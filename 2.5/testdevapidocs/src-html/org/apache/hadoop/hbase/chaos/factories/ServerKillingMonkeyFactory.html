018package org.apache.hadoop.hbase.chaos.factories;
019
020import org.apache.hadoop.hbase.chaos.actions.Action;
021import org.apache.hadoop.hbase.chaos.actions.DumpClusterStatusAction;
022import org.apache.hadoop.hbase.chaos.actions.ForceBalancerAction;
023import org.apache.hadoop.hbase.chaos.actions.GracefulRollingRestartRsAction;
024import org.apache.hadoop.hbase.chaos.actions.RestartActiveMasterAction;
025import org.apache.hadoop.hbase.chaos.actions.RestartRandomRsExceptMetaAction;
026import org.apache.hadoop.hbase.chaos.actions.RollingBatchRestartRsAction;
027import org.apache.hadoop.hbase.chaos.actions.RollingBatchSuspendResumeRsAction;
028import org.apache.hadoop.hbase.chaos.monkies.ChaosMonkey;
029import org.apache.hadoop.hbase.chaos.monkies.PolicyBasedChaosMonkey;
030import org.apache.hadoop.hbase.chaos.policies.CompositeSequentialPolicy;
031import org.apache.hadoop.hbase.chaos.policies.DoActionsOncePolicy;
032import org.apache.hadoop.hbase.chaos.policies.PeriodicRandomActionPolicy;
033
034/**
035 * Creates ChaosMonkeys for doing server restart actions, but not flush / compact / snapshot kind of
036 * actions.
037 */
038public class ServerKillingMonkeyFactory extends MonkeyFactory {
039
040  private long gracefulRollingRestartTSSLeepTime;
041  private long rollingBatchSuspendRSSleepTime;
042  private float rollingBatchSuspendtRSRatio;
043
044  @Override
045  public ChaosMonkey build() {
046    loadProperties();
047
048    // Destructive actions to mess things around. Cannot run batch restart
049    // @formatter:off
050    Action[] actions1 = new Action[] {
051      new RestartRandomRsExceptMetaAction(60000),
052      new RestartActiveMasterAction(5000),
053      // only allow 2 servers to be dead
054      new RollingBatchRestartRsAction(5000, 1.0f, 2, true),
055      new ForceBalancerAction(),
056      new GracefulRollingRestartRsAction(gracefulRollingRestartTSSLeepTime),
057      new RollingBatchSuspendResumeRsAction(rollingBatchSuspendRSSleepTime,
058          rollingBatchSuspendtRSRatio)
059    };
060    // @formatter:on
061
062    // Action to log more info for debugging
063    Action[] actions2 = new Action[] { new DumpClusterStatusAction() };
064
065    return new PolicyBasedChaosMonkey(properties, util,
066      new CompositeSequentialPolicy(new DoActionsOncePolicy(60 * 1000, actions1),
067        new PeriodicRandomActionPolicy(60 * 1000, actions1)),
068      new PeriodicRandomActionPolicy(60 * 1000, actions2));
069  }
070
071  private void loadProperties() {
072    gracefulRollingRestartTSSLeepTime =
073      Long.parseLong(this.properties.getProperty(MonkeyConstants.GRACEFUL_RESTART_RS_SLEEP_TIME,
074        MonkeyConstants.DEFAULT_GRACEFUL_RESTART_RS_SLEEP_TIME + ""));
075    rollingBatchSuspendRSSleepTime = Long
076      .parseLong(this.properties.getProperty(MonkeyConstants.ROLLING_BATCH_SUSPEND_RS_SLEEP_TIME,
077        MonkeyConstants.DEFAULT_ROLLING_BATCH_SUSPEND_RS_SLEEP_TIME + ""));
078    rollingBatchSuspendtRSRatio =
079      Float.parseFloat(this.properties.getProperty(MonkeyConstants.ROLLING_BATCH_SUSPEND_RS_RATIO,
080        MonkeyConstants.DEFAULT_ROLLING_BATCH_SUSPEND_RS_RATIO + ""));
081  }
082}
