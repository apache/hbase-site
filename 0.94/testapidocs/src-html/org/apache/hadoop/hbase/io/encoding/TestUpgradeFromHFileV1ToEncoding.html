017package org.apache.hadoop.hbase.io.encoding;
018
019import static org.apache.hadoop.hbase.io.encoding.TestChangingEncoding.CF;
020import static org.apache.hadoop.hbase.io.encoding.TestChangingEncoding.CF_BYTES;
021
022import java.util.concurrent.TimeUnit;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.HBaseTestingUtility;
028import org.apache.hadoop.hbase.HColumnDescriptor;
029import org.apache.hadoop.hbase.HConstants;
030import org.apache.hadoop.hbase.HTableDescriptor;
031import org.apache.hadoop.hbase.LargeTests;
032import org.apache.hadoop.hbase.client.HBaseAdmin;
033import org.apache.hadoop.hbase.io.hfile.HFile;
034import org.apache.hadoop.hbase.util.Bytes;
035import org.junit.AfterClass;
036import org.junit.BeforeClass;
037import org.junit.Test;
038import org.junit.experimental.categories.Category;
039
040@Category(LargeTests.class)
041public class TestUpgradeFromHFileV1ToEncoding {
042
043  private static final Log LOG =
044      LogFactory.getLog(TestUpgradeFromHFileV1ToEncoding.class);
045
046  private static final String TABLE = "UpgradeTable";
047  private static final byte[] TABLE_BYTES = Bytes.toBytes(TABLE);
048
049  private static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
050  private static final Configuration conf = TEST_UTIL.getConfiguration();
051
052  private static final int NUM_HFILE_V1_BATCHES = 10;
053  private static final int NUM_HFILE_V2_BATCHES = 20;
054
055  @BeforeClass
056  public static void setUpBeforeClass() throws Exception {
057    // Use a small flush size to create more HFiles.
058    conf.setInt(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 1024 * 1024);
059    conf.setInt(HFile.FORMAT_VERSION_KEY, 1); // Use HFile v1 initially
060    TEST_UTIL.startMiniCluster();
061    LOG.debug("Started an HFile v1 cluster");
062  }
063
064  @AfterClass
065  public static void tearDownAfterClass() throws Exception {
066    TEST_UTIL.shutdownMiniCluster();
067  }
068
069  @Test
070  public void testUpgrade() throws Exception {
071    int numBatches = 0;
072    HTableDescriptor htd = new HTableDescriptor(TABLE);
073    HColumnDescriptor hcd = new HColumnDescriptor(CF);
074    htd.addFamily(hcd);
075    HBaseAdmin admin = new HBaseAdmin(conf);
076    admin.createTable(htd);
077    admin.close();
078    for (int i = 0; i < NUM_HFILE_V1_BATCHES; ++i) {
079      TestChangingEncoding.writeTestDataBatch(conf, TABLE, numBatches++);
080    }
081    TEST_UTIL.shutdownMiniHBaseCluster();
082
083    conf.setInt(HFile.FORMAT_VERSION_KEY, 2);
084    TEST_UTIL.startMiniHBaseCluster(1, 1);
085    LOG.debug("Started an HFile v2 cluster");
086    admin = new HBaseAdmin(conf);
087    htd = admin.getTableDescriptor(TABLE_BYTES);
088    hcd = htd.getFamily(CF_BYTES);
089    hcd.setDataBlockEncoding(DataBlockEncoding.PREFIX);
090    admin.disableTable(TABLE);
091    admin.modifyColumn(TABLE, hcd);
092    admin.enableTable(TABLE);
093    admin.close();
094    for (int i = 0; i < NUM_HFILE_V2_BATCHES; ++i) {
095      TestChangingEncoding.writeTestDataBatch(conf, TABLE, numBatches++);
096    }
097
098    LOG.debug("Verifying all 'batches', both HFile v1 and encoded HFile v2");
099    verifyBatches(numBatches);
100
101    LOG.debug("Doing a manual compaction");
102    admin.compact(TABLE);
103    Thread.sleep(TimeUnit.SECONDS.toMillis(10));
104
105    LOG.debug("Verify all the data again");
106    verifyBatches(numBatches);
107  }
108
109  private void verifyBatches(int numBatches) throws Exception {
110    for (int i = 0; i < numBatches; ++i) {
111      TestChangingEncoding.verifyTestDataBatch(conf, TABLE, i);
112    }
113  }
114
115}
