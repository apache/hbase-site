047  private final static HBaseTestingUtility TEST_UTIL = new
048      HBaseTestingUtility();
049
050  @Test
051  public void testColumnCountGetFilter() throws IOException {
052    String family = "Family";
053    HTableDescriptor htd = new HTableDescriptor("testColumnCountGetFilter");
054    htd.addFamily(new HColumnDescriptor(family));
055    HRegionInfo info = new HRegionInfo(htd.getName(), null, null, false);
056    HRegion region = HRegion.createHRegion(info, TEST_UTIL.
057      getDataTestDir(), TEST_UTIL.getConfiguration(), htd);
058    try {
059      String valueString = "ValueString";
060      String row = "row-1";
061      List<String> columns = generateRandomWords(10000, "column");
062      Put p = new Put(Bytes.toBytes(row));
063      p.setWriteToWAL(false);
064      for (String column : columns) {
065        KeyValue kv = KeyValueTestUtil.create(row, family, column, 0, valueString);
066        p.add(kv);
067      }
068      region.put(p);
069
070      Get get = new Get(row.getBytes());
071      Filter filter = new ColumnCountGetFilter(100);
072      get.setFilter(filter);
073      Scan scan = new Scan(get);
074      InternalScanner scanner = region.getScanner(scan);
075      List<KeyValue> results = new ArrayList<KeyValue>();
076      scanner.next(results);
077      assertEquals(100, results.size());
078    } finally {
079      region.close();
080      region.getLog().closeAndDelete();
081    }
082
083    region.close();
084    region.getLog().closeAndDelete();
085  }

125  List<String> generateRandomWords(int numberOfWords, String suffix) {
126    Set<String> wordSet = new HashSet<String>();
127    for (int i = 0; i < numberOfWords; i++) {
128      int lengthOfWords = (int) (Math.random()*2) + 1;
129      char[] wordChar = new char[lengthOfWords];
130      for (int j = 0; j < wordChar.length; j++) {
131        wordChar[j] = (char) (Math.random() * 26 + 97);
132      }
133      String word;
134      if (suffix == null) {
135        word = new String(wordChar);
136      } else {
137        word = new String(wordChar) + suffix;
138      }
139      wordSet.add(word);
140    }
141    List<String> wordList = new ArrayList<String>(wordSet);
142    return wordList;
143  }
