018package org.apache.hadoop.hbase.errorhandling;
019
020import static org.junit.Assert.assertTrue;
021import static org.junit.Assert.fail;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.hbase.SmallTests;
026import org.junit.Test;
027import org.junit.experimental.categories.Category;
028import org.mockito.Mockito;
029
030/**
031 * Test that we propagate errors through an dispatcher exactly once via different failure
032 * injection mechanisms.
033 */
034@Category(SmallTests.class)
035public class TestForeignExceptionDispatcher {
036  private static final Log LOG = LogFactory.getLog(TestForeignExceptionDispatcher.class);
037
038  /**
039   * Exception thrown from the test
040   */
041  final ForeignException EXTEXN = new ForeignException("FORTEST", new IllegalArgumentException("FORTEST"));
042  final ForeignException EXTEXN2 = new ForeignException("FORTEST2", new IllegalArgumentException("FORTEST2"));
043
044  /**
045   * Tests that a dispatcher only dispatches only the first exception, and does not propagate
046   * subsequent exceptions.
047   */
048  @Test
049  public void testErrorPropagation() {
050    ForeignExceptionListener listener1 = Mockito.mock(ForeignExceptionListener.class);
051    ForeignExceptionListener listener2 = Mockito.mock(ForeignExceptionListener.class);
052    ForeignExceptionDispatcher dispatcher = new ForeignExceptionDispatcher();
053
054    // add the listeners
055    dispatcher.addListener(listener1);
056    dispatcher.addListener(listener2);
057
058    // create an artificial error
059    dispatcher.receive(EXTEXN);
060
061    // make sure the listeners got the error
062    Mockito.verify(listener1, Mockito.times(1)).receive(EXTEXN);
063    Mockito.verify(listener2, Mockito.times(1)).receive(EXTEXN);
064
065    // make sure that we get an exception
066    try {
067      dispatcher.rethrowException();
068      fail("Monitor should have thrown an exception after getting error.");
069    } catch (ForeignException ex) {
070      assertTrue("Got an unexpected exception:" + ex, ex.getCause() == EXTEXN.getCause());
071      LOG.debug("Got the testing exception!");
072    }
073
074    // push another error, which should be not be passed to listeners
075    dispatcher.receive(EXTEXN2);
076    Mockito.verify(listener1, Mockito.never()).receive(EXTEXN2);
077    Mockito.verify(listener2, Mockito.never()).receive(EXTEXN2);
078  }
079
080  @Test
081  public void testSingleDispatcherWithTimer() {
082    ForeignExceptionListener listener1 = Mockito.mock(ForeignExceptionListener.class);
083    ForeignExceptionListener listener2 = Mockito.mock(ForeignExceptionListener.class);
084
085    ForeignExceptionDispatcher monitor = new ForeignExceptionDispatcher();
086
087    // add the listeners
088    monitor.addListener(listener1);
089    monitor.addListener(listener2);
090
091    TimeoutExceptionInjector timer = new TimeoutExceptionInjector(monitor, 1000);
092    timer.start();
093    timer.trigger();
094
095    assertTrue("Monitor didn't get timeout", monitor.hasException());
096
097    // verify that that we propagated the error
098    Mockito.verify(listener1).receive(Mockito.any(ForeignException.class));
099    Mockito.verify(listener2).receive(Mockito.any(ForeignException.class));
100  }
101
102  /**
103   * Test that the dispatcher can receive an error via the timer mechanism.
104   */
105  @Test
106  public void testAttemptTimer() {
107    ForeignExceptionListener listener1 = Mockito.mock(ForeignExceptionListener.class);
108    ForeignExceptionListener listener2 = Mockito.mock(ForeignExceptionListener.class);
109    ForeignExceptionDispatcher orchestrator = new ForeignExceptionDispatcher();
110
111    // add the listeners
112    orchestrator.addListener(listener1);
113    orchestrator.addListener(listener2);
114
115    // now create a timer and check for that error
116    TimeoutExceptionInjector timer = new TimeoutExceptionInjector(orchestrator, 1000);
117    timer.start();
118    timer.trigger();
119    // make sure that we got the timer error
120    Mockito.verify(listener1, Mockito.times(1)).receive(Mockito.any(ForeignException.class));
121    Mockito.verify(listener2, Mockito.times(1)).receive(Mockito.any(ForeignException.class));
122  }
123}
