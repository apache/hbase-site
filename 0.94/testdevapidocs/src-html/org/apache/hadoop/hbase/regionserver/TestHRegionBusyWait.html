019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022import java.util.concurrent.atomic.AtomicBoolean;
023
024import org.apache.hadoop.hbase.MediumTests;
025import org.apache.hadoop.hbase.RegionTooBusyException;
026import org.apache.hadoop.hbase.client.Get;
027import org.apache.hadoop.hbase.util.Bytes;
028import org.junit.Test;
029import org.junit.experimental.categories.Category;
030
031/**
032 * TestHRegion with hbase.busy.wait.duration set to 1000 (1 second).
033 * We can't use parameterized test since TestHRegion is old fashion.
034 */
035@Category(MediumTests.class)
036@SuppressWarnings("deprecation")
037public class TestHRegionBusyWait extends TestHRegion {
038  public TestHRegionBusyWait() {
039    conf.set("hbase.busy.wait.duration", "1000");
040  }
041
042  /**
043   * Test RegionTooBusyException thrown when region is busy
044   */
045  @Test (timeout=2000)
046  public void testRegionTooBusy() throws IOException {
047    String method = "testRegionTooBusy";
048    byte[] tableName = Bytes.toBytes(method);
049    byte[] family = Bytes.toBytes("family");
050    region = initHRegion(tableName, method, conf, family);
051    final AtomicBoolean stopped = new AtomicBoolean(true);
052    Thread t = new Thread(new Runnable() {
053      @Override
054      public void run() {
055        try {
056          region.lock.writeLock().lock();
057          stopped.set(false);
058          while (!stopped.get()) {
059            Thread.sleep(100);
060          }
061        } catch (InterruptedException ie) {
062        } finally {
063          region.lock.writeLock().unlock();
064        }
065      }
066    });
067    t.start();
068    Get get = new Get(row);
069    try {
070      while (stopped.get()) {
071        Thread.sleep(100);
072      }
073      region.get(get, null);
074      fail("Should throw RegionTooBusyException");
075    } catch (InterruptedException ie) {
076      fail("test interrupted");
077    } catch (RegionTooBusyException e) {
078      // Good, expected
079    } finally {
080      stopped.set(true);
081      try {
082        t.join();
083      } catch (Throwable e) {
084      }
085
086      HRegion.closeHRegion(region);
087      region = null;
088    }
089  }
090}
