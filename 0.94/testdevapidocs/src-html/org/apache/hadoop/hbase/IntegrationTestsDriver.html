019package org.apache.hadoop.hbase;
020
021import java.io.IOException;
022import java.util.List;
023import java.util.Set;
024import java.util.regex.Pattern;
025
026import org.apache.commons.cli.CommandLine;
027import org.apache.hadoop.hbase.util.AbstractHBaseTool;
028import org.apache.hadoop.util.ToolRunner;
029import org.junit.internal.TextListener;
030import org.junit.runner.JUnitCore;
031import org.junit.runner.Result;
032
033import org.apache.commons.logging.Log;
034import org.apache.commons.logging.LogFactory;
035
036/**
037 * This class drives the Integration test suite execution. Executes all
038 * tests having @Category(IntegrationTests.class) annotation against an
039 * already deployed distributed cluster.
040 */
041public class IntegrationTestsDriver extends AbstractHBaseTool {
042  private static final String TESTS_ARG = "test";
043  private static final Log LOG = LogFactory.getLog(IntegrationTestsDriver.class);
044  private IntegrationTestFilter intTestFilter = new IntegrationTestFilter();
045
046  public static void main(String[] args) throws Exception {
047    int ret = ToolRunner.run(new IntegrationTestsDriver(), args);
048    System.exit(ret);
049  }
050
051  private class IntegrationTestFilter extends ClassTestFinder.TestClassFilter {
052    private Pattern testFilterRe = Pattern.compile(".*");
053    public IntegrationTestFilter() {
054      super(IntegrationTests.class);
055    }
056
057    public void setPattern(String pattern) {
058      testFilterRe = Pattern.compile(pattern);
059    }
060
061    @Override
062    public boolean isCandidateClass(Class<?> c) {
063      return super.isCandidateClass(c) && testFilterRe.matcher(c.getName()).find();
064    }
065  }
066
067  @Override
068  protected void addOptions() {
069    addOptWithArg(TESTS_ARG, "a Java regular expression to filter tests on");
070  }
071
072  @Override
073  protected void processOptions(CommandLine cmd) {
074    String testFilterString = cmd.getOptionValue(TESTS_ARG, null);
075    if (testFilterString != null) {
076      intTestFilter.setPattern(testFilterString);
077    }
078  }
079
080  /**
081   * Returns test classes annotated with @Category(IntegrationTests.class),
082   * according to the filter specific on the command line (if any).
083   */
084  private Class<?>[] findIntegrationTestClasses()
085    throws ClassNotFoundException, LinkageError, IOException {
086    ClassTestFinder.TestFileNameFilter nameFilter = new ClassTestFinder.TestFileNameFilter();
087    ClassFinder classFinder = new ClassFinder(nameFilter, intTestFilter);
088    Set<Class<?>> classes = classFinder.findClasses(true);
089    return classes.toArray(new Class<?>[classes.size()]);
090  }
091
092
093  @Override
094  protected int doWork() throws Exception {
095    //this is called from the command line, so we should set to use the distributed cluster
096    IntegrationTestingUtility.setUseDistributedCluster(conf);
097    Class<?>[] classes = findIntegrationTestClasses();
098    LOG.info("Found " + classes.length + " integration tests to run");
099
100    JUnitCore junit = new JUnitCore();
101    junit.addListener(new TextListener(System.out));
102    Result result = junit.run(classes);
103
104    return result.wasSuccessful() ? 0 : 1;
105  }
106
107}
