044/**
045 * <p>
046 * Tests scanning a snapshot. Tests various scan start and stop row scenarios.
047 * This is set in a scan and tested in a MapReduce job to see if that is handed
048 * over and done properly too.
049 * </p>
050 */
051@Category(LargeTests.class)
052public class TestTableSnapshotInputFormatScan {
053
054  static final Log LOG = LogFactory.getLog(TestTableSnapshotInputFormatScan.class);
055  static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
056
057  static final byte[] TABLE_NAME = Bytes.toBytes("scantest");
058  static final byte[] SNAPSHOT_NAME = Bytes.toBytes("scantest_snaphot");
059  static final byte[] INPUT_FAMILY = Bytes.toBytes("contents");
060  static final String KEY_STARTROW = "startRow";
061  static final String KEY_LASTROW = "stpRow";
062
063  private static HTable table = null;
064
065  @BeforeClass
066  public static void setUpBeforeClass() throws Exception {
067    // config snapshot support
068    TEST_UTIL.getConfiguration().setBoolean(
069        SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
070    TEST_UTIL.getConfiguration().setBoolean(
071        "hbase.master.enabletable.roundrobin", true);
072
073    // switch TIF to log at DEBUG level
074    TEST_UTIL.enableDebug(TableSnapshotInputFormat.class);
075
076    // start mini hbase cluster
077    TEST_UTIL.startMiniCluster(3);
078
079    // create and fill table
080    table = TEST_UTIL.createTable(TABLE_NAME, INPUT_FAMILY);
081    TEST_UTIL.createMultiRegions(table, INPUT_FAMILY);
082    TEST_UTIL.loadTable(table, INPUT_FAMILY);
083    TEST_UTIL.getHBaseAdmin().disableTable(TABLE_NAME);
084    TEST_UTIL.getHBaseAdmin().snapshot(SNAPSHOT_NAME, TABLE_NAME);
085    TEST_UTIL.getHBaseAdmin().enableTable(TABLE_NAME);
086
087    // start MR cluster
088    TEST_UTIL.startMiniMapReduceCluster();
089  }
090
091  @AfterClass
092  public static void tearDownAfterClass() throws Exception {
093    TEST_UTIL.shutdownMiniMapReduceCluster();
094    TEST_UTIL.shutdownMiniCluster();
095  }

162  /**
163   * Tests a MR scan using specific start and stop rows.
164   * 
165   * @throws IOException
166   * @throws ClassNotFoundException
167   * @throws InterruptedException
168   */
169  protected void testScan(String start, String stop, String last)
170      throws IOException, InterruptedException, ClassNotFoundException {
171    String jobName = "Scan" + (start != null ? start.toUpperCase() : "Empty")
172        + "To" + (stop != null ? stop.toUpperCase() : "Empty");
173    LOG.info("Before map/reduce startup - job " + jobName);
174    Configuration c = new Configuration(TEST_UTIL.getConfiguration());
175    Scan scan = new Scan();
176    scan.addFamily(INPUT_FAMILY);
177    if (start != null) {
178      scan.setStartRow(Bytes.toBytes(start));
179    }
180    c.set(KEY_STARTROW, start != null ? start : "");
181    if (stop != null) {
182      scan.setStopRow(Bytes.toBytes(stop));
183    }
184    c.set(KEY_LASTROW, last != null ? last : "");
185    LOG.info("scan before: " + scan);
186    Job job = new Job(c, jobName);
187
188    FileSystem fs = FileSystem.get(c);
189    Path tmpDir = new Path("/" + UUID.randomUUID());
190    fs.mkdirs(tmpDir);
191    try {
192      TableMapReduceUtil.initTableSnapshotMapperJob(Bytes.toString(SNAPSHOT_NAME),
193          scan, TestTableInputFormatScanBase.ScanMapper.class,
194          ImmutableBytesWritable.class, ImmutableBytesWritable.class, job,
195          false, tmpDir);
196      job.setReducerClass(TestTableInputFormatScanBase.ScanReducer.class);
197      job.setNumReduceTasks(1); // one to get final "first" and "last" key
198      FileOutputFormat.setOutputPath(job, new Path(job.getJobName()));
199      LOG.info("Started " + job.getJobName());
200      assertTrue(job.waitForCompletion(true));
201      LOG.info("After map/reduce completion - job " + jobName);
202    } finally {
203      fs.delete(tmpDir, true);
204    }
205
206  }
