019package org.apache.hadoop.hbase.rest;
020
021import org.apache.hadoop.hbase.SmallTests;
022import org.apache.hadoop.hbase.rest.metrics.RESTMetrics;
023import org.junit.Test;
024import org.junit.experimental.categories.Category;
025import static org.junit.Assert.*;
026
027/**
028 * Test RESTMetrics class
029 */
030@Category(SmallTests.class)
031public class TestRESTMetrics {
032
033  @Test
034  public void testRESTMetrics() throws InterruptedException {
035    long timeout = 2000;
036    RESTMetrics test = new RESTMetrics();
037    int incrementSucessfulGet = 20000;
038    int incrementSucessfulDelete = 3000000;
039    int incrementSucessfulPut = 3000000;
040    int incrementRequest = incrementSucessfulGet + incrementSucessfulDelete + incrementSucessfulPut;
041
042    int incrementFailedGetRequests = 100;
043    int incrementFailedDeleteRequests = 30;
044    int incrementFailedPutRequests = 2;
045
046    long start1 = System.currentTimeMillis();
047    test.doUpdates(null);
048
049    // started value
050    assertEquals(0, test.getRequests(), 0.01);
051    assertEquals(0, test.getSucessfulDeleteCount(), 0.01);
052    assertEquals(0, test.getSucessfulPutCount(), 0.01);
053    assertEquals(0, test.getSucessfulGetCount(), 0.01);
054    assertEquals(0, test.getFailedDeleteCount(), 0.01);
055    assertEquals(0, test.getFailedGetCount(), 0.01);
056    assertEquals(0, test.getFailedPutCount(), 0.01);
057
058    // sleep some seconds
059    Thread.sleep(timeout);
060    test.incrementRequests(incrementRequest);
061    test.incrementSucessfulGetRequests(incrementSucessfulGet);
062    test.incrementSucessfulDeleteRequests(incrementSucessfulDelete);
063    test.incrementSucessfulPutRequests(incrementSucessfulPut);
064    test.incrementFailedGetRequests(incrementFailedGetRequests);
065    test.incrementFailedDeleteRequests(incrementFailedDeleteRequests);
066    test.incrementFailedPutRequests(incrementFailedPutRequests);
067
068    test.doUpdates(null);
069
070    // The maximum time for stability test
071    long tmax = System.currentTimeMillis() - start1;
072
073    testData(tmax, timeout, test.getRequests(), incrementRequest);
074    testData(tmax, timeout, test.getSucessfulGetCount(), incrementSucessfulGet);
075    testData(tmax, timeout, test.getSucessfulDeleteCount(), incrementSucessfulDelete);
076    testData(tmax, timeout, test.getSucessfulPutCount(), incrementSucessfulPut);
077    testData(tmax, timeout, test.getFailedGetCount(), incrementFailedGetRequests);
078    testData(tmax, timeout, test.getFailedDeleteCount(), incrementFailedDeleteRequests);
079    testData(tmax, timeout, test.getFailedPutCount(), incrementFailedPutRequests);
080
081    test.shutdown();
082  }
083
084  // test minimum and maximum speed
085  private void testData(double tmax, long tmin, float value, double requests) {
086    assertTrue((requests / tmax) * 1000 <= value);
087    assertTrue((requests / tmin) * 1000 >= value);
088
089  }
090}
