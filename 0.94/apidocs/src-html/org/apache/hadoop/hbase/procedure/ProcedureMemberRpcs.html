018package org.apache.hadoop.hbase.procedure;
019
020import java.io.Closeable;
021import java.io.IOException;
022
023import org.apache.hadoop.classification.InterfaceAudience;
024import org.apache.hadoop.classification.InterfaceStability;
025import org.apache.hadoop.hbase.errorhandling.ForeignException;
026
027/**
028 * This is the notification interface for Procedures that encapsulates message passing from
029 * members to a coordinator.  Each of these calls should send a message to the coordinator.
030 */
031@InterfaceAudience.Public
032@InterfaceStability.Evolving
033public interface ProcedureMemberRpcs extends Closeable {
034
035  /**
036   * Initialize and start any threads or connections the member needs.
037   */
038  public void start(final String memberName, final ProcedureMember member);
039
040  /**
041   * Each subprocedure is being executed on a member.  This is the identifier for the member.
042   * @return the member name
043   */
044  public String getMemberName();
045
046  /**
047   * Notify the coordinator that we aborted the specified {@link Subprocedure}
048   *
049   * @param sub the {@link Subprocedure} we are aborting
050   * @param cause the reason why the member's subprocedure aborted
051   * @throws IOException thrown when the rpcs can't reach the other members of the procedure (and
052   *  thus can't recover).
053   */
054  public void sendMemberAborted(Subprocedure sub, ForeignException cause) throws IOException;
055
056  /**
057   * Notify the coordinator that the specified {@link Subprocedure} has acquired the locally required
058   * barrier condition.
059   *
060   * @param sub the specified {@link Subprocedure}
061   * @throws IOException if we can't reach the coordinator
062   */
063  public void sendMemberAcquired(Subprocedure sub) throws IOException;
064
065  /**
066   * Notify the coordinator that the specified {@link Subprocedure} has completed the work that
067   * needed to be done under the global barrier.
068   *
069   * @param sub the specified {@link Subprocedure}
070   * @throws IOException if we can't reach the coordinator
071   */
072  public void sendMemberCompleted(Subprocedure sub) throws IOException;
073}
