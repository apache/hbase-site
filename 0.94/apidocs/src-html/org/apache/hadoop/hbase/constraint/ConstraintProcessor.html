018package org.apache.hadoop.hbase.constraint;
019
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.List;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.hbase.CoprocessorEnvironment;
027import org.apache.hadoop.hbase.HTableDescriptor;
028import org.apache.hadoop.hbase.client.Put;
029import org.apache.hadoop.hbase.coprocessor.BaseRegionObserver;
030import org.apache.hadoop.hbase.coprocessor.ObserverContext;
031import org.apache.hadoop.hbase.coprocessor.RegionCoprocessorEnvironment;
032import org.apache.hadoop.hbase.regionserver.wal.WALEdit;
033
034/***
035 * Processes multiple {@link Constraint Constraints} on a given table.
036 * <p>
037 * This is an ease of use mechanism - all the functionality here could be
038 * implemented on any given system by a coprocessor.
039 */
040public class ConstraintProcessor extends BaseRegionObserver {
041
042  private static final Log LOG = LogFactory.getLog(ConstraintProcessor.class);
043
044  private final ClassLoader classloader;
045
046  private List<? extends Constraint> constraints = new ArrayList<Constraint>();
047
048  /**
049   * Create the constraint processor.
050   * <p>
051   * Stores the current classloader.
052   */
053  public ConstraintProcessor() {
054    classloader = this.getClass().getClassLoader();
055  }
056
057  @Override
058  public void start(CoprocessorEnvironment environment) {
059    // make sure we are on a region server
060    if (!(environment instanceof RegionCoprocessorEnvironment)) {
061      throw new IllegalArgumentException(
062          "Constraints only act on regions - started in an environment that was not a region");
063    }
064    RegionCoprocessorEnvironment env = (RegionCoprocessorEnvironment) environment;
065    HTableDescriptor desc = env.getRegion().getTableDesc();
066    // load all the constraints from the HTD
067    try {
068      this.constraints = Constraints.getConstraints(desc, classloader);
069    } catch (IOException e) {
070      throw new IllegalArgumentException(e);
071    }
072
073    if (LOG.isInfoEnabled()) {
074      LOG.info("Finished loading " + constraints.size()
075          + " user Constraints on table: " + new String(desc.getName()));
076    }
077
078  }
079
080  @Override
081  public void prePut(ObserverContext<RegionCoprocessorEnvironment> e, Put put,
082      WALEdit edit, boolean writeToWAL) throws IOException {
083    // check the put against the stored constraints
084    for (Constraint c : constraints) {
085      c.check(put);
086    }
087    // if we made it here, then the Put is valid
088  }
089}
