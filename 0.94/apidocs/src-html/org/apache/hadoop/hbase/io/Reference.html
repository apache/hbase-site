020package org.apache.hadoop.hbase.io;
021
022import java.io.DataInput;
023import java.io.DataOutput;
024import java.io.IOException;
025
026import org.apache.hadoop.fs.FSDataInputStream;
027import org.apache.hadoop.fs.FSDataOutputStream;
028import org.apache.hadoop.fs.FileSystem;
029import org.apache.hadoop.fs.Path;
030import org.apache.hadoop.hbase.HBaseFileSystem;
031import org.apache.hadoop.hbase.KeyValue;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.hbase.util.FSUtils;
034import org.apache.hadoop.io.Writable;
035
036/**
037 * A reference to the top or bottom half of a store file.  The file referenced
038 * lives under a different region.  References are made at region split time.
039 *
040 * <p>References work with a special half store file type.  References know how
041 * to write out the reference format in the file system and are whats juggled
042 * when references are mixed in with direct store files.  The half store file
043 * type is used reading the referred to file.
044 *
045 * <p>References to store files located over in some other region look like
046 * this in the file system
047 * <code>1278437856009925445.3323223323</code>:
048 * i.e. an id followed by hash of the referenced region.
049 * Note, a region is itself not splitable if it has instances of store file
050 * references.  References are cleaned up by compactions.
051 */
052public class Reference implements Writable {
053  private byte [] splitkey;
054  private Range region;
055
056  /**
057   * For split HStoreFiles, it specifies if the file covers the lower half or
058   * the upper half of the key range
059   */
060  public static enum Range {
061    /** HStoreFile contains upper half of key range */
062    top,
063    /** HStoreFile contains lower half of key range */
064    bottom
065  }
066
067  /**
068   * Constructor
069   * @param splitRow This is row we are splitting around.
070   * @param fr
071   */
072  public Reference(final byte [] splitRow, final Range fr) {
073    this.splitkey = splitRow == null?
074      null: KeyValue.createFirstOnRow(splitRow).getKey();
075    this.region = fr;
076  }
077
078  /**
079   * Used by serializations.
080   */
081  public Reference() {
082    this(null, Range.bottom);
083  }
084
085  /**
086   *
087   * @return Range
088   */
089  public Range getFileRegion() {
090    return this.region;
091  }
092
093  /**
094   * @return splitKey
095   */
096  public byte [] getSplitKey() {
097    return splitkey;
098  }
099
100  /**
101   * @see java.lang.Object#toString()
102   */
103  @Override
104  public String toString() {
105    return "" + this.region;
106  }
107
108  // Make it serializable.
109
110  public void write(DataOutput out) throws IOException {
111    // Write true if we're doing top of the file.
112    out.writeBoolean(isTopFileRegion(this.region));
113    Bytes.writeByteArray(out, this.splitkey);
114  }
115
116  public void readFields(DataInput in) throws IOException {
117    boolean tmp = in.readBoolean();
118    // If true, set region to top.
119    this.region = tmp? Range.top: Range.bottom;
120    this.splitkey = Bytes.readByteArray(in);
121  }
122
123  public static boolean isTopFileRegion(final Range r) {
124    return r.equals(Range.top);
125  }
126
127  public Path write(final FileSystem fs, final Path p)
128  throws IOException {
129    FSDataOutputStream out = HBaseFileSystem.createPathOnFileSystem(fs, p, false);
130    try {
131      write(out);
132    } finally {
133      out.close();
134    }
135    return p;
136  }
137
138  /**
139   * Read a Reference from FileSystem.
140   * @param fs
141   * @param p
142   * @return New Reference made from passed <code>p</code>
143   * @throws IOException
144   */
145  public static Reference read(final FileSystem fs, final Path p)
146  throws IOException {
147    FSDataInputStream in = fs.open(p);
148    try {
149      Reference r = new Reference();
150      r.readFields(in);
151      return r;
152    } finally {
153      in.close();
154    }
155  }
156}
