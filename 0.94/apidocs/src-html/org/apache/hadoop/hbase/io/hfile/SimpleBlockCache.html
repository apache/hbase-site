020package org.apache.hadoop.hbase.io.hfile;
021
022import java.lang.ref.ReferenceQueue;
023import java.lang.ref.SoftReference;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027
028import org.apache.hadoop.conf.Configuration;
029
030
031/**
032 * Simple one RFile soft reference cache.
033 */
034public class SimpleBlockCache implements BlockCache {
035  private static class Ref extends SoftReference<Cacheable> {
036    public BlockCacheKey blockId;
037    public Ref(BlockCacheKey blockId, Cacheable block, ReferenceQueue q) {
038      super(block, q);
039      this.blockId = blockId;
040    }
041  }
042  private Map<BlockCacheKey,Ref> cache =
043    new HashMap<BlockCacheKey,Ref>();
044
045  private ReferenceQueue q = new ReferenceQueue();
046  public int dumps = 0;
047
048  /**
049   * Constructor
050   */
051  public SimpleBlockCache() {
052    super();
053  }
054
055  void processQueue() {
056    Ref r;
057    while ( (r = (Ref)q.poll()) != null) {
058      cache.remove(r.blockId);
059      dumps++;
060    }
061  }
062
063  /**
064   * @return the size
065   */
066  public synchronized long size() {
067    processQueue();
068    return cache.size();
069  }
070
071  public synchronized Cacheable getBlock(BlockCacheKey cacheKey, boolean caching, boolean repeat) {
072    processQueue(); // clear out some crap.
073    Ref ref = cache.get(cacheKey);
074    if (ref == null)
075      return null;
076    return ref.get();
077  }
078
079  public synchronized void cacheBlock(BlockCacheKey cacheKey, Cacheable block) {
080    cache.put(cacheKey, new Ref(cacheKey, block, q));
081  }
082
083  public synchronized void cacheBlock(BlockCacheKey cacheKey, Cacheable block,
084      boolean inMemory) {
085    cache.put(cacheKey, new Ref(cacheKey, block, q));
086  }
087
088  @Override
089  public boolean evictBlock(BlockCacheKey cacheKey) {
090    return cache.remove(cacheKey) != null;
091  }
092
093  public void shutdown() {
094    // noop
095  }
096
097  @Override
098  public CacheStats getStats() {
099    // TODO: implement this if we ever actually use this block cache
100    return null;
101  }
102
103  @Override
104  public long getFreeSize() {
105    // TODO: implement this if we ever actually use this block cache
106    return 0;
107  }
108
109  @Override
110  public long getCurrentSize() {
111    // TODO: implement this if we ever actually use this block cache
112    return 0;
113  }
114
115  @Override
116  public long getEvictedCount() {
117    // TODO: implement this if we ever actually use this block cache
118    return 0;
119  }
120
121  @Override
122  public int evictBlocksByHfileName(String string) {
123    throw new UnsupportedOperationException();
124  }
125
126  @Override
127  public List<BlockCacheColumnFamilySummary> getBlockCacheColumnFamilySummaries(Configuration conf) {
128    throw new UnsupportedOperationException();
129  }
130
131  @Override
132  public long getBlockCount() {
133    // TODO: implement this if we ever actually use this block cache
134    return 0;
135  }
136
137}
