020package org.apache.hadoop.hbase.regionserver;
021
022import java.io.IOException;
023
024import org.apache.hadoop.fs.FileSystem;
025import org.apache.hadoop.hbase.HRegionInfo;
026import org.apache.hadoop.hbase.catalog.CatalogTracker;
027import org.apache.hadoop.hbase.ipc.RpcServer;
028import org.apache.hadoop.hbase.regionserver.wal.HLog;
029import org.apache.zookeeper.KeeperException;
030
031/**
032 * Services provided by {@link HRegionServer}
033 */
034public interface RegionServerServices extends OnlineRegions {
035  /**
036   * @return True if this regionserver is stopping.
037   */
038  public boolean isStopping();
039
040  /** @return the HLog for a particular region. Pass null for getting the
041   * default (common) WAL */
042  public HLog getWAL(HRegionInfo regionInfo) throws IOException;
043
044  /** @return get the default (common) WAL for the server*/
045  public HLog getWAL() throws IOException;
046
047  /**
048   * @return Implementation of {@link CompactionRequestor} or null.
049   */
050  public CompactionRequestor getCompactionRequester();
051
052  /**
053   * @return Implementation of {@link FlushRequester} or null.
054   */
055  public FlushRequester getFlushRequester();
056
057  /**
058   * @return the RegionServerAccounting for this Region Server
059   */
060  public RegionServerAccounting getRegionServerAccounting();
061
062  /**
063   * Tasks to perform after region open to complete deploy of region on
064   * regionserver
065   * 
066   * @param r Region to open.
067   * @param ct Instance of {@link CatalogTracker}
068   * @param daughter True if this is daughter of a split
069   * @throws KeeperException
070   * @throws IOException
071   */
072  public void postOpenDeployTasks(final HRegion r, final CatalogTracker ct,
073      final boolean daughter)
074  throws KeeperException, IOException;
075
076  /**
077   * Returns a reference to the region server's RPC server
078   */
079  public RpcServer getRpcServer();
080
081  /**
082   * Remove passed <code>hri</code> from the internal list of regions in transition on this
083   * regionserver.
084   * @param hri Region to remove.
085   * @return True if removed
086   */
087  public boolean removeFromRegionsInTransition(HRegionInfo hri);
088  /**
089   * @param hri
090   * @return True if the internal list of regions in transition includes the
091   *         passed <code>hri</code>.
092   */
093  public boolean containsKeyInRegionsInTransition(HRegionInfo hri);
094
095  /**
096   * @return Return the FileSystem object used by the regionserver
097   */
098  public FileSystem getFileSystem();
099
100  /**
101   * @return The RegionServer's "Leases" service
102   */
103  public Leases getLeases();
104}
