022package org.apache.hadoop.hbase.regionserver.wal;
023
024import java.io.IOException;
025
026import org.apache.hadoop.hbase.Coprocessor;
027import org.apache.hadoop.hbase.HRegionInfo;
028import org.apache.hadoop.hbase.coprocessor.*;
029import org.apache.hadoop.conf.Configuration;
030
031/**
032 * Implements the coprocessor environment and runtime support for coprocessors
033 * loaded within a {@link HLog}.
034 */
035public class WALCoprocessorHost
036    extends CoprocessorHost<WALCoprocessorHost.WALEnvironment> {
037  
038  /**
039   * Encapsulation of the environment of each coprocessor
040   */
041  static class WALEnvironment extends CoprocessorHost.Environment
042    implements WALCoprocessorEnvironment {
043
044    private HLog wal;
045
046    @Override
047    public HLog getWAL() {
048      return wal;
049    }
050
051    /**
052     * Constructor
053     * @param impl the coprocessor instance
054     * @param priority chaining priority
055     * @param seq load sequence
056     * @param conf configuration
057     * @param hlog HLog
058     */
059    public WALEnvironment(Class<?> implClass, final Coprocessor impl,
060        final int priority, final int seq, final Configuration conf,
061        final HLog hlog) {
062      super(impl, priority, seq, conf);
063      this.wal = hlog;
064    }
065  }
066
067  HLog wal;
068  /**
069   * Constructor
070   * @param log the write ahead log
071   * @param conf the configuration
072   */
073  public WALCoprocessorHost(final HLog log, final Configuration conf) {
074    this.wal = log;
075    // load system default cp's from configuration.
076    loadSystemCoprocessors(conf, WAL_COPROCESSOR_CONF_KEY);
077  }
078
079  @Override
080  public WALEnvironment createEnvironment(final Class<?> implClass,
081      final Coprocessor instance, final int priority, final int seq,
082      final Configuration conf) {
083    return new WALEnvironment(implClass, instance, priority, seq, conf,
084        this.wal);
085  }
086
087  /**
088   * @param info
089   * @param logKey
090   * @param logEdit
091   * @return true if default behavior should be bypassed, false otherwise
092   * @throws IOException
093   */
094  public boolean preWALWrite(HRegionInfo info, HLogKey logKey, WALEdit logEdit)
095      throws IOException {
096    boolean bypass = false;
097    ObserverContext<WALCoprocessorEnvironment> ctx = null;
098    for (WALEnvironment env: coprocessors) {
099      if (env.getInstance() instanceof
100          org.apache.hadoop.hbase.coprocessor.WALObserver) {
101        ctx = ObserverContext.createAndPrepare(env, ctx);
102        ((org.apache.hadoop.hbase.coprocessor.WALObserver)env.getInstance()).
103            preWALWrite(ctx, info, logKey, logEdit);
104        bypass |= ctx.shouldBypass();
105        if (ctx.shouldComplete()) {
106          break;
107        }
108      }
109    }
110    return bypass;
111  }
112
113  /**
114   * @param info
115   * @param logKey
116   * @param logEdit
117   * @throws IOException
118   */
119  public void postWALWrite(HRegionInfo info, HLogKey logKey, WALEdit logEdit)
120      throws IOException {
121    ObserverContext<WALCoprocessorEnvironment> ctx = null;
122    for (WALEnvironment env: coprocessors) {
123      if (env.getInstance() instanceof
124          org.apache.hadoop.hbase.coprocessor.WALObserver) {
125        ctx = ObserverContext.createAndPrepare(env, ctx);
126        ((org.apache.hadoop.hbase.coprocessor.WALObserver)env.getInstance()).
127            postWALWrite(ctx, info, logKey, logEdit);
128        if (ctx.shouldComplete()) {
129          break;
130        }
131      }
132    }
133  }
134}
