020package org.apache.hadoop.hbase.client;
021
022import java.io.IOException;
023import java.util.Map;
024
025import org.codehaus.jackson.map.ObjectMapper;
026
027/**
028 * Superclass for any type that maps to a potentially application-level query.
029 * (e.g. Put, Get, Delete, Scan, Next, etc.)
030 * Contains methods for exposure to logging and debugging tools.
031 */
032public abstract class Operation {
033  // TODO make this configurable
034  private static final int DEFAULT_MAX_COLS = 5;
035
036  /**
037   * Produces a Map containing a fingerprint which identifies the type and 
038   * the static schema components of a query (i.e. column families)
039   * @return a map containing fingerprint information (i.e. column families)
040   */
041  public abstract Map<String, Object> getFingerprint();
042
043  /**
044   * Produces a Map containing a summary of the details of a query 
045   * beyond the scope of the fingerprint (i.e. columns, rows...)
046   * @param maxCols a limit on the number of columns output prior to truncation
047   * @return a map containing parameters of a query (i.e. rows, columns...)
048   */
049  public abstract Map<String, Object> toMap(int maxCols);
050
051  /**
052   * Produces a Map containing a full summary of a query.
053   * @return a map containing parameters of a query (i.e. rows, columns...)
054   */
055  public Map<String, Object> toMap() {
056    return toMap(DEFAULT_MAX_COLS);
057  }
058
059  /**
060   * Produces a JSON object for fingerprint and details exposure in a
061   * parseable format.
062   * @param maxCols a limit on the number of columns to include in the JSON
063   * @return a JSONObject containing this Operation's information, as a string
064   */
065  public String toJSON(int maxCols) throws IOException {
066    ObjectMapper mapper = new ObjectMapper();
067    return mapper.writeValueAsString(toMap(maxCols));
068  }
069
070  /**
071   * Produces a JSON object sufficient for description of a query
072   * in a debugging or logging context.
073   * @return the produced JSON object, as a string
074   */
075  public String toJSON() throws IOException {
076    return toJSON(DEFAULT_MAX_COLS);
077  }
078
079  /**
080   * Produces a string representation of this Operation. It defaults to a JSON
081   * representation, but falls back to a string representation of the 
082   * fingerprint and details in the case of a JSON encoding failure.
083   * @param maxCols a limit on the number of columns output in the summary
084   * prior to truncation
085   * @return a JSON-parseable String
086   */
087  public String toString(int maxCols) {
088    /* for now this is merely a wrapper from producing a JSON string, but 
089     * toJSON is kept separate in case this is changed to be a less parsable
090     * pretty printed representation.
091     */
092    try {
093      return toJSON(maxCols);
094    } catch (IOException ioe) {
095      return toMap(maxCols).toString();
096    }
097  }
098
099  /**
100   * Produces a string representation of this Operation. It defaults to a JSON
101   * representation, but falls back to a string representation of the
102   * fingerprint and details in the case of a JSON encoding failure.
103   * @return String
104   */
105  @Override
106  public String toString() {
107    return toString(DEFAULT_MAX_COLS);
108  }
109}
