018package org.apache.hadoop.hbase.procedure;
019
020import java.io.IOException;
021import java.util.concurrent.Callable;
022import java.util.concurrent.CountDownLatch;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.hbase.errorhandling.ForeignException;
027import org.apache.hadoop.hbase.errorhandling.ForeignExceptionSnare;
028import org.apache.hadoop.hbase.errorhandling.ForeignExceptionDispatcher;
029import org.apache.hadoop.hbase.errorhandling.ForeignExceptionListener;
030import org.apache.hadoop.hbase.errorhandling.TimeoutExceptionInjector;
031
032/**
033 * Distributed procedure member's Subprocedure.  A procedure is sarted on a ProcedureCoordinator
034 * which communicates with ProcedureMembers who create and start its part of the Procedure.  This
035 * sub part is called a Subprocedure
036 *
037 * Users should subclass this and implement {@link #acquireBarrier()} (get local barrier for this
038 * member), {@link #insideBarrier()} (execute while globally barriered and release barrier) and
039 * {@link #cleanup(Exception)} (release state associated with subprocedure.)
040 *
041 * When submitted to a ProcedureMemeber, the call method is executed in a separate thread.
042 * Latches are use too block its progress and trigger continuations when barrier conditions are
043 * met.
044 *
045 * Exception that makes it out of calls to {@link #acquireBarrier()} or {@link #insideBarrier()}
046 * gets converted into {@link ForeignException}, which will get propagated to the
047 * {@link ProcedureCoordinator}.
048 *
049 * There is a category of procedure (ex: online-snapshots), and a user-specified instance-specific
050 * barrierName. (ex: snapshot121126).
051 */
052abstract public class Subprocedure implements Callable<Void> {
053  private static final Log LOG = LogFactory.getLog(Subprocedure.class);
054
055  // Name of the procedure
056  final private String barrierName;
057
058  //
059  // Execution state
060  //
061
062  /** wait on before allowing the in barrier phase to proceed */
063  private final CountDownLatch inGlobalBarrier;
064  /** counted down when the Subprocedure has completed */
065  private final CountDownLatch releasedLocalBarrier;
066
067  //
068  // Error handling
069  //
070  /** monitor to check for errors */
071  protected final ForeignExceptionDispatcher monitor;
072  /** frequency to check for errors (ms) */
073  protected final long wakeFrequency;
074  protected final TimeoutExceptionInjector executionTimeoutTimer;
075  protected final ProcedureMemberRpcs rpcs;
076
077  private volatile boolean complete = false;
078
079  /**
080   * @param member reference to the member managing this subprocedure
081   * @param procName name of the procedure this subprocedure is associated with
082   * @param monitor notified if there is an error in the subprocedure
083   * @param wakeFrequency time in millis to wake to check if there is an error via the monitor (in
084   *          milliseconds).
085   * @param timeout time in millis that will trigger a subprocedure abort if it has not completed
086   */
087  public Subprocedure(ProcedureMember member, String procName, ForeignExceptionDispatcher monitor,
088      long wakeFrequency, long timeout) {
089    // Asserts should be caught during unit testing
090    assert member != null : "procedure member should be non-null";
091    assert member.getRpcs() != null : "rpc handlers should be non-null";
092    assert procName != null : "procedure name should be non-null";
093    assert monitor != null : "monitor should be non-null";
094
095    // Default to a very large timeout
096    this.rpcs = member.getRpcs();
097    this.barrierName = procName;
098    this.monitor = monitor;
099    // forward any failures to coordinator.  Since this is a dispatcher, resend loops should not be
100    // possible.
101    this.monitor.addListener(new ForeignExceptionListener() {
102      @Override
103      public void receive(ForeignException ee) {
104        // if this is a notification from a remote source, just log
105        if (ee.isRemote()) {
106          LOG.debug("Was remote foreign exception, not redispatching error", ee);
107          return;
108        }
109
110        // if it is local, then send it to the coordinator
111        try {
112          rpcs.sendMemberAborted(Subprocedure.this, ee);
113        } catch (IOException e) {
114          // this will fail all the running procedures, since the connection is down
115          LOG.error("Can't reach controller, not propagating error", e);
116        }
117      }
118    });
119
120    this.wakeFrequency = wakeFrequency;
121    this.inGlobalBarrier = new CountDownLatch(1);
122    this.releasedLocalBarrier = new CountDownLatch(1);
123
124    // accept error from timer thread, this needs to be started.
125    this.executionTimeoutTimer = new TimeoutExceptionInjector(monitor, timeout);
126  }
127
128  public String getName() {
129     return barrierName;
130  }
131
132  public String getMemberName() {
133    return rpcs.getMemberName();
134  }
135
136  private void rethrowException() throws ForeignException {
137    monitor.rethrowException();
138  }
139
140  /**
141   * Execute the Subprocedure {@link #acquireBarrier()} and {@link #insideBarrier()} methods
142   * while keeping some state for other threads to access.
143   *
144   * This would normally be executed by the ProcedureMemeber when a acquire message comes from the
145   * coordinator.  Rpcs are used to spend message back to the coordinator after different phases
146   * are executed.  Any exceptions caught during the execution (except for InterrupedException) get
147   * converted and propagated to coordinator via {@link ProcedureMemberRpcs#sendMemberAborted(
148   * Subprocedure, ForeignException)}.
149   */
150  @SuppressWarnings("finally")
151  final public Void call() {
152    LOG.debug("Starting subprocedure '" + barrierName + "' with timeout " +
153        executionTimeoutTimer.getMaxTime() + "ms");
154    // start the execution timeout timer
155    executionTimeoutTimer.start();
156
157    try {
158      // start by checking for error first
159      rethrowException();
160      LOG.debug("Subprocedure '" + barrierName + "' starting 'acquire' stage");
161      acquireBarrier();
162      LOG.debug("Subprocedure '" + barrierName + "' locally acquired");
163
164      // vote yes to coordinator about being prepared
165      rpcs.sendMemberAcquired(this);
166      LOG.debug("Subprocedure '" + barrierName + "' coordinator notified of 'acquire', waiting on" +
167          " 'reached' or 'abort' from coordinator");
168
169      // wait for the procedure to reach global barrier before proceding
170      waitForReachedGlobalBarrier();
171      rethrowException(); // if Coordinator aborts, will bail from here with exception
172
173      // In traditional 2PC, if a member reaches this state the TX has been committed and the
174      // member is responsible for rolling forward and recovering and completing the subsequent
175      // operations in the case of failure.  It cannot rollback.
176      //
177      // This implementation is not 2PC since it can still rollback here, and thus has different
178      // semantics.
179
180      LOG.debug("Subprocedure '" + barrierName + "' received 'reached' from coordinator.");
181      insideBarrier();
182      LOG.debug("Subprocedure '" + barrierName + "' locally completed");
183
184      // Ack that the member has executed and released local barrier
185      rpcs.sendMemberCompleted(this);
186      LOG.debug("Subprocedure '" + barrierName + "' has notified controller of completion");
187
188      // make sure we didn't get an external exception
189      rethrowException();
190    } catch (Exception e) {
191      String msg = null;
192      if (e instanceof InterruptedException) {
193        msg = "Procedure '" + barrierName + "' aborting due to interrupt!" +
194            " Likely due to pool shutdown.";
195        Thread.currentThread().interrupt();
196      } else if (e instanceof ForeignException) {
197        msg = "Subprocedure '" + barrierName + "' aborting due to a ForeignException!";
198      } else {
199        msg = "Subprocedure '" + barrierName + "' failed!";
200      }
201      cancel(msg, e);
202
203      LOG.debug("Subprocedure '" + barrierName + "' running cleanup.");
204      cleanup(e);
205    } finally {
206      releasedLocalBarrier.countDown();
207
208      // tell the timer we are done, if we get here successfully
209      executionTimeoutTimer.complete();
210      complete = true;
211      LOG.debug("Subprocedure '" + barrierName + "' completed.");
212      return null;
213    }
214  }
215
216  boolean isComplete() {
217    return complete;
218  }
219
220  /**
221   * exposed for testing.
222   */
223  ForeignExceptionSnare getErrorCheckable() {
224    return this.monitor;
225  }
226
227  /**
228   * The implementation of this method should gather and hold required resources (locks, disk
229   * space, etc) to satisfy the Procedures barrier condition.  For example, this would be where
230   * to make all the regions on a RS on the quiescent for an procedure that required all regions
231   * to be globally quiesed.
232   *
233   * Users should override this method.  If a quiescent is not required, this is overkill but
234   * can still be used to execute a procedure on all members and to propagate any exceptions.
235   *
236   * @throws ForeignException
237   */
238  abstract public void acquireBarrier() throws ForeignException;
239
240  /**
241   * The implementation of this method should act with the assumption that the barrier condition
242   * has been satisfied.  Continuing the previous example, a condition could be that all RS's
243   * globally have been quiesced, and procedures that require this precondition could be
244   * implemented here.
245   *
246   * Users should override this method.  If quiescense is not required, this can be a no-op
247   *
248   * @throws ForeignException
249   */
250  abstract public void insideBarrier() throws ForeignException;
251
252  /**
253   * Users should override this method. This implementation of this method should rollback and
254   * cleanup any temporary or partially completed state that the {@link #acquireBarrier()} may have
255   * created.
256   * @param e
257   */
258  abstract public void cleanup(Exception e);
259
260  /**
261   * Method to cancel the Subprocedure by injecting an exception from and external source.
262   * @param cause
263   */
264  public void cancel(String msg, Throwable cause) {
265    LOG.error(msg, cause);
266    if (cause instanceof ForeignException) {
267      monitor.receive((ForeignException) cause);
268    } else {
269      monitor.receive(new ForeignException(getMemberName(), cause));
270    }
271  }
272
273  /**
274   * Callback for the member rpcs to call when the global barrier has been reached.  This
275   * unblocks the main subprocedure exectuion thread so that the Subprocedure's
276   * {@link #insideBarrier()} method can be run.
277   */
278  public void receiveReachedGlobalBarrier() {
279    inGlobalBarrier.countDown();
280  }
281
282  //
283  // Subprocedure Internal State interface
284  //
285
286  /**
287   * Wait for the reached global barrier notification.
288   *
289   * Package visibility for testing
290   *
291   * @throws ForeignException
292   * @throws InterruptedException
293   */
294  void waitForReachedGlobalBarrier() throws ForeignException, InterruptedException {
295    Procedure.waitForLatch(inGlobalBarrier, monitor, wakeFrequency,
296        barrierName + ":remote acquired");
297  }
298
299  /**
300   * Waits until the entire procedure has globally completed, or has been aborted.
301   * @throws ForeignException
302   * @throws InterruptedException
303   */
304  public void waitForLocallyCompleted() throws ForeignException, InterruptedException {
305    Procedure.waitForLatch(releasedLocalBarrier, monitor, wakeFrequency,
306        barrierName + ":completed");
307  }
308
309  /**
310   * Empty Subprocedure for testing.
311   *
312   * Must be public for stubbing used in testing to work.
313   */
314  public static class SubprocedureImpl extends Subprocedure {
315
316    public SubprocedureImpl(ProcedureMember member, String opName,
317        ForeignExceptionDispatcher monitor, long wakeFrequency, long timeout) {
318      super(member, opName, monitor, wakeFrequency, timeout);
319    }
320
321    @Override
322    public void acquireBarrier() throws ForeignException {}
323
324    @Override
325    public void insideBarrier() throws ForeignException {}
326
327    @Override
328    public void cleanup(Exception e) {}
329  };
