020package org.apache.hadoop.hbase.monitoring;
021
022import java.io.PrintWriter;
023import java.text.SimpleDateFormat;
024import java.util.Date;
025import java.util.LinkedList;
026import java.util.List;
027
028import com.google.common.base.Charsets;
029import com.google.common.base.Preconditions;
030import com.google.common.collect.Lists;
031
032/**
033 * A size-bounded repository of alerts, which are kept
034 * in a linked list. Alerts can be added, and they will
035 * automatically be removed one by one when the specified heap
036 * usage is exhausted.
037 */
038public class MemoryBoundedLogMessageBuffer {
039  private final long maxSizeBytes;
040  private long usage = 0;
041  private LinkedList<LogMessage> messages;
042  
043  public MemoryBoundedLogMessageBuffer(long maxSizeBytes) {
044    Preconditions.checkArgument(
045        maxSizeBytes > 0);
046    this.maxSizeBytes = maxSizeBytes;
047    this.messages = Lists.newLinkedList();
048  }
049  
050  /**
051   * Append the given message to this buffer, automatically evicting
052   * older messages until the desired memory limit is achieved.
053   */
054  public synchronized void add(String messageText) {
055    LogMessage message = new LogMessage(messageText, System.currentTimeMillis());
056    
057    usage += message.estimateHeapUsage();
058    messages.add(message);
059    while (usage > maxSizeBytes) {
060      LogMessage removed = messages.remove();
061      usage -= removed.estimateHeapUsage();
062      assert usage >= 0;
063    }
064  }
065  
066  /**
067   * Dump the contents of the buffer to the given stream.
068   */
069  public synchronized void dumpTo(PrintWriter out) {
070    SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
071
072    for (LogMessage msg : messages) {
073      out.write(df.format(new Date(msg.timestamp)));
074      out.write(" ");
075      out.println(new String(msg.message, Charsets.UTF_8));
076    }
077  }
078  
079  synchronized List<LogMessage> getMessages() {
080    // defensive copy
081    return Lists.newArrayList(messages);
082  }
083 
084  /**
085   * Estimate the number of bytes this buffer is currently
086   * using.
087   */
088  synchronized long estimateHeapUsage() {
089    return usage;
090  }
091  
092  private static class LogMessage {
093    /** the error text, encoded in bytes to save memory */
094    public final byte[] message;
095    public final long timestamp;
096    
097    /**
098     * Completely non-scientific estimate of how much one of these
099     * objects takes, along with the LinkedList overhead. This doesn't
100     * need to be exact, since we don't expect a ton of these alerts.
101     */
102    private static final long BASE_USAGE=100;
103    
104    public LogMessage(String message, long timestamp) {
105      this.message = message.getBytes(Charsets.UTF_8);
106      this.timestamp = timestamp;
107    }
108    
109    public long estimateHeapUsage() {
110      return message.length + BASE_USAGE;
111    }
112  }
113
114}
