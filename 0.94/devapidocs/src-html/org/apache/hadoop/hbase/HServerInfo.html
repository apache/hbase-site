020package org.apache.hadoop.hbase;
021
022import java.io.DataInput;
023import java.io.DataOutput;
024import java.io.IOException;
025import java.net.InetSocketAddress;
026
027import org.apache.hadoop.hbase.regionserver.HRegionServer;
028import org.apache.hadoop.io.VersionedWritable;
029import org.apache.hadoop.io.WritableComparable;
030
031
032/**
033 * HServerInfo is meta info about an {@link HRegionServer}.  It hosts the
034 * {@link HServerAddress}, its webui port, and its server startcode.  It was
035 * used to pass meta info about a server across an RPC but we've since made
036 * it so regionserver info is up in ZooKeeper and so this class is on its
037 * way out. It used to carry {@link HServerLoad} but as off HBase 0.92.0, the
038 * HServerLoad is passed independent of this class. Also, we now no longer pass
039 * the webui from regionserver to master (TODO: Fix).
040 * @deprecated Use {@link InetSocketAddress} and or {@link ServerName} and or
041 * {@link HServerLoad}
042 */
043public class HServerInfo extends VersionedWritable
044implements WritableComparable<HServerInfo> {
045  private static final byte VERSION = 1;
046  private HServerAddress serverAddress = new HServerAddress();
047  private long startCode;
048  private int webuiport;
049
050  public HServerInfo() {
051    super();
052  }
053
054  /**
055   * Constructor that creates a HServerInfo with a generated startcode
056   * @param serverAddress
057   * @param webuiport Port the webui runs on.
058   */
059  public HServerInfo(final HServerAddress serverAddress, final int webuiport) {
060    this(serverAddress, System.currentTimeMillis(), webuiport);
061  }
062
063  public HServerInfo(HServerAddress serverAddress, long startCode,
064      final int webuiport) {
065    this.serverAddress = serverAddress;
066    this.startCode = startCode;
067    this.webuiport = webuiport;
068  }
069
070  /**
071   * Copy-constructor
072   * @param other
073   */
074  public HServerInfo(HServerInfo other) {
075    this.serverAddress = new HServerAddress(other.getServerAddress());
076    this.startCode = other.getStartCode();
077    this.webuiport = other.getInfoPort();
078  }
079
080  /** @return the object version number */
081  public byte getVersion() {
082    return VERSION;
083  }
084
085  public synchronized HServerAddress getServerAddress() {
086    return new HServerAddress(serverAddress);
087  }
088
089  public synchronized long getStartCode() {
090    return startCode;
091  }
092
093  public int getInfoPort() {
094    return getWebuiPort();
095  }
096
097  public int getWebuiPort() {
098    return this.webuiport;
099  }
100
101  public String getHostname() {
102    return this.serverAddress.getHostname();
103  }
104
105  /**
106   * @return ServerName and load concatenated.
107   */
108  @Override
109  public synchronized String toString() {
110    return ServerName.getServerName(this.serverAddress.getHostnameAndPort(),
111      this.startCode);
112  }
113
114  @Override
115  public boolean equals(Object obj) {
116    if (this == obj) return true;
117    if (obj == null) return false;
118    if (getClass() != obj.getClass()) return false;
119    return compareTo((HServerInfo)obj) == 0;
120  }
121
122  @Override
123  public int hashCode() {
124    int code = this.serverAddress.hashCode();
125    code ^= this.webuiport;
126    code ^= this.startCode;
127    return code;
128  }
129
130  public void readFields(DataInput in) throws IOException {
131    super.readFields(in);
132    this.serverAddress.readFields(in);
133    this.startCode = in.readLong();
134    this.webuiport = in.readInt();
135  }
136
137  public void write(DataOutput out) throws IOException {
138    super.write(out);
139    this.serverAddress.write(out);
140    out.writeLong(this.startCode);
141    out.writeInt(this.webuiport);
142  }
143
144  public int compareTo(HServerInfo o) {
145    int compare = this.serverAddress.compareTo(o.getServerAddress());
146    if (compare != 0) return compare;
147    if (this.webuiport != o.getInfoPort()) return this.webuiport - o.getInfoPort();
148    if (this.startCode != o.getStartCode()) return (int)(this.startCode - o.getStartCode());
149    return 0;
150  }
151}
