021package org.apache.hadoop.hbase.zookeeper;
022
023import java.util.ArrayList;
024import java.util.List;
025import java.util.Map.Entry;
026import java.util.Properties;
027
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.hbase.HBaseConfiguration;
030
031/**
032 * Tool for reading a ZooKeeper server from HBase XML configuration producing
033 * the '-server host:port' argument to pass ZooKeeperMain.  This program
034 * emits either '-server HOST:PORT" where HOST is one of the zk ensemble
035 * members plus zk client port OR it emits '' if no zk servers found (Yes,
036 * it emits '-server' too).
037 */
038public class ZooKeeperMainServerArg {
039  public String parse(final Configuration c) {
040    // Note that we do not simply grab the property
041    // HConstants.ZOOKEEPER_QUORUM from the HBaseConfiguration because the
042    // user may be using a zoo.cfg file.
043    Properties zkProps = ZKConfig.makeZKProps(c);
044    String host = null;
045    String clientPort = null;
046    List<String> hosts = new ArrayList<String>();
047    for (Entry<Object, Object> entry: zkProps.entrySet()) {
048      String key = entry.getKey().toString().trim();
049      String value = entry.getValue().toString().trim();
050      if (key.startsWith("server.")) {
051        String[] parts = value.split(":");
052        hosts.add(parts[0]);
053      } else if (key.endsWith("clientPort")) {
054        clientPort = value;
055      }
056    }
057    if (hosts.isEmpty() || clientPort == null)
058      return null;
059    for (int i = 0; i < hosts.size(); i++) {
060      if (i > 0)
061        host += "," + hosts.get(i);
062      else
063        host = hosts.get(i);
064    }
065    return host != null ? host + ":" + clientPort : null;
066  }
067
068  /**
069   * Run the tool.
070   * @param args Command line arguments. First arg is path to zookeepers file.
071   */
072  public static void main(String args[]) {
073    Configuration conf = HBaseConfiguration.create();
074    String hostport = new ZooKeeperMainServerArg().parse(conf);
075    System.out.println((hostport == null || hostport.length() == 0)? "":
076      "-server " + hostport);
077  }
078}
