018package org.apache.hadoop.hbase.regionserver.compactions;
019
020import static org.junit.Assert.assertFalse;
021import static org.junit.Assert.assertTrue;
022
023import org.apache.hadoop.conf.Configuration;
024import org.apache.hadoop.hbase.HBaseClassTestRule;
025import org.apache.hadoop.hbase.HBaseTestingUtility;
026import org.apache.hadoop.hbase.testclassification.RegionServerTests;
027import org.apache.hadoop.hbase.testclassification.SmallTests;
028import org.junit.Before;
029import org.junit.BeforeClass;
030import org.junit.ClassRule;
031import org.junit.Test;
032import org.junit.experimental.categories.Category;
033
034@Category({RegionServerTests.class, SmallTests.class})
035public class TestOffPeakHours {
036
037  @ClassRule
038  public static final HBaseClassTestRule CLASS_RULE =
039      HBaseClassTestRule.forClass(TestOffPeakHours.class);
040
041  private static HBaseTestingUtility testUtil;
042
043  @BeforeClass
044  public static void setUpClass() {
045    testUtil = new HBaseTestingUtility();
046  }
047
048  private int hourOfDay;
049  private int hourPlusOne;
050  private int hourMinusOne;
051  private int hourMinusTwo;
052  private Configuration conf;
053
054  @Before
055  public void setUp() {
056    hourOfDay = 15;
057    hourPlusOne = ((hourOfDay+1)%24);
058    hourMinusOne = ((hourOfDay-1+24)%24);
059    hourMinusTwo = ((hourOfDay-2+24)%24);
060    conf = testUtil.getConfiguration();
061  }
062
063  @Test
064  public void testWithoutSettings() {
065    Configuration conf = testUtil.getConfiguration();
066    OffPeakHours target = OffPeakHours.getInstance(conf);
067    assertFalse(target.isOffPeakHour(hourOfDay));
068  }
069
070  @Test
071  public void testSetPeakHourToTargetTime() {
072    conf.setLong(CompactionConfiguration.HBASE_HSTORE_OFFPEAK_START_HOUR, hourMinusOne);
073    conf.setLong(CompactionConfiguration.HBASE_HSTORE_OFFPEAK_END_HOUR, hourPlusOne);
074    OffPeakHours target = OffPeakHours.getInstance(conf);
075    assertTrue(target.isOffPeakHour(hourOfDay));
076  }
077
078  @Test
079  public void testSetPeakHourOutsideCurrentSelection() {
080    conf.setLong(CompactionConfiguration.HBASE_HSTORE_OFFPEAK_START_HOUR, hourMinusTwo);
081    conf.setLong(CompactionConfiguration.HBASE_HSTORE_OFFPEAK_END_HOUR, hourMinusOne);
082    OffPeakHours target = OffPeakHours.getInstance(conf);
083    assertFalse(target.isOffPeakHour(hourOfDay));
084  }
085}
