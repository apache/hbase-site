018package org.apache.hadoop.hbase.master.balancer;
019
020import org.apache.hadoop.hbase.HBaseClassTestRule;
021import org.apache.hadoop.hbase.testclassification.LargeTests;
022import org.apache.hadoop.hbase.testclassification.MasterTests;
023import org.junit.ClassRule;
024import org.junit.Test;
025import org.junit.experimental.categories.Category;
026
027@Category({ MasterTests.class, LargeTests.class })
028public class TestStochasticLoadBalancerMidCluster extends BalancerTestBase {
029
030  @ClassRule
031  public static final HBaseClassTestRule CLASS_RULE =
032      HBaseClassTestRule.forClass(TestStochasticLoadBalancerMidCluster.class);
033
034  @Test
035  public void testMidCluster() {
036    int numNodes = 100;
037    int numRegions = 10000;
038    int numRegionsPerServer = 60; // all servers except one
039    int replication = 1;
040    int numTables = 40;
041    testWithCluster(numNodes, numRegions, numRegionsPerServer, replication, numTables, true, true);
042  }
043
044  @Test
045  public void testMidCluster2() {
046    int numNodes = 200;
047    int numRegions = 100000;
048    int numRegionsPerServer = 40; // all servers except one
049    int replication = 1;
050    int numTables = 400;
051    // num large num regions means may not always get to best balance with one run
052    boolean assertFullyBalanced = false;
053    testWithCluster(numNodes, numRegions, numRegionsPerServer, replication, numTables,
054      assertFullyBalanced, false);
055  }
056
057  @Test
058  public void testMidCluster3() {
059    int numNodes = 100;
060    int numRegions = 2000;
061    int numRegionsPerServer = 9; // all servers except one
062    int replication = 1;
063    int numTables = 110;
064    testWithCluster(numNodes, numRegions, numRegionsPerServer, replication, numTables, true, true);
065    // TODO(eclark): Make sure that the tables are well distributed.
066  }
067}
