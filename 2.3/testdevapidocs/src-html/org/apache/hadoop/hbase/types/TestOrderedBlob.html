018package org.apache.hadoop.hbase.types;
019
020import static org.junit.Assert.assertEquals;
021
022import org.apache.hadoop.hbase.HBaseClassTestRule;
023import org.apache.hadoop.hbase.testclassification.MiscTests;
024import org.apache.hadoop.hbase.testclassification.SmallTests;
025import org.apache.hadoop.hbase.util.Bytes;
026import org.apache.hadoop.hbase.util.PositionedByteRange;
027import org.apache.hadoop.hbase.util.SimplePositionedMutableByteRange;
028import org.junit.ClassRule;
029import org.junit.Test;
030import org.junit.experimental.categories.Category;
031
032@Category({MiscTests.class, SmallTests.class})
033public class TestOrderedBlob {
034
035  @ClassRule
036  public static final HBaseClassTestRule CLASS_RULE =
037      HBaseClassTestRule.forClass(TestOrderedBlob.class);
038
039  static final byte[][] VALUES = new byte[][] {
040    null, Bytes.toBytes(""), Bytes.toBytes("1"), Bytes.toBytes("22"), Bytes.toBytes("333"),
041    Bytes.toBytes("4444"), Bytes.toBytes("55555"), Bytes.toBytes("666666"),
042    Bytes.toBytes("7777777"), Bytes.toBytes("88888888"), Bytes.toBytes("999999999"),
043  };
044
045  @Test
046  public void testEncodedLength() {
047    PositionedByteRange buff = new SimplePositionedMutableByteRange(20);
048    for (DataType<byte[]> type : new OrderedBlob[] { OrderedBlob.ASCENDING, OrderedBlob.DESCENDING }) {
049      for (byte[] val : VALUES) {
050        buff.setPosition(0);
051        type.encode(buff, val);
052        assertEquals(
053          "encodedLength does not match actual, " + Bytes.toStringBinary(val),
054          buff.getPosition(), type.encodedLength(val));
055      }
056    }
057  }
058}
