018package org.apache.hadoop.hbase.rest.model;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertFalse;
022
023import java.util.Iterator;
024import org.apache.hadoop.hbase.HBaseClassTestRule;
025import org.apache.hadoop.hbase.testclassification.RestTests;
026import org.apache.hadoop.hbase.testclassification.SmallTests;
027import org.junit.ClassRule;
028import org.junit.experimental.categories.Category;
029
030@Category({RestTests.class, SmallTests.class})
031public class TestTableListModel extends TestModelBase<TableListModel> {
032
033  @ClassRule
034  public static final HBaseClassTestRule CLASS_RULE =
035      HBaseClassTestRule.forClass(TestTableListModel.class);
036
037  private static final String TABLE1 = "table1";
038  private static final String TABLE2 = "table2";
039  private static final String TABLE3 = "table3";
040
041  public TestTableListModel() throws Exception {
042    super(TableListModel.class);
043    AS_XML =
044      "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?><TableList><table " +
045          "name=\"table1\"/><table name=\"table2\"/><table name=\"table3\"/></TableList>";
046
047    AS_PB = "CgZ0YWJsZTEKBnRhYmxlMgoGdGFibGUz";
048
049    AS_JSON =
050      "{\"table\":[{\"name\":\"table1\"},{\"name\":\"table2\"},{\"name\":\"table3\"}]}";
051  }
052
053  @Override
054  protected TableListModel buildTestModel() {
055    TableListModel model = new TableListModel();
056    model.add(new TableModel(TABLE1));
057    model.add(new TableModel(TABLE2));
058    model.add(new TableModel(TABLE3));
059    return model;
060  }
061
062  @Override
063  protected void checkModel(TableListModel model) {
064    Iterator<TableModel> tables = model.getTables().iterator();
065    TableModel table = tables.next();
066    assertEquals(TABLE1, table.getName());
067    table = tables.next();
068    assertEquals(TABLE2, table.getName());
069    table = tables.next();
070    assertEquals(TABLE3, table.getName());
071    assertFalse(tables.hasNext());
072  }
073}
