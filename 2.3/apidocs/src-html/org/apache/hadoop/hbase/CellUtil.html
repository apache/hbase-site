169  /**
170   * Splits a column in {@code family:qualifier} form into separate byte arrays. An empty qualifier
171   * (ie, {@code fam:}) is parsed as <code>{ fam, EMPTY_BYTE_ARRAY }</code> while no delimiter (ie,
172   * {@code fam}) is parsed as an array of one element, <code>{ fam }</code>.
173   * <p>
174   * Don't forget, HBase DOES support empty qualifiers. (see HBASE-9549)
175   * </p>
176   * <p>
177   * Not recommend to be used as this is old-style API.
178   * </p>
179   * @param c The column.
180   * @return The parsed column.
181   */
182  public static byte[][] parseColumn(byte[] c) {
183    final int index = getDelimiter(c, 0, c.length, COLUMN_FAMILY_DELIMITER);
184    if (index == -1) {
185      // If no delimiter, return array of size 1
186      return new byte[][] { c };
187    } else if (index == c.length - 1) {
188      // family with empty qualifier, return array size 2
189      byte[] family = new byte[c.length - 1];
190      System.arraycopy(c, 0, family, 0, family.length);
191      return new byte[][] { family, HConstants.EMPTY_BYTE_ARRAY };
192    }
193    // Family and column, return array size 2
194    final byte[][] result = new byte[2][];
195    result[0] = new byte[index];
196    System.arraycopy(c, 0, result[0], 0, index);
197    final int len = c.length - (index + 1);
198    result[1] = new byte[len];
199    System.arraycopy(c, index + 1 /* Skip delimiter */, result[1], 0, len);
200    return result;
201  }

205  /**
206   * Copies the row to the given byte[]
207   * @param cell the cell whose row has to be copied
208   * @param destination the destination byte[] to which the row has to be copied
209   * @param destinationOffset the offset in the destination byte[]
210   * @return the offset of the byte[] after the copy has happened
211   */
212  public static int copyRowTo(Cell cell, byte[] destination, int destinationOffset) {
213    short rowLen = cell.getRowLength();
214    if (cell instanceof ByteBufferExtendedCell) {
215      ByteBufferUtils.copyFromBufferToArray(destination,
216          ((ByteBufferExtendedCell) cell).getRowByteBuffer(),
217          ((ByteBufferExtendedCell) cell).getRowPosition(), destinationOffset, rowLen);
218    } else {
219      System.arraycopy(cell.getRowArray(), cell.getRowOffset(), destination, destinationOffset,
220        rowLen);
221    }
222    return destinationOffset + rowLen;
223  }

1374  /**
1375   * Find length of common prefix in keys of the cells, considering key as byte[] if serialized in
1376   * {@link KeyValue}. The key format is &lt;2 bytes rk len&gt;&lt;rk&gt;&lt;1 byte cf
1377   * len&gt;&lt;cf&gt;&lt;qualifier&gt;&lt;8 bytes timestamp&gt;&lt;1 byte type&gt;
1378   * @param c1 the cell
1379   * @param c2 the cell
1380   * @param bypassFamilyCheck when true assume the family bytes same in both cells. Pass it as true
1381   *          when dealing with Cells in same CF so as to avoid some checks
1382   * @param withTsType when true check timestamp and type bytes also.
1383   * @return length of common prefix
1384   * @deprecated As of HBase-2.0. Will be removed in HBase-3.0
1385   */
1386  @Deprecated
1387  public static int findCommonPrefixInFlatKey(Cell c1, Cell c2, boolean bypassFamilyCheck,
1388    boolean withTsType) {
1389    // Compare the 2 bytes in RK length part
1390    short rLen1 = c1.getRowLength();
1391    short rLen2 = c2.getRowLength();
1392    int commonPrefix = KeyValue.ROW_LENGTH_SIZE;
1393    if (rLen1 != rLen2) {
1394      // early out when the RK length itself is not matching
1395      return ByteBufferUtils
1396        .findCommonPrefix(Bytes.toBytes(rLen1), 0, KeyValue.ROW_LENGTH_SIZE, Bytes.toBytes(rLen2),
1397          0, KeyValue.ROW_LENGTH_SIZE);
1398    }
1399    // Compare the RKs
1400    int rkCommonPrefix = 0;
1401    if (c1 instanceof ByteBufferExtendedCell && c2 instanceof ByteBufferExtendedCell) {
1402      rkCommonPrefix = ByteBufferUtils
1403        .findCommonPrefix(((ByteBufferExtendedCell) c1).getRowByteBuffer(),
1404          ((ByteBufferExtendedCell) c1).getRowPosition(), rLen1,
1405          ((ByteBufferExtendedCell) c2).getRowByteBuffer(),
1406          ((ByteBufferExtendedCell) c2).getRowPosition(), rLen2);
1407    } else {
1408      // There cannot be a case where one cell is BBCell and other is KeyValue. This flow comes
1409      // either
1410      // in flush or compactions. In flushes both cells are KV and in case of compaction it will be
1411      // either
1412      // KV or BBCell
1413      rkCommonPrefix = ByteBufferUtils
1414        .findCommonPrefix(c1.getRowArray(), c1.getRowOffset(), rLen1, c2.getRowArray(),
1415          c2.getRowOffset(), rLen2);
1416    }
1417    commonPrefix += rkCommonPrefix;
1418    if (rkCommonPrefix != rLen1) {
1419      // Early out when RK is not fully matching.
1420      return commonPrefix;
1421    }
1422    // Compare 1 byte CF length part
1423    byte fLen1 = c1.getFamilyLength();
1424    if (bypassFamilyCheck) {
1425      // This flag will be true when caller is sure that the family will be same for both the cells
1426      // Just make commonPrefix to increment by the family part
1427      commonPrefix += KeyValue.FAMILY_LENGTH_SIZE + fLen1;
1428    } else {
1429      byte fLen2 = c2.getFamilyLength();
1430      if (fLen1 != fLen2) {
1431        // early out when the CF length itself is not matching
1432        return commonPrefix;
1433      }
1434      // CF lengths are same so there is one more byte common in key part
1435      commonPrefix += KeyValue.FAMILY_LENGTH_SIZE;
1436      // Compare the CF names
1437      int fCommonPrefix;
1438      if (c1 instanceof ByteBufferExtendedCell && c2 instanceof ByteBufferExtendedCell) {
1439        fCommonPrefix = ByteBufferUtils
1440          .findCommonPrefix(((ByteBufferExtendedCell) c1).getFamilyByteBuffer(),
1441            ((ByteBufferExtendedCell) c1).getFamilyPosition(), fLen1,
1442            ((ByteBufferExtendedCell) c2).getFamilyByteBuffer(),
1443            ((ByteBufferExtendedCell) c2).getFamilyPosition(), fLen2);
1444      } else {
1445        fCommonPrefix = ByteBufferUtils
1446          .findCommonPrefix(c1.getFamilyArray(), c1.getFamilyOffset(), fLen1, c2.getFamilyArray(),
1447            c2.getFamilyOffset(), fLen2);
1448      }
1449      commonPrefix += fCommonPrefix;
1450      if (fCommonPrefix != fLen1) {
1451        return commonPrefix;
1452      }
1453    }
1454    // Compare the Qualifiers
1455    int qLen1 = c1.getQualifierLength();
1456    int qLen2 = c2.getQualifierLength();
1457    int qCommon;
1458    if (c1 instanceof ByteBufferExtendedCell && c2 instanceof ByteBufferExtendedCell) {
1459      qCommon = ByteBufferUtils
1460        .findCommonPrefix(((ByteBufferExtendedCell) c1).getQualifierByteBuffer(),
1461          ((ByteBufferExtendedCell) c1).getQualifierPosition(), qLen1,
1462          ((ByteBufferExtendedCell) c2).getQualifierByteBuffer(),
1463          ((ByteBufferExtendedCell) c2).getQualifierPosition(), qLen2);
1464    } else {
1465      qCommon = ByteBufferUtils
1466        .findCommonPrefix(c1.getQualifierArray(), c1.getQualifierOffset(), qLen1,
1467          c2.getQualifierArray(), c2.getQualifierOffset(), qLen2);
1468    }
1469    commonPrefix += qCommon;
1470    if (!withTsType || Math.max(qLen1, qLen2) != qCommon) {
1471      return commonPrefix;
1472    }
1473    // Compare the timestamp parts
1474    int tsCommonPrefix = ByteBufferUtils
1475      .findCommonPrefix(Bytes.toBytes(c1.getTimestamp()), 0, KeyValue.TIMESTAMP_SIZE,
1476        Bytes.toBytes(c2.getTimestamp()), 0, KeyValue.TIMESTAMP_SIZE);
1477    commonPrefix += tsCommonPrefix;
1478    if (tsCommonPrefix != KeyValue.TIMESTAMP_SIZE) {
1479      return commonPrefix;
1480    }
1481    // Compare the type
1482    if (c1.getTypeByte() == c2.getTypeByte()) {
1483      commonPrefix += KeyValue.TYPE_SIZE;
1484    }
1485    return commonPrefix;
1486  }
