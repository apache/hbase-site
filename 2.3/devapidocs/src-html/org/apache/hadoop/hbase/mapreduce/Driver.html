019package org.apache.hadoop.hbase.mapreduce;
020
021import org.apache.hadoop.hbase.HBaseInterfaceAudience;
022import org.apache.hadoop.hbase.mapreduce.replication.VerifyReplication;
023import org.apache.hadoop.hbase.mob.mapreduce.MobRefReporter;
024import org.apache.hadoop.hbase.snapshot.ExportSnapshot;
025import org.apache.hadoop.hbase.tool.BulkLoadHFilesTool;
026import org.apache.hadoop.util.ProgramDriver;
027import org.apache.yetus.audience.InterfaceAudience;
028import org.apache.yetus.audience.InterfaceStability;
029
030/**
031 * Driver for hbase mapreduce jobs. Select which to run by passing
032 * name of job to this main.
033 */
034@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
035@InterfaceStability.Stable
036public class Driver {
037
038  public static void main(String[] args) throws Throwable {
039    ProgramDriver pgd = new ProgramDriver();
040
041    pgd.addClass(RowCounter.NAME, RowCounter.class,
042      "Count rows in HBase table.");
043    pgd.addClass(CellCounter.NAME, CellCounter.class,
044      "Count cells in HBase table.");
045    pgd.addClass(Export.NAME, Export.class, "Write table data to HDFS.");
046    pgd.addClass(Import.NAME, Import.class, "Import data written by Export.");
047    pgd.addClass(ImportTsv.NAME, ImportTsv.class, "Import data in TSV format.");
048    pgd.addClass(BulkLoadHFilesTool.NAME, BulkLoadHFilesTool.class,
049                 "Complete a bulk data load.");
050    pgd.addClass(CopyTable.NAME, CopyTable.class,
051        "Export a table from local cluster to peer cluster.");
052    pgd.addClass(VerifyReplication.NAME, VerifyReplication.class, "Compare" +
053        " data from tables in two different clusters. It" +
054        " doesn't work for incrementColumnValues'd cells since" +
055        " timestamp is changed after appending to WAL.");
056    pgd.addClass(WALPlayer.NAME, WALPlayer.class, "Replay WAL files.");
057    pgd.addClass(ExportSnapshot.NAME, ExportSnapshot.class, "Export" +
058        " the specific snapshot to a given FileSystem.");
059    pgd.addClass(MobRefReporter.NAME, MobRefReporter.class, "Check the mob cells in a particular " +
060        "table and cf and confirm that the files they point to are correct.");
061
062    ProgramDriver.class.getMethod("driver", new Class [] {String[].class}).
063      invoke(pgd, new Object[]{args});
064  }
065}
