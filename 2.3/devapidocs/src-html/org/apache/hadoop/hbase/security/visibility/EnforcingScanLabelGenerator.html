018package org.apache.hadoop.hbase.security.visibility;
019
020import java.util.ArrayList;
021import java.util.HashSet;
022import java.util.List;
023import java.util.Set;
024
025import org.apache.yetus.audience.InterfaceAudience;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.hbase.security.User;
030
031/**
032 * This ScanLabelGenerator enforces a set of predefined authorizations for a
033 * given user, the set defined by the admin using the VisibilityClient admin
034 * interface or the set_auths shell command. Any authorizations requested with
035 * Scan#authorizations will be ignored.
036 */
037@InterfaceAudience.Private
038public class EnforcingScanLabelGenerator implements ScanLabelGenerator {
039
040  private static final Logger LOG = LoggerFactory.getLogger(EnforcingScanLabelGenerator.class);
041
042  private Configuration conf;
043  private VisibilityLabelsCache labelsCache;
044
045  public EnforcingScanLabelGenerator() {
046    this.labelsCache = VisibilityLabelsCache.get();
047  }
048
049  @Override
050  public void setConf(Configuration conf) {
051    this.conf = conf;
052  }
053
054  @Override
055  public Configuration getConf() {
056    return this.conf;
057  }
058
059  @Override
060  public List<String> getLabels(User user, Authorizations authorizations) {
061    String userName = user.getShortName();
062    if (authorizations != null) {
063      LOG.warn("Dropping authorizations requested by user " + userName + ": " + authorizations);
064    }
065    Set<String> auths = new HashSet<>();
066    auths.addAll(this.labelsCache.getUserAuths(userName));
067    auths.addAll(this.labelsCache.getGroupAuths(user.getGroupNames()));
068    return new ArrayList<>(auths);
069  }
070
071}
