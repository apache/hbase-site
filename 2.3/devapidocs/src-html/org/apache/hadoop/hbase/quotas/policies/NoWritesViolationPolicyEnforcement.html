017package org.apache.hadoop.hbase.quotas.policies;
018
019import org.apache.yetus.audience.InterfaceAudience;
020import org.apache.hadoop.hbase.client.Append;
021import org.apache.hadoop.hbase.client.Delete;
022import org.apache.hadoop.hbase.client.Increment;
023import org.apache.hadoop.hbase.client.Mutation;
024import org.apache.hadoop.hbase.client.Put;
025import org.apache.hadoop.hbase.quotas.SpaceLimitingException;
026import org.apache.hadoop.hbase.quotas.SpaceViolationPolicy;
027import org.apache.hadoop.hbase.quotas.SpaceViolationPolicyEnforcement;
028
029/**
030 * A {@link SpaceViolationPolicyEnforcement} implementation which disables all writes flowing
031 * into HBase. The enforcement counterpart to {@link SpaceViolationPolicy#NO_WRITES}.
032 */
033@InterfaceAudience.Private
034public class NoWritesViolationPolicyEnforcement extends DefaultViolationPolicyEnforcement {
035
036  @Override
037  public void enable() {}
038
039  @Override
040  public void disable() {}
041
042  @Override
043  public void check(Mutation m) throws SpaceLimitingException {
044    if (m instanceof Append || m instanceof Delete || m instanceof Increment || m instanceof Put) {
045      throw new SpaceLimitingException(getPolicyName(),
046          m.getClass().getSimpleName() + "s are disallowed due to a space quota.");
047    }
048  }
049
050  @Override
051  public String getPolicyName() {
052    return SpaceViolationPolicy.NO_WRITES.name();
053  }
054}
