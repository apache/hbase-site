019package org.apache.hadoop.hbase.replication.regionserver;
020
021import org.apache.hadoop.conf.Configuration;
022import org.apache.yetus.audience.InterfaceAudience;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025import org.apache.hadoop.hbase.replication.ReplicationQueueInfo;
026
027/**
028 * Constructs a {@link ReplicationSourceInterface}
029 */
030@InterfaceAudience.Private
031public class ReplicationSourceFactory {
032
033  private static final Logger LOG = LoggerFactory.getLogger(ReplicationSourceFactory.class);
034
035  static ReplicationSourceInterface create(Configuration conf, String queueId) {
036    ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(queueId);
037    boolean isQueueRecovered = replicationQueueInfo.isQueueRecovered();
038    ReplicationSourceInterface src;
039    try {
040      String defaultReplicationSourceImpl =
041          isQueueRecovered ? RecoveredReplicationSource.class.getCanonicalName()
042              : ReplicationSource.class.getCanonicalName();
043      Class<?> c = Class.forName(
044        conf.get("replication.replicationsource.implementation", defaultReplicationSourceImpl));
045      src = c.asSubclass(ReplicationSourceInterface.class).getDeclaredConstructor().newInstance();
046    } catch (Exception e) {
047      LOG.warn("Passed replication source implementation throws errors, "
048          + "defaulting to ReplicationSource",
049        e);
050      src = isQueueRecovered ? new RecoveredReplicationSource() : new ReplicationSource();
051    }
052    return src;
053  }
054}
