018package org.apache.hadoop.hbase.client;
019
020import org.apache.hadoop.hbase.client.metrics.ScanMetrics;
021import org.apache.yetus.audience.InterfaceAudience;
022
023/**
024 * The base interface for scan result consumer.
025 */
026@InterfaceAudience.Public
027public interface ScanResultConsumerBase {
028  /**
029   * Indicate that we hit an unrecoverable error and the scan operation is terminated.
030   * <p>
031   * We will not call {@link #onComplete()} after calling {@link #onError(Throwable)}.
032   */
033  void onError(Throwable error);
034
035  /**
036   * Indicate that the scan operation is completed normally.
037   */
038  void onComplete();
039
040  /**
041   * If {@code scan.isScanMetricsEnabled()} returns true, then this method will be called prior to
042   * all other methods in this interface to give you the {@link ScanMetrics} instance for this scan
043   * operation. The {@link ScanMetrics} instance will be updated on-the-fly during the scan, you can
044   * store it somewhere to get the metrics at any time if you want.
045   */
046  default void onScanMetricsCreated(ScanMetrics scanMetrics) {
047  }
048}
