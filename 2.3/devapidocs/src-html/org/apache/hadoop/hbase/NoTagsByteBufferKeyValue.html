018package org.apache.hadoop.hbase;
019
020import java.nio.ByteBuffer;
021
022import org.apache.hadoop.hbase.util.ByteBufferUtils;
023import org.apache.yetus.audience.InterfaceAudience;
024
025/**
026 * An extension of the ByteBufferKeyValue where the tags length is always 0
027 */
028@InterfaceAudience.Private
029public class NoTagsByteBufferKeyValue extends ByteBufferKeyValue {
030
031  public NoTagsByteBufferKeyValue(ByteBuffer buf, int offset, int length) {
032    super(buf, offset, length);
033  }
034
035  public NoTagsByteBufferKeyValue(ByteBuffer buf, int offset, int length, long seqId) {
036    super(buf, offset, length, seqId);
037  }
038
039  @Override
040  public byte[] getTagsArray() {
041    return HConstants.EMPTY_BYTE_ARRAY;
042  }
043
044  @Override
045  public int getTagsLength() {
046    return 0;
047  }
048
049  @Override
050  public int getSerializedSize(boolean withTags) {
051    return this.length;
052  }
053
054  @Override
055  public ExtendedCell deepClone() {
056    byte[] copy = new byte[this.length];
057    ByteBufferUtils.copyFromBufferToArray(copy, this.buf, this.offset, 0, this.length);
058    KeyValue kv = new NoTagsKeyValue(copy, 0, copy.length);
059    kv.setSequenceId(this.getSequenceId());
060    return kv;
061  }
062}
