94        this.conf.getInt(HConstants.COMPACTION_KV_MAX, HConstants.COMPACTION_KV_MAX_DEFAULT);

98        HConstants.MIN_KEEP_SEQID_PERIOD), HConstants.MIN_KEEP_SEQID_PERIOD);
99      this.dropCacheMajor = conf.getBoolean(MAJOR_COMPACTION_DROP_CACHE, true);
100     this.dropCacheMinor = conf.getBoolean(MINOR_COMPACTION_DROP_CACHE, true);

125     public long maxMVCCReadpoint = 0;

132   /**
133    * Extracts some details about the files to compact that are commonly needed by compactors.
134    * @param filesToCompact Files.
135    * @param allFiles Whether all files are included for compaction
136    * @return The result.
137    */
138   protected FileDetails getFileDetails(
139       Collection<StoreFile> filesToCompact, boolean allFiles) throws IOException {
140     FileDetails fd = new FileDetails();
141     long oldestHFileTimeStampToKeepMVCC = System.currentTimeMillis() - 
142       (1000L * 60 * 60 * 24 * this.keepSeqIdPeriod);  
143 
144     for (StoreFile file : filesToCompact) {
145       if(allFiles && (file.getModificationTimeStamp() < oldestHFileTimeStampToKeepMVCC)) {
146         // when isAllFiles is true, all files are compacted so we can calculate the smallest 
147         // MVCC value to keep
148         if(fd.minSeqIdToKeep < file.getMaxMemstoreTS()) {
149           fd.minSeqIdToKeep = file.getMaxMemstoreTS();
150         }
151       }
152       long seqNum = file.getMaxSequenceId();
153       fd.maxSeqId = Math.max(fd.maxSeqId, seqNum);
154       StoreFile.Reader r = file.getReader();
155       if (r == null) {
156         LOG.warn("Null reader for " + file.getPath());
157         continue;
158       }
159       // NOTE: use getEntries when compacting instead of getFilterEntries, otherwise under-sized
160       // blooms can cause progress to be miscalculated or if the user switches bloom
161       // type (e.g. from ROW to ROWCOL)
162       long keyCount = r.getEntries();
163       fd.maxKeyCount += keyCount;
164       // calculate the latest MVCC readpoint in any of the involved store files
165       Map<byte[], byte[]> fileInfo = r.loadFileInfo();
166       byte[] tmp = null;
167       // Get and set the real MVCCReadpoint for bulk loaded files, which is the
168       // SeqId number.
169       if (r.isBulkLoaded()) {
170         fd.maxMVCCReadpoint = Math.max(fd.maxMVCCReadpoint, r.getSequenceID());
171       }
172       else {
173         tmp = fileInfo.get(HFileWriterV2.MAX_MEMSTORE_TS_KEY);
174         if (tmp != null) {
175           fd.maxMVCCReadpoint = Math.max(fd.maxMVCCReadpoint, Bytes.toLong(tmp));
176         }
177       }
178       tmp = fileInfo.get(FileInfo.MAX_TAGS_LEN);
179       if (tmp != null) {
180         fd.maxTagsLength = Math.max(fd.maxTagsLength, Bytes.toInt(tmp));
181       }
182       // If required, calculate the earliest put timestamp of all involved storefiles.
183       // This is used to remove family delete marker during compaction.
184       long earliestPutTs = 0;
185       if (allFiles) {
186         tmp = fileInfo.get(StoreFile.EARLIEST_PUT_TS);
187         if (tmp == null) {
188           // There's a file with no information, must be an old one
189           // assume we have very old puts
190           fd.earliestPutTs = earliestPutTs = HConstants.OLDEST_TIMESTAMP;
191         } else {
192           earliestPutTs = Bytes.toLong(tmp);
193           fd.earliestPutTs = Math.min(fd.earliestPutTs, earliestPutTs);
194         }
195       }
196       if (LOG.isDebugEnabled()) {
197         LOG.debug("Compacting " + file +
198           ", keycount=" + keyCount +
199           ", bloomtype=" + r.getBloomFilterType().toString() +
200           ", size=" + TraditionalBinaryPrefix.long2String(r.length(), "", 1) +
201           ", encoding=" + r.getHFileReader().getDataBlockEncoding() +
202           ", compression=" + compactionCompression +
203           ", seqNum=" + seqNum +
204           (allFiles ? ", earliestPutTs=" + earliestPutTs: ""));
205       }
206     }
207     return fd;
208   }

270   protected List<Path> compact(final CompactionRequest request,
271       InternalScannerFactory scannerFactory, CellSinkFactory<T> sinkFactory,
272       ThroughputController throughputController, User user) throws IOException {
273     FileDetails fd = getFileDetails(request.getFiles(), request.isAllFiles());
274     this.progress = new CompactionProgress(fd.maxKeyCount);
275 
276     // Find the smallest read point across all the Scanners.
277     long smallestReadPoint = getSmallestReadPoint();
278 
279     List<StoreFileScanner> scanners;
280     Collection<StoreFile> readersToClose;
281     T writer = null;
282     boolean dropCache;
283     if (request.isMajor() || request.isAllFiles()) {
284       dropCache = this.dropCacheMajor;
285     } else {
286       dropCache = this.dropCacheMinor;
287     }
288 
289     if (this.conf.getBoolean("hbase.regionserver.compaction.private.readers", true)) {
290       // clone all StoreFiles, so we'll do the compaction on a independent copy of StoreFiles,
291       // HFiles, and their readers
292       readersToClose = new ArrayList<StoreFile>(request.getFiles().size());
293       for (StoreFile f : request.getFiles()) {
294         readersToClose.add(f.cloneForReader());
295       }
296       scanners = createFileScanners(readersToClose, smallestReadPoint, dropCache);
297     } else {
298       readersToClose = Collections.emptyList();
299       scanners = createFileScanners(request.getFiles(), smallestReadPoint, dropCache);
300     }
301     InternalScanner scanner = null;
302     boolean finished = false;
303     try {
304       /* Include deletes, unless we are doing a major compaction */
305       ScanType scanType = scannerFactory.getScanType(request);
306       scanner = preCreateCoprocScanner(request, scanType, fd.earliestPutTs, scanners, user,
307         smallestReadPoint);
308       if (scanner == null) {
309         scanner = scannerFactory.createScanner(scanners, scanType, fd, smallestReadPoint);
310       }
311       scanner = postCreateCoprocScanner(request, scanType, scanner, user);
312       if (scanner == null) {
313         // NULL scanner returned from coprocessor hooks means skip normal processing.
314         return new ArrayList<Path>();
315       }
316       boolean cleanSeqId = false;
317       if (fd.minSeqIdToKeep > 0) {
318         smallestReadPoint = Math.min(fd.minSeqIdToKeep, smallestReadPoint);
319         cleanSeqId = true;
320       }
321       writer = sinkFactory.createWriter(scanner, fd, dropCache);
322       finished =
323           performCompaction(scanner, writer, smallestReadPoint, cleanSeqId, throughputController);
324       if (!finished) {
325         throw new InterruptedIOException("Aborting compaction of store " + store + " in region "
326             + store.getRegionInfo().getRegionNameAsString() + " because it was interrupted.");
327       }
328     } finally {
329       Closeables.close(scanner, true);
330       for (StoreFile f : readersToClose) {
331         try {
332           f.closeReader(true);
333         } catch (IOException e) {
334           LOG.warn("Exception closing " + f, e);
335         }
336       }
337       if (!finished && writer != null) {
338         abortWriter(writer);
339       }
340     }
341     assert finished : "We should have exited the method on all error paths";
342     assert writer != null : "Writer should be non-null if no error";
343     return commitWriter(writer, fd, request);
344   }

351   /**
352    * Calls coprocessor, if any, to create compaction scanner - before normal scanner creation.
353    * @param request Compaction request.
354    * @param scanType Scan type.
355    * @param earliestPutTs Earliest put ts.
356    * @param scanners File scanners for compaction files.
357    * @param user the User
358    * @param readPoint the read point to help create scanner by Coprocessor if required.
359    * @return Scanner override by coprocessor; null if not overriding.
360    */
361   protected InternalScanner preCreateCoprocScanner(final CompactionRequest request,
362       final ScanType scanType, final long earliestPutTs, final List<StoreFileScanner> scanners,
363       User user, final long readPoint) throws IOException {
364     if (store.getCoprocessorHost() == null) {
365       return null;
366     }
367     if (user == null) {
368       return store.getCoprocessorHost().preCompactScannerOpen(store, scanners, scanType,
369         earliestPutTs, request, readPoint);
370     } else {
371       try {
372         return user.getUGI().doAs(new PrivilegedExceptionAction<InternalScanner>() {
373           @Override
374           public InternalScanner run() throws Exception {
375             return store.getCoprocessorHost().preCompactScannerOpen(store, scanners,
376               scanType, earliestPutTs, request, readPoint);
377           }
378         });
379       } catch (InterruptedException ie) {
380         InterruptedIOException iioe = new InterruptedIOException();
381         iioe.initCause(ie);
382         throw iioe;
383       }
384     }
385   }

398     }

404           @Override
405           public InternalScanner run() throws Exception {
406             return store.getCoprocessorHost().preCompact(store, scanner, scanType, request);
407           }

409       } catch (InterruptedException ie) {
410         InterruptedIOException iioe = new InterruptedIOException();
411         iioe.initCause(ie);
412         throw iioe;
413       }

417   /**
418    * Performs the compaction.
419    * @param scanner Where to read from.
420    * @param writer Where to write to.
421    * @param smallestReadPoint Smallest read point.
422    * @param cleanSeqId When true, remove seqId(used to be mvcc) value which is &lt;=
423    *          smallestReadPoint
424    * @return Whether compaction ended; false if it was interrupted for some reason.
425    */
426   protected boolean performCompaction(InternalScanner scanner, CellSink writer,
427       long smallestReadPoint, boolean cleanSeqId,
428       ThroughputController throughputController) throws IOException {
429     long bytesWritten = 0;
430     long bytesWrittenProgress = 0;
431     // Since scanner.next() can return 'false' but still be delivering data,
432     // we have to use a do/while loop.
433     List<Cell> cells = new ArrayList<Cell>();
434     long closeCheckInterval = HStore.getCloseCheckInterval();
435     long lastMillis = 0;
436     if (LOG.isDebugEnabled()) {
437       lastMillis = EnvironmentEdgeManager.currentTime();
438     }
439     String compactionName = ThroughputControlUtil.getNameForThrottling(store, "compaction");
440     long now = 0;
441     boolean hasMore;
442     ScannerContext scannerContext =
443         ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
444 
445     throughputController.start(compactionName);
446     try {
447       do {
448         hasMore = scanner.next(cells, scannerContext);
449         if (LOG.isDebugEnabled()) {
450           now = EnvironmentEdgeManager.currentTime();
451         }
452         // output to writer:
453         Cell lastCleanCell = null;
454         long lastCleanCellSeqId = 0;
455         for (Cell c : cells) {
456           if (cleanSeqId && c.getSequenceId() <= smallestReadPoint) {
457             lastCleanCell = c;
458             lastCleanCellSeqId = c.getSequenceId();
459             CellUtil.setSequenceId(c, 0);
460           } else {
461             lastCleanCell = null;
462             lastCleanCellSeqId = 0;
463           }
464           writer.append(c);
465           int len = KeyValueUtil.length(c);
466           ++progress.currentCompactedKVs;
467           progress.totalCompactedSize += len;
468           if (LOG.isDebugEnabled()) {
469             bytesWrittenProgress += len;
470           }
471           throughputController.control(compactionName, len);
472           // check periodically to see if a system stop is requested
473           if (closeCheckInterval > 0) {
474             bytesWritten += len;
475             if (bytesWritten > closeCheckInterval) {
476               bytesWritten = 0;
477               if (!store.areWritesEnabled()) {
478                 progress.cancel();
479                 return false;
480               }
481             }
482           }
483         }
484         if (lastCleanCell != null) {
485           // HBASE-16931, set back sequence id to avoid affecting scan order unexpectedly
486           CellUtil.setSequenceId(lastCleanCell, lastCleanCellSeqId);
487         }
488         // Log the progress of long running compactions every minute if
489         // logging at DEBUG level
490         if (LOG.isDebugEnabled()) {
491           if ((now - lastMillis) >= 60 * 1000) {
492             LOG.debug("Compaction progress: "
493                 + compactionName
494                 + " "
495                 + progress
496                 + String.format(", rate=%.2f kB/sec", (bytesWrittenProgress / 1024.0)
497                     / ((now - lastMillis) / 1000.0)) + ", throughputController is "
498                 + throughputController);
499             lastMillis = now;
500             bytesWrittenProgress = 0;
501           }
502         }
503         cells.clear();
504       } while (hasMore);
505     } catch (InterruptedException e) {
506       progress.cancel();
507       throw new InterruptedIOException("Interrupted while control throughput of compacting "
508           + compactionName);
509     } finally {
510       throughputController.finish(compactionName);
511     }
512     progress.complete();
513     return true;
514   }
