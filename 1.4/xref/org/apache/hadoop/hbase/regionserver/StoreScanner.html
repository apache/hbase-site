157   /** An internal constructor. */
158   protected StoreScanner(Store store, Scan scan, final ScanInfo scanInfo,
159       final NavigableSet<byte[]> columns, long readPt, boolean cacheBlocks) {
160     this.readPt = readPt;
161     this.store = store;
162     this.cacheBlocks = cacheBlocks;
163     get = scan.isGetScan();
164     int numCol = columns == null ? 0 : columns.size();
165     explicitColumnQuery = numCol > 0;
166     this.scan = scan;
167     this.columns = columns;
168     this.now = EnvironmentEdgeManager.currentTime();
169     this.oldestUnexpiredTS = scan.isRaw() ? 0L : now - scanInfo.getTtl();
170     this.minVersions = scanInfo.getMinVersions();
171 
172      // We look up row-column Bloom filters for multi-column queries as part of
173      // the seek operation. However, we also look the row-column Bloom filter
174      // for multi-row (non-"get") scans because this is not done in
175      // StoreFile.passesBloomFilter(Scan, SortedSet<byte[]>).
176      this.useRowColBloom = numCol > 1 || (!get && numCol == 1);
177 
178      this.maxRowSize = scanInfo.getTableMaxRowSize();
179      this.scanUsePread = scan.isSmall()? true: scanInfo.isUsePread();
180      this.cellsPerHeartbeatCheck = scanInfo.getCellsPerTimeoutCheck();
181      // Parallel seeking is on if the config allows and more there is more than one store file.
182      if (this.store != null && this.store.getStorefilesCount() > 1) {
183        RegionServerServices rsService = ((HStore)store).getHRegion().getRegionServerServices();
184        if (rsService != null && scanInfo.isParallelSeekEnabled()) {
185          this.parallelSeekEnabled = true;
186          this.executor = rsService.getExecutorService();
187        }
188      }
189   }

195   /**
196    * Opens a scanner across memstore, snapshot, and all StoreFiles. Assumes we
197    * are not in a compaction.
198    *
199    * @param store who we scan
200    * @param scan the spec
201    * @param columns which columns we are scanning
202    * @throws IOException
203    */
204   public StoreScanner(Store store, ScanInfo scanInfo, Scan scan, final NavigableSet<byte[]> columns,
205       long readPt)
206   throws IOException {
207     this(store, scan, scanInfo, columns, readPt, scan.getCacheBlocks());
208     if (columns != null && scan.isRaw()) {
209       throw new DoNotRetryIOException("Cannot specify any column for a raw scan");
210     }
211     matcher = UserScanQueryMatcher.create(scan, scanInfo, columns, oldestUnexpiredTS, now,
212       store.getCoprocessorHost());
213 
214     this.store.addChangedReaderObserver(this);
215 
216     try {
217       // Pass columns to try to filter out unnecessary StoreFiles.
218       List<KeyValueScanner> scanners = getScannersNoCompaction();
219 
220       // Seek all scanners to the start of the Row (or if the exact matching row
221       // key does not exist, then to the start of the next matching Row).
222       // Always check bloom filter to optimize the top row seek for delete
223       // family marker.
224       seekScanners(scanners, matcher.getStartKey(), explicitColumnQuery && lazySeekEnabledGlobally,
225         parallelSeekEnabled);
226 
227       // set storeLimit
228       this.storeLimit = scan.getMaxResultsPerColumnFamily();
229 
230       // set rowOffset
231       this.storeOffset = scan.getRowOffsetPerColumnFamily();
232       addCurrentScanners(scanners);
233       // Combine all seeked scanners with a heap
234       resetKVHeap(scanners, store.getComparator());
235     } catch (IOException e) {
236       // remove us from the HStore#changedReaderObservers here or we'll have no chance to
237       // and might cause memory leak
238       this.store.deleteChangedReaderObserver(this);
239       throw e;
240     }
241   }

277   private StoreScanner(Store store, ScanInfo scanInfo, Scan scan,
278       List<? extends KeyValueScanner> scanners, ScanType scanType, long smallestReadPoint,
279       long earliestPutTs, byte[] dropDeletesFromRow, byte[] dropDeletesToRow) throws IOException {
280     this(store, scan, scanInfo, null,
281         ((HStore) store).getHRegion().getReadpoint(IsolationLevel.READ_COMMITTED), false);
282     if (scan.hasFilter() || (scan.getStartRow() != null && scan.getStartRow().length > 0)
283         || (scan.getStopRow() != null && scan.getStopRow().length > 0)
284         || !scan.getTimeRange().isAllTime()) {
285       // use legacy query matcher since we do not consider the scan object in our code. Only used to
286       // keep compatibility for coprocessor.
287       matcher = LegacyScanQueryMatcher.create(scan, scanInfo, null, scanType, smallestReadPoint,
288         earliestPutTs, oldestUnexpiredTS, now, dropDeletesFromRow, dropDeletesToRow,
289         store.getCoprocessorHost());
290     } else {
291       matcher = CompactionScanQueryMatcher.create(scanInfo, scanType, smallestReadPoint,
292         earliestPutTs, oldestUnexpiredTS, now, dropDeletesFromRow, dropDeletesToRow,
293         store.getCoprocessorHost());
294     }
295 
296     // Filter the list of scanners using Bloom filters, time range, TTL, etc.
297     scanners = selectScannersFrom(scanners);
298 
299     // Seek all scanners to the initial key
300     seekScanners(scanners, matcher.getStartKey(), false, parallelSeekEnabled);
301     addCurrentScanners(scanners);
302     // Combine all seeked scanners with a heap
303     resetKVHeap(scanners, store.getComparator());
304   }

326   public StoreScanner(final Scan scan, ScanInfo scanInfo, ScanType scanType,
327       final NavigableSet<byte[]> columns, final List<KeyValueScanner> scanners, long earliestPutTs,
328       long readPt) throws IOException {
329     this(null, scan, scanInfo, columns, readPt, scan.getCacheBlocks());
330     if (scanType == ScanType.USER_SCAN) {
331       this.matcher = UserScanQueryMatcher.create(scan, scanInfo, columns, oldestUnexpiredTS, now,
332         null);
333     } else {
334       if (scan.hasFilter() || (scan.getStartRow() != null && scan.getStartRow().length > 0)
335           || (scan.getStopRow() != null && scan.getStopRow().length > 0)
336           || !scan.getTimeRange().isAllTime() || columns != null) {
337         // use legacy query matcher since we do not consider the scan object in our code. Only used
338         // to keep compatibility for coprocessor.
339         matcher = LegacyScanQueryMatcher.create(scan, scanInfo, columns, scanType, Long.MAX_VALUE,
340           earliestPutTs, oldestUnexpiredTS, now, null, null, store.getCoprocessorHost());
341       } else {
342         this.matcher = CompactionScanQueryMatcher.create(scanInfo, scanType, Long.MAX_VALUE,
343           earliestPutTs, oldestUnexpiredTS, now, null, null, null);
344       }
345     }
346 
347     // Seek all scanners to the initial key
348     seekScanners(scanners, matcher.getStartKey(), false, parallelSeekEnabled);
349     addCurrentScanners(scanners);
350     resetKVHeap(scanners, scanInfo.getComparator());
351   }

364   /**
365    * Seek the specified scanners with the given key
366    * @param scanners
367    * @param seekKey
368    * @param isLazy true if using lazy seek
369    * @param isParallelSeek true if using parallel seek
370    * @throws IOException
371    */
372   protected void seekScanners(List<? extends KeyValueScanner> scanners,
373       Cell seekKey, boolean isLazy, boolean isParallelSeek)
374       throws IOException {
375     // Seek all scanners to the start of the Row (or if the exact matching row
376     // key does not exist, then to the start of the next matching Row).
377     // Always check bloom filter to optimize the top row seek for delete
378     // family marker.
379     if (isLazy) {
380       for (KeyValueScanner scanner : scanners) {
381         scanner.requestSeek(seekKey, false, true);
382       }
383     } else {
384       if (!isParallelSeek) {
385         long totalScannersSoughtBytes = 0;
386         for (KeyValueScanner scanner : scanners) {
387           if (matcher.isUserScan() && totalScannersSoughtBytes >= maxRowSize) {
388             throw new RowTooBigException("Max row size allowed: " + maxRowSize
389               + ", but row is bigger than that");
390           }
391           scanner.seek(seekKey);
392           Cell c = scanner.peek();
393           if (c != null) {
394             totalScannersSoughtBytes += CellUtil.estimatedSerializedSizeOf(c);
395           }
396         }
397       } else {
398         parallelSeek(scanners, seekKey);
399       }
400     }
401   }

409   /**
410    * Filters the given list of scanners using Bloom filter, time range, and
411    * TTL.
412    */
413   protected List<KeyValueScanner> selectScannersFrom(
414       final List<? extends KeyValueScanner> allScanners) {
415     boolean memOnly;
416     boolean filesOnly;
417     if (scan instanceof InternalScan) {
418       InternalScan iscan = (InternalScan)scan;
419       memOnly = iscan.isCheckOnlyMemStore();
420       filesOnly = iscan.isCheckOnlyStoreFiles();
421     } else {
422       memOnly = false;
423       filesOnly = false;
424     }
425 
426     List<KeyValueScanner> scanners =
427         new ArrayList<KeyValueScanner>(allScanners.size());
428 
429     // We can only exclude store files based on TTL if minVersions is set to 0.
430     // Otherwise, we might have to return KVs that have technically expired.
431     long expiredTimestampCutoff = minVersions == 0 ? oldestUnexpiredTS: Long.MIN_VALUE;
432 
433     // include only those scan files which pass all filters
434     for (KeyValueScanner kvs : allScanners) {
435       boolean isFile = kvs.isFileScanner();
436       if ((!isFile && filesOnly) || (isFile && memOnly)) {
437         continue;
438       }
439 
440       if (kvs.shouldUseScanner(scan, store, expiredTimestampCutoff)) {
441         scanners.add(kvs);
442       } else {
443         kvs.close();
444       }
445     }
446     return scanners;
447   }

463   @Override
464   public void close() {
465     if (this.closing) {
466       return;
467     }
468     // Lets remove from observers as early as possible
469     // Under test, we dont have a this.store
470     if (this.store != null) {
471       this.store.deleteChangedReaderObserver(this);
472     }
473     // There is a race condition between close() and updateReaders(), during region flush. So,
474     // even though its just close, we will still acquire the flush lock, as a
475     // ConcurrentModificationException will abort the regionserver.
476     flushLock.lock();
477     try {
478       this.closing = true;
479       clearAndClose(scannersForDelayedClose);
480       clearAndClose(memStoreScannersAfterFlush);
481       // clear them at any case. In case scanner.next() was never called
482       // and there were some lease expiry we need to close all the scanners
483       // on the flushed files which are open
484       clearAndClose(flushedstoreFileScanners);
485     } finally {
486       flushLock.unlock();
487     }
488     if (this.heap != null)
489       this.heap.close();
490     this.heap = null; // CLOSED!
491     this.lastTop = null; // If both are null, we are closed.
492   }

507   /**
508    * Get the next row of values from this Store.
509    * @param outResult
510    * @param scannerContext
511    * @return true if there are more rows, false if scanner is done
512    */
513   @Override
514   public boolean next(List<Cell> outResult, ScannerContext scannerContext) throws IOException {
515     if (scannerContext == null) {
516       throw new IllegalArgumentException("Scanner context cannot be null");
517     }
518     boolean flushed = checkFlushed();
519     if (checkReseek(flushed)) {
520       return scannerContext.setScannerState(NextState.MORE_VALUES).hasMoreValues();
521     }
522 
523     // if the heap was left null, then the scanners had previously run out anyways, close and
524     // return.
525     if (this.heap == null) {
526       close();
527       return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
528     }
529 
530     Cell cell = this.heap.peek();
531     if (cell == null) {
532       close();
533       return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
534     }
535 
536     // only call setRow if the row changes; avoids confusing the query matcher
537     // if scanning intra-row
538 
539     // If no limits exists in the scope LimitScope.Between_Cells then we are sure we are changing
540     // rows. Else it is possible we are still traversing the same row so we must perform the row
541     // comparison.
542     if (!scannerContext.hasAnyLimit(LimitScope.BETWEEN_CELLS) || matcher.currentRow() == null) {
543       this.countPerRow = 0;
544       matcher.setToNewRow(cell);
545     }
546 
547     // Clear progress away unless invoker has indicated it should be kept.
548     if (!scannerContext.getKeepProgress()) scannerContext.clearProgress();
549 
550     // Only do a sanity-check if store and comparator are available.
551     KeyValue.KVComparator comparator =
552         store != null ? store.getComparator() : null;
553 
554     int count = 0;
555     long totalBytesRead = 0;
556 
557     LOOP: do {
558       // Update and check the time limit based on the configured value of cellsPerTimeoutCheck
559       if ((kvsScanned % cellsPerHeartbeatCheck == 0)) {
560         scannerContext.updateTimeProgress();
561         if (scannerContext.checkTimeLimit(LimitScope.BETWEEN_CELLS)) {
562           return scannerContext.setScannerState(NextState.TIME_LIMIT_REACHED).hasMoreValues();
563         }
564       }
565 
566       if (prevCell != cell) ++kvsScanned; // Do object compare - we set prevKV from the same heap.
567       checkScanOrder(prevCell, cell, comparator);
568       prevCell = cell;
569       scannerContext.setLastPeekedCell(cell);
570       topChanged = false;
571       ScanQueryMatcher.MatchCode qcode = matcher.match(cell);
572       switch (qcode) {
573         case INCLUDE:
574         case INCLUDE_AND_SEEK_NEXT_ROW:
575         case INCLUDE_AND_SEEK_NEXT_COL:
576 
577           Filter f = matcher.getFilter();
578           if (f != null) {
579             cell = f.transformCell(cell);
580           }
581 
582           this.countPerRow++;
583           if (storeLimit > -1 &&
584               this.countPerRow > (storeLimit + storeOffset)) {
585             // do what SEEK_NEXT_ROW does.
586             if (!matcher.moreRowsMayExistAfter(cell)) {
587               return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
588             }
589             // Setting the matcher.row = null, will mean that after the subsequent seekToNextRow()
590             // the heap.peek() will any way be in the next row. So the SQM.match(cell) need do
591             // another compareRow to say the current row is DONE
592             matcher.clearCurrentRow();
593             seekToNextRow(cell);
594             break LOOP;
595           }
596 
597           // add to results only if we have skipped #storeOffset kvs
598           // also update metric accordingly
599           if (this.countPerRow > storeOffset) {
600             outResult.add(cell);
601 
602             // Update local tracking information
603             count++;
604             totalBytesRead += CellUtil.estimatedSerializedSizeOf(cell);
605 
606             // Update the progress of the scanner context
607             scannerContext.incrementSizeProgress(CellUtil.estimatedHeapSizeOfWithoutTags(cell));
608             scannerContext.incrementBatchProgress(1);
609 
610             if (matcher.isUserScan() && totalBytesRead > maxRowSize) {
611               throw new RowTooBigException("Max row size allowed: " + maxRowSize
612                   + ", but the row is bigger than that.");
613             }
614           }
615 
616           if (qcode == ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_ROW) {
617             if (!matcher.moreRowsMayExistAfter(cell)) {
618               return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
619             }
620             // Setting the matcher.row = null, will mean that after the subsequent seekToNextRow()
621             // the heap.peek() will any way be in the next row. So the SQM.match(cell) need do
622             // another compareRow to say the current row is DONE
623             matcher.clearCurrentRow();
624             seekOrSkipToNextRow(cell);
625           } else if (qcode == ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_COL) {
626             seekOrSkipToNextColumn(cell);
627           } else {
628             this.heap.next();
629           }
630 
631           if (scannerContext.checkBatchLimit(LimitScope.BETWEEN_CELLS)) {
632             break LOOP;
633           }
634           if (scannerContext.checkSizeLimit(LimitScope.BETWEEN_CELLS)) {
635             break LOOP;
636           }
637           continue;
638 
639         case DONE:
640           // Optimization for Gets! If DONE, no more to get on this row, early exit!
641           if (this.scan.isGetScan()) {
642             // Then no more to this row... exit.
643             close();// Do all cleanup except heap.close()
644             return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
645           }
646           // We are sure that this row is done and we are in the next row.
647           // So subsequent StoresScanner.next() call need not do another compare
648           // and set the matcher.row
649           matcher.clearCurrentRow();
650           return scannerContext.setScannerState(NextState.MORE_VALUES).hasMoreValues();
651 
652         case DONE_SCAN:
653           close();
654           return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
655 
656         case SEEK_NEXT_ROW:
657           // This is just a relatively simple end of scan fix, to short-cut end
658           // us if there is an endKey in the scan.
659           if (!matcher.moreRowsMayExistAfter(cell)) {
660             return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
661           }
662           // Setting the matcher.row = null, will mean that after the subsequent seekToNextRow()
663           // the heap.peek() will any way be in the next row. So the SQM.match(cell) need do
664           // another compareRow to say the current row is DONE
665           matcher.clearCurrentRow();
666           seekOrSkipToNextRow(cell);
667           NextState stateAfterSeekNextRow = needToReturn(outResult);
668           if (stateAfterSeekNextRow != null) {
669             return scannerContext.setScannerState(stateAfterSeekNextRow).hasMoreValues();
670           }
671           break;
672 
673         case SEEK_NEXT_COL:
674           seekOrSkipToNextColumn(cell);
675           NextState stateAfterSeekNextColumn = needToReturn(outResult);
676           if (stateAfterSeekNextColumn != null) {
677             return scannerContext.setScannerState(stateAfterSeekNextColumn).hasMoreValues();
678           }
679           break;
680 
681         case SKIP:
682           this.heap.next();
683           break;
684 
685         case SEEK_NEXT_USING_HINT:
686           Cell nextKV = matcher.getNextKeyHint(cell);
687           if (nextKV != null && comparator.compare(nextKV, cell) > 0) {
688             seekAsDirection(nextKV);
689             NextState stateAfterSeekByHint = needToReturn(outResult);
690             if (stateAfterSeekByHint != null) {
691               return scannerContext.setScannerState(stateAfterSeekByHint).hasMoreValues();
692             }
693           } else {
694             heap.next();
695           }
696           break;
697 
698         default:
699           throw new RuntimeException("UNEXPECTED");
700       }
701     } while((cell = this.heap.peek()) != null);
702 
703     if (count > 0) {
704       return scannerContext.setScannerState(NextState.MORE_VALUES).hasMoreValues();
705     }
706 
707     // No more keys
708     close();
709     return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
710   }

747   /**
748    * See if we should actually SEEK or rather just SKIP to the next Cell (see HBASE-13109).
749    * ScanQueryMatcher may issue SEEK hints, such as seek to next column, next row,
750    * or seek to an arbitrary seek key. This method decides whether a seek is the most efficient
751    * _actual_ way to get us to the requested cell (SEEKs are more expensive than SKIP, SKIP,
752    * SKIP inside the current, loaded block).
753    * It does this by looking at the next indexed key of the current HFile. This key
754    * is then compared with the _SEEK_ key, where a SEEK key is an artificial 'last possible key
755    * on the row' (only in here, we avoid actually creating a SEEK key; in the compare we work with
756    * the current Cell but compare as though it were a seek key; see down in
757    * matcher.compareKeyForNextRow, etc). If the compare gets us onto the
758    * next block we *_SEEK, otherwise we just SKIP to the next requested cell.
759    *
760    * <p>Other notes:
761    * <ul>
762    * <li>Rows can straddle block boundaries</li>
763    * <li>Versions of columns can straddle block boundaries (i.e. column C1 at T1 might be in a
764    * different block than column C1 at T2)</li>
765    * <li>We want to SKIP if the chance is high that we'll find the desired Cell after a
766    * few SKIPs...</li>
767    * <li>We want to SEEK when the chance is high that we'll be able to seek
768    * past many Cells, especially if we know we need to go to the next block.</li>
769    * </ul>
770    * <p>A good proxy (best effort) to determine whether SKIP is better than SEEK is whether
771    * we'll likely end up seeking to the next block (or past the next block) to get our next column.
772    * Example:
773    * <pre>
774    * |    BLOCK 1              |     BLOCK 2                   |
775    * |  r1/c1, r1/c2, r1/c3    |    r1/c4, r1/c5, r2/c1        |
776    *                                   ^         ^
777    *                                   |         |
778    *                           Next Index Key   SEEK_NEXT_ROW (before r2/c1)
779    *
780    *
781    * |    BLOCK 1                       |     BLOCK 2                      |
782    * |  r1/c1/t5, r1/c1/t4, r1/c1/t3    |    r1/c1/t2, r1/c1/T1, r1/c2/T3  |
783    *                                            ^              ^
784    *                                            |              |
785    *                                    Next Index Key        SEEK_NEXT_COL
786    * </pre>
787    * Now imagine we want columns c1 and c3 (see first diagram above), the 'Next Index Key' of r1/c4
788    * is > r1/c3 so we should seek to get to the c1 on the next row, r2. In second case, say we only
789    * want one version of c1, after we have it, a SEEK_COL will be issued to get to c2. Looking at
790    * the 'Next Index Key', it would land us in the next block, so we should SEEK. In other scenarios
791    * where the SEEK will not land us in the next block, it is very likely better to issues a series
792    * of SKIPs.
793    * @param cell current cell
794    * @return true means skip to next row, false means not
795    */
796   @VisibleForTesting
797   protected boolean trySkipToNextRow(Cell cell) throws IOException {
798     Cell nextCell = null;
799     // used to guard against a changed next indexed key by doing a identity comparison
800     // when the identity changes we need to compare the bytes again
801     Cell previousIndexedKey = null;
802     do {
803       Cell nextIndexedKey = getNextIndexedKey();
804       if (nextIndexedKey != null && nextIndexedKey != KeyValueScanner.NO_NEXT_INDEXED_KEY &&
805           (nextIndexedKey == previousIndexedKey ||
806           matcher.compareKeyForNextRow(nextIndexedKey, cell) >= 0)) {
807         this.heap.next();
808         ++kvsScanned;
809         previousIndexedKey = nextIndexedKey;
810       } else {
811         return false;
812       }
813     } while ((nextCell = this.heap.peek()) != null && CellUtil.matchingRow(cell, nextCell));
814     return true;
815   }

863   public void updateReaders(List<StoreFile> sfs, List<KeyValueScanner> memStoreScanners) throws IOException {
864     if (CollectionUtils.isEmpty(sfs) && CollectionUtils.isEmpty(memStoreScanners)) {
865       return;
866     }
867     flushLock.lock();
868     try {
869       if (this.closing) {
870         // Lets close scanners created by caller, since close() won't notice this.
871         // memStoreScanners is immutable, so lets create a new list.
872         if (!CollectionUtils.isEmpty(memStoreScanners)) {
873           clearAndClose(new ArrayList<>(memStoreScanners));
874         }
875         return;
876       }
877       flushed = true;
878       final boolean isCompaction = false;
879       boolean usePread = get || scanUsePread;
880       // SEE HBASE-19468 where the flushed files are getting compacted even before a scanner
881       // calls next(). So its better we create scanners here rather than next() call. Ensure
882       // these scanners are properly closed() whether or not the scan is completed successfully
883       // Eagerly creating scanners so that we have the ref counting ticking on the newly created
884       // store files. In case of stream scanners this eager creation does not induce performance
885       // penalty because in scans (that uses stream scanners) the next() call is bound to happen.
886       List<KeyValueScanner> scanners = store.getScanners(sfs, cacheBlocks, get, usePread,
887         isCompaction, matcher, scan.getStartRow(), scan.getStopRow(), this.readPt, false);
888       flushedstoreFileScanners.addAll(scanners);
889       if (!CollectionUtils.isEmpty(memStoreScanners)) {
890         clearAndClose(memStoreScannersAfterFlush);
891         memStoreScannersAfterFlush.addAll(memStoreScanners);
892       }
893     } finally {
894       flushLock.unlock();
895     }

902     // All public synchronized API calls will call 'checkReseek' which will cause

919   }

922    * @param flushed indicates if there was a flush
923    * @return true if top of heap has changed (and KeyValueHeap has to try the
924    *         next KV)
925    * @throws IOException
926    */
927   protected boolean checkReseek(boolean flushed) throws IOException {
928     if (flushed && this.lastTop != null) {
929       resetScannerStack(this.lastTop);
930       if (this.heap.peek() == null
931           || store.getComparator().compareRows(this.lastTop, this.heap.peek()) != 0) {
932         LOG.info("Storescanner.peek() is changed where before = "
933             + this.lastTop.toString() + ",and after = " + this.heap.peek());
934         this.lastTop = null;
935         topChanged = true;
936         return true;
937       }
938       this.lastTop = null; // gone!
939     }
940     // else dont need to reseek
941     topChanged = false;
942     return false;

945   protected void resetScannerStack(Cell lastTopKey) throws IOException {
946     /* When we have the scan object, should we not pass it to getScanners()
947      * to get a limited set of scanners? We did so in the constructor and we
948      * could have done it now by storing the scan object from the constructor
949      */
950 
951     final boolean isCompaction = false;
952     boolean usePread = get || scanUsePread;
953     List<KeyValueScanner> scanners = null;
954     flushLock.lock();
955     try {
956       List<KeyValueScanner> allScanners =
957           new ArrayList<>(flushedstoreFileScanners.size() + memStoreScannersAfterFlush.size());
958       allScanners.addAll(flushedstoreFileScanners);
959       allScanners.addAll(memStoreScannersAfterFlush);
960       scanners = selectScannersFrom(allScanners);
961       // Clear the current set of flushed store files so that they don't get added again
962       flushedstoreFileScanners.clear();
963       memStoreScannersAfterFlush.clear();
964     } finally {
965       flushLock.unlock();
966     }
967 
968     // Seek the new scanners to the last key
969     seekScanners(scanners, lastTopKey, false, parallelSeekEnabled);
970     // remove the older memstore scanner
971     for (int i = 0; i < currentScanners.size(); i++) {
972       if (!currentScanners.get(i).isFileScanner()) {
973         scannersForDelayedClose.add(currentScanners.remove(i));
974         break;
975       }
976     }
977     // add the newly created scanners on the flushed files and the current active memstore scanner
978     addCurrentScanners(scanners);
979     // Combine all seeked scanners with a heap
980     resetKVHeap(this.currentScanners, store.getComparator());
981     // Reset the state of the Query Matcher and set to top row.
982     // Only reset and call setRow if the row changes; avoids confusing the
983     // query matcher if scanning intra-row.
984     Cell kv = heap.peek();
985     if (kv == null) {
986       kv = lastTopKey;
987     }
988     byte[] row = kv.getRowArray();
989     int offset = kv.getRowOffset();
990     short length = kv.getRowLength();
991     Cell currentRow = matcher.currentRow();
992 
993     if ((currentRow == null) || !Bytes.equals(row, offset, length, currentRow.getRowArray(),
994       currentRow.getRowOffset(), currentRow.getRowLength())) {
995       this.countPerRow = 0;
996       matcher.setToNewRow(kv);
997     }
998   }

1043   protected boolean checkFlushed() {
1044     // check the var without any lock. Suppose even if we see the old
1045     // value here still it is ok to continue because we will not be resetting
1046     // the heap but will continue with the referenced memstore's snapshot. For compactions
1047     // any way we don't need the updateReaders at all to happen as we still continue with
1048     // the older files

1052       if(!this.closing) {
1053         this.lastTop = this.peek();
1054       } else {
1055         return false;

1057       // reset the flag
1058       flushed = false;
1059       return true;
1060     }

1072   /**
1073    * Seek storefiles in parallel to optimize IO latency as much as possible
1074    * @param scanners the list {@link KeyValueScanner}s to be read from
1075    * @param kv the KeyValue on which the operation is being requested
1076    * @throws IOException
1077    */
1078   private void parallelSeek(final List<? extends KeyValueScanner>
1079       scanners, final Cell kv) throws IOException {
1080     if (scanners.isEmpty()) return;
1081     int storeFileScannerCount = scanners.size();
1082     CountDownLatch latch = new CountDownLatch(storeFileScannerCount);
1083     List<ParallelSeekHandler> handlers =
1084         new ArrayList<ParallelSeekHandler>(storeFileScannerCount);
1085     for (KeyValueScanner scanner : scanners) {
1086       if (scanner instanceof StoreFileScanner) {
1087         ParallelSeekHandler seekHandler = new ParallelSeekHandler(scanner, kv,
1088           this.readPt, latch);
1089         executor.submit(seekHandler);
1090         handlers.add(seekHandler);
1091       } else {
1092         scanner.seek(kv);
1093         latch.countDown();
1094       }
1095     }
1096 
1097     try {
1098       latch.await();
1099     } catch (InterruptedException ie) {
1100       throw (InterruptedIOException)new InterruptedIOException().initCause(ie);
1101     }
1102 
1103     for (ParallelSeekHandler handler : handlers) {
1104       if (handler.getErr() != null) {
1105         throw new IOException(handler.getErr());
1106       }
1107     }
1108   }
