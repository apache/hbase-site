31  /**
32   * Implementation of {@link MetricsTableLatencies} to track latencies for one table in a
33   * RegionServer.

35  @InterfaceAudience.Private
36  public class MetricsTableLatenciesImpl extends BaseSourceImpl implements MetricsTableLatencies {

38    private final HashMap<TableName,TableHistograms> histogramsByTable = new HashMap<>();
39  
40    @VisibleForTesting
41    public static class TableHistograms {
42      final MetricHistogram getTimeHisto;
43      final MetricHistogram incrementTimeHisto;
44      final MetricHistogram appendTimeHisto;
45      final MetricHistogram putTimeHisto;
46      final MetricHistogram putBatchTimeHisto;
47      final MetricHistogram deleteTimeHisto;
48      final MetricHistogram deleteBatchTimeHisto;
49      final MetricHistogram scanTimeHisto;
50      final MetricHistogram scanSizeHisto;
51  
52      TableHistograms(DynamicMetricsRegistry registry, TableName tn) {
53        getTimeHisto = registry.newTimeHistogram(qualifyMetricsName(tn, GET_TIME));
54        incrementTimeHisto = registry.newTimeHistogram(
55            qualifyMetricsName(tn, INCREMENT_TIME));
56        appendTimeHisto = registry.newTimeHistogram(qualifyMetricsName(tn, APPEND_TIME));
57        putTimeHisto = registry.newTimeHistogram(qualifyMetricsName(tn, PUT_TIME));
58        putBatchTimeHisto = registry.newTimeHistogram(qualifyMetricsName(tn, PUT_BATCH_TIME));
59        deleteTimeHisto = registry.newTimeHistogram(qualifyMetricsName(tn, DELETE_TIME));
60        deleteBatchTimeHisto = registry.newTimeHistogram(
61            qualifyMetricsName(tn, DELETE_BATCH_TIME));
62        scanTimeHisto = registry.newTimeHistogram(qualifyMetricsName(tn, SCAN_TIME));
63        scanSizeHisto = registry.newSizeHistogram(qualifyMetricsName(tn, SCAN_SIZE));
64      }
65  
66      public void updatePut(long time) {
67        putTimeHisto.add(time);
68      }
69  
70      public void updatePutBatch(long time) {
71        putBatchTimeHisto.add(time);
72      }
73  
74      public void updateDelete(long t) {
75        deleteTimeHisto.add(t);
76      }
77  
78      public void updateDeleteBatch(long t) {
79        deleteBatchTimeHisto.add(t);
80      }
81  
82      public void updateGet(long t) {
83        getTimeHisto.add(t);
84      }
85  
86      public void updateIncrement(long t) {
87        incrementTimeHisto.add(t);
88      }
89  
90      public void updateAppend(long t) {
91        appendTimeHisto.add(t);
92      }
93  
94      public void updateScanSize(long scanSize) {
95        scanSizeHisto.add(scanSize);
96      }
97  
98      public void updateScanTime(long t) {
99        scanTimeHisto.add(t);
100     }
101   }

103   @VisibleForTesting
104   public static String qualifyMetricsName(TableName tableName, String metric) {
105     StringBuilder sb = new StringBuilder();
106     sb.append("Namespace_").append(tableName.getNamespaceAsString());
107     sb.append("_table_").append(tableName.getQualifierAsString());
108     sb.append("_metric_").append(metric);
109     return sb.toString();
110   }
111 
112   @VisibleForTesting
113   public TableHistograms getOrCreateTableHistogram(String tableName) {
114     // TODO Java8's ConcurrentHashMap#computeIfAbsent would be stellar instead
115     final TableName tn = TableName.valueOf(tableName);
116     TableHistograms latency = histogramsByTable.get(tn);
117     if (latency == null) {
118       latency = new TableHistograms(getMetricsRegistry(), tn);
119       histogramsByTable.put(tn, latency);
120     }
121     return latency;
122   }

178   @Override
179   public void getMetrics(MetricsCollector metricsCollector, boolean all) {
180     MetricsRecordBuilder mrb = metricsCollector.addRecord(metricsName);
181     // source is registered in supers constructor, sometimes called before the whole initialization.
182     metricsRegistry.snapshot(mrb, all);
183     if (metricsAdapter != null) {
184       // snapshot MetricRegistry as well
185       metricsAdapter.snapshotAllMetrics(registry, mrb);
186     }
