78    /**
79     * Creates a reader worker for a given WAL queue. Reads WAL entries off a given queue, batches the
80     * entries, and puts them on a batch queue.
81     * @param manager replication manager
82     * @param replicationQueueInfo
83     * @param logQueue The WAL queue to read off of
84     * @param startPosition position in the first WAL to start reading from
85     * @param fs the files system to use
86     * @param conf configuration to use
87     * @param filter The filter to use while reading
88     * @param metrics replication metrics
89     */
90    public ReplicationSourceWALReaderThread(ReplicationSourceManager manager,
91        ReplicationQueueInfo replicationQueueInfo, PriorityBlockingQueue<Path> logQueue,
92        long startPosition,
93        FileSystem fs, Configuration conf, WALEntryFilter filter, MetricsSource metrics) {
94      this.replicationQueueInfo = replicationQueueInfo;
95      this.logQueue = logQueue;
96      this.currentPosition = startPosition;
97      this.fs = fs;
98      this.conf = conf;
99      this.filter = filter;
100     this.replicationBatchSizeCapacity =
101         this.conf.getLong("replication.source.size.capacity", 1024 * 1024 * 64);
102     this.replicationBatchCountCapacity = this.conf.getInt("replication.source.nb.capacity", 25000);
103     // memory used will be batchSizeCapacity * (nb.batches + 1)
104     // the +1 is for the current thread reading before placing onto the queue
105     int batchCount = conf.getInt("replication.source.nb.batches", 1);
106     this.replicationSourceManager = manager;
107     this.sleepForRetries =
108         this.conf.getLong("replication.source.sleepforretries", 1000);    // 1 second
109     this.maxRetriesMultiplier =
110         this.conf.getInt("replication.source.maxretriesmultiplier", 300); // 5 minutes @ 1 sec per
111     this.metrics = metrics;
112     this.entryBatchQueue = new LinkedBlockingQueue<>(batchCount);
113     LOG.info("peerClusterZnode=" + replicationQueueInfo.getPeerClusterZnode()
114         + ", ReplicationSourceWALReaderThread : " + replicationQueueInfo.getPeerId()
115         + " inited, replicationBatchSizeCapacity=" + replicationBatchSizeCapacity
116         + ", replicationBatchCountCapacity=" + replicationBatchCountCapacity
117         + ", replicationBatchQueueCapacity=" + batchCount);
118   }

120   @Override
121   public void run() {
122     int sleepMultiplier = 1;
123     while (isReaderRunning()) { // we only loop back here if something fatal happened to our stream
124       try (WALEntryStream entryStream =
125           new WALEntryStream(logQueue, fs, conf, currentPosition, metrics)) {
126         while (isReaderRunning()) { // loop here to keep reusing stream while we can
127           WALEntryBatch batch = null;
128           while (entryStream.hasNext()) {
129             if (batch == null) {
130               batch = new WALEntryBatch(replicationBatchCountCapacity, entryStream.getCurrentPath());
131             }
132             Entry entry = entryStream.next();
133             entry = filterEntry(entry);
134             if (entry != null) {
135               WALEdit edit = entry.getEdit();
136               if (edit != null && !edit.isEmpty()) {
137                 long entrySize = getEntrySize(entry);
138                 batch.addEntry(entry);
139                 replicationSourceManager.setPendingShipment(true);
140                 updateBatchStats(batch, entry, entryStream.getPosition(), entrySize);
141                 // Stop if too many entries or too big
142                 if (batch.getHeapSize() >= replicationBatchSizeCapacity
143                     || batch.getNbEntries() >= replicationBatchCountCapacity) {
144                   break;
145                 }
146               }
147             } else {
148               replicationSourceManager.logPositionAndCleanOldLogs(entryStream.getCurrentPath(),
149                 this.replicationQueueInfo.getPeerClusterZnode(),
150                 entryStream.getPosition(),
151                 this.replicationQueueInfo.isQueueRecovered(), false);
152             }
153           }
154           if (batch != null && (!batch.getLastSeqIds().isEmpty() || batch.getNbEntries() > 0)) {
155             if (LOG.isTraceEnabled()) {
156               LOG.trace(String.format("Read %s WAL entries eligible for replication",
157                 batch.getNbEntries()));
158             }
159             entryBatchQueue.put(batch);
160             sleepMultiplier = 1;
161           } else { // got no entries and didn't advance position in WAL
162             LOG.trace("Didn't read any new entries from WAL");
163             if (replicationQueueInfo.isQueueRecovered()) {
164               // we're done with queue recovery, shut ourself down
165               setReaderRunning(false);
166               // shuts down shipper thread immediately
167               entryBatchQueue.put(batch != null ? batch
168                   : new WALEntryBatch(replicationBatchCountCapacity, entryStream.getCurrentPath()));
169             } else {
170               Thread.sleep(sleepForRetries);
171             }
172           }
173           currentPosition = entryStream.getPosition();
174           entryStream.reset(); // reuse stream
175         }
176       } catch (IOException | WALEntryStreamRuntimeException e) { // stream related
177         if (sleepMultiplier < maxRetriesMultiplier) {
178           LOG.debug("Failed to read stream of replication entries: " + e);
179           sleepMultiplier++;
180         } else {
181           LOG.error("Failed to read stream of replication entries", e);
182           handleEofException(e);
183         }
184         Threads.sleep(sleepForRetries * sleepMultiplier);
185       } catch (InterruptedException e) {
186         LOG.trace("Interrupted while sleeping between WAL reads");
187         Thread.currentThread().interrupt();
188       }
189     }
190   }

192   // if we get an EOF due to a zero-length log, and there are other logs in queue
193   // (highly likely we've closed the current log), we've hit the max retries, and autorecovery is
194   // enabled, then dump the log
195   private void handleEofException(Exception e) {
196     if (e.getCause() instanceof EOFException && logQueue.size() > 1
197         && conf.getBoolean("replication.source.eof.autorecovery", false)) {
198       try {
199         if (fs.getFileStatus(logQueue.peek()).getLen() == 0) {
200           LOG.warn("Forcing removal of 0 length log in queue: " + logQueue.peek());
201           logQueue.remove();
202           currentPosition = 0;
203         }
204       } catch (IOException ioe) {
205         LOG.warn("Couldn't get file length information about log " + logQueue.peek());
206       }
207     }
208   }

210   public Path getCurrentPath() {
211     // if we've read some WAL entries, get the Path we read from
212     WALEntryBatch batchQueueHead = entryBatchQueue.peek();
213     if (batchQueueHead != null) {
214       return batchQueueHead.lastWalPath;
215     }
216     // otherwise, we must be currently reading from the head of the log queue
217     return logQueue.peek();
218   }

220   private Entry filterEntry(Entry entry) {
221     Entry filtered = filter.filter(entry);
222     if (entry != null && filtered == null) {
223       metrics.incrLogEditsFiltered();
224     }
225     return filtered;

238   private long getEntrySize(Entry entry) {
239     WALEdit edit = entry.getEdit();
240     return edit.heapSize() + calculateTotalSizeOfStoreFiles(edit);
241   }

243   private void updateBatchStats(WALEntryBatch batch, Entry entry, long entryPosition, long entrySize) {
244     WALEdit edit = entry.getEdit();
245     if (edit != null && !edit.isEmpty()) {
246       batch.incrementHeapSize(entrySize);
247       Pair<Integer, Integer> nbRowsAndHFiles = countDistinctRowKeysAndHFiles(edit);
248       batch.incrementNbRowKeys(nbRowsAndHFiles.getFirst());
249       batch.incrementNbHFiles(nbRowsAndHFiles.getSecond());
250     }
251     batch.lastWalPosition = entryPosition;
252   }

254   /**
255    * Count the number of different row keys in the given edit because of mini-batching. We assume
256    * that there's at least one Cell in the WALEdit.
257    * @param edit edit to count row keys from
258    * @return number of different row keys and HFiles
259    */
260   private Pair<Integer, Integer> countDistinctRowKeysAndHFiles(WALEdit edit) {
261     List<Cell> cells = edit.getCells();
262     int distinctRowKeys = 1;
263     int totalHFileEntries = 0;
264     Cell lastCell = cells.get(0);
265 
266     int totalCells = edit.size();
267     for (int i = 0; i < totalCells; i++) {
268       // Count HFiles to be replicated
269       if (CellUtil.matchingQualifier(cells.get(i), WALEdit.BULK_LOAD)) {
270         try {
271           BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cells.get(i));
272           List<StoreDescriptor> stores = bld.getStoresList();
273           int totalStores = stores.size();
274           for (int j = 0; j < totalStores; j++) {
275             totalHFileEntries += stores.get(j).getStoreFileList().size();
276           }
277         } catch (IOException e) {
278           LOG.error("Failed to deserialize bulk load entry from wal edit. "
279               + "Then its hfiles count will not be added into metric.");
280         }
281       }
282 
283       if (!CellUtil.matchingRow(cells.get(i), lastCell)) {
284         distinctRowKeys++;
285       }
286       lastCell = cells.get(i);
287     }
288 
289     Pair<Integer, Integer> result = new Pair<>(distinctRowKeys, totalHFileEntries);
290     return result;
291   }

293   /**
294    * Calculate the total size of all the store files
295    * @param edit edit to count row keys from
296    * @return the total size of the store files
297    */
298   private int calculateTotalSizeOfStoreFiles(WALEdit edit) {
299     List<Cell> cells = edit.getCells();
300     int totalStoreFilesSize = 0;
301 
302     int totalCells = edit.size();
303     for (int i = 0; i < totalCells; i++) {
304       if (CellUtil.matchingQualifier(cells.get(i), WALEdit.BULK_LOAD)) {
305         try {
306           BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cells.get(i));
307           List<StoreDescriptor> stores = bld.getStoresList();
308           int totalStores = stores.size();
309           for (int j = 0; j < totalStores; j++) {
310             totalStoreFilesSize += stores.get(j).getStoreFileSizeBytes();
311           }
312         } catch (IOException e) {
313           LOG.error("Failed to deserialize bulk load entry from wal edit. "
314               + "Size of HFiles part of cell will not be considered in replication "
315               + "request size calculation.",
316             e);
317         }
318       }
319     }
320     return totalStoreFilesSize;
321   }

337   /**
338    * Holds a batch of WAL entries to replicate, along with some statistics
339    *
340    */
341   static class WALEntryBatch {
342     private List<Entry> walEntries;
343     // last WAL that was read
344     private Path lastWalPath;
345     // position in WAL of last entry in this batch
346     private long lastWalPosition = 0;
347     // number of distinct row keys in this batch
348     private int nbRowKeys = 0;
349     // number of HFiles
350     private int nbHFiles = 0;
351     // heap size of data we need to replicate
352     private long heapSize = 0;
353     // save the last sequenceid for each region if the table has serial-replication scope
354     private Map<String, Long> lastSeqIds = new HashMap<>();
355 
356     /**
357      * @param walEntries
358      * @param lastWalPath Path of the WAL the last entry in this batch was read from
359      * @param lastWalPosition Position in the WAL the last entry in this batch was read from
360      */
361     private WALEntryBatch(int maxNbEntries, Path lastWalPath) {
362       this.walEntries = new ArrayList<>(maxNbEntries);
363       this.lastWalPath = lastWalPath;
364     }
365 
366     public void addEntry(Entry entry) {
367       walEntries.add(entry);
368     }
369 
370     /**
371      * @return the WAL Entries.
372      */
373     public List<Entry> getWalEntries() {
374       return walEntries;
375     }
376 
377     /**
378      * @return the path of the last WAL that was read.
379      */
380     public Path getLastWalPath() {
381       return lastWalPath;
382     }
383 
384     /**
385      * @return the position in the last WAL that was read.
386      */
387     public long getLastWalPosition() {
388       return lastWalPosition;
389     }
390 
391     public int getNbEntries() {
392       return walEntries.size();
393     }
394 
395     /**
396      * @return the number of distinct row keys in this batch
397      */
398     public int getNbRowKeys() {
399       return nbRowKeys;
400     }
401 
402     /**
403      * @return the number of HFiles in this batch
404      */
405     public int getNbHFiles() {
406       return nbHFiles;
407     }
408 
409     /**
410      * @return total number of operations in this batch
411      */
412     public int getNbOperations() {
413       return getNbRowKeys() + getNbHFiles();
414     }
415 
416     /**
417      * @return the heap size of this batch
418      */
419     public long getHeapSize() {
420       return heapSize;
421     }
422 
423     /**
424      * @return the last sequenceid for each region if the table has serial-replication scope
425      */
426     public Map<String, Long> getLastSeqIds() {
427       return lastSeqIds;
428     }

430     private void incrementNbRowKeys(int increment) {
431       nbRowKeys += increment;
432     }
433 
434     private void incrementNbHFiles(int increment) {
435       nbHFiles += increment;
436     }
437 
438     private void incrementHeapSize(long increment) {
439       heapSize += increment;
440     }

442     private void setLastPosition(String region, Long sequenceId) {
443       getLastSeqIds().put(region, sequenceId);
444     }
