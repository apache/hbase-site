63  /**
64   * <p>
65   * This class is responsible for replicating the edits coming
66   * from another cluster.
67   * </p><p>
68   * This replication process is currently waiting for the edits to be applied
69   * before the method can return. This means that the replication of edits
70   * is synchronized (after reading from WALs in ReplicationSource) and that a
71   * single region server cannot receive edits from two sources at the same time
72   * </p><p>
73   * This class uses the native HBase client in order to replicate entries.
74   * </p>
75   *
76   * TODO make this class more like ReplicationSource wrt log handling
77   */
78  @InterfaceAudience.Private
79  public class ReplicationSink {

134   /**
135    * Replicate this array of entries directly into the local cluster using the native client. Only
136    * operates against raw protobuf type saving on a conversion from pb to pojo.
137    * @param entries
138    * @param cells
139    * @param replicationClusterId Id which will uniquely identify source cluster FS client
140    *          configurations in the replication configuration directory
141    * @param sourceBaseNamespaceDirPath Path that point to the source cluster base namespace
142    *          directory
143    * @param sourceHFileArchiveDirPath Path that point to the source cluster hfile archive directory
144    * @throws IOException If failed to replicate the data
145    */
146   public void replicateEntries(List<WALEntry> entries, final CellScanner cells,
147       String replicationClusterId, String sourceBaseNamespaceDirPath,
148       String sourceHFileArchiveDirPath) throws IOException {
149     if (entries.isEmpty()) return;
150     if (cells == null) throw new NullPointerException("TODO: Add handling of null CellScanner");
151     // Very simple optimization where we batch sequences of rows going
152     // to the same table.
153     try {
154       long totalReplicated = 0;
155       // Map of table => list of Rows, grouped by cluster id, we only want to flushCommits once per
156       // invocation of this method per table and cluster id.
157       Map<TableName, Map<List<UUID>, List<Row>>> rowMap =
158           new TreeMap<TableName, Map<List<UUID>, List<Row>>>();
159 
160       Map<List<String>, Map<String, List<Pair<byte[], List<String>>>>> bulkLoadsPerClusters = null;
161       for (WALEntry entry : entries) {
162         TableName table =
163             TableName.valueOf(entry.getKey().getTableName().toByteArray());
164         Cell previousCell = null;
165         Mutation m = null;
166         int count = entry.getAssociatedCellCount();
167         for (int i = 0; i < count; i++) {
168           // Throw index out of bounds if our cell count is off
169           if (!cells.advance()) {
170             throw new ArrayIndexOutOfBoundsException("Expected=" + count + ", index=" + i);
171           }
172           Cell cell = cells.current();
173           // Handle bulk load hfiles replication
174           if (CellUtil.matchingQualifier(cell, WALEdit.BULK_LOAD)) {
175             BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cell);
176             if(bulkLoadsPerClusters == null) {
177               bulkLoadsPerClusters = new HashMap<>();
178             }
179             // Map of table name Vs list of pair of family and list of
180             // hfile paths from its namespace
181             Map<String, List<Pair<byte[], List<String>>>> bulkLoadHFileMap =
182               bulkLoadsPerClusters.get(bld.getClusterIdsList());
183             if (bulkLoadHFileMap == null) {
184               bulkLoadHFileMap = new HashMap<>();
185               bulkLoadsPerClusters.put(bld.getClusterIdsList(), bulkLoadHFileMap);
186             }
187             buildBulkLoadHFileMap(bulkLoadHFileMap, table, bld);
188           } else {
189             // Handle wal replication
190             if (isNewRowOrType(previousCell, cell)) {
191               // Create new mutation
192               m =
193                   CellUtil.isDelete(cell) ? new Delete(cell.getRowArray(), cell.getRowOffset(),
194                       cell.getRowLength()) : new Put(cell.getRowArray(), cell.getRowOffset(),
195                       cell.getRowLength());
196               List<UUID> clusterIds = new ArrayList<UUID>();
197               for (HBaseProtos.UUID clusterId : entry.getKey().getClusterIdsList()) {
198                 clusterIds.add(toUUID(clusterId));
199               }
200               m.setClusterIds(clusterIds);
201               addToHashMultiMap(rowMap, table, clusterIds, m);
202             }
203             if (CellUtil.isDelete(cell)) {
204               ((Delete) m).addDeleteMarker(cell);
205             } else {
206               ((Put) m).add(cell);
207             }
208             previousCell = cell;
209           }
210         }
211         totalReplicated++;
212       }
213 
214       // TODO Replicating mutations and bulk loaded data can be made parallel
215       if (!rowMap.isEmpty()) {
216         LOG.debug("Started replicating mutations.");
217         for (Entry<TableName, Map<List<UUID>, List<Row>>> entry : rowMap.entrySet()) {
218           batch(entry.getKey(), entry.getValue().values());
219         }
220         LOG.debug("Finished replicating mutations.");
221       }
222 
223       if(bulkLoadsPerClusters != null) {
224         for (Entry<List<String>, Map<String, List<Pair<byte[],
225           List<String>>>>> entry : bulkLoadsPerClusters.entrySet()) {
226           Map<String, List<Pair<byte[], List<String>>>> bulkLoadHFileMap = entry.getValue();
227           if (bulkLoadHFileMap != null && !bulkLoadHFileMap.isEmpty()) {
228             if(LOG.isDebugEnabled()) {
229               LOG.debug("Started replicating bulk loaded data from cluster ids: " +
230                 entry.getKey().toString());
231             }
232             HFileReplicator hFileReplicator =
233               new HFileReplicator(this.provider.getConf(this.conf, replicationClusterId),
234                 sourceBaseNamespaceDirPath, sourceHFileArchiveDirPath, bulkLoadHFileMap, conf,
235                 getConnection(), entry.getKey());
236             hFileReplicator.replicate();
237             if(LOG.isDebugEnabled()) {
238               LOG.debug("Finished replicating bulk loaded data from cluster id: " +
239                 entry.getKey().toString());
240             }
241           }
242         }
243       }
244 
245       int size = entries.size();
246       this.metrics.setAgeOfLastAppliedOp(entries.get(size - 1).getKey().getWriteTime());
247       this.metrics.applyBatch(size + hfilesReplicated, hfilesReplicated);
248       this.totalReplicatedEdits.addAndGet(totalReplicated);
249     } catch (IOException ex) {
250       LOG.error("Unable to accept edit because:", ex);
251       throw ex;
252     }
253   }

255   private void buildBulkLoadHFileMap(
256       final Map<String, List<Pair<byte[], List<String>>>> bulkLoadHFileMap, TableName table,
257       BulkLoadDescriptor bld) throws IOException {
258     List<StoreDescriptor> storesList = bld.getStoresList();
259     int storesSize = storesList.size();
260     for (int j = 0; j < storesSize; j++) {
261       StoreDescriptor storeDescriptor = storesList.get(j);
262       List<String> storeFileList = storeDescriptor.getStoreFileList();
263       int storeFilesSize = storeFileList.size();
264       hfilesReplicated += storeFilesSize;
265       for (int k = 0; k < storeFilesSize; k++) {
266         byte[] family = storeDescriptor.getFamilyName().toByteArray();
267 
268         // Build hfile relative path from its namespace
269         String pathToHfileFromNS = getHFilePath(table, bld, storeFileList.get(k), family);
270 
271         String tableName = table.getNameWithNamespaceInclAsString();
272         if (bulkLoadHFileMap.containsKey(tableName)) {
273           List<Pair<byte[], List<String>>> familyHFilePathsList = bulkLoadHFileMap.get(tableName);
274           boolean foundFamily = false;
275           for (int i = 0; i < familyHFilePathsList.size(); i++) {
276             Pair<byte[], List<String>> familyHFilePathsPair = familyHFilePathsList.get(i);
277             if (Bytes.equals(familyHFilePathsPair.getFirst(), family)) {
278               // Found family already present, just add the path to the existing list
279               familyHFilePathsPair.getSecond().add(pathToHfileFromNS);
280               foundFamily = true;
281               break;
282             }
283           }
284           if (!foundFamily) {
285             // Family not found, add this family and its hfile paths pair to the list
286             addFamilyAndItsHFilePathToTableInMap(family, pathToHfileFromNS, familyHFilePathsList);
287           }
288         } else {
289           // Add this table entry into the map
290           addNewTableEntryInMap(bulkLoadHFileMap, family, pathToHfileFromNS, tableName);
291         }
292       }
293     }
294   }

296   private void addFamilyAndItsHFilePathToTableInMap(byte[] family, String pathToHfileFromNS,
297       List<Pair<byte[], List<String>>> familyHFilePathsList) {
298     List<String> hfilePaths = new ArrayList<String>();
299     hfilePaths.add(pathToHfileFromNS);
300     familyHFilePathsList.add(new Pair<byte[], List<String>>(family, hfilePaths));
301   }

303   private void addNewTableEntryInMap(
304       final Map<String, List<Pair<byte[], List<String>>>> bulkLoadHFileMap, byte[] family,
305       String pathToHfileFromNS, String tableName) {
306     List<String> hfilePaths = new ArrayList<String>();
307     hfilePaths.add(pathToHfileFromNS);
308     Pair<byte[], List<String>> newFamilyHFilePathsPair =
309         new Pair<byte[], List<String>>(family, hfilePaths);
310     List<Pair<byte[], List<String>>> newFamilyHFilePathsList =
311         new ArrayList<Pair<byte[], List<String>>>();
312     newFamilyHFilePathsList.add(newFamilyHFilePathsPair);
313     bulkLoadHFileMap.put(tableName, newFamilyHFilePathsList);
314   }

316   private String getHFilePath(TableName table, BulkLoadDescriptor bld, String storeFile,
317       byte[] family) {
318     return new StringBuilder(100).append(table.getNamespaceAsString()).append(Path.SEPARATOR)
319         .append(table.getQualifierAsString()).append(Path.SEPARATOR)
320         .append(Bytes.toString(bld.getEncodedRegionName().toByteArray())).append(Path.SEPARATOR)
321         .append(Bytes.toString(family)).append(Path.SEPARATOR).append(storeFile).toString();
322   }

324   /**
325    * @param previousCell
326    * @param cell
327    * @return True if we have crossed over onto a new row or type
328    */
329   private boolean isNewRowOrType(final Cell previousCell, final Cell cell) {
330     return previousCell == null || previousCell.getTypeByte() != cell.getTypeByte() ||
331         !CellUtil.matchingRow(previousCell, cell);
332   }

334   private java.util.UUID toUUID(final HBaseProtos.UUID uuid) {
335     return new java.util.UUID(uuid.getMostSigBits(), uuid.getLeastSigBits());

338   /**
339    * Simple helper to a map from key to (a list of) values
340    * TODO: Make a general utility method
341    * @param map
342    * @param key1
343    * @param key2
344    * @param value
345    * @return the list of values corresponding to key1 and key2
346    */
347   private <K1, K2, V> List<V> addToHashMultiMap(Map<K1, Map<K2,List<V>>> map, K1 key1, K2 key2, V value) {
348     Map<K2,List<V>> innerMap = map.get(key1);
349     if (innerMap == null) {
350       innerMap = new HashMap<K2, List<V>>();
351       map.put(key1, innerMap);
352     }
353     List<V> values = innerMap.get(key2);
354     if (values == null) {
355       values = new ArrayList<V>();
356       innerMap.put(key2, values);
357     }
358     values.add(value);
359     return values;
360   }

362   /**
363    * stop the thread pool executor. It is called when the regionserver is stopped.
364    */
365   public void stopReplicationSinkServices() {
366     try {
367       if (this.sharedHtableCon != null) {
368         synchronized (sharedHtableConLock) {
369           if (this.sharedHtableCon != null) {
370             this.sharedHtableCon.close();
371             this.sharedHtableCon = null;
372           }
373         }
374       }
375     } catch (IOException e) {
376       LOG.warn("IOException while closing the connection", e); // ignoring as we are closing.
377     }
378   }

381   /**
382    * Do the changes and handle the pool
383    * @param tableName table to insert into
384    * @param allRows list of actions
385    * @throws IOException
386    */
387   protected void batch(TableName tableName, Collection<List<Row>> allRows) throws IOException {
388     if (allRows.isEmpty()) {
389       return;
390     }
391     Table table = null;
392     try {
393       Connection connection = getConnection();
394       table = connection.getTable(tableName);
395       for (List<Row> rows : allRows) {
396         table.batch(rows);
397       }
398     } catch (RetriesExhaustedWithDetailsException rewde) {
399       for (Throwable ex : rewde.getCauses()) {
400         if (ex instanceof TableNotFoundException) {
401           throw new TableNotFoundException("'" + tableName + "'");
402         }
403       }
404       throw rewde;
405     } catch (InterruptedException ix) {
406       throw (InterruptedIOException) new InterruptedIOException().initCause(ix);
407     } finally {
408       if (table != null) {
409         table.close();
410       }
411     }
412   }
413 
414   private Connection getConnection() throws IOException {
415     // See https://en.wikipedia.org/wiki/Double-checked_locking
416     Connection connection = sharedHtableCon;
417     if (connection == null) {
418       synchronized (sharedHtableConLock) {
419         connection = sharedHtableCon;
420         if (connection == null) {
421           connection = sharedHtableCon = ConnectionFactory.createConnection(conf);
422         }
423       }
424     }
425     return connection;
426   }
