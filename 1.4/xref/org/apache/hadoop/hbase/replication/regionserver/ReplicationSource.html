75  /**
76   * Class that handles the source of a replication stream.
77   * Currently does not handle more than 1 slave
78   * For each slave cluster it selects a random number of peers
79   * using a replication ratio. For example, if replication ration = 0.1
80   * and slave cluster has 100 region servers, 10 will be selected.
81   * <p>
82   * A stream is considered down when we cannot contact a region server on the
83   * peer cluster for more than 55 seconds by default.
84   * </p>
85   *
86   */
87  @InterfaceAudience.Private
88  public class ReplicationSource extends Thread implements ReplicationSourceInterface {

204   @Override
205   public void enqueueLog(Path log) {
206     String logPrefix = DefaultWALProvider.getWALPrefixFromWALName(log.getName());
207     PriorityBlockingQueue<Path> queue = queues.get(logPrefix);
208     if (queue == null) {
209       queue = new PriorityBlockingQueue<Path>(queueSizePerGroup, new LogsComparator());
210       queues.put(logPrefix, queue);
211       if (this.sourceRunning) {
212         // new wal group observed after source startup, start a new worker thread to track it
213         // notice: it's possible that log enqueued when this.running is set but worker thread
214         // still not launched, so it's necessary to check workerThreads before start the worker
215         final ReplicationSourceShipperThread worker =
216             new ReplicationSourceShipperThread(logPrefix, queue, replicationQueueInfo, this);
217         ReplicationSourceShipperThread extant = workerThreads.putIfAbsent(logPrefix, worker);
218         if (extant != null) {
219           LOG.debug("Someone has beat us to start a worker thread for wal group " + logPrefix);
220         } else {
221           LOG.debug("Starting up worker for wal group " + logPrefix);
222           worker.startup();
223         }
224       }
225     }
226     queue.put(log);
227     this.metrics.incrSizeOfLogQueue();
228     // This will log a warning for each new log that gets created above the warn threshold
229     int queueSize = queue.size();
230     if (queueSize > this.logQueueWarnThreshold) {
231       LOG.warn("WAL group " + logPrefix + " queue size: " + queueSize
232           + " exceeds value of replication.source.log.queue.warn: " + logQueueWarnThreshold);
233     }
234   }

236   @Override
237   public void addHFileRefs(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
238       throws ReplicationException {
239     String peerId = peerClusterZnode;
240     if (peerId.contains("-")) {
241       // peerClusterZnode will be in the form peerId + "-" + rsZNode.
242       // A peerId will not have "-" in its name, see HBASE-11394
243       peerId = peerClusterZnode.split("-")[0];
244     }
245     Map<TableName, List<String>> tableCFMap = replicationPeers.getPeer(peerId).getTableCFs();
246     if (tableCFMap != null) {
247       List<String> tableCfs = tableCFMap.get(tableName);
248       if (tableCFMap.containsKey(tableName)
249           && (tableCfs == null || tableCfs.contains(Bytes.toString(family)))) {
250         this.replicationQueues.addHFileRefs(peerId, pairs);
251         metrics.incrSizeOfHFileRefsQueue(pairs.size());
252       } else {
253         LOG.debug("HFiles will not be replicated belonging to the table " + tableName + " family "
254             + Bytes.toString(family) + " to peer id " + peerId);
255       }
256     } else {
257       // user has explicitly not defined any table cfs for replication, means replicate all the
258       // data
259       this.replicationQueues.addHFileRefs(peerId, pairs);
260       metrics.incrSizeOfHFileRefsQueue(pairs.size());
261     }
262   }

264   private void uninitialize() {
265     LOG.debug("Source exiting " + this.peerId);
266     metrics.clear();
267     if (replicationEndpoint.state() == Service.State.STARTING
268         || replicationEndpoint.state() == Service.State.RUNNING) {
269       replicationEndpoint.stopAndWait();
270     }

273   @Override
274   public void run() {
275     // mark we are running now
276     this.sourceRunning = true;
277     try {
278       // start the endpoint, connect to the cluster
279       Service.State state = replicationEndpoint.start().get();
280       if (state != Service.State.RUNNING) {
281         LOG.warn("ReplicationEndpoint was not started. Exiting");
282         uninitialize();
283         return;
284       }
285     } catch (Exception ex) {
286       LOG.warn("Error starting ReplicationEndpoint, exiting", ex);
287       throw new RuntimeException(ex);
288     }
289 
290     // get the WALEntryFilter from ReplicationEndpoint and add it to default filters
291     ArrayList<WALEntryFilter> filters = Lists.newArrayList(
292       (WALEntryFilter)new SystemTableWALEntryFilter());
293     WALEntryFilter filterFromEndpoint = this.replicationEndpoint.getWALEntryfilter();
294     if (filterFromEndpoint != null) {
295       filters.add(filterFromEndpoint);
296     }
297     this.walEntryFilter = new ChainWALEntryFilter(filters);
298 
299     int sleepMultiplier = 1;
300     // delay this until we are in an asynchronous thread
301     while (this.isSourceActive() && this.peerClusterId == null) {
302       this.peerClusterId = replicationEndpoint.getPeerUUID();
303       if (this.isSourceActive() && this.peerClusterId == null) {
304         if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
305           sleepMultiplier++;
306         }
307       }
308     }
309 
310     if (!this.isSourceActive()) {
311       return;
312     }
313 
314     // In rare case, zookeeper setting may be messed up. That leads to the incorrect
315     // peerClusterId value, which is the same as the source clusterId
316     if (clusterId.equals(peerClusterId) && !replicationEndpoint.canReplicateToSameCluster()) {
317       this.terminate("ClusterId " + clusterId + " is replicating to itself: peerClusterId "
318           + peerClusterId + " which is not allowed by ReplicationEndpoint:"
319           + replicationEndpoint.getClass().getName(), null, false);
320       this.manager.closeQueue(this);
321       return;
322     }
323     LOG.info("Replicating " + clusterId + " -> " + peerClusterId);
324     // start workers
325     for (Map.Entry<String, PriorityBlockingQueue<Path>> entry : queues.entrySet()) {
326       String walGroupId = entry.getKey();
327       PriorityBlockingQueue<Path> queue = entry.getValue();
328       final ReplicationSourceShipperThread worker =
329           new ReplicationSourceShipperThread(walGroupId, queue, replicationQueueInfo, this);
330       ReplicationSourceShipperThread extant = workerThreads.putIfAbsent(walGroupId, worker);
331       if (extant != null) {
332         LOG.debug("Someone has beat us to start a worker thread for wal group " + walGroupId);
333       } else {
334         LOG.debug("Starting up worker for wal group " + walGroupId);
335         worker.startup();
336       }
337     }
338   }

340   /**
341    * Do the sleeping logic
342    * @param msg Why we sleep
343    * @param sleepMultiplier by how many times the default sleeping time is augmented
344    * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
345    */
346   protected boolean sleepForRetries(String msg, int sleepMultiplier) {
347     try {
348       if (LOG.isTraceEnabled()) {
349         LOG.trace(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
350       }
351       Thread.sleep(this.sleepForRetries * sleepMultiplier);
352     } catch (InterruptedException e) {
353       LOG.debug("Interrupted while sleeping between retries");
354       Thread.currentThread().interrupt();
355     }
356     return sleepMultiplier < maxRetriesMultiplier;
357   }

391   public void terminate(String reason, Exception cause, boolean join) {
392     if (cause == null) {
393       LOG.info("Closing source "
394           + this.peerClusterZnode + " because: " + reason);
395 
396     } else {
397       LOG.error("Closing source " + this.peerClusterZnode
398           + " because an error occurred: " + reason, cause);
399     }
400     this.sourceRunning = false;
401     Collection<ReplicationSourceShipperThread> workers = workerThreads.values();
402     for (ReplicationSourceShipperThread worker : workers) {
403       worker.setWorkerState(WorkerState.STOPPED);
404       worker.entryReader.interrupt();
405       worker.interrupt();
406     }
407     ListenableFuture<Service.State> future = null;
408     if (this.replicationEndpoint != null) {
409       future = this.replicationEndpoint.stop();
410     }
411     if (join) {
412       for (ReplicationSourceShipperThread worker : workers) {
413         Threads.shutdown(worker, this.sleepForRetries);
414         LOG.info("ReplicationSourceWorker " + worker.getName() + " terminated");
415       }
416       if (future != null) {
417         try {
418           future.get(sleepForRetries * maxRetriesMultiplier, TimeUnit.MILLISECONDS);
419         } catch (Exception e) {
420           LOG.warn("Got exception while waiting for endpoint to shutdown for replication source :"
421               + this.peerClusterZnode,
422             e);
423         }
424       }
425     }
426   }

461     /**

465      * @return start time
466      */
467     private static long getTS(Path p) {
468       int tsIndex = p.getName().lastIndexOf('.') + 1;
469       return Long.parseLong(p.getName().substring(tsIndex));

475     StringBuilder sb = new StringBuilder();

482       Path currentPath = worker.getCurrentPath();
483       sb.append("walGroup [").append(walGroupId).append("]: ");
484       if (currentPath != null) {
485         sb.append("currently replicating from: ").append(currentPath).append(" at position: ")

487       } else {
488         sb.append("no replication ongoing, waiting for new log");
489       }

506   // This thread reads entries from a queue and ships them.
507   // Entries are placed onto the queue by ReplicationSourceWALReaderThread
508   public class ReplicationSourceShipperThread extends Thread {
509     ReplicationSourceInterface source;
510     String walGroupId;
511     PriorityBlockingQueue<Path> queue;
512     ReplicationQueueInfo replicationQueueInfo;
513     // Last position in the log that we sent to ZooKeeper
514     private long lastLoggedPosition = -1;
515     // Path of the current log
516     private volatile Path currentPath;
517     // Current state of the worker thread
518     private WorkerState state;
519     ReplicationSourceWALReaderThread entryReader;
520 
521     public ReplicationSourceShipperThread(String walGroupId,
522         PriorityBlockingQueue<Path> queue, ReplicationQueueInfo replicationQueueInfo,
523         ReplicationSourceInterface source) {
524       this.walGroupId = walGroupId;
525       this.queue = queue;
526       this.replicationQueueInfo = replicationQueueInfo;
527       this.source = source;
528     }
529 
530     @Override
531     public void run() {
532       setWorkerState(WorkerState.RUNNING);
533       // Loop until we close down
534       while (isWorkerActive()) {
535         int sleepMultiplier = 1;
536         // Sleep until replication is enabled again
537         if (!isPeerEnabled()) {
538           if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
539             sleepMultiplier++;
540           }
541           continue;
542         }
543         while (entryReader == null) {
544           if (sleepForRetries("Replication WAL entry reader thread not initialized",
545             sleepMultiplier)) {
546             sleepMultiplier++;
547           }
548           if (sleepMultiplier == maxRetriesMultiplier) {
549             LOG.warn("Replication WAL entry reader thread not initialized");
550           }
551         }
552 
553         try {
554           WALEntryBatch entryBatch = entryReader.take();
555           shipEdits(entryBatch);
556           if (replicationQueueInfo.isQueueRecovered() && entryBatch.getWalEntries().isEmpty()
557               && entryBatch.getLastSeqIds().isEmpty()) {
558             LOG.debug("Finished recovering queue for group " + walGroupId + " of peer "
559                 + peerClusterZnode);
560             metrics.incrCompletedRecoveryQueue();
561             setWorkerState(WorkerState.FINISHED);
562             continue;
563           }
564         } catch (InterruptedException e) {
565           LOG.trace("Interrupted while waiting for next replication entry batch", e);
566           Thread.currentThread().interrupt();
567         }
568       }
569 
570       if (replicationQueueInfo.isQueueRecovered() && getWorkerState() == WorkerState.FINISHED) {
571         // use synchronize to make sure one last thread will clean the queue
572         synchronized (this) {
573           Threads.sleep(100);// wait a short while for other worker thread to fully exit
574           boolean allOtherTaskDone = true;
575           for (ReplicationSourceShipperThread worker : workerThreads.values()) {
576             if (!worker.equals(this) && worker.getWorkerState() != WorkerState.FINISHED) {
577               allOtherTaskDone = false;
578               break;
579             }
580           }
581           if (allOtherTaskDone) {
582             manager.closeRecoveredQueue(this.source);
583             // stop replication endpoint
584             if (source instanceof ReplicationSource) {
585               ((ReplicationSource) source).replicationEndpoint.stop();
586             }
587             LOG.info("Finished recovering queue " + peerClusterZnode
588                 + " with the following stats: " + getStats());
589           }
590         }
591       }
592       // If the worker exits run loop without finishing it's task, mark it as stopped.
593       if (state != WorkerState.FINISHED) {
594         setWorkerState(WorkerState.STOPPED);
595       }
596     }

598     private void cleanUpHFileRefs(WALEdit edit) throws IOException {
599       String peerId = peerClusterZnode;
600       if (peerId.contains("-")) {
601         // peerClusterZnode will be in the form peerId + "-" + rsZNode.
602         // A peerId will not have "-" in its name, see HBASE-11394
603         peerId = peerClusterZnode.split("-")[0];
604       }
605       List<Cell> cells = edit.getCells();
606       int totalCells = cells.size();
607       for (int i = 0; i < totalCells; i++) {
608         Cell cell = cells.get(i);
609         if (CellUtil.matchingQualifier(cell, WALEdit.BULK_LOAD)) {
610           BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cell);
611           List<StoreDescriptor> stores = bld.getStoresList();
612           int totalStores = stores.size();
613           for (int j = 0; j < totalStores; j++) {
614             List<String> storeFileList = stores.get(j).getStoreFileList();
615             manager.cleanUpHFileRefs(peerId, storeFileList);
616             metrics.decrSizeOfHFileRefsQueue(storeFileList.size());
617           }
618         }
619       }
620     }

632     /**
633      * Do the shipping logic
634      */
635     protected void shipEdits(WALEntryBatch entryBatch) {
636       List<Entry> entries = entryBatch.getWalEntries();
637       long lastReadPosition = entryBatch.getLastWalPosition();
638       currentPath = entryBatch.getLastWalPath();
639       int sleepMultiplier = 0;
640       if (entries.isEmpty()) {
641         if (lastLoggedPosition != lastReadPosition) {
642           updateLogPosition(lastReadPosition);
643           // if there was nothing to ship and it's not an error
644           // set "ageOfLastShippedOp" to <now> to indicate that we're current
645           metrics.setAgeOfLastShippedOp(EnvironmentEdgeManager.currentTime(), walGroupId);
646         }
647         return;
648       }
649       int currentSize = (int) entryBatch.getHeapSize();
650       while (isWorkerActive()) {
651         try {
652           checkBandwidthChangeAndResetThrottler();
653           if (throttler.isEnabled()) {
654             long sleepTicks = throttler.getNextSleepInterval(currentSize);
655             if (sleepTicks > 0) {
656               try {
657                 if (LOG.isTraceEnabled()) {
658                   LOG.trace("To sleep " + sleepTicks + "ms for throttling control");
659                 }
660                 Thread.sleep(sleepTicks);
661               } catch (InterruptedException e) {
662                 LOG.debug("Interrupted while sleeping for throttling control");
663                 Thread.currentThread().interrupt();
664                 // current thread might be interrupted to terminate
665                 // directly go back to while() for confirm this
666                 continue;
667               }
668               // reset throttler's cycle start tick when sleep for throttling occurs
669               throttler.resetStartTick();
670             }
671           }
672           // create replicateContext here, so the entries can be GC'd upon return from this call
673           // stack
674           ReplicationEndpoint.ReplicateContext replicateContext =
675               new ReplicationEndpoint.ReplicateContext();
676           replicateContext.setEntries(entries).setSize(currentSize);
677           replicateContext.setWalGroupId(walGroupId);
678 
679           long startTimeNs = System.nanoTime();
680           // send the edits to the endpoint. Will block until the edits are shipped and acknowledged
681           boolean replicated = replicationEndpoint.replicate(replicateContext);
682           long endTimeNs = System.nanoTime();
683 
684           if (!replicated) {
685             continue;
686           } else {
687             sleepMultiplier = Math.max(sleepMultiplier - 1, 0);
688           }
689 
690           if (this.lastLoggedPosition != lastReadPosition) {
691             //Clean up hfile references
692             int size = entries.size();
693             for (int i = 0; i < size; i++) {
694               cleanUpHFileRefs(entries.get(i).getEdit());
695 
696               TableName tableName = entries.get(i).getKey().getTablename();
697               source.getSourceMetrics().setAgeOfLastShippedOpByTable(
698                 entries.get(i).getKey().getWriteTime(),
699                 tableName.getNameAsString());
700             }
701             //Log and clean up WAL logs
702             updateLogPosition(lastReadPosition);
703           }
704           if (throttler.isEnabled()) {
705             throttler.addPushSize(currentSize);
706           }
707           totalReplicatedEdits.addAndGet(entries.size());
708           totalReplicatedOperations.addAndGet(entryBatch.getNbOperations());
709           // FIXME check relationship between wal group and overall
710           metrics.shipBatch(entryBatch.getNbOperations(), currentSize, entryBatch.getNbHFiles());
711           metrics.setAgeOfLastShippedOp(entries.get(entries.size() - 1).getKey().getWriteTime(),
712             walGroupId);
713           if (LOG.isTraceEnabled()) {
714             LOG.trace("Replicated " + totalReplicatedEdits + " entries in total, or "
715                 + totalReplicatedOperations + " operations in "
716                 + ((endTimeNs - startTimeNs) / 1000000) + " ms");
717           }
718           break;
719         } catch (Exception ex) {
720           LOG.warn(replicationEndpoint.getClass().getName() + " threw unknown exception:"
721               + org.apache.hadoop.util.StringUtils.stringifyException(ex));
722           if (sleepForRetries("ReplicationEndpoint threw exception", sleepMultiplier)) {
723             sleepMultiplier++;
724           }
725         }
726       }
727     }

736     public void startup() {
737       String n = Thread.currentThread().getName();
738       Thread.UncaughtExceptionHandler handler = new Thread.UncaughtExceptionHandler() {
739         @Override
740         public void uncaughtException(final Thread t, final Throwable e) {
741           RSRpcServices.exitIfOOME(e);
742           LOG.error("Unexpected exception in ReplicationSourceWorkerThread," + " currentPath="
743               + getCurrentPath(), e);
744           stopper.stop("Unexpected exception in ReplicationSourceWorkerThread");
745         }
746       };
747       Threads.setDaemonThreadRunning(this, n + ".replicationSource." + walGroupId + ","
748           + peerClusterZnode, handler);
749       workerThreads.put(walGroupId, this);
750 
751       long startPosition = 0;
752 
753       if (this.replicationQueueInfo.isQueueRecovered()) {
754         startPosition = getRecoveredQueueStartPos(startPosition);
755         int numRetries = 0;
756         while (numRetries <= maxRetriesMultiplier) {
757           try {
758             locateRecoveredPaths();
759             break;
760           } catch (IOException e) {
761             LOG.error("Error while locating recovered queue paths, attempt #" + numRetries);
762             numRetries++;
763           }
764         }
765       }
766 
767       startWALReaderThread(n, handler, startPosition);
768     }

799     // Loops through the recovered queue and tries to find the location of each log
800     // this is necessary because the logs may have moved before recovery was initiated
801     private void locateRecoveredPaths() throws IOException {
802       boolean hasPathChanged = false;
803       PriorityBlockingQueue<Path> newPaths =
804           new PriorityBlockingQueue<Path>(queueSizePerGroup, new LogsComparator());
805       pathsLoop: for (Path path : queue) {
806         if (fs.exists(path)) { // still in same location, don't need to do anything
807           newPaths.add(path);
808           continue;
809         }
810         // Path changed - try to find the right path.
811         hasPathChanged = true;
812         if (stopper instanceof ReplicationSyncUp.DummyServer) {
813           // In the case of disaster/recovery, HMaster may be shutdown/crashed before flush data
814           // from .logs to .oldlogs. Loop into .logs folders and check whether a match exists
815           Path newPath = getReplSyncUpPath(path);
816           newPaths.add(newPath);
817           continue;
818         } else {
819           // See if Path exists in the dead RS folder (there could be a chain of failures
820           // to look at)
821           List<String> deadRegionServers = this.replicationQueueInfo.getDeadRegionServers();
822           LOG.info("NB dead servers : " + deadRegionServers.size());
823           final Path walDir = FSUtils.getWALRootDir(conf);
824           for (String curDeadServerName : deadRegionServers) {
825             final Path deadRsDirectory =
826                 new Path(walDir, DefaultWALProvider.getWALDirectoryName(curDeadServerName));
827             Path[] locs = new Path[] { new Path(deadRsDirectory, path.getName()), new Path(
828                 deadRsDirectory.suffix(DefaultWALProvider.SPLITTING_EXT), path.getName()) };
829             for (Path possibleLogLocation : locs) {
830               LOG.info("Possible location " + possibleLogLocation.toUri().toString());
831               if (manager.getFs().exists(possibleLogLocation)) {
832                 // We found the right new location
833                 LOG.info("Log " + path + " still exists at " + possibleLogLocation);
834                 newPaths.add(possibleLogLocation);
835                 continue pathsLoop;
836               }
837             }
838           }
839           // didn't find a new location
840           LOG.error(
841             String.format("WAL Path %s doesn't exist and couldn't find its new location", path));
842           newPaths.add(path);
843         }
844       }
845 
846       if (hasPathChanged) {
847         if (newPaths.size() != queue.size()) { // this shouldn't happen
848           LOG.error("Recovery queue size is incorrect");
849           throw new IOException("Recovery queue size error");
850         }
851         // put the correct locations in the queue
852         // since this is a recovered queue with no new incoming logs,
853         // there shouldn't be any concurrency issues
854         queue.clear();
855         for (Path path : newPaths) {
856           queue.add(path);
857         }
858       }
859     }

861     // N.B. the ReplicationSyncUp tool sets the manager.getWALDir to the root of the wal
862     // area rather than to the wal area for a particular region server.
863     private Path getReplSyncUpPath(Path path) throws IOException {
864       FileStatus[] rss = fs.listStatus(manager.getLogDir());
865       for (FileStatus rs : rss) {
866         Path p = rs.getPath();
867         FileStatus[] logs = fs.listStatus(p);
868         for (FileStatus log : logs) {
869           p = new Path(p, log.getPath().getName());
870           if (p.getName().equals(path.getName())) {
871             LOG.info("Log " + p.getName() + " found at " + p);
872             return p;
873           }
874         }
875       }
876       LOG.error("Didn't find path for: " + path.getName());
877       return path;
878     }
