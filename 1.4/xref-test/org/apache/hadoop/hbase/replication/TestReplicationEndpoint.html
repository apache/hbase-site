294   public void testMetricsSourceBaseSourcePassthrough(){
295     /*
296     The replication MetricsSource wraps a MetricsReplicationSourceSourceImpl
297     and a MetricsReplicationGlobalSourceSource, so that metrics get written to both namespaces.
298     Both of those classes wrap a MetricsReplicationSourceImpl that implements BaseSource, which
299     allows for custom JMX metrics.
300     This test checks to make sure the BaseSource decorator logic on MetricsSource actually calls down through
301     the two layers of wrapping to the actual BaseSource.
302     */
303     String id = "id";
304     DynamicMetricsRegistry mockRegistry = mock(DynamicMetricsRegistry.class);
305     MetricsReplicationSourceImpl singleRms = mock(MetricsReplicationSourceImpl.class);
306     when(singleRms.getMetricsRegistry()).thenReturn(mockRegistry);
307     MetricsReplicationSourceImpl globalRms = mock(MetricsReplicationSourceImpl.class);
308     when(globalRms.getMetricsRegistry()).thenReturn(mockRegistry);
309 
310     MetricsReplicationSourceSource singleSourceSource = new MetricsReplicationSourceSourceImpl(singleRms, id);
311     MetricsReplicationSourceSource globalSourceSource = new MetricsReplicationGlobalSourceSource(globalRms);
312     MetricsReplicationSourceSource spyglobalSourceSource = spy(globalSourceSource);
313     doNothing().when(spyglobalSourceSource).incrFailedRecoveryQueue();
314     Map<String, MetricsReplicationSourceSource> singleSourceSourceByTable = new HashMap<>();
315     MetricsSource source = new MetricsSource(id, singleSourceSource, spyglobalSourceSource,
316         singleSourceSourceByTable);
317     String gaugeName = "gauge";
318     String singleGaugeName = "source.id." + gaugeName;
319     long delta = 1;
320     String counterName = "counter";
321     String singleCounterName = "source.id." + counterName;
322     long count = 2;
323     source.decGauge(gaugeName, delta);
324     source.getMetricsContext();
325     source.getMetricsDescription();
326     source.getMetricsJmxContext();
327     source.getMetricsName();
328     source.incCounters(counterName, count);
329     source.incGauge(gaugeName, delta);
330     source.init();
331     source.removeMetric(gaugeName);
332     source.setGauge(gaugeName, delta);
333     source.updateHistogram(counterName, count);
334     source.incrFailedRecoveryQueue();
335 
336     verify(singleRms).decGauge(singleGaugeName, delta);
337     verify(globalRms).decGauge(gaugeName, delta);
338     verify(globalRms).getMetricsContext();
339     verify(globalRms).getMetricsJmxContext();
340     verify(globalRms).getMetricsName();
341     verify(singleRms).incCounters(singleCounterName, count);
342     verify(globalRms).incCounters(counterName, count);
343     verify(singleRms).incGauge(singleGaugeName, delta);
344     verify(globalRms).incGauge(gaugeName, delta);
345     verify(globalRms).init();
346     verify(singleRms).removeMetric(singleGaugeName);
347     verify(globalRms).removeMetric(gaugeName);
348     verify(singleRms).setGauge(singleGaugeName, delta);
349     verify(globalRms).setGauge(gaugeName, delta);
350     verify(singleRms).updateHistogram(singleCounterName, count);
351     verify(globalRms).updateHistogram(counterName, count);
352     verify(spyglobalSourceSource).incrFailedRecoveryQueue();
353 
354     // check singleSourceSourceByTable metrics.
355     // singleSourceSourceByTable map entry will be created only
356     // after calling #setAgeOfLastShippedOpByTable
357     boolean containsRandomNewTable = source.getSingleSourceSourceByTable()
358         .containsKey("RandomNewTable");
359     Assert.assertEquals(false, containsRandomNewTable);
360     source.setAgeOfLastShippedOpByTable(123L, "RandomNewTable");
361     containsRandomNewTable = source.getSingleSourceSourceByTable()
362       .containsKey("RandomNewTable");
363     Assert.assertEquals(true, containsRandomNewTable);
364     MetricsReplicationSourceSource msr = source.getSingleSourceSourceByTable()
365         .get("RandomNewTable");
366     // cannot put more concreate value here to verify because the age is arbitrary.
367     // as long as it's greater than 0, we see it as correct answer.
368     Assert.assertTrue(msr.getLastShippedAge() > 0);
369   }

527         public Entry filter(Entry entry) {
528           ArrayList<Cell> cells = entry.getEdit().getCells();
529           int size = cells.size();
530           for (int i = size-1; i >= 0; i--) {
531             Cell cell = cells.get(i);
532             if (!Bytes.equals(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
533               row, 0, row.length)) {
534               cells.remove(i);
535             }
536           }
537           return entry;
538         }
