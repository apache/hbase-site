257    * An example to scan using listener in unmarshaller for XML.
258    * @throws Exception the exception

260   @Test
261   public void testScanUsingListenerUnmarshallerXML() throws Exception {
262     StringBuilder builder = new StringBuilder();
263     builder.append("/*");
264     builder.append("?");
265     builder.append(Constants.SCAN_COLUMN + "=" + COLUMN_1);
266     builder.append("&");
267     builder.append(Constants.SCAN_LIMIT + "=10");
268     Response response = client.get("/" + TABLE + builder.toString(),
269       Constants.MIMETYPE_XML);
270     assertEquals(200, response.getCode());
271     assertEquals(Constants.MIMETYPE_XML, response.getHeader("content-type"));
272     JAXBContext context = JAXBContext.newInstance(ClientSideCellSetModel.class, RowModel.class,
273       CellModel.class);
274     Unmarshaller unmarshaller = context.createUnmarshaller();
275 
276     final ClientSideCellSetModel.Listener listener = new ClientSideCellSetModel.Listener() {
277       @Override
278       public void handleRowModel(ClientSideCellSetModel helper, RowModel row) {
279         assertTrue(row.getKey() != null);
280         assertTrue(row.getCells().size() > 0);
281       }
282     };
283 
284     // install the callback on all ClientSideCellSetModel instances
285     unmarshaller.setListener(new Unmarshaller.Listener() {
286         @Override
287         public void beforeUnmarshal(Object target, Object parent) {
288             if (target instanceof ClientSideCellSetModel) {
289                 ((ClientSideCellSetModel) target).setCellSetModelListener(listener);
290             }
291         }
292 
293         @Override
294         public void afterUnmarshal(Object target, Object parent) {
295             if (target instanceof ClientSideCellSetModel) {
296                 ((ClientSideCellSetModel) target).setCellSetModelListener(null);
297             }
298         }
299     });
300 
301     // create a new XML parser
302     SAXParserFactory factory = SAXParserFactory.newInstance();
303     factory.setNamespaceAware(true);
304     XMLReader reader = factory.newSAXParser().getXMLReader();
305     reader.setContentHandler(unmarshaller.getUnmarshallerHandler());
306     assertFalse(ClientSideCellSetModel.listenerInvoked);
307     reader.parse(new InputSource(response.getStream()));
308     assertTrue(ClientSideCellSetModel.listenerInvoked);
309 
310   }

328     int count = TestScannerResource.countCellSet(model);
329     assertEquals(20, count);

332     //Test scanning with no limit.
333     builder = new StringBuilder();
334     builder.append("/*");
335     builder.append("?");
336     builder.append(Constants.SCAN_COLUMN + "=" + COLUMN_2);
337     response = client.get("/" + TABLE + builder.toString(),
338       Constants.MIMETYPE_JSON);
339     assertEquals(200, response.getCode());
340     assertEquals(Constants.MIMETYPE_JSON, response.getHeader("content-type"));
341     model = mapper.readValue(response.getStream(), CellSetModel.class);
342     count = TestScannerResource.countCellSet(model);
343     assertEquals(expectedRows2, count);
344     checkRowsNotNull(model);
345 
346     //Test with start row and end row.
347     builder = new StringBuilder();
348     builder.append("/*");
349     builder.append("?");
350     builder.append(Constants.SCAN_COLUMN + "=" + COLUMN_1);
351     builder.append("&");
352     builder.append(Constants.SCAN_START_ROW + "=aaa");
353     builder.append("&");
354     builder.append(Constants.SCAN_END_ROW + "=aay");
355     response = client.get("/" + TABLE + builder.toString(),
356       Constants.MIMETYPE_JSON);
357     assertEquals(200, response.getCode());
358 
359     count = 0;
360     JsonFactory jfactory = new JsonFactory(mapper);
361     JsonParser jParser = jfactory.createParser(response.getStream());
362     boolean found = false;
363     while (jParser.nextToken() != JsonToken.END_OBJECT) {
364       if(jParser.getCurrentToken() == JsonToken.START_OBJECT && found) {
365         RowModel row = jParser.readValueAs(RowModel.class);
366         assertNotNull(row.getKey());
367         for (int i = 0; i < row.getCells().size(); i++) {
368           if (count == 0) {
369             assertEquals("aaa", Bytes.toString(row.getKey()));
370           }
371           if (count == 23) {
372             assertEquals("aax", Bytes.toString(row.getKey()));
373           }
374           count++;
375         }
376         jParser.skipChildren();
377       } else {
378         found = jParser.getCurrentToken() == JsonToken.START_ARRAY;
379       }
380     }
381     assertEquals(24, count);
382   }

423   /**
424    * Read protobuf stream.
425    * @param inputStream the input stream
426    * @return The number of rows in the cell set model.
427    * @throws IOException Signals that an I/O exception has occurred.
428    */
429   public int readProtobufStream(InputStream inputStream) throws IOException{
430     DataInputStream stream = new DataInputStream(inputStream);
431     CellSetModel model = null;
432     int rowCount = 0;
433     try {
434       while (true) {
435         byte[] lengthBytes = new byte[2];
436         int readBytes = stream.read(lengthBytes);
437         if (readBytes == -1) {
438           break;
439         }
440         assertEquals(2, readBytes);
441         int length = Bytes.toShort(lengthBytes);
442         byte[] cellset = new byte[length];
443         stream.read(cellset);
444         model = new CellSetModel();
445         model.getObjectFromMessage(cellset);
446         checkRowsNotNull(model);
447         rowCount = rowCount + TestScannerResource.countCellSet(model);
448       }
449     } catch (EOFException exp) {
450       exp.printStackTrace();
451     } finally {
452       stream.close();
453     }
454     return rowCount;
455   }

690   @XmlRootElement(name = "CellSet")

693 
694     private static final long serialVersionUID = 1L;
695 
696     /**
697      * This list is not a real list; instead it will notify a listener whenever JAXB has
698      * unmarshalled the next row.
699      */
700     @XmlElement(name="Row")
701     private List<RowModel> row;
702 
703     static boolean listenerInvoked = false;
704 
705     /**
706      * Install a listener for row model on this object. If l is null, the listener
707      * is removed again.
708      */
709     public void setCellSetModelListener(final Listener l) {
710         row = (l == null) ? null : new ArrayList<RowModel>() {
711         private static final long serialVersionUID = 1L;
712             @Override
713             public boolean add(RowModel o) {
714                 l.handleRowModel(ClientSideCellSetModel.this, o);
715                 listenerInvoked = true;
716                 return false;
717             }
718         };
719     }
720 
721     /**
722      * This listener is invoked every time a new row model is unmarshalled.
723      */
724     public static interface Listener {
725         void handleRowModel(ClientSideCellSetModel helper, RowModel rowModel);
726     }
