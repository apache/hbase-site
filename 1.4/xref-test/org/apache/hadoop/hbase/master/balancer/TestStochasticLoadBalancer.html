124   public void testKeepRegionLoad() throws Exception {
125 
126     ServerName sn = ServerName.valueOf("test:8080", 100);
127     int numClusterStatusToAdd = 20000;
128     for (int i = 0; i < numClusterStatusToAdd; i++) {
129       ServerLoad sl = mock(ServerLoad.class);
130 
131       RegionLoad rl = mock(RegionLoad.class);
132       when(rl.getStores()).thenReturn(i);
133 
134       Map<byte[], RegionLoad> regionLoadMap =
135           new TreeMap<byte[], RegionLoad>(Bytes.BYTES_COMPARATOR);
136       regionLoadMap.put(Bytes.toBytes(REGION_KEY), rl);
137       when(sl.getRegionsLoad()).thenReturn(regionLoadMap);
138 
139       ClusterStatus clusterStatus = mock(ClusterStatus.class);
140       when(clusterStatus.getServers()).thenReturn(Arrays.asList(sn));
141       when(clusterStatus.getLoad(sn)).thenReturn(sl);
142 
143       loadBalancer.setClusterStatus(clusterStatus);
144     }
145 
146     String regionNameAsString = HRegionInfo.getRegionNameAsString(Bytes.toBytes(REGION_KEY));
147     assertTrue(loadBalancer.loads.get(regionNameAsString) != null);
148     assertTrue(loadBalancer.loads.get(regionNameAsString).size() == 15);
149 
150     Queue<RegionLoad> loads = loadBalancer.loads.get(regionNameAsString);
151     int i = 0;
152     while(loads.size() > 0) {
153       RegionLoad rl = loads.remove();
154       assertEquals(i + (numClusterStatusToAdd - 15), rl.getStores());
155       i ++;
156     }
157   }

286   @Test
287   public void testCostAfterUndoAction() {
288     final int runs = 10;
289     loadBalancer.setConf(conf);
290     for (int[] mockCluster : clusterStateMocks) {
291       BaseLoadBalancer.Cluster cluster = mockCluster(mockCluster);
292       loadBalancer.initCosts(cluster);
293       for (int i = 0; i != runs; ++i) {
294         final double expectedCost = loadBalancer.computeCost(cluster, Double.MAX_VALUE);
295         Cluster.Action action = loadBalancer.nextAction(cluster);
296         cluster.doAction(action);
297         loadBalancer.updateCostsWithAction(cluster, action);
298         Cluster.Action undoAction = action.undoAction();
299         cluster.doAction(undoAction);
300         loadBalancer.updateCostsWithAction(cluster, undoAction);
301         final double actualCost = loadBalancer.computeCost(cluster, Double.MAX_VALUE);
302         assertEquals(expectedCost, actualCost, 0);
303       }
304     }
305   }

495     // check for the case where there are two hosts and with one rack, and where
496     // both the replicas are hosted on the same server
497     List<HRegionInfo> regions = randomRegions(1);
498     ServerName s1 = ServerName.valueOf("host1", 1000, 11111);
499     ServerName s2 = ServerName.valueOf("host11", 1000, 11111);
500     Map<ServerName, List<HRegionInfo>> map = new HashMap<ServerName, List<HRegionInfo>>();
501     map.put(s1, regions);
502     regions.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));

504     regions = randomRegions(1);
505     map.put(s2, regions);
506     assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null, null)));
507     // check for the case where there are two hosts on the same rack and there are two racks
508     // and both the replicas are on the same rack

510     regions = randomRegions(1);

512     regionsOnS2.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
513     map.put(s1, regions);
514     map.put(s2, regionsOnS2);
515     // add another server so that the cluster has some host on another rack
516     map.put(ServerName.valueOf("host2", 1000, 11111), randomRegions(1));
517     assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null,
518         new ForTestRackManagerOne())));
