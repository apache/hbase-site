020package org.apache.hadoop.hbase.rsgroup;
021
022import org.apache.commons.logging.Log;
023import org.apache.commons.logging.LogFactory;
024import org.apache.hadoop.hbase.IntegrationTestingUtility;
025import org.apache.hadoop.hbase.Waiter;
026import org.apache.hadoop.hbase.testclassification.IntegrationTests;
027import org.junit.After;
028import org.junit.Before;
029import org.junit.experimental.categories.Category;
030
031/**
032 * Runs all of the units tests defined in TestGroupBase
033 * as an integration test.
034 * Requires TestRSGroupBase.NUM_SLAVE_BASE servers to run.
035 */
036@Category(IntegrationTests.class)
037public class IntegrationTestRSGroup extends TestRSGroupsBase {
038  //Integration specific
039  private final static Log LOG = LogFactory.getLog(IntegrationTestRSGroup.class);
040  private static boolean initialized = false;
041
042  @Before
043  public void beforeMethod() throws Exception {
044    if(!initialized) {
045      LOG.info("Setting up IntegrationTestGroup");
046      LOG.info("Initializing cluster with " + NUM_SLAVES_BASE + " servers");
047      TEST_UTIL = new IntegrationTestingUtility();
048      ((IntegrationTestingUtility)TEST_UTIL).initializeCluster(NUM_SLAVES_BASE);
049      //set shared configs
050      admin = TEST_UTIL.getAdmin();
051      cluster = TEST_UTIL.getHBaseClusterInterface();
052      rsGroupAdmin = new VerifyingRSGroupAdminClient(rsGroupAdmin.newClient(TEST_UTIL.getConnection()),
053          TEST_UTIL.getConfiguration());
054      LOG.info("Done initializing cluster");
055      initialized = true;
056      //cluster may not be clean
057      //cleanup when initializing
058      afterMethod();
059    }
060  }
061
062  @After
063  public void afterMethod() throws Exception {
064    LOG.info("Cleaning up previous test run");
065    //cleanup previous artifacts
066    deleteTableIfNecessary();
067    deleteNamespaceIfNecessary();
068    deleteGroups();
069    admin.setBalancerRunning(true, true);
070
071    LOG.info("Restoring the cluster");
072    ((IntegrationTestingUtility)TEST_UTIL).restoreCluster();
073    LOG.info("Done restoring the cluster");
074
075    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
076      @Override
077      public boolean evaluate() throws Exception {
078        LOG.info("Waiting for cleanup to finish "+ rsGroupAdmin.listRSGroups());
079        //Might be greater since moving servers back to default
080        //is after starting a server
081        return rsGroupAdmin.getRSGroupInfo(RSGroupInfo.DEFAULT_GROUP).getServers().size()
082            >= NUM_SLAVES_BASE;
083      }
084    });
085
086    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
087      @Override
088      public boolean evaluate() throws Exception {
089        LOG.info("Waiting for regionservers to be registered "+ rsGroupAdmin.listRSGroups());
090        //Might be greater since moving servers back to default
091        //is after starting a server
092        return rsGroupAdmin.getRSGroupInfo(RSGroupInfo.DEFAULT_GROUP).getServers().size()
093            == getNumServers();
094      }
095    });
096
097    LOG.info("Done cleaning up previous test run");
098  }
099}
