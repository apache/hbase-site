153    rsGroupAdmin.addRSGroup(groupName);

155    assertTrue(observer.postAddRSGroupCalled);
156
157    admin.createNamespace(NamespaceDescriptor.create(nsName)

161      rsGroupAdmin.removeRSGroup(groupName);

175    assertTrue(observer.postRemoveRSGroupCalled);

177      admin.createNamespace(NamespaceDescriptor.create(nsName)
178        .addConfiguration(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP, "foo").build());

257  @Test
258  public void testMultiTableMove() throws Exception {
259    final TableName tableNameA = TableName.valueOf(tablePrefix + name.getMethodName() + "A");
260    final TableName tableNameB = TableName.valueOf(tablePrefix + name.getMethodName() + "B");
261    final byte[] familyNameBytes = Bytes.toBytes("f");
262    String newGroupName = getGroupName(name.getMethodName());
263    final RSGroupInfo newGroup = addGroup(newGroupName, 1);
264
265    TEST_UTIL.createTable(tableNameA, familyNameBytes);
266    TEST_UTIL.createTable(tableNameB, familyNameBytes);
267    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
268      @Override
269      public boolean evaluate() throws Exception {
270        List<String> regionsA = getTableRegionMap().get(tableNameA);
271        if (regionsA == null) {
272          return false;
273        }
274        List<String> regionsB = getTableRegionMap().get(tableNameB);
275        if (regionsB == null) {
276          return false;
277        }
278        return getTableRegionMap().get(tableNameA).size() >= 1 &&
279          getTableRegionMap().get(tableNameB).size() >= 1;
280      }
281    });
282
283    RSGroupInfo tableGrpA = rsGroupAdmin.getRSGroupInfoOfTable(tableNameA);
284    assertTrue(tableGrpA.getName().equals(RSGroupInfo.DEFAULT_GROUP));
285
286    RSGroupInfo tableGrpB = rsGroupAdmin.getRSGroupInfoOfTable(tableNameB);
287    assertTrue(tableGrpB.getName().equals(RSGroupInfo.DEFAULT_GROUP));
288    // change table's group
289    LOG.info("Moving table [" + tableNameA + "," + tableNameB + "] to " + newGroup.getName());
290    rsGroupAdmin.moveTables(Sets.newHashSet(tableNameA, tableNameB), newGroup.getName());
291
292    // verify group change
293    Assert.assertEquals(newGroup.getName(),
294      rsGroupAdmin.getRSGroupInfoOfTable(tableNameA).getName());
295
296    Assert.assertEquals(newGroup.getName(),
297      rsGroupAdmin.getRSGroupInfoOfTable(tableNameB).getName());
298
299    // verify tables' not exist in old group
300    Set<TableName> DefaultTables =
301      rsGroupAdmin.getRSGroupInfo(RSGroupInfo.DEFAULT_GROUP).getTables();
302    assertFalse(DefaultTables.contains(tableNameA));
303    assertFalse(DefaultTables.contains(tableNameB));
304
305    // verify tables' exist in new group
306    Set<TableName> newGroupTables = rsGroupAdmin.getRSGroupInfo(newGroupName).getTables();
307    assertTrue(newGroupTables.contains(tableNameA));
308    assertTrue(newGroupTables.contains(tableNameB));
309  }

311  @Test
312  public void testTableMoveTruncateAndDrop() throws Exception {
313    final byte[] familyNameBytes = Bytes.toBytes("f");
314    String newGroupName = getGroupName(name.getMethodName());
315    final RSGroupInfo newGroup = addGroup(newGroupName, 2);
316
317    TEST_UTIL.createMultiRegionTable(tableName, familyNameBytes, 5);
318    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
319      @Override
320      public boolean evaluate() throws Exception {
321        List<String> regions = getTableRegionMap().get(tableName);
322        if (regions == null) {
323          return false;
324        }
325
326        return getTableRegionMap().get(tableName).size() >= 5;
327      }
328    });
329
330    RSGroupInfo tableGrp = rsGroupAdmin.getRSGroupInfoOfTable(tableName);
331    assertTrue(tableGrp.getName().equals(RSGroupInfo.DEFAULT_GROUP));
332
333    // change table's group
334    LOG.info("Moving table " + tableName + " to " + newGroup.getName());
335    rsGroupAdmin.moveTables(Sets.newHashSet(tableName), newGroup.getName());
336
337    // verify group change
338    Assert.assertEquals(newGroup.getName(),
339      rsGroupAdmin.getRSGroupInfoOfTable(tableName).getName());
340
341    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
342      @Override
343      public boolean evaluate() throws Exception {
344        Map<ServerName, List<String>> serverMap = getTableServerRegionMap().get(tableName);
345        int count = 0;
346        if (serverMap != null) {
347          for (ServerName rs : serverMap.keySet()) {
348            if (newGroup.containsServer(rs.getAddress())) {
349              count += serverMap.get(rs).size();
350            }
351          }
352        }
353        return count == 5;
354      }
355    });
356
357    // test truncate
358    admin.disableTable(tableName);
359    admin.truncateTable(tableName, true);
360    Assert.assertEquals(1, rsGroupAdmin.getRSGroupInfo(newGroup.getName()).getTables().size());
361    Assert.assertEquals(tableName,
362      rsGroupAdmin.getRSGroupInfo(newGroup.getName()).getTables().first());
363
364    // verify removed table is removed from group
365    TEST_UTIL.deleteTable(tableName);
366    Assert.assertEquals(0, rsGroupAdmin.getRSGroupInfo(newGroup.getName()).getTables().size());
367
368    assertTrue(observer.preMoveTablesCalled);
369    assertTrue(observer.postMoveTablesCalled);
370  }

435  @Test
436  public void testRSGroupListDoesNotContainFailedTableCreation() throws Exception {
437    toggleQuotaCheckAndRestartMiniCluster(true);
438    String nsp = "np1";
439    NamespaceDescriptor nspDesc =
440      NamespaceDescriptor.create(nsp).addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "5")
441        .addConfiguration(TableNamespaceManager.KEY_MAX_TABLES, "2").build();
442    admin.createNamespace(nspDesc);
443    assertEquals(3, admin.listNamespaceDescriptors().length);
444    ColumnFamilyDescriptor fam1 = ColumnFamilyDescriptorBuilder.of("fam1");
445    TableDescriptor tableDescOne = TableDescriptorBuilder
446      .newBuilder(TableName.valueOf(nsp + TableName.NAMESPACE_DELIM + "table1"))
447      .setColumnFamily(fam1).build();
448    admin.createTable(tableDescOne);
449
450    TableDescriptor tableDescTwo = TableDescriptorBuilder
451      .newBuilder(TableName.valueOf(nsp + TableName.NAMESPACE_DELIM + "table2"))
452      .setColumnFamily(fam1).build();
453    boolean constraintViolated = false;
454
455    try {
456      admin.createTable(tableDescTwo, Bytes.toBytes("AAA"), Bytes.toBytes("ZZZ"), 6);
457      Assert.fail("Creation table should fail because of quota violation.");
458    } catch (Exception exp) {
459      assertTrue(exp instanceof IOException);
460      constraintViolated = true;
461    } finally {
462      assertTrue("Constraint not violated for table " + tableDescTwo.getTableName(),
463        constraintViolated);
464    }
465    List<RSGroupInfo> rsGroupInfoList = rsGroupAdmin.listRSGroups();
466    boolean foundTable2 = false;
467    boolean foundTable1 = false;
468    for (int i = 0; i < rsGroupInfoList.size(); i++) {
469      if (rsGroupInfoList.get(i).getTables().contains(tableDescTwo.getTableName())) {
470        foundTable2 = true;
471      }
472      if (rsGroupInfoList.get(i).getTables().contains(tableDescOne.getTableName())) {
473        foundTable1 = true;
474      }
475    }
476    assertFalse("Found table2 in rsgroup list.", foundTable2);
477    assertTrue("Did not find table1 in rsgroup list", foundTable1);
478
479    TEST_UTIL.deleteTable(tableDescOne.getTableName());
480    admin.deleteNamespace(nspDesc.getName());
481    toggleQuotaCheckAndRestartMiniCluster(false);
482
483  }

485  private void toggleQuotaCheckAndRestartMiniCluster(boolean enable) throws Exception {
486    TEST_UTIL.shutdownMiniCluster();
487    TEST_UTIL.getConfiguration().setBoolean(QuotaUtil.QUOTA_CONF_KEY, enable);

490      NUM_SLAVES_BASE - 1);
491    TEST_UTIL.getConfiguration().setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
492    initialize();
493  }
