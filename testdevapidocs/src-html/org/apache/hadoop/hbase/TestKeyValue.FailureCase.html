119  @Test
120  public void testPlainCompare() throws Exception {
121    final byte [] a = Bytes.toBytes("aaa");
122    final byte [] b = Bytes.toBytes("bbb");
123    final byte [] fam = Bytes.toBytes("col");
124    final byte [] qf = Bytes.toBytes("umn");
125    KeyValue aaa = new KeyValue(a, fam, qf, a);
126    KeyValue bbb = new KeyValue(b, fam, qf, b);
127    assertTrue(CellComparatorImpl.COMPARATOR.compare(aaa, bbb) < 0);
128    assertTrue(CellComparatorImpl.COMPARATOR.compare(bbb, aaa) > 0);
129    // Compare breaks if passed same ByteBuffer as both left and right arguments.
130    assertTrue(CellComparatorImpl.COMPARATOR.compare(bbb, bbb) == 0);
131    assertTrue(CellComparatorImpl.COMPARATOR.compare(aaa, aaa) == 0);
132    // Do compare with different timestamps.
133    aaa = new KeyValue(a, fam, qf, 1, a);
134    bbb = new KeyValue(a, fam, qf, 2, a);
135    assertTrue(CellComparatorImpl.COMPARATOR.compare(aaa, bbb) > 0);
136    assertTrue(CellComparatorImpl.COMPARATOR.compare(bbb, aaa) < 0);
137    assertTrue(CellComparatorImpl.COMPARATOR.compare(aaa, aaa) == 0);
138    // Do compare with different types.  Higher numbered types -- Delete
139    // should sort ahead of lower numbers; i.e. Put
140    aaa = new KeyValue(a, fam, qf, 1, KeyValue.Type.Delete, a);
141    bbb = new KeyValue(a, fam, qf, 1, a);
142    assertTrue(CellComparatorImpl.COMPARATOR.compare(aaa, bbb) < 0);
143    assertTrue(CellComparatorImpl.COMPARATOR.compare(bbb, aaa) > 0);
144    assertTrue(CellComparatorImpl.COMPARATOR.compare(aaa, aaa) == 0);
145  }

311  @Test
312  public void testCompareWithoutRow() {
313    final CellComparator c = CellComparatorImpl.COMPARATOR;
314    byte[] row = Bytes.toBytes("row");
315
316    byte[] fa = Bytes.toBytes("fa");
317    byte[] fami = Bytes.toBytes("fami");
318    byte[] fami1 = Bytes.toBytes("fami1");
319
320    byte[] qual0 = Bytes.toBytes("");
321    byte[] qual1 = Bytes.toBytes("qf1");
322    byte[] qual2 = Bytes.toBytes("qf2");
323    long ts = 1;
324
325    // 'fa:'
326    KeyValue kv_0 = new KeyValue(row, fa, qual0, ts, KeyValue.Type.Put);
327    // 'fami:'
328    KeyValue kv0_0 = new KeyValue(row, fami, qual0, ts, KeyValue.Type.Put);
329    // 'fami:qf1'
330    KeyValue kv0_1 = new KeyValue(row, fami, qual1, ts, KeyValue.Type.Put);
331    // 'fami:qf2'
332    KeyValue kv0_2 = new KeyValue(row, fami, qual2, ts, KeyValue.Type.Put);
333    // 'fami1:'
334    KeyValue kv1_0 = new KeyValue(row, fami1, qual0, ts, KeyValue.Type.Put);
335
336    // 'fami:qf1' < 'fami:qf2'
337    assertKVLessWithoutRow(c, 0, kv0_1, kv0_2);
338    // 'fami:qf1' < 'fami1:'
339    assertKVLessWithoutRow(c, 0, kv0_1, kv1_0);
340
341    // Test comparison by skipping the same prefix bytes.
342    /***
343     * KeyValue Format and commonLength:
344     * |_keyLen_|_valLen_|_rowLen_|_rowKey_|_famiLen_|_fami_|_Quali_|....
345     * ------------------|-------commonLength--------|--------------
346     */
347    int commonLength = KeyValue.ROW_LENGTH_SIZE + KeyValue.FAMILY_LENGTH_SIZE
348        + row.length;
349    // 'fa:' < 'fami:'. They have commonPrefix + 2 same prefix bytes.
350    assertKVLessWithoutRow(c, commonLength + 2, kv_0, kv0_0);
351    // 'fami:' < 'fami:qf1'. They have commonPrefix + 4 same prefix bytes.
352    assertKVLessWithoutRow(c, commonLength + 4, kv0_0, kv0_1);
353    // 'fami:qf1' < 'fami1:'. They have commonPrefix + 4 same prefix bytes.
354    assertKVLessWithoutRow(c, commonLength + 4, kv0_1, kv1_0);
355    // 'fami:qf1' < 'fami:qf2'. They have commonPrefix + 6 same prefix bytes.
356    assertKVLessWithoutRow(c, commonLength + 6, kv0_1, kv0_2);
357  }
