044import org.junit.Test;
045import org.junit.experimental.categories.Category;
046import org.slf4j.Logger;
047import org.slf4j.LoggerFactory;
048import org.apache.hbase.thirdparty.com.google.protobuf.BlockingRpcChannel;
049import org.apache.hbase.thirdparty.com.google.protobuf.ServiceException;
050
051@Category({ MasterTests.class, MediumTests.class })
052public class TestHMasterRPCException {
053
054  private static final Logger LOG = LoggerFactory.getLogger(TestHMasterRPCException.class);
055
056  private final HBaseTestingUtility testUtil = HBaseTestingUtility.createLocalHTU();
057
058  private HMaster master;
059
060  private RpcClient rpcClient;
061
062  @Before
063  public void setUp() throws Exception {
064    Configuration conf = testUtil.getConfiguration();
065    conf.set(HConstants.MASTER_PORT, "0");
066    conf.setInt(HConstants.ZK_SESSION_TIMEOUT, 2000);
067    testUtil.startMiniZKCluster();
068
069    ZKWatcher watcher = testUtil.getZooKeeperWatcher();
070    ZKUtil.createWithParents(watcher, watcher.znodePaths.masterAddressZNode, Bytes.toBytes("fake:123"));
071    master = new HMaster(conf);
072    rpcClient = RpcClientFactory.createClient(conf, HConstants.CLUSTER_ID_DEFAULT);
073  }
074
075  @After
076  public void tearDown() throws IOException {
077    if (rpcClient != null) {
078      rpcClient.close();
079    }
080    if (master != null) {
081      master.stopMaster();
082    }
083    testUtil.shutdownMiniZKCluster();
084  }
085
086  @Test
087  public void testRPCException() throws IOException, InterruptedException, KeeperException {
088    ServerName sm = master.getServerName();
089    boolean fakeZNodeDelete = false;
090    for (int i = 0; i < 20; i++) {
091      try {
092        BlockingRpcChannel channel = rpcClient.createBlockingRpcChannel(sm, User.getCurrent(), 0);
093        MasterProtos.MasterService.BlockingInterface stub =
094            MasterProtos.MasterService.newBlockingStub(channel);
095        assertTrue(stub.isMasterRunning(null, IsMasterRunningRequest.getDefaultInstance())
096            .getIsMasterRunning());
097        return;
098      } catch (ServiceException ex) {
099        IOException ie = ProtobufUtil.handleRemoteException(ex);
100        // No SocketTimeoutException here. RpcServer is already started after the construction of
101        // HMaster.
102        assertTrue(ie.getMessage().startsWith(
103          "org.apache.hadoop.hbase.ipc.ServerNotRunningYetException: Server is not running yet"));
104        LOG.info("Expected exception: ", ie);
105        if (!fakeZNodeDelete) {
106          testUtil.getZooKeeperWatcher().getRecoverableZooKeeper()
107              .delete(testUtil.getZooKeeperWatcher().znodePaths.masterAddressZNode, -1);
108          fakeZNodeDelete = true;
109        }
110      }
111      Thread.sleep(1000);
112    }
113  }
114}
