094  @Test
095  public void testCreateTableWithSingleReplica() throws Exception {
096    final int numRegions = 3;
097    final int numReplica = 1;
098    final TableName tableName = TableName.valueOf(name.getMethodName());
099    try {
100      HTableDescriptor desc = new HTableDescriptor(tableName);
101      desc.setRegionReplication(numReplica);
102      desc.addFamily(new HColumnDescriptor("family"));
103      ADMIN.createTable(desc, Bytes.toBytes("A"), Bytes.toBytes("Z"), numRegions);
104
105      validateNumberOfRowsInMeta(tableName, numRegions, ADMIN.getConnection());
106      List<HRegionInfo> hris = MetaTableAccessor.getTableRegions(
107        ADMIN.getConnection(), tableName);
108      assert(hris.size() == numRegions * numReplica);
109    } finally {
110      ADMIN.disableTable(tableName);
111      ADMIN.deleteTable(tableName);
112    }
113  }

115  @Test
116  public void testCreateTableWithMultipleReplicas() throws Exception {
117    final TableName tableName = TableName.valueOf(name.getMethodName());
118    final int numRegions = 3;
119    final int numReplica = 2;
120    try {
121      HTableDescriptor desc = new HTableDescriptor(tableName);
122      desc.setRegionReplication(numReplica);
123      desc.addFamily(new HColumnDescriptor("family"));
124      ADMIN.createTable(desc, Bytes.toBytes("A"), Bytes.toBytes("Z"), numRegions);
125      TEST_UTIL.waitTableEnabled(tableName);
126      validateNumberOfRowsInMeta(tableName, numRegions, ADMIN.getConnection());
127
128      List<HRegionInfo> hris = MetaTableAccessor.getTableRegions(ADMIN.getConnection(), tableName);
129      assert(hris.size() == numRegions * numReplica);
130      // check that the master created expected number of RegionState objects
131      for (int i = 0; i < numRegions; i++) {
132        for (int j = 0; j < numReplica; j++) {
133          HRegionInfo replica = RegionReplicaUtil.getRegionInfoForReplica(hris.get(i), j);
134          RegionState state = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager()
135              .getRegionStates().getRegionState(replica);
136          assert (state != null);
137        }
138      }
139
140      List<Result> metaRows = MetaTableAccessor.fullScanRegions(ADMIN.getConnection());
141      int numRows = 0;
142      for (Result result : metaRows) {
143        RegionLocations locations = MetaTableAccessor.getRegionLocations(result);
144        HRegionInfo hri = locations.getRegionLocation().getRegionInfo();
145        if (!hri.getTable().equals(tableName)) continue;
146        numRows += 1;
147        HRegionLocation[] servers = locations.getRegionLocations();
148        // have two locations for the replicas of a region, and the locations should be different
149        assert(servers.length == 2);
150        assert(!servers[0].equals(servers[1]));
151      }
152      assert(numRows == numRegions);
153
154      // The same verification of the meta as above but with the SnapshotOfRegionAssignmentFromMeta
155      // class
156      validateFromSnapshotFromMeta(TEST_UTIL, tableName, numRegions, numReplica,
157        ADMIN.getConnection());
158
159      // Now kill the master, restart it and see if the assignments are kept
160      ServerName master = TEST_UTIL.getHBaseClusterInterface().getClusterStatus().getMaster();
161      TEST_UTIL.getHBaseClusterInterface().stopMaster(master);
162      TEST_UTIL.getHBaseClusterInterface().waitForMasterToStop(master, 30000);
163      TEST_UTIL.getHBaseClusterInterface().startMaster(master.getHostname(), master.getPort());
164      TEST_UTIL.getHBaseClusterInterface().waitForActiveAndReadyMaster();
165      for (int i = 0; i < numRegions; i++) {
166        for (int j = 0; j < numReplica; j++) {
167          HRegionInfo replica = RegionReplicaUtil.getRegionInfoForReplica(hris.get(i), j);
168          RegionState state = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager()
169              .getRegionStates().getRegionState(replica);
170          assert (state != null);
171        }
172      }
173      validateFromSnapshotFromMeta(TEST_UTIL, tableName, numRegions, numReplica,
174        ADMIN.getConnection());
175
176      // Now shut the whole cluster down, and verify the assignments are kept so that the
177      // availability constraints are met.
178      TEST_UTIL.getConfiguration().setBoolean("hbase.master.startup.retainassign", true);
179      TEST_UTIL.shutdownMiniHBaseCluster();
180      TEST_UTIL.startMiniHBaseCluster(1, numSlaves);
181      TEST_UTIL.waitTableEnabled(tableName);
182      validateFromSnapshotFromMeta(TEST_UTIL, tableName, numRegions, numReplica,
183        ADMIN.getConnection());
184
185      // Now shut the whole cluster down, and verify regions are assigned even if there is only
186      // one server running
187      TEST_UTIL.shutdownMiniHBaseCluster();
188      TEST_UTIL.startMiniHBaseCluster(1, 1);
189      TEST_UTIL.waitTableEnabled(tableName);
190      validateSingleRegionServerAssignment(ADMIN.getConnection(), numRegions, numReplica);
191      for (int i = 1; i < numSlaves; i++) { //restore the cluster
192        TEST_UTIL.getMiniHBaseCluster().startRegionServer();
193      }
194
195      //check on alter table
196      ADMIN.disableTable(tableName);
197      assert(ADMIN.isTableDisabled(tableName));
198      //increase the replica
199      desc.setRegionReplication(numReplica + 1);
200      ADMIN.modifyTable(tableName, desc);
201      ADMIN.enableTable(tableName);
202      assert(ADMIN.isTableEnabled(tableName));
203      List<HRegionInfo> regions = TEST_UTIL.getMiniHBaseCluster().getMaster()
204          .getAssignmentManager().getRegionStates().getRegionsOfTable(tableName);
205      assert(regions.size() == numRegions * (numReplica + 1));
206
207      //decrease the replica(earlier, table was modified to have a replica count of numReplica + 1)
208      ADMIN.disableTable(tableName);
209      desc.setRegionReplication(numReplica);
210      ADMIN.modifyTable(tableName, desc);
211      ADMIN.enableTable(tableName);
212      assert(ADMIN.isTableEnabled(tableName));
213      regions = TEST_UTIL.getMiniHBaseCluster().getMaster()
214          .getAssignmentManager().getRegionStates().getRegionsOfTable(tableName);
215      assert(regions.size() == numRegions * numReplica);
216      //also make sure the meta table has the replica locations removed
217      hris = MetaTableAccessor.getTableRegions(ADMIN.getConnection(), tableName);
218      assert(hris.size() == numRegions * numReplica);
219      //just check that the number of default replica regions in the meta table are the same
220      //as the number of regions the table was created with, and the count of the
221      //replicas is numReplica for each region
222      Map<HRegionInfo, Integer> defaultReplicas = new HashMap<HRegionInfo, Integer>();
223      for (HRegionInfo hri : hris) {
224        Integer i;
225        HRegionInfo regionReplica0 = RegionReplicaUtil.getRegionInfoForDefaultReplica(hri);
226        defaultReplicas.put(regionReplica0,
227            (i = defaultReplicas.get(regionReplica0)) == null ? 1 : i + 1);
228      }
229      assert(defaultReplicas.size() == numRegions);
230      Collection<Integer> counts = new HashSet<Integer>(defaultReplicas.values());
231      assert(counts.size() == 1 && counts.contains(new Integer(numReplica)));
232    } finally {
233      ADMIN.disableTable(tableName);
234      ADMIN.deleteTable(tableName);
235    }
236  }

238  //@Test (TODO: enable when we have support for alter_table- HBASE-10361).
239  public void testIncompleteMetaTableReplicaInformation() throws Exception {
240    final TableName tableName = TableName.valueOf(name.getMethodName());
241    final int numRegions = 3;
242    final int numReplica = 2;
243    try {
244      // Create a table and let the meta table be updated with the location of the
245      // region locations.
246      HTableDescriptor desc = new HTableDescriptor(tableName);
247      desc.setRegionReplication(numReplica);
248      desc.addFamily(new HColumnDescriptor("family"));
249      ADMIN.createTable(desc, Bytes.toBytes("A"), Bytes.toBytes("Z"), numRegions);
250      TEST_UTIL.waitTableEnabled(tableName);
251      Set<byte[]> tableRows = new HashSet<byte[]>();
252      List<HRegionInfo> hris = MetaTableAccessor.getTableRegions(ADMIN.getConnection(), tableName);
253      for (HRegionInfo hri : hris) {
254        tableRows.add(hri.getRegionName());
255      }
256      ADMIN.disableTable(tableName);
257      // now delete one replica info from all the rows
258      // this is to make the meta appear to be only partially updated
259      Table metaTable = ADMIN.getConnection().getTable(TableName.META_TABLE_NAME);
260      for (byte[] row : tableRows) {
261        Delete deleteOneReplicaLocation = new Delete(row);
262        deleteOneReplicaLocation.addColumns(HConstants.CATALOG_FAMILY,
263          MetaTableAccessor.getServerColumn(1));
264        deleteOneReplicaLocation.addColumns(HConstants.CATALOG_FAMILY,
265          MetaTableAccessor.getSeqNumColumn(1));
266        deleteOneReplicaLocation.addColumns(HConstants.CATALOG_FAMILY,
267          MetaTableAccessor.getStartCodeColumn(1));
268        metaTable.delete(deleteOneReplicaLocation);
269      }
270      metaTable.close();
271      // even if the meta table is partly updated, when we re-enable the table, we should
272      // get back the desired number of replicas for the regions
273      ADMIN.enableTable(tableName);
274      assert(ADMIN.isTableEnabled(tableName));
275      List<HRegionInfo> regions = TEST_UTIL.getMiniHBaseCluster().getMaster()
276          .getAssignmentManager().getRegionStates().getRegionsOfTable(tableName);
277      assert(regions.size() == numRegions * numReplica);
278    } finally {
279      ADMIN.disableTable(tableName);
280      ADMIN.deleteTable(tableName);
281    }
282  }

292  private void validateNumberOfRowsInMeta(final TableName table, int numRegions,
293      Connection connection) throws IOException {
294    assert(ADMIN.tableExists(table));
295    final AtomicInteger count = new AtomicInteger();
296    Visitor visitor = new Visitor() {
297      @Override
298      public boolean visit(Result r) throws IOException {
299        if (MetaTableAccessor.getHRegionInfo(r).getTable().equals(table)) count.incrementAndGet();
300        return true;
301      }
302    };
303    MetaTableAccessor.fullScanRegions(connection, visitor);
304    assert(count.get() == numRegions);
305  }

307  private void validateFromSnapshotFromMeta(HBaseTestingUtility util, TableName table,
308      int numRegions, int numReplica, Connection connection) throws IOException {
309    SnapshotOfRegionAssignmentFromMeta snapshot = new SnapshotOfRegionAssignmentFromMeta(
310      connection);
311    snapshot.initialize();
312    Map<HRegionInfo, ServerName> regionToServerMap = snapshot.getRegionToRegionServerMap();
313    assert(regionToServerMap.size() == numRegions * numReplica + 1); //'1' for the namespace
314    Map<ServerName, List<HRegionInfo>> serverToRegionMap = snapshot.getRegionServerToRegionMap();
315    for (Map.Entry<ServerName, List<HRegionInfo>> entry : serverToRegionMap.entrySet()) {
316      if (entry.getKey().equals(util.getHBaseCluster().getMaster().getServerName())) {
317        continue;
318      }
319      List<HRegionInfo> regions = entry.getValue();
320      Set<byte[]> setOfStartKeys = new HashSet<byte[]>();
321      for (HRegionInfo region : regions) {
322        byte[] startKey = region.getStartKey();
323        if (region.getTable().equals(table)) {
324          setOfStartKeys.add(startKey); //ignore other tables
325          LOG.info("--STARTKEY " + new String(startKey)+"--");
326        }
327      }
328      // the number of startkeys will be equal to the number of regions hosted in each server
329      // (each server will be hosting one replica of a region)
330      assertEquals(numRegions, setOfStartKeys.size());
331    }
332  }

334  private void validateSingleRegionServerAssignment(Connection connection, int numRegions,
335      int numReplica) throws IOException {
336    SnapshotOfRegionAssignmentFromMeta snapshot = new SnapshotOfRegionAssignmentFromMeta(
337      connection);
338    snapshot.initialize();
339    Map<HRegionInfo, ServerName>  regionToServerMap = snapshot.getRegionToRegionServerMap();
340    assertEquals(regionToServerMap.size(), numRegions * numReplica + 1); //'1' for the namespace
341    Map<ServerName, List<HRegionInfo>> serverToRegionMap = snapshot.getRegionServerToRegionMap();
342    assertEquals(serverToRegionMap.keySet().size(), 2); // 1 rs + 1 master
343    for (Map.Entry<ServerName, List<HRegionInfo>> entry : serverToRegionMap.entrySet()) {
344      if (entry.getKey().equals(TEST_UTIL.getHBaseCluster().getMaster().getServerName())) {
345        continue;
346      }
347      assertEquals(entry.getValue().size(), numRegions * numReplica);
348    }
