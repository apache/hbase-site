070  public static class MyMaster extends HMaster {
071
072    public MyMaster(Configuration conf) throws IOException, KeeperException, InterruptedException {
073      super(conf);
074    }
075
076    @Override
077    protected void tryRegionServerReport(long reportStartTime, long reportEndTime) {
078      // do nothing
079    }
080
081    @Override
082    protected RSRpcServices createRpcServices() throws IOException {
083      return new MasterRpcServices(this) {
084
085        @Override
086        public RegionServerStartupResponse regionServerStartup(RpcController controller,
087          RegionServerStartupRequest request) throws ServiceException {
088          RegionServerStartupResponse resp = super.regionServerStartup(controller, request);
089          ServerManager serverManager = getServerManager();
090          // to let the region server actual online otherwise we can not assign meta region
091          new HashMap<>(serverManager.getOnlineServers()).forEach((sn, sm) -> {
092            if (sm.getLastReportTimestamp() <= 0) {
093              try {
094                serverManager.regionServerReport(sn,
095                  ServerMetricsBuilder.newBuilder(sn).setVersionNumber(sm.getVersionNumber())
096                    .setVersion(sm.getVersion())
097                      .setLastReportTimestamp(EnvironmentEdgeManager.currentTime())
098                    .build());
099              } catch (YouAreDeadException e) {
100                throw new UncheckedIOException(e);
101              }
102            }
103          });
104          return resp;
105        }
106      };
107    }
108  }

141  @Test
142  public void testClusterRequests() throws Exception {
143    // sending fake request to master to see how metric value has changed
144    RegionServerStatusProtos.RegionServerReportRequest.Builder request =
145      RegionServerStatusProtos.RegionServerReportRequest.newBuilder();
146    ServerName serverName = cluster.getMaster(0).getServerName();
147    request.setServer(ProtobufUtil.toServerName(serverName));
148    long expectedRequestNumber = 10000;
149
150    MetricsMasterSource masterSource = master.getMasterMetrics().getMetricsSource();
151    ClusterStatusProtos.ServerLoad sl = ClusterStatusProtos.ServerLoad.newBuilder()
152       .setTotalNumberOfRequests(expectedRequestNumber)
153       .setReadRequestsCount(expectedRequestNumber)
154       .setWriteRequestsCount(expectedRequestNumber).build();
155    request.setLoad(sl);
156
157    master.getMasterRpcServices().regionServerReport(null, request.build());
158    metricsHelper.assertCounter("cluster_requests", expectedRequestNumber, masterSource);
159    metricsHelper.assertCounter("cluster_read_requests", expectedRequestNumber, masterSource);
160    metricsHelper.assertCounter("cluster_write_requests", expectedRequestNumber, masterSource);
161
162    expectedRequestNumber = 15000;
163
164    sl = ClusterStatusProtos.ServerLoad.newBuilder()
165            .setTotalNumberOfRequests(expectedRequestNumber)
166            .setReadRequestsCount(expectedRequestNumber)
167            .setWriteRequestsCount(expectedRequestNumber).build();
168    request.setLoad(sl);
169
170    master.getMasterRpcServices().regionServerReport(null, request.build());
171    metricsHelper.assertCounter("cluster_requests", expectedRequestNumber, masterSource);
172    metricsHelper.assertCounter("cluster_read_requests", expectedRequestNumber, masterSource);
173    metricsHelper.assertCounter("cluster_write_requests", expectedRequestNumber, masterSource);
174  }

176  @Test
177  public void testDefaultMasterMetrics() throws Exception {
178    MetricsMasterSource masterSource = master.getMasterMetrics().getMetricsSource();
179    metricsHelper.assertGauge("numRegionServers", 1, masterSource);
180    metricsHelper.assertGauge("averageLoad", 1, masterSource);
181    metricsHelper.assertGauge("numDeadRegionServers", 0, masterSource);
182    metricsHelper.assertGauge("numDrainingRegionServers", 0, masterSource);
183
184    metricsHelper.assertGauge("masterStartTime", master.getMasterStartTime(), masterSource);
185    metricsHelper.assertGauge("masterActiveTime", master.getMasterActiveTime(), masterSource);
186
187    metricsHelper.assertTag("isActiveMaster", "true", masterSource);
188    metricsHelper.assertTag("serverName", master.getServerName().toString(), masterSource);
189    metricsHelper.assertTag("clusterId", master.getClusterId(), masterSource);
190    metricsHelper.assertTag("zookeeperQuorum", master.getZooKeeper().getQuorum(), masterSource);
191
192    metricsHelper.assertCounter(MetricsMasterSource.SERVER_CRASH_METRIC_PREFIX+"SubmittedCount",
193      0, masterSource);
194  }
