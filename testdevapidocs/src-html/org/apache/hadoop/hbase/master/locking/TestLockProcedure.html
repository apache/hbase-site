188  /**
189   * Returns immediately if the lock is acquired.
190   * @throws TimeoutException if lock couldn't be acquired.
191   */
192  private boolean awaitForLocked(long procId, long timeoutInMs) throws Exception {
193    long deadline = System.currentTimeMillis() + timeoutInMs;
194    while (System.currentTimeMillis() < deadline) {
195      LockHeartbeatResponse response = masterRpcService.lockHeartbeat(null,
196          LockHeartbeatRequest.newBuilder().setProcId(procId).build());
197      if (response.getLockStatus() == LockHeartbeatResponse.LockStatus.LOCKED) {
198        assertEquals(response.getTimeoutMs(), HEARTBEAT_TIMEOUT);
199        LOG.debug(String.format("Proc id %s acquired lock.", procId));
200        return true;
201      }
202      Thread.sleep(100);
203    }
204    return false;
205  }
206
207  private long queueLock(LockRequest lockRequest) throws ServiceException {
208    LockResponse response = masterRpcService.requestLock(null, lockRequest);
209    return response.getProcId();
210  }
211
212  private void sendHeartbeatAndCheckLocked(long procId, boolean isLocked) throws ServiceException {
213    LockHeartbeatResponse response = masterRpcService.lockHeartbeat(null,
214        LockHeartbeatRequest.newBuilder().setProcId(procId).build());
215    if (isLocked) {
216      assertEquals(LockHeartbeatResponse.LockStatus.LOCKED, response.getLockStatus());
217    } else {
218      assertEquals(LockHeartbeatResponse.LockStatus.UNLOCKED, response.getLockStatus());
219    }
220    LOG.debug(String.format("Proc id %s : %s.", procId, response.getLockStatus()));
221  }
222
223  private void releaseLock(long procId) throws ServiceException {
224    masterRpcService.lockHeartbeat(null,
225        LockHeartbeatRequest.newBuilder().setProcId(procId).setKeepAlive(false).build());
226  }

288  @Test
289  public void testMultipleLocks() throws Exception {
290    LockRequest nsLock = getNamespaceLock(namespace, testMethodName);
291    LockRequest tableLock1 = getTableExclusiveLock(tableName1, testMethodName);
292    LockRequest tableLock2 =  getTableExclusiveLock(tableName2, testMethodName);
293    LockRequest regionsLock1 = getRegionLock(tableRegions1, testMethodName);
294    LockRequest regionsLock2 = getRegionLock(tableRegions2, testMethodName);
295    // Acquire namespace lock, then queue other locks.
296    long nsProcId = queueLock(nsLock);
297    assertTrue(awaitForLocked(nsProcId, 2000));
298    sendHeartbeatAndCheckLocked(nsProcId, true);
299    long table1ProcId = queueLock(tableLock1);
300    long table2ProcId = queueLock(tableLock2);
301    long regions1ProcId = queueLock(regionsLock1);
302    long regions2ProcId = queueLock(regionsLock2);
303
304    // Assert tables & region locks are waiting because of namespace lock.
305    Thread.sleep(HEARTBEAT_TIMEOUT / 2);
306    sendHeartbeatAndCheckLocked(nsProcId, true);
307    sendHeartbeatAndCheckLocked(table1ProcId, false);
308    sendHeartbeatAndCheckLocked(table2ProcId, false);
309    sendHeartbeatAndCheckLocked(regions1ProcId, false);
310    sendHeartbeatAndCheckLocked(regions2ProcId, false);
311
312    // Release namespace lock and assert tables locks are acquired but not region lock
313    releaseLock(nsProcId);
314    assertTrue(awaitForLocked(table1ProcId, 2000));
315    assertTrue(awaitForLocked(table2ProcId, 2000));
316    sendHeartbeatAndCheckLocked(regions1ProcId, false);
317    sendHeartbeatAndCheckLocked(regions2ProcId, false);
318
319    // Release table1 lock and assert region lock is acquired.
320    releaseLock(table1ProcId);
321    sendHeartbeatAndCheckLocked(table1ProcId, false);
322    assertTrue(awaitForLocked(regions1ProcId, 2000));
323    sendHeartbeatAndCheckLocked(table2ProcId, true);
324    sendHeartbeatAndCheckLocked(regions2ProcId, false);
325
326    // Release table2 lock and assert region lock is acquired.
327    releaseLock(table2ProcId);
328    sendHeartbeatAndCheckLocked(table2ProcId, false);
329    assertTrue(awaitForLocked(regions2ProcId, 2000));
330    sendHeartbeatAndCheckLocked(regions1ProcId, true);
331    sendHeartbeatAndCheckLocked(regions2ProcId, true);
332
333    // Release region locks.
334    releaseLock(regions1ProcId);
335    releaseLock(regions2ProcId);
336    sendHeartbeatAndCheckLocked(regions1ProcId, false);
337    sendHeartbeatAndCheckLocked(regions2ProcId, false);
338    ProcedureTestingUtility.waitAllProcedures(procExec);
339    ProcedureTestingUtility.assertProcNotFailed(procExec, nsProcId);
340    ProcedureTestingUtility.assertProcNotFailed(procExec, table1ProcId);
341    ProcedureTestingUtility.assertProcNotFailed(procExec, table2ProcId);
342    ProcedureTestingUtility.assertProcNotFailed(procExec, regions1ProcId);
343    ProcedureTestingUtility.assertProcNotFailed(procExec, regions2ProcId);
344  }

352        TableName.valueOf("table"), org.apache.hadoop.hbase.procedure2.LockType.EXCLUSIVE, "desc", latch);

357    ProcedureTestingUtility.assertProcNotFailed(procExec, lockProc.getProcId());

360  // LockProcedures with latch are considered local locks.
361  @Test
362  public void testLocalLockTimeout() throws Exception {
363    CountDownLatch latch = new CountDownLatch(1);
364    // MasterRpcServices don't set latch with LockProcedure, so create one and submit it directly.
365    LockProcedure lockProc = new LockProcedure(UTIL.getConfiguration(),
366        TableName.valueOf("table"), LockType.EXCLUSIVE, "desc", latch);
367    procExec.submitProcedure(lockProc);
368    assertTrue(awaitForLocked(lockProc.getProcId(), 2000));
369    Thread.sleep(LOCAL_LOCKS_TIMEOUT / 2);
370    assertTrue(lockProc.isLocked());
371    Thread.sleep(2 * LOCAL_LOCKS_TIMEOUT);
372    assertFalse(lockProc.isLocked());
373    releaseLock(lockProc.getProcId());
374    ProcedureTestingUtility.waitProcedure(procExec, lockProc.getProcId());
375    ProcedureTestingUtility.assertProcNotFailed(procExec, lockProc.getProcId());
376  }

378  private void testRemoteLockRecovery(LockRequest lock) throws Exception {
379    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
380    final long procId = queueLock(lock);
381    assertTrue(awaitForLocked(procId, 2000));
382
383    // wait for proc Executor to die, then restart it and wait for Lock Procedure to get started.
384    ProcedureTestingUtility.waitProcedure(procExec, procId);
385    assertEquals(false, procExec.isRunning());
386    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
387    ProcedureTestingUtility.restart(procExec);
388    while (!procExec.isStarted(procId)) {
389      Thread.sleep(250);
390    }
391    assertEquals(true, procExec.isRunning());
392
393    // After recovery, remote locks should reacquire locks and function normally.
394    assertTrue(awaitForLocked(procId, 2000));
395    Thread.sleep(HEARTBEAT_TIMEOUT/2);
396    sendHeartbeatAndCheckLocked(procId, true);
397    Thread.sleep(HEARTBEAT_TIMEOUT/2);
398    sendHeartbeatAndCheckLocked(procId, true);
399    Thread.sleep(2 * HEARTBEAT_TIMEOUT);
400    sendHeartbeatAndCheckLocked(procId, false);
401    ProcedureTestingUtility.waitProcedure(procExec, procId);
402    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
403  }

423  @Test (timeout = 20000)
424  public void testLocalMasterLockRecovery() throws Exception {
425    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
426    CountDownLatch latch = new CountDownLatch(1);
427    LockProcedure lockProc = new LockProcedure(UTIL.getConfiguration(),
428        TableName.valueOf("table"), LockType.EXCLUSIVE, "desc", latch);
429    procExec.submitProcedure(lockProc);
430    assertTrue(latch.await(2000, TimeUnit.MILLISECONDS));
431
432    // wait for proc Executor to die, then restart it and wait for Lock Procedure to get started.
433    ProcedureTestingUtility.waitProcedure(procExec, lockProc.getProcId());
434    assertEquals(false, procExec.isRunning());
435    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
436    // remove zk lock node otherwise recovered lock will keep waiting on it.
437    ProcedureTestingUtility.restart(procExec);
438    while (!procExec.isStarted(lockProc.getProcId())) {
439      Thread.sleep(250);
440    }
441    assertEquals(true, procExec.isRunning());
442    ProcedureTestingUtility.waitProcedure(procExec, lockProc.getProcId());
443    Procedure<?> result = procExec.getResultOrProcedure(lockProc.getProcId());
444    assertTrue(result != null && !result.isFailed());
445    ProcedureTestingUtility.assertProcNotFailed(procExec, lockProc.getProcId());
446  }
