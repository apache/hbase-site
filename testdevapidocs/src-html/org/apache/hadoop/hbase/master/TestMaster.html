103  /**
104   * Return the region and current deployment for the region containing the given row. If the region
105   * cannot be found, returns null. If it is found, but not currently deployed, the second element
106   * of the pair may be null.
107   */
108  private Pair<RegionInfo, ServerName> getTableRegionForRow(HMaster master, TableName tableName,
109      byte[] rowKey) throws IOException {
110    final AtomicReference<Pair<RegionInfo, ServerName>> result = new AtomicReference<>(null);
111
112    ClientMetaTableAccessor.Visitor visitor = new ClientMetaTableAccessor.Visitor() {
113      @Override
114      public boolean visit(Result data) throws IOException {
115        if (data == null || data.size() <= 0) {
116          return true;
117        }
118        Pair<RegionInfo, ServerName> pair = new Pair<>(CatalogFamilyFormat.getRegionInfo(data),
119          CatalogFamilyFormat.getServerName(data, 0));
120        if (!pair.getFirst().getTable().equals(tableName)) {
121          return false;
122        }
123        result.set(pair);
124        return true;
125      }
126    };
127
128    MetaTableAccessor.scanMeta(master.getConnection(), visitor, tableName, rowKey, 1);
129    return result.get();
130  }

132  @Test
133  @SuppressWarnings("deprecation")
134  public void testMasterOpsWhileSplitting() throws Exception {
135    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
136    HMaster m = cluster.getMaster();
137
138    try (Table ht = TEST_UTIL.createTable(TABLENAME, FAMILYNAME)) {
139      assertTrue(m.getTableStateManager().isTableState(TABLENAME, TableState.State.ENABLED));
140      TEST_UTIL.loadTable(ht, FAMILYNAME, false);
141    }
142
143    List<Pair<RegionInfo, ServerName>> tableRegions = MetaTableAccessor.getTableRegionsAndLocations(
144        m.getConnection(), TABLENAME);
145    LOG.info("Regions after load: " + Joiner.on(',').join(tableRegions));
146    assertEquals(1, tableRegions.size());
147    assertArrayEquals(HConstants.EMPTY_START_ROW,
148        tableRegions.get(0).getFirst().getStartKey());
149    assertArrayEquals(HConstants.EMPTY_END_ROW,
150        tableRegions.get(0).getFirst().getEndKey());
151
152    // Now trigger a split and stop when the split is in progress
153    LOG.info("Splitting table");
154    TEST_UTIL.getAdmin().split(TABLENAME);
155
156    LOG.info("Making sure we can call getTableRegions while opening");
157    while (tableRegions.size() < 3) {
158      tableRegions = MetaTableAccessor.getTableRegionsAndLocations(m.getConnection(),
159          TABLENAME, false);
160      Thread.sleep(100);
161    }
162    LOG.info("Regions: " + Joiner.on(',').join(tableRegions));
163    // We have three regions because one is split-in-progress
164    assertEquals(3, tableRegions.size());
165    LOG.info("Making sure we can call getTableRegionClosest while opening");
166    Pair<RegionInfo, ServerName> pair = getTableRegionForRow(m, TABLENAME, Bytes.toBytes("cde"));
167    LOG.info("Result is: " + pair);
168    Pair<RegionInfo, ServerName> tableRegionFromName =
169        MetaTableAccessor.getRegion(m.getConnection(),
170          pair.getFirst().getRegionName());
171    assertTrue(RegionInfo.COMPARATOR.compare(tableRegionFromName.getFirst(), pair.getFirst()) == 0);
172  }

239  @Test
240  public void testFlushedSequenceIdPersistLoad() throws Exception {
241    Configuration conf = TEST_UTIL.getConfiguration();
242    int msgInterval = conf.getInt("hbase.regionserver.msginterval", 100);
243    // insert some data into META
244    TableName tableName = TableName.valueOf("testFlushSeqId");
245    TableDescriptor tableDescriptor = TableDescriptorBuilder.newBuilder(tableName)
246      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(Bytes.toBytes("cf"))).build();
247    Table table = TEST_UTIL.createTable(tableDescriptor, null);
248    // flush META region
249    TEST_UTIL.flush(TableName.META_TABLE_NAME);
250    // wait for regionserver report
251    Threads.sleep(msgInterval * 2);
252    // record flush seqid before cluster shutdown
253    Map<byte[], Long> regionMapBefore =
254        TEST_UTIL.getHBaseCluster().getMaster().getServerManager()
255            .getFlushedSequenceIdByRegion();
256    // restart hbase cluster which will cause flushed sequence id persist and reload
257    TEST_UTIL.getMiniHBaseCluster().shutdown();
258    TEST_UTIL.restartHBaseCluster(2);
259    TEST_UTIL.waitUntilNoRegionsInTransition();
260    // check equality after reloading flushed sequence id map
261    Map<byte[], Long> regionMapAfter =
262        TEST_UTIL.getHBaseCluster().getMaster().getServerManager()
263            .getFlushedSequenceIdByRegion();
264    assertTrue(regionMapBefore.equals(regionMapAfter));
265  }

267  @Test
268  public void testBlockingHbkc1WithLockFile() throws IOException {
269    // This is how the patch to the lock file is created inside in HBaseFsck. Too hard to use its
270    // actual method without disturbing HBaseFsck... Do the below mimic instead.
271    Path hbckLockPath = new Path(HBaseFsck.getTmpDir(TEST_UTIL.getConfiguration()),
272        HBaseFsck.HBCK_LOCK_FILE);
273    FileSystem fs = TEST_UTIL.getTestFileSystem();
274    assertTrue(fs.exists(hbckLockPath));
275    TEST_UTIL.getMiniHBaseCluster().
276        killMaster(TEST_UTIL.getMiniHBaseCluster().getMaster().getServerName());
277    assertTrue(fs.exists(hbckLockPath));
278    TEST_UTIL.getMiniHBaseCluster().startMaster();
279    TEST_UTIL.waitFor(30000, () -> TEST_UTIL.getMiniHBaseCluster().getMaster() != null &&
280        TEST_UTIL.getMiniHBaseCluster().getMaster().isInitialized());
281    assertTrue(fs.exists(hbckLockPath));
282    // Start a second Master. Should be fine.
283    TEST_UTIL.getMiniHBaseCluster().startMaster();
284    assertTrue(fs.exists(hbckLockPath));
285    fs.delete(hbckLockPath, true);
286    assertFalse(fs.exists(hbckLockPath));
287    // Kill all Masters.
288    TEST_UTIL.getMiniHBaseCluster().getLiveMasterThreads().stream().
289        map(sn -> sn.getMaster().getServerName()).forEach(sn -> {
290          try {
291            TEST_UTIL.getMiniHBaseCluster().killMaster(sn);
292          } catch (IOException e) {
293            e.printStackTrace();
294          }
295        });
296    // Start a new one.
297    TEST_UTIL.getMiniHBaseCluster().startMaster();
298    TEST_UTIL.waitFor(30000, () -> TEST_UTIL.getMiniHBaseCluster().getMaster() != null &&
299        TEST_UTIL.getMiniHBaseCluster().getMaster().isInitialized());
300    // Assert lock gets put in place again.
301    assertTrue(fs.exists(hbckLockPath));
302  }
