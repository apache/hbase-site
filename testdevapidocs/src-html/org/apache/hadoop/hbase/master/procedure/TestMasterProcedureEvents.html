087  @Test(timeout = 30000)
088  public void testMasterInitializedEvent() throws Exception {
089    final TableName tableName = TableName.valueOf(name.getMethodName());
090    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
091    ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
092
093    HRegionInfo hri = new HRegionInfo(tableName);
094    HTableDescriptor htd = new HTableDescriptor(tableName);
095    htd.addFamily(new HColumnDescriptor("f"));
096
097    while (!master.isInitialized()) Thread.sleep(250);
098    master.setInitialized(false); // fake it, set back later
099
100    // check event wait/wake
101    testProcedureEventWaitWake(master, master.getInitializedEvent(),
102      new CreateTableProcedure(procExec.getEnvironment(), htd, new HRegionInfo[] { hri }));
103  }

105  @Test(timeout = 30000)
106  public void testServerCrashProcedureEvent() throws Exception {
107    final TableName tableName = TableName.valueOf(name.getMethodName());
108    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
109    ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
110
111    while (!master.isServerCrashProcessingEnabled() || !master.isInitialized() ||
112        master.getAssignmentManager().getRegionStates().hasRegionsInTransition()) {
113      Thread.sleep(25);
114    }
115
116    UTIL.createTable(tableName, HBaseTestingUtility.COLUMNS[0]);
117    try (Table t = UTIL.getConnection().getTable(tableName)) {
118      // Load the table with a bit of data so some logs to split and some edits in each region.
119      UTIL.loadTable(t, HBaseTestingUtility.COLUMNS[0]);
120    }
121
122    master.setServerCrashProcessingEnabled(false);  // fake it, set back later
123
124    // Kill a server. Master will notice but do nothing other than add it to list of dead servers.
125    HRegionServer hrs = getServerWithRegions();
126    boolean carryingMeta = master.getAssignmentManager().isCarryingMeta(hrs.getServerName());
127    UTIL.getHBaseCluster().killRegionServer(hrs.getServerName());
128    hrs.join();
129
130    // Wait until the expiration of the server has arrived at the master. We won't process it
131    // by queuing a ServerCrashProcedure because we have disabled crash processing... but wait
132    // here so ServerManager gets notice and adds expired server to appropriate queues.
133    while (!master.getServerManager().isServerDead(hrs.getServerName())) Thread.sleep(10);
134
135    // Do some of the master processing of dead servers so when SCP runs, it has expected 'state'.
136    master.getServerManager().moveFromOnlineToDeadServers(hrs.getServerName());
137
138    // check event wait/wake
139    testProcedureEventWaitWake(master, master.getServerCrashProcessingEnabledEvent(),
140      new ServerCrashProcedure(procExec.getEnvironment(), hrs.getServerName(), true, carryingMeta));
141  }
142
143  private void testProcedureEventWaitWake(final HMaster master, final ProcedureEvent event,
144      final Procedure proc) throws Exception {
145    final ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
146    final MasterProcedureScheduler procSched = procExec.getEnvironment().getProcedureScheduler();
147
148    final long startPollCalls = procSched.getPollCalls();
149    final long startNullPollCalls = procSched.getNullPollCalls();
150
151    // check that nothing is in the event queue
152    LOG.debug("checking " + event);
153    assertEquals(false, event.isReady());
154    assertEquals(0, event.getSuspendedProcedures().size());
155
156    // submit the procedure
157    LOG.debug("submit " + proc);
158    long procId = procExec.submitProcedure(proc);
159
160    // wait until the event is in the queue (proc executed and got into suspended state)
161    LOG.debug("wait procedure suspended on " + event);
162    while (event.getSuspendedProcedures().size() < 1) Thread.sleep(25);
163
164    // check that the proc is in the event queue
165    LOG.debug("checking " + event + " size=" + event.getSuspendedProcedures().size());
166    assertEquals(false, event.isReady());
167    assertEquals(1, event.getSuspendedProcedures().size());
168
169    // wake the event
170    LOG.debug("wake " + event);
171    procSched.wakeEvent(event);
172    assertEquals(true, event.isReady());
173
174    // wait until proc completes
175    LOG.debug("waiting " + proc);
176    ProcedureTestingUtility.waitProcedure(procExec, procId);
177
178    // check that nothing is in the event queue and the event is not suspended
179    assertEquals(true, event.isReady());
180    assertEquals(0, event.getSuspendedProcedures().size());
181    LOG.debug("completed execution of " + proc +
182      " pollCalls=" + (procSched.getPollCalls() - startPollCalls) +
183      " nullPollCalls=" + (procSched.getNullPollCalls() - startNullPollCalls));
184  }
185
186  private HRegionServer getServerWithRegions() {
187    for (int i = 0; i < 3; ++i) {
188      HRegionServer hrs = UTIL.getHBaseCluster().getRegionServer(i);
189      if (hrs.getNumberOfOnlineRegions() > 0) {
190        return hrs;
191      }
192    }
193    return null;
194  }
