092  @Test(timeout = 60000)
093  public void testModifyNamespace() throws Exception {
094    final NamespaceDescriptor nsd = NamespaceDescriptor.create("testModifyNamespace").build();
095    final String nsKey1 = "hbase.namespace.quota.maxregions";
096    final String nsValue1before = "1111";
097    final String nsValue1after = "9999";
098    final String nsKey2 = "hbase.namespace.quota.maxtables";
099    final String nsValue2 = "10";
100    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
101
102    nsd.setConfiguration(nsKey1, nsValue1before);
103    createNamespaceForTesting(nsd);
104
105    // Before modify
106    NamespaceDescriptor currentNsDescriptor =
107        UTIL.getHBaseAdmin().getNamespaceDescriptor(nsd.getName());
108    assertEquals(currentNsDescriptor.getConfigurationValue(nsKey1), nsValue1before);
109    assertNull(currentNsDescriptor.getConfigurationValue(nsKey2));
110
111    // Update
112    nsd.setConfiguration(nsKey1, nsValue1after);
113    nsd.setConfiguration(nsKey2, nsValue2);
114
115    long procId1 = procExec.submitProcedure(
116      new ModifyNamespaceProcedure(procExec.getEnvironment(), nsd),
117      nonceGroup,
118      nonce);
119    // Wait the completion
120    ProcedureTestingUtility.waitProcedure(procExec, procId1);
121    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
122
123    // Verify the namespace is updated.
124    currentNsDescriptor =
125        UTIL.getHBaseAdmin().getNamespaceDescriptor(nsd.getName());
126    assertEquals(nsd.getConfigurationValue(nsKey1), nsValue1after);
127    assertEquals(currentNsDescriptor.getConfigurationValue(nsKey2), nsValue2);
128  }

130  @Test(timeout=60000)
131  public void testModifyNonExistNamespace() throws Exception {
132    final String namespaceName = "testModifyNonExistNamespace";
133    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
134
135    try {
136      NamespaceDescriptor nsDescriptor = UTIL.getHBaseAdmin().getNamespaceDescriptor(namespaceName);
137      assertNull(nsDescriptor);
138    } catch (NamespaceNotFoundException nsnfe) {
139      // Expected
140      LOG.debug("The namespace " + namespaceName + " does not exist.  This is expected.");
141    }
142
143    final NamespaceDescriptor nsd = NamespaceDescriptor.create(namespaceName).build();
144
145    long procId = procExec.submitProcedure(
146      new ModifyNamespaceProcedure(procExec.getEnvironment(), nsd),
147      nonceGroup,
148      nonce);
149    // Wait the completion
150    ProcedureTestingUtility.waitProcedure(procExec, procId);
151
152    // Expect fail with NamespaceNotFoundException
153    ProcedureInfo result = procExec.getResult(procId);
154    assertTrue(result.isFailed());
155    LOG.debug("modify namespace failed with exception: " + result.getExceptionFullMessage());
156    assertTrue(
157      ProcedureTestingUtility.getExceptionCause(result) instanceof NamespaceNotFoundException);
158  }

210  @Test(timeout = 60000)
211  public void testRecoveryAndDoubleExecution() throws Exception {
212    final NamespaceDescriptor nsd =
213        NamespaceDescriptor.create("testRecoveryAndDoubleExecution").build();
214    final String nsKey = "foo";
215    final String nsValue = "bar";
216    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
217
218    createNamespaceForTesting(nsd);
219    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
220    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
221
222    // Modify
223    nsd.setConfiguration(nsKey, nsValue);
224
225    // Start the Modify procedure && kill the executor
226    long procId = procExec.submitProcedure(
227      new ModifyNamespaceProcedure(procExec.getEnvironment(), nsd),
228      nonceGroup,
229      nonce);
230
231    // Restart the executor and execute the step twice
232    int numberOfSteps = ModifyNamespaceState.values().length;
233    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId, numberOfSteps);
234
235    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
236    // Validate
237    NamespaceDescriptor currentNsDescriptor =
238        UTIL.getHBaseAdmin().getNamespaceDescriptor(nsd.getName());
239    assertEquals(currentNsDescriptor.getConfigurationValue(nsKey), nsValue);
240  }
