041@Category({MasterTests.class, MediumTests.class})
042public class TestTruncateTableProcedure extends TestTableDDLProcedureBase {
043  private static final Log LOG = LogFactory.getLog(TestTruncateTableProcedure.class);
044
045  @Rule
046  public TestName name = new TestName();
047
048  @Test(timeout=60000)
049  public void testTruncateNotExistentTable() throws Exception {
050    final TableName tableName = TableName.valueOf(name.getMethodName());
051
052    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
053    long procId = ProcedureTestingUtility.submitAndWait(procExec,
054        new TruncateTableProcedure(procExec.getEnvironment(), tableName, true));
055
056    // Second delete should fail with TableNotFound
057    ProcedureInfo result = procExec.getResult(procId);
058    assertTrue(result.isFailed());
059    LOG.debug("Truncate failed with exception: " + result.getExceptionFullMessage());
060    assertTrue(ProcedureTestingUtility.getExceptionCause(result) instanceof TableNotFoundException);
061  }
062
063  @Test(timeout=60000)
064  public void testTruncateNotDisabledTable() throws Exception {
065    final TableName tableName = TableName.valueOf(name.getMethodName());
066
067    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
068    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f");
069
070    long procId = ProcedureTestingUtility.submitAndWait(procExec,
071        new TruncateTableProcedure(procExec.getEnvironment(), tableName, false));
072
073    // Second delete should fail with TableNotDisabled
074    ProcedureInfo result = procExec.getResult(procId);
075    assertTrue(result.isFailed());
076    LOG.debug("Truncate failed with exception: " + result.getExceptionFullMessage());
077    assertTrue(
078      ProcedureTestingUtility.getExceptionCause(result) instanceof TableNotDisabledException);
079  }

081  @Test(timeout=60000)
082  public void testSimpleTruncatePreserveSplits() throws Exception {
083    final TableName tableName = TableName.valueOf(name.getMethodName());
084    testSimpleTruncate(tableName, true);
085  }

087  @Test(timeout=60000)
088  public void testSimpleTruncateNoPreserveSplits() throws Exception {
089    final TableName tableName = TableName.valueOf(name.getMethodName());
090    testSimpleTruncate(tableName, false);
091  }
092
093  private void testSimpleTruncate(final TableName tableName, final boolean preserveSplits)
094      throws Exception {
095    final String[] families = new String[] { "f1", "f2" };
096    final byte[][] splitKeys = new byte[][] {
097      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
098    };
099
100    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
101      getMasterProcedureExecutor(), tableName, splitKeys, families);
102    // load and verify that there are rows in the table
103    MasterProcedureTestingUtility.loadData(
104      UTIL.getConnection(), tableName, 100, splitKeys, families);
105    assertEquals(100, UTIL.countRows(tableName));
106    // disable the table
107    UTIL.getAdmin().disableTable(tableName);
108
109    // truncate the table
110    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
111    long procId = ProcedureTestingUtility.submitAndWait(procExec,
112      new TruncateTableProcedure(procExec.getEnvironment(), tableName, preserveSplits));
113    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
114
115    UTIL.waitUntilAllRegionsAssigned(tableName);
116
117    // validate the table regions and layout
118    regions = UTIL.getAdmin().getTableRegions(tableName).toArray(new HRegionInfo[0]);
119    if (preserveSplits) {
120      assertEquals(1 + splitKeys.length, regions.length);
121    } else {
122      assertEquals(1, regions.length);
123    }
124    MasterProcedureTestingUtility.validateTableCreation(
125      UTIL.getHBaseCluster().getMaster(), tableName, regions, families);
126
127    // verify that there are no rows in the table
128    assertEquals(0, UTIL.countRows(tableName));
129
130    // verify that the table is read/writable
131    MasterProcedureTestingUtility.loadData(
132      UTIL.getConnection(), tableName, 50, splitKeys, families);
133    assertEquals(50, UTIL.countRows(tableName));
134  }

148  private void testRecoveryAndDoubleExecution(final TableName tableName,
149      final boolean preserveSplits) throws Exception {
150    final String[] families = new String[] { "f1", "f2" };
151
152    // create the table
153    final byte[][] splitKeys = new byte[][] {
154      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
155    };
156    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
157      getMasterProcedureExecutor(), tableName, splitKeys, families);
158    // load and verify that there are rows in the table
159    MasterProcedureTestingUtility.loadData(
160      UTIL.getConnection(), tableName, 100, splitKeys, families);
161    assertEquals(100, UTIL.countRows(tableName));
162    // disable the table
163    UTIL.getAdmin().disableTable(tableName);
164
165    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
166    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
167    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
168
169    // Start the Truncate procedure && kill the executor
170    long procId = procExec.submitProcedure(
171      new TruncateTableProcedure(procExec.getEnvironment(), tableName, preserveSplits));
172
173    // Restart the executor and execute the step twice
174    // NOTE: the 7 (number of TruncateTableState steps) is hardcoded,
175    //       so you have to look at this test at least once when you add a new step.
176    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId, 7);
177
178    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
179    UTIL.waitUntilAllRegionsAssigned(tableName);
180
181    // validate the table regions and layout
182    regions = UTIL.getAdmin().getTableRegions(tableName).toArray(new HRegionInfo[0]);
183    if (preserveSplits) {
184      assertEquals(1 + splitKeys.length, regions.length);
185    } else {
186      assertEquals(1, regions.length);
187    }
188    MasterProcedureTestingUtility.validateTableCreation(
189      UTIL.getHBaseCluster().getMaster(), tableName, regions, families);
190
191    // verify that there are no rows in the table
192    assertEquals(0, UTIL.countRows(tableName));
193
194    // verify that the table is read/writable
195    MasterProcedureTestingUtility.loadData(
196      UTIL.getConnection(), tableName, 50, splitKeys, families);
197    assertEquals(50, UTIL.countRows(tableName));
198  }
