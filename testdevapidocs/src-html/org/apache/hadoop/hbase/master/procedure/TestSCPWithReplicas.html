011package org.apache.hadoop.hbase.master.procedure;
012
013import static org.junit.Assert.fail;
014
015import java.io.IOException;
016import java.util.ArrayList;
017import java.util.List;
018
019import org.apache.hadoop.hbase.HBaseClassTestRule;
020import org.apache.hadoop.hbase.HBaseTestingUtility;
021import org.apache.hadoop.hbase.TableName;
022import org.apache.hadoop.hbase.client.RegionInfo;
023import org.apache.hadoop.hbase.client.RegionReplicaUtil;
024import org.apache.hadoop.hbase.client.Table;
025import org.apache.hadoop.hbase.regionserver.Region;
026import org.apache.hadoop.hbase.testclassification.LargeTests;
027import org.apache.hadoop.hbase.testclassification.MasterTests;
028import org.apache.hadoop.hbase.util.JVMClusterUtil.RegionServerThread;
029import org.junit.ClassRule;
030import org.junit.experimental.categories.Category;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034@Category({ MasterTests.class, LargeTests.class })
035public class TestSCPWithReplicas extends TestSCP {
036
037  @ClassRule
038  public static final HBaseClassTestRule CLASS_RULE =
039      HBaseClassTestRule.forClass(TestSCPWithReplicas.class);
040  private static final Logger LOG = LoggerFactory.getLogger(TestSCPWithReplicas.class);
041
042  @Override
043  protected void startMiniCluster() throws Exception {
044    // Start a cluster with 4 nodes because we have 3 replicas.
045    // So on a crash of a server still we can ensure that the
046    // replicas are distributed.
047    this.util.startMiniCluster(4);
048  }
049
050  @Override
051  protected Table createTable(final TableName tableName) throws IOException {
052    final Table t = this.util.createTable(tableName, HBaseTestingUtility.COLUMNS,
053      HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE, 3);
054    return t;
055  }
056
057  @Override
058  protected void assertReplicaDistributed(final Table t) {
059    // Assert all data came back.
060    List<RegionInfo> regionInfos = new ArrayList<>();
061    for (RegionServerThread rs : this.util.getMiniHBaseCluster().getRegionServerThreads()) {
062      regionInfos.clear();
063      for (Region r : rs.getRegionServer().getRegions(t.getName())) {
064        LOG.info("The region is " + r.getRegionInfo() + " the location is "
065            + rs.getRegionServer().getServerName());
066        if (contains(regionInfos, r.getRegionInfo())) {
067          LOG.error("Am exiting");
068          fail("Crashed replica regions should not be assigned to same region server");
069        } else {
070          regionInfos.add(r.getRegionInfo());
071        }
072      }
073    }
074  }
075
076  private boolean contains(List<RegionInfo> regionInfos, RegionInfo regionInfo) {
077    for (RegionInfo info : regionInfos) {
078      if (RegionReplicaUtil.isReplicasForSameRegion(info, regionInfo)) {
079        return true;
080      }
081    }
082    return false;
083  }
084}
