048@Category(MediumTests.class)
049public class TestSafemodeBringsDownMaster {
050  private static final Logger LOG = LoggerFactory.getLogger(TestSafemodeBringsDownMaster.class);
051
052  protected static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
053
054  private static void setupConf(Configuration conf) {
055    conf.setInt(MasterProcedureConstants.MASTER_PROCEDURE_THREADS, 1);
056    conf.set(BaseLoadBalancer.TABLES_ON_MASTER, "none");
057  }
058
059  @BeforeClass
060  public static void setupCluster() throws Exception {
061    setupConf(UTIL.getConfiguration());
062    UTIL.startMiniCluster(1);
063  }
064
065  @AfterClass
066  public static void cleanupTest() throws Exception {
067    try {
068      UTIL.shutdownMiniCluster();
069    } catch (Exception e) {
070      LOG.warn("failure shutting down cluster", e);
071    }
072  }
073
074  @Before
075  public void setup() throws Exception {
076    resetProcExecutorTestingKillFlag();
077  }
078
079  private ProcedureExecutor<MasterProcedureEnv> getMasterProcedureExecutor() {
080    return UTIL.getHBaseCluster().getMaster().getMasterProcedureExecutor();
081  }
082  private void resetProcExecutorTestingKillFlag() {
083    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
084    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
085    assertTrue("expected executor to be running", procExec.isRunning());
086  }
087
088  @After
089  public void tearDown() throws Exception {
090  }
091
092  @Test(timeout=60000)
093  public void testSafemodeBringsDownMaster() throws Exception {
094    final TableName tableName = TableName.valueOf("testSafemodeBringsDownMaster");
095    final byte[][] splitKeys = new byte[][] {
096      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
097    };
098    RegionInfo[] regions = MasterProcedureTestingUtility.createTable(
099        getMasterProcedureExecutor(), tableName, splitKeys, "f1", "f2");
100    MiniDFSCluster dfsCluster = UTIL.getDFSCluster();
101    DistributedFileSystem dfs = (DistributedFileSystem) dfsCluster.getFileSystem();
102    dfs.setSafeMode(HdfsConstants.SafeModeAction.SAFEMODE_ENTER);
103    final long timeOut = 180000;
104    long startTime = System.currentTimeMillis();
105    int index = -1;
106    do {
107      index = UTIL.getMiniHBaseCluster().getServerWithMeta();
108    } while (index == -1 &&
109      startTime + timeOut < System.currentTimeMillis());
110
111    if (index != -1){
112      UTIL.getMiniHBaseCluster().abortRegionServer(index);
113      UTIL.getMiniHBaseCluster().waitOnRegionServer(index);
114    }
115    UTIL.waitFor(timeOut, new Waiter.Predicate<Exception>() {
116      @Override
117      public boolean evaluate() throws Exception {
118        List<JVMClusterUtil.MasterThread> threads = UTIL.getMiniHBaseCluster().getLiveMasterThreads();
119        return threads == null || threads.isEmpty();
120      }
121    });
122    dfs.setSafeMode(HdfsConstants.SafeModeAction.SAFEMODE_LEAVE);
123  }
124}
