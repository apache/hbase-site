095  @Test(timeout = 60000)
096  public void testDeleteNamespace() throws Exception {
097    final String namespaceName = "testDeleteNamespace";
098    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
099
100    createNamespaceForTesting(namespaceName);
101
102    long procId = procExec.submitProcedure(
103      new DeleteNamespaceProcedure(procExec.getEnvironment(), namespaceName),
104      nonceGroup,
105      nonce);
106    // Wait the completion
107    ProcedureTestingUtility.waitProcedure(procExec, procId);
108    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
109
110    validateNamespaceNotExist(namespaceName);
111  }

114  public void testDeleteNonExistNamespace() throws Exception {
115    final String namespaceName = "testDeleteNonExistNamespace";
116    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
117
118    validateNamespaceNotExist(namespaceName);
119
120    long procId = procExec.submitProcedure(
121      new DeleteNamespaceProcedure(procExec.getEnvironment(), namespaceName),
122      nonceGroup,
123      nonce);
124    // Wait the completion
125    ProcedureTestingUtility.waitProcedure(procExec, procId);
126    // Expect fail with NamespaceNotFoundException
127    ProcedureInfo result = procExec.getResult(procId);
128    assertTrue(result.isFailed());
129    LOG.debug("Delete namespace failed with exception: " + result.getExceptionFullMessage());

131      ProcedureTestingUtility.getExceptionCause(result) instanceof NamespaceNotFoundException);

173  @Test(timeout=60000)
174  public void testDeleteSameNamespaceTwiceWithSameNonce() throws Exception {
175    final String namespaceName = "testDeleteSameNamespaceTwiceWithSameNonce";
176    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
177
178    createNamespaceForTesting(namespaceName);
179
180    long procId1 = procExec.submitProcedure(
181      new DeleteNamespaceProcedure(procExec.getEnvironment(), namespaceName),
182      nonceGroup,
183      nonce);
184    long procId2 = procExec.submitProcedure(
185      new DeleteNamespaceProcedure(procExec.getEnvironment(), namespaceName),
186      nonceGroup,
187      nonce);
188    // Wait the completion
189    ProcedureTestingUtility.waitProcedure(procExec, procId1);
190    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
191
192    validateNamespaceNotExist(namespaceName);
193
194    // Wait the completion and expect not fail - because it is the same proc
195    ProcedureTestingUtility.waitProcedure(procExec, procId2);
196    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
197    assertTrue(procId1 == procId2);
198  }

200  @Test(timeout = 60000)
201  public void testRecoveryAndDoubleExecution() throws Exception {
202    final String namespaceName = "testRecoveryAndDoubleExecution";
203    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
204
205    createNamespaceForTesting(namespaceName);
206
207    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
208    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
209
210    // Start the DeleteNamespace procedure && kill the executor
211    long procId = procExec.submitProcedure(
212      new DeleteNamespaceProcedure(procExec.getEnvironment(), namespaceName),
213      nonceGroup,
214      nonce);
215
216    // Restart the executor and execute the step twice
217    int numberOfSteps = DeleteNamespaceState.values().length;
218    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId, numberOfSteps);
219
220    // Validate the deletion of namespace
221    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
222    validateNamespaceNotExist(namespaceName);
223  }

225  @Test(timeout = 60000)
226  public void testRollbackAndDoubleExecution() throws Exception {
227    final String namespaceName = "testRollbackAndDoubleExecution";
228    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
229
230    createNamespaceForTesting(namespaceName);
231
232    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
233    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
234
235    // Start the DeleteNamespace procedure && kill the executor
236    LOG.info("SUBMIT DELTET");
237    long procId = procExec.submitProcedure(
238      new DeleteNamespaceProcedure(procExec.getEnvironment(), namespaceName),
239      nonceGroup,
240      nonce);
241
242    int numberOfSteps = 0; // failing at pre operation
243    MasterProcedureTestingUtility.testRollbackAndDoubleExecution(procExec, procId, numberOfSteps);
244
245    // Validate the namespace still exists
246    NamespaceDescriptor createdNsDescriptor=
247        UTIL.getHBaseAdmin().getNamespaceDescriptor(namespaceName);
248    assertNotNull(createdNsDescriptor);
249  }

251  private ProcedureExecutor<MasterProcedureEnv> getMasterProcedureExecutor() {
252    return UTIL.getHBaseCluster().getMaster().getMasterProcedureExecutor();
253  }
254
255  private void createNamespaceForTesting(final String namespaceName) throws Exception {
256    final NamespaceDescriptor nsd = NamespaceDescriptor.create(namespaceName).build();
257    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
258
259    long procId = procExec.submitProcedure(
260      new CreateNamespaceProcedure(procExec.getEnvironment(), nsd),
261      nonceGroup + 1,
262      nonce + 1);
263    // Wait the completion
264    ProcedureTestingUtility.waitProcedure(procExec, procId);
265    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
266  }

268  public static void validateNamespaceNotExist(final String nsName) throws IOException {
269    try {
270      NamespaceDescriptor nsDescriptor = UTIL.getHBaseAdmin().getNamespaceDescriptor(nsName);
271      assertNull(nsDescriptor);
272    } catch (NamespaceNotFoundException nsnfe) {
273      // Expected
274    }
275  }
