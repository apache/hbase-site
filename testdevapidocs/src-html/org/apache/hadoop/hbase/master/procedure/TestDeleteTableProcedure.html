119  @Test(timeout=60000)
120  public void testDeleteDeletedTable() throws Exception {
121    final TableName tableName = TableName.valueOf("testDeleteDeletedTable");
122    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
123
124    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
125      procExec, tableName, null, "f");
126    UTIL.getHBaseAdmin().disableTable(tableName);
127
128    // delete the table (that exists)
129    long procId1 = procExec.submitProcedure(
130        new DeleteTableProcedure(procExec.getEnvironment(), tableName), nonceGroup, nonce);
131    // delete the table (that will no longer exist)
132    long procId2 = procExec.submitProcedure(
133        new DeleteTableProcedure(procExec.getEnvironment(), tableName), nonceGroup + 1, nonce + 1);
134
135    // Wait the completion
136    ProcedureTestingUtility.waitProcedure(procExec, procId1);
137    ProcedureTestingUtility.waitProcedure(procExec, procId2);
138
139    // First delete should succeed
140    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
141    MasterProcedureTestingUtility.validateTableDeletion(
142      UTIL.getHBaseCluster().getMaster(), tableName);
143
144    // Second delete should fail with TableNotFound
145    ProcedureInfo result = procExec.getResult(procId2);
146    assertTrue(result.isFailed());
147    LOG.debug("Delete failed with exception: " + result.getExceptionFullMessage());
148    assertTrue(ProcedureTestingUtility.getExceptionCause(result) instanceof TableNotFoundException);
149  }

154    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
155
156    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
157      procExec, tableName, null, "f");

159
160    // delete the table (that exists)
161    long procId1 = procExec.submitProcedure(
162        new DeleteTableProcedure(procExec.getEnvironment(), tableName), nonceGroup, nonce);
163    // delete the table (that will no longer exist)
164    long procId2 = procExec.submitProcedure(
165        new DeleteTableProcedure(procExec.getEnvironment(), tableName), nonceGroup, nonce);
166
167    // Wait the completion
168    ProcedureTestingUtility.waitProcedure(procExec, procId1);
169    ProcedureTestingUtility.waitProcedure(procExec, procId2);

171    // First delete should succeed
172    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);

175
176    // Second delete should not fail, because it is the same delete
177    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
178    assertTrue(procId1 == procId2);
179  }

211  @Test(timeout=60000)
212  public void testRecoveryAndDoubleExecution() throws Exception {
213    final TableName tableName = TableName.valueOf("testRecoveryAndDoubleExecution");
214
215    // create the table
216    byte[][] splitKeys = null;
217    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
218      getMasterProcedureExecutor(), tableName, splitKeys, "f1", "f2");
219    UTIL.getHBaseAdmin().disableTable(tableName);
220
221    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
222    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
223    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
224
225    // Start the Delete procedure && kill the executor
226    long procId = procExec.submitProcedure(
227      new DeleteTableProcedure(procExec.getEnvironment(), tableName), nonceGroup, nonce);
228
229    // Restart the executor and execute the step twice
230    // NOTE: the 6 (number of DeleteTableState steps) is hardcoded,
231    //       so you have to look at this test at least once when you add a new step.
232    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(
233      procExec, procId, 6, DeleteTableState.values());
234
235    MasterProcedureTestingUtility.validateTableDeletion(
236      UTIL.getHBaseCluster().getMaster(), tableName);
237  }

241  }
