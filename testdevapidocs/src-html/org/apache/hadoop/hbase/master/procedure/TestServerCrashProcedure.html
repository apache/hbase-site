058  private long serverCrashSubmittedCount = 0;
059  private long serverCrashFailedCount = 0;

061  private void setupConf(Configuration conf) {
062    conf.setInt(MasterProcedureConstants.MASTER_PROCEDURE_THREADS, 1);
063    conf.set("hbase.balancer.tablesOnMaster", "none");
064    conf.setInt("hbase.client.retries.number", 3);
065  }

067  @Before
068  public void setup() throws Exception {
069    this.util = new HBaseTestingUtility();
070    setupConf(this.util.getConfiguration());
071    this.util.startMiniCluster(3);
072    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(
073      this.util.getHBaseCluster().getMaster().getMasterProcedureExecutor(), false);
074    serverCrashProcMetrics = this.util.getHBaseCluster().getMaster().getMasterMetrics()
075        .getServerCrashProcMetrics();
076  }
077
078  @After
079  public void tearDown() throws Exception {
080    MiniHBaseCluster cluster = this.util.getHBaseCluster();
081    HMaster master = cluster == null? null: cluster.getMaster();
082    if (master != null && master.getMasterProcedureExecutor() != null) {
083      ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(
084        master.getMasterProcedureExecutor(), false);
085    }
086    this.util.shutdownMiniCluster();
087  }

090  @Test
091  public void testCrashTargetRs() throws Exception {
092    testRecoveryAndDoubleExecution(false, false);
093  }
094
095  @Test
096  public void testRecoveryAndDoubleExecutionOnRsWithMeta() throws Exception {
097    testRecoveryAndDoubleExecution(true, true);
098  }
099
100  @Test
101  public void testRecoveryAndDoubleExecutionOnRsWithoutMeta() throws Exception {
102    testRecoveryAndDoubleExecution(false, true);
103  }
104
105  private long getSCPProcId(ProcedureExecutor<?> procExec) {
106    util.waitFor(30000, () -> !procExec.getProcedures().isEmpty());
107    return procExec.getActiveProcIds().stream().mapToLong(Long::longValue).min().getAsLong();
108  }
109
110  /**
111   * Run server crash procedure steps twice to test idempotency and that we are persisting all
112   * needed state.
113   */
114  private void testRecoveryAndDoubleExecution(boolean carryingMeta, boolean doubleExecution)
115      throws Exception {
116    final TableName tableName = TableName.valueOf(
117      "testRecoveryAndDoubleExecution-carryingMeta-" + carryingMeta);
118    final Table t = this.util.createTable(tableName, HBaseTestingUtility.COLUMNS,
119        HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
120    try {
121      // Load the table with a bit of data so some logs to split and some edits in each region.
122      this.util.loadTable(t, HBaseTestingUtility.COLUMNS[0]);
123      final int count = util.countRows(t);
124      assertTrue("expected some rows", count > 0);
125      final String checksum = util.checksumRows(t);
126      // Run the procedure executor outside the master so we can mess with it. Need to disable
127      // Master's running of the server crash processing.
128      final HMaster master = this.util.getHBaseCluster().getMaster();
129      final ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
130      // find the first server that match the request and executes the test
131      ServerName rsToKill = null;
132      for (RegionInfo hri : util.getAdmin().getRegions(tableName)) {
133        final ServerName serverName = AssignmentTestingUtil.getServerHoldingRegion(util, hri);
134        if (AssignmentTestingUtil.isServerHoldingMeta(util, serverName) == carryingMeta) {
135          rsToKill = serverName;
136          break;
137        }
138      }
139      // Enable test flags and then queue the crash procedure.
140      ProcedureTestingUtility.waitNoProcedureRunning(procExec);
141      if (doubleExecution) {
142        ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
143        // kill the RS
144        AssignmentTestingUtil.killRs(util, rsToKill);
145        long procId = getSCPProcId(procExec);
146        // Now run through the procedure twice crashing the executor on each step...
147        MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId);
148      } else {
149        // kill the RS
150        AssignmentTestingUtil.killRs(util, rsToKill);
151        long procId = getSCPProcId(procExec);
152        ProcedureTestingUtility.waitProcedure(procExec, procId);
153      }
154      // Assert all data came back.
155      assertEquals(count, util.countRows(t));
156      assertEquals(checksum, util.checksumRows(t));
157    } catch(Throwable throwable) {
158      LOG.error("Test failed!", throwable);
159      throw throwable;
160    } finally {
161      t.close();
162    }
163  }
164
165  private void collectMasterMetrics() {
166    serverCrashSubmittedCount = serverCrashProcMetrics.getSubmittedCounter().getCount();
167    serverCrashFailedCount = serverCrashProcMetrics.getFailedCounter().getCount();
168  }
