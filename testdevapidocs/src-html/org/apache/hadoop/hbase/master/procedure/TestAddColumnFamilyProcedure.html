041  @Test(timeout = 60000)
042  public void testAddColumnFamily() throws Exception {
043    final TableName tableName = TableName.valueOf("testAddColumnFamily");
044    final String cf1 = "cf1";
045    final String cf2 = "cf2";
046    final HColumnDescriptor columnDescriptor1 = new HColumnDescriptor(cf1);
047    final HColumnDescriptor columnDescriptor2 = new HColumnDescriptor(cf2);
048    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
049
050    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f3");
051
052    // Test 1: Add a column family online
053    long procId1 = procExec.submitProcedure(
054      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor1),
055      nonceGroup,
056      nonce);
057    // Wait the completion
058    ProcedureTestingUtility.waitProcedure(procExec, procId1);
059    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
060
061    MasterProcedureTestingUtility.validateColumnFamilyAddition(UTIL.getHBaseCluster().getMaster(),
062      tableName, cf1);
063
064    // Test 2: Add a column family offline
065    UTIL.getHBaseAdmin().disableTable(tableName);
066    long procId2 = procExec.submitProcedure(
067      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor2),
068      nonceGroup + 1,
069      nonce + 1);
070    // Wait the completion
071    ProcedureTestingUtility.waitProcedure(procExec, procId2);
072    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
073    MasterProcedureTestingUtility.validateColumnFamilyAddition(UTIL.getHBaseCluster().getMaster(),
074      tableName, cf2);
075  }

077  @Test(timeout=60000)
078  public void testAddSameColumnFamilyTwice() throws Exception {
079    final TableName tableName = TableName.valueOf("testAddColumnFamilyTwice");
080    final String cf2 = "cf2";
081    final HColumnDescriptor columnDescriptor = new HColumnDescriptor(cf2);
082
083    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
084
085    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f1");
086
087    // add the column family
088    long procId1 = procExec.submitProcedure(
089      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor),
090      nonceGroup,
091      nonce);
092    // Wait the completion
093    ProcedureTestingUtility.waitProcedure(procExec, procId1);
094    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
095    MasterProcedureTestingUtility.validateColumnFamilyAddition(UTIL.getHBaseCluster().getMaster(),
096      tableName, cf2);
097
098    // add the column family that exists
099    long procId2 = procExec.submitProcedure(
100      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor),
101      nonceGroup + 1,
102      nonce + 1);
103    // Wait the completion
104    ProcedureTestingUtility.waitProcedure(procExec, procId2);
105
106    // Second add should fail with InvalidFamilyOperationException
107    ProcedureInfo result = procExec.getResult(procId2);
108    assertTrue(result.isFailed());
109    LOG.debug("Add failed with exception: " + result.getExceptionFullMessage());
110    assertTrue(
111      ProcedureTestingUtility.getExceptionCause(result) instanceof InvalidFamilyOperationException);
112
113    // Do the same add the existing column family - this time offline
114    UTIL.getHBaseAdmin().disableTable(tableName);
115    long procId3 = procExec.submitProcedure(
116      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor),
117      nonceGroup + 2,
118      nonce + 2);
119    // Wait the completion
120    ProcedureTestingUtility.waitProcedure(procExec, procId3);
121
122    // Second add should fail with InvalidFamilyOperationException
123    result = procExec.getResult(procId3);
124    assertTrue(result.isFailed());
125    LOG.debug("Add failed with exception: " + result.getExceptionFullMessage());
126    assertTrue(
127      ProcedureTestingUtility.getExceptionCause(result) instanceof InvalidFamilyOperationException);
128  }

131  public void testAddSameColumnFamilyTwiceWithSameNonce() throws Exception {
132    final TableName tableName = TableName.valueOf("testAddSameColumnFamilyTwiceWithSameNonce");
133    final String cf2 = "cf2";
134    final HColumnDescriptor columnDescriptor = new HColumnDescriptor(cf2);
135
136    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
137
138    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f1");
139
140    // add the column family
141    long procId1 = procExec.submitProcedure(
142      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor),
143      nonceGroup,
144      nonce);
145    long procId2 = procExec.submitProcedure(
146      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor),
147      nonceGroup,
148      nonce);
149    // Wait the completion
150    ProcedureTestingUtility.waitProcedure(procExec, procId1);
151    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
152    MasterProcedureTestingUtility.validateColumnFamilyAddition(UTIL.getHBaseCluster().getMaster(),
153      tableName, cf2);
154
155    // Wait the completion and expect not fail - because it is the same proc
156    ProcedureTestingUtility.waitProcedure(procExec, procId2);
157    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
158    assertTrue(procId1 == procId2);
159  }

161  @Test(timeout = 60000)
162  public void testRecoveryAndDoubleExecutionOffline() throws Exception {
163    final TableName tableName = TableName.valueOf("testRecoveryAndDoubleExecutionOffline");
164    final String cf4 = "cf4";
165    final HColumnDescriptor columnDescriptor = new HColumnDescriptor(cf4);
166    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
167    // create the table
168    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f1", "f2", "f3");
169    UTIL.getHBaseAdmin().disableTable(tableName);
170
171    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
172    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
173
174    // Start the AddColumnFamily procedure && kill the executor
175    long procId = procExec.submitProcedure(
176      new AddColumnFamilyProcedure(procExec.getEnvironment(), tableName, columnDescriptor),
177      nonceGroup,
178      nonce);
179
180    // Restart the executor and execute the step twice
181    int numberOfSteps = AddColumnFamilyState.values().length;
182    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId, numberOfSteps);
183
184    MasterProcedureTestingUtility.validateColumnFamilyAddition(UTIL.getHBaseCluster().getMaster(),
185      tableName, cf4);
186  }
