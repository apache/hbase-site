113  @Test(timeout = 30000)
114  public void testServerCrashProcedureEvent() throws Exception {
115    TableName tableName = TableName.valueOf("testServerCrashProcedureEventTb");
116    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
117    ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
118
119    while (!master.isServerCrashProcessingEnabled() || !master.isInitialized() ||
120        master.getAssignmentManager().getRegionStates().isRegionsInTransition()) {
121      Thread.sleep(25);
122    }
123
124    UTIL.createTable(tableName, HBaseTestingUtility.COLUMNS[0]);
125    try (Table t = UTIL.getConnection().getTable(tableName)) {
126      // Load the table with a bit of data so some logs to split and some edits in each region.
127      UTIL.loadTable(t, HBaseTestingUtility.COLUMNS[0]);
128    }
129
130    master.setServerCrashProcessingEnabled(false);  // fake it, set back later
131
132    // Kill a server. Master will notice but do nothing other than add it to list of dead servers.
133    HRegionServer hrs = getServerWithRegions();
134    boolean carryingMeta = master.getAssignmentManager().isCarryingMeta(hrs.getServerName());
135    UTIL.getHBaseCluster().killRegionServer(hrs.getServerName());
136    hrs.join();
137
138    // Wait until the expiration of the server has arrived at the master. We won't process it
139    // by queuing a ServerCrashProcedure because we have disabled crash processing... but wait
140    // here so ServerManager gets notice and adds expired server to appropriate queues.
141    while (!master.getServerManager().isServerDead(hrs.getServerName())) Thread.sleep(10);
142
143    // Do some of the master processing of dead servers so when SCP runs, it has expected 'state'.
144    master.getServerManager().moveFromOnelineToDeadServers(hrs.getServerName());
145
146    // check event wait/wake
147    testProcedureEventWaitWake(master, master.getServerCrashProcessingEnabledEvent(),
148      new ServerCrashProcedure(procExec.getEnvironment(), hrs.getServerName(), true, carryingMeta));
149  }

151  private void testProcedureEventWaitWake(final HMaster master, final ProcedureEvent event,
152      final Procedure proc) throws Exception {
153    final ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
154    final MasterProcedureScheduler procSched = procExec.getEnvironment().getProcedureQueue();
155
156    final long startPollCalls = procSched.getPollCalls();
157    final long startNullPollCalls = procSched.getNullPollCalls();
158
159    // check that nothing is in the event queue
160    LOG.debug("checking " + event);
161    assertEquals(false, event.isReady());
162    assertEquals(0, event.size());
163
164    // submit the procedure
165    LOG.debug("submit " + proc);
166    long procId = procExec.submitProcedure(proc, HConstants.NO_NONCE, HConstants.NO_NONCE);
167
168    // wait until the event is in the queue (proc executed and got into suspended state)
169    LOG.debug("wait procedure suspended on " + event);
170    while (event.size() < 1) Thread.sleep(25);
171
172    // check that the proc is in the event queue
173    LOG.debug("checking " + event + " size=" + event.size());
174    assertEquals(false, event.isReady());
175    assertEquals(1, event.size());
176
177    // wake the event
178    LOG.debug("wake " + event);
179    procSched.wakeEvent(event);
180    assertEquals(true, event.isReady());
181
182    // wait until proc completes
183    LOG.debug("waiting " + proc);
184    ProcedureTestingUtility.waitProcedure(procExec, procId);
185
186    // check that nothing is in the event queue and the event is not suspended
187    assertEquals(true, event.isReady());
188    assertEquals(0, event.size());
189    LOG.debug("completed execution of " + proc +
190      " pollCalls=" + (procSched.getPollCalls() - startPollCalls) +
191      " nullPollCalls=" + (procSched.getNullPollCalls() - startNullPollCalls));
192  }

194  private HRegionServer getServerWithRegions() {
195    for (int i = 0; i < 3; ++i) {
196      HRegionServer hrs = UTIL.getHBaseCluster().getRegionServer(i);
197      if (hrs.getNumberOfOnlineRegions() > 0) {
198        return hrs;
199      }
200    }
201    return null;
202  }

204  @Test(timeout=30000)
205  public void testTimeoutEventProcedure() throws Exception {
206    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
207    ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
208    MasterProcedureScheduler procSched = procExec.getEnvironment().getProcedureQueue();
209
210    TestTimeoutEventProcedure proc = new TestTimeoutEventProcedure(1000, 5);
211    procExec.submitProcedure(proc);
212
213    ProcedureTestingUtility.waitProcedure(procExec, proc.getProcId());
214    ProcedureTestingUtility.assertIsAbortException(procExec.getResult(proc.getProcId()));
215  }

217  public static class TestTimeoutEventProcedure
218      extends Procedure<MasterProcedureEnv> implements TableProcedureInterface {
219    private final ProcedureEvent event = new ProcedureEvent("timeout-event");
220
221    private final AtomicInteger ntimeouts = new AtomicInteger(0);
222    private int maxTimeouts = 1;
223
224    public TestTimeoutEventProcedure() {}
225
226    public TestTimeoutEventProcedure(final int timeoutMsec, final int maxTimeouts) {
227      this.maxTimeouts = maxTimeouts;
228      setTimeout(timeoutMsec);
229      setOwner("test");
230    }
231
232    @Override
233    protected Procedure[] execute(final MasterProcedureEnv env)
234        throws ProcedureSuspendedException {
235      LOG.info("EXECUTE " + this + " ntimeouts=" + ntimeouts);
236      if (ntimeouts.get() > maxTimeouts) {
237        setAbortFailure("test", "give up after " + ntimeouts.get());
238        return null;
239      }
240
241      env.getProcedureQueue().suspendEvent(event);
242      if (env.getProcedureQueue().waitEvent(event, this)) {
243        setState(ProcedureState.WAITING_TIMEOUT);
244        throw new ProcedureSuspendedException();
245      }
246
247      return null;
248    }
249
250    @Override
251    protected void rollback(final MasterProcedureEnv env) {
252    }
253
254    @Override
255    protected boolean setTimeoutFailure(final MasterProcedureEnv env) {
256      int n = ntimeouts.incrementAndGet();
257      LOG.info("HANDLE TIMEOUT " + this + " ntimeouts=" + n);
258      setState(ProcedureState.RUNNABLE);
259      env.getProcedureQueue().wakeEvent(event);
260      return false;
261    }
262
263    @Override
264    public TableName getTableName() {
265      return TableName.valueOf("testtb");
266    }
267
268    @Override
269    public TableOperationType getTableOperationType() {
270      return TableOperationType.READ;
271    }
272
273    @Override
274    protected boolean abort(MasterProcedureEnv env) { return false; }
275
276    @Override
277    protected void serializeStateData(final OutputStream stream) throws IOException {
278    }
279
280    @Override
281    protected void deserializeStateData(final InputStream stream) throws IOException {
282    }
283  }
