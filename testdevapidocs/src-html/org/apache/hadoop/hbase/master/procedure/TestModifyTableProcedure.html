170

172    assertEquals(2, currentHtd.getFamiliesKeys().size());
173    assertFalse(currentHtd.hasFamily(Bytes.toBytes(cf2)));
174
175    // Test 2: Modify the table descriptor offline
176    UTIL.getAdmin().disableTable(tableName);
177    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
178
179    HTableDescriptor htd2 =
180        new HTableDescriptor(UTIL.getAdmin().getDescriptor(tableName));
181    htd2.removeFamily(Bytes.toBytes(cf3));
182    // Disable Sanity check
183    htd2.setConfiguration(TableDescriptorChecker.TABLE_SANITY_CHECKS, Boolean.FALSE.toString());
184
185    long procId2 =
186        ProcedureTestingUtility.submitAndWait(procExec,
187          new ModifyTableProcedure(procExec.getEnvironment(), htd2));

189
190    currentHtd = new HTableDescriptor(UTIL.getAdmin().getDescriptor(tableName));
191    assertEquals(1, currentHtd.getFamiliesKeys().size());
192    assertFalse(currentHtd.hasFamily(Bytes.toBytes(cf3)));
193
194    //Removing the last family will fail
195    HTableDescriptor htd3 =
196        new HTableDescriptor(UTIL.getAdmin().getDescriptor(tableName));
197    htd3.removeFamily(Bytes.toBytes(cf1));
198    long procId3 =
199        ProcedureTestingUtility.submitAndWait(procExec,
200            new ModifyTableProcedure(procExec.getEnvironment(), htd3));
201    final Procedure<?> result = procExec.getResult(procId3);
202    assertEquals(true, result.isFailed());
203    Throwable cause = ProcedureTestingUtility.getExceptionCause(result);
204    assertTrue("expected DoNotRetryIOException, got " + cause,
205        cause instanceof DoNotRetryIOException);
206    assertEquals(1, currentHtd.getFamiliesKeys().size());
207    assertTrue(currentHtd.hasFamily(Bytes.toBytes(cf1)));
208  }

293  @Test
294  public void testColumnFamilyAdditionTwiceWithNonce() throws Exception {
295    final TableName tableName = TableName.valueOf(name.getMethodName());
296    final String cf2 = "cf2";
297    final String cf3 = "cf3";
298    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
299
300    // create the table
301    RegionInfo[] regions =
302        MasterProcedureTestingUtility.createTable(procExec, tableName, null, "cf1", cf3);
303
304    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
305    // Modify multiple properties of the table.
306    TableDescriptor td = UTIL.getAdmin().getDescriptor(tableName);
307    TableDescriptor newTd =
308        TableDescriptorBuilder.newBuilder(td).setCompactionEnabled(!td.isCompactionEnabled())
309          .setColumnFamily(ColumnFamilyDescriptorBuilder.of(cf2)).build();
310
311    PerClientRandomNonceGenerator nonceGenerator = PerClientRandomNonceGenerator.get();
312    long nonceGroup = nonceGenerator.getNonceGroup();
313    long newNonce = nonceGenerator.newNonce();
314    NonceKey nonceKey = new NonceKey(nonceGroup, newNonce);
315    procExec.registerNonce(nonceKey);
316
317    // Start the Modify procedure && kill the executor
318    final long procId = procExec
319        .submitProcedure(new ModifyTableProcedure(procExec.getEnvironment(), newTd), nonceKey);
320
321    // Restart the executor after MODIFY_TABLE_UPDATE_TABLE_DESCRIPTOR and try to add column family
322    // as nonce are there , we should not fail
323    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId, new StepHook() {
324      @Override
325      public boolean execute(int step) throws IOException {
326        if (step == 3) {
327          return procId == UTIL.getHBaseCluster().getMaster().addColumn(tableName,
328            ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(cf2)).build(), nonceGroup,
329            newNonce);
330        }
331        return true;
332      }
333    });
334
335    //Try with different nonce, now it should fail the checks
336    try {
337      UTIL.getHBaseCluster().getMaster().addColumn(tableName,
338        ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(cf2)).build(), nonceGroup,
339        nonceGenerator.newNonce());
340      Assert.fail();
341    } catch (InvalidFamilyOperationException e) {
342    }
343
344    // Validate descriptor
345    TableDescriptor currentHtd = UTIL.getAdmin().getDescriptor(tableName);
346    assertEquals(!td.isCompactionEnabled(), currentHtd.isCompactionEnabled());
347    assertEquals(3, currentHtd.getColumnFamilyCount());
348    assertTrue(currentHtd.hasColumnFamily(Bytes.toBytes(cf2)));
349    assertTrue(currentHtd.hasColumnFamily(Bytes.toBytes(cf3)));
350
351    // cf2 should be added
352    MasterProcedureTestingUtility.validateTableCreation(UTIL.getHBaseCluster().getMaster(),
353      tableName, regions, "cf1", cf2, cf3);
354  }

420    final TableName tableName = TableName.valueOf(name.getMethodName());
421    UTIL.createTable(tableName, column_Family1);
422
423    class ConcurrentAddColumnFamily extends Thread {
424      TableName tableName = null;
425      ColumnFamilyDescriptor columnFamilyDescriptor;
426      boolean exception;
427
428      public ConcurrentAddColumnFamily(TableName tableName,
429          ColumnFamilyDescriptor columnFamilyDescriptor) {
430        this.tableName = tableName;
431        this.columnFamilyDescriptor = columnFamilyDescriptor;
432        this.exception = false;
433      }
434
435      public void run() {
436        try {
437          UTIL.getAdmin().addColumnFamily(tableName, columnFamilyDescriptor);
438        } catch (Exception e) {
439          if (e.getClass().equals(ConcurrentTableModificationException.class)) {
440            this.exception = true;
441          }
442        }
443      }
444    }
445    ColumnFamilyDescriptor columnFamilyDescriptor =
446      ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(column_Family2)).build();
447    ConcurrentAddColumnFamily t1 =
448      new ConcurrentAddColumnFamily(tableName, columnFamilyDescriptor);
449    columnFamilyDescriptor =
450      ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(column_Family3)).build();
451    ConcurrentAddColumnFamily t2 =
452      new ConcurrentAddColumnFamily(tableName, columnFamilyDescriptor);
453
454    t1.start();
455    t2.start();
456
457    t1.join();
458    t2.join();
459    int noOfColumnFamilies = UTIL.getAdmin().getDescriptor(tableName).getColumnFamilies().length;
460    assertTrue("Expected ConcurrentTableModificationException.",
461      ((t1.exception || t2.exception) && noOfColumnFamilies == 2) || noOfColumnFamilies == 3);
