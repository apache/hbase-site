071  @Test
072  public void testInfo() {
073    HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
074    MetricsMasterWrapperImpl info = new MetricsMasterWrapperImpl(master);
075    assertEquals(master.getSplitPlanCount(), info.getSplitPlanCount(), 0);
076    assertEquals(master.getMergePlanCount(), info.getMergePlanCount(), 0);
077    assertEquals(master.getAverageLoad(), info.getAverageLoad(), 0);
078    assertEquals(master.getClusterId(), info.getClusterId());
079    assertEquals(master.getMasterActiveTime(), info.getActiveTime());
080    assertEquals(master.getMasterStartTime(), info.getStartTime());
081    assertEquals(master.getMasterCoprocessors().length, info.getCoprocessors().length);
082    assertEquals(master.getServerManager().getOnlineServersList().size(), info.getNumRegionServers());
083    int regionServerCount =
084      NUM_RS + (LoadBalancer.isTablesOnMaster(TEST_UTIL.getConfiguration())? 1: 0);
085    assertEquals(regionServerCount, info.getNumRegionServers());
086
087    String zkServers = info.getZookeeperQuorum();
088    assertEquals(zkServers.split(",").length, TEST_UTIL.getZkCluster().getZooKeeperServerNum());
089
090    final int index = 3;
091    LOG.info("Stopping " + TEST_UTIL.getMiniHBaseCluster().getRegionServer(index));
092    TEST_UTIL.getMiniHBaseCluster().stopRegionServer(index, false);
093    TEST_UTIL.getMiniHBaseCluster().waitOnRegionServer(index);
094    // We stopped the regionserver but could take a while for the master to notice it so hang here
095    // until it does... then move forward to see if metrics wrapper notices.
096    while (TEST_UTIL.getHBaseCluster().getMaster().getServerManager().getOnlineServers().size() ==
097        regionServerCount ) {
098      Threads.sleep(10);
099    }
100    assertEquals(regionServerCount - 1, info.getNumRegionServers());
101    assertEquals(1, info.getNumDeadRegionServers());
102    // now we do not expose this information as WALProcedureStore is not the only ProcedureStore
103    // implementation any more.
104    assertEquals(0, info.getNumWALFiles());
105  }

107  @Test
108  public void testQuotaSnapshotConversion() {
109    MetricsMasterWrapperImpl info = new MetricsMasterWrapperImpl(
110        TEST_UTIL.getHBaseCluster().getMaster());
111    assertEquals(new SimpleImmutableEntry<Long,Long>(1024L, 2048L),
112        info.convertSnapshot(new SpaceQuotaSnapshot(
113            SpaceQuotaStatus.notInViolation(), 1024L, 2048L)));
114    assertEquals(new SimpleImmutableEntry<Long,Long>(4096L, 2048L),
115        info.convertSnapshot(new SpaceQuotaSnapshot(
116            new SpaceQuotaStatus(SpaceViolationPolicy.NO_INSERTS), 4096L, 2048L)));
117  }

119  /**
120   * tests online and offline region number
121   */
122  @Test
123  public void testOfflineRegion() throws Exception {
124    HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
125    MetricsMasterWrapperImpl info = new MetricsMasterWrapperImpl(master);
126    TableName table = TableName.valueOf("testRegionNumber");
127    try {
128      RegionInfo hri;
129      byte[] FAMILY = Bytes.toBytes("FAMILY");
130      TableDescriptor tableDescriptor = TableDescriptorBuilder.newBuilder(table)
131        .setColumnFamily(ColumnFamilyDescriptorBuilder.of(FAMILY)).build();
132      TEST_UTIL.getAdmin().createTable(tableDescriptor, Bytes.toBytes("A"),
133        Bytes.toBytes("Z"), 5);
134
135      // wait till the table is assigned
136      long timeoutTime = System.currentTimeMillis() + 1000;
137      while (true) {
138        List<RegionInfo> regions = master.getAssignmentManager().
139          getRegionStates().getRegionsOfTable(table);
140        if (regions.size() > 3) {
141          hri = regions.get(2);
142          break;
143        }
144        long now = System.currentTimeMillis();
145        if (now > timeoutTime) {
146          fail("Could not find an online region");
147        }
148        Thread.sleep(10);
149      }
150
151      PairOfSameType<Integer> regionNumberPair = info.getRegionCounts();
152      assertEquals(5, regionNumberPair.getFirst().intValue());
153      assertEquals(0, regionNumberPair.getSecond().intValue());
154
155      TEST_UTIL.getAdmin().offline(hri.getRegionName());
156
157      timeoutTime = System.currentTimeMillis() + 800;
158      RegionStates regionStates = master.getAssignmentManager().getRegionStates();
159      while (true) {
160        if (regionStates.getRegionByStateOfTable(table)
161            .get(RegionState.State.OFFLINE).contains(hri)) {
162          break;
163        }
164        long now = System.currentTimeMillis();
165        if (now > timeoutTime) {
166          fail("Failed to offline the region in time");
167          break;
168        }
169        Thread.sleep(10);
170      }
171      regionNumberPair = info.getRegionCounts();
172      assertEquals(4, regionNumberPair.getFirst().intValue());
173      assertEquals(1, regionNumberPair.getSecond().intValue());
174    } finally {
175      TEST_UTIL.deleteTable(table);
176    }
177  }
