085  @Test
086  public void testPlanComparator() {
087    Comparator<NormalizationPlan> comparator = new SimpleRegionNormalizer.PlanComparator();
088    NormalizationPlan splitPlan1 = new SplitNormalizationPlan(null, null);
089    NormalizationPlan splitPlan2 = new SplitNormalizationPlan(null, null);
090    NormalizationPlan mergePlan1 = new MergeNormalizationPlan(null, null);
091    NormalizationPlan mergePlan2 = new MergeNormalizationPlan(null, null);
092
093    assertTrue(comparator.compare(splitPlan1, splitPlan2) == 0);
094    assertTrue(comparator.compare(splitPlan2, splitPlan1) == 0);
095    assertTrue(comparator.compare(mergePlan1, mergePlan2) == 0);
096    assertTrue(comparator.compare(mergePlan2, mergePlan1) == 0);
097    assertTrue(comparator.compare(splitPlan1, mergePlan1) < 0);
098    assertTrue(comparator.compare(mergePlan1, splitPlan1) > 0);
099  }

479  @SuppressWarnings("MockitoCast")
480  protected void setupMocksForNormalizer(Map<byte[], Integer> regionSizes,
481                                         List<RegionInfo> RegionInfo) {
482    masterServices = Mockito.mock(MasterServices.class, RETURNS_DEEP_STUBS);
483    masterRpcServices = Mockito.mock(MasterRpcServices.class, RETURNS_DEEP_STUBS);
484
485    // for simplicity all regions are assumed to be on one server; doesn't matter to us
486    ServerName sn = ServerName.valueOf("localhost", 0, 1L);
487    when(masterServices.getAssignmentManager().getRegionStates().
488      getRegionsOfTable(any())).thenReturn(RegionInfo);
489    when(masterServices.getAssignmentManager().getRegionStates().
490      getRegionServerOfRegion(any())).thenReturn(sn);
491
492    for (Map.Entry<byte[], Integer> region : regionSizes.entrySet()) {
493      RegionMetrics regionLoad = Mockito.mock(RegionMetrics.class);
494      when(regionLoad.getRegionName()).thenReturn(region.getKey());
495      when(regionLoad.getStoreFileSize())
496        .thenReturn(new Size(region.getValue(), Size.Unit.MEGABYTE));
497
498      // this is possibly broken with jdk9, unclear if false positive or not
499      // suppress it for now, fix it when we get to running tests on 9
500      // see: http://errorprone.info/bugpattern/MockitoCast
501      when((Object) masterServices.getServerManager().getLoad(sn).
502        getRegionMetrics().get(region.getKey())).thenReturn(regionLoad);
503    }
504    try {
505      when(masterRpcServices.isSplitOrMergeEnabled(any(),
506        any())).thenReturn(
507          IsSplitOrMergeEnabledResponse.newBuilder().setEnabled(true).build());
508    } catch (ServiceException se) {
509      LOG.debug("error setting isSplitOrMergeEnabled switch", se);
510    }
511
512    normalizer.setMasterServices(masterServices);
513    normalizer.setMasterRpcServices(masterRpcServices);
514  }
