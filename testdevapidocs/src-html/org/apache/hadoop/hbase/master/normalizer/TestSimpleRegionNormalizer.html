387  @Test
388  public void testNormalizerCannotMergeNonAdjacentRegions() {
389    final TableName tableName = name.getTableName();
390    // create 5 regions with sizes to trigger merge of small regions. region ranges are:
391    // [, "aa"), ["aa", "aa1"), ["aa1", "aa1!"), ["aa1!", "aa2"), ["aa2", )
392    // Region ["aa", "aa1") and ["aa1!", "aa2") are not adjacent, they are not supposed to
393    // merged.
394    final byte[][] keys = {
395      null,
396      Bytes.toBytes("aa"),
397      Bytes.toBytes("aa1!"),
398      Bytes.toBytes("aa1"),
399      Bytes.toBytes("aa2"),
400      null,
401    };
402    final List<RegionInfo> regionInfos = createRegionInfos(tableName, keys);
403    final Map<byte[], Integer> regionSizes =
404      createRegionSizesMap(regionInfos, 3, 1, 1, 3, 5);
405    setupMocksForNormalizer(regionSizes, regionInfos);
406
407    // Compute the plan, no merge plan returned as they are not adjacent.
408    List<NormalizationPlan> plans = normalizer.computePlansForTable(tableName);
409    assertThat(plans, empty());
410  }

412  @SuppressWarnings("MockitoCast")
413  private void setupMocksForNormalizer(Map<byte[], Integer> regionSizes,
414    List<RegionInfo> regionInfoList) {
415    masterServices = Mockito.mock(MasterServices.class, RETURNS_DEEP_STUBS);
416
417    // for simplicity all regions are assumed to be on one server; doesn't matter to us
418    ServerName sn = ServerName.valueOf("localhost", 0, 0L);
419    when(masterServices.getAssignmentManager().getRegionStates()
420      .getRegionsOfTable(any())).thenReturn(regionInfoList);
421    when(masterServices.getAssignmentManager().getRegionStates()
422      .getRegionServerOfRegion(any())).thenReturn(sn);
423    when(masterServices.getAssignmentManager().getRegionStates()
424      .getRegionState(any(RegionInfo.class))).thenReturn(
425        RegionState.createForTesting(null, RegionState.State.OPEN));
426
427    for (Map.Entry<byte[], Integer> region : regionSizes.entrySet()) {
428      RegionMetrics regionLoad = Mockito.mock(RegionMetrics.class);
429      when(regionLoad.getRegionName()).thenReturn(region.getKey());
430      when(regionLoad.getStoreFileSize())
431        .thenReturn(new Size(region.getValue(), Size.Unit.MEGABYTE));
432
433      // this is possibly broken with jdk9, unclear if false positive or not
434      // suppress it for now, fix it when we get to running tests on 9
435      // see: http://errorprone.info/bugpattern/MockitoCast
436      when((Object) masterServices.getServerManager().getLoad(sn)
437        .getRegionMetrics().get(region.getKey())).thenReturn(regionLoad);
438    }
439
440    when(masterServices.isSplitOrMergeEnabled(any())).thenReturn(true);
441
442    normalizer = new SimpleRegionNormalizer();
443    normalizer.setConf(conf);
444    normalizer.setMasterServices(masterServices);
445  }

447  /**
448   * Create a list of {@link RegionInfo}s that represent a region chain of the specified length.
449   */
450  private static List<RegionInfo> createRegionInfos(final TableName tableName, final int length) {
451    if (length < 1) {
452      throw new IllegalStateException("length must be greater than or equal to 1.");
453    }
454
455    final byte[] startKey = Bytes.toBytes("aaaaa");
456    final byte[] endKey = Bytes.toBytes("zzzzz");
457    if (length == 1) {
458      return Collections.singletonList(createRegionInfo(tableName, startKey, endKey));
459    }
460
461    final byte[][] splitKeys = Bytes.split(startKey, endKey, length - 1);
462    final List<RegionInfo> ret = new ArrayList<>(length);
463    for (int i = 0; i < splitKeys.length - 1; i++) {
464      ret.add(createRegionInfo(tableName, splitKeys[i], splitKeys[i+1]));
465    }
466    return ret;
467  }
468
469  private static RegionInfo createRegionInfo(final TableName tableName, final byte[] startKey,
470    final byte[] endKey) {
471    return RegionInfoBuilder.newBuilder(tableName)
472      .setStartKey(startKey)
473      .setEndKey(endKey)
474      .setRegionId(generateRegionId())
475      .build();
476  }
477
478  private static long generateRegionId() {
479    return Instant.ofEpochMilli(EnvironmentEdgeManager.currentTime())
480      .minus(Period.ofDays(DEFAULT_MERGE_MIN_REGION_AGE_DAYS + 1))
481      .toEpochMilli();
482  }
483
484  private static List<RegionInfo> createRegionInfos(final TableName tableName,
485    final byte[][] splitKeys) {
486    final List<RegionInfo> ret = new ArrayList<>(splitKeys.length);
487    for (int i = 0; i < splitKeys.length - 1; i++) {
488      ret.add(createRegionInfo(tableName, splitKeys[i], splitKeys[i+1]));
489    }
490    return ret;
491  }
492
493  private static Map<byte[], Integer> createRegionSizesMap(final List<RegionInfo> regionInfos,
494    int... sizes) {
495    if (regionInfos.size() != sizes.length) {
496      throw new IllegalStateException("Parameter lengths must match.");
497    }
498
499    final Map<byte[], Integer> ret = new HashMap<>(regionInfos.size());
500    for (int i = 0; i < regionInfos.size(); i++) {
501      ret.put(regionInfos.get(i).getRegionName(), sizes[i]);
502    }
503    return ret;
504  }
