084  @Test
085  public void testPlanComparator() {
086    Comparator<NormalizationPlan> comparator = new SimpleRegionNormalizer.PlanComparator();
087    NormalizationPlan splitPlan1 = new SplitNormalizationPlan(null, null);
088    NormalizationPlan splitPlan2 = new SplitNormalizationPlan(null, null);
089    NormalizationPlan mergePlan1 = new MergeNormalizationPlan(null, null);
090    NormalizationPlan mergePlan2 = new MergeNormalizationPlan(null, null);
091
092    assertTrue(comparator.compare(splitPlan1, splitPlan2) == 0);
093    assertTrue(comparator.compare(splitPlan2, splitPlan1) == 0);
094    assertTrue(comparator.compare(mergePlan1, mergePlan2) == 0);
095    assertTrue(comparator.compare(mergePlan2, mergePlan1) == 0);
096    assertTrue(comparator.compare(splitPlan1, mergePlan1) < 0);
097    assertTrue(comparator.compare(mergePlan1, splitPlan1) > 0);
098  }

370  @SuppressWarnings("MockitoCast")
371  protected void setupMocksForNormalizer(Map<byte[], Integer> regionSizes,
372                                         List<RegionInfo> RegionInfo) {
373    masterServices = Mockito.mock(MasterServices.class, RETURNS_DEEP_STUBS);
374    masterRpcServices = Mockito.mock(MasterRpcServices.class, RETURNS_DEEP_STUBS);
375
376    // for simplicity all regions are assumed to be on one server; doesn't matter to us
377    ServerName sn = ServerName.valueOf("localhost", 0, 1L);
378    when(masterServices.getAssignmentManager().getRegionStates().
379      getRegionsOfTable(any())).thenReturn(RegionInfo);
380    when(masterServices.getAssignmentManager().getRegionStates().
381      getRegionServerOfRegion(any())).thenReturn(sn);
382
383    for (Map.Entry<byte[], Integer> region : regionSizes.entrySet()) {
384      RegionMetrics regionLoad = Mockito.mock(RegionMetrics.class);
385      when(regionLoad.getRegionName()).thenReturn(region.getKey());
386      when(regionLoad.getStoreFileSize())
387        .thenReturn(new Size(region.getValue(), Size.Unit.MEGABYTE));
388
389      // this is possibly broken with jdk9, unclear if false positive or not
390      // suppress it for now, fix it when we get to running tests on 9
391      // see: http://errorprone.info/bugpattern/MockitoCast
392      when((Object) masterServices.getServerManager().getLoad(sn).
393        getRegionMetrics().get(region.getKey())).thenReturn(regionLoad);
394    }
395    try {
396      when(masterRpcServices.isSplitOrMergeEnabled(any(),
397        any())).thenReturn(
398          IsSplitOrMergeEnabledResponse.newBuilder().setEnabled(true).build());
399    } catch (ServiceException se) {
400      LOG.debug("error setting isSplitOrMergeEnabled switch", se);
401    }
402
403    normalizer.setMasterServices(masterServices);
404    normalizer.setMasterRpcServices(masterRpcServices);
405  }
