051@Category({ MasterTests.class, MediumTests.class })
052public class TestRegionStateStore {

054  @ClassRule
055  public static final HBaseClassTestRule CLASS_RULE =
056    HBaseClassTestRule.forClass(TestRegionStateStore.class);

062  @Before
063  public void setup() throws Exception {
064    util = new HBaseTestingUtility();
065    util.startMiniCluster();

069  public void tearDown() throws Exception {
070    util.shutdownMiniCluster();

073  @Test
074  public void testVisitMetaForRegionExistingRegion() throws Exception {
075    final TableName tableName = TableName.valueOf("testVisitMetaForRegion");
076    util.createTable(tableName, "cf");
077    final List<HRegion> regions = util.getHBaseCluster().getRegions(tableName);
078    final String encodedName = regions.get(0).getRegionInfo().getEncodedName();
079    final RegionStateStore regionStateStore = util.getHBaseCluster().getMaster().
080      getAssignmentManager().getRegionStateStore();
081    final AtomicBoolean visitorCalled = new AtomicBoolean(false);
082    regionStateStore.visitMetaForRegion(encodedName, new RegionStateStore.RegionStateVisitor() {
083      @Override
084      public void visitRegionState(Result result, RegionInfo regionInfo, RegionState.State state,
085        ServerName regionLocation, ServerName lastHost, long openSeqNum) {
086        assertEquals(encodedName, regionInfo.getEncodedName());
087        visitorCalled.set(true);
088      }
089    });
090    assertTrue("Visitor has not been called.", visitorCalled.get());
091  }
092
093  @Test
094  public void testVisitMetaForBadRegionState() throws Exception {
095    final TableName tableName = TableName.valueOf("testVisitMetaForBadRegionState");
096    util.createTable(tableName, "cf");
097    final List<HRegion> regions = util.getHBaseCluster().getRegions(tableName);
098    final String encodedName = regions.get(0).getRegionInfo().getEncodedName();
099    final RegionStateStore regionStateStore = util.getHBaseCluster().getMaster().
100        getAssignmentManager().getRegionStateStore();
101
102    // add the BAD_STATE which does not exist in enum RegionState.State
103    Put put = new Put(regions.get(0).getRegionInfo().getRegionName(),
104        EnvironmentEdgeManager.currentTime());
105    put.addColumn(HConstants.CATALOG_FAMILY, HConstants.STATE_QUALIFIER,
106        Bytes.toBytes("BAD_STATE"));
107
108    try (Table table = util.getConnection().getTable(TableName.META_TABLE_NAME)) {
109      table.put(put);
110    }
111
112    final AtomicBoolean visitorCalled = new AtomicBoolean(false);
113    regionStateStore.visitMetaForRegion(encodedName, new RegionStateStore.RegionStateVisitor() {
114      @Override
115      public void visitRegionState(Result result, RegionInfo regionInfo,
116                                   RegionState.State state, ServerName regionLocation,
117                                   ServerName lastHost, long openSeqNum) {
118        assertEquals(encodedName, regionInfo.getEncodedName());
119        assertNull(state);
120        visitorCalled.set(true);
121      }
122    });
123    assertTrue("Visitor has not been called.", visitorCalled.get());
124  }
125
126  @Test
127  public void testVisitMetaForRegionNonExistingRegion() throws Exception {
128    final String encodedName = "fakeencodedregionname";
129    final RegionStateStore regionStateStore = util.getHBaseCluster().getMaster().
130      getAssignmentManager().getRegionStateStore();
131    final AtomicBoolean visitorCalled = new AtomicBoolean(false);
132    regionStateStore.visitMetaForRegion(encodedName, new RegionStateStore.RegionStateVisitor() {
133      @Override
134      public void visitRegionState(Result result, RegionInfo regionInfo, RegionState.State state,
135        ServerName regionLocation, ServerName lastHost, long openSeqNum) {
136        visitorCalled.set(true);
137      }
138    });
139    assertFalse("Visitor has been called, but it shouldn't.", visitorCalled.get());
140  }
