139  @Test(timeout=60000)
140  public void testSplitTableRegion() throws Exception {
141    final TableName tableName = TableName.valueOf(name.getMethodName());
142    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
143
144    RegionInfo [] regions = MasterProcedureTestingUtility.createTable(
145      procExec, tableName, null, ColumnFamilyName1, ColumnFamilyName2);
146    insertData(tableName);
147    int splitRowNum = startRowNum + rowCount / 2;
148    byte[] splitKey = Bytes.toBytes("" + splitRowNum);
149
150    assertTrue("not able to find a splittable region", regions != null);
151    assertTrue("not able to find a splittable region", regions.length == 1);
152
153    // collect AM metrics before test
154    collectAssignmentManagerMetrics();
155
156    // Split region of the table
157    long procId = procExec.submitProcedure(
158      new SplitTableRegionProcedure(procExec.getEnvironment(), regions[0], splitKey));
159    // Wait the completion
160    ProcedureTestingUtility.waitProcedure(procExec, procId);
161    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);

164

347    final TableName tableName = TableName.valueOf(name.getMethodName());
348    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
349
350    RegionInfo [] regions = MasterProcedureTestingUtility.createTable(
351      procExec, tableName, null, ColumnFamilyName1, ColumnFamilyName2);
352    insertData(tableName);
353    int splitRowNum = startRowNum + rowCount / 2;
354    byte[] splitKey = Bytes.toBytes("" + splitRowNum);
355
356    assertTrue("not able to find a splittable region", regions != null);
357    assertTrue("not able to find a splittable region", regions.length == 1);
358    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
359    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
360
361    // collect AM metrics before test
362    collectAssignmentManagerMetrics();
363
364    // Split region of the table
365    long procId = procExec.submitProcedure(
366      new SplitTableRegionProcedure(procExec.getEnvironment(), regions[0], splitKey));
367
368    // Failing before SPLIT_TABLE_REGION_UPDATE_META we should trigger the
369    // rollback
370    // NOTE: the 3 (number before SPLIT_TABLE_REGION_UPDATE_META step) is
371    // hardcoded, so you have to look at this test at least once when you add a new step.
372    int numberOfSteps = 3;

374    // check that we have only 1 region
375    assertEquals(1, UTIL.getHBaseAdmin().getTableRegions(tableName).size());
376    List<HRegion> daughters = UTIL.getMiniHBaseCluster().getRegions(tableName);
377    assertEquals(1, daughters.size());
378    verifyData(daughters.get(0), startRowNum, rowCount,
379    Bytes.toBytes(ColumnFamilyName1), Bytes.toBytes(ColumnFamilyName2));
380
381    assertEquals(splitSubmittedCount + 1, splitProcMetrics.getSubmittedCounter().getCount());
382    assertEquals(splitFailedCount + 1, splitProcMetrics.getFailedCounter().getCount());
383  }

451  private void insertData(final TableName tableName) throws IOException, InterruptedException {
452    Table t = UTIL.getConnection().getTable(tableName);
453    Put p;
454    for (int i= 0; i < rowCount / 2; i++) {
455      p = new Put(Bytes.toBytes("" + (startRowNum + i)));
456      p.addColumn(Bytes.toBytes(ColumnFamilyName1), Bytes.toBytes("q1"), Bytes.toBytes(i));
457      p.addColumn(Bytes.toBytes(ColumnFamilyName2), Bytes.toBytes("q2"), Bytes.toBytes(i));
458      t.put(p);
459      p = new Put(Bytes.toBytes("" + (startRowNum + rowCount - i - 1)));
460      p.addColumn(Bytes.toBytes(ColumnFamilyName1), Bytes.toBytes("q1"), Bytes.toBytes(i));
461      p.addColumn(Bytes.toBytes(ColumnFamilyName2), Bytes.toBytes("q2"), Bytes.toBytes(i));
462      t.put(p);
463      if (i % 5 == 0) {
464        UTIL.getAdmin().flush(tableName);
465      }
466    }
467  }
468
469  private void deleteData(
470      final TableName tableName,
471      final int startDeleteRowNum) throws IOException, InterruptedException {
472    Table t = UTIL.getConnection().getTable(tableName);
473    final int numRows = rowCount + startRowNum - startDeleteRowNum;
474    Delete d;
475    for (int i= startDeleteRowNum; i <= numRows + startDeleteRowNum; i++) {
476      d = new Delete(Bytes.toBytes("" + i));
477      t.delete(d);
478      if (i % 5 == 0) {
479        UTIL.getAdmin().flush(tableName);
480      }
481    }
482  }

484  private void verify(final TableName tableName, final int splitRowNum) throws IOException {
485    List<HRegion> daughters = UTIL.getMiniHBaseCluster().getRegions(tableName);
486    assertTrue(daughters.size() == 2);
487    LOG.info("Row Count = " + UTIL.countRows(tableName));
488    assertTrue(UTIL.countRows(tableName) == rowCount);
489    int startRow;
490    int numRows;
491    for (int i = 0; i < daughters.size(); i++) {
492      if (Bytes.compareTo(
493        daughters.get(i).getRegionInfo().getStartKey(), HConstants.EMPTY_BYTE_ARRAY) == 0) {
494        startRow = startRowNum; // first region
495        numRows = splitRowNum - startRowNum;
496      } else {
497        startRow = splitRowNum;
498        numRows = rowCount + startRowNum - splitRowNum;
499      }
500      verifyData(
501        daughters.get(i),
502        startRow,
503        numRows,
504        ColumnFamilyName1.getBytes(),
505        ColumnFamilyName2.getBytes());
506    }
507  }
508
509  private void verifyData(
510      final HRegion newReg,
511      final int startRow,
512      final int numRows,
513      final byte[]... families)
514      throws IOException {
515    for (int i = startRow; i < startRow + numRows; i++) {
516      byte[] row = Bytes.toBytes("" + i);
517      Get get = new Get(row);
518      Result result = newReg.get(get);
519      Cell[] raw = result.rawCells();
520      assertEquals(families.length, result.size());
521      for (int j = 0; j < families.length; j++) {
522        assertTrue(CellUtil.matchingRows(raw[j], row));
523        assertTrue(CellUtil.matchingFamily(raw[j], families[j]));
524      }
525    }
526  }
