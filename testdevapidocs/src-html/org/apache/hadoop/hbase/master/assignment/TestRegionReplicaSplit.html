055@Category({ RegionServerTests.class, LargeTests.class })
056public class TestRegionReplicaSplit {
057  @ClassRule
058  public static final HBaseClassTestRule CLASS_RULE =
059      HBaseClassTestRule.forClass(TestRegionReplicaSplit.class);
060  private static final Logger LOG = LoggerFactory.getLogger(TestRegionReplicaSplit.class);
061
062  private static final int NB_SERVERS = 4;
063  private static Table table;
064
065  private static final HBaseTestingUtility HTU = new HBaseTestingUtility();
066  private static final byte[] f = HConstants.CATALOG_FAMILY;
067
068  @BeforeClass
069  public static void beforeClass() throws Exception {
070    HTU.getConfiguration().setInt("hbase.master.wait.on.regionservers.mintostart", 3);
071    HTU.startMiniCluster(NB_SERVERS);
072    final TableName tableName = TableName.valueOf(TestRegionReplicaSplit.class.getSimpleName());
073
074    // Create table then get the single region for our new table.
075    createTable(tableName);
076  }
077
078  @Rule
079  public TestName name = new TestName();
080
081  private static void createTable(final TableName tableName) throws IOException {
082    TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(tableName);
083    builder.setRegionReplication(3);
084    // create a table with 3 replication
085
086    table = HTU.createTable(builder.build(), new byte[][] { f }, getSplits(2),
087      new Configuration(HTU.getConfiguration()));
088  }
089
090  private static byte[][] getSplits(int numRegions) {
091    RegionSplitter.UniformSplit split = new RegionSplitter.UniformSplit();
092    split.setFirstRow(Bytes.toBytes(0L));
093    split.setLastRow(Bytes.toBytes(Long.MAX_VALUE));
094    return split.split(numRegions);
095  }
096
097  @AfterClass
098  public static void afterClass() throws Exception {
099    HRegionServer.TEST_SKIP_REPORTING_TRANSITION = false;
100    table.close();
101    HTU.shutdownMiniCluster();
102  }
103
104  @Test
105  public void testRegionReplicaSplitRegionAssignment() throws Exception {
106    HTU.loadNumericRows(table, f, 0, 3);
107    // split the table
108    List<RegionInfo> regions = new ArrayList<RegionInfo>();
109    for (RegionServerThread rs : HTU.getMiniHBaseCluster().getRegionServerThreads()) {
110      for (Region r : rs.getRegionServer().getRegions(table.getName())) {
111        System.out.println("the region before split is is " + r.getRegionInfo()
112            + rs.getRegionServer().getServerName());
113        regions.add(r.getRegionInfo());
114      }
115    }
116    HTU.getAdmin().split(table.getName(), Bytes.toBytes(1));
117    int count = 0;
118    while (true) {
119      for (RegionServerThread rs : HTU.getMiniHBaseCluster().getRegionServerThreads()) {
120        for (Region r : rs.getRegionServer().getRegions(table.getName())) {
121          count++;
122        }
123      }
124      if (count >= 9) {
125        break;
126      }
127      count = 0;
128    }
129    List<ServerName> newRegionLocations = new ArrayList<ServerName>();
130    for (RegionServerThread rs : HTU.getMiniHBaseCluster().getRegionServerThreads()) {
131      RegionInfo prevInfo = null;
132      for (Region r : rs.getRegionServer().getRegions(table.getName())) {
133        if (!regions.contains(r.getRegionInfo())
134            && !RegionReplicaUtil.isDefaultReplica(r.getRegionInfo())) {
135          LOG.info("The region is " + r.getRegionInfo() + " the location is "
136              + rs.getRegionServer().getServerName());
137          if (!RegionReplicaUtil.isDefaultReplica(r.getRegionInfo())
138              && newRegionLocations.contains(rs.getRegionServer().getServerName())
139              && prevInfo != null
140              && Bytes.equals(prevInfo.getStartKey(), r.getRegionInfo().getStartKey())
141              && Bytes.equals(prevInfo.getEndKey(), r.getRegionInfo().getEndKey())) {
142            fail("Splitted regions should not be assigned to same region server");
143          } else {
144            prevInfo = r.getRegionInfo();
145            if (!RegionReplicaUtil.isDefaultReplica(r.getRegionInfo())
146                && !newRegionLocations.contains(rs.getRegionServer().getServerName())) {
147              newRegionLocations.add(rs.getRegionServer().getServerName());
148            }
149          }
150        }
151      }
152    }
153    // since we assign the daughter regions in round robin fashion, both the daugther region
154    // replicas will be assigned to two unique servers.
155    assertEquals("The new regions should be assigned to 3 unique servers ", 3,
156      newRegionLocations.size());
157  }
