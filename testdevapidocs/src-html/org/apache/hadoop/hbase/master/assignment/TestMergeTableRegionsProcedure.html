141  /**
142   * This tests two region merges
143   */
144  @Test
145  public void testMergeTwoRegions() throws Exception {
146    final TableName tableName = TableName.valueOf(this.name.getMethodName());
147    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
148
149    List<RegionInfo> tableRegions = createTable(tableName);
150
151    RegionInfo[] regionsToMerge = new RegionInfo[2];
152    regionsToMerge[0] = tableRegions.get(0);
153    regionsToMerge[1] = tableRegions.get(1);
154
155    // collect AM metrics before test
156    collectAssignmentManagerMetrics();
157
158    MergeTableRegionsProcedure proc =
159        new MergeTableRegionsProcedure(procExec.getEnvironment(), regionsToMerge, true);
160    long procId = procExec.submitProcedure(proc);
161    ProcedureTestingUtility.waitProcedure(procExec, procId);
162    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
163    assertRegionCount(tableName, initialRegionCount - 1);
164
165    assertEquals(mergeSubmittedCount + 1, mergeProcMetrics.getSubmittedCounter().getCount());
166    assertEquals(mergeFailedCount, mergeProcMetrics.getFailedCounter().getCount());
167    assertEquals(assignSubmittedCount + 1, assignProcMetrics.getSubmittedCounter().getCount());
168    assertEquals(assignFailedCount, assignProcMetrics.getFailedCounter().getCount());
169    assertEquals(unassignSubmittedCount + 2, unassignProcMetrics.getSubmittedCounter().getCount());
170    assertEquals(unassignFailedCount, unassignProcMetrics.getFailedCounter().getCount());
171
172    Pair<RegionInfo, RegionInfo> pair =
173      MetaTableAccessor.getRegionsFromMergeQualifier(UTIL.getConnection(),
174        proc.getMergedRegion().getRegionName());
175    assertTrue(pair.getFirst() != null && pair.getSecond() != null);
176
177    // Can I purge the merged regions from hbase:meta? Check that all went
178    // well by looking at the merged row up in hbase:meta. It should have no
179    // more mention of the merged regions; they are purged as last step in
180    // the merged regions cleanup.
181    UTIL.getHBaseCluster().getMaster().setCatalogJanitorEnabled(true);
182    UTIL.getHBaseCluster().getMaster().getCatalogJanitor().triggerNow();
183    while (pair != null && pair.getFirst() != null && pair.getSecond() != null) {
184      pair = MetaTableAccessor.getRegionsFromMergeQualifier(UTIL.getConnection(),
185          proc.getMergedRegion().getRegionName());
186    }
187  }

189  /**
190   * This tests two concurrent region merges
191   */
192  @Test
193  public void testMergeRegionsConcurrently() throws Exception {
194    final TableName tableName = TableName.valueOf("testMergeRegionsConcurrently");
195    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
196
197    List<RegionInfo> tableRegions = createTable(tableName);
198
199    RegionInfo[] regionsToMerge1 = new RegionInfo[2];
200    RegionInfo[] regionsToMerge2 = new RegionInfo[2];
201    regionsToMerge1[0] = tableRegions.get(0);
202    regionsToMerge1[1] = tableRegions.get(1);
203    regionsToMerge2[0] = tableRegions.get(2);
204    regionsToMerge2[1] = tableRegions.get(3);
205
206    // collect AM metrics before test
207    collectAssignmentManagerMetrics();
208
209    long procId1 = procExec.submitProcedure(new MergeTableRegionsProcedure(
210      procExec.getEnvironment(), regionsToMerge1, true));
211    long procId2 = procExec.submitProcedure(new MergeTableRegionsProcedure(
212      procExec.getEnvironment(), regionsToMerge2, true));
213    ProcedureTestingUtility.waitProcedure(procExec, procId1);
214    ProcedureTestingUtility.waitProcedure(procExec, procId2);
215    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
216    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
217    assertRegionCount(tableName, initialRegionCount - 2);
218
219    assertEquals(mergeSubmittedCount + 2, mergeProcMetrics.getSubmittedCounter().getCount());
220    assertEquals(mergeFailedCount, mergeProcMetrics.getFailedCounter().getCount());
221    assertEquals(assignSubmittedCount + 2, assignProcMetrics.getSubmittedCounter().getCount());
222    assertEquals(assignFailedCount, assignProcMetrics.getFailedCounter().getCount());
223    assertEquals(unassignSubmittedCount + 4, unassignProcMetrics.getSubmittedCounter().getCount());
224    assertEquals(unassignFailedCount, unassignProcMetrics.getFailedCounter().getCount());
225  }

246    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
247
248    assertRegionCount(tableName, initialRegionCount - 1);

252  public void testRollbackAndDoubleExecution() throws Exception {
253    final TableName tableName = TableName.valueOf("testRollbackAndDoubleExecution");
254    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
255
256    List<RegionInfo> tableRegions = createTable(tableName);
257
258    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
259    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
260
261    RegionInfo[] regionsToMerge = new RegionInfo[2];
262    regionsToMerge[0] = tableRegions.get(0);
263    regionsToMerge[1] = tableRegions.get(1);
264
265    long procId = procExec.submitProcedure(
266      new MergeTableRegionsProcedure(procExec.getEnvironment(), regionsToMerge, true));
267
268    // Failing before MERGE_TABLE_REGIONS_UPDATE_META we should trigger the rollback
269    // NOTE: the 5 (number before MERGE_TABLE_REGIONS_UPDATE_META step) is

275  private List<RegionInfo> createTable(final TableName tableName)
276      throws Exception {
277    HTableDescriptor desc = new HTableDescriptor(tableName);
278    desc.addFamily(new HColumnDescriptor(FAMILY));
279    byte[][] splitRows = new byte[initialRegionCount - 1][];
280    for (int i = 0; i < splitRows.length; ++i) {
281      splitRows[i] = Bytes.toBytes(String.format("%d", i));
282    }
283    admin.createTable(desc, splitRows);
284    return assertRegionCount(tableName, initialRegionCount);
285  }
286
287  public List<RegionInfo> assertRegionCount(final TableName tableName, final int nregions)
288      throws Exception {
289    UTIL.waitUntilNoRegionsInTransition();
290    List<RegionInfo> tableRegions = admin.getRegions(tableName);
291    assertEquals(nregions, tableRegions.size());
292    return tableRegions;
293  }
