046/**
047 * Testcase for HBASE-20792.
048 */
049@Category({ MediumTests.class, MasterTests.class })
050public class TestRegionMoveAndAbandon {
051  private static final Logger LOG = LoggerFactory.getLogger(TestRegionMoveAndAbandon.class);
052
053  @ClassRule
054  public static final HBaseClassTestRule CLASS_RULE =
055    HBaseClassTestRule.forClass(TestRegionMoveAndAbandon.class);
056
057  @Rule
058  public TestName name = new TestName();
059
060  private HBaseTestingUtil UTIL;
061  private SingleProcessHBaseCluster cluster;
062  private MiniZooKeeperCluster zkCluster;
063  private HRegionServer rs1;
064  private HRegionServer rs2;
065  private TableName tableName;
066  private RegionInfo regionInfo;
067
068  @Before
069  public void setup() throws Exception {
070    UTIL = new HBaseTestingUtil();
071    zkCluster = UTIL.startMiniZKCluster();
072    StartTestingClusterOption option =
073      StartTestingClusterOption.builder().numRegionServers(2).build();
074    cluster = UTIL.startMiniHBaseCluster(option);
075    rs1 = cluster.getRegionServer(0);
076    rs2 = cluster.getRegionServer(1);
077    assertEquals(2, cluster.getRegionServerThreads().size());
078    tableName = TableName.valueOf(name.getMethodName());
079    UTIL.createTable(tableName, Bytes.toBytes("cf"));
080    UTIL.waitTableAvailable(tableName, 30_000);
081    regionInfo = Iterables.getOnlyElement(cluster.getRegions(tableName)).getRegionInfo();
082  }
083
084  @After
085  public void teardown() throws Exception {
086    if (cluster != null) {
087      cluster.shutdown();
088      cluster = null;
089    }
090    if (zkCluster != null) {
091      zkCluster.shutdown();
092      zkCluster = null;
093    }
094  }
095
096  @Test
097  public void test() throws Exception {
098    LOG.info("Moving {} to {}", regionInfo, rs2.getServerName());
099    // Move to RS2
100    UTIL.moveRegionAndWait(regionInfo, rs2.getServerName());
101    LOG.info("Moving {} to {}", regionInfo, rs1.getServerName());
102    // Move to RS1
103    UTIL.moveRegionAndWait(regionInfo, rs1.getServerName());
104    LOG.info("Killing RS {}", rs1.getServerName());
105    // Stop RS1
106    cluster.killRegionServer(rs1.getServerName());
107    UTIL.waitFor(30_000, () -> rs1.isStopped() && !rs1.isAlive());
108    // Region should get moved to RS2
109    UTIL.waitTableAvailable(tableName, 60_000);
110    // Restart the master
111    LOG.info("Killing master {}", cluster.getMaster().getServerName());
112    cluster.killMaster(cluster.getMaster().getServerName());
113    // Stop RS2
114    LOG.info("Killing RS {}", rs2.getServerName());
115    cluster.killRegionServer(rs2.getServerName());
116    UTIL.waitFor(30_000, () -> rs2.isStopped() && !rs2.isAlive());
117    UTIL.waitFor(30_000, () -> rs1.isStopped() && !rs1.isAlive());
118    // make sure none of regionserver threads is alive.
119    UTIL.waitFor(30_000, () ->
120      UTIL.getMiniHBaseCluster().getLiveRegionServerThreads().isEmpty());
121    // Start up everything again
122    LOG.info("Starting cluster");
123    UTIL.getMiniHBaseCluster().startMaster();
124    UTIL.invalidateConnection();
125    UTIL.ensureSomeRegionServersAvailable(2);
126
127    UTIL.waitFor(30_000, new Waiter.Predicate<Exception>() {
128      @Override
129      public boolean evaluate() throws Exception {
130        try (Table nsTable = UTIL.getConnection().getTable(tableName)) {
131          // Doesn't matter what we're getting. We just want to make sure we can access the region
132          nsTable.get(new Get(Bytes.toBytes("a")));
133          return true;
134        }
135      }
136    });
137  }
138}
