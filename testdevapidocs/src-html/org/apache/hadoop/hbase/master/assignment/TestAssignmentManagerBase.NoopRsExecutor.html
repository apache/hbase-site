187  protected byte[] waitOnFuture(final Future<byte[]> future) throws Exception {
188    try {
189      return future.get(5, TimeUnit.SECONDS);
190    } catch (ExecutionException e) {
191      LOG.info("ExecutionException", e);
192      Exception ee = (Exception) e.getCause();
193      if (ee instanceof InterruptedIOException) {
194        for (Procedure<?> p : this.master.getMasterProcedureExecutor().getProcedures()) {
195          LOG.info(p.toStringDetails());
196        }
197      }
198      throw (Exception) e.getCause();
199    }

205  protected void bulkSubmit(TransitRegionStateProcedure[] procs) throws Exception {
206    Thread[] threads = new Thread[PROC_NTHREADS];
207    for (int i = 0; i < threads.length; ++i) {
208      final int threadId = i;
209      threads[i] = new Thread() {
210        @Override
211        public void run() {
212          TableName tableName = TableName.valueOf("table-" + threadId);
213          int n = (procs.length / threads.length);
214          int start = threadId * n;
215          int stop = start + n;
216          for (int j = start; j < stop; ++j) {
217            procs[j] = createAndSubmitAssign(tableName, j);
218          }
219        }
220      };
221      threads[i].start();
222    }
223    for (int i = 0; i < threads.length; ++i) {
224      threads[i].join();
225    }
226    for (int i = procs.length - 1; i >= 0 && procs[i] == null; --i) {
227      procs[i] = createAndSubmitAssign(TableName.valueOf("table-sync"), i);
228    }
229  }

231  protected TransitRegionStateProcedure createAndSubmitAssign(TableName tableName, int regionId) {
232    RegionInfo hri = createRegionInfo(tableName, regionId);
233    TransitRegionStateProcedure proc = createAssignProcedure(hri);
234    master.getMasterProcedureExecutor().submitProcedure(proc);
235    return proc;

247  protected TransitRegionStateProcedure createUnassignProcedure(RegionInfo hri) {
248    RegionStateNode regionNode = am.getRegionStates().getRegionStateNode(hri);
249    TransitRegionStateProcedure proc;
250    regionNode.lock();
251    try {
252      assertFalse(regionNode.isInTransition());
253      proc = TransitRegionStateProcedure
254        .unassign(master.getMasterProcedureExecutor().getEnvironment(), hri);
255      regionNode.setProcedure(proc);
256    } finally {
257      regionNode.unlock();
258    }
259    return proc;
260  }

285  protected class GoodRsExecutor extends NoopRsExecutor {
286    @Override
287    protected RegionOpeningState execOpenRegion(ServerName server, RegionOpenInfo openReq)
288        throws IOException {
289      RegionInfo hri = ProtobufUtil.toRegionInfo(openReq.getRegion());
290      long previousOpenSeqNum =
291        am.getRegionStates().getOrCreateRegionStateNode(hri).getOpenSeqNum();
292      sendTransitionReport(server, openReq.getRegion(), TransitionCode.OPENED,
293        previousOpenSeqNum + 2);
294      // Concurrency?
295      // Now update the state of our cluster in regionsToRegionServers.
296      SortedSet<byte[]> regions = regionsToRegionServers.get(server);
297      if (regions == null) {
298        regions = new ConcurrentSkipListSet<byte[]>(Bytes.BYTES_COMPARATOR);
299        regionsToRegionServers.put(server, regions);
300      }
301      if (regions.contains(hri.getRegionName())) {
302        throw new UnsupportedOperationException(hri.getRegionNameAsString());
303      }
304      regions.add(hri.getRegionName());
305      return RegionOpeningState.OPENED;
306    }
307
308    @Override
309    protected CloseRegionResponse execCloseRegion(ServerName server, byte[] regionName)
310        throws IOException {
311      RegionInfo hri = am.getRegionInfo(regionName);
312      sendTransitionReport(server, ProtobufUtil.toRegionInfo(hri), TransitionCode.CLOSED, -1);
313      return CloseRegionResponse.newBuilder().setClosed(true).build();
314    }

339  protected class SocketTimeoutRsExecutor extends GoodRsExecutor {
340    private final int maxSocketTimeoutRetries;
341    private final int maxServerRetries;
342
343    private ServerName lastServer;
344    private int sockTimeoutRetries;
345    private int serverRetries;
346
347    public SocketTimeoutRsExecutor(int maxSocketTimeoutRetries, int maxServerRetries) {
348      this.maxServerRetries = maxServerRetries;
349      this.maxSocketTimeoutRetries = maxSocketTimeoutRetries;
350    }
351
352    @Override
353    public ExecuteProceduresResponse sendRequest(ServerName server, ExecuteProceduresRequest req)
354        throws IOException {
355      // SocketTimeoutException should be a temporary problem
356      // unless the server will be declared dead.
357      if (sockTimeoutRetries++ < maxSocketTimeoutRetries) {
358        if (sockTimeoutRetries == 1) {
359          assertNotEquals(lastServer, server);
360        }
361        lastServer = server;
362        LOG.debug("Socket timeout for server=" + server + " retries=" + sockTimeoutRetries);
363        throw new SocketTimeoutException("simulate socket timeout");
364      } else if (serverRetries++ < maxServerRetries) {
365        LOG.info("Mark server=" + server + " as dead. serverRetries=" + serverRetries);
366        master.getServerManager().moveFromOnlineToDeadServers(server);
367        sockTimeoutRetries = 0;
368        throw new SocketTimeoutException("simulate socket timeout");
369      } else {
370        return super.sendRequest(server, req);
371      }
372    }
373  }

375  /**
376   * Takes open request and then returns nothing so acts like a RS that went zombie. No response (so
377   * proc is stuck/suspended on the Master and won't wake up.). We then send in a crash for this
378   * server after a few seconds; crash is supposed to take care of the suspended procedures.
379   */
380  protected class HangThenRSCrashExecutor extends GoodRsExecutor {
381    private int invocations;
382
383    @Override
384    protected RegionOpeningState execOpenRegion(final ServerName server, RegionOpenInfo openReq)
385        throws IOException {
386      if (this.invocations++ > 0) {
387        // Return w/o problem the second time through here.
388        return super.execOpenRegion(server, openReq);
389      }
390      // The procedure on master will just hang forever because nothing comes back
391      // from the RS in this case.
392      LOG.info("Return null response from serverName=" + server + "; means STUCK...TODO timeout");
393      executor.schedule(new Runnable() {
394        @Override
395        public void run() {
396          LOG.info("Sending in CRASH of " + server);
397          doCrash(server);
398        }
399      }, 1, TimeUnit.SECONDS);
400      return null;
401    }
402  }

404  /**
405   * Takes open request and then returns nothing so acts like a RS that went zombie. No response (so
406   * proc is stuck/suspended on the Master and won't wake up.). Different with
407   * HangThenRSCrashExecutor, HangThenRSCrashExecutor will create ServerCrashProcedure to handle the
408   * server crash. However, this HangThenRSRestartExecutor will restart RS directly, situation for
409   * RS crashed when SCP is not enabled.
410   */
411  protected class HangThenRSRestartExecutor extends GoodRsExecutor {
412    private int invocations;
413
414    @Override
415    protected RegionOpeningState execOpenRegion(final ServerName server, RegionOpenInfo openReq)
416        throws IOException {
417      if (this.invocations++ > 0) {
418        // Return w/o problem the second time through here.
419        return super.execOpenRegion(server, openReq);
420      }
421      // The procedure on master will just hang forever because nothing comes back
422      // from the RS in this case.
423      LOG.info("Return null response from serverName=" + server + "; means STUCK...TODO timeout");
424      executor.schedule(new Runnable() {
425        @Override
426        public void run() {
427          LOG.info("Restarting RS of " + server);
428          doRestart(server);
429        }
430      }, 1, TimeUnit.SECONDS);
431      return null;
432    }
433  }

436    public static final int TYPES_OF_FAILURE = 6;
437    private int invocations;
438
439    @Override
440    protected CloseRegionResponse execCloseRegion(ServerName server, byte[] regionName)

443        case 0:
444          throw new NotServingRegionException("Fake");

454        case 2:

458              LOG.info("Sending in CRASH of " + server);

463        case 3:
464          throw new ServerNotRunningYetException("Fake!");
465        case 4:
466          LOG.info("Returned null from serverName={}; means STUCK...TODO timeout", server);

476          return super.execCloseRegion(server, regionName);
477      }

481  protected class RandRsExecutor extends NoopRsExecutor {
482    private final Random rand = new Random();
483
484    @Override
485    public ExecuteProceduresResponse sendRequest(ServerName server, ExecuteProceduresRequest req)
486        throws IOException {
487      switch (rand.nextInt(5)) {
488        case 0:
489          throw new ServerNotRunningYetException("wait on server startup");
490        case 1:
491          throw new SocketTimeoutException("simulate socket timeout");
492        case 2:
493          throw new RemoteException("java.io.IOException", "unexpected exception");
494        default:
495          // fall out
496      }
497      return super.sendRequest(server, req);
498    }
499
500    @Override
501    protected RegionOpeningState execOpenRegion(final ServerName server, RegionOpenInfo openReq)
502        throws IOException {
503      RegionInfo hri = ProtobufUtil.toRegionInfo(openReq.getRegion());
504      long previousOpenSeqNum =
505        am.getRegionStates().getOrCreateRegionStateNode(hri).getOpenSeqNum();
506      switch (rand.nextInt(3)) {
507        case 0:
508          LOG.info("Return OPENED response");
509          sendTransitionReport(server, openReq.getRegion(), TransitionCode.OPENED,
510            previousOpenSeqNum + 2);
511          return OpenRegionResponse.RegionOpeningState.OPENED;
512        case 1:
513          LOG.info("Return transition report that FAILED_OPEN/FAILED_OPENING response");
514          sendTransitionReport(server, openReq.getRegion(), TransitionCode.FAILED_OPEN, -1);
515          return OpenRegionResponse.RegionOpeningState.FAILED_OPENING;
516        default:
517          // fall out
518      }
519      // The procedure on master will just hang forever because nothing comes back
520      // from the RS in this case.
521      LOG.info("Return null as response; means proc stuck so we send in a crash report after" +
522        " a few seconds...");
523      executor.schedule(new Runnable() {
524        @Override
525        public void run() {
526          LOG.info("Delayed CRASHING of " + server);
527          doCrash(server);
528        }
529      }, 5, TimeUnit.SECONDS);
530      return null;
531    }
532
533    @Override
534    protected CloseRegionResponse execCloseRegion(ServerName server, byte[] regionName)
535        throws IOException {
536      CloseRegionResponse.Builder resp = CloseRegionResponse.newBuilder();
537      boolean closed = rand.nextBoolean();
538      if (closed) {
539        RegionInfo hri = am.getRegionInfo(regionName);
540        sendTransitionReport(server, ProtobufUtil.toRegionInfo(hri), TransitionCode.CLOSED, -1);
541      }
542      resp.setClosed(closed);
543      return resp.build();
544    }
545  }
