217  protected void bulkSubmit(TransitRegionStateProcedure[] procs) throws Exception {
218    Thread[] threads = new Thread[PROC_NTHREADS];
219    for (int i = 0; i < threads.length; ++i) {
220      final int threadId = i;
221      threads[i] = new Thread() {
222        @Override
223        public void run() {
224          TableName tableName = TableName.valueOf("table-" + threadId);
225          int n = (procs.length / threads.length);
226          int start = threadId * n;
227          int stop = start + n;
228          for (int j = start; j < stop; ++j) {
229            procs[j] = createAndSubmitAssign(tableName, j);
230          }
231        }
232      };
233      threads[i].start();
234    }
235    for (int i = 0; i < threads.length; ++i) {
236      threads[i].join();
237    }
238    for (int i = procs.length - 1; i >= 0 && procs[i] == null; --i) {
239      procs[i] = createAndSubmitAssign(TableName.valueOf("table-sync"), i);
240    }
241  }

259  protected TransitRegionStateProcedure createUnassignProcedure(RegionInfo hri) {
260    RegionStateNode regionNode = am.getRegionStates().getRegionStateNode(hri);
261    TransitRegionStateProcedure proc;
262    regionNode.lock();
263    try {
264      assertFalse(regionNode.isInTransition());
265      proc = TransitRegionStateProcedure
266        .unassign(master.getMasterProcedureExecutor().getEnvironment(), hri);
267      regionNode.setProcedure(proc);
268    } finally {
269      regionNode.unlock();
270    }
271    return proc;
272  }

285  protected void doCrash(final ServerName serverName) {
286    this.master.getServerManager().moveFromOnlineToDeadServers(serverName);
287    this.am.submitServerCrash(serverName, false/* No WALs here */);
288    // add a new server to avoid killing all the region servers which may hang the UTs
289    ServerName newSn = ServerName.valueOf("localhost", 10000 + newRsAdded, 1);
290    newRsAdded++;
291    try {
292      this.master.getServerManager().regionServerReport(newSn, ServerMetricsBuilder.of(newSn));
293    } catch (YouAreDeadException e) {
294      // should not happen
295      throw new UncheckedIOException(e);
296    }
297  }

307  protected class GoodRsExecutor extends NoopRsExecutor {
308    @Override
309    protected RegionOpeningState execOpenRegion(ServerName server, RegionOpenInfo openReq)
310        throws IOException {
311      RegionInfo hri = ProtobufUtil.toRegionInfo(openReq.getRegion());
312      long previousOpenSeqNum =
313        am.getRegionStates().getOrCreateRegionStateNode(hri).getOpenSeqNum();
314      sendTransitionReport(server, openReq.getRegion(), TransitionCode.OPENED,
315        previousOpenSeqNum + 2);
316      // Concurrency?
317      // Now update the state of our cluster in regionsToRegionServers.
318      SortedSet<byte[]> regions = regionsToRegionServers.get(server);
319      if (regions == null) {
320        regions = new ConcurrentSkipListSet<byte[]>(Bytes.BYTES_COMPARATOR);
321        regionsToRegionServers.put(server, regions);
322      }
323      if (regions.contains(hri.getRegionName())) {
324        throw new UnsupportedOperationException(hri.getRegionNameAsString());
325      }
326      regions.add(hri.getRegionName());
327      return RegionOpeningState.OPENED;
328    }
329
330    @Override
331    protected CloseRegionResponse execCloseRegion(ServerName server, byte[] regionName)
332        throws IOException {
333      RegionInfo hri = am.getRegionInfo(regionName);
334      sendTransitionReport(server, ProtobufUtil.toRegionInfo(hri), TransitionCode.CLOSED, -1);
335      return CloseRegionResponse.newBuilder().setClosed(true).build();
336    }

361  protected class SocketTimeoutRsExecutor extends GoodRsExecutor {
362    private final int timeoutTimes;
363
364    private ServerName lastServer;
365    private int retries;
366
367    public SocketTimeoutRsExecutor(int timeoutTimes) {
368      this.timeoutTimes = timeoutTimes;
369    }
370
371    @Override
372    public ExecuteProceduresResponse sendRequest(ServerName server, ExecuteProceduresRequest req)
373        throws IOException {
374      // SocketTimeoutException should be a temporary problem
375      // unless the server will be declared dead.
376      retries++;
377      if (retries == 1) {
378        lastServer = server;
379      }
380      if (retries <= timeoutTimes) {
381        LOG.debug("Socket timeout for server=" + server + " retries=" + retries);
382        // should not change the server if the server is not dead yet.
383        assertEquals(lastServer, server);
384        if (retries == timeoutTimes) {
385          LOG.info("Mark server=" + server + " as dead. retries=" + retries);
386          master.getServerManager().moveFromOnlineToDeadServers(server);
387        }
388        throw new SocketTimeoutException("simulate socket timeout");
389      } else {
390        // should select another server
391        assertNotEquals(lastServer, server);
392        return super.sendRequest(server, req);
393      }
394    }
395  }

449  /**
450   * Takes open request and then returns nothing so acts like a RS that went zombie. No response (so
451   * proc is stuck/suspended on the Master and won't wake up.). We then send in a crash for this
452   * server after a few seconds; crash is supposed to take care of the suspended procedures.
453   */
454  protected class HangThenRSCrashExecutor extends GoodRsExecutor {
455    private int invocations;
456
457    @Override
458    protected RegionOpeningState execOpenRegion(final ServerName server, RegionOpenInfo openReq)
459        throws IOException {
460      if (this.invocations++ > 0) {
461        // Return w/o problem the second time through here.
462        return super.execOpenRegion(server, openReq);
463      }
464      // The procedure on master will just hang forever because nothing comes back
465      // from the RS in this case.
466      LOG.info("Return null response from serverName=" + server + "; means STUCK...TODO timeout");
467      executor.schedule(new Runnable() {
468        @Override
469        public void run() {
470          LOG.info("Sending in CRASH of " + server);
471          doCrash(server);
472        }
473      }, 1, TimeUnit.SECONDS);
474      return null;
475    }
476  }

555  protected class RandRsExecutor extends NoopRsExecutor {
556    private final Random rand = new Random();
557
558    @Override
559    public ExecuteProceduresResponse sendRequest(ServerName server, ExecuteProceduresRequest req)
560        throws IOException {
561      switch (rand.nextInt(5)) {
562        case 0:
563          throw new ServerNotRunningYetException("wait on server startup");
564        case 1:
565          throw new SocketTimeoutException("simulate socket timeout");
566        case 2:
567          throw new RemoteException("java.io.IOException", "unexpected exception");
568        default:
569          // fall out
570      }
571      return super.sendRequest(server, req);
572    }
573
574    @Override
575    protected RegionOpeningState execOpenRegion(final ServerName server, RegionOpenInfo openReq)
576        throws IOException {
577      RegionInfo hri = ProtobufUtil.toRegionInfo(openReq.getRegion());
578      long previousOpenSeqNum =
579        am.getRegionStates().getOrCreateRegionStateNode(hri).getOpenSeqNum();
580      switch (rand.nextInt(3)) {
581        case 0:
582          LOG.info("Return OPENED response");
583          sendTransitionReport(server, openReq.getRegion(), TransitionCode.OPENED,
584            previousOpenSeqNum + 2);
585          return OpenRegionResponse.RegionOpeningState.OPENED;
586        case 1:
587          LOG.info("Return transition report that FAILED_OPEN/FAILED_OPENING response");
588          sendTransitionReport(server, openReq.getRegion(), TransitionCode.FAILED_OPEN, -1);
589          return OpenRegionResponse.RegionOpeningState.FAILED_OPENING;
590        default:
591          // fall out
592      }
593      // The procedure on master will just hang forever because nothing comes back
594      // from the RS in this case.
595      LOG.info("Return null as response; means proc stuck so we send in a crash report after" +
596        " a few seconds...");
597      executor.schedule(new Runnable() {
598        @Override
599        public void run() {
600          LOG.info("Delayed CRASHING of " + server);
601          doCrash(server);
602        }
603      }, 5, TimeUnit.SECONDS);
604      return null;
605    }
606
607    @Override
608    protected CloseRegionResponse execCloseRegion(ServerName server, byte[] regionName)
609        throws IOException {
610      CloseRegionResponse.Builder resp = CloseRegionResponse.newBuilder();
611      boolean closed = rand.nextBoolean();
612      if (closed) {
613        RegionInfo hri = am.getRegionInfo(regionName);
614        sendTransitionReport(server, ProtobufUtil.toRegionInfo(hri), TransitionCode.CLOSED, -1);
615      }
616      resp.setClosed(closed);
617      return resp.build();
618    }
619  }
