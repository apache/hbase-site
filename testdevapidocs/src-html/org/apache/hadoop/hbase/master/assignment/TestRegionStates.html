089  private static void waitExecutorService(final int count) throws Exception {
090    for (int i = 0; i < count; ++i) {
091      executorService.take().get();
092    }

099  @Test
100  public void testRegionDoubleCreation() throws Exception {
101    // NOTE: RegionInfo sort by table first, so we are relying on that
102    final TableName TABLE_NAME_A = TableName.valueOf("testOrderedByTableA");
103    final TableName TABLE_NAME_B = TableName.valueOf("testOrderedByTableB");
104    final TableName TABLE_NAME_C = TableName.valueOf("testOrderedByTableC");
105    final RegionStates stateMap = new RegionStates();
106    final int NRUNS = 1000;
107    final int NSMALL_RUNS = 3;
108
109    // add some regions for table B
110    for (int i = 0; i < NRUNS; ++i) {
111      addRegionNode(stateMap, TABLE_NAME_B, i);
112    }
113    // re-add the regions for table B
114    for (int i = 0; i < NRUNS; ++i) {
115      addRegionNode(stateMap, TABLE_NAME_B, i);
116    }
117    waitExecutorService(NRUNS * 2);
118
119    // add two other tables A and C that will be placed before and after table B (sort order)
120    for (int i = 0; i < NSMALL_RUNS; ++i) {
121      addRegionNode(stateMap, TABLE_NAME_A, i);
122      addRegionNode(stateMap, TABLE_NAME_C, i);
123    }
124    waitExecutorService(NSMALL_RUNS * 2);
125    // check for the list of regions of the 3 tables
126    checkTableRegions(stateMap, TABLE_NAME_A, NSMALL_RUNS);
127    checkTableRegions(stateMap, TABLE_NAME_B, NRUNS);
128    checkTableRegions(stateMap, TABLE_NAME_C, NSMALL_RUNS);
129  }
130
131  private void checkTableRegions(final RegionStates stateMap,
132      final TableName tableName, final int nregions) {
133    List<RegionInfo> hris = stateMap.getRegionsOfTable(tableName, true);
134    assertEquals(nregions, hris.size());
135    for (int i = 1; i < hris.size(); ++i) {
136      long a = Bytes.toLong(hris.get(i - 1).getStartKey());
137      long b = Bytes.toLong(hris.get(i + 0).getStartKey());
138      assertEquals(b, a + 1);
139    }
140  }
141
142  private void addRegionNode(final RegionStates stateMap,
143      final TableName tableName, final long regionId) {
144    executorService.submit(new Callable<Object>() {
145      @Override
146      public Object call() {
147        return stateMap.getOrCreateRegionStateNode(RegionInfoBuilder.newBuilder(tableName)
148            .setStartKey(Bytes.toBytes(regionId))
149            .setEndKey(Bytes.toBytes(regionId + 1))
150            .setSplit(false)
151            .setRegionId(0)
152            .build());
153      }
154    });
155  }
156
157  private Object createRegionNode(final RegionStates stateMap,
158      final TableName tableName, final long regionId) {
159    return stateMap.getOrCreateRegionStateNode(createRegionInfo(tableName, regionId));
160  }
161
162  private RegionInfo createRegionInfo(final TableName tableName, final long regionId) {
163    return RegionInfoBuilder.newBuilder(tableName)
164        .setStartKey(Bytes.toBytes(regionId))
165        .setEndKey(Bytes.toBytes(regionId + 1))
166        .setSplit(false)
167        .setRegionId(0)
168        .build();
169  }

171  @Test
172  public void testPerf() throws Exception {
173    final TableName TABLE_NAME = TableName.valueOf("testPerf");
174    final int NRUNS = 1000000; // 1M
175    final RegionStates stateMap = new RegionStates();
176
177    long st = System.currentTimeMillis();
178    for (int i = 0; i < NRUNS; ++i) {
179      final int regionId = i;
180      executorService.submit(new Callable<Object>() {
181        @Override
182        public Object call() {
183          RegionInfo hri = createRegionInfo(TABLE_NAME, regionId);
184          return stateMap.getOrCreateRegionStateNode(hri);
185        }
186      });
187    }
188    waitExecutorService(NRUNS);
189    long et = System.currentTimeMillis();
190    LOG.info(String.format("PERF STATEMAP INSERT: %s %s/sec",
191      StringUtils.humanTimeDiff(et - st),
192      StringUtils.humanSize(NRUNS / ((et - st) / 1000.0f))));
193
194    st = System.currentTimeMillis();
195    for (int i = 0; i < NRUNS; ++i) {
196      final int regionId = i;
197      executorService.submit(new Callable<Object>() {
198        @Override
199        public Object call() {
200          RegionInfo hri = createRegionInfo(TABLE_NAME, regionId);
201          return stateMap.getRegionState(hri);
202        }
203      });
204    }
205
206    waitExecutorService(NRUNS);
207    et = System.currentTimeMillis();
208    LOG.info(String.format("PERF STATEMAP GET: %s %s/sec",
209      StringUtils.humanTimeDiff(et - st),
210      StringUtils.humanSize(NRUNS / ((et - st) / 1000.0f))));
211  }

214  public void testPerfSingleThread() {
215    final TableName TABLE_NAME = TableName.valueOf("testPerf");
216    final int NRUNS = 1 * 1000000; // 1M
217
218    final RegionStates stateMap = new RegionStates();

225        StringUtils.humanTimeDiff(et - st),

227  }
