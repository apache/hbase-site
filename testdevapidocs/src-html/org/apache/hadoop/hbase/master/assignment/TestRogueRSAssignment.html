083  private static final int initialRegionCount = 3;
084  private final static byte[] FAMILY = Bytes.toBytes("FAMILY");

086  private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
087  private static final Configuration conf = UTIL.getConfiguration();

089  private static MiniHBaseCluster cluster;

092  private static void setupConf(Configuration conf) {
093    // Reduce the maximum attempts to speed up the test
094    conf.setInt("hbase.assignment.maximum.attempts", 3);

096    conf.setInt("hbase.master.ping.server.retry.sleep.interval", 1);
097    conf.setInt(MasterProcedureConstants.MASTER_PROCEDURE_THREADS, 1);
098  }

102    setupConf(conf);

105    cluster = UTIL.getHBaseCluster();

107
108    admin = UTIL.getAdmin();
109    assertNotNull(admin);

111    master = cluster.getMaster();

139    admin.balancerSwitch(true, false);
140  }

142  /**
143   * Ignore this test, see HBASE-21421
144   */
145  @Test
146  @Ignore
147  public void testReportRSWithWrongRegion() throws Exception {
148    final TableName tableName = TableName.valueOf(this.name.getMethodName());
149
150    List<RegionInfo> tableRegions = createTable(tableName);
151
152    final ServerName sn = ServerName.parseVersionedServerName(
153        ServerName.valueOf("1.example.org", 1, EnvironmentEdgeManager.currentTime())
154          .getVersionedBytes());
155
156    // make fake request with a region assigned to different RS
157    RegionServerStatusProtos.RegionServerReportRequest.Builder request =
158        makeRSReportRequestWithRegions(sn, tableRegions.get(1));
159
160    // sending fake request to master
161    // TODO: replace YouAreDeadException with appropriate exception as and when necessary
162    exception.expect(ServiceException.class);
163    exception.expectCause(isA(YouAreDeadException.class));
164    RegionServerStatusProtos.RegionServerReportResponse response =
165        master.getMasterRpcServices().regionServerReport(null, request.build());
166  }
167
168  private RegionServerStatusProtos.RegionServerReportRequest.Builder
169      makeRSReportRequestWithRegions(final ServerName sn, RegionInfo... regions) {
170    ClusterStatusProtos.ServerLoad.Builder sl = ClusterStatusProtos.ServerLoad.newBuilder();
171    for (int i = 0; i < regions.length; i++) {
172      HBaseProtos.RegionSpecifier.Builder rs = HBaseProtos.RegionSpecifier.newBuilder();
173      rs.setType(HBaseProtos.RegionSpecifier.RegionSpecifierType.REGION_NAME);
174      rs.setValue(UnsafeByteOperations.unsafeWrap(regions[i].getRegionName()));
175
176      ClusterStatusProtos.RegionLoad.Builder rl = ClusterStatusProtos.RegionLoad.newBuilder()
177          .setRegionSpecifier(rs.build());
178
179      sl.addRegionLoads(i, rl.build());
180    }
181
182    return RegionServerStatusProtos.RegionServerReportRequest.newBuilder()
183              .setServer(ProtobufUtil.toServerName(sn))
184              .setLoad(sl);
185  }
186
187  private List<RegionInfo> createTable(final TableName tableName) throws Exception {
188    TableDescriptorBuilder tdBuilder = TableDescriptorBuilder.newBuilder(tableName);
189    tdBuilder.setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(FAMILY).build());
190
191    byte[][] rows = new byte[initialRegionCount - 1][];
192    for (int i = 0; i < rows.length; ++i) {
193      rows[i] = Bytes.toBytes(String.format("%d", i));
194    }
195    admin.createTable(tdBuilder.build(), rows);
196    return assertRegionCount(tableName, initialRegionCount);
197  }
198
199  private List<RegionInfo> assertRegionCount(final TableName tableName, final int nregions)
200      throws Exception {
201    UTIL.waitUntilNoRegionsInTransition();
202    List<RegionInfo> tableRegions = admin.getRegions(tableName);
203    assertEquals(nregions, tableRegions.size());
204    return tableRegions;
205  }
