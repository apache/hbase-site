178  /**
179   * Test that the contract from the master for checking on a snapshot are valid.
180   * <p>
181   * <ol>
182   * <li>If a snapshot fails with an error, we expect to get the source error.</li>
183   * <li>If there is no snapshot name supplied, we should get an error.</li>
184   * <li>If asking about a snapshot has hasn't occurred, you should get an error.</li>
185   * </ol>
186   */
187  @Test
188  public void testIsDoneContract() throws Exception {
189
190    IsSnapshotDoneRequest.Builder builder = IsSnapshotDoneRequest.newBuilder();
191
192    String snapshotName = "asyncExpectedFailureTest";
193
194    // check that we get an exception when looking up snapshot where one hasn't happened
195    SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
196      UnknownSnapshotException.class);
197
198    // and that we get the same issue, even if we specify a name
199    SnapshotDescription desc = SnapshotDescription.newBuilder()
200      .setName(snapshotName).setTable(TABLE_NAME.getNameAsString()).build();
201    builder.setSnapshot(desc);
202    SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
203      UnknownSnapshotException.class);
204
205    // set a mock handler to simulate a snapshot
206    DisabledTableSnapshotHandler mockHandler = Mockito.mock(DisabledTableSnapshotHandler.class);
207    Mockito.when(mockHandler.getException()).thenReturn(null);
208    Mockito.when(mockHandler.getSnapshot()).thenReturn(desc);
209    Mockito.when(mockHandler.isFinished()).thenReturn(Boolean.TRUE);
210    Mockito.when(mockHandler.getCompletionTimestamp())
211      .thenReturn(EnvironmentEdgeManager.currentTime());
212
213    master.getSnapshotManager()
214        .setSnapshotHandlerForTesting(TABLE_NAME, mockHandler);
215
216    // if we do a lookup without a snapshot name, we should fail - you should always know your name
217    builder = IsSnapshotDoneRequest.newBuilder();
218    SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
219      UnknownSnapshotException.class);
220
221    // then do the lookup for the snapshot that it is done
222    builder.setSnapshot(desc);
223    IsSnapshotDoneResponse response =
224      master.getMasterRpcServices().isSnapshotDone(null, builder.build());
225    assertTrue("Snapshot didn't complete when it should have.", response.getDone());
226
227    // now try the case where we are looking for a snapshot we didn't take
228    builder.setSnapshot(SnapshotDescription.newBuilder().setName("Not A Snapshot").build());
229    SnapshotTestingUtils.expectSnapshotDoneException(master, builder.build(),
230      UnknownSnapshotException.class);
231
232    // then create a snapshot to the fs and make sure that we can find it when checking done
233    snapshotName = "completed";
234    desc = createSnapshot(snapshotName);
235
236    builder.setSnapshot(desc);
237    response = master.getMasterRpcServices().isSnapshotDone(null, builder.build());
238    assertTrue("Completed, on-disk snapshot not found", response.getDone());
239  }

244    GetCompletedSnapshotsRequest request = GetCompletedSnapshotsRequest.newBuilder().build();
245    GetCompletedSnapshotsResponse response =

294  @Test
295  public void testGetCompletedSnapshotsWithCleanup() throws Exception {
296    // Enable auto snapshot cleanup for the cluster
297    SetSnapshotCleanupRequest setSnapshotCleanupRequest =
298        SetSnapshotCleanupRequest.newBuilder().setEnabled(true).build();
299    master.getMasterRpcServices().switchSnapshotCleanup(null, setSnapshotCleanupRequest);
300
301    // first check when there are no snapshots
302    GetCompletedSnapshotsRequest request = GetCompletedSnapshotsRequest.newBuilder().build();
303    GetCompletedSnapshotsResponse response =
304        master.getMasterRpcServices().getCompletedSnapshots(null, request);
305    assertEquals("Found unexpected number of snapshots", 0, response.getSnapshotsCount());
306
307    // NOTE: This is going to be flakey. Its timing based. For now made it more coarse
308    // so more likely to pass though we have to hang around longer.
309
310    // write one snapshot to the fs
311    createSnapshotWithTtl("snapshot_01", 5L);
312    createSnapshotWithTtl("snapshot_02", 100L);
313
314    // check that we get one snapshot
315    response = master.getMasterRpcServices().getCompletedSnapshots(null, request);
316    assertEquals("Found unexpected number of snapshots", 2, response.getSnapshotsCount());
317
318    // Check that 1 snapshot is auto cleaned after 5 sec of TTL expiration. Wait 10 seconds
319    // just in case.
320    Uninterruptibles.sleepUninterruptibly(10, TimeUnit.SECONDS);
321    response = master.getMasterRpcServices().getCompletedSnapshots(null, request);
322    assertEquals("Found unexpected number of snapshots", 1, response.getSnapshotsCount());
323  }

381  /**
382   * Test that the snapshot hfile archive cleaner works correctly. HFiles that are in snapshots
383   * should be retained, while those that are not in a snapshot should be deleted.
384   * @throws Exception on failure
385   */
386  @Test
387  public void testSnapshotHFileArchiving() throws Exception {
388    Admin admin = UTIL.getAdmin();
389    // make sure we don't fail on listing snapshots
390    SnapshotTestingUtils.assertNoSnapshots(admin);
391
392    // recreate test table with disabled compactions; otherwise compaction may happen before
393    // snapshot, the call after snapshot will be a no-op and checks will fail
394    UTIL.deleteTable(TABLE_NAME);
395    TableDescriptor td = TableDescriptorBuilder.newBuilder(TABLE_NAME)
396            .setColumnFamily(ColumnFamilyDescriptorBuilder.of(TEST_FAM))
397            .setCompactionEnabled(false)
398            .build();
399    UTIL.getAdmin().createTable(td);
400
401    // load the table
402    for (int i = 0; i < blockingStoreFiles / 2; i ++) {
403      UTIL.loadTable(UTIL.getConnection().getTable(TABLE_NAME), TEST_FAM);
404      UTIL.flush(TABLE_NAME);
405    }
406
407    // disable the table so we can take a snapshot
408    admin.disableTable(TABLE_NAME);
409
410    // take a snapshot of the table
411    String snapshotName = "snapshot";
412    String snapshotNameBytes = snapshotName;
413    admin.snapshot(snapshotName, TABLE_NAME);
414
415    LOG.info("After snapshot File-System state");
416    FSUtils.logFileSystemState(fs, rootDir, LOG);
417
418    // ensure we only have one snapshot
419    SnapshotTestingUtils.assertOneSnapshotThatMatches(admin, snapshotNameBytes, TABLE_NAME);
420
421    td = TableDescriptorBuilder.newBuilder(td)
422            .setCompactionEnabled(true)
423            .build();
424    // enable compactions now
425    admin.modifyTable(td);
426
427    // renable the table so we can compact the regions
428    admin.enableTable(TABLE_NAME);
429
430    // compact the files so we get some archived files for the table we just snapshotted
431    List<HRegion> regions = UTIL.getHBaseCluster().getRegions(TABLE_NAME);
432    for (HRegion region : regions) {
433      region.waitForFlushesAndCompactions(); // enable can trigger a compaction, wait for it.
434      region.compactStores(); // min is 2 so will compact and archive
435    }
436    List<RegionServerThread> regionServerThreads = UTIL.getMiniHBaseCluster()
437        .getRegionServerThreads();
438    HRegionServer hrs = null;
439    for (RegionServerThread rs : regionServerThreads) {
440      if (!rs.getRegionServer().getRegions(TABLE_NAME).isEmpty()) {
441        hrs = rs.getRegionServer();
442        break;
443      }
444    }
445    CompactedHFilesDischarger cleaner = new CompactedHFilesDischarger(100, null, hrs, false);
446    cleaner.chore();
447    LOG.info("After compaction File-System state");
448    FSUtils.logFileSystemState(fs, rootDir, LOG);
449
450    // make sure the cleaner has run
451    LOG.debug("Running hfile cleaners");
452    ensureHFileCleanersRun();
453    LOG.info("After cleaners File-System state: " + rootDir);
454    FSUtils.logFileSystemState(fs, rootDir, LOG);
455
456    // get the snapshot files for the table
457    Path snapshotTable = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
458    Set<String> snapshotHFiles = SnapshotReferenceUtil.getHFileNames(
459        UTIL.getConfiguration(), fs, snapshotTable);
460    // check that the files in the archive contain the ones that we need for the snapshot
461    LOG.debug("Have snapshot hfiles:");
462    for (String fileName : snapshotHFiles) {
463      LOG.debug(fileName);
464    }
465    // get the archived files for the table
466    Collection<String> archives = getHFiles(archiveDir, fs, TABLE_NAME);
467
468    // get the hfiles for the table
469    Collection<String> hfiles = getHFiles(rootDir, fs, TABLE_NAME);
470
471    // and make sure that there is a proper subset
472    for (String fileName : snapshotHFiles) {
473      boolean exist = archives.contains(fileName) || hfiles.contains(fileName);
474      assertTrue("Archived hfiles " + archives
475        + " and table hfiles " + hfiles + " is missing snapshot file:" + fileName, exist);
476    }
477
478    // delete the existing snapshot
479    admin.deleteSnapshot(snapshotNameBytes);
480    SnapshotTestingUtils.assertNoSnapshots(admin);
481
482    // make sure that we don't keep around the hfiles that aren't in a snapshot
483    // make sure we wait long enough to refresh the snapshot hfile
484    List<BaseHFileCleanerDelegate> delegates = UTIL.getMiniHBaseCluster().getMaster()
485        .getHFileCleaner().cleanersChain;
486    for (BaseHFileCleanerDelegate delegate: delegates) {
487      if (delegate instanceof SnapshotHFileCleaner) {
488        ((SnapshotHFileCleaner)delegate).getFileCacheForTesting().triggerCacheRefreshForTesting();
489      }
490    }
491    // run the cleaner again
492    LOG.debug("Running hfile cleaners");
493    ensureHFileCleanersRun();
494    LOG.info("After delete snapshot cleaners run File-System state");
495    FSUtils.logFileSystemState(fs, rootDir, LOG);
496
497    archives = getHFiles(archiveDir, fs, TABLE_NAME);
498    assertEquals("Still have some hfiles in the archive, when their snapshot has been deleted.", 0,
499      archives.size());
500  }
