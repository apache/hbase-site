119  /**
120   * This tests verifies LogCleaner works correctly with WALs and Procedure WALs located
121   * in the same oldWALs directory.
122   * Created files:
123   * - 2 invalid files
124   * - 5 old Procedure WALs
125   * - 30 old WALs from which 3 are in replication
126   * - 5 recent Procedure WALs
127   * - 1 recent WAL
128   * - 1 very new WAL (timestamp in future)
129   * - masterProcedureWALs subdirectory
130   * Files which should stay:
131   * - 3 replication WALs
132   * - 2 new WALs
133   * - 5 latest Procedure WALs
134   * - masterProcedureWALs subdirectory
135   */
136  @Test
137  public void testLogCleaning() throws Exception {
138    // set TTLs
139    long ttlWAL = 2000;
140    long ttlProcedureWAL = 4000;
141    conf.setLong("hbase.master.logcleaner.ttl", ttlWAL);
142    conf.setLong("hbase.master.procedurewalcleaner.ttl", ttlProcedureWAL);
143
144    HMaster.decorateMasterConfiguration(conf);
145    Server server = new DummyServer();
146    ReplicationQueueStorage queueStorage =
147        ReplicationStorageFactory.getReplicationQueueStorage(server.getZooKeeper(), conf);
148
149    String fakeMachineName = URLEncoder.encode(
150        server.getServerName().toString(), StandardCharsets.UTF_8.name());
151
152    final FileSystem fs = FileSystem.get(conf);
153    fs.mkdirs(OLD_PROCEDURE_WALS_DIR);
154
155    final long now = EnvironmentEdgeManager.currentTime();
156
157    // Case 1: 2 invalid files, which would be deleted directly
158    fs.createNewFile(new Path(OLD_WALS_DIR, "a"));
159    fs.createNewFile(new Path(OLD_WALS_DIR, fakeMachineName + "." + "a"));
160
161    // Case 2: 5 Procedure WALs that are old which would be deleted
162    for (int i = 1; i <= 5; i++) {
163      final Path fileName =
164          new Path(OLD_PROCEDURE_WALS_DIR, String.format("pv2-%020d.log", i));
165      fs.createNewFile(fileName);
166    }
167
168    // Sleep for sometime to get old procedure WALs
169    Thread.sleep(ttlProcedureWAL - ttlWAL);
170
171    // Case 3: old WALs which would be deletable
172    for (int i = 1; i <= 30; i++) {
173      Path fileName = new Path(OLD_WALS_DIR, fakeMachineName + "." + (now - i));
174      fs.createNewFile(fileName);
175      // Case 4: put 3 WALs in ZK indicating that they are scheduled for replication so these
176      // files would pass TimeToLiveLogCleaner but would be rejected by ReplicationLogCleaner
177      if (i % (30 / 3) == 0) {
178        queueStorage.addWAL(server.getServerName(), fakeMachineName, fileName.getName());
179        LOG.info("Replication log file: " + fileName);
180      }
181    }
182
183    // Case 5: 5 Procedure WALs that are new, will stay
184    for (int i = 6; i <= 10; i++) {
185      Path fileName =
186          new Path(OLD_PROCEDURE_WALS_DIR, String.format("pv2-%020d.log", i));
187      fs.createNewFile(fileName);
188    }
189
190    // Sleep for sometime to get newer modification time
191    Thread.sleep(ttlWAL);
192    fs.createNewFile(new Path(OLD_WALS_DIR, fakeMachineName + "." + now));
193
194    // Case 6: 1 newer WAL, not even deletable for TimeToLiveLogCleaner,
195    // so we are not going down the chain
196    fs.createNewFile(new Path(OLD_WALS_DIR, fakeMachineName + "." + (now + ttlWAL)));
197
198    FileStatus[] status = fs.listStatus(OLD_WALS_DIR);
199    LOG.info("File status: {}", Arrays.toString(status));
200
201    // There should be 34 files and 1 masterProcedureWALs directory
202    assertEquals(35, fs.listStatus(OLD_WALS_DIR).length);
203    // 10 procedure WALs
204    assertEquals(10, fs.listStatus(OLD_PROCEDURE_WALS_DIR).length);
205
206    LogCleaner cleaner = new LogCleaner(1000, server, conf, fs, OLD_WALS_DIR, POOL, null);
207    cleaner.chore();
208
209    // In oldWALs we end up with the current WAL, a newer WAL, the 3 old WALs which
210    // are scheduled for replication and masterProcedureWALs directory
211    TEST_UTIL.waitFor(1000, (Waiter.Predicate<Exception>) () -> 6 == fs
212        .listStatus(OLD_WALS_DIR).length);
213    // In masterProcedureWALs we end up with 5 newer Procedure WALs
214    TEST_UTIL.waitFor(1000, (Waiter.Predicate<Exception>) () -> 5 == fs
215        .listStatus(OLD_PROCEDURE_WALS_DIR).length);
216
217    if (LOG.isDebugEnabled()) {
218      FileStatus[] statusOldWALs = fs.listStatus(OLD_WALS_DIR);
219      FileStatus[] statusProcedureWALs = fs.listStatus(OLD_PROCEDURE_WALS_DIR);
220      LOG.debug("Kept log file for oldWALs: {}", Arrays.toString(statusOldWALs));
221      LOG.debug("Kept log file for masterProcedureWALs: {}",
222          Arrays.toString(statusProcedureWALs));
223    }
224  }

226  @Test
227  public void testZooKeeperRecoveryDuringGetListOfReplicators() throws Exception {
228    ReplicationLogCleaner cleaner = new ReplicationLogCleaner();
229
230    List<FileStatus> dummyFiles = Arrays.asList(
231      new FileStatus(100, false, 3, 100, EnvironmentEdgeManager.currentTime(), new Path("log1")),
232      new FileStatus(100, false, 3, 100, EnvironmentEdgeManager.currentTime(), new Path("log2"))
233    );
234
235    FaultyZooKeeperWatcher faultyZK =
236        new FaultyZooKeeperWatcher(conf, "testZooKeeperAbort-faulty", null);
237    final AtomicBoolean getListOfReplicatorsFailed = new AtomicBoolean(false);
238
239    try {
240      faultyZK.init(false);
241      ReplicationQueueStorage queueStorage = spy(ReplicationStorageFactory
242          .getReplicationQueueStorage(faultyZK, conf));
243      doAnswer(new Answer<Object>() {
244        @Override
245        public Object answer(InvocationOnMock invocation) throws Throwable {
246          try {
247            return invocation.callRealMethod();
248          } catch (ReplicationException e) {
249            LOG.debug("Caught Exception", e);
250            getListOfReplicatorsFailed.set(true);
251            throw e;
252          }
253        }
254      }).when(queueStorage).getAllWALs();
255
256      cleaner.setConf(conf, faultyZK, queueStorage);
257      // should keep all files due to a ConnectionLossException getting the queues znodes
258      cleaner.preClean();
259      Iterable<FileStatus> toDelete = cleaner.getDeletableFiles(dummyFiles);
260
261      assertTrue(getListOfReplicatorsFailed.get());
262      assertFalse(toDelete.iterator().hasNext());
263      assertFalse(cleaner.isStopped());
264
265      //zk recovery.
266      faultyZK.init(true);
267      cleaner.preClean();
268      Iterable<FileStatus> filesToDelete = cleaner.getDeletableFiles(dummyFiles);
269      Iterator<FileStatus> iter = filesToDelete.iterator();
270      assertTrue(iter.hasNext());
271      assertEquals(new Path("log1"), iter.next().getPath());
272      assertTrue(iter.hasNext());
273      assertEquals(new Path("log2"), iter.next().getPath());
274      assertFalse(iter.hasNext());
275
276    } finally {
277      faultyZK.close();
278    }
279  }

285  @Test
286  public void testZooKeeperNormal() throws Exception {
287    ReplicationLogCleaner cleaner = new ReplicationLogCleaner();
288
289    // Subtract 1000 from current time so modtime is for sure older
290    // than 'now'.
291    long modTime = EnvironmentEdgeManager.currentTime() - 1000;
292    List<FileStatus> dummyFiles = Arrays.asList(
293        new FileStatus(100, false, 3, 100, modTime, new Path("log1")),
294        new FileStatus(100, false, 3, 100, modTime, new Path("log2"))
295    );
296
297    ZKWatcher zkw = new ZKWatcher(conf, "testZooKeeperAbort-normal", null);
298    try {
299      cleaner.setConf(conf, zkw);
300      cleaner.preClean();
301      Iterable<FileStatus> filesToDelete = cleaner.getDeletableFiles(dummyFiles);
302      Iterator<FileStatus> iter = filesToDelete.iterator();
303      assertTrue(iter.hasNext());
304      assertEquals(new Path("log1"), iter.next().getPath());
305      assertTrue(iter.hasNext());
306      assertEquals(new Path("log2"), iter.next().getPath());
307      assertFalse(iter.hasNext());
308    } finally {
309      zkw.close();
310    }

313  @Test
314  public void testOnConfigurationChange() throws Exception {
315    // Prepare environments
316    Server server = new DummyServer();
317
318    FileSystem fs = TEST_UTIL.getDFSCluster().getFileSystem();
319    LogCleaner cleaner = new LogCleaner(3000, server, conf, fs, OLD_WALS_DIR, POOL, null);
320    int size = cleaner.getSizeOfCleaners();
321    assertEquals(LogCleaner.DEFAULT_OLD_WALS_CLEANER_THREAD_TIMEOUT_MSEC,
322        cleaner.getCleanerThreadTimeoutMsec());
323    // Create dir and files for test
324    int numOfFiles = 10;
325    createFiles(fs, OLD_WALS_DIR, numOfFiles);
326    FileStatus[] status = fs.listStatus(OLD_WALS_DIR);
327    assertEquals(numOfFiles, status.length);
328    // Start cleaner chore
329    Thread thread = new Thread(() -> cleaner.chore());
330    thread.setDaemon(true);
331    thread.start();
332    // change size of cleaners dynamically
333    int sizeToChange = 4;
334    long threadTimeoutToChange = 30 * 1000L;
335    conf.setInt(LogCleaner.OLD_WALS_CLEANER_THREAD_SIZE, size + sizeToChange);
336    conf.setLong(LogCleaner.OLD_WALS_CLEANER_THREAD_TIMEOUT_MSEC, threadTimeoutToChange);
337    cleaner.onConfigurationChange(conf);
338    assertEquals(sizeToChange + size, cleaner.getSizeOfCleaners());
339    assertEquals(threadTimeoutToChange, cleaner.getCleanerThreadTimeoutMsec());
340    // Stop chore
341    thread.join();
342    status = fs.listStatus(OLD_WALS_DIR);
343    assertEquals(0, status.length);
344  }

346  private void createFiles(FileSystem fs, Path parentDir, int numOfFiles) throws IOException {
347    for (int i = 0; i < numOfFiles; i++) {
348      // size of each file is 1M, 2M, or 3M
349      int xMega = 1 + ThreadLocalRandom.current().nextInt(1, 4);
350      try (FSDataOutputStream fsdos = fs.create(new Path(parentDir, "file-" + i))) {
351        byte[] M = RandomUtils.nextBytes(Math.toIntExact(FileUtils.ONE_MB * xMega));
352        fsdos.write(M);
353      }
354    }
355  }

375  static class FaultyZooKeeperWatcher extends ZKWatcher {
376    private RecoverableZooKeeper zk;
377
378    public FaultyZooKeeperWatcher(Configuration conf, String identifier, Abortable abortable)
379        throws ZooKeeperConnectionException, IOException {
380      super(conf, identifier, abortable);
381    }
382
383    public void init(boolean autoRecovery) throws Exception {
384      this.zk = spy(super.getRecoverableZooKeeper());
385      if (!autoRecovery) {
386        doThrow(new KeeperException.ConnectionLossException())
387          .when(zk).getChildren("/hbase/replication/rs", null);
388      }
389    }
390
391    @Override
392    public RecoverableZooKeeper getRecoverableZooKeeper() {
393      return zk;
394    }
