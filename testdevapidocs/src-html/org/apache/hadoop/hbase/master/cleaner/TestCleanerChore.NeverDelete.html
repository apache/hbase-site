138    AlwaysDelete delegate = (AlwaysDelete) chore.cleanersChain.get(0);

141

152    chore.chore();
153    // make sure we never checked the directory
154    Mockito.verify(spy, Mockito.never()).isFileDeletable(fStat);
155    Mockito.reset(spy);

184  /**
185   * While cleaning a directory, all the files in the directory may be deleted, but there may be
186   * another file added, in which case the directory shouldn't be deleted.
187   * @throws IOException on failure
188   */
189  @Test
190  public void testCleanerDoesNotDeleteDirectoryWithLateAddedFiles() throws IOException {
191    Stoppable stop = new StoppableImplementation();
192    Configuration conf = UTIL.getConfiguration();
193    final Path testDir = UTIL.getDataTestDir();
194    final FileSystem fs = UTIL.getTestFileSystem();
195    String confKey = "hbase.test.cleaner.delegates";
196    conf.set(confKey, AlwaysDelete.class.getName());
197
198    AllValidPaths chore = new AllValidPaths("test-file-cleaner", stop, conf, fs, testDir, confKey);
199    // spy on the delegate to ensure that we don't check for directories
200    AlwaysDelete delegate = (AlwaysDelete) chore.cleanersChain.get(0);
201    AlwaysDelete spy = Mockito.spy(delegate);
202    chore.cleanersChain.set(0, spy);
203
204    // create the directory layout in the directory to clean
205    final Path parent = new Path(testDir, "parent");
206    Path file = new Path(parent, "someFile");
207    fs.mkdirs(parent);
208    // touch a new file
209    fs.create(file).close();
210    assertTrue("Test file didn't get created.", fs.exists(file));
211    final Path addedFile = new Path(parent, "addedFile");
212
213    // when we attempt to delete the original file, add another file in the same directory
214    Mockito.doAnswer(new Answer<Boolean>() {
215      @Override
216      public Boolean answer(InvocationOnMock invocation) throws Throwable {
217        fs.create(addedFile).close();
218        FSUtils.logFileSystemState(fs, testDir, LOG);
219        return (Boolean) invocation.callRealMethod();
220      }
221    }).when(spy).isFileDeletable(Mockito.any());
222
223    // run the chore
224    chore.chore();
225
226    // make sure all the directories + added file exist, but the original file is deleted
227    assertTrue("Added file unexpectedly deleted", fs.exists(addedFile));
228    assertTrue("Parent directory deleted unexpectedly", fs.exists(parent));
229    assertFalse("Original file unexpectedly retained", fs.exists(file));
230    Mockito.verify(spy, Mockito.times(1)).isFileDeletable(Mockito.any());
231    Mockito.reset(spy);
232  }

357  @Test
358  public void testOnConfigurationChange() throws Exception {
359    Stoppable stop = new StoppableImplementation();
360    Configuration conf = UTIL.getConfiguration();
361    Path testDir = UTIL.getDataTestDir();
362    FileSystem fs = UTIL.getTestFileSystem();
363    String confKey = "hbase.test.cleaner.delegates";
364    conf.set(confKey, AlwaysDelete.class.getName());
365    conf.set(CleanerChore.CHORE_POOL_SIZE, "2");
366    AllValidPaths chore = new AllValidPaths("test-file-cleaner", stop, conf, fs, testDir, confKey);
367    chore.setEnabled(true);
368    // Create subdirs under testDir
369    int dirNums = 6;
370    Path[] subdirs = new Path[dirNums];
371    for (int i = 0; i < dirNums; i++) {
372      subdirs[i] = new Path(testDir, "subdir-" + i);
373      fs.mkdirs(subdirs[i]);
374    }
375    // Under each subdirs create 6 files
376    for (Path subdir : subdirs) {
377      createFiles(fs, subdir, 6);
378    }
379    // Start chore
380    Thread t = new Thread(() -> chore.chore());
381    t.setDaemon(true);
382    t.start();
383    // Change size of chore's pool
384    conf.set(CleanerChore.CHORE_POOL_SIZE, "4");
385    chore.onConfigurationChange(conf);
386    assertEquals(4, chore.getChorePoolSize());
387    // Stop chore
388    t.join();
389  }

391  private void createFiles(FileSystem fs, Path parentDir, int numOfFiles) throws IOException {
392    Random random = new Random();
393    for (int i = 0; i < numOfFiles; i++) {
394      int xMega = 1 + random.nextInt(3); // size of each file is between 1~3M
395      try (FSDataOutputStream fsdos = fs.create(new Path(parentDir, "file-" + i))) {
396        for (int m = 0; m < xMega; m++) {
397          byte[] M = new byte[1024 * 1024];
398          random.nextBytes(M);
399          fsdos.write(M);
400        }
401      }
402    }
403  }

405  private static class AllValidPaths extends CleanerChore<BaseHFileCleanerDelegate> {
406
407    public AllValidPaths(String name, Stoppable s, Configuration conf, FileSystem fs,
408        Path oldFileDir, String confkey) {
409      super(name, Integer.MAX_VALUE, s, conf, fs, oldFileDir, confkey);
410    }
411
412    // all paths are valid
413    @Override
414    protected boolean validate(Path file) {
415      return true;
416    }
417  };
