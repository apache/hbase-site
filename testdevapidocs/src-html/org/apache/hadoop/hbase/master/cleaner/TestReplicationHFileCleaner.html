116  @Test
117  public void testIsFileDeletable() throws IOException, ReplicationException {
118    // 1. Create a file
119    Path file = new Path(root, "testIsFileDeletableWithNoHFileRefs");
120    fs.createNewFile(file);
121    // 2. Assert file is successfully created
122    assertTrue("Test file not created!", fs.exists(file));
123    ReplicationHFileCleaner cleaner = new ReplicationHFileCleaner();
124    cleaner.setConf(conf);
125    // 3. Assert that file as is should be deletable
126    assertTrue("Cleaner should allow to delete this file as there is no hfile reference node "
127        + "for it in the queue.",
128      cleaner.isFileDeletable(fs.getFileStatus(file)));
129
130    List<String> files = new ArrayList<String>(1);
131    files.add(file.getName());
132    // 4. Add the file to hfile-refs queue
133    rq.addHFileRefs(peerId, files);
134    // 5. Assert file should not be deletable
135    assertFalse("Cleaner should not allow to delete this file as there is a hfile reference node "
136        + "for it in the queue.",
137      cleaner.isFileDeletable(fs.getFileStatus(file)));
138  }

140  @Test
141  public void testGetDeletableFiles() throws Exception {
142    // 1. Create two files and assert that they do not exist
143    Path notDeletablefile = new Path(root, "testGetDeletableFiles_1");
144    fs.createNewFile(notDeletablefile);
145    assertTrue("Test file not created!", fs.exists(notDeletablefile));
146    Path deletablefile = new Path(root, "testGetDeletableFiles_2");
147    fs.createNewFile(deletablefile);
148    assertTrue("Test file not created!", fs.exists(deletablefile));
149
150    List<FileStatus> files = new ArrayList<FileStatus>(2);
151    FileStatus f = new FileStatus();
152    f.setPath(deletablefile);
153    files.add(f);
154    f = new FileStatus();
155    f.setPath(notDeletablefile);
156    files.add(f);
157
158    List<String> hfiles = new ArrayList<>(1);
159    hfiles.add(notDeletablefile.getName());
160    // 2. Add one file to hfile-refs queue
161    rq.addHFileRefs(peerId, hfiles);
162
163    ReplicationHFileCleaner cleaner = new ReplicationHFileCleaner();
164    cleaner.setConf(conf);
165    Iterator<FileStatus> deletableFilesIterator = cleaner.getDeletableFiles(files).iterator();
166    int i = 0;
167    while (deletableFilesIterator.hasNext() && i < 2) {
168      i++;
169    }
170    // 5. Assert one file should not be deletable and it is present in the list returned
171    if (i > 2) {
172      fail("File " + notDeletablefile
173          + " should not be deletable as its hfile reference node is not added.");
174    }
175    assertTrue(deletableFilesIterator.next().getPath().equals(deletablefile));
176  }

196    Class<? extends ReplicationHFileCleaner> cleanerClass = cleaner.getClass();
197    Field rqc = cleanerClass.getDeclaredField("rqc");
198    rqc.setAccessible(true);
199    rqc.set(cleaner, replicationQueuesClient);
200
201    cleaner.isFileDeletable(fs.getFileStatus(file));
202  }
