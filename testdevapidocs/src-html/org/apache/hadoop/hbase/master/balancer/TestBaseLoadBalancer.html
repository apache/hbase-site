139  /**
140   * Tests the bulk assignment used during cluster startup.
141   *
142   * Round-robin. Should yield a balanced cluster so same invariant as the load
143   * balancer holds, all servers holding either floor(avg) or ceiling(avg).
144   *
145   * @throws Exception
146   */
147  @Test (timeout=180000)
148  public void testBulkAssignment() throws Exception {
149    List<ServerName> tmp = getListOfServerNames(randomServers(5, 0));
150    List<RegionInfo> hris = randomRegions(20);
151    hris.add(RegionInfoBuilder.FIRST_META_REGIONINFO);
152    tmp.add(master);
153    Map<ServerName, List<RegionInfo>> plans = loadBalancer.roundRobinAssignment(hris, tmp);
154    if (LoadBalancer.isTablesOnMaster(loadBalancer.getConf())) {
155      assertTrue(plans.get(master).contains(RegionInfoBuilder.FIRST_META_REGIONINFO));
156      assertEquals(1, plans.get(master).size());
157    }
158    int totalRegion = 0;
159    for (List<RegionInfo> regions: plans.values()) {
160      totalRegion += regions.size();
161    }
162    assertEquals(hris.size(), totalRegion);
163    for (int[] mock : regionsAndServersMocks) {
164      LOG.debug("testBulkAssignment with " + mock[0] + " regions and " + mock[1] + " servers");
165      List<RegionInfo> regions = randomRegions(mock[0]);
166      List<ServerAndLoad> servers = randomServers(mock[1], 0);
167      List<ServerName> list = getListOfServerNames(servers);
168      Map<ServerName, List<RegionInfo>> assignments =
169          loadBalancer.roundRobinAssignment(regions, list);
170      float average = (float) regions.size() / servers.size();
171      int min = (int) Math.floor(average);
172      int max = (int) Math.ceil(average);
173      if (assignments != null && !assignments.isEmpty()) {
174        for (List<RegionInfo> regionList : assignments.values()) {
175          assertTrue(regionList.size() == min || regionList.size() == max);
176        }
177      }
178      returnRegions(regions);
179      returnServers(list);
180    }
181  }

270  @Test (timeout=180000)
271  public void testRegionAvailability() throws Exception {
272    // Create a cluster with a few servers, assign them to specific racks
273    // then assign some regions. The tests should check whether moving a
274    // replica from one node to a specific other node or rack lowers the
275    // availability of the region or not
276
277    List<RegionInfo> list0 = new ArrayList<>();
278    List<RegionInfo> list1 = new ArrayList<>();
279    List<RegionInfo> list2 = new ArrayList<>();
280    // create a region (region1)
281    RegionInfo hri1 = RegionInfoBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
282        .setStartKey("key1".getBytes())
283        .setEndKey("key2".getBytes())
284        .setSplit(false)
285        .setRegionId(100)
286        .build();
287    // create a replica of the region (replica_of_region1)
288    RegionInfo hri2 = RegionReplicaUtil.getRegionInfoForReplica(hri1, 1);
289    // create a second region (region2)
290    RegionInfo hri3 = RegionInfoBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
291        .setStartKey("key2".getBytes())
292        .setEndKey("key3".getBytes())
293        .setSplit(false)
294        .setRegionId(101)
295        .build();
296    list0.add(hri1); //only region1
297    list1.add(hri2); //only replica_of_region1
298    list2.add(hri3); //only region2
299    Map<ServerName, List<RegionInfo>> clusterState = new LinkedHashMap<>();
300    clusterState.put(servers[0], list0); //servers[0] hosts region1
301    clusterState.put(servers[1], list1); //servers[1] hosts replica_of_region1
302    clusterState.put(servers[2], list2); //servers[2] hosts region2
303    // create a cluster with the above clusterState. The way in which the
304    // cluster is created (constructor code) would make sure the indices of
305    // the servers are in the order in which it is inserted in the clusterState
306    // map (linkedhashmap is important). A similar thing applies to the region lists
307    Cluster cluster = new Cluster(clusterState, null, null, rackManager);
308    // check whether a move of region1 from servers[0] to servers[1] would lower
309    // the availability of region1
310    assertTrue(cluster.wouldLowerAvailability(hri1, servers[1]));
311    // check whether a move of region1 from servers[0] to servers[2] would lower
312    // the availability of region1
313    assertTrue(!cluster.wouldLowerAvailability(hri1, servers[2]));
314    // check whether a move of replica_of_region1 from servers[0] to servers[2] would lower
315    // the availability of replica_of_region1
316    assertTrue(!cluster.wouldLowerAvailability(hri2, servers[2]));
317    // check whether a move of region2 from servers[0] to servers[1] would lower
318    // the availability of region2
319    assertTrue(!cluster.wouldLowerAvailability(hri3, servers[1]));
320
321    // now lets have servers[1] host replica_of_region2
322    list1.add(RegionReplicaUtil.getRegionInfoForReplica(hri3, 1));
323    // create a new clusterState with the above change
324    cluster = new Cluster(clusterState, null, null, rackManager);
325    // now check whether a move of a replica from servers[0] to servers[1] would lower
326    // the availability of region2
327    assertTrue(cluster.wouldLowerAvailability(hri3, servers[1]));
328
329    // start over again
330    clusterState.clear();
331    clusterState.put(servers[0], list0); //servers[0], rack1 hosts region1
332    clusterState.put(servers[5], list1); //servers[5], rack2 hosts replica_of_region1 and replica_of_region2
333    clusterState.put(servers[6], list2); //servers[6], rack2 hosts region2
334    clusterState.put(servers[10], new ArrayList<>()); //servers[10], rack3 hosts no region
335    // create a cluster with the above clusterState
336    cluster = new Cluster(clusterState, null, null, rackManager);
337    // check whether a move of region1 from servers[0],rack1 to servers[6],rack2 would
338    // lower the availability
339
340    assertTrue(cluster.wouldLowerAvailability(hri1, servers[0]));
341
342    // now create a cluster without the rack manager
343    cluster = new Cluster(clusterState, null, null, null);
344    // now repeat check whether a move of region1 from servers[0] to servers[6] would
345    // lower the availability
346    assertTrue(!cluster.wouldLowerAvailability(hri1, servers[6]));
347  }

418  /**
419   * Asserts a valid retained assignment plan.
420   * <p>
421   * Must meet the following conditions:
422   * <ul>
423   * <li>Every input region has an assignment, and to an online server
424   * <li>If a region had an existing assignment to a server with the same
425   * address a a currently online server, it will be assigned to it
426   * </ul>
427   * @param existing
428   * @param servers
429   * @param assignment
430   */
431  private void assertRetainedAssignment(Map<RegionInfo, ServerName> existing,
432      List<ServerName> servers, Map<ServerName, List<RegionInfo>> assignment) {
433    // Verify condition 1, every region assigned, and to online server
434    Set<ServerName> onlineServerSet = new TreeSet<>(servers);
435    Set<RegionInfo> assignedRegions = new TreeSet<>(RegionInfo.COMPARATOR);
436    for (Map.Entry<ServerName, List<RegionInfo>> a : assignment.entrySet()) {
437      assertTrue("Region assigned to server that was not listed as online",
438        onlineServerSet.contains(a.getKey()));
439      for (RegionInfo r : a.getValue())
440        assignedRegions.add(r);
441    }
442    assertEquals(existing.size(), assignedRegions.size());
443
444    // Verify condition 2, if server had existing assignment, must have same
445    Set<String> onlineHostNames = new TreeSet<>();
446    for (ServerName s : servers) {
447      onlineHostNames.add(s.getHostname());
448    }
449
450    for (Map.Entry<ServerName, List<RegionInfo>> a : assignment.entrySet()) {
451      ServerName assignedTo = a.getKey();
452      for (RegionInfo r : a.getValue()) {
453        ServerName address = existing.get(r);
454        if (address != null && onlineHostNames.contains(address.getHostname())) {
455          // this region was prevously assigned somewhere, and that
456          // host is still around, then it should be re-assigned on the
457          // same host
458          assertEquals(address.getHostname(), assignedTo.getHostname());
459        }
460      }
461    }
462  }
