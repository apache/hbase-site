120  @Test
121  public void testPickers() throws Exception {
122    TableName tableName = TableName.valueOf(name.getMethodName());
123    ColumnFamilyDescriptor columnFamilyDescriptor =
124        ColumnFamilyDescriptorBuilder.newBuilder(HConstants.CATALOG_FAMILY).build();
125    TableDescriptor desc = TableDescriptorBuilder
126        .newBuilder(tableName)
127        .setColumnFamily(columnFamilyDescriptor)
128        .build();
129    admin.createTable(desc, Bytes.toBytes("aaa"), Bytes.toBytes("zzz"), REGIONS);
130    TEST_UTIL.waitUntilAllRegionsAssigned(tableName);
131    TEST_UTIL.loadTable(admin.getConnection().getTable(tableName), HConstants.CATALOG_FAMILY);
132    admin.flush(tableName);
133
134    HMaster master = cluster.getMaster();
135    FavoredNodesManager fnm = master.getFavoredNodesManager();
136    ServerName masterServerName = master.getServerName();
137    List<ServerName> excludedServers = Lists.newArrayList(masterServerName);
138    final ServerName mostLoadedServer = getRSWithMaxRegions(tableName, excludedServers);
139    assertNotNull(mostLoadedServer);
140    int numRegions = getTableRegionsFromServer(tableName, mostLoadedServer).size();
141    excludedServers.add(mostLoadedServer);
142    // Lets find another server with more regions to calculate number of regions to move
143    ServerName source = getRSWithMaxRegions(tableName, excludedServers);
144    assertNotNull(source);
145    int regionsToMove = getTableRegionsFromServer(tableName, source).size()/2;
146
147    // Since move only works if the target is part of favored nodes of the region, lets get all
148    // regions that are movable to mostLoadedServer
149    List<RegionInfo> hris = getRegionsThatCanBeMoved(tableName, mostLoadedServer);
150    RegionStates rst = master.getAssignmentManager().getRegionStates();
151    for (int i = 0; i < regionsToMove; i++) {
152      final RegionInfo regionInfo = hris.get(i);
153      admin.move(regionInfo.getEncodedNameAsBytes(), mostLoadedServer);
154      LOG.info("Moving region: " + hris.get(i).getRegionNameAsString() + " to " + mostLoadedServer);
155      TEST_UTIL.waitFor(60000, new Waiter.Predicate<Exception>() {
156        @Override
157        public boolean evaluate() throws Exception {
158          return ServerName.isSameAddress(
159              rst.getRegionServerOfRegion(regionInfo), mostLoadedServer);
160        }
161      });
162    }
163    final int finalRegions = numRegions + regionsToMove;
164    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
165    TEST_UTIL.waitFor(60000, new Waiter.Predicate<Exception>() {
166      @Override
167      public boolean evaluate() throws Exception {
168        int numRegions = getTableRegionsFromServer(tableName, mostLoadedServer).size();
169        return (numRegions == finalRegions);
170      }
171    });
172    TEST_UTIL.getHBaseCluster().startRegionServerAndWait(60000);
173
174    Map<ServerName, List<RegionInfo>> serverAssignments = Maps.newHashMap();
175    ClusterMetrics status = admin.getClusterMetrics(EnumSet.of(Option.LIVE_SERVERS));
176    for (ServerName sn : status.getLiveServerMetrics().keySet()) {
177      if (!ServerName.isSameAddress(sn, masterServerName)) {
178        serverAssignments.put(sn, getTableRegionsFromServer(tableName, sn));
179      }
180    }
181    RegionHDFSBlockLocationFinder regionFinder = new RegionHDFSBlockLocationFinder();
182    regionFinder.setClusterMetrics(admin.getClusterMetrics(EnumSet.of(Option.LIVE_SERVERS)));
183    regionFinder.setConf(conf);
184    regionFinder.setClusterInfoProvider(
185      new MasterClusterInfoProvider(TEST_UTIL.getMiniHBaseCluster().getMaster()));
186    BalancerClusterState cluster =
187      new BalancerClusterState(serverAssignments, null, regionFinder, new RackManager(conf));
188    LoadOnlyFavoredStochasticBalancer balancer = (LoadOnlyFavoredStochasticBalancer) TEST_UTIL
189      .getMiniHBaseCluster().getMaster().getLoadBalancer().getInternalBalancer();
190
191    cluster.sortServersByRegionCount();
192    Integer[] servers = cluster.serverIndicesSortedByRegionCount;
193    LOG.info("Servers sorted by region count:" + Arrays.toString(servers));
194    LOG.info("Cluster dump: " + cluster);
195    if (!mostLoadedServer.equals(cluster.servers[servers[servers.length -1]])) {
196      LOG.error("Most loaded server: " + mostLoadedServer + " does not match: "
197          + cluster.servers[servers[servers.length -1]]);
198    }
199    assertEquals(mostLoadedServer, cluster.servers[servers[servers.length - 1]]);
200    FavoredStochasticBalancer.FavoredNodeLoadPicker loadPicker =
201      balancer.new FavoredNodeLoadPicker();
202    boolean userRegionPicked = false;
203    for (int i = 0; i < 100; i++) {
204      if (userRegionPicked) {
205        break;
206      } else {
207        BalanceAction action = loadPicker.generate(cluster);
208        if (action.getType() == BalanceAction.Type.MOVE_REGION) {
209          MoveRegionAction moveRegionAction = (MoveRegionAction) action;
210          RegionInfo region = cluster.regions[moveRegionAction.getRegion()];
211          assertNotEquals(-1, moveRegionAction.getToServer());
212          ServerName destinationServer = cluster.servers[moveRegionAction.getToServer()];
213          assertEquals(cluster.servers[moveRegionAction.getFromServer()], mostLoadedServer);
214          if (!region.getTable().isSystemTable()) {
215            List<ServerName> favNodes = fnm.getFavoredNodes(region);
216            assertTrue(favNodes.contains(
217              ServerName.valueOf(destinationServer.getAddress(), -1)));
218            userRegionPicked = true;
219          }
220        }
221      }
222    }
223    assertTrue("load picker did not pick expected regions in 100 iterations.", userRegionPicked);
224  }

226  /*
227   * A region can only be moved to one of its favored node. Hence this method helps us to
228   * get that list which makes it easy to write non-flaky tests.
229   */
230  private List<RegionInfo> getRegionsThatCanBeMoved(TableName tableName,
231      ServerName serverName) {
232    List<RegionInfo> regions = Lists.newArrayList();
233    RegionStates rst = cluster.getMaster().getAssignmentManager().getRegionStates();
234    FavoredNodesManager fnm = cluster.getMaster().getFavoredNodesManager();
235    for (RegionInfo regionInfo : fnm.getRegionsOfFavoredNode(serverName)) {
236      if (regionInfo.getTable().equals(tableName) &&
237          !ServerName.isSameAddress(rst.getRegionServerOfRegion(regionInfo), serverName)) {
238        regions.add(regionInfo);
239      }
240    }
241    return regions;
242  }
243
244  private List<RegionInfo> getTableRegionsFromServer(TableName tableName, ServerName source)
245      throws IOException {
246    List<RegionInfo> regionInfos = Lists.newArrayList();
247    HRegionServer regionServer = cluster.getRegionServer(source);
248    for (Region region : regionServer.getRegions(tableName)) {
249      regionInfos.add(region.getRegionInfo());
250    }
251    return regionInfos;
252  }
253
254  private ServerName getRSWithMaxRegions(TableName tableName, List<ServerName> excludeNodes)
255      throws IOException {
256
257    int maxRegions = 0;
258    ServerName maxLoadedServer = null;
259
260    for (JVMClusterUtil.RegionServerThread rst : cluster.getLiveRegionServerThreads()) {
261      List<HRegion> regions = rst.getRegionServer().getRegions(tableName);
262      LOG.debug("Server: " + rst.getRegionServer().getServerName() + " regions: " + regions.size());
263      if (regions.size() > maxRegions) {
264        if (excludeNodes == null ||
265            !doesMatchExcludeNodes(excludeNodes, rst.getRegionServer().getServerName())) {
266          maxRegions = regions.size();
267          maxLoadedServer = rst.getRegionServer().getServerName();
268        }
269      }
270    }
271    return maxLoadedServer;
272  }
273
274  private boolean doesMatchExcludeNodes(List<ServerName> excludeNodes, ServerName sn) {
275    for (ServerName excludeSN : excludeNodes) {
276      if (ServerName.isSameAddress(sn, excludeSN)) {
277        return true;
278      }
279    }
280    return false;
281  }
