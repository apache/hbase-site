077  /**
078   * Invariant is that all servers of a group have load between floor(avg) and
079   * ceiling(avg) number of regions.
080   */
081  protected void assertClusterAsBalanced(
082      ArrayListMultimap<String, ServerAndLoad> groupLoadMap) {
083    for (String gName : groupLoadMap.keySet()) {
084      List<ServerAndLoad> groupLoad = groupLoadMap.get(gName);
085      int numServers = groupLoad.size();
086      int numRegions = 0;
087      int maxRegions = 0;
088      int minRegions = Integer.MAX_VALUE;
089      for (ServerAndLoad server : groupLoad) {
090        int nr = server.getLoad();
091        if (nr > maxRegions) {
092          maxRegions = nr;
093        }
094        if (nr < minRegions) {
095          minRegions = nr;
096        }
097        numRegions += nr;
098      }
099      if (maxRegions - minRegions < 2) {
100        // less than 2 between max and min, can't balance
101        return;
102      }
103      int min = numRegions / numServers;
104      int max = numRegions % numServers == 0 ? min : min + 1;
105
106      for (ServerAndLoad server : groupLoad) {
107        assertTrue(server.getLoad() <= max);
108        assertTrue(server.getLoad() >= min);
109      }
110    }
111  }

132  /**
133   * Asserts a valid retained assignment plan.
134   * <p>
135   * Must meet the following conditions:
136   * <ul>
137   * <li>Every input region has an assignment, and to an online server
138   * <li>If a region had an existing assignment to a server with the same
139   * address a a currently online server, it will be assigned to it
140   * </ul>
141   */
142  protected void assertRetainedAssignment(
143      Map<RegionInfo, ServerName> existing, List<ServerName> servers,
144      Map<ServerName, List<RegionInfo>> assignment)
145      throws FileNotFoundException, IOException {
146    // Verify condition 1, every region assigned, and to online server
147    Set<ServerName> onlineServerSet = new TreeSet<>(servers);
148    Set<RegionInfo> assignedRegions = new TreeSet<>(RegionInfo.COMPARATOR);
149    for (Map.Entry<ServerName, List<RegionInfo>> a : assignment.entrySet()) {
150      assertTrue(
151          "Region assigned to server that was not listed as online",
152          onlineServerSet.contains(a.getKey()));
153      for (RegionInfo r : a.getValue()) {
154        assignedRegions.add(r);
155      }
156    }
157    assertEquals(existing.size(), assignedRegions.size());
158
159    // Verify condition 2, every region must be assigned to correct server.
160    Set<String> onlineHostNames = new TreeSet<>();
161    for (ServerName s : servers) {
162      onlineHostNames.add(s.getHostname());
163    }
164
165    for (Map.Entry<ServerName, List<RegionInfo>> a : assignment.entrySet()) {
166      ServerName currentServer = a.getKey();
167      for (RegionInfo r : a.getValue()) {
168        ServerName oldAssignedServer = existing.get(r);
169        TableName tableName = r.getTable();
170        String groupName =
171            tableDescs.get(tableName).getRegionServerGroup().orElse(RSGroupInfo.DEFAULT_GROUP);
172        assertTrue(StringUtils.isNotEmpty(groupName));
173        RSGroupInfo gInfo = getMockedGroupInfoManager().getRSGroup(groupName);
174        assertTrue("Region is not correctly assigned to group servers.",
175          gInfo.containsServer(currentServer.getAddress()));
176        if (oldAssignedServer != null &&
177            onlineHostNames.contains(oldAssignedServer.getHostname())) {
178          // this region was previously assigned somewhere, and that
179          // host is still around, then the host must have been is a
180          // different group.
181          if (!oldAssignedServer.getAddress().equals(currentServer.getAddress())) {
182            assertFalse(gInfo.containsServer(oldAssignedServer.getAddress()));
183          }
184        }
185      }
186    }
187  }

189  protected String printStats(
190      ArrayListMultimap<String, ServerAndLoad> groupBasedLoad) {
191    StringBuilder sb = new StringBuilder();
192    sb.append("\n");
193    for (String groupName : groupBasedLoad.keySet()) {
194      sb.append("Stats for group: " + groupName);
195      sb.append("\n");
196      sb.append(groupMap.get(groupName).getServers());
197      sb.append("\n");
198      List<ServerAndLoad> groupLoad = groupBasedLoad.get(groupName);
199      int numServers = groupLoad.size();
200      int totalRegions = 0;
201      sb.append("Per Server Load: \n");
202      for (ServerAndLoad sLoad : groupLoad) {
203        sb.append("Server :" + sLoad.getServerName() + " Load : "
204            + sLoad.getLoad() + "\n");
205        totalRegions += sLoad.getLoad();
206      }
207      sb.append(" Group Statistics : \n");
208      float average = (float) totalRegions / numServers;
209      int max = (int) Math.ceil(average);
210      int min = (int) Math.floor(average);
211      sb.append("[srvr=" + numServers + " rgns=" + totalRegions + " avg="
212          + average + " max=" + max + " min=" + min + "]");
213      sb.append("\n");
214      sb.append("===============================");
215      sb.append("\n");
216    }
217    return sb.toString();
218  }

220  protected ArrayListMultimap<String, ServerAndLoad> convertToGroupBasedMap(
221      final Map<ServerName, List<RegionInfo>> serversMap) throws IOException {
222    ArrayListMultimap<String, ServerAndLoad> loadMap = ArrayListMultimap
223        .create();
224    for (RSGroupInfo gInfo : getMockedGroupInfoManager().listRSGroups()) {
225      Set<Address> groupServers = gInfo.getServers();
226      for (Address hostPort : groupServers) {
227        ServerName actual = null;
228        for(ServerName entry: servers) {
229          if(entry.getAddress().equals(hostPort)) {
230            actual = entry;
231            break;
232          }
233        }
234        List<RegionInfo> regions = serversMap.get(actual);
235        assertTrue("No load for " + actual, regions != null);
236        loadMap.put(gInfo.getName(),
237            new ServerAndLoad(actual, regions.size()));
238      }
239    }
240    return loadMap;
241  }

243  protected ArrayListMultimap<String, ServerAndLoad> reconcile(
244      ArrayListMultimap<String, ServerAndLoad> previousLoad,
245      List<RegionPlan> plans) {
246    ArrayListMultimap<String, ServerAndLoad> result = ArrayListMultimap
247        .create();
248    result.putAll(previousLoad);
249    if (plans != null) {
250      for (RegionPlan plan : plans) {
251        ServerName source = plan.getSource();
252        updateLoad(result, source, -1);
253        ServerName destination = plan.getDestination();
254        updateLoad(result, destination, +1);
255      }
256    }
257    return result;
258  }
259
260  protected void updateLoad(
261      ArrayListMultimap<String, ServerAndLoad> previousLoad,
262      final ServerName sn, final int diff) {
263    for (String groupName : previousLoad.keySet()) {
264      ServerAndLoad newSAL = null;
265      ServerAndLoad oldSAL = null;
266      for (ServerAndLoad sal : previousLoad.get(groupName)) {
267        if (ServerName.isSameAddress(sn, sal.getServerName())) {
268          oldSAL = sal;
269          newSAL = new ServerAndLoad(sn, sal.getLoad() + diff);
270          break;
271        }
272      }
273      if (newSAL != null) {
274        previousLoad.remove(groupName, oldSAL);
275        previousLoad.put(groupName, newSAL);
276        break;
277      }
278    }
279  }

281  protected Map<ServerName, List<RegionInfo>> mockClusterServers() throws IOException {
282    assertTrue(servers.size() == regionAssignment.length);
283    Map<ServerName, List<RegionInfo>> assignment = new TreeMap<>();
284    for (int i = 0; i < servers.size(); i++) {
285      int numRegions = regionAssignment[i];
286      List<RegionInfo> regions = assignedRegions(numRegions, servers.get(i));
287      assignment.put(servers.get(i), regions);
288    }
289    return assignment;
290  }

292  /**
293   * Generate a list of regions evenly distributed between the tables.
294   *
295   * @param numRegions The number of regions to be generated.
296   * @return List of RegionInfo.
297   */
298  protected List<RegionInfo> randomRegions(int numRegions) {
299    List<RegionInfo> regions = new ArrayList<>(numRegions);
300    byte[] start = new byte[16];
301    byte[] end = new byte[16];
302    rand.nextBytes(start);
303    rand.nextBytes(end);
304    int regionIdx = rand.nextInt(tables.length);
305    for (int i = 0; i < numRegions; i++) {
306      Bytes.putInt(start, 0, numRegions << 1);
307      Bytes.putInt(end, 0, (numRegions << 1) + 1);
308      int tableIndex = (i + regionIdx) % tables.length;
309      regions.add(RegionInfoBuilder.newBuilder(tables[tableIndex])
310          .setStartKey(start)
311          .setEndKey(end)
312          .setSplit(false)
313          .setRegionId(regionId++)
314          .build());
315    }
316    return regions;
317  }

319  /**
320   * Generate assigned regions to a given server using group information.
321   *
322   * @param numRegions the num regions to generate
323   * @param sn the servername
324   * @return the list of regions
325   * @throws java.io.IOException Signals that an I/O exception has occurred.
326   */
327  protected List<RegionInfo> assignedRegions(int numRegions, ServerName sn) throws IOException {
328    List<RegionInfo> regions = new ArrayList<>(numRegions);
329    byte[] start = new byte[16];
330    byte[] end = new byte[16];
331    Bytes.putInt(start, 0, numRegions << 1);
332    Bytes.putInt(end, 0, (numRegions << 1) + 1);
333    for (int i = 0; i < numRegions; i++) {
334      TableName tableName = getTableName(sn);
335      regions.add(RegionInfoBuilder.newBuilder(tableName)
336          .setStartKey(start)
337          .setEndKey(end)
338          .setSplit(false)
339          .setRegionId(regionId++)
340          .build());
341    }
342    return regions;
343  }

355  /**
356   * Construct group info, with each group having at least one server.
357   * @param servers the servers
358   * @param groups the groups
359   * @return the map
360   */
361  protected static Map<String, RSGroupInfo> constructGroupInfo(List<ServerName> servers,
362      String[] groups) {
363    assertTrue(servers != null);
364    assertTrue(servers.size() >= groups.length);
365    int index = 0;
366    Map<String, RSGroupInfo> groupMap = new HashMap<>();
367    for (String grpName : groups) {
368      RSGroupInfo RSGroupInfo = new RSGroupInfo(grpName);
369      RSGroupInfo.addServer(servers.get(index).getAddress());
370      groupMap.put(grpName, RSGroupInfo);
371      index++;
372    }
373    while (index < servers.size()) {
374      int grpIndex = rand.nextInt(groups.length);
375      groupMap.get(groups[grpIndex]).addServer(servers.get(index).getAddress());
376      index++;
377    }
378    return groupMap;
379  }

381  /**
382   * Construct table descriptors evenly distributed between the groups.
383   * @param hasBogusTable there is a table that does not determine the group
384   * @return the list of table descriptors
385   */
386  protected static Map<TableName, TableDescriptor> constructTableDesc(boolean hasBogusTable) {
387    Map<TableName, TableDescriptor> tds = new HashMap<>();
388    int index = rand.nextInt(groups.length);
389    for (int i = 0; i < tables.length; i++) {
390      int grpIndex = (i + index) % groups.length;
391      String groupName = groups[grpIndex];
392      TableDescriptor htd =
393          TableDescriptorBuilder.newBuilder(tables[i]).setRegionServerGroup(groupName).build();
394      tds.put(htd.getTableName(), htd);
395    }
396    if (hasBogusTable) {
397      tds.put(table0, TableDescriptorBuilder.newBuilder(table0).setRegionServerGroup("").build());
398    }
399    return tds;
400  }

430  protected TableName getTableName(ServerName sn) throws IOException {
431    TableName tableName = null;
432    RSGroupInfoManager gm = getMockedGroupInfoManager();
433    RSGroupInfo groupOfServer = null;
434    for (RSGroupInfo gInfo : gm.listRSGroups()) {
435      if (gInfo.containsServer(sn.getAddress())) {
436        groupOfServer = gInfo;
437        break;
438      }
439    }
440
441    for (TableDescriptor desc : tableDescs.values()) {
442      Optional<String> optGroupName = desc.getRegionServerGroup();
443      if (optGroupName.isPresent() && optGroupName.get().endsWith(groupOfServer.getName())) {
444        tableName = desc.getTableName();
445      }
446    }
447    return tableName;
448  }
