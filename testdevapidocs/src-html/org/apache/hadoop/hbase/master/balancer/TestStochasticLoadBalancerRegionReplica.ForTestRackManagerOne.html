064  @Test
065  public void testReplicaCostForReplicas() {
066    Configuration conf = HBaseConfiguration.create();
067    CostFunction costFunction =
068        new RegionReplicaHostCostFunction(conf);
069
070    int[] servers = new int[] { 3, 3, 3, 3, 3 };
071    TreeMap<ServerName, List<RegionInfo>> clusterState = mockClusterServers(servers);
072
073    BalancerClusterState cluster;
074
075    cluster = new BalancerClusterState(clusterState, null, null, null);
076    costFunction.prepare(cluster);
077    double costWithoutReplicas = costFunction.cost();
078    assertEquals(0, costWithoutReplicas, 0);
079
080    // replicate the region from first server to the last server
081    RegionInfo replica1 =
082        RegionReplicaUtil.getRegionInfoForReplica(clusterState.firstEntry().getValue().get(0), 1);
083    clusterState.lastEntry().getValue().add(replica1);
084
085    cluster = new BalancerClusterState(clusterState, null, null, null);
086    costFunction.prepare(cluster);
087    double costWith1ReplicaDifferentServer = costFunction.cost();
088
089    assertEquals(0, costWith1ReplicaDifferentServer, 0);
090
091    // add a third replica to the last server
092    RegionInfo replica2 = RegionReplicaUtil.getRegionInfoForReplica(replica1, 2);
093    clusterState.lastEntry().getValue().add(replica2);
094
095    cluster = new BalancerClusterState(clusterState, null, null, null);
096    costFunction.prepare(cluster);
097    double costWith1ReplicaSameServer = costFunction.cost();
098
099    assertTrue(costWith1ReplicaDifferentServer < costWith1ReplicaSameServer);
100
101    // test with replication = 4 for following:
102
103    RegionInfo replica3;
104    Iterator<Map.Entry<ServerName, List<RegionInfo>>> it;
105    Map.Entry<ServerName, List<RegionInfo>> entry;
106
107    clusterState = mockClusterServers(servers);
108    it = clusterState.entrySet().iterator();
109    entry = it.next(); // first server
110    RegionInfo hri = entry.getValue().get(0);
111    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
112    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
113    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
114    entry.getValue().add(replica1);
115    entry.getValue().add(replica2);
116    it.next().getValue().add(replica3); // 2nd server
117
118    cluster = new BalancerClusterState(clusterState, null, null, null);
119    costFunction.prepare(cluster);
120    double costWith3ReplicasSameServer = costFunction.cost();
121
122    clusterState = mockClusterServers(servers);
123    hri = clusterState.firstEntry().getValue().get(0);
124    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
125    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
126    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
127
128    clusterState.firstEntry().getValue().add(replica1);
129    clusterState.lastEntry().getValue().add(replica2);
130    clusterState.lastEntry().getValue().add(replica3);
131
132    cluster = new BalancerClusterState(clusterState, null, null, null);
133    costFunction.prepare(cluster);
134    double costWith2ReplicasOnTwoServers = costFunction.cost();
135
136    assertTrue(costWith2ReplicasOnTwoServers < costWith3ReplicasSameServer);
137  }

139  @Test
140  public void testNeedsBalanceForColocatedReplicas() {
141    // check for the case where there are two hosts and with one rack, and where
142    // both the replicas are hosted on the same server
143    List<RegionInfo> regions = randomRegions(1);
144    ServerName s1 = ServerName.valueOf("host1", 1000, 11111);
145    ServerName s2 = ServerName.valueOf("host11", 1000, 11111);
146    Map<ServerName, List<RegionInfo>> map = new HashMap<>();
147    map.put(s1, regions);
148    regions.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
149    // until the step above s1 holds two replicas of a region
150    regions = randomRegions(1);
151    map.put(s2, regions);
152    assertTrue(loadBalancer.needsBalance(HConstants.ENSEMBLE_TABLE_NAME,
153      new BalancerClusterState(map, null, null, null)));
154    // check for the case where there are two hosts on the same rack and there are two racks
155    // and both the replicas are on the same rack
156    map.clear();
157    regions = randomRegions(1);
158    List<RegionInfo> regionsOnS2 = new ArrayList<>(1);
159    regionsOnS2.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
160    map.put(s1, regions);
161    map.put(s2, regionsOnS2);
162    // add another server so that the cluster has some host on another rack
163    map.put(ServerName.valueOf("host2", 1000, 11111), randomRegions(1));
164    assertTrue(
165      loadBalancer.needsBalance(HConstants.ENSEMBLE_TABLE_NAME,
166        new BalancerClusterState(map, null, null, new ForTestRackManagerOne())));
167  }
