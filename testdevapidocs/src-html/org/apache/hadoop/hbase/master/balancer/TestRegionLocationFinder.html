044@Category({MasterTests.class, SmallTests.class})
045public class TestRegionLocationFinder {
046  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
047  private static MiniHBaseCluster cluster;
048
049  private final static TableName tableName = TableName.valueOf("table");
050  private final static byte[] FAMILY = Bytes.toBytes("cf");
051  private static Table table;
052  private final static int ServerNum = 5;
053
054  private static RegionLocationFinder finder = new RegionLocationFinder();
055
056  @BeforeClass
057  public static void setUpBeforeClass() throws Exception {
058    cluster = TEST_UTIL.startMiniCluster(1, ServerNum);
059    table = TEST_UTIL.createTable(tableName, FAMILY, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
060    TEST_UTIL.waitTableAvailable(tableName, 1000);
061    TEST_UTIL.loadTable(table, FAMILY);
062
063    for (int i = 0; i < ServerNum; i++) {
064      HRegionServer server = cluster.getRegionServer(i);
065      for (Region region : server.getRegions(tableName)) {
066        region.flush(true);
067      }
068    }
069
070    finder.setConf(TEST_UTIL.getConfiguration());
071    finder.setServices(cluster.getMaster());
072    finder.setClusterStatus(cluster.getMaster().getClusterStatus());
073  }
074
075  @AfterClass
076  public static void tearDownAfterClass() throws Exception {
077    table.close();
078    TEST_UTIL.deleteTable(tableName);
079    TEST_UTIL.shutdownMiniCluster();
080  }
081
082  @Test
083  public void testInternalGetTopBlockLocation() throws Exception {
084    for (int i = 0; i < ServerNum; i++) {
085      HRegionServer server = cluster.getRegionServer(i);
086      for (Region region : server.getRegions(tableName)) {
087        // get region's hdfs block distribution by region and RegionLocationFinder,
088        // they should have same result
089        HDFSBlocksDistribution blocksDistribution1 = region.getHDFSBlocksDistribution();
090        HDFSBlocksDistribution blocksDistribution2 = finder.getBlockDistribution(region
091            .getRegionInfo());
092        assertEquals(blocksDistribution1.getUniqueBlocksTotalWeight(),
093          blocksDistribution2.getUniqueBlocksTotalWeight());
094        if (blocksDistribution1.getUniqueBlocksTotalWeight() != 0) {
095          assertEquals(blocksDistribution1.getTopHosts().get(0), blocksDistribution2.getTopHosts()
096              .get(0));
097        }
098      }
099    }
100  }
101
102  @Test
103  public void testMapHostNameToServerName() throws Exception {
104    List<String> topHosts = new ArrayList<>();
105    for (int i = 0; i < ServerNum; i++) {
106      HRegionServer server = cluster.getRegionServer(i);
107      String serverHost = server.getServerName().getHostname();
108      if (!topHosts.contains(serverHost)) {
109        topHosts.add(serverHost);
110      }
111    }
112    List<ServerName> servers = finder.mapHostNameToServerName(topHosts);
113    // mini cluster, all rs in one host
114    assertEquals(1, topHosts.size());
115    for (int i = 0; i < ServerNum; i++) {
116      ServerName server = cluster.getRegionServer(i).getServerName();
117      assertTrue(servers.contains(server));
118    }
119  }
120
121  @Test
122  public void testGetTopBlockLocations() throws Exception {
123    for (int i = 0; i < ServerNum; i++) {
124      HRegionServer server = cluster.getRegionServer(i);
125      for (Region region : server.getRegions(tableName)) {
126        List<ServerName> servers = finder.getTopBlockLocations(region
127            .getRegionInfo());
128        // test table may have empty region
129        if (region.getHDFSBlocksDistribution().getUniqueBlocksTotalWeight() == 0) {
130          continue;
131        }
132        List<String> topHosts = region.getHDFSBlocksDistribution().getTopHosts();
133        // rs and datanode may have different host in local machine test
134        if (!topHosts.contains(server.getServerName().getHostname())) {
135          continue;
136        }
137        for (int j = 0; j < ServerNum; j++) {
138          ServerName serverName = cluster.getRegionServer(j).getServerName();
139          assertTrue(servers.contains(serverName));
140        }
141      }
142    }
143  }
144
145  @Test
146  public void testRefreshAndWait() throws Exception {
147    finder.getCache().invalidateAll();
148    for (int i = 0; i < ServerNum; i++) {
149      HRegionServer server = cluster.getRegionServer(i);
150      List<Region> regions = server.getRegions(tableName);
151      if (regions.size() <= 0) {
152        continue;
153      }
154      List<RegionInfo> regionInfos = new ArrayList<>(regions.size());
155      for (Region region : regions) {
156        regionInfos.add(region.getRegionInfo());
157      }
158      finder.refreshAndWait(regionInfos);
159      for (RegionInfo regionInfo : regionInfos) {
160        assertNotNull(finder.getCache().getIfPresent(regionInfo));
161      }
162    }
163  }
