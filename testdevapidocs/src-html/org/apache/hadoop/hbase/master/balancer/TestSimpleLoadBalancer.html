126  /**
127   * Test the load balancing algorithm.
128   *
129   * Invariant is that all servers should be hosting either floor(average) or
130   * ceiling(average) at both table level and cluster level
131   */
132  @Test
133  public void testBalanceClusterOverall() throws Exception {
134    Map<TableName, Map<ServerName, List<RegionInfo>>> clusterLoad = new TreeMap<>();
135    for (int[] mockCluster : clusterStateMocks) {
136      Map<ServerName, List<RegionInfo>> clusterServers = mockClusterServers(mockCluster, 30);
137      List<ServerAndLoad> clusterList = convertToList(clusterServers);
138      clusterLoad.put(TableName.valueOf(name.getMethodName()), clusterServers);
139      HashMap<TableName, TreeMap<ServerName, List<RegionInfo>>> result =
140          mockClusterServersWithTables(clusterServers);
141      loadBalancer.setClusterLoad(clusterLoad);
142      List<RegionPlan> clusterplans = new ArrayList<>();
143      List<Pair<TableName, Integer>> regionAmountList = new ArrayList<>();
144      for (Map.Entry<TableName, TreeMap<ServerName, List<RegionInfo>>> mapEntry : result
145          .entrySet()) {
146        TableName tableName = mapEntry.getKey();
147        TreeMap<ServerName, List<RegionInfo>> servers = mapEntry.getValue();
148        List<ServerAndLoad> list = convertToList(servers);
149        LOG.info("Mock Cluster : " + printMock(list) + " " + printStats(list));
150        List<RegionPlan> partialplans = loadBalancer.balanceTable(tableName, servers);
151        if(partialplans != null) clusterplans.addAll(partialplans);
152        List<ServerAndLoad> balancedClusterPerTable = reconcile(list, partialplans, servers);
153        LOG.info("Mock Balance : " + printMock(balancedClusterPerTable));
154        assertClusterAsBalanced(balancedClusterPerTable);
155        for (Map.Entry<ServerName, List<RegionInfo>> entry : servers.entrySet()) {
156          returnRegions(entry.getValue());
157          returnServer(entry.getKey());
158        }
159      }
160      List<ServerAndLoad> balancedCluster = reconcile(clusterList, clusterplans, clusterServers);
161      assertTrue(assertClusterOverallAsBalanced(balancedCluster, result.keySet().size()));
162    }
163  }

184  private void testImpactOfBalanceClusterOverall(boolean useLoadOfAllTable) throws Exception {
185    Map<TableName, Map<ServerName, List<RegionInfo>>> clusterLoad = new TreeMap<>();
186    Map<ServerName, List<RegionInfo>> clusterServers =
187        mockUniformClusterServers(mockUniformCluster);
188    List<ServerAndLoad> clusterList = convertToList(clusterServers);
189    clusterLoad.put(TableName.valueOf(name.getMethodName()), clusterServers);
190    // use overall can achieve both table and cluster level balance
191    HashMap<TableName, TreeMap<ServerName, List<RegionInfo>>> LoadOfAllTable =
192        mockClusterServersWithTables(clusterServers);
193    if (useLoadOfAllTable) {
194      loadBalancer.setClusterLoad((Map) LoadOfAllTable);
195    } else {
196      loadBalancer.setClusterLoad(clusterLoad);
197    }
198    List<RegionPlan> clusterplans1 = new ArrayList<RegionPlan>();
199    List<Pair<TableName, Integer>> regionAmountList = new ArrayList<Pair<TableName, Integer>>();
200    for (Map.Entry<TableName, TreeMap<ServerName, List<RegionInfo>>> mapEntry : LoadOfAllTable
201        .entrySet()) {
202      TableName tableName = mapEntry.getKey();
203      TreeMap<ServerName, List<RegionInfo>> servers = mapEntry.getValue();
204      List<ServerAndLoad> list = convertToList(servers);
205      LOG.info("Mock Cluster : " + printMock(list) + " " + printStats(list));
206      List<RegionPlan> partialplans = loadBalancer.balanceTable(tableName, servers);
207      if (partialplans != null) clusterplans1.addAll(partialplans);
208      List<ServerAndLoad> balancedClusterPerTable = reconcile(list, partialplans, servers);
209      LOG.info("Mock Balance : " + printMock(balancedClusterPerTable));
210      assertClusterAsBalanced(balancedClusterPerTable);
211      for (Map.Entry<ServerName, List<RegionInfo>> entry : servers.entrySet()) {
212        returnRegions(entry.getValue());
213        returnServer(entry.getKey());
214      }
215    }
216    List<ServerAndLoad> balancedCluster1 = reconcile(clusterList, clusterplans1, clusterServers);
217    assertTrue(assertClusterOverallAsBalanced(balancedCluster1, LoadOfAllTable.keySet().size()));
218  }

220  @Test
221  public void testBalanceClusterOverallStrictly() throws Exception {
222    int[] regionNumOfTable1PerServer = { 3, 3, 4, 4, 4, 4, 5, 5, 5 };
223    int[] regionNumOfTable2PerServer = { 2, 2, 2, 2, 2, 2, 2, 2, 1 };
224    TreeMap<ServerName, List<RegionInfo>> serverRegionInfo = new TreeMap<>();
225    List<ServerAndLoad> serverAndLoads = new ArrayList<>();
226    for (int i = 0; i < regionNumOfTable1PerServer.length; i++) {
227      ServerName serverName = ServerName.valueOf("server" + i, 1000, -1);
228      List<RegionInfo> regions1 =
229          createRegions(regionNumOfTable1PerServer[i], TableName.valueOf("table1"));
230      List<RegionInfo> regions2 =
231          createRegions(regionNumOfTable2PerServer[i], TableName.valueOf("table2"));
232      regions1.addAll(regions2);
233      serverRegionInfo.put(serverName, regions1);
234      ServerAndLoad serverAndLoad = new ServerAndLoad(serverName,
235          regionNumOfTable1PerServer[i] + regionNumOfTable2PerServer[i]);
236      serverAndLoads.add(serverAndLoad);
237    }
238    HashMap<TableName, TreeMap<ServerName, List<RegionInfo>>> LoadOfAllTable =
239        mockClusterServersWithTables(serverRegionInfo);
240    loadBalancer.setClusterLoad((Map) LoadOfAllTable);
241    List<RegionPlan> partialplans = loadBalancer.balanceTable(TableName.valueOf("table1"),
242      LoadOfAllTable.get(TableName.valueOf("table1")));
243    List<ServerAndLoad> balancedServerLoads =
244        reconcile(serverAndLoads, partialplans, serverRegionInfo);
245    for (ServerAndLoad serverAndLoad : balancedServerLoads) {
246      assertEquals(6, serverAndLoad.getLoad());
247    }
248  }
