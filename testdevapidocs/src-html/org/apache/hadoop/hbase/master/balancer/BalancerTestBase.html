177  /**
178   * Invariant is that all servers have between floor(avg) and ceiling(avg)
179   * number of regions.
180   */
181  public void assertClusterAsBalanced(List<ServerAndLoad> servers) {
182    int numServers = servers.size();
183    int numRegions = 0;
184    int maxRegions = 0;
185    int minRegions = Integer.MAX_VALUE;
186    for (ServerAndLoad server : servers) {
187      int nr = server.getLoad();
188      if (nr > maxRegions) {
189        maxRegions = nr;
190      }
191      if (nr < minRegions) {
192        minRegions = nr;
193      }
194      numRegions += nr;
195    }
196    if (maxRegions - minRegions < 2) {
197      // less than 2 between max and min, can't balance
198      return;
199    }
200    int min = numRegions / numServers;
201    int max = numRegions % numServers == 0 ? min : min + 1;
202
203    for (ServerAndLoad server : servers) {
204      assertTrue(server.getLoad() >= 0);
205      assertTrue(server.getLoad() <= max);
206      assertTrue(server.getLoad() >= min);
207    }
208  }

210  /**
211   * Invariant is that all servers have between acceptable range
212   * number of regions.
213   */
214  public boolean assertClusterOverallAsBalanced(List<ServerAndLoad> servers, int tablenum) {
215    int numServers = servers.size();
216    int numRegions = 0;
217    int maxRegions = 0;
218    int minRegions = Integer.MAX_VALUE;
219    for (ServerAndLoad server : servers) {
220      int nr = server.getLoad();
221      if (nr > maxRegions) {
222        maxRegions = nr;
223      }
224      if (nr < minRegions) {
225        minRegions = nr;
226      }
227      numRegions += nr;
228    }
229    if (maxRegions - minRegions < 2) {
230      // less than 2 between max and min, can't balance
231      return true;
232    }
233    int min = numRegions / numServers;
234    int max = numRegions % numServers == 0 ? min : min + 1;
235
236    for (ServerAndLoad server : servers) {
237      if (server.getLoad() < 0 || server.getLoad() > max + tablenum/2 + 1  ||
238          server.getLoad() < min - tablenum/2 - 1) {
239        LOG.warn("server={}, load={}, max={}, tablenum={}, min={}",
240          server.getServerName(), server.getLoad(), max, tablenum, min);
241        return false;
242      }
243    }
244    return true;
245  }

247  /**
248   * Checks whether region replicas are not hosted on the same host.
249   */
250  public void assertRegionReplicaPlacement(Map<ServerName, List<RegionInfo>> serverMap, RackManager rackManager) {
251    TreeMap<String, Set<RegionInfo>> regionsPerHost = new TreeMap<>();
252    TreeMap<String, Set<RegionInfo>> regionsPerRack = new TreeMap<>();
253
254    for (Entry<ServerName, List<RegionInfo>> entry : serverMap.entrySet()) {
255      String hostname = entry.getKey().getHostname();
256      Set<RegionInfo> infos = regionsPerHost.get(hostname);
257      if (infos == null) {
258        infos = new HashSet<>();
259        regionsPerHost.put(hostname, infos);
260      }
261
262      for (RegionInfo info : entry.getValue()) {
263        RegionInfo primaryInfo = RegionReplicaUtil.getRegionInfoForDefaultReplica(info);
264        if (!infos.add(primaryInfo)) {
265          Assert.fail("Two or more region replicas are hosted on the same host after balance");
266        }
267      }
268    }
269
270    if (rackManager == null) {
271      return;
272    }
273
274    for (Entry<ServerName, List<RegionInfo>> entry : serverMap.entrySet()) {
275      String rack = rackManager.getRack(entry.getKey());
276      Set<RegionInfo> infos = regionsPerRack.get(rack);
277      if (infos == null) {
278        infos = new HashSet<>();
279        regionsPerRack.put(rack, infos);
280      }
281
282      for (RegionInfo info : entry.getValue()) {
283        RegionInfo primaryInfo = RegionReplicaUtil.getRegionInfoForDefaultReplica(info);
284        if (!infos.add(primaryInfo)) {
285          Assert.fail("Two or more region replicas are hosted on the same rack after balance");
286        }
287      }
288    }
289  }

297    float average = (float) totalRegions / numServers;
298    int max = (int) Math.ceil(average);
299    int min = (int) Math.floor(average);
300    return "[srvr=" + numServers + " rgns=" + totalRegions + " avg=" + average + " max=" + max

329  /**
330   * This assumes the RegionPlan HSI instances are the same ones in the map, so
331   * actually no need to even pass in the map, but I think it's clearer.
332   *
333   * @param list
334   * @param plans
335   * @return a list of all added {@link ServerAndLoad} values.
336   */
337  protected List<ServerAndLoad> reconcile(List<ServerAndLoad> list,
338                                          List<RegionPlan> plans,
339                                          Map<ServerName, List<RegionInfo>> servers) {
340    List<ServerAndLoad> result = new ArrayList<>(list.size());
341
342    Map<ServerName, ServerAndLoad> map = new HashMap<>(list.size());
343    for (ServerAndLoad sl : list) {
344      map.put(sl.getServerName(), sl);
345    }
346    if (plans != null) {
347      for (RegionPlan plan : plans) {
348        ServerName source = plan.getSource();
349
350        updateLoad(map, source, -1);
351        ServerName destination = plan.getDestination();
352        updateLoad(map, destination, +1);
353
354        servers.get(source).remove(plan.getRegionInfo());
355        servers.get(destination).add(plan.getRegionInfo());
356      }
357    }
358    result.clear();
359    result.addAll(map.values());
360    return result;
361  }

363  protected void updateLoad(final Map<ServerName, ServerAndLoad> map,
364                            final ServerName sn,
365                            final int diff) {
366    ServerAndLoad sal = map.get(sn);
367    if (sal == null) sal = new ServerAndLoad(sn, 0);
368    sal = new ServerAndLoad(sn, sal.getLoad() + diff);
369    map.put(sn, sal);
370  }

438  protected List<RegionInfo> createRegions(int numRegions, TableName tableName) {
439    List<RegionInfo> regions = new ArrayList<>(numRegions);
440    byte[] start = new byte[16];
441    byte[] end = new byte[16];
442    Random rand = ThreadLocalRandom.current();
443    rand.nextBytes(start);
444    rand.nextBytes(end);
445    for (int i = 0; i < numRegions; i++) {
446      Bytes.putInt(start, 0, numRegions << 1);
447      Bytes.putInt(end, 0, (numRegions << 1) + 1);
448      RegionInfo hri = RegionInfoBuilder.newBuilder(tableName)
449        .setStartKey(start)
450        .setEndKey(end)
451        .setSplit(false)
452        .build();
453      regions.add(hri);
454    }
455    return regions;
456  }

458  protected List<RegionInfo> randomRegions(int numRegions, int numTables) {
459    List<RegionInfo> regions = new ArrayList<>(numRegions);
460    byte[] start = new byte[16];
461    byte[] end = new byte[16];
462    Random rand = ThreadLocalRandom.current();
463    rand.nextBytes(start);
464    rand.nextBytes(end);
465    for (int i = 0; i < numRegions; i++) {
466      if (!regionQueue.isEmpty()) {
467        regions.add(regionQueue.poll());
468        continue;
469      }
470      Bytes.putInt(start, 0, numRegions << 1);
471      Bytes.putInt(end, 0, (numRegions << 1) + 1);
472      TableName tableName =
473          TableName.valueOf("table" + (numTables > 0 ? rand.nextInt(numTables) : i));
474      RegionInfo hri = RegionInfoBuilder.newBuilder(tableName)
475          .setStartKey(start)
476          .setEndKey(end)
477          .setSplit(false)
478          .setRegionId(regionId++)
479          .build();
480      regions.add(hri);
481    }
482    return regions;
483  }

553  protected void testWithCluster(Map<ServerName, List<RegionInfo>> serverMap,
554      RackManager rackManager, boolean assertFullyBalanced, boolean assertFullyBalancedForReplicas) {
555    List<ServerAndLoad> list = convertToList(serverMap);
556    LOG.info("Mock Cluster : " + printMock(list) + " " + printStats(list));
557
558    loadBalancer.setRackManager(rackManager);
559    // Run the balancer.
560    Map<TableName, Map<ServerName, List<RegionInfo>>> LoadOfAllTable =
561        (Map) mockClusterServersWithTables(serverMap);
562    List<RegionPlan> plans = loadBalancer.balanceCluster(LoadOfAllTable);
563    assertNotNull(plans);
564
565    // Check to see that this actually got to a stable place.
566    if (assertFullyBalanced || assertFullyBalancedForReplicas) {
567      // Apply the plan to the mock cluster.
568      List<ServerAndLoad> balancedCluster = reconcile(list, plans, serverMap);
569
570      // Print out the cluster loads to make debugging easier.
571      LOG.info("Mock Balance : " + printMock(balancedCluster));
572
573      if (assertFullyBalanced) {
574        assertClusterAsBalanced(balancedCluster);
575        LoadOfAllTable = (Map) mockClusterServersWithTables(serverMap);
576        List<RegionPlan> secondPlans = loadBalancer.balanceCluster(LoadOfAllTable);
577        assertNull(secondPlans);
578      }
579
580      if (assertFullyBalancedForReplicas) {
581        assertRegionReplicaPlacement(serverMap, rackManager);
582      }
583    }
584  }

586  protected Map<ServerName, List<RegionInfo>> createServerMap(int numNodes,
587                                                             int numRegions,
588                                                             int numRegionsPerServer,
589                                                             int replication,
590                                                             int numTables) {
591    //construct a cluster of numNodes, having  a total of numRegions. Each RS will hold
592    //numRegionsPerServer many regions except for the last one, which will host all the
593    //remaining regions
594    int[] cluster = new int[numNodes];
595    for (int i =0; i < numNodes; i++) {
596      cluster[i] = numRegionsPerServer;
597    }
598    cluster[cluster.length - 1] = numRegions - ((cluster.length - 1) * numRegionsPerServer);
599    Map<ServerName, List<RegionInfo>> clusterState = mockClusterServers(cluster, numTables);
600    if (replication > 0) {
601      // replicate the regions to the same servers
602      for (List<RegionInfo> regions : clusterState.values()) {
603        int length = regions.size();
604        for (int i = 0; i < length; i++) {
605          for (int r = 1; r < replication ; r++) {
606            regions.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(i), r));
607          }
608        }
609      }
610    }
611
612    return clusterState;
613  }
