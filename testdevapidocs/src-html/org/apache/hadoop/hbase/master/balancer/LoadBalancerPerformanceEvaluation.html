019package org.apache.hadoop.hbase.master.balancer;
020
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027import java.util.concurrent.TimeUnit;
028import org.apache.hadoop.hbase.HBaseCommonTestingUtil;
029import org.apache.hadoop.hbase.HBaseInterfaceAudience;
030import org.apache.hadoop.hbase.HConstants;
031import org.apache.hadoop.hbase.ServerName;
032import org.apache.hadoop.hbase.TableName;
033import org.apache.hadoop.hbase.client.RegionInfo;
034import org.apache.hadoop.hbase.client.RegionInfoBuilder;
035import org.apache.hadoop.hbase.master.LoadBalancer;
036import org.apache.hadoop.hbase.util.AbstractHBaseTool;
037import org.apache.hadoop.hbase.util.Bytes;
038import org.apache.yetus.audience.InterfaceAudience;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041import org.apache.hbase.thirdparty.com.google.common.base.Preconditions;
042import org.apache.hbase.thirdparty.com.google.common.base.Stopwatch;
043import org.apache.hbase.thirdparty.org.apache.commons.cli.CommandLine;
044import org.apache.hbase.thirdparty.org.apache.commons.cli.Option;
045
046/**
047 * Tool to test performance of different {@link org.apache.hadoop.hbase.master.LoadBalancer}
048 * implementations.
049 * Example command:
050 * $ bin/hbase org.apache.hadoop.hbase.master.balancer.LoadBalancerPerformanceEvaluation
051 *   -regions 1000 -servers 100
052 *   -load_balancer org.apache.hadoop.hbase.master.balancer.SimpleLoadBalancer
053 */
054@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
055public class LoadBalancerPerformanceEvaluation extends AbstractHBaseTool {
056  private static final Logger LOG =
057      LoggerFactory.getLogger(LoadBalancerPerformanceEvaluation.class.getName());
058
059  protected static final HBaseCommonTestingUtil UTIL = new HBaseCommonTestingUtil();
060
061  private static final int DEFAULT_NUM_REGIONS = 1000000;
062  private static Option NUM_REGIONS_OPT = new Option("regions", true,
063      "Number of regions to consider by load balancer. Default: " + DEFAULT_NUM_REGIONS);
064
065  private static final int DEFAULT_NUM_SERVERS = 1000;
066  private static Option NUM_SERVERS_OPT = new Option("servers", true,
067      "Number of servers to consider by load balancer. Default: " + DEFAULT_NUM_SERVERS);
068
069  private static final String DEFAULT_LOAD_BALANCER =
070      "org.apache.hadoop.hbase.master.balancer.StochasticLoadBalancer";
071  private static Option LOAD_BALANCER_OPT = new Option("load_balancer", true,
072      "Type of Load Balancer to use. Default: " + DEFAULT_LOAD_BALANCER);
073
074  private int numRegions;
075  private int numServers;
076  private String loadBalancerType;
077  private Class<?> loadBalancerClazz;
078
079  private LoadBalancer loadBalancer;
080
081  // data
082  private List<ServerName> servers;
083  private List<RegionInfo> regions;
084  private Map<RegionInfo, ServerName> regionServerMap;
085  private Map<TableName, Map<ServerName, List<RegionInfo>>> tableServerRegionMap;
086
087  // Non-default configurations.
088  private void setupConf() {
089    conf.setClass(HConstants.HBASE_MASTER_LOADBALANCER_CLASS, loadBalancerClazz,
090      LoadBalancer.class);
091    loadBalancer = LoadBalancerFactory.getLoadBalancer(conf);
092  }
093
094  private void generateRegionsAndServers() {
095    TableName tableName = TableName.valueOf("LoadBalancerPerfTable");
096    // regions
097    regions = new ArrayList<>(numRegions);
098    regionServerMap = new HashMap<>(numRegions);
099    for (int i = 0; i < numRegions; ++i) {
100      byte[] start = new byte[16];
101      byte[] end = new byte[16];
102
103      Bytes.putInt(start, 0, i);
104      Bytes.putInt(end, 0, i + 1);
105      RegionInfo hri = RegionInfoBuilder.newBuilder(tableName)
106        .setStartKey(start)
107        .setEndKey(end)
108        .setSplit(false)
109        .setRegionId(i)
110        .build();
111      regions.add(hri);
112      regionServerMap.put(hri, null);
113    }
114
115    // servers
116    servers = new ArrayList<>(numServers);
117    Map<ServerName, List<RegionInfo>> serverRegionMap = new HashMap<>(numServers);
118    for (int i = 0; i < numServers; ++i) {
119      ServerName sn = ServerName.valueOf("srv" + i, HConstants.DEFAULT_REGIONSERVER_PORT, i);
120      servers.add(sn);
121      serverRegionMap.put(sn, i == 0 ? regions : Collections.emptyList());
122    }
123    tableServerRegionMap = Collections.singletonMap(tableName, serverRegionMap);
124  }
125
126  @Override
127  protected void addOptions() {
128    addOption(NUM_REGIONS_OPT);
129    addOption(NUM_SERVERS_OPT);
130    addOption(LOAD_BALANCER_OPT);
131  }
132
133  @Override
134  protected void processOptions(CommandLine cmd) {
135    numRegions = getOptionAsInt(cmd, NUM_REGIONS_OPT.getOpt(), DEFAULT_NUM_REGIONS);
136    Preconditions.checkArgument(numRegions > 0, "Invalid number of regions!");
137
138    numServers = getOptionAsInt(cmd, NUM_SERVERS_OPT.getOpt(), DEFAULT_NUM_SERVERS);
139    Preconditions.checkArgument(numServers > 0, "Invalid number of servers!");
140
141    loadBalancerType = cmd.getOptionValue(LOAD_BALANCER_OPT.getOpt(), DEFAULT_LOAD_BALANCER);
142    Preconditions.checkArgument(!loadBalancerType.isEmpty(), "Invalid load balancer type!");
143
144    try {
145      loadBalancerClazz = Class.forName(loadBalancerType);
146    } catch (ClassNotFoundException e) {
147      System.err.println("Class '" + loadBalancerType + "' not found!");
148      System.exit(1);
149    }
150
151    setupConf();
152  }
153
154  private String formatResults(final String methodName, final long timeMillis) {
155    return String.format("Time for %-25s: %dms%n", methodName, timeMillis);
156  }
157
158  @Override
159  protected int doWork() throws Exception {
160    generateRegionsAndServers();
161
162    String methodName = "roundRobinAssignment";
163    LOG.info("Calling " + methodName);
164    Stopwatch watch = Stopwatch.createStarted();
165    loadBalancer.roundRobinAssignment(regions, servers);
166    System.out.print(formatResults(methodName, watch.elapsed(TimeUnit.MILLISECONDS)));
167
168    methodName = "retainAssignment";
169    LOG.info("Calling " + methodName);
170    watch.reset().start();
171    loadBalancer.retainAssignment(regionServerMap, servers);
172    System.out.print(formatResults(methodName, watch.elapsed(TimeUnit.MILLISECONDS)));
173
174    methodName = "balanceCluster";
175    LOG.info("Calling " + methodName);
176    watch.reset().start();
177
178    loadBalancer.balanceCluster(tableServerRegionMap);
179    System.out.print(formatResults(methodName, watch.elapsed(TimeUnit.MILLISECONDS)));
180
181    return EXIT_SUCCESS;
182  }
183
184  public static void main(String[] args) throws IOException {
185    LoadBalancerPerformanceEvaluation tool = new LoadBalancerPerformanceEvaluation();
186    tool.setConf(UTIL.getConfiguration());
187    tool.run(args);
188  }
189}
