018package org.apache.hadoop.hbase.master.balancer;
019
020import java.util.ArrayList;
021import java.util.List;
022import java.util.Map;
023import java.util.TreeMap;
024import org.apache.hadoop.hbase.HBaseClassTestRule;
025import org.apache.hadoop.hbase.ServerName;
026import org.apache.hadoop.hbase.client.RegionInfo;
027import org.apache.hadoop.hbase.testclassification.LargeTests;
028import org.apache.hadoop.hbase.testclassification.MasterTests;
029import org.junit.ClassRule;
030import org.junit.Test;
031import org.junit.experimental.categories.Category;
032
033@Category({ MasterTests.class, LargeTests.class })
034public class TestStochasticLoadBalancerRegionReplicaSameHosts extends BalancerTestBase {
035
036  @ClassRule
037  public static final HBaseClassTestRule CLASS_RULE =
038      HBaseClassTestRule.forClass(TestStochasticLoadBalancerRegionReplicaSameHosts.class);
039
040  @Test // Test is flakey. TODO: Fix!
041  public void testRegionReplicationOnMidClusterSameHosts() {
042    conf.setLong(StochasticLoadBalancer.MAX_STEPS_KEY, 2000000L);
043    conf.setLong("hbase.master.balancer.stochastic.maxRunningTime", 90 * 1000); // 90 sec
044    conf.setFloat("hbase.master.balancer.stochastic.maxMovePercent", 1.0f);
045    loadBalancer.setConf(conf);
046    int numHosts = 100;
047    int numRegions = 100 * 100;
048    int replication = 3; // 3 replicas per region
049    int numRegionsPerServer = 5;
050    int numTables = 10;
051    Map<ServerName, List<RegionInfo>> serverMap =
052        createServerMap(numHosts, numRegions, numRegionsPerServer, replication, numTables);
053    int numNodesPerHost = 4;
054
055    // create a new map with 4 RS per host.
056    Map<ServerName, List<RegionInfo>> newServerMap = new TreeMap<>(serverMap);
057    for (Map.Entry<ServerName, List<RegionInfo>> entry : serverMap.entrySet()) {
058      for (int i = 1; i < numNodesPerHost; i++) {
059        ServerName s1 = entry.getKey();
060        // create an RS for the same host
061        ServerName s2 = ServerName.valueOf(s1.getHostname(), s1.getPort() + i, 1);
062        newServerMap.put(s2, new ArrayList<>());
063      }
064    }
065
066    testWithCluster(newServerMap, null, true, true);
067  }
068}
