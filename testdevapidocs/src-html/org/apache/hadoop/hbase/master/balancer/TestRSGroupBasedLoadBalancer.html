128  /**
129   * Invariant is that all servers of a group have load between floor(avg) and
130   * ceiling(avg) number of regions.
131   */
132  private void assertClusterAsBalanced(
133      ArrayListMultimap<String, ServerAndLoad> groupLoadMap) {
134    for (String gName : groupLoadMap.keySet()) {
135      List<ServerAndLoad> groupLoad = groupLoadMap.get(gName);
136      int numServers = groupLoad.size();
137      int numRegions = 0;
138      int maxRegions = 0;
139      int minRegions = Integer.MAX_VALUE;
140      for (ServerAndLoad server : groupLoad) {
141        int nr = server.getLoad();
142        if (nr > maxRegions) {
143          maxRegions = nr;
144        }
145        if (nr < minRegions) {
146          minRegions = nr;
147        }
148        numRegions += nr;
149      }
150      if (maxRegions - minRegions < 2) {
151        // less than 2 between max and min, can't balance
152        return;
153      }
154      int min = numRegions / numServers;
155      int max = numRegions % numServers == 0 ? min : min + 1;
156
157      for (ServerAndLoad server : groupLoad) {
158        assertTrue(server.getLoad() <= max);
159        assertTrue(server.getLoad() >= min);
160      }
161    }
162  }

252   * Test BOGUS_SERVER_NAME among groups do not overwrite each other
253   * @throws Exception
254   */
255  @Test
256  public void testRoundRobinAssignment() throws Exception {
257    List<ServerName> onlineServers = new ArrayList<ServerName>(servers.size());
258    onlineServers.addAll(servers);
259    List<RegionInfo> regions = randomRegions(25);
260    int bogusRegion = 0;
261    for(RegionInfo region : regions){
262      String group = tableMap.get(region.getTable());
263      if("dg3".equals(group) || "dg4".equals(group)){
264        bogusRegion++;
265      }
266    }
267    Set<Address> offlineServers = new HashSet<Address>();
268    offlineServers.addAll(groupMap.get("dg3").getServers());
269    offlineServers.addAll(groupMap.get("dg4").getServers());
270    for(Iterator<ServerName> it =  onlineServers.iterator(); it.hasNext();){
271      ServerName server = it.next();
272      Address address = server.getAddress();
273      if(offlineServers.contains(address)){
274        it.remove();
275      }
276    }
277    Map<ServerName, List<RegionInfo>> assignments = loadBalancer
278        .roundRobinAssignment(regions, onlineServers);
279    assertEquals(bogusRegion, assignments.get(LoadBalancer.BOGUS_SERVER_NAME).size());
280  }

282  /**
283   * Asserts a valid retained assignment plan.
284   * <p>
285   * Must meet the following conditions:
286   * <ul>
287   * <li>Every input region has an assignment, and to an online server
288   * <li>If a region had an existing assignment to a server with the same
289   * address a a currently online server, it will be assigned to it
290   * </ul>
291   *
292   * @param existing
293   * @param assignment
294   * @throws java.io.IOException
295   * @throws java.io.FileNotFoundException
296   */
297  private void assertRetainedAssignment(
298      Map<RegionInfo, ServerName> existing, List<ServerName> servers,
299      Map<ServerName, List<RegionInfo>> assignment)
300      throws FileNotFoundException, IOException {
301    // Verify condition 1, every region assigned, and to online server
302    Set<ServerName> onlineServerSet = new TreeSet<>(servers);
303    Set<RegionInfo> assignedRegions = new TreeSet<>(RegionInfo.COMPARATOR);
304    for (Map.Entry<ServerName, List<RegionInfo>> a : assignment.entrySet()) {
305      assertTrue(
306          "Region assigned to server that was not listed as online",
307          onlineServerSet.contains(a.getKey()));
308      for (RegionInfo r : a.getValue())
309        assignedRegions.add(r);
310    }
311    assertEquals(existing.size(), assignedRegions.size());
312
313    // Verify condition 2, every region must be assigned to correct server.
314    Set<String> onlineHostNames = new TreeSet<>();
315    for (ServerName s : servers) {
316      onlineHostNames.add(s.getHostname());
317    }
318
319    for (Map.Entry<ServerName, List<RegionInfo>> a : assignment.entrySet()) {
320      ServerName currentServer = a.getKey();
321      for (RegionInfo r : a.getValue()) {
322        ServerName oldAssignedServer = existing.get(r);
323        TableName tableName = r.getTable();
324        String groupName =
325            getMockedGroupInfoManager().getRSGroupOfTable(tableName);
326        assertTrue(StringUtils.isNotEmpty(groupName));
327        RSGroupInfo gInfo = getMockedGroupInfoManager().getRSGroup(
328            groupName);
329        assertTrue(
330            "Region is not correctly assigned to group servers.",
331            gInfo.containsServer(currentServer.getAddress()));
332        if (oldAssignedServer != null
333            && onlineHostNames.contains(oldAssignedServer
334            .getHostname())) {
335          // this region was previously assigned somewhere, and that
336          // host is still around, then the host must have been is a
337          // different group.
338          if (!oldAssignedServer.getAddress().equals(currentServer.getAddress())) {
339            assertFalse(gInfo.containsServer(oldAssignedServer.getAddress()));
340          }
341        }
342      }
343    }
344  }

346  private String printStats(
347      ArrayListMultimap<String, ServerAndLoad> groupBasedLoad) {
348    StringBuffer sb = new StringBuffer();
349    sb.append("\n");
350    for (String groupName : groupBasedLoad.keySet()) {
351      sb.append("Stats for group: " + groupName);
352      sb.append("\n");
353      sb.append(groupMap.get(groupName).getServers());
354      sb.append("\n");
355      List<ServerAndLoad> groupLoad = groupBasedLoad.get(groupName);
356      int numServers = groupLoad.size();
357      int totalRegions = 0;
358      sb.append("Per Server Load: \n");
359      for (ServerAndLoad sLoad : groupLoad) {
360        sb.append("Server :" + sLoad.getServerName() + " Load : "
361            + sLoad.getLoad() + "\n");
362        totalRegions += sLoad.getLoad();
363      }
364      sb.append(" Group Statistics : \n");
365      float average = (float) totalRegions / numServers;
366      int max = (int) Math.ceil(average);
367      int min = (int) Math.floor(average);
368      sb.append("[srvr=" + numServers + " rgns=" + totalRegions + " avg="
369          + average + " max=" + max + " min=" + min + "]");
370      sb.append("\n");
371      sb.append("===============================");
372      sb.append("\n");
373    }
374    return sb.toString();
375  }

377  private ArrayListMultimap<String, ServerAndLoad> convertToGroupBasedMap(
378      final Map<ServerName, List<RegionInfo>> serversMap) throws IOException {
379    ArrayListMultimap<String, ServerAndLoad> loadMap = ArrayListMultimap
380        .create();
381    for (RSGroupInfo gInfo : getMockedGroupInfoManager().listRSGroups()) {
382      Set<Address> groupServers = gInfo.getServers();
383      for (Address hostPort : groupServers) {
384        ServerName actual = null;
385        for(ServerName entry: servers) {
386          if(entry.getAddress().equals(hostPort)) {
387            actual = entry;
388            break;
389          }
390        }
391        List<RegionInfo> regions = serversMap.get(actual);
392        assertTrue("No load for " + actual, regions != null);
393        loadMap.put(gInfo.getName(),
394            new ServerAndLoad(actual, regions.size()));
395      }
396    }
397    return loadMap;
398  }

400  private ArrayListMultimap<String, ServerAndLoad> reconcile(
401      ArrayListMultimap<String, ServerAndLoad> previousLoad,
402      List<RegionPlan> plans) {
403    ArrayListMultimap<String, ServerAndLoad> result = ArrayListMultimap
404        .create();
405    result.putAll(previousLoad);
406    if (plans != null) {
407      for (RegionPlan plan : plans) {
408        ServerName source = plan.getSource();
409        updateLoad(result, source, -1);
410        ServerName destination = plan.getDestination();
411        updateLoad(result, destination, +1);
412      }
413    }
414    return result;
415  }
416
417  private void updateLoad(
418      ArrayListMultimap<String, ServerAndLoad> previousLoad,
419      final ServerName sn, final int diff) {
420    for (String groupName : previousLoad.keySet()) {
421      ServerAndLoad newSAL = null;
422      ServerAndLoad oldSAL = null;
423      for (ServerAndLoad sal : previousLoad.get(groupName)) {
424        if (ServerName.isSameAddress(sn, sal.getServerName())) {
425          oldSAL = sal;
426          newSAL = new ServerAndLoad(sn, sal.getLoad() + diff);
427          break;
428        }
429      }
430      if (newSAL != null) {
431        previousLoad.remove(groupName, oldSAL);
432        previousLoad.put(groupName, newSAL);
433        break;
434      }
435    }
436  }

438  private Map<ServerName, List<RegionInfo>> mockClusterServers() throws IOException {
439    assertTrue(servers.size() == regionAssignment.length);
440    Map<ServerName, List<RegionInfo>> assignment = new TreeMap<>();
441    for (int i = 0; i < servers.size(); i++) {
442      int numRegions = regionAssignment[i];
443      List<RegionInfo> regions = assignedRegions(numRegions, servers.get(i));
444      assignment.put(servers.get(i), regions);
445    }
446    return assignment;

449  /**
450   * Generate a list of regions evenly distributed between the tables.
451   *
452   * @param numRegions The number of regions to be generated.
453   * @return List of RegionInfo.
454   */
455  private List<RegionInfo> randomRegions(int numRegions) {
456    List<RegionInfo> regions = new ArrayList<>(numRegions);
457    byte[] start = new byte[16];
458    byte[] end = new byte[16];
459    rand.nextBytes(start);
460    rand.nextBytes(end);
461    int regionIdx = rand.nextInt(tables.length);
462    for (int i = 0; i < numRegions; i++) {
463      Bytes.putInt(start, 0, numRegions << 1);
464      Bytes.putInt(end, 0, (numRegions << 1) + 1);
465      int tableIndex = (i + regionIdx) % tables.length;
466      regions.add(RegionInfoBuilder.newBuilder(tables[tableIndex])
467          .setStartKey(start)
468          .setEndKey(end)
469          .setSplit(false)
470          .setRegionId(regionId++)
471          .build());
472    }
473    return regions;
474  }

476  /**
477   * Generate assigned regions to a given server using group information.
478   *
479   * @param numRegions the num regions to generate
480   * @param sn the servername
481   * @return the list of regions
482   * @throws java.io.IOException Signals that an I/O exception has occurred.
483   */
484  private List<RegionInfo> assignedRegions(int numRegions, ServerName sn) throws IOException {
485    List<RegionInfo> regions = new ArrayList<>(numRegions);
486    byte[] start = new byte[16];
487    byte[] end = new byte[16];
488    Bytes.putInt(start, 0, numRegions << 1);
489    Bytes.putInt(end, 0, (numRegions << 1) + 1);
490    for (int i = 0; i < numRegions; i++) {
491      TableName tableName = getTableName(sn);
492      regions.add(RegionInfoBuilder.newBuilder(tableName)
493          .setStartKey(start)
494          .setEndKey(end)
495          .setSplit(false)
496          .setRegionId(regionId++)
497          .build());
498    }
499    return regions;
500  }

512  /**
513   * Construct group info, with each group having at least one server.
514   *
515   * @param servers the servers
516   * @param groups the groups
517   * @return the map
518   */
519  private static Map<String, RSGroupInfo> constructGroupInfo(
520      List<ServerName> servers, String[] groups) {
521    assertTrue(servers != null);
522    assertTrue(servers.size() >= groups.length);
523    int index = 0;
524    Map<String, RSGroupInfo> groupMap = new HashMap<>();
525    for (String grpName : groups) {
526      RSGroupInfo RSGroupInfo = new RSGroupInfo(grpName);
527      RSGroupInfo.addServer(servers.get(index).getAddress());
528      groupMap.put(grpName, RSGroupInfo);
529      index++;
530    }
531    while (index < servers.size()) {
532      int grpIndex = rand.nextInt(groups.length);
533      groupMap.get(groups[grpIndex]).addServer(
534          servers.get(index).getAddress());
535      index++;
536    }
537    return groupMap;
538  }

540  /**
541   * Construct table descriptors evenly distributed between the groups.
542   *
543   * @return the list
544   */
545  private static List<HTableDescriptor> constructTableDesc() {
546    List<HTableDescriptor> tds = Lists.newArrayList();
547    int index = rand.nextInt(groups.length);
548    for (int i = 0; i < tables.length; i++) {
549      HTableDescriptor htd = new HTableDescriptor(tables[i]);
550      int grpIndex = (i + index) % groups.length ;
551      String groupName = groups[grpIndex];
552      tableMap.put(tables[i], groupName);
553      tds.add(htd);
554    }
555    return tds;
556  }

594  private TableName getTableName(ServerName sn) throws IOException {
595    TableName tableName = null;
596    RSGroupInfoManager gm = getMockedGroupInfoManager();
597    RSGroupInfo groupOfServer = null;
598    for(RSGroupInfo gInfo : gm.listRSGroups()){
599      if(gInfo.containsServer(sn.getAddress())){
600        groupOfServer = gInfo;
601        break;
602      }
603    }
604
605    for(HTableDescriptor desc : tableDescs){
606      if(gm.getRSGroupOfTable(desc.getTableName()).endsWith(groupOfServer.getName())){
607        tableName = desc.getTableName();
608      }
609    }
610    return tableName;
611  }
