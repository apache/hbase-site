127    int numClusterStatusToAdd = 20000;
128    for (int i = 0; i < numClusterStatusToAdd; i++) {

294        loadBalancer.updateCostsWithAction(cluster, action);

415  @Test
416  public void testReplicaCostForReplicas() {
417    Configuration conf = HBaseConfiguration.create();
418    StochasticLoadBalancer.CostFunction
419        costFunction = new StochasticLoadBalancer.RegionReplicaHostCostFunction(conf);
420
421    int [] servers = new int[] {3,3,3,3,3};
422    TreeMap<ServerName, List<RegionInfo>> clusterState = mockClusterServers(servers);
423
424    BaseLoadBalancer.Cluster cluster;
425
426    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
427    costFunction.init(cluster);
428    double costWithoutReplicas = costFunction.cost();
429    assertEquals(0, costWithoutReplicas, 0);
430
431    // replicate the region from first server to the last server
432    RegionInfo replica1 = RegionReplicaUtil.getRegionInfoForReplica(
433      clusterState.firstEntry().getValue().get(0),1);
434    clusterState.lastEntry().getValue().add(replica1);
435
436    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
437    costFunction.init(cluster);
438    double costWith1ReplicaDifferentServer = costFunction.cost();
439
440    assertEquals(0, costWith1ReplicaDifferentServer, 0);
441
442    // add a third replica to the last server
443    RegionInfo replica2 = RegionReplicaUtil.getRegionInfoForReplica(replica1, 2);
444    clusterState.lastEntry().getValue().add(replica2);
445
446    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
447    costFunction.init(cluster);
448    double costWith1ReplicaSameServer = costFunction.cost();
449
450    assertTrue(costWith1ReplicaDifferentServer < costWith1ReplicaSameServer);
451
452    // test with replication = 4 for following:
453
454    RegionInfo replica3;
455    Iterator<Entry<ServerName, List<RegionInfo>>> it;
456    Entry<ServerName, List<RegionInfo>> entry;
457
458    clusterState = mockClusterServers(servers);
459    it = clusterState.entrySet().iterator();
460    entry = it.next(); //first server
461    RegionInfo hri = entry.getValue().get(0);
462    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
463    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
464    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
465    entry.getValue().add(replica1);
466    entry.getValue().add(replica2);
467    it.next().getValue().add(replica3); //2nd server
468
469    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
470    costFunction.init(cluster);
471    double costWith3ReplicasSameServer = costFunction.cost();
472
473    clusterState = mockClusterServers(servers);
474    hri = clusterState.firstEntry().getValue().get(0);
475    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
476    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
477    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
478
479    clusterState.firstEntry().getValue().add(replica1);
480    clusterState.lastEntry().getValue().add(replica2);
481    clusterState.lastEntry().getValue().add(replica3);
482
483    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
484    costFunction.init(cluster);
485    double costWith2ReplicasOnTwoServers = costFunction.cost();
486
487    assertTrue(costWith2ReplicasOnTwoServers < costWith3ReplicasSameServer);
488  }

492    // check for the case where there are two hosts and with one rack, and where
493    // both the replicas are hosted on the same server
494    List<RegionInfo> regions = randomRegions(1);
495    ServerName s1 = ServerName.valueOf("host1", 1000, 11111);
496    ServerName s2 = ServerName.valueOf("host11", 1000, 11111);
