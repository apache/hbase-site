123  private ServerMetrics mockServerMetricsWithCpRequests(ServerName server,
124      List<RegionInfo> regionsOnServer, long cpRequestCount) {
125    ServerMetrics serverMetrics = mock(ServerMetrics.class);
126    Map<byte[], RegionMetrics> regionLoadMap = new TreeMap<>(Bytes.BYTES_COMPARATOR);
127    for(RegionInfo info : regionsOnServer){
128      RegionMetrics rl = mock(RegionMetrics.class);
129      when(rl.getReadRequestCount()).thenReturn(0L);
130      when(rl.getCpRequestCount()).thenReturn(cpRequestCount);
131      when(rl.getWriteRequestCount()).thenReturn(0L);
132      when(rl.getMemStoreSize()).thenReturn(Size.ZERO);
133      when(rl.getStoreFileSize()).thenReturn(Size.ZERO);
134      regionLoadMap.put(info.getEncodedNameAsBytes(), rl);
135    }
136    when(serverMetrics.getRegionMetrics()).thenReturn(regionLoadMap);
137    return serverMetrics;
138  }

140  @Test
141  public void testCPRequestCost() {
142    // in order to pass needsBalance judgement
143    conf.setFloat("hbase.master.balancer.stochastic.cpRequestCost", 10000f);
144    loadBalancer.setConf(conf);
145    // mock cluster State
146    Map<ServerName, List<RegionInfo>> clusterState = new HashMap<ServerName, List<RegionInfo>>();
147    ServerName serverA = randomServer(3).getServerName();
148    ServerName serverB = randomServer(3).getServerName();
149    ServerName serverC = randomServer(3).getServerName();
150    List<RegionInfo> regionsOnServerA = randomRegions(3);
151    List<RegionInfo> regionsOnServerB = randomRegions(3);
152    List<RegionInfo> regionsOnServerC = randomRegions(3);
153    clusterState.put(serverA, regionsOnServerA);
154    clusterState.put(serverB, regionsOnServerB);
155    clusterState.put(serverC, regionsOnServerC);
156    // mock ClusterMetrics
157    Map<ServerName, ServerMetrics> serverMetricsMap = new TreeMap<>();
158    serverMetricsMap.put(serverA, mockServerMetricsWithCpRequests(serverA, regionsOnServerA, 0));
159    serverMetricsMap.put(serverB, mockServerMetricsWithCpRequests(serverB, regionsOnServerB, 0));
160    serverMetricsMap.put(serverC, mockServerMetricsWithCpRequests(serverC, regionsOnServerC, 0));
161    ClusterMetrics clusterStatus = mock(ClusterMetrics.class);
162    when(clusterStatus.getLiveServerMetrics()).thenReturn(serverMetricsMap);
163    loadBalancer.setClusterMetrics(clusterStatus);
164
165    // CPRequestCostFunction are Rate based, So doing setClusterMetrics again
166    // this time, regions on serverA with more cpRequestCount load
167    // serverA : 1000,1000,1000
168    // serverB : 0,0,0
169    // serverC : 0,0,0
170    // so should move two regions from serverA to serverB & serverC
171    serverMetricsMap = new TreeMap<>();
172    serverMetricsMap.put(serverA, mockServerMetricsWithCpRequests(serverA,
173        regionsOnServerA, 1000));
174    serverMetricsMap.put(serverB, mockServerMetricsWithCpRequests(serverB, regionsOnServerB, 0));
175    serverMetricsMap.put(serverC, mockServerMetricsWithCpRequests(serverC, regionsOnServerC, 0));
176    clusterStatus = mock(ClusterMetrics.class);
177    when(clusterStatus.getLiveServerMetrics()).thenReturn(serverMetricsMap);
178    loadBalancer.setClusterMetrics(clusterStatus);
179
180    List<RegionPlan> plans = loadBalancer.balanceCluster(clusterState);
181    Set<RegionInfo> regionsMoveFromServerA = new HashSet<>();
182    Set<ServerName> targetServers = new HashSet<>();
183    for(RegionPlan plan : plans) {
184      if(plan.getSource().equals(serverA)) {
185        regionsMoveFromServerA.add(plan.getRegionInfo());
186        targetServers.add(plan.getDestination());
187      }
188    }
189    // should move 2 regions from serverA, one moves to serverB, the other moves to serverC
190    assertEquals(2, regionsMoveFromServerA.size());
191    assertEquals(2, targetServers.size());
192    assertTrue(regionsOnServerA.containsAll(regionsMoveFromServerA));
193    // reset config
194    conf.setFloat("hbase.master.balancer.stochastic.cpRequestCost", 5f);
195    loadBalancer.setConf(conf);
196  }

335    final int runs = 10;

337    for (int[] mockCluster : clusterStateMocks) {
338      BaseLoadBalancer.Cluster cluster = mockCluster(mockCluster);

342        Cluster.Action action = loadBalancer.nextAction(cluster);
343        cluster.doAction(action);
344        loadBalancer.updateCostsWithAction(cluster, action);
345        Cluster.Action undoAction = action.undoAction();
346        cluster.doAction(undoAction);
347        loadBalancer.updateCostsWithAction(cluster, undoAction);
348        final double actualCost = loadBalancer.computeCost(cluster, Double.MAX_VALUE);
349        assertEquals(expectedCost, actualCost, 0);
350      }
351    }

381        new StochasticLoadBalancer.ReadRequestCostFunction(conf);

383    // read requests are treated as a rate so the average rate here is simply 1
384    assertEquals(1, rateResult, 0.01);
385
386    StochasticLoadBalancer.CPRequestCostFunction cpCostFunction =
387        new StochasticLoadBalancer.CPRequestCostFunction(conf);
388    rateResult = cpCostFunction.getRegionLoadCost(regionLoads);
389    // coprocessor requests are treated as a rate so the average rate here is simply 1
390    assertEquals(1, rateResult, 0.01);
391
392    StochasticLoadBalancer.StoreFileCostFunction storeFileCostFunction =
393        new StochasticLoadBalancer.StoreFileCostFunction(conf);
394    double result = storeFileCostFunction.getRegionLoadCost(regionLoads);
395    // storefile size cost is simply an average of it's value over time
396    assertEquals(2.5, result, 0.01);

458  // This mock allows us to test the LocalityCostFunction
459  private class MockCluster extends BaseLoadBalancer.Cluster {
460
461    private int[][] localities = null;   // [region][server] = percent of blocks
462
463    public MockCluster(int[][] regions) {
464
465      // regions[0] is an array where index = serverIndex an value = number of regions
466      super(mockClusterServers(regions[0], 1), null, null, null);
467
468      localities = new int[regions.length - 1][];
469      for (int i = 1; i < regions.length; i++) {
470        int regionIndex = i - 1;
471        localities[regionIndex] = new int[regions[i].length - 1];
472        regionIndexToServerIndex[regionIndex] = regions[i][0];
473        for (int j = 1; j < regions[i].length; j++) {
474          int serverIndex = j - 1;
475          localities[regionIndex][serverIndex] = regions[i][j] > 100 ? regions[i][j] % 100 : regions[i][j];
476        }
477      }
478    }
479
480    @Override
481    float getLocalityOfRegion(int region, int server) {
482      // convert the locality percentage to a fraction
483      return localities[region][server] / 100.0f;
484    }
485
486    @Override
487    public int getRegionSizeMB(int region) {
488      return 1;
489    }
