019package org.apache.hadoop.hbase.master;
020
021import static org.junit.Assert.assertEquals;
022import static org.junit.Assert.assertTrue;
023
024import java.util.List;
025import java.util.NavigableSet;
026
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.hbase.HBaseConfiguration;
029import org.apache.hadoop.hbase.HBaseTestingUtility;
030import org.apache.hadoop.hbase.MiniHBaseCluster;
031import org.apache.hadoop.hbase.TableName;
032import org.apache.hadoop.hbase.client.Admin;
033import org.apache.hadoop.hbase.client.RegionLocator;
034import org.apache.hadoop.hbase.client.Table;
035import org.apache.hadoop.hbase.client.TableState;
036import org.apache.hadoop.hbase.testclassification.LargeTests;
037import org.apache.hadoop.hbase.testclassification.MasterTests;
038import org.apache.hadoop.hbase.util.Bytes;
039import org.apache.hadoop.hbase.util.JVMClusterUtil.MasterThread;
040import org.junit.Rule;
041import org.junit.Test;
042import org.junit.experimental.categories.Category;
043import org.junit.rules.TestName;
044import org.slf4j.Logger;
045import org.slf4j.LoggerFactory;
046
047@Category({MasterTests.class, LargeTests.class})
048public class TestMasterRestartAfterDisablingTable {
049
050  private static final Logger LOG =
051      LoggerFactory.getLogger(TestMasterRestartAfterDisablingTable.class);
052
053  @Rule
054  public TestName name = new TestName();
055
056  @Test
057  public void testForCheckingIfEnableAndDisableWorksFineAfterSwitch()
058      throws Exception {
059    final int NUM_MASTERS = 2;
060    final int NUM_RS = 1;
061    final int NUM_REGIONS_TO_CREATE = 4;
062
063    // Start the cluster
064    log("Starting cluster");
065    Configuration conf = HBaseConfiguration.create();
066    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility(conf);
067    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
068    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
069    log("Waiting for active/ready master");
070    cluster.waitForActiveAndReadyMaster();
071
072    // Create a table with regions
073    final TableName tableName = TableName.valueOf(name.getMethodName());
074    byte[] family = Bytes.toBytes("family");
075    log("Creating table with " + NUM_REGIONS_TO_CREATE + " regions");
076    Table ht = TEST_UTIL.createMultiRegionTable(tableName, family, NUM_REGIONS_TO_CREATE);
077    int numRegions = -1;
078    try (RegionLocator r = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
079      numRegions = r.getStartKeys().length;
080    }
081    numRegions += 1; // catalogs
082    log("Waiting for no more RIT\n");
083    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
084    log("Disabling table\n");
085    TEST_UTIL.getAdmin().disableTable(tableName);
086
087    NavigableSet<String> regions = HBaseTestingUtility.getAllOnlineRegions(cluster);
088    assertEquals(
089        "The number of regions for the table tableRestart should be 0 and only"
090            + "the catalog and namespace tables should be present.", 2, regions.size());
091
092    List<MasterThread> masterThreads = cluster.getMasterThreads();
093    MasterThread activeMaster = null;
094    if (masterThreads.get(0).getMaster().isActiveMaster()) {
095      activeMaster = masterThreads.get(0);
096    } else {
097      activeMaster = masterThreads.get(1);
098    }
099    activeMaster.getMaster().stop(
100        "stopping the active master so that the backup can become active");
101    cluster.hbaseCluster.waitOnMaster(activeMaster);
102    cluster.waitForActiveAndReadyMaster();
103
104    assertTrue("The table should not be in enabled state",
105        cluster.getMaster().getTableStateManager().isTableState(
106        TableName.valueOf(name.getMethodName()), TableState.State.DISABLED,
107        TableState.State.DISABLING));
108    log("Enabling table\n");
109    // Need a new Admin, the previous one is on the old master
110    Admin admin = TEST_UTIL.getAdmin();
111    admin.enableTable(tableName);
112    admin.close();
113    log("Waiting for no more RIT\n");
114    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
115    log("Verifying there are " + numRegions + " assigned on cluster\n");
116    regions = HBaseTestingUtility.getAllOnlineRegions(cluster);
117    assertEquals("The assigned regions were not onlined after master"
118        + " switch except for the catalog and namespace tables.",
119          6, regions.size());
120    assertTrue("The table should be in enabled state",
121        cluster.getMaster().getTableStateManager()
122        .isTableState(TableName.valueOf(name.getMethodName()), TableState.State.ENABLED));
123    ht.close();
124    TEST_UTIL.shutdownMiniCluster();
125  }
126
127  private void log(String msg) {
128    LOG.debug("\n\nTRR: " + msg + "\n");
129  }
130}
