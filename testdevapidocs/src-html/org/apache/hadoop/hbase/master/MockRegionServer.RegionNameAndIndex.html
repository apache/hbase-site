140/**
141 * A mock RegionServer implementation.
142 * Use this when you can't bend Mockito to your liking (e.g. return null result
143 * when 'scanning' until master timesout and then return a coherent meta row
144 * result thereafter.  Have some facility for faking gets and scans.  See
145 * setGetResult(byte[], byte[], Result) for how to fill the backing data
146 * store that the get pulls from.
147 */
148class MockRegionServer implements AdminProtos.AdminService.BlockingInterface,
149    ClientProtos.ClientService.BlockingInterface, RegionServerServices {
150  private final ServerName sn;
151  private final ZKWatcher zkw;
152  private final Configuration conf;
153  private final Random random = new Random();
154
155  /**
156   * Map of regions to map of rows and {@link Result}. Used as data source when
157   * {@link #get(RpcController, ClientProtos.GetRequest)} is called. Because we have a byte
158   * key, need to use TreeMap and provide a Comparator.  Use
159   * {@link #setGetResult(byte[], byte[], Result)} filling this map.
160   */
161  private final Map<byte [], Map<byte [], Result>> gets = new TreeMap<>(Bytes.BYTES_COMPARATOR);
162
163  /**
164   * Map of regions to results to return when scanning.
165   */
166  private final Map<byte [], Result []> nexts = new TreeMap<>(Bytes.BYTES_COMPARATOR);
167
168  /**
169   * Data structure that holds regionname and index used scanning.
170   */
171  class RegionNameAndIndex {
172    private final byte[] regionName;
173    private int index = 0;
174
175    RegionNameAndIndex(final byte[] regionName) {
176      this.regionName = regionName;
177    }
178
179    byte[] getRegionName() {
180      return this.regionName;
181    }
182
183    int getThenIncrement() {
184      int currentIndex = this.index;
185      this.index++;
186      return currentIndex;
187    }
188  }
189
190  /**
191   * Outstanding scanners and their offset into <code>nexts</code>
192   */
193  private final Map<Long, RegionNameAndIndex> scannersAndOffsets = new HashMap<>();
194
195  /**
196   * @param sn Name of this mock regionserver
197   * @throws IOException
198   * @throws org.apache.hadoop.hbase.ZooKeeperConnectionException
199   */
200  MockRegionServer(final Configuration conf, final ServerName sn)
201  throws ZooKeeperConnectionException, IOException {
202    this.sn = sn;
203    this.conf = conf;
204    this.zkw = new ZKWatcher(conf, sn.toString(), this, true);
205  }
206
207  /**
208   * Use this method filling the backing data source used by
209   * {@link #get(RpcController, ClientProtos.GetRequest)}
210   * @param regionName the region name to assign
211   * @param row the row key
212   * @param r the single row result
213   */
214  void setGetResult(final byte [] regionName, final byte [] row, final Result r) {
215    Map<byte [], Result> value = this.gets.get(regionName);
216    if (value == null) {
217      // If no value already, create one.  Needs to be treemap because we are
218      // using byte array as key.   Not thread safe.
219      value = new TreeMap<>(Bytes.BYTES_COMPARATOR);
220      this.gets.put(regionName, value);
221    }
222    value.put(row, r);
223  }
224
225  /**
226   * Use this method to set what a scanner will reply as we next through
227   * @param regionName
228   * @param rs
229   */
230  void setNextResults(final byte [] regionName, final Result [] rs) {
231    this.nexts.put(regionName, rs);
232  }

249  public long openScanner(byte[] regionName, Scan scan) throws IOException {
250    long scannerId = this.random.nextLong();
251    this.scannersAndOffsets.put(scannerId, new RegionNameAndIndex(regionName));
252    return scannerId;
253  }
254
255  public Result next(long scannerId) throws IOException {
256    RegionNameAndIndex rnai = this.scannersAndOffsets.get(scannerId);
257    int index = rnai.getThenIncrement();
258    Result [] results = this.nexts.get(rnai.getRegionName());
259    if (results == null) return null;
260    return index < results.length? results[index]: null;
261  }
262
263  public Result [] next(long scannerId, int numberOfRows) throws IOException {
264    // Just return one result whatever they ask for.
265    Result r = next(scannerId);
266    return r == null? null: new Result [] {r};
267  }
268
269  public void close(final long scannerId) throws IOException {
270    this.scannersAndOffsets.remove(scannerId);
271  }

359  @Override
360  public GetResponse get(RpcController controller, GetRequest request)
361  throws ServiceException {
362    byte[] regionName = request.getRegion().getValue().toByteArray();
363    Map<byte [], Result> m = this.gets.get(regionName);
364    GetResponse.Builder builder = GetResponse.newBuilder();
365    if (m != null) {
366      byte[] row = request.getGet().getRow().toByteArray();
367      builder.setResult(ProtobufUtil.toResult(m.get(row)));
368    }
369    return builder.build();
370  }

378  @Override
379  public ScanResponse scan(RpcController controller, ScanRequest request)
380      throws ServiceException {
381    ScanResponse.Builder builder = ScanResponse.newBuilder();
382    try {
383      if (request.hasScan()) {
384        byte[] regionName = request.getRegion().getValue().toByteArray();
385        builder.setScannerId(openScanner(regionName, null));
386        builder.setMoreResults(true);
387      }
388      else {
389        long scannerId = request.getScannerId();
390        Result result = next(scannerId);
391        if (result != null) {
392          builder.addCellsPerResult(result.size());
393          List<CellScannable> results = new ArrayList<>(1);
394          results.add(result);
395          ((HBaseRpcController) controller).setCellScanner(CellUtil
396              .createCellScanner(results));
397          builder.setMoreResults(true);
398        }
399        else {
400          builder.setMoreResults(false);
401          close(scannerId);
402        }
403      }
404    } catch (IOException ie) {
405      throw new ServiceException(ie);
406    }
407    return builder.build();
408  }
