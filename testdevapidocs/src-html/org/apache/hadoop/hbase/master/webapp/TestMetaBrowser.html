091  @Test
092  public void noFilters() {
093    final String namespaceName = testNameRule.getMethodName();
094    final TableName a = TableName.valueOf("a");
095    final TableName b = TableName.valueOf(namespaceName, "b");
096
097    CompletableFuture.allOf(
098      createTable(a),
099      createNamespace(namespaceName).thenCompose(_void -> createTable(b, 2)))
100      .join();
101
102    final HttpServletRequest request = new MockRequestBuilder().build();
103    final List<RegionReplicaInfo> rows;
104    try (final MetaBrowser.Results results = new MetaBrowser(connection, request).getResults()) {
105      rows = IterableUtils.toList(results);
106    }
107    assertThat(rows, contains(
108      hasProperty("row", bytesAsStringBinary(startsWith(a + ",,"))),
109      hasProperty("row", bytesAsStringBinary(startsWith(b + ",,"))),
110      hasProperty("row", bytesAsStringBinary(startsWith(b + ",80000000")))));
111  }

186    final TableName b = TableName.valueOf(namespaceName, "b");
187

195      .build();

270  private ColumnFamilyDescriptor columnFamilyDescriptor() {
271    return ColumnFamilyDescriptorBuilder.of("f1");
272  }
273
274  private TableDescriptor tableDescriptor(final TableName tableName) {
275    return TableDescriptorBuilder.newBuilder(tableName)
276      .setColumnFamily(columnFamilyDescriptor())
277      .build();
278  }
279
280  private TableDescriptor tableDescriptor(final TableName tableName, final int replicaCount) {
281    return TableDescriptorBuilder.newBuilder(tableName)
282      .setRegionReplication(replicaCount)
283      .setColumnFamily(columnFamilyDescriptor())
284      .build();
285  }
286
287  private CompletableFuture<Void> createTable(final TableName tableName) {
288    return admin.createTable(tableDescriptor(tableName));
289  }
290
291  private CompletableFuture<Void> createTable(final TableName tableName, final int splitCount) {
292    return admin.createTable(
293      tableDescriptor(tableName),
294      new RegionSplitter.HexStringSplit().split(splitCount));
295  }
296
297  private CompletableFuture<Void> createTableWithReplicas(final TableName tableName,
298    final int replicaCount) {
299    return admin.createTable(tableDescriptor(tableName, replicaCount));
300  }
301
302  private CompletableFuture<Void> createNamespace(final String namespace) {
303    final NamespaceDescriptor descriptor = NamespaceDescriptor.create(namespace).build();
304    return admin.createNamespace(descriptor);
305  }

307  /**
308   * Helper for mocking an {@link HttpServletRequest} relevant to the test.
309   */
310  static class MockRequestBuilder {
311
312    private String limit = null;
313    private String regionState = null;
314    private String start = null;
315    private String table = null;
316
317    public MockRequestBuilder setLimit(final int value) {
318      this.limit = Integer.toString(value);
319      return this;
320    }
321
322    public MockRequestBuilder setLimit(final String value) {
323      this.limit = value;
324      return this;
325    }
326
327    public MockRequestBuilder setRegionState(final RegionState.State value) {
328      this.regionState = value.toString();
329      return this;
330    }
331
332    public MockRequestBuilder setRegionState(final String value) {
333      this.regionState = value;
334      return this;
335    }
336
337    public MockRequestBuilder setStart(final String value) {
338      this.start = value;
339      return this;
340    }
341
342    public MockRequestBuilder setTable(final String value) {
343      this.table = value;
344      return this;
345    }
346
347    public HttpServletRequest build() {
348      final HttpServletRequest request = mock(HttpServletRequest.class);
349      when(request.getRequestURI()).thenReturn("/table.jsp");
350      when(request.getParameter("name")).thenReturn("hbase%3Ameta");
351
352      when(request.getParameter("scan_limit")).thenReturn(limit);
353      when(request.getParameter("scan_region_state")).thenReturn(regionState);
354      when(request.getParameter("scan_start")).thenReturn(start);
355      when(request.getParameter("scan_table")).thenReturn(table);
356
357      return request;
358    }
359  }
