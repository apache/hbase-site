043@Category({MasterTests.class, MediumTests.class})

048  final ServerName hostname123_2 = ServerName.valueOf("127.0.0.1", 123, 4L);
049  final ServerName hostname1234 = ServerName.valueOf("127.0.0.2", 1234, 4L);

062  @Test public void testIsDead() {
063    DeadServer ds = new DeadServer();
064    ds.add(hostname123);
065    ds.notifyServer(hostname123);
066    assertTrue(ds.areDeadServersInProgress());
067    ds.finish(hostname123);
068    assertFalse(ds.areDeadServersInProgress());
069
070    ds.add(hostname1234);
071    ds.notifyServer(hostname1234);
072    assertTrue(ds.areDeadServersInProgress());
073    ds.finish(hostname1234);
074    assertFalse(ds.areDeadServersInProgress());
075
076    ds.add(hostname12345);
077    ds.notifyServer(hostname12345);
078    assertTrue(ds.areDeadServersInProgress());
079    ds.finish(hostname12345);
080    assertFalse(ds.areDeadServersInProgress());
081
082    // Already dead =       127.0.0.1,9090,112321
083    // Coming back alive =  127.0.0.1,9090,223341
084
085    final ServerName deadServer = ServerName.valueOf("127.0.0.1", 9090, 112321L);
086    assertFalse(ds.cleanPreviousInstance(deadServer));
087    ds.add(deadServer);
088    assertTrue(ds.isDeadServer(deadServer));
089    Set<ServerName> deadServerNames = ds.copyServerNames();
090    for (ServerName eachDeadServer : deadServerNames) {
091      Assert.assertNotNull(ds.getTimeOfDeath(eachDeadServer));
092    }
093    final ServerName deadServerHostComingAlive =
094        ServerName.valueOf("127.0.0.1", 9090, 223341L);
095    assertTrue(ds.cleanPreviousInstance(deadServerHostComingAlive));
096    assertFalse(ds.isDeadServer(deadServer));
097    assertFalse(ds.cleanPreviousInstance(deadServerHostComingAlive));
098  }

100  @Test(timeout = 15000)
101  public void testCrashProcedureReplay() {
102    HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
103    final ProcedureExecutor<MasterProcedureEnv> pExecutor = master.getMasterProcedureExecutor();
104    ServerCrashProcedure proc = new ServerCrashProcedure(
105      pExecutor.getEnvironment(), hostname123, false, false);
106
107    ProcedureTestingUtility.submitAndWait(pExecutor, proc);
108    
109    assertFalse(master.getServerManager().getDeadServers().areDeadServersInProgress());
110  }

112  @Test
113  public void testSortExtract(){
114    ManualEnvironmentEdge mee = new ManualEnvironmentEdge();
115    EnvironmentEdgeManager.injectEdge(mee);
116    mee.setValue(1);
117
118    DeadServer d = new DeadServer();
119
120
121    d.add(hostname123);
122    mee.incValue(1);
123    d.add(hostname1234);
124    mee.incValue(1);
125    d.add(hostname12345);
126
127    List<Pair<ServerName, Long>> copy = d.copyDeadServersSince(2L);
128    Assert.assertEquals(2, copy.size());
129
130    Assert.assertEquals(hostname1234, copy.get(0).getFirst());
131    Assert.assertEquals(new Long(2L), copy.get(0).getSecond());
132
133    Assert.assertEquals(hostname12345, copy.get(1).getFirst());
134    Assert.assertEquals(new Long(3L), copy.get(1).getSecond());
135
136    EnvironmentEdgeManager.reset();
137  }

139  @Test
140  public void testClean(){
141    DeadServer d = new DeadServer();
142    d.add(hostname123);
143
144    d.cleanPreviousInstance(hostname12345);
145    Assert.assertFalse(d.isEmpty());
146
147    d.cleanPreviousInstance(hostname1234);
148    Assert.assertFalse(d.isEmpty());
149
150    d.cleanPreviousInstance(hostname123_2);
151    Assert.assertTrue(d.isEmpty());
152  }
