223  @Test
224  public void testCleanParent() throws IOException, InterruptedException {
225    HBaseTestingUtility htu = new HBaseTestingUtility();
226    setRootDirAndCleanIt(htu, "testCleanParent");
227    MasterServices services = new MockMasterServices(htu);
228    try {
229      CatalogJanitor janitor = new CatalogJanitor(services);
230      // Create regions.
231      HTableDescriptor htd = new HTableDescriptor(TableName.valueOf("table"));
232      htd.addFamily(new HColumnDescriptor("f"));
233      HRegionInfo parent =
234        new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
235            Bytes.toBytes("eee"));
236      HRegionInfo splita =
237        new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
238            Bytes.toBytes("ccc"));
239      HRegionInfo splitb =
240        new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
241            Bytes.toBytes("eee"));
242      // Test that when both daughter regions are in place, that we do not
243      // remove the parent.
244      Result r = createResult(parent, splita, splitb);
245      // Add a reference under splitA directory so we don't clear out the parent.
246      Path rootdir = services.getMasterFileSystem().getRootDir();
247      Path tabledir =
248        FSUtils.getTableDir(rootdir, htd.getTableName());
249      Path storedir = HStore.getStoreHomedir(tabledir, splita,
250          htd.getColumnFamilies()[0].getName());
251      Reference ref = Reference.createTopReference(Bytes.toBytes("ccc"));
252      long now = System.currentTimeMillis();
253      // Reference name has this format: StoreFile#REF_NAME_PARSER
254      Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
255      FileSystem fs = services.getMasterFileSystem().getFileSystem();
256      Path path = ref.write(fs, p);
257      assertTrue(fs.exists(path));
258      assertFalse(janitor.cleanParent(parent, r));
259      // Remove the reference file and try again.
260      assertTrue(fs.delete(p, true));
261      assertTrue(janitor.cleanParent(parent, r));
262    } finally {
263      services.stop("shutdown");
264    }
265  }

291  /**
292   * Make sure parent with specified end key gets cleaned up even if daughter is cleaned up before it.
293   *
294   * @param rootDir the test case name, used as the HBase testing utility root
295   * @param lastEndKey the end key of the split parent
296   * @throws IOException
297   * @throws InterruptedException
298   */
299  private void parentWithSpecifiedEndKeyCleanedEvenIfDaughterGoneFirst(
300  final String rootDir, final byte[] lastEndKey)
301  throws IOException, InterruptedException {
302    HBaseTestingUtility htu = new HBaseTestingUtility();
303    setRootDirAndCleanIt(htu, rootDir);
304    MasterServices services = new MockMasterServices(htu);
305    CatalogJanitor janitor = new CatalogJanitor(services);
306    final HTableDescriptor htd = createHTableDescriptor();
307
308    // Create regions: aaa->{lastEndKey}, aaa->ccc, aaa->bbb, bbb->ccc, etc.
309
310    // Parent
311    HRegionInfo parent = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
312      lastEndKey);
313    // Sleep a second else the encoded name on these regions comes out
314    // same for all with same start key and made in same second.
315    Thread.sleep(1001);
316
317    // Daughter a
318    HRegionInfo splita = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
319      Bytes.toBytes("ccc"));
320    Thread.sleep(1001);
321    // Make daughters of daughter a; splitaa and splitab.
322    HRegionInfo splitaa = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
323      Bytes.toBytes("bbb"));
324    HRegionInfo splitab = new HRegionInfo(htd.getTableName(), Bytes.toBytes("bbb"),
325      Bytes.toBytes("ccc"));
326
327    // Daughter b
328    HRegionInfo splitb = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
329      lastEndKey);
330    Thread.sleep(1001);
331    // Make Daughters of daughterb; splitba and splitbb.
332    HRegionInfo splitba = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
333      Bytes.toBytes("ddd"));
334    HRegionInfo splitbb = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ddd"),
335    lastEndKey);
336
337    // First test that our Comparator works right up in CatalogJanitor.
338    // Just fo kicks.
339    SortedMap<HRegionInfo, Result> regions =
340      new TreeMap<HRegionInfo, Result>(new CatalogJanitor.SplitParentFirstComparator());
341    // Now make sure that this regions map sorts as we expect it to.
342    regions.put(parent, createResult(parent, splita, splitb));
343    regions.put(splitb, createResult(splitb, splitba, splitbb));
344    regions.put(splita, createResult(splita, splitaa, splitab));
345    // Assert its properly sorted.
346    int index = 0;
347    for (Map.Entry<HRegionInfo, Result> e: regions.entrySet()) {
348      if (index == 0) {
349        assertTrue(e.getKey().getEncodedName().equals(parent.getEncodedName()));
350      } else if (index == 1) {
351        assertTrue(e.getKey().getEncodedName().equals(splita.getEncodedName()));
352      } else if (index == 2) {
353        assertTrue(e.getKey().getEncodedName().equals(splitb.getEncodedName()));
354      }
355      index++;
356    }
357
358    // Now play around with the cleanParent function.  Create a ref from splita
359    // up to the parent.
360    Path splitaRef =
361      createReferences(services, htd, parent, splita, Bytes.toBytes("ccc"), false);
362    // Make sure actual super parent sticks around because splita has a ref.
363    assertFalse(janitor.cleanParent(parent, regions.get(parent)));
364
365    //splitba, and split bb, do not have dirs in fs.  That means that if
366    // we test splitb, it should get cleaned up.
367    assertTrue(janitor.cleanParent(splitb, regions.get(splitb)));
368
369    // Now remove ref from splita to parent... so parent can be let go and so
370    // the daughter splita can be split (can't split if still references).
371    // BUT make the timing such that the daughter gets cleaned up before we
372    // can get a chance to let go of the parent.
373    FileSystem fs = FileSystem.get(htu.getConfiguration());
374    assertTrue(fs.delete(splitaRef, true));
375    // Create the refs from daughters of splita.
376    Path splitaaRef =
377      createReferences(services, htd, splita, splitaa, Bytes.toBytes("bbb"), false);
378    Path splitabRef =
379      createReferences(services, htd, splita, splitab, Bytes.toBytes("bbb"), true);
380
381    // Test splita.  It should stick around because references from splitab, etc.
382    assertFalse(janitor.cleanParent(splita, regions.get(splita)));
383
384    // Now clean up parent daughter first.  Remove references from its daughters.
385    assertTrue(fs.delete(splitaaRef, true));
386    assertTrue(fs.delete(splitabRef, true));
387    assertTrue(janitor.cleanParent(splita, regions.get(splita)));
388
389    // Super parent should get cleaned up now both splita and splitb are gone.
390    assertTrue(janitor.cleanParent(parent, regions.get(parent)));
391
392    services.stop("test finished");
393    janitor.cancel(true);
394  }

417
418    // Daughter a
419    HRegionInfo splita = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),

426      Bytes.toBytes("ccc"), false);
427

431    Thread.sleep(1001);
432
433    final Map<HRegionInfo, Result> splitParents =
434        new TreeMap<HRegionInfo, Result>(new SplitParentFirstComparator());
435    splitParents.put(parent, createResult(parent, splita, splitb));

467  @Test
468  public void testSplitParentFirstComparator() {
469    SplitParentFirstComparator comp = new SplitParentFirstComparator();
470    final HTableDescriptor htd = createHTableDescriptor();
471
472    /*  Region splits:
473     *
474     *  rootRegion --- firstRegion --- firstRegiona
475     *              |               |- firstRegionb
476     *              |
477     *              |- lastRegion --- lastRegiona  --- lastRegionaa
478     *                             |                |- lastRegionab
479     *                             |- lastRegionb
480     *
481     *  rootRegion   :   []  - []
482     *  firstRegion  :   []  - bbb
483     *  lastRegion   :   bbb - []
484     *  firstRegiona :   []  - aaa
485     *  firstRegionb :   aaa - bbb
486     *  lastRegiona  :   bbb - ddd
487     *  lastRegionb  :   ddd - []
488     */
489
490    // root region
491    HRegionInfo rootRegion = new HRegionInfo(htd.getTableName(),
492      HConstants.EMPTY_START_ROW,
493      HConstants.EMPTY_END_ROW, true);
494    HRegionInfo firstRegion = new HRegionInfo(htd.getTableName(),
495      HConstants.EMPTY_START_ROW,
496      Bytes.toBytes("bbb"), true);
497    HRegionInfo lastRegion = new HRegionInfo(htd.getTableName(),
498      Bytes.toBytes("bbb"),
499      HConstants.EMPTY_END_ROW, true);
500
501    assertTrue(comp.compare(rootRegion, rootRegion) == 0);
502    assertTrue(comp.compare(firstRegion, firstRegion) == 0);
503    assertTrue(comp.compare(lastRegion, lastRegion) == 0);
504    assertTrue(comp.compare(rootRegion, firstRegion) < 0);
505    assertTrue(comp.compare(rootRegion, lastRegion) < 0);
506    assertTrue(comp.compare(firstRegion, lastRegion) < 0);
507
508    //first region split into a, b
509    HRegionInfo firstRegiona = new HRegionInfo(htd.getTableName(),
510      HConstants.EMPTY_START_ROW,
511      Bytes.toBytes("aaa"), true);
512    HRegionInfo firstRegionb = new HRegionInfo(htd.getTableName(),
513        Bytes.toBytes("aaa"),
514      Bytes.toBytes("bbb"), true);
515    //last region split into a, b
516    HRegionInfo lastRegiona = new HRegionInfo(htd.getTableName(),
517      Bytes.toBytes("bbb"),
518      Bytes.toBytes("ddd"), true);
519    HRegionInfo lastRegionb = new HRegionInfo(htd.getTableName(),
520      Bytes.toBytes("ddd"),
521      HConstants.EMPTY_END_ROW, true);
522
523    assertTrue(comp.compare(firstRegiona, firstRegiona) == 0);
524    assertTrue(comp.compare(firstRegionb, firstRegionb) == 0);
525    assertTrue(comp.compare(rootRegion, firstRegiona) < 0);
526    assertTrue(comp.compare(rootRegion, firstRegionb) < 0);
527    assertTrue(comp.compare(firstRegion, firstRegiona) < 0);
528    assertTrue(comp.compare(firstRegion, firstRegionb) < 0);
529    assertTrue(comp.compare(firstRegiona, firstRegionb) < 0);
530
531    assertTrue(comp.compare(lastRegiona, lastRegiona) == 0);
532    assertTrue(comp.compare(lastRegionb, lastRegionb) == 0);
533    assertTrue(comp.compare(rootRegion, lastRegiona) < 0);
534    assertTrue(comp.compare(rootRegion, lastRegionb) < 0);
535    assertTrue(comp.compare(lastRegion, lastRegiona) < 0);
536    assertTrue(comp.compare(lastRegion, lastRegionb) < 0);
537    assertTrue(comp.compare(lastRegiona, lastRegionb) < 0);
538
539    assertTrue(comp.compare(firstRegiona, lastRegiona) < 0);
540    assertTrue(comp.compare(firstRegiona, lastRegionb) < 0);
541    assertTrue(comp.compare(firstRegionb, lastRegiona) < 0);
542    assertTrue(comp.compare(firstRegionb, lastRegionb) < 0);
543
544    HRegionInfo lastRegionaa = new HRegionInfo(htd.getTableName(),
545      Bytes.toBytes("bbb"),
546      Bytes.toBytes("ccc"), false);
547    HRegionInfo lastRegionab = new HRegionInfo(htd.getTableName(),
548      Bytes.toBytes("ccc"),
549      Bytes.toBytes("ddd"), false);
550
551    assertTrue(comp.compare(lastRegiona, lastRegionaa) < 0);
552    assertTrue(comp.compare(lastRegiona, lastRegionab) < 0);
553    assertTrue(comp.compare(lastRegionaa, lastRegionab) < 0);

588    Path storedir = HStore.getStoreHomedir(tabledir, parent, htd.getColumnFamilies()[0].getName());
589    Path storeArchive = HFileArchiveUtil.getStoreArchivePath(services.getConfiguration(), parent,
590      tabledir, htd.getColumnFamilies()[0].getName());
591    LOG.debug("Table dir:" + tabledir);
592    LOG.debug("Store dir:" + storedir);
593    LOG.debug("Store archive dir:" + storeArchive);
594
595    // add a couple of store files that we can check for
596    FileStatus[] mockFiles = addMockStoreFiles(2, services, storedir);
597    // get the current store files for comparison
598    FileStatus[] storeFiles = fs.listStatus(storedir);
599    int index = 0;
600    for (FileStatus file : storeFiles) {
601      LOG.debug("Have store file:" + file.getPath());
602      assertEquals("Got unexpected store file", mockFiles[index].getPath(),
603        storeFiles[index].getPath());
604      index++;
605    }
606
607    // do the cleaning of the parent
608    assertTrue(janitor.cleanParent(parent, parentMetaRow));
609    LOG.debug("Finished cleanup of parent region");
610
611    // and now check to make sure that the files have actually been archived
612    FileStatus[] archivedStoreFiles = fs.listStatus(storeArchive);
613    logFiles("archived files", storeFiles);
614    logFiles("archived files", archivedStoreFiles);
615
616    assertArchiveEqualToOriginal(storeFiles, archivedStoreFiles, fs);
617
618    // cleanup
619    FSUtils.delete(fs, rootdir, true);
620    services.stop("Test finished");
621    janitor.cancel(true);
622  }

639  @Test

677      tabledir, htd.getColumnFamilies()[0].getName());
678    System.out.println("Old archive:" + storeArchive);

698    // and now check to make sure that the files have actually been archived

736  /**
737   * @param services Master services instance.
738   * @param htd
739   * @param parent
740   * @param daughter
741   * @param midkey
742   * @param top True if we are to write a 'top' reference.
743   * @return Path to reference we created.
744   * @throws IOException
745   */
746  private Path createReferences(final MasterServices services,
747      final HTableDescriptor htd, final HRegionInfo parent,
748      final HRegionInfo daughter, final byte [] midkey, final boolean top)
749  throws IOException {
750    Path rootdir = services.getMasterFileSystem().getRootDir();
751    Path tabledir = FSUtils.getTableDir(rootdir, parent.getTable());
752    Path storedir = HStore.getStoreHomedir(tabledir, daughter,
753      htd.getColumnFamilies()[0].getName());
754    Reference ref =
755      top? Reference.createTopReference(midkey): Reference.createBottomReference(midkey);
756    long now = System.currentTimeMillis();
757    // Reference name has this format: StoreFile#REF_NAME_PARSER
758    Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
759    FileSystem fs = services.getMasterFileSystem().getFileSystem();
760    ref.write(fs, p);
761    return p;
762  }

776  private MultiResponse buildMultiResponse(MultiRequest req) {
777    MultiResponse.Builder builder = MultiResponse.newBuilder();
778    RegionActionResult.Builder regionActionResultBuilder =
779        RegionActionResult.newBuilder();
780    ResultOrException.Builder roeBuilder = ResultOrException.newBuilder();
781    for (RegionAction regionAction: req.getRegionActionList()) {
782      regionActionResultBuilder.clear();
783      for (ClientProtos.Action action: regionAction.getActionList()) {
784        roeBuilder.clear();
785        roeBuilder.setResult(ClientProtos.Result.getDefaultInstance());
786        roeBuilder.setIndex(action.getIndex());
787        regionActionResultBuilder.addResultOrException(roeBuilder.build());
788      }
789      builder.addRegionActionResult(regionActionResultBuilder.build());
790    }
791    return builder.build();
792  }
