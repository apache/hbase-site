108  /**
109   * Test clearing a split parent.
110   */
111  @Test
112  public void testCleanParent() throws IOException, InterruptedException {
113    TableDescriptor td = createTableDescriptorForCurrentMethod();
114    // Create regions.
115    HRegionInfo parent =
116        new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"), Bytes.toBytes("eee"));
117    HRegionInfo splita =
118        new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"), Bytes.toBytes("ccc"));
119    HRegionInfo splitb =
120        new HRegionInfo(td.getTableName(), Bytes.toBytes("ccc"), Bytes.toBytes("eee"));
121    // Test that when both daughter regions are in place, that we do not remove the parent.
122    Result r = createResult(parent, splita, splitb);
123    // Add a reference under splitA directory so we don't clear out the parent.
124    Path rootdir = this.masterServices.getMasterFileSystem().getRootDir();
125    Path tabledir = FSUtils.getTableDir(rootdir, td.getTableName());
126    Path parentdir = new Path(tabledir, parent.getEncodedName());
127    Path storedir = HStore.getStoreHomedir(tabledir, splita, td.getColumnFamilies()[0].getName());
128    Reference ref = Reference.createTopReference(Bytes.toBytes("ccc"));
129    long now = System.currentTimeMillis();
130    // Reference name has this format: StoreFile#REF_NAME_PARSER
131    Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
132    FileSystem fs = this.masterServices.getMasterFileSystem().getFileSystem();
133    Path path = ref.write(fs, p);
134    assertTrue(fs.exists(path));
135    LOG.info("Created reference " + path);
136    // Add a parentdir for kicks so can check it gets removed by the catalogjanitor.
137    fs.mkdirs(parentdir);
138    assertFalse(this.janitor.cleanParent(parent, r));
139    ProcedureTestingUtility.waitAllProcedures(masterServices.getMasterProcedureExecutor());
140    assertTrue(fs.exists(parentdir));
141    // Remove the reference file and try again.
142    assertTrue(fs.delete(p, true));
143    assertTrue(this.janitor.cleanParent(parent, r));
144    // Parent cleanup is run async as a procedure. Make sure parentdir is removed.
145    ProcedureTestingUtility.waitAllProcedures(masterServices.getMasterProcedureExecutor());
146    assertTrue(!fs.exists(parentdir));
147  }

179  /**
180   * Make sure parent with specified end key gets cleaned up even if daughter is cleaned up before it.
181   *
182   * @param rootDir the test case name, used as the HBase testing utility root
183   * @param lastEndKey the end key of the split parent
184   */
185  private void parentWithSpecifiedEndKeyCleanedEvenIfDaughterGoneFirst(
186  final String rootDir, final byte[] lastEndKey)
187  throws IOException, InterruptedException {
188    TableDescriptor td = createTableDescriptorForCurrentMethod();
189    // Create regions: aaa->{lastEndKey}, aaa->ccc, aaa->bbb, bbb->ccc, etc.
190    HRegionInfo parent = new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"), lastEndKey);
191    // Sleep a second else the encoded name on these regions comes out
192    // same for all with same start key and made in same second.
193    Thread.sleep(1001);
194
195    // Daughter a
196    HRegionInfo splita =
197        new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"), Bytes.toBytes("ccc"));
198    Thread.sleep(1001);
199    // Make daughters of daughter a; splitaa and splitab.
200    HRegionInfo splitaa =
201        new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"), Bytes.toBytes("bbb"));
202    HRegionInfo splitab =
203        new HRegionInfo(td.getTableName(), Bytes.toBytes("bbb"), Bytes.toBytes("ccc"));
204
205    // Daughter b
206    HRegionInfo splitb =
207        new HRegionInfo(td.getTableName(), Bytes.toBytes("ccc"), lastEndKey);
208    Thread.sleep(1001);
209    // Make Daughters of daughterb; splitba and splitbb.
210    HRegionInfo splitba =
211        new HRegionInfo(td.getTableName(), Bytes.toBytes("ccc"), Bytes.toBytes("ddd"));
212    HRegionInfo splitbb =
213        new HRegionInfo(td.getTableName(), Bytes.toBytes("ddd"), lastEndKey);
214
215    // First test that our Comparator works right up in CatalogJanitor.
216    SortedMap<HRegionInfo, Result> regions =
217        new TreeMap<>(new CatalogJanitor.SplitParentFirstComparator());
218    // Now make sure that this regions map sorts as we expect it to.
219    regions.put(parent, createResult(parent, splita, splitb));
220    regions.put(splitb, createResult(splitb, splitba, splitbb));
221    regions.put(splita, createResult(splita, splitaa, splitab));
222    // Assert its properly sorted.
223    int index = 0;
224    for (Map.Entry<HRegionInfo, Result> e: regions.entrySet()) {
225      if (index == 0) {
226        assertTrue(e.getKey().getEncodedName().equals(parent.getEncodedName()));
227      } else if (index == 1) {
228        assertTrue(e.getKey().getEncodedName().equals(splita.getEncodedName()));
229      } else if (index == 2) {
230        assertTrue(e.getKey().getEncodedName().equals(splitb.getEncodedName()));
231      }
232      index++;
233    }
234
235    // Now play around with the cleanParent function. Create a ref from splita up to the parent.
236    Path splitaRef =
237        createReferences(this.masterServices, td, parent, splita, Bytes.toBytes("ccc"), false);
238    // Make sure actual super parent sticks around because splita has a ref.
239    assertFalse(janitor.cleanParent(parent, regions.get(parent)));
240
241    //splitba, and split bb, do not have dirs in fs.  That means that if
242    // we test splitb, it should get cleaned up.
243    assertTrue(janitor.cleanParent(splitb, regions.get(splitb)));
244
245    // Now remove ref from splita to parent... so parent can be let go and so
246    // the daughter splita can be split (can't split if still references).
247    // BUT make the timing such that the daughter gets cleaned up before we
248    // can get a chance to let go of the parent.
249    FileSystem fs = FileSystem.get(HTU.getConfiguration());
250    assertTrue(fs.delete(splitaRef, true));
251    // Create the refs from daughters of splita.
252    Path splitaaRef =
253      createReferences(this.masterServices, td, splita, splitaa, Bytes.toBytes("bbb"), false);
254    Path splitabRef =
255      createReferences(this.masterServices, td, splita, splitab, Bytes.toBytes("bbb"), true);
256
257    // Test splita.  It should stick around because references from splitab, etc.
258    assertFalse(janitor.cleanParent(splita, regions.get(splita)));
259
260    // Now clean up parent daughter first.  Remove references from its daughters.
261    assertTrue(fs.delete(splitaaRef, true));
262    assertTrue(fs.delete(splitabRef, true));
263    assertTrue(janitor.cleanParent(splita, regions.get(splita)));
264
265    // Super parent should get cleaned up now both splita and splitb are gone.
266    assertTrue(janitor.cleanParent(parent, regions.get(parent)));
267  }

270   * CatalogJanitor.scan() should not clean parent regions if their own
271   * parents are still referencing them. This ensures that grandparent regions
272   * do not point to deleted parent regions.
273   */
274  @Test
275  public void testScanDoesNotCleanRegionsWithExistingParents() throws Exception {
276    TableDescriptor td = createTableDescriptorForCurrentMethod();
277    // Create regions: aaa->{lastEndKey}, aaa->ccc, aaa->bbb, bbb->ccc, etc.
278
279    // Parent
280    HRegionInfo parent = new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"),
281            HConstants.EMPTY_BYTE_ARRAY, true);
282    // Sleep a second else the encoded name on these regions comes out
283    // same for all with same start key and made in same second.
284    Thread.sleep(1001);
285
286    // Daughter a
287    HRegionInfo splita =
288        new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"), Bytes.toBytes("ccc"), true);
289    Thread.sleep(1001);
290
291    // Make daughters of daughter a; splitaa and splitab.
292    HRegionInfo splitaa =
293        new HRegionInfo(td.getTableName(), Bytes.toBytes("aaa"), Bytes.toBytes("bbb"), false);
294    HRegionInfo splitab =
295        new HRegionInfo(td.getTableName(), Bytes.toBytes("bbb"), Bytes.toBytes("ccc"), false);
296
297    // Daughter b
298    HRegionInfo splitb =
299        new HRegionInfo(td.getTableName(), Bytes.toBytes("ccc"), HConstants.EMPTY_BYTE_ARRAY);
300    Thread.sleep(1001);
301
302    // Parent has daughters splita and splitb. Splita has daughters splitaa and splitab.
303    final Map<HRegionInfo, Result> splitParents = new TreeMap<>(new SplitParentFirstComparator());
304    splitParents.put(parent, createResult(parent, splita, splitb));
305    splita.setOffline(true); //simulate that splita goes offline when it is split
306    splitParents.put(splita, createResult(splita, splitaa, splitab));
307
308    final Map<HRegionInfo, Result> mergedRegions = new TreeMap<>();
309    CatalogJanitor spy = spy(this.janitor);
310    doReturn(new Triple<>(10, mergedRegions, splitParents)).when(spy).
311      getMergedRegionsAndSplitParents();
312
313    // Create ref from splita to parent
314    LOG.info("parent=" + parent.getShortNameToLog() + ", splita=" + splita.getShortNameToLog());
315    Path splitaRef =
316        createReferences(this.masterServices, td, parent, splita, Bytes.toBytes("ccc"), false);
317    LOG.info("Created reference " + splitaRef);
318
319    // Parent and splita should not be removed because a reference from splita to parent.
320    assertEquals(0, spy.scan());
321
322    // Now delete the ref
323    FileSystem fs = FileSystem.get(HTU.getConfiguration());
324    assertTrue(fs.delete(splitaRef, true));
325
326    //now, both parent, and splita can be deleted
327    assertEquals(2, spy.scan());
328  }

331   * Test that we correctly archive all the storefiles when a region is deleted
332   * @throws Exception
333   */
334  @Test
335  public void testSplitParentFirstComparator() {
336    SplitParentFirstComparator comp = new SplitParentFirstComparator();
337    TableDescriptor td = createTableDescriptorForCurrentMethod();
338
339    /*  Region splits:
340     *
341     *  rootRegion --- firstRegion --- firstRegiona
342     *              |               |- firstRegionb
343     *              |
344     *              |- lastRegion --- lastRegiona  --- lastRegionaa
345     *                             |                |- lastRegionab
346     *                             |- lastRegionb
347     *
348     *  rootRegion   :   []  - []
349     *  firstRegion  :   []  - bbb
350     *  lastRegion   :   bbb - []
351     *  firstRegiona :   []  - aaa
352     *  firstRegionb :   aaa - bbb
353     *  lastRegiona  :   bbb - ddd
354     *  lastRegionb  :   ddd - []
355     */
356
357    // root region
358    HRegionInfo rootRegion = new HRegionInfo(td.getTableName(),
359      HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW, true);
360    HRegionInfo firstRegion = new HRegionInfo(td.getTableName(),
361      HConstants.EMPTY_START_ROW, Bytes.toBytes("bbb"), true);
362    HRegionInfo lastRegion = new HRegionInfo(td.getTableName(),
363      Bytes.toBytes("bbb"), HConstants.EMPTY_END_ROW, true);
364
365    assertTrue(comp.compare(rootRegion, rootRegion) == 0);
366    assertTrue(comp.compare(firstRegion, firstRegion) == 0);
367    assertTrue(comp.compare(lastRegion, lastRegion) == 0);
368    assertTrue(comp.compare(rootRegion, firstRegion) < 0);
369    assertTrue(comp.compare(rootRegion, lastRegion) < 0);
370    assertTrue(comp.compare(firstRegion, lastRegion) < 0);
371
372    //first region split into a, b
373    HRegionInfo firstRegiona = new HRegionInfo(td.getTableName(),
374      HConstants.EMPTY_START_ROW, Bytes.toBytes("aaa"), true);
375    HRegionInfo firstRegionb = new HRegionInfo(td.getTableName(),
376        Bytes.toBytes("aaa"), Bytes.toBytes("bbb"), true);
377    //last region split into a, b
378    HRegionInfo lastRegiona = new HRegionInfo(td.getTableName(),
379      Bytes.toBytes("bbb"), Bytes.toBytes("ddd"), true);
380    HRegionInfo lastRegionb = new HRegionInfo(td.getTableName(),
381      Bytes.toBytes("ddd"), HConstants.EMPTY_END_ROW, true);
382
383    assertTrue(comp.compare(firstRegiona, firstRegiona) == 0);
384    assertTrue(comp.compare(firstRegionb, firstRegionb) == 0);
385    assertTrue(comp.compare(rootRegion, firstRegiona) < 0);
386    assertTrue(comp.compare(rootRegion, firstRegionb) < 0);
387    assertTrue(comp.compare(firstRegion, firstRegiona) < 0);
388    assertTrue(comp.compare(firstRegion, firstRegionb) < 0);
389    assertTrue(comp.compare(firstRegiona, firstRegionb) < 0);
390
391    assertTrue(comp.compare(lastRegiona, lastRegiona) == 0);
392    assertTrue(comp.compare(lastRegionb, lastRegionb) == 0);
393    assertTrue(comp.compare(rootRegion, lastRegiona) < 0);
394    assertTrue(comp.compare(rootRegion, lastRegionb) < 0);
395    assertTrue(comp.compare(lastRegion, lastRegiona) < 0);
396    assertTrue(comp.compare(lastRegion, lastRegionb) < 0);
397    assertTrue(comp.compare(lastRegiona, lastRegionb) < 0);
398
399    assertTrue(comp.compare(firstRegiona, lastRegiona) < 0);
400    assertTrue(comp.compare(firstRegiona, lastRegionb) < 0);
401    assertTrue(comp.compare(firstRegionb, lastRegiona) < 0);
402    assertTrue(comp.compare(firstRegionb, lastRegionb) < 0);
403
404    HRegionInfo lastRegionaa = new HRegionInfo(td.getTableName(),
405      Bytes.toBytes("bbb"), Bytes.toBytes("ccc"), false);
406    HRegionInfo lastRegionab = new HRegionInfo(td.getTableName(),
407      Bytes.toBytes("ccc"), Bytes.toBytes("ddd"), false);
408
409    assertTrue(comp.compare(lastRegiona, lastRegionaa) < 0);
410    assertTrue(comp.compare(lastRegiona, lastRegionab) < 0);
411    assertTrue(comp.compare(lastRegionaa, lastRegionab) < 0);
412  }

418    HRegionInfo parent = new HRegionInfo(td.getTableName(),
419        Bytes.toBytes("aaa"), Bytes.toBytes("eee"));

428    FileSystem fs = FileSystem.get(HTU.getConfiguration());
429    Path rootdir = this.masterServices.getMasterFileSystem().getRootDir();

435    Path storedir = HStore.getStoreHomedir(tabledir, parent, td.getColumnFamilies()[0].getName());
436    Path storeArchive =
437        HFileArchiveUtil.getStoreArchivePath(this.masterServices.getConfiguration(), parent,
438            tabledir, td.getColumnFamilies()[0].getName());
439    LOG.debug("Table dir:" + tabledir);
440    LOG.debug("Store dir:" + storedir);
441    LOG.debug("Store archive dir:" + storeArchive);
442
443    // add a couple of store files that we can check for
444    FileStatus[] mockFiles = addMockStoreFiles(2, this.masterServices, storedir);
445    // get the current store files for comparison
446    FileStatus[] storeFiles = fs.listStatus(storedir);
447    int index = 0;
448    for (FileStatus file : storeFiles) {
449      LOG.debug("Have store file:" + file.getPath());
450      assertEquals("Got unexpected store file", mockFiles[index].getPath(),
451        storeFiles[index].getPath());
452      index++;
453    }
454
455    // do the cleaning of the parent
456    assertTrue(janitor.cleanParent(parent, parentMetaRow));
457    Path parentDir = new Path(tabledir, parent.getEncodedName());
458    // Cleanup procedure runs async. Wait till it done.
459    ProcedureTestingUtility.waitAllProcedures(masterServices.getMasterProcedureExecutor());
460    assertTrue(!fs.exists(parentDir));
461    LOG.debug("Finished cleanup of parent region");
462
463    // and now check to make sure that the files have actually been archived
464    FileStatus[] archivedStoreFiles = fs.listStatus(storeArchive);
465    logFiles("archived files", storeFiles);
466    logFiles("archived files", archivedStoreFiles);
467
468    assertArchiveEqualToOriginal(storeFiles, archivedStoreFiles, fs);
469
470    // cleanup
471    FSUtils.delete(fs, rootdir, true);
472  }

536    addMockStoreFiles(2, this.masterServices, storedir);
537
538    // Do the cleaning of the parent
539    assertTrue(janitor.cleanParent(parent, r));
540    // Cleanup procedure runs async. Wait till it done.
541    ProcedureTestingUtility.waitAllProcedures(masterServices.getMasterProcedureExecutor());
542    assertTrue(!fs.exists(parentDir));
543
544    // and now check to make sure that the files have actually been archived
545    archivedStoreFiles = fs.listStatus(storeArchive);
546    assertArchiveEqualToOriginal(storeFiles, archivedStoreFiles, fs, true);
547  }

577  private Path createReferences(final MasterServices services,
578      final TableDescriptor td, final HRegionInfo parent,
579      final HRegionInfo daughter, final byte [] midkey, final boolean top)
580  throws IOException {
581    Path rootdir = services.getMasterFileSystem().getRootDir();
582    Path tabledir = FSUtils.getTableDir(rootdir, parent.getTable());
583    Path storedir = HStore.getStoreHomedir(tabledir, daughter,
584      td.getColumnFamilies()[0].getName());
585    Reference ref =
586      top? Reference.createTopReference(midkey): Reference.createBottomReference(midkey);
587    long now = System.currentTimeMillis();
588    // Reference name has this format: StoreFile#REF_NAME_PARSER
589    Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
590    FileSystem fs = services.getMasterFileSystem().getFileSystem();
591    ref.write(fs, p);
592    return p;
593  }
