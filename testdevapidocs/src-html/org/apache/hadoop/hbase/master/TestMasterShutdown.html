052@Category({MasterTests.class, LargeTests.class})
053public class TestMasterShutdown {
054  private static final Logger LOG = LoggerFactory.getLogger(TestMasterShutdown.class);
055
056  @ClassRule
057  public static final HBaseClassTestRule CLASS_RULE =
058      HBaseClassTestRule.forClass(TestMasterShutdown.class);

060  private HBaseTestingUtility htu;

062  @Before
063  public void shutdownCluster() throws IOException {
064    if (htu != null) {
065      // an extra check in case the test cluster was not terminated after HBaseClassTestRule's
066      // Timeout interrupted the test thread.
067      LOG.warn("found non-null TestingUtility -- previous test did not terminate cleanly.");
068      htu.shutdownMiniCluster();
069    }
070  }

072  /**
073   * Simple test of shutdown.
074   * <p>
075   * Starts with three masters.  Tells the active master to shutdown the cluster.
076   * Verifies that all masters are properly shutdown.
077   */
078  @Test
079  public void testMasterShutdown() throws Exception {
080    // Create config to use for this cluster
081    Configuration conf = HBaseConfiguration.create();
082
083    // Start the cluster
084    try {
085      htu = new HBaseTestingUtility(conf);
086      StartMiniClusterOption option = StartMiniClusterOption.builder()
087        .numMasters(3)
088        .numRegionServers(3)
089        .numDataNodes(3)
090        .build();
091      final MiniHBaseCluster cluster = htu.startMiniCluster(option);
092
093      // wait for all master thread to spawn and start their run loop.
094      final long thirtySeconds = TimeUnit.SECONDS.toMillis(30);
095      final long oneSecond = TimeUnit.SECONDS.toMillis(1);
096      assertNotEquals(-1, htu.waitFor(thirtySeconds, oneSecond, () -> {
097        final List<MasterThread> masterThreads = cluster.getMasterThreads();
098        return masterThreads != null
099          && masterThreads.size() >= 3
100          && masterThreads.stream().allMatch(Thread::isAlive);
101      }));
102
103      // find the active master
104      final HMaster active = cluster.getMaster();
105      assertNotNull(active);
106
107      // make sure the other two are backup masters
108      ClusterMetrics status = active.getClusterMetrics();
109      assertEquals(2, status.getBackupMasterNames().size());
110
111      // tell the active master to shutdown the cluster
112      active.shutdown();
113      assertNotEquals(-1, htu.waitFor(thirtySeconds, oneSecond,
114        () -> CollectionUtils.isEmpty(cluster.getLiveMasterThreads())));
115      assertNotEquals(-1, htu.waitFor(thirtySeconds, oneSecond,
116        () -> CollectionUtils.isEmpty(cluster.getLiveRegionServerThreads())));
117    } finally {
118      if (htu != null) {
119        htu.shutdownMiniCluster();
120        htu = null;
121      }
122    }
123  }

125  /**
126   * This test appears to be an intentional race between a thread that issues a shutdown RPC to the
127   * master, while the master is concurrently realizing it cannot initialize because there are no
128   * region servers available to it. The expected behavior is that master initialization is
129   * interruptable via said shutdown RPC.
130   */
131  @Test
132  public void testMasterShutdownBeforeStartingAnyRegionServer() throws Exception {
133    LocalHBaseCluster hbaseCluster = null;
134    try {
135      htu =  new HBaseTestingUtility(
136        createMasterShutdownBeforeStartingAnyRegionServerConfiguration());
137
138      // configure a cluster with
139      final StartMiniClusterOption options = StartMiniClusterOption.builder()
140        .numDataNodes(1)
141        .numMasters(1)
142        .numRegionServers(0)
143        .masterClass(HMaster.class)
144        .rsClass(MiniHBaseCluster.MiniHBaseClusterRegionServer.class)
145        .createRootDir(true)
146        .build();
147
148      // Can't simply `htu.startMiniCluster(options)` because that method waits for the master to
149      // start completely. However, this test's premise is that a partially started master should
150      // still respond to a shutdown RPC. So instead, we manage each component lifecycle
151      // independently.
152      // I think it's not worth refactoring HTU's helper methods just for this class.
153      htu.startMiniDFSCluster(options.getNumDataNodes());
154      htu.startMiniZKCluster(options.getNumZkServers());
155      htu.createRootDir();
156      hbaseCluster = new LocalHBaseCluster(htu.getConfiguration(), options.getNumMasters(),
157        options.getNumRegionServers(), options.getMasterClass(), options.getRsClass());
158      final MasterThread masterThread = hbaseCluster.getMasters().get(0);
159
160      final CompletableFuture<Void> shutdownFuture = CompletableFuture.runAsync(() -> {
161        // Switching to master registry exacerbated a race in the master bootstrap that can result
162        // in a lost shutdown command (HBASE-8422, HBASE-23836). The race is essentially because
163        // the server manager in HMaster is not initialized by the time shutdown() RPC (below) is
164        // made to the master. The suspected reason as to why it was uncommon before HBASE-18095
165        // is because the connection creation with ZK registry is so slow that by then the server
166        // manager is usually init'ed in time for the RPC to be made. For now, adding an explicit
167        // wait() in the test, waiting for the server manager to become available.
168        final long timeout = TimeUnit.MINUTES.toMillis(10);
169        assertNotEquals("timeout waiting for server manager to become available.",
170          -1, Waiter.waitFor(htu.getConfiguration(), timeout,
171            () -> masterThread.getMaster().getServerManager() != null));
172
173        // Master has come up far enough that we can terminate it without creating a zombie.
174        final long result = Waiter.waitFor(htu.getConfiguration(), timeout, 500, () -> {
175          final Configuration conf = createResponsiveZkConfig(htu.getConfiguration());
176          LOG.debug("Attempting to establish connection.");
177          final CompletableFuture<AsyncConnection> connFuture =
178            ConnectionFactory.createAsyncConnection(conf);
179          try (final AsyncConnection conn = connFuture.join()) {
180            LOG.debug("Sending shutdown RPC.");
181            try {
182              conn.getAdmin().shutdown().join();
183              LOG.debug("Shutdown RPC sent.");
184              return true;
185            } catch (CompletionException e) {
186              LOG.debug("Failure sending shutdown RPC.");
187            }
188          } catch (IOException|CompletionException e) {
189            LOG.debug("Failed to establish connection.");
190          } catch (Throwable e) {
191            LOG.info("Something unexpected happened.", e);
192          }
193          return false;
194        });
195        assertNotEquals("Failed to issue shutdown RPC after " + Duration.ofMillis(timeout),
196          -1, result);
197      });
198
199      masterThread.start();
200      shutdownFuture.join();
201      masterThread.join();
202    } finally {
203      if (hbaseCluster != null) {
204        hbaseCluster.shutdown();
205      }
206      if (htu != null) {
207        htu.shutdownMiniCluster();
208        htu = null;
209      }
210    }
211  }

213  /**
214   * Create a cluster configuration suitable for
215   * {@link #testMasterShutdownBeforeStartingAnyRegionServer()}.
216   */
217  private static Configuration createMasterShutdownBeforeStartingAnyRegionServerConfiguration() {
218    final Configuration conf = HBaseConfiguration.create();
219    // make sure the master will wait forever in the absence of a RS.
220    conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, 1);
221    // don't need a long write pipeline for this test.
222    conf.setInt("dfs.replication", 1);
223    return conf;
224  }

226  /**
227   * Create a new {@link Configuration} based on {@code baseConf} that has ZooKeeper connection
228   * settings tuned very aggressively. The resulting client is used within a retry loop, so there's
229   * no value in having the client itself do the retries. We want to iterate on the base
230   * configuration because we're waiting for the mini-cluster to start and set it's ZK client port.
231   *
232   * @return a new, configured {@link Configuration} instance.
233   */
234  private static Configuration createResponsiveZkConfig(final Configuration baseConf) {
235    final Configuration conf = HBaseConfiguration.create(baseConf);
236    conf.setInt(ReadOnlyZKClient.RECOVERY_RETRY, 3);
237    conf.setInt(ReadOnlyZKClient.RECOVERY_RETRY_INTERVAL_MILLIS, 100);
238    return conf;
239  }
