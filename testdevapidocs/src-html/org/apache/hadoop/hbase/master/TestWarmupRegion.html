050/**
051 * Run tests that use the HBase clients; {@link org.apache.hadoop.hbase.client.HTable}.
052 * Sets up the HBase mini cluster once at start and runs through all client tests.
053 * Each creates a table named for the method and does its stuff against that.
054 */
055@Category({MasterTests.class, LargeTests.class})
056@SuppressWarnings ("deprecation")
057public class TestWarmupRegion {
058  private static final Logger LOG = LoggerFactory.getLogger(TestWarmupRegion.class);
059  protected TableName TABLENAME = TableName.valueOf("testPurgeFutureDeletes");
060  protected final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
061  private static byte [] ROW = Bytes.toBytes("testRow");
062  private static byte [] FAMILY = Bytes.toBytes("testFamily");
063  private static byte [] QUALIFIER = Bytes.toBytes("testQualifier");
064  private static byte [] VALUE = Bytes.toBytes("testValue");
065  private static byte[] COLUMN = Bytes.toBytes("column");
066  private static int numRows = 10000;
067  protected static int SLAVES = 3;
068  private static MiniHBaseCluster myCluster;
069  private static Table table;
070
071  /**
072   * @throws java.lang.Exception
073   */
074  @BeforeClass
075  public static void setUpBeforeClass() throws Exception {
076    Configuration conf = TEST_UTIL.getConfiguration();
077    TEST_UTIL.startMiniCluster(SLAVES);
078  }
079
080  /**
081   * @throws java.lang.Exception
082   */
083  @AfterClass
084  public static void tearDownAfterClass() throws Exception {
085    TEST_UTIL.shutdownMiniCluster();
086  }
087
088  /**
089   * @throws java.lang.Exception
090   */
091  @Before
092  public void setUp() throws Exception {
093    table = TEST_UTIL.createTable(TABLENAME, FAMILY);
094
095    // future timestamp
096    for (int i = 0; i < numRows; i++) {
097      long ts = System.currentTimeMillis() * 2;
098      Put put = new Put(ROW, ts);
099      put.addColumn(FAMILY, COLUMN, VALUE);
100      table.put(put);
101    }
102
103    // major compaction, purged future deletes
104    TEST_UTIL.getAdmin().flush(TABLENAME);
105    TEST_UTIL.getAdmin().majorCompact(TABLENAME);
106
107    // waiting for the major compaction to complete
108    TEST_UTIL.waitFor(6000, new Waiter.Predicate<IOException>() {
109      @Override
110      public boolean evaluate() throws IOException {
111        return TEST_UTIL.getAdmin().getCompactionState(TABLENAME) ==
112            CompactionState.NONE;
113      }
114    });
115
116    table.close();
117  }
118
119
120  /**
121   * @throws java.lang.Exception
122   */
123  @After
124  public void tearDown() throws Exception {
125    // Nothing to do.
126  }
127
128  protected void runwarmup()  throws InterruptedException{
129    Thread thread = new Thread(new Runnable() {
130      @Override
131      public void run() {
132        HRegionServer rs = TEST_UTIL.getMiniHBaseCluster().getRegionServer(0);
133        HRegion region = TEST_UTIL.getMiniHBaseCluster().getRegions(TABLENAME).get(0);
134        RegionInfo info = region.getRegionInfo();
135
136        try {
137          HTableDescriptor htd = table.getTableDescriptor();
138          for (int i = 0; i < 10; i++) {
139            warmupHRegion(info, htd, rs.getWAL(info), rs.getConfiguration(), rs, null);
140          }
141
142        } catch (IOException ie) {
143          LOG.error("Failed warming up region " + info.getRegionNameAsString(), ie);
144        }
145      }
146    });
147    thread.start();
148    thread.join();
149  }
150
151  /**
152   * Basic client side validation of HBASE-4536
153   */
154   @Test
155   public void testWarmup() throws Exception {
156     int serverid = 0;
157     HRegion region = TEST_UTIL.getMiniHBaseCluster().getRegions(TABLENAME).get(0);
158     RegionInfo info = region.getRegionInfo();
159     runwarmup();
160     for (int i = 0; i < 10; i++) {
161       HRegionServer rs = TEST_UTIL.getMiniHBaseCluster().getRegionServer(serverid);
162       byte [] destName = Bytes.toBytes(rs.getServerName().toString());
163       assertTrue(destName != null);
164       LOG.info("i=" + i );
165       TEST_UTIL.getMiniHBaseCluster().getMaster().move(info.getEncodedNameAsBytes(), destName);
166       serverid = (serverid + 1) % 2;
167     }
168   }
169}
