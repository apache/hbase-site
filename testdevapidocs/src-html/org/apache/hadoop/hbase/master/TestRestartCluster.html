077  @Test
078  public void testClusterRestartFailOver() throws Exception {
079    UTIL.startMiniCluster(3);
080    UTIL.waitFor(60000, () -> UTIL.getMiniHBaseCluster().getMaster().isInitialized());
081    //wait for all SCPs finished
082    UTIL.waitFor(20000, () -> UTIL.getHBaseCluster().getMaster().getProcedures().stream()
083        .noneMatch(p -> p instanceof ServerCrashProcedure));
084    TableName tableName = TABLES[0];
085    ServerName testServer = UTIL.getHBaseCluster().getRegionServer(0).getServerName();
086    ServerStateNode serverNode = UTIL.getHBaseCluster().getMaster().getAssignmentManager()
087        .getRegionStates().getServerNode(testServer);
088    Assert.assertNotNull(serverNode);
089    Assert.assertTrue("serverNode should be ONLINE when cluster runs normally",
090      serverNode.isInState(ServerState.ONLINE));
091    UTIL.createMultiRegionTable(tableName, FAMILY);
092    UTIL.waitTableEnabled(tableName);
093    Table table = UTIL.getConnection().getTable(tableName);
094    for (int i = 0; i < 100; i++) {
095      UTIL.loadTable(table, FAMILY);
096    }
097    List<Integer> ports =
098        UTIL.getHBaseCluster().getMaster().getServerManager().getOnlineServersList().stream()
099            .map(serverName -> serverName.getPort()).collect(Collectors.toList());
100    LOG.info("Shutting down cluster");
101    UTIL.getHBaseCluster().killAll();
102    UTIL.getHBaseCluster().waitUntilShutDown();
103    LOG.info("Starting cluster the second time");
104    UTIL.restartHBaseCluster(3, ports);
105    UTIL.waitFor(10000, () -> UTIL.getHBaseCluster().getMaster().isInitialized());
106    serverNode = UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates()
107        .getServerNode(testServer);
108    Assert.assertNotNull("serverNode should not be null when restart whole cluster", serverNode);
109    Assert.assertFalse(serverNode.isInState(ServerState.ONLINE));
110    LOG.info("start to find the procedure of SCP for the severName we choose");
111    UTIL.waitFor(20000,
112      () -> UTIL.getHBaseCluster().getMaster().getProcedures().stream()
113          .anyMatch(procedure -> (procedure instanceof ServerCrashProcedure)
114              && ((ServerCrashProcedure) procedure).getServerName().equals(testServer)));
115    Assert.assertFalse("serverNode should not be ONLINE during SCP processing",
116      serverNode.isInState(ServerState.ONLINE));
117    LOG.info("start to submit the SCP for the same serverName {} which should fail", testServer);
118    Assert.assertFalse(
119      UTIL.getHBaseCluster().getMaster().getServerManager().expireServer(testServer));
120    Procedure procedure = UTIL.getHBaseCluster().getMaster().getProcedures().stream()
121        .filter(p -> (p instanceof ServerCrashProcedure)
122            && ((ServerCrashProcedure) p).getServerName().equals(testServer))
123        .findAny().get();
124    UTIL.waitFor(20000, () -> procedure.isFinished());
125    LOG.info("even when the SCP is finished, the duplicate SCP should not be scheduled for {}",
126      testServer);
127    Assert.assertFalse(
128      UTIL.getHBaseCluster().getMaster().getServerManager().expireServer(testServer));
129    serverNode = UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates()
130        .getServerNode(testServer);
131    Assert.assertNull("serverNode should be deleted after SCP finished", serverNode);
132  }

134  @Test
135  public void testClusterRestart() throws Exception {
136    UTIL.startMiniCluster(3);
137    while (!UTIL.getMiniHBaseCluster().getMaster().isInitialized()) {
138      Threads.sleep(1);
139    }
140    LOG.info("\n\nCreating tables");
141    for(TableName TABLE : TABLES) {
142      UTIL.createTable(TABLE, FAMILY);
143    }
144    for(TableName TABLE : TABLES) {
145      UTIL.waitTableEnabled(TABLE);
146    }
147
148    List<RegionInfo> allRegions = MetaTableAccessor.getAllRegions(UTIL.getConnection(), false);
149    assertEquals(3, allRegions.size());
150
151    LOG.info("\n\nShutting down cluster");
152    UTIL.shutdownMiniHBaseCluster();
153
154    LOG.info("\n\nSleeping a bit");
155    Thread.sleep(2000);
156
157    LOG.info("\n\nStarting cluster the second time");
158    UTIL.restartHBaseCluster(3);
159
160    // Need to use a new 'Configuration' so we make a new Connection.
161    // Otherwise we're reusing an Connection that has gone stale because
162    // the shutdown of the cluster also called shut of the connection.
163    allRegions = MetaTableAccessor.getAllRegions(UTIL.getConnection(), false);
164    assertEquals(3, allRegions.size());
165    LOG.info("\n\nWaiting for tables to be available");
166    for(TableName TABLE: TABLES) {
167      try {
168        UTIL.createTable(TABLE, FAMILY);
169        assertTrue("Able to create table that should already exist", false);
170      } catch(TableExistsException tee) {
171        LOG.info("Table already exists as expected");
172      }
173      UTIL.waitTableAvailable(TABLE);
174    }
175  }

178   * This tests retaining assignments on a cluster restart
179   */
180  @Test
181  public void testRetainAssignmentOnRestart() throws Exception {
182    UTIL.startMiniCluster(2);
183    // Turn off balancer
184    UTIL.getMiniHBaseCluster().getMaster().getMasterRpcServices().synchronousBalanceSwitch(false);
185    LOG.info("\n\nCreating tables");
186    for (TableName TABLE : TABLES) {
187      UTIL.createTable(TABLE, FAMILY);
188    }
189    for (TableName TABLE : TABLES) {
190      UTIL.waitTableEnabled(TABLE);
191    }
192
193    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
194    UTIL.waitUntilNoRegionsInTransition(120000);
195
196    // We don't have to use SnapshotOfRegionAssignmentFromMeta.
197    // We use it here because AM used to use it to load all user region placements
198    SnapshotOfRegionAssignmentFromMeta snapshot = new SnapshotOfRegionAssignmentFromMeta(
199      master.getConnection());
200    snapshot.initialize();
201    Map<RegionInfo, ServerName> regionToRegionServerMap
202      = snapshot.getRegionToRegionServerMap();
203
204    MiniHBaseCluster cluster = UTIL.getHBaseCluster();
205    List<JVMClusterUtil.RegionServerThread> threads = cluster.getLiveRegionServerThreads();
206    assertEquals(2, threads.size());
207    int[] rsPorts = new int[3];
208    for (int i = 0; i < 2; i++) {
209      rsPorts[i] = threads.get(i).getRegionServer().getServerName().getPort();
210    }
211    rsPorts[2] = cluster.getMaster().getServerName().getPort();
212    for (ServerName serverName: regionToRegionServerMap.values()) {
213      boolean found = false; // Test only, no need to optimize
214      for (int k = 0; k < 3 && !found; k++) {
215        found = serverName.getPort() == rsPorts[k];
216      }
217      assertTrue(found);
218    }
219
220    LOG.info("\n\nShutting down HBase cluster");
221    cluster.stopMaster(0);
222    cluster.shutdown();
223    cluster.waitUntilShutDown();
224
225    LOG.info("\n\nSleeping a bit");
226    Thread.sleep(2000);
227
228    LOG.info("\n\nStarting cluster the second time with the same ports");
229    try {
230      cluster.getConf().setInt(
231          ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, 3);
232      master = cluster.startMaster().getMaster();
233      for (int i = 0; i < 3; i++) {
234        cluster.getConf().setInt(HConstants.REGIONSERVER_PORT, rsPorts[i]);
235        cluster.startRegionServer();
236      }
237    } finally {
238      // Reset region server port so as not to conflict with other tests
239      cluster.getConf().setInt(HConstants.REGIONSERVER_PORT, 0);
240      cluster.getConf().setInt(
241        ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, 2);
242    }
243
244    // Make sure live regionservers are on the same host/port
245    List<ServerName> localServers = master.getServerManager().getOnlineServersList();
246    assertEquals(3, localServers.size());
247    for (int i = 0; i < 3; i++) {
248      boolean found = false;
249      for (ServerName serverName: localServers) {
250        if (serverName.getPort() == rsPorts[i]) {
251          found = true;
252          break;
253        }
254      }
255      assertTrue(found);
256    }
257
258    // Wait till master is initialized and all regions are assigned
259    for (TableName TABLE : TABLES) {
260      UTIL.waitTableAvailable(TABLE);
261    }
262
263    snapshot = new SnapshotOfRegionAssignmentFromMeta(master.getConnection());
264    snapshot.initialize();
265    Map<RegionInfo, ServerName> newRegionToRegionServerMap =
266      snapshot.getRegionToRegionServerMap();
267    assertEquals(regionToRegionServerMap.size(), newRegionToRegionServerMap.size());
268    for (Map.Entry<RegionInfo, ServerName> entry : newRegionToRegionServerMap.entrySet()) {
269      ServerName oldServer = regionToRegionServerMap.get(entry.getKey());
270      ServerName currentServer = entry.getValue();
271      LOG.info(
272        "Key=" + entry.getKey() + " oldServer=" + oldServer + ", currentServer=" + currentServer);
273      assertEquals(entry.getKey().toString(), oldServer.getAddress(), currentServer.getAddress());
274      assertNotEquals(oldServer.getStartcode(), currentServer.getStartcode());
275    }
276  }

278  @Test
279  public void testNewStartedRegionServerVersion() throws Exception {
280    UTIL.startMiniCluster(1);
281
282    // Start 3 new region server
283    Thread t = new Thread(() -> {
284      for (int i = 0; i < 3; i++) {
285        try {
286          JVMClusterUtil.RegionServerThread newRS = UTIL.getMiniHBaseCluster().startRegionServer();
287          newRS.waitForServerOnline();
288        } catch (IOException e) {
289          LOG.error("Failed to start a new RS", e);
290        }
291      }
292    });
293    t.start();
294
295    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
296    while (t.isAlive()) {
297      List<ServerName> serverNames = master.getServerManager().getOnlineServersList();
298      for (ServerName serverName : serverNames) {
299        assertNotEquals(0, master.getServerManager().getVersionNumber(serverName));
300      }
301      Thread.sleep(100);
302    }
303  }
