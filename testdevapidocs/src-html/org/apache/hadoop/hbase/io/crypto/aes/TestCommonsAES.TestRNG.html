043  // Validation for AES in CTR mode with a 128 bit key
044  // From NIST Special Publication 800-38A
045  @Test
046  public void testAESAlgorithm() throws Exception {
047    Configuration conf = HBaseConfiguration.create();
048    Cipher aes = Encryption.getCipher(conf, "AES");
049    assertEquals(aes.getKeyLength(), CommonsCryptoAES.KEY_LENGTH);
050    assertEquals(aes.getIvLength(), CommonsCryptoAES.IV_LENGTH);
051    Encryptor e = aes.getEncryptor();
052    e.setKey(new SecretKeySpec(Bytes.fromHex("2b7e151628aed2a6abf7158809cf4f3c"), "AES"));
053    e.setIv(Bytes.fromHex("f0f1f2f3f4f5f6f7f8f9fafbfcfdfeff"));
054
055    ByteArrayOutputStream out = new ByteArrayOutputStream();
056    OutputStream cout = e.createEncryptionStream(out);
057    cout.write(Bytes.fromHex("6bc1bee22e409f96e93d7e117393172a"));
058    cout.write(Bytes.fromHex("ae2d8a571e03ac9c9eb76fac45af8e51"));
059    cout.write(Bytes.fromHex("30c81c46a35ce411e5fbc1191a0a52ef"));
060    cout.write(Bytes.fromHex("f69f2445df4f9b17ad2b417be66c3710"));
061    cout.close();
062
063    ByteArrayInputStream in = new ByteArrayInputStream(out.toByteArray());
064    byte[] b = new byte[16];
065    IOUtils.readFully(in, b);
066    assertTrue("Failed #1", Bytes.equals(b, Bytes.fromHex("874d6191b620e3261bef6864990db6ce")));
067    IOUtils.readFully(in, b);
068    assertTrue("Failed #2", Bytes.equals(b, Bytes.fromHex("9806f66b7970fdff8617187bb9fffdff")));
069    IOUtils.readFully(in, b);
070    assertTrue("Failed #3", Bytes.equals(b, Bytes.fromHex("5ae4df3edbd5d35e5b4f09020db03eab")));
071    IOUtils.readFully(in, b);
072    assertTrue("Failed #4", Bytes.equals(b, Bytes.fromHex("1e031dda2fbe03d1792170a0f3009cee")));
073  }

075  @Test
076  public void testAlternateRNG() throws Exception {
077    Security.addProvider(new TestProvider());
078
079    Configuration conf = new Configuration();
080    conf.set(AES.RNG_ALGORITHM_KEY, "TestRNG");
081    conf.set(AES.RNG_PROVIDER_KEY, "TEST");
082    DefaultCipherProvider.getInstance().setConf(conf);
083
084    AES aes = new AES(DefaultCipherProvider.getInstance());
085    assertEquals("AES did not find alternate RNG", aes.getRNG().getAlgorithm(),
086      "TestRNG");
087  }
088
089  static class TestProvider extends Provider {
090    private static final long serialVersionUID = 1L;
091    public TestProvider() {
092      super("TEST", 1.0, "Test provider");
093      AccessController.doPrivileged(new PrivilegedAction<Object>() {
094        public Object run() {
095          put("SecureRandom.TestRNG", TestCommonsAES.class.getName() + "$TestRNG");
096          return null;
097        }
098      });
099    }
100  }
101
102  // Must be public for instantiation by the SecureRandom SPI
103  public static class TestRNG extends SecureRandomSpi {
104    private static final long serialVersionUID = 1L;
105    private SecureRandom rng;
106
107    public TestRNG() {
108      try {
109        rng = SecureRandom.getInstance("SHA1PRNG");
110      } catch (NoSuchAlgorithmException e) {
111        fail("Unable to create SecureRandom instance");
112      }
113    }
114
115    @Override
116    protected void engineSetSeed(byte[] seed) {
117      rng.setSeed(seed);
118    }
119
120    @Override
121    protected void engineNextBytes(byte[] bytes) {
122      rng.nextBytes(bytes);
123    }
124
125    @Override
126    protected byte[] engineGenerateSeed(int numBytes) {
127      return rng.generateSeed(numBytes);
128    }
129  }
