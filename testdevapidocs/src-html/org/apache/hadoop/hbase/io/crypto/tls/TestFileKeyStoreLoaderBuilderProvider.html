020import org.apache.hadoop.hbase.HBaseClassTestRule;
021import org.apache.hadoop.hbase.testclassification.SecurityTests;
022import org.apache.hadoop.hbase.testclassification.SmallTests;
023import org.junit.Assert;
024import org.junit.ClassRule;
025import org.junit.Test;
026import org.junit.experimental.categories.Category;
027
028/**
029 * This file has been copied from the Apache ZooKeeper project.
030 * @see <a href=
031 *      "https://github.com/apache/zookeeper/blob/master/zookeeper-server/src/test/java/org/apache/zookeeper/common/FileKeyStoreLoaderBuilderProviderTest.java">Base

034@Category({ SecurityTests.class, SmallTests.class })
035public class TestFileKeyStoreLoaderBuilderProvider {
036
037  @ClassRule
038  public static final HBaseClassTestRule CLASS_RULE =
039    HBaseClassTestRule.forClass(TestFileKeyStoreLoaderBuilderProvider.class);
040
041  @Test
042  public void testGetBuilderForJKSFileType() {
043    FileKeyStoreLoader.Builder<?> builder =
044      FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(KeyStoreFileType.JKS);
045    Assert.assertTrue(builder instanceof JKSFileLoader.Builder);
046  }
047
048  @Test
049  public void testGetBuilderForPEMFileType() {
050    FileKeyStoreLoader.Builder<?> builder =
051      FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(KeyStoreFileType.PEM);
052    Assert.assertTrue(builder instanceof PEMFileLoader.Builder);
053  }
054
055  @Test
056  public void testGetBuilderForPKCS12FileType() {
057    FileKeyStoreLoader.Builder<?> builder =
058      FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(KeyStoreFileType.PKCS12);
059    Assert.assertTrue(builder instanceof PKCS12FileLoader.Builder);
060  }
061
062  @Test(expected = NullPointerException.class)
063  public void testGetBuilderForNullFileType() {
064    FileKeyStoreLoaderBuilderProvider.getBuilderForKeyStoreFileType(null);
065  }
066}
