144    return a.get(RAND.nextInt(a.size()));

147  @Test
148  public void testBucketAllocator() throws BucketAllocatorException {
149    BucketAllocator mAllocator = cache.getAllocator();
150    /*
151     * Test the allocator first
152     */
153    final List<Integer> BLOCKSIZES = Arrays.asList(4 * 1024, 8 * 1024, 64 * 1024, 96 * 1024);
154
155    boolean full = false;
156    ArrayList<Long> allocations = new ArrayList<>();
157    // Fill the allocated extents by choosing a random blocksize. Continues selecting blocks until
158    // the cache is completely filled.
159    List<Integer> tmp = new ArrayList<>(BLOCKSIZES);
160    while (!full) {
161      Integer blockSize = null;
162      try {
163        blockSize = randFrom(tmp);
164        allocations.add(mAllocator.allocateBlock(blockSize));
165      } catch (CacheFullException cfe) {
166        tmp.remove(blockSize);
167        if (tmp.isEmpty()) full = true;
168      }
169    }
170
171    for (Integer blockSize : BLOCKSIZES) {
172      BucketSizeInfo bucketSizeInfo = mAllocator.roundUpToBucketSizeInfo(blockSize);
173      IndexStatistics indexStatistics = bucketSizeInfo.statistics();
174      assertEquals("unexpected freeCount for " + bucketSizeInfo, 0, indexStatistics.freeCount());
175    }
176
177    for (long offset : allocations) {
178      assertEquals(mAllocator.sizeOfAllocation(offset), mAllocator.freeBlock(offset));
179    }
180    assertEquals(0, mAllocator.getUsedSize());
181  }

199  // BucketCache.cacheBlock is async, it first adds block to ramCache and writeQueue, then writer
200  // threads will flush it to the bucket and put reference entry in backingMap.
201  private void cacheAndWaitUntilFlushedToBucket(BucketCache cache, BlockCacheKey cacheKey,
202      Cacheable block) throws InterruptedException {
203    cache.cacheBlock(cacheKey, block);
204    while (!cache.backingMap.containsKey(cacheKey)) {
205      Thread.sleep(100);
206    }
207  }

209  @Test
210  public void testMemoryLeak() throws Exception {
211    final BlockCacheKey cacheKey = new BlockCacheKey("dummy", 1L);
212    cacheAndWaitUntilFlushedToBucket(cache, cacheKey, new CacheTestUtils.ByteArrayCacheable(
213        new byte[10]));
214    long lockId = cache.backingMap.get(cacheKey).offset();
215    ReentrantReadWriteLock lock = cache.offsetLock.getLock(lockId);
216    lock.writeLock().lock();
217    Thread evictThread = new Thread("evict-block") {
218
219      @Override
220      public void run() {
221        cache.evictBlock(cacheKey);
222      }
223
224    };
225    evictThread.start();
226    cache.offsetLock.waitForWaiters(lockId, 1);
227    cache.blockEvicted(cacheKey, cache.backingMap.remove(cacheKey), true);
228    cacheAndWaitUntilFlushedToBucket(cache, cacheKey, new CacheTestUtils.ByteArrayCacheable(
229        new byte[10]));
230    lock.writeLock().unlock();
231    evictThread.join();
232    assertEquals(1L, cache.getBlockCount());
233    assertTrue(cache.getCurrentSize() > 0L);
234    assertTrue("We should have a block!", cache.iterator().hasNext());
235  }

237  @Test
238  public void testRetrieveFromFile() throws Exception {
239    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
240    Path testDir = TEST_UTIL.getDataTestDir();
241    TEST_UTIL.getTestFileSystem().mkdirs(testDir);
242
243    BucketCache bucketCache = new BucketCache("file:" + testDir + "/bucket.cache", capacitySize,
244        constructedBlockSize, constructedBlockSizes, writeThreads, writerQLen, testDir
245            + "/bucket.persistence");
246    long usedSize = bucketCache.getAllocator().getUsedSize();
247    assertTrue(usedSize == 0);
248
249    HFileBlockPair[] blocks = CacheTestUtils.generateHFileBlocks(constructedBlockSize, 1);
250    // Add blocks
251    for (HFileBlockPair block : blocks) {
252      bucketCache.cacheBlock(block.getBlockName(), block.getBlock());
253    }
254    for (HFileBlockPair block : blocks) {
255      cacheAndWaitUntilFlushedToBucket(bucketCache, block.getBlockName(), block.getBlock());
256    }
257    usedSize = bucketCache.getAllocator().getUsedSize();
258    assertTrue(usedSize != 0);
259    // persist cache to file
260    bucketCache.shutdown();
261
262    // restore cache from file
263    bucketCache = new BucketCache("file:" + testDir + "/bucket.cache", capacitySize,
264        constructedBlockSize, constructedBlockSizes, writeThreads, writerQLen, testDir
265            + "/bucket.persistence");
266    assertEquals(usedSize, bucketCache.getAllocator().getUsedSize());
267    // persist cache to file
268    bucketCache.shutdown();
269
270    // reconfig buckets sizes, the biggest bucket is small than constructedBlockSize (8k or 16k)
271    // so it can't restore cache from file
272    int[] smallBucketSizes = new int[] { 2 * 1024 + 1024, 4 * 1024 + 1024 };
273    bucketCache = new BucketCache("file:" + testDir + "/bucket.cache", capacitySize,
274        constructedBlockSize, smallBucketSizes, writeThreads,
275        writerQLen, testDir + "/bucket.persistence");
276    assertEquals(0, bucketCache.getAllocator().getUsedSize());
277    assertEquals(0, bucketCache.backingMap.size());
278
279    TEST_UTIL.cleanupTestDir();
280  }

376    Configuration conf = HBaseConfiguration.create();
377    checkConfigValues(conf, configMappings, expectedOutcomes);
378  }

380  private void checkConfigValues(Configuration conf, Map<String, float[]> configMap, boolean[] expectSuccess) throws IOException {
381    Set<String> configNames = configMap.keySet();
382    for (int i = 0; i < expectSuccess.length; i++) {
383      try {
384        for (String configName : configNames) {
385          conf.setFloat(configName, configMap.get(configName)[i]);
386        }
387        BucketCache cache = new BucketCache(ioEngineName, capacitySize, constructedBlockSize,
388            constructedBlockSizes, writeThreads, writerQLen, persistencePath, 100, conf);
389        assertTrue("Created BucketCache and expected it to succeed: " + expectSuccess[i] + ", but it actually was: " + !expectSuccess[i], expectSuccess[i]);
390      } catch (IllegalArgumentException e) {
391        assertFalse("Created BucketCache and expected it to succeed: " + expectSuccess[i] + ", but it actually was: " + !expectSuccess[i], expectSuccess[i]);
392      }
393    }
394  }

396  private void validateGetPartitionSize(BucketCache bucketCache, float partitionFactor, float minFactor) {
397    long expectedOutput = (long) Math.floor(bucketCache.getAllocator().getTotalSize() * partitionFactor * minFactor);
398    assertEquals(expectedOutput, bucketCache.getPartitionSize(partitionFactor));
399  }
