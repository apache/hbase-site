046@Category({IOTests.class, SmallTests.class})
047public class TestBucketWriterThread {
048  private BucketCache bc;
049  private BucketCache.WriterThread wt;
050  private BlockingQueue<RAMQueueEntry> q;
051  private Cacheable plainCacheable;
052  private BlockCacheKey plainKey;
053
054  /** A BucketCache that does not start its writer threads. */
055  private static class MockBucketCache extends BucketCache {
056
057    public MockBucketCache(String ioEngineName, long capacity, int blockSize, int[] bucketSizes,
058      int writerThreadNum, int writerQLen, String persistencePath, int ioErrorsTolerationDuration)
059      throws FileNotFoundException, IOException {
060      super(ioEngineName, capacity, blockSize, bucketSizes, writerThreadNum, writerQLen,
061        persistencePath, ioErrorsTolerationDuration, HBaseConfiguration.create());
062    }
063
064    @Override
065    protected void startWriterThreads() {
066      // intentional noop
067    }
068  }
069
070  /**
071   * Set up variables and get BucketCache and WriterThread into state where tests can  manually
072   * control the running of WriterThread and BucketCache is empty.
073   * @throws Exception
074   */
075  @Before
076  public void setUp() throws Exception {
077    // Arbitrary capacity.
078    final int capacity = 16;
079    // Run with one writer thread only. Means there will be one writer queue only too.  We depend
080    // on this in below.
081    final int writerThreadsCount = 1;
082    this.bc = new MockBucketCache("heap", capacity, 1, new int [] {1}, writerThreadsCount,
083      capacity, null, 100/*Tolerate ioerrors for 100ms*/);
084    assertEquals(writerThreadsCount, bc.writerThreads.length);
085    assertEquals(writerThreadsCount, bc.writerQueues.size());
086    // Get reference to our single WriterThread instance.
087    this.wt = bc.writerThreads[0];
088    this.q = bc.writerQueues.get(0);
089
090    wt.disableWriter();
091    this.plainKey = new BlockCacheKey("f", 0);
092    this.plainCacheable = Mockito.mock(Cacheable.class);
093
094    assertThat(bc.ramCache.isEmpty(), is(true));
095    assertTrue(q.isEmpty());
096  }
097
098  @After
099  public void tearDown() throws Exception {
100    if (this.bc != null) this.bc.shutdown();
101  }
102
103  /**
104   * Test non-error case just works.
105   * @throws FileNotFoundException
106   * @throws IOException
107   * @throws InterruptedException
108   */
109  @Test (timeout=30000)
110  public void testNonErrorCase() throws IOException, InterruptedException {
111    bc.cacheBlock(this.plainKey, this.plainCacheable);
112    doDrainOfOneEntry(this.bc, this.wt, this.q);
113  }
114
115  /**
116   * Pass through a too big entry and ensure it is cleared from queues and ramCache.
117   * Manually run the WriterThread.
118   * @throws InterruptedException
119   */
120  @Test
121  public void testTooBigEntry() throws InterruptedException {
122    Cacheable tooBigCacheable = Mockito.mock(Cacheable.class);
123    Mockito.when(tooBigCacheable.getSerializedLength()).thenReturn(Integer.MAX_VALUE);
124    this.bc.cacheBlock(this.plainKey, tooBigCacheable);
125    doDrainOfOneEntry(this.bc, this.wt, this.q);
126  }
127
128  /**
129   * Do IOE. Take the RAMQueueEntry that was on the queue, doctor it to throw exception, then
130   * put it back and process it.
131   * @throws IOException
132   * @throws InterruptedException
133   */
134  @SuppressWarnings("unchecked")
135  @Test (timeout=30000)
136  public void testIOE() throws IOException, InterruptedException {
137    this.bc.cacheBlock(this.plainKey, plainCacheable);
138    RAMQueueEntry rqe = q.remove();
139    RAMQueueEntry spiedRqe = Mockito.spy(rqe);
140    Mockito.doThrow(new IOException("Mocked!")).when(spiedRqe).
141      writeToCache((IOEngine)Mockito.any(), (BucketAllocator)Mockito.any(),
142        (UniqueIndexMap<Integer>)Mockito.any(), (AtomicLong)Mockito.any());
143    this.q.add(spiedRqe);
144    doDrainOfOneEntry(bc, wt, q);
145    // Cache disabled when ioes w/o ever healing.
146    assertTrue(!bc.isCacheEnabled());
147  }
148
149  /**
150   * Do Cache full exception
151   * @throws IOException
152   * @throws InterruptedException
153   */
154  @Test (timeout=30000)
155  public void testCacheFullException()
156      throws IOException, InterruptedException {
157    this.bc.cacheBlock(this.plainKey, plainCacheable);
158    RAMQueueEntry rqe = q.remove();
159    RAMQueueEntry spiedRqe = Mockito.spy(rqe);
160    final CacheFullException cfe = new CacheFullException(0, 0);
161    BucketEntry mockedBucketEntry = Mockito.mock(BucketEntry.class);
162    Mockito.doThrow(cfe).
163      doReturn(mockedBucketEntry).
164      when(spiedRqe).writeToCache((IOEngine)Mockito.any(), (BucketAllocator)Mockito.any(),
165        (UniqueIndexMap<Integer>)Mockito.any(), (AtomicLong)Mockito.any());
166    this.q.add(spiedRqe);
167    doDrainOfOneEntry(bc, wt, q);
168  }
169
170  private static void doDrainOfOneEntry(final BucketCache bc, final BucketCache.WriterThread wt,
171      final BlockingQueue<RAMQueueEntry> q)
172  throws InterruptedException {
173    List<RAMQueueEntry> rqes = BucketCache.getRAMQueueEntries(q, new ArrayList<>(1));
174    wt.doDrain(rqes);
175    assertTrue(q.isEmpty());
176    assertTrue(bc.ramCache.isEmpty());
177    assertEquals(0, bc.heapSize());
178  }
179}
