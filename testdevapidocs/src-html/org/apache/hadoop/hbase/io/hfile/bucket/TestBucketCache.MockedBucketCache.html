166  @Test
167  public void testBucketAllocator() throws BucketAllocatorException {
168    BucketAllocator mAllocator = cache.getAllocator();
169    /*
170     * Test the allocator first
171     */
172    final List<Integer> BLOCKSIZES = Arrays.asList(4 * 1024, 8 * 1024, 64 * 1024, 96 * 1024);
173
174    boolean full = false;
175    ArrayList<Long> allocations = new ArrayList<>();
176    // Fill the allocated extents by choosing a random blocksize. Continues selecting blocks until
177    // the cache is completely filled.
178    List<Integer> tmp = new ArrayList<>(BLOCKSIZES);
179    while (!full) {
180      Integer blockSize = null;
181      try {
182        blockSize = randFrom(tmp);
183        allocations.add(mAllocator.allocateBlock(blockSize));
184      } catch (CacheFullException cfe) {
185        tmp.remove(blockSize);
186        if (tmp.isEmpty()) full = true;
187      }
188    }
189
190    for (Integer blockSize : BLOCKSIZES) {
191      BucketSizeInfo bucketSizeInfo = mAllocator.roundUpToBucketSizeInfo(blockSize);
192      IndexStatistics indexStatistics = bucketSizeInfo.statistics();
193      assertEquals("unexpected freeCount for " + bucketSizeInfo, 0, indexStatistics.freeCount());
194    }
195
196    for (long offset : allocations) {
197      assertEquals(mAllocator.sizeOfAllocation(offset), mAllocator.freeBlock(offset));
198    }
199    assertEquals(0, mAllocator.getUsedSize());
200  }

218  public static void waitUntilFlushedToBucket(BucketCache cache, BlockCacheKey cacheKey)
219      throws InterruptedException {
220    while (!cache.backingMap.containsKey(cacheKey) || cache.ramCache.containsKey(cacheKey)) {
221      Thread.sleep(100);
222    }
223    Thread.sleep(1000);
224  }
225
226  public static void waitUntilAllFlushedToBucket(BucketCache cache) throws InterruptedException {
227    while (!cache.ramCache.isEmpty()) {
228      Thread.sleep(100);
229    }
230    Thread.sleep(1000);
231  }

233  // BucketCache.cacheBlock is async, it first adds block to ramCache and writeQueue, then writer
234  // threads will flush it to the bucket and put reference entry in backingMap.
235  private void cacheAndWaitUntilFlushedToBucket(BucketCache cache, BlockCacheKey cacheKey,
236      Cacheable block) throws InterruptedException {
237    cache.cacheBlock(cacheKey, block);
238    waitUntilFlushedToBucket(cache, cacheKey);
239  }

241  @Test
242  public void testMemoryLeak() throws Exception {
243    final BlockCacheKey cacheKey = new BlockCacheKey("dummy", 1L);
244    cacheAndWaitUntilFlushedToBucket(cache, cacheKey,
245      new CacheTestUtils.ByteArrayCacheable(new byte[10]));
246    long lockId = cache.backingMap.get(cacheKey).offset();
247    ReentrantReadWriteLock lock = cache.offsetLock.getLock(lockId);
248    lock.writeLock().lock();
249    Thread evictThread = new Thread("evict-block") {
250      @Override
251      public void run() {
252        cache.evictBlock(cacheKey);
253      }
254    };
255    evictThread.start();
256    cache.offsetLock.waitForWaiters(lockId, 1);
257    cache.blockEvicted(cacheKey, cache.backingMap.remove(cacheKey), true);
258    assertEquals(0, cache.getBlockCount());
259    cacheAndWaitUntilFlushedToBucket(cache, cacheKey,
260      new CacheTestUtils.ByteArrayCacheable(new byte[10]));
261    assertEquals(1, cache.getBlockCount());
262    lock.writeLock().unlock();
263    evictThread.join();
264    /**
265     * <pre>
266     * The asserts here before HBASE-21957 are:
267     * assertEquals(1L, cache.getBlockCount());
268     * assertTrue(cache.getCurrentSize() > 0L);
269     * assertTrue("We should have a block!", cache.iterator().hasNext());
270     *
271     * The asserts here after HBASE-21957 are:
272     * assertEquals(0, cache.getBlockCount());
273     * assertEquals(cache.getCurrentSize(), 0L);
274     *
275     * I think the asserts before HBASE-21957 is more reasonable,because
276     * {@link BucketCache#evictBlock} should only evict the {@link BucketEntry}
277     * it had seen, and newly added Block after the {@link BucketEntry}
278     * it had seen should not be evicted.
279     * </pre>
280     */
281    assertEquals(1L, cache.getBlockCount());
282    assertTrue(cache.getCurrentSize() > 0L);
283    assertTrue("We should have a block!", cache.iterator().hasNext());
284  }

323  private void testRetrievalUtils(Path testDir, String ioEngineName)
324          throws IOException, InterruptedException {
325    final String persistencePath = testDir + "/bucket.persistence";
326    BucketCache bucketCache = new BucketCache(ioEngineName, capacitySize, constructedBlockSize,
327            constructedBlockSizes, writeThreads, writerQLen, persistencePath);
328    try {
329      long usedSize = bucketCache.getAllocator().getUsedSize();
330      assertEquals(0, usedSize);
331      HFileBlockPair[] blocks = CacheTestUtils.generateHFileBlocks(constructedBlockSize, 1);
332      for (HFileBlockPair block : blocks) {
333        bucketCache.cacheBlock(block.getBlockName(), block.getBlock());
334      }
335      for (HFileBlockPair block : blocks) {
336        cacheAndWaitUntilFlushedToBucket(bucketCache, block.getBlockName(), block.getBlock());
337      }
338      usedSize = bucketCache.getAllocator().getUsedSize();
339      assertNotEquals(0, usedSize);
340      bucketCache.shutdown();
341      assertTrue(new File(persistencePath).exists());
342      bucketCache = new BucketCache(ioEngineName, capacitySize, constructedBlockSize,
343              constructedBlockSizes, writeThreads, writerQLen, persistencePath);
344      assertFalse(new File(persistencePath).exists());
345      assertEquals(usedSize, bucketCache.getAllocator().getUsedSize());
346    } finally {
347      bucketCache.shutdown();
348    }
349    assertTrue(new File(persistencePath).exists());
350  }

529  private void checkConfigValues(Configuration conf, Map<String, float[]> configMap, boolean[] expectSuccess) throws IOException {
530    Set<String> configNames = configMap.keySet();
531    for (int i = 0; i < expectSuccess.length; i++) {
532      try {
533        for (String configName : configNames) {
534          conf.setFloat(configName, configMap.get(configName)[i]);
535        }
536        BucketCache cache = new BucketCache(ioEngineName, capacitySize, constructedBlockSize,
537            constructedBlockSizes, writeThreads, writerQLen, null, 100, conf);
538        assertTrue("Created BucketCache and expected it to succeed: " + expectSuccess[i] + ", but it actually was: " + !expectSuccess[i], expectSuccess[i]);
539      } catch (IllegalArgumentException e) {
540        assertFalse("Created BucketCache and expected it to succeed: " + expectSuccess[i] + ", but it actually was: " + !expectSuccess[i], expectSuccess[i]);
541      }
542    }
543  }

695  /**
696   * This test is for HBASE-26295, {@link BucketEntry} which is restored from a persistence file
697   * could not be freed even if corresponding {@link HFileBlock} is evicted from
698   * {@link BucketCache}.
699   */
700  @Test
701  public void testFreeBucketEntryRestoredFromFile() throws Exception {
702    try {
703      final Path dataTestDir = createAndGetTestDir();
704
705      String ioEngineName = "file:" + dataTestDir + "/bucketNoRecycler.cache";
706      String persistencePath = dataTestDir + "/bucketNoRecycler.persistence";
707
708      BucketCache bucketCache = new BucketCache(ioEngineName, capacitySize, constructedBlockSize,
709          constructedBlockSizes, writeThreads, writerQLen, persistencePath);
710      long usedByteSize = bucketCache.getAllocator().getUsedSize();
711      assertEquals(0, usedByteSize);
712
713      HFileBlockPair[] hfileBlockPairs =
714          CacheTestUtils.generateHFileBlocks(constructedBlockSize, 1);
715      // Add blocks
716      for (HFileBlockPair hfileBlockPair : hfileBlockPairs) {
717        bucketCache.cacheBlock(hfileBlockPair.getBlockName(), hfileBlockPair.getBlock());
718      }
719
720      for (HFileBlockPair hfileBlockPair : hfileBlockPairs) {
721        cacheAndWaitUntilFlushedToBucket(bucketCache, hfileBlockPair.getBlockName(),
722          hfileBlockPair.getBlock());
723      }
724      usedByteSize = bucketCache.getAllocator().getUsedSize();
725      assertNotEquals(0, usedByteSize);
726      // persist cache to file
727      bucketCache.shutdown();
728      assertTrue(new File(persistencePath).exists());
729

732          constructedBlockSizes, writeThreads, writerQLen, persistencePath);

734      assertEquals(usedByteSize, bucketCache.getAllocator().getUsedSize());
735
736      for (HFileBlockPair hfileBlockPair : hfileBlockPairs) {
737        BlockCacheKey blockCacheKey = hfileBlockPair.getBlockName();
738        bucketCache.evictBlock(blockCacheKey);
739      }
740      assertEquals(0, bucketCache.getAllocator().getUsedSize());
741      assertEquals(0, bucketCache.backingMap.size());
742    } finally {
743      HBASE_TESTING_UTILITY.cleanupTestDir();
744    }
