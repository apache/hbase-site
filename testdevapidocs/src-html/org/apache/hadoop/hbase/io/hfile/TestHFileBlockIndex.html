152  /**
153   * A wrapper around a block reader which only caches the results of the last
154   * operation. Not thread-safe.
155   */
156  private static class BlockReaderWrapper implements HFile.CachingBlockReader {
157
158    private HFileBlock.FSReader realReader;
159    private long prevOffset;
160    private long prevOnDiskSize;
161    private boolean prevPread;
162    private HFileBlock prevBlock;
163
164    public int hitCount = 0;
165    public int missCount = 0;
166
167    public BlockReaderWrapper(HFileBlock.FSReader realReader) {
168      this.realReader = realReader;
169    }
170
171    @Override
172    public void returnBlock(HFileBlock block) {
173    }
174
175    @Override
176    public HFileBlock readBlock(long offset, long onDiskSize,
177        boolean cacheBlock, boolean pread, boolean isCompaction,
178        boolean updateCacheMetrics, BlockType expectedBlockType,
179        DataBlockEncoding expectedDataBlockEncoding)
180        throws IOException {
181      if (offset == prevOffset && onDiskSize == prevOnDiskSize &&
182          pread == prevPread) {
183        hitCount += 1;
184        return prevBlock;
185      }
186
187      missCount += 1;
188      prevBlock = realReader.readBlockData(offset, onDiskSize,
189          -1, pread);
190      prevOffset = offset;
191      prevOnDiskSize = onDiskSize;
192      prevPread = pread;
193
194      return prevBlock;
195    }
196  }

198  private void readIndex(boolean useTags) throws IOException {
199    long fileSize = fs.getFileStatus(path).getLen();
200    LOG.info("Size of " + path + ": " + fileSize);
201
202    FSDataInputStream istream = fs.open(path);
203    HFileContext meta = new HFileContextBuilder()
204                        .withHBaseCheckSum(true)
205                        .withIncludesMvcc(includesMemstoreTS)
206                        .withIncludesTags(useTags)
207                        .withCompression(compr)
208                        .build();
209    HFileBlock.FSReader blockReader = new HFileBlock.FSReaderImpl(istream, fs.getFileStatus(path)
210        .getLen(), meta);
211
212    BlockReaderWrapper brw = new BlockReaderWrapper(blockReader);
213    HFileBlockIndex.BlockIndexReader indexReader =
214        new HFileBlockIndex.CellBasedKeyBlockIndexReader(
215            CellComparator.COMPARATOR, numLevels, brw);
216
217    indexReader.readRootIndex(blockReader.blockRange(rootIndexOffset,
218        fileSize).nextBlockWithBlockType(BlockType.ROOT_INDEX), numRootEntries);
219
220    long prevOffset = -1;
221    int i = 0;
222    int expectedHitCount = 0;
223    int expectedMissCount = 0;
224    LOG.info("Total number of keys: " + keys.size());
225    for (byte[] key : keys) {
226      assertTrue(key != null);
227      assertTrue(indexReader != null);
228      KeyValue.KeyOnlyKeyValue keyOnlyKey = new KeyValue.KeyOnlyKeyValue(key, 0, key.length);
229      HFileBlock b =
230          indexReader.seekToDataBlock(keyOnlyKey, null, true,
231            true, false, null);
232      if (CellComparator.COMPARATOR.compare(keyOnlyKey, firstKeyInFile,
233          0, firstKeyInFile.length) < 0) {
234        assertTrue(b == null);
235        ++i;
236        continue;
237      }
238
239      String keyStr = "key #" + i + ", " + Bytes.toStringBinary(key);
240
241      assertTrue("seekToDataBlock failed for " + keyStr, b != null);
242
243      if (prevOffset == b.getOffset()) {
244        assertEquals(++expectedHitCount, brw.hitCount);
245      } else {
246        LOG.info("First key in a new block: " + keyStr + ", block offset: "
247            + b.getOffset() + ")");
248        assertTrue(b.getOffset() > prevOffset);
249        assertEquals(++expectedMissCount, brw.missCount);
250        prevOffset = b.getOffset();
251      }
252      ++i;
253    }
254
255    istream.close();
256  }

258  private void writeWholeIndex(boolean useTags) throws IOException {
259    assertEquals(0, keys.size());
260    HFileContext meta = new HFileContextBuilder()
261                        .withHBaseCheckSum(true)
262                        .withIncludesMvcc(includesMemstoreTS)
263                        .withIncludesTags(useTags)
264                        .withCompression(compr)
265                        .withBytesPerCheckSum(HFile.DEFAULT_BYTES_PER_CHECKSUM)
266                        .build();
267    HFileBlock.Writer hbw = new HFileBlock.Writer(null,
268        meta);
269    FSDataOutputStream outputStream = fs.create(path);
270    HFileBlockIndex.BlockIndexWriter biw =
271        new HFileBlockIndex.BlockIndexWriter(hbw, null, null);
272
273    for (int i = 0; i < NUM_DATA_BLOCKS; ++i) {
274      hbw.startWriting(BlockType.DATA).write(String.valueOf(rand.nextInt(1000)).getBytes());
275      long blockOffset = outputStream.getPos();
276      hbw.writeHeaderAndData(outputStream);
277
278      byte[] firstKey = null;
279      byte[] family = Bytes.toBytes("f");
280      byte[] qualifier = Bytes.toBytes("q");
281      for (int j = 0; j < 16; ++j) {
282        byte[] k =
283            new KeyValue(TestHFileWriterV2.randomOrderedKey(rand, i * 16 + j), family, qualifier,
284                EnvironmentEdgeManager.currentTime(), KeyValue.Type.Put).getKey();
285        keys.add(k);
286        if (j == 8) {
287          firstKey = k;
288        }
289      }
290      assertTrue(firstKey != null);
291      if (firstKeyInFile == null) {
292        firstKeyInFile = firstKey;
293      }
294      biw.addEntry(firstKey, blockOffset, hbw.getOnDiskSizeWithHeader());
295
296      writeInlineBlocks(hbw, outputStream, biw, false);
297    }
298    writeInlineBlocks(hbw, outputStream, biw, true);
299    rootIndexOffset = biw.writeIndexBlocks(outputStream);
300    outputStream.close();
301
302    numLevels = biw.getNumLevels();
303    numRootEntries = biw.getNumRootEntries();
304
305    LOG.info("Index written: numLevels=" + numLevels + ", numRootEntries=" +
306        numRootEntries + ", rootIndexOffset=" + rootIndexOffset);
307  }

309  private void writeInlineBlocks(HFileBlock.Writer hbw,
310      FSDataOutputStream outputStream, HFileBlockIndex.BlockIndexWriter biw,
311      boolean isClosing) throws IOException {
312    while (biw.shouldWriteBlock(isClosing)) {
313      long offset = outputStream.getPos();
314      biw.writeInlineBlock(hbw.startWriting(biw.getInlineBlockType()));
315      hbw.writeHeaderAndData(outputStream);
316      biw.blockWritten(offset, hbw.getOnDiskSizeWithHeader(),
317          hbw.getUncompressedSizeWithoutHeader());
318      LOG.info("Wrote an inline index block at " + offset + ", size " +
319          hbw.getOnDiskSizeWithHeader());
320    }
321  }

331  @Test
332  public void testSecondaryIndexBinarySearch() throws IOException {
333    int numTotalKeys = 99;
334    assertTrue(numTotalKeys % 2 == 1); // Ensure no one made this even.
335
336    // We only add odd-index keys into the array that we will binary-search.
337    int numSearchedKeys = (numTotalKeys - 1) / 2;
338
339    ByteArrayOutputStream baos = new ByteArrayOutputStream();
340    DataOutputStream dos = new DataOutputStream(baos);
341
342    dos.writeInt(numSearchedKeys);
343    int curAllEntriesSize = 0;
344    int numEntriesAdded = 0;
345
346    // Only odd-index elements of this array are used to keep the secondary
347    // index entries of the corresponding keys.
348    int secondaryIndexEntries[] = new int[numTotalKeys];
349
350    for (int i = 0; i < numTotalKeys; ++i) {
351      byte[] k = TestHFileWriterV2.randomOrderedKey(rand, i * 2);
352      KeyValue cell = new KeyValue(k, Bytes.toBytes("f"), Bytes.toBytes("q"), 
353          Bytes.toBytes("val"));
354      //KeyValue cell = new KeyValue.KeyOnlyKeyValue(k, 0, k.length);
355      keys.add(cell.getKey());
356      String msgPrefix = "Key #" + i + " (" + Bytes.toStringBinary(k) + "): ";
357      StringBuilder padding = new StringBuilder();
358      while (msgPrefix.length() + padding.length() < 70)
359        padding.append(' ');
360      msgPrefix += padding;
361      if (i % 2 == 1) {
362        dos.writeInt(curAllEntriesSize);
363        secondaryIndexEntries[i] = curAllEntriesSize;
364        LOG.info(msgPrefix + "secondary index entry #" + ((i - 1) / 2) +
365            ", offset " + curAllEntriesSize);
366        curAllEntriesSize += cell.getKey().length
367            + HFileBlockIndex.SECONDARY_INDEX_ENTRY_OVERHEAD;
368        ++numEntriesAdded;
369      } else {
370        secondaryIndexEntries[i] = -1;
371        LOG.info(msgPrefix + "not in the searched array");
372      }
373    }
374
375    // Make sure the keys are increasing.
376    for (int i = 0; i < keys.size() - 1; ++i)
377      assertTrue(CellComparator.COMPARATOR.compare(
378          new KeyValue.KeyOnlyKeyValue(keys.get(i), 0, keys.get(i).length),
379          new KeyValue.KeyOnlyKeyValue(keys.get(i + 1), 0, keys.get(i + 1).length)) < 0);
380
381    dos.writeInt(curAllEntriesSize);
382    assertEquals(numSearchedKeys, numEntriesAdded);
383    int secondaryIndexOffset = dos.size();
384    assertEquals(Bytes.SIZEOF_INT * (numSearchedKeys + 2),
385        secondaryIndexOffset);
386
387    for (int i = 1; i <= numTotalKeys - 1; i += 2) {
388      assertEquals(dos.size(),
389          secondaryIndexOffset + secondaryIndexEntries[i]);
390      long dummyFileOffset = getDummyFileOffset(i);
391      int dummyOnDiskSize = getDummyOnDiskSize(i);
392      LOG.debug("Storing file offset=" + dummyFileOffset + " and onDiskSize=" +
393          dummyOnDiskSize + " at offset " + dos.size());
394      dos.writeLong(dummyFileOffset);
395      dos.writeInt(dummyOnDiskSize);
396      LOG.debug("Stored key " + ((i - 1) / 2) +" at offset " + dos.size());
397      dos.write(keys.get(i));
398    }
399
400    dos.writeInt(curAllEntriesSize);
401
402    ByteBuffer nonRootIndex = ByteBuffer.wrap(baos.toByteArray());
403    for (int i = 0; i < numTotalKeys; ++i) {
404      byte[] searchKey = keys.get(i);
405      byte[] arrayHoldingKey = new byte[searchKey.length +
406                                        searchKey.length / 2];
407
408      // To make things a bit more interesting, store the key we are looking
409      // for at a non-zero offset in a new array.
410      System.arraycopy(searchKey, 0, arrayHoldingKey, searchKey.length / 2,
411            searchKey.length);
412
413      KeyValue.KeyOnlyKeyValue cell = new KeyValue.KeyOnlyKeyValue(
414          arrayHoldingKey, searchKey.length / 2, searchKey.length);
415      int searchResult = BlockIndexReader.binarySearchNonRootIndex(cell,
416          new MultiByteBuff(nonRootIndex), CellComparator.COMPARATOR);
417      String lookupFailureMsg = "Failed to look up key #" + i + " ("
418          + Bytes.toStringBinary(searchKey) + ")";
419
420      int expectedResult;
421      int referenceItem;
422
423      if (i % 2 == 1) {
424        // This key is in the array we search as the element (i - 1) / 2. Make
425        // sure we find it.
426        expectedResult = (i - 1) / 2;
427        referenceItem = i;
428      } else {
429        // This key is not in the array but between two elements on the array,
430        // in the beginning, or in the end. The result should be the previous
431        // key in the searched array, or -1 for i = 0.
432        expectedResult = i / 2 - 1;
433        referenceItem = i - 1;
434      }
435
436      assertEquals(lookupFailureMsg, expectedResult, searchResult);
437
438      // Now test we can get the offset and the on-disk-size using a
439      // higher-level API function.s
440      boolean locateBlockResult =
441          (BlockIndexReader.locateNonRootIndexEntry(new MultiByteBuff(nonRootIndex), cell,
442          CellComparator.COMPARATOR) != -1);
443
444      if (i == 0) {
445        assertFalse(locateBlockResult);
446      } else {
447        assertTrue(locateBlockResult);
448        String errorMsg = "i=" + i + ", position=" + nonRootIndex.position();
449        assertEquals(errorMsg, getDummyFileOffset(referenceItem),
450            nonRootIndex.getLong());
451        assertEquals(errorMsg, getDummyOnDiskSize(referenceItem),
452            nonRootIndex.getInt());
453      }
454    }

459  public void testBlockIndexChunk() throws IOException {
460    BlockIndexChunk c = new BlockIndexChunk();
461    ByteArrayOutputStream baos = new ByteArrayOutputStream();
462    int N = 1000;
463    int[] numSubEntriesAt = new int[N];
464    int numSubEntries = 0;
465    for (int i = 0; i < N; ++i) {
466      baos.reset();
467      DataOutputStream dos = new DataOutputStream(baos);
468      c.writeNonRoot(dos);
469      assertEquals(c.getNonRootSize(), dos.size());
470
471      baos.reset();
472      dos = new DataOutputStream(baos);
473      c.writeRoot(dos);
474      assertEquals(c.getRootSize(), dos.size());
475
476      byte[] k = TestHFileWriterV2.randomOrderedKey(rand, i);
477      numSubEntries += rand.nextInt(5) + 1;
478      keys.add(k);
479      c.add(k, getDummyFileOffset(i), getDummyOnDiskSize(i), numSubEntries);
480    }
481
482    // Test the ability to look up the entry that contains a particular
483    // deeper-level index block's entry ("sub-entry"), assuming a global
484    // 0-based ordering of sub-entries. This is needed for mid-key calculation.
485    for (int i = 0; i < N; ++i) {
486      for (int j = i == 0 ? 0 : numSubEntriesAt[i - 1];
487           j < numSubEntriesAt[i];
488           ++j) {
489        assertEquals(i, c.getEntryBySubEntry(j));
490      }
491    }

495  @Test
496  public void testHeapSizeForBlockIndex() throws IOException {
497    Class<HFileBlockIndex.BlockIndexReader> cl =
498        HFileBlockIndex.BlockIndexReader.class;
499    long expected = ClassSize.estimateBase(cl, false);
500
501    HFileBlockIndex.BlockIndexReader bi =

504

506    // int [] blockDataSizes) are all null they are not going to show up in the
507    // HeapSize calculation, so need to remove those array costs from expected.

512      expected = ClassSize.estimateBase(cl, true);
513      assertEquals(expected, actual);
514    }

537      blockCache.evictBlocksByHfileName(hfilePath.getName());
538
539      conf.setInt(HFileBlockIndex.MAX_CHUNK_SIZE_KEY, indexBlockSize);
540      Set<String> keyStrSet = new HashSet<String>();
541      byte[][] keys = new byte[NUM_KV][];
542      byte[][] values = new byte[NUM_KV][];
543
544      // Write the HFile
545      {
546        HFileContext meta = new HFileContextBuilder()
547                            .withBlockSize(SMALL_BLOCK_SIZE)
548                            .withCompression(compr)
549                            .build();
550        HFile.Writer writer =
551            HFile.getWriterFactory(conf, cacheConf)
552                .withPath(fs, hfilePath)
553                .withFileContext(meta)
554                .create();
555        Random rand = new Random(19231737);
556        byte[] family = Bytes.toBytes("f");
557        byte[] qualifier = Bytes.toBytes("q");
558        for (int i = 0; i < NUM_KV; ++i) {
559          byte[] row = TestHFileWriterV2.randomOrderedKey(rand, i);
560
561          // Key will be interpreted by KeyValue.KEY_COMPARATOR
562          KeyValue kv =
563              new KeyValue(row, family, qualifier, EnvironmentEdgeManager.currentTime(),
564                  TestHFileWriterV2.randomValue(rand));
565          byte[] k = kv.getKey();
566          writer.append(kv);
567          keys[i] = k;
568          values[i] = CellUtil.cloneValue(kv);
569          keyStrSet.add(Bytes.toStringBinary(k));
570          if (i > 0) {
571            assertTrue((CellComparator.COMPARATOR.compare(kv, keys[i - 1],
572                0, keys[i - 1].length)) > 0);
573          }
574        }
575
576        writer.close();
577      }
578
579      // Read the HFile
580      HFile.Reader reader = HFile.createReader(fs, hfilePath, cacheConf, conf);
581      assertEquals(expectedNumLevels,
582          reader.getTrailer().getNumDataIndexLevels());
583
584      assertTrue(Bytes.equals(keys[0], ((KeyValue)reader.getFirstKey()).getKey()));
585      assertTrue(Bytes.equals(keys[NUM_KV - 1], ((KeyValue)reader.getLastKey()).getKey()));
586      LOG.info("Last key: " + Bytes.toStringBinary(keys[NUM_KV - 1]));
587
588      for (boolean pread : new boolean[] { false, true }) {
589        HFileScanner scanner = reader.getScanner(true, pread);
590        for (int i = 0; i < NUM_KV; ++i) {
591          checkSeekTo(keys, scanner, i);
592          checkKeyValue("i=" + i, keys[i], values[i],
593              ByteBuffer.wrap(((KeyValue) scanner.getKey()).getKey()), scanner.getValue());
594        }
595        assertTrue(scanner.seekTo());
596        for (int i = NUM_KV - 1; i >= 0; --i) {
597          checkSeekTo(keys, scanner, i);
598          checkKeyValue("i=" + i, keys[i], values[i],
599              ByteBuffer.wrap(((KeyValue) scanner.getKey()).getKey()), scanner.getValue());
600        }
601      }
602
603      // Manually compute the mid-key and validate it.
604      HFile.Reader reader2 = reader;
605      HFileBlock.FSReader fsReader = reader2.getUncachedBlockReader();
606
607      HFileBlock.BlockIterator iter = fsReader.blockRange(0,
608          reader.getTrailer().getLoadOnOpenDataOffset());
609      HFileBlock block;
610      List<byte[]> blockKeys = new ArrayList<byte[]>();
611      while ((block = iter.nextBlock()) != null) {
612        if (block.getBlockType() != BlockType.LEAF_INDEX)
613          return;
614        ByteBuff b = block.getBufferReadOnly();
615        int n = b.getIntAfterPosition(0);
616        // One int for the number of items, and n + 1 for the secondary index.
617        int entriesOffset = Bytes.SIZEOF_INT * (n + 2);
618
619        // Get all the keys from the leaf index block. S
620        for (int i = 0; i < n; ++i) {
621          int keyRelOffset = b.getIntAfterPosition(Bytes.SIZEOF_INT * (i + 1));
622          int nextKeyRelOffset = b.getIntAfterPosition(Bytes.SIZEOF_INT * (i + 2));
623          int keyLen = nextKeyRelOffset - keyRelOffset;
624          int keyOffset = b.arrayOffset() + entriesOffset + keyRelOffset +
625              HFileBlockIndex.SECONDARY_INDEX_ENTRY_OVERHEAD;
626          byte[] blockKey = Arrays.copyOfRange(b.array(), keyOffset, keyOffset
627              + keyLen);
628          String blockKeyStr = Bytes.toString(blockKey);
629          blockKeys.add(blockKey);
630
631          // If the first key of the block is not among the keys written, we
632          // are not parsing the non-root index block format correctly.
633          assertTrue("Invalid block key from leaf-level block: " + blockKeyStr,
634              keyStrSet.contains(blockKeyStr));
635        }
636      }
637
638      // Validate the mid-key.
639      assertEquals(
640          Bytes.toStringBinary(blockKeys.get((blockKeys.size() - 1) / 2)),
641          reader.midkey());
642
643      assertEquals(UNCOMPRESSED_INDEX_SIZES[testI],
644          reader.getTrailer().getUncompressedDataIndexSize());
645
646      reader.close();
647      reader2.close();
648    }
649  }

657  private void assertArrayEqualsBuffer(String msgPrefix, byte[] arr,
658      ByteBuffer buf) {
659    assertEquals(msgPrefix + ": expected " + Bytes.toStringBinary(arr)
660        + ", actual " + Bytes.toStringBinary(buf), 0, Bytes.compareTo(arr, 0,
661        arr.length, buf.array(), buf.arrayOffset(), buf.limit()));
662  }

664  /** Check a key/value pair after it was read by the reader */
665  private void checkKeyValue(String msgPrefix, byte[] expectedKey,
666      byte[] expectedValue, ByteBuffer keyRead, ByteBuffer valueRead) {
667    if (!msgPrefix.isEmpty())
668      msgPrefix += ". ";
669
670    assertArrayEqualsBuffer(msgPrefix + "Invalid key", expectedKey, keyRead);
671    assertArrayEqualsBuffer(msgPrefix + "Invalid value", expectedValue,
672        valueRead);
673  }
