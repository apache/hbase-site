161  /**
162   * A wrapper around a block reader which only caches the results of the last
163   * operation. Not thread-safe.
164   */
165  private static class BlockReaderWrapper implements HFile.CachingBlockReader {
166
167    private HFileBlock.FSReader realReader;
168    private long prevOffset;
169    private long prevOnDiskSize;
170    private boolean prevPread;
171    private HFileBlock prevBlock;
172
173    public int hitCount = 0;
174    public int missCount = 0;
175
176    public BlockReaderWrapper(HFileBlock.FSReader realReader) {
177      this.realReader = realReader;
178    }
179
180    @Override
181    public HFileBlock readBlock(long offset, long onDiskSize,
182        boolean cacheBlock, boolean pread, boolean isCompaction,
183        boolean updateCacheMetrics, BlockType expectedBlockType,
184        DataBlockEncoding expectedDataBlockEncoding)
185        throws IOException {
186      if (offset == prevOffset && onDiskSize == prevOnDiskSize &&
187          pread == prevPread) {
188        hitCount += 1;
189        return prevBlock;
190      }
191
192      missCount += 1;
193      prevBlock = realReader.readBlockData(offset, onDiskSize, pread, false, true);
194      prevOffset = offset;
195      prevOnDiskSize = onDiskSize;
196      prevPread = pread;
197
198      return prevBlock;
199    }
200  }

202  private void readIndex(boolean useTags) throws IOException {
203    long fileSize = fs.getFileStatus(path).getLen();
204    LOG.info("Size of {}: {} compression={}", path, fileSize, compr.toString());
205
206    FSDataInputStream istream = fs.open(path);
207    HFileContext meta = new HFileContextBuilder()
208                        .withHBaseCheckSum(true)
209                        .withIncludesMvcc(includesMemstoreTS)
210                        .withIncludesTags(useTags)
211                        .withCompression(compr)
212                        .build();
213    ReaderContext context = new ReaderContextBuilder().withFileSystemAndPath(fs, path).build();
214    HFileBlock.FSReader blockReader = new HFileBlock.FSReaderImpl(context, meta,
215        ByteBuffAllocator.HEAP, conf);
216
217    BlockReaderWrapper brw = new BlockReaderWrapper(blockReader);
218    HFileBlockIndex.BlockIndexReader indexReader =
219        new HFileBlockIndex.CellBasedKeyBlockIndexReader(
220            CellComparatorImpl.COMPARATOR, numLevels);
221
222    indexReader.readRootIndex(blockReader.blockRange(rootIndexOffset,
223        fileSize).nextBlockWithBlockType(BlockType.ROOT_INDEX), numRootEntries);
224
225    long prevOffset = -1;
226    int i = 0;
227    int expectedHitCount = 0;
228    int expectedMissCount = 0;
229    LOG.info("Total number of keys: " + keys.size());
230    for (byte[] key : keys) {
231      assertTrue(key != null);
232      assertTrue(indexReader != null);
233      KeyValue.KeyOnlyKeyValue keyOnlyKey = new KeyValue.KeyOnlyKeyValue(key, 0, key.length);
234      HFileBlock b =
235          indexReader.seekToDataBlock(keyOnlyKey, null, true,
236            true, false, null, brw);
237      if (PrivateCellUtil.compare(CellComparatorImpl.COMPARATOR, keyOnlyKey, firstKeyInFile, 0,
238        firstKeyInFile.length) < 0) {
239        assertTrue(b == null);
240        ++i;
241        continue;
242      }
243
244      String keyStr = "key #" + i + ", " + Bytes.toStringBinary(key);
245
246      assertTrue("seekToDataBlock failed for " + keyStr, b != null);
247
248      if (prevOffset == b.getOffset()) {
249        assertEquals(++expectedHitCount, brw.hitCount);
250      } else {
251        LOG.info("First key in a new block: " + keyStr + ", block offset: "
252            + b.getOffset() + ")");
253        assertTrue(b.getOffset() > prevOffset);
254        assertEquals(++expectedMissCount, brw.missCount);
255        prevOffset = b.getOffset();
256      }
257      ++i;
258    }
259
260    istream.close();
261  }

263  private void writeWholeIndex(boolean useTags) throws IOException {
264    assertEquals(0, keys.size());
265    HFileContext meta = new HFileContextBuilder()
266                        .withHBaseCheckSum(true)
267                        .withIncludesMvcc(includesMemstoreTS)
268                        .withIncludesTags(useTags)
269                        .withCompression(compr)
270                        .withBytesPerCheckSum(HFile.DEFAULT_BYTES_PER_CHECKSUM)
271                        .build();
272    HFileBlock.Writer hbw = new HFileBlock.Writer(TEST_UTIL.getConfiguration(), null,
273        meta);
274    FSDataOutputStream outputStream = fs.create(path);
275    HFileBlockIndex.BlockIndexWriter biw =
276        new HFileBlockIndex.BlockIndexWriter(hbw, null, null);
277    for (int i = 0; i < NUM_DATA_BLOCKS; ++i) {
278      hbw.startWriting(BlockType.DATA).write(Bytes.toBytes(String.valueOf(RNG.nextInt(1000))));
279      long blockOffset = outputStream.getPos();
280      hbw.writeHeaderAndData(outputStream);
281
282      byte[] firstKey = null;
283      byte[] family = Bytes.toBytes("f");
284      byte[] qualifier = Bytes.toBytes("q");
285      for (int j = 0; j < 16; ++j) {
286        byte[] k =
287            new KeyValue(RandomKeyValueUtil.randomOrderedKey(RNG, i * 16 + j), family, qualifier,
288                EnvironmentEdgeManager.currentTime(), KeyValue.Type.Put).getKey();
289        keys.add(k);
290        if (j == 8) {
291          firstKey = k;
292        }
293      }
294      assertTrue(firstKey != null);
295      if (firstKeyInFile == null) {
296        firstKeyInFile = firstKey;
297      }
298      biw.addEntry(firstKey, blockOffset, hbw.getOnDiskSizeWithHeader());
299
300      writeInlineBlocks(hbw, outputStream, biw, false);
301    }
302    writeInlineBlocks(hbw, outputStream, biw, true);
303    rootIndexOffset = biw.writeIndexBlocks(outputStream);
304    outputStream.close();
305
306    numLevels = biw.getNumLevels();
307    numRootEntries = biw.getNumRootEntries();
308
309    LOG.info("Index written: numLevels=" + numLevels + ", numRootEntries=" +
310        numRootEntries + ", rootIndexOffset=" + rootIndexOffset);
311  }

313  private void writeInlineBlocks(HFileBlock.Writer hbw,
314      FSDataOutputStream outputStream, HFileBlockIndex.BlockIndexWriter biw,
315      boolean isClosing) throws IOException {
316    while (biw.shouldWriteBlock(isClosing)) {
317      long offset = outputStream.getPos();
318      biw.writeInlineBlock(hbw.startWriting(biw.getInlineBlockType()));
319      hbw.writeHeaderAndData(outputStream);
320      biw.blockWritten(offset, hbw.getOnDiskSizeWithHeader(),
321          hbw.getUncompressedSizeWithoutHeader());
322      LOG.info("Wrote an inline index block at " + offset + ", size " +
323          hbw.getOnDiskSizeWithHeader());
324    }
325  }

335  @Test
336  public void testSecondaryIndexBinarySearch() throws IOException {
337    int numTotalKeys = 99;
338    assertTrue(numTotalKeys % 2 == 1); // Ensure no one made this even.
339
340    // We only add odd-index keys into the array that we will binary-search.
341    int numSearchedKeys = (numTotalKeys - 1) / 2;
342
343    ByteArrayOutputStream baos = new ByteArrayOutputStream();
344    DataOutputStream dos = new DataOutputStream(baos);
345
346    dos.writeInt(numSearchedKeys);
347    int curAllEntriesSize = 0;
348    int numEntriesAdded = 0;
349
350    // Only odd-index elements of this array are used to keep the secondary
351    // index entries of the corresponding keys.
352    int secondaryIndexEntries[] = new int[numTotalKeys];
353
354    for (int i = 0; i < numTotalKeys; ++i) {
355      byte[] k = RandomKeyValueUtil.randomOrderedKey(RNG, i * 2);
356      KeyValue cell = new KeyValue(k, Bytes.toBytes("f"), Bytes.toBytes("q"),
357          Bytes.toBytes("val"));
358      //KeyValue cell = new KeyValue.KeyOnlyKeyValue(k, 0, k.length);
359      keys.add(cell.getKey());
360      String msgPrefix = "Key #" + i + " (" + Bytes.toStringBinary(k) + "): ";
361      StringBuilder padding = new StringBuilder();
362      while (msgPrefix.length() + padding.length() < 70)
363        padding.append(' ');
364      msgPrefix += padding;
365      if (i % 2 == 1) {
366        dos.writeInt(curAllEntriesSize);
367        secondaryIndexEntries[i] = curAllEntriesSize;
368        LOG.info(msgPrefix + "secondary index entry #" + ((i - 1) / 2) +
369            ", offset " + curAllEntriesSize);
370        curAllEntriesSize += cell.getKey().length
371            + HFileBlockIndex.SECONDARY_INDEX_ENTRY_OVERHEAD;
372        ++numEntriesAdded;
373      } else {
374        secondaryIndexEntries[i] = -1;
375        LOG.info(msgPrefix + "not in the searched array");
376      }
377    }
378
379    // Make sure the keys are increasing.
380    for (int i = 0; i < keys.size() - 1; ++i)
381      assertTrue(CellComparatorImpl.COMPARATOR.compare(
382          new KeyValue.KeyOnlyKeyValue(keys.get(i), 0, keys.get(i).length),
383          new KeyValue.KeyOnlyKeyValue(keys.get(i + 1), 0, keys.get(i + 1).length)) < 0);
384
385    dos.writeInt(curAllEntriesSize);
386    assertEquals(numSearchedKeys, numEntriesAdded);
387    int secondaryIndexOffset = dos.size();
388    assertEquals(Bytes.SIZEOF_INT * (numSearchedKeys + 2),
389        secondaryIndexOffset);
390
391    for (int i = 1; i <= numTotalKeys - 1; i += 2) {
392      assertEquals(dos.size(),
393          secondaryIndexOffset + secondaryIndexEntries[i]);
394      long dummyFileOffset = getDummyFileOffset(i);
395      int dummyOnDiskSize = getDummyOnDiskSize(i);
396      LOG.debug("Storing file offset=" + dummyFileOffset + " and onDiskSize=" +
397          dummyOnDiskSize + " at offset " + dos.size());
398      dos.writeLong(dummyFileOffset);
399      dos.writeInt(dummyOnDiskSize);
400      LOG.debug("Stored key " + ((i - 1) / 2) +" at offset " + dos.size());
401      dos.write(keys.get(i));
402    }
403
404    dos.writeInt(curAllEntriesSize);
405
406    ByteBuffer nonRootIndex = ByteBuffer.wrap(baos.toByteArray());
407    for (int i = 0; i < numTotalKeys; ++i) {
408      byte[] searchKey = keys.get(i);
409      byte[] arrayHoldingKey = new byte[searchKey.length +
410                                        searchKey.length / 2];
411
412      // To make things a bit more interesting, store the key we are looking
413      // for at a non-zero offset in a new array.
414      System.arraycopy(searchKey, 0, arrayHoldingKey, searchKey.length / 2,
415            searchKey.length);
416
417      KeyValue.KeyOnlyKeyValue cell = new KeyValue.KeyOnlyKeyValue(
418          arrayHoldingKey, searchKey.length / 2, searchKey.length);
419      int searchResult = BlockIndexReader.binarySearchNonRootIndex(cell,
420          new MultiByteBuff(nonRootIndex), CellComparatorImpl.COMPARATOR);
421      String lookupFailureMsg = "Failed to look up key #" + i + " ("
422          + Bytes.toStringBinary(searchKey) + ")";
423
424      int expectedResult;
425      int referenceItem;
426
427      if (i % 2 == 1) {
428        // This key is in the array we search as the element (i - 1) / 2. Make
429        // sure we find it.
430        expectedResult = (i - 1) / 2;
431        referenceItem = i;
432      } else {
433        // This key is not in the array but between two elements on the array,
434        // in the beginning, or in the end. The result should be the previous
435        // key in the searched array, or -1 for i = 0.
436        expectedResult = i / 2 - 1;
437        referenceItem = i - 1;
438      }
439
440      assertEquals(lookupFailureMsg, expectedResult, searchResult);
441
442      // Now test we can get the offset and the on-disk-size using a
443      // higher-level API function.s
444      boolean locateBlockResult =
445          (BlockIndexReader.locateNonRootIndexEntry(new MultiByteBuff(nonRootIndex), cell,
446          CellComparatorImpl.COMPARATOR) != -1);
447
448      if (i == 0) {
449        assertFalse(locateBlockResult);
450      } else {
451        assertTrue(locateBlockResult);
452        String errorMsg = "i=" + i + ", position=" + nonRootIndex.position();
453        assertEquals(errorMsg, getDummyFileOffset(referenceItem),
454            nonRootIndex.getLong());
455        assertEquals(errorMsg, getDummyOnDiskSize(referenceItem),
456            nonRootIndex.getInt());
457      }
458    }

462  @Test
463  public void testBlockIndexChunk() throws IOException {
464    BlockIndexChunk c = new BlockIndexChunk();
465    ByteArrayOutputStream baos = new ByteArrayOutputStream();
466    int N = 1000;
467    int[] numSubEntriesAt = new int[N];
468    int numSubEntries = 0;
469    for (int i = 0; i < N; ++i) {
470      baos.reset();
471      DataOutputStream dos = new DataOutputStream(baos);
472      c.writeNonRoot(dos);
473      assertEquals(c.getNonRootSize(), dos.size());
474
475      baos.reset();
476      dos = new DataOutputStream(baos);
477      c.writeRoot(dos);
478      assertEquals(c.getRootSize(), dos.size());
479
480      byte[] k = RandomKeyValueUtil.randomOrderedKey(RNG, i);
481      numSubEntries += RNG.nextInt(5) + 1;
482      keys.add(k);
483      c.add(k, getDummyFileOffset(i), getDummyOnDiskSize(i), numSubEntries);
484    }
485
486    // Test the ability to look up the entry that contains a particular
487    // deeper-level index block's entry ("sub-entry"), assuming a global
488    // 0-based ordering of sub-entries. This is needed for mid-key calculation.
489    for (int i = 0; i < N; ++i) {
490      for (int j = i == 0 ? 0 : numSubEntriesAt[i - 1];
491           j < numSubEntriesAt[i];
492           ++j) {
493        assertEquals(i, c.getEntryBySubEntry(j));
494      }
495    }
496  }

500  public void testHeapSizeForBlockIndex() throws IOException {
501    Class<HFileBlockIndex.BlockIndexReader> cl =
502        HFileBlockIndex.BlockIndexReader.class;
503    long expected = ClassSize.estimateBase(cl, false);

508

510    // int [] blockDataSizes) are all null they are not going to show up in the
511    // HeapSize calculation, so need to remove those array costs from expected.

516      expected = ClassSize.estimateBase(cl, true);
517      assertEquals(expected, actual);

615                .withFileContext(meta)
616                .create();
617        Random rand = new Random(19231737);
618        byte[] family = Bytes.toBytes("f");
619        byte[] qualifier = Bytes.toBytes("q");
620        for (int i = 0; i < NUM_KV; ++i) {
621          byte[] row = RandomKeyValueUtil.randomOrderedKey(rand, i);
622
623          // Key will be interpreted by KeyValue.KEY_COMPARATOR
624          KeyValue kv =
625              new KeyValue(row, family, qualifier, EnvironmentEdgeManager.currentTime(),
626                  RandomKeyValueUtil.randomValue(rand));
627          byte[] k = kv.getKey();
628          writer.append(kv);
629          keys[i] = k;
630          values[i] = CellUtil.cloneValue(kv);
631          keyStrSet.add(Bytes.toStringBinary(k));
632          if (i > 0) {
633            assertTrue((PrivateCellUtil.compare(CellComparatorImpl.COMPARATOR, kv, keys[i - 1],
634                0, keys[i - 1].length)) > 0);
635          }
636        }
637
638        writer.close();
639      }
640
641      // Read the HFile
642      HFile.Reader reader = HFile.createReader(fs, hfilePath, cacheConf, true, conf);
643      assertEquals(expectedNumLevels,
644          reader.getTrailer().getNumDataIndexLevels());
645
646      assertTrue(Bytes.equals(keys[0], ((KeyValue)reader.getFirstKey().get()).getKey()));
647      assertTrue(Bytes.equals(keys[NUM_KV - 1], ((KeyValue)reader.getLastKey().get()).getKey()));
648      LOG.info("Last key: " + Bytes.toStringBinary(keys[NUM_KV - 1]));
649
650      for (boolean pread : new boolean[] { false, true }) {
651        HFileScanner scanner = reader.getScanner(conf, true, pread);
652        for (int i = 0; i < NUM_KV; ++i) {
653          checkSeekTo(keys, scanner, i);
654          checkKeyValue("i=" + i, keys[i], values[i],
655              ByteBuffer.wrap(((KeyValue) scanner.getKey()).getKey()), scanner.getValue());
656        }
657        assertTrue(scanner.seekTo());
658        for (int i = NUM_KV - 1; i >= 0; --i) {
659          checkSeekTo(keys, scanner, i);
660          checkKeyValue("i=" + i, keys[i], values[i],
661              ByteBuffer.wrap(((KeyValue) scanner.getKey()).getKey()), scanner.getValue());
662        }
663      }
664
665      // Manually compute the mid-key and validate it.
666      HFile.Reader reader2 = reader;
667      HFileBlock.FSReader fsReader = reader2.getUncachedBlockReader();
668
669      HFileBlock.BlockIterator iter = fsReader.blockRange(0,
670          reader.getTrailer().getLoadOnOpenDataOffset());
671      HFileBlock block;
672      List<byte[]> blockKeys = new ArrayList<>();
673      while ((block = iter.nextBlock()) != null) {
674        if (block.getBlockType() != BlockType.LEAF_INDEX)
675          return;
676        ByteBuff b = block.getBufferReadOnly();
677        int n = b.getIntAfterPosition(0);
678        // One int for the number of items, and n + 1 for the secondary index.
679        int entriesOffset = Bytes.SIZEOF_INT * (n + 2);
680
681        // Get all the keys from the leaf index block. S
682        for (int i = 0; i < n; ++i) {
683          int keyRelOffset = b.getIntAfterPosition(Bytes.SIZEOF_INT * (i + 1));
684          int nextKeyRelOffset = b.getIntAfterPosition(Bytes.SIZEOF_INT * (i + 2));
685          int keyLen = nextKeyRelOffset - keyRelOffset;
686          int keyOffset = b.arrayOffset() + entriesOffset + keyRelOffset +
687              HFileBlockIndex.SECONDARY_INDEX_ENTRY_OVERHEAD;
688          byte[] blockKey = Arrays.copyOfRange(b.array(), keyOffset, keyOffset
689              + keyLen);
690          String blockKeyStr = Bytes.toString(blockKey);
691          blockKeys.add(blockKey);
692
693          // If the first key of the block is not among the keys written, we
694          // are not parsing the non-root index block format correctly.
695          assertTrue("Invalid block key from leaf-level block: " + blockKeyStr,
696              keyStrSet.contains(blockKeyStr));
697        }
698      }
699
700      // Validate the mid-key.
701      assertEquals(
702          Bytes.toStringBinary(blockKeys.get((blockKeys.size() - 1) / 2)),
703          reader.midKey());
704
705      assertEquals(UNCOMPRESSED_INDEX_SIZES[testI],
706          reader.getTrailer().getUncompressedDataIndexSize());
707
708      reader.close();
709      reader2.close();
710    }
711  }

714      throws IOException {

719  private void assertArrayEqualsBuffer(String msgPrefix, byte[] arr,
720      ByteBuffer buf) {
721    assertEquals(msgPrefix + ": expected " + Bytes.toStringBinary(arr)
722        + ", actual " + Bytes.toStringBinary(buf), 0, Bytes.compareTo(arr, 0,
723        arr.length, buf.array(), buf.arrayOffset(), buf.limit()));
724  }

726  /** Check a key/value pair after it was read by the reader */
727  private void checkKeyValue(String msgPrefix, byte[] expectedKey,
728      byte[] expectedValue, ByteBuffer keyRead, ByteBuffer valueRead) {
729    if (!msgPrefix.isEmpty())
730      msgPrefix += ". ";
731
732    assertArrayEqualsBuffer(msgPrefix + "Invalid key", expectedKey, keyRead);
733    assertArrayEqualsBuffer(msgPrefix + "Invalid value", expectedValue,
734        valueRead);
735  }

748  public void testIntermediateLevelIndicesWithLargeKeys(int minNumEntries) throws IOException {
749    Path hfPath = new Path(TEST_UTIL.getDataTestDir(),
750      "testIntermediateLevelIndicesWithLargeKeys.hfile");
751    int maxChunkSize = 1024;
752    FileSystem fs = FileSystem.get(conf);
753    CacheConfig cacheConf = new CacheConfig(conf);
754    conf.setInt(HFileBlockIndex.MAX_CHUNK_SIZE_KEY, maxChunkSize);
755    conf.setInt(HFileBlockIndex.MIN_INDEX_NUM_ENTRIES_KEY, minNumEntries);
756    HFileContext context = new HFileContextBuilder().withBlockSize(16).build();
757    HFile.Writer hfw = new HFile.WriterFactory(conf, cacheConf)
758            .withFileContext(context)
759            .withPath(fs, hfPath).create();
760    List<byte[]> keys = new ArrayList<>();
761
762    // This should result in leaf-level indices and a root level index
763    for (int i=0; i < 100; i++) {
764      byte[] rowkey = new byte[maxChunkSize + 1];
765      byte[] b = Bytes.toBytes(i);
766      System.arraycopy(b, 0, rowkey, rowkey.length - b.length, b.length);
767      keys.add(rowkey);
768      hfw.append(ExtendedCellBuilderFactory.create(CellBuilderType.DEEP_COPY)
769        .setRow(rowkey).setFamily(HConstants.EMPTY_BYTE_ARRAY)
770        .setQualifier(HConstants.EMPTY_BYTE_ARRAY)
771        .setTimestamp(HConstants.LATEST_TIMESTAMP)
772        .setType(KeyValue.Type.Maximum.getCode())
773        .setValue(HConstants.EMPTY_BYTE_ARRAY)
774        .build());
775    }
776    hfw.close();
777
778    HFile.Reader reader = HFile.createReader(fs, hfPath, cacheConf, true, conf);
779    // Scanner doesn't do Cells yet.  Fix.
780    HFileScanner scanner = reader.getScanner(conf, true, true);
781    for (int i = 0; i < keys.size(); ++i) {
782      scanner.seekTo(ExtendedCellBuilderFactory.create(CellBuilderType.DEEP_COPY)
783        .setRow(keys.get(i)).setFamily(HConstants.EMPTY_BYTE_ARRAY)
784        .setQualifier(HConstants.EMPTY_BYTE_ARRAY)
785        .setTimestamp(HConstants.LATEST_TIMESTAMP)
786        .setType(KeyValue.Type.Maximum.getCode())
787        .setValue(HConstants.EMPTY_BYTE_ARRAY)
788        .build());
789    }
790    reader.close();
791  }
