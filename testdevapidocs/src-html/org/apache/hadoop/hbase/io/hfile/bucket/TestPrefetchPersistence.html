086  private static final int NUM_VALID_KEY_TYPES = KeyValue.Type.values().length - 2;

100  final int writerQLen = BucketCache.DEFAULT_WRITER_QUEUE_ITEMS;

102  @Before
103  public void setup() throws IOException {
104    conf = TEST_UTIL.getConfiguration();
105    conf.setBoolean(CacheConfig.PREFETCH_BLOCKS_ON_OPEN_KEY, true);
106    testDir = TEST_UTIL.getDataTestDir();
107    TEST_UTIL.getTestFileSystem().mkdirs(testDir);
108    fs = HFileSystem.get(conf);
109  }
110
111  @Test
112  public void testPrefetchPersistence() throws Exception {
113    bucketCache = new BucketCache("file:" + testDir + "/bucket.cache", capacitySize,
114      constructedBlockSize, constructedBlockSizes, writeThreads, writerQLen,
115      testDir + "/bucket.persistence", 60 * 1000, conf);
116    cacheConf = new CacheConfig(conf, bucketCache);
117
118    long usedSize = bucketCache.getAllocator().getUsedSize();
119    assertEquals(0, usedSize);
120    assertTrue(new File(testDir + "/bucket.cache").exists());
121    // Load Cache
122    Path storeFile = writeStoreFile("TestPrefetch0");
123    Path storeFile2 = writeStoreFile("TestPrefetch1");
124    readStoreFile(storeFile, 0);
125    readStoreFile(storeFile2, 0);
126    usedSize = bucketCache.getAllocator().getUsedSize();
127    assertNotEquals(0, usedSize);
128
129    bucketCache.shutdown();
130    assertTrue(new File(testDir + "/bucket.persistence").exists());
131    bucketCache = new BucketCache("file:" + testDir + "/bucket.cache", capacitySize,
132      constructedBlockSize, constructedBlockSizes, writeThreads, writerQLen,
133      testDir + "/bucket.persistence", 60 * 1000, conf);
134    cacheConf = new CacheConfig(conf, bucketCache);
135    assertTrue(usedSize != 0);
136    readStoreFile(storeFile, 0);
137    readStoreFile(storeFile2, 0);
138    // Test Close Store File
139    closeStoreFile(storeFile2);
140    TEST_UTIL.cleanupTestDir();
141  }
142
143  public void closeStoreFile(Path path) throws Exception {
144    HFile.Reader reader = HFile.createReader(fs, path, cacheConf, true, conf);
145    assertTrue(bucketCache.fullyCachedFiles.containsKey(path.getName()));
146    reader.close(true);
147    assertFalse(bucketCache.fullyCachedFiles.containsKey(path.getName()));
148  }
149
150  public void readStoreFile(Path storeFilePath, long offset) throws Exception {
151    // Open the file
152    HFile.Reader reader = HFile.createReader(fs, storeFilePath, cacheConf, true, conf);
153
154    while (!reader.prefetchComplete()) {
155      // Sleep for a bit
156      Thread.sleep(1000);
157    }
158    HFileBlock block = reader.readBlock(offset, -1, false, true, false, true, null, null);
159    BlockCacheKey blockCacheKey = new BlockCacheKey(reader.getName(), offset);
160    BucketEntry be = bucketCache.backingMap.get(blockCacheKey);
161    boolean isCached = bucketCache.getBlock(blockCacheKey, true, false, true) != null;
162
163    if (
164      block.getBlockType() == BlockType.DATA || block.getBlockType() == BlockType.ROOT_INDEX
165        || block.getBlockType() == BlockType.INTERMEDIATE_INDEX
166    ) {
167      assertTrue(isCached);
168    }
169  }
170
171  public Path writeStoreFile(String fname) throws IOException {
172    Path storeFileParentDir = new Path(TEST_UTIL.getDataTestDir(), fname);
173    HFileContext meta = new HFileContextBuilder().withBlockSize(DATA_BLOCK_SIZE).build();
174    StoreFileWriter sfw = new StoreFileWriter.Builder(conf, cacheConf, fs)
175      .withOutputDir(storeFileParentDir).withFileContext(meta).build();
176    Random rand = ThreadLocalRandom.current();
177    final int rowLen = 32;
178    for (int i = 0; i < NUM_KV; ++i) {
179      byte[] k = RandomKeyValueUtil.randomOrderedKey(rand, i);
180      byte[] v = RandomKeyValueUtil.randomValue(rand);
181      int cfLen = rand.nextInt(k.length - rowLen + 1);
182      KeyValue kv = new KeyValue(k, 0, rowLen, k, rowLen, cfLen, k, rowLen + cfLen,
183        k.length - rowLen - cfLen, rand.nextLong(), generateKeyType(rand), v, 0, v.length);
184      sfw.append(kv);
185    }
186
187    sfw.close();
188    return sfw.getPath();
189  }
190
191  public static KeyValue.Type generateKeyType(Random rand) {
192    if (rand.nextBoolean()) {
193      // Let's make half of KVs puts.
194      return KeyValue.Type.Put;
195    } else {
196      KeyValue.Type keyType = KeyValue.Type.values()[1 + rand.nextInt(NUM_VALID_KEY_TYPES)];
197      if (keyType == KeyValue.Type.Minimum || keyType == KeyValue.Type.Maximum) {
198        throw new RuntimeException("Generated an invalid key type: " + keyType + ". "
199          + "Probably the layout of KeyValue.Type has changed.");
200      }
201      return keyType;
202    }
203  }
