112  private void readStoreFile(Path storeFilePath) throws Exception {
113    // Open the file reader with block cache disabled.
114    HFile.Reader reader = HFile.createReader(fs, storeFilePath, conf);
115    long offset = 0;
116    while (offset < reader.getTrailer().getLoadOnOpenDataOffset()) {
117      HFileBlock block = reader.readBlock(offset, -1, false, true, false, true, null, null);
118      offset += block.getOnDiskSizeWithHeader();
119    }
120  }

122  private Path writeStoreFile() throws IOException {
123    Path storeFileParentDir = new Path(TEST_UTIL.getDataTestDir(), "TestHFile");
124    HFileContext meta = new HFileContextBuilder().withBlockSize(64 * 1024).build();
125    StoreFileWriter sfw =
126        new StoreFileWriter.Builder(conf, fs).withOutputDir(storeFileParentDir)
127            .withComparator(CellComparatorImpl.COMPARATOR).withFileContext(meta).build();
128
129    final int rowLen = 32;
130    Random RNG = new Random();
131    for (int i = 0; i < 1000; ++i) {
132      byte[] k = RandomKeyValueUtil.randomOrderedKey(RNG, i);
133      byte[] v = RandomKeyValueUtil.randomValue(RNG);
134      int cfLen = RNG.nextInt(k.length - rowLen + 1);
135      KeyValue kv =
136          new KeyValue(k, 0, rowLen, k, rowLen, cfLen, k, rowLen + cfLen,
137              k.length - rowLen - cfLen, RNG.nextLong(), generateKeyType(RNG), v, 0, v.length);
138      sfw.append(kv);
139    }
140
141    sfw.close();
142    return sfw.getPath();
143  }

145  public static KeyValue.Type generateKeyType(Random rand) {
146    if (rand.nextBoolean()) {
147      // Let's make half of KVs puts.
148      return KeyValue.Type.Put;
149    } else {
150      KeyValue.Type keyType = KeyValue.Type.values()[1 + rand.nextInt(NUM_VALID_KEY_TYPES)];
151      if (keyType == KeyValue.Type.Minimum || keyType == KeyValue.Type.Maximum) {
152        throw new RuntimeException("Generated an invalid key type: " + keyType + ". "
153            + "Probably the layout of KeyValue.Type has changed.");
154      }
155      return keyType;
156    }
157  }

195  public static void truncateFile(FileSystem fs, Path src, Path dst) throws IOException {
196    FileStatus fst = fs.getFileStatus(src);
197    long len = fst.getLen();
198    len = len / 2 ;
199
200    // create a truncated hfile
201    FSDataOutputStream fdos = fs.create(dst);
202    byte[] buf = new byte[(int)len];
203    FSDataInputStream fdis = fs.open(src);
204    fdis.read(buf);
205    fdos.write(buf);
206    fdis.close();
207    fdos.close();
208  }

234  // write some records into the hfile
235  // write them twice
236  private int writeSomeRecords(Writer writer, int start, int n, boolean useTags)
237      throws IOException {
238    String value = "value";
239    KeyValue kv;
240    for (int i = start; i < (start + n); i++) {
241      String key = String.format(localFormatter, Integer.valueOf(i));
242      if (useTags) {
243        Tag t = new ArrayBackedTag((byte) 1, "myTag1");
244        Tag[] tags = new Tag[1];
245        tags[0] = t;
246        kv = new KeyValue(Bytes.toBytes(key), Bytes.toBytes("family"), Bytes.toBytes("qual"),
247            HConstants.LATEST_TIMESTAMP, Bytes.toBytes(value + key), tags);
248        writer.append(kv);
249      } else {
250        kv = new KeyValue(Bytes.toBytes(key), Bytes.toBytes("family"), Bytes.toBytes("qual"),
251            Bytes.toBytes(value + key));
252        writer.append(kv);
253      }
254    }
255    return (start + n);
256  }

262  // read the records and check
263  private int readAndCheckbytes(HFileScanner scanner, int start, int n)
264      throws IOException {
265    String value = "value";
266    int i = start;
267    for (; i < (start + n); i++) {
268      ByteBuffer key = ByteBuffer.wrap(((KeyValue)scanner.getKey()).getKey());
269      ByteBuffer val = scanner.getValue();
270      String keyStr = String.format(localFormatter, Integer.valueOf(i));
271      String valStr = value + keyStr;
272      KeyValue kv = new KeyValue(Bytes.toBytes(keyStr), Bytes.toBytes("family"),
273          Bytes.toBytes("qual"), Bytes.toBytes(valStr));
274      byte[] keyBytes = new KeyValue.KeyOnlyKeyValue(Bytes.toBytes(key), 0,
275          Bytes.toBytes(key).length).getKey();
276      assertTrue("bytes for keys do not match " + keyStr + " " +
277        Bytes.toString(Bytes.toBytes(key)),
278          Arrays.equals(kv.getKey(), keyBytes));
279      byte [] valBytes = Bytes.toBytes(val);
280      assertTrue("bytes for vals do not match " + valStr + " " +
281        Bytes.toString(valBytes),
282        Arrays.equals(Bytes.toBytes(valStr), valBytes));
283      if (!scanner.next()) {
284        break;
285      }
286    }
287    assertEquals(i, start + n - 1);
288    return (start + n);
289  }

308  /**
309   * test none codecs
310   * @param useTags
311   */
312  void basicWithSomeCodec(String codec, boolean useTags) throws IOException {
313    if (useTags) {
314      conf.setInt("hfile.format.version", 3);
315    }
316    Path  ncHFile = new Path(ROOT_DIR, "basic.hfile." + codec.toString() + useTags);
317    FSDataOutputStream fout = createFSOutput(ncHFile);
318    HFileContext meta = new HFileContextBuilder()
319                        .withBlockSize(minBlockSize)
320                        .withCompression(HFileWriterImpl.compressionByName(codec))
321                        .build();
322    Writer writer = HFile.getWriterFactory(conf, cacheConf)
323        .withOutputStream(fout)
324        .withFileContext(meta)
325        .withComparator(CellComparatorImpl.COMPARATOR)
326        .create();
327    LOG.info(Objects.toString(writer));
328    writeRecords(writer, useTags);
329    fout.close();
330    FSDataInputStream fin = fs.open(ncHFile);
331    Reader reader = HFile.createReaderFromStream(ncHFile, fs.open(ncHFile),
332      fs.getFileStatus(ncHFile).getLen(), cacheConf, conf);
333    System.out.println(cacheConf.toString());
334    // Load up the index.
335    reader.loadFileInfo();
336    // Get a scanner that caches and that does not use pread.
337    HFileScanner scanner = reader.getScanner(true, false);
338    // Align scanner at start of the file.
339    scanner.seekTo();
340    readAllRecords(scanner);
341    int seekTo = scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(50)));
342    System.out.println(seekTo);
343    assertTrue("location lookup failed",
344        scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(50))) == 0);
345    // read the key and see if it matches
346    ByteBuffer readKey = ByteBuffer.wrap(((KeyValue)scanner.getKey()).getKey());
347    assertTrue("seeked key does not match", Arrays.equals(getSomeKey(50),
348      Bytes.toBytes(readKey)));
349
350    scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(0)));
351    ByteBuffer val1 = scanner.getValue();
352    scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(0)));
353    ByteBuffer val2 = scanner.getValue();
354    assertTrue(Arrays.equals(Bytes.toBytes(val1), Bytes.toBytes(val2)));
355
356    reader.close();
357    fin.close();
358    fs.delete(ncHFile, true);
359  }

373    for (int i = 0; i < n; i++) {
374      writer.appendMetaBlock("HFileMeta" + i, new Writable() {
375        private int val;

380          out.write(("something to test" + val).getBytes());

385      }.setVal(i));
386    }

429    someReadingWithMetaBlock(reader);
430    fs.delete(mFile, true);
431    reader.close();
432    fin.close();
