136  private ByteBuffAllocator initAllocator(boolean reservoirEnabled, int bufSize, int bufCount,
137    int minAllocSize) {
138    Configuration that = HBaseConfiguration.create(conf);
139    that.setInt(BUFFER_SIZE_KEY, bufSize);
140    that.setInt(MAX_BUFFER_COUNT_KEY, bufCount);
141    // All ByteBuffers will be allocated from the buffers.
142    that.setInt(MIN_ALLOCATE_SIZE_KEY, minAllocSize);
143    return ByteBuffAllocator.create(that, reservoirEnabled);
144  }

146  private void fillByteBuffAllocator(ByteBuffAllocator alloc, int bufCount) {
147    // Fill the allocator with bufCount ByteBuffer
148    List<ByteBuff> buffs = new ArrayList<>();
149    for (int i = 0; i < bufCount; i++) {
150      buffs.add(alloc.allocateOneBuffer());
151      Assert.assertEquals(alloc.getFreeBufferCount(), 0);
152    }
153    buffs.forEach(ByteBuff::release);
154    Assert.assertEquals(alloc.getFreeBufferCount(), bufCount);
155  }

178  @Test
179  public void testReaderWithLRUBlockCache() throws Exception {
180    int bufCount = 1024, blockSize = 64 * 1024;
181    ByteBuffAllocator alloc = initAllocator(true, bufCount, blockSize, 0);
182    fillByteBuffAllocator(alloc, bufCount);
183    Path storeFilePath = writeStoreFile();
184    // Open the file reader with LRUBlockCache
185    BlockCache lru = new LruBlockCache(1024 * 1024 * 32, blockSize, true, conf);
186    CacheConfig cacheConfig = new CacheConfig(conf, null, lru, alloc);
187    HFile.Reader reader = HFile.createReader(fs, storeFilePath, cacheConfig, true, conf);
188    long offset = 0;
189    while (offset < reader.getTrailer().getLoadOnOpenDataOffset()) {
190      BlockCacheKey key = new BlockCacheKey(storeFilePath.getName(), offset);
191      HFileBlock block = reader.readBlock(offset, -1, true, true, false, true, null, null);
192      offset += block.getOnDiskSizeWithHeader();
193      // Ensure the block is an heap one.
194      Cacheable cachedBlock = lru.getBlock(key, false, false, true);
195      Assert.assertNotNull(cachedBlock);
196      Assert.assertTrue(cachedBlock instanceof HFileBlock);
197      Assert.assertFalse(((HFileBlock) cachedBlock).isSharedMem());
198      // Should never allocate off-heap block from allocator because ensure that it's LRU.
199      Assert.assertEquals(bufCount, alloc.getFreeBufferCount());
200      block.release(); // return back the ByteBuffer back to allocator.
201    }
202    reader.close();
203    Assert.assertEquals(bufCount, alloc.getFreeBufferCount());
204    alloc.clean();
205    lru.shutdown();
206  }

222  @Test
223  public void testReaderWithCombinedBlockCache() throws Exception {
224    int bufCount = 1024, blockSize = 64 * 1024;
225    ByteBuffAllocator alloc = initAllocator(true, bufCount, blockSize, 0);
226    fillByteBuffAllocator(alloc, bufCount);
227    Path storeFilePath = writeStoreFile();
228    // Open the file reader with CombinedBlockCache
229    BlockCache combined = initCombinedBlockCache("LRU");
230    conf.setBoolean(EVICT_BLOCKS_ON_CLOSE_KEY, true);
231    CacheConfig cacheConfig = new CacheConfig(conf, null, combined, alloc);
232    HFile.Reader reader = HFile.createReader(fs, storeFilePath, cacheConfig, true, conf);
233    long offset = 0;
234    while (offset < reader.getTrailer().getLoadOnOpenDataOffset()) {
235      BlockCacheKey key = new BlockCacheKey(storeFilePath.getName(), offset);
236      HFileBlock block = reader.readBlock(offset, -1, true, true, false, true, null, null);
237      offset += block.getOnDiskSizeWithHeader();
238      // Read the cached block.
239      Cacheable cachedBlock = combined.getBlock(key, false, false, true);
240      try {
241        Assert.assertNotNull(cachedBlock);
242        Assert.assertTrue(cachedBlock instanceof HFileBlock);
243        HFileBlock hfb = (HFileBlock) cachedBlock;
244        // Data block will be cached in BucketCache, so it should be an off-heap block.
245        if (hfb.getBlockType().isData()) {
246          Assert.assertTrue(hfb.isSharedMem());
247        } else {
248          // Non-data block will be cached in LRUBlockCache, so it must be an on-heap block.
249          Assert.assertFalse(hfb.isSharedMem());
250        }
251      } finally {
252        cachedBlock.release();
253      }
254      block.release(); // return back the ByteBuffer back to allocator.
255    }
256    reader.close();
257    combined.shutdown();
258    Assert.assertEquals(bufCount, alloc.getFreeBufferCount());
259    alloc.clean();
260  }

378  private long readAtOffsetWithAllocationAsserts(ByteBuffAllocator alloc, HFile.Reader reader,
379    long offset, boolean cacheBlock, BlockType blockType, boolean expectSharedMem)
380    throws IOException {
381    HFileBlock block;
382    try {
383      block = reader.readBlock(offset, -1, cacheBlock, true, false, true, blockType, null);
384    } catch (IOException e) {
385      if (e.getMessage().contains("Expected block type")) {
386        return -1;
387      }
388      throw e;
389    }
390
391    Assert.assertEquals(expectSharedMem, block.isSharedMem());
392
393    if (expectSharedMem) {
394      Assert.assertTrue(alloc.getFreeBufferCount() < alloc.getTotalBufferCount());
395    } else {
396      // Should never allocate off-heap block from allocator because ensure that it's LRU.
397      Assert.assertEquals(alloc.getTotalBufferCount(), alloc.getFreeBufferCount());
398    }
399
400    try {
401      return block.getOnDiskSizeWithHeader();
402    } finally {
403      block.release(); // return back the ByteBuffer back to allocator.
404    }
405  }

407  private void readStoreFile(Path storeFilePath, Configuration conf, ByteBuffAllocator alloc)
408    throws Exception {
409    // Open the file reader with block cache disabled.
410    CacheConfig cache = new CacheConfig(conf, null, null, alloc);

416      block.release(); // return back the ByteBuffer back to allocator.
417    }
418    reader.close();
419  }

421  private Path writeStoreFile() throws IOException {
422    Path storeFileParentDir = new Path(TEST_UTIL.getDataTestDir(), "TestHFile");
423    HFileContext meta = new HFileContextBuilder().withBlockSize(64 * 1024).build();
424    StoreFileWriter sfw = new StoreFileWriter.Builder(conf, fs).withOutputDir(storeFileParentDir)
425      .withFileContext(meta).build();
426    final int rowLen = 32;
427    Random rand = ThreadLocalRandom.current();
428    for (int i = 0; i < 1000; ++i) {
429      byte[] k = RandomKeyValueUtil.randomOrderedKey(rand, i);
430      byte[] v = RandomKeyValueUtil.randomValue(rand);
431      int cfLen = rand.nextInt(k.length - rowLen + 1);
432      KeyValue kv = new KeyValue(k, 0, rowLen, k, rowLen, cfLen, k, rowLen + cfLen,
433        k.length - rowLen - cfLen, rand.nextLong(), generateKeyType(rand), v, 0, v.length);
434      sfw.append(kv);
435    }
436
437    sfw.close();
438    return sfw.getPath();
439  }

441  public static KeyValue.Type generateKeyType(Random rand) {
442    if (rand.nextBoolean()) {
443      // Let's make half of KVs puts.
444      return KeyValue.Type.Put;
445    } else {
446      KeyValue.Type keyType = KeyValue.Type.values()[1 + rand.nextInt(NUM_VALID_KEY_TYPES)];
447      if (keyType == KeyValue.Type.Minimum || keyType == KeyValue.Type.Maximum) {
448        throw new RuntimeException("Generated an invalid key type: " + keyType + ". "
449          + "Probably the layout of KeyValue.Type has changed.");
450      }
451      return keyType;
452    }
453  }

496        .withBlockSize(minBlockSize).withColumnFamily(Bytes.toBytes(columnFamily))
497        .withTableName(Bytes.toBytes(tableName)).withHBaseCheckSum(false)
498        .withCompression(Compression.Algorithm.NONE).withCompressTags(false).build();
499    HFileWriterImpl writer =
500      new HFileWriterImpl(conf, cacheConf, path, mockedOutputStream, fileContext);
501    CellBuilder cellBuilder = CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY);
502    byte[] row = Bytes.toBytes("foo");
503    byte[] qualifier = Bytes.toBytes("qualifier");
504    byte[] cf = Bytes.toBytes(columnFamily);
505    byte[] val = Bytes.toBytes("fooVal");
506    long firstTS = 100L;
507    long secondTS = 101L;
508    Cell firstCell = cellBuilder.setRow(row).setValue(val).setTimestamp(firstTS)
509      .setQualifier(qualifier).setFamily(cf).setType(Cell.Type.Put).build();
510    Cell secondCell = cellBuilder.setRow(row).setValue(val).setTimestamp(secondTS)
511      .setQualifier(qualifier).setFamily(cf).setType(Cell.Type.Put).build();
512    // second Cell will sort "higher" than the first because later timestamps should come first
513    writer.append(firstCell);
514    try {
515      writer.append(secondCell);
516    } catch (IOException ie) {
517      String message = ie.getMessage();
518      Assert.assertTrue(message.contains("not lexically larger"));
519      Assert.assertTrue(message.contains(tableName));
520      Assert.assertTrue(message.contains(columnFamily));
521      return;
522    }
523    Assert.fail("Exception wasn't thrown even though Cells were appended in the wrong order!");
524  }

526  public static void truncateFile(FileSystem fs, Path src, Path dst) throws IOException {
527    FileStatus fst = fs.getFileStatus(src);
528    long len = fst.getLen();
529    len = len / 2;
530
531    // create a truncated hfile
532    FSDataOutputStream fdos = fs.create(dst);
533    byte[] buf = new byte[(int) len];
534    FSDataInputStream fdis = fs.open(src);
535    fdis.read(buf);
536    fdos.write(buf);
537    fdis.close();
538    fdos.close();
539  }

594  private int readAndCheckbytes(HFileScanner scanner, int start, int n) throws IOException {
595    String value = "value";
596    int i = start;
597    for (; i < (start + n); i++) {
598      ByteBuffer key = ByteBuffer.wrap(((KeyValue) scanner.getKey()).getKey());
599      ByteBuffer val = scanner.getValue();
600      String keyStr = String.format(localFormatter, Integer.valueOf(i));
601      String valStr = value + keyStr;
602      KeyValue kv = new KeyValue(Bytes.toBytes(keyStr), Bytes.toBytes("family"),
603        Bytes.toBytes("qual"), Bytes.toBytes(valStr));
604      byte[] keyBytes =
605        new KeyValue.KeyOnlyKeyValue(Bytes.toBytes(key), 0, Bytes.toBytes(key).length).getKey();
606      assertTrue("bytes for keys do not match " + keyStr + " " + Bytes.toString(Bytes.toBytes(key)),
607        Arrays.equals(kv.getKey(), keyBytes));
608      byte[] valBytes = Bytes.toBytes(val);
609      assertTrue("bytes for vals do not match " + valStr + " " + Bytes.toString(valBytes),
610        Arrays.equals(Bytes.toBytes(valStr), valBytes));
611      if (!scanner.next()) {
612        break;
613      }
614    }
615    assertEquals(i, start + n - 1);
616    return (start + n);
617  }

1031    HFile.Reader reader = HFile.createReader(fs, storeFilePath, cacheConfig, true, conf);
1032    long offset = 0;
1033    Cacheable cachedBlock = null;
1034    while (offset < reader.getTrailer().getLoadOnOpenDataOffset()) {
1035      BlockCacheKey key = new BlockCacheKey(storeFilePath.getName(), offset);
1036      HFileBlock block = reader.readBlock(offset, -1, true, true, false, true, null, null);
1037      offset += block.getOnDiskSizeWithHeader();
1038      // Read the cached block.
1039      cachedBlock = combined.getBlock(key, false, false, true);
1040      try {
1041        Assert.assertNotNull(cachedBlock);
1042        Assert.assertTrue(cachedBlock instanceof HFileBlock);
1043        HFileBlock hfb = (HFileBlock) cachedBlock;
1044        // Data block will be cached in BucketCache, so it should be an off-heap block.
1045        if (hfb.getBlockType().isData()) {
1046          Assert.assertTrue(hfb.isSharedMem());
1047        } else if (!l1CachePolicy.equals("TinyLfu")) {
1048          Assert.assertFalse(hfb.isSharedMem());
1049        }
1050      } finally {
1051        cachedBlock.release();
1052      }
1053      block.release(); // return back the ByteBuffer back to allocator.
1054    }
1055    reader.close();
1056    combined.shutdown();
1057    if (cachedBlock != null) {
1058      Assert.assertEquals(0, cachedBlock.refCnt());
1059    }
1060    Assert.assertEquals(bufCount, alloc.getFreeBufferCount());
1061    alloc.clean();
1062  }
