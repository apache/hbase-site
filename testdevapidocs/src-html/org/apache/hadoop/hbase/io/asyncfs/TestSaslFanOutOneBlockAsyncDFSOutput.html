122    for (String protection : Arrays.asList("authentication", "integrity", "privacy")) {

124        for (String cipherSuite : Arrays.asList("", CipherSuite.AES_CTR_NOPADDING.getName())) {

132  private static void setHdfsSecuredConfiguration(Configuration conf) throws Exception {
133    conf.set(DFS_NAMENODE_KERBEROS_PRINCIPAL_KEY, PRINCIPAL + "@" + KDC.getRealm());
134    conf.set(DFS_NAMENODE_KEYTAB_FILE_KEY, KEYTAB_FILE.getAbsolutePath());
135    conf.set(DFS_DATANODE_KERBEROS_PRINCIPAL_KEY, PRINCIPAL + "@" + KDC.getRealm());
136    conf.set(DFS_DATANODE_KEYTAB_FILE_KEY, KEYTAB_FILE.getAbsolutePath());
137    conf.set(DFS_WEB_AUTHENTICATION_KERBEROS_PRINCIPAL_KEY, HTTP_PRINCIPAL + "@" + KDC.getRealm());
138    conf.setBoolean(DFS_BLOCK_ACCESS_TOKEN_ENABLE_KEY, true);
139    conf.set(DFS_HTTP_POLICY_KEY, HttpConfig.Policy.HTTPS_ONLY.name());
140    conf.set(DFS_NAMENODE_HTTPS_ADDRESS_KEY, "localhost:0");
141    conf.set(DFS_DATANODE_HTTPS_ADDRESS_KEY, "localhost:0");
142
143    File keystoresDir = new File(TEST_UTIL.getDataTestDir("keystore").toUri().getPath());
144    keystoresDir.mkdirs();
145    String sslConfDir = KeyStoreTestUtil.getClasspathDir(TestGenerateDelegationToken.class);
146    KeyStoreTestUtil.setupSSLConfig(keystoresDir.getAbsolutePath(), sslConfDir, conf, false);
147
148    conf.setBoolean("ignore.secure.ports.for.testing", true);
149  }

151  private static void setUpKeyProvider(Configuration conf) throws Exception {
152    URI keyProviderUri =
153      new URI("jceks://file" + TEST_UTIL.getDataTestDir("test.jks").toUri().toString());
154    conf.set("dfs.encryption.key.provider.uri", keyProviderUri.toString());
155    KeyProvider keyProvider = KeyProviderFactory.get(keyProviderUri, conf);
156    keyProvider.createKey(TEST_KEY_NAME, KeyProvider.options(conf));
157    keyProvider.flush();
158    keyProvider.close();
159  }

199  @Before
200  public void setUp() throws Exception {
201    TEST_UTIL.getConfiguration().set("dfs.data.transfer.protection", protection);
202    if (StringUtils.isBlank(encryptionAlgorithm) && StringUtils.isBlank(cipherSuite)) {
203      TEST_UTIL.getConfiguration().setBoolean(DFS_ENCRYPT_DATA_TRANSFER_KEY, false);
204    } else {
205      TEST_UTIL.getConfiguration().setBoolean(DFS_ENCRYPT_DATA_TRANSFER_KEY, true);
206    }
207    if (StringUtils.isBlank(encryptionAlgorithm)) {
208      TEST_UTIL.getConfiguration().unset(DFS_DATA_ENCRYPTION_ALGORITHM_KEY);
209    } else {
210      TEST_UTIL.getConfiguration().set(DFS_DATA_ENCRYPTION_ALGORITHM_KEY, encryptionAlgorithm);
211    }
212    if (StringUtils.isBlank(cipherSuite)) {
213      TEST_UTIL.getConfiguration().unset(DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY);
214    } else {
215      TEST_UTIL.getConfiguration().set(DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY, cipherSuite);
216    }
217
218    TEST_UTIL.startMiniDFSCluster(3);
219    FS = TEST_UTIL.getDFSCluster().getFileSystem();
220    testDirOnTestFs = new Path("/" + name.getMethodName().replaceAll("[^0-9a-zA-Z]", "_"));
221    FS.mkdirs(testDirOnTestFs);
222    entryptionTestDirOnTestFs = new Path("/" + testDirOnTestFs.getName() + "_enc");
223    FS.mkdirs(entryptionTestDirOnTestFs);
224    createEncryptionZone();
225  }
226
227  @After

230  }

240  private void test(Path file) throws IOException, InterruptedException, ExecutionException {
241    EventLoop eventLoop = EVENT_LOOP_GROUP.next();
242    FanOutOneBlockAsyncDFSOutput out = FanOutOneBlockAsyncDFSOutputHelper.createOutput(FS, file,
243      true, false, (short) 3, FS.getDefaultBlockSize(), eventLoop, CHANNEL_CLASS);
244    TestFanOutOneBlockAsyncDFSOutput.writeAndVerify(FS, file, out);
245  }
246
247  @Test
248  public void test() throws IOException, InterruptedException, ExecutionException {
249    test(getTestFile());
250    test(getEncryptionTestFile());
251  }
