280  private void seekToTheKey(KeyValue expected, List<KeyValue> kvs, Cell toSeek)
281      throws IOException {
282    // create all seekers
283    List<DataBlockEncoder.EncodedSeeker> encodedSeekers = new ArrayList<>();
284    for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
285      if (encoding.getEncoder() == null) {
286        continue;
287      }
288      DataBlockEncoder encoder = encoding.getEncoder();
289      HFileContext meta = new HFileContextBuilder().withHBaseCheckSum(false)
290        .withIncludesMvcc(false).withIncludesTags(false)
291        .withCompression(Compression.Algorithm.NONE).build();
292      HFileBlockEncodingContext encodingContext = encoder.newDataBlockEncodingContext(conf,
293        encoding, HFILEBLOCK_DUMMY_HEADER, meta);
294      ByteBuffer encodedBuffer = TestDataBlockEncoders.encodeKeyValues(encoding, kvs,
295        encodingContext, this.useOffheapData);
296      DataBlockEncoder.EncodedSeeker seeker =
297        encoder.createSeeker(encoder.newDataBlockDecodingContext(conf, meta));
298      seeker.setCurrentBuffer(new SingleByteBuff(encodedBuffer));
299      encodedSeekers.add(seeker);
300    }
301    // test it!
302    // try a few random seeks
303    checkSeekingConsistency(encodedSeekers, toSeek, expected);
304  }
305
306  private void checkSeekingConsistency(List<DataBlockEncoder.EncodedSeeker> encodedSeekers,
307      Cell keyValue, KeyValue expected) {
308    for (DataBlockEncoder.EncodedSeeker seeker : encodedSeekers) {
309      seeker.seekToKeyInBlock(keyValue, false);
310      Cell keyValue2 = seeker.getCell();
311      assertEquals(expected, keyValue2);
312      seeker.rewind();
313    }
314  }
