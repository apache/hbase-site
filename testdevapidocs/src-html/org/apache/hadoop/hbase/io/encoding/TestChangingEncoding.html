056/**
057 * Tests changing data block encoding settings of a column family.
058 */
059@Category({IOTests.class, LargeTests.class})
060public class TestChangingEncoding {
061  private static final Log LOG = LogFactory.getLog(TestChangingEncoding.class);
062  static final String CF = "EncodingTestCF";
063  static final byte[] CF_BYTES = Bytes.toBytes(CF);
064
065  private static final int NUM_ROWS_PER_BATCH = 100;
066  private static final int NUM_COLS_PER_ROW = 20;
067
068  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
069  private static final Configuration conf = TEST_UTIL.getConfiguration();
070
071  private static final int TIMEOUT_MS = 600000;
072
073  private HColumnDescriptor hcd;
074
075  private TableName tableName;
076  private static final List<DataBlockEncoding> ENCODINGS_TO_ITERATE =
077      createEncodingsToIterate();
078
079  private static final List<DataBlockEncoding> createEncodingsToIterate() {
080    List<DataBlockEncoding> encodings = new ArrayList<>(Arrays.asList(DataBlockEncoding.values()));
081    encodings.add(DataBlockEncoding.NONE);
082    return Collections.unmodifiableList(encodings);
083  }
084
085  /** A zero-based index of the current batch of test data being written */
086  private int numBatchesWritten;
087
088  private void prepareTest(String testId) throws IOException {
089    tableName = TableName.valueOf("test_table_" + testId);
090    HTableDescriptor htd = new HTableDescriptor(tableName);
091    hcd = new HColumnDescriptor(CF);
092    htd.addFamily(hcd);
093    try (Admin admin = TEST_UTIL.getConnection().getAdmin()) {
094      admin.createTable(htd);
095    }
096    numBatchesWritten = 0;
097  }
098
099  @BeforeClass
100  public static void setUpBeforeClass() throws Exception {
101    // Use a small flush size to create more HFiles.
102    conf.setInt(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 1024 * 1024);
103    // ((Log4JLogger)RpcServerImplementation.LOG).getLogger().setLevel(Level.TRACE);
104    // ((Log4JLogger)RpcClient.LOG).getLogger().setLevel(Level.TRACE);
105    TEST_UTIL.startMiniCluster();
106  }
107
108  @AfterClass
109  public static void tearDownAfterClass() throws Exception {
110    TEST_UTIL.shutdownMiniCluster();
111  }
112
113  private static byte[] getRowKey(int batchId, int i) {
114    return Bytes.toBytes("batch" + batchId + "_row" + i);
115  }
116
117  private static byte[] getQualifier(int j) {
118    return Bytes.toBytes("col" + j);
119  }
120
121  private static byte[] getValue(int batchId, int i, int j) {
122    return Bytes.toBytes("value_for_" + Bytes.toString(getRowKey(batchId, i))
123        + "_col" + j);
124  }
125
126  static void writeTestDataBatch(TableName tableName,
127      int batchId) throws Exception {
128    LOG.debug("Writing test data batch " + batchId);
129    List<Put> puts = new ArrayList<>();
130    for (int i = 0; i < NUM_ROWS_PER_BATCH; ++i) {
131      Put put = new Put(getRowKey(batchId, i));
132      for (int j = 0; j < NUM_COLS_PER_ROW; ++j) {
133        put.addColumn(CF_BYTES, getQualifier(j), getValue(batchId, i, j));
134      }
135      put.setDurability(Durability.SKIP_WAL);
136      puts.add(put);
137    }
138    try (Connection conn = ConnectionFactory.createConnection(conf);
139        Table table = conn.getTable(tableName)) {
140      table.put(puts);
141    }
142  }
143
144  static void verifyTestDataBatch(TableName tableName,
145      int batchId) throws Exception {
146    LOG.debug("Verifying test data batch " + batchId);
147    Table table = TEST_UTIL.getConnection().getTable(tableName);
148    for (int i = 0; i < NUM_ROWS_PER_BATCH; ++i) {
149      Get get = new Get(getRowKey(batchId, i));
150      Result result = table.get(get);
151      for (int j = 0; j < NUM_COLS_PER_ROW; ++j) {
152        Cell kv = result.getColumnLatestCell(CF_BYTES, getQualifier(j));
153        assertTrue(CellUtil.matchingValue(kv, getValue(batchId, i, j)));
154      }
155    }
156    table.close();
157  }
158
159  private void writeSomeNewData() throws Exception {
160    writeTestDataBatch(tableName, numBatchesWritten);
161    ++numBatchesWritten;
162  }
163
164  private void verifyAllData() throws Exception {
165    for (int i = 0; i < numBatchesWritten; ++i) {
166      verifyTestDataBatch(tableName, i);
167    }
168  }
169
170  private void setEncodingConf(DataBlockEncoding encoding,
171      boolean onlineChange) throws Exception {
172    LOG.debug("Setting CF encoding to " + encoding + " (ordinal="
173      + encoding.ordinal() + "), onlineChange=" + onlineChange);
174    hcd.setDataBlockEncoding(encoding);
175    try (Admin admin = TEST_UTIL.getConnection().getAdmin()) {
176      if (!onlineChange) {
177        admin.disableTable(tableName);
178      }
179      admin.modifyColumnFamily(tableName, hcd);
180      if (!onlineChange) {
181        admin.enableTable(tableName);
182      }
183    }
184    // This is a unit test, not integration test. So let's
185    // wait for regions out of transition. Otherwise, for online
186    // encoding change, verification phase may be flaky because
187    // regions could be still in transition.
188    TEST_UTIL.waitUntilNoRegionsInTransition(TIMEOUT_MS);
189  }
190
191  @Test(timeout=TIMEOUT_MS)
192  public void testChangingEncoding() throws Exception {
193    prepareTest("ChangingEncoding");
194    for (boolean onlineChange : new boolean[]{false, true}) {
195      for (DataBlockEncoding encoding : ENCODINGS_TO_ITERATE) {
196        setEncodingConf(encoding, onlineChange);
197        writeSomeNewData();
198        verifyAllData();
199      }
200    }
201  }
202
203  @Test(timeout=TIMEOUT_MS)
204  public void testChangingEncodingWithCompaction() throws Exception {
205    prepareTest("ChangingEncodingWithCompaction");
206    for (boolean onlineChange : new boolean[]{false, true}) {
207      for (DataBlockEncoding encoding : ENCODINGS_TO_ITERATE) {
208        setEncodingConf(encoding, onlineChange);
209        writeSomeNewData();
210        verifyAllData();
211        compactAndWait();
212        verifyAllData();
213      }
214    }
215  }
216
217  private void compactAndWait() throws IOException, InterruptedException {
218    LOG.debug("Compacting table " + tableName);
219    HRegionServer rs = TEST_UTIL.getMiniHBaseCluster().getRegionServer(0);
220    Admin admin = TEST_UTIL.getAdmin();
221    admin.majorCompact(tableName);
222
223    // Waiting for the compaction to start, at least .5s.
224    final long maxWaitime = System.currentTimeMillis() + 500;
225    boolean cont;
226    do {
227      cont = rs.compactSplitThread.getCompactionQueueSize() == 0;
228      Threads.sleep(1);
229    } while (cont && System.currentTimeMillis() < maxWaitime);
230
231    while (rs.compactSplitThread.getCompactionQueueSize() > 0) {
232      Threads.sleep(1);
233    }
234    LOG.debug("Compaction queue size reached 0, continuing");
235  }
236
237  @Test
238  public void testCrazyRandomChanges() throws Exception {
239    prepareTest("RandomChanges");
240    Random rand = new Random(2934298742974297L);
241    for (int i = 0; i < 20; ++i) {
242      int encodingOrdinal = rand.nextInt(DataBlockEncoding.values().length);
243      DataBlockEncoding encoding = DataBlockEncoding.values()[encodingOrdinal];
244      setEncodingConf(encoding, rand.nextBoolean());
245      writeSomeNewData();
246      verifyAllData();
247    }
248  }
