103  private HFileBlockEncodingContext getEncodingContext(Compression.Algorithm algo,
104      DataBlockEncoding encoding) {
105    DataBlockEncoder encoder = encoding.getEncoder();
106    HFileContext meta = new HFileContextBuilder()
107                        .withHBaseCheckSum(false)
108                        .withIncludesMvcc(includesMemstoreTS)
109                        .withIncludesTags(includesTags)
110                        .withCompression(algo).build();
111    if (encoder != null) {
112      return encoder.newDataBlockEncodingContext(encoding, HFILEBLOCK_DUMMY_HEADER, meta);
113    } else {
114      return new HFileBlockDefaultEncodingContext(encoding, HFILEBLOCK_DUMMY_HEADER, meta);
115    }
116  }

184  /**
185   * Test seeking while file is encoded.
186   */
187  @Test
188  public void testSeekingOnSample() throws IOException {
189    List<KeyValue> sampleKv = generator.generateTestKeyValues(NUMBER_OF_KV, includesTags);
190
191    // create all seekers
192    List<DataBlockEncoder.EncodedSeeker> encodedSeekers = new ArrayList<>();
193    for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
194      LOG.info("Encoding: " + encoding);
195      DataBlockEncoder encoder = encoding.getEncoder();
196      if (encoder == null) {
197        continue;
198      }
199      LOG.info("Encoder: " + encoder);
200      ByteBuffer encodedBuffer = encodeKeyValues(encoding, sampleKv,
201          getEncodingContext(Compression.Algorithm.NONE, encoding), this.useOffheapData);
202      HFileContext meta = new HFileContextBuilder()
203                          .withHBaseCheckSum(false)
204                          .withIncludesMvcc(includesMemstoreTS)
205                          .withIncludesTags(includesTags)
206                          .withCompression(Compression.Algorithm.NONE)
207                          .build();
208      DataBlockEncoder.EncodedSeeker seeker = encoder.createSeeker(CellComparatorImpl.COMPARATOR,
209          encoder.newDataBlockDecodingContext(meta));
210      seeker.setCurrentBuffer(new SingleByteBuff(encodedBuffer));
211      encodedSeekers.add(seeker);
212    }
213    LOG.info("Testing it!");
214    // test it!
215    // try a few random seeks
216    for (boolean seekBefore : new boolean[] { false, true }) {
217      for (int i = 0; i < NUM_RANDOM_SEEKS; ++i) {
218        int keyValueId;
219        if (!seekBefore) {
220          keyValueId = randomizer.nextInt(sampleKv.size());
221        } else {
222          keyValueId = randomizer.nextInt(sampleKv.size() - 1) + 1;
223        }
224
225        KeyValue keyValue = sampleKv.get(keyValueId);
226        checkSeekingConsistency(encodedSeekers, seekBefore, keyValue);
227      }
228    }
229
230    // check edge cases
231    LOG.info("Checking edge cases");
232    checkSeekingConsistency(encodedSeekers, false, sampleKv.get(0));
233    for (boolean seekBefore : new boolean[] { false, true }) {
234      checkSeekingConsistency(encodedSeekers, seekBefore, sampleKv.get(sampleKv.size() - 1));
235      KeyValue midKv = sampleKv.get(sampleKv.size() / 2);
236      Cell lastMidKv =PrivateCellUtil.createLastOnRowCol(midKv);
237      checkSeekingConsistency(encodedSeekers, seekBefore, lastMidKv);
238    }
239    LOG.info("Done");
240  }

242  static ByteBuffer encodeKeyValues(DataBlockEncoding encoding, List<KeyValue> kvs,
243      HFileBlockEncodingContext encodingContext, boolean useOffheapData) throws IOException {
244    DataBlockEncoder encoder = encoding.getEncoder();
245    ByteArrayOutputStream baos = new ByteArrayOutputStream();
246    baos.write(HFILEBLOCK_DUMMY_HEADER);
247    DataOutputStream dos = new DataOutputStream(baos);
248    encoder.startBlockEncoding(encodingContext, dos);
249    for (KeyValue kv : kvs) {
250      encoder.encode(kv, encodingContext, dos);
251    }
252    encoder.endBlockEncoding(encodingContext, dos, baos.getBuffer());
253    byte[] encodedData = new byte[baos.size() - ENCODED_DATA_OFFSET];
254    System.arraycopy(baos.toByteArray(), ENCODED_DATA_OFFSET, encodedData, 0, encodedData.length);
255    if (useOffheapData) {
256      ByteBuffer bb = ByteBuffer.allocateDirect(encodedData.length);
257      bb.put(encodedData);
258      bb.rewind();
259      return bb;
260    }
261    return ByteBuffer.wrap(encodedData);
262  }

264  @Test
265  public void testNextOnSample() throws IOException {
266    List<KeyValue> sampleKv = generator.generateTestKeyValues(NUMBER_OF_KV, includesTags);
267
268    for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
269      if (encoding.getEncoder() == null) {
270        continue;
271      }
272      DataBlockEncoder encoder = encoding.getEncoder();
273      ByteBuffer encodedBuffer = encodeKeyValues(encoding, sampleKv,
274          getEncodingContext(Compression.Algorithm.NONE, encoding), this.useOffheapData);
275      HFileContext meta = new HFileContextBuilder()
276                          .withHBaseCheckSum(false)
277                          .withIncludesMvcc(includesMemstoreTS)
278                          .withIncludesTags(includesTags)
279                          .withCompression(Compression.Algorithm.NONE)
280                          .build();
281      DataBlockEncoder.EncodedSeeker seeker = encoder.createSeeker(CellComparatorImpl.COMPARATOR,
282          encoder.newDataBlockDecodingContext(meta));
283      seeker.setCurrentBuffer(new SingleByteBuff(encodedBuffer));
284      int i = 0;
285      do {
286        KeyValue expectedKeyValue = sampleKv.get(i);
287        Cell cell = seeker.getCell();
288        if (PrivateCellUtil.compareKeyIgnoresMvcc(CellComparatorImpl.COMPARATOR, expectedKeyValue,
289          cell) != 0) {
290          int commonPrefix = PrivateCellUtil
291              .findCommonPrefixInFlatKey(expectedKeyValue, cell, false, true);
292          fail(String.format("next() produces wrong results "
293              + "encoder: %s i: %d commonPrefix: %d" + "\n expected %s\n actual      %s", encoder
294              .toString(), i, commonPrefix, Bytes.toStringBinary(expectedKeyValue.getBuffer(),
295              expectedKeyValue.getKeyOffset(), expectedKeyValue.getKeyLength()), CellUtil.toString(
296              cell, false)));
297        }
298        i++;
299      } while (seeker.next());
300    }
301  }

303  /**
304   * Test whether the decompression of first key is implemented correctly.
305   * @throws IOException
306   */
307  @Test
308  public void testFirstKeyInBlockOnSample() throws IOException {
309    List<KeyValue> sampleKv = generator.generateTestKeyValues(NUMBER_OF_KV, includesTags);
310
311    for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
312      if (encoding.getEncoder() == null) {
313        continue;
314      }
315      DataBlockEncoder encoder = encoding.getEncoder();
316      ByteBuffer encodedBuffer = encodeKeyValues(encoding, sampleKv,
317          getEncodingContext(Compression.Algorithm.NONE, encoding), this.useOffheapData);
318      Cell key = encoder.getFirstKeyCellInBlock(new SingleByteBuff(encodedBuffer));
319      KeyValue firstKv = sampleKv.get(0);
320      if (0 != PrivateCellUtil.compareKeyIgnoresMvcc(CellComparatorImpl.COMPARATOR, key, firstKv)) {
321        int commonPrefix = PrivateCellUtil.findCommonPrefixInFlatKey(key, firstKv, false, true);
322        fail(String.format("Bug in '%s' commonPrefix %d", encoder.toString(), commonPrefix));
323      }
324    }
325  }

327  private void checkSeekingConsistency(List<DataBlockEncoder.EncodedSeeker> encodedSeekers,
328      boolean seekBefore, Cell keyValue) {
329    Cell expectedKeyValue = null;
330    ByteBuffer expectedKey = null;
331    ByteBuffer expectedValue = null;
332    for (DataBlockEncoder.EncodedSeeker seeker : encodedSeekers) {
333      seeker.seekToKeyInBlock(keyValue, seekBefore);
334      seeker.rewind();
335
336      Cell actualKeyValue = seeker.getCell();
337      ByteBuffer actualKey = null;
338      actualKey = ByteBuffer.wrap(((KeyValue) seeker.getKey()).getKey());
339      ByteBuffer actualValue = seeker.getValueShallowCopy();
340
341      if (expectedKeyValue != null) {
342        assertTrue(CellUtil.equals(expectedKeyValue, actualKeyValue));
343      } else {
344        expectedKeyValue = actualKeyValue;
345      }
346
347      if (expectedKey != null) {
348        assertEquals(expectedKey, actualKey);
349      } else {
350        expectedKey = actualKey;
351      }
352
353      if (expectedValue != null) {
354        assertEquals(expectedValue, actualValue);
355      } else {
356        expectedValue = actualValue;
357      }
358    }
359  }

361  private void testEncodersOnDataset(List<KeyValue> kvList, boolean includesMemstoreTS,
362      boolean includesTags) throws IOException {
363    ByteBuffer unencodedDataBuf = RedundantKVGenerator.convertKvToByteBuffer(kvList,
364        includesMemstoreTS);
365    HFileContext fileContext = new HFileContextBuilder().withIncludesMvcc(includesMemstoreTS)
366        .withIncludesTags(includesTags).build();
367    for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
368      DataBlockEncoder encoder = encoding.getEncoder();
369      if (encoder == null) {
370        continue;
371      }
372      HFileBlockEncodingContext encodingContext = new HFileBlockDefaultEncodingContext(encoding,
373          HFILEBLOCK_DUMMY_HEADER, fileContext);
374
375      ByteArrayOutputStream baos = new ByteArrayOutputStream();
376      baos.write(HFILEBLOCK_DUMMY_HEADER);
377      DataOutputStream dos = new DataOutputStream(baos);
378      encoder.startBlockEncoding(encodingContext, dos);
379      for (KeyValue kv : kvList) {
380        encoder.encode(kv, encodingContext, dos);
381      }
382      encoder.endBlockEncoding(encodingContext, dos, baos.getBuffer());
383      byte[] encodedData = baos.toByteArray();
384
385      testAlgorithm(encodedData, unencodedDataBuf, encoder);
386    }
387  }

389  @Test
390  public void testZeroByte() throws IOException {
391    List<KeyValue> kvList = new ArrayList<>();
392    byte[] row = Bytes.toBytes("abcd");
393    byte[] family = new byte[] { 'f' };
394    byte[] qualifier0 = new byte[] { 'b' };
395    byte[] qualifier1 = new byte[] { 'c' };
396    byte[] value0 = new byte[] { 'd' };
397    byte[] value1 = new byte[] { 0x00 };
398    if (includesTags) {
399      kvList.add(new KeyValue(row, family, qualifier0, 0, value0,
400          new Tag[] { new ArrayBackedTag((byte) 1, "value1") }));
401      kvList.add(new KeyValue(row, family, qualifier1, 0, value1,
402          new Tag[] { new ArrayBackedTag((byte) 1, "value1") }));
403    } else {
404      kvList.add(new KeyValue(row, family, qualifier0, 0, Type.Put, value0));
405      kvList.add(new KeyValue(row, family, qualifier1, 0, Type.Put, value1));
406    }
407    testEncodersOnDataset(kvList, includesMemstoreTS, includesTags);
408  }

410  private void testAlgorithm(byte[] encodedData, ByteBuffer unencodedDataBuf,
411      DataBlockEncoder encoder) throws IOException {
412    // decode
413    ByteArrayInputStream bais = new ByteArrayInputStream(encodedData, ENCODED_DATA_OFFSET,
414        encodedData.length - ENCODED_DATA_OFFSET);
415    DataInputStream dis = new DataInputStream(bais);
416    ByteBuffer actualDataset;
417    HFileContext meta = new HFileContextBuilder().withHBaseCheckSum(false)
418        .withIncludesMvcc(includesMemstoreTS).withIncludesTags(includesTags)
419        .withCompression(Compression.Algorithm.NONE).build();
420    actualDataset = encoder.decodeKeyValues(dis, encoder.newDataBlockDecodingContext(meta));
421    actualDataset.rewind();
422
423    // this is because in case of prefix tree the decoded stream will not have
424    // the
425    // mvcc in it.
426    assertEquals("Encoding -> decoding gives different results for " + encoder,
427        Bytes.toStringBinary(unencodedDataBuf), Bytes.toStringBinary(actualDataset));
428  }
