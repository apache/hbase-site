017package org.apache.hadoop.hbase.io.compress.zstd;
018
019import static org.junit.Assert.assertNotNull;
020import static org.junit.Assert.assertTrue;
021
022import java.io.File;
023import java.io.FileOutputStream;
024import java.io.IOException;
025import java.util.Random;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.fs.CommonConfigurationKeys;
028import org.apache.hadoop.hbase.HBaseClassTestRule;
029import org.apache.hadoop.hbase.io.compress.CompressionTestBase;
030import org.apache.hadoop.hbase.io.compress.DictionaryCache;
031import org.apache.hadoop.hbase.testclassification.SmallTests;
032import org.apache.hadoop.hbase.util.RandomDistribution;
033import org.junit.BeforeClass;
034import org.junit.ClassRule;
035import org.junit.Test;
036import org.junit.experimental.categories.Category;
037
038@Category(SmallTests.class)
039public class TestZstdDictionary extends CompressionTestBase {
040
041  @ClassRule
042  public static final HBaseClassTestRule CLASS_RULE =
043    HBaseClassTestRule.forClass(TestZstdDictionary.class);
044
045  private static final String DICTIONARY_PATH = DictionaryCache.RESOURCE_SCHEME + "zstd.test.dict";
046  // zstd.test.data compressed with zstd.test.dict at level 3 will produce a result of
047  // 358555 bytes
048  private static final int EXPECTED_COMPRESSED_SIZE = 358555;
049
050  private static byte[] TEST_DATA;
051
052  @BeforeClass
053  public static void setUp() throws Exception {
054    Configuration conf = new Configuration();
055    TEST_DATA = DictionaryCache.loadFromResource(conf,
056      DictionaryCache.RESOURCE_SCHEME + "zstd.test.data", /* maxSize */ 1024*1024);
057    assertNotNull("Failed to load test data", TEST_DATA);
058  }
059
060  @Test
061  public void test() throws Exception {
062    Configuration conf = new Configuration();
063    conf.setInt(CommonConfigurationKeys.IO_COMPRESSION_CODEC_ZSTD_LEVEL_KEY, 3);
064    conf.set(ZstdCodec.ZSTD_DICTIONARY_KEY, DICTIONARY_PATH);
065    ZstdCodec codec = new ZstdCodec();
066    codec.setConf(conf);
067    codecTest(codec, new byte[][] { TEST_DATA }, EXPECTED_COMPRESSED_SIZE);
068    // Assert that the dictionary was actually loaded
069    assertTrue("Dictionary was not loaded by codec", DictionaryCache.contains(DICTIONARY_PATH));
070  }
071
072  //
073  // For generating the test data in src/test/resources/
074  //
075
076  public static void main(String[] args) throws IOException {
077    // Write 1000 1k blocks for training to the specified file
078    // Train with:
079    //   zstd --train -B1024 -o <dictionary_file> <input_file>
080    if (args.length < 1) {
081      System.err.println("Usage: TestZstdCodec <outFile>");
082      System.exit(-1);
083    }
084    final RandomDistribution.DiscreteRNG rng =
085      new RandomDistribution.Zipf(new Random(), 0, Byte.MAX_VALUE, 2);
086    final File outFile = new File(args[0]);
087    final byte[] buffer = new byte[1024];
088    System.out.println("Generating " + outFile);
089    try (FileOutputStream os = new FileOutputStream(outFile)) {
090      for (int i = 0; i < 1000; i++) {
091        fill(rng, buffer);
092        os.write(buffer);
093      }
094    }
095    System.out.println("Done");
096  }
