047  @Test
048  public void testAllocateByteBuffToReadInto() {
049    int maxBuffersInPool = 10;
050    int bufSize = 6 * 1024;
051    ByteBuffAllocator alloc = new ByteBuffAllocator(true, maxBuffersInPool, bufSize, bufSize / 6);
052    assertEquals(0, alloc.getUsedBufferCount());
053
054    ByteBuff buff = alloc.allocate(10 * bufSize);
055    assertEquals(61440, alloc.getPoolAllocationBytes());
056    assertEquals(0, alloc.getHeapAllocationBytes());
057    assertEquals(10, alloc.getUsedBufferCount());
058    buff.release();
059    // When the request size is less than 1/6th of the pool buffer size. We should use on demand
060    // created on heap Buffer
061    buff = alloc.allocate(200);
062    assertTrue(buff.hasArray());
063    assertEquals(maxBuffersInPool, alloc.getFreeBufferCount());
064    assertEquals(maxBuffersInPool, alloc.getTotalBufferCount());
065    assertEquals(61440, alloc.getPoolAllocationBytes());
066    assertEquals(200, alloc.getHeapAllocationBytes());
067    assertEquals(10, alloc.getUsedBufferCount());
068    buff.release();
069    // When the request size is > 1/6th of the pool buffer size.
070    buff = alloc.allocate(1024);
071    assertFalse(buff.hasArray());
072    assertEquals(maxBuffersInPool - 1, alloc.getFreeBufferCount());
073    assertEquals(67584, alloc.getPoolAllocationBytes());
074    assertEquals(200, alloc.getHeapAllocationBytes());
075    assertEquals(10, alloc.getUsedBufferCount());
076    buff.release();// ByteBuff Recycler#free should put back the BB to pool.
077    assertEquals(maxBuffersInPool, alloc.getFreeBufferCount());
078    // Request size> pool buffer size
079    buff = alloc.allocate(7 * 1024);
080    assertFalse(buff.hasArray());
081    assertTrue(buff instanceof MultiByteBuff);
082    ByteBuffer[] bbs = buff.nioByteBuffers();
083    assertEquals(2, bbs.length);
084    assertTrue(bbs[0].isDirect());
085    assertTrue(bbs[1].isDirect());
086    assertEquals(6 * 1024, bbs[0].limit());
087    assertEquals(1024, bbs[1].limit());
088    assertEquals(maxBuffersInPool - 2, alloc.getFreeBufferCount());
089    assertEquals(79872, alloc.getPoolAllocationBytes());
090    assertEquals(200, alloc.getHeapAllocationBytes());
091    assertEquals(10, alloc.getUsedBufferCount());
092    buff.release();
093    assertEquals(maxBuffersInPool, alloc.getFreeBufferCount());
094
095    buff = alloc.allocate(6 * 1024 + 200);
096    assertFalse(buff.hasArray());
097    assertTrue(buff instanceof MultiByteBuff);
098    bbs = buff.nioByteBuffers();
099    assertEquals(2, bbs.length);
100    assertTrue(bbs[0].isDirect());
101    assertFalse(bbs[1].isDirect());
102    assertEquals(6 * 1024, bbs[0].limit());
103    assertEquals(200, bbs[1].limit());
104    assertEquals(maxBuffersInPool - 1, alloc.getFreeBufferCount());
105    assertEquals(86016, alloc.getPoolAllocationBytes());
106    assertEquals(400, alloc.getHeapAllocationBytes());
107    assertEquals(10, alloc.getUsedBufferCount());
108    buff.release();
109    assertEquals(maxBuffersInPool, alloc.getFreeBufferCount());
110
111    alloc.allocate(bufSize * (maxBuffersInPool - 1));
112    assertEquals(141312, alloc.getPoolAllocationBytes());
113    assertEquals(400, alloc.getHeapAllocationBytes());
114    assertEquals(10, alloc.getUsedBufferCount());
115
116    buff = alloc.allocate(20 * 1024);
117    assertFalse(buff.hasArray());
118    assertTrue(buff instanceof MultiByteBuff);
119    bbs = buff.nioByteBuffers();
120    assertEquals(2, bbs.length);
121    assertTrue(bbs[0].isDirect());
122    assertFalse(bbs[1].isDirect());
123    assertEquals(6 * 1024, bbs[0].limit());
124    assertEquals(14 * 1024, bbs[1].limit());
125    assertEquals(0, alloc.getFreeBufferCount());
126    assertEquals(147456, alloc.getPoolAllocationBytes());
127    assertEquals(14736, alloc.getHeapAllocationBytes());
128    assertEquals(10, alloc.getUsedBufferCount());
129
130    buff.release();
131    assertEquals(1, alloc.getFreeBufferCount());
132    alloc.allocateOneBuffer();
133    assertEquals(153600, alloc.getPoolAllocationBytes());
134    assertEquals(14736, alloc.getHeapAllocationBytes());
135    assertEquals(10, alloc.getUsedBufferCount());
136
137    buff = alloc.allocate(7 * 1024);
138    assertTrue(buff.hasArray());
139    assertTrue(buff instanceof SingleByteBuff);
140    assertEquals(7 * 1024, buff.nioByteBuffers()[0].limit());
141    assertEquals(153600, alloc.getPoolAllocationBytes());
142    assertEquals(21904, alloc.getHeapAllocationBytes());
143    assertEquals(10, alloc.getUsedBufferCount());
144    buff.release();
145  }

147  @Test
148  public void testNegativeAllocatedSize() {
149    int maxBuffersInPool = 10;
150    ByteBuffAllocator allocator = new ByteBuffAllocator(true, maxBuffersInPool, 6 * 1024, 1024);
151    try {
152      allocator.allocate(-1);
153      fail("Should throw exception when size < 0");

162  @Test
163  public void testAllocateOneBuffer() {
164    // Allocate from on-heap
165    ByteBuffAllocator allocator = ByteBuffAllocator.HEAP;
166    ByteBuff buf = allocator.allocateOneBuffer();
167    assertTrue(buf.hasArray());
168    assertEquals(ByteBuffAllocator.DEFAULT_BUFFER_SIZE, buf.remaining());
169    buf.release();
170
171    // Allocate from off-heap
172    int bufSize = 10;
173    allocator = new ByteBuffAllocator(true, 1, 10, 3);
174    buf = allocator.allocateOneBuffer();
175    assertFalse(buf.hasArray());
176    assertEquals(buf.remaining(), bufSize);
177    // The another one will be allocated from on-heap because the pool has only one ByteBuffer,
178    // and still not be cleaned.
179    ByteBuff buf2 = allocator.allocateOneBuffer();
180    assertTrue(buf2.hasArray());
181    assertEquals(buf2.remaining(), bufSize);
182    // free the first one
183    buf.release();
184    // The next one will be off-heap again.
185    buf = allocator.allocateOneBuffer();
186    assertFalse(buf.hasArray());
187    assertEquals(buf.remaining(), bufSize);
188    buf.release();
189  }

191  @Test
192  public void testReferenceCount() {
193    int bufSize = 64;
194    ByteBuffAllocator alloc = new ByteBuffAllocator(true, 2, bufSize, 3);
195    ByteBuff buf1 = alloc.allocate(bufSize * 2);
196    assertFalse(buf1.hasArray());
197    // The next one will be allocated from heap
198    ByteBuff buf2 = alloc.allocateOneBuffer();
199    assertTrue(buf2.hasArray());
200
201    // duplicate the buf2, if the dup released, buf2 will also be released (SingleByteBuffer)
202    ByteBuff dup2 = buf2.duplicate();
203    dup2.release();
204    assertEquals(0, buf2.refCnt());
205    assertEquals(0, dup2.refCnt());
206    assertEquals(0, alloc.getFreeBufferCount());
207    assertException(dup2::position);
208    assertException(buf2::position);
209
210    // duplicate the buf1, if the dup1 released, buf1 will also be released (MultipleByteBuffer)
211    ByteBuff dup1 = buf1.duplicate();
212    dup1.release();
213    assertEquals(0, buf1.refCnt());
214    assertEquals(0, dup1.refCnt());
215    assertEquals(2, alloc.getFreeBufferCount());
216    assertException(dup1::position);
217    assertException(buf1::position);
218
219    // slice the buf3, if the slice3 released, buf3 will also be released (SingleByteBuffer)
220    ByteBuff buf3 = alloc.allocateOneBuffer();
221    assertFalse(buf3.hasArray());
222    ByteBuff slice3 = buf3.slice();
223    slice3.release();
224    assertEquals(0, buf3.refCnt());
225    assertEquals(0, slice3.refCnt());
226    assertEquals(2, alloc.getFreeBufferCount());
227
228    // slice the buf4, if the slice4 released, buf4 will also be released (MultipleByteBuffer)
229    ByteBuff buf4 = alloc.allocate(bufSize * 2);
230    assertFalse(buf4.hasArray());
231    ByteBuff slice4 = buf4.slice();
232    slice4.release();
233    assertEquals(0, buf4.refCnt());
234    assertEquals(0, slice4.refCnt());
235    assertEquals(2, alloc.getFreeBufferCount());
236
237    // Test multiple reference for the same ByteBuff (SingleByteBuff)
238    ByteBuff buf5 = alloc.allocateOneBuffer();
239    ByteBuff slice5 = buf5.duplicate().duplicate().duplicate().slice().slice();
240    slice5.release();
241    assertEquals(0, buf5.refCnt());
242    assertEquals(0, slice5.refCnt());
243    assertEquals(2, alloc.getFreeBufferCount());
244    assertException(slice5::position);
245    assertException(buf5::position);
246
247    // Test multiple reference for the same ByteBuff (SingleByteBuff)
248    ByteBuff buf6 = alloc.allocate(bufSize >> 2);
249    ByteBuff slice6 = buf6.duplicate().duplicate().duplicate().slice().slice();
250    slice6.release();
251    assertEquals(0, buf6.refCnt());
252    assertEquals(0, slice6.refCnt());
253    assertEquals(2, alloc.getFreeBufferCount());
254
255    // Test retain the parent SingleByteBuff (duplicate)
256    ByteBuff parent = alloc.allocateOneBuffer();
257    ByteBuff child = parent.duplicate();
258    child.retain();
259    parent.release();
260    assertEquals(1, child.refCnt());
261    assertEquals(1, parent.refCnt());
262    assertEquals(1, alloc.getFreeBufferCount());
263    parent.release();
264    assertEquals(0, child.refCnt());
265    assertEquals(0, parent.refCnt());
266    assertEquals(2, alloc.getFreeBufferCount());
267
268    // Test retain parent MultiByteBuff (duplicate)
269    parent = alloc.allocate(bufSize << 1);
270    child = parent.duplicate();
271    child.retain();
272    parent.release();
273    assertEquals(1, child.refCnt());
274    assertEquals(1, parent.refCnt());
275    assertEquals(0, alloc.getFreeBufferCount());
276    parent.release();
277    assertEquals(0, child.refCnt());
278    assertEquals(0, parent.refCnt());
279    assertEquals(2, alloc.getFreeBufferCount());
280
281    // Test retain the parent SingleByteBuff (slice)
282    parent = alloc.allocateOneBuffer();
283    child = parent.slice();
284    child.retain();
285    parent.release();
286    assertEquals(1, child.refCnt());
287    assertEquals(1, parent.refCnt());
288    assertEquals(1, alloc.getFreeBufferCount());
289    parent.release();
290    assertEquals(0, child.refCnt());
291    assertEquals(0, parent.refCnt());
292    assertEquals(2, alloc.getFreeBufferCount());
293
294    // Test retain parent MultiByteBuff (slice)
295    parent = alloc.allocate(bufSize << 1);
296    child = parent.slice();
297    child.retain();
298    parent.release();
299    assertEquals(1, child.refCnt());
300    assertEquals(1, parent.refCnt());
301    assertEquals(0, alloc.getFreeBufferCount());
302    parent.release();
303    assertEquals(0, child.refCnt());
304    assertEquals(0, parent.refCnt());
305    assertEquals(2, alloc.getFreeBufferCount());
306  }

308  @Test
309  public void testReverseRef() {
310    int bufSize = 64;
311    ByteBuffAllocator alloc = new ByteBuffAllocator(true, 1, bufSize, 3);
312    ByteBuff buf1 = alloc.allocate(bufSize);
313    ByteBuff dup1 = buf1.duplicate();
314    assertEquals(1, buf1.refCnt());
315    assertEquals(1, dup1.refCnt());
316    buf1.release();
317    assertEquals(0, buf1.refCnt());
318    assertEquals(0, dup1.refCnt());
319    assertEquals(1, alloc.getFreeBufferCount());
320    assertException(buf1::position);
321    assertException(dup1::position);
322  }

324  @Test
325  public void testByteBuffUnsupportedMethods() {
326    int bufSize = 64;
327    ByteBuffAllocator alloc = new ByteBuffAllocator(true, 1, bufSize, 3);
328    ByteBuff buf = alloc.allocate(bufSize);
329    assertException(() -> buf.retain(2));
330    assertException(() -> buf.release(2));
331    assertException(() -> buf.touch());
332    assertException(() -> buf.touch(new Object()));
333  }

344  @Test
345  public void testDeprecatedConfigs() {
346    Configuration conf = new Configuration();
347    conf.setInt(ByteBuffAllocator.DEPRECATED_MAX_BUFFER_COUNT_KEY, 10);
348    conf.setInt(ByteBuffAllocator.DEPRECATED_BUFFER_SIZE_KEY, 1024);
349    ByteBuffAllocator allocator = ByteBuffAllocator.create(conf, true);
350    Assert.assertEquals(1024, allocator.getBufferSize());
351    Assert.assertEquals(10, allocator.getTotalBufferCount());
352
353    conf = new Configuration();
354    conf.setInt(ByteBuffAllocator.MAX_BUFFER_COUNT_KEY, 11);
355    conf.setInt(ByteBuffAllocator.BUFFER_SIZE_KEY, 2048);
356    allocator = ByteBuffAllocator.create(conf, true);
357    Assert.assertEquals(2048, allocator.getBufferSize());
358    Assert.assertEquals(11, allocator.getTotalBufferCount());
359
360    conf = new Configuration();
361    conf.setBoolean(ByteBuffAllocator.DEPRECATED_ALLOCATOR_POOL_ENABLED_KEY, false);
362    Assert.assertFalse(conf.getBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, true));
363    conf.setBoolean(ByteBuffAllocator.DEPRECATED_ALLOCATOR_POOL_ENABLED_KEY, true);
364    Assert.assertTrue(conf.getBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, false));
365    conf.setBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, true);
366    Assert.assertTrue(conf.getBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, false));
367    conf.setBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, false);
368    Assert.assertFalse(conf.getBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, true));
369  }
