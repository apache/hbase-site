019package org.apache.hadoop.hbase.procedure2;
020
021import java.io.IOException;
022import java.util.concurrent.CountDownLatch;
023import java.util.concurrent.TimeUnit;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027import org.apache.hadoop.hbase.HBaseCommonTestingUtility;
028import org.apache.hadoop.hbase.procedure2.store.NoopProcedureStore;
029import org.apache.hadoop.hbase.testclassification.SmallTests;
030import org.apache.hadoop.hbase.testclassification.MasterTests;
031import org.junit.After;
032import org.junit.Before;
033import org.junit.Test;
034import org.junit.experimental.categories.Category;
035
036import static org.junit.Assert.assertEquals;
037import static org.junit.Assert.assertFalse;
038import static org.junit.Assert.assertTrue;
039
040@Category({MasterTests.class, SmallTests.class})
041public class TestProcedureInMemoryChore {
042  private static final Log LOG = LogFactory.getLog(TestProcedureInMemoryChore.class);
043
044  private static final int PROCEDURE_EXECUTOR_SLOTS = 1;
045
046  private TestProcEnv procEnv;
047  private NoopProcedureStore procStore;
048  private ProcedureExecutor<TestProcEnv> procExecutor;
049
050  private HBaseCommonTestingUtility htu;
051
052  @Before
053  public void setUp() throws IOException {
054    htu = new HBaseCommonTestingUtility();
055
056    procEnv = new TestProcEnv();
057    procStore = new NoopProcedureStore();
058    procExecutor = new ProcedureExecutor(htu.getConfiguration(), procEnv, procStore);
059    procExecutor.testing = new ProcedureExecutor.Testing();
060    procStore.start(PROCEDURE_EXECUTOR_SLOTS);
061    procExecutor.start(PROCEDURE_EXECUTOR_SLOTS, true);
062  }
063
064  @After
065  public void tearDown() throws IOException {
066    procExecutor.stop();
067    procStore.stop(false);
068  }
069
070  @Test
071  public void testChoreAddAndRemove() throws Exception {
072    final int timeoutMSec = 50;
073    final int nCountDown = 5;
074
075    // submit the chore and wait for execution
076    CountDownLatch latch = new CountDownLatch(nCountDown);
077    TestLatchChore chore = new TestLatchChore(timeoutMSec, latch);
078    procExecutor.addChore(chore);
079    latch.await();
080
081    // remove the chore and verify it is no longer executed
082    procExecutor.removeChore(chore);
083    latch = new CountDownLatch(nCountDown);
084    chore.setLatch(latch);
085    latch.await(timeoutMSec * nCountDown, TimeUnit.MILLISECONDS);
086    LOG.info("chore latch count=" + latch.getCount());
087    assertTrue(latch.getCount() > 0);
088  }
089
090  public static class TestLatchChore extends ProcedureInMemoryChore<TestProcEnv> {
091    private CountDownLatch latch;
092
093    public TestLatchChore(final int timeoutMSec, final CountDownLatch latch) {
094      super(timeoutMSec);
095      setLatch(latch);
096    }
097
098    public void setLatch(final CountDownLatch latch) {
099      this.latch = latch;
100    }
101
102    @Override
103    protected void periodicExecute(final TestProcEnv env) {
104      latch.countDown();
105    }
106  }
107
108  private static class TestProcEnv {
109  }
110}
