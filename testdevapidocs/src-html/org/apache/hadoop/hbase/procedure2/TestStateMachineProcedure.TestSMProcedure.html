046  private static final Exception TEST_FAILURE_EXCEPTION = new Exception("test failure") {
047    @Override
048    public boolean equals(final Object other) {
049      if (this == other) return true;
050      if (!(other instanceof Exception)) return false;
051      // we are going to serialize the exception in the test,
052      // so the instance comparison will not match
053      return getMessage().equals(((Exception)other).getMessage());
054    }
055
056    @Override
057    public int hashCode() {
058      return getMessage().hashCode();
059    }
060  };

062  private static final int PROCEDURE_EXECUTOR_SLOTS = 1;
063
064  private ProcedureExecutor<TestProcEnv> procExecutor;
065  private ProcedureStore procStore;
066
067  private HBaseCommonTestingUtility htu;
068  private FileSystem fs;
069  private Path testDir;
070  private Path logDir;

072  @Before
073  public void setUp() throws IOException {
074    htu = new HBaseCommonTestingUtility();
075    testDir = htu.getDataTestDir();
076    fs = testDir.getFileSystem(htu.getConfiguration());
077
078    logDir = new Path(testDir, "proc-logs");
079    procStore = ProcedureTestingUtility.createWalStore(htu.getConfiguration(), logDir);
080    procExecutor = new ProcedureExecutor(htu.getConfiguration(), new TestProcEnv(), procStore);
081    procStore.start(PROCEDURE_EXECUTOR_SLOTS);
082    procExecutor.start(PROCEDURE_EXECUTOR_SLOTS, true);
083  }
084
085  @After
086  public void tearDown() throws IOException {
087    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExecutor, false);
088    assertTrue("expected executor to be running", procExecutor.isRunning());
089
090    procExecutor.stop();
091    procStore.stop(false);
092    fs.delete(logDir, true);
093  }
094
095  @Test
096  public void testAbortStuckProcedure() throws InterruptedException {
097    try {
098      procExecutor.getEnvironment().loop = true;
099      TestSMProcedure proc = new TestSMProcedure();
100      long procId = procExecutor.submitProcedure(proc);
101      Thread.sleep(1000 + (int) (Math.random() * 4001));
102      proc.abort(procExecutor.getEnvironment());
103      ProcedureTestingUtility.waitProcedure(procExecutor, procId);
104      assertEquals(true, proc.isFailed());
105    } finally {
106      procExecutor.getEnvironment().loop = false;
107    }
108  }
109
110  @Test
111  public void testChildOnLastStep() {
112    long procId = procExecutor.submitProcedure(new TestSMProcedure());
113    ProcedureTestingUtility.waitProcedure(procExecutor, procId);
114    assertEquals(3, procExecutor.getEnvironment().execCount.get());
115    assertEquals(0, procExecutor.getEnvironment().rollbackCount.get());
116    ProcedureTestingUtility.assertProcNotFailed(procExecutor, procId);
117  }

121    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExecutor, true);
122    long procId = procExecutor.submitProcedure(new TestSMProcedure());
123    ProcedureTestingUtility.testRecoveryAndDoubleExecution(procExecutor, procId);
124    assertEquals(6, procExecutor.getEnvironment().execCount.get());
125    assertEquals(0, procExecutor.getEnvironment().rollbackCount.get());

127  }

133    ProcedureTestingUtility.waitProcedure(procExecutor, procId);

141  public void testChildOnLastStepWithRollbackDoubleExecution() throws Exception {

152  public enum TestSMProcedureState { STEP_1, STEP_2 }
153
154  public static class TestSMProcedure
155      extends StateMachineProcedure<TestProcEnv, TestSMProcedureState> {
156    protected Flow executeFromState(TestProcEnv env, TestSMProcedureState state) {
157      LOG.info("EXEC " + state + " " + this);
158      env.execCount.incrementAndGet();
159      switch (state) {
160        case STEP_1:
161          if (!env.loop) {
162            setNextState(TestSMProcedureState.STEP_2);
163          }
164          break;
165        case STEP_2:
166          addChildProcedure(new SimpleChildProcedure());
167          return Flow.NO_MORE_STATE;
168      }
169      return Flow.HAS_MORE_STATE;
170    }
171
172    protected void rollbackState(TestProcEnv env, TestSMProcedureState state) {
173      LOG.info("ROLLBACK " + state + " " + this);
174      env.rollbackCount.incrementAndGet();
175    }
176
177    protected TestSMProcedureState getState(int stateId) {
178      return TestSMProcedureState.values()[stateId];
179    }
180
181    protected int getStateId(TestSMProcedureState state) {
182      return state.ordinal();
183    }
184
185    protected TestSMProcedureState getInitialState() {
186      return TestSMProcedureState.STEP_1;
187    }
188  }
189
190  public static class SimpleChildProcedure extends NoopProcedure<TestProcEnv> {
191    protected Procedure[] execute(TestProcEnv env) {
192      LOG.info("EXEC " + this);
193      env.execCount.incrementAndGet();
194      if (env.triggerChildRollback) {
195        setFailure("test-failure", TEST_FAILURE_EXCEPTION);
196      }
197      return null;
198    }
199
200    @Override
201    protected void rollback(TestProcEnv env) {
202      LOG.info("ROLLBACK " + this);
203      env.rollbackCount.incrementAndGet();
204    }
205  }
206
207  public class TestProcEnv {
208    AtomicInteger execCount = new AtomicInteger(0);
209    AtomicInteger rollbackCount = new AtomicInteger(0);
210    boolean triggerChildRollback = false;
211    boolean loop = false;
212  }
