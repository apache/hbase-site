050@Category({ MasterTests.class, MediumTests.class })
051public class TestWALProcedurePrettyPrinter {
052
053  @ClassRule
054  public static final HBaseClassTestRule CLASS_RULE =
055    HBaseClassTestRule.forClass(TestWALProcedurePrettyPrinter.class);
056
057  private static final Logger LOG = LoggerFactory.getLogger(TestWALProcedurePrettyPrinter.class);
058
059  private HBaseCommonTestingUtility htu;
060
061  private RegionProcedureStore store;
062
063  @Before
064  public void setUp() throws IOException {
065    htu = new HBaseCommonTestingUtility();
066    htu.getConfiguration().setBoolean(MemStoreLAB.USEMSLAB_KEY, false);
067    Path testDir = htu.getDataTestDir();
068    CommonFSUtils.setWALRootDir(htu.getConfiguration(), testDir);
069    store = RegionProcedureStoreTestHelper.createStore(htu.getConfiguration(), new LoadCounter());
070  }
071
072  @After
073  public void tearDown() throws IOException {
074    store.stop(true);
075    htu.cleanupTestDir();
076  }
077
078  @Test
079  public void test() throws Exception {
080    List<RegionProcedureStoreTestProcedure> procs = new ArrayList<>();
081    for (int i = 0; i < 10; i++) {
082      RegionProcedureStoreTestProcedure proc = new RegionProcedureStoreTestProcedure();
083      store.insert(proc, null);
084      procs.add(proc);
085    }
086    store.region.flush(true);
087    for (int i = 0; i < 5; i++) {
088      store.delete(procs.get(i).getProcId());
089    }
090    store.cleanup();
091    Path walParentDir = new Path(htu.getDataTestDir(),
092      RegionProcedureStore.MASTER_PROCEDURE_DIR + "/" + HConstants.HREGION_LOGDIR_NAME);
093    FileSystem fs = walParentDir.getFileSystem(htu.getConfiguration());
094    Path walDir = fs.listStatus(walParentDir)[0].getPath();
095    Path walFile = fs.listStatus(walDir)[0].getPath();
096    store.walRoller.requestRollAll();
097    store.walRoller.waitUntilWalRollFinished();
098    ByteArrayOutputStream bos = new ByteArrayOutputStream();
099    PrintStream out = new PrintStream(bos);
100    WALProcedurePrettyPrinter printer = new WALProcedurePrettyPrinter(out);
101    assertEquals(0, ToolRunner.run(htu.getConfiguration(), printer,
102      new String[] { fs.makeQualified(walFile).toString() }));
103    try (BufferedReader reader = new BufferedReader(
104      new InputStreamReader(new ByteArrayInputStream(bos.toByteArray()), StandardCharsets.UTF_8))) {
105      long inserted = 0;
106      long markedDeleted = 0;
107      long deleted = 0;
108      for (;;) {
109        String line = reader.readLine();
110        LOG.info(line);
111        if (line == null) {
112          break;
113        }
114        if (line.startsWith("\t")) {
115          if (line.startsWith("\tpid=")) {
116            inserted++;
117          } else {
118            assertEquals("\tmark deleted", line);
119            markedDeleted++;
120          }
121        } else if (line.contains("type=DeleteFamily")) {
122          deleted++;
123        }
124      }
125      assertEquals(10, inserted);
126      assertEquals(5, markedDeleted);
127      assertEquals(5, deleted);
128    }
129  }
130}
