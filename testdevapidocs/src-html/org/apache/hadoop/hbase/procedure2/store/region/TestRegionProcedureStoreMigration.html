058@SuppressWarnings("deprecation")
059@Category({ MasterTests.class, MediumTests.class })
060public class TestRegionProcedureStoreMigration {
061
062  @ClassRule
063  public static final HBaseClassTestRule CLASS_RULE =
064    HBaseClassTestRule.forClass(TestRegionProcedureStoreMigration.class);
065
066  private HBaseCommonTestingUtility htu;
067
068  private RegionProcedureStore store;
069
070  private WALProcedureStore walStore;
071
072  private ChoreService choreService;
073
074  private DirScanPool cleanerPool;
075
076  @Before
077  public void setUp() throws IOException {
078    htu = new HBaseCommonTestingUtility();
079    Configuration conf = htu.getConfiguration();
080    conf.setBoolean(MemStoreLAB.USEMSLAB_KEY, false);
081    // Runs on local filesystem. Test does not need sync. Turn off checks.
082    htu.getConfiguration().setBoolean(CommonFSUtils.UNSAFE_STREAM_CAPABILITY_ENFORCE, false);
083    Path testDir = htu.getDataTestDir();
084    CommonFSUtils.setWALRootDir(conf, testDir);
085    walStore = new WALProcedureStore(conf, new LeaseRecovery() {
086
087      @Override
088      public void recoverFileLease(FileSystem fs, Path path) throws IOException {
089      }
090    });
091    walStore.start(1);
092    walStore.recoverLease();
093    walStore.load(new LoadCounter());
094    choreService = new ChoreService(getClass().getSimpleName());
095    cleanerPool = new DirScanPool(htu.getConfiguration());
096  }
097
098  @After
099  public void tearDown() throws IOException {
100    if (store != null) {
101      store.stop(true);
102    }
103    walStore.stop(true);
104    cleanerPool.shutdownNow();
105    choreService.shutdown();
106    htu.cleanupTestDir();
107  }
108
109  @Test
110  public void test() throws IOException {
111    List<RegionProcedureStoreTestProcedure> procs = new ArrayList<>();
112    for (int i = 0; i < 10; i++) {
113      RegionProcedureStoreTestProcedure proc = new RegionProcedureStoreTestProcedure();
114      walStore.insert(proc, null);
115      procs.add(proc);
116    }
117    for (int i = 5; i < 10; i++) {
118      walStore.delete(procs.get(i).getProcId());
119    }
120    walStore.stop(true);
121    SortedSet<RegionProcedureStoreTestProcedure> loadedProcs =
122      new TreeSet<>((p1, p2) -> Long.compare(p1.getProcId(), p2.getProcId()));
123    MutableLong maxProcIdSet = new MutableLong(0);
124    store = RegionProcedureStoreTestHelper.createStore(htu.getConfiguration(), choreService,
125      cleanerPool, new ProcedureLoader() {
126
127        @Override
128        public void setMaxProcId(long maxProcId) {
129          maxProcIdSet.setValue(maxProcId);
130        }
131
132        @Override
133        public void load(ProcedureIterator procIter) throws IOException {
134          while (procIter.hasNext()) {
135            RegionProcedureStoreTestProcedure proc =
136              (RegionProcedureStoreTestProcedure) procIter.next();
137            loadedProcs.add(proc);
138          }
139        }
140
141        @Override
142        public void handleCorrupted(ProcedureIterator procIter) throws IOException {
143          if (procIter.hasNext()) {
144            fail("Found corrupted procedures");
145          }
146        }
147      });
148    assertEquals(10, maxProcIdSet.longValue());
149    assertEquals(5, loadedProcs.size());
150    int procId = 1;
151    for (RegionProcedureStoreTestProcedure proc : loadedProcs) {
152      assertEquals(procId, proc.getProcId());
153      procId++;
154    }
155    Path testDir = htu.getDataTestDir();
156    FileSystem fs = testDir.getFileSystem(htu.getConfiguration());
157    Path oldProcWALDir = new Path(testDir, WALProcedureStore.MASTER_PROCEDURE_LOGDIR);
158    // make sure the old proc wal directory has been deleted.
159    assertFalse(fs.exists(oldProcWALDir));
160  }
161
162  @Test
163  public void testMigrateWithUnsupportedProcedures() throws IOException {
164    AssignProcedure assignProc = new AssignProcedure();
165    assignProc.setProcId(1L);
166    assignProc.setRegionInfo(RegionInfoBuilder.newBuilder(TableName.valueOf("table")).build());
167    walStore.insert(assignProc, null);
168    walStore.stop(true);
169
170    try {
171      store = RegionProcedureStoreTestHelper.createStore(htu.getConfiguration(), choreService,
172        cleanerPool, new LoadCounter());
173      fail("Should fail since AssignProcedure is not supported");
174    } catch (HBaseIOException e) {
175      assertThat(e.getMessage(), startsWith("Unsupported"));
176    }
177  }
