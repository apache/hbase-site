040public class ProcedureWALPerformanceEvaluation extends AbstractHBaseTool {
041  protected static final HBaseCommonTestingUtility UTIL = new HBaseCommonTestingUtility();
042
043  // Command line options and defaults.
044  public static int DEFAULT_NUM_THREADS = 20;
045  public static Option NUM_THREADS_OPTION = new Option("threads", true,
046      "Number of parallel threads which will write insert/updates/deletes to WAL. Default: "
047      + DEFAULT_NUM_THREADS);
048  public static int DEFAULT_NUM_PROCS = 1000000;  // 1M
049  public static Option NUM_PROCS_OPTION = new Option("procs", true,
050      "Total number of procedures. Each procedure writes one insert and one update. Default: "
051      + DEFAULT_NUM_PROCS);
052  public static int DEFAULT_NUM_WALS = 0;
053  public static Option NUM_WALS_OPTION = new Option("wals", true,
054      "Number of WALs to write. If -ve or 0, uses " + WALProcedureStore.ROLL_THRESHOLD_CONF_KEY +
055          " conf to roll the logs. Default: " + DEFAULT_NUM_WALS);
056  public static int DEFAULT_STATE_SIZE = 1024;  // 1KB
057  public static Option STATE_SIZE_OPTION = new Option("size", true,
058      "Size of serialized state in bytes to write on update. Default: " + DEFAULT_STATE_SIZE
059          + "bytes");
060  public static Option SYNC_OPTION = new Option("sync", true,
061      "Type of sync to use when writing WAL contents to file system. Accepted values: hflush, "
062          + "hsync, nosync. Default: hflush");
063  public static String DEFAULT_SYNC_OPTION = "hflush";
064
065  public int numThreads;
066  public long numProcs;
067  public long numProcsPerWal = Long.MAX_VALUE;  // never roll wall based on this value.
068  public int numWals;
069  public String syncType;
070  public int stateSize;
071  static byte[] serializedState;
072  private WALProcedureStore store;
073
074  /** Used by {@link Worker}. */
075  private AtomicLong procIds = new AtomicLong(0);
076  private AtomicBoolean workersFailed = new AtomicBoolean(false);
077  // Timeout for worker threads.
078  private static final int WORKER_THREADS_TIMEOUT_SEC = 600;  // in seconds
079
080  // Non-default configurations.
081  private void setupConf() {
082    conf.setBoolean(WALProcedureStore.USE_HSYNC_CONF_KEY, "hsync".equals(syncType));
083    if (numWals > 0) {
084      conf.setLong(WALProcedureStore.ROLL_THRESHOLD_CONF_KEY, Long.MAX_VALUE);
085      numProcsPerWal = numProcs / numWals;
086    }
087  }
088
089  private void setupProcedureStore() throws IOException {
090    Path testDir = UTIL.getDataTestDir();
091    FileSystem fs = testDir.getFileSystem(conf);
092    Path logDir = new Path(testDir, "proc-logs");
093    System.out.println("Logs directory : " + logDir.toString());
094    fs.delete(logDir, true);
095    if ("nosync".equals(syncType)) {
096      store = new NoSyncWalProcedureStore(conf, fs, logDir);
097    } else {
098      store = ProcedureTestingUtility.createWalStore(conf, fs, logDir);
099    }
100    store.start(numThreads);
101    store.recoverLease();
102    store.load(new ProcedureTestingUtility.LoadCounter());
103    System.out.println("Starting new log : "
104        + store.getActiveLogs().get(store.getActiveLogs().size() - 1));
105  }
106
107  private void tearDownProcedureStore() {
108    store.stop(false);
109    try {
110      store.getFileSystem().delete(store.getLogDir(), true);
111    } catch (IOException e) {
112      System.err.println("Error: Couldn't delete log dir. You can delete it manually to free up "
113          + "disk space. Location: " + store.getLogDir().toString());
114      e.printStackTrace();
115    }
116  }
117
118  /**
119   * Processes and validates command line options.
120   */
121  @Override
122  public void processOptions(CommandLine cmd) {
123    numThreads = getOptionAsInt(cmd, NUM_THREADS_OPTION.getOpt(), DEFAULT_NUM_THREADS);
124    numProcs = getOptionAsInt(cmd, NUM_PROCS_OPTION.getOpt(), DEFAULT_NUM_PROCS);
125    numWals = getOptionAsInt(cmd, NUM_WALS_OPTION.getOpt(), DEFAULT_NUM_WALS);
126    syncType = cmd.getOptionValue(SYNC_OPTION.getOpt(), DEFAULT_SYNC_OPTION);
127    assert "hsync".equals(syncType) || "hflush".equals(syncType) || "nosync".equals(syncType):
128        "sync argument can only accept one of these three values: hsync, hflush, nosync";
129    stateSize = getOptionAsInt(cmd, STATE_SIZE_OPTION.getOpt(), DEFAULT_STATE_SIZE);
130    serializedState = new byte[stateSize];
131    setupConf();
132  }
133
134  @Override
135  public void addOptions() {
136    addOption(NUM_THREADS_OPTION);
137    addOption(NUM_PROCS_OPTION);
138    addOption(NUM_WALS_OPTION);
139    addOption(SYNC_OPTION);
140    addOption(STATE_SIZE_OPTION);
141  }
142
143  @Override
144  public int doWork() {
145    try {
146      setupProcedureStore();
147      ExecutorService executor = Executors.newFixedThreadPool(numThreads);
148      Future<?>[] futures = new Future<?>[numThreads];
149      // Start worker threads.
150      long start = System.currentTimeMillis();
151      for (int i = 0; i < numThreads; i++) {
152        futures[i] = executor.submit(this.new Worker(start));
153      }
154      boolean failure = false;
155      try {
156        for (Future<?> future : futures) {
157          long timeout = start + WORKER_THREADS_TIMEOUT_SEC * 1000 - System.currentTimeMillis();
158          failure |= (future.get(timeout, TimeUnit.MILLISECONDS).equals(EXIT_FAILURE));
159        }
160      } catch (Exception e) {
161        System.err.println("Exception in worker thread.");
162        e.printStackTrace();
163        return EXIT_FAILURE;
164      }
165      executor.shutdown();
166      if (failure) {
167        return EXIT_FAILURE;
168      }
169      long timeTaken = System.currentTimeMillis() - start;
170      System.out.println("******************************************");
171      System.out.println("Num threads    : " + numThreads);
172      System.out.println("Num procedures : " + numProcs);
173      System.out.println("Sync type      : " + syncType);
174      System.out.println("Time taken     : " + (timeTaken / 1000.0f) + "sec");
175      System.out.println("******************************************");
176      return EXIT_SUCCESS;
177    } catch (IOException e) {
178      e.printStackTrace();
179      return EXIT_FAILURE;
180    } finally {
181      tearDownProcedureStore();
182    }
183  }

189  /**
190   * Callable to generate load for wal by inserting/deleting/updating procedures.
191   * If procedure store fails to roll log file (throws IOException), all threads quit, and at
192   * least one returns value of {@link AbstractHBaseTool#EXIT_FAILURE}.
193   */
194  class Worker implements Callable<Integer> {
195    final long start;
196
197    public Worker(long start) {
198      this.start = start;
199    }
200
201    // TODO: Can also collect #procs, time taken by each thread to measure fairness.
202    @Override
203    public Integer call() throws IOException {
204      while (true) {
205        if (workersFailed.get()) {
206          return EXIT_FAILURE;
207        }
208        long procId = procIds.getAndIncrement();
209        if (procId >= numProcs) {
210          break;
211        }
212        if (procId != 0 && procId % 10000 == 0) {
213          long ms = System.currentTimeMillis() - start;
214          System.out.println("Wrote " + procId + " procedures in "
215              + StringUtils.humanTimeDiff(ms));
216        }
217        try{
218          if (procId > 0 && procId % numProcsPerWal == 0) {
219            store.rollWriterForTesting();
220            System.out.println("Starting new log : "
221                + store.getActiveLogs().get(store.getActiveLogs().size() - 1));
222          }
223        } catch (IOException ioe) {
224          // Ask other threads to quit too.
225          workersFailed.set(true);
226          System.err.println("Exception when rolling log file. Current procId = " + procId);
227          ioe.printStackTrace();
228          return EXIT_FAILURE;
229        }
230        ProcedureTestingUtility.TestProcedure proc =
231            new ProcedureTestingUtility.TestProcedure(procId);
232        proc.setData(serializedState);
233        store.insert(proc, null);
234        store.update(proc);
235      }
236      return EXIT_SUCCESS;
237    }
238  }

240  public class NoSyncWalProcedureStore extends WALProcedureStore {
241    public NoSyncWalProcedureStore(final Configuration conf, final FileSystem fs,
242        final Path logDir) {
243      super(conf, fs, logDir, new WALProcedureStore.LeaseRecovery() {
244        @Override
245        public void recoverFileLease(FileSystem fs, Path path) throws IOException {
246          // no-op
247        }
248      });
249    }
250
251    @Override
252    protected long syncSlots(FSDataOutputStream stream, ByteSlot[] slots, int offset, int count)
253        throws IOException {
254      long totalSynced = 0;
255      for (int i = 0; i < count; ++i) {
256        totalSynced += slots[offset + i].size();
257      }
258      return totalSynced;
259    }
260  }

265    tool.run(args);
