039  @Test
040  public void testSeqInsertAndDelete() {
041    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
042    assertTrue(tracker.isEmpty());
043
044    final int MIN_PROC = 1;
045    final int MAX_PROC = 1 << 10;
046
047    // sequential insert
048    for (int i = MIN_PROC; i < MAX_PROC; ++i) {
049      tracker.insert(i);
050
051      // All the proc that we inserted should not be deleted
052      for (int j = MIN_PROC; j <= i; ++j) {
053        assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(j));
054      }
055      // All the proc that are not yet inserted should be result as deleted
056      for (int j = i + 1; j < MAX_PROC; ++j) {
057        assertTrue(tracker.isDeleted(j) != ProcedureStoreTracker.DeleteState.NO);
058      }
059    }
060
061    // sequential delete
062    for (int i = MIN_PROC; i < MAX_PROC; ++i) {
063      tracker.delete(i);
064
065      // All the proc that we deleted should be deleted
066      for (int j = MIN_PROC; j <= i; ++j) {
067        assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(j));
068      }
069      // All the proc that are not yet deleted should be result as not deleted
070      for (int j = i + 1; j < MAX_PROC; ++j) {
071        assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(j));
072      }
073    }
074    assertTrue(tracker.isEmpty());
075  }

077  @Test
078  public void testPartialTracker() {
079    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
080    tracker.setPartialFlag(true);
081
082    // nothing in the tracker, the state is unknown
083    assertTrue(tracker.isEmpty());
084    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(1));
085    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(579));
086
087    // Mark 1 as deleted, now that is a known state
088    tracker.setDeleted(1, true);
089    tracker.dump();
090    assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(1));
091    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(2));
092    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(579));
093
094    // Mark 579 as non-deleted, now that is a known state
095    tracker.setDeleted(579, false);
096    assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(1));
097    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(2));
098    assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(579));
099    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(577));
100    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(580));
101
102    tracker.setDeleted(579, true);
103    tracker.setPartialFlag(false);
104    assertTrue(tracker.isEmpty());
105  }

107  @Test
108  public void testIsTracking() {
109    long[][] procIds = new long[][] {{4, 7}, {1024, 1027}, {8192, 8194}};
110    long[][] checkIds = new long[][] {{2, 8}, {1023, 1025}, {8193, 8191}};
111
112    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
113    for (int i = 0; i < procIds.length; ++i) {
114      long[] seq = procIds[i];
115      tracker.insert(seq[0]);
116      tracker.insert(seq[1]);
117    }
118
119    for (int i = 0; i < procIds.length; ++i) {
120      long[] check = checkIds[i];
121      long[] seq = procIds[i];
122      assertTrue(tracker.isTracking(seq[0], seq[1]));
123      assertTrue(tracker.isTracking(check[0], check[1]));
124      tracker.delete(seq[0]);
125      tracker.delete(seq[1]);
126      assertFalse(tracker.isTracking(seq[0], seq[1]));
127      assertFalse(tracker.isTracking(check[0], check[1]));
128    }
129
130    assertTrue(tracker.isEmpty());
131  }

133  @Test
134  public void testBasicCRUD() {
135    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
136    assertTrue(tracker.isEmpty());
137
138    long[] procs = new long[] { 1, 2, 3, 4, 5, 6 };
139
140    tracker.insert(procs[0]);
141    tracker.insert(procs[1], new long[] { procs[2], procs[3], procs[4] });
142    assertFalse(tracker.isEmpty());
143    assertTrue(tracker.isUpdated());
144
145    tracker.resetUpdates();
146    assertFalse(tracker.isUpdated());
147
148    for (int i = 0; i < 4; ++i) {
149      tracker.update(procs[i]);
150      assertFalse(tracker.isEmpty());
151      assertFalse(tracker.isUpdated());
152    }
153
154    tracker.update(procs[4]);
155    assertFalse(tracker.isEmpty());
156    assertTrue(tracker.isUpdated());
157
158    tracker.update(procs[5]);
159    assertFalse(tracker.isEmpty());
160    assertTrue(tracker.isUpdated());
161
162    for (int i = 0; i < 5; ++i) {
163      tracker.delete(procs[i]);
164      assertFalse(tracker.isEmpty());
165      assertTrue(tracker.isUpdated());
166    }
167    tracker.delete(procs[5]);
168    assertTrue(tracker.isEmpty());
169  }

171  @Test
172  public void testRandLoad() {
173    final int NPROCEDURES = 2500;
174    final int NRUNS = 5000;
175
176    final ProcedureStoreTracker tracker = new ProcedureStoreTracker();
177
178    Random rand = new Random(1);
179    for (int i = 0; i < NRUNS; ++i) {
180      assertTrue(tracker.isEmpty());
181
182      int count = 0;
183      while (count < NPROCEDURES) {
184        long procId = rand.nextLong();
185        if (procId < 1) continue;
186
187        tracker.setDeleted(procId, i % 2 == 0);
188        count++;
189      }
190
191      tracker.reset();
192    }
193  }

195  @Test
196  public void testLoad() {
197    final int MAX_PROCS = 1000;
198    final ProcedureStoreTracker tracker = new ProcedureStoreTracker();
199    for (int numProcs = 1; numProcs < MAX_PROCS; ++numProcs) {
200      for (int start = 1; start <= numProcs; ++start) {
201        assertTrue(tracker.isEmpty());
202
203        LOG.debug("loading " + numProcs + " procs from start=" + start);
204        for (int i = start; i <= numProcs; ++i) {
205          tracker.setDeleted(i, false);
206        }
207        for (int i = 1; i < start; ++i) {
208          tracker.setDeleted(i, false);
209        }
210
211        tracker.reset();
212      }
213    }
214  }

216  @Test
217  public void testDelete() {
218    final ProcedureStoreTracker tracker = new ProcedureStoreTracker();
219
220    long[] procIds = new long[] { 65, 1, 193 };
221    for (int i = 0; i < procIds.length; ++i) {
222      tracker.insert(procIds[i]);
223      tracker.dump();
224    }
225
226    for (int i = 0; i < (64 * 4); ++i) {
227      boolean hasProc = false;
228      for (int j = 0; j < procIds.length; ++j) {
229        if (procIds[j] == i) {
230          hasProc = true;
231          break;
232        }
233      }
234      if (hasProc) {
235        assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(i));
236      } else {
237        assertEquals("procId=" + i, ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(i));
238      }
239    }
