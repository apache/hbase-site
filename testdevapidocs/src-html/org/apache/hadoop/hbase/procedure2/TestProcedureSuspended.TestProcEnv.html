071  @Test(timeout=10000)
072  public void testSuspendWhileHoldingLocks() {
073    final AtomicBoolean lockA = new AtomicBoolean(false);
074    final AtomicBoolean lockB = new AtomicBoolean(false);
075
076    final TestLockProcedure p1keyA = new TestLockProcedure(lockA, "keyA", false, true);
077    final TestLockProcedure p2keyA = new TestLockProcedure(lockA, "keyA", false, true);
078    final TestLockProcedure p3keyB = new TestLockProcedure(lockB, "keyB", false, true);
079
080    procExecutor.submitProcedure(p1keyA);
081    procExecutor.submitProcedure(p2keyA);
082    procExecutor.submitProcedure(p3keyB);
083
084    // first run p1, p3 are able to run p2 is blocked by p1
085    waitAndAssertTimestamp(p1keyA, 1, 1);
086    waitAndAssertTimestamp(p2keyA, 0, -1);
087    waitAndAssertTimestamp(p3keyB, 1, 2);
088    assertEquals(true, lockA.get());
089    assertEquals(true, lockB.get());
090
091    // release p3
092    p3keyB.setThrowSuspend(false);
093    procExecutor.getScheduler().addFront(p3keyB);
094    waitAndAssertTimestamp(p1keyA, 1, 1);
095    waitAndAssertTimestamp(p2keyA, 0, -1);
096    waitAndAssertTimestamp(p3keyB, 2, 3);
097    assertEquals(true, lockA.get());
098
099    // wait until p3 is fully completed
100    ProcedureTestingUtility.waitProcedure(procExecutor, p3keyB);
101    assertEquals(false, lockB.get());
102
103    // rollback p2 and wait until is fully completed
104    p1keyA.setTriggerRollback(true);
105    procExecutor.getScheduler().addFront(p1keyA);
106    ProcedureTestingUtility.waitProcedure(procExecutor, p1keyA);
107
108    // p2 should start and suspend
109    waitAndAssertTimestamp(p1keyA, 4, 60000);
110    waitAndAssertTimestamp(p2keyA, 1, 7);
111    waitAndAssertTimestamp(p3keyB, 2, 3);
112    assertEquals(true, lockA.get());
113
114    // wait until p2 is fully completed
115    p2keyA.setThrowSuspend(false);
116    procExecutor.getScheduler().addFront(p2keyA);
117    ProcedureTestingUtility.waitProcedure(procExecutor, p2keyA);
118    waitAndAssertTimestamp(p1keyA, 4, 60000);
119    waitAndAssertTimestamp(p2keyA, 2, 8);
120    waitAndAssertTimestamp(p3keyB, 2, 3);
121    assertEquals(false, lockA.get());
122    assertEquals(false, lockB.get());
123  }

125  @Test(timeout=10000)
126  public void testYieldWhileHoldingLocks() {
127    final AtomicBoolean lock = new AtomicBoolean(false);
128
129    final TestLockProcedure p1 = new TestLockProcedure(lock, "key", true, false);
130    final TestLockProcedure p2 = new TestLockProcedure(lock, "key", true, false);
131
132    procExecutor.submitProcedure(p1);
133    procExecutor.submitProcedure(p2);
134
135    // try to execute a bunch of yield on p1, p2 should be blocked
136    while (p1.getTimestamps().size() < 100) Threads.sleep(10);
137    assertEquals(0, p2.getTimestamps().size());
138
139    // wait until p1 is completed
140    p1.setThrowYield(false);
141    ProcedureTestingUtility.waitProcedure(procExecutor, p1);
142
143    // try to execute a bunch of yield on p2
144    while (p2.getTimestamps().size() < 100) Threads.sleep(10);
145    assertEquals(p1.getTimestamps().get(p1.getTimestamps().size() - 1).longValue() + 1,
146      p2.getTimestamps().get(0).longValue());
147
148    // wait until p2 is completed
149    p1.setThrowYield(false);
150    ProcedureTestingUtility.waitProcedure(procExecutor, p1);
151  }
152
153  private void waitAndAssertTimestamp(TestLockProcedure proc, int size, int lastTs) {
154    final ArrayList<Long> timestamps = proc.getTimestamps();
155    while (timestamps.size() < size) Threads.sleep(10);
156    LOG.info(proc + " -> " + timestamps);
157    assertEquals(size, timestamps.size());
158    if (size > 0) {
159      assertEquals(lastTs, timestamps.get(timestamps.size() - 1).longValue());
160    }
161  }

163  public static class TestLockProcedure extends Procedure<TestProcEnv> {
164    private final ArrayList<Long> timestamps = new ArrayList<>();
165    private final String key;
166
167    private boolean triggerRollback = false;
168    private boolean throwSuspend = false;
169    private boolean throwYield = false;
170    private AtomicBoolean lock = null;
171    private boolean hasLock = false;
172
173    public TestLockProcedure(final AtomicBoolean lock, final String key,
174        final boolean throwYield, final boolean throwSuspend) {
175      this.lock = lock;
176      this.key = key;
177      this.throwYield = throwYield;
178      this.throwSuspend = throwSuspend;
179    }
180
181    public void setThrowYield(final boolean throwYield) {
182      this.throwYield = throwYield;
183    }
184
185    public void setThrowSuspend(final boolean throwSuspend) {
186      this.throwSuspend = throwSuspend;
187    }
188
189    public void setTriggerRollback(final boolean triggerRollback) {
190      this.triggerRollback = triggerRollback;
191    }
192
193    @Override
194    protected Procedure[] execute(final TestProcEnv env)
195        throws ProcedureYieldException, ProcedureSuspendedException {
196      LOG.info("EXECUTE " + this + " suspend " + (lock != null));
197      timestamps.add(env.nextTimestamp());
198      if (triggerRollback) {
199        setFailure(getClass().getSimpleName(), new Exception("injected failure"));
200      } else if (throwYield) {
201        throw new ProcedureYieldException();
202      } else if (throwSuspend) {
203        throw new ProcedureSuspendedException();
204      }
205      return null;
206    }
207
208    @Override
209    protected void rollback(final TestProcEnv env) {
210      LOG.info("ROLLBACK " + this);
211      timestamps.add(env.nextTimestamp() * 10000);
212    }
213
214    @Override
215    protected LockState acquireLock(final TestProcEnv env) {
216      if ((hasLock = lock.compareAndSet(false, true))) {
217        LOG.info("ACQUIRE LOCK " + this + " " + (hasLock));
218        return LockState.LOCK_ACQUIRED;
219      }
220      return LockState.LOCK_YIELD_WAIT;
221    }
222
223    @Override
224    protected void releaseLock(final TestProcEnv env) {
225      LOG.info("RELEASE LOCK " + this + " " + hasLock);
226      lock.set(false);
227      hasLock = false;
228    }
229
230    @Override
231    protected boolean holdLock(final TestProcEnv env) {
232      return true;
233    }
234
235    @Override
236    protected boolean hasLock(final TestProcEnv env) {
237      return hasLock;
238    }
239
240    public ArrayList<Long> getTimestamps() {
241      return timestamps;
242    }
243
244    @Override
245    protected void toStringClassDetails(StringBuilder builder) {
246      builder.append(getClass().getName());
247      builder.append("(" + key + ")");
248    }
249
250    @Override
251    protected boolean abort(TestProcEnv env) { return false; }
252
253    @Override
254    protected void serializeStateData(final OutputStream stream) throws IOException {
255    }
256
257    @Override
258    protected void deserializeStateData(final InputStream stream) throws IOException {
259    }
260  }
261
262  private static class TestProcEnv {
263    public final AtomicLong timestamp = new AtomicLong(0);
264
265    public long nextTimestamp() {
266      return timestamp.incrementAndGet();
267    }
268  }
