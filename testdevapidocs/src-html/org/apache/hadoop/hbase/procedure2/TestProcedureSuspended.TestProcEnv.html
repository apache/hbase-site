074  @Test
075  public void testSuspendWhileHoldingLocks() {
076    final AtomicBoolean lockA = new AtomicBoolean(false);
077    final AtomicBoolean lockB = new AtomicBoolean(false);
078
079    final TestLockProcedure p1keyA = new TestLockProcedure(lockA, "keyA", false, true);
080    final TestLockProcedure p2keyA = new TestLockProcedure(lockA, "keyA", false, true);
081    final TestLockProcedure p3keyB = new TestLockProcedure(lockB, "keyB", false, true);
082
083    procExecutor.submitProcedure(p1keyA);
084    procExecutor.submitProcedure(p2keyA);
085    procExecutor.submitProcedure(p3keyB);
086
087    // first run p1, p3 are able to run p2 is blocked by p1
088    waitAndAssertTimestamp(p1keyA, 1, 1);
089    waitAndAssertTimestamp(p2keyA, 0, -1);
090    waitAndAssertTimestamp(p3keyB, 1, 2);
091    assertEquals(true, lockA.get());
092    assertEquals(true, lockB.get());
093
094    // release p3
095    p3keyB.setThrowSuspend(false);
096    procExecutor.getScheduler().addFront(p3keyB);
097    waitAndAssertTimestamp(p1keyA, 1, 1);
098    waitAndAssertTimestamp(p2keyA, 0, -1);
099    waitAndAssertTimestamp(p3keyB, 2, 3);
100    assertEquals(true, lockA.get());
101
102    // wait until p3 is fully completed
103    ProcedureTestingUtility.waitProcedure(procExecutor, p3keyB);
104    assertEquals(false, lockB.get());
105
106    // rollback p2 and wait until is fully completed
107    p1keyA.setTriggerRollback(true);
108    procExecutor.getScheduler().addFront(p1keyA);
109    ProcedureTestingUtility.waitProcedure(procExecutor, p1keyA);
110
111    // p2 should start and suspend
112    waitAndAssertTimestamp(p1keyA, 4, 60000);
113    waitAndAssertTimestamp(p2keyA, 1, 7);
114    waitAndAssertTimestamp(p3keyB, 2, 3);
115    assertEquals(true, lockA.get());
116
117    // wait until p2 is fully completed
118    p2keyA.setThrowSuspend(false);
119    procExecutor.getScheduler().addFront(p2keyA);
120    ProcedureTestingUtility.waitProcedure(procExecutor, p2keyA);
121    waitAndAssertTimestamp(p1keyA, 4, 60000);
122    waitAndAssertTimestamp(p2keyA, 2, 8);
123    waitAndAssertTimestamp(p3keyB, 2, 3);
124    assertEquals(false, lockA.get());
125    assertEquals(false, lockB.get());
126  }

128  @Test
129  public void testYieldWhileHoldingLocks() {
130    final AtomicBoolean lock = new AtomicBoolean(false);
131
132    final TestLockProcedure p1 = new TestLockProcedure(lock, "key", true, false);
133    final TestLockProcedure p2 = new TestLockProcedure(lock, "key", true, false);
134
135    procExecutor.submitProcedure(p1);
136    procExecutor.submitProcedure(p2);
137
138    // try to execute a bunch of yield on p1, p2 should be blocked
139    while (p1.getTimestamps().size() < 100) Threads.sleep(10);
140    assertEquals(0, p2.getTimestamps().size());
141
142    // wait until p1 is completed
143    p1.setThrowYield(false);
144    ProcedureTestingUtility.waitProcedure(procExecutor, p1);
145
146    // try to execute a bunch of yield on p2
147    while (p2.getTimestamps().size() < 100) Threads.sleep(10);
148    assertEquals(p1.getTimestamps().get(p1.getTimestamps().size() - 1).longValue() + 1,
149      p2.getTimestamps().get(0).longValue());
150
151    // wait until p2 is completed
152    p1.setThrowYield(false);
153    ProcedureTestingUtility.waitProcedure(procExecutor, p1);
154  }
155
156  private void waitAndAssertTimestamp(TestLockProcedure proc, int size, int lastTs) {
157    final ArrayList<Long> timestamps = proc.getTimestamps();
158    while (timestamps.size() < size) Threads.sleep(10);
159    LOG.info(proc + " -> " + timestamps);
160    assertEquals(size, timestamps.size());
161    if (size > 0) {
162      assertEquals(lastTs, timestamps.get(timestamps.size() - 1).longValue());
163    }
164  }

166  public static class TestLockProcedure extends Procedure<TestProcEnv> {
167    private final ArrayList<Long> timestamps = new ArrayList<>();
168    private final String key;
169
170    private boolean triggerRollback = false;
171    private boolean throwSuspend = false;
172    private boolean throwYield = false;
173    private AtomicBoolean lock = null;
174    private boolean hasLock = false;
175
176    public TestLockProcedure(final AtomicBoolean lock, final String key,
177        final boolean throwYield, final boolean throwSuspend) {
178      this.lock = lock;
179      this.key = key;
180      this.throwYield = throwYield;
181      this.throwSuspend = throwSuspend;
182    }
183
184    public void setThrowYield(final boolean throwYield) {
185      this.throwYield = throwYield;
186    }
187
188    public void setThrowSuspend(final boolean throwSuspend) {
189      this.throwSuspend = throwSuspend;
190    }
191
192    public void setTriggerRollback(final boolean triggerRollback) {
193      this.triggerRollback = triggerRollback;
194    }
195
196    @Override
197    protected Procedure[] execute(final TestProcEnv env)
198        throws ProcedureYieldException, ProcedureSuspendedException {
199      LOG.info("EXECUTE " + this + " suspend " + (lock != null));
200      timestamps.add(env.nextTimestamp());
201      if (triggerRollback) {
202        setFailure(getClass().getSimpleName(), new Exception("injected failure"));
203      } else if (throwYield) {
204        throw new ProcedureYieldException();
205      } else if (throwSuspend) {
206        throw new ProcedureSuspendedException();
207      }
208      return null;
209    }
210
211    @Override
212    protected void rollback(final TestProcEnv env) {
213      LOG.info("ROLLBACK " + this);
214      timestamps.add(env.nextTimestamp() * 10000);
215    }
216
217    @Override
218    protected LockState acquireLock(final TestProcEnv env) {
219      if ((hasLock = lock.compareAndSet(false, true))) {
220        LOG.info("ACQUIRE LOCK " + this + " " + (hasLock));
221        return LockState.LOCK_ACQUIRED;
222      }
223      return LockState.LOCK_YIELD_WAIT;
224    }
225
226    @Override
227    protected void releaseLock(final TestProcEnv env) {
228      LOG.info("RELEASE LOCK " + this + " " + hasLock);
229      lock.set(false);
230    }
231
232    @Override
233    protected boolean holdLock(final TestProcEnv env) {
234      return true;
235    }
236
237    public ArrayList<Long> getTimestamps() {
238      return timestamps;
239    }
240
241    @Override
242    protected void toStringClassDetails(StringBuilder builder) {
243      builder.append(getClass().getName());
244      builder.append("(" + key + ")");
245    }
246
247    @Override
248    protected boolean abort(TestProcEnv env) { return false; }
249
250    @Override
251    protected void serializeStateData(ProcedureStateSerializer serializer)
252        throws IOException {
253    }
254
255    @Override
256    protected void deserializeStateData(ProcedureStateSerializer serializer)
257        throws IOException {
258    }
259  }
260
261  private static class TestProcEnv {
262    public final AtomicLong timestamp = new AtomicLong(0);
263
264    public long nextTimestamp() {
265      return timestamp.incrementAndGet();
266    }
267  }
