069  public static <TEnv> void restart(ProcedureExecutor<TEnv> procExecutor)
070      throws Exception {
071    restart(procExecutor, null, true);
072  }

074  public static <TEnv> void restart(ProcedureExecutor<TEnv> procExecutor,
075      Runnable beforeStartAction, boolean failOnCorrupted) throws Exception {
076    ProcedureStore procStore = procExecutor.getStore();
077    int storeThreads = procExecutor.getNumThreads();
078    int execThreads = procExecutor.getNumThreads();
079    // stop
080    procExecutor.stop();
081    procExecutor.join();
082    procStore.stop(false);
083    // nothing running...
084    if (beforeStartAction != null) {
085      beforeStartAction.run();
086    }
087    // re-start
088    procStore.start(storeThreads);
089    procExecutor.start(execThreads, failOnCorrupted);
090  }
091
092  public static void storeRestart(ProcedureStore procStore, ProcedureStore.ProcedureLoader loader)
093      throws Exception {
094    procStore.stop(false);
095    procStore.start(procStore.getNumThreads());
096    procStore.recoverLease();
097    procStore.load(loader);
098  }

100  public static void storeRestartAndAssert(ProcedureStore procStore, long maxProcId,
101      long runnableCount, int completedCount, int corruptedCount) throws Exception {
102    final LoadCounter loader = new LoadCounter();
103    storeRestart(procStore, loader);
104    assertEquals(maxProcId, loader.getMaxProcId());
105    assertEquals(runnableCount, loader.getRunnableCount());
106    assertEquals(completedCount, loader.getCompletedCount());
107    assertEquals(corruptedCount, loader.getCorruptedCount());
108  }

141  public static <TEnv> long submitAndWait(Configuration conf, TEnv env, Procedure<TEnv> proc)
142      throws IOException {
143    NoopProcedureStore procStore = new NoopProcedureStore();
144    ProcedureExecutor<TEnv> procExecutor = new ProcedureExecutor<TEnv>(conf, env, procStore);
145    procStore.start(1);
146    procExecutor.start(1, false);
147    try {
148      return submitAndWait(procExecutor, proc, HConstants.NO_NONCE, HConstants.NO_NONCE);
149    } finally {
150      procStore.stop(false);
151      procExecutor.stop();
152    }
153  }
154
155  public static <TEnv> long submitAndWait(ProcedureExecutor<TEnv> procExecutor, Procedure proc) {
156    return submitAndWait(procExecutor, proc, HConstants.NO_NONCE, HConstants.NO_NONCE);
157  }

159  public static <TEnv> long submitAndWait(ProcedureExecutor<TEnv> procExecutor, Procedure proc,
160      final long nonceGroup,
161      final long nonce) {
162    long procId = procExecutor.submitProcedure(proc, nonceGroup, nonce);
163    waitProcedure(procExecutor, procId);
164    return procId;
165  }

167  public static <TEnv> void waitProcedure(ProcedureExecutor<TEnv> procExecutor, Procedure proc) {
168    while (proc.getState() == ProcedureState.INITIALIZING) {
169      Threads.sleepWithoutInterrupt(250);
170    }
171    waitProcedure(procExecutor, proc.getProcId());
172  }
173
174  public static <TEnv> void waitProcedure(ProcedureExecutor<TEnv> procExecutor, long procId) {
175    while (!procExecutor.isFinished(procId) && procExecutor.isRunning()) {
176      Threads.sleepWithoutInterrupt(250);
177    }
178  }

180  public static <TEnv> void waitNoProcedureRunning(ProcedureExecutor<TEnv> procExecutor) {
181    int stableRuns = 0;
182    while (stableRuns < 10) {
183      if (procExecutor.getActiveExecutorCount() > 0 || procExecutor.getRunnableSet().size() > 0) {
184        stableRuns = 0;
185        Threads.sleepWithoutInterrupt(100);
186      } else {
187        stableRuns++;
188        Threads.sleepWithoutInterrupt(25);
189      }
190    }
191  }

210  public static void assertIsAbortException(final ProcedureInfo result) {
211    assertEquals(true, result.isFailed());
212    LOG.info(result.getException().getMessage());
213    Throwable cause = result.getException().getCause();
214    assertTrue("expected abort exception, got "+ cause, cause instanceof ProcedureAbortedException);

221    assertTrue("expected TimeoutIOException, got " + cause, cause instanceof TimeoutIOException);

224  public static void assertIsIllegalArgumentException(final ProcedureInfo result) {
225    assertEquals(true, result.isFailed());
226    LOG.info(result.getException().getMessage());
227    Throwable cause = getExceptionCause(result);
228    assertTrue("expected IllegalArgumentIOException, got " + cause,
229      cause instanceof IllegalArgumentIOException);
230  }
231
232  public static Throwable getExceptionCause(final ProcedureInfo procInfo) {
233    assert procInfo.isFailed();
234    Throwable cause = procInfo.getException().getCause();
235    return cause == null ? procInfo.getException() : cause;
236  }

309  public static class LoadCounter implements ProcedureStore.ProcedureLoader {
310    private final ArrayList<Procedure> corrupted = new ArrayList<Procedure>();
311    private final ArrayList<ProcedureInfo> completed = new ArrayList<ProcedureInfo>();
312    private final ArrayList<Procedure> runnable = new ArrayList<Procedure>();
313
314    private Set<Long> procIds;
315    private long maxProcId = 0;
316
317    public LoadCounter() {
318      this(null);
319    }
320
321    public LoadCounter(final Set<Long> procIds) {
322      this.procIds = procIds;
323    }
324
325    public void reset() {
326      reset(null);
327    }
328
329    public void reset(final Set<Long> procIds) {
330      corrupted.clear();
331      completed.clear();
332      runnable.clear();
333      this.procIds = procIds;
334      this.maxProcId = 0;
335    }
336
337    public long getMaxProcId() {
338      return maxProcId;
339    }
340
341    public ArrayList<Procedure> getRunnables() {
342      return runnable;
343    }
344
345    public int getRunnableCount() {
346      return runnable.size();
347    }
348
349    public ArrayList<ProcedureInfo> getCompleted() {
350      return completed;
351    }
352
353    public int getCompletedCount() {
354      return completed.size();
355    }
356
357    public int getLoadedCount() {
358      return runnable.size() + completed.size();
359    }
360
361    public ArrayList<Procedure> getCorrupted() {
362      return corrupted;
363    }
364
365    public int getCorruptedCount() {
366      return corrupted.size();
367    }
368
369    @Override
370    public void setMaxProcId(long maxProcId) {
371      this.maxProcId = maxProcId;
372    }
373
374    @Override
375    public void load(ProcedureIterator procIter) throws IOException {
376      while (procIter.hasNext()) {
377        long procId;
378        if (procIter.isNextCompleted()) {
379          ProcedureInfo proc = procIter.nextAsProcedureInfo();
380          procId = proc.getProcId();
381          LOG.debug("loading completed procId=" + procId + ": " + proc);
382          completed.add(proc);
383        } else {
384          Procedure proc = procIter.nextAsProcedure();
385          procId = proc.getProcId();
386          LOG.debug("loading runnable procId=" + procId + ": " + proc);
387          runnable.add(proc);
388        }
389        if (procIds != null) {
390          assertTrue("procId=" + procId + " unexpected", procIds.contains(procId));
391        }
392      }
393    }
394
395    @Override
396    public void handleCorrupted(ProcedureIterator procIter) throws IOException {
397      while (procIter.hasNext()) {
398        Procedure proc = procIter.nextAsProcedure();
399        LOG.debug("corrupted procId=" + proc.getProcId() + ": " + proc);
400        corrupted.add(proc);
401      }
402    }
403  }
