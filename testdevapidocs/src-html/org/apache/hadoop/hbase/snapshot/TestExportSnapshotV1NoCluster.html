043/**
044 * Test Export Snapshot Tool
045 * Tests V1 snapshots only. Used to ALSO test v2 but strange failure so separate the tests.
046 * See companion file for test of v2 snapshot.
047 * @see TestExportSnapshotV2NoCluster
048 */
049@Category({MapReduceTests.class, MediumTests.class})
050public class TestExportSnapshotV1NoCluster {
051  @ClassRule
052  public static final HBaseClassTestRule CLASS_RULE =
053      HBaseClassTestRule.forClass(TestExportSnapshotV1NoCluster.class);
054  private static final Logger LOG = LoggerFactory.getLogger(TestExportSnapshotV1NoCluster.class);
055
056  private HBaseCommonTestingUtility testUtil = new HBaseCommonTestingUtility();
057  private Path testDir;
058
059  @Before
060  public void setUpBefore() throws Exception {
061    this.testDir = setup(this.testUtil);
062  }
063
064  /**
065   * Setup for test. Returns path to test data dir.
066   */
067  static Path setup(HBaseCommonTestingUtility hctu) throws IOException {
068    // Make sure testDir is on LocalFileSystem
069    Path testDir =
070      hctu.getDataTestDir().makeQualified(URI.create("file:///"), new Path("/"));
071    FileSystem fs = testDir.getFileSystem(hctu.getConfiguration());
072    assertTrue("FileSystem '" + fs + "' is not local", fs instanceof LocalFileSystem);
073    hctu.getConfiguration().setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
074    hctu.getConfiguration().setInt("hbase.regionserver.msginterval", 100);
075    hctu.getConfiguration().setInt("hbase.client.pause", 250);
076    hctu.getConfiguration().setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 6);
077    hctu.getConfiguration().setBoolean("hbase.master.enabletable.roundrobin", true);
078    hctu.getConfiguration().setInt("mapreduce.map.maxattempts", 10);
079    hctu.getConfiguration().set(HConstants.HBASE_DIR, testDir.toString());
080    return testDir;
081  }
082
083  /**
084   * V1 snapshot test
085   */
086  @Test
087  public void testSnapshotWithRefsExportFileSystemState() throws Exception {
088    final SnapshotMock snapshotMock = new SnapshotMock(testUtil.getConfiguration(),
089      testDir.getFileSystem(testUtil.getConfiguration()), testDir);
090    final SnapshotMock.SnapshotBuilder builder = snapshotMock.createSnapshotV1("tableWithRefsV1",
091      "tableWithRefsV1");
092    testSnapshotWithRefsExportFileSystemState(builder, testUtil, testDir);
093  }
094
095  /**
096   * Generates a couple of regions for the specified SnapshotMock,
097   * and then it will run the export and verification.
098   */
099  static void testSnapshotWithRefsExportFileSystemState(SnapshotMock.SnapshotBuilder builder,
100      HBaseCommonTestingUtility testUtil, Path testDir) throws Exception {
101    Path[] r1Files = builder.addRegion();
102    Path[] r2Files = builder.addRegion();
103    builder.commit();
104    int snapshotFilesCount = r1Files.length + r2Files.length;
105    String snapshotName = builder.getSnapshotDescription().getName();
106    TableName tableName = builder.getTableDescriptor().getTableName();
107    TestExportSnapshot.testExportFileSystemState(testUtil.getConfiguration(),
108      tableName, snapshotName, snapshotName, snapshotFilesCount,
109      testDir, getDestinationDir(testUtil, testDir), false, null, true);
110  }
111
112  static Path getDestinationDir(HBaseCommonTestingUtility hctu, Path testDir) throws IOException {
113    FileSystem fs = FileSystem.get(hctu.getConfiguration());
114    Path path = new Path(new Path(testDir, "export-test"),
115      "export-" + System.currentTimeMillis()).makeQualified(fs.getUri(), testDir);
116    LOG.info("HDFS export destination path: " + path);
117    return path;
118  }
119}
