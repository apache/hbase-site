041public class MobSnapshotTestingUtils {
042
043  /**
044   * Create the Mob Table.
045   */
046  public static void createMobTable(final HBaseTestingUtility util,
047      final TableName tableName, int regionReplication,
048      final byte[]... families) throws IOException, InterruptedException {
049    createMobTable(util, tableName, SnapshotTestingUtils.getSplitKeys(),
050      regionReplication, families);
051  }
052
053  public static void createPreSplitMobTable(final HBaseTestingUtility util,
054      final TableName tableName, int nRegions, final byte[]... families)
055      throws IOException, InterruptedException {
056    createMobTable(util, tableName, SnapshotTestingUtils.getSplitKeys(nRegions),
057      1, families);
058  }
059
060  private static void createMobTable(final HBaseTestingUtility util,
061      final TableName tableName, final byte[][] splitKeys, int regionReplication,
062      final byte[]... families) throws IOException, InterruptedException {
063    HTableDescriptor htd = new HTableDescriptor(tableName);
064    htd.setRegionReplication(regionReplication);
065    for (byte[] family : families) {
066      HColumnDescriptor hcd = new HColumnDescriptor(family);
067      hcd.setMobEnabled(true);
068      hcd.setMobThreshold(0L);
069      htd.addFamily(hcd);
070    }
071    util.getAdmin().createTable(htd, splitKeys);
072    SnapshotTestingUtils.waitForTableToBeOnline(util, tableName);
073    assertEquals((splitKeys.length + 1) * regionReplication, util
074        .getAdmin().getTableRegions(tableName).size());
075  }
076
077  /**
078   * Create a Mob table.
079   *
080   * @param util
081   * @param tableName
082   * @param families
083   * @return An HTable instance for the created table.
084   * @throws IOException
085   */
086  public static Table createMobTable(final HBaseTestingUtility util,
087      final TableName tableName, final byte[]... families) throws IOException {
088    HTableDescriptor htd = new HTableDescriptor(tableName);
089    for (byte[] family : families) {
090      HColumnDescriptor hcd = new HColumnDescriptor(family);
091      // Disable blooms (they are on by default as of 0.95) but we disable them
092      // here because
093      // tests have hard coded counts of what to expect in block cache, etc.,
094      // and blooms being
095      // on is interfering.
096      hcd.setBloomFilterType(BloomType.NONE);
097      hcd.setMobEnabled(true);
098      hcd.setMobThreshold(0L);
099      htd.addFamily(hcd);
100    }
101    util.getAdmin().createTable(htd);
102    // HBaseAdmin only waits for regions to appear in hbase:meta we should wait
103    // until they are assigned
104    util.waitUntilAllRegionsAssigned(htd.getTableName());
105    return ConnectionFactory.createConnection(util.getConfiguration()).getTable(htd.getTableName());
106  }
107
108  /**
109   * Return the number of rows in the given table.
110   */
111  public static int countMobRows(final Table table, final byte[]... families) throws IOException {
112    Scan scan = new Scan();
113    for (byte[] family : families) {
114      scan.addFamily(family);
115    }
116    try (ResultScanner results = table.getScanner(scan)) {
117      int count = 0;
118      for (Result res; (res = results.next()) != null;) {
119        count++;
120        for (Cell cell : res.listCells()) {
121          // Verify the value
122          Assert.assertTrue(CellUtil.cloneValue(cell).length > 0);
123        }
124      }
125      return count;
126    }
127  }
128
129  public static void verifyMobRowCount(final HBaseTestingUtility util,
130      final TableName tableName, long expectedRows) throws IOException {
131
132    Table table = ConnectionFactory.createConnection(util.getConfiguration()).getTable(tableName);
133    try {
134      assertEquals(expectedRows, countMobRows(table));
135    } finally {
136      table.close();
137    }
138  }
