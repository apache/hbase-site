249  @Test
250  public void testSnapshotFailsOnNonExistantTable() throws Exception {
251    // make sure we don't fail on listing snapshots
252    SnapshotTestingUtils.assertNoSnapshots(admin);
253    TableName tableName = TableName.valueOf("_not_a_table");
254
255    // make sure the table doesn't exist
256    boolean fail = false;
257    do {
258      try {
259        admin.getDescriptor(tableName);
260        fail = true;
261        LOG.error("Table:" + tableName + " already exists, checking a new name");
262        tableName = TableName.valueOf(tableName + "!");
263      } catch (TableNotFoundException e) {
264        fail = false;
265      }
266    } while (fail);
267
268    // snapshot the non-existant table
269    try {
270      admin.snapshot("fail", tableName, SnapshotType.FLUSH);
271      fail("Snapshot succeeded even though there is not table.");
272    } catch (SnapshotCreationException e) {
273      LOG.info("Correctly failed to snapshot a non-existant table:" + e.getMessage());
274    }
275  }

277  /**
278   * Helper method for testing async snapshot operations. Just waits for the given snapshot to
279   * complete on the server by repeatedly checking the master.
280   * @param master the master running the snapshot
281   * @param snapshot the snapshot to check
282   * @param timeoutNanos the timeout in nano between checks to see if the snapshot is done
283   */
284  private static void waitForSnapshotToComplete(HMaster master,
285      SnapshotProtos.SnapshotDescription snapshot, long timeoutNanos) throws Exception {
286    final IsSnapshotDoneRequest request =
287      IsSnapshotDoneRequest.newBuilder().setSnapshot(snapshot).build();
288    long start = System.nanoTime();
289    while (System.nanoTime() - start < timeoutNanos) {
290      try {
291        IsSnapshotDoneResponse done = master.getMasterRpcServices().isSnapshotDone(null, request);
292        if (done.getDone()) {
293          return;
294        }
295      } catch (ServiceException e) {
296        // ignore UnknownSnapshotException, this is possible as for AsyncAdmin, the method will
297        // return immediately after sending out the request, no matter whether the master has
298        // processed the request or not.
299        if (!(e.getCause() instanceof UnknownSnapshotException)) {
300          throw e;
301        }
302      }
303
304      Thread.sleep(200);
305    }
306    throw new TimeoutException("Timeout waiting for snapshot " + snapshot + " to complete");
307  }

309  @Test
310  public void testAsyncFlushSnapshot() throws Exception {
311    SnapshotProtos.SnapshotDescription snapshot = SnapshotProtos.SnapshotDescription.newBuilder()
312        .setName("asyncSnapshot").setTable(TABLE_NAME.getNameAsString())
313        .setType(SnapshotProtos.SnapshotDescription.Type.FLUSH).build();
314
315    // take the snapshot async
316    admin.snapshotAsync(
317      new SnapshotDescription("asyncSnapshot", TABLE_NAME, SnapshotType.FLUSH));
318
319    // constantly loop, looking for the snapshot to complete
320    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
321    waitForSnapshotToComplete(master, snapshot, TimeUnit.MINUTES.toNanos(1));
322    LOG.info(" === Async Snapshot Completed ===");
323    UTIL.getHBaseCluster().getMaster().getMasterFileSystem().logFileSystemState(LOG);
324
325    // make sure we get the snapshot
326    SnapshotTestingUtils.assertOneSnapshotThatMatches(admin, snapshot);
327  }

329  @Test
330  public void testSnapshotStateAfterMerge() throws Exception {
331    int numRows = DEFAULT_NUM_ROWS;
332    // make sure we don't fail on listing snapshots
333    SnapshotTestingUtils.assertNoSnapshots(admin);
334    // load the table so we have some data
335    SnapshotTestingUtils.loadData(UTIL, TABLE_NAME, numRows, TEST_FAM);
336
337    // Take a snapshot
338    String snapshotBeforeMergeName = "snapshotBeforeMerge";
339    admin.snapshot(snapshotBeforeMergeName, TABLE_NAME, SnapshotType.FLUSH);
340
341    // Clone the table
342    TableName cloneBeforeMergeName = TableName.valueOf("cloneBeforeMerge");
343    admin.cloneSnapshot(snapshotBeforeMergeName, cloneBeforeMergeName);
344    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneBeforeMergeName);
345
346    // Merge two regions
347    List<RegionInfo> regions = admin.getRegions(TABLE_NAME);
348    Collections.sort(regions, new Comparator<RegionInfo>() {
349      @Override
350      public int compare(RegionInfo r1, RegionInfo r2) {
351        return Bytes.compareTo(r1.getStartKey(), r2.getStartKey());
352      }
353    });
354
355    int numRegions = admin.getRegions(TABLE_NAME).size();
356    int numRegionsAfterMerge = numRegions - 2;
357    admin.mergeRegionsAsync(regions.get(1).getEncodedNameAsBytes(),
358        regions.get(2).getEncodedNameAsBytes(), true);
359    admin.mergeRegionsAsync(regions.get(4).getEncodedNameAsBytes(),
360        regions.get(5).getEncodedNameAsBytes(), true);
361
362    // Verify that there's one region less
363    waitRegionsAfterMerge(numRegionsAfterMerge);
364    assertEquals(numRegionsAfterMerge, admin.getRegions(TABLE_NAME).size());
365
366    // Clone the table
367    TableName cloneAfterMergeName = TableName.valueOf("cloneAfterMerge");
368    admin.cloneSnapshot(snapshotBeforeMergeName, cloneAfterMergeName);
369    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneAfterMergeName);
370
371    verifyRowCount(UTIL, TABLE_NAME, numRows);
372    verifyRowCount(UTIL, cloneBeforeMergeName, numRows);
373    verifyRowCount(UTIL, cloneAfterMergeName, numRows);
374
375    // test that we can delete the snapshot
376    UTIL.deleteTable(cloneAfterMergeName);
377    UTIL.deleteTable(cloneBeforeMergeName);
378  }

441  private void waitRegionsAfterMerge(final long numRegionsAfterMerge)
442      throws IOException, InterruptedException {
443    // Verify that there's one region less
444    long startTime = System.currentTimeMillis();
445    while (admin.getRegions(TABLE_NAME).size() != numRegionsAfterMerge) {
446      // This may be flaky... if after 15sec the merge is not complete give up
447      // it will fail in the assertEquals(numRegionsAfterMerge).
448      if ((System.currentTimeMillis() - startTime) > 15000)
449        break;
450      Thread.sleep(100);
451    }
452    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, TABLE_NAME);
453  }
