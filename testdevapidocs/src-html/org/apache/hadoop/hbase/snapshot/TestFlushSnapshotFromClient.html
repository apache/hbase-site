083  protected static void setupConf(Configuration conf) {
084    // disable the ui
085    conf.setInt("hbase.regionsever.info.port", -1);
086    // change the flush size to a small amount, regulating number of store files
087    conf.setInt("hbase.hregion.memstore.flush.size", 25000);
088    // so make sure we get a compaction when doing a load, but keep around some
089    // files in the store
090    conf.setInt("hbase.hstore.compaction.min", 10);
091    conf.setInt("hbase.hstore.compactionThreshold", 10);
092    // block writes if we get to 12 store files
093    conf.setInt("hbase.hstore.blockingStoreFiles", 12);
094    // Enable snapshot
095    conf.setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
096    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
097      ConstantSizeRegionSplitPolicy.class.getName());

286  @Test (timeout=300000)
287  public void testSnapshotStateAfterMerge() throws Exception {
288    int numRows = DEFAULT_NUM_ROWS;
289    Admin admin = UTIL.getHBaseAdmin();
290    // make sure we don't fail on listing snapshots
291    SnapshotTestingUtils.assertNoSnapshots(admin);
292    // load the table so we have some data
293    SnapshotTestingUtils.loadData(UTIL, TABLE_NAME, numRows, TEST_FAM);
294
295    // Take a snapshot
296    String snapshotBeforeMergeName = "snapshotBeforeMerge";
297    admin.snapshot(snapshotBeforeMergeName, TABLE_NAME, SnapshotDescription.Type.FLUSH);
298
299    // Clone the table
300    TableName cloneBeforeMergeName = TableName.valueOf("cloneBeforeMerge");
301    admin.cloneSnapshot(snapshotBeforeMergeName, cloneBeforeMergeName);
302    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneBeforeMergeName);
303
304    // Merge two regions
305    List<HRegionInfo> regions = admin.getTableRegions(TABLE_NAME);
306    Collections.sort(regions, new Comparator<HRegionInfo>() {
307      public int compare(HRegionInfo r1, HRegionInfo r2) {
308        return Bytes.compareTo(r1.getStartKey(), r2.getStartKey());
309      }
310    });
311
312    int numRegions = admin.getTableRegions(TABLE_NAME).size();
313    int numRegionsAfterMerge = numRegions - 2;
314    admin.mergeRegions(regions.get(1).getEncodedNameAsBytes(),
315        regions.get(2).getEncodedNameAsBytes(), true);
316    admin.mergeRegions(regions.get(5).getEncodedNameAsBytes(),
317        regions.get(6).getEncodedNameAsBytes(), true);
318
319    // Verify that there's one region less
320    waitRegionsAfterMerge(numRegionsAfterMerge);
321    assertEquals(numRegionsAfterMerge, admin.getTableRegions(TABLE_NAME).size());
322
323    // Clone the table
324    TableName cloneAfterMergeName = TableName.valueOf("cloneAfterMerge");
325    admin.cloneSnapshot(snapshotBeforeMergeName, cloneAfterMergeName);
326    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneAfterMergeName);
327
328    verifyRowCount(UTIL, TABLE_NAME, numRows);
329    verifyRowCount(UTIL, cloneBeforeMergeName, numRows);
330    verifyRowCount(UTIL, cloneAfterMergeName, numRows);
331
332    // test that we can delete the snapshot
333    UTIL.deleteTable(cloneAfterMergeName);
334    UTIL.deleteTable(cloneBeforeMergeName);
335  }

400  /**
401   * Demonstrate that we reject snapshot requests if there is a snapshot already running on the
402   * same table currently running and that concurrent snapshots on different tables can both
403   * succeed concurretly.
404   */
405  @Test(timeout=300000)
406  public void testConcurrentSnapshottingAttempts() throws IOException, InterruptedException {
407    final TableName TABLE2_NAME = TableName.valueOf(TABLE_NAME + "2");
408
409    int ssNum = 20;
410    Admin admin = UTIL.getHBaseAdmin();
411    // make sure we don't fail on listing snapshots
412    SnapshotTestingUtils.assertNoSnapshots(admin);
413    // create second testing table
414    SnapshotTestingUtils.createTable(UTIL, TABLE2_NAME, TEST_FAM);
415    // load the table so we have some data
416    SnapshotTestingUtils.loadData(UTIL, TABLE_NAME, DEFAULT_NUM_ROWS, TEST_FAM);
417    SnapshotTestingUtils.loadData(UTIL, TABLE2_NAME, DEFAULT_NUM_ROWS, TEST_FAM);
418
419    final CountDownLatch toBeSubmitted = new CountDownLatch(ssNum);
420    // We'll have one of these per thread
421    class SSRunnable implements Runnable {
422      SnapshotDescription ss;
423      SSRunnable(SnapshotDescription ss) {
424        this.ss = ss;
425      }
426
427      @Override
428      public void run() {
429        try {
430          Admin admin = UTIL.getHBaseAdmin();
431          LOG.info("Submitting snapshot request: " + ClientSnapshotDescriptionUtils.toString(ss));
432          admin.takeSnapshotAsync(ss);
433        } catch (Exception e) {
434          LOG.info("Exception during snapshot request: " + ClientSnapshotDescriptionUtils.toString(
435              ss)
436              + ".  This is ok, we expect some", e);
437        }
438        LOG.info("Submitted snapshot request: " + ClientSnapshotDescriptionUtils.toString(ss));
439        toBeSubmitted.countDown();
440      }
441    };
442
443    // build descriptions
444    SnapshotDescription[] descs = new SnapshotDescription[ssNum];
445    for (int i = 0; i < ssNum; i++) {
446      SnapshotDescription.Builder builder = SnapshotDescription.newBuilder();
447      builder.setTable(((i % 2) == 0 ? TABLE_NAME : TABLE2_NAME).getNameAsString());
448      builder.setName("ss"+i);
449      builder.setType(SnapshotDescription.Type.FLUSH);
450      descs[i] = builder.build();
451    }
452
453    // kick each off its own thread
454    for (int i=0 ; i < ssNum; i++) {
455      new Thread(new SSRunnable(descs[i])).start();
456    }
457
458    // wait until all have been submitted
459    toBeSubmitted.await();
460
461    // loop until all are done.
462    while (true) {
463      int doneCount = 0;
464      for (SnapshotDescription ss : descs) {
465        try {
466          if (admin.isSnapshotFinished(ss)) {
467            doneCount++;
468          }
469        } catch (Exception e) {
470          LOG.warn("Got an exception when checking for snapshot " + ss.getName(), e);
471          doneCount++;
472        }
473      }
474      if (doneCount == descs.length) {
475        break;
476      }
477      Thread.sleep(100);
478    }
479
480    // dump for debugging
481    UTIL.getHBaseCluster().getMaster().getMasterFileSystem().logFileSystemState(LOG);
482
483    List<SnapshotDescription> taken = admin.listSnapshots();
484    int takenSize = taken.size();
485    LOG.info("Taken " + takenSize + " snapshots:  " + taken);
486    assertTrue("We expect at least 1 request to be rejected because of we concurrently" +
487        " issued many requests", takenSize < ssNum && takenSize > 0);
488
489    // Verify that there's at least one snapshot per table
490    int t1SnapshotsCount = 0;
491    int t2SnapshotsCount = 0;
492    for (SnapshotDescription ss : taken) {
493      if (TableName.valueOf(ss.getTable()).equals(TABLE_NAME)) {
494        t1SnapshotsCount++;
495      } else if (TableName.valueOf(ss.getTable()).equals(TABLE2_NAME)) {
496        t2SnapshotsCount++;
497      }
498    }
499    assertTrue("We expect at least 1 snapshot of table1 ", t1SnapshotsCount > 0);
500    assertTrue("We expect at least 1 snapshot of table2 ", t2SnapshotsCount > 0);
501
502    UTIL.deleteTable(TABLE2_NAME);
503  }

505  private void waitRegionsAfterMerge(final long numRegionsAfterMerge)
506      throws IOException, InterruptedException {
507    Admin admin = UTIL.getHBaseAdmin();
508    // Verify that there's one region less
509    long startTime = System.currentTimeMillis();
510    while (admin.getTableRegions(TABLE_NAME).size() != numRegionsAfterMerge) {
511      // This may be flaky... if after 15sec the merge is not complete give up
512      // it will fail in the assertEquals(numRegionsAfterMerge).
513      if ((System.currentTimeMillis() - startTime) > 15000)
514        break;
515      Thread.sleep(100);
516    }
517    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, TABLE_NAME);
518  }
