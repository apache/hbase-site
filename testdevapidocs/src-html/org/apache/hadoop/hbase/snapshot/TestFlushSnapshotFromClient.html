282  /**
283   * Helper method for testing async snapshot operations. Just waits for the given snapshot to
284   * complete on the server by repeatedly checking the master.
285   * @param master the master running the snapshot
286   * @param snapshot the snapshot to check
287   * @param sleep amount to sleep between checks to see if the snapshot is done
288   */
289  private static void waitForSnapshotToComplete(HMaster master,
290      SnapshotProtos.SnapshotDescription snapshot, long timeoutNanos) throws Exception {
291    final IsSnapshotDoneRequest request =
292      IsSnapshotDoneRequest.newBuilder().setSnapshot(snapshot).build();
293    long start = System.nanoTime();
294    while (System.nanoTime() - start < timeoutNanos) {
295      try {
296        IsSnapshotDoneResponse done = master.getMasterRpcServices().isSnapshotDone(null, request);
297        if (done.getDone()) {
298          return;
299        }
300      } catch (ServiceException e) {
301        // ignore UnknownSnapshotException, this is possible as for AsyncAdmin, the method will
302        // return immediately after sending out the request, no matter whether the master has
303        // processed the request or not.
304        if (!(e.getCause() instanceof UnknownSnapshotException)) {
305          throw e;
306        }
307      }
308
309      Thread.sleep(200);
310    }
311    throw new TimeoutException("Timeout waiting for snapshot " + snapshot + " to complete");
312  }

334  @Test
335  public void testSnapshotStateAfterMerge() throws Exception {
336    int numRows = DEFAULT_NUM_ROWS;
337    // make sure we don't fail on listing snapshots
338    SnapshotTestingUtils.assertNoSnapshots(admin);
339    // load the table so we have some data
340    SnapshotTestingUtils.loadData(UTIL, TABLE_NAME, numRows, TEST_FAM);
341
342    // Take a snapshot
343    String snapshotBeforeMergeName = "snapshotBeforeMerge";
344    admin.snapshot(snapshotBeforeMergeName, TABLE_NAME, SnapshotType.FLUSH);
345
346    // Clone the table
347    TableName cloneBeforeMergeName = TableName.valueOf("cloneBeforeMerge");
348    admin.cloneSnapshot(snapshotBeforeMergeName, cloneBeforeMergeName);
349    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneBeforeMergeName);
350
351    // Merge two regions
352    List<RegionInfo> regions = admin.getRegions(TABLE_NAME);
353    Collections.sort(regions, new Comparator<RegionInfo>() {
354      @Override
355      public int compare(RegionInfo r1, RegionInfo r2) {
356        return Bytes.compareTo(r1.getStartKey(), r2.getStartKey());
357      }
358    });
359
360    int numRegions = admin.getRegions(TABLE_NAME).size();
361    int numRegionsAfterMerge = numRegions - 2;
362    admin.mergeRegionsAsync(regions.get(1).getEncodedNameAsBytes(),
363        regions.get(2).getEncodedNameAsBytes(), true);
364    admin.mergeRegionsAsync(regions.get(4).getEncodedNameAsBytes(),
365        regions.get(5).getEncodedNameAsBytes(), true);
366
367    // Verify that there's one region less
368    waitRegionsAfterMerge(numRegionsAfterMerge);
369    assertEquals(numRegionsAfterMerge, admin.getRegions(TABLE_NAME).size());
370
371    // Clone the table
372    TableName cloneAfterMergeName = TableName.valueOf("cloneAfterMerge");
373    admin.cloneSnapshot(snapshotBeforeMergeName, cloneAfterMergeName);
374    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, cloneAfterMergeName);
375
376    verifyRowCount(UTIL, TABLE_NAME, numRows);
377    verifyRowCount(UTIL, cloneBeforeMergeName, numRows);
378    verifyRowCount(UTIL, cloneAfterMergeName, numRows);
379
380    // test that we can delete the snapshot
381    UTIL.deleteTable(cloneAfterMergeName);
382    UTIL.deleteTable(cloneBeforeMergeName);
383  }

446  /**
447   * Demonstrate that we reject snapshot requests if there is a snapshot already running on the
448   * same table currently running and that concurrent snapshots on different tables can both
449   * succeed concurretly.
450   */
451  @Test
452  public void testConcurrentSnapshottingAttempts() throws IOException, InterruptedException {
453    final TableName TABLE2_NAME = TableName.valueOf(TABLE_NAME + "2");
454
455    int ssNum = 20;
456    // make sure we don't fail on listing snapshots
457    SnapshotTestingUtils.assertNoSnapshots(admin);
458    // create second testing table
459    SnapshotTestingUtils.createTable(UTIL, TABLE2_NAME, TEST_FAM);
460    // load the table so we have some data
461    SnapshotTestingUtils.loadData(UTIL, TABLE_NAME, DEFAULT_NUM_ROWS, TEST_FAM);
462    SnapshotTestingUtils.loadData(UTIL, TABLE2_NAME, DEFAULT_NUM_ROWS, TEST_FAM);
463
464    final CountDownLatch toBeSubmitted = new CountDownLatch(ssNum);
465    // We'll have one of these per thread
466    class SSRunnable implements Runnable {
467      SnapshotDescription ss;
468      SSRunnable(SnapshotDescription ss) {
469        this.ss = ss;
470      }
471
472      @Override
473      public void run() {
474        try {
475          LOG.info("Submitting snapshot request: " + ClientSnapshotDescriptionUtils
476              .toString(ProtobufUtil.createHBaseProtosSnapshotDesc(ss)));
477          admin.snapshotAsync(ss);
478        } catch (Exception e) {
479          LOG.info("Exception during snapshot request: " + ClientSnapshotDescriptionUtils.toString(
480            ProtobufUtil.createHBaseProtosSnapshotDesc(ss))
481              + ".  This is ok, we expect some", e);
482        }
483        LOG.info("Submitted snapshot request: " + ClientSnapshotDescriptionUtils
484            .toString(ProtobufUtil.createHBaseProtosSnapshotDesc(ss)));
485        toBeSubmitted.countDown();
486      }
487    }
488
489    // build descriptions
490    SnapshotDescription[] descs = new SnapshotDescription[ssNum];
491    for (int i = 0; i < ssNum; i++) {
492      if(i % 2 ==0) {
493        descs[i] = new SnapshotDescription("ss" + i, TABLE_NAME, SnapshotType.FLUSH);
494      } else {
495        descs[i] = new SnapshotDescription("ss" + i, TABLE2_NAME, SnapshotType.FLUSH);
496      }
497    }
498
499    // kick each off its own thread
500    for (int i=0 ; i < ssNum; i++) {
501      new Thread(new SSRunnable(descs[i])).start();
502    }
503
504    // wait until all have been submitted
505    toBeSubmitted.await();
506
507    // loop until all are done.
508    while (true) {
509      int doneCount = 0;
510      for (SnapshotDescription ss : descs) {
511        try {
512          if (admin.isSnapshotFinished(ss)) {
513            doneCount++;
514          }
515        } catch (Exception e) {
516          LOG.warn("Got an exception when checking for snapshot " + ss.getName(), e);
517          doneCount++;
518        }
519      }
520      if (doneCount == descs.length) {
521        break;
522      }
523      Thread.sleep(100);
524    }
525
526    // dump for debugging
527    UTIL.getHBaseCluster().getMaster().getMasterFileSystem().logFileSystemState(LOG);
528
529    List<SnapshotDescription> taken = admin.listSnapshots();
530    int takenSize = taken.size();
531    LOG.info("Taken " + takenSize + " snapshots:  " + taken);
532    assertTrue("We expect at least 1 request to be rejected because of we concurrently" +
533        " issued many requests", takenSize < ssNum && takenSize > 0);
534
535    // Verify that there's at least one snapshot per table
536    int t1SnapshotsCount = 0;
537    int t2SnapshotsCount = 0;
538    for (SnapshotDescription ss : taken) {
539      if (ss.getTableName().equals(TABLE_NAME)) {
540        t1SnapshotsCount++;
541      } else if (ss.getTableName().equals(TABLE2_NAME)) {
542        t2SnapshotsCount++;
543      }
544    }
545    assertTrue("We expect at least 1 snapshot of table1 ", t1SnapshotsCount > 0);
546    assertTrue("We expect at least 1 snapshot of table2 ", t2SnapshotsCount > 0);
547
548    UTIL.deleteTable(TABLE2_NAME);
549  }

551  private void waitRegionsAfterMerge(final long numRegionsAfterMerge)
552      throws IOException, InterruptedException {
553    // Verify that there's one region less
554    long startTime = System.currentTimeMillis();
555    while (admin.getRegions(TABLE_NAME).size() != numRegionsAfterMerge) {
556      // This may be flaky... if after 15sec the merge is not complete give up
557      // it will fail in the assertEquals(numRegionsAfterMerge).
558      if ((System.currentTimeMillis() - startTime) > 15000)
559        break;
560      Thread.sleep(100);
561    }
562    SnapshotTestingUtils.waitForTableToBeOnline(UTIL, TABLE_NAME);
563  }
