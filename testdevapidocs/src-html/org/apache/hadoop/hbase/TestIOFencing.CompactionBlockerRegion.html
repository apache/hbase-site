063/**
064 * Test for the case where a regionserver going down has enough cycles to do damage to regions that
065 * have actually been assigned elsehwere.
066 * <p>
067 * If we happen to assign a region before it fully done with in its old location -- i.e. it is on
068 * two servers at the same time -- all can work fine until the case where the region on the dying
069 * server decides to compact or otherwise change the region file set. The region in its new location
070 * will then get a surprise when it tries to do something w/ a file removed by the region in its old
071 * location on dying server.
072 * <p>
073 * Making a test for this case is a little tough in that even if a file is deleted up on the
074 * namenode, if the file was opened before the delete, it will continue to let reads happen until
075 * something changes the state of cached blocks in the dfsclient that was already open (a block from
076 * the deleted file is cleaned from the datanode by NN).
077 * <p>
078 * What we will do below is do an explicit check for existence on the files listed in the region
079 * that has had some files removed because of a compaction. This sort of hurry's along and makes
080 * certain what is a chance occurance.
081 */
082@Category({MiscTests.class, LargeTests.class})
083public class TestIOFencing {
084  private static final Log LOG = LogFactory.getLog(TestIOFencing.class);
085  static {

087    // debugging (see HBASE-12285 for details).

096  public abstract static class CompactionBlockerRegion extends HRegion {
097    volatile int compactCount = 0;
098    volatile CountDownLatch compactionsBlocked = new CountDownLatch(0);
099    volatile CountDownLatch compactionsWaiting = new CountDownLatch(0);
100
101    @SuppressWarnings("deprecation")
102    public CompactionBlockerRegion(Path tableDir, WAL log,
103        FileSystem fs, Configuration confParam, RegionInfo info,
104        TableDescriptor htd, RegionServerServices rsServices) {
105      super(tableDir, log, fs, confParam, info, htd, rsServices);
106    }
107
108    public void stopCompactions() {
109      compactionsBlocked = new CountDownLatch(1);
110      compactionsWaiting = new CountDownLatch(1);
111    }
112
113    public void allowCompactions() {
114      LOG.debug("allowing compactions");
115      compactionsBlocked.countDown();
116    }
117    public void waitForCompactionToBlock() throws IOException {
118      try {
119        LOG.debug("waiting for compaction to block");
120        compactionsWaiting.await();
121        LOG.debug("compaction block reached");
122      } catch (InterruptedException ex) {
123        throw new IOException(ex);
124      }
125    }
126
127    @Override
128    public boolean compact(CompactionContext compaction, HStore store,
129        ThroughputController throughputController) throws IOException {
130      try {
131        return super.compact(compaction, store, throughputController);
132      } finally {
133        compactCount++;
134      }
135    }
136
137    @Override
138    public boolean compact(CompactionContext compaction, HStore store,
139        ThroughputController throughputController, User user) throws IOException {
140      try {
141        return super.compact(compaction, store, throughputController, user);
142      } finally {
143        compactCount++;
144      }
145    }
146
147    public int countStoreFiles() {
148      int count = 0;
149      for (HStore store : stores.values()) {
150        count += store.getStorefilesCount();
151      }
152      return count;
153    }
154  }

156  /**
157   * An override of HRegion that allows us park compactions in a holding pattern and
158   * then when appropriate for the test, allow them proceed again.
159   */
160  public static class BlockCompactionsInPrepRegion extends CompactionBlockerRegion {
161
162    public BlockCompactionsInPrepRegion(Path tableDir, WAL log,
163        FileSystem fs, Configuration confParam, RegionInfo info,
164        TableDescriptor htd, RegionServerServices rsServices) {
165      super(tableDir, log, fs, confParam, info, htd, rsServices);
166    }
167    @Override
168    protected void doRegionCompactionPrep() throws IOException {
169      compactionsWaiting.countDown();
170      try {
171        compactionsBlocked.await();
172      } catch (InterruptedException ex) {
173        throw new IOException();
174      }
175      super.doRegionCompactionPrep();
176    }
177  }

179  /**
180   * An override of HRegion that allows us park compactions in a holding pattern and
181   * then when appropriate for the test, allow them proceed again. This allows the compaction
182   * entry to go the WAL before blocking, but blocks afterwards
183   */
184  public static class BlockCompactionsInCompletionRegion extends CompactionBlockerRegion {
185    public BlockCompactionsInCompletionRegion(Path tableDir, WAL log,
186        FileSystem fs, Configuration confParam, RegionInfo info,
187        TableDescriptor htd, RegionServerServices rsServices) {
188      super(tableDir, log, fs, confParam, info, htd, rsServices);
189    }
190    @Override
191    protected HStore instantiateHStore(final ColumnFamilyDescriptor family) throws IOException {
192      return new BlockCompactionsInCompletionHStore(this, family, this.conf);
193    }
194  }
195
196  public static class BlockCompactionsInCompletionHStore extends HStore {
197    CompactionBlockerRegion r;
198    protected BlockCompactionsInCompletionHStore(HRegion region, ColumnFamilyDescriptor family,
199        Configuration confParam) throws IOException {
200      super(region, family, confParam);
201      r = (CompactionBlockerRegion) region;
202    }
203
204    @Override
205    protected void completeCompaction(Collection<HStoreFile> compactedFiles) throws IOException {
206      try {
207        r.compactionsWaiting.countDown();
208        r.compactionsBlocked.await();
209      } catch (InterruptedException ex) {
210        throw new IOException(ex);
211      }
212      super.completeCompaction(compactedFiles);
213    }
214  }

216  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
217  private final static TableName TABLE_NAME =
218      TableName.valueOf("tabletest");
219  private final static byte[] FAMILY = Bytes.toBytes("family");
220  private static final int FIRST_BATCH_COUNT = 4000;
221  private static final int SECOND_BATCH_COUNT = FIRST_BATCH_COUNT;

223  /**

227   * location changes the files out from underneath a region being served elsewhere.

249  public void doTest(Class<?> regionClass, MemoryCompactionPolicy policy) throws Exception {
250    Configuration c = TEST_UTIL.getConfiguration();
251    // Insert our custom region
252    c.setClass(HConstants.REGION_IMPL, regionClass, HRegion.class);
253    // Encourage plenty of flushes
254    c.setLong("hbase.hregion.memstore.flush.size", 25000);
255    c.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY, ConstantSizeRegionSplitPolicy.class.getName());
256    // Only run compaction when we tell it to
257    c.setInt("hbase.hstore.compaction.min",1);
258    c.setInt("hbase.hstore.compactionThreshold", 1000);
259    c.setLong("hbase.hstore.blockingStoreFiles", 1000);
260    // Compact quickly after we tell it to!
261    c.setInt("hbase.regionserver.thread.splitcompactcheckfrequency", 1000);
262    c.set(CompactingMemStore.COMPACTING_MEMSTORE_TYPE_KEY, String.valueOf(policy));
263    LOG.info("Starting mini cluster");
264    TEST_UTIL.startMiniCluster(1);
265    CompactionBlockerRegion compactingRegion = null;
266    Admin admin = null;
267    try {
268      LOG.info("Creating admin");
269      admin = TEST_UTIL.getConnection().getAdmin();
270      LOG.info("Creating table");
271      TEST_UTIL.createTable(TABLE_NAME, FAMILY);
272      Table table = TEST_UTIL.getConnection().getTable(TABLE_NAME);
273      LOG.info("Loading test table");
274      // Find the region
275      List<HRegion> testRegions = TEST_UTIL.getMiniHBaseCluster().findRegionsForTable(TABLE_NAME);
276      assertEquals(1, testRegions.size());
277      compactingRegion = (CompactionBlockerRegion)testRegions.get(0);
278      LOG.info("Blocking compactions");
279      compactingRegion.stopCompactions();
280      long lastFlushTime = compactingRegion.getEarliestFlushTimeForAllStores();
281      // Load some rows
282      TEST_UTIL.loadNumericRows(table, FAMILY, 0, FIRST_BATCH_COUNT);
283
284      // add a compaction from an older (non-existing) region to see whether we successfully skip
285      // those entries
286      HRegionInfo oldHri = new HRegionInfo(table.getName(),
287        HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
288      CompactionDescriptor compactionDescriptor = ProtobufUtil.toCompactionDescriptor(oldHri,
289        FAMILY, Lists.newArrayList(new Path("/a")), Lists.newArrayList(new Path("/b")),
290        new Path("store_dir"));
291      WALUtil.writeCompactionMarker(compactingRegion.getWAL(),
292          ((HRegion)compactingRegion).getReplicationScope(),
293        oldHri, compactionDescriptor, compactingRegion.getMVCC());
294
295      // Wait till flush has happened, otherwise there won't be multiple store files
296      long startWaitTime = System.currentTimeMillis();
297      while (compactingRegion.getEarliestFlushTimeForAllStores() <= lastFlushTime ||
298          compactingRegion.countStoreFiles() <= 1) {
299        LOG.info("Waiting for the region to flush " +
300          compactingRegion.getRegionInfo().getRegionNameAsString());
301        Thread.sleep(1000);
302        admin.flush(table.getName());
303        assertTrue("Timed out waiting for the region to flush",
304          System.currentTimeMillis() - startWaitTime < 30000);
305      }
306      assertTrue(compactingRegion.countStoreFiles() > 1);
307      final byte REGION_NAME[] = compactingRegion.getRegionInfo().getRegionName();
308      LOG.info("Asking for compaction");
309      admin.majorCompact(TABLE_NAME);
310      LOG.info("Waiting for compaction to be about to start");
311      compactingRegion.waitForCompactionToBlock();
312      LOG.info("Starting a new server");
313      RegionServerThread newServerThread = TEST_UTIL.getMiniHBaseCluster().startRegionServer();
314      final HRegionServer newServer = newServerThread.getRegionServer();
315      LOG.info("Killing region server ZK lease");
316      TEST_UTIL.expireRegionServerSession(0);
317      CompactionBlockerRegion newRegion = null;
318      startWaitTime = System.currentTimeMillis();
319      LOG.info("Waiting for the new server to pick up the region " + Bytes.toString(REGION_NAME));
320
321      // wait for region to be assigned and to go out of log replay if applicable
322      Waiter.waitFor(c, 60000, new Waiter.Predicate<Exception>() {
323        @Override
324        public boolean evaluate() throws Exception {
325          Region newRegion = newServer.getOnlineRegion(REGION_NAME);
326          return newRegion != null && !newRegion.isRecovering();
327        }
328      });
329
330      newRegion = (CompactionBlockerRegion)newServer.getOnlineRegion(REGION_NAME);
331
332      // After compaction of old region finishes on the server that was going down, make sure that
333      // all the files we expect are still working when region is up in new location.
334      FileSystem fs = newRegion.getFilesystem();
335      for (String f: newRegion.getStoreFileList(new byte [][] {FAMILY})) {
336        assertTrue("After compaction, does not exist: " + f, fs.exists(new Path(f)));
337      }
338      LOG.info("Allowing compaction to proceed");
339      compactingRegion.allowCompactions();
340      while (compactingRegion.compactCount == 0) {
341        Thread.sleep(1000);
342      }
343      // The server we killed stays up until the compaction that was started before it was killed
344      // completes. In logs you should see the old regionserver now going down.
345      LOG.info("Compaction finished");
346
347      // If we survive the split keep going...
348      // Now we make sure that the region isn't totally confused.  Load up more rows.
349      TEST_UTIL.loadNumericRows(table, FAMILY, FIRST_BATCH_COUNT,
350        FIRST_BATCH_COUNT + SECOND_BATCH_COUNT);
351      admin.majorCompact(TABLE_NAME);
352      startWaitTime = System.currentTimeMillis();
353      while (newRegion.compactCount == 0) {
354        Thread.sleep(1000);
355        assertTrue("New region never compacted",
356          System.currentTimeMillis() - startWaitTime < 180000);
357      }
358      int count;
359      for (int i = 0;; i++) {
360        try {
361          count = TEST_UTIL.countRows(table);
362          break;
363        } catch (DoNotRetryIOException e) {
364          // wait up to 30s
365          if (i >= 30 || !e.getMessage().contains("File does not exist")) {
366            throw e;
367          }
368          Thread.sleep(1000);
369        }
370      }
371      if (policy == MemoryCompactionPolicy.EAGER || policy == MemoryCompactionPolicy.ADAPTIVE) {
372        assertTrue(FIRST_BATCH_COUNT + SECOND_BATCH_COUNT >= count);
373      } else {
374        assertEquals(FIRST_BATCH_COUNT + SECOND_BATCH_COUNT, count);
375      }
376    } finally {
377      if (compactingRegion != null) {
378        compactingRegion.allowCompactions();
379      }
380      admin.close();
381      TEST_UTIL.shutdownMiniCluster();
382    }
383  }
