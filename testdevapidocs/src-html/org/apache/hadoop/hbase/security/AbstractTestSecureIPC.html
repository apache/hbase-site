098            @Override
099            public TestProtos.EchoResponseProto echo(RpcController controller,
100                                                     TestProtos.EchoRequestProto request)
101                throws ServiceException {
102              if (controller instanceof PayloadCarryingRpcController) {
103                PayloadCarryingRpcController pcrc = (PayloadCarryingRpcController) controller;
104                // If cells, scan them to check we are able to iterate what we were given and since
105                // this is
106                // an echo, just put them back on the controller creating a new block. Tests our
107                // block
108                // building.
109                CellScanner cellScanner = pcrc.cellScanner();
110                List<Cell> list = null;
111                if (cellScanner != null) {
112                  list = new ArrayList<Cell>();
113                  try {
114                    while (cellScanner.advance()) {
115                      list.add(cellScanner.current());
116                    }
117                  } catch (IOException e) {
118                    throw new ServiceException(e);
119                  }
120                }
121                cellScanner = CellUtil.createCellScanner(list);
122                ((PayloadCarryingRpcController) controller).setCellScanner(cellScanner);
123              }
124              return TestProtos.EchoResponseProto.newBuilder()
125                  .setMessage(request.getMessage()).build();
126            }
127          });

237  private UserGroupInformation loginKerberosPrincipal(String krbKeytab, String krbPrincipal)
238      throws Exception {
239    Configuration cnf = new Configuration();
240    cnf.set(CommonConfigurationKeys.HADOOP_SECURITY_AUTHENTICATION, "kerberos");

246  /**
247   * Sets up a RPC Server and a Client. Does a RPC checks the result. If an exception is thrown
248   * from the stub, this function will throw root cause of that exception.
249   */
250  private void callRpcService(User clientUser) throws Exception {
251    SecurityInfo securityInfoMock = Mockito.mock(SecurityInfo.class);
252    Mockito.when(securityInfoMock.getServerPrincipal())
253        .thenReturn(HBaseKerberosUtils.KRB_PRINCIPAL);
254    SecurityInfo.addInfo("TestProtobufRpcProto", securityInfoMock);
255
256    InetSocketAddress isa = new InetSocketAddress(HOST, 0);
257
258    RpcServerInterface rpcServer =
259        new RpcServer(null, "AbstractTestSecureIPC",
260            Lists.newArrayList(new RpcServer.BlockingServiceAndInterface(SERVICE, null)), isa,
261            serverConf, new FifoRpcScheduler(serverConf, 1));
262    rpcServer.start();
263    try (RpcClient rpcClient = RpcClientFactory.createClient(clientConf,
264        HConstants.DEFAULT_CLUSTER_ID.toString())) {
265      InetSocketAddress address = rpcServer.getListenerAddress();
266      if (address == null) {
267        throw new IOException("Listener channel is closed");
268      }
269      BlockingRpcChannel channel =
270          rpcClient.createBlockingRpcChannel(
271              ServerName.valueOf(address.getHostName(), address.getPort(),
272                  System.currentTimeMillis()), clientUser, 0);
273      TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface stub =
274          TestRpcServiceProtos.TestProtobufRpcProto.newBlockingStub(channel);
275      List<String> results = new ArrayList<>();
276      TestThread th1 = new TestThread(stub, results);
277      final Throwable exception[] = new Throwable[1];
278      Collections.synchronizedList(new ArrayList<Throwable>());
279      Thread.UncaughtExceptionHandler exceptionHandler =
280          new Thread.UncaughtExceptionHandler() {
281            public void uncaughtException(Thread th, Throwable ex) {
282              exception[0] = ex;
283            }
284          };
285      th1.setUncaughtExceptionHandler(exceptionHandler);
286      th1.start();
287      th1.join();
288      if (exception[0] != null) {
289        // throw root cause.
290        while (exception[0].getCause() != null) {
291          exception[0] = exception[0].getCause();
292        }
293        throw (Exception) exception[0];
294      }
295    } finally {
296      rpcServer.stop();
297    }
298  }

300  public static class TestThread extends Thread {
301      private final TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface stub;
302
303      private final List<String> results;
304
305          public TestThread(TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface stub, List<String> results) {
306          this.stub = stub;
307          this.results = results;
308        }
309
310          @Override
311      public void run() {
312          try {
313            int[] messageSize = new int[] {100, 1000, 10000};
314            for (int i = 0; i < messageSize.length; i++) {
315              String input = RandomStringUtils.random(messageSize[i]);
316              String result = stub.echo(null, TestProtos.EchoRequestProto.newBuilder()
317                  .setMessage(input).build()).getMessage();
318              assertEquals(input, result);
319            }
320          } catch (ServiceException e) {
321            throw new RuntimeException(e);
322          }
323        }
324    }
