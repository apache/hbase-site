565  /**
566   * A copy of the BlockingRpcCallback class for use locally. Only difference is that it makes
567   * use of non-shaded protobufs; i.e. refers to com.google.protobuf.* rather than to
568   * org.apache.hadoop.hbase.shaded.com.google.protobuf.*
569   */
570  private static class NonShadedBlockingRpcCallback<R> implements
571      com.google.protobuf.RpcCallback<R> {
572    private R result;
573    private boolean resultSet = false;
574
575    /**
576     * Called on completion of the RPC call with the response object, or {@code null} in the case of
577     * an error.
578     * @param parameter the response object or {@code null} if an error occurred
579     */
580    @Override
581    public void run(R parameter) {
582      synchronized (this) {
583        result = parameter;
584        resultSet = true;
585        this.notifyAll();
586      }
587    }
588
589    /**
590     * Returns the parameter passed to {@link #run(Object)} or {@code null} if a null value was
591     * passed.  When used asynchronously, this method will block until the {@link #run(Object)}
592     * method has been called.
593     * @return the response object or {@code null} if no response was passed
594     */
595    public synchronized R get() throws IOException {
596      while (!resultSet) {
597        try {
598          this.wait();
599        } catch (InterruptedException ie) {
600          InterruptedIOException exception = new InterruptedIOException(ie.getMessage());
601          exception.initCause(ie);
602          throw exception;
603        }
604      }
605      return result;
606    }
607  }
