139    public TokenServer(Configuration conf, HBaseTestingUtility TEST_UTIL) throws IOException {
140      this.conf = conf;
141      this.TEST_UTIL = TEST_UTIL;
142      this.startcode = EnvironmentEdgeManager.currentTime();
143      // Server to handle client requests.
144      String hostname =
145        Strings.domainNamePointerToHostName(DNS.getDefaultHost("default", "default"));
146      int port = 0;
147      // Creation of an ISA will force a resolve.
148      InetSocketAddress initialIsa = new InetSocketAddress(hostname, port);
149      if (initialIsa.getAddress() == null) {
150        throw new IllegalArgumentException("Failed resolve of " + initialIsa);
151      }
152      final List<BlockingServiceAndInterface> sai = new ArrayList<>(1);
153      // Make a proxy to go between the shaded Service that rpc expects and the
154      // non-shaded Service this CPEP is providing. This is because this test does a neat
155      // little trick of testing the CPEP Service by inserting it as RpcServer Service. This
156      // worked fine before we shaded PB. Now we need these proxies.
157      final BlockingService service =
158        AuthenticationProtos.AuthenticationService.newReflectiveBlockingService(this);
159      final org.apache.hbase.thirdparty.com.google.protobuf.BlockingService proxy =
160        new org.apache.hbase.thirdparty.com.google.protobuf.BlockingService() {
161          @Override
162          public Message callBlockingMethod(MethodDescriptor md,
163              org.apache.hbase.thirdparty.com.google.protobuf.RpcController controller,
164              Message param)
165              throws org.apache.hbase.thirdparty.com.google.protobuf.ServiceException {
166            com.google.protobuf.Descriptors.MethodDescriptor methodDescriptor =
167              service.getDescriptorForType().findMethodByName(md.getName());
168            com.google.protobuf.Message request = service.getRequestPrototype(methodDescriptor);
169            // TODO: Convert rpcController
170            com.google.protobuf.Message response = null;
171            try {
172              response = service.callBlockingMethod(methodDescriptor, null, request);
173            } catch (ServiceException e) {
174              throw new org.apache.hbase.thirdparty.com.google.protobuf.ServiceException(e);
175            }
176            return null;// Convert 'response'.
177          }

188          }
189
190          @Override
191          public Message getResponsePrototype(MethodDescriptor arg0) {
192            // TODO Auto-generated method stub

394        AuthenticationProtos.TokenIdentifier.Kind.HBASE_AUTH_TOKEN));

398    conf.set("hbase.security.authentication", "kerberos");
399    conf.setBoolean(HADOOP_SECURITY_AUTHORIZATION, true);
400    conf.set(RpcServerFactory.CUSTOM_RPC_SERVER_IMPL_CONF_KEY, rpcServerImpl);
401    server = new TokenServer(conf, TEST_UTIL);
402    serverThread = new Thread(server);
403    Threads.setDaemonThreadRunning(serverThread, "TokenServer:"+server.getServerName().toString());
404    // wait for startup
405    while (!server.isStarted() && !server.isStopped()) {
406      Thread.sleep(10);
407    }
408    server.rpcServer.refreshAuthManager(new PolicyProvider() {
409      @Override
410      public Service[] getServices() {
411        return new Service [] {
412          new Service("security.client.protocol.acl",
413            AuthenticationProtos.AuthenticationService.BlockingInterface.class)};
414      }
415    });
416    ZKClusterId.setClusterId(server.getZooKeeper(), clusterId);
417    secretManager = (AuthenticationTokenSecretManager)server.getSecretManager();
418    while(secretManager.getCurrentKey() == null) {
419      Thread.sleep(1);
420    }
421  }

511    } finally {
512      conn.close();

516  /**
517   * A copy of the BlockingRpcCallback class for use locally. Only difference is that it makes
518   * use of non-shaded protobufs; i.e. refers to com.google.protobuf.* rather than to
519   * org.apache.hbase.thirdparty.com.google.protobuf.*
520   */
521  private static class NonShadedBlockingRpcCallback<R> implements
522      com.google.protobuf.RpcCallback<R> {
523    private R result;
524    private boolean resultSet = false;
525
526    /**
527     * Called on completion of the RPC call with the response object, or {@code null} in the case of
528     * an error.
529     * @param parameter the response object or {@code null} if an error occurred
530     */
531    @Override
532    public void run(R parameter) {
533      synchronized (this) {
534        result = parameter;
535        resultSet = true;
536        this.notifyAll();
537      }
538    }
539
540    /**
541     * Returns the parameter passed to {@link #run(Object)} or {@code null} if a null value was
542     * passed.  When used asynchronously, this method will block until the {@link #run(Object)}
543     * method has been called.
544     * @return the response object or {@code null} if no response was passed
545     */
546    public synchronized R get() throws IOException {
547      while (!resultSet) {
548        try {
549          this.wait();
550        } catch (InterruptedException ie) {
551          InterruptedIOException exception = new InterruptedIOException(ie.getMessage());
552          exception.initCause(ie);
553          throw exception;
554        }
555      }
556      return result;
557    }
