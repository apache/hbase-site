048@Category({SecurityTests.class, MediumTests.class})
049public class TestVisibilityLabelsOpWithDifferentUsersNoACL {
050  private static final String PRIVATE = "private";
051  private static final String CONFIDENTIAL = "confidential";
052  private static final String SECRET = "secret";
053  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
054  private static Configuration conf;
055
056  @Rule
057  public final TestName TEST_NAME = new TestName();
058  private static User SUPERUSER;
059  private static User NORMAL_USER;
060  private static User NORMAL_USER1;
061
062  @BeforeClass
063  public static void setupBeforeClass() throws Exception {
064    // setup configuration
065    conf = TEST_UTIL.getConfiguration();
066    VisibilityTestUtil.enableVisiblityLabels(conf);
067    String currentUser = User.getCurrent().getName();
068    conf.set("hbase.superuser", "admin,"+currentUser);
069    TEST_UTIL.startMiniCluster(2);
070
071    // Wait for the labels table to become available
072    TEST_UTIL.waitTableEnabled(LABELS_TABLE_NAME.getName(), 50000);
073    SUPERUSER = User.createUserForTesting(conf, "admin", new String[] { "supergroup" });
074    NORMAL_USER = User.createUserForTesting(conf, "user1", new String[] {});
075    NORMAL_USER1 = User.createUserForTesting(conf, "user2", new String[] {});
076    addLabels();
077  }
078
079  @AfterClass
080  public static void tearDownAfterClass() throws Exception {
081    TEST_UTIL.shutdownMiniCluster();
082  }
083
084  @Test
085  public void testLabelsTableOpsWithDifferentUsers() throws Throwable {
086    PrivilegedExceptionAction<VisibilityLabelsResponse> action =
087        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
088      public VisibilityLabelsResponse run() throws Exception {
089        try (Connection conn = ConnectionFactory.createConnection(conf)) {
090          return VisibilityClient.setAuths(conn, new String[] { CONFIDENTIAL, PRIVATE }, "user1");
091        } catch (Throwable e) {
092        }
093        return null;
094      }
095    };
096    VisibilityLabelsResponse response = SUPERUSER.runAs(action);
097    assertTrue(response.getResult(0).getException().getValue().isEmpty());
098    assertTrue(response.getResult(1).getException().getValue().isEmpty());
099    
100    // Ideally this should not be allowed.  this operation should fail or do nothing.
101    action = new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
102      public VisibilityLabelsResponse run() throws Exception {
103        try (Connection conn = ConnectionFactory.createConnection(conf)) {
104          return VisibilityClient.setAuths(conn, new String[] { CONFIDENTIAL, PRIVATE }, "user3");
105        } catch (Throwable e) {
106        }
107        return null;
108      }
109    };
110    response = NORMAL_USER1.runAs(action);
111    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
112        .getResult(0).getException().getName());
113    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
114        .getResult(1).getException().getName());
115
116    PrivilegedExceptionAction<GetAuthsResponse> action1 =
117        new PrivilegedExceptionAction<GetAuthsResponse>() {
118      public GetAuthsResponse run() throws Exception {
119        try (Connection conn = ConnectionFactory.createConnection(conf)) {
120          return VisibilityClient.getAuths(conn, "user1");
121        } catch (Throwable e) {
122        }
123        return null;
124      }
125    };
126    GetAuthsResponse authsResponse = NORMAL_USER.runAs(action1);
127    assertTrue(authsResponse.getAuthList().isEmpty());
128    authsResponse = NORMAL_USER1.runAs(action1);
129    assertTrue(authsResponse.getAuthList().isEmpty());
130    authsResponse = SUPERUSER.runAs(action1);
131    List<String> authsList = new ArrayList<>(authsResponse.getAuthList().size());
132    for (ByteString authBS : authsResponse.getAuthList()) {
133      authsList.add(Bytes.toString(authBS.toByteArray()));
134    }
135    assertEquals(2, authsList.size());
136    assertTrue(authsList.contains(CONFIDENTIAL));
137    assertTrue(authsList.contains(PRIVATE));
138
139    PrivilegedExceptionAction<VisibilityLabelsResponse> action2 = 
140        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
141      public VisibilityLabelsResponse run() throws Exception {
142        try (Connection conn = ConnectionFactory.createConnection(conf)) {
143          return VisibilityClient.clearAuths(conn, new String[] {
144              CONFIDENTIAL, PRIVATE }, "user1");
145        } catch (Throwable e) {
146        }
147        return null;
148      }
149    };
150    response = NORMAL_USER1.runAs(action2);
151    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
152        .getResult(0).getException().getName());
153    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
154        .getResult(1).getException().getName());
155    response = SUPERUSER.runAs(action2);
156    assertTrue(response.getResult(0).getException().getValue().isEmpty());
157    assertTrue(response.getResult(1).getException().getValue().isEmpty());
158    authsResponse = SUPERUSER.runAs(action1);
159    assertTrue(authsResponse.getAuthList().isEmpty());
160  }
161
162  private static void addLabels() throws Exception {
163    PrivilegedExceptionAction<VisibilityLabelsResponse> action = 
164        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
165      public VisibilityLabelsResponse run() throws Exception {
166        String[] labels = { SECRET, CONFIDENTIAL, PRIVATE };
167        try (Connection conn = ConnectionFactory.createConnection(conf)) {
168          VisibilityClient.addLabels(conn, labels);
169        } catch (Throwable t) {
170          throw new IOException(t);
171        }
172        return null;
173      }
174    };
175    SUPERUSER.runAs(action);
176  }
177}
