051@Category({SecurityTests.class, MediumTests.class})
052/**
053 * Test visibility by setting 'hbase.security.visibility.mutations.checkauths' to true
054 */
055public class TestVisibilityWithCheckAuths {
056  private static final String TOPSECRET = "TOPSECRET";
057  private static final String PUBLIC = "PUBLIC";
058  public static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
059  private static final byte[] row1 = Bytes.toBytes("row1");
060  private final static byte[] fam = Bytes.toBytes("info");
061  private final static byte[] qual = Bytes.toBytes("qual");
062  private final static byte[] value = Bytes.toBytes("value");
063  public static Configuration conf;
064
065  @Rule
066  public final TestName TEST_NAME = new TestName();
067  public static User SUPERUSER;
068  public static User USER;
069  @BeforeClass
070  public static void setupBeforeClass() throws Exception {
071    // setup configuration
072    conf = TEST_UTIL.getConfiguration();
073    conf.setBoolean(HConstants.DISTRIBUTED_LOG_REPLAY_KEY, false);
074    VisibilityTestUtil.enableVisiblityLabels(conf);
075    conf.setBoolean(VisibilityConstants.CHECK_AUTHS_FOR_MUTATION, true);
076    conf.setClass(VisibilityUtils.VISIBILITY_LABEL_GENERATOR_CLASS, SimpleScanLabelGenerator.class,
077        ScanLabelGenerator.class);
078    conf.set("hbase.superuser", "admin");
079    TEST_UTIL.startMiniCluster(2);
080    SUPERUSER = User.createUserForTesting(conf, "admin", new String[] { "supergroup" });
081    USER = User.createUserForTesting(conf, "user", new String[]{});
082    // Wait for the labels table to become available
083    TEST_UTIL.waitTableEnabled(LABELS_TABLE_NAME.getName(), 50000);
084    addLabels();
085  }
086
087  @AfterClass
088  public static void tearDownAfterClass() throws Exception {
089    TEST_UTIL.shutdownMiniCluster();
090  }
091
092  public static void addLabels() throws Exception {
093    PrivilegedExceptionAction<VisibilityLabelsResponse> action =
094        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
095      @Override
096      public VisibilityLabelsResponse run() throws Exception {
097        String[] labels = { TOPSECRET };
098        try (Connection conn = ConnectionFactory.createConnection(conf)) {
099          VisibilityClient.addLabels(conn, labels);
100        } catch (Throwable t) {
101          throw new IOException(t);
102        }
103        return null;
104      }
105    };
106    SUPERUSER.runAs(action);
107  }
108
109  @Test
110  public void testVerifyAccessDeniedForInvalidUserAuths() throws Exception {
111    PrivilegedExceptionAction<VisibilityLabelsResponse> action =
112        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
113      @Override
114      public VisibilityLabelsResponse run() throws Exception {
115        try (Connection conn = ConnectionFactory.createConnection(conf)) {
116          return VisibilityClient.setAuths(conn, new String[] { TOPSECRET },
117              USER.getShortName());
118        } catch (Throwable e) {
119        }
120        return null;
121      }
122    };
123    SUPERUSER.runAs(action);
124    final TableName tableName = TableName.valueOf(TEST_NAME.getMethodName());
125    Admin hBaseAdmin = TEST_UTIL.getHBaseAdmin();
126    HColumnDescriptor colDesc = new HColumnDescriptor(fam);
127    colDesc.setMaxVersions(5);
128    HTableDescriptor desc = new HTableDescriptor(tableName);
129    desc.addFamily(colDesc);
130    hBaseAdmin.createTable(desc);
131    try {
132      TEST_UTIL.getHBaseAdmin().flush(tableName);
133      PrivilegedExceptionAction<Void> actiona = new PrivilegedExceptionAction<Void>() {
134        @Override
135        public Void run() throws Exception {
136          try (Connection connection = ConnectionFactory.createConnection(conf);
137               Table table = connection.getTable(tableName)) {
138            Put p = new Put(row1);
139            p.setCellVisibility(new CellVisibility(PUBLIC + "&" + TOPSECRET));
140            p.addColumn(fam, qual, 125l, value);
141            table.put(p);
142            Assert.fail("Testcase should fail with AccesDeniedException");
143          } catch (Throwable t) {
144            assertTrue(t.getMessage().contains("AccessDeniedException"));
145          }
146          return null;
147        }
148      };
149      USER.runAs(actiona);
150    } catch (Exception e) {
151      throw new IOException(e);
152    }
153  }

155  @Test
156  public void testLabelsWithAppend() throws Throwable {
157    PrivilegedExceptionAction<VisibilityLabelsResponse> action =
158        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
159      @Override
160      public VisibilityLabelsResponse run() throws Exception {
161        try (Connection conn = ConnectionFactory.createConnection(conf)) {
162          return VisibilityClient.setAuths(conn, new String[] { TOPSECRET },
163              USER.getShortName());
164        } catch (Throwable e) {
165        }
166        return null;
167      }
168    };
169    SUPERUSER.runAs(action);
170    final TableName tableName = TableName.valueOf(TEST_NAME.getMethodName());
171    try (Table table = TEST_UTIL.createTable(tableName, fam)) {
172      final byte[] row1 = Bytes.toBytes("row1");
173      final byte[] val = Bytes.toBytes("a");
174      PrivilegedExceptionAction<Void> actiona = new PrivilegedExceptionAction<Void>() {
175        @Override
176        public Void run() throws Exception {
177          try (Connection connection = ConnectionFactory.createConnection(conf);
178               Table table = connection.getTable(tableName)) {
179            Put put = new Put(row1);
180            put.addColumn(fam, qual, HConstants.LATEST_TIMESTAMP, val);
181            put.setCellVisibility(new CellVisibility(TOPSECRET));
182            table.put(put);
183          }
184          return null;
185        }
186      };
187      USER.runAs(actiona);
188      actiona = new PrivilegedExceptionAction<Void>() {
189        @Override
190        public Void run() throws Exception {
191          try (Connection connection = ConnectionFactory.createConnection(conf);
192               Table table = connection.getTable(tableName)) {
193            Append append = new Append(row1);
194            append.add(fam, qual, Bytes.toBytes("b"));
195            table.append(append);
196          }
197          return null;
198        }
199      };
200      USER.runAs(actiona);
201      actiona = new PrivilegedExceptionAction<Void>() {
202        @Override
203        public Void run() throws Exception {
204          try (Connection connection = ConnectionFactory.createConnection(conf);
205               Table table = connection.getTable(tableName)) {
206            Append append = new Append(row1);
207            append.add(fam, qual, Bytes.toBytes("c"));
208            append.setCellVisibility(new CellVisibility(PUBLIC));
209            table.append(append);
210            Assert.fail("Testcase should fail with AccesDeniedException");
211          } catch (Throwable t) {
212            assertTrue(t.getMessage().contains("AccessDeniedException"));
213          }
214          return null;
215        }
216      };
217      USER.runAs(actiona);
218    }
219  }
