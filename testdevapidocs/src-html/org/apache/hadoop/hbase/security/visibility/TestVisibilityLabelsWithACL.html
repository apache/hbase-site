080  @BeforeClass
081  public static void setupBeforeClass() throws Exception {
082    // setup configuration
083    conf = TEST_UTIL.getConfiguration();
084    SecureTestUtil.enableSecurity(conf);
085    conf.set("hbase.coprocessor.master.classes", AccessController.class.getName() + ","
086        + VisibilityController.class.getName());
087    conf.set("hbase.coprocessor.region.classes", AccessController.class.getName() + ","
088        + VisibilityController.class.getName());
089    TEST_UTIL.startMiniCluster(2);
090
091    TEST_UTIL.waitTableEnabled(AccessControlLists.ACL_TABLE_NAME.getName(), 50000);
092    // Wait for the labels table to become available
093    TEST_UTIL.waitTableEnabled(LABELS_TABLE_NAME.getName(), 50000);
094    addLabels();
095
096    // Create users for testing
097    SUPERUSER = User.createUserForTesting(conf, "admin", new String[] { "supergroup" });
098    NORMAL_USER1 = User.createUserForTesting(conf, "user1", new String[] {});
099    NORMAL_USER2 = User.createUserForTesting(conf, "user2", new String[] {});
100    // Grant users EXEC privilege on the labels table. For the purposes of this
101    // test, we want to insure that access is denied even with the ability to access
102    // the endpoint.
103    SecureTestUtil.grantOnTable(TEST_UTIL, NORMAL_USER1.getShortName(), LABELS_TABLE_NAME,
104      null, null, Permission.Action.EXEC);
105    SecureTestUtil.grantOnTable(TEST_UTIL, NORMAL_USER2.getShortName(), LABELS_TABLE_NAME,
106      null, null, Permission.Action.EXEC);
107  }

114  @Test
115  public void testScanForUserWithFewerLabelAuthsThanLabelsInScanAuthorizations() throws Throwable {
116    String[] auths = { SECRET };
117    String user = "user2";
118    VisibilityClient.setAuths(TEST_UTIL.getConnection(), auths, user);
119    TableName tableName = TableName.valueOf(TEST_NAME.getMethodName());
120    final Table table = createTableAndWriteDataWithLabels(tableName, SECRET + "&" + CONFIDENTIAL
121        + "&!" + PRIVATE, SECRET + "&!" + PRIVATE);
122    SecureTestUtil.grantOnTable(TEST_UTIL, NORMAL_USER2.getShortName(), tableName,
123      null, null, Permission.Action.READ);
124    PrivilegedExceptionAction<Void> scanAction = new PrivilegedExceptionAction<Void>() {
125      @Override
126      public Void run() throws Exception {
127        Scan s = new Scan();
128        s.setAuthorizations(new Authorizations(SECRET, CONFIDENTIAL));
129        try (Connection connection = ConnectionFactory.createConnection(conf);
130             Table t = connection.getTable(table.getName())) {
131          ResultScanner scanner = t.getScanner(s);
132          Result result = scanner.next();
133          assertTrue(!result.isEmpty());
134          assertTrue(Bytes.equals(Bytes.toBytes("row2"), result.getRow()));
135          result = scanner.next();
136          assertNull(result);
137        }
138        return null;
139      }
140    };
141    NORMAL_USER2.runAs(scanAction);
142  }

225  @Test
226  public void testLabelsTableOpsWithDifferentUsers() throws Throwable {
227    PrivilegedExceptionAction<VisibilityLabelsResponse> action = 
228        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
229      @Override
230      public VisibilityLabelsResponse run() throws Exception {
231        try (Connection conn = ConnectionFactory.createConnection(conf)) {
232          return VisibilityClient.addLabels(conn, new String[] { "l1", "l2" });
233        } catch (Throwable e) {
234        }
235        return null;
236      }
237    };
238    VisibilityLabelsResponse response = NORMAL_USER1.runAs(action);
239    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
240        .getResult(0).getException().getName());
241    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
242        .getResult(1).getException().getName());
243
244    action = new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
245      @Override
246      public VisibilityLabelsResponse run() throws Exception {
247        try (Connection conn = ConnectionFactory.createConnection(conf)) {
248          return VisibilityClient.setAuths(conn, new String[] { CONFIDENTIAL, PRIVATE }, "user1");
249        } catch (Throwable e) {
250        }
251        return null;
252      }
253    };
254    response = NORMAL_USER1.runAs(action);
255    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
256        .getResult(0).getException().getName());
257    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response
258        .getResult(1).getException().getName());
259
260    action = new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
261      @Override
262      public VisibilityLabelsResponse run() throws Exception {
263        try (Connection conn = ConnectionFactory.createConnection(conf)) {
264          return VisibilityClient.setAuths(conn, new String[] { CONFIDENTIAL, PRIVATE }, "user1");
265        } catch (Throwable e) {
266        }
267        return null;
268      }
269    };
270    response = SUPERUSER.runAs(action);
271    assertTrue(response.getResult(0).getException().getValue().isEmpty());
272    assertTrue(response.getResult(1).getException().getValue().isEmpty());
273
274    action = new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
275      @Override
276      public VisibilityLabelsResponse run() throws Exception {
277        try (Connection conn = ConnectionFactory.createConnection(conf)) {
278          return VisibilityClient.clearAuths(conn, new String[] {
279              CONFIDENTIAL, PRIVATE }, "user1");
280        } catch (Throwable e) {
281        }
282        return null;
283      }
284    };
285    response = NORMAL_USER1.runAs(action);
286    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response.getResult(0)
287        .getException().getName());
288    assertEquals("org.apache.hadoop.hbase.security.AccessDeniedException", response.getResult(1)
289        .getException().getName());
290
291    response = VisibilityClient.clearAuths(TEST_UTIL.getConnection(), new String[] { CONFIDENTIAL,
292      PRIVATE }, "user1");
293    assertTrue(response.getResult(0).getException().getValue().isEmpty());
294    assertTrue(response.getResult(1).getException().getValue().isEmpty());
295
296    VisibilityClient.setAuths(TEST_UTIL.getConnection(), new String[] { CONFIDENTIAL, PRIVATE },
297      "user3");
298    PrivilegedExceptionAction<GetAuthsResponse> action1 = 
299        new PrivilegedExceptionAction<GetAuthsResponse>() {
300      @Override
301      public GetAuthsResponse run() throws Exception {
302        try (Connection conn = ConnectionFactory.createConnection(conf)) {
303          return VisibilityClient.getAuths(conn, "user3");
304        } catch (Throwable e) {
305        }
306        return null;
307      }
308    };
309    GetAuthsResponse authsResponse = NORMAL_USER1.runAs(action1);
310    assertNull(authsResponse);
311    authsResponse = SUPERUSER.runAs(action1);
312    List<String> authsList = new ArrayList<>(authsResponse.getAuthList().size());
313    for (ByteString authBS : authsResponse.getAuthList()) {
314      authsList.add(Bytes.toString(authBS.toByteArray()));
315    }
316    assertEquals(2, authsList.size());
317    assertTrue(authsList.contains(CONFIDENTIAL));
318    assertTrue(authsList.contains(PRIVATE));
319  }

321  private static Table createTableAndWriteDataWithLabels(TableName tableName, String... labelExps)
322      throws Exception {
323    Table table = null;
324    try {
325      table = TEST_UTIL.createTable(tableName, fam);
326      int i = 1;
327      List<Put> puts = new ArrayList<>(labelExps.length);
328      for (String labelExp : labelExps) {
329        Put put = new Put(Bytes.toBytes("row" + i));
330        put.addColumn(fam, qual, HConstants.LATEST_TIMESTAMP, value);
331        put.setCellVisibility(new CellVisibility(labelExp));
332        puts.add(put);
333        i++;
334      }
335      table.put(puts);
336    } finally {
337      if (table != null) {
338        table.close();
339      }
340    }
341    return table;
342  }

349      throw new IOException(t);
