065/**
066 * This is a VisibilityLabelService where labels in Mutation's visibility
067 * expression will be persisted as Strings itself rather than ordinals in
068 * 'labels' table. Also there is no need to add labels to the system, prior to
069 * using them in Mutations/Authorizations.
070 */
071@InterfaceAudience.Private
072public class ExpAsStringVisibilityLabelServiceImpl implements VisibilityLabelService {
073
074  private static final Log LOG = LogFactory.getLog(ExpAsStringVisibilityLabelServiceImpl.class);

076  private static final byte[] DUMMY_VALUE = new byte[0];
077  private static final byte STRING_SERIALIZATION_FORMAT = 2;
078  private static final Tag STRING_SERIALIZATION_FORMAT_TAG = new ArrayBackedTag(
079      TagType.VISIBILITY_EXP_SERIALIZATION_FORMAT_TAG_TYPE,
080      new byte[] { STRING_SERIALIZATION_FORMAT });
081  private final ExpressionParser expressionParser = new ExpressionParser();
082  private final ExpressionExpander expressionExpander = new ExpressionExpander();
083  private Configuration conf;

087  @Override
088  public OperationStatus[] addLabels(List<byte[]> labels) throws IOException {
089    // Not doing specific label add. We will just add labels in Mutation
090    // visibility expression as it
091    // is along with every cell.
092    OperationStatus[] status = new OperationStatus[labels.size()];
093    for (int i = 0; i < labels.size(); i++) {
094      status[i] = new OperationStatus(OperationStatusCode.SUCCESS);
095    }
096    return status;
097  }

099  @Override
100  public OperationStatus[] setAuths(byte[] user, List<byte[]> authLabels) throws IOException {
101    assert labelsRegion != null;
102    OperationStatus[] finalOpStatus = new OperationStatus[authLabels.size()];
103    Put p = new Put(user);
104    for (byte[] auth : authLabels) {
105      p.addImmutable(LABELS_TABLE_FAMILY, auth, DUMMY_VALUE);
106    }
107    this.labelsRegion.put(p);
108    // This is a testing impl and so not doing any caching
109    for (int i = 0; i < authLabels.size(); i++) {
110      finalOpStatus[i] = new OperationStatus(OperationStatusCode.SUCCESS);
111    }
112    return finalOpStatus;
113  }

115  @Override
116  public OperationStatus[] clearAuths(byte[] user, List<byte[]> authLabels) throws IOException {
117    assert labelsRegion != null;
118    OperationStatus[] finalOpStatus = new OperationStatus[authLabels.size()];
119    List<String> currentAuths;
120    if (AuthUtil.isGroupPrincipal(Bytes.toString(user))) {
121      String group = AuthUtil.getGroupName(Bytes.toString(user));
122      currentAuths = this.getGroupAuths(new String[]{group}, true);
123    }
124    else {
125      currentAuths = this.getUserAuths(user, true);
126    }
127    Delete d = new Delete(user);
128    int i = 0;
129    for (byte[] authLabel : authLabels) {
130      String authLabelStr = Bytes.toString(authLabel);
131      if (currentAuths.contains(authLabelStr)) {
132        d.addColumns(LABELS_TABLE_FAMILY, authLabel);
133      } else {
134        // This label is not set for the user.
135        finalOpStatus[i] = new OperationStatus(OperationStatusCode.FAILURE,
136            new InvalidLabelException("Label '" + authLabelStr + "' is not set for the user "
137                + Bytes.toString(user)));
138      }
139      i++;
140    }
141    this.labelsRegion.delete(d);
142    // This is a testing impl and so not doing any caching
143    for (i = 0; i < authLabels.size(); i++) {
144      if (finalOpStatus[i] == null) {
145        finalOpStatus[i] = new OperationStatus(OperationStatusCode.SUCCESS);
146      }
147    }
148    return finalOpStatus;
149  }

151  @Override
152  public List<String> getUserAuths(byte[] user, boolean systemCall) throws IOException {
153    assert (labelsRegion != null || systemCall);
154    List<String> auths = new ArrayList<>();
155    Get get = new Get(user);
156    List<Cell> cells = null;
157    if (labelsRegion == null) {
158      Table table = null;
159      Connection connection = null;
160      try {
161        connection = ConnectionFactory.createConnection(conf);
162        table = connection.getTable(VisibilityConstants.LABELS_TABLE_NAME);
163        Result result = table.get(get);
164        cells = result.listCells();
165      } finally {
166        if (table != null) {
167          table.close();
168        }
169        if (connection != null){
170          connection.close();
171        }
172      }
173    } else {
174      cells = this.labelsRegion.get(get, false);
175    }
176    if (cells != null) {
177      for (Cell cell : cells) {
178        String auth = Bytes.toString(cell.getQualifierArray(), cell.getQualifierOffset(),
179          cell.getQualifierLength());
180        auths.add(auth);
181      }
182    }
183    return auths;
184  }

186  @Override
187  public List<String> getGroupAuths(String[] groups, boolean systemCall) throws IOException {
188    assert (labelsRegion != null || systemCall);
189    List<String> auths = new ArrayList<>();
190    if (groups != null && groups.length > 0) {
191      for (String group : groups) {
192        Get get = new Get(Bytes.toBytes(AuthUtil.toGroupEntry(group)));
193        List<Cell> cells = null;
194        if (labelsRegion == null) {
195          Table table = null;
196          Connection connection = null;
197          try {
198            connection = ConnectionFactory.createConnection(conf);
199            table = connection.getTable(VisibilityConstants.LABELS_TABLE_NAME);
200            Result result = table.get(get);
201            cells = result.listCells();
202          } finally {
203            if (table != null) {
204              table.close();
205              connection.close();
206            }
207          }
208        } else {
209          cells = this.labelsRegion.get(get, false);
210        }
211        if (cells != null) {
212          for (Cell cell : cells) {
213            String auth = Bytes.toString(cell.getQualifierArray(), cell.getQualifierOffset(),
214              cell.getQualifierLength());
215            auths.add(auth);
216          }
217        }
218      }
219    }
220    return auths;
221  }

229  @Override
230  public List<Tag> createVisibilityExpTags(String visExpression, boolean withSerializationFormat,
231      boolean checkAuths) throws IOException {
232    ExpressionNode node = null;
233    try {
234      node = this.expressionParser.parse(visExpression);
235    } catch (ParseException e) {
236      throw new IOException(e);
237    }
238    node = this.expressionExpander.expand(node);
239    List<Tag> tags = new ArrayList<>();
240    if (withSerializationFormat) {
241      tags.add(STRING_SERIALIZATION_FORMAT_TAG);
242    }
243    if (node instanceof NonLeafExpressionNode
244        && ((NonLeafExpressionNode) node).getOperator() == Operator.OR) {
245      for (ExpressionNode child : ((NonLeafExpressionNode) node).getChildExps()) {
246        tags.add(createTag(child));
247      }
248    } else {
249      tags.add(createTag(node));
250    }
251    return tags;
252  }

254  @Override
255  public VisibilityExpEvaluator getVisibilityExpEvaluator(Authorizations authorizations)
256      throws IOException {
257    // If a super user issues a get/scan, he should be able to scan the cells
258    // irrespective of the Visibility labels
259    if (isReadFromSystemAuthUser()) {
260      return new VisibilityExpEvaluator() {
261        @Override
262        public boolean evaluate(Cell cell) throws IOException {
263          return true;
264        }
265      };
266    }
267    List<String> authLabels = null;
268    for (ScanLabelGenerator scanLabelGenerator : scanLabelGenerators) {
269      try {
270        // null authorizations to be handled inside SLG impl.
271        authLabels = scanLabelGenerator.getLabels(VisibilityUtils.getActiveUser(), authorizations);
272        authLabels = (authLabels == null) ? new ArrayList<>() : authLabels;
273        authorizations = new Authorizations(authLabels);
274      } catch (Throwable t) {
275        LOG.error(t);
276        throw new IOException(t);
277      }
278    }
279    final List<String> authLabelsFinal = authLabels;
280    return new VisibilityExpEvaluator() {
281      @Override
282      public boolean evaluate(Cell cell) throws IOException {
283        boolean visibilityTagPresent = false;
284        // Save an object allocation where we can
285        if (cell.getTagsLength() > 0) {
286          Iterator<Tag> tagsItr = PrivateCellUtil.tagsIterator(cell);
287          while (tagsItr.hasNext()) {
288            boolean includeKV = true;
289            Tag tag = tagsItr.next();
290            if (tag.getType() == VISIBILITY_TAG_TYPE) {
291              visibilityTagPresent = true;
292              int offset = tag.getValueOffset();
293              int endOffset = offset + tag.getValueLength();
294              while (offset < endOffset) {
295                short len = getTagValuePartAsShort(tag, offset);
296                offset += 2;
297                if (len < 0) {
298                  // This is a NOT label.
299                  len = (short) (-1 * len);
300                  String label = getTagValuePartAsString(tag, offset, len);
301                  if (authLabelsFinal.contains(label)) {
302                    includeKV = false;
303                    break;
304                  }
305                } else {
306                  String label = getTagValuePartAsString(tag, offset, len);
307                  if (!authLabelsFinal.contains(label)) {
308                    includeKV = false;
309                    break;
310                  }
311                }
312                offset += len;
313              }
314              if (includeKV) {
315                // We got one visibility expression getting evaluated to true.
316                // Good to include this
317                // KV in the result then.
318                return true;
319              }
320            }
321          }
322        }
323        return !(visibilityTagPresent);
324      }
325    };
326  }

328  protected boolean isReadFromSystemAuthUser() throws IOException {
329    User user = VisibilityUtils.getActiveUser();
330    return havingSystemAuth(user);
331  }

333  private Tag createTag(ExpressionNode node) throws IOException {
334    ByteArrayOutputStream baos = new ByteArrayOutputStream();
335    DataOutputStream dos = new DataOutputStream(baos);
336    List<String> labels = new ArrayList<>();
337    List<String> notLabels = new ArrayList<>();
338    extractLabels(node, labels, notLabels);
339    Collections.sort(labels);
340    Collections.sort(notLabels);
341    // We will write the NOT labels 1st followed by normal labels
342    // Each of the label we will write with label length (as short 1st) followed
343    // by the label bytes.
344    // For a NOT node we will write the label length as -ve.
345    for (String label : notLabels) {
346      byte[] bLabel = Bytes.toBytes(label);
347      short length = (short) bLabel.length;
348      length = (short) (-1 * length);
349      dos.writeShort(length);
350      dos.write(bLabel);
351    }
352    for (String label : labels) {
353      byte[] bLabel = Bytes.toBytes(label);
354      dos.writeShort(bLabel.length);
355      dos.write(bLabel);
356    }
357    return new ArrayBackedTag(VISIBILITY_TAG_TYPE, baos.toByteArray());
358  }
359
360  private void extractLabels(ExpressionNode node, List<String> labels, List<String> notLabels) {
361    if (node.isSingleNode()) {
362      if (node instanceof NonLeafExpressionNode) {
363        // This is a NOT node.
364        LeafExpressionNode lNode = (LeafExpressionNode) ((NonLeafExpressionNode) node)
365            .getChildExps().get(0);
366        notLabels.add(lNode.getIdentifier());
367      } else {
368        labels.add(((LeafExpressionNode) node).getIdentifier());
369      }
370    } else {
371      // A non leaf expression of labels with & operator.
372      NonLeafExpressionNode nlNode = (NonLeafExpressionNode) node;
373      assert nlNode.getOperator() == Operator.AND;
374      List<ExpressionNode> childExps = nlNode.getChildExps();
375      for (ExpressionNode child : childExps) {
376        extractLabels(child, labels, notLabels);
377      }
378    }
379  }

399  @Override
400  public boolean havingSystemAuth(User user) throws IOException {
401    if (Superusers.isSuperUser(user)) {
402      return true;
403    }
404    Set<String> auths = new HashSet<>();
405    auths.addAll(this.getUserAuths(Bytes.toBytes(user.getShortName()), true));
406    auths.addAll(this.getGroupAuths(user.getGroupNames(), true));
407    return auths.contains(SYSTEM_LABEL);
408  }

410  @Override
411  public boolean matchVisibility(List<Tag> putTags, Byte putTagsFormat, List<Tag> deleteTags,
412      Byte deleteTagsFormat) throws IOException {
413    assert putTagsFormat == STRING_SERIALIZATION_FORMAT;
414    assert deleteTagsFormat == STRING_SERIALIZATION_FORMAT;
415    return checkForMatchingVisibilityTagsWithSortedOrder(putTags, deleteTags);
416  }
417
418  private static boolean checkForMatchingVisibilityTagsWithSortedOrder(List<Tag> putVisTags,
419      List<Tag> deleteVisTags) {
420    // Early out if there are no tags in both of cell and delete
421    if (putVisTags.isEmpty() && deleteVisTags.isEmpty()) {
422      return true;
423    }
424    boolean matchFound = false;
425    // If the size does not match. Definitely we are not comparing the equal
426    // tags.
427    if ((deleteVisTags.size()) == putVisTags.size()) {
428      for (Tag tag : deleteVisTags) {
429        matchFound = false;
430        for (Tag givenTag : putVisTags) {
431          if (Tag.matchingValue(tag, givenTag)) {
432            matchFound = true;
433            break;
434          }
435        }
436        if (!matchFound)
437          break;
438      }
439    }
440    return matchFound;
441  }

443  @Override
444  public byte[] encodeVisibilityForReplication(final List<Tag> tags, final Byte serializationFormat)
445      throws IOException {
446    if (tags.size() > 0 && (serializationFormat == null
447        || serializationFormat == STRING_SERIALIZATION_FORMAT)) {
448      return createModifiedVisExpression(tags);
449    }
450    return null;
451  }

453  /**
454   * @param tags - all the tags associated with the current Cell
455   * @return - the modified visibility expression as byte[]
456   */
457  private byte[] createModifiedVisExpression(final List<Tag> tags)
458      throws IOException {
459    StringBuilder visibilityString = new StringBuilder();
460    for (Tag tag : tags) {
461      if (tag.getType() == TagType.VISIBILITY_TAG_TYPE) {
462        if (visibilityString.length() != 0) {
463          visibilityString.append(VisibilityConstants.CLOSED_PARAN
464              + VisibilityConstants.OR_OPERATOR);
465        }
466        int offset = tag.getValueOffset();
467        int endOffset = offset + tag.getValueLength();
468        boolean expressionStart = true;
469        while (offset < endOffset) {
470          short len = getTagValuePartAsShort(tag, offset);
471          offset += 2;
472          if (len < 0) {
473            len = (short) (-1 * len);
474            String label = getTagValuePartAsString(tag, offset, len);
475            if (expressionStart) {
476              visibilityString.append(VisibilityConstants.OPEN_PARAN
477                  + VisibilityConstants.NOT_OPERATOR + CellVisibility.quote(label));
478            } else {
479              visibilityString.append(VisibilityConstants.AND_OPERATOR
480                  + VisibilityConstants.NOT_OPERATOR + CellVisibility.quote(label));
481            }
482          } else {
483            String label = getTagValuePartAsString(tag, offset, len);
484            if (expressionStart) {
485              visibilityString.append(VisibilityConstants.OPEN_PARAN + CellVisibility.quote(label));
486            } else {
487              visibilityString.append(VisibilityConstants.AND_OPERATOR
488                  + CellVisibility.quote(label));
489            }
490          }
491          expressionStart = false;
492          offset += len;
493        }
494      }
495    }
496    if (visibilityString.length() != 0) {
497      visibilityString.append(VisibilityConstants.CLOSED_PARAN);
498      // Return the string formed as byte[]
499      return Bytes.toBytes(visibilityString.toString());
500    }
501    return null;
502  }
503
504  private static short getTagValuePartAsShort(Tag t, int offset) {
505    if (t.hasArray()) {
506      return Bytes.toShort(t.getValueArray(), offset);
507    }
508    return ByteBufferUtils.toShort(t.getValueByteBuffer(), offset);
509  }
510
511  private static String getTagValuePartAsString(Tag t, int offset, int length) {
512    if (t.hasArray()) {
513      return Bytes.toString(t.getValueArray(), offset, length);
514    }
515    byte[] b = new byte[length];
516    ByteBufferUtils.copyFromBufferToArray(b, t.getValueByteBuffer(), offset, 0, length);
517    return Bytes.toString(b);
518  }
