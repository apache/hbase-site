081  @Test
082  public void testAddLabels() throws Throwable {
083    PrivilegedExceptionAction<VisibilityLabelsResponse> action =
084        new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
085      public VisibilityLabelsResponse run() throws Exception {
086        String[] labels = { "L1", SECRET, "L2", "invalid~", "L3" };
087        VisibilityLabelsResponse response = null;
088        try (Connection conn = ConnectionFactory.createConnection(conf)) {
089          response = VisibilityClient.addLabels(conn, labels);
090        } catch (Throwable e) {
091          fail("Should not have thrown exception");
092        }
093        List<RegionActionResult> resultList = response.getResultList();
094        assertEquals(5, resultList.size());
095        assertTrue(resultList.get(0).getException().getValue().isEmpty());
096        assertEquals("org.apache.hadoop.hbase.DoNotRetryIOException", resultList.get(1)
097            .getException().getName());
098        assertTrue(Bytes.toString(resultList.get(1).getException().getValue().toByteArray())
099            .contains(
100                "org.apache.hadoop.hbase.security.visibility.LabelAlreadyExistsException: "
101                    + "Label 'secret' already exists"));
102        assertTrue(resultList.get(2).getException().getValue().isEmpty());
103        assertTrue(resultList.get(3).getException().getValue().isEmpty());
104        assertTrue(resultList.get(4).getException().getValue().isEmpty());
105        return null;
106      }
107    };
108    SUPERUSER.runAs(action);
109  }

111  @Test(timeout = 60 * 1000)
112  public void testAddVisibilityLabelsOnRSRestart() throws Exception {
113    List<RegionServerThread> regionServerThreads = TEST_UTIL.getHBaseCluster()
114        .getRegionServerThreads();
115    for (RegionServerThread rsThread : regionServerThreads) {
116      rsThread.getRegionServer().abort("Aborting ");
117    }
118    // Start one new RS
119    RegionServerThread rs = TEST_UTIL.getHBaseCluster().startRegionServer();
120    waitForLabelsRegionAvailability(rs.getRegionServer());
121    final AtomicBoolean vcInitialized = new AtomicBoolean(true);
122    do {
123      PrivilegedExceptionAction<VisibilityLabelsResponse> action =
124          new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
125        public VisibilityLabelsResponse run() throws Exception {
126          String[] labels = { SECRET, CONFIDENTIAL, PRIVATE, "ABC", "XYZ" };
127          try (Connection conn = ConnectionFactory.createConnection(conf)) {
128            VisibilityLabelsResponse resp = VisibilityClient.addLabels(conn, labels);
129            List<RegionActionResult> results = resp.getResultList();
130            if (results.get(0).hasException()) {
131              NameBytesPair pair = results.get(0).getException();
132              Throwable t = ProtobufUtil.toException(pair);
133              LOG.debug("Got exception writing labels", t);
134              if (t instanceof VisibilityControllerNotReadyException) {
135                vcInitialized.set(false);
136                LOG.warn("VisibilityController was not yet initialized");
137                Threads.sleep(10);
138              } else {
139                vcInitialized.set(true);
140              }
141            } else LOG.debug("new labels added: " + resp);
142          } catch (Throwable t) {
143            throw new IOException(t);
144          }
145          return null;
146        }
147      };
148      SUPERUSER.runAs(action);
149    } while (!vcInitialized.get());
150    // Scan the visibility label
151    Scan s = new Scan();
152    s.setAuthorizations(new Authorizations(VisibilityUtils.SYSTEM_LABEL));
153
154    int i = 0;
155    try (Table ht = TEST_UTIL.getConnection().getTable(LABELS_TABLE_NAME);
156         ResultScanner scanner = ht.getScanner(s)) {
157      while (true) {
158        Result next = scanner.next();
159        if (next == null) {
160          break;
161        }
162        i++;
163      }
164    }
165    // One label is the "system" label.
166    Assert.assertEquals("The count should be 13", 13, i);
167  }

169  @Test
170  public void testListLabels() throws Throwable {
171    PrivilegedExceptionAction<ListLabelsResponse> action =
172        new PrivilegedExceptionAction<ListLabelsResponse>() {
173      public ListLabelsResponse run() throws Exception {
174        ListLabelsResponse response = null;
175        try (Connection conn = ConnectionFactory.createConnection(conf)) {
176          response = VisibilityClient.listLabels(conn, null);
177        } catch (Throwable e) {
178          fail("Should not have thrown exception");
179        }
180        // The addLabels() in setup added:
181        // { SECRET, TOPSECRET, CONFIDENTIAL, PUBLIC, PRIVATE, COPYRIGHT, ACCENT,
182        //  UNICODE_VIS_TAG, UC1, UC2 };
183        // The previous tests added 2 more labels: ABC, XYZ
184        // The 'system' label is excluded.
185        List<ByteString> labels = response.getLabelList();
186        assertEquals(12, labels.size());
187        assertTrue(labels.contains(ByteString.copyFrom(SECRET.getBytes())));
188        assertTrue(labels.contains(ByteString.copyFrom(TOPSECRET.getBytes())));
189        assertTrue(labels.contains(ByteString.copyFrom(CONFIDENTIAL.getBytes())));
190        assertTrue(labels.contains(ByteString.copyFrom("ABC".getBytes())));
191        assertTrue(labels.contains(ByteString.copyFrom("XYZ".getBytes())));
192        assertFalse(labels.contains(ByteString.copyFrom(SYSTEM_LABEL.getBytes())));
193        return null;
194      }
195    };
196    SUPERUSER.runAs(action);
197  }
