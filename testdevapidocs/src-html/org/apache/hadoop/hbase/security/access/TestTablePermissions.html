123  /**
124   * The AccessControlLists.addUserPermission may throw exception before closing the table.
125   */
126  private void addUserPermission(Configuration conf, UserPermission userPerm, Table t) throws IOException {
127    try {
128      AccessControlLists.addUserPermission(conf, userPerm, t);
129    } finally {
130      t.close();
131    }

134  @Test
135  public void testBasicWrite() throws Exception {
136    Configuration conf = UTIL.getConfiguration();
137    try (Connection connection = ConnectionFactory.createConnection(conf)) {
138      // add some permissions
139      addUserPermission(conf,
140          new UserPermission(Bytes.toBytes("george"), TEST_TABLE, null, (byte[])null,
141              UserPermission.Action.READ, UserPermission.Action.WRITE),
142              connection.getTable(AccessControlLists.ACL_TABLE_NAME));
143      addUserPermission(conf,
144          new UserPermission(Bytes.toBytes("hubert"), TEST_TABLE, null, (byte[])null,
145              UserPermission.Action.READ),
146          connection.getTable(AccessControlLists.ACL_TABLE_NAME));
147      addUserPermission(conf,
148          new UserPermission(Bytes.toBytes("humphrey"),
149              TEST_TABLE, TEST_FAMILY, TEST_QUALIFIER,
150              UserPermission.Action.READ),
151          connection.getTable(AccessControlLists.ACL_TABLE_NAME));
152    }
153    // retrieve the same
154    ListMultimap<String,TablePermission> perms =
155        AccessControlLists.getTablePermissions(conf, TEST_TABLE);
156    List<TablePermission> userPerms = perms.get("george");
157    assertNotNull("Should have permissions for george", userPerms);
158    assertEquals("Should have 1 permission for george", 1, userPerms.size());
159    TablePermission permission = userPerms.get(0);
160    assertEquals("Permission should be for " + TEST_TABLE,
161        TEST_TABLE, permission.getTableName());
162    assertNull("Column family should be empty", permission.getFamily());
163
164    // check actions
165    assertNotNull(permission.getActions());
166    assertEquals(2, permission.getActions().length);
167    List<TablePermission.Action> actions = Arrays.asList(permission.getActions());
168    assertTrue(actions.contains(TablePermission.Action.READ));
169    assertTrue(actions.contains(TablePermission.Action.WRITE));
170
171    userPerms = perms.get("hubert");
172    assertNotNull("Should have permissions for hubert", userPerms);
173    assertEquals("Should have 1 permission for hubert", 1, userPerms.size());
174    permission = userPerms.get(0);
175    assertEquals("Permission should be for " + TEST_TABLE,
176        TEST_TABLE, permission.getTableName());
177    assertNull("Column family should be empty", permission.getFamily());
178
179    // check actions
180    assertNotNull(permission.getActions());
181    assertEquals(1, permission.getActions().length);
182    actions = Arrays.asList(permission.getActions());
183    assertTrue(actions.contains(TablePermission.Action.READ));
184    assertFalse(actions.contains(TablePermission.Action.WRITE));
185
186    userPerms = perms.get("humphrey");
187    assertNotNull("Should have permissions for humphrey", userPerms);
188    assertEquals("Should have 1 permission for humphrey", 1, userPerms.size());
189    permission = userPerms.get(0);
190    assertEquals("Permission should be for " + TEST_TABLE,
191        TEST_TABLE, permission.getTableName());
192    assertTrue("Permission should be for family " + TEST_FAMILY,
193        Bytes.equals(TEST_FAMILY, permission.getFamily()));
194    assertTrue("Permission should be for qualifier " + TEST_QUALIFIER,
195        Bytes.equals(TEST_QUALIFIER, permission.getQualifier()));
196
197    // check actions
198    assertNotNull(permission.getActions());
199    assertEquals(1, permission.getActions().length);
200    actions = Arrays.asList(permission.getActions());
201    assertTrue(actions.contains(TablePermission.Action.READ));
202    assertFalse(actions.contains(TablePermission.Action.WRITE));
203
204    // table 2 permissions
205    try (Connection connection = ConnectionFactory.createConnection(conf);
206        Table table = connection.getTable(AccessControlLists.ACL_TABLE_NAME)) {
207      AccessControlLists.addUserPermission(conf,
208          new UserPermission(Bytes.toBytes("hubert"), TEST_TABLE2, null, (byte[])null,
209              TablePermission.Action.READ, TablePermission.Action.WRITE), table);
210    }
211    // check full load
212    Map<byte[], ListMultimap<String,TablePermission>> allPerms =
213        AccessControlLists.loadAll(conf);
214    assertEquals("Full permission map should have entries for both test tables",
215        2, allPerms.size());
216
217    userPerms = allPerms.get(TEST_TABLE.getName()).get("hubert");
218    assertNotNull(userPerms);
219    assertEquals(1, userPerms.size());
220    permission = userPerms.get(0);
221    assertEquals(TEST_TABLE, permission.getTableName());
222    assertEquals(1, permission.getActions().length);
223    assertEquals(TablePermission.Action.READ, permission.getActions()[0]);
224
225    userPerms = allPerms.get(TEST_TABLE2.getName()).get("hubert");
226    assertNotNull(userPerms);
227    assertEquals(1, userPerms.size());
228    permission = userPerms.get(0);
229    assertEquals(TEST_TABLE2, permission.getTableName());
230    assertEquals(2, permission.getActions().length);
231    actions = Arrays.asList(permission.getActions());
232    assertTrue(actions.contains(TablePermission.Action.READ));
233    assertTrue(actions.contains(TablePermission.Action.WRITE));
234  }

236  @Test
237  public void testPersistence() throws Exception {
238    Configuration conf = UTIL.getConfiguration();
239    try (Connection connection = ConnectionFactory.createConnection(conf)) {
240      addUserPermission(conf,
241          new UserPermission(Bytes.toBytes("albert"), TEST_TABLE, null,
242              (byte[])null, TablePermission.Action.READ), connection.getTable(AccessControlLists.ACL_TABLE_NAME));
243      addUserPermission(conf,
244          new UserPermission(Bytes.toBytes("betty"), TEST_TABLE, null,
245              (byte[])null, TablePermission.Action.READ,
246              TablePermission.Action.WRITE), connection.getTable(AccessControlLists.ACL_TABLE_NAME));
247      addUserPermission(conf,
248          new UserPermission(Bytes.toBytes("clark"),
249              TEST_TABLE, TEST_FAMILY,
250              TablePermission.Action.READ), connection.getTable(AccessControlLists.ACL_TABLE_NAME));
251      addUserPermission(conf,
252          new UserPermission(Bytes.toBytes("dwight"),
253              TEST_TABLE, TEST_FAMILY, TEST_QUALIFIER,
254              TablePermission.Action.WRITE), connection.getTable(AccessControlLists.ACL_TABLE_NAME));
255    }
256    // verify permissions survive changes in table metadata
257    ListMultimap<String,TablePermission> preperms =
258        AccessControlLists.getTablePermissions(conf, TEST_TABLE);
259
260    Table table = UTIL.getConnection().getTable(TEST_TABLE);
261    table.put(new Put(Bytes.toBytes("row1"))
262            .addColumn(TEST_FAMILY, TEST_QUALIFIER, Bytes.toBytes("v1")));
263    table.put(new Put(Bytes.toBytes("row2"))
264            .addColumn(TEST_FAMILY, TEST_QUALIFIER, Bytes.toBytes("v2")));
265    Admin admin = UTIL.getAdmin();
266    try {
267      admin.split(TEST_TABLE);
268    }
269    catch (IOException e) {
270      //although split fail, this may not affect following check
271      //In old Split API without AM2, if region's best split key is not found,
272      //there are not exception thrown. But in current API, exception
273      //will be thrown.
274      LOG.debug("region is not splittable, because " + e);
275    }
276
277    // wait for split
278    Thread.sleep(10000);
279
280    ListMultimap<String,TablePermission> postperms =
281        AccessControlLists.getTablePermissions(conf, TEST_TABLE);
282
283    checkMultimapEqual(preperms, postperms);
284  }

286  @Test
287  public void testSerialization() throws Exception {
288    Configuration conf = UTIL.getConfiguration();
289    ListMultimap<String,TablePermission> permissions = createPermissions();
290    byte[] permsData = AccessControlLists.writePermissionsAsBytes(permissions, conf);
291
292    ListMultimap<String, TablePermission> copy =
293        AccessControlLists.readPermissions(permsData, conf);
294
295    checkMultimapEqual(permissions, copy);
296  }

298  private ListMultimap<String,TablePermission> createPermissions() {
299    ListMultimap<String,TablePermission> permissions = ArrayListMultimap.create();
300    permissions.put("george", new TablePermission(TEST_TABLE, null,
301        TablePermission.Action.READ));
302    permissions.put("george", new TablePermission(TEST_TABLE, TEST_FAMILY,
303        TablePermission.Action.WRITE));
304    permissions.put("george", new TablePermission(TEST_TABLE2, null,
305        TablePermission.Action.READ));
306    permissions.put("hubert", new TablePermission(TEST_TABLE2, null,
307        TablePermission.Action.READ, TablePermission.Action.WRITE));
308    permissions.put("bruce",new TablePermission(TEST_NAMESPACE,
309        TablePermission.Action.READ));
310    return permissions;
311  }

313  public void checkMultimapEqual(ListMultimap<String,TablePermission> first,
314      ListMultimap<String,TablePermission> second) {
315    assertEquals(first.size(), second.size());
316    for (String key : first.keySet()) {
317      List<TablePermission> firstPerms = first.get(key);
318      List<TablePermission> secondPerms = second.get(key);
319      assertNotNull(secondPerms);
320      assertEquals(firstPerms.size(), secondPerms.size());
321      LOG.info("First permissions: "+firstPerms.toString());
322      LOG.info("Second permissions: "+secondPerms.toString());
323      for (TablePermission p : firstPerms) {
324        assertTrue("Permission "+p.toString()+" not found", secondPerms.contains(p));
325      }
326    }
327  }

329  @Test
330  public void testEquals() throws Exception {
331    TablePermission p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
332    TablePermission p2 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
333    assertTrue(p1.equals(p2));
334    assertTrue(p2.equals(p1));
335
336    p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ, TablePermission.Action.WRITE);
337    p2 = new TablePermission(TEST_TABLE, null, TablePermission.Action.WRITE, TablePermission.Action.READ);
338    assertTrue(p1.equals(p2));
339    assertTrue(p2.equals(p1));
340
341    p1 = new TablePermission(TEST_TABLE, TEST_FAMILY, TablePermission.Action.READ, TablePermission.Action.WRITE);
342    p2 = new TablePermission(TEST_TABLE, TEST_FAMILY, TablePermission.Action.WRITE, TablePermission.Action.READ);
343    assertTrue(p1.equals(p2));
344    assertTrue(p2.equals(p1));
345
346    p1 = new TablePermission(TEST_TABLE, TEST_FAMILY, TEST_QUALIFIER, TablePermission.Action.READ, TablePermission.Action.WRITE);
347    p2 = new TablePermission(TEST_TABLE, TEST_FAMILY, TEST_QUALIFIER, TablePermission.Action.WRITE, TablePermission.Action.READ);
348    assertTrue(p1.equals(p2));
349    assertTrue(p2.equals(p1));
350
351    p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
352    p2 = new TablePermission(TEST_TABLE, TEST_FAMILY, TablePermission.Action.READ);
353    assertFalse(p1.equals(p2));
354    assertFalse(p2.equals(p1));
355
356    p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
357    p2 = new TablePermission(TEST_TABLE, null, TablePermission.Action.WRITE);
358    assertFalse(p1.equals(p2));
359    assertFalse(p2.equals(p1));
360    p2 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ, TablePermission.Action.WRITE);
361    assertFalse(p1.equals(p2));
362    assertFalse(p2.equals(p1));
363
364    p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
365    p2 = new TablePermission(TEST_TABLE2, null, TablePermission.Action.READ);
366    assertFalse(p1.equals(p2));
367    assertFalse(p2.equals(p1));
368
369    p2 = new TablePermission(TEST_TABLE, null);
370    assertFalse(p1.equals(p2));
371    assertFalse(p2.equals(p1));
372
373    p1 = new TablePermission(TEST_NAMESPACE, TablePermission.Action.READ);
374    p2 = new TablePermission(TEST_NAMESPACE, TablePermission.Action.READ);
375    assertEquals(p1, p2);
376
377    p1 = new TablePermission(TEST_NAMESPACE, TablePermission.Action.READ);
378    p2 = new TablePermission(TEST_NAMESPACE2, TablePermission.Action.READ);
379    assertFalse(p1.equals(p2));
380    assertFalse(p2.equals(p1));
381  }

383  @Test
384  public void testGlobalPermission() throws Exception {
385    Configuration conf = UTIL.getConfiguration();
386
387    // add some permissions
388    try (Connection connection = ConnectionFactory.createConnection(conf)) {
389      addUserPermission(conf,
390          new UserPermission(Bytes.toBytes("user1"),
391              Permission.Action.READ, Permission.Action.WRITE), connection.getTable(AccessControlLists.ACL_TABLE_NAME));
392      addUserPermission(conf,
393          new UserPermission(Bytes.toBytes("user2"),
394              Permission.Action.CREATE), connection.getTable(AccessControlLists.ACL_TABLE_NAME));
395      addUserPermission(conf,
396          new UserPermission(Bytes.toBytes("user3"),
397              Permission.Action.ADMIN, Permission.Action.READ, Permission.Action.CREATE),
398          connection.getTable(AccessControlLists.ACL_TABLE_NAME));
399    }
400    ListMultimap<String,TablePermission> perms = AccessControlLists.getTablePermissions(conf, null);
401    List<TablePermission> user1Perms = perms.get("user1");
402    assertEquals("Should have 1 permission for user1", 1, user1Perms.size());
403    assertEquals("user1 should have WRITE permission",
404                 new Permission.Action[] { Permission.Action.READ, Permission.Action.WRITE },
405                 user1Perms.get(0).getActions());
406
407    List<TablePermission> user2Perms = perms.get("user2");
408    assertEquals("Should have 1 permission for user2", 1, user2Perms.size());
409    assertEquals("user2 should have CREATE permission",
410                 new Permission.Action[] { Permission.Action.CREATE },
411                 user2Perms.get(0).getActions());
412
413    List<TablePermission> user3Perms = perms.get("user3");
414    assertEquals("Should have 1 permission for user3", 1, user3Perms.size());
415    assertEquals("user3 should have ADMIN, READ, CREATE permission",
416                 new Permission.Action[] {
417                    Permission.Action.READ, Permission.Action.CREATE, Permission.Action.ADMIN
418                 },
419                 user3Perms.get(0).getActions());
420  }

422  @Test
423  public void testAuthManager() throws Exception {
424    Configuration conf = UTIL.getConfiguration();
425    /* test a race condition causing TableAuthManager to sometimes fail global permissions checks
426     * when the global cache is being updated
427     */
428    TableAuthManager authManager = TableAuthManager.getOrCreate(ZKW, conf);
429    // currently running user is the system user and should have global admin perms
430    User currentUser = User.getCurrent();
431    assertTrue(authManager.authorize(currentUser, Permission.Action.ADMIN));
432    try (Connection connection = ConnectionFactory.createConnection(conf)) {
433      for (int i=1; i<=50; i++) {
434        addUserPermission(conf, new UserPermission(Bytes.toBytes("testauth"+i),
435            Permission.Action.ADMIN, Permission.Action.READ, Permission.Action.WRITE),
436            connection.getTable(AccessControlLists.ACL_TABLE_NAME));
437        // make sure the system user still shows as authorized
438        assertTrue("Failed current user auth check on iter "+i,
439            authManager.authorize(currentUser, Permission.Action.ADMIN));
440      }
441    }
442  }
