154  @Test (timeout=120000)
155  public void testCellPermissions() throws Exception {
156    // store two sets of values, one store with a cell level ACL, and one without
157    verifyAllowed(new AccessTestAction() {
158      @Override
159      public Object run() throws Exception {
160        try(Connection connection = ConnectionFactory.createConnection(conf);
161            Table t = connection.getTable(TEST_TABLE.getTableName())) {
162          Put p;
163          // with ro ACL
164          p = new Put(TEST_ROW).addColumn(TEST_FAMILY, TEST_Q1, ZERO);
165          p.setACL(prepareCellPermissions(usersAndGroups, Action.READ));
166          t.put(p);
167          // with rw ACL
168          p = new Put(TEST_ROW).addColumn(TEST_FAMILY, TEST_Q2, ZERO);
169          p.setACL(prepareCellPermissions(usersAndGroups, Action.READ, Action.WRITE));
170          t.put(p);
171          // no ACL
172          p = new Put(TEST_ROW)
173              .addColumn(TEST_FAMILY, TEST_Q3, ZERO)
174              .addColumn(TEST_FAMILY, TEST_Q4, ZERO);
175          t.put(p);
176        }
177        return null;
178      }
179    }, USER_OWNER);
180
181    /* ---- Gets ---- */
182
183    AccessTestAction getQ1 = new AccessTestAction() {
184      @Override
185      public Object run() throws Exception {
186        Get get = new Get(TEST_ROW).addColumn(TEST_FAMILY, TEST_Q1);
187        try(Connection connection = ConnectionFactory.createConnection(conf);
188            Table t = connection.getTable(TEST_TABLE.getTableName())) {
189          return t.get(get).listCells();
190        }
191      }
192    };
193
194    AccessTestAction getQ2 = new AccessTestAction() {
195      @Override
196      public Object run() throws Exception {
197        Get get = new Get(TEST_ROW).addColumn(TEST_FAMILY, TEST_Q2);
198        try(Connection connection = ConnectionFactory.createConnection(conf);
199            Table t = connection.getTable(TEST_TABLE.getTableName())) {
200          return t.get(get).listCells();
201        }
202      }
203    };
204
205    AccessTestAction getQ3 = new AccessTestAction() {
206      @Override
207      public Object run() throws Exception {
208        Get get = new Get(TEST_ROW).addColumn(TEST_FAMILY, TEST_Q3);
209        try(Connection connection = ConnectionFactory.createConnection(conf);
210            Table t = connection.getTable(TEST_TABLE.getTableName())) {
211          return t.get(get).listCells();
212        }
213      }
214    };
215
216    AccessTestAction getQ4 = new AccessTestAction() {
217      @Override
218      public Object run() throws Exception {
219        Get get = new Get(TEST_ROW).addColumn(TEST_FAMILY, TEST_Q4);
220        try(Connection connection = ConnectionFactory.createConnection(conf);
221            Table t = connection.getTable(TEST_TABLE.getTableName())) {
222          return t.get(get).listCells();
223        }
224      }
225    };
226
227    // Confirm special read access set at cell level
228
229    verifyAllowed(getQ1, USER_OTHER, GROUP_USER);
230    verifyAllowed(getQ2, USER_OTHER, GROUP_USER);
231
232    // Confirm this access does not extend to other cells
233
234    verifyIfNull(getQ3, USER_OTHER, GROUP_USER);
235    verifyIfNull(getQ4, USER_OTHER, GROUP_USER);
236
237    /* ---- Scans ---- */
238
239    // check that a scan over the test data returns the expected number of KVs
240
241    final List<Cell> scanResults = Lists.newArrayList();
242
243    AccessTestAction scanAction = new AccessTestAction() {
244      @Override
245      public List<Cell> run() throws Exception {
246        Scan scan = new Scan();
247        scan.setStartRow(TEST_ROW);
248        scan.setStopRow(Bytes.add(TEST_ROW, new byte[]{ 0 } ));
249        scan.addFamily(TEST_FAMILY);
250        Connection connection = ConnectionFactory.createConnection(conf);
251        Table t = connection.getTable(TEST_TABLE.getTableName());
252        try {
253          ResultScanner scanner = t.getScanner(scan);
254          Result result = null;
255          do {
256            result = scanner.next();
257            if (result != null) {
258              scanResults.addAll(result.listCells());
259            }
260          } while (result != null);
261        } finally {
262          t.close();
263          connection.close();
264        }
265        return scanResults;
266      }
267    };
268
269    // owner will see all values
270    scanResults.clear();
271    verifyAllowed(scanAction, USER_OWNER);
272    assertEquals(4, scanResults.size());
273
274    // other user will see 2 values
275    scanResults.clear();
276    verifyAllowed(scanAction, USER_OTHER);
277    assertEquals(2, scanResults.size());
278
279    scanResults.clear();
280    verifyAllowed(scanAction, GROUP_USER);
281    assertEquals(2, scanResults.size());

341    verifyAllowed(incrementQ2newDenyACL, USER_OTHER);
342    // But not again after we denied ourselves write permission with an ACL
343    // update
344    verifyDenied(incrementQ2, USER_OTHER, GROUP_USER);

377  /**
378   * Insure we are not granting access in the absence of any cells found
379   * when scanning for covered cells.
380   */
381  @Test (timeout=120000)
382  public void testCoveringCheck() throws Exception {
383    // Grant read access to USER_OTHER
384    grantOnTable(TEST_UTIL, USER_OTHER.getShortName(), TEST_TABLE.getTableName(), TEST_FAMILY,
385      null, Action.READ);
386    // Grant read access to GROUP
387    grantOnTable(TEST_UTIL, AuthUtil.toGroupEntry(GROUP), TEST_TABLE.getTableName(), TEST_FAMILY,
388      null, Action.READ);
389
390    // A write by USER_OTHER should be denied.
391    // This is where we could have a big problem if there is an error in the
392    // covering check logic.
393    verifyUserDeniedForWrite(USER_OTHER, ZERO);
394    // A write by GROUP_USER from group GROUP should be denied.
395    verifyUserDeniedForWrite(GROUP_USER, ZERO);
396
397    // Add the cell
398    verifyAllowed(new AccessTestAction() {
399      @Override
400      public Object run() throws Exception {
401        try(Connection connection = ConnectionFactory.createConnection(conf);
402            Table t = connection.getTable(TEST_TABLE.getTableName())) {
403          Put p;
404          p = new Put(TEST_ROW).addColumn(TEST_FAMILY, TEST_Q1, ZERO);
405          t.put(p);
406        }
407        return null;
408      }
409    }, USER_OWNER);
410
411    // A write by USER_OTHER should still be denied, just to make sure
412    verifyUserDeniedForWrite(USER_OTHER, ONE);
413    // A write by GROUP_USER from group GROUP should still be denied
414    verifyUserDeniedForWrite(GROUP_USER, ONE);
415
416    // A read by USER_OTHER should be allowed, just to make sure
417    verifyUserAllowedForRead(USER_OTHER);
418    // A read by GROUP_USER from group GROUP should be allowed
419    verifyUserAllowedForRead(GROUP_USER);
420  }

430          t.put(p);
431        }
432        return null;
433      }
434    }, user);
435  }

449  private Map<String, Permission> prepareCellPermissions(String[] users, Action... action) {
450    Map<String, Permission> perms = new HashMap<>(2);
451    for (String user : users) {
452      perms.put(user, new Permission(action));
453    }
454    return perms;
455  }
