093  @BeforeClass
094  public static void setupBeforeClass() throws Exception {
095    // enable hdfs acl and set umask to 027
096    conf.setBoolean("dfs.namenode.acls.enabled", true);
097    conf.set("fs.permissions.umask-mode", "027");
098    // enable hbase hdfs acl feature
099    conf.setBoolean(SnapshotScannerHDFSAclHelper.ACL_SYNC_TO_HDFS_ENABLE, true);
100    // enable secure
101    conf.set(User.HBASE_SECURITY_CONF_KEY, "simple");
102    conf.set(SnapshotScannerHDFSAclHelper.SNAPSHOT_RESTORE_TMP_DIR,
103      SnapshotScannerHDFSAclHelper.SNAPSHOT_RESTORE_TMP_DIR_DEFAULT);
104    SecureTestUtil.enableSecurity(conf);
105    // add SnapshotScannerHDFSAclController coprocessor
106    conf.set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY,
107      conf.get(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY) + ","
108          + SnapshotScannerHDFSAclController.class.getName());
109
110    TEST_UTIL.startMiniCluster();
111    admin = TEST_UTIL.getAdmin();
112    rootDir = TEST_UTIL.getDefaultRootDirPath();
113    fs = rootDir.getFileSystem(conf);
114    unGrantUser = User.createUserForTesting(conf, UN_GRANT_USER, new String[] {});
115    helper = new SnapshotScannerHDFSAclHelper(conf, admin.getConnection());
116
117    // set hbase directory permission
118    FsPermission commonDirectoryPermission =
119        new FsPermission(conf.get(SnapshotScannerHDFSAclHelper.COMMON_DIRECTORY_PERMISSION,
120          SnapshotScannerHDFSAclHelper.COMMON_DIRECTORY_PERMISSION_DEFAULT));
121    Path path = rootDir;
122    while (path != null) {
123      fs.setPermission(path, commonDirectoryPermission);
124      path = path.getParent();
125    }
126    // set restore directory permission
127    Path restoreDir = new Path(SnapshotScannerHDFSAclHelper.SNAPSHOT_RESTORE_TMP_DIR_DEFAULT);
128    if (!fs.exists(restoreDir)) {
129      fs.mkdirs(restoreDir);
130      fs.setPermission(restoreDir,
131        new FsPermission(
132            conf.get(SnapshotScannerHDFSAclHelper.SNAPSHOT_RESTORE_DIRECTORY_PERMISSION,
133              SnapshotScannerHDFSAclHelper.SNAPSHOT_RESTORE_DIRECTORY_PERMISSION_DEFAULT)));
134    }
135    path = restoreDir.getParent();
136    while (path != null) {
137      fs.setPermission(path, commonDirectoryPermission);
138      path = path.getParent();
139    }
140
141    SnapshotScannerHDFSAclController coprocessor = TEST_UTIL.getHBaseCluster().getMaster()
142        .getMasterCoprocessorHost().findCoprocessor(SnapshotScannerHDFSAclController.class);

155    User grantUser = User.createUserForTesting(conf, grantUserName, new String[] {});
156    String namespace = name.getMethodName();
157    TableName table = TableName.valueOf(namespace, "t1");
158    String snapshot1 = namespace + "s1";
159    String snapshot2 = namespace + "s2";
160
161    TestHDFSAclHelper.createTableAndPut(TEST_UTIL, table);
162    admin.snapshot(snapshot1, table);
163    // grant G(R)
164    SecureTestUtil.grantGlobal(TEST_UTIL, grantUserName, READ);
165    TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot1, 6);
166    assertTrue(hasUserGlobalHdfsAcl(aclTable, grantUserName));
167    // grant G(W) with merging existing permissions
168    admin.grant(
169      new UserPermission(grantUserName, Permission.newBuilder().withActions(WRITE).build()), true);
170    TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot1, 6);
171    assertTrue(hasUserGlobalHdfsAcl(aclTable, grantUserName));
172    // grant G(W) without merging
173    SecureTestUtil.grantGlobal(TEST_UTIL, grantUserName, WRITE);
174    TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot1, -1);
175    assertFalse(hasUserGlobalHdfsAcl(aclTable, grantUserName));
176    // grant G(R)
177    SecureTestUtil.grantGlobal(TEST_UTIL, grantUserName, READ);
178    TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot1, 6);
179    // take a snapshot and ACLs are inherited automatically
180    admin.snapshot(snapshot2, table);
181    TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot2, 6);
182    assertTrue(hasUserGlobalHdfsAcl(aclTable, grantUserName));
183  }

334  @Test
335  public void testGrantTable() throws Exception {
336    final String grantUserName = name.getMethodName();
337    User grantUser = User.createUserForTesting(conf, grantUserName, new String[] {});
338
339    String namespace = name.getMethodName();
340    TableName table1 = TableName.valueOf(namespace, "t1");
341    String snapshot1 = namespace + "s1";
342    String snapshot2 = namespace + "s2";
343
344    try (Table t = TestHDFSAclHelper.createTable(TEST_UTIL, table1)) {
345      TestHDFSAclHelper.put(t);
346      admin.snapshot(snapshot1, table1);
347      // table owner can scan table snapshot
348      TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL,
349        User.createUserForTesting(conf, "owner", new String[] {}), snapshot1, 6);
350      // grant table1 family(R)
351      SecureTestUtil.grantOnTable(TEST_UTIL, grantUserName, table1, TestHDFSAclHelper.COLUMN1, null,
352        READ);
353      TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot1, -1);
354
355      // grant table1(R)
356      TestHDFSAclHelper.grantOnTable(TEST_UTIL, grantUserName, table1, READ);
357      TestHDFSAclHelper.put2(t);
358      admin.snapshot(snapshot2, table1);
359      // check scan snapshot
360      TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot1, 6);
361      TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot2, 10);
362      assertTrue(hasUserTableHdfsAcl(aclTable, grantUserName, table1));
363      checkUserAclEntry(helper.getTableRootPaths(table1, false), grantUserName, true, true);
364    }
365
366    // grant table1(W) with merging existing permissions
367    admin.grant(
368      new UserPermission(grantUserName, Permission.newBuilder(table1).withActions(WRITE).build()),
369      true);
370    assertTrue(hasUserTableHdfsAcl(aclTable, grantUserName, table1));
371    checkUserAclEntry(helper.getTableRootPaths(table1, false), grantUserName, true, true);
372
373    // grant table1(W) without merging existing permissions
374    TestHDFSAclHelper.grantOnTable(TEST_UTIL, grantUserName, table1, WRITE);
375    TestHDFSAclHelper.canUserScanSnapshot(TEST_UTIL, grantUser, snapshot1, -1);
376    assertFalse(hasUserTableHdfsAcl(aclTable, grantUserName, table1));
377    checkUserAclEntry(helper.getTableRootPaths(table1, false), grantUserName, false, false);
378  }

939    boolean accessAclEntry = false;
940    boolean defaultAclEntry = false;
941    if (fs.exists(path)) {
942      for (AclEntry aclEntry : fs.getAclStatus(path).getEntries()) {
943        String user = aclEntry.getName();
944        if (user != null && user.equals(userName)) {
945          if (aclEntry.getScope() == AclEntryScope.DEFAULT) {
946            defaultAclEntry = true;
947          } else if (aclEntry.getScope() == AclEntryScope.ACCESS) {
948            accessAclEntry = true;
949          }
950        }
951      }
952    }
953    String message = "require user: " + userName + ", path: " + path.toString() + " acl";
954    assertEquals(message, requireAccessAcl, accessAclEntry);
955    assertEquals(message, requireDefaultAcl, defaultAclEntry);
956  }

1062      int expectedRowCount) throws IOException, InterruptedException {

1068  private static PrivilegedExceptionAction<Void> getScanSnapshotAction(Configuration conf,
1069      String snapshotName, long expectedRowCount) {
1070    return () -> {
1071      try {
1072        Path restoreDir = new Path(SnapshotScannerHDFSAclHelper.SNAPSHOT_RESTORE_TMP_DIR_DEFAULT);
1073        Scan scan = new Scan();
1074        TableSnapshotScanner scanner =
1075            new TableSnapshotScanner(conf, restoreDir, snapshotName, scan);
1076        int rowCount = 0;
1077        while (true) {
1078          Result result = scanner.next();
1079          if (result == null) {
1080            break;
1081          }
1082          rowCount++;
1083        }
1084        scanner.close();
1085        assertEquals(expectedRowCount, rowCount);
1086      } catch (Exception e) {
1087        LOG.debug("Scan snapshot error, snapshot {}", snapshotName, e);
1088        assertEquals(expectedRowCount, -1);
1089      }
1090      return null;
1091    };
1092  }
