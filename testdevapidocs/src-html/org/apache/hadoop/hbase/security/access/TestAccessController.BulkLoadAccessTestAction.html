281  private static void setUpTableAndUserPermissions() throws Exception {
282    HTableDescriptor htd = new HTableDescriptor(TEST_TABLE);
283    HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAMILY);
284    hcd.setMaxVersions(100);
285    htd.addFamily(hcd);
286    htd.setOwner(USER_OWNER);
287    createTable(TEST_UTIL, htd, new byte[][] { Bytes.toBytes("s") });
288
289    HRegion region = TEST_UTIL.getHBaseCluster().getRegions(TEST_TABLE).get(0);
290    RegionCoprocessorHost rcpHost = region.getCoprocessorHost();
291    RCP_ENV = rcpHost.createEnvironment(ACCESS_CONTROLLER, Coprocessor.PRIORITY_HIGHEST, 1, conf);
292
293    // Set up initial grants
294
295    grantGlobal(TEST_UTIL, USER_ADMIN.getShortName(),
296      Permission.Action.ADMIN,
297      Permission.Action.CREATE,
298      Permission.Action.READ,
299      Permission.Action.WRITE);
300
301    grantOnTable(TEST_UTIL, USER_RW.getShortName(),
302      TEST_TABLE, TEST_FAMILY, null,
303      Permission.Action.READ,
304      Permission.Action.WRITE);
305
306    // USER_CREATE is USER_RW plus CREATE permissions
307    grantOnTable(TEST_UTIL, USER_CREATE.getShortName(),
308      TEST_TABLE, null, null,
309      Permission.Action.CREATE,
310      Permission.Action.READ,
311      Permission.Action.WRITE);
312
313    grantOnTable(TEST_UTIL, USER_RO.getShortName(),
314      TEST_TABLE, TEST_FAMILY, null,
315      Permission.Action.READ);
316
317    grantOnTable(TEST_UTIL, USER_ADMIN_CF.getShortName(),
318      TEST_TABLE, TEST_FAMILY,
319      null, Permission.Action.ADMIN, Permission.Action.CREATE);
320
321    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_ADMIN), Permission.Action.ADMIN);
322    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_CREATE), Permission.Action.CREATE);
323    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_READ), Permission.Action.READ);
324    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_WRITE), Permission.Action.WRITE);
325
326    assertEquals(5, AccessControlLists.getTablePermissions(conf, TEST_TABLE).size());
327    int size = 0;
328    try {
329      size = AccessControlClient.getUserPermissions(systemUserConnection, TEST_TABLE.toString())
330          .size();
331    } catch (Throwable e) {
332      LOG.error("error during call of AccessControlClient.getUserPermissions. ", e);
333      fail("error during call of AccessControlClient.getUserPermissions.");
334    }
335    assertEquals(5, size);
336  }

1062  public static class BulkLoadHelper {
1063    private final FileSystem fs;
1064    private final Path loadPath;
1065    private final Configuration conf;
1066
1067    public BulkLoadHelper(Path loadPath) throws IOException {
1068      fs = TEST_UTIL.getTestFileSystem();
1069      conf = TEST_UTIL.getConfiguration();
1070      loadPath = loadPath.makeQualified(fs);
1071      this.loadPath = loadPath;
1072    }
1073
1074    private void createHFile(Path path,
1075        byte[] family, byte[] qualifier,
1076        byte[] startKey, byte[] endKey, int numRows) throws IOException {
1077      HFile.Writer writer = null;
1078      long now = System.currentTimeMillis();
1079      try {
1080        HFileContext context = new HFileContextBuilder().build();
1081        writer = HFile.getWriterFactory(conf, new CacheConfig(conf)).withPath(fs, path)
1082            .withFileContext(context).create();
1083        // subtract 2 since numRows doesn't include boundary keys
1084        for (byte[] key : Bytes.iterateOnSplits(startKey, endKey, true, numRows - 2)) {
1085          KeyValue kv = new KeyValue(key, family, qualifier, now, key);
1086          writer.append(kv);
1087        }
1088      } finally {
1089        if (writer != null) {
1090          writer.close();
1091        }
1092      }
1093    }
1094
1095    private BulkLoadHelper initHFileData(byte[] family, byte[] qualifier, byte[][][] hfileRanges,
1096        int numRowsPerRange, FsPermission filePermission) throws Exception {
1097      Path familyDir = new Path(loadPath, Bytes.toString(family));
1098      fs.mkdirs(familyDir);
1099      int hfileIdx = 0;
1100      List<Path> hfiles = new ArrayList<>();
1101      for (byte[][] range : hfileRanges) {
1102        byte[] from = range[0];
1103        byte[] to = range[1];
1104        Path hfile = new Path(familyDir, "hfile_" + (hfileIdx++));
1105        hfiles.add(hfile);
1106        createHFile(hfile, family, qualifier, from, to, numRowsPerRange);
1107      }
1108      // set global read so RegionServer can move it
1109      setPermission(fs, loadPath, FS_PERMISSION_ALL);
1110      // Ensure the file permission as requested.
1111      for (Path hfile : hfiles) {
1112        setPermission(fs, hfile, filePermission);
1113      }
1114      return this;
1115    }
1116
1117    private void bulkLoadHFile(TableName tableName) throws Exception {
1118      try (Connection conn = ConnectionFactory.createConnection(conf);
1119          Admin admin = conn.getAdmin();
1120          RegionLocator locator = conn.getRegionLocator(tableName);
1121          Table table = conn.getTable(tableName)) {
1122        TEST_UTIL.waitUntilAllRegionsAssigned(tableName);
1123        LoadIncrementalHFiles loader = new LoadIncrementalHFiles(conf);
1124        loader.doBulkLoad(loadPath, admin, table, locator);
1125      }
1126    }
1127
1128    private static void setPermission(FileSystem fs, Path dir, FsPermission perm)
1129        throws IOException {
1130      if (!fs.getFileStatus(dir).isDirectory()) {
1131        fs.setPermission(dir, perm);
1132      } else {
1133        for (FileStatus el : fs.listStatus(dir)) {
1134          fs.setPermission(el.getPath(), perm);
1135          setPermission(fs, el.getPath(), perm);
1136        }
1137      }
1138    }
