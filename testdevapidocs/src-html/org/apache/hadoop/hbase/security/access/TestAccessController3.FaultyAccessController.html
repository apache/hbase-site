134  @BeforeClass
135  public static void setupBeforeClass() throws Exception {
136    // setup configuration
137    conf = TEST_UTIL.getConfiguration();
138    // Enable security
139    enableSecurity(conf);
140    String accessControllerClassName = FaultyAccessController.class.getName();
141    // In this particular test case, we can't use SecureBulkLoadEndpoint because its doAs will fail
142    // to move a file for a random user
143    conf.set(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY, accessControllerClassName);
144    // Verify enableSecurity sets up what we require
145    verifyConfiguration(conf);
146
147    // Enable EXEC permission checking
148    conf.setBoolean(AccessControlConstants.EXEC_PERMISSION_CHECKS_KEY, true);
149
150    TEST_UTIL.startMiniCluster();
151    MasterCoprocessorHost cpHost =
152      TEST_UTIL.getMiniHBaseCluster().getMaster().getMasterCoprocessorHost();
153    cpHost.load(FaultyAccessController.class, Coprocessor.PRIORITY_HIGHEST, conf);
154    ACCESS_CONTROLLER = (AccessController) cpHost.findCoprocessor(accessControllerClassName);
155    CP_ENV = cpHost.createEnvironment(ACCESS_CONTROLLER, Coprocessor.PRIORITY_HIGHEST, 1, conf);
156    RegionServerCoprocessorHost rsHost;
157    do {
158      rsHost = TEST_UTIL.getMiniHBaseCluster().getRegionServer(0)
159          .getRegionServerCoprocessorHost();
160    } while (rsHost == null);
161    RSCP_ENV = rsHost.createEnvironment(ACCESS_CONTROLLER, Coprocessor.PRIORITY_HIGHEST, 1, conf);
162
163    // Wait for the ACL table to become available
164    TEST_UTIL.waitUntilAllRegionsAssigned(AccessControlLists.ACL_TABLE_NAME);
165
166    // create a set of test users
167    SUPERUSER = User.createUserForTesting(conf, "admin", new String[] { "supergroup" });
168    USER_ADMIN = User.createUserForTesting(conf, "admin2", new String[0]);
169    USER_RW = User.createUserForTesting(conf, "rwuser", new String[0]);
170    USER_RO = User.createUserForTesting(conf, "rouser", new String[0]);
171    USER_OWNER = User.createUserForTesting(conf, "owner", new String[0]);
172    USER_CREATE = User.createUserForTesting(conf, "tbl_create", new String[0]);
173    USER_NONE = User.createUserForTesting(conf, "nouser", new String[0]);
174    USER_ADMIN_CF = User.createUserForTesting(conf, "col_family_admin", new String[0]);
175
176    USER_GROUP_ADMIN =
177        User.createUserForTesting(conf, "user_group_admin", new String[] { GROUP_ADMIN });
178    USER_GROUP_CREATE =
179        User.createUserForTesting(conf, "user_group_create", new String[] { GROUP_CREATE });
180    USER_GROUP_READ =
181        User.createUserForTesting(conf, "user_group_read", new String[] { GROUP_READ });
182    USER_GROUP_WRITE =
183        User.createUserForTesting(conf, "user_group_write", new String[] { GROUP_WRITE });
184
185    systemUserConnection = TEST_UTIL.getConnection();
186    setUpTableAndUserPermissions();
187  }

189  @AfterClass
190  public static void tearDownAfterClass() throws Exception {
191    HRegionServer rs = null;
192    for (JVMClusterUtil.RegionServerThread thread:
193      TEST_UTIL.getMiniHBaseCluster().getRegionServerThreads()) {
194      rs = thread.getRegionServer();
195    }
196    cleanUp();
197    TEST_UTIL.shutdownMiniCluster();
198    assertTrue("region server should have aborted due to FaultyAccessController", rs.isAborted());
199  }

201  private static void setUpTableAndUserPermissions() throws Exception {
202    HTableDescriptor htd = new HTableDescriptor(TEST_TABLE);
203    HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAMILY);
204    hcd.setMaxVersions(100);
205    htd.addFamily(hcd);
206    htd.setOwner(USER_OWNER);
207    createTable(TEST_UTIL, htd, new byte[][] { Bytes.toBytes("s") });
208
209    HRegion region = TEST_UTIL.getHBaseCluster().getRegions(TEST_TABLE).get(0);
210    RegionCoprocessorHost rcpHost = region.getCoprocessorHost();
211    RCP_ENV = rcpHost.createEnvironment(ACCESS_CONTROLLER, Coprocessor.PRIORITY_HIGHEST, 1, conf);
212
213    // Set up initial grants
214
215    grantGlobal(TEST_UTIL, USER_ADMIN.getShortName(),
216      Permission.Action.ADMIN,
217      Permission.Action.CREATE,
218      Permission.Action.READ,
219      Permission.Action.WRITE);
220
221    grantOnTable(TEST_UTIL, USER_RW.getShortName(),
222      TEST_TABLE, TEST_FAMILY, null,
223      Permission.Action.READ,
224      Permission.Action.WRITE);
225
226    // USER_CREATE is USER_RW plus CREATE permissions
227    grantOnTable(TEST_UTIL, USER_CREATE.getShortName(),
228      TEST_TABLE, null, null,
229      Permission.Action.CREATE,
230      Permission.Action.READ,
231      Permission.Action.WRITE);
232
233    grantOnTable(TEST_UTIL, USER_RO.getShortName(),
234      TEST_TABLE, TEST_FAMILY, null,
235      Permission.Action.READ);
236
237    grantOnTable(TEST_UTIL, USER_ADMIN_CF.getShortName(),
238      TEST_TABLE, TEST_FAMILY,
239      null, Permission.Action.ADMIN, Permission.Action.CREATE);
240
241    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_ADMIN), Permission.Action.ADMIN);
242    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_CREATE), Permission.Action.CREATE);
243    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_READ), Permission.Action.READ);
244    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_WRITE), Permission.Action.WRITE);
245
246    assertEquals(5, AccessControlLists.getTablePermissions(conf, TEST_TABLE).size());
247    try {
248      assertEquals(5, AccessControlClient.getUserPermissions(systemUserConnection,
249          TEST_TABLE.toString()).size());
250    } catch (Throwable e) {
251      LOG.error("error during call of AccessControlClient.getUserPermissions. ", e);
252    }
253  }
254
255  private static void cleanUp() throws Exception {
256    // Clean the _acl_ table
257    // TODO: Skipping delete because of access issues w/ AMv2.
258    // AMv1 seems to crash servers on exit too for same lack of
259    // auth perms but it gets hung up.
260    try {
261      deleteTable(TEST_UTIL, TEST_TABLE);
262    } catch (TableNotFoundException ex) {
263      // Test deleted the table, no problem
264      LOG.info("Test deleted table " + TEST_TABLE);
265    }
266    // Verify all table/namespace permissions are erased
267    assertEquals(0, AccessControlLists.getTablePermissions(conf, TEST_TABLE).size());
268    assertEquals(
269      0,
270      AccessControlLists.getNamespacePermissions(conf,
271        TEST_TABLE.getNamespaceAsString()).size());
272  }

274  @Test
275  public void testTableCreate() throws Exception {
276    AccessTestAction createTable = new AccessTestAction() {
277      @Override
278      public Object run() throws Exception {
279        HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
280        htd.addFamily(new HColumnDescriptor(TEST_FAMILY));
281        ACCESS_CONTROLLER.preCreateTable(ObserverContextImpl.createAndPrepare(CP_ENV), htd, null);
282        return null;
283      }
284    };
285
286    // verify that superuser can create tables
287    verifyAllowed(createTable, SUPERUSER, USER_ADMIN, USER_GROUP_CREATE);
288
289    // all others should be denied
290    verifyDenied(createTable, USER_CREATE, USER_RW, USER_RO, USER_NONE, USER_GROUP_ADMIN,
291      USER_GROUP_READ, USER_GROUP_WRITE);
292  }
