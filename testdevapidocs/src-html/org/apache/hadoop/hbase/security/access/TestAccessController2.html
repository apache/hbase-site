174  @Test (timeout=180000)
175  public void testCreateWithCorrectOwner() throws Exception {
176    // Create a test user
177    final User testUser = User.createUserForTesting(TEST_UTIL.getConfiguration(), "TestUser",
178      new String[0]);
179    // Grant the test user the ability to create tables
180    SecureTestUtil.grantGlobal(TEST_UTIL, testUser.getShortName(), Action.CREATE);
181    verifyAllowed(new AccessTestAction() {
182      @Override
183      public Object run() throws Exception {
184        HTableDescriptor desc = new HTableDescriptor(TEST_TABLE.getTableName());
185        desc.addFamily(new HColumnDescriptor(TEST_FAMILY));
186        try (Connection connection =
187            ConnectionFactory.createConnection(TEST_UTIL.getConfiguration(), testUser)) {
188          try (Admin admin = connection.getAdmin()) {
189            createTable(TEST_UTIL, admin, desc);
190          }
191        }
192        return null;
193      }
194    }, testUser);
195    TEST_UTIL.waitTableAvailable(TEST_TABLE.getTableName());
196    // Verify that owner permissions have been granted to the test user on the
197    // table just created
198    List<TablePermission> perms =
199      AccessControlLists.getTablePermissions(conf, TEST_TABLE.getTableName())
200       .get(testUser.getShortName());
201    assertNotNull(perms);
202    assertFalse(perms.isEmpty());
203    // Should be RWXCA
204    assertTrue(perms.get(0).implies(Permission.Action.READ));
205    assertTrue(perms.get(0).implies(Permission.Action.WRITE));
206    assertTrue(perms.get(0).implies(Permission.Action.EXEC));
207    assertTrue(perms.get(0).implies(Permission.Action.CREATE));
208    assertTrue(perms.get(0).implies(Permission.Action.ADMIN));
209  }

236  @Test (timeout=180000)
237  public void testACLTableAccess() throws Exception {
238    final Configuration conf = TEST_UTIL.getConfiguration();
239
240    // Superuser
241    User superUser = User.createUserForTesting(conf, "admin", new String[] { "supergroup" });
242
243    // Global users
244    User globalRead = User.createUserForTesting(conf, "globalRead", new String[0]);
245    User globalWrite = User.createUserForTesting(conf, "globalWrite", new String[0]);
246    User globalCreate = User.createUserForTesting(conf, "globalCreate", new String[0]);
247    User globalAdmin = User.createUserForTesting(conf, "globalAdmin", new String[0]);
248    SecureTestUtil.grantGlobal(TEST_UTIL, globalRead.getShortName(), Action.READ);
249    SecureTestUtil.grantGlobal(TEST_UTIL, globalWrite.getShortName(), Action.WRITE);
250    SecureTestUtil.grantGlobal(TEST_UTIL, globalCreate.getShortName(), Action.CREATE);
251    SecureTestUtil.grantGlobal(TEST_UTIL, globalAdmin.getShortName(), Action.ADMIN);
252
253    // Namespace users
254    User nsRead = User.createUserForTesting(conf, "nsRead", new String[0]);
255    User nsWrite = User.createUserForTesting(conf, "nsWrite", new String[0]);
256    User nsCreate = User.createUserForTesting(conf, "nsCreate", new String[0]);
257    User nsAdmin = User.createUserForTesting(conf, "nsAdmin", new String[0]);
258    SecureTestUtil.grantOnNamespace(TEST_UTIL, nsRead.getShortName(),
259      TEST_TABLE.getTableName().getNamespaceAsString(), Action.READ);
260    SecureTestUtil.grantOnNamespace(TEST_UTIL, nsWrite.getShortName(),
261      TEST_TABLE.getTableName().getNamespaceAsString(), Action.WRITE);
262    SecureTestUtil.grantOnNamespace(TEST_UTIL, nsCreate.getShortName(),
263      TEST_TABLE.getTableName().getNamespaceAsString(), Action.CREATE);
264    SecureTestUtil.grantOnNamespace(TEST_UTIL, nsAdmin.getShortName(),
265      TEST_TABLE.getTableName().getNamespaceAsString(), Action.ADMIN);
266
267    // Table users
268    User tableRead = User.createUserForTesting(conf, "tableRead", new String[0]);
269    User tableWrite = User.createUserForTesting(conf, "tableWrite", new String[0]);
270    User tableCreate = User.createUserForTesting(conf, "tableCreate", new String[0]);
271    User tableAdmin = User.createUserForTesting(conf, "tableAdmin", new String[0]);
272    SecureTestUtil.grantOnTable(TEST_UTIL, tableRead.getShortName(),
273      TEST_TABLE.getTableName(), null, null, Action.READ);
274    SecureTestUtil.grantOnTable(TEST_UTIL, tableWrite.getShortName(),
275      TEST_TABLE.getTableName(), null, null, Action.WRITE);
276    SecureTestUtil.grantOnTable(TEST_UTIL, tableCreate.getShortName(),
277      TEST_TABLE.getTableName(), null, null, Action.CREATE);
278    SecureTestUtil.grantOnTable(TEST_UTIL, tableAdmin.getShortName(),
279      TEST_TABLE.getTableName(), null, null, Action.ADMIN);
280
281    grantGlobal(TEST_UTIL, TESTGROUP_1_NAME, Action.WRITE);
282    try {
283      // Write tests
284
285      AccessTestAction writeAction = new AccessTestAction() {
286        @Override
287        public Object run() throws Exception {
288
289          try (Connection conn = ConnectionFactory.createConnection(conf);
290              Table t = conn.getTable(AccessControlLists.ACL_TABLE_NAME)) {
291            t.put(new Put(TEST_ROW).addColumn(AccessControlLists.ACL_LIST_FAMILY,
292                TEST_QUALIFIER, TEST_VALUE));
293            return null;
294          } finally {
295          }
296        }
297      };
298
299      // All writes to ACL table denied except for GLOBAL WRITE permission and superuser
300
301      verifyDenied(writeAction, globalAdmin, globalCreate, globalRead, TESTGROUP2_USER1);
302      verifyDenied(writeAction, nsAdmin, nsCreate, nsRead, nsWrite);
303      verifyDenied(writeAction, tableAdmin, tableCreate, tableRead, tableWrite);
304      verifyAllowed(writeAction, superUser, globalWrite, TESTGROUP1_USER1);
305    } finally {
306      revokeGlobal(TEST_UTIL, TESTGROUP_1_NAME, Action.WRITE);
307    }
308
309    grantGlobal(TEST_UTIL, TESTGROUP_1_NAME, Action.READ);
310    try {
311      // Read tests
312
313      AccessTestAction scanAction = new AccessTestAction() {
314        @Override
315        public Object run() throws Exception {
316          try (Connection conn = ConnectionFactory.createConnection(conf);
317              Table t = conn.getTable(AccessControlLists.ACL_TABLE_NAME)) {
318            ResultScanner s = t.getScanner(new Scan());
319            try {
320              for (Result r = s.next(); r != null; r = s.next()) {
321                // do nothing
322              }
323            } finally {
324              s.close();
325            }
326            return null;
327          }
328        }
329      };
330
331      // All reads from ACL table denied except for GLOBAL READ and superuser
332
333      verifyDenied(scanAction, globalAdmin, globalCreate, globalWrite, TESTGROUP2_USER1);
334      verifyDenied(scanAction, nsCreate, nsAdmin, nsRead, nsWrite);
335      verifyDenied(scanAction, tableCreate, tableAdmin, tableRead, tableWrite);
336      verifyAllowed(scanAction, superUser, globalRead, TESTGROUP1_USER1);
337    } finally {
338      revokeGlobal(TEST_UTIL, TESTGROUP_1_NAME, Action.READ);
339    }
340  }

495  @Test (timeout=180000)
496  public void testACLZNodeDeletion() throws Exception {
497    String baseAclZNode = "/hbase/acl/";
498    String ns = "testACLZNodeDeletionNamespace";
499    NamespaceDescriptor desc = NamespaceDescriptor.create(ns).build();
500    createNamespace(TEST_UTIL, desc);
501
502    final TableName table = TableName.valueOf(ns, "testACLZNodeDeletionTable");
503    final byte[] family = Bytes.toBytes("f1");
504    HTableDescriptor htd = new HTableDescriptor(table);
505    htd.addFamily(new HColumnDescriptor(family));
506    createTable(TEST_UTIL, htd);
507
508    // Namespace needs this, as they follow the lazy creation of ACL znode.
509    grantOnNamespace(TEST_UTIL, TESTGROUP1_USER1.getShortName(), ns, Action.ADMIN);
510    ZooKeeperWatcher zkw = TEST_UTIL.getMiniHBaseCluster().getMaster().getZooKeeper();
511    assertTrue("The acl znode for table should exist",  ZKUtil.checkExists(zkw, baseAclZNode +
512        table.getNameAsString()) != -1);
513    assertTrue("The acl znode for namespace should exist", ZKUtil.checkExists(zkw, baseAclZNode +
514        convertToNamespace(ns)) != -1);
515
516    revokeFromNamespace(TEST_UTIL, TESTGROUP1_USER1.getShortName(), ns, Action.ADMIN);
517    deleteTable(TEST_UTIL, table);
518    deleteNamespace(TEST_UTIL, ns);
519
520    assertTrue("The acl znode for table should have been deleted",
521        ZKUtil.checkExists(zkw, baseAclZNode + table.getNameAsString()) == -1);
522    assertTrue( "The acl znode for namespace should have been deleted",
523        ZKUtil.checkExists(zkw, baseAclZNode + convertToNamespace(ns)) == -1);
524  }
