084  @Test
085  public void testSaslClientUsesGivenRpcProtection() throws Exception {
086    Token<? extends TokenIdentifier> token = createTokenMockWithCredentials(DEFAULT_USER_NAME,
087        DEFAULT_USER_PASSWORD);
088    for (SaslUtil.QualityOfProtection qop : SaslUtil.QualityOfProtection.values()) {
089      String negotiatedQop = new HBaseSaslRpcClient(AuthMethod.DIGEST, token,
090          "principal/host@DOMAIN.COM", false, qop.name(), false) {
091        public String getQop() {
092          return saslProps.get(Sasl.QOP);
093        }
094      }.getQop();
095      assertEquals(negotiatedQop, qop.getSaslQop());
096    }
097  }

099  @Test
100  public void testSaslClientCallbackHandler() throws UnsupportedCallbackException {
101    final Token<? extends TokenIdentifier> token = createTokenMock();
102    when(token.getIdentifier()).thenReturn(Bytes.toBytes(DEFAULT_USER_NAME));
103    when(token.getPassword()).thenReturn(Bytes.toBytes(DEFAULT_USER_PASSWORD));

105    final NameCallback nameCallback = mock(NameCallback.class);
106    final PasswordCallback passwordCallback = mock(PasswordCallback.class);
107    final RealmCallback realmCallback = mock(RealmCallback.class);
108    final RealmChoiceCallback realmChoiceCallback = mock(RealmChoiceCallback.class);
109
110    Callback[] callbackArray = {nameCallback, passwordCallback, realmCallback, realmChoiceCallback};
111    final SaslClientCallbackHandler saslClCallbackHandler = new SaslClientCallbackHandler(token);
112    saslClCallbackHandler.handle(callbackArray);
113    verify(nameCallback).setName(anyString());
114    verify(realmCallback).setText(any());
115    verify(passwordCallback).setPassword(any());
116  }

133  @Test
134  public void testHBaseSaslRpcClientCreation() throws Exception {
135    //creation kerberos principal check section
136    assertFalse(assertSuccessCreationKerberosPrincipal(null));
137    assertFalse(assertSuccessCreationKerberosPrincipal("DOMAIN.COM"));
138    assertFalse(assertSuccessCreationKerberosPrincipal("principal/DOMAIN.COM"));
139    if (!assertSuccessCreationKerberosPrincipal("principal/localhost@DOMAIN.COM")) {
140      // XXX: This can fail if kerberos support in the OS is not sane, see HBASE-10107.
141      // For now, don't assert, just warn
142      LOG.warn("Could not create a SASL client with valid Kerberos credential");
143    }
144
145    //creation digest principal check section
146    assertFalse(assertSuccessCreationDigestPrincipal(null, null));
147    assertFalse(assertSuccessCreationDigestPrincipal("", ""));
148    assertFalse(assertSuccessCreationDigestPrincipal("", null));
149    assertFalse(assertSuccessCreationDigestPrincipal(null, ""));
150    assertTrue(assertSuccessCreationDigestPrincipal(DEFAULT_USER_NAME, DEFAULT_USER_PASSWORD));
151
152    //creation simple principal check section
153    assertFalse(assertSuccessCreationSimplePrincipal("", ""));
154    assertFalse(assertSuccessCreationSimplePrincipal(null, null));
155    assertFalse(assertSuccessCreationSimplePrincipal(DEFAULT_USER_NAME, DEFAULT_USER_PASSWORD));
156
157    //exceptions check section
158    assertTrue(assertIOExceptionThenSaslClientIsNull(DEFAULT_USER_NAME, DEFAULT_USER_PASSWORD));
159    assertTrue(assertIOExceptionWhenGetStreamsBeforeConnectCall(
160        DEFAULT_USER_NAME, DEFAULT_USER_PASSWORD));
161  }

163  @Test
164  public void testAuthMethodReadWrite() throws IOException {
165    DataInputBuffer in = new DataInputBuffer();
166    DataOutputBuffer out = new DataOutputBuffer();
167
168    assertAuthMethodRead(in, AuthMethod.SIMPLE);
169    assertAuthMethodRead(in, AuthMethod.KERBEROS);
170    assertAuthMethodRead(in, AuthMethod.DIGEST);
171
172    assertAuthMethodWrite(out, AuthMethod.SIMPLE);
173    assertAuthMethodWrite(out, AuthMethod.KERBEROS);
174    assertAuthMethodWrite(out, AuthMethod.DIGEST);
175  }
176
177  private void assertAuthMethodRead(DataInputBuffer in, AuthMethod authMethod)
178      throws IOException {
179    in.reset(new byte[] {authMethod.code}, 1);
180    assertEquals(authMethod, AuthMethod.read(in));
181  }
182
183  private void assertAuthMethodWrite(DataOutputBuffer out, AuthMethod authMethod)
184      throws IOException {
185    authMethod.write(out);
186    assertEquals(authMethod.code, out.getData()[0]);
187    out.reset();
188  }

190  private boolean assertIOExceptionWhenGetStreamsBeforeConnectCall(String principal,
191      String password) throws IOException {
192    boolean inState = false;
193    boolean outState = false;
194
195    HBaseSaslRpcClient rpcClient = new HBaseSaslRpcClient(AuthMethod.DIGEST,
196        createTokenMockWithCredentials(principal, password), principal, false) {
197      @Override
198      public SaslClient createDigestSaslClient(String[] mechanismNames,
199          String saslDefaultRealm, CallbackHandler saslClientCallbackHandler)
200              throws IOException {
201        return Mockito.mock(SaslClient.class);
202      }
203
204      @Override
205      public SaslClient createKerberosSaslClient(String[] mechanismNames,
206          String userFirstPart, String userSecondPart) throws IOException {
207        return Mockito.mock(SaslClient.class);
208      }
209    };
210
211    try {
212      rpcClient.getInputStream();
213    } catch(IOException ex) {
214      //Sasl authentication exchange hasn't completed yet
215      inState = true;
216    }
217
218    try {
219      rpcClient.getOutputStream();
220    } catch(IOException ex) {
221      //Sasl authentication exchange hasn't completed yet
222      outState = true;
223    }
224
225    return inState && outState;
226  }

287  private Token<? extends TokenIdentifier> createTokenMockWithCredentials(
288      String principal, String password)
289      throws IOException {
290    Token<? extends TokenIdentifier> token = createTokenMock();
291    if (!Strings.isNullOrEmpty(principal) && !Strings.isNullOrEmpty(password)) {
292      when(token.getIdentifier()).thenReturn(Bytes.toBytes(DEFAULT_USER_NAME));
293      when(token.getPassword()).thenReturn(Bytes.toBytes(DEFAULT_USER_PASSWORD));
294    }
295    return token;
296  }
