100  static LocalHBaseCluster createCluster(HBaseTestingUtil util, File keytabFile, MiniKdc kdc,
101    Map<String, char[]> userDatabase) throws Exception {
102    String servicePrincipal = "hbase/localhost";
103    String spnegoPrincipal = "HTTP/localhost";
104    kdc.createPrincipal(keytabFile, servicePrincipal);
105    util.startMiniZKCluster();
106
107    HBaseKerberosUtils.setSecuredConfiguration(util.getConfiguration(),
108      servicePrincipal + "@" + kdc.getRealm(), spnegoPrincipal + "@" + kdc.getRealm());
109    HBaseKerberosUtils.setSSLConfiguration(util, TestShadeSaslAuthenticationProvider.class);
110
111    util.getConfiguration().setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
112      TokenProvider.class.getName());
113    util.startMiniDFSCluster(1);
114    Path testDir = util.getDataTestDirOnTestFS("TestShadeSaslAuthenticationProvider");
115    USER_DATABASE_FILE = new Path(testDir, "user-db.txt");
116
117    createUserDBFile(USER_DATABASE_FILE.getFileSystem(CONF), USER_DATABASE_FILE, userDatabase);
118    CONF.set(ShadeSaslServerAuthenticationProvider.PASSWORD_FILE_KEY,
119      USER_DATABASE_FILE.toString());
120
121    Path rootdir = new Path(testDir, "hbase-root");
122    CommonFSUtils.setRootDir(CONF, rootdir);
123    LocalHBaseCluster cluster = new LocalHBaseCluster(CONF, 1);
124    return cluster;
125  }
126
127  static void createUserDBFile(FileSystem fs, Path p, Map<String, char[]> userDatabase)
128    throws IOException {
129    if (fs.exists(p)) {
130      fs.delete(p, true);
131    }
132    try (FSDataOutputStream out = fs.create(p); BufferedWriter writer =
133      new BufferedWriter(new OutputStreamWriter(out, StandardCharsets.UTF_8))) {
134      for (Entry<String, char[]> e : userDatabase.entrySet()) {
135        writer.write(e.getKey());
136        writer.write(ShadeSaslServerAuthenticationProvider.SEPARATOR);
137        writer.write(e.getValue());
138        writer.newLine();
139      }
140    }
141  }

181  @Before
182  public void createTable() throws Exception {
183    tableName = TableName.valueOf(name.getMethodName());
184
185    // Create a table and write a record as the service user (hbase)
186    UserGroupInformation serviceUgi = UserGroupInformation
187      .loginUserFromKeytabAndReturnUGI("hbase/localhost", KEYTAB_FILE.getAbsolutePath());
188    clusterId = serviceUgi.doAs(new PrivilegedExceptionAction<String>() {
189      @Override
190      public String run() throws Exception {
191        try (Connection conn = ConnectionFactory.createConnection(CONF);
192          Admin admin = conn.getAdmin();) {
193          admin.createTable(TableDescriptorBuilder.newBuilder(tableName)
194            .setColumnFamily(ColumnFamilyDescriptorBuilder.of("f1")).build());
195
196          UTIL.waitTableAvailable(tableName);
197
198          try (Table t = conn.getTable(tableName)) {
199            Put p = new Put(Bytes.toBytes("r1"));
200            p.addColumn(Bytes.toBytes("f1"), Bytes.toBytes("q1"), Bytes.toBytes("1"));
201            t.put(p);
202          }
203
204          return admin.getClusterMetrics().getClusterId();
205        }
206      }
207    });
208
209    assertNotNull(clusterId);
210  }

212  @Test
213  public void testPositiveAuthentication() throws Exception {
214    final Configuration clientConf = new Configuration(CONF);
215    try (Connection conn = ConnectionFactory.createConnection(clientConf)) {
216      UserGroupInformation user1 =
217        UserGroupInformation.createUserForTesting("user1", new String[0]);
218      user1.addToken(ShadeClientTokenUtil.obtainToken(conn, "user1", USER1_PASSWORD));
219      user1.doAs(new PrivilegedExceptionAction<Void>() {
220        @Override
221        public Void run() throws Exception {
222          try (Table t = conn.getTable(tableName)) {
223            Result r = t.get(new Get(Bytes.toBytes("r1")));
224            assertNotNull(r);
225            assertFalse("Should have read a non-empty Result", r.isEmpty());
226            final Cell cell = r.getColumnLatestCell(Bytes.toBytes("f1"), Bytes.toBytes("q1"));
227            assertTrue("Unexpected value", CellUtil.matchingValue(cell, Bytes.toBytes("1")));
228
229            return null;
230          }
231        }
232      });
233    }
234  }

236  @Test
237  public void testNegativeAuthentication() throws Exception {
238    List<Pair<String, Class<? extends Exception>>> params = new ArrayList<>();
239    // Master-based connection will fail to ask the master its cluster ID
240    // as a part of creating the Connection.
241    params.add(new Pair<String, Class<? extends Exception>>(MasterRegistry.class.getName(),
242      MasterRegistryFetchException.class));
243    // ZK based connection will fail on the master RPC
244    params.add(new Pair<String, Class<? extends Exception>>(
245      // ZKConnectionRegistry is package-private
246      HConstants.ZK_CONNECTION_REGISTRY_CLASS, RetriesExhaustedException.class));
247
248    params.forEach((pair) -> {
249      LOG.info("Running negative authentication test for client registry {}, expecting {}",
250        pair.getFirst(), pair.getSecond().getName());
251      // Validate that we can read that record back out as the user with our custom auth'n
252      final Configuration clientConf = new Configuration(CONF);
253      clientConf.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 3);
254      clientConf.set(HConstants.CLIENT_CONNECTION_REGISTRY_IMPL_CONF_KEY, pair.getFirst());
255      try (Connection conn = ConnectionFactory.createConnection(clientConf)) {
256        UserGroupInformation user1 =
257          UserGroupInformation.createUserForTesting("user1", new String[0]);
258        user1.addToken(
259          ShadeClientTokenUtil.obtainToken(conn, "user1", "not a real password".toCharArray()));
260
261        LOG.info("Executing request to HBase Master which should fail");
262        user1.doAs(new PrivilegedExceptionAction<Void>() {
263          @Override
264          public Void run() throws Exception {
265            try (Connection conn = ConnectionFactory.createConnection(clientConf);) {
266              conn.getAdmin().listTableDescriptors();
267              fail("Should not successfully authenticate with HBase");
268            } catch (Exception e) {
269              LOG.info("Caught exception in negative Master connectivity test", e);
270              assertEquals("Found unexpected exception", pair.getSecond(), e.getClass());
271              validateRootCause(Throwables.getRootCause(e));
272            }
273            return null;
274          }
275        });
276
277        LOG.info("Executing request to HBase RegionServer which should fail");
278        user1.doAs(new PrivilegedExceptionAction<Void>() {
279          @Override
280          public Void run() throws Exception {
281            // A little contrived because, with MasterRegistry, we'll still fail on talking
282            // to the HBase master before trying to talk to a RegionServer.
283            try (Connection conn = ConnectionFactory.createConnection(clientConf);
284              Table t = conn.getTable(tableName)) {
285              t.get(new Get(Bytes.toBytes("r1")));
286              fail("Should not successfully authenticate with HBase");
287            } catch (Exception e) {
288              LOG.info("Caught exception in negative RegionServer connectivity test", e);
289              assertEquals("Found unexpected exception", pair.getSecond(), e.getClass());
290              validateRootCause(Throwables.getRootCause(e));
291            }
292            return null;
293          }
294        });
295      } catch (InterruptedException e) {
296        LOG.error("Caught interrupted exception", e);
297        Thread.currentThread().interrupt();
298        return;
299      } catch (IOException e) {
300        throw new RuntimeException(e);
301      }
302    });
303  }
304
305  void validateRootCause(Throwable rootCause) {
306    LOG.info("Root cause was", rootCause);
307    if (rootCause instanceof RemoteException) {
308      RemoteException re = (RemoteException) rootCause;
309      IOException actualException = re.unwrapRemoteException();
310      assertEquals(InvalidToken.class, actualException.getClass());
311    } else {
312      StringWriter writer = new StringWriter();
313      rootCause.printStackTrace(new PrintWriter(writer));
314      String text = writer.toString();
315      assertTrue("Message did not contain expected text",
316        text.contains(InvalidToken.class.getName()));
317    }
318  }
