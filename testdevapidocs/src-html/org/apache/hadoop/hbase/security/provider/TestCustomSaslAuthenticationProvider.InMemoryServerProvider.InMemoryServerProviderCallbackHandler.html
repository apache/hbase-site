173   * used for delegation tokens.

175  public static class PasswordAuthTokenIdentifier extends TokenIdentifier {
176    public static final Text PASSWORD_AUTH_TOKEN = new Text("HBASE_PASSWORD_TEST_TOKEN");
177    private String username;
178
179    public PasswordAuthTokenIdentifier() {}
180
181    public PasswordAuthTokenIdentifier(String username) {
182      this.username = username;
183    }
184
185    @Override
186    public void readFields(DataInput in) throws IOException {
187      this.username = WritableUtils.readString(in);
188    }
189
190    @Override
191    public void write(DataOutput out) throws IOException {
192      WritableUtils.writeString(out, username);
193    }
194
195    @Override
196    public Text getKind() {
197      return PASSWORD_AUTH_TOKEN;
198    }
199
200    @Override
201    public UserGroupInformation getUser() {
202      if (username == null || "".equals(username)) {
203        return null;
204      }
205      return UserGroupInformation.createRemoteUser(username);
206    }
207  }

209  public static Token<? extends TokenIdentifier> createPasswordToken(
210      String username, String password, String clusterId) {
211    PasswordAuthTokenIdentifier id = new PasswordAuthTokenIdentifier(username);
212    Token<? extends TokenIdentifier> token = new Token<>(id.getBytes(), Bytes.toBytes(password),
213        id.getKind(), new Text(clusterId));
214    return token;

217  /**
218   * Client provider that finds custom Token in the user's UGI and authenticates with the server
219   * via DIGEST-MD5 using that password.
220   */
221  public static class InMemoryClientProvider extends AbstractSaslClientAuthenticationProvider {
222    public static final String MECHANISM = "DIGEST-MD5";
223    public static final SaslAuthMethod SASL_AUTH_METHOD = new SaslAuthMethod(
224        "IN_MEMORY", (byte)42, MECHANISM, AuthenticationMethod.TOKEN);
225
226    @Override
227    public SaslClient createClient(Configuration conf, InetAddress serverAddr,
228        SecurityInfo securityInfo, Token<? extends TokenIdentifier> token, boolean fallbackAllowed,
229        Map<String, String> saslProps) throws IOException {
230      return Sasl.createSaslClient(new String[] { MECHANISM }, null, null,
231          SaslUtil.SASL_DEFAULT_REALM, saslProps, new InMemoryClientProviderCallbackHandler(token));
232    }
233
234    public Optional<Token<? extends TokenIdentifier>> findToken(User user) {
235      List<Token<? extends TokenIdentifier>> tokens = user.getTokens().stream()
236        .filter((token) -> token.getKind().equals(PasswordAuthTokenIdentifier.PASSWORD_AUTH_TOKEN))
237        .collect(Collectors.toList());
238      if (tokens.isEmpty()) {
239        return Optional.empty();
240      }
241      if (tokens.size() > 1) {
242        throw new IllegalStateException("Cannot handle more than one PasswordAuthToken");
243      }
244      return Optional.of(tokens.get(0));
245    }
246
247    @Override
248    public SaslAuthMethod getSaslAuthMethod() {
249      return SASL_AUTH_METHOD;
250    }
251
252    /**
253     * Sasl CallbackHandler which extracts information from our custom token and places
254     * it into the Sasl objects.
255     */
256    public class InMemoryClientProviderCallbackHandler implements CallbackHandler {
257      private final Token<? extends TokenIdentifier> token;
258      public InMemoryClientProviderCallbackHandler(Token<? extends TokenIdentifier> token) {
259        this.token = token;
260      }
261
262      @Override
263      public void handle(Callback[] callbacks) throws UnsupportedCallbackException {
264        NameCallback nc = null;
265        PasswordCallback pc = null;
266        RealmCallback rc = null;
267        for (Callback callback : callbacks) {
268          if (callback instanceof RealmChoiceCallback) {
269            continue;
270          } else if (callback instanceof NameCallback) {
271            nc = (NameCallback) callback;
272          } else if (callback instanceof PasswordCallback) {
273            pc = (PasswordCallback) callback;
274          } else if (callback instanceof RealmCallback) {
275            rc = (RealmCallback) callback;
276          } else {
277            throw new UnsupportedCallbackException(callback, "Unrecognized SASL client callback");
278          }
279        }
280        if (nc != null) {
281          nc.setName(SaslUtil.encodeIdentifier(token.getIdentifier()));
282        }
283        if (pc != null) {
284          pc.setPassword(SaslUtil.encodePassword(token.getPassword()));
285        }
286        if (rc != null) {
287          rc.setText(rc.getDefaultText());
288        }
289      }
290    }
291
292    @Override
293    public UserInformation getUserInfo(User user) {
294      return null;
295    }
296  }

299   * Server provider which validates credentials from an in-memory database.
300   */
301  public static class InMemoryServerProvider extends InMemoryClientProvider
302      implements SaslServerAuthenticationProvider {
303
304    @Override
305    public AttemptingUserProvidingSaslServer createServer(
306        SecretManager<TokenIdentifier> secretManager,
307        Map<String, String> saslProps) throws IOException {
308      return new AttemptingUserProvidingSaslServer(
309          Sasl.createSaslServer(getSaslAuthMethod().getSaslMechanism(), null,
310              SaslUtil.SASL_DEFAULT_REALM, saslProps, new InMemoryServerProviderCallbackHandler()),
311        () -> null);
312    }
313
314    /**
315     * Pulls the correct password for the user who started the SASL handshake so that SASL
316     * can validate that the user provided the right password.
317     */
318    private class InMemoryServerProviderCallbackHandler implements CallbackHandler {
319
320      @Override
321      public void handle(Callback[] callbacks) throws InvalidToken, UnsupportedCallbackException {
322        NameCallback nc = null;
323        PasswordCallback pc = null;
324        AuthorizeCallback ac = null;
325        for (Callback callback : callbacks) {
326          if (callback instanceof AuthorizeCallback) {
327            ac = (AuthorizeCallback) callback;
328          } else if (callback instanceof NameCallback) {
329            nc = (NameCallback) callback;
330          } else if (callback instanceof PasswordCallback) {
331            pc = (PasswordCallback) callback;
332          } else if (callback instanceof RealmCallback) {
333            continue; // realm is ignored
334          } else {
335            throw new UnsupportedCallbackException(callback, "Unrecognized SASL Callback");
336          }
337        }
338        if (nc != null && pc != null) {
339          byte[] encodedId = SaslUtil.decodeIdentifier(nc.getDefaultName());
340          PasswordAuthTokenIdentifier id = new PasswordAuthTokenIdentifier();
341          try {
342            id.readFields(new DataInputStream(new ByteArrayInputStream(encodedId)));
343          } catch (IOException e) {
344            throw (InvalidToken) new InvalidToken(
345                "Can't de-serialize tokenIdentifier").initCause(e);
346          }
347          char[] actualPassword = SaslUtil.encodePassword(
348              Bytes.toBytes(getPassword(id.getUser().getUserName())));
349          pc.setPassword(actualPassword);
350        }
351        if (ac != null) {
352          String authid = ac.getAuthenticationID();
353          String authzid = ac.getAuthorizationID();
354          if (authid.equals(authzid)) {
355            ac.setAuthorized(true);
356          } else {
357            ac.setAuthorized(false);
358          }
359          if (ac.isAuthorized()) {
360            ac.setAuthorizedID(authzid);
361          }
362        }
363      }
364    }
365
366    @Override
367    public boolean supportsProtocolAuthentication() {
368      return false;
369    }
370
371    @Override
372    public UserGroupInformation getAuthorizedUgi(String authzId,
373        SecretManager<TokenIdentifier> secretManager) throws IOException {
374      UserGroupInformation authorizedUgi;
375      byte[] encodedId = SaslUtil.decodeIdentifier(authzId);
376      PasswordAuthTokenIdentifier tokenId = new PasswordAuthTokenIdentifier();
377      try {
378        tokenId.readFields(new DataInputStream(new ByteArrayInputStream(encodedId)));
379      } catch (IOException e) {
380        throw new IOException("Can't de-serialize PasswordAuthTokenIdentifier", e);
381      }
382      authorizedUgi = tokenId.getUser();
383      if (authorizedUgi == null) {
384        throw new AccessDeniedException(
385            "Can't retrieve username from tokenIdentifier.");
386      }
387      authorizedUgi.addTokenIdentifier(tokenId);
388      authorizedUgi.setAuthenticationMethod(getSaslAuthMethod().getAuthMethod());
389      return authorizedUgi;
390    }
391  }

393  /**
394   * Custom provider which can select our custom provider, amongst other tokens which
395   * may be available.
396   */
397  public static class InMemoryProviderSelector extends BuiltInProviderSelector {
398    private InMemoryClientProvider inMemoryProvider;
399
400    @Override
401    public void configure(Configuration conf,
402        Collection<SaslClientAuthenticationProvider> providers) {
403      super.configure(conf, providers);
404      Optional<SaslClientAuthenticationProvider> o = providers.stream()
405        .filter((p) -> p instanceof InMemoryClientProvider)
406        .findAny();
407
408      inMemoryProvider = (InMemoryClientProvider) o.orElseThrow(
409        () -> new RuntimeException("InMemoryClientProvider not found in available providers: "
410              + providers));
411    }
412
413    @Override
414    public Pair<SaslClientAuthenticationProvider, Token<? extends TokenIdentifier>> selectProvider(
415        String clusterId, User user) {
416      Pair<SaslClientAuthenticationProvider, Token<? extends TokenIdentifier>> superPair =
417          super.selectProvider(clusterId, user);
418
419      Optional<Token<? extends TokenIdentifier>> optional = inMemoryProvider.findToken(user);
420      if (optional.isPresent()) {
421        LOG.info("Using InMemoryClientProvider");
422        return new Pair<>(inMemoryProvider, optional.get());
423      }
424
425      LOG.info("InMemoryClientProvider not usable, falling back to {}", superPair);
426      return superPair;
427    }
428  }

445    FSUtils.setRootDir(util.getConfiguration(), rootdir);

508  public void createTable() throws Exception {
509    tableName = name.getTableName();
510
511    // Create a table and write a record as the service user (hbase)
512    UserGroupInformation serviceUgi = UserGroupInformation.loginUserFromKeytabAndReturnUGI(
513        "hbase/localhost", KEYTAB_FILE.getAbsolutePath());
514    clusterId = serviceUgi.doAs(new PrivilegedExceptionAction<String>() {

516        try (Connection conn = ConnectionFactory.createConnection(CONF);
517            Admin admin = conn.getAdmin();) {

519              .newBuilder(tableName)
520              .setColumnFamily(ColumnFamilyDescriptorBuilder.of("f1"))
521              .build());
522
523          UTIL.waitTableAvailable(tableName);
524
525          try (Table t = conn.getTable(tableName)) {
526            Put p = new Put(Bytes.toBytes("r1"));
527            p.addColumn(Bytes.toBytes("f1"), Bytes.toBytes("q1"), Bytes.toBytes("1"));
528            t.put(p);
529          }
530
531          return admin.getClusterMetrics().getClusterId();
532        }
533      }
534    });
535
536    assertNotNull(clusterId);
537  }

539  @Test
540  public void testPositiveAuthentication() throws Exception {
541    // Validate that we can read that record back out as the user with our custom auth'n
542    final Configuration clientConf = new Configuration(CONF);
543    UserGroupInformation user1 = UserGroupInformation.createUserForTesting(
544        "user1", new String[0]);
545    user1.addToken(createPasswordToken("user1", USER1_PASSWORD, clusterId));
546    user1.doAs(new PrivilegedExceptionAction<Void>() {
547      @Override public Void run() throws Exception {
548        try (Connection conn = ConnectionFactory.createConnection(clientConf);
549            Table t = conn.getTable(tableName)) {
550          Result r = t.get(new Get(Bytes.toBytes("r1")));
551          assertNotNull(r);
552          assertFalse("Should have read a non-empty Result", r.isEmpty());
553          final Cell cell = r.getColumnLatestCell(Bytes.toBytes("f1"), Bytes.toBytes("q1"));
554          assertTrue("Unexpected value", CellUtil.matchingValue(cell, Bytes.toBytes("1")));
555
556          return null;
557        }
558      }
559    });

562  @Test
563  public void testNegativeAuthentication() throws Exception {
564    // Validate that we can read that record back out as the user with our custom auth'n
565    final Configuration clientConf = new Configuration(CONF);
566    clientConf.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 3);
567    UserGroupInformation user1 = UserGroupInformation.createUserForTesting(
568        "user1", new String[0]);
569    user1.addToken(createPasswordToken("user1", "definitely not the password", clusterId));
570    user1.doAs(new PrivilegedExceptionAction<Void>() {
571      @Override public Void run() throws Exception {
572        // Depending on the registry in use, the following code can throw exceptions at different
573        // places. Master registry fails at the createConnection() step because the RPC to the
574        // master fails with sasl auth. With ZK registry, connection creation succeeds (since there
575        // is no RPC to HBase services involved) but the subsequent get() fails. The root cause
576        // should still be a SaslException in both the cases.
577        try (Connection conn = ConnectionFactory.createConnection(clientConf);
578            Table t = conn.getTable(tableName)) {
579          t.get(new Get(Bytes.toBytes("r1")));
580          fail("Should not successfully authenticate with HBase");
581        } catch (MasterRegistryFetchException mfe) {
582          Throwable cause = mfe.getCause();
583          assertTrue(cause.getMessage(), cause.getMessage().contains("SaslException"));
584        } catch (RetriesExhaustedException re) {
585          assertTrue(re.getMessage(), re.getMessage().contains("SaslException"));
586        } catch (Exception e) {
587          // Any other exception is unexpected.
588          fail("Unexpected exception caught, was expecting a authentication error: "
589              + Throwables.getStackTraceAsString(e));
590        }
591        return null;
592      }
593    });
