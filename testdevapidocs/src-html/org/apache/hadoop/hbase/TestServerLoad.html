021package org.apache.hadoop.hbase;
022
023import static org.junit.Assert.*;
024
025import org.apache.hadoop.hbase.shaded.protobuf.generated.ClusterStatusProtos;
026import org.apache.hadoop.hbase.shaded.protobuf.generated.HBaseProtos;
027import org.apache.hadoop.hbase.testclassification.MiscTests;
028import org.apache.hadoop.hbase.testclassification.SmallTests;
029import org.junit.Test;
030import org.junit.experimental.categories.Category;
031
032import org.apache.hadoop.hbase.shaded.com.google.protobuf.ByteString;
033
034@Category({MiscTests.class, SmallTests.class})
035public class TestServerLoad {
036
037  @Test
038  public void testRegionLoadAggregation() {
039    ServerLoad sl = new ServerLoad(createServerLoadProto());
040    assertEquals(13, sl.getStores());
041    assertEquals(114, sl.getStorefiles());
042    assertEquals(129, sl.getStoreUncompressedSizeMB());
043    assertEquals(504, sl.getRootIndexSizeKB());
044    assertEquals(820, sl.getStorefileSizeMB());
045    assertEquals(82, sl.getStorefileIndexSizeKB());
046    assertEquals(((long)Integer.MAX_VALUE)*2, sl.getReadRequestsCount());
047    assertEquals(300, sl.getFilteredReadRequestsCount());
048    
049  }
050 
051  @Test
052  public void testToString() {
053    ServerLoad sl = new ServerLoad(createServerLoadProto());
054    String slToString = sl.toString();
055    assertTrue(slToString.contains("numberOfStores=13"));
056    assertTrue(slToString.contains("numberOfStorefiles=114"));
057    assertTrue(slToString.contains("storefileUncompressedSizeMB=129"));
058    assertTrue(slToString.contains("storefileSizeMB=820")); 
059    assertTrue(slToString.contains("rootIndexSizeKB=504"));
060    assertTrue(slToString.contains("coprocessors=[]"));
061    assertTrue(slToString.contains("filteredReadRequestsCount=300"));
062  }
063
064  @Test
065  public void testRegionLoadWrapAroundAggregation() {
066    ServerLoad sl = new ServerLoad(createServerLoadProto());
067    long totalCount = ((long)Integer.MAX_VALUE)*2;
068    assertEquals(totalCount, sl.getReadRequestsCount());
069    assertEquals(totalCount, sl.getWriteRequestsCount());
070  }
071
072  private ClusterStatusProtos.ServerLoad createServerLoadProto() {
073    HBaseProtos.RegionSpecifier rSpecOne =
074        HBaseProtos.RegionSpecifier.newBuilder()
075            .setType(HBaseProtos.RegionSpecifier.RegionSpecifierType.ENCODED_REGION_NAME)
076            .setValue(ByteString.copyFromUtf8("ASDFGQWERT")).build();
077    HBaseProtos.RegionSpecifier rSpecTwo =
078        HBaseProtos.RegionSpecifier.newBuilder()
079            .setType(HBaseProtos.RegionSpecifier.RegionSpecifierType.ENCODED_REGION_NAME)
080            .setValue(ByteString.copyFromUtf8("QWERTYUIOP")).build();
081
082    ClusterStatusProtos.RegionLoad rlOne =
083        ClusterStatusProtos.RegionLoad.newBuilder().setRegionSpecifier(rSpecOne).setStores(10)
084            .setStorefiles(101).setStoreUncompressedSizeMB(106).setStorefileSizeMB(520)
085            .setFilteredReadRequestsCount(100)
086            .setStorefileIndexSizeKB(42).setRootIndexSizeKB(201).setReadRequestsCount(Integer.MAX_VALUE).setWriteRequestsCount(Integer.MAX_VALUE).build();
087    ClusterStatusProtos.RegionLoad rlTwo =
088        ClusterStatusProtos.RegionLoad.newBuilder().setRegionSpecifier(rSpecTwo).setStores(3)
089            .setStorefiles(13).setStoreUncompressedSizeMB(23).setStorefileSizeMB(300)
090            .setFilteredReadRequestsCount(200)
091            .setStorefileIndexSizeKB(40).setRootIndexSizeKB(303).setReadRequestsCount(Integer.MAX_VALUE).setWriteRequestsCount(Integer.MAX_VALUE).build();
092
093    ClusterStatusProtos.ServerLoad sl =
094        ClusterStatusProtos.ServerLoad.newBuilder().addRegionLoads(rlOne).
095          addRegionLoads(rlTwo).build();
096    return sl;
097  }
