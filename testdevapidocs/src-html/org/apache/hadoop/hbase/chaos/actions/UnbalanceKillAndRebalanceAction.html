019package org.apache.hadoop.hbase.chaos.actions;
020
021import java.util.ArrayList;
022import java.util.HashSet;
023import java.util.LinkedList;
024import java.util.List;
025import java.util.Set;
026
027import org.apache.commons.lang3.RandomUtils;
028import org.apache.hadoop.hbase.ClusterStatus;
029import org.apache.hadoop.hbase.ServerName;
030import org.junit.Assert;
031
032/** This action is too specific to put in ChaosMonkey; put it here */
033public class UnbalanceKillAndRebalanceAction extends Action {
034  /** Fractions of servers to get regions and live and die respectively; from all other
035   * servers, HOARD_FRC_OF_REGIONS will be removed to the above randomly */
036  private static final double FRC_SERVERS_THAT_HOARD_AND_LIVE = 0.1;
037  private static final double FRC_SERVERS_THAT_HOARD_AND_DIE = 0.1;
038  private static final double HOARD_FRC_OF_REGIONS = 0.8;
039  /** Waits between calling unbalance and killing servers, kills and rebalance, and rebalance
040   * and restarting the servers; to make sure these events have time to impact the cluster. */
041  private long waitForUnbalanceMilliSec;
042  private long waitForKillsMilliSec;
043  private long waitAfterBalanceMilliSec;
044  private boolean killMetaRs;
045
046  public UnbalanceKillAndRebalanceAction(long waitUnbalance, long waitKill, long waitAfterBalance,
047      boolean killMetaRs) {
048    super();
049    waitForUnbalanceMilliSec = waitUnbalance;
050    waitForKillsMilliSec = waitKill;
051    waitAfterBalanceMilliSec = waitAfterBalance;
052    this.killMetaRs = killMetaRs;
053  }
054
055  @Override
056  public void perform() throws Exception {
057    ClusterStatus status = this.cluster.getClusterStatus();
058    List<ServerName> victimServers = new LinkedList<>(status.getServers());
059    Set<ServerName> killedServers = new HashSet<>();
060
061    int liveCount = (int)Math.ceil(FRC_SERVERS_THAT_HOARD_AND_LIVE * victimServers.size());
062    int deadCount = (int)Math.ceil(FRC_SERVERS_THAT_HOARD_AND_DIE * victimServers.size());
063    Assert.assertTrue(
064        "There are not enough victim servers: " + victimServers.size(),
065        liveCount + deadCount < victimServers.size());
066    List<ServerName> targetServers = new ArrayList<>(liveCount);
067    for (int i = 0; i < liveCount + deadCount; ++i) {
068      int victimIx = RandomUtils.nextInt(0, victimServers.size());
069      targetServers.add(victimServers.remove(victimIx));
070    }
071    unbalanceRegions(status, victimServers, targetServers, HOARD_FRC_OF_REGIONS);
072    Thread.sleep(waitForUnbalanceMilliSec);
073    ServerName metaServer = cluster.getServerHoldingMeta();
074    for (ServerName targetServer: targetServers) {
075      // Don't keep killing servers if we're
076      // trying to stop the monkey.
077      if (context.isStopping()) {
078        break;
079      }
080      if (killedServers.size() >= liveCount) {
081        break;
082      }
083
084      if (!killMetaRs && targetServer.equals(metaServer)) {
085        LOG.info("Not killing server because it holds hbase:meta.");
086      } else {
087        killRs(targetServer);
088        killedServers.add(targetServer);
089      }
090    }
091
092    Thread.sleep(waitForKillsMilliSec);
093    forceBalancer();
094    Thread.sleep(waitAfterBalanceMilliSec);
095    for (ServerName server:killedServers) {
096      startRs(server);
097    }
098  }
099}
