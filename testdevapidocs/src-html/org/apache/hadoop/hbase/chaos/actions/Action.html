104  public void init(ActionContext context) throws IOException {
105    this.context = context;
106    cluster = context.getHBaseCluster();
107    initialStatus = cluster.getInitialClusterMetrics();
108    Collection<ServerName> regionServers = initialStatus.getLiveServerMetrics().keySet();
109    initialServers = regionServers.toArray(new ServerName[regionServers.size()]);
110
111    killMasterTimeout = cluster.getConf().getLong(KILL_MASTER_TIMEOUT_KEY,
112      KILL_MASTER_TIMEOUT_DEFAULT);
113    startMasterTimeout = cluster.getConf().getLong(START_MASTER_TIMEOUT_KEY,
114      START_MASTER_TIMEOUT_DEFAULT);
115    killRsTimeout = cluster.getConf().getLong(KILL_RS_TIMEOUT_KEY, KILL_RS_TIMEOUT_DEFAULT);
116    startRsTimeout = cluster.getConf().getLong(START_RS_TIMEOUT_KEY, START_RS_TIMEOUT_DEFAULT);
117    killZkNodeTimeout = cluster.getConf().getLong(KILL_ZK_NODE_TIMEOUT_KEY,
118      KILL_ZK_NODE_TIMEOUT_DEFAULT);
119    startZkNodeTimeout = cluster.getConf().getLong(START_ZK_NODE_TIMEOUT_KEY,
120      START_ZK_NODE_TIMEOUT_DEFAULT);
121    killDataNodeTimeout = cluster.getConf().getLong(KILL_DATANODE_TIMEOUT_KEY,

134  /** Returns current region servers - active master */
135  protected ServerName[] getCurrentServers() throws IOException {
136    ClusterMetrics clusterStatus = cluster.getClusterMetrics();
137    Collection<ServerName> regionServers = clusterStatus.getLiveServerMetrics().keySet();
138    int count = regionServers == null ? 0 : regionServers.size();
139    if (count <= 0) {
140      return new ServerName [] {};
141    }
142    ServerName master = clusterStatus.getMasterName();
143    Set<ServerName> masters = new HashSet<ServerName>();
144    masters.add(master);
145    masters.addAll(clusterStatus.getBackupMasterNames());
146    ArrayList<ServerName> tmp = new ArrayList<>(count);
147    tmp.addAll(regionServers);
148    tmp.removeAll(masters);
149    return tmp.toArray(new ServerName[tmp.size()]);
150  }

152  protected void killMaster(ServerName server) throws IOException {
153    LOG.info("Killing master " + server);
154    cluster.killMaster(server);
155    cluster.waitForMasterToStop(server, killMasterTimeout);
156    LOG.info("Killed master " + server);
157  }

162    cluster.waitForActiveAndReadyMaster(startMasterTimeout);
163    LOG.info("Started master " + server.getHostname());
164  }

226  protected void unbalanceRegions(ClusterMetrics clusterStatus,
227      List<ServerName> fromServers, List<ServerName> toServers,
228      double fractionOfRegions) throws Exception {
229    List<byte[]> victimRegions = new LinkedList<>();
230    for (Map.Entry<ServerName, ServerMetrics> entry
231      : clusterStatus.getLiveServerMetrics().entrySet()) {
232      ServerName sn = entry.getKey();
233      ServerMetrics serverLoad = entry.getValue();
234      // Ugh.
235      List<byte[]> regions = new LinkedList<>(serverLoad.getRegionMetrics().keySet());
236      int victimRegionCount = (int)Math.ceil(fractionOfRegions * regions.size());
237      LOG.debug("Removing " + victimRegionCount + " regions from " + sn);
238      for (int i = 0; i < victimRegionCount; ++i) {
239        int victimIx = RandomUtils.nextInt(0, regions.size());
240        String regionId = HRegionInfo.encodeRegionName(regions.remove(victimIx));
241        victimRegions.add(Bytes.toBytes(regionId));
242      }
243    }
244
245    LOG.info("Moving " + victimRegions.size() + " regions from " + fromServers.size()
246        + " servers to " + toServers.size() + " different servers");
247    Admin admin = this.context.getHBaseIntegrationTestingUtility().getAdmin();
248    for (byte[] victimRegion : victimRegions) {
249      // Don't keep moving regions if we're
250      // trying to stop the monkey.
251      if (context.isStopping()) {
252        break;
253      }
254      int targetIx = RandomUtils.nextInt(0, toServers.size());
255      admin.move(victimRegion, toServers.get(targetIx));
256    }
257  }
258
259  protected void forceBalancer() throws Exception {
260    Admin admin = this.context.getHBaseIntegrationTestingUtility().getAdmin();
261    boolean result = false;
262    try {
263      result = admin.balance();
264    } catch (Exception e) {
265      LOG.warn("Got exception while doing balance ", e);
266    }
267    if (!result) {
268      LOG.error("Balancer didn't succeed");
269    }
270  }

276  /**
277   * Apply a transform to all columns in a given table. If there are no columns in a table or if the context is stopping does nothing.
278   * @param tableName the table to modify
279   * @param transform the modification to perform. Callers will have the column name as a string and a column family builder available to them
280   */
281  protected void modifyAllTableColumns(TableName tableName, BiConsumer<String, ColumnFamilyDescriptorBuilder> transform) throws IOException {
282    HBaseTestingUtility util = this.context.getHBaseIntegrationTestingUtility();
283    Admin admin = util.getAdmin();
284
285    TableDescriptor tableDescriptor = admin.getDescriptor(tableName);
286    ColumnFamilyDescriptor[] columnDescriptors = tableDescriptor.getColumnFamilies();
287
288    if (columnDescriptors == null || columnDescriptors.length == 0) {
289      return;
290    }
291
292    TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(tableDescriptor);
293    for (ColumnFamilyDescriptor descriptor : columnDescriptors) {
294      ColumnFamilyDescriptorBuilder cfd = ColumnFamilyDescriptorBuilder.newBuilder(descriptor);
295      transform.accept(descriptor.getNameAsString(), cfd);
296      builder.modifyColumnFamily(cfd.build());
297    }
298
299    // Don't try the modify if we're stopping
300    if (this.context.isStopping()) {
301      return;
302    }
303    admin.modifyTable(builder.build());
304  }
305
306  /**
307   * Apply a transform to all columns in a given table. If there are no columns in a table or if the context is stopping does nothing.
308   * @param tableName the table to modify
309   * @param transform the modification to perform on each column family descriptor builder
310   */
311  protected void modifyAllTableColumns(TableName tableName, Consumer<ColumnFamilyDescriptorBuilder> transform) throws IOException {
312    modifyAllTableColumns(tableName, (name, cfd) -> transform.accept(cfd));
313  }
