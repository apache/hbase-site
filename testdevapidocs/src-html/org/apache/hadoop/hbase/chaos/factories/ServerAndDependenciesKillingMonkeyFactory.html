018package org.apache.hadoop.hbase.chaos.factories;
019
020import org.apache.hadoop.hbase.chaos.actions.Action;
021import org.apache.hadoop.hbase.chaos.actions.DumpClusterStatusAction;
022import org.apache.hadoop.hbase.chaos.actions.DumpHdfsClusterStatusAction;
023import org.apache.hadoop.hbase.chaos.actions.ForceBalancerAction;
024import org.apache.hadoop.hbase.chaos.actions.GracefulRollingRestartRsAction;
025import org.apache.hadoop.hbase.chaos.actions.RestartActiveMasterAction;
026import org.apache.hadoop.hbase.chaos.actions.RestartActiveNameNodeAction;
027import org.apache.hadoop.hbase.chaos.actions.RestartRandomDataNodeAction;
028import org.apache.hadoop.hbase.chaos.actions.RestartRandomRsExceptMetaAction;
029import org.apache.hadoop.hbase.chaos.actions.RestartRandomZKNodeAction;
030import org.apache.hadoop.hbase.chaos.actions.RollingBatchRestartRsAction;
031import org.apache.hadoop.hbase.chaos.actions.RollingBatchSuspendResumeRsAction;
032import org.apache.hadoop.hbase.chaos.monkies.ChaosMonkey;
033import org.apache.hadoop.hbase.chaos.monkies.PolicyBasedChaosMonkey;
034import org.apache.hadoop.hbase.chaos.policies.CompositeSequentialPolicy;
035import org.apache.hadoop.hbase.chaos.policies.DoActionsOncePolicy;
036import org.apache.hadoop.hbase.chaos.policies.PeriodicRandomActionPolicy;
037
038/**
039 * Creates ChaosMonkeys for doing server restart actions, but not flush / compact / snapshot kind of
040 * actions.
041 */
042public class ServerAndDependenciesKillingMonkeyFactory extends MonkeyFactory {
043
044  private long gracefulRollingRestartTSSLeepTime;
045  private long rollingBatchSuspendRSSleepTime;
046  private float rollingBatchSuspendtRSRatio;
047
048  @Override
049  public ChaosMonkey build() {
050    loadProperties();
051
052    // Destructive actions to mess things around. Cannot run batch restart.
053    // @formatter:off
054    Action[] actions1 = new Action[] {
055      new RestartRandomRsExceptMetaAction(60000),
056      new RestartActiveMasterAction(5000),
057      // only allow 2 servers to be dead.
058      new RollingBatchRestartRsAction(5000, 1.0f, 2, true),
059      new ForceBalancerAction(),
060      new RestartActiveNameNodeAction(60000),
061      new RestartRandomDataNodeAction(60000),
062      new RestartRandomZKNodeAction(60000),
063      new GracefulRollingRestartRsAction(gracefulRollingRestartTSSLeepTime),
064      new RollingBatchSuspendResumeRsAction(rollingBatchSuspendRSSleepTime,
065          rollingBatchSuspendtRSRatio)
066    };
067    // @formatter:on
068
069    // Action to log more info for debugging
070    Action[] actions2 =
071      new Action[] { new DumpClusterStatusAction(), new DumpHdfsClusterStatusAction() };
072
073    return new PolicyBasedChaosMonkey(properties, util,
074      new CompositeSequentialPolicy(new DoActionsOncePolicy(60 * 1000, actions1),
075        new PeriodicRandomActionPolicy(60 * 1000, actions1)),
076      new PeriodicRandomActionPolicy(60 * 1000, actions2));
077  }
078
079  private void loadProperties() {
080    gracefulRollingRestartTSSLeepTime =
081      Long.parseLong(this.properties.getProperty(MonkeyConstants.GRACEFUL_RESTART_RS_SLEEP_TIME,
082        MonkeyConstants.DEFAULT_GRACEFUL_RESTART_RS_SLEEP_TIME + ""));
083    rollingBatchSuspendRSSleepTime = Long
084      .parseLong(this.properties.getProperty(MonkeyConstants.ROLLING_BATCH_SUSPEND_RS_SLEEP_TIME,
085        MonkeyConstants.DEFAULT_ROLLING_BATCH_SUSPEND_RS_SLEEP_TIME + ""));
086    rollingBatchSuspendtRSRatio =
087      Float.parseFloat(this.properties.getProperty(MonkeyConstants.ROLLING_BATCH_SUSPEND_RS_RATIO,
088        MonkeyConstants.DEFAULT_ROLLING_BATCH_SUSPEND_RS_RATIO + ""));
089  }
090}
