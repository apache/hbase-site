117  private void runCommit(String... members) throws Exception {
118    // make sure we just have an empty list
119    if (members == null) {
120      members = new String[0];
121    }
122    List<String> expected = Arrays.asList(members);
123
124    // setup the constants
125    ZKWatcher coordZkw = newZooKeeperWatcher();
126    String opDescription = "coordination test - " + members.length + " cohort members";
127
128    // start running the controller
129    ZKProcedureCoordinator coordinatorComms = new ZKProcedureCoordinator(
130        coordZkw, opDescription, COORDINATOR_NODE_NAME);
131    ThreadPoolExecutor pool = ProcedureCoordinator.defaultPool(COORDINATOR_NODE_NAME, POOL_SIZE, KEEP_ALIVE);
132    ProcedureCoordinator coordinator = new ProcedureCoordinator(coordinatorComms, pool) {
133      @Override
134      public Procedure createProcedure(ForeignExceptionDispatcher fed, String procName, byte[] procArgs,
135          List<String> expectedMembers) {
136        return Mockito.spy(super.createProcedure(fed, procName, procArgs, expectedMembers));
137      }
138    };
139
140    // build and start members
141    // NOTE: There is a single subprocedure builder for all members here.
142    SubprocedureFactory subprocFactory = Mockito.mock(SubprocedureFactory.class);
143    List<Pair<ProcedureMember, ZKProcedureMemberRpcs>> procMembers = new ArrayList<>(members.length);
144    // start each member
145    for (String member : members) {
146      ZKWatcher watcher = newZooKeeperWatcher();
147      ZKProcedureMemberRpcs comms = new ZKProcedureMemberRpcs(watcher, opDescription);
148      ThreadPoolExecutor pool2 = ProcedureMember.defaultPool(member, 1, KEEP_ALIVE);
149      ProcedureMember procMember = new ProcedureMember(comms, pool2, subprocFactory);
150      procMembers.add(new Pair<>(procMember, comms));
151      comms.start(member, procMember);
152    }
153
154    // setup mock member subprocedures
155    final List<Subprocedure> subprocs = new ArrayList<>();
156    for (int i = 0; i < procMembers.size(); i++) {
157      ForeignExceptionDispatcher cohortMonitor = new ForeignExceptionDispatcher();
158      Subprocedure commit = Mockito
159      .spy(new SubprocedureImpl(procMembers.get(i).getFirst(), opName, cohortMonitor,
160          WAKE_FREQUENCY, TIMEOUT));
161      subprocs.add(commit);
162    }
163
164    // link subprocedure to buildNewOperation invocation.
165    final AtomicInteger i = new AtomicInteger(0); // NOTE: would be racy if not an AtomicInteger
166    Mockito.when(subprocFactory.buildSubprocedure(Mockito.eq(opName),
167        (byte[]) Mockito.argThat(new ArrayEquals(data)))).thenAnswer(
168      new Answer<Subprocedure>() {
169        @Override
170        public Subprocedure answer(InvocationOnMock invocation) throws Throwable {
171          int index = i.getAndIncrement();
172          LOG.debug("Task size:" + subprocs.size() + ", getting:" + index);
173          Subprocedure commit = subprocs.get(index);
174          return commit;
175        }
176      });
177
178    // setup spying on the coordinator
179//    Procedure proc = Mockito.spy(procBuilder.createProcedure(coordinator, opName, data, expected));
180//    Mockito.when(procBuilder.build(coordinator, opName, data, expected)).thenReturn(proc);
181
182    // start running the operation
183    Procedure task = coordinator.startProcedure(new ForeignExceptionDispatcher(), opName, data, expected);
184//    assertEquals("Didn't mock coordinator task", proc, task);
185
186    // verify all things ran as expected
187//    waitAndVerifyProc(proc, once, once, never(), once, false);
188    waitAndVerifyProc(task, once, once, never(), once, false);
189    verifyCohortSuccessful(expected, subprocFactory, subprocs, once, once, never(), once, false);
190
191    // close all the things
192    closeAll(coordinator, coordinatorComms, procMembers);
193  }

195  /**
196   * Test a distributed commit with multiple cohort members, where one of the cohort members has a
197   * timeout exception during the prepare stage.
198   */
199  @Test
200  public void testMultiCohortWithMemberTimeoutDuringPrepare() throws Exception {
201    String opDescription = "error injection coordination";
202    String[] cohortMembers = new String[] { "one", "two", "three" };
203    List<String> expected = Lists.newArrayList(cohortMembers);
204    // error constants
205    final int memberErrorIndex = 2;
206    final CountDownLatch coordinatorReceivedErrorLatch = new CountDownLatch(1);
207
208    // start running the coordinator and its controller
209    ZKWatcher coordinatorWatcher = newZooKeeperWatcher();
210    ZKProcedureCoordinator coordinatorController = new ZKProcedureCoordinator(
211        coordinatorWatcher, opDescription, COORDINATOR_NODE_NAME);
212    ThreadPoolExecutor pool = ProcedureCoordinator.defaultPool(COORDINATOR_NODE_NAME, POOL_SIZE, KEEP_ALIVE);
213    ProcedureCoordinator coordinator = spy(new ProcedureCoordinator(coordinatorController, pool));
214
215    // start a member for each node
216    SubprocedureFactory subprocFactory = Mockito.mock(SubprocedureFactory.class);
217    List<Pair<ProcedureMember, ZKProcedureMemberRpcs>> members = new ArrayList<>(expected.size());
218    for (String member : expected) {
219      ZKWatcher watcher = newZooKeeperWatcher();
220      ZKProcedureMemberRpcs controller = new ZKProcedureMemberRpcs(watcher, opDescription);
221      ThreadPoolExecutor pool2 = ProcedureMember.defaultPool(member, 1, KEEP_ALIVE);
222      ProcedureMember mem = new ProcedureMember(controller, pool2, subprocFactory);
223      members.add(new Pair<>(mem, controller));
224      controller.start(member, mem);
225    }
226
227    // setup mock subprocedures
228    final List<Subprocedure> cohortTasks = new ArrayList<>();
229    final int[] elem = new int[1];
230    for (int i = 0; i < members.size(); i++) {
231      ForeignExceptionDispatcher cohortMonitor = new ForeignExceptionDispatcher();
232      final ProcedureMember comms = members.get(i).getFirst();
233      Subprocedure commit = Mockito
234      .spy(new SubprocedureImpl(comms, opName, cohortMonitor, WAKE_FREQUENCY, TIMEOUT));
235      // This nasty bit has one of the impls throw a TimeoutException
236      Mockito.doAnswer(new Answer<Void>() {
237        @Override
238        public Void answer(InvocationOnMock invocation) throws Throwable {
239          int index = elem[0];
240          if (index == memberErrorIndex) {
241            LOG.debug("Sending error to coordinator");
242            ForeignException remoteCause = new ForeignException("TIMER",
243                new TimeoutException("subprocTimeout" , 1, 2, 0));
244            Subprocedure r = ((Subprocedure) invocation.getMock());
245            LOG.error("Remote commit failure, not propagating error:" + remoteCause);
246            comms.receiveAbortProcedure(r.getName(), remoteCause);
247            assertEquals(r.isComplete(), true);
248            // don't complete the error phase until the coordinator has gotten the error
249            // notification (which ensures that we never progress past prepare)
250            try {
251              Procedure.waitForLatch(coordinatorReceivedErrorLatch, new ForeignExceptionDispatcher(),
252                  WAKE_FREQUENCY, "coordinator received error");
253            } catch (InterruptedException e) {
254              LOG.debug("Wait for latch interrupted, done:" + (coordinatorReceivedErrorLatch.getCount() == 0));
255              // reset the interrupt status on the thread
256              Thread.currentThread().interrupt();
257            }
258          }
259          elem[0] = ++index;
260          return null;
261        }
262      }).when(commit).acquireBarrier();
263      cohortTasks.add(commit);
264    }
265
266    // pass out a task per member
267    final AtomicInteger taskIndex = new AtomicInteger();
268    Mockito.when(
269      subprocFactory.buildSubprocedure(Mockito.eq(opName),
270        (byte[]) Mockito.argThat(new ArrayEquals(data)))).thenAnswer(
271      new Answer<Subprocedure>() {
272        @Override
273        public Subprocedure answer(InvocationOnMock invocation) throws Throwable {
274          int index = taskIndex.getAndIncrement();
275          Subprocedure commit = cohortTasks.get(index);
276          return commit;
277        }
278      });
279
280    // setup spying on the coordinator
281    ForeignExceptionDispatcher coordinatorTaskErrorMonitor = Mockito
282        .spy(new ForeignExceptionDispatcher());
283    Procedure coordinatorTask = Mockito.spy(new Procedure(coordinator,
284        coordinatorTaskErrorMonitor, WAKE_FREQUENCY, TIMEOUT,
285        opName, data, expected));
286    when(coordinator.createProcedure(any(), eq(opName), eq(data), anyListOf(String.class)))
287      .thenReturn(coordinatorTask);
288    // count down the error latch when we get the remote error
289    Mockito.doAnswer(new Answer<Void>() {
290      @Override
291      public Void answer(InvocationOnMock invocation) throws Throwable {
292        // pass on the error to the master
293        invocation.callRealMethod();
294        // then count down the got error latch
295        coordinatorReceivedErrorLatch.countDown();
296        return null;
297      }
298    }).when(coordinatorTask).receive(Mockito.any());
299
300    // ----------------------------
301    // start running the operation
302    // ----------------------------
303
304    Procedure task = coordinator.startProcedure(coordinatorTaskErrorMonitor, opName, data, expected);
305    assertEquals("Didn't mock coordinator task", coordinatorTask, task);
306
307    // wait for the task to complete
308    try {
309      task.waitForCompleted();
310    } catch (ForeignException fe) {
311      // this may get caught or may not
312    }
313
314    // -------------
315    // verification
316    // -------------
317
318    // always expect prepared, never committed, and possible to have cleanup and finish (racy since
319    // error case)
320    waitAndVerifyProc(coordinatorTask, once, never(), once, atMost(1), true);
321    verifyCohortSuccessful(expected, subprocFactory, cohortTasks, once, never(), once,
322      once, true);
323
324    // close all the open things
325    closeAll(coordinator, coordinatorController, members);
326  }

328  /**
329   * Wait for the coordinator task to complete, and verify all the mocks
330   * @param task to wait on
331   * @throws Exception on unexpected failure
332   */
333  private void waitAndVerifyProc(Procedure proc, VerificationMode prepare,
334      VerificationMode commit, VerificationMode cleanup, VerificationMode finish, boolean opHasError)
335      throws Exception {
336    boolean caughtError = false;
337    try {
338      proc.waitForCompleted();
339    } catch (ForeignException fe) {
340      caughtError = true;
341    }
342    // make sure that the task called all the expected phases
343    Mockito.verify(proc, prepare).sendGlobalBarrierStart();
344    Mockito.verify(proc, commit).sendGlobalBarrierReached();
345    Mockito.verify(proc, finish).sendGlobalBarrierComplete();
346    assertEquals("Operation error state was unexpected", opHasError, proc.getErrorMonitor()
347        .hasException());
348    assertEquals("Operation error state was unexpected", opHasError, caughtError);
349
350  }
351
352  /**
353   * Wait for the coordinator task to complete, and verify all the mocks
354   * @param task to wait on
355   * @throws Exception on unexpected failure
356   */
357  private void waitAndVerifySubproc(Subprocedure op, VerificationMode prepare,
358      VerificationMode commit, VerificationMode cleanup, VerificationMode finish, boolean opHasError)
359      throws Exception {
360    boolean caughtError = false;
361    try {
362      op.waitForLocallyCompleted();
363    } catch (ForeignException fe) {
364      caughtError = true;
365    }
366    // make sure that the task called all the expected phases
367    Mockito.verify(op, prepare).acquireBarrier();
368    Mockito.verify(op, commit).insideBarrier();
369    // We cannot guarantee that cleanup has run so we don't check it.
370
371    assertEquals("Operation error state was unexpected", opHasError, op.getErrorCheckable()
372        .hasException());
373    assertEquals("Operation error state was unexpected", opHasError, caughtError);
374
375  }
376
377  private void verifyCohortSuccessful(List<String> cohortNames,
378      SubprocedureFactory subprocFactory, Iterable<Subprocedure> cohortTasks,
379      VerificationMode prepare, VerificationMode commit, VerificationMode cleanup,
380      VerificationMode finish, boolean opHasError) throws Exception {
381
382    // make sure we build the correct number of cohort members
383    Mockito.verify(subprocFactory, Mockito.times(cohortNames.size())).buildSubprocedure(
384      Mockito.eq(opName), (byte[]) Mockito.argThat(new ArrayEquals(data)));
385    // verify that we ran each of the operations cleanly
386    int j = 0;
387    for (Subprocedure op : cohortTasks) {
388      LOG.debug("Checking mock:" + (j++));
389      waitAndVerifySubproc(op, prepare, commit, cleanup, finish, opHasError);
390    }
391  }

393  private void closeAll(
394      ProcedureCoordinator coordinator,
395      ZKProcedureCoordinator coordinatorController,
396      List<Pair<ProcedureMember, ZKProcedureMemberRpcs>> cohort)
397      throws IOException {
398    // make sure we close all the resources
399    for (Pair<ProcedureMember, ZKProcedureMemberRpcs> member : cohort) {
400      member.getFirst().close();
401      member.getSecond().close();
402    }
403    coordinator.close();
404    coordinatorController.close();
405  }
