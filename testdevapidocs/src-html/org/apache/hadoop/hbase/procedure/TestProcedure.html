040/**
041 * Demonstrate how Procedure handles single members, multiple members, and errors semantics
042 */
043@Category({MasterTests.class, SmallTests.class})
044public class TestProcedure {

046  ProcedureCoordinator coord;
047
048  @Before
049  public void setup() {
050    coord = mock(ProcedureCoordinator.class);
051    final ProcedureCoordinatorRpcs comms = mock(ProcedureCoordinatorRpcs.class);
052    when(coord.getRpcs()).thenReturn(comms); // make it not null
053  }

055  class LatchedProcedure extends Procedure {
056    CountDownLatch startedAcquireBarrier = new CountDownLatch(1);
057    CountDownLatch startedDuringBarrier = new CountDownLatch(1);
058    CountDownLatch completedProcedure = new CountDownLatch(1);
059
060    public LatchedProcedure(ProcedureCoordinator coord, ForeignExceptionDispatcher monitor,
061        long wakeFreq, long timeout, String opName, byte[] data,
062        List<String> expectedMembers) {
063      super(coord, monitor, wakeFreq, timeout, opName, data, expectedMembers);
064    }
065
066    @Override
067    public void sendGlobalBarrierStart() {
068      startedAcquireBarrier.countDown();
069    }
070
071    @Override
072    public void sendGlobalBarrierReached() {
073      startedDuringBarrier.countDown();
074    }
075
076    @Override
077    public void sendGlobalBarrierComplete() {
078      completedProcedure.countDown();
079    }
080  };
081
082  /**
083   * With a single member, verify ordered execution.  The Coordinator side is run in a separate
084   * thread so we can only trigger from members and wait for particular state latches.
085   */
086  @Test(timeout = 60000)
087  public void testSingleMember() throws Exception {
088    // The member
089    List<String> members =  new ArrayList<>();
090    members.add("member");
091    LatchedProcedure proc = new LatchedProcedure(coord, new ForeignExceptionDispatcher(), 100,
092        Integer.MAX_VALUE, "op", null, members);
093    final LatchedProcedure procspy = spy(proc);
094    // coordinator: start the barrier procedure
095    new Thread() {
096      public void run() {
097        procspy.call();
098      }
099    }.start();
100
101    // coordinator: wait for the barrier to be acquired, then send start barrier
102    proc.startedAcquireBarrier.await();
103
104    // we only know that {@link Procedure#sendStartBarrier()} was called, and others are blocked.
105    verify(procspy).sendGlobalBarrierStart();
106    verify(procspy, never()).sendGlobalBarrierReached();
107    verify(procspy, never()).sendGlobalBarrierComplete();
108    verify(procspy, never()).barrierAcquiredByMember(anyString());
109
110    // member: trigger global barrier acquisition
111    proc.barrierAcquiredByMember(members.get(0));
112
113    // coordinator: wait for global barrier to be acquired.
114    proc.acquiredBarrierLatch.await();
115    verify(procspy).sendGlobalBarrierStart(); // old news
116
117    // since two threads, we cannot guarantee that {@link Procedure#sendSatsifiedBarrier()} was
118    // or was not called here.
119
120    // member: trigger global barrier release
121    proc.barrierReleasedByMember(members.get(0), new byte[0]);
122
123    // coordinator: wait for procedure to be completed
124    proc.completedProcedure.await();
125    verify(procspy).sendGlobalBarrierReached();
126    verify(procspy).sendGlobalBarrierComplete();
127    verify(procspy, never()).receive(any(ForeignException.class));
128  }

130  @Test(timeout = 60000)
131  public void testMultipleMember() throws Exception {
132    // 2 members
133    List<String> members =  new ArrayList<>();
134    members.add("member1");
135    members.add("member2");
136
137    LatchedProcedure proc = new LatchedProcedure(coord, new ForeignExceptionDispatcher(), 100,
138        Integer.MAX_VALUE, "op", null, members);
139    final LatchedProcedure procspy = spy(proc);
140    // start the barrier procedure
141    new Thread() {
142      public void run() {
143        procspy.call();
144      }
145    }.start();
146
147    // coordinator: wait for the barrier to be acquired, then send start barrier
148    procspy.startedAcquireBarrier.await();
149
150    // we only know that {@link Procedure#sendStartBarrier()} was called, and others are blocked.
151    verify(procspy).sendGlobalBarrierStart();
152    verify(procspy, never()).sendGlobalBarrierReached();
153    verify(procspy, never()).sendGlobalBarrierComplete();
154    verify(procspy, never()).barrierAcquiredByMember(anyString()); // no externals
155
156    // member0: [1/2] trigger global barrier acquisition.
157    procspy.barrierAcquiredByMember(members.get(0));
158
159    // coordinator not satisified.
160    verify(procspy).sendGlobalBarrierStart();
161    verify(procspy, never()).sendGlobalBarrierReached();
162    verify(procspy, never()).sendGlobalBarrierComplete();
163
164    // member 1: [2/2] trigger global barrier acquisition.
165    procspy.barrierAcquiredByMember(members.get(1));
166
167    // coordinator: wait for global barrier to be acquired.
168    procspy.startedDuringBarrier.await();
169    verify(procspy).sendGlobalBarrierStart(); // old news
170
171    // member 1, 2: trigger global barrier release
172    procspy.barrierReleasedByMember(members.get(0), new byte[0]);
173    procspy.barrierReleasedByMember(members.get(1), new byte[0]);
174
175    // coordinator wait for procedure to be completed
176    procspy.completedProcedure.await();
177    verify(procspy).sendGlobalBarrierReached();
178    verify(procspy).sendGlobalBarrierComplete();
179    verify(procspy, never()).receive(any(ForeignException.class));
180  }

182  @Test(timeout = 60000)
183  public void testErrorPropagation() throws Exception {
184    List<String> members =  new ArrayList<>();
185    members.add("member");
186    Procedure proc = new Procedure(coord, new ForeignExceptionDispatcher(), 100,
187        Integer.MAX_VALUE, "op", null, members);
188    final Procedure procspy = spy(proc);
189
190    ForeignException cause = new ForeignException("SRC", "External Exception");
191    proc.receive(cause);
192
193    // start the barrier procedure
194    Thread t = new Thread() {
195      public void run() {
196        procspy.call();
197      }
198    };
199    t.start();
200    t.join();
201
202    verify(procspy, never()).sendGlobalBarrierStart();
203    verify(procspy, never()).sendGlobalBarrierReached();
204    verify(procspy).sendGlobalBarrierComplete();
205  }

207  @Test(timeout = 60000)
208  public void testBarrieredErrorPropagation() throws Exception {
209    List<String> members =  new ArrayList<>();
210    members.add("member");
211    LatchedProcedure proc = new LatchedProcedure(coord, new ForeignExceptionDispatcher(), 100,
212        Integer.MAX_VALUE, "op", null, members);
213    final LatchedProcedure procspy = spy(proc);
214
215    // start the barrier procedure
216    Thread t = new Thread() {
217      public void run() {
218        procspy.call();
219      }
220    };
221    t.start();
222
223    // now test that we can put an error in before the commit phase runs
224    procspy.startedAcquireBarrier.await();
225    ForeignException cause = new ForeignException("SRC", "External Exception");
226    procspy.receive(cause);
227    procspy.barrierAcquiredByMember(members.get(0));
228    t.join();
229
230    // verify state of all the object
231    verify(procspy).sendGlobalBarrierStart();
232    verify(procspy).sendGlobalBarrierComplete();
233    verify(procspy, never()).sendGlobalBarrierReached();
234  }
