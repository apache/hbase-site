056/**
057 * A test tool that uses multiple threads to read and write multifamily rows into a table, verifying
058 * that reads never see partially-complete writes
059 */
060@InterfaceAudience.Private
061public class AcidGuaranteesTestTool extends AbstractHBaseTool {
062
063  private static final Logger LOG = LoggerFactory.getLogger(AcidGuaranteesTestTool.class);
064
065  public static final TableName TABLE_NAME = TableName.valueOf("TestAcidGuarantees");
066  public static final byte[] FAMILY_A = Bytes.toBytes("A");
067  public static final byte[] FAMILY_B = Bytes.toBytes("B");
068  public static final byte[] FAMILY_C = Bytes.toBytes("C");
069  public static final byte[] QUALIFIER_NAME = Bytes.toBytes("data");
070
071  public static final byte[][] FAMILIES = new byte[][] { FAMILY_A, FAMILY_B, FAMILY_C };
072
073  public static int NUM_COLS_TO_CHECK = 50;
074
075  private ExecutorService sharedPool;
076
077  private long millisToRun;
078  private int numWriters;
079  private int numGetters;
080  private int numScanners;
081  private int numUniqueRows;
082  private boolean crazyFlush;
083  private boolean useMob;
084
085  private ExecutorService createThreadPool() {
086    int maxThreads = 256;
087    int coreThreads = 128;
088
089    long keepAliveTime = 60;
090    BlockingQueue<Runnable> workQueue = new LinkedBlockingQueue<Runnable>(
091        maxThreads * HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS);
092
093    ThreadPoolExecutor tpe = new ThreadPoolExecutor(coreThreads, maxThreads, keepAliveTime,
094        TimeUnit.SECONDS, workQueue, Threads.newDaemonThreadFactory(toString() + "-shared"));
095    tpe.allowCoreThreadTimeOut(true);
096    return tpe;
097  }
098
099  @Override
100  protected void addOptions() {
101    addOptWithArg("millis", "time limit in milliseconds");
102    addOptWithArg("numWriters", "number of write threads");
103    addOptWithArg("numGetters", "number of get threads");
104    addOptWithArg("numScanners", "number of scan threads");
105    addOptWithArg("numUniqueRows", "number of unique rows to test");
106    addOptNoArg("crazyFlush",
107      "if specified we will flush continuously otherwise will flush every minute");
108    addOptNoArg("useMob", "if specified we will enable mob on the first column family");
109  }
110
111  @Override
112  protected void processOptions(CommandLine cmd) {
113    millisToRun = getOptionAsLong(cmd, "millis", 5000);
114    numWriters = getOptionAsInt(cmd, "numWriters", 50);
115    numGetters = getOptionAsInt(cmd, "numGetters", 2);
116    numScanners = getOptionAsInt(cmd, "numScanners", 2);
117    numUniqueRows = getOptionAsInt(cmd, "numUniqueRows", 3);
118    crazyFlush = cmd.hasOption("crazyFlush");
119    useMob = cmd.hasOption("useMob");
120  }
121
122  @Override
123  protected int doWork() throws Exception {
124    sharedPool = createThreadPool();
125    try (Connection conn = ConnectionFactory.createConnection(getConf())) {
126      runTestAtomicity(conn.getAdmin());
127    } finally {
128      sharedPool.shutdown();
129    }
130    return 0;
131  }

133  /**
134   * Thread that does random full-row writes into a table.
135   */
136  public static class AtomicityWriter extends RepeatingTestThread {
137    Random rand = new Random();
138    byte data[] = new byte[10];
139    byte[][] targetRows;
140    byte[][] targetFamilies;
141    Connection connection;
142    Table table;
143    AtomicLong numWritten = new AtomicLong();
144
145    public AtomicityWriter(TestContext ctx, byte[][] targetRows, byte[][] targetFamilies,
146        ExecutorService pool) throws IOException {
147      super(ctx);
148      this.targetRows = targetRows;
149      this.targetFamilies = targetFamilies;
150      connection = ConnectionFactory.createConnection(ctx.getConf(), pool);
151      table = connection.getTable(TABLE_NAME);
152    }
153
154    public void doAnAction() throws Exception {
155      // Pick a random row to write into
156      byte[] targetRow = targetRows[rand.nextInt(targetRows.length)];
157      Put p = new Put(targetRow);
158      rand.nextBytes(data);
159
160      for (byte[] family : targetFamilies) {
161        for (int i = 0; i < NUM_COLS_TO_CHECK; i++) {
162          byte qualifier[] = Bytes.toBytes("col" + i);
163          p.addColumn(family, qualifier, data);
164        }
165      }
166      table.put(p);
167      numWritten.getAndIncrement();
168    }
169
170    @Override
171    public void workDone() throws IOException {
172      try {
173        table.close();
174      } finally {
175        connection.close();
176      }
177    }
178  }
179
180  /**
181   * Thread that does single-row reads in a table, looking for partially completed rows.
182   */
183  public static class AtomicGetReader extends RepeatingTestThread {
184    byte[] targetRow;
185    byte[][] targetFamilies;
186    Connection connection;
187    Table table;
188    int numVerified = 0;
189    AtomicLong numRead = new AtomicLong();
190
191    public AtomicGetReader(TestContext ctx, byte[] targetRow, byte[][] targetFamilies,
192        ExecutorService pool) throws IOException {
193      super(ctx);
194      this.targetRow = targetRow;
195      this.targetFamilies = targetFamilies;
196      connection = ConnectionFactory.createConnection(ctx.getConf(), pool);
197      table = connection.getTable(TABLE_NAME);
198    }
199
200    public void doAnAction() throws Exception {
201      Get g = new Get(targetRow);
202      Result res = table.get(g);
203      byte[] gotValue = null;
204      if (res.getRow() == null) {
205        // Trying to verify but we didn't find the row - the writing
206        // thread probably just hasn't started writing yet, so we can
207        // ignore this action
208        return;
209      }
210
211      for (byte[] family : targetFamilies) {
212        for (int i = 0; i < NUM_COLS_TO_CHECK; i++) {
213          byte qualifier[] = Bytes.toBytes("col" + i);
214          byte thisValue[] = res.getValue(family, qualifier);
215          if (gotValue != null && !Bytes.equals(gotValue, thisValue)) {
216            gotFailure(gotValue, res);
217          }
218          numVerified++;
219          gotValue = thisValue;
220        }
221      }
222      numRead.getAndIncrement();
223    }
224
225    @Override
226    public void workDone() throws IOException {
227      try {
228        table.close();
229      } finally {
230        connection.close();
231      }
232    }
233
234    private void gotFailure(byte[] expected, Result res) {
235      StringBuilder msg = new StringBuilder();
236      msg.append("Failed after ").append(numVerified).append("!");
237      msg.append("Expected=").append(Bytes.toStringBinary(expected));
238      msg.append("Got:\n");
239      for (Cell kv : res.listCells()) {
240        msg.append(kv.toString());
241        msg.append(" val= ");
242        msg.append(Bytes.toStringBinary(CellUtil.cloneValue(kv)));
243        msg.append("\n");
244      }
245      throw new RuntimeException(msg.toString());
246    }
247  }

249  /**
250   * Thread that does full scans of the table looking for any partially completed rows.
251   */
252  public static class AtomicScanReader extends RepeatingTestThread {
253    byte[][] targetFamilies;
254    Table table;
255    Connection connection;
256    AtomicLong numScans = new AtomicLong();
257    AtomicLong numRowsScanned = new AtomicLong();
258
259    public AtomicScanReader(TestContext ctx, byte[][] targetFamilies, ExecutorService pool)
260        throws IOException {
261      super(ctx);
262      this.targetFamilies = targetFamilies;
263      connection = ConnectionFactory.createConnection(ctx.getConf(), pool);
264      table = connection.getTable(TABLE_NAME);
265    }
266
267    public void doAnAction() throws Exception {
268      Scan s = new Scan();
269      for (byte[] family : targetFamilies) {
270        s.addFamily(family);
271      }
272      ResultScanner scanner = table.getScanner(s);
273
274      for (Result res : scanner) {
275        byte[] gotValue = null;
276
277        for (byte[] family : targetFamilies) {
278          for (int i = 0; i < NUM_COLS_TO_CHECK; i++) {
279            byte qualifier[] = Bytes.toBytes("col" + i);
280            byte thisValue[] = res.getValue(family, qualifier);
281            if (gotValue != null && !Bytes.equals(gotValue, thisValue)) {
282              gotFailure(gotValue, res);
283            }
284            gotValue = thisValue;
285          }
286        }
287        numRowsScanned.getAndIncrement();
288      }
289      numScans.getAndIncrement();
290    }
291
292    @Override
293    public void workDone() throws IOException {
294      try {
295        table.close();
296      } finally {
297        connection.close();
298      }
299    }
300
301    private void gotFailure(byte[] expected, Result res) {
302      StringBuilder msg = new StringBuilder();
303      msg.append("Failed after ").append(numRowsScanned).append("!");
304      msg.append("Expected=").append(Bytes.toStringBinary(expected));
305      msg.append("Got:\n");
306      for (Cell kv : res.listCells()) {
307        msg.append(kv.toString());
308        msg.append(" val= ");
309        msg.append(Bytes.toStringBinary(CellUtil.cloneValue(kv)));
310        msg.append("\n");
311      }
312      throw new RuntimeException(msg.toString());
313    }
314  }

316  private void createTableIfMissing(Admin admin, boolean useMob) throws IOException {
317    if (!admin.tableExists(TABLE_NAME)) {
318      TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(TABLE_NAME);
319      Stream.of(FAMILIES).map(ColumnFamilyDescriptorBuilder::of)
320          .forEachOrdered(builder::addColumnFamily);
321      admin.createTable(builder.build());
322    }
323    ColumnFamilyDescriptor cfd = admin.getDescriptor(TABLE_NAME).getColumnFamilies()[0];
324    if (cfd.isMobEnabled() != useMob) {
325      admin.modifyColumnFamily(TABLE_NAME, ColumnFamilyDescriptorBuilder.newBuilder(cfd)
326          .setMobEnabled(useMob).setMobThreshold(4).build());
327    }
328  }
329
330  private void runTestAtomicity(Admin admin) throws Exception {
331    createTableIfMissing(admin, useMob);
332    TestContext ctx = new TestContext(conf);
333
334    byte rows[][] = new byte[numUniqueRows][];
335    for (int i = 0; i < numUniqueRows; i++) {
336      rows[i] = Bytes.toBytes("test_row_" + i);
337    }
338
339    List<AtomicityWriter> writers = Lists.newArrayList();
340    for (int i = 0; i < numWriters; i++) {
341      AtomicityWriter writer = new AtomicityWriter(ctx, rows, FAMILIES, sharedPool);
342      writers.add(writer);
343      ctx.addThread(writer);
344    }
345    // Add a flusher
346    ctx.addThread(new RepeatingTestThread(ctx) {
347      public void doAnAction() throws Exception {
348        try {
349          admin.flush(TABLE_NAME);
350        } catch (IOException ioe) {
351          LOG.warn("Ignoring exception while flushing: " + StringUtils.stringifyException(ioe));
352        }
353        // Flushing has been a source of ACID violations previously (see HBASE-2856), so ideally,
354        // we would flush as often as possible. On a running cluster, this isn't practical:
355        // (1) we will cause a lot of load due to all the flushing and compacting
356        // (2) we cannot change the flushing/compacting related Configuration options to try to
357        // alleviate this
358        // (3) it is an unrealistic workload, since no one would actually flush that often.
359        // Therefore, let's flush every minute to have more flushes than usual, but not overload
360        // the running cluster.
361        if (!crazyFlush) {
362          Thread.sleep(60000);
363        }
364      }
365    });
366
367    List<AtomicGetReader> getters = Lists.newArrayList();
368    for (int i = 0; i < numGetters; i++) {
369      AtomicGetReader getter =
370          new AtomicGetReader(ctx, rows[i % numUniqueRows], FAMILIES, sharedPool);
371      getters.add(getter);
372      ctx.addThread(getter);
373    }
374
375    List<AtomicScanReader> scanners = Lists.newArrayList();
376    for (int i = 0; i < numScanners; i++) {
377      AtomicScanReader scanner = new AtomicScanReader(ctx, FAMILIES, sharedPool);
378      scanners.add(scanner);
379      ctx.addThread(scanner);
380    }
381
382    ctx.startThreads();
383    ctx.waitFor(millisToRun);
384    ctx.stop();
385
386    LOG.info("Finished test. Writers:");
387    for (AtomicityWriter writer : writers) {
388      LOG.info("  wrote " + writer.numWritten.get());
389    }
390    LOG.info("Readers:");
391    for (AtomicGetReader reader : getters) {
392      LOG.info("  read " + reader.numRead.get());
393    }
394    LOG.info("Scanners:");
395    for (AtomicScanReader scanner : scanners) {
396      LOG.info("  scanned " + scanner.numScans.get());
397      LOG.info("  verified " + scanner.numRowsScanned.get() + " rows");
398    }
399  }

401  public static void main(String[] args) {
402    Configuration c = HBaseConfiguration.create();
403    int status;
404    try {
405      AcidGuaranteesTestTool test = new AcidGuaranteesTestTool();
406      status = ToolRunner.run(c, test, args);
407    } catch (Exception e) {
408      LOG.error("Exiting due to error", e);
409      status = -1;
410    }
411    System.exit(status);
412  }
