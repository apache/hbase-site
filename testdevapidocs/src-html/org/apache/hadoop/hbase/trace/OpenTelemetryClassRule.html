018package org.apache.hadoop.hbase.trace;
019
020import io.opentelemetry.api.GlobalOpenTelemetry;
021import io.opentelemetry.api.OpenTelemetry;
022import io.opentelemetry.api.trace.propagation.W3CTraceContextPropagator;
023import io.opentelemetry.context.propagation.ContextPropagators;
024import io.opentelemetry.sdk.OpenTelemetrySdk;
025import io.opentelemetry.sdk.testing.exporter.InMemorySpanExporter;
026import io.opentelemetry.sdk.testing.junit4.OpenTelemetryRule;
027import io.opentelemetry.sdk.trace.SdkTracerProvider;
028import io.opentelemetry.sdk.trace.data.SpanData;
029import io.opentelemetry.sdk.trace.export.SimpleSpanProcessor;
030import java.util.List;
031import org.junit.rules.ExternalResource;
032
033/**
034 * <p>Like {@link OpenTelemetryRule}, except modeled after the junit5 implementation
035 * {@code OpenTelemetryExtension}. Use this class when you need to make asserts on {@link SpanData}
036 * created on a MiniCluster. Make sure this rule initialized before the MiniCluster so that it can
037 * register its instance of {@link OpenTelemetry} as the global instance before any server-side
038 * component can call {@link TraceUtil#getGlobalTracer()}.</p>
039 * <p>For example:</p>
040 * <pre>{@code
041 * public class TestMyClass {
042 *   private static final OpenTelemetryClassRule otelClassRule =
043 *     OpenTelemetryClassRule.create();
044 *   private static final MiniClusterRule miniClusterRule =
045 *     MiniClusterRule.newBuilder().build();
046 *   protected static final ConnectionRule connectionRule =
047 *     new ConnectionRule(miniClusterRule::createConnection);
048 *
049 *   @ClassRule
050 *   public static final TestRule classRule = RuleChain.outerRule(otelClassRule)
051 *     .around(miniClusterRule)
052 *     .around(connectionRule);
053 *
054 *   @Rule
055 *   public final OpenTelemetryTestRule otelTestRule =
056 *     new OpenTelemetryTestRule(otelClassRule);
057 *
058 *   @Test
059 *   public void myTest() {
060 *     // ...
061 *     // do something that makes spans
062 *     final List<SpanData> spans = otelClassRule.getSpans();
063 *     // make assertions on them
064 *   }
065 * }
066 * }</pre>
067 *
068 * @see <a href="https://github.com/open-telemetry/opentelemetry-java/blob/9a330d0/sdk/testing/src/main/java/io/opentelemetry/sdk/testing/junit5/OpenTelemetryExtension.java">junit5/OpenTelemetryExtension.java</a>
069 */
070public final class OpenTelemetryClassRule extends ExternalResource {
071
072  public static OpenTelemetryClassRule create() {
073    InMemorySpanExporter spanExporter = InMemorySpanExporter.create();
074
075    SdkTracerProvider tracerProvider =
076      SdkTracerProvider.builder()
077        .addSpanProcessor(SimpleSpanProcessor.create(spanExporter))
078        .build();
079
080    OpenTelemetrySdk openTelemetry =
081      OpenTelemetrySdk.builder()
082        .setPropagators(ContextPropagators.create(W3CTraceContextPropagator.getInstance()))
083        .setTracerProvider(tracerProvider)
084        .build();
085
086    return new OpenTelemetryClassRule(openTelemetry, spanExporter);
087  }
088
089  private final OpenTelemetrySdk openTelemetry;
090  private final InMemorySpanExporter spanExporter;
091
092  private OpenTelemetryClassRule(
093    final OpenTelemetrySdk openTelemetry,
094    final InMemorySpanExporter spanExporter
095  ) {
096    this.openTelemetry = openTelemetry;
097    this.spanExporter = spanExporter;
098  }
099
100  /** Returns the {@link OpenTelemetry} created by this Rule. */
101  public OpenTelemetry getOpenTelemetry() {
102    return openTelemetry;
103  }
104
105  /** Returns all the exported {@link SpanData} so far. */
106  public List<SpanData> getSpans() {
107    return spanExporter.getFinishedSpanItems();
108  }
109
110  /**
111   * Clears the collected exported {@link SpanData}.
112   */
113  public void clearSpans() {
114    spanExporter.reset();
115  }
116
117  @Override
118  protected void before() throws Throwable {
119    GlobalOpenTelemetry.resetForTest();
120    GlobalOpenTelemetry.set(openTelemetry);
121  }
122
123  @Override
124  protected void after() {
125    GlobalOpenTelemetry.resetForTest();
126  }
127}
