051@Category(IntegrationTests.class)
052public class IntegrationTestSendTraceRequests extends AbstractHBaseTool {
053
054  public static final String TABLE_ARG = "t";
055  public static final String CF_ARG = "f";
056
057  public static final String TABLE_NAME_DEFAULT = "SendTracesTable";
058  public static final String COLUMN_FAMILY_DEFAULT = "D";
059  private TableName tableName = TableName.valueOf(TABLE_NAME_DEFAULT);
060  private byte[] familyName = Bytes.toBytes(COLUMN_FAMILY_DEFAULT);
061  private IntegrationTestingUtility util;
062  private Random random = new Random();
063  private Admin admin;
064  private SpanReceiverHost receiverHost;
065
066  public static void main(String[] args) throws Exception {
067    Configuration configuration = HBaseConfiguration.create();
068    IntegrationTestingUtility.setUseDistributedCluster(configuration);
069    IntegrationTestSendTraceRequests tool = new IntegrationTestSendTraceRequests();
070    ToolRunner.run(configuration, tool, args);
071  }
072
073  @Override
074  protected void addOptions() {
075    addOptWithArg(TABLE_ARG, "The table name to target.  Will be created if not there already.");
076    addOptWithArg(CF_ARG, "The family to target");
077  }
078
079  @Override
080  public void processOptions(CommandLine cmd) {
081    String tableNameString = cmd.getOptionValue(TABLE_ARG, TABLE_NAME_DEFAULT);
082    String familyString = cmd.getOptionValue(CF_ARG, COLUMN_FAMILY_DEFAULT);
083
084    this.tableName = TableName.valueOf(tableNameString);
085    this.familyName = Bytes.toBytes(familyString);
086  }
087
088  @Override
089  public int doWork() throws Exception {
090    internalDoWork();
091    return 0;
092  }
093
094  @Test
095  public void internalDoWork() throws Exception {
096    util = createUtil();
097    admin = util.getHBaseAdmin();
098    setupReceiver();
099
100    deleteTable();
101    createTable();
102    LinkedBlockingQueue<Long> rks = insertData();
103
104    ExecutorService service = Executors.newFixedThreadPool(20);
105    doScans(service, rks);
106    doGets(service, rks);
107
108    service.shutdown();
109    service.awaitTermination(100, TimeUnit.SECONDS);
110    Thread.sleep(90000);
111    receiverHost.closeReceivers();
112    util.restoreCluster();
113    util = null;
114  }
115
116  private void doScans(ExecutorService service, final LinkedBlockingQueue<Long> rks) {
117
118      for (int i = 0; i < 100; i++) {
119        Runnable runnable = new Runnable() {
120          private TraceScope innerScope = null;
121          private final LinkedBlockingQueue<Long> rowKeyQueue = rks;
122          @Override
123          public void run() {
124            ResultScanner rs = null;
125            try {
126              innerScope = Trace.startSpan("Scan", Sampler.ALWAYS);
127              Table ht = util.getConnection().getTable(tableName);
128              Scan s = new Scan();
129              s.setStartRow(Bytes.toBytes(rowKeyQueue.take()));
130              s.setBatch(7);
131              rs = ht.getScanner(s);
132              // Something to keep the jvm from removing the loop.
133              long accum = 0;
134
135              for(int x = 0; x < 1000; x++) {
136                Result r = rs.next();
137                accum |= Bytes.toLong(r.getRow());
138              }
139
140              innerScope.getSpan().addTimelineAnnotation("Accum result = " + accum);
141
142              ht.close();
143              ht = null;
144            } catch (IOException e) {
145              e.printStackTrace();
146
147              innerScope.getSpan().addKVAnnotation(
148                  Bytes.toBytes("exception"),
149                  Bytes.toBytes(e.getClass().getSimpleName()));
150
151            } catch (Exception e) {
152            } finally {
153              if (innerScope != null) innerScope.close();
154              if (rs != null) rs.close();
155            }
156
157          }
158        };
159        service.submit(runnable);
160      }
161
162  }
163
164  private void doGets(ExecutorService service, final LinkedBlockingQueue<Long> rowKeys)
165      throws IOException {
166    for (int i = 0; i < 100; i++) {
167      Runnable runnable = new Runnable() {
168        private TraceScope innerScope = null;
169        private final LinkedBlockingQueue<Long> rowKeyQueue = rowKeys;
170
171        @Override
172        public void run() {
173
174
175          Table ht = null;
176          try {
177            ht = util.getConnection().getTable(tableName);
178          } catch (IOException e) {
179            e.printStackTrace();
180          }
181
182          long accum = 0;
183          for (int x = 0; x < 5; x++) {
184            try {
185              innerScope = Trace.startSpan("gets", Sampler.ALWAYS);
186              long rk = rowKeyQueue.take();
187              Result r1 = ht.get(new Get(Bytes.toBytes(rk)));
188              if (r1 != null) {
189                accum |= Bytes.toLong(r1.getRow());
190              }
191              Result r2 = ht.get(new Get(Bytes.toBytes(rk)));
192              if (r2 != null) {
193                accum |= Bytes.toLong(r2.getRow());
194              }
195              innerScope.getSpan().addTimelineAnnotation("Accum = " + accum);
196
197            } catch (IOException e) {
198              // IGNORED
199            } catch (InterruptedException ie) {
200              // IGNORED
201            } finally {
202              if (innerScope != null) innerScope.close();
203            }
204          }
205
206        }
207      };
208      service.submit(runnable);
209    }
210  }
211
212  private void createTable() throws IOException {
213    TraceScope createScope = null;
214    try {
215      createScope = Trace.startSpan("createTable", Sampler.ALWAYS);
216      util.createTable(tableName, familyName);
217    } finally {
218      if (createScope != null) createScope.close();
219    }
220  }
221
222  private void deleteTable() throws IOException {
223    TraceScope deleteScope = null;
224
225    try {
226      if (admin.tableExists(tableName)) {
227        deleteScope = Trace.startSpan("deleteTable", Sampler.ALWAYS);
228        util.deleteTable(tableName);
229      }
230    } finally {
231      if (deleteScope != null) deleteScope.close();
232    }
233  }
234
235  private LinkedBlockingQueue<Long> insertData() throws IOException, InterruptedException {
236    LinkedBlockingQueue<Long> rowKeys = new LinkedBlockingQueue<Long>(25000);
237    BufferedMutator ht = util.getConnection().getBufferedMutator(this.tableName);
238    byte[] value = new byte[300];
239    for (int x = 0; x < 5000; x++) {
240      TraceScope traceScope = Trace.startSpan("insertData", Sampler.ALWAYS);
241      try {
242        for (int i = 0; i < 5; i++) {
243          long rk = random.nextLong();
244          rowKeys.add(rk);
245          Put p = new Put(Bytes.toBytes(rk));
246          for (int y = 0; y < 10; y++) {
247            random.nextBytes(value);
248            p.addColumn(familyName, Bytes.toBytes(random.nextLong()), value);
249          }
250          ht.mutate(p);
251        }
252        if ((x % 1000) == 0) {
253          admin.flush(tableName);
254        }
255      } finally {
256        traceScope.close();
257      }
258    }
259    admin.flush(tableName);
260    return rowKeys;
261  }
262
263  private IntegrationTestingUtility createUtil() throws Exception {
264    Configuration conf = getConf();
265    if (this.util == null) {
266      IntegrationTestingUtility u;
267      if (conf == null) {
268        u = new IntegrationTestingUtility();
269      } else {
270        u = new IntegrationTestingUtility(conf);
271      }
272      util = u;
273      util.initializeCluster(1);
274
275    }
276    return this.util;
277  }
278
279  private void setupReceiver() {
280    Configuration conf = new Configuration(util.getConfiguration());
281    conf.setBoolean("hbase.zipkin.is-in-client-mode", true);
282
283    this.receiverHost = SpanReceiverHost.getInstance(conf);
284  }
