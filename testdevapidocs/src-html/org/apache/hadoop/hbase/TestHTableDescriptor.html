095  @Test
096  public void testPb() throws DeserializationException, IOException {
097    HTableDescriptor htd = new HTableDescriptor(TableName.META_TABLE_NAME);
098    final int v = 123;
099    htd.setMaxFileSize(v);
100    htd.setDurability(Durability.ASYNC_WAL);
101    htd.setReadOnly(true);
102    htd.setRegionReplication(2);
103    byte [] bytes = htd.toByteArray();
104    HTableDescriptor deserializedHtd = HTableDescriptor.parseFrom(bytes);
105    assertEquals(htd, deserializedHtd);
106    assertEquals(v, deserializedHtd.getMaxFileSize());
107    assertTrue(deserializedHtd.isReadOnly());
108    assertEquals(Durability.ASYNC_WAL, deserializedHtd.getDurability());
109    assertEquals(deserializedHtd.getRegionReplication(), 2);
110  }

133  @Test
134  public void testSetListRemoveCP() throws Exception {
135    HTableDescriptor desc = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
136    // simple CP
137    String className1 = "org.apache.hadoop.hbase.coprocessor.SimpleRegionObserver";
138    String className2 = "org.apache.hadoop.hbase.coprocessor.SampleRegionWALObserver";
139    // Check that any coprocessor is present.
140    assertTrue(desc.getCoprocessors().isEmpty());
141
142    // Add the 1 coprocessor and check if present.
143    desc.addCoprocessor(className1);
144    assertTrue(desc.getCoprocessors().size() == 1);
145    assertTrue(desc.getCoprocessors().contains(className1));
146
147    // Add the 2nd coprocessor and check if present.
148    // remove it and check that it is gone
149    desc.addCoprocessor(className2);
150    assertTrue(desc.getCoprocessors().size() == 2);
151    assertTrue(desc.getCoprocessors().contains(className2));
152
153    // Remove one and check
154    desc.removeCoprocessor(className1);
155    assertTrue(desc.getCoprocessors().size() == 1);
156    assertFalse(desc.getCoprocessors().contains(className1));
157    assertTrue(desc.getCoprocessors().contains(className2));
158
159    // Remove the last and check
160    desc.removeCoprocessor(className2);
161    assertTrue(desc.getCoprocessors().isEmpty());
162    assertFalse(desc.getCoprocessors().contains(className1));
163    assertFalse(desc.getCoprocessors().contains(className2));
164  }

278  @Test
279  public void testModifyFamily() {
280    HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
281    byte[] familyName = Bytes.toBytes("cf");
282    HColumnDescriptor hcd = new HColumnDescriptor(familyName);
283    hcd.setBlocksize(1000);
284    hcd.setDFSReplication((short) 3);
285    htd.addFamily(hcd);
286    assertEquals(1000, htd.getFamily(familyName).getBlocksize());
287    assertEquals(3, htd.getFamily(familyName).getDFSReplication());
288    hcd = new HColumnDescriptor(familyName);
289    hcd.setBlocksize(2000);
290    hcd.setDFSReplication((short) 1);
291    htd.modifyFamily(hcd);
292    assertEquals(2000, htd.getFamily(familyName).getBlocksize());
293    assertEquals(1, htd.getFamily(familyName).getDFSReplication());
294  }
