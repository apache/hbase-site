254  @Test (timeout = 60000)
255  public void testOnlyReplayingFlushStartDoesNotHoldUpRegionClose() throws IOException {
256    // load some data to primary and flush
257    int start = 0;
258    LOG.info("-- Writing some data to primary from " +  start + " to " + (start+100));
259    putData(primaryRegion, Durability.SYNC_WAL, start, 100, cq, families);
260    LOG.info("-- Flushing primary, creating 3 files for 3 stores");
261    primaryRegion.flush(true);
262
263    // now replay the edits and the flush marker
264    reader = createWALReaderForPrimary();
265
266    LOG.info("-- Replaying edits and flush events in secondary");
267    while (true) {
268      WAL.Entry entry = reader.next();
269      if (entry == null) {
270        break;
271      }
272      FlushDescriptor flushDesc
273        = WALEdit.getFlushDescriptor(entry.getEdit().getCells().get(0));
274      if (flushDesc != null) {
275        if (flushDesc.getAction() == FlushAction.START_FLUSH) {
276          LOG.info("-- Replaying flush start in secondary");
277          secondaryRegion.replayWALFlushStartMarker(flushDesc);
278        } else if (flushDesc.getAction() == FlushAction.COMMIT_FLUSH) {
279          LOG.info("-- NOT Replaying flush commit in secondary");
280        }
281      } else {
282        replayEdit(secondaryRegion, entry);
283      }
284    }
285
286    assertTrue(rss.getRegionServerAccounting().getGlobalMemStoreDataSize() > 0);
287    // now close the region which should not cause hold because of un-committed flush
288    secondaryRegion.close();
289
290    // verify that the memstore size is back to what it was
291    assertEquals(0, rss.getRegionServerAccounting().getGlobalMemStoreDataSize());

294  static int replayEdit(HRegion region, WAL.Entry entry) throws IOException {
295    if (WALEdit.isMetaEditFamily(entry.getEdit().getCells().get(0))) {
296      return 0; // handled elsewhere
297    }
298    Put put = new Put(CellUtil.cloneRow(entry.getEdit().getCells().get(0)));
299    for (Cell cell : entry.getEdit().getCells()) put.add(cell);
300    put.setDurability(Durability.SKIP_WAL);
301    MutationReplay mutation = new MutationReplay(MutationType.PUT, put, 0, 0);
302    region.batchReplay(new MutationReplay[] {mutation},
303      entry.getKey().getSequenceId());
304    return Integer.parseInt(Bytes.toString(put.getRow()));
305  }

336    // initiate a secondary region with some data.
337
338    // load some data to primary and flush. 3 flushes and some more unflushed data
339    putDataWithFlushes(primaryRegion, 100, 300, 100);
340

342    LOG.info("-- Compacting primary, only 1 store");
343    primaryRegion.compactStore(Bytes.toBytes("cf1"),
344      NoLimitThroughputController.INSTANCE);
345
346    // now replay the edits and the flush marker
347    reader = createWALReaderForPrimary();
348
349    LOG.info("-- Replaying edits and flush events in secondary");
350    int lastReplayed = 0;
351    int expectedStoreFileCount = 0;
352    while (true) {
353      WAL.Entry entry = reader.next();
354      if (entry == null) {
355        break;
356      }
357      FlushDescriptor flushDesc
358      = WALEdit.getFlushDescriptor(entry.getEdit().getCells().get(0));
359      CompactionDescriptor compactionDesc
360      = WALEdit.getCompaction(entry.getEdit().getCells().get(0));
361      if (flushDesc != null) {
362        // first verify that everything is replayed and visible before flush event replay
363        verifyData(secondaryRegion, 0, lastReplayed, cq, families);
364        HStore store = secondaryRegion.getStore(Bytes.toBytes("cf1"));
365        long storeMemstoreSize = store.getMemStoreSize().getHeapSize();
366        long regionMemstoreSize = secondaryRegion.getMemStoreSize();
367        long storeFlushableSize = store.getFlushableSize().getHeapSize();
368        long storeSize = store.getSize();
369        long storeSizeUncompressed = store.getStoreSizeUncompressed();
370        if (flushDesc.getAction() == FlushAction.START_FLUSH) {
371          LOG.info("-- Replaying flush start in secondary");
372          PrepareFlushResult result = secondaryRegion.replayWALFlushStartMarker(flushDesc);
373          assertNull(result.result);
374          assertEquals(result.flushOpSeqId, flushDesc.getFlushSequenceNumber());
375
376          // assert that the store memstore is smaller now
377          long newStoreMemstoreSize = store.getMemStoreSize().getHeapSize();
378          LOG.info("Memstore size reduced by:"
379              + StringUtils.humanReadableInt(newStoreMemstoreSize - storeMemstoreSize));
380          assertTrue(storeMemstoreSize > newStoreMemstoreSize);
381
382        } else if (flushDesc.getAction() == FlushAction.COMMIT_FLUSH) {
383          LOG.info("-- Replaying flush commit in secondary");
384          secondaryRegion.replayWALFlushCommitMarker(flushDesc);
385
386          // assert that the flush files are picked
387          expectedStoreFileCount++;
388          for (HStore s : secondaryRegion.getStores()) {
389            assertEquals(expectedStoreFileCount, s.getStorefilesCount());
390          }
391          long newFlushableSize = store.getFlushableSize().getHeapSize();
392          assertTrue(storeFlushableSize > newFlushableSize);
393
394          // assert that the region memstore is smaller now
395          long newRegionMemstoreSize = secondaryRegion.getMemStoreSize();
396          assertTrue(regionMemstoreSize > newRegionMemstoreSize);
397
398          // assert that the store sizes are bigger
399          assertTrue(store.getSize() > storeSize);
400          assertTrue(store.getStoreSizeUncompressed() > storeSizeUncompressed);
401          assertEquals(store.getSize(), store.getStorefilesSize());
402        }
403        // after replay verify that everything is still visible
404        verifyData(secondaryRegion, 0, lastReplayed+1, cq, families);
405      } else if (compactionDesc != null) {
406        secondaryRegion.replayWALCompactionMarker(compactionDesc, true, false, Long.MAX_VALUE);
407
408        // assert that the compaction is applied
409        for (HStore store : secondaryRegion.getStores()) {
410          if (store.getColumnFamilyName().equals("cf1")) {
411            assertEquals(1, store.getStorefilesCount());
412          } else {
413            assertEquals(expectedStoreFileCount, store.getStorefilesCount());
414          }
415        }
416      } else {
417        lastReplayed = replayEdit(secondaryRegion, entry);;
418      }
419    }
420
421    assertEquals(400-1, lastReplayed);
422    LOG.info("-- Verifying edits from secondary");
423    verifyData(secondaryRegion, 0, 400, cq, families);
424
425    LOG.info("-- Verifying edits from primary. Ensuring that files are not deleted");
426    verifyData(primaryRegion, 0, lastReplayed, cq, families);
427    for (HStore store : primaryRegion.getStores()) {
428      if (store.getColumnFamilyName().equals("cf1")) {
429        assertEquals(1, store.getStorefilesCount());
430      } else {
431        assertEquals(expectedStoreFileCount, store.getStorefilesCount());
432      }
433    }

1664   * @throws IOException */
1665  private void putDataWithFlushes(HRegion region, int flushInterval,
1666      int numRows, int numRowsAfterFlush) throws IOException {

1671      LOG.info("-- Flushing primary, creating 3 files for 3 stores");
1672      region.flush(true);
1673    }

1675    putData(region, Durability.SYNC_WAL, start, numRowsAfterFlush, cq, families);
