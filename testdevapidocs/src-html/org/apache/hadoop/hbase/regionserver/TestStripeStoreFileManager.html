123  @Test
124  public void testRowKeyBefore() throws Exception {
125    StripeStoreFileManager manager = createManager();
126    StoreFile l0File = createFile(), l0File2 = createFile();
127    manager.insertNewFiles(al(l0File));
128    manager.insertNewFiles(al(l0File2));
129    // Get candidate files.
130    Iterator<StoreFile> sfs = manager.getCandidateFilesForRowKeyBefore(KV_B);
131    sfs.next();
132    sfs.remove();
133    // Suppose we found a candidate in this file... make sure L0 file remaining is not removed.
134    sfs = manager.updateCandidateFilesForRowKeyBefore(sfs, KV_B, KV_A);
135    assertTrue(sfs.hasNext());
136    // Now add some stripes (remove L0 file too)
137    MockStoreFile stripe0a = createFile(0, 100, OPEN_KEY, KEY_B),
138        stripe1 = createFile(KEY_B, OPEN_KEY);
139    manager.addCompactionResults(al(l0File), al(stripe0a, stripe1));
140    manager.removeCompactedFiles(al(l0File));
141    // If we want a key <= KEY_A, we should get everything except stripe1.
142    ArrayList<StoreFile> sfsDump = dumpIterator(manager.getCandidateFilesForRowKeyBefore(KV_A));
143    assertEquals(2, sfsDump.size());
144    assertTrue(sfsDump.contains(stripe0a));
145    assertFalse(sfsDump.contains(stripe1));
146    // If we want a key <= KEY_B, we should get everything since lower bound is inclusive.
147    sfsDump = dumpIterator(manager.getCandidateFilesForRowKeyBefore(KV_B));
148    assertEquals(3, sfsDump.size());
149    assertTrue(sfsDump.contains(stripe1));
150    // For KEY_D, we should also get everything.
151    sfsDump = dumpIterator(manager.getCandidateFilesForRowKeyBefore(KV_D));
152    assertEquals(3, sfsDump.size());
153    // Suppose in the first file we found candidate with KEY_C.
154    // Then, stripe0 no longer matters and should be removed, but stripe1 should stay.
155    sfs = manager.getCandidateFilesForRowKeyBefore(KV_D);
156    sfs.next(); // Skip L0 file.
157    sfs.remove();
158    sfs = manager.updateCandidateFilesForRowKeyBefore(sfs, KV_D, KV_C);
159    assertEquals(stripe1, sfs.next());
160    assertFalse(sfs.hasNext());
161    // Add one more, later, file to stripe0, remove the last annoying L0 file.
162    // This file should be returned in preference to older L0 file; also, after we get
163    // a candidate from the first file, the old one should not be removed.
164    StoreFile stripe0b = createFile(0, 101, OPEN_KEY, KEY_B);
165    manager.addCompactionResults(al(l0File2), al(stripe0b));
166    manager.removeCompactedFiles(al(l0File2));
167    sfs = manager.getCandidateFilesForRowKeyBefore(KV_A);
168    assertEquals(stripe0b, sfs.next());
169    sfs.remove();
170    sfs = manager.updateCandidateFilesForRowKeyBefore(sfs, KV_A, KV_A);
171    assertEquals(stripe0a, sfs.next());
172  }

225  /**
226   * Verifies scenario for finding a split point.
227   * @param splitPointAfter Stripe to expect the split point at/after.
228   * @param shouldSplitStripe If true, the split point is expected in the middle of the above
229   *                          stripe; if false, should be at the end.
230   * @param splitRatioToVerify Maximum split imbalance ratio.
231   * @param sizes Stripe sizes.
232   */
233  private void verifySplitPointScenario(int splitPointAfter, boolean shouldSplitStripe,
234      float splitRatioToVerify, int... sizes) throws Exception {
235    assertTrue(sizes.length > 1);
236    ArrayList<StoreFile> sfs = new ArrayList<>();
237    for (int sizeIx = 0; sizeIx < sizes.length; ++sizeIx) {
238      byte[] startKey = (sizeIx == 0) ? OPEN_KEY : Bytes.toBytes(sizeIx - 1);
239      byte[] endKey = (sizeIx == sizes.length - 1) ? OPEN_KEY : Bytes.toBytes(sizeIx);
240      MockStoreFile sf = createFile(sizes[sizeIx], 0, startKey, endKey);
241      sf.splitPoint = Bytes.toBytes(-sizeIx); // set split point to the negative index
242      sfs.add(sf);
243    }
244
245    Configuration conf = HBaseConfiguration.create();
246    if (splitRatioToVerify != 0) {
247      conf.setFloat(StripeStoreConfig.MAX_REGION_SPLIT_IMBALANCE_KEY, splitRatioToVerify);
248    }
249    StripeStoreFileManager manager = createManager(al(), conf);
250    manager.addCompactionResults(al(), sfs);
251    int result = Bytes.toInt(manager.getSplitPoint());
252    // Either end key and thus positive index, or "middle" of the file and thus negative index.
253    assertEquals(splitPointAfter * (shouldSplitStripe ? -1 : 1), result);
254  }

256  private static byte[] keyAfter(byte[] key) {
257    return Arrays.copyOf(key, key.length + 1);
258  }

367    StripeStoreFileManager manager = createManager();
368    // First, add some L0 files and "compact" one with new stripe creation.
369    StoreFile sf_L0_0a = createFile(), sf_L0_0b = createFile();

376    verifyInvalidCompactionScenario(manager, al(sf_L0_0a), al(createFile(OPEN_KEY, KEY_B),
377        createFile(KEY_B, OPEN_KEY), createFile(KEY_A, KEY_D)));
378    verifyInvalidCompactionScenario(manager, al(sf_L0_0a), al(createFile(OPEN_KEY, KEY_B),
379        createFile(KEY_A, KEY_B), createFile(KEY_B, OPEN_KEY)));
380
381    StoreFile sf_i2B_0 = createFile(OPEN_KEY, KEY_B);
382    StoreFile sf_B2C_0 = createFile(KEY_B, KEY_C);
383    StoreFile sf_C2i_0 = createFile(KEY_C, OPEN_KEY);

385    manager.removeCompactedFiles(al(sf_L0_0a));
386    verifyAllFiles(manager, al(sf_L0_0b, sf_i2B_0, sf_B2C_0, sf_C2i_0));

388    // Add another l0 file, "compact" both L0 into two stripes

397    // Try compacting with invalid file (no metadata) - should add files to L0.
398    StoreFile sf_L0_2 = createFile(null, null);
399    manager.addCompactionResults(al(), al(sf_L0_2));
400    manager.removeCompactedFiles(al());
401    verifyAllFiles(manager, al(sf_i2B_0, sf_B2C_0, sf_C2i_0, sf_i2B_1, sf_B2C_1, sf_L0_2));
402    // Remove it...
403    manager.addCompactionResults(al(sf_L0_2), al());
404    manager.removeCompactedFiles(al(sf_L0_2));
405
406    // Do regular compaction in the first stripe.
407    StoreFile sf_i2B_3 = createFile(OPEN_KEY, KEY_B);
408    manager.addCompactionResults(al(sf_i2B_0, sf_i2B_1), al(sf_i2B_3));
409    manager.removeCompactedFiles(al(sf_i2B_0, sf_i2B_1));
410    verifyAllFiles(manager, al(sf_B2C_0, sf_C2i_0, sf_B2C_1, sf_i2B_3));
411
412    // Rebalance two stripes.
413    StoreFile sf_B2D_4 = createFile(KEY_B, KEY_D);
414    StoreFile sf_D2i_4 = createFile(KEY_D, OPEN_KEY);
415    manager.addCompactionResults(al(sf_B2C_0, sf_C2i_0, sf_B2C_1), al(sf_B2D_4, sf_D2i_4));
416    manager.removeCompactedFiles(al(sf_B2C_0, sf_C2i_0, sf_B2C_1));
417    verifyAllFiles(manager, al(sf_i2B_3, sf_B2D_4, sf_D2i_4));
418
419    // Split the first stripe.
420    StoreFile sf_i2A_5 = createFile(OPEN_KEY, KEY_A);
421    StoreFile sf_A2B_5 = createFile(KEY_A, KEY_B);
422    manager.addCompactionResults(al(sf_i2B_3), al(sf_i2A_5, sf_A2B_5));
423    manager.removeCompactedFiles(al(sf_i2B_3));
424    verifyAllFiles(manager, al(sf_B2D_4, sf_D2i_4, sf_i2A_5, sf_A2B_5));
425
426    // Split the middle stripe.
427    StoreFile sf_B2C_6 = createFile(KEY_B, KEY_C);
428    StoreFile sf_C2D_6 = createFile(KEY_C, KEY_D);
429    manager.addCompactionResults(al(sf_B2D_4), al(sf_B2C_6, sf_C2D_6));
430    manager.removeCompactedFiles(al(sf_B2D_4));
431    verifyAllFiles(manager, al(sf_D2i_4, sf_i2A_5, sf_A2B_5, sf_B2C_6, sf_C2D_6));
432
433    // Merge two different middle stripes.
434    StoreFile sf_A2C_7 = createFile(KEY_A, KEY_C);
435    manager.addCompactionResults(al(sf_A2B_5, sf_B2C_6), al(sf_A2C_7));
436    manager.removeCompactedFiles(al(sf_A2B_5, sf_B2C_6));
437    verifyAllFiles(manager, al(sf_D2i_4, sf_i2A_5, sf_C2D_6, sf_A2C_7));

441    manager.addCompactionResults(al(sf_i2A_5, sf_A2C_7), al(sf_i2C_8));
442    manager.removeCompactedFiles(al(sf_i2A_5, sf_A2C_7));
443    verifyAllFiles(manager, al(sf_D2i_4, sf_C2D_6, sf_i2C_8));

457    StoreFile sf_i2c = createFile(OPEN_KEY, KEY_C), sf_c2i = createFile(KEY_C, OPEN_KEY);

500  @Test
501  public void testPriority() throws Exception {
502    // Expected priority, file limit, stripe count, files per stripe, l0 files.
503    testPriorityScenario(5,    5, 0, 0, 0);
504    testPriorityScenario(2,    5, 0, 0, 3);
505    testPriorityScenario(4,   25, 5, 1, 0); // example case.
506    testPriorityScenario(3,   25, 5, 1, 1); // L0 files counts for all stripes.
507    testPriorityScenario(3,   25, 5, 2, 0); // file to each stripe - same as one L0 file.
508    testPriorityScenario(2,   25, 5, 4, 0); // 1 is priority user, so 2 is returned.
509    testPriorityScenario(2,   25, 5, 4, 4); // don't return higher than user unless over limit.
510    testPriorityScenario(2,   25, 5, 1, 10); // same.
511    testPriorityScenario(0,   25, 5, 4, 5); // at limit.
512    testPriorityScenario(-5,  25, 5, 6, 0); // over limit!
513    testPriorityScenario(-1,  25, 0, 0, 26); // over limit with just L0
514  }

516  private void testPriorityScenario(int expectedPriority,
517      int limit, int stripes, int filesInStripe, int l0Files) throws Exception
518  {
519    final byte[][] keys = { KEY_A, KEY_B, KEY_C, KEY_D };
520    assertTrue(stripes <= keys.length + 1);
521    Configuration conf = TEST_UTIL.getConfiguration();
522    conf.setInt("hbase.hstore.blockingStoreFiles", limit);
523    StripeStoreFileManager sfm = createManager(al(), conf);
524    for (int i = 0; i < l0Files; ++i) {
525      sfm.insertNewFiles(al(createFile()));
526    }
527    for (int i = 0; i < filesInStripe; ++i) {
528      ArrayList<StoreFile> stripe = new ArrayList<>();
529      for (int j = 0; j < stripes; ++j) {
530        stripe.add(createFile(
531            (j == 0) ? OPEN_KEY : keys[j - 1], (j == stripes - 1) ? OPEN_KEY : keys[j]));
532      }
533      sfm.addCompactionResults(al(), stripe);
534    }
535    assertEquals(expectedPriority, sfm.getStoreCompactionPriority());
536  }
