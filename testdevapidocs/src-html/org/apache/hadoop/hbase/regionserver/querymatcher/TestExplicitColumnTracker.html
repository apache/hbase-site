038@Category({ RegionServerTests.class, SmallTests.class })
039public class TestExplicitColumnTracker {
040
041  private final byte[] col1 = Bytes.toBytes("col1");
042  private final byte[] col2 = Bytes.toBytes("col2");
043  private final byte[] col3 = Bytes.toBytes("col3");
044  private final byte[] col4 = Bytes.toBytes("col4");
045  private final byte[] col5 = Bytes.toBytes("col5");
046
047  private void runTest(int maxVersions, TreeSet<byte[]> trackColumns, List<byte[]> scannerColumns,
048      List<MatchCode> expected) throws IOException {
049    ColumnTracker exp = new ExplicitColumnTracker(trackColumns, 0, maxVersions, Long.MIN_VALUE);
050
051    // Initialize result
052    List<ScanQueryMatcher.MatchCode> result = new ArrayList<>(scannerColumns.size());
053
054    long timestamp = 0;
055    // "Match"
056    for (byte[] col : scannerColumns) {
057      result.add(ScanQueryMatcher.checkColumn(exp, col, 0, col.length, ++timestamp,
058        KeyValue.Type.Put.getCode(), false));
059    }
060
061    assertEquals(expected.size(), result.size());
062    for (int i = 0; i < expected.size(); i++) {
063      assertEquals(expected.get(i), result.get(i));
064    }
065  }
066
067  @Test
068  public void testGetSingleVersion() throws IOException {
069    // Create tracker
070    TreeSet<byte[]> columns = new TreeSet<>(Bytes.BYTES_COMPARATOR);
071    // Looking for every other
072    columns.add(col2);
073    columns.add(col4);
074    List<MatchCode> expected = new ArrayList<>(5);
075    expected.add(ScanQueryMatcher.MatchCode.SEEK_NEXT_COL); // col1
076    expected.add(ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_COL); // col2
077    expected.add(ScanQueryMatcher.MatchCode.SEEK_NEXT_COL); // col3
078    expected.add(ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_ROW); // col4
079    expected.add(ScanQueryMatcher.MatchCode.SEEK_NEXT_ROW); // col5
080    int maxVersions = 1;
081
082    // Create "Scanner"
083    List<byte[]> scanner = new ArrayList<>(5);
084    scanner.add(col1);
085    scanner.add(col2);
086    scanner.add(col3);
087    scanner.add(col4);
088    scanner.add(col5);
089
090    runTest(maxVersions, columns, scanner, expected);
091  }

148  @Test
149  public void testStackOverflow() throws IOException {
150    int maxVersions = 1;
151    TreeSet<byte[]> columns = new TreeSet<>(Bytes.BYTES_COMPARATOR);
152    for (int i = 0; i < 100000; i++) {
153      columns.add(Bytes.toBytes("col" + i));
154    }
155
156    ColumnTracker explicit = new ExplicitColumnTracker(columns, 0, maxVersions, Long.MIN_VALUE);
157    for (int i = 0; i < 100000; i += 2) {
158      byte[] col = Bytes.toBytes("col" + i);
159      ScanQueryMatcher.checkColumn(explicit, col, 0, col.length, 1, KeyValue.Type.Put.getCode(),
160        false);
161    }
162    explicit.reset();
163
164    for (int i = 1; i < 100000; i += 2) {
165      byte[] col = Bytes.toBytes("col" + i);
166      ScanQueryMatcher.checkColumn(explicit, col, 0, col.length, 1, KeyValue.Type.Put.getCode(),
167        false);
168    }
169  }

174  @Test
175  public void testInfiniteLoop() throws IOException {
176    TreeSet<byte[]> columns = new TreeSet<>(Bytes.BYTES_COMPARATOR);
177    columns.addAll(Arrays.asList(new byte[][] { col2, col3, col5 }));
178    List<byte[]> scanner = Arrays.<byte[]> asList(new byte[][] { col1, col4 });
179    List<ScanQueryMatcher.MatchCode> expected = Arrays.<ScanQueryMatcher.MatchCode> asList(
180      new ScanQueryMatcher.MatchCode[] { ScanQueryMatcher.MatchCode.SEEK_NEXT_COL,
181          ScanQueryMatcher.MatchCode.SEEK_NEXT_COL });
182    runTest(1, columns, scanner, expected);
183  }
