073  private Table createTable() throws IOException {
074    TableName tableName = TableName.valueOf(name.getMethodName());
075    HTableDescriptor table = new HTableDescriptor(tableName);
076    HColumnDescriptor fam = new HColumnDescriptor(FAMILY);
077    fam.setNewVersionBehavior(true);
078    fam.setMaxVersions(3);
079    table.addFamily(fam);
080    TEST_UTIL.getHBaseAdmin().createTable(table);
081    return TEST_UTIL.getConnection().getTable(tableName);
082  }

084  @Test
085  public void testPutAndDeleteVersions() throws IOException {
086    try (Table t = createTable()) {
087      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000001, value));
088      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000002, value));
089      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000003, value));
090      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000004, value));
091      t.delete(new Delete(ROW).addColumns(FAMILY, col1, 2000000));
092      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000000, value));
093      TEST_UTIL.getAdmin().flush(t.getName());
094      Result r = t.get(new Get(ROW).setMaxVersions(3));
095      assertEquals(1, r.size());
096      assertEquals(1000000, r.rawCells()[0].getTimestamp());
097    }
098  }

218  @Test
219  public void testDeleteFamily() throws IOException {
220    try (Table t = createTable()) {
221
222      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000001, value));
223      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000002, value));
224      t.put(new Put(ROW).addColumn(FAMILY, col2, 1000002, value));
225      t.put(new Put(ROW).addColumn(FAMILY, col3, 1000001, value));
226
227      t.delete(new Delete(ROW).addFamily(FAMILY, 2000000));
228
229      t.put(new Put(ROW).addColumn(FAMILY, col3, 1500002, value));
230      t.put(new Put(ROW).addColumn(FAMILY, col2, 1500001, value));
231      t.put(new Put(ROW).addColumn(FAMILY, col1, 1500001, value));
232      t.put(new Put(ROW).addColumn(FAMILY, col1, 1500002, value));
233      TEST_UTIL.getAdmin().flush(t.getName());
234      Result r = t.get(new Get(ROW).setMaxVersions(3));
235      assertEquals(4, r.size());
236      assertEquals(1500002, r.rawCells()[0].getTimestamp());
237      assertEquals(1500001, r.rawCells()[1].getTimestamp());
238      assertEquals(1500001, r.rawCells()[2].getTimestamp());
239      assertEquals(1500002, r.rawCells()[3].getTimestamp());
240
241      t.delete(new Delete(ROW).addFamilyVersion(FAMILY, 1500001));
242
243      r = t.get(new Get(ROW).setMaxVersions(3));
244      assertEquals(2, r.size());
245      assertEquals(1500002, r.rawCells()[0].getTimestamp());
246      assertEquals(1500002, r.rawCells()[1].getTimestamp());
247
248      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000001, value));
249      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000002, value));
250      t.put(new Put(ROW).addColumn(FAMILY, col2, 1000002, value));
251      t.put(new Put(ROW).addColumn(FAMILY, col3, 1000001, value));
252      TEST_UTIL.getAdmin().flush(t.getName());
253      r = t.get(new Get(ROW).setMaxVersions(3));
254      assertEquals(6, r.size());
255      assertEquals(1500002, r.rawCells()[0].getTimestamp());
256      assertEquals(1000002, r.rawCells()[1].getTimestamp());
257      assertEquals(1000001, r.rawCells()[2].getTimestamp());
258      assertEquals(1000002, r.rawCells()[3].getTimestamp());
259      assertEquals(1500002, r.rawCells()[4].getTimestamp());
260      assertEquals(1000001, r.rawCells()[5].getTimestamp());
261    }
262  }

323  @Test
324  public void testRawScanAndMajorCompaction() throws IOException {
325    try (Table t = createTable()) {
326      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000001, value));
327      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000002, value));
328      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000003, value));
329      t.put(new Put(ROW).addColumn(FAMILY, col1, 1000004, value));
330
331      t.delete(new Delete(ROW).addColumn(FAMILY, col1, 1000004));
332      t.delete(new Delete(ROW).addColumn(FAMILY, col1, 1000003));
333
334      try (ResultScanner scannner = t.getScanner(new Scan().setRaw(true).setMaxVersions())) {
335        Result r = scannner.next();
336        assertNull(scannner.next());
337        assertEquals(6, r.size());
338      }
339      TEST_UTIL.getAdmin().flush(t.getName());
340      try (ResultScanner scannner = t.getScanner(new Scan().setRaw(true).setMaxVersions())) {
341        Result r = scannner.next();
342        assertNull(scannner.next());
343        assertEquals(6, r.size());
344      }
345      TEST_UTIL.getAdmin().majorCompact(t.getName());
346      Threads.sleep(5000);
347      try (ResultScanner scannner = t.getScanner(new Scan().setRaw(true).setMaxVersions())) {
348        Result r = scannner.next();
349        assertNull(scannner.next());
350        assertEquals(1, r.size());
351        assertEquals(1000002, r.rawCells()[0].getTimestamp());
352      }
353    }
