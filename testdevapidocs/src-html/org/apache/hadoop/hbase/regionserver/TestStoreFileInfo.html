019package org.apache.hadoop.hbase.regionserver;
020
021import static org.junit.Assert.*;
022
023import java.io.IOException;
024
025import org.apache.hadoop.fs.Path;
026import org.apache.hadoop.hbase.HBaseTestingUtility;
027import org.apache.hadoop.hbase.testclassification.RegionServerTests;
028import org.apache.hadoop.hbase.testclassification.SmallTests;
029import org.apache.hadoop.hbase.io.HFileLink;
030import org.junit.Test;
031import org.junit.experimental.categories.Category;
032
033/**
034 * Test HStoreFile
035 */
036@Category({RegionServerTests.class, SmallTests.class})
037public class TestStoreFileInfo {
038  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
039
040  /**
041   * Validate that we can handle valid tables with '.', '_', and '-' chars.
042   */
043  @Test
044  public void testStoreFileNames() {
045    String[] legalHFileLink = { "MyTable_02=abc012-def345", "MyTable_02.300=abc012-def345",
046      "MyTable_02-400=abc012-def345", "MyTable_02-400.200=abc012-def345",
047      "MyTable_02=abc012-def345_SeqId_1_", "MyTable_02=abc012-def345_SeqId_20_" };
048    for (String name: legalHFileLink) {
049      assertTrue("should be a valid link: " + name, HFileLink.isHFileLink(name));
050      assertTrue("should be a valid StoreFile" + name, StoreFileInfo.validateStoreFileName(name));
051      assertFalse("should not be a valid reference: " + name, StoreFileInfo.isReference(name));
052
053      String refName = name + ".6789";
054      assertTrue("should be a valid link reference: " + refName,
055          StoreFileInfo.isReference(refName));
056      assertTrue("should be a valid StoreFile" + refName,
057          StoreFileInfo.validateStoreFileName(refName));
058    }
059
060    String[] illegalHFileLink = { ".MyTable_02=abc012-def345", "-MyTable_02.300=abc012-def345",
061      "MyTable_02-400=abc0_12-def345", "MyTable_02-400.200=abc012-def345...." };
062    for (String name: illegalHFileLink) {
063      assertFalse("should not be a valid link: " + name, HFileLink.isHFileLink(name));
064    }
065  }
066
067  @Test
068  public void testEqualsWithLink() throws IOException {
069    Path origin = new Path("/origin");
070    Path tmp = new Path("/tmp");
071    Path mob = new Path("/mob");
072    Path archive = new Path("/archive");
073    HFileLink link1 = new HFileLink(new Path(origin, "f1"), new Path(tmp, "f1"),
074      new Path(mob, "f1"), new Path(archive, "f1"));
075    HFileLink link2 = new HFileLink(new Path(origin, "f1"), new Path(tmp, "f1"),
076      new Path(mob, "f1"), new Path(archive, "f1"));
077
078    StoreFileInfo info1 = new StoreFileInfo(TEST_UTIL.getConfiguration(),
079      TEST_UTIL.getTestFileSystem(), null, link1);
080    StoreFileInfo info2 = new StoreFileInfo(TEST_UTIL.getConfiguration(),
081      TEST_UTIL.getTestFileSystem(), null, link2);
082
083    assertEquals(info1, info2);
084    assertEquals(info1.hashCode(), info2.hashCode());
085  }
086}
