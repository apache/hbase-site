052/**
053 * Test many concurrent appenders to an WAL while rolling the log.
054 */
055@Category({RegionServerTests.class, MediumTests.class})
056public class TestLogRollingNoCluster {
057  @Rule public final TestRule timeout = CategoryBasedTimeout.builder().withTimeout(this.getClass()).
058      withLookingForStuckThread(true).build();
059  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
060  private final static byte [] EMPTY_1K_ARRAY = new byte[1024];
061  private static final int NUM_THREADS = 100; // Spin up this many threads
062  private static final int NUM_ENTRIES = 100; // How many entries to write
063
064  /** ProtobufLogWriter that simulates higher latencies in sync() call */
065  public static class HighLatencySyncWriter extends  ProtobufLogWriter {
066    @Override
067    public void sync() throws IOException {
068      Threads.sleep(ThreadLocalRandom.current().nextInt(10));
069      super.sync();
070      Threads.sleep(ThreadLocalRandom.current().nextInt(10));
071    }
072  }
073
074  /**
075   * Spin up a bunch of threads and have them all append to a WAL.  Roll the
076   * WAL frequently to try and trigger NPE.
077   * @throws IOException
078   * @throws InterruptedException
079   */
080  @Test
081  public void testContendedLogRolling() throws Exception {
082    TEST_UTIL.startMiniDFSCluster(3);
083    Path dir = TEST_UTIL.getDataTestDirOnTestFS();
084
085    // The implementation needs to know the 'handler' count.
086    TEST_UTIL.getConfiguration().setInt(HConstants.REGION_SERVER_HANDLER_COUNT, NUM_THREADS);
087    final Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
088    conf.set(WALFactory.WAL_PROVIDER, "filesystem");
089    FSUtils.setRootDir(conf, dir);
090    conf.set("hbase.regionserver.hlog.writer.impl", HighLatencySyncWriter.class.getName());
091    final WALFactory wals = new WALFactory(conf, null, TestLogRollingNoCluster.class.getName());
092    final WAL wal = wals.getWAL(new byte[]{}, null);
093    
094    Appender [] appenders = null;
095
096    final int numThreads = NUM_THREADS;
097    appenders = new Appender[numThreads];
098    try {
099      for (int i = 0; i < numThreads; i++) {
100        // Have each appending thread write 'count' entries
101        appenders[i] = new Appender(wal, i, NUM_ENTRIES);
102      }
103      for (int i = 0; i < numThreads; i++) {
104        appenders[i].start();
105      }
106      for (int i = 0; i < numThreads; i++) {
107        //ensure that all threads are joined before closing the wal
108        appenders[i].join();
109      }
110    } finally {
111      wals.close();
112    }
113    for (int i = 0; i < numThreads; i++) {
114      assertFalse(appenders[i].isException());
115    }
116    TEST_UTIL.shutdownMiniDFSCluster();
117  }
118
119  /**
120   * Appender thread.  Appends to passed wal file.
121   */
122  static class Appender extends Thread {
123    private final Logger log;
124    private final WAL wal;
125    private final int count;
126    private Exception e = null;
127
128    Appender(final WAL wal, final int index, final int count) {
129      super("" + index);
130      this.wal = wal;
131      this.count = count;
132      this.log = LoggerFactory.getLogger("Appender:" + getName());
133    }
134
135    /**
136     * @return Call when the thread is done.
137     */
138    boolean isException() {
139      return !isAlive() && this.e != null;
140    }
141
142    Exception getException() {
143      return this.e;
144    }
145
146    @Override
147    public void run() {
148      this.log.info(getName() +" started");
149      final MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
150      try {
151        for (int i = 0; i < this.count; i++) {
152          long now = System.currentTimeMillis();
153          // Roll every ten edits
154          if (i % 10 == 0) {
155            this.wal.rollWriter();
156          }
157          WALEdit edit = new WALEdit();
158          byte[] bytes = Bytes.toBytes(i);
159          edit.add(new KeyValue(bytes, bytes, bytes, now, EMPTY_1K_ARRAY));
160          final HRegionInfo hri = HRegionInfo.FIRST_META_REGIONINFO;
161          final HTableDescriptor htd = TEST_UTIL.getMetaTableDescriptor();
162          NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
163          for(byte[] fam : htd.getFamiliesKeys()) {
164            scopes.put(fam, 0);
165          }
166          final long txid = wal.append(hri, new WALKeyImpl(hri.getEncodedNameAsBytes(),
167              TableName.META_TABLE_NAME, now, mvcc, scopes), edit, true);
168          Threads.sleep(ThreadLocalRandom.current().nextInt(5));
169          wal.sync(txid);
170        }
171        String msg = getName() + " finished";
172        if (isException())
173          this.log.info(msg, getException());
174        else
175          this.log.info(msg);
176      } catch (Exception e) {
177        this.e = e;
178        log.info("Caught exception from Appender:" + getName(), e);
179      } finally {
180        // Call sync on our log.else threads just hang out.
181        try {
182          this.wal.sync();
183        } catch (IOException e) {
184          throw new RuntimeException(e);
185        }
186      }
187    }
188  }
