125  @Test
126  public void testBrokenWriter() throws Exception {
127    RegionServerServices services = mock(RegionServerServices.class);
128    when(services.getConfiguration()).thenReturn(CONF);
129    TableDescriptor td = TableDescriptorBuilder.newBuilder(TableName.valueOf("table"))
130        .setColumnFamily(ColumnFamilyDescriptorBuilder.of("row")).build();
131    RegionInfo ri = RegionInfoBuilder.newBuilder(td.getTableName()).build();
132    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
133    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
134    for (byte[] fam : td.getColumnFamilyNames()) {
135      scopes.put(fam, 0);
136    }
137    long timestamp = System.currentTimeMillis();
138    String testName = currentTest.getMethodName();
139    AtomicInteger failedCount = new AtomicInteger(0);
140    try (LogRoller roller = new LogRoller(services);
141        AsyncFSWAL wal = new AsyncFSWAL(FS, CommonFSUtils.getWALRootDir(CONF), DIR.toString(),
142            testName, CONF, null, true, null, null, GROUP, CHANNEL_CLASS) {
143
144          @Override
145          protected AsyncWriter createWriterInstance(Path path) throws IOException {
146            AsyncWriter writer = super.createWriterInstance(path);
147            return new AsyncWriter() {
148
149              @Override
150              public void close() throws IOException {
151                writer.close();
152              }
153
154              @Override
155              public long getLength() {
156                return writer.getLength();
157              }
158
159              @Override
160              public long getSyncedLength() {
161                return writer.getSyncedLength();
162              }
163
164              @Override
165              public CompletableFuture<Long> sync(boolean forceSync) {
166                CompletableFuture<Long> result = writer.sync(forceSync);
167                if (failedCount.incrementAndGet() < 1000) {
168                  CompletableFuture<Long> future = new CompletableFuture<>();
169                  FutureUtils.addListener(result,
170                    (r, e) -> future.completeExceptionally(new IOException("Inject Error")));
171                  return future;
172                } else {
173                  return result;
174                }
175              }
176
177              @Override
178              public void append(Entry entry) {
179                writer.append(entry);
180              }
181            };
182          }
183        }) {
184      wal.init();
185      roller.addWAL(wal);
186      roller.start();
187      int numThreads = 10;
188      AtomicReference<Exception> error = new AtomicReference<>();
189      Thread[] threads = new Thread[numThreads];
190      for (int i = 0; i < 10; i++) {
191        final int index = i;
192        threads[index] = new Thread("Write-Thread-" + index) {
193
194          @Override
195          public void run() {
196            byte[] row = Bytes.toBytes("row" + index);
197            WALEdit cols = new WALEdit();
198            cols.add(new KeyValue(row, row, row, timestamp + index, row));
199            WALKeyImpl key = new WALKeyImpl(ri.getEncodedNameAsBytes(), td.getTableName(),
200                SequenceId.NO_SEQUENCE_ID, timestamp, WALKey.EMPTY_UUIDS, HConstants.NO_NONCE,
201                HConstants.NO_NONCE, mvcc, scopes);
202            try {
203              wal.append(ri, key, cols, true);
204            } catch (IOException e) {
205              // should not happen
206              throw new UncheckedIOException(e);
207            }
208            try {
209              wal.sync();
210            } catch (IOException e) {
211              error.set(e);
212            }
213          }
214        };
215      }
216      for (Thread t : threads) {
217        t.start();
218      }
219      for (Thread t : threads) {
220        t.join();
221      }
222      assertNull(error.get());
223    }
224  }
