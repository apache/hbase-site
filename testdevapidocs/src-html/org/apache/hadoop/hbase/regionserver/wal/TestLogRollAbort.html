059/**
060 * Tests for conditions that should trigger RegionServer aborts when
061 * rolling the current WAL fails.
062 */
063@Category({RegionServerTests.class, MediumTests.class})
064public class TestLogRollAbort {
065  private static final Log LOG = LogFactory.getLog(TestLogRolling.class);
066  private static MiniDFSCluster dfsCluster;
067  private static Admin admin;
068  private static MiniHBaseCluster cluster;
069  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
070
071  /* For the split-then-roll test */
072  private static final Path HBASEDIR = new Path("/hbase");
073  private static final Path OLDLOGDIR = new Path(HBASEDIR, HConstants.HREGION_OLDLOGDIR_NAME);
074
075  // Need to override this setup so we can edit the config before it gets sent
076  // to the HDFS & HBase cluster startup.
077  @BeforeClass
078  public static void setUpBeforeClass() throws Exception {
079    // Tweak default timeout values down for faster recovery
080    TEST_UTIL.getConfiguration().setInt(
081        "hbase.regionserver.logroll.errors.tolerated", 2);
082    TEST_UTIL.getConfiguration().setInt("hbase.rpc.timeout", 10 * 1000);
083
084    // Increase the amount of time between client retries
085    TEST_UTIL.getConfiguration().setLong("hbase.client.pause", 5 * 1000);
086
087    // make sure log.hflush() calls syncFs() to open a pipeline
088    TEST_UTIL.getConfiguration().setBoolean("dfs.support.append", true);
089    // lower the namenode & datanode heartbeat so the namenode
090    // quickly detects datanode failures
091    TEST_UTIL.getConfiguration().setInt("dfs.namenode.heartbeat.recheck-interval", 5000);
092    TEST_UTIL.getConfiguration().setInt("dfs.heartbeat.interval", 1);
093    // the namenode might still try to choose the recently-dead datanode
094    // for a pipeline, so try to a new pipeline multiple times
095    TEST_UTIL.getConfiguration().setInt("dfs.client.block.write.retries", 10);
096  }
097
098  private Configuration conf;
099  private FileSystem fs;
100
101  @Before
102  public void setUp() throws Exception {
103    TEST_UTIL.startMiniCluster(2);
104
105    cluster = TEST_UTIL.getHBaseCluster();
106    dfsCluster = TEST_UTIL.getDFSCluster();
107    admin = TEST_UTIL.getHBaseAdmin();
108    conf = TEST_UTIL.getConfiguration();
109    fs = TEST_UTIL.getDFSCluster().getFileSystem();
110
111    // disable region rebalancing (interferes with log watching)
112    cluster.getMaster().balanceSwitch(false);
113    FSUtils.setRootDir(conf, HBASEDIR);
114  }

116  @After
117  public void tearDown() throws Exception {
118    TEST_UTIL.shutdownMiniCluster();
119  }

121  /**
122   * Tests that RegionServer aborts if we hit an error closing the WAL when
123   * there are unsynced WAL edits.  See HBASE-4282.
124   */
125  @Test
126  public void testRSAbortWithUnflushedEdits() throws Exception {
127    LOG.info("Starting testRSAbortWithUnflushedEdits()");
128
129    // When the hbase:meta table can be opened, the region servers are running
130    TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME).close();
131
132    // Create the test table and open it
133    TableName tableName = TableName.valueOf(this.getClass().getSimpleName());
134    HTableDescriptor desc = new HTableDescriptor(tableName);
135    desc.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
136
137    admin.createTable(desc);
138    Table table = TEST_UTIL.getConnection().getTable(desc.getTableName());
139    try {
140      HRegionServer server = TEST_UTIL.getRSForFirstRegionInTable(tableName);
141      WAL log = server.getWAL(null);
142
143      // don't run this test without append support (HDFS-200 & HDFS-142)
144      assertTrue("Need append support for this test",
145        FSUtils.isAppendSupported(TEST_UTIL.getConfiguration()));
146
147      Put p = new Put(Bytes.toBytes("row2001"));
148      p.addColumn(HConstants.CATALOG_FAMILY, Bytes.toBytes("col"), Bytes.toBytes(2001));
149      table.put(p);
150
151      log.sync();
152
153      p = new Put(Bytes.toBytes("row2002"));
154      p.addColumn(HConstants.CATALOG_FAMILY, Bytes.toBytes("col"), Bytes.toBytes(2002));
155      table.put(p);
156
157      dfsCluster.restartDataNodes();
158      LOG.info("Restarted datanodes");
159
160      try {
161        log.rollWriter(true);
162      } catch (FailedLogCloseException flce) {
163        // Expected exception.  We used to expect that there would be unsynced appends but this
164        // not reliable now that sync plays a roll in wall rolling.  The above puts also now call
165        // sync.
166      } catch (Throwable t) {
167        LOG.fatal("FAILED TEST: Got wrong exception", t);
168      }
169    } finally {
170      table.close();
171    }
172  }

174  /**
175   * Tests the case where a RegionServer enters a GC pause,
176   * comes back online after the master declared it dead and started to split.
177   * Want log rolling after a master split to fail. See HBASE-2312.
178   */
179  @Test (timeout=300000)
180  public void testLogRollAfterSplitStart() throws IOException {
181    LOG.info("Verify wal roll after split starts will fail.");
182    String logName = "testLogRollAfterSplitStart";
183    Path thisTestsDir = new Path(HBASEDIR, DefaultWALProvider.getWALDirectoryName(logName));
184    final WALFactory wals = new WALFactory(conf, null, logName);
185
186    try {
187      // put some entries in an WAL
188      TableName tableName =
189          TableName.valueOf(this.getClass().getName());
190      HRegionInfo regioninfo = new HRegionInfo(tableName,
191          HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
192      final WAL log = wals.getWAL(regioninfo.getEncodedNameAsBytes(),
193          regioninfo.getTable().getNamespace());
194      MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl(1);
195
196      final int total = 20;
197      for (int i = 0; i < total; i++) {
198        WALEdit kvs = new WALEdit();
199        kvs.add(new KeyValue(Bytes.toBytes(i), tableName.getName(), tableName.getName()));
200        HTableDescriptor htd = new HTableDescriptor(tableName);
201        htd.addFamily(new HColumnDescriptor("column"));
202        log.append(htd, regioninfo, new WALKey(regioninfo.getEncodedNameAsBytes(), tableName,
203            System.currentTimeMillis(), mvcc), kvs, true);
204      }
205      // Send the data to HDFS datanodes and close the HDFS writer
206      log.sync();
207      ((FSHLog) log).replaceWriter(((FSHLog)log).getOldPath(), null, null, null);
208
209      /* code taken from MasterFileSystem.getLogDirs(), which is called from MasterFileSystem.splitLog()
210       * handles RS shutdowns (as observed by the splitting process)
211       */
212      // rename the directory so a rogue RS doesn't create more WALs
213      Path rsSplitDir = thisTestsDir.suffix(DefaultWALProvider.SPLITTING_EXT);
214      if (!fs.rename(thisTestsDir, rsSplitDir)) {
215        throw new IOException("Failed fs.rename for log split: " + thisTestsDir);
216      }
217      LOG.debug("Renamed region directory: " + rsSplitDir);
218
219      LOG.debug("Processing the old log files.");
220      WALSplitter.split(HBASEDIR, rsSplitDir, OLDLOGDIR, fs, conf, wals);
221
222      LOG.debug("Trying to roll the WAL.");
223      try {
224        log.rollWriter();
225        Assert.fail("rollWriter() did not throw any exception.");
226      } catch (IOException ioe) {
227        if (ioe.getCause() instanceof FileNotFoundException) {
228          LOG.info("Got the expected exception: ", ioe.getCause());
229        } else {
230          Assert.fail("Unexpected exception: " + ioe);
231        }
232      }
233    } finally {
234      wals.close();
235      if (fs.exists(thisTestsDir)) {
236        fs.delete(thisTestsDir, true);
237      }
238    }
239  }
