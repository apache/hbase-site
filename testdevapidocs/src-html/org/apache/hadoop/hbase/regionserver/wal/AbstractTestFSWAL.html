154  protected void addEdits(WAL log, RegionInfo hri, HTableDescriptor htd, int times,
155      MultiVersionConcurrencyControl mvcc, NavigableMap<byte[], Integer> scopes)
156      throws IOException {
157    final byte[] row = Bytes.toBytes("row");
158    for (int i = 0; i < times; i++) {
159      long timestamp = System.currentTimeMillis();
160      WALEdit cols = new WALEdit();
161      cols.add(new KeyValue(row, row, row, timestamp, row));
162      WALKey key = new WALKey(hri.getEncodedNameAsBytes(), htd.getTableName(),
163          WALKey.NO_SEQUENCE_ID, timestamp, WALKey.EMPTY_UUIDS, HConstants.NO_NONCE,
164          HConstants.NO_NONCE, mvcc, scopes);
165      log.append(hri, key, cols, true);
166    }
167    log.sync();
168  }

171   * helper method to simulate region flush for a WAL.
172   * @param wal
173   * @param regionEncodedName

175  protected void flushRegion(WAL wal, byte[] regionEncodedName, Set<byte[]> flushedFamilyNames) {
176    wal.startCacheFlush(regionEncodedName, flushedFamilyNames);
177    wal.completeCacheFlush(regionEncodedName);

180  /**
181   * tests the log comparator. Ensure that we are not mixing meta logs with non-meta logs (throws
182   * exception if we do). Comparison is based on the timestamp present in the wal name.
183   * @throws Exception
184   */
185  @Test
186  public void testWALComparator() throws Exception {
187    AbstractFSWAL<?> wal1 = null;
188    AbstractFSWAL<?> walMeta = null;
189    try {
190      wal1 = newWAL(FS, FSUtils.getWALRootDir(CONF), DIR.toString(), HConstants.HREGION_OLDLOGDIR_NAME,
191        CONF, null, true, null, null);
192      LOG.debug("Log obtained is: " + wal1);
193      Comparator<Path> comp = wal1.LOG_NAME_COMPARATOR;
194      Path p1 = wal1.computeFilename(11);
195      Path p2 = wal1.computeFilename(12);
196      // comparing with itself returns 0
197      assertTrue(comp.compare(p1, p1) == 0);
198      // comparing with different filenum.
199      assertTrue(comp.compare(p1, p2) < 0);
200      walMeta =
201          newWAL(FS, FSUtils.getWALRootDir(CONF), DIR.toString(), HConstants.HREGION_OLDLOGDIR_NAME,
202            CONF, null, true, null, AbstractFSWALProvider.META_WAL_PROVIDER_ID);
203      Comparator<Path> compMeta = walMeta.LOG_NAME_COMPARATOR;
204
205      Path p1WithMeta = walMeta.computeFilename(11);
206      Path p2WithMeta = walMeta.computeFilename(12);
207      assertTrue(compMeta.compare(p1WithMeta, p1WithMeta) == 0);
208      assertTrue(compMeta.compare(p1WithMeta, p2WithMeta) < 0);
209      // mixing meta and non-meta logs gives error
210      boolean ex = false;
211      try {
212        comp.compare(p1WithMeta, p2);
213      } catch (IllegalArgumentException e) {
214        ex = true;
215      }
216      assertTrue("Comparator doesn't complain while checking meta log files", ex);
217      boolean exMeta = false;
218      try {
219        compMeta.compare(p1WithMeta, p2);
220      } catch (IllegalArgumentException e) {
221        exMeta = true;
222      }
223      assertTrue("Meta comparator doesn't complain while checking log files", exMeta);
224    } finally {
225      if (wal1 != null) {
226        wal1.close();
227      }
228      if (walMeta != null) {
229        walMeta.close();
230      }
231    }

234  /**
235   * On rolling a wal after reaching the threshold, {@link WAL#rollWriter()} returns the list of
236   * regions which should be flushed in order to archive the oldest wal file.
237   * <p>
238   * This method tests this behavior by inserting edits and rolling the wal enough times to reach
239   * the max number of logs threshold. It checks whether we get the "right regions" for flush on
240   * rolling the wal.
241   * @throws Exception
242   */
243  @Test
244  public void testFindMemStoresEligibleForFlush() throws Exception {
245    LOG.debug("testFindMemStoresEligibleForFlush");
246    Configuration conf1 = HBaseConfiguration.create(CONF);
247    conf1.setInt("hbase.regionserver.maxlogs", 1);
248    AbstractFSWAL<?> wal = newWAL(FS, FSUtils.getWALRootDir(conf1), DIR.toString(),
249      HConstants.HREGION_OLDLOGDIR_NAME, conf1, null, true, null, null);
250    HTableDescriptor t1 =
251        new HTableDescriptor(TableName.valueOf("t1")).addFamily(new HColumnDescriptor("row"));
252    HTableDescriptor t2 =
253        new HTableDescriptor(TableName.valueOf("t2")).addFamily(new HColumnDescriptor("row"));
254    RegionInfo hri1 = RegionInfoBuilder.newBuilder(t1.getTableName())
255        .setStartKey(HConstants.EMPTY_START_ROW)
256        .setEndKey(HConstants.EMPTY_END_ROW)
257        .build();
258    RegionInfo hri2 = RegionInfoBuilder.newBuilder(t2.getTableName())
259        .setStartKey(HConstants.EMPTY_START_ROW)
260        .setEndKey(HConstants.EMPTY_END_ROW)
261        .build();
262    // add edits and roll the wal
263    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
264    NavigableMap<byte[], Integer> scopes1 = new TreeMap<>(Bytes.BYTES_COMPARATOR);
265    for (byte[] fam : t1.getFamiliesKeys()) {
266      scopes1.put(fam, 0);
267    }
268    NavigableMap<byte[], Integer> scopes2 = new TreeMap<>(Bytes.BYTES_COMPARATOR);
269    for (byte[] fam : t2.getFamiliesKeys()) {
270      scopes2.put(fam, 0);
271    }
272    try {
273      addEdits(wal, hri1, t1, 2, mvcc, scopes1);
274      wal.rollWriter();
275      // add some more edits and roll the wal. This would reach the log number threshold
276      addEdits(wal, hri1, t1, 2, mvcc, scopes1);
277      wal.rollWriter();
278      // with above rollWriter call, the max logs limit is reached.
279      assertTrue(wal.getNumRolledLogFiles() == 2);
280
281      // get the regions to flush; since there is only one region in the oldest wal, it should
282      // return only one region.
283      byte[][] regionsToFlush = wal.findRegionsToForceFlush();
284      assertEquals(1, regionsToFlush.length);
285      assertEquals(hri1.getEncodedNameAsBytes(), regionsToFlush[0]);
286      // insert edits in second region
287      addEdits(wal, hri2, t2, 2, mvcc, scopes2);
288      // get the regions to flush, it should still read region1.
289      regionsToFlush = wal.findRegionsToForceFlush();
290      assertEquals(regionsToFlush.length, 1);
291      assertEquals(hri1.getEncodedNameAsBytes(), regionsToFlush[0]);
292      // flush region 1, and roll the wal file. Only last wal which has entries for region1 should
293      // remain.
294      flushRegion(wal, hri1.getEncodedNameAsBytes(), t1.getFamiliesKeys());
295      wal.rollWriter();
296      // only one wal should remain now (that is for the second region).
297      assertEquals(1, wal.getNumRolledLogFiles());
298      // flush the second region
299      flushRegion(wal, hri2.getEncodedNameAsBytes(), t2.getFamiliesKeys());
300      wal.rollWriter(true);
301      // no wal should remain now.
302      assertEquals(0, wal.getNumRolledLogFiles());
303      // add edits both to region 1 and region 2, and roll.
304      addEdits(wal, hri1, t1, 2, mvcc, scopes1);
305      addEdits(wal, hri2, t2, 2, mvcc, scopes2);
306      wal.rollWriter();
307      // add edits and roll the writer, to reach the max logs limit.
308      assertEquals(1, wal.getNumRolledLogFiles());
309      addEdits(wal, hri1, t1, 2, mvcc, scopes1);
310      wal.rollWriter();
311      // it should return two regions to flush, as the oldest wal file has entries
312      // for both regions.
313      regionsToFlush = wal.findRegionsToForceFlush();
314      assertEquals(2, regionsToFlush.length);
315      // flush both regions
316      flushRegion(wal, hri1.getEncodedNameAsBytes(), t1.getFamiliesKeys());
317      flushRegion(wal, hri2.getEncodedNameAsBytes(), t2.getFamiliesKeys());
318      wal.rollWriter(true);
319      assertEquals(0, wal.getNumRolledLogFiles());
320      // Add an edit to region1, and roll the wal.
321      addEdits(wal, hri1, t1, 2, mvcc, scopes1);
322      // tests partial flush: roll on a partial flush, and ensure that wal is not archived.
323      wal.startCacheFlush(hri1.getEncodedNameAsBytes(), t1.getFamiliesKeys());
324      wal.rollWriter();
325      wal.completeCacheFlush(hri1.getEncodedNameAsBytes());
326      assertEquals(1, wal.getNumRolledLogFiles());
327    } finally {
328      if (wal != null) {
329        wal.close();
330      }
331    }
332  }

350  /**
351   * Test flush for sure has a sequence id that is beyond the last edit appended. We do this by
352   * slowing appends in the background ring buffer thread while in foreground we call flush. The
353   * addition of the sync over HRegion in flush should fix an issue where flush was returning before
354   * all of its appends had made it out to the WAL (HBASE-11109).
355   * @throws IOException
356   * @see <a href="https://issues.apache.org/jira/browse/HBASE-11109">HBASE-11109</a>
357   */
358  @Test
359  public void testFlushSequenceIdIsGreaterThanAllEditsInHFile() throws IOException {
360    String testName = currentTest.getMethodName();
361    final TableName tableName = TableName.valueOf(testName);
362    final RegionInfo hri = RegionInfoBuilder.newBuilder(tableName).build();
363    final byte[] rowName = tableName.getName();
364    final HTableDescriptor htd = new HTableDescriptor(tableName);
365    htd.addFamily(new HColumnDescriptor("f"));
366    HRegion r = HBaseTestingUtility.createRegionAndWAL(hri, TEST_UTIL.getDefaultRootDirPath(),
367      TEST_UTIL.getConfiguration(), htd);
368    HBaseTestingUtility.closeRegionAndWAL(r);
369    final int countPerFamily = 10;
370    final AtomicBoolean goslow = new AtomicBoolean(false);
371    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
372    for (byte[] fam : htd.getFamiliesKeys()) {
373      scopes.put(fam, 0);
374    }
375    // subclass and doctor a method.
376    AbstractFSWAL<?> wal = newSlowWAL(FS, FSUtils.getWALRootDir(CONF), DIR.toString(), testName, CONF,
377      null, true, null, null, new Runnable() {
378
379        @Override
380        public void run() {
381          if (goslow.get()) {
382            Threads.sleep(100);
383            LOG.debug("Sleeping before appending 100ms");
384          }
385        }
386      });
387    HRegion region = HRegion.openHRegion(TEST_UTIL.getConfiguration(),
388      TEST_UTIL.getTestFileSystem(), TEST_UTIL.getDefaultRootDirPath(), hri, htd, wal);
389    EnvironmentEdge ee = EnvironmentEdgeManager.getDelegate();
390    try {
391      List<Put> puts = null;
392      for (HColumnDescriptor hcd : htd.getFamilies()) {
393        puts =
394            TestWALReplay.addRegionEdits(rowName, hcd.getName(), countPerFamily, ee, region, "x");
395      }
396
397      // Now assert edits made it in.
398      final Get g = new Get(rowName);
399      Result result = region.get(g);
400      assertEquals(countPerFamily * htd.getFamilies().size(), result.size());
401
402      // Construct a WALEdit and add it a few times to the WAL.
403      WALEdit edits = new WALEdit();
404      for (Put p : puts) {
405        CellScanner cs = p.cellScanner();
406        while (cs.advance()) {
407          edits.add(cs.current());
408        }
409      }
410      // Add any old cluster id.
411      List<UUID> clusterIds = new ArrayList<>(1);
412      clusterIds.add(UUID.randomUUID());
413      // Now make appends run slow.
414      goslow.set(true);
415      for (int i = 0; i < countPerFamily; i++) {
416        final RegionInfo info = region.getRegionInfo();
417        final WALKey logkey = new WALKey(info.getEncodedNameAsBytes(), tableName,
418            System.currentTimeMillis(), clusterIds, -1, -1, region.getMVCC(), scopes);
419        wal.append(info, logkey, edits, true);
420        region.getMVCC().completeAndWait(logkey.getWriteEntry());
421      }
422      region.flush(true);
423      // FlushResult.flushSequenceId is not visible here so go get the current sequence id.
424      long currentSequenceId = region.getReadPoint(null);
425      // Now release the appends
426      goslow.set(false);
427      assertTrue(currentSequenceId >= region.getReadPoint(null));
428    } finally {
429      region.close(true);
430      wal.close();
431    }
432  }
