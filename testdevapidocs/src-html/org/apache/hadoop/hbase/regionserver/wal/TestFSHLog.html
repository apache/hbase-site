095  @Test
096  public void testSyncRunnerIndexOverflow() throws IOException, NoSuchFieldException,
097      SecurityException, IllegalArgumentException, IllegalAccessException {
098    final String name = this.name.getMethodName();
099    FSHLog log = new FSHLog(FS, FSUtils.getRootDir(CONF), name, HConstants.HREGION_OLDLOGDIR_NAME,
100        CONF, null, true, null, null);
101    try {
102      Field ringBufferEventHandlerField = FSHLog.class.getDeclaredField("ringBufferEventHandler");
103      ringBufferEventHandlerField.setAccessible(true);
104      FSHLog.RingBufferEventHandler ringBufferEventHandler =
105          (FSHLog.RingBufferEventHandler) ringBufferEventHandlerField.get(log);
106      Field syncRunnerIndexField =
107          FSHLog.RingBufferEventHandler.class.getDeclaredField("syncRunnerIndex");
108      syncRunnerIndexField.setAccessible(true);
109      syncRunnerIndexField.set(ringBufferEventHandler, Integer.MAX_VALUE - 1);
110      HTableDescriptor htd =
111          new HTableDescriptor(TableName.valueOf(this.name.getMethodName())).addFamily(new HColumnDescriptor("row"));
112      NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
113      for (byte[] fam : htd.getFamiliesKeys()) {
114        scopes.put(fam, 0);
115      }
116      HRegionInfo hri =
117          new HRegionInfo(htd.getTableName(), HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
118      MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
119      for (int i = 0; i < 10; i++) {
120        addEdits(log, hri, htd, 1, mvcc, scopes);
121      }
122    } finally {
123      log.close();
124    }
125  }

127  /**
128   * Test case for https://issues.apache.org/jira/browse/HBASE-16721
129   */
130  @Test (timeout = 30000)
131  public void testUnflushedSeqIdTracking() throws IOException, InterruptedException {
132    final String name = this.name.getMethodName();
133    final byte[] b = Bytes.toBytes("b");
134
135    final AtomicBoolean startHoldingForAppend = new AtomicBoolean(false);
136    final CountDownLatch holdAppend = new CountDownLatch(1);
137    final CountDownLatch flushFinished = new CountDownLatch(1);
138    final CountDownLatch putFinished = new CountDownLatch(1);
139
140    try (FSHLog log =
141        new FSHLog(FS, FSUtils.getRootDir(CONF), name, HConstants.HREGION_OLDLOGDIR_NAME, CONF,
142            null, true, null, null)) {
143
144      log.registerWALActionsListener(new WALActionsListener() {
145        @Override
146        public void visitLogEntryBeforeWrite(WALKey logKey, WALEdit logEdit)
147            throws IOException {
148          if (startHoldingForAppend.get()) {
149            try {
150              holdAppend.await();
151            } catch (InterruptedException e) {
152              LOG.error(e.toString(), e);
153            }
154          }
155        }
156      });
157
158      // open a new region which uses this WAL
159      HTableDescriptor htd =
160          new HTableDescriptor(TableName.valueOf(this.name.getMethodName())).addFamily(new HColumnDescriptor(b));
161      HRegionInfo hri =
162          new HRegionInfo(htd.getTableName(), HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
163      ChunkCreator.initialize(MemStoreLABImpl.CHUNK_SIZE_DEFAULT, false, 0, 0, 0, null);
164      final HRegion region = TEST_UTIL.createLocalHRegion(hri, htd, log);
165      ExecutorService exec = Executors.newFixedThreadPool(2);
166
167      // do a regular write first because of memstore size calculation.
168      region.put(new Put(b).addColumn(b, b,b));
169
170      startHoldingForAppend.set(true);
171      exec.submit(new Runnable() {
172        @Override
173        public void run() {
174          try {
175            region.put(new Put(b).addColumn(b, b,b));
176            putFinished.countDown();
177          } catch (IOException e) {
178            LOG.error(e.toString(), e);
179          }
180        }
181      });
182
183      // give the put a chance to start
184      Threads.sleep(3000);
185
186      exec.submit(new Runnable() {
187        @Override
188        public void run() {
189          try {
190            HRegion.FlushResult flushResult = region.flush(true);
191            LOG.info("Flush result:" +  flushResult.getResult());
192            LOG.info("Flush succeeded:" +  flushResult.isFlushSucceeded());
193            flushFinished.countDown();
194          } catch (IOException e) {
195            LOG.error(e.toString(), e);
196          }
197        }
198      });
199
200      // give the flush a chance to start. Flush should have got the region lock, and
201      // should have been waiting on the mvcc complete after this.
202      Threads.sleep(3000);
203
204      // let the append to WAL go through now that the flush already started
205      holdAppend.countDown();
206      putFinished.await();
207      flushFinished.await();
208
209      // check whether flush went through
210      assertEquals("Region did not flush?", 1, region.getStoreFileList(new byte[][]{b}).size());
211
212      // now check the region's unflushed seqIds.
213      long seqId = log.getEarliestMemStoreSeqNum(hri.getEncodedNameAsBytes());
214      assertEquals("Found seqId for the region which is already flushed",
215          HConstants.NO_SEQNUM, seqId);
216
217      region.close();
218    }
219  }
