182  @Test
183  public void testReplayEditsAfterRegionMovedWithMultiCF() throws Exception {
184    final TableName tableName =
185        TableName.valueOf("testReplayEditsAfterRegionMovedWithMultiCF");
186    byte[] family1 = Bytes.toBytes("cf1");
187    byte[] family2 = Bytes.toBytes("cf2");
188    byte[] qualifier = Bytes.toBytes("q");
189    byte[] value = Bytes.toBytes("testV");
190    byte[][] familys = { family1, family2 };
191    TEST_UTIL.createTable(tableName, familys);
192    Table htable = TEST_UTIL.getConnection().getTable(tableName);
193    Put put = new Put(Bytes.toBytes("r1"));
194    put.addColumn(family1, qualifier, value);
195    htable.put(put);
196    ResultScanner resultScanner = htable.getScanner(new Scan());
197    int count = 0;
198    while (resultScanner.next() != null) {
199      count++;
200    }
201    resultScanner.close();
202    assertEquals(1, count);
203
204    MiniHBaseCluster hbaseCluster = TEST_UTIL.getMiniHBaseCluster();
205    List<HRegion> regions = hbaseCluster.getRegions(tableName);
206    assertEquals(1, regions.size());
207
208    // move region to another regionserver
209    Region destRegion = regions.get(0);
210    int originServerNum = hbaseCluster.getServerWith(destRegion.getRegionInfo().getRegionName());
211    assertTrue("Please start more than 1 regionserver",
212        hbaseCluster.getRegionServerThreads().size() > 1);
213    int destServerNum = 0;
214    while (destServerNum == originServerNum) {
215      destServerNum++;
216    }
217    HRegionServer originServer = hbaseCluster.getRegionServer(originServerNum);
218    HRegionServer destServer = hbaseCluster.getRegionServer(destServerNum);
219    // move region to destination regionserver
220    TEST_UTIL.moveRegionAndWait(destRegion.getRegionInfo(), destServer.getServerName());
221
222    // delete the row
223    Delete del = new Delete(Bytes.toBytes("r1"));
224    htable.delete(del);
225    resultScanner = htable.getScanner(new Scan());
226    count = 0;
227    while (resultScanner.next() != null) {
228      count++;
229    }
230    resultScanner.close();
231    assertEquals(0, count);
232
233    // flush region and make major compaction
234    Region region =  destServer.getOnlineRegion(destRegion.getRegionInfo().getRegionName());
235    region.flush(true);
236    // wait to complete major compaction
237    for (Store store : region.getStores()) {
238      store.triggerMajorCompaction();
239    }
240    region.compact(true);
241
242    // move region to origin regionserver
243    TEST_UTIL.moveRegionAndWait(destRegion.getRegionInfo(), originServer.getServerName());
244    // abort the origin regionserver
245    originServer.abort("testing");
246
247    // see what we get
248    Result result = htable.get(new Get(Bytes.toBytes("r1")));
249    if (result != null) {
250      assertTrue("Row is deleted, but we get" + result.toString(),
251          (result == null) || result.isEmpty());
252    }
253    resultScanner.close();
254  }

456  /**
457   * Test writing edits into an HRegion, closing it, splitting logs, opening
458   * Region again.  Verify seqids.
459   * @throws IOException
460   * @throws IllegalAccessException
461   * @throws NoSuchFieldException
462   * @throws IllegalArgumentException
463   * @throws SecurityException
464   */
465  @Test
466  public void testReplayEditsWrittenViaHRegion()
467  throws IOException, SecurityException, IllegalArgumentException,
468      NoSuchFieldException, IllegalAccessException, InterruptedException {
469    final TableName tableName =
470        TableName.valueOf("testReplayEditsWrittenViaHRegion");
471    final HRegionInfo hri = createBasic3FamilyHRegionInfo(tableName);
472    final Path basedir = FSUtils.getTableDir(this.hbaseRootDir, tableName);
473    deleteDir(basedir);
474    final byte[] rowName = tableName.getName();
475    final int countPerFamily = 10;
476    final HTableDescriptor htd = createBasic3FamilyHTD(tableName);
477    HRegion region3 = HBaseTestingUtility.createRegionAndWAL(hri, hbaseRootDir, this.conf, htd);
478    HBaseTestingUtility.closeRegionAndWAL(region3);
479    // Write countPerFamily edits into the three families.  Do a flush on one
480    // of the families during the load of edits so its seqid is not same as
481    // others to test we do right thing when different seqids.
482    WAL wal = createWAL(this.conf, hbaseRootDir, logName);
483    HRegion region = HRegion.openHRegion(this.conf, this.fs, hbaseRootDir, hri, htd, wal);
484    long seqid = region.getOpenSeqNum();
485    boolean first = true;
486    for (HColumnDescriptor hcd: htd.getFamilies()) {
487      addRegionEdits(rowName, hcd.getName(), countPerFamily, this.ee, region, "x");
488      if (first) {
489        // If first, so we have at least one family w/ different seqid to rest.
490        region.flush(true);
491        first = false;
492      }
493    }
494    // Now assert edits made it in.
495    final Get g = new Get(rowName);
496    Result result = region.get(g);
497    assertEquals(countPerFamily * htd.getFamilies().size(),
498      result.size());
499    // Now close the region (without flush), split the log, reopen the region and assert that
500    // replay of log has the correct effect, that our seqids are calculated correctly so
501    // all edits in logs are seen as 'stale'/old.
502    region.close(true);
503    wal.shutdown();
504    runWALSplit(this.conf);
505    WAL wal2 = createWAL(this.conf, hbaseRootDir, logName);
506    HRegion region2 = HRegion.openHRegion(conf, this.fs, hbaseRootDir, hri, htd, wal2);
507    long seqid2 = region2.getOpenSeqNum();
508    assertTrue(seqid + result.size() < seqid2);
509    final Result result1b = region2.get(g);
510    assertEquals(result.size(), result1b.size());
511
512    // Next test.  Add more edits, then 'crash' this region by stealing its wal
513    // out from under it and assert that replay of the log adds the edits back
514    // correctly when region is opened again.
515    for (HColumnDescriptor hcd: htd.getFamilies()) {
516      addRegionEdits(rowName, hcd.getName(), countPerFamily, this.ee, region2, "y");
517    }
518    // Get count of edits.
519    final Result result2 = region2.get(g);
520    assertEquals(2 * result.size(), result2.size());
521    wal2.sync();
522    final Configuration newConf = HBaseConfiguration.create(this.conf);
523    User user = HBaseTestingUtility.getDifferentUser(newConf,
524      tableName.getNameAsString());
525    user.runAs(new PrivilegedExceptionAction<Object>() {
526      @Override
527      public Object run() throws Exception {
528        runWALSplit(newConf);
529        FileSystem newFS = FileSystem.get(newConf);
530        // Make a new wal for new region open.
531        WAL wal3 = createWAL(newConf, hbaseRootDir, logName);
532        final AtomicInteger countOfRestoredEdits = new AtomicInteger(0);
533        HRegion region3 = new HRegion(basedir, wal3, newFS, newConf, hri, htd, null) {
534          @Override
535          protected void restoreEdit(HStore s, Cell cell, MemstoreSize memstoreSize) {
536            super.restoreEdit(s, cell, memstoreSize);
537            countOfRestoredEdits.incrementAndGet();
538          }
539        };
540        long seqid3 = region3.initialize();
541        Result result3 = region3.get(g);
542        // Assert that count of cells is same as before crash.
543        assertEquals(result2.size(), result3.size());
544        assertEquals(htd.getFamilies().size() * countPerFamily,
545          countOfRestoredEdits.get());
546
547        // I can't close wal1.  Its been appropriated when we split.
548        region3.close();
549        wal3.close();
550        return null;
551      }
552    });
553  }

573  @Test

654   * Test that we could recover the data correctly after aborting flush. In the
655   * test, first we abort flush after writing some data, then writing more data
656   * and flush again, at last verify the data.
657   * @throws IOException
658   */
659  @Test
660  public void testReplayEditsAfterAbortingFlush() throws IOException {
661    final TableName tableName =
662        TableName.valueOf("testReplayEditsAfterAbortingFlush");
663    final HRegionInfo hri = createBasic3FamilyHRegionInfo(tableName);
664    final Path basedir = FSUtils.getTableDir(this.hbaseRootDir, tableName);
665    deleteDir(basedir);
666    final HTableDescriptor htd = createBasic3FamilyHTD(tableName);
667    HRegion region3 = HBaseTestingUtility.createRegionAndWAL(hri, hbaseRootDir, this.conf, htd);
668    HBaseTestingUtility.closeRegionAndWAL(region3);
669    // Write countPerFamily edits into the three families. Do a flush on one
670    // of the families during the load of edits so its seqid is not same as
671    // others to test we do right thing when different seqids.
672    WAL wal = createWAL(this.conf, hbaseRootDir, logName);
673    RegionServerServices rsServices = Mockito.mock(RegionServerServices.class);
674    Mockito.doReturn(false).when(rsServices).isAborted();
675    when(rsServices.getServerName()).thenReturn(ServerName.valueOf("foo", 10, 10));
676    Configuration customConf = new Configuration(this.conf);
677    customConf.set(DefaultStoreEngine.DEFAULT_STORE_FLUSHER_CLASS_KEY,
678        CustomStoreFlusher.class.getName());
679    HRegion region =
680      HRegion.openHRegion(this.hbaseRootDir, hri, htd, wal, customConf, rsServices, null);
681    int writtenRowCount = 10;
682    List<HColumnDescriptor> families = new ArrayList<>(htd.getFamilies());
683    for (int i = 0; i < writtenRowCount; i++) {
684      Put put = new Put(Bytes.toBytes(tableName + Integer.toString(i)));
685      put.addColumn(families.get(i % families.size()).getName(), Bytes.toBytes("q"),
686          Bytes.toBytes("val"));
687      region.put(put);
688    }
689
690    // Now assert edits made it in.
691    RegionScanner scanner = region.getScanner(new Scan());
692    assertEquals(writtenRowCount, getScannedCount(scanner));
693
694    // Let us flush the region
695    CustomStoreFlusher.throwExceptionWhenFlushing.set(true);
696    try {
697      region.flush(true);
698      fail("Injected exception hasn't been thrown");
699    } catch (Throwable t) {
700      LOG.info("Expected simulated exception when flushing region,"
701          + t.getMessage());
702      // simulated to abort server
703      Mockito.doReturn(true).when(rsServices).isAborted();
704      region.setClosing(false); // region normally does not accept writes after
705      // DroppedSnapshotException. We mock around it for this test.
706    }
707    // writing more data
708    int moreRow = 10;
709    for (int i = writtenRowCount; i < writtenRowCount + moreRow; i++) {
710      Put put = new Put(Bytes.toBytes(tableName + Integer.toString(i)));
711      put.addColumn(families.get(i % families.size()).getName(), Bytes.toBytes("q"),
712          Bytes.toBytes("val"));
713      region.put(put);
714    }
715    writtenRowCount += moreRow;
716    // call flush again
717    CustomStoreFlusher.throwExceptionWhenFlushing.set(false);
718    try {
719      region.flush(true);
720    } catch (IOException t) {
721      LOG.info("Expected exception when flushing region because server is stopped,"
722          + t.getMessage());
723    }
724
725    region.close(true);
726    wal.shutdown();
727
728    // Let us try to split and recover
729    runWALSplit(this.conf);
730    WAL wal2 = createWAL(this.conf, hbaseRootDir, logName);
731    Mockito.doReturn(false).when(rsServices).isAborted();
732    HRegion region2 =
733      HRegion.openHRegion(this.hbaseRootDir, hri, htd, wal2, this.conf, rsServices, null);
734    scanner = region2.getScanner(new Scan());
735    assertEquals(writtenRowCount, getScannedCount(scanner));
736  }

738  private int getScannedCount(RegionScanner scanner) throws IOException {
739    int scannedCount = 0;
740    List<Cell> results = new ArrayList<>();
741    while (true) {
742      boolean existMore = scanner.next(results);
743      if (!results.isEmpty())
744        scannedCount++;
745      if (!existMore)
746        break;
747      results.clear();
748    }
749    return scannedCount;
750  }

773    // Add 1k to each family.

776    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);

832                  Mockito.mock(MonitoredTask.class), writeFlushWalMarker);

834              return fs;
835            }
836          };
837          // The seq id this region has opened up with

839
840          // The mvcc readpoint of from inserting data.
841          long writePoint = mvcc.getWritePoint();

848          Result result = region.get(get);
849          // Make sure we only see the good edits

855        }

857      }

1163  private void addWALEdits(final TableName tableName, final HRegionInfo hri, final byte[] rowName,

1165      final HTableDescriptor htd, final MultiVersionConcurrencyControl mvcc,

1174  static List<Put> addRegionEdits(final byte[] rowName, final byte[] family, final int count,
1175      EnvironmentEdge ee, final Region r, final String qualifierPrefix) throws IOException {
1176    List<Put> puts = new ArrayList<>();
1177    for (int j = 0; j < count; j++) {
1178      byte[] qualifier = Bytes.toBytes(qualifierPrefix + Integer.toString(j));
1179      Put p = new Put(rowName);
1180      p.addColumn(family, qualifier, ee.currentTime(), rowName);
1181      r.put(p);
1182      puts.add(p);
1183    }
1184    return puts;
