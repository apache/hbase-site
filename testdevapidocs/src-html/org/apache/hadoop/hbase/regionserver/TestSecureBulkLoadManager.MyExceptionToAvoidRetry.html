101  /**
102   * After a secure bulkload finished , there is a clean-up for FileSystems used in the bulkload.
103   * Sometimes, FileSystems used in the finished bulkload might also be used in other bulkload
104   * calls, or there are other FileSystems created by the same user, they could be closed by a
105   * FileSystem.closeAllForUGI call. So during the clean-up, those FileSystems need to be used
106   * later can not get closed ,or else a race condition occurs.
107   *
108   * testForRaceCondition tests the case that two secure bulkload calls from the same UGI go
109   * into two different regions and one bulkload finishes earlier when the other bulkload still
110   * needs its FileSystems, checks that both bulkloads succeed.
111   */
112  @Test
113  public void testForRaceCondition() throws Exception {
114    Consumer<HRegion> fsCreatedListener = new Consumer<HRegion>() {
115      @Override
116      public void accept(HRegion hRegion) {
117        if (hRegion.getRegionInfo().containsRow(key3)) {
118          Threads.shutdown(ealierBulkload);/// wait util the other bulkload finished
119        }
120      }
121    } ;
122    testUtil.getMiniHBaseCluster().getRegionServerThreads().get(0).getRegionServer()
123        .secureBulkLoadManager.setFsCreatedListener(fsCreatedListener);
124    /// create table
125    testUtil.createTable(TABLE,FAMILY,Bytes.toByteArrays(SPLIT_ROWKEY));
126
127    /// prepare files
128    Path rootdir = testUtil.getMiniHBaseCluster().getRegionServerThreads().get(0)
129        .getRegionServer().getRootDir();
130    Path dir1 = new Path(rootdir, "dir1");
131    prepareHFile(dir1, key1, value1);
132    Path dir2 = new Path(rootdir, "dir2");
133    prepareHFile(dir2, key3, value3);
134
135    /// do bulkload
136    final AtomicReference<Throwable> t1Exception = new AtomicReference<>();
137    final AtomicReference<Throwable> t2Exception = new AtomicReference<>();
138    ealierBulkload = new Thread(new Runnable() {
139      @Override
140      public void run() {
141        try {
142          doBulkloadWithoutRetry(dir1);
143        } catch (Exception e) {
144          LOG.error("bulk load failed .",e);
145          t1Exception.set(e);
146        }
147      }
148    });
149    laterBulkload = new Thread(new Runnable() {
150      @Override
151      public void run() {
152        try {
153          doBulkloadWithoutRetry(dir2);
154        } catch (Exception e) {
155          LOG.error("bulk load failed .",e);
156          t2Exception.set(e);
157        }
158      }
159    });
160    ealierBulkload.start();
161    laterBulkload.start();
162    Threads.shutdown(ealierBulkload);
163    Threads.shutdown(laterBulkload);
164    Assert.assertNull(t1Exception.get());
165    Assert.assertNull(t2Exception.get());
166
167    /// check bulkload ok
168    Get get1 = new Get(key1);
169    Get get3 = new Get(key3);
170    Table t = testUtil.getConnection().getTable(TABLE);
171    Result r = t.get(get1);
172    Assert.assertArrayEquals(r.getValue(FAMILY, COLUMN), value1);
173    r = t.get(get3);
174    Assert.assertArrayEquals(r.getValue(FAMILY, COLUMN), value3);
175
176  }

178  /**
179   * A trick is used to make sure server-side failures( if any ) not being covered up by a client
180   * retry. Since BulkLoadHFilesTool.bulkLoad keeps performing bulkload calls as long as the
181   * HFile queue is not empty, while server-side exceptions in the doAs block do not lead
182   * to a client exception, a bulkload will always succeed in this case by default, thus client
183   * will never be aware that failures have ever happened . To avoid this kind of retry ,
184   * a MyExceptionToAvoidRetry exception is thrown after bulkLoadPhase finished and caught
185   * silently outside the doBulkLoad call, so that the bulkLoadPhase would be called exactly
186   * once, and server-side failures, if any ,can be checked via data.
187   */
188  class MyExceptionToAvoidRetry extends DoNotRetryIOException {
189
190    private static final long serialVersionUID = -6802760664998771151L;
191  }
192
193  private void doBulkloadWithoutRetry(Path dir) throws Exception {
194    BulkLoadHFilesTool h = new BulkLoadHFilesTool(conf) {
195
196      @Override
197      protected void bulkLoadPhase(AsyncClusterConnection conn, TableName tableName,
198          Deque<LoadQueueItem> queue, Multimap<ByteBuffer, LoadQueueItem> regionGroups,
199          boolean copyFiles, Map<LoadQueueItem, ByteBuffer> item2RegionMap) throws IOException {
200        super.bulkLoadPhase(conn, tableName, queue, regionGroups, copyFiles, item2RegionMap);
201        throw new MyExceptionToAvoidRetry(); // throw exception to avoid retry
202      }
203    };
204    try {
205      h.bulkLoad(TABLE, dir);
206      Assert.fail("MyExceptionToAvoidRetry is expected");
207    } catch (MyExceptionToAvoidRetry e) { //expected
208    }
209  }

211  private void prepareHFile(Path dir, byte[] key, byte[] value) throws Exception {
212    TableDescriptor desc = testUtil.getAdmin().getDescriptor(TABLE);
213    ColumnFamilyDescriptor family = desc.getColumnFamily(FAMILY);
214    Compression.Algorithm compression = HFile.DEFAULT_COMPRESSION_ALGORITHM;
215
216    CacheConfig writerCacheConf = new CacheConfig(conf, family, null);
217    writerCacheConf.setCacheDataOnWrite(false);
218    HFileContext hFileContext = new HFileContextBuilder()
219        .withIncludesMvcc(false)
220        .withIncludesTags(true)
221        .withCompression(compression)
222        .withCompressTags(family.isCompressTags())
223        .withChecksumType(HStore.getChecksumType(conf))
224        .withBytesPerCheckSum(HStore.getBytesPerChecksum(conf))
225        .withBlockSize(family.getBlocksize())
226        .withHBaseCheckSum(true)
227        .withDataBlockEncoding(family.getDataBlockEncoding())
228        .withEncryptionContext(Encryption.Context.NONE)
229        .withCreateTime(EnvironmentEdgeManager.currentTime())
230        .build();
231    StoreFileWriter.Builder builder =
232        new StoreFileWriter.Builder(conf, writerCacheConf, dir.getFileSystem(conf))
233        .withOutputDir(new Path(dir, family.getNameAsString()))
234        .withBloomType(family.getBloomFilterType())
235        .withMaxKeyCount(Integer.MAX_VALUE)
236        .withFileContext(hFileContext);
237    StoreFileWriter writer = builder.build();
238
239    Put put = new Put(key);
240    put.addColumn(FAMILY, COLUMN, value);
241    for (Cell c : put.get(FAMILY, COLUMN)) {
242      writer.append(c);
243    }
244
245    writer.close();
246  }
