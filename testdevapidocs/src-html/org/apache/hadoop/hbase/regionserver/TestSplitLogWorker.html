161  private void waitForCounter(LongAdder ctr, long oldval, long newval, long timems)
162      throws Exception {
163    assertTrue("ctr=" + ctr.sum() + ", oldval=" + oldval + ", newval=" + newval,
164      waitForCounterBoolean(ctr, oldval, newval, timems));
165  }
166
167  private boolean waitForCounterBoolean(final LongAdder ctr, final long oldval, long newval,
168      long timems) throws Exception {
169
170    return waitForCounterBoolean(ctr, oldval, newval, timems, true);
171  }
172
173  private boolean waitForCounterBoolean(final LongAdder ctr, final long oldval, final long newval,
174      long timems, boolean failIfTimeout) throws Exception {
175
176    long timeWaited = TEST_UTIL.waitFor(timems, 10, failIfTimeout,
177      new Waiter.Predicate<Exception>() {
178      @Override
179      public boolean evaluate() throws Exception {
180            return (ctr.sum() >= newval);
181      }
182    });
183
184    if( timeWaited > 0) {
185      // when not timed out
186      assertEquals(newval, ctr.sum());
187    }
188    return true;
189  }

242  public void testAcquireTaskAtStartup() throws Exception {
243    LOG.info("testAcquireTaskAtStartup");
244    SplitLogCounters.resetCounters();
245    final String TATAS = "tatas";
246    final ServerName RS = ServerName.valueOf("rs,1,1");
247    RegionServerServices mockedRS = getRegionServer(RS);
248    zkw.getRecoverableZooKeeper().create(ZKSplitLog.getEncodedNodeName(zkw, TATAS),

250        Ids.OPEN_ACL_UNSAFE,

252

254        new SplitLogWorker(ds, TEST_UTIL.getConfiguration(), mockedRS, neverEndingTask);
255    slw.start();
256    try {
257      waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 0, 1, WAIT_TIME);

261    } finally {
262     stopSplitLogWorker(slw);
263    }
264  }

266  private void stopSplitLogWorker(final SplitLogWorker slw)
267  throws InterruptedException {
268    if (slw != null) {
269      slw.stop();
270      slw.worker.join(WAIT_TIME);
271      if (slw.worker.isAlive()) {
272        assertTrue(("Could not stop the worker thread slw=" + slw) == null);
273      }
274    }
275  }

281    final String TRFT = "trft";
282    final ServerName SVR1 = ServerName.valueOf("svr1,1,1");
283    final ServerName SVR2 = ServerName.valueOf("svr2,1,1");
284    zkw.getRecoverableZooKeeper().create(ZKSplitLog.getEncodedNodeName(zkw, TRFT),
285      new SplitLogTask.Unassigned(MANAGER).toByteArray(),
286        Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
287    RegionServerServices mockedRS1 = getRegionServer(SVR1);
288    RegionServerServices mockedRS2 = getRegionServer(SVR2);
289    SplitLogWorker slw1 =
290        new SplitLogWorker(ds, TEST_UTIL.getConfiguration(), mockedRS1, neverEndingTask);
291    SplitLogWorker slw2 =
292        new SplitLogWorker(ds, TEST_UTIL.getConfiguration(), mockedRS2, neverEndingTask);
293    slw1.start();
294    slw2.start();
295    try {
296      waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 0, 1, WAIT_TIME);
297      // Assert that either the tot_wkr_failed_to_grab_task_owned count was set of if
298      // not it, that we fell through to the next counter in line and it was set.
299      assertTrue(waitForCounterBoolean(SplitLogCounters.tot_wkr_failed_to_grab_task_owned, 0, 1,
300          WAIT_TIME, false) ||
301        SplitLogCounters.tot_wkr_failed_to_grab_task_lost_race.sum() == 1);
302      byte [] bytes = ZKUtil.getData(zkw, ZKSplitLog.getEncodedNodeName(zkw, TRFT));
303      SplitLogTask slt = SplitLogTask.parseFrom(bytes);
304      assertTrue(slt.isOwned(SVR1) || slt.isOwned(SVR2));
305    } finally {
306      stopSplitLogWorker(slw1);
307      stopSplitLogWorker(slw2);
308    }
309  }

311  @Test(timeout=60000)
312  public void testPreemptTask() throws Exception {
313    LOG.info("testPreemptTask");
314    SplitLogCounters.resetCounters();
315    final ServerName SRV = ServerName.valueOf("tpt_svr,1,1");
316    final String PATH = ZKSplitLog.getEncodedNodeName(zkw, "tpt_task");
317    RegionServerServices mockedRS = getRegionServer(SRV);
318    SplitLogWorker slw =
319        new SplitLogWorker(ds, TEST_UTIL.getConfiguration(), mockedRS, neverEndingTask);
320    slw.start();
321    try {
322      Thread.yield(); // let the worker start
323      Thread.sleep(1000);
324      waitForCounter(SplitLogCounters.tot_wkr_task_grabing, 0, 1, WAIT_TIME);
325
326      // this time create a task node after starting the splitLogWorker
327      zkw.getRecoverableZooKeeper().create(PATH,
328        new SplitLogTask.Unassigned(MANAGER).toByteArray(),
329        Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
330
331      waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 0, 1, WAIT_TIME);
332      assertEquals(1, slw.getTaskReadySeq());
333      byte [] bytes = ZKUtil.getData(zkw, PATH);
334      SplitLogTask slt = SplitLogTask.parseFrom(bytes);
335      assertTrue(slt.isOwned(SRV));
336      slt = new SplitLogTask.Owned(MANAGER);
337      ZKUtil.setData(zkw, PATH, slt.toByteArray());
338      waitForCounter(SplitLogCounters.tot_wkr_preempt_task, 0, 1, WAIT_TIME);
339    } finally {
340      stopSplitLogWorker(slw);
341    }
342  }

344  @Test(timeout=60000)
345  public void testMultipleTasks() throws Exception {
346    LOG.info("testMultipleTasks");
347    SplitLogCounters.resetCounters();
348    final ServerName SRV = ServerName.valueOf("tmt_svr,1,1");
349    final String PATH1 = ZKSplitLog.getEncodedNodeName(zkw, "tmt_task");
350    RegionServerServices mockedRS = getRegionServer(SRV);
351    SplitLogWorker slw =
352        new SplitLogWorker(ds, TEST_UTIL.getConfiguration(), mockedRS, neverEndingTask);
353    slw.start();
354    try {
355      Thread.yield(); // let the worker start
356      Thread.sleep(100);
357      waitForCounter(SplitLogCounters.tot_wkr_task_grabing, 0, 1, WAIT_TIME);
358
359      SplitLogTask unassignedManager =
360        new SplitLogTask.Unassigned(MANAGER);
361      zkw.getRecoverableZooKeeper().create(PATH1, unassignedManager.toByteArray(),
362        Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
363
364      waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 0, 1, WAIT_TIME);
365      // now the worker is busy doing the above task
366
367      // create another task
368      final String PATH2 = ZKSplitLog.getEncodedNodeName(zkw, "tmt_task_2");
369      zkw.getRecoverableZooKeeper().create(PATH2, unassignedManager.toByteArray(),
370        Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
371
372      // preempt the first task, have it owned by another worker
373      final ServerName anotherWorker = ServerName.valueOf("another-worker,1,1");
374      SplitLogTask slt = new SplitLogTask.Owned(anotherWorker);
375      ZKUtil.setData(zkw, PATH1, slt.toByteArray());
376      waitForCounter(SplitLogCounters.tot_wkr_preempt_task, 0, 1, WAIT_TIME);
377
378      waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 1, 2, WAIT_TIME);
379      assertEquals(2, slw.getTaskReadySeq());
380      byte [] bytes = ZKUtil.getData(zkw, PATH2);
381      slt = SplitLogTask.parseFrom(bytes);
382      assertTrue(slt.isOwned(SRV));
383    } finally {
384      stopSplitLogWorker(slw);
385    }
386  }

388  @Test(timeout=60000)
389  public void testRescan() throws Exception {
390    LOG.info("testRescan");
391    SplitLogCounters.resetCounters();
392    final ServerName SRV = ServerName.valueOf("svr,1,1");
393    RegionServerServices mockedRS = getRegionServer(SRV);
394    slw = new SplitLogWorker(ds, TEST_UTIL.getConfiguration(), mockedRS, neverEndingTask);
395    slw.start();
396    Thread.yield(); // let the worker start
397    Thread.sleep(100);
398
399    String task = ZKSplitLog.getEncodedNodeName(zkw, "task");
400    SplitLogTask slt = new SplitLogTask.Unassigned(MANAGER);
401    zkw.getRecoverableZooKeeper().create(task,slt.toByteArray(), Ids.OPEN_ACL_UNSAFE,
402      CreateMode.PERSISTENT);
403
404    waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 0, 1, WAIT_TIME);
405    // now the worker is busy doing the above task
406
407    // preempt the task, have it owned by another worker
408    ZKUtil.setData(zkw, task, slt.toByteArray());
409    waitForCounter(SplitLogCounters.tot_wkr_preempt_task, 0, 1, WAIT_TIME);
410
411    // create a RESCAN node
412    String rescan = ZKSplitLog.getEncodedNodeName(zkw, "RESCAN");
413    rescan = zkw.getRecoverableZooKeeper().create(rescan, slt.toByteArray(), Ids.OPEN_ACL_UNSAFE,
414      CreateMode.PERSISTENT_SEQUENTIAL);
415
416    waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 1, 2, WAIT_TIME);
417    // RESCAN node might not have been processed if the worker became busy
418    // with the above task. preempt the task again so that now the RESCAN
419    // node is processed
420    ZKUtil.setData(zkw, task, slt.toByteArray());
421    waitForCounter(SplitLogCounters.tot_wkr_preempt_task, 1, 2, WAIT_TIME);
422    waitForCounter(SplitLogCounters.tot_wkr_task_acquired_rescan, 0, 1, WAIT_TIME);
423
424    List<String> nodes = ZKUtil.listChildrenNoWatch(zkw, zkw.znodePaths.splitLogZNode);
425    LOG.debug(nodes);
426    int num = 0;
427    for (String node : nodes) {
428      num++;
429      if (node.startsWith("RESCAN")) {
430        String name = ZKSplitLog.getEncodedNodeName(zkw, node);
431        String fn = ZKSplitLog.getFileName(name);
432        byte [] data = ZKUtil.getData(zkw, ZKUtil.joinZNode(zkw.znodePaths.splitLogZNode, fn));
433        slt = SplitLogTask.parseFrom(data);
434        assertTrue(slt.toString(), slt.isDone(SRV));
435      }
436    }
437    assertEquals(2, num);
438  }

477    LOG.info("testAcquireMultiTasks");
478    SplitLogCounters.resetCounters();
479    final String TATAS = "tatas";
480    final ServerName RS = ServerName.valueOf("rs,1,1");
481    final ServerName RS2 = ServerName.valueOf("rs,1,2");
482    final int maxTasks = 3;
483    Configuration testConf = HBaseConfiguration.create(TEST_UTIL.getConfiguration());
484    testConf.setInt("hbase.regionserver.wal.max.splitters", maxTasks);
485    RegionServerServices mockedRS = getRegionServer(RS);
486
487    // create two RS nodes
488    String rsPath = ZKUtil.joinZNode(zkw.znodePaths.rsZNode, RS.getServerName());
489    zkw.getRecoverableZooKeeper().create(rsPath, null, Ids.OPEN_ACL_UNSAFE, CreateMode.EPHEMERAL);
490    rsPath = ZKUtil.joinZNode(zkw.znodePaths.rsZNode, RS2.getServerName());
491    zkw.getRecoverableZooKeeper().create(rsPath, null, Ids.OPEN_ACL_UNSAFE, CreateMode.EPHEMERAL);
492
493    for (int i = 0; i < maxTasks; i++) {
494      zkw.getRecoverableZooKeeper().create(ZKSplitLog.getEncodedNodeName(zkw, TATAS + i),
495        new SplitLogTask.Unassigned(ServerName.valueOf("mgr,1,1")).toByteArray(),
496          Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
497    }
498
499    SplitLogWorker slw = new SplitLogWorker(ds, testConf, mockedRS, neverEndingTask);
500    slw.start();
501    try {
502      int acquiredTasks = 0;
503      waitForCounter(SplitLogCounters.tot_wkr_task_acquired, 0, 2, WAIT_TIME);
504      for (int i = 0; i < maxTasks; i++) {
505        byte[] bytes = ZKUtil.getData(zkw, ZKSplitLog.getEncodedNodeName(zkw, TATAS + i));
506        SplitLogTask slt = SplitLogTask.parseFrom(bytes);
507        if (slt.isOwned(RS)) {
508          acquiredTasks++;
509        }
510      }
511      assertEquals(2, acquiredTasks);
512    } finally {
513      stopSplitLogWorker(slw);
514    }
515  }
