144   * Verify that you can stop a long-running compaction
145   * (used during RS shutdown)
146   * @throws Exception
147   */
148  @Test
149  public void testInterruptCompaction() throws Exception {
150    assertEquals(0, count());
151
152    // lower the polling interval for this test
153    int origWI = HStore.closeCheckInterval;
154    HStore.closeCheckInterval = 10*1000; // 10 KB
155
156    try {
157      // Create a couple store files w/ 15KB (over 10KB interval)
158      int jmax = (int) Math.ceil(15.0/compactionThreshold);
159      byte [] pad = new byte[1000]; // 1 KB chunk
160      for (int i = 0; i < compactionThreshold; i++) {
161        Table loader = new RegionAsTable(r);
162        Put p = new Put(Bytes.add(STARTROW, Bytes.toBytes(i)));
163        p.setDurability(Durability.SKIP_WAL);
164        for (int j = 0; j < jmax; j++) {
165          p.addColumn(COLUMN_FAMILY, Bytes.toBytes(j), pad);
166        }
167        HBaseTestCase.addContent(loader, Bytes.toString(COLUMN_FAMILY));
168        loader.put(p);
169        r.flush(true);
170      }
171
172      HRegion spyR = spy(r);
173      doAnswer(new Answer() {
174        @Override
175        public Object answer(InvocationOnMock invocation) throws Throwable {
176          r.writestate.writesEnabled = false;
177          return invocation.callRealMethod();
178        }
179      }).when(spyR).doRegionCompactionPrep();
180
181      // force a minor compaction, but not before requesting a stop
182      spyR.compactStores();
183
184      // ensure that the compaction stopped, all old files are intact,
185      Store s = r.stores.get(COLUMN_FAMILY);
186      assertEquals(compactionThreshold, s.getStorefilesCount());
187      assertTrue(s.getStorefilesSize() > 15*1000);
188      // and no new store files persisted past compactStores()
189      // only one empty dir exists in temp dir
190      FileStatus[] ls = r.getFilesystem().listStatus(r.getRegionFileSystem().getTempDir());
191      assertEquals(1, ls.length);
192      Path storeTempDir = new Path(r.getRegionFileSystem().getTempDir(), Bytes.toString(COLUMN_FAMILY));
193      assertTrue(r.getFilesystem().exists(storeTempDir));
194      ls = r.getFilesystem().listStatus(storeTempDir);
195      assertEquals(0, ls.length);
196    } finally {
197      // don't mess up future tests
198      r.writestate.writesEnabled = true;
199      HStore.closeCheckInterval = origWI;
200
201      // Delete all Store information once done using
202      for (int i = 0; i < compactionThreshold; i++) {
203        Delete delete = new Delete(Bytes.add(STARTROW, Bytes.toBytes(i)));
204        byte [][] famAndQf = {COLUMN_FAMILY, null};
205        delete.addFamily(famAndQf[0]);
206        r.delete(delete);
207      }
208      r.flush(true);
209
210      // Multiple versions allowed for an entry, so the delete isn't enough
211      // Lower TTL and expire to ensure that all our entries have been wiped
212      final int ttl = 1000;
213      for (Store hstore: this.r.stores.values()) {
214        HStore store = (HStore)hstore;
215        ScanInfo old = store.getScanInfo();
216        ScanInfo si = new ScanInfo(old.getConfiguration(), old.getFamily(), old.getMinVersions(),
217            old.getMaxVersions(), ttl, old.getKeepDeletedCells(), HConstants.DEFAULT_BLOCKSIZE, 0,
218            old.getComparator(), old.isNewVersionBehavior());
219        store.setScanInfo(si);
220      }
221      Thread.sleep(ttl);
222
223      r.compact(true);
224      assertEquals(0, count());
225    }
226  }

228  private int count() throws IOException {
229    int count = 0;
230    for (HStoreFile f: this.r.stores.
231        get(COLUMN_FAMILY_TEXT).getStorefiles()) {
232      HFileScanner scanner = f.getReader().getScanner(false, false);
233      if (!scanner.seekTo()) {
234        continue;
235      }
236      do {
237        count++;
238      } while(scanner.next());
239    }
240    return count;
241  }

255    int nfiles = 10;
256    for (int i = 0; i < nfiles; i++) {
257      createStoreFile(r);
258    }
259    HStore store = r.getStore(COLUMN_FAMILY);
260
261    Collection<HStoreFile> storeFiles = store.getStorefiles();
262    DefaultCompactor tool = (DefaultCompactor)store.storeEngine.getCompactor();
263    tool.compactForTesting(storeFiles, false);
264
265    // Now lets corrupt the compacted file.
266    FileSystem fs = store.getFileSystem();
267    // default compaction policy created one and only one new compacted file
268    Path dstPath = store.getRegionFileSystem().createTempName();
269    FSDataOutputStream stream = fs.create(dstPath, null, true, 512, (short)3, (long)1024, null);
270    stream.writeChars("CORRUPT FILE!!!!");
271    stream.close();
272    Path origPath = store.getRegionFileSystem().commitStoreFile(
273      Bytes.toString(COLUMN_FAMILY), dstPath);
274
275    try {
276      ((HStore)store).moveFileIntoPlace(origPath);
277    } catch (Exception e) {
278      // The complete compaction should fail and the corrupt file should remain
279      // in the 'tmp' directory;
280      assertTrue(fs.exists(origPath));
281      assertFalse(fs.exists(dstPath));
282      System.out.println("testCompactionWithCorruptResult Passed");
283      return;
284    }
285    fail("testCompactionWithCorruptResult failed since no exception was" +
286        "thrown while completing a corrupt file");

545  /** Test compaction priority management and multiple compactions per store (HBASE-8665). */
546  @Test
547  public void testCompactionQueuePriorities() throws Exception {
548    // Setup a compact/split thread on a mock server.
549    final Configuration conf = HBaseConfiguration.create();
550    HRegionServer mockServer = mock(HRegionServer.class);
551    when(mockServer.isStopped()).thenReturn(false);
552    when(mockServer.getConfiguration()).thenReturn(conf);
553    when(mockServer.getChoreService()).thenReturn(new ChoreService("test"));
554    CompactSplit cst = new CompactSplit(mockServer);
555    when(mockServer.getCompactSplitThread()).thenReturn(cst);
556    //prevent large compaction thread pool stealing job from small compaction queue.
557    cst.shutdownLongCompactions();
558    // Set up the region mock that redirects compactions.
559    HRegion r = mock(HRegion.class);
560    when(
561      r.compact(any(CompactionContext.class), any(HStore.class),
562        any(ThroughputController.class), any(User.class))).then(new Answer<Boolean>() {
563      @Override
564      public Boolean answer(InvocationOnMock invocation) throws Throwable {
565        invocation.getArgumentAt(0, CompactionContext.class).compact(
566          invocation.getArgumentAt(2, ThroughputController.class), null);
567        return true;
568      }
569    });
570
571    // Set up store mocks for 2 "real" stores and the one we use for blocking CST.
572    ArrayList<Integer> results = new ArrayList<>();
573    StoreMockMaker sm = new StoreMockMaker(results), sm2 = new StoreMockMaker(results);
574    HStore store = sm.createStoreMock("store1"), store2 = sm2.createStoreMock("store2");
575    BlockingStoreMockMaker blocker = new BlockingStoreMockMaker();
576
577    // First, block the compaction thread so that we could muck with queue.
578    cst.requestSystemCompaction(r, blocker.createStoreMock(1, "b-pri1"), "b-pri1");
579    BlockingStoreMockMaker.BlockingCompactionContext currentBlock = blocker.waitForBlocking();
580
581    // Add 4 files to store1, 3 to store2, and queue compactions; pri 3 and 4 respectively.
582    for (int i = 0; i < 4; ++i) {
583      sm.notCompacting.add(createFile());
584    }
585    cst.requestSystemCompaction(r, store, "s1-pri3");
586    for (int i = 0; i < 3; ++i) {
587      sm2.notCompacting.add(createFile());
588    }
589    cst.requestSystemCompaction(r, store2, "s2-pri4");
590    // Now add 2 more files to store1 and queue compaction - pri 1.
591    for (int i = 0; i < 2; ++i) {
592      sm.notCompacting.add(createFile());
593    }
594    cst.requestSystemCompaction(r, store, "s1-pri1");
595    // Finally add blocking compaction with priority 2.
596    cst.requestSystemCompaction(r, blocker.createStoreMock(2, "b-pri2"), "b-pri2");
597
598    // Unblock the blocking compaction; we should run pri1 and become block again in pri2.
599    currentBlock.unblock();
600    currentBlock = blocker.waitForBlocking();
601    // Pri1 should have "compacted" all 6 files.
602    assertEquals(1, results.size());
603    assertEquals(6, results.get(0).intValue());
604    // Add 2 files to store 1 (it has 2 files now).
605    for (int i = 0; i < 2; ++i) {
606      sm.notCompacting.add(createFile());
607    }
608    // Now we have pri4 for store 2 in queue, and pri3 for store1; store1's current priority
609    // is 5, however, so it must not preempt store 2. Add blocking compaction at the end.
610    cst.requestSystemCompaction(r, blocker.createStoreMock(7, "b-pri7"), "b-pri7");
611    currentBlock.unblock();
612    currentBlock = blocker.waitForBlocking();
613    assertEquals(3, results.size());
614    assertEquals(3, results.get(1).intValue()); // 3 files should go before 2 files.
615    assertEquals(2, results.get(2).intValue());
616
617    currentBlock.unblock();
618    cst.interruptIfNecessary();
619  }

623   * to hfile1, then write 10 cells (with different time stamp) to the same
624   * qualifier and flush to hfile2. The latest cell (cell-A) in hfile1 and the
625   * oldest cell (cell-B) in hfile2 are with the same time stamp but different
626   * sequence id, and will get scanned successively during compaction.
627   * <p/>
628   * We set compaction.kv.max to 10 so compaction will scan 10 versions each
629   * round, meanwhile we set keepSeqIdPeriod=0 in {@link DummyCompactor} so all
630   * 10 versions of hfile2 will be written out with seqId cleaned (set to 0)
631   * including cell-B, then when scanner goes to cell-A it will cause a scan
632   * out-of-order assertion error before HBASE-16931
633   *
634   * @throws Exception
635   *           if error occurs during the test
636   */
637  @Test
638  public void testCompactionSeqId() throws Exception {
639    final byte[] ROW = Bytes.toBytes("row");
640    final byte[] QUALIFIER = Bytes.toBytes("qualifier");
641
642    long timestamp = 10000;
643
644    // row1/cf:a/10009/Put/vlen=2/seqid=11 V: v9
645    // row1/cf:a/10008/Put/vlen=2/seqid=10 V: v8
646    // row1/cf:a/10007/Put/vlen=2/seqid=9 V: v7
647    // row1/cf:a/10006/Put/vlen=2/seqid=8 V: v6
648    // row1/cf:a/10005/Put/vlen=2/seqid=7 V: v5
649    // row1/cf:a/10004/Put/vlen=2/seqid=6 V: v4
650    // row1/cf:a/10003/Put/vlen=2/seqid=5 V: v3
651    // row1/cf:a/10002/Put/vlen=2/seqid=4 V: v2
652    // row1/cf:a/10001/Put/vlen=2/seqid=3 V: v1
653    // row1/cf:a/10000/Put/vlen=2/seqid=2 V: v0
654    for (int i = 0; i < 10; i++) {
655      Put put = new Put(ROW);
656      put.addColumn(FAMILY, QUALIFIER, timestamp + i, Bytes.toBytes("v" + i));
657      r.put(put);
658    }
659    r.flush(true);
660
661    // row1/cf:a/10018/Put/vlen=3/seqid=16 V: v18
662    // row1/cf:a/10017/Put/vlen=3/seqid=17 V: v17
663    // row1/cf:a/10016/Put/vlen=3/seqid=18 V: v16
664    // row1/cf:a/10015/Put/vlen=3/seqid=19 V: v15
665    // row1/cf:a/10014/Put/vlen=3/seqid=20 V: v14
666    // row1/cf:a/10013/Put/vlen=3/seqid=21 V: v13
667    // row1/cf:a/10012/Put/vlen=3/seqid=22 V: v12
668    // row1/cf:a/10011/Put/vlen=3/seqid=23 V: v11
669    // row1/cf:a/10010/Put/vlen=3/seqid=24 V: v10
670    // row1/cf:a/10009/Put/vlen=2/seqid=25 V: v9
671    for (int i = 18; i > 8; i--) {
672      Put put = new Put(ROW);
673      put.addColumn(FAMILY, QUALIFIER, timestamp + i, Bytes.toBytes("v" + i));
674      r.put(put);
675    }
676    r.flush(true);
677    r.compact(true);
678  }
