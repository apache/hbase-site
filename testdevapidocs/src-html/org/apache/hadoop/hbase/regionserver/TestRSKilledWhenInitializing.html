058/**
059 * Tests that a regionserver that dies after reporting for duty gets removed
060 * from list of online regions. See HBASE-9593.
061 */
062@Category({RegionServerTests.class, MediumTests.class})
063public class TestRSKilledWhenInitializing {
064  private static final Log LOG = LogFactory.getLog(TestRSKilledWhenInitializing.class);
065  @Rule public TestName testName = new TestName();
066  @Rule public final TestRule timeout = CategoryBasedTimeout.builder().
067    withTimeout(this.getClass()).withLookingForStuckThread(true).build();
068
069  // This boolean needs to be globally available. It is used below in our
070  // mocked up regionserver so it knows when to die.
071  private static AtomicBoolean masterActive = new AtomicBoolean(false);
072  // Ditto for this variable. It also is used in the mocked regionserver class.
073  private static final AtomicReference<ServerName> killedRS = new AtomicReference<ServerName>();
074
075  private static final int NUM_MASTERS = 1;
076  private static final int NUM_RS = 2;
077
078  /**
079   * Test verifies whether a region server is removed from online servers list in master if it went
080   * down after registering with master. Test will TIMEOUT if an error!!!!
081   * @throws Exception
082   */
083  @Test
084  public void testRSTerminationAfterRegisteringToMasterBeforeCreatingEphemeralNode()
085  throws Exception {
086    // Create config to use for this cluster
087    Configuration conf = HBaseConfiguration.create();
088    conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, 1);
089    // Start the cluster
090    final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility(conf);
091    TEST_UTIL.startMiniDFSCluster(3);
092    TEST_UTIL.startMiniZKCluster();
093    TEST_UTIL.createRootDir();
094    final LocalHBaseCluster cluster =
095        new LocalHBaseCluster(conf, NUM_MASTERS, NUM_RS, HMaster.class,
096            RegisterAndDieRegionServer.class);
097    final MasterThread master = startMaster(cluster.getMasters().get(0));
098    try {
099      // Master is up waiting on RegionServers to check in. Now start RegionServers.
100      for (int i = 0; i < NUM_RS; i++) {
101        cluster.getRegionServers().get(i).start();
102      }
103      // Expected total regionservers depends on whether Master can host regions or not.
104      int expectedTotalRegionServers = NUM_RS + (LoadBalancer.isTablesOnMaster(conf)? 1: 0);
105      List<ServerName> onlineServersList = null;
106      do {
107        onlineServersList = master.getMaster().getServerManager().getOnlineServersList();
108      } while (onlineServersList.size() < expectedTotalRegionServers);
109      // Wait until killedRS is set. Means RegionServer is starting to go down.
110      while (killedRS.get() == null) {
111        Threads.sleep(1);
112      }
113      // Wait on the RegionServer to fully die.
114      while (cluster.getLiveRegionServers().size() >= expectedTotalRegionServers) {
115        Threads.sleep(1);
116      }
117      // Make sure Master is fully up before progressing. Could take a while if regions
118      // being reassigned.
119      while (!master.getMaster().isInitialized()) {
120        Threads.sleep(1);
121      }
122
123      // Now in steady state. How many regions open? Master should have too many regionservers
124      // showing still. The downed RegionServer should still be showing as registered.
125      assertTrue(master.getMaster().getServerManager().isServerOnline(killedRS.get()));
126      // Find non-meta region (namespace?) and assign to the killed server. That'll trigger cleanup.
127      Map<RegionInfo, ServerName> assignments = null;
128      do {
129        assignments = master.getMaster().getAssignmentManager().getRegionStates().getRegionAssignments();
130      } while (assignments == null || assignments.size() < 2);
131      RegionInfo hri = null;
132      for (Map.Entry<RegionInfo, ServerName> e: assignments.entrySet()) {
133        if (e.getKey().isMetaRegion()) continue;
134        hri = e.getKey();
135        break;
136      }
137      // Try moving region to the killed server. It will fail. As by-product, we will
138      // remove the RS from Master online list because no corresponding znode.
139      assertEquals(expectedTotalRegionServers,
140        master.getMaster().getServerManager().getOnlineServersList().size());
141      LOG.info("Move " + hri.getEncodedName() + " to " + killedRS.get());
142      master.getMaster().move(hri.getEncodedNameAsBytes(),
143          Bytes.toBytes(killedRS.get().toString()));
144      // Wait until the RS no longer shows as registered in Master.
145      while (onlineServersList.size() > (NUM_RS + 1)) {
146        Thread.sleep(100);
147        onlineServersList = master.getMaster().getServerManager().getOnlineServersList();
148      }
149    } finally {
150      // Shutdown is messy with complaints about fs being closed. Why? TODO.
151      cluster.shutdown();
152      cluster.join();
153      TEST_UTIL.shutdownMiniDFSCluster();
154      TEST_UTIL.shutdownMiniZKCluster();
155      TEST_UTIL.cleanupTestDir();
156    }
157  }
158
159  /**
160   * Start Master. Get as far as the state where Master is waiting on
161   * RegionServers to check in, then return.
162   */
163  private MasterThread startMaster(MasterThread master) {
164    master.start();
165    // It takes a while until ServerManager creation to happen inside Master startup.
166    while (master.getMaster().getServerManager() == null) {
167      continue;
168    }
169    // Set a listener for the waiting-on-RegionServers state. We want to wait
170    // until this condition before we leave this method and start regionservers.
171    final AtomicBoolean waiting = new AtomicBoolean(false);
172    if (master.getMaster().getServerManager() == null) throw new NullPointerException("SM");
173    master.getMaster().getServerManager().registerListener(new ServerListener() {
174      @Override
175      public void waiting() {
176        waiting.set(true);
177      }
178    });
179    // Wait until the Master gets to place where it is waiting on RegionServers to check in.
180    while (!waiting.get()) {
181      continue;
182    }
183    // Set the global master-is-active; gets picked up by regionservers later.
184    masterActive.set(true);
185    return master;
186  }
187
188  /**
189   * A RegionServer that reports for duty and then immediately dies if it is the first to receive
190   * the response to a reportForDuty. When it dies, it clears its ephemeral znode which the master
191   * notices and so removes the region from its set of online regionservers.
192   */
193  static class RegisterAndDieRegionServer extends MiniHBaseCluster.MiniHBaseClusterRegionServer {
194    public RegisterAndDieRegionServer(Configuration conf, CoordinatedStateManager cp)
195    throws IOException, InterruptedException {
196      super(conf, cp);
197    }
198
199    @Override
200    protected void handleReportForDutyResponse(RegionServerStartupResponse c)
201    throws IOException {
202      if (killedRS.compareAndSet(null, getServerName())) {
203        // Make sure Master is up so it will see the removal of the ephemeral znode for this RS.
204        while (!masterActive.get()) {
205          Threads.sleep(100);
206        }
207        super.kill();
208      } else {
209        super.handleReportForDutyResponse(c);
210      }
211    }
212  }
