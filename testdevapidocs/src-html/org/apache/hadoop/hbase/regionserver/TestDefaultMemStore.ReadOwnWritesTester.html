116  @Test
117  public void testPutSameCell() {
118    byte[] bytes = Bytes.toBytes(getName());
119    KeyValue kv = new KeyValue(bytes, bytes, bytes, bytes);
120    long sizeChangeForFirstCell = this.memstore.add(kv);
121    long sizeChangeForSecondCell = this.memstore.add(kv);
122    // make sure memstore size increase won't double-count MSLAB chunk size
123    assertEquals(AbstractMemStore.heapSizeChange(kv, true), sizeChangeForFirstCell);
124    Segment segment = this.memstore.getActive();
125    MemStoreLAB msLab = segment.getMemStoreLAB();
126    if (msLab != null) {
127      // make sure memstore size increased even when writing the same cell, if using MSLAB
128      assertEquals(segment.getCellLength(kv), sizeChangeForSecondCell);
129      // make sure chunk size increased even when writing the same cell, if using MSLAB
130      if (msLab instanceof HeapMemStoreLAB) {
131        assertEquals(2 * segment.getCellLength(kv),
132          ((HeapMemStoreLAB) msLab).getCurrentChunk().getNextFreeOffset());
133      }
134    } else {
135      // make sure no memstore size change w/o MSLAB
136      assertEquals(0, sizeChangeForSecondCell);
137    }

144  @Test
145  public void testScanAcrossSnapshot() throws IOException {
146    int rowCount = addRows(this.memstore);
147    List<KeyValueScanner> memstorescanners = this.memstore.getScanners(0);
148    Scan scan = new Scan();
149    List<Cell> result = new ArrayList<Cell>();
150    Configuration conf = HBaseConfiguration.create();
151    ScanInfo scanInfo =
152        new ScanInfo(conf, null, 0, 1, HConstants.LATEST_TIMESTAMP, KeepDeletedCells.FALSE, 0,
153            this.memstore.getComparator());
154    ScanType scanType = ScanType.USER_SCAN;
155    StoreScanner s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
156    int count = 0;
157    try {
158      while (s.next(result)) {
159        LOG.info(result);
160        count++;
161        // Row count is same as column count.
162        assertEquals(rowCount, result.size());
163        result.clear();
164      }
165    } finally {
166      s.close();
167    }
168    assertEquals(rowCount, count);
169    for (KeyValueScanner scanner : memstorescanners) {
170      scanner.close();
171    }
172
173    memstorescanners = this.memstore.getScanners(mvcc.getReadPoint());
174    // Now assert can count same number even if a snapshot mid-scan.
175    s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
176    count = 0;
177    try {
178      while (s.next(result)) {
179        LOG.info(result);
180        // Assert the stuff is coming out in right order.
181        assertTrue(CellUtil.matchingRow(result.get(0), Bytes.toBytes(count)));
182        count++;
183        // Row count is same as column count.
184        assertEquals(rowCount, result.size());
185        if (count == 2) {
186          this.memstore.snapshot();
187          LOG.info("Snapshotted");
188        }
189        result.clear();
190      }
191    } finally {
192      s.close();
193    }
194    assertEquals(rowCount, count);
195    for (KeyValueScanner scanner : memstorescanners) {
196      scanner.close();
197    }
198    memstorescanners = this.memstore.getScanners(mvcc.getReadPoint());
199    // Assert that new values are seen in kvset as we scan.
200    long ts = System.currentTimeMillis();
201    s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
202    count = 0;
203    int snapshotIndex = 5;
204    try {
205      while (s.next(result)) {
206        LOG.info(result);
207        // Assert the stuff is coming out in right order.
208        assertTrue(CellUtil.matchingRow(result.get(0), Bytes.toBytes(count)));
209        // Row count is same as column count.
210        assertEquals("count=" + count + ", result=" + result, rowCount, result.size());
211        count++;
212        if (count == snapshotIndex) {
213          MemStoreSnapshot snapshot = this.memstore.snapshot();
214          this.memstore.clearSnapshot(snapshot.getId());
215          // Added more rows into kvset.  But the scanner wont see these rows.
216          addRows(this.memstore, ts);
217          LOG.info("Snapshotted, cleared it and then added values (which wont be seen)");
218        }
219        result.clear();
220      }
221    } finally {
222      s.close();
223    }
224    assertEquals(rowCount, count);
225  }

232  @Test
233  public void testScanAcrossSnapshot2() throws IOException, CloneNotSupportedException {
234    // we are going to the scanning across snapshot with two kvs
235    // kv1 should always be returned before kv2
236    final byte[] one = Bytes.toBytes(1);
237    final byte[] two = Bytes.toBytes(2);
238    final byte[] f = Bytes.toBytes("f");
239    final byte[] q = Bytes.toBytes("q");
240    final byte[] v = Bytes.toBytes(3);
241
242    final KeyValue kv1 = new KeyValue(one, f, q, v);
243    final KeyValue kv2 = new KeyValue(two, f, q, v);
244
245    // use case 1: both kvs in kvset
246    this.memstore.add(kv1.clone());
247    this.memstore.add(kv2.clone());
248    verifyScanAcrossSnapshot2(kv1, kv2);
249
250    // use case 2: both kvs in snapshot
251    this.memstore.snapshot();
252    verifyScanAcrossSnapshot2(kv1, kv2);
253
254    // use case 3: first in snapshot second in kvset
255    this.memstore = new DefaultMemStore();
256    this.memstore.add(kv1.clone());
257    this.memstore.snapshot();
258    this.memstore.add(kv2.clone());
259    verifyScanAcrossSnapshot2(kv1, kv2);
260  }
261
262  protected void verifyScanAcrossSnapshot2(KeyValue kv1, KeyValue kv2)
263      throws IOException {
264    List<KeyValueScanner> memstorescanners = this.memstore.getScanners(mvcc.getReadPoint());
265    assertEquals(1, memstorescanners.size());
266    final KeyValueScanner scanner = memstorescanners.get(0);
267    scanner.seek(KeyValueUtil.createFirstOnRow(HConstants.EMPTY_START_ROW));
268    assertEquals(kv1, scanner.next());
269    assertEquals(kv2, scanner.next());
270    assertNull(scanner.next());
271  }
272
273  protected void assertScannerResults(KeyValueScanner scanner, KeyValue[] expected)
274      throws IOException {
275    scanner.seek(KeyValueUtil.createFirstOnRow(new byte[]{}));
276    List<Cell> returned = Lists.newArrayList();
277
278    while (true) {
279      Cell next = scanner.next();
280      if (next == null) break;
281      returned.add(next);
282    }
283
284    assertTrue(
285        "Got:\n" + Joiner.on("\n").join(returned) +
286        "\nExpected:\n" + Joiner.on("\n").join(expected),
287        Iterables.elementsEqual(Arrays.asList(expected), returned));
288    assertNull(scanner.peek());
289  }

336    final byte[] row = Bytes.toBytes(1);
337    final byte[] f = Bytes.toBytes("family");
338    final byte[] q1 = Bytes.toBytes("q1");
339    final byte[] q2 = Bytes.toBytes("q2");
340    final byte[] v1 = Bytes.toBytes("value1");
341    final byte[] v2 = Bytes.toBytes("value2");
342
343    // INSERT 1: Write both columns val1
344    MultiVersionConcurrencyControl.WriteEntry w =
345        mvcc.begin();
346
347    KeyValue kv11 = new KeyValue(row, f, q1, v1);
348    kv11.setSequenceId(w.getWriteNumber());
349    memstore.add(kv11);
350
351    KeyValue kv12 = new KeyValue(row, f, q2, v1);
352    kv12.setSequenceId(w.getWriteNumber());
353    memstore.add(kv12);
354    mvcc.completeAndWait(w);
355
356    // BEFORE STARTING INSERT 2, SEE FIRST KVS
357    KeyValueScanner s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
358    assertScannerResults(s, new KeyValue[]{kv11, kv12});
359
360    // START INSERT 2: Write both columns val2
361    w = mvcc.begin();
362    KeyValue kv21 = new KeyValue(row, f, q1, v2);
363    kv21.setSequenceId(w.getWriteNumber());
364    memstore.add(kv21);
365
366    KeyValue kv22 = new KeyValue(row, f, q2, v2);
367    kv22.setSequenceId(w.getWriteNumber());
368    memstore.add(kv22);
369
370    // BEFORE COMPLETING INSERT 2, SEE FIRST KVS
371    s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
372    assertScannerResults(s, new KeyValue[]{kv11, kv12});
373
374    // COMPLETE INSERT 2
375    mvcc.completeAndWait(w);
376
377    // NOW SHOULD SEE NEW KVS IN ADDITION TO OLD KVS.
378    // See HBASE-1485 for discussion about what we should do with
379    // the duplicate-TS inserts
380    s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
381    assertScannerResults(s, new KeyValue[]{kv21, kv11, kv22, kv12});
382  }

433  private static class ReadOwnWritesTester extends Thread {
434    static final int NUM_TRIES = 1000;
435
436    final byte[] row;
437
438    final byte[] f = Bytes.toBytes("family");
439    final byte[] q1 = Bytes.toBytes("q1");
440
441    final MultiVersionConcurrencyControl mvcc;
442    final MemStore memstore;
443    final AtomicLong startSeqNum;
444
445    AtomicReference<Throwable> caughtException;
446
447
448    public ReadOwnWritesTester(int id,
449                               MemStore memstore,
450                               MultiVersionConcurrencyControl mvcc,
451                               AtomicReference<Throwable> caughtException,
452                               AtomicLong startSeqNum)
453    {
454      this.mvcc = mvcc;
455      this.memstore = memstore;
456      this.caughtException = caughtException;
457      row = Bytes.toBytes(id);
458      this.startSeqNum = startSeqNum;
459    }
460
461    @Override
462    public void run() {
463      try {
464        internalRun();
465      } catch (Throwable t) {
466        caughtException.compareAndSet(null, t);
467      }
468    }
469
470    private void internalRun() throws IOException {
471      for (long i = 0; i < NUM_TRIES && caughtException.get() == null; i++) {
472        MultiVersionConcurrencyControl.WriteEntry w =
473            mvcc.begin();
474
475        // Insert the sequence value (i)
476        byte[] v = Bytes.toBytes(i);
477
478        KeyValue kv = new KeyValue(row, f, q1, i, v);
479        kv.setSequenceId(w.getWriteNumber());
480        memstore.add(kv);
481        mvcc.completeAndWait(w);
482
483        // Assert that we can read back
484        KeyValueScanner s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
485        s.seek(kv);
486
487        Cell ret = s.next();
488        assertNotNull("Didnt find own write at all", ret);
489        assertEquals("Didnt read own writes",
490                     kv.getTimestamp(), ret.getTimestamp());
491      }
492    }
493  }

495  @Test
496  public void testReadOwnWritesUnderConcurrency() throws Throwable {
497
498    int NUM_THREADS = 8;
499
500    ReadOwnWritesTester threads[] = new ReadOwnWritesTester[NUM_THREADS];
501    AtomicReference<Throwable> caught = new AtomicReference<Throwable>();

503    for (int i = 0; i < NUM_THREADS; i++) {
504      threads[i] = new ReadOwnWritesTester(i, memstore, mvcc, caught, this.startSeqNum);

507

509      threads[i].join();
510    }
511

513      throw caught.get();
514    }

806  /**
807   * Test a pathological pattern that shows why we can't currently
808   * use the MSLAB for upsert workloads. This test inserts data
809   * in the following pattern:
810   *
811   * - row0001 through row1000 (fills up one 2M Chunk)
812   * - row0002 through row1001 (fills up another 2M chunk, leaves one reference
813   *   to the first chunk
814   * - row0003 through row1002 (another chunk, another dangling reference)
815   *
816   * This causes OOME pretty quickly if we use MSLAB for upsert
817   * since each 2M chunk is held onto by a single reference.
818   */
819  @Test
820  public void testUpsertMSLAB() throws Exception {
821    Configuration conf = HBaseConfiguration.create();
822    conf.setBoolean(SegmentFactory.USEMSLAB_KEY, true);
823    memstore = new DefaultMemStore(conf, CellComparator.COMPARATOR);
824
825    int ROW_SIZE = 2048;
826    byte[] qualifier = new byte[ROW_SIZE - 4];
827
828    MemoryMXBean bean = ManagementFactory.getMemoryMXBean();
829    for (int i = 0; i < 3; i++) { System.gc(); }
830    long usageBefore = bean.getHeapMemoryUsage().getUsed();
831
832    long size = 0;
833    long ts=0;
834
835    for (int newValue = 0; newValue < 1000; newValue++) {
836      for (int row = newValue; row < newValue + 1000; row++) {
837        byte[] rowBytes = Bytes.toBytes(row);
838        size += memstore.updateColumnValue(rowBytes, FAMILY, qualifier, newValue, ++ts);
839      }
840    }
841    System.out.println("Wrote " + ts + " vals");
842    for (int i = 0; i < 3; i++) { System.gc(); }
843    long usageAfter = bean.getHeapMemoryUsage().getUsed();
844    System.out.println("Memory used: " + (usageAfter - usageBefore)
845        + " (heapsize: " + memstore.heapSize() +
846        " size: " + size + ")");
847  }
