089  @Test
090  public void testReusingChunks() {
091    MemStoreLAB mslab = new MemStoreLABImpl(conf);
092    int expectedOff = 0;
093    ByteBuffer lastBuffer = null;
094    final byte[] rk = Bytes.toBytes("r1");
095    final byte[] cf = Bytes.toBytes("f");
096    final byte[] q = Bytes.toBytes("q");
097    // Randomly allocate some bytes
098    final Random rand = ThreadLocalRandom.current();
099    for (int i = 0; i < 100; i++) {
100      int valSize = rand.nextInt(1000);
101      KeyValue kv = new KeyValue(rk, cf, q, new byte[valSize]);
102      int size = kv.getSerializedSize();
103      ByteBufferKeyValue newKv = (ByteBufferKeyValue) mslab.copyCellInto(kv);
104      if (newKv.getBuffer() != lastBuffer) {
105        expectedOff = 4;
106        lastBuffer = newKv.getBuffer();
107      }
108      assertEquals(expectedOff, newKv.getOffset());
109      assertTrue("Allocation overruns buffer",
110          newKv.getOffset() + size <= newKv.getBuffer().capacity());
111      expectedOff += size;
112    }
113    // chunks will be put back to pool after close
114    mslab.close();
115    int chunkCount = chunkCreator.getPoolSize();
116    assertTrue(chunkCount > 0);
117    // reconstruct mslab
118    mslab = new MemStoreLABImpl(conf);
119    // chunk should be got from the pool, so we can reuse it.
120    KeyValue kv = new KeyValue(rk, cf, q, new byte[10]);
121    mslab.copyCellInto(kv);
122    assertEquals(chunkCount - 1, chunkCreator.getPoolSize());
123  }

125  @Test
126  public void testPuttingBackChunksAfterFlushing() throws UnexpectedStateException {
127    byte[] row = Bytes.toBytes("testrow");
128    byte[] fam = Bytes.toBytes("testfamily");
129    byte[] qf1 = Bytes.toBytes("testqualifier1");
130    byte[] qf2 = Bytes.toBytes("testqualifier2");
131    byte[] qf3 = Bytes.toBytes("testqualifier3");
132    byte[] qf4 = Bytes.toBytes("testqualifier4");
133    byte[] qf5 = Bytes.toBytes("testqualifier5");
134    byte[] val = Bytes.toBytes("testval");
135
136    DefaultMemStore memstore = new DefaultMemStore();
137
138    // Setting up memstore
139    memstore.add(new KeyValue(row, fam, qf1, val), null);
140    memstore.add(new KeyValue(row, fam, qf2, val), null);
141    memstore.add(new KeyValue(row, fam, qf3, val), null);
142
143    // Creating a snapshot
144    MemStoreSnapshot snapshot = memstore.snapshot();
145    assertEquals(3, memstore.getSnapshot().getCellsCount());
146
147    // Adding value to "new" memstore
148    assertEquals(0, memstore.getActive().getCellsCount());
149    memstore.add(new KeyValue(row, fam, qf4, val), null);
150    memstore.add(new KeyValue(row, fam, qf5, val), null);
151    assertEquals(2, memstore.getActive().getCellsCount());
152    // close the scanner - this is how the snapshot will be used
153    for(KeyValueScanner scanner : snapshot.getScanners()) {
154      scanner.close();
155    }
156    memstore.clearSnapshot(snapshot.getId());
157
158    int chunkCount = chunkCreator.getPoolSize();
159    assertTrue(chunkCount > 0);
160
161  }

163  @Test
164  public void testPuttingBackChunksWithOpeningScanner()
165      throws IOException {
166    byte[] row = Bytes.toBytes("testrow");
167    byte[] fam = Bytes.toBytes("testfamily");
168    byte[] qf1 = Bytes.toBytes("testqualifier1");
169    byte[] qf2 = Bytes.toBytes("testqualifier2");
170    byte[] qf3 = Bytes.toBytes("testqualifier3");
171    byte[] qf4 = Bytes.toBytes("testqualifier4");
172    byte[] qf5 = Bytes.toBytes("testqualifier5");
173    byte[] qf6 = Bytes.toBytes("testqualifier6");
174    byte[] qf7 = Bytes.toBytes("testqualifier7");
175    byte[] val = Bytes.toBytes("testval");
176
177    DefaultMemStore memstore = new DefaultMemStore();
178
179    // Setting up memstore
180    memstore.add(new KeyValue(row, fam, qf1, val), null);
181    memstore.add(new KeyValue(row, fam, qf2, val), null);
182    memstore.add(new KeyValue(row, fam, qf3, val), null);
183
184    // Creating a snapshot
185    MemStoreSnapshot snapshot = memstore.snapshot();
186    assertEquals(3, memstore.getSnapshot().getCellsCount());
187
188    // Adding value to "new" memstore
189    assertEquals(0, memstore.getActive().getCellsCount());
190    memstore.add(new KeyValue(row, fam, qf4, val), null);
191    memstore.add(new KeyValue(row, fam, qf5, val), null);
192    assertEquals(2, memstore.getActive().getCellsCount());
193
194    // opening scanner before clear the snapshot
195    List<KeyValueScanner> scanners = memstore.getScanners(0);
196    // Shouldn't putting back the chunks to pool,since some scanners are opening
197    // based on their data
198    // close the snapshot scanner
199    for(KeyValueScanner scanner : snapshot.getScanners()) {
200      scanner.close();
201    }
202    memstore.clearSnapshot(snapshot.getId());
203
204    assertTrue(chunkCreator.getPoolSize() == 0);
205
206    // Chunks will be put back to pool after close scanners;
207    for (KeyValueScanner scanner : scanners) {
208      scanner.close();
209    }
210    assertTrue(chunkCreator.getPoolSize() > 0);
211
212    // clear chunks
213    chunkCreator.clearChunksInPool();
214
215    // Creating another snapshot
216    snapshot = memstore.snapshot();
217    // Adding more value
218    memstore.add(new KeyValue(row, fam, qf6, val), null);
219    memstore.add(new KeyValue(row, fam, qf7, val), null);
220    // opening scanners
221    scanners = memstore.getScanners(0);
222    // close scanners before clear the snapshot
223    for (KeyValueScanner scanner : scanners) {
224      scanner.close();
225    }
226    // Since no opening scanner, the chunks of snapshot should be put back to
227    // pool
228    // close the snapshot scanners
229    for(KeyValueScanner scanner : snapshot.getScanners()) {
230      scanner.close();
231    }
232    memstore.clearSnapshot(snapshot.getId());
233    assertTrue(chunkCreator.getPoolSize() > 0);
234  }

236  @Test
237  public void testPutbackChunksMultiThreaded() throws Exception {
238    final int maxCount = 10;
239    final int initialCount = 5;
240    final int chunkSize = 40;
241    final int valSize = 7;
242    ChunkCreator oldCreator = ChunkCreator.getInstance();
243    ChunkCreator newCreator = new ChunkCreator(chunkSize, false, 400, 1, 0.5f, null, 0);
244    assertEquals(initialCount, newCreator.getPoolSize());
245    assertEquals(maxCount, newCreator.getMaxCount());
246    // Replace the global ref with the new one we created.
247    // Used it for the testing. Later in finally we put
248    // back the original
249    ChunkCreator.instance = newCreator;
250
251    final KeyValue kv = new KeyValue(Bytes.toBytes("r"), Bytes.toBytes("f"), Bytes.toBytes("q"),
252        new byte[valSize]);
253    final AtomicReference<Throwable> exceptionRef = new AtomicReference<Throwable>();
254    try {
255      Runnable r = new Runnable() {
256        @Override
257        public void run() {
258          try {
259            MemStoreLAB memStoreLAB = new MemStoreLABImpl(conf);
260            for (int i = 0; i < maxCount; i++) {
261              // Try allocate size = chunkSize. Means every
262              // allocate call will result in a new chunk
263              memStoreLAB.copyCellInto(kv);
264            }
265            // Close MemStoreLAB so that all chunks will be tried to be put back to pool
266            memStoreLAB.close();
267          } catch (Throwable execption) {
268            exceptionRef.set(execption);
269          }
270        }
271      };
272      Thread t1 = new Thread(r);
273      Thread t2 = new Thread(r);
274      Thread t3 = new Thread(r);
275      t1.start();
276      t2.start();
277      t3.start();
278      t1.join();
279      t2.join();
280      t3.join();
281      assertTrue(exceptionRef.get() == null);
282      assertTrue(newCreator.getPoolSize() <= maxCount && newCreator.getPoolSize() > 0);
283    } finally {
284      ChunkCreator.instance = oldCreator;
285    }
286  }

288  // This test is for HBASE-26142, which throws NPE when indexChunksPool is null.
289  @Test
290  public void testNoIndexChunksPoolOrNoDataChunksPool() throws Exception {
291    final int maxCount = 10;
292    final int initialCount = 5;
293    final int newChunkSize = 40;
294    final int valSize = 7;
295
296    ChunkCreator oldCreator = ChunkCreator.getInstance();
297    try {
298      // Test dataChunksPool is not null and indexChunksPool is null
299      ChunkCreator newCreator = new ChunkCreator(newChunkSize, false, 400, 1, 0.5f, null, 0);
300      assertEquals(initialCount, newCreator.getPoolSize());
301      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
302      assertEquals(maxCount, newCreator.getMaxCount());
303      assertEquals(0, newCreator.getMaxCount(ChunkType.INDEX_CHUNK));
304      assertTrue(newCreator.getDataChunksPool() != null);
305      assertTrue(newCreator.getIndexChunksPool() == null);
306      ChunkCreator.instance = newCreator;
307      final KeyValue kv = new KeyValue(Bytes.toBytes("r"), Bytes.toBytes("f"), Bytes.toBytes("q"),
308          new byte[valSize]);
309
310      MemStoreLAB memStoreLAB = new MemStoreLABImpl(conf);
311      memStoreLAB.copyCellInto(kv);
312      memStoreLAB.close();
313      assertEquals(initialCount, newCreator.getPoolSize());
314      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
315
316      Chunk dataChunk = newCreator.getChunk();
317      assertTrue(dataChunk.isDataChunk());
318      assertTrue(dataChunk.isFromPool());
319      assertEquals(initialCount - 1, newCreator.getPoolSize());
320      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
321      newCreator.putbackChunks(Collections.singleton(dataChunk.getId()));
322      assertEquals(initialCount, newCreator.getPoolSize());
323      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
324
325      // We set ChunkCreator.indexChunkSize to 0, but we want to get a IndexChunk
326      try {
327        newCreator.getChunk(ChunkType.INDEX_CHUNK);
328        fail();
329      } catch (IllegalArgumentException e) {
330      }
331
332      Chunk jumboChunk = newCreator.getJumboChunk(newChunkSize + 10);
333      assertTrue(jumboChunk.isJumbo());
334      assertTrue(!jumboChunk.isFromPool());
335      assertEquals(initialCount, newCreator.getPoolSize());
336      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
337
338      // Test both dataChunksPool and indexChunksPool are null
339      newCreator = new ChunkCreator(newChunkSize, false, 400, 0, 0.5f, null, 0);
340      assertEquals(0, newCreator.getPoolSize());
341      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
342      assertEquals(0, newCreator.getMaxCount());
343      assertEquals(0, newCreator.getMaxCount(ChunkType.INDEX_CHUNK));
344      assertTrue(newCreator.getDataChunksPool() == null);
345      assertTrue(newCreator.getIndexChunksPool() == null);
346      ChunkCreator.instance = newCreator;
347
348      memStoreLAB = new MemStoreLABImpl(conf);
349      memStoreLAB.copyCellInto(kv);
350      memStoreLAB.close();
351      assertEquals(0, newCreator.getPoolSize());
352      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
353
354      dataChunk = newCreator.getChunk();
355      assertTrue(dataChunk.isDataChunk());
356      assertTrue(!dataChunk.isFromPool());
357      assertEquals(0, newCreator.getPoolSize());
358      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
359
360      try {
361        // We set ChunkCreator.indexChunkSize to 0, but we want to get a IndexChunk
362        newCreator.getChunk(ChunkType.INDEX_CHUNK);
363        fail();
364      } catch (IllegalArgumentException e) {
365      }
366
367      jumboChunk = newCreator.getJumboChunk(newChunkSize + 10);
368      assertTrue(jumboChunk.isJumbo());
369      assertTrue(!jumboChunk.isFromPool());
370      assertEquals(0, newCreator.getPoolSize());
371      assertEquals(0, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
372
373      // Test dataChunksPool is null and indexChunksPool is not null
374      newCreator = new ChunkCreator(newChunkSize, false, 400, 1, 0.5f, null, 1);
375      assertEquals(0, newCreator.getPoolSize());
376      assertEquals(initialCount, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
377      assertEquals(0, newCreator.getMaxCount());
378      assertEquals(maxCount, newCreator.getMaxCount(ChunkType.INDEX_CHUNK));
379      assertTrue(newCreator.getDataChunksPool() == null);
380      assertTrue(newCreator.getIndexChunksPool() != null);
381      assertEquals(newCreator.getChunkSize(ChunkType.DATA_CHUNK),
382        newCreator.getChunkSize(ChunkType.INDEX_CHUNK));
383      ChunkCreator.instance = newCreator;
384
385      memStoreLAB = new MemStoreLABImpl(conf);
386      memStoreLAB.copyCellInto(kv);
387      memStoreLAB.close();
388      assertEquals(0, newCreator.getPoolSize());
389      assertEquals(initialCount, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
390
391      dataChunk = newCreator.getChunk();
392      assertTrue(dataChunk.isDataChunk());
393      assertTrue(!dataChunk.isFromPool());
394      assertEquals(0, newCreator.getPoolSize());
395      assertEquals(initialCount, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
396
397      Chunk indexChunk =
398          newCreator.getChunk(ChunkType.INDEX_CHUNK);
399      assertEquals(0, newCreator.getPoolSize());
400      assertEquals(initialCount - 1, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
401      assertTrue(indexChunk.isIndexChunk());
402      assertTrue(indexChunk.isFromPool());
403      newCreator.putbackChunks(Collections.singleton(indexChunk.getId()));
404      assertEquals(0, newCreator.getPoolSize());
405      assertEquals(initialCount, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
406
407      jumboChunk = newCreator.getJumboChunk(newChunkSize + 10);
408      assertTrue(jumboChunk.isJumbo());
409      assertTrue(!jumboChunk.isFromPool());
410      assertEquals(0, newCreator.getPoolSize());
411      assertEquals(initialCount, newCreator.getPoolSize(ChunkType.INDEX_CHUNK));
412    } finally {
413      ChunkCreator.instance = oldCreator;
414    }
415
416    // Test both dataChunksPool and indexChunksPool are not null
417    assertTrue(ChunkCreator.getInstance().getDataChunksPool() != null);
418    assertTrue(ChunkCreator.getInstance().getIndexChunksPool() != null);
419    Chunk dataChunk = ChunkCreator.getInstance().getChunk();
420    assertTrue(dataChunk.isDataChunk());
421    assertTrue(dataChunk.isFromPool());
422    Chunk indexChunk = ChunkCreator.getInstance().getChunk(
423      ChunkType.INDEX_CHUNK);
424    assertTrue(indexChunk.isIndexChunk());
425    assertTrue(indexChunk.isFromPool());
426    Chunk jumboChunk =
427        ChunkCreator.getInstance().getJumboChunk(ChunkCreator.getInstance().getChunkSize() + 10);
428    assertTrue(jumboChunk.isJumbo());
429    assertTrue(!jumboChunk.isFromPool());
430  }
