180  public void majorCompactionWithDataBlockEncoding(boolean inCacheOnly) throws Exception {
181    Map<HStore, HFileDataBlockEncoder> replaceBlockCache = new HashMap<>();
182    for (HStore store : r.getStores()) {
183      HFileDataBlockEncoder blockEncoder = store.getDataBlockEncoder();
184      replaceBlockCache.put(store, blockEncoder);
185      final DataBlockEncoding inCache = DataBlockEncoding.PREFIX;
186      final DataBlockEncoding onDisk = inCacheOnly ? DataBlockEncoding.NONE : inCache;
187      ((HStore) store).setDataBlockEncoderInTest(new HFileDataBlockEncoderImpl(onDisk));
188    }
189
190    majorCompaction();
191
192    // restore settings
193    for (Entry<HStore, HFileDataBlockEncoder> entry : replaceBlockCache.entrySet()) {
194      ((HStore) entry.getKey()).setDataBlockEncoderInTest(entry.getValue());
195    }
196  }

198  private void majorCompaction() throws Exception {
199    createStoreFile(r);
200    for (int i = 0; i < compactionThreshold; i++) {
201      createStoreFile(r);
202    }
203    // Add more content.
204    HTestConst.addContent(new RegionAsTable(r), Bytes.toString(COLUMN_FAMILY));
205
206    // Now there are about 5 versions of each column.
207    // Default is that there only 3 (MAXVERSIONS) versions allowed per column.
208    //
209    // Assert == 3 when we ask for versions.
210    Result result = r.get(new Get(STARTROW).addFamily(COLUMN_FAMILY_TEXT).readVersions(100));
211    assertEquals(compactionThreshold, result.size());
212
213    // see if CompactionProgress is in place but null
214    for (HStore store : r.getStores()) {
215      assertNull(store.getCompactionProgress());
216    }
217
218    r.flush(true);
219    r.compact(true);
220
221    // see if CompactionProgress has done its thing on at least one store
222    int storeCount = 0;
223    for (HStore store : r.getStores()) {
224      CompactionProgress progress = store.getCompactionProgress();
225      if (progress != null) {
226        ++storeCount;
227        assertTrue(progress.currentCompactedKVs > 0);
228        assertTrue(progress.getTotalCompactingKVs() > 0);
229      }
230      assertTrue(storeCount > 0);
231    }
232
233    // look at the second row
234    // Increment the least significant character so we get to next row.
235    byte[] secondRowBytes = START_KEY_BYTES.clone();
236    secondRowBytes[START_KEY_BYTES.length - 1]++;
237
238    // Always 3 versions if that is what max versions is.
239    result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).readVersions(100));
240    LOG.debug(
241      "Row " + Bytes.toStringBinary(secondRowBytes) + " after " + "initial compaction: " + result);
242    assertEquals("Invalid number of versions of row " + Bytes.toStringBinary(secondRowBytes) + ".",
243      compactionThreshold, result.size());
244
245    // Now add deletes to memstore and then flush it.
246    // That will put us over
247    // the compaction threshold of 3 store files. Compacting these store files
248    // should result in a compacted store file that has no references to the
249    // deleted row.
250    LOG.debug("Adding deletes to memstore and flushing");
251    Delete delete = new Delete(secondRowBytes, System.currentTimeMillis());
252    byte[][] famAndQf = { COLUMN_FAMILY, null };
253    delete.addFamily(famAndQf[0]);
254    r.delete(delete);
255
256    // Assert deleted.
257    result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).readVersions(100));
258    assertTrue("Second row should have been deleted", result.isEmpty());
259
260    r.flush(true);
261
262    result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).readVersions(100));
263    assertTrue("Second row should have been deleted", result.isEmpty());
264
265    // Add a bit of data and flush. Start adding at 'bbb'.
266    createSmallerStoreFile(this.r);
267    r.flush(true);
268    // Assert that the second row is still deleted.
269    result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).readVersions(100));
270    assertTrue("Second row should still be deleted", result.isEmpty());
271
272    // Force major compaction.
273    r.compact(true);
274    assertEquals(1, r.getStore(COLUMN_FAMILY_TEXT).getStorefiles().size());
275
276    result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).readVersions(100));
277    assertTrue("Second row should still be deleted", result.isEmpty());
278
279    // Make sure the store files do have some 'aaa' keys in them -- exactly 3.
280    // Also, that compacted store files do not have any secondRowBytes because
281    // they were deleted.
282    verifyCounts(3, 0);
283
284    // Multiple versions allowed for an entry, so the delete isn't enough
285    // Lower TTL and expire to ensure that all our entries have been wiped
286    final int ttl = 1000;
287    for (HStore store : r.getStores()) {
288      ScanInfo old = store.getScanInfo();
289      ScanInfo si = old.customize(old.getMaxVersions(), ttl, old.getKeepDeletedCells());
290      store.setScanInfo(si);
291    }
292    Thread.sleep(1000);
293
294    r.compact(true);
295    int count = count();
296    assertEquals("Should not see anything after TTL has expired", 0, count);
297  }

299  @Test
300  public void testTimeBasedMajorCompaction() throws Exception {
301    // create 2 storefiles and force a major compaction to reset the time
302    int delay = 10 * 1000; // 10 sec
303    float jitterPct = 0.20f; // 20%
304    conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, delay);
305    conf.setFloat("hbase.hregion.majorcompaction.jitter", jitterPct);
306
307    HStore s = ((HStore) r.getStore(COLUMN_FAMILY));
308    s.storeEngine.getCompactionPolicy().setConf(conf);
309    try {
310      createStoreFile(r);
311      createStoreFile(r);
312      r.compact(true);
313
314      // add one more file & verify that a regular compaction won't work
315      createStoreFile(r);
316      r.compact(false);
317      assertEquals(2, s.getStorefilesCount());
318
319      // ensure that major compaction time is deterministic
320      RatioBasedCompactionPolicy c =
321        (RatioBasedCompactionPolicy) s.storeEngine.getCompactionPolicy();
322      Collection<HStoreFile> storeFiles = s.getStorefiles();
323      long mcTime = c.getNextMajorCompactTime(storeFiles);
324      for (int i = 0; i < 10; ++i) {
325        assertEquals(mcTime, c.getNextMajorCompactTime(storeFiles));
326      }
327
328      // ensure that the major compaction time is within the variance
329      long jitter = Math.round(delay * jitterPct);
330      assertTrue(delay - jitter <= mcTime && mcTime <= delay + jitter);
331
332      // wait until the time-based compaction interval
333      Thread.sleep(mcTime);
334
335      // trigger a compaction request and ensure that it's upgraded to major
336      r.compact(false);
337      assertEquals(1, s.getStorefilesCount());
338    } finally {
339      // reset the timed compaction settings
340      conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, 1000 * 60 * 60 * 24);
341      conf.setFloat("hbase.hregion.majorcompaction.jitter", 0.20F);
342      // run a major to reset the cache
343      createStoreFile(r);
344      r.compact(true);
345      assertEquals(1, s.getStorefilesCount());
346    }
347  }

349  private void verifyCounts(int countRow1, int countRow2) throws Exception {
350    int count1 = 0;
351    int count2 = 0;
352    for (HStoreFile f : r.getStore(COLUMN_FAMILY_TEXT).getStorefiles()) {
353      HFileScanner scanner = f.getReader().getScanner(false, false);
354      scanner.seekTo();
355      do {
356        byte[] row = CellUtil.cloneRow(scanner.getCell());
357        if (Bytes.equals(row, STARTROW)) {
358          count1++;
359        } else if (Bytes.equals(row, secondRowBytes)) {
360          count2++;
361        }
362      } while (scanner.next());
363    }
364    assertEquals(countRow1, count1);
365    assertEquals(countRow2, count2);
366  }

368  private int count() throws IOException {
369    int count = 0;
370    for (HStoreFile f : r.getStore(COLUMN_FAMILY_TEXT).getStorefiles()) {
371      HFileScanner scanner = f.getReader().getScanner(false, false);
372      if (!scanner.seekTo()) {
373        continue;
374      }
375      do {
376        count++;
377      } while (scanner.next());
378    }
379    return count;
380  }

436  /**
437   * Test that on a major compaction, if all cells are expired or deleted, then we'll end up with no
438   * product. Make sure scanner over region returns right answer in this case - and that it just
439   * basically works.
440   * @throws IOException
441   */
442  @Test
443  public void testMajorCompactingToNoOutputWithReverseScan() throws IOException {
444    createStoreFile(r);
445    for (int i = 0; i < compactionThreshold; i++) {
446      createStoreFile(r);
447    }
448    // Now delete everything.
449    Scan scan = new Scan();
450    scan.setReversed(true);
451    InternalScanner s = r.getScanner(scan);
452    do {
453      List<Cell> results = new ArrayList<>();
454      boolean result = s.next(results);
455      assertTrue(!results.isEmpty());
456      r.delete(new Delete(CellUtil.cloneRow(results.get(0))));
457      if (!result) {
458        break;
459      }
460    } while (true);
461    s.close();
462    // Flush
463    r.flush(true);
464    // Major compact.
465    r.compact(true);
466    scan = new Scan();
467    scan.setReversed(true);
468    s = r.getScanner(scan);
469    int counter = 0;
470    do {
471      List<Cell> results = new ArrayList<>();
472      boolean result = s.next(results);
473      if (!result) {
474        break;
475      }
476      counter++;
477    } while (true);
478    s.close();
479    assertEquals(0, counter);
480  }

482  private void testMajorCompactingWithDeletes(KeepDeletedCells keepDeletedCells)
483    throws IOException {
484    createStoreFile(r);
485    for (int i = 0; i < compactionThreshold; i++) {
486      createStoreFile(r);
487    }
488    // Now delete everything.
489    InternalScanner s = r.getScanner(new Scan());
490    int originalCount = 0;
491    do {
492      List<Cell> results = new ArrayList<>();
493      boolean result = s.next(results);
494      r.delete(new Delete(CellUtil.cloneRow(results.get(0))));
495      if (!result) break;
496      originalCount++;
497    } while (true);
498    s.close();
499    // Flush
500    r.flush(true);
501
502    for (HStore store : this.r.stores.values()) {
503      ScanInfo old = store.getScanInfo();
504      ScanInfo si = old.customize(old.getMaxVersions(), old.getTtl(), keepDeletedCells);
505      store.setScanInfo(si);
506    }
507    // Major compact.
508    r.compact(true);
509    s = r.getScanner(new Scan().setRaw(true));
510    int counter = 0;
511    do {
512      List<Cell> results = new ArrayList<>();
513      boolean result = s.next(results);
514      if (!result) break;
515      counter++;
516    } while (true);
517    assertEquals(keepDeletedCells == KeepDeletedCells.TRUE ? originalCount : 0, counter);
