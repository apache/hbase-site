130  /**
131   * Verify that you can stop a long-running compaction
132   * (used during RS shutdown)
133   * @throws Exception
134   */
135  @Test
136  public void testInterruptCompaction() throws Exception {
137    assertEquals(0, count());
138
139    // lower the polling interval for this test
140    int origWI = HStore.closeCheckInterval;
141    HStore.closeCheckInterval = 10*1000; // 10 KB
142
143    try {
144      // Create a couple store files w/ 15KB (over 10KB interval)
145      int jmax = (int) Math.ceil(15.0/compactionThreshold);
146      byte [] pad = new byte[1000]; // 1 KB chunk
147      for (int i = 0; i < compactionThreshold; i++) {
148        Table loader = new RegionAsTable(r);
149        Put p = new Put(Bytes.add(STARTROW, Bytes.toBytes(i)));
150        p.setDurability(Durability.SKIP_WAL);
151        for (int j = 0; j < jmax; j++) {
152          p.addColumn(COLUMN_FAMILY, Bytes.toBytes(j), pad);
153        }
154        HBaseTestCase.addContent(loader, Bytes.toString(COLUMN_FAMILY));
155        loader.put(p);
156        r.flush(true);
157      }
158
159      HRegion spyR = spy(r);
160      doAnswer(new Answer() {
161        @Override
162        public Object answer(InvocationOnMock invocation) throws Throwable {
163          r.writestate.writesEnabled = false;
164          return invocation.callRealMethod();
165        }
166      }).when(spyR).doRegionCompactionPrep();
167
168      // force a minor compaction, but not before requesting a stop
169      spyR.compactStores();
170
171      // ensure that the compaction stopped, all old files are intact,
172      Store s = r.stores.get(COLUMN_FAMILY);
173      assertEquals(compactionThreshold, s.getStorefilesCount());
174      assertTrue(s.getStorefilesSize() > 15*1000);
175      // and no new store files persisted past compactStores()
176      FileStatus[] ls = r.getFilesystem().listStatus(r.getRegionFileSystem().getTempDir());
177      assertEquals(0, ls.length);
178
179    } finally {
180      // don't mess up future tests
181      r.writestate.writesEnabled = true;
182      HStore.closeCheckInterval = origWI;
183
184      // Delete all Store information once done using
185      for (int i = 0; i < compactionThreshold; i++) {
186        Delete delete = new Delete(Bytes.add(STARTROW, Bytes.toBytes(i)));
187        byte [][] famAndQf = {COLUMN_FAMILY, null};
188        delete.addFamily(famAndQf[0]);
189        r.delete(delete);
190      }
191      r.flush(true);
192
193      // Multiple versions allowed for an entry, so the delete isn't enough
194      // Lower TTL and expire to ensure that all our entries have been wiped
195      final int ttl = 1000;
196      for (Store hstore: this.r.stores.values()) {
197        HStore store = (HStore)hstore;
198        ScanInfo old = store.getScanInfo();
199        ScanInfo si = new ScanInfo(old.getConfiguration(), old.getFamily(),
200            old.getMinVersions(), old.getMaxVersions(), ttl,
201            old.getKeepDeletedCells(), 0, old.getComparator());
202        store.setScanInfo(si);
203      }
204      Thread.sleep(ttl);
205
206      r.compact(true);
207      assertEquals(0, count());
208    }
209  }

211  private int count() throws IOException {
212    int count = 0;
213    for (StoreFile f: this.r.stores.
214        get(COLUMN_FAMILY_TEXT).getStorefiles()) {
215      HFileScanner scanner = f.getReader().getScanner(false, false);
216      if (!scanner.seekTo()) {
217        continue;
218      }
219      do {
220        count++;
221      } while(scanner.next());
222    }
223    return count;
224  }

236  @Test
237  public void testCompactionWithCorruptResult() throws Exception {
238    int nfiles = 10;
239    for (int i = 0; i < nfiles; i++) {
240      createStoreFile(r);
241    }
242    HStore store = (HStore) r.getStore(COLUMN_FAMILY);
243
244    Collection<StoreFile> storeFiles = store.getStorefiles();
245    DefaultCompactor tool = (DefaultCompactor)store.storeEngine.getCompactor();
246    tool.compactForTesting(storeFiles, false);
247
248    // Now lets corrupt the compacted file.
249    FileSystem fs = store.getFileSystem();
250    // default compaction policy created one and only one new compacted file
251    Path dstPath = store.getRegionFileSystem().createTempName();
252    FSDataOutputStream stream = fs.create(dstPath, null, true, 512, (short)3, (long)1024, null);
253    stream.writeChars("CORRUPT FILE!!!!");
254    stream.close();
255    Path origPath = store.getRegionFileSystem().commitStoreFile(
256      Bytes.toString(COLUMN_FAMILY), dstPath);
257
258    try {
259      ((HStore)store).moveFileIntoPlace(origPath);
260    } catch (Exception e) {
261      // The complete compaction should fail and the corrupt file should remain
262      // in the 'tmp' directory;
263      assert (fs.exists(origPath));
264      assert (!fs.exists(dstPath));
265      System.out.println("testCompactionWithCorruptResult Passed");
266      return;
267    }
268    fail("testCompactionWithCorruptResult failed since no exception was" +
269        "thrown while completing a corrupt file");
270  }

337  private class StoreMockMaker extends StatefulStoreMockMaker {
338    public ArrayList<StoreFile> compacting = new ArrayList<StoreFile>();
339    public ArrayList<StoreFile> notCompacting = new ArrayList<StoreFile>();
340    private ArrayList<Integer> results;

342    public StoreMockMaker(ArrayList<Integer> results) {
343      this.results = results;

346    public class TestCompactionContext extends CompactionContext {
347      private List<StoreFile> selectedFiles;
348      public TestCompactionContext(List<StoreFile> selectedFiles) {
349        super();
350        this.selectedFiles = selectedFiles;
351      }
352
353      @Override
354      public List<StoreFile> preSelect(List<StoreFile> filesCompacting) {
355        return new ArrayList<StoreFile>();
356      }
357
358      @Override
359      public boolean select(List<StoreFile> filesCompacting, boolean isUserCompaction,
360          boolean mayUseOffPeak, boolean forceMajor) throws IOException {
361        this.request = new CompactionRequest(selectedFiles);
362        this.request.setPriority(getPriority());
363        return true;
364      }
365
366      @Override
367      public List<Path> compact(ThroughputController throughputController, User user)
368          throws IOException {
369        finishCompaction(this.selectedFiles);
370        return new ArrayList<Path>();
371      }
372    }
373
374    @Override
375    public synchronized CompactionContext selectCompaction() {
376      CompactionContext ctx = new TestCompactionContext(new ArrayList<StoreFile>(notCompacting));
377      compacting.addAll(notCompacting);
378      notCompacting.clear();
379      try {
380        ctx.select(null, false, false, false);
381      } catch (IOException ex) {
382        fail("Shouldn't happen");
383      }
384      return ctx;
385    }
386
387    @Override
388    public synchronized void cancelCompaction(Object object) {
389      TestCompactionContext ctx = (TestCompactionContext)object;
390      compacting.removeAll(ctx.selectedFiles);
391      notCompacting.addAll(ctx.selectedFiles);
392    }
393
394    public synchronized void finishCompaction(List<StoreFile> sfs) {
395      if (sfs.isEmpty()) return;
396      synchronized (results) {
397        results.add(sfs.size());
398      }
399      compacting.removeAll(sfs);
400    }
401
402    @Override
403    public int getPriority() {
404      return 7 - compacting.size() - notCompacting.size();
405    }

464    public BlockingCompactionContext waitForBlocking() {

486  /** Test compaction priority management and multiple compactions per store (HBASE-8665). */
487  @Test
488  public void testCompactionQueuePriorities() throws Exception {
489    // Setup a compact/split thread on a mock server.
490    final Configuration conf = HBaseConfiguration.create();
491    HRegionServer mockServer = mock(HRegionServer.class);
492    when(mockServer.isStopped()).thenReturn(false);
493    when(mockServer.getConfiguration()).thenReturn(conf);
494    when(mockServer.getChoreService()).thenReturn(new ChoreService("test"));
495    CompactSplitThread cst = new CompactSplitThread(mockServer);
496    when(mockServer.getCompactSplitThread()).thenReturn(cst);
497    //prevent large compaction thread pool stealing job from small compaction queue.
498    cst.shutdownLongCompactions();
499    // Set up the region mock that redirects compactions.
500    HRegion r = mock(HRegion.class);
501    when(
502      r.compact(any(CompactionContext.class), any(Store.class),
503        any(ThroughputController.class), any(User.class))).then(new Answer<Boolean>() {
504      @Override
505      public Boolean answer(InvocationOnMock invocation) throws Throwable {
506        invocation.getArgumentAt(0, CompactionContext.class).compact(
507          invocation.getArgumentAt(2, ThroughputController.class), null);
508        return true;
509      }
510    });
511
512    // Set up store mocks for 2 "real" stores and the one we use for blocking CST.
513    ArrayList<Integer> results = new ArrayList<Integer>();
514    StoreMockMaker sm = new StoreMockMaker(results), sm2 = new StoreMockMaker(results);
515    Store store = sm.createStoreMock("store1"), store2 = sm2.createStoreMock("store2");
516    BlockingStoreMockMaker blocker = new BlockingStoreMockMaker();
517
518    // First, block the compaction thread so that we could muck with queue.
519    cst.requestSystemCompaction(r, blocker.createStoreMock(1, "b-pri1"), "b-pri1");
520    BlockingStoreMockMaker.BlockingCompactionContext currentBlock = blocker.waitForBlocking();
521
522    // Add 4 files to store1, 3 to store2, and queue compactions; pri 3 and 4 respectively.
523    for (int i = 0; i < 4; ++i) {
524      sm.notCompacting.add(createFile());
525    }
526    cst.requestSystemCompaction(r, store, "s1-pri3");
527    for (int i = 0; i < 3; ++i) {
528      sm2.notCompacting.add(createFile());
529    }
530    cst.requestSystemCompaction(r, store2, "s2-pri4");
531    // Now add 2 more files to store1 and queue compaction - pri 1.
532    for (int i = 0; i < 2; ++i) {
533      sm.notCompacting.add(createFile());
534    }
535    cst.requestSystemCompaction(r, store, "s1-pri1");
536    // Finally add blocking compaction with priority 2.
537    cst.requestSystemCompaction(r, blocker.createStoreMock(2, "b-pri2"), "b-pri2");
538
539    // Unblock the blocking compaction; we should run pri1 and become block again in pri2.
540    currentBlock.unblock();
541    currentBlock = blocker.waitForBlocking();
542    // Pri1 should have "compacted" all 6 files.
543    assertEquals(1, results.size());
544    assertEquals(6, results.get(0).intValue());
545    // Add 2 files to store 1 (it has 2 files now).
546    for (int i = 0; i < 2; ++i) {
547      sm.notCompacting.add(createFile());
548    }
549    // Now we have pri4 for store 2 in queue, and pri3 for store1; store1's current priority
550    // is 5, however, so it must not preempt store 2. Add blocking compaction at the end.
551    cst.requestSystemCompaction(r, blocker.createStoreMock(7, "b-pri7"), "b-pri7");
552    currentBlock.unblock();
553    currentBlock = blocker.waitForBlocking();
554    assertEquals(3, results.size());
555    assertEquals(3, results.get(1).intValue()); // 3 files should go before 2 files.
556    assertEquals(2, results.get(2).intValue());
557
558    currentBlock.unblock();
559    cst.interruptIfNecessary();
560  }

571  /**
572   * Simple {@link CompactionRequest} on which you can wait until the requested compaction finishes.
573   */
574  public static class TrackableCompactionRequest extends CompactionRequest {
575    private CountDownLatch done;
576
577    /**
578     * Constructor for a custom compaction. Uses the setXXX methods to update the state of the
579     * compaction before being used.
580     */
581    public TrackableCompactionRequest(CountDownLatch finished) {
582      super();
583      this.done = finished;
584    }
585
586    @Override
587    public void afterExecute() {
588      super.afterExecute();
589      this.done.countDown();
590    }
591  }
