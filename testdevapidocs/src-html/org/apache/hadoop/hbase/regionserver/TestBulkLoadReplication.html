093/**
094 * Integration test for bulk load replication. Defines three clusters, with the following
095 * replication topology: "1 <-> 2 <-> 3" (active-active between 1 and 2, and active-active between
096 * 2 and 3).
097 *
098 * For each of defined test clusters, it performs a bulk load, asserting values on bulk loaded file
099 * gets replicated to other two peers. Since we are doing 3 bulk loads, with the given replication
100 * topology all these bulk loads should get replicated only once on each peer. To assert this,
101 * this test defines a preBulkLoad coprocessor and adds it to all test table regions, on each of the
102 * clusters. This CP counts the amount of times bulk load actually gets invoked, certifying
103 * we are not entering the infinite loop condition addressed by HBASE-22380.
104 */
105@Category({ ReplicationTests.class, MediumTests.class})
106public class TestBulkLoadReplication extends TestReplicationBase {

149    LOG.info("Setup Zk to same one from UTIL1 and UTIL2");
150    UTIL3.setZkCluster(UTIL1.getZkCluster());
151    UTIL3.startMiniCluster(NUM_SLAVES1);

153    TableDescriptor table = TableDescriptorBuilder.newBuilder(tableName)
154      .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(famName)
155        .setMobEnabled(true)
156        .setMobThreshold(4000)
157        .setScope(HConstants.REPLICATION_SCOPE_GLOBAL).build())
158      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(noRepfamName)).build();
159
160    Connection connection3 = ConnectionFactory.createConnection(CONF3);
161    try (Admin admin3 = connection3.getAdmin()) {

163    }
164    UTIL3.waitUntilAllRegionsAssigned(tableName);
165    htable3 = connection3.getTable(tableName);

170  public void setUpBase() throws Exception {
171    //"super.setUpBase()" already sets replication from 1->2,
172    //then on the subsequent lines, sets 2->1, 2->3 and 3->2.

174    super.setUpBase();

180    //adds cluster3 as a remote peer on cluster2
181    UTIL2.getAdmin().addReplicationPeer(PEER_ID3, peer3Config);
182    //adds cluster2 as a remote peer on cluster3
183    UTIL3.getAdmin().addReplicationPeer(PEER_ID2, peer2Config);
184    setupCoprocessor(UTIL1, "cluster1");
185    setupCoprocessor(UTIL2, "cluster2");
186    setupCoprocessor(UTIL3, "cluster3");
187    BULK_LOADS_COUNT = new AtomicInteger(0);

195  private void setupCoprocessor(HBaseTestingUtility cluster, String name){
196    cluster.getHBaseCluster().getRegions(tableName).forEach(r -> {
197      try {
198        TestBulkLoadReplication.BulkReplicationTestObserver cp = r.getCoprocessorHost().
199          findCoprocessor(TestBulkLoadReplication.BulkReplicationTestObserver.class);
200        if(cp == null) {
201          r.getCoprocessorHost().
202            load(TestBulkLoadReplication.BulkReplicationTestObserver.class, 0,
203              cluster.getConfiguration());
204          cp = r.getCoprocessorHost().
205            findCoprocessor(TestBulkLoadReplication.BulkReplicationTestObserver.class);
206          cp.clusterName = cluster.getClusterKey();
207        }
208      } catch (Exception e){
209        LOG.error(e.getMessage(), e);
210      }
211    });
212  }

223  private static void setupBulkLoadConfigsForCluster(Configuration config,
224    String clusterReplicationId) throws Exception {

234  @Test
235  public void testBulkLoadReplicationActiveActive() throws Exception {
236    Table peer1TestTable = UTIL1.getConnection().getTable(TestReplicationBase.tableName);
237    Table peer2TestTable = UTIL2.getConnection().getTable(TestReplicationBase.tableName);
238    Table peer3TestTable = UTIL3.getConnection().getTable(TestReplicationBase.tableName);
239    byte[] row = Bytes.toBytes("001");
240    byte[] value = Bytes.toBytes("v1");
241    assertBulkLoadConditions(row, value, UTIL1, peer1TestTable, peer2TestTable, peer3TestTable);
242    row = Bytes.toBytes("002");
243    value = Bytes.toBytes("v2");
244    assertBulkLoadConditions(row, value, UTIL2, peer1TestTable, peer2TestTable, peer3TestTable);
245    row = Bytes.toBytes("003");
246    value = Bytes.toBytes("v3");
247    assertBulkLoadConditions(row, value, UTIL3, peer1TestTable, peer2TestTable, peer3TestTable);
248    //Additional wait to make sure no extra bulk load happens
249    Thread.sleep(400);
250    //We have 3 bulk load events (1 initiated on each cluster).
251    //Each event gets 3 counts (the originator cluster, plus the two peers),
252    //so BULK_LOADS_COUNT expected value is 3 * 3 = 9.
253    assertEquals(9, BULK_LOADS_COUNT.get());
254  }

256  @Test
257  public void testPartionedMOBCompactionBulkLoadDoesntReplicate() throws Exception {
258    Path path = createMobFiles(UTIL3);
259    ColumnFamilyDescriptor descriptor =
260      new ColumnFamilyDescriptorBuilder.ModifyableColumnFamilyDescriptor(famName);
261    ExecutorService pool = null;
262    try {
263      pool = Executors.newFixedThreadPool(1);
264      PartitionedMobCompactor compactor =
265        new PartitionedMobCompactor(UTIL3.getConfiguration(), UTIL3.getTestFileSystem(), tableName,
266          descriptor, pool);
267      BULK_LOAD_LATCH = new CountDownLatch(1);
268      BULK_LOADS_COUNT.set(0);
269      compactor.compact(Arrays.asList(UTIL3.getTestFileSystem().listStatus(path)), true);
270      assertTrue(BULK_LOAD_LATCH.await(1, TimeUnit.SECONDS));
271      Thread.sleep(400);
272      assertEquals(1, BULK_LOADS_COUNT.get());
273    } finally {
274      if(pool != null && !pool.isTerminated()) {
275        pool.shutdownNow();
276      }
277    }
278  }

281  private void assertBulkLoadConditions(byte[] row, byte[] value,
282      HBaseTestingUtility utility, Table...tables) throws Exception {
283    BULK_LOAD_LATCH = new CountDownLatch(3);
284    bulkLoadOnCluster(row, value, utility);
285    assertTrue(BULK_LOAD_LATCH.await(1, TimeUnit.MINUTES));
286    assertTableHasValue(tables[0], row, value);
287    assertTableHasValue(tables[1], row, value);
288    assertTableHasValue(tables[2], row, value);
289  }

291  private void bulkLoadOnCluster(byte[] row, byte[] value,
292      HBaseTestingUtility cluster) throws Exception {
293    String bulkLoadFilePath = createHFileForFamilies(row, value, cluster.getConfiguration());
294    copyToHdfs(bulkLoadFilePath, cluster.getDFSCluster());
295    BulkLoadHFilesTool bulkLoadHFilesTool = new BulkLoadHFilesTool(cluster.getConfiguration());
296    bulkLoadHFilesTool.bulkLoad(tableName, BULK_LOAD_BASE_DIR);
297  }
298
299  private void copyToHdfs(String bulkLoadFilePath, MiniDFSCluster cluster) throws Exception {
300    Path bulkLoadDir = new Path(BULK_LOAD_BASE_DIR, "f");
301    cluster.getFileSystem().mkdirs(bulkLoadDir);
302    cluster.getFileSystem().copyFromLocalFile(new Path(bulkLoadFilePath), bulkLoadDir);
303  }

305  private void assertTableHasValue(Table table, byte[] row, byte[] value) throws Exception {
306    Get get = new Get(row);
307    Result result = table.get(get);
308    assertTrue(result.advance());
309    assertEquals(Bytes.toString(value), Bytes.toString(result.value()));
310  }

312  private String createHFileForFamilies(byte[] row, byte[] value,
313      Configuration clusterConfig) throws IOException {
314    CellBuilder cellBuilder = CellBuilderFactory.create(CellBuilderType.DEEP_COPY);
315    cellBuilder.setRow(row)
316      .setFamily(TestReplicationBase.famName)
317      .setQualifier(Bytes.toBytes("1"))
318      .setValue(value)
319      .setType(Cell.Type.Put);
320
321    HFile.WriterFactory hFileFactory = HFile.getWriterFactoryNoCache(clusterConfig);
322    // TODO We need a way to do this without creating files
323    File hFileLocation = testFolder.newFile();
324    FSDataOutputStream out =
325      new FSDataOutputStream(new FileOutputStream(hFileLocation), null);
326    try {
327      hFileFactory.withOutputStream(out);
328      hFileFactory.withFileContext(new HFileContext());
329      HFile.Writer writer = hFileFactory.create();
330      try {
331        writer.append(new KeyValue(cellBuilder.build()));
332      } finally {
333        writer.close();
334      }
335    } finally {
336      out.close();
337    }
338    return hFileLocation.getAbsoluteFile().getAbsolutePath();
339  }

341  private Path createMobFiles(HBaseTestingUtility util) throws IOException {
342    Path testDir = FSUtils.getRootDir(util.getConfiguration());
343    Path mobTestDir = new Path(testDir, MobConstants.MOB_DIR_NAME);
344    Path basePath = new Path(new Path(mobTestDir, tableName.getNameAsString()), "f");
345    HFileContext meta = new HFileContextBuilder().withBlockSize(8 * 1024).build();
346    MobFileName mobFileName = null;
347    byte[] mobFileStartRow = new byte[32];
348    for (byte rowKey : Bytes.toBytes("01234")) {
349      mobFileName = MobFileName.create(mobFileStartRow, MobUtils.formatDate(new Date()),
350        UUID.randomUUID().toString().replaceAll("-", ""));
351      StoreFileWriter mobFileWriter =
352        new StoreFileWriter.Builder(util.getConfiguration(),
353          new CacheConfig(util.getConfiguration()), util.getTestFileSystem()).withFileContext(meta)
354          .withFilePath(new Path(basePath, mobFileName.getFileName())).build();
355      long now = System.currentTimeMillis();
356      try {
357        for (int i = 0; i < 10; i++) {
358          byte[] key = Bytes.add(Bytes.toBytes(rowKey), Bytes.toBytes(i));
359          byte[] dummyData = new byte[5000];
360          new Random().nextBytes(dummyData);
361          mobFileWriter.append(
362            new KeyValue(key, famName, Bytes.toBytes("1"), now, KeyValue.Type.Put, dummyData));
363        }
364      } finally {
365        mobFileWriter.close();
366      }
367    }
368    return basePath;
369  }

371  public static class BulkReplicationTestObserver implements RegionCoprocessor {
372
373    String clusterName;
374    AtomicInteger bulkLoadCounts = new AtomicInteger();
375
376    @Override
377    public Optional<RegionObserver> getRegionObserver() {
378      return Optional.of(new RegionObserver() {
379
380        @Override
381        public void postBulkLoadHFile(ObserverContext<RegionCoprocessorEnvironment> ctx,
382          List<Pair<byte[], String>> stagingFamilyPaths, Map<byte[], List<Path>> finalPaths)
383            throws IOException {
384          BULK_LOAD_LATCH.countDown();
385          BULK_LOADS_COUNT.incrementAndGet();
386          LOG.debug("Another file bulk loaded. Total for {}: {}", clusterName,
387            bulkLoadCounts.addAndGet(1));
388        }
389      });
390    }
391  }
