043@Category(MediumTests.class)
044public class TestDeleteMobTable {
045
046  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
047  private final static byte[] FAMILY = Bytes.toBytes("family");
048  private final static byte[] QF = Bytes.toBytes("qualifier");
049  private static Random random = new Random();
050
051  @BeforeClass
052  public static void setUpBeforeClass() throws Exception {
053    TEST_UTIL.startMiniCluster(1);
054  }
055
056  @AfterClass
057  public static void tearDownAfterClass() throws Exception {
058    TEST_UTIL.shutdownMiniCluster();
059  }
060
061  /**
062   * Generate the mob value.
063   *
064   * @param size
065   *          the size of the value
066   * @return the mob value generated
067   */
068  private static byte[] generateMobValue(int size) {
069    byte[] mobVal = new byte[size];
070    random.nextBytes(mobVal);
071    return mobVal;
072  }
073
074  private HTableDescriptor createTableDescriptor(TableName tableName, boolean hasMob) {
075    HTableDescriptor htd = new HTableDescriptor(tableName);
076    HColumnDescriptor hcd = new HColumnDescriptor(FAMILY);
077    if (hasMob) {
078      hcd.setMobEnabled(true);
079      hcd.setMobThreshold(0);
080    }
081    htd.addFamily(hcd);
082    return htd;
083  }
084
085  private Table createTableWithOneFile(HTableDescriptor htd) throws IOException {
086    Table table = TEST_UTIL.createTable(htd, null);
087    try {
088      // insert data
089      byte[] value = generateMobValue(10);
090      byte[] row = Bytes.toBytes("row");
091      Put put = new Put(row);
092      put.addColumn(FAMILY, QF, EnvironmentEdgeManager.currentTime(), value);
093      table.put(put);
094
095      // create an hfile
096      TEST_UTIL.getAdmin().flush(htd.getTableName());
097    } catch (IOException e) {
098      table.close();
099      throw e;
100    }
101    return table;
102  }
103
104  @Test
105  public void testDeleteMobTable() throws Exception {
106    TableName tn = TableName.valueOf("testDeleteMobTable");
107    HTableDescriptor htd = createTableDescriptor(tn, true);
108    HColumnDescriptor hcd = htd.getFamily(FAMILY);
109
110    String fileName = null;
111    Table table = createTableWithOneFile(htd);
112    try {
113      // the mob file exists
114      Assert.assertEquals(1, countMobFiles(tn, hcd.getNameAsString()));
115      Assert.assertEquals(0, countArchiveMobFiles(tn, hcd.getNameAsString()));
116      fileName = assertHasOneMobRow(table, tn, hcd.getNameAsString());
117      Assert.assertFalse(mobArchiveExist(tn, hcd.getNameAsString(), fileName));
118      Assert.assertTrue(mobTableDirExist(tn));
119    } finally {
120      table.close();
121      TEST_UTIL.deleteTable(tn);
122    }
123
124    Assert.assertFalse(TEST_UTIL.getAdmin().tableExists(tn));
125    Assert.assertEquals(0, countMobFiles(tn, hcd.getNameAsString()));
126    Assert.assertEquals(1, countArchiveMobFiles(tn, hcd.getNameAsString()));
127    Assert.assertTrue(mobArchiveExist(tn, hcd.getNameAsString(), fileName));
128    Assert.assertFalse(mobTableDirExist(tn));
129  }
130
131  @Test
132  public void testDeleteNonMobTable() throws Exception {
133    TableName tn = TableName.valueOf("testDeleteNonMobTable");
134    HTableDescriptor htd = createTableDescriptor(tn, false);
135    HColumnDescriptor hcd = htd.getFamily(FAMILY);
136
137    Table table = createTableWithOneFile(htd);
138    try {
139      // the mob file doesn't exist
140      Assert.assertEquals(0, countMobFiles(tn, hcd.getNameAsString()));
141      Assert.assertEquals(0, countArchiveMobFiles(tn, hcd.getNameAsString()));
142      Assert.assertFalse(mobTableDirExist(tn));
143    } finally {
144      table.close();
145      TEST_UTIL.deleteTable(tn);
146    }
147
148    Assert.assertFalse(TEST_UTIL.getAdmin().tableExists(tn));
149    Assert.assertEquals(0, countMobFiles(tn, hcd.getNameAsString()));
150    Assert.assertEquals(0, countArchiveMobFiles(tn, hcd.getNameAsString()));
151    Assert.assertFalse(mobTableDirExist(tn));
152  }
153
154  @Test
155  public void testMobFamilyDelete() throws Exception {
156    TableName tn = TableName.valueOf("testMobFamilyDelete");
157    HTableDescriptor htd = createTableDescriptor(tn, true);
158    HColumnDescriptor hcd = htd.getFamily(FAMILY);
159    htd.addFamily(new HColumnDescriptor(Bytes.toBytes("family2")));
160
161    Table table = createTableWithOneFile(htd);
162    try {
163      // the mob file exists
164      Assert.assertEquals(1, countMobFiles(tn, hcd.getNameAsString()));
165      Assert.assertEquals(0, countArchiveMobFiles(tn, hcd.getNameAsString()));
166      String fileName = assertHasOneMobRow(table, tn, hcd.getNameAsString());
167      Assert.assertFalse(mobArchiveExist(tn, hcd.getNameAsString(), fileName));
168      Assert.assertTrue(mobTableDirExist(tn));
169
170      TEST_UTIL.getAdmin().deleteColumnFamily(tn, FAMILY);
171
172      Assert.assertEquals(0, countMobFiles(tn, hcd.getNameAsString()));
173      Assert.assertEquals(1, countArchiveMobFiles(tn, hcd.getNameAsString()));
174      Assert.assertTrue(mobArchiveExist(tn, hcd.getNameAsString(), fileName));
175      Assert.assertFalse(mobColumnFamilyDirExist(tn, hcd.getNameAsString()));
176    } finally {
177      table.close();
178      TEST_UTIL.deleteTable(tn);
179    }
180  }
181
182  private int countMobFiles(TableName tn, String familyName) throws IOException {
183    FileSystem fs = TEST_UTIL.getTestFileSystem();
184    Path mobFileDir = MobUtils.getMobFamilyPath(TEST_UTIL.getConfiguration(), tn, familyName);
185    if (fs.exists(mobFileDir)) {
186      return fs.listStatus(mobFileDir).length;
187    }
188    return 0;
189  }
190
191  private int countArchiveMobFiles(TableName tn, String familyName)
192      throws IOException {
193    FileSystem fs = TEST_UTIL.getTestFileSystem();
194    Path storePath = HFileArchiveUtil.getStoreArchivePath(TEST_UTIL.getConfiguration(), tn,
195        MobUtils.getMobRegionInfo(tn).getEncodedName(), familyName);
196    if (fs.exists(storePath)) {
197      return fs.listStatus(storePath).length;
198    }
199    return 0;
200  }
201
202  private boolean mobTableDirExist(TableName tn) throws IOException {
203    FileSystem fs = TEST_UTIL.getTestFileSystem();
204    Path tableDir = FSUtils.getTableDir(MobUtils.getMobHome(TEST_UTIL.getConfiguration()), tn);
205    return fs.exists(tableDir);
206  }
207
208  private boolean mobColumnFamilyDirExist(TableName tn, String familyName) throws IOException {
209    FileSystem fs = TEST_UTIL.getTestFileSystem();
210    Path mobFamilyDir = MobUtils.getMobFamilyPath(TEST_UTIL.getConfiguration(), tn, familyName);
211    return fs.exists(mobFamilyDir);
212  }
213
214  private boolean mobArchiveExist(TableName tn, String familyName, String fileName)
215      throws IOException {
216    FileSystem fs = TEST_UTIL.getTestFileSystem();
217    Path storePath = HFileArchiveUtil.getStoreArchivePath(TEST_UTIL.getConfiguration(), tn,
218        MobUtils.getMobRegionInfo(tn).getEncodedName(), familyName);
219    return fs.exists(new Path(storePath, fileName));
220  }
221
222  private String assertHasOneMobRow(Table table, TableName tn, String familyName)
223      throws IOException {
224    Scan scan = new Scan();
225    scan.setAttribute(MobConstants.MOB_SCAN_RAW, Bytes.toBytes(Boolean.TRUE));
226    ResultScanner rs = table.getScanner(scan);
227    Result r = rs.next();
228    Assert.assertNotNull(r);
229    String fileName = MobUtils.getMobFileName(r.getColumnLatestCell(FAMILY, QF));
230    Path filePath = new Path(
231        MobUtils.getMobFamilyPath(TEST_UTIL.getConfiguration(), tn, familyName), fileName);
232    FileSystem fs = TEST_UTIL.getTestFileSystem();
233    Assert.assertTrue(fs.exists(filePath));
234    r = rs.next();
235    Assert.assertNull(r);
236    return fileName;
237  }
