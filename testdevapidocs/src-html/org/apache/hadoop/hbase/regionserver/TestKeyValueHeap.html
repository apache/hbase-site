039@Category({RegionServerTests.class, SmallTests.class})
040public class TestKeyValueHeap extends HBaseTestCase {
041  private byte[] row1 = Bytes.toBytes("row1");
042  private byte[] fam1 = Bytes.toBytes("fam1");
043  private byte[] col1 = Bytes.toBytes("col1");
044  private byte[] data = Bytes.toBytes("data");
045
046  private byte[] row2 = Bytes.toBytes("row2");
047  private byte[] fam2 = Bytes.toBytes("fam2");
048  private byte[] col2 = Bytes.toBytes("col2");
049
050  private byte[] col3 = Bytes.toBytes("col3");
051  private byte[] col4 = Bytes.toBytes("col4");
052  private byte[] col5 = Bytes.toBytes("col5");
053
054  // Variable name encoding. kv<row#><fam#><col#>
055  Cell kv111 = new KeyValue(row1, fam1, col1, data);
056  Cell kv112 = new KeyValue(row1, fam1, col2, data);
057  Cell kv113 = new KeyValue(row1, fam1, col3, data);
058  Cell kv114 = new KeyValue(row1, fam1, col4, data);
059  Cell kv115 = new KeyValue(row1, fam1, col5, data);
060  Cell kv121 = new KeyValue(row1, fam2, col1, data);
061  Cell kv122 = new KeyValue(row1, fam2, col2, data);
062  Cell kv211 = new KeyValue(row2, fam1, col1, data);
063  Cell kv212 = new KeyValue(row2, fam1, col2, data);
064  Cell kv213 = new KeyValue(row2, fam1, col3, data);
065
066  TestScanner s1 = new TestScanner(Arrays.asList(kv115, kv211, kv212));
067  TestScanner s2 = new TestScanner(Arrays.asList(kv111, kv112));
068  TestScanner s3 = new TestScanner(Arrays.asList(kv113, kv114, kv121, kv122, kv213));
069
070  List<KeyValueScanner> scanners = new ArrayList<>(Arrays.asList(s1, s2, s3));
071
072  /*
073   * Uses {@code scanners} to build a KeyValueHeap, iterates over it and asserts that returned
074   * Cells are same as {@code expected}.
075   * @return List of Cells returned from scanners.
076   */
077  public List<Cell> assertCells(List<Cell> expected, List<KeyValueScanner> scanners)
078      throws IOException {
079    //Creating KeyValueHeap
080    KeyValueHeap kvh = new KeyValueHeap(scanners, CellComparatorImpl.COMPARATOR);
081
082    List<Cell> actual = new ArrayList<>();
083    while(kvh.peek() != null){
084      actual.add(kvh.next());
085    }
086
087    assertEquals(expected, actual);
088    return actual;
089  }

096  @Test
097  public void testSorted() throws IOException{
098    //Cases that need to be checked are:
099    //1. The "smallest" Cell is in the same scanners as current
100    //2. Current scanner gets empty
101
102    List<Cell> expected = Arrays.asList(
103        kv111, kv112, kv113, kv114, kv115, kv121, kv122, kv211, kv212, kv213);
104
105    List<Cell> actual = assertCells(expected, scanners);
106
107    //Check if result is sorted according to Comparator
108    for(int i=0; i<actual.size()-1; i++){
109      int ret = CellComparatorImpl.COMPARATOR.compare(actual.get(i), actual.get(i+1));
110      assertTrue(ret < 0);
111    }
112  }

114  @Test
115  public void testSeek() throws IOException {
116    //Cases:
117    //1. Seek Cell that is not in scanner
118    //2. Check that smallest that is returned from a seek is correct
119
120    List<Cell> expected = Arrays.asList(kv211);
121
122    //Creating KeyValueHeap
123    KeyValueHeap kvh =
124      new KeyValueHeap(scanners, CellComparatorImpl.COMPARATOR);
125
126    Cell seekKv = new KeyValue(row2, fam1, null, null);
127    kvh.seek(seekKv);
128
129    List<Cell> actual = Arrays.asList(kvh.peek());
130
131    assertEquals("Expected = " + Arrays.toString(expected.toArray())
132        + "\n Actual = " + Arrays.toString(actual.toArray()), expected, actual);
133  }

135  @Test
136  public void testScannerLeak() throws IOException {
137    // Test for unclosed scanners (HBASE-1927)
138
139    TestScanner s4 = new TestScanner(new ArrayList<>());
140    scanners.add(s4);
141
142    //Creating KeyValueHeap
143    KeyValueHeap kvh = new KeyValueHeap(scanners, CellComparatorImpl.COMPARATOR);
144
145    while(kvh.next() != null);
146    // Once the internal scanners go out of Cells, those will be removed from KVHeap's priority
147    // queue and added to a Set for lazy close. The actual close will happen only on KVHeap#close()
148    assertEquals(4, kvh.scannersForDelayedClose.size());
149    assertTrue(kvh.scannersForDelayedClose.contains(s1));
150    assertTrue(kvh.scannersForDelayedClose.contains(s2));
151    assertTrue(kvh.scannersForDelayedClose.contains(s3));
152    assertTrue(kvh.scannersForDelayedClose.contains(s4));
153    kvh.close();
154    for(KeyValueScanner scanner : scanners) {
155      assertTrue(((TestScanner)scanner).isClosed());
156    }
157  }

159  @Test
160  public void testScannerException() throws IOException {
161    // Test for NPE issue when exception happens in scanners (HBASE-13835)
162
163    TestScanner s1 = new SeekTestScanner(Arrays.asList(kv115, kv211, kv212));
164    TestScanner s2 = new SeekTestScanner(Arrays.asList(kv111, kv112));
165    TestScanner s3 = new SeekTestScanner(Arrays.asList(kv113, kv114, kv121, kv122, kv213));
166    TestScanner s4 = new SeekTestScanner(new ArrayList<>());
167
168    List<KeyValueScanner> scanners = new ArrayList<>(Arrays.asList(s1, s2, s3, s4));
169
170    // Creating KeyValueHeap
171    KeyValueHeap kvh = new KeyValueHeap(scanners, CellComparatorImpl.COMPARATOR);
172
173    try {
174      for (KeyValueScanner scanner : scanners) {
175        ((SeekTestScanner) scanner).setRealSeekDone(false);
176      }
177      while (kvh.next() != null);
178      // The pollRealKV should throw IOE.
179      assertTrue(false);
180    } catch (IOException ioe) {
181      kvh.close();
182    }
183
184    // It implies there is no NPE thrown from kvh.close() if getting here
185    for (KeyValueScanner scanner : scanners) {
186      // Verify that close is called and only called once for each scanner
187      assertTrue(((SeekTestScanner) scanner).isClosed());
188      assertEquals(((SeekTestScanner) scanner).getClosedNum(), 1);
189    }
190  }

192  @Test
193  public void testPriorityId() throws IOException {
194    Cell kv113A = new KeyValue(row1, fam1, col3, Bytes.toBytes("aaa"));
195    Cell kv113B = new KeyValue(row1, fam1, col3, Bytes.toBytes("bbb"));
196    {
197      TestScanner scan1 = new TestScanner(Arrays.asList(kv111, kv112, kv113A), 1);
198      TestScanner scan2 = new TestScanner(Arrays.asList(kv113B), 2);
199      List<Cell> expected = Arrays.asList(kv111, kv112, kv113B, kv113A);
200      assertCells(expected, new ArrayList<>(Arrays.asList(scan1, scan2)));
201    }
202    {
203      TestScanner scan1 = new TestScanner(Arrays.asList(kv111, kv112, kv113A), 2);
204      TestScanner scan2 = new TestScanner(Arrays.asList(kv113B), 1);
205      List<Cell> expected = Arrays.asList(kv111, kv112, kv113A, kv113B);
206      assertCells(expected, new ArrayList<>(Arrays.asList(scan1, scan2)));
207    }
208  }

210  private static class TestScanner extends CollectionBackedScanner {
211    private boolean closed = false;
212    private long scannerOrder = 0;
213
214    public TestScanner(List<Cell> list) {
215      super(list);
216    }
217
218    public TestScanner(List<Cell> list, long scannerOrder) {
219      this(list);
220      this.scannerOrder = scannerOrder;
221    }
222
223    @Override
224    public long getScannerOrder() {
225      return scannerOrder;
226    }
227
228    @Override
229    public void close(){
230      closed = true;
231    }
232
233    public boolean isClosed() {
234      return closed;
235    }
236  }
237
238  private static class SeekTestScanner extends TestScanner {
239    private int closedNum = 0;
240    private boolean realSeekDone = true;
241
242    public SeekTestScanner(List<Cell> list) {
243      super(list);
244    }
245
246    @Override
247    public void close() {
248      super.close();
249      closedNum++;
250    }
251
252    public int getClosedNum() {
253      return closedNum;
254    }
255
256    @Override
257    public boolean realSeekDone() {
258      return realSeekDone;
259    }
260
261    public void setRealSeekDone(boolean done) {
262      realSeekDone = done;
263    }
264
265    @Override
266    public void enforceSeek() throws IOException {
267      throw new IOException("enforceSeek must not be called on a " + "non-lazy scanner");
268    }
269  }
