277  @Test
278  public void testTableQueryMeterSwitch() {
279    TableName tn1 = TableName.valueOf("table1");
280    Configuration conf = new Configuration(false);
281    // disable
282    rsm.updateReadQueryMeter(tn1, 500L);
283    assertFalse(HELPER.checkGaugeExists("ServerReadQueryPerSecond_count", serverSource));
284    rsm.updateWriteQueryMeter(tn1, 500L);
285    assertFalse(HELPER.checkGaugeExists("ServerWriteQueryPerSecond_count", serverSource));
286
287    // enable
288    conf.setBoolean(MetricsRegionServer.RS_ENABLE_SERVER_QUERY_METER_METRICS_KEY, true);
289    rsm = new MetricsRegionServer(wrapper, conf, null);
290    serverSource = rsm.getMetricsSource();
291    rsm.updateReadQueryMeter(tn1, 500L);
292    assertTrue(HELPER.checkGaugeExists("ServerWriteQueryPerSecond_count", serverSource));
293    HELPER.assertGauge("ServerReadQueryPerSecond_count", 500L, serverSource);
294    assertTrue(HELPER.checkGaugeExists("ServerWriteQueryPerSecond_count", serverSource));
295    rsm.updateWriteQueryMeter(tn1, 500L);
296    HELPER.assertGauge("ServerWriteQueryPerSecond_count", 500L, serverSource);
297  }
