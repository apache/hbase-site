147  private List<KeyValue> createSortedKeyValues(Random rand, int n) {
148    List<KeyValue> kvList = new ArrayList<>(n);
149    for (int i = 0; i < n; ++i)
150      kvList.add(RandomKeyValueUtil.randomKeyValue(rand));

155  @Test
156  public void testCompoundBloomFilter() throws IOException {
157    conf.setBoolean(BloomFilterFactory.IO_STOREFILE_BLOOM_ENABLED, true);
158    for (int t = 0; t < NUM_TESTS; ++t) {
159      conf.setFloat(BloomFilterFactory.IO_STOREFILE_BLOOM_ERROR_RATE,
160          (float) TARGET_ERROR_RATES[t]);
161
162      testIdMsg = "in test #" + t + ":";
163      Random generationRand = new Random(GENERATION_SEED);
164      List<KeyValue> kvs = createSortedKeyValues(generationRand, NUM_KV[t]);
165      BloomType bt = BLOOM_TYPES[t];
166      Path sfPath = writeStoreFile(t, bt, kvs);
167      readStoreFile(t, bt, kvs, sfPath);
168    }
169  }

171  /**
172   * Validates the false positive ratio by computing its z-value and comparing
173   * it to the provided threshold.
174   *
175   * @param falsePosRate experimental positive rate
176   * @param nTrials the number of Bloom filter checks
177   * @param zValueBoundary z-value boundary, positive for an upper bound and
178   *          negative for a lower bound
179   * @param cbf the compound Bloom filter we are using
180   * @param additionalMsg additional message to include in log output and
181   *          assertion failures
182   */
183  private void validateFalsePosRate(double falsePosRate, int nTrials,
184      double zValueBoundary, CompoundBloomFilter cbf, String additionalMsg) {
185    double p = BloomFilterFactory.getErrorRate(conf);
186    double zValue = (falsePosRate - p) / Math.sqrt(p * (1 - p) / nTrials);
187
188    String assortedStatsStr = " (targetErrorRate=" + p + ", falsePosRate="
189        + falsePosRate + ", nTrials=" + nTrials + ")";
190    LOG.info("z-value is " + zValue + assortedStatsStr);
191
192    boolean isUpperBound = zValueBoundary > 0;
193
194    if (isUpperBound && zValue > zValueBoundary ||
195        !isUpperBound && zValue < zValueBoundary) {
196      String errorMsg = "False positive rate z-value " + zValue + " is "
197          + (isUpperBound ? "higher" : "lower") + " than " + zValueBoundary
198          + assortedStatsStr + ". Per-chunk stats:\n"
199          + cbf.formatTestingStats();
200      fail(errorMsg + additionalMsg);
201    }
202  }

204  private void readStoreFile(int t, BloomType bt, List<KeyValue> kvs,
205      Path sfPath) throws IOException {
206    HStoreFile sf = new HStoreFile(fs, sfPath, conf, cacheConf, bt, true);
207    sf.initReader();
208    StoreFileReader r = sf.getReader();
209    final boolean pread = true; // does not really matter
210    StoreFileScanner scanner = r.getStoreFileScanner(true, pread, false, 0, 0, false);
211
212    {
213      // Test for false negatives (not allowed).
214      int numChecked = 0;
215      for (KeyValue kv : kvs) {
216        byte[] row = CellUtil.cloneRow(kv);
217        boolean present = isInBloom(scanner, row, CellUtil.cloneQualifier(kv));
218        assertTrue(testIdMsg + " Bloom filter false negative on row "
219            + Bytes.toStringBinary(row) + " after " + numChecked
220            + " successful checks", present);
221        ++numChecked;
222      }
223    }
224
225    // Test for false positives (some percentage allowed). We test in two modes:
226    // "fake lookup" which ignores the key distribution, and production mode.
227    for (boolean fakeLookupEnabled : new boolean[] { true, false }) {
228      BloomFilterUtil.setFakeLookupMode(fakeLookupEnabled);
229      try {
230        String fakeLookupModeStr = ", fake lookup is " + (fakeLookupEnabled ?
231            "enabled" : "disabled");
232        CompoundBloomFilter cbf = (CompoundBloomFilter) r.getGeneralBloomFilter();
233        cbf.enableTestingStats();
234        int numFalsePos = 0;
235        Random rand = new Random(EVALUATION_SEED);
236        int nTrials = NUM_KV[t] * 10;
237        for (int i = 0; i < nTrials; ++i) {
238          byte[] query = RandomKeyValueUtil.randomRowOrQualifier(rand);
239          if (isInBloom(scanner, query, bt, rand)) {
240            numFalsePos += 1;
241          }
242        }
243        double falsePosRate = numFalsePos * 1.0 / nTrials;
244        LOG.debug(String.format(testIdMsg
245            + " False positives: %d out of %d (%f)",
246            numFalsePos, nTrials, falsePosRate) + fakeLookupModeStr);
247
248        // Check for obvious Bloom filter crashes.
249        assertTrue("False positive is too high: " + falsePosRate + " (greater "
250            + "than " + TOO_HIGH_ERROR_RATE + ")" + fakeLookupModeStr,
251            falsePosRate < TOO_HIGH_ERROR_RATE);
252
253        // Now a more precise check to see if the false positive rate is not
254        // too high. The reason we use a relaxed restriction for the real-world
255        // case as opposed to the "fake lookup" case is that our hash functions
256        // are not completely independent.
257
258        double maxZValue = fakeLookupEnabled ? 1.96 : 2.5;
259        validateFalsePosRate(falsePosRate, nTrials, maxZValue, cbf,
260            fakeLookupModeStr);
261
262        // For checking the lower bound we need to eliminate the last chunk,
263        // because it is frequently smaller and the false positive rate in it
264        // is too low. This does not help if there is only one under-sized
265        // chunk, though.
266        int nChunks = cbf.getNumChunks();
267        if (nChunks > 1) {
268          numFalsePos -= cbf.getNumPositivesForTesting(nChunks - 1);
269          nTrials -= cbf.getNumQueriesForTesting(nChunks - 1);
270          falsePosRate = numFalsePos * 1.0 / nTrials;
271          LOG.info(testIdMsg + " False positive rate without last chunk is " +
272              falsePosRate + fakeLookupModeStr);
273        }
274
275        validateFalsePosRate(falsePosRate, nTrials, -2.58, cbf,
276            fakeLookupModeStr);
277      } finally {
278        BloomFilterUtil.setFakeLookupMode(false);
279      }
280    }
281
282    r.close(true); // end of test so evictOnClose
283  }

285  private boolean isInBloom(StoreFileScanner scanner, byte[] row, BloomType bt,
286      Random rand) {
287    return isInBloom(scanner, row, RandomKeyValueUtil.randomRowOrQualifier(rand));
288  }

290  private boolean isInBloom(StoreFileScanner scanner, byte[] row,
291      byte[] qualifier) {
292    Scan scan = new Scan().withStartRow(row).withStopRow(row, true);
293    scan.addColumn(Bytes.toBytes(RandomKeyValueUtil.COLUMN_FAMILY_NAME), qualifier);
294    HStore store = mock(HStore.class);
295    when(store.getColumnFamilyDescriptor())
296        .thenReturn(ColumnFamilyDescriptorBuilder.of(RandomKeyValueUtil.COLUMN_FAMILY_NAME));
297    return scanner.shouldUseScanner(scan, store, Long.MIN_VALUE);
298  }

300  private Path writeStoreFile(int t, BloomType bt, List<KeyValue> kvs)
301      throws IOException {
302    conf.setInt(BloomFilterFactory.IO_STOREFILE_BLOOM_BLOCK_SIZE,
303        BLOOM_BLOCK_SIZES[t]);
304    conf.setBoolean(CacheConfig.CACHE_BLOCKS_ON_WRITE_KEY, true);
305    cacheConf = new CacheConfig(conf);
306    HFileContext meta = new HFileContextBuilder().withBlockSize(BLOCK_SIZES[t]).build();
307    StoreFileWriter w = new StoreFileWriter.Builder(conf, cacheConf, fs)
308            .withOutputDir(TEST_UTIL.getDataTestDir())
309            .withBloomType(bt)
310            .withFileContext(meta)
311            .build();
312
313    assertTrue(w.hasGeneralBloom());
314    assertTrue(w.getGeneralBloomWriter() instanceof CompoundBloomFilterWriter);
315    CompoundBloomFilterWriter cbbf =
316        (CompoundBloomFilterWriter) w.getGeneralBloomWriter();
317
318    int keyCount = 0;
319    KeyValue prev = null;
320    LOG.debug("Total keys/values to insert: " + kvs.size());
321    for (KeyValue kv : kvs) {
322      w.append(kv);
323
324      // Validate the key count in the Bloom filter.
325      boolean newKey = true;
326      if (prev != null) {
327        newKey = !(bt == BloomType.ROW ? CellUtil.matchingRows(kv,
328            prev) : CellUtil.matchingRowColumn(kv, prev));
329      }
330      if (newKey)
331        ++keyCount;
332      assertEquals(keyCount, cbbf.getKeyCount());
333
334      prev = kv;
335    }
336    w.close();
337
338    return w.getPath();
339  }

341  @Test
342  public void testCompoundBloomSizing() {
343    int bloomBlockByteSize = 4096;
344    int bloomBlockBitSize = bloomBlockByteSize * 8;
345    double targetErrorRate = 0.01;
346    long maxKeysPerChunk = BloomFilterUtil.idealMaxKeys(bloomBlockBitSize,
347        targetErrorRate);
348
349    long bloomSize1 = bloomBlockByteSize * 8;
350    long bloomSize2 = BloomFilterUtil.computeBitSize(maxKeysPerChunk,
351        targetErrorRate);
352
353    double bloomSizeRatio = (bloomSize2 * 1.0 / bloomSize1);
354    assertTrue(Math.abs(bloomSizeRatio - 0.9999) < 0.0001);
355  }

357  @Test
358  public void testCreateKey() {
359    byte[] row = "myRow".getBytes();
360    byte[] qualifier = "myQualifier".getBytes();
361    // Mimic what Storefile.createBloomKeyValue() does
362    byte[] rowKey = KeyValueUtil.createFirstOnRow(row, 0, row.length, new byte[0], 0, 0, row, 0, 0).getKey();
363    byte[] rowColKey = KeyValueUtil.createFirstOnRow(row, 0, row.length,
364        new byte[0], 0, 0, qualifier, 0, qualifier.length).getKey();
365    KeyValue rowKV = KeyValueUtil.createKeyValueFromKey(rowKey);
366    KeyValue rowColKV = KeyValueUtil.createKeyValueFromKey(rowColKey);
367    assertEquals(rowKV.getTimestamp(), rowColKV.getTimestamp());
368    assertEquals(Bytes.toStringBinary(rowKV.getRowArray(), rowKV.getRowOffset(),
369      rowKV.getRowLength()), Bytes.toStringBinary(rowColKV.getRowArray(), rowColKV.getRowOffset(),
370      rowColKV.getRowLength()));
371    assertEquals(0, rowKV.getQualifierLength());
372  }
