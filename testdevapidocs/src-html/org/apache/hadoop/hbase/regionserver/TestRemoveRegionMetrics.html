020import org.apache.hadoop.conf.Configuration;
021import org.apache.hadoop.hbase.CompatibilityFactory;

027import org.apache.hadoop.hbase.TableName;
028import org.apache.hadoop.hbase.client.Admin;
029import org.apache.hadoop.hbase.client.Put;
030import org.apache.hadoop.hbase.client.RegionLocator;
031import org.apache.hadoop.hbase.client.Table;
032import org.apache.hadoop.hbase.test.MetricsAssertHelper;
033import org.apache.hadoop.hbase.testclassification.LargeTests;
034
035import org.apache.hadoop.hbase.testclassification.RegionServerTests;
036import org.apache.hadoop.hbase.util.Bytes;
037import org.apache.hadoop.hbase.util.Threads;
038import org.junit.BeforeClass;
039import org.junit.Test;
040import org.junit.experimental.categories.Category;
041
042import java.io.IOException;
043
044@Category({RegionServerTests.class, LargeTests.class})
045public class TestRemoveRegionMetrics {
046
047  private static MiniHBaseCluster cluster;
048  private static Configuration conf;
049  private static HBaseTestingUtility TEST_UTIL;
050  private static MetricsAssertHelper metricsHelper;
051
052  @BeforeClass
053  public static void startCluster() throws Exception {
054    metricsHelper = CompatibilityFactory.getInstance(MetricsAssertHelper.class);
055    TEST_UTIL = new HBaseTestingUtility();
056    conf = TEST_UTIL.getConfiguration();
057    conf.getLong("hbase.splitlog.max.resubmit", 0);
058    // Make the failure test faster
059    conf.setInt("zookeeper.recovery.retry", 0);
060    conf.setInt(HConstants.REGIONSERVER_INFO_PORT, -1);
061
062    TEST_UTIL.startMiniCluster(1, 2);
063    cluster = TEST_UTIL.getHBaseCluster();
064
065    cluster.waitForActiveAndReadyMaster();
066
067    while (cluster.getLiveRegionServerThreads().size() < 2) {
068      Threads.sleep(100);
069    }
070  }
071
072
073  @Test
074  public void testMoveRegion() throws IOException, InterruptedException {
075    String tableNameString = "testMoveRegion";
076    TableName tableName = TableName.valueOf(tableNameString);
077    Table t = TEST_UTIL.createTable(tableName, Bytes.toBytes("D"));
078    TEST_UTIL.waitUntilAllRegionsAssigned(t.getName());
079    Admin admin = TEST_UTIL.getHBaseAdmin();
080    HRegionInfo regionInfo;
081    byte[] row =  Bytes.toBytes("r1");
082
083
084    for (int i = 0; i < 30; i++) {
085      boolean moved = false;
086      try (RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
087        regionInfo = locator.getRegionLocation(row, true).getRegionInfo();
088      }
089
090      int currentServerIdx = cluster.getServerWith(regionInfo.getRegionName());
091      int destServerIdx = (currentServerIdx +1)% cluster.getLiveRegionServerThreads().size();
092      HRegionServer currentServer = cluster.getRegionServer(currentServerIdx);
093      HRegionServer destServer = cluster.getRegionServer(destServerIdx);
094      byte[] destServerName = Bytes.toBytes(destServer.getServerName().getServerName());
095
096
097      // Do a put. The counters should be non-zero now
098      Put p = new Put(row);
099      p.addColumn(Bytes.toBytes("D"), Bytes.toBytes("Zero"), Bytes.toBytes("VALUE"));
100      t.put(p);
101
102
103      MetricsRegionAggregateSource currentAgg = currentServer.getRegion(regionInfo.getRegionName())
104          .getMetrics()
105          .getSource()
106          .getAggregateSource();
107
108      String prefix = "namespace_"+ NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR+
109          "_table_"+tableNameString +
110          "_region_" + regionInfo.getEncodedName()+
111          "_metric";
112
113      metricsHelper.assertCounter(prefix + "_mutateCount", 1, currentAgg);
114
115
116      try {
117        admin.move(regionInfo.getEncodedNameAsBytes(), destServerName);
118        moved = true;
119        Thread.sleep(5000);
120      } catch (IOException ioe) {
121        moved = false;
122      }
123      TEST_UTIL.waitUntilAllRegionsAssigned(t.getName());
124
125      if (moved) {
126        MetricsRegionAggregateSource destAgg = destServer.getRegion(regionInfo.getRegionName())
127            .getMetrics()
128            .getSource()
129            .getAggregateSource();
130        metricsHelper.assertCounter(prefix + "_mutateCount", 0, destAgg);
131      }
132    }
133
134    TEST_UTIL.deleteTable(tableName);
135
136  }
137}
