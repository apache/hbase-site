037/**
038 * A test similar to TestHRegion, but with in-memory flush families.
039 * Also checks wal truncation after in-memory compaction.
040 */
041@Category({VerySlowRegionServerTests.class, LargeTests.class})
042public class TestHRegionWithInMemoryFlush extends TestHRegion {
043
044  @ClassRule
045  public static final HBaseClassTestRule CLASS_RULE =
046      HBaseClassTestRule.forClass(TestHRegionWithInMemoryFlush.class);
047
048  /**
049   * @return A region on which you must call
050   *         {@link HBaseTestingUtility#closeRegionAndWAL(HRegion)} when done.
051   */
052  @Override
053  public HRegion initHRegion(TableName tableName, byte[] startKey, byte[] stopKey,
054      boolean isReadOnly, Durability durability, WAL wal, byte[]... families) throws IOException {
055    boolean[] inMemory = new boolean[families.length];
056    for(int i = 0; i < inMemory.length; i++) {
057      inMemory[i] = true;
058    }
059    ChunkCreator.initialize(MemStoreLAB.CHUNK_SIZE_DEFAULT, false, 0, 0,
060      0, null, MemStoreLAB.INDEX_CHUNK_SIZE_PERCENTAGE_DEFAULT);
061    return TEST_UTIL.createLocalHRegionWithInMemoryFlags(tableName, startKey, stopKey,
062        isReadOnly, durability, wal, inMemory, families);
063  }
064
065  @Override int getTestCountForTestWritesWhileScanning() {
066    return 10;
067  }
068
069  /**
070   * testWritesWhileScanning is flakey when called out of this class. Need to dig in. Meantime
071   * go easy on it. See if that helps.
072   */
073  @Override int getNumQualifiersForTestWritesWhileScanning() {
074    return 10;
075  }
076
077  /**
078   * A test case of HBASE-21041
079   * @throws Exception Exception
080   */
081  @Override
082  @Test
083  public void testFlushAndMemstoreSizeCounting() throws Exception {
084    byte[] family = Bytes.toBytes("family");
085    this.region = initHRegion(tableName, method, CONF, family);
086    final WALFactory wals = new WALFactory(CONF, method);
087    int count = 0;
088    try {
089      for (byte[] row : HBaseTestingUtility.ROWS) {
090        Put put = new Put(row);
091        put.addColumn(family, family, row);
092        region.put(put);
093        //In memory flush every 1000 puts
094        if (count++ % 1000 == 0) {
095          ((CompactingMemStore) (region.getStore(family).memstore))
096              .flushInMemory();
097        }
098      }
099      region.flush(true);
100      // After flush, data size should be zero
101      Assert.assertEquals(0, region.getMemStoreDataSize());
102      // After flush, a new active mutable segment is created, so the heap size
103      // should equal to MutableSegment.DEEP_OVERHEAD
104      Assert.assertEquals(MutableSegment.DEEP_OVERHEAD, region.getMemStoreHeapSize());
105      // After flush, offheap size should be zero
106      Assert.assertEquals(0, region.getMemStoreOffHeapSize());
107
108    } finally {
109      HBaseTestingUtility.closeRegionAndWAL(this.region);
110      this.region = null;
111      wals.close();
112    }
113  }
114}
