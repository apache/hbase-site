048  @Test
049  public void testMvcc() throws Exception {
050    ServerNonceManager nm = createManager();
051    final long group = 100;
052    final long nonce = 1;
053    final long initMvcc = 999;
054    assertTrue(nm.startOperation(group, nonce, createStoppable()));
055    nm.addMvccToOperationContext(group, nonce, initMvcc);
056    nm.endOperation(group, nonce, true);
057    assertEquals(initMvcc, nm.getMvccFromOperationContext(group, nonce));
058    long newMvcc = initMvcc + 1;
059    for (long newNonce = nonce + 1; newNonce != (nonce + 5); ++newNonce) {
060      assertTrue(nm.startOperation(group, newNonce, createStoppable()));
061      nm.addMvccToOperationContext(group, newNonce, newMvcc);
062      nm.endOperation(group, newNonce, true);
063      assertEquals(newMvcc, nm.getMvccFromOperationContext(group, newNonce));
064      ++newMvcc;
065    }
066    assertEquals(initMvcc, nm.getMvccFromOperationContext(group, nonce));
067  }

069  @Test
070  public void testNormalStartEnd() throws Exception {
071    final long[] numbers = new long[] { NO_NONCE, 1, 2, Long.MAX_VALUE, Long.MIN_VALUE };
072    ServerNonceManager nm = createManager();
073    for (int i = 0; i < numbers.length; ++i) {
074      for (int j = 0; j < numbers.length; ++j) {
075        assertTrue(nm.startOperation(numbers[i], numbers[j], createStoppable()));
076      }
077    }
078    // Should be able to start operation the second time w/o nonces.
079    for (int i = 0; i < numbers.length; ++i) {
080      assertTrue(nm.startOperation(numbers[i], NO_NONCE, createStoppable()));
081    }
082    // Fail all operations - should be able to restart.
083    for (int i = 0; i < numbers.length; ++i) {
084      for (int j = 0; j < numbers.length; ++j) {
085        nm.endOperation(numbers[i], numbers[j], false);
086        assertTrue(nm.startOperation(numbers[i], numbers[j], createStoppable()));
087      }
088    }
089    // Succeed all operations - should not be able to restart, except for NO_NONCE.
090    for (int i = 0; i < numbers.length; ++i) {
091      for (int j = 0; j < numbers.length; ++j) {
092        nm.endOperation(numbers[i], numbers[j], true);
093        assertEquals(numbers[j] == NO_NONCE,
094            nm.startOperation(numbers[i], numbers[j], createStoppable()));
095      }
096    }
097  }

108  @Test
109  public void testCleanup() throws Exception {
110    ManualEnvironmentEdge edge = new ManualEnvironmentEdge();
111    EnvironmentEdgeManager.injectEdge(edge);
112    try {
113      ServerNonceManager nm = createManager(6);
114      ScheduledChore cleanup = nm.createCleanupScheduledChore(Mockito.mock(Stoppable.class));
115      edge.setValue(1);
116      assertTrue(nm.startOperation(NO_NONCE, 1, createStoppable()));
117      assertTrue(nm.startOperation(NO_NONCE, 2, createStoppable()));
118      assertTrue(nm.startOperation(NO_NONCE, 3, createStoppable()));
119      edge.setValue(2);
120      nm.endOperation(NO_NONCE, 1, true);
121      edge.setValue(4);
122      nm.endOperation(NO_NONCE, 2, true);
123      edge.setValue(9);
124      cleanup.choreForTesting();
125      // Nonce 1 has been cleaned up.
126      assertTrue(nm.startOperation(NO_NONCE, 1, createStoppable()));
127      // Nonce 2 has not been cleaned up.
128      assertFalse(nm.startOperation(NO_NONCE, 2, createStoppable()));
129      // Nonce 3 was active and active ops should never be cleaned up; try to end and start.
130      nm.endOperation(NO_NONCE, 3, false);
131      assertTrue(nm.startOperation(NO_NONCE, 3, createStoppable()));
132      edge.setValue(11);
133      cleanup.choreForTesting();
134      // Now, nonce 2 has been cleaned up.
135      assertTrue(nm.startOperation(NO_NONCE, 2, createStoppable()));
136    } finally {
137      EnvironmentEdgeManager.reset();
138    }
139  }

141  @Test
142  public void testWalNonces() throws Exception {
143    ManualEnvironmentEdge edge = new ManualEnvironmentEdge();
144    EnvironmentEdgeManager.injectEdge(edge);
145    try {
146      ServerNonceManager nm = createManager(6);
147      ScheduledChore cleanup = nm.createCleanupScheduledChore(Mockito.mock(Stoppable.class));
148      // Add nonces from WAL, including dups.
149      edge.setValue(12);
150      nm.reportOperationFromWal(NO_NONCE, 1, 8);
151      nm.reportOperationFromWal(NO_NONCE, 2, 2);
152      nm.reportOperationFromWal(NO_NONCE, 3, 5);
153      nm.reportOperationFromWal(NO_NONCE, 3, 6);
154      // WAL nonces should prevent cross-server conflicts.
155      assertFalse(nm.startOperation(NO_NONCE, 1, createStoppable()));
156      // Make sure we ignore very old nonces, but not borderline old nonces.
157      assertTrue(nm.startOperation(NO_NONCE, 2, createStoppable()));
158      assertFalse(nm.startOperation(NO_NONCE, 3, createStoppable()));
159      // Make sure grace period is counted from recovery time.
160      edge.setValue(17);
161      cleanup.choreForTesting();
162      assertFalse(nm.startOperation(NO_NONCE, 1, createStoppable()));
163      assertFalse(nm.startOperation(NO_NONCE, 3, createStoppable()));
164      edge.setValue(19);
165      cleanup.choreForTesting();
166      assertTrue(nm.startOperation(NO_NONCE, 1, createStoppable()));
167      assertTrue(nm.startOperation(NO_NONCE, 3, createStoppable()));
168    } finally {
169      EnvironmentEdgeManager.reset();
170    }
171  }

173  @Test
174  public void testConcurrentAttempts() throws Exception {
175    final ServerNonceManager nm = createManager();
176
177    nm.startOperation(NO_NONCE, 1, createStoppable());
178    TestRunnable tr = new TestRunnable(nm, 1, false, createStoppable());
179    Thread t = tr.start();
180    waitForThreadToBlockOrExit(t);
181    nm.endOperation(NO_NONCE, 1, true); // operation succeeded
182    t.join(); // thread must now unblock and not proceed (result checked inside).
183    tr.propagateError();
184
185    nm.startOperation(NO_NONCE, 2, createStoppable());
186    tr = new TestRunnable(nm, 2, true, createStoppable());
187    t = tr.start();
188    waitForThreadToBlockOrExit(t);
189    nm.endOperation(NO_NONCE, 2, false);
190    t.join(); // thread must now unblock and allow us to proceed (result checked inside).
191    tr.propagateError();
192    nm.endOperation(NO_NONCE, 2, true); // that is to say we should be able to end operation
193
194    nm.startOperation(NO_NONCE, 3, createStoppable());
195    tr = new TestRunnable(nm, 4, true, createStoppable());
196    tr.start().join();  // nonce 3 must have no bearing on nonce 4
197    tr.propagateError();
198  }

200  @Test
201  public void testStopWaiting() throws Exception {
202    final ServerNonceManager nm = createManager();
203    nm.setConflictWaitIterationMs(1);
204    Stoppable stoppingStoppable = createStoppable();
205    Mockito.when(stoppingStoppable.isStopped()).thenAnswer(new Answer<Boolean>() {
206      AtomicInteger answer = new AtomicInteger(3);
207      @Override
208      public Boolean answer(InvocationOnMock invocation) throws Throwable {
209        return 0 < answer.decrementAndGet();
210      }
211    });
212
213    nm.startOperation(NO_NONCE, 1, createStoppable());
214    TestRunnable tr = new TestRunnable(nm, 1, null, stoppingStoppable);
215    Thread t = tr.start();
216    waitForThreadToBlockOrExit(t);
217    // thread must eventually throw
218    t.join();
219    tr.propagateError();
220  }
221
222  private void waitForThreadToBlockOrExit(Thread t) throws InterruptedException {
223    for (int i = 9; i >= 0; --i) {
224      if (t.getState() == Thread.State.TIMED_WAITING || t.getState() == Thread.State.WAITING
225          || t.getState() == Thread.State.BLOCKED || t.getState() == Thread.State.TERMINATED) {
226        return;
227      }
228      if (i > 0) Thread.sleep(300);
229    }
230    // Thread didn't block in 3 seconds. What is it doing? Continue the test, we'd rather
231    // have a very strange false positive then false negative due to timing.
232  }

234  private static class TestRunnable implements Runnable {
235    public final CountDownLatch startedLatch = new CountDownLatch(1); // It's the final countdown!
236
237    private final ServerNonceManager nm;
238    private final long nonce;
239    private final Boolean expected;
240    private final Stoppable stoppable;
241
242    private Throwable throwable = null;
243
244    public TestRunnable(ServerNonceManager nm, long nonce, Boolean expected, Stoppable stoppable) {
245      this.nm = nm;
246      this.nonce = nonce;
247      this.expected = expected;
248      this.stoppable = stoppable;
249    }
250
251    public void propagateError() throws Exception {
252      if (throwable == null) return;
253      throw new Exception(throwable);
254    }
255
256    public Thread start() {
257      Thread t = new Thread(this);
258      t = Threads.setDaemonThreadRunning(t);
259      try {
260        startedLatch.await();
261      } catch (InterruptedException e) {
262        fail("Unexpected");
263      }
264      return t;
265    }
266
267    @Override
268    public void run() {
269      startedLatch.countDown();
270      boolean shouldThrow = expected == null;
271      boolean hasThrown = true;
272      try {
273        boolean result = nm.startOperation(NO_NONCE, nonce, stoppable);
274        hasThrown = false;
275        if (!shouldThrow) {
276          assertEquals(expected.booleanValue(), result);
277        }
278      } catch (Throwable t) {
279        if (!shouldThrow) {
280          throwable = t;
281        }
282      }
283      if (shouldThrow && !hasThrown) {
284        throwable = new AssertionError("Should have thrown");
285      }
286    }
287  }
288
289  private Stoppable createStoppable() {
290    Stoppable s = Mockito.mock(Stoppable.class);
291    Mockito.when(s.isStopped()).thenReturn(false);
292    return s;
293  }
294
295  private ServerNonceManager createManager() {
296    return createManager(null);
297  }
298
299  private ServerNonceManager createManager(Integer gracePeriod) {
300    Configuration conf = HBaseConfiguration.create();
301    if (gracePeriod != null) {
302      conf.setInt(ServerNonceManager.HASH_NONCE_GRACE_PERIOD_KEY, gracePeriod.intValue());
303    }
304    return new ServerNonceManager(conf);
305  }
