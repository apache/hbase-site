126  /**
127   * During compaction, cells smaller than the threshold won't be affected.
128   */
129  @Test
130  public void testSmallerValue() throws Exception {
131    init(UTIL.getConfiguration(), 500);
132    byte[] dummyData = makeDummyData(300); // smaller than mob threshold
133    Table loader = new RegionAsTable(region);
134    // one hfile per row
135    for (int i = 0; i < compactionThreshold; i++) {
136      Put p = createPut(i, dummyData);
137      loader.put(p);
138      region.flush(true);
139    }
140    assertEquals("Before compaction: store files", compactionThreshold, countStoreFiles());
141    assertEquals("Before compaction: mob file count", 0, countMobFiles());
142    assertEquals("Before compaction: rows", compactionThreshold, UTIL.countRows(region));
143    assertEquals("Before compaction: mob rows", 0, countMobRows());
144
145    region.compactStores();
146
147    assertEquals("After compaction: store files", 1, countStoreFiles());
148    assertEquals("After compaction: mob file count", 0, countMobFiles());
149    assertEquals("After compaction: referenced mob file count", 0, countReferencedMobFiles());
150    assertEquals("After compaction: rows", compactionThreshold, UTIL.countRows(region));
151    assertEquals("After compaction: mob rows", 0, countMobRows());
152  }

168    assertEquals("Before compaction: mob file count", compactionThreshold, countMobFiles());
169    assertEquals("Before compaction: rows", compactionThreshold, UTIL.countRows(region));
170    assertEquals("Before compaction: mob rows", compactionThreshold, countMobRows());
171    assertEquals("Before compaction: number of mob cells", compactionThreshold,
172        countMobCellsInMetadata());

175    region.initialize();

178    assertEquals("After compaction: store files", 1, countStoreFiles());
179    assertEquals("After compaction: mob file count", compactionThreshold, countMobFiles());
180    assertEquals("After compaction: referenced mob file count", 0, countReferencedMobFiles());
181    assertEquals("After compaction: rows", compactionThreshold, UTIL.countRows(region));
182    assertEquals("After compaction: mob rows", 0, countMobRows());
183  }

185  /**
186   * This test will first generate store files, then bulk load them and trigger the compaction.
187   * When compaction, the cell value will be larger than the threshold.
188   */
189  @Test
190  public void testMobCompactionWithBulkload() throws Exception {
191    // The following will produce store files of 600.
192    init(UTIL.getConfiguration(), 300);
193    byte[] dummyData = makeDummyData(600);
194
195    Path hbaseRootDir = FSUtils.getRootDir(conf);
196    Path basedir = new Path(hbaseRootDir, htd.getNameAsString());
197    List<Pair<byte[], String>> hfiles = new ArrayList<>(1);
198    for (int i = 0; i < compactionThreshold; i++) {
199      Path hpath = new Path(basedir, "hfile" + i);
200      hfiles.add(Pair.newPair(COLUMN_FAMILY, hpath.toString()));
201      createHFile(hpath, i, dummyData);
202    }
203
204    // The following will bulk load the above generated store files and compact, with 600(fileSize)
205    // > 300(threshold)
206    Map<byte[], List<Path>> map = region.bulkLoadHFiles(hfiles, true, null);
207    assertTrue("Bulkload result:", !map.isEmpty());
208    assertEquals("Before compaction: store files", compactionThreshold, countStoreFiles());
209    assertEquals("Before compaction: mob file count", 0, countMobFiles());
210    assertEquals("Before compaction: rows", compactionThreshold, UTIL.countRows(region));
211    assertEquals("Before compaction: mob rows", 0, countMobRows());
212    assertEquals("Before compaction: referenced mob file count", 0, countReferencedMobFiles());
213
214    region.compactStores();
215
216    assertEquals("After compaction: store files", 1, countStoreFiles());
217    assertEquals("After compaction: mob file count:", 1, countMobFiles());
218    assertEquals("After compaction: rows", compactionThreshold, UTIL.countRows(region));
219    assertEquals("After compaction: mob rows", compactionThreshold, countMobRows());
220    assertEquals("After compaction: referenced mob file count", 1, countReferencedMobFiles());
221    assertEquals("After compaction: number of mob cells", compactionThreshold,
222        countMobCellsInMetadata());
223  }

225  @Test
226  public void testMajorCompactionAfterDelete() throws Exception {
227    init(UTIL.getConfiguration(), 100);
228    byte[] dummyData = makeDummyData(200); // larger than mob threshold
229    Table loader = new RegionAsTable(region);
230    // create hfiles and mob hfiles but don't trigger compaction
231    int numHfiles = compactionThreshold - 1;
232    byte[] deleteRow = Bytes.add(STARTROW, Bytes.toBytes(0));
233    for (int i = 0; i < numHfiles; i++) {
234      Put p = createPut(i, dummyData);
235      loader.put(p);
236      region.flush(true);
237    }
238    assertEquals("Before compaction: store files", numHfiles, countStoreFiles());
239    assertEquals("Before compaction: mob file count", numHfiles, countMobFiles());
240    assertEquals("Before compaction: rows", numHfiles, UTIL.countRows(region));
241    assertEquals("Before compaction: mob rows", numHfiles, countMobRows());
242    assertEquals("Before compaction: number of mob cells", numHfiles, countMobCellsInMetadata());
243    // now let's delete some cells that contain mobs
244    Delete delete = new Delete(deleteRow);
245    delete.addFamily(COLUMN_FAMILY);
246    region.delete(delete);
247    region.flush(true);
248
249    assertEquals("Before compaction: store files", numHfiles + 1, countStoreFiles());
250    assertEquals("Before compaction: mob files", numHfiles, countMobFiles());
251    // region.compactStores();
252    region.compact(true);
253    assertEquals("After compaction: store files", 1, countStoreFiles());
254    // still have original mob hfiles and now added a mob del file
255    assertEquals("After compaction: mob files", numHfiles + 1, countMobFiles());
256
257    Scan scan = new Scan();
258    scan.setRaw(true);
259    InternalScanner scanner = region.getScanner(scan);
260    List<Cell> results = new ArrayList<>();
261    scanner.next(results);
262    int deleteCount = 0;
263    while (!results.isEmpty()) {
264      for (Cell c : results) {
265        if (c.getTypeByte() == KeyValue.Type.DeleteFamily.getCode()) {
266          deleteCount++;
267          assertTrue(Bytes.equals(CellUtil.cloneRow(c), deleteRow));
268        }
269      }
270      results.clear();
271      scanner.next(results);
272    }
273    // assert the delete mark is retained after the major compaction
274    assertEquals(1, deleteCount);
275    scanner.close();
276    // assert the deleted cell is not counted
277    assertEquals("The cells in mob files", numHfiles - 1, countMobCellsInMobFiles(1));
278  }

294  private long countMobCellsInMetadata() throws IOException {
295    long mobCellsCount = 0;
296    Path mobDirPath = MobUtils.getMobFamilyPath(conf, htd.getTableName(), hcd.getNameAsString());
297    Configuration copyOfConf = new Configuration(conf);
298    copyOfConf.setFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY, 0f);
299    CacheConfig cacheConfig = new CacheConfig(copyOfConf);
300    if (fs.exists(mobDirPath)) {
301      FileStatus[] files = UTIL.getTestFileSystem().listStatus(mobDirPath);
302      for (FileStatus file : files) {
303        StoreFile sf = new HStoreFile(fs, file.getPath(), conf, cacheConfig, BloomType.NONE, true);
304        sf.initReader();
305        Map<byte[], byte[]> fileInfo = sf.getReader().loadFileInfo();
306        byte[] count = fileInfo.get(StoreFile.MOB_CELLS_COUNT);
307        assertTrue(count != null);
308        mobCellsCount += Bytes.toLong(count);
309      }
310    }
311    return mobCellsCount;
312  }

314  private Put createPut(int rowIdx, byte[] dummyData) throws IOException {
315    Put p = new Put(Bytes.add(STARTROW, Bytes.toBytes(rowIdx)));
316    p.setDurability(Durability.SKIP_WAL);
317    p.addColumn(COLUMN_FAMILY, Bytes.toBytes("colX"), dummyData);
318    return p;
319  }

321  /**
322   * Create an HFile with the given number of bytes
323   */
324  private void createHFile(Path path, int rowIdx, byte[] dummyData) throws IOException {
325    HFileContext meta = new HFileContextBuilder().build();
326    HFile.Writer writer = HFile.getWriterFactory(conf, new CacheConfig(conf)).withPath(fs, path)
327        .withFileContext(meta).create();
328    long now = System.currentTimeMillis();
329    try {
330      KeyValue kv = new KeyValue(Bytes.add(STARTROW, Bytes.toBytes(rowIdx)), COLUMN_FAMILY,
331          Bytes.toBytes("colX"), now, dummyData);
332      writer.append(kv);
333    } finally {
334      writer.appendFileInfo(StoreFile.BULKLOAD_TIME_KEY, Bytes.toBytes(System.currentTimeMillis()));
335      writer.close();
336    }
337  }

339  private int countMobRows() throws IOException {
340    Scan scan = new Scan();
341    // Do not retrieve the mob data when scanning
342    scan.setAttribute(MobConstants.MOB_SCAN_RAW, Bytes.toBytes(Boolean.TRUE));
343    InternalScanner scanner = region.getScanner(scan);
344
345    int scannedCount = 0;
346    List<Cell> results = new ArrayList<>();
347    boolean hasMore = true;
348    while (hasMore) {
349      hasMore = scanner.next(results);
350      for (Cell c : results) {
351        if (MobUtils.isMobReferenceCell(c)) {
352          scannedCount++;
353        }
354      }
355      results.clear();
356    }
357    scanner.close();
358
359    return scannedCount;
360  }

368  private int countReferencedMobFiles() throws IOException {
369    Scan scan = new Scan();
370    // Do not retrieve the mob data when scanning
371    scan.setAttribute(MobConstants.MOB_SCAN_RAW, Bytes.toBytes(Boolean.TRUE));
372    InternalScanner scanner = region.getScanner(scan);
373
374    List<Cell> kvs = new ArrayList<>();
375    boolean hasMore = true;
376    String fileName;
377    Set<String> files = new HashSet<>();
378    do {
379      kvs.clear();
380      hasMore = scanner.next(kvs);
381      for (Cell kv : kvs) {
382        if (!MobUtils.isMobReferenceCell(kv)) {
383          continue;
384        }
385        if (!MobUtils.hasValidMobRefCellValue(kv)) {
386          continue;
387        }
388        int size = MobUtils.getMobValueLength(kv);
389        if (size <= mobCellThreshold) {
390          continue;
391        }
392        fileName = MobUtils.getMobFileName(kv);
393        if (fileName.isEmpty()) {
394          continue;
395        }
396        files.add(fileName);
397        Path familyPath = MobUtils.getMobFamilyPath(conf, htd.getTableName(),
398            hcd.getNameAsString());
399        assertTrue(fs.exists(new Path(familyPath, fileName)));
400      }
401    } while (hasMore);
402
403    scanner.close();
404
405    return files.size();
406  }

408  private int countMobCellsInMobFiles(int expectedNumDelfiles) throws IOException {
409    Configuration copyOfConf = new Configuration(conf);
410    copyOfConf.setFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY, 0f);
411    CacheConfig cacheConfig = new CacheConfig(copyOfConf);
412    Path mobDirPath = MobUtils.getMobFamilyPath(conf, htd.getTableName(), hcd.getNameAsString());
413    List<StoreFile> sfs = new ArrayList<>();
414    int numDelfiles = 0;
415    int size = 0;
416    if (fs.exists(mobDirPath)) {
417      for (FileStatus f : fs.listStatus(mobDirPath)) {
418        StoreFile sf = new HStoreFile(fs, f.getPath(), conf, cacheConfig, BloomType.NONE, true);
419        sfs.add(sf);
420        if (StoreFileInfo.isDelFile(sf.getPath())) {
421          numDelfiles++;
422        }
423      }
424
425      List<StoreFileScanner> scanners = StoreFileScanner.getScannersForStoreFiles(sfs, false, true, false, false,
426          HConstants.LATEST_TIMESTAMP);
427      Scan scan = new Scan();
428      scan.setMaxVersions(hcd.getMaxVersions());
429      long timeToPurgeDeletes = Math.max(conf.getLong("hbase.hstore.time.to.purge.deletes", 0), 0);
430      long ttl = HStore.determineTTLFromFamily(hcd);
431      ScanInfo scanInfo = new ScanInfo(copyOfConf, hcd, ttl, timeToPurgeDeletes,
432        CellComparator.COMPARATOR);
433      StoreScanner scanner = new StoreScanner(scan, scanInfo, ScanType.COMPACT_DROP_DELETES, null,
434          scanners, 0L, HConstants.LATEST_TIMESTAMP);
435      try {
436        size += UTIL.countRows(scanner);
437      } finally {
438        scanner.close();
439      }
440    }
441    // assert the number of the existing del files
442    assertEquals(expectedNumDelfiles, numDelfiles);
443    return size;
444  }
