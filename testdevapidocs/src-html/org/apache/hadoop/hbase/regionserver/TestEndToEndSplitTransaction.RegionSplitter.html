089  /**
090   * Tests that the client sees meta table changes as atomic during splits
091   */
092  @Test
093  public void testFromClientSideWhileSplitting() throws Throwable {
094    LOG.info("Starting testFromClientSideWhileSplitting");
095    final TableName tableName = TableName.valueOf(name.getMethodName());
096    final byte[] FAMILY = Bytes.toBytes("family");
097
098    //SplitTransaction will update the meta table by offlining the parent region, and adding info
099    //for daughters.
100    Table table = TEST_UTIL.createTable(tableName, FAMILY);
101
102    Stoppable stopper = new StoppableImplementation();
103    RegionSplitter regionSplitter = new RegionSplitter(table);
104    RegionChecker regionChecker = new RegionChecker(CONF, stopper, tableName);
105    final ChoreService choreService = new ChoreService("TEST_SERVER");
106
107    choreService.scheduleChore(regionChecker);
108    regionSplitter.start();
109
110    //wait until the splitter is finished
111    regionSplitter.join();
112    stopper.stop(null);
113
114    if (regionChecker.ex != null) {
115      throw new AssertionError("regionChecker", regionChecker.ex);
116    }
117
118    if (regionSplitter.ex != null) {
119      throw new AssertionError("regionSplitter", regionSplitter.ex);
120    }
121
122    //one final check
123    regionChecker.verify();
124  }

126  static class RegionSplitter extends Thread {
127    final Connection connection;
128    Throwable ex;
129    Table table;
130    TableName tableName;
131    byte[] family;
132    Admin admin;
133    HRegionServer rs;
134
135    RegionSplitter(Table table) throws IOException {
136      this.table = table;
137      this.tableName = table.getName();
138      this.family = table.getTableDescriptor().getFamiliesKeys().iterator().next();
139      admin = TEST_UTIL.getAdmin();
140      rs = TEST_UTIL.getMiniHBaseCluster().getRegionServer(0);
141      connection = TEST_UTIL.getConnection();
142    }
143
144    @Override
145    public void run() {
146      try {
147        Random random = new Random();
148        for (int i= 0; i< 5; i++) {
149          List<RegionInfo> regions =
150              MetaTableAccessor.getTableRegions(connection, tableName, true);
151          if (regions.isEmpty()) {
152            continue;
153          }
154          int regionIndex = random.nextInt(regions.size());
155
156          //pick a random region and split it into two
157          RegionInfo region = Iterators.get(regions.iterator(), regionIndex);
158
159          //pick the mid split point
160          int start = 0, end = Integer.MAX_VALUE;
161          if (region.getStartKey().length > 0) {
162            start = Bytes.toInt(region.getStartKey());
163          }
164          if (region.getEndKey().length > 0) {
165            end = Bytes.toInt(region.getEndKey());
166          }
167          int mid = start + ((end - start) / 2);
168          byte[] splitPoint = Bytes.toBytes(mid);
169
170          //put some rows to the regions
171          addData(start);
172          addData(mid);
173
174          flushAndBlockUntilDone(admin, rs, region.getRegionName());
175          compactAndBlockUntilDone(admin, rs, region.getRegionName());
176
177          log("Initiating region split for:" + region.getRegionNameAsString());
178          try {
179            admin.splitRegion(region.getRegionName(), splitPoint);
180            //wait until the split is complete
181            blockUntilRegionSplit(CONF, 50000, region.getRegionName(), true);
182
183          } catch (NotServingRegionException ex) {
184            //ignore
185          }
186        }
187      } catch (Throwable ex) {
188        this.ex = ex;
189      }
190    }
191
192    void addData(int start) throws IOException {
193      List<Put> puts = new ArrayList<>();
194      for (int i=start; i< start + 100; i++) {
195        Put put = new Put(Bytes.toBytes(i));
196        put.addColumn(family, family, Bytes.toBytes(i));
197        puts.add(put);
198      }
199      table.put(puts);
200    }
201  }

203  /**
204   * Checks regions using MetaTableAccessor and HTable methods
205   */
206  static class RegionChecker extends ScheduledChore {
207    Connection connection;
208    Configuration conf;
209    TableName tableName;
210    Throwable ex;
211
212    RegionChecker(Configuration conf, Stoppable stopper, TableName tableName) throws IOException {
213      super("RegionChecker", stopper, 10);
214      this.conf = conf;
215      this.tableName = tableName;
216
217      this.connection = ConnectionFactory.createConnection(conf);
218    }
219
220    /** verify region boundaries obtained from MetaScanner */
221    void verifyRegionsUsingMetaTableAccessor() throws Exception {
222      List<RegionInfo> regionList = MetaTableAccessor.getTableRegions(connection, tableName, true);
223      verifyTableRegions(regionList.stream().collect(Collectors.toCollection(() -> new TreeSet<>(RegionInfo.COMPARATOR))));
224      regionList = MetaTableAccessor.getAllRegions(connection, true);
225      verifyTableRegions(regionList.stream().collect(Collectors.toCollection(() -> new TreeSet<>(RegionInfo.COMPARATOR))));
226    }
227
228    /** verify region boundaries obtained from HTable.getStartEndKeys() */
229    void verifyRegionsUsingHTable() throws IOException {
230      Table table = null;
231      try {
232        //HTable.getStartEndKeys()
233        table = connection.getTable(tableName);
234
235        try(RegionLocator rl = connection.getRegionLocator(tableName)) {
236          Pair<byte[][], byte[][]> keys = rl.getStartEndKeys();
237          verifyStartEndKeys(keys);
238
239          Set<RegionInfo> regions = new TreeSet<>(RegionInfo.COMPARATOR);
240          for (HRegionLocation loc : rl.getAllRegionLocations()) {
241            regions.add(loc.getRegionInfo());
242          }
243          verifyTableRegions(regions);
244        }
245
246      } finally {
247        IOUtils.closeQuietly(table);
248      }
249    }
250
251    void verify() throws Exception {
252      verifyRegionsUsingMetaTableAccessor();
253      verifyRegionsUsingHTable();
254    }
255
256    void verifyTableRegions(Set<RegionInfo> regions) {
257      log("Verifying " + regions.size() + " regions: " + regions);
258
259      byte[][] startKeys = new byte[regions.size()][];
260      byte[][] endKeys = new byte[regions.size()][];
261
262      int i=0;
263      for (RegionInfo region : regions) {
264        startKeys[i] = region.getStartKey();
265        endKeys[i] = region.getEndKey();
266        i++;
267      }
268
269      Pair<byte[][], byte[][]> keys = new Pair<>(startKeys, endKeys);
270      verifyStartEndKeys(keys);
271    }
272
273    void verifyStartEndKeys(Pair<byte[][], byte[][]> keys) {
274      byte[][] startKeys = keys.getFirst();
275      byte[][] endKeys = keys.getSecond();
276      assertEquals(startKeys.length, endKeys.length);
277      assertTrue("Found 0 regions for the table", startKeys.length > 0);
278
279      assertArrayEquals("Start key for the first region is not byte[0]",
280          HConstants.EMPTY_START_ROW, startKeys[0]);
281      byte[] prevEndKey = HConstants.EMPTY_START_ROW;
282
283      // ensure that we do not have any gaps
284      for (int i=0; i<startKeys.length; i++) {
285        assertArrayEquals(
286            "Hole in hbase:meta is detected. prevEndKey=" + Bytes.toStringBinary(prevEndKey)
287                + " ,regionStartKey=" + Bytes.toStringBinary(startKeys[i]), prevEndKey,
288            startKeys[i]);
289        prevEndKey = endKeys[i];
290      }
291      assertArrayEquals("End key for the last region is not byte[0]", HConstants.EMPTY_END_ROW,
292          endKeys[endKeys.length - 1]);
293    }
294
295    @Override
296    protected void chore() {
297      try {
298        verify();
299      } catch (Throwable ex) {
300        this.ex = ex;
301        getStopper().stop("caught exception");
302      }
303    }
304  }

312  public static void flushAndBlockUntilDone(Admin admin, HRegionServer rs, byte[] regionName)
313      throws IOException, InterruptedException {
314    log("flushing region: " + Bytes.toStringBinary(regionName));
315    admin.flushRegion(regionName);
316    log("blocking until flush is complete: " + Bytes.toStringBinary(regionName));
317    Threads.sleepWithoutInterrupt(500);
318    while (rs.getOnlineRegion(regionName).getMemStoreSize() > 0) {
319      Threads.sleep(50);
320    }
321  }
322
323  public static void compactAndBlockUntilDone(Admin admin, HRegionServer rs, byte[] regionName)
324      throws IOException, InterruptedException {
325    log("Compacting region: " + Bytes.toStringBinary(regionName));
326    admin.majorCompactRegion(regionName);
327    log("blocking until compaction is complete: " + Bytes.toStringBinary(regionName));
328    Threads.sleepWithoutInterrupt(500);
329    outer: for (;;) {
330      for (Store store : rs.getOnlineRegion(regionName).getStores()) {
331        if (store.getStorefilesCount() > 1) {
332          Threads.sleep(50);
333          continue outer;
334        }
335      }
336      break;
337    }
338  }

340  /** Blocks until the region split is complete in hbase:meta and region server opens the daughters */
341  public static void blockUntilRegionSplit(Configuration conf, long timeout,
342      final byte[] regionName, boolean waitForDaughters)
343      throws IOException, InterruptedException {
344    long start = System.currentTimeMillis();
345    log("blocking until region is split:" +  Bytes.toStringBinary(regionName));
346    RegionInfo daughterA = null, daughterB = null;
347    try (Connection conn = ConnectionFactory.createConnection(conf);
348        Table metaTable = conn.getTable(TableName.META_TABLE_NAME)) {
349      Result result = null;
350      RegionInfo region = null;
351      while ((System.currentTimeMillis() - start) < timeout) {
352        result = metaTable.get(new Get(regionName));
353        if (result == null) {
354          break;
355        }
356
357        region = MetaTableAccessor.getRegionInfo(result);
358        if (region.isSplitParent()) {
359          log("found parent region: " + region.toString());
360          PairOfSameType<RegionInfo> pair = MetaTableAccessor.getDaughterRegions(result);
361          daughterA = pair.getFirst();
362          daughterB = pair.getSecond();
363          break;
364        }
365        Threads.sleep(100);
366      }
367      if (daughterA == null || daughterB == null) {
368        throw new IOException("Failed to get daughters, daughterA=" + daughterA + ", daughterB=" +
369          daughterB + ", timeout=" + timeout + ", result=" + result + ", regionName=" +
370          Bytes.toString(regionName) + ", region=" + region);
371      }
372
373      //if we are here, this means the region split is complete or timed out
374      if (waitForDaughters) {
375        long rem = timeout - (System.currentTimeMillis() - start);
376        blockUntilRegionIsInMeta(conn, rem, daughterA);
377
378        rem = timeout - (System.currentTimeMillis() - start);
379        blockUntilRegionIsInMeta(conn, rem, daughterB);
380
381        rem = timeout - (System.currentTimeMillis() - start);
382        blockUntilRegionIsOpened(conf, rem, daughterA);
383
384        rem = timeout - (System.currentTimeMillis() - start);
385        blockUntilRegionIsOpened(conf, rem, daughterB);
386      }
387    }
388  }
389
390  public static void blockUntilRegionIsInMeta(Connection conn, long timeout, RegionInfo hri)
391      throws IOException, InterruptedException {
392    log("blocking until region is in META: " + hri.getRegionNameAsString());
393    long start = System.currentTimeMillis();
394    while (System.currentTimeMillis() - start < timeout) {
395      HRegionLocation loc = MetaTableAccessor.getRegionLocation(conn, hri);
396      if (loc != null && !loc.getRegionInfo().isOffline()) {
397        log("found region in META: " + hri.getRegionNameAsString());
398        break;
399      }
400      Threads.sleep(10);
401    }
402  }
403
404  public static void blockUntilRegionIsOpened(Configuration conf, long timeout, RegionInfo hri)
405      throws IOException, InterruptedException {
406    log("blocking until region is opened for reading:" + hri.getRegionNameAsString());
407    long start = System.currentTimeMillis();
408    try (Connection conn = ConnectionFactory.createConnection(conf);
409        Table table = conn.getTable(hri.getTable())) {
410      byte[] row = hri.getStartKey();
411      // Check for null/empty row. If we find one, use a key that is likely to be in first region.
412      if (row == null || row.length <= 0) row = new byte[] { '0' };
413      Get get = new Get(row);
414      while (System.currentTimeMillis() - start < timeout) {
415        try {
416          table.get(get);
417          break;
418        } catch (IOException ex) {
419          // wait some more
420        }
421        Threads.sleep(10);
422      }
423    }
