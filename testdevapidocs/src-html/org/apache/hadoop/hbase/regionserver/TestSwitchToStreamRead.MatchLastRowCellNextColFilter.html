053@Category({ RegionServerTests.class, MediumTests.class })
054public class TestSwitchToStreamRead {
055
056  @ClassRule
057  public static final HBaseClassTestRule CLASS_RULE =
058    HBaseClassTestRule.forClass(TestSwitchToStreamRead.class);
059
060  private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
061
062  private static TableName TABLE_NAME = TableName.valueOf("stream");
063
064  private static byte[] FAMILY = Bytes.toBytes("cf");
065
066  private static byte[] QUAL = Bytes.toBytes("cq");
067
068  private static String VALUE_PREFIX;
069
070  private static HRegion REGION;
071
072  @Before
073  public void setUp() throws IOException {
074    UTIL.getConfiguration().setLong(StoreScanner.STORESCANNER_PREAD_MAX_BYTES, 2048);
075    StringBuilder sb = new StringBuilder(256);
076    for (int i = 0; i < 255; i++) {
077      sb.append((char) ThreadLocalRandom.current().nextInt('A', 'z' + 1));
078    }
079    VALUE_PREFIX = sb.append("-").toString();
080    REGION = UTIL.createLocalHRegion(
081      TableDescriptorBuilder.newBuilder(TABLE_NAME)
082        .setColumnFamily(
083          ColumnFamilyDescriptorBuilder.newBuilder(FAMILY).setBlocksize(1024).build())
084        .build(),
085      null, null);
086    for (int i = 0; i < 900; i++) {
087      REGION
088        .put(new Put(Bytes.toBytes(i)).addColumn(FAMILY, QUAL, Bytes.toBytes(VALUE_PREFIX + i)));
089    }
090    REGION.flush(true);
091    for (int i = 900; i < 1000; i++) {
092      REGION
093        .put(new Put(Bytes.toBytes(i)).addColumn(FAMILY, QUAL, Bytes.toBytes(VALUE_PREFIX + i)));
094    }
095  }
096
097  @After
098  public void tearDown() throws IOException {
099    REGION.close(true);
100    UTIL.cleanupTestDir();
101  }
102
103  @Test
104  public void test() throws IOException {
105    try (RegionScannerImpl scanner = REGION.getScanner(new Scan())) {
106      StoreScanner storeScanner =
107        (StoreScanner) (scanner).getStoreHeapForTesting().getCurrentForTesting();
108      for (KeyValueScanner kvs : storeScanner.getAllScannersForTesting()) {
109        if (kvs instanceof StoreFileScanner) {
110          StoreFileScanner sfScanner = (StoreFileScanner) kvs;
111          // starting from pread so we use shared reader here.
112          assertTrue(sfScanner.getReader().getReaderContext()
113            .getReaderType() == ReaderType.PREAD);
114        }
115      }
116      List<Cell> cells = new ArrayList<>();
117      for (int i = 0; i < 500; i++) {
118        assertTrue(scanner.next(cells));
119        Result result = Result.create(cells);
120        assertEquals(VALUE_PREFIX + i, Bytes.toString(result.getValue(FAMILY, QUAL)));
121        cells.clear();
122        scanner.shipped();
123      }
124      for (KeyValueScanner kvs : storeScanner.getAllScannersForTesting()) {
125        if (kvs instanceof StoreFileScanner) {
126          StoreFileScanner sfScanner = (StoreFileScanner) kvs;
127          // we should have convert to use stream read now.
128          assertFalse(sfScanner.getReader().getReaderContext()
129            .getReaderType() == ReaderType.PREAD);
130        }
131      }
132      for (int i = 500; i < 1000; i++) {
133        assertEquals(i != 999, scanner.next(cells));
134        Result result = Result.create(cells);
135        assertEquals(VALUE_PREFIX + i, Bytes.toString(result.getValue(FAMILY, QUAL)));
136        cells.clear();
137        scanner.shipped();
138      }
139    }
140    // make sure all scanners are closed.
141    for (HStoreFile sf : REGION.getStore(FAMILY).getStorefiles()) {
142      assertFalse(sf.isReferencedInReads());
143    }
144  }

146  public static final class MatchLastRowKeyFilter extends FilterBase {
147
148    @Override
149    public boolean filterRowKey(Cell cell) throws IOException {
150      return Bytes.toInt(cell.getRowArray(), cell.getRowOffset()) != 999;
151    }
152  }
153
154  private void testFilter(Filter filter) throws IOException {
155    try (RegionScannerImpl scanner = REGION.getScanner(new Scan().setFilter(filter))) {
156      StoreScanner storeScanner =
157        (StoreScanner) (scanner).getStoreHeapForTesting().getCurrentForTesting();
158      for (KeyValueScanner kvs : storeScanner.getAllScannersForTesting()) {
159        if (kvs instanceof StoreFileScanner) {
160          StoreFileScanner sfScanner = (StoreFileScanner) kvs;
161          // starting from pread so we use shared reader here.
162          assertTrue(sfScanner.getReader().getReaderContext()
163            .getReaderType() == ReaderType.PREAD);
164        }
165      }
166      List<Cell> cells = new ArrayList<>();
167      // should return before finishing the scan as we want to switch from pread to stream
168      assertTrue(scanner.next(cells,
169        ScannerContext.newBuilder().setTimeLimit(LimitScope.BETWEEN_CELLS, -1).build()));
170      assertTrue(cells.isEmpty());
171      scanner.shipped();
172
173      for (KeyValueScanner kvs : storeScanner.getAllScannersForTesting()) {
174        if (kvs instanceof StoreFileScanner) {
175          StoreFileScanner sfScanner = (StoreFileScanner) kvs;
176          // we should have convert to use stream read now.
177          assertFalse(sfScanner.getReader().getReaderContext()
178            .getReaderType() == ReaderType.PREAD);
179        }
180      }
181      assertFalse(scanner.next(cells,
182        ScannerContext.newBuilder().setTimeLimit(LimitScope.BETWEEN_CELLS, -1).build()));
183      Result result = Result.create(cells);
184      assertEquals(VALUE_PREFIX + 999, Bytes.toString(result.getValue(FAMILY, QUAL)));
185      cells.clear();
186      scanner.shipped();
187    }
188    // make sure all scanners are closed.
189    for (HStoreFile sf : REGION.getStore(FAMILY).getStorefiles()) {
190      assertFalse(sf.isReferencedInReads());
191    }
192  }
193
194  // We use a different logic to implement filterRowKey, where we will keep calling kvHeap.next
195  // until the row key is changed. And there we can only use NoLimitScannerContext so we can not
196  // make the upper layer return immediately. Simply do not use NoLimitScannerContext will lead to
197  // an infinite loop. Need to dig more, the code are way too complicated...
198  @Ignore
199  @Test
200  public void testFilterRowKey() throws IOException {
201    testFilter(new MatchLastRowKeyFilter());
202  }
