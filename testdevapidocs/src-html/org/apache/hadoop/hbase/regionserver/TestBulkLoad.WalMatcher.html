110  @Test
111  public void verifyBulkLoadEvent() throws IOException {
112    TableName tableName = TableName.valueOf("test", "test");
113    List<Pair<byte[], String>> familyPaths = withFamilyPathsFor(family1);
114    byte[] familyName = familyPaths.get(0).getFirst();
115    String storeFileName = familyPaths.get(0).getSecond();
116    storeFileName = (new Path(storeFileName)).getName();
117    List<String> storeFileNames = new ArrayList<>();
118    storeFileNames.add(storeFileName);
119    when(log.appendMarker(any(), any(),
120      argThat(bulkLogWalEdit(WALEdit.BULK_LOAD, tableName.toBytes(), familyName, storeFileNames))))
121        .thenAnswer(new Answer() {
122          @Override
123          public Object answer(InvocationOnMock invocation) {
124            WALKeyImpl walKey = invocation.getArgument(1);
125            MultiVersionConcurrencyControl mvcc = walKey.getMvcc();
126            if (mvcc != null) {
127              MultiVersionConcurrencyControl.WriteEntry we = mvcc.begin();
128              walKey.setWriteEntry(we);
129            }
130            return 01L;
131          }
132        });
133    testRegionWithFamiliesAndSpecifiedTableName(tableName, family1)
134        .bulkLoadHFiles(familyPaths, false, null);
135    verify(log).sync(anyLong());
136  }

260  private HRegion testRegionWithFamiliesAndSpecifiedTableName(TableName tableName,
261    byte[]... families) throws IOException {
262    RegionInfo hRegionInfo = RegionInfoBuilder.newBuilder(tableName).build();
263    TableDescriptorBuilder.ModifyableTableDescriptor tableDescriptor =
264      new TableDescriptorBuilder.ModifyableTableDescriptor(tableName);
265
266    for (byte[] family : families) {
267      tableDescriptor.setColumnFamily(
268        new ColumnFamilyDescriptorBuilder.ModifyableColumnFamilyDescriptor(family));
269    }
270    ChunkCreator.initialize(MemStoreLABImpl.CHUNK_SIZE_DEFAULT, false, 0, 0, 0, null);
271    // TODO We need a way to do this without creating files
272    return HRegion.createHRegion(hRegionInfo, new Path(testFolder.newFolder().toURI()), conf,
273      tableDescriptor, log);
274  }

285  private List<Pair<byte[], String>> withFamilyPathsFor(byte[]... families) throws IOException {
286    List<Pair<byte[], String>> familyPaths = getBlankFamilyPaths();
287    for (byte[] family : families) {
288      familyPaths.add(new Pair<>(family, createHFileForFamilies(family)));
289    }
290    return familyPaths;
291  }
292
293  private String createHFileForFamilies(byte[] family) throws IOException {
294    HFile.WriterFactory hFileFactory = HFile.getWriterFactoryNoCache(conf);
295    // TODO We need a way to do this without creating files
296    File hFileLocation = testFolder.newFile();
297    FSDataOutputStream out = new FSDataOutputStream(new FileOutputStream(hFileLocation), null);
298    try {
299      hFileFactory.withOutputStream(out);
300      hFileFactory.withFileContext(new HFileContextBuilder().build());
301      HFile.Writer writer = hFileFactory.create();
302      try {
303        writer.append(new KeyValue(ExtendedCellBuilderFactory.create(CellBuilderType.DEEP_COPY)
304          .setRow(randomBytes)
305          .setFamily(family)
306          .setQualifier(randomBytes)
307          .setTimestamp(0L)
308          .setType(KeyValue.Type.Put.getCode())
309          .setValue(randomBytes)
310          .build()));
311      } finally {
312        writer.close();
313      }
314    } finally {
315      out.close();
316    }
317    return hFileLocation.getAbsoluteFile().getAbsolutePath();
318  }

324  private static Matcher<WALEdit> bulkLogWalEdit(byte[] typeBytes, byte[] tableName,
325      byte[] familyName, List<String> storeFileNames) {
326    return new WalMatcher(typeBytes, tableName, familyName, storeFileNames);
327  }

329  private static class WalMatcher extends TypeSafeMatcher<WALEdit> {
330    private final byte[] typeBytes;
331    private final byte[] tableName;
332    private final byte[] familyName;
333    private final List<String> storeFileNames;
334
335    public WalMatcher(byte[] typeBytes) {
336      this(typeBytes, null, null, null);
337    }
338
339    public WalMatcher(byte[] typeBytes, byte[] tableName, byte[] familyName,
340        List<String> storeFileNames) {
341      this.typeBytes = typeBytes;
342      this.tableName = tableName;
343      this.familyName = familyName;
344      this.storeFileNames = storeFileNames;
345    }
346
347    @Override
348    protected boolean matchesSafely(WALEdit item) {
349      assertTrue(Arrays.equals(CellUtil.cloneQualifier(item.getCells().get(0)), typeBytes));
350      BulkLoadDescriptor desc;
351      try {
352        desc = WALEdit.getBulkLoadDescriptor(item.getCells().get(0));
353      } catch (IOException e) {
354        return false;
355      }
356      assertNotNull(desc);
357
358      if (tableName != null) {
359        assertTrue(Bytes.equals(ProtobufUtil.toTableName(desc.getTableName()).getName(),
360          tableName));
361      }
362
363      if(storeFileNames != null) {
364        int index=0;
365        StoreDescriptor store = desc.getStores(0);
366        assertTrue(Bytes.equals(store.getFamilyName().toByteArray(), familyName));
367        assertTrue(Bytes.equals(Bytes.toBytes(store.getStoreHomeDir()), familyName));
368        assertEquals(storeFileNames.size(), store.getStoreFileCount());
369      }
370
371      return true;
372    }
373
374    @Override
375    public void describeTo(Description description) {
376
377    }
378  }
