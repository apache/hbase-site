053/**
054 * Tests that need to spin up a cluster testing an {@link HRegion}.  Use
055 * {@link TestHRegion} if you don't need a cluster, if you can test w/ a
056 * standalone {@link HRegion}.
057 */
058@Category({RegionServerTests.class, MediumTests.class})
059public class TestHRegionOnCluster {
060
061  @ClassRule
062  public static final HBaseClassTestRule CLASS_RULE =
063      HBaseClassTestRule.forClass(TestHRegionOnCluster.class);
064
065  private static final Logger LOG = LoggerFactory.getLogger(TestHRegionOnCluster.class);
066  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
067
068  @Rule
069  public TestName name = new TestName();
070
071  @Test
072  public void testDataCorrectnessReplayingRecoveredEdits() throws Exception {
073    final int NUM_RS = 3;
074    Admin hbaseAdmin = null;
075    TEST_UTIL.startMiniCluster(NUM_RS);
076
077    try {
078      final TableName tableName = TableName.valueOf(name.getMethodName());
079      final byte[] FAMILY = Bytes.toBytes("family");
080      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
081      HMaster master = cluster.getMaster();
082
083      // Create table
084      TableDescriptorBuilder.ModifyableTableDescriptor tableDescriptor =
085        new TableDescriptorBuilder.ModifyableTableDescriptor(tableName);
086      tableDescriptor.setColumnFamily(
087        new ColumnFamilyDescriptorBuilder.ModifyableColumnFamilyDescriptor(FAMILY));
088      hbaseAdmin = master.getConnection().getAdmin();
089      hbaseAdmin.createTable(tableDescriptor);
090
091      assertTrue(hbaseAdmin.isTableAvailable(tableName));
092
093      // Put data: r1->v1
094      LOG.info("Loading r1 to v1 into " + tableName);
095      Table table = TEST_UTIL.getConnection().getTable(tableName);
096      putDataAndVerify(table, "r1", FAMILY, "v1", 1);
097
098      TEST_UTIL.waitUntilAllRegionsAssigned(table.getName());
099      // Move region to target server
100
101      RegionInfo regionInfo;
102      try (RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
103        regionInfo = locator.getRegionLocation(Bytes.toBytes("r1")).getRegion();
104      }
105
106      int originServerNum = cluster.getServerWith(regionInfo.getRegionName());
107      HRegionServer originServer = cluster.getRegionServer(originServerNum);
108      int targetServerNum = (originServerNum + 1) % NUM_RS;
109      HRegionServer targetServer = cluster.getRegionServer(targetServerNum);
110      assertFalse(originServer.equals(targetServer));
111
112      TEST_UTIL.waitUntilAllRegionsAssigned(table.getName());
113      LOG.info("Moving " + regionInfo.getEncodedName() + " to " + targetServer.getServerName());
114      hbaseAdmin.move(regionInfo.getEncodedNameAsBytes(), targetServer.getServerName());
115      do {
116        Thread.sleep(1);
117      } while (cluster.getServerWith(regionInfo.getRegionName()) == originServerNum);
118
119      // Put data: r2->v2
120      LOG.info("Loading r2 to v2 into " + tableName);
121      putDataAndVerify(table, "r2", FAMILY, "v2", 2);
122
123      TEST_UTIL.waitUntilAllRegionsAssigned(table.getName());
124      // Move region to origin server
125      LOG.info("Moving " + regionInfo.getEncodedName() + " to " + originServer.getServerName());
126      hbaseAdmin.move(regionInfo.getEncodedNameAsBytes(), originServer.getServerName());
127      do {
128        Thread.sleep(1);
129      } while (cluster.getServerWith(regionInfo.getRegionName()) == targetServerNum);
130
131      // Put data: r3->v3
132      LOG.info("Loading r3 to v3 into " + tableName);
133      putDataAndVerify(table, "r3", FAMILY, "v3", 3);
134
135      // Kill target server
136      LOG.info("Killing target server " + targetServer.getServerName());
137      targetServer.kill();
138      cluster.getRegionServerThreads().get(targetServerNum).join();
139      // Wait until finish processing of shutdown
140      while (master.getServerManager().areDeadServersInProgress()) {
141        Thread.sleep(5);
142      }
143      // Kill origin server
144      LOG.info("Killing origin server " + targetServer.getServerName());
145      originServer.kill();
146      cluster.getRegionServerThreads().get(originServerNum).join();
147
148      // Put data: r4->v4
149      LOG.info("Loading r4 to v4 into " + tableName);
150      putDataAndVerify(table, "r4", FAMILY, "v4", 4);
151
152    } finally {
153      if (hbaseAdmin != null) hbaseAdmin.close();
154      TEST_UTIL.shutdownMiniCluster();
155    }
156  }
157
158  private void putDataAndVerify(Table table, String row, byte[] family,
159      String value, int verifyNum) throws IOException {
160    System.out.println("=========Putting data :" + row);
161    Put put = new Put(Bytes.toBytes(row));
162    put.addColumn(family, Bytes.toBytes("q1"), Bytes.toBytes(value));
163    table.put(put);
164    ResultScanner resultScanner = table.getScanner(new Scan());
165    List<Result> results = new ArrayList<>();
166    while (true) {
167      Result r = resultScanner.next();
168      if (r == null)
169        break;
170      results.add(r);
171    }
172    resultScanner.close();
173    if (results.size() != verifyNum) {
174      System.out.println(results);
175    }
176    assertEquals(verifyNum, results.size());
177  }
