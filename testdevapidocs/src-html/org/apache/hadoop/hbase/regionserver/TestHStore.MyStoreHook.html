232  @Test
233  public void testFlushSizeSizing() throws Exception {
234    LOG.info("Setting up a faulty file system that cannot write in " +
235      this.name.getMethodName());
236    final Configuration conf = HBaseConfiguration.create();
237    // Only retry once.
238    conf.setInt("hbase.hstore.flush.retries.number", 1);
239    User user = User.createUserForTesting(conf, this.name.getMethodName(),
240      new String[]{"foo"});
241    // Inject our faulty LocalFileSystem
242    conf.setClass("fs.file.impl", FaultyFileSystem.class, FileSystem.class);
243    user.runAs(new PrivilegedExceptionAction<Object>() {
244      @Override
245      public Object run() throws Exception {
246        // Make sure it worked (above is sensitive to caching details in hadoop core)
247        FileSystem fs = FileSystem.get(conf);
248        assertEquals(FaultyFileSystem.class, fs.getClass());
249        FaultyFileSystem ffs = (FaultyFileSystem)fs;
250
251        // Initialize region
252        init(name.getMethodName(), conf);
253
254        MemStoreSize size = store.memstore.getFlushableSize();
255        assertEquals(0, size.getDataSize());
256        LOG.info("Adding some data");
257        MemStoreSizing kvSize = new MemStoreSizing();
258        store.add(new KeyValue(row, family, qf1, 1, (byte[]) null), kvSize);
259        // add the heap size of active (mutable) segment
260        kvSize.incMemStoreSize(0, MutableSegment.DEEP_OVERHEAD);
261        size = store.memstore.getFlushableSize();
262        assertEquals(kvSize, size);
263        // Flush.  Bug #1 from HBASE-10466.  Make sure size calculation on failed flush is right.
264        try {
265          LOG.info("Flushing");
266          flushStore(store, id++);
267          fail("Didn't bubble up IOE!");
268        } catch (IOException ioe) {
269          assertTrue(ioe.getMessage().contains("Fault injected"));
270        }
271        // due to snapshot, change mutable to immutable segment
272        kvSize.incMemStoreSize(0,
273            CSLMImmutableSegment.DEEP_OVERHEAD_CSLM-MutableSegment.DEEP_OVERHEAD);
274        size = store.memstore.getFlushableSize();
275        assertEquals(kvSize, size);
276        MemStoreSizing kvSize2 = new MemStoreSizing();
277        store.add(new KeyValue(row, family, qf2, 2, (byte[])null), kvSize2);
278        kvSize2.incMemStoreSize(0, MutableSegment.DEEP_OVERHEAD);
279        // Even though we add a new kv, we expect the flushable size to be 'same' since we have
280        // not yet cleared the snapshot -- the above flush failed.
281        assertEquals(kvSize, size);
282        ffs.fault.set(false);
283        flushStore(store, id++);
284        size = store.memstore.getFlushableSize();
285        // Size should be the foreground kv size.
286        assertEquals(kvSize2, size);
287        flushStore(store, id++);
288        size = store.memstore.getFlushableSize();
289        assertEquals(0, size.getDataSize());
290        assertEquals(MutableSegment.DEEP_OVERHEAD, size.getHeapSize());
291        return null;
292      }
293    });
294  }

334   * @param minVersions the MIN_VERSIONS for the column family
335   */
336  public void testDeleteExpiredStoreFiles(int minVersions) throws Exception {
337    int storeFileNum = 4;
338    int ttl = 4;
339    IncrementingEnvironmentEdge edge = new IncrementingEnvironmentEdge();
340    EnvironmentEdgeManagerTestHelper.injectEdge(edge);
341
342    Configuration conf = HBaseConfiguration.create();
343    // Enable the expired store file deletion
344    conf.setBoolean("hbase.store.delete.expired.storefile", true);
345    // Set the compaction threshold higher to avoid normal compactions.
346    conf.setInt(CompactionConfiguration.HBASE_HSTORE_COMPACTION_MIN_KEY, 5);
347
348    init(name.getMethodName() + "-" + minVersions, conf, ColumnFamilyDescriptorBuilder
349        .newBuilder(family).setMinVersions(minVersions).setTimeToLive(ttl).build());
350
351    long storeTtl = this.store.getScanInfo().getTtl();
352    long sleepTime = storeTtl / storeFileNum;
353    long timeStamp;
354    // There are 4 store files and the max time stamp difference among these
355    // store files will be (this.store.ttl / storeFileNum)
356    for (int i = 1; i <= storeFileNum; i++) {
357      LOG.info("Adding some data for the store file #" + i);
358      timeStamp = EnvironmentEdgeManager.currentTime();
359      this.store.add(new KeyValue(row, family, qf1, timeStamp, (byte[]) null), null);
360      this.store.add(new KeyValue(row, family, qf2, timeStamp, (byte[]) null), null);
361      this.store.add(new KeyValue(row, family, qf3, timeStamp, (byte[]) null), null);
362      flush(i);
363      edge.incrementTime(sleepTime);
364    }
365
366    // Verify the total number of store files
367    assertEquals(storeFileNum, this.store.getStorefiles().size());
368
369     // Each call will find one expired store file and delete it before compaction happens.
370     // There will be no compaction due to threshold above. Last file will not be replaced.
371    for (int i = 1; i <= storeFileNum - 1; i++) {
372      // verify the expired store file.
373      assertFalse(this.store.requestCompaction().isPresent());
374      Collection<HStoreFile> sfs = this.store.getStorefiles();
375      // Ensure i files are gone.
376      if (minVersions == 0) {
377        assertEquals(storeFileNum - i, sfs.size());
378        // Ensure only non-expired files remain.
379        for (HStoreFile sf : sfs) {
380          assertTrue(sf.getReader().getMaxTimestamp() >= (edge.currentTime() - storeTtl));
381        }
382      } else {
383        assertEquals(storeFileNum, sfs.size());
384      }
385      // Let the next store file expired.
386      edge.incrementTime(sleepTime);
387    }
388    assertFalse(this.store.requestCompaction().isPresent());
389
390    Collection<HStoreFile> sfs = this.store.getStorefiles();
391    // Assert the last expired file is not removed.
392    if (minVersions == 0) {
393      assertEquals(1, sfs.size());
394    }
395    long ts = sfs.iterator().next().getReader().getMaxTimestamp();
396    assertTrue(ts < (edge.currentTime() - storeTtl));
397
398    for (HStoreFile sf : sfs) {
399      sf.closeStoreFile(true);
400    }
401  }

430  private static long getLowestTimeStampFromFS(FileSystem fs,
431      final Collection<HStoreFile> candidates) throws IOException {
432    long minTs = Long.MAX_VALUE;
433    if (candidates.isEmpty()) {
434      return minTs;
435    }
436    Path[] p = new Path[candidates.size()];
437    int i = 0;
438    for (HStoreFile sf : candidates) {
439      p[i] = sf.getPath();
440      ++i;
441    }
442
443    FileStatus[] stats = fs.listStatus(p);
444    if (stats == null || stats.length == 0) {
445      return minTs;
446    }
447    for (FileStatus s : stats) {
448      minTs = Math.min(minTs, s.getModificationTime());
449    }
450    return minTs;
451  }
