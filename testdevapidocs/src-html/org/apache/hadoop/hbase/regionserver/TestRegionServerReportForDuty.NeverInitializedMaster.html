126  /**
127   * Tests region server should backoff to report for duty if master is not ready.
128   */
129  @Test
130  public void testReportForDutyBackoff() throws IOException, InterruptedException {
131    cluster.getConfiguration().set(HConstants.MASTER_IMPL, NeverInitializedMaster.class.getName());
132    master = cluster.addMaster();
133    master.start();
134
135    LogCapturer capturer =
136      new LogCapturer((org.apache.logging.log4j.core.Logger) org.apache.logging.log4j.LogManager
137        .getLogger(HRegionServer.class));
138    // Set sleep interval relatively low so that exponential backoff is more demanding.
139    int msginterval = 100;
140    cluster.getConfiguration().setInt("hbase.regionserver.msginterval", msginterval);
141    rs = cluster.addRegionServer();
142    rs.start();
143
144    int interval = 10_000;
145    Thread.sleep(interval);
146    capturer.stopCapturing();
147    String output = capturer.getOutput();
148    LOG.info("{}", output);
149    String failMsg = "reportForDuty failed;";
150    int count = StringUtils.countMatches(output, failMsg);
151
152    // Following asserts the actual retry number is in range (expectedRetry/2, expectedRetry*2).
153    // Ideally we can assert the exact retry count. We relax here to tolerate contention error.
154    int expectedRetry = (int)Math.ceil(Math.log(interval - msginterval));
155    assertTrue(String.format("reportForDuty retries %d times, less than expected min %d",
156        count, expectedRetry / 2), count > expectedRetry / 2);
157    assertTrue(String.format("reportForDuty retries %d times, more than expected max %d",
158        count, expectedRetry * 2), count < expectedRetry * 2);
159  }

161  /**
162   * Tests region sever reportForDuty with backup master becomes primary master after
163   * the first master goes away.
164   */
165  @Test
166  public void testReportForDutyWithMasterChange() throws Exception {
167
168    // Start a master and wait for it to become the active/primary master.
169    // Use a random unique port
170    cluster.getConfiguration().setInt(HConstants.MASTER_PORT, HBaseTestingUtility.randomFreePort());
171    // master has a rs. defaultMinToStart = 2
172    boolean tablesOnMaster = LoadBalancer.isTablesOnMaster(testUtil.getConfiguration());
173    cluster.getConfiguration().setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, tablesOnMaster? 2: 1);
174    cluster.getConfiguration().setInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART, tablesOnMaster? 2: 1);
175    master = cluster.addMaster();
176    rs = cluster.addRegionServer();
177    LOG.debug("Starting master: " + master.getMaster().getServerName());
178    master.start();
179    rs.start();
180
181    waitForClusterOnline(master);
182
183    // Add a 2nd region server
184    cluster.getConfiguration().set(HConstants.REGION_SERVER_IMPL, MyRegionServer.class.getName());
185    rs2 = cluster.addRegionServer();
186    // Start the region server. This region server will refresh RPC connection
187    // from the current active master to the next active master before completing
188    // reportForDuty
189    LOG.debug("Starting 2nd region server: " + rs2.getRegionServer().getServerName());
190    rs2.start();
191
192    waitForSecondRsStarted();
193
194    // Stop the current master.
195    master.getMaster().stop("Stopping master");
196
197    // Start a new master and use another random unique port
198    // Also let it wait for exactly 2 region severs to report in.
199    // TODO: Add handling bindexception. Random port is not enough!!! Flakie test!
200    cluster.getConfiguration().setInt(HConstants.MASTER_PORT, HBaseTestingUtility.randomFreePort());
201    cluster.getConfiguration().setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART,
202      tablesOnMaster? 3: 2);
203    cluster.getConfiguration().setInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART,
204      tablesOnMaster? 3: 2);
205    backupMaster = cluster.addMaster();
206    LOG.debug("Starting new master: " + backupMaster.getMaster().getServerName());
207    backupMaster.start();
208
209    waitForClusterOnline(backupMaster);
210
211    // Do some checking/asserts here.
212    assertTrue(backupMaster.getMaster().isActiveMaster());
213    assertTrue(backupMaster.getMaster().isInitialized());
214    assertEquals(backupMaster.getMaster().getServerManager().getOnlineServersList().size(),
215      tablesOnMaster? 3: 2);
216
217  }

219  /**
220   * Tests region sever reportForDuty with RS RPC retry
221   */
222  @Test
223  public void testReportForDutyWithRSRpcRetry() throws Exception {
224    ScheduledThreadPoolExecutor scheduledThreadPoolExecutor = new ScheduledThreadPoolExecutor(1,
225      new ThreadFactoryBuilder().setNameFormat("RSDelayedStart-pool-%d").setDaemon(true)
226        .setUncaughtExceptionHandler(Threads.LOGGING_EXCEPTION_HANDLER).build());
227
228    // Start a master and wait for it to become the active/primary master.
229    // Use a random unique port
230    cluster.getConfiguration().setInt(HConstants.MASTER_PORT, HBaseTestingUtility.randomFreePort());
231    // Override the default RS RPC retry interval of 100ms to 300ms
232    cluster.getConfiguration().setLong("hbase.regionserver.rpc.retry.interval", 300);
233    // master has a rs. defaultMinToStart = 2
234    boolean tablesOnMaster = LoadBalancer.isTablesOnMaster(testUtil.getConfiguration());
235    cluster.getConfiguration().setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART,
236      tablesOnMaster ? 2 : 1);
237    cluster.getConfiguration().setInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART,
238      tablesOnMaster ? 2 : 1);
239    master = cluster.addMaster();
240    rs = cluster.addRegionServer();
241    LOG.debug("Starting master: " + master.getMaster().getServerName());
242    master.start();
243    // Delay the RS start so that the meta assignment fails in first attempt and goes to retry block
244    scheduledThreadPoolExecutor.schedule(new Runnable() {
245      @Override
246      public void run() {
247        rs.start();
248      }
249    }, 1000, TimeUnit.MILLISECONDS);
250
251    waitForClusterOnline(master);
252  }

269      tablesOnMaster ? 2 : 1);
270    cluster.getConfiguration().setInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART,
271      tablesOnMaster ? 2 : 1);
272
273    // Inject manual environment edge for clock skew computation between RS and master
274    ManualEnvironmentEdge edge = new ManualEnvironmentEdge();
275    EnvironmentEdgeManager.injectEdge(edge);
276    master = cluster.addMaster();
277    rs = cluster.addRegionServer();
278    LOG.debug("Starting master: " + master.getMaster().getServerName());
279    master.start();
280    rs.start();
281
282    waitForClusterOnline(master);

285  private void waitForClusterOnline(MasterThread master) throws InterruptedException {
286    while (true) {
287      if (master.getMaster().isInitialized()) {
288        break;
289      }
290      Thread.sleep(SLEEP_INTERVAL);
291      LOG.debug("Waiting for master to come online ...");
292    }
293    rs.waitForServerOnline();
294  }

296  private void waitForSecondRsStarted() throws InterruptedException {
297    while (true) {
298      if (((MyRegionServer) rs2.getRegionServer()).getRpcStubCreatedFlag() == true) {
299        break;
300      }
301      Thread.sleep(SLEEP_INTERVAL);
302      LOG.debug("Waiting 2nd RS to be started ...");
303    }
304  }

306  // Create a Region Server that provide a hook so that we can wait for the master switch over
307  // before continuing reportForDuty to the mater.
308  // The idea is that we get a RPC connection to the first active master, then we wait.
309  // The first master goes down, the second master becomes the active master. The region
310  // server continues reportForDuty. It should succeed with the new master.
311  public static class MyRegionServer extends MiniHBaseClusterRegionServer {
312
313    private ServerName sn;
314    // This flag is to make sure this rs has obtained the rpcStub to the first master.
315    // The first master will go down after this.
316    private boolean rpcStubCreatedFlag = false;
317    private boolean masterChanged = false;
318
319    public MyRegionServer(Configuration conf) throws IOException, KeeperException,
320        InterruptedException {
321      super(conf);
322    }
323
324    @Override
325    protected synchronized ServerName createRegionServerStatusStub(boolean refresh) {
326      sn = super.createRegionServerStatusStub(refresh);
327      rpcStubCreatedFlag = true;
328
329      // Wait for master switch over. Only do this for the second region server.
330      while (!masterChanged) {
331        ServerName newSn = super.getMasterAddressTracker().getMasterAddress(true);
332        if (newSn != null && !newSn.equals(sn)) {
333          masterChanged = true;
334          break;
335        }
336        try {
337          Thread.sleep(SLEEP_INTERVAL);
338        } catch (InterruptedException e) {
339          return null;
340        }
341        LOG.debug("Waiting for master switch over ... ");
342      }
343      return sn;
344    }
345
346    public boolean getRpcStubCreatedFlag() {
347      return rpcStubCreatedFlag;
348    }
349  }
