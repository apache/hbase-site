035  @Test
036  public void testCompactionRatio() throws IOException {
037    /**
038     * NOTE: these tests are specific to describe the implementation of the
039     * current compaction algorithm.  Developed to ensure that refactoring
040     * doesn't implicitly alter this.
041     */
042    long tooBig = maxSize + 1;
043
044    // default case. preserve user ratio on size
045    compactEquals(sfCreate(100,50,23,12,12), 23, 12, 12);
046    // less than compact threshold = don't compact
047    compactEquals(sfCreate(100,50,25,12,12) /* empty */);
048    // greater than compact size = skip those
049    compactEquals(sfCreate(tooBig, tooBig, 700, 700, 700), 700, 700, 700);
050    // big size + threshold
051    compactEquals(sfCreate(tooBig, tooBig, 700,700) /* empty */);
052    // small files = don't care about ratio
053    compactEquals(sfCreate(7,1,1), 7,1,1);
054
055    // don't exceed max file compact threshold
056    // note:  file selection starts with largest to smallest.
057    compactEquals(sfCreate(7, 6, 5, 4, 3, 2, 1), 5, 4, 3, 2, 1);
058
059    compactEquals(sfCreate(50, 10, 10 ,10, 10), 10, 10, 10, 10);
060
061    compactEquals(sfCreate(10, 10, 10, 10, 50), 10, 10, 10, 10);
062
063    compactEquals(sfCreate(251, 253, 251, maxSize -1), 251, 253, 251);
064
065    compactEquals(sfCreate(maxSize -1,maxSize -1,maxSize -1) /* empty */);
066
067    // Always try and compact something to get below blocking storefile count
068    this.conf.setLong("hbase.hstore.compaction.min.size", 1);
069    store.storeEngine.getCompactionPolicy().setConf(conf);
070    compactEquals(sfCreate(512,256,128,64,32,16,8,4,2,1), 4,2,1);
071    this.conf.setLong("hbase.hstore.compaction.min.size", minSize);
072    store.storeEngine.getCompactionPolicy().setConf(conf);
073
074    /* MAJOR COMPACTION */
075    // if a major compaction has been forced, then compact everything
076    compactEquals(sfCreate(50,25,12,12), true, 50, 25, 12, 12);
077    // also choose files < threshold on major compaction
078    compactEquals(sfCreate(12,12), true, 12, 12);
079    // even if one of those files is too big
080    compactEquals(sfCreate(tooBig, 12,12), true, tooBig, 12, 12);
081    // don't exceed max file compact threshold, even with major compaction
082    store.forceMajor = true;
083    compactEquals(sfCreate(7, 6, 5, 4, 3, 2, 1), 5, 4, 3, 2, 1);
084    store.forceMajor = false;
085    // if we exceed maxCompactSize, downgrade to minor
086    // if not, it creates a 'snowball effect' when files >> maxCompactSize:
087    // the last file in compaction is the aggregate of all previous compactions
088    compactEquals(sfCreate(100,50,23,12,12), true, 23, 12, 12);
089    conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, 1);
090    conf.setFloat("hbase.hregion.majorcompaction.jitter", 0);
091    store.storeEngine.getCompactionPolicy().setConf(conf);
092    try {
093      // trigger an aged major compaction
094      compactEquals(sfCreate(50,25,12,12), 50, 25, 12, 12);
095      // major sure exceeding maxCompactSize also downgrades aged minors
096      compactEquals(sfCreate(100,50,23,12,12), 23, 12, 12);
097    } finally {
098      conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, 1000*60*60*24);
099      conf.setFloat("hbase.hregion.majorcompaction.jitter", 0.20F);
100    }
101
102    /* REFERENCES == file is from a region that was split */
103    // treat storefiles that have references like a major compaction
104    compactEquals(sfCreate(true, 100,50,25,12,12), 100, 50, 25, 12, 12);
105    // reference files shouldn't obey max threshold
106    compactEquals(sfCreate(true, tooBig, 12,12), tooBig, 12, 12);
107    // reference files should obey max file compact to avoid OOM
108    compactEquals(sfCreate(true, 7, 6, 5, 4, 3, 2, 1), 7, 6, 5, 4, 3);
109
110    // empty case
111    compactEquals(new ArrayList<StoreFile>() /* empty */);
112    // empty case (because all files are too big)
113    compactEquals(sfCreate(tooBig, tooBig) /* empty */);
114  }

116  @Test
117  public void testOffPeakCompactionRatio() throws IOException {
118    /*
119     * NOTE: these tests are specific to describe the implementation of the
120     * current compaction algorithm.  Developed to ensure that refactoring
121     * doesn't implicitly alter this.
122     */
123    // set an off-peak compaction threshold
124    this.conf.setFloat("hbase.hstore.compaction.ratio.offpeak", 5.0F);
125    store.storeEngine.getCompactionPolicy().setConf(this.conf);
126    // Test with and without the flag.
127    compactEquals(sfCreate(999, 50, 12, 12, 1), false, true, 50, 12, 12, 1);
128    compactEquals(sfCreate(999, 50, 12, 12, 1), 12, 12, 1);
129  }

131  @Test
132  public void testStuckStoreCompaction() throws IOException {
133    // Select the smallest compaction if the store is stuck.
134    compactEquals(sfCreate(99,99,99,99,99,99, 30,30,30,30), 30, 30, 30);
135    // If not stuck, standard policy applies.
136    compactEquals(sfCreate(99,99,99,99,99, 30,30,30,30), 99, 30, 30, 30, 30);
137
138    // Add sufficiently small files to compaction, though
139    compactEquals(sfCreate(99,99,99,99,99,99, 30,30,30,15), 30, 30, 30, 15);
140    // Prefer earlier compaction to latter if the benefit is not significant
141    compactEquals(sfCreate(99,99,99,99, 30,26,26,29,25,25), 30, 26, 26);
142    // Prefer later compaction if the benefit is significant.
143    compactEquals(sfCreate(99,99,99,99, 27,27,27,20,20,20), 20, 20, 20);
144  }

146  @Test
147  public void testCompactionEmptyHFile() throws IOException {
148    // Set TTL
149    ScanInfo oldScanInfo = store.getScanInfo();
150    ScanInfo newScanInfo = new ScanInfo(oldScanInfo.getConfiguration(), oldScanInfo.getFamily(),
151        oldScanInfo.getMinVersions(), oldScanInfo.getMaxVersions(), 600,
152        oldScanInfo.getKeepDeletedCells(), oldScanInfo.getTimeToPurgeDeletes(),
153        oldScanInfo.getComparator());
154    store.setScanInfo(newScanInfo);
155    // Do not compact empty store file
156    List<StoreFile> candidates = sfCreate(0);
157    for (StoreFile file : candidates) {
158      if (file instanceof MockStoreFile) {
159        MockStoreFile mockFile = (MockStoreFile) file;
160        mockFile.setTimeRangeTracker(new TimeRangeTracker(-1, -1));
161        mockFile.setEntries(0);
162      }
163    }
164    // Test Default compactions
165    CompactionRequest result = ((RatioBasedCompactionPolicy) store.storeEngine
166        .getCompactionPolicy()).selectCompaction(candidates,
167        new ArrayList<StoreFile>(), false, false, false);
168    Assert.assertTrue(result.getFiles().size() == 0);
169    store.setScanInfo(oldScanInfo);
170  }
