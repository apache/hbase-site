083  public static class PingHandler extends PingProtos.PingService implements RegionCoprocessor {
084    private int counter = 0;
085
086    @Override
087    public void start(CoprocessorEnvironment env) throws IOException {
088      if (env instanceof RegionCoprocessorEnvironment) {
089        return;
090      }
091      throw new CoprocessorException("Must be loaded on a table region!");
092    }
093
094    @Override
095    public void stop(CoprocessorEnvironment env) throws IOException {
096      // Nothing to do.
097    }
098
099    @Override
100    public void ping(RpcController controller, PingRequest request,
101        RpcCallback<PingResponse> done) {
102      this.counter++;
103      done.run(PingResponse.newBuilder().setPong("pong").build());
104    }
105
106    @Override
107    public void count(RpcController controller, CountRequest request,
108        RpcCallback<CountResponse> done) {
109      done.run(CountResponse.newBuilder().setCount(this.counter).build());
110    }
111
112    @Override
113    public void increment(RpcController controller,
114        IncrementCountRequest request, RpcCallback<IncrementCountResponse> done) {
115      this.counter += request.getDiff();
116      done.run(IncrementCountResponse.newBuilder().setCount(this.counter).build());
117    }
118
119    @Override
120    public void hello(RpcController controller, HelloRequest request,
121        RpcCallback<HelloResponse> done) {
122      if (!request.hasName()) {
123        done.run(HelloResponse.newBuilder().setResponse(WHOAREYOU).build());
124      } else if (request.getName().equals(NOBODY)) {
125        done.run(HelloResponse.newBuilder().build());
126      } else {
127        done.run(HelloResponse.newBuilder().setResponse(HELLO + request.getName()).build());
128      }
129    }
130
131    @Override
132    public void noop(RpcController controller, NoopRequest request,
133        RpcCallback<NoopResponse> done) {
134      done.run(NoopResponse.newBuilder().build());
135    }
136
137    @Override
138    public Iterable<Service> getServices() {
139      return Collections.singleton(this);
140    }
141  }

190  @Test
191  public void testSingleProxy() throws Throwable {
192    Table table = util.getConnection().getTable(TEST_TABLE);
193    Map<byte [], String> results = ping(table, null, null);
194    // There are three regions so should get back three results.
195    assertEquals(3, results.size());
196    for (Map.Entry<byte [], String> e: results.entrySet()) {
197      assertEquals("Invalid custom protocol response", "pong", e.getValue());
198    }
199    hello(table, "George", HELLO + "George");
200    LOG.info("Did george");
201    hello(table, null, "Who are you?");
202    LOG.info("Who are you");
203    hello(table, NOBODY, null);
204    LOG.info(NOBODY);
205    Map<byte [], Integer> intResults = table.coprocessorService(PingProtos.PingService.class,
206      null, null,
207      new Batch.Call<PingProtos.PingService, Integer>() {
208        @Override
209        public Integer call(PingProtos.PingService instance) throws IOException {
210          CoprocessorRpcUtils.BlockingRpcCallback<PingProtos.CountResponse> rpcCallback =
211            new CoprocessorRpcUtils.BlockingRpcCallback<>();
212          instance.count(null, PingProtos.CountRequest.newBuilder().build(), rpcCallback);
213          return rpcCallback.get().getCount();
214        }
215      });
216    int count = -1;
217    for (Map.Entry<byte [], Integer> e: intResults.entrySet()) {
218      assertTrue(e.getValue() > 0);
219      count = e.getValue();
220    }
221    final int diff = 5;
222    intResults = table.coprocessorService(PingProtos.PingService.class,
223      null, null,
224      new Batch.Call<PingProtos.PingService, Integer>() {
225        @Override
226        public Integer call(PingProtos.PingService instance) throws IOException {
227          CoprocessorRpcUtils.BlockingRpcCallback<PingProtos.IncrementCountResponse> rpcCallback =
228            new CoprocessorRpcUtils.BlockingRpcCallback<>();
229          instance.increment(null,
230              PingProtos.IncrementCountRequest.newBuilder().setDiff(diff).build(),
231            rpcCallback);
232          return rpcCallback.get().getCount();
233        }
234      });
235    // There are three regions so should get back three results.
236    assertEquals(3, results.size());
237    for (Map.Entry<byte [], Integer> e: intResults.entrySet()) {
238      assertEquals(e.getValue().intValue(), count + diff);
239    }
240    table.close();
241  }

257  private Map<byte [], String> hello(final Table table, final String send, final byte [] start,
258          final byte [] end) throws ServiceException, Throwable {
259    return table.coprocessorService(PingProtos.PingService.class,
260        start, end,
261        new Batch.Call<PingProtos.PingService, String>() {
262          @Override
263          public String call(PingProtos.PingService instance) throws IOException {
264            CoprocessorRpcUtils.BlockingRpcCallback<PingProtos.HelloResponse> rpcCallback =
265              new CoprocessorRpcUtils.BlockingRpcCallback<>();
266            PingProtos.HelloRequest.Builder builder = PingProtos.HelloRequest.newBuilder();
267            if (send != null) {
268              builder.setName(send);
269            }
270            instance.hello(null, builder.build(), rpcCallback);
271            PingProtos.HelloResponse r = rpcCallback.get();
272            return r != null && r.hasResponse()? r.getResponse(): null;
273          }
274        });
275  }

278          final byte [] end) throws ServiceException, Throwable {
279    return table.coprocessorService(PingProtos.PingService.class,

285              new CoprocessorRpcUtils.BlockingRpcCallback<>();
286            PingProtos.HelloRequest.Builder builder = PingProtos.HelloRequest.newBuilder();

291            return r != null && r.hasResponse()? r.getResponse(): null;
292          }

314  public void testSingleMethod() throws Throwable {
315    try (Table table = util.getConnection().getTable(TEST_TABLE);
316        RegionLocator locator = util.getConnection().getRegionLocator(TEST_TABLE)) {
317      Map<byte [], String> results = table.coprocessorService(PingProtos.PingService.class,
318        null, ROW_A,
319        new Batch.Call<PingProtos.PingService, String>() {
320          @Override
321          public String call(PingProtos.PingService instance) throws IOException {
322            CoprocessorRpcUtils.BlockingRpcCallback<PingProtos.PingResponse> rpcCallback =
323              new CoprocessorRpcUtils.BlockingRpcCallback<>();
324            instance.ping(null, PingProtos.PingRequest.newBuilder().build(), rpcCallback);
325            return rpcCallback.get().getPong();
326          }
327        });
328      // Should have gotten results for 1 of the three regions only since we specified
329      // rows from 1 region
330      assertEquals(1, results.size());
331      verifyRegionResults(locator, results, ROW_A);
332
333      final String name = "NAME";
334      results = hello(table, name, null, ROW_A);
335      // Should have gotten results for 1 of the three regions only since we specified
336      // rows from 1 region
337      assertEquals(1, results.size());
338      verifyRegionResults(locator, results, "Hello, NAME", ROW_A);
339    }
340  }

342  @Test
343  public void testRowRange() throws Throwable {
344    try (Table table = util.getConnection().getTable(TEST_TABLE);
345        RegionLocator locator = util.getConnection().getRegionLocator(TEST_TABLE)) {
346      for (HRegionLocation e: locator.getAllRegionLocations()) {
347        LOG.info("Region " + e.getRegion().getRegionNameAsString()
348            + ", servername=" + e.getServerName());
349      }
350      // Here are what regions looked like on a run:
351      //
352      // test,,1355943549657.c65d4822d8bdecc033a96451f3a0f55d.
353      // test,bbb,1355943549661.110393b070dd1ed93441e0bc9b3ffb7e.
354      // test,ccc,1355943549665.c3d6d125141359cbbd2a43eaff3cdf74.
355
356      Map<byte [], String> results = ping(table, null, ROW_A);
357      // Should contain first region only.
358      assertEquals(1, results.size());
359      verifyRegionResults(locator, results, ROW_A);
360
361      // Test start row + empty end
362      results = ping(table, ROW_BC, null);
363      assertEquals(2, results.size());
364      // should contain last 2 regions
365      HRegionLocation loc = locator.getRegionLocation(ROW_A, true);
366      assertNull("Should be missing region for row aaa (prior to start row)",
367        results.get(loc.getRegion().getRegionName()));
368      verifyRegionResults(locator, results, ROW_B);
369      verifyRegionResults(locator, results, ROW_C);
370
371      // test empty start + end
372      results = ping(table, null, ROW_BC);
373      // should contain the first 2 regions
374      assertEquals(2, results.size());
375      verifyRegionResults(locator, results, ROW_A);
376      verifyRegionResults(locator, results, ROW_B);
377      loc = locator.getRegionLocation(ROW_C, true);
378      assertNull("Should be missing region for row ccc (past stop row)",
379          results.get(loc.getRegion().getRegionName()));
380
381      // test explicit start + end
382      results = ping(table, ROW_AB, ROW_BC);
383      // should contain first 2 regions
384      assertEquals(2, results.size());
385      verifyRegionResults(locator, results, ROW_A);
386      verifyRegionResults(locator, results, ROW_B);
387      loc = locator.getRegionLocation(ROW_C, true);
388      assertNull("Should be missing region for row ccc (past stop row)",
389          results.get(loc.getRegion().getRegionName()));
390
391      // test single region
392      results = ping(table, ROW_B, ROW_BC);
393      // should only contain region bbb
394      assertEquals(1, results.size());
395      verifyRegionResults(locator, results, ROW_B);
396      loc = locator.getRegionLocation(ROW_A, true);
397      assertNull("Should be missing region for row aaa (prior to start)",
398          results.get(loc.getRegion().getRegionName()));
399      loc = locator.getRegionLocation(ROW_C, true);
400      assertNull("Should be missing region for row ccc (past stop row)",
401          results.get(loc.getRegion().getRegionName()));
402    }
403  }

468  private void verifyRegionResults(RegionLocator table, Map<byte[],String> results, byte[] row)
469          throws Exception {
470    verifyRegionResults(table, results, "pong", row);
471  }

473  private void verifyRegionResults(RegionLocator regionLocator, Map<byte[], String> results,
474          String expected, byte[] row) throws Exception {
475    for (Map.Entry<byte [], String> e: results.entrySet()) {
476      LOG.info("row=" + Bytes.toString(row) + ", expected=" + expected +
477        ", result key=" + Bytes.toString(e.getKey()) +
478        ", value=" + e.getValue());
479    }
480    HRegionLocation loc = regionLocator.getRegionLocation(row, true);
481    byte[] region = loc.getRegion().getRegionName();
482    assertTrue("Results should contain region " +
483      Bytes.toStringBinary(region) + " for row '" + Bytes.toStringBinary(row)+ "'",
484      results.containsKey(region));
485    assertEquals("Invalid result for row '"+Bytes.toStringBinary(row)+"'",
486      expected, results.get(region));
487  }
