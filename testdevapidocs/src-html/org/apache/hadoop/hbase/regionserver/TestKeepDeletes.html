092  /**
093   * Make sure that deleted rows are retained.
094   * Family delete markers are deleted.
095   * Column Delete markers are versioned
096   * Time range scan of deleted rows are possible
097   */
098  @Test
099  public void testBasicScenario() throws Exception {
100    // keep 3 versions, rows do not expire
101    HTableDescriptor htd = hbu.createTableDescriptor(name.getMethodName(), 0, 3,
102        HConstants.FOREVER, KeepDeletedCells.TRUE);
103    HRegion region = hbu.createLocalHRegion(htd, null, null);
104
105    long ts = EnvironmentEdgeManager.currentTime();
106    Put p = new Put(T1, ts);
107    p.addColumn(c0, c0, T1);
108    region.put(p);
109    p = new Put(T1, ts+1);
110    p.addColumn(c0, c0, T2);
111    region.put(p);
112    p = new Put(T1, ts+2);
113    p.addColumn(c0, c0, T3);
114    region.put(p);
115    p = new Put(T1, ts+4);
116    p.addColumn(c0, c0, T4);
117    region.put(p);
118
119    // now place a delete marker at ts+2
120    Delete d = new Delete(T1, ts+2);
121    region.delete(d);
122
123    // a raw scan can see the delete markers
124    // (one for each column family)
125    assertEquals(3, countDeleteMarkers(region));
126
127    // get something *before* the delete marker
128    Get g = new Get(T1);
129    g.setMaxVersions();
130    g.setTimeRange(0L, ts+2);
131    Result r = region.get(g);
132    checkResult(r, c0, c0, T2,T1);
133
134    // flush
135    region.flush(true);
136
137    // yep, T2 still there, T1 gone
138    r = region.get(g);
139    checkResult(r, c0, c0, T2);
140
141    // major compact
142    region.compact(true);
143    region.compact(true);
144
145    // one delete marker left (the others did not
146    // have older puts)
147    assertEquals(1, countDeleteMarkers(region));
148
149    // still there (even after multiple compactions)
150    r = region.get(g);
151    checkResult(r, c0, c0, T2);
152
153    // a timerange that includes the delete marker won't see past rows
154    g.setTimeRange(0L, ts+4);
155    r = region.get(g);
156    assertTrue(r.isEmpty());
157
158    // two more puts, this will expire the older puts.
159    p = new Put(T1, ts+5);
160    p.addColumn(c0, c0, T5);
161    region.put(p);
162    p = new Put(T1, ts+6);
163    p.addColumn(c0, c0, T6);
164    region.put(p);
165
166    // also add an old put again
167    // (which is past the max versions)
168    p = new Put(T1, ts);
169    p.addColumn(c0, c0, T1);
170    region.put(p);
171    r = region.get(g);
172    assertTrue(r.isEmpty());
173
174    region.flush(true);
175    region.compact(true);
176    region.compact(true);
177
178    // verify that the delete marker itself was collected
179    region.put(p);
180    r = region.get(g);
181    checkResult(r, c0, c0, T1);
182    assertEquals(0, countDeleteMarkers(region));
183
184    HBaseTestingUtility.closeRegionAndWAL(region);
185  }

315  /**
316   * Verify that "raw" scanning mode return delete markers and deletes rows.
317   */
318  @Test
319  public void testRawScan() throws Exception {
320    HTableDescriptor htd = hbu.createTableDescriptor(name.getMethodName(), 0, 3,
321        HConstants.FOREVER, KeepDeletedCells.TRUE);
322    Region region = hbu.createLocalHRegion(htd, null, null);
323
324    long ts = EnvironmentEdgeManager.currentTime();
325    Put p = new Put(T1, ts);
326    p.addColumn(c0, c0, T1);
327    region.put(p);
328    p = new Put(T1, ts+2);
329    p.addColumn(c0, c0, T2);
330    region.put(p);
331    p = new Put(T1, ts+4);
332    p.addColumn(c0, c0, T3);
333    region.put(p);
334
335    Delete d = new Delete(T1, ts+1);
336    region.delete(d);
337
338    d = new Delete(T1, ts+2);
339    d.addColumn(c0, c0, ts+2);
340    region.delete(d);
341
342    d = new Delete(T1, ts+3);
343    d.addColumns(c0, c0, ts+3);
344    region.delete(d);
345
346    Scan s = new Scan();
347    s.setRaw(true);
348    s.setMaxVersions();
349    InternalScanner scan = region.getScanner(s);
350    List<Cell> kvs = new ArrayList<>();
351    scan.next(kvs);
352    assertEquals(8, kvs.size());
353    assertTrue(PrivateCellUtil.isDeleteFamily(kvs.get(0)));
354    assertArrayEquals(CellUtil.cloneValue(kvs.get(1)), T3);
355    assertTrue(CellUtil.isDelete(kvs.get(2)));
356    assertTrue(CellUtil.isDelete(kvs.get(3))); // .isDeleteType());
357    assertArrayEquals(CellUtil.cloneValue(kvs.get(4)), T2);
358    assertArrayEquals(CellUtil.cloneValue(kvs.get(5)), T1);
359    // we have 3 CFs, so there are two more delete markers
360    assertTrue(PrivateCellUtil.isDeleteFamily(kvs.get(6)));
361    assertTrue(PrivateCellUtil.isDeleteFamily(kvs.get(7)));
362
363    // verify that raw scans honor the passed timerange
364    s = new Scan();
365    s.setRaw(true);
366    s.setMaxVersions();
367    s.setTimeRange(0, 1);
368    scan = region.getScanner(s);
369    kvs = new ArrayList<>();
370    scan.next(kvs);
371    // nothing in this interval, not even delete markers
372    assertTrue(kvs.isEmpty());
373
374    // filter new delete markers
375    s = new Scan();
376    s.setRaw(true);
377    s.setMaxVersions();
378    s.setTimeRange(0, ts+2);
379    scan = region.getScanner(s);
380    kvs = new ArrayList<>();
381    scan.next(kvs);
382    assertEquals(4, kvs.size());
383    assertTrue(PrivateCellUtil.isDeleteFamily(kvs.get(0)));
384    assertArrayEquals(CellUtil.cloneValue(kvs.get(1)), T1);
385    // we have 3 CFs
386    assertTrue(PrivateCellUtil.isDeleteFamily(kvs.get(2)));
387    assertTrue(PrivateCellUtil.isDeleteFamily(kvs.get(3)));
388
389    // filter old delete markers
390    s = new Scan();
391    s.setRaw(true);
392    s.setMaxVersions();
393    s.setTimeRange(ts+3, ts+5);
394    scan = region.getScanner(s);
395    kvs = new ArrayList<>();
396    scan.next(kvs);
397    assertEquals(2, kvs.size());
398    assertArrayEquals(CellUtil.cloneValue(kvs.get(0)), T3);
399    assertTrue(CellUtil.isDelete(kvs.get(1)));
400
401
402    HBaseTestingUtility.closeRegionAndWAL(region);
403  }

449   * Test delete marker removal from store files.

451  @Test
452  public void testDeleteMarkerExpiration() throws Exception {
453    HTableDescriptor htd = hbu.createTableDescriptor(name.getMethodName(), 0, 1,
454        HConstants.FOREVER, KeepDeletedCells.TRUE);
455    HRegion region = hbu.createLocalHRegion(htd, null, null);
456

458
459    Put p = new Put(T1, ts);
460    p.addColumn(c0, c0, T1);
461    region.put(p);
462
463    // a put into another store (CF) should have no effect
464    p = new Put(T1, ts-10);
465    p.addColumn(c1, c0, T1);
466    region.put(p);
467
468    // all the following deletes affect the put
469    Delete d = new Delete(T1, ts);
470    d.addColumns(c0, c0, ts);
471    region.delete(d);
472
473    d = new Delete(T1, ts);
474    d.addFamily(c0, ts);
475    region.delete(d);
476
477    d = new Delete(T1, ts);
478    d.addColumn(c0, c0, ts+1);
479    region.delete(d);
480
481    d = new Delete(T1, ts);
482    d.addColumn(c0, c0, ts+2);
483    region.delete(d);
484
485    // 1 family marker, 1 column marker, 2 version markers
486    assertEquals(4, countDeleteMarkers(region));
487
488    region.flush(true);
489    assertEquals(4, countDeleteMarkers(region));
490    region.compact(false);
491    assertEquals(4, countDeleteMarkers(region));
492
493    // another put will push out the earlier put...
494    p = new Put(T1, ts+3);
495    p.addColumn(c0, c0, T1);
496    region.put(p);
497
498    region.flush(true);
499    // no markers are collected, since there is an affected put
500    region.compact(true);
501    assertEquals(4, countDeleteMarkers(region));
502
503    // the last collections collected the earlier put
504    // so after this collection all markers
505    region.compact(true);
506    assertEquals(0, countDeleteMarkers(region));
507
508    HBaseTestingUtility.closeRegionAndWAL(region);
509  }

669  /**
670   * Verify that column/version delete makers are sorted
671   * with their respective puts and removed correctly by
672   * versioning (i.e. not relying on the store earliestPutTS).
673   */
674  @Test
675  public void testDeleteMarkerVersioning() throws Exception {
676    HTableDescriptor htd = hbu.createTableDescriptor(name.getMethodName(), 0, 1,
677        HConstants.FOREVER, KeepDeletedCells.TRUE);
678    HRegion region = hbu.createLocalHRegion(htd, null, null);
679
680    long ts = EnvironmentEdgeManager.currentTime();
681    Put p = new Put(T1, ts);
682    p.addColumn(c0, c0, T1);
683    region.put(p);
684
685    // this prevents marker collection based on earliestPut
686    // (cannot keep earliest put per column in the store file)
687    p = new Put(T1, ts-10);
688    p.addColumn(c0, c1, T1);
689    region.put(p);
690
691    Delete d = new Delete(T1, ts);
692    // test corner case (Put and Delete have same TS)
693    d.addColumns(c0, c0, ts);
694    region.delete(d);
695
696    d = new Delete(T1, ts+1);
697    d.addColumn(c0, c0, ts+1);
698    region.delete(d);
699
700    d = new Delete(T1, ts+3);
701    d.addColumn(c0, c0, ts+3);
702    region.delete(d);
703
704    region.flush(true);
705    region.compact(true);
706    region.compact(true);
707    assertEquals(3, countDeleteMarkers(region));
708
709    // add two more puts, since max version is 1
710    // the 2nd put (and all delete markers following)
711    // will be removed.
712    p = new Put(T1, ts+2);
713    p.addColumn(c0, c0, T2);
714    region.put(p);
715
716    // delete, put, delete, delete, put
717    assertEquals(3, countDeleteMarkers(region));
718
719    p = new Put(T1, ts+3);
720    p.addColumn(c0, c0, T3);
721    region.put(p);
722
723    // This is potentially questionable behavior.
724    // This could be changed by not letting the ScanQueryMatcher
725    // return SEEK_NEXT_COL if a put is past VERSIONS, but instead
726    // return SKIP if the store has KEEP_DELETED_CELLS set.
727    //
728    // As it stands, the 1 here is correct here.
729    // There are two puts, VERSIONS is one, so after the 1st put the scanner
730    // knows that there can be no more KVs (put or delete) that have any effect.
731    //
732    // delete, put, put | delete, delete
733    assertEquals(1, countDeleteMarkers(region));
734
735    // flush cache only sees what is in the memstore
736    region.flush(true);
737
738    // Here we have the three markers again, because the flush above
739    // removed the 2nd put before the file is written.
740    // So there's only one put, and hence the deletes already in the store
741    // files cannot be removed safely.
742    // delete, put, delete, delete
743    assertEquals(3, countDeleteMarkers(region));
744
745    region.compact(true);
746    assertEquals(3, countDeleteMarkers(region));
747
748    // add one more put
749    p = new Put(T1, ts+4);
750    p.addColumn(c0, c0, T4);
751    region.put(p);
752
753    region.flush(true);
754    // one trailing delete marker remains (but only one)
755    // because delete markers do not increase the version count
756    assertEquals(1, countDeleteMarkers(region));
757    region.compact(true);
758    region.compact(true);
759    assertEquals(1, countDeleteMarkers(region));
760
761    HBaseTestingUtility.closeRegionAndWAL(region);
762  }

839    // all puts now are just retained because of min versions = 3
840

842    Delete d = new Delete(T1, ts-1);
843    region.delete(d);
844    // and a column delete marker
845    d = new Delete(T1, ts-2);
846    d.addColumns(c0, c0, ts-1);

939      long time, byte[]... vals) throws IOException {

949  private int countDeleteMarkers(HRegion region) throws IOException {
950    Scan s = new Scan();
951    s.setRaw(true);
952    // use max versions from the store(s)
953    s.setMaxVersions(region.getStores().iterator().next().getScanInfo().getMaxVersions());
954    InternalScanner scan = region.getScanner(s);
955    List<Cell> kvs = new ArrayList<>();
956    int res = 0;
957    boolean hasMore;
958    do {
959      hasMore = scan.next(kvs);
960      for (Cell kv : kvs) {
961        if(CellUtil.isDelete(kv)) res++;
962      }
963      kvs.clear();
964    } while (hasMore);
965    scan.close();
966    return res;
967  }

969  private void checkResult(Result r, byte[] fam, byte[] col, byte[] ... vals) {
970    assertEquals(r.size(), vals.length);
971    List<Cell> kvs = r.getColumnCells(fam, col);
972    assertEquals(kvs.size(), vals.length);
973    for (int i=0;i<vals.length;i++) {
974      assertArrayEquals(CellUtil.cloneValue(kvs.get(i)), vals[i]);
975    }
