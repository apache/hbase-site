073/**
074 * Here we test to make sure that scans return the expected Results when the server is sending the
075 * Client heartbeat messages. Heartbeat messages are essentially keep-alive messages (they prevent
076 * the scanner on the client side from timing out). A heartbeat message is sent from the server to
077 * the client when the server has exceeded the time limit during the processing of the scan. When
078 * the time limit is reached, the server will return to the Client whatever Results it has
079 * accumulated (potentially empty).
080 */
081@Category(MediumTests.class)
082public class TestScannerHeartbeatMessages {
083

093  private static int NUM_ROWS = 5;
094  private static byte[] ROW = Bytes.toBytes("testRow");
095  private static byte[][] ROWS = HTestConst.makeNAscii(ROW, NUM_ROWS);

098  private static byte[] FAMILY = Bytes.toBytes("testFamily");
099  private static byte[][] FAMILIES = HTestConst.makeNAscii(FAMILY, NUM_FAMILIES);

108  // The time limit should be based on the rpc timeout at client, or the client will regards
109  // the request as timeout before server return a heartbeat.
110  private static int SERVER_TIMEOUT = 60000;
111
112  // Time, in milliseconds, that the client will wait for a response from the server before timing
113  // out. This value is used server side to determine when it is necessary to send a heartbeat
114  // message to the client. Time limit will be 500 ms.
115  private static int CLIENT_TIMEOUT = 1000;
116
117  // In this test, we sleep after reading each row. So we should make sure after we get some number
118  // of rows and sleep same times we must reach time limit, and do not timeout after next sleeping.
119  private static int DEFAULT_ROW_SLEEP_TIME = 300;
120
121  // Similar with row sleep time.
122  private static int DEFAULT_CF_SLEEP_TIME = 300;

143  static Table createTestTable(TableName name, byte[][] rows, byte[][] families,
144      byte[][] qualifiers, byte[] cellValue) throws IOException {
145    Table ht = TEST_UTIL.createTable(name, families);
146    List<Put> puts = createPuts(rows, families, qualifiers, cellValue);
147    ht.put(puts);
148    ht.getConfiguration().setInt(HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD, CLIENT_TIMEOUT);
149    return ht;

152  /**
153   * Make puts to put the input value into each combination of row, family, and qualifier
154   * @param rows
155   * @param families
156   * @param qualifiers
157   * @param value
158   * @return
159   * @throws IOException
160   */
161  static ArrayList<Put> createPuts(byte[][] rows, byte[][] families, byte[][] qualifiers,
162      byte[] value) throws IOException {
163    Put put;
164    ArrayList<Put> puts = new ArrayList<>();
165
166    for (int row = 0; row < rows.length; row++) {
167      put = new Put(rows[row]);
168      for (int fam = 0; fam < families.length; fam++) {
169        for (int qual = 0; qual < qualifiers.length; qual++) {
170          KeyValue kv = new KeyValue(rows[row], families[fam], qualifiers[qual], qual, value);
171          put.add(kv);
172        }
173      }
174      puts.add(put);
175    }
176
177    return puts;
178  }

195  /**
196   * Run the test callable when heartbeats are enabled/disabled. We expect all tests to only pass
197   * when heartbeat messages are enabled (otherwise the test is pointless). When heartbeats are
198   * disabled, the test should throw an exception.
199   * @param testCallable
200   * @throws InterruptedException
201   */
202  private void testImportanceOfHeartbeats(Callable<Void> testCallable) throws InterruptedException {
203    HeartbeatRPCServices.heartbeatsEnabled = true;
204
205    try {
206      testCallable.call();
207    } catch (Exception e) {
208      fail("Heartbeat messages are enabled, exceptions should NOT be thrown. Exception trace:"
209          + ExceptionUtils.getStackTrace(e));
210    }
211
212    HeartbeatRPCServices.heartbeatsEnabled = false;
213    try {
214      testCallable.call();
215    } catch (Exception e) {
216      return;
217    } finally {
218      HeartbeatRPCServices.heartbeatsEnabled = true;
219    }
220    fail("Heartbeats messages are disabled, an exception should be thrown. If an exception "
221        + " is not thrown, the test case is not testing the importance of heartbeat messages");
222  }

261
262        // Copy the scan before each test. When a scan object is used by a scanner, some of its
263        // fields may be changed such as start row
264        Scan scanCopy = new Scan(baseScan);
265        testEquivalenceOfScanWithHeartbeats(scanCopy, -1, DEFAULT_CF_SLEEP_TIME, false);
266        scanCopy = new Scan(baseScan);
267        testEquivalenceOfScanWithHeartbeats(scanCopy, -1, DEFAULT_CF_SLEEP_TIME, true);
268        return null;
269      }
270    });
271  }

273  public static class SparseFilter extends FilterBase {
274
275    @Override
276    public ReturnCode filterCell(final Cell v) throws IOException {
277      try {
278        Thread.sleep(CLIENT_TIMEOUT / 2 + 100);
279      } catch (InterruptedException e) {
280        Thread.currentThread().interrupt();
281      }
282      return Bytes.equals(CellUtil.cloneRow(v), ROWS[NUM_ROWS - 1]) ? ReturnCode.INCLUDE
283          : ReturnCode.SKIP;
284    }
285
286    public static Filter parseFrom(final byte[] pbBytes) {
287      return new SparseFilter();
288    }
289  }

291  /**
292   * Test the case that there is a filter which filters most of cells
293   * @throws Exception
294   */
295  @Test
296  public void testHeartbeatWithSparseFilter() throws Exception {
297    testImportanceOfHeartbeats(new Callable<Void>() {
298      @Override
299      public Void call() throws Exception {
300        Scan scan = new Scan();
301        scan.setMaxResultSize(Long.MAX_VALUE);
302        scan.setCaching(Integer.MAX_VALUE);
303        scan.setFilter(new SparseFilter());
304        ResultScanner scanner = TABLE.getScanner(scan);
305        int num = 0;
306        while (scanner.next() != null) {
307          num++;
308        }
309        assertEquals(1, num);
310        scanner.close();
311
312        scan = new Scan();
313        scan.setMaxResultSize(Long.MAX_VALUE);
314        scan.setCaching(Integer.MAX_VALUE);
315        scan.setFilter(new SparseFilter());
316        scan.setAllowPartialResults(true);
317        scanner = TABLE.getScanner(scan);
318        num = 0;
319        while (scanner.next() != null) {
320          num++;
321        }
322        assertEquals(NUM_FAMILIES * NUM_QUALIFIERS, num);
323        scanner.close();
324
325        return null;
326      }
327    });
328  }

330  /**
331   * Test the equivalence of a scan versus the same scan executed when heartbeat messages are
332   * necessary
333   * @param scan The scan configuration being tested
334   * @param rowSleepTime The time to sleep between fetches of row cells
335   * @param cfSleepTime The time to sleep between fetches of column family cells
336   * @param sleepBeforeCf set to true when column family sleeps should occur before the cells for
337   *          that column family are fetched
338   * @throws Exception
339   */
340  private void testEquivalenceOfScanWithHeartbeats(final Scan scan, int rowSleepTime,
341      int cfSleepTime, boolean sleepBeforeCf) throws Exception {
342    disableSleeping();
343    final ResultScanner scanner = TABLE.getScanner(scan);
344    final ResultScanner scannerWithHeartbeats = TABLE.getScanner(scan);
345
346    Result r1 = null;
347    Result r2 = null;
348
349    while ((r1 = scanner.next()) != null) {
350      // Enforce the specified sleep conditions during calls to the heartbeat scanner
351      configureSleepTime(rowSleepTime, cfSleepTime, sleepBeforeCf);
352      r2 = scannerWithHeartbeats.next();
353      disableSleeping();
354
355      assertTrue(r2 != null);
356      try {
357        Result.compareResults(r1, r2);
358      } catch (Exception e) {
359        fail(e.getMessage());
360      }
361    }
362
363    assertTrue(scannerWithHeartbeats.next() == null);
364    scanner.close();
365    scannerWithHeartbeats.close();
366  }

368  /**
369   * Helper method for setting the time to sleep between rows and column families. If a sleep time
370   * is negative then that sleep will be disabled
371   * @param rowSleepTime
372   * @param cfSleepTime
373   */
374  private static void configureSleepTime(int rowSleepTime, int cfSleepTime, boolean sleepBeforeCf) {
375    HeartbeatHRegion.sleepBetweenRows = rowSleepTime > 0;
376    HeartbeatHRegion.rowSleepTime = rowSleepTime;
377
378    HeartbeatHRegion.sleepBetweenColumnFamilies = cfSleepTime > 0;
379    HeartbeatHRegion.columnFamilySleepTime = cfSleepTime;
380    HeartbeatHRegion.sleepBeforeColumnFamily = sleepBeforeCf;
381  }
382
383  /**
384   * Disable the sleeping mechanism server side.
385   */
386  private static void disableSleeping() {
387    HeartbeatHRegion.sleepBetweenRows = false;
388    HeartbeatHRegion.sleepBetweenColumnFamilies = false;
389  }

425  /**
426   * Custom HRegion class that instantiates {@link RegionScanner}s with configurable sleep times
427   * between fetches of row Results and/or column family cells. Useful for emulating an instance
428   * where the server is taking a long time to process a client's scan request
429   */
430  private static class HeartbeatHRegion extends HRegion {
431    // Row sleeps occur AFTER each row worth of cells is retrieved.
432    private static volatile int rowSleepTime = DEFAULT_ROW_SLEEP_TIME;
433    private static volatile boolean sleepBetweenRows = false;
434
435    // The sleep for column families can be initiated before or after we fetch the cells for the
436    // column family. If the sleep occurs BEFORE then the time limits will be reached inside
437    // StoreScanner while we are fetching individual cells. If the sleep occurs AFTER then the time
438    // limit will be reached inside RegionScanner after all the cells for a column family have been
439    // retrieved.
440    private static volatile boolean sleepBeforeColumnFamily = false;
441    private static volatile int columnFamilySleepTime = DEFAULT_CF_SLEEP_TIME;
442    private static volatile boolean sleepBetweenColumnFamilies = false;
443
444    public HeartbeatHRegion(Path tableDir, WAL wal, FileSystem fs, Configuration confParam,
445        RegionInfo regionInfo, TableDescriptor htd, RegionServerServices rsServices) {
446      super(tableDir, wal, fs, confParam, regionInfo, htd, rsServices);
447    }
448
449    public HeartbeatHRegion(HRegionFileSystem fs, WAL wal, Configuration confParam,
450        TableDescriptor htd, RegionServerServices rsServices) {
451      super(fs, wal, confParam, htd, rsServices);
452    }
453
454    private static void columnFamilySleep() {
455      if (sleepBetweenColumnFamilies) {
456        Threads.sleepWithoutInterrupt(columnFamilySleepTime);
457      }
458    }
459
460    private static void rowSleep() {
461      if (sleepBetweenRows) {
462        Threads.sleepWithoutInterrupt(rowSleepTime);
463      }
464    }
465
466    // Instantiate the custom heartbeat region scanners
467    @Override
468    protected RegionScanner instantiateRegionScanner(Scan scan,
469        List<KeyValueScanner> additionalScanners, long nonceGroup, long nonce) throws IOException {
470      if (scan.isReversed()) {
471        if (scan.getFilter() != null) {
472          scan.getFilter().setReversed(true);
473        }
474        return new HeartbeatReversedRegionScanner(scan, additionalScanners, this);
475      }
476      return new HeartbeatRegionScanner(scan, additionalScanners, this);
477    }
478  }

480  /**
481   * Custom ReversedRegionScanner that can be configured to sleep between retrievals of row Results
482   * and/or column family cells
483   */
484  private static class HeartbeatReversedRegionScanner extends ReversedRegionScannerImpl {
485    HeartbeatReversedRegionScanner(Scan scan, List<KeyValueScanner> additionalScanners,
486        HRegion region) throws IOException {
487      super(scan, additionalScanners, region);
488    }
489
490    @Override
491    public boolean nextRaw(List<Cell> outResults, ScannerContext context) throws IOException {
492      boolean moreRows = super.nextRaw(outResults, context);
493      HeartbeatHRegion.rowSleep();
494      return moreRows;
495    }
496
497    @Override
498    protected void initializeKVHeap(List<KeyValueScanner> scanners,
499        List<KeyValueScanner> joinedScanners, HRegion region) throws IOException {
500      this.storeHeap =
501          new HeartbeatReversedKVHeap(scanners, (CellComparatorImpl) region.getCellComparator());
502      if (!joinedScanners.isEmpty()) {
503        this.joinedHeap = new HeartbeatReversedKVHeap(joinedScanners,
504            (CellComparatorImpl) region.getCellComparator());
505      }
506    }
507  }

538  /**
539   * Custom KV Heap that can be configured to sleep/wait in between retrievals of column family
540   * cells. Useful for testing
541   */
542  private static final class HeartbeatKVHeap extends KeyValueHeap {
543    public HeartbeatKVHeap(List<? extends KeyValueScanner> scanners, CellComparator comparator)
544        throws IOException {
545      super(scanners, comparator);
546    }
547
548    HeartbeatKVHeap(List<? extends KeyValueScanner> scanners, KVScannerComparator comparator)
549        throws IOException {
550      super(scanners, comparator);
551    }
552
553    @Override
554    public boolean next(List<Cell> result, ScannerContext context) throws IOException {
555      if (HeartbeatHRegion.sleepBeforeColumnFamily) HeartbeatHRegion.columnFamilySleep();
556      boolean moreRows = super.next(result, context);
557      if (!HeartbeatHRegion.sleepBeforeColumnFamily) HeartbeatHRegion.columnFamilySleep();
558      return moreRows;
559    }
560  }
