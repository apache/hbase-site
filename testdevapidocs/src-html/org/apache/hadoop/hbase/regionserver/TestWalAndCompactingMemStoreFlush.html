075  private HRegion initHRegion(String callingMethod, Configuration conf) throws IOException {
076    int i = 0;
077    TableDescriptorBuilder.ModifyableTableDescriptor tableDescriptor =
078      new TableDescriptorBuilder.ModifyableTableDescriptor(TABLENAME);
079    for (byte[] family : FAMILIES) {
080      ColumnFamilyDescriptorBuilder.ModifyableColumnFamilyDescriptor familyDescriptor =
081        new ColumnFamilyDescriptorBuilder.ModifyableColumnFamilyDescriptor(family);
082      // even column families are going to have compacted memstore
083      if (i % 2 == 0) {
084        familyDescriptor.setInMemoryCompaction(MemoryCompactionPolicy
085            .valueOf(conf.get(CompactingMemStore.COMPACTING_MEMSTORE_TYPE_KEY)));
086      } else {
087        familyDescriptor.setInMemoryCompaction(MemoryCompactionPolicy.NONE);
088      }
089      tableDescriptor.setColumnFamily(familyDescriptor);
090      i++;
091    }
092
093    HRegionInfo info = new HRegionInfo(TABLENAME, null, null, false);
094    Path path = new Path(DIR, callingMethod);
095    HRegion region = HBaseTestingUtility.createRegionAndWAL(info, path, conf,
096      tableDescriptor, false);
097    region.regionServicesForStores = Mockito.spy(region.regionServicesForStores);
098    ThreadPoolExecutor pool = (ThreadPoolExecutor) Executors.newFixedThreadPool(1);
099    Mockito.when(region.regionServicesForStores.getInMemoryCompactionPool()).thenReturn(pool);
100    region.initialize(null);
101    return region;
102  }

104  // A helper function to create puts.
105  private Put createPut(int familyNum, int putNum) {
106    byte[] qf  = Bytes.toBytes("q" + familyNum);
107    byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
108    byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
109    Put p = new Put(row);
110    p.addColumn(FAMILIES[familyNum - 1], qf, val);
111    return p;
112  }
113
114  // A helper function to create double puts, so something can be compacted later.
115  private Put createDoublePut(int familyNum, int putNum) {
116    byte[] qf  = Bytes.toBytes("q" + familyNum);
117    byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
118    byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
119    Put p = new Put(row);
120    // add twice with different timestamps
121    p.addColumn(FAMILIES[familyNum - 1], qf, 10, val);
122    p.addColumn(FAMILIES[familyNum - 1], qf, 20, val);
123    return p;
124  }

140  @Test
141  public void testSelectiveFlushWithEager() throws IOException {
142    // Set up the configuration
143    conf.setLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 300 * 1024);
144    conf.setLong(FlushLargeStoresPolicy.HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN, 75 * 1024);
145    // set memstore to do data compaction
146    conf.set(CompactingMemStore.COMPACTING_MEMSTORE_TYPE_KEY,
147        String.valueOf(MemoryCompactionPolicy.EAGER));
148
149    // Intialize the region
150    HRegion region = initHRegion("testSelectiveFlushWithEager", conf);
151    verifyInMemoryFlushSize(region);
152    // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
153    for (int i = 1; i <= 1200; i++) {
154      region.put(createPut(1, i));        // compacted memstore, all the keys are unique
155
156      if (i <= 100) {
157        region.put(createPut(2, i));
158        if (i <= 50) {
159          // compacted memstore, subject for compaction due to duplications
160          region.put(createDoublePut(3, i));
161        }
162      }
163    }
164
165    // Now add more puts for CF2, so that we only flush CF2 (DefaultMemStore) to disk
166    for (int i = 100; i < 2000; i++) {
167      region.put(createPut(2, i));
168    }
169
170    long totalMemstoreSize = region.getMemStoreDataSize();
171
172    // Find the smallest LSNs for edits wrt to each CF.
173    long smallestSeqCF1PhaseI = region.getOldestSeqIdOfStore(FAMILY1);
174    long smallestSeqCF2PhaseI = region.getOldestSeqIdOfStore(FAMILY2);
175    long smallestSeqCF3PhaseI = region.getOldestSeqIdOfStore(FAMILY3);
176
177    // Find the sizes of the memstores of each CF.
178    MemStoreSize cf1MemstoreSizePhaseI = region.getStore(FAMILY1).getMemStoreSize();
179    MemStoreSize cf2MemstoreSizePhaseI = region.getStore(FAMILY2).getMemStoreSize();
180    MemStoreSize cf3MemstoreSizePhaseI = region.getStore(FAMILY3).getMemStoreSize();
181
182    // Get the overall smallest LSN in the region's memstores.
183    long smallestSeqInRegionCurrentMemstorePhaseI = getWAL(region)
184        .getEarliestMemStoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
185
186    String s = "\n\n----------------------------------\n"
187        + "Upon initial insert and before any flush, size of CF1 is:"
188        + cf1MemstoreSizePhaseI + ", is CF1 compacted memstore?:"
189        + region.getStore(FAMILY1).isSloppyMemStore() + ". Size of CF2 is:"
190        + cf2MemstoreSizePhaseI + ", is CF2 compacted memstore?:"
191        + region.getStore(FAMILY2).isSloppyMemStore() + ". Size of CF3 is:"
192        + cf3MemstoreSizePhaseI + ", is CF3 compacted memstore?:"
193        + region.getStore(FAMILY3).isSloppyMemStore() + "\n";
194
195    // The overall smallest LSN in the region's memstores should be the same as
196    // the LSN of the smallest edit in CF1
197    assertEquals(smallestSeqCF1PhaseI, smallestSeqInRegionCurrentMemstorePhaseI);
198
199    // Some other sanity checks.
200    assertTrue(smallestSeqCF1PhaseI < smallestSeqCF2PhaseI);
201    assertTrue(smallestSeqCF2PhaseI < smallestSeqCF3PhaseI);
202    assertTrue(cf1MemstoreSizePhaseI.getDataSize() > 0);
203    assertTrue(cf2MemstoreSizePhaseI.getDataSize() > 0);
204    assertTrue(cf3MemstoreSizePhaseI.getDataSize() > 0);
205
206    // The total memstore size should be the same as the sum of the sizes of
207    // memstores of CF1, CF2 and CF3.
208    String msg = "totalMemstoreSize="+totalMemstoreSize +
209        " cf1MemstoreSizePhaseI="+cf1MemstoreSizePhaseI +
210        " cf2MemstoreSizePhaseI="+cf2MemstoreSizePhaseI +
211        " cf3MemstoreSizePhaseI="+cf3MemstoreSizePhaseI ;
212    assertEquals(msg, totalMemstoreSize, cf1MemstoreSizePhaseI.getDataSize()
213        + cf2MemstoreSizePhaseI.getDataSize() + cf3MemstoreSizePhaseI.getDataSize());
214
215    // Flush!!!!!!!!!!!!!!!!!!!!!!
216    // We have big compacting memstore CF1 and two small memstores:
217    // CF2 (not compacted) and CF3 (compacting)
218    // All together they are above the flush size lower bound.
219    // Since CF1 and CF3 should be flushed to memory (not to disk),
220    // CF2 is going to be flushed to disk.
221    // CF1 - nothing to compact (but flattening), CF3 - should be twice compacted
222    CompactingMemStore cms1 = (CompactingMemStore) ((HStore) region.getStore(FAMILY1)).memstore;
223    CompactingMemStore cms3 = (CompactingMemStore) ((HStore) region.getStore(FAMILY3)).memstore;
224    cms1.flushInMemory();
225    cms3.flushInMemory();
226    region.flush(false);
227
228    // Recalculate everything
229    MemStoreSize cf1MemstoreSizePhaseII = region.getStore(FAMILY1).getMemStoreSize();
230    MemStoreSize cf2MemstoreSizePhaseII = region.getStore(FAMILY2).getMemStoreSize();
231    MemStoreSize cf3MemstoreSizePhaseII = region.getStore(FAMILY3).getMemStoreSize();
232
233    long smallestSeqInRegionCurrentMemstorePhaseII = getWAL(region)
234        .getEarliestMemStoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
235    // Find the smallest LSNs for edits wrt to each CF.
236    long smallestSeqCF1PhaseII = region.getOldestSeqIdOfStore(FAMILY1);
237    long smallestSeqCF2PhaseII = region.getOldestSeqIdOfStore(FAMILY2);
238    long smallestSeqCF3PhaseII = region.getOldestSeqIdOfStore(FAMILY3);
239
240    s = s + "\n----After first flush! CF1 should be flushed to memory, but not compacted.---\n"
241        + "Size of CF1 is:" + cf1MemstoreSizePhaseII + ", size of CF2 is:" + cf2MemstoreSizePhaseII
242        + ", size of CF3 is:" + cf3MemstoreSizePhaseII + "\n";
243
244    // CF1 was flushed to memory, but there is nothing to compact, and CF1 was flattened
245    assertTrue(cf1MemstoreSizePhaseII.getDataSize() == cf1MemstoreSizePhaseI.getDataSize());
246    assertTrue(cf1MemstoreSizePhaseII.getHeapSize() < cf1MemstoreSizePhaseI.getHeapSize());
247
248    // CF2 should become empty
249    assertEquals(0, cf2MemstoreSizePhaseII.getDataSize());
250    assertEquals(MutableSegment.DEEP_OVERHEAD, cf2MemstoreSizePhaseII.getHeapSize());
251
252    // verify that CF3 was flushed to memory and was compacted (this is approximation check)
253    assertTrue(cf3MemstoreSizePhaseI.getDataSize() > cf3MemstoreSizePhaseII.getDataSize());
254    assertTrue(
255        cf3MemstoreSizePhaseI.getHeapSize() / 2 > cf3MemstoreSizePhaseII.getHeapSize());
256
257    // Now the smallest LSN in the region should be the same as the smallest
258    // LSN in the memstore of CF1.
259    assertEquals(smallestSeqInRegionCurrentMemstorePhaseII, smallestSeqCF1PhaseI);
260
261    // Now add more puts for CF1, so that we also flush CF1 to disk instead of
262    // memory in next flush
263    for (int i = 1200; i < 3000; i++) {
264      region.put(createPut(1, i));
265    }
266
267    s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseII
268        + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseII + ", " +
269        "the smallest sequence in CF2:"
270        + smallestSeqCF2PhaseII +", the smallest sequence in CF3:" + smallestSeqCF3PhaseII + "\n";
271
272    // How much does the CF1 memstore occupy? Will be used later.
273    MemStoreSize cf1MemstoreSizePhaseIII = region.getStore(FAMILY1).getMemStoreSize();
274    long smallestSeqCF1PhaseIII = region.getOldestSeqIdOfStore(FAMILY1);
275
276    s = s + "----After more puts into CF1 its size is:" + cf1MemstoreSizePhaseIII
277        + ", and its sequence is:" + smallestSeqCF1PhaseIII + " ----\n" ;
278
279
280    // Flush!!!!!!!!!!!!!!!!!!!!!!
281    // Flush again, CF1 is flushed to disk
282    // CF2 is flushed to disk, because it is not in-memory compacted memstore
283    // CF3 is flushed empty to memory (actually nothing happens to CF3)
284    region.flush(false);
285
286    // Recalculate everything
287    MemStoreSize cf1MemstoreSizePhaseIV = region.getStore(FAMILY1).getMemStoreSize();
288    MemStoreSize cf2MemstoreSizePhaseIV = region.getStore(FAMILY2).getMemStoreSize();
289    MemStoreSize cf3MemstoreSizePhaseIV = region.getStore(FAMILY3).getMemStoreSize();
290
291    long smallestSeqInRegionCurrentMemstorePhaseIV = getWAL(region)
292        .getEarliestMemStoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
293    long smallestSeqCF1PhaseIV = region.getOldestSeqIdOfStore(FAMILY1);
294    long smallestSeqCF2PhaseIV = region.getOldestSeqIdOfStore(FAMILY2);
295    long smallestSeqCF3PhaseIV = region.getOldestSeqIdOfStore(FAMILY3);
296
297    s = s + "----After SECOND FLUSH, CF1 size is:" + cf1MemstoreSizePhaseIV + ", CF2 size is:"
298        + cf2MemstoreSizePhaseIV + " and CF3 size is:" + cf3MemstoreSizePhaseIV
299        + "\n";
300
301    s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseIV
302        + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseIV + ", " +
303        "the smallest sequence in CF2:"
304        + smallestSeqCF2PhaseIV +", the smallest sequence in CF3:" + smallestSeqCF3PhaseIV
305        + "\n";
306
307    // CF1's pipeline component (inserted before first flush) should be flushed to disk
308    // CF2 should be flushed to disk
309    assertTrue(cf1MemstoreSizePhaseIII.getDataSize() > cf1MemstoreSizePhaseIV.getDataSize());
310    assertEquals(0, cf2MemstoreSizePhaseIV.getDataSize());
311    assertEquals(MutableSegment.DEEP_OVERHEAD, cf2MemstoreSizePhaseIV.getHeapSize());
312
313    // CF3 shouldn't have been touched.
314    assertEquals(cf3MemstoreSizePhaseIV, cf3MemstoreSizePhaseII);
315
316    // the smallest LSN of CF3 shouldn't change
317    assertEquals(smallestSeqCF3PhaseII, smallestSeqCF3PhaseIV);
318
319    // CF3 should be bottleneck for WAL
320    assertEquals(s, smallestSeqInRegionCurrentMemstorePhaseIV, smallestSeqCF3PhaseIV);
321
322    // Flush!!!!!!!!!!!!!!!!!!!!!!
323    // Trying to clean the existing memstores, CF2 all flushed to disk. The single
324    // memstore segment in the compaction pipeline of CF1 and CF3 should be flushed to disk.
325    region.flush(true);
326
327    // Recalculate everything
328    MemStoreSize cf1MemstoreSizePhaseV = region.getStore(FAMILY1).getMemStoreSize();
329    MemStoreSize cf2MemstoreSizePhaseV = region.getStore(FAMILY2).getMemStoreSize();
330    MemStoreSize cf3MemstoreSizePhaseV = region.getStore(FAMILY3).getMemStoreSize();
331    long smallestSeqInRegionCurrentMemstorePhaseV = getWAL(region)
332        .getEarliestMemStoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
333
334    assertEquals(0, cf1MemstoreSizePhaseV.getDataSize());
335    assertEquals(MutableSegment.DEEP_OVERHEAD, cf1MemstoreSizePhaseV.getHeapSize());
336    assertEquals(0, cf2MemstoreSizePhaseV.getDataSize());
337    assertEquals(MutableSegment.DEEP_OVERHEAD, cf2MemstoreSizePhaseV.getHeapSize());
338    assertEquals(0, cf3MemstoreSizePhaseV.getDataSize());
339    assertEquals(MutableSegment.DEEP_OVERHEAD, cf3MemstoreSizePhaseV.getHeapSize());
340
341    // What happens when we hit the memstore limit, but we are not able to find
342    // any Column Family above the threshold?
343    // In that case, we should flush all the CFs.
344
345    // The memstore limit is 100*1024 and the column family flush threshold is
346    // around 25*1024. We try to just hit the memstore limit with each CF's
347    // memstore being below the CF flush threshold.
348    for (int i = 1; i <= 300; i++) {
349      region.put(createPut(1, i));
350      region.put(createPut(2, i));
351      region.put(createPut(3, i));
352      region.put(createPut(4, i));
353      region.put(createPut(5, i));
354    }
355
356    region.flush(false);
357
358    s = s + "----AFTER THIRD AND FORTH FLUSH, The smallest sequence in region WAL is: "
359        + smallestSeqInRegionCurrentMemstorePhaseV
360        + ". After additional inserts and last flush, the entire region size is:" + region
361        .getMemStoreDataSize()
362        + "\n----------------------------------\n";
363
364    // Since we won't find any CF above the threshold, and hence no specific
365    // store to flush, we should flush all the memstores
366    // Also compacted memstores are flushed to disk.
367    assertEquals(0, region.getMemStoreDataSize());
368    System.out.println(s);
369    HBaseTestingUtility.closeRegionAndWAL(region);
370  }

374  @Test
375  public void testSelectiveFlushWithIndexCompaction() throws IOException {
376    /*------------------------------------------------------------------------------*/
377    /* SETUP */
378    // Set up the configuration
379    conf.setLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 300 * 1024);
380    conf.setLong(FlushLargeStoresPolicy.HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN, 75 * 1024);
381    conf.setDouble(CompactingMemStore.IN_MEMORY_FLUSH_THRESHOLD_FACTOR_KEY, 0.5);
382    // set memstore to index-compaction
383    conf.set(CompactingMemStore.COMPACTING_MEMSTORE_TYPE_KEY,
384        String.valueOf(MemoryCompactionPolicy.BASIC));
385
386    // Initialize the region
387    HRegion region = initHRegion("testSelectiveFlushWithIndexCompaction", conf);
388    verifyInMemoryFlushSize(region);
389    /*------------------------------------------------------------------------------*/
390    /* PHASE I - insertions */
391    // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
392    for (int i = 1; i <= 1200; i++) {
393      region.put(createPut(1, i));        // compacted memstore
394      if (i <= 100) {
395        region.put(createPut(2, i));
396        if (i <= 50) {
397          region.put(createDoublePut(3, i)); // subject for in-memory compaction
398        }
399      }
400    }
401    // Now add more puts for CF2, so that we only flush CF2 to disk
402    for (int i = 100; i < 2000; i++) {
403      region.put(createPut(2, i));
404    }
405
406    /*------------------------------------------------------------------------------*/
407    /*------------------------------------------------------------------------------*/
408    /* PHASE I - collect sizes */
409    long totalMemstoreSizePhaseI = region.getMemStoreDataSize();
410    // Find the smallest LSNs for edits wrt to each CF.
411    long smallestSeqCF1PhaseI = region.getOldestSeqIdOfStore(FAMILY1);
412    long smallestSeqCF2PhaseI = region.getOldestSeqIdOfStore(FAMILY2);
413    long smallestSeqCF3PhaseI = region.getOldestSeqIdOfStore(FAMILY3);
414    // Find the sizes of the memstores of each CF.
415    MemStoreSize cf1MemstoreSizePhaseI = region.getStore(FAMILY1).getMemStoreSize();
416    MemStoreSize cf2MemstoreSizePhaseI = region.getStore(FAMILY2).getMemStoreSize();
417    MemStoreSize cf3MemstoreSizePhaseI = region.getStore(FAMILY3).getMemStoreSize();
418    // Get the overall smallest LSN in the region's memstores.
419    long smallestSeqInRegionCurrentMemstorePhaseI = getWAL(region)
420        .getEarliestMemStoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
421
422    /*------------------------------------------------------------------------------*/
423    /* PHASE I - validation */
424    // The overall smallest LSN in the region's memstores should be the same as
425    // the LSN of the smallest edit in CF1
426    assertEquals(smallestSeqCF1PhaseI, smallestSeqInRegionCurrentMemstorePhaseI);
427    // Some other sanity checks.
428    assertTrue(smallestSeqCF1PhaseI < smallestSeqCF2PhaseI);
429    assertTrue(smallestSeqCF2PhaseI < smallestSeqCF3PhaseI);
430    assertTrue(cf1MemstoreSizePhaseI.getDataSize() > 0);
431    assertTrue(cf2MemstoreSizePhaseI.getDataSize() > 0);
432    assertTrue(cf3MemstoreSizePhaseI.getDataSize() > 0);
433
434    // The total memstore size should be the same as the sum of the sizes of
435    // memstores of CF1, CF2 and CF3.
436    assertEquals(totalMemstoreSizePhaseI, cf1MemstoreSizePhaseI.getDataSize()
437        + cf2MemstoreSizePhaseI.getDataSize() + cf3MemstoreSizePhaseI.getDataSize());
438
439    /*------------------------------------------------------------------------------*/
440    /* PHASE I - Flush */
441    // First Flush in Test!!!!!!!!!!!!!!!!!!!!!!
442    // CF1, CF2, CF3, all together they are above the flush size lower bound.
443    // Since CF1 and CF3 are compacting, CF2 is going to be flushed to disk.
444    // CF1 and CF3 - flushed to memory and flatten explicitly
445    region.flush(false);
446    CompactingMemStore cms1 = (CompactingMemStore) ((HStore) region.getStore(FAMILY1)).memstore;
447    CompactingMemStore cms3 = (CompactingMemStore) ((HStore) region.getStore(FAMILY3)).memstore;
448    cms1.flushInMemory();
449    cms3.flushInMemory();
450
451    // CF3/CF1 should be merged so wait here to be sure the compaction is done
452    while (((CompactingMemStore) ((HStore) region.getStore(FAMILY1)).memstore)
453        .isMemStoreFlushingInMemory()) {
454      Threads.sleep(10);
455    }
456    while (((CompactingMemStore) ((HStore) region.getStore(FAMILY3)).memstore)
457        .isMemStoreFlushingInMemory()) {
458      Threads.sleep(10);
459    }
460
461    /*------------------------------------------------------------------------------*/
462    /*------------------------------------------------------------------------------*/
463    /* PHASE II - collect sizes */
464    // Recalculate everything
465    MemStoreSize cf1MemstoreSizePhaseII = region.getStore(FAMILY1).getMemStoreSize();
466    MemStoreSize cf2MemstoreSizePhaseII = region.getStore(FAMILY2).getMemStoreSize();
467    MemStoreSize cf3MemstoreSizePhaseII = region.getStore(FAMILY3).getMemStoreSize();
468    long smallestSeqInRegionCurrentMemstorePhaseII = getWAL(region)
469        .getEarliestMemStoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
470    // Find the smallest LSNs for edits wrt to each CF.
471    long smallestSeqCF3PhaseII = region.getOldestSeqIdOfStore(FAMILY3);
472    long totalMemstoreSizePhaseII = region.getMemStoreDataSize();
473
474    /*------------------------------------------------------------------------------*/
475    /* PHASE II - validation */
476    // CF1 was flushed to memory, should be flattened and take less space
477    assertEquals(cf1MemstoreSizePhaseII.getDataSize() , cf1MemstoreSizePhaseI.getDataSize());
478    assertTrue(cf1MemstoreSizePhaseII.getHeapSize() < cf1MemstoreSizePhaseI.getHeapSize());
479    // CF2 should become empty
480    assertEquals(0, cf2MemstoreSizePhaseII.getDataSize());
481    assertEquals(MutableSegment.DEEP_OVERHEAD, cf2MemstoreSizePhaseII.getHeapSize());
482    // verify that CF3 was flushed to memory and was not compacted (this is an approximation check)
483    // if compacted CF# should be at least twice less because its every key was duplicated
484    assertEquals(cf3MemstoreSizePhaseII.getDataSize() , cf3MemstoreSizePhaseI.getDataSize());
485    assertTrue(cf3MemstoreSizePhaseI.getHeapSize() / 2 < cf3MemstoreSizePhaseII.getHeapSize());
486
487    // Now the smallest LSN in the region should be the same as the smallest
488    // LSN in the memstore of CF1.
489    assertEquals(smallestSeqInRegionCurrentMemstorePhaseII, smallestSeqCF1PhaseI);
490    // The total memstore size should be the same as the sum of the sizes of
491    // memstores of CF1, CF2 and CF3. Counting the empty active segments in CF1/2/3 and pipeline
492    // items in CF1/2

952   * memstore instance. It takes the updatesLock exclusively, pushes active into the pipeline,
953   * releases updatesLock and compacts the pipeline.
954   */

956    private final HRegion stressedRegion;
957    private final int startNumber;
958
959    ConcurrentPutRunnable(HRegion r, int i) {
960      this.stressedRegion = r;
961      this.startNumber = i;
962    }
963
964    @Override
965    public void run() {
966
967      try {
968        int dummy = startNumber / 10000;
969        System.out.print("Thread " + dummy + " with start number " + startNumber + " starts\n");
970        // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
971        for (int i = startNumber; i <= startNumber + 3000; i++) {
972          stressedRegion.put(createPut(1, i));
973          if (i <= startNumber + 2000) {
974            stressedRegion.put(createPut(2, i));
975            if (i <= startNumber + 1000) {
976              stressedRegion.put(createPut(3, i));
977            }
978          }
979        }
980        System.out.print("Thread with start number " + startNumber + " continues to more puts\n");
981        // Now add more puts for CF2, so that we only flush CF2 to disk
982        for (int i = startNumber + 3000; i < startNumber + 5000; i++) {
983          stressedRegion.put(createPut(2, i));
984        }
985        // And add more puts for CF1
986        for (int i = startNumber + 5000; i < startNumber + 7000; i++) {
987          stressedRegion.put(createPut(1, i));
988        }
989        System.out.print("Thread with start number " + startNumber + " flushes\n");
990        // flush (IN MEMORY) one of the stores (each thread flushes different store)
991        // and wait till the flush and the following action are done
992        if (startNumber == 0) {
993          ((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY1)).memstore)
994              .flushInMemory();
995          while (((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY1)).memstore)
996              .isMemStoreFlushingInMemory()) {
997            Threads.sleep(10);
998          }
999        }
1000        if (startNumber == 10000) {
1001          ((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY2)).memstore).flushInMemory();
1002          while (((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY2)).memstore)
1003              .isMemStoreFlushingInMemory()) {
1004            Threads.sleep(10);
1005          }
1006        }
1007        if (startNumber == 20000) {
1008          ((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY3)).memstore).flushInMemory();
1009          while (((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY3)).memstore)
1010              .isMemStoreFlushingInMemory()) {
1011            Threads.sleep(10);
1012          }
1013        }
1014        System.out.print("Thread with start number " + startNumber + " finishes\n");
1015      } catch (IOException e) {
1016        assert false;
1017      }
1018    }
1019  }
