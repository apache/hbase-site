059/**
060 * Tests for the region server status page and its template.
061 */
062@Category({RegionServerTests.class, SmallTests.class})
063public class TestRSStatusServlet {
064  private static final Log LOG = LogFactory.getLog(TestRSStatusServlet.class);
065  private HRegionServer rs;
066  private RSRpcServices rpcServices;
067  private RpcServerInterface rpcServer;
068
069  static final int FAKE_IPC_PORT = 1585;
070  static final int FAKE_WEB_PORT = 1586;
071
072  private final ServerName fakeServerName =
073      ServerName.valueOf("localhost", FAKE_IPC_PORT, 11111);
074  private final GetServerInfoResponse fakeResponse =
075    ResponseConverter.buildGetServerInfoResponse(fakeServerName, FAKE_WEB_PORT);
076
077  private final ServerName fakeMasterAddress =
078      ServerName.valueOf("localhost", 60010, 1212121212);
079
080  @Rule
081  public TestName name = new TestName();
082
083  @Before
084  public void setupBasicMocks() throws IOException, ServiceException {
085    rs = Mockito.mock(HRegionServer.class);
086    rpcServices = Mockito.mock(RSRpcServices.class);
087    rpcServer = Mockito.mock(RpcServerInterface.class);
088    Mockito.doReturn(HBaseConfiguration.create())
089      .when(rs).getConfiguration();
090    Mockito.doReturn(rpcServices).when(rs).getRSRpcServices();
091    Mockito.doReturn(rpcServer).when(rs).getRpcServer();
092    Mockito.doReturn(fakeResponse).when(rpcServices).getServerInfo(
093      (RpcController)Mockito.any(), (GetServerInfoRequest)Mockito.any());
094    // Fake ZKW
095    ZooKeeperWatcher zkw = Mockito.mock(ZooKeeperWatcher.class);
096    Mockito.doReturn("fakequorum").when(zkw).getQuorum();
097    Mockito.doReturn(zkw).when(rs).getZooKeeper();
098
099    // Fake CacheConfig
100    LOG.warn("The " + HConstants.HFILE_BLOCK_CACHE_SIZE_KEY + " is set to 0");
101    CacheConfig cacheConf = Mockito.mock(CacheConfig.class);
102    Mockito.doReturn(null).when(cacheConf).getBlockCache();
103    Mockito.doReturn(cacheConf).when(rs).getCacheConfig();
104
105    // Fake MasterAddressTracker
106    MasterAddressTracker mat = Mockito.mock(MasterAddressTracker.class);
107    Mockito.doReturn(fakeMasterAddress).when(mat).getMasterAddress();
108    Mockito.doReturn(mat).when(rs).getMasterAddressTracker();
109
110    MetricsRegionServer rms = Mockito.mock(MetricsRegionServer.class);
111    Mockito.doReturn(new MetricsRegionServerWrapperStub()).when(rms).getRegionServerWrapper();
112    Mockito.doReturn(rms).when(rs).getRegionServerMetrics();
113
114    MetricsHBaseServer ms = Mockito.mock(MetricsHBaseServer.class);
115    Mockito.doReturn(new MetricsHBaseServerWrapperStub()).when(ms).getHBaseServerWrapper();
116    Mockito.doReturn(ms).when(rpcServer).getMetrics();
117  }
118
119  @Test
120  public void testBasic() throws IOException, ServiceException {
121    new RSStatusTmpl().render(new StringWriter(), rs);
122  }
123
124  @Test
125  public void testWithRegions() throws IOException, ServiceException {
126    HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
127    List<RegionInfo> regions = Lists.newArrayList(
128        RegionInfoBuilder.newBuilder(htd.getTableName())
129            .setStartKey(Bytes.toBytes("a"))
130            .setEndKey(Bytes.toBytes("d"))
131            .build(),
132        RegionInfoBuilder.newBuilder(htd.getTableName())
133            .setStartKey(Bytes.toBytes("d"))
134            .setEndKey(Bytes.toBytes("z"))
135            .build()
136        );
137    Mockito.doReturn(ResponseConverter.buildGetOnlineRegionResponse(
138      regions)).when(rpcServices).getOnlineRegion((RpcController)Mockito.any(),
139        (GetOnlineRegionRequest)Mockito.any());
140    new RSStatusTmpl().render(new StringWriter(), rs);
141  }
142}
