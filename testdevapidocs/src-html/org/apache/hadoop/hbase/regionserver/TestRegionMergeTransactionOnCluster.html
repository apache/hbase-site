138  public void testWholesomeMerge() throws Exception {
139    LOG.info("Starting " + name.getMethodName());
140    final TableName tableName = TableName.valueOf(name.getMethodName());
141
142    try {
143      // Create table and load data.
144      Table table = createTableAndLoadData(MASTER, tableName);
145      // Merge 1st and 2nd region
146      mergeRegionsAndVerifyRegionNum(MASTER, tableName, 0, 1, INITIAL_REGION_NUM - 1);
147
148      // Merge 2nd and 3th region
149      PairOfSameType<RegionInfo> mergedRegions =
150        mergeRegionsAndVerifyRegionNum(MASTER, tableName, 1, 2, INITIAL_REGION_NUM - 2);
151
152      verifyRowCount(table, ROWSIZE);
153
154      // Randomly choose one of the two merged regions
155      RegionInfo hri = RandomUtils.nextBoolean() ? mergedRegions.getFirst() : mergedRegions.getSecond();
156      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
157      AssignmentManager am = cluster.getMaster().getAssignmentManager();
158      RegionStates regionStates = am.getRegionStates();
159
160      // We should not be able to assign it again
161      am.assign(hri);
162      assertFalse("Merged region can't be assigned", regionStates.isRegionInTransition(hri));
163
164      // We should not be able to unassign it either
165      am.unassign(hri);
166      assertFalse("Merged region can't be unassigned", regionStates.isRegionInTransition(hri));
167
168      table.close();
169    } finally {
170      TEST_UTIL.deleteTable(tableName);
171    }
172  }

201  @Test
202  public void testCleanMergeReference() throws Exception {
203    LOG.info("Starting " + name.getMethodName());
204    ADMIN.catalogJanitorSwitch(false);
205    final TableName tableName = TableName.valueOf(name.getMethodName());
206    try {
207      // Create table and load data.
208      Table table = createTableAndLoadData(MASTER, tableName);
209      // Merge 1st and 2nd region
210      mergeRegionsAndVerifyRegionNum(MASTER, tableName, 0, 1, INITIAL_REGION_NUM - 1);
211      verifyRowCount(table, ROWSIZE);
212      table.close();
213
214      List<Pair<RegionInfo, ServerName>> tableRegions = MetaTableAccessor
215          .getTableRegionsAndLocations(MASTER.getConnection(), tableName);
216      RegionInfo mergedRegionInfo = tableRegions.get(0).getFirst();
217      TableDescriptor tableDescriptor = MASTER.getTableDescriptors().get(
218          tableName);
219      Result mergedRegionResult = MetaTableAccessor.getRegionResult(
220        MASTER.getConnection(), mergedRegionInfo.getRegionName());
221
222      // contains merge reference in META
223      assertTrue(MetaTableAccessor.hasMergeRegions(mergedRegionResult.rawCells()));
224
225      // merging regions' directory are in the file system all the same
226      List<RegionInfo> p = MetaTableAccessor.getMergeRegions(mergedRegionResult.rawCells());
227      RegionInfo regionA = p.get(0);
228      RegionInfo regionB = p.get(1);
229      FileSystem fs = MASTER.getMasterFileSystem().getFileSystem();
230      Path rootDir = MASTER.getMasterFileSystem().getRootDir();
231
232      Path tabledir = FSUtils.getTableDir(rootDir, mergedRegionInfo.getTable());
233      Path regionAdir = new Path(tabledir, regionA.getEncodedName());
234      Path regionBdir = new Path(tabledir, regionB.getEncodedName());
235      assertTrue(fs.exists(regionAdir));
236      assertTrue(fs.exists(regionBdir));
237
238      ColumnFamilyDescriptor[] columnFamilies = tableDescriptor.getColumnFamilies();
239      HRegionFileSystem hrfs = new HRegionFileSystem(
240        TEST_UTIL.getConfiguration(), fs, tabledir, mergedRegionInfo);
241      int count = 0;
242      for(ColumnFamilyDescriptor colFamily : columnFamilies) {
243        count += hrfs.getStoreFiles(colFamily.getName()).size();
244      }
245      ADMIN.compactRegion(mergedRegionInfo.getRegionName());
246      // clean up the merged region store files
247      // wait until merged region have reference file
248      long timeout = System.currentTimeMillis() + waitTime;
249      int newcount = 0;
250      while (System.currentTimeMillis() < timeout) {
251        for(ColumnFamilyDescriptor colFamily : columnFamilies) {
252          newcount += hrfs.getStoreFiles(colFamily.getName()).size();
253        }
254        if(newcount > count) {
255          break;
256        }
257        Thread.sleep(50);
258      }
259      assertTrue(newcount > count);
260      List<RegionServerThread> regionServerThreads = TEST_UTIL.getHBaseCluster()
261          .getRegionServerThreads();
262      for (RegionServerThread rs : regionServerThreads) {
263        CompactedHFilesDischarger cleaner = new CompactedHFilesDischarger(100, null,
264            rs.getRegionServer(), false);
265        cleaner.chore();
266        Thread.sleep(1000);
267      }
268      while (System.currentTimeMillis() < timeout) {
269        int newcount1 = 0;
270        for(ColumnFamilyDescriptor colFamily : columnFamilies) {
271          newcount1 += hrfs.getStoreFiles(colFamily.getName()).size();
272        }
273        if(newcount1 <= 1) {
274          break;
275        }
276        Thread.sleep(50);
277      }
278      // run CatalogJanitor to clean merge references in hbase:meta and archive the
279      // files of merging regions
280      int cleaned = 0;
281      while (cleaned == 0) {
282        cleaned = ADMIN.runCatalogJanitor();
283        LOG.debug("catalog janitor returned " + cleaned);
284        Thread.sleep(50);
285        // Cleanup is async so wait till all procedures are done running.
286        ProcedureTestingUtility.waitNoProcedureRunning(
287            TEST_UTIL.getMiniHBaseCluster().getMaster().getMasterProcedureExecutor());
288      }
289      // We used to check for existence of region in fs but sometimes the region dir was
290      // cleaned up by the time we got here making the test sometimes flakey.
291      assertTrue(cleaned > 0);
292
293      // Wait around a bit to give stuff a chance to complete.
294      while (true) {
295        mergedRegionResult = MetaTableAccessor
296          .getRegionResult(TEST_UTIL.getConnection(), mergedRegionInfo.getRegionName());
297        if (MetaTableAccessor.hasMergeRegions(mergedRegionResult.rawCells())) {
298          LOG.info("Waiting on cleanup of merge columns {}",
299            Arrays.asList(mergedRegionResult.rawCells()).stream().
300              map(c -> c.toString()).collect(Collectors.joining(",")));
301          Threads.sleep(50);
302        } else {
303          break;
304        }
305      }
306      assertFalse(MetaTableAccessor.hasMergeRegions(mergedRegionResult.rawCells()));
307    } finally {
308      ADMIN.catalogJanitorSwitch(true);
309      TEST_UTIL.deleteTable(tableName);
310    }
311  }

313  /**
314   * This test tests 1, merging region not online;
315   * 2, merging same two regions; 3, merging unknown regions.
316   * They are in one test case so that we don't have to create
317   * many tables, and these tests are simple.
318   */
319  @Test
320  public void testMerge() throws Exception {
321    LOG.info("Starting " + name.getMethodName());
322    final TableName tableName = TableName.valueOf(name.getMethodName());
323    final Admin admin = TEST_UTIL.getAdmin();
324
325    try {
326      // Create table and load data.
327      Table table = createTableAndLoadData(MASTER, tableName);
328      AssignmentManager am = MASTER.getAssignmentManager();
329      List<RegionInfo> regions = am.getRegionStates().getRegionsOfTable(tableName);
330      // Fake offline one region
331      RegionInfo a = regions.get(0);
332      RegionInfo b = regions.get(1);
333      am.unassign(b);
334      am.offlineRegion(b);
335      try {
336        // Merge offline region. Region a is offline here
337        FutureUtils.get(
338          admin.mergeRegionsAsync(a.getEncodedNameAsBytes(), b.getEncodedNameAsBytes(), false));
339        fail("Offline regions should not be able to merge");
340      } catch (DoNotRetryRegionException ie) {
341        System.out.println(ie);
342        assertTrue(ie instanceof MergeRegionException);
343      }
344
345      try {
346        // Merge the same region: b and b.
347        FutureUtils
348          .get(admin.mergeRegionsAsync(b.getEncodedNameAsBytes(), b.getEncodedNameAsBytes(), true));
349        fail("A region should not be able to merge with itself, even forcfully");
350      } catch (IOException ie) {
351        assertTrue("Exception should mention regions not online",
352          StringUtils.stringifyException(ie).contains("region to itself") &&
353            ie instanceof MergeRegionException);
354      }
355
356      try {
357        // Merge unknown regions
358        FutureUtils.get(admin.mergeRegionsAsync(Bytes.toBytes("-f1"), Bytes.toBytes("-f2"), true));
359        fail("Unknown region could not be merged");
360      } catch (IOException ie) {
361        assertTrue("UnknownRegionException should be thrown", ie instanceof UnknownRegionException);
362      }
363      table.close();
364    } finally {
365      TEST_UTIL.deleteTable(tableName);
366    }
367  }

369  @Test
370  public void testMergeWithReplicas() throws Exception {
371    final TableName tableName = TableName.valueOf(name.getMethodName());
372    try {
373      // Create table and load data.
374      Table table = createTableAndLoadData(MASTER, tableName, 5, 2);
375      List<Pair<RegionInfo, ServerName>> initialRegionToServers =
376        MetaTableAccessor.getTableRegionsAndLocations(TEST_UTIL.getConnection(), tableName);
377      // Merge 1st and 2nd region
378      PairOfSameType<RegionInfo> mergedRegions =
379        mergeRegionsAndVerifyRegionNum(MASTER, tableName, 0, 2, 5 * 2 - 2);
380      List<Pair<RegionInfo, ServerName>> currentRegionToServers =
381        MetaTableAccessor.getTableRegionsAndLocations(TEST_UTIL.getConnection(), tableName);
382      List<RegionInfo> initialRegions = new ArrayList<>();
383      for (Pair<RegionInfo, ServerName> p : initialRegionToServers) {
384        initialRegions.add(p.getFirst());
385      }
386      List<RegionInfo> currentRegions = new ArrayList<>();
387      for (Pair<RegionInfo, ServerName> p : currentRegionToServers) {
388        currentRegions.add(p.getFirst());
389      }
390      assertTrue(initialRegions.contains(mergedRegions.getFirst())); //this is the first region
391      assertTrue(initialRegions.contains(RegionReplicaUtil
392        .getRegionInfoForReplica(mergedRegions.getFirst(), 1))); //this is the replica of the first region
393      assertTrue(initialRegions.contains(mergedRegions.getSecond())); //this is the second region
394      assertTrue(initialRegions.contains(RegionReplicaUtil
395        .getRegionInfoForReplica(mergedRegions.getSecond(), 1))); //this is the replica of the second region
396      assertTrue(!initialRegions.contains(currentRegions.get(0))); //this is the new region
397      assertTrue(!initialRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(currentRegions.get(0), 1))); //replica of the new region
398      assertTrue(currentRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(currentRegions.get(0), 1))); //replica of the new region
399      assertTrue(!currentRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(mergedRegions.getFirst(), 1))); //replica of the merged region
400      assertTrue(!currentRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(mergedRegions.getSecond(), 1))); //replica of the merged region
401      table.close();
402    } finally {
403      TEST_UTIL.deleteTable(tableName);
404    }
405  }

407  private PairOfSameType<RegionInfo> mergeRegionsAndVerifyRegionNum(
408      HMaster master, TableName tablename,
409      int regionAnum, int regionBnum, int expectedRegionNum) throws Exception {
410    PairOfSameType<RegionInfo> mergedRegions =
411      requestMergeRegion(master, tablename, regionAnum, regionBnum);
412    waitAndVerifyRegionNum(master, tablename, expectedRegionNum);
413    return mergedRegions;
414  }

416  private PairOfSameType<RegionInfo> requestMergeRegion(
417      HMaster master, TableName tablename,
418      int regionAnum, int regionBnum) throws Exception {
419    List<Pair<RegionInfo, ServerName>> tableRegions = MetaTableAccessor
420        .getTableRegionsAndLocations(
421            TEST_UTIL.getConnection(), tablename);
422    RegionInfo regionA = tableRegions.get(regionAnum).getFirst();
423    RegionInfo regionB = tableRegions.get(regionBnum).getFirst();
424    ADMIN.mergeRegionsAsync(
425      regionA.getEncodedNameAsBytes(),
426      regionB.getEncodedNameAsBytes(), false);
427    return new PairOfSameType<>(regionA, regionB);
428  }

430  private void waitAndVerifyRegionNum(HMaster master, TableName tablename,
431      int expectedRegionNum) throws Exception {
432    List<Pair<RegionInfo, ServerName>> tableRegionsInMeta;
433    List<RegionInfo> tableRegionsInMaster;
434    long timeout = System.currentTimeMillis() + waitTime;
435    while (System.currentTimeMillis() < timeout) {
436      tableRegionsInMeta =
437          MetaTableAccessor.getTableRegionsAndLocations(TEST_UTIL.getConnection(), tablename);
438      tableRegionsInMaster =
439          master.getAssignmentManager().getRegionStates().getRegionsOfTable(tablename);
440      LOG.info(Objects.toString(tableRegionsInMaster));
441      LOG.info(Objects.toString(tableRegionsInMeta));
442      int tableRegionsInMetaSize = tableRegionsInMeta.size();
443      int tableRegionsInMasterSize = tableRegionsInMaster.size();
444      if (tableRegionsInMetaSize == expectedRegionNum
445          && tableRegionsInMasterSize == expectedRegionNum) {
446        break;
447      }
448      Thread.sleep(250);
449    }
450
451    tableRegionsInMeta = MetaTableAccessor.getTableRegionsAndLocations(
452        TEST_UTIL.getConnection(), tablename);
453    LOG.info("Regions after merge:" + Joiner.on(',').join(tableRegionsInMeta));
454    assertEquals(expectedRegionNum, tableRegionsInMeta.size());
455  }

462  private Table createTableAndLoadData(HMaster master, TableName tablename,
463      int numRegions, int replication) throws Exception {
464    assertTrue("ROWSIZE must > numregions:" + numRegions, ROWSIZE > numRegions);
465    byte[][] splitRows = new byte[numRegions - 1][];
466    for (int i = 0; i < splitRows.length; i++) {
467      splitRows[i] = ROWS[(i + 1) * ROWSIZE / numRegions];
468    }
469
470    Table table = TEST_UTIL.createTable(tablename, FAMILYNAME, splitRows);
471    LOG.info("Created " + table.getName());
472    if (replication > 1) {
473      HBaseTestingUtility.setReplicas(ADMIN, tablename, replication);
474      LOG.info("Set replication of " + replication + " on " + table.getName());
475    }
476    loadData(table);
477    LOG.info("Loaded " + table.getName());
478    verifyRowCount(table, ROWSIZE);
479    LOG.info("Verified " + table.getName());
480
481    List<Pair<RegionInfo, ServerName>> tableRegions;
482    TEST_UTIL.waitUntilAllRegionsAssigned(tablename);
483    LOG.info("All regions assigned for table - " + table.getName());
484    tableRegions = MetaTableAccessor.getTableRegionsAndLocations(
485        TEST_UTIL.getConnection(), tablename);
486    assertEquals("Wrong number of regions in table " + tablename,
487        numRegions * replication, tableRegions.size());
488    LOG.info(tableRegions.size() + "Regions after load: " + Joiner.on(',').join(tableRegions));
489    assertEquals(numRegions * replication, tableRegions.size());
490    return table;
491  }

493  private static byte[][] makeN(byte[] base, int n) {
494    byte[][] ret = new byte[n][];
495    for (int i = 0; i < n; i++) {
496      ret[i] = Bytes.add(base, Bytes.toBytes(String.format("%04d", i)));
497    }
498    return ret;
499  }

501  private void loadData(Table table) throws IOException {
502    for (int i = 0; i < ROWSIZE; i++) {
503      Put put = new Put(ROWS[i]);
504      put.addColumn(FAMILYNAME, QUALIFIER, Bytes.toBytes(i));
505      table.put(put);
506    }
507  }

509  private void verifyRowCount(Table table, int expectedRegionNum)
510      throws IOException {
511    ResultScanner scanner = table.getScanner(new Scan());
512    int rowCount = 0;
513    while (scanner.next() != null) {
514      rowCount++;
515    }
516    assertEquals(expectedRegionNum, rowCount);
517    scanner.close();
518  }

521  public static class MyMaster extends HMaster {
522    public MyMaster(Configuration conf) throws IOException, KeeperException, InterruptedException {
523      super(conf);
524    }
525
526    @Override
527    protected RSRpcServices createRpcServices() throws IOException {
528      return new MyMasterRpcServices(this);
529    }

532  static class MyMasterRpcServices extends MasterRpcServices {
533    static AtomicBoolean enabled = new AtomicBoolean(false);
534
535    private HMaster myMaster;
536    public MyMasterRpcServices(HMaster master) throws IOException {
537      super(master);
538      myMaster = master;
539    }
540
541    @Override
542    public ReportRegionStateTransitionResponse reportRegionStateTransition(RpcController c,
543        ReportRegionStateTransitionRequest req) throws ServiceException {
544      ReportRegionStateTransitionResponse resp = super.reportRegionStateTransition(c, req);
545      if (enabled.get() && req.getTransition(0).getTransitionCode()
546          == TransitionCode.READY_TO_MERGE && !resp.hasErrorMessage()) {
547        RegionStates regionStates = myMaster.getAssignmentManager().getRegionStates();
548        for (RegionState regionState: regionStates.getRegionsStateInTransition()) {
549          // Find the merging_new region and remove it
550          if (regionState.isMergingNew()) {
551            regionStates.deleteRegion(regionState.getRegion());
552          }
553        }
554      }
555      return resp;
556    }
557  }
