137  @Test
138  public void testWholesomeMerge() throws Exception {
139    LOG.info("Starting testWholesomeMerge");
140    final TableName tableName =
141        TableName.valueOf("testWholesomeMerge");
142
143    // Create table and load data.
144    Table table = createTableAndLoadData(MASTER, tableName);
145    // Merge 1st and 2nd region
146    mergeRegionsAndVerifyRegionNum(MASTER, tableName, 0, 1,
147        INITIAL_REGION_NUM - 1);
148
149    // Merge 2nd and 3th region
150    PairOfSameType<HRegionInfo> mergedRegions =
151      mergeRegionsAndVerifyRegionNum(MASTER, tableName, 1, 2,
152        INITIAL_REGION_NUM - 2);
153
154    verifyRowCount(table, ROWSIZE);
155
156    // Randomly choose one of the two merged regions
157    HRegionInfo hri = RandomUtils.nextBoolean() ?
158      mergedRegions.getFirst() : mergedRegions.getSecond();
159    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
160    AssignmentManager am = cluster.getMaster().getAssignmentManager();
161    RegionStates regionStates = am.getRegionStates();
162    long start = EnvironmentEdgeManager.currentTime();
163    while (!regionStates.isRegionInState(hri, State.MERGED)) {
164      assertFalse("Timed out in waiting one merged region to be in state MERGED",
165        EnvironmentEdgeManager.currentTime() - start > 60000);
166      Thread.sleep(500);
167    }
168
169    // We should not be able to assign it again
170    am.assign(hri, true);
171    assertFalse("Merged region can't be assigned",
172      regionStates.isRegionInTransition(hri));
173    assertTrue(regionStates.isRegionInState(hri, State.MERGED));
174
175    // We should not be able to unassign it either
176    am.unassign(hri, null);
177    assertFalse("Merged region can't be unassigned",
178      regionStates.isRegionInTransition(hri));
179    assertTrue(regionStates.isRegionInState(hri, State.MERGED));
180
181    table.close();
182  }

207  @Test
208  public void testCleanMergeReference() throws Exception {
209    LOG.info("Starting testCleanMergeReference");
210    ADMIN.enableCatalogJanitor(false);
211    try {
212      final TableName tableName =
213          TableName.valueOf("testCleanMergeReference");
214      // Create table and load data.
215      Table table = createTableAndLoadData(MASTER, tableName);
216      // Merge 1st and 2nd region
217      mergeRegionsAndVerifyRegionNum(MASTER, tableName, 0, 1,
218          INITIAL_REGION_NUM - 1);
219      verifyRowCount(table, ROWSIZE);
220      table.close();
221
222      List<Pair<HRegionInfo, ServerName>> tableRegions = MetaTableAccessor
223          .getTableRegionsAndLocations(MASTER.getConnection(), tableName);
224      HRegionInfo mergedRegionInfo = tableRegions.get(0).getFirst();
225      HTableDescriptor tableDescriptor = MASTER.getTableDescriptors().get(
226          tableName);
227      Result mergedRegionResult = MetaTableAccessor.getRegionResult(
228        MASTER.getConnection(), mergedRegionInfo.getRegionName());
229
230      // contains merge reference in META
231      assertTrue(mergedRegionResult.getValue(HConstants.CATALOG_FAMILY,
232          HConstants.MERGEA_QUALIFIER) != null);
233      assertTrue(mergedRegionResult.getValue(HConstants.CATALOG_FAMILY,
234          HConstants.MERGEB_QUALIFIER) != null);
235
236      // merging regions' directory are in the file system all the same
237      PairOfSameType<HRegionInfo> p = MetaTableAccessor.getMergeRegions(mergedRegionResult);
238      HRegionInfo regionA = p.getFirst();
239      HRegionInfo regionB = p.getSecond();
240      FileSystem fs = MASTER.getMasterFileSystem().getFileSystem();
241      Path rootDir = MASTER.getMasterFileSystem().getRootDir();
242
243      Path tabledir = FSUtils.getTableDir(rootDir, mergedRegionInfo.getTable());
244      Path regionAdir = new Path(tabledir, regionA.getEncodedName());
245      Path regionBdir = new Path(tabledir, regionB.getEncodedName());
246      assertTrue(fs.exists(regionAdir));
247      assertTrue(fs.exists(regionBdir));
248
249      HColumnDescriptor[] columnFamilies = tableDescriptor.getColumnFamilies();
250      HRegionFileSystem hrfs = new HRegionFileSystem(
251        TEST_UTIL.getConfiguration(), fs, tabledir, mergedRegionInfo);
252      int count = 0;
253      for(HColumnDescriptor colFamily : columnFamilies) {
254        count += hrfs.getStoreFiles(colFamily.getName()).size();
255      }
256      ADMIN.compactRegion(mergedRegionInfo.getRegionName());
257      // clean up the merged region store files
258      // wait until merged region have reference file
259      long timeout = System.currentTimeMillis() + waitTime;
260      int newcount = 0;
261      while (System.currentTimeMillis() < timeout) {
262        for(HColumnDescriptor colFamily : columnFamilies) {
263          newcount += hrfs.getStoreFiles(colFamily.getName()).size();
264        }
265        if(newcount > count) {
266          break;
267        }
268        Thread.sleep(50);
269      }
270      assertTrue(newcount > count);
271      List<RegionServerThread> regionServerThreads = TEST_UTIL.getHBaseCluster()
272          .getRegionServerThreads();
273      for (RegionServerThread rs : regionServerThreads) {
274        CompactedHFilesDischarger cleaner = new CompactedHFilesDischarger(100, null,
275            rs.getRegionServer(), false);
276        cleaner.chore();
277        Thread.sleep(1000);
278      }
279      while (System.currentTimeMillis() < timeout) {
280        int newcount1 = 0;
281        for(HColumnDescriptor colFamily : columnFamilies) {
282          newcount1 += hrfs.getStoreFiles(colFamily.getName()).size();
283        }
284        if(newcount1 <= 1) {
285          break;
286        }
287        Thread.sleep(50);
288      }
289      // run CatalogJanitor to clean merge references in hbase:meta and archive the
290      // files of merging regions
291      int cleaned = 0;
292      while (cleaned == 0) {
293        cleaned = ADMIN.runCatalogScan();
294        LOG.debug("catalog janitor returned " + cleaned);
295        Thread.sleep(50);
296      }
297      assertFalse(regionAdir.toString(), fs.exists(regionAdir));
298      assertFalse(regionBdir.toString(), fs.exists(regionBdir));
299      assertTrue(cleaned > 0);
300
301      mergedRegionResult = MetaTableAccessor.getRegionResult(
302        TEST_UTIL.getConnection(), mergedRegionInfo.getRegionName());
303      assertFalse(mergedRegionResult.getValue(HConstants.CATALOG_FAMILY,
304          HConstants.MERGEA_QUALIFIER) != null);
305      assertFalse(mergedRegionResult.getValue(HConstants.CATALOG_FAMILY,
306          HConstants.MERGEB_QUALIFIER) != null);
307
308    } finally {
309      ADMIN.enableCatalogJanitor(true);
310    }
311  }

313  /**
314   * This test tests 1, merging region not online;
315   * 2, merging same two regions; 3, merging unknown regions.
316   * They are in one test case so that we don't have to create
317   * many tables, and these tests are simple.
318   */
319  @Test
320  public void testMerge() throws Exception {
321    LOG.info("Starting testMerge");
322    final TableName tableName = TableName.valueOf("testMerge");
323
324    try {
325      // Create table and load data.
326      Table table = createTableAndLoadData(MASTER, tableName);
327      RegionStates regionStates = MASTER.getAssignmentManager().getRegionStates();
328      List<HRegionInfo> regions = regionStates.getRegionsOfTable(tableName);
329      // Fake offline one region
330      HRegionInfo a = regions.get(0);
331      HRegionInfo b = regions.get(1);
332      regionStates.regionOffline(a);
333      try {
334        // Merge offline region. Region a is offline here
335        ADMIN.mergeRegions(a.getEncodedNameAsBytes(), b.getEncodedNameAsBytes(), false);
336        fail("Offline regions should not be able to merge");
337      } catch (IOException ie) {
338        System.out.println(ie);
339        assertTrue("Exception should mention regions not online",
340          StringUtils.stringifyException(ie).contains("regions not online")
341            && ie instanceof MergeRegionException);
342      }
343      try {
344        // Merge the same region: b and b.
345        ADMIN.mergeRegions(b.getEncodedNameAsBytes(), b.getEncodedNameAsBytes(), true);
346        fail("A region should not be able to merge with itself, even forcifully");
347      } catch (IOException ie) {
348        assertTrue("Exception should mention regions not online",
349          StringUtils.stringifyException(ie).contains("region to itself")
350            && ie instanceof MergeRegionException);
351      }
352      try {
353        // Merge unknown regions
354        ADMIN.mergeRegions(Bytes.toBytes("-f1"), Bytes.toBytes("-f2"), true);
355        fail("Unknown region could not be merged");
356      } catch (IOException ie) {
357        assertTrue("UnknownRegionException should be thrown",
358          ie instanceof UnknownRegionException);
359      }
360      table.close();
361    } finally {
362      TEST_UTIL.deleteTable(tableName);
363    }
364  }

366  @Test
367  public void testMergeWithReplicas() throws Exception {
368    final TableName tableName = TableName.valueOf("testMergeWithReplicas");
369    // Create table and load data.
370    createTableAndLoadData(MASTER, tableName, 5, 2);
371    List<Pair<HRegionInfo, ServerName>> initialRegionToServers =
372        MetaTableAccessor.getTableRegionsAndLocations(
373            TEST_UTIL.getConnection(), tableName);
374    // Merge 1st and 2nd region
375    PairOfSameType<HRegionInfo> mergedRegions = mergeRegionsAndVerifyRegionNum(MASTER, tableName,
376        0, 2, 5 * 2 - 2);
377    List<Pair<HRegionInfo, ServerName>> currentRegionToServers =
378        MetaTableAccessor.getTableRegionsAndLocations(
379            TEST_UTIL.getConnection(), tableName);
380    List<HRegionInfo> initialRegions = new ArrayList<HRegionInfo>();
381    for (Pair<HRegionInfo, ServerName> p : initialRegionToServers) {
382      initialRegions.add(p.getFirst());
383    }
384    List<HRegionInfo> currentRegions = new ArrayList<HRegionInfo>();
385    for (Pair<HRegionInfo, ServerName> p : currentRegionToServers) {
386      currentRegions.add(p.getFirst());
387    }
388    assertTrue(initialRegions.contains(mergedRegions.getFirst())); //this is the first region
389    assertTrue(initialRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(
390        mergedRegions.getFirst(), 1))); //this is the replica of the first region
391    assertTrue(initialRegions.contains(mergedRegions.getSecond())); //this is the second region
392    assertTrue(initialRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(
393        mergedRegions.getSecond(), 1))); //this is the replica of the second region
394    assertTrue(!initialRegions.contains(currentRegions.get(0))); //this is the new region
395    assertTrue(!initialRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(
396        currentRegions.get(0), 1))); //replica of the new region
397    assertTrue(currentRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(
398        currentRegions.get(0), 1))); //replica of the new region
399    assertTrue(!currentRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(
400        mergedRegions.getFirst(), 1))); //replica of the merged region
401    assertTrue(!currentRegions.contains(RegionReplicaUtil.getRegionInfoForReplica(
402        mergedRegions.getSecond(), 1))); //replica of the merged region
403  }

405  private PairOfSameType<HRegionInfo> mergeRegionsAndVerifyRegionNum(
406      HMaster master, TableName tablename,
407      int regionAnum, int regionBnum, int expectedRegionNum) throws Exception {
408    PairOfSameType<HRegionInfo> mergedRegions =
409      requestMergeRegion(master, tablename, regionAnum, regionBnum);
410    waitAndVerifyRegionNum(master, tablename, expectedRegionNum);
411    return mergedRegions;
412  }

414  private PairOfSameType<HRegionInfo> requestMergeRegion(
415      HMaster master, TableName tablename,
416      int regionAnum, int regionBnum) throws Exception {
417    List<Pair<HRegionInfo, ServerName>> tableRegions = MetaTableAccessor
418        .getTableRegionsAndLocations(
419            TEST_UTIL.getConnection(), tablename);
420    HRegionInfo regionA = tableRegions.get(regionAnum).getFirst();
421    HRegionInfo regionB = tableRegions.get(regionBnum).getFirst();
422    ADMIN.mergeRegions(
423      regionA.getEncodedNameAsBytes(),
424      regionB.getEncodedNameAsBytes(), false);
425    return new PairOfSameType<HRegionInfo>(regionA, regionB);
426  }

428  private void waitAndVerifyRegionNum(HMaster master, TableName tablename,
429      int expectedRegionNum) throws Exception {
430    List<Pair<HRegionInfo, ServerName>> tableRegionsInMeta;
431    List<HRegionInfo> tableRegionsInMaster;
432    long timeout = System.currentTimeMillis() + waitTime;
433    while (System.currentTimeMillis() < timeout) {
434      tableRegionsInMeta = MetaTableAccessor.getTableRegionsAndLocations(
435        TEST_UTIL.getConnection(), tablename);
436      tableRegionsInMaster = master.getAssignmentManager().getRegionStates()
437          .getRegionsOfTable(tablename);
438      if (tableRegionsInMeta.size() == expectedRegionNum
439          && tableRegionsInMaster.size() == expectedRegionNum) {
440        break;
441      }
442      Thread.sleep(250);
443    }
444
445    tableRegionsInMeta = MetaTableAccessor.getTableRegionsAndLocations(
446        TEST_UTIL.getConnection(), tablename);
447    LOG.info("Regions after merge:" + Joiner.on(',').join(tableRegionsInMeta));
448    assertEquals(expectedRegionNum, tableRegionsInMeta.size());
449  }

451  private Table createTableAndLoadData(HMaster master, TableName tablename)
452      throws Exception {
453    return createTableAndLoadData(master, tablename, INITIAL_REGION_NUM, 1);
454  }

456  private Table createTableAndLoadData(HMaster master, TableName tablename,
457      int numRegions, int replication) throws Exception {
458    assertTrue("ROWSIZE must > numregions:" + numRegions, ROWSIZE > numRegions);
459    byte[][] splitRows = new byte[numRegions - 1][];
460    for (int i = 0; i < splitRows.length; i++) {
461      splitRows[i] = ROWS[(i + 1) * ROWSIZE / numRegions];
462    }
463
464    Table table = TEST_UTIL.createTable(tablename, FAMILYNAME, splitRows);
465    LOG.info("Created " + table.getName());
466    if (replication > 1) {
467      HBaseTestingUtility.setReplicas(ADMIN, tablename, replication);
468      LOG.info("Set replication of " + replication + " on " + table.getName());
469    }
470    loadData(table);
471    LOG.info("Loaded " + table.getName());
472    verifyRowCount(table, ROWSIZE);
473    LOG.info("Verified " + table.getName());
474
475    // sleep here is an ugly hack to allow region transitions to finish
476    long timeout = System.currentTimeMillis() + waitTime;
477    List<Pair<HRegionInfo, ServerName>> tableRegions;
478    while (System.currentTimeMillis() < timeout) {
479      tableRegions = MetaTableAccessor.getTableRegionsAndLocations(
480          TEST_UTIL.getConnection(), tablename);
481      if (tableRegions.size() == numRegions * replication)
482        break;
483      Thread.sleep(250);
484    }
485    LOG.info("Getting regions of " + table.getName());
486    tableRegions = MetaTableAccessor.getTableRegionsAndLocations(
487        TEST_UTIL.getConnection(), tablename);
488    LOG.info("Regions after load: " + Joiner.on(',').join(tableRegions));
489    assertEquals(numRegions * replication, tableRegions.size());
490    return table;
491  }

493  private static byte[][] makeN(byte[] base, int n) {
494    byte[][] ret = new byte[n][];
495    for (int i = 0; i < n; i++) {
496      ret[i] = Bytes.add(base, Bytes.toBytes(String.format("%04d", i)));
497    }
498    return ret;
499  }

501  private void loadData(Table table) throws IOException {
502    for (int i = 0; i < ROWSIZE; i++) {
503      Put put = new Put(ROWS[i]);
504      put.addColumn(FAMILYNAME, QUALIFIER, Bytes.toBytes(i));
505      table.put(put);
506    }
507  }

509  private void verifyRowCount(Table table, int expectedRegionNum)
510      throws IOException {
511    ResultScanner scanner = table.getScanner(new Scan());
512    int rowCount = 0;
513    while (scanner.next() != null) {
514      rowCount++;
515    }
516    assertEquals(expectedRegionNum, rowCount);
517    scanner.close();
518  }

520  // Make it public so that JVMClusterUtil can access it.
521  public static class MyMaster extends HMaster {
522    public MyMaster(Configuration conf, CoordinatedStateManager cp)
523      throws IOException, KeeperException,
524        InterruptedException {
525      super(conf, cp);
526    }
527
528    @Override
529    protected RSRpcServices createRpcServices() throws IOException {
530      return new MyMasterRpcServices(this);
531    }
532  }
533
534  static class MyMasterRpcServices extends MasterRpcServices {
535    static AtomicBoolean enabled = new AtomicBoolean(false);
536
537    private HMaster myMaster;
538    public MyMasterRpcServices(HMaster master) throws IOException {
539      super(master);
540      myMaster = master;
541    }
542
543    @Override
544    public ReportRegionStateTransitionResponse reportRegionStateTransition(RpcController c,
545        ReportRegionStateTransitionRequest req) throws ServiceException {
546      ReportRegionStateTransitionResponse resp = super.reportRegionStateTransition(c, req);
547      if (enabled.get() && req.getTransition(0).getTransitionCode()
548          == TransitionCode.READY_TO_MERGE && !resp.hasErrorMessage()) {
549        RegionStates regionStates = myMaster.getAssignmentManager().getRegionStates();
550        for (RegionState regionState: regionStates.getRegionsInTransition()) {
551          // Find the merging_new region and remove it
552          if (regionState.isMergingNew()) {
553            regionStates.deleteRegion(regionState.getRegion());
554          }
555        }
556      }
557      return resp;
558    }
559  }
