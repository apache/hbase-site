053/**
054 * An implementation of {@link Table} that sits directly on a Region; it decorates the passed in
055 * Region instance with the Table API. Some API is not implemented yet (throws
056 * {@link UnsupportedOperationException}) mostly because no need as yet or it necessitates copying
057 * a load of code local from RegionServer.
058 * 
059 * <p>Use as an instance of a {@link Table} in-the-small -- no networking or servers
060 * necessary -- or to write a test that can run directly against the datastore and then
061 * over the network.
062 */
063public class RegionAsTable implements Table {
064  private final Region region;
065
066  /**
067   * @param region Region to decorate with Table API.
068   */
069  public RegionAsTable(final Region region) {
070    this.region = region;
071  }
072
073  @Override
074  public TableName getName() {
075    return this.region.getTableDesc().getTableName();
076  }
077
078  @Override
079  public Configuration getConfiguration() {
080    throw new UnsupportedOperationException();
081  }
082
083  @Override
084  public HTableDescriptor getTableDescriptor() throws IOException {
085    return this.region.getTableDesc();
086  }
087
088  @Override
089  public boolean exists(Get get) throws IOException {
090    if (!get.isCheckExistenceOnly()) throw new IllegalArgumentException();
091    return get(get) != null;
092  }
093
094  @Override
095  public boolean[] existsAll(List<Get> gets) throws IOException {
096    boolean [] results = new boolean[gets.size()];
097    int index = 0;
098    for (Get get: gets) {
099      results[index++] = exists(get);
100    }
101    return results;
102  }
103
104  @Override
105  public void batch(List<? extends Row> actions, Object[] results)
106  throws IOException, InterruptedException {
107    throw new UnsupportedOperationException();
108  }
109
110  @Override
111  public <R> void batchCallback(List<? extends Row> actions, Object[] results,
112      Callback<R> callback)
113  throws IOException, InterruptedException {
114    throw new UnsupportedOperationException();
115  }
116
117  @Override
118  public Result get(Get get) throws IOException {
119    return this.region.get(get);
120  }
121
122  @Override
123  public Result[] get(List<Get> gets) throws IOException {
124    Result [] results = new Result[gets.size()];
125    int index = 0;
126    for (Get get: gets) {
127      results[index++] = get(get);
128    }
129    return results;
130  }
131
132  static class RegionScannerToResultScannerAdaptor implements ResultScanner {
133    private static final Result [] EMPTY_RESULT_ARRAY = new Result[0];
134    private final RegionScanner regionScanner;
135
136    RegionScannerToResultScannerAdaptor(final RegionScanner regionScanner) {
137      this.regionScanner = regionScanner;
138    }
139
140    @Override
141    public Iterator<Result> iterator() {
142      throw new UnsupportedOperationException();
143    }
144
145    @Override
146    public Result next() throws IOException {
147      List<Cell> cells = new ArrayList<Cell>();
148      return regionScanner.next(cells)? Result.create(cells): null;
149    }
150
151    @Override
152    public Result[] next(int nbRows) throws IOException {
153      List<Result> results = new ArrayList<Result>(nbRows);
154      for (int i = 0; i < nbRows; i++) {
155        Result result = next();
156        if (result == null) break;
157        results.add(result);
158      }
159      return results.toArray(EMPTY_RESULT_ARRAY);
160    }
161
162    @Override
163    public void close() {
164      try {
165        regionScanner.close();
166      } catch (IOException e) {
167        throw new RuntimeException(e);
168      }
169    }
170
171    @Override
172    public boolean renewLease() {
173      throw new UnsupportedOperationException();
174    }
175  };
176
177  @Override
178  public ResultScanner getScanner(Scan scan) throws IOException {
179    return new RegionScannerToResultScannerAdaptor(this.region.getScanner(scan));
180  }
