079  @BeforeClass
080  public static void before() throws Exception {
081    // Reduce the hdfs block size and prefetch to trigger the file-link reopen
082    // when the file is moved to archive (e.g. compaction)
083    HTU.getConfiguration().setInt(DFSConfigKeys.DFS_BLOCK_SIZE_KEY, 8192);
084    HTU.getConfiguration().setInt(DFSConfigKeys.DFS_CLIENT_READ_PREFETCH_SIZE_KEY, 1);
085    HTU.getConfiguration().setInt(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 128 * 1024 * 1024);
086
087    HTU.startMiniCluster(NB_SERVERS);
088    final TableName tableName = TableName.valueOf(TestRegionReplicas.class.getSimpleName());
089
090    // Create table then get the single region for our new table.
091    table = HTU.createTable(tableName, f);
092
093    try (RegionLocator locator = HTU.getConnection().getRegionLocator(tableName)) {
094      hriPrimary = locator.getRegionLocation(row, false).getRegionInfo();
095    }
096
097    // mock a secondary region info to open
098    hriSecondary = new HRegionInfo(hriPrimary.getTable(), hriPrimary.getStartKey(),
099        hriPrimary.getEndKey(), hriPrimary.isSplit(), hriPrimary.getRegionId(), 1);
100
101    // No master
102    TestRegionServerNoMaster.stopMasterAndAssignMeta(HTU);
103  }

196  private void assertGet(Region region, int value, boolean expect) throws IOException {
197    byte[] row = Bytes.toBytes(String.valueOf(value));
198    Get get = new Get(row);
199    Result result = region.get(get);
200    if (expect) {
201      Assert.assertArrayEquals(row, result.getValue(f, null));
202    } else {
203      result.isEmpty();
204    }
205  }

207  // build a mock rpc
208  private void assertGetRpc(HRegionInfo info, int value, boolean expect)
209      throws IOException, org.apache.hadoop.hbase.shaded.com.google.protobuf.ServiceException {
210    byte[] row = Bytes.toBytes(String.valueOf(value));
211    Get get = new Get(row);
212    ClientProtos.GetRequest getReq = RequestConverter.buildGetRequest(info.getRegionName(), get);
213    ClientProtos.GetResponse getResp =  getRS().getRSRpcServices().get(null, getReq);
214    Result result = ProtobufUtil.toResult(getResp.getResult());
215    if (expect) {
216      Assert.assertArrayEquals(row, result.getValue(f, null));
217    } else {
218      result.isEmpty();
219    }
220  }

227  @Test(timeout = 300000)
228  public void testRefresStoreFiles() throws Exception {
229    // enable store file refreshing
230    final int refreshPeriod = 2000; // 2 sec
231    HTU.getConfiguration().setInt("hbase.hstore.compactionThreshold", 100);
232    HTU.getConfiguration().setInt(StorefileRefresherChore.REGIONSERVER_STOREFILE_REFRESH_PERIOD,
233      refreshPeriod);
234    // restart the region server so that it starts the refresher chore
235    restartRegionServer();
236
237    try {
238      LOG.info("Opening the secondary region " + hriSecondary.getEncodedName());
239      openRegion(HTU, getRS(), hriSecondary);
240
241      //load some data to primary
242      LOG.info("Loading data to primary region");
243      HTU.loadNumericRows(table, f, 0, 1000);
244      // assert that we can read back from primary
245      Assert.assertEquals(1000, HTU.countRows(table));
246      // flush so that region replica can read
247      LOG.info("Flushing primary region");
248      Region region = getRS().getRegionByEncodedName(hriPrimary.getEncodedName());
249      region.flush(true);
250      HRegion primaryRegion = (HRegion) region;
251
252      // ensure that chore is run
253      LOG.info("Sleeping for " + (4 * refreshPeriod));
254      Threads.sleep(4 * refreshPeriod);
255
256      LOG.info("Checking results from secondary region replica");
257      Region secondaryRegion = getRS().getFromOnlineRegions(hriSecondary.getEncodedName());
258      Assert.assertEquals(1, secondaryRegion.getStore(f).getStorefilesCount());
259
260      assertGet(secondaryRegion, 42, true);
261      assertGetRpc(hriSecondary, 42, true);
262      assertGetRpc(hriSecondary, 1042, false);
263
264      //load some data to primary
265      HTU.loadNumericRows(table, f, 1000, 1100);
266      region = getRS().getRegionByEncodedName(hriPrimary.getEncodedName());
267      region.flush(true);
268
269      HTU.loadNumericRows(table, f, 2000, 2100);
270      region = getRS().getRegionByEncodedName(hriPrimary.getEncodedName());
271      region.flush(true);
272
273      // ensure that chore is run
274      Threads.sleep(4 * refreshPeriod);
275
276      assertGetRpc(hriSecondary, 42, true);
277      assertGetRpc(hriSecondary, 1042, true);
278      assertGetRpc(hriSecondary, 2042, true);
279
280      // ensure that we see the 3 store files
281      Assert.assertEquals(3, secondaryRegion.getStore(f).getStorefilesCount());
282
283      // force compaction
284      HTU.compact(table.getName(), true);
285
286      long wakeUpTime = System.currentTimeMillis() + 4 * refreshPeriod;
287      while (System.currentTimeMillis() < wakeUpTime) {
288        assertGetRpc(hriSecondary, 42, true);
289        assertGetRpc(hriSecondary, 1042, true);
290        assertGetRpc(hriSecondary, 2042, true);
291        Threads.sleep(10);
292      }
293
294      // ensure that we see the compacted file only
295      // This will be 4 until the cleaner chore runs
296      Assert.assertEquals(4, secondaryRegion.getStore(f).getStorefilesCount());
297
298    } finally {
299      HTU.deleteNumericRows(table, HConstants.CATALOG_FAMILY, 0, 1000);
300      closeRegion(HTU, getRS(), hriSecondary);
301    }
302  }

304  @Ignore
305  @Test(timeout = 300000)
306  public void testFlushAndCompactionsInPrimary() throws Exception {
307
308    long runtime = 30 * 1000;
309    // enable store file refreshing
310    final int refreshPeriod = 100; // 100ms refresh is a lot
311    HTU.getConfiguration().setInt("hbase.hstore.compactionThreshold", 3);
312    HTU.getConfiguration().setInt(StorefileRefresherChore.REGIONSERVER_STOREFILE_REFRESH_PERIOD, refreshPeriod);
313    // restart the region server so that it starts the refresher chore
314    restartRegionServer();
315    final int startKey = 0, endKey = 1000;
316
317    try {
318      openRegion(HTU, getRS(), hriSecondary);
319
320      //load some data to primary so that reader won't fail
321      HTU.loadNumericRows(table, f, startKey, endKey);
322      TestRegionServerNoMaster.flushRegion(HTU, hriPrimary);
323      // ensure that chore is run
324      Threads.sleep(2 * refreshPeriod);
325
326      final AtomicBoolean running = new AtomicBoolean(true);
327      @SuppressWarnings("unchecked")
328      final AtomicReference<Exception>[] exceptions = new AtomicReference[3];
329      for (int i=0; i < exceptions.length; i++) {
330        exceptions[i] = new AtomicReference<>();
331      }
332
333      Runnable writer = new Runnable() {
334        int key = startKey;
335        @Override
336        public void run() {
337          try {
338            while (running.get()) {
339              byte[] data = Bytes.toBytes(String.valueOf(key));
340              Put put = new Put(data);
341              put.addColumn(f, null, data);
342              table.put(put);
343              key++;
344              if (key == endKey) key = startKey;
345            }
346          } catch (Exception ex) {
347            LOG.warn(ex);
348            exceptions[0].compareAndSet(null, ex);
349          }
350        }
351      };
352
353      Runnable flusherCompactor = new Runnable() {
354        Random random = new Random();
355        @Override
356        public void run() {
357          try {
358            while (running.get()) {
359              // flush or compact
360              if (random.nextBoolean()) {
361                TestRegionServerNoMaster.flushRegion(HTU, hriPrimary);
362              } else {
363                HTU.compact(table.getName(), random.nextBoolean());
364              }
365            }
366          } catch (Exception ex) {
367            LOG.warn(ex);
368            exceptions[1].compareAndSet(null, ex);
369          }
370        }
371      };
372
373      Runnable reader = new Runnable() {
374        Random random = new Random();
375        @Override
376        public void run() {
377          try {
378            while (running.get()) {
379              // whether to do a close and open
380              if (random.nextInt(10) == 0) {
381                try {
382                  closeRegion(HTU, getRS(), hriSecondary);
383                } catch (Exception ex) {
384                  LOG.warn("Failed closing the region " + hriSecondary + " "  + StringUtils.stringifyException(ex));
385                  exceptions[2].compareAndSet(null, ex);
386                }
387                try {
388                  openRegion(HTU, getRS(), hriSecondary);
389                } catch (Exception ex) {
390                  LOG.warn("Failed opening the region " + hriSecondary + " "  + StringUtils.stringifyException(ex));
391                  exceptions[2].compareAndSet(null, ex);
392                }
393              }
394
395              int key = random.nextInt(endKey - startKey) + startKey;
396              assertGetRpc(hriSecondary, key, true);
397            }
398          } catch (Exception ex) {
399            LOG.warn("Failed getting the value in the region " + hriSecondary + " "  + StringUtils.stringifyException(ex));
400            exceptions[2].compareAndSet(null, ex);
401          }
402        }
403      };
404
405      LOG.info("Starting writer and reader");
406      ExecutorService executor = Executors.newFixedThreadPool(3);
407      executor.submit(writer);
408      executor.submit(flusherCompactor);
409      executor.submit(reader);
410
411      // wait for threads
412      Threads.sleep(runtime);
413      running.set(false);
414      executor.shutdown();
415      executor.awaitTermination(30, TimeUnit.SECONDS);
416
417      for (AtomicReference<Exception> exRef : exceptions) {
418        Assert.assertNull(exRef.get());
419      }
420    } finally {
421      HTU.deleteNumericRows(table, HConstants.CATALOG_FAMILY, startKey, endKey);
422      closeRegion(HTU, getRS(), hriSecondary);
423    }
424  }

426  @Test(timeout = 300000)
427  public void testVerifySecondaryAbilityToReadWithOnFiles() throws Exception {
428    // disable the store file refresh chore (we do this by hand)
429    HTU.getConfiguration().setInt(StorefileRefresherChore.REGIONSERVER_STOREFILE_REFRESH_PERIOD, 0);
430    restartRegionServer();
431
432    try {
433      LOG.info("Opening the secondary region " + hriSecondary.getEncodedName());
434      openRegion(HTU, getRS(), hriSecondary);
435
436      // load some data to primary
437      LOG.info("Loading data to primary region");
438      for (int i = 0; i < 3; ++i) {
439        HTU.loadNumericRows(table, f, i * 1000, (i + 1) * 1000);
440        Region region = getRS().getRegionByEncodedName(hriPrimary.getEncodedName());
441        region.flush(true);
442      }
443
444      Region primaryRegion = getRS().getFromOnlineRegions(hriPrimary.getEncodedName());
445      Assert.assertEquals(3, primaryRegion.getStore(f).getStorefilesCount());
446
447      // Refresh store files on the secondary
448      Region secondaryRegion = getRS().getFromOnlineRegions(hriSecondary.getEncodedName());
449      secondaryRegion.getStore(f).refreshStoreFiles();
450      Assert.assertEquals(3, secondaryRegion.getStore(f).getStorefilesCount());
451
452      // force compaction
453      LOG.info("Force Major compaction on primary region " + hriPrimary);
454      primaryRegion.compact(true);
455      Assert.assertEquals(1, primaryRegion.getStore(f).getStorefilesCount());
456      List<RegionServerThread> regionServerThreads = HTU.getMiniHBaseCluster()
457          .getRegionServerThreads();
458      HRegionServer hrs = null;
459      for (RegionServerThread rs : regionServerThreads) {
460        if (rs.getRegionServer()
461            .getOnlineRegion(primaryRegion.getRegionInfo().getRegionName()) != null) {
462          hrs = rs.getRegionServer();
463          break;
464        }
465      }
466      CompactedHFilesDischarger cleaner =
467          new CompactedHFilesDischarger(100, null, hrs, false);
468      cleaner.chore();
469      // scan all the hfiles on the secondary.
470      // since there are no read on the secondary when we ask locations to
471      // the NN a FileNotFound exception will be returned and the FileLink
472      // should be able to deal with it giving us all the result we expect.
473      int keys = 0;
474      int sum = 0;
475      for (StoreFile sf: secondaryRegion.getStore(f).getStorefiles()) {
476        // Our file does not exist anymore. was moved by the compaction above.
477        LOG.debug(getRS().getFileSystem().exists(sf.getPath()));
478        Assert.assertFalse(getRS().getFileSystem().exists(sf.getPath()));
479
480        HFileScanner scanner = sf.getReader().getScanner(false, false);
481        scanner.seekTo();
482        do {
483          keys++;
484
485          Cell cell = scanner.getCell();
486          sum += Integer.parseInt(Bytes.toString(cell.getRowArray(),
487            cell.getRowOffset(), cell.getRowLength()));
488        } while (scanner.next());
489      }
490      Assert.assertEquals(3000, keys);
491      Assert.assertEquals(4498500, sum);
492    } finally {
493      HTU.deleteNumericRows(table, HConstants.CATALOG_FAMILY, 0, 1000);
494      closeRegion(HTU, getRS(), hriSecondary);
495    }
496  }
