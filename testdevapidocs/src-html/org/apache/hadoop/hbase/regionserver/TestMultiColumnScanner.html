061/**
062 * Tests optimized scanning of multiple columns.
063 */
064@RunWith(Parameterized.class)
065@Category({RegionServerTests.class, MediumTests.class})
066public class TestMultiColumnScanner {
067
068  private static final Log LOG = LogFactory.getLog(TestMultiColumnScanner.class);
069
070  private static final String TABLE_NAME =
071      TestMultiColumnScanner.class.getSimpleName();
072
073  static final int MAX_VERSIONS = 50;
074
075  private static final String FAMILY = "CF";
076  private static final byte[] FAMILY_BYTES = Bytes.toBytes(FAMILY);
077
078  /**
079   * The size of the column qualifier set used. Increasing this parameter
080   * exponentially increases test time.
081   */
082  private static final int NUM_COLUMNS = 8;
083
084  private static final int MAX_COLUMN_BIT_MASK = 1 << NUM_COLUMNS - 1;
085  private static final int NUM_FLUSHES = 10;
086  private static final int NUM_ROWS = 20;
087
088  /** A large value of type long for use as a timestamp */
089  private static final long BIG_LONG = 9111222333444555666L;
090
091  /**
092   * Timestamps to test with. Cannot use {@link Long#MAX_VALUE} here, because
093   * it will be replaced by an timestamp auto-generated based on the time.
094   */
095  private static final long[] TIMESTAMPS = new long[] { 1, 3, 5,
096      Integer.MAX_VALUE, BIG_LONG, Long.MAX_VALUE - 1 };
097
098  /** The probability that a column is skipped in a store file. */
099  private static final double COLUMN_SKIP_IN_STORE_FILE_PROB = 0.7;
100
101  /** The probability of skipping a column in a single row */
102  private static final double COLUMN_SKIP_IN_ROW_PROB = 0.1;
103
104  /** The probability of skipping a column everywhere */
105  private static final double COLUMN_SKIP_EVERYWHERE_PROB = 0.1;
106
107  /** The probability to delete a row/column pair */
108  private static final double DELETE_PROBABILITY = 0.02;
109
110  private final static HBaseTestingUtility TEST_UTIL = HBaseTestingUtility.createLocalHTU();
111
112  private final Compression.Algorithm comprAlgo;
113  private final BloomType bloomType;
114  private final DataBlockEncoding dataBlockEncoding;
115
116  // Some static sanity-checking.
117  static {
118    assertTrue(BIG_LONG > 0.9 * Long.MAX_VALUE); // Guard against typos.
119
120    // Ensure TIMESTAMPS are sorted.
121    for (int i = 0; i < TIMESTAMPS.length - 1; ++i)
122      assertTrue(TIMESTAMPS[i] < TIMESTAMPS[i + 1]);
123  }
124
125  @Parameters
126  public static final Collection<Object[]> parameters() {
127    List<Object[]> parameters = new ArrayList<>();
128    for (Object[] bloomAndCompressionParams :
129        HBaseTestingUtility.BLOOM_AND_COMPRESSION_COMBINATIONS) {
130      for (boolean useDataBlockEncoding : new boolean[]{false, true}) {
131        parameters.add(ArrayUtils.add(bloomAndCompressionParams,
132            useDataBlockEncoding));
133      }
134    }
135    return parameters;
136  }
137
138  public TestMultiColumnScanner(Compression.Algorithm comprAlgo,
139      BloomType bloomType, boolean useDataBlockEncoding) {
140    this.comprAlgo = comprAlgo;
141    this.bloomType = bloomType;
142    this.dataBlockEncoding = useDataBlockEncoding ? DataBlockEncoding.PREFIX :
143        DataBlockEncoding.NONE;
144  }
145
146  @Test
147  public void testMultiColumnScanner() throws IOException {
148    HRegion region = TEST_UTIL.createTestRegion(TABLE_NAME,
149        new HColumnDescriptor(FAMILY)
150            .setCompressionType(comprAlgo)
151            .setBloomFilterType(bloomType)
152            .setMaxVersions(MAX_VERSIONS)
153            .setDataBlockEncoding(dataBlockEncoding)
154    );
155    List<String> rows = sequentialStrings("row", NUM_ROWS);
156    List<String> qualifiers = sequentialStrings("qual", NUM_COLUMNS);
157    List<KeyValue> kvs = new ArrayList<>();
158    Set<String> keySet = new HashSet<>();
159
160    // A map from <row>_<qualifier> to the most recent delete timestamp for
161    // that column.
162    Map<String, Long> lastDelTimeMap = new HashMap<>();
163
164    Random rand = new Random(29372937L);
165    Set<String> rowQualSkip = new HashSet<>();
166
167    // Skip some columns in some rows. We need to test scanning over a set
168    // of columns when some of the columns are not there.
169    for (String row : rows)
170      for (String qual : qualifiers)
171        if (rand.nextDouble() < COLUMN_SKIP_IN_ROW_PROB) {
172          LOG.info("Skipping " + qual + " in row " + row);
173          rowQualSkip.add(rowQualKey(row, qual));
174        }
175
176    // Also skip some columns in all rows.
177    for (String qual : qualifiers)
178      if (rand.nextDouble() < COLUMN_SKIP_EVERYWHERE_PROB) {
179        LOG.info("Skipping " + qual + " in all rows");
180        for (String row : rows)
181          rowQualSkip.add(rowQualKey(row, qual));
182      }
183
184    for (int iFlush = 0; iFlush < NUM_FLUSHES; ++iFlush) {
185      for (String qual : qualifiers) {
186        // This is where we decide to include or not include this column into
187        // this store file, regardless of row and timestamp.
188        if (rand.nextDouble() < COLUMN_SKIP_IN_STORE_FILE_PROB)
189          continue;
190
191        byte[] qualBytes = Bytes.toBytes(qual);
192        for (String row : rows) {
193          Put p = new Put(Bytes.toBytes(row));
194          for (long ts : TIMESTAMPS) {
195            String value = createValue(row, qual, ts);
196            KeyValue kv = KeyValueTestUtil.create(row, FAMILY, qual, ts,
197                value);
198            assertEquals(kv.getTimestamp(), ts);
199            p.add(kv);
200            String keyAsString = kv.toString();
201            if (!keySet.contains(keyAsString)) {
202              keySet.add(keyAsString);
203              kvs.add(kv);
204            }
205          }
206          region.put(p);
207
208          Delete d = new Delete(Bytes.toBytes(row));
209          boolean deletedSomething = false;
210          for (long ts : TIMESTAMPS)
211            if (rand.nextDouble() < DELETE_PROBABILITY) {
212              d.addColumns(FAMILY_BYTES, qualBytes, ts);
213              String rowAndQual = row + "_" + qual;
214              Long whenDeleted = lastDelTimeMap.get(rowAndQual);
215              lastDelTimeMap.put(rowAndQual, whenDeleted == null ? ts
216                  : Math.max(ts, whenDeleted));
217              deletedSomething = true;
218            }
219          if (deletedSomething)
220            region.delete(d);
221        }
222      }
223      region.flush(true);
224    }
225
226    Collections.sort(kvs, CellComparatorImpl.COMPARATOR);
227    for (int maxVersions = 1; maxVersions <= TIMESTAMPS.length; ++maxVersions) {
228      for (int columnBitMask = 1; columnBitMask <= MAX_COLUMN_BIT_MASK; ++columnBitMask) {
229        Scan scan = new Scan();
230        scan.setMaxVersions(maxVersions);
231        Set<String> qualSet = new TreeSet<>();
232        {
233          int columnMaskTmp = columnBitMask;
234          for (String qual : qualifiers) {
235            if ((columnMaskTmp & 1) != 0) {
236              scan.addColumn(FAMILY_BYTES, Bytes.toBytes(qual));
237              qualSet.add(qual);
238            }
239            columnMaskTmp >>= 1;
240          }
241          assertEquals(0, columnMaskTmp);
242        }
243
244        InternalScanner scanner = region.getScanner(scan);
245        List<Cell> results = new ArrayList<>();
246
247        int kvPos = 0;
248        int numResults = 0;
249        String queryInfo = "columns queried: " + qualSet + " (columnBitMask="
250            + columnBitMask + "), maxVersions=" + maxVersions;
251
252        while (scanner.next(results) || results.size() > 0) {
253          for (Cell kv : results) {
254            while (kvPos < kvs.size()
255                && !matchesQuery(kvs.get(kvPos), qualSet, maxVersions,
256                    lastDelTimeMap)) {
257              ++kvPos;
258            }
259            String rowQual = getRowQualStr(kv);
260            String deleteInfo = "";
261            Long lastDelTS = lastDelTimeMap.get(rowQual);
262            if (lastDelTS != null) {
263              deleteInfo = "; last timestamp when row/column " + rowQual
264                  + " was deleted: " + lastDelTS;
265            }
266            assertTrue("Scanner returned additional key/value: " + kv + ", "
267                + queryInfo + deleteInfo + ";", kvPos < kvs.size());
268            assertTrue("Scanner returned wrong key/value; " + queryInfo
269                + deleteInfo + ";", CellUtil.equalsIgnoreMvccVersion(kvs.get(kvPos), (kv)));
270            ++kvPos;
271            ++numResults;
272          }
273          results.clear();
274        }
275        for (; kvPos < kvs.size(); ++kvPos) {
276          KeyValue remainingKV = kvs.get(kvPos);
277          assertFalse("Matching column not returned by scanner: "
278              + remainingKV + ", " + queryInfo + ", results returned: "
279              + numResults, matchesQuery(remainingKV, qualSet, maxVersions,
280              lastDelTimeMap));
281        }
282      }
283    }
284    assertTrue("This test is supposed to delete at least some row/column " +
285        "pairs", lastDelTimeMap.size() > 0);
286    LOG.info("Number of row/col pairs deleted at least once: " +
287       lastDelTimeMap.size());
288    HBaseTestingUtility.closeRegionAndWAL(region);
289  }
290
291  private static String getRowQualStr(Cell kv) {
292    String rowStr = Bytes.toString(CellUtil.cloneRow(kv));
293    String qualStr = Bytes.toString(CellUtil.cloneQualifier(kv));
294    return rowStr + "_" + qualStr;
295  }
296
297  private static boolean matchesQuery(KeyValue kv, Set<String> qualSet,
298      int maxVersions, Map<String, Long> lastDelTimeMap) {
299    Long lastDelTS = lastDelTimeMap.get(getRowQualStr(kv));
300    long ts = kv.getTimestamp();
301    return qualSet.contains(qualStr(kv))
302        && ts >= TIMESTAMPS[TIMESTAMPS.length - maxVersions]
303        && (lastDelTS == null || ts > lastDelTS);
304  }
305
306  private static String qualStr(KeyValue kv) {
307    return Bytes.toString(kv.getQualifierArray(), kv.getQualifierOffset(),
308        kv.getQualifierLength());
309  }
310
311  private static String rowQualKey(String row, String qual) {
312    return row + "_" + qual;
313  }
314
315  static String createValue(String row, String qual, long ts) {
316    return "value_for_" + row + "_" + qual + "_" + ts;
317  }
318
319  private static List<String> sequentialStrings(String prefix, int n) {
320    List<String> lst = new ArrayList<>();
321    for (int i = 0; i < n; ++i) {
322      StringBuilder sb = new StringBuilder();
323      sb.append(prefix + i);
324
325      // Make column length depend on i.
326      int iBitShifted = i;
327      while (iBitShifted != 0) {
328        sb.append((iBitShifted & 1) == 0 ? 'a' : 'b');
329        iBitShifted >>= 1;
330      }
331
332      lst.add(sb.toString());
333    }
334
335    return lst;
336  }
