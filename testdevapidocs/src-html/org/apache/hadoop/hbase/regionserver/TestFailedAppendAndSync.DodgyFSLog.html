105  // Dodgy WAL. Will throw exceptions when flags set.
106  class DodgyFSLog extends FSHLog {
107    volatile boolean throwSyncException = false;
108    volatile boolean throwAppendException = false;
109    final AtomicLong rolls = new AtomicLong(0);
110
111    public DodgyFSLog(FileSystem fs, Path root, String logDir, Configuration conf)
112        throws IOException {
113      super(fs, root, logDir, conf);
114    }
115
116    @Override
117    public byte[][] rollWriter(boolean force) throws FailedLogCloseException, IOException {
118      byte[][] regions = super.rollWriter(force);
119      rolls.getAndIncrement();
120      return regions;
121    }
122
123    @Override
124    protected Writer createWriterInstance(Path path) throws IOException {
125      final Writer w = super.createWriterInstance(path);
126      return new Writer() {
127        @Override
128        public void close() throws IOException {
129          w.close();
130        }
131
132        @Override
133        public void sync(boolean forceSync) throws IOException {
134          if (throwSyncException) {
135            throw new IOException("FAKE! Failed to replace a bad datanode...");
136          }
137          w.sync(forceSync);
138        }
139
140        @Override
141        public void append(Entry entry) throws IOException {
142          if (throwAppendException) {
143            throw new IOException("FAKE! Failed to replace a bad datanode...");
144          }
145          w.append(entry);
146        }
147
148        @Override
149        public long getLength() {
150          return w.getLength();
151        }
152      };
153    }

155  /**
156   * Reproduce locking up that happens when we get an exceptions appending and syncing.
157   * See HBASE-14317.
158   * First I need to set up some mocks for Server and RegionServerServices. I also need to
159   * set up a dodgy WAL that will throw an exception when we go to append to it.
160   */
161  @Test
162  public void testLockupAroundBadAssignSync() throws IOException {
163    // Make up mocked server and services.
164    Server server = mock(Server.class);
165    when(server.getConfiguration()).thenReturn(CONF);
166    when(server.isStopped()).thenReturn(false);
167    when(server.isAborted()).thenReturn(false);
168    RegionServerServices services = mock(RegionServerServices.class);
169    // OK. Now I have my mocked up Server and RegionServerServices and my dodgy WAL, go ahead with
170    // the test.
171    FileSystem fs = FileSystem.get(CONF);
172    Path rootDir = new Path(dir + getName());
173    DodgyFSLog dodgyWAL = new DodgyFSLog(fs, rootDir, getName(), CONF);
174    dodgyWAL.init();
175    LogRoller logRoller = new LogRoller(server, services);
176    logRoller.addWAL(dodgyWAL);
177    logRoller.start();
178
179    boolean threwOnSync = false;
180    boolean threwOnAppend = false;
181    boolean threwOnBoth = false;
182
183    HRegion region = initHRegion(tableName, null, null, dodgyWAL);
184    try {
185      // Get some random bytes.
186      byte[] value = Bytes.toBytes(getName());
187      try {
188        // First get something into memstore
189        Put put = new Put(value);
190        put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("1"), value);
191        region.put(put);
192      } catch (IOException ioe) {
193        fail();
194      }
195      long rollsCount = dodgyWAL.rolls.get();
196      try {
197        dodgyWAL.throwAppendException = true;
198        dodgyWAL.throwSyncException = false;
199        Put put = new Put(value);
200        put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("3"), value);
201        region.put(put);
202      } catch (IOException ioe) {
203        threwOnAppend = true;
204      }
205      while (rollsCount == dodgyWAL.rolls.get()) {
206        Threads.sleep(100);
207      }
208      rollsCount = dodgyWAL.rolls.get();
209
210      // When we get to here.. we should be ok. A new WAL has been put in place. There were no
211      // appends to sync. We should be able to continue.
212
213      try {
214        dodgyWAL.throwAppendException = true;
215        dodgyWAL.throwSyncException = true;
216        Put put = new Put(value);
217        put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("4"), value);
218        region.put(put);
219      } catch (IOException ioe) {
220        threwOnBoth = true;
221      }
222      while (rollsCount == dodgyWAL.rolls.get()) {
223        Threads.sleep(100);
224      }
225
226      // Again, all should be good. New WAL and no outstanding unsync'd edits so we should be able
227      // to just continue.
228
229      // So, should be no abort at this stage. Verify.
230      Mockito.verify(server, Mockito.atLeast(0)).abort(Mockito.anyString(),
231        Mockito.any(Throwable.class));
232      try {
233        dodgyWAL.throwAppendException = false;
234        dodgyWAL.throwSyncException = true;
235        Put put = new Put(value);
236        put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("2"), value);
237        region.put(put);
238      } catch (IOException ioe) {
239        threwOnSync = true;
240      }
241      // An append in the WAL but the sync failed is a server abort condition. That is our
242      // current semantic. Verify. It takes a while for abort to be called. Just hang here till it
243      // happens. If it don't we'll timeout the whole test. That is fine.
244      while (true) {
245        try {
246          Mockito.verify(server, Mockito.atLeast(1)).abort(Mockito.anyString(),
247            Mockito.any(Throwable.class));
248          break;
249        } catch (WantedButNotInvoked t) {
250          Threads.sleep(1);
251        }
252      }
253    } finally {
254      // To stop logRoller, its server has to say it is stopped.
255      Mockito.when(server.isStopped()).thenReturn(true);
256      if (logRoller != null) logRoller.close();
257      if (region != null) {
258        try {
259          region.close(true);
260        } catch (DroppedSnapshotException e) {
261          LOG.info("On way out; expected!", e);
262        }
263      }
264      if (dodgyWAL != null) dodgyWAL.close();
265      assertTrue("The regionserver should have thrown an exception", threwOnBoth);
266      assertTrue("The regionserver should have thrown an exception", threwOnAppend);
267      assertTrue("The regionserver should have thrown an exception", threwOnSync);
268    }
269  }
