158  private void requestSplitRegion(
159      final HRegionServer rsServer,
160      final Region region,
161      final byte[] midKey) throws IOException {
162    long procId = cluster.getMaster().splitRegion(region.getRegionInfo(), midKey, 0, 0);
163    // wait
164    if (procId != -1) {
165      // wait for the split to complete or get interrupted.  If the split completes successfully,
166      // the procedure will return true; if the split fails, the procedure would throw exception.
167      //
168      while (!rsServer.isProcedureFinished(procId)) {
169        try {
170          Thread.sleep(1000);
171        } catch (InterruptedException e) {
172          throw new IOException("Split region interrupted.");
173        }
174      }
175    } else {
176      throw new IOException ("Request split region failed.");
177    }
178  }

235
236    for (int i = 0; cluster.getRegions(tableName).isEmpty() && i < 100; i++) {
237      Thread.sleep(100);
238    }
239    assertEquals(1, cluster.getRegions(tableName).size());
240
241    HRegion region = cluster.getRegions(tableName).get(0);
242    Store store = region.getStore(cf);
243    int regionServerIndex = cluster.getServerWith(region.getRegionInfo().getRegionName());
244    HRegionServer regionServer = cluster.getRegionServer(regionServerIndex);
245
246    Table t = TESTING_UTIL.getConnection().getTable(tableName);
247    // insert data
248    insertData(tableName, admin, t);
249    insertData(tableName, admin, t);
250
251    int fileNum = store.getStorefiles().size();
252    // 0, Compaction Request
253    store.triggerMajorCompaction();
254    CompactionContext cc = store.requestCompaction();
255    assertNotNull(cc);
256    // 1, A timeout split
257    // 1.1 close region
258    assertEquals(2, region.close(false).get(cf).size());
259    // 1.2 rollback and Region initialize again
260    region.initialize();
261
262    // 2, Run Compaction cc
263    assertFalse(region.compact(cc, store, NoLimitThroughputController.INSTANCE));
264    assertTrue(fileNum > store.getStorefiles().size());
265
266    // 3, Split
267    requestSplitRegion(regionServer, region, Bytes.toBytes("row3"));
268    assertEquals(2, cluster.getRegions(tableName).size());

582      tableExists = MetaTableAccessor.tableExists(regionServer.getConnection(),
583          tableName);
584      assertEquals("The specified table should be present.", true, tableExists);
585      // exists works on stale and we see the put after the flush
586      byte[] b1 = "row1".getBytes();
587      Get g = new Get(b1);
588      g.setConsistency(Consistency.STRONG);
589      // The following GET will make a trip to the meta to get the new location of the 1st daughter
590      // In the process it will also get the location of the replica of the daughter (initially
591      // pointing to the parent's replica)
592      Result r = t.get(g);
593      Assert.assertFalse(r.isStale());
594      LOG.info("exists stale after flush done");
595
596      SlowMeCopro.getCdl().set(new CountDownLatch(1));
597      g = new Get(b1);
598      g.setConsistency(Consistency.TIMELINE);
599      // This will succeed because in the previous GET we get the location of the replica
600      r = t.get(g);
601      Assert.assertTrue(r.isStale());
602      SlowMeCopro.getCdl().get().countDown();
603    } finally {
604      SlowMeCopro.getCdl().get().countDown();
605      admin.setBalancerRunning(true, false);
606      cluster.getMaster().setCatalogJanitorEnabled(true);
607      t.close();
608    }
609  }

672      // store files

676      // check dirs

679          FSUtils.getTableStoreFilePathMap(null, fs, rootDir, tableName);

683      hri = region.getRegionInfo(); // split parent
684      AssignmentManager am = cluster.getMaster().getAssignmentManager();

792  /**
793   * Ensure single table region is not on same server as the single hbase:meta table
794   * region.
795   * @param admin
796   * @param hri
797   * @return Index of the server hosting the single table region
798   * @throws UnknownRegionException
799   * @throws MasterNotRunningException
800   * @throws org.apache.hadoop.hbase.ZooKeeperConnectionException
801   * @throws InterruptedException
802   */
803  private int ensureTableRegionNotOnSameServerAsMeta(final Admin admin,
804      final HRegionInfo hri)
805  throws IOException, MasterNotRunningException,
806  ZooKeeperConnectionException, InterruptedException {
807    // Now make sure that the table region is not on same server as that hosting
808    // hbase:meta  We don't want hbase:meta replay polluting our test when we later crash
809    // the table region serving server.
810    int metaServerIndex = cluster.getServerWithMeta();
811    assertTrue(metaServerIndex == -1); // meta is on master now
812    HRegionServer metaRegionServer = cluster.getMaster();
813    int tableRegionIndex = cluster.getServerWith(hri.getRegionName());
814    assertTrue(tableRegionIndex != -1);
815    HRegionServer tableRegionServer = cluster.getRegionServer(tableRegionIndex);
816    if (metaRegionServer.getServerName().equals(tableRegionServer.getServerName())) {
817      HRegionServer hrs = getOtherRegionServer(cluster, metaRegionServer);
818      assertNotNull(hrs);
819      assertNotNull(hri);
820      LOG.info("Moving " + hri.getRegionNameAsString() + " from " +
821        metaRegionServer.getServerName() + " to " +
822        hrs.getServerName() + "; metaServerIndex=" + metaServerIndex);
823      admin.move(hri.getEncodedNameAsBytes(), Bytes.toBytes(hrs.getServerName().toString()));
824    }
825    // Wait till table region is up on the server that is NOT carrying hbase:meta.
826    for (int i = 0; i < 100; i++) {
827      tableRegionIndex = cluster.getServerWith(hri.getRegionName());
828      if (tableRegionIndex != -1 && tableRegionIndex != metaServerIndex) break;
829      LOG.debug("Waiting on region move off the hbase:meta server; current index " +
830        tableRegionIndex + " and metaServerIndex=" + metaServerIndex);
831      Thread.sleep(100);
832    }
833    assertTrue("Region not moved off hbase:meta server", tableRegionIndex != -1
834        && tableRegionIndex != metaServerIndex);
835    // Verify for sure table region is not on same server as hbase:meta
836    tableRegionIndex = cluster.getServerWith(hri.getRegionName());
837    assertTrue(tableRegionIndex != -1);
838    assertNotSame(metaServerIndex, tableRegionIndex);
839    return tableRegionIndex;
840  }

842  /**
843   * Find regionserver other than the one passed.
844   * Can't rely on indexes into list of regionservers since crashed servers
845   * occupy an index.
846   * @param cluster
847   * @param notThisOne
848   * @return A regionserver that is not <code>notThisOne</code> or null if none
849   * found
850   */
851  private HRegionServer getOtherRegionServer(final MiniHBaseCluster cluster,
852      final HRegionServer notThisOne) {
853    for (RegionServerThread rst: cluster.getRegionServerThreads()) {
854      HRegionServer hrs = rst.getRegionServer();
855      if (hrs.getServerName().equals(notThisOne.getServerName())) continue;
856      if (hrs.isStopping() || hrs.isStopped()) continue;
857      return hrs;
858    }
859    return null;
860  }

926  static class MyMasterRpcServices extends MasterRpcServices {
927    static AtomicBoolean enabled = new AtomicBoolean(false);
928
929    private HMaster myMaster;
930    public MyMasterRpcServices(HMaster master) throws IOException {
931      super(master);
932      myMaster = master;
933    }
934
935    @Override
936    public ReportRegionStateTransitionResponse reportRegionStateTransition(RpcController c,
937        ReportRegionStateTransitionRequest req) throws ServiceException {
938      ReportRegionStateTransitionResponse resp = super.reportRegionStateTransition(c, req);
939      if (enabled.get() && req.getTransition(0).getTransitionCode().equals(
940          TransitionCode.READY_TO_SPLIT) && !resp.hasErrorMessage()) {
941        RegionStates regionStates = myMaster.getAssignmentManager().getRegionStates();
942        for (RegionState regionState: regionStates.getRegionsInTransition()) {
943          // Find the merging_new region and remove it
944          if (regionState.isSplittingNew()) {
945            regionStates.deleteRegion(regionState.getRegion());
946          }
947        }
948      }
949      return resp;
950    }
