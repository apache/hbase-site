150    // write data to one memstore and two store files
151    FileSystem fs = TEST_UTIL.getTestFileSystem();
152    Path hfilePath = new Path(new Path(
153        TEST_UTIL.getDataTestDir("testReversibleKeyValueHeap"), "regionname"),
154        "familyname");

156    HFileContextBuilder hcBuilder = new HFileContextBuilder();
157    hcBuilder.withBlockSize(2 * 1024);
158    HFileContext hFileContext = hcBuilder.build();
159    StoreFileWriter writer1 = new StoreFileWriter.Builder(
160        TEST_UTIL.getConfiguration(), cacheConf, fs).withOutputDir(
161        hfilePath).withFileContext(hFileContext).build();
162    StoreFileWriter writer2 = new StoreFileWriter.Builder(
163        TEST_UTIL.getConfiguration(), cacheConf, fs).withOutputDir(
164        hfilePath).withFileContext(hFileContext).build();
165
166    MemStore memstore = new DefaultMemStore();
167    writeMemstoreAndStoreFiles(memstore, new StoreFileWriter[] { writer1,
168        writer2 });
169
170    StoreFile sf1 = new StoreFile(fs, writer1.getPath(), TEST_UTIL.getConfiguration(), cacheConf,
171        BloomType.NONE, true);
172
173    StoreFile sf2 = new StoreFile(fs, writer2.getPath(), TEST_UTIL.getConfiguration(), cacheConf,
174        BloomType.NONE, true);
175    /**
176     * Test without MVCC
177     */
178    int startRowNum = ROWSIZE / 2;
179    ReversedKeyValueHeap kvHeap = getReversibleKeyValueHeap(memstore, sf1, sf2,
180        ROWS[startRowNum], MAXMVCC);
181    internalTestSeekAndNextForReversibleKeyValueHeap(kvHeap, startRowNum);
182
183    startRowNum = ROWSIZE - 1;
184    kvHeap = getReversibleKeyValueHeap(memstore, sf1, sf2,
185        HConstants.EMPTY_START_ROW, MAXMVCC);
186    internalTestSeekAndNextForReversibleKeyValueHeap(kvHeap, startRowNum);
187
188    /**
189     * Test with MVCC
190     */
191    for (int readPoint = 0; readPoint < MAXMVCC; readPoint++) {
192      LOG.info("Setting read point to " + readPoint);
193      startRowNum = ROWSIZE - 1;
194      kvHeap = getReversibleKeyValueHeap(memstore, sf1, sf2,
195          HConstants.EMPTY_START_ROW, readPoint);
196      for (int i = startRowNum; i >= 0; i--) {
197        if (i - 2 < 0) break;
198        i = i - 2;
199        kvHeap.seekToPreviousRow(KeyValueUtil.createFirstOnRow(ROWS[i + 1]));
200        Pair<Integer, Integer> nextReadableNum = getNextReadableNumWithBackwardScan(
201            i, 0, readPoint);
202        if (nextReadableNum == null) break;
203        KeyValue expecedKey = makeKV(nextReadableNum.getFirst(),
204            nextReadableNum.getSecond());
205        assertEquals(expecedKey, kvHeap.peek());
206        i = nextReadableNum.getFirst();
207        int qualNum = nextReadableNum.getSecond();
208        if (qualNum + 1 < QUALSIZE) {
209          kvHeap.backwardSeek(makeKV(i, qualNum + 1));
210          nextReadableNum = getNextReadableNumWithBackwardScan(i, qualNum + 1,
211              readPoint);
212          if (nextReadableNum == null) break;
213          expecedKey = makeKV(nextReadableNum.getFirst(),
214              nextReadableNum.getSecond());
215          assertEquals(expecedKey, kvHeap.peek());
216          i = nextReadableNum.getFirst();
217          qualNum = nextReadableNum.getSecond();
218        }
219
220        kvHeap.next();
221
222        if (qualNum + 1 >= QUALSIZE) {
223          nextReadableNum = getNextReadableNumWithBackwardScan(i - 1, 0,
224              readPoint);
225        } else {
226          nextReadableNum = getNextReadableNumWithBackwardScan(i, qualNum + 1,
227              readPoint);
228        }
229        if (nextReadableNum == null) break;
230        expecedKey = makeKV(nextReadableNum.getFirst(),
231            nextReadableNum.getSecond());
232        assertEquals(expecedKey, kvHeap.peek());
233        i = nextReadableNum.getFirst();
234      }
235    }

240    // write data to one memstore and two store files
241    FileSystem fs = TEST_UTIL.getTestFileSystem();
242    Path hfilePath = new Path(new Path(
243        TEST_UTIL.getDataTestDir("testReversibleStoreScanner"), "regionname"),

249    StoreFileWriter writer1 = new StoreFileWriter.Builder(

251        hfilePath).withFileContext(hFileContext).build();
252    StoreFileWriter writer2 = new StoreFileWriter.Builder(
253        TEST_UTIL.getConfiguration(), cacheConf, fs).withOutputDir(
254        hfilePath).withFileContext(hFileContext).build();
255
256    MemStore memstore = new DefaultMemStore();
257    writeMemstoreAndStoreFiles(memstore, new StoreFileWriter[] { writer1,
258        writer2 });
259
260    StoreFile sf1 = new StoreFile(fs, writer1.getPath(), TEST_UTIL.getConfiguration(), cacheConf,
261        BloomType.NONE, true);
262
263    StoreFile sf2 = new StoreFile(fs, writer2.getPath(), TEST_UTIL.getConfiguration(), cacheConf,
264        BloomType.NONE, true);
265
266    ScanType scanType = ScanType.USER_SCAN;
267    ScanInfo scanInfo =
268        new ScanInfo(TEST_UTIL.getConfiguration(), FAMILYNAME, 0, Integer.MAX_VALUE, Long.MAX_VALUE,
269            KeepDeletedCells.FALSE, HConstants.DEFAULT_BLOCKSIZE, 0, CellComparator.COMPARATOR);
270

272    Scan scan = new Scan();
273    scan.setReversed(true);
274    StoreScanner storeScanner = getReversibleStoreScanner(memstore, sf1, sf2,
275        scan, scanType, scanInfo, MAXMVCC);
276    verifyCountAndOrder(storeScanner, QUALSIZE * ROWSIZE, ROWSIZE, false);
277
278    // Case 2.Test reversed scan with a specified start row
279    int startRowNum = ROWSIZE / 2;
280    byte[] startRow = ROWS[startRowNum];
281    scan.withStartRow(startRow);
282    storeScanner = getReversibleStoreScanner(memstore, sf1, sf2, scan,
283        scanType, scanInfo, MAXMVCC);
284    verifyCountAndOrder(storeScanner, QUALSIZE * (startRowNum + 1),
285        startRowNum + 1, false);
286
287    // Case 3.Test reversed scan with a specified start row and specified
288    // qualifiers
289    assertTrue(QUALSIZE > 2);
290    scan.addColumn(FAMILYNAME, QUALS[0]);
291    scan.addColumn(FAMILYNAME, QUALS[2]);
292    storeScanner = getReversibleStoreScanner(memstore, sf1, sf2, scan,

295        false);
296

298    for (int readPoint = 0; readPoint < MAXMVCC; readPoint++) {
299      LOG.info("Setting read point to " + readPoint);
300      storeScanner = getReversibleStoreScanner(memstore, sf1, sf2, scan,
301          scanType, scanInfo, readPoint);
302      int expectedRowCount = 0;
303      int expectedKVCount = 0;
304      for (int i = startRowNum; i >= 0; i--) {
305        int kvCount = 0;
306        if (makeMVCC(i, 0) <= readPoint) {
307          kvCount++;
308        }
309        if (makeMVCC(i, 2) <= readPoint) {
310          kvCount++;
311        }
312        if (kvCount > 0) {
313          expectedRowCount++;
314          expectedKVCount += kvCount;
315        }
316      }
317      verifyCountAndOrder(storeScanner, expectedKVCount, expectedRowCount,
318          false);
319    }

349    byte[][] specifiedQualifiers = { QUALS[1], QUALS[2] };
350    for (byte[] specifiedQualifier : specifiedQualifiers)
351      scan.addColumn(FAMILYNAME, specifiedQualifier);
352    scanner = region.getScanner(scan);
353    verifyCountAndOrder(scanner, ROWSIZE * 2, ROWSIZE, false);
354
355    // Case4:Specify qualifiers + Two families
356    for (byte[] specifiedQualifier : specifiedQualifiers)
357      scan.addColumn(FAMILYNAME2, specifiedQualifier);
358    scanner = region.getScanner(scan);
359    verifyCountAndOrder(scanner, ROWSIZE * 2 * 2, ROWSIZE, false);
360
361    // Case5: Case4 + specify start row
362    int startRowNum = ROWSIZE * 3 / 4;
363    scan.withStartRow(ROWS[startRowNum]);
364    scanner = region.getScanner(scan);
365    verifyCountAndOrder(scanner, (startRowNum + 1) * 2 * 2, (startRowNum + 1),
366        false);
367
368    // Case6: Case4 + specify stop row
369    int stopRowNum = ROWSIZE / 4;
370    scan.withStartRow(HConstants.EMPTY_BYTE_ARRAY);
371    scan.withStopRow(ROWS[stopRowNum]);
372    scanner = region.getScanner(scan);
373    verifyCountAndOrder(scanner, (ROWSIZE - stopRowNum - 1) * 2 * 2, (ROWSIZE
374        - stopRowNum - 1), false);
375
376    // Case7: Case4 + specify start row + specify stop row
377    scan.withStartRow(ROWS[startRowNum]);
378    scanner = region.getScanner(scan);
379    verifyCountAndOrder(scanner, (startRowNum - stopRowNum) * 2 * 2,
380        (startRowNum - stopRowNum), false);
381
382    // Case8: Case7 + SingleColumnValueFilter
383    int valueNum = startRowNum % VALUESIZE;
384    Filter filter = new SingleColumnValueFilter(FAMILYNAME,
385        specifiedQualifiers[0], CompareOp.EQUAL, VALUES[valueNum]);
386    scan.setFilter(filter);
387    scanner = region.getScanner(scan);
388    int unfilteredRowNum = (startRowNum - stopRowNum) / VALUESIZE
389        + (stopRowNum / VALUESIZE == valueNum ? 0 : 1);
390    verifyCountAndOrder(scanner, unfilteredRowNum * 2 * 2, unfilteredRowNum,
391        false);
392
393    // Case9: Case7 + PageFilter
394    int pageSize = 10;
395    filter = new PageFilter(pageSize);
396    scan.setFilter(filter);
397    scanner = region.getScanner(scan);
398    int expectedRowNum = pageSize;
399    verifyCountAndOrder(scanner, expectedRowNum * 2 * 2, expectedRowNum, false);
400
401    // Case10: Case7 + FilterList+MUST_PASS_ONE
402    SingleColumnValueFilter scvFilter1 = new SingleColumnValueFilter(
403        FAMILYNAME, specifiedQualifiers[0], CompareOp.EQUAL, VALUES[0]);
404    SingleColumnValueFilter scvFilter2 = new SingleColumnValueFilter(
405        FAMILYNAME, specifiedQualifiers[0], CompareOp.EQUAL, VALUES[1]);
406    expectedRowNum = 0;
407    for (int i = startRowNum; i > stopRowNum; i--) {
408      if (i % VALUESIZE == 0 || i % VALUESIZE == 1) {
409        expectedRowNum++;
410      }
411    }
412    filter = new FilterList(Operator.MUST_PASS_ONE, scvFilter1, scvFilter2);
413    scan.setFilter(filter);
414    scanner = region.getScanner(scan);
415    verifyCountAndOrder(scanner, expectedRowNum * 2 * 2, expectedRowNum, false);
416
417    // Case10: Case7 + FilterList+MUST_PASS_ALL
418    filter = new FilterList(Operator.MUST_PASS_ALL, scvFilter1, scvFilter2);
419    expectedRowNum = 0;
420    scan.setFilter(filter);
421    scanner = region.getScanner(scan);
422    verifyCountAndOrder(scanner, expectedRowNum * 2 * 2, expectedRowNum, false);
423  }

426      StoreFile sf1, StoreFile sf2, Scan scan, ScanType scanType,
427      ScanInfo scanInfo, int readPoint) throws IOException {
428    List<KeyValueScanner> scanners = getScanners(memstore, sf1, sf2, null,
429        false, readPoint);

437        scanType, columns, scanners);
438    return storeScanner;
439  }

441  private void verifyCountAndOrder(InternalScanner scanner,
442      int expectedKVCount, int expectedRowCount, boolean forward)
443      throws IOException {
444    List<Cell> kvList = new ArrayList<>();
445    Result lastResult = null;
446    int rowCount = 0;
447    int kvCount = 0;
448    try {
449      while (scanner.next(kvList)) {
450        if (kvList.isEmpty()) continue;
451        rowCount++;
452        kvCount += kvList.size();
453        if (lastResult != null) {
454          Result curResult = Result.create(kvList);
455          assertEquals("LastResult:" + lastResult + "CurResult:" + curResult,
456              forward,
457              Bytes.compareTo(curResult.getRow(), lastResult.getRow()) > 0);
458        }
459        lastResult = Result.create(kvList);
460        kvList.clear();
461      }
462    } finally {
463      scanner.close();
464    }
465    if (!kvList.isEmpty()) {
466      rowCount++;
467      kvCount += kvList.size();
468      kvList.clear();
469    }
470    assertEquals(expectedKVCount, kvCount);
471    assertEquals(expectedRowCount, rowCount);
472  }

474  private void internalTestSeekAndNextForReversibleKeyValueHeap(
475      ReversedKeyValueHeap kvHeap, int startRowNum) throws IOException {
476    // Test next and seek
477    for (int i = startRowNum; i >= 0; i--) {
478      if (i % 2 == 1 && i - 2 >= 0) {
479        i = i - 2;
480        kvHeap.seekToPreviousRow(KeyValueUtil.createFirstOnRow(ROWS[i + 1]));
481      }
482      for (int j = 0; j < QUALSIZE; j++) {
483        if (j % 2 == 1 && (j + 1) < QUALSIZE) {
484          j = j + 1;
485          kvHeap.backwardSeek(makeKV(i, j));
486        }
487        assertEquals(makeKV(i, j), kvHeap.peek());
488        kvHeap.next();
489      }
490    }
491    assertEquals(null, kvHeap.peek());
492  }

504  private List<KeyValueScanner> getScanners(MemStore memstore, StoreFile sf1,
505      StoreFile sf2, byte[] startRow, boolean doSeek, int readPoint)
506      throws IOException {
507    List<StoreFileScanner> fileScanners = StoreFileScanner
508        .getScannersForStoreFiles(Lists.newArrayList(sf1, sf2), false, true,
509            false, false, readPoint);
510    List<KeyValueScanner> memScanners = memstore.getScanners(readPoint);
511    List<KeyValueScanner> scanners = new ArrayList<>(fileScanners.size() + 1);
512    scanners.addAll(fileScanners);
513    scanners.addAll(memScanners);
514
515    if (doSeek) {
516      if (Bytes.equals(HConstants.EMPTY_START_ROW, startRow)) {
517        for (KeyValueScanner scanner : scanners) {
518          scanner.seekToLastRow();
519        }
520      } else {
521        KeyValue startKey = KeyValueUtil.createFirstOnRow(startRow);
522        for (KeyValueScanner scanner : scanners) {
523          scanner.backwardSeek(startKey);
524        }
525      }
526    }
527    return scanners;
528  }

530  private void seekTestOfReversibleKeyValueScanner(KeyValueScanner scanner)
531      throws IOException {
532    /**
533     * Test without MVCC
534     */
535    // Test seek to last row
536    assertTrue(scanner.seekToLastRow());
537    assertEquals(makeKV(ROWSIZE - 1, 0), scanner.peek());
538
539    // Test backward seek in three cases
540    // Case1: seek in the same row in backwardSeek
541    KeyValue seekKey = makeKV(ROWSIZE - 2, QUALSIZE - 2);
542    assertTrue(scanner.backwardSeek(seekKey));
543    assertEquals(seekKey, scanner.peek());
544
545    // Case2: seek to the previous row in backwardSeek
546    int seekRowNum = ROWSIZE - 2;
547    assertTrue(scanner.backwardSeek(KeyValueUtil.createLastOnRow(ROWS[seekRowNum])));
548    KeyValue expectedKey = makeKV(seekRowNum - 1, 0);
549    assertEquals(expectedKey, scanner.peek());
550
551    // Case3: unable to backward seek
552    assertFalse(scanner.backwardSeek(KeyValueUtil.createLastOnRow(ROWS[0])));
553    assertEquals(null, scanner.peek());
554
555    // Test seek to previous row
556    seekRowNum = ROWSIZE - 4;
557    assertTrue(scanner.seekToPreviousRow(KeyValueUtil
558        .createFirstOnRow(ROWS[seekRowNum])));
559    expectedKey = makeKV(seekRowNum - 1, 0);
560    assertEquals(expectedKey, scanner.peek());
561
562    // Test seek to previous row for the first row
563    assertFalse(scanner.seekToPreviousRow(makeKV(0, 0)));
564    assertEquals(null, scanner.peek());
565
566  }

568  private void seekTestOfReversibleKeyValueScannerWithMVCC(
569      List<? extends KeyValueScanner> scanners, int readPoint) throws IOException {
570  /**
571   * Test with MVCC
572   */
573    // Test seek to last row
574    KeyValue expectedKey = getNextReadableKeyValueWithBackwardScan(
575        ROWSIZE - 1, 0, readPoint);
576    boolean res = false;
577    for (KeyValueScanner scanner : scanners) {
578      res |= scanner.seekToLastRow();
579    }
580    assertEquals(expectedKey != null, res);
581    res = false;
582    for (KeyValueScanner scanner : scanners) {
583      res |= (expectedKey.equals(scanner.peek()));
584    }
585    assertTrue(res);
586
587      // Test backward seek in two cases
588      // Case1: seek in the same row in backwardSeek
589      expectedKey = getNextReadableKeyValueWithBackwardScan(ROWSIZE - 2,
590          QUALSIZE - 2, readPoint);
591    res = false;
592    for (KeyValueScanner scanner : scanners) {
593      res |= scanner.backwardSeek(expectedKey);
594    }
595    assertEquals(expectedKey != null, res);
596    res = false;
597    for (KeyValueScanner scanner : scanners) {
598      res |= (expectedKey.equals(scanner.peek()));
599    }
600    assertTrue(res);
601
602      // Case2: seek to the previous row in backwardSeek
603    int seekRowNum = ROWSIZE - 3;
604    res = false;
605    for (KeyValueScanner scanner : scanners) {
606      res |= scanner.backwardSeek(expectedKey);
607    }
608    res = false;
609    for (KeyValueScanner scanner : scanners) {
610      res |= (expectedKey.equals(scanner.peek()));
611    }
612    assertTrue(res);
613
614      // Test seek to previous row
615      seekRowNum = ROWSIZE - 4;
616      expectedKey = getNextReadableKeyValueWithBackwardScan(seekRowNum - 1, 0,
617          readPoint);
618    res = false;
619    for (KeyValueScanner scanner : scanners) {
620      res |= scanner.seekToPreviousRow(KeyValueUtil.createFirstOnRow(ROWS[seekRowNum]));
621    }
622    assertEquals(expectedKey != null, res);
623    res = false;
624    for (KeyValueScanner scanner : scanners) {
625      res |= (expectedKey.equals(scanner.peek()));
626    }
627    assertTrue(res);
628  }

630  private KeyValue getNextReadableKeyValueWithBackwardScan(int startRowNum,
631      int startQualNum, int readPoint) {
632    Pair<Integer, Integer> nextReadableNum = getNextReadableNumWithBackwardScan(
633        startRowNum, startQualNum, readPoint);
634    if (nextReadableNum == null)
635      return null;
636    return makeKV(nextReadableNum.getFirst(), nextReadableNum.getSecond());
637  }
638
639  private Pair<Integer, Integer> getNextReadableNumWithBackwardScan(
640      int startRowNum, int startQualNum, int readPoint) {
641    Pair<Integer, Integer> nextReadableNum = null;
642    boolean findExpected = false;
643    for (int i = startRowNum; i >= 0; i--) {
644      for (int j = (i == startRowNum ? startQualNum : 0); j < QUALSIZE; j++) {
645        if (makeMVCC(i, j) <= readPoint) {
646          nextReadableNum = new Pair<>(i, j);
647          findExpected = true;
648          break;
649        }
650      }
651      if (findExpected)
652        break;
653    }
654    return nextReadableNum;
655  }

657  private static void loadDataToRegion(Region region, byte[] additionalFamily)
658      throws IOException {
659    for (int i = 0; i < ROWSIZE; i++) {
660      Put put = new Put(ROWS[i]);
661      for (int j = 0; j < QUALSIZE; j++) {
662        put.add(makeKV(i, j));
663        // put additional family
664        put.add(makeKV(i, j, additionalFamily));
665      }
666      region.put(put);
667      if (i == ROWSIZE / 3 || i == ROWSIZE * 2 / 3) {
668        region.flush(true);
669      }
670    }
671  }

673  private static void writeMemstoreAndStoreFiles(MemStore memstore,
674      final StoreFileWriter[] writers) throws IOException {
675    try {
676      for (int i = 0; i < ROWSIZE; i++) {
677        for (int j = 0; j < QUALSIZE; j++) {
678          if (i % 2 == 0) {
679            memstore.add(makeKV(i, j), null);
680          } else {
681            writers[(i + j) % writers.length].append(makeKV(i, j));
682          }
683        }
684      }
685    } finally {
686      for (int i = 0; i < writers.length; i++) {
687        writers[i].close();
688      }
689    }
690  }

696        for (int j = 0; j < QUALSIZE; j++) {

705  private static void writeMemstore(MemStore memstore) throws IOException {
706    // Add half of the keyvalues to memstore
707    for (int i = 0; i < ROWSIZE; i++) {
708      for (int j = 0; j < QUALSIZE; j++) {
709        if ((i + j) % 2 == 0) {
710          memstore.add(makeKV(i, j), null);
711        }
712      }
713    }
714    memstore.snapshot();
715    // Add another half of the keyvalues to snapshot
716    for (int i = 0; i < ROWSIZE; i++) {
717      for (int j = 0; j < QUALSIZE; j++) {
718        if ((i + j) % 2 == 1) {
719          memstore.add(makeKV(i, j), null);
720        }
721      }
722    }
723  }
