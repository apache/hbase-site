092  @Test
093  public void testCompactedHFilesCleaner() throws Exception {
094    // Create the cleaner object
095    CompactedHFilesDischarger cleaner =
096        new CompactedHFilesDischarger(1000, (Stoppable) null, rss, false);
097    // Add some data to the region and do some flushes
098    for (int i = 1; i < 10; i++) {
099      Put p = new Put(Bytes.toBytes("row" + i));
100      p.addColumn(fam, qual1, val);
101      region.put(p);
102    }
103    // flush them
104    region.flush(true);
105    for (int i = 11; i < 20; i++) {
106      Put p = new Put(Bytes.toBytes("row" + i));
107      p.addColumn(fam, qual1, val);
108      region.put(p);
109    }
110    // flush them
111    region.flush(true);
112    for (int i = 21; i < 30; i++) {
113      Put p = new Put(Bytes.toBytes("row" + i));
114      p.addColumn(fam, qual1, val);
115      region.put(p);
116    }
117    // flush them
118    region.flush(true);
119
120    HStore store = region.getStore(fam);
121    assertEquals(3, store.getStorefilesCount());
122
123    Collection<HStoreFile> storefiles = store.getStorefiles();
124    Collection<HStoreFile> compactedfiles =
125        store.getStoreEngine().getStoreFileManager().getCompactedfiles();
126    // None of the files should be in compacted state.
127    for (HStoreFile file : storefiles) {
128      assertFalse(file.isCompactedAway());
129    }
130    // Try to run the cleaner without compaction. there should not be any change
131    cleaner.chore();
132    storefiles = store.getStorefiles();
133    // None of the files should be in compacted state.
134    for (HStoreFile file : storefiles) {
135      assertFalse(file.isCompactedAway());
136    }
137    // now do some compaction
138    region.compact(true);
139    // Still the flushed files should be present until the cleaner runs. But the state of it should
140    // be in COMPACTED state
141    assertEquals(1, store.getStorefilesCount());
142    assertEquals(3,
143      ((HStore) store).getStoreEngine().getStoreFileManager().getCompactedfiles().size());
144
145    // Run the cleaner
146    cleaner.chore();
147    assertEquals(1, store.getStorefilesCount());
148    storefiles = store.getStorefiles();
149    for (HStoreFile file : storefiles) {
150      // Should not be in compacted state
151      assertFalse(file.isCompactedAway());
152    }
153    compactedfiles = ((HStore) store).getStoreEngine().getStoreFileManager().getCompactedfiles();
154    assertTrue(compactedfiles.isEmpty());
155    
156  }

229  @Test
230  public void testCleanerWithParallelScanners() throws Exception {
231    // Create the cleaner object
232    CompactedHFilesDischarger cleaner =
233        new CompactedHFilesDischarger(1000, (Stoppable) null, rss, false);
234    // Add some data to the region and do some flushes
235    for (int i = 1; i < 10; i++) {
236      Put p = new Put(Bytes.toBytes("row" + i));
237      p.addColumn(fam, qual1, val);
238      region.put(p);
239    }
240    // flush them
241    region.flush(true);
242    for (int i = 11; i < 20; i++) {
243      Put p = new Put(Bytes.toBytes("row" + i));
244      p.addColumn(fam, qual1, val);
245      region.put(p);
246    }
247    // flush them
248    region.flush(true);
249    for (int i = 21; i < 30; i++) {
250      Put p = new Put(Bytes.toBytes("row" + i));
251      p.addColumn(fam, qual1, val);
252      region.put(p);
253    }
254    // flush them
255    region.flush(true);
256
257    HStore store = region.getStore(fam);
258    assertEquals(3, store.getStorefilesCount());
259
260    Collection<HStoreFile> storefiles = store.getStorefiles();
261    Collection<HStoreFile> compactedfiles =
262        store.getStoreEngine().getStoreFileManager().getCompactedfiles();
263    // None of the files should be in compacted state.
264    for (HStoreFile file : storefiles) {
265      assertFalse(file.isCompactedAway());
266    }
267    startScannerThreads();
268    // Do compaction
269    region.compact(true);
270
271    storefiles = store.getStorefiles();
272    int usedReaderCount = 0;
273    int unusedReaderCount = 0;
274    for (HStoreFile file : storefiles) {
275      if (file.getRefCount() == 0) {
276        unusedReaderCount++;
277      }
278    }
279    compactedfiles = store.getStoreEngine().getStoreFileManager().getCompactedfiles();
280    for(HStoreFile file : compactedfiles) {
281      assertEquals("Refcount should be 3", 3, ((HStoreFile) file).getRefCount());
282      usedReaderCount++;
283    }
284    // The newly compacted file will not be used by any scanner
285    assertEquals("unused reader count should be 1", 1, unusedReaderCount);
286    assertEquals("used reader count should be 3", 3, usedReaderCount);
287    // now run the cleaner
288    cleaner.chore();
289    countDown();
290    // No change in the number of store files as none of the compacted files could be cleaned up
291    assertEquals(1, store.getStorefilesCount());
292    assertEquals(3,
293      ((HStore) store).getStoreEngine().getStoreFileManager().getCompactedfiles().size());
294    while (scanCompletedCounter.get() != 3) {
295      Thread.sleep(100);
296    }
297    // reset
298    latch = new CountDownLatch(3);
299    scanCompletedCounter.set(0);
300    counter.set(0);
301    // Try creating a new scanner and it should use only the new file created after compaction
302    startScannerThreads();
303    storefiles = store.getStorefiles();
304    usedReaderCount = 0;
305    unusedReaderCount = 0;
306    for (HStoreFile file : storefiles) {
307      if (file.getRefCount() == 3) {
308        usedReaderCount++;
309      }
310    }
311    compactedfiles = ((HStore) store).getStoreEngine().getStoreFileManager().getCompactedfiles();
312    for (HStoreFile file : compactedfiles) {
313      assertEquals("Refcount should be 0", 0, file.getRefCount());
314      unusedReaderCount++;
315    }
316    // Though there are files we are not using them for reads
317    assertEquals("unused reader count should be 3", 3, unusedReaderCount);
318    assertEquals("used reader count should be 1", 1, usedReaderCount);
319    countDown();
320    while (scanCompletedCounter.get() != 3) {
321      Thread.sleep(100);
322    }
323    // Run the cleaner again
324    cleaner.chore();
325    // Now the cleaner should be able to clear it up because there are no active readers
326    assertEquals(1, store.getStorefilesCount());
327    storefiles = store.getStorefiles();
328    for (HStoreFile file : storefiles) {
329      // Should not be in compacted state
330      assertFalse(file.isCompactedAway());
331    }
332    compactedfiles = ((HStore) store).getStoreEngine().getStoreFileManager().getCompactedfiles();
333    assertTrue(compactedfiles.isEmpty());
334  }
335
336  protected void countDown() {
337    // count down 3 times
338    latch.countDown();
339    latch.countDown();
340    latch.countDown();
341  }
342
343  protected void startScannerThreads() throws InterruptedException {
344    // Start parallel scan threads
345    ScanThread[] scanThreads = new ScanThread[3];
346    for (int i = 0; i < 3; i++) {
347      scanThreads[i] = new ScanThread((HRegion) region);
348    }
349    for (ScanThread thread : scanThreads) {
350      thread.start();
351    }
352    while (counter.get() != 3) {
353      Thread.sleep(100);
354    }
355  }
356
357  private static class ScanThread extends Thread {
358    private final HRegion region;
359
360    public ScanThread(HRegion region) {
361      this.region = region;
362    }
363
364    @Override
365    public void run() {
366      try {
367        initiateScan(region);
368      } catch (IOException e) {
369        // do nothing
370      }
371    }
372
373    private void initiateScan(HRegion region) throws IOException {
374      Scan scan = new Scan();
375      scan.setCaching(1);
376      RegionScanner resScanner = null;
377      try {
378        resScanner = region.getScanner(scan);
379        List<Cell> results = new ArrayList<>();
380        boolean next = resScanner.next(results);
381        try {
382          counter.incrementAndGet();
383          latch.await();
384        } catch (InterruptedException e) {
385        }
386        while (!next) {
387          resScanner.next(results);
388        }
389      } finally {
390        scanCompletedCounter.incrementAndGet();
391        resScanner.close();
392      }
393    }
394  }
