059@Category({ RegionServerTests.class, MediumTests.class })
060public class TestFIFOCompactionPolicy {
061
062  @ClassRule
063  public static final HBaseClassTestRule CLASS_RULE =
064      HBaseClassTestRule.forClass(TestFIFOCompactionPolicy.class);
065
066  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
067
068  private final TableName tableName = TableName.valueOf(getClass().getSimpleName());
069
070  private final byte[] family = Bytes.toBytes("f");
071
072  private final byte[] qualifier = Bytes.toBytes("q");
073
074  @Rule
075  public ExpectedException error = ExpectedException.none();
076
077  private HStore getStoreWithName(TableName tableName) {
078    MiniHBaseCluster cluster = TEST_UTIL.getMiniHBaseCluster();
079    List<JVMClusterUtil.RegionServerThread> rsts = cluster.getRegionServerThreads();
080    for (int i = 0; i < cluster.getRegionServerThreads().size(); i++) {
081      HRegionServer hrs = rsts.get(i).getRegionServer();
082      for (HRegion region : hrs.getRegions(tableName)) {
083        return region.getStores().iterator().next();
084      }
085    }
086    return null;
087  }
088
089  private HStore prepareData() throws IOException {
090    Admin admin = TEST_UTIL.getAdmin();
091    TableDescriptor desc = TableDescriptorBuilder.newBuilder(tableName)
092        .setValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
093          FIFOCompactionPolicy.class.getName())
094        .setValue(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
095          DisabledRegionSplitPolicy.class.getName())
096        .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(family).setTimeToLive(1).build())
097        .build();
098    admin.createTable(desc);
099    Table table = TEST_UTIL.getConnection().getTable(tableName);
100    TimeOffsetEnvironmentEdge edge =
101        (TimeOffsetEnvironmentEdge) EnvironmentEdgeManager.getDelegate();
102    for (int i = 0; i < 10; i++) {
103      for (int j = 0; j < 10; j++) {
104        byte[] value = new byte[128 * 1024];
105        ThreadLocalRandom.current().nextBytes(value);
106        table.put(new Put(Bytes.toBytes(i * 10 + j)).addColumn(family, qualifier, value));
107      }
108      admin.flush(tableName);
109      edge.increment(1001);
110    }
111    return getStoreWithName(tableName);
112  }
113
114  @BeforeClass
115  public static void setEnvironmentEdge() throws Exception {
116    EnvironmentEdge ee = new TimeOffsetEnvironmentEdge();
117    EnvironmentEdgeManager.injectEdge(ee);
118    Configuration conf = TEST_UTIL.getConfiguration();
119    conf.setInt(HStore.BLOCKING_STOREFILES_KEY, 10000);
120    TEST_UTIL.startMiniCluster(1);
121  }
122
123  @AfterClass
124  public static void resetEnvironmentEdge() throws Exception {
125    TEST_UTIL.shutdownMiniCluster();
126    EnvironmentEdgeManager.reset();
127  }
128
129  @Test
130  public void testPurgeExpiredFiles() throws Exception {
131    HStore store = prepareData();
132    assertEquals(10, store.getStorefilesCount());
133    TEST_UTIL.getAdmin().majorCompact(tableName);
134    TEST_UTIL.waitFor(30000, new ExplainingPredicate<Exception>() {
135
136      @Override
137      public boolean evaluate() throws Exception {
138        return store.getStorefilesCount() == 1;
139      }
140
141      @Override
142      public String explainFailure() throws Exception {
143        return "The store file count " + store.getStorefilesCount() + " is still greater than 1";
144      }
145    });
146  }

148  @Test
149  public void testSanityCheckTTL() throws IOException {
150    error.expect(DoNotRetryIOException.class);
151    error.expectMessage("Default TTL is not supported");
152    TableName tableName = TableName.valueOf(getClass().getSimpleName() + "-TTL");
153    TableDescriptor desc = TableDescriptorBuilder.newBuilder(tableName)
154        .setValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
155          FIFOCompactionPolicy.class.getName())
156        .setValue(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
157          DisabledRegionSplitPolicy.class.getName())
158        .setColumnFamily(ColumnFamilyDescriptorBuilder.of(family)).build();
159    TEST_UTIL.getAdmin().createTable(desc);
160  }
161
162  @Test
163  public void testSanityCheckMinVersion() throws IOException {
164    error.expect(DoNotRetryIOException.class);
165    error.expectMessage("MIN_VERSION > 0 is not supported for FIFO compaction");
166    TableName tableName = TableName.valueOf(getClass().getSimpleName() + "-MinVersion");
167    TableDescriptor desc = TableDescriptorBuilder.newBuilder(tableName)
168        .setValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
169          FIFOCompactionPolicy.class.getName())
170        .setValue(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
171          DisabledRegionSplitPolicy.class.getName())
172        .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(family).setTimeToLive(1)
173            .setMinVersions(1).build())
174        .build();
175    TEST_UTIL.getAdmin().createTable(desc);
176  }

178  @Test
179  public void testSanityCheckBlockingStoreFiles() throws IOException {
180    error.expect(DoNotRetryIOException.class);
181    error.expectMessage("Blocking file count 'hbase.hstore.blockingStoreFiles'");
182    error.expectMessage("is below recommended minimum of 1000 for column family");
183    TableName tableName = TableName.valueOf(getClass().getSimpleName() + "-BlockingStoreFiles");
184    TableDescriptor desc = TableDescriptorBuilder.newBuilder(tableName)
185        .setValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
186          FIFOCompactionPolicy.class.getName())
187        .setValue(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
188          DisabledRegionSplitPolicy.class.getName())
189        .setValue(HStore.BLOCKING_STOREFILES_KEY, "10")
190        .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(family).setTimeToLive(1).build())
191        .build();
192    TEST_UTIL.getAdmin().createTable(desc);
193  }
