150  @Test
151  public void testSingleStripeCompaction() throws Exception {
152    // Create a special policy that only compacts single stripes, using standard methods.
153    Configuration conf = HBaseConfiguration.create();
154    // Test depends on this not being set to pass.  Default breaks test.  TODO: Revisit.
155    conf.unset("hbase.hstore.compaction.min.size");
156    conf.setFloat(CompactionConfiguration.HBASE_HSTORE_COMPACTION_RATIO_KEY, 1.0F);
157    conf.setInt(StripeStoreConfig.MIN_FILES_KEY, 3);
158    conf.setInt(StripeStoreConfig.MAX_FILES_KEY, 4);
159    conf.setLong(StripeStoreConfig.SIZE_TO_SPLIT_KEY, 1000); // make sure the are no splits
160    StoreConfigInformation sci = mock(StoreConfigInformation.class);
161    StripeStoreConfig ssc = new StripeStoreConfig(conf, sci);
162    StripeCompactionPolicy policy = new StripeCompactionPolicy(conf, sci, ssc) {
163      @Override
164      public StripeCompactionRequest selectCompaction(StripeInformationProvider si,
165          List<StoreFile> filesCompacting, boolean isOffpeak) throws IOException {
166        if (!filesCompacting.isEmpty()) return null;
167        return selectSingleStripeCompaction(si, false, false, isOffpeak);
168      }
169
170      @Override
171      public boolean needsCompactions(
172          StripeInformationProvider si, List<StoreFile> filesCompacting) {
173        if (!filesCompacting.isEmpty()) return false;
174        return needsSingleStripeCompaction(si);
175      }
176    };
177
178    // No compaction due to min files or ratio
179    StripeInformationProvider si = createStripesWithSizes(0, 0,
180        new Long[] { 2L }, new Long[] { 3L, 3L }, new Long[] { 5L, 1L });
181    verifyNoCompaction(policy, si);
182    // No compaction due to min files or ratio - will report needed, but not do any.
183    si = createStripesWithSizes(0, 0,
184        new Long[] { 2L }, new Long[] { 3L, 3L }, new Long[] { 5L, 1L, 1L });
185    assertNull(policy.selectCompaction(si, al(), false));
186    assertTrue(policy.needsCompactions(si, al()));
187    // One stripe has possible compaction
188    si = createStripesWithSizes(0, 0,
189        new Long[] { 2L }, new Long[] { 3L, 3L }, new Long[] { 5L, 4L, 3L });
190    verifySingleStripeCompaction(policy, si, 2, null);
191    // Several stripes have possible compactions; choose best quality (removes most files)
192    si = createStripesWithSizes(0, 0,
193        new Long[] { 3L, 2L, 2L }, new Long[] { 2L, 2L, 1L }, new Long[] { 3L, 2L, 2L, 1L });
194    verifySingleStripeCompaction(policy, si, 2, null);
195    si = createStripesWithSizes(0, 0,
196        new Long[] { 5L }, new Long[] { 3L, 2L, 2L, 1L }, new Long[] { 3L, 2L, 2L });
197    verifySingleStripeCompaction(policy, si, 1, null);
198    // Or with smallest files, if the count is the same 
199    si = createStripesWithSizes(0, 0,
200        new Long[] { 3L, 3L, 3L }, new Long[] { 3L, 1L, 2L }, new Long[] { 3L, 2L, 2L });
201    verifySingleStripeCompaction(policy, si, 1, null);
202    // Verify max count is respected.
203    si = createStripesWithSizes(0, 0, new Long[] { 5L }, new Long[] { 5L, 4L, 4L, 4L, 4L });
204    List<StoreFile> sfs = si.getStripes().get(1).subList(1, 5);
205    verifyCompaction(policy, si, sfs, null, 1, null, si.getStartRow(1), si.getEndRow(1), true);
206    // Verify ratio is applied.
207    si = createStripesWithSizes(0, 0, new Long[] { 5L }, new Long[] { 50L, 4L, 4L, 4L, 4L });
208    sfs = si.getStripes().get(1).subList(1, 5);
209    verifyCompaction(policy, si, sfs, null, 1, null, si.getStartRow(1), si.getEndRow(1), true);
210  }

276    Configuration conf = HBaseConfiguration.create();
277    // Test depends on this not being set to pass.  Default breaks test.  TODO: Revisit.
278    conf.unset("hbase.hstore.compaction.min.size");
279    // First test everything with default split count of 2, then split into more.
280    conf.setInt(StripeStoreConfig.MIN_FILES_KEY, 2);
281    Long[] toSplit = new Long[] { defaultSplitSize - 2, 1L, 1L };
282    Long[] noSplit = new Long[] { defaultSplitSize - 2, 1L };
283    long splitTargetSize = (long)(defaultSplitSize / defaultSplitCount);
284    // Don't split if not eligible for compaction.
285    StripeCompactionPolicy.StripeInformationProvider si =
286        createStripesWithSizes(0, 0, new Long[] { defaultSplitSize - 2, 2L });
287    assertNull(createPolicy(conf).selectCompaction(si, al(), false));
288    // Make sure everything is eligible.
289    conf.setFloat(CompactionConfiguration.HBASE_HSTORE_COMPACTION_RATIO_KEY, 500f);
290    StripeCompactionPolicy policy = createPolicy(conf);
291    verifyWholeStripesCompaction(policy, si, 0, 0, null, 2, splitTargetSize);
292    // Add some extra stripes...
293    si = createStripesWithSizes(0, 0, noSplit, noSplit, toSplit);
294    verifyWholeStripesCompaction(policy, si, 2, 2, null, 2, splitTargetSize);
295    // In the middle.
296    si = createStripesWithSizes(0, 0, noSplit, toSplit, noSplit);
297    verifyWholeStripesCompaction(policy, si, 1, 1, null, 2, splitTargetSize);
298    // No split-off with different config (larger split size).
299    // However, in this case some eligible stripe will just be compacted alone.
300    StripeCompactionPolicy specPolicy = createPolicy(
301        conf, defaultSplitSize + 1, defaultSplitCount, defaultInitialCount, false);
302    verifySingleStripeCompaction(specPolicy, si, 1, null);
303  }

344  @Test
345  public void testMergeExpiredFiles() throws Exception {
346    ManualEnvironmentEdge edge = new ManualEnvironmentEdge();
347    long now = defaultTtl + 2;
348    edge.setValue(now);
349    EnvironmentEdgeManager.injectEdge(edge);
350    try {
351      StoreFile expiredFile = createFile(), notExpiredFile = createFile();
352      when(expiredFile.getReader().getMaxTimestamp()).thenReturn(now - defaultTtl - 1);
353      when(notExpiredFile.getReader().getMaxTimestamp()).thenReturn(now - defaultTtl + 1);
354      List<StoreFile> expired = Lists.newArrayList(expiredFile, expiredFile);
355      List<StoreFile> notExpired = Lists.newArrayList(notExpiredFile, notExpiredFile);
356      List<StoreFile> mixed = Lists.newArrayList(expiredFile, notExpiredFile);
357
358      StripeCompactionPolicy policy = createPolicy(HBaseConfiguration.create(),
359          defaultSplitSize, defaultSplitCount, defaultInitialCount, true);
360      // Merge expired if there are eligible stripes.
361      StripeCompactionPolicy.StripeInformationProvider si =
362          createStripesWithFiles(expired, expired, expired);
363      verifyWholeStripesCompaction(policy, si, 0, 2, null, 1, Long.MAX_VALUE, false);
364      // Don't merge if nothing expired.
365      si = createStripesWithFiles(notExpired, notExpired, notExpired);
366      assertNull(policy.selectCompaction(si, al(), false));
367      // Merge one expired stripe with next.
368      si = createStripesWithFiles(notExpired, expired, notExpired);
369      verifyWholeStripesCompaction(policy, si, 1, 2, null, 1, Long.MAX_VALUE, false);
370      // Merge the biggest run out of multiple options.
371      // Merge one expired stripe with next.
372      si = createStripesWithFiles(notExpired, expired, notExpired, expired, expired, notExpired);
373      verifyWholeStripesCompaction(policy, si, 3, 4, null, 1, Long.MAX_VALUE, false);
374      // Stripe with a subset of expired files is not merged.
375      si = createStripesWithFiles(expired, expired, notExpired, expired, mixed);
376      verifyWholeStripesCompaction(policy, si, 0, 1, null, 1, Long.MAX_VALUE, false);
377    } finally {
378      EnvironmentEdgeManager.reset();
379    }
380  }

421  @Test
422  public void testSingleStripeDropDeletes() throws Exception {
423    Configuration conf = HBaseConfiguration.create();
424    // Test depends on this not being set to pass.  Default breaks test.  TODO: Revisit.
425    conf.unset("hbase.hstore.compaction.min.size");
426    StripeCompactionPolicy policy = createPolicy(conf);
427    // Verify the deletes can be dropped if there are no L0 files.
428    Long[][] stripes = new Long[][] { new Long[] { 3L, 2L, 2L, 2L }, new Long[] { 6L } };
429    StripeInformationProvider si = createStripesWithSizes(0, 0, stripes);
430    verifySingleStripeCompaction(policy, si, 0, true);
431    // But cannot be dropped if there are.
432    si = createStripesWithSizes(2, 2, stripes);
433    verifySingleStripeCompaction(policy, si, 0, false);
434    // Unless there are enough to cause L0 compaction.
435    si = createStripesWithSizes(6, 2, stripes);
436    ConcatenatedLists<StoreFile> sfs = new ConcatenatedLists<StoreFile>();
437    sfs.addSublist(si.getLevel0Files());
438    sfs.addSublist(si.getStripes().get(0));
439    verifyCompaction(
440        policy, si, sfs, si.getStartRow(0), si.getEndRow(0), si.getStripeBoundaries());
441    // If we cannot actually compact all files in some stripe, L0 is chosen.
442    si = createStripesWithSizes(6, 2,
443        new Long[][] { new Long[] { 10L, 1L, 1L, 1L, 1L }, new Long[] { 12L } });
444    verifyCompaction(policy, si, si.getLevel0Files(), null, null, si.getStripeBoundaries());
445    // even if L0 has no file
446    // if all files of stripe aren't selected, delete must not be dropped.
447    stripes = new Long[][] { new Long[] { 100L, 3L, 2L, 2L, 2L }, new Long[] { 6L } };
448    si = createStripesWithSizes(0, 0, stripes);
449    List<StoreFile> compact_file = new ArrayList<StoreFile>();
450    Iterator<StoreFile> iter = si.getStripes().get(0).listIterator(1);
451    while (iter.hasNext()) {
452        compact_file.add(iter.next());
453    }
454    verifyCompaction(policy, si, compact_file, false, 1, null, si.getStartRow(0), si.getEndRow(0), true);
455  }

457  /********* HELPER METHODS ************/
458  private static StripeCompactionPolicy createPolicy(
459      Configuration conf) throws Exception {
460    return createPolicy(conf, defaultSplitSize, defaultSplitCount, defaultInitialCount, false);
461  }

463  private static StripeCompactionPolicy createPolicy(Configuration conf,
464      long splitSize, float splitCount, int initialCount, boolean hasTtl) throws Exception {
465    conf.setLong(StripeStoreConfig.SIZE_TO_SPLIT_KEY, splitSize);
466    conf.setFloat(StripeStoreConfig.SPLIT_PARTS_KEY, splitCount);
467    conf.setInt(StripeStoreConfig.INITIAL_STRIPE_COUNT_KEY, initialCount);
468    StoreConfigInformation sci = mock(StoreConfigInformation.class);
469    when(sci.getStoreFileTtl()).thenReturn(hasTtl ? defaultTtl : Long.MAX_VALUE);
470    StripeStoreConfig ssc = new StripeStoreConfig(conf, sci);
471    return new StripeCompactionPolicy(conf, sci, ssc);

530  /**
531   * Verify arbitrary compaction.
532   * @param policy Policy to test.
533   * @param si Stripe information pre-set with stripes to test.
534   * @param sfs Files that should be compacted.
535   * @param dropDeletesFrom Row from which to drop deletes.
536   * @param dropDeletesTo Row to which to drop deletes.
537   * @param boundaries Expected target stripe boundaries.
538   */
539  private void verifyCompaction(StripeCompactionPolicy policy, StripeInformationProvider si,
540      Collection<StoreFile> sfs, byte[] dropDeletesFrom, byte[] dropDeletesTo,
541      final List<byte[]> boundaries) throws Exception {
542    StripeCompactor sc = mock(StripeCompactor.class);
543    assertTrue(policy.needsCompactions(si, al()));
544    StripeCompactionPolicy.StripeCompactionRequest scr = policy.selectCompaction(si, al(), false);
545    verifyCollectionsEqual(sfs, scr.getRequest().getFiles());
546    scr.execute(sc, NoLimitThroughputController.INSTANCE, null);
547    verify(sc, times(1)).compact(eq(scr.getRequest()), argThat(new ArgumentMatcher<List<byte[]>>() {
548      @Override
549      public boolean matches(Object argument) {
550        @SuppressWarnings("unchecked")
551        List<byte[]> other = (List<byte[]>) argument;
552        if (other.size() != boundaries.size()) return false;
553        for (int i = 0; i < other.size(); ++i) {
554          if (!Bytes.equals(other.get(i), boundaries.get(i))) return false;
555        }
556        return true;
557      }
558    }), dropDeletesFrom == null ? isNull(byte[].class) : aryEq(dropDeletesFrom),
559      dropDeletesTo == null ? isNull(byte[].class) : aryEq(dropDeletesTo),
560      any(NoLimitThroughputController.class), any(User.class));
561  }

589  /** Verify arbitrary flush. */
590  protected void verifyFlush(StripeCompactionPolicy policy, StripeInformationProvider si,
591      KeyValue[] input, KeyValue[][] expected, byte[][] boundaries) throws IOException {
592    StoreFileWritersCapture writers = new StoreFileWritersCapture();
593    StripeStoreFlusher.StripeFlushRequest req = policy.selectFlush(CellComparator.COMPARATOR, si,
594      input.length);
595    StripeMultiFileWriter mw = req.createWriter();
596    mw.init(null, writers);
597    for (KeyValue kv : input) {
598      mw.append(kv);
599    }
600    boolean hasMetadata = boundaries != null;
601    mw.commitWriters(0, false);
602    writers.verifyKvs(expected, true, hasMetadata);
603    if (hasMetadata) {
604      writers.verifyBoundaries(boundaries);
605    }
606  }

694  private static StripeInformationProvider createStripes(List<byte[]> boundaries,
695      List<List<Long>> stripeSizes, List<Long> l0Sizes) throws Exception {
696    List<List<StoreFile>> stripeFiles = new ArrayList<List<StoreFile>>(stripeSizes.size());
697    for (List<Long> sizes : stripeSizes) {
698      List<StoreFile> sfs = new ArrayList<StoreFile>();
699      for (Long size : sizes) {
700        sfs.add(createFile(size));
701      }
702      stripeFiles.add(sfs);
703    }
704    List<StoreFile> l0Files = new ArrayList<StoreFile>();
705    for (Long size : l0Sizes) {
706      l0Files.add(createFile(size));
707    }
708    return createStripesWithFiles(boundaries, stripeFiles, l0Files);
709  }

711  /**
712   * This method actually does all the work.
713   */
714  private static StripeInformationProvider createStripesWithFiles(List<byte[]> boundaries,
715      List<List<StoreFile>> stripeFiles, List<StoreFile> l0Files) throws Exception {
716    ArrayList<ImmutableList<StoreFile>> stripes = new ArrayList<ImmutableList<StoreFile>>();
717    ArrayList<byte[]> boundariesList = new ArrayList<byte[]>();
718    StripeInformationProvider si = mock(StripeInformationProvider.class);
719    if (!stripeFiles.isEmpty()) {
720      assert stripeFiles.size() == (boundaries.size() + 1);
721      boundariesList.add(OPEN_KEY);
722      for (int i = 0; i <= boundaries.size(); ++i) {
723        byte[] startKey = ((i == 0) ? OPEN_KEY : boundaries.get(i - 1));
724        byte[] endKey = ((i == boundaries.size()) ? OPEN_KEY : boundaries.get(i));
725        boundariesList.add(endKey);
726        for (StoreFile sf : stripeFiles.get(i)) {
727          setFileStripe(sf, startKey, endKey);
728        }
729        stripes.add(ImmutableList.copyOf(stripeFiles.get(i)));
730        when(si.getStartRow(eq(i))).thenReturn(startKey);
731        when(si.getEndRow(eq(i))).thenReturn(endKey);
732      }
733    }
734    ConcatenatedLists<StoreFile> sfs = new ConcatenatedLists<StoreFile>();
735    sfs.addAllSublists(stripes);
736    sfs.addSublist(l0Files);
737    when(si.getStorefiles()).thenReturn(sfs);
738    when(si.getStripes()).thenReturn(stripes);
739    when(si.getStripeBoundaries()).thenReturn(boundariesList);
740    when(si.getStripeCount()).thenReturn(stripes.size());
741    when(si.getLevel0Files()).thenReturn(l0Files);
742    return si;
743  }

803  private static class Scanner implements InternalScanner {
804    private final ArrayList<KeyValue> kvs;
805
806    public Scanner(KeyValue... kvs) {
807      this.kvs = new ArrayList<KeyValue>(Arrays.asList(kvs));
808    }
809
810    @Override
811    public boolean next(List<Cell> results) throws IOException {
812      if (kvs.isEmpty()) return false;
813      results.add(kvs.remove(0));
814      return !kvs.isEmpty();
815    }
816
817    @Override
818    public boolean next(List<Cell> result, ScannerContext scannerContext)
819        throws IOException {
820      return next(result);
821    }

824    public void close() throws IOException {
825    }
