066@RunWith(Parameterized.class)
067@Category({ RegionServerTests.class, SmallTests.class })
068public class TestDateTieredCompactor {
069
070  private static final byte[] NAME_OF_THINGS = Bytes.toBytes("foo");
071
072  private static final TableName TABLE_NAME = TableName.valueOf(NAME_OF_THINGS, NAME_OF_THINGS);
073
074  private static final KeyValue KV_A = new KeyValue(Bytes.toBytes("aaa"), 100L);
075
076  private static final KeyValue KV_B = new KeyValue(Bytes.toBytes("bbb"), 200L);
077
078  private static final KeyValue KV_C = new KeyValue(Bytes.toBytes("ccc"), 300L);
079
080  private static final KeyValue KV_D = new KeyValue(Bytes.toBytes("ddd"), 400L);
081
082  @Parameters(name = "{index}: usePrivateReaders={0}")
083  public static Iterable<Object[]> data() {
084    return Arrays.asList(new Object[] { true }, new Object[] { false });
085  }
086
087  @Parameter
088  public boolean usePrivateReaders;
089
090  private DateTieredCompactor createCompactor(StoreFileWritersCapture writers,
091      final KeyValue[] input, List<HStoreFile> storefiles) throws Exception {
092    Configuration conf = HBaseConfiguration.create();
093    conf.setBoolean("hbase.regionserver.compaction.private.readers", usePrivateReaders);
094    final Scanner scanner = new Scanner(input);
095    // Create store mock that is satisfactory for compactor.
096    HColumnDescriptor col = new HColumnDescriptor(NAME_OF_THINGS);
097    ScanInfo si = new ScanInfo(conf, col, Long.MAX_VALUE, 0, CellComparatorImpl.COMPARATOR);
098    HStore store = mock(HStore.class);
099    when(store.getStorefiles()).thenReturn(storefiles);
100    when(store.getColumnFamilyDescriptor()).thenReturn(col);
101    when(store.getScanInfo()).thenReturn(si);
102    when(store.areWritesEnabled()).thenReturn(true);
103    when(store.getFileSystem()).thenReturn(mock(FileSystem.class));
104    when(store.getRegionInfo()).thenReturn(new HRegionInfo(TABLE_NAME));
105    when(store.createWriterInTmp(anyLong(), any(Compression.Algorithm.class), anyBoolean(),
106      anyBoolean(), anyBoolean(), anyBoolean())).thenAnswer(writers);
107    when(store.getComparator()).thenReturn(CellComparatorImpl.COMPARATOR);
108    OptionalLong maxSequenceId = StoreUtils.getMaxSequenceIdInList(storefiles);
109    when(store.getMaxSequenceId()).thenReturn(maxSequenceId);
110
111    return new DateTieredCompactor(conf, store) {
112      @Override
113      protected InternalScanner createScanner(HStore store, ScanInfo scanInfo,
114          List<StoreFileScanner> scanners, long smallestReadPoint, long earliestPutTs,
115          byte[] dropDeletesFromRow, byte[] dropDeletesToRow) throws IOException {
116        return scanner;
117      }
118
119      @Override
120      protected InternalScanner createScanner(HStore store, ScanInfo scanInfo,
121          List<StoreFileScanner> scanners, ScanType scanType, long smallestReadPoint,
122          long earliestPutTs) throws IOException {
123        return scanner;
124      }
125    };
126  }
127
128  private void verify(KeyValue[] input, List<Long> boundaries, KeyValue[][] output,
129      boolean allFiles) throws Exception {
130    StoreFileWritersCapture writers = new StoreFileWritersCapture();
131    HStoreFile sf1 = createDummyStoreFile(1L);
132    HStoreFile sf2 = createDummyStoreFile(2L);
133    DateTieredCompactor dtc = createCompactor(writers, input, Arrays.asList(sf1, sf2));
134    List<Path> paths = dtc.compact(new CompactionRequestImpl(Arrays.asList(sf1)),
135      boundaries.subList(0, boundaries.size() - 1), NoLimitThroughputController.INSTANCE, null);
136    writers.verifyKvs(output, allFiles, boundaries);
137    if (allFiles) {
138      assertEquals(output.length, paths.size());
139    }
140  }
141
142  @SuppressWarnings("unchecked")
143  private static <T> T[] a(T... a) {
144    return a;
145  }
146
147  @Test
148  public void test() throws Exception {
149    verify(a(KV_A, KV_B, KV_C, KV_D), Arrays.asList(100L, 200L, 300L, 400L, 500L),
150      a(a(KV_A), a(KV_B), a(KV_C), a(KV_D)), true);
151    verify(a(KV_A, KV_B, KV_C, KV_D), Arrays.asList(Long.MIN_VALUE, 200L, Long.MAX_VALUE),
152      a(a(KV_A), a(KV_B, KV_C, KV_D)), false);
153    verify(a(KV_A, KV_B, KV_C, KV_D), Arrays.asList(Long.MIN_VALUE, Long.MAX_VALUE),
154      new KeyValue[][] { a(KV_A, KV_B, KV_C, KV_D) }, false);
155  }
156
157  @Test
158  public void testEmptyOutputFile() throws Exception {
159    StoreFileWritersCapture writers = new StoreFileWritersCapture();
160    CompactionRequestImpl request = createDummyRequest();
161    DateTieredCompactor dtc = createCompactor(writers, new KeyValue[0],
162      new ArrayList<>(request.getFiles()));
163    List<Path> paths = dtc.compact(request, Arrays.asList(Long.MIN_VALUE, Long.MAX_VALUE),
164      NoLimitThroughputController.INSTANCE, null);
165    assertEquals(1, paths.size());
166    List<StoreFileWritersCapture.Writer> dummyWriters = writers.getWriters();
167    assertEquals(1, dummyWriters.size());
168    StoreFileWritersCapture.Writer dummyWriter = dummyWriters.get(0);
169    assertTrue(dummyWriter.kvs.isEmpty());
170    assertTrue(dummyWriter.hasMetadata);
171  }
