061/**
062 * Tests that archiving compacted files behaves correctly when encountering exceptions.
063 */
064@Category(MediumTests.class)
065public class TestCompactionArchiveIOException {
066  private static final String ERROR_FILE = "fffffffffffffffffdeadbeef";
067
068  public HBaseTestingUtility testUtil;
069
070  private Path testDir;
071
072  @Rule
073  public TestName name = new TestName();
074
075  @Before
076  public void setup() throws Exception {
077    testUtil = new HBaseTestingUtility();
078    testUtil.startMiniDFSCluster(1);
079    testDir = testUtil.getDataTestDirOnTestFS();
080    FSUtils.setRootDir(testUtil.getConfiguration(), testDir);
081  }
082
083  @After
084  public void tearDown() throws Exception {
085    testUtil.cleanupTestDir();
086    testUtil.shutdownMiniDFSCluster();
087  }
088
089  @Test
090  public void testRemoveCompactedFilesWithException() throws Exception {
091    byte[] fam = Bytes.toBytes("f");
092    byte[] col = Bytes.toBytes("c");
093    byte[] val = Bytes.toBytes("val");
094
095    final TableName tableName = TableName.valueOf(name.getMethodName());
096    HTableDescriptor htd = new HTableDescriptor(tableName);
097    htd.addFamily(new HColumnDescriptor(fam));
098    HRegionInfo info = new HRegionInfo(tableName, null, null, false);
099    final HRegion region = initHRegion(htd, info);
100    RegionServerServices rss = mock(RegionServerServices.class);
101    List<Region> regions = new ArrayList<Region>();
102    regions.add(region);
103    when(rss.getOnlineRegions()).thenReturn(regions);
104
105    // Create the cleaner object
106    final CompactedHFilesDischarger cleaner =
107        new CompactedHFilesDischarger(1000, (Stoppable) null, rss, false);
108    // Add some data to the region and do some flushes
109    int batchSize = 10;
110    int fileCount = 10;
111    for (int f = 0; f < fileCount; f++) {
112      int start = f * batchSize;
113      for (int i = start; i < start + batchSize; i++) {
114        Put p = new Put(Bytes.toBytes("row" + i));
115        p.addColumn(fam, col, val);
116        region.put(p);
117      }
118      // flush them
119      region.flush(true);
120    }
121
122    HStore store = (HStore) region.getStore(fam);
123    assertEquals(fileCount, store.getStorefilesCount());
124
125    Collection<StoreFile> storefiles = store.getStorefiles();
126    // None of the files should be in compacted state.
127    for (StoreFile file : storefiles) {
128      assertFalse(file.isCompactedAway());
129    }
130
131    StoreFileManager fileManager = store.getStoreEngine().getStoreFileManager();
132    Collection<StoreFile> initialCompactedFiles = fileManager.getCompactedfiles();
133    assertTrue(initialCompactedFiles == null || initialCompactedFiles.isEmpty());
134
135    // Do compaction
136    region.compact(true);
137
138    // all prior store files should now be compacted
139    Collection<StoreFile> compactedFilesPreClean = fileManager.getCompactedfiles();
140    assertNotNull(compactedFilesPreClean);
141    assertTrue(compactedFilesPreClean.size() > 0);
142
143    // add the dummy file to the store directory
144    HRegionFileSystem regionFS = region.getRegionFileSystem();
145    Path errFile = regionFS.getStoreFilePath(Bytes.toString(fam), ERROR_FILE);
146    FSDataOutputStream out = regionFS.getFileSystem().create(errFile);
147    out.writeInt(1);
148    out.close();
149
150    StoreFile errStoreFile = new MockStoreFile(testUtil, errFile, 1, 0, false, 1);
151    fileManager.addCompactionResults(
152        ImmutableList.of(errStoreFile), ImmutableList.<StoreFile>of());
153
154    // cleanup compacted files
155    cleaner.chore();
156
157    // make sure the compacted files are cleared
158    Collection<StoreFile> compactedFilesPostClean = fileManager.getCompactedfiles();
159    assertEquals(1, compactedFilesPostClean.size());
160    for (StoreFile origFile : compactedFilesPreClean) {
161      assertFalse(compactedFilesPostClean.contains(origFile));
162    }
163
164    // close the region
165    try {
166      region.close();
167    } catch (FailedArchiveException e) {
168      // expected due to errorfile
169      assertEquals(1, e.getFailedFiles().size());
170      assertEquals(ERROR_FILE, e.getFailedFiles().iterator().next().getName());
171    }
172  }
173
174  private HRegion initHRegion(HTableDescriptor htd, HRegionInfo info)
175      throws IOException {
176    Configuration conf = testUtil.getConfiguration();
177    Path tableDir = FSUtils.getTableDir(testDir, htd.getTableName());
178    Path regionDir = new Path(tableDir, info.getEncodedName());
179    Path storeDir = new Path(regionDir, htd.getColumnFamilies()[0].getNameAsString());
180
181
182    FileSystem errFS = spy(testUtil.getTestFileSystem());
183    // Prior to HBASE-16964, when an exception is thrown archiving any compacted file,
184    // none of the other files are cleared from the compactedfiles list.
185    // Simulate this condition with a dummy file
186    doThrow(new IOException("Error for test"))
187        .when(errFS).rename(eq(new Path(storeDir, ERROR_FILE)), any(Path.class));
188
189    HRegionFileSystem fs = new HRegionFileSystem(conf, errFS, tableDir, info);
190    final Configuration walConf = new Configuration(conf);
191    FSUtils.setRootDir(walConf, tableDir);
192    final WALFactory wals = new WALFactory(walConf, null, "log_" + info.getEncodedName());
193    HRegion region =
194        new HRegion(fs, wals.getWAL(info.getEncodedNameAsBytes(), info.getTable().getNamespace()),
195            conf, htd, null);
196
197    region.initialize();
198
199    return region;
200  }
