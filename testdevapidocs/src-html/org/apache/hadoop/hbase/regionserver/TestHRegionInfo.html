066  @Test
067  public void testReadAndWriteHRegionInfoFile() throws IOException, InterruptedException {
068    HBaseTestingUtility htu = new HBaseTestingUtility();
069    HRegionInfo hri = HRegionInfo.FIRST_META_REGIONINFO;
070    Path basedir = htu.getDataTestDir();
071    // Create a region.  That'll write the .regioninfo file.
072    FSTableDescriptors fsTableDescriptors = new FSTableDescriptors(htu.getConfiguration());
073    HRegion r = HBaseTestingUtility.createRegionAndWAL(hri, basedir, htu.getConfiguration(),
074        fsTableDescriptors.get(TableName.META_TABLE_NAME));
075    // Get modtime on the file.
076    long modtime = getModTime(r);
077    HBaseTestingUtility.closeRegionAndWAL(r);
078    Thread.sleep(1001);
079    r = HRegion.openHRegion(basedir, hri, fsTableDescriptors.get(TableName.META_TABLE_NAME),
080        null, htu.getConfiguration());
081    // Ensure the file is not written for a second time.
082    long modtime2 = getModTime(r);
083    assertEquals(modtime, modtime2);
084    // Now load the file.
085    org.apache.hadoop.hbase.client.RegionInfo deserializedHri = HRegionFileSystem.loadRegionInfoFileContent(
086        r.getRegionFileSystem().getFileSystem(), r.getRegionFileSystem().getRegionDir());
087    assertTrue(org.apache.hadoop.hbase.client.RegionInfo.COMPARATOR.compare(hri, deserializedHri) == 0);
088    HBaseTestingUtility.closeRegionAndWAL(r);
089  }
090
091  long getModTime(final HRegion r) throws IOException {
092    FileStatus[] statuses = r.getRegionFileSystem().getFileSystem().listStatus(
093      new Path(r.getRegionFileSystem().getRegionDir(), HRegionFileSystem.REGION_INFO_FILE));
094    assertTrue(statuses != null && statuses.length == 1);
095    return statuses[0].getModificationTime();
096  }

122  @Test
123  public void testContainsRange() {
124    HTableDescriptor tableDesc = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
125    HRegionInfo hri = new HRegionInfo(
126        tableDesc.getTableName(), Bytes.toBytes("a"), Bytes.toBytes("g"));
127    // Single row range at start of region
128    assertTrue(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("a")));
129    // Fully contained range
130    assertTrue(hri.containsRange(Bytes.toBytes("b"), Bytes.toBytes("c")));
131    // Range overlapping start of region
132    assertTrue(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("c")));
133    // Fully contained single-row range
134    assertTrue(hri.containsRange(Bytes.toBytes("c"), Bytes.toBytes("c")));
135    // Range that overlaps end key and hence doesn't fit
136    assertFalse(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("g")));
137    // Single row range on end key
138    assertFalse(hri.containsRange(Bytes.toBytes("g"), Bytes.toBytes("g")));
139    // Single row range entirely outside
140    assertFalse(hri.containsRange(Bytes.toBytes("z"), Bytes.toBytes("z")));
141
142    // Degenerate range
143    try {
144      hri.containsRange(Bytes.toBytes("z"), Bytes.toBytes("a"));
145      fail("Invalid range did not throw IAE");
146    } catch (IllegalArgumentException iae) {
147    }
148  }

166  @Test
167  public void testComparator() {
168    final TableName tableName = TableName.valueOf(name.getMethodName());
169    byte[] empty = new byte[0];
170    HRegionInfo older = new HRegionInfo(tableName, empty, empty, false, 0L);
171    HRegionInfo newer = new HRegionInfo(tableName, empty, empty, false, 1L);
172    assertTrue(older.compareTo(newer) < 0);
173    assertTrue(newer.compareTo(older) > 0);
174    assertEquals(0, older.compareTo(older));
175    assertEquals(0, newer.compareTo(newer));
176
177    HRegionInfo a = new HRegionInfo(TableName.valueOf("a"), null, null);
178    HRegionInfo b = new HRegionInfo(TableName.valueOf("b"), null, null);
179    assertNotEquals(0, a.compareTo(b));
180    HTableDescriptor t = new HTableDescriptor(TableName.valueOf("t"));
181    byte [] midway = Bytes.toBytes("midway");
182    a = new HRegionInfo(t.getTableName(), null, midway);
183    b = new HRegionInfo(t.getTableName(), midway, null);
184    assertTrue(a.compareTo(b) < 0);
185    assertTrue(b.compareTo(a) > 0);
186    assertEquals(a, a);
187    assertEquals(0, a.compareTo(a));
188    a = new HRegionInfo(t.getTableName(), Bytes.toBytes("a"), Bytes.toBytes("d"));
189    b = new HRegionInfo(t.getTableName(), Bytes.toBytes("e"), Bytes.toBytes("g"));
190    assertTrue(a.compareTo(b) < 0);
191    a = new HRegionInfo(t.getTableName(), Bytes.toBytes("aaaa"), Bytes.toBytes("dddd"));
192    b = new HRegionInfo(t.getTableName(), Bytes.toBytes("e"), Bytes.toBytes("g"));
193    assertTrue(a.compareTo(b) < 0);
194    a = new HRegionInfo(t.getTableName(), Bytes.toBytes("aaaa"), Bytes.toBytes("dddd"));
195    b = new HRegionInfo(t.getTableName(), Bytes.toBytes("aaaa"), Bytes.toBytes("eeee"));
196    assertTrue(a.compareTo(b) < 0);
197
198  }

229  public void testParseName() throws IOException {
230    final TableName tableName = TableName.valueOf(name.getMethodName());
231    byte[] startKey = Bytes.toBytes("startKey");
232    long regionId = System.currentTimeMillis();
233    int replicaId = 42;
234
235    // test without replicaId
236    byte[] regionName = HRegionInfo.createRegionName(tableName, startKey, regionId, false);
237
238    byte[][] fields = HRegionInfo.parseRegionName(regionName);
239    assertArrayEquals(Bytes.toString(fields[0]),tableName.getName(), fields[0]);
240    assertArrayEquals(Bytes.toString(fields[1]),startKey, fields[1]);
241    assertArrayEquals(Bytes.toString(fields[2]), Bytes.toBytes(Long.toString(regionId)),fields[2]);
242    assertEquals(3, fields.length);
243
244    // test with replicaId
245    regionName = HRegionInfo.createRegionName(tableName, startKey, regionId,
246      replicaId, false);
247
248    fields = HRegionInfo.parseRegionName(regionName);
249    assertArrayEquals(Bytes.toString(fields[0]),tableName.getName(), fields[0]);
250    assertArrayEquals(Bytes.toString(fields[1]),startKey, fields[1]);
251    assertArrayEquals(Bytes.toString(fields[2]), Bytes.toBytes(Long.toString(regionId)),fields[2]);
252    assertArrayEquals(Bytes.toString(fields[3]), Bytes.toBytes(
253      String.format(HRegionInfo.REPLICA_ID_FORMAT, replicaId)), fields[3]);

257  public void testConvert() {
258    final TableName tableName = TableName.valueOf("ns1:" + name.getMethodName());
259    byte[] startKey = Bytes.toBytes("startKey");
260    byte[] endKey = Bytes.toBytes("endKey");
261    boolean split = false;
262    long regionId = System.currentTimeMillis();
263    int replicaId = 42;
264
265
266    HRegionInfo hri = new HRegionInfo(tableName, startKey, endKey, split,
267      regionId, replicaId);
268
269    // convert two times, compare
270    HRegionInfo convertedHri = HRegionInfo.convert(HRegionInfo.convert(hri));
271
272    assertEquals(hri, convertedHri);
273
274    // test convert RegionInfo without replicaId
275    RegionInfo info = RegionInfo.newBuilder()
276      .setTableName(HBaseProtos.TableName.newBuilder()
277        .setQualifier(UnsafeByteOperations.unsafeWrap(tableName.getQualifier()))
278        .setNamespace(UnsafeByteOperations.unsafeWrap(tableName.getNamespace()))
279        .build())
280      .setStartKey(UnsafeByteOperations.unsafeWrap(startKey))
281      .setEndKey(UnsafeByteOperations.unsafeWrap(endKey))
282      .setSplit(split)
283      .setRegionId(regionId)
284      .build();
285
286    convertedHri = HRegionInfo.convert(info);
287    HRegionInfo expectedHri = new HRegionInfo(tableName, startKey, endKey, split,
288      regionId, 0); // expecting default replicaId
289
290    assertEquals(expectedHri, convertedHri);

292  @Test
293  public void testRegionDetailsForDisplay() throws IOException {
294    byte[] startKey = new byte[] {0x01, 0x01, 0x02, 0x03};
295    byte[] endKey = new byte[] {0x01, 0x01, 0x02, 0x04};
296    Configuration conf = new Configuration();
297    conf.setBoolean("hbase.display.keys", false);
298    HRegionInfo h = new HRegionInfo(TableName.valueOf(name.getMethodName()), startKey, endKey);
299    checkEquality(h, conf);
300    // check HRIs with non-default replicaId
301    h = new HRegionInfo(TableName.valueOf(name.getMethodName()), startKey, endKey, false,
302        System.currentTimeMillis(), 1);
303    checkEquality(h, conf);
304    Assert.assertArrayEquals(HRegionInfo.HIDDEN_END_KEY,
305        HRegionInfo.getEndKeyForDisplay(h, conf));
306    Assert.assertArrayEquals(HRegionInfo.HIDDEN_START_KEY,
307        HRegionInfo.getStartKeyForDisplay(h, conf));
308
309    RegionState state = RegionState.createForTesting(h, RegionState.State.OPEN);
310    String descriptiveNameForDisplay =
311        HRegionInfo.getDescriptiveNameFromRegionStateForDisplay(state, conf);
312    checkDescriptiveNameEquality(descriptiveNameForDisplay,state.toDescriptiveString(), startKey);
313
314    conf.setBoolean("hbase.display.keys", true);
315    Assert.assertArrayEquals(endKey, HRegionInfo.getEndKeyForDisplay(h, conf));
316    Assert.assertArrayEquals(startKey, HRegionInfo.getStartKeyForDisplay(h, conf));
317    Assert.assertEquals(state.toDescriptiveString(),
318        HRegionInfo.getDescriptiveNameFromRegionStateForDisplay(state, conf));
319  }
320
321  private void checkDescriptiveNameEquality(String descriptiveNameForDisplay, String origDesc,
322      byte[] startKey) {
323    // except for the "hidden-start-key" substring everything else should exactly match
324    String firstPart = descriptiveNameForDisplay.substring(0,
325        descriptiveNameForDisplay.indexOf(new String(HRegionInfo.HIDDEN_START_KEY)));
326    String secondPart = descriptiveNameForDisplay.substring(
327        descriptiveNameForDisplay.indexOf(new String(HRegionInfo.HIDDEN_START_KEY)) +
328        HRegionInfo.HIDDEN_START_KEY.length);
329    String firstPartOrig = origDesc.substring(0,
330        origDesc.indexOf(Bytes.toStringBinary(startKey)));
331    String secondPartOrig = origDesc.substring(
332        origDesc.indexOf(Bytes.toStringBinary(startKey)) +
333        Bytes.toStringBinary(startKey).length());
334    assert(firstPart.equals(firstPartOrig));
335    assert(secondPart.equals(secondPartOrig));
336  }
337
338  private void checkEquality(HRegionInfo h, Configuration conf) throws IOException {
339    byte[] modifiedRegionName = HRegionInfo.getRegionNameForDisplay(h, conf);
340    byte[][] modifiedRegionNameParts = HRegionInfo.parseRegionName(modifiedRegionName);
341    byte[][] regionNameParts = HRegionInfo.parseRegionName(h.getRegionName());
342
343    //same number of parts
344    assert(modifiedRegionNameParts.length == regionNameParts.length);
345
346    for (int i = 0; i < regionNameParts.length; i++) {
347      // all parts should match except for [1] where in the modified one,
348      // we should have "hidden_start_key"
349      if (i != 1) {
350        Assert.assertArrayEquals(regionNameParts[i], modifiedRegionNameParts[i]);
351      } else {
352        assertNotEquals(regionNameParts[i][0], modifiedRegionNameParts[i][0]);
353        Assert.assertArrayEquals(modifiedRegionNameParts[1],
354            HRegionInfo.getStartKeyForDisplay(h, conf));
355      }
356    }
357  }
