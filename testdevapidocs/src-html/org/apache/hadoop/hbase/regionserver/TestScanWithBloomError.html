060/**
061 * Test a multi-column scanner when there is a Bloom filter false-positive.
062 * This is needed for the multi-column Bloom filter optimization.
063 */
064@RunWith(Parameterized.class)
065@Category({RegionServerTests.class, SmallTests.class})
066public class TestScanWithBloomError {
067
068  private static final Log LOG =
069    LogFactory.getLog(TestScanWithBloomError.class);
070
071  private static final String TABLE_NAME = "ScanWithBloomError";
072  private static final String FAMILY = "myCF";
073  private static final byte[] FAMILY_BYTES = Bytes.toBytes(FAMILY);
074  private static final String ROW = "theRow";
075  private static final String QUALIFIER_PREFIX = "qual";
076  private static final byte[] ROW_BYTES = Bytes.toBytes(ROW);
077  private static NavigableSet<Integer> allColIds = new TreeSet<>();
078  private Region region;
079  private BloomType bloomType;
080  private FileSystem fs;
081  private Configuration conf;
082
083  private final static HBaseTestingUtility TEST_UTIL = HBaseTestingUtility.createLocalHTU();
084
085  @Parameters
086  public static final Collection<Object[]> parameters() {
087    List<Object[]> configurations = new ArrayList<>();
088    for (BloomType bloomType : BloomType.values()) {
089      configurations.add(new Object[] { bloomType });
090    }
091    return configurations;
092  }
093
094  public TestScanWithBloomError(BloomType bloomType) {
095    this.bloomType = bloomType;
096  }
097
098  @Before
099  public void setUp() throws IOException{
100    conf = TEST_UTIL.getConfiguration();
101    fs = FileSystem.get(conf);
102  }
103
104  @Test
105  public void testThreeStoreFiles() throws IOException {
106    region = TEST_UTIL.createTestRegion(TABLE_NAME,
107        new HColumnDescriptor(FAMILY)
108            .setCompressionType(Compression.Algorithm.GZ)
109            .setBloomFilterType(bloomType)
110            .setMaxVersions(TestMultiColumnScanner.MAX_VERSIONS));
111    createStoreFile(new int[] {1, 2, 6});
112    createStoreFile(new int[] {1, 2, 3, 7});
113    createStoreFile(new int[] {1, 9});
114    scanColSet(new int[]{1, 4, 6, 7}, new int[]{1, 6, 7});
115
116    HBaseTestingUtility.closeRegionAndWAL(region);
117  }
118
119  private void scanColSet(int[] colSet, int[] expectedResultCols)
120      throws IOException {
121    LOG.info("Scanning column set: " + Arrays.toString(colSet));
122    Scan scan = new Scan(ROW_BYTES, ROW_BYTES);
123    addColumnSetToScan(scan, colSet);
124    RegionScannerImpl scanner = (RegionScannerImpl) region.getScanner(scan);
125    KeyValueHeap storeHeap = scanner.getStoreHeapForTesting();
126    assertEquals(0, storeHeap.getHeap().size());
127    StoreScanner storeScanner =
128        (StoreScanner) storeHeap.getCurrentForTesting();
129    @SuppressWarnings({ "unchecked", "rawtypes" })
130    List<StoreFileScanner> scanners = (List<StoreFileScanner>)
131        (List) storeScanner.getAllScannersForTesting();
132
133    // Sort scanners by their HFile's modification time.
134    Collections.sort(scanners, new Comparator<StoreFileScanner>() {
135      @Override
136      public int compare(StoreFileScanner s1, StoreFileScanner s2) {
137        Path p1 = s1.getReader().getHFileReader().getPath();
138        Path p2 = s2.getReader().getHFileReader().getPath();
139        long t1, t2;
140        try {
141          t1 = fs.getFileStatus(p1).getModificationTime();
142          t2 = fs.getFileStatus(p2).getModificationTime();
143        } catch (IOException ex) {
144          throw new RuntimeException(ex);
145        }
146        return t1 < t2 ? -1 : t1 == t2 ? 1 : 0;
147      }
148    });
149
150    StoreFileReader lastStoreFileReader = null;
151    for (StoreFileScanner sfScanner : scanners)
152      lastStoreFileReader = sfScanner.getReader();
153
154    new HFilePrettyPrinter(conf).run(new String[]{ "-m", "-p", "-f",
155        lastStoreFileReader.getHFileReader().getPath().toString()});
156
157    // Disable Bloom filter for the last store file. The disabled Bloom filter
158    // will always return "true".
159    LOG.info("Disabling Bloom filter for: "
160        + lastStoreFileReader.getHFileReader().getName());
161    lastStoreFileReader.disableBloomFilterForTesting();
162
163    List<Cell> allResults = new ArrayList<>();
164
165    { // Limit the scope of results.
166      List<Cell> results = new ArrayList<>();
167      while (scanner.next(results) || results.size() > 0) {
168        allResults.addAll(results);
169        results.clear();
170      }
171    }
172
173    List<Integer> actualIds = new ArrayList<>();
174    for (Cell kv : allResults) {
175      String qual = Bytes.toString(CellUtil.cloneQualifier(kv));
176      assertTrue(qual.startsWith(QUALIFIER_PREFIX));
177      actualIds.add(Integer.valueOf(qual.substring(
178          QUALIFIER_PREFIX.length())));
179    }
180    List<Integer> expectedIds = new ArrayList<>();
181    for (int expectedId : expectedResultCols)
182      expectedIds.add(expectedId);
183
184    LOG.info("Column ids returned: " + actualIds + ", expected: "
185        + expectedIds);
186    assertEquals(expectedIds.toString(), actualIds.toString());
187  }
188
189  private void addColumnSetToScan(Scan scan, int[] colIds) {
190    for (int colId : colIds) {
191      scan.addColumn(FAMILY_BYTES,
192          Bytes.toBytes(qualFromId(colId)));
193    }
194  }
195
196  private String qualFromId(int colId) {
197    return QUALIFIER_PREFIX + colId;
198  }
199
200  private void createStoreFile(int[] colIds)
201      throws IOException {
202    Put p = new Put(ROW_BYTES);
203    for (int colId : colIds) {
204      long ts = Long.MAX_VALUE;
205      String qual = qualFromId(colId);
206      allColIds.add(colId);
207      KeyValue kv = KeyValueTestUtil.create(ROW, FAMILY,
208          qual, ts, TestMultiColumnScanner.createValue(ROW, qual, ts));
209      p.add(kv);
210    }
211    region.put(p);
212    region.flush(true);
213  }
