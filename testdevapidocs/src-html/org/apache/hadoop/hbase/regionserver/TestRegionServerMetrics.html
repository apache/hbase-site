106  @Test
107  public void testRequestCount() throws Exception {
108    String tableNameString = "testRequestCount";
109    TableName tName = TableName.valueOf(tableNameString);
110    byte[] cfName = Bytes.toBytes("d");
111    byte[] row = Bytes.toBytes("rk");
112    byte[] qualifier = Bytes.toBytes("qual");
113    byte[] initValue = Bytes.toBytes("Value");
114
115    TEST_UTIL.createTable(tName, cfName);
116
117    Connection connection = TEST_UTIL.getConnection();
118    connection.getTable(tName).close(); //wait for the table to come up.
119
120    // Do a first put to be sure that the connection is established, meta is there and so on.
121    Table table = connection.getTable(tName);
122    Put p = new Put(row);
123    p.addColumn(cfName, qualifier, initValue);
124    table.put(p);
125
126    metricsRegionServer.getRegionServerWrapper().forceRecompute();
127    long requests = metricsHelper.getCounter("totalRequestCount", serverSource);
128    long readRequests = metricsHelper.getCounter("readRequestCount", serverSource);
129    long writeRequests = metricsHelper.getCounter("writeRequestCount", serverSource);
130
131    for (int i=0; i< 30; i++) {
132      table.put(p);
133    }
134
135    metricsRegionServer.getRegionServerWrapper().forceRecompute();
136    metricsHelper.assertCounter("totalRequestCount", requests + 30, serverSource);
137    metricsHelper.assertCounter("readRequestCount", readRequests, serverSource);
138    metricsHelper.assertCounter("writeRequestCount", writeRequests + 30, serverSource);
139
140    Get g = new Get(row);
141    for (int i=0; i< 10; i++) {
142      table.get(g);
143    }
144
145    metricsRegionServer.getRegionServerWrapper().forceRecompute();
146    metricsHelper.assertCounter("totalRequestCount", requests + 40, serverSource);
147    metricsHelper.assertCounter("readRequestCount", readRequests + 10, serverSource);
148    metricsHelper.assertCounter("writeRequestCount", writeRequests + 30, serverSource);
149
150    try (RegionLocator locator = connection.getRegionLocator(tName)) {
151      for ( HRegionLocation location: locator.getAllRegionLocations()) {
152        HRegionInfo i = location.getRegionInfo();
153        MetricsRegionAggregateSource agg = rs.getRegion(i.getRegionName())
154            .getMetrics()
155            .getSource()
156            .getAggregateSource();
157        String prefix = "namespace_"+NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR+
158            "_table_"+tableNameString +
159            "_region_" + i.getEncodedName()+
160            "_metric";
161        metricsHelper.assertCounter(prefix + "_getNumOps", 10, agg);
162        metricsHelper.assertCounter(prefix + "_mutateCount", 31, agg);
163      }
164    }
165    List<Get> gets = new ArrayList<Get>();
166    for (int i=0; i< 10; i++) {
167      gets.add(new Get(row));
168    }
169    table.get(gets);
170
171    metricsRegionServer.getRegionServerWrapper().forceRecompute();
172    metricsHelper.assertCounter("totalRequestCount", requests + 50, serverSource);
173    metricsHelper.assertCounter("readRequestCount", readRequests + 20, serverSource);
174    metricsHelper.assertCounter("writeRequestCount", writeRequests + 30, serverSource);
175
176    List<Put> puts = new ArrayList<>();
177    for (int i=0; i< 30; i++) {
178      puts.add(p);
179    }
180    table.put(puts);
181
182    metricsRegionServer.getRegionServerWrapper().forceRecompute();
183    metricsHelper.assertCounter("totalRequestCount", requests + 80, serverSource);
184    metricsHelper.assertCounter("readRequestCount", readRequests + 20, serverSource);
185    metricsHelper.assertCounter("writeRequestCount", writeRequests + 60, serverSource);
186
187    table.close();
188  }

450  @Test
451  public void testMobMetrics() throws IOException, InterruptedException {
452    String tableNameString = "testMobMetrics";
453    TableName tableName = TableName.valueOf(tableNameString);
454    byte[] cf = Bytes.toBytes("d");
455    byte[] qualifier = Bytes.toBytes("qual");
456    byte[] val = Bytes.toBytes("mobdata");
457    int numHfiles = conf.getInt("hbase.hstore.compactionThreshold", 3) - 1;
458    HTableDescriptor htd = new HTableDescriptor(tableName);
459    HColumnDescriptor hcd = new HColumnDescriptor(cf);
460    hcd.setMobEnabled(true);
461    hcd.setMobThreshold(0);
462    htd.addFamily(hcd);
463    Connection connection = ConnectionFactory.createConnection(conf);
464    Admin admin = connection.getAdmin();
465    HTable t = TEST_UTIL.createTable(htd, new byte[0][0], conf);
466    Region region = rs.getOnlineRegions(tableName).get(0);
467    t.setAutoFlush(true, true);
468    for (int insertCount = 0; insertCount < numHfiles; insertCount++) {
469      Put p = new Put(Bytes.toBytes(insertCount));
470      p.addColumn(cf, qualifier, val);
471      t.put(p);
472      admin.flush(tableName);
473    }
474    metricsRegionServer.getRegionServerWrapper().forceRecompute();
475    metricsHelper.assertCounter("mobFlushCount", numHfiles, serverSource);
476    Scan scan = new Scan(Bytes.toBytes(0), Bytes.toBytes(2));
477    ResultScanner scanner = t.getScanner(scan);
478    scanner.next(100);
479    numScanNext++;  // this is an ugly construct
480    scanner.close();
481    metricsRegionServer.getRegionServerWrapper().forceRecompute();
482    metricsHelper.assertCounter("mobScanCellsCount", 2, serverSource);
483    region.getTableDesc().getFamily(cf).setMobThreshold(100);
484    ((HRegion)region).initialize();
485    region.compact(true);
486    metricsRegionServer.getRegionServerWrapper().forceRecompute();
487    metricsHelper.assertCounter("cellsCountCompactedFromMob", numHfiles,
488        serverSource);
489    metricsHelper.assertCounter("cellsCountCompactedToMob", 0, serverSource);
490    scanner = t.getScanner(scan);
491    scanner.next(100);
492    numScanNext++;  // this is an ugly construct
493    metricsRegionServer.getRegionServerWrapper().forceRecompute();
494    // metrics are reset by the region initialization
495    metricsHelper.assertCounter("mobScanCellsCount", 0, serverSource);
496    for (int insertCount = numHfiles;
497        insertCount < 2 * numHfiles - 1; insertCount++) {
498      Put p = new Put(Bytes.toBytes(insertCount));
499      p.addColumn(cf, qualifier, val);
500      t.put(p);
501      admin.flush(tableName);
502    }
503    region.getTableDesc().getFamily(cf).setMobThreshold(0);
504    ((HRegion)region).initialize();
505    region.compact(true);
506    metricsRegionServer.getRegionServerWrapper().forceRecompute();
507    // metrics are reset by the region initialization
508    metricsHelper.assertCounter("cellsCountCompactedFromMob", 0, serverSource);
509    metricsHelper.assertCounter("cellsCountCompactedToMob", 2 * numHfiles - 1,
510        serverSource);
511    t.close();
512    admin.close();
513    connection.close();
514  }

516  @Test
517  @Ignore
518  public void testRangeCountMetrics() throws Exception {
519    String tableNameString = "testRangeCountMetrics";
520    final long[] timeranges =
521        { 1, 3, 10, 30, 100, 300, 1000, 3000, 10000, 30000, 60000, 120000, 300000, 600000 };
522    final String timeRangeType = "TimeRangeCount";
523    final String timeRangeMetricName = "Mutate";
524    boolean timeRangeCountUpdated = false;
525
526    TableName tName = TableName.valueOf(tableNameString);
527    byte[] cfName = Bytes.toBytes("d");
528    byte[] row = Bytes.toBytes("rk");
529    byte[] qualifier = Bytes.toBytes("qual");
530    byte[] initValue = Bytes.toBytes("Value");
531
532    TEST_UTIL.createTable(tName, cfName);
533
534    Connection connection = TEST_UTIL.getConnection();
535    connection.getTable(tName).close(); // wait for the table to come up.
536
537    // Do a first put to be sure that the connection is established, meta is there and so on.
538    Table table = connection.getTable(tName);
539    Put p = new Put(row);
540    p.addColumn(cfName, qualifier, initValue);
541    table.put(p);
542
543    // do some puts and gets
544    for (int i = 0; i < 10; i++) {
545      table.put(p);
546    }
547
548    Get g = new Get(row);
549    for (int i = 0; i < 10; i++) {
550      table.get(g);
551    }
552
553    metricsRegionServer.getRegionServerWrapper().forceRecompute();
554
555    // Check some time range counters were updated
556    long prior = 0;
557
558    String dynamicMetricName;
559    for (int i = 0; i < timeranges.length; i++) {
560      dynamicMetricName =
561          timeRangeMetricName + "_" + timeRangeType + "_" + prior + "-" + timeranges[i];
562      if (metricsHelper.checkCounterExists(dynamicMetricName, serverSource)) {
563        long count = metricsHelper.getGaugeLong(dynamicMetricName, serverSource);
564        if (count > 0) {
565          timeRangeCountUpdated = true;
566          break;
567        }
568      }
569      prior = timeranges[i];
570    }
571    dynamicMetricName =
572        timeRangeMetricName + "_" + timeRangeType + "_" + timeranges[timeranges.length - 1] + "-inf";
573    if (metricsHelper.checkCounterExists(dynamicMetricName, serverSource)) {
574      long count = metricsHelper.getCounter(dynamicMetricName, serverSource);
575      if (count > 0) {
576        timeRangeCountUpdated = true;
577      }
578    }
579    assertEquals(true, timeRangeCountUpdated);
580
581    table.close();
582  }
