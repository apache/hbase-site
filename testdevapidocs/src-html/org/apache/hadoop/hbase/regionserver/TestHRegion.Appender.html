460  /**
461   * Test we do not lose data if we fail a flush and then close.
462   * Part of HBase-10466.  Tests the following from the issue description:
463   * "Bug 1: Wrong calculation of HRegion.memstoreSize: When a flush fails, data to be flushed is
464   * kept in each MemStore's snapshot and wait for next flush attempt to continue on it. But when
465   * the next flush succeeds, the counter of total memstore size in HRegion is always deduced by
466   * the sum of current memstore sizes instead of snapshots left from previous failed flush. This
467   * calculation is problematic that almost every time there is failed flush, HRegion.memstoreSize
468   * gets reduced by a wrong value. If region flush could not proceed for a couple cycles, the size
469   * in current memstore could be much larger than the snapshot. It's likely to drift memstoreSize
470   * much smaller than expected. In extreme case, if the error accumulates to even bigger than
471   * HRegion's memstore size limit, any further flush is skipped because flush does not do anything
472   * if memstoreSize is not larger than 0."
473   * @throws Exception
474   */
475  @Test
476  public void testFlushSizeAccounting() throws Exception {
477    final Configuration conf = HBaseConfiguration.create(CONF);
478    final WAL wal = createWALCompatibleWithFaultyFileSystem(method, conf, tableName);
479    // Only retry once.
480    conf.setInt("hbase.hstore.flush.retries.number", 1);
481    final User user =
482      User.createUserForTesting(conf, method, new String[]{"foo"});
483    // Inject our faulty LocalFileSystem
484    conf.setClass("fs.file.impl", FaultyFileSystem.class, FileSystem.class);
485    user.runAs(new PrivilegedExceptionAction<Object>() {
486      @Override
487      public Object run() throws Exception {
488        // Make sure it worked (above is sensitive to caching details in hadoop core)
489        FileSystem fs = FileSystem.get(conf);
490        Assert.assertEquals(FaultyFileSystem.class, fs.getClass());
491        FaultyFileSystem ffs = (FaultyFileSystem)fs;
492        HRegion region = null;
493        try {
494          // Initialize region
495          region = initHRegion(tableName, null, null, false, Durability.SYNC_WAL, wal,
496              COLUMN_FAMILY_BYTES);
497          long size = region.getMemStoreSize();
498          Assert.assertEquals(0, size);
499          // Put one item into memstore.  Measure the size of one item in memstore.
500          Put p1 = new Put(row);
501          p1.add(new KeyValue(row, COLUMN_FAMILY_BYTES, qual1, 1, (byte[]) null));
502          region.put(p1);
503          final long sizeOfOnePut = region.getMemStoreSize();
504          // Fail a flush which means the current memstore will hang out as memstore 'snapshot'.
505          try {
506            LOG.info("Flushing");
507            region.flush(true);
508            Assert.fail("Didn't bubble up IOE!");
509          } catch (DroppedSnapshotException dse) {
510            // What we are expecting
511            region.closing.set(false); // this is needed for the rest of the test to work
512          }
513          // Make it so all writes succeed from here on out
514          ffs.fault.set(false);
515          // Check sizes.  Should still be the one entry.
