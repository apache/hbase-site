018package org.apache.hadoop.hbase.types;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023import org.apache.hadoop.hbase.Cell;
024import org.apache.hadoop.hbase.CellBuilderType;
025import org.apache.hadoop.hbase.CellUtil;
026import org.apache.hadoop.hbase.ExtendedCellBuilderFactory;
027import org.apache.hadoop.hbase.KeyValue;
028import org.apache.hadoop.hbase.testclassification.MiscTests;
029import org.apache.hadoop.hbase.testclassification.SmallTests;
030import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
031import org.apache.hadoop.hbase.protobuf.generated.CellProtos;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.hbase.util.PositionedByteRange;
034import org.apache.hadoop.hbase.util.SimplePositionedByteRange;
035import org.junit.Test;
036import org.junit.experimental.categories.Category;
037
038@Category({SmallTests.class, MiscTests.class})
039public class TestPBCell {
040
041  private static final PBCell CODEC = new PBCell();
042
043  /**
044   * Basic test to verify utility methods in {@link PBType} and delegation to protobuf works.
045   */
046  @Test
047  public void testRoundTrip() {
048    final Cell cell = new KeyValue(Bytes.toBytes("row"), Bytes.toBytes("fam"),
049        Bytes.toBytes("qual"), Bytes.toBytes("val"));
050    CellProtos.Cell c = ProtobufUtil.toCell(cell), decoded;
051    PositionedByteRange pbr = new SimplePositionedByteRange(c.getSerializedSize());
052    pbr.setPosition(0);
053    int encodedLength = CODEC.encode(pbr, c);
054    pbr.setPosition(0);
055    decoded = CODEC.decode(pbr);
056    assertEquals(encodedLength, pbr.getPosition());
057    assertTrue(CellUtil.equals(cell, ProtobufUtil
058        .toCell(ExtendedCellBuilderFactory.create(CellBuilderType.SHALLOW_COPY), decoded)));
059  }
060}
