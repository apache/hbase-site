065  @BeforeClass
066  public static void setUpBeforeClass() throws Exception {
067    Configuration conf = HBaseConfiguration.create();
068    conf.set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY, MyObserver.class.getName());
069    UTIL = new HBaseTestingUtility(conf);
070    UTIL.startMiniCluster(MASTERS, SLAVES);
071    CLUSTER = UTIL.getHBaseCluster();
072    CLUSTER.waitForActiveAndReadyMaster();
073    ADMIN = UTIL.getAdmin();
074    // Kill one region server
075    List<RegionServerThread> rsts = CLUSTER.getLiveRegionServerThreads();
076    RegionServerThread rst = rsts.get(rsts.size() - 1);
077    DEAD = rst.getRegionServer();
078    DEAD.stop("Test dead servers status");
079    while (rst.isAlive()) {
080      Thread.sleep(500);
081    }
082  }

084  @Test

089    Assert.assertEquals(origin.getClusterId(), defaults.getClusterId());
090    Assert.assertTrue(origin.getAverageLoad() == defaults.getAverageLoad());
091    Assert.assertTrue(origin.getBackupMastersSize() == defaults.getBackupMastersSize());
092    Assert.assertTrue(origin.getDeadServersSize() == defaults.getDeadServersSize());
093    Assert.assertTrue(origin.getRegionsCount() == defaults.getRegionsCount());
094    Assert.assertTrue(origin.getServersSize() == defaults.getServersSize());
095    Assert.assertTrue(origin.getMasterInfoPort() == defaults.getMasterInfoPort());
096    Assert.assertTrue(origin.equals(defaults));
097  }

130  @Test
131  public void testLiveAndDeadServersStatus() throws Exception {
132    // Count the number of live regionservers
133    List<RegionServerThread> regionserverThreads = CLUSTER.getLiveRegionServerThreads();
134    int numRs = 0;
135    int len = regionserverThreads.size();
136    for (int i = 0; i < len; i++) {
137      if (regionserverThreads.get(i).isAlive()) {
138        numRs++;
139      }
140    }
141    // Depending on the (random) order of unit execution we may run this unit before the
142    // minicluster is fully up and recovered from the RS shutdown done during test init.
143    Waiter.waitFor(CLUSTER.getConfiguration(), 10 * 1000, 100, new Predicate<Exception>() {
144      @Override
145      public boolean evaluate() throws Exception {
146        ClusterStatus status = ADMIN.getClusterStatus(EnumSet.of(Option.LIVE_SERVERS));
147        Assert.assertNotNull(status);
148        return status.getRegionsCount() > 0;
149      }
150    });
151    // Retrieve live servers and dead servers info.
152    EnumSet<Option> options = EnumSet.of(Option.LIVE_SERVERS, Option.DEAD_SERVERS);
153    ClusterStatus status = ADMIN.getClusterStatus(options);
154    Assert.assertNotNull(status);
155    Assert.assertNotNull(status.getServers());
156    // exclude a dead region server
157    Assert.assertEquals(SLAVES -1, numRs);
158    // live servers = nums of regionservers
159    // By default, HMaster don't carry any regions so it won't report its load.
160    // Hence, it won't be in the server list.
161    Assert.assertEquals(status.getServers().size(), numRs);
162    Assert.assertTrue(status.getRegionsCount() > 0);
163    Assert.assertNotNull(status.getDeadServerNames());
164    Assert.assertEquals(1, status.getDeadServersSize());
165    ServerName deadServerName = status.getDeadServerNames().iterator().next();
166    Assert.assertEquals(DEAD.getServerName(), deadServerName);
167  }

169  @Test
170  public void testMasterAndBackupMastersStatus() throws Exception {
171    // get all the master threads
172    List<MasterThread> masterThreads = CLUSTER.getMasterThreads();
173    int numActive = 0;
174    int activeIndex = 0;
175    ServerName activeName = null;
176    HMaster active = null;
177    for (int i = 0; i < masterThreads.size(); i++) {
178      if (masterThreads.get(i).getMaster().isActiveMaster()) {
179        numActive++;
180        activeIndex = i;
181        active = masterThreads.get(activeIndex).getMaster();
182        activeName = active.getServerName();
183      }
184    }
185    Assert.assertNotNull(active);
186    Assert.assertEquals(1, numActive);
187    Assert.assertEquals(MASTERS, masterThreads.size());
188    // Retrieve master and backup masters infos only.
189    EnumSet<Option> options = EnumSet.of(Option.MASTER, Option.BACKUP_MASTERS);
190    ClusterStatus status = ADMIN.getClusterStatus(options);
191    Assert.assertTrue(status.getMaster().equals(activeName));
192    Assert.assertEquals(MASTERS - 1, status.getBackupMastersSize());
193  }

214  @Test
215  public void testObserver() throws IOException {
216    int preCount = MyObserver.PRE_COUNT.get();
217    int postCount = MyObserver.POST_COUNT.get();
218    Assert.assertTrue(Stream.of(ADMIN.getClusterStatus().getMasterCoprocessors())
219        .anyMatch(s -> s.equals(MyObserver.class.getSimpleName())));
220    Assert.assertEquals(preCount + 1, MyObserver.PRE_COUNT.get());
221    Assert.assertEquals(postCount + 1, MyObserver.POST_COUNT.get());
222  }
223
224  public static class MyObserver implements MasterCoprocessor, MasterObserver {
225    private static final AtomicInteger PRE_COUNT = new AtomicInteger(0);
226    private static final AtomicInteger POST_COUNT = new AtomicInteger(0);
227
228    @Override public Optional<MasterObserver> getMasterObserver() {
229      return Optional.of(this);
230    }
231
232    @Override public void preGetClusterStatus(ObserverContext<MasterCoprocessorEnvironment> ctx)
233        throws IOException {
234      PRE_COUNT.incrementAndGet();
235    }
236
237    @Override public void postGetClusterStatus(ObserverContext<MasterCoprocessorEnvironment> ctx,
238        ClusterStatus status) throws IOException {
239      POST_COUNT.incrementAndGet();
240    }
241  }
