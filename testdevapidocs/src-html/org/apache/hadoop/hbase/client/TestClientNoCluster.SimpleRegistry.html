186    localConfig.setInt("hbase.client.pause", pause);
187    localConfig.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 10);
188    // Set the operation timeout to be < the pause.  Expectation is that after first pause, we will
189    // fail out of the rpc because the rpc timeout will have been set to the operation tiemout

191    // all ten of them -- and we'll get the RetriesExhaustedException exception.
192    localConfig.setInt(HConstants.HBASE_CLIENT_META_OPERATION_TIMEOUT, pause - 1);
193    Connection connection = ConnectionFactory.createConnection(localConfig);
194    Table table = connection.getTable(TableName.META_TABLE_NAME);
195    Throwable t = null;
196    try {
197      // An exists call turns into a get w/ a flag.
198      table.exists(new Get(Bytes.toBytes("abc")));
199    } catch (SocketTimeoutException e) {
200      // I expect this exception.
201      LOG.info("Got expected exception", e);

203    } catch (RetriesExhaustedException e) {
204      // This is the old, unwanted behavior.  If we get here FAIL!!!

370  static MultiResponse doMultiResponse(final SortedMap<byte [], Pair<HRegionInfo, ServerName>> meta,
371      final AtomicLong sequenceids, final MultiRequest request) {
372    // Make a response to match the request.  Act like there were no failures.
373    ClientProtos.MultiResponse.Builder builder = ClientProtos.MultiResponse.newBuilder();
374    // Per Region.
375    RegionActionResult.Builder regionActionResultBuilder =
376        RegionActionResult.newBuilder();
377    ResultOrException.Builder roeBuilder = ResultOrException.newBuilder();
378    for (RegionAction regionAction: request.getRegionActionList()) {
379      regionActionResultBuilder.clear();
380      // Per Action in a Region.
381      for (ClientProtos.Action action: regionAction.getActionList()) {
382        roeBuilder.clear();
383        // Return empty Result and proper index as result.
384        roeBuilder.setResult(ClientProtos.Result.getDefaultInstance());
385        roeBuilder.setIndex(action.getIndex());
386        regionActionResultBuilder.addResultOrException(roeBuilder.build());
387      }
388      builder.addRegionActionResult(regionActionResultBuilder.build());
389    }
390    return builder.build();
391  }

499  static ScanResponse doMetaScanResponse(final SortedMap<byte [], Pair<HRegionInfo, ServerName>> meta,
500      final AtomicLong sequenceids, final ScanRequest request) {
501    ScanResponse.Builder builder = ScanResponse.newBuilder();
502    int max = request.getNumberOfRows();
503    int count = 0;
504    Map<byte [], Pair<HRegionInfo, ServerName>> tail =
505      request.hasScan()? meta.tailMap(request.getScan().getStartRow().toByteArray()): meta;
506      ClientProtos.Result.Builder resultBuilder = ClientProtos.Result.newBuilder();
507    for (Map.Entry<byte [], Pair<HRegionInfo, ServerName>> e: tail.entrySet()) {
508      // Can be 0 on open of a scanner -- i.e. rpc to setup scannerid only.
509      if (max <= 0) break;
510      if (++count > max) break;
511      HRegionInfo hri = e.getValue().getFirst();
512      ByteString row = ByteStringer.wrap(hri.getRegionName());
513      resultBuilder.clear();
514      resultBuilder.addCell(getRegionInfo(row, hri));
515      resultBuilder.addCell(getServer(row, e.getValue().getSecond()));
516      resultBuilder.addCell(getStartCode(row));
517      builder.addResults(resultBuilder.build());
518      // Set more to false if we are on the last region in table.
519      if (hri.getEndKey().length <= 0) builder.setMoreResults(false);
520      else builder.setMoreResults(true);
521    }
522    // If no scannerid, set one.
523    builder.setScannerId(request.hasScannerId()?
524      request.getScannerId(): sequenceids.incrementAndGet());
525    return builder.build();
526  }

528  static GetResponse doMetaGetResponse(final SortedMap<byte [], Pair<HRegionInfo, ServerName>> meta,
529      final GetRequest request) {
530    ClientProtos.Result.Builder resultBuilder = ClientProtos.Result.newBuilder();
531    ByteString row = request.getGet().getRow();
532    Pair<HRegionInfo, ServerName> p = meta.get(row.toByteArray());
533    if (p != null) {
534      resultBuilder.addCell(getRegionInfo(row, p.getFirst()));
535      resultBuilder.addCell(getServer(row, p.getSecond()));
536    }
537    resultBuilder.addCell(getStartCode(row));
538    GetResponse.Builder builder = GetResponse.newBuilder();
539    builder.setResult(resultBuilder.build());
540    return builder.build();
541  }

548  static boolean isMetaRegion(final byte [] name, final RegionSpecifierType type) {
549    switch (type) {
550    case REGION_NAME:
551      return Bytes.equals(HRegionInfo.FIRST_META_REGIONINFO.getRegionName(), name);
552    case ENCODED_REGION_NAME:
553      return Bytes.equals(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes(), name);
554    default: throw new UnsupportedOperationException();
555    }

597  /**
598   * Format passed integer.  Zero-pad.
599   * Copied from hbase-server PE class and small amendment.  Make them share.
600   * @param number
601   * @return Returns zero-prefixed 10-byte wide decimal version of passed
602   * number (Does absolute in case number is negative).
603   */
604  private static byte [] format(final long number) {
605    byte [] b = new byte[10];
606    long d = number;
607    for (int i = b.length - 1; i >= 0; i--) {
608      b[i] = (byte)((d % 10) + '0');
609      d /= 10;
610    }
611    return b;
612  }

616   * @param namespaceSpan
617   * @return <code>count</code> regions
618   */
619  private static HRegionInfo [] makeHRegionInfos(final byte [] tableName, final int count,
620      final long namespaceSpan) {
621    byte [] startKey = HConstants.EMPTY_BYTE_ARRAY;
622    byte [] endKey = HConstants.EMPTY_BYTE_ARRAY;
623    long interval = namespaceSpan / count;
624    HRegionInfo [] hris = new HRegionInfo[count];
625    for (int i = 0; i < count; i++) {
626      if (i == 0) {
627        endKey = format(interval);
628      } else {
629        startKey = endKey;
630        if (i == count - 1) endKey = HConstants.EMPTY_BYTE_ARRAY;
631        else endKey = format((i + 1) * interval);
632      }
633      hris[i] = new HRegionInfo(TableName.valueOf(tableName), startKey, endKey);
634    }
635    return hris;
636  }

642  private static ServerName [] makeServerNames(final int count) {
643    ServerName [] sns = new ServerName[count];
644    for (int i = 0; i < count; i++) {
645      sns[i] = ServerName.valueOf("" + i + ".example.org", 16010, i);
646    }

650  /**
651   * Comparator for meta row keys.
652   */
653  private static class MetaRowsComparator implements Comparator<byte []> {
654    private final CellComparator delegate = CellComparator.META_COMPARATOR;
655    @Override
656    public int compare(byte[] left, byte[] right) {
657      return delegate.compareRows(new KeyValue.KeyOnlyKeyValue(left), right, 0, right.length);
658    }
659  }

661  /**
662   * Create up a map that is keyed by meta row name and whose value is the HRegionInfo and
663   * ServerName to return for this row.
664   * @return Map with faked hbase:meta content in it.
665   */
666  static SortedMap<byte [], Pair<HRegionInfo, ServerName>> makeMeta(final byte [] tableName,
667      final int regionCount, final long namespaceSpan, final int serverCount) {
668    // I need a comparator for meta rows so we sort properly.
669    SortedMap<byte [], Pair<HRegionInfo, ServerName>> meta =
670      new ConcurrentSkipListMap<byte[], Pair<HRegionInfo,ServerName>>(new MetaRowsComparator());
671    HRegionInfo [] hris = makeHRegionInfos(tableName, regionCount, namespaceSpan);
672    ServerName [] serverNames = makeServerNames(serverCount);
673    int per = regionCount / serverCount;
674    int count = 0;
675    for (HRegionInfo hri: hris) {
676      Pair<HRegionInfo, ServerName> p =
677        new Pair<HRegionInfo, ServerName>(hri, serverNames[count++ / per]);
678      meta.put(hri.getRegionName(), p);
679    }
680    return meta;
681  }

683  /**
684   * Code for each 'client' to run.
685   *
686   * @param id
687   * @param c
688   * @param sharedConnection
689   * @throws IOException
690   */
691  static void cycle(int id, final Configuration c, final Connection sharedConnection) throws IOException {
692    long namespaceSpan = c.getLong("hbase.test.namespace.span", 1000000);
693    long startTime = System.currentTimeMillis();
694    final int printInterval = 100000;
695    Random rd = new Random(id);
696    boolean get = c.getBoolean("hbase.test.do.gets", false);
697    TableName tableName = TableName.valueOf(BIG_USER_TABLE);
698    if (get) {
699      try (Table table = sharedConnection.getTable(tableName)){
700        Stopwatch stopWatch = new Stopwatch();
701        stopWatch.start();
702        for (int i = 0; i < namespaceSpan; i++) {
703          byte [] b = format(rd.nextLong());
704          Get g = new Get(b);
705          table.get(g);
706          if (i % printInterval == 0) {
707            LOG.info("Get " + printInterval + "/" + stopWatch.elapsedMillis());
708            stopWatch.reset();
709            stopWatch.start();
710          }
711        }
712        LOG.info("Finished a cycle putting " + namespaceSpan + " in " +
713            (System.currentTimeMillis() - startTime) + "ms");
714      }
715    } else {
716      try (BufferedMutator mutator = sharedConnection.getBufferedMutator(tableName)) {
717        Stopwatch stopWatch = new Stopwatch();
718        stopWatch.start();
719        for (int i = 0; i < namespaceSpan; i++) {
720          byte [] b = format(rd.nextLong());
721          Put p = new Put(b);
722          p.addColumn(HConstants.CATALOG_FAMILY, b, b);
723          mutator.mutate(p);
724          if (i % printInterval == 0) {
725            LOG.info("Put " + printInterval + "/" + stopWatch.elapsedMillis());
726            stopWatch.reset();
727            stopWatch.start();
728          }
729        }
730        LOG.info("Finished a cycle putting " + namespaceSpan + " in " +
731            (System.currentTimeMillis() - startTime) + "ms");
732        }
733    }
734  }

736  @Override
737  public int run(String[] arg0) throws Exception {
738    int errCode = 0;
739    // TODO: Make command options.
740    // How many servers to fake.
741    final int servers = 1;
742    // How many regions to put on the faked servers.
743    final int regions = 100000;
744    // How many 'keys' in the faked regions.
745    final long namespaceSpan = 50000000;
746    // How long to take to pause after doing a put; make this long if you want to fake a struggling
747    // server.
748    final long multiPause = 0;
749    // Check args make basic sense.
750    if ((namespaceSpan < regions) || (regions < servers)) {
751      throw new IllegalArgumentException("namespaceSpan=" + namespaceSpan + " must be > regions=" +
752        regions + " which must be > servers=" + servers);
753    }
754
755    // Set my many servers and many regions faking connection in place.
756    getConf().set("hbase.client.connection.impl",
757      ManyServersManyRegionsConnection.class.getName());
758    // Use simple kv registry rather than zk
759    getConf().set("hbase.client.registry.impl", SimpleRegistry.class.getName());
760    // When to report fails.  Default is we report the 10th.  This means we'll see log everytime
761    // an exception is thrown -- usually RegionTooBusyException when we have more than
762    // hbase.test.multi.too.many requests outstanding at any time.
763    getConf().setInt("hbase.client.start.log.errors.counter", 0);
764
765    // Ugly but this is only way to pass in configs.into ManyServersManyRegionsConnection class.
766    getConf().setInt("hbase.test.regions", regions);
767    getConf().setLong("hbase.test.namespace.span", namespaceSpan);
768    getConf().setLong("hbase.test.servers", servers);
769    getConf().set("hbase.test.tablename", Bytes.toString(BIG_USER_TABLE));
770    getConf().setLong("hbase.test.multi.pause.when.done", multiPause);
771    // Let there be ten outstanding requests at a time before we throw RegionBusyException.
772    getConf().setInt("hbase.test.multi.too.many", 10);
773    final int clients = 2;
774
775    // Have them all share the same connection so they all share the same instance of
776    // ManyServersManyRegionsConnection so I can keep an eye on how many requests by server.
777    final ExecutorService pool = Executors.newCachedThreadPool(Threads.getNamedThreadFactory("p"));
778      // Executors.newFixedThreadPool(servers * 10, Threads.getNamedThreadFactory("p"));
779    // Share a connection so I can keep counts in the 'server' on concurrency.
780    final Connection sharedConnection = ConnectionFactory.createConnection(getConf()/*, pool*/);
781    try {
782      Thread [] ts = new Thread[clients];
783      for (int j = 0; j < ts.length; j++) {
784        final int id = j;
785        ts[j] = new Thread("" + j) {
786          final Configuration c = getConf();
787
788          @Override
789          public void run() {
790            try {
791              cycle(id, c, sharedConnection);
792            } catch (IOException e) {
793              e.printStackTrace();
794            }
795          }
796        };
797        ts[j].start();
798      }
799      for (int j = 0; j < ts.length; j++) {
800        ts[j].join();
801      }
802    } finally {
803      sharedConnection.close();
804    }
805    return errCode;
806  }
