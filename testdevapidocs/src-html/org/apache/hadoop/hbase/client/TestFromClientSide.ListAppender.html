562  /*
563   * Wait on table split.  May return because we waited long enough on the split
564   * and it didn't happen.  Caller should check.
565   * @param t
566   * @return Map of table regions; caller needs to check table actually split.
567   */
568  private List<HRegionLocation> waitOnSplit(final Table t)
569  throws IOException {
570    try (RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(t.getName())) {
571      List<HRegionLocation> regions = locator.getAllRegionLocations();
572      int originalCount = regions.size();
573      for (int i = 0; i < TEST_UTIL.getConfiguration().getInt("hbase.test.retries", 30); i++) {
574        Thread.currentThread();
575        try {
576          Thread.sleep(1000);
577        } catch (InterruptedException e) {
578          e.printStackTrace();
579        }
580        regions = locator.getAllRegionLocations();
581        if (regions.size() > originalCount)
582          break;
