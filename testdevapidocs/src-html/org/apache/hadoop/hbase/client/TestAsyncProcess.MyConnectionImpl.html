328  class MyAsyncProcessWithReplicas extends MyAsyncProcess {
329    private Set<byte[]> failures = new TreeSet<byte[]>(new Bytes.ByteArrayComparator());
330    private long primarySleepMs = 0, replicaSleepMs = 0;
331    private Map<ServerName, Long> customPrimarySleepMs = new HashMap<ServerName, Long>();
332    private final AtomicLong replicaCalls = new AtomicLong(0);
333
334    public void addFailures(HRegionInfo... hris) {
335      for (HRegionInfo hri : hris) {
336        failures.add(hri.getRegionName());
337      }
338    }
339
340    public long getReplicaCallCount() {
341      return replicaCalls.get();
342    }
343
344    public void setPrimaryCallDelay(ServerName server, long primaryMs) {
345      customPrimarySleepMs.put(server, primaryMs);
346    }
347
348    public MyAsyncProcessWithReplicas(ClusterConnection hc, Configuration conf) {
349      super(hc, conf);
350    }
351
352    public void setCallDelays(long primaryMs, long replicaMs) {
353      this.primarySleepMs = primaryMs;
354      this.replicaSleepMs = replicaMs;
355    }
356
357    @Override
358    protected RpcRetryingCaller<AbstractResponse> createCaller(
359        CancellableRegionServerCallable payloadCallable, int rpcTimeout) {
360      MultiServerCallable<Row> callable = (MultiServerCallable) payloadCallable;
361      final MultiResponse mr = createMultiResponse(
362          callable.getMulti(), nbMultiResponse, nbActions, new ResponseGenerator() {
363            @Override
364            public void addResponse(MultiResponse mr, byte[] regionName, Action<Row> a) {
365              if (failures.contains(regionName)) {
366                mr.add(regionName, a.getOriginalIndex(), failure);
367              } else {
368                boolean isStale = !RegionReplicaUtil.isDefaultReplica(a.getReplicaId());
369                mr.add(regionName, a.getOriginalIndex(),
370                    Result.create(new Cell[0], null, isStale));
371              }
372            }
373          });
374      // Currently AsyncProcess either sends all-replica, or all-primary request.
375      final boolean isDefault = RegionReplicaUtil.isDefaultReplica(
376          callable.getMulti().actions.values().iterator().next().iterator().next().getReplicaId());
377      final ServerName server = ((MultiServerCallable<?>)callable).getServerName();
378      String debugMsg = "Call to " + server + ", primary=" + isDefault + " with "
379          + callable.getMulti().actions.size() + " entries: ";
380      for (byte[] region : callable.getMulti().actions.keySet()) {
381        debugMsg += "[" + Bytes.toStringBinary(region) + "], ";
382      }
383      LOG.debug(debugMsg);
384      if (!isDefault) {
385        replicaCalls.incrementAndGet();
386      }
387
388      return new RpcRetryingCallerImpl<AbstractResponse>(100, 10, 9) {
389        @Override
390        public MultiResponse callWithoutRetries(RetryingCallable<AbstractResponse> callable,
391                                                int callTimeout)
392        throws IOException, RuntimeException {
393          long sleep = -1;
394          if (isDefault) {
395            Long customSleep = customPrimarySleepMs.get(server);
396            sleep = (customSleep == null ? primarySleepMs : customSleep.longValue());
397          } else {
398            sleep = replicaSleepMs;
399          }
400          if (sleep != 0) {
401            try {
402              Thread.sleep(sleep);
403            } catch (InterruptedException e) {
404            }
405          }
406          return mr;
407        }
408      };
409    }
410  }

412  static MultiResponse createMultiResponse(final MultiAction<Row> multi,
413      AtomicInteger nbMultiResponse, AtomicInteger nbActions, ResponseGenerator gen) {
414    final MultiResponse mr = new MultiResponse();
415    nbMultiResponse.incrementAndGet();
416    for (Map.Entry<byte[], List<Action<Row>>> entry : multi.actions.entrySet()) {
417      byte[] regionName = entry.getKey();
418      for (Action<Row> a : entry.getValue()) {
419        nbActions.incrementAndGet();
420        gen.addResponse(mr, regionName, a);
421      }
422    }
423    return mr;
424  }
