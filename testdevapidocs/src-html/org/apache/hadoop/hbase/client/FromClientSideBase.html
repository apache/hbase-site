080  /**
081   * JUnit does not provide an easy way to run a hook after each parameterized run. Without that
082   * there is no easy way to restart the test cluster after each parameterized run. Annotation
083   * BeforeParam does not work either because it runs before parameterization and hence does not
084   * have access to the test parameters (which is weird).
085   *
086   * This *hack* checks if the current instance of test cluster configuration has the passed
087   * parameterized configs. In such a case, we can just reuse the cluster for test and do not need
088   * to initialize from scratch. While this is a hack, it saves a ton of time for the full
089   * test and de-flakes it.
090   */
091  protected static boolean isSameParameterizedCluster(Class<?> registryImpl, int numHedgedReqs) {
092    if (TEST_UTIL == null) {
093      return false;
094    }
095    Configuration conf = TEST_UTIL.getConfiguration();
096    Class<?> confClass = conf.getClass(HConstants.CLIENT_CONNECTION_REGISTRY_IMPL_CONF_KEY,
097      ZKConnectionRegistry.class);
098    int hedgedReqConfig = conf.getInt(MasterRegistry.MASTER_REGISTRY_HEDGED_REQS_FANOUT_KEY,
099      AbstractRpcBasedConnectionRegistry.HEDGED_REQS_FANOUT_DEFAULT);
100    return confClass.getName().equals(registryImpl.getName()) && numHedgedReqs == hedgedReqConfig;
101  }

103  protected static final void initialize(Class<?> registryImpl, int numHedgedReqs, Class<?>... cps)
104    throws Exception {
105    // initialize() is called for every unit test, however we only want to reset the cluster state
106    // at the end of every parameterized run.
107    if (isSameParameterizedCluster(registryImpl, numHedgedReqs)) {
108      return;
109    }
110    // Uncomment the following lines if more verbosity is needed for
111    // debugging (see HBASE-12285 for details).
112    // ((Log4JLogger)RpcServer.LOG).getLogger().setLevel(Level.ALL);
113    // ((Log4JLogger)RpcClient.LOG).getLogger().setLevel(Level.ALL);
114    // ((Log4JLogger)ScannerCallable.LOG).getLogger().setLevel(Level.ALL);
115    // make sure that we do not get the same ts twice, see HBASE-19731 for more details.
116    EnvironmentEdgeManager.injectEdge(new NonRepeatedEnvironmentEdge());
117    if (TEST_UTIL != null) {
118      // We reached end of a parameterized run, clean up.
119      TEST_UTIL.shutdownMiniCluster();
120    }
121    TEST_UTIL = new HBaseTestingUtil();
122    Configuration conf = TEST_UTIL.getConfiguration();
123    conf.setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
124      Arrays.stream(cps).map(Class::getName).toArray(String[]::new));
125    conf.setBoolean(TableDescriptorChecker.TABLE_SANITY_CHECKS, true); // enable for below tests
126    conf.setClass(HConstants.CLIENT_CONNECTION_REGISTRY_IMPL_CONF_KEY, registryImpl,
127        ConnectionRegistry.class);
128    Preconditions.checkArgument(numHedgedReqs > 0);
129    conf.setInt(MasterRegistry.MASTER_REGISTRY_HEDGED_REQS_FANOUT_KEY, numHedgedReqs);
130    StartTestingClusterOption.Builder builder = StartTestingClusterOption.builder();
131    // Multiple masters needed only when hedged reads for master registry are enabled.
132    builder.numMasters(numHedgedReqs > 1 ? 3 : 1).numRegionServers(SLAVES);
133    TEST_UTIL.startMiniCluster(builder.build());
134  }

174  protected ResultScanner buildScanner(String keyPrefix, String value, Table ht)
175    throws IOException {
176    // OurFilterList allFilters = new OurFilterList();
177    FilterList allFilters = new FilterList(/* FilterList.Operator.MUST_PASS_ALL */);
178    allFilters.addFilter(new PrefixFilter(Bytes.toBytes(keyPrefix)));
179    SingleColumnValueFilter filter = new SingleColumnValueFilter(Bytes
180      .toBytes("trans-tags"), Bytes.toBytes("qual2"), CompareOperator.EQUAL, Bytes
181      .toBytes(value));
182    filter.setFilterIfMissing(true);
183    allFilters.addFilter(filter);
184
185    // allFilters.addFilter(new
186    // RowExcludingSingleColumnValueFilter(Bytes.toBytes("trans-tags"),
187    // Bytes.toBytes("qual2"), CompareOp.EQUAL, Bytes.toBytes(value)));
188
189    Scan scan = new Scan();
190    scan.addFamily(Bytes.toBytes("trans-blob"));
191    scan.addFamily(Bytes.toBytes("trans-type"));
192    scan.addFamily(Bytes.toBytes("trans-date"));
193    scan.addFamily(Bytes.toBytes("trans-tags"));
194    scan.addFamily(Bytes.toBytes("trans-group"));
195    scan.setFilter(allFilters);
196
197    return ht.getScanner(scan);
198  }

233  /*
234   * @param key
235   * @param op
236   * @param startRow
237   * @return Scan with RowFilter that does CompareOp op on passed key.
238   */
239  protected Scan createScanWithRowFilter(final byte [] key,
240    final byte [] startRow, CompareOperator op) {
241    // Make sure key is of some substance... non-null and > than first key.
242    assertTrue(key != null && key.length > 0 &&
243      Bytes.BYTES_COMPARATOR.compare(key, new byte [] {'a', 'a', 'a'}) >= 0);
244    LOG.info("Key=" + Bytes.toString(key));
245    Scan s = startRow == null? new Scan(): new Scan().withStartRow(startRow);
246    Filter f = new RowFilter(op, new BinaryComparator(key));
247    f = new WhileMatchFilter(f);
248    s.setFilter(f);
249    return s;
250  }

257  protected List<HRegionLocation> splitTable(final Table t) throws IOException {
258    // Split this table in two.
259    Admin admin = TEST_UTIL.getAdmin();
260    admin.split(t.getName());
261    // Is it right closing this admin?
262    admin.close();
263    List<HRegionLocation> regions = waitOnSplit(t);
264    assertTrue(regions.size() > 1);
265    return regions;
266  }

268  /*
269   * Wait on table split.  May return because we waited long enough on the split
270   * and it didn't happen.  Caller should check.
271   * @param t
272   * @return Map of table regions; caller needs to check table actually split.
273   */
274  private List<HRegionLocation> waitOnSplit(final Table t) throws IOException {
275    try (RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(t.getName())) {
276      List<HRegionLocation> regions = locator.getAllRegionLocations();
277      int originalCount = regions.size();
278      for (int i = 0; i < TEST_UTIL.getConfiguration().getInt("hbase.test.retries", 30); i++) {
279        try {
280          Thread.sleep(1000);
281        } catch (InterruptedException e) {
282          e.printStackTrace();
283        }
284        regions = locator.getAllRegionLocations();
285        if (regions.size() > originalCount) {
286          break;
287        }
288      }
289      return regions;
290    }
291  }

300  byte [][] makeNAscii(byte [] base, int n) {
301    if(n > 256) {
302      return makeNBig(base, n);
303    }
304    byte [][] ret = new byte[n][];
305    for(int i=0;i<n;i++) {
306      byte [] tail = Bytes.toBytes(Integer.toString(i));
307      ret[i] = Bytes.add(base, tail);
308    }
309    return ret;
310  }

312  protected byte [][] makeN(byte [] base, int n) {
313    if (n > 256) {
314      return makeNBig(base, n);
315    }
316    byte [][] ret = new byte[n][];
317    for(int i=0;i<n;i++) {
318      ret[i] = Bytes.add(base, new byte[]{(byte)i});
319    }
320    return ret;
321  }
322
323  protected byte [][] makeNBig(byte [] base, int n) {
324    byte [][] ret = new byte[n][];
325    for(int i = 0; i < n; i++) {
326      int byteA = (i % 256);
327      int byteB = (i >> 8);
328      ret[i] = Bytes.add(base, new byte[]{(byte)byteB,(byte)byteA});
329    }
330    return ret;
331  }

341  protected static boolean equals(byte [] left, byte [] right) {
342    if (left == null && right == null) {
343      return true;
344    }
345    if (left == null && right.length == 0) {
346      return true;
347    }
348    if (right == null && left.length == 0) {
349      return true;
350    }
351    return Bytes.equals(left, right);
352  }
