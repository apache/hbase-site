054 * Test class to verify that metadata is consistent before and after a snapshot attempt.

065  private static final byte[] MAX_VERSIONS_FAM = Bytes.toBytes(MAX_VERSIONS_FAM_STR);

067  private static final String COMPRESSED_FAM_STR = "fam_compressed";
068  private static final byte[] COMPRESSED_FAM = Bytes.toBytes(COMPRESSED_FAM_STR);

071  private static final byte[] BLOCKSIZE_FAM = Bytes.toBytes(BLOCKSIZE_FAM_STR);

076  private static final String TEST_CONF_CUSTOM_VALUE = "TestCustomConf";
077  private static final String TEST_CUSTOM_VALUE = "TestCustomValue";

114  private static void setupConf(Configuration conf) {
115    // enable snapshot support
116    conf.setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
117    // disable the ui
118    conf.setInt("hbase.regionsever.info.port", -1);
119    // change the flush size to a small amount, regulating number of store files
120    conf.setInt("hbase.hregion.memstore.flush.size", 25000);
121    // so make sure we get a compaction when doing a load, but keep around
122    // some files in the store
123    conf.setInt("hbase.hstore.compaction.min", 10);
124    conf.setInt("hbase.hstore.compactionThreshold", 10);
125    // block writes if we get to 12 store files
126    conf.setInt("hbase.hstore.blockingStoreFiles", 12);
127    conf.setInt("hbase.regionserver.msginterval", 100);
128    conf.setBoolean("hbase.master.enabletable.roundrobin", true);
129    // Avoid potentially aggressive splitting which would cause snapshot to fail
130    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
131      ConstantSizeRegionSplitPolicy.class.getName());
132  }

145  /*
146   *  Create a table that has non-default properties so we can see if they hold
147   */
148  private void createTableWithNonDefaultProperties() throws Exception {
149    final long startTime = System.currentTimeMillis();
150    final String sourceTableNameAsString = STRING_TABLE_NAME + startTime;
151    originalTableName = TableName.valueOf(sourceTableNameAsString);
152
153    // enable replication on a column family
154    HColumnDescriptor maxVersionsColumn = new HColumnDescriptor(MAX_VERSIONS_FAM);
155    HColumnDescriptor bloomFilterColumn = new HColumnDescriptor(BLOOMFILTER_FAM);
156    HColumnDescriptor dataBlockColumn = new HColumnDescriptor(COMPRESSED_FAM);
157    HColumnDescriptor blockSizeColumn = new HColumnDescriptor(BLOCKSIZE_FAM);
158
159    maxVersionsColumn.setMaxVersions(MAX_VERSIONS);
160    bloomFilterColumn.setBloomFilterType(BLOOM_TYPE);
161    dataBlockColumn.setDataBlockEncoding(DATA_BLOCK_ENCODING_TYPE);
162    blockSizeColumn.setBlocksize(BLOCK_SIZE);
163
164    HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(sourceTableNameAsString));
165    htd.addFamily(maxVersionsColumn);
166    htd.addFamily(bloomFilterColumn);
167    htd.addFamily(dataBlockColumn);
168    htd.addFamily(blockSizeColumn);
169    htd.setValue(TEST_CUSTOM_VALUE, TEST_CUSTOM_VALUE);
170    htd.setConfiguration(TEST_CONF_CUSTOM_VALUE, TEST_CONF_CUSTOM_VALUE);
171    assertTrue(htd.getConfiguration().size() > 0);
172
173    admin.createTable(htd);
174    Table original = UTIL.getConnection().getTable(originalTableName);
175    originalTableName = TableName.valueOf(sourceTableNameAsString);
176    originalTableDescriptor = admin.getTableDescriptor(originalTableName);
177    originalTableDescription = originalTableDescriptor.toStringCustomizedValues();
178
179    original.close();
180  }

183  /**
184   * Verify that the describe for a cloned table matches the describe from the original.
185   */
186  @Test (timeout=300000)
187  public void testDescribeMatchesAfterClone() throws Exception {
188    // Clone the original table
189    final String clonedTableNameAsString = "clone" + originalTableName;
190    final TableName clonedTableName = TableName.valueOf(clonedTableNameAsString);
191    final String snapshotNameAsString = "snapshot" + originalTableName
192        + System.currentTimeMillis();
193    final byte[] snapshotName = Bytes.toBytes(snapshotNameAsString);
194
195    // restore the snapshot into a cloned table and examine the output
196    List<byte[]> familiesList = new ArrayList<byte[]>();
197    Collections.addAll(familiesList, families);
198
199    // Create a snapshot in which all families are empty
200    SnapshotTestingUtils.createSnapshotAndValidate(admin, originalTableName, null,
201      familiesList, snapshotNameAsString, rootDir, fs, /* onlineSnapshot= */ false);
202
203    admin.cloneSnapshot(snapshotName, clonedTableName);
204    Table clonedTable = UTIL.getConnection().getTable(clonedTableName);
205    HTableDescriptor cloneHtd = admin.getTableDescriptor(clonedTableName);
206    assertEquals(
207      originalTableDescription.replace(originalTableName.getNameAsString(),clonedTableNameAsString),
208      cloneHtd.toStringCustomizedValues());
209
210    // Verify the custom fields
211    assertEquals(originalTableDescriptor.getValues().size(),
212                        cloneHtd.getValues().size());
213    assertEquals(originalTableDescriptor.getConfiguration().size(),
214                        cloneHtd.getConfiguration().size());
215    assertEquals(cloneHtd.getValue(TEST_CUSTOM_VALUE), TEST_CUSTOM_VALUE);
216    assertEquals(cloneHtd.getConfigurationValue(TEST_CONF_CUSTOM_VALUE), TEST_CONF_CUSTOM_VALUE);
217    assertEquals(originalTableDescriptor.getValues(), cloneHtd.getValues());
218    assertEquals(originalTableDescriptor.getConfiguration(), cloneHtd.getConfiguration());
219
220    admin.enableTable(originalTableName);
221    clonedTable.close();
222  }

251  private void runRestoreWithAdditionalMetadata(boolean changeMetadata) throws Exception {
252    runRestoreWithAdditionalMetadata(changeMetadata, true);
253  }
254
255  private void runRestoreWithAdditionalMetadata(boolean changeMetadata, boolean addData)
256      throws Exception {
257
258    if (admin.isTableDisabled(originalTableName)) {
259      admin.enableTable(originalTableName);
260    }
261
262    // populate it with data
263    final byte[] familyForUpdate = BLOCKSIZE_FAM;
264
265    List<byte[]> familiesWithDataList = new ArrayList<byte[]>();
266    List<byte[]> emptyFamiliesList = new ArrayList<byte[]>();
267    if (addData) {
268      Table original = UTIL.getConnection().getTable(originalTableName);
269      UTIL.loadTable(original, familyForUpdate); // family arbitrarily chosen
270      original.close();
271
272      for (byte[] family : families) {
273        if (family != familyForUpdate) {
274          emptyFamiliesList.add(family);
275        }
276      }
277      familiesWithDataList.add(familyForUpdate);
278    } else {
279      Collections.addAll(emptyFamiliesList, families);
280    }
281
282    // take a "disabled" snapshot
283    final String snapshotNameAsString = "snapshot" + originalTableName
284        + System.currentTimeMillis();
285    final byte[] snapshotName = Bytes.toBytes(snapshotNameAsString);
286
287    SnapshotTestingUtils.createSnapshotAndValidate(admin, originalTableName,
288      familiesWithDataList, emptyFamiliesList, snapshotNameAsString, rootDir, fs,
289      /* onlineSnapshot= */ false);
290
291    admin.enableTable(originalTableName);
292
293    if (changeMetadata) {
294      final String newFamilyNameAsString = "newFamily" + System.currentTimeMillis();
295      final byte[] newFamilyName = Bytes.toBytes(newFamilyNameAsString);
296
297      admin.disableTable(originalTableName);
298      HColumnDescriptor hcd = new HColumnDescriptor(newFamilyName);
299      admin.addColumnFamily(originalTableName, hcd);
300      assertTrue("New column family was not added.",
301        admin.getTableDescriptor(originalTableName).toString().contains(newFamilyNameAsString));
302    }
303
304    // restore it
305    if (!admin.isTableDisabled(originalTableName)) {
306      admin.disableTable(originalTableName);
307    }
308
309    admin.restoreSnapshot(snapshotName);
310    admin.enableTable(originalTableName);
311
312    // verify that the descrption is reverted
313    Table original = UTIL.getConnection().getTable(originalTableName);
314    try {
315      assertTrue(originalTableDescriptor.equals(admin.getTableDescriptor(originalTableName)));
316      assertTrue(originalTableDescriptor.equals(original.getTableDescriptor()));
317    } finally {
318      original.close();
319    }
320  }
