102  public static class SlowMeCopro implements RegionCoprocessor, RegionObserver {
103    static final AtomicLong sleepTime = new AtomicLong(0);
104    static final AtomicBoolean slowDownNext = new AtomicBoolean(false);
105    static final AtomicInteger countOfNext = new AtomicInteger(0);
106    private static final AtomicReference<CountDownLatch> primaryCdl =
107        new AtomicReference<>(new CountDownLatch(0));
108    private static final AtomicReference<CountDownLatch> secondaryCdl =
109        new AtomicReference<>(new CountDownLatch(0));
110    Random r = new Random();

112    }

114    @Override
115    public Optional<RegionObserver> getRegionObserver() {
116      return Optional.of(this);
117    }
118
119    @Override
120    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
121                         final Get get, final List<Cell> results) throws IOException {
122      slowdownCode(e);
123    }
124
125    @Override
126    public void preScannerOpen(final ObserverContext<RegionCoprocessorEnvironment> e,
127        final Scan scan) throws IOException {
128      slowdownCode(e);
129    }
130
131    @Override
132    public boolean preScannerNext(final ObserverContext<RegionCoprocessorEnvironment> e,
133        final InternalScanner s, final List<Result> results,
134        final int limit, final boolean hasMore) throws IOException {
135      //this will slow down a certain next operation if the conditions are met. The slowness
136      //will allow the call to go to a replica
137      if (slowDownNext.get()) {
138        //have some "next" return successfully from the primary; hence countOfNext checked
139        if (countOfNext.incrementAndGet() == 2) {
140          sleepTime.set(2000);
141          slowdownCode(e);
142        }
143      }
144      return true;
145    }
146
147    private void slowdownCode(final ObserverContext<RegionCoprocessorEnvironment> e) {
148      if (e.getEnvironment().getRegion().getRegionInfo().getReplicaId() == 0) {
149        LOG.info("We're the primary replicas.");
150        CountDownLatch latch = getPrimaryCdl().get();
151        try {
152          if (sleepTime.get() > 0) {
153            LOG.info("Sleeping for " + sleepTime.get() + " ms");
154            Thread.sleep(sleepTime.get());
155          } else if (latch.getCount() > 0) {
156            LOG.info("Waiting for the counterCountDownLatch");
157            latch.await(2, TimeUnit.MINUTES); // To help the tests to finish.
158            if (latch.getCount() > 0) {
159              throw new RuntimeException("Can't wait more");
160            }
161          }
162        } catch (InterruptedException e1) {
163          LOG.error(e1.toString(), e1);
164        }
165      } else {
166        LOG.info("We're not the primary replicas.");
167        CountDownLatch latch = getSecondaryCdl().get();
168        try {
169          if (latch.getCount() > 0) {
170            LOG.info("Waiting for the secondary counterCountDownLatch");
171            latch.await(2, TimeUnit.MINUTES); // To help the tests to finish.
172            if (latch.getCount() > 0) {
173              throw new RuntimeException("Can't wait more");
174            }
175          }
176        } catch (InterruptedException e1) {
177          LOG.error(e1.toString(), e1);
178        }
179      }
180    }
181
182    public static AtomicReference<CountDownLatch> getPrimaryCdl() {
183      return primaryCdl;
184    }
185
186    public static AtomicReference<CountDownLatch> getSecondaryCdl() {
187      return secondaryCdl;
188    }
189  }

260  private void openRegion(HRegionInfo hri) throws Exception {
261    try {
262      if (isRegionOpened(hri)) return;
263    } catch (Exception e){}
264    // first version is '0'
265    AdminProtos.OpenRegionRequest orr = RequestConverter.buildOpenRegionRequest(
266      getRS().getServerName(), hri, null);
267    AdminProtos.OpenRegionResponse responseOpen = getRS().getRSRpcServices().openRegion(null, orr);
268    Assert.assertEquals(1, responseOpen.getOpeningStateCount());
269    Assert.assertEquals(AdminProtos.OpenRegionResponse.RegionOpeningState.OPENED,
270        responseOpen.getOpeningState(0));
271    checkRegionIsOpened(hri);
272  }
273
274  private void closeRegion(HRegionInfo hri) throws Exception {
275    AdminProtos.CloseRegionRequest crr = ProtobufUtil.buildCloseRegionRequest(
276      getRS().getServerName(), hri.getRegionName());
277    AdminProtos.CloseRegionResponse responseClose = getRS()
278        .getRSRpcServices().closeRegion(null, crr);
279    Assert.assertTrue(responseClose.getClosed());
280
281    checkRegionIsClosed(hri.getEncodedName());
282  }
283
284  private void checkRegionIsOpened(HRegionInfo hri) throws Exception {
285    while (!getRS().getRegionsInTransitionInRS().isEmpty()) {
286      Thread.sleep(1);
287    }
288  }

290  private boolean isRegionOpened(HRegionInfo hri) throws Exception {
291    return getRS().getRegionByEncodedName(hri.getEncodedName()).isAvailable();
292  }

294  private void checkRegionIsClosed(String encodedRegionName) throws Exception {
295
296    while (!getRS().getRegionsInTransitionInRS().isEmpty()) {
297      Thread.sleep(1);
298    }

301      Assert.assertFalse(getRS().getRegionByEncodedName(encodedRegionName).isAvailable());
302    } catch (NotServingRegionException expected) {
303      // That's how it work: if the region is closed we have an exception.
304    }
305
306    // We don't delete the znode here, because there is not always a znode.
307  }

461  @Test
462  public void testUseRegionWithReplica() throws Exception {
463    byte[] b1 = "testUseRegionWithReplica".getBytes();
464    openRegion(hriSecondary);
465
466    try {
467      // A simple put works, even if there here a second replica
468      Put p = new Put(b1);
469      p.addColumn(f, b1, b1);
470      table.put(p);
471      LOG.info("Put done");
472
473      // A get works and is not stale
474      Get g = new Get(b1);
475      Result r = table.get(g);
476      Assert.assertFalse(r.isStale());
477      Assert.assertFalse(r.getColumnCells(f, b1).isEmpty());
478      LOG.info("get works and is not stale done");
479
480      // Even if it we have to wait a little on the main region
481      SlowMeCopro.sleepTime.set(2000);
482      g = new Get(b1);
483      r = table.get(g);
484      Assert.assertFalse(r.isStale());
485      Assert.assertFalse(r.getColumnCells(f, b1).isEmpty());
486      SlowMeCopro.sleepTime.set(0);
487      LOG.info("sleep and is not stale done");
488
489      // But if we ask for stale we will get it
490      SlowMeCopro.getPrimaryCdl().set(new CountDownLatch(1));
491      g = new Get(b1);
492      g.setConsistency(Consistency.TIMELINE);
493      r = table.get(g);
494      Assert.assertTrue(r.isStale());
495      Assert.assertTrue(r.getColumnCells(f, b1).isEmpty());
496      SlowMeCopro.getPrimaryCdl().get().countDown();
497
498      LOG.info("stale done");
499
500      // exists works and is not stale
501      g = new Get(b1);
502      g.setCheckExistenceOnly(true);
503      r = table.get(g);
504      Assert.assertFalse(r.isStale());
505      Assert.assertTrue(r.getExists());
506      LOG.info("exists not stale done");
507
508      // exists works on stale but don't see the put
509      SlowMeCopro.getPrimaryCdl().set(new CountDownLatch(1));
510      g = new Get(b1);
511      g.setCheckExistenceOnly(true);
512      g.setConsistency(Consistency.TIMELINE);
513      r = table.get(g);
514      Assert.assertTrue(r.isStale());
515      Assert.assertFalse("The secondary has stale data", r.getExists());
516      SlowMeCopro.getPrimaryCdl().get().countDown();
517      LOG.info("exists stale before flush done");
518
519      flushRegion(hriPrimary);
520      flushRegion(hriSecondary);
521      LOG.info("flush done");
522      Thread.sleep(1000 + REFRESH_PERIOD * 2);
523
524      // get works and is not stale
525      SlowMeCopro.getPrimaryCdl().set(new CountDownLatch(1));
526      g = new Get(b1);
527      g.setConsistency(Consistency.TIMELINE);
528      r = table.get(g);
529      Assert.assertTrue(r.isStale());
530      Assert.assertFalse(r.isEmpty());
531      SlowMeCopro.getPrimaryCdl().get().countDown();
532      LOG.info("stale done");
533
534      // exists works on stale and we see the put after the flush
535      SlowMeCopro.getPrimaryCdl().set(new CountDownLatch(1));
536      g = new Get(b1);
537      g.setCheckExistenceOnly(true);
538      g.setConsistency(Consistency.TIMELINE);
539      r = table.get(g);
540      Assert.assertTrue(r.isStale());
541      Assert.assertTrue(r.getExists());
542      SlowMeCopro.getPrimaryCdl().get().countDown();
543      LOG.info("exists stale after flush done");
544
545    } finally {
546      SlowMeCopro.getPrimaryCdl().get().countDown();
547      SlowMeCopro.sleepTime.set(0);
548      Delete d = new Delete(b1);
549      table.delete(d);
550      closeRegion(hriSecondary);
551    }
552  }

554  @Test
555  public void testHedgedRead() throws Exception {
556    byte[] b1 = "testHedgedRead".getBytes();
557    openRegion(hriSecondary);
558
559    try {
560      // A simple put works, even if there here a second replica
561      Put p = new Put(b1);
562      p.addColumn(f, b1, b1);
563      table.put(p);
564      LOG.info("Put done");
565
566      // A get works and is not stale
567      Get g = new Get(b1);
568      Result r = table.get(g);
569      Assert.assertFalse(r.isStale());
570      Assert.assertFalse(r.getColumnCells(f, b1).isEmpty());
571      LOG.info("get works and is not stale done");
572
573      //reset
574      ClusterConnection connection = (ClusterConnection) HTU.getConnection();
575      Counter hedgedReadOps = connection.getConnectionMetrics().hedgedReadOps;
576      Counter hedgedReadWin = connection.getConnectionMetrics().hedgedReadWin;
577      hedgedReadOps.dec(hedgedReadOps.getCount());
578      hedgedReadWin.dec(hedgedReadWin.getCount());
579
580      // Wait a little on the main region, just enough to happen once hedged read
581      // and hedged read did not returned faster
582      int primaryCallTimeoutMicroSecond = connection.getConnectionConfiguration().getPrimaryCallTimeoutMicroSecond();
583      SlowMeCopro.sleepTime.set(TimeUnit.MICROSECONDS.toMillis(primaryCallTimeoutMicroSecond));
584      SlowMeCopro.getSecondaryCdl().set(new CountDownLatch(1));
585      g = new Get(b1);
586      g.setConsistency(Consistency.TIMELINE);
587      r = table.get(g);
588      Assert.assertFalse(r.isStale());
589      Assert.assertFalse(r.getColumnCells(f, b1).isEmpty());
590      Assert.assertEquals(1, hedgedReadOps.getCount());
591      Assert.assertEquals(0, hedgedReadWin.getCount());
592      SlowMeCopro.sleepTime.set(0);
593      SlowMeCopro.getSecondaryCdl().get().countDown();
594      LOG.info("hedged read occurred but not faster");
595
596
597      // But if we ask for stale we will get it and hedged read returned faster
598      SlowMeCopro.getPrimaryCdl().set(new CountDownLatch(1));
599      g = new Get(b1);
600      g.setConsistency(Consistency.TIMELINE);
601      r = table.get(g);
602      Assert.assertTrue(r.isStale());
603      Assert.assertTrue(r.getColumnCells(f, b1).isEmpty());
604      Assert.assertEquals(2, hedgedReadOps.getCount());
605      Assert.assertEquals(1, hedgedReadWin.getCount());
606      SlowMeCopro.getPrimaryCdl().get().countDown();
607      LOG.info("hedged read occurred and faster");
608
609    } finally {
610      SlowMeCopro.getPrimaryCdl().get().countDown();
611      SlowMeCopro.getSecondaryCdl().get().countDown();
612      SlowMeCopro.sleepTime.set(0);
613      Delete d = new Delete(b1);
614      table.delete(d);
615      closeRegion(hriSecondary);
616    }
617  }

634      LOG.debug("PUT done");
635      flushRegion(hriPrimary);
636      LOG.info("flush done");
637
638      Thread.sleep(1000 + REFRESH_PERIOD * 2);
639
640      AsyncProcess ap = ((ClusterConnection) HTU.getConnection()).getAsyncProcess();
641
642      // Make primary slowdown
643      SlowMeCopro.getPrimaryCdl().set(new CountDownLatch(1));
644
645      List<Get> gets = new ArrayList<>();
646      Get g = new Get(b1);
647      g.setCheckExistenceOnly(true);
648      g.setConsistency(Consistency.TIMELINE);
649      gets.add(g);
650      g = new Get(b2);
651      g.setCheckExistenceOnly(true);
652      g.setConsistency(Consistency.TIMELINE);
653      gets.add(g);
654      Object[] results = new Object[2];
655
656      int operationTimeout = ((ClusterConnection) HTU.getConnection()).getConnectionConfiguration().getOperationTimeout();
657      int readTimeout = ((ClusterConnection) HTU.getConnection()).getConnectionConfiguration().getReadRpcTimeout();
658      AsyncProcessTask task = AsyncProcessTask.newBuilder()
659              .setPool(HTable.getDefaultExecutor(HTU.getConfiguration()))
660              .setTableName(table.getName())
661              .setRowAccess(gets)
662              .setResults(results)
663              .setOperationTimeout(operationTimeout)
664              .setRpcTimeout(readTimeout)
665              .build();
666      AsyncRequestFuture reqs = ap.submit(task);
667      reqs.waitUntilDone();
668      // verify we got the right results back
669      for (Object r : results) {
670        Assert.assertTrue(((Result)r).isStale());
671        Assert.assertTrue(((Result)r).getExists());
672      }
673      Set<CancellableRegionServerCallable> set =
674          ((AsyncRequestFutureImpl<?>)reqs).getCallsInProgress();
675      // verify we did cancel unneeded calls
676      Assert.assertTrue(!set.isEmpty());
677      for (CancellableRegionServerCallable m : set) {
678        Assert.assertTrue(m.isCancelled());
679      }
680    } finally {
681      SlowMeCopro.getPrimaryCdl().get().countDown();
682      SlowMeCopro.sleepTime.set(0);
683      SlowMeCopro.slowDownNext.set(false);
684      SlowMeCopro.countOfNext.set(0);
685      for (int i = 0; i < 2; i++) {
686        byte[] b1 = Bytes.toBytes("testCancelOfMultiGet" + i);
687        Delete d = new Delete(b1);
688        table.delete(d);
689      }
690      closeRegion(hriSecondary);
691    }

760  private void runMultipleScansOfOneType(boolean reversed, boolean small) throws Exception {
761    openRegion(hriSecondary);
762    int NUMROWS = 100;
763    int NUMCOLS = 10;
764    try {
765      for (int i = 0; i < NUMROWS; i++) {
766        byte[] b1 = Bytes.toBytes("testUseRegionWithReplica" + i);
767        for (int col = 0; col < NUMCOLS; col++) {
768          Put p = new Put(b1);
769          String qualifier = "qualifer" + col;
770          KeyValue kv = new KeyValue(b1, f, qualifier.getBytes());
771          p.add(kv);
772          table.put(p);
773        }
774      }
775      LOG.debug("PUT done");
776      int caching = 20;
777      long maxResultSize = Long.MAX_VALUE;
778
779      byte[] start;
780      if (reversed) start = Bytes.toBytes("testUseRegionWithReplica" + (NUMROWS - 1));
781      else start = Bytes.toBytes("testUseRegionWithReplica" + 0);
782
783      scanWithReplicas(reversed, small, Consistency.TIMELINE, caching, maxResultSize,
784        start, NUMROWS, NUMCOLS, false, false);
785
786      // Even if we were to slow the server down, unless we ask for stale
787      // we won't get it
788      SlowMeCopro.sleepTime.set(5000);
789      scanWithReplicas(reversed, small, Consistency.STRONG, caching, maxResultSize, start, NUMROWS,
790        NUMCOLS, false, false);
791      SlowMeCopro.sleepTime.set(0);
792
793      flushRegion(hriPrimary);
794      LOG.info("flush done");
795      Thread.sleep(1000 + REFRESH_PERIOD * 2);
796
797      //Now set the flag to get a response even if stale
798      SlowMeCopro.sleepTime.set(5000);
799      scanWithReplicas(reversed, small, Consistency.TIMELINE, caching, maxResultSize,
800        start, NUMROWS, NUMCOLS, true, false);
801      SlowMeCopro.sleepTime.set(0);
802
803      // now make some 'next' calls slow
804      SlowMeCopro.slowDownNext.set(true);
805      SlowMeCopro.countOfNext.set(0);
806      scanWithReplicas(reversed, small, Consistency.TIMELINE, caching, maxResultSize, start,
807        NUMROWS, NUMCOLS, true, true);
808      SlowMeCopro.slowDownNext.set(false);
809      SlowMeCopro.countOfNext.set(0);
810
811      // Make sure we do not get stale data..
812      SlowMeCopro.sleepTime.set(5000);
813      scanWithReplicas(reversed, small, Consistency.STRONG, caching, maxResultSize,
814        start, NUMROWS, NUMCOLS, false, false);
815      SlowMeCopro.sleepTime.set(0);
816
817      // While the next calls are slow, set maxResultSize to 1 so that some partial results will be
818      // returned from the server before the replica switch occurs.
819      maxResultSize = 1;
820      SlowMeCopro.slowDownNext.set(true);
821      SlowMeCopro.countOfNext.set(0);
822      scanWithReplicas(reversed, small, Consistency.TIMELINE, caching, maxResultSize, start,
823        NUMROWS, NUMCOLS, true, true);
824      maxResultSize = Long.MAX_VALUE;
825      SlowMeCopro.slowDownNext.set(false);
826      SlowMeCopro.countOfNext.set(0);
827    } finally {
828      SlowMeCopro.getPrimaryCdl().get().countDown();
829      SlowMeCopro.sleepTime.set(0);
830      SlowMeCopro.slowDownNext.set(false);
831      SlowMeCopro.countOfNext.set(0);
832      for (int i = 0; i < NUMROWS; i++) {
833        byte[] b1 = Bytes.toBytes("testUseRegionWithReplica" + i);
834        Delete d = new Delete(b1);
835        table.delete(d);
836      }
837      closeRegion(hriSecondary);
838    }
839  }

841  private void scanWithReplicas(boolean reversed, boolean small, Consistency consistency,
842      int caching, long maxResultSize, byte[] startRow, int numRows, int numCols,
843      boolean staleExpected, boolean slowNext)
844          throws Exception {
845    Scan scan = new Scan(startRow);
846    scan.setCaching(caching);
847    scan.setMaxResultSize(maxResultSize);
848    scan.setReversed(reversed);
849    scan.setSmall(small);
850    scan.setConsistency(consistency);
851    ResultScanner scanner = table.getScanner(scan);
852    Iterator<Result> iter = scanner.iterator();
853
854    // Maps of row keys that we have seen so far
855    HashMap<String, Boolean> map = new HashMap<>();
856
857    // Tracked metrics
858    int rowCount = 0;
859    int cellCount = 0;
860    int countOfStale = 0;
861
862    while (iter.hasNext()) {
863      rowCount++;
864      Result r = iter.next();
865      String row = new String(r.getRow());
866
867      if (map.containsKey(row)) {
868        throw new Exception("Unexpected scan result. Repeated row " + Bytes.toString(r.getRow()));
869      }
870
871      map.put(row, true);
872
873      for (Cell cell : r.rawCells()) {
874        cellCount++;
875      }
876
877      if (!slowNext) Assert.assertTrue(r.isStale() == staleExpected);
878      if (r.isStale()) countOfStale++;
879    }
880    Assert.assertTrue("Count of rows " + rowCount + " num rows expected " + numRows,
881      rowCount == numRows);
882    Assert.assertTrue("Count of cells: " + cellCount + " cells expected: " + numRows * numCols,
883      cellCount == (numRows * numCols));
884
885    if (slowNext) {
886      LOG.debug("Count of Stale " + countOfStale);
887      Assert.assertTrue(countOfStale > 1);
888
889      // If the scan was configured in such a way that a full row was NOT retrieved before the
890      // replica switch occurred, then it is possible that all rows were stale
891      if (maxResultSize != Long.MAX_VALUE) {
892        Assert.assertTrue(countOfStale <= numRows);
893      } else {
894        Assert.assertTrue(countOfStale < numRows);
895      }
896    }
897  }
