102  @Test
103  public void testCompactionTimestamps() throws Exception {
104    TableName tableName = TableName.valueOf(name.getMethodName());
105    TableDescriptor htd = TableDescriptorBuilder.newBuilder(tableName)
106      .setColumnFamily(ColumnFamilyDescriptorBuilder.of("fam1")).build();
107    ADMIN.createTable(htd);
108    Table table = TEST_UTIL.getConnection().getTable(htd.getTableName());
109    long ts = ADMIN.getLastMajorCompactionTimestamp(tableName);
110    assertEquals(0, ts);
111    Put p = new Put(Bytes.toBytes("row1"));
112    p.addColumn(Bytes.toBytes("fam1"), Bytes.toBytes("fam1"), Bytes.toBytes("fam1"));
113    table.put(p);
114    ts = ADMIN.getLastMajorCompactionTimestamp(tableName);
115    // no files written -> no data
116    assertEquals(0, ts);
117
118    ADMIN.flush(tableName);
119    ts = ADMIN.getLastMajorCompactionTimestamp(tableName);
120    // still 0, we flushed a file, but no major compaction happened
121    assertEquals(0, ts);
122
123    byte[] regionName;
124    try (RegionLocator l = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
125      regionName = l.getAllRegionLocations().get(0).getRegion().getRegionName();
126    }
127    long ts1 = ADMIN.getLastMajorCompactionTimestampForRegion(regionName);
128    assertEquals(ts, ts1);
129    p = new Put(Bytes.toBytes("row2"));
130    p.addColumn(Bytes.toBytes("fam1"), Bytes.toBytes("fam1"), Bytes.toBytes("fam1"));
131    table.put(p);
132    ADMIN.flush(tableName);
133    ts = ADMIN.getLastMajorCompactionTimestamp(tableName);
134    // make sure the region API returns the same value, as the old file is still around
135    assertEquals(ts1, ts);
136
137    TEST_UTIL.compact(tableName, true);
138    table.put(p);
139    // forces a wait for the compaction
140    ADMIN.flush(tableName);
141    ts = ADMIN.getLastMajorCompactionTimestamp(tableName);
142    // after a compaction our earliest timestamp will have progressed forward
143    assertTrue(ts > ts1);
144
145    // region api still the same
146    ts1 = ADMIN.getLastMajorCompactionTimestampForRegion(regionName);
147    assertEquals(ts, ts1);
148    table.put(p);
149    ADMIN.flush(tableName);
150    ts = ADMIN.getLastMajorCompactionTimestamp(tableName);
151    assertEquals(ts, ts1);
152    table.close();
153  }

218  private int count(ResultScanner scanner) throws IOException {
219    int rows = 0;
220    while (scanner.next() != null) {
221      rows++;
222    }
223    return rows;
224  }

226  private void splitTest(byte[] splitPoint, byte[][] familyNames, int[] rowCounts, int numVersions,
227      int blockSize, boolean async) throws Exception {
228    TableName tableName = TableName.valueOf("testForceSplit");
229    StringBuilder sb = new StringBuilder();
230    // Add tail to String so can see better in logs where a test is running.
231    for (int i = 0; i < rowCounts.length; i++) {
232      sb.append("_").append(Integer.toString(rowCounts[i]));
233    }
234    assertFalse(ADMIN.tableExists(tableName));
235    try (final Table table = TEST_UTIL.createTable(tableName, familyNames, numVersions, blockSize);
236        final RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
237
238      int rowCount = 0;
239      byte[] q = new byte[0];
240
241      // insert rows into column families. The number of rows that have values
242      // in a specific column family is decided by rowCounts[familyIndex]
243      for (int index = 0; index < familyNames.length; index++) {
244        ArrayList<Put> puts = new ArrayList<>(rowCounts[index]);
245        for (int i = 0; i < rowCounts[index]; i++) {
246          byte[] k = Bytes.toBytes(i);
247          Put put = new Put(k);
248          put.addColumn(familyNames[index], q, k);
249          puts.add(put);
250        }
251        table.put(puts);
252
253        if (rowCount < rowCounts[index]) {
254          rowCount = rowCounts[index];
255        }
256      }
257
258      // get the initial layout (should just be one region)
259      List<HRegionLocation> m = locator.getAllRegionLocations();
260      LOG.info("Initial regions (" + m.size() + "): " + m);
261      assertTrue(m.size() == 1);
262
263      // Verify row count
264      Scan scan = new Scan();
265      int rows;
266      try (ResultScanner scanner = table.getScanner(scan)) {
267        rows = count(scanner);
268      }
269      assertEquals(rowCount, rows);
270
271      // Have an outstanding scan going on to make sure we can scan over splits.
272      scan = new Scan();
273      try (ResultScanner scanner = table.getScanner(scan)) {
274        // Scan first row so we are into first region before split happens.
275        scanner.next();
276
277        // Split the table
278        if (async) {
279          if (splitPoint != null) {
280            ADMIN.split(tableName, splitPoint);
281          } else {
282            ADMIN.split(tableName);
283          }
284          final AtomicInteger count = new AtomicInteger(0);
285          Thread t = new Thread("CheckForSplit") {
286            @Override
287            public void run() {
288              for (int i = 0; i < 45; i++) {
289                try {
290                  sleep(1000);
291                } catch (InterruptedException e) {
292                  continue;
293                }
294                // check again
295                List<HRegionLocation> regions = null;
296                try {
297                  regions = locator.getAllRegionLocations();
298                } catch (IOException e) {
299                  LOG.warn("get location failed", e);
300                }
301                if (regions == null) {
302                  continue;
303                }
304                count.set(regions.size());
305                if (count.get() >= 2) {
306                  LOG.info("Found: " + regions);
307                  break;
308                }
309                LOG.debug("Cycle waiting on split");
310              }
311              LOG.debug("CheckForSplit thread exited, current region count: " + count.get());
312            }
313          };
314          t.setPriority(Thread.NORM_PRIORITY - 2);
315          t.start();
316          t.join();
317        } else {
318          // Sync split region, no need to create a thread to check
319          ADMIN.splitRegionAsync(m.get(0).getRegion().getRegionName(), splitPoint).get();
320        }
321        // Verify row count
322        rows = 1 + count(scanner); // We counted one row above.
323      }
324      assertEquals(rowCount, rows);
325
326      List<HRegionLocation> regions = null;
327      try {
328        regions = locator.getAllRegionLocations();
329      } catch (IOException e) {
330        e.printStackTrace();
331      }
332      assertEquals(2, regions.size());
333      if (splitPoint != null) {
334        // make sure the split point matches our explicit configuration
335        assertEquals(Bytes.toString(splitPoint),
336          Bytes.toString(regions.get(0).getRegion().getEndKey()));
337        assertEquals(Bytes.toString(splitPoint),
338          Bytes.toString(regions.get(1).getRegion().getStartKey()));
339        LOG.debug("Properly split on " + Bytes.toString(splitPoint));
340      } else {
341        if (familyNames.length > 1) {
342          int splitKey = Bytes.toInt(regions.get(0).getRegion().getEndKey());
343          // check if splitKey is based on the largest column family
344          // in terms of it store size
345          int deltaForLargestFamily = Math.abs(rowCount / 2 - splitKey);
346          LOG.debug("SplitKey=" + splitKey + "&deltaForLargestFamily=" + deltaForLargestFamily +
347            ", r=" + regions.get(0).getRegion());
348          for (int index = 0; index < familyNames.length; index++) {
349            int delta = Math.abs(rowCounts[index] / 2 - splitKey);
350            if (delta < deltaForLargestFamily) {
351              assertTrue("Delta " + delta + " for family " + index + " should be at least " +
352                "deltaForLargestFamily " + deltaForLargestFamily, false);
353            }
354          }
355        }
356      }
357      TEST_UTIL.deleteTable(tableName);
358    }
359  }

361  @Test
362  public void testSplitAndMergeWithReplicaTable() throws Exception {
363    // The test tries to directly split replica regions and directly merge replica regions. These
364    // are not allowed. The test validates that. Then the test does a valid split/merge of allowed
365    // regions.
366    // Set up a table with 3 regions and replication set to 3
367    TableName tableName = TableName.valueOf(name.getMethodName());
368    byte[] cf = Bytes.toBytes("f");
369    TableDescriptor desc = TableDescriptorBuilder.newBuilder(tableName).setRegionReplication(3)
370      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(cf)).build();
371    byte[][] splitRows = new byte[2][];
372    splitRows[0] = new byte[] { (byte) '4' };
373    splitRows[1] = new byte[] { (byte) '7' };
374    TEST_UTIL.getAdmin().createTable(desc, splitRows);
375    List<HRegion> oldRegions;
376    do {
377      oldRegions = TEST_UTIL.getHBaseCluster().getRegions(tableName);
378      Thread.sleep(10);
379    } while (oldRegions.size() != 9); // 3 regions * 3 replicas
380    // write some data to the table
381    Table ht = TEST_UTIL.getConnection().getTable(tableName);
382    List<Put> puts = new ArrayList<>();
383    byte[] qualifier = Bytes.toBytes("c");
384    Put put = new Put(new byte[] { (byte) '1' });
385    put.addColumn(cf, qualifier, Bytes.toBytes("100"));
386    puts.add(put);
387    put = new Put(new byte[] { (byte) '6' });
388    put.addColumn(cf, qualifier, Bytes.toBytes("100"));
389    puts.add(put);
390    put = new Put(new byte[] { (byte) '8' });
391    put.addColumn(cf, qualifier, Bytes.toBytes("100"));
392    puts.add(put);
393    ht.put(puts);
394    ht.close();
395    List<Pair<RegionInfo, ServerName>> regions =
396      MetaTableAccessor.getTableRegionsAndLocations(TEST_UTIL.getConnection(), tableName);
397    boolean gotException = false;
398    // the element at index 1 would be a replica (since the metareader gives us ordered
399    // regions). Try splitting that region via the split API . Should fail
400    try {
401      FutureUtils.get(
402        TEST_UTIL.getAdmin().splitRegionAsync(regions.get(1).getFirst().getRegionName()));
403    } catch (IllegalArgumentException ex) {
404      gotException = true;
405    }
406    assertTrue(gotException);
407    gotException = false;
408    // the element at index 1 would be a replica (since the metareader gives us ordered
409    // regions). Try splitting that region via a different split API (the difference is
410    // this API goes direct to the regionserver skipping any checks in the admin). Should fail
411    try {
412      FutureUtils.get(TEST_UTIL.getAdmin().splitRegionAsync(
413        regions.get(1).getFirst().getEncodedNameAsBytes(), new byte[] { (byte) '1' }));
414    } catch (IllegalArgumentException ex) {
415      gotException = true;
416    }
417    assertTrue(gotException);
418
419    gotException = false;
420    // testing Sync split operation
421    try {
422      FutureUtils.get(TEST_UTIL.getAdmin()
423        .splitRegionAsync(regions.get(1).getFirst().getRegionName(), new byte[] { (byte) '1' }));
424    } catch (IllegalArgumentException ex) {
425      gotException = true;
426    }
427    assertTrue(gotException);
428
429    gotException = false;
430    // Try merging a replica with another. Should fail.
431    try {
432      FutureUtils.get(TEST_UTIL.getAdmin().mergeRegionsAsync(
433        regions.get(1).getFirst().getEncodedNameAsBytes(),
434        regions.get(2).getFirst().getEncodedNameAsBytes(),
435        true));
436    } catch (IllegalArgumentException m) {
437      gotException = true;
438    }
439    assertTrue(gotException);
440    // Try going to the master directly (that will skip the check in admin)
441    try {
442      byte[][] nameofRegionsToMerge = new byte[2][];
443      nameofRegionsToMerge[0] = regions.get(1).getFirst().getEncodedNameAsBytes();
444      nameofRegionsToMerge[1] = regions.get(2).getFirst().getEncodedNameAsBytes();
445      MergeTableRegionsRequest request = RequestConverter.buildMergeTableRegionsRequest(
446        nameofRegionsToMerge, true, HConstants.NO_NONCE, HConstants.NO_NONCE);
447      TEST_UTIL.getMiniHBaseCluster().getMaster().getMasterRpcServices().mergeTableRegions(null,
448        request);
449    } catch (org.apache.hbase.thirdparty.com.google.protobuf.ServiceException m) {
450      Throwable t = m.getCause();
451      do {
452        if (t instanceof MergeRegionException) {
453          gotException = true;
454          break;
455        }
456        t = t.getCause();
457      } while (t != null);
458    }
459    assertTrue(gotException);
460  }

468   * Test DFS replication for column families, where one CF has default replication(3) and the other
469   * is set to 1.

471  @Test
472  public void testHFileReplication() throws Exception {
473    final TableName tableName = TableName.valueOf(this.name.getMethodName());
474    String fn1 = "rep1";
475    String fn = "defaultRep";
476    TableDescriptor htd = TableDescriptorBuilder.newBuilder(tableName)
477      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(fn))
478      .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes(fn1))
479        .setDFSReplication((short) 1).build())
480      .build();
481    Table table = TEST_UTIL.createTable(htd, null);
482    TEST_UTIL.waitTableAvailable(tableName);
483    Put p = new Put(Bytes.toBytes("defaultRep_rk"));
484    byte[] q1 = Bytes.toBytes("q1");
485    byte[] v1 = Bytes.toBytes("v1");
486    p.addColumn(Bytes.toBytes(fn), q1, v1);
487    List<Put> puts = new ArrayList<>(2);
488    puts.add(p);
489    p = new Put(Bytes.toBytes("rep1_rk"));
490    p.addColumn(Bytes.toBytes(fn1), q1, v1);
491    puts.add(p);
492    try {
493      table.put(puts);
494      ADMIN.flush(tableName);
495
496      List<HRegion> regions = TEST_UTIL.getMiniHBaseCluster().getRegions(tableName);
497      for (HRegion r : regions) {
498        HStore store = r.getStore(Bytes.toBytes(fn));
499        for (HStoreFile sf : store.getStorefiles()) {
500          assertTrue(sf.toString().contains(fn));
501          assertTrue("Column family " + fn + " should have 3 copies",
502            CommonFSUtils.getDefaultReplication(TEST_UTIL.getTestFileSystem(),
503              sf.getPath()) == (sf.getFileInfo().getFileStatus().getReplication()));
504        }
505
506        store = r.getStore(Bytes.toBytes(fn1));
507        for (HStoreFile sf : store.getStorefiles()) {
508          assertTrue(sf.toString().contains(fn1));
509          assertTrue("Column family " + fn1 + " should have only 1 copy",
510            1 == sf.getFileInfo().getFileStatus().getReplication());
511        }
512      }
513    } finally {
514      if (ADMIN.isTableEnabled(tableName)) {
515        ADMIN.disableTable(tableName);
516        ADMIN.deleteTable(tableName);
517      }
518    }
519  }

537      RegionInfo mergedChildRegion = null;

540      tableRegions = ADMIN.getRegions(tableName);
541      assertEquals(3, ADMIN.getRegions(tableName).size());
542      regionA = tableRegions.get(0);
543      regionB = tableRegions.get(1);
544      regionC = tableRegions.get(2);
545      // TODO convert this to version that is synchronous (See HBASE-16668)
546      ADMIN.mergeRegionsAsync(regionA.getRegionName(), regionB.getRegionName(),
547        false).get(60, TimeUnit.SECONDS);
548
549      tableRegions = ADMIN.getRegions(tableName);
550
551      assertEquals(2, tableRegions.size());
552      for (RegionInfo ri : tableRegions) {
553        if (regionC.compareTo(ri) != 0) {
554          mergedChildRegion = ri;
555          break;
556        }
557      }
558
559      assertNotNull(mergedChildRegion);
560      // Need to wait GC for merged child region is done.
561      HMaster services = TEST_UTIL.getHBaseCluster().getMaster();
562      CatalogJanitor cj = services.getCatalogJanitor();
563      cj.cleanMergeQualifier(mergedChildRegion);
564      // Wait until all procedures settled down
565      while (!services.getMasterProcedureExecutor().getActiveProcIds().isEmpty()) {
566        Thread.sleep(200);
567      }
568
569      // TODO convert this to version that is synchronous (See HBASE-16668)
570      ADMIN.mergeRegionsAsync(regionC.getEncodedNameAsBytes(),
571        mergedChildRegion.getEncodedNameAsBytes(), false)
572        .get(60, TimeUnit.SECONDS);
573
574      assertEquals(1, ADMIN.getRegions(tableName).size());
575    } finally {
576      ADMIN.disableTable(tableName);
577      ADMIN.deleteTable(tableName);
578    }
