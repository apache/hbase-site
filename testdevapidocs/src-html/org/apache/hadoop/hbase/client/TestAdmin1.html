283      ht.get(get);

285      ok = false;
286    }
287    assertTrue(ok);

429  @Test (timeout=300000)
430  public void testCompactionTimestamps() throws Exception {
431    HColumnDescriptor fam1 = new HColumnDescriptor("fam1");
432    final TableName tableName = TableName.valueOf(name.getMethodName());
433    HTableDescriptor htd = new HTableDescriptor(tableName);
434    htd.addFamily(fam1);
435    this.admin.createTable(htd);
436    Table table = TEST_UTIL.getConnection().getTable(htd.getTableName());
437    long ts = this.admin.getLastMajorCompactionTimestamp(tableName);
438    assertEquals(0, ts);
439    Put p = new Put(Bytes.toBytes("row1"));
440    p.addColumn(Bytes.toBytes("fam1"), Bytes.toBytes("fam1"), Bytes.toBytes("fam1"));
441    table.put(p);
442    ts = this.admin.getLastMajorCompactionTimestamp(tableName);
443    // no files written -> no data
444    assertEquals(0, ts);
445
446    this.admin.flush(tableName);
447    ts = this.admin.getLastMajorCompactionTimestamp(tableName);
448    // still 0, we flushed a file, but no major compaction happened
449    assertEquals(0, ts);
450
451    byte[] regionName;
452    try (RegionLocator l = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
453      regionName = l.getAllRegionLocations().get(0).getRegionInfo().getRegionName();
454    }
455    long ts1 = this.admin.getLastMajorCompactionTimestampForRegion(regionName);
456    assertEquals(ts, ts1);
457    p = new Put(Bytes.toBytes("row2"));
458    p.addColumn(Bytes.toBytes("fam1"), Bytes.toBytes("fam1"), Bytes.toBytes("fam1"));
459    table.put(p);
460    this.admin.flush(tableName);
461    ts = this.admin.getLastMajorCompactionTimestamp(tableName);
462    // make sure the region API returns the same value, as the old file is still around
463    assertEquals(ts1, ts);
464
465    TEST_UTIL.compact(tableName, true);
466    table.put(p);
467    // forces a wait for the compaction
468    this.admin.flush(tableName);
469    ts = this.admin.getLastMajorCompactionTimestamp(tableName);
470    // after a compaction our earliest timestamp will have progressed forward
471    assertTrue(ts > ts1);
472
473    // region api still the same
474    ts1 = this.admin.getLastMajorCompactionTimestampForRegion(regionName);
475    assertEquals(ts, ts1);
476    table.put(p);
477    this.admin.flush(tableName);
478    ts = this.admin.getLastMajorCompactionTimestamp(tableName);
479    assertEquals(ts, ts1);
480    table.close();
481  }

499  @Test (timeout=300000)
500  public void testOnlineChangeTableSchema() throws IOException, InterruptedException {
501    final TableName tableName = TableName.valueOf(name.getMethodName());
502    HTableDescriptor [] tables = admin.listTables();
503    int numTables = tables.length;
504    TEST_UTIL.createTable(tableName, HConstants.CATALOG_FAMILY).close();
505    tables = this.admin.listTables();
506    assertEquals(numTables + 1, tables.length);
507
508    // FIRST, do htabledescriptor changes.
509    HTableDescriptor htd = this.admin.getTableDescriptor(tableName);
510    // Make a copy and assert copy is good.
511    HTableDescriptor copy = new HTableDescriptor(htd);
512    assertTrue(htd.equals(copy));
513    // Now amend the copy. Introduce differences.
514    long newFlushSize = htd.getMemStoreFlushSize() / 2;
515    if (newFlushSize <=0) {
516      newFlushSize = HTableDescriptor.DEFAULT_MEMSTORE_FLUSH_SIZE / 2;
517    }
518    copy.setMemStoreFlushSize(newFlushSize);
519    final String key = "anyoldkey";
520    assertTrue(htd.getValue(key) == null);
521    copy.setValue(key, key);
522    boolean expectedException = false;
523    try {
524      admin.modifyTable(tableName, copy);
525    } catch (TableNotDisabledException re) {
526      expectedException = true;
527    }
528    assertFalse(expectedException);
529    HTableDescriptor modifiedHtd = new HTableDescriptor(this.admin.getTableDescriptor(tableName));
530    assertFalse(htd.equals(modifiedHtd));
531    assertTrue(copy.equals(modifiedHtd));
532    assertEquals(newFlushSize, modifiedHtd.getMemStoreFlushSize());
533    assertEquals(key, modifiedHtd.getValue(key));
534
535    // Now work on column family changes.
536    int countOfFamilies = modifiedHtd.getFamilies().size();
537    assertTrue(countOfFamilies > 0);
538    HColumnDescriptor hcd = modifiedHtd.getFamilies().iterator().next();
539    int maxversions = hcd.getMaxVersions();
540    final int newMaxVersions = maxversions + 1;
541    hcd.setMaxVersions(newMaxVersions);
542    final byte [] hcdName = hcd.getName();
543    expectedException = false;
544    try {
545      this.admin.modifyColumnFamily(tableName, hcd);
546    } catch (TableNotDisabledException re) {
547      expectedException = true;
548    }
549    assertFalse(expectedException);
550    modifiedHtd = this.admin.getTableDescriptor(tableName);
551    HColumnDescriptor modifiedHcd = modifiedHtd.getFamily(hcdName);
552    assertEquals(newMaxVersions, modifiedHcd.getMaxVersions());
553
554    // Try adding a column
555    assertFalse(this.admin.isTableDisabled(tableName));
556    final String xtracolName = "xtracol";
557    HColumnDescriptor xtracol = new HColumnDescriptor(xtracolName);
558    xtracol.setValue(xtracolName, xtracolName);
559    expectedException = false;
560    try {
561      this.admin.addColumnFamily(tableName, xtracol);
562    } catch (TableNotDisabledException re) {
563      expectedException = true;
564    }
565    // Add column should work even if the table is enabled
566    assertFalse(expectedException);
567    modifiedHtd = this.admin.getTableDescriptor(tableName);
568    hcd = modifiedHtd.getFamily(xtracol.getName());
569    assertTrue(hcd != null);
570    assertTrue(hcd.getValue(xtracolName).equals(xtracolName));
571
572    // Delete the just-added column.
573    this.admin.deleteColumnFamily(tableName, xtracol.getName());
574    modifiedHtd = this.admin.getTableDescriptor(tableName);
575    hcd = modifiedHtd.getFamily(xtracol.getName());
576    assertTrue(hcd == null);
577
578    // Delete the table
579    this.admin.disableTable(tableName);
580    this.admin.deleteTable(tableName);
581    this.admin.listTables();
582    assertFalse(this.admin.tableExists(tableName));
583  }

585  protected void verifyRoundRobinDistribution(ClusterConnection c, RegionLocator regionLocator, int
586      expectedRegions) throws IOException {
587    int numRS = c.getCurrentNrHRS();
588    List<HRegionLocation> regions = regionLocator.getAllRegionLocations();
589    Map<ServerName, List<RegionInfo>> server2Regions = new HashMap<>();
590    for (HRegionLocation loc : regions) {
591      ServerName server = loc.getServerName();
592      List<RegionInfo> regs = server2Regions.get(server);
593      if (regs == null) {
594        regs = new ArrayList<>();
595        server2Regions.put(server, regs);
596      }
597      regs.add(loc.getRegionInfo());
598    }
599    boolean tablesOnMaster = LoadBalancer.isTablesOnMaster(TEST_UTIL.getConfiguration());
600    if (tablesOnMaster) {
601      // Ignore the master region server,
602      // which contains less regions by intention.
603      numRS--;
604    }
605    float average = (float) expectedRegions/numRS;
606    int min = (int)Math.floor(average);
607    int max = (int)Math.ceil(average);
608    for (List<RegionInfo> regionList : server2Regions.values()) {
609      assertTrue("numRS=" + numRS + ", min=" + min + ", max=" + max +
610        ", size=" + regionList.size() + ", tablesOnMaster=" + tablesOnMaster,
611      regionList.size() == min || regionList.size() == max);
612    }
613  }

986  void splitTest(byte[] splitPoint, byte[][] familyNames, int[] rowCounts,
987    int numVersions, int blockSize, boolean async) throws Exception {
988    TableName tableName = TableName.valueOf("testForceSplit");
989    StringBuilder sb = new StringBuilder();
990    // Add tail to String so can see better in logs where a test is running.
991    for (int i = 0; i < rowCounts.length; i++) {
992      sb.append("_").append(Integer.toString(rowCounts[i]));
993    }
994    assertFalse(admin.tableExists(tableName));
995    try (final Table table = TEST_UTIL.createTable(tableName, familyNames,
996      numVersions, blockSize);
997      final RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
998
999      int rowCount = 0;
1000      byte[] q = new byte[0];
1001
1002      // insert rows into column families. The number of rows that have values
1003      // in a specific column family is decided by rowCounts[familyIndex]
1004      for (int index = 0; index < familyNames.length; index++) {
1005        ArrayList<Put> puts = new ArrayList<>(rowCounts[index]);
1006        for (int i = 0; i < rowCounts[index]; i++) {
1007          byte[] k = Bytes.toBytes(i);
1008          Put put = new Put(k);
1009          put.addColumn(familyNames[index], q, k);
1010          puts.add(put);
1011        }
1012        table.put(puts);
1013
1014        if (rowCount < rowCounts[index]) {
1015          rowCount = rowCounts[index];
1016        }
1017      }
1018
1019      // get the initial layout (should just be one region)
1020      List<HRegionLocation> m = locator.getAllRegionLocations();
1021      LOG.info("Initial regions (" + m.size() + "): " + m);
1022      assertTrue(m.size() == 1);
1023
1024      // Verify row count
1025      Scan scan = new Scan();
1026      ResultScanner scanner = table.getScanner(scan);
1027      int rows = 0;
1028      for (@SuppressWarnings("unused") Result result : scanner) {
1029        rows++;
1030      }
1031      scanner.close();
1032      assertEquals(rowCount, rows);
1033
1034      // Have an outstanding scan going on to make sure we can scan over splits.
1035      scan = new Scan();
1036      scanner = table.getScanner(scan);
1037      // Scan first row so we are into first region before split happens.
1038      scanner.next();
1039
1040      // Split the table
1041      if (async) {
1042        this.admin.split(tableName, splitPoint);
1043        final AtomicInteger count = new AtomicInteger(0);
1044        Thread t = new Thread("CheckForSplit") {
1045          @Override public void run() {
1046            for (int i = 0; i < 45; i++) {
1047              try {
1048                sleep(1000);
1049              } catch (InterruptedException e) {
1050                continue;
1051              }
1052              // check again
1053              List<HRegionLocation> regions = null;
1054              try {
1055                regions = locator.getAllRegionLocations();
1056              } catch (IOException e) {
1057                e.printStackTrace();
1058              }
1059              if (regions == null) continue;
1060              count.set(regions.size());
1061              if (count.get() >= 2) {
1062                LOG.info("Found: " + regions);
1063                break;
1064              }
1065              LOG.debug("Cycle waiting on split");
1066            }
1067            LOG.debug("CheckForSplit thread exited, current region count: " + count.get());
1068          }
1069        };
1070        t.setPriority(Thread.NORM_PRIORITY - 2);
1071        t.start();
1072        t.join();
1073      } else {
1074        // Sync split region, no need to create a thread to check
1075        ((HBaseAdmin)admin).splitRegionSync(m.get(0).getRegionInfo().getRegionName(), splitPoint);
1076      }
1077
1078      // Verify row count
1079      rows = 1; // We counted one row above.
1080      for (@SuppressWarnings("unused") Result result : scanner) {
1081        rows++;
1082        if (rows > rowCount) {
1083          scanner.close();
1084          assertTrue("Scanned more than expected (" + rowCount + ")", false);
1085        }
1086      }
1087      scanner.close();
1088      assertEquals(rowCount, rows);
1089
1090      List<HRegionLocation> regions = null;
1091      try {
1092        regions = locator.getAllRegionLocations();
1093      } catch (IOException e) {
1094        e.printStackTrace();
1095      }
1096      assertEquals(2, regions.size());
1097      if (splitPoint != null) {
1098        // make sure the split point matches our explicit configuration
1099        assertEquals(Bytes.toString(splitPoint),
1100            Bytes.toString(regions.get(0).getRegionInfo().getEndKey()));
1101        assertEquals(Bytes.toString(splitPoint),
1102            Bytes.toString(regions.get(1).getRegionInfo().getStartKey()));
1103        LOG.debug("Properly split on " + Bytes.toString(splitPoint));
1104      } else {
1105        if (familyNames.length > 1) {
1106          int splitKey = Bytes.toInt(regions.get(0).getRegionInfo().getEndKey());
1107          // check if splitKey is based on the largest column family
1108          // in terms of it store size
1109          int deltaForLargestFamily = Math.abs(rowCount / 2 - splitKey);
1110          LOG.debug("SplitKey=" + splitKey + "&deltaForLargestFamily=" + deltaForLargestFamily +
1111              ", r=" + regions.get(0).getRegionInfo());
1112          for (int index = 0; index < familyNames.length; index++) {
1113            int delta = Math.abs(rowCounts[index] / 2 - splitKey);
1114            if (delta < deltaForLargestFamily) {
1115              assertTrue("Delta " + delta + " for family " + index + " should be at least "
1116                      + "deltaForLargestFamily " + deltaForLargestFamily, false);
1117            }
1118          }
1119        }
1120      }
1121      TEST_UTIL.deleteTable(tableName);
1122    }
1123  }

1313    p = new Put(Bytes.toBytes("rep1_rk"));
1314    p.addColumn(Bytes.toBytes(fn1), q1, v1);

1334              .getFileStatus().getReplication());

1337    } finally {
