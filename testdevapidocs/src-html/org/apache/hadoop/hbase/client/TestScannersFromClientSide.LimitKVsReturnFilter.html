114  /**
115   * JUnit does not provide an easy way to run a hook after each parameterized run. Without that
116   * there is no easy way to restart the test cluster after each parameterized run. Annotation
117   * BeforeParam does not work either because it runs before parameterization and hence does not
118   * have access to the test parameters (which is weird).
119   *
120   * This *hack* checks if the current instance of test cluster configuration has the passed
121   * parameterized configs. In such a case, we can just reuse the cluster for test and do not need
122   * to initialize from scratch. While this is a hack, it saves a ton of time for the full
123   * test and de-flakes it.
124   */
125  private static boolean isSameParameterizedCluster(Class<?> registryImpl, int numHedgedReqs) {
126    // initialize() is called for every unit test, however we only want to reset the cluster state
127    // at the end of every parameterized run.
128    if (TEST_UTIL == null) {
129      return false;
130    }
131    Configuration conf = TEST_UTIL.getConfiguration();
132    Class<?> confClass = conf.getClass(HConstants.CLIENT_CONNECTION_REGISTRY_IMPL_CONF_KEY,
133      ZKConnectionRegistry.class);
134    int hedgedReqConfig = conf.getInt(MasterRegistry.MASTER_REGISTRY_HEDGED_REQS_FANOUT_KEY,
135      MasterRegistry.MASTER_REGISTRY_HEDGED_REQS_FANOUT_DEFAULT);
136    return confClass.getName().equals(registryImpl.getName()) && numHedgedReqs == hedgedReqConfig;
137  }
138
139  public TestScannersFromClientSide(Class<?> registryImpl, int numHedgedReqs) throws Exception {
140    if (isSameParameterizedCluster(registryImpl, numHedgedReqs)) {
141      return;
142    }
143    if (TEST_UTIL != null) {
144      // We reached the end of a parameterized run, clean up the cluster.
145      TEST_UTIL.shutdownMiniCluster();
146    }
147    TEST_UTIL = new HBaseTestingUtility();
148    Configuration conf = TEST_UTIL.getConfiguration();
149    conf.setLong(HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY, 10 * 1024 * 1024);
150    conf.setClass(HConstants.CLIENT_CONNECTION_REGISTRY_IMPL_CONF_KEY, registryImpl,
151        ConnectionRegistry.class);
152    Preconditions.checkArgument(numHedgedReqs > 0);
153    conf.setInt(MasterRegistry.MASTER_REGISTRY_HEDGED_REQS_FANOUT_KEY, numHedgedReqs);
154    StartMiniClusterOption.Builder builder = StartMiniClusterOption.builder();
155    // Multiple masters needed only when hedged reads for master registry are enabled.
156    builder.numMasters(numHedgedReqs > 1 ? 3 : 1).numRegionServers(3);
157    TEST_UTIL.startMiniCluster(builder.build());
158  }

233  @Test
234  public void testMaxResultSizeIsSetToDefault() throws Exception {
235    final TableName tableName = name.getTableName();
236    Table ht = TEST_UTIL.createTable(tableName, FAMILY);
237
238    // The max result size we expect the scan to use by default.
239    long expectedMaxResultSize =
240        TEST_UTIL.getConfiguration().getLong(HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY,
241          HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE);
242
243    int numRows = 5;
244    byte[][] ROWS = HTestConst.makeNAscii(ROW, numRows);
245
246    int numQualifiers = 10;
247    byte[][] QUALIFIERS = HTestConst.makeNAscii(QUALIFIER, numQualifiers);
248
249    // Specify the cell size such that a single row will be larger than the default
250    // value of maxResultSize. This means that Scan RPCs should return at most a single
251    // result back to the client.
252    int cellSize = (int) (expectedMaxResultSize / (numQualifiers - 1));
253    byte[] cellValue = Bytes.createMaxByteArray(cellSize);
254
255    Put put;
256    List<Put> puts = new ArrayList<>();
257    for (int row = 0; row < ROWS.length; row++) {
258      put = new Put(ROWS[row]);
259      for (int qual = 0; qual < QUALIFIERS.length; qual++) {
260        KeyValue kv = new KeyValue(ROWS[row], FAMILY, QUALIFIERS[qual], cellValue);
261        put.add(kv);
262      }
263      puts.add(put);
264    }
265    ht.put(puts);
266
267    // Create a scan with the default configuration.
268    Scan scan = new Scan();
269
270    try (ResultScanner scanner = ht.getScanner(scan)) {
271      assertThat(scanner, instanceOf(AsyncTableResultScanner.class));
272      scanner.next();
273      AsyncTableResultScanner s = (AsyncTableResultScanner) scanner;
274      // The scanner should have, at most, a single result in its cache. If there more results
275      // exists
276      // in the cache it means that more than the expected max result size was fetched.
277      assertTrue("The cache contains: " + s.getCacheSize() + " results", s.getCacheSize() <= 1);
278    }
279  }

356  private void verifyExpectedCounts(Table table, Scan scan, int expectedRowCount,
357      int expectedCellCount) throws Exception {
358    ResultScanner scanner = table.getScanner(scan);
359
360    int rowCount = 0;
361    int cellCount = 0;
362    Result r = null;
363    while ((r = scanner.next()) != null) {
364      rowCount++;
365      cellCount += r.rawCells().length;
366    }
367
368    assertTrue("Expected row count: " + expectedRowCount + " Actual row count: " + rowCount,
369        expectedRowCount == rowCount);
370    assertTrue("Expected cell count: " + expectedCellCount + " Actual cell count: " + cellCount,
371        expectedCellCount == cellCount);
372    scanner.close();
373  }

735  static void verifyResult(Result result, List<Cell> expKvList, boolean toLog,
736      String msg) {
737
738    LOG.info(msg);
739    LOG.info("Expected count: " + expKvList.size());
740    LOG.info("Actual count: " + result.size());
741    if (expKvList.isEmpty()) {
742      return;
743    }
744
745    int i = 0;
746    for (Cell kv : result.rawCells()) {
747      if (i >= expKvList.size()) {
748        break;  // we will check the size later
749      }
750
751      Cell kvExp = expKvList.get(i++);
752      if (toLog) {
753        LOG.info("get kv is: " + kv.toString());
754        LOG.info("exp kv is: " + kvExp.toString());
755      }
756      assertTrue("Not equal", kvExp.equals(kv));
757    }
758
759    assertEquals(expKvList.size(), result.size());
760  }

924    private int cellCount = 0;

926    @Override
927    public ReturnCode filterCell(Cell v) throws IOException {
928      if (cellCount >= 6) {
929        cellCount++;
930        return ReturnCode.SKIP;
931      }
932      cellCount++;
933      return ReturnCode.INCLUDE;
934    }

937    public boolean filterAllRemaining() throws IOException {
938      if (cellCount < 7) {
939        return false;
940      }
941      cellCount++;
942      return true;
943    }

947      return this.getClass().getSimpleName();
948    }
