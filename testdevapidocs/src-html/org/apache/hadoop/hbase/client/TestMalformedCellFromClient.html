103  /**
104   * The purpose of this ut is to check the consistency between the exception and results.
105   * If the RetriesExhaustedWithDetailsException contains the whole batch,
106   * each result should be of IOE. Otherwise, the row operation which is not in the exception
107   * should have a true result.
108   */
109  @Test
110  public void testRegionException() throws InterruptedException, IOException {
111    List<Row> batches = new ArrayList<>();
112    batches.add(new Put(Bytes.toBytes("good")).addColumn(FAMILY, null, new byte[10]));
113    // the rm is used to prompt the region exception.
114    // see RSRpcServices#multi
115    RowMutations rm = new RowMutations(Bytes.toBytes("fail"));
116    rm.add(new Put(rm.getRow()).addColumn(FAMILY, null, new byte[CELL_SIZE]));
117    batches.add(rm);
118    Object[] results = new Object[batches.size()];
119
120    try (Table table = TEST_UTIL.getConnection().getTable(TABLE_NAME)) {
121      Throwable exceptionByCaught = null;
122      try {
123        table.batch(batches, results);
124        fail("Where is the exception? We put the malformed cells!!!");
125      } catch (RetriesExhaustedException e) {
126        exceptionByCaught = e.getCause();
127      }
128      for (Object obj : results) {
129        assertNotNull(obj);
130      }
131      assertEquals(Result.class, results[0].getClass());
132      assertEquals(exceptionByCaught.getClass(), results[1].getClass());
133      Result result = table.get(new Get(Bytes.toBytes("good")));
134      assertEquals(1, result.size());
135      Cell cell = result.getColumnLatestCell(FAMILY, null);
136      assertTrue(Bytes.equals(CellUtil.cloneValue(cell), new byte[10]));
137    }
138  }

140  /**
141   * This test verifies region exception doesn't corrupt the results of batch. The prescription is
142   * shown below. 1) honor the action result rather than region exception. If the action have both
143   * of true result and region exception, the action is fine as the exception is caused by other
144   * actions which are in the same region. 2) honor the action exception rather than region
145   * exception. If the action have both of action exception and region exception, we deal with the
146   * action exception only. If we also handle the region exception for the same action, it will
147   * introduce the negative count of actions in progress. The AsyncRequestFuture#waitUntilDone will
148   * block forever. If the RetriesExhaustedWithDetailsException contains the whole batch, each
149   * result should be of IOE. Otherwise, the row operation which is not in the exception should have
150   * a true result. The no-cluster test is in TestAsyncProcessWithRegionException.
151   */
152  @Test
153  public void testRegionExceptionByAsync() throws Exception {
154    List<Row> batches = new ArrayList<>();
155    batches.add(new Put(Bytes.toBytes("good")).addColumn(FAMILY, null, new byte[10]));
156    // the rm is used to prompt the region exception.
157    // see RSRpcServices#multi
158    RowMutations rm = new RowMutations(Bytes.toBytes("fail"));
159    rm.add(new Put(rm.getRow()).addColumn(FAMILY, null, new byte[CELL_SIZE]));
160    batches.add(rm);
161    try (AsyncConnection asyncConnection = ConnectionFactory
162      .createAsyncConnection(TEST_UTIL.getConfiguration()).get()) {
163      AsyncTable<AdvancedScanResultConsumer> table = asyncConnection.getTable(TABLE_NAME);
164      List<CompletableFuture<AdvancedScanResultConsumer>> results = table.batch(batches);
165      assertEquals(2, results.size());
166      try {
167        results.get(1).get();
168        fail("Where is the exception? We put the malformed cells!!!");
169      } catch (ExecutionException e) {
170        // pass
171      }
172      Result result = table.get(new Get(Bytes.toBytes("good"))).get();
173      assertEquals(1, result.size());
174      Cell cell = result.getColumnLatestCell(FAMILY, null);
175      assertTrue(Bytes.equals(CellUtil.cloneValue(cell), new byte[10]));
176    }
177  }

179  /**
180   * The invalid cells is in rm. The rm should fail but the subsequent mutations should succeed.
181   * Currently, we have no client api to submit the request consisting of condition-rm and mutation.
182   * Hence, this test build the request manually.
183   */
184  @Test
185  public void testAtomicOperations() throws Exception {
186    RowMutations rm = new RowMutations(Bytes.toBytes("fail"));
187    rm.add(new Put(rm.getRow()).addColumn(FAMILY, null, new byte[CELL_SIZE]));
188    rm.add(new Put(rm.getRow()).addColumn(FAMILY, null, new byte[10]));
189    Put put = new Put(Bytes.toBytes("good")).addColumn(FAMILY, null, new byte[10]);
190
191    // build the request
192    HRegion r = TEST_UTIL.getMiniHBaseCluster().getRegions(TABLE_NAME).get(0);
193    ClientProtos.MultiRequest request =
194      ClientProtos.MultiRequest.newBuilder(createRequest(rm, r.getRegionInfo().getRegionName()))
195        .addRegionAction(ClientProtos.RegionAction.newBuilder().setRegion(RequestConverter
196          .buildRegionSpecifier(HBaseProtos.RegionSpecifier.RegionSpecifierType.REGION_NAME,
197            r.getRegionInfo().getRegionName())).addAction(ClientProtos.Action.newBuilder()
198          .setMutation(
199            ProtobufUtil.toMutationNoData(ClientProtos.MutationProto.MutationType.PUT, put))))
200        .build();
201
202    List<Cell> cells = new ArrayList<>();
203    for (Mutation m : rm.getMutations()) {
204      cells.addAll(m.getCellList(FAMILY));
205    }
206    cells.addAll(put.getCellList(FAMILY));
207    assertEquals(3, cells.size());
208    HBaseRpcController controller = Mockito.mock(HBaseRpcController.class);
209    Mockito.when(controller.cellScanner()).thenReturn(CellUtil.createCellScanner(cells));
210    HRegionServer rs = TEST_UTIL.getMiniHBaseCluster().getRegionServer(
211      TEST_UTIL.getMiniHBaseCluster()
212        .getServerHoldingRegion(TABLE_NAME, r.getRegionInfo().getRegionName()));
213
214    ClientProtos.MultiResponse response = rs.getRSRpcServices().multi(controller, request);
215    assertEquals(2, response.getRegionActionResultCount());
216    assertTrue(response.getRegionActionResultList().get(0).hasException());
217    assertFalse(response.getRegionActionResultList().get(1).hasException());
218    assertEquals(1, response.getRegionActionResultList().get(1).getResultOrExceptionCount());
219    assertTrue(
220      response.getRegionActionResultList().get(1).getResultOrExceptionList().get(0).hasResult());
221    try (Table table = TEST_UTIL.getConnection().getTable(TABLE_NAME)) {
222      Result result = table.get(new Get(Bytes.toBytes("good")));
223      assertEquals(1, result.size());
224      Cell cell = result.getColumnLatestCell(FAMILY, null);
225      assertTrue(Bytes.equals(CellUtil.cloneValue(cell), new byte[10]));
226    }
227  }

229  private static ClientProtos.MultiRequest createRequest(RowMutations rm, byte[] regionName)
230    throws IOException {
231    ClientProtos.RegionAction.Builder builder = RequestConverter
232      .getRegionActionBuilderWithRegion(ClientProtos.RegionAction.newBuilder(), regionName);
233    builder.setAtomic(true);
234    ClientProtos.Action.Builder actionBuilder = ClientProtos.Action.newBuilder();
235    ClientProtos.MutationProto.Builder mutationBuilder = ClientProtos.MutationProto.newBuilder();
236    ClientProtos.Condition condition = ProtobufUtil.toCondition(rm.getRow(), FAMILY, null,
237      CompareOperator.EQUAL, new byte[10], null, null);
238    for (Mutation mutation : rm.getMutations()) {
239      ClientProtos.MutationProto.MutationType mutateType = null;
240      if (mutation instanceof Put) {
241        mutateType = ClientProtos.MutationProto.MutationType.PUT;
242      } else if (mutation instanceof Delete) {
243        mutateType = ClientProtos.MutationProto.MutationType.DELETE;
244      } else {
245        throw new DoNotRetryIOException(
246          "RowMutations supports only put and delete, not " + mutation.getClass().getName());
247      }
248      mutationBuilder.clear();
249      ClientProtos.MutationProto mp =
250        ProtobufUtil.toMutationNoData(mutateType, mutation, mutationBuilder);
251      actionBuilder.clear();
252      actionBuilder.setMutation(mp);
253      builder.addAction(actionBuilder.build());
254    }
255    ClientProtos.MultiRequest request = ClientProtos.MultiRequest.newBuilder()
256        .addRegionAction(builder.setCondition(condition).build()).build();
257    return request;
258  }

260  /**
261   * This test depends on how regionserver process the batch ops.
262   * 1) group the put/delete until meeting the increment
263   * 2) process the batch of put/delete
264   * 3) process the increment
265   * see RSRpcServices#doNonAtomicRegionMutation
266   */
267  @Test
268  public void testNonAtomicOperations() throws InterruptedException, IOException {
269    Increment inc = new Increment(Bytes.toBytes("good")).addColumn(FAMILY, null, 100);
270    List<Row> batches = new ArrayList<>();
271    // the first and second puts will be group by regionserver
272    batches.add(new Put(Bytes.toBytes("fail")).addColumn(FAMILY, null, new byte[CELL_SIZE]));
273    batches.add(new Put(Bytes.toBytes("fail")).addColumn(FAMILY, null, new byte[CELL_SIZE]));
274    // this Increment should succeed
275    batches.add(inc);
276    // this put should succeed
277    batches.add(new Put(Bytes.toBytes("good")).addColumn(FAMILY, null, new byte[1]));
278    Object[] objs = new Object[batches.size()];
279    try (Table table = TEST_UTIL.getConnection().getTable(TABLE_NAME)) {
280      table.batch(batches, objs);
281      fail("Where is the exception? We put the malformed cells!!!");
282    } catch (RetriesExhaustedException e) {
283      Throwable error = e.getCause();
284      for (;;) {
285        assertNotNull("Can not find a DoNotRetryIOException on stack trace", error);
286        if (error instanceof DoNotRetryIOException) {
287          break;
288        }
289        error = error.getCause();
290      }
291    } finally {
292      assertObjects(objs, batches.size());
293      assertTrue(objs[0] instanceof IOException);
294      assertTrue(objs[1] instanceof IOException);
295      assertEquals(Result.class, objs[2].getClass());
296      assertEquals(Result.class, objs[3].getClass());
297    }
298  }

300  @Test
301  public void testRowMutations() throws InterruptedException, IOException {
302    Put put = new Put(Bytes.toBytes("good")).addColumn(FAMILY, null, new byte[1]);
303    List<Row> batches = new ArrayList<>();
304    RowMutations mutations = new RowMutations(Bytes.toBytes("fail"));
305    // the first and second puts will be group by regionserver
306    mutations.add(new Put(Bytes.toBytes("fail")).addColumn(FAMILY, null, new byte[CELL_SIZE]));
307    mutations.add(new Put(Bytes.toBytes("fail")).addColumn(FAMILY, null, new byte[CELL_SIZE]));
308    batches.add(mutations);
309    // this bm should succeed
310    mutations = new RowMutations(Bytes.toBytes("good"));
311    mutations.add(put);
312    mutations.add(put);
313    batches.add(mutations);
314    Object[] objs = new Object[batches.size()];
315    try (Table table = TEST_UTIL.getConnection().getTable(TABLE_NAME)) {
316      table.batch(batches, objs);
317      fail("Where is the exception? We put the malformed cells!!!");
318    } catch (RetriesExhaustedException e) {
319      Throwable error = e.getCause();
320      for (;;) {
321        assertNotNull("Can not find a DoNotRetryIOException on stack trace", error);
322        if (error instanceof DoNotRetryIOException) {
323          break;
324        }
325        error = error.getCause();
326      }
327    } finally {
328      assertObjects(objs, batches.size());
329      assertTrue(objs[0] instanceof IOException);
330      assertEquals(Result.class, objs[1].getClass());
331    }
332  }

334  private static void assertObjects(Object[] objs, int expectedSize) {
335    int count = 0;
336    for (Object obj : objs) {
337      assertNotNull(obj);
338      ++count;
339    }
340    assertEquals(expectedSize, count);
341  }
