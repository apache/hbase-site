020import java.io.IOException;
021
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.RegionLocations;
024import org.apache.hadoop.hbase.TableName;
025import org.apache.hadoop.hbase.HRegionInfo;
026import org.apache.hadoop.hbase.HRegionLocation;
027import org.apache.hadoop.hbase.ServerName;
028import org.apache.hadoop.hbase.ZooKeeperConnectionException;
029import org.apache.hadoop.hbase.protobuf.generated.AdminProtos;
030import org.apache.hadoop.hbase.protobuf.generated.ClientProtos;
031import org.apache.hadoop.hbase.ipc.RpcControllerFactory;

034/**
035 * {@link ClusterConnection} testing utility.
036 */
037public class HConnectionTestingUtility {
038  /*
039   * Not part of {@link HBaseTestingUtility} because this class is not
040   * in same package as {@link HConnection}.  Would have to reveal ugly
041   * {@link ConnectionImplementation} innards to HBaseTestingUtility to give it access.
042   */
043  /**
044   * Get a Mocked {@link HConnection} that goes with the passed <code>conf</code>
045   * configuration instance.  Minimally the mock will return
046   * <code>conf</conf> when {@link ClusterConnection#getConfiguration()} is invoked.
047   * Be sure to shutdown the connection when done by calling
048   * {@link Connection#close()} else it will stick around; this is probably not what you want.
049   * @param conf configuration
050   * @return HConnection object for <code>conf</code>
051   * @throws ZooKeeperConnectionException
052   */
053  public static ClusterConnection getMockedConnection(final Configuration conf)
054  throws ZooKeeperConnectionException {
055    ConnectionImplementation connection = Mockito.mock(ConnectionImplementation.class);
056    Mockito.when(connection.getConfiguration()).thenReturn(conf);
057    return connection;
058  }
059
060  /**
061   * Calls {@link #getMockedConnection(Configuration)} and then mocks a few
062   * more of the popular {@link ClusterConnection} methods so they do 'normal'
063   * operation (see return doc below for list). Be sure to shutdown the
064   * connection when done by calling {@link Connection#close()} else it will stick around;
065   * this is probably not what you want.
066   *
067   * @param conf Configuration to use
068   * @param admin An AdminProtocol; can be null but is usually
069   * itself a mock.
070   * @param client A ClientProtocol; can be null but is usually
071   * itself a mock.
072   * @param sn ServerName to include in the region location returned by this
073   * <code>connection</code>
074   * @param hri HRegionInfo to include in the location returned when
075   * getRegionLocator is called on the mocked connection
076   * @return Mock up a connection that returns a {@link Configuration} when
077   * {@link ClusterConnection#getConfiguration()} is called, a 'location' when
078   * {@link ClusterConnection#getRegionLocation(org.apache.hadoop.hbase.TableName, byte[], boolean)}
079   * is called,
080   * and that returns the passed {@link AdminProtos.AdminService.BlockingInterface} instance when
081   * {@link ClusterConnection#getAdmin(ServerName)} is called, returns the passed
082   * {@link ClientProtos.ClientService.BlockingInterface} instance when
083   * {@link ClusterConnection#getClient(ServerName)} is called (Be sure to call
084   * {@link Connection#close()} when done with this mocked Connection.
085   * @throws IOException
086   */
087  public static ClusterConnection getMockedConnectionAndDecorate(final Configuration conf,
088      final AdminProtos.AdminService.BlockingInterface admin,
089      final ClientProtos.ClientService.BlockingInterface client,
090      final ServerName sn, final HRegionInfo hri)
091  throws IOException {
092    ConnectionImplementation c = Mockito.mock(ConnectionImplementation.class);
093    Mockito.when(c.getConfiguration()).thenReturn(conf);
094    Mockito.doNothing().when(c).close();
095    // Make it so we return a particular location when asked.
096    final HRegionLocation loc = new HRegionLocation(hri, sn);
097    Mockito.when(c.getRegionLocation((TableName) Mockito.any(),
098        (byte[]) Mockito.any(), Mockito.anyBoolean())).
099      thenReturn(loc);
100    Mockito.when(c.locateRegion((TableName) Mockito.any(), (byte[]) Mockito.any())).
101      thenReturn(loc);
102    Mockito.when(c.locateRegion((TableName) Mockito.any(), (byte[]) Mockito.any(),
103        Mockito.anyBoolean(), Mockito.anyBoolean(),  Mockito.anyInt()))
104        .thenReturn(new RegionLocations(loc));
105    if (admin != null) {
106      // If a call to getAdmin, return this implementation.
107      Mockito.when(c.getAdmin(Mockito.any(ServerName.class))).
108        thenReturn(admin);
109    }
110    if (client != null) {
111      // If a call to getClient, return this client.
112      Mockito.when(c.getClient(Mockito.any(ServerName.class))).
113        thenReturn(client);
114    }
115    NonceGenerator ng = Mockito.mock(NonceGenerator.class);
116    Mockito.when(c.getNonceGenerator()).thenReturn(ng);
117    Mockito.when(c.getAsyncProcess()).thenReturn(
118      new AsyncProcess(c, conf, null, RpcRetryingCallerFactory.instantiate(conf), false,
119          RpcControllerFactory.instantiate(conf)));
120    Mockito.when(c.getNewRpcRetryingCallerFactory(conf)).thenReturn(
121        RpcRetryingCallerFactory.instantiate(conf,
122            RetryingCallerInterceptorFactory.NO_OP_INTERCEPTOR, null));
123    HTableInterface t = Mockito.mock(HTableInterface.class);
124    Mockito.when(c.getTable((TableName)Mockito.any())).thenReturn(t);
125    ResultScanner rs = Mockito.mock(ResultScanner.class);
126    Mockito.when(t.getScanner((Scan)Mockito.any())).thenReturn(rs);
127    return c;
128  }
129
130  /**
131   * Get a Mockito spied-upon {@link ClusterConnection} that goes with the passed
132   * <code>conf</code> configuration instance.
133   * Be sure to shutdown the connection when done by calling
134   * {@link Connection#close()} else it will stick around; this is probably not what you want.
135   * @param conf configuration
136   * @return HConnection object for <code>conf</code>
137   * @throws ZooKeeperConnectionException
138   * @see @link
139   * {http://mockito.googlecode.com/svn/branches/1.6/javadoc/org/mockito/Mockito.html#spy(T)}
140   */
141  public static ClusterConnection getSpiedConnection(final Configuration conf)
142  throws IOException {
143    ConnectionImplementation connection =
144      Mockito.spy(new ConnectionImplementation(conf, null, null));
145    return connection;
146  }
147}
