129  @Test
130  public void testRollWALWALWriter() throws Exception {
131    setUpforLogRolling();
132    String className = this.getClass().getName();
133    StringBuilder v = new StringBuilder(className);
134    while (v.length() < 1000) {
135      v.append(className);
136    }
137    byte[] value = Bytes.toBytes(v.toString());
138    HRegionServer regionServer = startAndWriteData(tableName, value);
139    LOG.info("after writing there are "
140        + AbstractFSWALProvider.getNumRolledLogFiles(regionServer.getWAL(null)) + " log files");
141
142    // flush all regions
143    for (HRegion r : regionServer.getOnlineRegionsLocalContext()) {
144      r.flush(true);
145    }
146    admin.rollWALWriter(regionServer.getServerName()).join();
147    int count = AbstractFSWALProvider.getNumRolledLogFiles(regionServer.getWAL(null));
148    LOG.info("after flushing all regions and rolling logs there are " +
149        count + " log files");
150    assertTrue(("actual count: " + count), count <= 2);
151  }

153  private void setUpforLogRolling() {
154    // Force a region split after every 768KB
155    TEST_UTIL.getConfiguration().setLong(HConstants.HREGION_MAX_FILESIZE,
156        768L * 1024L);
157
158    // We roll the log after every 32 writes
159    TEST_UTIL.getConfiguration().setInt("hbase.regionserver.maxlogentries", 32);
160
161    TEST_UTIL.getConfiguration().setInt(
162        "hbase.regionserver.logroll.errors.tolerated", 2);
163    TEST_UTIL.getConfiguration().setInt("hbase.rpc.timeout", 10 * 1000);
164
165    // For less frequently updated regions flush after every 2 flushes
166    TEST_UTIL.getConfiguration().setInt(
167        "hbase.hregion.memstore.optionalflushcount", 2);
168
169    // We flush the cache after every 8192 bytes
170    TEST_UTIL.getConfiguration().setInt(HConstants.HREGION_MEMSTORE_FLUSH_SIZE,
171        8192);
172
173    // Increase the amount of time between client retries
174    TEST_UTIL.getConfiguration().setLong("hbase.client.pause", 10 * 1000);
175
176    // Reduce thread wake frequency so that other threads can get
177    // a chance to run.
178    TEST_UTIL.getConfiguration().setInt(HConstants.THREAD_WAKE_FREQUENCY,
179        2 * 1000);
180
181    /**** configuration for testLogRollOnDatanodeDeath ****/
182    // lower the namenode & datanode heartbeat so the namenode
183    // quickly detects datanode failures
184    TEST_UTIL.getConfiguration().setInt("dfs.namenode.heartbeat.recheck-interval", 5000);
185    TEST_UTIL.getConfiguration().setInt("dfs.heartbeat.interval", 1);
186    // the namenode might still try to choose the recently-dead datanode
187    // for a pipeline, so try to a new pipeline multiple times
188    TEST_UTIL.getConfiguration().setInt("dfs.client.block.write.retries", 30);
189    TEST_UTIL.getConfiguration().setInt(
190        "hbase.regionserver.hlog.tolerable.lowreplication", 2);
191    TEST_UTIL.getConfiguration().setInt(
192        "hbase.regionserver.hlog.lowreplication.rolllimit", 3);
193  }

195  private HRegionServer startAndWriteData(TableName tableName, byte[] value) throws Exception {
196    createTableWithDefaultConf(tableName);
197    RawAsyncTable table = ASYNC_CONN.getRawTable(tableName);
198    HRegionServer regionServer = TEST_UTIL.getRSForFirstRegionInTable(tableName);
199    for (int i = 1; i <= 256; i++) { // 256 writes should cause 8 log rolls
200      Put put = new Put(Bytes.toBytes("row" + String.format("%1$04d", i)));
201      put.addColumn(FAMILY, null, value);
202      table.put(put).join();
203      if (i % 32 == 0) {
204        // After every 32 writes sleep to let the log roller run
205        try {
206          Thread.sleep(2000);
207        } catch (InterruptedException e) {
208          // continue
209        }
210      }
211    }
212    return regionServer;
213  }

215  @Test
216  public void testGetRegionLoads() throws Exception {
217    // Turn off the balancer
218    admin.setBalancerOn(false).join();
219    TableName[] tables =
220        new TableName[] { TableName.valueOf(tableName.getNameAsString() + "1"),
221            TableName.valueOf(tableName.getNameAsString() + "2"),
222            TableName.valueOf(tableName.getNameAsString() + "3") };
223    createAndLoadTable(tables);
224    // Sleep to wait region server report
225    Thread.sleep(TEST_UTIL.getConfiguration().getInt("hbase.regionserver.msginterval", 3 * 1000) * 2);
226    // Check if regions match with the regionLoad from the server
227    Collection<ServerName> servers = admin.getRegionServers().get();
228    for (ServerName serverName : servers) {
229      List<RegionInfo> regions = admin.getOnlineRegions(serverName).get();
230      checkRegionsAndRegionLoads(regions, admin.getRegionLoads(serverName).get());
231    }
232
233    // Check if regionLoad matches the table's regions and nothing is missed
234    for (TableName table : tables) {
235      List<RegionInfo> tableRegions = admin.getTableRegions(table).get();
236      List<RegionLoad> regionLoads = Lists.newArrayList();
237      for (ServerName serverName : servers) {
238        regionLoads.addAll(admin.getRegionLoads(serverName, Optional.of(table)).get());
239      }
240      checkRegionsAndRegionLoads(tableRegions, regionLoads);
241    }
242
243    // Check RegionLoad matches the regionLoad from ClusterStatus
244    ClusterStatus clusterStatus = admin.getClusterStatus(EnumSet.of(Option.LIVE_SERVERS)).get();
245    for (ServerName serverName : clusterStatus.getServers()) {
246      ServerLoad serverLoad = clusterStatus.getLoad(serverName);
247      compareRegionLoads(serverLoad.getRegionsLoad().values(), admin.getRegionLoads(serverName)
248          .get());
249    }
250  }
251
252  private void compareRegionLoads(Collection<RegionLoad> regionLoadCluster,
253      Collection<RegionLoad> regionLoads) {
254
255    assertEquals("No of regionLoads from clusterStatus and regionloads from RS doesn't match",
256      regionLoadCluster.size(), regionLoads.size());
257
258    for (RegionLoad loadCluster : regionLoadCluster) {
259      boolean matched = false;
260      for (RegionLoad load : regionLoads) {
261        if (Bytes.equals(loadCluster.getName(), load.getName())) {
262          matched = true;
263          continue;
264        }
265      }
266      assertTrue("The contents of region load from cluster and server should match", matched);
267    }
268  }
269
270  private void checkRegionsAndRegionLoads(Collection<RegionInfo> regions,
271      Collection<RegionLoad> regionLoads) {
272
273    assertEquals("No of regions and regionloads doesn't match", regions.size(), regionLoads.size());
274
275    Map<byte[], RegionLoad> regionLoadMap = Maps.newTreeMap(Bytes.BYTES_COMPARATOR);
276    for (RegionLoad regionLoad : regionLoads) {
277      regionLoadMap.put(regionLoad.getName(), regionLoad);
278    }
279    for (RegionInfo info : regions) {
280      assertTrue("Region not in regionLoadMap region:" + info.getRegionNameAsString()
281          + " regionMap: " + regionLoadMap, regionLoadMap.containsKey(info.getRegionName()));
282    }
283  }

285  private void createAndLoadTable(TableName[] tables) {
286    for (TableName table : tables) {
287      TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(table);
288      builder.addColumnFamily(ColumnFamilyDescriptorBuilder.of(FAMILY));
289      admin.createTable(builder.build(), Bytes.toBytes("aaaaa"), Bytes.toBytes("zzzzz"), 16).join();
290      RawAsyncTable asyncTable = ASYNC_CONN.getRawTable(table);
291      List<Put> puts = new ArrayList<>();
292      for (byte[] row : HBaseTestingUtility.ROWS) {
293        puts.add(new Put(row).addColumn(FAMILY, Bytes.toBytes("q"), Bytes.toBytes("v")));
294      }
295      asyncTable.putAll(puts).join();
296    }
297  }
