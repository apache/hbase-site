098  protected static void setupConf(Configuration conf) {
099    // disable the ui
100    conf.setInt("hbase.regionsever.info.port", -1);
101    // change the flush size to a small amount, regulating number of store files
102    conf.setInt("hbase.hregion.memstore.flush.size", 25000);
103    // so make sure we get a compaction when doing a load, but keep around some
104    // files in the store
105    conf.setInt("hbase.hstore.compaction.min", 10);
106    conf.setInt("hbase.hstore.compactionThreshold", 10);
107    // block writes if we get to 12 store files
108    conf.setInt("hbase.hstore.blockingStoreFiles", 12);
109    // Enable snapshot
110    conf.setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
111    conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
112        ConstantSizeRegionSplitPolicy.class.getName());
113
114  }

160  @Test
161  public void testSnapshotDeletionWithRegex() throws Exception {
162    Admin admin = UTIL.getAdmin();
163    // make sure we don't fail on listing snapshots
164    SnapshotTestingUtils.assertNoSnapshots(admin);
165
166    // put some stuff in the table
167    Table table = UTIL.getConnection().getTable(TABLE_NAME);
168    UTIL.loadTable(table, TEST_FAM);
169    table.close();
170
171    String snapshot1 = "TableSnapshot1";
172    admin.snapshot(snapshot1, TABLE_NAME);
173    LOG.debug("Snapshot1 completed.");
174
175    String snapshot2 = "TableSnapshot2";
176    admin.snapshot(snapshot2, TABLE_NAME);
177    LOG.debug("Snapshot2 completed.");
178
179    String snapshot3 = "3rdTableSnapshot";
180    admin.snapshot(snapshot3, TABLE_NAME);
181    LOG.debug(snapshot3 + " completed.");
182
183    // delete the first two snapshots
184    admin.deleteSnapshots(Pattern.compile("TableSnapshot.*"));
185    List<SnapshotDescription> snapshots = admin.listSnapshots();
186    assertEquals(1, snapshots.size());
187    assertEquals(snapshot3, snapshots.get(0).getName());
188
189    admin.deleteSnapshot(snapshot3);
190    admin.close();
191  }

196  @Test
197  public void testOfflineTableSnapshot() throws Exception {
198    Admin admin = UTIL.getAdmin();
199    // make sure we don't fail on listing snapshots
200    SnapshotTestingUtils.assertNoSnapshots(admin);
201
202    // put some stuff in the table
203    Table table = UTIL.getConnection().getTable(TABLE_NAME);
204    UTIL.loadTable(table, TEST_FAM, false);
205
206    LOG.debug("FS state before disable:");
207    CommonFSUtils.logFileSystemState(UTIL.getTestFileSystem(),
208      CommonFSUtils.getRootDir(UTIL.getConfiguration()), LOG);
209    // XXX if this is flakey, might want to consider using the async version and looping as
210    // disableTable can succeed and still timeout.
211    admin.disableTable(TABLE_NAME);
212
213    LOG.debug("FS state before snapshot:");
214    CommonFSUtils.logFileSystemState(UTIL.getTestFileSystem(),
215      CommonFSUtils.getRootDir(UTIL.getConfiguration()), LOG);
216
217    // take a snapshot of the disabled table
218    final String SNAPSHOT_NAME = "offlineTableSnapshot";
219    String snapshot = SNAPSHOT_NAME;
220
221    admin.snapshot(new SnapshotDescription(SNAPSHOT_NAME, TABLE_NAME,
222        SnapshotType.DISABLED, null, -1, SnapshotManifestV1.DESCRIPTOR_VERSION, null));
223    LOG.debug("Snapshot completed.");
224
225    // make sure we have the snapshot
226    List<SnapshotDescription> snapshots =
227        SnapshotTestingUtils.assertOneSnapshotThatMatches(admin, snapshot, TABLE_NAME);
228
229    // make sure its a valid snapshot
230    FileSystem fs = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getFileSystem();
231    Path rootDir = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getRootDir();
232    LOG.debug("FS state after snapshot:");
233    CommonFSUtils.logFileSystemState(UTIL.getTestFileSystem(),
234      CommonFSUtils.getRootDir(UTIL.getConfiguration()), LOG);
235    SnapshotTestingUtils.confirmSnapshotValid(
236      ProtobufUtil.createHBaseProtosSnapshotDesc(snapshots.get(0)), TABLE_NAME, TEST_FAM,
237      rootDir, admin, fs);
238
239    admin.deleteSnapshot(snapshot);
240    snapshots = admin.listSnapshots();
241    SnapshotTestingUtils.assertNoSnapshots(admin);
242  }

244  @Test
245  public void testSnapshotFailsOnNonExistantTable() throws Exception {
246    Admin admin = UTIL.getAdmin();
247    // make sure we don't fail on listing snapshots
248    SnapshotTestingUtils.assertNoSnapshots(admin);
249    String tableName = "_not_a_table";
250
251    // make sure the table doesn't exist
252    boolean fail = false;
253    do {
254      try {
255        admin.getDescriptor(TableName.valueOf(tableName));
256        fail = true;
257        LOG.error("Table:" + tableName + " already exists, checking a new name");
258        tableName = tableName + "!";
259      } catch (TableNotFoundException e) {
260        fail = false;
261      }
262    } while (fail);
263
264    // snapshot the non-existant table
265    try {
266      admin.snapshot("fail", TableName.valueOf(tableName));
267      fail("Snapshot succeeded even though there is not table.");
268    } catch (SnapshotCreationException e) {
269      LOG.info("Correctly failed to snapshot a non-existant table:" + e.getMessage());
270    }
271  }

317  @Test
318  public void testListTableSnapshots() throws Exception {
319    Admin admin = null;
320    final TableName tableName = TableName.valueOf(name.getMethodName());
321    try {
322      admin = UTIL.getAdmin();
323
324      HTableDescriptor htd = new HTableDescriptor(tableName);
325      UTIL.createTable(htd, new byte[][] { TEST_FAM }, UTIL.getConfiguration());
326
327      String table1Snapshot1 = "Table1Snapshot1";
328      admin.snapshot(table1Snapshot1, TABLE_NAME);
329      LOG.debug("Snapshot1 completed.");
330
331      String table1Snapshot2 = "Table1Snapshot2";
332      admin.snapshot(table1Snapshot2, TABLE_NAME);
333      LOG.debug("Snapshot2 completed.");
334
335      String table2Snapshot1 = "Table2Snapshot1";
336      admin.snapshot(table2Snapshot1, tableName);
337      LOG.debug(table2Snapshot1 + " completed.");
338
339      List<SnapshotDescription> listTableSnapshots =
340          admin.listTableSnapshots(Pattern.compile("test.*"), MATCH_ALL);
341      List<String> listTableSnapshotNames = new ArrayList<>();
342      assertEquals(3, listTableSnapshots.size());
343      for (SnapshotDescription s : listTableSnapshots) {
344        listTableSnapshotNames.add(s.getName());
345      }
346      assertTrue(listTableSnapshotNames.contains(table1Snapshot1));
347      assertTrue(listTableSnapshotNames.contains(table1Snapshot2));
348      assertTrue(listTableSnapshotNames.contains(table2Snapshot1));
349    } finally {
350      if (admin != null) {
351        try {
352          admin.deleteSnapshots(Pattern.compile("Table.*"));
353        } catch (SnapshotDoesNotExistException ignore) {
354        }
355        if (admin.tableExists(tableName)) {
356          UTIL.deleteTable(tableName);
357        }
358        admin.close();
359      }
360    }
361  }
