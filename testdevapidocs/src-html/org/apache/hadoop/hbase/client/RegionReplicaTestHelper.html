023import java.io.IOException;

034
035final class RegionReplicaTestHelper {
036
037  private RegionReplicaTestHelper() {
038  }
039
040  // waits for all replicas to have region location
041  static void waitUntilAllMetaReplicasHavingRegionLocation(Configuration conf,
042      AsyncRegistry registry, int regionReplication) throws IOException {
043    Waiter.waitFor(conf, conf.getLong("hbase.client.sync.wait.timeout.msec", 60000), 200, true,
044      new ExplainingPredicate<IOException>() {
045        @Override
046        public String explainFailure() throws IOException {
047          return "Not all meta replicas get assigned";
048        }
049
050        @Override
051        public boolean evaluate() throws IOException {
052          try {
053            RegionLocations locs = registry.getMetaRegionLocation().get();
054            if (locs.size() < regionReplication) {
055              return false;
056            }
057            for (int i = 0; i < regionReplication; i++) {
058              if (locs.getRegionLocation(i) == null) {
059                return false;
060              }
061            }
062            return true;
063          } catch (Exception e) {
064            TestZKAsyncRegistry.LOG.warn("Failed to get meta region locations", e);
065            return false;
066          }
067        }
068      });
069  }
070
071  static Optional<ServerName> getRSCarryingReplica(HBaseTestingUtility util, TableName tableName,
072      int replicaId) {
073    return util.getHBaseCluster().getRegionServerThreads().stream().map(t -> t.getRegionServer())
074      .filter(rs -> rs.getRegions(tableName).stream()
075        .anyMatch(r -> r.getRegionInfo().getReplicaId() == replicaId))
076      .findAny().map(rs -> rs.getServerName());
077  }
078
079  /**
080   * Return the new location.
081   */
082  static ServerName moveRegion(HBaseTestingUtility util, HRegionLocation currentLoc)
083      throws Exception {
084    ServerName serverName = currentLoc.getServerName();
085    RegionInfo regionInfo = currentLoc.getRegion();
086    TableName tableName = regionInfo.getTable();
087    int replicaId = regionInfo.getReplicaId();
088    ServerName newServerName = util.getHBaseCluster().getRegionServerThreads().stream()
089      .map(t -> t.getRegionServer().getServerName()).filter(sn -> !sn.equals(serverName)).findAny()
090      .get();
091    util.getAdmin().move(regionInfo.getEncodedNameAsBytes(), newServerName);
092    util.waitFor(30000, new ExplainingPredicate<Exception>() {
093
094      @Override
095      public boolean evaluate() throws Exception {
096        Optional<ServerName> newServerName = getRSCarryingReplica(util, tableName, replicaId);
097        return newServerName.isPresent() && !newServerName.get().equals(serverName);
098      }
099
100      @Override
101      public String explainFailure() throws Exception {
102        return regionInfo.getRegionNameAsString() + " is still on " + serverName;
103      }
104    });
105    return newServerName;
106  }
107
108  interface Locator {
109    RegionLocations getRegionLocations(TableName tableName, int replicaId, boolean reload)
110        throws Exception;
111
112    void updateCachedLocationOnError(HRegionLocation loc, Throwable error) throws Exception;
113  }
114
115  static void testLocator(HBaseTestingUtility util, TableName tableName, Locator locator)
116      throws Exception {
117    RegionLocations locs =
118      locator.getRegionLocations(tableName, RegionReplicaUtil.DEFAULT_REPLICA_ID, false);
119    assertEquals(3, locs.size());
120    for (int i = 0; i < 3; i++) {
121      HRegionLocation loc = locs.getRegionLocation(i);
122      assertNotNull(loc);
123      ServerName serverName = getRSCarryingReplica(util, tableName, i).get();
124      assertEquals(serverName, loc.getServerName());
125    }
126    ServerName newServerName = moveRegion(util, locs.getDefaultRegionLocation());
127    // The cached location should not be changed
128    assertEquals(locs.getDefaultRegionLocation().getServerName(),
129      locator.getRegionLocations(tableName, RegionReplicaUtil.DEFAULT_REPLICA_ID, false)
130        .getDefaultRegionLocation().getServerName());
131    // should get the new location when reload = true
132    assertEquals(newServerName,
133      locator.getRegionLocations(tableName, RegionReplicaUtil.DEFAULT_REPLICA_ID, true)
134        .getDefaultRegionLocation().getServerName());
135    // the cached location should be replaced
136    assertEquals(newServerName,
137      locator.getRegionLocations(tableName, RegionReplicaUtil.DEFAULT_REPLICA_ID, false)
138        .getDefaultRegionLocation().getServerName());
139
140    ServerName newServerName1 = moveRegion(util, locs.getRegionLocation(1));
141    ServerName newServerName2 = moveRegion(util, locs.getRegionLocation(2));
142
143    // The cached location should not be change
144    assertEquals(locs.getRegionLocation(1).getServerName(),
145      locator.getRegionLocations(tableName, 1, false).getRegionLocation(1).getServerName());
146    // clear the cached location for replica 1
147    locator.updateCachedLocationOnError(locs.getRegionLocation(1), new NotServingRegionException());
148    // the cached location for replica 2 should not be changed
149    assertEquals(locs.getRegionLocation(2).getServerName(),
150      locator.getRegionLocations(tableName, 2, false).getRegionLocation(2).getServerName());
151    // should get the new location as we have cleared the old location
152    assertEquals(newServerName1,
153      locator.getRegionLocations(tableName, 1, false).getRegionLocation(1).getServerName());
154    // as we will get the new location for replica 2 at once, we should also get the new location
155    // for replica 2
156    assertEquals(newServerName2,
157      locator.getRegionLocations(tableName, 2, false).getRegionLocation(2).getServerName());
158  }
