052@Category({MediumTests.class, ClientTests.class})
053public class TestClientOperationInterrupt {
054  private static final Log LOG = LogFactory.getLog(TestClientOperationInterrupt.class);
055
056  private static HBaseTestingUtility util;
057  private static final TableName tableName = TableName.valueOf("test");
058  private static final byte[] dummy = Bytes.toBytes("dummy");
059  private static final byte[] row1 = Bytes.toBytes("r1");
060  private static final byte[] test = Bytes.toBytes("test");
061  private static Configuration conf;
062
063  public static class TestCoprocessor implements RegionCoprocessor, RegionObserver {
064    @Override
065    public Optional<RegionObserver> getRegionObserver() {
066      return Optional.of(this);
067    }
068
069    @Override
070    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
071                         final Get get, final List<Cell> results) throws IOException {
072      Threads.sleep(2500);
073    }
074  }

077  @BeforeClass
078  public static void setUpBeforeClass() throws Exception {
079    conf = HBaseConfiguration.create();
080    conf.setStrings(CoprocessorHost.USER_REGION_COPROCESSOR_CONF_KEY,
081        TestCoprocessor.class.getName());
082    util = new HBaseTestingUtility(conf);
083    util.startMiniCluster();
084
085    Admin admin = util.getAdmin();
086    if (admin.tableExists(tableName)) {
087      if (admin.isTableEnabled(tableName)) {
088        admin.disableTable(tableName);
089      }
090      admin.deleteTable(tableName);
091    }
092    Table ht = util.createTable(tableName, new byte[][]{dummy, test});
093
094    Put p = new Put(row1);
095    p.addColumn(dummy, dummy, dummy);
096    ht.put(p);
097  }

100  @Test
101  public void testInterrupt50Percent() throws IOException, InterruptedException {
102    final AtomicInteger noEx = new AtomicInteger(0);
103    final AtomicInteger badEx = new AtomicInteger(0);
104    final AtomicInteger noInt = new AtomicInteger(0);
105    final AtomicInteger done = new AtomicInteger(0);
106    List<Thread> threads = new ArrayList<>();
107
108    final int nbThread = 100;
109
110    for (int i = 0; i < nbThread; i++) {
111      Thread t = new Thread() {
112        @Override
113        public void run() {
114          try {
115            Table ht = util.getConnection().getTable(tableName);
116            Result r = ht.get(new Get(row1));
117            noEx.incrementAndGet();
118          } catch (IOException e) {
119            LOG.info("exception", e);
120            if (!(e instanceof InterruptedIOException) || (e instanceof SocketTimeoutException)) {
121              badEx.incrementAndGet();
122            } else {
123              if (Thread.currentThread().isInterrupted()) {
124                noInt.incrementAndGet();
125                LOG.info("The thread should NOT be with the 'interrupt' status.");
126              }
127            }
128          } finally {
129            done.incrementAndGet();
130          }
131        }
132      };
133      t.setName("TestClientOperationInterrupt #" + i);
134      threads.add(t);
135      t.start();
136    }
137
138    for (int i = 0; i < nbThread / 2; i++) {
139      threads.get(i).interrupt();
140    }
141
142
143    boolean stillAlive = true;
144    while (stillAlive) {
145      stillAlive = false;
146      for (Thread t : threads) {
147        if (t.isAlive()) {
148          stillAlive = true;
149        }
150      }
151      Threads.sleep(10);
152    }
153
154    Assert.assertFalse(Thread.currentThread().isInterrupted());
155
156    Assert.assertTrue(" noEx: " + noEx.get() + ", badEx=" + badEx.get() + ", noInt=" + noInt.get(),
157        noEx.get() == nbThread / 2 && badEx.get() == 0);
158
159    // The problem here is that we need the server to free its handlers to handle all operations
160    while (done.get() != nbThread){
161      Thread.sleep(1);
162    }
163
164    Table ht = util.getConnection().getTable(tableName);
165    Result r = ht.get(new Get(row1));
166    Assert.assertFalse(r.isEmpty());
167  }
