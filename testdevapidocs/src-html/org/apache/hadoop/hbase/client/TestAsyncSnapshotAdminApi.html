056  @Test
057  public void testTakeSnapshot() throws Exception {
058    Admin syncAdmin = TEST_UTIL.getAdmin();
059
060    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
061    for (int i = 0; i < 3000; i++) {
062      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
063        Bytes.toBytes(i)));
064    }
065
066    admin.snapshot(snapshotName1, tableName).get();
067    admin.snapshot(snapshotName2, tableName).get();
068    List<SnapshotDescription> snapshots = syncAdmin.listSnapshots();
069    Collections.sort(snapshots, (snap1, snap2) -> {
070      Assert.assertNotNull(snap1);
071      Assert.assertNotNull(snap1.getName());
072      Assert.assertNotNull(snap2);
073      Assert.assertNotNull(snap2.getName());
074      return snap1.getName().compareTo(snap2.getName());
075    });
076
077    Assert.assertEquals(snapshotName1, snapshots.get(0).getName());
078    Assert.assertEquals(tableName, snapshots.get(0).getTableName());
079    Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(0).getType());
080    Assert.assertEquals(snapshotName2, snapshots.get(1).getName());
081    Assert.assertEquals(tableName, snapshots.get(1).getTableName());
082    Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(1).getType());
083  }

085  @Test
086  public void testCloneSnapshot() throws Exception {
087    TableName tableName2 = TableName.valueOf("testCloneSnapshot2");
088    Admin syncAdmin = TEST_UTIL.getAdmin();
089
090    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
091    for (int i = 0; i < 3000; i++) {
092      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
093        Bytes.toBytes(i)));
094    }
095
096    admin.snapshot(snapshotName1, tableName).get();
097    List<SnapshotDescription> snapshots = syncAdmin.listSnapshots();
098    Assert.assertEquals(snapshots.size(), 1);
099    Assert.assertEquals(snapshotName1, snapshots.get(0).getName());
100    Assert.assertEquals(tableName, snapshots.get(0).getTableName());
101    Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(0).getType());
102
103    // cloneSnapshot into a existed table.
104    boolean failed = false;
105    try {
106      admin.cloneSnapshot(snapshotName1, tableName).get();
107    } catch (Exception e) {
108      failed = true;
109    }
110    Assert.assertTrue(failed);
111
112    // cloneSnapshot into a new table.
113    Assert.assertTrue(!syncAdmin.tableExists(tableName2));
114    admin.cloneSnapshot(snapshotName1, tableName2).get();
115    syncAdmin.tableExists(tableName2);
116  }

118  private void assertResult(TableName tableName, int expectedRowCount) throws IOException {
119    try (Table table = TEST_UTIL.getConnection().getTable(tableName)) {
120      Scan scan = new Scan();
121      try (ResultScanner scanner = table.getScanner(scan)) {
122        Result result;
123        int rowCount = 0;
124        while ((result = scanner.next()) != null) {
125          Assert.assertArrayEquals(result.getRow(), Bytes.toBytes(rowCount));
126          Assert.assertArrayEquals(result.getValue(Bytes.toBytes("f1"), Bytes.toBytes("cq")),
127            Bytes.toBytes(rowCount));
128          rowCount += 1;
129        }
130        Assert.assertEquals(rowCount, expectedRowCount);
131      }
132    }
133  }

136  public void testRestoreSnapshot() throws Exception {
137    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
138    for (int i = 0; i < 3000; i++) {
139      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
140        Bytes.toBytes(i)));
141    }
142    Assert.assertEquals(admin.listSnapshots().get().size(), 0);
143
144    admin.snapshot(snapshotName1, tableName).get();
145    admin.snapshot(snapshotName2, tableName).get();
146    Assert.assertEquals(admin.listSnapshots().get().size(), 2);
147
148    admin.disableTable(tableName).get();
149    admin.restoreSnapshot(snapshotName1, true).get();
150    admin.enableTable(tableName).get();
151    assertResult(tableName, 3000);
152
153    admin.disableTable(tableName).get();
154    admin.restoreSnapshot(snapshotName2, false).get();
155    admin.enableTable(tableName).get();
156    assertResult(tableName, 3000);
157  }

159  @Test
160  public void testListSnapshots() throws Exception {
161    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
162    for (int i = 0; i < 3000; i++) {
163      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
164        Bytes.toBytes(i)));
165    }
166    Assert.assertEquals(admin.listSnapshots().get().size(), 0);
167
168    admin.snapshot(snapshotName1, tableName).get();
169    admin.snapshot(snapshotName2, tableName).get();
170    admin.snapshot(snapshotName3, tableName).get();
171    Assert.assertEquals(admin.listSnapshots().get().size(), 3);
172
173    Assert.assertEquals(admin.listSnapshots(Optional.of(Pattern.compile("(.*)"))).get().size(), 3);
174    Assert.assertEquals(admin.listSnapshots(Optional.of(Pattern.compile("snapshotName(\\d+)")))
175        .get().size(), 3);
176    Assert.assertEquals(admin.listSnapshots(Optional.of(Pattern.compile("snapshotName[1|3]")))
177        .get().size(), 2);
178    Assert.assertEquals(admin.listSnapshots(Optional.of(Pattern.compile("snapshot(.*)"))).get()
179        .size(), 3);
180    Assert.assertEquals(
181      admin.listTableSnapshots(Pattern.compile("testListSnapshots"), Pattern.compile("s(.*)")).get()
182          .size(),
183      3);
184    Assert.assertEquals(
185      admin.listTableSnapshots(Pattern.compile("fakeTableName"), Pattern.compile("snap(.*)")).get()
186          .size(),
187      0);
188    Assert.assertEquals(
189      admin.listTableSnapshots(Pattern.compile("test(.*)"), Pattern.compile("snap(.*)[1|3]")).get()
190          .size(),
191      2);
192  }

196    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
197    for (int i = 0; i < 3000; i++) {
