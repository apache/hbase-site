044@Category({ LargeTests.class, ClientTests.class })
045public class TestHTableMultiplexerFlushCache {
046  private static final Log LOG = LogFactory.getLog(TestHTableMultiplexerFlushCache.class);
047  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
048  private static byte[] FAMILY = Bytes.toBytes("testFamily");
049  private static byte[] QUALIFIER1 = Bytes.toBytes("testQualifier_1");
050  private static byte[] QUALIFIER2 = Bytes.toBytes("testQualifier_2");
051  private static byte[] VALUE1 = Bytes.toBytes("testValue1");
052  private static byte[] VALUE2 = Bytes.toBytes("testValue2");
053  private static int SLAVES = 3;
054  private static int PER_REGIONSERVER_QUEUE_SIZE = 100000;
055
056  /**
057   * @throws java.lang.Exception
058   */
059  @BeforeClass
060  public static void setUpBeforeClass() throws Exception {
061    TEST_UTIL.startMiniCluster(SLAVES);
062  }
063
064  /**
065   * @throws java.lang.Exception
066   */
067  @AfterClass
068  public static void tearDownAfterClass() throws Exception {
069    TEST_UTIL.shutdownMiniCluster();
070  }
071
072  private static void checkExistence(final Table htable, final byte[] row, final byte[] family,
073      final byte[] quality,
074      final byte[] value) throws Exception {
075    // verify that the Get returns the correct result
076    TEST_UTIL.waitFor(30000, new Waiter.Predicate<Exception>() {
077      @Override
078      public boolean evaluate() throws Exception {
079        Result r;
080        Get get = new Get(row);
081        get.addColumn(family, quality);
082        r = htable.get(get);
083        return r != null && r.getValue(family, quality) != null
084            && Bytes.toStringBinary(value).equals(
085            Bytes.toStringBinary(r.getValue(family, quality)));
086      }
087    });
088  }
089
090  @Test
091  public void testOnRegionChange() throws Exception {
092    TableName TABLE = TableName.valueOf("testOnRegionChange");
093    final int NUM_REGIONS = 10;
094    Table htable = TEST_UTIL.createTable(TABLE, new byte[][] { FAMILY }, 3,
095      Bytes.toBytes("aaaaa"), Bytes.toBytes("zzzzz"), NUM_REGIONS);
096
097    HTableMultiplexer multiplexer = new HTableMultiplexer(TEST_UTIL.getConfiguration(), 
098      PER_REGIONSERVER_QUEUE_SIZE);
099    
100    try (RegionLocator r = TEST_UTIL.getConnection().getRegionLocator(TABLE)) {
101      byte[][] startRows = r.getStartKeys();
102      byte[] row = startRows[1];
103      assertTrue("2nd region should not start with empty row", row != null && row.length > 0);
104
105      Put put = new Put(row).addColumn(FAMILY, QUALIFIER1, VALUE1);
106      assertTrue("multiplexer.put returns", multiplexer.put(TABLE, put));
107
108      checkExistence(htable, row, FAMILY, QUALIFIER1, VALUE1);
109
110      // Now let's shutdown the regionserver and let regions moved to other servers.
111      HRegionLocation loc = r.getRegionLocation(row);
112      MiniHBaseCluster hbaseCluster = TEST_UTIL.getHBaseCluster();
113      hbaseCluster.stopRegionServer(loc.getServerName());
114      TEST_UTIL.waitUntilAllRegionsAssigned(TABLE);
115
116      // put with multiplexer.
117      put = new Put(row).addColumn(FAMILY, QUALIFIER2, VALUE2);
118      assertTrue("multiplexer.put returns", multiplexer.put(TABLE, put));
119
120      checkExistence(htable, row, FAMILY, QUALIFIER2, VALUE2);
121    }
122  }

124  @Test
125  public void testOnRegionMove() throws Exception {
126    // This test is doing near exactly the same thing that testOnRegionChange but avoiding the
127    // potential to get a ConnectionClosingException. By moving the region, we can be certain that
128    // the connection is still valid and that the implementation is correctly handling an invalid
129    // Region cache (and not just tearing down the entire connection).
130    TableName TABLE = TableName.valueOf("testOnRegionMove");
131    final int NUM_REGIONS = 10;
132    HTable htable = TEST_UTIL.createTable(TABLE, new byte[][] { FAMILY }, 3,
133      Bytes.toBytes("aaaaa"), Bytes.toBytes("zzzzz"), NUM_REGIONS);
134
135    HTableMultiplexer multiplexer = new HTableMultiplexer(TEST_UTIL.getConfiguration(),
136      PER_REGIONSERVER_QUEUE_SIZE);
137
138    final RegionLocator regionLocator = TEST_UTIL.getConnection().getRegionLocator(TABLE);
139    Pair<byte[][],byte[][]> startEndRows = regionLocator.getStartEndKeys();
140    byte[] row = startEndRows.getFirst()[1];
141    assertTrue("2nd region should not start with empty row", row != null && row.length > 0);
142
143    Put put = new Put(row).addColumn(FAMILY, QUALIFIER1, VALUE1);
144    assertTrue("multiplexer.put returns", multiplexer.put(TABLE, put));
145
146    checkExistence(htable, row, FAMILY, QUALIFIER1, VALUE1);
147
148    final HRegionLocation loc = regionLocator.getRegionLocation(row);
149    final MiniHBaseCluster hbaseCluster = TEST_UTIL.getHBaseCluster();
150    // The current server for the region we're writing to
151    final ServerName originalServer = loc.getServerName();
152    ServerName newServer = null;
153    // Find a new server to move that region to
154    for (int i = 0; i < SLAVES; i++) {
155      HRegionServer rs = hbaseCluster.getRegionServer(0);
156      if (!rs.getServerName().equals(originalServer.getServerName())) {
157        newServer = rs.getServerName();
158        break;
159      }
160    }
161    assertNotNull("Did not find a new RegionServer to use", newServer);
162
163    // Move the region
164    LOG.info("Moving " + loc.getRegionInfo().getEncodedName() + " from " + originalServer
165        +  " to " + newServer);
166    TEST_UTIL.getHBaseAdmin().move(loc.getRegionInfo().getEncodedNameAsBytes(),
167        Bytes.toBytes(newServer.getServerName()));
168
169    TEST_UTIL.waitUntilAllRegionsAssigned(TABLE);
170
171    // Send a new Put
172    put = new Put(row).addColumn(FAMILY, QUALIFIER2, VALUE2);
173    assertTrue("multiplexer.put returns", multiplexer.put(TABLE, put));
174
175    // We should see the update make it to the new server eventually
176    checkExistence(htable, row, FAMILY, QUALIFIER2, VALUE2);
177  }
