532   * @return Scan with RowFilter that does CompareOp op on passed key.

534  private Scan createScanWithRowFilter(final byte [] key,
535      final byte [] startRow, CompareOperator op) {
536    // Make sure key is of some substance... non-null and > than first key.
537    assertTrue(key != null && key.length > 0 &&
538      Bytes.BYTES_COMPARATOR.compare(key, new byte [] {'a', 'a', 'a'}) >= 0);
539    LOG.info("Key=" + Bytes.toString(key));
540    Scan s = startRow == null? new Scan(): new Scan(startRow);
541    Filter f = new RowFilter(op, new BinaryComparator(key));
542    f = new WhileMatchFilter(f);
543    s.setFilter(f);
544    return s;
545  }

576      List<HRegionLocation> regions = locator.getAllRegionLocations();
577      int originalCount = regions.size();
578      for (int i = 0; i < TEST_UTIL.getConfiguration().getInt("hbase.test.retries", 30); i++) {

583          e.printStackTrace();
584        }

586        if (regions.size() > originalCount)
587          break;
588      }
