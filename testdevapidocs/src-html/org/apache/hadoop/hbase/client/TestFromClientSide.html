204  public void testKeepDeletedCells() throws Exception {
205    final TableName tableName = name.getTableName();
206    final byte[] FAMILY = Bytes.toBytes("family");
207    final byte[] C0 = Bytes.toBytes("c0");
208
209    final byte[] T1 = Bytes.toBytes("T1");
210    final byte[] T2 = Bytes.toBytes("T2");
211    final byte[] T3 = Bytes.toBytes("T3");
212
213    TableDescriptor tableDescriptor = TableDescriptorBuilder.newBuilder(tableName)
214      .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(FAMILY)
215        .setKeepDeletedCells(KeepDeletedCells.TRUE).setMaxVersions(3).build())
216      .build();
217    TEST_UTIL.getAdmin().createTable(tableDescriptor);
218    try (Table h = TEST_UTIL.getConnection().getTable(tableName)) {
219      long ts = System.currentTimeMillis();
220      Put p = new Put(T1, ts);
221      p.addColumn(FAMILY, C0, T1);
222      h.put(p);
223      p = new Put(T1, ts + 2);
224      p.addColumn(FAMILY, C0, T2);
225      h.put(p);
226      p = new Put(T1, ts + 4);
227      p.addColumn(FAMILY, C0, T3);
228      h.put(p);
229
230      Delete d = new Delete(T1, ts + 3);
231      h.delete(d);
232
233      d = new Delete(T1, ts + 3);
234      d.addColumns(FAMILY, C0, ts + 3);
235      h.delete(d);
236
237      Get g = new Get(T1);
238      // does *not* include the delete
239      g.setTimeRange(0, ts + 3);
240      Result r = h.get(g);
241      assertArrayEquals(T2, r.getValue(FAMILY, C0));
242
243      Scan s = new Scan().withStartRow(T1);
244      s.setTimeRange(0, ts + 3);
245      s.readAllVersions();
246      ResultScanner scanner = h.getScanner(s);
247      Cell[] kvs = scanner.next().rawCells();
248      assertArrayEquals(T2, CellUtil.cloneValue(kvs[0]));
249      assertArrayEquals(T1, CellUtil.cloneValue(kvs[1]));
250      scanner.close();
251
252      s = new Scan().withStartRow(T1);
253      s.setRaw(true);
254      s.readAllVersions();
255      scanner = h.getScanner(s);
256      kvs = scanner.next().rawCells();
257      assertTrue(PrivateCellUtil.isDeleteFamily(kvs[0]));
258      assertArrayEquals(T3, CellUtil.cloneValue(kvs[1]));
259      assertTrue(CellUtil.isDelete(kvs[2]));
260      assertArrayEquals(T2, CellUtil.cloneValue(kvs[3]));
261      assertArrayEquals(T1, CellUtil.cloneValue(kvs[4]));
262      scanner.close();
263    }

275
276    try (Table table = TEST_UTIL.createTable(tableName, FAMILY)) {
277      // future timestamp
278      long ts = System.currentTimeMillis() * 2;
279      Put put = new Put(ROW, ts);
280      put.addColumn(FAMILY, COLUMN, VALUE);
281      table.put(put);
282
283      Get get = new Get(ROW);
284      Result result = table.get(get);
285      assertArrayEquals(VALUE, result.getValue(FAMILY, COLUMN));
286
287      Delete del = new Delete(ROW);

289      table.delete(del);
290
291      get = new Get(ROW);
292      result = table.get(get);
293      assertNull(result.getValue(FAMILY, COLUMN));
294
295      // major compaction, purged future deletes
296      TEST_UTIL.getAdmin().flush(tableName);
297      TEST_UTIL.getAdmin().majorCompact(tableName);
298
299      // waiting for the major compaction to complete
300      TEST_UTIL.waitFor(6000,
301        () -> TEST_UTIL.getAdmin().getCompactionState(tableName) == CompactionState.NONE);
302
303      put = new Put(ROW, ts);
304      put.addColumn(FAMILY, COLUMN, VALUE);
305      table.put(put);
306
307      get = new Get(ROW);
