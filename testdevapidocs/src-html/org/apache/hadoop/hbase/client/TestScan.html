053  @Test
054  public void testAttributesSerialization() throws IOException {
055    Scan scan = new Scan();
056    scan.setAttribute("attribute1", Bytes.toBytes("value1"));
057    scan.setAttribute("attribute2", Bytes.toBytes("value2"));
058    scan.setAttribute("attribute3", Bytes.toBytes("value3"));
059
060    ClientProtos.Scan scanProto = ProtobufUtil.toScan(scan);
061
062    Scan scan2 = ProtobufUtil.toScan(scanProto);
063
064    Assert.assertNull(scan2.getAttribute("absent"));
065    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value1"), scan2.getAttribute("attribute1")));
066    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value2"), scan2.getAttribute("attribute2")));
067    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value3"), scan2.getAttribute("attribute3")));
068    Assert.assertEquals(3, scan2.getAttributesMap().size());
069  }

112    assertEquals(get.getPriority(), scan.getPriority());

115  @Test
116  public void testScanAttributes() {
117    Scan scan = new Scan();
118    Assert.assertTrue(scan.getAttributesMap().isEmpty());
119    Assert.assertNull(scan.getAttribute("absent"));
120
121    scan.setAttribute("absent", null);
122    Assert.assertTrue(scan.getAttributesMap().isEmpty());
123    Assert.assertNull(scan.getAttribute("absent"));
124
125    // adding attribute
126    scan.setAttribute("attribute1", Bytes.toBytes("value1"));
127    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value1"), scan.getAttribute("attribute1")));
128    Assert.assertEquals(1, scan.getAttributesMap().size());
129    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value1"),
130        scan.getAttributesMap().get("attribute1")));
131
132    // overriding attribute value
133    scan.setAttribute("attribute1", Bytes.toBytes("value12"));
134    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value12"), scan.getAttribute("attribute1")));
135    Assert.assertEquals(1, scan.getAttributesMap().size());
136    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value12"),
137        scan.getAttributesMap().get("attribute1")));
138
139    // adding another attribute
140    scan.setAttribute("attribute2", Bytes.toBytes("value2"));
141    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value2"), scan.getAttribute("attribute2")));
142    Assert.assertEquals(2, scan.getAttributesMap().size());
143    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value2"),
144        scan.getAttributesMap().get("attribute2")));
145
146    // removing attribute
147    scan.setAttribute("attribute2", null);
148    Assert.assertNull(scan.getAttribute("attribute2"));
149    Assert.assertEquals(1, scan.getAttributesMap().size());
150    Assert.assertNull(scan.getAttributesMap().get("attribute2"));
151
152    // removing non-existed attribute
153    scan.setAttribute("attribute2", null);
154    Assert.assertNull(scan.getAttribute("attribute2"));
155    Assert.assertEquals(1, scan.getAttributesMap().size());
156    Assert.assertNull(scan.getAttributesMap().get("attribute2"));
157
158    // removing another attribute
159    scan.setAttribute("attribute1", null);
160    Assert.assertNull(scan.getAttribute("attribute1"));
161    Assert.assertTrue(scan.getAttributesMap().isEmpty());
162    Assert.assertNull(scan.getAttributesMap().get("attribute1"));
163  }

221  @Test
222  public void testScanCopyConstructor() throws Exception {
223    Scan scan = new Scan();
224
225    scan.addColumn(Bytes.toBytes("cf"), Bytes.toBytes("q"))
226        .setACL("test_user", new Permission(Permission.Action.READ))
227        .setAllowPartialResults(true)
228        .setAsyncPrefetch(false)
229        .setAttribute("test_key", Bytes.toBytes("test_value"))
230        .setAuthorizations(new Authorizations("test_label"))
231        .setBatch(10)
232        .setCacheBlocks(false)
233        .setCaching(10)
234        .setConsistency(Consistency.TIMELINE)
235        .setFilter(new FilterList())
236        .setId("scan_copy_constructor")
237        .setIsolationLevel(IsolationLevel.READ_COMMITTED)
238        .setLimit(100)
239        .setLoadColumnFamiliesOnDemand(false)
240        .setMaxResultSize(100)
241        .setMaxResultsPerColumnFamily(1000)
242        .readVersions(9999)
243        .setMvccReadPoint(5)
244        .setNeedCursorResult(true)
245        .setPriority(1)
246        .setRaw(true)
247        .setReplicaId(3)
248        .setReversed(true)
249        .setRowOffsetPerColumnFamily(5)
250        .setRowPrefixFilter(Bytes.toBytes("row_"))
251        .setScanMetricsEnabled(true)
252        .setSmall(true)
253        .setReadType(ReadType.STREAM)
254        .withStartRow(Bytes.toBytes("row_1"))
255        .withStopRow(Bytes.toBytes("row_2"))
256        .setTimeRange(0, 13);
257
258    // create a copy of existing scan object
259    Scan scanCopy = new Scan(scan);
260
261    // validate fields of copied scan object match with the original scan object
262    assertEquals(scan.getACL(), scanCopy.getACL());
263    assertEquals(scan.getAllowPartialResults(), scanCopy.getAllowPartialResults());
264    assertEquals(scan.getAttribute("test_key"), scanCopy.getAttribute("test_key"));
265    assertEquals(scan.getAttributeSize(), scanCopy.getAttributeSize());
266    assertEquals(scan.getAttributesMap(), scanCopy.getAttributesMap());
267    assertEquals(scan.getAuthorizations().getLabels(), scanCopy.getAuthorizations().getLabels());

269    assertEquals(scan.getCacheBlocks(), scanCopy.getCacheBlocks());
270    assertEquals(scan.getCaching(), scanCopy.getCaching());
271    assertEquals(scan.getConsistency(), scanCopy.getConsistency());
272    assertEquals(scan.getFamilies().length, scanCopy.getFamilies().length);
273    assertEquals(scan.getFamilies()[0], scanCopy.getFamilies()[0]);

282    assertEquals(scan.getMaxResultsPerColumnFamily(), scanCopy.getMaxResultsPerColumnFamily());
