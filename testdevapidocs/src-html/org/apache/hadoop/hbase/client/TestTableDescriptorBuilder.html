106  public void testPb() throws DeserializationException, IOException {
107    final int v = 123;
108    TableDescriptor htd
109      = TableDescriptorBuilder.newBuilder(TableName.META_TABLE_NAME)
110          .setMaxFileSize(v)

145   * Test cps in the table description
146   * @throws Exception
147   */
148  @Test
149  public void testSetListRemoveCP() throws Exception {
150    TableDescriptor desc
151      = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName())).build();
152    // Check that any coprocessor is present.
153    assertTrue(desc.getCoprocessors().isEmpty());
154
155    // simple CP
156    String className1 = "org.apache.hadoop.hbase.coprocessor.SimpleRegionObserver";
157    String className2 = "org.apache.hadoop.hbase.coprocessor.SampleRegionWALObserver";
158    desc = TableDescriptorBuilder.newBuilder(desc)
159            .addCoprocessor(className1) // Add the 1 coprocessor and check if present.
160            .build();
161    assertTrue(desc.getCoprocessors().size() == 1);
162    assertTrue(desc.getCoprocessors().contains(className1));
163
164    desc = TableDescriptorBuilder.newBuilder(desc)
165            // Add the 2nd coprocessor and check if present.
166            // remove it and check that it is gone
167            .addCoprocessor(className2)
168            .build();
169    assertTrue(desc.getCoprocessors().size() == 2);
170    assertTrue(desc.getCoprocessors().contains(className2));
171
172    desc = TableDescriptorBuilder.newBuilder(desc)
173            // Remove one and check
174            .removeCoprocessor(className1)
175            .build();
176    assertTrue(desc.getCoprocessors().size() == 1);
177    assertFalse(desc.getCoprocessors().contains(className1));
178    assertTrue(desc.getCoprocessors().contains(className2));
179
180    desc = TableDescriptorBuilder.newBuilder(desc)
181            // Remove the last and check
182            .removeCoprocessor(className2)
183            .build();
184    assertTrue(desc.getCoprocessors().isEmpty());
185    assertFalse(desc.getCoprocessors().contains(className1));
186    assertFalse(desc.getCoprocessors().contains(className2));
187  }

195    byte[] key = Bytes.toBytes("Some");
196    byte[] value = Bytes.toBytes("value");
197    TableDescriptor desc

199            .setValue(key, value)
200            .build();
201    assertTrue(Bytes.equals(value, desc.getValue(key)));

204            .build();
205    assertTrue(desc.getValue(key) == null);

285  @Test
286  public void testModifyFamily() {
287    byte[] familyName = Bytes.toBytes("cf");
288    ColumnFamilyDescriptor hcd = ColumnFamilyDescriptorBuilder.newBuilder(familyName)
289            .setBlocksize(1000)
290            .setDFSReplication((short) 3)
291            .build();
292    TableDescriptor htd
293      = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
294              .addColumnFamily(hcd)
295              .build();
296
297    assertEquals(1000, htd.getColumnFamily(familyName).getBlocksize());
298    assertEquals(3, htd.getColumnFamily(familyName).getDFSReplication());
299    hcd = ColumnFamilyDescriptorBuilder.newBuilder(familyName)
300            .setBlocksize(2000)
301            .setDFSReplication((short) 1)
302            .build();
303    htd = TableDescriptorBuilder.newBuilder(htd)
304              .modifyColumnFamily(hcd)
305              .build();
306    assertEquals(2000, htd.getColumnFamily(familyName).getBlocksize());
307    assertEquals(1, htd.getColumnFamily(familyName).getDFSReplication());
308  }

345  @Test
346  public void testSerialReplicationScope() {
347    HColumnDescriptor hcdWithScope = new HColumnDescriptor(Bytes.toBytes("cf0"));
348    hcdWithScope.setScope(HConstants.REPLICATION_SCOPE_SERIAL);
349    HColumnDescriptor hcdWithoutScope = new HColumnDescriptor(Bytes.toBytes("cf1"));
350    TableDescriptor htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
351            .addColumnFamily(hcdWithoutScope)
352            .build();
353    assertFalse(htd.hasSerialReplicationScope());
354
355    htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
356            .addColumnFamily(hcdWithScope)
357            .build();
358    assertTrue(htd.hasSerialReplicationScope());
359
360    htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
361            .addColumnFamily(hcdWithScope)
362            .addColumnFamily(hcdWithoutScope)
363            .build();
364    assertTrue(htd.hasSerialReplicationScope());
365  }
