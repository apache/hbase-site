061@Category({ MasterTests.class, MediumTests.class })
062public class TestEnableTable {

082  @Test(timeout = 300000)
083  public void testEnableTableWithNoRegionServers() throws Exception {
084    final TableName tableName = TableName.valueOf(name.getMethodName());
085    final MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
086    final HMaster m = cluster.getMaster();
087    final Admin admin = TEST_UTIL.getAdmin();
088    final HTableDescriptor desc = new HTableDescriptor(tableName);
089    desc.addFamily(new HColumnDescriptor(FAMILYNAME));
090    admin.createTable(desc);
091    admin.disableTable(tableName);
092    TEST_UTIL.waitTableDisabled(tableName.getName());
093
094    admin.enableTable(tableName);
095    TEST_UTIL.waitTableEnabled(tableName);
096    // disable once more
097    admin.disableTable(tableName);
098
099    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
100    // now stop region servers
101    JVMClusterUtil.RegionServerThread rs = cluster.getRegionServerThreads().get(0);
102    rs.getRegionServer().stop("stop");
103    cluster.waitForRegionServerToStop(rs.getRegionServer().getServerName(), 10000);
104
105    // We used to enable the table here but AMv2 would hang waiting on a RS to check-in.
106    // Revisit.
107
108    JVMClusterUtil.RegionServerThread rs2 = cluster.startRegionServer();
109    cluster.waitForRegionServerToStart(rs2.getRegionServer().getServerName().getHostname(),
110        rs2.getRegionServer().getServerName().getPort(), 60000);
111
112    LOG.debug("Now enabling table " + tableName);
113    admin.enableTable(tableName);
114    assertTrue(admin.isTableEnabled(tableName));
115
116    List<HRegionInfo> regions = TEST_UTIL.getAdmin().getTableRegions(tableName);
117    assertEquals(1, regions.size());
118    for (HRegionInfo region : regions) {
119      TEST_UTIL.getAdmin().assign(region.getEncodedNameAsBytes());
120    }
121    LOG.debug("Waiting for table assigned " + tableName);
122    TEST_UTIL.waitUntilAllRegionsAssigned(tableName);
123    List<HRegionInfo> onlineRegions = admin.getOnlineRegions(
124        rs2.getRegionServer().getServerName());
125    ArrayList<HRegionInfo> tableRegions = filterTableRegions(tableName, onlineRegions);
126    assertEquals(1, tableRegions.size());
127  }
128
129  private ArrayList<HRegionInfo> filterTableRegions(final TableName tableName,
130      List<HRegionInfo> onlineRegions) {
131    return Lists.newArrayList(Iterables.filter(onlineRegions, new Predicate<HRegionInfo>() {
132      @Override
133      public boolean apply(HRegionInfo input) {
134        return input.getTable().equals(tableName);
135      }
136    }));
137  }

139  /**
140   * We were only clearing rows that had a hregioninfo column in hbase:meta.  Mangled rows that
141   * were missing the hregioninfo because of error were being left behind messing up any
142   * subsequent table made with the same name. HBASE-12980
143   * @throws IOException
144   * @throws InterruptedException
145   */
146  @Test(timeout=60000)
147  public void testDeleteForSureClearsAllTableRowsFromMeta()
148  throws IOException, InterruptedException {
149    final TableName tableName = TableName.valueOf(name.getMethodName());
150    final Admin admin = TEST_UTIL.getAdmin();
151    final HTableDescriptor desc = new HTableDescriptor(tableName);
152    desc.addFamily(new HColumnDescriptor(FAMILYNAME));
153    try {
154      createTable(TEST_UTIL, desc, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
155    } catch (Exception e) {
156      e.printStackTrace();
157      fail("Got an exception while creating " + tableName);
158    }
159    // Now I have a nice table, mangle it by removing the HConstants.REGIONINFO_QUALIFIER_STR
160    // content from a few of the rows.
161    try (Table metaTable = TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME)) {
162      try (ResultScanner scanner = metaTable.getScanner(
163        MetaTableAccessor.getScanForTableName(TEST_UTIL.getConnection(), tableName))) {
164        for (Result result : scanner) {
165          // Just delete one row.
166          Delete d = new Delete(result.getRow());
167          d.addColumn(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
168          LOG.info("Mangled: " + d);
169          metaTable.delete(d);
170          break;
171        }
172      }
173      admin.disableTable(tableName);
174      TEST_UTIL.waitTableDisabled(tableName.getName());
175      // Rely on the coprocessor based latch to make the operation synchronous.
176      try {
177        deleteTable(TEST_UTIL, tableName);
178      } catch (Exception e) {
179        e.printStackTrace();
180        fail("Got an exception while deleting " + tableName);
181      }
182      int rowCount = 0;
183      try (ResultScanner scanner =
184          metaTable.getScanner(MetaTableAccessor.getScanForTableName(TEST_UTIL.getConnection(), tableName))) {
185        for (Result result : scanner) {
186          LOG.info("Found when none expected: " + result);
187          rowCount++;
188        }
189      }
190      assertEquals(0, rowCount);
191    }
192  }

194  public  static class MasterSyncObserver implements MasterCoprocessor, MasterObserver {
195    volatile CountDownLatch tableCreationLatch = null;
196    volatile CountDownLatch tableDeletionLatch = null;
197
198    @Override
199    public Optional<MasterObserver> getMasterObserver() {
200      return Optional.of(this);
201    }
202
203    @Override
204    public void postCompletedCreateTableAction(
205        final ObserverContext<MasterCoprocessorEnvironment> ctx,
206        final TableDescriptor desc,
207        final RegionInfo[] regions) throws IOException {
208      // the AccessController test, some times calls only and directly the
209      // postCompletedCreateTableAction()
210      if (tableCreationLatch != null) {
211        tableCreationLatch.countDown();
212      }
213    }
214
215    @Override
216    public void postCompletedDeleteTableAction(
217        final ObserverContext<MasterCoprocessorEnvironment> ctx,
218        final TableName tableName)
219    throws IOException {
220      // the AccessController test, some times calls only and directly the postDeleteTableHandler()
221      if (tableDeletionLatch != null) {
222        tableDeletionLatch.countDown();
223      }
224    }
225  }

227  public static void createTable(HBaseTestingUtility testUtil,
228    HTableDescriptor htd, byte [][] splitKeys)
229  throws Exception {
230    // NOTE: We need a latch because admin is not sync,
231    // so the postOp coprocessor method may be called after the admin operation returned.
232    MasterSyncObserver observer = testUtil.getHBaseCluster().getMaster()
233      .getMasterCoprocessorHost().findCoprocessor(MasterSyncObserver.class);
234    observer.tableCreationLatch = new CountDownLatch(1);
235    Admin admin = testUtil.getAdmin();
236    if (splitKeys != null) {
237      admin.createTable(htd, splitKeys);
238    } else {
239      admin.createTable(htd);
240    }
241    observer.tableCreationLatch.await();
242    observer.tableCreationLatch = null;
243    testUtil.waitUntilAllRegionsAssigned(htd.getTableName());
244  }
245
246  public static void deleteTable(HBaseTestingUtility testUtil, TableName tableName)
247  throws Exception {
248    // NOTE: We need a latch because admin is not sync,
249    // so the postOp coprocessor method may be called after the admin operation returned.
250    MasterSyncObserver observer = testUtil.getHBaseCluster().getMaster()
251      .getMasterCoprocessorHost().findCoprocessor(MasterSyncObserver.class);
252    observer.tableDeletionLatch = new CountDownLatch(1);
253    Admin admin = testUtil.getAdmin();
254    try {
255      admin.disableTable(tableName);
256    } catch (Exception e) {
257      LOG.debug("Table: " + tableName + " already disabled, so just deleting it.");
258    }
259    admin.deleteTable(tableName);
260    observer.tableDeletionLatch.await();
261    observer.tableDeletionLatch = null;
262  }
