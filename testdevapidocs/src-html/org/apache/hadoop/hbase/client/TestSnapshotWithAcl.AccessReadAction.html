048@Category({ MediumTests.class, ClientTests.class })
049public class TestSnapshotWithAcl extends SecureTestUtil {
050
051  public TableName TEST_TABLE = TableName.valueOf(UUID.randomUUID().toString());
052
053  private static final int ROW_COUNT = 30000;
054
055  private static byte[] TEST_FAMILY = Bytes.toBytes("f1");
056  private static byte[] TEST_QUALIFIER = Bytes.toBytes("cq");
057  private static byte[] TEST_ROW = Bytes.toBytes(0);
058  private static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
059  private static Configuration conf;
060  private static HBaseAdmin admin = null;
061
062  // user is table owner. will have all permissions on table
063  private static User USER_OWNER;
064  // user with rw permissions on column family.
065  private static User USER_RW;
066  // user with read-only permissions
067  private static User USER_RO;
068  // user with none permissions
069  private static User USER_NONE;
070
071  static class AccessReadAction implements AccessTestAction {
072
073    private TableName tableName;
074
075    public AccessReadAction(TableName tableName) {
076      this.tableName = tableName;
077    }
078
079    @Override
080    public Object run() throws Exception {
081      Get g = new Get(TEST_ROW);
082      g.addFamily(TEST_FAMILY);
083      try (Connection conn = ConnectionFactory.createConnection(conf)) {
084        try (Table t = conn.getTable(tableName)) {
085          t.get(g);
086        }
087      }
088      return null;
089    }
090  }
091
092  static class AccessWriteAction implements AccessTestAction {
093    private TableName tableName;
094
095    public AccessWriteAction(TableName tableName) {
096      this.tableName = tableName;
097    }
098
099    @Override
100    public Object run() throws Exception {
101      Put p = new Put(TEST_ROW);
102      p.addColumn(TEST_FAMILY, TEST_QUALIFIER, Bytes.toBytes(0));
103      try (Connection conn = ConnectionFactory.createConnection(conf)) {
104        try (Table t = conn.getTable(tableName)) {
105          t.put(p);
106        }
107      }
108      return null;
109    }
110  }
111
112
113  @BeforeClass
114  public static void setupBeforeClass() throws Exception {
115    conf = TEST_UTIL.getConfiguration();
116    // Enable security
117    enableSecurity(conf);
118    conf.set(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY, AccessController.class.getName());
119    // Verify enableSecurity sets up what we require
120    verifyConfiguration(conf);
121    // Enable EXEC permission checking
122    conf.setBoolean(AccessControlConstants.EXEC_PERMISSION_CHECKS_KEY, true);
123    TEST_UTIL.startMiniCluster();
124    TEST_UTIL.waitUntilAllRegionsAssigned(AccessControlLists.ACL_TABLE_NAME);
125    MasterCoprocessorHost cpHost =
126        TEST_UTIL.getMiniHBaseCluster().getMaster().getMasterCoprocessorHost();
127    cpHost.load(AccessController.class, Coprocessor.PRIORITY_HIGHEST, conf);
128
129    USER_OWNER = User.createUserForTesting(conf, "owner", new String[0]);
130    USER_RW = User.createUserForTesting(conf, "rwuser", new String[0]);
131    USER_RO = User.createUserForTesting(conf, "rouser", new String[0]);
132    USER_NONE = User.createUserForTesting(conf, "usernone", new String[0]);
133  }
134
135  @Before
136  public void setUp() throws Exception {
137    admin = TEST_UTIL.getHBaseAdmin();
138    HTableDescriptor htd = new HTableDescriptor(TEST_TABLE);
139    HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAMILY);
140    hcd.setMaxVersions(100);
141    htd.addFamily(hcd);
142    htd.setOwner(USER_OWNER);
143    admin.createTable(htd, new byte[][] { Bytes.toBytes("s") });
144    TEST_UTIL.waitTableEnabled(TEST_TABLE);
145
146    grantOnTable(TEST_UTIL, USER_RW.getShortName(), TEST_TABLE, TEST_FAMILY, null,
147            Permission.Action.READ, Permission.Action.WRITE);
148
149    grantOnTable(TEST_UTIL, USER_RO.getShortName(), TEST_TABLE, TEST_FAMILY, null,
150            Permission.Action.READ);
151  }
152
153  private void loadData() throws IOException {
154    try (Connection conn = ConnectionFactory.createConnection(conf)) {
155      try (Table t = conn.getTable(TEST_TABLE)) {
156        for (int i = 0; i < ROW_COUNT; i++) {
157          Put put = new Put(Bytes.toBytes(i));
158          put.addColumn(TEST_FAMILY, TEST_QUALIFIER, Bytes.toBytes(i));
159          t.put(put);
160        }
161      }
162    }
163  }
164
165  @AfterClass
166  public static void tearDownAfterClass() throws Exception {
167    TEST_UTIL.shutdownMiniCluster();
168  }
169
170  private void verifyRows(TableName tableName) throws IOException {
171    try (Connection conn = ConnectionFactory.createConnection(conf)) {
172      try (Table t = conn.getTable(tableName)) {
173        try (ResultScanner scanner = t.getScanner(new Scan())) {
174          Result result;
175          int rowCount = 0;
176          while ((result = scanner.next()) != null) {
177            byte[] value = result.getValue(TEST_FAMILY, TEST_QUALIFIER);
178            Assert.assertArrayEquals(value, Bytes.toBytes(rowCount++));
179          }
180          Assert.assertEquals(rowCount, ROW_COUNT);
181        }
182      }
183    }
184  }
185
186  @Test
187  public void testRestoreSnapshot() throws Exception {
188    verifyAllowed(new AccessReadAction(TEST_TABLE), USER_OWNER, USER_RO, USER_RW);
189    verifyDenied(new AccessReadAction(TEST_TABLE), USER_NONE);
190    verifyAllowed(new AccessWriteAction(TEST_TABLE), USER_OWNER, USER_RW);
191    verifyDenied(new AccessWriteAction(TEST_TABLE), USER_RO, USER_NONE);
192
193    loadData();
194    verifyRows(TEST_TABLE);
195
196    String snapshotName1 = UUID.randomUUID().toString();
197    admin.snapshot(snapshotName1, TEST_TABLE);
198
199    // clone snapshot with restoreAcl true.
200    TableName tableName1 = TableName.valueOf(UUID.randomUUID().toString());
201    admin.cloneSnapshot(snapshotName1, tableName1, true);
202    verifyRows(tableName1);
203    verifyAllowed(new AccessReadAction(tableName1), USER_OWNER, USER_RO, USER_RW);
204    verifyDenied(new AccessReadAction(tableName1), USER_NONE);
205    verifyAllowed(new AccessWriteAction(tableName1), USER_OWNER, USER_RW);
206    verifyDenied(new AccessWriteAction(tableName1), USER_RO, USER_NONE);
207
208    // clone snapshot with restoreAcl false.
209    TableName tableName2 = TableName.valueOf(UUID.randomUUID().toString());
210    admin.cloneSnapshot(snapshotName1, tableName2, false);
211    verifyRows(tableName2);
212    verifyAllowed(new AccessReadAction(tableName2), USER_OWNER);
213    verifyDenied(new AccessReadAction(tableName2), USER_NONE, USER_RO, USER_RW);
214    verifyAllowed(new AccessWriteAction(tableName2), USER_OWNER);
215    verifyDenied(new AccessWriteAction(tableName2), USER_RO, USER_RW, USER_NONE);
216
217    // remove read permission for USER_RO.
218    revokeFromTable(TEST_UTIL, USER_RO.getShortName(), TEST_TABLE, TEST_FAMILY, null,
219      Permission.Action.READ);
220    verifyAllowed(new AccessReadAction(TEST_TABLE), USER_OWNER, USER_RW);
221    verifyDenied(new AccessReadAction(TEST_TABLE), USER_RO, USER_NONE);
222    verifyAllowed(new AccessWriteAction(TEST_TABLE), USER_OWNER, USER_RW);
223    verifyDenied(new AccessWriteAction(TEST_TABLE), USER_RO, USER_NONE);
224
225    // restore snapshot with restoreAcl false.
226    admin.disableTable(TEST_TABLE);
227    admin.restoreSnapshot(snapshotName1, false, false);
228    admin.enableTable(TEST_TABLE);
229    verifyAllowed(new AccessReadAction(TEST_TABLE), USER_OWNER, USER_RW);
230    verifyDenied(new AccessReadAction(TEST_TABLE), USER_RO, USER_NONE);
231    verifyAllowed(new AccessWriteAction(TEST_TABLE), USER_OWNER, USER_RW);
232    verifyDenied(new AccessWriteAction(TEST_TABLE), USER_RO, USER_NONE);
233
234    // restore snapshot with restoreAcl true.
235    admin.disableTable(TEST_TABLE);
236    admin.restoreSnapshot(snapshotName1, false, true);
237    admin.enableTable(TEST_TABLE);
238    verifyAllowed(new AccessReadAction(TEST_TABLE), USER_OWNER, USER_RO, USER_RW);
239    verifyDenied(new AccessReadAction(TEST_TABLE), USER_NONE);
240    verifyAllowed(new AccessWriteAction(TEST_TABLE), USER_OWNER, USER_RW);
241    verifyDenied(new AccessWriteAction(TEST_TABLE), USER_RO, USER_NONE);
242  }
