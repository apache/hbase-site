043  @Test
044  public void testAppendCopyConstructor() throws IOException {
045    Append origin = new Append(Bytes.toBytes("ROW-01"));
046    origin.setPriority(100);
047    byte[] family = Bytes.toBytes("CF-01");
048
049    origin.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
050      .setRow(origin.getRow())
051      .setFamily(family)
052      .setQualifier(Bytes.toBytes("q"))
053      .setType(DataType.Put)
054      .setValue(Bytes.toBytes(100))
055      .build());
056    origin.addColumn(family, Bytes.toBytes("q0"), Bytes.toBytes("value"));
057    origin.setTimeRange(100, 1000);
058    Append clone = new Append(origin);
059    assertEquals(origin, clone);
060    origin.addColumn(family, Bytes.toBytes("q1"), Bytes.toBytes("value"));
061
062    //They should have different cell lists
063    assertNotEquals(origin.getCellList(family), clone.getCellList(family));
064  }

137  private void assertEquals(Mutation origin, Mutation clone) {
138    Assert.assertEquals(origin.getFamilyCellMap().size(), clone.getFamilyCellMap().size());
139    for (byte[] family : origin.getFamilyCellMap().keySet()) {
140      List<Cell> originCells = origin.getCellList(family);
141      List<Cell> cloneCells = clone.getCellList(family);
142      Assert.assertEquals(originCells.size(), cloneCells.size());
143      for (int i = 0; i != cloneCells.size(); ++i) {
144        Cell originCell = originCells.get(i);
145        Cell cloneCell = cloneCells.get(i);
146        assertTrue(CellUtil.equals(originCell, cloneCell));
147        assertTrue(CellUtil.matchingValue(originCell, cloneCell));
148      }
149    }
150    Assert.assertEquals(origin.getAttributesMap().size(), clone.getAttributesMap().size());
151    for (String name : origin.getAttributesMap().keySet()) {
152      byte[] originValue = origin.getAttributesMap().get(name);
153      byte[] cloneValue = clone.getAttributesMap().get(name);
154      assertTrue(Bytes.equals(originValue, cloneValue));
155    }
156    Assert.assertEquals(origin.getTimeStamp(), clone.getTimeStamp());
157    Assert.assertEquals(origin.getPriority(), clone.getPriority());
158    if (origin instanceof Append) {
159      assertEquals(((Append)origin).getTimeRange(), ((Append)clone).getTimeRange());
160    }
161    if (origin instanceof Increment) {
162      assertEquals(((Increment)origin).getTimeRange(), ((Increment)clone).getTimeRange());
163    }
164  }
165
166  private static void assertEquals(TimeRange origin, TimeRange clone) {
167    Assert.assertEquals(origin.getMin(), clone.getMin());
168    Assert.assertEquals(origin.getMax(), clone.getMax());
169  }

171  // HBASE-14881
172  @Test
173  public void testRowIsImmutableOrNot() {
174    byte[] rowKey = Bytes.toBytes("immutable");
175
176    // Test when row key is immutable
177    Put putRowIsImmutable = new Put(rowKey, true);
178    assertTrue(rowKey == putRowIsImmutable.getRow());  // No local copy is made
179
180    // Test when row key is not immutable
181    Put putRowIsNotImmutable = new Put(rowKey, 1000L, false);
182    assertTrue(rowKey != putRowIsNotImmutable.getRow());  // A local copy is made
183  }

185  // HBASE-14882
186  @Test
187  public void testAddImmutableToPut() throws IOException {
188    byte[] row        = Bytes.toBytes("immutable-row");
189    byte[] family     = Bytes.toBytes("immutable-family");
190
191    byte[] qualifier0 = Bytes.toBytes("immutable-qualifier-0");
192    byte[] value0     = Bytes.toBytes("immutable-value-0");
193
194    byte[] qualifier1 = Bytes.toBytes("immutable-qualifier-1");
195    byte[] value1     = Bytes.toBytes("immutable-value-1");
196    long   ts1        = 5000L;
197
198    // "true" indicates that the input row is immutable
199    Put put = new Put(row, true);
200    put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
201            .setRow(row)
202            .setFamily(family)
203            .setQualifier(qualifier0)
204            .setTimestamp(put.getTimeStamp())
205            .setType(DataType.Put)
206            .setValue(value0)
207            .build())
208        .add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
209            .setRow(row)
210            .setFamily(family)
211            .setQualifier(qualifier1)
212            .setTimestamp(ts1)
213            .setType(DataType.Put)
214            .setValue(value1)
215            .build());
216
217    // Verify the cell of family:qualifier0
218    Cell cell0 = put.get(family, qualifier0).get(0);
219
220    // Verify no local copy is made for family, qualifier or value
221    assertTrue(cell0.getFamilyArray()    == family);
222    assertTrue(cell0.getQualifierArray() == qualifier0);
223    assertTrue(cell0.getValueArray()     == value0);
224
225    // Verify timestamp
226    assertTrue(cell0.getTimestamp()      == put.getTimeStamp());
227
228    // Verify the cell of family:qualifier1
229    Cell cell1 = put.get(family, qualifier1).get(0);
230
231    // Verify no local copy is made for family, qualifier or value
232    assertTrue(cell1.getFamilyArray()    == family);
233    assertTrue(cell1.getQualifierArray() == qualifier1);
234    assertTrue(cell1.getValueArray()     == value1);
235
236    // Verify timestamp
237    assertTrue(cell1.getTimestamp()      == ts1);
238  }
