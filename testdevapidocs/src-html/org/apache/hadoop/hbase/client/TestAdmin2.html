112  @Test (timeout=300000)
113  public void testCreateBadTables() throws IOException {
114    String msg = null;
115    try {
116      this.admin.createTable(new HTableDescriptor(TableName.META_TABLE_NAME));
117    } catch(TableExistsException e) {
118      msg = e.toString();
119    }
120    assertTrue("Unexcepted exception message " + msg, msg != null &&
121      msg.startsWith(TableExistsException.class.getName()) &&
122      msg.contains(TableName.META_TABLE_NAME.getNameAsString()));
123
124    // Now try and do concurrent creation with a bunch of threads.
125    final HTableDescriptor threadDesc =
126      new HTableDescriptor(TableName.valueOf("threaded_testCreateBadTables"));
127    threadDesc.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
128    int count = 10;
129    Thread [] threads = new Thread [count];
130    final AtomicInteger successes = new AtomicInteger(0);
131    final AtomicInteger failures = new AtomicInteger(0);
132    final Admin localAdmin = this.admin;
133    for (int i = 0; i < count; i++) {
134      threads[i] = new Thread(Integer.toString(i)) {
135        @Override
136        public void run() {
137          try {
138            localAdmin.createTable(threadDesc);
139            successes.incrementAndGet();
140          } catch (TableExistsException e) {
141            failures.incrementAndGet();
142          } catch (IOException e) {
143            throw new RuntimeException("Failed threaded create" + getName(), e);
144          }
145        }
146      };
147    }
148    for (int i = 0; i < count; i++) {
149      threads[i].start();
150    }
151    for (int i = 0; i < count; i++) {
152      while(threads[i].isAlive()) {
153        try {
154          Thread.sleep(100);
155        } catch (InterruptedException e) {
156          // continue
157        }
158      }
159    }
160    // All threads are now dead.  Count up how many tables were created and
161    // how many failed w/ appropriate exception.
162    assertEquals(1, successes.get());
163    assertEquals(count - 1, failures.get());
164  }

537    setUpforLogRolling();
538    String className = this.getClass().getName();

554    LOG.info("after flushing all regions and rolling logs there are " +
555        count + " log files");

559  private void setUpforLogRolling() {
560    // Force a region split after every 768KB
561    TEST_UTIL.getConfiguration().setLong(HConstants.HREGION_MAX_FILESIZE,
562        768L * 1024L);
563
564    // We roll the log after every 32 writes
565    TEST_UTIL.getConfiguration().setInt("hbase.regionserver.maxlogentries", 32);
566
567    TEST_UTIL.getConfiguration().setInt(
568        "hbase.regionserver.logroll.errors.tolerated", 2);
569    TEST_UTIL.getConfiguration().setInt("hbase.rpc.timeout", 10 * 1000);
570
571    // For less frequently updated regions flush after every 2 flushes
572    TEST_UTIL.getConfiguration().setInt(
573        "hbase.hregion.memstore.optionalflushcount", 2);
574
575    // We flush the cache after every 8192 bytes
576    TEST_UTIL.getConfiguration().setInt(HConstants.HREGION_MEMSTORE_FLUSH_SIZE,
577        8192);
578
579    // Increase the amount of time between client retries
580    TEST_UTIL.getConfiguration().setLong("hbase.client.pause", 10 * 1000);
581
582    // Reduce thread wake frequency so that other threads can get
583    // a chance to run.
584    TEST_UTIL.getConfiguration().setInt(HConstants.THREAD_WAKE_FREQUENCY,
585        2 * 1000);
586
587    /**** configuration for testLogRollOnDatanodeDeath ****/
588    // make sure log.hflush() calls syncFs() to open a pipeline
589    TEST_UTIL.getConfiguration().setBoolean("dfs.support.append", true);
590    // lower the namenode & datanode heartbeat so the namenode
591    // quickly detects datanode failures
592    TEST_UTIL.getConfiguration().setInt("dfs.namenode.heartbeat.recheck-interval", 5000);
593    TEST_UTIL.getConfiguration().setInt("dfs.heartbeat.interval", 1);
594    // the namenode might still try to choose the recently-dead datanode
595    // for a pipeline, so try to a new pipeline multiple times
596    TEST_UTIL.getConfiguration().setInt("dfs.client.block.write.retries", 30);
597    TEST_UTIL.getConfiguration().setInt(
598        "hbase.regionserver.hlog.tolerable.lowreplication", 2);
599    TEST_UTIL.getConfiguration().setInt(
600        "hbase.regionserver.hlog.lowreplication.rolllimit", 3);
601  }

603  private HRegionServer startAndWriteData(TableName tableName, byte[] value)
604  throws IOException, InterruptedException {
605    // When the hbase:meta table can be opened, the region servers are running
606    TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME).close();
607
608    // Create the test table and open it
609    HTableDescriptor desc = new HTableDescriptor(tableName);

611    admin.createTable(desc);
612    Table table = TEST_UTIL.getConnection().getTable(tableName);
613
614    HRegionServer regionServer = TEST_UTIL.getRSForFirstRegionInTable(tableName);
615    for (int i = 1; i <= 256; i++) { // 256 writes should cause 8 log rolls
616      Put put = new Put(Bytes.toBytes("row" + String.format("%1$04d", i)));
617      put.addColumn(HConstants.CATALOG_FAMILY, null, value);
618      table.put(put);
619      if (i % 32 == 0) {
620        // After every 32 writes sleep to let the log roller run
621        try {
622          Thread.sleep(2000);
623        } catch (InterruptedException e) {
624          // continue
625        }
626      }
627    }
628
629    table.close();
630    return regionServer;

713

725
