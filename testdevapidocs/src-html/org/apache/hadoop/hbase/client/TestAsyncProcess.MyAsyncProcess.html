295      return multiAction.actions.values().stream()
296              .flatMap(v -> v.stream())
297              .map(action -> action.getAction())
298              .filter(row -> row instanceof Mutation)
299              .mapToLong(row -> ((Mutation) row).heapSize())
300              .sum();

355  class MyAsyncProcessWithReplicas extends MyAsyncProcess {
356    private Set<byte[]> failures = new TreeSet<>(new Bytes.ByteArrayComparator());
357    private long primarySleepMs = 0, replicaSleepMs = 0;
358    private Map<ServerName, Long> customPrimarySleepMs = new HashMap<>();
359    private final AtomicLong replicaCalls = new AtomicLong(0);
360
361    public void addFailures(HRegionInfo... hris) {
362      for (HRegionInfo hri : hris) {
363        failures.add(hri.getRegionName());
364      }

368      return replicaCalls.get();
369    }
370
371    public void setPrimaryCallDelay(ServerName server, long primaryMs) {
372      customPrimarySleepMs.put(server, primaryMs);
373    }

375    public MyAsyncProcessWithReplicas(ClusterConnection hc, Configuration conf) {
376      super(hc, conf);
377    }
378
379    public void setCallDelays(long primaryMs, long replicaMs) {
380      this.primarySleepMs = primaryMs;
381      this.replicaSleepMs = replicaMs;
382    }
383
384    @Override
385    protected RpcRetryingCaller<AbstractResponse> createCaller(
386        CancellableRegionServerCallable payloadCallable, int rpcTimeout) {
387      MultiServerCallable callable = (MultiServerCallable) payloadCallable;
388      final MultiResponse mr = createMultiResponse(
389          callable.getMulti(), nbMultiResponse, nbActions, new ResponseGenerator() {
390            @Override
391            public void addResponse(MultiResponse mr, byte[] regionName, Action a) {
392              if (failures.contains(regionName)) {
393                mr.add(regionName, a.getOriginalIndex(), failure);
394              } else {
395                boolean isStale = !RegionReplicaUtil.isDefaultReplica(a.getReplicaId());
396                mr.add(regionName, a.getOriginalIndex(),
397                    Result.create(new Cell[0], null, isStale));
398              }
399            }
400          });
401      // Currently AsyncProcess either sends all-replica, or all-primary request.
402      final boolean isDefault = RegionReplicaUtil.isDefaultReplica(
403          callable.getMulti().actions.values().iterator().next().iterator().next().getReplicaId());
404      final ServerName server = ((MultiServerCallable)callable).getServerName();
405      String debugMsg = "Call to " + server + ", primary=" + isDefault + " with "
406          + callable.getMulti().actions.size() + " entries: ";
407      for (byte[] region : callable.getMulti().actions.keySet()) {
408        debugMsg += "[" + Bytes.toStringBinary(region) + "], ";
409      }
410      LOG.debug(debugMsg);
411      if (!isDefault) {
412        replicaCalls.incrementAndGet();
413      }
414
415      return new RpcRetryingCallerImpl<AbstractResponse>(100, 500, 10, 9) {
416        @Override
417        public MultiResponse callWithoutRetries(RetryingCallable<AbstractResponse> callable,
418                                                int callTimeout)
419        throws IOException, RuntimeException {
420          long sleep = -1;
421          if (isDefault) {
422            Long customSleep = customPrimarySleepMs.get(server);
423            sleep = (customSleep == null ? primarySleepMs : customSleep.longValue());
424          } else {
425            sleep = replicaSleepMs;
426          }
427          if (sleep != 0) {
428            try {
429              Thread.sleep(sleep);
430            } catch (InterruptedException e) {
431            }
432          }
433          return mr;
434        }
435      };
436    }

439  static MultiResponse createMultiResponse(final MultiAction multi,
440      AtomicInteger nbMultiResponse, AtomicInteger nbActions, ResponseGenerator gen) {
441    final MultiResponse mr = new MultiResponse();
442    nbMultiResponse.incrementAndGet();
443    for (Map.Entry<byte[], List<Action>> entry : multi.actions.entrySet()) {
444      byte[] regionName = entry.getKey();
445      for (Action a : entry.getValue()) {
446        nbActions.incrementAndGet();
447        gen.addResponse(mr, regionName, a);
448      }
449    }
450    return mr;
451  }

616  private void doSubmitRequest(long maxHeapSizePerRequest, long putsHeapSize) throws Exception {
617    ClusterConnection conn = createHConnection();
618    final String defaultClazz = conn.getConfiguration().get(RequestControllerFactory.REQUEST_CONTROLLER_IMPL_CONF_KEY);
619    final long defaultHeapSizePerRequest = conn.getConfiguration().getLong(
620      SimpleRequestController.HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE,
621      SimpleRequestController.DEFAULT_HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE);
622    conn.getConfiguration().set(RequestControllerFactory.REQUEST_CONTROLLER_IMPL_CONF_KEY,
623      SimpleRequestController.class.getName());
624    conn.getConfiguration().setLong(SimpleRequestController.HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE, maxHeapSizePerRequest);
625
626    // sn has two regions
627    long putSizeSN = 0;
628    long putSizeSN2 = 0;
629    List<Put> puts = new ArrayList<>();
630    while ((putSizeSN + putSizeSN2) <= putsHeapSize) {
631      Put put1 = new Put(DUMMY_BYTES_1);
632      put1.addColumn(DUMMY_BYTES_1, DUMMY_BYTES_1, DUMMY_BYTES_1);
633      Put put2 = new Put(DUMMY_BYTES_2);
634      put2.addColumn(DUMMY_BYTES_2, DUMMY_BYTES_2, DUMMY_BYTES_2);
635      Put put3 = new Put(DUMMY_BYTES_3);
636      put3.addColumn(DUMMY_BYTES_3, DUMMY_BYTES_3, DUMMY_BYTES_3);
637      putSizeSN += (put1.heapSize() + put2.heapSize());
638      putSizeSN2 += put3.heapSize();

645    int minCountSn2Request = (int) calculateRequestCount(putSizeSN2, maxHeapSizePerRequest);
646    LOG.info("Total put count:" + puts.size() + ", putSizeSN:"+ putSizeSN + ", putSizeSN2:" + putSizeSN2

648      + ", minCountSnRequest:" + minCountSnRequest
649      + ", minCountSn2Request:" + minCountSn2Request);

664        }
665        MyAsyncRequestFutureImpl ars = (MyAsyncRequestFutureImpl) req;

671        }

674      assertEquals(true, minCountSnRequest <= actualSnReqCount);
