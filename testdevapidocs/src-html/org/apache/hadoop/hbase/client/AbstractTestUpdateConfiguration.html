018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021import java.nio.file.Files;
022import java.nio.file.Path;
023import java.nio.file.Paths;
024import java.nio.file.StandardCopyOption;
025import org.apache.hadoop.hbase.HBaseTestingUtil;
026import org.apache.hadoop.hbase.util.JVMClusterUtil.RegionServerThread;
027
028/**
029 * Base class to test Configuration Update logic. It wraps up things needed to test configuration
030 * change and provides utility methods for test cluster setup, updating/restoring configuration
031 * file.
032 */
033public abstract class AbstractTestUpdateConfiguration {
034  private static final String SERVER_CONFIG = "hbase-site.xml";
035  private static final String OVERRIDE_SERVER_CONFIG = "override-hbase-site.xml";
036  private static final String BACKUP_SERVER_CONFIG = "backup-hbase-site.xml";
037
038  private static Path configFileUnderTestDataDir;
039  private static Path overrideConfigFileUnderTestDataDir;
040  private static Path backupConfigFileUnderTestDataDir;
041
042  protected static void setUpConfigurationFiles(final HBaseTestingUtil testUtil) throws Exception {
043    // Before this change, the test will update hbase-site.xml under target/test-classes and
044    // trigger a config reload. Since target/test-classes/hbase-site.xml is being used by
045    // other testing cases at the same time, this update will break other testing cases so it will
046    // be flakey in nature.
047    // To avoid this, the change is to make target/test-classes/hbase-site.xml immutable. A new
048    // hbase-site.xml will be created under its test data directory, i.e,
049    // hbase-server/target/test-data/UUID, this new file will be added as a resource for the
050    // config, new update will be applied to this new file and only visible to this specific test
051    // case. The target/test-classes/hbase-site.xml will not be changed during the test.
052
053    String absoluteDataPath = testUtil.getDataTestDir().toString();
054
055    // Create test-data directories.
056    Files.createDirectories(Paths.get(absoluteDataPath));
057
058    // Copy hbase-site.xml from target/test-class to target/test-data/UUID directory.
059    Path configFile = Paths.get("target", "test-classes", SERVER_CONFIG);
060    configFileUnderTestDataDir = Paths.get(absoluteDataPath, SERVER_CONFIG);
061    Files.copy(configFile, configFileUnderTestDataDir);
062
063    // Copy override config file overrider-hbase-site.xml from target/test-class to
064    // target/test-data/UUID directory.
065    Path overrideConfigFile = Paths.get("target", "test-classes", OVERRIDE_SERVER_CONFIG);
066    overrideConfigFileUnderTestDataDir = Paths.get(absoluteDataPath, OVERRIDE_SERVER_CONFIG);
067    Files.copy(overrideConfigFile, overrideConfigFileUnderTestDataDir);
068
069    backupConfigFileUnderTestDataDir = Paths.get(absoluteDataPath, BACKUP_SERVER_CONFIG);
070
071    // Add the new custom config file to Configuration
072    testUtil.getConfiguration().addResource(testUtil.getDataTestDir(SERVER_CONFIG));
073  }
074
075  protected static void addResourceToRegionServerConfiguration(final HBaseTestingUtil testUtil) {
076    // When RegionServer is created in MiniHBaseCluster, it uses HBaseConfiguration.create(conf) of
077    // the master Configuration. The create() just copies config params over, it does not do
078    // a clone for a historic reason. Properties such as resources are lost during this process.
079    // Exposing a new method in HBaseConfiguration causes confusion. Instead, the new hbase-site.xml
080    // under test-data directory is added to RegionServer's configuration as a workaround.
081    for (RegionServerThread rsThread : testUtil.getMiniHBaseCluster().getRegionServerThreads()) {
082      rsThread.getRegionServer().getConfiguration()
083        .addResource(testUtil.getDataTestDir(SERVER_CONFIG));
084    }
085  }
086
087  /**
088   * Replace the hbase-site.xml file under this test's data directory with the content of the
089   * override-hbase-site.xml file. Stashes the current existing file so that it can be restored
090   * using {@link #restoreHBaseSiteXML()}.
091   * @throws IOException if an I/O error occurs
092   */
093  protected void replaceHBaseSiteXML() throws IOException {
094    // make a backup of hbase-site.xml
095    Files.copy(configFileUnderTestDataDir, backupConfigFileUnderTestDataDir,
096      StandardCopyOption.REPLACE_EXISTING);
097    // update hbase-site.xml by overwriting it
098    Files.copy(overrideConfigFileUnderTestDataDir, configFileUnderTestDataDir,
099      StandardCopyOption.REPLACE_EXISTING);
100  }
101
102  /**
103   * Restores the hbase-site.xml file that was stashed by a previous call to
104   * {@link #replaceHBaseSiteXML()}.
105   * @throws IOException if an I/O error occurs
106   */
107  protected void restoreHBaseSiteXML() throws IOException {
108    // restore hbase-site.xml
109    Files.copy(backupConfigFileUnderTestDataDir, configFileUnderTestDataDir,
110      StandardCopyOption.REPLACE_EXISTING);
111  }
112}
