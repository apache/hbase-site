126  /**
127   * A dummy RpcChannel implementation that intercepts the GetClusterId() RPC calls and injects
128   * errors. All other RPCs are ignored.
129   */
130  public static final class RpcChannelImpl implements RpcChannel {
131
132    @Override
133    public void callMethod(MethodDescriptor method, RpcController controller, Message request,
134      Message responsePrototype, RpcCallback<Message> done) {
135      if (!method.getName().equals("GetClusterId")) {
136        // On RPC failures, MasterRegistry internally runs getMasters() RPC to keep the master list
137        // fresh. We do not want to intercept those RPCs here and double count.
138        return;
139      }
140      // simulate the asynchronous behavior otherwise all logic will perform in the same thread...
141      EXECUTOR.execute(() -> {
142        int index = CALLED.getAndIncrement();
143        if (index == BAD_RESP_INDEX) {
144          done.run(GetClusterIdResponse.getDefaultInstance());
145        } else if (GOOD_RESP_INDEXS.contains(index)) {
146          done.run(RESP);
147        } else {
148          controller.setFailed("inject error");
149          done.run(null);
150        }
151      });
152    }
153  }

155  private AbstractRpcBasedConnectionRegistry createRegistry(int hedged) throws IOException {
156    Configuration conf = UTIL.getConfiguration();
157    conf.setInt(HEDGED_REQS_FANOUT_CONFIG_NAME, hedged);
158    return new AbstractRpcBasedConnectionRegistry(conf, HEDGED_REQS_FANOUT_CONFIG_NAME,
159      INITIAL_DELAY_SECS_CONFIG_NAME, REFRESH_INTERVAL_SECS_CONFIG_NAME,
160      MIN_REFRESH_INTERVAL_SECS_CONFIG_NAME) {
161
162      @Override
163      protected Set<ServerName> getBootstrapNodes(Configuration conf) throws IOException {
164        return BOOTSTRAP_NODES;
165      }
166
167      @Override
168      protected CompletableFuture<Set<ServerName>> fetchEndpoints() {
169        return CompletableFuture.completedFuture(BOOTSTRAP_NODES);
170      }
171
172      @Override
173      public String getConnectionString() {
174        return "unimplemented";
175      }
176    };
177  }

189      .mapToObj(i -> ServerName.valueOf("localhost", (10000 + 100 * i), ServerName.NON_STARTCODE))
190      .collect(Collectors.toSet());

205  private <T> T logIfError(CompletableFuture<T> future) throws IOException {
206    try {
207      return FutureUtils.get(future);
208    } catch (Throwable t) {
209      LOG.warn("", t);
210      throw t;
211    }
212  }

214  @Test
215  public void testAllFailNoHedged() throws IOException {
216    try (AbstractRpcBasedConnectionRegistry registry = createRegistry(1)) {
217      assertThrows(IOException.class, () -> logIfError(registry.getClusterId()));
218      assertEquals(10, CALLED.get());
219    }
220  }
221
222  @Test
223  public void testAllFailHedged3() throws IOException {
224    BAD_RESP_INDEX = 5;
225    try (AbstractRpcBasedConnectionRegistry registry = createRegistry(3)) {
226      assertThrows(IOException.class, () -> logIfError(registry.getClusterId()));
227      assertEquals(10, CALLED.get());
228    }
229  }

231  @Test
232  public void testFirstSucceededNoHedge() throws IOException {
233    GOOD_RESP_INDEXS =
234      IntStream.range(0, 10).mapToObj(Integer::valueOf).collect(Collectors.toSet());
235    // will be set to 1
236    try (AbstractRpcBasedConnectionRegistry registry = createRegistry(0)) {
237      String clusterId = logIfError(registry.getClusterId());
238      assertEquals(RESP.getClusterId(), clusterId);
239      assertEquals(1, CALLED.get());
240    }
241  }
242
243  @Test
244  public void testSecondRoundSucceededHedge4() throws IOException {
245    GOOD_RESP_INDEXS = Collections.singleton(6);
246    try (AbstractRpcBasedConnectionRegistry registry = createRegistry(4)) {
247      String clusterId = logIfError(registry.getClusterId());
248      assertEquals(RESP.getClusterId(), clusterId);
249      UTIL.waitFor(5000, () -> CALLED.get() == 8);
250    }
251  }
252
253  @Test
254  public void testSucceededWithLargestHedged() throws IOException, InterruptedException {
255    GOOD_RESP_INDEXS = Collections.singleton(5);
256    try (AbstractRpcBasedConnectionRegistry registry = createRegistry(Integer.MAX_VALUE)) {
257      String clusterId = logIfError(registry.getClusterId());
258      assertEquals(RESP.getClusterId(), clusterId);
259      UTIL.waitFor(5000, () -> CALLED.get() == 10);
260      Thread.sleep(1000);
261      // make sure we do not send more
262      assertEquals(10, CALLED.get());
263    }
264  }
