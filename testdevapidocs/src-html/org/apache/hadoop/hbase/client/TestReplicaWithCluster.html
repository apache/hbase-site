087  /**
088   * This copro is used to synchronize the tests.
089   */
090  public static class SlowMeCopro implements RegionCoprocessor, RegionObserver {
091    static final AtomicLong sleepTime = new AtomicLong(0);
092    static final AtomicReference<CountDownLatch> cdl = new AtomicReference<>(new CountDownLatch(0));
093
094    public SlowMeCopro() {
095    }
096
097    @Override
098    public Optional<RegionObserver> getRegionObserver() {
099      return Optional.of(this);
100    }
101
102    @Override
103    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
104                         final Get get, final List<Cell> results) throws IOException {
105
106      if (e.getEnvironment().getRegion().getRegionInfo().getReplicaId() == 0) {
107        CountDownLatch latch = cdl.get();
108        try {
109          if (sleepTime.get() > 0) {
110            LOG.info("Sleeping for " + sleepTime.get() + " ms");
111            Thread.sleep(sleepTime.get());
112          } else if (latch.getCount() > 0) {
113            LOG.info("Waiting for the counterCountDownLatch");
114            latch.await(2, TimeUnit.MINUTES); // To help the tests to finish.
115            if (latch.getCount() > 0) {
116              throw new RuntimeException("Can't wait more");
117            }
118          }
119        } catch (InterruptedException e1) {
120          LOG.error(e1);
121        }
122      } else {
123        LOG.info("We're not the primary replicas.");
124      }
125    }
126  }

177  /**
178   * This copro is used to slow down the primary meta region scan a bit
179   */
180  public static class RegionServerHostingPrimayMetaRegionSlowOrStopCopro
181      implements RegionCoprocessor, RegionObserver {
182    static boolean slowDownPrimaryMetaScan = false;
183    static boolean throwException = false;
184
185    @Override
186    public Optional<RegionObserver> getRegionObserver() {
187      return Optional.of(this);
188    }
189
190    @Override
191    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
192        final Get get, final List<Cell> results) throws IOException {
193
194      int replicaId = e.getEnvironment().getRegion().getRegionInfo().getReplicaId();
195
196      // Fail for the primary replica, but not for meta
197      if (throwException) {
198        if (!e.getEnvironment().getRegion().getRegionInfo().isMetaRegion() && (replicaId == 0)) {
199          LOG.info("Get, throw Region Server Stopped Exceptoin for region " + e.getEnvironment()
200              .getRegion().getRegionInfo());
201          throw new RegionServerStoppedException("Server " +
202            ((HasRegionServerServices)e.getEnvironment()).getRegionServerServices().getServerName()
203                  + " not running");
204        }
205      } else {
206        LOG.info("Get, We're replica region " + replicaId);
207      }
208    }
209
210    @Override
211    public RegionScanner preScannerOpen(final ObserverContext<RegionCoprocessorEnvironment> e,
212        final Scan scan, final RegionScanner s) throws IOException {
213
214      int replicaId = e.getEnvironment().getRegion().getRegionInfo().getReplicaId();
215
216      // Slow down with the primary meta region scan
217      if (e.getEnvironment().getRegion().getRegionInfo().isMetaRegion() && (replicaId == 0)) {
218        if (slowDownPrimaryMetaScan) {
219          LOG.info("Scan with primary meta region, slow down a bit");
220          try {
221            Thread.sleep(META_SCAN_TIMEOUT_IN_MILLISEC - 50);
222          } catch (InterruptedException ie) {
223            // Ingore
224          }
225        }
226
227        // Fail for the primary replica
228        if (throwException) {
229          LOG.info("Scan, throw Region Server Stopped Exceptoin for replica " + e.getEnvironment()
230              .getRegion().getRegionInfo());
231
232          throw new RegionServerStoppedException("Server " +
233            ((HasRegionServerServices)e.getEnvironment()).getRegionServerServices().getServerName()
234               + " not running");
235        } else {
236          LOG.info("Scan, We're replica region " + replicaId);
237        }
238      } else {
239        LOG.info("Scan, We're replica region " + replicaId);
240      }
241
242      return null;
243    }
244  }

292  public void testCreateDeleteTable() throws IOException {
293    // Create table then get the single region for our new table.
294    HTableDescriptor hdt = HTU.createTableDescriptor("testCreateDeleteTable");
295    hdt.setRegionReplication(NB_SERVERS);
296    hdt.addCoprocessor(SlowMeCopro.class.getName());

300    p.addColumn(f, row, row);
301    table.put(p);
302
303    Get g = new Get(row);
304    Result r = table.get(g);
305    Assert.assertFalse(r.isStale());
306
307    try {
308      // But if we ask for stale we will get it
309      SlowMeCopro.cdl.set(new CountDownLatch(1));
310      g = new Get(row);
311      g.setConsistency(Consistency.TIMELINE);
312      r = table.get(g);
313      Assert.assertTrue(r.isStale());
314      SlowMeCopro.cdl.get().countDown();
315    } finally {
316      SlowMeCopro.cdl.get().countDown();
317      SlowMeCopro.sleepTime.set(0);
318    }
319
320    HTU.getAdmin().disableTable(hdt.getTableName());
321    HTU.deleteTable(hdt.getTableName());

540      SlowMeCopro.cdl.get().countDown();
541      SlowMeCopro.sleepTime.set(0);
542    }

545    HTU.deleteTable(hdt.getTableName());

708  // This test is to simulate the case that the meta region and the primary user region
709  // are down, hbase client is able to access user replica regions and return stale data.
710  // Meta replica is enabled to show the case that the meta replica region could be out of sync
711  // with the primary meta region.
712  @Test
713  public void testReplicaGetWithPrimaryAndMetaDown() throws IOException, InterruptedException {
714    HTU.getAdmin().setBalancerRunning(false, true);
715
716    ((ConnectionImplementation)HTU.getAdmin().getConnection()).setUseMetaReplicas(true);
717
718    // Create table then get the single region for our new table.
719    HTableDescriptor hdt = HTU.createTableDescriptor("testReplicaGetWithPrimaryAndMetaDown");
720    hdt.setRegionReplication(2);
721    try {
722
723      Table table = HTU.createTable(hdt, new byte[][] { f }, null);
724
725      // Get Meta location
726      RegionLocations mrl = ((ClusterConnection) HTU.getConnection())
727          .locateRegion(TableName.META_TABLE_NAME,
728              HConstants.EMPTY_START_ROW, false, false);
729
730      // Get user table location
731      RegionLocations url = ((ClusterConnection) HTU.getConnection())
732          .locateRegion(hdt.getTableName(), row, false, false);
733
734      // Make sure that user primary region is co-hosted with the meta region
735      if (!url.getDefaultRegionLocation().getServerName().equals(
736          mrl.getDefaultRegionLocation().getServerName())) {
737        HTU.moveRegionAndWait(url.getDefaultRegionLocation().getRegionInfo(),
738            mrl.getDefaultRegionLocation().getServerName());
739      }
740
741      // Make sure that the user replica region is not hosted by the same region server with
742      // primary
743      if (url.getRegionLocation(1).getServerName().equals(mrl.getDefaultRegionLocation()
744          .getServerName())) {
745        HTU.moveRegionAndWait(url.getRegionLocation(1).getRegionInfo(),
746            url.getDefaultRegionLocation().getServerName());
747      }
748
749      // Wait until the meta table is updated with new location info
750      while (true) {
751        mrl = ((ClusterConnection) HTU.getConnection())
752            .locateRegion(TableName.META_TABLE_NAME, HConstants.EMPTY_START_ROW, false, false);
753
754        // Get user table location
755        url = ((ClusterConnection) HTU.getConnection())
756            .locateRegion(hdt.getTableName(), row, false, true);
757
758        LOG.info("meta locations " + mrl);
759        LOG.info("table locations " + url);
760        ServerName a = url.getDefaultRegionLocation().getServerName();
761        ServerName b = mrl.getDefaultRegionLocation().getServerName();
762        if(a.equals(b)) {
763          break;
764        } else {
765          LOG.info("Waiting for new region info to be updated in meta table");
766          Thread.sleep(100);
767        }
768      }
769
770      Put p = new Put(row);
771      p.addColumn(f, row, row);
772      table.put(p);
773
774      // Flush so it can be picked by the replica refresher thread
775      HTU.flush(table.getName());
776
777      // Sleep for some time until data is picked up by replicas
778      try {
779        Thread.sleep(2 * REFRESH_PERIOD);
780      } catch (InterruptedException e1) {
781        LOG.error(e1);
782      }
783
784      // Simulating the RS down
785      RegionServerHostingPrimayMetaRegionSlowOrStopCopro.throwException = true;
786
787      // The first Get is supposed to succeed
788      Get g = new Get(row);
789      g.setConsistency(Consistency.TIMELINE);
790      Result r = table.get(g);
791      Assert.assertTrue(r.isStale());
792
793      // The second Get will succeed as well
794      r = table.get(g);
795      Assert.assertTrue(r.isStale());
796
797    } finally {
798      ((ConnectionImplementation)HTU.getAdmin().getConnection()).setUseMetaReplicas(false);
799      RegionServerHostingPrimayMetaRegionSlowOrStopCopro.throwException = false;
800      HTU.getAdmin().setBalancerRunning(true, true);
801      HTU.getAdmin().disableTable(hdt.getTableName());
802      HTU.deleteTable(hdt.getTableName());
803    }
