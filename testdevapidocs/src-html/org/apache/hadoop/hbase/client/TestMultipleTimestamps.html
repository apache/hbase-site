178    Integer[] putColumns = new Integer[] { 1, 3, 5};
179    Long[] putTimestamps = new Long[] {1L, 2L, 3L, 4L, 5L};
180
181    Integer[] scanRows = new Integer[] {5, 7};
182    Integer[] scanColumns = new Integer[] {3, 4, 5};
183    Long[] scanTimestamps = new Long[] {2l, 3L};
184    int scanMaxVersions = 2;
185
186    put(ht, FAMILY, putRows, putColumns, putTimestamps);
187
188    TEST_UTIL.flush(tableName);
189    Scan scan = new Scan();
190    scan.setMaxVersions(10);
191    ResultScanner scanner = ht.getScanner(scan);
192    while (true) {
193      Result r = scanner.next();
194      if (r == null) break;
195      LOG.info("r=" + r);

197    scanner = scan(ht, FAMILY, scanRows, scanColumns, scanTimestamps, scanMaxVersions);
198
199    Cell[] kvs;
200
201    // This looks like wrong answer.  Should be 2.  Even then we are returning wrong result,
202    // timestamps that are 3 whereas should be 2 since min is inclusive.
203    kvs = scanner.next().rawCells();

207    checkOneCell(kvs[2], FAMILY, 5, 5, 3);
208    checkOneCell(kvs[3], FAMILY, 5, 5, 2);

211    checkOneCell(kvs[0], FAMILY, 7, 3, 3);
212    checkOneCell(kvs[1], FAMILY, 7, 3, 2);
213    checkOneCell(kvs[2], FAMILY, 7, 5, 3);
214    checkOneCell(kvs[3], FAMILY, 7, 5, 2);

288  public void testWithVersionDeletes(boolean flushTables) throws IOException {
289    LOG.info(name.getMethodName() + "_"+ (flushTables ? "flush" : "noflush"));
290    final TableName tableName = TableName.valueOf(name.getMethodName() + "_" + (flushTables ?
291            "flush" : "noflush"));
292    byte [] FAMILY = Bytes.toBytes("event_log");
293    byte [][] FAMILIES = new byte[][] { FAMILY };
294
295    // create table; set versions to max...
296    Table ht = TEST_UTIL.createTable(tableName, FAMILIES, Integer.MAX_VALUE);
297
298    // For row:0, col:0: insert versions 1 through 5.
299    putNVersions(ht, FAMILY, 0, 0, 1, 5);
300
301    if (flushTables) {
302      TEST_UTIL.flush(tableName);
303    }
304
305    // delete version 4.
306    deleteOneVersion(ht, FAMILY, 0, 0, 4);
307
308    // request a bunch of versions including the deleted version. We should
309    // only get back entries for the versions that exist.
310    Cell kvs[] = getNVersions(ht, FAMILY, 0, 0,
311        Arrays.asList(2L, 3L, 4L, 5L));
312    assertEquals(3, kvs.length);
313    checkOneCell(kvs[0], FAMILY, 0, 0, 5);
314    checkOneCell(kvs[1], FAMILY, 0, 0, 3);
315    checkOneCell(kvs[2], FAMILY, 0, 0, 2);
316
317    ht.close();
318  }

397  /**
398   * Assert that the passed in KeyValue has expected contents for the
399   * specified row, column & timestamp.
400   */
401  private void checkOneCell(Cell kv, byte[] cf,
402      int rowIdx, int colIdx, long ts) {
403
404    String ctx = "rowIdx=" + rowIdx + "; colIdx=" + colIdx + "; ts=" + ts;
405
406    assertEquals("Row mismatch which checking: " + ctx,
407        "row:"+ rowIdx, Bytes.toString(CellUtil.cloneRow(kv)));
408
409    assertEquals("ColumnFamily mismatch while checking: " + ctx,
410        Bytes.toString(cf), Bytes.toString(CellUtil.cloneFamily(kv)));
411
412    assertEquals("Column qualifier mismatch while checking: " + ctx,
413        "column:" + colIdx,
414        Bytes.toString(CellUtil.cloneQualifier(kv)));
415
416    assertEquals("Timestamp mismatch while checking: " + ctx,
417        ts, kv.getTimestamp());
418
419    assertEquals("Value mismatch while checking: " + ctx,
420        "value-version-" + ts, Bytes.toString(CellUtil.cloneValue(kv)));
421  }

423  /**
424   * Uses the TimestampFilter on a Get to request a specified list of
425   * versions for the row/column specified by rowIdx & colIdx.
426   *
427   */
428  private  Cell[] getNVersions(Table ht, byte[] cf, int rowIdx,
429      int colIdx, List<Long> versions)
430  throws IOException {
431    byte row[] = Bytes.toBytes("row:" + rowIdx);
432    byte column[] = Bytes.toBytes("column:" + colIdx);
433    Get get = new Get(row);
434    get.addColumn(cf, column);
435    get.setMaxVersions();
436    get.setTimeRange(Collections.min(versions), Collections.max(versions)+1);
437    Result result = ht.get(get);
438
439    return result.rawCells();
440  }
441
442  private  ResultScanner scan(Table ht, byte[] cf,
443      Integer[] rowIndexes, Integer[] columnIndexes,
444      Long[] versions, int maxVersions)
445  throws IOException {
446    Arrays.asList(rowIndexes);
447    byte startRow[] = Bytes.toBytes("row:" +
448        Collections.min( Arrays.asList(rowIndexes)));
449    byte endRow[] = Bytes.toBytes("row:" +
450        Collections.max( Arrays.asList(rowIndexes))+1);
451    Scan scan = new Scan(startRow, endRow);
452    for (Integer colIdx: columnIndexes) {
453      byte column[] = Bytes.toBytes("column:" + colIdx);
454      scan.addColumn(cf, column);
455    }
456    scan.setMaxVersions(maxVersions);
457    scan.setTimeRange(Collections.min(Arrays.asList(versions)),
458        Collections.max(Arrays.asList(versions))+1);
459    ResultScanner scanner = ht.getScanner(scan);
460    return scanner;
461  }
462
463  private void put(Table ht, byte[] cf, Integer[] rowIndexes,
464      Integer[] columnIndexes, Long[] versions)
465  throws IOException {
466    for (int rowIdx: rowIndexes) {
467      byte row[] = Bytes.toBytes("row:" + rowIdx);
468      Put put = new Put(row);
469      put.setDurability(Durability.SKIP_WAL);
470      for(int colIdx: columnIndexes) {
471        byte column[] = Bytes.toBytes("column:" + colIdx);
472        for (long version: versions) {
473          put.addColumn(cf, column, version, Bytes.toBytes("value-version-" +
474                  version));
475        }
476      }
477      ht.put(put);
478    }
479  }

481  /**
482   * Insert in specific row/column versions with timestamps
483   * versionStart..versionEnd.
484   */
485  private void putNVersions(Table ht, byte[] cf, int rowIdx, int colIdx,
486      long versionStart, long versionEnd)
487  throws IOException {
488    byte row[] = Bytes.toBytes("row:" + rowIdx);
489    byte column[] = Bytes.toBytes("column:" + colIdx);
490    Put put = new Put(row);
491    put.setDurability(Durability.SKIP_WAL);
492
493    for (long idx = versionStart; idx <= versionEnd; idx++) {
494      put.addColumn(cf, column, idx, Bytes.toBytes("value-version-" + idx));
495    }
496
497    ht.put(put);
498  }

500  /**
501   * For row/column specified by rowIdx/colIdx, delete the cell
502   * corresponding to the specified version.
503   */
504  private void deleteOneVersion(Table ht, byte[] cf, int rowIdx,
505      int colIdx, long version)
506  throws IOException {
507    byte row[] = Bytes.toBytes("row:" + rowIdx);
508    byte column[] = Bytes.toBytes("column:" + colIdx);
509    Delete del = new Delete(row);
510    del.addColumn(cf, column, version);
511    ht.delete(del);
512  }

514  /**

518  private void deleteAllVersionsBefore(Table ht, byte[] cf, int rowIdx,
519      int colIdx, long version)
520  throws IOException {
521    byte row[] = Bytes.toBytes("row:" + rowIdx);
522    byte column[] = Bytes.toBytes("column:" + colIdx);

524    del.addColumns(cf, column, version);
525    ht.delete(del);
526  }
