065  public static void blockUntilSplitFinished(HBaseTestingUtility util, TableName tableName,
066      int expectedRegionSize) throws Exception {
067    for (int i = 0; i < 100; i++) {
068      List<HRegionInfo> hRegionInfoList = util.getAdmin().getTableRegions(tableName);
069      if (hRegionInfoList.size() >= expectedRegionSize) {
070        break;
071      }
072      Thread.sleep(1000);
073    }
074  }

099  public static void createTableAndSnapshot(HBaseTestingUtility util, TableName tableName,
100      String snapshotName, int numRegions)
101      throws Exception {
102    try {
103      util.deleteTable(tableName);
104    } catch(Exception ex) {
105      // ignore
106    }
107
108    if (numRegions > 1) {
109      util.createTable(tableName, FAMILIES, 1, bbb, yyy, numRegions);
110    } else {
111      util.createTable(tableName, FAMILIES);
112    }
113    Admin admin = util.getAdmin();
114
115    // put some stuff in the table
116    Table table = util.getConnection().getTable(tableName);
117    util.loadTable(table, FAMILIES);
118
119    Path rootDir = FSUtils.getRootDir(util.getConfiguration());
120    FileSystem fs = rootDir.getFileSystem(util.getConfiguration());
121
122    SnapshotTestingUtils.createSnapshotAndValidate(admin, tableName,
123        Arrays.asList(FAMILIES), null, snapshotName, rootDir, fs, true);
124
125    // load different values
126    byte[] value = Bytes.toBytes("after_snapshot_value");
127    util.loadTable(table, FAMILIES, value);
128
129    // cause flush to create new files in the region
130    admin.flush(tableName);
131    table.close();
132  }

134  @Test
135  public void testNoDuplicateResultsWhenSplitting() throws Exception {
136    setupCluster();
137    TableName tableName = TableName.valueOf("testNoDuplicateResultsWhenSplitting");
138    String snapshotName = "testSnapshotBug";
139    try {
140      if (UTIL.getAdmin().tableExists(tableName)) {
141        UTIL.deleteTable(tableName);
142      }
143
144      UTIL.createTable(tableName, FAMILIES);
145      Admin admin = UTIL.getAdmin();
146
147      // put some stuff in the table
148      Table table = UTIL.getConnection().getTable(tableName);
149      UTIL.loadTable(table, FAMILIES);
150
151      // split to 2 regions
152      admin.split(tableName, Bytes.toBytes("eee"));
153      blockUntilSplitFinished(UTIL, tableName, 2);
154
155      Path rootDir = FSUtils.getRootDir(UTIL.getConfiguration());
156      FileSystem fs = rootDir.getFileSystem(UTIL.getConfiguration());
157
158      SnapshotTestingUtils.createSnapshotAndValidate(admin, tableName,
159        Arrays.asList(FAMILIES), null, snapshotName, rootDir, fs, true);
160
161      // load different values
162      byte[] value = Bytes.toBytes("after_snapshot_value");
163      UTIL.loadTable(table, FAMILIES, value);
164
165      // cause flush to create new files in the region
166      admin.flush(tableName);
167      table.close();
168
169      Path restoreDir = UTIL.getDataTestDirOnTestFS(snapshotName);
170      Scan scan = new Scan().withStartRow(bbb).withStopRow(yyy); // limit the scan
171
172      TableSnapshotScanner scanner =
173          new TableSnapshotScanner(UTIL.getConfiguration(), restoreDir, snapshotName, scan);
174
175      verifyScanner(scanner, bbb, yyy);
176      scanner.close();
177    } finally {
178      UTIL.getAdmin().deleteSnapshot(snapshotName);
179      UTIL.deleteTable(tableName);
180      tearDownCluster();
181    }
182  }

199  @Test
200  public void testScannerWithRestoreScanner() throws Exception {
201    setupCluster();
202    TableName tableName = TableName.valueOf("testScanner");
203    String snapshotName = "testScannerWithRestoreScanner";
204    try {
205      createTableAndSnapshot(UTIL, tableName, snapshotName, 50);
206      Path restoreDir = UTIL.getDataTestDirOnTestFS(snapshotName);
207      Scan scan = new Scan(bbb, yyy); // limit the scan
208
209      Configuration conf = UTIL.getConfiguration();
210      Path rootDir = FSUtils.getRootDir(conf);
211
212      TableSnapshotScanner scanner0 =
213          new TableSnapshotScanner(conf, restoreDir, snapshotName, scan);
214      verifyScanner(scanner0, bbb, yyy);
215      scanner0.close();
216
217      // restore snapshot.
218      RestoreSnapshotHelper.copySnapshotForScanner(conf, fs, rootDir, restoreDir, snapshotName);
219
220      // scan the snapshot without restoring snapshot
221      TableSnapshotScanner scanner =
222          new TableSnapshotScanner(conf, rootDir, restoreDir, snapshotName, scan, true);
223      verifyScanner(scanner, bbb, yyy);
224      scanner.close();
225
226      // check whether the snapshot has been deleted by the close of scanner.
227      scanner = new TableSnapshotScanner(conf, rootDir, restoreDir, snapshotName, scan, true);
228      verifyScanner(scanner, bbb, yyy);
229      scanner.close();
230
231      // restore snapshot again.
232      RestoreSnapshotHelper.copySnapshotForScanner(conf, fs, rootDir, restoreDir, snapshotName);
233
234      // check whether the snapshot has been deleted by the close of scanner.
235      scanner = new TableSnapshotScanner(conf, rootDir, restoreDir, snapshotName, scan, true);
236      verifyScanner(scanner, bbb, yyy);
237      scanner.close();
238    } finally {
239      UTIL.getAdmin().deleteSnapshot(snapshotName);
240      UTIL.deleteTable(tableName);
241      tearDownCluster();
242    }
243  }

245  private void testScanner(HBaseTestingUtility util, String snapshotName, int numRegions,
246      boolean shutdownCluster) throws Exception {
247    setupCluster();
248    TableName tableName = TableName.valueOf("testScanner");
249    try {
250      createTableAndSnapshot(util, tableName, snapshotName, numRegions);
251
252      if (shutdownCluster) {
253        util.shutdownMiniHBaseCluster();
254      }
255
256      Path restoreDir = util.getDataTestDirOnTestFS(snapshotName);
257      Scan scan = new Scan(bbb, yyy); // limit the scan
258
259      TableSnapshotScanner scanner = new TableSnapshotScanner(UTIL.getConfiguration(), restoreDir,
260        snapshotName, scan);
261
262      verifyScanner(scanner, bbb, yyy);
263      scanner.close();
264    } finally {
265      if (!shutdownCluster) {
266        util.getAdmin().deleteSnapshot(snapshotName);
267        util.deleteTable(tableName);
268        tearDownCluster();
269      }
270    }
271  }

273  private void verifyScanner(ResultScanner scanner, byte[] startRow, byte[] stopRow)
274      throws IOException, InterruptedException {
275
276    HBaseTestingUtility.SeenRowTracker rowTracker =
277        new HBaseTestingUtility.SeenRowTracker(startRow, stopRow);
278
279    while (true) {
280      Result result = scanner.next();
281      if (result == null) {
282        break;
283      }
284      verifyRow(result);
285      rowTracker.addRow(result.getRow());
286    }
287
288    // validate all rows are seen
289    rowTracker.validate();
290  }
291
292  private static void verifyRow(Result result) throws IOException {
293    byte[] row = result.getRow();
294    CellScanner scanner = result.cellScanner();
295    while (scanner.advance()) {
296      Cell cell = scanner.current();
297
298      //assert that all Cells in the Result have the same key
299     Assert.assertEquals(0, Bytes.compareTo(row, 0, row.length,
300         cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()));
301    }
302
303    for (int j = 0; j < FAMILIES.length; j++) {
304      byte[] actual = result.getValue(FAMILIES[j], FAMILIES[j]);
305      Assert.assertArrayEquals("Row in snapshot does not match, expected:" + Bytes.toString(row)
306          + " ,actual:" + Bytes.toString(actual), row, actual);
307    }
308  }
