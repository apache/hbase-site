045@Category({LargeTests.class, ClientTests.class})
046public class TestTableSnapshotScanner {
047
048  private static final Log LOG = LogFactory.getLog(TestTableSnapshotInputFormat.class);
049  private final HBaseTestingUtility UTIL = new HBaseTestingUtility();
050  private static final int NUM_REGION_SERVERS = 2;
051  private static final byte[][] FAMILIES = {Bytes.toBytes("f1"), Bytes.toBytes("f2")};
052  public static byte[] bbb = Bytes.toBytes("bbb");
053  public static byte[] yyy = Bytes.toBytes("yyy");
054
055  private FileSystem fs;
056  private Path rootDir;
057
058  public void setupCluster() throws Exception {
059    setupConf(UTIL.getConfiguration());
060    UTIL.startMiniCluster(NUM_REGION_SERVERS, true);
061    rootDir = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getRootDir();
062    fs = rootDir.getFileSystem(UTIL.getConfiguration());
063  }
064
065  public void tearDownCluster() throws Exception {
066    UTIL.shutdownMiniCluster();
067  }
068
069  private static void setupConf(Configuration conf) {
070    // Enable snapshot
071    conf.setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
072  }
073
074  @After
075  public void tearDown() throws Exception {
076  }
077
078  public static void createTableAndSnapshot(HBaseTestingUtility util, TableName tableName,
079      String snapshotName, int numRegions)
080      throws Exception {
081    try {
082      util.deleteTable(tableName);
083    } catch(Exception ex) {
084      // ignore
085    }
086
087    if (numRegions > 1) {
088      util.createTable(tableName, FAMILIES, 1, bbb, yyy, numRegions);
089    } else {
090      util.createTable(tableName, FAMILIES);
091    }
092    Admin admin = util.getHBaseAdmin();
093
094    // put some stuff in the table
095    Table table = util.getConnection().getTable(tableName);
096    util.loadTable(table, FAMILIES);
097
098    Path rootDir = FSUtils.getRootDir(util.getConfiguration());
099    FileSystem fs = rootDir.getFileSystem(util.getConfiguration());
100
101    SnapshotTestingUtils.createSnapshotAndValidate(admin, tableName,
102        Arrays.asList(FAMILIES), null, snapshotName, rootDir, fs, true);
103
104    // load different values
105    byte[] value = Bytes.toBytes("after_snapshot_value");
106    util.loadTable(table, FAMILIES, value);
107
108    // cause flush to create new files in the region
109    admin.flush(tableName);
110    table.close();
111  }

113  @Test
114  public void testWithSingleRegion() throws Exception {
115    testScanner(UTIL, "testWithSingleRegion", 1, false);
116  }
117
118  @Test
119  public void testWithMultiRegion() throws Exception {
120    testScanner(UTIL, "testWithMultiRegion", 10, false);
121  }
122
123  @Test
124  public void testWithOfflineHBaseMultiRegion() throws Exception {
125    testScanner(UTIL, "testWithMultiRegion", 20, true);
126  }
127
128  private void testScanner(HBaseTestingUtility util, String snapshotName, int numRegions,
129      boolean shutdownCluster) throws Exception {
130    setupCluster();
131    TableName tableName = TableName.valueOf("testScanner");
132    try {
133      createTableAndSnapshot(util, tableName, snapshotName, numRegions);
134
135      if (shutdownCluster) {
136        util.shutdownMiniHBaseCluster();
137      }
138
139      Path restoreDir = util.getDataTestDirOnTestFS(snapshotName);
140      Scan scan = new Scan(bbb, yyy); // limit the scan
141
142      TableSnapshotScanner scanner = new TableSnapshotScanner(UTIL.getConfiguration(), restoreDir,
143        snapshotName, scan);
144
145      verifyScanner(scanner, bbb, yyy);
146      scanner.close();
147    } finally {
148      if (!shutdownCluster) {
149        util.getHBaseAdmin().deleteSnapshot(snapshotName);
150        util.deleteTable(tableName);
151        tearDownCluster();
152      }
153    }
154  }
155
156  private void verifyScanner(ResultScanner scanner, byte[] startRow, byte[] stopRow)
157      throws IOException, InterruptedException {
158
159    HBaseTestingUtility.SeenRowTracker rowTracker =
160        new HBaseTestingUtility.SeenRowTracker(startRow, stopRow);
161
162    while (true) {
163      Result result = scanner.next();
164      if (result == null) {
165        break;
166      }
167      verifyRow(result);
168      rowTracker.addRow(result.getRow());
169    }
170
171    // validate all rows are seen
172    rowTracker.validate();
173  }
174
175  private static void verifyRow(Result result) throws IOException {
176    byte[] row = result.getRow();
177    CellScanner scanner = result.cellScanner();
178    while (scanner.advance()) {
179      Cell cell = scanner.current();
180
181      //assert that all Cells in the Result have the same key
182     Assert.assertEquals(0, Bytes.compareTo(row, 0, row.length,
183         cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()));
184    }
185
186    for (int j = 0; j < FAMILIES.length; j++) {
187      byte[] actual = result.getValue(FAMILIES[j], FAMILIES[j]);
188      Assert.assertArrayEquals("Row in snapshot does not match, expected:" + Bytes.toString(row)
189          + " ,actual:" + Bytes.toString(actual), row, actual);
190    }
191  }
