055@Category({MediumTests.class, ClientTests.class})
056public class TestClientTimeouts {
057  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
058  protected static int SLAVES = 1;
059
060 /**
061   * @throws java.lang.Exception
062   */
063  @BeforeClass
064  public static void setUpBeforeClass() throws Exception {
065    TEST_UTIL.startMiniCluster(SLAVES);
066    // Set the custom RPC client with random timeouts as the client
067    TEST_UTIL.getConfiguration().set(
068        RpcClientFactory.CUSTOM_RPC_CLIENT_IMPL_CONF_KEY,
069        RandomTimeoutRpcClient.class.getName());
070  }
071
072  /**
073   * @throws java.lang.Exception
074   */
075  @AfterClass
076  public static void tearDownAfterClass() throws Exception {
077    TEST_UTIL.shutdownMiniCluster();
078  }
079
080  /**
081   * Test that a client that fails an RPC to the master retries properly and
082   * doesn't throw any unexpected exceptions.
083   * @throws Exception
084   */
085  @Test
086  public void testAdminTimeout() throws Exception {
087    boolean lastFailed = false;
088    int initialInvocations = RandomTimeoutBlockingRpcChannel.invokations.get();
089    RandomTimeoutRpcClient rpcClient = (RandomTimeoutRpcClient) RpcClientFactory
090        .createClient(TEST_UTIL.getConfiguration(), TEST_UTIL.getClusterKey());
091
092    try {
093      for (int i = 0; i < 5 || (lastFailed && i < 100); ++i) {
094        lastFailed = false;
095        // Ensure the HBaseAdmin uses a new connection by changing Configuration.
096        Configuration conf = HBaseConfiguration.create(TEST_UTIL.getConfiguration());
097        conf.set(HConstants.HBASE_CLIENT_INSTANCE_ID, String.valueOf(-1));
098        Admin admin = null;
099        Connection connection = null;
100        try {
101          connection = ConnectionFactory.createConnection(conf);
102          admin = connection.getAdmin();
103          // run some admin commands
104          HBaseAdmin.available(conf);
105          admin.setBalancerRunning(false, false);
106        } catch (ZooKeeperConnectionException ex) {
107          // Since we are randomly throwing SocketTimeoutExceptions, it is possible to get
108          // a ZooKeeperConnectionException.  It's a bug if we get other exceptions.
109          lastFailed = true;
110        } finally {
111          if(admin != null) {
112            admin.close();
113            if (admin.getConnection().isClosed()) {
114              rpcClient = (RandomTimeoutRpcClient) RpcClientFactory
115                  .createClient(TEST_UTIL.getConfiguration(), TEST_UTIL.getClusterKey());
116            }
117          }
118          if(connection != null) {
119            connection.close();
120          }
121        }
122      }
123      // Ensure the RandomTimeoutRpcEngine is actually being used.
124      assertFalse(lastFailed);
125      assertTrue(RandomTimeoutBlockingRpcChannel.invokations.get() > initialInvocations);
126    } finally {
127      rpcClient.close();
128    }
129  }
130
131  /**
132   * Rpc Channel implementation with RandomTimeoutBlockingRpcChannel
133   */
134  public static class RandomTimeoutRpcClient extends BlockingRpcClient {
135    public RandomTimeoutRpcClient(Configuration conf, String clusterId, SocketAddress localAddr,
136        MetricsConnection metrics) {
137      super(conf, clusterId, localAddr, metrics);
138    }
139
140    // Return my own instance, one that does random timeouts
141    @Override
142    public BlockingRpcChannel createBlockingRpcChannel(ServerName sn,
143        User ticket, int rpcTimeout) throws UnknownHostException {
144      return new RandomTimeoutBlockingRpcChannel(this, sn, ticket, rpcTimeout);
145    }
146  }
147
148  /**
149   * Blocking rpc channel that goes via hbase rpc.
150   */
151  static class RandomTimeoutBlockingRpcChannel
152      extends AbstractRpcClient.BlockingRpcChannelImplementation {
153    private static final Random RANDOM = new Random(System.currentTimeMillis());
154    public static final double CHANCE_OF_TIMEOUT = 0.3;
155    private static AtomicInteger invokations = new AtomicInteger();
156
157    RandomTimeoutBlockingRpcChannel(final BlockingRpcClient rpcClient, final ServerName sn,
158        final User ticket, final int rpcTimeout) {
159      super(rpcClient, new InetSocketAddress(sn.getHostname(), sn.getPort()), ticket, rpcTimeout);
160    }
161
162    @Override
163    public Message callBlockingMethod(MethodDescriptor md,
164        RpcController controller, Message param, Message returnType)
165        throws ServiceException {
166      invokations.getAndIncrement();
167      if (RANDOM.nextFloat() < CHANCE_OF_TIMEOUT) {
168        // throw a ServiceException, becuase that is the only exception type that
169        // {@link ProtobufRpcEngine} throws.  If this RpcEngine is used with a different
170        // "actual" type, this may not properly mimic the underlying RpcEngine.
171        throw new ServiceException(new SocketTimeoutException("fake timeout"));
172      }
173      return super.callBlockingMethod(md, controller, param, returnType);
174    }
175  }
