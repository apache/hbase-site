186  @Test
187  public void testMaxResultSizeIsSetToDefault() throws Exception {
188    final TableName tableName = TableName.valueOf(name.getMethodName());
189    Table ht = TEST_UTIL.createTable(tableName, FAMILY);
190
191    // The max result size we expect the scan to use by default.
192    long expectedMaxResultSize =
193        TEST_UTIL.getConfiguration().getLong(HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY,
194          HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE);
195
196    int numRows = 5;
197    byte[][] ROWS = HTestConst.makeNAscii(ROW, numRows);
198
199    int numQualifiers = 10;
200    byte[][] QUALIFIERS = HTestConst.makeNAscii(QUALIFIER, numQualifiers);
201
202    // Specify the cell size such that a single row will be larger than the default
203    // value of maxResultSize. This means that Scan RPCs should return at most a single
204    // result back to the client.
205    int cellSize = (int) (expectedMaxResultSize / (numQualifiers - 1));
206    byte[] cellValue = Bytes.createMaxByteArray(cellSize);
207
208    Put put;
209    List<Put> puts = new ArrayList<>();
210    for (int row = 0; row < ROWS.length; row++) {
211      put = new Put(ROWS[row]);
212      for (int qual = 0; qual < QUALIFIERS.length; qual++) {
213        KeyValue kv = new KeyValue(ROWS[row], FAMILY, QUALIFIERS[qual], cellValue);
214        put.add(kv);
215      }
216      puts.add(put);
217    }
218    ht.put(puts);
219
220    // Create a scan with the default configuration.
221    Scan scan = new Scan();
222
223    ResultScanner scanner = ht.getScanner(scan);
224    assertTrue(scanner instanceof ClientScanner);
225    ClientScanner clientScanner = (ClientScanner) scanner;
226
227    // Call next to issue a single RPC to the server
228    scanner.next();
229
230    // The scanner should have, at most, a single result in its cache. If there more results exists
231    // in the cache it means that more than the expected max result size was fetched.
232    assertTrue("The cache contains: " + clientScanner.getCacheSize() + " results",
233      clientScanner.getCacheSize() <= 1);
234  }

269   * Run through a variety of test configurations with a small scan
270   * @param table
271   * @param reversed
272   * @param rows
273   * @param columns
274   * @throws Exception
275   */
276  private void testSmallScan(Table table, boolean reversed, int rows, int columns) throws Exception {
277    Scan baseScan = new Scan();
278    baseScan.setReversed(reversed);
279    baseScan.setSmall(true);
280
281    Scan scan = new Scan(baseScan);
282    verifyExpectedCounts(table, scan, rows, columns);
283
284    scan = new Scan(baseScan);
285    scan.setMaxResultSize(1);
286    verifyExpectedCounts(table, scan, rows, columns);
287
288    scan = new Scan(baseScan);
289    scan.setMaxResultSize(1);
290    scan.setCaching(Integer.MAX_VALUE);
291    verifyExpectedCounts(table, scan, rows, columns);
292  }
293
294  private void verifyExpectedCounts(Table table, Scan scan, int expectedRowCount,
295      int expectedCellCount) throws Exception {
296    ResultScanner scanner = table.getScanner(scan);
297
298    int rowCount = 0;
299    int cellCount = 0;
300    Result r = null;
301    while ((r = scanner.next()) != null) {
302      rowCount++;
303      cellCount += r.rawCells().length;
304    }
305
306    assertTrue("Expected row count: " + expectedRowCount + " Actual row count: " + rowCount,
307        expectedRowCount == rowCount);
308    assertTrue("Expected cell count: " + expectedCellCount + " Actual cell count: " + cellCount,
309        expectedCellCount == cellCount);
310    scanner.close();
311  }

612      loc = locator.getRegionLocation(ROW);
613    }
614    HRegionInfo hri = loc.getRegionInfo();
615    MiniHBaseCluster cluster = TEST_UTIL.getMiniHBaseCluster();
616    byte[] regionName = hri.getRegionName();
617    int i = cluster.getServerWith(regionName);
618    HRegionServer rs = cluster.getRegionServer(i);
619    LOG.info("Unassigning " + hri);
620    TEST_UTIL.getAdmin().unassign(hri.getRegionName(), true);
621    long startTime = EnvironmentEdgeManager.currentTime();
622    long timeOut = 10000;
623    boolean offline = false;
624    while (true) {
625      if (rs.getOnlineRegion(regionName) == null) {
626        offline = true;
627        break;
628      }
629      assertTrue("Timed out in closing the testing region",
630        EnvironmentEdgeManager.currentTime() < startTime + timeOut);
631    }
632    assertTrue(offline);
633    LOG.info("Assigning " + hri);
634    TEST_UTIL.getAdmin().assign(hri.getRegionName());
635    startTime = EnvironmentEdgeManager.currentTime();
636    while (true) {
637      rs = cluster.getRegionServer(cluster.getServerWith(regionName));
638      if (rs != null && rs.getOnlineRegion(regionName) != null) {
639        offline = false;
640        break;
641      }
642      assertTrue("Timed out in open the testing region",
643        EnvironmentEdgeManager.currentTime() < startTime + timeOut);
644    }
645    assertFalse(offline);
646
647    // c0:0, c1:1
648    kvListExp = new ArrayList<>();
649    kvListExp.add(new KeyValue(ROW, FAMILY, QUALIFIERS[0], 0, VALUE));
650    kvListExp.add(new KeyValue(ROW, FAMILY, QUALIFIERS[1], 1, VALUE));
651    result = scanner.next();
652    verifyResult(result, kvListExp, toLog, "Testing scan on re-opened region");
653  }

690  private void testAsyncScanner(TableName table, int rowNumber, int familyNumber,
691      int qualifierNumber, int caching, Consumer<Boolean> listener) throws Exception {
692    assert rowNumber > 0;
693    assert familyNumber > 0;
694    assert qualifierNumber > 0;
695    byte[] row = Bytes.toBytes("r");
696    byte[] family = Bytes.toBytes("f");
697    byte[] qualifier = Bytes.toBytes("q");
698    byte[][] rows = makeNAsciiWithZeroPrefix(row, rowNumber);
699    byte[][] families = makeNAsciiWithZeroPrefix(family, familyNumber);
700    byte[][] qualifiers = makeNAsciiWithZeroPrefix(qualifier, qualifierNumber);
701
702    Table ht = TEST_UTIL.createTable(table, families);
703
704    boolean toLog = true;
705    List<Cell> kvListExp = new ArrayList<>();
706
707    List<Put> puts = new ArrayList<>();
708    for (byte[] r : rows) {
709      Put put = new Put(r);
710      for (byte[] f : families) {
711        for (byte[] q : qualifiers) {
712          KeyValue kv = new KeyValue(r, f, q, 1, VALUE);
713          put.add(kv);
714          kvListExp.add(kv);
715        }
716      }
717      puts.add(put);
718      if (puts.size() > 1000) {
719        ht.put(puts);
720        puts.clear();
721      }
722    }
723    if (!puts.isEmpty()) {
724      ht.put(puts);
725      puts.clear();
726    }
727
728    Scan scan = new Scan();
729    scan.setAsyncPrefetch(true);
730    if (caching > 0) {
731      scan.setCaching(caching);
732    }
733    try (ResultScanner scanner = ht.getScanner(scan)) {
734      assertTrue("Not instance of async scanner",scanner instanceof ClientAsyncPrefetchScanner);
735      ((ClientAsyncPrefetchScanner) scanner).setPrefetchListener(listener);
736      List<Cell> kvListScan = new ArrayList<>();
737      Result result;
738      boolean first = true;
739      int actualRows = 0;
740      while ((result = scanner.next()) != null) {
741        ++actualRows;
742        // waiting for cache. see HBASE-17376
743        if (first) {
744          TimeUnit.SECONDS.sleep(1);
745          first = false;
746        }
747        for (Cell kv : result.listCells()) {
748          kvListScan.add(kv);
749        }
750      }
751      assertEquals(rowNumber, actualRows);
752      // These cells may have different rows but it is ok. The Result#getRow
753      // isn't used in the verifyResult()
754      result = Result.create(kvListScan);
755      verifyResult(result, kvListExp, toLog, "Testing async scan");
756    }
757
758    TEST_UTIL.deleteTable(table);
759  }

761  private static byte[][] makeNAsciiWithZeroPrefix(byte[] base, int n) {
762    int maxLength = Integer.toString(n).length();
763    byte [][] ret = new byte[n][];
764    for (int i = 0; i < n; i++) {
765      int length = Integer.toString(i).length();
766      StringBuilder buf = new StringBuilder(Integer.toString(i));
767      IntStream.range(0, maxLength - length).forEach(v -> buf.insert(0, "0"));
768      byte[] tail = Bytes.toBytes(buf.toString());
769      ret[i] = Bytes.add(base, tail);
770    }
771    return ret;
772  }
773
774  static void verifyResult(Result result, List<Cell> expKvList, boolean toLog,
775      String msg) {
776
777    LOG.info(msg);
778    LOG.info("Expected count: " + expKvList.size());
779    LOG.info("Actual count: " + result.size());
780    if (expKvList.isEmpty())
781      return;
782
783    int i = 0;
784    for (Cell kv : result.rawCells()) {
785      if (i >= expKvList.size()) {
786        break;  // we will check the size later
787      }
788
789      Cell kvExp = expKvList.get(i++);
790      if (toLog) {
791        LOG.info("get kv is: " + kv.toString());
792        LOG.info("exp kv is: " + kvExp.toString());
793      }
794      assertTrue("Not equal", kvExp.equals(kv));
795    }
796
797    assertEquals(expKvList.size(), result.size());
798  }
