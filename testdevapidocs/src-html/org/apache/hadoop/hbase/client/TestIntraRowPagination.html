036/**
037 * Test scan/get offset and limit settings within one row through HRegion API.
038 */
039@Category({SmallTests.class, ClientTests.class})
040public class TestIntraRowPagination {
041
042  @ClassRule
043  public static final HBaseClassTestRule CLASS_RULE =
044      HBaseClassTestRule.forClass(TestIntraRowPagination.class);
045
046  private static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
047
048  /**
049   * Test from client side for scan with maxResultPerCF set
050   */
051  @Test
052  public void testScanLimitAndOffset() throws Exception {
053    //byte [] TABLE = HTestConst.DEFAULT_TABLE_BYTES;
054    byte [][] ROWS = HTestConst.makeNAscii(HTestConst.DEFAULT_ROW_BYTES, 2);
055    byte [][] FAMILIES = HTestConst.makeNAscii(HTestConst.DEFAULT_CF_BYTES, 3);
056    byte [][] QUALIFIERS = HTestConst.makeNAscii(HTestConst.DEFAULT_QUALIFIER_BYTES, 10);
057
058    TableDescriptorBuilder.ModifyableTableDescriptor tableDescriptor =
059      new TableDescriptorBuilder.ModifyableTableDescriptor(
060        TableName.valueOf(HTestConst.DEFAULT_TABLE_BYTES));
061
062    RegionInfo info = RegionInfoBuilder.newBuilder(HTestConst.DEFAULT_TABLE).build();
063    for (byte[] family : FAMILIES) {
064      ColumnFamilyDescriptor familyDescriptor =
065        new ColumnFamilyDescriptorBuilder.ModifyableColumnFamilyDescriptor(family);
066      tableDescriptor.setColumnFamily(familyDescriptor);
067    }
068    HRegion region = HBaseTestingUtility.createRegionAndWAL(info, TEST_UTIL.getDataTestDir(),
069        TEST_UTIL.getConfiguration(), tableDescriptor);
070    try {
071      Put put;
072      Scan scan;
073      Result result;
074      boolean toLog = true;
075
076      List<Cell> kvListExp = new ArrayList<>();
077
078      int storeOffset = 1;
079      int storeLimit = 3;
080      for (int r = 0; r < ROWS.length; r++) {
081        put = new Put(ROWS[r]);
082        for (int c = 0; c < FAMILIES.length; c++) {
083          for (int q = 0; q < QUALIFIERS.length; q++) {
084            KeyValue kv = new KeyValue(ROWS[r], FAMILIES[c], QUALIFIERS[q], 1,
085                HTestConst.DEFAULT_VALUE_BYTES);
086            put.add(kv);
087            if (storeOffset <= q && q < storeOffset + storeLimit) {
088              kvListExp.add(kv);
089            }
090          }
091        }
092        region.put(put);
093      }
094
095      scan = new Scan();
096      scan.setRowOffsetPerColumnFamily(storeOffset);
097      scan.setMaxResultsPerColumnFamily(storeLimit);
098      RegionScanner scanner = region.getScanner(scan);
099      List<Cell> kvListScan = new ArrayList<>();
100      List<Cell> results = new ArrayList<>();
101      while (scanner.next(results) || !results.isEmpty()) {
102        kvListScan.addAll(results);
103        results.clear();
104      }
105      result = Result.create(kvListScan);
106      TestScannersFromClientSide.verifyResult(result, kvListExp, toLog,
107          "Testing scan with storeOffset and storeLimit");
108    } finally {
109      HBaseTestingUtility.closeRegionAndWAL(region);
110    }
111  }
