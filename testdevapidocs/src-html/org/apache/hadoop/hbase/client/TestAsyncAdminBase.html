022import java.util.Arrays;
023import java.util.List;
024import java.util.Optional;
025import java.util.concurrent.ExecutionException;

040import org.junit.Before;
041import org.junit.BeforeClass;
042import org.junit.Rule;
043import org.junit.rules.TestName;
044import org.junit.runner.RunWith;
045import org.junit.runners.Parameterized;
046import org.junit.runners.Parameterized.Parameter;
047import org.junit.runners.Parameterized.Parameters;
048
049/**
050 * Class to test AsyncAdmin.
051 */
052public abstract class TestAsyncAdminBase {
053
054  protected static final Log LOG = LogFactory.getLog(TestAsyncAdminBase.class);
055  protected final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
056  protected static final byte[] FAMILY = Bytes.toBytes("testFamily");
057  protected static final byte[] FAMILY_0 = Bytes.toBytes("cf0");
058  protected static final byte[] FAMILY_1 = Bytes.toBytes("cf1");
059
060  protected static AsyncConnection ASYNC_CONN;
061  protected AsyncAdmin admin;
062
063  @Parameter
064  public Supplier<AsyncAdmin> getAdmin;
065
066  private static AsyncAdmin getRawAsyncAdmin() {
067    return ASYNC_CONN.getAdmin();
068  }
069
070  private static AsyncAdmin getAsyncAdmin() {
071    return ASYNC_CONN.getAdmin(ForkJoinPool.commonPool());
072  }
073
074  @Parameters
075  public static List<Object[]> params() {
076    return Arrays.asList(new Supplier<?>[] { TestAsyncAdminBase::getRawAsyncAdmin },
077      new Supplier<?>[] { TestAsyncAdminBase::getAsyncAdmin });
078  }
079
080  @Rule
081  public TestName testName = new TestName();
082  protected TableName tableName;
083
084  @BeforeClass
085  public static void setUpBeforeClass() throws Exception {
086    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_RPC_TIMEOUT_KEY, 60000);
087    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT, 120000);
088    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 2);
089    TEST_UTIL.getConfiguration().setInt(START_LOG_ERRORS_AFTER_COUNT_KEY, 0);
090    TEST_UTIL.startMiniCluster(2);
091    ASYNC_CONN = ConnectionFactory.createAsyncConnection(TEST_UTIL.getConfiguration()).get();
092  }
093
094  @AfterClass
095  public static void tearDownAfterClass() throws Exception {
096    IOUtils.closeQuietly(ASYNC_CONN);
097    TEST_UTIL.shutdownMiniCluster();
098  }
099
100  @Before
101  public void setUp() throws Exception {
102    admin = getAdmin.get();
103    String methodName = testName.getMethodName();
104    tableName = TableName.valueOf(methodName.substring(0, methodName.length() - 3));
105  }
106
107  @After
108  public void tearDown() throws Exception {
109    admin.listTableNames(Optional.of(Pattern.compile(tableName.getNameAsString() + ".*")), false)
110        .whenCompleteAsync((tables, err) -> {
111          if (tables != null) {
112            tables.forEach(table -> {
113              try {
114                admin.disableTable(table).join();
115              } catch (Exception e) {
116                LOG.debug("Table: " + tableName + " already disabled, so just deleting it.");
117              }
118              admin.deleteTable(table).join();
119            });
120          }
121        }, ForkJoinPool.commonPool()).join();
122  }
123
124  protected void createTableWithDefaultConf(TableName tableName) {
125    createTableWithDefaultConf(tableName, Optional.empty());
126  }
127
128  protected void createTableWithDefaultConf(TableName tableName, Optional<byte[][]> splitKeys) {
129    createTableWithDefaultConf(tableName, splitKeys, FAMILY);
130  }
131
132  protected void createTableWithDefaultConf(TableName tableName, Optional<byte[][]> splitKeys,
133      byte[]... families) {
134    TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(tableName);
135    for (byte[] family : families) {
136      builder.addColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(family).build());
137    }
138    admin.createTable(builder.build(), splitKeys).join();
139  }
140}
