118  public static class SuspendProcedure extends
119      ProcedureTestingUtility.NoopProcedure<MasterProcedureEnv> implements TableProcedureInterface {
120    public SuspendProcedure() {
121      super();
122    }
123
124    @SuppressWarnings({ "rawtypes", "unchecked" })
125    @Override
126    protected Procedure[] execute(final MasterProcedureEnv env) throws ProcedureSuspendedException {
127      // Always suspend the procedure
128      throw new ProcedureSuspendedException();
129    }
130
131    @Override
132    public TableName getTableName() {
133      return TABLE_NAME;
134    }
135
136    @Override
137    public TableOperationType getTableOperationType() {
138      return TableOperationType.READ;
139    }
140  }

142  @Test
143  public void testBypassProcedure() throws Exception {
144    // SuspendProcedure
145    final SuspendProcedure proc = new SuspendProcedure();
146    long procId = procExec.submitProcedure(proc);
147    Thread.sleep(500);
148
149    // bypass the procedure
150    List<Long> pids = Arrays.<Long> asList(procId);
151    List<Boolean> results = getHbck().bypassProcedure(pids, 30000, false, false);
152    assertTrue("Failed to by pass procedure!", results.get(0));
153    TEST_UTIL.waitFor(5000, () -> proc.isSuccess() && proc.isBypass());
154    LOG.info("{} finished", proc);
155  }

157  @Test
158  public void testSetTableStateInMeta() throws Exception {
159    Hbck hbck = getHbck();
160    // set table state to DISABLED
161    hbck.setTableStateInMeta(new TableState(TABLE_NAME, TableState.State.DISABLED));
162    // Method {@link Hbck#setTableStateInMeta()} returns previous state, which in this case
163    // will be DISABLED
164    TableState prevState =
165      hbck.setTableStateInMeta(new TableState(TABLE_NAME, TableState.State.ENABLED));
166    assertTrue("Incorrect previous state! expeced=DISABLED, found=" + prevState.getState(),
167      prevState.isDisabled());
168  }

170  @Test
171  public void testAssigns() throws Exception {
172    Hbck hbck = getHbck();
173    try (Admin admin = TEST_UTIL.getConnection().getAdmin()) {
174      List<RegionInfo> regions = admin.getRegions(TABLE_NAME);
175      for (RegionInfo ri : regions) {
176        RegionState rs = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager()
177          .getRegionStates().getRegionState(ri.getEncodedName());
178        LOG.info("RS: {}", rs.toString());
179      }
180      List<Long> pids =
181        hbck.unassigns(regions.stream().map(r -> r.getEncodedName()).collect(Collectors.toList()));
182      waitOnPids(pids);
183      for (RegionInfo ri : regions) {
184        RegionState rs = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager()
185          .getRegionStates().getRegionState(ri.getEncodedName());
186        LOG.info("RS: {}", rs.toString());
187        assertTrue(rs.toString(), rs.isClosed());
188      }
189      pids =
190        hbck.assigns(regions.stream().map(r -> r.getEncodedName()).collect(Collectors.toList()));
191      waitOnPids(pids);
192      for (RegionInfo ri : regions) {
193        RegionState rs = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager()
194          .getRegionStates().getRegionState(ri.getEncodedName());
195        LOG.info("RS: {}", rs.toString());
196        assertTrue(rs.toString(), rs.isOpened());
197      }
198      // What happens if crappy region list passed?
199      pids = hbck.assigns(
200        Arrays.stream(new String[] { "a", "some rubbish name" }).collect(Collectors.toList()));
201      for (long pid : pids) {
202        assertEquals(org.apache.hadoop.hbase.procedure2.Procedure.NO_PROC_ID, pid);
203      }
204    }
205  }

207  @Test
208  public void testScheduleSCP() throws Exception {
209    HRegionServer testRs = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME);
210    TEST_UTIL.loadTable(TEST_UTIL.getConnection().getTable(TABLE_NAME), Bytes.toBytes("family1"),
211      true);
212    ServerName serverName = testRs.getServerName();
213    Hbck hbck = getHbck();
214    List<Long> pids =
215      hbck.scheduleServerCrashProcedure(Arrays.asList(ProtobufUtil.toServerName(serverName)));
216    assertTrue(pids.get(0) > 0);
217    LOG.info("pid is {}", pids.get(0));
218
219    List<Long> newPids =
220      hbck.scheduleServerCrashProcedure(Arrays.asList(ProtobufUtil.toServerName(serverName)));
221    assertTrue(newPids.get(0) < 0);
222    LOG.info("pid is {}", newPids.get(0));
223    waitOnPids(pids);
224  }
225
226  private void waitOnPids(List<Long> pids) {
227    TEST_UTIL.waitFor(60000, () -> pids.stream().allMatch(procExec::isFinished));
228  }
