086  private static class MockClientScanner extends ClientScanner {
087
088    private boolean rpcFinished = false;
089    private boolean rpcFinishedFired = false;
090
091    public MockClientScanner(final Configuration conf, final Scan scan, final TableName tableName,
092        ClusterConnection connection, RpcRetryingCallerFactory rpcFactory,
093        RpcControllerFactory controllerFactory, ExecutorService pool, int primaryOperationTimeout)
094        throws IOException {
095      super(conf, scan, tableName, connection, rpcFactory, controllerFactory, pool,
096          primaryOperationTimeout);
097    }
098
099    @Override
100    protected boolean nextScanner(int nbRows, final boolean done) throws IOException {
101      if (!rpcFinished) {
102        return super.nextScanner(nbRows, done);
103      }
104
105      // Enforce that we don't short-circuit more than once
106      if (rpcFinishedFired) {
107        throw new RuntimeException("Expected nextScanner to only be called once after " +
108            " short-circuit was triggered.");
109      }
110      rpcFinishedFired = true;
111      return false;
112    }
113
114    @Override
115    protected ScannerCallableWithReplicas getScannerCallable(byte [] localStartKey,
116        int nbRows) {
117      scan.setStartRow(localStartKey);
118      ScannerCallable s =
119          new ScannerCallable(getConnection(), getTable(), scan, this.scanMetrics,
120              this.rpcControllerFactory);
121      s.setCaching(nbRows);
122      ScannerCallableWithReplicas sr = new ScannerCallableWithReplicas(getTable(), getConnection(),
123       s, pool, primaryOperationTimeout, scan,
124       getRetries(), scannerTimeout, caching, conf, caller);
125      return sr;
126    }
127
128    public void setRpcFinished(boolean rpcFinished) {
129      this.rpcFinished = rpcFinished;
130    }
131
132    @Override
133    protected void initCache() {
134      initSyncCache();
135    }
136
137    @Override public Result next() throws IOException {
138      return nextWithSyncCache();
139    }
140  }

142  @Test
143  @SuppressWarnings("unchecked")
144  public void testNoResultsHint() throws IOException {
145    final Result[] results = new Result[1];
146    KeyValue kv1 = new KeyValue("row".getBytes(), "cf".getBytes(), "cq".getBytes(), 1,
147        Type.Maximum);
148    results[0] = Result.create(new Cell[] {kv1});
149
150    RpcRetryingCaller<Result[]> caller = Mockito.mock(RpcRetryingCaller.class);
151
152    Mockito.when(rpcFactory.<Result[]> newCaller()).thenReturn(caller);
153    Mockito.when(caller.callWithoutRetries(Mockito.any(RetryingCallable.class),
154      Mockito.anyInt())).thenAnswer(new Answer<Result[]>() {
155        private int count = 0;
156        @Override
157        public Result[] answer(InvocationOnMock invocation) throws Throwable {
158            ScannerCallableWithReplicas callable = invocation.getArgumentAt(0,
159                ScannerCallableWithReplicas.class);
160          switch (count) {
161            case 0: // initialize
162            case 2: // close
163              count++;
164              return null;
165            case 1:
166              count++;
167              callable.setHasMoreResultsContext(false);
168              return results;
169            default:
170              throw new RuntimeException("Expected only 2 invocations");
171          }
172        }
173    });
174
175    // Set a much larger cache and buffer size than we'll provide
176    scan.setCaching(100);
177    scan.setMaxResultSize(1000*1000);
178
179    try (MockClientScanner scanner = new MockClientScanner(conf, scan, TableName.valueOf("table"),
180        clusterConn, rpcFactory, controllerFactory, pool, Integer.MAX_VALUE)) {
181
182      scanner.setRpcFinished(true);
183
184      InOrder inOrder = Mockito.inOrder(caller);
185
186      scanner.loadCache();
187
188      // One more call due to initializeScannerInConstruction()
189      inOrder.verify(caller, Mockito.times(2)).callWithoutRetries(
190          Mockito.any(RetryingCallable.class), Mockito.anyInt());
191
192      assertEquals(1, scanner.cache.size());
193      Result r = scanner.cache.poll();
194      assertNotNull(r);
195      CellScanner cs = r.cellScanner();
196      assertTrue(cs.advance());
197      assertEquals(kv1, cs.current());
198      assertFalse(cs.advance());
199    }
200  }

506  /**
507   * Tests the case where all replicas of a region throw an exception. It should not cause a hang
508   * but the exception should propagate to the client
509   */
510  @Test (timeout = 30000)
511  public void testExceptionsFromReplicasArePropagated() throws IOException {
512    scan.setConsistency(Consistency.TIMELINE);
513
514    // Mock a caller which calls the callable for ScannerCallableWithReplicas,
515    // but throws an exception for the actual scanner calls via callWithRetries.
516    rpcFactory = new MockRpcRetryingCallerFactory(conf);
517    conf.set(RpcRetryingCallerFactory.CUSTOM_CALLER_CONF_KEY,
518      MockRpcRetryingCallerFactory.class.getName());
519
520    // mock 3 replica locations
521    when(clusterConn.locateRegion((TableName)any(), (byte[])any(), anyBoolean(),
522      anyBoolean(), anyInt())).thenReturn(new RegionLocations(null, null, null));
523
524    try (MockClientScanner scanner = new MockClientScanner(conf, scan, TableName.valueOf("table"),
525      clusterConn, rpcFactory, new RpcControllerFactory(conf), pool, Integer.MAX_VALUE)) {
526      Iterator<Result> iter = scanner.iterator();
527      while (iter.hasNext()) {
528        iter.next();
529      }
530      fail("Should have failed with RetriesExhaustedException");
531    } catch (RetriesExhaustedException expected) {
532
533    }
534  }

536  public static class MockRpcRetryingCallerFactory extends RpcRetryingCallerFactory {
537
538    public MockRpcRetryingCallerFactory(Configuration conf) {
539      super(conf);
540    }
541
542    @Override
543    public <T> RpcRetryingCaller<T> newCaller() {
544      return new RpcRetryingCaller<T>() {
545        @Override
546        public void cancel() {
547        }
548        @Override
549        public T callWithRetries(RetryingCallable<T> callable, int callTimeout)
550            throws IOException, RuntimeException {
551          throw new IOException("Scanner exception");
552        }
553
554        @Override
555        public T callWithoutRetries(RetryingCallable<T> callable, int callTimeout)
556            throws IOException, RuntimeException {
557          try {
558            return callable.call(callTimeout);
559          } catch (IOException e) {
560            throw e;
561          } catch (Exception e) {
562            throw new RuntimeException(e);
563          }
564        }
565      };
566    }
