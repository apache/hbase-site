045/**
046 * Class to test asynchronous procedure admin operations.
047 */
048@RunWith(Parameterized.class)
049@Category({ LargeTests.class, ClientTests.class })
050public class TestAsyncProcedureAdminApi extends TestAsyncAdminBase {
051
052  @BeforeClass
053  public static void setUpBeforeClass() throws Exception {
054    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_RPC_TIMEOUT_KEY, 60000);
055    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT, 120000);
056    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 2);
057    TEST_UTIL.getConfiguration().setInt(START_LOG_ERRORS_AFTER_COUNT_KEY, 0);
058    TEST_UTIL.getConfiguration().set(ProcedureManagerHost.MASTER_PROCEDURE_CONF_KEY,
059      SimpleMasterProcedureManager.class.getName());
060    TEST_UTIL.getConfiguration().set(ProcedureManagerHost.REGIONSERVER_PROCEDURE_CONF_KEY,
061      SimpleRSProcedureManager.class.getName());
062    TEST_UTIL.getConfiguration().setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
063    TEST_UTIL.startMiniCluster(2);
064    ASYNC_CONN = ConnectionFactory.createAsyncConnection(TEST_UTIL.getConfiguration()).get();
065  }
066
067  @Test
068  public void testExecProcedure() throws Exception {
069    String snapshotString = "offlineTableSnapshot";
070    try {
071      Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("cf"));
072      for (int i = 0; i < 100; i++) {
073        Put put = new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("cf"), null, Bytes.toBytes(i));
074        table.put(put);
075      }
076      // take a snapshot of the enabled table
077      Map<String, String> props = new HashMap<>();
078      props.put("table", tableName.getNameAsString());
079      admin.execProcedure(SnapshotManager.ONLINE_SNAPSHOT_CONTROLLER_DESCRIPTION, snapshotString,
080        props).get();
081      LOG.debug("Snapshot completed.");
082    } finally {
083      admin.deleteSnapshot(snapshotString).join();
084      TEST_UTIL.deleteTable(tableName);
085    }
086  }
087
088  @Test
089  public void testExecProcedureWithRet() throws Exception {
090    byte[] result = admin.execProcedureWithRet(SimpleMasterProcedureManager.SIMPLE_SIGNATURE,
091      "myTest2", new HashMap<>()).get();
092    assertArrayEquals("Incorrect return data from execProcedure",
093      SimpleMasterProcedureManager.SIMPLE_DATA.getBytes(), result);
094  }
095
096  @Test
097  public void listProcedure() throws Exception {
098    String procList = admin.getProcedures().get();
099    assertTrue(procList.startsWith("["));
100  }
101
102  @Test
103  public void isProcedureFinished() throws Exception {
104    boolean failed = false;
105    try {
106      admin.isProcedureFinished("fake-signature", "fake-instance", new HashMap<>()).get();
107    } catch (Exception e) {
108      failed = true;
109    }
110    Assert.assertTrue(failed);
111  }
112
113  @Test
114  public void abortProcedure() throws Exception {
115    Random randomGenerator = new Random();
116    long procId = randomGenerator.nextLong();
117    boolean abortResult = admin.abortProcedure(procId, true).get();
118    assertFalse(abortResult);
119  }
