052@Category(LargeTests.class)
053public class TestCheckAndMutateWithByteBuff {
054  private static final Logger LOG = LoggerFactory.getLogger(TestCheckAndMutateWithByteBuff.class);
055
056  @ClassRule
057  public static final HBaseClassTestRule CLASS_RULE =
058    HBaseClassTestRule.forClass(TestCheckAndMutateWithByteBuff.class);
059
060  @Rule
061  public TestName name = new TestName();
062
063  private static final byte[] CF = Bytes.toBytes("CF");
064  private static final HBaseTestingUtil TEST_UTIL = new HBaseTestingUtil();
065  private static final Configuration conf = TEST_UTIL.getConfiguration();
066  private static Admin admin = null;
067
068  @BeforeClass
069  public static void setupBeforeClass() throws Exception {
070    conf.set(HConstants.REGION_IMPL, TestCheckAndMutateRegion.class.getName());
071    conf.set(ByteBuffAllocator.BYTEBUFF_ALLOCATOR_CLASS,
072      DeallocateRewriteByteBuffAllocator.class.getName());
073    conf.setBoolean(ByteBuffAllocator.ALLOCATOR_POOL_ENABLED_KEY, true);
074    conf.setInt(ByteBuffAllocator.MIN_ALLOCATE_SIZE_KEY, 1);
075    conf.setInt(BlockCacheFactory.BUCKET_CACHE_WRITER_THREADS_KEY, 20);
076    conf.setInt(ByteBuffAllocator.BUFFER_SIZE_KEY, 1024);
077    conf.set(HConstants.BUCKET_CACHE_IOENGINE_KEY, "offheap");
078    conf.setInt(HConstants.BUCKET_CACHE_SIZE_KEY, 64);
079    conf.setInt("hbase.client.retries.number", 1);
080    TEST_UTIL.startMiniCluster();
081    admin = TEST_UTIL.getAdmin();
082  }
083
084  @AfterClass
085  public static void tearDownAfterClass() throws Exception {
086    TEST_UTIL.shutdownMiniCluster();
087  }
088
089  @Test
090  public void testCheckAndMutateWithByteBuff() throws Exception {
091    Table testTable = createTable(TableName.valueOf(name.getMethodName()));
092    byte[] checkRow = Bytes.toBytes("checkRow");
093    byte[] checkQualifier = Bytes.toBytes("cq");
094    byte[] checkValue = Bytes.toBytes("checkValue");
095
096    Put put = new Put(checkRow);
097    put.addColumn(CF, checkQualifier, checkValue);
098    testTable.put(put);
099    admin.flush(testTable.getName());
100
101    assertTrue(testTable.checkAndMutate(checkRow, CF).qualifier(checkQualifier).
102      ifEquals(checkValue)
103      .thenPut(new Put(checkRow).addColumn(CF, Bytes.toBytes("q1"),
104        Bytes.toBytes("testValue"))));
105  }
106
107  private Table createTable(TableName tableName)
108    throws IOException {
109    TableDescriptor td = TableDescriptorBuilder.newBuilder(tableName)
110      .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(CF).setBlocksize(100).build())
111      .build();
112    return TEST_UTIL.createTable(td, null);
113  }
114
115  /**
116   * An override of HRegion to allow sleep after get(), waiting for the release of DBB
117   */
118  public static class TestCheckAndMutateRegion extends HRegion {
119    public TestCheckAndMutateRegion(Path tableDir, WAL log, FileSystem fs, Configuration confParam,
120      RegionInfo info, TableDescriptor htd, RegionServerServices rsServices) {
121      super(tableDir, log, fs, confParam, info, htd, rsServices);
122    }
123
124    public TestCheckAndMutateRegion(HRegionFileSystem fs, WAL wal, Configuration confParam,
125      TableDescriptor htd, RegionServerServices rsServices) {
126      super(fs, wal, confParam, htd, rsServices);
127    }
128
129    @Override
130    public List<Cell> get(Get get, boolean withCoprocessor) throws IOException {
131      List<Cell> cells = super.get(get, withCoprocessor);
132      sleep(600);
133      return cells;
134    }
135  }
