048@Category({ ClientTests.class, MediumTests.class })
049public class TestSeparateClientZKCluster {
050  private static final Logger LOG = LoggerFactory.getLogger(TestSeparateClientZKCluster.class);
051  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
052  private static final File clientZkDir =
053    new File(TEST_UTIL.getDataTestDir("TestSeparateClientZKCluster").toString());
054  private static final int ZK_SESSION_TIMEOUT = 5000;

061  private final byte[] newVal = Bytes.toBytes("v2");

068      HBaseClassTestRule.forClass(TestSeparateClientZKCluster.class);

070  @BeforeClass
071  public static void beforeAllTests() throws Exception {
072    int clientZkPort = 21828;
073    clientZkCluster = new MiniZooKeeperCluster(TEST_UTIL.getConfiguration());
074    clientZkCluster.setDefaultClientPort(clientZkPort);
075    clientZkCluster.startup(clientZkDir);
076    // reduce the retry number and start log counter
077    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 2);
078    TEST_UTIL.getConfiguration().setInt("hbase.client.start.log.errors.counter", -1);
079    TEST_UTIL.getConfiguration().setInt("zookeeper.recovery.retry", 1);
080    // core settings for testing client ZK cluster
081    TEST_UTIL.getConfiguration().set(HConstants.CLIENT_ZOOKEEPER_QUORUM, HConstants.LOCALHOST);
082    TEST_UTIL.getConfiguration().setInt(HConstants.CLIENT_ZOOKEEPER_CLIENT_PORT, clientZkPort);
083    // reduce zk session timeout to easier trigger session expiration
084    TEST_UTIL.getConfiguration().setInt(HConstants.ZK_SESSION_TIMEOUT, ZK_SESSION_TIMEOUT);
085    // Start a cluster with 2 masters and 3 regionservers.
086    StartMiniClusterOption option = StartMiniClusterOption.builder()
087        .numMasters(2).numRegionServers(3).numDataNodes(3).build();
088    TEST_UTIL.startMiniCluster(option);
089  }

091  @AfterClass
092  public static void afterAllTests() throws Exception {
093    TEST_UTIL.shutdownMiniCluster();
094    clientZkCluster.shutdown();
095    FileUtils.deleteDirectory(clientZkDir);
096  }

098  @Test
099  public void testBasicOperation() throws Exception {
100    TableName tn = TableName.valueOf(name.getMethodName());
101    // create table
102    Connection conn = TEST_UTIL.getConnection();
103    try (Admin admin = conn.getAdmin(); Table table = conn.getTable(tn)) {
104      ColumnFamilyDescriptorBuilder cfDescBuilder =
105        ColumnFamilyDescriptorBuilder.newBuilder(family);
106      TableDescriptorBuilder tableDescBuilder =
107        TableDescriptorBuilder.newBuilder(tn).setColumnFamily(cfDescBuilder.build());
108      admin.createTable(tableDescBuilder.build());
109      // test simple get and put
110      Put put = new Put(row);
111      put.addColumn(family, qualifier, value);
112      table.put(put);
113      Get get = new Get(row);
114      Result result = table.get(get);
115      LOG.debug("Result: " + Bytes.toString(result.getValue(family, qualifier)));
116      Assert.assertArrayEquals(value, result.getValue(family, qualifier));
117    }
118  }

120  @Test
121  public void testMasterSwitch() throws Exception {
122    // get an admin instance and issue some request first
123    Connection conn = TEST_UTIL.getConnection();
124    try (Admin admin = conn.getAdmin()) {
125      LOG.debug("Tables: " + admin.listTableDescriptors());
126      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
127      // switch active master
128      HMaster master = cluster.getMaster();
129      master.stopMaster();
130      while (!master.isShutDown()) {
131        Thread.sleep(200);
132      }
133      while (cluster.getMaster() == null || !cluster.getMaster().isInitialized()) {
134        Thread.sleep(200);
135      }
136      // confirm client access still works
137      Assert.assertTrue(admin.balance(false));
138    }
139  }

141  @Test
142  public void testMetaRegionMove() throws Exception {
143    TableName tn = TableName.valueOf(name.getMethodName());
144    // create table
145    Connection conn = TEST_UTIL.getConnection();
146    try (Admin admin = conn.getAdmin();
147        Table table = conn.getTable(tn);
148        RegionLocator locator = conn.getRegionLocator(tn)) {
149      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
150      ColumnFamilyDescriptorBuilder cfDescBuilder =
151        ColumnFamilyDescriptorBuilder.newBuilder(family);
152      TableDescriptorBuilder tableDescBuilder =
153        TableDescriptorBuilder.newBuilder(tn).setColumnFamily(cfDescBuilder.build());
154      admin.createTable(tableDescBuilder.build());
155      // issue some requests to cache the region location
156      Put put = new Put(row);
157      put.addColumn(family, qualifier, value);
158      table.put(put);
159      Get get = new Get(row);
160      Result result = table.get(get);
161      // move meta region and confirm client could detect
162      ServerName destServerName = null;
163      for (RegionServerThread rst : cluster.getLiveRegionServerThreads()) {
164        ServerName name = rst.getRegionServer().getServerName();
165        if (!name.equals(cluster.getServerHoldingMeta())) {
166          destServerName = name;
167          break;
168        }
169      }
170      admin.move(RegionInfoBuilder.FIRST_META_REGIONINFO.getEncodedNameAsBytes(), destServerName);
171      LOG.debug("Finished moving meta");
172      // invalidate client cache
173      RegionInfo region = locator.getRegionLocation(row).getRegion();
174      ServerName currentServer = cluster.getServerHoldingRegion(tn, region.getRegionName());
175      for (RegionServerThread rst : cluster.getLiveRegionServerThreads()) {
176        ServerName name = rst.getRegionServer().getServerName();
177        if (!name.equals(currentServer)) {
178          destServerName = name;
179          break;
180        }
181      }
182      admin.move(region.getEncodedNameAsBytes(), destServerName);
183      LOG.debug("Finished moving user region");
184      put = new Put(row);
185      put.addColumn(family, qualifier, newVal);
186      table.put(put);
187      result = table.get(get);
188      LOG.debug("Result: " + Bytes.toString(result.getValue(family, qualifier)));
189      Assert.assertArrayEquals(newVal, result.getValue(family, qualifier));
190    }
191  }

193  @Test
194  public void testMetaMoveDuringClientZkClusterRestart() throws Exception {
195    TableName tn = TableName.valueOf(name.getMethodName());
196    // create table
197    Connection conn = TEST_UTIL.getConnection();
198    try (Admin admin = conn.getAdmin(); Table table = conn.getTable(tn)) {
199      ColumnFamilyDescriptorBuilder cfDescBuilder =
200        ColumnFamilyDescriptorBuilder.newBuilder(family);
201      TableDescriptorBuilder tableDescBuilder =
202        TableDescriptorBuilder.newBuilder(tn).setColumnFamily(cfDescBuilder.build());
203      admin.createTable(tableDescBuilder.build());
204      // put some data
205      Put put = new Put(row);
206      put.addColumn(family, qualifier, value);
207      table.put(put);
208      // invalid connection cache
209      conn.clearRegionLocationCache();
210      // stop client zk cluster
211      clientZkCluster.shutdown();
212      // stop current meta server and confirm the server shutdown process
213      // is not affected by client ZK crash
214      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
215      int metaServerId = cluster.getServerWithMeta();
216      HRegionServer metaServer = cluster.getRegionServer(metaServerId);
217      metaServer.stop("Stop current RS holding meta region");
218      while (!metaServer.isShutDown()) {
219        Thread.sleep(200);
220      }
221      // wait for meta region online
222      AssignmentTestingUtil.waitForAssignment(cluster.getMaster().getAssignmentManager(),
223        RegionInfoBuilder.FIRST_META_REGIONINFO);
224      // wait some long time to make sure we will retry sync data to client ZK until data set
225      Thread.sleep(10000);
226      clientZkCluster.startup(clientZkDir);
227      // new request should pass
228      Get get = new Get(row);
229      Result result = table.get(get);
230      LOG.debug("Result: " + Bytes.toString(result.getValue(family, qualifier)));
231      Assert.assertArrayEquals(value, result.getValue(family, qualifier));
232    }
233  }

238    ColumnFamilyDescriptorBuilder cfDescBuilder = ColumnFamilyDescriptorBuilder.newBuilder(family);
239    TableDescriptorBuilder tableDescBuilder =
240        TableDescriptorBuilder.newBuilder(tn).setColumnFamily(cfDescBuilder.build());
241    try (AsyncConnection ASYNC_CONN =
242        ConnectionFactory.createAsyncConnection(TEST_UTIL.getConfiguration()).get()) {
243      ASYNC_CONN.getAdmin().createTable(tableDescBuilder.build()).get();
244      AsyncTable<?> table = ASYNC_CONN.getTable(tn);
245      // put some data
246      Put put = new Put(row);
247      put.addColumn(family, qualifier, value);
248      table.put(put).get();
249      // get and verify
250      Get get = new Get(row);
251      Result result = table.get(get).get();
252      LOG.debug("Result: " + Bytes.toString(result.getValue(family, qualifier)));
253      Assert.assertArrayEquals(value, result.getValue(family, qualifier));
254    }
