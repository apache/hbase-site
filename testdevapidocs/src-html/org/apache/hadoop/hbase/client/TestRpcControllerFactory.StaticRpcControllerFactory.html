052  public static class StaticRpcControllerFactory extends RpcControllerFactory {
053
054    public StaticRpcControllerFactory(Configuration conf) {
055      super(conf);
056    }
057
058    @Override
059    public PayloadCarryingRpcController newController() {
060      return new CountingRpcController(super.newController());
061    }
062
063    @Override
064    public PayloadCarryingRpcController newController(final CellScanner cellScanner) {
065      return new CountingRpcController(super.newController(cellScanner));
066    }
067
068    @Override
069    public PayloadCarryingRpcController newController(final List<CellScannable> cellIterables) {
070      return new CountingRpcController(super.newController(cellIterables));
071    }
072  }

074  public static class CountingRpcController extends DelegatingPayloadCarryingRpcController {
075
076    private static AtomicInteger INT_PRIORITY = new AtomicInteger();
077    private static AtomicInteger TABLE_PRIORITY = new AtomicInteger();
078
079    public CountingRpcController(PayloadCarryingRpcController delegate) {
080      super(delegate);
081    }
082
083    @Override
084    public void setPriority(int priority) {
085      super.setPriority(priority);
086      INT_PRIORITY.incrementAndGet();
087    }
088
089    @Override
090    public void setPriority(TableName tn) {
091      super.setPriority(tn);
092      // ignore counts for system tables - it could change and we really only want to check on what
093      // the client should change
094      if (!tn.isSystemTable()) {
095        TABLE_PRIORITY.incrementAndGet();
096      }
097
098    }
099  }

119  /**
120   * check some of the methods and make sure we are incrementing each time. Its a bit tediuous to
121   * cover all methods here and really is a bit brittle since we can always add new methods but
122   * won't be sure to add them here. So we just can cover the major ones.
123   * @throws Exception on failure
124   */
125  @Test
126  public void testCountController() throws Exception {
127    Configuration conf = new Configuration(UTIL.getConfiguration());
128    // setup our custom controller
129    conf.set(RpcControllerFactory.CUSTOM_CONTROLLER_CONF_KEY,
130      StaticRpcControllerFactory.class.getName());
131
132    TableName name = TableName.valueOf("testcustomcontroller");
133    UTIL.createTable(name, fam1).close();
134
135    // change one of the connection properties so we get a new Connection with our configuration
136    conf.setInt(HConstants.HBASE_RPC_TIMEOUT_KEY, HConstants.DEFAULT_HBASE_RPC_TIMEOUT + 1);
137
138    Connection connection = ConnectionFactory.createConnection(conf);
139    Table table = connection.getTable(name);
140    byte[] row = Bytes.toBytes("row");
141    Put p = new Put(row);
142    p.addColumn(fam1, fam1, Bytes.toBytes("val0"));
143    table.put(p);
144
145    Integer counter = 1;
146    counter = verifyCount(counter);
147
148    Delete d = new Delete(row);
149    d.addColumn(fam1, fam1);
150    table.delete(d);
151    counter = verifyCount(counter);
152
153    Put p2 = new Put(row);
154    p2.addColumn(fam1, Bytes.toBytes("qual"), Bytes.toBytes("val1"));
155    table.batch(Lists.newArrayList(p, p2), null);
156    // this only goes to a single server, so we don't need to change the count here
157    counter = verifyCount(counter);
158
159    Append append = new Append(row);
160    append.add(fam1, fam1, Bytes.toBytes("val2"));
161    table.append(append);
162    counter = verifyCount(counter);
163
164    // and check the major lookup calls as well
165    Get g = new Get(row);
166    table.get(g);
167    counter = verifyCount(counter);
168
169    ResultScanner scan = table.getScanner(fam1);
170    scan.next();
171    scan.close();
172    counter = verifyCount(counter);
173
174    Get g2 = new Get(row);
175    table.get(Lists.newArrayList(g, g2));
176    // same server, so same as above for not changing count
177    counter = verifyCount(counter);
178
179    // make sure all the scanner types are covered
180    Scan scanInfo = new Scan(row);
181    // regular small
182    scanInfo.setSmall(true);
183    counter = doScan(table, scanInfo, counter);
184
185    // reversed, small
186    scanInfo.setReversed(true);
187    counter = doScan(table, scanInfo, counter);
188
189    // reversed, regular
190    scanInfo.setSmall(false);
191    counter = doScan(table, scanInfo, counter);
192
193    table.close();
194    connection.close();
195  }
196
197  int doScan(Table table, Scan scan, int expectedCount) throws IOException {
198    ResultScanner results = table.getScanner(scan);
199    results.next();
200    results.close();
201    return verifyCount(expectedCount);
202  }
203
204  int verifyCount(Integer counter) {
205    assertEquals(counter.intValue(), CountingRpcController.TABLE_PRIORITY.get());
206    assertEquals(0, CountingRpcController.INT_PRIORITY.get());
207    return counter + 1;
208  }
