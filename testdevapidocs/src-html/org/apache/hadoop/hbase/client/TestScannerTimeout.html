041/**
042 * Test various scanner timeout issues.
043 */
044@Category({LargeTests.class, ClientTests.class})
045public class TestScannerTimeout {
046
047  private final static HBaseTestingUtility
048      TEST_UTIL = new HBaseTestingUtility();
049
050  private static final Log LOG = LogFactory.getLog(TestScannerTimeout.class);
051  private final static byte[] SOME_BYTES = Bytes.toBytes("f");
052  private final static TableName TABLE_NAME = TableName.valueOf("t");
053  private final static int NB_ROWS = 10;
054  // Be careful w/ what you set this timer to... it can get in the way of
055  // the mini cluster coming up -- the verification in particular.
056  private final static int THREAD_WAKE_FREQUENCY = 1000;
057  private final static int SCANNER_TIMEOUT = 15000;
058  private final static int SCANNER_CACHING = 5;
059
060   /**
061   * @throws java.lang.Exception
062   */
063  @BeforeClass
064  public static void setUpBeforeClass() throws Exception {
065    Configuration c = TEST_UTIL.getConfiguration();
066    c.setInt(HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD, SCANNER_TIMEOUT);
067    c.setInt(HConstants.THREAD_WAKE_FREQUENCY, THREAD_WAKE_FREQUENCY);
068    // We need more than one region server for this test
069    TEST_UTIL.startMiniCluster(2);
070    Table table = TEST_UTIL.createTable(TABLE_NAME, SOME_BYTES);
071    for (int i = 0; i < NB_ROWS; i++) {
072      Put put = new Put(Bytes.toBytes(i));
073      put.addColumn(SOME_BYTES, SOME_BYTES, SOME_BYTES);
074      table.put(put);
075    }
076    table.close();
077  }
078
079  /**
080   * @throws java.lang.Exception
081   */
082  @AfterClass
083  public static void tearDownAfterClass() throws Exception {
084    TEST_UTIL.shutdownMiniCluster();
085  }
086
087  /**
088   * @throws java.lang.Exception
089   */
090  @Before
091  public void setUp() throws Exception {
092    TEST_UTIL.ensureSomeNonStoppedRegionServersAvailable(2);
093  }
094
095  /**
096   * Test that scanner can continue even if the region server it was reading
097   * from failed. Before 2772, it reused the same scanner id.
098   * @throws Exception
099   */
100  @Test(timeout=300000)
101  public void test2772() throws Exception {
102    LOG.info("START************ test2772");
103    HRegionServer rs = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME);
104    Scan scan = new Scan();
105    // Set a very high timeout, we want to test what happens when a RS
106    // fails but the region is recovered before the lease times out.
107    // Since the RS is already created, this conf is client-side only for
108    // this new table
109    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
110    conf.setInt(HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD, SCANNER_TIMEOUT * 100);
111
112    Connection connection = ConnectionFactory.createConnection(conf);
113    Table higherScanTimeoutTable = connection.getTable(TABLE_NAME);
114    ResultScanner r = higherScanTimeoutTable.getScanner(scan);
115    // This takes way less than SCANNER_TIMEOUT*100
116    rs.abort("die!");
117    Result[] results = r.next(NB_ROWS);
118    assertEquals(NB_ROWS, results.length);
119    r.close();
120    higherScanTimeoutTable.close();
121    connection.close();
122    LOG.info("END ************ test2772");
123
124  }
125
126  /**
127   * Test that scanner won't miss any rows if the region server it was reading
128   * from failed. Before 3686, it would skip rows in the scan.
129   * @throws Exception
130   */
131  @Test(timeout=300000)
132  public void test3686a() throws Exception {
133    LOG.info("START ************ TEST3686A---1");
134    HRegionServer rs = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME);
135    LOG.info("START ************ TEST3686A---1111");
136
137    Scan scan = new Scan();
138    scan.setCaching(SCANNER_CACHING);
139    LOG.info("************ TEST3686A");
140    MetaTableAccessor.fullScanMetaAndPrint(TEST_UTIL.getAdmin().getConnection());
141    // Set a very high timeout, we want to test what happens when a RS
142    // fails but the region is recovered before the lease times out.
143    // Since the RS is already created, this conf is client-side only for
144    // this new table
145    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
146    conf.setInt(
147        HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD, SCANNER_TIMEOUT*100);
148    Connection connection = ConnectionFactory.createConnection(conf);
149    Table table = connection.getTable(TABLE_NAME);
150    LOG.info("START ************ TEST3686A---22");
151
152    ResultScanner r = table.getScanner(scan);
153    LOG.info("START ************ TEST3686A---33");
154
155    int count = 1;
156    r.next();
157    LOG.info("START ************ TEST3686A---44");
158
159    // Kill after one call to next(), which got 5 rows.
160    rs.abort("die!");
161    while(r.next() != null) {
162      count ++;
163    }
164    assertEquals(NB_ROWS, count);
165    r.close();
166    table.close();
167    connection.close();
168    LOG.info("************ END TEST3686A");
169  }
170
171  /**
172   * Make sure that no rows are lost if the scanner timeout is longer on the
173   * client than the server, and the scan times out on the server but not the
174   * client.
175   * @throws Exception
176   */
177  @Test(timeout=300000)
178  public void test3686b() throws Exception {
179    LOG.info("START ************ test3686b");
180    HRegionServer rs = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME);
181    Scan scan = new Scan();
182    scan.setCaching(SCANNER_CACHING);
183    // Set a very high timeout, we want to test what happens when a RS
184    // fails but the region is recovered before the lease times out.
185    // Since the RS is already created, this conf is client-side only for
186    // this new table
187    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
188    conf.setInt(HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD, SCANNER_TIMEOUT * 100);
189    Connection connection = ConnectionFactory.createConnection(conf);
190    Table higherScanTimeoutTable = connection.getTable(TABLE_NAME);
191    ResultScanner r = higherScanTimeoutTable.getScanner(scan);
192    int count = 1;
193    r.next();
194    // Sleep, allowing the scan to timeout on the server but not on the client.
195    Thread.sleep(SCANNER_TIMEOUT+2000);
196    while(r.next() != null) {
197      count ++;
198    }
199    assertEquals(NB_ROWS, count);
200    r.close();
201    higherScanTimeoutTable.close();
202    connection.close();
203    LOG.info("END ************ END test3686b");
204
205  }
