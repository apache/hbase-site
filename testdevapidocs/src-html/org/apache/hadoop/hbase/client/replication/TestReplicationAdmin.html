104   * all interactions with ZK work

107  @Test
108  public void testAddRemovePeer() throws Exception {
109    ReplicationPeerConfig rpc1 = new ReplicationPeerConfig();
110    rpc1.setClusterKey(KEY_ONE);
111    ReplicationPeerConfig rpc2 = new ReplicationPeerConfig();
112    rpc2.setClusterKey(KEY_SECOND);
113    // Add a valid peer
114    admin.addPeer(ID_ONE, rpc1, null);
115    // try adding the same (fails)
116    try {
117      admin.addPeer(ID_ONE, rpc1, null);
118    } catch (Exception e) {
119      // OK!
120    }
121    assertEquals(1, admin.getPeersCount());
122    // Try to remove an inexisting peer
123    try {
124      admin.removePeer(ID_SECOND);
125      fail();
126    } catch (Exception iae) {
127      // OK!
128    }
129    assertEquals(1, admin.getPeersCount());
130    // Add a second since multi-slave is supported
131    try {
132      admin.addPeer(ID_SECOND, rpc2, null);
133    } catch (Exception iae) {
134      fail();
135    }
136    assertEquals(2, admin.getPeersCount());
137    // Remove the first peer we added
138    admin.removePeer(ID_ONE);
139    assertEquals(1, admin.getPeersCount());
140    admin.removePeer(ID_SECOND);
141    assertEquals(0, admin.getPeersCount());
142  }

238  @Test
239  public void testAppendPeerTableCFs() throws Exception {
240    ReplicationPeerConfig rpc = new ReplicationPeerConfig();
241    rpc.setClusterKey(KEY_ONE);
242    final TableName tableName1 = TableName.valueOf(name.getMethodName() + "t1");
243    final TableName tableName2 = TableName.valueOf(name.getMethodName() + "t2");
244    final TableName tableName3 = TableName.valueOf(name.getMethodName() + "t3");
245    final TableName tableName4 = TableName.valueOf(name.getMethodName() + "t4");
246    final TableName tableName5 = TableName.valueOf(name.getMethodName() + "t5");
247    final TableName tableName6 = TableName.valueOf(name.getMethodName() + "t6");
248
249    // Add a valid peer
250    hbaseAdmin.addReplicationPeer(ID_ONE, rpc);
251
252    // Update peer config, not replicate all user tables
253    rpc = hbaseAdmin.getReplicationPeerConfig(ID_ONE);
254    rpc.setReplicateAllUserTables(false);
255    hbaseAdmin.updateReplicationPeerConfig(ID_ONE, rpc);
256
257    Map<TableName, List<String>> tableCFs = new HashMap<>();
258    tableCFs.put(tableName1, null);
259    admin.appendPeerTableCFs(ID_ONE, tableCFs);
260    Map<TableName, List<String>> result =
261      ReplicationPeerConfigUtil.parseTableCFsFromConfig(admin.getPeerTableCFs(ID_ONE));
262    assertEquals(1, result.size());
263    assertEquals(true, result.containsKey(tableName1));
264    assertNull(result.get(tableName1));
265
266    // append table t2 to replication
267    tableCFs.clear();
268    tableCFs.put(tableName2, null);
269    admin.appendPeerTableCFs(ID_ONE, tableCFs);
270    result = ReplicationPeerConfigUtil.parseTableCFsFromConfig(admin.getPeerTableCFs(ID_ONE));
271    assertEquals(2, result.size());
272    assertTrue("Should contain t1", result.containsKey(tableName1));
273    assertTrue("Should contain t2", result.containsKey(tableName2));
274    assertNull(result.get(tableName1));
275    assertNull(result.get(tableName2));
276
277    // append table column family: f1 of t3 to replication
278    tableCFs.clear();
279    tableCFs.put(tableName3, new ArrayList<>());
280    tableCFs.get(tableName3).add("f1");
281    admin.appendPeerTableCFs(ID_ONE, tableCFs);
282    result = ReplicationPeerConfigUtil.parseTableCFsFromConfig(admin.getPeerTableCFs(ID_ONE));
283    assertEquals(3, result.size());
284    assertTrue("Should contain t1", result.containsKey(tableName1));
285    assertTrue("Should contain t2", result.containsKey(tableName2));
286    assertTrue("Should contain t3", result.containsKey(tableName3));
287    assertNull(result.get(tableName1));
288    assertNull(result.get(tableName2));
289    assertEquals(1, result.get(tableName3).size());
290    assertEquals("f1", result.get(tableName3).get(0));
291
292    tableCFs.clear();
293    tableCFs.put(tableName4, new ArrayList<>());
294    tableCFs.get(tableName4).add("f1");
295    tableCFs.get(tableName4).add("f2");
296    admin.appendPeerTableCFs(ID_ONE, tableCFs);
297    result = ReplicationPeerConfigUtil.parseTableCFsFromConfig(admin.getPeerTableCFs(ID_ONE));
298    assertEquals(4, result.size());
299    assertTrue("Should contain t1", result.containsKey(tableName1));
300    assertTrue("Should contain t2", result.containsKey(tableName2));
301    assertTrue("Should contain t3", result.containsKey(tableName3));
302    assertTrue("Should contain t4", result.containsKey(tableName4));
303    assertNull(result.get(tableName1));
304    assertNull(result.get(tableName2));
305    assertEquals(1, result.get(tableName3).size());
306    assertEquals("f1", result.get(tableName3).get(0));
307    assertEquals(2, result.get(tableName4).size());
308    assertEquals("f1", result.get(tableName4).get(0));
309    assertEquals("f2", result.get(tableName4).get(1));
310
311    // append "table5" => [], then append "table5" => ["f1"]
312    tableCFs.clear();
313    tableCFs.put(tableName5, new ArrayList<>());
314    admin.appendPeerTableCFs(ID_ONE, tableCFs);
315    tableCFs.clear();
316    tableCFs.put(tableName5, new ArrayList<>());
317    tableCFs.get(tableName5).add("f1");
318    admin.appendPeerTableCFs(ID_ONE, tableCFs);
319    result = ReplicationPeerConfigUtil.parseTableCFsFromConfig(admin.getPeerTableCFs(ID_ONE));
320    assertEquals(5, result.size());
321    assertTrue("Should contain t5", result.containsKey(tableName5));
322    // null means replication all cfs of tab5
323    assertNull(result.get(tableName5));
324
325    // append "table6" => ["f1"], then append "table6" => []
326    tableCFs.clear();
327    tableCFs.put(tableName6, new ArrayList<>());
328    tableCFs.get(tableName6).add("f1");
329    admin.appendPeerTableCFs(ID_ONE, tableCFs);
330    tableCFs.clear();
331    tableCFs.put(tableName6, new ArrayList<>());
332    admin.appendPeerTableCFs(ID_ONE, tableCFs);
333    result = ReplicationPeerConfigUtil.parseTableCFsFromConfig(admin.getPeerTableCFs(ID_ONE));
334    assertEquals(6, result.size());
335    assertTrue("Should contain t6", result.containsKey(tableName6));
336    // null means replication all cfs of tab6
337    assertNull(result.get(tableName6));
338
339    admin.removePeer(ID_ONE);
340  }
