018package org.apache.hadoop.hbase.client.trace.hamcrest;
019
020import static org.apache.hadoop.hbase.client.trace.hamcrest.AttributesMatchers.containsEntry;
021import static org.hamcrest.Matchers.equalTo;
022import static org.hamcrest.Matchers.is;
023import io.opentelemetry.api.common.Attributes;
024import io.opentelemetry.api.trace.SpanKind;
025import io.opentelemetry.api.trace.StatusCode;
026import io.opentelemetry.sdk.trace.data.EventData;
027import io.opentelemetry.sdk.trace.data.SpanData;
028import io.opentelemetry.sdk.trace.data.StatusData;
029import io.opentelemetry.semconv.trace.attributes.SemanticAttributes;
030import java.time.Duration;
031import java.util.Objects;
032import org.hamcrest.Description;
033import org.hamcrest.FeatureMatcher;
034import org.hamcrest.Matcher;
035import org.hamcrest.TypeSafeMatcher;
036
037/**
038 * Helper methods for matching against instances of {@link SpanData}.
039 */
040public final class SpanDataMatchers {
041
042  private SpanDataMatchers() { }
043
044  public static Matcher<SpanData> hasAttributes(Matcher<Attributes> matcher) {
045    return new FeatureMatcher<SpanData, Attributes>(
046      matcher, "SpanData having attributes that ", "attributes"
047    ) {
048      @Override protected Attributes featureValueOf(SpanData item) {
049        return item.getAttributes();
050      }
051    };
052  }
053
054  public static Matcher<SpanData> hasDuration(Matcher<Duration> matcher) {
055    return new FeatureMatcher<SpanData, Duration>(
056      matcher, "SpanData having duration that ", "duration") {
057      @Override
058      protected Duration featureValueOf(SpanData item) {
059        return Duration.ofNanos(item.getEndEpochNanos() - item.getStartEpochNanos());
060      }
061    };
062  }
063
064  public static Matcher<SpanData> hasEnded() {
065    return new TypeSafeMatcher<SpanData>() {
066      @Override protected boolean matchesSafely(SpanData item) {
067        return item.hasEnded();
068      }
069      @Override public void describeTo(Description description) {
070        description.appendText("SpanData that hasEnded");
071      }
072    };
073  }
074
075  public static Matcher<SpanData> hasEvents(Matcher<Iterable<? super EventData>> matcher) {
076    return new FeatureMatcher<SpanData, Iterable<? super EventData>>(
077      matcher, "SpanData having events that", "events") {
078      @Override protected Iterable<? super EventData> featureValueOf(SpanData item) {
079        return item.getEvents();
080      }
081    };
082  }
083
084  public static Matcher<SpanData> hasExceptionWithType(Matcher<? super String> matcher) {
085    return hasException(containsEntry(is(SemanticAttributes.EXCEPTION_TYPE), matcher));
086  }
087
088  public static Matcher<SpanData> hasException(Matcher<? super Attributes> matcher) {
089    return new FeatureMatcher<SpanData, Attributes>(matcher,
090      "SpanData having Exception with Attributes that", "exception attributes") {
091      @Override protected Attributes featureValueOf(SpanData actual) {
092        return actual.getEvents()
093          .stream()
094          .filter(e -> Objects.equals(SemanticAttributes.EXCEPTION_EVENT_NAME, e.getName()))
095          .map(EventData::getAttributes)
096          .findFirst()
097          .orElse(null);
098      }
099    };
100  }
101
102  public static Matcher<SpanData> hasKind(SpanKind kind) {
103    return new FeatureMatcher<SpanData, SpanKind>(
104      equalTo(kind), "SpanData with kind that", "SpanKind") {
105      @Override protected SpanKind featureValueOf(SpanData item) {
106        return item.getKind();
107      }
108    };
109  }
110
111  public static Matcher<SpanData> hasName(String name) {
112    return hasName(equalTo(name));
113  }
114
115  public static Matcher<SpanData> hasName(Matcher<String> matcher) {
116    return new FeatureMatcher<SpanData, String>(matcher, "SpanKind with a name that", "name") {
117      @Override protected String featureValueOf(SpanData item) {
118        return item.getName();
119      }
120    };
121  }
122
123  public static Matcher<SpanData> hasParentSpanId(String parentSpanId) {
124    return hasParentSpanId(equalTo(parentSpanId));
125  }
126
127  public static Matcher<SpanData> hasParentSpanId(SpanData parent) {
128    return hasParentSpanId(parent.getSpanId());
129  }
130
131  public static Matcher<SpanData> hasParentSpanId(Matcher<String> matcher) {
132    return new FeatureMatcher<SpanData, String>(matcher, "SpanKind with a parentSpanId that",
133      "parentSpanId"
134    ) {
135      @Override protected String featureValueOf(SpanData item) {
136        return item.getParentSpanId();
137      }
138    };
139  }
140
141  public static Matcher<SpanData> hasStatusWithCode(StatusCode statusCode) {
142    final Matcher<StatusCode> matcher = is(equalTo(statusCode));
143    return new TypeSafeMatcher<SpanData>() {
144      @Override protected boolean matchesSafely(SpanData item) {
145        final StatusData statusData = item.getStatus();
146        return statusData != null
147          && statusData.getStatusCode() != null
148          && matcher.matches(statusData.getStatusCode());
149      }
150      @Override public void describeTo(Description description) {
151        description.appendText("SpanData with StatusCode that ").appendDescriptionOf(matcher);
152      }
153    };
154  }
155
156  public static Matcher<SpanData> hasTraceId(String traceId) {
157    return hasTraceId(is(equalTo(traceId)));
158  }
159
160  public static Matcher<SpanData> hasTraceId(Matcher<String> matcher) {
161    return new FeatureMatcher<SpanData, String>(
162      matcher, "SpanData with a traceId that ", "traceId") {
163      @Override protected String featureValueOf(SpanData item) {
164        return item.getTraceId();
165      }
166    };
167  }
168}
