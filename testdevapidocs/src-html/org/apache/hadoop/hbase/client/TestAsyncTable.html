116  @Test
117  public void testSimple() throws Exception {
118    AsyncTableBase table = getTable.get();
119    table.put(new Put(row).addColumn(FAMILY, QUALIFIER, VALUE)).get();
120    assertTrue(table.exists(new Get(row).addColumn(FAMILY, QUALIFIER)).get());
121    Result result = table.get(new Get(row).addColumn(FAMILY, QUALIFIER)).get();
122    assertArrayEquals(VALUE, result.getValue(FAMILY, QUALIFIER));
123    table.delete(new Delete(row)).get();
124    result = table.get(new Get(row).addColumn(FAMILY, QUALIFIER)).get();
125    assertTrue(result.isEmpty());
126    assertFalse(table.exists(new Get(row).addColumn(FAMILY, QUALIFIER)).get());
127  }

129  private byte[] concat(byte[] base, int index) {
130    return Bytes.toBytes(Bytes.toString(base) + "-" + index);

133  @Test
134  public void testSimpleMultiple() throws Exception {
135    AsyncTableBase table = getTable.get();
136    int count = 100;
137    CountDownLatch putLatch = new CountDownLatch(count);
138    IntStream.range(0, count).forEach(
139      i -> table.put(new Put(concat(row, i)).addColumn(FAMILY, QUALIFIER, concat(VALUE, i)))
140          .thenAccept(x -> putLatch.countDown()));
141    putLatch.await();
142    BlockingQueue<Boolean> existsResp = new ArrayBlockingQueue<>(count);
143    IntStream.range(0, count)
144        .forEach(i -> table.exists(new Get(concat(row, i)).addColumn(FAMILY, QUALIFIER))
145            .thenAccept(x -> existsResp.add(x)));
146    for (int i = 0; i < count; i++) {
147      assertTrue(existsResp.take());
148    }
149    BlockingQueue<Pair<Integer, Result>> getResp = new ArrayBlockingQueue<>(count);
150    IntStream.range(0, count)
151        .forEach(i -> table.get(new Get(concat(row, i)).addColumn(FAMILY, QUALIFIER))
152            .thenAccept(x -> getResp.add(Pair.newPair(i, x))));
153    for (int i = 0; i < count; i++) {
154      Pair<Integer, Result> pair = getResp.take();
155      assertArrayEquals(concat(VALUE, pair.getFirst()),
156        pair.getSecond().getValue(FAMILY, QUALIFIER));
157    }
158    CountDownLatch deleteLatch = new CountDownLatch(count);
159    IntStream.range(0, count).forEach(
160      i -> table.delete(new Delete(concat(row, i))).thenAccept(x -> deleteLatch.countDown()));
161    deleteLatch.await();
162    IntStream.range(0, count)
163        .forEach(i -> table.exists(new Get(concat(row, i)).addColumn(FAMILY, QUALIFIER))
164            .thenAccept(x -> existsResp.add(x)));
165    for (int i = 0; i < count; i++) {
166      assertFalse(existsResp.take());
167    }
168    IntStream.range(0, count)
169        .forEach(i -> table.get(new Get(concat(row, i)).addColumn(FAMILY, QUALIFIER))
170            .thenAccept(x -> getResp.add(Pair.newPair(i, x))));
171    for (int i = 0; i < count; i++) {
172      Pair<Integer, Result> pair = getResp.take();
173      assertTrue(pair.getSecond().isEmpty());
174    }
175  }

177  @Test
178  public void testIncrement() throws InterruptedException, ExecutionException {
179    AsyncTableBase table = getTable.get();
180    int count = 100;
181    CountDownLatch latch = new CountDownLatch(count);
182    AtomicLong sum = new AtomicLong(0L);
183    IntStream.range(0, count)
184        .forEach(i -> table.incrementColumnValue(row, FAMILY, QUALIFIER, 1).thenAccept(x -> {
185          sum.addAndGet(x);
186          latch.countDown();
187        }));
188    latch.await();
189    assertEquals(count, Bytes.toLong(
190      table.get(new Get(row).addColumn(FAMILY, QUALIFIER)).get().getValue(FAMILY, QUALIFIER)));
191    assertEquals((1 + count) * count / 2, sum.get());
192  }

194  @Test
195  public void testAppend() throws InterruptedException, ExecutionException {
196    AsyncTableBase table = getTable.get();
197    int count = 10;
198    CountDownLatch latch = new CountDownLatch(count);
199    char suffix = ':';
200    AtomicLong suffixCount = new AtomicLong(0L);
201    IntStream.range(0, count).forEachOrdered(
202      i -> table.append(new Append(row).addColumn(FAMILY, QUALIFIER, Bytes.toBytes("" + i + suffix)))
203          .thenAccept(r -> {
204            suffixCount.addAndGet(Bytes.toString(r.getValue(FAMILY, QUALIFIER)).chars()
205                .filter(x -> x == suffix).count());
206            latch.countDown();
207          }));
208    latch.await();
209    assertEquals((1 + count) * count / 2, suffixCount.get());
210    String value = Bytes.toString(
211      table.get(new Get(row).addColumn(FAMILY, QUALIFIER)).get().getValue(FAMILY, QUALIFIER));
212    int[] actual = Arrays.asList(value.split("" + suffix)).stream().mapToInt(Integer::parseInt)
213        .sorted().toArray();
214    assertArrayEquals(IntStream.range(0, count).toArray(), actual);
215  }

217  @Test
218  public void testCheckAndPut() throws InterruptedException, ExecutionException {
219    AsyncTableBase table = getTable.get();
220    AtomicInteger successCount = new AtomicInteger(0);
221    AtomicInteger successIndex = new AtomicInteger(-1);
222    int count = 10;
223    CountDownLatch latch = new CountDownLatch(count);
224    IntStream.range(0, count)
225        .forEach(i -> table.checkAndMutate(row, FAMILY).qualifier(QUALIFIER).ifNotExists()
226            .thenPut(new Put(row).addColumn(FAMILY, QUALIFIER, concat(VALUE, i))).thenAccept(x -> {
227              if (x) {
228                successCount.incrementAndGet();
229                successIndex.set(i);
230              }
231              latch.countDown();
232            }));
233    latch.await();
234    assertEquals(1, successCount.get());
235    String actual = Bytes.toString(table.get(new Get(row)).get().getValue(FAMILY, QUALIFIER));
236    assertTrue(actual.endsWith(Integer.toString(successIndex.get())));
237  }

239  @Test
240  public void testCheckAndDelete() throws InterruptedException, ExecutionException {
241    AsyncTableBase table = getTable.get();
242    int count = 10;
243    CountDownLatch putLatch = new CountDownLatch(count + 1);
244    table.put(new Put(row).addColumn(FAMILY, QUALIFIER, VALUE)).thenRun(() -> putLatch.countDown());
245    IntStream.range(0, count)
246        .forEach(i -> table.put(new Put(row).addColumn(FAMILY, concat(QUALIFIER, i), VALUE))
247            .thenRun(() -> putLatch.countDown()));
248    putLatch.await();
249
250    AtomicInteger successCount = new AtomicInteger(0);
251    AtomicInteger successIndex = new AtomicInteger(-1);
252    CountDownLatch deleteLatch = new CountDownLatch(count);
253    IntStream.range(0, count)
254        .forEach(i -> table.checkAndMutate(row, FAMILY).qualifier(QUALIFIER).ifEquals(VALUE)
255            .thenDelete(
256              new Delete(row).addColumn(FAMILY, QUALIFIER).addColumn(FAMILY, concat(QUALIFIER, i)))
257            .thenAccept(x -> {
258              if (x) {
259                successCount.incrementAndGet();
260                successIndex.set(i);
261              }
262              deleteLatch.countDown();
263            }));
264    deleteLatch.await();
265    assertEquals(1, successCount.get());
266    Result result = table.get(new Get(row)).get();
267    IntStream.range(0, count).forEach(i -> {
268      if (i == successIndex.get()) {
269        assertFalse(result.containsColumn(FAMILY, concat(QUALIFIER, i)));
270      } else {
271        assertArrayEquals(VALUE, result.getValue(FAMILY, concat(QUALIFIER, i)));
272      }
273    });
274  }

276  @Test
277  public void testMutateRow() throws InterruptedException, ExecutionException, IOException {
278    AsyncTableBase table = getTable.get();
279    RowMutations mutation = new RowMutations(row);
280    mutation.add(new Put(row).addColumn(FAMILY, concat(QUALIFIER, 1), VALUE));
281    table.mutateRow(mutation).get();
282    Result result = table.get(new Get(row)).get();
283    assertArrayEquals(VALUE, result.getValue(FAMILY, concat(QUALIFIER, 1)));
284
285    mutation = new RowMutations(row);
286    mutation.add(new Delete(row).addColumn(FAMILY, concat(QUALIFIER, 1)));
287    mutation.add(new Put(row).addColumn(FAMILY, concat(QUALIFIER, 2), VALUE));
288    table.mutateRow(mutation).get();
289    result = table.get(new Get(row)).get();
290    assertNull(result.getValue(FAMILY, concat(QUALIFIER, 1)));
291    assertArrayEquals(VALUE, result.getValue(FAMILY, concat(QUALIFIER, 2)));
292  }

294  @Test
295  public void testCheckAndMutate() throws InterruptedException, ExecutionException {
296    AsyncTableBase table = getTable.get();
297    int count = 10;
298    CountDownLatch putLatch = new CountDownLatch(count + 1);
299    table.put(new Put(row).addColumn(FAMILY, QUALIFIER, VALUE)).thenRun(() -> putLatch.countDown());
300    IntStream.range(0, count)
301        .forEach(i -> table.put(new Put(row).addColumn(FAMILY, concat(QUALIFIER, i), VALUE))
302            .thenRun(() -> putLatch.countDown()));
303    putLatch.await();
304
305    AtomicInteger successCount = new AtomicInteger(0);
306    AtomicInteger successIndex = new AtomicInteger(-1);
307    CountDownLatch mutateLatch = new CountDownLatch(count);
308    IntStream.range(0, count).forEach(i -> {
309      RowMutations mutation = new RowMutations(row);
310      try {
311        mutation.add(new Delete(row).addColumn(FAMILY, QUALIFIER));
312        mutation.add(new Put(row).addColumn(FAMILY, concat(QUALIFIER, i), concat(VALUE, i)));
313      } catch (IOException e) {
314        throw new UncheckedIOException(e);
315      }
316      table.checkAndMutate(row, FAMILY).qualifier(QUALIFIER).ifEquals(VALUE).thenMutate(mutation)
317          .thenAccept(x -> {
318            if (x) {
319              successCount.incrementAndGet();
320              successIndex.set(i);
321            }
322            mutateLatch.countDown();
323          });
324    });
325    mutateLatch.await();
326    assertEquals(1, successCount.get());
327    Result result = table.get(new Get(row)).get();
328    IntStream.range(0, count).forEach(i -> {
329      if (i == successIndex.get()) {
330        assertArrayEquals(concat(VALUE, i), result.getValue(FAMILY, concat(QUALIFIER, i)));
331      } else {
332        assertArrayEquals(VALUE, result.getValue(FAMILY, concat(QUALIFIER, i)));
333      }
334    });
335  }
