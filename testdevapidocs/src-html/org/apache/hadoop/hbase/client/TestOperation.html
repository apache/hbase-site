299  /**
300   * Test the client Operations' JSON encoding to ensure that produced JSON is
301   * parseable and that the details are present and not corrupted.
302   * @throws IOException
303   */
304  @Test
305  public void testOperationJSON()
306      throws IOException {
307    // produce a Scan Operation
308    Scan scan = new Scan(ROW);
309    scan.addColumn(FAMILY, QUALIFIER);
310    // get its JSON representation, and parse it
311    String json = scan.toJSON();
312    Map<String, Object> parsedJSON = mapper.readValue(json, HashMap.class);
313    // check for the row
314    assertEquals("startRow incorrect in Scan.toJSON()",
315        Bytes.toStringBinary(ROW), parsedJSON.get("startRow"));
316    // check for the family and the qualifier.
317    List familyInfo = (List) ((Map) parsedJSON.get("families")).get(
318        Bytes.toStringBinary(FAMILY));
319    assertNotNull("Family absent in Scan.toJSON()", familyInfo);
320    assertEquals("Qualifier absent in Scan.toJSON()", 1, familyInfo.size());
321    assertEquals("Qualifier incorrect in Scan.toJSON()",
322        Bytes.toStringBinary(QUALIFIER),
323        familyInfo.get(0));
324
325    // produce a Get Operation
326    Get get = new Get(ROW);
327    get.addColumn(FAMILY, QUALIFIER);
328    // get its JSON representation, and parse it
329    json = get.toJSON();
330    parsedJSON = mapper.readValue(json, HashMap.class);
331    // check for the row
332    assertEquals("row incorrect in Get.toJSON()",
333        Bytes.toStringBinary(ROW), parsedJSON.get("row"));
334    // check for the family and the qualifier.
335    familyInfo = (List) ((Map) parsedJSON.get("families")).get(
336        Bytes.toStringBinary(FAMILY));
337    assertNotNull("Family absent in Get.toJSON()", familyInfo);
338    assertEquals("Qualifier absent in Get.toJSON()", 1, familyInfo.size());
339    assertEquals("Qualifier incorrect in Get.toJSON()",
340        Bytes.toStringBinary(QUALIFIER),
341        familyInfo.get(0));
342
343    // produce a Put operation
344    Put put = new Put(ROW);
345    put.addColumn(FAMILY, QUALIFIER, VALUE);
346    // get its JSON representation, and parse it
347    json = put.toJSON();
348    parsedJSON = mapper.readValue(json, HashMap.class);
349    // check for the row
350    assertEquals("row absent in Put.toJSON()",
351        Bytes.toStringBinary(ROW), parsedJSON.get("row"));
352    // check for the family and the qualifier.
353    familyInfo = (List) ((Map) parsedJSON.get("families")).get(
354        Bytes.toStringBinary(FAMILY));
355    assertNotNull("Family absent in Put.toJSON()", familyInfo);
356    assertEquals("KeyValue absent in Put.toJSON()", 1, familyInfo.size());
357    Map kvMap = (Map) familyInfo.get(0);
358    assertEquals("Qualifier incorrect in Put.toJSON()",
359        Bytes.toStringBinary(QUALIFIER),
360        kvMap.get("qualifier"));
361    assertEquals("Value length incorrect in Put.toJSON()",
362        VALUE.length, kvMap.get("vlen"));
363
364    // produce a Delete operation
365    Delete delete = new Delete(ROW);
366    delete.addColumn(FAMILY, QUALIFIER);
367    // get its JSON representation, and parse it
368    json = delete.toJSON();
369    parsedJSON = mapper.readValue(json, HashMap.class);
370    // check for the row
371    assertEquals("row absent in Delete.toJSON()",
372        Bytes.toStringBinary(ROW), parsedJSON.get("row"));
373    // check for the family and the qualifier.
374    familyInfo = (List) ((Map) parsedJSON.get("families")).get(
375        Bytes.toStringBinary(FAMILY));
376    assertNotNull("Family absent in Delete.toJSON()", familyInfo);
377    assertEquals("KeyValue absent in Delete.toJSON()", 1, familyInfo.size());
378    kvMap = (Map) familyInfo.get(0);
379    assertEquals("Qualifier incorrect in Delete.toJSON()",
380        Bytes.toStringBinary(QUALIFIER), kvMap.get("qualifier"));
381  }

383  @Test
384  public void testPutCreationWithByteBuffer() {
385    Put p = new Put(ROW);
386    List<Cell> c = p.get(FAMILY, QUALIFIER);
387    Assert.assertEquals(0, c.size());
388    Assert.assertEquals(HConstants.LATEST_TIMESTAMP, p.getTimeStamp());
389
390    p.addColumn(FAMILY, ByteBuffer.wrap(QUALIFIER), 1984L, ByteBuffer.wrap(VALUE));
391    c = p.get(FAMILY, QUALIFIER);
392    Assert.assertEquals(1, c.size());
393    Assert.assertEquals(1984L, c.get(0).getTimestamp());
394    Assert.assertArrayEquals(VALUE, CellUtil.cloneValue(c.get(0)));
395    Assert.assertEquals(HConstants.LATEST_TIMESTAMP, p.getTimeStamp());
396    Assert.assertEquals(0, CellComparatorImpl.COMPARATOR.compare(c.get(0), new KeyValue(c.get(0))));
397
398    p = new Put(ROW);
399    p.addColumn(FAMILY, ByteBuffer.wrap(QUALIFIER), 2013L, null);
400    c = p.get(FAMILY, QUALIFIER);
401    Assert.assertEquals(1, c.size());
402    Assert.assertEquals(2013L, c.get(0).getTimestamp());
403    Assert.assertArrayEquals(new byte[]{}, CellUtil.cloneValue(c.get(0)));
404    Assert.assertEquals(HConstants.LATEST_TIMESTAMP, p.getTimeStamp());
405    Assert.assertEquals(0, CellComparatorImpl.COMPARATOR.compare(c.get(0), new KeyValue(c.get(0))));
406
407    p = new Put(ByteBuffer.wrap(ROW));
408    p.addColumn(FAMILY, ByteBuffer.wrap(QUALIFIER), 2001L, null);
409    c = p.get(FAMILY, QUALIFIER);
410    Assert.assertEquals(1, c.size());
411    Assert.assertEquals(2001L, c.get(0).getTimestamp());
412    Assert.assertArrayEquals(new byte[]{}, CellUtil.cloneValue(c.get(0)));
413    Assert.assertArrayEquals(ROW, CellUtil.cloneRow(c.get(0)));
414    Assert.assertEquals(HConstants.LATEST_TIMESTAMP, p.getTimeStamp());
415    Assert.assertEquals(0, CellComparatorImpl.COMPARATOR.compare(c.get(0), new KeyValue(c.get(0))));
416
417    p = new Put(ByteBuffer.wrap(ROW), 1970L);
418    p.addColumn(FAMILY, ByteBuffer.wrap(QUALIFIER), 2001L, null);
419    c = p.get(FAMILY, QUALIFIER);
420    Assert.assertEquals(1, c.size());
421    Assert.assertEquals(2001L, c.get(0).getTimestamp());
422    Assert.assertArrayEquals(new byte[]{}, CellUtil.cloneValue(c.get(0)));
423    Assert.assertArrayEquals(ROW, CellUtil.cloneRow(c.get(0)));
424    Assert.assertEquals(1970L, p.getTimeStamp());
425    Assert.assertEquals(0, CellComparatorImpl.COMPARATOR.compare(c.get(0), new KeyValue(c.get(0))));
426  }

445        OperationWithAttributes.class,
446        Mutation.class,
447        Query.class,
448        Delete.class,
449        Increment.class,
450        Append.class,

456  }
