085  /**
086   * Test from client side for TimestampsFilter.
087   *
088   * The TimestampsFilter provides the ability to request cells (KeyValues)
089   * whose timestamp/version is in the specified list of timestamps/version.
090   *
091   * @throws Exception
092   */
093  @Test
094  public void testTimestampsFilter() throws Exception {
095    byte [] TABLE = Bytes.toBytes("testTimestampsFilter");
096    byte [] FAMILY = Bytes.toBytes("event_log");
097    byte [][] FAMILIES = new byte[][] { FAMILY };
098    Cell kvs[];
099
100    // create table; set versions to max...
101    Table ht = TEST_UTIL.createTable(TableName.valueOf(TABLE), FAMILIES, Integer.MAX_VALUE);
102
103    for (int rowIdx = 0; rowIdx < 5; rowIdx++) {
104      for (int colIdx = 0; colIdx < 5; colIdx++) {
105        // insert versions 201..300
106        putNVersions(ht, FAMILY, rowIdx, colIdx, 201, 300);
107        // insert versions 1..100
108        putNVersions(ht, FAMILY, rowIdx, colIdx, 1, 100);
109      }
110    }
111
112    // do some verification before flush
113    verifyInsertedValues(ht, FAMILY);
114
115    TEST_UTIL.flush();
116
117    // do some verification after flush
118    verifyInsertedValues(ht, FAMILY);
119
120    // Insert some more versions after flush. These should be in memstore.
121    // After this we should have data in both memstore & HFiles.
122    for (int rowIdx = 0; rowIdx < 5; rowIdx++) {
123      for (int colIdx = 0; colIdx < 5; colIdx++) {
124        putNVersions(ht, FAMILY, rowIdx, colIdx, 301, 400);
125        putNVersions(ht, FAMILY, rowIdx, colIdx, 101, 200);
126      }
127    }
128
129    for (int rowIdx = 0; rowIdx < 5; rowIdx++) {
130      for (int colIdx = 0; colIdx < 5; colIdx++) {
131        kvs = getNVersions(ht, FAMILY, rowIdx, colIdx,
132                           Arrays.asList(505L, 5L, 105L, 305L, 205L));
133        assertEquals(4, kvs.length);
134        checkOneCell(kvs[0], FAMILY, rowIdx, colIdx, 305);
135        checkOneCell(kvs[1], FAMILY, rowIdx, colIdx, 205);
136        checkOneCell(kvs[2], FAMILY, rowIdx, colIdx, 105);
137        checkOneCell(kvs[3], FAMILY, rowIdx, colIdx, 5);
138      }
139    }
140
141    // Request an empty list of versions using the Timestamps filter;
142    // Should return none.
143    kvs = getNVersions(ht, FAMILY, 2, 2, new ArrayList<Long>());
144    assertEquals(0, kvs == null? 0: kvs.length);
145
146    //
147    // Test the filter using a Scan operation
148    // Scan rows 0..4. For each row, get all its columns, but only
149    // those versions of the columns with the specified timestamps.
150    Result[] results = scanNVersions(ht, FAMILY, 0, 4,
151                                     Arrays.asList(6L, 106L, 306L));
152    assertEquals("# of rows returned from scan", 5, results.length);
153    for (int rowIdx = 0; rowIdx < 5; rowIdx++) {
154      kvs = results[rowIdx].rawCells();
155      // each row should have 5 columns.
156      // And we have requested 3 versions for each.
157      assertEquals("Number of KeyValues in result for row:" + rowIdx,
158                   3*5, kvs.length);
159      for (int colIdx = 0; colIdx < 5; colIdx++) {
160        int offset = colIdx * 3;
161        checkOneCell(kvs[offset + 0], FAMILY, rowIdx, colIdx, 306);
162        checkOneCell(kvs[offset + 1], FAMILY, rowIdx, colIdx, 106);
163        checkOneCell(kvs[offset + 2], FAMILY, rowIdx, colIdx, 6);
164      }
165    }
166    ht.close();
167  }

169  @Test
170  public void testMultiColumns() throws Exception {
171    byte [] TABLE = Bytes.toBytes("testTimestampsFilterMultiColumns");
172    byte [] FAMILY = Bytes.toBytes("event_log");
173    byte [][] FAMILIES = new byte[][] { FAMILY };
174
175    // create table; set versions to max...
176    Table ht = TEST_UTIL.createTable(TableName.valueOf(TABLE), FAMILIES, Integer.MAX_VALUE);
177
178    Put p = new Put(Bytes.toBytes("row"));
179    p.addColumn(FAMILY, Bytes.toBytes("column0"), (long) 3, Bytes.toBytes("value0-3"));
180    p.addColumn(FAMILY, Bytes.toBytes("column1"), (long) 3, Bytes.toBytes("value1-3"));
181    p.addColumn(FAMILY, Bytes.toBytes("column2"), (long) 1, Bytes.toBytes("value2-1"));
182    p.addColumn(FAMILY, Bytes.toBytes("column2"), (long) 2, Bytes.toBytes("value2-2"));
183    p.addColumn(FAMILY, Bytes.toBytes("column2"), (long) 3, Bytes.toBytes("value2-3"));
184    p.addColumn(FAMILY, Bytes.toBytes("column3"), (long) 2, Bytes.toBytes("value3-2"));
185    p.addColumn(FAMILY, Bytes.toBytes("column4"), (long) 1, Bytes.toBytes("value4-1"));
186    p.addColumn(FAMILY, Bytes.toBytes("column4"), (long) 2, Bytes.toBytes("value4-2"));
187    p.addColumn(FAMILY, Bytes.toBytes("column4"), (long) 3, Bytes.toBytes("value4-3"));
188    ht.put(p);
189
190    ArrayList<Long> timestamps = new ArrayList<Long>();
191    timestamps.add(new Long(3));
192    TimestampsFilter filter = new TimestampsFilter(timestamps);
193
194    Get g = new Get(Bytes.toBytes("row"));
195    g.setFilter(filter);
196    g.setMaxVersions();
197    g.addColumn(FAMILY, Bytes.toBytes("column2"));
198    g.addColumn(FAMILY, Bytes.toBytes("column4"));
199
200    Result result = ht.get(g);
201    for (Cell kv : result.listCells()) {
202      System.out.println("found row " + Bytes.toString(CellUtil.cloneRow(kv)) +
203          ", column " + Bytes.toString(CellUtil.cloneQualifier(kv)) + ", value "
204          + Bytes.toString(CellUtil.cloneValue(kv)));
205    }
206
207    assertEquals(result.listCells().size(), 2);
208    assertTrue(CellUtil.matchingValue(result.listCells().get(0), Bytes.toBytes("value2-3")));
209    assertTrue(CellUtil.matchingValue(result.listCells().get(1), Bytes.toBytes("value4-3")));
210
211    ht.close();
212  }

214  /**
215   * Test TimestampsFilter in the presence of version deletes.
216   *
217   * @throws Exception
218   */
219  @Test
220  public void testWithVersionDeletes() throws Exception {
221
222    // first test from memstore (without flushing).
223    testWithVersionDeletes(false);
224
225    // run same test against HFiles (by forcing a flush).
226    testWithVersionDeletes(true);
227  }
228
229  private void testWithVersionDeletes(boolean flushTables) throws IOException {
230    byte [] TABLE = Bytes.toBytes("testWithVersionDeletes_" +
231                                   (flushTables ? "flush" : "noflush")); 
232    byte [] FAMILY = Bytes.toBytes("event_log");
233    byte [][] FAMILIES = new byte[][] { FAMILY };
234
235    // create table; set versions to max...
236    Table ht = TEST_UTIL.createTable(TableName.valueOf(TABLE), FAMILIES, Integer.MAX_VALUE);
237
238    // For row:0, col:0: insert versions 1 through 5.
239    putNVersions(ht, FAMILY, 0, 0, 1, 5);
240
241    // delete version 4.
242    deleteOneVersion(ht, FAMILY, 0, 0, 4);
243
244    if (flushTables) {
245      TEST_UTIL.flush();
246    }
247
248    // request a bunch of versions including the deleted version. We should
249    // only get back entries for the versions that exist.
250    Cell kvs[] = getNVersions(ht, FAMILY, 0, 0, Arrays.asList(2L, 3L, 4L, 5L));
251    assertEquals(3, kvs.length);
252    checkOneCell(kvs[0], FAMILY, 0, 0, 5);
253    checkOneCell(kvs[1], FAMILY, 0, 0, 3);
254    checkOneCell(kvs[2], FAMILY, 0, 0, 2);
255
256    ht.close();
257  }

259  private void verifyInsertedValues(Table ht, byte[] cf) throws IOException {
260    for (int rowIdx = 0; rowIdx < 5; rowIdx++) {
261      for (int colIdx = 0; colIdx < 5; colIdx++) {
262        // ask for versions that exist.
263        Cell[] kvs = getNVersions(ht, cf, rowIdx, colIdx,
264                                      Arrays.asList(5L, 300L, 6L, 80L));
265        assertEquals(4, kvs.length);
266        checkOneCell(kvs[0], cf, rowIdx, colIdx, 300);
267        checkOneCell(kvs[1], cf, rowIdx, colIdx, 80);
268        checkOneCell(kvs[2], cf, rowIdx, colIdx, 6);
269        checkOneCell(kvs[3], cf, rowIdx, colIdx, 5);
270
271        // ask for versions that do not exist.
272        kvs = getNVersions(ht, cf, rowIdx, colIdx,
273                           Arrays.asList(101L, 102L));
274        assertEquals(0, kvs == null? 0: kvs.length);
275
276        // ask for some versions that exist and some that do not.
277        kvs = getNVersions(ht, cf, rowIdx, colIdx,
278                           Arrays.asList(1L, 300L, 105L, 70L, 115L));
279        assertEquals(3, kvs.length);
280        checkOneCell(kvs[0], cf, rowIdx, colIdx, 300);
281        checkOneCell(kvs[1], cf, rowIdx, colIdx, 70);
282        checkOneCell(kvs[2], cf, rowIdx, colIdx, 1);
283      }
284    }
285  }

287  /**
288   * Assert that the passed in KeyValue has expected contents for the
289   * specified row, column & timestamp.
290   */
291  private void checkOneCell(Cell kv, byte[] cf,
292                             int rowIdx, int colIdx, long ts) {
293
294    String ctx = "rowIdx=" + rowIdx + "; colIdx=" + colIdx + "; ts=" + ts;
295
296    assertEquals("Row mismatch which checking: " + ctx,
297                 "row:"+ rowIdx, Bytes.toString(CellUtil.cloneRow(kv)));
298
299    assertEquals("ColumnFamily mismatch while checking: " + ctx,
300                 Bytes.toString(cf), Bytes.toString(CellUtil.cloneFamily(kv)));
301
302    assertEquals("Column qualifier mismatch while checking: " + ctx,
303                 "column:" + colIdx,
304                  Bytes.toString(CellUtil.cloneQualifier(kv)));
305
306    assertEquals("Timestamp mismatch while checking: " + ctx,
307                 ts, kv.getTimestamp());
308
309    assertEquals("Value mismatch while checking: " + ctx,
310                 "value-version-" + ts, Bytes.toString(CellUtil.cloneValue(kv)));
311  }

313  /**
314   * Uses the TimestampFilter on a Get to request a specified list of
315   * versions for the row/column specified by rowIdx & colIdx.
316   *
317   */
318  private  Cell[] getNVersions(Table ht, byte[] cf, int rowIdx,
319                                   int colIdx, List<Long> versions)
320    throws IOException {
321    byte row[] = Bytes.toBytes("row:" + rowIdx);
322    byte column[] = Bytes.toBytes("column:" + colIdx);
323    Filter filter = new TimestampsFilter(versions);
324    Get get = new Get(row);
325    get.addColumn(cf, column);
326    get.setFilter(filter);
327    get.setMaxVersions();
328    Result result = ht.get(get);
329
330    return result.rawCells();
331  }
332
333  /**
334   * Uses the TimestampFilter on a Scan to request a specified list of
335   * versions for the rows from startRowIdx to endRowIdx (both inclusive).
336   */
337  private Result[] scanNVersions(Table ht, byte[] cf, int startRowIdx,
338                                 int endRowIdx, List<Long> versions)
339    throws IOException {
340    byte startRow[] = Bytes.toBytes("row:" + startRowIdx);
341    byte endRow[] = Bytes.toBytes("row:" + endRowIdx + 1); // exclusive
342    Filter filter = new TimestampsFilter(versions);
343    Scan scan = new Scan(startRow, endRow);
344    scan.setFilter(filter);
345    scan.setMaxVersions();
346    ResultScanner scanner = ht.getScanner(scan);
347    return scanner.next(endRowIdx - startRowIdx + 1);
348  }
349
350  /**
351   * Insert in specific row/column versions with timestamps
352   * versionStart..versionEnd.
353   */
354  private void putNVersions(Table ht, byte[] cf, int rowIdx, int colIdx,
355                            long versionStart, long versionEnd)
356      throws IOException {
357    byte row[] = Bytes.toBytes("row:" + rowIdx);
358    byte column[] = Bytes.toBytes("column:" + colIdx);
359    Put put = new Put(row);
360    put.setDurability(Durability.SKIP_WAL);
361
362    for (long idx = versionStart; idx <= versionEnd; idx++) {
363      put.addColumn(cf, column, idx, Bytes.toBytes("value-version-" + idx));
364    }
365
366    ht.put(put);
367  }
368
369  /**
370   * For row/column specified by rowIdx/colIdx, delete the cell
371   * corresponding to the specified version.
372   */
373  private void deleteOneVersion(Table ht, byte[] cf, int rowIdx,
374                                int colIdx, long version)
375    throws IOException {
376    byte row[] = Bytes.toBytes("row:" + rowIdx);
377    byte column[] = Bytes.toBytes("column:" + colIdx);
378    Delete del = new Delete(row);
379    del.addColumn(cf, column, version);
380    ht.delete(del);
381  }
