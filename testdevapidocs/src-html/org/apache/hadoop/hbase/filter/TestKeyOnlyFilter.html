018package org.apache.hadoop.hbase.filter;
019
020import static org.junit.Assert.assertTrue;
021
022import java.nio.ByteBuffer;
023import java.util.ArrayList;
024import java.util.Collection;
025import java.util.List;
026
027import org.apache.hadoop.hbase.CellUtil;
028import org.apache.hadoop.hbase.KeyValue;
029import org.apache.hadoop.hbase.KeyValue.Type;
030import org.apache.hadoop.hbase.KeyValueUtil;
031import org.apache.hadoop.hbase.TestCellUtil.ByteBufferCellImpl;
032import org.apache.hadoop.hbase.filter.KeyOnlyFilter.KeyOnlyByteBufferCell;
033import org.apache.hadoop.hbase.filter.KeyOnlyFilter.KeyOnlyCell;
034import org.apache.hadoop.hbase.testclassification.MiscTests;
035import org.apache.hadoop.hbase.testclassification.SmallTests;
036import org.apache.hadoop.hbase.util.Bytes;
037import org.junit.Test;
038import org.junit.experimental.categories.Category;
039import org.junit.runner.RunWith;
040import org.junit.runners.Parameterized;
041import org.junit.runners.Parameterized.Parameters;
042
043@Category({ MiscTests.class, SmallTests.class })
044@RunWith(Parameterized.class)
045public class TestKeyOnlyFilter {
046
047  private final boolean lenAsVal;
048
049  @Parameters
050  public static Collection<Object[]> parameters() {
051    List<Object[]> paramList = new ArrayList<Object[]>(2);
052    {
053      paramList.add(new Object[] { false });
054      paramList.add(new Object[] { true });
055    }
056    return paramList;
057  }
058
059  public TestKeyOnlyFilter(boolean lenAsVal) {
060    this.lenAsVal = lenAsVal;
061  }
062
063  @Test
064  public void testKeyOnly() throws Exception {
065    byte[] r = Bytes.toBytes("row1");
066    byte[] f = Bytes.toBytes("cf1");
067    byte[] q = Bytes.toBytes("qual1");
068    byte[] v = Bytes.toBytes("val1");
069    byte[] tags = Bytes.toBytes("tag1");
070    KeyValue kv = new KeyValue(r, f, q, 0, q.length, 1234L, Type.Put, v, 0,
071        v.length, tags);
072
073    ByteBuffer buffer = ByteBuffer.wrap(kv.getBuffer());
074    ByteBufferCellImpl bbCell = new ByteBufferCellImpl(buffer, 0,
075        buffer.remaining());
076
077    // KV format: <keylen:4><valuelen:4><key:keylen><value:valuelen>
078    // Rebuild as: <keylen:4><0:4><key:keylen>
079    int dataLen = lenAsVal ? Bytes.SIZEOF_INT : 0;
080    int keyOffset = (2 * Bytes.SIZEOF_INT);
081    int keyLen = KeyValueUtil.keyLength(kv);
082    byte[] newBuffer = new byte[keyLen + keyOffset + dataLen];
083    Bytes.putInt(newBuffer, 0, keyLen);
084    Bytes.putInt(newBuffer, Bytes.SIZEOF_INT, dataLen);
085    KeyValueUtil.appendKeyTo(kv, newBuffer, keyOffset);
086    if (lenAsVal) {
087      Bytes.putInt(newBuffer, newBuffer.length - dataLen, kv.getValueLength());
088    }
089    KeyValue KeyOnlyKeyValue = new KeyValue(newBuffer);
090
091    KeyOnlyCell keyOnlyCell = new KeyOnlyCell(kv, lenAsVal);
092    KeyOnlyByteBufferCell keyOnlyByteBufferedCell = new KeyOnlyByteBufferCell(
093        bbCell, lenAsVal);
094
095    assertTrue(CellUtil.matchingRows(KeyOnlyKeyValue, keyOnlyCell));
096    assertTrue(CellUtil.matchingRows(KeyOnlyKeyValue, keyOnlyByteBufferedCell));
097
098    assertTrue(CellUtil.matchingFamily(KeyOnlyKeyValue, keyOnlyCell));
099    assertTrue(CellUtil
100        .matchingFamily(KeyOnlyKeyValue, keyOnlyByteBufferedCell));
101
102    assertTrue(CellUtil.matchingQualifier(KeyOnlyKeyValue, keyOnlyCell));
103    assertTrue(CellUtil.matchingQualifier(KeyOnlyKeyValue,
104        keyOnlyByteBufferedCell));
105
106    assertTrue(CellUtil.matchingValue(KeyOnlyKeyValue, keyOnlyCell));
107    assertTrue(KeyOnlyKeyValue.getValueLength() == keyOnlyByteBufferedCell
108        .getValueLength());
109    if (keyOnlyByteBufferedCell.getValueLength() > 0) {
110      assertTrue(CellUtil.matchingValue(KeyOnlyKeyValue,
111          keyOnlyByteBufferedCell));
112    }
113
114    assertTrue(KeyOnlyKeyValue.getTimestamp() == keyOnlyCell.getTimestamp());
115    assertTrue(KeyOnlyKeyValue.getTimestamp() == keyOnlyByteBufferedCell
116        .getTimestamp());
117
118    assertTrue(KeyOnlyKeyValue.getTypeByte() == keyOnlyCell.getTypeByte());
119    assertTrue(KeyOnlyKeyValue.getTypeByte() == keyOnlyByteBufferedCell
120        .getTypeByte());
121
122    assertTrue(KeyOnlyKeyValue.getTagsLength() == keyOnlyCell.getTagsLength());
123    assertTrue(KeyOnlyKeyValue.getTagsLength() == keyOnlyByteBufferedCell
124        .getTagsLength());
125  }
