069  private static final long STAMP_BASE = System.currentTimeMillis();
070  private static final long[] STAMPS = {
071    STAMP_BASE-100, STAMP_BASE-200, STAMP_BASE-300

074  private static final byte[][] BAD_VALS = {
075    Bytes.toBytes("bad1"), Bytes.toBytes("bad2"), Bytes.toBytes("bad3")

083  @Before
084  public void setUp() throws Exception {
085    testVals = makeTestVals();

098    tableDescriptor.setColumnFamily(hcd1);
099    RegionInfo info = RegionInfoBuilder.newBuilder(tableDescriptor.getTableName()).build();
100    this.region = HBaseTestingUtility.createRegionAndWAL(info, TEST_UTIL.getDataTestDir(),

102    addData();
103  }

110  private void addData() throws IOException {
111    Put put = new Put(ROWS[0]);
112    // add in an entry for each stamp, with 2 as a "good" value
113    put.addColumn(FAMILIES[0], QUALIFIER, STAMPS[0], BAD_VALS[0]);
114    put.addColumn(FAMILIES[0], QUALIFIER, STAMPS[1], BAD_VALS[1]);
115    put.addColumn(FAMILIES[0], QUALIFIER, STAMPS[2], MATCH_VAL);
116    // add in entries for stamps 0 and 2.
117    // without a value check both will be "accepted"
118    // with one 2 will be accepted(since the corresponding ts entry
119    // has a matching value
120    put.addColumn(FAMILIES[1], QUALIFIER, STAMPS[0], BAD_VALS[0]);
121    put.addColumn(FAMILIES[1], QUALIFIER, STAMPS[2], BAD_VALS[2]);
122
123    this.region.put(put);
124
125    put = new Put(ROWS[1]);
126    put.addColumn(FAMILIES[0], QUALIFIER, STAMPS[0], BAD_VALS[0]);
127    // there is no corresponding timestamp for this so it should never pass
128    put.addColumn(FAMILIES[0], QUALIFIER, STAMPS[2], MATCH_VAL);
129    // if we reverse the qualifiers this one should pass
130    put.addColumn(FAMILIES[1], QUALIFIER, STAMPS[0], MATCH_VAL);
131    // should pass
132    put.addColumn(FAMILIES[1], QUALIFIER, STAMPS[1], BAD_VALS[2]);
133
134    this.region.put(put);
135  }

137  private List<KeyValue> makeTestVals() {
138    List<KeyValue> testVals = new ArrayList<>();
139    testVals.add(new KeyValue(ROWS[0], FAMILIES[0], QUALIFIER, STAMPS[0], BAD_VALS[0]));
140    testVals.add(new KeyValue(ROWS[0], FAMILIES[0], QUALIFIER, STAMPS[1], BAD_VALS[1]));
141    testVals.add(new KeyValue(ROWS[0], FAMILIES[1], QUALIFIER, STAMPS[1], BAD_VALS[2]));
142    testVals.add(new KeyValue(ROWS[0], FAMILIES[1], QUALIFIER, STAMPS[0], MATCH_VAL));
143    testVals.add(new KeyValue(ROWS[0], FAMILIES[1], QUALIFIER, STAMPS[2], BAD_VALS[2]));
144
145    return testVals;
146  }

148  /**
149   * This shouldn't be confused with TestFilter#verifyScan
150   * as expectedKeys is not the per row total, but the scan total
151   *
152   * @param s
153   * @param expectedRows
154   * @param expectedCells
155   * @throws IOException
156   */
157  private void verifyScan(Scan s, long expectedRows, long expectedCells)
158  throws IOException {
159    InternalScanner scanner = this.region.getScanner(s);
160    List<Cell> results = new ArrayList<>();
161    int i = 0;
162    int cells = 0;
163    for (boolean done = true; done; i++) {
164      done = scanner.next(results);
165      Arrays.sort(results.toArray(new Cell[results.size()]),
166          CellComparatorImpl.COMPARATOR);
167      LOG.info("counter=" + i + ", " + results);
168      if (results.isEmpty()) break;
169      cells += results.size();
170      assertTrue("Scanned too many rows! Only expected " + expectedRows +
171          " total but already scanned " + (i+1), expectedRows > i);
172      assertTrue("Expected " + expectedCells + " cells total but " +
173          "already scanned " + cells, expectedCells >= cells);
174      results.clear();
175    }
176    assertEquals("Expected " + expectedRows + " rows but scanned " + i +
177        " rows", expectedRows, i);
178    assertEquals("Expected " + expectedCells + " cells but scanned " + cells +
179            " cells", expectedCells, cells);
180  }

182  /**
183   * Test scans using a DependentColumnFilter
184   */
185  @Test
186  public void testScans() throws Exception {
187    Filter filter = new DependentColumnFilter(FAMILIES[0], QUALIFIER);
188
189    Scan scan = new Scan();
190    scan.setFilter(filter);
191    scan.readVersions(Integer.MAX_VALUE);
192
193    verifyScan(scan, 2, 8);
194
195    // drop the filtering cells
196    filter = new DependentColumnFilter(FAMILIES[0], QUALIFIER, true);
197    scan = new Scan();
198    scan.setFilter(filter);
199    scan.readVersions(Integer.MAX_VALUE);
200
201    verifyScan(scan, 2, 3);
202
203    // include a comparator operation
204    filter = new DependentColumnFilter(FAMILIES[0], QUALIFIER, false,
205    CompareOperator.EQUAL, new BinaryComparator(MATCH_VAL));
206    scan = new Scan();
207    scan.setFilter(filter);
208    scan.readVersions(Integer.MAX_VALUE);
209
210    /*
211     * expecting to get the following 3 cells
212     * row 0
213     *   put.add(FAMILIES[0], QUALIFIER, STAMPS[2], MATCH_VAL);
214     *   put.add(FAMILIES[1], QUALIFIER, STAMPS[2], BAD_VALS[2]);
215     * row 1
216     *   put.add(FAMILIES[0], QUALIFIER, STAMPS[2], MATCH_VAL);
217     */
218    verifyScan(scan, 2, 3);
219
220    // include a comparator operation and drop comparator
221    filter = new DependentColumnFilter(FAMILIES[0], QUALIFIER, true,
222    CompareOperator.EQUAL, new BinaryComparator(MATCH_VAL));
223    scan = new Scan();
224    scan.setFilter(filter);
225    scan.readVersions(Integer.MAX_VALUE);
226
227    /*
228     * expecting to get the following 1 cell
229     * row 0
230     *   put.add(FAMILIES[1], QUALIFIER, STAMPS[2], BAD_VALS[2]);
231     */
232    verifyScan(scan, 1, 1);
233
234  }

236  /**
237   * Test that the filter correctly drops rows without a corresponding timestamp
238   *
239   * @throws Exception
240   */
241  @Test
242  public void testFilterDropping() throws Exception {
243    Filter filter = new DependentColumnFilter(FAMILIES[0], QUALIFIER);
244    List<Cell> accepted = new ArrayList<>();
245    for(Cell val : testVals) {
246      if(filter.filterCell(val) == ReturnCode.INCLUDE) {
247        accepted.add(val);
248      }
249    }
250    assertEquals("check all values accepted from filterCell", 5, accepted.size());
251
252    filter.filterRowCells(accepted);
253    assertEquals("check filterRow(List<KeyValue>) dropped cell without corresponding column entry", 4, accepted.size());
254
255    // start do it again with dependent column dropping on
256    filter = new DependentColumnFilter(FAMILIES[1], QUALIFIER, true);
257    accepted.clear();
258    for(KeyValue val : testVals) {
259        if(filter.filterCell(val) == ReturnCode.INCLUDE) {
260          accepted.add(val);
261        }
262      }
263      assertEquals("check the filtering column cells got dropped", 2, accepted.size());
264
265      filter.filterRowCells(accepted);
266      assertEquals("check cell retention", 2, accepted.size());
267  }

269  /**
270   * Test for HBASE-8794. Avoid NullPointerException in DependentColumnFilter.toString().
271   */
272  @Test
273  public void testToStringWithNullComparator() {
274    // Test constructor that implicitly sets a null comparator
275    Filter filter = new DependentColumnFilter(FAMILIES[0], QUALIFIER);
276    assertNotNull(filter.toString());
277    assertTrue("check string contains 'null' as compatator is null",
278      filter.toString().contains("null"));
279
280    // Test constructor with explicit null comparator
281    filter = new DependentColumnFilter(FAMILIES[0], QUALIFIER, true, CompareOperator.EQUAL, null);
282    assertNotNull(filter.toString());
283    assertTrue("check string contains 'null' as compatator is null",
284      filter.toString().contains("null"));
285  }
