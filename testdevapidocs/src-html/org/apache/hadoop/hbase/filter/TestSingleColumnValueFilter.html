057  private static final Pattern QUICK_PATTERN = Pattern.compile("QuIcK", Pattern.CASE_INSENSITIVE | Pattern.DOTALL);

074  private Filter basicFilterNew() {

091    return new SingleColumnValueFilter(COLUMN_FAMILY, COLUMN_QUALIFIER,
092      CompareOp.EQUAL,
093      new RegexStringComparator(QUICK_REGEX));

097    return new SingleColumnValueFilter(COLUMN_FAMILY, COLUMN_QUALIFIER,
098        CompareOp.EQUAL,
099        new RegexStringComparator(pattern.pattern(), pattern.flags()));
100  }

102  @Test
103  public void testLongComparator() throws IOException {
104    Filter filter = new SingleColumnValueFilter(COLUMN_FAMILY,
105        COLUMN_QUALIFIER, CompareOp.GREATER, new LongComparator(100L));
106    KeyValue kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER,
107      Bytes.toBytes(1L));
108    assertTrue("less than", filter.filterKeyValue(kv) == Filter.ReturnCode.NEXT_ROW);
109    filter.reset();
110    byte[] buffer = kv.getBuffer();
111    Cell c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
112    assertTrue("less than", filter.filterKeyValue(c) == Filter.ReturnCode.NEXT_ROW);
113    filter.reset();
114
115    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER,
116      Bytes.toBytes(100L));
117    assertTrue("Equals 100", filter.filterKeyValue(kv) == Filter.ReturnCode.NEXT_ROW);
118    filter.reset();
119    buffer = kv.getBuffer();
120    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
121    assertTrue("Equals 100", filter.filterKeyValue(c) == Filter.ReturnCode.NEXT_ROW);
122    filter.reset();
123
124    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER,
125      Bytes.toBytes(120L));
126    assertTrue("include 120", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
127    filter.reset();
128    buffer = kv.getBuffer();
129    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
130    assertTrue("include 120", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
131  }

133  private void basicFilterTests(SingleColumnValueFilter filter)
134      throws Exception {
135    KeyValue kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, VAL_2);
136    assertTrue("basicFilter1", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
137    byte[] buffer = kv.getBuffer();
138    Cell c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
139    assertTrue("basicFilter1", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
140    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, VAL_3);
141    assertTrue("basicFilter2", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
142    buffer = kv.getBuffer();
143    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
144    assertTrue("basicFilter2", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
145    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, VAL_4);
146    assertTrue("basicFilter3", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
147    buffer = kv.getBuffer();
148    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
149    assertTrue("basicFilter3", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
150    assertFalse("basicFilterNotNull", filter.filterRow());
151    filter.reset();
152    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, VAL_1);
153    assertTrue("basicFilter4", filter.filterKeyValue(kv) == Filter.ReturnCode.NEXT_ROW);
154    buffer = kv.getBuffer();
155    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
156    assertTrue("basicFilter4", filter.filterKeyValue(c) == Filter.ReturnCode.NEXT_ROW);
157    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, VAL_2);
158    assertTrue("basicFilter4", filter.filterKeyValue(kv) == Filter.ReturnCode.NEXT_ROW);
159    buffer = kv.getBuffer();
160    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
161    assertTrue("basicFilter4", filter.filterKeyValue(c) == Filter.ReturnCode.NEXT_ROW);
162    assertFalse("basicFilterAllRemaining", filter.filterAllRemaining());
163    assertTrue("basicFilterNotNull", filter.filterRow());
164    filter.reset();
165    filter.setLatestVersionOnly(false);
166    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, VAL_1);
167    assertTrue("basicFilter5", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
168    buffer = kv.getBuffer();
169    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
170    assertTrue("basicFilter5", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
171    kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, VAL_2);
172    assertTrue("basicFilter5", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
173    buffer = kv.getBuffer();
174    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
175    assertTrue("basicFilter5", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
176    assertFalse("basicFilterNotNull", filter.filterRow());
177  }

179  private void nullFilterTests(Filter filter) throws Exception {
180    ((SingleColumnValueFilter) filter).setFilterIfMissing(true);
181    KeyValue kv = new KeyValue(ROW, COLUMN_FAMILY, COLUMN_QUALIFIER, FULLSTRING_1);
182    assertTrue("null1", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
183    byte[] buffer = kv.getBuffer();
184    Cell c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
185    assertTrue("null1", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
186    assertFalse("null1FilterRow", filter.filterRow());
187    filter.reset();
188    kv = new KeyValue(ROW, COLUMN_FAMILY, Bytes.toBytes("qual2"), FULLSTRING_2);
189    assertTrue("null2", filter.filterKeyValue(kv) == Filter.ReturnCode.INCLUDE);
190    buffer = kv.getBuffer();
191    c = new ByteBufferedCellImpl(ByteBuffer.wrap(buffer), 0, buffer.length);
192    assertTrue("null2", filter.filterKeyValue(c) == Filter.ReturnCode.INCLUDE);
193    assertTrue("null2FilterRow", filter.filterRow());
194  }

212    assertFalse("substrFilterNotNull", filter.filterRow());
