048/*
049 * This test is for the optimization added in HBASE-15243.
050 * FilterList with two MultiRowRangeFilter's is constructed using Operator.MUST_PASS_ONE.
051 */
052@Category(MediumTests.class)
053public class TestFilterListOrOperatorWithBlkCnt {
054
055  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
056  private static final Log LOG = LogFactory.getLog(TestFilterListOrOperatorWithBlkCnt.class);
057  private byte[] family = Bytes.toBytes("family");
058  private byte[] qf = Bytes.toBytes("qf");
059  private byte[] value = Bytes.toBytes("val");
060  private TableName tableName;
061  private int numRows = 10000;
062
063  @Rule
064  public TestName name = new TestName();
065
066  /**
067   * @throws Exception
068   */
069  @BeforeClass
070  public static void setUpBeforeClass() throws Exception {
071    long blkSize = 4096;
072    /*
073     * dfs block size is adjusted so that the specified number of rows would result in
074     * multiple blocks (8 for this test).
075     * Later in the test, assertion is made on the number of blocks read.
076     */
077    TEST_UTIL.getConfiguration().setLong("dfs.blocksize", blkSize);
078    TEST_UTIL.getConfiguration().setLong("dfs.bytes-per-checksum", blkSize);
079    TEST_UTIL.startMiniCluster();
080  }
081
082  /**
083   * @throws Exception
084   */
085  @AfterClass
086  public static void tearDownAfterClass() throws Exception {
087    TEST_UTIL.shutdownMiniCluster();
088  }
089
090  private static long getBlkAccessCount() {
091    return HFile.DATABLOCK_READ_COUNT.sum();
092  }
093
094  @Test
095  public void testMultiRowRangeWithFilterListOrOperatorWithBlkCnt() throws IOException {
096    tableName = TableName.valueOf(name.getMethodName());
097    Table ht = TEST_UTIL.createTable(tableName, family, Integer.MAX_VALUE);
098    generateRows(numRows, ht, family, qf, value);
099
100    Scan scan = new Scan();
101    scan.setMaxVersions();
102    long blocksStart = getBlkAccessCount();
103
104    List<RowRange> ranges1 = new ArrayList<RowRange>();
105    ranges1.add(new RowRange(Bytes.toBytes(10), true, Bytes.toBytes(15), false));
106    ranges1.add(new RowRange(Bytes.toBytes(9980), true, Bytes.toBytes(9985), false));
107
108    MultiRowRangeFilter filter1 = new MultiRowRangeFilter(ranges1);
109
110    List<RowRange> ranges2 = new ArrayList<RowRange>();
111    ranges2.add(new RowRange(Bytes.toBytes(15), true, Bytes.toBytes(20), false));
112    ranges2.add(new RowRange(Bytes.toBytes(9985), true, Bytes.toBytes(9990), false));
113
114    MultiRowRangeFilter filter2 = new MultiRowRangeFilter(ranges2);
115
116    FilterList filterList = new FilterList(FilterList.Operator.MUST_PASS_ONE);
117    filterList.addFilter(filter1);
118    filterList.addFilter(filter2);
119    scan.setFilter(filterList);
120    int resultsSize = getResultsSize(ht, scan);
121    LOG.info("found " + resultsSize + " results");
122    List<Cell> results1 = getScanResult(Bytes.toBytes(10), Bytes.toBytes(20), ht);
123    List<Cell> results2 = getScanResult(Bytes.toBytes(9980), Bytes.toBytes(9990), ht);
124
125    assertEquals(results1.size() + results2.size(), resultsSize);
126    long blocksEnd = getBlkAccessCount();
127    long diff = blocksEnd - blocksStart;
128    LOG.info("Diff in number of blocks " + diff);
129    /*
130     * Verify that we don't read all the blocks (8 in total).
131     */
132    assertEquals(4, diff);
133
134    ht.close();
135  }
136
137  private void generateRows(int numberOfRows, Table ht, byte[] family, byte[] qf, byte[] value)
138      throws IOException {
139    for (int i = 0; i < numberOfRows; i++) {
140      byte[] row = Bytes.toBytes(i);
141      Put p = new Put(row);
142      p.addColumn(family, qf, value);
143      p.setDurability(Durability.SKIP_WAL);
144      ht.put(p);
145    }
146    TEST_UTIL.flush();
147  }
148
149  private List<Cell> getScanResult(byte[] startRow, byte[] stopRow, Table ht) throws IOException {
150    Scan scan = new Scan();
151    scan.setMaxVersions();
152    if(!Bytes.toString(startRow).isEmpty()) {
153      scan.setStartRow(startRow);
154    }
155    if(!Bytes.toString(stopRow).isEmpty()) {
156      scan.setStopRow(stopRow);
157    }
158    ResultScanner scanner = ht.getScanner(scan);
159    List<Cell> kvList = new ArrayList<Cell>();
160    Result r;
161    while ((r = scanner.next()) != null) {
162      for (Cell kv : r.listCells()) {
163        kvList.add(kv);
164      }
165    }
166    return kvList;
167  }
168
169  private int getResultsSize(Table ht, Scan scan) throws IOException {
170    ResultScanner scanner = ht.getScanner(scan);
171    List<Cell> results = new ArrayList<Cell>();
172    Result r;
173    while ((r = scanner.next()) != null) {
174      for (Cell kv : r.listCells()) {
175        results.add(kv);
176      }
177    }
178    return results.size();
179  }
