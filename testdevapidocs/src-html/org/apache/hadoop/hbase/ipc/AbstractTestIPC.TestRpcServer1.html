087  // We are using the test TestRpcServiceProtos generated classes and Service because they are
088  // available and basic with methods like 'echo', and ping. Below we make a blocking service
089  // by passing in implementation of blocking interface. We use this service in all tests that
090  // follow.
091  static final BlockingService SERVICE =
092      TestRpcServiceProtos.TestProtobufRpcProto
093          .newReflectiveBlockingService(new TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface() {
094
095            @Override
096            public EmptyResponseProto ping(RpcController controller, EmptyRequestProto request)
097                throws ServiceException {
098              return null;
099            }
100
101            @Override
102            public EmptyResponseProto error(RpcController controller, EmptyRequestProto request)
103                throws ServiceException {
104              return null;
105            }
106
107            @Override
108            public EchoResponseProto echo(RpcController controller, EchoRequestProto request)
109                throws ServiceException {
110              if (controller instanceof PayloadCarryingRpcController) {
111                PayloadCarryingRpcController pcrc = (PayloadCarryingRpcController) controller;
112                // If cells, scan them to check we are able to iterate what we were given and since
113                // this is
114                // an echo, just put them back on the controller creating a new block. Tests our
115                // block
116                // building.
117                CellScanner cellScanner = pcrc.cellScanner();
118                List<Cell> list = null;
119                if (cellScanner != null) {
120                  list = new ArrayList<>();
121                  try {
122                    while (cellScanner.advance()) {
123                      list.add(cellScanner.current());
124                    }
125                  } catch (IOException e) {
126                    throw new ServiceException(e);
127                  }
128                }
129                cellScanner = CellUtil.createCellScanner(list);
130                ((PayloadCarryingRpcController) controller).setCellScanner(cellScanner);
131              }
132              return EchoResponseProto.newBuilder().setMessage(request.getMessage()).build();
133            }
134          });

198   * It is hard to verify the compression is actually happening under the wraps. Hope that if

201   * @throws IOException

206  @Test
207  public void testCompressCellBlock() throws IOException, InterruptedException, SecurityException,
208      NoSuchMethodException, ServiceException {
209    Configuration conf = new Configuration(HBaseConfiguration.create());
210    conf.set("hbase.client.rpc.compressor", GzipCodec.class.getCanonicalName());
211    List<Cell> cells = new ArrayList<>();
212    int count = 3;
213    for (int i = 0; i < count; i++) {
214      cells.add(CELL);
215    }
216    TestRpcServer rpcServer = new TestRpcServer();
217    try (AbstractRpcClient client = createRpcClient(conf)) {
218      rpcServer.start();
219      MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
220      EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
221      PayloadCarryingRpcController pcrc =
222          new PayloadCarryingRpcController(CellUtil.createCellScanner(cells));
223      InetSocketAddress address = rpcServer.getListenerAddress();
224      if (address == null) {
225        throw new IOException("Listener channel is closed");
226      }
227      Pair<Message, CellScanner> r =
228          client.call(pcrc, md, param, md.getOutputType().toProto(), User.getCurrent(), address,
229              new MetricsConnection.CallStats());
230      int index = 0;
231      while (r.getSecond().advance()) {
232        assertTrue(CELL.equals(r.getSecond().current()));
233        index++;
234      }
235      assertEquals(count, index);
236    } finally {
237      rpcServer.stop();
238    }

282      }

290    } finally {
291      rpcServer.stop();
292      verify(scheduler).stop();

309      InetSocketAddress address = rpcServer.getListenerAddress();

367  /**
368   * Tests that the RpcServer creates & dispatches CallRunner object to scheduler with non-null
369   * remoteAddress set to its Call Object
370   * @throws ServiceException
371   */
372  @Test
373  public void testRpcServerForNotNullRemoteAddressInCallObject() throws IOException,
374      ServiceException {
375    final RpcScheduler scheduler = new FifoRpcScheduler(CONF, 1);
376    final TestRpcServer1 rpcServer = new TestRpcServer1(scheduler);
377    final InetSocketAddress localAddr = new InetSocketAddress("localhost", 0);
378    try (AbstractRpcClient client = new RpcClientImpl(CONF, HConstants.CLUSTER_ID_DEFAULT,
379        localAddr, null)) {
380      rpcServer.start();
381      final InetSocketAddress isa = rpcServer.getListenerAddress();
382      if (isa == null) {
383        throw new IOException("Listener channel is closed");
384      }
385      final BlockingRpcChannel channel = client.createBlockingRpcChannel(
386          ServerName.valueOf(isa.getHostName(), isa.getPort(), System.currentTimeMillis()),
387          User.getCurrent(), 0);
388      BlockingInterface stub = TestRpcServiceProtos.TestProtobufRpcProto.newBlockingStub(channel);
389      final EchoRequestProto echoRequest =
390          EchoRequestProto.newBuilder().setMessage("GetRemoteAddress").build();
391      final EchoResponseProto echoResponse = stub.echo(null, echoRequest);
392      Assert.assertEquals(localAddr.getAddress().getHostAddress(), echoResponse.getMessage());
393    } finally {
394      rpcServer.stop();
395    }
396  }

400    AbstractRpcClient client =
401        (AbstractRpcClient) RpcClientFactory.createClient(CONF, "AbstractTestIPC");
402    final InetSocketAddress address = InetSocketAddress.createUnresolved("localhost", 0);

410        .getCause() instanceof CallTimeoutException);
