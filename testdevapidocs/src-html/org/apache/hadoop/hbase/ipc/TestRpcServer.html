018package org.apache.hadoop.hbase.ipc;

020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertFalse;
022import static org.junit.Assert.assertNotNull;
023import static org.junit.Assert.assertNull;
024import static org.junit.Assert.assertTrue;
025
026import java.nio.ByteBuffer;
027
028import org.apache.hadoop.hbase.io.ByteBufferPool;
029import org.apache.hadoop.hbase.ipc.RpcServer.CallCleanup;
030import org.apache.hadoop.hbase.nio.ByteBuff;
031import org.apache.hadoop.hbase.nio.MultiByteBuff;
032import org.apache.hadoop.hbase.nio.SingleByteBuff;
033import org.apache.hadoop.hbase.testclassification.RPCTests;
034import org.apache.hadoop.hbase.testclassification.SmallTests;
035import org.apache.hadoop.hbase.util.Pair;
036import org.junit.Test;
037import org.junit.experimental.categories.Category;
038
039@Category({ RPCTests.class, SmallTests.class })
040public class TestRpcServer {
041
042  @Test
043  public void testAllocateByteBuffToReadInto() throws Exception {
044    int maxBuffersInPool = 10;
045    ByteBufferPool pool = new ByteBufferPool(6 * 1024, maxBuffersInPool);
046    initPoolWithAllBuffers(pool, maxBuffersInPool);
047    ByteBuff buff = null;
048    Pair<ByteBuff, CallCleanup> pair;
049    // When the request size is less than 1/6th of the pool buffer size. We should use on demand
050    // created on heap Buffer
051    pair = RpcServer.allocateByteBuffToReadInto(pool, RpcServer.getMinSizeForReservoirUse(pool),
052        200);
053    buff = pair.getFirst();
054    assertTrue(buff.hasArray());
055    assertEquals(maxBuffersInPool, pool.getQueueSize());
056    assertNull(pair.getSecond());
057    // When the request size is > 1/6th of the pool buffer size.
058    pair = RpcServer.allocateByteBuffToReadInto(pool, RpcServer.getMinSizeForReservoirUse(pool),
059        1024);
060    buff = pair.getFirst();
061    assertFalse(buff.hasArray());
062    assertEquals(maxBuffersInPool - 1, pool.getQueueSize());
063    assertNotNull(pair.getSecond());
064    pair.getSecond().run();// CallCleanup#run should put back the BB to pool.
065    assertEquals(maxBuffersInPool, pool.getQueueSize());
066    // Request size> pool buffer size
067    pair = RpcServer.allocateByteBuffToReadInto(pool, RpcServer.getMinSizeForReservoirUse(pool),
068        7 * 1024);
069    buff = pair.getFirst();
070    assertFalse(buff.hasArray());
071    assertTrue(buff instanceof MultiByteBuff);
072    ByteBuffer[] bbs = ((MultiByteBuff) buff).getEnclosingByteBuffers();
073    assertEquals(2, bbs.length);
074    assertTrue(bbs[0].isDirect());
075    assertTrue(bbs[1].isDirect());
076    assertEquals(6 * 1024, bbs[0].limit());
077    assertEquals(1024, bbs[1].limit());
078    assertEquals(maxBuffersInPool - 2, pool.getQueueSize());
079    assertNotNull(pair.getSecond());
080    pair.getSecond().run();// CallCleanup#run should put back the BB to pool.
081    assertEquals(maxBuffersInPool, pool.getQueueSize());
082
083    pair = RpcServer.allocateByteBuffToReadInto(pool, RpcServer.getMinSizeForReservoirUse(pool),
084        6 * 1024 + 200);
085    buff = pair.getFirst();
086    assertFalse(buff.hasArray());
087    assertTrue(buff instanceof MultiByteBuff);
088    bbs = ((MultiByteBuff) buff).getEnclosingByteBuffers();
089    assertEquals(2, bbs.length);
090    assertTrue(bbs[0].isDirect());
091    assertFalse(bbs[1].isDirect());
092    assertEquals(6 * 1024, bbs[0].limit());
093    assertEquals(200, bbs[1].limit());
094    assertEquals(maxBuffersInPool - 1, pool.getQueueSize());
095    assertNotNull(pair.getSecond());
096    pair.getSecond().run();// CallCleanup#run should put back the BB to pool.
097    assertEquals(maxBuffersInPool, pool.getQueueSize());
098
099    ByteBuffer[] buffers = new ByteBuffer[maxBuffersInPool - 1];
100    for (int i = 0; i < maxBuffersInPool - 1; i++) {
101      buffers[i] = pool.getBuffer();
102    }
103    pair = RpcServer.allocateByteBuffToReadInto(pool, RpcServer.getMinSizeForReservoirUse(pool),
104        20 * 1024);
105    buff = pair.getFirst();
106    assertFalse(buff.hasArray());
107    assertTrue(buff instanceof MultiByteBuff);
108    bbs = ((MultiByteBuff) buff).getEnclosingByteBuffers();
109    assertEquals(2, bbs.length);
110    assertTrue(bbs[0].isDirect());
111    assertFalse(bbs[1].isDirect());
112    assertEquals(6 * 1024, bbs[0].limit());
113    assertEquals(14 * 1024, bbs[1].limit());
114    assertEquals(0, pool.getQueueSize());
115    assertNotNull(pair.getSecond());
116    pair.getSecond().run();// CallCleanup#run should put back the BB to pool.
117    assertEquals(1, pool.getQueueSize());
118    pool.getBuffer();
119    pair = RpcServer.allocateByteBuffToReadInto(pool, RpcServer.getMinSizeForReservoirUse(pool),
120        7 * 1024);
121    buff = pair.getFirst();
122    assertTrue(buff.hasArray());
123    assertTrue(buff instanceof SingleByteBuff);
124    assertEquals(7 * 1024, ((SingleByteBuff) buff).getEnclosingByteBuffer().limit());
125    assertNull(pair.getSecond());
126  }
127
128  private void initPoolWithAllBuffers(ByteBufferPool pool, int maxBuffersInPool) {
129    ByteBuffer[] buffers = new ByteBuffer[maxBuffersInPool];
130    // Just call getBuffer() on pool 'maxBuffersInPool' so as to init all buffers and then put back
131    // all. Makes pool with max #buffers.
132    for (int i = 0; i < maxBuffersInPool; i++) {
133      buffers[i] = pool.getBuffer();
134    }
135    for (ByteBuffer buf : buffers) {
136      pool.putbackBuffer(buf);
137    }
138  }
139}
