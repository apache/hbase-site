046/**
047 * Test for testing protocol buffer based RPC mechanism. This test depends on test.proto definition
048 * of types in <code>src/test/protobuf/test.proto</code> and protobuf service definition from
049 * <code>src/test/protobuf/test_rpc_service.proto</code>
050 */
051@Category({ RPCTests.class, MediumTests.class })
052public class TestProtoBufRpc {
053  public final static String ADDRESS = "localhost";
054  public static int PORT = 0;
055  private InetSocketAddress isa;
056  private Configuration conf;
057  private RpcServerInterface server;
058
059  @Before
060  public void setUp() throws IOException { // Setup server for both protocols
061    this.conf = HBaseConfiguration.create();
062    Logger log = Logger.getLogger("org.apache.hadoop.ipc.HBaseServer");
063    log.setLevel(Level.DEBUG);
064    log = Logger.getLogger("org.apache.hadoop.ipc.HBaseServer.trace");
065    log.setLevel(Level.TRACE);
066    // Create server side implementation
067    // Get RPC server for server side implementation
068    this.server = RpcServerFactory.createRpcServer(null, "testrpc",
069        Lists.newArrayList(new RpcServer.BlockingServiceAndInterface(SERVICE, null)),
070        new InetSocketAddress(ADDRESS, PORT), conf, new FifoRpcScheduler(conf, 10));
071    InetSocketAddress address = server.getListenerAddress();
072    if (address == null) {
073      throw new IOException("Listener channel is closed");
074    }
075    this.isa = address;
076    this.server.start();
077  }
078
079  @After
080  public void tearDown() throws Exception {
081    server.stop();
082  }
083
084  @Test (expected=org.apache.hadoop.hbase.shaded.com.google.protobuf.ServiceException.class
085      /*Thrown when we call stub.error*/)
086  public void testProtoBufRpc() throws Exception {
087    RpcClient rpcClient = RpcClientFactory.createClient(conf, HConstants.CLUSTER_ID_DEFAULT);
088    try {
089      BlockingInterface stub = newBlockingStub(rpcClient, this.isa);
090      // Test ping method
091      TestProtos.EmptyRequestProto emptyRequest = TestProtos.EmptyRequestProto.newBuilder().build();
092      stub.ping(null, emptyRequest);
093
094      // Test echo method
095      EchoRequestProto echoRequest = EchoRequestProto.newBuilder().setMessage("hello").build();
096      EchoResponseProto echoResponse = stub.echo(null, echoRequest);
097      assertEquals(echoResponse.getMessage(), "hello");
098
099      stub.error(null, emptyRequest);
100      fail("Expected exception is not thrown");
101    } finally {
102      rpcClient.close();
103    }
104  }
105}
