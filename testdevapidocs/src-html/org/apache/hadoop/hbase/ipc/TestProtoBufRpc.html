051/**
052 * Test for testing protocol buffer based RPC mechanism. This test depends on test.proto definition
053 * of types in <code>src/test/protobuf/test.proto</code> and protobuf service definition from
054 * <code>src/test/protobuf/test_rpc_service.proto</code>
055 */
056@RunWith(Parameterized.class)
057@Category({ RPCTests.class, SmallTests.class })
058public class TestProtoBufRpc {
059
060  @ClassRule
061  public static final HBaseClassTestRule CLASS_RULE =
062      HBaseClassTestRule.forClass(TestProtoBufRpc.class);
063
064  public final static String ADDRESS = "localhost";
065  private static int PORT = 0;
066  private InetSocketAddress isa;
067  private Configuration conf;
068  private RpcServerInterface server;
069
070  @Parameters(name = "{index}: rpcServerImpl={0}")
071  public static Collection<Object[]> parameters() {
072    return Arrays.asList(new Object[] { SimpleRpcServer.class.getName() },
073        new Object[] { NettyRpcServer.class.getName() });
074  }
075
076  @Parameter(0)
077  public String rpcServerImpl;
078
079  @Before
080  public void setUp() throws IOException { // Setup server for both protocols
081    this.conf = HBaseConfiguration.create();
082    this.conf.set(RpcServerFactory.CUSTOM_RPC_SERVER_IMPL_CONF_KEY,
083        rpcServerImpl);
084    Logger log = Logger.getLogger("org.apache.hadoop.ipc.HBaseServer");
085    log.setLevel(Level.DEBUG);
086    log = Logger.getLogger("org.apache.hadoop.ipc.HBaseServer.trace");
087    log.setLevel(Level.TRACE);
088    // Create server side implementation
089    // Get RPC server for server side implementation
090    this.server = RpcServerFactory.createRpcServer(null, "testrpc",
091        Lists.newArrayList(new RpcServer.BlockingServiceAndInterface(SERVICE, null)),
092        new InetSocketAddress(ADDRESS, PORT), conf, new FifoRpcScheduler(conf, 10));
093    InetSocketAddress address = server.getListenerAddress();
094    if (address == null) {
095      throw new IOException("Listener channel is closed");
096    }
097    this.isa = address;
098    this.server.start();
099  }
100
101  @After
102  public void tearDown() throws Exception {
103    server.stop();
104  }
105
106  @Test (expected=org.apache.hbase.thirdparty.com.google.protobuf.ServiceException.class
107      /*Thrown when we call stub.error*/)
108  public void testProtoBufRpc() throws Exception {
109    RpcClient rpcClient = RpcClientFactory.createClient(conf, HConstants.CLUSTER_ID_DEFAULT);
110    try {
111      BlockingInterface stub = newBlockingStub(rpcClient, this.isa);
112      // Test ping method
113      TestProtos.EmptyRequestProto emptyRequest = TestProtos.EmptyRequestProto.newBuilder().build();
114      stub.ping(null, emptyRequest);
115
116      // Test echo method
117      EchoRequestProto echoRequest = EchoRequestProto.newBuilder().setMessage("hello").build();
118      EchoResponseProto echoResponse = stub.echo(null, echoRequest);
119      assertEquals("hello", echoResponse.getMessage());
120
121      stub.error(null, emptyRequest);
122      fail("Expected exception is not thrown");
123    } finally {
124      rpcClient.close();
125    }
126  }
127}
