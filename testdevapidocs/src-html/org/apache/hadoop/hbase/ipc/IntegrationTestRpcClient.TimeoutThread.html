091  class Cluster {
092    Random random = new Random();
093    ReadWriteLock lock = new ReentrantReadWriteLock();
094    HashMap<InetSocketAddress, RpcServer> rpcServers = new HashMap<>();
095    List<RpcServer> serverList = new ArrayList<>();
096    int maxServers;
097    int minServers;
098
099    Cluster(int minServers, int maxServers) {
100      this.minServers = minServers;
101      this.maxServers = maxServers;
102    }
103
104    RpcServer startServer() throws IOException {
105      lock.writeLock().lock();
106      try {
107        if (rpcServers.size() >= maxServers) {
108          return null;
109        }
110
111        RpcServer rpcServer = RpcServerFactory.createRpcServer(null,
112            "testRpcServer", Lists
113                .newArrayList(new BlockingServiceAndInterface(SERVICE, null)),
114            new InetSocketAddress("localhost", 0), conf, new FifoRpcScheduler(
115                conf, 1));
116        rpcServer.start();
117        InetSocketAddress address = rpcServer.getListenerAddress();
118        if (address == null) {
119          throw new IOException("Listener channel is closed");
120        }
121        rpcServers.put(address, rpcServer);
122        serverList.add(rpcServer);
123        LOG.info("Started server: " + address);
124        return rpcServer;
125      } finally {
126        lock.writeLock().unlock();
127      }
128    }
129
130    void stopRandomServer() throws Exception {
131      lock.writeLock().lock();
132      RpcServer rpcServer = null;
133      try {
134        if (rpcServers.size() <= minServers) {
135          return;
136        }
137        int size = rpcServers.size();
138        int rand = random.nextInt(size);
139        rpcServer = serverList.remove(rand);
140        InetSocketAddress address = rpcServer.getListenerAddress();
141        if (address == null) {
142          // Throw exception here. We can't remove this instance from the server map because
143          // we no longer have access to its map key
144          throw new IOException("Listener channel is closed");
145        }
146        rpcServers.remove(address);
147
148        if (rpcServer != null) {
149          stopServer(rpcServer);
150        }
151      } finally {
152        lock.writeLock().unlock();
153      }
154    }
155
156    void stopServer(RpcServer rpcServer) throws InterruptedException {
157      InetSocketAddress address = rpcServer.getListenerAddress();
158      LOG.info("Stopping server: " + address);
159      rpcServer.stop();
160      rpcServer.join();
161      LOG.info("Stopped server: " + address);
162    }
163
164    void stopRunning() throws InterruptedException {
165      lock.writeLock().lock();
166      try {
167        for (RpcServer rpcServer : serverList) {
168          stopServer(rpcServer);
169        }
170
171      } finally {
172        lock.writeLock().unlock();
173      }
174    }
175
176    RpcServer getRandomServer() {
177      lock.readLock().lock();
178      try {
179        int size = rpcServers.size();
180        int rand = random.nextInt(size);
181        return serverList.get(rand);
182      } finally {
183        lock.readLock().unlock();
184      }
185    }
186  }

188  static class MiniChaosMonkey extends Thread {
189    AtomicBoolean running = new  AtomicBoolean(true);
190    Random random = new Random();
191    AtomicReference<Exception> exception = new AtomicReference<>(null);
192    Cluster cluster;
193
194    public MiniChaosMonkey(Cluster cluster) {
195      this.cluster = cluster;
196    }
197
198    @Override
199    public void run() {
200      while (running.get()) {
201        switch (random.nextInt() % 2) {
202        case 0: //start a server
203          try {
204            cluster.startServer();
205          } catch (Exception e) {
206            LOG.warn(e);
207            exception.compareAndSet(null, e);
208          }
209          break;
210
211        case 1: // stop a server
212          try {
213            cluster.stopRandomServer();
214          } catch (Exception e) {
215            LOG.warn(e);
216            exception.compareAndSet(null, e);
217          }
218        default:
219        }
220
221        Threads.sleep(100);
222      }
223    }
224
225    void stopRunning() {
226      running.set(false);
227    }
228
229    void rethrowException() throws Exception {
230      if (exception.get() != null) {
231        throw exception.get();
232      }
233    }
234  }

236  static class SimpleClient extends Thread {
237    AbstractRpcClient<?> rpcClient;
238    AtomicBoolean running = new  AtomicBoolean(true);
239    AtomicBoolean sending = new AtomicBoolean(false);
240    AtomicReference<Throwable> exception = new AtomicReference<>(null);
241    Cluster cluster;
242    String id;
243    long numCalls = 0;
244    Random random = new Random();
245
246    public SimpleClient(Cluster cluster, AbstractRpcClient<?> rpcClient, String id) {
247      this.cluster = cluster;
248      this.rpcClient = rpcClient;
249      this.id = id;
250      this.setName(id);
251    }
252
253    @Override
254    public void run() {
255      while (running.get()) {
256        boolean isBigPayload = random.nextBoolean();
257        String message = isBigPayload ? BIG_PAYLOAD : id + numCalls;
258        EchoRequestProto param = EchoRequestProto.newBuilder().setMessage(message).build();
259        EchoResponseProto ret;
260        RpcServer server = cluster.getRandomServer();
261        try {
262          sending.set(true);
263          BlockingInterface stub = newBlockingStub(rpcClient, server.getListenerAddress());
264          ret = stub.echo(null, param);
265        } catch (Exception e) {
266          LOG.warn(e);
267          continue; // expected in case connection is closing or closed
268        }
269
270        try {
271          assertNotNull(ret);
272          assertEquals(message, ret.getMessage());
273        } catch (Throwable t) {
274          exception.compareAndSet(null, t);
275        }
276
277        numCalls++;
278      }
279    }
280
281    void stopRunning() {
282      running.set(false);
283    }
284    boolean isSending() {
285      return sending.get();
286    }
287
288    void rethrowException() throws Throwable {
289      if (exception.get() != null) {
290        throw exception.get();
291      }
292    }
293  }

361  static class TimeoutThread extends Thread {
362    long timeout;
363    public TimeoutThread(long timeout) {
364      this.timeout = timeout;
365    }
366
367    @Override
368    public void run() {
369      try {
370        Thread.sleep(timeout);
371        Threads.printThreadInfo(System.err, "TEST TIMEOUT STACK DUMP");
372        System.exit(1); // a timeout happened
373      } catch (InterruptedException e) {
374        // this is what we want
375      }
376    }
377
378    // runs in the same thread context but injects a timeout thread which will exit the JVM on
379    // timeout
380    static void runWithTimeout(Callable<?> callable, long timeout) throws Exception {
381      TimeoutThread thread = new TimeoutThread(timeout);
382      thread.start();
383      callable.call();
384      thread.interrupt();
385    }
386  }

388  public void testRpcWithChaosMonkey(boolean isSyncClient) throws Throwable {
389    LOG.info("Starting test");
390    Cluster cluster = new Cluster(10, 100);
391    for (int i = 0; i < 10; i++) {
392      cluster.startServer();
393    }
394
395    ArrayList<SimpleClient> clients = new ArrayList<>(30);
396
397    // all threads should share the same rpc client
398    AbstractRpcClient<?> rpcClient = createRpcClient(conf, isSyncClient);
399
400    for (int i = 0; i < 30; i++) {
401      String clientId = "client_" + i + "_";
402      LOG.info("Starting client: " + clientId);
403      SimpleClient client = new SimpleClient(cluster, rpcClient, clientId);
404      client.start();
405      clients.add(client);
406    }
407
408    LOG.info("Starting MiniChaosMonkey");
409    MiniChaosMonkey cm = new MiniChaosMonkey(cluster);
410    cm.start();
411
412    Threads.sleep(30000);
413
414    LOG.info("Stopping MiniChaosMonkey");
415    cm.stopRunning();
416    cm.join();
417    cm.rethrowException();
418
419    LOG.info("Stopping clients");
420    for (SimpleClient client : clients) {
421      LOG.info("Stopping client: " + client.id);
422      LOG.info(client.id + " numCalls:" + client.numCalls);
423      client.stopRunning();
424      client.join();
425      client.rethrowException();
426      assertTrue(client.numCalls > 10);
427    }
428
429    LOG.info("Stopping RpcClient");
430    rpcClient.close();
431
432    LOG.info("Stopping Cluster");
433    cluster.stopRunning();
434  }
