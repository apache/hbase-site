037@Category({RPCTests.class, SmallTests.class})
038public class TestRpcMetrics {
039  public MetricsAssertHelper HELPER = CompatibilityFactory.getInstance(MetricsAssertHelper.class);

041  @Test
042  public void testFactory() {
043    MetricsHBaseServer masterMetrics = new MetricsHBaseServer("HMaster", new MetricsHBaseServerWrapperStub());
044    MetricsHBaseServerSource masterSource = masterMetrics.getMetricsSource();
045
046    MetricsHBaseServer rsMetrics = new MetricsHBaseServer("HRegionServer", new MetricsHBaseServerWrapperStub());
047    MetricsHBaseServerSource rsSource = rsMetrics.getMetricsSource();
048
049
050    assertEquals("master", masterSource.getMetricsContext());
051    assertEquals("regionserver", rsSource.getMetricsContext());
052
053    assertEquals("Master,sub=IPC", masterSource.getMetricsJmxContext());
054    assertEquals("RegionServer,sub=IPC", rsSource.getMetricsJmxContext());
055
056    assertEquals("Master", masterSource.getMetricsName());
057    assertEquals("RegionServer", rsSource.getMetricsName());
058  }
059
060  /**
061   * This test makes sure that the numbers from a MetricsHBaseServerWrapper are correctly exported
062   * to hadoop metrics 2 system.
063   */
064  @Test
065  public void testWrapperSource() {
066    MetricsHBaseServer mrpc = new MetricsHBaseServer("HMaster", new MetricsHBaseServerWrapperStub());
067    MetricsHBaseServerSource serverSource = mrpc.getMetricsSource();
068    HELPER.assertGauge("queueSize", 101, serverSource);
069    HELPER.assertGauge("numCallsInGeneralQueue", 102, serverSource);
070    HELPER.assertGauge("numCallsInReplicationQueue", 103, serverSource);
071    HELPER.assertGauge("numCallsInPriorityQueue", 104, serverSource);
072    HELPER.assertGauge("numOpenConnections", 105, serverSource);
073    HELPER.assertGauge("numActiveHandler", 106, serverSource);
074  }
075
076  /**
077   * Test to make sure that all the actively called method on MetricsHBaseServer work.
078   */
079  @Test
080  public void testSourceMethods() {
081    MetricsHBaseServer mrpc = new MetricsHBaseServer("HMaster", new MetricsHBaseServerWrapperStub());
082    MetricsHBaseServerSource serverSource = mrpc.getMetricsSource();
083
084    for (int i=0; i < 12; i++) {
085      mrpc.authenticationFailure();
086    }
087    for (int i=0; i < 13; i++) {
088      mrpc.authenticationSuccess();
089    }
090    HELPER.assertCounter("authenticationFailures", 12, serverSource);
091    HELPER.assertCounter("authenticationSuccesses", 13, serverSource);
092
093
094
095    for (int i=0; i < 14; i++) {
096      mrpc.authorizationSuccess();
097    }
098    for (int i=0; i < 15; i++) {
099      mrpc.authorizationFailure();
100    }
101    HELPER.assertCounter("authorizationSuccesses", 14, serverSource);
102    HELPER.assertCounter("authorizationFailures", 15, serverSource);
103
104
105    mrpc.dequeuedCall(100);
106    mrpc.processedCall(101);
107    mrpc.totalCall(102);
108    HELPER.assertCounter("queueCallTime_NumOps", 1, serverSource);
109    HELPER.assertCounter("processCallTime_NumOps", 1, serverSource);
110    HELPER.assertCounter("totalCallTime_NumOps", 1, serverSource);
111
112    mrpc.sentBytes(103);
113    mrpc.sentBytes(103);
114    mrpc.sentBytes(103);
115
116    mrpc.receivedBytes(104);
117    mrpc.receivedBytes(104);
118
119    HELPER.assertCounter("sentBytes", 309, serverSource);
120    HELPER.assertCounter("receivedBytes", 208, serverSource);
121
122    mrpc.receivedRequest(105);
123    mrpc.sentResponse(106);
124    HELPER.assertCounter("requestSize_NumOps", 1, serverSource);
125    HELPER.assertCounter("responseSize_NumOps", 1, serverSource);
126
127    mrpc.exception(null);
128    HELPER.assertCounter("exceptions", 1, serverSource);
129
130    mrpc.exception(new RegionMovedException(ServerName.parseServerName("localhost:60020"), 100));
131    mrpc.exception(new RegionTooBusyException());
132    mrpc.exception(new OutOfOrderScannerNextException());
133    mrpc.exception(new NotServingRegionException());
134    HELPER.assertCounter("exceptions.RegionMovedException", 1, serverSource);
135    HELPER.assertCounter("exceptions.RegionTooBusyException", 1, serverSource);
136    HELPER.assertCounter("exceptions.OutOfOrderScannerNextException", 1, serverSource);
137    HELPER.assertCounter("exceptions.NotServingRegionException", 1, serverSource);
138    HELPER.assertCounter("exceptions", 5, serverSource);
139  }

141  @Test
142  public void testServerContextNameWithHostName() {
143    String[] masterServerNames = { "master/node-xyz/10.19.250.253:16020",
144        "master/node-regionserver-xyz/10.19.250.253:16020", "HMaster/node-xyz/10.19.250.253:16020",
145        "HMaster/node-regionserver-xyz/10.19.250.253:16020" };
146
147    String[] regionServerNames = { "regionserver/node-xyz/10.19.250.253:16020",
148        "regionserver/node-master1-xyz/10.19.250.253:16020",
149        "HRegionserver/node-xyz/10.19.250.253:16020",
150        "HRegionserver/node-master1-xyz/10.19.250.253:16020" };
151
152    MetricsHBaseServerSource masterSource = null;
153    for (String serverName : masterServerNames) {
154      masterSource = new MetricsHBaseServer(serverName, new MetricsHBaseServerWrapperStub())
155          .getMetricsSource();
156      assertEquals("master", masterSource.getMetricsContext());
157      assertEquals("Master,sub=IPC", masterSource.getMetricsJmxContext());
158      assertEquals("Master", masterSource.getMetricsName());
159    }
160
161    MetricsHBaseServerSource rsSource = null;
162    for (String serverName : regionServerNames) {
163      rsSource = new MetricsHBaseServer(serverName, new MetricsHBaseServerWrapperStub())
164          .getMetricsSource();
165      assertEquals("regionserver", rsSource.getMetricsContext());
166      assertEquals("RegionServer,sub=IPC", rsSource.getMetricsJmxContext());
167      assertEquals("RegionServer", rsSource.getMetricsName());
168    }
169  }
