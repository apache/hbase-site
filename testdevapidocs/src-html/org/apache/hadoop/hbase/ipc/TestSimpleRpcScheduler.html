117    try {
118      Field ExecutorField = scheduler.getClass().getDeclaredField("callExecutor");
119      ExecutorField.setAccessible(true);

121      RpcExecutor rpcExecutor = (RpcExecutor)ExecutorField.get(scheduler);
122
123      Field handlerCountField = rpcExecutor.getClass().getSuperclass().getSuperclass().getDeclaredField("handlerCount");
124
125      handlerCountField.setAccessible(true);
126      handlerCountField.set(rpcExecutor, 0);

130      numCallQueuesField.setAccessible(true);
131      numCallQueuesField.set(rpcExecutor, 1);

139      LOG.error("No such field exception"+e);

183  @Test
184  public void testHandlerIsolation() throws IOException, InterruptedException {
185
186    CallRunner generalTask = createMockTask();
187    CallRunner priorityTask = createMockTask();
188    CallRunner replicationTask = createMockTask();
189    List<CallRunner> tasks = ImmutableList.of(
190        generalTask,
191        priorityTask,
192        replicationTask);
193    Map<CallRunner, Integer> qos = ImmutableMap.of(
194        generalTask, 0,
195        priorityTask, HConstants.HIGH_QOS + 1,
196        replicationTask, HConstants.REPLICATION_QOS);
197    PriorityFunction qosFunction = mock(PriorityFunction.class);
198    final Map<CallRunner, Thread> handlerThreads = Maps.newHashMap();
199    final CountDownLatch countDownLatch = new CountDownLatch(tasks.size());
200    Answer<Void> answerToRun = new Answer<Void>() {
201      @Override
202      public Void answer(InvocationOnMock invocationOnMock) throws Throwable {
203        synchronized (handlerThreads) {
204          handlerThreads.put(
205              (CallRunner) invocationOnMock.getMock(),
206              Thread.currentThread());
207        }
208        countDownLatch.countDown();
209        return null;
210      }
211    };
212    for (CallRunner task : tasks) {
213      task.setStatus(new MonitoredRPCHandlerImpl());
214      doAnswer(answerToRun).when(task).run();
215    }
216
217    RpcScheduler scheduler = new SimpleRpcScheduler(
218        conf, 1, 1 ,1, qosFunction, HConstants.HIGH_QOS);
219    scheduler.init(CONTEXT);
220    scheduler.start();
221    for (CallRunner task : tasks) {
222      when(qosFunction.getPriority((RPCProtos.RequestHeader) anyObject(),
223        (Message) anyObject(), (User) anyObject()))
224          .thenReturn(qos.get(task));
225      scheduler.dispatch(task);
226    }
227    for (CallRunner task : tasks) {
228      verify(task, timeout(1000)).run();
229    }
230    scheduler.stop();
231
232    // Tests that these requests are handled by three distinct threads.
233    countDownLatch.await();
234    assertEquals(3, ImmutableSet.copyOf(handlerThreads.values()).size());
235  }

252  private void testRpcScheduler(final String queueType) throws Exception {
253    Configuration schedConf = HBaseConfiguration.create();
254    schedConf.set(RpcExecutor.CALL_QUEUE_TYPE_CONF_KEY, queueType);
255
256    PriorityFunction priority = mock(PriorityFunction.class);
257    when(priority.getPriority(any(RequestHeader.class),
258      any(Message.class), any(User.class)))
259      .thenReturn(HConstants.NORMAL_QOS);
260
261    RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 1, 1, 1, priority,
262                                                    HConstants.QOS_THRESHOLD);
263    try {
264      scheduler.start();
265
266      CallRunner smallCallTask = mock(CallRunner.class);
267      ServerCall smallCall = mock(ServerCall.class);
268      RequestHeader smallHead = RequestHeader.newBuilder().setCallId(1).build();
269      when(smallCallTask.getRpcCall()).thenReturn(smallCall);
270      when(smallCall.getHeader()).thenReturn(smallHead);
271      when(smallCall.getRequestUser()).thenReturn(Optional.empty());
272
273      CallRunner largeCallTask = mock(CallRunner.class);
274      ServerCall largeCall = mock(ServerCall.class);
275      RequestHeader largeHead = RequestHeader.newBuilder().setCallId(50).build();
276      when(largeCallTask.getRpcCall()).thenReturn(largeCall);
277      when(largeCall.getHeader()).thenReturn(largeHead);
278      when(largeCall.getRequestUser()).thenReturn(Optional.empty());
279
280      CallRunner hugeCallTask = mock(CallRunner.class);
281      ServerCall hugeCall = mock(ServerCall.class);
282      RequestHeader hugeHead = RequestHeader.newBuilder().setCallId(100).build();
283      when(hugeCallTask.getRpcCall()).thenReturn(hugeCall);
284      when(hugeCall.getHeader()).thenReturn(hugeHead);
285      when(hugeCall.getRequestUser()).thenReturn(Optional.empty());
286
287      when(priority.getDeadline(eq(smallHead), any(Message.class))).thenReturn(0L);
288      when(priority.getDeadline(eq(largeHead), any(Message.class))).thenReturn(50L);
289      when(priority.getDeadline(eq(hugeHead), any(Message.class))).thenReturn(100L);
290
291      final ArrayList<Integer> work = new ArrayList<>();
292      doAnswerTaskExecution(smallCallTask, work, 10, 250);
293      doAnswerTaskExecution(largeCallTask, work, 50, 250);
294      doAnswerTaskExecution(hugeCallTask, work, 100, 250);
295
296      scheduler.dispatch(smallCallTask);
297      scheduler.dispatch(smallCallTask);
298      scheduler.dispatch(smallCallTask);
299      scheduler.dispatch(hugeCallTask);
300      scheduler.dispatch(smallCallTask);
301      scheduler.dispatch(largeCallTask);
302      scheduler.dispatch(smallCallTask);
303      scheduler.dispatch(smallCallTask);
304
305      while (work.size() < 8) {
306        Thread.sleep(100);
307      }
308
309      int seqSum = 0;
310      int totalTime = 0;
311      for (int i = 0; i < work.size(); ++i) {
312        LOG.debug("Request i=" + i + " value=" + work.get(i));
313        seqSum += work.get(i);
314        totalTime += seqSum;
315      }
316      LOG.debug("Total Time: " + totalTime);
317
318      // -> [small small small huge small large small small]
319      // -> NO REORDER   [10 10 10 100 10 50 10 10] -> 930 (FIFO Queue)
320      // -> WITH REORDER [10 10 10 10 10 10 50 100] -> 530 (Deadline Queue)
321      if (queueType.equals(RpcExecutor.CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE)) {
322        assertEquals(530, totalTime);
323      } else if (queueType.equals(RpcExecutor.CALL_QUEUE_TYPE_FIFO_CONF_VALUE)) {
324        assertEquals(930, totalTime);
325      }
326    } finally {
327      scheduler.stop();
328    }
329  }

349  public void testScanQueues() throws Exception {
350    Configuration schedConf = HBaseConfiguration.create();
351    schedConf.setFloat(RpcExecutor.CALL_QUEUE_HANDLER_FACTOR_CONF_KEY, 1.0f);
352    schedConf.setFloat(RWQueueRpcExecutor.CALL_QUEUE_READ_SHARE_CONF_KEY, 0.7f);

358
359    RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 3, 1, 1, priority,
360                                                    HConstants.QOS_THRESHOLD);
361    try {
362      scheduler.start();
363
364      CallRunner putCallTask = mock(CallRunner.class);
365      ServerCall putCall = mock(ServerCall.class);
366      putCall.param = RequestConverter.buildMutateRequest(
367          Bytes.toBytes("abc"), new Put(Bytes.toBytes("row")));
368      RequestHeader putHead = RequestHeader.newBuilder().setMethodName("mutate").build();
369      when(putCallTask.getRpcCall()).thenReturn(putCall);
370      when(putCall.getHeader()).thenReturn(putHead);
371      when(putCall.getParam()).thenReturn(putCall.param);
372      when(putCall.getRequestUser()).thenReturn(Optional.empty());
373
374      CallRunner getCallTask = mock(CallRunner.class);
375      ServerCall getCall = mock(ServerCall.class);
376      RequestHeader getHead = RequestHeader.newBuilder().setMethodName("get").build();
377      when(getCallTask.getRpcCall()).thenReturn(getCall);
378      when(getCall.getHeader()).thenReturn(getHead);
379      when(getCall.getRequestUser()).thenReturn(Optional.empty());
380
381      CallRunner scanCallTask = mock(CallRunner.class);
382      ServerCall scanCall = mock(ServerCall.class);
383      scanCall.param = ScanRequest.newBuilder().setScannerId(1).build();
384      RequestHeader scanHead = RequestHeader.newBuilder().setMethodName("scan").build();
385      when(scanCallTask.getRpcCall()).thenReturn(scanCall);
386      when(scanCall.getHeader()).thenReturn(scanHead);
387      when(scanCall.getParam()).thenReturn(scanCall.param);
388      when(scanCall.getRequestUser()).thenReturn(Optional.empty());
389
390      ArrayList<Integer> work = new ArrayList<>();
391      doAnswerTaskExecution(putCallTask, work, 1, 1000);
392      doAnswerTaskExecution(getCallTask, work, 2, 1000);
393      doAnswerTaskExecution(scanCallTask, work, 3, 1000);
394
395      // There are 3 queues: [puts], [gets], [scans]
396      // so the calls will be interleaved
397      scheduler.dispatch(putCallTask);
398      scheduler.dispatch(putCallTask);
399      scheduler.dispatch(putCallTask);
400      scheduler.dispatch(getCallTask);
401      scheduler.dispatch(getCallTask);
402      scheduler.dispatch(getCallTask);
403      scheduler.dispatch(scanCallTask);
404      scheduler.dispatch(scanCallTask);
405      scheduler.dispatch(scanCallTask);
406
407      while (work.size() < 6) {
408        Thread.sleep(100);
409      }
410
411      for (int i = 0; i < work.size() - 2; i += 3) {
412        assertNotEquals(work.get(i + 0), work.get(i + 1));
413        assertNotEquals(work.get(i + 0), work.get(i + 2));
414        assertNotEquals(work.get(i + 1), work.get(i + 2));
415      }
416    } finally {
417      scheduler.stop();
418    }

421  private void doAnswerTaskExecution(final CallRunner callTask,
422      final ArrayList<Integer> results, final int value, final int sleepInterval) {
423    callTask.setStatus(new MonitoredRPCHandlerImpl());
424    doAnswer(new Answer<Object>() {
425      @Override
426      public Object answer(InvocationOnMock invocation) {
427        synchronized (results) {
428          results.add(value);
429        }
430        Threads.sleepWithoutInterrupt(sleepInterval);
431        return null;
432      }
433    }).when(callTask).run();
434  }

484    private final BlockingQueue<Long> timeQ = new LinkedBlockingQueue<>();
485
486    private long offset;
487
488    private final Set<String> threadNamePrefixs = new HashSet<>();
489
490    @Override
491    public long currentTime() {
492      for (String threadNamePrefix : threadNamePrefixs) {
493        String threadName = Thread.currentThread().getName();
494        if (threadName.startsWith(threadNamePrefix)) {
495          return timeQ.poll().longValue() + offset;
496        }
497      }
498      return System.currentTimeMillis();
499    }

602        } catch (InterruptedException e) {
603        }
604      }

606      public RpcCall getRpcCall() {

608      }

610      public void drop() {
611      }
612    };
