261  private void testCompactionAtMergeSize(final String tableName,
262      final long mergeSize, final CompactionType type, final boolean isForceAllFiles,
263      final boolean createDelFiles, final Date date, final MobCompactPartitionPolicy policy,
264      final long mergeSizeMultiFactor)
265      throws Exception {
266    resetConf();
267    init(tableName);
268    int count = 10;
269    // create 10 mob files.
270    createStoreFiles(basePath, family, qf, count, Type.Put, date);
271
272    if (createDelFiles) {
273      // create 10 del files
274      createStoreFiles(basePath, family, qf, count, Type.Delete, date);
275    }
276
277    Calendar calendar =  Calendar.getInstance();
278    Date firstDayOfCurrentWeek = MobUtils.getFirstDayOfWeek(calendar, new Date());
279
280    listFiles();
281    List<String> expectedStartKeys = new ArrayList<>();
282    for(FileStatus file : mobFiles) {
283      if(file.getLen() < mergeSize * mergeSizeMultiFactor) {
284        String fileName = file.getPath().getName();
285        String startKey = fileName.substring(0, 32);
286
287        // If the policy is monthly and files are in current week, they will be skipped
288        // in minor compcation.
289        boolean skipCompaction = false;
290        if (policy == MobCompactPartitionPolicy.MONTHLY) {
291          String fileDateStr = MobFileName.getDateFromName(fileName);
292          Date fileDate;
293          try {
294            fileDate = MobUtils.parseDate(fileDateStr);
295          } catch (ParseException e)  {
296            LOG.warn("Failed to parse date " + fileDateStr, e);
297            fileDate = new Date();
298          }
299          if (!fileDate.before(firstDayOfCurrentWeek)) {
300            skipCompaction = true;
301          }
302        }
303
304        // If it is not an major mob compaction and del files are there,
305        // these mob files wont be compacted.
306        if (isForceAllFiles || (!createDelFiles && !skipCompaction)) {
307          expectedStartKeys.add(startKey);
308        }
309      }
310    }
311
312    // Set the policy
313    this.hcd.setMobCompactPartitionPolicy(policy);
314    // set the mob compaction mergeable threshold
315    conf.setLong(MobConstants.MOB_COMPACTION_MERGEABLE_THRESHOLD, mergeSize);
316    testSelectFiles(tableName, type, isForceAllFiles, expectedStartKeys);
317    // go back to the default daily policy
318    this.hcd.setMobCompactPartitionPolicy(MobCompactPartitionPolicy.DAILY);
319  }

403   * Tests the selectFiles
404   * @param tableName the table name
405   * @param type the expected compaction type
406   * @param isForceAllFiles whether all the mob files are selected
407   * @param expected the expected start keys
408   */
409  private void testSelectFiles(String tableName, final CompactionType type,
410    final boolean isForceAllFiles, final List<String> expected) throws IOException {
411    PartitionedMobCompactor compactor = new PartitionedMobCompactor(conf, fs,
412      TableName.valueOf(tableName), hcd, pool) {

414      public List<Path> compact(List<FileStatus> files, boolean isForceAllFiles)

417          return null;
418        }
419        PartitionedMobCompactionRequest request = select(files, isForceAllFiles);
420        // assert the compaction type
421        Assert.assertEquals(type, request.type);
422        // assert get the right partitions
423        compareCompactedPartitions(expected, request.compactionPartitions);
424        // assert get the right del files
425        compareDelFiles(request.delFiles);
426        return null;
427      }
428    };
429    compactor.compact(allFiles, isForceAllFiles);
430  }

455      }

457    compactor.compact(allFiles, isForceAllFiles);

478  private void compareCompactedPartitions(List<String> expected,

517  private void createStoreFiles(Path basePath, String family, String qualifier, int count,
518      Type type, boolean sameStartKey, final Date date) throws IOException {
519    HFileContext meta = new HFileContextBuilder().withBlockSize(8 * 1024).build();
520    String startKey = "row_";
521    MobFileName mobFileName = null;
522    for (int i = 0; i < count; i++) {
523      byte[] startRow;
524      if (sameStartKey) {
525        // When creating multiple files under one partition, suffix needs to be different.
526        startRow = Bytes.toBytes(startKey);
527        mobSuffix = UUID.randomUUID().toString().replaceAll("-", "");
528        delSuffix = UUID.randomUUID().toString().replaceAll("-", "") + "_del";
529      } else {
530        startRow = Bytes.toBytes(startKey + i);
531      }
532      if(type.equals(Type.Delete)) {
533        mobFileName = MobFileName.create(startRow, MobUtils.formatDate(date), delSuffix);
534      }
535      if(type.equals(Type.Put)){
536        mobFileName = MobFileName.create(startRow, MobUtils.formatDate(date), mobSuffix);
537      }
538      StoreFileWriter mobFileWriter = new StoreFileWriter.Builder(conf, cacheConf, fs)
539      .withFileContext(meta).withFilePath(new Path(basePath, mobFileName.getFileName())).build();
540      writeStoreFile(mobFileWriter, startRow, Bytes.toBytes(family), Bytes.toBytes(qualifier),
541          type, (i+1)*1000);
542    }
543  }

545  /**
546   * Writes data to store file.
547   * @param writer the store file writer
548   * @param row the row key
549   * @param family the family name
550   * @param qualifier the column qualifier
551   * @param type the key type
552   * @param size the size of value
553   */
554  private static void writeStoreFile(final StoreFileWriter writer, byte[]row, byte[] family,
555      byte[] qualifier, Type type, int size) throws IOException {
556    long now = System.currentTimeMillis();
557    try {
558      byte[] dummyData = new byte[size];
559      new Random().nextBytes(dummyData);
560      writer.append(new KeyValue(row, family, qualifier, now, type, dummyData));
561    } finally {
562      writer.close();
563    }
564  }

566  /**
567   * Gets the number of del cell in the del files
568   * @param paths the del file paths
569   * @return the cell size
570   */
571  private int countDelCellsInDelFiles(List<Path> paths) throws IOException {
572    List<StoreFile> sfs = new ArrayList<StoreFile>();
573    int size = 0;
574    for(Path path : paths) {
575      StoreFile sf = new StoreFile(fs, path, conf, cacheConf, BloomType.NONE);
576      sfs.add(sf);
577    }
578    List scanners = StoreFileScanner.getScannersForStoreFiles(sfs, false, true,
579        false, false, HConstants.LATEST_TIMESTAMP);
580    Scan scan = new Scan();
581    scan.setMaxVersions(hcd.getMaxVersions());
582    long timeToPurgeDeletes = Math.max(conf.getLong("hbase.hstore.time.to.purge.deletes", 0), 0);
583    long ttl = HStore.determineTTLFromFamily(hcd);
584    ScanInfo scanInfo = new ScanInfo(conf, hcd, ttl, timeToPurgeDeletes, CellComparator.COMPARATOR);
585    StoreScanner scanner = new StoreScanner(scan, scanInfo, ScanType.COMPACT_RETAIN_DELETES, null,
586        scanners, 0L, HConstants.LATEST_TIMESTAMP);
587    List<Cell> results = new ArrayList<>();
588    boolean hasMore = true;
589
590    while (hasMore) {
591      hasMore = scanner.next(results);
592      size += results.size();
593      results.clear();
594    }
595    scanner.close();
596    return size;
597  }

599  private static ExecutorService createThreadPool() {
600    int maxThreads = 10;
601    long keepAliveTime = 60;
602    final SynchronousQueue<Runnable> queue = new SynchronousQueue<Runnable>();
603    ThreadPoolExecutor pool = new ThreadPoolExecutor(1, maxThreads, keepAliveTime,
604      TimeUnit.SECONDS, queue, Threads.newDaemonThreadFactory("MobFileCompactionChore"),
605      new RejectedExecutionHandler() {
606        @Override
607        public void rejectedExecution(Runnable r, ThreadPoolExecutor executor) {
608          try {
609            // waiting for a thread to pick up instead of throwing exceptions.
610            queue.put(r);
611          } catch (InterruptedException e) {
612            throw new RejectedExecutionException(e);
613          }
614        }
615      });
616    ((ThreadPoolExecutor) pool).allowCoreThreadTimeOut(true);
617    return pool;
618  }
