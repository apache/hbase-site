018package org.apache.hadoop.hbase.mob;
019
020import static org.junit.Assert.assertEquals;
021
022import java.util.Random;
023
024import org.apache.hadoop.fs.FileStatus;
025import org.apache.hadoop.fs.Path;
026import org.apache.hadoop.hbase.HBaseTestingUtility;
027import org.apache.hadoop.hbase.HColumnDescriptor;
028import org.apache.hadoop.hbase.HTableDescriptor;
029import org.apache.hadoop.hbase.TableName;
030import org.apache.hadoop.hbase.client.*;
031import org.apache.hadoop.hbase.testclassification.MediumTests;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.util.ToolRunner;
034import org.junit.After;
035import org.junit.AfterClass;
036import org.junit.Before;
037import org.junit.BeforeClass;
038import org.junit.Test;
039import org.junit.experimental.categories.Category;
040
041@Category(MediumTests.class)
042public class TestExpiredMobFileCleaner {
043
044  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
045  private final static TableName tableName = TableName.valueOf("TestExpiredMobFileCleaner");
046  private final static String family = "family";
047  private final static byte[] row1 = Bytes.toBytes("row1");
048  private final static byte[] row2 = Bytes.toBytes("row2");
049  private final static byte[] qf = Bytes.toBytes("qf");
050
051  private static BufferedMutator table;
052  private static Admin admin;
053
054  @BeforeClass
055  public static void setUpBeforeClass() throws Exception {
056    TEST_UTIL.getConfiguration().setInt("hbase.master.info.port", 0);
057    TEST_UTIL.getConfiguration().setBoolean("hbase.regionserver.info.port.auto", true);
058
059    TEST_UTIL.getConfiguration().setInt("hfile.format.version", 3);
060  }
061
062  @AfterClass
063  public static void tearDownAfterClass() throws Exception {
064
065  }
066
067  @Before
068  public void setUp() throws Exception {
069    TEST_UTIL.startMiniCluster(1);
070  }
071
072  @After
073  public void tearDown() throws Exception {
074    admin.disableTable(tableName);
075    admin.deleteTable(tableName);
076    admin.close();
077    TEST_UTIL.shutdownMiniCluster();
078    TEST_UTIL.getTestFileSystem().delete(TEST_UTIL.getDataTestDir(), true);
079  }
080
081  private void init() throws Exception {
082    HTableDescriptor desc = new HTableDescriptor(tableName);
083    HColumnDescriptor hcd = new HColumnDescriptor(family);
084    hcd.setMobEnabled(true);
085    hcd.setMobThreshold(3L);
086    hcd.setMaxVersions(4);
087    desc.addFamily(hcd);
088
089    admin = TEST_UTIL.getHBaseAdmin();
090    admin.createTable(desc);
091    table = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration())
092            .getBufferedMutator(tableName);
093  }
094
095  private void modifyColumnExpiryDays(int expireDays) throws Exception {
096    HColumnDescriptor hcd = new HColumnDescriptor(family);
097    hcd.setMobEnabled(true);
098    hcd.setMobThreshold(3L);
099    // change ttl as expire days to make some row expired
100    int timeToLive = expireDays * secondsOfDay();
101    hcd.setTimeToLive(timeToLive);
102
103    admin.modifyColumnFamily(tableName, hcd);
104  }
105
106  private void putKVAndFlush(BufferedMutator table, byte[] row, byte[] value, long ts)
107      throws Exception {
108
109    Put put = new Put(row, ts);
110    put.addColumn(Bytes.toBytes(family), qf, value);
111    table.mutate(put);
112
113    table.flush();
114    admin.flush(tableName);
115  }
116
117  /**
118   * Creates a 3 day old hfile and an 1 day old hfile then sets expiry to 2 days.
119   * Verifies that the 3 day old hfile is removed but the 1 day one is still present
120   * after the expiry based cleaner is run.
121   */
122  @Test
123  public void testCleaner() throws Exception {
124    init();
125
126    Path mobDirPath = MobUtils.getMobFamilyPath(TEST_UTIL.getConfiguration(), tableName, family);
127
128    byte[] dummyData = makeDummyData(600);
129    long ts = System.currentTimeMillis() - 3 * secondsOfDay() * 1000; // 3 days before
130    putKVAndFlush(table, row1, dummyData, ts);
131    FileStatus[] firstFiles = TEST_UTIL.getTestFileSystem().listStatus(mobDirPath);
132    //the first mob file
133    assertEquals("Before cleanup without delay 1", 1, firstFiles.length);
134    String firstFile = firstFiles[0].getPath().getName();
135
136    ts = System.currentTimeMillis() - 1 * secondsOfDay() * 1000; // 1 day before
137    putKVAndFlush(table, row2, dummyData, ts);
138    FileStatus[] secondFiles = TEST_UTIL.getTestFileSystem().listStatus(mobDirPath);
139    //now there are 2 mob files
140    assertEquals("Before cleanup without delay 2", 2, secondFiles.length);
141    String f1 = secondFiles[0].getPath().getName();
142    String f2 = secondFiles[1].getPath().getName();
143    String secondFile = f1.equals(firstFile) ? f2 : f1;
144
145    modifyColumnExpiryDays(2); // ttl = 2, make the first row expired
146
147    //run the cleaner
148    String[] args = new String[2];
149    args[0] = tableName.getNameAsString();
150    args[1] = family;
151    ToolRunner.run(TEST_UTIL.getConfiguration(), new ExpiredMobFileCleaner(), args);
152
153    FileStatus[] filesAfterClean = TEST_UTIL.getTestFileSystem().listStatus(mobDirPath);
154    String lastFile = filesAfterClean[0].getPath().getName();
155    //the first mob fie is removed
156    assertEquals("After cleanup without delay 1", 1, filesAfterClean.length);
157    assertEquals("After cleanup without delay 2", secondFile, lastFile);
158  }
159
160  private int secondsOfDay() {
161    return 24 * 3600;
162  }
163
164  private byte[] makeDummyData(int size) {
165    byte [] dummyData = new byte[size];
166    new Random().nextBytes(dummyData);
167    return dummyData;
168  }
169}
