053public class TestMobFileCache {

056  public static final HBaseClassTestRule CLASS_RULE =
057      HBaseClassTestRule.forClass(TestMobFileCache.class);

059  private HBaseTestingUtility UTIL;
060  private HRegion region;

062  private MobFileCache mobFileCache;
063  private Date currentDate = new Date();
064  private static final String TEST_CACHE_SIZE = "2";
065  private static final int EXPECTED_CACHE_SIZE_ZERO = 0;
066  private static final int EXPECTED_CACHE_SIZE_ONE = 1;
067  private static final int EXPECTED_CACHE_SIZE_TWO = 2;
068  private static final int EXPECTED_CACHE_SIZE_THREE = 3;
069  private static final long EXPECTED_REFERENCE_ONE = 1;
070  private static final long EXPECTED_REFERENCE_TWO = 2;
071
072  private static final String TABLE = "tableName";
073  private static final String FAMILY1 = "family1";
074  private static final String FAMILY2 = "family2";
075  private static final String FAMILY3 = "family3";

077  private static final byte[] ROW = Bytes.toBytes("row");
078  private static final byte[] ROW2 = Bytes.toBytes("row2");
079  private static final byte[] VALUE = Bytes.toBytes("value");
080  private static final byte[] VALUE2 = Bytes.toBytes("value2");
081  private static final byte[] QF1 = Bytes.toBytes("qf1");
082  private static final byte[] QF2 = Bytes.toBytes("qf2");
083  private static final byte[] QF3 = Bytes.toBytes("qf3");

123  /**
124   * Create the mob store file.
125   */
126  private Path createMobStoreFile(String family) throws IOException {
127    return createMobStoreFile(HBaseConfiguration.create(), family);
128  }
129
130  /**
131   * Create the mob store file
132   */
133  private Path createMobStoreFile(Configuration conf, String family) throws IOException {
134    ColumnFamilyDescriptor columnFamilyDescriptor =
135      ColumnFamilyDescriptorBuilder
136        .newBuilder(Bytes.toBytes(family))
137        .setMaxVersions(4)
138        .setMobEnabled(true).build();
139    return createMobStoreFile(columnFamilyDescriptor);
140  }
141
142  /**
143   * Create the mob store file
144   */
145  private Path createMobStoreFile(ColumnFamilyDescriptor columnFamilyDescriptor)
146      throws IOException {
147    // Setting up a Store
148    TableName tn = TableName.valueOf(TABLE);
149    TableDescriptorBuilder tableDescriptorBuilder =
150      TableDescriptorBuilder.newBuilder(tn);
151    tableDescriptorBuilder.setColumnFamily(columnFamilyDescriptor);
152    HMobStore mobStore = (HMobStore) region.getStore(columnFamilyDescriptor.getName());
153    KeyValue key1 = new KeyValue(ROW, columnFamilyDescriptor.getName(), QF1, 1, VALUE);
154    KeyValue key2 = new KeyValue(ROW, columnFamilyDescriptor.getName(), QF2, 1, VALUE);
155    KeyValue key3 = new KeyValue(ROW2, columnFamilyDescriptor.getName(), QF3, 1, VALUE2);
156    KeyValue[] keys = new KeyValue[] { key1, key2, key3 };
157    int maxKeyCount = keys.length;
158    HRegionInfo regionInfo = new HRegionInfo(tn);
159    StoreFileWriter mobWriter = mobStore.createWriterInTmp(currentDate,
160      maxKeyCount, columnFamilyDescriptor.getCompactionCompressionType(),
161      regionInfo.getStartKey(), false);
162    Path mobFilePath = mobWriter.getPath();
163    String fileName = mobFilePath.getName();
164    mobWriter.append(key1);
165    mobWriter.append(key2);
166    mobWriter.append(key3);
167    mobWriter.close();
168    String targetPathName = MobUtils.formatDate(currentDate);
169    Path targetPath = new Path(mobStore.getPath(), targetPathName);
170    mobStore.commitFile(mobFilePath, targetPath);
171    return new Path(targetPath, fileName);
172  }
173
174  @Test
175  public void testMobFileCache() throws Exception {
176    FileSystem fs = FileSystem.get(conf);
177    Path file1Path = createMobStoreFile(FAMILY1);
178    Path file2Path = createMobStoreFile(FAMILY2);
179    Path file3Path = createMobStoreFile(FAMILY3);
180
181    CacheConfig cacheConf = new CacheConfig(conf);
182    // Before open one file by the MobFileCache
183    assertEquals(EXPECTED_CACHE_SIZE_ZERO, mobFileCache.getCacheSize());
184    // Open one file by the MobFileCache
185    CachedMobFile cachedMobFile1 = (CachedMobFile) mobFileCache.openFile(
186        fs, file1Path, cacheConf);
187    assertEquals(EXPECTED_CACHE_SIZE_ONE, mobFileCache.getCacheSize());
188    assertNotNull(cachedMobFile1);
189    assertEquals(EXPECTED_REFERENCE_TWO, cachedMobFile1.getReferenceCount());
190
191    // The evict is also managed by a schedule thread pool.
192    // And its check period is set as 3600 seconds by default.
193    // This evict should get the lock at the most time
194    mobFileCache.evict();  // Cache not full, evict it
195    assertEquals(EXPECTED_CACHE_SIZE_ONE, mobFileCache.getCacheSize());
196    assertEquals(EXPECTED_REFERENCE_TWO, cachedMobFile1.getReferenceCount());
197
198    mobFileCache.evictFile(file1Path.getName());  // Evict one file
199    assertEquals(EXPECTED_CACHE_SIZE_ZERO, mobFileCache.getCacheSize());
200    assertEquals(EXPECTED_REFERENCE_ONE, cachedMobFile1.getReferenceCount());
201
202    cachedMobFile1.close();  // Close the cached mob file
203
204    // Reopen three cached file
205    cachedMobFile1 = (CachedMobFile) mobFileCache.openFile(
206        fs, file1Path, cacheConf);
207    assertEquals(EXPECTED_CACHE_SIZE_ONE, mobFileCache.getCacheSize());
208    CachedMobFile cachedMobFile2 = (CachedMobFile) mobFileCache.openFile(
209        fs, file2Path, cacheConf);
210    assertEquals(EXPECTED_CACHE_SIZE_TWO, mobFileCache.getCacheSize());
211    CachedMobFile cachedMobFile3 = (CachedMobFile) mobFileCache.openFile(
212        fs, file3Path, cacheConf);
213    // Before the evict
214    // Evict the cache, should close the first file 1
215    assertEquals(EXPECTED_CACHE_SIZE_THREE, mobFileCache.getCacheSize());
216    assertEquals(EXPECTED_REFERENCE_TWO, cachedMobFile1.getReferenceCount());
217    assertEquals(EXPECTED_REFERENCE_TWO, cachedMobFile2.getReferenceCount());
218    assertEquals(EXPECTED_REFERENCE_TWO, cachedMobFile3.getReferenceCount());
219    mobFileCache.evict();
220    assertEquals(EXPECTED_CACHE_SIZE_ONE, mobFileCache.getCacheSize());
221    assertEquals(EXPECTED_REFERENCE_ONE, cachedMobFile1.getReferenceCount());
222    assertEquals(EXPECTED_REFERENCE_ONE, cachedMobFile2.getReferenceCount());
223    assertEquals(EXPECTED_REFERENCE_TWO, cachedMobFile3.getReferenceCount());
224  }
