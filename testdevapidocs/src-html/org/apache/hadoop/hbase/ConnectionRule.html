018package org.apache.hadoop.hbase;
019
020import java.io.IOException;
021import java.util.concurrent.CompletableFuture;
022import java.util.function.Supplier;
023import org.apache.hadoop.hbase.client.AsyncConnection;
024import org.apache.hadoop.hbase.client.Connection;
025import org.junit.ClassRule;
026import org.junit.Rule;
027import org.junit.rules.ExternalResource;
028
029/**
030 * A {@link Rule} that manages the lifecycle of an instance of {@link AsyncConnection}. Can be used
031 * in either the {@link Rule} or {@link ClassRule} positions.
032 * </p>
033 * Use in combination with {@link MiniClusterRule}, for example:
034 *
035 * <pre>{@code
036 *   public class TestMyClass {
037 *     private static final MiniClusterRule miniClusterRule = MiniClusterRule.newBuilder().build();
038 *     private static final ConnectionRule connectionRule =
039 *       ConnectionRule.createAsyncConnectionRule(miniClusterRule::createConnection);
040 *
041 *     @ClassRule
042 *     public static final TestRule rule = RuleChain
043 *       .outerRule(miniClusterRule)
044 *       .around(connectionRule);
045 *   }
046 * }</pre>
047 */
048public final class ConnectionRule extends ExternalResource {
049
050  private final Supplier<Connection> connectionSupplier;
051  private final Supplier<CompletableFuture<AsyncConnection>> asyncConnectionSupplier;
052
053  private Connection connection;
054  private AsyncConnection asyncConnection;
055
056  public static ConnectionRule createConnectionRule(
057    final Supplier<Connection> connectionSupplier
058  ) {
059    return new ConnectionRule(connectionSupplier, null);
060  }
061
062  public static ConnectionRule createAsyncConnectionRule(
063    final Supplier<CompletableFuture<AsyncConnection>> asyncConnectionSupplier
064  ) {
065    return new ConnectionRule(null, asyncConnectionSupplier);
066  }
067
068  public static ConnectionRule createConnectionRule(
069    final Supplier<Connection> connectionSupplier,
070    final Supplier<CompletableFuture<AsyncConnection>> asyncConnectionSupplier
071  ) {
072    return new ConnectionRule(connectionSupplier, asyncConnectionSupplier);
073  }
074
075  private ConnectionRule(
076    final Supplier<Connection> connectionSupplier,
077    final Supplier<CompletableFuture<AsyncConnection>> asyncConnectionSupplier
078  ) {
079    this.connectionSupplier = connectionSupplier;
080    this.asyncConnectionSupplier = asyncConnectionSupplier;
081  }
082
083  public Connection getConnection() {
084    if (connection == null) {
085      throw new IllegalStateException(
086        "ConnectionRule not initialized with a synchronous connection.");
087    }
088    return connection;
089  }
090
091  public AsyncConnection getAsyncConnection() {
092    if (asyncConnection == null) {
093      throw new IllegalStateException(
094        "ConnectionRule not initialized with an asynchronous connection.");
095    }
096    return asyncConnection;
097  }
098
099  @Override
100  protected void before() throws Throwable {
101    if (connectionSupplier != null) {
102      this.connection = connectionSupplier.get();
103    }
104    if (asyncConnectionSupplier != null) {
105      this.asyncConnection = asyncConnectionSupplier.get().join();
106    }
107    if (connection == null && asyncConnection != null) {
108      this.connection = asyncConnection.toConnection();
109    }
110  }
111
112  @Override
113  protected void after() {
114    CompletableFuture<Void> closeConnection = CompletableFuture.runAsync(() -> {
115      if (this.connection != null) {
116        try {
117          connection.close();
118        } catch (IOException e) {
119          throw new RuntimeException(e);
120        }
121      }
122    });
123    CompletableFuture<Void> closeAsyncConnection = CompletableFuture.runAsync(() -> {
124      if (this.asyncConnection != null) {
125        try {
126          asyncConnection.close();
127        } catch (IOException e) {
128          throw new RuntimeException(e);
129        }
130      }
131    });
132    CompletableFuture.allOf(closeConnection, closeAsyncConnection).join();
133  }
134}
