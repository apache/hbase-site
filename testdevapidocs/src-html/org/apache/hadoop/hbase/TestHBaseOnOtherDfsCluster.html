018package org.apache.hadoop.hbase;
019
020import org.apache.hadoop.fs.FileSystem;
021import org.apache.hadoop.fs.Path;
022import org.apache.hadoop.hbase.client.Put;
023import org.apache.hadoop.hbase.client.Table;
024import org.apache.hadoop.hbase.testclassification.MediumTests;
025import org.apache.hadoop.hbase.util.Bytes;
026import org.apache.hadoop.hdfs.MiniDFSCluster;
027import org.junit.Rule;
028import org.junit.Test;
029import org.junit.experimental.categories.Category;
030import org.junit.rules.TestName;
031
032import java.util.UUID;
033
034import static org.junit.Assert.assertEquals;
035import static org.junit.Assert.assertTrue;
036
037/**
038 * Test that an HBase cluster can run on top of an existing MiniDfsCluster
039 */
040@Category(MediumTests.class)
041public class TestHBaseOnOtherDfsCluster {
042  @Rule
043  public TestName name = new TestName();
044
045  @Test
046  public void testOveralyOnOtherCluster() throws Exception {
047    // just run HDFS
048    HBaseTestingUtility util1 = new HBaseTestingUtility();
049    MiniDFSCluster dfs = util1.startMiniDFSCluster(1);
050
051    // run HBase on that HDFS
052    HBaseTestingUtility util2 = new HBaseTestingUtility();
053    // set the dfs
054    util2.setDFSCluster(dfs, false);
055    util2.startMiniCluster();
056
057    //ensure that they are pointed at the same place
058    FileSystem fs = dfs.getFileSystem();
059    FileSystem targetFs = util2.getDFSCluster().getFileSystem();
060    assertFsSameUri(fs, targetFs);
061
062    fs = FileSystem.get(util1.getConfiguration());
063    targetFs = FileSystem.get(util2.getConfiguration());
064    assertFsSameUri(fs, targetFs);
065
066    Path randomFile = new Path("/"+UUID.randomUUID());
067    assertTrue(targetFs.createNewFile(randomFile));
068    assertTrue(fs.exists(randomFile));
069
070    // do a simple create/write to ensure the cluster works as expected
071    byte[] family = Bytes.toBytes("testfamily");
072    final TableName tablename = TableName.valueOf(name.getMethodName());
073    Table table = util2.createTable(tablename, family);
074    Put p = new Put(new byte[] { 1, 2, 3 });
075    p.addColumn(family, null, new byte[] { 1 });
076    table.put(p);
077
078    // shutdown and make sure cleanly shutting down
079    util2.shutdownMiniCluster();
080    util1.shutdownMiniDFSCluster();
081  }
082
083  private void assertFsSameUri(FileSystem sourceFs, FileSystem targetFs) {
084    Path source = new Path(sourceFs.getUri());
085    Path target = new Path(targetFs.getUri());
086    assertEquals(source, target);
087  }
088}
