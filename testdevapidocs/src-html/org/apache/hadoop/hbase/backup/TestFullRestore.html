038  /**
039   * Verify that a single table is restored to a new table
040   * @throws Exception
041   */
042  @Test
043  public void testFullRestoreSingle() throws Exception {
044
045    LOG.info("test full restore on a single table empty table");
046
047    List<TableName> tables = Lists.newArrayList(table1);
048    String backupId = fullTableBackup(tables);
049    assertTrue(checkSucceeded(backupId));
050
051    LOG.info("backup complete");
052
053    TableName[] tableset = new TableName[] { table1 };
054    TableName[] tablemap = new TableName[] { table1_restore };
055    BackupAdmin client = getBackupAdmin();
056    client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupId, false,
057      tableset, tablemap, false));
058    HBaseAdmin hba = TEST_UTIL.getHBaseAdmin();
059    assertTrue(hba.tableExists(table1_restore));
060    TEST_UTIL.deleteTable(table1_restore);
061    hba.close();
062  }
