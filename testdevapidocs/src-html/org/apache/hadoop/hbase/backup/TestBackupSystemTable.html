090  public void testUpdateReadDeleteBackupStatus() throws IOException {
091    BackupInfo ctx = createBackupInfo();
092    table.updateBackupInfo(ctx);
093    BackupInfo readCtx = table.readBackupInfo(ctx.getBackupId());
094    assertTrue(compare(ctx, readCtx));
095    // try fake backup id
096    readCtx = table.readBackupInfo("fake");
097    assertNull(readCtx);
098    // delete backup info
099    table.deleteBackupInfo(ctx.getBackupId());
100    readCtx = table.readBackupInfo(ctx.getBackupId());
101    assertNull(readCtx);
102    cleanBackupTable();
103  }

114  private void cleanBackupTable() throws IOException {
115    Admin admin = UTIL.getHBaseAdmin();
116    admin.disableTable(BackupSystemTable.getTableName(conf));
117    admin.truncateTable(BackupSystemTable.getTableName(conf), true);
118    if (admin.isTableDisabled(BackupSystemTable.getTableName(conf))) {
119      admin.enableTable(BackupSystemTable.getTableName(conf));
120    }
121  }

134

185  @Test
186  public void testRegionServerLastLogRollResults() throws IOException {
187    String[] servers = new String[] { "server1", "server2", "server3" };
188    Long[] timestamps = new Long[] { 100L, 102L, 107L };
189
190    for (int i = 0; i < servers.length; i++) {
191      table.writeRegionServerLastLogRollResult(servers[i], timestamps[i], "root");
192    }
193
194    HashMap<String, Long> result = table.readRegionServerLastLogRollResult("root");
195    assertTrue(servers.length == result.size());
196    Set<String> keys = result.keySet();
197    String[] keysAsArray = new String[keys.size()];
198    keys.toArray(keysAsArray);
199    Arrays.sort(keysAsArray);
200
201    for (int i = 0; i < keysAsArray.length; i++) {
202      assertEquals(keysAsArray[i], servers[i]);
203      Long ts1 = timestamps[i];
204      Long ts2 = result.get(keysAsArray[i]);
205      assertEquals(ts1, ts2);
206    }
207
208    cleanBackupTable();
209  }

211  @Test
212  public void testIncrementalBackupTableSet() throws IOException {
213    TreeSet<TableName> tables1 = new TreeSet<>();
214
215    tables1.add(TableName.valueOf("t1"));
216    tables1.add(TableName.valueOf("t2"));
217    tables1.add(TableName.valueOf("t3"));
218
219    TreeSet<TableName> tables2 = new TreeSet<>();
220
221    tables2.add(TableName.valueOf("t3"));
222    tables2.add(TableName.valueOf("t4"));
223    tables2.add(TableName.valueOf("t5"));
224
225    table.addIncrementalBackupTableSet(tables1, "root");
226    BackupSystemTable table = new BackupSystemTable(conn);
227    TreeSet<TableName> res1 = (TreeSet<TableName>) table.getIncrementalBackupTableSet("root");
228    assertTrue(tables1.size() == res1.size());
229    Iterator<TableName> desc1 = tables1.descendingIterator();
230    Iterator<TableName> desc2 = res1.descendingIterator();
231    while (desc1.hasNext()) {
232      assertEquals(desc1.next(), desc2.next());
233    }
234
235    table.addIncrementalBackupTableSet(tables2, "root");
236    TreeSet<TableName> res2 = (TreeSet<TableName>) table.getIncrementalBackupTableSet("root");
237    assertTrue((tables2.size() + tables1.size() - 1) == res2.size());
238
239    tables1.addAll(tables2);
240
241    desc1 = tables1.descendingIterator();
242    desc2 = res2.descendingIterator();
243
244    while (desc1.hasNext()) {
245      assertEquals(desc1.next(), desc2.next());
246    }
247    cleanBackupTable();
248
249  }

252  public void testRegionServerLogTimestampMap() throws IOException {
253    TreeSet<TableName> tables = new TreeSet<>();
254
255    tables.add(TableName.valueOf("t1"));
256    tables.add(TableName.valueOf("t2"));
257    tables.add(TableName.valueOf("t3"));
258
259    HashMap<String, Long> rsTimestampMap = new HashMap<String, Long>();
260
261    rsTimestampMap.put("rs1:100", 100L);
262    rsTimestampMap.put("rs2:100", 101L);
263    rsTimestampMap.put("rs3:100", 103L);
264
265    table.writeRegionServerLogTimestamp(tables, rsTimestampMap, "root");
266
267    HashMap<TableName, HashMap<String, Long>> result = table.readLogTimestampMap("root");
268
269    assertTrue(tables.size() == result.size());
270
271    for (TableName t : tables) {
272      HashMap<String, Long> rstm = result.get(t);
273      assertNotNull(rstm);
274      assertEquals(rstm.get("rs1:100"), new Long(100L));
275      assertEquals(rstm.get("rs2:100"), new Long(101L));
276      assertEquals(rstm.get("rs3:100"), new Long(103L));
277    }
278
279    Set<TableName> tables1 = new TreeSet<>();
280
281    tables1.add(TableName.valueOf("t3"));
282    tables1.add(TableName.valueOf("t4"));
283    tables1.add(TableName.valueOf("t5"));
284
285    HashMap<String, Long> rsTimestampMap1 = new HashMap<String, Long>();
286
287    rsTimestampMap1.put("rs1:100", 200L);
288    rsTimestampMap1.put("rs2:100", 201L);
289    rsTimestampMap1.put("rs3:100", 203L);
290
291    table.writeRegionServerLogTimestamp(tables1, rsTimestampMap1, "root");
292
293    result = table.readLogTimestampMap("root");
294
295    assertTrue(5 == result.size());
296
297    for (TableName t : tables) {
298      HashMap<String, Long> rstm = result.get(t);
299      assertNotNull(rstm);
300      if (t.equals(TableName.valueOf("t3")) == false) {

302        assertEquals(rstm.get("rs2:100"), new Long(101L));
303        assertEquals(rstm.get("rs3:100"), new Long(103L));
304      } else {
305        assertEquals(rstm.get("rs1:100"), new Long(200L));
306        assertEquals(rstm.get("rs2:100"), new Long(201L));
307        assertEquals(rstm.get("rs3:100"), new Long(203L));
308      }
309    }
310
311    for (TableName t : tables1) {
312      HashMap<String, Long> rstm = result.get(t);
313      assertNotNull(rstm);
314      assertEquals(rstm.get("rs1:100"), new Long(200L));
315      assertEquals(rstm.get("rs2:100"), new Long(201L));
316      assertEquals(rstm.get("rs3:100"), new Long(203L));

488  private boolean compare(BackupInfo one, BackupInfo two) {
489    return one.getBackupId().equals(two.getBackupId()) && one.getType().equals(two.getType())
490        && one.getBackupRootDir().equals(two.getBackupRootDir())
491        && one.getStartTs() == two.getStartTs() && one.getCompleteTs() == two.getCompleteTs();
492  }
493
494  private BackupInfo createBackupInfo() {
495
496    BackupInfo ctxt =
497        new BackupInfo("backup_" + System.nanoTime(), BackupType.FULL, new TableName[] {
498            TableName.valueOf("t1"), TableName.valueOf("t2"), TableName.valueOf("t3") },
499            "/hbase/backup");
500    ctxt.setStartTs(System.currentTimeMillis());
501    ctxt.setCompleteTs(System.currentTimeMillis() + 1);
502    return ctxt;
503  }
504
505  private List<BackupInfo> createBackupInfoList(int size) {
506    List<BackupInfo> list = new ArrayList<BackupInfo>();
507    for (int i = 0; i < size; i++) {
508      list.add(createBackupInfo());
509      try {
510        Thread.sleep(10);
511      } catch (InterruptedException e) {
512        e.printStackTrace();
513      }
514    }
515    return list;
516  }
