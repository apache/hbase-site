042/**
043 * 1. Create table t1, t2
044 * 2. Load data to t1, t2
045 * 3 Full backup t1, t2
046 * 4 Delete t2
047 * 5 Load data to t1
048 * 6 Incremental backup t1
049 */
050@Category(LargeTests.class)
051public class TestIncrementalBackupDeleteTable extends TestBackupBase {
052  private static final Logger LOG = LoggerFactory.getLogger(TestIncrementalBackupDeleteTable.class);
053
054  // implement all test cases in 1 test since incremental backup/restore has dependencies
055  @Test
056  public void testIncBackupDeleteTable() throws Exception {
057    // #1 - create full backup for all tables
058    LOG.info("create full backup image for all tables");
059
060    List<TableName> tables = Lists.newArrayList(table1, table2);
061    HBaseAdmin admin = null;
062    Connection conn = ConnectionFactory.createConnection(conf1);
063    admin = (HBaseAdmin) conn.getAdmin();
064    BackupAdminImpl client = new BackupAdminImpl(conn);
065
066    BackupRequest request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
067    String backupIdFull = client.backupTables(request);
068
069    assertTrue(checkSucceeded(backupIdFull));
070
071    // #2 - insert some data to table table1
072    HTable t1 = (HTable) conn.getTable(table1);
073    Put p1;
074    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
075      p1 = new Put(Bytes.toBytes("row-t1" + i));
076      p1.addColumn(famName, qualName, Bytes.toBytes("val" + i));
077      t1.put(p1);
078    }
079
080    Assert.assertEquals(TEST_UTIL.countRows(t1), NB_ROWS_IN_BATCH * 2);
081    t1.close();
082
083    // Delete table table2
084    admin.disableTable(table2);
085    admin.deleteTable(table2);
086
087    // #3 - incremental backup for table1
088    tables = Lists.newArrayList(table1);
089    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
090    String backupIdIncMultiple = client.backupTables(request);
091    assertTrue(checkSucceeded(backupIdIncMultiple));
092
093    // #4 - restore full backup for all tables, without overwrite
094    TableName[] tablesRestoreFull = new TableName[] { table1, table2 };
095
096    TableName[] tablesMapFull = new TableName[] { table1_restore, table2_restore };
097
098    client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupIdFull, false,
099      tablesRestoreFull, tablesMapFull, false));
100
101    // #5.1 - check tables for full restore
102    HBaseAdmin hAdmin = TEST_UTIL.getHBaseAdmin();
103    assertTrue(hAdmin.tableExists(table1_restore));
104    assertTrue(hAdmin.tableExists(table2_restore));
105
106    // #5.2 - checking row count of tables for full restore
107    HTable hTable = (HTable) conn.getTable(table1_restore);
108    Assert.assertEquals(TEST_UTIL.countRows(hTable), NB_ROWS_IN_BATCH);
109    hTable.close();
110
111    hTable = (HTable) conn.getTable(table2_restore);
112    Assert.assertEquals(TEST_UTIL.countRows(hTable), NB_ROWS_IN_BATCH);
113    hTable.close();
114
115    // #6 - restore incremental backup for table1
116    TableName[] tablesRestoreIncMultiple = new TableName[] { table1 };
117    TableName[] tablesMapIncMultiple = new TableName[] { table1_restore };
118    client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupIdIncMultiple,
119      false, tablesRestoreIncMultiple, tablesMapIncMultiple, true));
120
121    hTable = (HTable) conn.getTable(table1_restore);
122    Assert.assertEquals(TEST_UTIL.countRows(hTable), NB_ROWS_IN_BATCH * 2);
123    hTable.close();
124    admin.close();
125    conn.close();
126  }
