045/**
046 * Create multiple backups for two tables: table1, table2 then perform 1 delete
047 */
048@Category(LargeTests.class)
049public class TestBackupMultipleDeletes extends TestBackupBase {
050
051  @ClassRule
052  public static final HBaseClassTestRule CLASS_RULE =
053      HBaseClassTestRule.forClass(TestBackupMultipleDeletes.class);
054
055  private static final Logger LOG = LoggerFactory.getLogger(TestBackupMultipleDeletes.class);
056
057  @Test
058  public void testBackupMultipleDeletes() throws Exception {
059    // #1 - create full backup for all tables
060    LOG.info("create full backup image for all tables");
061    List<TableName> tables = Lists.newArrayList(table1, table2);
062    HBaseAdmin admin = null;
063    Connection conn = ConnectionFactory.createConnection(conf1);
064    admin = (HBaseAdmin) conn.getAdmin();
065    BackupAdmin client = new BackupAdminImpl(conn);
066    BackupRequest request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
067    String backupIdFull = client.backupTables(request);
068    assertTrue(checkSucceeded(backupIdFull));
069    // #2 - insert some data to table table1
070    HTable t1 = (HTable) conn.getTable(table1);
071    Put p1;
072    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
073      p1 = new Put(Bytes.toBytes("row-t1" + i));
074      p1.addColumn(famName, qualName, Bytes.toBytes("val" + i));
075      t1.put(p1);
076    }
077    Assert.assertEquals(TEST_UTIL.countRows(t1), NB_ROWS_IN_BATCH * 2);
078    t1.close();
079    // #3 - incremental backup for table1
080    tables = Lists.newArrayList(table1);
081    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
082    String backupIdInc1 = client.backupTables(request);
083    assertTrue(checkSucceeded(backupIdInc1));
084    // #4 - insert some data to table table2
085    HTable t2 = (HTable) conn.getTable(table2);
086    Put p2 = null;
087    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
088      p2 = new Put(Bytes.toBytes("row-t2" + i));
089      p2.addColumn(famName, qualName, Bytes.toBytes("val" + i));
090      t2.put(p2);
091    }
092    // #5 - incremental backup for table1, table2
093    tables = Lists.newArrayList(table1, table2);
094    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
095    String backupIdInc2 = client.backupTables(request);
096    assertTrue(checkSucceeded(backupIdInc2));
097    // #6 - insert some data to table table1
098    t1 = (HTable) conn.getTable(table1);
099    for (int i = NB_ROWS_IN_BATCH; i < 2 * NB_ROWS_IN_BATCH; i++) {
100      p1 = new Put(Bytes.toBytes("row-t1" + i));
101      p1.addColumn(famName, qualName, Bytes.toBytes("val" + i));
102      t1.put(p1);
103    }
104    // #7 - incremental backup for table1
105    tables = Lists.newArrayList(table1);
106    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
107    String backupIdInc3 = client.backupTables(request);
108    assertTrue(checkSucceeded(backupIdInc3));
109    // #8 - insert some data to table table2
110    t2 = (HTable) conn.getTable(table2);
111    for (int i = NB_ROWS_IN_BATCH; i < 2 * NB_ROWS_IN_BATCH; i++) {
112      p2 = new Put(Bytes.toBytes("row-t1" + i));
113      p2.addColumn(famName, qualName, Bytes.toBytes("val" + i));
114      t2.put(p2);
115    }
116    // #9 - incremental backup for table1, table2
117    tables = Lists.newArrayList(table1, table2);
118    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
119    String backupIdInc4 = client.backupTables(request);
120    assertTrue(checkSucceeded(backupIdInc4));
121    // #10 full backup for table3
122    tables = Lists.newArrayList(table3);
123    request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
124    String backupIdFull2 = client.backupTables(request);
125    assertTrue(checkSucceeded(backupIdFull2));
126    // #11 - incremental backup for table3
127    tables = Lists.newArrayList(table3);
128    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
129    String backupIdInc5 = client.backupTables(request);
130    assertTrue(checkSucceeded(backupIdInc5));
131    LOG.error("Delete backupIdInc2");
132    client.deleteBackups(new String[] { backupIdInc2 });
133    LOG.error("Delete backupIdInc2 done");
134    List<BackupInfo> list = client.getHistory(100);
135    // First check number of backup images before and after
136    assertEquals(4, list.size());
137    // then verify that no backupIdInc2,3,4
138    Set<String> ids = new HashSet<String>();
139    ids.add(backupIdInc2);
140    ids.add(backupIdInc3);
141    ids.add(backupIdInc4);
142    for (BackupInfo info : list) {
143      String backupId = info.getBackupId();
144      if (ids.contains(backupId)) {
145        assertTrue(false);
146      }
147    }
148    // Verify that backupInc5 contains only table3
149    boolean found = false;
150    for (BackupInfo info : list) {
151      String backupId = info.getBackupId();
152      if (backupId.equals(backupIdInc5)) {
153        assertTrue(info.getTables().size() == 1);
154        assertEquals(table3, info.getTableNames().get(0));
155        found = true;
156      }
157    }
158    assertTrue(found);
159    admin.close();
160    conn.close();
161  }
