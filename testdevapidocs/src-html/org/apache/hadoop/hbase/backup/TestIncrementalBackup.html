053@Category(LargeTests.class)
054@RunWith(Parameterized.class)
055public class TestIncrementalBackup extends TestBackupBase {
056  private static final Log LOG = LogFactory.getLog(TestIncrementalBackup.class);
057
058  @Parameterized.Parameters
059  public static Collection<Object[]> data() {
060    provider = "multiwal";
061    List<Object[]> params = new ArrayList<Object[]>();
062    params.add(new Object[] { Boolean.TRUE });
063    return params;
064  }
065
066  public TestIncrementalBackup(Boolean b) {
067  }
068
069  // implement all test cases in 1 test since incremental backup/restore has dependencies
070  @Test
071  public void TestIncBackupRestore() throws Exception {
072
073    int ADD_ROWS = 99;
074    // #1 - create full backup for all tables
075    LOG.info("create full backup image for all tables");
076
077    List<TableName> tables = Lists.newArrayList(table1, table2);
078    final byte[] fam3Name = Bytes.toBytes("f3");
079    table1Desc.addFamily(new HColumnDescriptor(fam3Name));
080    HBaseTestingUtility.modifyTableSync(TEST_UTIL.getAdmin(), table1Desc);
081
082    Connection conn = ConnectionFactory.createConnection(conf1);
083    int NB_ROWS_FAM3 = 6;
084    insertIntoTable(conn, table1, fam3Name, 3, NB_ROWS_FAM3).close();
085
086    HBaseAdmin admin = null;
087    admin = (HBaseAdmin) conn.getAdmin();
088    BackupAdminImpl client = new BackupAdminImpl(conn);
089
090    BackupRequest request = createBackupRequest(BackupType.FULL, tables, BACKUP_ROOT_DIR);
091    String backupIdFull = client.backupTables(request);
092
093    assertTrue(checkSucceeded(backupIdFull));
094
095    // #2 - insert some data to table
096    HTable t1 = insertIntoTable(conn, table1, famName, 1, ADD_ROWS);
097    LOG.debug("writing " + ADD_ROWS + " rows to " + table1);
098
099    Assert.assertEquals(TEST_UTIL.countRows(t1), NB_ROWS_IN_BATCH + ADD_ROWS + NB_ROWS_FAM3);
100    t1.close();
101    LOG.debug("written " + ADD_ROWS + " rows to " + table1);
102
103    HTable t2 = (HTable) conn.getTable(table2);
104    Put p2;
105    for (int i = 0; i < 5; i++) {
106      p2 = new Put(Bytes.toBytes("row-t2" + i));
107      p2.addColumn(famName, qualName, Bytes.toBytes("val" + i));
108      t2.put(p2);
109    }
110
111    Assert.assertEquals(TEST_UTIL.countRows(t2), NB_ROWS_IN_BATCH + 5);
112    t2.close();
113    LOG.debug("written " + 5 + " rows to " + table2);
114    // split table1
115    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
116    List<HRegion> regions = cluster.getRegions(table1);
117
118    byte[] name = regions.get(0).getRegionInfo().getRegionName();
119    long startSplitTime = EnvironmentEdgeManager.currentTime();
120    try {
121      admin.splitRegion(name);
122    } catch (IOException e) {
123      //although split fail, this may not affect following check
124      //In old split without AM2, if region's best split key is not found,
125      //there are not exception thrown. But in current API, exception
126      //will be thrown.
127      LOG.debug("region is not splittable, because " + e);
128    }
129
130    while (!admin.isTableAvailable(table1)) {
131      Thread.sleep(100);
132    }
133
134    long endSplitTime = EnvironmentEdgeManager.currentTime();
135
136    // split finished
137    LOG.debug("split finished in =" + (endSplitTime - startSplitTime));
138
139    // #3 - incremental backup for multiple tables
140    tables = Lists.newArrayList(table1, table2);
141    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
142    String backupIdIncMultiple = client.backupTables(request);
143    assertTrue(checkSucceeded(backupIdIncMultiple));
144
145    // add column family f2 to table1
146    final byte[] fam2Name = Bytes.toBytes("f2");
147    table1Desc.addFamily(new HColumnDescriptor(fam2Name));
148    // drop column family f3
149    table1Desc.removeFamily(fam3Name);
150    HBaseTestingUtility.modifyTableSync(TEST_UTIL.getAdmin(), table1Desc);
151
152    int NB_ROWS_FAM2 = 7;
153    HTable t3 = insertIntoTable(conn, table1, fam2Name, 2, NB_ROWS_FAM2);
154    t3.close();
155
156    // #3 - incremental backup for multiple tables
157    request = createBackupRequest(BackupType.INCREMENTAL, tables, BACKUP_ROOT_DIR);
158    String backupIdIncMultiple2 = client.backupTables(request);
159    assertTrue(checkSucceeded(backupIdIncMultiple2));
160
161    // #4 - restore full backup for all tables, without overwrite
162    TableName[] tablesRestoreFull = new TableName[] { table1, table2 };
163
164    TableName[] tablesMapFull = new TableName[] { table1_restore, table2_restore };
165
166    LOG.debug("Restoring full " + backupIdFull);
167    client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupIdFull, false,
168      tablesRestoreFull, tablesMapFull, false));
169
170    // #5.1 - check tables for full restore
171    HBaseAdmin hAdmin = TEST_UTIL.getHBaseAdmin();
172    assertTrue(hAdmin.tableExists(table1_restore));
173    assertTrue(hAdmin.tableExists(table2_restore));
174
175    hAdmin.close();
176
177    // #5.2 - checking row count of tables for full restore
178    HTable hTable = (HTable) conn.getTable(table1_restore);
179    Assert.assertEquals(TEST_UTIL.countRows(hTable), NB_ROWS_IN_BATCH + NB_ROWS_FAM3);
180    hTable.close();
181
182    hTable = (HTable) conn.getTable(table2_restore);
183    Assert.assertEquals(TEST_UTIL.countRows(hTable), NB_ROWS_IN_BATCH);
184    hTable.close();
185
186    // #6 - restore incremental backup for multiple tables, with overwrite
187    TableName[] tablesRestoreIncMultiple = new TableName[] { table1, table2 };
188    TableName[] tablesMapIncMultiple = new TableName[] { table1_restore, table2_restore };
189    client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupIdIncMultiple2,
190      false, tablesRestoreIncMultiple, tablesMapIncMultiple, true));
191
192    hTable = (HTable) conn.getTable(table1_restore);
193    LOG.debug("After incremental restore: " + hTable.getDescriptor());
194    LOG.debug("f1 has " + TEST_UTIL.countRows(hTable, famName) + " rows");
195    Assert.assertEquals(TEST_UTIL.countRows(hTable, famName), NB_ROWS_IN_BATCH + ADD_ROWS);
196    LOG.debug("f2 has " + TEST_UTIL.countRows(hTable, fam2Name) + " rows");
197    Assert.assertEquals(TEST_UTIL.countRows(hTable, fam2Name), NB_ROWS_FAM2);
198    hTable.close();
199
200    hTable = (HTable) conn.getTable(table2_restore);
201    Assert.assertEquals(TEST_UTIL.countRows(hTable), NB_ROWS_IN_BATCH + 5);
202    hTable.close();
203
204    admin.close();
205    conn.close();
206
207  }
