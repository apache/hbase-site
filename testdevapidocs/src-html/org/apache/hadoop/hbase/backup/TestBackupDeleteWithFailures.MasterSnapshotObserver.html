051/**
052 * This class is only a base for other integration-level backup tests. Do not add tests here.
053 * TestBackupSmallTests is where tests that don't require bring machines up/down should go All other
054 * tests should have their own classes and extend this one
055 */
056@Category(LargeTests.class)
057public class TestBackupDeleteWithFailures extends TestBackupBase{
058
059  private static final Log LOG = LogFactory.getLog(TestBackupDeleteWithFailures.class);
060
061
062
063  public static enum Failure {
064    NO_FAILURES,
065    PRE_SNAPSHOT_FAILURE,
066    PRE_DELETE_SNAPSHOT_FAILURE,
067    POST_DELETE_SNAPSHOT_FAILURE
068  }
069
070  public static class MasterSnapshotObserver implements MasterObserver {
071
072
073    List<Failure> failures = new ArrayList<Failure>();
074
075    public void setFailures(Failure ... f) {
076      failures.clear();
077      for (int i = 0; i < f.length; i++) {
078        failures.add(f[i]);
079      }
080    }
081
082    @Override
083    public void preSnapshot(final ObserverContext<MasterCoprocessorEnvironment> ctx,
084        final SnapshotDescription snapshot, final HTableDescriptor hTableDescriptor)
085        throws IOException
086    {
087       if (failures.contains(Failure.PRE_SNAPSHOT_FAILURE)) {
088         throw new IOException ("preSnapshot");
089       }
090    }
091
092    @Override
093    public void preDeleteSnapshot(ObserverContext<MasterCoprocessorEnvironment> ctx,
094        SnapshotDescription snapshot) throws IOException {
095      if (failures.contains(Failure.PRE_DELETE_SNAPSHOT_FAILURE)) {
096        throw new IOException ("preDeleteSnapshot");
097      }
098    }
099
100    @Override
101    public void postDeleteSnapshot(ObserverContext<MasterCoprocessorEnvironment> ctx,
102        SnapshotDescription snapshot) throws IOException {
103      if (failures.contains(Failure.POST_DELETE_SNAPSHOT_FAILURE)) {
104        throw new IOException ("postDeleteSnapshot");
105      }
106    }
107
108  }
109
110  /**
111   * @throws java.lang.Exception
112   */
113  @Override
114  @Before
115  public void setUp() throws Exception {
116    conf1.set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY,
117      MasterSnapshotObserver.class.getName());
118    conf1.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 1);
119    super.setUp();
120  }
121
122
123  private MasterSnapshotObserver getMasterSnapshotObserver() {
124    return (MasterSnapshotObserver)TEST_UTIL.getHBaseCluster().getMaster()
125      .getMasterCoprocessorHost().findCoprocessor(MasterSnapshotObserver.class.getName());
126  }
127
128  @Test
129  public void testBackupDeleteWithFailures() throws Exception
130  {
131     testBackupDeleteWithFailuresAfter(1, Failure.PRE_DELETE_SNAPSHOT_FAILURE);
132     testBackupDeleteWithFailuresAfter(0, Failure.POST_DELETE_SNAPSHOT_FAILURE);
133     testBackupDeleteWithFailuresAfter(1, Failure.PRE_SNAPSHOT_FAILURE);
134  }
135
136  private void testBackupDeleteWithFailuresAfter(int expected, Failure ...failures) throws Exception {
137    LOG.info("test repair backup delete on a single table with data and failures "+ failures[0]);
138    List<TableName> tableList = Lists.newArrayList(table1);
139    String backupId = fullTableBackup(tableList);
140    assertTrue(checkSucceeded(backupId));
141    LOG.info("backup complete");
142    String[] backupIds = new String[] { backupId };
143    BackupSystemTable table = new BackupSystemTable(TEST_UTIL.getConnection());
144    BackupInfo info = table.readBackupInfo(backupId);
145    Path path = new Path(info.getBackupRootDir(), backupId);
146    FileSystem fs = FileSystem.get(path.toUri(), conf1);
147    assertTrue(fs.exists(path));
148
149    Connection conn = TEST_UTIL.getConnection();
150    Admin admin = conn.getAdmin();
151    MasterSnapshotObserver observer = getMasterSnapshotObserver();
152
153    observer.setFailures(failures);
154    try {
155      getBackupAdmin().deleteBackups(backupIds);
156    } catch(IOException e) {
157      if(expected != 1) assertTrue(false);
158    }
159
160    // Verify that history length == expected after delete failure
161    assertTrue (table.getBackupHistory().size() == expected);
162
163    String[] ids = table.getListOfBackupIdsFromDeleteOperation();
164
165    // Verify that we still have delete record in backup system table
166    if(expected == 1) {
167      assertTrue(ids.length == 1);
168      assertTrue(ids[0].equals(backupId));
169    } else {
170      assertNull(ids);
171    }
172
173    // Now run repair command to repair "failed" delete operation
174    String[] args = new String[] {"repair"};
175
176    observer.setFailures(Failure.NO_FAILURES);
177
178    // Run repair
179    int ret = ToolRunner.run(conf1, new BackupDriver(), args);
180    assertTrue(ret == 0);
181    // Verify that history length == 0
182    assertTrue (table.getBackupHistory().size() == 0);
183    ids = table.getListOfBackupIdsFromDeleteOperation();
184
185    // Verify that we do not have delete record in backup system table
186    assertNull(ids);
187
188    table.close();
189    admin.close();
190  }
