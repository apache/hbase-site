034@Category({MiscTests.class, SmallTests.class})
035public class TestServerName {
036  @Test
037  public void testGetHostNameMinusDomain() {
038    assertEquals("2607:f0d0:1002:51::4",
039      ServerName.getHostNameMinusDomain("2607:f0d0:1002:51::4"));
040    assertEquals("2607:f0d0:1002:0051:0000:0000:0000:0004",
041        ServerName.getHostNameMinusDomain("2607:f0d0:1002:0051:0000:0000:0000:0004"));
042    assertEquals("1.1.1.1", ServerName.getHostNameMinusDomain("1.1.1.1"));
043    assertEquals("x", ServerName.getHostNameMinusDomain("x"));
044    assertEquals("x", ServerName.getHostNameMinusDomain("x.y.z"));
045    assertEquals("asf000", ServerName.getHostNameMinusDomain("asf000.sp2.ygridcore.net"));
046    ServerName sn = ServerName.valueOf("asf000.sp2.ygridcore.net", 1, 1);
047    assertEquals("asf000.sp2.ygridcore.net,1,1", sn.toString());
048  }
049
050  @Test
051  public void testShortString() {
052    ServerName sn = ServerName.valueOf("asf000.sp2.ygridcore.net", 1, 1);
053    assertEquals("asf000:1", sn.toShortString());
054    sn = ServerName.valueOf("2607:f0d0:1002:0051:0000:0000:0000:0004", 1, 1);
055    assertEquals("2607:f0d0:1002:0051:0000:0000:0000:0004:1", sn.toShortString());
056    sn = ServerName.valueOf("1.1.1.1", 1, 1);
057    assertEquals("1.1.1.1:1", sn.toShortString());
058  }
059
060  @Test
061  public void testRegexPatterns() {
062    assertTrue(Pattern.matches(Addressing.VALID_PORT_REGEX, "123"));
063    assertFalse(Pattern.matches(Addressing.VALID_PORT_REGEX, ""));
064    assertTrue(ServerName.SERVERNAME_PATTERN.matcher("www1.example.org,1234,567").matches());
065    ServerName.parseServerName("a.b.c,58102,1319771740322");
066    ServerName.parseServerName("192.168.1.199,58102,1319771740322");
067    ServerName.parseServerName("a.b.c:58102");
068    ServerName.parseServerName("192.168.1.199:58102");
069  }
070
071  @Test public void testParseOfBytes() {
072    final String snStr = "www.example.org,1234,5678";
073    ServerName sn = ServerName.valueOf(snStr);
074    byte [] versionedBytes = sn.getVersionedBytes();
075    assertEquals(sn.toString(), ServerName.parseVersionedServerName(versionedBytes).toString());
076    final String hostnamePortStr = sn.getHostAndPort();
077    byte [] bytes = Bytes.toBytes(hostnamePortStr);
078    String expecting =
079      hostnamePortStr.replace(":", ServerName.SERVERNAME_SEPARATOR) +
080      ServerName.SERVERNAME_SEPARATOR + ServerName.NON_STARTCODE;
081    assertEquals(expecting, ServerName.parseVersionedServerName(bytes).toString());
082  }
083
084  @Test
085  public void testServerName() {
086    ServerName sn = ServerName.valueOf("www.example.org", 1234, 5678);
087    ServerName sn2 = ServerName.valueOf("www.example.org", 1234, 5678);
088    ServerName sn3 = ServerName.valueOf("www.example.org", 1234, 56789);
089    assertTrue(sn.equals(sn2));
090    assertFalse(sn.equals(sn3));
091    assertEquals(sn.hashCode(), sn2.hashCode());
092    assertNotSame(sn.hashCode(), sn3.hashCode());
093    assertEquals(sn.toString(),
094      ServerName.valueOf("www.example.org", 1234, 5678).toString());
095    assertEquals(sn.toString(),
096      ServerName.valueOf("www.example.org:1234", 5678).toString());
097    assertEquals(sn.toString(),
098      "www.example.org" + ServerName.SERVERNAME_SEPARATOR + "1234" +
099      ServerName.SERVERNAME_SEPARATOR + "5678");
100  }
101
102  @Test
103  public void testHostNameCaseSensitivity() {
104    ServerName lower = ServerName.valueOf("www.example.org", 1234, 5678);
105    ServerName upper = ServerName.valueOf("www.EXAMPLE.org", 1234, 5678);
106    assertEquals(0, lower.compareTo(upper));
107    assertEquals(0, upper.compareTo(lower));
108    assertEquals(lower.hashCode(), upper.hashCode());
109    assertTrue(lower.equals(upper));
110    assertTrue(upper.equals(lower));
111    assertTrue(ServerName.isSameAddress(lower, upper));
112  }
