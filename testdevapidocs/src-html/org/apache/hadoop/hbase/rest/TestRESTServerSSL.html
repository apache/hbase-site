045@Category({ RestTests.class, MediumTests.class})
046public class TestRESTServerSSL {
047
048  @ClassRule
049  public static final HBaseClassTestRule CLASS_RULE =
050      HBaseClassTestRule.forClass(TestRESTServerSSL.class);
051
052  private static final String KEY_STORE_PASSWORD = "myKSPassword";
053  private static final String TRUST_STORE_PASSWORD = "myTSPassword";
054
055  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
056  private static final HBaseRESTTestingUtility REST_TEST_UTIL = new HBaseRESTTestingUtility();
057  private static Client sslClient;
058  private static File keyDir;
059  private Configuration conf;
060
061  @BeforeClass
062  public static void beforeClass() throws Exception {
063    // Workaround for jdk8 252 bug. See https://github.com/bcgit/bc-java/issues/941
064    // Below is a workaround described in above URL. Issue fingered first in comments in
065    // HBASE-25920 Support Hadoop 3.3.1
066    AlgorithmId.get("PBEWithSHA1AndDESede");
067    keyDir = initKeystoreDir();
068    KeyPair keyPair = KeyStoreTestUtil.generateKeyPair("RSA");
069    X509Certificate serverCertificate = KeyStoreTestUtil.generateCertificate(
070      "CN=localhost, O=server", keyPair, 30, "SHA1withRSA");
071
072    generateTrustStore("jks", serverCertificate);
073    generateTrustStore("jceks", serverCertificate);
074    generateTrustStore("pkcs12", serverCertificate);
075
076    generateKeyStore("jks", keyPair, serverCertificate);
077    generateKeyStore("jceks", keyPair, serverCertificate);
078    generateKeyStore("pkcs12", keyPair, serverCertificate);
079
080    TEST_UTIL.startMiniCluster();
081  }
082
083  @AfterClass
084  public static void afterClass() throws Exception {
085    // this will also delete the generated test keystore / teststore files,
086    // as we were placing them under the dataTestDir used by the minicluster
087    TEST_UTIL.shutdownMiniCluster();
088  }
089
090  @Before
091  public void beforeEachTest() {
092    conf = new Configuration(TEST_UTIL.getConfiguration());
093    conf.set(Constants.REST_SSL_ENABLED, "true");
094    conf.set(Constants.REST_SSL_KEYSTORE_KEYPASSWORD, KEY_STORE_PASSWORD);
095    conf.set(Constants.REST_SSL_KEYSTORE_PASSWORD, KEY_STORE_PASSWORD);
096    conf.set(Constants.REST_SSL_TRUSTSTORE_PASSWORD, TRUST_STORE_PASSWORD);
097  }
098
099  @After
100  public void tearDownAfterTest() {
101    REST_TEST_UTIL.shutdownServletContainer();
102  }
103
104  @Test
105  public void testSslConnection() throws Exception {
106    startRESTServerWithDefaultKeystoreType();
107
108    Response response = sslClient.get("/version", Constants.MIMETYPE_TEXT);
109    assertEquals(200, response.getCode());
110  }
111
112  @Test(expected = org.apache.http.client.ClientProtocolException.class)
113  public void testNonSslClientDenied() throws Exception {
114    startRESTServerWithDefaultKeystoreType();
115
116    Cluster localCluster = new Cluster().add("localhost", REST_TEST_UTIL.getServletPort());
117    Client nonSslClient = new Client(localCluster, false);
118
119    nonSslClient.get("/version");
120  }

123  public void testSslConnectionUsingKeystoreFormatJKS() throws Exception {

148  private static File initKeystoreDir() {
149    String dataTestDir = TEST_UTIL.getDataTestDir().toString();
150    File keystoreDir = new File(dataTestDir, TestRESTServerSSL.class.getSimpleName() + "_keys");
151    keystoreDir.mkdirs();
152    return keystoreDir;
153  }
154
155  private static void generateKeyStore(String keyStoreType, KeyPair keyPair,
156    X509Certificate serverCertificate) throws Exception {
157    String keyStorePath = getKeystoreFilePath(keyStoreType);
158    KeyStoreTestUtil.createKeyStore(keyStorePath, KEY_STORE_PASSWORD, KEY_STORE_PASSWORD,
159      "serverKS", keyPair.getPrivate(), serverCertificate, keyStoreType);
160  }
161
162  private static void generateTrustStore(String trustStoreType, X509Certificate serverCertificate)
163    throws Exception {
164    String trustStorePath = getTruststoreFilePath(trustStoreType);
165    KeyStoreTestUtil.createTrustStore(trustStorePath, TRUST_STORE_PASSWORD, "serverTS",
166      serverCertificate, trustStoreType);
167  }
168
169  private static String getKeystoreFilePath(String keyStoreType) {
170    return String.format("%s/serverKS.%s", keyDir.getAbsolutePath(), keyStoreType);
171  }
172
173  private static String getTruststoreFilePath(String trustStoreType) {
174    return String.format("%s/serverTS.%s", keyDir.getAbsolutePath(), trustStoreType);
175  }
176
177  private void startRESTServerWithDefaultKeystoreType() throws Exception {
178    conf.set(Constants.REST_SSL_KEYSTORE_STORE, getKeystoreFilePath("jks"));
179    conf.set(Constants.REST_SSL_TRUSTSTORE_STORE, getTruststoreFilePath("jks"));
180
181    REST_TEST_UTIL.startServletContainer(conf);
182    Cluster localCluster = new Cluster().add("localhost", REST_TEST_UTIL.getServletPort());
183    sslClient = new Client(localCluster, getTruststoreFilePath("jks"),
184      Optional.of(TRUST_STORE_PASSWORD), Optional.empty());
185  }
186
187  private void startRESTServer(String storeType) throws Exception {
188    conf.set(Constants.REST_SSL_KEYSTORE_TYPE, storeType);
189    conf.set(Constants.REST_SSL_KEYSTORE_STORE, getKeystoreFilePath(storeType));
190
191    conf.set(Constants.REST_SSL_TRUSTSTORE_STORE, getTruststoreFilePath(storeType));
192    conf.set(Constants.REST_SSL_TRUSTSTORE_TYPE, storeType);
193
194    REST_TEST_UTIL.startServletContainer(conf);
195    Cluster localCluster = new Cluster().add("localhost", REST_TEST_UTIL.getServletPort());
196    sslClient = new Client(localCluster, getTruststoreFilePath(storeType),
197                           Optional.of(TRUST_STORE_PASSWORD), Optional.of(storeType));
198  }
