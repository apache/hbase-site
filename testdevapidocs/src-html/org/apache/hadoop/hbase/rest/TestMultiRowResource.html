056@Category({RestTests.class, MediumTests.class})
057@RunWith(Parameterized.class)
058public class TestMultiRowResource {
059
060  private static final TableName TABLE = TableName.valueOf("TestRowResource");
061  private static final String CFA = "a";
062  private static final String CFB = "b";
063  private static final String COLUMN_1 = CFA + ":1";
064  private static final String COLUMN_2 = CFB + ":2";
065  private static final String ROW_1 = "testrow5";
066  private static final String VALUE_1 = "testvalue5";
067  private static final String ROW_2 = "testrow6";
068  private static final String VALUE_2 = "testvalue6";
069
070
071  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
072  private static final HBaseRESTTestingUtility REST_TEST_UTIL = new HBaseRESTTestingUtility();
073
074  private static Client client;
075  private static JAXBContext context;
076  private static Marshaller marshaller;
077  private static Unmarshaller unmarshaller;
078  private static Configuration conf;
079
080  private static Header extraHdr = null;
081  private static boolean csrfEnabled = true;
082
083  @Parameterized.Parameters
084  public static Collection<Object[]> data() {
085    List<Object[]> params = new ArrayList<Object[]>();
086    params.add(new Object[] {Boolean.TRUE});
087    params.add(new Object[] {Boolean.FALSE});
088    return params;
089  }
090
091  public TestMultiRowResource(Boolean csrf) {
092    csrfEnabled = csrf;
093  }
094
095
096  @BeforeClass
097  public static void setUpBeforeClass() throws Exception {
098    conf = TEST_UTIL.getConfiguration();
099    conf.setBoolean(RESTServer.REST_CSRF_ENABLED_KEY, csrfEnabled);
100    extraHdr = new BasicHeader(RESTServer.REST_CSRF_CUSTOM_HEADER_DEFAULT, "");
101    TEST_UTIL.startMiniCluster();
102    REST_TEST_UTIL.startServletContainer(conf);
103    context = JAXBContext.newInstance(
104            CellModel.class,
105            CellSetModel.class,
106            RowModel.class);
107    marshaller = context.createMarshaller();
108    unmarshaller = context.createUnmarshaller();
109    client = new Client(new Cluster().add("localhost", REST_TEST_UTIL.getServletPort()));
110    Admin admin = TEST_UTIL.getHBaseAdmin();
111    if (admin.tableExists(TABLE)) {
112      return;
113    }
114    HTableDescriptor htd = new HTableDescriptor(TABLE);
115    htd.addFamily(new HColumnDescriptor(CFA));
116    htd.addFamily(new HColumnDescriptor(CFB));
117    admin.createTable(htd);
118  }
119
120  @AfterClass
121  public static void tearDownAfterClass() throws Exception {
122    REST_TEST_UTIL.shutdownServletContainer();
123    TEST_UTIL.shutdownMiniCluster();
124  }

127  @Test
128  public void testMultiCellGetJSON() throws IOException, JAXBException {
129    String row_5_url = "/" + TABLE + "/" + ROW_1 + "/" + COLUMN_1;
130    String row_6_url = "/" + TABLE + "/" + ROW_2 + "/" + COLUMN_2;
131
132
133    StringBuilder path = new StringBuilder();
134    path.append("/");
135    path.append(TABLE);
136    path.append("/multiget/?row=");
137    path.append(ROW_1);
138    path.append("&row=");
139    path.append(ROW_2);
140
141    if (csrfEnabled) {
142      Response response = client.post(row_5_url, Constants.MIMETYPE_BINARY, Bytes.toBytes(VALUE_1));
143      assertEquals(400, response.getCode());
144    }
145
146    client.post(row_5_url, Constants.MIMETYPE_BINARY, Bytes.toBytes(VALUE_1), extraHdr);
147    client.post(row_6_url, Constants.MIMETYPE_BINARY, Bytes.toBytes(VALUE_2), extraHdr);
148
149
150    Response response = client.get(path.toString(), Constants.MIMETYPE_JSON);
151    assertEquals(response.getCode(), 200);
152    assertEquals(Constants.MIMETYPE_JSON, response.getHeader("content-type"));
153
154    client.delete(row_5_url, extraHdr);
155    client.delete(row_6_url, extraHdr);
156
157  }

186  @Test
187  public void testMultiCellGetWithColsJSON() throws IOException, JAXBException {
188    String row_5_url = "/" + TABLE + "/" + ROW_1 + "/" + COLUMN_1;
189    String row_6_url = "/" + TABLE + "/" + ROW_2 + "/" + COLUMN_2;
190
191    StringBuilder path = new StringBuilder();
192    path.append("/");
193    path.append(TABLE);
194    path.append("/multiget");
195    path.append("/" + COLUMN_1 + "," + CFB);
196    path.append("?row=");
197    path.append(ROW_1);
198    path.append("&row=");
199    path.append(ROW_2);
200
201    client.post(row_5_url, Constants.MIMETYPE_BINARY, Bytes.toBytes(VALUE_1), extraHdr);
202    client.post(row_6_url, Constants.MIMETYPE_BINARY, Bytes.toBytes(VALUE_2), extraHdr);
203
204    Response response = client.get(path.toString(), Constants.MIMETYPE_JSON);
205    assertEquals(response.getCode(), 200);
206    ObjectMapper mapper =
207        new JacksonProvider().locateMapper(CellSetModel.class, MediaType.APPLICATION_JSON_TYPE);
208    CellSetModel cellSet = (CellSetModel) mapper.readValue(response.getBody(), CellSetModel.class);
209    assertEquals(2, cellSet.getRows().size());
210    assertEquals(ROW_1, Bytes.toString(cellSet.getRows().get(0).getKey()));
211    assertEquals(VALUE_1, Bytes.toString(cellSet.getRows().get(0).getCells().get(0).getValue()));
212    assertEquals(ROW_2, Bytes.toString(cellSet.getRows().get(1).getKey()));
213    assertEquals(VALUE_2, Bytes.toString(cellSet.getRows().get(1).getCells().get(0).getValue()));
214
215    client.delete(row_5_url, extraHdr);
216    client.delete(row_6_url, extraHdr);
217
218  }
