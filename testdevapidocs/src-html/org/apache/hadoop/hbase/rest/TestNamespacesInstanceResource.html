114  private static byte[] toXML(NamespacesInstanceModel model) throws JAXBException {
115    StringWriter writer = new StringWriter();
116    context.createMarshaller().marshal(model, writer);
117    return Bytes.toBytes(writer.toString());
118  }
119
120  @SuppressWarnings("unchecked")
121  private static <T> T fromXML(byte[] content)
122      throws JAXBException {
123    return (T) context.createUnmarshaller().unmarshal(new ByteArrayInputStream(content));
124  }

126  private NamespaceDescriptor findNamespace(Admin admin, String namespaceName) throws IOException{
127    NamespaceDescriptor[] nd = admin.listNamespaceDescriptors();
128    for (NamespaceDescriptor namespaceDescriptor : nd) {
129      if (namespaceDescriptor.getName().equals(namespaceName)) {
130        return namespaceDescriptor;
131      }
132    }
133    return null;
134  }
135
136  private void checkNamespaceProperties(NamespaceDescriptor nd, Map<String,String> testProps){
137    checkNamespaceProperties(nd.getConfiguration(), testProps);
138  }
139
140  private void checkNamespaceProperties(Map<String,String> namespaceProps,
141      Map<String,String> testProps){
142    assertTrue(namespaceProps.size() == testProps.size());
143    for (String key: testProps.keySet()) {
144      assertEquals(testProps.get(key), namespaceProps.get(key));
145    }
146  }
147
148  private void checkNamespaceTables(List<TableModel> namespaceTables, List<String> testTables){
149    assertEquals(namespaceTables.size(), testTables.size());
150    for (TableModel namespaceTable : namespaceTables) {
151      String tableName = namespaceTable.getName();
152      assertTrue(testTables.contains(tableName));
153    }
154  }

177  @Test
178  public void testGetNamespaceTablesAndCannotDeleteNamespace() throws IOException, JAXBException {
179    Admin admin = TEST_UTIL.getAdmin();
180    String nsName = "TestNamespacesInstanceResource5";
181    Response response;
182
183    // Create namespace via admin.
184    NamespaceDescriptor.Builder nsBuilder = NamespaceDescriptor.create(nsName);
185    NamespaceDescriptor nsd = nsBuilder.build();
186    nsd.setConfiguration("key1", "value1");
187    admin.createNamespace(nsd);
188
189    // Create two tables via admin.
190    TableName tn1 = TableName.valueOf(nsName + ":table1");
191    TableDescriptorBuilder tableDescriptorBuilder =
192      TableDescriptorBuilder.newBuilder(tn1);
193    ColumnFamilyDescriptor columnFamilyDescriptor =
194      ColumnFamilyDescriptorBuilder.newBuilder(Bytes.toBytes("cf1")).build();
195    tableDescriptorBuilder.setColumnFamily(columnFamilyDescriptor);
196    admin.createTable(tableDescriptorBuilder.build());
197    TableName tn2 = TableName.valueOf(nsName + ":table2");
198    tableDescriptorBuilder = TableDescriptorBuilder.newBuilder(tn2);
199    tableDescriptorBuilder.setColumnFamily(columnFamilyDescriptor);
200    admin.createTable(tableDescriptorBuilder.build());
201
202    Map<String, String> nsProperties = new HashMap<>();
203    nsProperties.put("key1", "value1");
204    List<String> nsTables = Arrays.asList("table1", "table2");
205
206    // Check get namespace properties as XML, JSON and Protobuf.
207    String namespacePath = "/namespaces/" + nsName;
208    response = client.get(namespacePath);
209    assertEquals(200, response.getCode());
210
211    response = client.get(namespacePath, Constants.MIMETYPE_XML);
212    assertEquals(200, response.getCode());
213    NamespacesInstanceModel model = fromXML(response.getBody());
214    checkNamespaceProperties(model.getProperties(), nsProperties);
215
216    response = client.get(namespacePath, Constants.MIMETYPE_JSON);
217    assertEquals(200, response.getCode());
218    model = jsonMapper.readValue(response.getBody(), NamespacesInstanceModel.class);
219    checkNamespaceProperties(model.getProperties(), nsProperties);
220
221    response = client.get(namespacePath, Constants.MIMETYPE_PROTOBUF);
222    assertEquals(200, response.getCode());
223    model.getObjectFromMessage(response.getBody());
224    checkNamespaceProperties(model.getProperties(), nsProperties);
225
226    // Check get namespace tables as XML, JSON and Protobuf.
227    namespacePath = "/namespaces/" + nsName + "/tables";
228    response = client.get(namespacePath);
229    assertEquals(200, response.getCode());
230
231    response = client.get(namespacePath, Constants.MIMETYPE_XML);
232    assertEquals(200, response.getCode());
233    TableListModel tablemodel = fromXML(response.getBody());
234    checkNamespaceTables(tablemodel.getTables(), nsTables);
235
236    response = client.get(namespacePath, Constants.MIMETYPE_JSON);
237    assertEquals(200, response.getCode());
238    tablemodel = jsonMapper.readValue(response.getBody(), TableListModel.class);
239    checkNamespaceTables(tablemodel.getTables(), nsTables);
240
241    response = client.get(namespacePath, Constants.MIMETYPE_PROTOBUF);
242    assertEquals(200, response.getCode());
243    tablemodel.setTables(new ArrayList<>());
244    tablemodel.getObjectFromMessage(response.getBody());
245    checkNamespaceTables(tablemodel.getTables(), nsTables);
246
247    // Check cannot delete namespace via REST because it contains tables.
248    response = client.delete(namespacePath);
249    namespacePath = "/namespaces/" + nsName;
250    assertEquals(503, response.getCode());
251  }

294  @Test
295  public void testNamespaceCreateAndDeleteXMLAndJSON() throws IOException, JAXBException {
296    String namespacePath1 = "/namespaces/" + NAMESPACE1;
297    String namespacePath2 = "/namespaces/" + NAMESPACE2;
298    NamespacesInstanceModel model1;
299    NamespacesInstanceModel model2;
300    Response response;
301
302    // Check that namespaces don't exist via non-REST call.
303    Admin admin = TEST_UTIL.getAdmin();
304    assertNull(findNamespace(admin, NAMESPACE1));
305    assertNull(findNamespace(admin, NAMESPACE2));
306
307    model1 = testNamespacesInstanceModel.buildTestModel(NAMESPACE1, NAMESPACE1_PROPS);
308    testNamespacesInstanceModel.checkModel(model1, NAMESPACE1, NAMESPACE1_PROPS);
309    model2 = testNamespacesInstanceModel.buildTestModel(NAMESPACE2, NAMESPACE2_PROPS);
310    testNamespacesInstanceModel.checkModel(model2, NAMESPACE2, NAMESPACE2_PROPS);
311
312    // Test cannot PUT (alter) non-existent namespace.
313    response = client.put(namespacePath1, Constants.MIMETYPE_XML, toXML(model1));
314    assertEquals(403, response.getCode());
315    String jsonString = jsonMapper.writeValueAsString(model2);
316    response = client.put(namespacePath2, Constants.MIMETYPE_JSON, Bytes.toBytes(jsonString));
317    assertEquals(403, response.getCode());
318
319    // Test cannot create tables when in read only mode.
320    conf.set("hbase.rest.readonly", "true");
321    response = client.post(namespacePath1, Constants.MIMETYPE_XML, toXML(model1));
322    assertEquals(403, response.getCode());
323    jsonString = jsonMapper.writeValueAsString(model2);
324    response = client.post(namespacePath2, Constants.MIMETYPE_JSON, Bytes.toBytes(jsonString));
325    assertEquals(403, response.getCode());
326    NamespaceDescriptor nd1 = findNamespace(admin, NAMESPACE1);
327    NamespaceDescriptor nd2 = findNamespace(admin, NAMESPACE2);
328    assertNull(nd1);
329    assertNull(nd2);
330    conf.set("hbase.rest.readonly", "false");
331
332    // Create namespace via XML and JSON.
333    response = client.post(namespacePath1, Constants.MIMETYPE_XML, toXML(model1));
334    assertEquals(201, response.getCode());
335    jsonString = jsonMapper.writeValueAsString(model2);
336    response = client.post(namespacePath2, Constants.MIMETYPE_JSON, Bytes.toBytes(jsonString));
337    assertEquals(201, response.getCode());
338    //check passing null content-type with a payload returns 415
339    Header[] nullHeaders = null;
340    response = client.post(namespacePath1, nullHeaders, toXML(model1));
341    assertEquals(415, response.getCode());
342    response = client.post(namespacePath1, nullHeaders, Bytes.toBytes(jsonString));
343    assertEquals(415, response.getCode());
344
345    // Check that created namespaces correctly.
346    nd1 = findNamespace(admin, NAMESPACE1);
347    nd2 = findNamespace(admin, NAMESPACE2);
348    assertNotNull(nd1);
349    assertNotNull(nd2);
350    checkNamespaceProperties(nd1, NAMESPACE1_PROPS);
351    checkNamespaceProperties(nd1, NAMESPACE1_PROPS);
352
353    // Test cannot delete tables when in read only mode.
354    conf.set("hbase.rest.readonly", "true");
355    response = client.delete(namespacePath1);
356    assertEquals(403, response.getCode());
357    response = client.delete(namespacePath2);
358    assertEquals(403, response.getCode());
359    nd1 = findNamespace(admin, NAMESPACE1);
360    nd2 = findNamespace(admin, NAMESPACE2);
361    assertNotNull(nd1);
362    assertNotNull(nd2);
363    conf.set("hbase.rest.readonly", "false");
364
365    // Delete namespaces via XML and JSON.
366    response = client.delete(namespacePath1);
367    assertEquals(200, response.getCode());
368    response = client.delete(namespacePath2);
369    assertEquals(200, response.getCode());
370    nd1 = findNamespace(admin, NAMESPACE1);
371    nd2 = findNamespace(admin, NAMESPACE2);
372    assertNull(nd1);
373    assertNull(nd2);
374  }

376  @Test
377  public void testNamespaceCreateAndDeletePBAndNoBody() throws IOException {
378    String namespacePath3 = "/namespaces/" + NAMESPACE3;
379    String namespacePath4 = "/namespaces/" + NAMESPACE4;
380    NamespacesInstanceModel model3;
381    NamespacesInstanceModel model4;
382    Response response;
383
384    // Check that namespaces don't exist via non-REST call.
385    Admin admin = TEST_UTIL.getAdmin();
386    assertNull(findNamespace(admin, NAMESPACE3));
387    assertNull(findNamespace(admin, NAMESPACE4));
388
389    model3 = testNamespacesInstanceModel.buildTestModel(NAMESPACE3, NAMESPACE3_PROPS);
390    testNamespacesInstanceModel.checkModel(model3, NAMESPACE3, NAMESPACE3_PROPS);
391    model4 = testNamespacesInstanceModel.buildTestModel(NAMESPACE4, NAMESPACE4_PROPS);
392    testNamespacesInstanceModel.checkModel(model4, NAMESPACE4, NAMESPACE4_PROPS);
393
394    //Defines null headers for use in tests where no body content is provided, so that we set
395    // no content-type in the request
396    Header[] nullHeaders = null;
397
398    // Test cannot PUT (alter) non-existent namespace.
399    response = client.put(namespacePath3, nullHeaders, new byte[]{});
400    assertEquals(403, response.getCode());
401    response = client.put(namespacePath4, Constants.MIMETYPE_PROTOBUF,
402      model4.createProtobufOutput());
403    assertEquals(403, response.getCode());
404
405    // Test cannot create tables when in read only mode.
406    conf.set("hbase.rest.readonly", "true");
407    response = client.post(namespacePath3, nullHeaders, new byte[]{});
408    assertEquals(403, response.getCode());
409    response = client.put(namespacePath4, Constants.MIMETYPE_PROTOBUF,
410      model4.createProtobufOutput());
411    assertEquals(403, response.getCode());
412    NamespaceDescriptor nd3 = findNamespace(admin, NAMESPACE3);
413    NamespaceDescriptor nd4 = findNamespace(admin, NAMESPACE4);
414    assertNull(nd3);
415    assertNull(nd4);
416    conf.set("hbase.rest.readonly", "false");
417
418    // Create namespace with no body and binary content type.
419    response = client.post(namespacePath3, nullHeaders, new byte[]{});
420    assertEquals(201, response.getCode());
421    // Create namespace with protobuf content-type.
422    response = client.post(namespacePath4, Constants.MIMETYPE_PROTOBUF,
423      model4.createProtobufOutput());
424    assertEquals(201, response.getCode());
425    //check setting unsupported content-type returns 415
426    response = client.post(namespacePath3, Constants.MIMETYPE_BINARY, new byte[]{});
427    assertEquals(415, response.getCode());
428
429    // Check that created namespaces correctly.
430    nd3 = findNamespace(admin, NAMESPACE3);
431    nd4 = findNamespace(admin, NAMESPACE4);
432    assertNotNull(nd3);
433    assertNotNull(nd4);
434    checkNamespaceProperties(nd3, new HashMap<>());
435    checkNamespaceProperties(nd4, NAMESPACE4_PROPS);
436
437    // Check cannot post tables that already exist.
438    response = client.post(namespacePath3, nullHeaders, new byte[]{});
439    assertEquals(403, response.getCode());
440    response = client.post(namespacePath4, Constants.MIMETYPE_PROTOBUF,
441      model4.createProtobufOutput());
442    assertEquals(403, response.getCode());
443
444    // Check cannot post tables when in read only mode.
445    conf.set("hbase.rest.readonly", "true");
446    response = client.delete(namespacePath3);
447    assertEquals(403, response.getCode());
448    response = client.delete(namespacePath4);
449    assertEquals(403, response.getCode());
450    nd3 = findNamespace(admin, NAMESPACE3);
451    nd4 = findNamespace(admin, NAMESPACE4);
452    assertNotNull(nd3);
453    assertNotNull(nd4);
454    conf.set("hbase.rest.readonly", "false");
455
456    // Delete namespaces via XML and JSON.
457    response = client.delete(namespacePath3);
458    assertEquals(200, response.getCode());
459    response = client.delete(namespacePath4);
460    assertEquals(200, response.getCode());
461    nd3 = findNamespace(admin, NAMESPACE3);
462    nd4 = findNamespace(admin, NAMESPACE4);
463    assertNull(nd3);
464    assertNull(nd4);
465  }
