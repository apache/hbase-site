020package org.apache.hadoop.hbase.rest.model;
021
022import java.util.Iterator;
023
024import org.apache.hadoop.hbase.testclassification.RestTests;
025import org.apache.hadoop.hbase.testclassification.SmallTests;
026
027import org.junit.experimental.categories.Category;
028
029@Category({RestTests.class, SmallTests.class})
030public class TestTableListModel extends TestModelBase<TableListModel> {
031  private static final String TABLE1 = "table1";
032  private static final String TABLE2 = "table2";
033  private static final String TABLE3 = "table3";
034
035  public TestTableListModel() throws Exception {
036    super(TableListModel.class);
037    AS_XML =
038      "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?><TableList><table " +
039          "name=\"table1\"/><table name=\"table2\"/><table name=\"table3\"/></TableList>";
040
041    AS_PB = "CgZ0YWJsZTEKBnRhYmxlMgoGdGFibGUz";
042
043    AS_JSON =
044      "{\"table\":[{\"name\":\"table1\"},{\"name\":\"table2\"},{\"name\":\"table3\"}]}";
045  }
046
047  protected TableListModel buildTestModel() {
048    TableListModel model = new TableListModel();
049    model.add(new TableModel(TABLE1));
050    model.add(new TableModel(TABLE2));
051    model.add(new TableModel(TABLE3));
052    return model;
053  }
054
055  protected void checkModel(TableListModel model) {
056    Iterator<TableModel> tables = model.getTables().iterator();
057    TableModel table = tables.next();
058    assertEquals(table.getName(), TABLE1);
059    table = tables.next();
060    assertEquals(table.getName(), TABLE2);
061    table = tables.next();
062    assertEquals(table.getName(), TABLE3);
063    assertFalse(tables.hasNext());
064  }
065}
