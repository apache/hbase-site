018package org.apache.hadoop.hbase.rest.model;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023import org.apache.hadoop.hbase.HBaseClassTestRule;
024import org.apache.hadoop.hbase.testclassification.RestTests;
025import org.apache.hadoop.hbase.testclassification.SmallTests;
026import org.apache.hadoop.hbase.util.Bytes;
027import org.junit.ClassRule;
028import org.junit.experimental.categories.Category;
029
030@Category({RestTests.class, SmallTests.class})
031public class TestScannerModel extends TestModelBase<ScannerModel> {
032
033  @ClassRule
034  public static final HBaseClassTestRule CLASS_RULE =
035      HBaseClassTestRule.forClass(TestScannerModel.class);
036
037  private static final String PRIVATE = "private";
038  private static final String PUBLIC = "public";
039  private static final byte[] START_ROW = Bytes.toBytes("abracadabra");
040  private static final byte[] END_ROW = Bytes.toBytes("zzyzx");
041  private static final byte[] COLUMN1 = Bytes.toBytes("column1");
042  private static final byte[] COLUMN2 = Bytes.toBytes("column2:foo");
043  private static final long START_TIME = 1245219839331L;
044  private static final long END_TIME = 1245393318192L;
045  private static final int CACHING = 1000;
046  private static final int BATCH = 100;
047  private static final boolean CACHE_BLOCKS = false;
048
049  public TestScannerModel() throws Exception {
050    super(ScannerModel.class);
051    AS_XML = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>"
052        + "<Scanner batch=\"100\" cacheBlocks=\"false\" caching=\"1000\" endRow=\"enp5eng=\" "
053        + "endTime=\"1245393318192\" maxVersions=\"2147483647\" startRow=\"YWJyYWNhZGFicmE=\" "
054        + "startTime=\"1245219839331\">"
055        + "<column>Y29sdW1uMQ==</column><column>Y29sdW1uMjpmb28=</column>"
056        + "<labels>private</labels><labels>public</labels>"
057        + "</Scanner>";
058
059    AS_JSON = "{\"batch\":100,\"caching\":1000,\"cacheBlocks\":false,\"endRow\":\"enp5eng=\","
060        + "\"endTime\":1245393318192,\"maxVersions\":2147483647,\"startRow\":\"YWJyYWNhZGFicmE=\","
061        + "\"startTime\":1245219839331,\"column\":[\"Y29sdW1uMQ==\",\"Y29sdW1uMjpmb28=\"],"
062        +"\"labels\":[\"private\",\"public\"]"
063        +"}";
064
065    AS_PB = "CgthYnJhY2FkYWJyYRIFenp5engaB2NvbHVtbjEaC2NvbHVtbjI6Zm9vIGQo47qL554kMLDi57mf"
066        + "JDj/////B0joB1IHcHJpdmF0ZVIGcHVibGljWAA=";
067  }
068
069  @Override
070  protected ScannerModel buildTestModel() {
071    ScannerModel model = new ScannerModel();
072    model.setStartRow(START_ROW);
073    model.setEndRow(END_ROW);
074    model.addColumn(COLUMN1);
075    model.addColumn(COLUMN2);
076    model.setStartTime(START_TIME);
077    model.setEndTime(END_TIME);
078    model.setBatch(BATCH);
079    model.setCaching(CACHING);
080    model.addLabel(PRIVATE);
081    model.addLabel(PUBLIC);
082    model.setCacheBlocks(CACHE_BLOCKS);
083    return model;
084  }
085
086  @Override
087  protected void checkModel(ScannerModel model) {
088    assertTrue(Bytes.equals(model.getStartRow(), START_ROW));
089    assertTrue(Bytes.equals(model.getEndRow(), END_ROW));
090    boolean foundCol1 = false, foundCol2 = false;
091    for (byte[] column : model.getColumns()) {
092      if (Bytes.equals(column, COLUMN1)) {
093        foundCol1 = true;
094      } else if (Bytes.equals(column, COLUMN2)) {
095        foundCol2 = true;
096      }
097    }
098    assertTrue(foundCol1);
099    assertTrue(foundCol2);
100    assertEquals(START_TIME, model.getStartTime());
101    assertEquals(END_TIME, model.getEndTime());
102    assertEquals(BATCH, model.getBatch());
103    assertEquals(CACHING, model.getCaching());
104    assertEquals(CACHE_BLOCKS, model.getCacheBlocks());
105    boolean foundLabel1 = false;
106    boolean foundLabel2 = false;
107    if (model.getLabels() != null && model.getLabels().size() > 0) {
108      for (String label : model.getLabels()) {
109        if (label.equals(PRIVATE)) {
110          foundLabel1 = true;
111        } else if (label.equals(PUBLIC)) {
112          foundLabel2 = true;
113        }
114      }
115      assertTrue(foundLabel1);
116      assertTrue(foundLabel2);
117    }
118  }
119
120}
