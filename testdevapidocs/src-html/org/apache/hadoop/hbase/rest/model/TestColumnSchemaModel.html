020package org.apache.hadoop.hbase.rest.model;
021
022import org.apache.hadoop.hbase.testclassification.RestTests;
023import org.apache.hadoop.hbase.testclassification.SmallTests;
024import org.junit.experimental.categories.Category;
025
026@Category({RestTests.class, SmallTests.class})
027public class TestColumnSchemaModel extends TestModelBase<ColumnSchemaModel> {
028
029  protected static final String COLUMN_NAME = "testcolumn";
030  protected static final boolean BLOCKCACHE = true;
031  protected static final int BLOCKSIZE = 16384;
032  protected static final String BLOOMFILTER = "NONE";
033  protected static final String COMPRESSION = "GZ";
034  protected static final boolean IN_MEMORY = false;
035  protected static final int TTL = 86400;
036  protected static final int VERSIONS = 1;
037
038  public TestColumnSchemaModel() throws Exception {
039    super(ColumnSchemaModel.class);
040    AS_XML =
041      "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?><ColumnSchema " +
042          "name=\"testcolumn\" BLOCKSIZE=\"16384\" BLOOMFILTER=\"NONE\" BLOCKCACHE=\"true\" " +
043          "COMPRESSION=\"GZ\" VERSIONS=\"1\" TTL=\"86400\" IN_MEMORY=\"false\"/>";
044
045    AS_JSON =
046      "{\"name\":\"testcolumn\",\"BLOCKSIZE\":\"16384\",\"BLOOMFILTER\":\"NONE\"," +
047          "\"BLOCKCACHE\":\"true\",\"COMPRESSION\":\"GZ\",\"VERSIONS\":\"1\"," +
048          "\"TTL\":\"86400\",\"IN_MEMORY\":\"false\"}";
049  }
050
051  protected ColumnSchemaModel buildTestModel() {
052    ColumnSchemaModel model = new ColumnSchemaModel();
053    model.setName(COLUMN_NAME);
054    model.__setBlocksize(BLOCKSIZE);
055    model.__setBloomfilter(BLOOMFILTER);
056    model.__setBlockcache(BLOCKCACHE);
057    model.__setCompression(COMPRESSION);
058    model.__setVersions(VERSIONS);
059    model.__setTTL(TTL);
060    model.__setInMemory(IN_MEMORY);
061    return model;
062  }
063
064  protected void checkModel(ColumnSchemaModel model) {
065    assertEquals(model.getName(), COLUMN_NAME);
066    assertEquals(model.__getBlockcache(), BLOCKCACHE);
067    assertEquals(model.__getBlocksize(), BLOCKSIZE);
068    assertEquals(model.__getBloomfilter(), BLOOMFILTER);
069    assertTrue(model.__getCompression().equalsIgnoreCase(COMPRESSION));
070    assertEquals(model.__getInMemory(), IN_MEMORY);
071    assertEquals(model.__getTTL(), TTL);
072    assertEquals(model.__getVersions(), VERSIONS);
073  }
074
075  public void testFromPB() throws Exception {
076  }
077}
