017package org.apache.hadoop.hbase.rest.client;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertTrue;
021import static org.junit.Assert.fail;
022import static org.mockito.Mockito.mock;
023import static org.mockito.Mockito.when;
024
025import java.io.IOException;
026import javax.xml.bind.UnmarshalException;
027
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.hadoop.hbase.HBaseConfiguration;
031import org.apache.hadoop.hbase.rest.Constants;
032import org.apache.hadoop.hbase.rest.model.StorageClusterVersionModel;
033import org.apache.hadoop.hbase.testclassification.SmallTests;
034import org.apache.hadoop.util.StringUtils;
035import org.junit.Test;
036import org.junit.experimental.categories.Category;
037
038/**
039 * Test class for {@link RemoteAdmin} to verify XML is parsed in a certain manner.
040 */
041@Category(SmallTests.class)
042public class TestXmlParsing {
043  private static final Log LOG = LogFactory.getLog(TestXmlParsing.class);
044
045  @Test
046  public void testParsingClusterVersion() throws Exception {
047    final String xml = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>"
048        + "<ClusterVersion Version=\"2.0.0\"/>";
049    Client client = mock(Client.class);
050    RemoteAdmin admin = new RemoteAdmin(client, HBaseConfiguration.create(), null);
051    Response resp = new Response(200, null, xml.getBytes());
052
053    when(client.get("/version/cluster", Constants.MIMETYPE_XML)).thenReturn(resp);
054
055    StorageClusterVersionModel cv = admin.getClusterVersion();
056    assertEquals("2.0.0", cv.getVersion());
057  }
058
059  @Test
060  public void testFailOnExternalEntities() throws Exception {
061    final String externalEntitiesXml =
062        "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
063        + " <!DOCTYPE foo [ <!ENTITY xxe SYSTEM \"/tmp/foo\"> ] >"
064        + " <ClusterVersion>&xee;</ClusterVersion>";
065    Client client = mock(Client.class);
066    RemoteAdmin admin = new RemoteAdmin(client, HBaseConfiguration.create(), null);
067    Response resp = new Response(200, null, externalEntitiesXml.getBytes());
068
069    when(client.get("/version/cluster", Constants.MIMETYPE_XML)).thenReturn(resp);
070
071    try {
072      admin.getClusterVersion();
073      fail("Expected getClusterVersion() to throw an exception");
074    } catch (IOException e) {
075      assertEquals("Cause of exception ought to be a failure to parse the stream due to our " +
076          "invalid external entity. Make sure this isn't just a false positive due to " +
077          "implementation. see HBASE-19020.", UnmarshalException.class, e.getCause().getClass());
078      final String exceptionText = StringUtils.stringifyException(e);
079      final String expectedText = "\"xee\"";
080      LOG.debug("exception text: '" + exceptionText + "'", e);
081      assertTrue("Exception does not contain expected text", exceptionText.contains(expectedText));
082    }
083  }
084}
