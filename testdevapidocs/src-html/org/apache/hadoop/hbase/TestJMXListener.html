022
023import javax.management.MBeanServerConnection;
024import javax.management.remote.JMXConnector;
025import javax.management.remote.JMXConnectorFactory;
026
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.hbase.coprocessor.CoprocessorHost;
029import org.apache.hadoop.hbase.testclassification.MediumTests;
030import org.apache.hadoop.hbase.testclassification.MiscTests;
031import org.apache.hadoop.hbase.util.JVMClusterUtil;
032import org.junit.AfterClass;
033import org.junit.Assert;
034import static org.junit.Assert.fail;
035import org.junit.BeforeClass;
036import org.junit.Rule;
037import org.junit.Test;
038import org.junit.experimental.categories.Category;
039import org.junit.rules.ExpectedException;
040import org.slf4j.Logger;
041import org.slf4j.LoggerFactory;
042
043
044
045@Category({MiscTests.class, MediumTests.class})
046public class TestJMXListener {
047  private static final Logger LOG = LoggerFactory.getLogger(TestJMXListener.class);
048  private static HBaseTestingUtility UTIL = new HBaseTestingUtility();
049  private static int connectorPort = UTIL.randomFreePort();
050
051  @BeforeClass
052  public static void setupBeforeClass() throws Exception {
053    Configuration conf = UTIL.getConfiguration();
054
055    conf.set(CoprocessorHost.REGIONSERVER_COPROCESSOR_CONF_KEY,
056      JMXListener.class.getName());
057    conf.setInt("regionserver.rmi.registry.port", connectorPort);
058
059    UTIL.startMiniCluster();
060  }
061
062  @AfterClass
063  public static void tearDownAfterClass() throws Exception {
064    UTIL.shutdownMiniCluster();
065  }
066
067  @Test
068  public void testStart() throws Exception {
069    JMXConnector connector = JMXConnectorFactory.connect(
070      JMXListener.buildJMXServiceURL(connectorPort,connectorPort));
071
072    MBeanServerConnection mb = connector.getMBeanServerConnection();
073    String domain = mb.getDefaultDomain();
074    Assert.assertTrue("default domain is not correct",
075      !domain.isEmpty());
076    connector.close();
077
078  }
079
080  //shutdown hbase only. then try connect, IOException expected
081  @Rule
082  public ExpectedException expectedEx = ExpectedException.none();
083  @Test
084  public void testStop() throws Exception {
085    MiniHBaseCluster cluster = UTIL.getHBaseCluster();
086    LOG.info("shutdown hbase cluster...");
087    cluster.shutdown();
088    LOG.info("wait for the hbase cluster shutdown...");
089    cluster.waitUntilShutDown();
090
091    JMXConnector connector = JMXConnectorFactory.newJMXConnector(
092      JMXListener.buildJMXServiceURL(connectorPort,connectorPort), null);
093    expectedEx.expect(IOException.class);
094    connector.connect();
095
096  }
097
098  @Test
099  public void testGetRegionServerCoprocessors() throws Exception {
100    for (JVMClusterUtil.RegionServerThread rs : UTIL.getHBaseCluster().getRegionServerThreads()) {
101      boolean find = false;
102      for (String s : rs.getRegionServer().getRegionServerCoprocessors()) {
103        if (s.equals(JMXListener.class.getSimpleName())) {
104          find = true;
105          break;
106        }
107      }
108      if (!find) {
109        fail("where is the JMXListener?");
110      }
111    }
112  }
113}
