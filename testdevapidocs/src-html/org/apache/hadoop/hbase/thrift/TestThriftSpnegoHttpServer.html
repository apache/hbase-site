078 * Start the HBase Thrift HTTP server on a random port through the command-line
079 * interface and talk to it from client side with SPNEGO security enabled.

099  private static void setupUser(SimpleKdcServer kdc, File keytab, String principal)
100      throws KrbException {
101    kdc.createPrincipal(principal);
102    kdc.exportPrincipal(principal, keytab);
103  }
104
105  private static SimpleKdcServer buildMiniKdc() throws Exception {
106    SimpleKdcServer kdc = new SimpleKdcServer();
107
108    final File target = new File(System.getProperty("user.dir"), "target");
109    File kdcDir = new File(target, TestThriftSpnegoHttpServer.class.getSimpleName());
110    if (kdcDir.exists()) {
111      FileUtils.deleteDirectory(kdcDir);
112    }
113    kdcDir.mkdirs();
114    kdc.setWorkDir(kdcDir);
115
116    kdc.setKdcHost(HConstants.LOCALHOST);
117    int kdcPort = HBaseTestingUtility.randomFreePort();
118    kdc.setAllowTcp(true);
119    kdc.setAllowUdp(false);
120    kdc.setKdcTcpPort(kdcPort);
121
122    LOG.info("Starting KDC server at " + HConstants.LOCALHOST + ":" + kdcPort);
123
124    kdc.init();
125
126    return kdc;
127  }
128
129  private static void addSecurityConfigurations(Configuration conf) {
130    KerberosName.setRules("DEFAULT");
131
132    HBaseKerberosUtils.setKeytabFileForTesting(serverKeytab.getAbsolutePath());
133    HBaseKerberosUtils.setSecuredConfiguration(conf, serverPrincipal, spnegoServerPrincipal);
134
135    conf.setBoolean(THRIFT_SUPPORT_PROXYUSER_KEY, true);
136    conf.setBoolean(USE_HTTP_CONF_KEY, true);
137    conf.set("hadoop.proxyuser.hbase.hosts", "*");
138    conf.set("hadoop.proxyuser.hbase.groups", "*");
139
140    conf.set(THRIFT_KERBEROS_PRINCIPAL_KEY, serverPrincipal);
141    conf.set(THRIFT_KEYTAB_FILE_KEY, serverKeytab.getAbsolutePath());
142    conf.set(THRIFT_SPNEGO_PRINCIPAL_KEY, spnegoServerPrincipal);
143    conf.set(THRIFT_SPNEGO_KEYTAB_FILE_KEY, spnegoServerKeytab.getAbsolutePath());
144  }

146  @BeforeClass
147  public static void setUpBeforeClass() throws Exception {
148    final File target = new File(System.getProperty("user.dir"), "target");
149    assertTrue(target.exists());
150
151    File keytabDir = new File(target, TestThriftSpnegoHttpServer.class.getSimpleName() +
152      "_keytabs");
153    if (keytabDir.exists()) {
154      FileUtils.deleteDirectory(keytabDir);
155    }
156    keytabDir.mkdirs();
157
158    kdc = buildMiniKdc();
159    kdc.start();
160
161    clientPrincipal = "client@" + kdc.getKdcConfig().getKdcRealm();
162    clientKeytab = new File(keytabDir, clientPrincipal + ".keytab");
163    setupUser(kdc, clientKeytab, clientPrincipal);
164
165    serverPrincipal = "hbase/" + HConstants.LOCALHOST + "@" + kdc.getKdcConfig().getKdcRealm();
166    serverKeytab = new File(keytabDir, serverPrincipal.replace('/', '_') + ".keytab");
167    setupUser(kdc, serverKeytab, serverPrincipal);
168
169    spnegoServerPrincipal = "HTTP/" + HConstants.LOCALHOST + "@" + kdc.getKdcConfig().getKdcRealm();
170    spnegoServerKeytab = new File(keytabDir, spnegoServerPrincipal.replace('/', '_') + ".keytab");
171    setupUser(kdc, spnegoServerKeytab, spnegoServerPrincipal);
172
173    TEST_UTIL.getConfiguration().setBoolean(USE_HTTP_CONF_KEY, true);
174    TEST_UTIL.getConfiguration().setBoolean("hbase.table.sanity.checks", false);
175    addSecurityConfigurations(TEST_UTIL.getConfiguration());
176
177    TestThriftHttpServer.setUpBeforeClass();
178  }

193  @Override
194  void talkToThriftServer(String url, int customHeaderSize) throws Exception {
195    // Close httpClient and THttpClient automatically on any failures
196    try (
197        CloseableHttpClient httpClient = createHttpClient();
198        THttpClient tHttpClient = new THttpClient(url, httpClient)
199    ) {
200      tHttpClient.open();
201      if (customHeaderSize > 0) {
202        StringBuilder sb = new StringBuilder();
203        for (int i = 0; i < customHeaderSize; i++) {
204          sb.append("a");
205        }
206        tHttpClient.setCustomHeader(HttpHeaders.USER_AGENT, sb.toString());
207      }
208
209      TProtocol prot = new TBinaryProtocol(tHttpClient);
210      Hbase.Client client = new Hbase.Client(prot);
211      if (!tableCreated) {
212        TestThriftServer.createTestTables(client);
213        tableCreated = true;
214      }
215      TestThriftServer.checkTableList(client);
216    }
217  }
218
219  private CloseableHttpClient createHttpClient() throws Exception {
220    final Subject clientSubject = JaasKrbUtil.loginUsingKeytab(clientPrincipal, clientKeytab);
221    final Set<Principal> clientPrincipals = clientSubject.getPrincipals();
222    // Make sure the subject has a principal
223    assertFalse(clientPrincipals.isEmpty());
224
225    // Get a TGT for the subject (might have many, different encryption types). The first should
226    // be the default encryption type.
227    Set<KerberosTicket> privateCredentials =
228        clientSubject.getPrivateCredentials(KerberosTicket.class);
229    assertFalse(privateCredentials.isEmpty());
230    KerberosTicket tgt = privateCredentials.iterator().next();
231    assertNotNull(tgt);
232
233    // The name of the principal
234    final String clientPrincipalName = clientPrincipals.iterator().next().getName();
235
236    return Subject.doAs(clientSubject, new PrivilegedExceptionAction<CloseableHttpClient>() {
237      @Override
238      public CloseableHttpClient run() throws Exception {
239        // Logs in with Kerberos via GSS
240        GSSManager gssManager = GSSManager.getInstance();
241        // jGSS Kerberos login constant
242        Oid oid = new Oid("1.2.840.113554.1.2.2");
243        GSSName gssClient = gssManager.createName(clientPrincipalName, GSSName.NT_USER_NAME);
244        GSSCredential credential = gssManager.createCredential(gssClient,
245            GSSCredential.DEFAULT_LIFETIME, oid, GSSCredential.INITIATE_ONLY);
246
247        Lookup<AuthSchemeProvider> authRegistry = RegistryBuilder.<AuthSchemeProvider>create()
248            .register(AuthSchemes.SPNEGO, new SPNegoSchemeFactory(true, true))
249            .build();
250
251        BasicCredentialsProvider credentialsProvider = new BasicCredentialsProvider();
252        credentialsProvider.setCredentials(AuthScope.ANY, new KerberosCredentials(credential));
253
254        return HttpClients.custom()
255            .setDefaultAuthSchemeRegistry(authRegistry)
256            .setDefaultCredentialsProvider(credentialsProvider)
257            .build();
258      }
259    });
260  }
