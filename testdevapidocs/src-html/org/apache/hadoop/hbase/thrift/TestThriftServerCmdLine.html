065/**
066 * Start the HBase Thrift server on a random port through the command-line
067 * interface and talk to it from client side.
068 */
069@Category({ClientTests.class, LargeTests.class})
070@RunWith(Parameterized.class)
071public class TestThriftServerCmdLine {
072
073  @ClassRule
074  public static final HBaseClassTestRule CLASS_RULE =
075      HBaseClassTestRule.forClass(TestThriftServerCmdLine.class);

078      LoggerFactory.getLogger(TestThriftServerCmdLine.class);

096  @Parameters
097  public static Collection<Object[]> getParameters() {
098    Collection<Object[]> parameters = new ArrayList<>();
099    for (ImplType implType : ImplType.values()) {
100      for (boolean specifyFramed : new boolean[] {false, true}) {
101        for (boolean specifyBindIP : new boolean[] {false, true}) {
102          if (specifyBindIP && !implType.canSpecifyBindIP) {
103            continue;
104          }
105          for (boolean specifyCompact : new boolean[] {false, true}) {
106            parameters.add(new Object[] {
107              implType, specifyFramed, specifyBindIP, specifyCompact
108            });
109          }
110        }
111      }
112    }
113    return parameters;
114  }

147  private void startCmdLineThread(final String[] args) {
148    LOG.info("Starting HBase Thrift server with command line: " + Joiner.on(" ").join(args));
149
150    cmdLineException = null;
151    cmdLineThread = new Thread(new Runnable() {
152      @Override
153      public void run() {
154        try {
155          thriftServer.run(args);
156        } catch (Exception e) {
157          LOG.error("Error when start thrift server", e);
158          cmdLineException = e;
159        }
160      }
161    });
162    cmdLineThread.setName(ThriftServer.class.getSimpleName() +
163        "-cmdline");
164    cmdLineThread.start();
165  }
166
167  protected ThriftServer createThriftServer() {
168    return new ThriftServer(TEST_UTIL.getConfiguration());
169  }
170
171  private int getRandomPort() {
172    return HBaseTestingUtility.randomFreePort();
173  }
174
175  /**
176   * Server can fail to bind if clashing address. Add retrying until we get a good server.
177   */
178  ThriftServer createBoundServer() {
179    return createBoundServer(false);
180  }
181
182  /**
183   * @param clash This param is just so we can manufacture a port clash so we can test the
184   *   code does the right thing when this happens during actual test runs. Ugly but works.
185   * @see TestBindExceptionHandling#testPortClash()
186   */
187  ThriftServer createBoundServer(boolean clash) {
188    boolean testClashOfFirstPort = clash;
189    List<String> args = new ArrayList<>();
190    ServerSocket ss = null;
191    for (int i = 0; i < 100; i++) {
192      args.clear();
193      if (implType != null) {
194        String serverTypeOption = implType.toString();
195        assertTrue(serverTypeOption.startsWith("-"));
196        args.add(serverTypeOption);
197      }
198      port = getRandomPort();
199      if (testClashOfFirstPort) {
200        // Test what happens if already something bound to the socket.
201        // Occupy the random port we just pulled.
202        try {
203          ss = new ServerSocket();
204          ss.bind(new InetSocketAddress(port));
205        } catch (IOException ioe) {
206          LOG.warn("Failed bind", ioe);
207        }
208        testClashOfFirstPort = false;
209      }
210      args.add("-" + PORT_OPTION);
211      args.add(String.valueOf(port));
212      args.add("-" + INFOPORT_OPTION);
213      int infoPort = getRandomPort();
214      args.add(String.valueOf(infoPort));
215
216      if (specifyFramed) {
217        args.add("-" + FRAMED_OPTION);
218      }
219      if (specifyBindIP) {
220        args.add("-" + BIND_OPTION);
221        args.add(InetAddress.getLoopbackAddress().getHostName());
222      }
223      if (specifyCompact) {
224        args.add("-" + COMPACT_OPTION);
225      }
226      args.add("start");
227
228      thriftServer = createThriftServer();
229      startCmdLineThread(args.toArray(new String[args.size()]));
230      // wait up to 10s for the server to start
231      for (int ii = 0; ii < 100 && (thriftServer.tserver == null); ii++) {
232        Threads.sleep(100);
233      }
234      if (cmdLineException instanceof TTransportException &&
235          cmdLineException.getCause() instanceof BindException) {
236        LOG.info("Trying new port", cmdLineException);
237        cmdLineException =  null;
238        thriftServer.stop();
239        continue;
240      }
241      break;
242    }
243    if (ss != null) {
244      try {
245        ss.close();
246      } catch (IOException ioe) {
247        LOG.warn("Failed close", ioe);
248      }
249    }
250    Class<? extends TServer> expectedClass = implType != null ?
251      implType.serverClass : TBoundedThreadPoolServer.class;
252    assertEquals(expectedClass, thriftServer.tserver.getClass());
253    LOG.info("Server={}", args);
254    return thriftServer;
255  }
256
257  @Test
258  public void testRunThriftServer() throws Exception {
259    ThriftServer thriftServer = createBoundServer();
260    try {
261      talkToThriftServer();
262    } catch (Exception ex) {
263      clientSideException = ex;
264      LOG.info("Exception", ex);
265    } finally {
266      stopCmdLineThread();
267      thriftServer.stop();
268    }
269
270    if (clientSideException != null) {
271      LOG.error("Thrift client threw an exception. Parameters:" +
272          getParametersString(), clientSideException);
273      throw new Exception(clientSideException);
274    }
275  }
276
277  protected static volatile boolean tableCreated = false;
278
279  protected void talkToThriftServer() throws Exception {
280    LOG.info("Talking to port=" + this.port);
281    TSocket sock = new TSocket(InetAddress.getLoopbackAddress().getHostName(), port);
282    TTransport transport = sock;
283    if (specifyFramed || implType.isAlwaysFramed) {
284      transport = new TFramedTransport(transport);
285    }
286
287    sock.open();
288    try {
289      TProtocol prot;
290      if (specifyCompact) {
291        prot = new TCompactProtocol(transport);
292      } else {
293        prot = new TBinaryProtocol(transport);
294      }
295
296      Hbase.Client client = new Hbase.Client(prot);
297      if (!tableCreated){
298        TestThriftServer.createTestTables(client);
299        tableCreated = true;
300      }
301      TestThriftServer.checkTableList(client);
302
303    } finally {
304      sock.close();
305    }
306  }
307
308  private void stopCmdLineThread() throws Exception {
309    LOG.debug("Stopping " + implType.simpleClassName() + " Thrift server");
310    thriftServer.stop();
311    cmdLineThread.join();
312    if (cmdLineException != null) {
313      LOG.error("Command-line invocation of HBase Thrift server threw an " +
314          "exception", cmdLineException);
315      throw new Exception(cmdLineException);
316    }
317  }
