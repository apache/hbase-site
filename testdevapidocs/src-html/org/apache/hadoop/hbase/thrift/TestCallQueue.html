019package org.apache.hadoop.hbase.thrift;
020
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.concurrent.LinkedBlockingQueue;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.hbase.CompatibilitySingletonFactory;
029import org.apache.hadoop.hbase.HBaseTestingUtility;
030import org.apache.hadoop.hbase.test.MetricsAssertHelper;
031import org.apache.hadoop.hbase.testclassification.ClientTests;
032import org.apache.hadoop.hbase.testclassification.SmallTests;
033import org.junit.Test;
034import org.junit.experimental.categories.Category;
035import org.junit.runner.RunWith;
036import org.junit.runners.Parameterized;
037import org.junit.runners.Parameterized.Parameters;
038
039/**
040 * Unit testing for CallQueue, a part of the
041 * org.apache.hadoop.hbase.thrift package.
042 */
043@Category({ClientTests.class, SmallTests.class})
044@RunWith(Parameterized.class)
045public class TestCallQueue {
046
047  private static final Log LOG = LogFactory.getLog(TestCallQueue.class);
048  private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
049
050  private static final MetricsAssertHelper metricsHelper =
051      CompatibilitySingletonFactory.getInstance(MetricsAssertHelper.class);
052
053  private int elementsAdded;
054  private int elementsRemoved;
055
056  @Parameters
057  public static Collection<Object[]> getParameters() {
058    Collection<Object[]> parameters = new ArrayList<>();
059    for (int elementsAdded : new int[] {100, 200, 300}) {
060      for (int elementsRemoved : new int[] {0, 20, 100}) {
061        parameters.add(new Object[]{new Integer(elementsAdded),
062                                    new Integer(elementsRemoved)});
063      }
064    }
065    return parameters;
066  }
067
068  public TestCallQueue(int elementsAdded, int elementsRemoved) {
069    this.elementsAdded = elementsAdded;
070    this.elementsRemoved = elementsRemoved;
071    LOG.debug("elementsAdded:" + elementsAdded +
072              " elementsRemoved:" + elementsRemoved);
073
074  }
075
076  @Test(timeout = 60000)
077  public void testPutTake() throws Exception {
078    ThriftMetrics metrics = createMetrics();
079    CallQueue callQueue = new CallQueue(new LinkedBlockingQueue<>(), metrics);
080    for (int i = 0; i < elementsAdded; ++i) {
081      callQueue.put(createDummyRunnable());
082    }
083    for (int i = 0; i < elementsRemoved; ++i) {
084      callQueue.take();
085    }
086    verifyMetrics(metrics, "timeInQueue_num_ops", elementsRemoved);
087  }
088
089  @Test(timeout = 60000)
090  public void testOfferPoll() throws Exception {
091    ThriftMetrics metrics = createMetrics();
092    CallQueue callQueue = new CallQueue(new LinkedBlockingQueue<>(), metrics);
093    for (int i = 0; i < elementsAdded; ++i) {
094      callQueue.offer(createDummyRunnable());
095    }
096    for (int i = 0; i < elementsRemoved; ++i) {
097      callQueue.poll();
098    }
099    verifyMetrics(metrics, "timeInQueue_num_ops", elementsRemoved);
100  }
101
102  private static ThriftMetrics createMetrics() throws Exception {
103    Configuration conf = UTIL.getConfiguration();
104    ThriftMetrics m = new ThriftMetrics(conf, ThriftMetrics.ThriftServerType.ONE);
105    m.getSource().init();
106    return m;
107  }
108
109
110  private static void verifyMetrics(ThriftMetrics metrics, String name, int expectValue)
111      throws Exception { 
112      metricsHelper.assertCounter(name, expectValue, metrics.getSource());
113  }
114
115  private static Runnable createDummyRunnable() {
116    return new Runnable() {
117      @Override
118      public void run() {
119      }
120    };
121  }
122
123}
