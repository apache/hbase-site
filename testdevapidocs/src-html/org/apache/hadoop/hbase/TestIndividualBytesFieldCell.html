040  @BeforeClass
041  public static void testConstructorAndVerify() {
042    // Immutable inputs
043    byte[] row       = Bytes.toBytes("immutable-row");
044    byte[] family    = Bytes.toBytes("immutable-family");
045    byte[] qualifier = Bytes.toBytes("immutable-qualifier");
046    byte[] value     = Bytes.toBytes("immutable-value");
047    byte[] tags      = Bytes.toBytes("immutable-tags");
048
049    // Other inputs
050    long timestamp = 5000L;
051    long seqId     = 0L;
052    KeyValue.Type type      = KeyValue.Type.Put;
053
054    ic0 = new IndividualBytesFieldCell(row, family, qualifier, timestamp, type, seqId, value, tags);
055    kv0 = new KeyValue(row, family, qualifier, timestamp, type, value, tags);
056
057    // Verify if no local copy is made for row, family, qualifier, value or tags.
058    assertTrue(ic0.getRowArray()       == row);
059    assertTrue(ic0.getFamilyArray()    == family);
060    assertTrue(ic0.getQualifierArray() == qualifier);
061    assertTrue(ic0.getValueArray()     == value);
062    assertTrue(ic0.getTagsArray()      == tags);
063
064    // Verify others.
065    assertEquals(timestamp     , ic0.getTimestamp());
066    assertEquals(seqId         , ic0.getSequenceId());
067    assertEquals(type.getCode(), ic0.getTypeByte());
068
069    // Verify offsets of backing byte arrays are always 0.
070    assertEquals(0, ic0.getRowOffset());
071    assertEquals(0, ic0.getFamilyOffset());
072    assertEquals(0, ic0.getQualifierOffset());
073    assertEquals(0, ic0.getValueOffset());
074    assertEquals(0, ic0.getTagsOffset());
075  }

077  // Verify clone() and deepClone()
078  @Test
079  public void testClone() throws CloneNotSupportedException {
080    // Verify clone. Only shadow copies are made for backing byte arrays.
081    IndividualBytesFieldCell cloned = (IndividualBytesFieldCell) ic0.clone();
082    assertTrue(cloned.getRowArray()       == ic0.getRowArray());
083    assertTrue(cloned.getFamilyArray()    == ic0.getFamilyArray());
084    assertTrue(cloned.getQualifierArray() == ic0.getQualifierArray());
085    assertTrue(cloned.getValueArray()     == ic0.getValueArray());
086    assertTrue(cloned.getTagsArray()      == ic0.getTagsArray());
087
088    // Verify if deep clone returns a KeyValue object
089    assertTrue(ic0.deepClone() instanceof KeyValue);
090  }

092  /**
093   * Verify KeyValue format related functions: write() and getSerializedSize().
094   * Should have the same behaviors as {@link KeyValue}.
095   */
096  @Test
097  public void testWriteIntoKeyValueFormat() throws IOException {
098    // Verify getSerializedSize().
099    assertEquals(kv0.getSerializedSize(true),  ic0.getSerializedSize(true));   // with tags
100    assertEquals(kv0.getSerializedSize(false), ic0.getSerializedSize(false));  // without tags
101
102    // Verify writing into ByteBuffer.
103    ByteBuffer bbufIC = ByteBuffer.allocate(ic0.getSerializedSize(true));
104    ic0.write(bbufIC, 0);
105
106    ByteBuffer bbufKV = ByteBuffer.allocate(kv0.getSerializedSize(true));
107    kv0.write(bbufKV, 0);
108
109    assertTrue(bbufIC.equals(bbufKV));
110
111    // Verify writing into OutputStream.
112    testWriteIntoOutputStream(ic0, kv0, true);   // with tags
113    testWriteIntoOutputStream(ic0, kv0, false);  // without tags
114  }

116  /**
117   * @param ic An instance of IndividualBytesFieldCell to compare.
118   * @param kv An instance of KeyValue to compare.
119   * @param withTags Whether to write tags.
120   * @throws IOException
121   */
122  private void testWriteIntoOutputStream(IndividualBytesFieldCell ic, KeyValue kv, boolean withTags)
123          throws IOException {
124    ByteArrayOutputStream outIC = new ByteArrayOutputStream(ic.getSerializedSize(withTags));
125    ByteArrayOutputStream outKV = new ByteArrayOutputStream(kv.getSerializedSize(withTags));
126
127    assertEquals(kv.write(outKV, withTags), ic.write(outIC, withTags));  // compare the number of bytes written
128    assertArrayEquals(outKV.getBuffer(), outIC.getBuffer());             // compare the underlying byte array
129  }

131  /**
132   * Verify getXXXArray() and getXXXLength() when family/qualifier/value/tags are null.
133   * Should have the same behaviors as {@link KeyValue}.
134   */
135  @Test
136  public void testNullFamilyQualifierValueTags() {
137    byte[] row = Bytes.toBytes("row1");
138
139    long timestamp = 5000L;
140    long seqId     = 0L;
141    KeyValue.Type type      = KeyValue.Type.Put;
142
143    // Test when following fields are null.
144    byte[] family    = null;
145    byte[] qualifier = null;
146    byte[] value     = null;
147    byte[] tags      = null;
148
149    Cell ic1 = new IndividualBytesFieldCell(row, family, qualifier, timestamp, type, seqId, value, tags);
150
151    Cell kv1 = new KeyValue(row, family, qualifier, timestamp, type, value, tags);
152    byte[] familyArrayInKV    = Bytes.copy(kv1.getFamilyArray()   , kv1.getFamilyOffset()   , kv1.getFamilyLength());
153    byte[] qualifierArrayInKV = Bytes.copy(kv1.getQualifierArray(), kv1.getQualifierOffset(), kv1.getQualifierLength());
154    byte[] valueArrayInKV     = Bytes.copy(kv1.getValueArray()    , kv1.getValueOffset()    , kv1.getValueLength());
155    byte[] tagsArrayInKV      = Bytes.copy(kv1.getTagsArray()     , kv1.getTagsOffset()     , kv1.getTagsLength());
156
157    // getXXXArray() for family, qualifier, value and tags are supposed to return empty byte array, rather than null.
158    assertArrayEquals(familyArrayInKV   , ic1.getFamilyArray());
159    assertArrayEquals(qualifierArrayInKV, ic1.getQualifierArray());
160    assertArrayEquals(valueArrayInKV    , ic1.getValueArray());
161    assertArrayEquals(tagsArrayInKV     , ic1.getTagsArray());
162
163    // getXXXLength() for family, qualifier, value and tags are supposed to return 0.
164    assertEquals(kv1.getFamilyLength()   , ic1.getFamilyLength());
165    assertEquals(kv1.getQualifierLength(), ic1.getQualifierLength());
166    assertEquals(kv1.getValueLength()    , ic1.getValueLength());
167    assertEquals(kv1.getTagsLength()     , ic1.getTagsLength());
168  }

236  @Test
237  public void testWriteTag() throws IOException {
238    byte[] tags = Bytes.toBytes("---tags---");
239    int tagOffset = 3;
240    int length = 4;
241    IndividualBytesFieldCell cell
242        = new IndividualBytesFieldCell(Bytes.toBytes("row"), 0, 3,
243          Bytes.toBytes("family"), 0, 6,
244          Bytes.toBytes("qualifier"), 0, 9,
245          0L, KeyValue.Type.Put, 0,
246          Bytes.toBytes("value"), 0, 5,
247          tags, tagOffset, length);
248
249    try (ByteArrayOutputStream output = new ByteArrayOutputStream(300)) {
250      cell.write(output, true);
251      byte[] buf = output.toByteArray();
252      assertEquals(cell.getSerializedSize(true), buf.length);
253    }
254  }
