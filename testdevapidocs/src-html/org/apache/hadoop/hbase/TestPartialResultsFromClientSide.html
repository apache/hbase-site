060/**
061 * These tests are focused on testing how partial results appear to a client. Partial results are
062 * {@link Result}s that contain only a portion of a row's complete list of cells. Partial results
063 * are formed when the server breaches its maximum result size when trying to service a client's RPC
064 * request. It is the responsibility of the scanner on the client side to recognize when partial
065 * results have been returned and to take action to form the complete results.
066 * <p>
067 * Unless the flag {@link Scan#setAllowPartialResults(boolean)} has been set to true, the caller of
068 * {@link ResultScanner#next()} should never see partial results.
069 */
070@Category(MediumTests.class)

259  public void testOrderingOfCellsInPartialResults(final Scan basePartialScan) throws Exception {
260    // Scan that retrieves results in pieces (partials). By setting allowPartialResults to be true
261    // the results will NOT be reconstructed and instead the caller will see the partial results
262    // returned by the server
263    Scan partialScan = new Scan(basePartialScan);
264    partialScan.setAllowPartialResults(true);
265    ResultScanner partialScanner = TABLE.getScanner(partialScan);
266
267    // Scan that retrieves all table results in single RPC request
268    Scan oneShotScan = new Scan(basePartialScan);
269    oneShotScan.setMaxResultSize(Long.MAX_VALUE);
270    oneShotScan.setCaching(ROWS.length);
271    ResultScanner oneShotScanner = TABLE.getScanner(oneShotScan);
272
273    Result oneShotResult = oneShotScanner.next();
274    Result partialResult = null;
275    int iterationCount = 0;
276
277    while (oneShotResult != null && oneShotResult.rawCells() != null) {
278      List<Cell> aggregatePartialCells = new ArrayList<>();
279      do {
280        partialResult = partialScanner.next();
281        assertTrue("Partial Result is null. iteration: " + iterationCount, partialResult != null);
282        assertTrue("Partial cells are null. iteration: " + iterationCount,
283            partialResult.rawCells() != null);
284
285        for (Cell c : partialResult.rawCells()) {
286          aggregatePartialCells.add(c);
287        }
288      } while (partialResult.mayHaveMoreCellsInRow());
289
290      assertTrue("Number of cells differs. iteration: " + iterationCount,
291          oneShotResult.rawCells().length == aggregatePartialCells.size());
292      final Cell[] oneShotCells = oneShotResult.rawCells();
293      for (int cell = 0; cell < oneShotCells.length; cell++) {
294        Cell oneShotCell = oneShotCells[cell];
295        Cell partialCell = aggregatePartialCells.get(cell);
296
297        assertTrue("One shot cell was null", oneShotCell != null);
298        assertTrue("Partial cell was null", partialCell != null);
299        assertTrue("Cell differs. oneShotCell:" + oneShotCell + " partialCell:" + partialCell,
300            oneShotCell.equals(partialCell));
301      }
302
303      oneShotResult = oneShotScanner.next();
304      iterationCount++;
305    }
306
307    assertTrue(partialScanner.next() == null);
308
309    partialScanner.close();
310    oneShotScanner.close();
311  }

333  public void testExpectedNumberOfCellsPerPartialResult(Scan baseScan, int expectedNumberOfCells)
334      throws Exception {
335
336    if (LOG.isInfoEnabled()) LOG.info("groupSize:" + expectedNumberOfCells);
337
338    // Use the cellHeapSize to set maxResultSize such that we know how many cells to expect back
339    // from the call. The returned results should NOT exceed expectedNumberOfCells but may be less
340    // than it in cases where expectedNumberOfCells is not an exact multiple of the number of
341    // columns in the table.
342    Scan scan = new Scan(baseScan);
343    scan.setAllowPartialResults(true);
344    scan.setMaxResultSize(getResultSizeForNumberOfCells(expectedNumberOfCells));
345
346    ResultScanner scanner = TABLE.getScanner(scan);
347    Result result = null;
348    byte[] prevRow = null;
349    while ((result = scanner.next()) != null) {
350      assertTrue(result.rawCells() != null);
351
352      // Cases when cell count won't equal expectedNumberOfCells:
353      // 1. Returned result is the final result needed to form the complete result for that row
354      // 2. It is the first result we have seen for that row and thus may have been fetched as
355      // the last group of cells that fit inside the maxResultSize
356      assertTrue(
357          "Result's cell count differed from expected number. result: " + result,
358          result.rawCells().length == expectedNumberOfCells || !result.mayHaveMoreCellsInRow()
359              || !Bytes.equals(prevRow, result.getRow()));
360      prevRow = result.getRow();
361    }
362
363    scanner.close();
364  }

366  /**
367   * @return The approximate heap size of a cell in the test table. All cells should have
368   *         approximately the same heap size, so the value is cached to avoid repeating the
369   *         calculation
370   * @throws Exception
371   */
372  private long getCellHeapSize() throws Exception {
373    if (CELL_HEAP_SIZE == -1) {
374      // Do a partial scan that will return a single result with a single cell
375      Scan scan = new Scan();
376      scan.setMaxResultSize(2);
377      scan.setAllowPartialResults(true);
378      ResultScanner scanner = TABLE.getScanner(scan);
379
380      Result result = scanner.next();
381
382      assertTrue(result != null);
383      assertTrue(result.rawCells() != null);
384      assertTrue(result.rawCells().length == 1);
385
386      // Estimate the cell heap size. One difference is that on server side, the KV Heap size is
387      // estimated differently in case the cell is backed up by MSLAB byte[] (no overhead for
388      // backing array). Thus below calculation is a bit brittle.
389      CELL_HEAP_SIZE = PrivateCellUtil.estimatedHeapSizeOf(result.rawCells()[0])
390          - (ClassSize.ARRAY+3);
391      if (LOG.isInfoEnabled()) LOG.info("Cell heap size: " + CELL_HEAP_SIZE);
392      scanner.close();
393    }
394
395    return CELL_HEAP_SIZE;
396  }

401   *         the server to return exactly numberOfCells cells

403   */
404  private long getResultSizeForNumberOfCells(int numberOfCells) throws Exception {

549  /**
550   * Examine the interaction between the maxResultSize and caching. If the caching limit is reached
551   * before the maxResultSize limit, we should not see partial results. On the other hand, if the
552   * maxResultSize limit is reached before the caching limit, it is likely that partial results will
553   * be seen.
554   * @throws Exception
555   */
556  @Test
557  public void testPartialResultsAndCaching() throws Exception {
558    for (int caching = 1; caching <= NUM_ROWS; caching++) {
559      for (int maxResultRows = 0; maxResultRows <= NUM_ROWS; maxResultRows++) {
560        testPartialResultsAndCaching(maxResultRows, caching);
561      }
562    }

565  /**
566   * @param resultSizeRowLimit The row limit that will be enforced through maxResultSize
567   * @param cachingRowLimit The row limit that will be enforced through caching
568   * @throws Exception
569   */
570  public void testPartialResultsAndCaching(int resultSizeRowLimit, int cachingRowLimit)
571      throws Exception {
572    Scan scan = new Scan();
573    scan.setAllowPartialResults(true);
574
575    // The number of cells specified in the call to getResultSizeForNumberOfCells is offset to
576    // ensure that the result size we specify is not an exact multiple of the number of cells
577    // in a row. This ensures that partial results will be returned when the result size limit
578    // is reached before the caching limit.
579    int cellOffset = NUM_COLS / 3;
580    long maxResultSize = getResultSizeForNumberOfCells(resultSizeRowLimit * NUM_COLS + cellOffset);
581    scan.setMaxResultSize(maxResultSize);
582    scan.setCaching(cachingRowLimit);
583
584    ResultScanner scanner = TABLE.getScanner(scan);
585    ClientScanner clientScanner = (ClientScanner) scanner;
586    Result r = null;
587
588    // Approximate the number of rows we expect will fit into the specified max rsult size. If this
589    // approximation is less than caching, then we expect that the max result size limit will be
590    // hit before the caching limit and thus partial results may be seen
591    boolean expectToSeePartialResults = resultSizeRowLimit < cachingRowLimit;
592    while ((r = clientScanner.next()) != null) {
593      assertTrue(!r.mayHaveMoreCellsInRow() || expectToSeePartialResults);
594    }
595
596    scanner.close();
597  }

599  /**
600   * Make puts to put the input value into each combination of row, family, and qualifier
601   * @param rows
602   * @param families
603   * @param qualifiers
604   * @param value
605   * @return
606   * @throws IOException
607   */
608  static ArrayList<Put> createPuts(byte[][] rows, byte[][] families, byte[][] qualifiers,
609      byte[] value) throws IOException {
610    Put put;
611    ArrayList<Put> puts = new ArrayList<>();
612
613    for (int row = 0; row < rows.length; row++) {
614      put = new Put(rows[row]);
615      for (int fam = 0; fam < families.length; fam++) {
616        for (int qual = 0; qual < qualifiers.length; qual++) {
617          KeyValue kv = new KeyValue(rows[row], families[fam], qualifiers[qual], qual, value);
618          put.add(kv);
619        }
620      }
621      puts.add(put);
622    }
623
624    return puts;
625  }

627  /**
628   * Make key values to represent each possible combination of family and qualifier in the specified
629   * row.
630   * @param row
631   * @param families
632   * @param qualifiers
633   * @param value
634   * @return
635   */
636  static ArrayList<Cell> createKeyValuesForRow(byte[] row, byte[][] families, byte[][] qualifiers,
637      byte[] value) {
638    ArrayList<Cell> outList = new ArrayList<>();
639    for (int fam = 0; fam < families.length; fam++) {
640      for (int qual = 0; qual < qualifiers.length; qual++) {
641        outList.add(new KeyValue(row, families[fam], qualifiers[qual], qual, value));
642      }
643    }
644    return outList;
645  }

647  /**
648   * Verifies that result contains all the key values within expKvList. Fails the test otherwise
649   * @param result
650   * @param expKvList
651   * @param msg
652   */
653  static void verifyResult(Result result, List<Cell> expKvList, String msg) {
654    if (LOG.isInfoEnabled()) {
655      LOG.info(msg);
656      LOG.info("Expected count: " + expKvList.size());
657      LOG.info("Actual count: " + result.size());
658    }
659
660    if (expKvList.isEmpty()) return;
661
662    int i = 0;
663    for (Cell kv : result.rawCells()) {
664      if (i >= expKvList.size()) {
665        break; // we will check the size later
666      }
667
668      Cell kvExp = expKvList.get(i++);
669      assertTrue("Not equal. get kv: " + kv.toString() + " exp kv: " + kvExp.toString(),
670          kvExp.equals(kv));
671    }
672
673    assertEquals(expKvList.size(), result.size());
674  }

712    // Open scanner before deletes
713    ResultScanner scanner =
714        tmpTable.getScanner(new Scan().setMaxResultSize(1).setAllowPartialResults(true));
715    // now the openScanner will also fetch data and will be executed lazily, i.e, only openScanner
716    // when you call next, so here we need to make a next call to open scanner. The maxResultSize
717    // limit can make sure that we will not fetch all the data at once, so the test sill works.
718    int scannerCount = scanner.next().rawCells().length;
719    Delete delete1 = new Delete(rows[0]);
720    delete1.addColumn(families[0], qualifiers[0], 0);
721    tmpTable.delete(delete1);
722
723    Delete delete2 = new Delete(rows[1]);
724    delete2.addColumn(families[1], qualifiers[1], 1);
725    tmpTable.delete(delete2);
726
727    // Should see all cells because scanner was opened prior to deletes
728    scannerCount += countCellsFromScanner(scanner);
729    int expectedCount = numRows * numFamilies * numQualifiers;
730    assertTrue("scannerCount: " + scannerCount + " expectedCount: " + expectedCount,
731        scannerCount == expectedCount);
732
733    // Minus 2 for the two cells that were deleted
734    scanner = tmpTable.getScanner(new Scan().setMaxResultSize(1).setAllowPartialResults(true));
735    scannerCount = countCellsFromScanner(scanner);
736    expectedCount = numRows * numFamilies * numQualifiers - 2;
737    assertTrue("scannerCount: " + scannerCount + " expectedCount: " + expectedCount,
738        scannerCount == expectedCount);
739
740    scanner = tmpTable.getScanner(new Scan().setMaxResultSize(1).setAllowPartialResults(true));
741    scannerCount = scanner.next().rawCells().length;
742    // Put in 2 new rows. The timestamps differ from the deleted rows
743    Put put1 = new Put(rows[0]);
744    put1.add(new KeyValue(rows[0], families[0], qualifiers[0], 1, value));
745    tmpTable.put(put1);
746
747    Put put2 = new Put(rows[1]);
748    put2.add(new KeyValue(rows[1], families[1], qualifiers[1], 2, value));
749    tmpTable.put(put2);
750
751    // Scanner opened prior to puts. Cell count shouldn't have changed
752    scannerCount += countCellsFromScanner(scanner);
753    expectedCount = numRows * numFamilies * numQualifiers - 2;
754    assertTrue("scannerCount: " + scannerCount + " expectedCount: " + expectedCount,
755        scannerCount == expectedCount);
756
757    // Now the scanner should see the cells that were added by puts
758    scanner = tmpTable.getScanner(new Scan().setMaxResultSize(1).setAllowPartialResults(true));
759    scannerCount = countCellsFromScanner(scanner);
760    expectedCount = numRows * numFamilies * numQualifiers;
761    assertTrue("scannerCount: " + scannerCount + " expectedCount: " + expectedCount,
762        scannerCount == expectedCount);
763
764    TEST_UTIL.deleteTable(tableName);
765  }

767  /**
768   * Exhausts the scanner by calling next repetitively. Once completely exhausted, close scanner and
769   * return total cell count
770   * @param scanner
771   * @return
772   * @throws Exception
773   */
774  private int countCellsFromScanner(ResultScanner scanner) throws Exception {
775    Result result = null;
776    int numCells = 0;
777    while ((result = scanner.next()) != null) {
778      numCells += result.rawCells().length;
779    }
780
781    scanner.close();
782    return numCells;
783  }
