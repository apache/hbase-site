199  /**
200   * Split the known table in half.  (this is hard coded for this test suite)
201   */
202  private void forceSplit(TableName table) {
203    try {
204      // need to call regions server to by synchronous but isn't visible.
205      HRegionServer hrs = util.getRSForFirstRegionInTable(table);
206
207      for (HRegionInfo hri :
208          ProtobufUtil.getOnlineRegions(hrs.getRSRpcServices())) {
209        if (hri.getTable().equals(table)) {
210          // splitRegion doesn't work if startkey/endkey are null
211          ProtobufUtil.split(null, hrs.getRSRpcServices(), hri, rowkey(ROWCOUNT / 2));
212        }
213      }
214
215      // verify that split completed.
216      int regions;
217      do {
218        regions = 0;
219        for (HRegionInfo hri :
220            ProtobufUtil.getOnlineRegions(hrs.getRSRpcServices())) {
221          if (hri.getTable().equals(table)) {
222            regions++;
223          }
224        }
225        if (regions != 2) {
226          LOG.info("Taking some time to complete split...");
227          Thread.sleep(250);
228        }
229      } while (regions != 2);
230    } catch (IOException e) {
231      e.printStackTrace();
232    } catch (InterruptedException e) {
233      e.printStackTrace();
234    }
235  }

401  /**
402   * This test exercises the path where there is a split after initial
403   * validation but before the atomic bulk load call. We cannot use presplitting
404   * to test this path, so we actually inject a split just before the atomic
405   * region load.
406   */
407  @Test (timeout=120000)
408  public void testSplitWhileBulkLoadPhase() throws Exception {
409    final TableName table = TableName.valueOf(name.getMethodName());
410    try (Connection connection = ConnectionFactory.createConnection(util.getConfiguration())) {
411      setupTable(connection, table, 10);
412      populateTable(connection, table,1);
413      assertExpectedTable(table, ROWCOUNT, 1);
414
415      // Now let's cause trouble.  This will occur after checks and cause bulk
416      // files to fail when attempt to atomically import.  This is recoverable.
417      final AtomicInteger attemptedCalls = new AtomicInteger();
418      LoadIncrementalHFiles lih2 = new LoadIncrementalHFiles(util.getConfiguration()) {
419        @Override
420        protected void bulkLoadPhase(final Table htable, final Connection conn,
421            ExecutorService pool, Deque<LoadQueueItem> queue,
422            final Multimap<ByteBuffer, LoadQueueItem> regionGroups, boolean copyFile,
423            Map<LoadQueueItem, ByteBuffer> item2RegionMap)
424                throws IOException {
425          int i = attemptedCalls.incrementAndGet();
426          if (i == 1) {
427            // On first attempt force a split.
428            forceSplit(table);
429          }
430          super.bulkLoadPhase(htable, conn, pool, queue, regionGroups, copyFile, item2RegionMap);
431        }
432      };
433
434      // create HFiles for different column families
435      try (Table t = connection.getTable(table);
436          RegionLocator locator = connection.getRegionLocator(table);
437          Admin admin = connection.getAdmin()) {
438        Path bulk = buildBulkFiles(table, 2);
439        lih2.doBulkLoad(bulk, admin, t, locator);
440      }
441
442      // check that data was loaded
443      // The three expected attempts are 1) failure because need to split, 2)
444      // load of split top 3) load of split bottom
445      assertEquals(attemptedCalls.get(), 3);
446      assertExpectedTable(table, ROWCOUNT, 2);
447    }
448  }

610
611    // Mess it up by leaving a hole in the hbase:meta
612    List<HRegionInfo> regionInfos = MetaTableAccessor.getTableRegions(connection, tableName);
613    for (HRegionInfo regionInfo : regionInfos) {
614      if (Bytes.equals(regionInfo.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
615        MetaTableAccessor.deleteRegion(connection, regionInfo);
616        break;
617      }
618    }
619
620    try (Table t = connection.getTable(tableName);
621        RegionLocator locator = connection.getRegionLocator(tableName);
622        Admin admin = connection.getAdmin()) {
623      loader.doBulkLoad(dir, admin, t, locator);
624    } catch (Exception e) {
625      LOG.error("exception=", e);
626      assertTrue("IOException expected", e instanceof IOException);
627    }
628
629    table.close();
630
631    // Make sure at least the one region that still exists can be found.
632    regionInfos = MetaTableAccessor.getTableRegions(connection, tableName);
633    assertTrue(regionInfos.size() >= 1);
634
635    this.assertExpectedTable(connection, tableName, ROWCOUNT, 2);
636    connection.close();
637  }

639  /**
640   * Checks that all columns have the expected value and that there is the
641   * expected number of rows.
642   * @throws IOException
643   */
644  void assertExpectedTable(final Connection connection, TableName table, int count, int value)
645  throws IOException {
646    HTableDescriptor [] htds = util.getAdmin().listTables(table.getNameAsString());
647    assertEquals(htds.length, 1);
648    Table t = null;
649    try {
650      t = connection.getTable(table);
651      Scan s = new Scan();
652      ResultScanner sr = t.getScanner(s);
653      int i = 0;
654      for (Result r : sr) {
655        i++;
656        for (NavigableMap<byte[], byte[]> nm : r.getNoVersionMap().values()) {
657          for (byte[] val : nm.values()) {
658            assertTrue(Bytes.equals(val, value(value)));
659          }
660        }
661      }
662      assertEquals(count, i);
663    } catch (IOException e) {
664      fail("Failed due to exception");
665    } finally {
666      if (t != null) t.close();
667    }
668  }
