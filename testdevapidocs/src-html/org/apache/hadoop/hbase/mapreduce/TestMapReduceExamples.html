051  /**
052   * Test SampleUploader from examples
053   */
054
055  @SuppressWarnings("unchecked")
056  @Test
057  public void testSampleUploader() throws Exception {
058
059    Configuration configuration = new Configuration();
060    Uploader uploader = new Uploader();
061    Mapper<LongWritable, Text, ImmutableBytesWritable, Put>.Context ctx = mock(Context.class);
062    doAnswer(new Answer<Void>() {
063
064      @Override
065      public Void answer(InvocationOnMock invocation) throws Throwable {
066        ImmutableBytesWritable writer = (ImmutableBytesWritable) invocation.getArguments()[0];
067        Put put = (Put) invocation.getArguments()[1];
068        assertEquals("row", Bytes.toString(writer.get()));
069        assertEquals("row", Bytes.toString(put.getRow()));
070        return null;
071      }
072    }).when(ctx).write(any(ImmutableBytesWritable.class), any(Put.class));
073
074    uploader.map(null, new Text("row,family,qualifier,value"), ctx);
075
076    Path dir = util.getDataTestDirOnTestFS("testSampleUploader");
077
078    String[] args = { dir.toString(), "simpleTable" };
079    Job job = SampleUploader.configureJob(configuration, args);
080    assertEquals(SequenceFileInputFormat.class, job.getInputFormatClass());
081
082  }

084  /**
085   * Test main method of SampleUploader.
086   */
087  @Test
088  public void testMainSampleUploader() throws Exception {
089    PrintStream oldPrintStream = System.err;
090    SecurityManager SECURITY_MANAGER = System.getSecurityManager();
091    LauncherSecurityManager newSecurityManager= new LauncherSecurityManager();
092    System.setSecurityManager(newSecurityManager);
093    ByteArrayOutputStream data = new ByteArrayOutputStream();
094    String[] args = {};
095    System.setErr(new PrintStream(data));
096    try {
097      System.setErr(new PrintStream(data));
098
099      try {
100        SampleUploader.main(args);
101        fail("should be SecurityException");
102      } catch (SecurityException e) {
103        assertEquals(-1, newSecurityManager.getExitCode());
104        assertTrue(data.toString().contains("Wrong number of arguments:"));
105        assertTrue(data.toString().contains("Usage: SampleUploader <input> <tablename>"));
106      }
107
108    } finally {
109      System.setErr(oldPrintStream);
110      System.setSecurityManager(SECURITY_MANAGER);
111    }
112
113  }

115  /**
116   * Test IndexBuilder from examples
117   */
118  @SuppressWarnings("unchecked")
119  @Test
120  public void testIndexBuilder() throws Exception {
121    Configuration configuration = new Configuration();
122    String[] args = { "tableName", "columnFamily", "column1", "column2" };
123    IndexBuilder.configureJob(configuration, args);
124    assertEquals("tableName", configuration.get("index.tablename"));
125    assertEquals("tableName", configuration.get(TableInputFormat.INPUT_TABLE));
126    assertEquals("column1,column2", configuration.get("index.fields"));
127
128    Map map = new Map();
129    ImmutableBytesWritable rowKey = new ImmutableBytesWritable(Bytes.toBytes("test"));
130    Mapper<ImmutableBytesWritable, Result, ImmutableBytesWritable, Put>.Context ctx =
131        mock(Context.class);
132    when(ctx.getConfiguration()).thenReturn(configuration);
133    doAnswer(new Answer<Void>() {
134
135      @Override
136      public Void answer(InvocationOnMock invocation) throws Throwable {
137        ImmutableBytesWritable writer = (ImmutableBytesWritable) invocation.getArguments()[0];
138        Put put = (Put) invocation.getArguments()[1];
139        assertEquals("tableName-column1", Bytes.toString(writer.get()));
140        assertEquals("test", Bytes.toString(put.getRow()));
141        return null;
142      }
143    }).when(ctx).write(any(ImmutableBytesWritable.class), any(Put.class));
144    Result result = mock(Result.class);
145    when(result.getValue(Bytes.toBytes("columnFamily"), Bytes.toBytes("column1"))).thenReturn(
146        Bytes.toBytes("test"));
147    map.setup(ctx);
148    map.map(rowKey, result, ctx);
149  }
