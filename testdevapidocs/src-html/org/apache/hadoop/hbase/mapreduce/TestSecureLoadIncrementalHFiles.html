020package org.apache.hadoop.hbase.mapreduce;
021
022import org.apache.hadoop.hbase.HConstants;
023import org.apache.hadoop.hbase.codec.KeyValueCodecWithTags;
024import org.apache.hadoop.hbase.testclassification.LargeTests;
025import org.apache.hadoop.hbase.testclassification.MapReduceTests;
026import org.apache.hadoop.hbase.security.UserProvider;
027import org.apache.hadoop.hbase.security.access.AccessControlLists;
028import org.apache.hadoop.hbase.security.access.SecureTestUtil;
029
030import org.junit.BeforeClass;
031import org.junit.experimental.categories.Category;
032
033/**
034 * Reruns TestLoadIncrementalHFiles using LoadIncrementalHFiles in secure mode.
035 * This suite is unable to verify the security handoff/turnover
036 * as miniCluster is running as system user thus has root privileges
037 * and delegation tokens don't seem to work on miniDFS.
038 *
039 * Thus SecureBulkload can only be completely verified by running
040 * integration tests against a secure cluster. This suite is still
041 * invaluable as it verifies the other mechanisms that need to be
042 * supported as part of a LoadIncrementalFiles call.
043 */
044@Category({MapReduceTests.class, LargeTests.class})
045public class TestSecureLoadIncrementalHFiles extends  TestLoadIncrementalHFiles{
046
047  @BeforeClass
048  public static void setUpBeforeClass() throws Exception {
049    // set the always on security provider
050    UserProvider.setUserProviderForTesting(util.getConfiguration(),
051      HadoopSecurityEnabledUserProviderForTesting.class);
052    // setup configuration
053    SecureTestUtil.enableSecurity(util.getConfiguration());
054    util.getConfiguration().setInt(
055        LoadIncrementalHFiles.MAX_FILES_PER_REGION_PER_FAMILY,
056        MAX_FILES_PER_REGION_PER_FAMILY);
057    // change default behavior so that tag values are returned with normal rpcs
058    util.getConfiguration().set(HConstants.RPC_CODEC_CONF_KEY,
059        KeyValueCodecWithTags.class.getCanonicalName());
060
061    util.startMiniCluster();
062
063    // Wait for the ACL table to become available
064    util.waitTableEnabled(AccessControlLists.ACL_TABLE_NAME);
065
066    setupNamespace();
067  }
068
069}
