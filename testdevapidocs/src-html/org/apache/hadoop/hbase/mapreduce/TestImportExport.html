442   * Create a simple table, run an Export Job on it, Import with filtering on,  verify counts,
443   * attempt with invalid values.
444   */
445  @Test
446  public void testWithFilter() throws Exception {
447    // Create simple table to export
448    HTableDescriptor desc = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
449    desc.addFamily(new HColumnDescriptor(FAMILYA).setMaxVersions(5));
450    UTIL.getAdmin().createTable(desc);
451    Table exportTable = UTIL.getConnection().getTable(desc.getTableName());
452
453    Put p1 = new Put(ROW1);
454    p1.addColumn(FAMILYA, QUAL, now, QUAL);
455    p1.addColumn(FAMILYA, QUAL, now + 1, QUAL);
456    p1.addColumn(FAMILYA, QUAL, now + 2, QUAL);
457    p1.addColumn(FAMILYA, QUAL, now + 3, QUAL);
458    p1.addColumn(FAMILYA, QUAL, now + 4, QUAL);
459
460    // Having another row would actually test the filter.
461    Put p2 = new Put(ROW2);
462    p2.addColumn(FAMILYA, QUAL, now, QUAL);
463
464    exportTable.put(Arrays.asList(p1, p2));
465
466    // Export the simple table
467    String[] args = new String[] { name.getMethodName(), FQ_OUTPUT_DIR, "1000" };
468    assertTrue(runExport(args));
469
470    // Import to a new table
471    final String IMPORT_TABLE = name.getMethodName() + "import";
472    desc = new HTableDescriptor(TableName.valueOf(IMPORT_TABLE));
473    desc.addFamily(new HColumnDescriptor(FAMILYA).setMaxVersions(5));
474    UTIL.getAdmin().createTable(desc);
475
476    Table importTable = UTIL.getConnection().getTable(desc.getTableName());
477    args = new String[] { "-D" + Import.FILTER_CLASS_CONF_KEY + "=" + PrefixFilter.class.getName(),
478        "-D" + Import.FILTER_ARGS_CONF_KEY + "=" + Bytes.toString(ROW1), IMPORT_TABLE,
479        FQ_OUTPUT_DIR,
480        "1000" };
481    assertTrue(runImport(args));
482
483    // get the count of the source table for that time range
484    PrefixFilter filter = new PrefixFilter(ROW1);
485    int count = getCount(exportTable, filter);
486
487    Assert.assertEquals("Unexpected row count between export and import tables", count,
488      getCount(importTable, null));
489
490    // and then test that a broken command doesn't bork everything - easier here because we don't
491    // need to re-run the export job
492
493    args = new String[] { "-D" + Import.FILTER_CLASS_CONF_KEY + "=" + Filter.class.getName(),
494        "-D" + Import.FILTER_ARGS_CONF_KEY + "=" + Bytes.toString(ROW1) + "", name.getMethodName(),
495        FQ_OUTPUT_DIR, "1000" };
496    assertFalse(runImport(args));
497
498    // cleanup
499    exportTable.close();
500    importTable.close();
501  }

503  /**
504   * Count the number of keyvalues in the specified table for the given timerange
505   * @param start
506   * @param end
507   * @param table
508   * @return
509   * @throws IOException
510   */
511  private int getCount(Table table, Filter filter) throws IOException {
512    Scan scan = new Scan();
513    scan.setFilter(filter);
514    ResultScanner results = table.getScanner(scan);
515    int count = 0;
516    for (Result res : results) {
517      count += res.size();
518    }
519    results.close();
520    return count;
521  }

702  /**
703   * This listens to the {@link #visitLogEntryBeforeWrite(HRegionInfo, WALKey, WALEdit)} to
704   * identify that an entry is written to the Write Ahead Log for the given table.
705   */
706  private static class TableWALActionListener extends WALActionsListener.Base {
707
708    private HRegionInfo regionInfo;
709    private boolean isVisited = false;
710
711    public TableWALActionListener(HRegionInfo region) {
712      this.regionInfo = region;
713    }
714
715    @Override
716    public void visitLogEntryBeforeWrite(WALKey logKey, WALEdit logEdit) {
717      if (logKey.getTablename().getNameAsString().equalsIgnoreCase(
718          this.regionInfo.getTable().getNameAsString()) && (!logEdit.isMetaEdit())) {
719        isVisited = true;
720      }
721    }
722
723    public boolean isWALVisited() {
724      return isVisited;
725    }
726  }
