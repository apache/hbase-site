055  private static void delete(File file) throws IOException {
056    if (file.getAbsolutePath().length() < 5) {
057      throw new IllegalArgumentException(
058        MessageFormat.format("Path [{0}] is too short, not deleting",
059                             file.getAbsolutePath()));
060    }
061    if (file.exists()) {
062      if (file.isDirectory()) {
063        File[] children = file.listFiles();
064        if (children != null) {
065          for (File child : children) {
066            delete(child);
067          }
068        }
069      }
070      if (!file.delete()) {
071        throw new RuntimeException(
072          MessageFormat.format("Could not delete path [{0}]",
073                               file.getAbsolutePath()));
074      }
075    }
076  }

086  @Test
087  public void testExistingManifest() throws Exception {
088    File dir = new File(System.getProperty("test.build.dir", "target/test-dir"),
089                        TestJarFinder.class.getName() + "-testExistingManifest");
090    delete(dir);
091    dir.mkdirs();
092
093    File metaInfDir = new File(dir, "META-INF");
094    metaInfDir.mkdirs();
095    File manifestFile = new File(metaInfDir, "MANIFEST.MF");
096    Manifest manifest = new Manifest();
097    OutputStream os = new FileOutputStream(manifestFile);
098    manifest.write(os);
099    os.close();
100
101    File propsFile = new File(dir, "props.properties");
102    Writer writer = new FileWriter(propsFile);
103    new Properties().store(writer, "");
104    writer.close();
105    ByteArrayOutputStream baos = new ByteArrayOutputStream();
106    JarOutputStream zos = new JarOutputStream(baos);
107    JarFinder.jarDir(dir, "", zos);
108    JarInputStream jis =
109      new JarInputStream(new ByteArrayInputStream(baos.toByteArray()));
110    Assert.assertNotNull(jis.getManifest());
111    jis.close();
112  }
