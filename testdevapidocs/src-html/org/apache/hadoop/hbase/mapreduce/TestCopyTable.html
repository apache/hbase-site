092  private void doCopyTableTest(boolean bulkload) throws Exception {
093    final TableName tableName1 = TableName.valueOf(name.getMethodName() + "1");
094    final TableName tableName2 = TableName.valueOf(name.getMethodName() + "2");
095    final byte[] FAMILY = Bytes.toBytes("family");
096    final byte[] COLUMN1 = Bytes.toBytes("c1");
097
098    try (Table t1 = TEST_UTIL.createTable(tableName1, FAMILY);
099         Table t2 = TEST_UTIL.createTable(tableName2, FAMILY)) {
100      // put rows into the first table
101      loadData(t1, FAMILY, COLUMN1);
102
103      CopyTable copy = new CopyTable();
104      int code;
105      if (bulkload) {
106        code = ToolRunner.run(new Configuration(TEST_UTIL.getConfiguration()),
107            copy, new String[] { "--new.name=" + tableName2.getNameAsString(),
108              "--bulkload", tableName1.getNameAsString() });
109      } else {
110        code = ToolRunner.run(new Configuration(TEST_UTIL.getConfiguration()),
111            copy, new String[] { "--new.name=" + tableName2.getNameAsString(),
112            tableName1.getNameAsString() });
113      }
114      assertEquals("copy job failed", 0, code);
115
116      // verify the data was copied into table 2
117      verifyRows(t2, FAMILY, COLUMN1);
118    } finally {
119      TEST_UTIL.deleteTable(tableName1);
120      TEST_UTIL.deleteTable(tableName2);
121    }
122  }

124  private void doCopyTableTestWithMob(boolean bulkload) throws Exception {
125    final TableName tableName1 = TableName.valueOf(name.getMethodName() + "1");
126    final TableName tableName2 = TableName.valueOf(name.getMethodName() + "2");
127    final byte[] FAMILY = Bytes.toBytes("mob");
128    final byte[] COLUMN1 = Bytes.toBytes("c1");
129
130    ColumnFamilyDescriptorBuilder cfd = ColumnFamilyDescriptorBuilder.newBuilder(FAMILY);
131
132    cfd.setMobEnabled(true);
133    cfd.setMobThreshold(5);
134    TableDescriptor desc1 = TableDescriptorBuilder.newBuilder(tableName1)
135            .setColumnFamily(cfd.build())
136            .build();
137    TableDescriptor desc2 = TableDescriptorBuilder.newBuilder(tableName2)
138            .setColumnFamily(cfd.build())
139            .build();
140
141    try (Table t1 = TEST_UTIL.createTable(desc1, null);
142         Table t2 = TEST_UTIL.createTable(desc2, null);) {
143
144      // put rows into the first table
145      for (int i = 0; i < 10; i++) {
146        Put p = new Put(Bytes.toBytes("row" + i));
147        p.addColumn(FAMILY, COLUMN1, COLUMN1);
148        t1.put(p);
149      }
150
151      CopyTable copy = new CopyTable();
152
153      int code;
154      if (bulkload) {
155        code = ToolRunner.run(new Configuration(TEST_UTIL.getConfiguration()),
156            copy, new String[] { "--new.name=" + tableName2.getNameAsString(),
157              "--bulkload", tableName1.getNameAsString() });
158      } else {
159        code = ToolRunner.run(new Configuration(TEST_UTIL.getConfiguration()),
160            copy, new String[] { "--new.name=" + tableName2.getNameAsString(),
161            tableName1.getNameAsString() });
162      }
163      assertEquals("copy job failed", 0, code);
164
165      // verify the data was copied into table 2
166      for (int i = 0; i < 10; i++) {
167        Get g = new Get(Bytes.toBytes("row" + i));
168        Result r = t2.get(g);
169        assertEquals(1, r.size());
170        assertTrue(CellUtil.matchingQualifier(r.rawCells()[0], COLUMN1));
171        assertEquals("compare row values between two tables",
172              t1.getDescriptor().getValue("row" + i),
173              t2.getDescriptor().getValue("row" + i));
174      }
175
176      assertEquals("compare count of mob rows after table copy", MobTestUtil.countMobRows(t1),
177              MobTestUtil.countMobRows(t2));
178      assertEquals("compare count of mob row values between two tables",
179              t1.getDescriptor().getValues().size(),
180              t2.getDescriptor().getValues().size());
181      assertTrue("The mob row count is 0 but should be > 0",
182              MobTestUtil.countMobRows(t2) > 0);
183    } finally {
184      TEST_UTIL.deleteTable(tableName1);
185      TEST_UTIL.deleteTable(tableName2);
186    }
187  }

221  @Test
222  public void testStartStopRow() throws Exception {
223    final TableName tableName1 = TableName.valueOf(name.getMethodName() + "1");
224    final TableName tableName2 = TableName.valueOf(name.getMethodName() + "2");
225    final byte[] FAMILY = Bytes.toBytes("family");
226    final byte[] COLUMN1 = Bytes.toBytes("c1");
227    final byte[] row0 = Bytes.toBytesBinary("\\x01row0");
228    final byte[] row1 = Bytes.toBytesBinary("\\x01row1");
229    final byte[] row2 = Bytes.toBytesBinary("\\x01row2");
230
231    try (Table t1 = TEST_UTIL.createTable(tableName1, FAMILY);
232            Table t2 = TEST_UTIL.createTable(tableName2, FAMILY)) {
233
234      // put rows into the first table
235      Put p = new Put(row0);
236      p.addColumn(FAMILY, COLUMN1, COLUMN1);
237      t1.put(p);
238      p = new Put(row1);
239      p.addColumn(FAMILY, COLUMN1, COLUMN1);
240      t1.put(p);
241      p = new Put(row2);
242      p.addColumn(FAMILY, COLUMN1, COLUMN1);
243      t1.put(p);
244
245      CopyTable copy = new CopyTable();
246      assertEquals(0, ToolRunner.run(new Configuration(TEST_UTIL.getConfiguration()),
247              copy, new String[]{"--new.name=" + tableName2, "--startrow=\\x01row1",
248                "--stoprow=\\x01row2", tableName1.getNameAsString()}));
249
250      // verify the data was copied into table 2
251      // row1 exist, row0, row2 do not exist
252      Get g = new Get(row1);
253      Result r = t2.get(g);
254      assertEquals(1, r.size());
255      assertTrue(CellUtil.matchingQualifier(r.rawCells()[0], COLUMN1));
256
257      g = new Get(row0);
258      r = t2.get(g);
259      assertEquals(0, r.size());
260
261      g = new Get(row2);
262      r = t2.get(g);
263      assertEquals(0, r.size());
264
265    } finally {
266      TEST_UTIL.deleteTable(tableName1);
267      TEST_UTIL.deleteTable(tableName2);
268    }
269  }

274  @Test
275  public void testRenameFamily() throws Exception {
276    final TableName sourceTable = TableName.valueOf(name.getMethodName() + "source");
277    final TableName targetTable = TableName.valueOf(name.getMethodName() + "-target");
278
279    byte[][] families = { FAMILY_A, FAMILY_B };
280
281    Table t = TEST_UTIL.createTable(sourceTable, families);
282    Table t2 = TEST_UTIL.createTable(targetTable, families);
283    Put p = new Put(ROW1);
284    p.addColumn(FAMILY_A, QUALIFIER, Bytes.toBytes("Data11"));
285    p.addColumn(FAMILY_B, QUALIFIER, Bytes.toBytes("Data12"));
286    p.addColumn(FAMILY_A, QUALIFIER, Bytes.toBytes("Data13"));
287    t.put(p);
288    p = new Put(ROW2);
289    p.addColumn(FAMILY_B, QUALIFIER, Bytes.toBytes("Dat21"));
290    p.addColumn(FAMILY_A, QUALIFIER, Bytes.toBytes("Data22"));
291    p.addColumn(FAMILY_B, QUALIFIER, Bytes.toBytes("Data23"));
292    t.put(p);
293
294    long currentTime = EnvironmentEdgeManager.currentTime();
295    String[] args = new String[] { "--new.name=" + targetTable, "--families=a:b", "--all.cells",
296      "--starttime=" + (currentTime - 100000), "--endtime=" + (currentTime + 100000),
297      "--versions=1", sourceTable.getNameAsString() };
298    assertNull(t2.get(new Get(ROW1)).getRow());
299
300    assertTrue(runCopy(args));
301
302    assertNotNull(t2.get(new Get(ROW1)).getRow());
303    Result res = t2.get(new Get(ROW1));
304    byte[] b1 = res.getValue(FAMILY_B, QUALIFIER);
305    assertEquals("Data13", Bytes.toString(b1));
306    assertNotNull(t2.get(new Get(ROW2)).getRow());
307    res = t2.get(new Get(ROW2));
308    b1 = res.getValue(FAMILY_A, QUALIFIER);
309    // Data from the family of B is not copied
310    assertNull(b1);
311
312  }

347  private void loadData(Table t, byte[] family, byte[] column) throws IOException {
348    for (int i = 0; i < 10; i++) {
349      byte[] row = Bytes.toBytes("row" + i);
350      Put p = new Put(row);
351      p.addColumn(family, column, row);
352      t.put(p);
353    }
354  }
355
356  private void verifyRows(Table t, byte[] family, byte[] column) throws IOException {
357    for (int i = 0; i < 10; i++) {
358      byte[] row = Bytes.toBytes("row" + i);
359      Get g = new Get(row).addFamily(family);
360      Result r = t.get(g);
361      Assert.assertNotNull(r);
362      Assert.assertEquals(1, r.size());
363      Cell cell = r.rawCells()[0];
364      Assert.assertTrue(CellUtil.matchingQualifier(cell, column));
365      Assert.assertEquals(Bytes.compareTo(cell.getValueArray(), cell.getValueOffset(),
366        cell.getValueLength(), row, 0, row.length), 0);
367    }
368  }

370  private Table createTable(TableName tableName, byte[] family, boolean isMob) throws IOException {
371    if (isMob) {
372      ColumnFamilyDescriptor cfd = ColumnFamilyDescriptorBuilder.newBuilder(family)
373          .setMobEnabled(true).setMobThreshold(1).build();
374      TableDescriptor desc =
375          TableDescriptorBuilder.newBuilder(tableName).setColumnFamily(cfd).build();
376      return TEST_UTIL.createTable(desc, null);
377    } else {
378      return TEST_UTIL.createTable(tableName, family);
379    }
380  }

382  private void testCopyTableBySnapshot(String tablePrefix, boolean bulkLoad, boolean isMob)
383      throws Exception {
384    TableName table1 = TableName.valueOf(tablePrefix + 1);
385    TableName table2 = TableName.valueOf(tablePrefix + 2);
386    Table t1 = createTable(table1, FAMILY_A, isMob);
387    Table t2 = createTable(table2, FAMILY_A, isMob);
388    loadData(t1, FAMILY_A, Bytes.toBytes("qualifier"));
389    String snapshot = tablePrefix + "_snapshot";
390    TEST_UTIL.getAdmin().snapshot(snapshot, table1);
391    boolean success;
392    if (bulkLoad) {
393      success =
394          runCopy(new String[] { "--snapshot", "--new.name=" + table2, "--bulkload", snapshot });
395    } else {
396      success = runCopy(new String[] { "--snapshot", "--new.name=" + table2, snapshot });
397    }
398    Assert.assertTrue(success);
399    verifyRows(t2, FAMILY_A, Bytes.toBytes("qualifier"));
400  }
