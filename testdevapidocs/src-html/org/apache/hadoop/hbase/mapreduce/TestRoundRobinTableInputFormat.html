044/**
045 * Basic test of {@link RoundRobinTableInputFormat}; i.e. RRTIF.
046 */
047@Category({ SmallTests.class })
048public class TestRoundRobinTableInputFormat {
049  @ClassRule
050  public static final HBaseClassTestRule CLASS_RULE =
051    HBaseClassTestRule.forClass(TestRoundRobinTableInputFormat.class);
052
053  private static final int SERVERS_COUNT = 5;
054  private static final String[] KEYS = { "aa", "ab", "ac", "ad", "ae", "ba", "bb", "bc", "bd", "be",
055    "ca", "cb", "cc", "cd", "ce", "da", "db", "dc", "dd", "de", "ea", "eb", "ec", "ed", "ee", "fa",
056    "fb", "fc", "fd", "fe", "ga", "gb", "gc", "gd", "ge", "ha", "hb", "hc", "hd", "he", "ia", "ib",
057    "ic", "id", "ie", "ja", "jb", "jc", "jd", "je", "jf" };
058
059  /**
060   * Test default behavior.
061   */
062  @Test
063  public void testRoundRobinSplit() throws IOException, InterruptedException {
064    final List<InputSplit> splits = createSplits();
065    Collections.shuffle(splits);
066    List<InputSplit> sortedSplits = new RoundRobinTableInputFormat().roundRobin(splits);
067    testDistribution(sortedSplits);
068    // Now test that order is preserved even after being passed through the SplitComparator
069    // that sorts InputSplit by length as is done up in Hadoop in JobSubmitter.
070    List<InputSplit> copy = new ArrayList<>(sortedSplits);
071    Arrays.sort(copy.toArray(new InputSplit[0]), new SplitComparator());
072    // Assert the sort is retained even after passing through SplitComparator.
073    for (int i = 0; i < sortedSplits.size(); i++) {
074      TableSplit sortedTs = (TableSplit) sortedSplits.get(i);
075      TableSplit copyTs = (TableSplit) copy.get(i);
076      assertEquals(sortedTs.getEncodedRegionName(), copyTs.getEncodedRegionName());
077    }
078  }
079
080  /**
081   * @return Splits made out of {@link #KEYS}. Splits are for five Servers. Length is ZERO!
082   */
083  private List<InputSplit> createSplits() {
084    List<InputSplit> splits = new ArrayList<>(KEYS.length - 1);
085    for (int i = 0; i < KEYS.length - 1; i++) {
086      InputSplit split =
087        new TableSplit(TableName.valueOf("test"), new Scan(), Bytes.toBytes(KEYS[i]),
088          Bytes.toBytes(KEYS[i + 1]), String.valueOf(i % SERVERS_COUNT + 1), "", 0);
089      splits.add(split);
090    }
091    return splits;
092  }
093
094  private void testDistribution(List<InputSplit> list) throws IOException, InterruptedException {
095    for (int i = 0; i < KEYS.length / SERVERS_COUNT; i++) {
096      int[] counts = new int[SERVERS_COUNT];
097      for (int j = i * SERVERS_COUNT; j < i * SERVERS_COUNT + SERVERS_COUNT; j++) {
098        counts[Integer.parseInt(list.get(j).getLocations()[0]) - 1]++;
099      }
100      for (int value : counts) {
101        assertEquals(value, 1);
102      }
103    }
104  }
105
106  /**
107   * Private comparator copied from private JobSubmmiter Hadoop class...
108   * hadoop/hadoop-mapreduce-project/hadoop-mapreduce-client/hadoop-mapreduce-client-core/src/main/java/org/apache/hadoop/mapreduce/JobSubmitter.java
109   * Used so we can run the sort done up in JobSubmitter here in tests.
110   */
111  private static class SplitComparator implements Comparator<InputSplit> {
112    @Override
113    public int compare(InputSplit o1, InputSplit o2) {
114      try {
115        return Long.compare(o1.getLength(), o2.getLength());
116      } catch (IOException | InterruptedException e) {
117        throw new RuntimeException("exception in compare", e);
118      }
119    }
120  }
121
122  /**
123   * Assert that lengths are descending. RRTIF writes lengths in descending order so any subsequent
124   * sort using dump SplitComparator as is done in JobSubmitter up in Hadoop keeps our RRTIF
125   * ordering.
126   */
127  private void assertLengthDescending(List<InputSplit> list)
128    throws IOException, InterruptedException {
129    long previousLength = Long.MAX_VALUE;
130    for (InputSplit is : list) {
131      long length = is.getLength();
132      assertTrue(previousLength + " " + length, previousLength > length);
133      previousLength = length;
134    }
135  }
136
137  /**
138   * Test that configure/unconfigure set and properly undo the HBASE_REGIONSIZECALCULATOR_ENABLE
139   * configuration.
140   */
141  @Test
142  public void testConfigureUnconfigure() {
143    Configuration configuration = HBaseConfiguration.create();
144    RoundRobinTableInputFormat rrtif = new RoundRobinTableInputFormat();
145    rrtif.setConf(configuration);
146    JobContext jobContext = Mockito.mock(JobContext.class);
147    Mockito.when(jobContext.getConfiguration()).thenReturn(configuration);
148    // Assert when done, HBASE_REGIONSIZECALCULATOR_ENABLE is still unset.
149    configuration.unset(RoundRobinTableInputFormat.HBASE_REGIONSIZECALCULATOR_ENABLE);
150    rrtif.configure();
151    rrtif.unconfigure();
152    String value = configuration.get(RoundRobinTableInputFormat.HBASE_REGIONSIZECALCULATOR_ENABLE);
153    assertNull(value);
154    // Assert HBASE_REGIONSIZECALCULATOR_ENABLE is still false when done.
155    checkRetainsBooleanValue(jobContext, rrtif, false);
156    // Assert HBASE_REGIONSIZECALCULATOR_ENABLE is still true when done.
157    checkRetainsBooleanValue(jobContext, rrtif, true);
158  }
159
160  private void checkRetainsBooleanValue(JobContext jobContext, RoundRobinTableInputFormat rrtif,
161    final boolean b) {
162    jobContext.getConfiguration()
163      .setBoolean(RoundRobinTableInputFormat.HBASE_REGIONSIZECALCULATOR_ENABLE, b);
164    rrtif.configure();
165    rrtif.unconfigure();
166    String value = jobContext.getConfiguration()
167      .get(RoundRobinTableInputFormat.HBASE_REGIONSIZECALCULATOR_ENABLE);
168    assertEquals(b, Boolean.valueOf(value));
169  }
