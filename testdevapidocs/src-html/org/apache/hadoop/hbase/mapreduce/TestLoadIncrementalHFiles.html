308  private void runTest(String testName, HTableDescriptor htd, BloomType bloomType,
309      boolean preCreateTable, byte[][] tableSplitKeys, byte[][][] hfileRanges, boolean useMap,
310      boolean copyFiles) throws Exception {
311    Path dir = util.getDataTestDirOnTestFS(testName);
312    FileSystem fs = util.getTestFileSystem();
313    dir = dir.makeQualified(fs);
314    Path familyDir = new Path(dir, Bytes.toString(FAMILY));
315
316    int hfileIdx = 0;
317    Map<byte[], List<Path>> map = null;
318    List<Path> list = null;
319    if (useMap || copyFiles) {
320      list = new ArrayList<>();
321    }
322    if (useMap) {
323      map = new TreeMap<byte[], List<Path>>(Bytes.BYTES_COMPARATOR);
324      map.put(FAMILY, list);
325    }
326    for (byte[][] range : hfileRanges) {
327      byte[] from = range[0];
328      byte[] to = range[1];
329      Path path = new Path(familyDir, "hfile_" + hfileIdx++);
330      HFileTestUtil.createHFile(util.getConfiguration(), fs, path, FAMILY, QUALIFIER, from, to, 1000);
331      if (useMap) {
332        list.add(path);
333      }
334    }
335    int expectedRows = hfileIdx * 1000;
336
337    if (preCreateTable || map != null) {
338      util.getHBaseAdmin().createTable(htd, tableSplitKeys);
339    }
340
341    final TableName tableName = htd.getTableName();
342    Configuration conf = util.getConfiguration();
343    if (copyFiles) {
344      conf.setBoolean(LoadIncrementalHFiles.ALWAYS_COPY_FILES, true);
345    }
346    LoadIncrementalHFiles loader = new LoadIncrementalHFiles(conf);
347    String [] args= {dir.toString(), tableName.toString()};
348    if (useMap) {
349      loader.run(null, map, tableName);
350    } else {
351      loader.run(args);
352    }
353
354    if (copyFiles) {
355      for (Path p : list) {
356        assertTrue(fs.exists(p));
357      }
358    }
359
360    Table table = util.getConnection().getTable(tableName);
361    try {
362      assertEquals(expectedRows, util.countRows(table));
363    } finally {
364      table.close();
365    }
366
367    // verify staging folder has been cleaned up
368    Path stagingBasePath = new Path(FSUtils.getRootDir(util.getConfiguration()), HConstants.BULKLOAD_STAGING_DIR_NAME);
369    if(fs.exists(stagingBasePath)) {
370      FileStatus[] files = fs.listStatus(stagingBasePath);
371      for(FileStatus file : files) {
372        assertTrue("Folder=" + file.getPath() + " is not cleaned up.",
373          file.getPath().getName() != "DONOTERASE");
374      }
375    }
376
377    util.deleteTable(tableName);
378  }

507  private static void createRandomDataFile(FileSystem fs, Path path, int size)
508      throws IOException {
509    FSDataOutputStream stream = fs.create(path);
510    try {
511      byte[] data = new byte[1024];
512      for (int i = 0; i < data.length; ++i) {
513        data[i] = (byte)(i & 0xff);
514      }
515      while (size >= data.length) {
516        stream.write(data, 0, data.length);
517        size -= data.length;
518      }
519      if (size > 0) {
520        stream.write(data, 0, size);
521      }
522    } finally {
523      stream.close();
524    }
525  }

595  private int verifyHFile(Path p) throws IOException {
596    Configuration conf = util.getConfiguration();
597    HFile.Reader reader = HFile.createReader(
598        p.getFileSystem(conf), p, new CacheConfig(conf), conf);
599    reader.loadFileInfo();
600    HFileScanner scanner = reader.getScanner(false, false);
601    scanner.seekTo();
602    int count = 0;
603    do {
604      count++;
605    } while (scanner.next());
606    assertTrue(count > 0);
607    reader.close();
608    return count;
609  }

620  public void testInferBoundaries() {
621    TreeMap<byte[], Integer> map = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
622
623    /* Toy example
624     *     c---------i            o------p          s---------t     v------x
625     * a------e    g-----k   m-------------q   r----s            u----w
626     *
627     * Should be inferred as:
628     * a-----------------k   m-------------q   r--------------t  u---------x
629     *
630     * The output should be (m,r,u)
631     */
632
633    String first;
634    String last;
635
636    first = "a"; last = "e";
637    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
638
639    first = "r"; last = "s";
640    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
641
642    first = "o"; last = "p";
643    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
644
645    first = "g"; last = "k";
646    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
647
648    first = "v"; last = "x";
649    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
650
651    first = "c"; last = "i";
652    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
653
654    first = "m"; last = "q";
655    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
656
657    first = "s"; last = "t";
658    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
659
660    first = "u"; last = "w";
661    addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
662
663    byte[][] keysArray = LoadIncrementalHFiles.inferBoundaries(map);
664    byte[][] compare = new byte[3][];
665    compare[0] = "m".getBytes();
666    compare[1] = "r".getBytes();
667    compare[2] = "u".getBytes();
668
669    assertEquals(keysArray.length, 3);
670
671    for (int row = 0; row<keysArray.length; row++){
672      assertArrayEquals(keysArray[row], compare[row]);
673    }
