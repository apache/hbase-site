019package org.apache.hadoop.hbase.mapreduce;
020
021import org.apache.hadoop.hbase.shaded.com.google.common.base.Function;
022import org.apache.hadoop.hbase.shaded.com.google.common.collect.ImmutableList;
023import org.apache.hadoop.hbase.shaded.com.google.common.collect.Multimaps;

027import org.apache.hadoop.hbase.client.Scan;
028import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
029import org.apache.hadoop.hbase.snapshot.SnapshotTestingUtils;
030import org.apache.hadoop.hbase.testclassification.LargeTests;
031import org.apache.hadoop.hbase.testclassification.VerySlowMapReduceTests;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.hbase.util.FSUtils;
034import org.apache.hadoop.mapreduce.Job;
035import org.junit.Before;
036import org.junit.BeforeClass;
037import org.junit.experimental.categories.Category;
038
039import java.io.IOException;
040import java.util.Collection;
041import java.util.List;
042import java.util.Map;
043
044@Category({ VerySlowMapReduceTests.class, LargeTests.class })
045public class TestMultiTableSnapshotInputFormat extends MultiTableInputFormatTestBase {
046
047  protected Path restoreDir;
048
049  @BeforeClass
050  public static void setUpSnapshots() throws Exception {
051
052    TEST_UTIL.enableDebug(MultiTableSnapshotInputFormat.class);
053    TEST_UTIL.enableDebug(MultiTableSnapshotInputFormatImpl.class);
054
055    // take a snapshot of every table we have.
056    for (String tableName : TABLES) {
057      SnapshotTestingUtils
058          .createSnapshotAndValidate(TEST_UTIL.getAdmin(), TableName.valueOf(tableName),
059              ImmutableList.of(MultiTableInputFormatTestBase.INPUT_FAMILY), null,
060              snapshotNameForTable(tableName), FSUtils.getRootDir(TEST_UTIL.getConfiguration()),
061              TEST_UTIL.getTestFileSystem(), true);
062    }
063  }
064
065  @Before
066  public void setUp() throws Exception {
067    this.restoreDir = TEST_UTIL.getRandomDir();
068  }
069
070  @Override
071  protected void initJob(List<Scan> scans, Job job) throws IOException {
072    TableMapReduceUtil
073        .initMultiTableSnapshotMapperJob(getSnapshotScanMapping(scans), ScanMapper.class,
074            ImmutableBytesWritable.class, ImmutableBytesWritable.class, job, true, restoreDir);
075  }
076
077  protected Map<String, Collection<Scan>> getSnapshotScanMapping(final List<Scan> scans) {
078    return Multimaps.index(scans, new Function<Scan, String>() {
079      @Nullable
080      @Override
081      public String apply(Scan input) {
082        return snapshotNameForTable(
083            Bytes.toStringBinary(input.getAttribute(Scan.SCAN_ATTRIBUTES_TABLE_NAME)));
084      }
085    }).asMap();
086  }
087
088  public static String snapshotNameForTable(String tableName) {
089    return tableName + "_snapshot";
090  }
091
092}
