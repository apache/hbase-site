078  private static byte[][] generateSplits(int numRows, int numRegions) {
079    byte[][] splitRows = new byte[numRegions-1][];
080    for (int i = 1; i < numRegions; i++) {
081      splitRows[i-1] = Bytes.toBytes(numRows * i / numRegions);
082    }
083    return splitRows;
084  }

086  @Test
087  public void testSyncTable() throws Exception {
088    final TableName sourceTableName = TableName.valueOf(name.getMethodName() + "_source");
089    final TableName targetTableName = TableName.valueOf(name.getMethodName() + "_target");
090    Path testDir = TEST_UTIL.getDataTestDirOnTestFS("testSyncTable");
091
092    writeTestData(sourceTableName, targetTableName);
093    hashSourceTable(sourceTableName, testDir);
094    Counters syncCounters = syncTables(sourceTableName, targetTableName, testDir);
095    assertEqualTables(90, sourceTableName, targetTableName);
096
097    assertEquals(60, syncCounters.findCounter(Counter.ROWSWITHDIFFS).getValue());
098    assertEquals(10, syncCounters.findCounter(Counter.SOURCEMISSINGROWS).getValue());
099    assertEquals(10, syncCounters.findCounter(Counter.TARGETMISSINGROWS).getValue());
100    assertEquals(50, syncCounters.findCounter(Counter.SOURCEMISSINGCELLS).getValue());
101    assertEquals(50, syncCounters.findCounter(Counter.TARGETMISSINGCELLS).getValue());
102    assertEquals(20, syncCounters.findCounter(Counter.DIFFERENTCELLVALUES).getValue());
103
104    TEST_UTIL.deleteTable(sourceTableName);
105    TEST_UTIL.deleteTable(targetTableName);
106    TEST_UTIL.cleanupDataTestDirOnTestFS();
107  }

109  private void assertEqualTables(int expectedRows, TableName sourceTableName,
110      TableName targetTableName) throws Exception {
111    Table sourceTable = TEST_UTIL.getConnection().getTable(sourceTableName);
112    Table targetTable = TEST_UTIL.getConnection().getTable(targetTableName);
113
114    ResultScanner sourceScanner = sourceTable.getScanner(new Scan());
115    ResultScanner targetScanner = targetTable.getScanner(new Scan());
116
117    for (int i = 0; i < expectedRows; i++) {
118      Result sourceRow = sourceScanner.next();
119      Result targetRow = targetScanner.next();
120
121      LOG.debug("SOURCE row: " + (sourceRow == null ? "null" : Bytes.toInt(sourceRow.getRow()))
122          + " cells:" + sourceRow);
123      LOG.debug("TARGET row: " + (targetRow == null ? "null" : Bytes.toInt(targetRow.getRow()))
124          + " cells:" + targetRow);
125
126      if (sourceRow == null) {
127        Assert.fail("Expected " + expectedRows
128            + " source rows but only found " + i);
129      }
130      if (targetRow == null) {
131        Assert.fail("Expected " + expectedRows
132            + " target rows but only found " + i);
133      }
134      Cell[] sourceCells = sourceRow.rawCells();
135      Cell[] targetCells = targetRow.rawCells();
136      if (sourceCells.length != targetCells.length) {
137        LOG.debug("Source cells: " + Arrays.toString(sourceCells));
138        LOG.debug("Target cells: " + Arrays.toString(targetCells));
139        Assert.fail("Row " + Bytes.toInt(sourceRow.getRow())
140            + " has " + sourceCells.length
141            + " cells in source table but " + targetCells.length
142            + " cells in target table");
143      }
144      for (int j = 0; j < sourceCells.length; j++) {
145        Cell sourceCell = sourceCells[j];
146        Cell targetCell = targetCells[j];
147        try {
148          if (!CellUtil.matchingRow(sourceCell, targetCell)) {
149            Assert.fail("Rows don't match");
150          }
151          if (!CellUtil.matchingFamily(sourceCell, targetCell)) {
152            Assert.fail("Families don't match");
153          }
154          if (!CellUtil.matchingQualifier(sourceCell, targetCell)) {
155            Assert.fail("Qualifiers don't match");
156          }
157          if (!CellUtil.matchingTimestamp(sourceCell, targetCell)) {
158            Assert.fail("Timestamps don't match");
159          }
160          if (!CellUtil.matchingValue(sourceCell, targetCell)) {
161            Assert.fail("Values don't match");
162          }
163        } catch (Throwable t) {
164          LOG.debug("Source cell: " + sourceCell + " target cell: " + targetCell);
165          Throwables.propagate(t);
166        }
167      }
168    }
169    Result sourceRow = sourceScanner.next();
170    if (sourceRow != null) {
171      Assert.fail("Source table has more than " + expectedRows
172          + " rows.  Next row: " + Bytes.toInt(sourceRow.getRow()));
173    }
174    Result targetRow = targetScanner.next();
175    if (targetRow != null) {
176      Assert.fail("Target table has more than " + expectedRows
177          + " rows.  Next row: " + Bytes.toInt(targetRow.getRow()));
178    }
179    sourceScanner.close();
180    targetScanner.close();
181    sourceTable.close();
182    targetTable.close();
183  }

185  private Counters syncTables(TableName sourceTableName, TableName targetTableName,
186      Path testDir) throws Exception {
187    SyncTable syncTable = new SyncTable(TEST_UTIL.getConfiguration());
188    int code = syncTable.run(new String[] {
189        testDir.toString(),
190        sourceTableName.getNameAsString(),
191        targetTableName.getNameAsString()
192        });
193    assertEquals("sync table job failed", 0, code);
194
195    LOG.info("Sync tables completed");
196    return syncTable.counters;
197  }

199  private void hashSourceTable(TableName sourceTableName, Path testDir)
200      throws Exception, IOException {
201    int numHashFiles = 3;
202    long batchSize = 100;  // should be 2 batches per region
203    int scanBatch = 1;
204    HashTable hashTable = new HashTable(TEST_UTIL.getConfiguration());
205    int code = hashTable.run(new String[] {
206        "--batchsize=" + batchSize,
207        "--numhashfiles=" + numHashFiles,
208        "--scanbatch=" + scanBatch,
209        sourceTableName.getNameAsString(),
210        testDir.toString()});
211    assertEquals("hash table job failed", 0, code);
212
213    FileSystem fs = TEST_UTIL.getTestFileSystem();
214
215    HashTable.TableHash tableHash = HashTable.TableHash.read(fs.getConf(), testDir);
216    assertEquals(sourceTableName.getNameAsString(), tableHash.tableName);
217    assertEquals(batchSize, tableHash.batchSize);
218    assertEquals(numHashFiles, tableHash.numHashFiles);
219    assertEquals(numHashFiles - 1, tableHash.partitions.size());
220
221    LOG.info("Hash table completed");
222  }

224  private void writeTestData(TableName sourceTableName, TableName targetTableName)
225      throws Exception {
226    final byte[] family = Bytes.toBytes("family");
227    final byte[] column1 = Bytes.toBytes("c1");
228    final byte[] column2 = Bytes.toBytes("c2");
229    final byte[] value1 = Bytes.toBytes("val1");
230    final byte[] value2 = Bytes.toBytes("val2");
231    final byte[] value3 = Bytes.toBytes("val3");
232
233    int numRows = 100;
234    int sourceRegions = 10;
235    int targetRegions = 6;
236
237    Table sourceTable = TEST_UTIL.createTable(sourceTableName,
238        family, generateSplits(numRows, sourceRegions));
239
240    Table targetTable = TEST_UTIL.createTable(targetTableName,
241        family, generateSplits(numRows, targetRegions));
242
243    long timestamp = 1430764183454L;
244
245    int rowIndex = 0;
246    // a bunch of identical rows
247    for (; rowIndex < 40; rowIndex++) {
248      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
249      sourcePut.addColumn(family, column1, timestamp, value1);
250      sourcePut.addColumn(family, column2, timestamp, value2);
251      sourceTable.put(sourcePut);
252
253      Put targetPut = new Put(Bytes.toBytes(rowIndex));
254      targetPut.addColumn(family, column1, timestamp, value1);
255      targetPut.addColumn(family, column2, timestamp, value2);
256      targetTable.put(targetPut);
257    }
258    // some rows only in the source table
259    // ROWSWITHDIFFS: 10
260    // TARGETMISSINGROWS: 10
261    // TARGETMISSINGCELLS: 20
262    for (; rowIndex < 50; rowIndex++) {
263      Put put = new Put(Bytes.toBytes(rowIndex));
264      put.addColumn(family, column1, timestamp, value1);
265      put.addColumn(family, column2, timestamp, value2);
266      sourceTable.put(put);
267    }
268    // some rows only in the target table
269    // ROWSWITHDIFFS: 10
270    // SOURCEMISSINGROWS: 10
271    // SOURCEMISSINGCELLS: 20
272    for (; rowIndex < 60; rowIndex++) {
273      Put put = new Put(Bytes.toBytes(rowIndex));
274      put.addColumn(family, column1, timestamp, value1);
275      put.addColumn(family, column2, timestamp, value2);
276      targetTable.put(put);
277    }
278    // some rows with 1 missing cell in target table
279    // ROWSWITHDIFFS: 10
280    // TARGETMISSINGCELLS: 10
281    for (; rowIndex < 70; rowIndex++) {
282      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
283      sourcePut.addColumn(family, column1, timestamp, value1);
284      sourcePut.addColumn(family, column2, timestamp, value2);
285      sourceTable.put(sourcePut);
286
287      Put targetPut = new Put(Bytes.toBytes(rowIndex));
288      targetPut.addColumn(family, column1, timestamp, value1);
289      targetTable.put(targetPut);
290    }
291    // some rows with 1 missing cell in source table
292    // ROWSWITHDIFFS: 10
293    // SOURCEMISSINGCELLS: 10
294    for (; rowIndex < 80; rowIndex++) {
295      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
296      sourcePut.addColumn(family, column1, timestamp, value1);
297      sourceTable.put(sourcePut);
298
299      Put targetPut = new Put(Bytes.toBytes(rowIndex));
300      targetPut.addColumn(family, column1, timestamp, value1);
301      targetPut.addColumn(family, column2, timestamp, value2);
302      targetTable.put(targetPut);
303    }
304    // some rows differing only in timestamp
305    // ROWSWITHDIFFS: 10
306    // SOURCEMISSINGCELLS: 20
307    // TARGETMISSINGCELLS: 20
308    for (; rowIndex < 90; rowIndex++) {
309      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
310      sourcePut.addColumn(family, column1, timestamp, column1);
311      sourcePut.addColumn(family, column2, timestamp, value2);
312      sourceTable.put(sourcePut);
313
314      Put targetPut = new Put(Bytes.toBytes(rowIndex));
315      targetPut.addColumn(family, column1, timestamp+1, column1);
316      targetPut.addColumn(family, column2, timestamp-1, value2);
317      targetTable.put(targetPut);
318    }
319    // some rows with different values
320    // ROWSWITHDIFFS: 10
321    // DIFFERENTCELLVALUES: 20
322    for (; rowIndex < numRows; rowIndex++) {
323      Put sourcePut = new Put(Bytes.toBytes(rowIndex));
324      sourcePut.addColumn(family, column1, timestamp, value1);
325      sourcePut.addColumn(family, column2, timestamp, value2);
326      sourceTable.put(sourcePut);
327
328      Put targetPut = new Put(Bytes.toBytes(rowIndex));
329      targetPut.addColumn(family, column1, timestamp, value3);
330      targetPut.addColumn(family, column2, timestamp, value3);
331      targetTable.put(targetPut);
332    }
333
334    sourceTable.close();
335    targetTable.close();
336  }
