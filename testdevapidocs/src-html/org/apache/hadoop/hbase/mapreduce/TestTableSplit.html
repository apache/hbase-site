018package org.apache.hadoop.hbase.mapreduce;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023import java.util.HashSet;
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.HBaseClassTestRule;
026import org.apache.hadoop.hbase.TableName;
027import org.apache.hadoop.hbase.testclassification.MapReduceTests;
028import org.apache.hadoop.hbase.testclassification.SmallTests;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.apache.hadoop.util.ReflectionUtils;
031import org.junit.Assert;
032import org.junit.ClassRule;
033import org.junit.Rule;
034import org.junit.Test;
035import org.junit.experimental.categories.Category;
036import org.junit.rules.TestName;
037
038@Category({MapReduceTests.class, SmallTests.class})
039public class TestTableSplit {
040
041  @ClassRule
042  public static final HBaseClassTestRule CLASS_RULE =
043      HBaseClassTestRule.forClass(TestTableSplit.class);
044
045  @Rule
046  public TestName name = new TestName();
047
048  @Test
049  public void testHashCode() {
050    TableSplit split1 = new TableSplit(TableName.valueOf(name.getMethodName()),
051        Bytes.toBytes("row-start"),
052            Bytes.toBytes("row-end"), "location");
053    TableSplit split2 = new TableSplit(TableName.valueOf(name.getMethodName()),
054        Bytes.toBytes("row-start"),
055            Bytes.toBytes("row-end"), "location");
056    assertEquals (split1, split2);
057    assertTrue   (split1.hashCode() == split2.hashCode());
058    HashSet<TableSplit> set = new HashSet<>(2);
059    set.add(split1);
060    set.add(split2);
061    assertTrue(set.size() == 1);
062  }
063
064  /**
065   * length of region should not influence hashcode
066   * */
067  @Test
068  public void testHashCode_length() {
069    TableSplit split1 = new TableSplit(TableName.valueOf(name.getMethodName()),
070        Bytes.toBytes("row-start"),
071            Bytes.toBytes("row-end"), "location", 1984);
072    TableSplit split2 = new TableSplit(TableName.valueOf(name.getMethodName()),
073        Bytes.toBytes("row-start"),
074            Bytes.toBytes("row-end"), "location", 1982);
075
076    assertEquals (split1, split2);
077    assertTrue   (split1.hashCode() == split2.hashCode());
078    HashSet<TableSplit> set = new HashSet<>(2);
079    set.add(split1);
080    set.add(split2);
081    assertTrue(set.size() == 1);
082  }
083
084  /**
085   * Length of region need to be properly serialized.
086   * */
087  @Test
088  public void testLengthIsSerialized() throws Exception {
089    TableSplit split1 = new TableSplit(TableName.valueOf(name.getMethodName()),
090        Bytes.toBytes("row-start"),
091            Bytes.toBytes("row-end"), "location", 666);
092
093    TableSplit deserialized = new TableSplit(TableName.valueOf(name.getMethodName()),
094        Bytes.toBytes("row-start2"),
095            Bytes.toBytes("row-end2"), "location1");
096    ReflectionUtils.copy(new Configuration(), split1, deserialized);
097
098    Assert.assertEquals(666, deserialized.getLength());
099  }
100
101  @Test
102  public void testToString() {
103    TableSplit split =
104        new TableSplit(TableName.valueOf(name.getMethodName()), Bytes.toBytes("row-start"),
105            Bytes.toBytes("row-end"), "location");
106    String str =
107        "HBase table split(table name: " + name.getMethodName() + ", scan: , start row: row-start, "
108            + "end row: row-end, region location: location, "
109            + "encoded region name: )";
110    Assert.assertEquals(str, split.toString());
111
112    split =
113        new TableSplit(TableName.valueOf(name.getMethodName()), null, Bytes.toBytes("row-start"),
114            Bytes.toBytes("row-end"), "location", "encoded-region-name", 1000L);
115    str =
116        "HBase table split(table name: " + name.getMethodName() + ", scan: , start row: row-start, "
117            + "end row: row-end, region location: location, "
118            + "encoded region name: encoded-region-name)";
119    Assert.assertEquals(str, split.toString());
120
121    split = new TableSplit((TableName) null, null, null, null);
122    str =
123        "HBase table split(table name: null, scan: , start row: null, "
124            + "end row: null, region location: null, "
125            + "encoded region name: )";
126    Assert.assertEquals(str, split.toString());
127
128    split = new TableSplit((TableName) null, null, null, null, null, null, 1000L);
129    str =
130        "HBase table split(table name: null, scan: , start row: null, "
131            + "end row: null, region location: null, "
132            + "encoded region name: null)";
133    Assert.assertEquals(str, split.toString());
134  }
135}
