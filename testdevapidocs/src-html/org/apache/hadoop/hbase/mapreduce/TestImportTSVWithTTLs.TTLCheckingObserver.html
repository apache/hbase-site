056@Category({MapReduceTests.class, LargeTests.class})
057public class TestImportTSVWithTTLs implements Configurable {
058
059  protected static final Log LOG = LogFactory.getLog(TestImportTSVWithTTLs.class);
060  protected static final String NAME = TestImportTsv.class.getSimpleName();
061  protected static HBaseTestingUtility util = new HBaseTestingUtility();
062
063  /**
064   * Delete the tmp directory after running doMROnTableTest. Boolean. Default is
065   * false.
066   */
067  protected static final String DELETE_AFTER_LOAD_CONF = NAME + ".deleteAfterLoad";
068
069  /**
070   * Force use of combiner in doMROnTableTest. Boolean. Default is true.
071   */
072  protected static final String FORCE_COMBINER_CONF = NAME + ".forceCombiner";
073
074  private final String FAMILY = "FAM";
075  private static Configuration conf;
076
077  @Rule
078  public TestName name = new TestName();
079
080  @Override
081  public Configuration getConf() {
082    return util.getConfiguration();
083  }
084
085  @Override
086  public void setConf(Configuration conf) {
087    throw new IllegalArgumentException("setConf not supported");
088  }
089
090  @BeforeClass
091  public static void provisionCluster() throws Exception {
092    conf = util.getConfiguration();
093    // We don't check persistence in HFiles in this test, but if we ever do we will
094    // need this where the default hfile version is not 3 (i.e. 0.98)
095    conf.setInt("hfile.format.version", 3);
096    conf.set("hbase.coprocessor.region.classes", TTLCheckingObserver.class.getName());
097    util.startMiniCluster();
098  }
099
100  @AfterClass
101  public static void releaseCluster() throws Exception {
102    util.shutdownMiniCluster();
103  }
104
105  @Test
106  public void testMROnTable() throws Exception {
107    final TableName tableName = TableName.valueOf(name.getMethodName() + UUID.randomUUID());
108
109    // Prepare the arguments required for the test.
110    String[] args = new String[] {
111        "-D" + ImportTsv.MAPPER_CONF_KEY
112            + "=org.apache.hadoop.hbase.mapreduce.TsvImporterMapper",
113        "-D" + ImportTsv.COLUMNS_CONF_KEY + "=HBASE_ROW_KEY,FAM:A,FAM:B,HBASE_CELL_TTL",
114        "-D" + ImportTsv.SEPARATOR_CONF_KEY + "=\u001b", tableName.getNameAsString() };
115    String data = "KEY\u001bVALUE1\u001bVALUE2\u001b1000000\n";
116    util.createTable(tableName, FAMILY);
117    doMROnTableTest(util, FAMILY, data, args, 1);
118    util.deleteTable(tableName);
119  }
120
121  protected static Tool doMROnTableTest(HBaseTestingUtility util, String family, String data,
122      String[] args, int valueMultiplier) throws Exception {
123    TableName table = TableName.valueOf(args[args.length - 1]);
124    Configuration conf = new Configuration(util.getConfiguration());
125
126    // populate input file
127    FileSystem fs = FileSystem.get(conf);
128    Path inputPath = fs.makeQualified(new Path(util
129        .getDataTestDirOnTestFS(table.getNameAsString()), "input.dat"));
130    FSDataOutputStream op = fs.create(inputPath, true);
131    op.write(Bytes.toBytes(data));
132    op.close();
133    LOG.debug(String.format("Wrote test data to file: %s", inputPath));
134
135    if (conf.getBoolean(FORCE_COMBINER_CONF, true)) {
136      LOG.debug("Forcing combiner.");
137      conf.setInt("mapreduce.map.combine.minspills", 1);
138    }
139
140    // run the import
141    List<String> argv = new ArrayList<String>(Arrays.asList(args));
142    argv.add(inputPath.toString());
143    Tool tool = new ImportTsv();
144    LOG.debug("Running ImportTsv with arguments: " + argv);
145    try {
146      // Job will fail if observer rejects entries without TTL
147      assertEquals(0, ToolRunner.run(conf, tool, argv.toArray(args)));
148    } finally {
149      // Clean up
150      if (conf.getBoolean(DELETE_AFTER_LOAD_CONF, true)) {
151        LOG.debug("Deleting test subdirectory");
152        util.cleanupDataTestDirOnTestFS(table.getNameAsString());
153      }
154    }
155
156    return tool;
157  }
158
159  public static class TTLCheckingObserver extends BaseRegionObserver {
160
161    @Override
162    public void prePut(ObserverContext<RegionCoprocessorEnvironment> e, Put put, WALEdit edit,
163        Durability durability) throws IOException {
164      Region region = e.getEnvironment().getRegion();
165      if (!region.getRegionInfo().isMetaTable()
166          && !region.getRegionInfo().getTable().isSystemTable()) {
167        // The put carries the TTL attribute
168        if (put.getTTL() != Long.MAX_VALUE) {
169          return;
170        }
171        throw new IOException("Operation does not have TTL set");
172      }
173    }
174  }
