016package org.apache.hadoop.hbase.mapreduce;
017
018import org.apache.hadoop.conf.Configuration;
019import org.apache.hadoop.hbase.HBaseTestingUtility;
020import org.apache.hadoop.hbase.TableName;
021import org.apache.hadoop.hbase.testclassification.MapReduceTests;
022import org.apache.hadoop.hbase.testclassification.MediumTests;
023import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
024import org.apache.hadoop.hbase.util.Bytes;
025import org.junit.AfterClass;
026import org.junit.BeforeClass;
027import org.junit.Rule;
028import org.junit.Test;
029import org.junit.experimental.categories.Category;
030import org.junit.rules.TestName;
031
032import static org.junit.Assert.assertEquals;
033
034@Category({MapReduceTests.class, MediumTests.class})
035public class TestHRegionPartitioner {
036  private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
037
038  @Rule
039  public TestName name = new TestName();
040
041  @BeforeClass
042  public static void beforeClass() throws Exception {
043    UTIL.startMiniCluster();
044  }
045
046  @AfterClass
047  public static void afterClass() throws Exception {
048    UTIL.shutdownMiniCluster();
049  }
050
051  /**
052   * Test HRegionPartitioner
053   */
054  @Test (timeout=300000)
055  public void testHRegionPartitioner() throws Exception {
056
057    byte[][] families = { Bytes.toBytes("familyA"), Bytes.toBytes("familyB") };
058
059    UTIL.createTable(TableName.valueOf(name.getMethodName()), families, 1,
060    Bytes.toBytes("aa"), Bytes.toBytes("cc"), 3);
061
062    HRegionPartitioner<Long, Long> partitioner = new HRegionPartitioner<Long, Long>();
063    Configuration configuration = UTIL.getConfiguration();
064    configuration.set(TableOutputFormat.OUTPUT_TABLE, name.getMethodName());
065    partitioner.setConf(configuration);
066    ImmutableBytesWritable writable = new ImmutableBytesWritable(Bytes.toBytes("bb"));
067
068    assertEquals(1, partitioner.getPartition(writable, 10L, 3));
069    assertEquals(0, partitioner.getPartition(writable, 10L, 1));
070  }
071}
