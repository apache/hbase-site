104  public void testPlainCompare() throws Exception {
105    final byte [] a = Bytes.toBytes("aaa");
106    final byte [] b = Bytes.toBytes("bbb");
107    final byte [] fam = Bytes.toBytes("col");
108    final byte [] qf = Bytes.toBytes("umn");
109    KeyValue aaa = new KeyValue(a, fam, qf, a);
110    KeyValue bbb = new KeyValue(b, fam, qf, b);
111    assertTrue(CellComparator.COMPARATOR.compare(aaa, bbb) < 0);
112    assertTrue(CellComparator.COMPARATOR.compare(bbb, aaa) > 0);
113    // Compare breaks if passed same ByteBuffer as both left and right arguments.
114    assertTrue(CellComparator.COMPARATOR.compare(bbb, bbb) == 0);
115    assertTrue(CellComparator.COMPARATOR.compare(aaa, aaa) == 0);
116    // Do compare with different timestamps.
117    aaa = new KeyValue(a, fam, qf, 1, a);
118    bbb = new KeyValue(a, fam, qf, 2, a);
119    assertTrue(CellComparator.COMPARATOR.compare(aaa, bbb) > 0);
120    assertTrue(CellComparator.COMPARATOR.compare(bbb, aaa) < 0);
121    assertTrue(CellComparator.COMPARATOR.compare(aaa, aaa) == 0);
122    // Do compare with different types.  Higher numbered types -- Delete
123    // should sort ahead of lower numbers; i.e. Put
124    aaa = new KeyValue(a, fam, qf, 1, KeyValue.Type.Delete, a);
125    bbb = new KeyValue(a, fam, qf, 1, a);
126    assertTrue(CellComparator.COMPARATOR.compare(aaa, bbb) < 0);
127    assertTrue(CellComparator.COMPARATOR.compare(bbb, aaa) > 0);
128    assertTrue(CellComparator.COMPARATOR.compare(aaa, aaa) == 0);
129  }

290  public void testCompareWithoutRow() {
291    final CellComparator c = CellComparator.COMPARATOR;
292    byte[] row = Bytes.toBytes("row");
293
294    byte[] fa = Bytes.toBytes("fa");
295    byte[] fami = Bytes.toBytes("fami");
296    byte[] fami1 = Bytes.toBytes("fami1");
297
298    byte[] qual0 = Bytes.toBytes("");
299    byte[] qual1 = Bytes.toBytes("qf1");
300    byte[] qual2 = Bytes.toBytes("qf2");
301    long ts = 1;
302
303    // 'fa:'
304    KeyValue kv_0 = new KeyValue(row, fa, qual0, ts, Type.Put);
305    // 'fami:'
306    KeyValue kv0_0 = new KeyValue(row, fami, qual0, ts, Type.Put);
307    // 'fami:qf1'
308    KeyValue kv0_1 = new KeyValue(row, fami, qual1, ts, Type.Put);
309    // 'fami:qf2'
310    KeyValue kv0_2 = new KeyValue(row, fami, qual2, ts, Type.Put);
311    // 'fami1:'
312    KeyValue kv1_0 = new KeyValue(row, fami1, qual0, ts, Type.Put);
313
314    // 'fami:qf1' < 'fami:qf2'
315    assertKVLessWithoutRow(c, 0, kv0_1, kv0_2);
316    // 'fami:qf1' < 'fami1:'
317    assertKVLessWithoutRow(c, 0, kv0_1, kv1_0);
318
319    // Test comparison by skipping the same prefix bytes.
320    /***
321     * KeyValue Format and commonLength:
322     * |_keyLen_|_valLen_|_rowLen_|_rowKey_|_famiLen_|_fami_|_Quali_|....
323     * ------------------|-------commonLength--------|--------------
324     */
325    int commonLength = KeyValue.ROW_LENGTH_SIZE + KeyValue.FAMILY_LENGTH_SIZE
326        + row.length;
327    // 'fa:' < 'fami:'. They have commonPrefix + 2 same prefix bytes.
328    assertKVLessWithoutRow(c, commonLength + 2, kv_0, kv0_0);
329    // 'fami:' < 'fami:qf1'. They have commonPrefix + 4 same prefix bytes.
330    assertKVLessWithoutRow(c, commonLength + 4, kv0_0, kv0_1);
331    // 'fami:qf1' < 'fami1:'. They have commonPrefix + 4 same prefix bytes.
332    assertKVLessWithoutRow(c, commonLength + 4, kv0_1, kv1_0);
333    // 'fami:qf1' < 'fami:qf2'. They have commonPrefix + 6 same prefix bytes.
334    assertKVLessWithoutRow(c, commonLength + 6, kv0_1, kv0_2);
335  }

565    KeyValue kvA1 = new KeyValue(Bytes.toBytes("key"), Bytes.toBytes("cf"), Bytes.toBytes("qualA"),
566        Bytes.toBytes("1"));
567    KeyValue kvA2 = new KeyValue(Bytes.toBytes("key"), Bytes.toBytes("cf"), Bytes.toBytes("qualA"),

570    MockKeyValue mkvA2 = new MockKeyValue(kvA2);
571    ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
572    DataOutputStream os = new DataOutputStream(byteArrayOutputStream);
573    ByteBufferUtils.putInt(os, KeyValueUtil.getSerializedSize(mkvA1, true));
574    KeyValueUtil.oswrite(mkvA1, os, true);
575    ByteBufferUtils.putInt(os, KeyValueUtil.getSerializedSize(mkvA2, true));
576    KeyValueUtil.oswrite(mkvA2, os, true);
577    DataInputStream is = new DataInputStream(new ByteArrayInputStream(
578        byteArrayOutputStream.toByteArray()));
579    KeyValue deSerKV1 = KeyValueUtil.iscreate(is, true);
580    assertTrue(kvA1.equals(deSerKV1));
581    KeyValue deSerKV2 = KeyValueUtil.iscreate(is, true);
582    assertTrue(kvA2.equals(deSerKV2));
