048/**
049 * Test HBASE-3694 whether the GlobalMemStoreSize is the same as the summary
050 * of all the online region's MemStoreSize
051 */
052@Category({MiscTests.class, MediumTests.class})
053public class TestGlobalMemStoreSize {
054
055  @ClassRule
056  public static final HBaseClassTestRule CLASS_RULE =
057      HBaseClassTestRule.forClass(TestGlobalMemStoreSize.class);
058
059  private static final Logger LOG = LoggerFactory.getLogger(TestGlobalMemStoreSize.class);
060  private static int regionServerNum = 4;
061  private static int regionNum = 16;
062  // total region num = region num + meta regions
063  private static int totalRegionNum = regionNum + 1;
064
065  private HBaseTestingUtil TEST_UTIL;
066  private SingleProcessHBaseCluster cluster;
067
068  @Rule
069  public TestName name = new TestName();
070
071  /**
072   * Test the global mem store size in the region server is equal to sum of each
073   * region's mem store size
074   * @throws Exception
075   */
076  @Test
077  public void testGlobalMemStore() throws Exception {
078    // Start the cluster
079    LOG.info("Starting cluster");
080    Configuration conf = HBaseConfiguration.create();
081    TEST_UTIL = new HBaseTestingUtil(conf);
082    TEST_UTIL.startMiniCluster(regionServerNum);
083    cluster = TEST_UTIL.getHBaseCluster();
084    LOG.info("Waiting for active/ready master");
085    cluster.waitForActiveAndReadyMaster();
086
087    // Create a table with regions
088    final TableName table = TableName.valueOf(name.getMethodName());
089    byte [] family = Bytes.toBytes("family");
090    LOG.info("Creating table with " + regionNum + " regions");
091    Table ht = TEST_UTIL.createMultiRegionTable(table, family, regionNum);
092    int numRegions = -1;
093    try (RegionLocator r = TEST_UTIL.getConnection().getRegionLocator(table)) {
094      numRegions = r.getStartKeys().length;
095    }
096    assertEquals(regionNum,numRegions);
097    waitForAllRegionsAssigned();
098
099    for (HRegionServer server : getOnlineRegionServers()) {
100      long globalMemStoreSize = 0;
101      for (RegionInfo regionInfo :
102          ProtobufUtil.getOnlineRegions(null, server.getRSRpcServices())) {
103        globalMemStoreSize += server.getRegion(regionInfo.getEncodedName()).getMemStoreDataSize();
104      }
105      assertEquals(server.getRegionServerAccounting().getGlobalMemStoreDataSize(),
106        globalMemStoreSize);
107    }
108
109    // check the global memstore size after flush
110    int i = 0;
111    for (HRegionServer server : getOnlineRegionServers()) {
112      LOG.info("Starting flushes on " + server.getServerName() +
113        ", size=" + server.getRegionServerAccounting().getGlobalMemStoreDataSize());
114
115      for (RegionInfo regionInfo :
116          ProtobufUtil.getOnlineRegions(null, server.getRSRpcServices())) {
117        HRegion r = server.getRegion(regionInfo.getEncodedName());
118        flush(r, server);
119      }
120      LOG.info("Post flush on " + server.getServerName());
121      long now = EnvironmentEdgeManager.currentTime();
122      long timeout = now + 1000;
123      while(server.getRegionServerAccounting().getGlobalMemStoreDataSize() != 0 &&
124          timeout < EnvironmentEdgeManager.currentTime()) {
125        Threads.sleep(10);
126      }
127      long size = server.getRegionServerAccounting().getGlobalMemStoreDataSize();
128      if (size > 0) {
129        // If size > 0, see if its because the meta region got edits while
130        // our test was running....
131        for (RegionInfo regionInfo :
132            ProtobufUtil.getOnlineRegions(null, server.getRSRpcServices())) {
133          HRegion r = server.getRegion(regionInfo.getEncodedName());
134          long l = r.getMemStoreDataSize();
135          if (l > 0) {
136            // Only meta could have edits at this stage.  Give it another flush
137            // clear them.
138            assertTrue(regionInfo.isMetaRegion());
139            LOG.info(r.toString() + " " + l + ", reflushing");
140            r.flush(true);
141          }
142        }
143      }
144      size = server.getRegionServerAccounting().getGlobalMemStoreDataSize();
145      assertEquals("Server=" + server.getServerName() + ", i=" + i++, 0, size);
146    }
147
148    ht.close();
149    TEST_UTIL.shutdownMiniCluster();
150  }
151
152  /**
153   * Flush and log stats on flush
154   * @param r
155   * @param server
156   * @throws IOException
157   */
158  private void flush(final HRegion r, final HRegionServer server)
159  throws IOException {
160    LOG.info("Flush " + r.toString() + " on " + server.getServerName() +
161      ", " +  r.flush(true) + ", size=" +
162      server.getRegionServerAccounting().getGlobalMemStoreDataSize());
163  }
164
165  private List<HRegionServer> getOnlineRegionServers() {
166    List<HRegionServer> list = new ArrayList<>();
167    for (JVMClusterUtil.RegionServerThread rst :
168          cluster.getRegionServerThreads()) {
169      if (rst.getRegionServer().isOnline()) {
170        list.add(rst.getRegionServer());
171      }
172    }
173    return list;
174  }
175
176  /**
177   * Wait until all the regions are assigned.
178   */
179  private void waitForAllRegionsAssigned() throws IOException {
180    while (true) {
181      int regionCount = HBaseTestingUtil.getAllOnlineRegions(cluster).size();
182      if (regionCount >= totalRegionNum) break;
183      LOG.debug("Waiting for there to be "+ totalRegionNum
184        +" regions, but there are " + regionCount + " right now.");
185      try {
186        Thread.sleep(100);
187      } catch (InterruptedException e) {}
188    }
189  }
190}
