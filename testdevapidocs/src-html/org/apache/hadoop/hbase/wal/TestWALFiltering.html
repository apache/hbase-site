051@Category({RegionServerTests.class, MediumTests.class})
052public class TestWALFiltering {
053  private static final int NUM_MASTERS = 1;
054  private static final int NUM_RS = 4;
055
056  private static final TableName TABLE_NAME =
057      TableName.valueOf("TestWALFiltering");
058  private static final byte[] CF1 = Bytes.toBytes("MyCF1");
059  private static final byte[] CF2 = Bytes.toBytes("MyCF2");
060  private static final byte[][] FAMILIES = { CF1, CF2 };
061
062  private HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
063
064  @Before
065  public void setUp() throws Exception {
066    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
067    fillTable();
068  }
069
070  @After
071  public void tearDown() throws Exception {
072    TEST_UTIL.shutdownMiniCluster();
073  }
074
075  private void fillTable() throws IOException, InterruptedException {
076    Table table = TEST_UTIL.createTable(TABLE_NAME, FAMILIES, 3,
077        Bytes.toBytes("row0"), Bytes.toBytes("row99"), NUM_RS);
078    Random rand = new Random(19387129L);
079    for (int iStoreFile = 0; iStoreFile < 4; ++iStoreFile) {
080      for (int iRow = 0; iRow < 100; ++iRow) {
081        final byte[] row = Bytes.toBytes("row" + iRow);
082        Put put = new Put(row);
083        Delete del = new Delete(row);
084        for (int iCol = 0; iCol < 10; ++iCol) {
085          final byte[] cf = rand.nextBoolean() ? CF1 : CF2;
086          final long ts = Math.abs(rand.nextInt());
087          final byte[] qual = Bytes.toBytes("col" + iCol);
088          if (rand.nextBoolean()) {
089            final byte[] value = Bytes.toBytes("value_for_row_" + iRow +
090                "_cf_" + Bytes.toStringBinary(cf) + "_col_" + iCol + "_ts_" +
091                ts + "_random_" + rand.nextLong());
092            put.addColumn(cf, qual, ts, value);
093          } else if (rand.nextDouble() < 0.8) {
094            del.addColumn(cf, qual, ts);
095          } else {
096            del.addColumn(cf, qual, ts);
097          }
098        }
099        table.put(put);
100        table.delete(del);
101      }
102    }
103    TEST_UTIL.waitUntilAllRegionsAssigned(TABLE_NAME);
104  }
105
106  @Test
107  public void testFlushedSequenceIdsSentToHMaster()
108  throws IOException, InterruptedException,
109  org.apache.hadoop.hbase.shaded.com.google.protobuf.ServiceException, ServiceException {
110    SortedMap<byte[], Long> allFlushedSequenceIds = new TreeMap<>(Bytes.BYTES_COMPARATOR);
111    for (int i = 0; i < NUM_RS; ++i) {
112      flushAllRegions(i);
113    }
114    Thread.sleep(10000);
115    HMaster master = TEST_UTIL.getMiniHBaseCluster().getMaster();
116    for (int i = 0; i < NUM_RS; ++i) {
117      for (byte[] regionName : getRegionsByServer(i)) {
118        if (allFlushedSequenceIds.containsKey(regionName)) {
119          GetLastFlushedSequenceIdRequest req =
120            RequestConverter.buildGetLastFlushedSequenceIdRequest(regionName);
121
122          assertEquals((long)allFlushedSequenceIds.get(regionName),
123            master.getMasterRpcServices().getLastFlushedSequenceId(
124              null, req).getLastFlushedSequenceId());
125        }
126      }
127    }
128  }
129
130  private List<byte[]> getRegionsByServer(int rsId) throws IOException {
131    List<byte[]> regionNames = Lists.newArrayList();
132    HRegionServer hrs = getRegionServer(rsId);
133    for (Region r : hrs.getRegions(TABLE_NAME)) {
134      regionNames.add(r.getRegionInfo().getRegionName());
135    }
136    return regionNames;
137  }
138
139  private HRegionServer getRegionServer(int rsId) {
140    return TEST_UTIL.getMiniHBaseCluster().getRegionServer(rsId);
141  }
142
143  private void flushAllRegions(int rsId)
144  throws ServiceException,
145  org.apache.hadoop.hbase.shaded.com.google.protobuf.ServiceException, IOException {
146    HRegionServer hrs = getRegionServer(rsId);
147    for (byte[] regionName : getRegionsByServer(rsId)) {
148      FlushRegionRequest request =
149        RequestConverter.buildFlushRegionRequest(regionName);
150      hrs.getRSRpcServices().flushRegion(null, request);
151    }
152  }
