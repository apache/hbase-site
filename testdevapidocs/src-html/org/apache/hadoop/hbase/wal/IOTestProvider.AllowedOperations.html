044/**
045 * A WAL Provider that returns a single thread safe WAL that optionally can skip parts of our
046 * normal interactions with HDFS.
047 *
048 * This implementation picks a directory in HDFS based on the same mechanisms as the 
049 * {@link FSHLogProvider}. Users can configure how much interaction
050 * we have with HDFS with the configuration property "hbase.wal.iotestprovider.operations".
051 * The value should be a comma separated list of allowed operations:
052 * <ul>
053 *   <li><em>append</em>   : edits will be written to the underlying filesystem
054 *   <li><em>sync</em>     : wal syncs will result in hflush calls
055 *   <li><em>fileroll</em> : roll requests will result in creating a new file on the underlying
056 *                           filesystem.
057 * </ul>
058 * Additionally, the special cases "all" and "none" are recognized.
059 * If ommited, the value defaults to "all."
060 * Behavior is undefined if "all" or "none" are paired with additional values. Behavior is also
061 * undefined if values not listed above are included.
062 *
063 * Only those operations listed will occur between the returned WAL and HDFS. All others
064 * will be no-ops.
065 *
066 * Note that in the case of allowing "append" operations but not allowing "fileroll", the returned
067 * WAL will just keep writing to the same file. This won't avoid all costs associated with file
068 * management over time, becaue the data set size may result in additional HDFS block allocations.
069 *
070 */
071@InterfaceAudience.Private
072public class IOTestProvider implements WALProvider {
073  private static final Log LOG = LogFactory.getLog(IOTestProvider.class);
074
075  private static final String ALLOWED_OPERATIONS = "hbase.wal.iotestprovider.operations";
076  private enum AllowedOperations {
077    all,
078    append,
079    sync,
080    fileroll,
081    none;
082  }
083
084  private FSHLog log = null;
085
086  /**
087   * @param factory factory that made us, identity used for FS layout. may not be null
088   * @param conf may not be null
089   * @param listeners may be null
090   * @param providerId differentiate between providers from one facotry, used for FS layout. may be
091   *                   null
092   */
093  @Override
094  public void init(final WALFactory factory, final Configuration conf,
095      final List<WALActionsListener> listeners, String providerId) throws IOException {
096    if (null != log) {
097      throw new IllegalStateException("WALProvider.init should only be called once.");
098    }
099    if (null == providerId) {
100      providerId = DEFAULT_PROVIDER_ID;
101    }
102    final String logPrefix = factory.factoryId + WAL_FILE_NAME_DELIMITER + providerId;
103    log = new IOTestWAL(FSUtils.getWALFileSystem(conf), FSUtils.getWALRootDir(conf),
104        AbstractFSWALProvider.getWALDirectoryName(factory.factoryId),
105        HConstants.HREGION_OLDLOGDIR_NAME, conf, listeners, true, logPrefix,
106        META_WAL_PROVIDER_ID.equals(providerId) ? META_WAL_PROVIDER_ID : null);
107  }
108
109  @Override
110  public List<WAL> getWALs() throws IOException {
111    List<WAL> wals = new ArrayList<>(1);
112    wals.add(log);
113    return wals;
114  }
115
116  @Override
117  public WAL getWAL(final byte[] identifier, byte[] namespace) throws IOException {
118   return log;
119  }
120
121  @Override
122  public void close() throws IOException {
123    log.close();
124  }
125
126  @Override
127  public void shutdown() throws IOException {
128    log.shutdown();
129  }
130
131  private static class IOTestWAL extends FSHLog {
132
133    private final boolean doFileRolls;
134
135    // Used to differntiate between roll calls before and after we finish construction.
136    private final boolean initialized;
137
138    /**
139     * Create an edit log at the given <code>dir</code> location.
140     *
141     * You should never have to load an existing log. If there is a log at
142     * startup, it should have already been processed and deleted by the time the
143     * WAL object is started up.
144     *
145     * @param fs filesystem handle
146     * @param rootDir path to where logs and oldlogs
147     * @param logDir dir where wals are stored
148     * @param archiveDir dir where wals are archived
149     * @param conf configuration to use
150     * @param listeners Listeners on WAL events. Listeners passed here will
151     * be registered before we do anything else; e.g. the
152     * Constructor {@link #rollWriter()}.
153     * @param failIfWALExists If true IOException will be thrown if files related to this wal
154     *        already exist.
155     * @param prefix should always be hostname and port in distributed env and
156     *        it will be URL encoded before being used.
157     *        If prefix is null, "wal" will be used
158     * @param suffix will be url encoded. null is treated as empty. non-empty must start with
159     *        {@link AbstractFSWALProvider#WAL_FILE_NAME_DELIMITER}
160     * @throws IOException
161     */
162    public IOTestWAL(final FileSystem fs, final Path rootDir, final String logDir,
163        final String archiveDir, final Configuration conf,
164        final List<WALActionsListener> listeners,
165        final boolean failIfWALExists, final String prefix, final String suffix)
166        throws IOException {
167      super(fs, rootDir, logDir, archiveDir, conf, listeners, failIfWALExists, prefix, suffix);
168      Collection<String> operations = conf.getStringCollection(ALLOWED_OPERATIONS);
169      doFileRolls = operations.isEmpty() || operations.contains(AllowedOperations.all.name()) ||
170          operations.contains(AllowedOperations.fileroll.name());
171      initialized = true;
172      LOG.info("Initialized with file rolling " + (doFileRolls ? "enabled" : "disabled"));
173    }
174
175    private Writer noRollsWriter;
176
177    // creatWriterInstance is where the new pipeline is set up for doing file rolls
178    // if we are skipping it, just keep returning the same writer.
179    @Override
180    protected Writer createWriterInstance(final Path path) throws IOException {
181      // we get called from the FSHLog constructor (!); always roll in this case since
182      // we don't know yet if we're supposed to generally roll and
183      // we need an initial file in the case of doing appends but no rolls.
184      if (!initialized || doFileRolls) {
185        LOG.info("creating new writer instance.");
186        final ProtobufLogWriter writer = new IOTestWriter();
187        writer.init(fs, path, conf, false);
188        if (!initialized) {
189          LOG.info("storing initial writer instance in case file rolling isn't allowed.");
190          noRollsWriter = writer;
191        }
192        return writer;
193      } else {
194        LOG.info("WAL rolling disabled, returning the first writer.");
195        // Initial assignment happens during the constructor call, so there ought not be
196        // a race for first assignment.
197        return noRollsWriter;
198      }
199    }
200  }
201
202  /**
203   * Presumes init will be called by a single thread prior to any access of other methods.
204   */
205  private static class IOTestWriter extends ProtobufLogWriter {
206    private boolean doAppends;
207    private boolean doSyncs;
208
209    @Override
210    public void init(FileSystem fs, Path path, Configuration conf, boolean overwritable) throws IOException {
211      Collection<String> operations = conf.getStringCollection(ALLOWED_OPERATIONS);
212      if (operations.isEmpty() || operations.contains(AllowedOperations.all.name())) {
213        doAppends = doSyncs = true;
214      } else if (operations.contains(AllowedOperations.none.name())) {
215        doAppends = doSyncs = false;
216      } else {
217        doAppends = operations.contains(AllowedOperations.append.name());
218        doSyncs = operations.contains(AllowedOperations.sync.name());
219      }
220      LOG.info("IOTestWriter initialized with appends " + (doAppends ? "enabled" : "disabled") +
221          " and syncs " + (doSyncs ? "enabled" : "disabled"));
222      super.init(fs, path, conf, overwritable);
223    }
224
225    @Override
226    protected String getWriterClassName() {
227      return ProtobufLogWriter.class.getSimpleName();
228    }
229
230    @Override
231    public void append(Entry entry) throws IOException {
232      if (doAppends) {
233        super.append(entry);
234      }
235    }
236
237    @Override
238    public void sync() throws IOException {
239      if (doSyncs) {
240        super.sync();
241      }
242    }
243  }
244
245  @Override
246  public long getNumLogFiles() {
247    return this.log.getNumLogFiles();
248  }
249
250  @Override
251  public long getLogFileSize() {
252    return this.log.getLogFileSize();
253  }
