175      infos[i] = RegionInfoBuilder.newBuilder(tableName).setStartKey(Bytes.toBytes("" + i))

179    }

181    scopes.put(Bytes.toBytes("column"), 0);
182
183
184    // Add edits for three regions.
185    for (int ii = 0; ii < howmany; ii++) {
186      for (int i = 0; i < howmany; i++) {
187        final WAL log =
188            wals.getWAL(infos[i]);
189        for (int j = 0; j < howmany; j++) {
190          WALEdit edit = new WALEdit();
191          byte [] family = Bytes.toBytes("column");
192          byte [] qualifier = Bytes.toBytes(Integer.toString(j));
193          byte [] column = Bytes.toBytes("column:" + Integer.toString(j));
194          edit.add(new KeyValue(rowName, family, qualifier,
195              System.currentTimeMillis(), column));
196          LOG.info("Region " + i + ": " + edit);
197          WALKeyImpl walKey =  new WALKeyImpl(infos[i].getEncodedNameAsBytes(), tableName,
198              System.currentTimeMillis(), mvcc, scopes);
199          log.append(infos[i], walKey, edit, true);
200          walKey.getWriteEntry();
201        }
202        log.sync();
203        log.rollWriter(true);
204      }
205    }
206    wals.shutdown();
207    // The below calculation of logDir relies on insider information... WALSplitter should be connected better

213    List<Path> splits = WALSplitter.split(hbaseWALDir, logDir, oldLogDir, fs, conf, wals);
214    verifySplits(splits, howmany);

222  public void Broken_testSync() throws Exception {
223    TableName tableName = TableName.valueOf(currentTest.getMethodName());
224    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl(1);
225    // First verify that using streams all works.
226    Path p = new Path(dir, currentTest.getMethodName() + ".fsdos");
227    FSDataOutputStream out = fs.create(p);
228    out.write(tableName.getName());
229    Method syncMethod = null;
230    try {
231      syncMethod = out.getClass().getMethod("hflush", new Class<?> []{});
232    } catch (NoSuchMethodException e) {
233      try {
234        syncMethod = out.getClass().getMethod("sync", new Class<?> []{});
235      } catch (NoSuchMethodException ex) {
236        fail("This version of Hadoop supports neither Syncable.sync() " +
237            "nor Syncable.hflush().");
238      }
239    }
240    syncMethod.invoke(out, new Object[]{});
241    FSDataInputStream in = fs.open(p);
242    assertTrue(in.available() > 0);
243    byte [] buffer = new byte [1024];
244    int read = in.read(buffer);
245    assertEquals(tableName.getName().length, read);
246    out.close();
247    in.close();
248
249    final int total = 20;
250    WAL.Reader reader = null;
251
252    try {
253      RegionInfo info = RegionInfoBuilder.newBuilder(tableName).build();
254      NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
255      scopes.put(tableName.getName(), 0);
256      final WAL wal = wals.getWAL(info);
257
258      for (int i = 0; i < total; i++) {
259        WALEdit kvs = new WALEdit();
260        kvs.add(new KeyValue(Bytes.toBytes(i), tableName.getName(), tableName.getName()));
261        wal.append(info, new WALKeyImpl(info.getEncodedNameAsBytes(), tableName,
262            System.currentTimeMillis(), mvcc, scopes), kvs, true);
263      }
264      // Now call sync and try reading.  Opening a Reader before you sync just
265      // gives you EOFE.
266      wal.sync();
267      // Open a Reader.
268      Path walPath = AbstractFSWALProvider.getCurrentFileName(wal);
269      reader = wals.createReader(fs, walPath);
270      int count = 0;
271      WAL.Entry entry = new WAL.Entry();
272      while ((entry = reader.next(entry)) != null) count++;
273      assertEquals(total, count);
274      reader.close();
275      // Add test that checks to see that an open of a Reader works on a file
276      // that has had a sync done on it.
277      for (int i = 0; i < total; i++) {
278        WALEdit kvs = new WALEdit();
279        kvs.add(new KeyValue(Bytes.toBytes(i), tableName.getName(), tableName.getName()));
280        wal.append(info, new WALKeyImpl(info.getEncodedNameAsBytes(), tableName,
281            System.currentTimeMillis(), mvcc, scopes), kvs, true);
282      }
283      wal.sync();
284      reader = wals.createReader(fs, walPath);
285      count = 0;
286      while((entry = reader.next(entry)) != null) count++;
287      assertTrue(count >= total);
288      reader.close();
289      // If I sync, should see double the edits.
290      wal.sync();
291      reader = wals.createReader(fs, walPath);
292      count = 0;
293      while((entry = reader.next(entry)) != null) count++;
294      assertEquals(total * 2, count);
295      reader.close();
296      // Now do a test that ensures stuff works when we go over block boundary,
297      // especially that we return good length on file.
298      final byte [] value = new byte[1025 * 1024];  // Make a 1M value.
299      for (int i = 0; i < total; i++) {
300        WALEdit kvs = new WALEdit();
301        kvs.add(new KeyValue(Bytes.toBytes(i), tableName.getName(), value));
302        wal.append(info, new WALKeyImpl(info.getEncodedNameAsBytes(), tableName,
303            System.currentTimeMillis(), mvcc, scopes), kvs,  true);
304      }
305      // Now I should have written out lots of blocks.  Sync then read.
306      wal.sync();
307      reader = wals.createReader(fs, walPath);
308      count = 0;
309      while((entry = reader.next(entry)) != null) count++;
310      assertEquals(total * 3, count);
311      reader.close();
312      // shutdown and ensure that Reader gets right length also.
313      wal.shutdown();
314      reader = wals.createReader(fs, walPath);
315      count = 0;
316      while((entry = reader.next(entry)) != null) count++;
317      assertEquals(total * 3, count);
318      reader.close();
319    } finally {
320      if (reader != null) reader.close();
321    }
322  }

324  private void verifySplits(final List<Path> splits, final int howmany)
325  throws IOException {
326    assertEquals(howmany * howmany, splits.size());
327    for (int i = 0; i < splits.size(); i++) {
328      LOG.info("Verifying=" + splits.get(i));
329      WAL.Reader reader = wals.createReader(fs, splits.get(i));
330      try {
331        int count = 0;
332        String previousRegion = null;
333        long seqno = -1;
334        WAL.Entry entry = new WAL.Entry();
335        while((entry = reader.next(entry)) != null) {
336          WALKey key = entry.getKey();
337          String region = Bytes.toString(key.getEncodedRegionName());
338          // Assert that all edits are for same region.
339          if (previousRegion != null) {
340            assertEquals(previousRegion, region);
341          }
342          LOG.info("oldseqno=" + seqno + ", newseqno=" + key.getSequenceId());
343          assertTrue(seqno < key.getSequenceId());
344          seqno = key.getSequenceId();
345          previousRegion = region;
346          count++;
347        }
348        assertEquals(howmany, count);
349      } finally {
350        reader.close();
351      }
352    }
353  }

356   * We pass different values to recoverFileLease() so that different code paths are covered
357   *
358   * For this test to pass, requires:
359   * 1. HDFS-200 (append support)
360   * 2. HDFS-988 (SafeMode should freeze file operations
361   *              [FSNamesystem.nextGenerationStampForBlock])
362   * 3. HDFS-142 (on restart, maintain pendingCreates)
363   */
364  @Test (timeout=300000)
365  public void testAppendClose() throws Exception {
366    TableName tableName =
367        TableName.valueOf(currentTest.getMethodName());
368    RegionInfo regionInfo = RegionInfoBuilder.newBuilder(tableName).build();
369
370    WAL wal = wals.getWAL(regionInfo);
371    int total = 20;
372
373    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
374    scopes.put(tableName.getName(), 0);
375    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
376    for (int i = 0; i < total; i++) {
377      WALEdit kvs = new WALEdit();
378      kvs.add(new KeyValue(Bytes.toBytes(i), tableName.getName(), tableName.getName()));
379      wal.append(regionInfo, new WALKeyImpl(regionInfo.getEncodedNameAsBytes(), tableName,
380          System.currentTimeMillis(), mvcc, scopes),
381        kvs, true);
382    }
383    // Now call sync to send the data to HDFS datanodes
384    wal.sync();
385     int namenodePort = cluster.getNameNodePort();
386    final Path walPath = AbstractFSWALProvider.getCurrentFileName(wal);
387
388
389    // Stop the cluster.  (ensure restart since we're sharing MiniDFSCluster)
390    try {
391      DistributedFileSystem dfs = cluster.getFileSystem();
392      dfs.setSafeMode(HdfsConstants.SafeModeAction.SAFEMODE_ENTER);
393      TEST_UTIL.shutdownMiniDFSCluster();
394      try {
395        // wal.writer.close() will throw an exception,
396        // but still call this since it closes the LogSyncer thread first
397        wal.shutdown();
398      } catch (IOException e) {
399        LOG.info(e.toString(), e);
400      }
401      fs.close(); // closing FS last so DFSOutputStream can't call close
402      LOG.info("STOPPED first instance of the cluster");
403    } finally {
404      // Restart the cluster
405      while (cluster.isClusterUp()){
406        LOG.error("Waiting for cluster to go down");
407        Thread.sleep(1000);
408      }
409      assertFalse(cluster.isClusterUp());
410      cluster = null;
411      for (int i = 0; i < 100; i++) {
412        try {
413          cluster = TEST_UTIL.startMiniDFSClusterForTestWAL(namenodePort);
414          break;
415        } catch (BindException e) {
416          LOG.info("Sleeping.  BindException bringing up new cluster");
417          Threads.sleep(1000);
418        }
419      }
420      cluster.waitActive();
421      fs = cluster.getFileSystem();
422      LOG.info("STARTED second instance.");
423    }
424
425    // set the lease period to be 1 second so that the
426    // namenode triggers lease recovery upon append request
427    Method setLeasePeriod = cluster.getClass()
428      .getDeclaredMethod("setLeasePeriod", new Class[]{Long.TYPE, Long.TYPE});
429    setLeasePeriod.setAccessible(true);
430    setLeasePeriod.invoke(cluster, 1000L, 1000L);
431    try {
432      Thread.sleep(1000);
433    } catch (InterruptedException e) {
434      LOG.info(e.toString(), e);
435    }
436
437    // Now try recovering the log, like the HMaster would do
438    final FileSystem recoveredFs = fs;
439    final Configuration rlConf = conf;
440
441    class RecoverLogThread extends Thread {
442      public Exception exception = null;
443      @Override
444      public void run() {
445          try {
446            FSUtils.getInstance(fs, rlConf)
447              .recoverFileLease(recoveredFs, walPath, rlConf, null);
448          } catch (IOException e) {
449            exception = e;
450          }
451      }
452    }
453
454    RecoverLogThread t = new RecoverLogThread();
455    t.start();
456    // Timeout after 60 sec. Without correct patches, would be an infinite loop
457    t.join(60 * 1000);
458    if(t.isAlive()) {
459      t.interrupt();
460      throw new Exception("Timed out waiting for WAL.recoverLog()");
461    }
462
463    if (t.exception != null)
464      throw t.exception;
465
466    // Make sure you can read all the content
467    WAL.Reader reader = wals.createReader(fs, walPath);
468    int count = 0;
469    WAL.Entry entry = new WAL.Entry();
470    while (reader.next(entry) != null) {
471      count++;
472      assertTrue("Should be one KeyValue per WALEdit",
473                  entry.getEdit().getCells().size() == 1);
474    }
475    assertEquals(total, count);
476    reader.close();
477
478    // Reset the lease period
479    setLeasePeriod.invoke(cluster, new Object[]{new Long(60000), new Long(3600000)});
480  }

587          entry.getKey().getEncodedRegionName()));
588        assertTrue(htd.getTableName().equals(entry.getKey().getTablename()));
589        assertTrue(Bytes.equals(row, 0, row.length, val.getRowArray(), val.getRowOffset(),
590          val.getRowLength()));
591        assertEquals((byte) (idx + '0'), CellUtil.cloneValue(val)[0]);
592        System.out.println(entry.getKey() + " " + val);
593        idx++;
594      }
595    } finally {
596      if (reader != null) {
597        reader.close();
598      }
599    }

602  /**
603   * Test that we can visit entries before they are appended
604   * @throws Exception
605   */
606  @Test
607  public void testVisitors() throws Exception {
608    final int COL_COUNT = 10;
609    final TableName tableName = TableName.valueOf(currentTest.getMethodName());
610    final byte [] row = Bytes.toBytes("row");
611    final DumbWALActionsListener visitor = new DumbWALActionsListener();
612    final MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl(1);
613    long timestamp = System.currentTimeMillis();
614    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
615    scopes.put(Bytes.toBytes("column"), 0);
616
617    RegionInfo hri = RegionInfoBuilder.newBuilder(tableName).build();
618    final WAL log = wals.getWAL(hri);
619    log.registerWALActionsListener(visitor);
620    for (int i = 0; i < COL_COUNT; i++) {
621      WALEdit cols = new WALEdit();
622      cols.add(new KeyValue(row, Bytes.toBytes("column"),
623          Bytes.toBytes(Integer.toString(i)),
624          timestamp, new byte[]{(byte) (i + '0')}));
625      log.append(hri, new WALKeyImpl(hri.getEncodedNameAsBytes(), tableName,
626          System.currentTimeMillis(), mvcc, scopes), cols, true);
627    }
628    log.sync();
629    assertEquals(COL_COUNT, visitor.increments);
630    log.unregisterWALActionsListener(visitor);
631    WALEdit cols = new WALEdit();
632    cols.add(new KeyValue(row, Bytes.toBytes("column"),
633        Bytes.toBytes(Integer.toString(11)),
634        timestamp, new byte[]{(byte) (11 + '0')}));
635    log.append(hri, new WALKeyImpl(hri.getEncodedNameAsBytes(), tableName,
636        System.currentTimeMillis(), mvcc, scopes), cols, true);
637    log.sync();
638    assertEquals(COL_COUNT, visitor.increments);
639  }
