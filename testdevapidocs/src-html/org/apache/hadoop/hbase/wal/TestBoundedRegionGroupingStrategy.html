168  /**
169   * Ensure that we can use Set.add to deduplicate WALs
170   */
171  @Test
172  public void setMembershipDedups() throws IOException {
173    final int temp = CONF.getInt(NUM_REGION_GROUPS, DEFAULT_NUM_REGION_GROUPS);
174    WALFactory wals = null;
175    try {
176      CONF.setInt(NUM_REGION_GROUPS, temp * 4);
177      // Set HDFS root directory for storing WAL
178      FSUtils.setRootDir(CONF, TEST_UTIL.getDataTestDirOnTestFS());
179
180      wals = new WALFactory(CONF, null, "setMembershipDedups");
181      final Set<WAL> seen = new HashSet<>(temp * 4);
182      final Random random = new Random();
183      int count = 0;
184      // we know that this should see one of the wals more than once
185      for (int i = 0; i < temp * 8; i++) {
186        final WAL maybeNewWAL = wals.getWAL(Bytes.toBytes(random.nextInt()), null);
187        LOG.info("Iteration " + i + ", checking wal " + maybeNewWAL);
188        if (seen.add(maybeNewWAL)) {
189          count++;
190        }
191      }
192      assertEquals("received back a different number of WALs that are not equal() to each other "
193          + "than the bound we placed.",
194        temp * 4, count);
195    } finally {
196      if (wals != null) {
197        wals.close();
198      }
199      CONF.setInt(NUM_REGION_GROUPS, temp);
200    }
201  }
