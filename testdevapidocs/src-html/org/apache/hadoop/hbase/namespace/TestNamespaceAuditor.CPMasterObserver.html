334    String nsp1 = prefix + "_regiontest";
335    final int initialRegions = 3;
336    NamespaceDescriptor nspDesc =
337        NamespaceDescriptor.create(nsp1)
338            .addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "" + initialRegions)
339            .addConfiguration(TableNamespaceManager.KEY_MAX_TABLES, "2").build();

342    byte[] columnFamily = Bytes.toBytes("info");
343    HTableDescriptor tableDescOne = new HTableDescriptor(tableTwo);
344    tableDescOne.addFamily(new HColumnDescriptor(columnFamily));
345    ADMIN.createTable(tableDescOne, Bytes.toBytes("0"), Bytes.toBytes("9"), initialRegions);
346    Connection connection = ConnectionFactory.createConnection(UTIL.getConfiguration());
347    try (Table table = connection.getTable(tableTwo)) {
348      UTIL.loadNumericRows(table, Bytes.toBytes("info"), 1000, 1999);
349    }
350    ADMIN.flush(tableTwo);
351    List<HRegionInfo> hris = ADMIN.getTableRegions(tableTwo);
352    assertEquals(initialRegions, hris.size());
353    Collections.sort(hris);
354    Future<?> f = ADMIN.mergeRegionsAsync(
355      hris.get(0).getEncodedNameAsBytes(),
356      hris.get(1).getEncodedNameAsBytes(),
357      false);
358    f.get(10, TimeUnit.SECONDS);
359
360    hris = ADMIN.getTableRegions(tableTwo);
361    assertEquals(initialRegions - 1, hris.size());
362    Collections.sort(hris);
363    ADMIN.split(tableTwo, Bytes.toBytes("3"));
364    // Not much we can do here until we have split return a Future.
365    Threads.sleep(5000);
366    hris = ADMIN.getTableRegions(tableTwo);
367    assertEquals(initialRegions, hris.size());
368    Collections.sort(hris);
369
370    // Fail region merge through Coprocessor hook
371    MiniHBaseCluster cluster = UTIL.getHBaseCluster();
372    MasterCoprocessorHost cpHost = cluster.getMaster().getMasterCoprocessorHost();
373    Coprocessor coprocessor = cpHost.findCoprocessor(CPMasterObserver.class);
374    CPMasterObserver masterObserver = (CPMasterObserver) coprocessor;
375    masterObserver.failMerge(true);
376
377    f = ADMIN.mergeRegionsAsync(
378      hris.get(1).getEncodedNameAsBytes(),
379      hris.get(2).getEncodedNameAsBytes(),
380      false);
381    try {
382      f.get(10, TimeUnit.SECONDS);
383      fail("Merge was supposed to fail!");
384    } catch (ExecutionException ee) {
385      // Expected.
386    }
387    hris = ADMIN.getTableRegions(tableTwo);
388    assertEquals(initialRegions, hris.size());
389    Collections.sort(hris);
390    // verify that we cannot split
391    HRegionInfo hriToSplit2 = hris.get(1);
392    try {
393      ADMIN.split(tableTwo, Bytes.toBytes("6"));
394      fail();
395    } catch (DoNotRetryRegionException e) {
396      // Expected
397    }
398    Thread.sleep(2000);
399    assertEquals(initialRegions, ADMIN.getTableRegions(tableTwo).size());

404   * namespace quota cache. Now correct the failure and recreate the table with same name.

407  @Test

409    String nsp1 = prefix + "_testRecreateTable";
410    NamespaceDescriptor nspDesc =
411        NamespaceDescriptor.create(nsp1)
412            .addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "20")
413            .addConfiguration(TableNamespaceManager.KEY_MAX_TABLES, "1").build();
414    ADMIN.createNamespace(nspDesc);
415    final TableName tableOne = TableName.valueOf(nsp1 + TableName.NAMESPACE_DELIM + "table1");
416    byte[] columnFamily = Bytes.toBytes("info");
417    HTableDescriptor tableDescOne = new HTableDescriptor(tableOne);
418    tableDescOne.addFamily(new HColumnDescriptor(columnFamily));
419    MasterSyncObserver.throwExceptionInPreCreateTableAction = true;
420    try {
421      try {
422        ADMIN.createTable(tableDescOne);
423        fail("Table " + tableOne.toString() + "creation should fail.");
424      } catch (Exception exp) {
425        LOG.error(exp.toString(), exp);
426      }
427      assertFalse(ADMIN.tableExists(tableOne));
428
429      NamespaceTableAndRegionInfo nstate = getNamespaceState(nsp1);
430      assertEquals("First table creation failed in namespace so number of tables in namespace "
431          + "should be 0.", 0, nstate.getTables().size());
432
433      MasterSyncObserver.throwExceptionInPreCreateTableAction = false;
434      try {
435        ADMIN.createTable(tableDescOne);
436      } catch (Exception e) {
437        fail("Table " + tableOne.toString() + "creation should succeed.");
438        LOG.error(e.toString(), e);
439      }
440      assertTrue(ADMIN.tableExists(tableOne));
441      nstate = getNamespaceState(nsp1);
442      assertEquals("First table was created successfully so table size in namespace should "
443          + "be one now.", 1, nstate.getTables().size());
444    } finally {
445      MasterSyncObserver.throwExceptionInPreCreateTableAction = false;
446      if (ADMIN.tableExists(tableOne)) {
447        ADMIN.disableTable(tableOne);
448        deleteTable(tableOne);
449      }
450      ADMIN.deleteNamespace(nsp1);
451    }
452  }

527  public static void waitForQuotaInitialize(final HBaseTestingUtility util) throws Exception {
528    util.waitFor(60000, new Waiter.Predicate<Exception>() {
529      @Override
530      public boolean evaluate() throws Exception {
531        HMaster master = util.getHBaseCluster().getMaster();
532        if (master == null) {
533          return false;
534        }
535        MasterQuotaManager quotaManager = master.getMasterQuotaManager();
536        return quotaManager != null && quotaManager.isQuotaInitialized();
537      }
538    });

584  private void deleteTable(final TableName tableName) throws Exception {
585    // NOTE: We need a latch because admin is not sync,
586    // so the postOp coprocessor method may be called after the admin operation returned.
587    MasterSyncObserver observer = UTIL.getHBaseCluster().getMaster()
588      .getMasterCoprocessorHost().findCoprocessor(MasterSyncObserver.class);
589    ADMIN.deleteTable(tableName);
590    observer.tableDeletionLatch.await();
591  }

673    String nsp = prefix + "_testRestoreSnapshot";
674    NamespaceDescriptor nspDesc =
675        NamespaceDescriptor.create(nsp)
676            .addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "10").build();
677    ADMIN.createNamespace(nspDesc);
678    assertNotNull("Namespace descriptor found null.", ADMIN.getNamespaceDescriptor(nsp));
679    TableName tableName1 = TableName.valueOf(nsp + TableName.NAMESPACE_DELIM + "table1");
680    HTableDescriptor tableDescOne = new HTableDescriptor(tableName1);
681    HColumnDescriptor fam1 = new HColumnDescriptor("fam1");
682    tableDescOne.addFamily(fam1);
683    ADMIN.createTable(tableDescOne, Bytes.toBytes("AAA"), Bytes.toBytes("ZZZ"), 4);
684
685    NamespaceTableAndRegionInfo nstate = getNamespaceState(nsp);
686    assertEquals("Intial region count should be 4.", 4, nstate.getRegionCount());
687
688    String snapshot = "snapshot_testRestoreSnapshot";

690
691    List<HRegionInfo> regions = ADMIN.getTableRegions(tableName1);

708  public void testRestoreSnapshotQuotaExceed() throws Exception {
709    String nsp = prefix + "_testRestoreSnapshotQuotaExceed";
710    NamespaceDescriptor nspDesc =
711        NamespaceDescriptor.create(nsp)
712            .addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "10").build();
713    ADMIN.createNamespace(nspDesc);
714    NamespaceDescriptor ndesc = ADMIN.getNamespaceDescriptor(nsp);
715    assertNotNull("Namespace descriptor found null.", ndesc);
716    TableName tableName1 = TableName.valueOf(nsp + TableName.NAMESPACE_DELIM + "table1");
717    HTableDescriptor tableDescOne = new HTableDescriptor(tableName1);
718    HColumnDescriptor fam1 = new HColumnDescriptor("fam1");
719    tableDescOne.addFamily(fam1);
720
721    ADMIN.createTable(tableDescOne, Bytes.toBytes("AAA"), Bytes.toBytes("ZZZ"), 4);
722
723    NamespaceTableAndRegionInfo nstate = getNamespaceState(nsp);
724    assertEquals("Intial region count should be 4.", 4, nstate.getRegionCount());
725
726    String snapshot = "snapshot_testRestoreSnapshotQuotaExceed";
727    // snapshot has 4 regions
728    ADMIN.snapshot(snapshot, tableName1);
729    // recreate table with 1 region and set max regions to 3 for namespace
730    ADMIN.disableTable(tableName1);
731    ADMIN.deleteTable(tableName1);
732    ADMIN.createTable(tableDescOne);
733    ndesc.setConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "3");
734    ADMIN.modifyNamespace(ndesc);
735
736    ADMIN.disableTable(tableName1);
737    try {
738      ADMIN.restoreSnapshot(snapshot);
739      fail("Region quota is exceeded so QuotaExceededException should be thrown but HBaseAdmin"
740          + " wraps IOException into RestoreSnapshotException");
741    } catch (RestoreSnapshotException ignore) {
742      assertTrue(ignore.getCause() instanceof QuotaExceededException);
743    }
744    assertEquals(1, getNamespaceState(nsp).getRegionCount());
745    ADMIN.enableTable(tableName1);
746    ADMIN.deleteSnapshot(snapshot);
747  }
