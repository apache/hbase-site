018package org.apache.hadoop.hbase.constraint;
019
020import org.apache.hadoop.conf.Configuration;
021import org.apache.hadoop.hbase.client.Put;
022
023/**
024 * Test Constraint to check to make sure the configuration is set
025 */
026public class CheckConfigurationConstraint extends BaseConstraint {
027
028  private static String key = "testKey";
029  private static String value = "testValue";
030
031  public static Configuration getConfiguration() {
032    Configuration conf = new Configuration(false);
033    conf.set(key, value);
034    return conf;
035  }
036
037  @Override
038  public void check(Put p) {
039    // NOOP
040  }
041
042  @Override
043  public void setConf(Configuration conf) {
044    super.setConf(conf);
045    if (conf != null) {
046      String val = conf.get(key);
047      if (val == null || !val.equals(value))
048        throw new IllegalArgumentException(
049            "Configuration was not passed correctly");
050      // and then check to make sure we got a fresh config by checking for a
051      // hadoop-based config value, and if we don't find it, its fine
052      if (conf.getRaw("fs.file.impl") != null)
053        throw new IllegalArgumentException(
054            "Configuration was created using 'new Configuration()', should be "
055                + "done via 'new Configuration(false) to exclude defaut hadoop "
056                + "configurations values.");
057    }
058  }
059
060}
