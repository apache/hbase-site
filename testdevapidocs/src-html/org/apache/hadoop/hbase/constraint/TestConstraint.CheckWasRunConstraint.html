050@Category({MiscTests.class, MediumTests.class})

053  @ClassRule
054  public static final HBaseClassTestRule CLASS_RULE =
055      HBaseClassTestRule.forClass(TestConstraint.class);

057  private static final Logger LOG = LoggerFactory
058      .getLogger(TestConstraint.class);

060  private static HBaseTestingUtility util;
061  private static final TableName tableName = TableName.valueOf("test");
062  private static final byte[] dummy = Bytes.toBytes("dummy");
063  private static final byte[] row1 = Bytes.toBytes("r1");
064  private static final byte[] test = Bytes.toBytes("test");
065
066  @BeforeClass
067  public static void setUpBeforeClass() throws Exception {
068    util = new HBaseTestingUtility();
069    util.getConfiguration().setBoolean(CoprocessorHost.ABORT_ON_ERROR_KEY, false);
070    util.startMiniCluster();
071  }

073  /**
074   * Test that we run a passing constraint
075   * @throws Exception
076   */
077  @SuppressWarnings("unchecked")
078  @Test
079  public void testConstraintPasses() throws Exception {
080    // create the table
081    // it would be nice if this was also a method on the util
082    HTableDescriptor desc = new HTableDescriptor(tableName);
083    for (byte[] family : new byte[][] { dummy, test }) {
084      desc.addFamily(new HColumnDescriptor(family));
085    }
086    // add a constraint
087    Constraints.add(desc, CheckWasRunConstraint.class);
088
089    util.getAdmin().createTable(desc);
090    Table table = util.getConnection().getTable(tableName);
091    try {
092      // test that we don't fail on a valid put
093      Put put = new Put(row1);
094      byte[] value = Bytes.toBytes(Integer.toString(10));
095      byte[] qualifier = new byte[0];
096      put.addColumn(dummy, qualifier, value);
097      table.put(put);
098    } finally {
099      table.close();
100    }
101    assertTrue(CheckWasRunConstraint.wasRun);
102  }
103
104  /**
105   * Test that constraints will fail properly
106   * @throws Exception
107   */
108  @SuppressWarnings("unchecked")
109  @Test
110  public void testConstraintFails() throws Exception {
111
112    // create the table
113    // it would be nice if this was also a method on the util
114    HTableDescriptor desc = new HTableDescriptor(tableName);
115    for (byte[] family : new byte[][] { dummy, test }) {
116      desc.addFamily(new HColumnDescriptor(family));
117    }
118
119    // add a constraint that is sure to fail
120    Constraints.add(desc, AllFailConstraint.class);
121
122    util.getAdmin().createTable(desc);
123    Table table = util.getConnection().getTable(tableName);
124
125    // test that we do fail on violation
126    Put put = new Put(row1);
127    byte[] qualifier = new byte[0];
128    put.addColumn(dummy, qualifier, Bytes.toBytes("fail"));
129    LOG.warn("Doing put in table");
130    try {
131      table.put(put);
132      fail("This put should not have suceeded - AllFailConstraint was not run!");
133    } catch (ConstraintException e) {
134      // expected
135    }
136    table.close();
137  }

139  /**
140   * Check that if we just disable one constraint, then
141   * @throws Throwable
142   */
143  @SuppressWarnings("unchecked")
144  @Test
145  public void testDisableConstraint() throws Throwable {
146    // create the table
147    HTableDescriptor desc = new HTableDescriptor(tableName);
148    // add a family to the table
149    for (byte[] family : new byte[][] { dummy, test }) {
150      desc.addFamily(new HColumnDescriptor(family));
151    }
152    // add a constraint to make sure it others get run
153    Constraints.add(desc, CheckWasRunConstraint.class);
154
155    // Add Constraint to check
156    Constraints.add(desc, AllFailConstraint.class);
157
158    // and then disable the failing constraint
159    Constraints.disableConstraint(desc, AllFailConstraint.class);
160
161    util.getAdmin().createTable(desc);
162    Table table = util.getConnection().getTable(tableName);
163    try {
164      // test that we don't fail because its disabled
165      Put put = new Put(row1);
166      byte[] qualifier = new byte[0];
167      put.addColumn(dummy, qualifier, Bytes.toBytes("pass"));
168      table.put(put);
169    } finally {
170      table.close();
171    }
172    assertTrue(CheckWasRunConstraint.wasRun);
173  }

175  /**
176   * Test that if we disable all constraints, then nothing gets run
177   * @throws Throwable
178   */
179  @SuppressWarnings("unchecked")
180  @Test
181  public void testDisableConstraints() throws Throwable {
182    // create the table
183    HTableDescriptor desc = new HTableDescriptor(tableName);
184    // add a family to the table
185    for (byte[] family : new byte[][] { dummy, test }) {
186      desc.addFamily(new HColumnDescriptor(family));
187    }
188    // add a constraint to check to see if is run
189    Constraints.add(desc, CheckWasRunConstraint.class);
190
191    // then disable all the constraints
192    Constraints.disable(desc);
193
194    util.getAdmin().createTable(desc);
195    Table table = util.getConnection().getTable(tableName);
196    try {
197      // test that we do fail on violation
198      Put put = new Put(row1);
199      byte[] qualifier = new byte[0];
200      put.addColumn(dummy, qualifier, Bytes.toBytes("pass"));
201      LOG.warn("Doing put in table");
202      table.put(put);
203    } finally {
204      table.close();
205    }
206    assertFalse(CheckWasRunConstraint.wasRun);
207  }

209  /**
210   * Check to make sure a constraint is unloaded when it fails
211   * @throws Exception
212   */
213  @Test
214  public void testIsUnloaded() throws Exception {
215    // create the table
216    HTableDescriptor desc = new HTableDescriptor(tableName);
217    // add a family to the table
218    for (byte[] family : new byte[][] { dummy, test }) {
219      desc.addFamily(new HColumnDescriptor(family));
220    }
221    // make sure that constraints are unloaded
222    Constraints.add(desc, RuntimeFailConstraint.class);
223    // add a constraint to check to see if is run
224    Constraints.add(desc, CheckWasRunConstraint.class);
225    CheckWasRunConstraint.wasRun = false;
226
227    util.getAdmin().createTable(desc);
228    Table table = util.getConnection().getTable(tableName);
229
230    // test that we do fail on violation
231    Put put = new Put(row1);
232    byte[] qualifier = new byte[0];
233    put.addColumn(dummy, qualifier, Bytes.toBytes("pass"));
234
235    try{
236    table.put(put);
237    fail("RuntimeFailConstraint wasn't triggered - this put shouldn't work!");
238    } catch (Exception e) {// NOOP
239    }
240
241    // try the put again, this time constraints are not used, so it works
242    table.put(put);
243    // and we make sure that constraints were not run...
244    assertFalse(CheckWasRunConstraint.wasRun);
245    table.close();
246  }
