054/**
055 * Test MetaTableAccessor but without spinning up a cluster.
056 * We mock regionserver back and forth (we do spin up a zk cluster).
057 */
058@Category({MiscTests.class, MediumTests.class})
059public class TestMetaTableAccessorNoCluster {
060  private static final Log LOG = LogFactory.getLog(TestMetaTableAccessorNoCluster.class);
061  private static final  HBaseTestingUtility UTIL = new HBaseTestingUtility();
062  private static final Abortable ABORTABLE = new Abortable() {
063    boolean aborted = false;
064    @Override
065    public void abort(String why, Throwable e) {
066      LOG.info(why, e);
067      this.aborted = true;
068      throw new RuntimeException(e);
069    }
070    @Override
071    public boolean isAborted()  {
072      return this.aborted;
073    }
074  };
075
076  @Before
077  public void before() throws Exception {
078    UTIL.startMiniZKCluster();
079  }
080
081  @After
082  public void after() throws IOException {
083    UTIL.shutdownMiniZKCluster();
084  }

086  @Test
087  public void testGetHRegionInfo() throws IOException {
088    assertNull(MetaTableAccessor.getHRegionInfo(new Result()));
089
090    List<Cell> kvs = new ArrayList<>();
091    Result r = Result.create(kvs);
092    assertNull(MetaTableAccessor.getHRegionInfo(r));
093
094    byte [] f = HConstants.CATALOG_FAMILY;
095    // Make a key value that doesn't have the expected qualifier.
096    kvs.add(new KeyValue(HConstants.EMPTY_BYTE_ARRAY, f,
097      HConstants.SERVER_QUALIFIER, f));
098    r = Result.create(kvs);
099    assertNull(MetaTableAccessor.getHRegionInfo(r));
100    // Make a key that does not have a regioninfo value.
101    kvs.add(new KeyValue(HConstants.EMPTY_BYTE_ARRAY, f,
102      HConstants.REGIONINFO_QUALIFIER, f));
103    HRegionInfo hri = MetaTableAccessor.getHRegionInfo(Result.create(kvs));
104    assertTrue(hri == null);
105    // OK, give it what it expects
106    kvs.clear();
107    kvs.add(new KeyValue(HConstants.EMPTY_BYTE_ARRAY, f,
108      HConstants.REGIONINFO_QUALIFIER,
109      HRegionInfo.FIRST_META_REGIONINFO.toByteArray()));
110    hri = MetaTableAccessor.getHRegionInfo(Result.create(kvs));
111    assertNotNull(hri);
112    assertTrue(hri.equals(HRegionInfo.FIRST_META_REGIONINFO));
113  }

115  /**
116   * Test that MetaTableAccessor will ride over server throwing
117   * "Server not running" IOEs.
118   * @see @link {https://issues.apache.org/jira/browse/HBASE-3446}
119   * @throws IOException
120   * @throws InterruptedException
121   */
122  @Test
123  public void testRideOverServerNotRunning()
124      throws IOException, InterruptedException, ServiceException {
125    // Need a zk watcher.
126    ZooKeeperWatcher zkw = new ZooKeeperWatcher(UTIL.getConfiguration(),
127      this.getClass().getSimpleName(), ABORTABLE, true);
128    // This is a servername we use in a few places below.
129    ServerName sn = ServerName.valueOf("example.com", 1234, System.currentTimeMillis());
130
131    ClusterConnection connection = null;
132    try {
133      // Mock an ClientProtocol. Our mock implementation will fail a few
134      // times when we go to open a scanner.
135      final ClientProtos.ClientService.BlockingInterface implementation =
136        Mockito.mock(ClientProtos.ClientService.BlockingInterface.class);
137      // When scan called throw IOE 'Server not running' a few times
138      // before we return a scanner id.  Whats WEIRD is that these
139      // exceptions do not show in the log because they are caught and only
140      // printed if we FAIL.  We eventually succeed after retry so these don't
141      // show.  We will know if they happened or not because we will ask
142      // mockito at the end of this test to verify that scan was indeed
143      // called the wanted number of times.
144      List<Cell> kvs = new ArrayList<>();
145      final byte [] rowToVerify = Bytes.toBytes("rowToVerify");
146      kvs.add(new KeyValue(rowToVerify,
147        HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER,
148        HRegionInfo.FIRST_META_REGIONINFO.toByteArray()));
149      kvs.add(new KeyValue(rowToVerify,
150        HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER,
151        Bytes.toBytes(sn.getHostAndPort())));
152      kvs.add(new KeyValue(rowToVerify,
153        HConstants.CATALOG_FAMILY, HConstants.STARTCODE_QUALIFIER,
154        Bytes.toBytes(sn.getStartcode())));
155      final List<CellScannable> cellScannables = new ArrayList<>(1);
156      cellScannables.add(Result.create(kvs));
157      final ScanResponse.Builder builder = ScanResponse.newBuilder();
158      for (CellScannable result : cellScannables) {
159        builder.addCellsPerResult(((Result)result).size());
160      }
161      Mockito.when(implementation.scan((RpcController) Mockito.any(), (ScanRequest) Mockito.any()))
162          .thenThrow(new ServiceException("Server not running (1 of 3)"))
163          .thenThrow(new ServiceException("Server not running (2 of 3)"))
164          .thenThrow(new ServiceException("Server not running (3 of 3)"))
165          .thenAnswer(new Answer<ScanResponse>() {
166            public ScanResponse answer(InvocationOnMock invocation) throws Throwable {
167              ((HBaseRpcController) invocation.getArguments()[0]).setCellScanner(CellUtil
168                  .createCellScanner(cellScannables));
169              return builder.setScannerId(1234567890L).setMoreResults(false).build();
170            }
171          });
172      // Associate a spied-upon Connection with UTIL.getConfiguration.  Need
173      // to shove this in here first so it gets picked up all over; e.g. by
174      // HTable.
175      connection = HConnectionTestingUtility.getSpiedConnection(UTIL.getConfiguration());
176      
177      // Fix the location lookup so it 'works' though no network.  First
178      // make an 'any location' object.
179      final HRegionLocation anyLocation =
180        new HRegionLocation(HRegionInfo.FIRST_META_REGIONINFO, sn);
181      final RegionLocations rl = new RegionLocations(anyLocation);
182      // Return the RegionLocations object when locateRegion
183      // The ugly format below comes of 'Important gotcha on spying real objects!' from
184      // http://mockito.googlecode.com/svn/branches/1.6/javadoc/org/mockito/Mockito.html
185      Mockito.doReturn(rl).when
186      (connection).locateRegion((TableName)Mockito.any(), (byte[])Mockito.any(),
187              Mockito.anyBoolean(), Mockito.anyBoolean(), Mockito.anyInt());
188
189      // Now shove our HRI implementation into the spied-upon connection.
190      Mockito.doReturn(implementation).
191        when(connection).getClient(Mockito.any(ServerName.class));
192
193      // Scan meta for user tables and verify we got back expected answer.
194      NavigableMap<HRegionInfo, Result> hris =
195        MetaTableAccessor.getServerUserRegions(connection, sn);
196      assertEquals(1, hris.size());
197      assertTrue(hris.firstEntry().getKey().equals(HRegionInfo.FIRST_META_REGIONINFO));
198      assertTrue(Bytes.equals(rowToVerify, hris.firstEntry().getValue().getRow()));
199      // Finally verify that scan was called four times -- three times
200      // with exception and then on 4th attempt we succeed
201      Mockito.verify(implementation, Mockito.times(4)).
202        scan((RpcController)Mockito.any(), (ScanRequest)Mockito.any());
203    } finally {
204      if (connection != null && !connection.isClosed()) connection.close();
205      zkw.close();
206    }
207  }
