107  private static List<ServerName> getServersFromRack(Map<String, Integer> rackToServerCount) {
108    List<ServerName> chosenServers = new ArrayList<>();
109    for (Map.Entry<String, Integer> entry : rackToServerCount.entrySet()) {

111      for (int i = 0; i < entry.getValue(); i++) {
112        chosenServers.add(servers.get(i));
113      }
114    }
115    return chosenServers;

253  @Test
254  public void testSecondaryAndTertiaryPlacementWithMoreThanOneServerInPrimaryRack() {
255    // Test the case where there is only one server in one rack and another rack
256    // has more servers. We try to choose secondary/tertiary on different
257    // racks than what the primary is on. But if the other rack doesn't have
258    // enough nodes to have both secondary/tertiary RSs, the tertiary is placed
259    // on the same rack as the primary server is on
260    Map<String,Integer> rackToServerCount = new HashMap<>();
261    rackToServerCount.put("rack1", 2);
262    rackToServerCount.put("rack2", 1);
263    Triple<Map<RegionInfo, ServerName>, FavoredNodeAssignmentHelper, List<RegionInfo>>
264      primaryRSMapAndHelper = secondaryAndTertiaryRSPlacementHelper(6, rackToServerCount);
265    FavoredNodeAssignmentHelper helper = primaryRSMapAndHelper.getSecond();
266    Map<RegionInfo, ServerName> primaryRSMap = primaryRSMapAndHelper.getFirst();
267    List<RegionInfo> regions = primaryRSMapAndHelper.getThird();
268    assertTrue(primaryRSMap.size() == 6);
269    Map<RegionInfo, ServerName[]> secondaryAndTertiaryMap =
270          helper.placeSecondaryAndTertiaryRS(primaryRSMap);
271    assertTrue(secondaryAndTertiaryMap.size() == regions.size());
272    for (RegionInfo region : regions) {
273      ServerName s = primaryRSMap.get(region);
274      ServerName secondaryRS = secondaryAndTertiaryMap.get(region)[0];
275      ServerName tertiaryRS = secondaryAndTertiaryMap.get(region)[1];
276      Set<String> racks = Sets.newHashSet(rackManager.getRack(s));
277      racks.add(rackManager.getRack(secondaryRS));
278      racks.add(rackManager.getRack(tertiaryRS));
279      assertTrue(racks.size() >= 2);
280    }
281  }

283  private Triple<Map<RegionInfo, ServerName>, FavoredNodeAssignmentHelper, List<RegionInfo>>
284  secondaryAndTertiaryRSPlacementHelper(
285      int regionCount, Map<String, Integer> rackToServerCount) {
286    Map<RegionInfo, ServerName> primaryRSMap = new HashMap<RegionInfo, ServerName>();
287    List<ServerName> servers = getServersFromRack(rackToServerCount);
288    FavoredNodeAssignmentHelper helper = new FavoredNodeAssignmentHelper(servers, rackManager);
289    Map<ServerName, List<RegionInfo>> assignmentMap =
290        new HashMap<ServerName, List<RegionInfo>>();
291    helper.initialize();
292    // create regions
293    List<RegionInfo> regions = new ArrayList<>(regionCount);
294    for (int i = 0; i < regionCount; i++) {
295      regions.add(RegionInfoBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
296          .setStartKey(Bytes.toBytes(i))
297          .setEndKey(Bytes.toBytes(i + 1))
298          .build());
299    }
300    // place the regions
301    helper.placePrimaryRSAsRoundRobin(assignmentMap, primaryRSMap, regions);
302    return new Triple<>(primaryRSMap, helper, regions);
303  }

305  private void primaryRSPlacement(int regionCount, Map<RegionInfo, ServerName> primaryRSMap,
306      int firstRackSize, int secondRackSize, int thirdRackSize) {
307    Map<String,Integer> rackToServerCount = new HashMap<>();
308    rackToServerCount.put("rack1", firstRackSize);
309    rackToServerCount.put("rack2", secondRackSize);
310    rackToServerCount.put("rack3", thirdRackSize);
311    List<ServerName> servers = getServersFromRack(rackToServerCount);
312    FavoredNodeAssignmentHelper helper = new FavoredNodeAssignmentHelper(servers,
313        rackManager);
314    helper.initialize();
315
316    assertTrue(helper.canPlaceFavoredNodes());
317
318    Map<ServerName, List<RegionInfo>> assignmentMap = new HashMap<>();
319    if (primaryRSMap == null) primaryRSMap = new HashMap<>();
320    // create some regions
321    List<RegionInfo> regions = new ArrayList<>(regionCount);
322    for (int i = 0; i < regionCount; i++) {
323      regions.add(RegionInfoBuilder.newBuilder(TableName.valueOf("foobar"))
324          .setStartKey(Bytes.toBytes(i))
325          .setEndKey(Bytes.toBytes(i + 1))
326          .build());
327    }
328    // place those regions in primary RSs
329    helper.placePrimaryRSAsRoundRobin(assignmentMap, primaryRSMap, regions);
330
331    // we should have all the regions nicely spread across the racks
332    int regionsOnRack1 = 0;
333    int regionsOnRack2 = 0;
334    int regionsOnRack3 = 0;
335    for (RegionInfo region : regions) {
336      if (rackManager.getRack(primaryRSMap.get(region)).equals("rack1")) {
337        regionsOnRack1++;
338      } else if (rackManager.getRack(primaryRSMap.get(region)).equals("rack2")) {
339        regionsOnRack2++;
340      } else if (rackManager.getRack(primaryRSMap.get(region)).equals("rack3")) {
341        regionsOnRack3++;
342      }
343    }
344    // Verify that the regions got placed in the way we expect (documented in
345    // FavoredNodeAssignmentHelper#placePrimaryRSAsRoundRobin)
346    checkNumRegions(regionCount, firstRackSize, secondRackSize, thirdRackSize, regionsOnRack1,
347        regionsOnRack2, regionsOnRack3, assignmentMap);
348  }

350  private void checkNumRegions(int regionCount, int firstRackSize, int secondRackSize,
351      int thirdRackSize, int regionsOnRack1, int regionsOnRack2, int regionsOnRack3,
352      Map<ServerName, List<RegionInfo>> assignmentMap) {
353    //The regions should be distributed proportionately to the racksizes
354    //Verify the ordering was as expected by inserting the racks and regions
355    //in sorted maps. The keys being the racksize and numregions; values are
356    //the relative positions of the racksizes and numregions respectively
357    SortedMap<Integer, Integer> rackMap = new TreeMap<>();
358    rackMap.put(firstRackSize, 1);
359    rackMap.put(secondRackSize, 2);
360    rackMap.put(thirdRackSize, 3);
361    SortedMap<Integer, Integer> regionMap = new TreeMap<>();
362    regionMap.put(regionsOnRack1, 1);
363    regionMap.put(regionsOnRack2, 2);
364    regionMap.put(regionsOnRack3, 3);
365    assertTrue(printProportions(firstRackSize, secondRackSize, thirdRackSize,
366        regionsOnRack1, regionsOnRack2, regionsOnRack3),
367        rackMap.get(firstRackSize) == regionMap.get(regionsOnRack1));
368    assertTrue(printProportions(firstRackSize, secondRackSize, thirdRackSize,
369        regionsOnRack1, regionsOnRack2, regionsOnRack3),
370        rackMap.get(secondRackSize) == regionMap.get(regionsOnRack2));
371    assertTrue(printProportions(firstRackSize, secondRackSize, thirdRackSize,
372        regionsOnRack1, regionsOnRack2, regionsOnRack3),
373        rackMap.get(thirdRackSize) == regionMap.get(regionsOnRack3));
374  }
375
376  private String printProportions(int firstRackSize, int secondRackSize,
377      int thirdRackSize, int regionsOnRack1, int regionsOnRack2, int regionsOnRack3) {
378    return "The rack sizes " + firstRackSize + " " + secondRackSize
379        + " " + thirdRackSize + " " + regionsOnRack1 + " " + regionsOnRack2 +
380        " " + regionsOnRack3;
381  }

640  private void checkFNRacks(List<ServerName> fnList, ServerName genFN) {

648  }
