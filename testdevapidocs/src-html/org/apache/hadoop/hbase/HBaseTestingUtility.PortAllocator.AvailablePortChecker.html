229  public static boolean available(int port) {
230    ServerSocket ss = null;
231    DatagramSocket ds = null;
232    try {
233      ss = new ServerSocket(port);
234      ss.setReuseAddress(true);
235      ds = new DatagramSocket(port);
236      ds.setReuseAddress(true);
237      return true;
238    } catch (IOException e) {
239      // Do nothing
240    } finally {
241      if (ds != null) {
242        ds.close();
243      }
244
245      if (ss != null) {
246        try {
247          ss.close();
248        } catch (IOException e) {
249          /* should not be thrown */
250        }
251      }
252    }
253
254    return false;
255  }

390  /**
391   * Home our data in a dir under {@link #DEFAULT_BASE_TEST_DIRECTORY}.
392   * Give it a random name so can have many concurrent tests running if
393   * we need to.  It needs to amend the {@link #TEST_DIRECTORY_KEY}
394   * System property, as it's what minidfscluster bases
395   * it data dir on.  Moding a System property is not the way to do concurrent
396   * instances -- another instance could grab the temporary
397   * value unintentionally -- but not anything can do about it at moment;
398   * single instance only is how the minidfscluster works.
399   *
400   * We also create the underlying directory for
401   *  hadoop.log.dir, mapreduce.cluster.local.dir and hadoop.tmp.dir, and set the values
402   *  in the conf, and as a system property for hadoop.tmp.dir
403   *
404   * @return The calculated data test build directory, if newly-created.
405   */
406  @Override
407  protected Path setupDataTestDir() {
408    Path testPath = super.setupDataTestDir();
409    if (null == testPath) {
410      return null;
411    }
412
413    createSubDirAndSystemProperty(
414      "hadoop.log.dir",
415      testPath, "hadoop-log-dir");
416
417    // This is defaulted in core-default.xml to /tmp/hadoop-${user.name}, but
418    //  we want our own value to ensure uniqueness on the same machine
419    createSubDirAndSystemProperty(
420      "hadoop.tmp.dir",
421      testPath, "hadoop-tmp-dir");
422
423    // Read and modified in org.apache.hadoop.mapred.MiniMRCluster
424    createSubDir(
425      "mapreduce.cluster.local.dir",
426      testPath, "mapred-local-dir");
427
428    return testPath;
429  }

431  private void createSubDirAndSystemProperty(
432    String propertyName, Path parent, String subDirName){
433
434    String sysValue = System.getProperty(propertyName);
435
436    if (sysValue != null) {
437      // There is already a value set. So we do nothing but hope
438      //  that there will be no conflicts
439      LOG.info("System.getProperty(\""+propertyName+"\") already set to: "+
440        sysValue + " so I do NOT create it in " + parent);
441      String confValue = conf.get(propertyName);
442      if (confValue != null && !confValue.endsWith(sysValue)){
443       LOG.warn(
444         propertyName + " property value differs in configuration and system: "+
445         "Configuration="+confValue+" while System="+sysValue+
446         " Erasing configuration value by system value."
447       );
448      }
449      conf.set(propertyName, sysValue);
450    } else {
451      // Ok, it's not set, so we create it as a subdirectory
452      createSubDir(propertyName, parent, subDirName);
453      System.setProperty(propertyName, conf.get(propertyName));
454    }
455  }

531    //The file system can be either local, mini dfs, or if the configuration
532    //is supplied externally, it can be an external cluster FS. If it is a local
533    //file system, the tests should use getBaseTestDir, otherwise, we can use
534    //the working directory, and create a unique sub dir there
535    FileSystem fs = getTestFileSystem();
536    Path newDataTestDir;

538    if (fs.getUri().getScheme().equals(FileSystem.getLocal(conf).getUri().getScheme())) {
539      newDataTestDir = new Path(getDataTestDir(), randomStr);
540      File dataTestDir = new File(newDataTestDir.toString());
541      if (deleteOnExit()) dataTestDir.deleteOnExit();
542    } else {
543      Path base = getBaseTestDirOnTestFS();
544      newDataTestDir = new Path(base, randomStr);
545      if (deleteOnExit()) fs.deleteOnExit(newDataTestDir);
546    }
547    return newDataTestDir;
548  }

979  /**
980   * Starts up mini hbase cluster.  Usually used after call to
981   * {@link #startMiniCluster(int, int)} when doing stepped startup of clusters.
982   * Usually you won't want this.  You'll usually want {@link #startMiniCluster()}.
983   * @param numMasters
984   * @param numSlaves
985   * @param create Whether to create a
986   * root or data directory path or not; will overwrite if exists already.
987   * @return Reference to the hbase mini hbase cluster.
988   * @throws IOException
989   * @throws InterruptedException
990   * @see {@link #startMiniCluster()}
991   */
992  public MiniHBaseCluster startMiniHBaseCluster(final int numMasters,
993        final int numSlaves, Class<? extends HMaster> masterClass,
994        Class<? extends MiniHBaseCluster.MiniHBaseClusterRegionServer> regionserverClass,
995        boolean create, boolean withWALDir)
996  throws IOException, InterruptedException {
997    // Now do the mini hbase cluster.  Set the hbase.rootdir in config.
998    createRootDir(create);
999    if (withWALDir) {
1000      createWALRootDir();
1001    }
1002    // Set the hbase.fs.tmp.dir config to make sure that we have some default value. This is
1003    // for tests that do not read hbase-defaults.xml
1004    setHBaseFsTmpDir();
1005
1006    // These settings will make the server waits until this exact number of
1007    // regions servers are connected.
1008    if (conf.getInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, -1) == -1) {
1009      conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, numSlaves);
1010    }
1011    if (conf.getInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART, -1) == -1) {
1012      conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART, numSlaves);
1013    }
1014
1015    Configuration c = new Configuration(this.conf);
1016    TraceUtil.initTracer(c);
1017    this.hbaseCluster =
1018        new MiniHBaseCluster(c, numMasters, numSlaves, masterClass, regionserverClass);
1019    // Don't leave here till we've done a successful scan of the hbase:meta
1020    Table t = getConnection().getTable(TableName.META_TABLE_NAME);
1021    ResultScanner s = t.getScanner(new Scan());
1022    while (s.next() != null) {
1023      continue;
1024    }
1025    s.close();
1026    t.close();
1027
1028    getAdmin(); // create immediately the hbaseAdmin
1029    LOG.info("Minicluster is up");
1030
1031    return (MiniHBaseCluster)this.hbaseCluster;
1032  }

1577   */
1578  @SuppressWarnings("serial")
1579  public static void modifyTableSync(Admin admin, TableDescriptor desc)
1580      throws IOException, InterruptedException {
1581    admin.modifyTable(desc);
1582    Pair<Integer, Integer> status = new Pair<Integer, Integer>() {{
1583      setFirst(0);
1584      setSecond(0);
1585    }};
1586    int i = 0;
1587    do {
1588      status = admin.getAlterStatus(desc.getTableName());
1589      if (status.getSecond() != 0) {
1590        LOG.debug(status.getSecond() - status.getFirst() + "/" + status.getSecond()
1591          + " regions updated.");
1592        Thread.sleep(1 * 1000l);
1593      } else {
1594        LOG.debug("All regions updated.");
1595        break;
1596      }
1597    } while (status.getFirst() != 0 && i++ < 500);
1598    if (status.getFirst() != 0) {
1599      throw new IOException("Failed to update all regions even after 500 seconds.");
1600    }
1601  }

1937  /** A tracker for tracking and validating table rows
1938   * generated with {@link HBaseTestingUtility#loadTable(Table, byte[])}
1939   */
1940  public static class SeenRowTracker {
1941    int dim = 'z' - 'a' + 1;
1942    int[][][] seenRows = new int[dim][dim][dim]; //count of how many times the row is seen
1943    byte[] startRow;
1944    byte[] stopRow;
1945
1946    public SeenRowTracker(byte[] startRow, byte[] stopRow) {
1947      this.startRow = startRow;
1948      this.stopRow = stopRow;
1949    }
1950
1951    void reset() {
1952      for (byte[] row : ROWS) {
1953        seenRows[i(row[0])][i(row[1])][i(row[2])] = 0;
1954      }
1955    }

1957    int i(byte b) {
1958      return b - 'a';
1959    }

1966     * all other rows none
1967     */

1970        for (byte b2 = 'a'; b2 <= 'z'; b2++) {
1971          for (byte b3 = 'a'; b3 <= 'z'; b3++) {
1972            int count = seenRows[i(b1)][i(b2)][i(b3)];
1973            int expectedCount = 0;
1974            if (Bytes.compareTo(new byte[] {b1,b2,b3}, startRow) >= 0
1975                && Bytes.compareTo(new byte[] {b1,b2,b3}, stopRow) < 0) {
1976              expectedCount = 1;
1977            }
1978            if (count != expectedCount) {
1979              String row = new String(new byte[] {b1,b2,b3});
1980              throw new RuntimeException("Row:" + row + " has a seen count of " + count + " " +
1981                  "instead of " + expectedCount);
1982            }
1983          }
1984        }
1985      }

2648  /**
2649   * Expire a ZooKeeper session as recommended in ZooKeeper documentation
2650   * http://hbase.apache.org/book.html#trouble.zookeeper
2651   * There are issues when doing this:
2652   * [1] http://www.mail-archive.com/dev@zookeeper.apache.org/msg01942.html
2653   * [2] https://issues.apache.org/jira/browse/ZOOKEEPER-1105
2654   *
2655   * @param nodeZK - the ZK watcher to expire
2656   * @param checkStatus - true to check if we can create a Table with the
2657   *                    current configuration.
2658   */
2659  public void expireSession(ZKWatcher nodeZK, boolean checkStatus)
2660    throws Exception {
2661    Configuration c = new Configuration(this.conf);
2662    String quorumServers = ZKConfig.getZKQuorumServersString(c);
2663    ZooKeeper zk = nodeZK.getRecoverableZooKeeper().getZooKeeper();
2664    byte[] password = zk.getSessionPasswd();
2665    long sessionID = zk.getSessionId();
2666
2667    // Expiry seems to be asynchronous (see comment from P. Hunt in [1]),
2668    //  so we create a first watcher to be sure that the
2669    //  event was sent. We expect that if our watcher receives the event
2670    //  other watchers on the same machine will get is as well.
2671    // When we ask to close the connection, ZK does not close it before
2672    //  we receive all the events, so don't have to capture the event, just
2673    //  closing the connection should be enough.
2674    ZooKeeper monitor = new ZooKeeper(quorumServers,
2675      1000, new org.apache.zookeeper.Watcher(){
2676      @Override
2677      public void process(WatchedEvent watchedEvent) {
2678        LOG.info("Monitor ZKW received event="+watchedEvent);
2679      }
2680    } , sessionID, password);
2681
2682    // Making it expire
2683    ZooKeeper newZK = new ZooKeeper(quorumServers,
2684        1000, EmptyWatcher.instance, sessionID, password);
2685
2686    //ensure that we have connection to the server before closing down, otherwise
2687    //the close session event will be eaten out before we start CONNECTING state
2688    long start = System.currentTimeMillis();
2689    while (newZK.getState() != States.CONNECTED
2690         && System.currentTimeMillis() - start < 1000) {
2691       Thread.sleep(1);
2692    }
2693    newZK.close();
2694    LOG.info("ZK Closed Session 0x" + Long.toHexString(sessionID));
2695
2696    // Now closing & waiting to be sure that the clients get it.
2697    monitor.close();
2698
2699    if (checkStatus) {
2700      getConnection().getTable(TableName.META_TABLE_NAME).close();
2701    }
2702  }

2815  /*
2816   * Retrieves a splittable region randomly from tableName
2817   *
2818   * @param tableName name of table
2819   * @param maxAttempts maximum number of attempts, unlimited for value of -1
2820   * @return the HRegion chosen, null if none was found within limit of maxAttempts
2821   */
2822  public HRegion getSplittableRegion(TableName tableName, int maxAttempts) {
2823    List<HRegion> regions = getHBaseCluster().getRegions(tableName);
2824    int regCount = regions.size();
2825    Set<Integer> attempted = new HashSet<>();
2826    int idx;
2827    int attempts = 0;
2828    do {
2829      regions = getHBaseCluster().getRegions(tableName);
2830      if (regCount != regions.size()) {
2831        // if there was region movement, clear attempted Set
2832        attempted.clear();
2833      }
2834      regCount = regions.size();
2835      // There are chances that before we get the region for the table from an RS the region may
2836      // be going for CLOSE.  This may be because online schema change is enabled
2837      if (regCount > 0) {
2838        idx = random.nextInt(regCount);
2839        // if we have just tried this region, there is no need to try again
2840        if (attempted.contains(idx))
2841          continue;
2842        try {
2843          regions.get(idx).checkSplit();
2844          return regions.get(idx);
2845        } catch (Exception ex) {
2846          LOG.warn("Caught exception", ex);
2847          attempted.add(idx);
2848        }
2849      }
2850      attempts++;
2851    } while (maxAttempts == -1 || attempts < maxAttempts);
2852    return null;
2853  }

3702  /**
3703   * Creates a pre-split table for load testing. If the table already exists,
3704   * logs a warning and continues.
3705   * @return the number of regions the table was split into
3706   */
3707  public static int createPreSplitLoadTestTable(Configuration conf,
3708      TableDescriptor td, ColumnFamilyDescriptor[] cds,
3709      SplitAlgorithm splitter, int numRegionsPerServer) throws IOException {
3710    TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(td);
3711    for (ColumnFamilyDescriptor cd : cds) {
3712      if (!td.hasColumnFamily(cd.getName())) {
3713        builder.addColumnFamily(cd);
3714      }
3715    }
3716    td = builder.build();
3717    int totalNumberOfRegions = 0;
3718    Connection unmanagedConnection = ConnectionFactory.createConnection(conf);
3719    Admin admin = unmanagedConnection.getAdmin();
3720
3721    try {
3722      // create a table a pre-splits regions.
3723      // The number of splits is set as:
3724      //    region servers * regions per region server).
3725      int numberOfServers =
3726          admin.getClusterStatus(EnumSet.of(Option.LIVE_SERVERS)).getServers()
3727              .size();
3728      if (numberOfServers == 0) {
3729        throw new IllegalStateException("No live regionservers");
3730      }
3731
3732      totalNumberOfRegions = numberOfServers * numRegionsPerServer;
3733      LOG.info("Number of live regionservers: " + numberOfServers + ", " +
3734          "pre-splitting table into " + totalNumberOfRegions + " regions " +
3735          "(regions per server: " + numRegionsPerServer + ")");
3736
3737      byte[][] splits = splitter.split(
3738          totalNumberOfRegions);
3739
3740      admin.createTable(td, splits);
3741    } catch (MasterNotRunningException e) {
3742      LOG.error("Master not running", e);
3743      throw new IOException(e);
3744    } catch (TableExistsException e) {
3745      LOG.warn("Table " + td.getTableName() +
3746          " already exists, continuing");
3747    } finally {
3748      admin.close();
3749      unmanagedConnection.close();
3750    }
3751    return totalNumberOfRegions;
3752  }

4072          throw e;
4073        }
4074        LOG.error("BindException encountered when setting up MiniKdc. Trying again.");
4075        bindException = true;
4076      }
