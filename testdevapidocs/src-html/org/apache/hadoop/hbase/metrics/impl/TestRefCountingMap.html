021package org.apache.hadoop.hbase.metrics.impl;
022
023import static org.junit.Assert.assertEquals;
024import static org.junit.Assert.assertNotNull;
025import static org.junit.Assert.assertNull;
026import static org.junit.Assert.assertTrue;
027
028import java.util.Collection;
029import java.util.Set;
030
031import org.apache.hadoop.hbase.testclassification.SmallTests;
032import org.junit.Before;
033import org.junit.Test;
034import org.junit.experimental.categories.Category;
035
036import org.apache.hadoop.hbase.shaded.com.google.common.collect.Lists;
037
038@Category(SmallTests.class)
039public class TestRefCountingMap {
040
041  private RefCountingMap<String, String> map;
042
043  @Before
044  public void setUp() {
045    map = new RefCountingMap<>();
046  }
047
048  @Test
049  public void testPutGet() {
050    map.put("foo", () -> "foovalue");
051
052    String v = map.get("foo");
053    assertNotNull(v);
054    assertEquals("foovalue", v);
055  }
056
057  @Test
058  public void testPutMulti() {
059    String v1 = map.put("foo", () -> "foovalue");
060    String v2 =  map.put("foo", () -> "foovalue2");
061    String v3 = map.put("foo", () -> "foovalue3");
062
063    String v = map.get("foo");
064    assertEquals("foovalue", v);
065    assertEquals(v, v1);
066    assertEquals(v, v2);
067    assertEquals(v, v3);
068  }
069
070  @Test
071  public void testPutRemove() {
072    map.put("foo", () -> "foovalue");
073    String v = map.remove("foo");
074    assertNull(v);
075    v = map.get("foo");
076    assertNull(v);
077  }
078
079  @Test
080  public void testPutRemoveMulti() {
081    map.put("foo", () -> "foovalue");
082    map.put("foo", () -> "foovalue2");
083    map.put("foo", () -> "foovalue3");
084
085    // remove 1
086    String v = map.remove("foo");
087    assertEquals("foovalue", v);
088
089    // remove 2
090    v = map.remove("foo");
091    assertEquals("foovalue", v);
092
093    // remove 3
094    v = map.remove("foo");
095    assertNull(v);
096    v = map.get("foo");
097    assertNull(v);
098  }
099
100  @Test
101  public void testSize() {
102    assertEquals(0, map.size());
103
104    // put a key
105    map.put("foo", () -> "foovalue");
106    assertEquals(1, map.size());
107
108    // put a different key
109    map.put("bar", () -> "foovalue2");
110    assertEquals(2, map.size());
111
112    // put the same key again
113    map.put("bar", () -> "foovalue3");
114    assertEquals(2, map.size()); // map should be same size
115  }
116
117  @Test
118  public void testClear() {
119    map.put("foo", () -> "foovalue");
120    map.put("bar", () -> "foovalue2");
121    map.put("baz", () -> "foovalue3");
122
123    map.clear();
124
125    assertEquals(0, map.size());
126  }
127
128
129  @Test
130  public void testKeySet() {
131    map.put("foo", () -> "foovalue");
132    map.put("bar", () -> "foovalue2");
133    map.put("baz", () -> "foovalue3");
134
135    Set<String> keys = map.keySet();
136    assertEquals(3, keys.size());
137
138    Lists.newArrayList("foo", "bar", "baz").stream().forEach( v -> assertTrue(keys.contains(v)));
139  }
140
141  @Test
142  public void testValues() {
143    map.put("foo", () -> "foovalue");
144    map.put("foo", () -> "foovalue2");
145    map.put("bar", () -> "foovalue3");
146    map.put("baz", () -> "foovalue4");
147
148    Collection<String> values = map.values();
149    assertEquals(3, values.size());
150
151    Lists.newArrayList("foovalue", "foovalue3", "foovalue4").stream().forEach(
152        v -> assertTrue(values.contains(v)));
153  }
154}
