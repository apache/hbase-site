018package org.apache.hadoop.hbase.metrics;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertNull;
022
023import org.apache.hadoop.hbase.HBaseClassTestRule;
024import org.apache.hadoop.hbase.testclassification.MetricsTests;
025import org.apache.hadoop.hbase.testclassification.SmallTests;
026import org.apache.hadoop.metrics2.lib.MutableFastCounter;
027import org.apache.hadoop.metrics2.lib.MutableGaugeLong;
028import org.junit.BeforeClass;
029import org.junit.ClassRule;
030import org.junit.Test;
031import org.junit.experimental.categories.Category;
032
033/**
034 *  Test of default BaseSource for hadoop 2
035 */
036@Category({MetricsTests.class, SmallTests.class})
037public class TestBaseSourceImpl {
038
039  @ClassRule
040  public static final HBaseClassTestRule CLASS_RULE =
041      HBaseClassTestRule.forClass(TestBaseSourceImpl.class);
042
043  private static BaseSourceImpl bmsi;
044
045  @BeforeClass
046  public static void setUp() throws Exception {
047    bmsi = new BaseSourceImpl("TestName", "test description", "testcontext", "TestContext");
048  }
049
050  @Test
051  public void testSetGauge() throws Exception {
052    bmsi.setGauge("testset", 100);
053    assertEquals(100, ((MutableGaugeLong) bmsi.metricsRegistry.get("testset")).value());
054    bmsi.setGauge("testset", 300);
055    assertEquals(300, ((MutableGaugeLong) bmsi.metricsRegistry.get("testset")).value());
056
057  }
058
059  @Test
060  public void testIncGauge() throws Exception {
061    bmsi.incGauge("testincgauge", 100);
062    assertEquals(100, ((MutableGaugeLong) bmsi.metricsRegistry.get("testincgauge")).value());
063    bmsi.incGauge("testincgauge", 100);
064    assertEquals(200, ((MutableGaugeLong) bmsi.metricsRegistry.get("testincgauge")).value());
065
066  }
067
068  @Test
069  public void testDecGauge() throws Exception {
070    bmsi.decGauge("testdec", 100);
071    assertEquals(-100, ((MutableGaugeLong) bmsi.metricsRegistry.get("testdec")).value());
072    bmsi.decGauge("testdec", 100);
073    assertEquals(-200, ((MutableGaugeLong) bmsi.metricsRegistry.get("testdec")).value());
074
075  }
076
077  @Test
078  public void testIncCounters() throws Exception {
079    bmsi.incCounters("testinccounter", 100);
080    assertEquals(100, ((MutableFastCounter) bmsi.metricsRegistry.get("testinccounter")).value());
081    bmsi.incCounters("testinccounter", 100);
082    assertEquals(200, ((MutableFastCounter) bmsi.metricsRegistry.get("testinccounter")).value());
083
084  }
085
086  @Test
087  public void testRemoveMetric() throws Exception {
088    bmsi.setGauge("testrmgauge", 100);
089    bmsi.removeMetric("testrmgauge");
090    assertNull(bmsi.metricsRegistry.get("testrmgauge"));
091  }
092
093}
