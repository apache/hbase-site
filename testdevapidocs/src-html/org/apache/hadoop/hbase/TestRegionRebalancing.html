054/**
055 * Test whether region re-balancing works. (HBASE-71)
056 */
057@Ignore // This is broken since new RegionServers does proper average of regions
058// and because Master is treated as a regionserver though it hosts two regions only.
059@Category({FlakeyTests.class, LargeTests.class})
060@RunWith(value = Parameterized.class)
061public class TestRegionRebalancing {

063  @Parameters
064  public static Collection<Object[]> data() {
065    Object[][] balancers =
066        new String[][] { { "org.apache.hadoop.hbase.master.balancer.SimpleLoadBalancer" },
067            { "org.apache.hadoop.hbase.master.balancer.StochasticLoadBalancer" } };
068    return Arrays.asList(balancers);
069  }

071  private static final byte[] FAMILY_NAME = Bytes.toBytes("col");
072  private static final Log LOG = LogFactory.getLog(TestRegionRebalancing.class);
073  private final HBaseTestingUtility UTIL = new HBaseTestingUtility();
074  private RegionLocator regionLocator;
075  private HTableDescriptor desc;
076  private String balancerName;

078  public TestRegionRebalancing(String balancerName) {
079    this.balancerName = balancerName;

083  @After

088  @Before
089  public void before() throws Exception {
090    UTIL.getConfiguration().set("hbase.master.loadbalancer.class", this.balancerName);
091    // set minCostNeedBalance to 0, make sure balancer run
092    UTIL.startMiniCluster(1);
093    this.desc = new HTableDescriptor(TableName.valueOf("test"));
094    this.desc.addFamily(new HColumnDescriptor(FAMILY_NAME));

097  /**
098   * For HBASE-71. Try a few different configurations of starting and stopping
099   * region servers to see if the assignment or regions is pretty balanced.
100   * @throws IOException
101   * @throws InterruptedException
102   */
103  @Test (timeout=300000)
104  public void testRebalanceOnRegionServerNumberChange()
105  throws IOException, InterruptedException {
106    try(Connection connection = ConnectionFactory.createConnection(UTIL.getConfiguration());
107        Admin admin = connection.getAdmin()) {
108      admin.createTable(this.desc, Arrays.copyOfRange(HBaseTestingUtility.KEYS,
109          1, HBaseTestingUtility.KEYS.length));
110      this.regionLocator = connection.getRegionLocator(this.desc.getTableName());
111
112      MetaTableAccessor.fullScanMetaAndPrint(admin.getConnection());
113
114      assertEquals("Test table should have right number of regions",
115        HBaseTestingUtility.KEYS.length,
116        this.regionLocator.getStartKeys().length);
117
118      // verify that the region assignments are balanced to start out
119      assertRegionsAreBalanced();
120
121      // add a region server - total of 2
122      LOG.info("Started second server=" +
123        UTIL.getHBaseCluster().startRegionServer().getRegionServer().getServerName());
124      UTIL.getHBaseCluster().getMaster().balance();
125      assertRegionsAreBalanced();
126
127      // On a balanced cluster, calling balance() should return true
128      assert(UTIL.getHBaseCluster().getMaster().balance() == true);
129
130      // if we add a server, then the balance() call should return true
131      // add a region server - total of 3
132      LOG.info("Started third server=" +
133          UTIL.getHBaseCluster().startRegionServer().getRegionServer().getServerName());
134      assert(UTIL.getHBaseCluster().getMaster().balance() == true);
135      assertRegionsAreBalanced();
136
137      // kill a region server - total of 2
138      LOG.info("Stopped third server=" + UTIL.getHBaseCluster().stopRegionServer(2, false));
139      UTIL.getHBaseCluster().waitOnRegionServer(2);
140      waitOnCrashProcessing();
141      UTIL.getHBaseCluster().getMaster().balance();
142      assertRegionsAreBalanced();
143
144      // start two more region servers - total of 4
145      LOG.info("Readding third server=" +
146          UTIL.getHBaseCluster().startRegionServer().getRegionServer().getServerName());
147      LOG.info("Added fourth server=" +
148          UTIL.getHBaseCluster().startRegionServer().getRegionServer().getServerName());
149      waitOnCrashProcessing();
150      assert(UTIL.getHBaseCluster().getMaster().balance() == true);
151      assertRegionsAreBalanced();
152      for (int i = 0; i < 6; i++){
153        LOG.info("Adding " + (i + 5) + "th region server");
154        UTIL.getHBaseCluster().startRegionServer();
155      }
156      assert(UTIL.getHBaseCluster().getMaster().balance() == true);
157      assertRegionsAreBalanced();
158      regionLocator.close();
159    }
160  }

162  /**
163   * Wait on crash processing. Balancer won't run if processing a crashed server.
164   */
165  private void waitOnCrashProcessing() {
166    while (UTIL.getHBaseCluster().getMaster().getServerManager().areDeadServersInProgress()) {
167      LOG.info("Waiting on processing of crashed server before proceeding...");
168      Threads.sleep(1000);
169    }
170  }
171
172  /**
173   * Determine if regions are balanced. Figure out the total, divide by the
174   * number of online servers, then test if each server is +/- 1 of average
175   * rounded up.
176   */
177  private void assertRegionsAreBalanced() throws IOException {
178    // TODO: Fix this test.  Old balancer used to run with 'slop'.  New
179    // balancer does not.
180    boolean success = false;
181    float slop = (float)UTIL.getConfiguration().getFloat("hbase.regions.slop", 0.1f);
182    if (slop <= 0) slop = 1;
183
184    for (int i = 0; i < 5; i++) {
185      success = true;
186      // make sure all the regions are reassigned before we test balance
187      waitForAllRegionsAssigned();
188
189      long regionCount = UTIL.getMiniHBaseCluster().countServedRegions();
190      List<HRegionServer> servers = getOnlineRegionServers();
191      double avg = UTIL.getHBaseCluster().getMaster().getAverageLoad();
192      int avgLoadPlusSlop = (int)Math.ceil(avg * (1 + slop));
193      int avgLoadMinusSlop = (int)Math.floor(avg * (1 - slop)) - 1;
194      LOG.debug("There are " + servers.size() + " servers and " + regionCount
195        + " regions. Load Average: " + avg + " low border: " + avgLoadMinusSlop
196        + ", up border: " + avgLoadPlusSlop + "; attempt: " + i);
197
198      for (HRegionServer server : servers) {
199        int serverLoad =
200          ProtobufUtil.getOnlineRegions(server.getRSRpcServices()).size();
201        LOG.debug(server.getServerName() + " Avg: " + avg + " actual: " + serverLoad);
202        if (!(avg > 2.0 && serverLoad <= avgLoadPlusSlop
203            && serverLoad >= avgLoadMinusSlop)) {
204          for (RegionInfo hri :
205              ProtobufUtil.getOnlineRegions(server.getRSRpcServices())) {
206            if (hri.isMetaRegion()) serverLoad--;
207            // LOG.debug(hri.getRegionNameAsString());
208          }
209          if (!(serverLoad <= avgLoadPlusSlop && serverLoad >= avgLoadMinusSlop)) {
210            LOG.debug(server.getServerName() + " Isn't balanced!!! Avg: " + avg +
211                " actual: " + serverLoad + " slop: " + slop);
212            success = false;
213            break;
214          }
215        }
216      }
217
218      if (!success) {
219        // one or more servers are not balanced. sleep a little to give it a
220        // chance to catch up. then, go back to the retry loop.
221        try {
222          Thread.sleep(10000);
223        } catch (InterruptedException e) {}
224
225        UTIL.getHBaseCluster().getMaster().balance();
226        continue;
227      }
228
229      // if we get here, all servers were balanced, so we should just return.
230      return;
231    }
232    // if we get here, we tried 5 times and never got to short circuit out of
233    // the retry loop, so this is a failure.
234    fail("After 5 attempts, region assignments were not balanced.");
235  }
236
237  private List<HRegionServer> getOnlineRegionServers() {
238    List<HRegionServer> list = new ArrayList<>();
239    for (JVMClusterUtil.RegionServerThread rst :
240        UTIL.getHBaseCluster().getRegionServerThreads()) {
241      if (rst.getRegionServer().isOnline()) {
242        list.add(rst.getRegionServer());
243      }
244    }
245    return list;
246  }
247
248  /**
249   * Wait until all the regions are assigned.
250   */
251  private void waitForAllRegionsAssigned() throws IOException {
252    int totalRegions = HBaseTestingUtility.KEYS.length;
253    while (UTIL.getMiniHBaseCluster().countServedRegions() < totalRegions) {
254    // while (!cluster.getMaster().allRegionsAssigned()) {
255      LOG.debug("Waiting for there to be "+ totalRegions +" regions, but there are "
256        + UTIL.getMiniHBaseCluster().countServedRegions() + " right now.");
257      try {
258        Thread.sleep(200);
259      } catch (InterruptedException e) {}
260    }
261    UTIL.waitUntilNoRegionsInTransition();
262  }
