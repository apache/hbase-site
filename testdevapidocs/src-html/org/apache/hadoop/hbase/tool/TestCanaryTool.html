086  @Test
087  public void testBasicZookeeperCanaryWorks() throws Exception {
088    Integer port =
089        Iterables.getOnlyElement(testingUtility.getZkCluster().getClientPortList(), null);
090    testingUtility.getConfiguration().set(HConstants.ZOOKEEPER_QUORUM,
091        "localhost:" + port + "/hbase");
092    ExecutorService executor = new ScheduledThreadPoolExecutor(2);
093    Canary.ZookeeperStdOutSink sink = spy(new Canary.ZookeeperStdOutSink());
094    Canary canary = new Canary(executor, sink);
095    String[] args = { "-t", "10000", "-zookeeper" };
096    assertEquals(0, ToolRunner.run(testingUtility.getConfiguration(), canary, args));
097
098    String baseZnode = testingUtility.getConfiguration()
099        .get(HConstants.ZOOKEEPER_ZNODE_PARENT, HConstants.DEFAULT_ZOOKEEPER_ZNODE_PARENT);
100    verify(sink, atLeastOnce())
101        .publishReadTiming(eq(baseZnode), eq("localhost:" + port), anyLong());
102  }

104  @Test
105  public void testBasicCanaryWorks() throws Exception {
106    final TableName tableName = TableName.valueOf(name.getMethodName());
107    Table table = testingUtility.createTable(tableName, new byte[][] { FAMILY });
108    // insert some test rows
109    for (int i=0; i<1000; i++) {
110      byte[] iBytes = Bytes.toBytes(i);
111      Put p = new Put(iBytes);
112      p.addColumn(FAMILY, COLUMN, iBytes);
113      table.put(p);
114    }
115    ExecutorService executor = new ScheduledThreadPoolExecutor(1);
116    Canary.RegionStdOutSink sink = spy(new Canary.RegionStdOutSink());
117    Canary canary = new Canary(executor, sink);
118    String[] args = { "-writeSniffing", "-t", "10000", name.getMethodName() };
119    assertEquals(0, ToolRunner.run(testingUtility.getConfiguration(), canary, args));
120    assertEquals("verify no read error count", 0, canary.getReadFailures().size());
121    assertEquals("verify no write error count", 0, canary.getWriteFailures().size());
122    verify(sink, atLeastOnce()).publishReadTiming(isA(ServerName.class), isA(HRegionInfo.class),
123      isA(ColumnFamilyDescriptor.class), anyLong());
124  }

126  @Test
127  @Ignore("Intermittent argument matching failures, see HBASE-18813")
128  public void testReadTableTimeouts() throws Exception {
129    final TableName [] tableNames = new TableName[2];
130    tableNames[0] = TableName.valueOf(name.getMethodName() + "1");
131    tableNames[1] = TableName.valueOf(name.getMethodName() + "2");
132    // Create 2 test tables.
133    for (int j = 0; j<2; j++) {
134      Table table = testingUtility.createTable(tableNames[j], new byte[][] { FAMILY });
135      // insert some test rows
136      for (int i=0; i<1000; i++) {
137        byte[] iBytes = Bytes.toBytes(i + j);
138        Put p = new Put(iBytes);
139        p.addColumn(FAMILY, COLUMN, iBytes);
140        table.put(p);
141      }
142    }
143    ExecutorService executor = new ScheduledThreadPoolExecutor(1);
144    Canary.RegionStdOutSink sink = spy(new Canary.RegionStdOutSink());
145    Canary canary = new Canary(executor, sink);
146    String configuredTimeoutStr = tableNames[0].getNameAsString() + "=" + Long.MAX_VALUE + "," +
147      tableNames[1].getNameAsString() + "=0";
148    String[] args = { "-readTableTimeouts", configuredTimeoutStr, name.getMethodName() + "1", name.getMethodName() + "2"};
149    assertEquals(0, ToolRunner.run(testingUtility.getConfiguration(), canary, args));
150    verify(sink, times(tableNames.length)).initializeAndGetReadLatencyForTable(isA(String.class));
151    for (int i=0; i<2; i++) {
152      assertNotEquals("verify non-null read latency", null, sink.getReadLatencyMap().get(tableNames[i].getNameAsString()));
153      assertNotEquals("verify non-zero read latency", 0L, sink.getReadLatencyMap().get(tableNames[i].getNameAsString()));
154    }
155    // One table's timeout is set for 0 ms and thus, should lead to an error.
156    verify(mockAppender, times(1)).doAppend(argThat(new ArgumentMatcher<LoggingEvent>() {
157      @Override
158      public boolean matches(LoggingEvent argument) {
159        return ((LoggingEvent) argument).getRenderedMessage().contains("exceeded the configured read timeout.");
160      }
161    }));
162    verify(mockAppender, times(2)).doAppend(argThat(new ArgumentMatcher<LoggingEvent>() {
163      @Override
164      public boolean matches(LoggingEvent argument) {
165        return argument.getRenderedMessage().contains("The configured read timeout was");
166      }
167    }));
168  }

215  @Test
216  public void testRawScanConfig() throws Exception {
217    final TableName tableName = TableName.valueOf(name.getMethodName());
218    Table table = testingUtility.createTable(tableName, new byte[][] { FAMILY });
219    // insert some test rows
220    for (int i=0; i<1000; i++) {
221      byte[] iBytes = Bytes.toBytes(i);
222      Put p = new Put(iBytes);
223      p.addColumn(FAMILY, COLUMN, iBytes);
224      table.put(p);
225    }
226    ExecutorService executor = new ScheduledThreadPoolExecutor(1);
227    Canary.RegionStdOutSink sink = spy(new Canary.RegionStdOutSink());
228    Canary canary = new Canary(executor, sink);
229    String[] args = { "-t", "10000", name.getMethodName() };
230    org.apache.hadoop.conf.Configuration conf =
231      new org.apache.hadoop.conf.Configuration(testingUtility.getConfiguration());
232    conf.setBoolean(HConstants.HBASE_CANARY_READ_RAW_SCAN_KEY, true);
233    assertEquals(0, ToolRunner.run(conf, canary, args));
234    verify(sink, atLeastOnce())
235        .publishReadTiming(isA(ServerName.class), isA(HRegionInfo.class),
236        isA(ColumnFamilyDescriptor.class), anyLong());
237    assertEquals("verify no read error count", 0, canary.getReadFailures().size());
238  }
239
240  private void runRegionserverCanary() throws Exception {
241    ExecutorService executor = new ScheduledThreadPoolExecutor(1);
242    Canary canary = new Canary(executor, new Canary.RegionServerStdOutSink());
243    String[] args = { "-t", "10000", "-regionserver"};
244    assertEquals(0, ToolRunner.run(testingUtility.getConfiguration(), canary, args));
245    assertEquals("verify no read error count", 0, canary.getReadFailures().size());
246  }
