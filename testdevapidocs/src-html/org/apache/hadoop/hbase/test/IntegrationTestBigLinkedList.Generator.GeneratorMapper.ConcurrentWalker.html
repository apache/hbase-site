124/**
125 * <p>
126 * This is an integration test borrowed from goraci, written by Keith Turner,
127 * which is in turn inspired by the Accumulo test called continous ingest (ci).
128 * The original source code can be found here:
129 * <ul>
130 *   <li>https://github.com/keith-turner/goraci</li>
131 *   <li>https://github.com/enis/goraci/</li>
132 * </ul>
133 * </p>
134 * <p>
135 * Apache Accumulo [0] has a simple test suite that verifies that data is not
136 * lost at scale. This test suite is called continuous ingest. This test runs
137 * many ingest clients that continually create linked lists containing 25
138 * million nodes. At some point the clients are stopped and a map reduce job is
139 * run to ensure no linked list has a hole. A hole indicates data was lost.
140 * </p>
141 * <p>
142 * The nodes in the linked list are random. This causes each linked list to
143 * spread across the table. Therefore if one part of a table loses data, then it
144 * will be detected by references in another part of the table.
145 * </p>
146 * <p>
147 * <h3>THE ANATOMY OF THE TEST</h3>
148 *
149 * Below is rough sketch of how data is written. For specific details look at
150 * the Generator code.
151 * </p>
152 * <p>
153 * <ol>
154 * <li>Write out 1 million nodes</li>
155 * <li>Flush the client</li>
156 * <li>Write out 1 million that reference previous million</li>
157 * <li>If this is the 25th set of 1 million nodes, then update 1st set of
158 * million to point to last</li>
159 * <li>goto 1</li>
160 * </ol>
161 * </p>
162 * <p>
163 * The key is that nodes only reference flushed nodes. Therefore a node should
164 * never reference a missing node, even if the ingest client is killed at any
165 * point in time.
166 * </p>
167 * <p>
168 * When running this test suite w/ Accumulo there is a script running in
169 * parallel called the Aggitator that randomly and continuously kills server
170 * processes. The outcome was that many data loss bugs were found in Accumulo
171 * by doing this. This test suite can also help find bugs that impact uptime
172 * and stability when run for days or weeks.
173 * </p>
174 * <p>
175 * This test suite consists the following
176 * <ul>
177 * <li>a few Java programs</li>
178 * <li>a little helper script to run the java programs</li>
179 * <li>a maven script to build it</li>
180 * </ul>
181 * </p>
182 * <p>
183 * When generating data, its best to have each map task generate a multiple of
184 * 25 million. The reason for this is that circular linked list are generated
185 * every 25M. Not generating a multiple in 25M will result in some nodes in the
186 * linked list not having references. The loss of an unreferenced node can not
187 * be detected.
188 * </p>
189 * <p>
190 * <h3>Below is a description of the Java programs</h3>
191 * <ul>
192 * <li>
193 * {@code Generator} - A map only job that generates data. As stated previously, its best to
194 * generate data in multiples of 25M. An option is also available to allow concurrent walkers to
195 * select and walk random flushed loops during this phase.
196 * </li>
197 * <li>
198 * {@code Verify} - A map reduce job that looks for holes. Look at the counts after running.
199 * {@code REFERENCED} and {@code UNREFERENCED} are ok, any {@code UNDEFINED} counts are bad. Do not
200 * run at the same time as the Generator.
201 * </li>
202 * <li>
203 * {@code Walker} - A standalone program that start following a linked list and emits timing info.
204 * </li>
205 * <li>
206 * {@code Print} - A standalone program that prints nodes in the linked list
207 * </li>
208 * <li>
209 * {@code Delete} - A standalone program that deletes a single node
210 * </li>
211 * </ul>
212 *
213 * This class can be run as a unit test, as an integration test, or from the command line
214 * </p>
215 * <p>
216 * ex:
217 * <pre>
218 * ./hbase org.apache.hadoop.hbase.test.IntegrationTestBigLinkedList
219 *    loop 2 1 100000 /temp 1 1000 50 1 0
220 * </pre>
221 * </p>
222 */

525      @Override
526      protected void map(BytesWritable key, NullWritable value, Context output) throws IOException {
527        current[i] = new byte[key.getLength()];
528        System.arraycopy(key.getBytes(), 0, current[i], 0, key.getLength());
529        if (++i == current.length) {
530          LOG.debug("Persisting current.length={}, count={}, id={}, current={}, i=",
531            current.length, count, Bytes.toStringBinary(id), Bytes.toStringBinary(current[0]), i);
532          persist(output, count, prev, current, id);
533          i = 0;
534
535          if (first == null) {
536            first = current;
537          }
538          prev = current;
539          current = new byte[this.width][];
540
541          count += current.length;
542          output.setStatus("Count " + count);
543
544          if (count % wrap == 0) {
545            // this block of code turns the 1 million linked list of length 25 into one giant
546            //circular linked list of 25 million
547            circularLeftShift(first);
548            persist(output, -1, prev, first, null);
549            // At this point the entire loop has been flushed so we can add one of its nodes to the
550            // concurrent walker
551            if (numWalkers > 0) {
552              addFlushed(key.getBytes());
553              if (walkers.isEmpty()) {
554                startWalkers(numWalkers, conf, output);
555              }
556            }
557            first = null;
558            prev = null;
559          }
560        }
561      }

563      private static <T> void circularLeftShift(T[] first) {
564        T ez = first[0];
565        System.arraycopy(first, 1, first, 0, first.length - 1);
566        first[first.length - 1] = ez;
567      }

569      private void addFlushed(byte[] rowKey) {
570        synchronized (flushedLoops) {
571          flushedLoops.add(Bytes.toLong(rowKey));
572          flushedLoops.notifyAll();
573        }
574      }

576      protected void persist(Context output, long count, byte[][] prev, byte[][] current, byte[] id)
577          throws IOException {
578        for (int i = 0; i < current.length; i++) {
579
580          if (i % 100 == 0) {
581            // Tickle progress every so often else maprunner will think us hung
582            output.progress();
583          }
584
585          Put put = new Put(current[i]);
586          put.addColumn(FAMILY_NAME, COLUMN_PREV, prev == null ? NO_KEY : prev[i]);
587
588          if (count >= 0) {
589            put.addColumn(FAMILY_NAME, COLUMN_COUNT, Bytes.toBytes(count + i));
590          }
591          if (id != null) {
592            put.addColumn(FAMILY_NAME, COLUMN_CLIENT, id);
593          }
594          // See if we are to write multiple columns.
595          if (this.multipleUnevenColumnFamilies) {
596            // Use any column name.
597            put.addColumn(TINY_FAMILY_NAME, TINY_FAMILY_NAME, this.tinyValue);
598            // Use any column name.
599            put.addColumn(BIG_FAMILY_NAME, BIG_FAMILY_NAME, this.bigValue);
600          }
601          mutator.mutate(put);
602        }
603
604        mutator.flush();
605      }

607      private void startWalkers(int numWalkers, Configuration conf, Context context) {
608        LOG.info("Starting " + numWalkers + " concurrent walkers");
609        for (int i = 0; i < numWalkers; i++) {
610          Thread walker = new Thread(new ContinuousConcurrentWalker(conf, context));
611          walker.start();
612          walkers.add(walker);
613        }
614      }

616      private void joinWalkers() {
617        walkersStop = true;
618        synchronized (flushedLoops) {
619          flushedLoops.notifyAll();
620        }
621        for (Thread walker : walkers) {
622          try {
623            walker.join();

631       * Randomly selects and walks a random flushed loop concurrently with the Generator Mapper by
632       * spawning ConcurrentWalker's with specified StartNodes. These ConcurrentWalker's are
633       * configured to only log erroneous nodes.
634       */

637
638        ConcurrentWalker walker;
639        Configuration conf;
640        Context context;
641        Random rand;
642
643        public ContinuousConcurrentWalker(Configuration conf, Context context) {
644          this.conf = conf;
645          this.context = context;
646          rand = new Random();
647        }
648
649        @Override
650        public void run() {
651          while (!walkersStop) {
652            try {
653              long node = selectLoop();
654              try {
655                walkLoop(node);
656              } catch (IOException e) {
657                context.getCounter(Counts.IOEXCEPTION).increment(1l);
658                return;
659              }
660            } catch (InterruptedException e) {
661              return;
662            }
663          }
664        }

666        private void walkLoop(long node) throws IOException {
667          walker = new ConcurrentWalker(context);
668          walker.setConf(conf);
669          walker.run(node, wrap);
670        }

672        private long selectLoop () throws InterruptedException{
673          synchronized (flushedLoops) {
674            while (flushedLoops.isEmpty() && !walkersStop) {
675              flushedLoops.wait();
676            }
677            if (walkersStop) {
678              throw new InterruptedException();
679            }
680            return flushedLoops.get(rand.nextInt(flushedLoops.size()));
681          }
682        }
683      }

694          byte[] startKey = Bytes.toBytes(startKeyIn);

697          Table table = connection.getTable(getTableName(getConf()));

704          if (node == null) {

706            throw new IOException("Start node not found: " + startKeyIn);
707          }
708          while (numQueries < maxQueries) {
709            numQueries++;
710            byte[] prev = node.prev;
711            long t1 = System.currentTimeMillis();
712            node = getNode(prev, table, node);
713            long t2 = System.currentTimeMillis();
714            if (node == null) {

719                  Bytes.toStringBinary(node.key));

721            } else {
722              // Increment for successful walk
723              context.getCounter(Counts.SUCCESS).increment(1l);
724            }
725          }
726          table.close();

728        }

1234        // TODO check for more than one def, should not happen
1235        StringBuilder refsSb = null;
1236        if (defCount == 0 || refs.size() != 1) {
1237          String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
1238          refsSb = dumpExtraInfoOnRefs(key, context, refs);
1239          LOG.error("LinkedListError: key=" + keyString + ", reference(s)=" +
1240            (refsSb != null? refsSb.toString(): ""));
1241        }
1242        if (lostFamilies) {
1243          String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
1244          LOG.error("LinkedListError: key=" + keyString + ", lost big or tiny families");
1245          context.getCounter(Counts.LOST_FAMILIES).increment(1);
1246          context.write(key, LOSTFAM);
1247        }
1248
1249        if (defCount == 0 && refs.size() > 0) {
1250          // This is bad, found a node that is referenced but not defined. It must have been
1251          // lost, emit some info about this node for debugging purposes.
1252          // Write out a line per reference. If more than one, flag it.;
1253          for (int i = 0; i < refs.size(); i++) {
1254            byte[] bs = refs.get(i);
1255            int ordinal;
1256            if (i <= 0) {
1257              ordinal = Counts.UNDEFINED.ordinal();
1258              context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
1259              context.getCounter(Counts.UNDEFINED).increment(1);
1260            } else {
1261              ordinal = Counts.EXTRA_UNDEF_REFERENCES.ordinal();
1262              context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
1263            }
1264          }
1265          if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
1266            // Print out missing row; doing get on reference gives info on when the referencer
1267            // was added which can help a little debugging. This info is only available in mapper
1268            // output -- the 'Linked List error Key...' log message above. What we emit here is
1269            // useless for debugging.
1270            String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
1271            context.getCounter("undef", keyString).increment(1);
1272          }
1273        } else if (defCount > 0 && refs.isEmpty()) {
1274          // node is defined but not referenced
1275          context.write(key, UNREF);
1276          context.getCounter(Counts.UNREFERENCED).increment(1);
1277          if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
1278            String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
1279            context.getCounter("unref", keyString).increment(1);
1280          }
1281        } else {
1282          if (refs.size() > 1) {
1283            // Skip first reference.
1284            for (int i = 1; i < refs.size(); i++) {
1285              context.write(key,
1286                new BytesWritable(addPrefixFlag(Counts.EXTRAREFERENCES.ordinal(), refs.get(i))));
1287            }
1288            context.getCounter(Counts.EXTRAREFERENCES).increment(refs.size() - 1);
1289          }
1290          // node is defined and referenced
1291          context.getCounter(Counts.REFERENCED).increment(1);
1292        }
1293      }

1340                ", key row regionLocation=" + keyRegionLocation);
1341            refsSb.append(comma);
1342            comma = ",";

1423    public boolean verify(long expectedReferenced) throws Exception {
1424      if (job == null) {
1425        throw new IllegalStateException("You should call run() first");
1426      }
1427
1428      Counters counters = job.getCounters();
1429      if (counters == null) {

1431              + " This is commonly a result of insufficient YARN configuration.");

1433      }

1436      boolean success = verifyExpectedValues(expectedReferenced, counters);

1439        // We found counter objects which imply failure

1449    /**
1450     * Verify the values in the Counters against the expected number of entries written.
1451     *
1452     * @param expectedReferenced
1453     *          Expected number of referenced entrires
1454     * @param counters
1455     *          The Job's Counters object
1456     * @return True if the values match what's expected, false otherwise
1457     */
1458    protected boolean verifyExpectedValues(long expectedReferenced, Counters counters) {
1459      final Counter referenced = counters.findCounter(Counts.REFERENCED);
1460      final Counter unreferenced = counters.findCounter(Counts.UNREFERENCED);
1461      boolean success = true;
1462
1463      if (expectedReferenced != referenced.getValue()) {
1464        LOG.error("Expected referenced count does not match with actual referenced count. " +
1465            "expected referenced=" + expectedReferenced + " ,actual=" + referenced.getValue());
1466        success = false;
1467      }
1468
1469      if (unreferenced.getValue() > 0) {
1470        final Counter multiref = counters.findCounter(Counts.EXTRAREFERENCES);
1471        boolean couldBeMultiRef = (multiref.getValue() == unreferenced.getValue());
1472        LOG.error("Unreferenced nodes were not expected. Unreferenced count=" + unreferenced.getValue()
1473            + (couldBeMultiRef ? "; could be due to duplicate random numbers" : ""));
1474        success = false;
1475      }
1476
1477      return success;
1478    }

1484     *          The Job's counters

1491

1496
1497      if (lostfamilies.getValue() > 0) {
1498        LOG.error("Found nodes which lost big or tiny families, count=" + lostfamilies.getValue());
1499        success = false;
1500      }
1501
1502      return success;
