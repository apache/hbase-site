041package org.apache.hadoop.hbase.test.util.warc;
042
043import java.io.ByteArrayOutputStream;
044import java.io.DataInput;
045import java.io.DataOutput;
046import java.io.IOException;
047import java.util.LinkedHashMap;
048import java.util.Map;
049import java.util.regex.Pattern;
050
051/**
052 * Immutable implementation of a record in a WARC file. You create a {@link WARCRecord}
053 * by parsing it out of a {@link DataInput} stream.
054 *
055 * The file format is documented in the
056 * [ISO Standard](http://bibnum.bnf.fr/warc/WARC_ISO_28500_version1_latestdraft.pdf).
057 * In a nutshell, it's a textual format consisting of lines delimited by `\r\n`.
058 * Each record has the following structure:
059 *
060 * 1. A line indicating the WARC version number, such as `WARC/1.0`.
061 * 2. Several header lines (in key-value format, similar to HTTP or email headers),
062 *    giving information about the record. The header is terminated by an empty line.
063 * 3. A body consisting of raw bytes (the number of bytes is indicated in one of the headers).
064 * 4. A final separator of `\r\n\r\n` before the next record starts.
065 *
066 * There are various different types of records, as documented on
067 * {@link Header#getRecordType()}.
068 */
069public class WARCRecord {
070
071  public static final String WARC_VERSION = "WARC/1.0";
072  private static final Pattern VERSION_PATTERN = Pattern.compile("WARC/[0-9\\.]+");
073  private static final Pattern CONTINUATION_PATTERN = Pattern.compile("^[\\t ]+.*");
074  private static final String CRLF = "\r\n";
075  private static final byte[] CRLF_BYTES = { 13, 10 };
076
077  private final Header header;
078  private final byte[] content;
079
080  /**
081   * Creates a new WARCRecord by parsing it out of a {@link DataInput} stream.
082   * @param in The input source from which one record will be read.
083   */
084  public WARCRecord(DataInput in) throws IOException {
085    header = readHeader(in);
086    content = new byte[header.getContentLength()];
087    in.readFully(content);
088    readSeparator(in);
089  }
090
091  private static Header readHeader(DataInput in) throws IOException {
092    String versionLine = readLine(in);
093    if (!VERSION_PATTERN.matcher(versionLine).matches()) {
094      throw new IllegalStateException("Expected WARC version, but got: " + versionLine);
095    }
096
097    LinkedHashMap<String, String> headers = new LinkedHashMap<String, String>();
098    String line, fieldName = null;
099
100    do {
101      line = readLine(in);
102      if (fieldName != null && CONTINUATION_PATTERN.matcher(line).matches()) {
103        headers.put(fieldName, headers.get(fieldName) + line);
104      } else if (!line.isEmpty()) {
105        String[] field = line.split(":", 2);
106        if (field.length < 2) {
107          throw new IllegalStateException("Malformed header line: " + line);
108        }
109        fieldName = field[0].trim();
110        headers.put(fieldName, field[1].trim());
111      }
112    } while (!line.isEmpty());
113
114    return new Header(headers);
115  }
116
117  private static String readLine(DataInput in) throws IOException {
118    ByteArrayOutputStream out = new ByteArrayOutputStream();
119    boolean seenCR = false, seenCRLF = false;
120    while (!seenCRLF) {
121      byte b = in.readByte();
122      if (!seenCR && b == 13) {
123        seenCR = true;
124      } else if (seenCR && b == 10) {
125        seenCRLF = true;
126      } else {
127        seenCR = false;
128        out.write(b);
129      }
130    }
131    return out.toString("UTF-8");
132  }
133
134  private static void readSeparator(DataInput in) throws IOException {
135    byte[] sep = new byte[4];
136    in.readFully(sep);
137    if (sep[0] != 13 || sep[1] != 10 || sep[2] != 13 || sep[3] != 10) {
138      throw new IllegalStateException(String.format(
139        "Expected final separator CR LF CR LF, but got: %d %d %d %d",
140        sep[0], sep[1], sep[2], sep[3]));
141    }
142  }
143
144  /**
145   * Returns the parsed header structure of the WARC record.
146   */
147  public Header getHeader() {
148    return header;
149  }
150
151  /**
152   * Returns the body of the record, as an unparsed raw array of bytes. The content
153   * of the body depends on the type of record (see {@link Header#getRecordType()}).
154   * For example, in the case of a `response` type header, the body consists of the
155   * full HTTP response returned by the server (HTTP headers followed by the body).
156   */
157  public byte[] getContent() {
158    return content;
159  }
160
161  /**
162   * Writes this record to a {@link DataOutput} stream. The output may, in some edge
163   * cases, be not byte-for-byte identical to what was parsed from a {@link DataInput}.
164   * However it has the same meaning and should not lose any information.
165   * @param out The output stream to which this record should be appended.
166   */
167  public void write(DataOutput out) throws IOException {
168    header.write(out);
169    out.write(CRLF_BYTES);
170    out.write(content);
171    out.write(CRLF_BYTES);
172    out.write(CRLF_BYTES);
173  }
174
175  /**
176   * Returns a human-readable string representation of the record.
177   */
178  @Override
179  public String toString() {
180    return header.toString();
181  }
182
183  /**
184   * Contains the parsed headers of a {@link WARCRecord}. Each record contains a number
185   * of headers in key-value format, where some header keys are standardised, but
186   * nonstandard ones can be added.
187   *
188   * The documentation of the methods in this class is excerpted from the
189   * [WARC 1.0 specification](http://bibnum.bnf.fr/warc/WARC_ISO_28500_version1_latestdraft.pdf).
190   * Please see the specification for more detail.
191   */
192  public final static class Header {
193    private final Map<String, String> fields;
194
195    private Header(Map<String, String> fields) {
196      this.fields = fields;
197    }
198
199    /**
200     * Returns the type of WARC record (the value of the `WARC-Type` header field).
201     * WARC 1.0 defines the following record types: (for full definitions, see the
202     * [spec](http://bibnum.bnf.fr/warc/WARC_ISO_28500_version1_latestdraft.pdf))
203     *
204     *  *  `warcinfo`: Describes the records that follow it, up through end of file,
205     *     end of input, or until next `warcinfo` record. Typically, this appears once and
206     *     at the beginning of a WARC file. For a web archive, it often contains information
207     *     about the web crawl which generated the following records.
208     *
209     *     The format of this descriptive record block may vary, though the use of the
210     *     `"application/warc-fields"` content-type is recommended. (...)
211     *
212     *  *  `response`: The record should contain a complete scheme-specific response, including
213     *     network protocol information where possible. For a target-URI of the `http` or
214     *     `https` schemes, a `response` record block should contain the full HTTP
215     *     response received over the network, including headers. That is, it contains the
216     *     'Response' message defined by section 6 of HTTP/1.1 (RFC2616).
217     *
218     *     The WARC record's Content-Type field should contain the value defined by HTTP/1.1,
219     *     `"application/http;msgtype=response"`. The payload of the record is defined as its
220     *     'entity-body' (per RFC2616), with any transfer-encoding removed.
221     *
222     *  *  `resource`: The record contains a resource, without full protocol response
223     *     information. For example: a file directly retrieved from a locally accessible
224     *     repository or the result of a networked retrieval where the protocol information
225     *     has been discarded. For a target-URI of the `http` or `https` schemes, a `resource`
226     *     record block shall contain the returned 'entity-body' (per RFC2616, with any
227     *     transfer-encodings removed), possibly truncated.
228     *
229     *  *  `request`: The record holds the details of a complete scheme-specific request,
230     *     including network protocol information where possible. For a target-URI of the
231     *     `http` or `https` schemes, a `request` record block should contain the full HTTP
232     *     request sent over the network, including headers. That is, it contains the
233     *     'Request' message defined by section 5 of HTTP/1.1 (RFC2616).
234     *
235     *     The WARC record's Content-Type field should contain the value defined by HTTP/1.1,
236     *     `"application/http;msgtype=request"`. The payload of a `request` record with a
237     *     target-URI of scheme `http` or `https` is defined as its 'entity-body' (per
238     *     RFC2616), with any transfer-encoding removed.
239     *
240     *  *  `metadata`: The record contains content created in order to further describe,
241     *     explain, or accompany a harvested resource, in ways not covered by other record
242     *     types. A `metadata` record will almost always refer to another record of another
243     *     type, with that other record holding original harvested or transformed content.
244     *
245     *     The format of the metadata record block may vary. The `"application/warc-fields"`
246     *     format may be used.
247     *
248     *  *  `revisit`: The record describes the revisitation of content already archived,
249     *     and might include only an abbreviated content body which has to be interpreted
250     *     relative to a previous record. Most typically, a `revisit` record is used
251     *     instead of a `response` or `resource` record to indicate that the content
252     *     visited was either a complete or substantial duplicate of material previously
253     *     archived.
254     *
255     *     A `revisit` record shall contain a WARC-Profile field which determines the
256     *     interpretation of the record's fields and record block. Please see the
257     *     specification for details.
258     *
259     *  *  `conversion`: The record shall contain an alternative version of another
260     *     record's content that was created as the result of an archival process.
261     *     Typically, this is used to hold content transformations that maintain viability
262     *     of content after widely available rendering tools for the originally stored
263     *     format disappear. As needed, the original content may be migrated (transformed)
264     *     to a more viable format in order to keep the information usable with current
265     *     tools while minimizing loss of information.
266     *
267     *  *  `continuation`: Record blocks from `continuation` records must be appended to
268     *     corresponding prior record blocks (eg. from other WARC files) to create the
269     *     logically complete full-sized original record. That is, `continuation`
270     *     records are used when a record that would otherwise cause a WARC file size to
271     *     exceed a desired limit is broken into segments. A continuation record shall
272     *     contain the named fields `WARC-Segment-Origin-ID` and `WARC-Segment-Number`,
273     *     and the last `continuation` record of a series shall contain a
274     *     `WARC-Segment-Total-Length` field. Please see the specification for details.
275     *
276     *  *  Other record types may be added in future, so this list is not exclusive.
277     *
278     * @return The record's `WARC-Type` header field, as a string.
279     */
280    public String getRecordType() {
281      return fields.get("WARC-Type");
282    }
283
284    /**
285     * A 14-digit UTC timestamp formatted according to YYYY-MM-DDThh:mm:ssZ, described
286     * in the W3C profile of ISO8601. The timestamp shall represent the instant that
287     * data capture for record creation began. Multiple records written as part of a
288     * single capture event shall use the same WARC-Date, even though the times of
289     * their writing will not be exactly synchronized.
290     *
291     * @return The record's `WARC-Date` header field, as a string.
292     */
293    public String getDateString() {
294      return fields.get("WARC-Date");
295    }
296
297    /**
298     * An identifier assigned to the current record that is globally unique for its
299     * period of intended use. No identifier scheme is mandated by this specification,
300     * but each record-id shall be a legal URI and clearly indicate a documented and
301     * registered scheme to which it conforms (e.g., via a URI scheme prefix such as
302     * `http:` or `urn:`).
303     *
304     * @return The record's `WARC-Record-ID` header field, as a string.
305     */
306    public String getRecordID() {
307      return fields.get("WARC-Record-ID");
308    }
309
310    /**
311     * The MIME type (RFC2045) of the information contained in the record's block. For
312     * example, in HTTP request and response records, this would be `application/http`
313     * as per section 19.1 of RFC2616 (or `application/http; msgtype=request` and
314     * `application/http; msgtype=response` respectively).
315     *
316     * In particular, the content-type is *not* the value of the HTTP Content-Type
317     * header in an HTTP response, but a MIME type to describe the full archived HTTP
318     * message (hence `application/http` if the block contains request or response
319     * headers).
320     *
321     * @return The record's `Content-Type` header field, as a string.
322     */
323    public String getContentType() {
324      return fields.get("Content-Type");
325    }
326
327    /**
328     * The original URI whose capture gave rise to the information content in this record.
329     * In the context of web harvesting, this is the URI that was the target of a
330     * crawler's retrieval request. For a `revisit` record, it is the URI that was the
331     * target of a retrieval request. Indirectly, such as for a `metadata`, or `conversion`
332     * record, it is a copy of the `WARC-Target-URI` appearing in the original record to
333     * which the newer record pertains. The URI in this value shall be properly escaped
334     * according to RFC3986, and written with no internal whitespace.
335     *
336     * @return The record's `WARC-Target-URI` header field, as a string.
337     */
338    public String getTargetURI() {
339      return fields.get("WARC-Target-URI");
340    }
341
342    /**
343     * The number of bytes in the body of the record, similar to RFC2616.
344     *
345     * @return The record's `Content-Length` header field, parsed into an int.
346     */
347    public int getContentLength() {
348      String lengthStr = fields.get("Content-Length");
349      if (lengthStr == null) {
350        throw new IllegalStateException("Missing Content-Length header");
351      }
352      try {
353        return Integer.parseInt(lengthStr);
354      } catch (NumberFormatException e) {
355        throw new IllegalStateException("Malformed Content-Length header: " + lengthStr);
356      }
357    }
358
359    /**
360     * Returns the value of a selected header field, or null if there is no header with
361     * that field name.
362     * @param field The name of the header to return (case-sensitive).
363     * @return The value associated with that field name, or null if not present.
364     */
365    public String getField(String field) {
366      return fields.get(field);
367    }
368
369    /**
370     * Appends this header to a {@link DataOutput} stream, in WARC/1.0 format.
371     * @param out The data output to which the header should be written.
372     */
373    public void write(DataOutput out) throws IOException {
374      out.write(toString().getBytes("UTF-8"));
375    }
376
377    /**
378     * Formats this header in WARC/1.0 format, consisting of a version line followed
379     * by colon-delimited key-value pairs, and `\r\n` line endings.
380     */
381    @Override
382    public String toString() {
383      StringBuilder buf = new StringBuilder();
384      buf.append(WARC_VERSION);
385      buf.append(CRLF);
386      for (Map.Entry<String, String> field : fields.entrySet()) {
387        buf.append(field.getKey());
388        buf.append(": ");
389        buf.append(field.getValue());
390        buf.append(CRLF);
391      }
392      return buf.toString();
393    }
394  }
