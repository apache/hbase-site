101/**
102 * This integration test loads successful resource retrieval records from the Common Crawl
103 * (https://commoncrawl.org/) public dataset into an HBase table and writes records that can be used
104 * to later verify the presence and integrity of those records.
105 * <p>
106 * Run like: <blockquote> ./bin/hbase org.apache.hadoop.hbase.test.IntegrationTestLoadCommonCrawl
107 * \<br>
108 * &nbsp;&nbsp; -Dfs.s3n.awsAccessKeyId=&lt;AWS access key&gt; \<br>
109 * &nbsp;&nbsp; -Dfs.s3n.awsSecretAccessKey=&lt;AWS secret key&gt; \<br>
110 * &nbsp;&nbsp; /path/to/test-CC-MAIN-2021-10-warc.paths.gz \<br>
111 * &nbsp;&nbsp; /path/to/tmp/warc-loader-output </blockquote>
112 * <p>
113 * Access to the Common Crawl dataset in S3 is made available to anyone by Amazon AWS, but Hadoop's
114 * S3N filesystem still requires valid access credentials to initialize.
115 * <p>
116 * The input path can either specify a directory or a file. The file may optionally be compressed
117 * with gzip. If a directory, the loader expects the directory to contain one or more WARC files
118 * from the Common Crawl dataset. If a file, the loader expects a list of Hadoop S3N URIs which
119 * point to S3 locations for one or more WARC files from the Common Crawl dataset, one URI per line.
120 * Lines should be terminated with the UNIX line terminator.
121 * <p>
122 * Included in hbase-it/src/test/resources/CC-MAIN-2021-10-warc.paths.gz is a list of all WARC files
123 * comprising the Q1 2021 crawl archive. There are 64,000 WARC files in this data set, each
124 * containing ~1GB of gzipped data. The WARC files contain several record types, such as metadata,
125 * request, and response, but we only load the response record types. If the HBase table schema does
126 * not specify compression (by default) there is roughly a 10x expansion. Loading the full crawl
127 * archive results in a table approximately 640 TB in size.
128 * <p>
129 * The loader can optionally drive read load during ingest by incrementing counters for each URL
130 * discovered in content. Add <tt>-DIntegrationTestLoadCommonCrawl.increments=true</tt> to the
131 * command line to enable.
132 * <p>
133 * You can also split the Loader and Verify stages:
134 * <p>
135 * Load with: <blockquote> ./bin/hbase
136 * 'org.apache.hadoop.hbase.test.IntegrationTestLoadCommonCrawl$Loader' \<br>
137 * &nbsp;&nbsp; -files /path/to/hadoop-aws.jar \<br>
138 * &nbsp;&nbsp; -Dfs.s3n.awsAccessKeyId=&lt;AWS access key&gt; \<br>
139 * &nbsp;&nbsp; -Dfs.s3n.awsSecretAccessKey=&lt;AWS secret key&gt; \<br>
140 * &nbsp;&nbsp; /path/to/test-CC-MAIN-2021-10-warc.paths.gz \<br>
141 * &nbsp;&nbsp; /path/to/tmp/warc-loader-output </blockquote>
142 * <p>
143 * Note: The hadoop-aws jar will be needed at runtime to instantiate the S3N filesystem. Use the
144 * <tt>-files</tt> ToolRunner argument to add it.
145 * <p>
146 * Verify with: <blockquote> ./bin/hbase
147 * 'org.apache.hadoop.hbase.test.IntegrationTestLoadCommonCrawl$Verify' \<br>
148 * &nbsp;&nbsp; /path/to/tmp/warc-loader-output </blockquote>
149 * <p>
150 */

446  public static class Loader extends Configured implements Tool {
447
448    private static final Logger LOG = LoggerFactory.getLogger(Loader.class);
449    private static final String USAGE = "Loader <warInputDir | warFileList> <outputDir>";
450
451    void createSchema(final TableName tableName) throws IOException {
452
453      try (Connection conn = ConnectionFactory.createConnection(getConf());
454        Admin admin = conn.getAdmin()) {
455        if (!admin.tableExists(tableName)) {
456
457          ColumnFamilyDescriptorBuilder contentFamilyBuilder =
458            ColumnFamilyDescriptorBuilder.newBuilder(CONTENT_FAMILY_NAME).setMaxVersions(1000)
459              .setDataBlockEncoding(DataBlockEncoding.NONE).setBloomFilterType(BloomType.ROW);
460
461          ColumnFamilyDescriptorBuilder infoFamilyBuilder =
462            ColumnFamilyDescriptorBuilder.newBuilder(INFO_FAMILY_NAME).setMaxVersions(1000)
463              .setDataBlockEncoding(DataBlockEncoding.ROW_INDEX_V1)
464              .setBloomFilterType(BloomType.ROWCOL).setBlocksize(8 * 1024);
465
466          ColumnFamilyDescriptorBuilder urlFamilyBuilder =
467            ColumnFamilyDescriptorBuilder.newBuilder(URL_FAMILY_NAME).setMaxVersions(1000)
468              .setDataBlockEncoding(DataBlockEncoding.ROW_INDEX_V1)
469              .setBloomFilterType(BloomType.ROWCOL).setBlocksize(8 * 1024);
470
471          Set<ColumnFamilyDescriptor> families = new HashSet<>();
472          families.add(contentFamilyBuilder.build());
473          families.add(infoFamilyBuilder.build());
474          families.add(urlFamilyBuilder.build());
475
476          TableDescriptor tableDescriptor =
477            TableDescriptorBuilder.newBuilder(tableName).setColumnFamilies(families).build();
478
479          if (
480            getConf().getBoolean(HBaseTestingUtil.PRESPLIT_TEST_TABLE_KEY,
481              HBaseTestingUtil.PRESPLIT_TEST_TABLE)
482          ) {
483            int numberOfServers = admin.getRegionServers().size();
484            if (numberOfServers == 0) {
485              throw new IllegalStateException("No live regionservers");
486            }
487            int regionsPerServer = getConf().getInt(HBaseTestingUtil.REGIONS_PER_SERVER_KEY,
488              HBaseTestingUtil.DEFAULT_REGIONS_PER_SERVER);
489            int totalNumberOfRegions = numberOfServers * regionsPerServer;
490            LOG.info("Creating test table: " + tableDescriptor);
491            LOG.info("Number of live regionservers: " + numberOfServers + ", "
492              + "pre-splitting table into " + totalNumberOfRegions + " regions "
493              + "(default regions per server: " + regionsPerServer + ")");
494            byte[][] splits = new RegionSplitter.UniformSplit().split(totalNumberOfRegions);
495            admin.createTable(tableDescriptor, splits);
496          } else {
497            LOG.info("Creating test table: " + tableDescriptor);
498            admin.createTable(tableDescriptor);
499          }
500        }
501      } catch (MasterNotRunningException e) {
502        LOG.error("Master not running", e);
503        throw new IOException(e);
504      }
505    }
506
507    int run(final Path warcFileInput, final Path outputDir)
508      throws IOException, ClassNotFoundException, InterruptedException {
509
510      createSchema(getTablename(getConf()));
511
512      final Job job = Job.getInstance(getConf());
513      job.setJobName(Loader.class.getName());
514      job.setNumReduceTasks(0);
515      job.setJarByClass(getClass());
516      job.setMapperClass(LoaderMapper.class);
517      job.setInputFormatClass(WARCInputFormat.class);
518      final FileSystem fs = FileSystem.get(warcFileInput.toUri(), getConf());
519      if (fs.getFileStatus(warcFileInput).isDirectory()) {
520        LOG.info("Using directory as WARC input path: " + warcFileInput);
521        FileInputFormat.setInputPaths(job, warcFileInput);
522      } else if (warcFileInput.toUri().getScheme().equals("file")) {
523        LOG.info("Getting WARC input paths from file: " + warcFileInput);
524        final List<Path> paths = new ArrayList<Path>();
525        try (FSDataInputStream is = fs.open(warcFileInput)) {
526          InputStreamReader reader;
527          if (warcFileInput.getName().toLowerCase().endsWith(".gz")) {
528            reader = new InputStreamReader(new GZIPInputStream(is));
529          } else {
530            reader = new InputStreamReader(is);
531          }
532          try (BufferedReader br = new BufferedReader(reader)) {
533            String line;
534            while ((line = br.readLine()) != null) {
535              paths.add(new Path(line));
536            }
537          }
538        }
539        LOG.info("Read " + paths.size() + " WARC input paths from " + warcFileInput);
540        FileInputFormat.setInputPaths(job, paths.toArray(new Path[paths.size()]));
541      } else {
542        FileInputFormat.setInputPaths(job, warcFileInput);
543      }
544      job.setOutputFormatClass(SequenceFileOutputFormat.class);
545      SequenceFileOutputFormat.setOutputPath(job, outputDir);
546      SequenceFileOutputFormat.setOutputCompressionType(job, CompressionType.BLOCK);
547      job.setOutputKeyClass(HBaseKeyWritable.class);
548      job.setOutputValueClass(BytesWritable.class);
549      TableMapReduceUtil.addDependencyJars(job);
550      // Increase max attempts because S3 might throttle aggressively and ultimately fail a task
551      job.getConfiguration().setInt("mapred.map.max.attempts", 100);
552      job.getConfiguration().setInt("mapreduce.map.maxattempts", 100);
553
554      boolean success = job.waitForCompletion(true);
555      if (!success) {
556        LOG.error("Failure during job " + job.getJobID());
557      }
558
559      final Counters counters = job.getCounters();
560      for (Counts c : Counts.values()) {
561        long value = counters.findCounter(c).getValue();
562        if (value != 0) {
563          LOG.info(c + ": " + value);
564        }
565      }
566
567      return success ? 0 : 1;
568    }
569
570    @Override
571    public int run(String[] args) throws Exception {
572      if (args.length < 2) {
573        System.err.println(USAGE);
574        return 1;
575      }
576      try {
577        Path warcFileInput = new Path(args[0]);
578        Path outputDir = new Path(args[1]);
579        return run(warcFileInput, outputDir);
580      } catch (NumberFormatException e) {
581        System.err.println("Parsing loader arguments failed: " + e.getMessage());
582        System.err.println(USAGE);
583        return 1;
584      }
585    }

591    public static class LoaderMapper
592      extends Mapper<LongWritable, WARCWritable, HBaseKeyWritable, BytesWritable> {
593
594      protected AsyncConnection conn;
595      protected AsyncTable<ScanResultConsumer> table;
596      protected ExecutorService executor;
597      protected AtomicLong inflight = new AtomicLong();
598      protected boolean doIncrements;
599
600      @Override
601      protected void setup(final Context context) throws IOException, InterruptedException {
602        executor = Executors.newWorkStealingPool(Runtime.getRuntime().availableProcessors());
603        Configuration conf = context.getConfiguration();
604        doIncrements = conf.getBoolean(INCREMENTS_NAME_KEY, DEFAULT_INCREMENTS);
605        try {
606          conn = ConnectionFactory.createAsyncConnection(conf).get();
607          table = conn.getTable(getTablename(conf), executor);
608        } catch (ExecutionException e) {
609          throw new IOException(e);
610        }
611      }
612
613      @Override
614      protected void cleanup(final Context context) throws IOException, InterruptedException {
615
616        while (inflight.get() != 0) {
617          LOG.info("Operations in flight, waiting");
618          Thread.sleep(INFLIGHT_PAUSE_MS);
619        }
620
621        // Shut down the executor
622        executor.shutdown();
623        if (!executor.awaitTermination(1, TimeUnit.MINUTES)) {
624          LOG.warn("Pool did not shut down cleanly");
625        }
626        // Close the connection
627        try {
628          conn.close();
629        } catch (Exception e) {
630          LOG.warn("Exception closing Connection", e);
631        }
632      }
633
634      @Override
635      protected void map(final LongWritable key, final WARCWritable value, final Context output)
636        throws IOException, InterruptedException {
637        final WARCRecord.Header warcHeader = value.getRecord().getHeader();
638        final String recordID = warcHeader.getRecordID();
639        final String targetURI = warcHeader.getTargetURI();
640        if (warcHeader.getRecordType().equals("response") && targetURI != null) {
641          final String contentType = warcHeader.getField("WARC-Identified-Payload-Type");
642          if (contentType != null) {
643            // Make row key
644            byte[] rowKey;
645            try {
646              rowKey = rowKeyFromTargetURI(targetURI);
647            } catch (IllegalArgumentException e) {
648              LOG.debug("Could not make a row key for record " + recordID + ", ignoring", e);
649              return;
650            } catch (URISyntaxException e) {
651              LOG.warn(
652                "Could not parse URI \"" + targetURI + "\" for record " + recordID + ", ignoring");
653              return;
654            }
655
656            // Get the content and calculate the CRC64
657            final byte[] content = value.getRecord().getContent();
658            final CRC64 crc = new CRC64();
659            crc.update(content);
660            final long crc64 = crc.getValue();
661            LOG.info("{}: content {} bytes, crc64={}", targetURI, content.length,
662              Bytes.toHex(Bytes.toBytes(crc64)));
663
664            // Store to HBase
665            final long ts = getSequence();
666            final Put put = new Put(rowKey);
667            put.addColumn(CONTENT_FAMILY_NAME, CONTENT_QUALIFIER, ts, content);
668            put.addColumn(INFO_FAMILY_NAME, CONTENT_LENGTH_QUALIFIER, ts,
669              Bytes.toBytes(content.length));
670            put.addColumn(INFO_FAMILY_NAME, CONTENT_TYPE_QUALIFIER, ts, Bytes.toBytes(contentType));
671            put.addColumn(INFO_FAMILY_NAME, CRC_QUALIFIER, ts, Bytes.toBytes(crc64));
672            put.addColumn(INFO_FAMILY_NAME, TARGET_URI_QUALIFIER, ts, Bytes.toBytes(targetURI));
673            put.addColumn(INFO_FAMILY_NAME, DATE_QUALIFIER, ts,
674              Bytes.toBytes(warcHeader.getDateString()));
675            final String ipAddr = warcHeader.getField("WARC-IP-Address");
676            if (ipAddr != null) {
677              put.addColumn(INFO_FAMILY_NAME, IP_ADDRESS_QUALIFIER, ts, Bytes.toBytes(ipAddr));
678            }
679            long pending = inflight.incrementAndGet();
680            while (pending > MAX_INFLIGHT) {
681              LOG.info("Too many operations in flight, waiting");
682              Thread.sleep(INFLIGHT_PAUSE_MS);
683              pending = inflight.get();
684            }
685            final long putStartTime = System.currentTimeMillis();
686            final CompletableFuture<Void> putFuture = table.put(put);
687            putFuture.thenRun(() -> {
688              inflight.decrementAndGet();
689              if (!putFuture.isCompletedExceptionally()) {
690                output.getCounter(Counts.RPC_TIME_MS)
691                  .increment(System.currentTimeMillis() - putStartTime);
692                output.getCounter(Counts.RPC_BYTES_WRITTEN).increment(put.heapSize());
693              }
694            });
695
696            // Write records out for later verification, one per HBase field except for the
697            // content record, which will be verified by CRC64.
698            output.write(
699              new HBaseKeyWritable(rowKey, INFO_FAMILY_NAME, CONTENT_LENGTH_QUALIFIER, ts),
700              new BytesWritable(Bytes.toBytes(content.length)));
701            output.write(new HBaseKeyWritable(rowKey, INFO_FAMILY_NAME, CONTENT_TYPE_QUALIFIER, ts),
702              new BytesWritable(Bytes.toBytes(contentType)));
703            output.write(new HBaseKeyWritable(rowKey, INFO_FAMILY_NAME, CRC_QUALIFIER, ts),
704              new BytesWritable(Bytes.toBytes(crc64)));
705            output.write(new HBaseKeyWritable(rowKey, INFO_FAMILY_NAME, TARGET_URI_QUALIFIER, ts),
706              new BytesWritable(Bytes.toBytes(targetURI)));
707            output.write(new HBaseKeyWritable(rowKey, INFO_FAMILY_NAME, DATE_QUALIFIER, ts),
708              new BytesWritable(Bytes.toBytes(warcHeader.getDateString())));
709            if (ipAddr != null) {
710              output.write(new HBaseKeyWritable(rowKey, INFO_FAMILY_NAME, IP_ADDRESS_QUALIFIER, ts),
711                new BytesWritable(Bytes.toBytes(ipAddr)));
712            }
713
714            if (doIncrements) {
715              // The URLs cf is not tracked for correctness. For now it is used only to exercise
716              // Increments, to drive some read load during ingest. They can be verified with a
717              // reducer to sum increments per row and then compare the final count to the table
718              // data. This is left as a future exercise.
719              final byte[] refQual = Bytes.add(REF_QUALIFIER, SEP, rowKey);
720              for (String refUri : extractUrls(content)) {
721                try {
722                  byte[] urlRowKey = rowKeyFromTargetURI(refUri);
723                  LOG.debug("  -> {}", refUri);
724                  final Increment increment = new Increment(urlRowKey);
725                  increment.setTimestamp(ts);
726                  increment.addColumn(URL_FAMILY_NAME, refQual, 1);
727                  pending = inflight.incrementAndGet();
728                  while (pending > MAX_INFLIGHT) {
729                    LOG.info("Too many operations in flight, waiting");
730                    Thread.sleep(INFLIGHT_PAUSE_MS);
731                    pending = inflight.get();
732                  }
733                  final long incrStartTime = System.currentTimeMillis();
734                  final CompletableFuture<Result> incrFuture = table.increment(increment);
735                  incrFuture.thenRun(() -> {
736                    inflight.decrementAndGet();
737                    if (!incrFuture.isCompletedExceptionally()) {
738                      output.getCounter(Counts.RPC_TIME_MS)
739                        .increment(System.currentTimeMillis() - incrStartTime);
740                      output.getCounter(Counts.RPC_BYTES_WRITTEN).increment(increment.heapSize());
741                    }
742                  });
743                } catch (IllegalArgumentException | URISyntaxException e) {
744                  LOG.debug("Could not make a row key for URI " + refUri + ", ignoring", e);
745                }
746              }
747            }
748          }
749        }
750      }
751    }

774      job.setOutputKeyClass(NullWritable.class);
775      job.setOutputValueClass(NullWritable.class);
776      TableMapReduceUtil.addDependencyJars(job);
777
778      boolean success = job.waitForCompletion(true);
779      if (!success) {
780        LOG.error("Failure during job " + job.getJobID());
781      }
782
783      final Counters counters = job.getCounters();
784      for (Counts c : Counts.values()) {
785        long value = counters.findCounter(c).getValue();
786        if (value != 0) {
787          LOG.info(c + ": " + value);
788        }
789      }
790      if (counters.findCounter(Counts.UNREFERENCED).getValue() > 0) {
791        LOG.error("Nonzero UNREFERENCED count from job " + job.getJobID());
792        success = false;
793      }
794      if (counters.findCounter(Counts.CORRUPT).getValue() > 0) {
795        LOG.error("Nonzero CORRUPT count from job " + job.getJobID());
796        success = false;
797      }
798
799      return success ? 0 : 1;
800    }

816    public static class VerifyMapper
817      extends Mapper<HBaseKeyWritable, BytesWritable, NullWritable, NullWritable> {
818
819      protected Connection conn;
820      protected Table table;
821
822      @Override
823      protected void setup(final Context context) throws IOException, InterruptedException {
824        Configuration conf = context.getConfiguration();
825        conn = ConnectionFactory.createConnection(conf);
826        table = conn.getTable(getTablename(conf));
827      }
828
829      @Override
830      protected void cleanup(final Context context) throws IOException, InterruptedException {
831        // Close the table
832        try {
833          table.close();
834        } catch (Exception e) {
835          LOG.warn("Exception closing table", e);
836        }
837        // Close the connection
838        try {
839          conn.close();
840        } catch (Exception e) {
841          LOG.warn("Exception closing Connection", e);
842        }
843      }
844
845      @Override
846      protected void map(final HBaseKeyWritable key, final BytesWritable value,
847        final Context output) throws IOException, InterruptedException {
848        final byte[] row = Bytes.copy(key.getRowArray(), key.getRowOffset(), key.getRowLength());
849        final byte[] family =
850          Bytes.copy(key.getFamilyArray(), key.getFamilyOffset(), key.getFamilyLength());
851        final byte[] qualifier =
852          Bytes.copy(key.getQualifierArray(), key.getQualifierOffset(), key.getQualifierLength());
853        final long ts = key.getTimestamp();
854
855        if (Bytes.equals(INFO_FAMILY_NAME, family) && Bytes.equals(CRC_QUALIFIER, qualifier)) {
856          final long expectedCRC64 = Bytes.toLong(value.getBytes(), 0, value.getLength());
857          final Get get = new Get(row).setTimestamp(ts).addFamily(CONTENT_FAMILY_NAME)
858            .addFamily(INFO_FAMILY_NAME);
859          final long startTime = System.currentTimeMillis();
860          Result r;
861          try {
862            r = table.get(get);
863            output.getCounter(Counts.RPC_TIME_MS).increment(System.currentTimeMillis() - startTime);
864          } catch (Exception e) {
865            LOG.error("Row " + Bytes.toStringBinary(row) + ": exception", e);
866            output.getCounter(Counts.UNREFERENCED).increment(1);
867            return;
868          }
869          final byte[] crcBytes = r.getValue(INFO_FAMILY_NAME, CRC_QUALIFIER);
870          if (crcBytes == null) {
871            LOG.error("Row " + Bytes.toStringBinary(row) + ": missing i:c");
872            output.getCounter(Counts.UNREFERENCED).increment(1);
873            return;
874          }
875          if (Bytes.toLong(crcBytes) != expectedCRC64) {
876            LOG.error("Row " + Bytes.toStringBinary(row) + ": i:c mismatch");
877            output.getCounter(Counts.CORRUPT).increment(1);
878            return;
879          }
880          // If we fell through to here all verification checks have succeeded for the info
881          // record.
882          output.getCounter(Counts.REFERENCED).increment(1);
883          final byte[] content = r.getValue(CONTENT_FAMILY_NAME, CONTENT_QUALIFIER);
884          if (content == null) {
885            LOG.error("Row " + Bytes.toStringBinary(row) + ": missing content");
886            output.getCounter(Counts.UNREFERENCED).increment(1);
887            return;
888          } else {
889            final CRC64 crc = new CRC64();
890            crc.update(content);
891            if (crc.getValue() != expectedCRC64) {
892              LOG.error("Row " + Bytes.toStringBinary(row) + ": corrupt content");
893              output.getCounter(Counts.CORRUPT).increment(1);
894              return;
895            }
896          }
897          // If we fell through to here all verification checks have succeeded for the content
898          // record.
899          output.getCounter(Counts.REFERENCED).increment(1);
900        } else {
901          final long startTime = System.currentTimeMillis();
902          final Get get = new Get(row).setTimestamp(ts).addColumn(family, qualifier);
903          Result r;
904          try {
905            r = table.get(get);
906            output.getCounter(Counts.RPC_TIME_MS).increment(System.currentTimeMillis() - startTime);
907          } catch (Exception e) {
908            LOG.error("Row " + Bytes.toStringBinary(row) + ": exception", e);
909            output.getCounter(Counts.UNREFERENCED).increment(1);
910            return;
911          }
912          final byte[] bytes = r.getValue(family, qualifier);
913          if (bytes == null) {
914            LOG.error("Row " + Bytes.toStringBinary(row) + ": missing "
915              + Bytes.toStringBinary(family) + ":" + Bytes.toStringBinary(qualifier));
916            output.getCounter(Counts.UNREFERENCED).increment(1);
917            return;
918          }
919          if (!Bytes.equals(bytes, 0, bytes.length, value.getBytes(), 0, value.getLength())) {
920            LOG.error("Row " + Bytes.toStringBinary(row) + ": " + Bytes.toStringBinary(family) + ":"
921              + Bytes.toStringBinary(qualifier) + " mismatch");
922            output.getCounter(Counts.CORRUPT).increment(1);
923            return;
924          }
925          // If we fell through to here all verification checks have succeeded for the info
926          // record.
927          output.getCounter(Counts.REFERENCED).increment(1);
928        }
929      }
930    }

934  private static final int shift = 8;

936  private static long getSequence() {
937    long t = EnvironmentEdgeManager.currentTime();
938    t <<= shift;
939    t |= (counter.getAndIncrement() % (1 << shift));
940    return t;
941  }

943  private static byte[] rowKeyFromTargetURI(final String targetUri)
944    throws IOException, URISyntaxException, IllegalArgumentException {
945    final URI uri = new URI(targetUri);
946    // Ignore the scheme
947    // Reverse the components of the hostname
948    String reversedHost;
949    if (uri.getHost() != null) {
950      final StringBuilder sb = new StringBuilder();
951      final String[] hostComponents = uri.getHost().split("\\.");
952      for (int i = hostComponents.length - 1; i >= 0; i--) {
953        sb.append(hostComponents[i]);
954        if (i != 0) {
955          sb.append('.');
956        }
957      }
958      reversedHost = sb.toString();
959    } else {
960      throw new IllegalArgumentException("URI is missing host component");
961    }
962    final ByteArrayOutputStream os = new ByteArrayOutputStream();
963    os.write(reversedHost.getBytes(StandardCharsets.UTF_8));
964    if (uri.getPort() >= 0) {
965      os.write(String.format(":%d", uri.getPort()).getBytes(StandardCharsets.UTF_8));
966    }
967    os.write((byte) '|');
968    if (uri.getPath() != null) {
969      os.write(uri.getPath().getBytes(StandardCharsets.UTF_8));
970    }
971    if (uri.getQuery() != null) {
972      os.write(String.format("?%s", uri.getQuery()).getBytes(StandardCharsets.UTF_8));
973    }
974    if (uri.getFragment() != null) {
975      os.write(String.format("#%s", uri.getFragment()).getBytes(StandardCharsets.UTF_8));
976    }
977    if (os.size() > HConstants.MAX_ROW_LENGTH) {
978      throw new IllegalArgumentException(
979        "Key would be too large (length=" + os.size() + ", limit=" + HConstants.MAX_ROW_LENGTH);
980    }
981    return os.toByteArray();
982  }

984  static final Pattern URL_PATTERN = Pattern.compile(
985    "\\b((https?|ftp|file)://|(www|ftp)\\.)" + "[\\-A-Z0-9+&@#/%?=~_|$!:,\\.;]*[A-Z0-9+&@#/%=~_|$]",
986    Pattern.CASE_INSENSITIVE);
987
988  private static Collection<String> extractUrls(byte[] content) {
989    final Set<String> list = new HashSet<>(); // uniques
990    final Matcher m = URL_PATTERN.matcher(new String(content, StandardCharsets.UTF_8));
991    while (m.find()) {
992      list.add(m.group());
993    }
994    return list;
995  }
