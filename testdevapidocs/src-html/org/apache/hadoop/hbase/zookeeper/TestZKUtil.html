019package org.apache.hadoop.hbase.zookeeper;
020
021import java.io.IOException;
022import java.util.List;
023
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.HBaseConfiguration;
026import org.apache.hadoop.hbase.HConstants;
027import org.apache.hadoop.hbase.ZooKeeperConnectionException;
028import org.apache.hadoop.hbase.security.Superusers;
029import org.apache.hadoop.hbase.testclassification.SmallTests;
030import org.apache.zookeeper.ZooDefs.Perms;
031import org.apache.zookeeper.data.ACL;
032import org.apache.zookeeper.data.Id;
033import org.junit.Assert;
034import org.junit.Test;
035import org.junit.experimental.categories.Category;
036
037/**
038 *
039 */
040@Category({SmallTests.class})
041public class TestZKUtil {
042
043  @Test
044  public void testGetZooKeeperClusterKey() {
045    Configuration conf = HBaseConfiguration.create();
046    conf.set(HConstants.ZOOKEEPER_QUORUM, "\tlocalhost\n");
047    conf.set(HConstants.ZOOKEEPER_CLIENT_PORT, "3333");
048    conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "hbase");
049    String clusterKey = ZKUtil.getZooKeeperClusterKey(conf, "test");
050    Assert.assertTrue(!clusterKey.contains("\t") && !clusterKey.contains("\n"));
051    Assert.assertEquals("localhost:3333:hbase,test", clusterKey);
052  }
053  
054  @Test
055  public void testCreateACL() throws ZooKeeperConnectionException, IOException {
056    Configuration conf = HBaseConfiguration.create();
057    conf.set(Superusers.SUPERUSER_CONF_KEY, "user1,@group1,user2,@group2,user3");
058    String node = "/hbase/testCreateACL";
059    ZooKeeperWatcher watcher = new ZooKeeperWatcher(conf, node, null, false);
060    List<ACL> aclList = ZKUtil.createACL(watcher, node, true);
061    Assert.assertEquals(aclList.size(), 4); // 3+1, since ACL will be set for the creator by default
062    Assert.assertTrue(!aclList.contains(new ACL(Perms.ALL, new Id("auth", "@group1")))
063        && !aclList.contains(new ACL(Perms.ALL, new Id("auth", "@group2"))));
064    Assert.assertTrue(aclList.contains(new ACL(Perms.ALL, new Id("auth", "user1")))
065        && aclList.contains(new ACL(Perms.ALL, new Id("auth", "user2")))
066        && aclList.contains(new ACL(Perms.ALL, new Id("auth", "user3"))));
067  }
068}
