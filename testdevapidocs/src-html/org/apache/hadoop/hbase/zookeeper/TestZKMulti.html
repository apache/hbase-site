080  @Test (timeout=60000)
081  public void testSimpleMulti() throws Exception {
082    // null multi
083    ZKUtil.multiOrSequential(zkw, null, false);
084
085    // empty multi
086    ZKUtil.multiOrSequential(zkw, new LinkedList<>(), false);
087
088    // single create
089    String path = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testSimpleMulti");
090    LinkedList<ZKUtilOp> singleCreate = new LinkedList<>();
091    singleCreate.add(ZKUtilOp.createAndFailSilent(path, new byte[0]));
092    ZKUtil.multiOrSequential(zkw, singleCreate, false);
093    assertTrue(ZKUtil.checkExists(zkw, path) != -1);
094
095    // single setdata
096    LinkedList<ZKUtilOp> singleSetData = new LinkedList<>();
097    byte [] data = Bytes.toBytes("foobar");
098    singleSetData.add(ZKUtilOp.setData(path, data));
099    ZKUtil.multiOrSequential(zkw, singleSetData, false);
100    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path), data));
101
102    // single delete
103    LinkedList<ZKUtilOp> singleDelete = new LinkedList<>();
104    singleDelete.add(ZKUtilOp.deleteNodeFailSilent(path));
105    ZKUtil.multiOrSequential(zkw, singleDelete, false);
106    assertTrue(ZKUtil.checkExists(zkw, path) == -1);
107  }

109  @Test (timeout=60000)
110  public void testComplexMulti() throws Exception {
111    String path1 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testComplexMulti1");
112    String path2 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testComplexMulti2");
113    String path3 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testComplexMulti3");
114    String path4 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testComplexMulti4");
115    String path5 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testComplexMulti5");
116    String path6 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testComplexMulti6");
117    // create 4 nodes that we'll setData on or delete later
118    LinkedList<ZKUtilOp> create4Nodes = new LinkedList<>();
119    create4Nodes.add(ZKUtilOp.createAndFailSilent(path1, Bytes.toBytes(path1)));
120    create4Nodes.add(ZKUtilOp.createAndFailSilent(path2, Bytes.toBytes(path2)));
121    create4Nodes.add(ZKUtilOp.createAndFailSilent(path3, Bytes.toBytes(path3)));
122    create4Nodes.add(ZKUtilOp.createAndFailSilent(path4, Bytes.toBytes(path4)));
123    ZKUtil.multiOrSequential(zkw, create4Nodes, false);
124    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path1), Bytes.toBytes(path1)));
125    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path2), Bytes.toBytes(path2)));
126    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path3), Bytes.toBytes(path3)));
127    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path4), Bytes.toBytes(path4)));
128
129    // do multiple of each operation (setData, delete, create)
130    LinkedList<ZKUtilOp> ops = new LinkedList<>();
131    // setData
132    ops.add(ZKUtilOp.setData(path1, Bytes.add(Bytes.toBytes(path1), Bytes.toBytes(path1))));
133    ops.add(ZKUtilOp.setData(path2, Bytes.add(Bytes.toBytes(path2), Bytes.toBytes(path2))));
134    // delete
135    ops.add(ZKUtilOp.deleteNodeFailSilent(path3));
136    ops.add(ZKUtilOp.deleteNodeFailSilent(path4));
137    // create
138    ops.add(ZKUtilOp.createAndFailSilent(path5, Bytes.toBytes(path5)));
139    ops.add(ZKUtilOp.createAndFailSilent(path6, Bytes.toBytes(path6)));
140    ZKUtil.multiOrSequential(zkw, ops, false);
141    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path1),
142      Bytes.add(Bytes.toBytes(path1), Bytes.toBytes(path1))));
143    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path2),
144      Bytes.add(Bytes.toBytes(path2), Bytes.toBytes(path2))));
145    assertTrue(ZKUtil.checkExists(zkw, path3) == -1);
146    assertTrue(ZKUtil.checkExists(zkw, path4) == -1);
147    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path5), Bytes.toBytes(path5)));
148    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path6), Bytes.toBytes(path6)));
149  }

151  @Test (timeout=60000)
152  public void testSingleFailure() throws Exception {
153    // try to delete a node that doesn't exist
154    boolean caughtNoNode = false;
155    String path = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testSingleFailureZ");
156    LinkedList<ZKUtilOp> ops = new LinkedList<>();
157    ops.add(ZKUtilOp.deleteNodeFailSilent(path));
158    try {
159      ZKUtil.multiOrSequential(zkw, ops, false);
160    } catch (KeeperException.NoNodeException nne) {
161      caughtNoNode = true;
162    }
163    assertTrue(caughtNoNode);
164
165    // try to setData on a node that doesn't exist
166    caughtNoNode = false;
167    ops = new LinkedList<>();
168    ops.add(ZKUtilOp.setData(path, Bytes.toBytes(path)));
169    try {
170      ZKUtil.multiOrSequential(zkw, ops, false);
171    } catch (KeeperException.NoNodeException nne) {
172      caughtNoNode = true;
173    }
174    assertTrue(caughtNoNode);
175
176    // try to create on a node that already exists
177    boolean caughtNodeExists = false;
178    ops = new LinkedList<>();
179    ops.add(ZKUtilOp.createAndFailSilent(path, Bytes.toBytes(path)));
180    ZKUtil.multiOrSequential(zkw, ops, false);
181    try {
182      ZKUtil.multiOrSequential(zkw, ops, false);
183    } catch (KeeperException.NodeExistsException nee) {
184      caughtNodeExists = true;
185    }
186    assertTrue(caughtNodeExists);
187  }

191    // try a multi where all but one operation succeeds
192    String pathA = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testSingleFailureInMultiA");
193    String pathB = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testSingleFailureInMultiB");

212  @Test (timeout=60000)
213  public void testMultiFailure() throws Exception {
214    String pathX = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testMultiFailureX");
215    String pathY = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testMultiFailureY");
216    String pathZ = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testMultiFailureZ");
217    // create X that we will use to fail create later
218    LinkedList<ZKUtilOp> ops = new LinkedList<>();
219    ops.add(ZKUtilOp.createAndFailSilent(pathX, Bytes.toBytes(pathX)));
220    ZKUtil.multiOrSequential(zkw, ops, false);
221
222    // fail one of each create ,setData, delete
223    String pathV = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testMultiFailureV");
224    String pathW = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "testMultiFailureW");
225    ops = new LinkedList<>();
226    ops.add(ZKUtilOp.createAndFailSilent(pathX, Bytes.toBytes(pathX))); // fail  -- already exists
227    ops.add(ZKUtilOp.setData(pathY, Bytes.toBytes(pathY))); // fail -- doesn't exist
228    ops.add(ZKUtilOp.deleteNodeFailSilent(pathZ)); // fail -- doesn't exist
229    ops.add(ZKUtilOp.createAndFailSilent(pathX, Bytes.toBytes(pathV))); // pass
230    ops.add(ZKUtilOp.createAndFailSilent(pathX, Bytes.toBytes(pathW))); // pass
231    boolean caughtNodeExists = false;
232    try {
233      ZKUtil.multiOrSequential(zkw, ops, false);
234    } catch (KeeperException.NodeExistsException nee) {
235      // check first operation that fails throws exception
236      caughtNodeExists = true;
237    }
238    assertTrue(caughtNodeExists);
239    // check that no modifications were made
240    assertFalse(ZKUtil.checkExists(zkw, pathX) == -1);
241    assertTrue(ZKUtil.checkExists(zkw, pathY) == -1);
242    assertTrue(ZKUtil.checkExists(zkw, pathZ) == -1);
243    assertTrue(ZKUtil.checkExists(zkw, pathW) == -1);
244    assertTrue(ZKUtil.checkExists(zkw, pathV) == -1);
245
246    // test that with multiple failures, throws an exception corresponding to first failure in list
247    ops = new LinkedList<>();
248    ops.add(ZKUtilOp.setData(pathY, Bytes.toBytes(pathY))); // fail -- doesn't exist
249    ops.add(ZKUtilOp.createAndFailSilent(pathX, Bytes.toBytes(pathX))); // fail -- exists
250    boolean caughtNoNode = false;
251    try {
252      ZKUtil.multiOrSequential(zkw, ops, false);
253    } catch (KeeperException.NoNodeException nne) {
254      // check first operation that fails throws exception
255      caughtNoNode = true;
256    }
257    assertTrue(caughtNoNode);
258    // check that no modifications were made
259    assertFalse(ZKUtil.checkExists(zkw, pathX) == -1);
260    assertTrue(ZKUtil.checkExists(zkw, pathY) == -1);
261    assertTrue(ZKUtil.checkExists(zkw, pathZ) == -1);
262    assertTrue(ZKUtil.checkExists(zkw, pathW) == -1);
263    assertTrue(ZKUtil.checkExists(zkw, pathV) == -1);
264  }

266  @Test (timeout=60000)
267  public void testRunSequentialOnMultiFailure() throws Exception {
268    String path1 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "runSequential1");
269    String path2 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "runSequential2");
270    String path3 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "runSequential3");
271    String path4 = ZNodePaths.joinZNode(zkw.znodePaths.baseZNode, "runSequential4");
272
273    // create some nodes that we will use later
274    LinkedList<ZKUtilOp> ops = new LinkedList<>();
275    ops.add(ZKUtilOp.createAndFailSilent(path1, Bytes.toBytes(path1)));
276    ops.add(ZKUtilOp.createAndFailSilent(path2, Bytes.toBytes(path2)));
277    ZKUtil.multiOrSequential(zkw, ops, false);
278
279    // test that, even with operations that fail, the ones that would pass will pass
280    // with runSequentialOnMultiFailure
281    ops = new LinkedList<>();
282    ops.add(ZKUtilOp.setData(path1, Bytes.add(Bytes.toBytes(path1), Bytes.toBytes(path1)))); // pass
283    ops.add(ZKUtilOp.deleteNodeFailSilent(path2)); // pass
284    ops.add(ZKUtilOp.deleteNodeFailSilent(path3)); // fail -- node doesn't exist
285    ops.add(ZKUtilOp.createAndFailSilent(path4,
286      Bytes.add(Bytes.toBytes(path4), Bytes.toBytes(path4)))); // pass
287    ZKUtil.multiOrSequential(zkw, ops, true);
288    assertTrue(Bytes.equals(ZKUtil.getData(zkw, path1),
289      Bytes.add(Bytes.toBytes(path1), Bytes.toBytes(path1))));
290    assertTrue(ZKUtil.checkExists(zkw, path2) == -1);
291    assertTrue(ZKUtil.checkExists(zkw, path3) == -1);
292    assertFalse(ZKUtil.checkExists(zkw, path4) == -1);
293  }

367  private void createZNodeTree(String rootZNode) throws KeeperException,
368      InterruptedException {
369    List<Op> opList = new ArrayList<>();
370    opList.add(Op.create(rootZNode, new byte[0], Ids.OPEN_ACL_UNSAFE,
371        CreateMode.PERSISTENT));
372    int level = 0;
373    String parentZNode = rootZNode;
374    while (level < 10) {
375      // define parent node
376      parentZNode = parentZNode + "/" + level;
377      opList.add(Op.create(parentZNode, new byte[0], Ids.OPEN_ACL_UNSAFE,
378          CreateMode.PERSISTENT));
379      int elements = 0;
380      // add elements to the parent node
381      while (elements < level) {
382        opList.add(Op.create(parentZNode + "/" + elements, new byte[0],
383            Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT));
384        elements++;
385      }
386      level++;
387    }
388    zkw.getRecoverableZooKeeper().multi(opList);
389  }
