103public void assertTColumnValuesEqual(List<TColumnValue> columnValuesA,
104    List<TColumnValue> columnValuesB) {
105  assertEquals(columnValuesA.size(), columnValuesB.size());
106  Comparator<TColumnValue> comparator = new Comparator<TColumnValue>() {
107    @Override
108    public int compare(TColumnValue o1, TColumnValue o2) {
109      return Bytes.compareTo(Bytes.add(o1.getFamily(), o1.getQualifier()),
110          Bytes.add(o2.getFamily(), o2.getQualifier()));
111    }
112  };
113  Collections.sort(columnValuesA, comparator);
114  Collections.sort(columnValuesB, comparator);
115
116  for (int i = 0; i < columnValuesA.size(); i++) {
117    TColumnValue a = columnValuesA.get(i);
118    TColumnValue b = columnValuesB.get(i);
119    assertArrayEquals(a.getFamily(), b.getFamily());
120    assertArrayEquals(a.getQualifier(), b.getQualifier());
121    assertArrayEquals(a.getValue(), b.getValue());
122  }
123}

153private static void createLabels() throws IOException, InterruptedException {
154  PrivilegedExceptionAction<VisibilityLabelsResponse> action =
155      new PrivilegedExceptionAction<VisibilityLabelsResponse>() {
156    public VisibilityLabelsResponse run() throws Exception {
157      String[] labels = { SECRET, CONFIDENTIAL, PRIVATE, PUBLIC, TOPSECRET };
158      try (Connection conn = ConnectionFactory.createConnection(conf)) {
159        VisibilityClient.addLabels(conn, labels);
160      } catch (Throwable t) {
161        throw new IOException(t);
162      }
163      return null;
164    }
165  };
166  SUPERUSER.runAs(action);
167}

173  } catch (Throwable t) {

192@Test
193public void testScanWithVisibilityLabels() throws Exception {
194  ThriftHBaseServiceHandler handler = createHandler();
195  ByteBuffer table = wrap(tableAname);
196
197  // insert data
198  TColumnValue columnValue = new TColumnValue(wrap(familyAname),
199      wrap(qualifierAname), wrap(valueAname));
200  List<TColumnValue> columnValues = new ArrayList<TColumnValue>(1);
201  columnValues.add(columnValue);
202  for (int i = 0; i < 10; i++) {
203    TPut put = new TPut(wrap(("testScan" + i).getBytes()), columnValues);
204    if (i == 5) {
205      put.setCellVisibility(new TCellVisibility().setExpression(PUBLIC));
206    } else {
207      put.setCellVisibility(new TCellVisibility().setExpression("(" + SECRET
208          + "|" + CONFIDENTIAL + ")" + "&" + "!" + TOPSECRET));
209    }
210    handler.put(table, put);
211  }
212
213  // create scan instance
214  TScan scan = new TScan();
215  List<TColumn> columns = new ArrayList<TColumn>(1);
216  TColumn column = new TColumn();
217  column.setFamily(familyAname);
218  column.setQualifier(qualifierAname);
219  columns.add(column);
220  scan.setColumns(columns);
221  scan.setStartRow("testScan".getBytes());
222  scan.setStopRow("testScan\uffff".getBytes());
223
224  TAuthorization tauth = new TAuthorization();
225  List<String> labels = new ArrayList<String>(2);
226  labels.add(SECRET);
227  labels.add(PRIVATE);
228  tauth.setLabels(labels);
229  scan.setAuthorizations(tauth);
230  // get scanner and rows
231  int scanId = handler.openScanner(table, scan);
232  List<TResult> results = handler.getScannerRows(scanId, 10);
233  assertEquals(9, results.size());
234  Assert.assertFalse(Bytes.equals(results.get(5).getRow(),
235      ("testScan" + 5).getBytes()));
236  for (int i = 0; i < 9; i++) {
237    if (i < 5) {
238      assertArrayEquals(("testScan" + i).getBytes(), results.get(i).getRow());
239    } else if (i == 5) {
240      continue;
241    } else {
242      assertArrayEquals(("testScan" + (i + 1)).getBytes(), results.get(i)
243          .getRow());
244    }
245  }
246
247  // check that we are at the end of the scan
248  results = handler.getScannerRows(scanId, 9);
249  assertEquals(0, results.size());
250
251  // close scanner and check that it was indeed closed
252  handler.closeScanner(scanId);
253  try {
254    handler.getScannerRows(scanId, 9);
255    fail("Scanner id should be invalid");
256  } catch (TIllegalArgument e) {
257  }
258}

450/**
451 * Padding numbers to make comparison of sort order easier in a for loop
452 * 
453 * @param n
454 *          The number to pad.
455 * @param pad
456 *          The length to pad up to.
457 * @return The padded number as a string.
458 */
459private String pad(int n, byte pad) {
460  String res = Integer.toString(n);
461  while (res.length() < pad)
462    res = "0" + res;
463  return res;
464}
