114  public void assertTColumnValuesEqual(List<TColumnValue> columnValuesA,
115      List<TColumnValue> columnValuesB) {
116    assertEquals(columnValuesA.size(), columnValuesB.size());
117    Comparator<TColumnValue> comparator = new Comparator<TColumnValue>() {
118      @Override
119      public int compare(TColumnValue o1, TColumnValue o2) {
120        return Bytes.compareTo(Bytes.add(o1.getFamily(), o1.getQualifier()),
121            Bytes.add(o2.getFamily(), o2.getQualifier()));
122      }
123    };
124    Collections.sort(columnValuesA, comparator);
125    Collections.sort(columnValuesB, comparator);
126
127    for (int i = 0; i < columnValuesA.size(); i++) {
128      TColumnValue a = columnValuesA.get(i);
129      TColumnValue b = columnValuesB.get(i);
130      assertTColumnValueEqual(a, b);
131    }
132  }
133
134  public void assertTColumnValueEqual(TColumnValue a, TColumnValue b) {
135    assertArrayEquals(a.getFamily(), b.getFamily());
136    assertArrayEquals(a.getQualifier(), b.getQualifier());
137    assertArrayEquals(a.getValue(), b.getValue());
138  }

544  @Test
545  public void testScan() throws Exception {
546    ThriftHBaseServiceHandler handler = createHandler();
547    ByteBuffer table = wrap(tableAname);
548
549    // insert data
550    TColumnValue columnValue = new TColumnValue(wrap(familyAname), wrap(qualifierAname),
551      wrap(valueAname));
552    List<TColumnValue> columnValues = new ArrayList<TColumnValue>(1);
553    columnValues.add(columnValue);
554    for (int i = 0; i < 10; i++) {
555      TPut put = new TPut(wrap(("testScan" + i).getBytes()), columnValues);
556      handler.put(table, put);
557    }
558
559    // create scan instance
560    TScan scan = new TScan();
561    List<TColumn> columns = new ArrayList<TColumn>(1);
562    TColumn column = new TColumn();
563    column.setFamily(familyAname);
564    column.setQualifier(qualifierAname);
565    columns.add(column);
566    scan.setColumns(columns);
567    scan.setStartRow("testScan".getBytes());
568    scan.setStopRow("testScan\uffff".getBytes());
569
570    // get scanner and rows
571    int scanId = handler.openScanner(table, scan);
572    List<TResult> results = handler.getScannerRows(scanId, 10);
573    assertEquals(10, results.size());
574    for (int i = 0; i < 10; i++) {
575      // check if the rows are returned and in order
576      assertArrayEquals(("testScan" + i).getBytes(), results.get(i).getRow());
577    }
578
579    // check that we are at the end of the scan
580    results = handler.getScannerRows(scanId, 10);
581    assertEquals(0, results.size());
582
583    // close scanner and check that it was indeed closed
584    handler.closeScanner(scanId);
585    try {
586      handler.getScannerRows(scanId, 10);
587      fail("Scanner id should be invalid");
588    } catch (TIllegalArgument e) {
589    }
590  }

797   * @param n  The number to pad.
798   * @param pad  The length to pad up to.
