018package org.apache.hadoop.hbase.codec.prefixtree.row.data;
019
020import java.util.ArrayList;
021import java.util.List;
022
023import org.apache.hadoop.hbase.KeyValue;
024import org.apache.hadoop.hbase.KeyValueUtil;
025import org.apache.hadoop.hbase.Tag;
026import org.apache.hadoop.hbase.ArrayBackedTag;
027import org.apache.hadoop.hbase.codec.prefixtree.PrefixTreeBlockMeta;
028import org.apache.hadoop.hbase.codec.prefixtree.row.BaseTestRowData;
029import org.apache.hadoop.hbase.codec.prefixtree.scanner.CellScannerPosition;
030import org.apache.hadoop.hbase.codec.prefixtree.scanner.CellSearcher;
031import org.apache.hadoop.hbase.util.Bytes;
032import org.junit.Assert;
033
034import org.apache.hadoop.hbase.shaded.com.google.common.collect.Lists;
035
036public class TestRowDataTrivialWithTags extends BaseTestRowData{
037  static byte[] rA = Bytes.toBytes("rA"), rB = Bytes.toBytes("rB"),// turn "r"
038                                                                   // into a
039                                                                   // branch for
040                                                                   // the
041                                                                   // Searcher
042                                                                   // tests
043      cf = Bytes.toBytes("fam"), cq0 = Bytes.toBytes("q0"), v0 = Bytes.toBytes("v0");
044
045  static long ts = 55L;
046
047  static List<KeyValue> d = Lists.newArrayList();
048  static {
049    List<Tag> tagList = new ArrayList<>(2);
050    Tag t = new ArrayBackedTag((byte) 1, "visisbility");
051    tagList.add(t);
052    t = new ArrayBackedTag((byte) 2, "ACL");
053    tagList.add(t);
054    d.add(new KeyValue(rA, cf, cq0, ts, v0, tagList));
055    d.add(new KeyValue(rB, cf, cq0, ts, v0, tagList));
056  }
057
058  @Override
059  public List<KeyValue> getInputs() {
060    return d;
061  }
062
063  @Override
064  public void individualBlockMetaAssertions(PrefixTreeBlockMeta blockMeta) {
065    // node[0] -> root[r]
066    // node[1] -> leaf[A], etc
067    Assert.assertEquals(2, blockMeta.getRowTreeDepth());
068  }
069
070  @Override
071  public void individualSearcherAssertions(CellSearcher searcher) {
072    /**
073     * The searcher should get a token mismatch on the "r" branch. Assert that
074     * it skips not only rA, but rB as well.
075     */
076    KeyValue afterLast = KeyValueUtil.createFirstOnRow(Bytes.toBytes("zzz"));
077    CellScannerPosition position = searcher.positionAtOrAfter(afterLast);
078    Assert.assertEquals(CellScannerPosition.AFTER_LAST, position);
079    Assert.assertNull(searcher.current());
080  }
081}
