053/**
054 * Integration test that verifies lazy CF loading during scans by doing repeated scans
055 * with this feature while multiple threads are continuously writing values; and
056 * verifying the result.
057 */
058@Category(IntegrationTests.class)
059public class IntegrationTestLazyCfLoading {
060  private static final TableName TABLE_NAME =
061      TableName.valueOf(IntegrationTestLazyCfLoading.class.getSimpleName());
062  private static final String TIMEOUT_KEY = "hbase.%s.timeout";
063  private static final String ENCODING_KEY = "hbase.%s.datablock.encoding";
064
065  /** A soft test timeout; duration of the test, as such, depends on number of keys to put. */
066  private static final int DEFAULT_TIMEOUT_MINUTES = 10;
067
068  private static final int NUM_SERVERS = 1;
069  /** Set regions per server low to ensure splits happen during test */
070  private static final int REGIONS_PER_SERVER = 3;
071  private static final int KEYS_TO_WRITE_PER_SERVER = 20000;
072  private static final int WRITER_THREADS = 10;
073  private static final int WAIT_BETWEEN_SCANS_MS = 1000;
074
075  private static final Log LOG = LogFactory.getLog(IntegrationTestLazyCfLoading.class);
076  private IntegrationTestingUtility util = new IntegrationTestingUtility();
077  private final DataGenerator dataGen = new DataGenerator();
078
079  /** Custom LoadTestDataGenerator. Uses key generation and verification from
080   * LoadTestKVGenerator. Creates 3 column families; one with an integer column to
081   * filter on, the 2nd one with an integer column that matches the first integer column (for
082   * test-specific verification), and byte[] value that is used for general verification; and
083   * the third one with just the value.
084   */
085  private static class DataGenerator extends LoadTestDataGenerator {
086    private static final int MIN_DATA_SIZE = 4096;
087    private static final int MAX_DATA_SIZE = 65536;
088    public static final byte[] ESSENTIAL_CF = Bytes.toBytes("essential");
089    public static final byte[] JOINED_CF1 = Bytes.toBytes("joined");
090    public static final byte[] JOINED_CF2 = Bytes.toBytes("joined2");
091    public static final byte[] FILTER_COLUMN = Bytes.toBytes("filter");
092    public static final byte[] VALUE_COLUMN = Bytes.toBytes("val");
093    public static final long ACCEPTED_VALUE = 1L;
094
095    private static final Map<byte[], byte[][]> columnMap = new TreeMap<byte[], byte[][]>(
096        Bytes.BYTES_COMPARATOR);
097
098    private final AtomicLong expectedNumberOfKeys = new AtomicLong(0);
099    private final AtomicLong totalNumberOfKeys = new AtomicLong(0);
100
101    public DataGenerator() {
102      super(MIN_DATA_SIZE, MAX_DATA_SIZE);
103      columnMap.put(ESSENTIAL_CF, new byte[][] { FILTER_COLUMN });
104      columnMap.put(JOINED_CF1, new byte[][] { FILTER_COLUMN, VALUE_COLUMN });
105      columnMap.put(JOINED_CF2, new byte[][] { VALUE_COLUMN });
106    }
107
108    public long getExpectedNumberOfKeys() {
109      return expectedNumberOfKeys.get();
110    }
111
112    public long getTotalNumberOfKeys() {
113      return totalNumberOfKeys.get();
114    }
115
116    @Override
117    public byte[] getDeterministicUniqueKey(long keyBase) {
118      return LoadTestKVGenerator.md5PrefixedKey(keyBase).getBytes();
119    }
120
121    @Override
122    public byte[][] getColumnFamilies() {
123      return columnMap.keySet().toArray(new byte[columnMap.size()][]);
124    }
125
126    @Override
127    public byte[][] generateColumnsForCf(byte[] rowKey, byte[] cf) {
128      return columnMap.get(cf);
129    }
130
131    @Override
132    public byte[] generateValue(byte[] rowKey, byte[] cf, byte[] column) {
133      if (Bytes.BYTES_COMPARATOR.compare(column, FILTER_COLUMN) == 0) {
134        // Random deterministic way to make some values "on" and others "off" for filters.
135        long value = Long.parseLong(Bytes.toString(rowKey, 0, 4), 16) & ACCEPTED_VALUE;
136        if (Bytes.BYTES_COMPARATOR.compare(cf, ESSENTIAL_CF) == 0) {
137          totalNumberOfKeys.incrementAndGet();
138          if (value == ACCEPTED_VALUE) {
139            expectedNumberOfKeys.incrementAndGet();
140          }
141        }
142        return Bytes.toBytes(value);
143      } else if (Bytes.BYTES_COMPARATOR.compare(column, VALUE_COLUMN) == 0) {
144        return kvGenerator.generateRandomSizeValue(rowKey, cf, column);
145      }
146      String error = "Unknown column " + Bytes.toString(column);
147      assert false : error;
148      throw new InvalidParameterException(error);
149    }
150
151    @Override
152    public boolean verify(byte[] rowKey, byte[] cf, byte[] column, byte[] value) {
153      if (Bytes.BYTES_COMPARATOR.compare(column, FILTER_COLUMN) == 0) {
154        // Relies on the filter from getScanFilter being used.
155        return Bytes.toLong(value) == ACCEPTED_VALUE;
156      } else if (Bytes.BYTES_COMPARATOR.compare(column, VALUE_COLUMN) == 0) {
157        return LoadTestKVGenerator.verify(value, rowKey, cf, column);
158      }
159      return false; // some bogus value from read, we don't expect any such thing.
160    }
161
162    @Override
163    public boolean verify(byte[] rowKey, byte[] cf, Set<byte[]> columnSet) {
164      return columnMap.get(cf).length == columnSet.size();
165    }
166
167    public Filter getScanFilter() {
168      SingleColumnValueFilter scf = new SingleColumnValueFilter(ESSENTIAL_CF, FILTER_COLUMN,
169          CompareFilter.CompareOp.EQUAL, Bytes.toBytes(ACCEPTED_VALUE));
170      scf.setFilterIfMissing(true);
171      return scf;
172    }
173  }
174
175  @Before
176  public void setUp() throws Exception {
177    LOG.info("Initializing cluster with " + NUM_SERVERS + " servers");
178    util.initializeCluster(NUM_SERVERS);
179    LOG.info("Done initializing cluster");
180    createTable();
181    // after table creation, ACLs need time to be propagated to RSs in a secure deployment
182    // so we sleep a little bit because we don't have a good way to know when permissions
183    // are received by RSs
184    Thread.sleep(3000);
185  }
186
187  private void createTable() throws Exception {
188    deleteTable();
189    LOG.info("Creating table");
190    Configuration conf = util.getConfiguration();
191    String encodingKey = String.format(ENCODING_KEY, this.getClass().getSimpleName());
192    DataBlockEncoding blockEncoding = DataBlockEncoding.valueOf(conf.get(encodingKey, "FAST_DIFF"));
193    HTableDescriptor htd = new HTableDescriptor(TABLE_NAME);
194    for (byte[] cf : dataGen.getColumnFamilies()) {
195      HColumnDescriptor hcd = new HColumnDescriptor(cf);
196      hcd.setDataBlockEncoding(blockEncoding);
197      htd.addFamily(hcd);
198    }
199    int serverCount = util.getHBaseClusterInterface().getClusterStatus().getServersSize();
200    byte[][] splits = new RegionSplitter.HexStringSplit().split(serverCount * REGIONS_PER_SERVER);
201    util.getHBaseAdmin().createTable(htd, splits);
202    LOG.info("Created table");
203  }
204
205  private void deleteTable() throws Exception {
206    if (util.getHBaseAdmin().tableExists(TABLE_NAME)) {
207      LOG.info("Deleting table");
208      util.deleteTable(TABLE_NAME);
209      LOG.info("Deleted table");
210    }
211  }
212
213  @After
214  public void tearDown() throws Exception {
215    deleteTable();
216    LOG.info("Restoring the cluster");
217    util.restoreCluster();
218    LOG.info("Done restoring the cluster");
219  }

221  @Test
222  public void testReadersAndWriters() throws Exception {
223    Configuration conf = util.getConfiguration();
224    String timeoutKey = String.format(TIMEOUT_KEY, this.getClass().getSimpleName());
225    long maxRuntime = conf.getLong(timeoutKey, DEFAULT_TIMEOUT_MINUTES);
226    long serverCount = util.getHBaseClusterInterface().getClusterStatus().getServersSize();
227    long keysToWrite = serverCount * KEYS_TO_WRITE_PER_SERVER;
228    Connection connection = ConnectionFactory.createConnection(conf);
229    Table table = connection.getTable(TABLE_NAME);
230
231    // Create multi-threaded writer and start it. We write multiple columns/CFs and verify
232    // their integrity, therefore multi-put is necessary.
233    MultiThreadedWriter writer =
234      new MultiThreadedWriter(dataGen, conf, TABLE_NAME);
235    writer.setMultiPut(true);
236
237    LOG.info("Starting writer; the number of keys to write is " + keysToWrite);
238    // TODO : Need to see if tag support has to be given here in the integration test suite
239    writer.start(1, keysToWrite, WRITER_THREADS);
240
241    // Now, do scans.
242    long now = EnvironmentEdgeManager.currentTime();
243    long timeLimit = now + (maxRuntime * 60000);
244    boolean isWriterDone = false;
245    while (now < timeLimit && !isWriterDone) {
246      LOG.info("Starting the scan; wrote approximately "
247        + dataGen.getTotalNumberOfKeys() + " keys");
248      isWriterDone = writer.isDone();
249      if (isWriterDone) {
250        LOG.info("Scanning full result, writer is done");
251      }
252      Scan scan = new Scan();
253      for (byte[] cf : dataGen.getColumnFamilies()) {
254        scan.addFamily(cf);
255      }
256      scan.setFilter(dataGen.getScanFilter());
257      scan.setLoadColumnFamiliesOnDemand(true);
258      // The number of keys we can expect from scan - lower bound (before scan).
259      // Not a strict lower bound - writer knows nothing about filters, so we report
260      // this from generator. Writer might have generated the value but not put it yet.
261      long onesGennedBeforeScan = dataGen.getExpectedNumberOfKeys();
262      long startTs = EnvironmentEdgeManager.currentTime();
263      ResultScanner results = table.getScanner(scan);
264      long resultCount = 0;
265      Result result = null;
266      // Verify and count the results.
267      while ((result = results.next()) != null) {
268        boolean isOk = writer.verifyResultAgainstDataGenerator(result, true, true);
269        Assert.assertTrue("Failed to verify [" + Bytes.toString(result.getRow())+ "]", isOk);
270        ++resultCount;
271      }
272      long timeTaken = EnvironmentEdgeManager.currentTime() - startTs;
273      // Verify the result count.
274      long onesGennedAfterScan = dataGen.getExpectedNumberOfKeys();
275      Assert.assertTrue("Read " + resultCount + " keys when at most " + onesGennedAfterScan
276        + " were generated ", onesGennedAfterScan >= resultCount);
277      if (isWriterDone) {
278        Assert.assertTrue("Read " + resultCount + " keys; the writer is done and "
279          + onesGennedAfterScan + " keys were generated", onesGennedAfterScan == resultCount);
280      } else if (onesGennedBeforeScan * 0.9 > resultCount) {
281        LOG.warn("Read way too few keys (" + resultCount + "/" + onesGennedBeforeScan
282          + ") - there might be a problem, or the writer might just be slow");
283      }
284      LOG.info("Scan took " + timeTaken + "ms");
285      if (!isWriterDone) {
286        Thread.sleep(WAIT_BETWEEN_SCANS_MS);
287        now = EnvironmentEdgeManager.currentTime();
288      }
289    }
290    Assert.assertEquals("There are write failures", 0, writer.getNumWriteFailures());
291    Assert.assertTrue("Writer is not done", isWriterDone);
292    // Assert.fail("Boom!");
293    connection.close();
294  }
