085  static Table createTestTable(TableName name, byte[][] rows, byte[][] families,
086      byte[][] qualifiers, byte[] cellValue) throws IOException {
087    Table ht = TEST_UTIL.createTable(name, families);
088    List<Put> puts = createPuts(rows, families, qualifiers, cellValue);
089    ht.put(puts);
090
091    return ht;
092  }

099  /**
100   * Make puts to put the input value into each combination of row, family, and qualifier
101   * @param rows
102   * @param families
103   * @param qualifiers
104   * @param value
105   * @return
106   * @throws IOException
107   */
108  static ArrayList<Put> createPuts(byte[][] rows, byte[][] families, byte[][] qualifiers,
109      byte[] value) throws IOException {
110    Put put;
111    ArrayList<Put> puts = new ArrayList<>();
112
113    for (int row = 0; row < rows.length; row++) {
114      put = new Put(rows[row]);
115      for (int fam = 0; fam < families.length; fam++) {
116        for (int qual = 0; qual < qualifiers.length; qual++) {
117          KeyValue kv = new KeyValue(rows[row], families[fam], qualifiers[qual], qual, value);
118          put.add(kv);
119        }
120      }
121      puts.add(put);
122    }
123
124    return puts;
125  }

127  /**
128   * @return The approximate heap size of a cell in the test table. All cells should have
129   *         approximately the same heap size, so the value is cached to avoid repeating the
130   *         calculation
131   * @throws Exception
132   */
133  private long getCellHeapSize() throws Exception {
134    if (CELL_HEAP_SIZE == -1) {
135      // Do a partial scan that will return a single result with a single cell
136      Scan scan = new Scan();
137      scan.setMaxResultSize(1);
138      scan.setAllowPartialResults(true);
139      ResultScanner scanner = TABLE.getScanner(scan);
140
141      Result result = scanner.next();
142
143      assertTrue(result != null);
144      assertTrue(result.rawCells() != null);
145      assertTrue(result.rawCells().length == 1);
146
147      CELL_HEAP_SIZE = PrivateCellUtil.estimatedHeapSizeOf(result.rawCells()[0]);
148      scanner.close();
149    }
150
151    return CELL_HEAP_SIZE;
152  }

164  private void testRowsSeenMetric(boolean async) throws Exception {
165    // Base scan configuration
166    Scan baseScan;
167    baseScan = new Scan();
168    baseScan.setScanMetricsEnabled(true);
169    baseScan.setAsyncPrefetch(async);
170    testRowsSeenMetric(baseScan);
171
172    // Test case that only a single result will be returned per RPC to the serer
173    baseScan.setCaching(1);
174    testRowsSeenMetric(baseScan);
175
176    // Test case that partial results are returned from the server. At most one cell will be
177    // contained in each response
178    baseScan.setMaxResultSize(1);
179    testRowsSeenMetric(baseScan);
180
181    // Test case that size limit is set such that a few cells are returned per partial result from
182    // the server
183    baseScan.setCaching(NUM_ROWS);
184    baseScan.setMaxResultSize(getCellHeapSize() * (NUM_COLS - 1));
185    testRowsSeenMetric(baseScan);
186  }

188  public void testRowsSeenMetric(Scan baseScan) throws Exception {
189    Scan scan;
190    scan = new Scan(baseScan);
191    testMetric(scan, ServerSideScanMetrics.COUNT_OF_ROWS_SCANNED_KEY_METRIC_NAME, NUM_ROWS);
192
193    for (int i = 0; i < ROWS.length - 1; i++) {
194      scan = new Scan(baseScan);
195      scan.withStartRow(ROWS[0]);
196      scan.withStopRow(ROWS[i + 1]);
197      testMetric(scan, ServerSideScanMetrics.COUNT_OF_ROWS_SCANNED_KEY_METRIC_NAME, i + 1);
198    }
199
200    for (int i = ROWS.length - 1; i > 0; i--) {
201      scan = new Scan(baseScan);
202      scan.withStartRow(ROWS[i - 1]);
203      scan.withStopRow(ROWS[ROWS.length - 1]);
204      testMetric(scan, ServerSideScanMetrics.COUNT_OF_ROWS_SCANNED_KEY_METRIC_NAME, ROWS.length - i);
205    }
206
207    // The filter should filter out all rows, but we still expect to see every row.
208    Filter filter = new RowFilter(CompareOperator.EQUAL, new BinaryComparator("xyz".getBytes()));
209    scan = new Scan(baseScan);
210    scan.setFilter(filter);
211    testMetric(scan, ServerSideScanMetrics.COUNT_OF_ROWS_SCANNED_KEY_METRIC_NAME, ROWS.length);
212
213    // Filter should pass on all rows
214    SingleColumnValueFilter singleColumnValueFilter =
215        new SingleColumnValueFilter(FAMILIES[0], QUALIFIERS[0], CompareOperator.EQUAL, VALUE);
216    scan = new Scan(baseScan);
217    scan.setFilter(singleColumnValueFilter);
218    testMetric(scan, ServerSideScanMetrics.COUNT_OF_ROWS_SCANNED_KEY_METRIC_NAME, ROWS.length);
219
220    // Filter should filter out all rows
221    singleColumnValueFilter =
222        new SingleColumnValueFilter(FAMILIES[0], QUALIFIERS[0], CompareOperator.NOT_EQUAL, VALUE);
223    scan = new Scan(baseScan);
224    scan.setFilter(singleColumnValueFilter);
225    testMetric(scan, ServerSideScanMetrics.COUNT_OF_ROWS_SCANNED_KEY_METRIC_NAME, ROWS.length);
226  }

228  @Test
229  public void testRowsFilteredMetric() throws Exception {
230    // Base scan configuration
231    Scan baseScan;
232    baseScan = new Scan();
233    baseScan.setScanMetricsEnabled(true);
234
235    // Test case where scan uses default values
236    testRowsFilteredMetric(baseScan);
237
238    // Test case where at most one Result is retrieved per RPC
239    baseScan.setCaching(1);
240    testRowsFilteredMetric(baseScan);
241
242    // Test case where size limit is very restrictive and partial results will be returned from
243    // server
244    baseScan.setMaxResultSize(1);
245    testRowsFilteredMetric(baseScan);
246
247    // Test a case where max result size limits response from server to only a few cells (not all
248    // cells from the row)
249    baseScan.setCaching(NUM_ROWS);
250    baseScan.setMaxResultSize(getCellHeapSize() * (NUM_COLS - 1));
251    testRowsSeenMetric(baseScan);
252  }

254  public void testRowsFilteredMetric(Scan baseScan) throws Exception {
255    testRowsFilteredMetric(baseScan, null, 0);
256
257    // Row filter doesn't match any row key. All rows should be filtered
258    Filter filter = new RowFilter(CompareOperator.EQUAL, new BinaryComparator("xyz".getBytes()));
259    testRowsFilteredMetric(baseScan, filter, ROWS.length);
260
261    // Filter will return results containing only the first key. Number of entire rows filtered
262    // should be 0.
263    filter = new FirstKeyOnlyFilter();
264    testRowsFilteredMetric(baseScan, filter, 0);
265
266    // Column prefix will find some matching qualifier on each row. Number of entire rows filtered
267    // should be 0
268    filter = new ColumnPrefixFilter(QUALIFIERS[0]);
269    testRowsFilteredMetric(baseScan, filter, 0);
270
271    // Column prefix will NOT find any matching qualifier on any row. All rows should be filtered
272    filter = new ColumnPrefixFilter("xyz".getBytes());
273    testRowsFilteredMetric(baseScan, filter, ROWS.length);
274
275    // Matching column value should exist in each row. No rows should be filtered.
276    filter = new SingleColumnValueFilter(FAMILIES[0], QUALIFIERS[0], CompareOperator.EQUAL, VALUE);
277    testRowsFilteredMetric(baseScan, filter, 0);
278
279    // No matching column value should exist in any row. Filter all rows
280    filter = new SingleColumnValueFilter(FAMILIES[0], QUALIFIERS[0],
281      CompareOperator.NOT_EQUAL, VALUE);
282    testRowsFilteredMetric(baseScan, filter, ROWS.length);
283
284    List<Filter> filters = new ArrayList<>();
285    filters.add(new RowFilter(CompareOperator.EQUAL, new BinaryComparator(ROWS[0])));
286    filters.add(new RowFilter(CompareOperator.EQUAL, new BinaryComparator(ROWS[3])));
287    int numberOfMatchingRowFilters = filters.size();
288    filter = new FilterList(Operator.MUST_PASS_ONE, filters);
289    testRowsFilteredMetric(baseScan, filter, ROWS.length - numberOfMatchingRowFilters);
290    filters.clear();
291
292    // Add a single column value exclude filter for each column... The net effect is that all
293    // columns will be excluded when scanning on the server side. This will result in an empty cell
294    // array in RegionScanner#nextInternal which should be interpreted as a row being filtered.
295    for (int family = 0; family < FAMILIES.length; family++) {
296      for (int qualifier = 0; qualifier < QUALIFIERS.length; qualifier++) {
297        filters.add(new SingleColumnValueExcludeFilter(FAMILIES[family], QUALIFIERS[qualifier],
298          CompareOperator.EQUAL, VALUE));
299      }
300    }
301    filter = new FilterList(Operator.MUST_PASS_ONE, filters);
302    testRowsFilteredMetric(baseScan, filter, ROWS.length);
303  }

305  public void testRowsFilteredMetric(Scan baseScan, Filter filter, int expectedNumFiltered)
306      throws Exception {
307    Scan scan = new Scan(baseScan);
308    if (filter != null) scan.setFilter(filter);
309    testMetric(scan, ServerSideScanMetrics.COUNT_OF_ROWS_FILTERED_KEY_METRIC_NAME, expectedNumFiltered);
310  }

312  /**
313   * Run the scan to completetion and check the metric against the specified value
314   * @param scan
315   * @param metricKey
316   * @param expectedValue
317   * @throws Exception
318   */
319  public void testMetric(Scan scan, String metricKey, long expectedValue) throws Exception {
320    assertTrue("Scan should be configured to record metrics", scan.isScanMetricsEnabled());
321    ResultScanner scanner = TABLE.getScanner(scan);
322    // Iterate through all the results
323    while (scanner.next() != null) {
324
325    }
326    scanner.close();
327    ScanMetrics metrics = scanner.getScanMetrics();
328    assertTrue("Metrics are null", metrics != null);
329    assertTrue("Metric : " + metricKey + " does not exist", metrics.hasCounter(metricKey));
330    final long actualMetricValue = metrics.getCounter(metricKey).get();
331    assertEquals("Metric: " + metricKey + " Expected: " + expectedValue + " Actual: "
332        + actualMetricValue, expectedValue, actualMetricValue);
