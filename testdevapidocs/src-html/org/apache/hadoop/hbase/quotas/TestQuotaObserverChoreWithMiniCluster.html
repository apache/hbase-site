114  @Test
115  public void testTableViolatesQuota() throws Exception {
116    TableName tn = helper.createTableWithRegions(10);
117
118    final long sizeLimit = 2L * SpaceQuotaHelperForTests.ONE_MEGABYTE;
119    final SpaceViolationPolicy violationPolicy = SpaceViolationPolicy.NO_INSERTS;
120    QuotaSettings settings = QuotaSettingsFactory.limitTableSpace(tn, sizeLimit, violationPolicy);
121    TEST_UTIL.getAdmin().setQuota(settings);
122
123    // Write more data than should be allowed
124    helper.writeData(tn, 3L * SpaceQuotaHelperForTests.ONE_MEGABYTE);
125
126    Map<TableName,SpaceQuotaSnapshot> quotaSnapshots = snapshotNotifier.copySnapshots();
127    boolean foundSnapshot = false;
128    while (!foundSnapshot) {
129      if (quotaSnapshots.isEmpty()) {
130        LOG.info("Found no violated quotas, sleeping and retrying. Current reports: "
131            + master.getMasterQuotaManager().snapshotRegionSizes());
132        sleepWithInterrupt(DEFAULT_WAIT_MILLIS);
133        quotaSnapshots = snapshotNotifier.copySnapshots();
134      } else {
135        Entry<TableName,SpaceQuotaSnapshot> entry = Iterables.getOnlyElement(quotaSnapshots.entrySet());
136        assertEquals(tn, entry.getKey());
137        final SpaceQuotaSnapshot snapshot = entry.getValue();
138        if (!snapshot.getQuotaStatus().isInViolation()) {
139          LOG.info("Found a snapshot, but it was not yet in violation. " + snapshot);
140          sleepWithInterrupt(DEFAULT_WAIT_MILLIS);
141          quotaSnapshots = snapshotNotifier.copySnapshots();
142        } else {
143          foundSnapshot = true;
144        }
145      }
146    }
147
148    Entry<TableName,SpaceQuotaSnapshot> entry = Iterables.getOnlyElement(quotaSnapshots.entrySet());
149    assertEquals(tn, entry.getKey());
150    final SpaceQuotaSnapshot snapshot = entry.getValue();
151    assertEquals("Snapshot was " + snapshot, violationPolicy, snapshot.getQuotaStatus().getPolicy());
152    assertEquals(sizeLimit, snapshot.getLimit());
153    assertTrue(
154        "The usage should be greater than the limit, but were " + snapshot.getUsage() + " and "
155        + snapshot.getLimit() + ", respectively", snapshot.getUsage() > snapshot.getLimit());
156  }

158  @Test
159  public void testNamespaceViolatesQuota() throws Exception {
160    final String namespace = testName.getMethodName();
161    final Admin admin = TEST_UTIL.getAdmin();
162    // Ensure the namespace exists
163    try {
164      admin.getNamespaceDescriptor(namespace);
165    } catch (NamespaceNotFoundException e) {
166      NamespaceDescriptor desc = NamespaceDescriptor.create(namespace).build();
167      admin.createNamespace(desc);
168    }
169
170    TableName tn1 = helper.createTableWithRegions(namespace, 5);
171    TableName tn2 = helper.createTableWithRegions(namespace, 5);
172    TableName tn3 = helper.createTableWithRegions(namespace, 5);
173
174    final long sizeLimit = 5L * SpaceQuotaHelperForTests.ONE_MEGABYTE;
175    final SpaceViolationPolicy violationPolicy = SpaceViolationPolicy.DISABLE;
176    QuotaSettings settings = QuotaSettingsFactory.limitNamespaceSpace(namespace, sizeLimit, violationPolicy);
177    admin.setQuota(settings);
178
179    helper.writeData(tn1, 2L * SpaceQuotaHelperForTests.ONE_MEGABYTE);
180    admin.flush(tn1);
181    Map<TableName,SpaceQuotaSnapshot> snapshots = snapshotNotifier.copySnapshots();
182    for (int i = 0; i < 5; i++) {
183      // Check a few times to make sure we don't prematurely move to violation
184      assertEquals(
185          "Should not see any quota violations after writing 2MB of data", 0,
186          numSnapshotsInViolation(snapshots));
187      try {
188        Thread.sleep(DEFAULT_WAIT_MILLIS);
189      } catch (InterruptedException e) {
190        LOG.debug("Interrupted while sleeping." , e);
191      }
192      snapshots = snapshotNotifier.copySnapshots();
193    }
194
195    helper.writeData(tn2, 2L * SpaceQuotaHelperForTests.ONE_MEGABYTE);
196    admin.flush(tn2);
197    snapshots = snapshotNotifier.copySnapshots();
198    for (int i = 0; i < 5; i++) {
199      // Check a few times to make sure we don't prematurely move to violation
200      assertEquals("Should not see any quota violations after writing 4MB of data", 0,
201          numSnapshotsInViolation(snapshots));
202      try {
203        Thread.sleep(DEFAULT_WAIT_MILLIS);
204      } catch (InterruptedException e) {
205        LOG.debug("Interrupted while sleeping." , e);
206      }
207      snapshots = snapshotNotifier.copySnapshots();
208    }
209
210    // Writing the final 2MB of data will push the namespace over the 5MB limit (6MB in total)
211    // and should push all three tables in the namespace into violation.
212    helper.writeData(tn3, 2L * SpaceQuotaHelperForTests.ONE_MEGABYTE);
213    admin.flush(tn3);
214    snapshots = snapshotNotifier.copySnapshots();
215    while (numSnapshotsInViolation(snapshots) < 3) {
216      LOG.debug("Saw fewer violations than desired (expected 3): " + snapshots
217          + ". Current reports: " + master.getMasterQuotaManager().snapshotRegionSizes());
218      try {
219        Thread.sleep(DEFAULT_WAIT_MILLIS);
220      } catch (InterruptedException e) {
221        LOG.debug("Interrupted while sleeping.", e);
222        Thread.currentThread().interrupt();
223      }
224      snapshots = snapshotNotifier.copySnapshots();
225    }
226
227    SpaceQuotaSnapshot snapshot1 = snapshots.remove(tn1);
228    assertNotNull("tn1 should be in violation", snapshot1);
229    assertEquals(violationPolicy, snapshot1.getQuotaStatus().getPolicy());
230    SpaceQuotaSnapshot snapshot2 = snapshots.remove(tn2);
231    assertNotNull("tn2 should be in violation", snapshot2);
232    assertEquals(violationPolicy, snapshot2.getQuotaStatus().getPolicy());
233    SpaceQuotaSnapshot snapshot3 = snapshots.remove(tn3);
234    assertNotNull("tn3 should be in violation", snapshot3);
235    assertEquals(violationPolicy, snapshot3.getQuotaStatus().getPolicy());
236    assertTrue("Unexpected additional quota violations: " + snapshots, snapshots.isEmpty());
237  }

239  @Test
240  public void testTableQuotaOverridesNamespaceQuota() throws Exception {
241    final String namespace = testName.getMethodName();
242    final Admin admin = TEST_UTIL.getAdmin();
243    // Ensure the namespace exists
244    try {
245      admin.getNamespaceDescriptor(namespace);
246    } catch (NamespaceNotFoundException e) {
247      NamespaceDescriptor desc = NamespaceDescriptor.create(namespace).build();
248      admin.createNamespace(desc);
249    }
250
251    TableName tn1 = helper.createTableWithRegions(namespace, 5);
252    TableName tn2 = helper.createTableWithRegions(namespace, 5);
253
254    final long namespaceSizeLimit = 3L * SpaceQuotaHelperForTests.ONE_MEGABYTE;
255    final SpaceViolationPolicy namespaceViolationPolicy = SpaceViolationPolicy.DISABLE;
256    QuotaSettings namespaceSettings = QuotaSettingsFactory.limitNamespaceSpace(namespace,
257        namespaceSizeLimit, namespaceViolationPolicy);
258    admin.setQuota(namespaceSettings);
259
260    helper.writeData(tn1, 2L * SpaceQuotaHelperForTests.ONE_MEGABYTE);
261    admin.flush(tn1);
262    Map<TableName,SpaceQuotaSnapshot> snapshots = snapshotNotifier.copySnapshots();
263    for (int i = 0; i < 5; i++) {
264      // Check a few times to make sure we don't prematurely move to violation
265      assertEquals("Should not see any quota violations after writing 2MB of data: " + snapshots, 0,
266          numSnapshotsInViolation(snapshots));
267      try {
268        Thread.sleep(DEFAULT_WAIT_MILLIS);
269      } catch (InterruptedException e) {
270        LOG.debug("Interrupted while sleeping." , e);
271      }
272      snapshots = snapshotNotifier.copySnapshots();
273    }
274
275    helper.writeData(tn2, 2L * SpaceQuotaHelperForTests.ONE_MEGABYTE);
276    admin.flush(tn2);
277    snapshots = snapshotNotifier.copySnapshots();
278    while (numSnapshotsInViolation(snapshots) < 2) {
279      LOG.debug("Saw fewer violations than desired (expected 2): " + snapshots
280          + ". Current reports: " + master.getMasterQuotaManager().snapshotRegionSizes());
281      try {
282        Thread.sleep(DEFAULT_WAIT_MILLIS);
283      } catch (InterruptedException e) {
284        LOG.debug("Interrupted while sleeping.", e);
285        Thread.currentThread().interrupt();
286      }
287      snapshots = snapshotNotifier.copySnapshots();
288    }
289
290    SpaceQuotaSnapshot actualPolicyTN1 = snapshots.get(tn1);
291    assertNotNull("Expected to see violation policy for tn1", actualPolicyTN1);
292    assertEquals(namespaceViolationPolicy, actualPolicyTN1.getQuotaStatus().getPolicy());
293    SpaceQuotaSnapshot actualPolicyTN2 = snapshots.get(tn2);
294    assertNotNull("Expected to see violation policy for tn2", actualPolicyTN2);
295    assertEquals(namespaceViolationPolicy, actualPolicyTN2.getQuotaStatus().getPolicy());
296
297    // Override the namespace quota with a table quota
298    final long tableSizeLimit = SpaceQuotaHelperForTests.ONE_MEGABYTE;
299    final SpaceViolationPolicy tableViolationPolicy = SpaceViolationPolicy.NO_INSERTS;
300    QuotaSettings tableSettings = QuotaSettingsFactory.limitTableSpace(tn1, tableSizeLimit,
301        tableViolationPolicy);
302    admin.setQuota(tableSettings);
303
304    // Keep checking for the table quota policy to override the namespace quota
305    while (true) {
306      snapshots = snapshotNotifier.copySnapshots();
307      SpaceQuotaSnapshot actualTableSnapshot = snapshots.get(tn1);
308      assertNotNull("Violation policy should never be null", actualTableSnapshot);
309      if (tableViolationPolicy != actualTableSnapshot.getQuotaStatus().getPolicy()) {
310        LOG.debug("Saw unexpected table violation policy, waiting and re-checking.");
311        try {
312          Thread.sleep(DEFAULT_WAIT_MILLIS);
313        } catch (InterruptedException e) {
314          LOG.debug("Interrupted while sleeping");
315          Thread.currentThread().interrupt();
316        }
317        continue;
318      }
319      assertEquals(tableViolationPolicy, actualTableSnapshot.getQuotaStatus().getPolicy());
320      break;
321    }
322
323    // This should not change with the introduction of the table quota for tn1
324    actualPolicyTN2 = snapshots.get(tn2);
325    assertNotNull("Expected to see violation policy for tn2", actualPolicyTN2);
326    assertEquals(namespaceViolationPolicy, actualPolicyTN2.getQuotaStatus().getPolicy());
327  }

360  @Test
361  public void testFilterRegions() throws Exception {
362    Map<TableName,Integer> mockReportedRegions = new HashMap<>();
363    // Can't mock because of primitive int as a return type -- Mockito
364    // can only handle an Integer.
365    TablesWithQuotas tables = new TablesWithQuotas(TEST_UTIL.getConnection(),
366        TEST_UTIL.getConfiguration()) {
367      @Override
368      int getNumReportedRegions(TableName table, QuotaSnapshotStore<TableName> tableStore) {
369        Integer i = mockReportedRegions.get(table);
370        if (i == null) {
371          return 0;
372        }
373        return i;
374      }
375    };
376
377    // Create the tables
378    TableName tn1 = helper.createTableWithRegions(20);
379    TableName tn2 = helper.createTableWithRegions(20);
380    TableName tn3 = helper.createTableWithRegions(20);
381
382    // Add them to the Tables with Quotas object
383    tables.addTableQuotaTable(tn1);
384    tables.addTableQuotaTable(tn2);
385    tables.addTableQuotaTable(tn3);
386
387    // Mock the number of regions reported
388    mockReportedRegions.put(tn1, 10); // 50%
389    mockReportedRegions.put(tn2, 19); // 95%
390    mockReportedRegions.put(tn3, 20); // 100%
391
392    // Argument is un-used
393    tables.filterInsufficientlyReportedTables(null);
394    // The default of 95% reported should prevent tn1 from appearing
395    assertEquals(new HashSet<>(Arrays.asList(tn2, tn3)), tables.getTableQuotaTables());
396  }

398  @Test
399  public void testFetchSpaceQuota() throws Exception {
400    Multimap<TableName,QuotaSettings> tables = helper.createTablesWithSpaceQuotas();
401    // Can pass in an empty map, we're not consulting it.
402    chore.initializeSnapshotStores(Collections.emptyMap());
403    // All tables that were created should have a quota defined.
404    for (Entry<TableName,QuotaSettings> entry : tables.entries()) {
405      final TableName table = entry.getKey();
406      final QuotaSettings qs = entry.getValue();
407
408      assertTrue("QuotaSettings was an instance of " + qs.getClass(),
409          qs instanceof SpaceLimitSettings);
410
411      SpaceQuota spaceQuota = null;
412      if (qs.getTableName() != null) {
413        spaceQuota = chore.getTableSnapshotStore().getSpaceQuota(table);
414        assertNotNull("Could not find table space quota for " + table, spaceQuota);
415      } else if (qs.getNamespace() != null) {
416        spaceQuota = chore.getNamespaceSnapshotStore().getSpaceQuota(table.getNamespaceAsString());
417        assertNotNull("Could not find namespace space quota for " + table.getNamespaceAsString(), spaceQuota);
418      } else {
419        fail("Expected table or namespace space quota");
420      }
421
422      final SpaceLimitSettings sls = (SpaceLimitSettings) qs;
423      assertEquals(sls.getProto().getQuota(), spaceQuota);
424    }
425
426    TableName tableWithoutQuota = helper.createTable();
427    assertNull(chore.getTableSnapshotStore().getSpaceQuota(tableWithoutQuota));
428  }

430  private int numSnapshotsInViolation(Map<TableName,SpaceQuotaSnapshot> snapshots) {
431    int sum = 0;
432    for (SpaceQuotaSnapshot snapshot : snapshots.values()) {
433      if (snapshot.getQuotaStatus().isInViolation()) {
434        sum++;
435      }
436    }
437    return sum;
438  }

446    }
