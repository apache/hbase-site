041  private final static Logger LOG = LoggerFactory.getLogger(ThrottleQuotaTestUtil.class);
042  private static ManualEnvironmentEdge envEdge = new ManualEnvironmentEdge();
043  private final static int REFRESH_TIME = 30 * 60000;
044  static {
045    envEdge.setValue(EnvironmentEdgeManager.currentTime());
046    // only active the envEdge for quotas package
047    EnvironmentEdgeManagerTestHelper.injectEdgeForPackage(envEdge,
048      ThrottleQuotaTestUtil.class.getPackage().getName());

056  static int doPuts(int maxOps, byte[] family, byte[] qualifier, final Table... tables) {
057    return doPuts(maxOps, -1, family, qualifier, tables);
058  }
059
060  static int doPuts(int maxOps, int valueSize, byte[] family, byte[] qualifier,
061    final Table... tables) {
062    int count = 0;
063    try {
064      while (count < maxOps) {
065        Put put = new Put(Bytes.toBytes("row-" + count));
066        byte[] value;
067        if (valueSize < 0) {
068          value = Bytes.toBytes("data-" + count);
069        } else {
070          value = generateValue(valueSize);
071        }
072        put.addColumn(family, qualifier, value);
073        for (final Table table : tables) {
074          table.put(put);
075        }
076        count += tables.length;
077      }
078    } catch (IOException e) {
079      LOG.error("put failed after nRetries=" + count, e);
080    }
081    return count;
082  }
083
084  private static byte[] generateValue(int valueSize) {
085    byte[] bytes = new byte[valueSize];
086    for (int i = 0; i < valueSize; i++) {
087      bytes[i] = 'a';
088    }
089    return bytes;
090  }
091
092  static long doGets(int maxOps, final Table... tables) {
093    int count = 0;
094    try {
095      while (count < maxOps) {
096        Get get = new Get(Bytes.toBytes("row-" + count));
097        for (final Table table : tables) {
098          table.get(get);
099        }
100        count += tables.length;
101      }
102    } catch (IOException e) {
103      LOG.error("get failed after nRetries=" + count, e);
104    }
105    return count;
106  }

133  private static void triggerCacheRefresh(HBaseTestingUtil testUtil, boolean bypass,
134    boolean userLimiter, boolean tableLimiter, boolean nsLimiter, boolean rsLimiter,
135    boolean exceedThrottleQuota, final TableName... tables) throws Exception {
136    envEdge.incValue(2 * REFRESH_TIME);
137    for (RegionServerThread rst : testUtil.getMiniHBaseCluster().getRegionServerThreads()) {
138      RegionServerRpcQuotaManager quotaManager =
139        rst.getRegionServer().getRegionServerRpcQuotaManager();
140      QuotaCache quotaCache = quotaManager.getQuotaCache();
141      quotaCache.triggerCacheRefresh();
142      Thread.sleep(250);
143      testUtil.waitFor(60000, 250, new ExplainingPredicate<Exception>() {
144
145        @Override
146        public boolean evaluate() throws Exception {
147          boolean isUpdated = true;
148          for (TableName table : tables) {
149            if (userLimiter) {
150              boolean isUserBypass =
151                quotaCache.getUserLimiter(User.getCurrent().getUGI(), table).isBypass();
152              if (isUserBypass != bypass) {
153                LOG.info(
154                  "User limiter for user={}, table={} not refreshed, bypass expected {}, actual {}",
155                  User.getCurrent(), table, bypass, isUserBypass);
156                envEdge.incValue(100);
157                isUpdated = false;
158                break;
159              }
160            }
161            if (tableLimiter) {
162              boolean isTableBypass = quotaCache.getTableLimiter(table).isBypass();
163              if (isTableBypass != bypass) {
164                LOG.info("Table limiter for table={} not refreshed, bypass expected {}, actual {}",
165                  table, bypass, isTableBypass);
166                envEdge.incValue(100);
167                isUpdated = false;
168                break;
169              }
170            }
171            if (nsLimiter) {
172              boolean isNsBypass =
173                quotaCache.getNamespaceLimiter(table.getNamespaceAsString()).isBypass();
174              if (isNsBypass != bypass) {
175                LOG.info(
176                  "Namespace limiter for namespace={} not refreshed, bypass expected {}, actual {}",
177                  table.getNamespaceAsString(), bypass, isNsBypass);
178                envEdge.incValue(100);
179                isUpdated = false;
180                break;
181              }
182            }
183          }
184          if (rsLimiter) {
185            boolean rsIsBypass = quotaCache
186              .getRegionServerQuotaLimiter(QuotaTableUtil.QUOTA_REGION_SERVER_ROW_KEY).isBypass();
187            if (rsIsBypass != bypass) {
188              LOG.info("RegionServer limiter not refreshed, bypass expected {}, actual {}", bypass,
189                rsIsBypass);
190              envEdge.incValue(100);
191              isUpdated = false;
192            }
193          }
194          if (exceedThrottleQuota) {
195            if (quotaCache.isExceedThrottleQuotaEnabled() != bypass) {
196              LOG.info("ExceedThrottleQuotaEnabled not refreshed, bypass expected {}, actual {}",
197                bypass, quotaCache.isExceedThrottleQuotaEnabled());
198              envEdge.incValue(100);
199              isUpdated = false;
200            }
201          }
202          if (isUpdated) {
203            return true;
204          }
205          quotaCache.triggerCacheRefresh();
206          return false;
207        }
208
209        @Override
210        public String explainFailure() throws Exception {
211          return "Quota cache is still not refreshed";
212        }
213      });
214
215      LOG.debug("QuotaCache");
216      LOG.debug(Objects.toString(quotaCache.getNamespaceQuotaCache()));
217      LOG.debug(Objects.toString(quotaCache.getTableQuotaCache()));
218      LOG.debug(Objects.toString(quotaCache.getUserQuotaCache()));
219      LOG.debug(Objects.toString(quotaCache.getRegionServerQuotaCache()));
220    }
221  }

227  static void clearQuotaCache(HBaseTestingUtil testUtil) {
228    for (RegionServerThread rst : testUtil.getMiniHBaseCluster().getRegionServerThreads()) {
229      RegionServerRpcQuotaManager quotaManager =
230        rst.getRegionServer().getRegionServerRpcQuotaManager();
231      QuotaCache quotaCache = quotaManager.getQuotaCache();
232      quotaCache.getNamespaceQuotaCache().clear();
233      quotaCache.getTableQuotaCache().clear();
234      quotaCache.getUserQuotaCache().clear();
235      quotaCache.getRegionServerQuotaCache().clear();
236    }
