017package org.apache.hadoop.hbase.quotas;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertFalse;
021import static org.junit.Assert.assertNotNull;
022
023import org.apache.hadoop.hbase.TableName;
024import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
025import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProtos.SetQuotaRequest;
026import org.apache.hadoop.hbase.shaded.protobuf.generated.QuotaProtos.SpaceLimitRequest;
027import org.apache.hadoop.hbase.shaded.protobuf.generated.QuotaProtos.SpaceQuota;
028import org.apache.hadoop.hbase.testclassification.SmallTests;
029import org.junit.Test;
030import org.junit.experimental.categories.Category;
031
032/**
033 * Test class for {@link SpaceLimitSettings}.
034 */
035@Category({SmallTests.class})
036public class TestSpaceLimitSettings {
037
038  @Test(expected = IllegalArgumentException.class)
039  public void testInvalidTableQuotaSizeLimit() {
040    new SpaceLimitSettings(TableName.valueOf("foo"), -1, SpaceViolationPolicy.NO_INSERTS);
041  }
042
043  @Test(expected = NullPointerException.class)
044  public void testNullTableName() {
045    TableName tn = null;
046    new SpaceLimitSettings(tn, 1, SpaceViolationPolicy.NO_INSERTS);
047  }
048
049  @Test(expected = NullPointerException.class)
050  public void testNullTableViolationPolicy() {
051    new SpaceLimitSettings(TableName.valueOf("foo"), 1, null);
052  }
053
054  @Test(expected = IllegalArgumentException.class)
055  public void testInvalidNamespaceQuotaSizeLimit() {
056    new SpaceLimitSettings("foo_ns", -1, SpaceViolationPolicy.NO_INSERTS);
057  }
058
059  @Test(expected = NullPointerException.class)
060  public void testNullNamespace() {
061    String ns = null;
062    new SpaceLimitSettings(ns, 1, SpaceViolationPolicy.NO_INSERTS);
063  }
064
065  @Test(expected = NullPointerException.class)
066  public void testNullNamespaceViolationPolicy() {
067    new SpaceLimitSettings("foo_ns", 1, null);
068  }
069
070  @Test
071  public void testTableQuota() {
072    final TableName tableName = TableName.valueOf("foo");
073    final long sizeLimit = 1024 * 1024;
074    final SpaceViolationPolicy policy = SpaceViolationPolicy.NO_WRITES;
075    SpaceLimitSettings settings = new SpaceLimitSettings(tableName, sizeLimit, policy);
076    SetQuotaRequest proto = QuotaSettings.buildSetQuotaRequestProto(settings);
077
078    assertFalse("User should be missing", proto.hasUserName());
079    assertFalse("Namespace should be missing", proto.hasNamespace());
080    assertEquals(ProtobufUtil.toProtoTableName(tableName), proto.getTableName());
081    SpaceLimitRequest spaceLimitReq = proto.getSpaceLimit();
082    assertNotNull("SpaceLimitRequest was null", spaceLimitReq);
083    SpaceQuota spaceQuota = spaceLimitReq.getQuota();
084    assertNotNull("SpaceQuota was null", spaceQuota);
085    assertEquals(sizeLimit, spaceQuota.getSoftLimit());
086    assertEquals(ProtobufUtil.toProtoViolationPolicy(policy), spaceQuota.getViolationPolicy());
087
088    assertEquals(QuotaType.SPACE, settings.getQuotaType());
089
090    SpaceLimitSettings copy = new SpaceLimitSettings(tableName, sizeLimit, policy);
091    assertEquals(settings, copy);
092    assertEquals(settings.hashCode(), copy.hashCode());
093  }
094
095  @Test
096  public void testNamespaceQuota() {
097    final String namespace = "foo_ns";
098    final long sizeLimit = 1024 * 1024;
099    final SpaceViolationPolicy policy = SpaceViolationPolicy.NO_WRITES;
100    SpaceLimitSettings settings = new SpaceLimitSettings(namespace, sizeLimit, policy);
101    SetQuotaRequest proto = QuotaSettings.buildSetQuotaRequestProto(settings);
102
103    assertFalse("User should be missing", proto.hasUserName());
104    assertFalse("TableName should be missing", proto.hasTableName());
105    assertEquals(namespace, proto.getNamespace());
106    SpaceLimitRequest spaceLimitReq = proto.getSpaceLimit();
107    assertNotNull("SpaceLimitRequest was null", spaceLimitReq);
108    SpaceQuota spaceQuota = spaceLimitReq.getQuota();
109    assertNotNull("SpaceQuota was null", spaceQuota);
110    assertEquals(sizeLimit, spaceQuota.getSoftLimit());
111    assertEquals(ProtobufUtil.toProtoViolationPolicy(policy), spaceQuota.getViolationPolicy());
112
113    assertEquals(QuotaType.SPACE, settings.getQuotaType());
114
115    SpaceLimitSettings copy = new SpaceLimitSettings(namespace, sizeLimit, policy);
116    assertEquals(settings, copy);
117    assertEquals(settings.hashCode(), copy.hashCode());
118  }
119}
