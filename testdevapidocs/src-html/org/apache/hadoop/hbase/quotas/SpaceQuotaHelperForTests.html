161  /**
162   * Verifies that the given policy on the given table has been violated
163   */
164  void verifyViolation(SpaceViolationPolicy policyToViolate, TableName tn, Mutation m)
165      throws Exception {
166    // But let's try a few times to get the exception before failing
167    boolean sawError = false;
168    String msg = "";
169    for (int i = 0; i < NUM_RETRIES && !sawError; i++) {
170      try (Table table = testUtil.getConnection().getTable(tn)) {
171        if (m instanceof Put) {
172          table.put((Put) m);
173        } else if (m instanceof Delete) {
174          table.delete((Delete) m);
175        } else if (m instanceof Append) {
176          table.append((Append) m);
177        } else if (m instanceof Increment) {
178          table.increment((Increment) m);
179        } else {
180          fail(
181              "Failed to apply " + m.getClass().getSimpleName() +
182                  " to the table. Programming error");
183        }
184        LOG.info("Did not reject the " + m.getClass().getSimpleName() + ", will sleep and retry");
185        Thread.sleep(2000);
186      } catch (Exception e) {
187        msg = StringUtils.stringifyException(e);
188        if ((policyToViolate.equals(SpaceViolationPolicy.DISABLE)
189            && e instanceof TableNotEnabledException) || msg.contains(policyToViolate.name())) {
190          LOG.info("Got the expected exception={}", msg);
191          sawError = true;
192          break;
193        } else {
194          LOG.warn("Did not get the expected exception, will sleep and retry", e);
195          Thread.sleep(2000);
196        }
197      }
198    }
199    if (!sawError) {
200      try (Table quotaTable = testUtil.getConnection().getTable(QuotaUtil.QUOTA_TABLE_NAME)) {
201        ResultScanner scanner = quotaTable.getScanner(new Scan());
202        Result result = null;
203        LOG.info("Dumping contents of hbase:quota table");
204        while ((result = scanner.next()) != null) {
205          LOG.info(Bytes.toString(result.getRow()) + " => " + result.toString());
206        }
207        scanner.close();
208      }
209    } else {
210      if (policyToViolate.equals(SpaceViolationPolicy.DISABLE)) {
211        assertTrue(
212            msg.contains("TableNotEnabledException") || msg.contains(policyToViolate.name()));
213      } else {
214        assertTrue("Expected exception message to contain the word '" + policyToViolate.name()
215                + "', but was " + msg,
216            msg.contains(policyToViolate.name()));
217      }
218    }
219    assertTrue(
220        "Expected to see an exception writing data to a table exceeding its quota", sawError);
221  }

226  void verifyNoViolation(TableName tn, Mutation m) throws Exception {
227    // But let's try a few times to write data before failing
228    boolean sawSuccess = false;
229    for (int i = 0; i < NUM_RETRIES && !sawSuccess; i++) {
230      try (Table table = testUtil.getConnection().getTable(tn)) {
231        if (m instanceof Put) {
232          table.put((Put) m);

239        } else {
240          fail("Failed to apply " + m.getClass().getSimpleName() + " to the table."
241              + " Programming error");
242        }
243        sawSuccess = true;
244      } catch (Exception e) {
245        LOG.info("Rejected the " + m.getClass().getSimpleName() + ", will sleep and retry");

257        scanner.close();
258      }
259    }

292  /**
293   * Removes all quotas defined in the HBase quota table.
294   */
295  void removeAllQuotas(Connection conn) throws IOException {
296    // Wait for the quota table to be created
297    if (!conn.getAdmin().tableExists(QuotaUtil.QUOTA_TABLE_NAME)) {
298      waitForQuotaTable(conn);
299    } else {
300      // Or, clean up any quotas from previous test runs.
301      QuotaRetriever scanner = QuotaRetriever.open(conn.getConfiguration());
302      try {
303        for (QuotaSettings quotaSettings : scanner) {
304          final String namespace = quotaSettings.getNamespace();
305          final TableName tableName = quotaSettings.getTableName();
306          final String userName = quotaSettings.getUserName();
307          if (namespace != null) {
308            LOG.debug("Deleting quota for namespace: " + namespace);
309            QuotaUtil.deleteNamespaceQuota(conn, namespace);
310          } else if (tableName != null) {
311            LOG.debug("Deleting quota for table: " + tableName);
312            QuotaUtil.deleteTableQuota(conn, tableName);
313          } else if (userName != null) {
314            LOG.debug("Deleting quota for user: " + userName);
315            QuotaUtil.deleteUserQuota(conn, userName);
316          }
317        }
318      } finally {
319        if (scanner != null) {
320          scanner.close();
321        }
322      }
323    }
324  }

369  void writeData(TableName tn, long sizeInBytes, byte[] qual) throws IOException {
370    final Connection conn = testUtil.getConnection();
371    final Table table = conn.getTable(tn);
372    try {
373      List<Put> updates = new ArrayList<>();
374      long bytesToWrite = sizeInBytes;
375      long rowKeyId = 0L;
376      final StringBuilder sb = new StringBuilder();
377      final Random r = new Random();
378      while (bytesToWrite > 0L) {
379        sb.setLength(0);
380        sb.append(Long.toString(rowKeyId));
381        // Use the reverse counter as the rowKey to get even spread across all regions
382        Put p = new Put(Bytes.toBytes(sb.reverse().toString()));
383        byte[] value = new byte[SIZE_PER_VALUE];
384        r.nextBytes(value);
385        p.addColumn(Bytes.toBytes(F1), qual, value);
386        updates.add(p);
387
388        // Batch ~13KB worth of updates
389        if (updates.size() > 50) {
390          table.put(updates);
391          updates.clear();
392        }
393
394        // Just count the value size, ignore the size of rowkey + column
395        bytesToWrite -= SIZE_PER_VALUE;
396        rowKeyId++;
397      }
398
399      // Write the final batch
400      if (!updates.isEmpty()) {
401        table.put(updates);
402      }
403
404      LOG.debug("Data was written to HBase");
405      // Push the data to disk.
406      testUtil.getAdmin().flush(tn);
407      LOG.debug("Data flushed to disk");
408    } finally {
409      table.close();
410    }
411  }

419  Multimap<TableName, QuotaSettings> createTablesWithSpaceQuotas() throws Exception {
420    final Admin admin = testUtil.getAdmin();
421    final Multimap<TableName, QuotaSettings> tablesWithQuotas = HashMultimap.create();
422
423    final TableName tn1 = createTable();
424    final TableName tn2 = createTable();
425
426    NamespaceDescriptor nd = createNamespace();
427    final TableName tn3 = createTableInNamespace(nd);
428    final TableName tn4 = createTableInNamespace(nd);
429    final TableName tn5 = createTableInNamespace(nd);
430
431    final long sizeLimit1 = 1024L * 1024L * 1024L * 1024L * 5L; // 5TB
432    final SpaceViolationPolicy violationPolicy1 = SpaceViolationPolicy.NO_WRITES;
433    QuotaSettings qs1 = QuotaSettingsFactory.limitTableSpace(tn1, sizeLimit1, violationPolicy1);
434    tablesWithQuotas.put(tn1, qs1);
435    admin.setQuota(qs1);
436
437    final long sizeLimit2 = 1024L * 1024L * 1024L * 200L; // 200GB
438    final SpaceViolationPolicy violationPolicy2 = SpaceViolationPolicy.NO_WRITES_COMPACTIONS;
439    QuotaSettings qs2 = QuotaSettingsFactory.limitTableSpace(tn2, sizeLimit2, violationPolicy2);
440    tablesWithQuotas.put(tn2, qs2);
441    admin.setQuota(qs2);
442
443    final long sizeLimit3 = 1024L * 1024L * 1024L * 1024L * 100L; // 100TB
444    final SpaceViolationPolicy violationPolicy3 = SpaceViolationPolicy.NO_INSERTS;
445    QuotaSettings qs3 = QuotaSettingsFactory.limitNamespaceSpace(
446        nd.getName(), sizeLimit3, violationPolicy3);
447    tablesWithQuotas.put(tn3, qs3);
448    tablesWithQuotas.put(tn4, qs3);
449    tablesWithQuotas.put(tn5, qs3);
450    admin.setQuota(qs3);
451
452    final long sizeLimit4 = 1024L * 1024L * 1024L * 5L; // 5GB
453    final SpaceViolationPolicy violationPolicy4 = SpaceViolationPolicy.NO_INSERTS;
454    QuotaSettings qs4 = QuotaSettingsFactory.limitTableSpace(tn5, sizeLimit4, violationPolicy4);
455    // Override the ns quota for tn5, import edge-case to catch table quota taking
456    // precedence over ns quota.
457    tablesWithQuotas.put(tn5, qs4);
458    admin.setQuota(qs4);
459
460    return tablesWithQuotas;
461  }

488  TableName createTableWithRegions(Admin admin, String namespace, int numRegions) throws Exception {
489    final TableName tn = getNextTableName(namespace);
490
491    // Delete the old table
492    if (admin.tableExists(tn)) {
493      admin.disableTable(tn);
494      admin.deleteTable(tn);
495    }
496
497    // Create the table
498    TableDescriptor tableDesc = TableDescriptorBuilder.newBuilder(tn)
499        .setColumnFamily(ColumnFamilyDescriptorBuilder.of(F1)).build();
500    if (numRegions == 1) {
501      admin.createTable(tableDesc);
502    } else {
503      admin.createTable(tableDesc, Bytes.toBytes("0"), Bytes.toBytes("9"), numRegions);
504    }
505    return tn;
506  }

527  void partitionTablesByQuotaTarget(Multimap<TableName,QuotaSettings> quotas,
528      Set<TableName> tablesWithTableQuota, Set<TableName> tablesWithNamespaceQuota) {
529    // Partition the tables with quotas by table and ns quota
530    for (Entry<TableName, QuotaSettings> entry : quotas.entries()) {
531      SpaceLimitSettings settings = (SpaceLimitSettings) entry.getValue();
532      TableName tn = entry.getKey();
533      if (settings.getTableName() != null) {
534        tablesWithTableQuota.add(tn);
535      }
536      if (settings.getNamespace() != null) {
537        tablesWithNamespaceQuota.add(tn);
538      }
539
540      if (settings.getTableName() == null && settings.getNamespace() == null) {
541        fail("Unexpected table name with null tableName and namespace: " + tn);
542      }
543    }
544  }

566  /**
567   * Abstraction to simplify the case where a test needs to verify a certain state
568   * on a {@code SpaceQuotaSnapshot}. This class fails-fast when there is no such
569   * snapshot obtained from the Master. As such, it is not useful to verify the
570   * lack of a snapshot.
571   */
572  static abstract class SpaceQuotaSnapshotPredicate implements Predicate<Exception> {
573    private final Connection conn;
574    private final TableName tn;
575    private final String ns;
576
577    SpaceQuotaSnapshotPredicate(Connection conn, TableName tn) {
578      this(Objects.requireNonNull(conn), Objects.requireNonNull(tn), null);
579    }
580
581    SpaceQuotaSnapshotPredicate(Connection conn, String ns) {
582      this(Objects.requireNonNull(conn), null, Objects.requireNonNull(ns));
583    }
584
585    SpaceQuotaSnapshotPredicate(Connection conn, TableName tn, String ns) {
586      if ((null != tn && null != ns) || (null == tn && null == ns)) {
587        throw new IllegalArgumentException(
588            "One of TableName and Namespace must be non-null, and the other null");
589      }
590      this.conn = conn;
591      this.tn = tn;
592      this.ns = ns;
593    }
594
595    @Override
596    public boolean evaluate() throws Exception {
597      SpaceQuotaSnapshot snapshot;
598      if (null == ns) {
599        snapshot = (SpaceQuotaSnapshot) conn.getAdmin().getCurrentSpaceQuotaSnapshot(tn);
600      } else {
601        snapshot = (SpaceQuotaSnapshot) conn.getAdmin().getCurrentSpaceQuotaSnapshot(ns);
602      }
603
604      LOG.debug("Saw quota snapshot for " + (null == tn ? ns : tn) + ": " + snapshot);
605      if (null == snapshot) {
606        return false;
607      }
608      return evaluate(snapshot);
609    }
610
611    /**
612     * Must determine if the given {@code SpaceQuotaSnapshot} meets some criteria.
613     *
614     * @param snapshot a non-null snapshot obtained from the HBase Master
615     * @return true if the criteria is met, false otherwise
616     */
617    abstract boolean evaluate(SpaceQuotaSnapshot snapshot) throws Exception;
618  }
619
620  /**
621   * Predicate that waits for all store files in a table to have no compacted files.
622   */
623  static class NoFilesToDischarge implements Predicate<Exception> {
624    private final MiniHBaseCluster cluster;
625    private final TableName tn;
626
627    NoFilesToDischarge(MiniHBaseCluster cluster, TableName tn) {
628      this.cluster = cluster;
629      this.tn = tn;
630    }
631
632    @Override
633    public boolean evaluate() throws Exception {
634      for (HRegion region : cluster.getRegions(tn)) {
635        for (HStore store : region.getStores()) {
636          Collection<HStoreFile> files =
637              store.getStoreEngine().getStoreFileManager().getCompactedfiles();
638          if (null != files && !files.isEmpty()) {
639            LOG.debug(region.getRegionInfo().getEncodedName() + " still has compacted files");
640            return false;
641          }
642        }
643      }
644      return true;
645    }
646  }
