124  /**
125   * Removes all quotas defined in the HBase quota table.
126   */
127  void removeAllQuotas(Connection conn) throws IOException, InterruptedException {
128    // Wait for the quota table to be created
129    if (!conn.getAdmin().tableExists(QuotaUtil.QUOTA_TABLE_NAME)) {
130      waitForQuotaTable(conn);
131    } else {
132      // Or, clean up any quotas from previous test runs.
133      QuotaRetriever scanner = QuotaRetriever.open(conn.getConfiguration());
134      try {
135        for (QuotaSettings quotaSettings : scanner) {
136          final String namespace = quotaSettings.getNamespace();
137          final TableName tableName = quotaSettings.getTableName();
138          if (namespace != null) {
139            LOG.debug("Deleting quota for namespace: " + namespace);
140            QuotaUtil.deleteNamespaceQuota(conn, namespace);
141          } else {
142            assert tableName != null;
143            LOG.debug("Deleting quota for table: "+ tableName);
144            QuotaUtil.deleteTableQuota(conn, tableName);
145          }
146        }
147      } finally {
148        if (scanner != null) {
149          scanner.close();
150        }
151      }
152    }
153  }

155  QuotaSettings getTableSpaceQuota(Connection conn, TableName tn) throws IOException {
156    try (QuotaRetriever scanner = QuotaRetriever.open(
157        conn.getConfiguration(), new QuotaFilter().setTableFilter(tn.getNameAsString()))) {
158      for (QuotaSettings setting : scanner) {
159        if (setting.getTableName().equals(tn) && setting.getQuotaType() == QuotaType.SPACE) {
160          return setting;
161        }
162      }
163      return null;
164    }
165  }

198  void writeData(TableName tn, long sizeInBytes, byte[] qual) throws IOException {
199    final Connection conn = testUtil.getConnection();
200    final Table table = conn.getTable(tn);
201    try {
202      List<Put> updates = new ArrayList<>();
203      long bytesToWrite = sizeInBytes;
204      long rowKeyId = 0L;
205      final StringBuilder sb = new StringBuilder();
206      final Random r = new Random();
207      while (bytesToWrite > 0L) {
208        sb.setLength(0);
209        sb.append(Long.toString(rowKeyId));
210        // Use the reverse counter as the rowKey to get even spread across all regions
211        Put p = new Put(Bytes.toBytes(sb.reverse().toString()));
212        byte[] value = new byte[SIZE_PER_VALUE];
213        r.nextBytes(value);
214        p.addColumn(Bytes.toBytes(F1), qual, value);
215        updates.add(p);
216
217        // Batch ~13KB worth of updates
218        if (updates.size() > 50) {
219          table.put(updates);
220          updates.clear();
221        }
222
223        // Just count the value size, ignore the size of rowkey + column
224        bytesToWrite -= SIZE_PER_VALUE;
225        rowKeyId++;
226      }
227
228      // Write the final batch
229      if (!updates.isEmpty()) {
230        table.put(updates);
231      }
232
233      LOG.debug("Data was written to HBase");
234      // Push the data to disk.
235      testUtil.getAdmin().flush(tn);
236      LOG.debug("Data flushed to disk");
237    } finally {
238      table.close();
239    }
240  }

248  Multimap<TableName, QuotaSettings> createTablesWithSpaceQuotas() throws Exception {
249    final Admin admin = testUtil.getAdmin();
250    final Multimap<TableName, QuotaSettings> tablesWithQuotas = HashMultimap.create();
251
252    final TableName tn1 = createTable();
253    final TableName tn2 = createTable();
254
255    NamespaceDescriptor nd = createNamespace();
256    final TableName tn3 = createTableInNamespace(nd);
257    final TableName tn4 = createTableInNamespace(nd);
258    final TableName tn5 = createTableInNamespace(nd);
259
260    final long sizeLimit1 = 1024L * 1024L * 1024L * 1024L * 5L; // 5TB
261    final SpaceViolationPolicy violationPolicy1 = SpaceViolationPolicy.NO_WRITES;
262    QuotaSettings qs1 = QuotaSettingsFactory.limitTableSpace(tn1, sizeLimit1, violationPolicy1);
263    tablesWithQuotas.put(tn1, qs1);
264    admin.setQuota(qs1);
265
266    final long sizeLimit2 = 1024L * 1024L * 1024L * 200L; // 200GB
267    final SpaceViolationPolicy violationPolicy2 = SpaceViolationPolicy.NO_WRITES_COMPACTIONS;
268    QuotaSettings qs2 = QuotaSettingsFactory.limitTableSpace(tn2, sizeLimit2, violationPolicy2);
269    tablesWithQuotas.put(tn2, qs2);
270    admin.setQuota(qs2);
271
272    final long sizeLimit3 = 1024L * 1024L * 1024L * 1024L * 100L; // 100TB
273    final SpaceViolationPolicy violationPolicy3 = SpaceViolationPolicy.NO_INSERTS;
274    QuotaSettings qs3 = QuotaSettingsFactory.limitNamespaceSpace(
275        nd.getName(), sizeLimit3, violationPolicy3);
276    tablesWithQuotas.put(tn3, qs3);
277    tablesWithQuotas.put(tn4, qs3);
278    tablesWithQuotas.put(tn5, qs3);
279    admin.setQuota(qs3);
280
281    final long sizeLimit4 = 1024L * 1024L * 1024L * 5L; // 5GB
282    final SpaceViolationPolicy violationPolicy4 = SpaceViolationPolicy.NO_INSERTS;
283    QuotaSettings qs4 = QuotaSettingsFactory.limitTableSpace(tn5, sizeLimit4, violationPolicy4);
284    // Override the ns quota for tn5, import edge-case to catch table quota taking
285    // precedence over ns quota.
286    tablesWithQuotas.put(tn5, qs4);
287    admin.setQuota(qs4);
288
289    return tablesWithQuotas;
290  }

317  TableName createTableWithRegions(Admin admin, String namespace, int numRegions) throws Exception {
318    final TableName tn = getNextTableName(namespace);
319
320    // Delete the old table
321    if (admin.tableExists(tn)) {
322      admin.disableTable(tn);
323      admin.deleteTable(tn);
324    }
325
326    // Create the table
327    TableDescriptor tableDesc = TableDescriptorBuilder.newBuilder(tn)
328        .setColumnFamily(ColumnFamilyDescriptorBuilder.of(F1)).build();
329    if (numRegions == 1) {
330      admin.createTable(tableDesc);
331    } else {
332      admin.createTable(tableDesc, Bytes.toBytes("0"), Bytes.toBytes("9"), numRegions);
333    }
334    return tn;
335  }

356  void partitionTablesByQuotaTarget(Multimap<TableName,QuotaSettings> quotas,
357      Set<TableName> tablesWithTableQuota, Set<TableName> tablesWithNamespaceQuota) {
358    // Partition the tables with quotas by table and ns quota
359    for (Entry<TableName, QuotaSettings> entry : quotas.entries()) {
360      SpaceLimitSettings settings = (SpaceLimitSettings) entry.getValue();
361      TableName tn = entry.getKey();
362      if (settings.getTableName() != null) {
363        tablesWithTableQuota.add(tn);
364      }
365      if (settings.getNamespace() != null) {
366        tablesWithNamespaceQuota.add(tn);
367      }
368
369      if (settings.getTableName() == null && settings.getNamespace() == null) {
370        fail("Unexpected table name with null tableName and namespace: " + tn);
371      }
372    }
373  }

376   * Bulk-loads a number of files with a number of rows to the given table.
377   */
378  ClientServiceCallable<Boolean> generateFileToLoad(
379      TableName tn, int numFiles, int numRowsPerFile) throws Exception {
380    Connection conn = testUtil.getConnection();
381    FileSystem fs = testUtil.getTestFileSystem();
382    Configuration conf = testUtil.getConfiguration();
383    Path baseDir = new Path(fs.getHomeDirectory(), testName.getMethodName() + "_files");
384    fs.mkdirs(baseDir);
385    final List<Pair<byte[], String>> famPaths = new ArrayList<>();
386    for (int i = 1; i <= numFiles; i++) {
387      Path hfile = new Path(baseDir, "file" + i);
388      TestHRegionServerBulkLoad.createHFile(
389          fs, hfile, Bytes.toBytes(SpaceQuotaHelperForTests.F1), Bytes.toBytes("my"),
390          Bytes.toBytes("file"), numRowsPerFile);
391      famPaths.add(new Pair<>(Bytes.toBytes(SpaceQuotaHelperForTests.F1), hfile.toString()));
392    }
393
394    // bulk load HFiles
395    Table table = conn.getTable(tn);
396    final String bulkToken = new SecureBulkLoadClient(conf, table).prepareBulkLoad(conn);
397    return new ClientServiceCallable<Boolean>(
398        conn, tn, Bytes.toBytes("row"), new RpcControllerFactory(conf).newController(),
399        HConstants.PRIORITY_UNSET) {
400      @Override
401     public Boolean rpcCall() throws Exception {
402        SecureBulkLoadClient secureClient = null;
403        byte[] regionName = getLocation().getRegion().getRegionName();
404        try (Table table = conn.getTable(getTableName())) {
405          secureClient = new SecureBulkLoadClient(conf, table);
406          return secureClient.secureBulkLoadHFiles(getStub(), famPaths, regionName,
407                true, null, bulkToken);
408        }
409      }
410    };
411  }

413  /**
414   * Abstraction to simplify the case where a test needs to verify a certain state
415   * on a {@code SpaceQuotaSnapshot}. This class fails-fast when there is no such
416   * snapshot obtained from the Master. As such, it is not useful to verify the
417   * lack of a snapshot.
418   */
419  static abstract class SpaceQuotaSnapshotPredicate implements Predicate<Exception> {
420    private final Connection conn;
421    private final TableName tn;
422    private final String ns;
423
424    SpaceQuotaSnapshotPredicate(Connection conn, TableName tn) {
425      this(Objects.requireNonNull(conn), Objects.requireNonNull(tn), null);
426    }
427
428    SpaceQuotaSnapshotPredicate(Connection conn, String ns) {
429      this(Objects.requireNonNull(conn), null, Objects.requireNonNull(ns));
430    }
431
432    SpaceQuotaSnapshotPredicate(Connection conn, TableName tn, String ns) {
433      if ((null != tn && null != ns) || (null == tn && null == ns)) {
434        throw new IllegalArgumentException(
435            "One of TableName and Namespace must be non-null, and the other null");
436      }
437      this.conn = conn;
438      this.tn = tn;
439      this.ns = ns;
440    }
441
442    @Override
443    public boolean evaluate() throws Exception {
444      SpaceQuotaSnapshot snapshot;
445      if (null == ns) {
446        snapshot = QuotaTableUtil.getCurrentSnapshot(conn, tn);
447      } else {
448        snapshot = QuotaTableUtil.getCurrentSnapshot(conn, ns);
449      }
450
451      LOG.debug("Saw quota snapshot for " + (null == tn ? ns : tn) + ": " + snapshot);
452      if (null == snapshot) {
453        return false;
454      }
455      return evaluate(snapshot);
456    }
457
458    /**
459     * Must determine if the given {@code SpaceQuotaSnapshot} meets some criteria.
460     *
461     * @param snapshot a non-null snapshot obtained from the HBase Master
462     * @return true if the criteria is met, false otherwise
463     */
464    abstract boolean evaluate(SpaceQuotaSnapshot snapshot) throws Exception;
465  }

467  /**
468   * Predicate that waits for all store files in a table to have no compacted files.
469   */
470  static class NoFilesToDischarge implements Predicate<Exception> {
471    private final MiniHBaseCluster cluster;
472    private final TableName tn;
473
474    NoFilesToDischarge(MiniHBaseCluster cluster, TableName tn) {
475      this.cluster = cluster;
476      this.tn = tn;
477    }
478
479    @Override
480    public boolean evaluate() throws Exception {
481      for (HRegion region : cluster.getRegions(tn)) {
482        for (HStore store : region.getStores()) {
483          Collection<HStoreFile> files =
484              store.getStoreEngine().getStoreFileManager().getCompactedfiles();
485          if (null != files && !files.isEmpty()) {
486            LOG.debug(region.getRegionInfo().getEncodedName() + " still has compacted files");
487            return false;
488          }
489        }
490      }
491      return true;
492    }
493  }
