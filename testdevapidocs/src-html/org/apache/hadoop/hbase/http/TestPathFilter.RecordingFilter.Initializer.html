020import java.io.BufferedReader;
021import java.io.IOException;
022import java.io.InputStreamReader;
023import java.net.URL;
024import java.net.URLConnection;
025import java.util.Set;
026import java.util.TreeSet;
027
028import javax.servlet.Filter;
029import javax.servlet.FilterChain;
030import javax.servlet.FilterConfig;
031import javax.servlet.ServletException;
032import javax.servlet.ServletRequest;
033import javax.servlet.ServletResponse;
034import javax.servlet.http.HttpServletRequest;

036import org.apache.commons.logging.Log;
037import org.apache.commons.logging.LogFactory;
038import org.apache.hadoop.conf.Configuration;
039import org.apache.hadoop.hbase.testclassification.MiscTests;
040import org.apache.hadoop.hbase.testclassification.SmallTests;
041import org.apache.hadoop.net.NetUtils;
042import org.junit.Test;
043import org.junit.experimental.categories.Category;
044
045@Category({MiscTests.class, SmallTests.class})
046public class TestPathFilter extends HttpServerFunctionalTest {
047  private static final Log LOG = LogFactory.getLog(HttpServer.class);
048  static final Set<String> RECORDS = new TreeSet<>();
049
050  /** A very simple filter that records accessed uri's */
051  static public class RecordingFilter implements Filter {
052    private FilterConfig filterConfig = null;
053
054    @Override
055    public void init(FilterConfig filterConfig) {
056      this.filterConfig = filterConfig;
057    }
058
059    @Override
060    public void destroy() {
061      this.filterConfig = null;
062    }
063
064    @Override
065    public void doFilter(ServletRequest request, ServletResponse response,
066        FilterChain chain) throws IOException, ServletException {
067      if (filterConfig == null)
068         return;
069
070      String uri = ((HttpServletRequest)request).getRequestURI();
071      LOG.info("filtering " + uri);
072      RECORDS.add(uri);
073      chain.doFilter(request, response);
074    }
075
076    /** Configuration for RecordingFilter */
077    static public class Initializer extends FilterInitializer {
078      public Initializer() {}
079
080      @Override
081      public void initFilter(FilterContainer container, Configuration conf) {
082        container.addFilter("recording", RecordingFilter.class.getName(), null);
083      }
084    }
085  }
086
087
088  /** access a url, ignoring some IOException such as the page does not exist */
089  static void access(String urlstring) throws IOException {
090    LOG.warn("access " + urlstring);
091    URL url = new URL(urlstring);
092
093    URLConnection connection = url.openConnection();
094    connection.connect();
095
096    try {
097      BufferedReader in = new BufferedReader(new InputStreamReader(
098          connection.getInputStream()));
099      try {
100        for(; in.readLine() != null; );
101      } finally {
102        in.close();
103      }
104    } catch(IOException ioe) {
105      LOG.warn("urlstring=" + urlstring, ioe);
106    }
107  }
108
109  @Test
110  public void testPathSpecFilters() throws Exception {
111    Configuration conf = new Configuration();
112
113    //start a http server with CountingFilter
114    conf.set(HttpServer.FILTER_INITIALIZERS_PROPERTY,
115        RecordingFilter.Initializer.class.getName());
116    String[] pathSpecs = { "/path", "/path/*" };
117    HttpServer http = createTestServer(conf, pathSpecs);
118    http.start();
119
120    final String baseURL = "/path";
121    final String baseSlashURL = "/path/";
122    final String addedURL = "/path/nodes";
123    final String addedSlashURL = "/path/nodes/";
124    final String longURL = "/path/nodes/foo/job";
125    final String rootURL = "/";
126    final String allURL = "/*";
127
128    final String[] filteredUrls = {baseURL, baseSlashURL, addedURL,
129        addedSlashURL, longURL};
130    final String[] notFilteredUrls = {rootURL, allURL};
131
132    // access the urls and verify our paths specs got added to the
133    // filters
134    final String prefix = "http://"
135        + NetUtils.getHostPortString(http.getConnectorAddress(0));
136    try {
137      for(int i = 0; i < filteredUrls.length; i++) {
138        access(prefix + filteredUrls[i]);
139      }
140      for(int i = 0; i < notFilteredUrls.length; i++) {
141        access(prefix + notFilteredUrls[i]);
142      }
143    } finally {
144      http.stop();
145    }
146
147    LOG.info("RECORDS = " + RECORDS);
148
149    //verify records
150    for(int i = 0; i < filteredUrls.length; i++) {
151      assertTrue(RECORDS.remove(filteredUrls[i]));
152    }
153    assertTrue(RECORDS.isEmpty());
154  }
155}
