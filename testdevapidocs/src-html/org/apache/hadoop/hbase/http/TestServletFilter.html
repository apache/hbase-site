046@Category({MiscTests.class, SmallTests.class})
047public class TestServletFilter extends HttpServerFunctionalTest {
048  private static final Log LOG = LogFactory.getLog(HttpServer.class);
049  static volatile String uri = null; 
050
051  /** A very simple filter which record the uri filtered. */
052  static public class SimpleFilter implements Filter {
053    private FilterConfig filterConfig = null;
054
055    @Override
056    public void init(FilterConfig filterConfig) throws ServletException {
057      this.filterConfig = filterConfig;
058    }
059
060    @Override
061    public void destroy() {
062      this.filterConfig = null;
063    }
064
065    @Override
066    public void doFilter(ServletRequest request, ServletResponse response,
067        FilterChain chain) throws IOException, ServletException {
068      if (filterConfig == null)
069         return;
070
071      uri = ((HttpServletRequest)request).getRequestURI();
072      LOG.info("filtering " + uri);
073      chain.doFilter(request, response);
074    }
075
076    /** Configuration for the filter */
077    static public class Initializer extends FilterInitializer {
078      public Initializer() {}
079
080      @Override
081      public void initFilter(FilterContainer container, Configuration conf) {
082        container.addFilter("simple", SimpleFilter.class.getName(), null);
083      }
084    }
085  }
086  
087  
088  /** access a url, ignoring some IOException such as the page does not exist */
089  static void access(String urlstring) throws IOException {
090    LOG.warn("access " + urlstring);
091    URL url = new URL(urlstring);
092    URLConnection connection = url.openConnection();
093    connection.connect();
094    
095    try {
096      BufferedReader in = new BufferedReader(new InputStreamReader(
097          connection.getInputStream()));
098      try {
099        for(; in.readLine() != null; );
100      } finally {
101        in.close();
102      }
103    } catch(IOException ioe) {
104      LOG.warn("urlstring=" + urlstring, ioe);
105    }
106  }

108  @Test
109  @Ignore
110  //From stack
111  // Its a 'foreign' test, one that came in from hadoop when we copy/pasted http
112  // It's second class. Could comment it out if only failing test (as per @nkeywal â€“ sort of)
113  public void testServletFilter() throws Exception {
114    Configuration conf = new Configuration();
115    
116    //start a http server with CountingFilter
117    conf.set(HttpServer.FILTER_INITIALIZERS_PROPERTY,
118        SimpleFilter.Initializer.class.getName());
119    HttpServer http = createTestServer(conf);
120    http.start();
121
122    final String fsckURL = "/fsck";
123    final String stacksURL = "/stacks";
124    final String ajspURL = "/a.jsp";
125    final String logURL = "/logs/a.log";
126    final String hadooplogoURL = "/static/hadoop-logo.jpg";
127    
128    final String[] urls = {fsckURL, stacksURL, ajspURL, logURL, hadooplogoURL};
129    final Random ran = new Random();
130    final int[] sequence = new int[50];
131
132    //generate a random sequence and update counts 
133    for(int i = 0; i < sequence.length; i++) {
134      sequence[i] = ran.nextInt(urls.length);
135    }
136
137    //access the urls as the sequence
138    final String prefix = "http://"
139        + NetUtils.getHostPortString(http.getConnectorAddress(0));
140    try {
141      for(int i = 0; i < sequence.length; i++) {
142        access(prefix + urls[sequence[i]]);
143
144        //make sure everything except fsck get filtered
145        if (sequence[i] == 0) {
146          assertEquals(null, uri);
147        } else {
148          assertEquals(urls[sequence[i]], uri);
149          uri = null;
150        }
151      }
152    } finally {
153      http.stop();
154    }
155  }

159    public void init(FilterConfig arg0) throws ServletException {
160      throw new ServletException("Throwing the exception from Filter init");
161    }
162
163    /** Configuration for the filter */
164    static public class Initializer extends FilterInitializer {
165      public Initializer() {
166      }
167
168      @Override
169      public void initFilter(FilterContainer container, Configuration conf) {
170        container.addFilter("simple", ErrorFilter.class.getName(), null);
171      }
172    }
173  }

175  @Test
176  public void testServletFilterWhenInitThrowsException() throws Exception {
177    Configuration conf = new Configuration();
178    // start a http server with ErrorFilter
179    conf.set(HttpServer.FILTER_INITIALIZERS_PROPERTY,
180        ErrorFilter.Initializer.class.getName());
181    HttpServer http = createTestServer(conf);
182    try {
183      http.start();
184      fail("expecting exception");
185    } catch (IOException e) {
186      assertTrue( e.getMessage().contains("Problem in starting http server. Server handlers failed"));
187    }
188  }
