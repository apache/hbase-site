113  /**
114   * Sets up {@link MiniKdc} for testing security.
115   * Copied from HBaseTestingUtility#setupMiniKdc().
116   */
117  static private MiniKdc setupMiniKdc() throws Exception {
118    Properties conf = MiniKdc.createConf();
119    conf.put(MiniKdc.DEBUG, true);
120    MiniKdc kdc = null;
121    File dir = null;
122    // There is time lag between selecting a port and trying to bind with it. It's possible that
123    // another service captures the port in between which'll result in BindException.
124    boolean bindException;
125    int numTries = 0;
126    do {
127      try {
128        bindException = false;
129        dir = new File(htu.getDataTestDir("kdc").toUri().getPath());
130        kdc = new MiniKdc(conf, dir);
131        kdc.start();
132      } catch (BindException e) {
133        FileUtils.deleteDirectory(dir);  // clean directory
134        numTries++;
135        if (numTries == 3) {
136          log.error("Failed setting up MiniKDC. Tried " + numTries + " times.");
137          throw e;
138        }
139        log.error("BindException encountered when setting up MiniKdc. Trying again.");
140        bindException = true;
141      }
142    } while (bindException);
143    return kdc;
144  }

146  static private void setupSSL(File base) throws Exception {
147    Configuration conf = new Configuration();
148    conf.set(DFSConfigKeys.DFS_HTTP_POLICY_KEY, HttpConfig.Policy.HTTPS_ONLY.name());
149    conf.set(DFSConfigKeys.DFS_NAMENODE_HTTPS_ADDRESS_KEY, "localhost:0");
150    conf.set(DFSConfigKeys.DFS_DATANODE_HTTPS_ADDRESS_KEY, "localhost:0");
151
152    keystoresDir = base.getAbsolutePath();
153    sslConfDir = KeyStoreTestUtil.getClasspathDir(TestLogLevel.class);
154    KeyStoreTestUtil.setupSSLConfig(keystoresDir, sslConfDir, conf, false);
155
156    sslConf = getSslConfig();
157  }

159  /**
160   * Get the SSL configuration.
161   * This method is copied from KeyStoreTestUtil#getSslConfig() in Hadoop.
162   * @return {@link Configuration} instance with ssl configs loaded.
163   */
164  private static Configuration getSslConfig(){
165    Configuration sslConf = new Configuration(false);
166    String sslServerConfFile = "ssl-server.xml";
167    String sslClientConfFile = "ssl-client.xml";
168    sslConf.addResource(sslServerConfFile);
169    sslConf.addResource(sslClientConfFile);
170    sslConf.set(SSLFactory.SSL_SERVER_CONF_KEY, sslServerConfFile);
171    sslConf.set(SSLFactory.SSL_CLIENT_CONF_KEY, sslClientConfFile);
172    return sslConf;
173  }

184  /**
185   * Test client command line options. Does not validate server behavior.
186   * @throws Exception if commands return unexpected results.
187   */
188  @Test
189  public void testCommandOptions() throws Exception {
190    final String className = this.getClass().getName();
191
192    assertFalse(validateCommand(new String[] {"-foo" }));
193    // fail due to insufficient number of arguments
194    assertFalse(validateCommand(new String[] {}));
195    assertFalse(validateCommand(new String[] {"-getlevel" }));
196    assertFalse(validateCommand(new String[] {"-setlevel" }));
197    assertFalse(validateCommand(new String[] {"-getlevel", "foo.bar:8080" }));
198
199    // valid command arguments
200    assertTrue(validateCommand(
201        new String[] {"-getlevel", "foo.bar:8080", className }));
202    assertTrue(validateCommand(
203        new String[] {"-setlevel", "foo.bar:8080", className, "DEBUG" }));
204    assertTrue(validateCommand(
205        new String[] {"-getlevel", "foo.bar:8080", className }));
206    assertTrue(validateCommand(
207        new String[] {"-setlevel", "foo.bar:8080", className, "DEBUG" }));
208
209    // fail due to the extra argument
210    assertFalse(validateCommand(
211        new String[] {"-getlevel", "foo.bar:8080", className, "blah" }));
212    assertFalse(validateCommand(
213        new String[] {"-setlevel", "foo.bar:8080", className, "DEBUG", "blah" }));
214    assertFalse(validateCommand(
215        new String[] {"-getlevel", "foo.bar:8080", className, "-setlevel", "foo.bar:8080",
216          className }));
217  }

219  /**
220   * Check to see if a command can be accepted.
221   *
222   * @param args a String array of arguments
223   * @return true if the command can be accepted, false if not.
224   */
225  private boolean validateCommand(String[] args) {
226    CLI cli = new CLI(clientConf);
227    try {
228      cli.parseArguments(args);
229    } catch (HadoopIllegalArgumentException e) {
230      return false;
231    } catch (Exception e) {
232      // this is used to verify the command arguments only.
233      // no HadoopIllegalArgumentException = the arguments are good.
234      return true;
235    }
236    return true;
237  }

239  /**
240   * Creates and starts a Jetty server binding at an ephemeral port to run
241   * LogLevel servlet.
242   * @param protocol "http" or "https"
243   * @param isSpnego true if SPNEGO is enabled
244   * @return a created HttpServer object
245   * @throws Exception if unable to create or start a Jetty server
246   */
247  private HttpServer createServer(String protocol, boolean isSpnego)
248      throws Exception {
249    HttpServer.Builder builder = new HttpServer.Builder()
250        .setName("..")
251        .addEndpoint(new URI(protocol + "://localhost:0"))
252        .setFindPort(true)
253        .setConf(serverConf);
254    if (isSpnego) {
255      // Set up server Kerberos credentials.
256      // Since the server may fall back to simple authentication,
257      // use ACL to make sure the connection is Kerberos/SPNEGO authenticated.
258      builder.setSecurityEnabled(true)
259          .setUsernameConfKey(PRINCIPAL)
260          .setKeytabConfKey(KEYTAB)
261          .setACL(new AccessControlList("client"));
262    }
263
264    // if using HTTPS, configure keystore/truststore properties.
265    if (protocol.equals(LogLevel.PROTOCOL_HTTPS)) {
266      builder = builder.
267          keyPassword(sslConf.get("ssl.server.keystore.keypassword"))
268          .keyStore(sslConf.get("ssl.server.keystore.location"),
269              sslConf.get("ssl.server.keystore.password"),
270              sslConf.get("ssl.server.keystore.type", "jks"))
271          .trustStore(sslConf.get("ssl.server.truststore.location"),
272              sslConf.get("ssl.server.truststore.password"),
273              sslConf.get("ssl.server.truststore.type", "jks"));
274    }
275
276    HttpServer server = builder.build();
277    server.start();
278    return server;
279  }

287  /**
288   * Run both client and server using the given protocol.
289   *
290   * @param bindProtocol specify either http or https for server
291   * @param connectProtocol specify either http or https for client
292   * @param isSpnego true if SPNEGO is enabled
293   * @throws Exception if client can't accesss server.
294   */
295  private void testDynamicLogLevel(final String bindProtocol, final String connectProtocol,
296      final boolean isSpnego, final String newLevel)
297      throws Exception {
298    if (!LogLevel.isValidProtocol(bindProtocol)) {
299      throw new Exception("Invalid server protocol " + bindProtocol);
300    }
301    if (!LogLevel.isValidProtocol(connectProtocol)) {
302      throw new Exception("Invalid client protocol " + connectProtocol);
303    }
304    Level oldLevel = log.getEffectiveLevel();
305    assertNotEquals("Get default Log Level which shouldn't be ERROR.",
306        Level.ERROR, oldLevel);
307
308    // configs needed for SPNEGO at server side
309    if (isSpnego) {
310      serverConf.set(PRINCIPAL, HTTP_PRINCIPAL);
311      serverConf.set(KEYTAB, KEYTAB_FILE.getAbsolutePath());
312      serverConf.set(CommonConfigurationKeysPublic.HADOOP_SECURITY_AUTHENTICATION, "kerberos");
313      serverConf.setBoolean(CommonConfigurationKeys.HADOOP_SECURITY_AUTHORIZATION, true);
314      UserGroupInformation.setConfiguration(serverConf);
315    } else {
316      serverConf.set(CommonConfigurationKeysPublic.HADOOP_SECURITY_AUTHENTICATION, "simple");
317      serverConf.setBoolean(CommonConfigurationKeys.HADOOP_SECURITY_AUTHORIZATION, false);
318      UserGroupInformation.setConfiguration(serverConf);
319    }
320
321    final HttpServer server = createServer(bindProtocol, isSpnego);
322    // get server port
323    final String authority = NetUtils.getHostPortString(server.getConnectorAddress(0));
324
325    String keytabFilePath = KEYTAB_FILE.getAbsolutePath();
326
327    UserGroupInformation clientUGI = UserGroupInformation.
328        loginUserFromKeytabAndReturnUGI(clientPrincipal, keytabFilePath);
329    try {
330      clientUGI.doAs((PrivilegedExceptionAction<Void>) () -> {
331        // client command line
332        getLevel(connectProtocol, authority);
333        setLevel(connectProtocol, authority, newLevel);
334        return null;
335      });
336    } finally {
337      clientUGI.logoutUserFromKeytab();
338      server.stop();
339    }
340
341    // restore log level
342    GenericTestUtils.setLogLevel(log, oldLevel);
343  }

347   * class.
348   *
349   * @param protocol specify either http or https
350   * @param authority daemon's web UI address
351   * @throws Exception if unable to connect

353  private void getLevel(String protocol, String authority) throws Exception {

359  /**
360   * Run LogLevel command line to start a client to set log level of this test
361   * class to debug.
362   *
363   * @param protocol specify either http or https
364   * @param authority daemon's web UI address
365   * @throws Exception if unable to run or log level does not change as expected
366   */
367  private void setLevel(String protocol, String authority, String newLevel)
368      throws Exception {
369    String[] setLevelArgs = {"-setlevel", authority, logName, newLevel, "-protocol", protocol};
370    CLI cli = new CLI(clientConf);
371    cli.run(setLevelArgs);
372
373    assertEquals("new level not equal to expected: ", newLevel.toUpperCase(),
374        log.getEffectiveLevel().toString());
375  }
