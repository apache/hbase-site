095  @Test
096  public void testRegionSplit() throws Exception {
097    TableName tableName = TableName.valueOf(name.getMethodName());
098    UTIL.getAdmin().createTable(
099      TableDescriptorBuilder.newBuilder(tableName).setColumnFamily(ColumnFamilyDescriptorBuilder
100        .newBuilder(CF).setScope(HConstants.REPLICATION_SCOPE_GLOBAL).build()).build());
101    UTIL.waitTableAvailable(tableName);
102    try (Table table = UTIL.getConnection().getTable(tableName)) {
103      for (int i = 0; i < 100; i++) {
104        table.put(new Put(Bytes.toBytes(i)).addColumn(CF, CQ, Bytes.toBytes(i)));
105      }
106    }
107    UTIL.flush(tableName);
108    RegionInfo region = UTIL.getAdmin().getRegions(tableName).get(0);
109    UTIL.getAdmin().splitRegionAsync(region.getEncodedNameAsBytes(), Bytes.toBytes(50)).get(30,
110      TimeUnit.SECONDS);
111    UTIL.waitUntilNoRegionsInTransition(30000);
112    List<RegionInfo> regions = UTIL.getAdmin().getRegions(tableName);
113    assertEquals(2, regions.size());
114    try (Table table = UTIL.getConnection().getTable(tableName)) {
115      for (int i = 0; i < 100; i++) {
116        table.put(new Put(Bytes.toBytes(i)).addColumn(CF, CQ, Bytes.toBytes(i)));
117      }
118    }
119    enablePeerAndWaitUntilReplicationDone(200);
120    Map<String, Long> regionsToSeqId = new HashMap<>();
121    regionsToSeqId.put(region.getEncodedName(), -1L);
122    regions.stream().map(RegionInfo::getEncodedName).forEach(n -> regionsToSeqId.put(n, -1L));
123    try (WAL.Reader reader =
124      WALFactory.createReader(UTIL.getTestFileSystem(), logPath, UTIL.getConfiguration())) {
125      int count = 0;
126      for (Entry entry;;) {
127        entry = reader.next();
128        if (entry == null) {
129          break;
130        }
131        String encodedName = Bytes.toString(entry.getKey().getEncodedRegionName());
132        Long seqId = regionsToSeqId.get(encodedName);
133        assertNotNull(
134          "Unexcepted entry " + entry + ", expected regions " + region + ", or " + regions, seqId);
135        assertTrue("Sequence id go backwards from " + seqId + " to " +
136          entry.getKey().getSequenceId() + " for " + encodedName,
137          entry.getKey().getSequenceId() >= seqId.longValue());
138        if (count < 100) {
139          assertEquals(encodedName + " is pushed before parent " + region.getEncodedName(),
140            region.getEncodedName(), encodedName);
141        } else {
142          assertNotEquals(region.getEncodedName(), encodedName);
143        }
144        count++;
145      }
146      assertEquals(200, count);
147    }
148  }

150  @Test
151  public void testRegionMerge() throws Exception {
152    byte[] splitKey = Bytes.toBytes(50);
153    TableName tableName = TableName.valueOf(name.getMethodName());
154    UTIL.getAdmin().createTable(
155      TableDescriptorBuilder.newBuilder(tableName)
156        .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(CF)
157          .setScope(HConstants.REPLICATION_SCOPE_GLOBAL).build())
158        .build(),
159      new byte[][] { splitKey });
160    UTIL.waitTableAvailable(tableName);
161    try (Table table = UTIL.getConnection().getTable(tableName)) {
162      for (int i = 0; i < 100; i++) {
163        table.put(new Put(Bytes.toBytes(i)).addColumn(CF, CQ, Bytes.toBytes(i)));
164      }
165    }
166    List<RegionInfo> regions = UTIL.getAdmin().getRegions(tableName);
167    UTIL.getAdmin()
168      .mergeRegionsAsync(
169        regions.stream().map(RegionInfo::getEncodedNameAsBytes).toArray(byte[][]::new), false)
170      .get(30, TimeUnit.SECONDS);
171    UTIL.waitUntilNoRegionsInTransition(30000);
172    List<RegionInfo> regionsAfterMerge = UTIL.getAdmin().getRegions(tableName);
173    assertEquals(1, regionsAfterMerge.size());
174    try (Table table = UTIL.getConnection().getTable(tableName)) {
175      for (int i = 0; i < 100; i++) {
176        table.put(new Put(Bytes.toBytes(i)).addColumn(CF, CQ, Bytes.toBytes(i)));
177      }
178    }
179    enablePeerAndWaitUntilReplicationDone(200);
180    Map<String, Long> regionsToSeqId = new HashMap<>();
181    RegionInfo region = regionsAfterMerge.get(0);
182    regionsToSeqId.put(region.getEncodedName(), -1L);
183    regions.stream().map(RegionInfo::getEncodedName).forEach(n -> regionsToSeqId.put(n, -1L));
184    try (WAL.Reader reader =
185      WALFactory.createReader(UTIL.getTestFileSystem(), logPath, UTIL.getConfiguration())) {
186      int count = 0;
187      for (Entry entry;;) {
188        entry = reader.next();
189        if (entry == null) {
190          break;
191        }
192        String encodedName = Bytes.toString(entry.getKey().getEncodedRegionName());
193        Long seqId = regionsToSeqId.get(encodedName);
194        assertNotNull(
195          "Unexcepted entry " + entry + ", expected regions " + region + ", or " + regions, seqId);
196        assertTrue("Sequence id go backwards from " + seqId + " to " +
197          entry.getKey().getSequenceId() + " for " + encodedName,
198          entry.getKey().getSequenceId() >= seqId.longValue());
199        if (count < 100) {
200          assertNotEquals(
201            encodedName + " is pushed before parents " +
202              regions.stream().map(RegionInfo::getEncodedName).collect(Collectors.joining(" and ")),
203            region.getEncodedName(), encodedName);
204        } else {
205          assertEquals(region.getEncodedName(), encodedName);
206        }
207        count++;
208      }
209      assertEquals(200, count);
210    }
211  }
