046@Category({ ReplicationTests.class, MediumTests.class })
047public class TestZKReplicationPeerStorage {
048
049  private static final HBaseZKTestingUtility UTIL = new HBaseZKTestingUtility();
050
051  private static ZKReplicationPeerStorage STORAGE;
052
053  @BeforeClass
054  public static void setUp() throws Exception {
055    UTIL.startMiniZKCluster();
056    STORAGE = new ZKReplicationPeerStorage(UTIL.getZooKeeperWatcher(), UTIL.getConfiguration());
057  }
058
059  @AfterClass
060  public static void tearDown() throws IOException {
061    UTIL.shutdownMiniZKCluster();
062  }
063
064  private Set<String> randNamespaces(Random rand) {
065    return Stream.generate(() -> Long.toHexString(rand.nextLong())).limit(rand.nextInt(5))
066        .collect(toSet());
067  }
068
069  private Map<TableName, List<String>> randTableCFs(Random rand) {
070    int size = rand.nextInt(5);
071    Map<TableName, List<String>> map = new HashMap<>();
072    for (int i = 0; i < size; i++) {
073      TableName tn = TableName.valueOf(Long.toHexString(rand.nextLong()));
074      List<String> cfs = Stream.generate(() -> Long.toHexString(rand.nextLong()))
075          .limit(rand.nextInt(5)).collect(toList());
076      map.put(tn, cfs);
077    }
078    return map;
079  }
080
081  private ReplicationPeerConfig getConfig(int seed) {
082    Random rand = new Random(seed);
083    return ReplicationPeerConfig.newBuilder().setClusterKey(Long.toHexString(rand.nextLong()))
084        .setReplicationEndpointImpl(Long.toHexString(rand.nextLong()))
085        .setNamespaces(randNamespaces(rand)).setExcludeNamespaces(randNamespaces(rand))
086        .setTableCFsMap(randTableCFs(rand)).setReplicateAllUserTables(rand.nextBoolean())
087        .setBandwidth(rand.nextInt(1000)).build();
088  }
089
090  private void assertSetEquals(Set<String> expected, Set<String> actual) {
091    if (expected == null || expected.size() == 0) {
092      assertTrue(actual == null || actual.size() == 0);
093      return;
094    }
095    assertEquals(expected.size(), actual.size());
096    expected.forEach(s -> assertTrue(actual.contains(s)));
097  }
098
099  private void assertMapEquals(Map<TableName, List<String>> expected,
100      Map<TableName, List<String>> actual) {
101    if (expected == null || expected.size() == 0) {
102      assertTrue(actual == null || actual.size() == 0);
103      return;
104    }
105    assertEquals(expected.size(), actual.size());
106    expected.forEach((expectedTn, expectedCFs) -> {
107      List<String> actualCFs = actual.get(expectedTn);
108      if (expectedCFs == null || expectedCFs.size() == 0) {
109        assertTrue(actual.containsKey(expectedTn));
110        assertTrue(actualCFs == null || actualCFs.size() == 0);
111      } else {
112        assertNotNull(actualCFs);
113        assertEquals(expectedCFs.size(), actualCFs.size());
114        for (Iterator<String> expectedIt = expectedCFs.iterator(), actualIt = actualCFs.iterator();
115          expectedIt.hasNext();) {
116          assertEquals(expectedIt.next(), actualIt.next());
117        }
118      }
119    });
120  }
121
122  private void assertConfigEquals(ReplicationPeerConfig expected, ReplicationPeerConfig actual) {
123    assertEquals(expected.getClusterKey(), actual.getClusterKey());
124    assertEquals(expected.getReplicationEndpointImpl(), actual.getReplicationEndpointImpl());
125    assertSetEquals(expected.getNamespaces(), actual.getNamespaces());
126    assertSetEquals(expected.getExcludeNamespaces(), actual.getExcludeNamespaces());
127    assertMapEquals(expected.getTableCFsMap(), actual.getTableCFsMap());
128    assertMapEquals(expected.getExcludeTableCFsMap(), actual.getExcludeTableCFsMap());
129    assertEquals(expected.replicateAllUserTables(), actual.replicateAllUserTables());
130    assertEquals(expected.getBandwidth(), actual.getBandwidth());
131  }
132
133  @Test
134  public void test() throws ReplicationException {
135    int peerCount = 10;
136    for (int i = 0; i < peerCount; i++) {
137      STORAGE.addPeer(Integer.toString(i), getConfig(i), i % 2 == 0);
138    }
139    List<String> peerIds = STORAGE.listPeerIds();
140    assertEquals(peerCount, peerIds.size());
141    for (String peerId : peerIds) {
142      int seed = Integer.parseInt(peerId);
143      assertConfigEquals(getConfig(seed), STORAGE.getPeerConfig(peerId));
144    }
145    for (int i = 0; i < peerCount; i++) {
146      STORAGE.updatePeerConfig(Integer.toString(i), getConfig(i + 1));
147    }
148    for (String peerId : peerIds) {
149      int seed = Integer.parseInt(peerId);
150      assertConfigEquals(getConfig(seed + 1), STORAGE.getPeerConfig(peerId));
151    }
152    for (int i = 0; i < peerCount; i++) {
153      assertEquals(i % 2 == 0, STORAGE.isPeerEnabled(Integer.toString(i)));
154    }
155    for (int i = 0; i < peerCount; i++) {
156      STORAGE.setPeerState(Integer.toString(i), i % 2 != 0);
157    }
158    for (int i = 0; i < peerCount; i++) {
159      assertEquals(i % 2 != 0, STORAGE.isPeerEnabled(Integer.toString(i)));
160    }
161    String toRemove = Integer.toString(peerCount / 2);
162    STORAGE.removePeer(toRemove);
163    peerIds = STORAGE.listPeerIds();
164    assertEquals(peerCount - 1, peerIds.size());
165    assertFalse(peerIds.contains(toRemove));
166
167    try {
168      STORAGE.getPeerConfig(toRemove);
169      fail("Should throw a ReplicationException when get peer config of a peerId");
170    } catch (ReplicationException e) {
171    }
172  }
