268  /**
269   * Test replicateEntries with a bulk load entry for 25 HFiles
270   */
271  @Test
272  public void testReplicateEntriesForHFiles() throws Exception {
273    Path dir = TEST_UTIL.getDataTestDirOnTestFS("testReplicateEntries");
274    Path familyDir = new Path(dir, Bytes.toString(FAM_NAME1));
275    int numRows = 10;
276
277    List<Path> p = new ArrayList<>(1);
278
279    // 1. Generate 25 hfile ranges
280    Random rng = new SecureRandom();
281    Set<Integer> numbers = new HashSet<>();
282    while (numbers.size() < 50) {
283      numbers.add(rng.nextInt(1000));
284    }
285    List<Integer> numberList = new ArrayList<>(numbers);
286    Collections.sort(numberList);
287    Map<String, Long> storeFilesSize = new HashMap<String, Long>(1);
288
289    // 2. Create 25 hfiles
290    Configuration conf = TEST_UTIL.getConfiguration();
291    FileSystem fs = dir.getFileSystem(conf);
292    Iterator<Integer> numbersItr = numberList.iterator();
293    for (int i = 0; i < 25; i++) {
294      Path hfilePath = new Path(familyDir, "hfile_" + i);
295      HFileTestUtil.createHFile(conf, fs, hfilePath, FAM_NAME1, FAM_NAME1,
296        Bytes.toBytes(numbersItr.next()), Bytes.toBytes(numbersItr.next()), numRows);
297      p.add(hfilePath);
298      storeFilesSize.put(hfilePath.getName(), fs.getFileStatus(hfilePath).getLen());
299    }
300
301    // 3. Create a BulkLoadDescriptor and a WALEdit
302    Map<byte[], List<Path>> storeFiles = new HashMap<>(1);
303    storeFiles.put(FAM_NAME1, p);
304    WALEdit edit = null;
305    WALProtos.BulkLoadDescriptor loadDescriptor = null;
306
307    try (Connection c = ConnectionFactory.createConnection(conf);
308        RegionLocator l = c.getRegionLocator(TABLE_NAME1)) {
309      HRegionInfo regionInfo = l.getAllRegionLocations().get(0).getRegionInfo();
310      loadDescriptor =
311          ProtobufUtil.toBulkLoadDescriptor(TABLE_NAME1,
312            ByteStringer.wrap(regionInfo.getEncodedNameAsBytes()), storeFiles, storeFilesSize, 1);
313      edit = WALEdit.createBulkLoadEvent(regionInfo, loadDescriptor);
314    }
315    List<WALEntry> entries = new ArrayList<WALEntry>(1);
316
317    // 4. Create a WALEntryBuilder
318    WALEntry.Builder builder = createWALEntryBuilder(TABLE_NAME1);
319
320    // 5. Copy the hfile to the path as it is in reality
321    for (int i = 0; i < 25; i++) {
322      String pathToHfileFromNS =
323          new StringBuilder(100).append(TABLE_NAME1.getNamespaceAsString()).append(Path.SEPARATOR)
324              .append(Bytes.toString(TABLE_NAME1.getName())).append(Path.SEPARATOR)
325              .append(Bytes.toString(loadDescriptor.getEncodedRegionName().toByteArray()))
326              .append(Path.SEPARATOR).append(Bytes.toString(FAM_NAME1)).append(Path.SEPARATOR)
327              .append("hfile_" + i).toString();
328      String dst = baseNamespaceDir + Path.SEPARATOR + pathToHfileFromNS;
329
330      FileUtil.copy(fs, p.get(0), fs, new Path(dst), false, conf);
331    }
332
333    entries.add(builder.build());
334    ResultScanner scanRes = null;
335    try {
336      Scan scan = new Scan();
337      scanRes = table1.getScanner(scan);
338      // 6. Assert no existing data in table
339      assertEquals(0, scanRes.next(numRows).length);
340      // 7. Replicate the bulk loaded entry
341      SINK.replicateEntries(entries, CellUtil.createCellScanner(edit.getCells().iterator()),
342        replicationClusterId, baseNamespaceDir, hfileArchiveDir);
343      scanRes = table1.getScanner(scan);
344      // 8. Assert data is replicated
345      assertEquals(numRows, scanRes.next(numRows).length);
346    } finally {
347      if (scanRes != null) {
348        scanRes.close();
349      }
350    }
351  }

353  private WALEntry createEntry(TableName table, int row,  KeyValue.Type type, List<Cell> cells) {
354    byte[] fam = table.equals(TABLE_NAME1) ? FAM_NAME1 : FAM_NAME2;
355    byte[] rowBytes = Bytes.toBytes(row);
356    // Just make sure we don't get the same ts for two consecutive rows with
357    // same key
358    try {
359      Thread.sleep(1);
360    } catch (InterruptedException e) {
361      LOG.info("Was interrupted while sleep, meh", e);
362    }
363    final long now = System.currentTimeMillis();
364    KeyValue kv = null;
365    if(type.getCode() == KeyValue.Type.Put.getCode()) {
366      kv = new KeyValue(rowBytes, fam, fam, now,
367          KeyValue.Type.Put, Bytes.toBytes(row));
368    } else if (type.getCode() == KeyValue.Type.DeleteColumn.getCode()) {
369        kv = new KeyValue(rowBytes, fam, fam,
370            now, KeyValue.Type.DeleteColumn);
371    } else if (type.getCode() == KeyValue.Type.DeleteFamily.getCode()) {
372        kv = new KeyValue(rowBytes, fam, null,
373            now, KeyValue.Type.DeleteFamily);
374    }
375    WALEntry.Builder builder = createWALEntryBuilder(table);
376    cells.add(kv);
377
378    return builder.build();
379  }
380
381  private WALEntry.Builder createWALEntryBuilder(TableName table) {
382    WALEntry.Builder builder = WALEntry.newBuilder();
383    builder.setAssociatedCellCount(1);
384    WALKey.Builder keyBuilder = WALKey.newBuilder();
385    UUID.Builder uuidBuilder = UUID.newBuilder();
386    uuidBuilder.setLeastSigBits(HConstants.DEFAULT_CLUSTER_ID.getLeastSignificantBits());
387    uuidBuilder.setMostSigBits(HConstants.DEFAULT_CLUSTER_ID.getMostSignificantBits());
388    keyBuilder.setClusterId(uuidBuilder.build());
389    keyBuilder.setTableName(ByteStringer.wrap(table.getName()));
390    keyBuilder.setWriteTime(System.currentTimeMillis());
391    keyBuilder.setEncodedRegionName(ByteStringer.wrap(HConstants.EMPTY_BYTE_ARRAY));
392    keyBuilder.setLogSequenceNumber(-1);
393    builder.setKey(keyBuilder.build());
394    return builder;
395  }
