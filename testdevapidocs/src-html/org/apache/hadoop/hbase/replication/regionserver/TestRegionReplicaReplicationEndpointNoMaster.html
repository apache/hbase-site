159  public static class WALEditCopro implements WALCoprocessor, WALObserver {
160    public WALEditCopro() {
161      entries.clear();
162    }
163
164    @Override
165    public Optional<WALObserver> getWALObserver() {
166      return Optional.of(this);
167    }
168
169    @Override
170    public void postWALWrite(ObserverContext<? extends WALCoprocessorEnvironment> ctx,
171                             RegionInfo info, WALKey logKey, WALEdit logEdit) throws IOException {
172      // only keep primary region's edits
173      if (logKey.getTableName().equals(tableName) && info.getReplicaId() == 0) {
174        // Presume type is a WALKeyImpl
175        entries.add(new Entry((WALKeyImpl)logKey, logEdit));
176      }
177    }
178  }

180  @Test
181  public void testReplayCallable() throws Exception {
182    // tests replaying the edits to a secondary region replica using the Callable directly
183    openRegion(HTU, rs0, hriSecondary);
184
185    // load some data to primary
186    HTU.loadNumericRows(table, f, 0, 1000);
187
188    Assert.assertEquals(1000, entries.size());
189    try (AsyncClusterConnection conn = ClusterConnectionFactory
190      .createAsyncClusterConnection(HTU.getConfiguration(), null, User.getCurrent())) {
191      // replay the edits to the secondary using replay callable
192      replicateUsingCallable(conn, entries);
193    }
194
195    Region region = rs0.getRegion(hriSecondary.getEncodedName());
196    HTU.verifyNumericRows(region, f, 0, 1000);
197
198    HTU.deleteNumericRows(table, f, 0, 1000);
199    closeRegion(HTU, rs0, hriSecondary);
200  }
201
202  private void replicateUsingCallable(AsyncClusterConnection connection, Queue<Entry> entries)
203      throws IOException, ExecutionException, InterruptedException {
204    Entry entry;
205    while ((entry = entries.poll()) != null) {
206      byte[] row = CellUtil.cloneRow(entry.getEdit().getCells().get(0));
207      RegionLocations locations = connection.getRegionLocations(tableName, row, true).get();
208      connection
209        .replay(tableName, locations.getRegionLocation(1).getRegion().getEncodedNameAsBytes(), row,
210          Collections.singletonList(entry), 1, Integer.MAX_VALUE, TimeUnit.SECONDS.toNanos(10))
211        .get();
212    }
213  }

215  @Test
216  public void testReplayCallableWithRegionMove() throws Exception {
217    // tests replaying the edits to a secondary region replica using the Callable directly while
218    // the region is moved to another location.It tests handling of RME.
219    try (AsyncClusterConnection conn = ClusterConnectionFactory
220      .createAsyncClusterConnection(HTU.getConfiguration(), null, User.getCurrent())) {
221      openRegion(HTU, rs0, hriSecondary);
222      // load some data to primary
223      HTU.loadNumericRows(table, f, 0, 1000);
224
225      Assert.assertEquals(1000, entries.size());
226
227      // replay the edits to the secondary using replay callable
228      replicateUsingCallable(conn, entries);
229
230      Region region = rs0.getRegion(hriSecondary.getEncodedName());
231      HTU.verifyNumericRows(region, f, 0, 1000);
232
233      HTU.loadNumericRows(table, f, 1000, 2000); // load some more data to primary
234
235      // move the secondary region from RS0 to RS1
236      closeRegion(HTU, rs0, hriSecondary);
237      openRegion(HTU, rs1, hriSecondary);
238
239      // replicate the new data
240      replicateUsingCallable(conn, entries);
241
242      region = rs1.getRegion(hriSecondary.getEncodedName());
243      // verify the new data. old data may or may not be there
244      HTU.verifyNumericRows(region, f, 1000, 2000);
245
246      HTU.deleteNumericRows(table, f, 0, 2000);
247      closeRegion(HTU, rs1, hriSecondary);
248    }
249  }

251  @Test
252  public void testRegionReplicaReplicationEndpointReplicate() throws Exception {
253    // tests replaying the edits to a secondary region replica using the RRRE.replicate()
254    openRegion(HTU, rs0, hriSecondary);
255    RegionReplicaReplicationEndpoint replicator = new RegionReplicaReplicationEndpoint();
256
257    ReplicationEndpoint.Context context = mock(ReplicationEndpoint.Context.class);
258    when(context.getConfiguration()).thenReturn(HTU.getConfiguration());
259    when(context.getMetrics()).thenReturn(mock(MetricsSource.class));
260    when(context.getServer()).thenReturn(rs0);
261    when(context.getTableDescriptors()).thenReturn(rs0.getTableDescriptors());
262    replicator.init(context);
263    replicator.startAsync();
264
265    //load some data to primary
266    HTU.loadNumericRows(table, f, 0, 1000);
267
268    Assert.assertEquals(1000, entries.size());
269    // replay the edits to the secondary using replay callable
270    final String fakeWalGroupId = "fakeWALGroup";
271    replicator.replicate(new ReplicateContext().setEntries(Lists.newArrayList(entries))
272        .setWalGroupId(fakeWalGroupId));
273    replicator.stop();
274    Region region = rs0.getRegion(hriSecondary.getEncodedName());
275    HTU.verifyNumericRows(region, f, 0, 1000);
276
277    HTU.deleteNumericRows(table, f, 0, 1000);
278    closeRegion(HTU, rs0, hriSecondary);
279  }
