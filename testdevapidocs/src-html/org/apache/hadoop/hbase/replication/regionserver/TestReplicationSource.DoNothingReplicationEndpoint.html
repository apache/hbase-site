152  @Test
153  public void testWALEntryFilter() throws IOException {
154    // To get the fully constructed default WALEntryFilter, need to create a ReplicationSource
155    // instance and init it.
156    ReplicationSource rs = new ReplicationSource();
157    UUID uuid = UUID.randomUUID();
158    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
159    ReplicationPeer mockPeer = Mockito.mock(ReplicationPeer.class);
160    Mockito.when(mockPeer.getConfiguration()).thenReturn(conf);
161    Mockito.when(mockPeer.getPeerBandwidth()).thenReturn(0L);
162    ReplicationPeerConfig peerConfig = Mockito.mock(ReplicationPeerConfig.class);
163    Mockito.when(peerConfig.getReplicationEndpointImpl()).
164      thenReturn(DoNothingReplicationEndpoint.class.getName());
165    Mockito.when(mockPeer.getPeerConfig()).thenReturn(peerConfig);
166    ReplicationSourceManager manager = Mockito.mock(ReplicationSourceManager.class);
167    Mockito.when(manager.getTotalBufferUsed()).thenReturn(new AtomicLong());
168    String queueId = "qid";
169    RegionServerServices rss =
170      TEST_UTIL.createMockRegionServerService(ServerName.parseServerName("a.b.c,1,1"));
171    rs.init(conf, null, manager, null, mockPeer, rss, queueId,
172      uuid, p -> OptionalLong.empty(), new MetricsSource(queueId));
173    try {
174      rs.startup();
175      TEST_UTIL.waitFor(30000, () -> rs.getWalEntryFilter() != null);
176      WALEntryFilter wef = rs.getWalEntryFilter();
177      // Test non-system WAL edit.
178      WALEdit we = new WALEdit().add(CellBuilderFactory.create(CellBuilderType.DEEP_COPY).
179        setRow(HConstants.EMPTY_START_ROW).
180        setFamily(HConstants.CATALOG_FAMILY).
181        setType(Cell.Type.Put).build());
182      WAL.Entry e = new WAL.Entry(new WALKeyImpl(HConstants.EMPTY_BYTE_ARRAY,
183        TableName.valueOf("test"), -1, -1, uuid), we);
184      assertTrue(wef.filter(e) == e);
185      // Test system WAL edit.
186      e = new WAL.Entry(
187        new WALKeyImpl(HConstants.EMPTY_BYTE_ARRAY, TableName.META_TABLE_NAME, -1, -1, uuid),
188          we);
189      assertNull(wef.filter(e));
190    } finally {
191      rs.terminate("Done");
192      rss.stop("Done");
193    }
194  }

196  /**
197   * Sanity check that we can move logs around while we are reading
198   * from them. Should this test fail, ReplicationSource would have a hard
199   * time reading logs that are being archived.
200   */
201  // This tests doesn't belong in here... it is not about ReplicationSource.
202  @Test
203  public void testLogMoving() throws Exception{
204    Path logPath = new Path(logDir, "log");
205    if (!FS.exists(logDir)) {
206      FS.mkdirs(logDir);
207    }
208    if (!FS.exists(oldLogDir)) {
209      FS.mkdirs(oldLogDir);
210    }
211    WALProvider.Writer writer = WALFactory.createWALWriter(FS, logPath,
212        TEST_UTIL.getConfiguration());
213    for(int i = 0; i < 3; i++) {
214      byte[] b = Bytes.toBytes(Integer.toString(i));
215      KeyValue kv = new KeyValue(b,b,b);
216      WALEdit edit = new WALEdit();
217      edit.add(kv);
218      WALKeyImpl key = new WALKeyImpl(b, TableName.valueOf(b), 0, 0,
219          HConstants.DEFAULT_CLUSTER_ID);
220      writer.append(new WAL.Entry(key, edit));
221      writer.sync(false);
222    }
223    writer.close();
224
225    WAL.Reader reader = WALFactory.createReader(FS, logPath, TEST_UTIL.getConfiguration());
226    WAL.Entry entry = reader.next();
227    assertNotNull(entry);
228
229    Path oldLogPath = new Path(oldLogDir, "log");
230    FS.rename(logPath, oldLogPath);
231
232    entry = reader.next();
233    assertNotNull(entry);
234
235    reader.next();
236    entry = reader.next();
237
238    assertNull(entry);
239    reader.close();
240  }

271  /**
272   * Tests that recovered queues are preserved on a regionserver shutdown.
273   * See HBASE-18192
274   */
275  @Test
276  public void testServerShutdownRecoveredQueue() throws Exception {
277    try {
278      // Ensure single-threaded WAL
279      conf.set("hbase.wal.provider", "defaultProvider");
280      conf.setInt("replication.sleep.before.failover", 2000);
281      // Introduces a delay in regionserver shutdown to give the race condition a chance to kick in.
282      conf.set(HConstants.REGION_SERVER_IMPL, ShutdownDelayRegionServer.class.getName());
283      MiniHBaseCluster cluster = TEST_UTIL.startMiniCluster(2);
284      TEST_UTIL_PEER.startMiniCluster(1);
285
286      HRegionServer serverA = cluster.getRegionServer(0);
287      final ReplicationSourceManager managerA =
288          serverA.getReplicationSourceService().getReplicationManager();
289      HRegionServer serverB = cluster.getRegionServer(1);
290      final ReplicationSourceManager managerB =
291          serverB.getReplicationSourceService().getReplicationManager();
292      final Admin admin = TEST_UTIL.getAdmin();
293
294      final String peerId = "TestPeer";
295      admin.addReplicationPeer(peerId,
296        ReplicationPeerConfig.newBuilder().setClusterKey(TEST_UTIL_PEER.getClusterKey()).build());
297      // Wait for replication sources to come up
298      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
299        @Override public boolean evaluate() {
300          return !(managerA.getSources().isEmpty() || managerB.getSources().isEmpty());
301        }
302      });
303      // Disabling peer makes sure there is at least one log to claim when the server dies
304      // The recovered queue will also stay there until the peer is disabled even if the
305      // WALs it contains have no data.
306      admin.disableReplicationPeer(peerId);
307
308      // Stopping serverA
309      // It's queues should be claimed by the only other alive server i.e. serverB
310      cluster.stopRegionServer(serverA.getServerName());
311      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
312        @Override public boolean evaluate() throws Exception {
313          return managerB.getOldSources().size() == 1;
314        }
315      });
316
317      final HRegionServer serverC = cluster.startRegionServer().getRegionServer();
318      serverC.waitForServerOnline();
319      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
320        @Override public boolean evaluate() throws Exception {
321          return serverC.getReplicationSourceService() != null;
322        }
323      });
324      final ReplicationSourceManager managerC =
325          ((Replication) serverC.getReplicationSourceService()).getReplicationManager();
326      // Sanity check
327      assertEquals(0, managerC.getOldSources().size());
328
329      // Stopping serverB
330      // Now serverC should have two recovered queues:
331      // 1. The serverB's normal queue
332      // 2. serverA's recovered queue on serverB
333      cluster.stopRegionServer(serverB.getServerName());
334      Waiter.waitFor(conf, 20000,
335        (Waiter.Predicate<Exception>) () -> managerC.getOldSources().size() == 2);
336      admin.enableReplicationPeer(peerId);
337      Waiter.waitFor(conf, 20000,
338        (Waiter.Predicate<Exception>) () -> managerC.getOldSources().size() == 0);
339    } finally {
340      conf.set(HConstants.REGION_SERVER_IMPL, HRegionServer.class.getName());
341    }
342  }

344  /**
345   * Regionserver implementation that adds a delay on the graceful shutdown.
346   */
347  public static class ShutdownDelayRegionServer extends HRegionServer {
348    public ShutdownDelayRegionServer(Configuration conf) throws IOException {
349      super(conf);
350    }
351
352    @Override
353    protected void stopServiceThreads() {
354      // Add a delay before service threads are shutdown.
355      // This will keep the zookeeper connection alive for the duration of the delay.
356      LOG.info("Adding a delay to the regionserver shutdown");
357      try {
358        Thread.sleep(2000);
359      } catch (InterruptedException ex) {
360        LOG.error("Interrupted while sleeping");
361      }
362      super.stopServiceThreads();
363    }
364  }

403  @Test
404  public void testRecoveredReplicationSourceShipperGetPosition() throws Exception {
405    String walGroupId = "fake-wal-group-id";
406    ServerName serverName = ServerName.valueOf("www.example.com", 12006, 1524679704418L);
407    ServerName deadServer = ServerName.valueOf("www.deadServer.com", 12006, 1524679704419L);
408    PriorityBlockingQueue<Path> queue = new PriorityBlockingQueue<>();
409    queue.put(new Path("/www/html/test"));
410    RecoveredReplicationSource source = Mockito.mock(RecoveredReplicationSource.class);
411    Server server = Mockito.mock(Server.class);
412    Mockito.when(server.getServerName()).thenReturn(serverName);
413    Mockito.when(source.getServer()).thenReturn(server);
414    Mockito.when(source.getServerWALsBelongTo()).thenReturn(deadServer);
415    ReplicationQueueStorage storage = Mockito.mock(ReplicationQueueStorage.class);
416    Mockito.when(storage.getWALPosition(Mockito.eq(serverName), Mockito.any(), Mockito.any()))
417      .thenReturn(1001L);
418    Mockito.when(storage.getWALPosition(Mockito.eq(deadServer), Mockito.any(), Mockito.any()))
419      .thenReturn(-1L);
420    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
421    conf.setInt("replication.source.maxretriesmultiplier", -1);
422    RecoveredReplicationSourceShipper shipper =
423      new RecoveredReplicationSourceShipper(conf, walGroupId, queue, source, storage);
424    assertEquals(1001L, shipper.getStartPosition());
425  }
