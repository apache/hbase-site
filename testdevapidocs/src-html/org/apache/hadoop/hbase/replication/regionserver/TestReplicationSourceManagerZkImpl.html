045/**
046 * Tests the ReplicationSourceManager with ReplicationQueueZkImpl's and

064  // Tests the naming convention of adopted queues for ReplicationQueuesZkImpl
065  @Test
066  public void testNodeFailoverDeadServerParsing() throws Exception {
067    LOG.debug("testNodeFailoverDeadServerParsing");
068    conf.setBoolean(HConstants.ZOOKEEPER_USEMULTI, true);
069    final Server server = new DummyServer("ec2-54-234-230-108.compute-1.amazonaws.com");
070    ReplicationQueues repQueues =
071      ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(conf, server,
072        server.getZooKeeper()));
073    repQueues.init(server.getServerName().toString());
074    // populate some znodes in the peer znode
075    files.add("log1");
076    files.add("log2");
077    for (String file : files) {
078      repQueues.addLog("1", file);
079    }
080
081    // create 3 DummyServers
082    Server s1 = new DummyServer("ip-10-8-101-114.ec2.internal");
083    Server s2 = new DummyServer("ec2-107-20-52-47.compute-1.amazonaws.com");
084    Server s3 = new DummyServer("ec2-23-20-187-167.compute-1.amazonaws.com");
085
086    // simulate three servers fail sequentially
087    ReplicationQueues rq1 =
088      ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s1.getConfiguration(), s1,
089        s1.getZooKeeper()));
090    rq1.init(s1.getServerName().toString());
091    Map<String, Set<String>> testMap =
092      rq1.claimQueues(server.getServerName().getServerName());
093    ReplicationQueues rq2 =
094      ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s2.getConfiguration(), s2,
095        s2.getZooKeeper()));
096    rq2.init(s2.getServerName().toString());
097    testMap = rq2.claimQueues(s1.getServerName().getServerName());
098    ReplicationQueues rq3 =
099      ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s3.getConfiguration(), s3,
100        s3.getZooKeeper()));
101    rq3.init(s3.getServerName().toString());
102    testMap = rq3.claimQueues(s2.getServerName().getServerName());
103
104    ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(testMap.keySet().iterator().next());
105    List<String> result = replicationQueueInfo.getDeadRegionServers();
106
107    // verify
108    assertTrue(result.contains(server.getServerName().getServerName()));
109    assertTrue(result.contains(s1.getServerName().getServerName()));
110    assertTrue(result.contains(s2.getServerName().getServerName()));
111
112    server.stop("");
113  }
114
115  @Test
116  public void testFailoverDeadServerCversionChange() throws Exception {
117    LOG.debug("testFailoverDeadServerCversionChange");
118
119    conf.setBoolean(HConstants.ZOOKEEPER_USEMULTI, true);
120    final Server s0 = new DummyServer("cversion-change0.example.org");
121    ReplicationQueues repQueues =
122      ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(conf, s0,
123        s0.getZooKeeper()));
124    repQueues.init(s0.getServerName().toString());
125    // populate some znodes in the peer znode
126    files.add("log1");
127    files.add("log2");
128    for (String file : files) {
129      repQueues.addLog("1", file);
130    }
131    // simulate queue transfer
132    Server s1 = new DummyServer("cversion-change1.example.org");
133    ReplicationQueues rq1 =
134      ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s1.getConfiguration(), s1,
135        s1.getZooKeeper()));
136    rq1.init(s1.getServerName().toString());
137
138    ReplicationQueuesClientZKImpl client =
139      (ReplicationQueuesClientZKImpl)ReplicationFactory.getReplicationQueuesClient(
140        new ReplicationQueuesClientArguments(s1.getConfiguration(), s1, s1.getZooKeeper()));
141
142    int v0 = client.getQueuesZNodeCversion();
143    rq1.claimQueues(s0.getServerName().getServerName());
144    int v1 = client.getQueuesZNodeCversion();
145    // cversion should increase by 1 since a child node is deleted
146    assertEquals(v0 + 1, v1);
147
148    s0.stop("");
149  }
