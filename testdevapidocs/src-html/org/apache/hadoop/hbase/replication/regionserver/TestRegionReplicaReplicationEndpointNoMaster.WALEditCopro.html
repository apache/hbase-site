165  @Test (timeout = 240000)
166  public void testReplayCallable() throws Exception {
167    // tests replaying the edits to a secondary region replica using the Callable directly
168    openRegion(HTU, rs0, hriSecondary);
169    ClusterConnection connection =
170        (ClusterConnection) ConnectionFactory.createConnection(HTU.getConfiguration());
171
172    //load some data to primary
173    HTU.loadNumericRows(table, f, 0, 1000);
174
175    Assert.assertEquals(1000, entries.size());
176    // replay the edits to the secondary using replay callable
177    replicateUsingCallable(connection, entries);
178
179    Region region = rs0.getFromOnlineRegions(hriSecondary.getEncodedName());
180    HTU.verifyNumericRows(region, f, 0, 1000);
181
182    HTU.deleteNumericRows(table, f, 0, 1000);
183    closeRegion(HTU, rs0, hriSecondary);
184    connection.close();
185  }
186
187  private void replicateUsingCallable(ClusterConnection connection, Queue<Entry> entries)
188      throws IOException, RuntimeException {
189    Entry entry;
190    while ((entry = entries.poll()) != null) {
191      byte[] row = CellUtil.cloneRow(entry.getEdit().getCells().get(0));
192      RegionLocations locations = connection.locateRegion(tableName, row, true, true);
193      RegionReplicaReplayCallable callable = new RegionReplicaReplayCallable(connection,
194        RpcControllerFactory.instantiate(connection.getConfiguration()),
195        table.getName(), locations.getRegionLocation(1),
196        locations.getRegionLocation(1).getRegionInfo(), row, Lists.newArrayList(entry),
197        new AtomicLong());
198
199      RpcRetryingCallerFactory factory = RpcRetryingCallerFactory.instantiate(
200        connection.getConfiguration());
201      factory.<ReplicateWALEntryResponse> newCaller().callWithRetries(callable, 10000);
202    }
203  }

205  @Test (timeout = 240000)
206  public void testReplayCallableWithRegionMove() throws Exception {
207    // tests replaying the edits to a secondary region replica using the Callable directly while
208    // the region is moved to another location.It tests handling of RME.
209    openRegion(HTU, rs0, hriSecondary);
210    ClusterConnection connection =
211        (ClusterConnection) ConnectionFactory.createConnection(HTU.getConfiguration());
212    //load some data to primary
213    HTU.loadNumericRows(table, f, 0, 1000);
214
215    Assert.assertEquals(1000, entries.size());
216    // replay the edits to the secondary using replay callable
217    replicateUsingCallable(connection, entries);
218
219    Region region = rs0.getFromOnlineRegions(hriSecondary.getEncodedName());
220    HTU.verifyNumericRows(region, f, 0, 1000);
221
222    HTU.loadNumericRows(table, f, 1000, 2000); // load some more data to primary
223
224    // move the secondary region from RS0 to RS1
225    closeRegion(HTU, rs0, hriSecondary);
226    openRegion(HTU, rs1, hriSecondary);
227
228    // replicate the new data
229    replicateUsingCallable(connection, entries);
230
231    region = rs1.getFromOnlineRegions(hriSecondary.getEncodedName());
232    // verify the new data. old data may or may not be there
233    HTU.verifyNumericRows(region, f, 1000, 2000);
234
235    HTU.deleteNumericRows(table, f, 0, 2000);
236    closeRegion(HTU, rs1, hriSecondary);
237    connection.close();
238  }

272  @Test (timeout = 240000)
273  public void testReplayedEditsAreSkipped() throws Exception {
274    openRegion(HTU, rs0, hriSecondary);
275    ClusterConnection connection =
276        (ClusterConnection) ConnectionFactory.createConnection(HTU.getConfiguration());
277    RegionReplicaReplicationEndpoint replicator = new RegionReplicaReplicationEndpoint();
278
279    ReplicationEndpoint.Context context = mock(ReplicationEndpoint.Context.class);
280    when(context.getConfiguration()).thenReturn(HTU.getConfiguration());
281    when(context.getMetrics()).thenReturn(mock(MetricsSource.class));
282
283    ReplicationPeer mockPeer = mock(ReplicationPeer.class);
284    when(mockPeer.getNamespaces()).thenReturn(null);
285    when(mockPeer.getTableCFs()).thenReturn(null);
286    when(mockPeer.getPeerConfig()).thenReturn(new ReplicationPeerConfig());
287    when(context.getReplicationPeer()).thenReturn(mockPeer);
288
289    replicator.init(context);
290    replicator.startAsync();
291
292    // test the filter for the RE, not actual replication
293    WALEntryFilter filter = replicator.getWALEntryfilter();
294
295    //load some data to primary
296    HTU.loadNumericRows(table, f, 0, 1000);
297
298    Assert.assertEquals(1000, entries.size());
299    for (Entry e: entries) {
300      Cell _c = e.getEdit().getCells().get(0);
301      if (Integer.parseInt(
302        Bytes.toString(_c.getValueArray(), _c.getValueOffset(), _c.getValueLength())) % 2 == 0) {
303        e.getKey().setOrigLogSeqNum(1); // simulate dist log replay by setting orig seq id
304      }
305    }
306
307    long skipped = 0, replayed = 0;
308    for (Entry e : entries) {
309      if (filter.filter(e) == null) {
310        skipped++;
311      } else {
312        replayed++;
313      }
314    }
315
316    assertEquals(500, skipped);
317    assertEquals(500, replayed);
318
319    HTU.deleteNumericRows(table, f, 0, 1000);
320    closeRegion(HTU, rs0, hriSecondary);
321    connection.close();
322  }
