047@Category({ ReplicationTests.class, MediumTests.class})
048public class TestReplicationSourceManagerJoin extends TestReplicationBase {
049
050  @ClassRule
051  public static final HBaseClassTestRule CLASS_RULE =
052    HBaseClassTestRule.forClass(TestReplicationSourceManagerJoin.class);
053
054  @Rule
055  public TestName testName = new TestName();
056
057  @BeforeClass
058  public static void setUpBeforeClass() throws Exception {
059    // NUM_SLAVES1 is presumed 2 in below.
060    NUM_SLAVES1 = 2;
061    TestReplicationBase.setUpBeforeClass();
062  }
063
064  @Test
065  public void testReplicationSourcesTerminate() throws Exception {
066    // Create table in source cluster only, let TableNotFoundException block peer to avoid
067    // recovered source end.
068    TableName tableName = TableName.valueOf(testName.getMethodName());
069    TableDescriptor td = TableDescriptorBuilder.newBuilder(tableName)
070      .setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(famName)
071        .setScope(HConstants.REPLICATION_SCOPE_GLOBAL).build())
072      .setColumnFamily(ColumnFamilyDescriptorBuilder.of(noRepfamName)).build();
073    hbaseAdmin.createTable(td);
074    assertFalse(UTIL2.getAdmin().tableExists(tableName));
075    Table table = UTIL1.getConnection().getTable(tableName);
076    // load data
077    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
078      table.put(new Put(Bytes.toBytes(i)).addColumn(famName, row, row));
079    }
080    // Kill rs holding table region. There are only TWO servers. We depend on it.
081    Optional<HRegionServer> server =
082      UTIL1.getMiniHBaseCluster().getLiveRegionServerThreads().stream()
083      .map(JVMClusterUtil.RegionServerThread::getRegionServer)
084      .filter(rs -> !rs.getRegions(tableName).isEmpty()).findAny();
085    assertTrue(server.isPresent());
086    server.get().abort("stopping for test");
087
088    UTIL1.waitFor(60000, () -> 1 == UTIL1.getMiniHBaseCluster().getNumLiveRegionServers());
089    UTIL1.waitTableAvailable(tableName);
090    // Wait for recovered source running
091    HRegionServer rs =
092      UTIL1.getMiniHBaseCluster().getLiveRegionServerThreads().get(0).getRegionServer();
093    ReplicationSourceManager manager = rs.getReplicationSourceService().getReplicationManager();
094    UTIL1.waitFor(60000, () -> !manager.getOldSources().isEmpty());
095
096    assertFalse(manager.getSources().isEmpty());
097    assertFalse(manager.getOldSources().isEmpty());
098
099    // Check all sources running before manager.join(), terminated after manager.join().
100    Stream.concat(manager.getSources().stream(), manager.getOldSources().stream())
101      .filter(src -> src instanceof ReplicationSource)
102      .forEach(src -> assertTrue(((ReplicationSource) src).sourceRunning));
103    manager.join();
104    Stream.concat(manager.getSources().stream(), manager.getOldSources().stream())
105      .filter(src -> src instanceof ReplicationSource)
106      .forEach(src -> assertFalse(((ReplicationSource) src).sourceRunning));
107  }
