081/**
082 * Simple test of sink-side wal entry filter facility.
083 */
084@Category({ReplicationTests.class, SmallTests.class})
085public class TestWALEntrySinkFilter {
086  private static final Logger LOG = LoggerFactory.getLogger(TestReplicationSink.class);
087  @Rule public TestName name = new TestName();
088  @Rule public final TestRule timeout = CategoryBasedTimeout.builder().
089      withTimeout(this.getClass()).
090      withLookingForStuckThread(true).
091      build();
092  static final int BOUNDARY = 5;
093  static final AtomicInteger UNFILTERED = new AtomicInteger();
094  static final AtomicInteger FILTERED = new AtomicInteger();
095
096  /**
097   * Implemetentation of Stoppable to pass into ReplicationSink.
098   */
099  private static Stoppable STOPPABLE = new Stoppable() {
100    private final AtomicBoolean stop = new AtomicBoolean(false);
101
102    @Override
103    public boolean isStopped() {
104      return this.stop.get();
105    }
106
107    @Override
108    public void stop(String why) {
109      LOG.info("STOPPING BECAUSE: " + why);
110      this.stop.set(true);
111    }
112  };

114  /**
115   * Test filter.
116   * Filter will filter out any write time that is <= 5 (BOUNDARY). We count how many items we
117   * filter out and we count how many cells make it through for distribution way down below in the
118   * Table#batch implementation. Puts in place a custom DevNullConnection so we can insert our
119   * counting Table.
120   * @throws IOException
121   */
122  @Test
123  public void testWALEntryFilter() throws IOException {
124    Configuration conf = HBaseConfiguration.create();
125    // Make it so our filter is instantiated on construction of ReplicationSink.
126    conf.setClass(WALEntrySinkFilter.WAL_ENTRY_FILTER_KEY,
127        IfTimeIsGreaterThanBOUNDARYWALEntrySinkFilterImpl.class, WALEntrySinkFilter.class);
128    conf.setClass("hbase.client.connection.impl", DevNullConnection.class,
129        Connection.class);
130    ReplicationSink sink = new ReplicationSink(conf, STOPPABLE);
131    // Create some dumb walentries.
132    List< org.apache.hadoop.hbase.shaded.protobuf.generated.AdminProtos.WALEntry > entries =
133        new ArrayList<>();
134    AdminProtos.WALEntry.Builder entryBuilder = AdminProtos.WALEntry.newBuilder();
135    // Need a tablename.
136    ByteString tableName =
137        ByteString.copyFromUtf8(TableName.valueOf(this.name.getMethodName()).toString());
138    // Add WALEdit Cells to Cells List. The way edits arrive at the sink is with protos
139    // describing the edit with all Cells from all edits aggregated in a single CellScanner.
140    final List<Cell> cells = new ArrayList<>();
141    int count = BOUNDARY * 2;
142    for(int i = 0; i < count; i++) {
143      byte [] bytes = Bytes.toBytes(i);
144      // Create a wal entry. Everything is set to the current index as bytes or int/long.
145      entryBuilder.clear();
146      entryBuilder.setKey(entryBuilder.getKeyBuilder().
147          setLogSequenceNumber(i).
148          setEncodedRegionName(ByteString.copyFrom(bytes)).
149          setWriteTime(i).
150          setTableName(tableName).build());
151      // Lets have one Cell associated with each WALEdit.
152      entryBuilder.setAssociatedCellCount(1);
153      entries.add(entryBuilder.build());
154      // We need to add a Cell per WALEdit to the cells array.
155      CellBuilder cellBuilder = CellBuilderFactory.create(CellBuilderType.DEEP_COPY);
156      // Make cells whose row, family, cell, value, and ts are == 'i'.
157      Cell cell = cellBuilder.
158          setRow(bytes).
159          setFamily(bytes).
160          setQualifier(bytes).
161          setType(Cell.Type.Put).
162          setTimestamp(i).
163          setValue(bytes).build();
164      cells.add(cell);
165    }
166    // Now wrap our cells array in a CellScanner that we can pass in to replicateEntries. It has
167    // all Cells from all the WALEntries made above.
168    CellScanner cellScanner = new CellScanner() {
169      // Set to -1 because advance gets called before current.
170      int index = -1;
171
172      @Override
173      public Cell current() {
174        return cells.get(index);
175      }
176
177      @Override
178      public boolean advance() throws IOException {
179        index++;
180        return index < cells.size();
181      }
182    };
183    // Call our sink.
184    sink.replicateEntries(entries, cellScanner, null, null, null);
185    // Check what made it through and what was filtered.
186    assertTrue(FILTERED.get() > 0);
187    assertTrue(UNFILTERED.get() > 0);
188    assertEquals(count, FILTERED.get() + UNFILTERED.get());
189  }

191  /**
192   * Simple filter that will filter out any entry wholse writeTime is <= 5.
193   */
194  public static class IfTimeIsGreaterThanBOUNDARYWALEntrySinkFilterImpl implements WALEntrySinkFilter {
195    public IfTimeIsGreaterThanBOUNDARYWALEntrySinkFilterImpl() {}
196
197    @Override
198    public void init(Connection connection) {
199      // Do nothing.
200    }
201
202    @Override
203    public boolean filter(TableName table, long writeTime) {
204      boolean b = writeTime <= BOUNDARY;
205      if (b) {
206        FILTERED.incrementAndGet();
207      }
208      return b;
209    }
210  }
