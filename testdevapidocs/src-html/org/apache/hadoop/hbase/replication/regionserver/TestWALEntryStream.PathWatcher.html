181  @Test
182  public void testAppendsWithRolls() throws Exception {
183    appendToLogAndSync();
184    long oldPos;
185    try (WALEntryStream entryStream =
186        new WALEntryStream(walQueue, fs, CONF, 0, log, null, new MetricsSource("1"))) {
187      // There's one edit in the log, read it. Reading past it needs to throw exception
188      assertTrue(entryStream.hasNext());
189      WAL.Entry entry = entryStream.peek();
190      assertSame(entry, entryStream.next());
191      assertNotNull(entry);
192      assertFalse(entryStream.hasNext());
193      assertNull(entryStream.peek());
194      assertNull(entryStream.next());
195      oldPos = entryStream.getPosition();
196    }
197
198    appendToLogAndSync();
199
200    try (WALEntryStream entryStream = new WALEntryStream(walQueue, fs, CONF, oldPos,
201        log, null, new MetricsSource("1"))) {
202      // Read the newly added entry, make sure we made progress
203      WAL.Entry entry = entryStream.next();
204      assertNotEquals(oldPos, entryStream.getPosition());
205      assertNotNull(entry);
206      oldPos = entryStream.getPosition();
207    }
208
209    // We rolled but we still should see the end of the first log and get that item
210    appendToLogAndSync();
211    log.rollWriter();
212    appendToLogAndSync();
213
214    try (WALEntryStream entryStream = new WALEntryStream(walQueue, fs, CONF, oldPos,
215        log, null, new MetricsSource("1"))) {
216      WAL.Entry entry = entryStream.next();
217      assertNotEquals(oldPos, entryStream.getPosition());
218      assertNotNull(entry);
219
220      // next item should come from the new log
221      entry = entryStream.next();
222      assertNotEquals(oldPos, entryStream.getPosition());
223      assertNotNull(entry);
224
225      // no more entries to read
226      assertFalse(entryStream.hasNext());
227      oldPos = entryStream.getPosition();
228    }
229  }

231  /**
232   * Tests that if after a stream is opened, more entries come in and then the log is rolled, we
233   * don't mistakenly dequeue the current log thinking we're done with it
234   */
235  @Test
236  public void testLogrollWhileStreaming() throws Exception {
237    appendToLog("1");
238    appendToLog("2");// 2
239    try (WALEntryStream entryStream =
240        new WALEntryStream(walQueue, fs, CONF, 0, log, null, new MetricsSource("1"))) {
241      assertEquals("1", getRow(entryStream.next()));
242
243      appendToLog("3"); // 3 - comes in after reader opened
244      log.rollWriter(); // log roll happening while we're reading
245      appendToLog("4"); // 4 - this append is in the rolled log
246
247      assertEquals("2", getRow(entryStream.next()));
248      assertEquals(2, walQueue.size()); // we should not have dequeued yet since there's still an
249                                        // entry in first log
250      assertEquals("3", getRow(entryStream.next())); // if implemented improperly, this would be 4
251                                                     // and 3 would be skipped
252      assertEquals("4", getRow(entryStream.next())); // 4
253      assertEquals(1, walQueue.size()); // now we've dequeued and moved on to next log properly
254      assertFalse(entryStream.hasNext());
255    }
256  }

478    try (WALEntryStream entryStream =

484    }
485
486    // start up a reader
487    Path walPath = walQueue.peek();
488    ReplicationSource source = mockReplicationSource(false, CONF);
489    AtomicInteger invokeCount = new AtomicInteger(0);
490    AtomicBoolean enabled = new AtomicBoolean(false);
491    when(source.isPeerEnabled()).then(i -> {
492      invokeCount.incrementAndGet();
493      return enabled.get();
494    });
495
496    ReplicationSourceWALReader reader =
497      new ReplicationSourceWALReader(fs, CONF, walQueue, 0, getDummyFilter(), source);
498    reader.start();

503    TEST_UTIL.waitFor(30000, () -> invokeCount.get() >= 5);
504    // confirm that we can read nothing if the peer is disabled
505    assertFalse(future.isDone());
506    // then enable the peer, we should get the batch
507    enabled.set(true);
508    WALEntryBatch entryBatch = future.get();
509
510    // should've batched up our entries
511    assertNotNull(entryBatch);
512    assertEquals(3, entryBatch.getWalEntries().size());
513    assertEquals(position, entryBatch.getLastWalPosition());
514    assertEquals(walPath, entryBatch.getLastWalPath());
515    assertEquals(3, entryBatch.getNbRowKeys());

523  private void appendToLog(String key) throws IOException {
524    final long txid = log.append(info,
525      new WALKeyImpl(info.getEncodedNameAsBytes(), tableName, System.currentTimeMillis(),
526          mvcc, scopes), getWALEdit(key), true);
527    log.sync(txid);
528  }

530  private void appendEntriesToLogAndSync(int count) throws IOException {
531    long txid = -1L;
532    for (int i = 0; i < count; i++) {
533      txid = appendToLog(1);
534    }
535    log.sync(txid);
536  }

543    long txid = appendToLog(count);
544    log.sync(txid);
545  }

547  private long appendToLog(int count) throws IOException {
548    return log.append(info, new WALKeyImpl(info.getEncodedNameAsBytes(), tableName,
549      System.currentTimeMillis(), mvcc, scopes), getWALEdits(count), true);
550  }

552  private WALEdit getWALEdits(int count) {
553    WALEdit edit = new WALEdit();
554    for (int i = 0; i < count; i++) {
555      edit.add(new KeyValue(Bytes.toBytes(System.currentTimeMillis()), family, qualifier,
556          System.currentTimeMillis(), qualifier));
557    }
558    return edit;
559  }

565    return edit;

589  @Test
590  public void testReadBeyondCommittedLength() throws IOException, InterruptedException {
591    appendToLog("1");
592    appendToLog("2");
593    long size = log.getLogFileSizeIfBeingWritten(walQueue.peek()).getAsLong();
594    AtomicLong fileLength = new AtomicLong(size - 1);
595    try (WALEntryStream entryStream = new WALEntryStream(walQueue, fs, CONF, 0,
596        p -> OptionalLong.of(fileLength.get()), null, new MetricsSource("1"))) {
597      assertTrue(entryStream.hasNext());
598      assertNotNull(entryStream.next());
599      // can not get log 2
600      assertFalse(entryStream.hasNext());
601      Thread.sleep(1000);
602      entryStream.reset();
603      // still can not get log 2
604      assertFalse(entryStream.hasNext());
605
606      // can get log 2 now
607      fileLength.set(size);
608      entryStream.reset();
609      assertTrue(entryStream.hasNext());
610      assertNotNull(entryStream.next());
611
612      assertFalse(entryStream.hasNext());
613    }
