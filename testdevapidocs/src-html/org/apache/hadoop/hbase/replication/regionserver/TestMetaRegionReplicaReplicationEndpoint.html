124   * Assert that the ReplicationSource for hbase:meta gets created when hbase:meta is opened.
125   */
126  @Test
127  public void testHBaseMetaReplicationSourceCreatedOnOpen() throws Exception {
128    SingleProcessHBaseCluster cluster = HTU.getMiniHBaseCluster();
129    HRegionServer hrs = cluster.getRegionServer(cluster.getServerHoldingMeta());
130    // Replicate a row to prove all working.
131    testHBaseMetaReplicatesOneRow(0);
132    assertTrue(isMetaRegionReplicaReplicationSource(hrs));
133    // Now move the hbase:meta and make sure the ReplicationSource is in both places.
134    HRegionServer hrsOther = null;
135    for (int i = 0; i < cluster.getNumLiveRegionServers(); i++) {
136      hrsOther = cluster.getRegionServer(i);
137      if (hrsOther.getServerName().equals(hrs.getServerName())) {
138        hrsOther = null;
139        continue;
140      }
141      break;
142    }
143    assertNotNull(hrsOther);
144    assertFalse(isMetaRegionReplicaReplicationSource(hrsOther));
145    Region meta = null;
146    for (Region region : hrs.getOnlineRegionsLocalContext()) {
147      if (region.getRegionInfo().isMetaRegion()) {
148        meta = region;
149        break;
150      }
151    }
152    assertNotNull(meta);
153    HTU.moveRegionAndWait(meta.getRegionInfo(), hrsOther.getServerName());
154    // Assert that there is a ReplicationSource in both places now.
155    assertTrue(isMetaRegionReplicaReplicationSource(hrs));
156    assertTrue(isMetaRegionReplicaReplicationSource(hrsOther));
157    // Replicate to show stuff still works.
158    testHBaseMetaReplicatesOneRow(1);
159    // Now pretend a few hours have gone by... roll the meta WAL in original location... Move the
160    // meta back and retry replication. See if it works.
161    hrs.getWAL(meta.getRegionInfo()).rollWriter(true);
162    testHBaseMetaReplicatesOneRow(2);
163    hrs.getWAL(meta.getRegionInfo()).rollWriter(true);
164    testHBaseMetaReplicatesOneRow(3);
165  }

289  private void verifyReplication(TableName tableName, int regionReplication, final int startRow,
290    final int endRow, final byte[] family, final boolean present) throws Exception {
291    // find the regions
292    final Region[] regions = new Region[regionReplication];
293
294    for (int i = 0; i < NB_SERVERS; i++) {
295      HRegionServer rs = HTU.getMiniHBaseCluster().getRegionServer(i);
296      List<HRegion> onlineRegions = rs.getRegions(tableName);
297      for (HRegion region : onlineRegions) {
298        regions[region.getRegionInfo().getReplicaId()] = region;
299      }
300    }
301
302    for (Region region : regions) {
303      assertNotNull(region);
304    }
305
306    for (int i = 1; i < regionReplication; i++) {
307      final Region region = regions[i];
308      // wait until all the data is replicated to all secondary regions
309      Waiter.waitFor(HTU.getConfiguration(), 90000, 1000, new Waiter.Predicate<Exception>() {
310        @Override
311        public boolean evaluate() throws Exception {
312          LOG.info("verifying replication for region replica:" + region.getRegionInfo());
313          try {
314            HTU.verifyNumericRows(region, family, startRow, endRow, present);
315          } catch (Throwable ex) {
316            LOG.warn("Verification from secondary region is not complete yet", ex);
317            // still wait
318            return false;
319          }
320          return true;
321        }
322      });
323    }
324  }

327   * Replicas come online after primary.
328   */
329  private void waitForMetaReplicasToOnline() throws IOException {
330    final RegionLocator regionLocator =
331      HTU.getConnection().getRegionLocator(TableName.META_TABLE_NAME);
332    HTU.waitFor(10000,
333      // getRegionLocations returns an entry for each replica but if unassigned, entry is null.

337    List<HRegionLocation> locations = regionLocator.getRegionLocations(HConstants.EMPTY_START_ROW);
338    LOG.info("Found locations {}", locations);
339    assertEquals(numOfMetaReplica, locations.size());
340  }

387  /**
388   * Verify when a Table is deleted from primary, then there are no references in replicas
389   * (because they get the delete of the table rows too).
390   */
391  private void verifyDeletedReplication(TableName tableName, int regionReplication,
392    final TableName deletedTableName) {
393    final Region[] regions = getAllRegions(tableName, regionReplication);
394
395    // Start count at '1' so we skip default, primary replica and only look at secondaries.
396    for (int i = 1; i < regionReplication; i++) {
397      final Region region = regions[i];
398      // wait until all the data is replicated to all secondary regions
399      Waiter.waitFor(HTU.getConfiguration(), 30000, 1000, new Waiter.Predicate<Exception>() {
400        @Override
401        public boolean evaluate() throws Exception {
402          LOG.info("Verifying replication for region replica {}", region.getRegionInfo());
403          try (RegionScanner rs = region.getScanner(new Scan())) {
404            List<Cell> cells = new ArrayList<>();
405            while (rs.next(cells)) {
406              continue;
407            }
408            return doesNotContain(cells, deletedTableName);
409          } catch (Throwable ex) {
410            LOG.warn("Verification from secondary region is not complete yet", ex);
411            // still wait
412            return false;
413          }
414        }
415      });
416    }
417  }

419  /**
420   * Cells are from hbase:meta replica so will start w/ 'tableName,'; i.e. the tablename followed
421   * by HConstants.DELIMITER. Make sure the deleted table is no longer present in passed
422   * <code>cells</code>.
423   */
424  private boolean doesNotContain(List<Cell> cells, TableName tableName) {
425    for (Cell cell : cells) {
426      String row = Bytes.toString(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
427      if (row.startsWith(tableName.toString() + HConstants.DELIMITER)) {
428        return false;
429      }
430    }
431    return true;
432  }

434  /**
435   * Verify Replicas have results (exactly).
436   */
437  private void verifyReplication(TableName tableName, int regionReplication,
438    List<Result> contains) {
439    final Region[] regions = getAllRegions(tableName, regionReplication);
440
441    // Start count at '1' so we skip default, primary replica and only look at secondaries.
442    for (int i = 1; i < regionReplication; i++) {
443      final Region region = regions[i];
444      // wait until all the data is replicated to all secondary regions
445      Waiter.waitFor(HTU.getConfiguration(), 30000, 1000, new Waiter.Predicate<Exception>() {
446        @Override
447        public boolean evaluate() throws Exception {
448          LOG.info("Verifying replication for region replica {}", region.getRegionInfo());
449          try (RegionScanner rs = region.getScanner(new Scan())) {
450            List<Cell> cells = new ArrayList<>();
451            while (rs.next(cells)) {
452              continue;
453            }
454            return contains(contains, cells);
455          } catch (Throwable ex) {
456            LOG.warn("Verification from secondary region is not complete yet", ex);
457            // still wait
458            return false;
459          }
460        }
461      });
462    }
463  }

465  /**
466   * Presumes sorted Cells. Verify that <code>cells</code> has <code>contains</code> at least.
467   */
468  static boolean contains(List<Result> contains, List<Cell> cells) throws IOException {
469    CellScanner containsScanner = CellUtil.createCellScanner(contains);
470    CellScanner cellsScanner = CellUtil.createCellScanner(cells);
471    int matches = 0;
472    int count = 0;
473    while (containsScanner.advance()) {
474      while (cellsScanner.advance()) {
475        count++;
476        LOG.info("{} {}", containsScanner.current(), cellsScanner.current());
477        if (containsScanner.current().equals(cellsScanner.current())) {
478          matches++;
479          break;
480        }
481      }
482    }
483    return !containsScanner.advance() && matches >= 1 && count >= matches && count == cells.size();
484  }

493  private void primaryNoChangeReplicaIncrease(final long[] before, final long[] after) {
494    assertEquals(before[RegionInfo.DEFAULT_REPLICA_ID],
495      after[RegionInfo.DEFAULT_REPLICA_ID]);
496
497    for (int i = 1; i < after.length; i ++) {
498      assertTrue(after[i] > before[i]);
499    }

502  private void primaryIncreaseReplicaNoChange(final long[] before, final long[] after) {
503    // There are read requests increase for primary meta replica.
504    assertTrue(after[RegionInfo.DEFAULT_REPLICA_ID] >
505      before[RegionInfo.DEFAULT_REPLICA_ID]);
506
507    // No change for replica regions
508    for (int i = 1; i < after.length; i ++) {
509      assertEquals(before[i], after[i]);
510    }
511  }

522  @Test
523  public void testHBaseMetaReplicaGets() throws Exception {
524
525    TableName tn = TableName.valueOf(this.name.getMethodName());
526    final Region[] metaRegions = getAllRegions(TableName.META_TABLE_NAME, numOfMetaReplica);
527    long[] readReqsForMetaReplicas = new long[numOfMetaReplica];
528    long[] readReqsForMetaReplicasAfterGet = new long[numOfMetaReplica];
529    long[] readReqsForMetaReplicasAfterMove = new long[numOfMetaReplica];
530    long[] readReqsForMetaReplicasAfterSecondMove = new long[numOfMetaReplica];
531    long[] readReqsForMetaReplicasAfterThirdGet = new long[numOfMetaReplica];
532    Region userRegion = null;
533    HRegionServer srcRs = null;
534    HRegionServer destRs = null;
535
536    try (Table table = HTU.createTable(tn, HConstants.CATALOG_FAMILY,
537      Arrays.copyOfRange(HBaseTestingUtil.KEYS, 1, HBaseTestingUtil.KEYS.length))) {
538      verifyReplication(TableName.META_TABLE_NAME, numOfMetaReplica, getMetaCells(table.getName()));
539      // load different values
540      HTU.loadTable(table, new byte[][] { HConstants.CATALOG_FAMILY }, VALUE);
541      for (int i = 0; i < NB_SERVERS; i++) {
542        HRegionServer rs = HTU.getMiniHBaseCluster().getRegionServer(i);
543        List<HRegion> onlineRegions = rs.getRegions(tn);
544        if (onlineRegions.size() > 0) {
545          userRegion = onlineRegions.get(0);
546          srcRs = rs;
547          if (i > 0) {
548            destRs = HTU.getMiniHBaseCluster().getRegionServer(0);
549          } else {
550            destRs = HTU.getMiniHBaseCluster().getRegionServer(1);
551          }
552        }
553      }
554
555      getMetaReplicaReadRequests(metaRegions, readReqsForMetaReplicas);
556
557      Configuration c = new Configuration(HTU.getConfiguration());
558      c.setBoolean(HConstants.USE_META_REPLICAS, true);
559      c.set(LOCATOR_META_REPLICAS_MODE, "LoadBalance");
560      Connection connection = ConnectionFactory.createConnection(c);
561      Table tableForGet = connection.getTable(tn);
562      byte[][] getRows = new byte[HBaseTestingUtil.KEYS.length][];
563
564      int i = 0;
565      for (byte[] key : HBaseTestingUtil.KEYS) {
566        getRows[i] = key;
567        i++;
568      }
569      getRows[0] = Bytes.toBytes("aaa");
570      doNGets(tableForGet, getRows);
571
572      getMetaReplicaReadRequests(metaRegions, readReqsForMetaReplicasAfterGet);
573
574      // There is no read requests increase for primary meta replica.
575      // For rest of meta replicas, there are more reads against them.
576      primaryNoChangeReplicaIncrease(readReqsForMetaReplicas, readReqsForMetaReplicasAfterGet);
577
578      // move one of regions so it meta cache may be invalid.
579      HTU.moveRegionAndWait(userRegion.getRegionInfo(), destRs.getServerName());
580
581      doNGets(tableForGet, getRows);
582
583      getMetaReplicaReadRequests(metaRegions, readReqsForMetaReplicasAfterMove);
584
585      // There are read requests increase for primary meta replica.
586      // For rest of meta replicas, there is no change as regionMove will tell the new location
587      primaryIncreaseReplicaNoChange(readReqsForMetaReplicasAfterGet,
588        readReqsForMetaReplicasAfterMove);
589      // Move region again.
590      HTU.moveRegionAndWait(userRegion.getRegionInfo(), srcRs.getServerName());
591
592      // Wait until moveRegion cache timeout.
593      while (destRs.getMovedRegion(userRegion.getRegionInfo().getEncodedName()) != null) {
594        Thread.sleep(1000);
595      }
596
597      getMetaReplicaReadRequests(metaRegions, readReqsForMetaReplicasAfterSecondMove);
598
599      // There are read requests increase for primary meta replica.
600      // For rest of meta replicas, there is no change.
601      primaryIncreaseReplicaNoChange(readReqsForMetaReplicasAfterMove,
602        readReqsForMetaReplicasAfterSecondMove);
603
604      doNGets(tableForGet, getRows);
605
606      getMetaReplicaReadRequests(metaRegions, readReqsForMetaReplicasAfterThirdGet);
607
608      // Since it gets RegionNotServedException, it will go to primary for the next lookup.
609      // There are read requests increase for primary meta replica.
610      // For rest of meta replicas, there is no change.
611      primaryIncreaseReplicaNoChange(readReqsForMetaReplicasAfterSecondMove,
612        readReqsForMetaReplicasAfterThirdGet);
613    }
614  }
