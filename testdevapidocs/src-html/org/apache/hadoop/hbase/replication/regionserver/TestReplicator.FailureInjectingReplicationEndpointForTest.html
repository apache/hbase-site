066  @Test
067  public void testReplicatorBatching() throws Exception {
068    // Clear the tables
069    truncateTable(UTIL1, tableName);
070    truncateTable(UTIL2, tableName);
071
072    // Replace the peer set up for us by the base class with a wrapper for this test
073    hbaseAdmin.addReplicationPeer("testReplicatorBatching",
074      new ReplicationPeerConfig().setClusterKey(UTIL2.getClusterKey())
075          .setReplicationEndpointImpl(ReplicationEndpointForTest.class.getName()));
076
077    ReplicationEndpointForTest.setBatchCount(0);
078    ReplicationEndpointForTest.setEntriesCount(0);
079    try {
080      ReplicationEndpointForTest.pause();
081      try {
082        // Queue up a bunch of cells of size 8K. Because of RPC size limits, they will all
083        // have to be replicated separately.
084        final byte[] valueBytes = new byte[8 * 1024];
085        for (int i = 0; i < NUM_ROWS; i++) {
086          htable1.put(new Put(Bytes.toBytes("row" + Integer.toString(i))).addColumn(famName, null,
087            valueBytes));
088        }
089      } finally {
090        ReplicationEndpointForTest.resume();
091      }
092
093      // Wait for replication to complete.
094      Waiter.waitFor(CONF1, 60000, new Waiter.ExplainingPredicate<Exception>() {
095        @Override
096        public boolean evaluate() throws Exception {
097          LOG.info("Count=" + ReplicationEndpointForTest.getBatchCount());
098          return ReplicationEndpointForTest.getBatchCount() >= NUM_ROWS;
099        }
100
101        @Override
102        public String explainFailure() throws Exception {
103          return "We waited too long for expected replication of " + NUM_ROWS + " entries";
104        }
105      });
106
107      assertEquals("We sent an incorrect number of batches", NUM_ROWS,
108        ReplicationEndpointForTest.getBatchCount());
109      assertEquals("We did not replicate enough rows", NUM_ROWS, UTIL2.countRows(htable2));
110    } finally {
111      hbaseAdmin.removeReplicationPeer("testReplicatorBatching");
112    }
113  }

173  public static class ReplicationEndpointForTest extends HBaseInterClusterReplicationEndpoint {
174
175    protected static AtomicInteger batchCount = new AtomicInteger(0);
176    protected static int entriesCount;
177    private static final Object latch = new Object();
178    private static AtomicBoolean useLatch = new AtomicBoolean(false);
179
180    public static void resume() {
181      useLatch.set(false);
182      synchronized (latch) {
183        latch.notifyAll();
184      }
185    }
186
187    public static void pause() {
188      useLatch.set(true);
189    }
190
191    public static void await() throws InterruptedException {
192      if (useLatch.get()) {
193        LOG.info("Waiting on latch");
194        synchronized (latch) {
195          latch.wait();
196        }
197        LOG.info("Waited on latch, now proceeding");
198      }
199    }
200
201    public static int getBatchCount() {
202      return batchCount.get();
203    }
204
205    public static void setBatchCount(int i) {
206      LOG.info("SetBatchCount=" + i + ", old=" + getBatchCount());
207      batchCount.set(i);
208    }
209
210    public static int getEntriesCount() {
211      return entriesCount;
212    }
213
214    public static void setEntriesCount(int i) {
215      LOG.info("SetEntriesCount=" + i);
216      entriesCount = i;
217    }
218
219    @Override
220    public boolean replicate(ReplicateContext replicateContext) {
221      try {
222        await();
223      } catch (InterruptedException e) {
224        LOG.warn("Interrupted waiting for latch", e);
225      }
226      return super.replicate(replicateContext);
227    }
228
229    @Override
230    protected Callable<Integer> createReplicator(List<Entry> entries, int ordinal) {
231      return () -> {
232        int batchIndex = replicateEntries(entries, ordinal);
233        entriesCount += entries.size();
234        int count = batchCount.incrementAndGet();
235        LOG.info(
236          "Completed replicating batch " + System.identityHashCode(entries) + " count=" + count);
237        return batchIndex;
238      };
239    }
240  }
241
242  public static class FailureInjectingReplicationEndpointForTest
243      extends ReplicationEndpointForTest {
244    private final AtomicBoolean failNext = new AtomicBoolean(false);
245
246    @Override
247    protected Callable<Integer> createReplicator(List<Entry> entries, int ordinal) {
248      return () -> {
249        if (failNext.compareAndSet(false, true)) {
250          int batchIndex = replicateEntries(entries, ordinal);
251          entriesCount += entries.size();
252          int count = batchCount.incrementAndGet();
253          LOG.info(
254            "Completed replicating batch " + System.identityHashCode(entries) + " count=" + count);
255          return batchIndex;
256        } else if (failNext.compareAndSet(true, false)) {
257          throw new ServiceException("Injected failure");
258        }
259        return ordinal;
260      };
261    }
262  }
