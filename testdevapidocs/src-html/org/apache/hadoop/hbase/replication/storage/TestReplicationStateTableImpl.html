053@Category({ ReplicationTests.class, MediumTests.class })
054public class TestReplicationStateTableImpl extends TestReplicationStateBasic {
055
056  @ClassRule
057  public static final HBaseClassTestRule CLASS_RULE =
058      HBaseClassTestRule.forClass(TestReplicationStateTableImpl.class);
059
060  private static Configuration conf;
061  private static HBaseTestingUtility utility = new HBaseTestingUtility();
062  private static ZKWatcher zkw;
063  private static Connection connection;
064
065  @BeforeClass
066  public static void setUpBeforeClass() throws Exception {
067    conf = utility.getConfiguration();
068    conf.setBoolean(HConstants.REPLICATION_BULKLOAD_ENABLE_KEY, true);
069    conf.set(HConstants.REPLICATION_CLUSTER_ID, "12345");
070    utility.startMiniCluster();
071
072    // After the HBase Mini cluster startup, we set the storage implementation to table based
073    // implementation. Otherwise, we cannot setup the HBase Mini Cluster because the master will
074    // list peers before finish its initialization, and if master cannot finish initialization, the
075    // meta cannot be online, in other hand, if meta cannot be online, the list peers never success
076    // when using table based replication. a dead loop happen.
077    // Our UTs are written for testing storage layer, so no problem here.
078    conf.set(ReplicationStorageFactory.REPLICATION_PEER_STORAGE_IMPL,
079      TableReplicationPeerStorage.class.getName());
080    conf.set(ReplicationStorageFactory.REPLICATION_QUEUE_STORAGE_IMPL,
081      TableReplicationQueueStorage.class.getName());
082
083    zkw = utility.getZooKeeperWatcher();
084    connection = ConnectionFactory.createConnection(conf);
085
086    KEY_ONE = initPeerClusterState("/hbase1");
087    KEY_TWO = initPeerClusterState("/hbase2");
088  }
089
090  private static String initPeerClusterState(String baseZKNode)
091      throws IOException, KeeperException {
092    // Add a dummy region server and set up the cluster id
093    Configuration testConf = new Configuration(conf);
094    testConf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, baseZKNode);
095    ZKWatcher zkw1 = new ZKWatcher(testConf, "test1", null);
096    String fakeRs = ZNodePaths.joinZNode(zkw1.znodePaths.rsZNode, "hostname1.example.org:1234");
097    ZKUtil.createWithParents(zkw1, fakeRs);
098    ZKClusterId.setClusterId(zkw1, new ClusterId());
099    return ZKConfig.getZooKeeperClusterKey(testConf);
100  }
101
102  @Before
103  public void setUp() throws IOException {
104    rqs = ReplicationStorageFactory.getReplicationQueueStorage(zkw, conf);
105    rp = ReplicationFactory.getReplicationPeers(zkw, conf);
106    OUR_KEY = ZKConfig.getZooKeeperClusterKey(conf);
107
108    // Create hbase:replication meta table.
109    try (Admin admin = connection.getAdmin()) {
110      TableDescriptor table =
111          TableReplicationStorageBase.createReplicationTableDescBuilder(conf).build();
112      admin.createTable(table);
113    }
114  }
115
116  @After
117  public void tearDown() throws KeeperException, IOException {
118    // Drop the hbase:replication meta table.
119    utility.deleteTable(TableReplicationStorageBase.REPLICATION_TABLE);
120  }
121
122  @AfterClass
123  public static void tearDownAfterClass() throws Exception {
124    if (connection != null) {
125      IOUtils.closeQuietly(connection);
126    }
127    utility.shutdownMiniZKCluster();
128  }
129}
