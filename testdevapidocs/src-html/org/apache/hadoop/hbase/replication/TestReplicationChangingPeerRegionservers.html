043/**
044 * Test handling of changes to the number of a peer's regionservers.
045 */
046@Category({ReplicationTests.class, LargeTests.class})
047public class TestReplicationChangingPeerRegionservers extends TestReplicationBase {
048
049  private static final Log LOG = LogFactory.getLog(TestReplicationChangingPeerRegionservers.class);
050
051  /**
052   * @throws java.lang.Exception
053   */
054  @Before
055  public void setUp() throws Exception {
056    // Starting and stopping replication can make us miss new logs,
057    // rolling like this makes sure the most recent one gets added to the queue
058    for (JVMClusterUtil.RegionServerThread r :
059                          utility1.getHBaseCluster().getRegionServerThreads()) {
060      utility1.getHBaseAdmin().rollWALWriter(r.getRegionServer().getServerName());
061    }
062    utility1.deleteTableData(tableName);
063    // truncating the table will send one Delete per row to the slave cluster
064    // in an async fashion, which is why we cannot just call deleteTableData on
065    // utility2 since late writes could make it to the slave in some way.
066    // Instead, we truncate the first table and wait for all the Deletes to
067    // make it to the slave.
068    Scan scan = new Scan();
069    int lastCount = 0;
070    for (int i = 0; i < NB_RETRIES; i++) {
071      if (i == NB_RETRIES - 1) {
072        fail("Waited too much time for truncate");
073      }
074      ResultScanner scanner = htable2.getScanner(scan);
075      Result[] res = scanner.next(NB_ROWS_IN_BIG_BATCH);
076      scanner.close();
077      if (res.length != 0) {
078        if (res.length < lastCount) {
079          i--; // Don't increment timeout if we make progress
080        }
081        lastCount = res.length;
082        LOG.info("Still got " + res.length + " rows");
083        Thread.sleep(SLEEP_TIME);
084      } else {
085        break;
086      }
087    }
088  }
089
090  @Test(timeout = 300000)
091  public void testChangingNumberOfPeerRegionServers() throws IOException, InterruptedException {
092
093    LOG.info("testSimplePutDelete");
094    MiniHBaseCluster peerCluster = utility2.getMiniHBaseCluster();
095    int numRS = peerCluster.getRegionServerThreads().size();
096
097    doPutTest(Bytes.toBytes(1));
098
099    int rsToStop = peerCluster.getServerWithMeta() == 0 ? 1 : 0;
100    peerCluster.stopRegionServer(rsToStop);
101    peerCluster.waitOnRegionServer(rsToStop);
102
103    // Sanity check
104    assertEquals(numRS - 1, peerCluster.getRegionServerThreads().size());
105
106    doPutTest(Bytes.toBytes(2));
107
108    peerCluster.startRegionServer();
109
110    // Sanity check
111    assertEquals(numRS, peerCluster.getRegionServerThreads().size());
112
113    doPutTest(Bytes.toBytes(3));
114
115  }
116
117  private void doPutTest(byte[] row) throws IOException, InterruptedException {
118    Put put = new Put(row);
119    put.addColumn(famName, row, row);
120
121    if (htable1 == null) {
122      htable1 = utility1.getConnection().getTable(tableName);
123    }
124
125    htable1.put(put);
126
127    Get get = new Get(row);
128    for (int i = 0; i < NB_RETRIES; i++) {
129      if (i == NB_RETRIES - 1) {
130        fail("Waited too much time for put replication");
131      }
132      Result res = htable2.get(get);
133      if (res.size() == 0) {
134        LOG.info("Row not available");
135        Thread.sleep(SLEEP_TIME);
136      } else {
137        assertArrayEquals(res.value(), row);
138        break;
139      }
140    }
141  }
