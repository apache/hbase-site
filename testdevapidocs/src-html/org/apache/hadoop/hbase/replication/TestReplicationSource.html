101  /**
102   * Sanity check that we can move logs around while we are reading
103   * from them. Should this test fail, ReplicationSource would have a hard
104   * time reading logs that are being archived.
105   * @throws Exception
106   */
107  @Test
108  public void testLogMoving() throws Exception{
109    Path logPath = new Path(logDir, "log");
110    if (!FS.exists(logDir)) FS.mkdirs(logDir);
111    if (!FS.exists(oldLogDir)) FS.mkdirs(oldLogDir);
112    WALProvider.Writer writer = WALFactory.createWALWriter(FS, logPath,
113        TEST_UTIL.getConfiguration());
114    for(int i = 0; i < 3; i++) {
115      byte[] b = Bytes.toBytes(Integer.toString(i));
116      KeyValue kv = new KeyValue(b,b,b);
117      WALEdit edit = new WALEdit();
118      edit.add(kv);
119      WALKey key = new WALKey(b, TableName.valueOf(b), 0, 0,
120          HConstants.DEFAULT_CLUSTER_ID);
121      writer.append(new WAL.Entry(key, edit));
122      writer.sync();
123    }
124    writer.close();
125
126    WAL.Reader reader = WALFactory.createReader(FS, logPath, TEST_UTIL.getConfiguration());
127    WAL.Entry entry = reader.next();
128    assertNotNull(entry);
129
130    Path oldLogPath = new Path(oldLogDir, "log");
131    FS.rename(logPath, oldLogPath);
132
133    entry = reader.next();
134    assertNotNull(entry);
135
136    entry = reader.next();
137    entry = reader.next();
138
139    assertNull(entry);
140    reader.close();
141  }

143  /**
144   * Tests that {@link ReplicationSource#terminate(String)} will timeout properly
145   */
146  @Test
147  public void testTerminateTimeout() throws Exception {
148    ReplicationSource source = new ReplicationSource();
149    ReplicationEndpoint replicationEndpoint = new HBaseInterClusterReplicationEndpoint() {
150      @Override
151      protected void doStart() {
152        notifyStarted();
153      }
154
155      @Override
156      protected void doStop() {
157        // not calling notifyStopped() here causes the caller of stop() to get a Future that never
158        // completes
159      }
160    };
161    replicationEndpoint.startAsync();
162    ReplicationPeers mockPeers = Mockito.mock(ReplicationPeers.class);
163    ReplicationPeer mockPeer = Mockito.mock(ReplicationPeer.class);
164    Mockito.when(mockPeer.getPeerBandwidth()).thenReturn(0L);
165    Configuration testConf = HBaseConfiguration.create();
166    testConf.setInt("replication.source.maxretriesmultiplier", 1);
167    ReplicationSourceManager manager = Mockito.mock(ReplicationSourceManager.class);
168    Mockito.when(manager.getTotalBufferUsed()).thenReturn(new AtomicLong());
169    source.init(testConf, null, manager, null, mockPeers, null, "testPeer",
170        null, replicationEndpoint, null);
171    ExecutorService executor = Executors.newSingleThreadExecutor();
172    Future<?> future = executor.submit(new Runnable() {
173
174      @Override
175      public void run() {
176        source.terminate("testing source termination");
177      }
178    });
179    long sleepForRetries = testConf.getLong("replication.source.sleepforretries", 1000);
180    Waiter.waitFor(testConf, sleepForRetries * 2, new Predicate<Exception>() {
181
182      @Override
183      public boolean evaluate() throws Exception {
184        return future.isDone();
185      }
186
187    });
188
189  }

191  /**
192   * Tests that recovered queues are preserved on a regionserver shutdown.
193   * See HBASE-18192
194   * @throws Exception
195   */
196  @Test
197  public void testServerShutdownRecoveredQueue() throws Exception {
198    try {
199      // Ensure single-threaded WAL
200      conf.set("hbase.wal.provider", "defaultProvider");
201      conf.setInt("replication.sleep.before.failover", 2000);
202      // Introduces a delay in regionserver shutdown to give the race condition a chance to kick in.
203      conf.set(HConstants.REGION_SERVER_IMPL, ShutdownDelayRegionServer.class.getName());
204      MiniHBaseCluster cluster = TEST_UTIL.startMiniCluster(2);
205      TEST_UTIL_PEER.startMiniCluster(1);
206
207      HRegionServer serverA = cluster.getRegionServer(0);
208      final ReplicationSourceManager managerA =
209          ((Replication) serverA.getReplicationSourceService()).getReplicationManager();
210      HRegionServer serverB = cluster.getRegionServer(1);
211      final ReplicationSourceManager managerB =
212          ((Replication) serverB.getReplicationSourceService()).getReplicationManager();
213      final Admin admin = TEST_UTIL.getAdmin();
214
215      final String peerId = "TestPeer";
216      admin.addReplicationPeer(peerId,
217          new ReplicationPeerConfig().setClusterKey(TEST_UTIL_PEER.getClusterKey()));
218      // Wait for replication sources to come up
219      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
220        @Override public boolean evaluate() throws Exception {
221          return !(managerA.getSources().isEmpty() || managerB.getSources().isEmpty());
222        }
223      });
224      // Disabling peer makes sure there is at least one log to claim when the server dies
225      // The recovered queue will also stay there until the peer is disabled even if the
226      // WALs it contains have no data.
227      admin.disableReplicationPeer(peerId);
228
229      // Stopping serverA
230      // It's queues should be claimed by the only other alive server i.e. serverB
231      cluster.stopRegionServer(serverA.getServerName());
232      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
233        @Override public boolean evaluate() throws Exception {
234          return managerB.getOldSources().size() == 1;
235        }
236      });
237
238      final HRegionServer serverC = cluster.startRegionServer().getRegionServer();
239      serverC.waitForServerOnline();
240      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
241        @Override public boolean evaluate() throws Exception {
242          return serverC.getReplicationSourceService() != null;
243        }
244      });
245      final ReplicationSourceManager managerC =
246          ((Replication) serverC.getReplicationSourceService()).getReplicationManager();
247      // Sanity check
248      assertEquals(0, managerC.getOldSources().size());
249
250      // Stopping serverB
251      // Now serverC should have two recovered queues:
252      // 1. The serverB's normal queue
253      // 2. serverA's recovered queue on serverB
254      cluster.stopRegionServer(serverB.getServerName());
255      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
256        @Override public boolean evaluate() throws Exception {
257          return managerC.getOldSources().size() == 2;
258        }
259      });
260      admin.enableReplicationPeer(peerId);
261      Waiter.waitFor(conf, 20000, new Waiter.Predicate<Exception>() {
262        @Override public boolean evaluate() throws Exception {
263          return managerC.getOldSources().size() == 0;
264        }
265      });
266    } finally {
267      conf.set(HConstants.REGION_SERVER_IMPL, HRegionServer.class.getName());
268    }
269  }

271  /**
272   * Regionserver implementation that adds a delay on the graceful shutdown.
273   */
274  public static class ShutdownDelayRegionServer extends HRegionServer {
275    public ShutdownDelayRegionServer(Configuration conf) throws IOException, InterruptedException {
276      super(conf);
277    }
278
279    public ShutdownDelayRegionServer(Configuration conf, CoordinatedStateManager csm)
280        throws IOException, InterruptedException {
281      super(conf, csm);
282    }
283
284    @Override
285    protected void stopServiceThreads() {
286      // Add a delay before service threads are shutdown.
287      // This will keep the zookeeper connection alive for the duration of the delay.
288      LOG.info("Adding a delay to the regionserver shutdown");
289      try {
290        Thread.sleep(2000);
291      } catch (InterruptedException ex) {
292        LOG.error("Interrupted while sleeping");
293      }
294      super.stopServiceThreads();
295    }
296  }
