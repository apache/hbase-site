018package org.apache.hadoop.hbase.replication;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023import java.util.EnumSet;
024import java.util.List;
025import org.apache.hadoop.hbase.ClusterMetrics;
026import org.apache.hadoop.hbase.ClusterMetrics.Option;
027import org.apache.hadoop.hbase.HBaseClassTestRule;
028import org.apache.hadoop.hbase.ServerName;
029import org.apache.hadoop.hbase.client.Admin;
030import org.apache.hadoop.hbase.client.Put;
031import org.apache.hadoop.hbase.testclassification.MediumTests;
032import org.apache.hadoop.hbase.testclassification.ReplicationTests;
033import org.apache.hadoop.hbase.util.Bytes;
034import org.junit.ClassRule;
035import org.junit.Test;
036import org.junit.experimental.categories.Category;
037
038@Category({ ReplicationTests.class, MediumTests.class })
039public class TestReplicationStatusBothNormalAndRecoveryLagging extends TestReplicationBase {
040
041  @ClassRule
042  public static final HBaseClassTestRule CLASS_RULE =
043    HBaseClassTestRule.forClass(TestReplicationStatusBothNormalAndRecoveryLagging.class);
044
045  @Test
046  public void testReplicationStatusBothNormalAndRecoveryLagging() throws Exception {
047    utility2.shutdownMiniHBaseCluster();
048    // add some values to cluster 1
049    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
050      Put p = new Put(Bytes.toBytes("row" + i));
051      p.addColumn(famName, Bytes.toBytes("col1"), Bytes.toBytes("val" + i));
052      htable1.put(p);
053    }
054    Thread.sleep(10000);
055    restartHBaseCluster(utility1, 1);
056    Admin hbaseAdmin = utility1.getAdmin();
057    ServerName serverName = utility1.getHBaseCluster().getRegionServer(0).getServerName();
058    Thread.sleep(10000);
059    // add more values to cluster 1, these should cause normal queue to lag
060    for (int i = 0; i < NB_ROWS_IN_BATCH; i++) {
061      Put p = new Put(Bytes.toBytes("row" + i));
062      p.addColumn(famName, Bytes.toBytes("col1"), Bytes.toBytes("val" + i));
063      htable1.put(p);
064    }
065    Thread.sleep(10000);
066    ClusterMetrics metrics = hbaseAdmin.getClusterMetrics(EnumSet.of(Option.LIVE_SERVERS));
067    List<ReplicationLoadSource> loadSources =
068      metrics.getLiveServerMetrics().get(serverName).getReplicationLoadSourceList();
069    assertEquals(2, loadSources.size());
070    boolean foundRecovery = false;
071    boolean foundNormal = false;
072    for (ReplicationLoadSource loadSource : loadSources) {
073      if (loadSource.isRecovered()) {
074        foundRecovery = true;
075      } else {
076        foundNormal = true;
077      }
078      assertTrue(loadSource.hasEditsSinceRestart());
079      assertEquals(0, loadSource.getTimestampOfLastShippedOp());
080      assertTrue(loadSource.getReplicationLag() > 0);
081    }
082    assertTrue("No normal queue found.", foundNormal);
083    assertTrue("No recovery queue found.", foundRecovery);
084  }
085}
