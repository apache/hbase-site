131  @Test(timeout=300000)
132  public void testMultiSlaveReplication() throws Exception {
133    LOG.info("testCyclicReplication");
134    MiniHBaseCluster master = utility1.startMiniCluster();
135    utility2.startMiniCluster();
136    utility3.startMiniCluster();
137    ReplicationAdmin admin1 = new ReplicationAdmin(conf1);
138
139    utility1.getHBaseAdmin().createTable(table);
140    utility2.getHBaseAdmin().createTable(table);
141    utility3.getHBaseAdmin().createTable(table);
142    Table htable1 = utility1.getConnection().getTable(tableName);
143    htable1.setWriteBufferSize(1024);
144    Table htable2 = utility2.getConnection().getTable(tableName);
145    htable2.setWriteBufferSize(1024);
146    Table htable3 = utility3.getConnection().getTable(tableName);
147    htable3.setWriteBufferSize(1024);
148
149    ReplicationPeerConfig rpc = new ReplicationPeerConfig();
150    rpc.setClusterKey(utility2.getClusterKey());
151    admin1.addPeer("1", rpc, null);
152
153    // put "row" and wait 'til it got around, then delete
154    putAndWait(row, famName, htable1, htable2);
155    deleteAndWait(row, htable1, htable2);
156    // check it wasn't replication to cluster 3
157    checkRow(row,0,htable3);
158
159    putAndWait(row2, famName, htable1, htable2);
160
161    // now roll the region server's logs
162    rollWALAndWait(utility1, htable1.getName(), row2);
163
164    // after the log was rolled put a new row
165    putAndWait(row3, famName, htable1, htable2);
166
167    rpc = new ReplicationPeerConfig();
168    rpc.setClusterKey(utility3.getClusterKey());
169    admin1.addPeer("2", rpc, null);
170
171    // put a row, check it was replicated to all clusters
172    putAndWait(row1, famName, htable1, htable2, htable3);
173    // delete and verify
174    deleteAndWait(row1, htable1, htable2, htable3);
175
176    // make sure row2 did not get replicated after
177    // cluster 3 was added
178    checkRow(row2,0,htable3);
179
180    // row3 will get replicated, because it was in the
181    // latest log
182    checkRow(row3,1,htable3);
183
184    Put p = new Put(row);
185    p.addColumn(famName, row, row);
186    htable1.put(p);
187    // now roll the logs again
188    rollWALAndWait(utility1, htable1.getName(), row);
189
190    // cleanup "row2", also conveniently use this to wait replication
191    // to finish
192    deleteAndWait(row2, htable1, htable2, htable3);
193    // Even if the log was rolled in the middle of the replication
194    // "row" is still replication.
195    checkRow(row, 1, htable2);
196    // Replication thread of cluster 2 may be sleeping, and since row2 is not there in it, 
197    // we should wait before checking.
198    checkWithWait(row, 1, htable3);
199
200    // cleanup the rest
201    deleteAndWait(row, htable1, htable2, htable3);
202    deleteAndWait(row3, htable1, htable2, htable3);
203
204    utility3.shutdownMiniCluster();
205    utility2.shutdownMiniCluster();
206    utility1.shutdownMiniCluster();
207  }

209  private void rollWALAndWait(final HBaseTestingUtility utility, final TableName table,
210      final byte[] row) throws IOException {
211    final Admin admin = utility.getHBaseAdmin();
212    final MiniHBaseCluster cluster = utility.getMiniHBaseCluster();
213
214    // find the region that corresponds to the given row.
215    HRegion region = null;
216    for (HRegion candidate : cluster.getRegions(table)) {
217      if (HRegion.rowIsInRange(candidate.getRegionInfo(), row)) {
218        region = candidate;
219        break;
220      }
221    }
222    assertNotNull("Couldn't find the region for row '" + Arrays.toString(row) + "'", region);
223
224    final CountDownLatch latch = new CountDownLatch(1);
225
226    // listen for successful log rolls
227    final WALActionsListener listener = new WALActionsListener.Base() {
228          @Override
229          public void postLogRoll(final Path oldPath, final Path newPath) throws IOException {
230            latch.countDown();
231          }
232        };
233    region.getWAL().registerWALActionsListener(listener);
234
235    // request a roll
236    admin.rollWALWriter(cluster.getServerHoldingRegion(region.getTableDesc().getTableName(),
237      region.getRegionInfo().getRegionName()));
238
239    // wait
240    try {
241      latch.await();
242    } catch (InterruptedException exception) {
243      LOG.warn("Interrupted while waiting for the wal of '" + region + "' to roll. If later " +
244          "replication tests fail, it's probably because we should still be waiting.");
245      Thread.currentThread().interrupt();
246    }
247    region.getWAL().unregisterWALActionsListener(listener);
248  }

251  private void checkWithWait(byte[] row, int count, Table table) throws Exception {
252    Get get = new Get(row);
253    for (int i = 0; i < NB_RETRIES; i++) {
254      if (i == NB_RETRIES - 1) {
255        fail("Waited too much time while getting the row.");
256      }
257      boolean rowReplicated = false;
258      Result res = table.get(get);
259      if (res.size() >= 1) {
260        LOG.info("Row is replicated");
261        rowReplicated = true;
262        assertEquals("Table '" + table + "' did not have the expected number of  results.",
263            count, res.size());
264        break;
265      }
266      if (rowReplicated) {
267        break;
268      } else {
269        Thread.sleep(SLEEP_TIME);
270      }
271    }
272  }
273  
274  private void checkRow(byte[] row, int count, Table... tables) throws IOException {
275    Get get = new Get(row);
276    for (Table table : tables) {
277      Result res = table.get(get);
278      assertEquals("Table '" + table + "' did not have the expected number of results.",
279          count, res.size());
280    }
281  }
282
283  private void deleteAndWait(byte[] row, Table source, Table... targets)
284  throws Exception {
285    Delete del = new Delete(row);
286    source.delete(del);
287
288    Get get = new Get(row);
289    for (int i = 0; i < NB_RETRIES; i++) {
290      if (i==NB_RETRIES-1) {
291        fail("Waited too much time for del replication");
292      }
293      boolean removedFromAll = true;
294      for (Table target : targets) {
295        Result res = target.get(get);
296        if (res.size() >= 1) {
297          LOG.info("Row not deleted");
298          removedFromAll = false;
299          break;
300        }
301      }
302      if (removedFromAll) {
303        break;
304      } else {
305        Thread.sleep(SLEEP_TIME);        
306      }
307    }
308  }
309
310  private void putAndWait(byte[] row, byte[] fam, Table source, Table... targets)
311  throws Exception {
312    Put put = new Put(row);
313    put.addColumn(fam, row, row);
314    source.put(put);
315
316    Get get = new Get(row);
317    for (int i = 0; i < NB_RETRIES; i++) {
318      if (i==NB_RETRIES-1) {
319        fail("Waited too much time for put replication");
320      }
321      boolean replicatedToAll = true;
322      for (Table target : targets) {
323        Result res = target.get(get);
324        if (res.size() == 0) {
325          LOG.info("Row not available");
326          replicatedToAll = false;
327          break;
328        } else {
329          assertArrayEquals(res.value(), row);
330        }
331      }
332      if (replicatedToAll) {
333        break;
334      } else {
335        Thread.sleep(SLEEP_TIME);
336      }
337    }
338  }
