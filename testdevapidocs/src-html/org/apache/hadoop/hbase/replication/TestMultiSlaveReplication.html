136  @Test
137  public void testMultiSlaveReplication() throws Exception {
138    LOG.info("testCyclicReplication");
139    MiniHBaseCluster master = utility1.startMiniCluster();
140    utility2.startMiniCluster();
141    utility3.startMiniCluster();
142    try (Connection conn = ConnectionFactory.createConnection(conf1);
143      Admin admin1 = conn.getAdmin()) {
144      utility1.getAdmin().createTable(table);
145      utility2.getAdmin().createTable(table);
146      utility3.getAdmin().createTable(table);
147      Table htable1 = utility1.getConnection().getTable(tableName);
148      Table htable2 = utility2.getConnection().getTable(tableName);
149      Table htable3 = utility3.getConnection().getTable(tableName);
150
151      ReplicationPeerConfig rpc = new ReplicationPeerConfig();
152      rpc.setClusterKey(utility2.getClusterKey());
153      admin1.addReplicationPeer("1", rpc);
154
155      // put "row" and wait 'til it got around, then delete
156      putAndWait(row, famName, htable1, htable2);
157      deleteAndWait(row, htable1, htable2);
158      // check it wasn't replication to cluster 3
159      checkRow(row, 0, htable3);
160
161      putAndWait(row2, famName, htable1, htable2);
162
163      // now roll the region server's logs
164      rollWALAndWait(utility1, htable1.getName(), row2);
165
166      // after the log was rolled put a new row
167      putAndWait(row3, famName, htable1, htable2);
168
169      rpc = new ReplicationPeerConfig();
170      rpc.setClusterKey(utility3.getClusterKey());
171      admin1.addReplicationPeer("2", rpc);
172
173      // put a row, check it was replicated to all clusters
174      putAndWait(row1, famName, htable1, htable2, htable3);
175      // delete and verify
176      deleteAndWait(row1, htable1, htable2, htable3);
177
178      // make sure row2 did not get replicated after
179      // cluster 3 was added
180      checkRow(row2, 0, htable3);
181
182      // row3 will get replicated, because it was in the
183      // latest log
184      checkRow(row3, 1, htable3);
185
186      Put p = new Put(row);
187      p.addColumn(famName, row, row);
188      htable1.put(p);
189      // now roll the logs again
190      rollWALAndWait(utility1, htable1.getName(), row);
191
192      // cleanup "row2", also conveniently use this to wait replication
193      // to finish
194      deleteAndWait(row2, htable1, htable2, htable3);
195      // Even if the log was rolled in the middle of the replication
196      // "row" is still replication.
197      checkRow(row, 1, htable2);
198      // Replication thread of cluster 2 may be sleeping, and since row2 is not there in it,
199      // we should wait before checking.
200      checkWithWait(row, 1, htable3);
201
202      // cleanup the rest
203      deleteAndWait(row, htable1, htable2, htable3);
204      deleteAndWait(row3, htable1, htable2, htable3);
205
206      utility3.shutdownMiniCluster();
207      utility2.shutdownMiniCluster();
208      utility1.shutdownMiniCluster();
209    }
210  }

212  private void rollWALAndWait(final HBaseTestingUtility utility, final TableName table,
213      final byte[] row) throws IOException {
214    final Admin admin = utility.getAdmin();
215    final MiniHBaseCluster cluster = utility.getMiniHBaseCluster();
216
217    // find the region that corresponds to the given row.
218    HRegion region = null;
219    for (HRegion candidate : cluster.getRegions(table)) {
220      if (HRegion.rowIsInRange(candidate.getRegionInfo(), row)) {
221        region = candidate;
222        break;
223      }
224    }
225    assertNotNull("Couldn't find the region for row '" + Arrays.toString(row) + "'", region);
226
227    final CountDownLatch latch = new CountDownLatch(1);
228
229    // listen for successful log rolls
230    final WALActionsListener listener = new WALActionsListener() {
231          @Override
232          public void postLogRoll(final Path oldPath, final Path newPath) throws IOException {
233            latch.countDown();
234          }
235        };
236    region.getWAL().registerWALActionsListener(listener);
237
238    // request a roll
239    admin.rollWALWriter(cluster.getServerHoldingRegion(region.getTableDescriptor().getTableName(),
240      region.getRegionInfo().getRegionName()));
241
242    // wait
243    try {
244      latch.await();
245    } catch (InterruptedException exception) {
246      LOG.warn("Interrupted while waiting for the wal of '" + region + "' to roll. If later " +
247          "replication tests fail, it's probably because we should still be waiting.");
248      Thread.currentThread().interrupt();
249    }
250    region.getWAL().unregisterWALActionsListener(listener);
251  }

254  private void checkWithWait(byte[] row, int count, Table table) throws Exception {
255    Get get = new Get(row);
256    for (int i = 0; i < NB_RETRIES; i++) {
257      if (i == NB_RETRIES - 1) {
258        fail("Waited too much time while getting the row.");
259      }
260      boolean rowReplicated = false;
261      Result res = table.get(get);
262      if (res.size() >= 1) {
263        LOG.info("Row is replicated");
264        rowReplicated = true;
265        assertEquals("Table '" + table + "' did not have the expected number of  results.",
266            count, res.size());
267        break;
268      }
269      if (rowReplicated) {
270        break;
271      } else {
272        Thread.sleep(SLEEP_TIME);
273      }
274    }
275  }
276
277  private void checkRow(byte[] row, int count, Table... tables) throws IOException {
278    Get get = new Get(row);
279    for (Table table : tables) {
280      Result res = table.get(get);
281      assertEquals("Table '" + table + "' did not have the expected number of results.",
282          count, res.size());
283    }
284  }
285
286  private void deleteAndWait(byte[] row, Table source, Table... targets)
287  throws Exception {
288    Delete del = new Delete(row);
289    source.delete(del);
290
291    Get get = new Get(row);
292    for (int i = 0; i < NB_RETRIES; i++) {
293      if (i==NB_RETRIES-1) {
294        fail("Waited too much time for del replication");
295      }
296      boolean removedFromAll = true;
297      for (Table target : targets) {
298        Result res = target.get(get);
299        if (res.size() >= 1) {
300          LOG.info("Row not deleted");
301          removedFromAll = false;
302          break;
303        }
304      }
305      if (removedFromAll) {
306        break;
307      } else {
308        Thread.sleep(SLEEP_TIME);
309      }
310    }
311  }
312
313  private void putAndWait(byte[] row, byte[] fam, Table source, Table... targets)
314  throws Exception {
315    Put put = new Put(row);
316    put.addColumn(fam, row, row);
317    source.put(put);
318
319    Get get = new Get(row);
320    for (int i = 0; i < NB_RETRIES; i++) {
321      if (i==NB_RETRIES-1) {
322        fail("Waited too much time for put replication");
323      }
324      boolean replicatedToAll = true;
325      for (Table target : targets) {
326        Result res = target.get(get);
327        if (res.isEmpty()) {
328          LOG.info("Row not available");
329          replicatedToAll = false;
330          break;
331        } else {
332          assertArrayEquals(res.value(), row);
333        }
334      }
335      if (replicatedToAll) {
336        break;
337      } else {
338        Thread.sleep(SLEEP_TIME);
339      }
340    }
341  }
