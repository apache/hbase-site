192            .getClusterStatus(EnumSet.of(Option.LIVE_SERVERS));

209  }

264  /**
265   * Tests the cyclic replication scenario of 0 -> 1 -> 2 -> 0 by adding and deleting rows to a
266   * table in each clusters and ensuring that the each of these clusters get the appropriate
267   * mutations. It also tests the grouping scenario where a cluster needs to replicate the edits
268   * originating from itself and also the edits that it received using replication from a different
269   * cluster. The scenario is explained in HBASE-9158
270   */
271  @Test(timeout = 300000)
272  public void testCyclicReplication2() throws Exception {
273    LOG.info("testCyclicReplication2");
274    int numClusters = 3;
275    Table[] htables = null;
276    try {
277      startMiniClusters(numClusters);
278      createTableOnClusters(table);
279
280      // Test the replication scenario of 0 -> 1 -> 2 -> 0
281      addPeer("1", 0, 1);
282      addPeer("1", 1, 2);
283      addPeer("1", 2, 0);
284
285      htables = getHTablesOnClusters(tableName);
286
287      // put "row" and wait 'til it got around
288      putAndWait(row, famName, htables[0], htables[2]);
289      putAndWait(row1, famName, htables[1], htables[0]);
290      putAndWait(row2, famName, htables[2], htables[1]);
291
292      deleteAndWait(row, htables[0], htables[2]);
293      deleteAndWait(row1, htables[1], htables[0]);
294      deleteAndWait(row2, htables[2], htables[1]);
295
296      int[] expectedCounts = new int[] { 3, 3, 3 };
297      validateCounts(htables, put, expectedCounts);
298      validateCounts(htables, delete, expectedCounts);
299
300      // Test HBASE-9158
301      disablePeer("1", 2);
302      // we now have an edit that was replicated into cluster originating from
303      // cluster 0
304      putAndWait(row3, famName, htables[0], htables[1]);
305      // now add a local edit to cluster 1
306      htables[1].put(new Put(row4).addColumn(famName, row4, row4));
307      // re-enable replication from cluster 2 to cluster 0
308      enablePeer("1", 2);
309      // without HBASE-9158 the edit for row4 would have been marked with
310      // cluster 0's id
311      // and hence not replicated to cluster 0
312      wait(row4, htables[0], false);
313    } finally {
314      close(htables);
315      shutDownMiniClusters();
316    }
317  }

475  @SuppressWarnings("resource")
476  private void startMiniClusters(int numClusters) throws Exception {
477    Random random = new Random();
478    utilities = new HBaseTestingUtility[numClusters];
479    configurations = new Configuration[numClusters];
480    for (int i = 0; i < numClusters; i++) {
481      Configuration conf = new Configuration(baseConfiguration);
482      conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/" + i + random.nextInt());
483      HBaseTestingUtility utility = new HBaseTestingUtility(conf);
484      if (i == 0) {
485        utility.startMiniZKCluster();
486        miniZK = utility.getZkCluster();
487      } else {
488        utility.setZkCluster(miniZK);
489      }
490      utility.startMiniCluster();
491      utilities[i] = utility;
492      configurations[i] = conf;
493      new ZKWatcher(conf, "cluster" + i, null, true);
494    }
495  }
496
497  private void shutDownMiniClusters() throws Exception {
498    int numClusters = utilities.length;
499    for (int i = numClusters - 1; i >= 0; i--) {
500      if (utilities[i] != null) {
501        utilities[i].shutdownMiniCluster();
502      }
503    }
504    miniZK.shutdown();
505  }

513  private void addPeer(String id, int masterClusterNumber,
514      int slaveClusterNumber) throws Exception {
515    try (Admin admin = ConnectionFactory.createConnection(configurations[masterClusterNumber])
516        .getAdmin()) {
517      admin.addReplicationPeer(id,
518        new ReplicationPeerConfig().setClusterKey(utilities[slaveClusterNumber].getClusterKey()));
519    }
520  }

522  private void addPeer(String id, int masterClusterNumber, int slaveClusterNumber, String tableCfs)
523      throws Exception {
524    try (Admin admin =
525        ConnectionFactory.createConnection(configurations[masterClusterNumber]).getAdmin()) {
526      admin.addReplicationPeer(
527        id,
528        new ReplicationPeerConfig().setClusterKey(utilities[slaveClusterNumber].getClusterKey())
529            .setReplicateAllUserTables(false)
530            .setTableCFsMap(ReplicationPeerConfigUtil.parseTableCFsFromConfig(tableCfs)));
531    }
532  }

571  private void validateCounts(Table[] htables, byte[] type,
572      int[] expectedCounts) throws IOException {
573    for (int i = 0; i < htables.length; i++) {
574      assertEquals(Bytes.toString(type) + " were replicated back ",
575          expectedCounts[i], getCount(htables[i], type));
576    }

579  private int getCount(Table t, byte[] type) throws IOException {
580    Get test = new Get(row);
581    test.setAttribute("count", new byte[] {});
582    Result res = t.get(test);
583    return Bytes.toInt(res.getValue(count, type));
584  }

602      int[] slaveNumbers, byte[] row, byte[] fam, Table[] tables, byte[][][] hfileRanges,
603      int numOfRows, int[] expectedCounts, boolean toValidate) throws Exception {

632  private void wait(int slaveNumber, Table target, int expectedCount)
633      throws IOException, InterruptedException {
634    int count = 0;
635    for (int i = 0; i < NB_RETRIES; i++) {
636      if (i == NB_RETRIES - 1) {
637        fail("Waited too much time for bulkloaded data replication. Current count=" + count
638            + ", expected count=" + expectedCount);
639      }
640      count = utilities[slaveNumber].countRows(target);
641      if (count != expectedCount) {
642        LOG.info("Waiting more time for bulkloaded data replication.");
643        Thread.sleep(SLEEP_TIME);
644      } else {
645        break;
646      }
647    }
648  }
649
650  private void wait(byte[] row, Table target, boolean isDeleted) throws Exception {
651    Get get = new Get(row);
652    for (int i = 0; i < NB_RETRIES; i++) {
653      if (i == NB_RETRIES - 1) {
654        fail("Waited too much time for replication. Row:" + Bytes.toString(row)
655            + ". IsDeleteReplication:" + isDeleted);
656      }
657      Result res = target.get(get);
658      boolean sleep = isDeleted ? res.size() > 0 : res.isEmpty();
659      if (sleep) {
660        LOG.info("Waiting for more time for replication. Row:"
661            + Bytes.toString(row) + ". IsDeleteReplication:" + isDeleted);
662        Thread.sleep(SLEEP_TIME);
663      } else {
664        if (!isDeleted) {
665          assertArrayEquals(res.value(), row);
666        }
667        LOG.info("Obtained row:"
668            + Bytes.toString(row) + ". IsDeleteReplication:" + isDeleted);
669        break;
670      }
671    }
672  }

674  private void rollWALAndWait(final HBaseTestingUtility utility, final TableName table,
675      final byte[] row) throws IOException {
676    final Admin admin = utility.getAdmin();
677    final MiniHBaseCluster cluster = utility.getMiniHBaseCluster();
678
679    // find the region that corresponds to the given row.
680    HRegion region = null;
681    for (HRegion candidate : cluster.getRegions(table)) {
682      if (HRegion.rowIsInRange(candidate.getRegionInfo(), row)) {
683        region = candidate;
684        break;
685      }
686    }
687    assertNotNull("Couldn't find the region for row '" + Arrays.toString(row) + "'", region);
688
689    final CountDownLatch latch = new CountDownLatch(1);
690
691    // listen for successful log rolls
692    final WALActionsListener listener = new WALActionsListener() {
693          @Override
694          public void postLogRoll(final Path oldPath, final Path newPath) throws IOException {
695            latch.countDown();
696          }
697        };
698    region.getWAL().registerWALActionsListener(listener);
699
700    // request a roll
701    admin.rollWALWriter(cluster.getServerHoldingRegion(region.getTableDescriptor().getTableName(),
702      region.getRegionInfo().getRegionName()));
703
704    // wait
705    try {
706      latch.await();
707    } catch (InterruptedException exception) {
708      LOG.warn("Interrupted while waiting for the wal of '" + region + "' to roll. If later " +
709          "replication tests fail, it's probably because we should still be waiting.");
710      Thread.currentThread().interrupt();
711    }
712    region.getWAL().unregisterWALActionsListener(listener);
713  }

715  /**
716   * Use a coprocessor to count puts and deletes. as KVs would be replicated back with the same
717   * timestamp there is otherwise no way to count them.
718   */
719  public static class CoprocessorCounter implements RegionCoprocessor, RegionObserver {
720    private int nCount = 0;
721    private int nDelete = 0;
722
723    @Override
724    public Optional<RegionObserver> getRegionObserver() {
725      return Optional.of(this);
726    }
727
728    @Override
729    public void prePut(final ObserverContext<RegionCoprocessorEnvironment> e, final Put put,
730        final WALEdit edit, final Durability durability) throws IOException {
731      nCount++;
732    }
733
734    @Override
735    public void postDelete(final ObserverContext<RegionCoprocessorEnvironment> c,
736        final Delete delete, final WALEdit edit, final Durability durability) throws IOException {
737      nDelete++;
738    }
739
740    @Override
741    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> c,
742        final Get get, final List<Cell> result) throws IOException {
743      if (get.getAttribute("count") != null) {
744        result.clear();
745        // order is important!
746        result.add(new KeyValue(count, count, delete, Bytes.toBytes(nDelete)));
747        result.add(new KeyValue(count, count, put, Bytes.toBytes(nCount)));
748        c.bypass();
749      }
750    }
751  }
