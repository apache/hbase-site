173  @Test
174  public void testReplicationWithCellTags() throws Exception {
175    LOG.info("testSimplePutDelete");
176    Put put = new Put(ROW);
177    put.setAttribute("visibility", Bytes.toBytes("myTag3"));
178    put.addColumn(FAMILY, ROW, ROW);
179
180    htable1 = utility1.getConnection().getTable(TABLE_NAME);
181    htable1.put(put);
182
183    Get get = new Get(ROW);
184    try {
185      for (int i = 0; i < NB_RETRIES; i++) {
186        if (i == NB_RETRIES - 1) {
187          fail("Waited too much time for put replication");
188        }
189        Result res = htable2.get(get);
190        if (res.isEmpty()) {
191          LOG.info("Row not available");
192          Thread.sleep(SLEEP_TIME);
193        } else {
194          assertArrayEquals(ROW, res.value());
195          assertEquals(1, TestCoprocessorForTagsAtSink.tags.size());
196          Tag tag = TestCoprocessorForTagsAtSink.tags.get(0);
197          assertEquals(TAG_TYPE, tag.getType());
198          break;
199        }
200      }
201    } finally {
202      TestCoprocessorForTagsAtSink.tags = null;
203    }
204  }

206  public static class TestCoprocessorForTagsAtSource implements RegionCoprocessor, RegionObserver {
207    @Override
208    public Optional<RegionObserver> getRegionObserver() {
209      return Optional.of(this);
210    }
211
212    @Override
213    public void prePut(final ObserverContext<RegionCoprocessorEnvironment> e, final Put put,
214        final WALEdit edit, final Durability durability) throws IOException {
215      byte[] attribute = put.getAttribute("visibility");
216      byte[] cf = null;
217      List<Cell> updatedCells = new ArrayList<>();
218      if (attribute != null) {
219        for (List<? extends Cell> edits : put.getFamilyCellMap().values()) {
220          for (Cell cell : edits) {
221            KeyValue kv = KeyValueUtil.ensureKeyValue(cell);
222            if (cf == null) {
223              cf = CellUtil.cloneFamily(kv);
224            }
225            Tag tag = new ArrayBackedTag(TAG_TYPE, attribute);
226            List<Tag> tagList = new ArrayList<>(1);
227            tagList.add(tag);
228
229            KeyValue newKV = new KeyValue(CellUtil.cloneRow(kv), 0, kv.getRowLength(),
230                CellUtil.cloneFamily(kv), 0, kv.getFamilyLength(), CellUtil.cloneQualifier(kv), 0,
231                kv.getQualifierLength(), kv.getTimestamp(),
232                KeyValue.Type.codeToType(kv.getTypeByte()), CellUtil.cloneValue(kv), 0,
233                kv.getValueLength(), tagList);
234            ((List<Cell>) updatedCells).add(newKV);
235          }
236        }
237        put.getFamilyCellMap().remove(cf);
238        // Update the family map
239        put.getFamilyCellMap().put(cf, updatedCells);
240      }
241    }
242  }

244  public static class TestCoprocessorForTagsAtSink implements RegionCoprocessor, RegionObserver {
245    public static List<Tag> tags = null;
246
247    @Override
248    public Optional<RegionObserver> getRegionObserver() {
249      return Optional.of(this);
250    }
251
252    @Override
253    public void postGetOp(ObserverContext<RegionCoprocessorEnvironment> e, Get get,
254        List<Cell> results) throws IOException {
255      if (results.size() > 0) {
256        // Check tag presence in the 1st cell in 1st Result
257        if (!results.isEmpty()) {
258          Cell cell = results.get(0);
259          tags = PrivateCellUtil.getTags(cell);
260        }
261      }
262    }
