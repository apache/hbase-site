095  /**
096   * Does {@link MetaTableAccessor#getRegion(Connection, byte[])} and a write
097   * against hbase:meta while its hosted server is restarted to prove our retrying
098   * works.
099   * @throws IOException
100   * @throws InterruptedException
101   */
102  @Test public void testRetrying()
103  throws IOException, InterruptedException {
104    final TableName name =
105        TableName.valueOf("testRetrying");
106    LOG.info("Started " + name);
107    Table t = UTIL.createMultiRegionTable(name, HConstants.CATALOG_FAMILY);
108    int regionCount = -1;
109    try (RegionLocator r = UTIL.getConnection().getRegionLocator(name)) {
110      regionCount = r.getStartKeys().length;
111    }
112    // Test it works getting a region from just made user table.
113    final List<HRegionInfo> regions =
114      testGettingTableRegions(connection, name, regionCount);
115    MetaTask reader = new MetaTask(connection, "reader") {
116      @Override
117      void metaTask() throws Throwable {
118        testGetRegion(connection, regions.get(0));
119        LOG.info("Read " + regions.get(0).getEncodedName());
120      }
121    };
122    MetaTask writer = new MetaTask(connection, "writer") {
123      @Override
124      void metaTask() throws Throwable {
125        MetaTableAccessor.addRegionToMeta(connection, regions.get(0));
126        LOG.info("Wrote " + regions.get(0).getEncodedName());
127      }
128    };
129    reader.start();
130    writer.start();
131
132    // We're gonna check how it takes. If it takes too long, we will consider
133    //  it as a fail. We can't put that in the @Test tag as we want to close
134    //  the threads nicely
135    final long timeOut = 180000;
136    long startTime = System.currentTimeMillis();
137
138    try {
139      // Make sure reader and writer are working.
140      assertTrue(reader.isProgressing());
141      assertTrue(writer.isProgressing());
142
143      // Kill server hosting meta -- twice  . See if our reader/writer ride over the
144      // meta moves.  They'll need to retry.
145      for (int i = 0; i < 2; i++) {
146        LOG.info("Restart=" + i);
147        UTIL.ensureSomeRegionServersAvailable(2);
148        int index = -1;
149        do {
150          index = UTIL.getMiniHBaseCluster().getServerWithMeta();
151        } while (index == -1 &&
152          startTime + timeOut < System.currentTimeMillis());
153
154        if (index != -1){
155          UTIL.getMiniHBaseCluster().abortRegionServer(index);
156          UTIL.getMiniHBaseCluster().waitOnRegionServer(index);
157        }
158      }
159
160      assertTrue("reader: " + reader.toString(), reader.isProgressing());
161      assertTrue("writer: " + writer.toString(), writer.isProgressing());
162    } catch (IOException e) {
163      throw e;
164    } finally {
165      reader.stop = true;
166      writer.stop = true;
167      reader.join();
168      writer.join();
169      t.close();
170    }
171    long exeTime = System.currentTimeMillis() - startTime;
172    assertTrue("Timeout: test took " + exeTime / 1000 + " sec", exeTime < timeOut);
173  }

175  /**
176   * Thread that runs a MetaTableAccessor task until asked stop.
177   */
178  abstract static class MetaTask extends Thread {
179    boolean stop = false;
180    int count = 0;
181    Throwable t = null;
182    final Connection connection;
183
184    MetaTask(final Connection connection, final String name) {
185      super(name);
186      this.connection = connection;
187    }
188
189    @Override
190    public void run() {
191      try {
192        while(!this.stop) {
193          LOG.info("Before " + this.getName()+ ", count=" + this.count);
194          metaTask();
195          this.count += 1;
196          LOG.info("After " + this.getName() + ", count=" + this.count);
197          Thread.sleep(100);
198        }
199      } catch (Throwable t) {
200        LOG.info(this.getName() + " failed", t);
201        this.t = t;
202      }
203    }
204
205    boolean isProgressing() throws InterruptedException {
206      int currentCount = this.count;
207      while(currentCount == this.count) {
208        if (!isAlive()) return false;
209        if (this.t != null) return false;
210        Thread.sleep(10);
211      }
212      return true;
213    }
214
215    @Override
216    public String toString() {
217      return "count=" + this.count + ", t=" +
218        (this.t == null? "null": this.t.toString());
219    }
220
221    abstract void metaTask() throws Throwable;
222  }

371      // check whether the primary is still there

378      // check whether the primary is still there
379      assertMetaLocation(meta, primary.getRegionName(), serverName0, seqNum0, 0, true);
380      // check whether the replica 1 is still there
381      assertMetaLocation(meta, primary.getRegionName(), serverName1, seqNum1, 1, true);
382      // now check for replica 1
383      assertMetaLocation(meta, primary.getRegionName(), serverName100, seqNum100, 100, true);
384    } finally {
385      meta.close();
386    }

584    HRegionInfo mergedRegionInfo = new HRegionInfo(TableName.valueOf("table_foo"),
585      HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW, false, regionId, 0);
586
587    ServerName sn = ServerName.valueOf("bar", 0, 0);
588    Table meta = MetaTableAccessor.getMetaHTable(connection);
589    try {
590      List<HRegionInfo> regionInfos = Lists.newArrayList(regionInfoA, regionInfoB);
591      MetaTableAccessor.addRegionsToMeta(connection, regionInfos, 1);
592
593      // write the serverName column with a big current time, but set the masters time as even
594      // bigger. When region merge deletes the rows for regionA and regionB, the serverName columns
595      // should not be seen by the following get
596      long serverNameTime = EnvironmentEdgeManager.currentTime()   + 100000000;
597      long masterSystemTime = EnvironmentEdgeManager.currentTime() + 123456789;
598
599      // write the serverName columns
600      MetaTableAccessor.updateRegionLocation(connection, regionInfoA, sn, 1, serverNameTime);
601
602      // assert that we have the serverName column with expected ts
603      Get get = new Get(mergedRegionInfo.getRegionName());
604      Result result = meta.get(get);
605      Cell serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
606          MetaTableAccessor.getServerColumn(0));
607      assertNotNull(serverCell);
608      assertEquals(serverNameTime, serverCell.getTimestamp());
609
610      // now merge the regions, effectively deleting the rows for region a and b.
611      MetaTableAccessor.mergeRegions(connection, mergedRegionInfo,
612        regionInfoA, regionInfoB, sn, 1, masterSystemTime);
613
614      result = meta.get(get);
615      serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
616          MetaTableAccessor.getServerColumn(0));
617      Cell startCodeCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
618        MetaTableAccessor.getStartCodeColumn(0));
619      Cell seqNumCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
620        MetaTableAccessor.getSeqNumColumn(0));
621      assertNull(serverCell);
622      assertNull(startCodeCell);
623      assertNull(seqNumCell);
624    } finally {
625      meta.close();
626    }
627  }

637  public static class SpyingRpcScheduler extends DelegatingRpcScheduler {
638    long numPriorityCalls = 0;
639
640    public SpyingRpcScheduler(RpcScheduler delegate) {
641      super(delegate);
642    }
643
644    @Override
645    public boolean dispatch(CallRunner task) throws IOException, InterruptedException {
646      int priority = task.getCall().getPriority();
647
648      if (priority > HConstants.QOS_THRESHOLD) {
649        numPriorityCalls++;
650      }
651      return super.dispatch(task);
652    }
653  }
