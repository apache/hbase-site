101    Path testdir = UTIL.getDataTestDir(name.getMethodName());
102    TableDescriptor htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName())).build();
103    FileSystem fs = FileSystem.get(UTIL.getConfiguration());

105    Path p0 = fstd.updateTableDescriptor(htd);
106    int i0 = FSTableDescriptors.getTableInfoSequenceId(p0);
107    Path p1 = fstd.updateTableDescriptor(htd);
108    // Assert we cleaned up the old file.
109    assertTrue(!fs.exists(p0));
110    int i1 = FSTableDescriptors.getTableInfoSequenceId(p1);
111    assertTrue(i1 == i0 + 1);
112    Path p2 = fstd.updateTableDescriptor(htd);
113    // Assert we cleaned up the old file.
114    assertTrue(!fs.exists(p1));
115    int i2 = FSTableDescriptors.getTableInfoSequenceId(p2);
116    assertTrue(i2 == i1 + 1);
117    Path p3 = fstd.updateTableDescriptor(htd);
118    // Assert we cleaned up the old file.
119    assertTrue(!fs.exists(p2));
120    int i3 = FSTableDescriptors.getTableInfoSequenceId(p3);
121    assertTrue(i3 == i2 + 1);
122    TableDescriptor descriptor = fstd.get(htd.getTableName());
123    assertEquals(descriptor, htd);
124  }

126  @Test
127  public void testFormatTableInfoSequenceId() {
128    Path p0 = assertWriteAndReadSequenceId(0);
129    // Assert p0 has format we expect.
130    StringBuilder sb = new StringBuilder();
131    for (int i = 0; i < FSTableDescriptors.WIDTH_OF_SEQUENCE_ID; i++) {
132      sb.append("0");
133    }
134    assertEquals(FSTableDescriptors.TABLEINFO_FILE_PREFIX + "." + sb.toString(),
135      p0.getName());
136    // Check a few more.
137    Path p2 = assertWriteAndReadSequenceId(2);
138    Path p10000 = assertWriteAndReadSequenceId(10000);
139    // Get a .tablinfo that has no sequenceid suffix.
140    Path p = new Path(p0.getParent(), FSTableDescriptors.TABLEINFO_FILE_PREFIX);
141    FileStatus fs = new FileStatus(0, false, 0, 0, 0, p);
142    FileStatus fs0 = new FileStatus(0, false, 0, 0, 0, p0);
143    FileStatus fs2 = new FileStatus(0, false, 0, 0, 0, p2);
144    FileStatus fs10000 = new FileStatus(0, false, 0, 0, 0, p10000);
145    Comparator<FileStatus> comparator = FSTableDescriptors.TABLEINFO_FILESTATUS_COMPARATOR;
146    assertTrue(comparator.compare(fs, fs0) > 0);
147    assertTrue(comparator.compare(fs0, fs2) > 0);
148    assertTrue(comparator.compare(fs2, fs10000) > 0);
149  }

184    final String name = this.name.getMethodName();
185    FileSystem fs = FileSystem.get(UTIL.getConfiguration());
186    Path rootdir = UTIL.getDataTestDir(name);
187    FSTableDescriptors fstd = new FSTableDescriptors(UTIL.getConfiguration(), fs, rootdir);
188    TableDescriptor htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name)).build();
189    Path descriptorFile = fstd.updateTableDescriptor(htd);
190    try (FSDataOutputStream out = fs.create(descriptorFile, true)) {
191      out.write(TableDescriptorBuilder.toByteArray(htd));
192    }
193    FSTableDescriptors fstd2 = new FSTableDescriptors(UTIL.getConfiguration(), fs, rootdir);
194    TableDescriptor td2 = fstd2.get(htd.getTableName());
195    assertEquals(htd, td2);
196    FileStatus descriptorFile2 =
197        FSTableDescriptors.getTableInfoPath(fs, fstd2.getTableDir(htd.getTableName()));
198    byte[] buffer = TableDescriptorBuilder.toByteArray(htd);
199    try (FSDataInputStream in = fs.open(descriptorFile2.getPath())) {
200      in.readFully(buffer);
201    }
202    TableDescriptor td3 = TableDescriptorBuilder.parseFrom(buffer);
203    assertEquals(htd, td3);
204  }

206  @Test public void testTableDescriptors()
207  throws IOException, InterruptedException {
208    final String name = this.name.getMethodName();
209    FileSystem fs = FileSystem.get(UTIL.getConfiguration());
210    // Cleanup old tests if any debris laying around.
211    Path rootdir = new Path(UTIL.getDataTestDir(), name);
212    FSTableDescriptors htds = new FSTableDescriptors(UTIL.getConfiguration(), fs, rootdir) {
213      @Override
214      public TableDescriptor get(TableName tablename)
215          throws TableExistsException, FileNotFoundException, IOException {
216        LOG.info(tablename + ", cachehits=" + this.cachehits);
217        return super.get(tablename);
218      }
219    };
220    final int count = 10;
221    // Write out table infos.
222    for (int i = 0; i < count; i++) {
223      htds.createTableDescriptor(TableDescriptorBuilder.newBuilder(TableName.valueOf(name + i)).build());
224    }
225
226    for (int i = 0; i < count; i++) {
227      assertTrue(htds.get(TableName.valueOf(name + i)) !=  null);
228    }
229    for (int i = 0; i < count; i++) {
230      assertTrue(htds.get(TableName.valueOf(name + i)) !=  null);
231    }
232    // Update the table infos
233    for (int i = 0; i < count; i++) {
234      TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(TableName.valueOf(name + i));
235      builder.addColumnFamily(ColumnFamilyDescriptorBuilder.of("" + i));
236      htds.updateTableDescriptor(builder.build());
237    }
238    // Wait a while so mod time we write is for sure different.
239    Thread.sleep(100);
240    for (int i = 0; i < count; i++) {
241      assertTrue(htds.get(TableName.valueOf(name + i)) !=  null);
242    }
243    for (int i = 0; i < count; i++) {
244      assertTrue(htds.get(TableName.valueOf(name + i)) !=  null);
245    }
246    assertEquals(count * 4, htds.invocations);
247    assertTrue("expected=" + (count * 2) + ", actual=" + htds.cachehits,
248      htds.cachehits >= (count * 2));
249  }

306  @Test
307  public void testCacheConsistency()
308    throws IOException, InterruptedException {
309    final String name = this.name.getMethodName();
310    FileSystem fs = FileSystem.get(UTIL.getConfiguration());
311    // Cleanup old tests if any debris laying around.
312    Path rootdir = new Path(UTIL.getDataTestDir(), name);
313    FSTableDescriptors chtds = new FSTableDescriptorsTest(UTIL.getConfiguration(), fs, rootdir);
314    FSTableDescriptors nonchtds = new FSTableDescriptorsTest(UTIL.getConfiguration(), fs,
315      rootdir, false, false);
316
317    final int count = 10;
318    // Write out table infos via non-cached FSTableDescriptors
319    for (int i = 0; i < count; i++) {
320      nonchtds.createTableDescriptor(TableDescriptorBuilder.newBuilder(TableName.valueOf(name + i)).build());
321    }
322
323    // Calls to getAll() won't increase the cache counter, do per table.
324    for (int i = 0; i < count; i++) {
325      assertTrue(chtds.get(TableName.valueOf(name + i)) !=  null);
326    }
327
328    assertTrue(nonchtds.getAll().size() == chtds.getAll().size());
329
330    // add a new entry for hbase:meta
331    TableDescriptor htd = TableDescriptorBuilder.newBuilder(TableName.META_TABLE_NAME).build();
332    nonchtds.createTableDescriptor(htd);
333
334    // hbase:meta will only increase the cachehit by 1
335    assertTrue(nonchtds.getAll().size() == chtds.getAll().size());
336
337    for (Map.Entry<String, TableDescriptor> entry: nonchtds.getAll().entrySet()) {
338      String t = (String) entry.getKey();
339      TableDescriptor nchtd = entry.getValue();
340      assertTrue("expected " + htd.toString() +
341                   " got: " + chtds.get(TableName.valueOf(t)).toString(),
342                 (nchtd.equals(chtds.get(TableName.valueOf(t)))));
343    }
344  }

370  @Test
371  public void testTableInfoFileStatusComparator() {
372    FileStatus bare =
373      new FileStatus(0, false, 0, 0, -1,
374        new Path("/tmp", FSTableDescriptors.TABLEINFO_FILE_PREFIX));
375    FileStatus future =
376      new FileStatus(0, false, 0, 0, -1,
377        new Path("/tmp/tablinfo." + System.currentTimeMillis()));
378    FileStatus farFuture =
379      new FileStatus(0, false, 0, 0, -1,
380        new Path("/tmp/tablinfo." + System.currentTimeMillis() + 1000));
381    FileStatus [] alist = {bare, future, farFuture};
382    FileStatus [] blist = {bare, farFuture, future};
383    FileStatus [] clist = {farFuture, bare, future};
384    Comparator<FileStatus> c = FSTableDescriptors.TABLEINFO_FILESTATUS_COMPARATOR;
385    Arrays.sort(alist, c);
386    Arrays.sort(blist, c);
387    Arrays.sort(clist, c);
388    // Now assert all sorted same in way we want.
389    for (int i = 0; i < alist.length; i++) {
390      assertTrue(alist[i].equals(blist[i]));
391      assertTrue(blist[i].equals(clist[i]));
392      assertTrue(clist[i].equals(i == 0? farFuture: i == 1? future: bare));
393    }
394  }
