096  /**
097   * This creates a table and then corrupts a mob file.  Hbck should quarantine the file.
098   */
099  @Test(timeout=180000)
100  public void testQuarantineCorruptMobFile() throws Exception {
101    TableName table = TableName.valueOf(name.getMethodName());
102    try {
103      setupMobTable(table);
104      assertEquals(ROWKEYS.length, countRows());
105      admin.flush(table);
106
107      FileSystem fs = FileSystem.get(conf);
108      Path mobFile = getFlushedMobFile(fs, table);
109      admin.disableTable(table);
110      // create new corrupt mob file.
111      String corruptMobFile = createMobFileName(mobFile.getName());
112      Path corrupt = new Path(mobFile.getParent(), corruptMobFile);
113      TestHFile.truncateFile(fs, mobFile, corrupt);
114      LOG.info("Created corrupted mob file " + corrupt);
115      HBaseFsck.debugLsr(conf, FSUtils.getRootDir(conf));
116      HBaseFsck.debugLsr(conf, MobUtils.getMobHome(conf));
117
118      // A corrupt mob file doesn't abort the start of regions, so we can enable the table.
119      admin.enableTable(table);
120      HBaseFsck res = HbckTestingUtil.doHFileQuarantine(conf, table);
121      assertEquals(res.getRetCode(), 0);
122      HFileCorruptionChecker hfcc = res.getHFilecorruptionChecker();
123      assertEquals(hfcc.getHFilesChecked(), 4);
124      assertEquals(hfcc.getCorrupted().size(), 0);
125      assertEquals(hfcc.getFailures().size(), 0);
126      assertEquals(hfcc.getQuarantined().size(), 0);
127      assertEquals(hfcc.getMissing().size(), 0);
128      assertEquals(hfcc.getMobFilesChecked(), 5);
129      assertEquals(hfcc.getCorruptedMobFiles().size(), 1);
130      assertEquals(hfcc.getFailureMobFiles().size(), 0);
131      assertEquals(hfcc.getQuarantinedMobFiles().size(), 1);
132      assertEquals(hfcc.getMissedMobFiles().size(), 0);
133      String quarantinedMobFile = hfcc.getQuarantinedMobFiles().iterator().next().getName();
134      assertEquals(corruptMobFile, quarantinedMobFile);
135    } finally {
136      cleanupTable(table);
137    }
138  }
