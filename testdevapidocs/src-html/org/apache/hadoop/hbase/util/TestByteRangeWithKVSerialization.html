018package org.apache.hadoop.hbase.util;
019
020import java.util.ArrayList;
021import java.util.List;
022
023import org.apache.hadoop.hbase.KeyValue;
024import org.apache.hadoop.hbase.Tag;
025import org.apache.hadoop.hbase.testclassification.MiscTests;
026import org.apache.hadoop.hbase.testclassification.SmallTests;
027import org.junit.Assert;
028import org.junit.Test;
029import org.junit.experimental.categories.Category;
030
031@Category({MiscTests.class, SmallTests.class})
032public class TestByteRangeWithKVSerialization {
033
034  static void writeCell(PositionedByteRange pbr, KeyValue kv) throws Exception {
035    pbr.putInt(kv.getKeyLength());
036    pbr.putInt(kv.getValueLength());
037    pbr.put(kv.getBuffer(), kv.getKeyOffset(), kv.getKeyLength());
038    pbr.put(kv.getBuffer(), kv.getValueOffset(), kv.getValueLength());
039    int tagsLen = kv.getTagsLength();
040    pbr.put((byte) (tagsLen >> 8 & 0xff));
041    pbr.put((byte) (tagsLen & 0xff));
042    pbr.put(kv.getTagsArray(), kv.getTagsOffset(), tagsLen);
043    pbr.putVLong(kv.getSequenceId());
044  }
045
046  static KeyValue readCell(PositionedByteRange pbr) throws Exception {
047    int kvStartPos = pbr.getPosition();
048    int keyLen = pbr.getInt();
049    int valLen = pbr.getInt();
050    pbr.setPosition(pbr.getPosition() + keyLen + valLen); // Skip the key and value section
051    int tagsLen = ((pbr.get() & 0xff) << 8) ^ (pbr.get() & 0xff);
052    pbr.setPosition(pbr.getPosition() + tagsLen); // Skip the tags section
053    long mvcc = pbr.getVLong();
054    KeyValue kv = new KeyValue(pbr.getBytes(), kvStartPos,
055        (int) KeyValue.getKeyValueDataStructureSize(keyLen, valLen, tagsLen));
056    kv.setSequenceId(mvcc);
057    return kv;
058  }
059
060  @Test
061  public void testWritingAndReadingCells() throws Exception {
062    final byte[] FAMILY = Bytes.toBytes("f1");
063    final byte[] QUALIFIER = Bytes.toBytes("q1");
064    final byte[] VALUE = Bytes.toBytes("v");
065    int kvCount = 1000000;
066    List<KeyValue> kvs = new ArrayList<KeyValue>(kvCount);
067    int totalSize = 0;
068    Tag[] tags = new Tag[] { new Tag((byte) 1, "tag1") };
069    for (int i = 0; i < kvCount; i++) {
070      KeyValue kv = new KeyValue(Bytes.toBytes(i), FAMILY, QUALIFIER, i, VALUE, tags);
071      kv.setSequenceId(i);
072      kvs.add(kv);
073      totalSize += kv.getLength() + Bytes.SIZEOF_LONG;
074    }
075    PositionedByteRange pbr = new SimplePositionedMutableByteRange(totalSize);
076    for (KeyValue kv : kvs) {
077      writeCell(pbr, kv);
078    }
079
080    PositionedByteRange pbr1 = new SimplePositionedMutableByteRange(pbr.getBytes(), 0,
081        pbr.getPosition());
082    for (int i = 0; i < kvCount; i++) {
083      KeyValue kv = readCell(pbr1);
084      KeyValue kv1 = kvs.get(i);
085      Assert.assertTrue(kv.equals(kv1));
086      Assert.assertTrue(Bytes.equals(kv.getValueArray(), kv.getValueOffset(), kv.getValueLength(),
087          kv1.getValueArray(), kv1.getValueOffset(), kv1.getValueLength()));
088      Assert.assertTrue(Bytes.equals(kv.getTagsArray(), kv.getTagsOffset(),
089          kv.getTagsLength(), kv1.getTagsArray(), kv1.getTagsOffset(),
090          kv1.getTagsLength()));
091      Assert.assertEquals(kv1.getSequenceId(), kv.getSequenceId());
092    }
093  }
094}
