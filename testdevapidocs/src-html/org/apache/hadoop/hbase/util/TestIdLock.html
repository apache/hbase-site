020package org.apache.hadoop.hbase.util;
021
022import static org.junit.Assert.assertTrue;
023
024import java.util.Map;
025import java.util.Random;
026import java.util.concurrent.Callable;
027import java.util.concurrent.ConcurrentHashMap;
028import java.util.concurrent.ExecutorCompletionService;
029import java.util.concurrent.ExecutorService;
030import java.util.concurrent.Executors;
031import java.util.concurrent.Future;
032import java.util.concurrent.TimeUnit;
033
034import org.apache.hadoop.hbase.testclassification.MediumTests;
035import org.apache.hadoop.hbase.testclassification.MiscTests;
036import org.junit.Test;
037import org.junit.experimental.categories.Category;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041@Category({MiscTests.class, MediumTests.class})
042// Medium as it creates 100 threads; seems better to run it isolated
043public class TestIdLock {
044
045  private static final Logger LOG = LoggerFactory.getLogger(TestIdLock.class);
046
047  private static final int NUM_IDS = 16;
048  private static final int NUM_THREADS = 128;
049  private static final int NUM_SECONDS = 15;
050
051  private IdLock idLock = new IdLock();
052
053  private Map<Long, String> idOwner = new ConcurrentHashMap<>();
054
055  private class IdLockTestThread implements Callable<Boolean> {
056
057    private String clientId;
058
059    public IdLockTestThread(String clientId) {
060      this.clientId = clientId;
061    }
062
063    @Override
064    public Boolean call() throws Exception {
065      Thread.currentThread().setName(clientId);
066      Random rand = new Random();
067      long endTime = System.currentTimeMillis() + NUM_SECONDS * 1000;
068      while (System.currentTimeMillis() < endTime) {
069        long id = rand.nextInt(NUM_IDS);
070
071        IdLock.Entry lockEntry = idLock.getLockEntry(id);
072        try {
073          int sleepMs = 1 + rand.nextInt(4);
074          String owner = idOwner.get(id);
075          if (owner != null) {
076            LOG.error("Id " + id + " already taken by " + owner + ", "
077                + clientId + " failed");
078            return false;
079          }
080
081          idOwner.put(id, clientId);
082          Thread.sleep(sleepMs);
083          idOwner.remove(id);
084
085        } finally {
086          idLock.releaseLockEntry(lockEntry);
087        }
088      }
089      return true;
090    }
091
092  }
093
094  @Test
095  public void testMultipleClients() throws Exception {
096    ExecutorService exec = Executors.newFixedThreadPool(NUM_THREADS);
097    try {
098      ExecutorCompletionService<Boolean> ecs = new ExecutorCompletionService<>(exec);
099      for (int i = 0; i < NUM_THREADS; ++i)
100        ecs.submit(new IdLockTestThread("client_" + i));
101      for (int i = 0; i < NUM_THREADS; ++i) {
102        Future<Boolean> result = ecs.take();
103        assertTrue(result.get());
104      }
105      idLock.assertMapEmpty();
106    } finally {
107      exec.shutdown();
108      exec.awaitTermination(5000, TimeUnit.MILLISECONDS);
109    }
110  }
