046@Category({MiscTests.class, SmallTests.class})
047public class TestRegionSizeCalculator {
048
049  private Configuration configuration = new Configuration();
050  private final long megabyte = 1024L * 1024L;
051
052  @Test
053  public void testSimpleTestCase() throws Exception {
054
055    RegionLocator regionLocator = mockRegionLocator("region1", "region2", "region3");
056
057    Admin admin = mockAdmin(
058      mockServer(
059        mockRegion("region1", 123),
060        mockRegion("region3", 1232)
061      ),
062      mockServer(
063        mockRegion("region2",  54321),
064        mockRegion("otherTableRegion", 110)
065      )
066    );
067
068    RegionSizeCalculator calculator = new RegionSizeCalculator(regionLocator, admin);
069
070    assertEquals(123 * megabyte, calculator.getRegionSize("region1".getBytes()));
071    assertEquals(54321 * megabyte, calculator.getRegionSize("region2".getBytes()));
072    assertEquals(1232 * megabyte, calculator.getRegionSize("region3".getBytes()));
073    // if region is not inside our table, it should return 0
074    assertEquals(0 * megabyte, calculator.getRegionSize("otherTableRegion".getBytes()));
075
076    assertEquals(3, calculator.getRegionSizeMap().size());
077  }
078
079
080  /**
081   * When size of region in megabytes is larger than largest possible integer there could be
082   * error caused by lost of precision.
083   * */
084  @Test
085  public void testLargeRegion() throws Exception {
086
087    RegionLocator regionLocator = mockRegionLocator("largeRegion");
088
089    Admin admin = mockAdmin(
090      mockServer(
091        mockRegion("largeRegion", Integer.MAX_VALUE)
092      )
093    );
094
095    RegionSizeCalculator calculator = new RegionSizeCalculator(regionLocator, admin);
096
097    assertEquals(((long) Integer.MAX_VALUE) * megabyte, calculator.getRegionSize("largeRegion".getBytes()));
098  }
099
100  /** When calculator is disabled, it should return 0 for each request.*/
101  @Test
102  public void testDisabled() throws Exception {
103    String regionName = "cz.goout:/index.html";
104    RegionLocator table = mockRegionLocator(regionName);
105
106    Admin admin = mockAdmin(
107      mockServer(
108        mockRegion(regionName, 999)
109      )
110    );
111
112    //first request on enabled calculator
113    RegionSizeCalculator calculator = new RegionSizeCalculator(table, admin);
114    assertEquals(999 * megabyte, calculator.getRegionSize(regionName.getBytes()));
115
116    //then disabled calculator.
117    configuration.setBoolean(RegionSizeCalculator.ENABLE_REGIONSIZECALCULATOR, false);
118    RegionSizeCalculator disabledCalculator = new RegionSizeCalculator(table, admin);
119    assertEquals(0 * megabyte, disabledCalculator.getRegionSize(regionName.getBytes()));
120
121    assertEquals(0, disabledCalculator.getRegionSizeMap().size());
122  }

124  /**
125   * Makes some table with given region names.
126   * */
127  private RegionLocator mockRegionLocator(String... regionNames) throws IOException {
128    RegionLocator mockedTable = Mockito.mock(RegionLocator.class);
129    when(mockedTable.getName()).thenReturn(TableName.valueOf("sizeTestTable"));
130    List<HRegionLocation> regionLocations = new ArrayList<>();
131    when(mockedTable.getAllRegionLocations()).thenReturn(regionLocations);
132
133    for (String regionName : regionNames) {
134      HRegionInfo info = Mockito.mock(HRegionInfo.class);
135      when(info.getRegionName()).thenReturn(regionName.getBytes());
136      regionLocations.add(new HRegionLocation(info, null));//we are not interested in values
137    }
138
139    return mockedTable;
140  }
141
142  /**
143   * Creates mock returning ClusterStatus info about given servers.
144  */
145  private Admin mockAdmin(ServerLoad... servers) throws Exception {
146    //get clusterstatus
147    Admin mockAdmin = Mockito.mock(Admin.class);
148    ClusterStatus clusterStatus = mockCluster(servers);
149    when(mockAdmin.getConfiguration()).thenReturn(configuration);
150    when(mockAdmin.getClusterStatus()).thenReturn(clusterStatus);
151    return mockAdmin;
152  }
153
154  /**
155   * Creates mock of region with given name and size.
156   *
157   * @param  fileSizeMb number of megabytes occupied by region in file store in megabytes
158   * */
159  private RegionLoad mockRegion(String regionName, int fileSizeMb) {
160    RegionLoad region = Mockito.mock(RegionLoad.class);
161    when(region.getName()).thenReturn(regionName.getBytes());
162    when(region.getNameAsString()).thenReturn(regionName);
163    when(region.getStorefileSizeMB()).thenReturn(fileSizeMb);
164    return region;
165  }
166
167  private ClusterStatus mockCluster(ServerLoad[] servers) {
168    List<ServerName> serverNames = new ArrayList<ServerName>();
169
170    ClusterStatus clusterStatus = Mockito.mock(ClusterStatus.class);
171    when(clusterStatus.getServers()).thenReturn(serverNames);
172
173    int serverCounter = 0;
174    for (ServerLoad server : servers) {
175      ServerName serverName = mock(ServerName.class);
176      when(serverName.getServerName()).thenReturn("server" + (serverCounter++));
177      serverNames.add(serverName);
178      when(clusterStatus.getLoad(serverName)).thenReturn(server);
179    }
180
181    return clusterStatus;
182  }
183
184  /** Creates mock of region server with given regions*/
185  private ServerLoad mockServer(RegionLoad... regions) {
186    ServerLoad serverLoad = Mockito.mock(ServerLoad.class);
187    Map<byte[], RegionLoad> regionMap = new TreeMap<byte[], RegionLoad>(Bytes.BYTES_COMPARATOR);
188
189    for (RegionLoad regionName : regions) {
190      regionMap.put(regionName.getName(), regionName);
191    }
192
193    when(serverLoad.getRegionsLoad()).thenReturn(regionMap);
194    return serverLoad;
195  }
