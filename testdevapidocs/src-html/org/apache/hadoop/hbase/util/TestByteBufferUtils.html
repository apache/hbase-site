080  private static void setUnsafe(String fieldName, boolean value) throws Exception {
081    Field field = ByteBufferUtils.class.getDeclaredField(fieldName);
082    field.setAccessible(true);
083    Field modifiersField = Field.class.getDeclaredField("modifiers");
084    modifiersField.setAccessible(true);
085    int oldModifiers = field.getModifiers();
086    modifiersField.setInt(field, oldModifiers & ~Modifier.FINAL);
087    try {
088      field.set(null, value);
089    } finally {
090      modifiersField.setInt(field, oldModifiers);
091    }
092  }

094  static void disableUnsafe() throws Exception {
095    if (ByteBufferUtils.UNSAFE_AVAIL) {
096      setUnsafe(UNSAFE_AVAIL_NAME, false);
097    }
098    if (ByteBufferUtils.UNSAFE_UNALIGNED) {
099      setUnsafe(UNSAFE_UNALIGNED_NAME, false);
100    }
101    assertFalse(ByteBufferUtils.UNSAFE_AVAIL);
102    assertFalse(ByteBufferUtils.UNSAFE_UNALIGNED);
103  }

105  static void detectAvailabilityOfUnsafe() throws Exception {
106    if (ByteBufferUtils.UNSAFE_AVAIL != UnsafeAvailChecker.isAvailable()) {
107      setUnsafe(UNSAFE_AVAIL_NAME, UnsafeAvailChecker.isAvailable());
108    }
109    if (ByteBufferUtils.UNSAFE_UNALIGNED != UnsafeAvailChecker.unaligned()) {
110      setUnsafe(UNSAFE_UNALIGNED_NAME, UnsafeAvailChecker.unaligned());
111    }
112    assertEquals(ByteBufferUtils.UNSAFE_AVAIL, UnsafeAvailChecker.isAvailable());
113    assertEquals(ByteBufferUtils.UNSAFE_UNALIGNED, UnsafeAvailChecker.unaligned());
114  }

138  private static void addNumber(Set<Long> a, long l) {
139    if (l != Long.MIN_VALUE) {
140      a.add(l - 1);
141    }
142    a.add(l);
143    if (l != Long.MAX_VALUE) {
144      a.add(l + 1);
145    }
146    for (long divisor = 3; divisor <= 10; ++divisor) {
147      for (long delta = -1; delta <= 1; ++delta) {
148        a.add(l / divisor + delta);
149      }
150    }
151  }

153  static {
154    SortedSet<Long> a = new TreeSet<>();
155    for (int i = 0; i <= 63; ++i) {
156      long v = (-1L) << i;
157      assertTrue(v < 0);
158      addNumber(a, v);
159      v = (1L << i) - 1;
160      assertTrue(v >= 0);
161      addNumber(a, v);
162    }
163
164    testNumbers = Collections.unmodifiableSet(a);
165    System.err.println("Testing variable-length long serialization using: "
166        + testNumbers + " (count: " + testNumbers.size() + ")");
167    assertEquals(1753, testNumbers.size());
168    assertEquals(Long.MIN_VALUE, a.first().longValue());
169    assertEquals(Long.MAX_VALUE, a.last().longValue());
170  }

434  private void testCopyFromSrcToDestWithThreads(Object input, Object output,
435    List<Integer> lengthes, List<Integer> offsets) throws InterruptedException {
436    assertTrue((input instanceof ByteBuffer) || (input instanceof byte[]));
437    assertTrue((output instanceof ByteBuffer) || (output instanceof byte[]));
438    assertEquals(lengthes.size(), offsets.size());
439
440    final int threads = lengthes.size();
441    CountDownLatch latch = new CountDownLatch(1);
442    List<Runnable> exes = new ArrayList<>(threads);
443    int oldInputPos = (input instanceof ByteBuffer) ? ((ByteBuffer) input).position() : 0;
444    int oldOutputPos = (output instanceof ByteBuffer) ? ((ByteBuffer) output).position() : 0;
445    for (int i = 0; i != threads; ++i) {
446      int offset = offsets.get(i);
447      int length = lengthes.get(i);
448      exes.add(() -> {
449        try {
450          latch.await();
451          if (input instanceof ByteBuffer && output instanceof byte[]) {
452            ByteBufferUtils.copyFromBufferToArray((byte[]) output,
453                (ByteBuffer) input, offset, offset, length);
454          }
455          if (input instanceof byte[] && output instanceof ByteBuffer) {
456            ByteBufferUtils.copyFromArrayToBuffer((ByteBuffer) output,
457                offset, (byte[]) input, offset, length);
458          }
459          if (input instanceof ByteBuffer && output instanceof ByteBuffer) {
460            ByteBufferUtils.copyFromBufferToBuffer((ByteBuffer) input,
461                (ByteBuffer) output, offset, offset, length);
462          }
463        } catch (InterruptedException ex) {
464          throw new RuntimeException(ex);
465        }
466      });
467    }
468    ExecutorService service = Executors.newFixedThreadPool(threads);
469    exes.forEach(service::execute);
470    latch.countDown();
471    service.shutdown();
472    assertTrue(service.awaitTermination(5, TimeUnit.SECONDS));
473    if (input instanceof ByteBuffer) {
474      assertEquals(oldInputPos, ((ByteBuffer) input).position());
475    }
476    if (output instanceof ByteBuffer) {
477      assertEquals(oldOutputPos, ((ByteBuffer) output).position());
478    }
479    String inputString = (input instanceof ByteBuffer) ?
480      Bytes.toString(Bytes.toBytes((ByteBuffer) input)) : Bytes.toString((byte[]) input);
481    String outputString = (output instanceof ByteBuffer) ?
482      Bytes.toString(Bytes.toBytes((ByteBuffer) output)) : Bytes.toString((byte[]) output);
483    assertEquals(inputString, outputString);
484  }

486  @Test
487  public void testCopyFromSrcToDestWithThreads() throws InterruptedException {
488    List<byte[]> words = Arrays.asList(
489      Bytes.toBytes("with"),
490      Bytes.toBytes("great"),
491      Bytes.toBytes("power"),
492      Bytes.toBytes("comes"),
493      Bytes.toBytes("great"),
494      Bytes.toBytes("responsibility")
495    );
496    List<Integer> lengthes = words.stream().map(v -> v.length).collect(Collectors.toList());
497    List<Integer> offsets = new ArrayList<>(words.size());
498    for (int i = 0; i != words.size(); ++i) {
499      offsets.add(words.subList(0, i).stream().mapToInt(v -> v.length).sum());
500    }
501
502    int totalSize = words.stream().mapToInt(v -> v.length).sum();
503    byte[] fullContent = new byte[totalSize];
504    int offset = 0;
505    for (byte[] w : words) {
506      offset = Bytes.putBytes(fullContent, offset, w, 0, w.length);
507    }
508
509    // test copyFromBufferToArray
510    for (ByteBuffer input : Arrays.asList(
511            ByteBuffer.allocateDirect(totalSize),
512            ByteBuffer.allocate(totalSize))) {
513      words.forEach(input::put);
514      byte[] output = new byte[totalSize];
515      testCopyFromSrcToDestWithThreads(input, output, lengthes, offsets);
516    }
517
518    // test copyFromArrayToBuffer
519    for (ByteBuffer output : Arrays.asList(
520            ByteBuffer.allocateDirect(totalSize),
521            ByteBuffer.allocate(totalSize))) {
522      byte[] input = fullContent;
523      testCopyFromSrcToDestWithThreads(input, output, lengthes, offsets);
524    }
525
526    // test copyFromBufferToBuffer
527    for (ByteBuffer input : Arrays.asList(
528            ByteBuffer.allocateDirect(totalSize),
529            ByteBuffer.allocate(totalSize))) {
530      words.forEach(input::put);
531      for (ByteBuffer output : Arrays.asList(
532            ByteBuffer.allocateDirect(totalSize),
533            ByteBuffer.allocate(totalSize))) {
534        testCopyFromSrcToDestWithThreads(input, output, lengthes, offsets);
535      }
536    }
537  }

571  @Test
572  public void testCompareTo() {
573    ByteBuffer bb1 = ByteBuffer.allocate(135);
574    ByteBuffer bb2 = ByteBuffer.allocate(135);
575    byte[] b = new byte[71];
576    fillBB(bb1, (byte) 5);
577    fillBB(bb2, (byte) 5);
578    fillArray(b, (byte) 5);
579    assertEquals(0, ByteBufferUtils.compareTo(bb1, 0, bb1.remaining(), bb2, 0, bb2.remaining()));
580    assertTrue(ByteBufferUtils.compareTo(bb1, 0, bb1.remaining(), b, 0, b.length) > 0);
581    bb2.put(134, (byte) 6);
582    assertTrue(ByteBufferUtils.compareTo(bb1, 0, bb1.remaining(), bb2, 0, bb2.remaining()) < 0);
583    bb2.put(6, (byte) 4);
584    assertTrue(ByteBufferUtils.compareTo(bb1, 0, bb1.remaining(), bb2, 0, bb2.remaining()) > 0);
585    // Assert reverse comparing BB and bytearray works.
586    ByteBuffer bb3 = ByteBuffer.allocate(135);
587    fillBB(bb3, (byte)0);
588    byte[] b3 = new byte[135];
589    fillArray(b3, (byte)1);
590    int result = ByteBufferUtils.compareTo(b3, 0, b3.length, bb3, 0, bb3.remaining());
591    assertTrue(result > 0);
592    result = ByteBufferUtils.compareTo(bb3, 0, bb3.remaining(), b3, 0, b3.length);
593    assertTrue(result < 0);
594
595    byte[] b4 = Bytes.toBytes("123");
596    ByteBuffer bb4 = ByteBuffer.allocate(10 + b4.length);
597    for (int i = 10; i < (bb4.capacity()); ++i) {
598      bb4.put(i, b4[i - 10]);
599    }
600    result = ByteBufferUtils.compareTo(b4, 0, b4.length, bb4, 10, b4.length);
601    assertEquals(0, result);
602  }
