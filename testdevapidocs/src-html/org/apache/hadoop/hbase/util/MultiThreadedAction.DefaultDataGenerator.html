076  /**
077   * Default implementation of LoadTestDataGenerator that uses LoadTestKVGenerator, fixed
078   * set of column families, and random number of columns in range. The table for it can
079   * be created manually or, for example, via
080   * {@link org.apache.hadoop.hbase.HBaseTestingUtility#createPreSplitLoadTestTable(Configuration, TableName, byte[],
081   * org.apache.hadoop.hbase.io.compress.Compression.Algorithm, org.apache.hadoop.hbase.io.encoding.DataBlockEncoding)}
082   */
083  public static class DefaultDataGenerator extends LoadTestDataGenerator {
084    private byte[][] columnFamilies = null;
085    private int minColumnsPerKey;
086    private int maxColumnsPerKey;
087    private final Random random = new Random();
088
089    public DefaultDataGenerator(int minValueSize, int maxValueSize,
090        int minColumnsPerKey, int maxColumnsPerKey, byte[]... columnFamilies) {
091      super(minValueSize, maxValueSize);
092      this.columnFamilies = columnFamilies;
093      this.minColumnsPerKey = minColumnsPerKey;
094      this.maxColumnsPerKey = maxColumnsPerKey;
095    }
096
097    public DefaultDataGenerator(byte[]... columnFamilies) {
098      // Default values for tests that didn't care to provide theirs.
099      this(256, 1024, 1, 10, columnFamilies);
100    }
101
102    @Override
103    public byte[] getDeterministicUniqueKey(long keyBase) {
104      return LoadTestKVGenerator.md5PrefixedKey(keyBase).getBytes();
105    }
106
107    @Override
108    public byte[][] getColumnFamilies() {
109      return columnFamilies;
110    }
111
112    @Override
113    public byte[][] generateColumnsForCf(byte[] rowKey, byte[] cf) {
114      int numColumns = minColumnsPerKey + random.nextInt(maxColumnsPerKey - minColumnsPerKey + 1);
115      byte[][] columns = new byte[numColumns][];
116      for (int i = 0; i < numColumns; ++i) {
117        columns[i] = Integer.toString(i).getBytes();
118      }
119      return columns;
120    }
121
122    @Override
123    public byte[] generateValue(byte[] rowKey, byte[] cf, byte[] column) {
124      return kvGenerator.generateRandomSizeValue(rowKey, cf, column);
125    }
126
127    @Override
128    public boolean verify(byte[] rowKey, byte[] cf, byte[] column, byte[] value) {
129      return LoadTestKVGenerator.verify(value, rowKey, cf, column);
130    }
131
132    @Override
133    public boolean verify(byte[] rowKey, byte[] cf, Set<byte[]> columnSet) {
134      return (columnSet.size() >= minColumnsPerKey) && (columnSet.size() <= maxColumnsPerKey);
135    }
136  }

164  private static String formatTime(long elapsedTime) {
165    String format = String.format("%%0%dd", 2);
166    elapsedTime = elapsedTime / 1000;
167    String seconds = String.format(format, elapsedTime % 60);
168    String minutes = String.format(format, (elapsedTime % 3600) / 60);
169    String hours = String.format(format, elapsedTime / 3600);
170    String time =  hours + ":" + minutes + ":" + seconds;
171    return time;
172  }

174  /** Asynchronously reports progress */
175  private class ProgressReporter implements Runnable {
176
177    private String reporterId = "";
178
179    public ProgressReporter(String id) {
180      this.reporterId = id;
181    }
182
183    @Override
184    public void run() {
185      long startTime = System.currentTimeMillis();
186      long priorNumKeys = 0;
187      long priorCumulativeOpTime = 0;
188      int priorAverageKeysPerSecond = 0;
189
190      // Give other threads time to start.
191      Threads.sleep(REPORTING_INTERVAL_MS);
192
193      while (numThreadsWorking.get() != 0) {
194        String threadsLeft =
195            "[" + reporterId + ":" + numThreadsWorking.get() + "] ";
196        if (numKeys.get() == 0) {
197          LOG.info(threadsLeft + "Number of keys = 0");
198        } else {
199          long numKeys = MultiThreadedAction.this.numKeys.get();
200          long time = System.currentTimeMillis() - startTime;
201          long totalOpTime = totalOpTimeMs.get();
202
203          long numKeysDelta = numKeys - priorNumKeys;
204          long totalOpTimeDelta = totalOpTime - priorCumulativeOpTime;
205
206          double averageKeysPerSecond =
207              (time > 0) ? (numKeys * 1000 / time) : 0;
208
209          LOG.info(threadsLeft
210              + "Keys="
211              + numKeys
212              + ", cols="
213              + StringUtils.humanReadableInt(numCols.get())
214              + ", time="
215              + formatTime(time)
216              + ((numKeys > 0 && time > 0) ? (" Overall: [" + "keys/s= "
217                  + numKeys * 1000 / time + ", latency="
218                  + String.format("%.2f", (double)totalOpTime / (double)numKeys)
219                  + " ms]") : "")
220              + ((numKeysDelta > 0) ? (" Current: [" + "keys/s="
221                  + numKeysDelta * 1000 / REPORTING_INTERVAL_MS + ", latency="
222                  + String.format("%.2f", (double)totalOpTimeDelta / (double)numKeysDelta)
223                  + " ms]") : "")
224              + progressInfo());
225
226          if (streamingCounters) {
227            printStreamingCounters(numKeysDelta,
228                averageKeysPerSecond - priorAverageKeysPerSecond);
229          }
230
231          priorNumKeys = numKeys;
232          priorCumulativeOpTime = totalOpTime;
233          priorAverageKeysPerSecond = (int) averageKeysPerSecond;
234        }
235
236        Threads.sleep(REPORTING_INTERVAL_MS);
237      }
238    }
239
240    private void printStreamingCounters(long numKeysDelta,
241        double avgKeysPerSecondDelta) {
242      // Write stats in a format that can be interpreted as counters by
243      // streaming map-reduce jobs.
244      System.err.println("reporter:counter:numKeys," + reporterId + ","
245          + numKeysDelta);
246      System.err.println("reporter:counter:numCols," + reporterId + ","
247          + numCols.get());
248      System.err.println("reporter:counter:avgKeysPerSecond," + reporterId
249          + "," + (long) (avgKeysPerSecondDelta));
250    }
251  }

253  public void close() {
254    if (connection != null) {
255      try {
256        connection.close();
257      } catch (Exception ex) {
258        LOG.warn("Could not close the connection: " + ex);
259      }
260    }
261  }
262
263  public void waitForFinish() {
264    while (numThreadsWorking.get() != 0) {
265      Threads.sleepWithoutInterrupt(1000);
266    }
267    close();
268  }
269
270  public boolean isDone() {
271    return (numThreadsWorking.get() == 0);
272  }

274  protected void startThreads(Collection<? extends Thread> threads) {
275    numThreadsWorking.addAndGet(threads.size());
276    for (Thread thread : threads) {
277      thread.start();

316  /**
317   * Verifies the result from get or scan using the dataGenerator (that was presumably
318   * also used to generate said result).
319   * @param verifyValues verify that values in the result make sense for row/cf/column combination
320   * @param verifyCfAndColumnIntegrity verify that cf/column set in the result is complete. Note
321   *                                   that to use this multiPut should be used, or verification
322   *                                   has to happen after writes, otherwise there can be races.
323   * @return
324   */
325  public boolean verifyResultAgainstDataGenerator(Result result, boolean verifyValues,
326      boolean verifyCfAndColumnIntegrity) {
327    String rowKeyStr = Bytes.toString(result.getRow());
328    // See if we have any data at all.
329    if (result.isEmpty()) {
330      LOG.error("Error checking data for key [" + rowKeyStr + "], no data returned");
331      printLocations(result);
332      return false;
333    }
334
335    if (!verifyValues && !verifyCfAndColumnIntegrity) {
336      return true; // as long as we have something, we are good.
337    }
338
339    // See if we have all the CFs.
340    byte[][] expectedCfs = dataGenerator.getColumnFamilies();
341    if (verifyCfAndColumnIntegrity && (expectedCfs.length != result.getMap().size())) {
342      LOG.error("Error checking data for key [" + rowKeyStr
343        + "], bad family count: " + result.getMap().size());
344      printLocations(result);
345      return false;
346    }
347
348    // Verify each column family from get in the result.
349    for (byte[] cf : result.getMap().keySet()) {
350      String cfStr = Bytes.toString(cf);
351      Map<byte[], byte[]> columnValues = result.getFamilyMap(cf);
352      if (columnValues == null) {
353        LOG.error("Error checking data for key [" + rowKeyStr
354          + "], no data for family [" + cfStr + "]]");
355        printLocations(result);
356        return false;
357      }
358
359      Map<String, MutationType> mutateInfo = null;
360      if (verifyCfAndColumnIntegrity || verifyValues) {
361        if (!columnValues.containsKey(MUTATE_INFO)) {
362          LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
363            + cfStr + "], column [" + Bytes.toString(MUTATE_INFO) + "]; value is not found");
364          printLocations(result);
365          return false;
366        }
367
368        long cfHash = Arrays.hashCode(cf);
369        // Verify deleted columns, and make up column counts if deleted
370        byte[] mutateInfoValue = columnValues.remove(MUTATE_INFO);
371        mutateInfo = parseMutateInfo(mutateInfoValue);
372        for (Map.Entry<String, MutationType> mutate: mutateInfo.entrySet()) {
373          if (mutate.getValue() == MutationType.DELETE) {
374            byte[] column = Bytes.toBytes(mutate.getKey());
375            long columnHash = Arrays.hashCode(column);
376            long hashCode = cfHash + columnHash;
377            if (hashCode % 2 == 0) {
378              if (columnValues.containsKey(column)) {
379                LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
380                  + cfStr + "], column [" + mutate.getKey() + "]; should be deleted");
381                printLocations(result);
382                return false;
383              }
384              byte[] hashCodeBytes = Bytes.toBytes(hashCode);
385              columnValues.put(column, hashCodeBytes);
386            }
387          }
388        }
389
390        // Verify increment
391        if (!columnValues.containsKey(INCREMENT)) {
392          LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
393            + cfStr + "], column [" + Bytes.toString(INCREMENT) + "]; value is not found");
394          printLocations(result);
395          return false;
396        }
397        long currentValue = Bytes.toLong(columnValues.remove(INCREMENT));
398        if (verifyValues) {
399          long amount = mutateInfo.isEmpty() ? 0 : cfHash;
400          long originalValue = Arrays.hashCode(result.getRow());
401          long extra = currentValue - originalValue;
402          if (extra != 0 && (amount == 0 || extra % amount != 0)) {
403            LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
404              + cfStr + "], column [increment], extra [" + extra + "], amount [" + amount + "]");
405            printLocations(result);
406            return false;
407          }
408          if (amount != 0 && extra != amount) {
409            LOG.warn("Warning checking data for key [" + rowKeyStr + "], column family ["
410              + cfStr + "], column [increment], incremented [" + (extra / amount) + "] times");
411          }
412        }
413
414        // See if we have correct columns.
415        if (verifyCfAndColumnIntegrity
416            && !dataGenerator.verify(result.getRow(), cf, columnValues.keySet())) {
417          String colsStr = "";
418          for (byte[] col : columnValues.keySet()) {
419            if (colsStr.length() > 0) {
420              colsStr += ", ";
421            }
422            colsStr += "[" + Bytes.toString(col) + "]";
423          }
424          LOG.error("Error checking data for key [" + rowKeyStr
425            + "], bad columns for family [" + cfStr + "]: " + colsStr);
426          printLocations(result);
427          return false;
428        }
429        // See if values check out.
430        if (verifyValues) {
431          for (Map.Entry<byte[], byte[]> kv : columnValues.entrySet()) {
432            String column = Bytes.toString(kv.getKey());
433            MutationType mutation = mutateInfo.get(column);
434            boolean verificationNeeded = true;
435            byte[] bytes = kv.getValue();
436            if (mutation != null) {
437              boolean mutationVerified = true;
438              long columnHash = Arrays.hashCode(kv.getKey());
439              long hashCode = cfHash + columnHash;
440              byte[] hashCodeBytes = Bytes.toBytes(hashCode);
441              if (mutation == MutationType.APPEND) {
442                int offset = bytes.length - hashCodeBytes.length;
443                mutationVerified = offset > 0 && Bytes.equals(hashCodeBytes,
444                  0, hashCodeBytes.length, bytes, offset, hashCodeBytes.length);
445                if (mutationVerified) {
446                  int n = 1;
447                  while (true) {
448                    int newOffset = offset - hashCodeBytes.length;
449                    if (newOffset < 0 || !Bytes.equals(hashCodeBytes, 0,
450                        hashCodeBytes.length, bytes, newOffset, hashCodeBytes.length)) {
451                      break;
452                    }
453                    offset = newOffset;
454                    n++;
455                  }
456                  if (n > 1) {
457                    LOG.warn("Warning checking data for key [" + rowKeyStr + "], column family ["
458                      + cfStr + "], column [" + column + "], appended [" + n + "] times");
459                  }
460                  byte[] dest = new byte[offset];
461                  System.arraycopy(bytes, 0, dest, 0, offset);
462                  bytes = dest;
463                }
464              } else if (hashCode % 2 == 0) { // checkAndPut
465                mutationVerified = Bytes.equals(bytes, hashCodeBytes);
466                verificationNeeded = false;
467              }
468              if (!mutationVerified) {
469                LOG.error("Error checking data for key [" + rowKeyStr
470                  + "], mutation checking failed for column family [" + cfStr + "], column ["
471                  + column + "]; mutation [" + mutation + "], hashCode ["
472                  + hashCode + "], verificationNeeded ["
473                  + verificationNeeded + "]");
474                printLocations(result);
475                return false;
476              }
477            } // end of mutation checking
478            if (verificationNeeded &&
479                !dataGenerator.verify(result.getRow(), cf, kv.getKey(), bytes)) {
480              LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
481                + cfStr + "], column [" + column + "], mutation [" + mutation
482                + "]; value of length " + bytes.length);
483              printLocations(result);
484              return false;
485            }
486          }
487        }
488      }
489    }
490    return true;
491  }

493  private void printLocations(Result r) {
494    RegionLocations rl = null;
495    if (r == null) {
496      LOG.info("FAILED FOR null Result");
497      return;
498    }
499    LOG.info("FAILED FOR " + resultToString(r) + " Stale " + r.isStale());
500    if (r.getRow() == null) {
501      return;
502    }
503    try {
504      rl = ((ClusterConnection)connection).locateRegion(tableName, r.getRow(), true, true);
505    } catch (IOException e) {
506      LOG.warn("Couldn't get locations for row " + Bytes.toString(r.getRow()));
507    }
508    HRegionLocation locations[] = rl.getRegionLocations();
509    for (HRegionLocation h : locations) {
510      LOG.info("LOCATION " + h);
511    }
512  }

514  private String resultToString(Result result) {
515    StringBuilder sb = new StringBuilder();
516    sb.append("cells=");
517    if(result.isEmpty()) {
518      sb.append("NONE");
519      return sb.toString();
520    }
521    sb.append("{");
522    boolean moreThanOne = false;
523    for(Cell cell : result.listCells()) {
524      if(moreThanOne) {
525        sb.append(", ");
526      } else {
527        moreThanOne = true;
528      }
529      sb.append(CellUtil.toString(cell, true));
530    }
531    sb.append("}");
532    return sb.toString();
533  }

535  // Parse mutate info into a map of <column name> => <update action>
536  private Map<String, MutationType> parseMutateInfo(byte[] mutateInfo) {
537    Map<String, MutationType> mi = new HashMap<>();
538    if (mutateInfo != null) {
539      String mutateInfoStr = Bytes.toString(mutateInfo);
540      String[] mutations = mutateInfoStr.split("#");
541      for (String mutation: mutations) {
542        if (mutation.isEmpty()) continue;
543        Preconditions.checkArgument(mutation.contains(":"),
544          "Invalid mutation info " + mutation);
545        int p = mutation.indexOf(":");
546        String column = mutation.substring(0, p);
547        MutationType type = MutationType.valueOf(
548          Integer.parseInt(mutation.substring(p+1)));
549        mi.put(column, type);
550      }
551    }
552    return mi;
553  }
