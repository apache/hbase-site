019package org.apache.hadoop.hbase.util;
020
021import java.util.concurrent.locks.ReentrantLock;
022
023import org.apache.hadoop.hbase.testclassification.MiscTests;
024import org.apache.hadoop.hbase.testclassification.SmallTests;
025import org.junit.Assert;
026import org.junit.Test;
027import org.junit.experimental.categories.Category;
028
029@Category({MiscTests.class, SmallTests.class})
030public class TestKeyLocker {
031  @Test
032  public void testLocker(){
033    KeyLocker<String> locker = new KeyLocker<String>();
034    ReentrantLock lock1 = locker.acquireLock("l1");
035    Assert.assertTrue(lock1.isHeldByCurrentThread());
036
037    ReentrantLock lock2 = locker.acquireLock("l2");
038    Assert.assertTrue(lock2.isHeldByCurrentThread());
039    Assert.assertTrue(lock1 != lock2);
040
041    // same key = same lock
042    ReentrantLock lock20 = locker.acquireLock("l2");
043    Assert.assertTrue(lock20 == lock2);
044    Assert.assertTrue(lock2.isHeldByCurrentThread());
045    Assert.assertTrue(lock20.isHeldByCurrentThread());
046
047    // Locks are still reentrant; so with 2 acquires we want two unlocks
048    lock20.unlock();
049    Assert.assertTrue(lock20.isHeldByCurrentThread());
050
051    lock2.unlock();
052    Assert.assertFalse(lock20.isHeldByCurrentThread());
053
054    // The lock object will be garbage-collected
055    // if you free its reference for a long time,
056    // and you will get a new one at the next time.
057    int lock2Hash = System.identityHashCode(lock2);
058    lock2 = null;
059    lock20 = null;
060
061    System.gc();
062    System.gc();
063    System.gc();
064
065    ReentrantLock lock200 = locker.acquireLock("l2");
066    Assert.assertNotEquals(lock2Hash, System.identityHashCode(lock200));
067    lock200.unlock();
068    Assert.assertFalse(lock200.isHeldByCurrentThread());
069
070    // first lock is still there
071    Assert.assertTrue(lock1.isHeldByCurrentThread());
072    lock1.unlock();
073    Assert.assertFalse(lock1.isHeldByCurrentThread());
074  }
075}
