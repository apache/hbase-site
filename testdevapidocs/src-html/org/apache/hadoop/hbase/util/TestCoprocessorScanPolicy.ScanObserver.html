102  @Test
103  public void testBaseCases() throws Exception {
104    TableName tableName = TableName.valueOf("baseCases");
105    if (TEST_UTIL.getAdmin().tableExists(tableName)) {
106      TEST_UTIL.deleteTable(tableName);
107    }
108    Table t = TEST_UTIL.createTable(tableName, F, 10);
109    // insert 3 versions
110    long now = EnvironmentEdgeManager.currentTime();
111    Put p = new Put(R);
112    p.addColumn(F, Q, now, Q);
113    t.put(p);
114    p = new Put(R);
115    p.addColumn(F, Q, now + 1, Q);
116    t.put(p);
117    p = new Put(R);
118    p.addColumn(F, Q, now + 2, Q);
119    t.put(p);
120
121    Get g = new Get(R);
122    g.readVersions(10);
123    Result r = t.get(g);
124    assertEquals(3, r.size());
125
126    TEST_UTIL.flush(tableName);
127    TEST_UTIL.compact(tableName, true);
128
129    // still visible after a flush/compaction
130    r = t.get(g);
131    assertEquals(3, r.size());
132
133    // set the version override to 2
134    p = new Put(R);
135    p.setAttribute("versions", new byte[] {});
136    p.addColumn(F, tableName.getName(), Bytes.toBytes(2));
137    t.put(p);
138
139    // only 2 versions now
140    r = t.get(g);
141    assertEquals(2, r.size());
142
143    TEST_UTIL.flush(tableName);
144    TEST_UTIL.compact(tableName, true);
145
146    // still 2 versions after a flush/compaction
147    r = t.get(g);
148    assertEquals(2, r.size());
149
150    // insert a new version
151    p.addColumn(F, Q, now + 3, Q);
152    t.put(p);
153
154    // still 2 versions
155    r = t.get(g);
156    assertEquals(2, r.size());
157
158    t.close();
159  }

161  @Test
162  public void testTTL() throws Exception {
163    TableName tableName = TableName.valueOf("testTTL");
164    if (TEST_UTIL.getAdmin().tableExists(tableName)) {
165      TEST_UTIL.deleteTable(tableName);
166    }
167    Table t = TEST_UTIL.createTable(tableName, F, 10);
168    long now = EnvironmentEdgeManager.currentTime();
169    ManualEnvironmentEdge me = new ManualEnvironmentEdge();
170    me.setValue(now);
171    EnvironmentEdgeManagerTestHelper.injectEdge(me);
172    // 2s in the past
173    long ts = now - 2000;
174
175    Put p = new Put(R);
176    p.addColumn(F, Q, ts, Q);
177    t.put(p);
178    p = new Put(R);
179    p.addColumn(F, Q, ts + 1, Q);
180    t.put(p);
181
182    // Set the TTL override to 3s
183    p = new Put(R);
184    p.setAttribute("ttl", new byte[] {});
185    p.addColumn(F, tableName.getName(), Bytes.toBytes(3000L));
186    t.put(p);
187    // these two should still be there
188    Get g = new Get(R);
189    g.readAllVersions();
190    Result r = t.get(g);
191    // still there?
192    assertEquals(2, r.size());
193
194    TEST_UTIL.flush(tableName);
195    TEST_UTIL.compact(tableName, true);
196
197    g = new Get(R);
198    g.readAllVersions();
199    r = t.get(g);
200    // still there?
201    assertEquals(2, r.size());
202
203    // roll time forward 2s.
204    me.setValue(now + 2000);
205    // now verify that data eventually does expire
206    g = new Get(R);
207    g.readAllVersions();
208    r = t.get(g);
209    // should be gone now
210    assertEquals(0, r.size());
211    t.close();
212    EnvironmentEdgeManager.reset();
213  }

215  public static class ScanObserver implements RegionCoprocessor, RegionObserver {
216    private final ConcurrentMap<TableName, Long> ttls = new ConcurrentHashMap<>();
217    private final ConcurrentMap<TableName, Integer> versions = new ConcurrentHashMap<>();
218
219    @Override
220    public Optional<RegionObserver> getRegionObserver() {
221      return Optional.of(this);
222    }
223
224    // lame way to communicate with the coprocessor,
225    // since it is loaded by a different class loader
226    @Override
227    public void prePut(final ObserverContext<RegionCoprocessorEnvironment> c, final Put put,
228        final WALEdit edit, final Durability durability) throws IOException {
229      if (put.getAttribute("ttl") != null) {
230        Cell cell = put.getFamilyCellMap().values().stream().findFirst().get().get(0);
231        ttls.put(
232          TableName.valueOf(Bytes.toString(cell.getQualifierArray(), cell.getQualifierOffset(),
233            cell.getQualifierLength())),
234          Bytes.toLong(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength()));
235        c.bypass();
236      } else if (put.getAttribute("versions") != null) {
237        Cell cell = put.getFamilyCellMap().values().stream().findFirst().get().get(0);
238        versions.put(
239          TableName.valueOf(Bytes.toString(cell.getQualifierArray(), cell.getQualifierOffset(),
240            cell.getQualifierLength())),
241          Bytes.toInt(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength()));
242        c.bypass();
243      }
244    }
245
246    private InternalScanner wrap(Store store, InternalScanner scanner) {
247      Long ttl = this.ttls.get(store.getTableName());
248      Integer version = this.versions.get(store.getTableName());
249      return new DelegatingInternalScanner(scanner) {
250
251        private byte[] row;
252
253        private byte[] qualifier;
254
255        private int count;
256
257        private Predicate<Cell> checkTtl(long now, long ttl) {
258          return c -> now - c.getTimestamp() > ttl;
259        }
260
261        private Predicate<Cell> checkVersion(Cell firstCell, int version) {
262          if (version == 0) {
263            return c -> true;
264          } else {
265            if (row == null || !CellUtil.matchingRow(firstCell, row)) {
266              row = CellUtil.cloneRow(firstCell);
267              // reset qualifier as there is a row change
268              qualifier = null;
269            }
270            return c -> {
271              if (qualifier != null && CellUtil.matchingQualifier(c, qualifier)) {
272                if (count >= version) {
273                  return true;
274                }
275                count++;
276                return false;
277              } else { // qualifier switch
278                qualifier = CellUtil.cloneQualifier(c);
279                count = 1;
280                return false;
281              }
282            };
283          }
284        }
285
286        @Override
287        public boolean next(List<Cell> result, ScannerContext scannerContext) throws IOException {
288          boolean moreRows = scanner.next(result, scannerContext);
289          if (result.isEmpty()) {
290            return moreRows;
291          }
292          long now = EnvironmentEdgeManager.currentTime();
293          Predicate<Cell> predicate = null;
294          if (ttl != null) {
295            predicate = checkTtl(now, ttl);
296          }
297          if (version != null) {
298            Predicate<Cell> vp = checkVersion(result.get(0), version);
299            if (predicate != null) {
300              predicate = predicate.and(vp);
301            } else {
302              predicate = vp;
303            }
304          }
305          if (predicate != null) {
306            result.removeIf(predicate);
307          }
308          return moreRows;
309        }
310      };
311    }
312
313    @Override
314    public InternalScanner preFlush(ObserverContext<RegionCoprocessorEnvironment> c, Store store,
315        InternalScanner scanner, FlushLifeCycleTracker tracker) throws IOException {
316      return wrap(store, scanner);
317    }
318
319    @Override
320    public InternalScanner preCompact(ObserverContext<RegionCoprocessorEnvironment> c, Store store,
321        InternalScanner scanner, ScanType scanType, CompactionLifeCycleTracker tracker,
322        CompactionRequest request) throws IOException {
323      return wrap(store, scanner);
324    }
325
326    @Override
327    public void preGetOp(ObserverContext<RegionCoprocessorEnvironment> c, Get get,
328        List<Cell> result) throws IOException {
329      TableName tableName = c.getEnvironment().getRegion().getTableDescriptor().getTableName();
330      Long ttl = this.ttls.get(tableName);
331      if (ttl != null) {
332        get.setTimeRange(EnvironmentEdgeManager.currentTime() - ttl, get.getTimeRange().getMax());
333      }
334      Integer version = this.versions.get(tableName);
335      if (version != null) {
336        get.readVersions(version);
337      }
338    }
339
340    @Override
341    public RegionScanner preScannerOpen(ObserverContext<RegionCoprocessorEnvironment> c, Scan scan,
342        RegionScanner s) throws IOException {
343      Region region = c.getEnvironment().getRegion();
344      TableName tableName = region.getTableDescriptor().getTableName();
345      Long ttl = this.ttls.get(tableName);
346      if (ttl != null) {
347        scan.setTimeRange(EnvironmentEdgeManager.currentTime() - ttl, scan.getTimeRange().getMax());
348      }
349      Integer version = this.versions.get(tableName);
350      if (version != null) {
351        scan.readVersions(version);
352      }
353      return region.getScanner(scan);
354    }
355  }
