430  /* Overrides compaction to only output rows with keys that are even numbers */
431  public static class EvenOnlyCompactor implements RegionCoprocessor, RegionObserver {
432    long lastCompaction;
433    long lastFlush;
434
435    @Override
436    public Optional<RegionObserver> getRegionObserver() {
437      return Optional.of(this);
438    }
439
440    @Override
441    public InternalScanner preCompact(ObserverContext<RegionCoprocessorEnvironment> e, Store store,
442        InternalScanner scanner, ScanType scanType, CompactionLifeCycleTracker tracker,
443        CompactionRequest request) {
444      return new InternalScanner() {
445
446        @Override
447        public boolean next(List<Cell> results, ScannerContext scannerContext) throws IOException {
448          List<Cell> internalResults = new ArrayList<>();
449          boolean hasMore;
450          do {
451            hasMore = scanner.next(internalResults, scannerContext);
452            if (!internalResults.isEmpty()) {
453              long row = Bytes.toLong(CellUtil.cloneValue(internalResults.get(0)));
454              if (row % 2 == 0) {
455                // return this row
456                break;
457              }
458              // clear and continue
459              internalResults.clear();
460            }
461          } while (hasMore);
462
463          if (!internalResults.isEmpty()) {
464            results.addAll(internalResults);
465          }
466          return hasMore;
467        }
468
469        @Override
470        public void close() throws IOException {
471          scanner.close();
472        }
473      };
474    }
475
476    @Override
477    public void postCompact(ObserverContext<RegionCoprocessorEnvironment> e, Store store,
478        StoreFile resultFile, CompactionLifeCycleTracker tracker, CompactionRequest request) {
479      lastCompaction = EnvironmentEdgeManager.currentTime();
480    }
481
482    @Override
483    public void postFlush(ObserverContext<RegionCoprocessorEnvironment> e,
484        FlushLifeCycleTracker tracker) {
485      lastFlush = EnvironmentEdgeManager.currentTime();
486    }
487  }

490   * Tests overriding compaction handling via coprocessor hooks
491   * @throws Exception

493  @Test
494  public void testCompactionOverride() throws Exception {
495    final TableName compactTable = TableName.valueOf(name.getMethodName());
496    Admin admin = util.getAdmin();
497    if (admin.tableExists(compactTable)) {
498      admin.disableTable(compactTable);
499      admin.deleteTable(compactTable);
500    }
501
502    HTableDescriptor htd = new HTableDescriptor(compactTable);
503    htd.addFamily(new HColumnDescriptor(A));
504    htd.addCoprocessor(EvenOnlyCompactor.class.getName());
505    admin.createTable(htd);
506
507    Table table = util.getConnection().getTable(compactTable);
508    for (long i = 1; i <= 10; i++) {
509      byte[] iBytes = Bytes.toBytes(i);
510      Put put = new Put(iBytes);
511      put.setDurability(Durability.SKIP_WAL);
512      put.addColumn(A, A, iBytes);
513      table.put(put);
514    }
515
516    HRegion firstRegion = cluster.getRegions(compactTable).get(0);
517    Coprocessor cp = firstRegion.getCoprocessorHost().findCoprocessor(EvenOnlyCompactor.class);
518    assertNotNull("EvenOnlyCompactor coprocessor should be loaded", cp);
519    EvenOnlyCompactor compactor = (EvenOnlyCompactor) cp;
520
521    // force a compaction
522    long ts = System.currentTimeMillis();
523    admin.flush(compactTable);
524    // wait for flush
525    for (int i = 0; i < 10; i++) {
526      if (compactor.lastFlush >= ts) {
527        break;
528      }
529      Thread.sleep(1000);
530    }
531    assertTrue("Flush didn't complete", compactor.lastFlush >= ts);
532    LOG.debug("Flush complete");
533
534    ts = compactor.lastFlush;
535    admin.majorCompact(compactTable);
536    // wait for compaction
537    for (int i = 0; i < 30; i++) {
538      if (compactor.lastCompaction >= ts) {
539        break;
540      }
541      Thread.sleep(1000);
542    }
543    LOG.debug("Last compaction was at " + compactor.lastCompaction);
544    assertTrue("Compaction didn't complete", compactor.lastCompaction >= ts);
545
546    // only even rows should remain
547    ResultScanner scanner = table.getScanner(new Scan());
548    try {
549      for (long i = 2; i <= 10; i += 2) {
550        Result r = scanner.next();
551        assertNotNull(r);
552        assertFalse(r.isEmpty());
553        byte[] iBytes = Bytes.toBytes(i);
554        assertArrayEquals("Row should be " + i, r.getRow(), iBytes);
555        assertArrayEquals("Value should be " + i, r.getValue(A, A), iBytes);
556      }
557    } finally {
558      scanner.close();
559    }
560    table.close();
561  }

768  // check each region whether the coprocessor upcalls are called or not.
769  private void verifyMethodResult(Class<?> coprocessor, String methodName[], TableName tableName,
770      Object value[]) throws IOException {
771    try {
772      for (JVMClusterUtil.RegionServerThread t : cluster.getRegionServerThreads()) {
773        if (!t.isAlive() || t.getRegionServer().isAborted() || t.getRegionServer().isStopping()) {
774          continue;
775        }
776        for (RegionInfo r : ProtobufUtil
777            .getOnlineRegions(t.getRegionServer().getRSRpcServices())) {
778          if (!r.getTable().equals(tableName)) {
779            continue;
780          }
781          RegionCoprocessorHost cph =
782              t.getRegionServer().getOnlineRegion(r.getRegionName()).getCoprocessorHost();
783
784          Coprocessor cp = cph.findCoprocessor(coprocessor.getName());
785          assertNotNull(cp);
786          for (int i = 0; i < methodName.length; ++i) {
787            Method m = coprocessor.getMethod(methodName[i]);
788            Object o = m.invoke(cp);
789            assertTrue("Result of " + coprocessor.getName() + "." + methodName[i]
790                    + " is expected to be " + value[i].toString() + ", while we get "
791                    + o.toString(), o.equals(value[i]));
792          }
793        }
794      }
795    } catch (Exception e) {
796      throw new IOException(e.toString());
797    }
798  }

800  private static void createHFile(Configuration conf, FileSystem fs, Path path, byte[] family,
801      byte[] qualifier) throws IOException {
802    HFileContext context = new HFileContextBuilder().build();
803    HFile.Writer writer = HFile.getWriterFactory(conf, new CacheConfig(conf)).withPath(fs, path)
804        .withFileContext(context).create();
805    long now = System.currentTimeMillis();
806    try {
807      for (int i = 1; i <= 9; i++) {
808        KeyValue kv =
809            new KeyValue(Bytes.toBytes(i + ""), family, qualifier, now, Bytes.toBytes(i + ""));
810        writer.append(kv);
811      }
812    } finally {
813      writer.close();
814    }

817  private static class PreWALAppendWALActionsListener implements WALActionsListener {
818    boolean[] walKeysCorrect = {false, false, false, false};
819
820    @Override
821    public void postAppend(long entryLen, long elapsedTimeMillis,
822                           WALKey logKey, WALEdit logEdit) throws IOException {
823      for (int k = 0; k < 4; k++) {
824        if (!walKeysCorrect[k]) {
825          walKeysCorrect[k] = Arrays.equals(SimpleRegionObserver.WAL_EXTENDED_ATTRIBUTE_BYTES,
826              logKey.getExtendedAttribute(Integer.toString(k + 1)));
827        }
828      }
829    }
830
831    boolean[] getWalKeysCorrectArray() {
832      return walKeysCorrect;
833    }
