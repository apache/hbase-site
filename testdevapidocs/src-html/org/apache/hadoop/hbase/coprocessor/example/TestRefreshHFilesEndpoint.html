059@Category(MediumTests.class)
060public class TestRefreshHFilesEndpoint {
061
062  @ClassRule
063  public static final HBaseClassTestRule CLASS_RULE =
064      HBaseClassTestRule.forClass(TestRefreshHFilesEndpoint.class);
065
066  private static final Logger LOG = LoggerFactory.getLogger(TestRefreshHFilesEndpoint.class);
067  private static final HBaseTestingUtility HTU = new HBaseTestingUtility();
068  private static final int NUM_RS = 2;
069  private static final TableName TABLE_NAME = TableName.valueOf("testRefreshRegionHFilesEP");
070  private static final byte[] FAMILY = Bytes.toBytes("family");
071  private static final byte[] QUALIFIER = Bytes.toBytes("qualifier");
072  private static final byte[][] SPLIT_KEY = new byte[][] { Bytes.toBytes("30") };
073  private static final int NUM_ROWS = 5;
074  private static final String HFILE_NAME = "123abcdef";
075
076  private static Configuration CONF = HTU.getConfiguration();
077  private static MiniHBaseCluster cluster;
078  private static Table table;
079
080  public static void setUp(String regionImpl) {
081    try {
082      CONF.set(HConstants.REGION_IMPL, regionImpl);
083      CONF.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 2);
084
085      CONF.setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY, RefreshHFilesEndpoint.class.getName());
086      cluster = HTU.startMiniCluster(NUM_RS);
087
088      // Create table
089      table = HTU.createTable(TABLE_NAME, FAMILY, SPLIT_KEY);
090
091      // this will create 2 regions spread across slaves
092      HTU.loadNumericRows(table, FAMILY, 1, 20);
093      HTU.flush(TABLE_NAME);
094    } catch (Exception ex) {
095      LOG.error("Couldn't finish setup", ex);
096    }
097  }
098
099  @After
100  public void tearDown() throws Exception {
101    HTU.shutdownMiniCluster();
102  }
103
104  @Test
105  public void testRefreshRegionHFilesEndpoint() throws Exception {
106    setUp(HRegion.class.getName());
107    MasterFileSystem mfs = HTU.getMiniHBaseCluster().getMaster().getMasterFileSystem();
108    Path tableDir = FSUtils.getTableDir(mfs.getRootDir(), TABLE_NAME);
109    for (Region region : cluster.getRegions(TABLE_NAME)) {
110      Path regionDir = new Path(tableDir, region.getRegionInfo().getEncodedName());
111      Path familyDir = new Path(regionDir, Bytes.toString(FAMILY));
112      HFileTestUtil
113        .createHFile(HTU.getConfiguration(), HTU.getTestFileSystem(), new Path(familyDir, HFILE_NAME), FAMILY,
114                     QUALIFIER, Bytes.toBytes("50"), Bytes.toBytes("60"), NUM_ROWS);
115    }
116    assertEquals(2, HTU.getNumHFiles(TABLE_NAME, FAMILY));
117    callRefreshRegionHFilesEndPoint();
118    assertEquals(4, HTU.getNumHFiles(TABLE_NAME, FAMILY));
119  }
120
121  @Test(expected = IOException.class)
122  public void testRefreshRegionHFilesEndpointWithException() throws IOException {
123    setUp(HRegionForRefreshHFilesEP.class.getName());
124    callRefreshRegionHFilesEndPoint();
125  }
126
127  private void callRefreshRegionHFilesEndPoint() throws IOException {
128    try {
129      RefreshHFilesClient refreshHFilesClient = new RefreshHFilesClient(CONF);
130      refreshHFilesClient.refreshHFiles(TABLE_NAME);
131    } catch (RetriesExhaustedException rex) {
132      if (rex.getCause() instanceof IOException)
133        throw new IOException();
134    } catch (Throwable ex) {
135      LOG.error(ex.toString(), ex);
136      fail("Couldn't call the RefreshRegionHFilesEndpoint");
137    }
138  }
139
140  public static class HRegionForRefreshHFilesEP extends HRegion {
141    HStoreWithFaultyRefreshHFilesAPI store;
142
143    public HRegionForRefreshHFilesEP(final Path tableDir, final WAL wal, final FileSystem fs,
144                                     final Configuration confParam, final RegionInfo regionInfo,
145                                     final TableDescriptor htd, final RegionServerServices rsServices) {
146      super(tableDir, wal, fs, confParam, regionInfo, htd, rsServices);
147    }
148
149    @Override
150    public List<HStore> getStores() {
151      List<HStore> list = new ArrayList<>(stores.size());
152      /**
153       * This is used to trigger the custom definition (faulty)
154       * of refresh HFiles API.
155       */
156      try {
157        if (this.store == null) {
158          store = new HStoreWithFaultyRefreshHFilesAPI(this,
159              ColumnFamilyDescriptorBuilder.of(FAMILY), this.conf);
160        }
161        list.add(store);
162      } catch (IOException ioe) {
163        LOG.info("Couldn't instantiate custom store implementation", ioe);
164      }
165
166      list.addAll(stores.values());
167      return list;
168    }
169  }
170
171  public static class HStoreWithFaultyRefreshHFilesAPI extends HStore {
172    public HStoreWithFaultyRefreshHFilesAPI(final HRegion region, final ColumnFamilyDescriptor family,
173                                            final Configuration confParam) throws IOException {
174      super(region, family, confParam);
175    }
176
177    @Override
178    public void refreshStoreFiles() throws IOException {
179      throw new IOException();
180    }
181  }
