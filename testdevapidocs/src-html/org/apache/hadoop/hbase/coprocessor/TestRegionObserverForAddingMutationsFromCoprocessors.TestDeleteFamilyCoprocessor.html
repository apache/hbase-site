066  private static HBaseTestingUtility util;

095  private void createTable(String coprocessor) throws IOException {
096    HTableDescriptor htd = new HTableDescriptor(tableName)
097        .addFamily(new HColumnDescriptor(dummy))
098        .addFamily(new HColumnDescriptor(test))
099        .addCoprocessor(coprocessor);
100    util.getAdmin().createTable(htd);
101  }

103  /**
104   * Test various multiput operations.
105   * @throws Exception
106   */
107  @Test
108  public void testMulti() throws Exception {
109    createTable(TestMultiMutationCoprocessor.class.getName());
110
111    try (Table t = util.getConnection().getTable(tableName)) {
112      t.put(new Put(row1).addColumn(test, dummy, dummy));
113      assertRowCount(t, 3);
114    }
115  }

117  /**
118   * Tests that added mutations from coprocessors end up in the WAL.
119   */
120  @Test
121  public void testCPMutationsAreWrittenToWALEdit() throws Exception {
122    createTable(TestMultiMutationCoprocessor.class.getName());
123
124    try (Table t = util.getConnection().getTable(tableName)) {
125      t.put(new Put(row1).addColumn(test, dummy, dummy));
126      assertRowCount(t, 3);
127    }
128
129    assertNotNull(TestWALObserver.savedEdit);
130    assertEquals(4, TestWALObserver.savedEdit.getCells().size());
131  }

133  private static void assertRowCount(Table t, int expected) throws IOException {
134    try (ResultScanner scanner = t.getScanner(new Scan())) {
135      int i = 0;
136      for (Result r: scanner) {
137        LOG.info(r.toString());
138        i++;
139      }
140      assertEquals(expected, i);
141    }
142  }

146    createTable(TestDeleteCellCoprocessor.class.getName());
147
148    try (Table t = util.getConnection().getTable(tableName)) {
149      t.put(Lists.newArrayList(
150        new Put(row1).addColumn(test, dummy, dummy),
151        new Put(row2).addColumn(test, dummy, dummy),
152        new Put(row3).addColumn(test, dummy, dummy)
153          ));
154
155      assertRowCount(t, 3);
156
157      t.delete(new Delete(test).addColumn(test, dummy)); // delete non-existing row
158      assertRowCount(t, 1);

198  public static class TestMultiMutationCoprocessor implements RegionCoprocessor, RegionObserver {
199    @Override
200    public Optional<RegionObserver> getRegionObserver() {
201      return Optional.of(this);
202    }
203
204    @Override
205    public void preBatchMutate(ObserverContext<RegionCoprocessorEnvironment> c,
206        MiniBatchOperationInProgress<Mutation> miniBatchOp) throws IOException {
207      Mutation mut = miniBatchOp.getOperation(0);
208      List<Cell> cells = mut.getFamilyCellMap().get(test);
209      Put[] puts = new Put[] {
210          new Put(row1).addColumn(test, dummy, cells.get(0).getTimestamp(),
211            Bytes.toBytes("cpdummy")),
212          new Put(row2).addColumn(test, dummy, cells.get(0).getTimestamp(), dummy),
213          new Put(row3).addColumn(test, dummy, cells.get(0).getTimestamp(), dummy),
214      };
215      LOG.info("Putting:" + Arrays.toString(puts));
216      miniBatchOp.addOperationsFromCP(0, puts);
217    }
218  }

220  public static class TestDeleteCellCoprocessor implements RegionCoprocessor, RegionObserver {
221    @Override
222    public Optional<RegionObserver> getRegionObserver() {
223      return Optional.of(this);
224    }
225
226    @Override
227    public void preBatchMutate(ObserverContext<RegionCoprocessorEnvironment> c,
228        MiniBatchOperationInProgress<Mutation> miniBatchOp) throws IOException {
229      Mutation mut = miniBatchOp.getOperation(0);
230
231      if (mut instanceof Delete) {
232        List<Cell> cells = mut.getFamilyCellMap().get(test);
233        Delete[] deletes = new Delete[] {
234            // delete only 2 rows
235            new Delete(row1).addColumns(test, dummy, cells.get(0).getTimestamp()),
236            new Delete(row2).addColumns(test, dummy, cells.get(0).getTimestamp()),
237        };
238        LOG.info("Deleting:" + Arrays.toString(deletes));
239        miniBatchOp.addOperationsFromCP(0, deletes);
240      }
241    }
242  }

292  public static class TestWALObserver implements WALCoprocessor, WALObserver {
293    static WALEdit savedEdit = null;
294
295    @Override
296    public Optional<WALObserver> getWALObserver() {
297      return Optional.of(this);
298    }
299
300    @Override
301    public void postWALWrite(ObserverContext<? extends WALCoprocessorEnvironment> ctx,
302                             RegionInfo info, WALKey logKey, WALEdit logEdit) throws IOException {
303      if (info.getTable().equals(TableName.valueOf("testCPMutationsAreWrittenToWALEdit"))) {
304        savedEdit = logEdit;
305      }
306    }
