186  HRegion initHRegion(byte[] tableName, String callingMethod, Configuration conf,
187      byte[]... families) throws IOException {
188    TableDescriptorBuilder builder =
189      TableDescriptorBuilder.newBuilder(TableName.valueOf(tableName));
190    for (byte[] family : families) {
191      builder.setColumnFamily(
192        ColumnFamilyDescriptorBuilder.of(family));
193    }
194    TableDescriptor tableDescriptor = builder.build();
195    ChunkCreator.initialize(MemStoreLAB.CHUNK_SIZE_DEFAULT, false, 0, 0,
196      0, null, MemStoreLAB.INDEX_CHUNK_SIZE_PERCENTAGE_DEFAULT);
197    RegionInfo info = RegionInfoBuilder.newBuilder(tableDescriptor.getTableName()).build();
198    Path path = new Path(DIR + callingMethod);
199    WAL wal = HBaseTestingUtil.createWal(conf, path, info);
200    HRegion r = HRegion.createHRegion(info, path, conf, tableDescriptor, wal);
201    // this following piece is a hack. currently a coprocessorHost
202    // is secretly loaded at OpenRegionHandler. we don't really
203    // start a region server here, so just manually create cphost
204    // and set it to region.
205    RegionCoprocessorHost host = new RegionCoprocessorHost(r, null, conf);
206    r.setCoprocessorHost(host);
207    return r;
208  }

210  @Test
211  public void testRegionObserverScanTimeStacking() throws Exception {
212    byte[] ROW = Bytes.toBytes("testRow");
213    byte[] TABLE = Bytes.toBytes(getClass().getName());
214    byte[] A = Bytes.toBytes("A");
215    byte[][] FAMILIES = new byte[][] { A };
216
217    // Use new HTU to not overlap with the DFS cluster started in #CompactionStacking
218    Configuration conf = new HBaseTestingUtil().getConfiguration();
219    HRegion region = initHRegion(TABLE, getClass().getName(), conf, FAMILIES);
220    RegionCoprocessorHost h = region.getCoprocessorHost();
221    h.load(NoDataFromScan.class, Coprocessor.PRIORITY_HIGHEST, conf);
222    h.load(EmptyRegionObsever.class, Coprocessor.PRIORITY_USER, conf);
223
224    Put put = new Put(ROW);
225    put.addColumn(A, A, A);
226    region.put(put);
227
228    Get get = new Get(ROW);
229    Result r = region.get(get);
230    assertNull(
231      "Got an unexpected number of rows - "
232        + "no data should be returned with the NoDataFromScan coprocessor. Found: " + r,
233      r.listCells());
234    HBaseTestingUtil.closeRegionAndWAL(region);
235  }

265  /*
266   * Custom HRegion which uses CountDownLatch to signal the completion of compaction
267   */
268  public static class CompactionCompletionNotifyingRegion extends HRegion {
269    private static volatile CountDownLatch compactionStateChangeLatch = null;
270
271    @SuppressWarnings("deprecation")
272    public CompactionCompletionNotifyingRegion(Path tableDir, WAL log,
273        FileSystem fs, Configuration confParam, RegionInfo info,
274        TableDescriptor htd, RegionServerServices rsServices) {
275      super(tableDir, log, fs, confParam, info, htd, rsServices);
276    }
277
278    public CountDownLatch getCompactionStateChangeLatch() {
279      if (compactionStateChangeLatch == null) {
280        compactionStateChangeLatch = new CountDownLatch(1);
281      }
282      return compactionStateChangeLatch;
283    }
284
285    @Override
286    public boolean compact(CompactionContext compaction, HStore store,
287      ThroughputController throughputController) throws IOException {
288      boolean ret = super.compact(compaction, store, throughputController);
289      if (ret) {
290        compactionStateChangeLatch.countDown();
291      }
292      return ret;
293    }
294
295    @Override
296    public boolean compact(CompactionContext compaction, HStore store,
297        ThroughputController throughputController, User user) throws IOException {
298      boolean ret = super.compact(compaction, store, throughputController, user);
299      if (ret) compactionStateChangeLatch.countDown();
300      return ret;
301    }
302  }

304  /**
305   * Unfortunately, the easiest way to test this is to spin up a mini-cluster since we want to do
306   * the usual compaction mechanism on the region, rather than going through the backdoor to the
307   * region
308   */
309  @Test
310  public void testRegionObserverCompactionTimeStacking() throws Exception {
311    // setup a mini cluster so we can do a real compaction on a region
312    Configuration conf = UTIL.getConfiguration();
313    conf.setClass(HConstants.REGION_IMPL, CompactionCompletionNotifyingRegion.class, HRegion.class);
314    conf.setInt("hbase.hstore.compaction.min", 2);
315    UTIL.startMiniCluster();
316    byte[] ROW = Bytes.toBytes("testRow");
317    byte[] A = Bytes.toBytes("A");
318    TableDescriptor tableDescriptor =
319      TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
320        .setColumnFamily(ColumnFamilyDescriptorBuilder.of(A))
321        .setCoprocessor(CoprocessorDescriptorBuilder
322          .newBuilder(EmptyRegionObsever.class.getName()).setJarPath(null)
323          .setPriority(Coprocessor.PRIORITY_USER).setProperties(Collections.emptyMap()).build())
324        .setCoprocessor(CoprocessorDescriptorBuilder
325          .newBuilder(NoDataFromCompaction.class.getName()).setJarPath(null)
326          .setPriority(Coprocessor.PRIORITY_HIGHEST).setProperties(Collections.emptyMap()).build())
327        .build();
328
329    Admin admin = UTIL.getAdmin();
330    admin.createTable(tableDescriptor);
331
332    Table table = UTIL.getConnection().getTable(tableDescriptor.getTableName());
333
334    // put a row and flush it to disk
335    Put put = new Put(ROW);
336    put.addColumn(A, A, A);
337    table.put(put);
338
339    HRegionServer rs = UTIL.getRSForFirstRegionInTable(tableDescriptor.getTableName());
340    List<HRegion> regions = rs.getRegions(tableDescriptor.getTableName());
341    assertEquals("More than 1 region serving test table with 1 row", 1, regions.size());
342    Region region = regions.get(0);
343    admin.flushRegion(region.getRegionInfo().getRegionName());
344    CountDownLatch latch = ((CompactionCompletionNotifyingRegion)region)
345        .getCompactionStateChangeLatch();
346
347    // put another row and flush that too
348    put = new Put(Bytes.toBytes("anotherrow"));
349    put.addColumn(A, A, A);
350    table.put(put);
351    admin.flushRegion(region.getRegionInfo().getRegionName());
352
353    // run a compaction, which normally would should get rid of the data
354    // wait for the compaction checker to complete
355    latch.await();
356    // check both rows to ensure that they aren't there
357    Get get = new Get(ROW);
358    Result r = table.get(get);
359    assertNull(
360      "Got an unexpected number of rows - "
361        + "no data should be returned with the NoDataFromScan coprocessor. Found: " + r,
362      r.listCells());
363
364    get = new Get(Bytes.toBytes("anotherrow"));
365    r = table.get(get);
366    assertNull(
367      "Got an unexpected number of rows - "
368        + "no data should be returned with the NoDataFromScan coprocessor Found: " + r,
369      r.listCells());
370
371    table.close();
372    UTIL.shutdownMiniCluster();
373  }
