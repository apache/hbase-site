184  private void verifyWritesSeen(final WAL log, final SampleRegionWALCoprocessor cp,
185      final boolean seesLegacy) throws Exception {
186    RegionInfo hri = createBasicHRegionInfo(Bytes.toString(TEST_TABLE));
187    TableDescriptor htd = createBasic3FamilyHTD(Bytes
188        .toString(TEST_TABLE));
189    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
190    for (byte[] fam : htd.getColumnFamilyNames()) {
191      scopes.put(fam, 0);
192    }
193    Path basedir = new Path(this.hbaseRootDir, Bytes.toString(TEST_TABLE));
194    deleteDir(basedir);
195    fs.mkdirs(new Path(basedir, hri.getEncodedName()));
196
197    // TEST_FAMILY[0] shall be removed from WALEdit.
198    // TEST_FAMILY[1] value shall be changed.
199    // TEST_FAMILY[2] shall be added to WALEdit, although it's not in the put.
200    cp.setTestValues(TEST_TABLE, TEST_ROW, TEST_FAMILY[0], TEST_QUALIFIER[0],
201        TEST_FAMILY[1], TEST_QUALIFIER[1], TEST_FAMILY[2], TEST_QUALIFIER[2]);
202
203    assertFalse(cp.isPreWALWriteCalled());
204    assertFalse(cp.isPostWALWriteCalled());
205
206    // TEST_FAMILY[2] is not in the put, however it shall be added by the tested
207    // coprocessor.
208    // Use a Put to create familyMap.
209    Put p = creatPutWith2Families(TEST_ROW);
210
211    Map<byte[], List<Cell>> familyMap = p.getFamilyCellMap();
212    WALEdit edit = new WALEdit();
213    edit.add(familyMap);
214
215    boolean foundFamily0 = false;
216    boolean foundFamily2 = false;
217    boolean modifiedFamily1 = false;
218
219    List<Cell> cells = edit.getCells();
220
221    for (Cell cell : cells) {
222      if (Arrays.equals(CellUtil.cloneFamily(cell), TEST_FAMILY[0])) {
223        foundFamily0 = true;
224      }
225      if (Arrays.equals(CellUtil.cloneFamily(cell), TEST_FAMILY[2])) {
226        foundFamily2 = true;
227      }
228      if (Arrays.equals(CellUtil.cloneFamily(cell), TEST_FAMILY[1])) {
229        if (!Arrays.equals(CellUtil.cloneValue(cell), TEST_VALUE[1])) {
230          modifiedFamily1 = true;
231        }
232      }
233    }
234    assertTrue(foundFamily0);
235    assertFalse(foundFamily2);
236    assertFalse(modifiedFamily1);
237
238    // it's where WAL write cp should occur.
239    long now = EnvironmentEdgeManager.currentTime();
240    // we use HLogKey here instead of WALKeyImpl directly to support legacy coprocessors.
241    long txid = log.appendData(hri, new WALKeyImpl(hri.getEncodedNameAsBytes(), hri.getTable(), now,
242      new MultiVersionConcurrencyControl(), scopes), edit);
243    log.sync(txid);
244
245    // the edit shall have been change now by the coprocessor.
246    foundFamily0 = false;
247    foundFamily2 = false;
248    modifiedFamily1 = false;
249    for (Cell cell : cells) {
250      if (Arrays.equals(CellUtil.cloneFamily(cell), TEST_FAMILY[0])) {
251        foundFamily0 = true;
252      }
253      if (Arrays.equals(CellUtil.cloneFamily(cell), TEST_FAMILY[2])) {
254        foundFamily2 = true;
255      }
256      if (Arrays.equals(CellUtil.cloneFamily(cell), TEST_FAMILY[1])) {
257        if (!Arrays.equals(CellUtil.cloneValue(cell), TEST_VALUE[1])) {
258          modifiedFamily1 = true;
259        }
260      }
261    }
262    assertFalse(foundFamily0);
263    assertTrue(foundFamily2);
264    assertTrue(modifiedFamily1);
265
266    assertTrue(cp.isPreWALWriteCalled());
267    assertTrue(cp.isPostWALWriteCalled());
268  }

270  /**
271   * Coprocessors shouldn't get notice of empty waledits.
272   */
273  @Test
274  public void testEmptyWALEditAreNotSeen() throws Exception {
275    RegionInfo hri = createBasicHRegionInfo(Bytes.toString(TEST_TABLE));
276    TableDescriptor htd = createBasic3FamilyHTD(Bytes.toString(TEST_TABLE));
277    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
278    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
279    for(byte[] fam : htd.getColumnFamilyNames()) {
280      scopes.put(fam, 0);
281    }
282    WAL log = wals.getWAL(null);
283    try {
284      SampleRegionWALCoprocessor cp = getCoprocessor(log, SampleRegionWALCoprocessor.class);
285
286      cp.setTestValues(TEST_TABLE, null, null, null, null, null, null, null);
287
288      assertFalse(cp.isPreWALWriteCalled());
289      assertFalse(cp.isPostWALWriteCalled());
290
291      final long now = EnvironmentEdgeManager.currentTime();
292      long txid = log.appendData(hri,
293        new WALKeyImpl(hri.getEncodedNameAsBytes(), hri.getTable(), now, mvcc, scopes),
294        new WALEdit());
295      log.sync(txid);
296
297      assertFalse("Empty WALEdit should skip coprocessor evaluation.", cp.isPreWALWriteCalled());
298      assertFalse("Empty WALEdit should skip coprocessor evaluation.", cp.isPostWALWriteCalled());
299    } finally {
300      log.close();
301    }
302  }

304  /**
305   * Test WAL replay behavior with WALObserver.
306   */
307  @Test
308  public void testWALCoprocessorReplay() throws Exception {
309    // WAL replay is handled at HRegion::replayRecoveredEdits(), which is
310    // ultimately called by HRegion::initialize()
311    TableName tableName = TableName.valueOf(currentTest.getMethodName());
312    TableDescriptor htd = getBasic3FamilyHTableDescriptor(tableName);
313    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
314    // final HRegionInfo hri =
315    // createBasic3FamilyHRegionInfo(Bytes.toString(tableName));
316    // final HRegionInfo hri1 =
317    // createBasic3FamilyHRegionInfo(Bytes.toString(tableName));
318    RegionInfo hri = RegionInfoBuilder.newBuilder(tableName).build();
319
320    final Path basedir = CommonFSUtils.getTableDir(this.hbaseRootDir, tableName);
321    deleteDir(basedir);
322    fs.mkdirs(new Path(basedir, hri.getEncodedName()));
323
324    final Configuration newConf = HBaseConfiguration.create(this.conf);
325
326    // WAL wal = new WAL(this.fs, this.dir, this.oldLogDir, this.conf);
327    WAL wal = wals.getWAL(null);
328    // Put p = creatPutWith2Families(TEST_ROW);
329    WALEdit edit = new WALEdit();
330    long now = EnvironmentEdgeManager.currentTime();
331    final int countPerFamily = 1000;
332    NavigableMap<byte[], Integer> scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
333    for (byte[] fam : htd.getColumnFamilyNames()) {
334      scopes.put(fam, 0);
335    }
336    for (byte[] fam : htd.getColumnFamilyNames()) {
337      addWALEdits(tableName, hri, TEST_ROW, fam, countPerFamily,
338        EnvironmentEdgeManager.getDelegate(), wal, scopes, mvcc);
339    }
340    wal.appendData(hri, new WALKeyImpl(hri.getEncodedNameAsBytes(), tableName, now, mvcc, scopes),
341      edit);
342    // sync to fs.
343    wal.sync();
344
345    User user = HBaseTestingUtil.getDifferentUser(newConf,
346        ".replay.wal.secondtime");
347    user.runAs(new PrivilegedExceptionAction<Void>() {
348      @Override
349      public Void run() throws Exception {
350        Path p = runWALSplit(newConf);
351        LOG.info("WALSplit path == " + p);
352        // Make a new wal for new region open.
353        final WALFactory wals2 = new WALFactory(conf,
354            ServerName.valueOf(currentTest.getMethodName() + "2", 16010,
355              EnvironmentEdgeManager.currentTime())
356                .toString());
357        WAL wal2 = wals2.getWAL(null);
358        HRegion region = HRegion.openHRegion(newConf, FileSystem.get(newConf), hbaseRootDir,
359            hri, htd, wal2, TEST_UTIL.getHBaseCluster().getRegionServer(0), null);
360
361        SampleRegionWALCoprocessor cp2 =
362          region.getCoprocessorHost().findCoprocessor(SampleRegionWALCoprocessor.class);
363        // TODO: asserting here is problematic.
364        assertNotNull(cp2);
365        assertTrue(cp2.isPreWALRestoreCalled());
366        assertTrue(cp2.isPostWALRestoreCalled());
367        region.close();
368        wals2.close();
369        return null;
370      }
371    });
372  }

426    Put p = new Put(row);

433  private Path runWALSplit(final Configuration c) throws IOException {
434    List<Path> splits = WALSplitter.split(
435      hbaseRootDir, logDir, oldLogDir, FileSystem.get(c), c, wals);
436    // Split should generate only 1 file since there's only 1 region
437    assertEquals(1, splits.size());
438    // Make sure the file exists
439    assertTrue(fs.exists(splits.get(0)));
440    LOG.info("Split file=" + splits.get(0));
441    return splits.get(0);
442  }

444  private void addWALEdits(final TableName tableName, final RegionInfo hri, final byte[] rowName,
445      final byte[] family, final int count, EnvironmentEdge ee, final WAL wal,
446      final NavigableMap<byte[], Integer> scopes, final MultiVersionConcurrencyControl mvcc)
447      throws IOException {
448    String familyStr = Bytes.toString(family);
449    long txid = -1;
450    for (int j = 0; j < count; j++) {
451      byte[] qualifierBytes = Bytes.toBytes(Integer.toString(j));
452      byte[] columnBytes = Bytes.toBytes(familyStr + ":" + Integer.toString(j));
453      WALEdit edit = new WALEdit();
454      edit.add(new KeyValue(rowName, family, qualifierBytes, ee.currentTime(), columnBytes));
455      // uses WALKeyImpl instead of HLogKey on purpose. will only work for tests where we don't care
456      // about legacy coprocessors
457      txid = wal.appendData(hri,
458        new WALKeyImpl(hri.getEncodedNameAsBytes(), tableName, ee.currentTime(), mvcc), edit);
459    }
460    if (-1 != txid) {
461      wal.sync(txid);
462    }
463  }

465  private TableDescriptor getBasic3FamilyHTableDescriptor(TableName tableName) {
466    TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(tableName);
467    Arrays.stream(TEST_FAMILY).map(ColumnFamilyDescriptorBuilder::of)
468        .forEachOrdered(builder::setColumnFamily);
469    return builder.build();
470  }

472  private TableDescriptor createBasic3FamilyHTD(String tableName) {
473    return TableDescriptorBuilder.newBuilder(TableName.valueOf(tableName))
474        .setColumnFamily(ColumnFamilyDescriptorBuilder.of("a"))
475        .setColumnFamily(ColumnFamilyDescriptorBuilder.of("b"))
476        .setColumnFamily(ColumnFamilyDescriptorBuilder.of("c")).build();
477  }
