062@Category({CoprocessorTests.class, MediumTests.class})
063public class TestBatchCoprocessorEndpoint {
064  private static final Log LOG = LogFactory.getLog(TestBatchCoprocessorEndpoint.class);
065
066  private static final TableName TEST_TABLE =
067      TableName.valueOf("TestTable");
068  private static final byte[] TEST_FAMILY = Bytes.toBytes("TestFamily");
069  private static final byte[] TEST_QUALIFIER = Bytes.toBytes("TestQualifier");
070  private static byte[] ROW = Bytes.toBytes("testRow");
071
072  private static final int ROWSIZE = 20;
073  private static final int rowSeperator1 = 5;
074  private static final int rowSeperator2 = 12;
075  private static byte[][] ROWS = makeN(ROW, ROWSIZE);
076
077  private static HBaseTestingUtility util = new HBaseTestingUtility();
078
079  @BeforeClass
080  public static void setupBeforeClass() throws Exception {
081    // set configure to indicate which cp should be loaded
082    Configuration conf = util.getConfiguration();
083    conf.setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
084        org.apache.hadoop.hbase.coprocessor.ColumnAggregationEndpoint.class.getName(),
085        ProtobufCoprocessorService.class.getName(),
086        ColumnAggregationEndpointWithErrors.class.getName(),
087        ColumnAggregationEndpointNullResponse.class.getName());
088    conf.setStrings(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY,
089        ProtobufCoprocessorService.class.getName());
090    util.startMiniCluster(2);
091    Admin admin = util.getHBaseAdmin();
092    HTableDescriptor desc = new HTableDescriptor(TEST_TABLE);
093    desc.addFamily(new HColumnDescriptor(TEST_FAMILY));
094    admin.createTable(desc, new byte[][]{ROWS[rowSeperator1], ROWS[rowSeperator2]});
095    util.waitUntilAllRegionsAssigned(TEST_TABLE);
096    admin.close();
097
098    Table table = util.getConnection().getTable(TEST_TABLE);
099    for (int i = 0; i < ROWSIZE; i++) {
100      Put put = new Put(ROWS[i]);
101      put.addColumn(TEST_FAMILY, TEST_QUALIFIER, Bytes.toBytes(i));
102      table.put(put);
103    }
104    table.close();
105  }
106
107  @AfterClass
108  public static void tearDownAfterClass() throws Exception {
109    util.shutdownMiniCluster();
110  }

112  @Test
113  public void testAggregationNullResponse() throws Throwable {
114    Table table = util.getConnection().getTable(TEST_TABLE);
115    ColumnAggregationNullResponseSumRequest.Builder builder =
116        ColumnAggregationNullResponseSumRequest
117        .newBuilder();
118    builder.setFamily(ByteStringer.wrap(TEST_FAMILY));
119    if (TEST_QUALIFIER != null && TEST_QUALIFIER.length > 0) {
120      builder.setQualifier(ByteStringer.wrap(TEST_QUALIFIER));
121    }
122    Map<byte[], ColumnAggregationNullResponseSumResponse> results =
123        table.batchCoprocessorService(
124            ColumnAggregationServiceNullResponse.getDescriptor().findMethodByName("sum"),
125            builder.build(), ROWS[0], ROWS[ROWS.length - 1],
126            ColumnAggregationNullResponseSumResponse.getDefaultInstance());
127
128    int sumResult = 0;
129    int expectedResult = 0;
130    for (Map.Entry<byte[], ColumnAggregationNullResponseSumResponse> e :
131        results.entrySet()) {
132      LOG.info("Got value " + e.getValue().getSum() + " for region "
133          + Bytes.toStringBinary(e.getKey()));
134      sumResult += e.getValue().getSum();
135    }
136    for (int i = 0; i < rowSeperator2; i++) {
137      expectedResult += i;
138    }
139    assertEquals("Invalid result", expectedResult, sumResult);
140    table.close();
141  }

143  private static byte[][] makeN(byte[] base, int n) {
144    byte[][] ret = new byte[n][];
145    for (int i = 0; i < n; i++) {
146      ret[i] = Bytes.add(base, Bytes.toBytes(String.format("%02d", i)));
147    }
148    return ret;
149  }
150
151  private Map<byte[], SumResponse> sum(final Table table, final byte[] family,
152      final byte[] qualifier, final byte[] start, final byte[] end) throws ServiceException,
153      Throwable {
154    ColumnAggregationProtos.SumRequest.Builder builder = ColumnAggregationProtos.SumRequest
155        .newBuilder();
156    builder.setFamily(ByteStringer.wrap(family));
157    if (qualifier != null && qualifier.length > 0) {
158      builder.setQualifier(ByteStringer.wrap(qualifier));
159    }
160    return table.batchCoprocessorService(
161        ColumnAggregationProtos.ColumnAggregationService.getDescriptor().findMethodByName("sum"),
162        builder.build(), start, end, ColumnAggregationProtos.SumResponse.getDefaultInstance());
163  }

172    for (Map.Entry<byte[], SumResponse> e : results.entrySet()) {

174          + Bytes.toStringBinary(e.getKey()));
175      sumResult += e.getValue().getSum();

177    for (int i = 0; i < ROWSIZE; i++) {
178      expectedResult += i;
179    }
180    assertEquals("Invalid result", expectedResult, sumResult);
181
182    results.clear();
183
184    // scan: for region 2 and region 3
185    results = sum(table, TEST_FAMILY, TEST_QUALIFIER, ROWS[rowSeperator1],
186        ROWS[ROWS.length - 1]);
187    sumResult = 0;
188    expectedResult = 0;
189    for (Map.Entry<byte[], SumResponse> e : results.entrySet()) {
190      LOG.info("Got value " + e.getValue().getSum() + " for region "
191          + Bytes.toStringBinary(e.getKey()));
192      sumResult += e.getValue().getSum();
193    }
194    for (int i = rowSeperator1; i < ROWSIZE; i++) {
195      expectedResult += i;
196    }
197    assertEquals("Invalid result", expectedResult, sumResult);
198    table.close();
199  }

234  @Test
235  public void testAggregationWithErrors() throws Throwable {
236    Table table = util.getConnection().getTable(TEST_TABLE);
237    final Map<byte[], ColumnAggregationWithErrorsSumResponse> results =
238        Collections.synchronizedMap(
239            new TreeMap<byte[], ColumnAggregationWithErrorsSumResponse>(
240                Bytes.BYTES_COMPARATOR
241            ));
242    ColumnAggregationWithErrorsSumRequest.Builder builder =
243        ColumnAggregationWithErrorsSumRequest
244        .newBuilder();
245    builder.setFamily(ByteStringer.wrap(TEST_FAMILY));
246    if (TEST_QUALIFIER != null && TEST_QUALIFIER.length > 0) {
247      builder.setQualifier(ByteStringer.wrap(TEST_QUALIFIER));
248    }
249
250    boolean hasError = false;
251    try {
252      table.batchCoprocessorService(
253          ColumnAggregationWithErrorsProtos.ColumnAggregationServiceWithErrors.getDescriptor()
254              .findMethodByName("sum"),
255          builder.build(), ROWS[0], ROWS[ROWS.length - 1],
256          ColumnAggregationWithErrorsSumResponse.getDefaultInstance(),
257          new Batch.Callback<ColumnAggregationWithErrorsSumResponse>() {
258
259            @Override
260            public void update(byte[] region, byte[] row,
261                ColumnAggregationWithErrorsSumResponse result) {
262              results.put(region, result);
263            }
264          });
265    } catch (Throwable t) {
266      LOG.info("Exceptions in coprocessor service", t);
267      hasError = true;
268    }
269
270    int sumResult = 0;
271    int expectedResult = 0;
272    for (Map.Entry<byte[], ColumnAggregationWithErrorsSumResponse> e : results.entrySet()) {
273      LOG.info("Got value " + e.getValue().getSum() + " for region "
274          + Bytes.toStringBinary(e.getKey()));
275      sumResult += e.getValue().getSum();
276    }
277    for (int i = 0; i < rowSeperator2; i++) {
278      expectedResult += i;
279    }
280    assertEquals("Invalid result", expectedResult, sumResult);
281    assertTrue(hasError);
282    table.close();
283  }
