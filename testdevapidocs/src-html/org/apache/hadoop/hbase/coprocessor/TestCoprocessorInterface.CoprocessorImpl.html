287  @Test
288  public void testSharedData() throws IOException {
289    TableName tableName = TableName.valueOf(name.getMethodName());
290    byte [][] families = { fam1, fam2, fam3 };
291
292    Configuration hc = initConfig();
293    HRegion region = initHRegion(tableName, name.getMethodName(), hc, new Class<?>[]{}, families);
294
295    for (int i = 0; i < 3; i++) {
296      HBaseTestCase.addContent(region, fam3);
297      region.flush(true);
298    }
299
300    region.compact(false);
301
302    region = reopenRegion(region, CoprocessorImpl.class, CoprocessorII.class);
303
304    Coprocessor c = region.getCoprocessorHost().findCoprocessor(CoprocessorImpl.class);
305    Coprocessor c2 = region.getCoprocessorHost().findCoprocessor(CoprocessorII.class);
306    Object o = ((CoprocessorImpl)c).getSharedData().get("test1");
307    Object o2 = ((CoprocessorII)c2).getSharedData().get("test2");
308    assertNotNull(o);
309    assertNotNull(o2);
310    // to coprocessors get different sharedDatas
311    assertFalse(((CoprocessorImpl)c).getSharedData() == ((CoprocessorII)c2).getSharedData());
312    c = region.getCoprocessorHost().findCoprocessor(CoprocessorImpl.class);
313    c2 = region.getCoprocessorHost().findCoprocessor(CoprocessorII.class);
314    // make sure that all coprocessor of a class have identical sharedDatas
315    assertTrue(((CoprocessorImpl)c).getSharedData().get("test1") == o);
316    assertTrue(((CoprocessorII)c2).getSharedData().get("test2") == o2);
317
318    // now have all Environments fail
319    try {
320      byte [] r = region.getRegionInfo().getStartKey();
321      if (r == null || r.length <= 0) {
322        // Its the start row.  Can't ask for null.  Ask for minimal key instead.
323        r = new byte [] {0};
324      }
325      Get g = new Get(r);
326      region.get(g);
327      fail();
328    } catch (org.apache.hadoop.hbase.DoNotRetryIOException xc) {
329    }
330    assertNull(region.getCoprocessorHost().findCoprocessor(CoprocessorII.class));
331    c = region.getCoprocessorHost().findCoprocessor(CoprocessorImpl.class);
332    assertTrue(((CoprocessorImpl)c).getSharedData().get("test1") == o);
333    c = c2 = null;
334    // perform a GC
335    System.gc();
336    // reopen the region
337    region = reopenRegion(region, CoprocessorImpl.class, CoprocessorII.class);
338    c = region.getCoprocessorHost().findCoprocessor(CoprocessorImpl.class);
339    // CPimpl is unaffected, still the same reference
340    assertTrue(((CoprocessorImpl)c).getSharedData().get("test1") == o);
341    c2 = region.getCoprocessorHost().findCoprocessor(CoprocessorII.class);
342    // new map and object created, hence the reference is different
343    // hence the old entry was indeed removed by the GC and new one has been created
344    Object o3 = ((CoprocessorII)c2).getSharedData().get("test2");
345    assertFalse(o3 == o2);
346    HBaseTestingUtility.closeRegionAndWAL(region);
347  }

349  @Test
350  public void testCoprocessorInterface() throws IOException {
351    TableName tableName = TableName.valueOf(name.getMethodName());
352    byte [][] families = { fam1, fam2, fam3 };
353
354    Configuration hc = initConfig();
355    HRegion region = initHRegion(tableName, name.getMethodName(), hc,
356      new Class<?>[]{CoprocessorImpl.class}, families);
357    for (int i = 0; i < 3; i++) {
358      HBaseTestCase.addContent(region, fam3);
359      region.flush(true);
360    }
361
362    region.compact(false);
363
364    // HBASE-4197
365    Scan s = new Scan();
366    RegionScanner scanner = region.getCoprocessorHost().postScannerOpen(s, region.getScanner(s));
367    assertTrue(scanner instanceof CustomScanner);
368    // this would throw an exception before HBASE-4197
369    scanner.next(new ArrayList<>());
370
371    HBaseTestingUtility.closeRegionAndWAL(region);
372    Coprocessor c = region.getCoprocessorHost().findCoprocessor(CoprocessorImpl.class);
373
374    assertTrue("Coprocessor not started", ((CoprocessorImpl)c).wasStarted());
375    assertTrue("Coprocessor not stopped", ((CoprocessorImpl)c).wasStopped());
376    assertTrue(((CoprocessorImpl)c).wasOpened());
377    assertTrue(((CoprocessorImpl)c).wasClosed());
378    assertTrue(((CoprocessorImpl)c).wasFlushed());
379    assertTrue(((CoprocessorImpl)c).wasCompacted());
380  }

382  HRegion reopenRegion(final HRegion closedRegion, Class<?> ... implClasses)
383      throws IOException {
384    //RegionInfo info = new RegionInfo(tableName, null, null, false);
385    HRegion r = HRegion.openHRegion(closedRegion, null);
386
387    // this following piece is a hack. currently a coprocessorHost
388    // is secretly loaded at OpenRegionHandler. we don't really
389    // start a region server here, so just manually create cphost
390    // and set it to region.
391    Configuration conf = TEST_UTIL.getConfiguration();
392    RegionCoprocessorHost host = new RegionCoprocessorHost(r,
393        Mockito.mock(RegionServerServices.class), conf);
394    r.setCoprocessorHost(host);
395
396    for (Class<?> implClass : implClasses) {
397      host.load((Class<? extends RegionCoprocessor>) implClass, Coprocessor.PRIORITY_USER, conf);
398    }
399    // we need to manually call pre- and postOpen here since the
400    // above load() is not the real case for CP loading. A CP is
401    // expected to be loaded by default from 1) configuration; or 2)
402    // HTableDescriptor. If it's loaded after HRegion initialized,
403    // the pre- and postOpen() won't be triggered automatically.
404    // Here we have to call pre and postOpen explicitly.
405    host.preOpen();
406    host.postOpen();
407    return r;
408  }

410  HRegion initHRegion (TableName tableName, String callingMethod,
411      Configuration conf, Class<?> [] implClasses, byte [][] families)
412      throws IOException {
413    HTableDescriptor htd = new HTableDescriptor(tableName);
414    for(byte [] family : families) {
415      htd.addFamily(new HColumnDescriptor(family));
416    }
417    ChunkCreator.initialize(MemStoreLABImpl.CHUNK_SIZE_DEFAULT, false, 0, 0, 0, null);
418    RegionInfo info = RegionInfoBuilder.newBuilder(tableName)
419        .setStartKey(null)
420        .setEndKey(null)
421        .setSplit(false)
422        .build();
423    Path path = new Path(DIR + callingMethod);
424    HRegion r = HBaseTestingUtility.createRegionAndWAL(info, path, conf, htd);
425
426    // this following piece is a hack.
427    RegionCoprocessorHost host =
428        new RegionCoprocessorHost(r, Mockito.mock(RegionServerServices.class), conf);
429    r.setCoprocessorHost(host);
430
431    for (Class<?> implClass : implClasses) {
432      host.load((Class<? extends RegionCoprocessor>) implClass, Coprocessor.PRIORITY_USER, conf);
433      Coprocessor c = host.findCoprocessor(implClass.getName());
434      assertNotNull(c);
435    }
436
437    // Here we have to call pre and postOpen explicitly.
438    host.preOpen();
439    host.postOpen();
440    return r;
441  }

443  private Configuration initConfig() {
444    // Always compact if there is more than one store file.
445    TEST_UTIL.getConfiguration().setInt("hbase.hstore.compactionThreshold", 2);
446    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD, 10 * 1000);
447    // Increase the amount of time between client retries
448    TEST_UTIL.getConfiguration().setLong("hbase.client.pause", 15 * 1000);
449    // This size should make it so we always split using the addContent
450    // below.  After adding all data, the first region is 1.3M
451    TEST_UTIL.getConfiguration().setLong(HConstants.HREGION_MAX_FILESIZE,
452        1024 * 128);
453    TEST_UTIL.getConfiguration().setBoolean("hbase.testing.nocluster",
454        true);
455    TEST_UTIL.getConfiguration().setBoolean(CoprocessorHost.ABORT_ON_ERROR_KEY, false);
456
457    return TEST_UTIL.getConfiguration();
458  }
