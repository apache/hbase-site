067  @Test
068  public void testDoubleLoadingAndPriorityValue() {
069    final Configuration conf = HBaseConfiguration.create();
070    final String key = "KEY";
071    final String coprocessor = "org.apache.hadoop.hbase.coprocessor.SimpleRegionObserver";
072
073    CoprocessorHost<RegionCoprocessor, CoprocessorEnvironment<RegionCoprocessor>> host;
074    host = new CoprocessorHostForTest<>(conf);
075    int overridePriority = Integer.MAX_VALUE - 1;
076
077    final String coprocessor_v3 =
078      SimpleRegionObserverV3.class.getName() + "|" + overridePriority;
079
080    // Try and load a coprocessor three times
081    conf.setStrings(key, coprocessor, coprocessor, coprocessor,
082        SimpleRegionObserverV2.class.getName(), coprocessor_v3);
083    host.loadSystemCoprocessors(conf, key);
084
085    // Three coprocessors(SimpleRegionObserver, SimpleRegionObserverV2,
086    // SimpleRegionObserverV3) loaded
087    Assert.assertEquals(3, host.coprocEnvironments.size());
088
089    // Check the priority value
090    CoprocessorEnvironment<?> simpleEnv = host.findCoprocessorEnvironment(
091        SimpleRegionObserver.class.getName());
092    CoprocessorEnvironment<?> simpleEnv_v2 = host.findCoprocessorEnvironment(
093        SimpleRegionObserverV2.class.getName());
094    CoprocessorEnvironment<?> simpleEnv_v3 = host.findCoprocessorEnvironment(
095      SimpleRegionObserverV3.class.getName());
096
097    assertNotNull(simpleEnv);
098    assertNotNull(simpleEnv_v2);
099    assertNotNull(simpleEnv_v3);
100    assertEquals(Coprocessor.PRIORITY_SYSTEM, simpleEnv.getPriority());
101    assertEquals(Coprocessor.PRIORITY_SYSTEM + 1, simpleEnv_v2.getPriority());
102    assertEquals(overridePriority, simpleEnv_v3.getPriority());
103  }

105  @Test
106  public void testLoadSystemCoprocessorWithPath() throws Exception {
107    Configuration conf = TEST_UTIL.getConfiguration();
108    final String key = "KEY";
109    final String testClassName = "TestSystemCoprocessor";
110    final String testClassNameWithPriorityAndPath = testClassName + "PriorityAndPath";
111
112    File jarFile = buildCoprocessorJar(testClassName);
113    File jarFileWithPriorityAndPath = buildCoprocessorJar(testClassNameWithPriorityAndPath);
114
115    try {
116      CoprocessorHost<RegionCoprocessor, CoprocessorEnvironment<RegionCoprocessor>> host;
117      host = new CoprocessorHostForTest<>(conf);
118
119      // make a string of coprocessor with only priority
120      int overridePriority = Integer.MAX_VALUE - 1;
121      final String coprocessorWithPriority =
122        SimpleRegionObserverV3.class.getName() + "|" + overridePriority;
123      // make a string of coprocessor with path but no priority
124      final String coprocessorWithPath =
125        String.format("%s|%s|%s", testClassName, "", jarFile.getAbsolutePath());
126      // make a string of coprocessor with priority and path
127      final String coprocessorWithPriorityAndPath = String
128        .format("%s|%s|%s", testClassNameWithPriorityAndPath, (overridePriority - 1),
129          jarFileWithPriorityAndPath.getAbsolutePath());
130
131      // Try and load a system coprocessors
132      conf.setStrings(key, SimpleRegionObserverV2.class.getName(), coprocessorWithPriority,
133        coprocessorWithPath, coprocessorWithPriorityAndPath);
134      host.loadSystemCoprocessors(conf, key);
135
136      // first loaded system coprocessor with default priority
137      CoprocessorEnvironment<?> simpleEnv =
138        host.findCoprocessorEnvironment(SimpleRegionObserverV2.class.getName());
139      assertNotNull(simpleEnv);
140      assertEquals(Coprocessor.PRIORITY_SYSTEM, simpleEnv.getPriority());
141
142      // external system coprocessor with default priority
143      CoprocessorEnvironment<?> coprocessorEnvironmentWithPath =
144        host.findCoprocessorEnvironment(testClassName);
145      assertNotNull(coprocessorEnvironmentWithPath);
146      assertEquals(Coprocessor.PRIORITY_SYSTEM + 1, coprocessorEnvironmentWithPath.getPriority());
147
148      // system coprocessor with configured priority
149      CoprocessorEnvironment<?> coprocessorEnvironmentWithPriority =
150        host.findCoprocessorEnvironment(SimpleRegionObserverV3.class.getName());
151      assertNotNull(coprocessorEnvironmentWithPriority);
152      assertEquals(overridePriority, coprocessorEnvironmentWithPriority.getPriority());
153
154      // external system coprocessor with override priority
155      CoprocessorEnvironment<?> coprocessorEnvironmentWithPriorityAndPath =
156        host.findCoprocessorEnvironment(testClassNameWithPriorityAndPath);
157      assertNotNull(coprocessorEnvironmentWithPriorityAndPath);
158      assertEquals(overridePriority - 1, coprocessorEnvironmentWithPriorityAndPath.getPriority());
159    } finally {
160      if (jarFile.exists()) {
161        jarFile.delete();
162      }
163      if (jarFileWithPriorityAndPath.exists()) {
164        jarFileWithPriorityAndPath.delete();
165      }
166    }
167  }

213  private static class CoprocessorHostForTest<E extends Coprocessor> extends
214      CoprocessorHost<E, CoprocessorEnvironment<E>> {
215    final Configuration cpHostConf;
216
217    public CoprocessorHostForTest(Configuration conf) {
218      super(new TestAbortable());
219      cpHostConf = conf;
220    }
221
222    @Override
223    public E checkAndGetInstance(Class<?> implClass)
224        throws InstantiationException, IllegalAccessException {
225      try {
226        return (E) implClass.getDeclaredConstructor().newInstance();
227      } catch (InvocationTargetException | NoSuchMethodException e) {
228        throw (InstantiationException) new InstantiationException().initCause(e);
229      }
230    }
231
232    @Override
233    public CoprocessorEnvironment<E> createEnvironment(final E instance, final int priority,
234        int sequence, Configuration conf) {
235      return new BaseEnvironment<>(instance, priority, 0, cpHostConf);
236    }
237  }

239  private File buildCoprocessorJar(String className) throws Exception {
240    String dataTestDir = TEST_UTIL.getDataTestDir().toString();
241    String code = String.format("import org.apache.hadoop.hbase.coprocessor.*; public class %s"
242      + " implements RegionCoprocessor {}", className);
243    return ClassLoaderTestHelper.buildJar(dataTestDir, className, code);
244  }
