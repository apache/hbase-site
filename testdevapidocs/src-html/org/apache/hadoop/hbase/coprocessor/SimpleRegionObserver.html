143    // this only makes sure that leases and locks are available to coprocessors

145    RegionCoprocessorEnvironment re = (RegionCoprocessorEnvironment)e;
146    Leases leases = re.getRegionServerServices().getLeases();
147    leases.createLease(re.getRegion().getRegionInfo().getRegionNameAsString(), 2000, null);

341  @Override
342  public void postGetOp(final ObserverContext<RegionCoprocessorEnvironment> c, final Get get,
343      final List<Cell> results) {
344    RegionCoprocessorEnvironment e = c.getEnvironment();
345    assertNotNull(e);
346    assertNotNull(e.getRegion());
347    assertNotNull(get);
348    assertNotNull(results);
349    if (e.getRegion().getTableDescriptor().getTableName().equals(
350        TestRegionObserverInterface.TEST_TABLE)) {
351      boolean foundA = false;
352      boolean foundB = false;
353      boolean foundC = false;
354      for (Cell kv: results) {
355        if (CellUtil.matchingFamily(kv, TestRegionObserverInterface.A)) {
356          foundA = true;
357        }
358        if (CellUtil.matchingFamily(kv, TestRegionObserverInterface.B)) {
359          foundB = true;
360        }
361        if (CellUtil.matchingFamily(kv, TestRegionObserverInterface.C)) {
362          foundC = true;
363        }
364      }
365      assertTrue(foundA);
366      assertTrue(foundB);
367      assertTrue(foundC);
368    }
369    ctPostGet.incrementAndGet();
370  }

610  @Override
611  public void preBulkLoadHFile(ObserverContext<RegionCoprocessorEnvironment> ctx,
612                               List<Pair<byte[], String>> familyPaths) throws IOException {
613    RegionCoprocessorEnvironment e = ctx.getEnvironment();
614    assertNotNull(e);
615    assertNotNull(e.getRegion());
616    if (e.getRegion().getTableDescriptor().getTableName().equals(
617        TestRegionObserverInterface.TEST_TABLE)) {
618      assertNotNull(familyPaths);
619      assertEquals(1,familyPaths.size());
620      assertArrayEquals(familyPaths.get(0).getFirst(), TestRegionObserverInterface.A);
