018package org.apache.hadoop.hbase.hbtop;
019
020import java.util.ArrayList;
021import java.util.List;
022import java.util.Objects;
023import java.util.Optional;
024import java.util.stream.Collectors;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.conf.Configured;
027import org.apache.hadoop.hbase.HBaseConfiguration;
028import org.apache.hadoop.hbase.HBaseInterfaceAudience;
029import org.apache.hadoop.hbase.hbtop.field.Field;
030import org.apache.hadoop.hbase.hbtop.field.FieldInfo;
031import org.apache.hadoop.hbase.hbtop.mode.Mode;
032import org.apache.hadoop.hbase.hbtop.screen.Screen;
033import org.apache.hadoop.util.Tool;
034import org.apache.hadoop.util.ToolRunner;
035import org.apache.yetus.audience.InterfaceAudience;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038
039import org.apache.hbase.thirdparty.org.apache.commons.cli.CommandLine;
040import org.apache.hbase.thirdparty.org.apache.commons.cli.DefaultParser;
041import org.apache.hbase.thirdparty.org.apache.commons.cli.HelpFormatter;
042import org.apache.hbase.thirdparty.org.apache.commons.cli.Options;
043
044
045/**
046 * A real-time monitoring tool for HBase like Unix top command.
047 */
048@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
049public class HBTop extends Configured implements Tool {
050
051  private static final Logger LOGGER = LoggerFactory.getLogger(HBTop.class);
052
053  public HBTop() {
054    this(HBaseConfiguration.create());
055  }
056
057  public HBTop(Configuration conf) {
058    super(Objects.requireNonNull(conf));
059  }
060
061  @Override
062  public int run(String[] args) throws Exception {
063    long initialRefreshDelay = 3 * 1000;
064    Mode initialMode = Mode.REGION;
065    List<Field> initialFields = null;
066    Field initialSortField = null;
067    Boolean initialAscendingSort = null;
068    List<RecordFilter> initialFilters = null;
069    long numberOfIterations = Long.MAX_VALUE;
070    boolean batchMode = false;
071    try {
072      Options opts = getOptions();
073      CommandLine commandLine = new DefaultParser().parse(opts, args);
074
075      if (commandLine.hasOption("help")) {
076        printUsage(opts);
077        return 0;
078      }
079
080      if (commandLine.hasOption("mode")) {
081        String mode = commandLine.getOptionValue("mode");
082        switch (mode) {
083          case "n":
084            initialMode = Mode.NAMESPACE;
085            break;
086
087          case "t":
088            initialMode = Mode.TABLE;
089            break;
090
091          case "r":
092            initialMode = Mode.REGION;
093            break;
094
095          case "s":
096            initialMode = Mode.REGION_SERVER;
097            break;
098
099          case "u":
100            initialMode = Mode.USER;
101            break;
102
103          case "c":
104            initialMode = Mode.CLIENT;
105            break;
106
107          default:
108            LOGGER.warn("Mode set invalid, using default");
109            break;
110        }
111      }
112
113      if (commandLine.hasOption("outputFieldNames")) {
114        initialMode.getFieldInfos().forEach(f -> System.out.println(f.getField().getHeader()));
115        return 0;
116      }
117
118      if (commandLine.hasOption("delay")) {
119        int delay = 0;
120        try {
121          delay = Integer.parseInt(commandLine.getOptionValue("delay"));
122        } catch (NumberFormatException ignored) {
123        }
124
125        if (delay < 1) {
126          LOGGER.warn("Delay set too low or invalid, using default");
127        } else {
128          initialRefreshDelay = delay * 1000L;
129        }
130      }
131
132      if (commandLine.hasOption("numberOfIterations")) {
133        try {
134          numberOfIterations = Long.parseLong(commandLine.getOptionValue("numberOfIterations"));
135        } catch (NumberFormatException ignored) {
136          LOGGER.warn("The number of iterations set invalid, ignoring");
137        }
138      }
139
140      if (commandLine.hasOption("sortField")) {
141        String sortField = commandLine.getOptionValue("sortField");
142
143        String field;
144        boolean ascendingSort;
145        if (sortField.startsWith("+")) {
146          field = sortField.substring(1);
147          ascendingSort = false;
148        } else if (sortField.startsWith("-")) {
149          field = sortField.substring(1);
150          ascendingSort = true;
151        } else {
152          field = sortField;
153          ascendingSort = false;
154        }
155
156        Optional<FieldInfo> fieldInfo = initialMode.getFieldInfos().stream()
157          .filter(f -> f.getField().getHeader().equals(field)).findFirst();
158        if (fieldInfo.isPresent()) {
159          initialSortField = fieldInfo.get().getField();
160          initialAscendingSort = ascendingSort;
161        } else {
162          LOGGER.warn("The specified sort field " + field + " is not found, using default");
163        }
164      }
165
166      if (commandLine.hasOption("fields")) {
167        String[] fields = commandLine.getOptionValue("fields").split(",");
168        initialFields = new ArrayList<>();
169        for (String field : fields) {
170          Optional<FieldInfo> fieldInfo = initialMode.getFieldInfos().stream()
171            .filter(f -> f.getField().getHeader().equals(field)).findFirst();
172          if (fieldInfo.isPresent()) {
173            initialFields.add(fieldInfo.get().getField());
174          } else {
175            LOGGER.warn("The specified field " + field + " is not found, ignoring");
176          }
177        }
178      }
179
180      if (commandLine.hasOption("filters")) {
181        String[] filters = commandLine.getOptionValue("filters").split(",");
182        List<Field> fields = initialMode.getFieldInfos().stream().map(FieldInfo::getField)
183          .collect(Collectors.toList());
184        for (String filter : filters) {
185          RecordFilter f = RecordFilter.parse(filter, fields, false);
186          if (f != null) {
187            if (initialFilters == null) {
188              initialFilters = new ArrayList<>();
189            }
190            initialFilters.add(f);
191          } else {
192            LOGGER.warn("The specified filter " + filter + " is invalid, ignoring");
193          }
194        }
195      }
196
197      if (commandLine.hasOption("batchMode")) {
198        batchMode = true;
199      }
200    } catch (Exception e) {
201      LOGGER.error("Unable to parse options", e);
202      return 1;
203    }
204
205    try (Screen screen = new Screen(getConf(), initialRefreshDelay, initialMode, initialFields,
206      initialSortField, initialAscendingSort, initialFilters, numberOfIterations, batchMode)) {
207      screen.run();
208    }
209
210    return 0;
211  }
212
213  private Options getOptions() {
214    Options opts = new Options();
215    opts.addOption("h", "help", false,
216      "Print usage; for help while the tool is running press 'h'");
217    opts.addOption("d", "delay", true,
218      "The refresh delay (in seconds); default is 3 seconds");
219    opts.addOption("m", "mode", true,
220      "The mode; n (Namespace)|t (Table)|r (Region)|s (RegionServer)|u (User)"
221        + "|c (Client), default is r");
222    opts.addOption("n", "numberOfIterations", true,
223      "The number of iterations");
224    opts.addOption("s", "sortField", true,
225      "The initial sort field. You can prepend a `+' or `-' to the field name to also override"
226        + " the sort direction. A leading `+' will force sorting high to low, whereas a `-' will"
227        + " ensure a low to high ordering");
228    opts.addOption("O", "outputFieldNames", false,
229      "Print each of the available field names on a separate line, then quit");
230    opts.addOption("f", "fields", true,
231      "Show only the given fields. Specify comma separated fields to show multiple fields");
232    opts.addOption("i", "filters", true,
233      "The initial filters. Specify comma separated filters to set multiple filters");
234    opts.addOption("b", "batchMode", false,
235      "Starts hbtop in Batch mode, which could be useful for sending output from hbtop to other"
236        + " programs or to a file. In this mode, hbtop will not accept input and runs until the"
237        + " iterations limit you've set with the `-n' command-line option or until killed");
238    return opts;
239  }
240
241  private void printUsage(Options opts) {
242    new HelpFormatter().printHelp("hbase hbtop [opts] [-D<property=value>]*", opts);
243    System.out.println("");
244    System.out.println(" Note: -D properties will be applied to the conf used.");
245    System.out.println("  For example:");
246    System.out.println("   -Dhbase.client.zookeeper.quorum=<zookeeper quorum>");
247    System.out.println("   -Dzookeeper.znode.parent=<znode parent>");
248    System.out.println("");
249  }
250
251  public static void main(String[] args) throws Exception {
252    int res = ToolRunner.run(new HBTop(), args);
253    System.exit(res);
254  }
