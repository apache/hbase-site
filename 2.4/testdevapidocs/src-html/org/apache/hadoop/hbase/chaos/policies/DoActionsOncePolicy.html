019package org.apache.hadoop.hbase.chaos.policies;
020
021import java.util.ArrayList;
022import java.util.Arrays;
023import java.util.List;
024
025import org.apache.hadoop.hbase.chaos.actions.Action;
026import org.apache.hadoop.util.StringUtils;
027
028/** A policy which performs a sequence of actions deterministically. */
029public class DoActionsOncePolicy extends PeriodicPolicy {
030  private List<Action> actions;
031
032  public DoActionsOncePolicy(long periodMs, List<Action> actions) {
033    super(periodMs);
034    this.actions = new ArrayList<>(actions);
035  }
036
037  public DoActionsOncePolicy(long periodMs, Action... actions) {
038    this(periodMs, Arrays.asList(actions));
039  }
040
041  @Override
042  protected void runOneIteration() {
043    if (actions.isEmpty()) {
044      this.stop("done");
045      return;
046    }
047    Action action = actions.remove(0);
048
049    try {
050      action.perform();
051    } catch (Exception ex) {
052      LOG.warn("Exception occurred during performing action: "
053          + StringUtils.stringifyException(ex));
054    }
055  }
056
057  @Override
058  public void init(PolicyContext context) throws Exception {
059    super.init(context);
060    for (Action action : actions) {
061      action.init(this.context);
062    }
063  }
064}
