035/**
036 * An optional metrics registry class for creating and maintaining a
037 * collection of MetricsMutables, making writing metrics source easier.
038 * NOTE: this is a copy of org.apache.hadoop.metrics2.lib.MetricsRegistry with added one
039 *       feature: metrics can be removed. When HADOOP-8313 is fixed, usages of this class
040 *       should be substituted with org.apache.hadoop.metrics2.lib.MetricsRegistry.
041 *       This implementation also provides handy methods for creating metrics
042 *       dynamically.
043 *       Another difference is that metricsMap implementation is substituted with
044 *       thread-safe map, as we allow dynamic metrics additions/removals.
045 */
046@InterfaceAudience.Private
047public class DynamicMetricsRegistry {
048  private static final Logger LOG = LoggerFactory.getLogger(DynamicMetricsRegistry.class);
049
050  private final ConcurrentMap<String, MutableMetric> metricsMap =
051      Maps.newConcurrentMap();
052  private final ConcurrentMap<String, MetricsTag> tagsMap =
053      Maps.newConcurrentMap();
054  private final MetricsInfo metricsInfo;
055  private final DefaultMetricsSystemHelper helper = new DefaultMetricsSystemHelper();

058    "_min",
059    "_max",

210  @InterfaceAudience.Private
211  public MutableRate newRate(String name, String desc,
212      boolean extended, boolean returnExisting) {
213    if (returnExisting) {
214      MutableMetric rate = metricsMap.get(name);
215      if (rate != null) {
216        if (rate instanceof MutableRate) {
217          return (MutableRate) rate;
218        }
219
220        throw new MetricsException("Unexpected metrics type "+ rate.getClass()
221                                   +" for "+ name);
222      }
223    }
224    MutableRate ret = new MutableRate(name, desc, extended);
225    return addNewMetricIfAbsent(name, ret, MutableRate.class);
226  }

293  /**
294   * Add sample to a stat metric by name.
295   * @param name  of the metric
296   * @param value of the snapshot to add
297   */
298  public void add(String name, long value) {
299    MutableMetric m = metricsMap.get(name);
300
301    if (m != null) {
302      if (m instanceof MutableStat) {
303        ((MutableStat) m).add(value);
304      }
305      else {
306        throw new MetricsException("Unsupported add(value) for metric "+ name);
307      }
308    }
309    else {
310      metricsMap.put(name, newRate(name)); // default is a rate metric
311      add(name, value);
312    }
313  }

348  /**
349   * Add a tag to the metrics
350   * @param info  metadata of the tag
351   * @param value of the tag
352   * @param override existing tag if true
353   * @return the registry (for keep adding tags etc.)
354   */
355  public DynamicMetricsRegistry tag(MetricsInfo info, String value, boolean override) {
356    MetricsTag tag = Interns.tag(info, value);
357
358    if (!override) {
359      MetricsTag existing = tagsMap.putIfAbsent(info.name(), tag);
360      if (existing != null) {
361        throw new MetricsException("Tag "+ info.name() +" already exists!");
362      }
363      return this;
364    }
365
366    tagsMap.put(info.name(), tag);
367
368    return this;
369  }

383  /**
384   * Sample all the mutable metrics and put the snapshot in the builder
385   * @param builder to contain the metrics snapshot
386   * @param all get all the metrics even if the values are not changed.
387   */
388  public void snapshot(MetricsRecordBuilder builder, boolean all) {
389    for (MetricsTag tag : tags()) {
390      builder.add(tag);
391    }
392    for (MutableMetric metric : metrics()) {
393      metric.snapshot(builder, all);
394    }
395  }

418  /**
419   * Get a MetricMutableGaugeLong from the storage.  If it is not there atomically put it.
420   *
421   * @param gaugeName              name of the gauge to create or get.
422   * @param potentialStartingValue value of the new gauge if we have to create it.
423   */
424  public MutableGaugeLong getGauge(String gaugeName, long potentialStartingValue) {
425    //Try and get the guage.
426    MutableMetric metric = metricsMap.get(gaugeName);
427
428    //If it's not there then try and put a new one in the storage.
429    if (metric == null) {
430
431      //Create the potential new gauge.
432      MutableGaugeLong newGauge = new MutableGaugeLong(new MetricsInfoImpl(gaugeName, ""),
433              potentialStartingValue);
434
435      // Try and put the gauge in.  This is atomic.
436      metric = metricsMap.putIfAbsent(gaugeName, newGauge);
437
438      //If the value we get back is null then the put was successful and we will return that.
439      //otherwise gaugeLong should contain the thing that was in before the put could be completed.
440      if (metric == null) {
441        return newGauge;
442      }
443    }
444
445    if (!(metric instanceof MutableGaugeLong)) {
446      throw new MetricsException("Metric already exists in registry for metric name: " + gaugeName +
447              " and not of type MetricMutableGaugeLong");
448    }
449
450    return (MutableGaugeLong) metric;
451  }

487  /**
488   * Get a MetricMutableCounterLong from the storage.  If it is not there atomically put it.
489   *
490   * @param counterName            Name of the counter to get
491   * @param potentialStartingValue starting value if we have to create a new counter
492   */
493  public MutableFastCounter getCounter(String counterName, long potentialStartingValue) {
494    //See getGauge for description on how this works.
495    MutableMetric counter = metricsMap.get(counterName);
496    if (counter == null) {
497      MutableFastCounter newCounter =
498              new MutableFastCounter(new MetricsInfoImpl(counterName, ""), potentialStartingValue);
499      counter = metricsMap.putIfAbsent(counterName, newCounter);
500      if (counter == null) {
501        return newCounter;
502      }
503    }
504
505
506    if (!(counter instanceof MutableCounter)) {
507      throw new MetricsException("Metric already exists in registry for metric name: " +
508              counterName + " and not of type MutableCounter");
509    }
510
511    return (MutableFastCounter) counter;
512  }

514  public MutableHistogram getHistogram(String histoName) {
515    //See getGauge for description on how this works.
516    MutableMetric histo = metricsMap.get(histoName);
517    if (histo == null) {
518      MutableHistogram newCounter =
519          new MutableHistogram(new MetricsInfoImpl(histoName, ""));
520      histo = metricsMap.putIfAbsent(histoName, newCounter);
521      if (histo == null) {
522        return newCounter;
523      }
524    }
525
526
527    if (!(histo instanceof MutableHistogram)) {
528      throw new MetricsException("Metric already exists in registry for metric name: " +
529          histoName + " and not of type MutableHistogram");
530    }
531
532    return (MutableHistogram) histo;
533  }

535  private<T extends MutableMetric> T addNewMetricIfAbsent(String name, T ret,
536      Class<T> metricClass) {
537    //If the value we get back is null then the put was successful and we will
538    // return that. Otherwise metric should contain the thing that was in
539    // before the put could be completed.
540    MutableMetric metric = metricsMap.putIfAbsent(name, ret);
541    if (metric == null) {
542      return ret;
543    }
544
545    return returnExistingWithCast(metric, metricClass, name);
546  }
547
548  @SuppressWarnings("unchecked")
549  private<T> T returnExistingWithCast(MutableMetric metric,
550                                      Class<T> metricClass, String name) {
551    if (!metricClass.isAssignableFrom(metric.getClass())) {
552      throw new MetricsException("Metric already exists in registry for metric name: " +
553              name + " and not of type " + metricClass +
554              " but instead of type " + metric.getClass());
555    }
556
557    return (T) metric;
558  }

563    }
