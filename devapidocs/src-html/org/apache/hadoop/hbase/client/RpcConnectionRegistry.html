018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021import java.util.Set;
022import java.util.concurrent.CompletableFuture;
023import java.util.stream.Collectors;
024import org.apache.commons.lang3.StringUtils;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.HBaseInterfaceAudience;
027import org.apache.hadoop.hbase.ServerName;
028import org.apache.yetus.audience.InterfaceAudience;
029
030import org.apache.hbase.thirdparty.com.google.common.base.Splitter;
031
032import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
033import org.apache.hadoop.hbase.shaded.protobuf.generated.RegistryProtos.ClientMetaService;
034import org.apache.hadoop.hbase.shaded.protobuf.generated.RegistryProtos.GetBootstrapNodesRequest;
035import org.apache.hadoop.hbase.shaded.protobuf.generated.RegistryProtos.GetBootstrapNodesResponse;
036
037/**
038 * Rpc based connection registry. It will make use of the {@link ClientMetaService} to get registry
039 * information.
040 * <p/>
041 * It needs bootstrap node list when start up, and then it will use {@link ClientMetaService} to
042 * refresh the bootstrap node list periodically.
043 * <p/>
044 * Usually, you could set masters as the bootstrap nodes,as they will also implement the
045 * {@link ClientMetaService}, and then, we will switch to use region servers after refreshing the
046 * bootstrap nodes.
047 */
048@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
049public class RpcConnectionRegistry extends AbstractRpcBasedConnectionRegistry {
050
051  /** Configuration key that controls the fan out of requests **/
052  public static final String HEDGED_REQS_FANOUT_KEY = "hbase.client.bootstrap.hedged.fanout";
053
054  /**
055   * As end users could configure any nodes in a cluster as the initial bootstrap nodes, it is
056   * possible that different end users will configure the same machine which makes the machine over
057   * load. So we should have a shorter delay for the initial refresh, to let users quickly switch to
058   * the bootstrap nodes we want them to connect to.
059   * <p/>
060   * The default value for initial refresh delay is 1/10 of periodic refresh interval.
061   */
062  public static final String INITIAL_REFRESH_DELAY_SECS =
063    "hbase.client.bootstrap.initial_refresh_delay_secs";
064
065  public static final String PERIODIC_REFRESH_INTERVAL_SECS =
066    "hbase.client.bootstrap.refresh_interval_secs";
067
068  public static final String MIN_SECS_BETWEEN_REFRESHES =
069    "hbase.client.bootstrap.min_secs_between_refreshes";
070
071  public static final String BOOTSTRAP_NODES = "hbase.client.bootstrap.servers";
072
073  private static final char ADDRS_CONF_SEPARATOR = ',';
074
075  RpcConnectionRegistry(Configuration conf) throws IOException {
076    super(conf, HEDGED_REQS_FANOUT_KEY, INITIAL_REFRESH_DELAY_SECS, PERIODIC_REFRESH_INTERVAL_SECS,
077      MIN_SECS_BETWEEN_REFRESHES);
078  }
079
080  @Override
081  protected Set<ServerName> getBootstrapNodes(Configuration conf) throws IOException {
082    // try get bootstrap nodes config first
083    String configuredBootstrapNodes = conf.get(BOOTSTRAP_NODES);
084    if (!StringUtils.isBlank(configuredBootstrapNodes)) {
085      return Splitter.on(ADDRS_CONF_SEPARATOR).trimResults().splitToStream(configuredBootstrapNodes)
086        .map(addr -> ServerName.valueOf(addr, ServerName.NON_STARTCODE))
087        .collect(Collectors.toSet());
088    } else {
089      // otherwise, just use master addresses
090      return MasterRegistry.parseMasterAddrs(conf);
091    }
092  }
093
094  private static Set<ServerName> transformServerNames(GetBootstrapNodesResponse resp) {
095    return resp.getServerNameList().stream().map(ProtobufUtil::toServerName)
096      .collect(Collectors.toSet());
097  }
098
099  private CompletableFuture<Set<ServerName>> getBootstrapNodes() {
100    return this
101      .<GetBootstrapNodesResponse> call(
102        (c, s, d) -> s.getBootstrapNodes(c, GetBootstrapNodesRequest.getDefaultInstance(), d),
103        r -> r.getServerNameCount() != 0, "getBootstrapNodes()")
104      .thenApply(RpcConnectionRegistry::transformServerNames);
105  }
106
107  @Override
108  protected CompletableFuture<Set<ServerName>> fetchEndpoints() {
109    return getBootstrapNodes();
110  }
111}
