076  /**
077   * Calculate pause time. Built on {@link HConstants#RETRY_BACKOFF}.
078   * @param pause time to pause
079   * @param tries amount of tries
080   * @return How long to wait after <code>tries</code> retries
081   */
082  public static long getPauseTime(final long pause, final int tries) {
083    int ntries = tries;
084    if (ntries >= HConstants.RETRY_BACKOFF.length) {
085      ntries = HConstants.RETRY_BACKOFF.length - 1;
086    }
087    if (ntries < 0) {
088      ntries = 0;
089    }
090
091    long normalPause = pause * HConstants.RETRY_BACKOFF[ntries];
092    // 1% possible jitter
093    long jitter = (long) (normalPause * ThreadLocalRandom.current().nextFloat() * 0.01f);
094    return normalPause + jitter;
095  }
096
097  /**
098   * Adds / subs an up to 50% jitter to a pause time. Minimum is 1.
099   * @param pause the expected pause.
100   * @param jitter the jitter ratio, between 0 and 1, exclusive.
101   */
102  public static long addJitter(final long pause, final float jitter) {
103    float lag = pause * (ThreadLocalRandom.current().nextFloat() - 0.5f) * jitter;
104    long newPause = pause + (long) lag;
105    if (newPause <= 0) {
106      return 1;
107    }
108    return newPause;
109  }

121  /**
122   * Changes the configuration to set the number of retries needed when using Connection internally,
123   * e.g. for updating catalog tables, etc. Call this method before we create any Connections.
124   * @param c The Configuration instance to set the retries into.
125   * @param log Used to log what we set in here.
126   */
127  public static void setServerSideHConnectionRetriesConfig(final Configuration c, final String sn,
128      final Log log) {
129    // TODO: Fix this. Not all connections from server side should have 10 times the retries.
130    int hcRetries = c.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
131      HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
132    // Go big. Multiply by 10. If we can't get to meta after this many retries
133    // then something seriously wrong.
134    int serversideMultiplier = c.getInt("hbase.client.serverside.retries.multiplier", 10);
135    int retries = hcRetries * serversideMultiplier;
136    c.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, retries);
137    log.info(sn + " server-side Connection retries=" + retries);
138  }

140  /**
141   * Creates a short-circuit connection that can bypass the RPC layer (serialization,
142   * deserialization, networking, etc..) when talking to a local server.
143   * @param conf the current configuration
144   * @param pool the thread pool to use for batch operations
145   * @param user the user the connection is for
146   * @param serverName the local server name
147   * @param admin the admin interface of the local server
148   * @param client the client interface of the local server
149   * @return an short-circuit connection.
150   * @throws IOException if IO failure occurred
151   */
152  public static ClusterConnection createShortCircuitConnection(final Configuration conf,
153      ExecutorService pool, User user, final ServerName serverName,
154      final AdminService.BlockingInterface admin, final ClientService.BlockingInterface client)
155      throws IOException {
156    if (user == null) {
157      user = UserProvider.instantiate(conf).getCurrent();
158    }
159    return new ConnectionImplementation(conf, pool, user) {
160      @Override
161      public AdminService.BlockingInterface getAdmin(ServerName sn) throws IOException {
162        return serverName.equals(sn) ? admin : super.getAdmin(sn);
163      }
164
165      @Override
166      public ClientService.BlockingInterface getClient(ServerName sn) throws IOException {
167        return serverName.equals(sn) ? client : super.getClient(sn);
168      }
169
170      @Override
171      public MasterKeepAliveConnection getKeepAliveMasterService()
172          throws MasterNotRunningException {
173        if (!(client instanceof MasterService.BlockingInterface)) {
174          return super.getKeepAliveMasterService();
175        } else {
176          return new ShortCircuitMasterConnection((MasterService.BlockingInterface) client);
177        }
178      }
179    };
180  }

214  /**
215   * Get a unique key for the rpc stub to the given server.
216   */
217  static String getStubKey(String serviceName, ServerName serverName, boolean hostnameCanChange) {
218    // Sometimes, servers go down and they come back up with the same hostname but a different
219    // IP address. Force a resolution of the rsHostname by trying to instantiate an
220    // InetSocketAddress, and this way we will rightfully get a new stubKey.
221    // Also, include the hostname in the key so as to take care of those cases where the
222    // DNS name is different but IP address remains the same.
223    String hostname = serverName.getHostname();
224    int port = serverName.getPort();
225    if (hostnameCanChange) {
226      try {
227        InetAddress ip = InetAddress.getByName(hostname);
228        return serviceName + "@" + hostname + "-" + ip.getHostAddress() + ":" + port;
229      } catch (UnknownHostException e) {
230        LOG.warn("Can not resolve " + hostname + ", please check your network", e);
231      }
232    }
233    return serviceName + "@" + hostname + ":" + port;
234  }

255  // A byte array in which all elements are the max byte, and it is used to
256  // construct closest front row
257  static final byte[] MAX_BYTE_ARRAY = Bytes.createMaxByteArray(9);
258
259  /**
260   * Create the closest row after the specified row
261   */
262  static byte[] createClosestRowAfter(byte[] row) {
263    return Arrays.copyOf(row, row.length + 1);
264  }
265
266  /**
267   * Create a row before the specified row and very close to the specified row.
268   */
269  static byte[] createCloseRowBefore(byte[] row) {
270    if (row.length == 0) {
271      return MAX_BYTE_ARRAY;
272    }
273    if (row[row.length - 1] == 0) {
274      return Arrays.copyOf(row, row.length - 1);
275    } else {
276      byte[] nextRow = new byte[row.length + MAX_BYTE_ARRAY.length];
277      System.arraycopy(row, 0, nextRow, 0, row.length - 1);
278      nextRow[row.length - 1] = (byte) ((row[row.length - 1] & 0xFF) - 1);
279      System.arraycopy(MAX_BYTE_ARRAY, 0, nextRow, row.length, MAX_BYTE_ARRAY.length);
280      return nextRow;
281    }
282  }

300  static Throwable translateException(Throwable t) {
301    if (t instanceof UndeclaredThrowableException && t.getCause() != null) {
302      t = t.getCause();
303    }
304    if (t instanceof RemoteException) {
305      t = ((RemoteException) t).unwrapRemoteException();
306    }
307    if (t instanceof ServiceException && t.getCause() != null) {
308      t = translateException(t.getCause());
309    }
310    return t;
311  }

313  static long calcEstimatedSize(Result rs) {
314    long estimatedHeapSizeOfResult = 0;
315    // We don't make Iterator here
316    for (Cell cell : rs.rawCells()) {
317      estimatedHeapSizeOfResult += CellUtil.estimatedHeapSizeOf(cell);
318    }
319    return estimatedHeapSizeOfResult;
320  }

322  static Result filterCells(Result result, Cell keepCellsAfter) {
323    if (keepCellsAfter == null) {
324      // do not need to filter
325      return result;
326    }
327    // not the same row
328    if (!CellUtil.matchingRow(keepCellsAfter, result.getRow(), 0, result.getRow().length)) {
329      return result;
330    }
331    Cell[] rawCells = result.rawCells();
332    int index = Arrays.binarySearch(rawCells, keepCellsAfter, CellComparator::compareWithoutRow);
333    if (index < 0) {
334      index = -index - 1;
335    } else {
336      index++;
337    }
338    if (index == 0) {
339      return result;
340    }
341    if (index == rawCells.length) {
342      return null;
343    }
344    return Result.create(Arrays.copyOfRange(rawCells, index, rawCells.length), null,
345      result.isStale(), result.mayHaveMoreCellsInRow());
346  }

362  static RegionLocateType getLocateType(Scan scan) {
363    if (scan.isReversed()) {
364      if (isEmptyStartRow(scan.getStartRow())) {
365        return RegionLocateType.BEFORE;
366      } else {
367        return scan.includeStartRow() ? RegionLocateType.CURRENT : RegionLocateType.BEFORE;
368      }
369    } else {
370      return scan.includeStartRow() ? RegionLocateType.CURRENT : RegionLocateType.AFTER;
371    }
372  }

374  static boolean noMoreResultsForScan(Scan scan, HRegionInfo info) {
375    if (isEmptyStopRow(info.getEndKey())) {
376      return true;
377    }
378    if (isEmptyStopRow(scan.getStopRow())) {
379      return false;
380    }
381    int c = Bytes.compareTo(info.getEndKey(), scan.getStopRow());
382    // 1. if our stop row is less than the endKey of the region
383    // 2. if our stop row is equal to the endKey of the region and we do not include the stop row
384    // for scan.
385    return c > 0 || (c == 0 && !scan.includeStopRow());
386  }
387
388  static boolean noMoreResultsForReverseScan(Scan scan, HRegionInfo info) {
389    if (isEmptyStartRow(info.getStartKey())) {
390      return true;
391    }
392    if (isEmptyStopRow(scan.getStopRow())) {
393      return false;
394    }
395    // no need to test the inclusive of the stop row as the start key of a region is included in
396    // the region.
397    return Bytes.compareTo(info.getStartKey(), scan.getStopRow()) <= 0;
398  }

400  static <T> CompletableFuture<List<T>> allOf(List<CompletableFuture<T>> futures) {
401    return CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]))
402        .thenApply(v -> futures.stream().map(f -> f.getNow(null)).collect(toList()));
403  }

450  static void updateResultsMetrics(ScanMetrics scanMetrics, Result[] rrs,
451      boolean isRegionServerRemote) {
452    if (scanMetrics == null || rrs == null || rrs.length == 0) {
453      return;
454    }
455    long resultSize = 0;
456    for (Result rr : rrs) {
457      for (Cell cell : rr.rawCells()) {
458        resultSize += CellUtil.estimatedSerializedSizeOf(cell);
459      }
460    }
461    scanMetrics.countOfBytesInResults.addAndGet(resultSize);
462    if (isRegionServerRemote) {
463      scanMetrics.countOfBytesInRemoteResults.addAndGet(resultSize);
464    }
465  }

468   * Use the scan metrics returned by the server to add to the identically named counters in the
469   * client side metrics. If a counter does not exist with the same name as the server side metric,

473    if (scanMetrics == null || response == null || !response.hasScanMetrics()) {
474      return;
475    }
476    ResponseConverter.getScanMetrics(response).forEach(scanMetrics::addToCounter);
477  }
