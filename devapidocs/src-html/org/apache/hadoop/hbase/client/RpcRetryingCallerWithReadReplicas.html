054/**
055 * Caller that goes to replica if the primary region does no answer within a configurable
056 * timeout. If the timeout is reached, it calls all the secondary replicas, and returns
057 * the first answer. If the answer comes from one of the secondary replica, it will
058 * be marked as stale.
059 */
060@InterfaceAudience.Private
061public class RpcRetryingCallerWithReadReplicas {
062  private static final Log LOG = LogFactory.getLog(RpcRetryingCallerWithReadReplicas.class);
063
064  protected final ExecutorService pool;
065  protected final ClusterConnection cConnection;
066  protected final Configuration conf;
067  protected final Get get;
068  protected final TableName tableName;
069  protected final int timeBeforeReplicas;
070  private final int callTimeout;
071  private final int retries;
072  private final RpcControllerFactory rpcControllerFactory;
073  private final RpcRetryingCallerFactory rpcRetryingCallerFactory;
074
075  public RpcRetryingCallerWithReadReplicas(
076      RpcControllerFactory rpcControllerFactory, TableName tableName,
077      ClusterConnection cConnection, final Get get,
078      ExecutorService pool, int retries, int callTimeout,
079      int timeBeforeReplicas) {
080    this.rpcControllerFactory = rpcControllerFactory;
081    this.tableName = tableName;
082    this.cConnection = cConnection;
083    this.conf = cConnection.getConfiguration();
084    this.get = get;
085    this.pool = pool;
086    this.retries = retries;
087    this.callTimeout = callTimeout;
088    this.timeBeforeReplicas = timeBeforeReplicas;
089    this.rpcRetryingCallerFactory = new RpcRetryingCallerFactory(conf);
090  }
091
092  /**
093   * A RegionServerCallable that takes into account the replicas, i.e.
094   * - the call can be on any replica
095   * - we need to stop retrying when the call is completed
096   * - we can be interrupted
097   */
098  class ReplicaRegionServerCallable extends RegionServerCallable<Result> implements Cancellable {
099    final int id;
100    private final PayloadCarryingRpcController controller;
101
102    public ReplicaRegionServerCallable(int id, HRegionLocation location) {
103      super(RpcRetryingCallerWithReadReplicas.this.cConnection,
104          RpcRetryingCallerWithReadReplicas.this.tableName, get.getRow());
105      this.id = id;
106      this.location = location;
107      this.controller = rpcControllerFactory.newController();
108      controller.setPriority(tableName);
109    }
110
111    @Override
112    public void cancel() {
113      controller.startCancel();
114    }
115
116    /**
117     * Two responsibilities
118     * - if the call is already completed (by another replica) stops the retries.
119     * - set the location to the right region, depending on the replica.
120     */
121    @Override
122    public void prepare(final boolean reload) throws IOException {
123      if (controller.isCanceled()) return;
124
125      if (Thread.interrupted()) {
126        throw new InterruptedIOException();
127      }
128
129      if (reload || location == null) {
130        RegionLocations rl = getRegionLocations(false, id, cConnection, tableName, get.getRow());
131        location = id < rl.size() ? rl.getRegionLocation(id) : null;
132      }
133
134      if (location == null || location.getServerName() == null) {
135        // With this exception, there will be a retry. The location can be null for a replica
136        //  when the table is created or after a split.
137        throw new HBaseIOException("There is no location for replica id #" + id);
138      }
139
140      ServerName dest = location.getServerName();
141
142      setStub(cConnection.getClient(dest));
143    }
144
145    @Override
146    public Result call(int callTimeout) throws Exception {
147      if (controller.isCanceled()) return null;
148
149      if (Thread.interrupted()) {
150        throw new InterruptedIOException();
151      }
152
153      byte[] reg = location.getRegionInfo().getRegionName();
154
155      ClientProtos.GetRequest request =
156          RequestConverter.buildGetRequest(reg, get);
157      controller.setCallTimeout(callTimeout);
158
159      try {
160        ClientProtos.GetResponse response = getStub().get(controller, request);
161        if (response == null) {
162          return null;
163        }
164        return ProtobufUtil.toResult(response.getResult());
165      } catch (ServiceException se) {
166        throw ProtobufUtil.getRemoteException(se);
167      }
168    }
169
170    @Override
171    public boolean isCancelled() {
172      return controller.isCanceled();
173    }
174  }
175
176  /**
177   * <p>
178   * Algo:
179   * - we put the query into the execution pool.
180   * - after x ms, if we don't have a result, we add the queries for the secondary replicas
181   * - we take the first answer
182   * - when done, we cancel what's left. Cancelling means:
183   * - removing from the pool if the actual call was not started
184   * - interrupting the call if it has started
185   * Client side, we need to take into account
186   * - a call is not executed immediately after being put into the pool
187   * - a call is a thread. Let's not multiply the number of thread by the number of replicas.
188   * Server side, if we can cancel when it's still in the handler pool, it's much better, as a call
189   * can take some i/o.
190   * </p>
191   * Globally, the number of retries, timeout and so on still applies, but it's per replica,
192   * not global. We continue until all retries are done, or all timeouts are exceeded.
193   */
194  public synchronized Result call()
195      throws DoNotRetryIOException, InterruptedIOException, RetriesExhaustedException {
196    boolean isTargetReplicaSpecified = (get.getReplicaId() >= 0);
197
198    RegionLocations rl = getRegionLocations(true, (isTargetReplicaSpecified ? get.getReplicaId()
199        : RegionReplicaUtil.DEFAULT_REPLICA_ID), cConnection, tableName, get.getRow());
200    ResultBoundedCompletionService<Result> cs =
201        new ResultBoundedCompletionService<Result>(this.rpcRetryingCallerFactory, pool, rl.size());
202
203    if(isTargetReplicaSpecified) {
204      addCallsForReplica(cs, rl, get.getReplicaId(), get.getReplicaId());
205    } else {
206      addCallsForReplica(cs, rl, 0, 0);
207      try {
208        // wait for the timeout to see whether the primary responds back
209        Future<Result> f = cs.poll(timeBeforeReplicas, TimeUnit.MICROSECONDS); // Yes, microseconds
210        if (f != null) {
211          return f.get(); //great we got a response
212        }
213      } catch (ExecutionException e) {
214        throwEnrichedException(e, retries);
215      } catch (CancellationException e) {
216        throw new InterruptedIOException();
217      } catch (InterruptedException e) {
218        throw new InterruptedIOException();
219      }
220
221      // submit call for the all of the secondaries at once
222      addCallsForReplica(cs, rl, 1, rl.size() - 1);
223    }
224
225    try {
226      try {
227        Future<Result> f = cs.take();
228        return f.get();
229      } catch (ExecutionException e) {
230        throwEnrichedException(e, retries);
231      }
232    } catch (CancellationException e) {
233      throw new InterruptedIOException();
234    } catch (InterruptedException e) {
235      throw new InterruptedIOException();
236    } finally {
237      // We get there because we were interrupted or because one or more of the
238      // calls succeeded or failed. In all case, we stop all our tasks.
239      cs.cancelAll();
240    }
241
242    return null; // unreachable
243  }
244
245  /**
246   * Extract the real exception from the ExecutionException, and throws what makes more
247   * sense.
248   */
249  static void throwEnrichedException(ExecutionException e, int retries)
250      throws RetriesExhaustedException, DoNotRetryIOException {
251    Throwable t = e.getCause();
252    assert t != null; // That's what ExecutionException is about: holding an exception
253
254    if (t instanceof RetriesExhaustedException) {
255      throw (RetriesExhaustedException) t;
256    }
257
258    if (t instanceof DoNotRetryIOException) {
259      throw (DoNotRetryIOException) t;
260    }
261
262    RetriesExhaustedException.ThrowableWithExtraContext qt =
263        new RetriesExhaustedException.ThrowableWithExtraContext(t,
264            EnvironmentEdgeManager.currentTime(), null);
265
266    List<RetriesExhaustedException.ThrowableWithExtraContext> exceptions =
267        Collections.singletonList(qt);
268
269    throw new RetriesExhaustedException(retries, exceptions);
270  }
271
272  /**
273   * Creates the calls and submit them
274   *
275   * @param cs  - the completion service to use for submitting
276   * @param rl  - the region locations
277   * @param min - the id of the first replica, inclusive
278   * @param max - the id of the last replica, inclusive.
279   */
280  private void addCallsForReplica(ResultBoundedCompletionService<Result> cs,
281                                 RegionLocations rl, int min, int max) {
282    for (int id = min; id <= max; id++) {
283      HRegionLocation hrl = rl.getRegionLocation(id);
284      ReplicaRegionServerCallable callOnReplica = new ReplicaRegionServerCallable(id, hrl);
285      cs.submit(callOnReplica, callTimeout, id);
286    }
287  }
288
289  static RegionLocations getRegionLocations(boolean useCache, int replicaId,
290                 ClusterConnection cConnection, TableName tableName, byte[] row)
291      throws RetriesExhaustedException, DoNotRetryIOException, InterruptedIOException {
292
293    RegionLocations rl;
294    try {
295      if (!useCache) {
296        rl = cConnection.relocateRegion(tableName, row, replicaId);
297      } else {
298        rl = cConnection.locateRegion(tableName, row, useCache, true, replicaId);
299      }
300    } catch (DoNotRetryIOException e) {
301      throw e;
302    } catch (RetriesExhaustedException e) {
303      throw e;
304    } catch (InterruptedIOException e) {
305      throw e;
306    } catch (IOException e) {
307      throw new RetriesExhaustedException("Can't get the location", e);
308    }
309    if (rl == null) {
310      throw new RetriesExhaustedException("Can't get the locations");
311    }
312
313    return rl;
314  }
