012package org.apache.hadoop.hbase.client;
013
014import org.apache.hadoop.conf.Configuration;
015import org.apache.hadoop.hbase.HConstants;
016import org.apache.hadoop.hbase.classification.InterfaceAudience;
017
018import com.google.common.annotations.VisibleForTesting;
019
020/**
021 * Configuration parameters for the connection.
022 * Configuration is a heavy weight registry that does a lot of string operations and regex matching.
023 * Method calls into Configuration account for high CPU usage and have huge performance impact.
024 * This class caches connection-related configuration values in the  ConnectionConfiguration
025 * object so that expensive conf.getXXX() calls are avoided every time HTable, etc is instantiated.
026 * see HBASE-12128
027 */
028@InterfaceAudience.Private
029public class ConnectionConfiguration {
030
031  public static final String WRITE_BUFFER_SIZE_KEY = "hbase.client.write.buffer";
032  public static final long WRITE_BUFFER_SIZE_DEFAULT = 2097152;
033  public static final String MAX_KEYVALUE_SIZE_KEY = "hbase.client.keyvalue.maxsize";
034  public static final int MAX_KEYVALUE_SIZE_DEFAULT = -1;
035
036  private final long writeBufferSize;
037  private final int metaOperationTimeout;
038  private final int operationTimeout;
039  private final int scannerCaching;
040  private final long scannerMaxResultSize;
041  private final int primaryCallTimeoutMicroSecond;
042  private final int replicaCallTimeoutMicroSecondScan;
043  private final int metaReplicaCallTimeoutMicroSecondScan;
044  private final int retries;
045  private final int maxKeyValueSize;
046  private final int rpcTimeout;
047  private final int readRpcTimeout;
048  private final int writeRpcTimeout;
049  // toggle for async/sync prefetch
050  private final boolean clientScannerAsyncPrefetch;
051
052    /**
053   * Constructor
054   * @param conf Configuration object
055   */
056  ConnectionConfiguration(Configuration conf) {
057    this.writeBufferSize = conf.getLong(WRITE_BUFFER_SIZE_KEY, WRITE_BUFFER_SIZE_DEFAULT);
058
059    this.metaOperationTimeout = conf.getInt(HConstants.HBASE_CLIENT_META_OPERATION_TIMEOUT,
060        HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
061
062    this.operationTimeout = conf.getInt(
063      HConstants.HBASE_CLIENT_OPERATION_TIMEOUT, HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
064
065    this.scannerCaching = conf.getInt(
066      HConstants.HBASE_CLIENT_SCANNER_CACHING, HConstants.DEFAULT_HBASE_CLIENT_SCANNER_CACHING);
067
068    this.scannerMaxResultSize =
069        conf.getLong(HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY,
070            HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE);
071
072    this.primaryCallTimeoutMicroSecond =
073        conf.getInt("hbase.client.primaryCallTimeout.get", 10000); // 10ms
074
075    this.replicaCallTimeoutMicroSecondScan =
076        conf.getInt("hbase.client.replicaCallTimeout.scan", 1000000); // 1000 ms
077
078    this.metaReplicaCallTimeoutMicroSecondScan =
079        conf.getInt(HConstants.HBASE_CLIENT_MEAT_REPLICA_SCAN_TIMEOUT,
080            HConstants.HBASE_CLIENT_MEAT_REPLICA_SCAN_TIMEOUT_DEFAULT);
081
082    this.retries = conf.getInt(
083       HConstants.HBASE_CLIENT_RETRIES_NUMBER, HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
084
085    this.clientScannerAsyncPrefetch = conf.getBoolean(
086       Scan.HBASE_CLIENT_SCANNER_ASYNC_PREFETCH, Scan.DEFAULT_HBASE_CLIENT_SCANNER_ASYNC_PREFETCH);
087
088    this.maxKeyValueSize = conf.getInt(MAX_KEYVALUE_SIZE_KEY, MAX_KEYVALUE_SIZE_DEFAULT);
089
090    this.rpcTimeout =
091        conf.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY, HConstants.DEFAULT_HBASE_RPC_TIMEOUT);
092
093    this.readRpcTimeout = conf.getInt(HConstants.HBASE_RPC_READ_TIMEOUT_KEY,
094        conf.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY, HConstants.DEFAULT_HBASE_RPC_TIMEOUT));
095
096    this.writeRpcTimeout = conf.getInt(HConstants.HBASE_RPC_WRITE_TIMEOUT_KEY,
097        conf.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY, HConstants.DEFAULT_HBASE_RPC_TIMEOUT));
098  }
099
100  /**
101   * Constructor
102   * This is for internal testing purpose (using the default value).
103   * In real usage, we should read the configuration from the Configuration object.
104   */
105  @VisibleForTesting
106  protected ConnectionConfiguration() {
107    this.writeBufferSize = WRITE_BUFFER_SIZE_DEFAULT;
108    this.metaOperationTimeout = HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT;
109    this.operationTimeout = HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT;
110    this.scannerCaching = HConstants.DEFAULT_HBASE_CLIENT_SCANNER_CACHING;
111    this.scannerMaxResultSize = HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE;
112    this.primaryCallTimeoutMicroSecond = 10000;
113    this.replicaCallTimeoutMicroSecondScan = 1000000;
114    this.metaReplicaCallTimeoutMicroSecondScan =
115        HConstants.HBASE_CLIENT_MEAT_REPLICA_SCAN_TIMEOUT_DEFAULT;
116    this.retries = HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER;
117    this.clientScannerAsyncPrefetch = Scan.DEFAULT_HBASE_CLIENT_SCANNER_ASYNC_PREFETCH;
118    this.maxKeyValueSize = MAX_KEYVALUE_SIZE_DEFAULT;
119    this.readRpcTimeout = HConstants.DEFAULT_HBASE_RPC_TIMEOUT;
120    this.writeRpcTimeout = HConstants.DEFAULT_HBASE_RPC_TIMEOUT;
121    this.rpcTimeout = HConstants.DEFAULT_HBASE_RPC_TIMEOUT;
122  }
123
124  public int getReadRpcTimeout() {
125    return readRpcTimeout;
126  }
127
128  public int getWriteRpcTimeout() {
129    return writeRpcTimeout;
130  }
131
132  public long getWriteBufferSize() {
133    return writeBufferSize;
134  }
135
136  public int getMetaOperationTimeout() {
137    return metaOperationTimeout;
138  }
139
140  public int getOperationTimeout() {
141    return operationTimeout;
142  }
143
144  public int getScannerCaching() {
145    return scannerCaching;
146  }
147
148  public int getPrimaryCallTimeoutMicroSecond() {
149    return primaryCallTimeoutMicroSecond;
150  }
151
152  public int getReplicaCallTimeoutMicroSecondScan() {
153    return replicaCallTimeoutMicroSecondScan;
154  }
155
156  public int getMetaReplicaCallTimeoutMicroSecondScan() {
157    return metaReplicaCallTimeoutMicroSecondScan;
158  }
159
160  public int getRetriesNumber() {
161    return retries;
162  }
163
164  public int getMaxKeyValueSize() {
165    return maxKeyValueSize;
166  }
167
168  public long getScannerMaxResultSize() {
169    return scannerMaxResultSize;
170  }
171
172  public boolean isClientScannerAsyncPrefetch() {
173    return clientScannerAsyncPrefetch;
174  }
175
176  public int getRpcTimeout() {
177    return rpcTimeout;
178  }
179
180}
