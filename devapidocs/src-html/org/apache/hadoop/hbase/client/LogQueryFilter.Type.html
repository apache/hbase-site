020package org.apache.hadoop.hbase.client;
021
022import org.apache.commons.lang3.builder.EqualsBuilder;
023import org.apache.commons.lang3.builder.HashCodeBuilder;
024import org.apache.commons.lang3.builder.ToStringBuilder;
025import org.apache.yetus.audience.InterfaceAudience;
026
027/**
028 * Slow/Large Log Query Filter with all filter and limit parameters
029 * Used by Admin API: getSlowLogResponses
030 */
031@InterfaceAudience.Private
032public class LogQueryFilter {
033
034  private String regionName;
035  private String clientAddress;
036  private String tableName;
037  private String userName;
038  private int limit = 10;
039  private Type type = Type.SLOW_LOG;
040
041  public enum Type {
042    SLOW_LOG,
043    LARGE_LOG
044  }
045
046  public String getRegionName() {
047    return regionName;
048  }
049
050  public void setRegionName(String regionName) {
051    this.regionName = regionName;
052  }
053
054  public String getClientAddress() {
055    return clientAddress;
056  }
057
058  public void setClientAddress(String clientAddress) {
059    this.clientAddress = clientAddress;
060  }
061
062  public String getTableName() {
063    return tableName;
064  }
065
066  public void setTableName(String tableName) {
067    this.tableName = tableName;
068  }
069
070  public String getUserName() {
071    return userName;
072  }
073
074  public void setUserName(String userName) {
075    this.userName = userName;
076  }
077
078  public int getLimit() {
079    return limit;
080  }
081
082  public void setLimit(int limit) {
083    this.limit = limit;
084  }
085
086  public Type getType() {
087    return type;
088  }
089
090  public void setType(Type type) {
091    this.type = type;
092  }
093
094  @Override
095  public boolean equals(Object o) {
096    if (this == o) {
097      return true;
098    }
099
100    if (o == null || getClass() != o.getClass()) {
101      return false;
102    }
103
104    LogQueryFilter that = (LogQueryFilter) o;
105
106    return new EqualsBuilder()
107      .append(limit, that.limit)
108      .append(regionName, that.regionName)
109      .append(clientAddress, that.clientAddress)
110      .append(tableName, that.tableName)
111      .append(userName, that.userName)
112      .append(type, that.type)
113      .isEquals();
114  }
115
116  @Override
117  public int hashCode() {
118    return new HashCodeBuilder(17, 37)
119      .append(regionName)
120      .append(clientAddress)
121      .append(tableName)
122      .append(userName)
123      .append(limit)
124      .append(type)
125      .toHashCode();
126  }
127
128  @Override
129  public String toString() {
130    return new ToStringBuilder(this)
131      .append("regionName", regionName)
132      .append("clientAddress", clientAddress)
133      .append("tableName", tableName)
134      .append("userName", userName)
135      .append("limit", limit)
136      .append("type", type)
137      .toString();
138  }
139}
