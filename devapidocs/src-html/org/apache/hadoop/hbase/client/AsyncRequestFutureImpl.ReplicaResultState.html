061/**
062 * The context, and return value, for a single submit/submitAll call.
063 * Note on how this class (one AP submit) works. Initially, all requests are split into groups
064 * by server; request is sent to each server in parallel; the RPC calls are not async so a
065 * thread per server is used. Every time some actions fail, regions/locations might have
066 * changed, so we re-group them by server and region again and send these groups in parallel
067 * too. The result, in case of retries, is a "tree" of threads, with parent exiting after
068 * scheduling children. This is why lots of code doesn't require any synchronization.
069 */
070@InterfaceAudience.Private
071class AsyncRequestFutureImpl<CResult> implements AsyncRequestFuture {
072
073  private static final Logger LOG = LoggerFactory.getLogger(AsyncRequestFutureImpl.class);

077  /**
078   * Runnable (that can be submitted to thread pool) that waits for when it's time
079   * to issue replica calls, finds region replicas, groups the requests by replica and
080   * issues the calls (on separate threads, via sendMultiAction).
081   * This is done on a separate thread because we don't want to wait on user thread for
082   * our asynchronous call, and usually we have to wait before making replica calls.
083   */
084  private final class ReplicaCallIssuingRunnable implements Runnable {
085    private final long startTime;
086    private final List<Action> initialActions;
087
088    public ReplicaCallIssuingRunnable(List<Action> initialActions, long startTime) {
089      this.initialActions = initialActions;
090      this.startTime = startTime;
091    }
092
093    @Override
094    public void run() {
095      boolean done = false;
096      if (asyncProcess.primaryCallTimeoutMicroseconds > 0) {
097        try {
098          done = waitUntilDone(startTime * 1000L + asyncProcess.primaryCallTimeoutMicroseconds);
099        } catch (InterruptedException ex) {
100          LOG.error("Replica thread was interrupted - no replica calls: " + ex.getMessage());
101          return;
102        }
103      }
104      if (done) return; // Done within primary timeout
105      Map<ServerName, MultiAction> actionsByServer = new HashMap<>();
106      List<Action> unknownLocActions = new ArrayList<>();
107      if (replicaGetIndices == null) {
108        for (int i = 0; i < results.length; ++i) {
109          addReplicaActions(i, actionsByServer, unknownLocActions);
110        }
111      } else {
112        for (int replicaGetIndice : replicaGetIndices) {
113          addReplicaActions(replicaGetIndice, actionsByServer, unknownLocActions);
114        }
115      }
116      if (!actionsByServer.isEmpty()) {
117        sendMultiAction(actionsByServer, 1, null, unknownLocActions.isEmpty());
118      }
119      if (!unknownLocActions.isEmpty()) {
120        actionsByServer = new HashMap<>();
121        for (Action action : unknownLocActions) {
122          addReplicaActionsAgain(action, actionsByServer);
123        }
124        // Some actions may have completely failed, they are handled inside addAgain.
125        if (!actionsByServer.isEmpty()) {
126          sendMultiAction(actionsByServer, 1, null, true);
127        }
128      }
129    }
130
131    /**
132     * Add replica actions to action map by server.
133     * @param index Index of the original action.
134     * @param actionsByServer The map by server to add it to.
135     */
136    private void addReplicaActions(int index, Map<ServerName, MultiAction> actionsByServer,
137                                   List<Action> unknownReplicaActions) {
138      if (results[index] != null) return; // opportunistic. Never goes from non-null to null.
139      Action action = initialActions.get(index);
140      RegionLocations loc = findAllLocationsOrFail(action, true);
141      if (loc == null) return;
142      HRegionLocation[] locs = loc.getRegionLocations();
143      if (locs.length == 1) {
144        LOG.warn("No replicas found for " + action.getAction());
145        return;
146      }
147      synchronized (replicaResultLock) {
148        // Don't run replica calls if the original has finished. We could do it e.g. if
149        // original has already failed before first replica call (unlikely given retries),
150        // but that would require additional synchronization w.r.t. returning to caller.
151        if (results[index] != null) return;
152        // We set the number of calls here. After that any path must call setResult/setError.
153        // True even for replicas that are not found - if we refuse to send we MUST set error.
154        updateResult(index, new ReplicaResultState(locs.length));
155      }
156      for (int i = 1; i < locs.length; ++i) {
157        Action replicaAction = new Action(action, i);
158        if (locs[i] != null) {
159          asyncProcess.addAction(locs[i].getServerName(), locs[i].getRegionInfo().getRegionName(),
160              replicaAction, actionsByServer, nonceGroup);
161        } else {
162          unknownReplicaActions.add(replicaAction);
163        }
164      }
165    }
166
167    private void addReplicaActionsAgain(
168        Action action, Map<ServerName, MultiAction> actionsByServer) {
169      if (action.getReplicaId() == RegionReplicaUtil.DEFAULT_REPLICA_ID) {
170        throw new AssertionError("Cannot have default replica here");
171      }
172      HRegionLocation loc = getReplicaLocationOrFail(action);
173      if (loc == null) return;
174      asyncProcess.addAction(loc.getServerName(), loc.getRegionInfo().getRegionName(),
175          action, actionsByServer, nonceGroup);
176    }
177  }

179  /**
180   * Runnable (that can be submitted to thread pool) that submits MultiAction to a
181   * single server. The server call is synchronous, therefore we do it on a thread pool.
182   */
183  @VisibleForTesting
184  final class SingleServerRequestRunnable implements Runnable {
185    private final MultiAction multiAction;
186    private final int numAttempt;
187    private final ServerName server;
188    private final Set<CancellableRegionServerCallable> callsInProgress;
189    @VisibleForTesting
190    SingleServerRequestRunnable(
191        MultiAction multiAction, int numAttempt, ServerName server,
192        Set<CancellableRegionServerCallable> callsInProgress) {
193      this.multiAction = multiAction;
194      this.numAttempt = numAttempt;
195      this.server = server;
196      this.callsInProgress = callsInProgress;
197    }
198
199    @Override
200    public void run() {
201      AbstractResponse res = null;
202      CancellableRegionServerCallable callable = currentCallable;
203      try {
204        // setup the callable based on the actions, if we don't have one already from the request
205        if (callable == null) {
206          callable = createCallable(server, tableName, multiAction);
207        }
208        RpcRetryingCaller<AbstractResponse> caller = asyncProcess.createCaller(callable,rpcTimeout);
209        try {
210          if (callsInProgress != null) {
211            callsInProgress.add(callable);
212          }
213          res = caller.callWithoutRetries(callable, operationTimeout);
214          if (res == null) {
215            // Cancelled
216            return;
217          }
218        } catch (IOException e) {
219          // The service itself failed . It may be an error coming from the communication
220          //   layer, but, as well, a functional error raised by the server.
221          receiveGlobalFailure(multiAction, server, numAttempt, e);
222          return;
223        } catch (Throwable t) {
224          // This should not happen. Let's log & retry anyway.
225          LOG.error("#" + asyncProcess.id + ", Caught throwable while calling. This is unexpected." +
226              " Retrying. Server is " + server + ", tableName=" + tableName, t);
227          receiveGlobalFailure(multiAction, server, numAttempt, t);
228          return;
229        }
230        if (res.type() == AbstractResponse.ResponseType.MULTI) {
231          // Normal case: we received an answer from the server, and it's not an exception.
232          receiveMultiAction(multiAction, server, (MultiResponse) res, numAttempt);
233        } else {
234          if (results != null) {
235            SingleResponse singleResponse = (SingleResponse) res;
236            updateResult(0, singleResponse.getEntry());
237          }
238          decActionCounter(1);
239        }
240      } catch (Throwable t) {
241        // Something really bad happened. We are on the send thread that will now die.
242        LOG.error("Internal AsyncProcess #" + asyncProcess.id + " error for "
243            + tableName + " processing for " + server, t);
244        throw new RuntimeException(t);
245      } finally {
246        asyncProcess.decTaskCounters(multiAction.getRegions(), server);
247        if (callsInProgress != null && callable != null && res != null) {
248          callsInProgress.remove(callable);
249        }
250      }
251    }
252  }

302  /** Sync point for calls to multiple replicas for the same user request (Get).
303   * Created and put in the results array (we assume replica calls require results) when
304   * the replica calls are launched. See results for details of this process.
305   * POJO, all fields are public. To modify them, the object itself is locked. */
306  private static class ReplicaResultState {
307    public ReplicaResultState(int callCount) {
308      this.callCount = callCount;
309    }
310
311    /** Number of calls outstanding, or 0 if a call succeeded (even with others outstanding). */
312    int callCount;
313    /** Errors for which it is not decided whether we will report them to user. If one of the
314     * calls succeeds, we will discard the errors that may have happened in the other calls. */
315    BatchErrors replicaErrors = null;
316
317    @Override
318    public String toString() {
319      return "[call count " + callCount + "; errors " + replicaErrors + "]";
320    }

323  public AsyncRequestFutureImpl(AsyncProcessTask task, List<Action> actions,
324      long nonceGroup, AsyncProcess asyncProcess) {
325    this.pool = task.getPool();
326    this.callback = task.getCallback();
327    this.nonceGroup = nonceGroup;
328    this.tableName = task.getTableName();
329    this.actionsInProgress.set(actions.size());
330    if (task.getResults() == null) {
331      results = task.getNeedResults() ? new Object[actions.size()] : null;
332    } else {
333      if (task.getResults().length != actions.size()) {
334        throw new AssertionError("results.length");
335      }
336      this.results = task.getResults();
337      for (int i = 0; i != this.results.length; ++i) {
338        results[i] = null;
339      }
340    }
341    List<Integer> replicaGetIndices = null;
342    boolean hasAnyReplicaGets = false;
343    if (results != null) {
344      // Check to see if any requests might require replica calls.
345      // We expect that many requests will consist of all or no multi-replica gets; in such
346      // cases we would just use a boolean (hasAnyReplicaGets). If there's a mix, we will
347      // store the list of action indexes for which replica gets are possible, and set
348      // hasAnyReplicaGets to true.
349      boolean hasAnyNonReplicaReqs = false;
350      int posInList = 0;
351      for (Action action : actions) {
352        boolean isReplicaGet = AsyncProcess.isReplicaGet(action.getAction());
353        if (isReplicaGet) {
354          hasAnyReplicaGets = true;
355          if (hasAnyNonReplicaReqs) { // Mixed case
356            if (replicaGetIndices == null) {
357              replicaGetIndices = new ArrayList<>(actions.size() - 1);
358            }
359            replicaGetIndices.add(posInList);
360          }
361        } else if (!hasAnyNonReplicaReqs) {
362          // The first non-multi-replica request in the action list.
363          hasAnyNonReplicaReqs = true;
364          if (posInList > 0) {
365            // Add all the previous requests to the index lists. We know they are all
366            // replica-gets because this is the first non-multi-replica request in the list.
367            replicaGetIndices = new ArrayList<>(actions.size() - 1);
368            for (int i = 0; i < posInList; ++i) {
369              replicaGetIndices.add(i);
370            }
371          }
372        }
373        ++posInList;
374      }
375    }
376    this.hasAnyReplicaGets = hasAnyReplicaGets;
377    if (replicaGetIndices != null) {
378      this.replicaGetIndices = new int[replicaGetIndices.size()];
379      int i = 0;
380      for (Integer el : replicaGetIndices) {
381        this.replicaGetIndices[i++] = el;
382      }
383    } else {
384      this.replicaGetIndices = null;
385    }
386    this.callsInProgress = !hasAnyReplicaGets ? null :
387        Collections.newSetFromMap(
388            new ConcurrentHashMap<CancellableRegionServerCallable, Boolean>());
389    this.asyncProcess = asyncProcess;
390    this.errorsByServer = createServerErrorTracker();
391    this.errors = new BatchErrors();
392    this.operationTimeout = task.getOperationTimeout();
393    this.rpcTimeout = task.getRpcTimeout();
394    this.currentCallable = task.getCallable();
395    if (task.getCallable() == null) {
396      tracker = new RetryingTimeTracker().start();
397    }
398  }

411  /**
412   * Group a list of actions per region servers, and send them.
413   *
414   * @param currentActions - the list of row to submit
415   * @param numAttempt - the current numAttempt (first attempt is 1)
416   */
417  void groupAndSendMultiAction(List<Action> currentActions, int numAttempt) {
418    Map<ServerName, MultiAction> actionsByServer = new HashMap<>();
419
420    boolean isReplica = false;
421    List<Action> unknownReplicaActions = null;
422    for (Action action : currentActions) {
423      RegionLocations locs = findAllLocationsOrFail(action, true);
424      if (locs == null) continue;
425      boolean isReplicaAction = !RegionReplicaUtil.isDefaultReplica(action.getReplicaId());
426      if (isReplica && !isReplicaAction) {
427        // This is the property of the current implementation, not a requirement.
428        throw new AssertionError("Replica and non-replica actions in the same retry");
429      }
430      isReplica = isReplicaAction;
431      HRegionLocation loc = locs.getRegionLocation(action.getReplicaId());
432      if (loc == null || loc.getServerName() == null) {
433        if (isReplica) {
434          if (unknownReplicaActions == null) {
435            unknownReplicaActions = new ArrayList<>(1);
436          }
437          unknownReplicaActions.add(action);
438        } else {
439          // TODO: relies on primary location always being fetched
440          manageLocationError(action, null);
441        }
442      } else {
443        byte[] regionName = loc.getRegionInfo().getRegionName();
444        AsyncProcess.addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
445      }
446    }
447    boolean doStartReplica = (numAttempt == 1 && !isReplica && hasAnyReplicaGets);
448    boolean hasUnknown = unknownReplicaActions != null && !unknownReplicaActions.isEmpty();
449
450    if (!actionsByServer.isEmpty()) {
451      // If this is a first attempt to group and send, no replicas, we need replica thread.
452      sendMultiAction(actionsByServer, numAttempt, (doStartReplica && !hasUnknown)
453          ? currentActions : null, numAttempt > 1 && !hasUnknown);
454    }
455
456    if (hasUnknown) {
457      actionsByServer = new HashMap<>();
458      for (Action action : unknownReplicaActions) {
459        HRegionLocation loc = getReplicaLocationOrFail(action);
460        if (loc == null) continue;
461        byte[] regionName = loc.getRegionInfo().getRegionName();
462        AsyncProcess.addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
463      }
464      if (!actionsByServer.isEmpty()) {
465        sendMultiAction(
466            actionsByServer, numAttempt, doStartReplica ? currentActions : null, true);
467      }
468    }
469  }

471  private HRegionLocation getReplicaLocationOrFail(Action action) {
472    // We are going to try get location once again. For each action, we'll do it once
473    // from cache, because the previous calls in the loop might populate it.
474    int replicaId = action.getReplicaId();
475    RegionLocations locs = findAllLocationsOrFail(action, true);
476    if (locs == null) return null; // manageError already called
477    HRegionLocation loc = locs.getRegionLocation(replicaId);
478    if (loc == null || loc.getServerName() == null) {
479      locs = findAllLocationsOrFail(action, false);
480      if (locs == null) return null; // manageError already called
481      loc = locs.getRegionLocation(replicaId);
482    }
483    if (loc == null || loc.getServerName() == null) {
484      manageLocationError(action, null);
485      return null;
486    }
487    return loc;
488  }
489
490  private void manageLocationError(Action action, Exception ex) {
491    String msg = "Cannot get replica " + action.getReplicaId()
492        + " location for " + action.getAction();
493    LOG.error(msg);
494    if (ex == null) {
495      ex = new IOException(msg);
496    }
497    manageError(action.getOriginalIndex(), action.getAction(),
498        Retry.NO_LOCATION_PROBLEM, ex, null);
499  }
500
501  private RegionLocations findAllLocationsOrFail(Action action, boolean useCache) {
502    if (action.getAction() == null) throw new IllegalArgumentException("#" + asyncProcess.id +
503        ", row cannot be null");
504    RegionLocations loc = null;
505    try {
506      loc = asyncProcess.connection.locateRegion(
507          tableName, action.getAction().getRow(), useCache, true, action.getReplicaId());
508    } catch (IOException ex) {
509      manageLocationError(action, ex);
510    }
511    return loc;
512  }

514  /**
515   * Send a multi action structure to the servers, after a delay depending on the attempt
516   * number. Asynchronous.
517   *
518   * @param actionsByServer the actions structured by regions
519   * @param numAttempt the attempt number.
520   * @param actionsForReplicaThread original actions for replica thread; null on non-first call.
521   */
522  void sendMultiAction(Map<ServerName, MultiAction> actionsByServer,
523                               int numAttempt, List<Action> actionsForReplicaThread, boolean reuseThread) {
524    // Run the last item on the same thread if we are already on a send thread.
525    // We hope most of the time it will be the only item, so we can cut down on threads.
526    int actionsRemaining = actionsByServer.size();
527    // This iteration is by server (the HRegionLocation comparator is by server portion only).
528    for (Map.Entry<ServerName, MultiAction> e : actionsByServer.entrySet()) {
529      ServerName server = e.getKey();
530      MultiAction multiAction = e.getValue();
531      Collection<? extends Runnable> runnables = getNewMultiActionRunnable(server, multiAction,
532          numAttempt);
533      // make sure we correctly count the number of runnables before we try to reuse the send
534      // thread, in case we had to split the request into different runnables because of backoff
535      if (runnables.size() > actionsRemaining) {
536        actionsRemaining = runnables.size();
537      }
538
539      // run all the runnables
540      // HBASE-17475: Do not reuse the thread after stack reach a certain depth to prevent stack overflow
541      // for now, we use HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER to control the depth
542      for (Runnable runnable : runnables) {
543        if ((--actionsRemaining == 0) && reuseThread
544            && numAttempt % HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER != 0) {
545          runnable.run();
546        } else {
547          try {
548            pool.submit(runnable);
549          } catch (Throwable t) {
550            if (t instanceof RejectedExecutionException) {
551              // This should never happen. But as the pool is provided by the end user,
552              // let's secure this a little.
553              LOG.warn("#" + asyncProcess.id + ", the task was rejected by the pool. This is unexpected." +
554                  " Server is " + server.getServerName(), t);
555            } else {
556              // see #HBASE-14359 for more details
557              LOG.warn("Caught unexpected exception/error: ", t);
558            }
559            asyncProcess.decTaskCounters(multiAction.getRegions(), server);
560            // We're likely to fail again, but this will increment the attempt counter,
561            // so it will finish.
562            receiveGlobalFailure(multiAction, server, numAttempt, t);
563          }
564        }
565      }
566    }
567
568    if (actionsForReplicaThread != null) {
569      startWaitingForReplicaCalls(actionsForReplicaThread);
570    }
571  }

573  private Collection<? extends Runnable> getNewMultiActionRunnable(ServerName server,
574                                                                   MultiAction multiAction,
575                                                                   int numAttempt) {
576    // no stats to manage, just do the standard action
577    if (asyncProcess.connection.getStatisticsTracker() == null) {
578      if (asyncProcess.connection.getConnectionMetrics() != null) {
579        asyncProcess.connection.getConnectionMetrics().incrNormalRunners();
580      }
581      asyncProcess.incTaskCounters(multiAction.getRegions(), server);
582      SingleServerRequestRunnable runnable = createSingleServerRequest(
583              multiAction, numAttempt, server, callsInProgress);
584      Tracer tracer = Tracer.curThreadTracer();
585
586      if (tracer == null) {
587        return Collections.singletonList(runnable);
588      } else {
589        return Collections.singletonList(tracer.wrap(runnable, "AsyncProcess.sendMultiAction"));
590      }
591    }
592
593    // group the actions by the amount of delay
594    Map<Long, DelayingRunner> actions = new HashMap<>(multiAction.size());
595
596    // split up the actions
597    for (Map.Entry<byte[], List<Action>> e : multiAction.actions.entrySet()) {
598      Long backoff = getBackoff(server, e.getKey());
599      DelayingRunner runner = actions.get(backoff);
600      if (runner == null) {
601        actions.put(backoff, new DelayingRunner(backoff, e));
602      } else {
603        runner.add(e);
604      }
605    }
606
607    List<Runnable> toReturn = new ArrayList<>(actions.size());
608    for (DelayingRunner runner : actions.values()) {
609      asyncProcess.incTaskCounters(runner.getActions().getRegions(), server);
610      String traceText = "AsyncProcess.sendMultiAction";
611      Runnable runnable = createSingleServerRequest(runner.getActions(), numAttempt, server, callsInProgress);
612      // use a delay runner only if we need to sleep for some time
613      if (runner.getSleepTime() > 0) {
614        runner.setRunner(runnable);
615        traceText = "AsyncProcess.clientBackoff.sendMultiAction";
616        runnable = runner;
617        if (asyncProcess.connection.getConnectionMetrics() != null) {
618          asyncProcess.connection.getConnectionMetrics().incrDelayRunners();
619          asyncProcess.connection.getConnectionMetrics().updateDelayInterval(runner.getSleepTime());
620        }
621      } else {
622        if (asyncProcess.connection.getConnectionMetrics() != null) {
623          asyncProcess.connection.getConnectionMetrics().incrNormalRunners();
624        }
625      }
626      runnable = TraceUtil.wrap(runnable, traceText);
627      toReturn.add(runnable);
628
629    }
630    return toReturn;
631  }

667  /**
668   * Check that we can retry acts accordingly: logs, set the error status.
669   *
670   * @param originalIndex the position in the list sent
671   * @param row           the row
672   * @param canRetry      if false, we won't retry whatever the settings.
673   * @param throwable     the throwable, if any (can be null)
674   * @param server        the location, if any (can be null)
675   * @return true if the action can be retried, false otherwise.
676   */
677  Retry manageError(int originalIndex, Row row, Retry canRetry,
678                                        Throwable throwable, ServerName server) {
679    if (canRetry == Retry.YES
680        && throwable != null && throwable instanceof DoNotRetryIOException) {
681      canRetry = Retry.NO_NOT_RETRIABLE;
682    }
683
684    if (canRetry != Retry.YES) {
685      // Batch.Callback<Res> was not called on failure in 0.94. We keep this.
686      setError(originalIndex, row, throwable, server);
687    } else if (isActionComplete(originalIndex, row)) {
688      canRetry = Retry.NO_OTHER_SUCCEEDED;
689    }
690    return canRetry;
691  }

693  /**
694   * Resubmit all the actions from this multiaction after a failure.
695   *
696   * @param rsActions  the actions still to do from the initial list
697   * @param server   the destination
698   * @param numAttempt the number of attempts so far
699   * @param t the throwable (if any) that caused the resubmit
700   */
701  private void receiveGlobalFailure(
702      MultiAction rsActions, ServerName server, int numAttempt, Throwable t) {
703    errorsByServer.reportServerError(server);
704    Retry canRetry = errorsByServer.canTryMore(numAttempt)
705        ? Retry.YES : Retry.NO_RETRIES_EXHAUSTED;
706
707    cleanServerCache(server, t);
708    int failed = 0;
709    int stopped = 0;
710    List<Action> toReplay = new ArrayList<>();
711    for (Map.Entry<byte[], List<Action>> e : rsActions.actions.entrySet()) {
712      byte[] regionName = e.getKey();
713      byte[] row = e.getValue().get(0).getAction().getRow();
714      // Do not use the exception for updating cache because it might be coming from
715      // any of the regions in the MultiAction.
716      updateCachedLocations(server, regionName, row,
717        ClientExceptionsUtil.isMetaClearingException(t) ? null : t);
718      for (Action action : e.getValue()) {
719        Retry retry = manageError(
720            action.getOriginalIndex(), action.getAction(), canRetry, t, server);
721        if (retry == Retry.YES) {
722          toReplay.add(action);
723        } else if (retry == Retry.NO_OTHER_SUCCEEDED) {
724          ++stopped;
725        } else {
726          ++failed;
727        }
728      }
729    }
730
731    if (toReplay.isEmpty()) {
732      logNoResubmit(server, numAttempt, rsActions.size(), t, failed, stopped);
733    } else {
734      resubmit(server, toReplay, numAttempt, rsActions.size(), t);
735    }
736  }

738  /**
739   * Log as much info as possible, and, if there is something to replay,
740   * submit it again after a back off sleep.
741   */
742  private void resubmit(ServerName oldServer, List<Action> toReplay,
743                        int numAttempt, int failureCount, Throwable throwable) {
744    // We have something to replay. We're going to sleep a little before.
745
746    // We have two contradicting needs here:
747    //  1) We want to get the new location after having slept, as it may change.
748    //  2) We want to take into account the location when calculating the sleep time.
749    //  3) If all this is just because the response needed to be chunked try again FAST.
750    // It should be possible to have some heuristics to take the right decision. Short term,
751    //  we go for one.
752    boolean retryImmediately = throwable instanceof RetryImmediatelyException;
753    int nextAttemptNumber = retryImmediately ? numAttempt : numAttempt + 1;
754    long backOffTime;
755    if (retryImmediately) {
756      backOffTime = 0;
757    } else if (throwable instanceof CallQueueTooBigException) {
758      // Give a special check on CQTBE, see #HBASE-17114
759      backOffTime = errorsByServer.calculateBackoffTime(oldServer, asyncProcess.pauseForCQTBE);
760    } else {
761      backOffTime = errorsByServer.calculateBackoffTime(oldServer, asyncProcess.pause);
762    }
763    if (numAttempt > asyncProcess.startLogErrorsCnt) {
764      // We use this value to have some logs when we have multiple failures, but not too many
765      //  logs, as errors are to be expected when a region moves, splits and so on
766      LOG.info(createLog(numAttempt, failureCount, toReplay.size(),
767          oldServer, throwable, backOffTime, true, null, -1, -1));
768    }
769
770    try {
771      if (backOffTime > 0) {
772        Thread.sleep(backOffTime);
773      }
774    } catch (InterruptedException e) {
775      LOG.warn("#" + asyncProcess.id + ", not sent: " + toReplay.size() + " operations, " + oldServer, e);
776      Thread.currentThread().interrupt();
777      return;
778    }
779
780    groupAndSendMultiAction(toReplay, nextAttemptNumber);
781  }

783  private void logNoResubmit(ServerName oldServer, int numAttempt,
784                             int failureCount, Throwable throwable, int failed, int stopped) {
785    if (failureCount != 0 || numAttempt > asyncProcess.startLogErrorsCnt + 1) {
786      String timeStr = new Date(errorsByServer.getStartTrackingTime()).toString();
787      String logMessage = createLog(numAttempt, failureCount, 0, oldServer,
788          throwable, -1, false, timeStr, failed, stopped);
789      if (failed != 0) {
790        // Only log final failures as warning
791        LOG.warn(logMessage);
792      } else {
793        LOG.info(logMessage);
794      }
795    }
796  }

798  /**
799   * Called when we receive the result of a server query.
800   *
801   * @param multiAction    - the multiAction we sent
802   * @param server       - the location. It's used as a server name.
803   * @param responses      - the response, if any
804   * @param numAttempt     - the attempt
805   */
806  private void receiveMultiAction(MultiAction multiAction,
807                                  ServerName server, MultiResponse responses, int numAttempt) {
808    assert responses != null;
809
810    Map<byte[], MultiResponse.RegionResult> results = responses.getResults();
811    updateStats(server, results);
812
813    // Success or partial success
814    // Analyze detailed results. We can still have individual failures to be redo.
815    // two specific throwables are managed:
816    //  - DoNotRetryIOException: we continue to retry for other actions
817    //  - RegionMovedException: we update the cache with the new region location
818
819    List<Action> toReplay = new ArrayList<>();
820    Throwable lastException = null;
821    int failureCount = 0;
822    int failed = 0;
823    int stopped = 0;
824    Retry retry = null;
825    // Go by original action.
826    for (Map.Entry<byte[], List<Action>> regionEntry : multiAction.actions.entrySet()) {
827      byte[] regionName = regionEntry.getKey();
828
829      Throwable regionException = responses.getExceptions().get(regionName);
830      cleanServerCache(server, regionException);
831
832      Map<Integer, Object> regionResults =
833        results.containsKey(regionName) ? results.get(regionName).result : Collections.emptyMap();
834      boolean regionFailureRegistered = false;
835      for (Action sentAction : regionEntry.getValue()) {
836        Object result = regionResults.get(sentAction.getOriginalIndex());
837        if (result == null) {
838          if (regionException == null) {
839            LOG.error("Server sent us neither results nor exceptions for "
840              + Bytes.toStringBinary(regionName)
841              + ", numAttempt:" + numAttempt);
842            regionException = new RuntimeException("Invalid response");
843          }
844          // If the row operation encounters the region-lever error, the exception of action may be
845          // null.
846          result = regionException;
847        }
848        // Failure: retry if it's make sense else update the errors lists
849        if (result instanceof Throwable) {
850          Throwable actionException = (Throwable) result;
851          Row row = sentAction.getAction();
852          lastException = regionException != null ? regionException
853            : ClientExceptionsUtil.findException(actionException);
854          // Register corresponding failures once per server/once per region.
855          if (!regionFailureRegistered) {
856            regionFailureRegistered = true;
857            updateCachedLocations(server, regionName, row.getRow(), actionException);
858          }
859          if (retry == null) {
860            errorsByServer.reportServerError(server);
861            // We determine canRetry only once for all calls, after reporting server failure.
862            retry = errorsByServer.canTryMore(numAttempt) ?
863              Retry.YES : Retry.NO_RETRIES_EXHAUSTED;
864          }
865          ++failureCount;
866          switch (manageError(sentAction.getOriginalIndex(), row, retry, actionException,
867            server)) {
868            case YES:
869              toReplay.add(sentAction);
870              break;
871            case NO_OTHER_SUCCEEDED:
872              ++stopped;
873              break;
874            default:
875              ++failed;
876              break;
877          }
878        } else {
879          invokeCallBack(regionName, sentAction.getAction().getRow(), (CResult) result);
880          setResult(sentAction, result);
881        }
882      }
883    }
884    if (toReplay.isEmpty()) {
885      logNoResubmit(server, numAttempt, failureCount, lastException, failed, stopped);
886    } else {
887      resubmit(server, toReplay, numAttempt, failureCount, lastException);
888    }
889  }

985  /**
986   * Sets the non-error result from a particular action.
987   * @param action Action (request) that the server responded to.
988   * @param result The result.
989   */
990  private void setResult(Action action, Object result) {
991    if (result == null) {
992      throw new RuntimeException("Result cannot be null");
993    }
994    ReplicaResultState state = null;
995    boolean isStale = !RegionReplicaUtil.isDefaultReplica(action.getReplicaId());
996    int index = action.getOriginalIndex();
997    if (results == null) {
998      decActionCounter(index);
999      return; // Simple case, no replica requests.
1000    }
1001    state = trySetResultSimple(index, action.getAction(), false, result, null, isStale);
1002    if (state == null) {
1003      return; // Simple case, no replica requests.
1004    }
1005    // At this point we know that state is set to replica tracking class.
1006    // It could be that someone else is also looking at it; however, we know there can
1007    // only be one state object, and only one thread can set callCount to 0. Other threads
1008    // will either see state with callCount 0 after locking it; or will not see state at all
1009    // we will replace it with the result.
1010    synchronized (state) {
1011      if (state.callCount == 0) {
1012        return; // someone already set the result
1013      }
1014      state.callCount = 0;
1015    }
1016    synchronized (replicaResultLock) {
1017      if (results[index] != state) {
1018        throw new AssertionError("We set the callCount but someone else replaced the result");
1019      }
1020      updateResult(index, result);
1021    }
1022
1023    decActionCounter(index);
1024  }

1026  /**
1027   * Sets the error from a particular action.
1028   * @param index Original action index.
1029   * @param row Original request.
1030   * @param throwable The resulting error.
1031   * @param server The source server.
1032   */
1033  private void setError(int index, Row row, Throwable throwable, ServerName server) {
1034    ReplicaResultState state = null;
1035    if (results == null) {
1036      // Note that we currently cannot have replica requests with null results. So it shouldn't
1037      // happen that multiple replica calls will call dAC for same actions with results == null.
1038      // Only one call per action should be present in this case.
1039      errors.add(throwable, row, server);
1040      decActionCounter(index);
1041      return; // Simple case, no replica requests.
1042    }
1043    state = trySetResultSimple(index, row, true, throwable, server, false);
1044    if (state == null) {
1045      return; // Simple case, no replica requests.
1046    }
1047    BatchErrors target = null; // Error will be added to final errors, or temp replica errors.
1048    boolean isActionDone = false;
1049    synchronized (state) {
1050      switch (state.callCount) {
1051        case 0: return; // someone already set the result
1052        case 1: { // All calls failed, we are the last error.
1053          target = errors;
1054          isActionDone = true;
1055          break;
1056        }
1057        default: {
1058          assert state.callCount > 1;
1059          if (state.replicaErrors == null) {
1060            state.replicaErrors = new BatchErrors();
1061          }
1062          target = state.replicaErrors;
1063          break;
1064        }
1065      }
1066      --state.callCount;
1067    }
1068    target.add(throwable, row, server);
1069    if (isActionDone) {
1070      if (state.replicaErrors != null) { // last call, no need to lock
1071        errors.merge(state.replicaErrors);
1072      }
1073      // See setResult for explanations.
1074      synchronized (replicaResultLock) {
1075        if (results[index] != state) {
1076          throw new AssertionError("We set the callCount but someone else replaced the result");
1077        }
1078        updateResult(index, throwable);
1079      }
1080      decActionCounter(index);
1081    }
1082  }

1084  /**
1085   * Checks if the action is complete; used on error to prevent needless retries.
1086   * Does not synchronize, assuming element index/field accesses are atomic.
1087   * This is an opportunistic optimization check, doesn't have to be strict.
1088   * @param index Original action index.
1089   * @param row Original request.
1090   */
1091  private boolean isActionComplete(int index, Row row) {
1092    if (!AsyncProcess.isReplicaGet(row)) return false;
1093    Object resObj = results[index];
1094    return (resObj != null) && (!(resObj instanceof ReplicaResultState)
1095        || ((ReplicaResultState)resObj).callCount == 0);
1096  }

1098  /**
1099   * Tries to set the result or error for a particular action as if there were no replica calls.
1100   * @return null if successful; replica state if there were in fact replica calls.
1101   */
1102  private ReplicaResultState trySetResultSimple(int index, Row row, boolean isError,
1103                                                Object result, ServerName server, boolean isFromReplica) {
1104    Object resObj = null;
1105    if (!AsyncProcess.isReplicaGet(row)) {
1106      if (isFromReplica) {
1107        throw new AssertionError("Unexpected stale result for " + row);
1108      }
1109      updateResult(index, result);
1110    } else {
1111      synchronized (replicaResultLock) {
1112        resObj = results[index];
1113        if (resObj == null) {
1114          if (isFromReplica) {
1115            throw new AssertionError("Unexpected stale result for " + row);
1116          }
1117          updateResult(index, result);
1118        }
1119      }
1120    }
1121
1122    ReplicaResultState rrs =
1123        (resObj instanceof ReplicaResultState) ? (ReplicaResultState)resObj : null;
1124    if (rrs == null && isError) {
1125      // The resObj is not replica state (null or already set).
1126      errors.add((Throwable)result, row, server);
1127    }
1128
1129    if (resObj == null) {
1130      // resObj is null - no replica calls were made.
1131      decActionCounter(index);
1132      return null;
1133    }
1134    return rrs;
1135  }

1137  private void decActionCounter(int index) {
1138    long actionsRemaining = actionsInProgress.decrementAndGet();
1139    if (actionsRemaining < 0) {
1140      String error = buildDetailedErrorMsg("Incorrect actions in progress", index);
1141      throw new AssertionError(error);
1142    } else if (actionsRemaining == 0) {
1143      synchronized (actionsInProgress) {
1144        actionsInProgress.notifyAll();
1145      }
1146    }
1147  }

1149  private String buildDetailedErrorMsg(String string, int index) {
1150    StringBuilder error = new StringBuilder(128);
1151    error.append(string).append("; called for ").append(index).append(", actionsInProgress ")
1152        .append(actionsInProgress.get()).append("; replica gets: ");
1153    if (replicaGetIndices != null) {
1154      for (int i = 0; i < replicaGetIndices.length; ++i) {
1155        error.append(replicaGetIndices[i]).append(", ");
1156      }
1157    } else {
1158      error.append(hasAnyReplicaGets ? "all" : "none");
1159    }
1160    error.append("; results ");
1161    if (results != null) {
1162      for (int i = 0; i < results.length; ++i) {
1163        Object o = results[i];
1164        error.append(((o == null) ? "null" : o.toString())).append(", ");
1165      }
1166    }
1167    return error.toString();
1168  }

1185  private boolean waitUntilDone(long cutoff) throws InterruptedException {
1186    boolean hasWait = cutoff != Long.MAX_VALUE;
1187    long lastLog = EnvironmentEdgeManager.currentTime();
1188    long currentInProgress;
1189    while (0 != (currentInProgress = actionsInProgress.get())) {
1190      long now = EnvironmentEdgeManager.currentTime();
1191      if (hasWait && (now * 1000L) > cutoff) {
1192        return false;
1193      }
1194      if (!hasWait) { // Only log if wait is infinite.
1195        if (now > lastLog + 10000) {
1196          lastLog = now;
1197          LOG.info("#" + asyncProcess.id + ", waiting for " + currentInProgress
1198              + "  actions to finish on table: " + tableName);
1199        }
1200      }
1201      synchronized (actionsInProgress) {
1202        if (actionsInProgress.get() == 0) break;
1203        if (!hasWait) {
1204          actionsInProgress.wait(10);
1205        } else {
1206          long waitMicroSecond = Math.min(100000L, (cutoff - now * 1000L));
1207          TimeUnit.MICROSECONDS.timedWait(actionsInProgress, waitMicroSecond);
1208        }
1209      }
1210    }
