037/**
038 * Factory to create an AsyncRpcRetryCaller.
039 */
040@InterfaceAudience.Private
041class AsyncRpcRetryingCallerFactory {
042
043  private final AsyncConnectionImpl conn;
044
045  private final HashedWheelTimer retryTimer;
046
047  public AsyncRpcRetryingCallerFactory(AsyncConnectionImpl conn, HashedWheelTimer retryTimer) {
048    this.conn = conn;
049    this.retryTimer = retryTimer;
050  }
051
052  private abstract class BuilderBase {
053
054    protected long pauseNs = conn.connConf.getPauseNs();
055
056    protected int maxAttempts = retries2Attempts(conn.connConf.getMaxRetries());
057
058    protected int startLogErrorsCnt = conn.connConf.getStartLogErrorsCnt();
059  }
060
061  public class SingleRequestCallerBuilder<T> extends BuilderBase {
062
063    private TableName tableName;
064
065    private byte[] row;
066
067    private AsyncSingleRequestRpcRetryingCaller.Callable<T> callable;
068
069    private long operationTimeoutNs = -1L;
070
071    private long rpcTimeoutNs = -1L;
072
073    private RegionLocateType locateType = RegionLocateType.CURRENT;
074
075    public SingleRequestCallerBuilder<T> table(TableName tableName) {
076      this.tableName = tableName;
077      return this;
078    }
079
080    public SingleRequestCallerBuilder<T> row(byte[] row) {
081      this.row = row;
082      return this;
083    }
084
085    public SingleRequestCallerBuilder<T> action(
086        AsyncSingleRequestRpcRetryingCaller.Callable<T> callable) {
087      this.callable = callable;
088      return this;
089    }
090
091    public SingleRequestCallerBuilder<T> operationTimeout(long operationTimeout, TimeUnit unit) {
092      this.operationTimeoutNs = unit.toNanos(operationTimeout);
093      return this;
094    }
095
096    public SingleRequestCallerBuilder<T> rpcTimeout(long rpcTimeout, TimeUnit unit) {
097      this.rpcTimeoutNs = unit.toNanos(rpcTimeout);
098      return this;
099    }
100
101    public SingleRequestCallerBuilder<T> locateType(RegionLocateType locateType) {
102      this.locateType = locateType;
103      return this;
104    }
105
106    public SingleRequestCallerBuilder<T> pause(long pause, TimeUnit unit) {
107      this.pauseNs = unit.toNanos(pause);
108      return this;
109    }
110
111    public SingleRequestCallerBuilder<T> maxAttempts(int maxAttempts) {
112      this.maxAttempts = maxAttempts;
113      return this;
114    }
115
116    public SingleRequestCallerBuilder<T> startLogErrorsCnt(int startLogErrorsCnt) {
117      this.startLogErrorsCnt = startLogErrorsCnt;
118      return this;
119    }
120
121    public AsyncSingleRequestRpcRetryingCaller<T> build() {
122      return new AsyncSingleRequestRpcRetryingCaller<>(retryTimer, conn,
123          checkNotNull(tableName, "tableName is null"), checkNotNull(row, "row is null"),
124          checkNotNull(locateType, "locateType is null"), checkNotNull(callable, "action is null"),
125          pauseNs, maxAttempts, operationTimeoutNs, rpcTimeoutNs, startLogErrorsCnt);
126    }
127
128    /**
129     * Shortcut for {@code build().call()}
130     */
131    public CompletableFuture<T> call() {
132      return build().call();
133    }
134  }
135
136  /**
137   * Create retry caller for single action, such as get, put, delete, etc.
138   */
139  public <T> SingleRequestCallerBuilder<T> single() {
140    return new SingleRequestCallerBuilder<>();
141  }

150
151    private RawScanResultConsumer consumer;
152
153    private ClientService.Interface stub;
154
155    private HRegionLocation loc;
156
157    private long scanTimeoutNs;
158
159    private long rpcTimeoutNs;
160
161    public ScanSingleRegionCallerBuilder id(long scannerId) {
162      this.scannerId = scannerId;
163      return this;
164    }
165
166    public ScanSingleRegionCallerBuilder setScan(Scan scan) {
167      this.scan = scan;
168      return this;
169    }
170
171    public ScanSingleRegionCallerBuilder resultCache(ScanResultCache resultCache) {
172      this.resultCache = resultCache;
173      return this;
174    }
175
176    public ScanSingleRegionCallerBuilder consumer(RawScanResultConsumer consumer) {
177      this.consumer = consumer;
178      return this;
179    }
180
181    public ScanSingleRegionCallerBuilder stub(ClientService.Interface stub) {
182      this.stub = stub;
183      return this;
184    }
185
186    public ScanSingleRegionCallerBuilder location(HRegionLocation loc) {
187      this.loc = loc;
188      return this;
189    }
190
191    public ScanSingleRegionCallerBuilder scanTimeout(long scanTimeout, TimeUnit unit) {
192      this.scanTimeoutNs = unit.toNanos(scanTimeout);
193      return this;
194    }
195
196    public ScanSingleRegionCallerBuilder rpcTimeout(long rpcTimeout, TimeUnit unit) {
197      this.rpcTimeoutNs = unit.toNanos(rpcTimeout);
198      return this;
199    }
200
201    public ScanSingleRegionCallerBuilder pause(long pause, TimeUnit unit) {
202      this.pauseNs = unit.toNanos(pause);
203      return this;
204    }
205
206    public ScanSingleRegionCallerBuilder maxAttempts(int maxAttempts) {
207      this.maxAttempts = maxAttempts;
208      return this;
209    }
210
211    public ScanSingleRegionCallerBuilder startLogErrorsCnt(int startLogErrorsCnt) {
212      this.startLogErrorsCnt = startLogErrorsCnt;
213      return this;
214    }
215
216    public AsyncScanSingleRegionRpcRetryingCaller build() {
217      checkArgument(scannerId >= 0, "invalid scannerId %d", scannerId);
218      return new AsyncScanSingleRegionRpcRetryingCaller(retryTimer, conn,
219          checkNotNull(scan, "scan is null"), scannerId,
220          checkNotNull(resultCache, "resultCache is null"),
221          checkNotNull(consumer, "consumer is null"), checkNotNull(stub, "stub is null"),
222          checkNotNull(loc, "location is null"), pauseNs, maxAttempts, scanTimeoutNs, rpcTimeoutNs,
223          startLogErrorsCnt);
224    }
225
226    /**
227     * Short cut for {@code build().start(HBaseRpcController, ScanResponse)}.
228     */
229    public CompletableFuture<Boolean> start(HBaseRpcController controller,
230        ScanResponse respWhenOpen) {
231      return build().start(controller, respWhenOpen);
232    }
233  }
