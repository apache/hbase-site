123  private enum ScanControllerState {
124    INITIALIZED, SUSPENDED, TERMINATED, DESTROYED
125  }
126
127  // Since suspend and terminate should only be called within onNext or onHeartbeat(see the comments
128  // of RawScanResultConsumer.onNext and onHeartbeat), we need to add some check to prevent invalid
129  // usage. We use two things to prevent invalid usage:
130  // 1. Record the thread that construct the ScanControllerImpl instance. We will throw an
131  // IllegalStateException if the caller thread is not this thread.
132  // 2. The ControllerState. The initial state is INITIALIZED, if you call suspend, the state will
133  // be transformed to SUSPENDED, and if you call terminate, the state will be transformed to
134  // TERMINATED. And when we are back from onNext or onHeartbeat in the onComplete method, we will
135  // call destroy to get the current state and set the state to DESTROYED. And when user calls
136  // suspend or terminate, we will check if the current state is INITIALIZED, if not we will throw
137  // an IllegalStateException. Notice that the DESTROYED state is necessary as you may not call
138  // suspend or terminate so the state will still be INITIALIZED when back from onNext or
139  // onHeartbeat. We need another state to replace the INITIALIZED state to prevent the controller
140  // to be used in the future.
141  // Notice that, the public methods of this class is supposed to be called by upper layer only, and
142  // package private methods can only be called within the implementation of
143  // AsyncScanSingleRegionRpcRetryingCaller.
144  private final class ScanControllerImpl implements AdvancedScanResultConsumer.ScanController {
145
146    // Make sure the methods are only called in this thread.
147    private final Thread callerThread;
148
149    private final Optional<Cursor> cursor;
150
151    // INITIALIZED -> SUSPENDED -> DESTROYED
152    // INITIALIZED -> TERMINATED -> DESTROYED
153    // INITIALIZED -> DESTROYED
154    // If the state is incorrect we will throw IllegalStateException.
155    private ScanControllerState state = ScanControllerState.INITIALIZED;
156
157    private ScanResumerImpl resumer;
158
159    public ScanControllerImpl(Optional<Cursor> cursor) {
160      this.callerThread = Thread.currentThread();
161      this.cursor = cursor;
162    }
163
164    private void preCheck() {
165      Preconditions.checkState(Thread.currentThread() == callerThread,
166        "The current thread is %s, expected thread is %s, " +
167            "you should not call this method outside onNext or onHeartbeat",
168        Thread.currentThread(), callerThread);
169      Preconditions.checkState(state.equals(ScanControllerState.INITIALIZED),
170        "Invalid Stopper state %s", state);
171    }
172
173    @Override
174    public ScanResumer suspend() {
175      preCheck();
176      state = ScanControllerState.SUSPENDED;
177      ScanResumerImpl resumer = new ScanResumerImpl();
178      this.resumer = resumer;
179      return resumer;
180    }
181
182    @Override
183    public void terminate() {
184      preCheck();
185      state = ScanControllerState.TERMINATED;
186    }
187
188    // return the current state, and set the state to DESTROYED.
189    ScanControllerState destroy() {
190      ScanControllerState state = this.state;
191      this.state = ScanControllerState.DESTROYED;
192      return state;
193    }
194
195    @Override
196    public Optional<Cursor> cursor() {
197        return cursor;
198    }
199  }

201  private enum ScanResumerState {
202    INITIALIZED, SUSPENDED, RESUMED
203  }
204
205  // The resume method is allowed to be called in another thread so here we also use the
206  // ResumerState to prevent race. The initial state is INITIALIZED, and in most cases, when back
207  // from onNext or onHeartbeat, we will call the prepare method to change the state to SUSPENDED,
208  // and when user calls resume method, we will change the state to RESUMED. But the resume method
209  // could be called in other thread, and in fact, user could just do this:
210  // controller.suspend().resume()
211  // This is strange but valid. This means the scan could be resumed before we call the prepare
212  // method to do the actual suspend work. So in the resume method, we will check if the state is
213  // INTIALIZED, if it is, then we will just set the state to RESUMED and return. And in prepare
214  // method, if the state is RESUMED already, we will just return an let the scan go on.
215  // Notice that, the public methods of this class is supposed to be called by upper layer only, and
216  // package private methods can only be called within the implementation of
217  // AsyncScanSingleRegionRpcRetryingCaller.
218  private final class ScanResumerImpl implements AdvancedScanResultConsumer.ScanResumer {
219
220    // INITIALIZED -> SUSPENDED -> RESUMED
221    // INITIALIZED -> RESUMED
222    private ScanResumerState state = ScanResumerState.INITIALIZED;
223
224    private ScanResponse resp;
225
226    private int numberOfCompleteRows;
227
228    // If the scan is suspended successfully, we need to do lease renewal to prevent it being closed
229    // by RS due to lease expire. It is a one-time timer task so we need to schedule a new task
230    // every time when the previous task is finished. There could also be race as the renewal is
231    // executed in the timer thread, so we also need to check the state before lease renewal. If the
232    // state is RESUMED already, we will give up lease renewal and also not schedule the next lease
233    // renewal task.
234    private Timeout leaseRenewer;
235
236    @Override
237    public void resume() {
238      // just used to fix findbugs warnings. In fact, if resume is called before prepare, then we
239      // just return at the first if condition without loading the resp and numValidResuls field. If
240      // resume is called after suspend, then it is also safe to just reference resp and
241      // numValidResults after the synchronized block as no one will change it anymore.
242      ScanResponse localResp;
243      int localNumberOfCompleteRows;
244      synchronized (this) {
245        if (state == ScanResumerState.INITIALIZED) {
246          // user calls this method before we call prepare, so just set the state to
247          // RESUMED, the implementation will just go on.
248          state = ScanResumerState.RESUMED;
249          return;
250        }
251        if (state == ScanResumerState.RESUMED) {
252          // already resumed, give up.
253          return;
254        }
255        state = ScanResumerState.RESUMED;
256        if (leaseRenewer != null) {
257          leaseRenewer.cancel();
258        }
259        localResp = this.resp;
260        localNumberOfCompleteRows = this.numberOfCompleteRows;
261      }
262      completeOrNext(localResp, localNumberOfCompleteRows);
263    }
264
265    private void scheduleRenewLeaseTask() {
266      leaseRenewer = retryTimer.newTimeout(t -> tryRenewLease(), scannerLeaseTimeoutPeriodNs / 2,
267        TimeUnit.NANOSECONDS);
268    }
269
270    private synchronized void tryRenewLease() {
271      // the scan has already been resumed, give up
272      if (state == ScanResumerState.RESUMED) {
273        return;
274      }
275      renewLease();
276      // schedule the next renew lease task again as this is a one-time task.
277      scheduleRenewLeaseTask();
278    }
279
280    // return false if the scan has already been resumed. See the comment above for ScanResumerImpl
281    // for more details.
282    synchronized boolean prepare(ScanResponse resp, int numberOfCompleteRows) {
283      if (state == ScanResumerState.RESUMED) {
284        // user calls resume before we actually suspend the scan, just continue;
285        return false;
286      }
287      state = ScanResumerState.SUSPENDED;
288      this.resp = resp;
289      this.numberOfCompleteRows = numberOfCompleteRows;
290      // if there are no more results in region then the scanner at RS side will be closed
291      // automatically so we do not need to renew lease.
292      if (resp.getMoreResultsInRegion()) {
293        // schedule renew lease task
294        scheduleRenewLeaseTask();
295      }
296      return true;
297    }
298  }

300  public AsyncScanSingleRegionRpcRetryingCaller(HashedWheelTimer retryTimer,
301      AsyncConnectionImpl conn, Scan scan, ScanMetrics scanMetrics, long scannerId,
302      ScanResultCache resultCache, AdvancedScanResultConsumer consumer, Interface stub,
303      HRegionLocation loc, boolean isRegionServerRemote, long scannerLeaseTimeoutPeriodNs,
304      long pauseNs, int maxAttempts, long scanTimeoutNs, long rpcTimeoutNs, int startLogErrorsCnt) {
305    this.retryTimer = retryTimer;
306    this.scan = scan;
307    this.scanMetrics = scanMetrics;
308    this.scannerId = scannerId;
309    this.resultCache = resultCache;
310    this.consumer = consumer;
311    this.stub = stub;
312    this.loc = loc;
313    this.regionServerRemote = isRegionServerRemote;
314    this.scannerLeaseTimeoutPeriodNs = scannerLeaseTimeoutPeriodNs;
315    this.pauseNs = pauseNs;
316    this.maxAttempts = maxAttempts;
317    this.scanTimeoutNs = scanTimeoutNs;
318    this.rpcTimeoutNs = rpcTimeoutNs;
319    this.startLogErrorsCnt = startLogErrorsCnt;
320    if (scan.isReversed()) {
321      completeWhenNoMoreResultsInRegion = this::completeReversedWhenNoMoreResultsInRegion;
322    } else {
323      completeWhenNoMoreResultsInRegion = this::completeWhenNoMoreResultsInRegion;
324    }

326    this.controller = conn.rpcControllerFactory.newController();
327    this.exceptions = new ArrayList<>();

330  private long elapsedMs() {
331    return TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - nextCallStartNs);
332  }

334  private long remainingTimeNs() {
335    return scanTimeoutNs - (System.nanoTime() - nextCallStartNs);

338  private void closeScanner() {
339    incRPCCallsMetrics(scanMetrics, regionServerRemote);
340    resetController(controller, rpcTimeoutNs);
341    ScanRequest req = RequestConverter.buildScanRequest(this.scannerId, 0, true, false);
342    stub.scan(controller, req, resp -> {
343      if (controller.failed()) {
344        LOG.warn("Call to " + loc.getServerName() + " for closing scanner id = " + scannerId +
345            " for " + loc.getRegion().getEncodedName() + " of " +
346            loc.getRegion().getTable() + " failed, ignore, probably already closed",
347          controller.getFailed());
348      }
349    });
350  }

352  private void completeExceptionally(boolean closeScanner) {
353    resultCache.clear();
354    if (closeScanner) {
355      closeScanner();
356    }
357    future.completeExceptionally(new RetriesExhaustedException(tries - 1, exceptions));
358  }

360  private void completeNoMoreResults() {
361    future.complete(false);
362  }

364  private void completeWithNextStartRow(byte[] row, boolean inclusive) {
365    scan.withStartRow(row, inclusive);
366    future.complete(true);
367  }

369  private void completeWhenError(boolean closeScanner) {
370    incRPCRetriesMetrics(scanMetrics, closeScanner);
371    resultCache.clear();
372    if (closeScanner) {
373      closeScanner();
374    }
375    if (nextStartRowWhenError != null) {
376      scan.withStartRow(nextStartRowWhenError, includeNextStartRowWhenError);
377    }
378    future.complete(true);
379  }
380
381  private void onError(Throwable error) {
382    error = translateException(error);
383    if (tries > startLogErrorsCnt) {
384      LOG.warn("Call to " + loc.getServerName() + " for scanner id = " + scannerId + " for " +
385          loc.getRegion().getEncodedName() + " of " + loc.getRegion().getTable() +
386          " failed, , tries = " + tries + ", maxAttempts = " + maxAttempts + ", timeout = " +
387          TimeUnit.NANOSECONDS.toMillis(scanTimeoutNs) + " ms, time elapsed = " + elapsedMs() +
388          " ms",
389        error);
390    }
391    boolean scannerClosed = error instanceof UnknownScannerException ||
392        error instanceof NotServingRegionException || error instanceof RegionServerStoppedException;
393    RetriesExhaustedException.ThrowableWithExtraContext qt =
394        new RetriesExhaustedException.ThrowableWithExtraContext(error,
395            EnvironmentEdgeManager.currentTime(), "");
396    exceptions.add(qt);
397    if (tries >= maxAttempts) {
398      completeExceptionally(!scannerClosed);
399      return;
400    }
401    long delayNs;
402    if (scanTimeoutNs > 0) {
403      long maxDelayNs = remainingTimeNs() - SLEEP_DELTA_NS;
404      if (maxDelayNs <= 0) {
405        completeExceptionally(!scannerClosed);
406        return;
407      }
408      delayNs = Math.min(maxDelayNs, getPauseTime(pauseNs, tries - 1));
409    } else {
410      delayNs = getPauseTime(pauseNs, tries - 1);
411    }
412    if (scannerClosed) {
413      completeWhenError(false);
414      return;
415    }
416    if (error instanceof OutOfOrderScannerNextException || error instanceof ScannerResetException) {
417      completeWhenError(true);
418      return;
419    }
420    if (error instanceof DoNotRetryIOException) {
421      completeExceptionally(true);
422      return;
423    }
424    tries++;
425    retryTimer.newTimeout(t -> call(), delayNs, TimeUnit.NANOSECONDS);
426  }

428  private void updateNextStartRowWhenError(Result result) {
429    nextStartRowWhenError = result.getRow();
430    includeNextStartRowWhenError = result.mayHaveMoreCellsInRow();
431  }

433  private void completeWhenNoMoreResultsInRegion() {
434    if (noMoreResultsForScan(scan, loc.getRegion())) {
435      completeNoMoreResults();
436    } else {
437      completeWithNextStartRow(loc.getRegion().getEndKey(), true);
438    }
439  }

441  private void completeReversedWhenNoMoreResultsInRegion() {
442    if (noMoreResultsForReverseScan(scan, loc.getRegion())) {
443      completeNoMoreResults();
444    } else {
445      completeWithNextStartRow(loc.getRegion().getStartKey(), false);
446    }
447  }

449  private void completeOrNext(ScanResponse resp, int numberOfCompleteRows) {
450    if (resp.hasMoreResults() && !resp.getMoreResults()) {
451      // RS tells us there is no more data for the whole scan
452      completeNoMoreResults();
453      return;
454    }
455    if (scan.getLimit() > 0) {
456      // The RS should have set the moreResults field in ScanResponse to false when we have reached
457      // the limit, so we add an assert here.
458      int newLimit = scan.getLimit() - numberOfCompleteRows;
459      assert newLimit > 0;
460      scan.setLimit(newLimit);
461    }
462    // as in 2.0 this value will always be set
463    if (!resp.getMoreResultsInRegion()) {
464      completeWhenNoMoreResultsInRegion.run();
465      return;
466    }
467    next();
468  }

470  private void onComplete(HBaseRpcController controller, ScanResponse resp) {
471    if (controller.failed()) {
472      onError(controller.getFailed());
473      return;
474    }
475    updateServerSideMetrics(scanMetrics, resp);
476    boolean isHeartbeatMessage = resp.hasHeartbeatMessage() && resp.getHeartbeatMessage();
477    Result[] rawResults;
478    Result[] results;
479    int numberOfCompleteRowsBefore = resultCache.numberOfCompleteRows();
480    try {
481      rawResults = ResponseConverter.getResults(controller.cellScanner(), resp);
482      updateResultsMetrics(scanMetrics, rawResults, isHeartbeatMessage);
483      results = resultCache.addAndGet(
484        Optional.ofNullable(rawResults).orElse(ScanResultCache.EMPTY_RESULT_ARRAY),
485        isHeartbeatMessage);
486    } catch (IOException e) {
487      // We can not retry here. The server has responded normally and the call sequence has been
488      // increased so a new scan with the same call sequence will cause an
489      // OutOfOrderScannerNextException. Let the upper layer open a new scanner.
490      LOG.warn("decode scan response failed", e);
491      completeWhenError(true);
492      return;
493    }
494
495    ScanControllerImpl scanController;
496    if (results.length > 0) {
497      scanController = new ScanControllerImpl(
498          resp.hasCursor() ? Optional.of(ProtobufUtil.toCursor(resp.getCursor()))
499              : Optional.empty());
500      updateNextStartRowWhenError(results[results.length - 1]);
501      consumer.onNext(results, scanController);
502    } else {
503      Optional<Cursor> cursor = Optional.empty();
504      if (resp.hasCursor()) {
505        cursor = Optional.of(ProtobufUtil.toCursor(resp.getCursor()));
506      } else if (scan.isNeedCursorResult() && rawResults.length > 0) {
507        // It is size limit exceed and we need to return the last Result's row.
508        // When user setBatch and the scanner is reopened, the server may return Results that
509        // user has seen and the last Result can not be seen because the number is not enough.
510        // So the row keys of results may not be same, we must use the last one.
511        cursor = Optional.of(new Cursor(rawResults[rawResults.length - 1].getRow()));
512      }
513      scanController = new ScanControllerImpl(cursor);
514      if (isHeartbeatMessage || cursor.isPresent()) {
515        // only call onHeartbeat if server tells us explicitly this is a heartbeat message, or we
516        // want to pass a cursor to upper layer.
517        consumer.onHeartbeat(scanController);
518      }
519    }
520    ScanControllerState state = scanController.destroy();
521    if (state == ScanControllerState.TERMINATED) {
522      if (resp.getMoreResultsInRegion()) {
523        // we have more results in region but user request to stop the scan, so we need to close the
524        // scanner explicitly.
525        closeScanner();
526      }
527      completeNoMoreResults();
528      return;
529    }
530    int numberOfCompleteRows = resultCache.numberOfCompleteRows() - numberOfCompleteRowsBefore;
531    if (state == ScanControllerState.SUSPENDED) {
532      if (scanController.resumer.prepare(resp, numberOfCompleteRows)) {
533        return;
534      }
535    }
536    completeOrNext(resp, numberOfCompleteRows);
537  }

539  private void call() {
540    // As we have a call sequence for scan, it is useless to have a different rpc timeout which is
541    // less than the scan timeout. If the server does not respond in time(usually this will not
542    // happen as we have heartbeat now), we will get an OutOfOrderScannerNextException when
543    // resending the next request and the only way to fix this is to close the scanner and open a
544    // new one.
545    long callTimeoutNs;
546    if (scanTimeoutNs > 0) {
547      long remainingNs = scanTimeoutNs - (System.nanoTime() - nextCallStartNs);
548      if (remainingNs <= 0) {
549        completeExceptionally(true);
550        return;
551      }
552      callTimeoutNs = remainingNs;
553    } else {
554      callTimeoutNs = 0L;
555    }
556    incRPCCallsMetrics(scanMetrics, regionServerRemote);
557    if (tries > 1) {
558      incRPCRetriesMetrics(scanMetrics, regionServerRemote);
559    }
560    resetController(controller, callTimeoutNs);
561    ScanRequest req = RequestConverter.buildScanRequest(scannerId, scan.getCaching(), false,
562      nextCallSeq, false, false, scan.getLimit());
563    stub.scan(controller, req, resp -> onComplete(controller, resp));
564  }
565
566  private void next() {
567    nextCallSeq++;
568    tries = 1;
569    exceptions.clear();
570    nextCallStartNs = System.nanoTime();
571    call();
572  }
573
574  private void renewLease() {
575    incRPCCallsMetrics(scanMetrics, regionServerRemote);
576    nextCallSeq++;
577    resetController(controller, rpcTimeoutNs);
578    ScanRequest req =
579        RequestConverter.buildScanRequest(scannerId, 0, false, nextCallSeq, false, true, -1);
580    stub.scan(controller, req, resp -> {
581    });
582  }

585   * Now we will also fetch some cells along with the scanner id when opening a scanner, so we also
586   * need to process the ScanResponse for the open scanner request. The HBaseRpcController for the
587   * open scanner request is also needed because we may have some data in the CellScanner which is
588   * contained in the controller.

593    onComplete(controller, respWhenOpen);
594    return future;
