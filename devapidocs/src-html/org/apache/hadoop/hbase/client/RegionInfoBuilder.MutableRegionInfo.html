125  /**
126   * An implementation of RegionInfo that adds mutable methods so can build a RegionInfo instance.

128  @InterfaceAudience.Private

130    /**
131     * The new format for a region name contains its encodedName at the end.
132     * The encoded name also serves as the directory name for the region
133     * in the filesystem.
134     *
135     * New region name format:
136     *    &lt;tablename>,,&lt;startkey>,&lt;regionIdTimestamp>.&lt;encodedName>.
137     * where,
138     *    &lt;encodedName> is a hex version of the MD5 hash of
139     *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
140     *
141     * The old region name format:
142     *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
143     * For region names in the old format, the encoded name is a 32-bit
144     * JenkinsHash integer value (in its decimal notation, string form).
145     *<p>
146     * **NOTE**
147     *
148     * The first hbase:meta region, and regions created by an older
149     * version of HBase (0.20 or prior) will continue to use the
150     * old region name format.
151     */
152
153    // This flag is in the parent of a split while the parent is still referenced
154    // by daughter regions.  We USED to set this flag when we disabled a table
155    // but now table state is kept up in zookeeper as of 0.90.0 HBase.
156    private boolean offLine = false;
157    private boolean split = false;
158    private final long regionId;
159    private final int replicaId;
160    private final byte[] regionName;
161    private final byte[] startKey;
162    private final byte[] endKey;
163    private final int hashCode;
164    private final String encodedName;
165    private final byte[] encodedNameAsBytes;
166    private final TableName tableName;
167
168    private static int generateHashCode(final TableName tableName, final byte[] startKey,
169        final byte[] endKey, final long regionId,
170        final int replicaId, boolean offLine, byte[] regionName) {
171      int result = Arrays.hashCode(regionName);
172      result ^= regionId;
173      result ^= Arrays.hashCode(checkStartKey(startKey));
174      result ^= Arrays.hashCode(checkEndKey(endKey));
175      result ^= Boolean.valueOf(offLine).hashCode();
176      result ^= Arrays.hashCode(tableName.getName());
177      result ^= replicaId;
178      return result;
179    }
180
181    private static byte[] checkStartKey(byte[] startKey) {
182      return startKey == null? HConstants.EMPTY_START_ROW: startKey;
183    }
184
185    private static byte[] checkEndKey(byte[] endKey) {
186      return endKey == null? HConstants.EMPTY_END_ROW: endKey;
187    }
188
189    private static TableName checkTableName(TableName tableName) {
190      if (tableName == null) {
191        throw new IllegalArgumentException("TableName cannot be null");
192      }
193      return tableName;
194    }
195
196    private static int checkReplicaId(int regionId) {
197      if (regionId > MAX_REPLICA_ID) {
198        throw new IllegalArgumentException("ReplicaId cannot be greater than" + MAX_REPLICA_ID);
199      }
200      return regionId;
201    }
202
203    /**
204     * Private constructor used constructing MutableRegionInfo for the
205     * first meta regions
206     */
207    private MutableRegionInfo(long regionId, TableName tableName, int replicaId) {
208      this(tableName,
209          HConstants.EMPTY_START_ROW,
210          HConstants.EMPTY_END_ROW,
211          false,
212          regionId,
213          replicaId,
214          false,
215          RegionInfo.createRegionName(tableName, null, regionId, replicaId, false));
216    }
217
218    MutableRegionInfo(final TableName tableName, final byte[] startKey,
219        final byte[] endKey, final boolean split, final long regionId,
220        final int replicaId, boolean offLine, byte[] regionName) {
221      this(checkTableName(tableName),
222          checkStartKey(startKey),
223          checkEndKey(endKey),
224          split, regionId,
225          checkReplicaId(replicaId),
226          offLine,
227          regionName,
228          RegionInfo.encodeRegionName(regionName));
229    }
230
231    MutableRegionInfo(final TableName tableName, final byte[] startKey,
232        final byte[] endKey, final boolean split, final long regionId,
233        final int replicaId, boolean offLine, byte[] regionName, String encodedName) {
234      this.tableName = checkTableName(tableName);
235      this.startKey = checkStartKey(startKey);
236      this.endKey = checkEndKey(endKey);
237      this.split = split;
238      this.regionId = regionId;
239      this.replicaId = checkReplicaId(replicaId);
240      this.offLine = offLine;
241      if (ArrayUtils.isEmpty(regionName)) {
242        this.regionName = RegionInfo.createRegionName(this.tableName, this.startKey, this.regionId, this.replicaId,
243            !this.tableName.equals(TableName.META_TABLE_NAME));
244        this.encodedName = RegionInfo.encodeRegionName(this.regionName);
245      } else {
246        this.regionName = regionName;
247        this.encodedName = encodedName;
248      }
249      this.hashCode = generateHashCode(
250          this.tableName,
251          this.startKey,
252          this.endKey,
253          this.regionId,
254          this.replicaId,
255          this.offLine,
256          this.regionName);
257      this.encodedNameAsBytes = Bytes.toBytes(this.encodedName);
258    }

333    /**
334     * Returns true if the given inclusive range of rows is fully contained
335     * by this region. For example, if the region is foo,a,g and this is
336     * passed ["b","c"] or ["a","c"] it will return true, but if this is passed
337     * ["b","z"] it will return false.
338     * @throws IllegalArgumentException if the range passed is invalid (ie. end &lt; start)
339     */
340    @Override
341    public boolean containsRange(byte[] rangeStartKey, byte[] rangeEndKey) {
342      if (Bytes.compareTo(rangeStartKey, rangeEndKey) > 0) {
343        throw new IllegalArgumentException(
344        "Invalid range: " + Bytes.toStringBinary(rangeStartKey) +
345        " > " + Bytes.toStringBinary(rangeEndKey));
346      }
347
348      boolean firstKeyInRange = Bytes.compareTo(rangeStartKey, startKey) >= 0;
349      boolean lastKeyInRange =
350        Bytes.compareTo(rangeEndKey, endKey) < 0 ||
351        Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY);
352      return firstKeyInRange && lastKeyInRange;
353    }
354
355    /**
356     * Return true if the given row falls in this region.
357     */
358    @Override
359    public boolean containsRow(byte[] row) {
360      return Bytes.compareTo(row, startKey) >= 0 &&
361        (Bytes.compareTo(row, endKey) < 0 ||
362         Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY));
363    }

423    /**

426    @Override
427    public boolean isSplitParent() {
428      if (!isSplit()) return false;
429      if (!isOffline()) {

432      return true;

448    public String toString() {
449      return "{ENCODED => " + getEncodedName() + ", " +
450        HConstants.NAME + " => '" + Bytes.toStringBinary(this.regionName)
451        + "', STARTKEY => '" +
452        Bytes.toStringBinary(this.startKey) + "', ENDKEY => '" +
453        Bytes.toStringBinary(this.endKey) + "'" +
454        (isOffline()? ", OFFLINE => true": "") +
455        (isSplit()? ", SPLIT => true": "") +
456        ((replicaId > 0)? ", REPLICA_ID => " + replicaId : "") + "}";
457    }
