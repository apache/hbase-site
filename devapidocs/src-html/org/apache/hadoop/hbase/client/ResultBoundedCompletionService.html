019package org.apache.hadoop.hbase.client;
020
021import java.util.concurrent.ExecutionException;
022import java.util.concurrent.Executor;
023import java.util.concurrent.RunnableFuture;
024import java.util.concurrent.TimeUnit;
025import java.util.concurrent.TimeoutException;
026
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.htrace.Trace;
029
030/**
031 * A completion service for the RpcRetryingCallerFactory.
032 * Keeps the list of the futures, and allows to cancel them all.
033 * This means as well that it can be used for a small set of tasks only.
034 * <br>Implementation is not Thread safe.
035 */
036@InterfaceAudience.Private
037public class ResultBoundedCompletionService<V> {
038  private final RpcRetryingCallerFactory retryingCallerFactory;
039  private final Executor executor;
040  private final QueueingFuture<V>[] tasks; // all the tasks
041  private volatile QueueingFuture<V> completed = null;
042  private volatile boolean cancelled = false;
043  
044  class QueueingFuture<T> implements RunnableFuture<T> {
045    private final RetryingCallable<T> future;
046    private T result = null;
047    private ExecutionException exeEx = null;
048    private volatile boolean cancelled = false;
049    private final int callTimeout;
050    private final RpcRetryingCaller<T> retryingCaller;
051    private boolean resultObtained = false;
052
053
054    public QueueingFuture(RetryingCallable<T> future, int callTimeout) {
055      this.future = future;
056      this.callTimeout = callTimeout;
057      this.retryingCaller = retryingCallerFactory.<T>newCaller();
058    }
059
060    @SuppressWarnings("unchecked")
061    @Override
062    public void run() {
063      try {
064        if (!cancelled) {
065          result = this.retryingCaller.callWithRetries(future, callTimeout);
066          resultObtained = true;
067        }
068      } catch (Throwable t) {
069        exeEx = new ExecutionException(t);
070      } finally {
071        synchronized (tasks) {
072          // If this wasn't canceled then store the result.
073          if (!cancelled && completed == null) {
074            completed = (QueueingFuture<V>) QueueingFuture.this;
075          }
076
077          // Notify just in case there was someone waiting and this was canceled.
078          // That shouldn't happen but better safe than sorry.
079          tasks.notify();
080        }
081      }
082    }
083    @Override
084    public boolean cancel(boolean mayInterruptIfRunning) {
085      if (resultObtained || exeEx != null) return false;
086      retryingCaller.cancel();
087      if (future instanceof Cancellable) ((Cancellable)future).cancel();
088      cancelled = true;
089      return true;
090    }
091
092    @Override
093    public boolean isCancelled() {
094      return cancelled;
095    }
096
097    @Override
098    public boolean isDone() {
099      return resultObtained || exeEx != null;
100    }
101
102    @Override
103    public T get() throws InterruptedException, ExecutionException {
104      try {
105        return get(1000, TimeUnit.DAYS);
106      } catch (TimeoutException e) {
107        throw new RuntimeException("You did wait for 1000 days here?", e);
108      }
109    }
110
111    @Override
112    public T get(long timeout, TimeUnit unit)
113        throws InterruptedException, ExecutionException, TimeoutException {
114      synchronized (tasks) {
115        if (resultObtained) {
116          return result;
117        }
118        if (exeEx != null) {
119          throw exeEx;
120        }
121        unit.timedWait(tasks, timeout);
122      }
123      if (resultObtained) {
124        return result;
125      }
126      if (exeEx != null) {
127        throw exeEx;
128      }
129
130      throw new TimeoutException("timeout=" + timeout + ", " + unit);
131    }
132  }
133
134  @SuppressWarnings("unchecked")
135  public ResultBoundedCompletionService(
136      RpcRetryingCallerFactory retryingCallerFactory, Executor executor,
137      int maxTasks) {
138    this.retryingCallerFactory = retryingCallerFactory;
139    this.executor = executor;
140    this.tasks = new QueueingFuture[maxTasks];
141  }
142
143
144  public void submit(RetryingCallable<V> task, int callTimeout, int id) {
145    QueueingFuture<V> newFuture = new QueueingFuture<V>(task, callTimeout);
146    executor.execute(Trace.wrap(newFuture));
147    tasks[id] = newFuture;
148  }
149
150  public QueueingFuture<V> take() throws InterruptedException {
151    synchronized (tasks) {
152      while (completed == null && !cancelled) tasks.wait();
153    }
154    return completed;
155  }
156
157  public QueueingFuture<V> poll(long timeout, TimeUnit unit) throws InterruptedException {
158    synchronized (tasks) {
159      if (completed == null && !cancelled) unit.timedWait(tasks, timeout);
160    }
161    return completed;
162  }
163
164  public void cancelAll() {
165    // Grab the lock on tasks so that cancelled is visible everywhere
166    synchronized (tasks) {
167      cancelled = true;
168    }
169    for (QueueingFuture<V> future : tasks) {
170      if (future != null) future.cancel(true);
171    }
172  }
173}
