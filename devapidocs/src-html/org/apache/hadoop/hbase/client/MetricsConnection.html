047/**
048 * This class is for maintaining the various connection statistics and publishing them through
049 * the metrics interfaces.
050 *
051 * This class manages its own {@link MetricRegistry} and {@link JmxReporter} so as to not
052 * conflict with other uses of Yammer Metrics within the client application. Instantiating
053 * this class implicitly creates and "starts" instances of these classes; be sure to call
054 * {@link #shutdown()} to terminate the thread pools they allocate.
055 */
056@InterfaceAudience.Private
057public class MetricsConnection implements StatisticTrackable {
058
059  /** Set this key to {@code true} to enable metrics collection of client requests. */
060  public static final String CLIENT_SIDE_METRICS_ENABLED_KEY = "hbase.client.metrics.enable";
061
062  /**
063   * Set to specify a custom scope for the metrics published through {@link MetricsConnection}.
064   * The scope is added to JMX MBean objectName, and defaults to a combination of the Connection's
065   * clusterId and hashCode. For example, a default value for a connection to cluster "foo" might
066   * be "foo-7d9d0818", where "7d9d0818" is the hashCode of the underlying AsyncConnectionImpl.
067   * Users may set this key to give a more contextual name for this scope. For example, one might
068   * want to differentiate a read connection from a write connection by setting the scopes to
069   * "foo-read" and "foo-write" respectively.
070   *
071   * Scope is the only thing that lends any uniqueness to the metrics. Care should be taken to
072   * avoid using the same scope for multiple Connections, otherwise the metrics may aggregate in
073   * unforeseen ways.
074   */
075  public static final String METRICS_SCOPE_KEY = "hbase.client.metrics.scope";
076
077  /**
078   * Returns the scope for a MetricsConnection based on the configured {@link #METRICS_SCOPE_KEY}
079   * or by generating a default from the passed clusterId and connectionObj's hashCode.
080   * @param conf          configuration for the connection
081   * @param clusterId     clusterId for the connection
082   * @param connectionObj either a Connection or AsyncConnectionImpl, the instance
083   *                      creating this MetricsConnection.
084   */
085  static String getScope(Configuration conf, String clusterId, Object connectionObj) {
086    return conf.get(METRICS_SCOPE_KEY,
087      clusterId + "@" + Integer.toHexString(connectionObj.hashCode()));
088  }

160  protected static final class CallTracker {
161    private final String name;
162    final Timer callTimer;
163    final Histogram reqHist;
164    final Histogram respHist;
165
166    private CallTracker(MetricRegistry registry, String name, String subName, String scope) {
167      StringBuilder sb = new StringBuilder(CLIENT_SVC).append("_").append(name);
168      if (subName != null) {
169        sb.append("(").append(subName).append(")");
170      }
171      this.name = sb.toString();
172      this.callTimer = registry.timer(name(MetricsConnection.class,
173        DRTN_BASE + this.name, scope));
174      this.reqHist = registry.histogram(name(MetricsConnection.class,
175        REQ_BASE + this.name, scope));
176      this.respHist = registry.histogram(name(MetricsConnection.class,
177        RESP_BASE + this.name, scope));
178    }
179
180    private CallTracker(MetricRegistry registry, String name, String scope) {
181      this(registry, name, null, scope);
182    }
183
184    public void updateRpc(CallStats stats) {
185      this.callTimer.update(stats.getCallTimeMs(), TimeUnit.MILLISECONDS);
186      this.reqHist.update(stats.getRequestSizeBytes());
187      this.respHist.update(stats.getResponseSizeBytes());
188    }
189
190    @Override
191    public String toString() {
192      return "CallTracker:" + name;
193    }
194  }

242  protected ConcurrentHashMap<ServerName, ConcurrentMap<byte[], RegionStats>> serverStats
243          = new ConcurrentHashMap<>();
244
245  public void updateServerStats(ServerName serverName, byte[] regionName,
246                                Object r) {
247    if (!(r instanceof Result)) {
248      return;
249    }
250    Result result = (Result) r;
251    RegionLoadStats stats = result.getStats();
252    if (stats == null) {
253      return;
254    }
255    updateRegionStats(serverName, regionName, stats);
256  }
257
258  @Override
259  public void updateRegionStats(ServerName serverName, byte[] regionName, RegionLoadStats stats) {
260    String name = serverName.getServerName() + "," + Bytes.toStringBinary(regionName);
261    ConcurrentMap<byte[], RegionStats> rsStats = computeIfAbsent(serverStats, serverName,
262      () -> new ConcurrentSkipListMap<>(Bytes.BYTES_COMPARATOR));
263    RegionStats regionStats =
264        computeIfAbsent(rsStats, regionName, () -> new RegionStats(this.registry, name));
265    regionStats.update(stats);
266  }

327  // These maps are used to cache references to the metric instances that are managed by the
328  // registry. I don't think their use perfectly removes redundant allocations, but it's
329  // a big improvement over calling registry.newMetric each time.
330  protected final ConcurrentMap<String, Timer> rpcTimers =
331      new ConcurrentHashMap<>(CAPACITY, LOAD_FACTOR, CONCURRENCY_LEVEL);
332  protected final ConcurrentMap<String, Histogram> rpcHistograms =
333      new ConcurrentHashMap<>(CAPACITY * 2 /* tracking both request and response sizes */,
334          LOAD_FACTOR, CONCURRENCY_LEVEL);

340  MetricsConnection(String scope, Supplier<ThreadPoolExecutor> batchPool,
341      Supplier<ThreadPoolExecutor> metaPool) {
342    this.scope = scope;
343    this.registry = new MetricRegistry();
344    this.registry.register(getExecutorPoolName(),
345        new RatioGauge() {
346          @Override
347          protected Ratio getRatio() {
348            ThreadPoolExecutor pool = batchPool.get();
349            if (pool == null) {
350              return Ratio.of(0, 0);
351            }
352            return Ratio.of(pool.getActiveCount(), pool.getMaximumPoolSize());
353          }
354        });
355    this.registry.register(getMetaPoolName(),
356        new RatioGauge() {
357          @Override
358          protected Ratio getRatio() {
359            ThreadPoolExecutor pool = metaPool.get();
360            if (pool == null) {
361              return Ratio.of(0, 0);
362            }
363            return Ratio.of(pool.getActiveCount(), pool.getMaximumPoolSize());
364          }
365        });
366    this.metaCacheHits = registry.counter(name(this.getClass(), "metaCacheHits", scope));
367    this.metaCacheMisses = registry.counter(name(this.getClass(), "metaCacheMisses", scope));
368    this.metaCacheNumClearServer = registry.counter(name(this.getClass(),
369      "metaCacheNumClearServer", scope));
370    this.metaCacheNumClearRegion = registry.counter(name(this.getClass(),
371      "metaCacheNumClearRegion", scope));
372    this.hedgedReadOps = registry.counter(name(this.getClass(), "hedgedReadOps", scope));
373    this.hedgedReadWin = registry.counter(name(this.getClass(), "hedgedReadWin", scope));
374    this.getTracker = new CallTracker(this.registry, "Get", scope);
375    this.scanTracker = new CallTracker(this.registry, "Scan", scope);
376    this.appendTracker = new CallTracker(this.registry, "Mutate", "Append", scope);
377    this.deleteTracker = new CallTracker(this.registry, "Mutate", "Delete", scope);
378    this.incrementTracker = new CallTracker(this.registry, "Mutate", "Increment", scope);
379    this.putTracker = new CallTracker(this.registry, "Mutate", "Put", scope);
380    this.multiTracker = new CallTracker(this.registry, "Multi", scope);
381    this.runnerStats = new RunnerStats(this.registry);
382    this.concurrentCallsPerServerHist = registry.histogram(name(MetricsConnection.class,
383      "concurrentCallsPerServer", scope));
384    this.numActionsPerServerHist = registry.histogram(name(MetricsConnection.class,
385      "numActionsPerServer", scope));
386    this.nsLookups = registry.counter(name(this.getClass(), NS_LOOKUPS, scope));
387    this.nsLookupsFailed = registry.counter(name(this.getClass(), NS_LOOKUPS_FAILED, scope));
388
389    this.reporter = JmxReporter.forRegistry(this.registry).build();
390    this.reporter.start();
391  }

462   * Get a metric for {@code key} from {@code map}, or create it with {@code factory}.
463   */
464  private <T> T getMetric(String key, ConcurrentMap<String, T> map, NewMetric<T> factory) {

468  /** Update call stats for non-critical-path methods */
469  private void updateRpcGeneric(String methodName, CallStats stats) {
470    getMetric(DRTN_BASE + methodName, rpcTimers, timerFactory)
471        .update(stats.getCallTimeMs(), TimeUnit.MILLISECONDS);
472    getMetric(REQ_BASE + methodName, rpcHistograms, histogramFactory)
473        .update(stats.getRequestSizeBytes());
474    getMetric(RESP_BASE + methodName, rpcHistograms, histogramFactory)
475        .update(stats.getResponseSizeBytes());
476  }

478  /** Report RPC context to metrics system. */
479  public void updateRpc(MethodDescriptor method, Message param, CallStats stats) {
480    int callsPerServer = stats.getConcurrentCallsPerServer();
481    if (callsPerServer > 0) {
482      concurrentCallsPerServerHist.update(callsPerServer);
483    }
484    // Update the counter that tracks RPCs by type.
485    final String methodName = method.getService().getName() + "_" + method.getName();
486    getMetric(CNT_BASE + methodName, rpcCounters, counterFactory).inc();
487    // this implementation is tied directly to protobuf implementation details. would be better
488    // if we could dispatch based on something static, ie, request Message type.
489    if (method.getService() == ClientService.getDescriptor()) {
490      switch(method.getIndex()) {
491        case 0:
492          assert "Get".equals(method.getName());
493          getTracker.updateRpc(stats);
494          return;
495        case 1:
496          assert "Mutate".equals(method.getName());
497          final MutationType mutationType = ((MutateRequest) param).getMutation().getMutateType();
498          switch(mutationType) {
499            case APPEND:
500              appendTracker.updateRpc(stats);
501              return;
502            case DELETE:
503              deleteTracker.updateRpc(stats);
504              return;
505            case INCREMENT:
506              incrementTracker.updateRpc(stats);
507              return;
508            case PUT:
509              putTracker.updateRpc(stats);
510              return;
511            default:
512              throw new RuntimeException("Unrecognized mutation type " + mutationType);
513          }
514        case 2:
515          assert "Scan".equals(method.getName());
516          scanTracker.updateRpc(stats);
517          return;
518        case 3:
519          assert "BulkLoadHFile".equals(method.getName());
520          // use generic implementation
521          break;
522        case 4:
523          assert "PrepareBulkLoad".equals(method.getName());
524          // use generic implementation
525          break;
526        case 5:
527          assert "CleanupBulkLoad".equals(method.getName());
528          // use generic implementation
529          break;
530        case 6:
531          assert "ExecService".equals(method.getName());
532          // use generic implementation
533          break;
534        case 7:
535          assert "ExecRegionServerService".equals(method.getName());
536          // use generic implementation
537          break;
538        case 8:
539          assert "Multi".equals(method.getName());
540          numActionsPerServerHist.update(stats.getNumActionsPerServer());
541          multiTracker.updateRpc(stats);
542          return;
543        default:
544          throw new RuntimeException("Unrecognized ClientService RPC type " + method.getFullName());
545      }
546    }
547    // Fallback to dynamic registry lookup for DDL methods.
548    updateRpcGeneric(methodName, stats);
549  }
