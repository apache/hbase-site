046/**
047 * This class is for maintaining the various connection statistics and publishing them through
048 * the metrics interfaces.
049 *
050 * This class manages its own {@link MetricRegistry} and {@link JmxReporter} so as to not
051 * conflict with other uses of Yammer Metrics within the client application. Instantiating
052 * this class implicitly creates and "starts" instances of these classes; be sure to call
053 * {@link #shutdown()} to terminate the thread pools they allocate.
054 */
055@InterfaceAudience.Private
056public class MetricsConnection implements StatisticTrackable {
057

110  @VisibleForTesting
111  protected static final class CallTracker {
112    private final String name;
113    @VisibleForTesting final Timer callTimer;
114    @VisibleForTesting final Histogram reqHist;
115    @VisibleForTesting final Histogram respHist;
116
117    private CallTracker(MetricRegistry registry, String name, String subName, String scope) {
118      StringBuilder sb = new StringBuilder(CLIENT_SVC).append("_").append(name);
119      if (subName != null) {
120        sb.append("(").append(subName).append(")");
121      }
122      this.name = sb.toString();
123      this.callTimer = registry.timer(name(MetricsConnection.class,
124        DRTN_BASE + this.name, scope));
125      this.reqHist = registry.histogram(name(MetricsConnection.class,
126        REQ_BASE + this.name, scope));
127      this.respHist = registry.histogram(name(MetricsConnection.class,
128        RESP_BASE + this.name, scope));
129    }
130
131    private CallTracker(MetricRegistry registry, String name, String scope) {
132      this(registry, name, null, scope);
133    }
134
135    public void updateRpc(CallStats stats) {
136      this.callTimer.update(stats.getCallTimeMs(), TimeUnit.MILLISECONDS);
137      this.reqHist.update(stats.getRequestSizeBytes());
138      this.respHist.update(stats.getResponseSizeBytes());
139    }
140
141    @Override
142    public String toString() {
143      return "CallTracker:" + name;
144    }
145  }

194  @VisibleForTesting
195  protected ConcurrentHashMap<ServerName, ConcurrentMap<byte[], RegionStats>> serverStats
196          = new ConcurrentHashMap<>();
197
198  public void updateServerStats(ServerName serverName, byte[] regionName,
199                                Object r) {
200    if (!(r instanceof Result)) {
201      return;
202    }
203    Result result = (Result) r;
204    RegionLoadStats stats = result.getStats();
205    if (stats == null) {
206      return;
207    }
208    updateRegionStats(serverName, regionName, stats);
209  }
210
211  @Override
212  public void updateRegionStats(ServerName serverName, byte[] regionName, RegionLoadStats stats) {
213    String name = serverName.getServerName() + "," + Bytes.toStringBinary(regionName);
214    ConcurrentMap<byte[], RegionStats> rsStats = computeIfAbsent(serverStats, serverName,
215      () -> new ConcurrentSkipListMap<>(Bytes.BYTES_COMPARATOR));
216    RegionStats regionStats =
217        computeIfAbsent(rsStats, regionName, () -> new RegionStats(this.registry, name));
218    regionStats.update(stats);
219  }

275  // These maps are used to cache references to the metric instances that are managed by the
276  // registry. I don't think their use perfectly removes redundant allocations, but it's
277  // a big improvement over calling registry.newMetric each time.
278  @VisibleForTesting protected final ConcurrentMap<String, Timer> rpcTimers =
279      new ConcurrentHashMap<>(CAPACITY, LOAD_FACTOR, CONCURRENCY_LEVEL);
280  @VisibleForTesting protected final ConcurrentMap<String, Histogram> rpcHistograms =
281      new ConcurrentHashMap<>(CAPACITY * 2 /* tracking both request and response sizes */,
282          LOAD_FACTOR, CONCURRENCY_LEVEL);
283  private final ConcurrentMap<String, Counter> cacheDroppingExceptions =
284    new ConcurrentHashMap<>(CAPACITY, LOAD_FACTOR, CONCURRENCY_LEVEL);

286  MetricsConnection(final ConnectionImplementation conn) {
287    this.scope = conn.toString();
288    this.registry = new MetricRegistry();
289
290    this.registry.register(getExecutorPoolName(),
291        new RatioGauge() {
292          @Override
293          protected Ratio getRatio() {
294            ThreadPoolExecutor batchPool = (ThreadPoolExecutor) conn.getCurrentBatchPool();
295            if (batchPool == null) {
296              return Ratio.of(0, 0);
297            }
298            return Ratio.of(batchPool.getActiveCount(), batchPool.getMaximumPoolSize());
299          }
300        });
301    this.registry.register(getMetaPoolName(),
302        new RatioGauge() {
303          @Override
304          protected Ratio getRatio() {
305            ThreadPoolExecutor metaPool = (ThreadPoolExecutor) conn.getCurrentMetaLookupPool();
306            if (metaPool == null) {
307              return Ratio.of(0, 0);
308            }
309            return Ratio.of(metaPool.getActiveCount(), metaPool.getMaximumPoolSize());
310          }
311        });
312    this.metaCacheHits = registry.counter(name(this.getClass(), "metaCacheHits", scope));
313    this.metaCacheMisses = registry.counter(name(this.getClass(), "metaCacheMisses", scope));
314    this.metaCacheNumClearServer = registry.counter(name(this.getClass(),
315      "metaCacheNumClearServer", scope));
316    this.metaCacheNumClearRegion = registry.counter(name(this.getClass(),
317      "metaCacheNumClearRegion", scope));
318    this.getTracker = new CallTracker(this.registry, "Get", scope);
319    this.scanTracker = new CallTracker(this.registry, "Scan", scope);
320    this.appendTracker = new CallTracker(this.registry, "Mutate", "Append", scope);
321    this.deleteTracker = new CallTracker(this.registry, "Mutate", "Delete", scope);
322    this.incrementTracker = new CallTracker(this.registry, "Mutate", "Increment", scope);
323    this.putTracker = new CallTracker(this.registry, "Mutate", "Put", scope);
324    this.multiTracker = new CallTracker(this.registry, "Multi", scope);
325    this.runnerStats = new RunnerStats(this.registry);
326
327    this.reporter = JmxReporter.forRegistry(this.registry).build();
328    this.reporter.start();
329  }

391  /**
392   * Get a metric for {@code key} from {@code map}, or create it with {@code factory}.
393   */
394  private <T> T getMetric(String key, ConcurrentMap<String, T> map, NewMetric<T> factory) {
395    return computeIfAbsent(map, key, () -> factory.newMetric(getClass(), key, scope));
396  }

398  /** Update call stats for non-critical-path methods */
399  private void updateRpcGeneric(MethodDescriptor method, CallStats stats) {
400    final String methodName = method.getService().getName() + "_" + method.getName();
401    getMetric(DRTN_BASE + methodName, rpcTimers, timerFactory)
402        .update(stats.getCallTimeMs(), TimeUnit.MILLISECONDS);
403    getMetric(REQ_BASE + methodName, rpcHistograms, histogramFactory)
404        .update(stats.getRequestSizeBytes());
405    getMetric(RESP_BASE + methodName, rpcHistograms, histogramFactory)
406        .update(stats.getResponseSizeBytes());
407  }

409  /** Report RPC context to metrics system. */
410  public void updateRpc(MethodDescriptor method, Message param, CallStats stats) {
411    // this implementation is tied directly to protobuf implementation details. would be better
412    // if we could dispatch based on something static, ie, request Message type.
413    if (method.getService() == ClientService.getDescriptor()) {
414      switch(method.getIndex()) {
415      case 0:
416        assert "Get".equals(method.getName());
417        getTracker.updateRpc(stats);
418        return;
419      case 1:
420        assert "Mutate".equals(method.getName());
421        final MutationType mutationType = ((MutateRequest) param).getMutation().getMutateType();
422        switch(mutationType) {
423        case APPEND:
424          appendTracker.updateRpc(stats);
425          return;
426        case DELETE:
427          deleteTracker.updateRpc(stats);
428          return;
429        case INCREMENT:
430          incrementTracker.updateRpc(stats);
431          return;
432        case PUT:
433          putTracker.updateRpc(stats);
434          return;
435        default:
436          throw new RuntimeException("Unrecognized mutation type " + mutationType);
437        }
438      case 2:
439        assert "Scan".equals(method.getName());
440        scanTracker.updateRpc(stats);
441        return;
442      case 3:
443        assert "BulkLoadHFile".equals(method.getName());
444        // use generic implementation
445        break;
446      case 4:
447        assert "PrepareBulkLoad".equals(method.getName());
448        // use generic implementation
449        break;
450      case 5:
451        assert "CleanupBulkLoad".equals(method.getName());
452        // use generic implementation
453        break;
454      case 6:
455        assert "ExecService".equals(method.getName());
456        // use generic implementation
457        break;
458      case 7:
459        assert "ExecRegionServerService".equals(method.getName());
460        // use generic implementation
461        break;
462      case 8:
463        assert "Multi".equals(method.getName());
464        multiTracker.updateRpc(stats);
465        return;
466      default:
467        throw new RuntimeException("Unrecognized ClientService RPC type " + method.getFullName());
468      }
469    }
470    // Fallback to dynamic registry lookup for DDL methods.
471    updateRpcGeneric(method, stats);
472  }

474  public void incrCacheDroppingExceptions(Object exception) {
475    getMetric(CACHE_BASE +
476      (exception == null? UNKNOWN_EXCEPTION : exception.getClass().getSimpleName()),
477      cacheDroppingExceptions, counterFactory).inc();
478  }
