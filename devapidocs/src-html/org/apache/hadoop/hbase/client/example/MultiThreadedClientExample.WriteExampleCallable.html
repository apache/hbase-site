056/**
057 * Example on how to use HBase's {@link Connection} and {@link Table} in a
058 * multi-threaded environment. Each table is a light weight object
059 * that is created and thrown away. Connections are heavy weight objects
060 * that hold on to zookeeper connections, async processes, and other state.
061 *
062 * <pre>
063 * Usage:
064 * bin/hbase org.apache.hadoop.hbase.client.example.MultiThreadedClientExample testTableName 500000
065 * </pre>
066 *
067 * <p>
068 * The table should already be created before running the command.
069 * This example expects one column family named d.
070 * </p>
071 * <p>
072 * This is meant to show different operations that are likely to be
073 * done in a real world application. These operations are:
074 * </p>
075 *
076 * <ul>
077 *   <li>
078 *     30% of all operations performed are batch writes.
079 *     30 puts are created and sent out at a time.
080 *     The response for all puts is waited on.
081 *   </li>
082 *   <li>
083 *     20% of all operations are single writes.
084 *     A single put is sent out and the response is waited for.
085 *   </li>
086 *   <li>
087 *     50% of all operations are scans.
088 *     These scans start at a random place and scan up to 100 rows.
089 *   </li>
090 * </ul>
091 *
092 */
093public class MultiThreadedClientExample extends Configured implements Tool {
094  private static final Logger LOG = LoggerFactory.getLogger(MultiThreadedClientExample.class);
095  private static final int DEFAULT_NUM_OPERATIONS = 500000;
096
097  /**
098   * The name of the column family.
099   *
100   * d for default.
101   */
102  private static final byte[] FAMILY = Bytes.toBytes("d");
103
104  /**
105   * For the example we're just using one qualifier.
106   */
107  private static final byte[] QUAL = Bytes.toBytes("test");
108
109  private final ExecutorService internalPool;
110
111  private final int threads;
112
113  public MultiThreadedClientExample() throws IOException {
114    // Base number of threads.
115    // This represents the number of threads you application has
116    // that can be interacting with an hbase client.
117    this.threads = Runtime.getRuntime().availableProcessors() * 4;
118
119    // Daemon threads are great for things that get shut down.
120    ThreadFactory threadFactory = new ThreadFactoryBuilder()
121        .setDaemon(true).setNameFormat("internal-pol-%d").build();
122
123
124    this.internalPool = Executors.newFixedThreadPool(threads, threadFactory);
125  }
126
127  @Override
128  public int run(String[] args) throws Exception {
129
130    if (args.length < 1 || args.length > 2) {
131      System.out.println("Usage: " + this.getClass().getName() + " tableName [num_operations]");
132      return -1;
133    }
134
135    final TableName tableName = TableName.valueOf(args[0]);
136    int numOperations = DEFAULT_NUM_OPERATIONS;
137
138    // the second arg is the number of operations to send.
139    if (args.length == 2) {
140      numOperations = Integer.parseInt(args[1]);
141    }
142
143    // Threads for the client only.
144    //
145    // We don't want to mix hbase and business logic.
146    //
147    ExecutorService service = new ForkJoinPool(threads * 2);
148
149    // Create two different connections showing how it's possible to
150    // separate different types of requests onto different connections
151    final Connection writeConnection = ConnectionFactory.createConnection(getConf(), service);
152    final Connection readConnection = ConnectionFactory.createConnection(getConf(), service);
153
154    // At this point the entire cache for the region locations is full.
155    // Only do this if the number of regions in a table is easy to fit into memory.
156    //
157    // If you are interacting with more than 25k regions on a client then it's probably not good
158    // to do this at all.
159    warmUpConnectionCache(readConnection, tableName);
160    warmUpConnectionCache(writeConnection, tableName);
161
162    List<Future<Boolean>> futures = new ArrayList<>(numOperations);
163    for (int i = 0; i < numOperations; i++) {
164      double r = ThreadLocalRandom.current().nextDouble();
165      Future<Boolean> f;
166
167      // For the sake of generating some synthetic load this queues
168      // some different callables.
169      // These callables are meant to represent real work done by your application.
170      if (r < .30) {
171        f = internalPool.submit(new WriteExampleCallable(writeConnection, tableName));
172      } else if (r < .50) {
173        f = internalPool.submit(new SingleWriteExampleCallable(writeConnection, tableName));
174      } else {
175        f = internalPool.submit(new ReadExampleCallable(writeConnection, tableName));
176      }
177      futures.add(f);
178    }
179
180    // Wait a long time for all the reads/writes to complete
181    for (Future<Boolean> f : futures) {
182      f.get(10, TimeUnit.MINUTES);
183    }
184
185    // Clean up after our selves for cleanliness
186    internalPool.shutdownNow();
187    service.shutdownNow();
188    return 0;
189  }
190
191  private void warmUpConnectionCache(Connection connection, TableName tn) throws IOException {
192    try (RegionLocator locator = connection.getRegionLocator(tn)) {
193      LOG.info(
194          "Warmed up region location cache for " + tn
195              + " got " + locator.getAllRegionLocations().size());
196    }
197  }

199  /**
200   * Class that will show how to send batches of puts at the same time.
201   */
202  public static class WriteExampleCallable implements Callable<Boolean> {
203    private final Connection connection;
204    private final TableName tableName;
205
206    public WriteExampleCallable(Connection connection, TableName tableName) {
207      this.connection = connection;
208      this.tableName = tableName;
209    }
210
211    @Override
212    public Boolean call() throws Exception {
213
214      // Table implements Closable so we use the try with resource structure here.
215      // https://docs.oracle.com/javase/tutorial/essential/exceptions/tryResourceClose.html
216      try (Table t = connection.getTable(tableName)) {
217        byte[] value = Bytes.toBytes(Double.toString(ThreadLocalRandom.current().nextDouble()));
218        int rows = 30;
219
220        // Array to put the batch
221        ArrayList<Put> puts = new ArrayList<>(rows);
222        for (int i = 0; i < 30; i++) {
223          byte[] rk = Bytes.toBytes(ThreadLocalRandom.current().nextLong());
224          Put p = new Put(rk);
225          p.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
226                .setRow(rk)
227                .setFamily(FAMILY)
228                .setQualifier(QUAL)
229                .setTimestamp(p.getTimeStamp())
230                .setType(Cell.Type.Put)
231                .setValue(value)
232                .build());
233          puts.add(p);
234        }
235
236        // now that we've assembled the batch it's time to push it to hbase.
237        t.put(puts);
238      }
239      return true;
240    }
241  }

243  /**
244   * Class to show how to send a single put.
245   */
246  public static class SingleWriteExampleCallable implements Callable<Boolean> {
247    private final Connection connection;
248    private final TableName tableName;
249
250    public SingleWriteExampleCallable(Connection connection, TableName tableName) {
251      this.connection = connection;
252      this.tableName = tableName;
253    }
254
255    @Override
256    public Boolean call() throws Exception {
257      try (Table t = connection.getTable(tableName)) {
258
259        byte[] value = Bytes.toBytes(Double.toString(ThreadLocalRandom.current().nextDouble()));
260        byte[] rk = Bytes.toBytes(ThreadLocalRandom.current().nextLong());
261        Put p = new Put(rk);
262        p.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
263                .setRow(rk)
264                .setFamily(FAMILY)
265                .setQualifier(QUAL)
266                .setTimestamp(p.getTimeStamp())
267                .setType(Type.Put)
268                .setValue(value)
269                .build());
270        t.put(p);
271      }
272      return true;
273    }
274  }

277  /**
278   * Class to show how to scan some rows starting at a random location.
279   */
280  public static class ReadExampleCallable implements Callable<Boolean> {
281    private final Connection connection;
282    private final TableName tableName;
283
284    public ReadExampleCallable(Connection connection, TableName tableName) {
285      this.connection = connection;
286      this.tableName = tableName;
287    }
288
289    @Override
290    public Boolean call() throws Exception {
291
292      // total length in bytes of all read rows.
293      int result = 0;
294
295      // Number of rows the scan will read before being considered done.
296      int toRead = 100;
297      try (Table t = connection.getTable(tableName)) {
298        byte[] rk = Bytes.toBytes(ThreadLocalRandom.current().nextLong());
299        Scan s = new Scan(rk);
300
301        // This filter will keep the values from being sent accross the wire.
302        // This is good for counting or other scans that are checking for
303        // existence and don't rely on the value.
304        s.setFilter(new KeyOnlyFilter());
305
306        // Don't go back to the server for every single row.
307        // We know these rows are small. So ask for 20 at a time.
308        // This would be application specific.
309        //
310        // The goal is to reduce round trips but asking for too
311        // many rows can lead to GC problems on client and server sides.
312        s.setCaching(20);
313
314        // Don't use the cache. While this is a silly test program it's still good to be
315        // explicit that scans normally don't use the block cache.
316        s.setCacheBlocks(false);
317
318        // Open up the scanner and close it automatically when done.
319        try (ResultScanner rs = t.getScanner(s)) {
320
321          // Now go through rows.
322          for (Result r : rs) {
323            // Keep track of things size to simulate doing some real work.
324            result += r.getRow().length;
325            toRead -= 1;
326
327            // Most online applications won't be
328            // reading the entire table so this break
329            // simulates small to medium size scans,
330            // without needing to know an end row.
331            if (toRead <= 0)  {
332              break;
333            }
334          }
335        }
336      }
337      return result > 0;
338    }
339  }
340
341  public static void main(String[] args) throws Exception {
342    ToolRunner.run(new MultiThreadedClientExample(), args);
343  }
