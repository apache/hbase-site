534  /**
535   * Sets replication scope all & only the columns already in the builder. Columns added later won't
536   * be backfilled with replication scope.
537   * @param scope replication scope
538   * @return a TableDescriptorBuilder
539   */
540  public TableDescriptorBuilder setReplicationScope(int scope) {
541    Map<byte[], ColumnFamilyDescriptor> newFamilies = new TreeMap<>(Bytes.BYTES_RAWCOMPARATOR);
542    newFamilies.putAll(desc.families);
543    newFamilies
544        .forEach((cf, cfDesc) -> {
545          desc.removeColumnFamily(cf);
546          desc.setColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(cfDesc).setScope(scope)
547              .build());
548        });
549    return this;
550  }

561  /**
562   * TODO: make this private after removing the HTableDescriptor
563   */
564  @InterfaceAudience.Private
565  public static class ModifyableTableDescriptor
566          implements TableDescriptor, Comparable<ModifyableTableDescriptor> {
567
568    private final TableName name;
569
570    /**
571     * A map which holds the metadata information of the table. This metadata
572     * includes values like IS_META, SPLIT_POLICY, MAX_FILE_SIZE,
573     * READONLY, MEMSTORE_FLUSHSIZE etc...
574     */
575    private final Map<Bytes, Bytes> values = new HashMap<>();
576
577    /**
578     * Maps column family name to the respective FamilyDescriptors
579     */
580    private final Map<byte[], ColumnFamilyDescriptor> families
581            = new TreeMap<>(Bytes.BYTES_RAWCOMPARATOR);
582
583    /**
584     * Construct a table descriptor specifying a TableName object
585     *
586     * @param name Table name.
587     * TODO: make this private after removing the HTableDescriptor
588     */
589    @InterfaceAudience.Private
590    public ModifyableTableDescriptor(final TableName name) {
591      this(name, Collections.EMPTY_LIST, Collections.EMPTY_MAP);
592    }
593
594    private ModifyableTableDescriptor(final TableDescriptor desc) {
595      this(desc.getTableName(), Arrays.asList(desc.getColumnFamilies()), desc.getValues());
596    }
597
598    /**
599     * Construct a table descriptor by cloning the descriptor passed as a
600     * parameter.
601     * <p>
602     * Makes a deep copy of the supplied descriptor.
603     * @param name The new name
604     * @param desc The descriptor.
605     * TODO: make this private after removing the HTableDescriptor
606     */
607    @InterfaceAudience.Private
608    @Deprecated // only used by HTableDescriptor. remove this method if HTD is removed
609    public ModifyableTableDescriptor(final TableName name, final TableDescriptor desc) {
610      this(name, Arrays.asList(desc.getColumnFamilies()), desc.getValues());
611    }
612
613    private ModifyableTableDescriptor(final TableName name, final Collection<ColumnFamilyDescriptor> families,
614            Map<Bytes, Bytes> values) {
615      this.name = name;
616      families.forEach(c -> this.families.put(c.getName(), ColumnFamilyDescriptorBuilder.copy(c)));
617      this.values.putAll(values);
618      this.values.put(IS_META_KEY,
619        new Bytes(Bytes.toBytes(Boolean.toString(name.equals(TableName.META_TABLE_NAME)))));
620    }
621
622    /**
623     * Checks if this table is <code> hbase:meta </code> region.
624     *
625     * @return true if this table is <code> hbase:meta </code> region
626     */
627    @Override
628    public boolean isMetaRegion() {
629      return getOrDefault(IS_META_KEY, Boolean::valueOf, false);
630    }
631
632    /**
633     * Checks if the table is a <code>hbase:meta</code> table
634     *
635     * @return true if table is <code> hbase:meta </code> region.
636     */
637    @Override
638    public boolean isMetaTable() {
639      return isMetaRegion();
640    }
641
642    @Override
643    public Bytes getValue(Bytes key) {
644      Bytes rval = values.get(key);
645      return rval == null ? null : new Bytes(rval.copyBytes());
646    }
647
648    @Override
649    public String getValue(String key) {
650      Bytes rval = values.get(new Bytes(Bytes.toBytes(key)));
651      return rval == null ? null : Bytes.toString(rval.get(), rval.getOffset(), rval.getLength());
652    }
653
654    @Override
655    public byte[] getValue(byte[] key) {
656      Bytes value = values.get(new Bytes(key));
657      return value == null ? null : value.copyBytes();
658    }
659
660    private <T> T getOrDefault(Bytes key, Function<String, T> function, T defaultValue) {
661      Bytes value = values.get(key);
662      if (value == null) {
663        return defaultValue;
664      } else {
665        return function.apply(Bytes.toString(value.get(), value.getOffset(), value.getLength()));
666      }
667    }

1150    private StringBuilder getValues(boolean printDefaults) {
1151      StringBuilder s = new StringBuilder();
1152
1153      // step 1: set partitioning and pruning
1154      Set<Bytes> reservedKeys = new TreeSet<>();
1155      Set<Bytes> userKeys = new TreeSet<>();
1156      for (Map.Entry<Bytes, Bytes> entry : values.entrySet()) {
1157        if (entry.getKey() == null || entry.getKey().get() == null) {
1158          continue;
1159        }
1160        String key = Bytes.toString(entry.getKey().get());
1161        // in this section, print out reserved keywords + coprocessor info
1162        if (!RESERVED_KEYWORDS.contains(entry.getKey()) && !key.startsWith("coprocessor$")) {
1163          userKeys.add(entry.getKey());
1164          continue;
1165        }
1166        // only print out IS_META if true
1167        String value = Bytes.toString(entry.getValue().get());
1168        if (key.equalsIgnoreCase(IS_META)) {
1169          if (Boolean.valueOf(value) == false) {
1170            continue;
1171          }
1172        }
1173        // see if a reserved key is a default value. may not want to print it out
1174        if (printDefaults
1175                || !DEFAULT_VALUES.containsKey(key)
1176                || !DEFAULT_VALUES.get(key).equalsIgnoreCase(value)) {
1177          reservedKeys.add(entry.getKey());
1178        }
1179      }
1180
1181      // early exit optimization
1182      boolean hasAttributes = !reservedKeys.isEmpty() || !userKeys.isEmpty();
1183      if (!hasAttributes) {
1184        return s;
1185      }
1186
1187      s.append(", {");
1188      // step 2: printing attributes
1189      if (hasAttributes) {
1190        s.append("TABLE_ATTRIBUTES => {");
1191
1192        // print all reserved keys first
1193        boolean printCommaForAttr = false;
1194        for (Bytes k : reservedKeys) {
1195          String key = Bytes.toString(k.get());
1196          String value = Bytes.toStringBinary(values.get(k).get());
1197          if (printCommaForAttr) {
1198            s.append(", ");
1199          }
1200          printCommaForAttr = true;
1201          s.append(key);
1202          s.append(" => ");
1203          s.append('\'').append(value).append('\'');
1204        }
1205
1206        if (!userKeys.isEmpty()) {
1207          // print all non-reserved as a separate subset
1208          if (printCommaForAttr) {
1209            s.append(", ");
1210          }
1211          s.append(HConstants.METADATA).append(" => ");
1212          s.append("{");
1213          boolean printCommaForCfg = false;
1214          for (Bytes k : userKeys) {
1215            String key = Bytes.toString(k.get());
1216            String value = Bytes.toStringBinary(values.get(k).get());
1217            if (printCommaForCfg) {
1218              s.append(", ");
1219            }
1220            printCommaForCfg = true;
1221            s.append('\'').append(key).append('\'');
1222            s.append(" => ");
1223            s.append('\'').append(value).append('\'');
1224          }
1225          s.append("}");
1226        }
1227
1228        s.append("}");
1229      }
1230
1231      s.append("}"); // end METHOD
1232      return s;
1233    }

1398    /**
1399     * Add a table coprocessor to this table. The coprocessor type must be
1400     * org.apache.hadoop.hbase.coprocessor.RegionObserver or Endpoint. It won't
1401     * check if the class can be loaded or not. Whether a coprocessor is
1402     * loadable or not will be determined when a region is opened.
1403     *
1404     * @throws IOException any illegal parameter key/value
1405     * @return the modifyable TD
1406     */
1407    public ModifyableTableDescriptor setCoprocessor(CoprocessorDescriptor cp)
1408            throws IOException {
1409      checkHasCoprocessor(cp.getClassName());
1410      if (cp.getPriority() < 0) {
1411        throw new IOException("Priority must be bigger than or equal with zero, current:"
1412          + cp.getPriority());
1413      }
1414      // Validate parameter kvs and then add key/values to kvString.
1415      StringBuilder kvString = new StringBuilder();
1416      for (Map.Entry<String, String> e : cp.getProperties().entrySet()) {
1417        if (!e.getKey().matches(CP_HTD_ATTR_VALUE_PARAM_KEY_PATTERN)) {
1418          throw new IOException("Illegal parameter key = " + e.getKey());
1419        }
1420        if (!e.getValue().matches(CP_HTD_ATTR_VALUE_PARAM_VALUE_PATTERN)) {
1421          throw new IOException("Illegal parameter (" + e.getKey()
1422                  + ") value = " + e.getValue());
1423        }
1424        if (kvString.length() != 0) {
1425          kvString.append(',');
1426        }
1427        kvString.append(e.getKey());
1428        kvString.append('=');
1429        kvString.append(e.getValue());
1430      }
1431
1432      String value = cp.getJarPath().orElse("")
1433              + "|" + cp.getClassName() + "|" + Integer.toString(cp.getPriority()) + "|"
1434              + kvString.toString();
1435      return setCoprocessorToMap(value);
1436    }

1466    /**
1467     * Add coprocessor to values Map
1468     * @param specStr The Coprocessor specification all in in one String
1469     * @return Returns <code>this</code>
1470     */
1471    private ModifyableTableDescriptor setCoprocessorToMap(final String specStr) {
1472      if (specStr == null) {
1473        return this;
1474      }
1475      // generate a coprocessor key
1476      int maxCoprocessorNumber = 0;
1477      Matcher keyMatcher;
1478      for (Map.Entry<Bytes, Bytes> e : this.values.entrySet()) {
1479        keyMatcher = CP_HTD_ATTR_KEY_PATTERN.matcher(Bytes.toString(e.getKey().get()));
1480        if (!keyMatcher.matches()) {
1481          continue;
1482        }
1483        maxCoprocessorNumber = Math.max(Integer.parseInt(keyMatcher.group(1)), maxCoprocessorNumber);
1484      }
1485      maxCoprocessorNumber++;
1486      String key = "coprocessor$" + Integer.toString(maxCoprocessorNumber);
1487      return setValue(new Bytes(Bytes.toBytes(key)), new Bytes(Bytes.toBytes(specStr)));
1488    }

1505     * className
1506     *
1507     * @return The list of co-processors classNames
1508     */
1509    @Override
1510    public List<CoprocessorDescriptor> getCoprocessorDescriptors() {
1511      List<CoprocessorDescriptor> result = new ArrayList<>();
1512      for (Map.Entry<Bytes, Bytes> e: getValues().entrySet()) {
1513        String key = Bytes.toString(e.getKey().get()).trim();
1514        if (CP_HTD_ATTR_KEY_PATTERN.matcher(key).matches()) {
1515          toCoprocessorDescriptor(Bytes.toString(e.getValue().get()).trim())
1516            .ifPresent(result::add);
1517        }
1518      }
1519      return result;
1520    }

1522    /**
1523     * Remove a coprocessor from those set on the table
1524     *
1525     * @param className Class name of the co-processor
1526     */
1527    public void removeCoprocessor(String className) {
1528      Bytes match = null;
1529      Matcher keyMatcher;
1530      Matcher valueMatcher;
1531      for (Map.Entry<Bytes, Bytes> e : this.values
1532              .entrySet()) {
1533        keyMatcher = CP_HTD_ATTR_KEY_PATTERN.matcher(Bytes.toString(e
1534                .getKey().get()));
1535        if (!keyMatcher.matches()) {
1536          continue;
1537        }
1538        valueMatcher = CP_HTD_ATTR_VALUE_PATTERN.matcher(Bytes
1539                .toString(e.getValue().get()));
1540        if (!valueMatcher.matches()) {
1541          continue;
1542        }
1543        // get className and compare
1544        String clazz = valueMatcher.group(2).trim(); // classname is the 2nd field
1545        // remove the CP if it is present
1546        if (clazz.equals(className.trim())) {
1547          match = e.getKey();
1548          break;
1549        }
1550      }
1551      // if we found a match, remove it
1552      if (match != null) {
1553        ModifyableTableDescriptor.this.removeValue(match);
1554      }
1555    }

1596    /**
1597     * @param bytes A pb serialized {@link ModifyableTableDescriptor} instance
1598     * with pb magic prefix
1599     * @return An instance of {@link ModifyableTableDescriptor} made from
1600     * <code>bytes</code>
1601     * @throws DeserializationException
1602     * @see #toByteArray()
1603     */
1604    private static TableDescriptor parseFrom(final byte[] bytes)
1605            throws DeserializationException {
1606      if (!ProtobufUtil.isPBMagicPrefix(bytes)) {
1607        throw new DeserializationException("Expected PB encoded ModifyableTableDescriptor");
1608      }
1609      int pblen = ProtobufUtil.lengthOfPBMagic();
1610      HBaseProtos.TableSchema.Builder builder = HBaseProtos.TableSchema.newBuilder();
1611      try {
1612        ProtobufUtil.mergeFrom(builder, bytes, pblen, bytes.length - pblen);
1613        return ProtobufUtil.toTableDescriptor(builder.build());
1614      } catch (IOException e) {
1615        throw new DeserializationException(e);
1616      }
1617    }

1635  private static Optional<CoprocessorDescriptor> toCoprocessorDescriptor(String spec) {
1636    Matcher matcher = CP_HTD_ATTR_VALUE_PATTERN.matcher(spec);
1637    if (matcher.matches()) {
1638      // jar file path can be empty if the cp class can be loaded
1639      // from class loader.
1640      String path = matcher.group(1).trim().isEmpty() ?
1641        null : matcher.group(1).trim();
1642      String className = matcher.group(2).trim();
1643      if (className.isEmpty()) {
1644        return Optional.empty();
1645      }
1646      String priorityStr = matcher.group(3).trim();
1647      int priority = priorityStr.isEmpty() ?
1648        Coprocessor.PRIORITY_USER : Integer.parseInt(priorityStr);
1649      String cfgSpec = null;
1650      try {
1651        cfgSpec = matcher.group(4);
1652      } catch (IndexOutOfBoundsException ex) {
1653        // ignore
1654      }
1655      Map<String, String> ourConf = new TreeMap<>();
1656      if (cfgSpec != null && !cfgSpec.trim().equals("|")) {
1657        cfgSpec = cfgSpec.substring(cfgSpec.indexOf('|') + 1);
1658        Matcher m = CP_HTD_ATTR_VALUE_PARAM_PATTERN.matcher(cfgSpec);
1659        while (m.find()) {
1660          ourConf.put(m.group(1), m.group(2));
1661        }
1662      }
1663      return Optional.of(CoprocessorDescriptorBuilder.newBuilder(className)
1664        .setJarPath(path)
1665        .setPriority(priority)
1666        .setProperties(ourConf)
1667        .build());
1668    }
1669    return Optional.empty();
1670  }
