039/**
040 * Factory to create an AsyncRpcRetryCaller.
041 */
042@InterfaceAudience.Private
043class AsyncRpcRetryingCallerFactory {
044
045  private final AsyncConnectionImpl conn;
046
047  private final HashedWheelTimer retryTimer;
048
049  public AsyncRpcRetryingCallerFactory(AsyncConnectionImpl conn, HashedWheelTimer retryTimer) {
050    this.conn = conn;
051    this.retryTimer = retryTimer;
052  }
053
054  private abstract class BuilderBase {
055
056    protected long pauseNs = conn.connConf.getPauseNs();
057
058    protected int maxAttempts = retries2Attempts(conn.connConf.getMaxRetries());
059
060    protected int startLogErrorsCnt = conn.connConf.getStartLogErrorsCnt();
061  }
062
063  public class SingleRequestCallerBuilder<T> extends BuilderBase {
064
065    private TableName tableName;
066
067    private byte[] row;
068
069    private AsyncSingleRequestRpcRetryingCaller.Callable<T> callable;
070
071    private long operationTimeoutNs = -1L;
072
073    private long rpcTimeoutNs = -1L;
074
075    private RegionLocateType locateType = RegionLocateType.CURRENT;
076
077    public SingleRequestCallerBuilder<T> table(TableName tableName) {
078      this.tableName = tableName;
079      return this;
080    }
081
082    public SingleRequestCallerBuilder<T> row(byte[] row) {
083      this.row = row;
084      return this;
085    }
086
087    public SingleRequestCallerBuilder<T> action(
088        AsyncSingleRequestRpcRetryingCaller.Callable<T> callable) {
089      this.callable = callable;
090      return this;
091    }
092
093    public SingleRequestCallerBuilder<T> operationTimeout(long operationTimeout, TimeUnit unit) {
094      this.operationTimeoutNs = unit.toNanos(operationTimeout);
095      return this;
096    }
097
098    public SingleRequestCallerBuilder<T> rpcTimeout(long rpcTimeout, TimeUnit unit) {
099      this.rpcTimeoutNs = unit.toNanos(rpcTimeout);
100      return this;
101    }
102
103    public SingleRequestCallerBuilder<T> locateType(RegionLocateType locateType) {
104      this.locateType = locateType;
105      return this;
106    }
107
108    public SingleRequestCallerBuilder<T> pause(long pause, TimeUnit unit) {
109      this.pauseNs = unit.toNanos(pause);
110      return this;
111    }
112
113    public SingleRequestCallerBuilder<T> maxAttempts(int maxAttempts) {
114      this.maxAttempts = maxAttempts;
115      return this;
116    }
117
118    public SingleRequestCallerBuilder<T> startLogErrorsCnt(int startLogErrorsCnt) {
119      this.startLogErrorsCnt = startLogErrorsCnt;
120      return this;
121    }
122
123    public AsyncSingleRequestRpcRetryingCaller<T> build() {
124      return new AsyncSingleRequestRpcRetryingCaller<>(retryTimer, conn,
125          checkNotNull(tableName, "tableName is null"), checkNotNull(row, "row is null"),
126          checkNotNull(locateType, "locateType is null"), checkNotNull(callable, "action is null"),
127          pauseNs, maxAttempts, operationTimeoutNs, rpcTimeoutNs, startLogErrorsCnt);
128    }
129
130    /**
131     * Shortcut for {@code build().call()}
132     */
133    public CompletableFuture<T> call() {
134      return build().call();
135    }
136  }
137
138  /**
139   * Create retry caller for single action, such as get, put, delete, etc.
140   */
141  public <T> SingleRequestCallerBuilder<T> single() {
142    return new SingleRequestCallerBuilder<>();
143  }

194      this.scanTimeoutNs = unit.toNanos(scanTimeout);
195      return this;
196    }
197
198    public ScanSingleRegionCallerBuilder rpcTimeout(long rpcTimeout, TimeUnit unit) {
199      this.rpcTimeoutNs = unit.toNanos(rpcTimeout);
200      return this;

214      this.startLogErrorsCnt = startLogErrorsCnt;
215      return this;
216    }
217
218    public AsyncScanSingleRegionRpcRetryingCaller build() {
219      checkArgument(scannerId >= 0, "invalid scannerId %d", scannerId);

222          checkNotNull(resultCache, "resultCache is null"),

224          checkNotNull(loc, "location is null"), pauseNs, maxAttempts, scanTimeoutNs, rpcTimeoutNs,

230     */
231    public CompletableFuture<Boolean> start(HBaseRpcController controller,
232        ScanResponse respWhenOpen) {
233      return build().start(controller, respWhenOpen);
234    }
235  }
