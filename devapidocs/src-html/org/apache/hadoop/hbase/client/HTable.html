129  @InterfaceAudience.Private
130  public static ThreadPoolExecutor getDefaultExecutor(Configuration conf) {
131    int maxThreads = conf.getInt("hbase.htable.threads.max", Integer.MAX_VALUE);
132    if (maxThreads == 0) {
133      maxThreads = 1; // is there a better default?
134    }
135    int corePoolSize = conf.getInt("hbase.htable.threads.coresize", 1);
136    long keepAliveTime = conf.getLong("hbase.htable.threads.keepalivetime", 60);
137
138    // Using the "direct handoff" approach, new threads will only be created
139    // if it is necessary and will grow unbounded. This could be bad but in HCM
140    // we only create as many Runnables as there are region servers. It means
141    // it also scales when new region servers are added.
142    ThreadPoolExecutor pool = new ThreadPoolExecutor(corePoolSize, maxThreads, keepAliveTime,
143      TimeUnit.SECONDS, new SynchronousQueue<>(), Threads.newDaemonThreadFactory("htable"));
144    pool.allowCoreThreadTimeOut(true);
145    return pool;
146  }

148  /**
149   * Creates an object to access a HBase table.
150   * Used by HBase internally.  DO NOT USE. See {@link ConnectionFactory} class comment for how to
151   * get a {@link Table} instance (use {@link Table} instead of {@link HTable}).
152   * @param connection Connection to be used.
153   * @param builder The table builder
154   * @param rpcCallerFactory The RPC caller factory
155   * @param rpcControllerFactory The RPC controller factory
156   * @param pool ExecutorService to be used.
157   */
158  @InterfaceAudience.Private
159  protected HTable(final ClusterConnection connection,
160      final TableBuilderBase builder,
161      final RpcRetryingCallerFactory rpcCallerFactory,
162      final RpcControllerFactory rpcControllerFactory,
163      final ExecutorService pool) {
164    if (connection == null || connection.isClosed()) {
165      throw new IllegalArgumentException("Connection is null or closed.");
166    }
167    this.connection = connection;
168    this.configuration = connection.getConfiguration();
169    this.connConfiguration = connection.getConnectionConfiguration();
170    if (pool == null) {
171      this.pool = getDefaultExecutor(this.configuration);
172      this.cleanupPoolOnClose = true;
173    } else {
174      this.pool = pool;
175      this.cleanupPoolOnClose = false;
176    }
177    if (rpcCallerFactory == null) {
178      this.rpcCallerFactory = connection.getNewRpcRetryingCallerFactory(configuration);
179    } else {
180      this.rpcCallerFactory = rpcCallerFactory;
181    }
182
183    if (rpcControllerFactory == null) {
184      this.rpcControllerFactory = RpcControllerFactory.instantiate(configuration);
185    } else {
186      this.rpcControllerFactory = rpcControllerFactory;
187    }
188
189    this.tableName = builder.tableName;
190    this.operationTimeout = builder.operationTimeout;
191    this.rpcTimeout = builder.rpcTimeout;
192    this.readRpcTimeout = builder.readRpcTimeout;
193    this.writeRpcTimeout = builder.writeRpcTimeout;
194    this.scannerCaching = connConfiguration.getScannerCaching();
195    this.scannerMaxResultSize = connConfiguration.getScannerMaxResultSize();
196
197    // puts need to track errors globally due to how the APIs currently work.
198    multiAp = this.connection.getAsyncProcess();
199    this.locator = new HRegionLocator(tableName, connection);
200  }

268  /**
269   * Get the corresponding start keys and regions for an arbitrary range of
270   * keys.
271   * <p>
272   * @param startKey Starting row in range, inclusive
273   * @param endKey Ending row in range
274   * @param includeEndKey true if endRow is inclusive, false if exclusive
275   * @param reload true to reload information or false to use cached information
276   * @return A pair of list of start keys and list of HRegionLocations that
277   *         contain the specified range
278   * @throws IOException if a remote or network exception occurs
279   */
280  private Pair<List<byte[]>, List<HRegionLocation>> getKeysAndRegionsInRange(
281      final byte[] startKey, final byte[] endKey, final boolean includeEndKey,
282      final boolean reload) throws IOException {
283    final boolean endKeyIsEndOfTable = Bytes.equals(endKey,HConstants.EMPTY_END_ROW);
284    if ((Bytes.compareTo(startKey, endKey) > 0) && !endKeyIsEndOfTable) {
285      throw new IllegalArgumentException(
286        "Invalid range: " + Bytes.toStringBinary(startKey) +
287        " > " + Bytes.toStringBinary(endKey));
288    }
289    List<byte[]> keysInRange = new ArrayList<>();
290    List<HRegionLocation> regionsInRange = new ArrayList<>();
291    byte[] currentKey = startKey;
292    do {
293      HRegionLocation regionLocation = getRegionLocator().getRegionLocation(currentKey, reload);
294      keysInRange.add(currentKey);
295      regionsInRange.add(regionLocation);
296      currentKey = regionLocation.getRegionInfo().getEndKey();
297    } while (!Bytes.equals(currentKey, HConstants.EMPTY_END_ROW)
298        && (endKeyIsEndOfTable || Bytes.compareTo(currentKey, endKey) < 0
299            || (includeEndKey && Bytes.compareTo(currentKey, endKey) == 0)));
300    return new Pair<>(keysInRange, regionsInRange);
301  }

303  /**
304   * The underlying {@link HTable} must not be closed.
305   * {@link Table#getScanner(Scan)} has other usage details.
306   */
307  @Override
308  public ResultScanner getScanner(Scan scan) throws IOException {
309    if (scan.getCaching() <= 0) {
310      scan.setCaching(scannerCaching);
311    }
312    if (scan.getMaxResultSize() <= 0) {
313      scan.setMaxResultSize(scannerMaxResultSize);
314    }
315    if (scan.getMvccReadPoint() > 0) {
316      // it is not supposed to be set by user, clear
317      scan.resetMvccReadPoint();
318    }
319    Boolean async = scan.isAsyncPrefetch();
320    if (async == null) {
321      async = connConfiguration.isClientScannerAsyncPrefetch();
322    }
323
324    if (scan.isReversed()) {
325      return new ReversedClientScanner(getConfiguration(), scan, getName(),
326        this.connection, this.rpcCallerFactory, this.rpcControllerFactory,
327        pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
328    } else {
329      if (async) {
330        return new ClientAsyncPrefetchScanner(getConfiguration(), scan, getName(), this.connection,
331            this.rpcCallerFactory, this.rpcControllerFactory,
332            pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
333      } else {
334        return new ClientSimpleScanner(getConfiguration(), scan, getName(), this.connection,
335            this.rpcCallerFactory, this.rpcControllerFactory,
336            pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
337      }
338    }
339  }

372  private Result get(Get get, final boolean checkExistenceOnly) throws IOException {
373    // if we are changing settings to the get, clone it.
374    if (get.isCheckExistenceOnly() != checkExistenceOnly || get.getConsistency() == null) {
375      get = ReflectionUtils.newInstance(get.getClass(), get);
376      get.setCheckExistenceOnly(checkExistenceOnly);
377      if (get.getConsistency() == null){
378        get.setConsistency(DEFAULT_CONSISTENCY);
379      }
380    }
381
382    if (get.getConsistency() == Consistency.STRONG) {
383      final Get configuredGet = get;
384      ClientServiceCallable<Result> callable = new ClientServiceCallable<Result>(this.connection, getName(),
385          get.getRow(), this.rpcControllerFactory.newController(), get.getPriority()) {
386        @Override
387        protected Result rpcCall() throws Exception {
388          ClientProtos.GetRequest request = RequestConverter.buildGetRequest(
389              getLocation().getRegionInfo().getRegionName(), configuredGet);
390          ClientProtos.GetResponse response = doGet(request);
391          return response == null? null:
392            ProtobufUtil.toResult(response.getResult(), getRpcControllerCellScanner());
393        }
394      };
395      return rpcCallerFactory.<Result>newCaller(readRpcTimeout).callWithRetries(callable,
396          this.operationTimeout);
397    }
398
399    // Call that takes into account the replica
400    RpcRetryingCallerWithReadReplicas callable = new RpcRetryingCallerWithReadReplicas(
401        rpcControllerFactory, tableName, this.connection, get, pool,
402        connConfiguration.getRetriesNumber(),
403        operationTimeout, readRpcTimeout,
404        connConfiguration.getPrimaryCallTimeoutMicroSecond());
405    return callable.call(operationTimeout);
406  }

409   * {@inheritDoc}
410   */
411  @Override
412  public Result[] get(List<Get> gets) throws IOException {
413    if (gets.size() == 1) {
414      return new Result[]{get(gets.get(0))};
415    }
416    try {
417      Object[] r1 = new Object[gets.size()];
418      batch((List<? extends Row>)gets, r1, readRpcTimeout);
419      // Translate.
420      Result [] results = new Result[r1.length];
421      int i = 0;
422      for (Object obj: r1) {
423        // Batch ensures if there is a failure we get an exception instead
424        results[i++] = (Result)obj;
425      }
426      return results;
427    } catch (InterruptedException e) {
428      throw (InterruptedIOException)new InterruptedIOException().initCause(e);
429    }
430  }

432  /**
433   * {@inheritDoc}
434   */
435  @Override
436  public void batch(final List<? extends Row> actions, final Object[] results)
437      throws InterruptedException, IOException {
438    int rpcTimeout = writeRpcTimeout;
439    boolean hasRead = false;
440    boolean hasWrite = false;
441    for (Row action : actions) {
442      if (action instanceof Mutation) {
443        hasWrite = true;
444      } else {
445        hasRead = true;
446      }
447      if (hasRead && hasWrite) {
448        break;
449      }
450    }
451    if (hasRead && !hasWrite) {
452      rpcTimeout = readRpcTimeout;
453    }
454    batch(actions, results, rpcTimeout);
455  }

457  public void batch(final List<? extends Row> actions, final Object[] results, int rpcTimeout)
458      throws InterruptedException, IOException {
459    AsyncProcessTask task = AsyncProcessTask.newBuilder()
460            .setPool(pool)
461            .setTableName(tableName)
462            .setRowAccess(actions)
463            .setResults(results)
464            .setRpcTimeout(rpcTimeout)
465            .setOperationTimeout(operationTimeout)
466            .setSubmittedRows(AsyncProcessTask.SubmittedRows.ALL)
467            .build();
468    AsyncRequestFuture ars = multiAp.submit(task);
469    ars.waitUntilDone();
470    if (ars.hasError()) {
471      throw ars.getErrors();
472    }
473  }

485  public static <R> void doBatchWithCallback(List<? extends Row> actions, Object[] results,
486    Callback<R> callback, ClusterConnection connection, ExecutorService pool, TableName tableName)
487    throws InterruptedIOException, RetriesExhaustedWithDetailsException {
488    int operationTimeout = connection.getConnectionConfiguration().getOperationTimeout();
489    int writeTimeout = connection.getConfiguration().getInt(HConstants.HBASE_RPC_WRITE_TIMEOUT_KEY,
490        connection.getConfiguration().getInt(HConstants.HBASE_RPC_TIMEOUT_KEY,
491            HConstants.DEFAULT_HBASE_RPC_TIMEOUT));
492    AsyncProcessTask<R> task = AsyncProcessTask.newBuilder(callback)
493            .setPool(pool)
494            .setTableName(tableName)
495            .setRowAccess(actions)
496            .setResults(results)
497            .setOperationTimeout(operationTimeout)
498            .setRpcTimeout(writeTimeout)
499            .setSubmittedRows(AsyncProcessTask.SubmittedRows.ALL)
500            .build();
501    AsyncRequestFuture ars = connection.getAsyncProcess().submit(task);
502    ars.waitUntilDone();
503    if (ars.hasError()) {
504      throw ars.getErrors();
505    }

543  /**
544   * {@inheritDoc}
545   */
546  @Override
547  public void delete(final List<Delete> deletes)
548  throws IOException {
549    Object[] results = new Object[deletes.size()];
550    try {
551      batch(deletes, results, writeRpcTimeout);
552    } catch (InterruptedException e) {
553      throw (InterruptedIOException)new InterruptedIOException().initCause(e);
554    } finally {
555      // mutate list so that it is empty for complete success, or contains only failed records
556      // results are returned in the same order as the requests in list walk the list backwards,
557      // so we can remove from list without impacting the indexes of earlier members
558      for (int i = results.length - 1; i>=0; i--) {
559        // if result is not null, it succeeded
560        if (results[i] instanceof Result) {
561          deletes.remove(i);
562        }
563      }
564    }
565  }

607  /**
608   * {@inheritDoc}
609   */
610  @Override
611  public void mutateRow(final RowMutations rm) throws IOException {
612    CancellableRegionServerCallable<MultiResponse> callable =
613      new CancellableRegionServerCallable<MultiResponse>(this.connection, getName(), rm.getRow(),
614          rpcControllerFactory.newController(), writeRpcTimeout, new RetryingTimeTracker().start(), rm.getMaxPriority()){
615      @Override
616      protected MultiResponse rpcCall() throws Exception {
617        RegionAction.Builder regionMutationBuilder = RequestConverter.buildRegionAction(
618            getLocation().getRegionInfo().getRegionName(), rm);
619        regionMutationBuilder.setAtomic(true);
620        MultiRequest request =
621            MultiRequest.newBuilder().addRegionAction(regionMutationBuilder.build()).build();
622        ClientProtos.MultiResponse response = doMulti(request);
623        ClientProtos.RegionActionResult res = response.getRegionActionResultList().get(0);
624        if (res.hasException()) {
625          Throwable ex = ProtobufUtil.toException(res.getException());
626          if (ex instanceof IOException) {
627            throw (IOException) ex;
628          }
629          throw new IOException("Failed to mutate row: " + Bytes.toStringBinary(rm.getRow()), ex);
630        }
631        return ResponseConverter.getResults(request, response, getRpcControllerCellScanner());
632      }
633    };
634    AsyncProcessTask task = AsyncProcessTask.newBuilder()
635            .setPool(pool)
636            .setTableName(tableName)
637            .setRowAccess(rm.getMutations())
638            .setCallable(callable)
639            .setRpcTimeout(writeRpcTimeout)
640            .setOperationTimeout(operationTimeout)

865  private boolean doCheckAndMutate(final byte [] row, final byte [] family, final byte [] qualifier,
866                                final String opName, final byte [] value, final RowMutations rm)
867  throws IOException {
868    CancellableRegionServerCallable<MultiResponse> callable =
869    new CancellableRegionServerCallable<MultiResponse>(connection, getName(), rm.getRow(),
870    rpcControllerFactory.newController(), writeRpcTimeout, new RetryingTimeTracker().start(), rm.getMaxPriority()) {
871      @Override
872      protected MultiResponse rpcCall() throws Exception {
873        CompareType compareType = CompareType.valueOf(opName);
874        MultiRequest request = RequestConverter.buildMutateRequest(
875        getLocation().getRegionInfo().getRegionName(), row, family, qualifier,
876        new BinaryComparator(value), compareType, rm);
877        ClientProtos.MultiResponse response = doMulti(request);
878        ClientProtos.RegionActionResult res = response.getRegionActionResultList().get(0);
879        if (res.hasException()) {
880          Throwable ex = ProtobufUtil.toException(res.getException());
881          if (ex instanceof IOException) {
882            throw (IOException)ex;
883          }
884          throw new IOException("Failed to checkAndMutate row: "+
885          Bytes.toStringBinary(rm.getRow()), ex);
886        }
887        return ResponseConverter.getResults(request, response, getRpcControllerCellScanner());
888      }
889    };
890
891    /**
892     *  Currently, we use one array to store 'processed' flag which is returned by server.
893     *  It is excessive to send such a large array, but that is required by the framework right now
894     * */
895    Object[] results = new Object[rm.getMutations().size()];
896    AsyncProcessTask task = AsyncProcessTask.newBuilder()
897    .setPool(pool)
898    .setTableName(tableName)
899    .setRowAccess(rm.getMutations())
900    .setResults(results)
901    .setCallable(callable)
902    // TODO any better timeout?
903    .setRpcTimeout(Math.max(readRpcTimeout, writeRpcTimeout))
904    .setOperationTimeout(operationTimeout)
905    .setSubmittedRows(AsyncProcessTask.SubmittedRows.ALL)
906    .build();
907    AsyncRequestFuture ars = multiAp.submit(task);
908    ars.waitUntilDone();
909    if (ars.hasError()) {
910      throw ars.getErrors();
911    }
912
913    return ((Result)results[0]).getExists();
914  }

949  @Override
950  public boolean[] existsAll(final List<Get> gets) throws IOException {
951    if (gets.isEmpty()) return new boolean[]{};
952    if (gets.size() == 1) return new boolean[]{exists(gets.get(0))};
953
954    ArrayList<Get> exists = new ArrayList<>(gets.size());
955    for (Get g: gets){
956      Get ge = new Get(g);
957      ge.setCheckExistenceOnly(true);
958      exists.add(ge);
959    }
960
961    Object[] r1= new Object[exists.size()];
962    try {
963      batch(exists, r1, readRpcTimeout);
964    } catch (InterruptedException e) {
965      throw (InterruptedIOException)new InterruptedIOException().initCause(e);
966    }
967
968    // translate.
969    boolean[] results = new boolean[r1.length];
970    int i = 0;
971    for (Object o : r1) {
972      // batch ensures if there is a failure we get an exception instead
973      results[i++] = ((Result)o).getExists();
974    }
975
976    return results;
977  }

996  @Override
997  public void close() throws IOException {
998    if (this.closed) {
999      return;
1000    }
1001    if (cleanupPoolOnClose) {
1002      this.pool.shutdown();
1003      try {
1004        boolean terminated = false;
1005        do {
1006          // wait until the pool has terminated
1007          terminated = this.pool.awaitTermination(60, TimeUnit.SECONDS);
1008        } while (!terminated);
1009      } catch (InterruptedException e) {
1010        this.pool.shutdownNow();
1011        LOG.warn("waitForTermination interrupted");
1012      }
1013    }
1014    this.closed = true;
1015  }

1083   * {@inheritDoc}
1084   */
1085  @Override
1086  public <T extends Service, R> void coprocessorService(final Class<T> service,
1087      byte[] startKey, byte[] endKey, final Batch.Call<T,R> callable,
1088      final Batch.Callback<R> callback) throws ServiceException, Throwable {
1089    // get regions covered by the row range
1090    List<byte[]> keys = getStartKeysInRange(startKey, endKey);
1091    Map<byte[],Future<R>> futures = new TreeMap<>(Bytes.BYTES_COMPARATOR);
1092    for (final byte[] r : keys) {
1093      final RegionCoprocessorRpcChannel channel =
1094          new RegionCoprocessorRpcChannel(connection, tableName, r);
1095      Future<R> future = pool.submit(new Callable<R>() {
1096        @Override
1097        public R call() throws Exception {
1098          T instance =
1099              org.apache.hadoop.hbase.protobuf.ProtobufUtil.newServiceStub(service, channel);
1100          R result = callable.call(instance);
1101          byte[] region = channel.getLastRegion();
1102          if (callback != null) {
1103            callback.update(region, r, result);
1104          }
1105          return result;
1106        }
1107      });
1108      futures.put(r, future);
1109    }
1110    for (Map.Entry<byte[],Future<R>> e : futures.entrySet()) {
1111      try {
1112        e.getValue().get();
1113      } catch (ExecutionException ee) {
1114        LOG.warn("Error calling coprocessor service " + service.getName() + " for row "
1115            + Bytes.toStringBinary(e.getKey()), ee);
1116        throw ee.getCause();
1117      } catch (InterruptedException ie) {
1118        throw new InterruptedIOException("Interrupted calling coprocessor service "
1119            + service.getName() + " for row " + Bytes.toStringBinary(e.getKey())).initCause(ie);
1120      }
1121    }
1122  }

1202  /**
1203   * {@inheritDoc}
1204   */
1205  @Override
1206  public <R extends Message> void batchCoprocessorService(
1207      final Descriptors.MethodDescriptor methodDescriptor, final Message request,
1208      byte[] startKey, byte[] endKey, final R responsePrototype, final Callback<R> callback)
1209      throws ServiceException, Throwable {
1210
1211    if (startKey == null) {
1212      startKey = HConstants.EMPTY_START_ROW;
1213    }
1214    if (endKey == null) {
1215      endKey = HConstants.EMPTY_END_ROW;
1216    }
1217    // get regions covered by the row range
1218    Pair<List<byte[]>, List<HRegionLocation>> keysAndRegions =
1219        getKeysAndRegionsInRange(startKey, endKey, true);
1220    List<byte[]> keys = keysAndRegions.getFirst();
1221    List<HRegionLocation> regions = keysAndRegions.getSecond();
1222
1223    // check if we have any calls to make
1224    if (keys.isEmpty()) {
1225      LOG.info("No regions were selected by key range start=" + Bytes.toStringBinary(startKey) +
1226          ", end=" + Bytes.toStringBinary(endKey));
1227      return;
1228    }
1229
1230    List<RegionCoprocessorServiceExec> execs = new ArrayList<>(keys.size());
1231    final Map<byte[], RegionCoprocessorServiceExec> execsByRow = new TreeMap<>(Bytes.BYTES_COMPARATOR);
1232    for (int i = 0; i < keys.size(); i++) {
1233      final byte[] rowKey = keys.get(i);
1234      final byte[] region = regions.get(i).getRegionInfo().getRegionName();
1235      RegionCoprocessorServiceExec exec =
1236          new RegionCoprocessorServiceExec(region, rowKey, methodDescriptor, request);
1237      execs.add(exec);
1238      execsByRow.put(rowKey, exec);
1239    }
1240
1241    // tracking for any possible deserialization errors on success callback
1242    // TODO: it would be better to be able to reuse AsyncProcess.BatchErrors here
1243    final List<Throwable> callbackErrorExceptions = new ArrayList<>();
1244    final List<Row> callbackErrorActions = new ArrayList<>();
1245    final List<String> callbackErrorServers = new ArrayList<>();
1246    Object[] results = new Object[execs.size()];
1247
1248    AsyncProcess asyncProcess =
1249        new AsyncProcess(connection, configuration,
1250            RpcRetryingCallerFactory.instantiate(configuration, connection.getStatisticsTracker()),
1251            true, RpcControllerFactory.instantiate(configuration));
1252
1253    Callback<ClientProtos.CoprocessorServiceResult> resultsCallback
1254    = (byte[] region, byte[] row, ClientProtos.CoprocessorServiceResult serviceResult) -> {
1255      if (LOG.isTraceEnabled()) {
1256        LOG.trace("Received result for endpoint " + methodDescriptor.getFullName() +
1257            ": region=" + Bytes.toStringBinary(region) +
1258            ", row=" + Bytes.toStringBinary(row) +
1259            ", value=" + serviceResult.getValue().getValue());
1260      }
1261      try {
1262        Message.Builder builder = responsePrototype.newBuilderForType();
1263        org.apache.hadoop.hbase.protobuf.ProtobufUtil.mergeFrom(builder,
1264            serviceResult.getValue().getValue().toByteArray());
1265        callback.update(region, row, (R) builder.build());
1266      } catch (IOException e) {
1267        LOG.error("Unexpected response type from endpoint " + methodDescriptor.getFullName(),
1268            e);
1269        callbackErrorExceptions.add(e);
1270        callbackErrorActions.add(execsByRow.get(row));
1271        callbackErrorServers.add("null");
1272      }
1273    };
1274    AsyncProcessTask<ClientProtos.CoprocessorServiceResult> task = AsyncProcessTask.newBuilder(resultsCallback)
1275            .setPool(pool)
1276            .setTableName(tableName)
1277            .setRowAccess(execs)
1278            .setResults(results)
1279            .setRpcTimeout(readRpcTimeout)
1280            .setOperationTimeout(operationTimeout)
1281            .setSubmittedRows(AsyncProcessTask.SubmittedRows.ALL)
1282            .build();
1283    AsyncRequestFuture future = asyncProcess.submit(task);
1284    future.waitUntilDone();
1285
1286    if (future.hasError()) {
1287      throw future.getErrors();
1288    } else if (!callbackErrorExceptions.isEmpty()) {
