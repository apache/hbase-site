583  public void createTable(TableDescriptor desc, byte [] startKey,
584      byte [] endKey, int numRegions)
585  throws IOException {
586    if(numRegions < 3) {
587      throw new IllegalArgumentException("Must create at least three regions");
588    } else if(Bytes.compareTo(startKey, endKey) >= 0) {
589      throw new IllegalArgumentException("Start key must be smaller than end key");
590    }
591    if (numRegions == 3) {
592      createTable(desc, new byte[][]{startKey, endKey});
593      return;
594    }
595    byte [][] splitKeys = Bytes.split(startKey, endKey, numRegions - 3);
596    if(splitKeys == null || splitKeys.length != numRegions - 1) {
597      throw new IllegalArgumentException("Unable to split key range into enough regions");
598    }
599    createTable(desc, splitKeys);
600  }

609  public Future<Void> createTableAsync(final TableDescriptor desc, final byte[][] splitKeys)
610      throws IOException {
611    if (desc.getTableName() == null) {
612      throw new IllegalArgumentException("TableName cannot be null");
613    }
614    if (splitKeys != null && splitKeys.length > 0) {
615      Arrays.sort(splitKeys, Bytes.BYTES_COMPARATOR);
616      // Verify there are no duplicate split keys
617      byte[] lastKey = null;
618      for (byte[] splitKey : splitKeys) {
619        if (Bytes.compareTo(splitKey, HConstants.EMPTY_BYTE_ARRAY) == 0) {
620          throw new IllegalArgumentException(
621              "Empty split key must not be passed in the split keys.");
622        }
623        if (lastKey != null && Bytes.equals(splitKey, lastKey)) {
624          throw new IllegalArgumentException("All split keys must be unique, " +
625            "found duplicate: " + Bytes.toStringBinary(splitKey) +
626            ", " + Bytes.toStringBinary(lastKey));
627        }
628        lastKey = splitKey;
629      }
630    }
631
632    CreateTableResponse response = executeCallable(
633      new MasterCallable<CreateTableResponse>(getConnection(), getRpcControllerFactory()) {
634        @Override
635        protected CreateTableResponse rpcCall() throws Exception {
636          setPriority(desc.getTableName());
637          CreateTableRequest request = RequestConverter.buildCreateTableRequest(
638            desc, splitKeys, ng.getNonceGroup(), ng.newNonce());

640        }

1285  /**
1286   * Compact a table.
1287   * Asynchronous operation.
1288   *
1289   * @param tableName table or region to compact
1290   * @param columnFamily column family within a table or region
1291   * @param major True if we are to do a major compaction.
1292   * @param compactType {@link org.apache.hadoop.hbase.client.CompactType}
1293   * @throws IOException if a remote or network exception occurs
1294   */
1295  private void compact(final TableName tableName, final byte[] columnFamily,final boolean major,
1296                       CompactType compactType) throws IOException {
1297    switch (compactType) {
1298      case MOB:
1299        ServerName master = getMasterAddress();
1300        compact(master, getMobRegionInfo(tableName), major, columnFamily);
1301        break;
1302      case NORMAL:
1303      default:
1304        ZooKeeperWatcher zookeeper = null;
1305        try {
1306          checkTableExists(tableName);
1307          zookeeper = new ZooKeeperWatcher(conf, ZK_IDENTIFIER_PREFIX + connection.toString(),
1308                  new ThrowableAbortable());
1309          List<Pair<HRegionInfo, ServerName>> pairs;
1310          if (TableName.META_TABLE_NAME.equals(tableName)) {
1311            pairs = new MetaTableLocator().getMetaRegionsAndLocations(zookeeper);
1312          } else {
1313            pairs = MetaTableAccessor.getTableRegionsAndLocations(connection, tableName);
1314          }
1315          for (Pair<HRegionInfo, ServerName> pair: pairs) {
1316            if (pair.getFirst().isOffline()) continue;
1317            if (pair.getSecond() == null) continue;
1318            try {
1319              compact(pair.getSecond(), pair.getFirst(), major, columnFamily);
1320            } catch (NotServingRegionException e) {
1321              if (LOG.isDebugEnabled()) {
1322                LOG.debug("Trying to" + (major ? " major" : "") + " compact " +
1323                        pair.getFirst() + ": " +
1324                        StringUtils.stringifyException(e));
1325              }
1326            }
1327          }
1328        } finally {
1329          if (zookeeper != null) {
1330            zookeeper.close();
1331          }
1332        }
1333        break;
1334    }
1335  }

1822  @Override
1823  public void split(final TableName tableName, final byte [] splitPoint) throws IOException {
1824    ZooKeeperWatcher zookeeper = null;
1825    try {
1826      checkTableExists(tableName);
1827      zookeeper = new ZooKeeperWatcher(conf, ZK_IDENTIFIER_PREFIX + connection.toString(),
1828        new ThrowableAbortable());
1829      List<Pair<HRegionInfo, ServerName>> pairs;
1830      if (TableName.META_TABLE_NAME.equals(tableName)) {
1831        pairs = new MetaTableLocator().getMetaRegionsAndLocations(zookeeper);
1832      } else {
1833        pairs = MetaTableAccessor.getTableRegionsAndLocations(connection, tableName);
1834      }
1835      if (splitPoint == null) {
1836        LOG.info("SplitPoint is null, will find bestSplitPoint from Region");
1837      }
1838      for (Pair<HRegionInfo, ServerName> pair: pairs) {
1839        // May not be a server for a particular row
1840        if (pair.getSecond() == null) continue;
1841        HRegionInfo r = pair.getFirst();
1842        // check for parents
1843        if (r.isSplitParent()) continue;
1844        // if a split point given, only split that particular region
1845        if (r.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID ||
1846           (splitPoint != null && !r.containsRow(splitPoint))) continue;
1847        // call out to master to do split now
1848        splitRegionAsync(pair.getFirst(), splitPoint);
1849      }

1928   * @param regionName Name of a region.
1929   * @return a pair of HRegionInfo and ServerName if <code>regionName</code> is
1930   *  a verified region name (we call {@link
1931   *  MetaTableAccessor#getRegionLocation(Connection, byte[])}
1932   *  else null.
1933   * Throw IllegalArgumentException if <code>regionName</code> is null.
1934   * @throws IOException
1935   */
1936  Pair<HRegionInfo, ServerName> getRegion(final byte[] regionName) throws IOException {
1937    if (regionName == null) {
1938      throw new IllegalArgumentException("Pass a table name or region name");
1939    }
1940    Pair<HRegionInfo, ServerName> pair =
1941      MetaTableAccessor.getRegion(connection, regionName);
1942    if (pair == null) {
1943      final AtomicReference<Pair<HRegionInfo, ServerName>> result = new AtomicReference<>(null);
1944      final String encodedName = Bytes.toString(regionName);
1945      MetaTableAccessor.Visitor visitor = new MetaTableAccessor.Visitor() {
1946        @Override
1947        public boolean visit(Result data) throws IOException {
1948          HRegionInfo info = MetaTableAccessor.getHRegionInfo(data);
1949          if (info == null) {
1950            LOG.warn("No serialized HRegionInfo in " + data);
1951            return true;
1952          }
1953          RegionLocations rl = MetaTableAccessor.getRegionLocations(data);
1954          boolean matched = false;
1955          ServerName sn = null;
1956          if (rl != null) {
1957            for (HRegionLocation h : rl.getRegionLocations()) {
1958              if (h != null && encodedName.equals(h.getRegionInfo().getEncodedName())) {
1959                sn = h.getServerName();
1960                info = h.getRegionInfo();
1961                matched = true;
1962              }
1963            }
1964          }
1965          if (!matched) return true;
1966          result.set(new Pair<>(info, sn));
1967          return false; // found the region, stop
1968        }
1969      };
1970
1971      MetaTableAccessor.fullScanRegions(connection, visitor);
1972      pair = result.get();
1973    }
1974    return pair;
1975  }

2693  public void restoreSnapshot(final String snapshotName, final boolean takeFailSafeSnapshot,
2694      final boolean restoreAcl) throws IOException, RestoreSnapshotException {
2695    TableName tableName = getTableNameBeforeRestoreSnapshot(snapshotName);
2696
2697    // The table does not exists, switch to clone.
2698    if (!tableExists(tableName)) {
2699      cloneSnapshot(snapshotName, tableName, restoreAcl);
2700      return;
2701    }
2702
2703    // Check if the table is disabled
2704    if (!isTableDisabled(tableName)) {
2705      throw new TableNotDisabledException(tableName);
2706    }
2707
2708    // Take a snapshot of the current state
2709    String failSafeSnapshotSnapshotName = null;
2710    if (takeFailSafeSnapshot) {
2711      failSafeSnapshotSnapshotName = conf.get("hbase.snapshot.restore.failsafe.name",
2712        "hbase-failsafe-{snapshot.name}-{restore.timestamp}");
2713      failSafeSnapshotSnapshotName = failSafeSnapshotSnapshotName
2714        .replace("{snapshot.name}", snapshotName)
2715        .replace("{table.name}", tableName.toString().replace(TableName.NAMESPACE_DELIM, '.'))
2716        .replace("{restore.timestamp}", String.valueOf(EnvironmentEdgeManager.currentTime()));
2717      LOG.info("Taking restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
2718      snapshot(failSafeSnapshotSnapshotName, tableName);
2719    }
2720
2721    try {
2722      // Restore snapshot
2723      get(
2724        internalRestoreSnapshotAsync(snapshotName, tableName, restoreAcl),
2725        syncWaitTimeout,
2726        TimeUnit.MILLISECONDS);
2727    } catch (IOException e) {
2728      // Something went wrong during the restore...
2729      // if the pre-restore snapshot is available try to rollback
2730      if (takeFailSafeSnapshot) {
2731        try {
2732          get(
2733            internalRestoreSnapshotAsync(failSafeSnapshotSnapshotName, tableName, restoreAcl),
2734            syncWaitTimeout,
2735            TimeUnit.MILLISECONDS);
2736          String msg = "Restore snapshot=" + snapshotName +
2737            " failed. Rollback to snapshot=" + failSafeSnapshotSnapshotName + " succeeded.";
2738          LOG.error(msg, e);
2739          throw new RestoreSnapshotException(msg, e);
2740        } catch (IOException ex) {
2741          String msg = "Failed to restore and rollback to snapshot=" + failSafeSnapshotSnapshotName;
2742          LOG.error(msg, ex);
2743          throw new RestoreSnapshotException(msg, e);
2744        }
2745      } else {
2746        throw new RestoreSnapshotException("Failed to restore snapshot=" + snapshotName, e);
2747      }
2748    }
2749
2750    // If the restore is succeeded, delete the pre-restore snapshot
2751    if (takeFailSafeSnapshot) {
2752      try {
2753        LOG.info("Deleting restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
2754        deleteSnapshot(failSafeSnapshotSnapshotName);
2755      } catch (IOException e) {
2756        LOG.error("Unable to remove the failsafe snapshot: " + failSafeSnapshotSnapshotName, e);
2757      }
2758    }
2759  }

3790      final MetaTableAccessor.Visitor visitor = new MetaTableAccessor.Visitor() {
3791        @Override

3793          RegionLocations list = MetaTableAccessor.getRegionLocations(rowResult);

3795            LOG.warn("No serialized HRegionInfo in " + rowResult);
3796            return true;
3797          }
3798          HRegionLocation l = list.getRegionLocation();
3799          if (l == null) {
3800            return true;
3801          }
3802          if (!l.getRegionInfo().getTable().equals(desc.getTableName())) {
3803            return false;
3804          }
3805          if (l.getRegionInfo().isOffline() || l.getRegionInfo().isSplit()) return true;
3806          HRegionLocation[] locations = list.getRegionLocations();
3807          for (HRegionLocation location : locations) {
3808            if (location == null) continue;
3809            ServerName serverName = location.getServerName();
3810            // Make sure that regions are assigned to server
3811            if (serverName != null && serverName.getHostAndPort() != null) {
3812              actualRegCount.incrementAndGet();
3813            }
3814          }
3815          return true;
3816        }
3817      };
3818
3819      int tries = 0;
3820      int numRegs = (splitKeys == null ? 1 : splitKeys.length + 1) * desc.getRegionReplication();
3821      while (EnvironmentEdgeManager.currentTime() < deadlineTs) {
3822        actualRegCount.set(0);
3823        MetaTableAccessor.scanMetaForTableRegions(getAdmin().getConnection(), visitor,
3824          desc.getTableName());
3825        if (actualRegCount.get() == numRegs) {
3826          // all the regions are online
3827          return;
3828        }
3829
3830        try {
3831          Thread.sleep(getAdmin().getPauseTime(tries++));
3832        } catch (InterruptedException e) {
3833          throw new InterruptedIOException("Interrupted when opening" + " regions; "
3834              + actualRegCount.get() + " of " + numRegs + " regions processed so far");
3835        }
3836      }
3837      throw new TimeoutException("Only " + actualRegCount.get() + " of " + numRegs
3838          + " regions are online; retries exhausted.");

4146   * Copies the REPLICATION_SCOPE of table descriptor passed as an argument. Before copy, the method
4147   * ensures that the name of table and column-families should match.
4148   * @param peerHtd descriptor on peer cluster
4149   * @param localHtd - The HTableDescriptor of table from source cluster.
4150   * @return true If the name of table and column families match and REPLICATION_SCOPE copied
4151   *         successfully. false If there is any mismatch in the names.
4152   */
4153  private boolean copyReplicationScope(final HTableDescriptor peerHtd,
4154      final HTableDescriptor localHtd) {
4155    // Copy the REPLICATION_SCOPE only when table names and the names of
4156    // Column-Families are same.
4157    int result = peerHtd.getTableName().compareTo(localHtd.getTableName());
4158
4159    if (result == 0) {
4160      Iterator<HColumnDescriptor> remoteHCDIter = peerHtd.getFamilies().iterator();
4161      Iterator<HColumnDescriptor> localHCDIter = localHtd.getFamilies().iterator();
4162

4164        HColumnDescriptor remoteHCD = remoteHCDIter.next();
4165        HColumnDescriptor localHCD = localHCDIter.next();
4166

4168        String localHCDName = localHCD.getNameAsString();
4169
4170        if (remoteHCDName.equals(localHCDName)) {
4171          remoteHCD.setScope(localHCD.getScope());
4172        } else {
4173          result = -1;
4174          break;
4175        }
4176      }
4177      if (remoteHCDIter.hasNext() || localHCDIter.hasNext()) {
4178        return false;
4179      }
4180    }
4181
4182    return result == 0;
4183  }
