035/**
036 * Used to perform CheckAndMutate operations.
037 * <p>
038 * Use the builder class to instantiate a CheckAndMutate object.
039 * This builder class is fluent style APIs, the code are like:
040 * <pre>
041 * <code>
042 * // A CheckAndMutate operation where do the specified action if the column (specified by the
043 * // family and the qualifier) of the row equals to the specified value
044 * CheckAndMutate checkAndMutate = CheckAndMutate.newBuilder(row)
045 *   .ifEquals(family, qualifier, value)
046 *   .build(put);
047 *
048 * // A CheckAndMutate operation where do the specified action if the column (specified by the
049 * // family and the qualifier) of the row doesn't exist
050 * CheckAndMutate checkAndMutate = CheckAndMutate.newBuilder(row)
051 *   .ifNotExists(family, qualifier)
052 *   .build(put);
053 *
054 * // A CheckAndMutate operation where do the specified action if the row matches the filter
055 * CheckAndMutate checkAndMutate = CheckAndMutate.newBuilder(row)
056 *   .ifMatches(filter)
057 *   .build(delete);
058 * </code>
059 * </pre>
060 */
061@InterfaceAudience.Public
062@InterfaceStability.Evolving
063public final class CheckAndMutate extends Mutation {
064
065  /**
066   * A builder class for building a CheckAndMutate object.
067   */
068  @InterfaceAudience.Public
069  @InterfaceStability.Evolving
070  public static final class Builder {
071    private final byte[] row;
072    private byte[] family;
073    private byte[] qualifier;
074    private CompareOperator op;
075    private byte[] value;
076    private Filter filter;
077    private TimeRange timeRange;
078
079    private Builder(byte[] row) {
080      this.row = Preconditions.checkNotNull(row, "row is null");
081    }
082
083    /**
084     * Check for lack of column
085     *
086     * @param family family to check
087     * @param qualifier qualifier to check
088     * @return the CheckAndMutate object
089     */
090    public Builder ifNotExists(byte[] family, byte[] qualifier) {
091      return ifEquals(family, qualifier, null);
092    }
093
094    /**
095     * Check for equality
096     *
097     * @param family family to check
098     * @param qualifier qualifier to check
099     * @param value the expected value
100     * @return the CheckAndMutate object
101     */
102    public Builder ifEquals(byte[] family, byte[] qualifier, byte[] value) {
103      return ifMatches(family, qualifier, CompareOperator.EQUAL, value);
104    }
105
106    /**
107     * @param family family to check
108     * @param qualifier qualifier to check
109     * @param compareOp comparison operator to use
110     * @param value the expected value
111     * @return the CheckAndMutate object
112     */
113    public Builder ifMatches(byte[] family, byte[] qualifier, CompareOperator compareOp,
114      byte[] value) {
115      this.family = Preconditions.checkNotNull(family, "family is null");
116      this.qualifier = qualifier;
117      this.op = Preconditions.checkNotNull(compareOp, "compareOp is null");
118      this.value = value;
119      return this;
120    }
121
122    /**
123     * @param filter filter to check
124     * @return the CheckAndMutate object
125     */
126    public Builder ifMatches(Filter filter) {
127      this.filter = Preconditions.checkNotNull(filter, "filter is null");
128      return this;
129    }
130
131    /**
132     * @param timeRange time range to check
133     * @return the CheckAndMutate object
134     */
135    public Builder timeRange(TimeRange timeRange) {
136      this.timeRange = timeRange;
137      return this;
138    }
139
140    private void preCheck(Row action) {
141      Preconditions.checkNotNull(action, "action is null");
142      if (!Bytes.equals(row, action.getRow())) {
143        throw new IllegalArgumentException("The row of the action <" +
144          Bytes.toStringBinary(action.getRow()) + "> doesn't match the original one <" +
145          Bytes.toStringBinary(this.row) + ">");
146      }
147      Preconditions.checkState(op != null || filter != null, "condition is null. You need to"
148        + " specify the condition by calling ifNotExists/ifEquals/ifMatches before building a"
149        + " CheckAndMutate object");
150    }
151
152    /**
153     * @param put data to put if check succeeds
154     * @return a CheckAndMutate object
155     */
156    public CheckAndMutate build(Put put) {
157      preCheck(put);
158      if (filter != null) {
159        return new CheckAndMutate(row, filter, timeRange, put);
160      } else {
161        return new CheckAndMutate(row, family, qualifier, op, value, timeRange, put);
162      }
163    }
164
165    /**
166     * @param delete data to delete if check succeeds
167     * @return a CheckAndMutate object
168     */
169    public CheckAndMutate build(Delete delete) {
170      preCheck(delete);
171      if (filter != null) {
172        return new CheckAndMutate(row, filter, timeRange, delete);
173      } else {
174        return new CheckAndMutate(row, family, qualifier, op, value, timeRange, delete);
175      }
176    }
177
178    /**
179     * @param increment data to increment if check succeeds
180     * @return a CheckAndMutate object
181     */
182    public CheckAndMutate build(Increment increment) {
183      preCheck(increment);
184      if (filter != null) {
185        return new CheckAndMutate(row, filter, timeRange, increment);
186      } else {
187        return new CheckAndMutate(row, family, qualifier, op, value, timeRange, increment);
188      }
189    }
190
191    /**
192     * @param append data to append if check succeeds
193     * @return a CheckAndMutate object
194     */
195    public CheckAndMutate build(Append append) {
196      preCheck(append);
197      if (filter != null) {
198        return new CheckAndMutate(row, filter, timeRange, append);
199      } else {
200        return new CheckAndMutate(row, family, qualifier, op, value, timeRange, append);
201      }
202    }
203
204    /**
205     * @param mutation mutations to perform if check succeeds
206     * @return a CheckAndMutate object
207     */
208    public CheckAndMutate build(RowMutations mutation) {
209      preCheck(mutation);
210      if (filter != null) {
211        return new CheckAndMutate(row, filter, timeRange, mutation);
212      } else {
213        return new CheckAndMutate(row, family, qualifier, op, value, timeRange, mutation);
214      }
215    }
216  }
217
218  /**
219   * returns a builder object to build a CheckAndMutate object
220   *
221   * @param row row
222   * @return a builder object
223   */
224  public static Builder newBuilder(byte[] row) {
225    return new Builder(row);
226  }
227
228  private final byte[] family;
229  private final byte[] qualifier;
230  private final CompareOperator op;
231  private final byte[] value;
232  private final Filter filter;
233  private final TimeRange timeRange;
234  private final Row action;
235
236  private CheckAndMutate(byte[] row, byte[] family, byte[] qualifier,final CompareOperator op,
237    byte[] value, TimeRange timeRange, Row action) {
238    super(row, HConstants.LATEST_TIMESTAMP, Collections.emptyNavigableMap());
239    this.family = family;
240    this.qualifier = qualifier;
241    this.op = op;
242    this.value = value;
243    this.filter = null;
244    this.timeRange = timeRange != null ? timeRange : TimeRange.allTime();
245    this.action = action;
246  }
247
248  private CheckAndMutate(byte[] row, Filter filter, TimeRange timeRange, Row action) {
249    super(row, HConstants.LATEST_TIMESTAMP, Collections.emptyNavigableMap());
250    this.family = null;
251    this.qualifier = null;
252    this.op = null;
253    this.value = null;
254    this.filter = filter;
255    this.timeRange = timeRange != null ? timeRange : TimeRange.allTime();
256    this.action = action;
257  }
