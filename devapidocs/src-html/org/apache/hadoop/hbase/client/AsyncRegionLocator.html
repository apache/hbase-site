052/**
053 * The asynchronous region locator.
054 */
055@InterfaceAudience.Private
056class AsyncRegionLocator {
057
058  private static final Logger LOG = LoggerFactory.getLogger(AsyncRegionLocator.class);
059
060  private final HashedWheelTimer retryTimer;
061
062  private final AsyncConnectionImpl conn;
063
064  private final AsyncMetaRegionLocator metaRegionLocator;
065
066  private final AsyncNonMetaRegionLocator nonMetaRegionLocator;
067
068  AsyncRegionLocator(AsyncConnectionImpl conn, HashedWheelTimer retryTimer) {
069    this.conn = conn;
070    this.metaRegionLocator = new AsyncMetaRegionLocator(conn.registry);
071    this.nonMetaRegionLocator = new AsyncNonMetaRegionLocator(conn);
072    this.retryTimer = retryTimer;
073  }
074
075  private <T> CompletableFuture<T> withTimeout(CompletableFuture<T> future, long timeoutNs,
076    Supplier<String> timeoutMsg) {
077    if (future.isDone() || timeoutNs <= 0) {
078      return future;
079    }
080    Timeout timeoutTask = retryTimer.newTimeout(t -> {
081      if (future.isDone()) {
082        return;
083      }
084      future.completeExceptionally(new TimeoutIOException(timeoutMsg.get()));
085    }, timeoutNs, TimeUnit.NANOSECONDS);
086    FutureUtils.addListener(future, (loc, error) -> {
087      if (error != null && error.getClass() != TimeoutIOException.class) {
088        // cancel timeout task if we are not completed by it.
089        timeoutTask.cancel();
090      }
091    });
092    return future;
093  }
094
095  private boolean isMeta(TableName tableName) {
096    return TableName.isMetaTableName(tableName);
097  }
098
099  private <T> CompletableFuture<T> tracedLocationFuture(Supplier<CompletableFuture<T>> action,
100    Function<T, List<String>> getRegionNames, TableName tableName, String methodName) {
101    Span span = createTableSpan("AsyncRegionLocator." + methodName, tableName);
102    try (Scope scope = span.makeCurrent()) {
103      CompletableFuture<T> future = action.get();
104      FutureUtils.addListener(future, (resp, error) -> {
105        if (error != null) {
106          TraceUtil.setError(span, error);
107        } else {
108          List<String> regionNames = getRegionNames.apply(resp);
109          if (!regionNames.isEmpty()) {
110            span.setAttribute(REGION_NAMES_KEY, regionNames);
111          }
112          span.setStatus(StatusCode.OK);
113        }
114        span.end();
115      });
116      return future;
117    }
118  }
119
120  private List<String> getRegionName(RegionLocations locs) {
121    List<String> names = new ArrayList<>();
122    for (HRegionLocation loc : locs.getRegionLocations()) {
123      if (loc != null) {
124        names.add(loc.getRegion().getRegionNameAsString());
125      }
126    }
127    return names;
128  }
129
130  CompletableFuture<RegionLocations> getRegionLocations(TableName tableName, byte[] row,
131    RegionLocateType type, boolean reload, long timeoutNs) {
132    return tracedLocationFuture(() -> {
133      CompletableFuture<RegionLocations> future = isMeta(tableName) ?
134        metaRegionLocator.getRegionLocations(RegionReplicaUtil.DEFAULT_REPLICA_ID, reload) :
135        nonMetaRegionLocator.getRegionLocations(tableName, row,
136          RegionReplicaUtil.DEFAULT_REPLICA_ID, type, reload);
137      return withTimeout(future, timeoutNs,
138        () -> "Timeout(" + TimeUnit.NANOSECONDS.toMillis(timeoutNs) +
139          "ms) waiting for region locations for " + tableName + ", row='" +
140          Bytes.toStringBinary(row) + "'");
141    }, this::getRegionName, tableName, "getRegionLocations");
142  }
143
144  CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row,
145    int replicaId, RegionLocateType type, boolean reload, long timeoutNs) {
146    return tracedLocationFuture(() -> {
147      // meta region can not be split right now so we always call the same method.
148      // Change it later if the meta table can have more than one regions.
149      CompletableFuture<HRegionLocation> future = new CompletableFuture<>();
150      CompletableFuture<RegionLocations> locsFuture =
151        isMeta(tableName) ? metaRegionLocator.getRegionLocations(replicaId, reload) :
152          nonMetaRegionLocator.getRegionLocations(tableName, row, replicaId, type, reload);
153      addListener(locsFuture, (locs, error) -> {
154        if (error != null) {
155          future.completeExceptionally(error);
156          return;
157        }
158        HRegionLocation loc = locs.getRegionLocation(replicaId);
159        if (loc == null) {
160          future.completeExceptionally(
161            new RegionOfflineException("No location for " + tableName + ", row='" +
162              Bytes.toStringBinary(row) + "', locateType=" + type + ", replicaId=" + replicaId));
163        } else if (loc.getServerName() == null) {
164          future.completeExceptionally(
165            new RegionOfflineException("No server address listed for region '" +
166              loc.getRegion().getRegionNameAsString() + ", row='" + Bytes.toStringBinary(row) +
167              "', locateType=" + type + ", replicaId=" + replicaId));
168        } else {
169          future.complete(loc);
170        }
171      });
172      return withTimeout(future, timeoutNs,
173        () -> "Timeout(" + TimeUnit.NANOSECONDS.toMillis(timeoutNs) +
174          "ms) waiting for region location for " + tableName + ", row='" +
175          Bytes.toStringBinary(row) + "', replicaId=" + replicaId);
176    }, loc -> Arrays.asList(loc.getRegion().getRegionNameAsString()), tableName,
177      "getRegionLocation");
178  }
179
180  CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row,
181    int replicaId, RegionLocateType type, long timeoutNs) {
182    return getRegionLocation(tableName, row, replicaId, type, false, timeoutNs);
183  }
184
185  CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row,
186    RegionLocateType type, boolean reload, long timeoutNs) {
187    return getRegionLocation(tableName, row, RegionReplicaUtil.DEFAULT_REPLICA_ID, type, reload,
188      timeoutNs);
189  }
190
191  CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row,
192    RegionLocateType type, long timeoutNs) {
193    return getRegionLocation(tableName, row, type, false, timeoutNs);
194  }
195
196  void updateCachedLocationOnError(HRegionLocation loc, Throwable exception) {
197    if (loc.getRegion().isMetaRegion()) {
198      metaRegionLocator.updateCachedLocationOnError(loc, exception);
199    } else {
200      nonMetaRegionLocator.updateCachedLocationOnError(loc, exception);
201    }
202  }
203
204  void clearCache(TableName tableName) {
205    TraceUtil.trace(() -> {
206      LOG.debug("Clear meta cache for {}", tableName);
207      if (tableName.equals(META_TABLE_NAME)) {
208        metaRegionLocator.clearCache();
209      } else {
210        nonMetaRegionLocator.clearCache(tableName);
211      }
212    }, () -> createTableSpan("AsyncRegionLocator.clearCache", tableName));
213  }
214
215  void clearCache(ServerName serverName) {
216    TraceUtil.trace(() -> {
217      LOG.debug("Clear meta cache for {}", serverName);
218      metaRegionLocator.clearCache(serverName);
219      nonMetaRegionLocator.clearCache(serverName);
220      conn.getConnectionMetrics().ifPresent(MetricsConnection::incrMetaCacheNumClearServer);
221    }, () -> createSpan("AsyncRegionLocator.clearCache").setAttribute(SERVER_NAME_KEY,
222      serverName.getServerName()));
223  }
224
225  void clearCache() {
226    TraceUtil.trace(() -> {
227      metaRegionLocator.clearCache();
228      nonMetaRegionLocator.clearCache();
229    }, "AsyncRegionLocator.clearCache");
230  }

232  AsyncNonMetaRegionLocator getNonMetaRegionLocator() {
233    return nonMetaRegionLocator;
234  }

236  // only used for testing whether we have cached the location for a region.
237  RegionLocations getRegionLocationInCache(TableName tableName, byte[] row) {
238    if (TableName.isMetaTableName(tableName)) {
239      return metaRegionLocator.getRegionLocationInCache();
240    } else {
241      return nonMetaRegionLocator.getRegionLocationInCache(tableName, row);
242    }
243  }
244
245  // only used for testing whether we have cached the location for a table.
246  int getNumberOfCachedRegionLocations(TableName tableName) {
247    if (TableName.isMetaTableName(tableName)) {
248      return metaRegionLocator.getNumberOfCachedRegionLocations();
249    } else {
250      return nonMetaRegionLocator.getNumberOfCachedRegionLocations(tableName);
251    }
252  }
