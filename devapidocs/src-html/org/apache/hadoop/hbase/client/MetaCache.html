040/**
041 * A cache implementation for region locations from meta.
042 */
043@InterfaceAudience.Private
044public class MetaCache {
045
046  private static final Log LOG = LogFactory.getLog(MetaCache.class);
047
048  /**
049   * Map of table to table {@link HRegionLocation}s.
050   */
051  private final ConcurrentMap<TableName, ConcurrentNavigableMap<byte[], RegionLocations>>
052  cachedRegionLocations =
053  new CopyOnWriteArrayMap<>();
054
055  // The presence of a server in the map implies it's likely that there is an
056  // entry in cachedRegionLocations that map to this server; but the absence
057  // of a server in this map guarentees that there is no entry in cache that
058  // maps to the absent server.
059  // The access to this attribute must be protected by a lock on cachedRegionLocations
060  private final Set<ServerName> cachedServers = new CopyOnWriteArraySet<>();
061
062  private final MetricsConnection metrics;
063
064  public MetaCache(MetricsConnection metrics) {
065    this.metrics = metrics;
066  }
067
068  /**
069   * Search the cache for a location that fits our table and row key.
070   * Return null if no suitable region is located.
071   *
072   * @return Null or region location found in cache.
073   */
074  public RegionLocations getCachedLocation(final TableName tableName, final byte [] row) {
075    ConcurrentNavigableMap<byte[], RegionLocations> tableLocations =
076      getTableLocations(tableName);
077
078    Entry<byte[], RegionLocations> e = tableLocations.floorEntry(row);
079    if (e == null) {
080      if (metrics != null) metrics.incrMetaCacheMiss();
081      return null;
082    }
083    RegionLocations possibleRegion = e.getValue();
084
085    // make sure that the end key is greater than the row we're looking
086    // for, otherwise the row actually belongs in the next region, not
087    // this one. the exception case is when the endkey is
088    // HConstants.EMPTY_END_ROW, signifying that the region we're
089    // checking is actually the last region in the table.
090    byte[] endKey = possibleRegion.getRegionLocation().getRegionInfo().getEndKey();
091    // Here we do direct Bytes.compareTo and not doing CellComparator/MetaCellComparator path.
092    // MetaCellComparator is for comparing against data in META table which need special handling.
093    // Not doing that is ok for this case because
094    // 1. We are getting the Region location for the given row in non META tables only. The compare
095    // checks the given row is within the end key of the found region. So META regions are not
096    // coming in here.
097    // 2. Even if META region comes in, its end key will be empty byte[] and so Bytes.equals(endKey,
098    // HConstants.EMPTY_END_ROW) check itself will pass.
099    if (Bytes.equals(endKey, HConstants.EMPTY_END_ROW) ||
100        Bytes.compareTo(endKey, 0, endKey.length, row, 0, row.length) > 0) {
101      if (metrics != null) metrics.incrMetaCacheHit();
102      return possibleRegion;
103    }
104
105    // Passed all the way through, so we got nothing - complete cache miss
106    if (metrics != null) metrics.incrMetaCacheMiss();
107    return null;
108  }

110  /**
111   * Put a newly discovered HRegionLocation into the cache.
112   * @param tableName The table name.
113   * @param source the source of the new location
114   * @param location the new location
115   */
116  public void cacheLocation(final TableName tableName, final ServerName source,
117      final HRegionLocation location) {
118    assert source != null;
119    byte [] startKey = location.getRegionInfo().getStartKey();
120    ConcurrentMap<byte[], RegionLocations> tableLocations = getTableLocations(tableName);
121    RegionLocations locations = new RegionLocations(new HRegionLocation[] {location}) ;
122    RegionLocations oldLocations = tableLocations.putIfAbsent(startKey, locations);
123    boolean isNewCacheEntry = (oldLocations == null);
124    if (isNewCacheEntry) {
125      if (LOG.isTraceEnabled()) {
126        LOG.trace("Cached location: " + location);
127      }
128      addToCachedServers(locations);
129      return;
130    }
131
132    // If the server in cache sends us a redirect, assume it's always valid.
133    HRegionLocation oldLocation = oldLocations.getRegionLocation(
134      location.getRegionInfo().getReplicaId());
135    boolean force = oldLocation != null && oldLocation.getServerName() != null
136        && oldLocation.getServerName().equals(source);
137
138    // For redirect if the number is equal to previous
139    // record, the most common case is that first the region was closed with seqNum, and then
140    // opened with the same seqNum; hence we will ignore the redirect.
141    // There are so many corner cases with various combinations of opens and closes that
142    // an additional counter on top of seqNum would be necessary to handle them all.
143    RegionLocations updatedLocations = oldLocations.updateLocation(location, false, force);
144    if (oldLocations != updatedLocations) {
145      boolean replaced = tableLocations.replace(startKey, oldLocations, updatedLocations);
146      if (replaced && LOG.isTraceEnabled()) {
147        LOG.trace("Changed cached location to: " + location);
148      }
149      addToCachedServers(updatedLocations);
150    }
151  }

153  /**
154   * Put a newly discovered HRegionLocation into the cache.
155   * @param tableName The table name.
156   * @param locations the new locations
157   */
158  public void cacheLocation(final TableName tableName, final RegionLocations locations) {
159    byte [] startKey = locations.getRegionLocation().getRegionInfo().getStartKey();
160    ConcurrentMap<byte[], RegionLocations> tableLocations = getTableLocations(tableName);
161    RegionLocations oldLocation = tableLocations.putIfAbsent(startKey, locations);
162    boolean isNewCacheEntry = (oldLocation == null);
163    if (isNewCacheEntry) {
164      if (LOG.isTraceEnabled()) {
165        LOG.trace("Cached location: " + locations);
166      }
167      addToCachedServers(locations);
168      return;
169    }
170
171    // merge old and new locations and add it to the cache
172    // Meta record might be stale - some (probably the same) server has closed the region
173    // with later seqNum and told us about the new location.
174    RegionLocations mergedLocation = oldLocation.mergeLocations(locations);
175    boolean replaced = tableLocations.replace(startKey, oldLocation, mergedLocation);
176    if (replaced && LOG.isTraceEnabled()) {
177      LOG.trace("Merged cached locations: " + mergedLocation);
178    }
179    addToCachedServers(locations);
180  }
181
182  private void addToCachedServers(RegionLocations locations) {
183    for (HRegionLocation loc : locations.getRegionLocations()) {
184      if (loc != null) {
185        cachedServers.add(loc.getServerName());
186      }
187    }
188  }
189
190  /**
191   * @param tableName
192   * @return Map of cached locations for passed <code>tableName</code>
193   */
194  private ConcurrentNavigableMap<byte[], RegionLocations>
195    getTableLocations(final TableName tableName) {
196    // find the map of cached locations for this table
197    ConcurrentNavigableMap<byte[], RegionLocations> result;
198    result = this.cachedRegionLocations.get(tableName);
199    // if tableLocations for this table isn't built yet, make one
200    if (result == null) {
201      result = new CopyOnWriteArrayMap<>(Bytes.BYTES_COMPARATOR);
202      ConcurrentNavigableMap<byte[], RegionLocations> old =
203          this.cachedRegionLocations.putIfAbsent(tableName, result);
204      if (old != null) {
205        return old;
206      }
207    }
208    return result;
209  }

211  /**
212   * Check the region cache to see whether a region is cached yet or not.
213   * @param tableName tableName
214   * @param row row
215   * @return Region cached or not.
216   */
217  public boolean isRegionCached(TableName tableName, final byte[] row) {
218    RegionLocations location = getCachedLocation(tableName, row);
219    return location != null;
220  }

222  /**
223   * Return the number of cached region for a table. It will only be called
224   * from a unit test.
225   */
226  public int getNumberOfCachedRegionLocations(final TableName tableName) {
227    Map<byte[], RegionLocations> tableLocs = this.cachedRegionLocations.get(tableName);
228    if (tableLocs == null) {
229      return 0;
230    }
231    int numRegions = 0;
232    for (RegionLocations tableLoc : tableLocs.values()) {
233      numRegions += tableLoc.numNonNullElements();
234    }
235    return numRegions;
236  }

242    this.cachedRegionLocations.clear();
243    this.cachedServers.clear();
244  }

246  /**
247   * Delete all cached entries of a server.
248   */
249  public void clearCache(final ServerName serverName) {
250    if (!this.cachedServers.contains(serverName)) {
251      return;
252    }
253
254    boolean deletedSomething = false;
255    synchronized (this.cachedServers) {
256      // We block here, because if there is an error on a server, it's likely that multiple
257      //  threads will get the error  simultaneously. If there are hundreds of thousand of
258      //  region location to check, it's better to do this only once. A better pattern would
259      //  be to check if the server is dead when we get the region location.
260      if (!this.cachedServers.contains(serverName)) {
261        return;
262      }
263      for (ConcurrentMap<byte[], RegionLocations> tableLocations : cachedRegionLocations.values()){
264        for (Entry<byte[], RegionLocations> e : tableLocations.entrySet()) {
265          RegionLocations regionLocations = e.getValue();
266          if (regionLocations != null) {
267            RegionLocations updatedLocations = regionLocations.removeByServer(serverName);
268            if (updatedLocations != regionLocations) {
269              if (updatedLocations.isEmpty()) {
270                deletedSomething |= tableLocations.remove(e.getKey(), regionLocations);
271              } else {
272                deletedSomething |= tableLocations.replace(e.getKey(), regionLocations,
273                    updatedLocations);
274              }
275            }
276          }
277        }
278      }
279      this.cachedServers.remove(serverName);
280    }
281    if (deletedSomething && LOG.isTraceEnabled()) {
282      LOG.trace("Removed all cached region locations that map to " + serverName);
283    }
284  }

286  /**
287   * Delete all cached entries of a table.
288   */
289  public void clearCache(final TableName tableName) {
290    if (LOG.isTraceEnabled()) {
291      LOG.trace("Removed all cached region locations for table " + tableName);
292    }
293    this.cachedRegionLocations.remove(tableName);
294  }

296  /**
297   * Delete a cached location, no matter what it is. Called when we were told to not use cache.
298   * @param tableName tableName
299   * @param row
300   */
301  public void clearCache(final TableName tableName, final byte [] row, int replicaId) {
302    ConcurrentMap<byte[], RegionLocations> tableLocations = getTableLocations(tableName);
303
304    boolean removed = false;
305    RegionLocations regionLocations = getCachedLocation(tableName, row);
306    if (regionLocations != null) {
307      HRegionLocation toBeRemoved = regionLocations.getRegionLocation(replicaId);
308      RegionLocations updatedLocations = regionLocations.remove(replicaId);
309      if (updatedLocations != regionLocations) {
310        byte[] startKey = regionLocations.getRegionLocation().getRegionInfo().getStartKey();
311        if (updatedLocations.isEmpty()) {
312          removed = tableLocations.remove(startKey, regionLocations);
313        } else {
314          removed = tableLocations.replace(startKey, regionLocations, updatedLocations);
315        }
316      }
317
318      if (removed && LOG.isTraceEnabled() && toBeRemoved != null) {
319        LOG.trace("Removed " + toBeRemoved + " from cache");
320      }
321    }
322  }

324  /**
325   * Delete a cached location, no matter what it is. Called when we were told to not use cache.
326   * @param tableName tableName
327   * @param row
328   */
329  public void clearCache(final TableName tableName, final byte [] row) {
330    ConcurrentMap<byte[], RegionLocations> tableLocations = getTableLocations(tableName);
331
332    RegionLocations regionLocations = getCachedLocation(tableName, row);
333    if (regionLocations != null) {
334      byte[] startKey = regionLocations.getRegionLocation().getRegionInfo().getStartKey();
335      boolean removed = tableLocations.remove(startKey, regionLocations);
336      if (removed && LOG.isTraceEnabled()) {
337        LOG.trace("Removed " + regionLocations + " from cache");
338      }
339    }
340  }

342  /**
343   * Delete a cached location for a table, row and server
344   */
345  public void clearCache(final TableName tableName, final byte [] row, ServerName serverName) {
346    ConcurrentMap<byte[], RegionLocations> tableLocations = getTableLocations(tableName);
347
348    RegionLocations regionLocations = getCachedLocation(tableName, row);
349    if (regionLocations != null) {
350      RegionLocations updatedLocations = regionLocations.removeByServer(serverName);
351      if (updatedLocations != regionLocations) {
352        byte[] startKey = regionLocations.getRegionLocation().getRegionInfo().getStartKey();
353        boolean removed = false;
354        if (updatedLocations.isEmpty()) {
355          removed = tableLocations.remove(startKey, regionLocations);
356        } else {
357          removed = tableLocations.replace(startKey, regionLocations, updatedLocations);
358        }
359        if (removed && LOG.isTraceEnabled()) {
360          LOG.trace("Removed locations of table: " + tableName + " ,row: " + Bytes.toString(row)
361            + " mapping to server: " + serverName + " from cache");
362        }
363      }
364    }
365  }

367  /**
368   * Deletes the cached location of the region if necessary, based on some error from source.
369   * @param hri The region in question.
370   */
371  public void clearCache(HRegionInfo hri) {
372    ConcurrentMap<byte[], RegionLocations> tableLocations = getTableLocations(hri.getTable());
373    RegionLocations regionLocations = tableLocations.get(hri.getStartKey());
374    if (regionLocations != null) {
375      HRegionLocation oldLocation = regionLocations.getRegionLocation(hri.getReplicaId());
376      if (oldLocation == null) return;
377      RegionLocations updatedLocations = regionLocations.remove(oldLocation);
378      boolean removed = false;
379      if (updatedLocations != regionLocations) {
380        if (updatedLocations.isEmpty()) {
381          removed = tableLocations.remove(hri.getStartKey(), regionLocations);
382        } else {
383          removed = tableLocations.replace(hri.getStartKey(), regionLocations, updatedLocations);
384        }
385        if (removed && LOG.isTraceEnabled()) {
386          LOG.trace("Removed " + oldLocation + " from cache");
387        }
388      }
389    }
390  }
391
392  public void clearCache(final HRegionLocation location) {
393    if (location == null) {
394      return;
395    }
396    TableName tableName = location.getRegionInfo().getTable();
397    ConcurrentMap<byte[], RegionLocations> tableLocations = getTableLocations(tableName);
398    RegionLocations regionLocations = tableLocations.get(location.getRegionInfo().getStartKey());
399    if (regionLocations != null) {
400      RegionLocations updatedLocations = regionLocations.remove(location);
401      boolean removed = false;
402      if (updatedLocations != regionLocations) {
403        if (updatedLocations.isEmpty()) {
404          removed = tableLocations.remove(location.getRegionInfo().getStartKey(), regionLocations);
405        } else {
406          removed = tableLocations.replace(location.getRegionInfo().getStartKey(), regionLocations,
407              updatedLocations);
408        }
409        if (removed && LOG.isTraceEnabled()) {
410          LOG.trace("Removed " + location + " from cache");
411        }
412      }
413    }
