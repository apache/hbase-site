046/**
047 * Callable that handles the <code>multi</code> method call going against a single
048 * regionserver; i.e. A RegionServerCallable for the multi call (It is NOT a
049 * RegionServerCallable that goes against multiple regions).
050 * @param <R>
051 */
052@InterfaceAudience.Private
053class MultiServerCallable extends CancellableRegionServerCallable<MultiResponse> {
054  private MultiAction multiAction;
055  private boolean cellBlock;
056
057  MultiServerCallable(final ClusterConnection connection, final TableName tableName,
058      final ServerName location, final MultiAction multi, RpcController rpcController,
059      int rpcTimeout, RetryingTimeTracker tracker, int priority) {
060    super(connection, tableName, null, rpcController, rpcTimeout, tracker, priority);
061    this.multiAction = multi;
062    // RegionServerCallable has HRegionLocation field, but this is a multi-region request.
063    // Using region info from parent HRegionLocation would be a mistake for this class; so
064    // we will store the server here, and throw if someone tries to obtain location/regioninfo.
065    this.location = new HRegionLocation(null, location);
066    this.cellBlock = isCellBlock();
067  }
068
069  public void reset(ServerName location, MultiAction multiAction) {
070    this.location = new HRegionLocation(null, location);
071    this.multiAction = multiAction;
072    this.cellBlock = isCellBlock();
073  }
074
075  @Override
076  protected HRegionLocation getLocation() {
077    throw new RuntimeException("Cannot get region location for multi-region request");
078  }
079
080  @Override
081  public HRegionInfo getHRegionInfo() {
082    throw new RuntimeException("Cannot get region info for multi-region request");
083  }
084
085  MultiAction getMulti() {
086    return this.multiAction;
087  }
088
089  @Override
090  protected MultiResponse rpcCall() throws Exception {
091    int countOfActions = this.multiAction.size();
092    if (countOfActions <= 0) throw new DoNotRetryIOException("No Actions");
093    MultiRequest.Builder multiRequestBuilder = MultiRequest.newBuilder();
094    RegionAction.Builder regionActionBuilder = RegionAction.newBuilder();
095    ClientProtos.Action.Builder actionBuilder = ClientProtos.Action.newBuilder();
096    MutationProto.Builder mutationBuilder = MutationProto.newBuilder();
097
098    // Pre-size. Presume at least a KV per Action. There are likely more.
099    List<CellScannable> cells =
100        (this.cellBlock ? new ArrayList<CellScannable>(countOfActions) : null);
101
102    long nonceGroup = multiAction.getNonceGroup();
103    if (nonceGroup != HConstants.NO_NONCE) {
104      multiRequestBuilder.setNonceGroup(nonceGroup);
105    }
106    // Index to track RegionAction within the MultiRequest
107    int regionActionIndex = -1;
108    // Map from a created RegionAction to the original index for a RowMutations within
109    // its original list of actions
110    Map<Integer, Integer> rowMutationsIndexMap = new HashMap<>();
111    // The multi object is a list of Actions by region. Iterate by region.
112    for (Map.Entry<byte[], List<Action>> e: this.multiAction.actions.entrySet()) {
113      final byte [] regionName = e.getKey();
114      final List<Action> actions = e.getValue();
115      regionActionBuilder.clear();
116      regionActionBuilder.setRegion(RequestConverter.buildRegionSpecifier(
117          HBaseProtos.RegionSpecifier.RegionSpecifierType.REGION_NAME, regionName));
118
119      int rowMutations = 0;
120      for (Action action : actions) {
121        Row row = action.getAction();
122        // Row Mutations are a set of Puts and/or Deletes all to be applied atomically
123        // on the one row. We do separate RegionAction for each RowMutations.
124        // We maintain a map to keep track of this RegionAction and the original Action index.
125        if (row instanceof RowMutations) {
126          RowMutations rms = (RowMutations)row;
127          if (this.cellBlock) {
128            // Build a multi request absent its Cell payload. Send data in cellblocks.
129            regionActionBuilder = RequestConverter.buildNoDataRegionAction(regionName, rms, cells,
130              regionActionBuilder, actionBuilder, mutationBuilder);
131          } else {
132            regionActionBuilder = RequestConverter.buildRegionAction(regionName, rms);
133          }
134          regionActionBuilder.setAtomic(true);
135          multiRequestBuilder.addRegionAction(regionActionBuilder.build());
136          regionActionIndex++;
137          rowMutationsIndexMap.put(regionActionIndex, action.getOriginalIndex());
138          rowMutations++;
139        }
140      }
141
142      if (actions.size() > rowMutations) {
143        if (this.cellBlock) {
144          // Send data in cellblocks. The call to buildNoDataRegionAction will skip RowMutations.
145          // They have already been handled above. Guess at count of cells
146          regionActionBuilder = RequestConverter.buildNoDataRegionAction(regionName, actions, cells,
147            regionActionBuilder, actionBuilder, mutationBuilder);
148        } else {
149          regionActionBuilder = RequestConverter.buildRegionAction(regionName, actions,
150            regionActionBuilder, actionBuilder, mutationBuilder);
151        }
152        multiRequestBuilder.addRegionAction(regionActionBuilder.build());
153        regionActionIndex++;
154      }
155    }
156
157    if (cells != null) {
158      setRpcControllerCellScanner(CellUtil.createCellScanner(cells));
159    }
160    ClientProtos.MultiRequest requestProto = multiRequestBuilder.build();
161    ClientProtos.MultiResponse responseProto = getStub().multi(getRpcController(), requestProto);
162    if (responseProto == null) return null; // Occurs on cancel
163    return ResponseConverter.getResults(requestProto, rowMutationsIndexMap, responseProto,
164      getRpcControllerCellScanner());
165  }
166
167  /**
168   * @return True if we should send data in cellblocks.  This is an expensive call.  Cache the
169   * result if you can rather than call each time.
170   */
171  private boolean isCellBlock() {
172    // This is not exact -- the configuration could have changed on us after connection was set up
173    // but it will do for now.
174    ClusterConnection conn = getConnection();
175    return conn.hasCellBlockSupport();
176  }
177
178  @Override
179  public void prepare(boolean reload) throws IOException {
180    // Use the location we were given in the constructor rather than go look it up.
181    setStub(getConnection().getClient(this.location.getServerName()));
182  }
183
184  @VisibleForTesting
185  ServerName getServerName() {
186    return location.getServerName();
187  }
