018package org.apache.hadoop.hbase.client;
019
020import com.google.protobuf.RpcCallback;
021import java.io.IOException;
022import java.io.InterruptedIOException;
023import org.apache.yetus.audience.InterfaceAudience;
024
025/**
026 * For implementation coprocessor related methods in {@link Table} and {@link Admin} interface.
027 * @deprecated since 3.0.0, will be removed in 4.0.0 along with the coprocessor related methods in
028 *             {@link Table} and {@link Admin} interface.
029 */
030@Deprecated
031@InterfaceAudience.Private
032class CoprocessorBlockingRpcCallback<R> implements RpcCallback<R> {
033  private R result;
034  private boolean resultSet = false;
035
036  /**
037   * Called on completion of the RPC call with the response object, or {@code null} in the case of
038   * an error.
039   * @param parameter the response object or {@code null} if an error occurred
040   */
041  @Override
042  public void run(R parameter) {
043    synchronized (this) {
044      result = parameter;
045      resultSet = true;
046      this.notifyAll();
047    }
048  }
049
050  /**
051   * Returns the parameter passed to {@link #run(Object)} or {@code null} if a null value was
052   * passed. When used asynchronously, this method will block until the {@link #run(Object)} method
053   * has been called.
054   * @return the response object or {@code null} if no response was passed
055   */
056  public synchronized R get() throws IOException {
057    while (!resultSet) {
058      try {
059        this.wait();
060      } catch (InterruptedException ie) {
061        InterruptedIOException exception = new InterruptedIOException(ie.getMessage());
062        exception.initCause(ie);
063        throw exception;
064      }
065    }
066    return result;
067  }
068}
