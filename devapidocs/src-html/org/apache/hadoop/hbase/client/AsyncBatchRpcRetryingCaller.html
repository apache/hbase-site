065/**
066 * Retry caller for batch.
067 * <p>
068 * Notice that, the {@link #operationTimeoutNs} is the total time limit now which is the same with
069 * other single operations
070 * <p>
071 * And the {@link #maxAttempts} is a limit for each single operation in the batch logically. In the
072 * implementation, we will record a {@code tries} parameter for each operation group, and if it is
073 * split to several groups when retrying, the sub groups will inherit the {@code tries}. You can
074 * imagine that the whole retrying process is a tree, and the {@link #maxAttempts} is the limit of
075 * the depth of the tree.
076 */
077@InterfaceAudience.Private
078class AsyncBatchRpcRetryingCaller<T> {
079
080  private static final Log LOG = LogFactory.getLog(AsyncBatchRpcRetryingCaller.class);

108  // we can not use HRegionLocation as the map key because the hashCode and equals method of
109  // HRegionLocation only consider serverName.
110  private static final class RegionRequest {
111
112    public final HRegionLocation loc;
113
114    public final ConcurrentLinkedQueue<Action> actions = new ConcurrentLinkedQueue<>();
115
116    public RegionRequest(HRegionLocation loc) {
117      this.loc = loc;
118    }
119  }

121  private static final class ServerRequest {
122
123    public final ConcurrentMap<byte[], RegionRequest> actionsByRegion =
124        new ConcurrentSkipListMap<>(Bytes.BYTES_COMPARATOR);
125
126    public void addAction(HRegionLocation loc, Action action) {
127      computeIfAbsent(actionsByRegion, loc.getRegionInfo().getRegionName(),
128        () -> new RegionRequest(loc)).actions.add(action);
129    }

132  public AsyncBatchRpcRetryingCaller(HashedWheelTimer retryTimer, AsyncConnectionImpl conn,
133      TableName tableName, List<? extends Row> actions, long pauseNs, int maxAttempts,
134      long operationTimeoutNs, long rpcTimeoutNs, int startLogErrorsCnt) {
135    this.retryTimer = retryTimer;
136    this.conn = conn;
137    this.tableName = tableName;
138    this.pauseNs = pauseNs;
139    this.maxAttempts = maxAttempts;
140    this.operationTimeoutNs = operationTimeoutNs;
141    this.rpcTimeoutNs = rpcTimeoutNs;
142    this.startLogErrorsCnt = startLogErrorsCnt;
143
144    this.actions = new ArrayList<>(actions.size());
145    this.futures = new ArrayList<>(actions.size());
146    this.action2Future = new IdentityHashMap<>(actions.size());
147    for (int i = 0, n = actions.size(); i < n; i++) {
148      Row rawAction = actions.get(i);
149      Action action = new Action(rawAction, i);
150      if (rawAction instanceof Append || rawAction instanceof Increment) {
151        action.setNonce(conn.getNonceGenerator().newNonce());
152      }
153      this.actions.add(action);
154      CompletableFuture<T> future = new CompletableFuture<>();
155      futures.add(future);
156      action2Future.put(action, future);
157    }
158    this.action2Errors = new IdentityHashMap<>();
159    this.startNs = System.nanoTime();
160  }
161
162  private long remainingTimeNs() {
163    return operationTimeoutNs - (System.nanoTime() - startNs);
164  }
165
166  private List<ThrowableWithExtraContext> removeErrors(Action action) {
167    synchronized (action2Errors) {
168      return action2Errors.remove(action);
169    }
170  }
171
172  private void logException(int tries, Supplier<Stream<RegionRequest>> regionsSupplier,
173      Throwable error, ServerName serverName) {
174    if (tries > startLogErrorsCnt) {
175      String regions =
176          regionsSupplier.get().map(r -> "'" + r.loc.getRegionInfo().getRegionNameAsString() + "'")
177              .collect(Collectors.joining(",", "[", "]"));
178      LOG.warn("Process batch for " + regions + " in " + tableName + " from " + serverName
179          + " failed, tries=" + tries,
180        error);
181    }
182  }
183
184  private String getExtraContextForError(ServerName serverName) {
185    return serverName != null ? serverName.getServerName() : "";
186  }
187
188  private void addError(Action action, Throwable error, ServerName serverName) {
189    List<ThrowableWithExtraContext> errors;
190    synchronized (action2Errors) {
191      errors = action2Errors.computeIfAbsent(action, k -> new ArrayList<>());
192    }
193    errors.add(new ThrowableWithExtraContext(error, EnvironmentEdgeManager.currentTime(),
194        getExtraContextForError(serverName)));
195  }
196
197  private void addError(Iterable<Action> actions, Throwable error, ServerName serverName) {
198    actions.forEach(action -> addError(action, error, serverName));
199  }
200
201  private void failOne(Action action, int tries, Throwable error, long currentTime, String extras) {
202    CompletableFuture<T> future = action2Future.get(action);
203    if (future.isDone()) {
204      return;
205    }
206    ThrowableWithExtraContext errorWithCtx =
207        new ThrowableWithExtraContext(error, currentTime, extras);
208    List<ThrowableWithExtraContext> errors = removeErrors(action);
209    if (errors == null) {
210      errors = Collections.singletonList(errorWithCtx);
211    } else {
212      errors.add(errorWithCtx);
213    }
214    future.completeExceptionally(new RetriesExhaustedException(tries - 1, errors));
215  }
216
217  private void failAll(Stream<Action> actions, int tries, Throwable error, ServerName serverName) {
218    long currentTime = EnvironmentEdgeManager.currentTime();
219    String extras = getExtraContextForError(serverName);
220    actions.forEach(action -> failOne(action, tries, error, currentTime, extras));
221  }
222
223  private void failAll(Stream<Action> actions, int tries) {
224    actions.forEach(action -> {
225      CompletableFuture<T> future = action2Future.get(action);
226      if (future.isDone()) {
227        return;
228      }
229      future.completeExceptionally(new RetriesExhaustedException(tries,
230          Optional.ofNullable(removeErrors(action)).orElse(Collections.emptyList())));
231    });
232  }
233
234  private ClientProtos.MultiRequest buildReq(Map<byte[], RegionRequest> actionsByRegion,
235      List<CellScannable> cells, Map<Integer, Integer> rowMutationsIndexMap) throws IOException {
236    ClientProtos.MultiRequest.Builder multiRequestBuilder = ClientProtos.MultiRequest.newBuilder();
237    ClientProtos.RegionAction.Builder regionActionBuilder = ClientProtos.RegionAction.newBuilder();
238    ClientProtos.Action.Builder actionBuilder = ClientProtos.Action.newBuilder();
239    ClientProtos.MutationProto.Builder mutationBuilder = ClientProtos.MutationProto.newBuilder();
240    for (Map.Entry<byte[], RegionRequest> entry : actionsByRegion.entrySet()) {
241      long nonceGroup = conn.getNonceGenerator().getNonceGroup();
242      // multiRequestBuilder will be populated with region actions.
243      // rowMutationsIndexMap will be non-empty after the call if there is RowMutations in the
244      // action list.
245      RequestConverter.buildNoDataRegionActions(entry.getKey(),
246        entry.getValue().actions, cells, multiRequestBuilder, regionActionBuilder, actionBuilder,
247        mutationBuilder, nonceGroup, rowMutationsIndexMap);
248    }
249    return multiRequestBuilder.build();
250  }
251
252  @SuppressWarnings("unchecked")
253  private void onComplete(Action action, RegionRequest regionReq, int tries, ServerName serverName,
254      RegionResult regionResult, List<Action> failedActions) {
255    Object result = regionResult.result.get(action.getOriginalIndex());
256    if (result == null) {
257      LOG.error("Server " + serverName + " sent us neither result nor exception for row '"
258          + Bytes.toStringBinary(action.getAction().getRow()) + "' of "
259          + regionReq.loc.getRegionInfo().getRegionNameAsString());
260      addError(action, new RuntimeException("Invalid response"), serverName);
261      failedActions.add(action);
262    } else if (result instanceof Throwable) {
263      Throwable error = translateException((Throwable) result);
264      logException(tries, () -> Stream.of(regionReq), error, serverName);
265      if (error instanceof DoNotRetryIOException || tries >= maxAttempts) {
266        failOne(action, tries, error, EnvironmentEdgeManager.currentTime(),
267          getExtraContextForError(serverName));
268      } else {
269        failedActions.add(action);
270      }
271    } else {
272      action2Future.get(action).complete((T) result);
273    }
274  }
275
276  private void onComplete(Map<byte[], RegionRequest> actionsByRegion, int tries,
277      ServerName serverName, MultiResponse resp) {
278    List<Action> failedActions = new ArrayList<>();
279    actionsByRegion.forEach((rn, regionReq) -> {
280      RegionResult regionResult = resp.getResults().get(rn);
281      if (regionResult != null) {
282        regionReq.actions.forEach(
283          action -> onComplete(action, regionReq, tries, serverName, regionResult, failedActions));
284      } else {
285        Throwable t = resp.getException(rn);
286        Throwable error;
287        if (t == null) {
288          LOG.error(
289            "Server sent us neither results nor exceptions for " + Bytes.toStringBinary(rn));
290          error = new RuntimeException("Invalid response");
291        } else {
292          error = translateException(t);
293          logException(tries, () -> Stream.of(regionReq), error, serverName);
294          conn.getLocator().updateCachedLocation(regionReq.loc, error);
295          if (error instanceof DoNotRetryIOException || tries >= maxAttempts) {
296            failAll(regionReq.actions.stream(), tries, error, serverName);
297            return;
298          }
299          addError(regionReq.actions, error, serverName);
300          failedActions.addAll(regionReq.actions);
301        }
302      }
303    });
304    if (!failedActions.isEmpty()) {
305      tryResubmit(failedActions.stream(), tries);
306    }
307  }

309  private void send(Map<ServerName, ServerRequest> actionsByServer, int tries) {
310    long remainingNs;
311    if (operationTimeoutNs > 0) {
312      remainingNs = remainingTimeNs();
313      if (remainingNs <= 0) {
314        failAll(actionsByServer.values().stream().flatMap(m -> m.actionsByRegion.values().stream())
315            .flatMap(r -> r.actions.stream()),
316          tries);
317        return;
318      }
319    } else {
320      remainingNs = Long.MAX_VALUE;
321    }
322    actionsByServer.forEach((sn, serverReq) -> {
323      ClientService.Interface stub;
324      try {
325        stub = conn.getRegionServerStub(sn);
326      } catch (IOException e) {
327        onError(serverReq.actionsByRegion, tries, e, sn);
328        return;
329      }
330      ClientProtos.MultiRequest req;
331      List<CellScannable> cells = new ArrayList<>();
332      // Map from a created RegionAction to the original index for a RowMutations within
333      // the original list of actions. This will be used to process the results when there
334      // is RowMutations in the action list.
335      Map<Integer, Integer> rowMutationsIndexMap = new HashMap<>();
336      try {
337        req = buildReq(serverReq.actionsByRegion, cells, rowMutationsIndexMap);
338      } catch (IOException e) {
339        onError(serverReq.actionsByRegion, tries, e, sn);
340        return;
341      }
342      HBaseRpcController controller = conn.rpcControllerFactory.newController();
343      resetController(controller, Math.min(rpcTimeoutNs, remainingNs));
344      if (!cells.isEmpty()) {
345        controller.setCellScanner(createCellScanner(cells));
346      }
347      stub.multi(controller, req, resp -> {
348        if (controller.failed()) {
349          onError(serverReq.actionsByRegion, tries, controller.getFailed(), sn);
350        } else {
351          try {
352            onComplete(serverReq.actionsByRegion, tries, sn, ResponseConverter.getResults(req,
353              rowMutationsIndexMap, resp, controller.cellScanner()));
354          } catch (Exception e) {
355            onError(serverReq.actionsByRegion, tries, e, sn);
356            return;
357          }
358        }
359      });
360    });
361  }
362
363  private void onError(Map<byte[], RegionRequest> actionsByRegion, int tries, Throwable t,
364      ServerName serverName) {
365    Throwable error = translateException(t);
366    logException(tries, () -> actionsByRegion.values().stream(), error, serverName);
367    if (error instanceof DoNotRetryIOException || tries >= maxAttempts) {
368      failAll(actionsByRegion.values().stream().flatMap(r -> r.actions.stream()), tries, error,
369        serverName);
370      return;
371    }
372    List<Action> copiedActions = actionsByRegion.values().stream().flatMap(r -> r.actions.stream())
373        .collect(Collectors.toList());
374    addError(copiedActions, error, serverName);
375    tryResubmit(copiedActions.stream(), tries);
376  }
377
378  private void tryResubmit(Stream<Action> actions, int tries) {
379    long delayNs;
380    if (operationTimeoutNs > 0) {
381      long maxDelayNs = remainingTimeNs() - SLEEP_DELTA_NS;
382      if (maxDelayNs <= 0) {
383        failAll(actions, tries);
384        return;
385      }
386      delayNs = Math.min(maxDelayNs, getPauseTime(pauseNs, tries - 1));
387    } else {
388      delayNs = getPauseTime(pauseNs, tries - 1);
389    }
390    retryTimer.newTimeout(t -> groupAndSend(actions, tries + 1), delayNs, TimeUnit.NANOSECONDS);
391  }
392
393  private void groupAndSend(Stream<Action> actions, int tries) {
394    long locateTimeoutNs;
395    if (operationTimeoutNs > 0) {
396      locateTimeoutNs = remainingTimeNs();
397      if (locateTimeoutNs <= 0) {
398        failAll(actions, tries);
399        return;
400      }
401    } else {
402      locateTimeoutNs = -1L;
403    }
404    ConcurrentMap<ServerName, ServerRequest> actionsByServer = new ConcurrentHashMap<>();
405    ConcurrentLinkedQueue<Action> locateFailed = new ConcurrentLinkedQueue<>();
406    CompletableFuture.allOf(actions
407        .map(action -> conn.getLocator().getRegionLocation(tableName, action.getAction().getRow(),
408          RegionLocateType.CURRENT, locateTimeoutNs).whenComplete((loc, error) -> {
409            if (error != null) {
410              error = translateException(error);
411              if (error instanceof DoNotRetryIOException) {
412                failOne(action, tries, error, EnvironmentEdgeManager.currentTime(), "");
413                return;
414              }
415              addError(action, error, null);
416              locateFailed.add(action);
417            } else {
418              computeIfAbsent(actionsByServer, loc.getServerName(), ServerRequest::new)
419                  .addAction(loc, action);
420            }
421          }))
422        .toArray(CompletableFuture[]::new)).whenComplete((v, r) -> {
423          if (!actionsByServer.isEmpty()) {
424            send(actionsByServer, tries);
425          }
426          if (!locateFailed.isEmpty()) {
427            tryResubmit(locateFailed.stream(), tries);
428          }
429        });
430  }
431
432  public List<CompletableFuture<T>> call() {
433    groupAndSend(actions.stream(), 1);
434    return futures;
435  }
