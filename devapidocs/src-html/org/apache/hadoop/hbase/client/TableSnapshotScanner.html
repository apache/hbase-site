019package org.apache.hadoop.hbase.client;
020
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.List;
025import java.util.UUID;
026
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.fs.FileSystem;
029import org.apache.hadoop.fs.Path;
030import org.apache.hadoop.hbase.CellUtil;
031import org.apache.hadoop.hbase.PrivateCellUtil;
032import org.apache.hadoop.hbase.snapshot.RestoreSnapshotHelper;
033import org.apache.hadoop.hbase.util.FSUtils;
034import org.apache.yetus.audience.InterfaceAudience;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037
038/**
039 * A Scanner which performs a scan over snapshot files. Using this class requires copying the
040 * snapshot to a temporary empty directory, which will copy the snapshot reference files into that
041 * directory. Actual data files are not copied.
042 *
043 * <p>
044 * This also allows one to run the scan from an
045 * online or offline hbase cluster. The snapshot files can be exported by using the
046 * org.apache.hadoop.hbase.snapshot.ExportSnapshot tool,
047 * to a pure-hdfs cluster, and this scanner can be used to
048 * run the scan directly over the snapshot files. The snapshot should not be deleted while there
049 * are open scanners reading from snapshot files.
050 *
051 * <p>
052 * An internal RegionScanner is used to execute the {@link Scan} obtained
053 * from the user for each region in the snapshot.
054 * <p>
055 * HBase owns all the data and snapshot files on the filesystem. Only the HBase user can read from
056 * snapshot files and data files. HBase also enforces security because all the requests are handled
057 * by the server layer, and the user cannot read from the data files directly. To read from snapshot
058 * files directly from the file system, the user who is running the MR job must have sufficient
059 * permissions to access snapshot and reference files. This means that to run mapreduce over
060 * snapshot files, the job has to be run as the HBase user or the user must have group or other
061 * priviledges in the filesystem (See HBASE-8369). Note that, given other users access to read from
062 * snapshot/data files will completely circumvent the access control enforced by HBase.
063 * See org.apache.hadoop.hbase.mapreduce.TableSnapshotInputFormat.
064 */
065@InterfaceAudience.Private
066public class TableSnapshotScanner extends AbstractClientScanner {
067
068  private static final Logger LOG = LoggerFactory.getLogger(TableSnapshotScanner.class);
069
070  private Configuration conf;
071  private String snapshotName;
072  private FileSystem fs;
073  private Path rootDir;
074  private Path restoreDir;
075  private Scan scan;
076  private ArrayList<RegionInfo> regions;
077  private TableDescriptor htd;
078
079  private ClientSideRegionScanner currentRegionScanner  = null;
080  private int currentRegion = -1;
081
082  /**
083   * Creates a TableSnapshotScanner.
084   * @param conf the configuration
085   * @param restoreDir a temporary directory to copy the snapshot files into. Current user should
086   * have write permissions to this directory, and this should not be a subdirectory of rootdir.
087   * The scanner deletes the contents of the directory once the scanner is closed.
088   * @param snapshotName the name of the snapshot to read from
089   * @param scan a Scan representing scan parameters
090   * @throws IOException in case of error
091   */
092  public TableSnapshotScanner(Configuration conf, Path restoreDir,
093      String snapshotName, Scan scan) throws IOException {
094    this(conf, FSUtils.getRootDir(conf), restoreDir, snapshotName, scan);
095  }
096
097  /**
098   * Creates a TableSnapshotScanner.
099   * @param conf the configuration
100   * @param rootDir root directory for HBase.
101   * @param restoreDir a temporary directory to copy the snapshot files into. Current user should
102   * have write permissions to this directory, and this should not be a subdirectory of rootdir.
103   * The scanner deletes the contents of the directory once the scanner is closed.
104   * @param snapshotName the name of the snapshot to read from
105   * @param scan a Scan representing scan parameters
106   * @throws IOException in case of error
107   */
108  public TableSnapshotScanner(Configuration conf, Path rootDir,
109      Path restoreDir, String snapshotName, Scan scan) throws IOException {
110    this.conf = conf;
111    this.snapshotName = snapshotName;
112    this.rootDir = rootDir;
113    // restoreDir will be deleted in close(), use a unique sub directory
114    this.restoreDir = new Path(restoreDir, UUID.randomUUID().toString());
115    this.scan = scan;
116    this.fs = rootDir.getFileSystem(conf);
117    init();
118  }
119
120  private void init() throws IOException {
121    final RestoreSnapshotHelper.RestoreMetaChanges meta =
122      RestoreSnapshotHelper.copySnapshotForScanner(
123        conf, fs, rootDir, restoreDir, snapshotName);
124    final List<RegionInfo> restoredRegions = meta.getRegionsToAdd();
125
126    htd = meta.getTableDescriptor();
127    regions = new ArrayList<>(restoredRegions.size());
128    for (RegionInfo hri : restoredRegions) {
129      if (hri.isOffline() && (hri.isSplit() || hri.isSplitParent())) {
130        continue;
131      }
132      if (PrivateCellUtil.overlappingKeys(scan.getStartRow(), scan.getStopRow(), hri.getStartKey(),
133        hri.getEndKey())) {
134        regions.add(hri);
135      }
136    }
137
138    // sort for regions according to startKey.
139    Collections.sort(regions, RegionInfo.COMPARATOR);
140    initScanMetrics(scan);
141  }
142
143  @Override
144  public Result next() throws IOException {
145    Result result = null;
146    while (true) {
147      if (currentRegionScanner == null) {
148        currentRegion++;
149        if (currentRegion >= regions.size()) {
150          return null;
151        }
152
153        RegionInfo hri = regions.get(currentRegion);
154        currentRegionScanner = new ClientSideRegionScanner(conf, fs,
155          restoreDir, htd, hri, scan, scanMetrics);
156        if (this.scanMetrics != null) {
157          this.scanMetrics.countOfRegions.incrementAndGet();
158        }
159      }
160
161      try {
162        result = currentRegionScanner.next();
163        if (result != null) {
164          return result;
165        }
166      } finally {
167        if (result == null) {
168          currentRegionScanner.close();
169          currentRegionScanner = null;
170        }
171      }
172    }
173  }
174
175  @Override
176  public void close() {
177    if (currentRegionScanner != null) {
178      currentRegionScanner.close();
179    }
180    try {
181      fs.delete(this.restoreDir, true);
182    } catch (IOException ex) {
183      LOG.warn("Could not delete restore directory for the snapshot:" + ex);
184    }
185  }
186
187  @Override
188  public boolean renewLease() {
189    throw new UnsupportedOperationException();
190  }
191
192}
