019package org.apache.hadoop.hbase.client;
020
021import org.apache.hadoop.conf.Configuration;
022import org.apache.hadoop.hbase.TableName;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.classification.InterfaceStability;
025
026import java.io.Closeable;
027import java.io.IOException;
028import java.util.List;
029
030/**
031 * <p>Used to communicate with a single HBase table similar to {@link Table} but meant for
032 * batched, asynchronous puts. Obtain an instance from a {@link Connection} and call
033 * {@link #close()} afterwards. Customizations can be applied to the {@code BufferedMutator} via
034 * the {@link BufferedMutatorParams}.
035 * </p>
036 *
037 * <p>Exception handling with asynchronously via the {@link BufferedMutator.ExceptionListener}.
038 * The default implementation is to throw the exception upon receipt. This behavior can be
039 * overridden with a custom implementation, provided as a parameter with
040 * {@link BufferedMutatorParams#listener(BufferedMutator.ExceptionListener)}.</p>
041 *
042 * <p>Map/Reduce jobs are good use cases for using {@code BufferedMutator}. Map/reduce jobs
043 * benefit from batching, but have no natural flush point. {@code BufferedMutator} receives the
044 * puts from the M/R job and will batch puts based on some heuristic, such as the accumulated size
045 * of the puts, and submit batches of puts asynchronously so that the M/R logic can continue
046 * without interruption.
047 * </p>
048 *
049 * <p>{@code BufferedMutator} can also be used on more exotic circumstances. Map/Reduce batch jobs
050 * will have a single {@code BufferedMutator} per thread. A single {@code BufferedMutator} can
051 * also be effectively used in high volume online systems to batch puts, with the caveat that
052 * extreme circumstances, such as JVM or machine failure, may cause some data loss.</p>
053 *
054 * <p>NOTE: This class replaces the functionality that used to be available via
055 * HTable#setAutoFlush(boolean) set to {@code false}.
056 * </p>
057 *
058 * <p>See also the {@code BufferedMutatorExample} in the hbase-examples module.</p>
059 * @see ConnectionFactory
060 * @see Connection
061 * @since 1.0.0
062 */
063@InterfaceAudience.Public
064@InterfaceStability.Evolving
065public interface BufferedMutator extends Closeable {
066  /**
067   * Gets the fully qualified table name instance of the table that this BufferedMutator writes to.
068   */
069  TableName getName();
070
071  /**
072   * Returns the {@link org.apache.hadoop.conf.Configuration} object used by this instance.
073   * <p>
074   * The reference returned is not a copy, so any change made to it will
075   * affect this instance.
076   */
077  Configuration getConfiguration();
078
079  /**
080   * Sends a {@link Mutation} to the table. The mutations will be buffered and sent over the
081   * wire as part of a batch. Currently only supports {@link Put} and {@link Delete} mutations.
082   *
083   * @param mutation The data to send.
084   * @throws IOException if a remote or network exception occurs.
085   */
086  void mutate(Mutation mutation) throws IOException;
087
088  /**
089   * Send some {@link Mutation}s to the table. The mutations will be buffered and sent over the
090   * wire as part of a batch. There is no guarantee of sending entire content of {@code mutations}
091   * in a single batch; it will be broken up according to the write buffer capacity.
092   *
093   * @param mutations The data to send.
094   * @throws IOException if a remote or network exception occurs.
095   */
096  void mutate(List<? extends Mutation> mutations) throws IOException;
097
098  /**
099   * Performs a {@link #flush()} and releases any resources held.
100   *
101   * @throws IOException if a remote or network exception occurs.
102   */
103  @Override
104  void close() throws IOException;
105
106  /**
107   * Executes all the buffered, asynchronous {@link Mutation} operations and waits until they
108   * are done.
109   *
110   * @throws IOException if a remote or network exception occurs.
111   */
112  void flush() throws IOException;
113
114  /**
115   * Returns the maximum size in bytes of the write buffer for this HTable.
116   * <p>
117   * The default value comes from the configuration parameter {@code hbase.client.write.buffer}.
118   * @return The size of the write buffer in bytes.
119   */
120  long getWriteBufferSize();
121
122  /**
123   * Set rpc timeout for this mutator instance
124   */
125  void setRpcTimeout(int timeout);
126
127  /**
128   * Set operation timeout for this mutator instance
129   */
130  void setOperationTimeout(int timeout);
131
132  /**
133   * Listens for asynchronous exceptions on a {@link BufferedMutator}.
134   */
135  @InterfaceAudience.Public
136  @InterfaceStability.Evolving
137  interface ExceptionListener {
138    public void onException(RetriesExhaustedWithDetailsException exception,
139        BufferedMutator mutator) throws RetriesExhaustedWithDetailsException;
140  }
141}
