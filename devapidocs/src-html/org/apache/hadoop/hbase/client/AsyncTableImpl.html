033/**
034 * The implementation of AsyncTable. Based on {@link RawAsyncTable}.
035 */
036@InterfaceAudience.Private
037class AsyncTableImpl implements AsyncTable {
038
039  private final RawAsyncTable rawTable;
040
041  private final ExecutorService pool;
042
043  private final long defaultScannerMaxResultSize;
044
045  AsyncTableImpl(AsyncConnectionImpl conn, RawAsyncTable rawTable, ExecutorService pool) {
046    this.rawTable = rawTable;
047    this.pool = pool;
048    this.defaultScannerMaxResultSize = conn.connConf.getScannerMaxResultSize();
049  }
050

081  @Override

086  private <T> CompletableFuture<T> wrap(CompletableFuture<T> future) {
087    CompletableFuture<T> asyncFuture = new CompletableFuture<>();
088    future.whenCompleteAsync((r, e) -> {
089      if (e != null) {
090        asyncFuture.completeExceptionally(e);
091      } else {
092        asyncFuture.complete(r);
093      }
094    }, pool);
095    return asyncFuture;
096  }
097

099  public CompletableFuture<Result> get(Get get) {

101  }

150
151  private long resultSize2CacheSize(long maxResultSize) {
152    // * 2 if possible
153    return maxResultSize > Long.MAX_VALUE / 2 ? maxResultSize : maxResultSize * 2;
154  }
155
156  @Override
157  public ResultScanner getScanner(Scan scan) {
158    return new AsyncTableResultScanner(rawTable, ReflectionUtils.newInstance(scan.getClass(), scan),
159        resultSize2CacheSize(
160          scan.getMaxResultSize() > 0 ? scan.getMaxResultSize() : defaultScannerMaxResultSize));
161  }
162
163  private void scan0(Scan scan, ScanResultConsumer consumer) {
164    try (ResultScanner scanner = getScanner(scan)) {
165      for (Result result; (result = scanner.next()) != null;) {
166        if (!consumer.onNext(result)) {
167          break;
168        }
169      }
170      consumer.onComplete();
171    } catch (IOException e) {
172      consumer.onError(e);
173    }
174  }
175
176  @Override
177  public void scan(Scan scan, ScanResultConsumer consumer) {
178    pool.execute(() -> scan0(scan, consumer));
179  }
180
181  @Override
182  public List<CompletableFuture<Result>> get(List<Get> gets) {
183    return rawTable.get(gets).stream().map(this::wrap).collect(toList());
184  }

189  }
190

193    return rawTable.delete(deletes).stream().map(this::wrap).collect(toList());
194  }
195
196  @Override
197  public <T> List<CompletableFuture<T>> batch(List<? extends Row> actions) {
198    return rawTable.<T> batch(actions).stream().map(this::wrap).collect(toList());
199  }
