018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021import java.util.concurrent.TimeUnit;
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
024import org.apache.yetus.audience.InterfaceAudience;
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028import org.apache.hbase.thirdparty.com.google.common.base.Preconditions;
029
030/**
031 * Thread safe utility that keeps registry end points used by {@link ConnectionRegistry} up to date.
032 * By default the refresh happens periodically (configured via {@code intervalSecsConfigName}). The
033 * refresh can also be triggered on demand via {@link #refreshNow()}. To prevent a flood of
034 * on-demand refreshes we expect that any attempts two should be spaced at least
035 * {@code minIntervalSecsConfigName} seconds apart.
036 */
037@InterfaceAudience.Private
038final class RegistryEndpointsRefresher {
039
040  private static final Logger LOG = LoggerFactory.getLogger(RegistryEndpointsRefresher.class);
041
042  private static final int PERIODIC_REFRESH_INTERVAL_SECS_DEFAULT = 300;
043
044  private static final int MIN_SECS_BETWEEN_REFRESHES_DEFAULT = 60;
045
046  private final Thread thread;
047  private final Refresher refresher;
048  private final long periodicRefreshMs;
049  private final long minTimeBetweenRefreshesMs;
050
051  private boolean refreshNow = false;
052  private boolean stopped = false;
053
054  synchronized void stop() {
055    stopped = true;
056    notifyAll();
057  }
058
059  // The main loop for the refresh thread.
060  private void mainLoop() {
061    long lastRefreshTime = EnvironmentEdgeManager.currentTime();
062    for (;;) {
063      synchronized (this) {
064        for (;;) {
065          if (stopped) {
066            LOG.info("Registry end points refresher loop exited.");
067            return;
068          }
069          // if refreshNow is true, then we will wait until minTimeBetweenRefreshesMs elapsed,
070          // otherwise wait until periodicRefreshMs elapsed
071          long waitTime = (refreshNow ? minTimeBetweenRefreshesMs : periodicRefreshMs) -
072            (EnvironmentEdgeManager.currentTime() - lastRefreshTime);
073          if (waitTime <= 0) {
074            break;
075          }
076          try {
077            wait(waitTime);
078          } catch (InterruptedException e) {
079            LOG.warn("Interrupted during wait", e);
080            Thread.currentThread().interrupt();
081            continue;
082          }
083        }
084        // we are going to refresh, reset this flag
085        refreshNow = false;
086      }
087      LOG.debug("Attempting to refresh registry end points");
088      try {
089        refresher.refresh();
090      } catch (IOException e) {
091        LOG.warn("Error refresh registry end points", e);
092      }
093      // We do not think it is a big deal to fail one time, so no matter what is refresh result, we
094      // just update this refresh time and wait for the next round. If later this becomes critical,
095      // could change to only update this value when we have done a successful refreshing.
096      lastRefreshTime = EnvironmentEdgeManager.currentTime();
097      LOG.debug("Finished refreshing registry end points");
098    }
099  }
100
101  @FunctionalInterface
102  public interface Refresher {
103
104    void refresh() throws IOException;
105  }
106
107  private RegistryEndpointsRefresher(long periodicRefreshMs, long minTimeBetweenRefreshesMs,
108    Refresher refresher) {
109    this.periodicRefreshMs = periodicRefreshMs;
110    this.minTimeBetweenRefreshesMs = minTimeBetweenRefreshesMs;
111    this.refresher = refresher;
112    thread = new Thread(this::mainLoop);
113    thread.setName("Registry-endpoints-refresh-end-points");
114    thread.setDaemon(true);
115    thread.start();
116  }
117
118  /**
119   * Notifies the refresher thread to refresh the configuration. This does not guarantee a refresh.
120   * See class comment for details.
121   */
122  synchronized void refreshNow() {
123    refreshNow = true;
124    notifyAll();
125  }
126
127  /**
128   * Create a {@link RegistryEndpointsRefresher}. If the interval secs configured via
129   * {@code intervalSecsConfigName} is less than zero, will return null here, which means disable
130   * refreshing of endpoints.
131   */
132  static RegistryEndpointsRefresher create(Configuration conf, String intervalSecsConfigName,
133    String minIntervalSecsConfigName, Refresher refresher) {
134    long periodicRefreshMs = TimeUnit.SECONDS
135      .toMillis(conf.getLong(intervalSecsConfigName, PERIODIC_REFRESH_INTERVAL_SECS_DEFAULT));
136    if (periodicRefreshMs <= 0) {
137      return null;
138    }
139    long minTimeBetweenRefreshesMs = TimeUnit.SECONDS
140      .toMillis(conf.getLong(minIntervalSecsConfigName, MIN_SECS_BETWEEN_REFRESHES_DEFAULT));
141    Preconditions.checkArgument(minTimeBetweenRefreshesMs < periodicRefreshMs);
142    return new RegistryEndpointsRefresher(periodicRefreshMs, minTimeBetweenRefreshesMs, refresher);
143  }
144}
