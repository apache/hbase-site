058/**
059 * Fetch the registry data from zookeeper.
060 */
061@InterfaceAudience.Private
062class ZKAsyncRegistry implements AsyncRegistry {
063
064  private static final Log LOG = LogFactory.getLog(ZKAsyncRegistry.class);
065
066  private final CuratorFramework zk;
067
068  private final ZNodePaths znodePaths;
069
070  ZKAsyncRegistry(Configuration conf) {
071    this.znodePaths = new ZNodePaths(conf);
072    int zkSessionTimeout = conf.getInt(ZK_SESSION_TIMEOUT, DEFAULT_ZK_SESSION_TIMEOUT);
073    int zkRetry = conf.getInt("zookeeper.recovery.retry", 30);
074    int zkRetryIntervalMs = conf.getInt("zookeeper.recovery.retry.intervalmill", 1000);
075    this.zk = CuratorFrameworkFactory.builder()
076        .connectString(ZKConfig.getZKQuorumServersString(conf)).sessionTimeoutMs(zkSessionTimeout)
077        .retryPolicy(new RetryNTimes(zkRetry, zkRetryIntervalMs))
078        .threadFactory(
079          Threads.newDaemonThreadFactory(String.format("ZKClusterRegistry-0x%08x", hashCode())))
080        .build();
081    this.zk.start();
082  }
083
084  private interface CuratorEventProcessor<T> {
085    T process(CuratorEvent event) throws Exception;
086  }
087
088  private static <T> CompletableFuture<T> exec(BackgroundPathable<?> opBuilder, String path,
089      CuratorEventProcessor<T> processor) {
090    CompletableFuture<T> future = new CompletableFuture<>();
091    try {
092      opBuilder.inBackground((client, event) -> {
093        try {
094          future.complete(processor.process(event));
095        } catch (Exception e) {
096          future.completeExceptionally(e);
097        }
098      }).withUnhandledErrorListener((msg, e) -> future.completeExceptionally(e)).forPath(path);
099    } catch (Exception e) {
100      future.completeExceptionally(e);
101    }
102    return future;
103  }
104
105  private static String getClusterId(CuratorEvent event) throws DeserializationException {
106    byte[] data = event.getData();
107    if (data == null || data.length == 0) {
108      return null;
109    }
110    data = removeMetaData(data);
111    return ClusterId.parseFrom(data).toString();
112  }
113
114  @Override
115  public CompletableFuture<String> getClusterId() {
116    return exec(zk.getData(), znodePaths.clusterIdZNode, ZKAsyncRegistry::getClusterId);
117  }
118
119  @VisibleForTesting
120  CuratorFramework getCuratorFramework() {
121    return zk;
122  }
123
124  private static ZooKeeperProtos.MetaRegionServer getMetaProto(CuratorEvent event)
125      throws IOException {
126    byte[] data = event.getData();
127    if (data == null || data.length == 0) {
128      return null;
129    }
130    data = removeMetaData(data);
131    int prefixLen = lengthOfPBMagic();
132    return ZooKeeperProtos.MetaRegionServer.parser().parseFrom(data, prefixLen,
133      data.length - prefixLen);
134  }
135
136  private static void tryComplete(MutableInt remaining, HRegionLocation[] locs,
137      CompletableFuture<RegionLocations> future) {
138    remaining.decrement();
139    if (remaining.intValue() > 0) {
140      return;
141    }
142    future.complete(new RegionLocations(locs));
143  }
144
145  private Pair<RegionState.State, ServerName> getStateAndServerName(
146      ZooKeeperProtos.MetaRegionServer proto) {
147    RegionState.State state;
148    if (proto.hasState()) {
149      state = RegionState.State.convert(proto.getState());
150    } else {
151      state = RegionState.State.OPEN;
152    }
153    HBaseProtos.ServerName snProto = proto.getServer();
154    return Pair.newPair(state,
155      ServerName.valueOf(snProto.getHostName(), snProto.getPort(), snProto.getStartCode()));
156  }
157
158  @Override
159  public CompletableFuture<RegionLocations> getMetaRegionLocation() {
160    CompletableFuture<RegionLocations> future = new CompletableFuture<>();
161    HRegionLocation[] locs = new HRegionLocation[znodePaths.metaReplicaZNodes.size()];
162    MutableInt remaining = new MutableInt(locs.length);
163    znodePaths.metaReplicaZNodes.forEach((replicaId, path) -> {
164      if (replicaId == DEFAULT_REPLICA_ID) {
165        exec(zk.getData(), path, ZKAsyncRegistry::getMetaProto).whenComplete((proto, error) -> {
166          if (error != null) {
167            future.completeExceptionally(error);
168            return;
169          }
170          if (proto == null) {
171            future.completeExceptionally(new IOException("Meta znode is null"));
172            return;
173          }
174          Pair<RegionState.State, ServerName> stateAndServerName = getStateAndServerName(proto);
175          if (stateAndServerName.getFirst() != RegionState.State.OPEN) {
176            future.completeExceptionally(
177              new IOException("Meta region is in state " + stateAndServerName.getFirst()));
178            return;
179          }
180          locs[DEFAULT_REPLICA_ID] = new HRegionLocation(
181              getRegionInfoForDefaultReplica(FIRST_META_REGIONINFO),
182              stateAndServerName.getSecond());
183          tryComplete(remaining, locs, future);
184        });
185      } else {
186        exec(zk.getData(), path, ZKAsyncRegistry::getMetaProto).whenComplete((proto, error) -> {
187          if (future.isDone()) {
188            return;
189          }
190          if (error != null) {
191            LOG.warn("Failed to fetch " + path, error);
192            locs[replicaId] = null;
193          } else if (proto == null) {
194            LOG.warn("Meta znode for replica " + replicaId + " is null");
195            locs[replicaId] = null;
196          } else {
197            Pair<RegionState.State, ServerName> stateAndServerName = getStateAndServerName(proto);
198            if (stateAndServerName.getFirst() != RegionState.State.OPEN) {
199              LOG.warn("Meta region for replica " + replicaId + " is in state "
200                  + stateAndServerName.getFirst());
201              locs[replicaId] = null;
202            } else {
203              locs[replicaId] = new HRegionLocation(
204                  getRegionInfoForReplica(FIRST_META_REGIONINFO, replicaId),
205                  stateAndServerName.getSecond());
206            }
207          }
208          tryComplete(remaining, locs, future);
209        });
210      }
211    });
212    return future;
213  }

215  private static int getCurrentNrHRS(CuratorEvent event) {
216    Stat stat = event.getStat();
217    return stat != null ? stat.getNumChildren() : 0;
218  }
219
220  @Override
221  public CompletableFuture<Integer> getCurrentNrHRS() {
222    return exec(zk.checkExists(), znodePaths.rsZNode, ZKAsyncRegistry::getCurrentNrHRS);
223  }
224
225  private static ZooKeeperProtos.Master getMasterProto(CuratorEvent event) throws IOException {
226    byte[] data = event.getData();
227    if (data == null || data.length == 0) {
228      return null;
229    }
230    data = removeMetaData(data);
231    int prefixLen = lengthOfPBMagic();
232    return ZooKeeperProtos.Master.parser().parseFrom(data, prefixLen, data.length - prefixLen);
233  }
234
235  @Override
236  public CompletableFuture<ServerName> getMasterAddress() {
237    return exec(zk.getData(), znodePaths.masterAddressZNode, ZKAsyncRegistry::getMasterProto)
238        .thenApply(proto -> {
239          if (proto == null) {
240            return null;
241          }
242          HBaseProtos.ServerName snProto = proto.getMaster();
243          return ServerName.valueOf(snProto.getHostName(), snProto.getPort(),
244            snProto.getStartCode());
245        });
246  }
247
248  @Override
249  public CompletableFuture<Integer> getMasterInfoPort() {
250    return exec(zk.getData(), znodePaths.masterAddressZNode, ZKAsyncRegistry::getMasterProto)
251        .thenApply(proto -> proto != null ? proto.getInfoPort() : 0);
252  }
253
254  @Override
255  public void close() {
256    zk.close();
257  }
