337  /**
338   * @param b Family name.
339   * @return <code>b</code>
340   * @throws IllegalArgumentException If not null and not a legitimate family
341   * name: i.e. 'printable' and ends in a ':' (Null passes are allowed because
342   * <code>b</code> can be null when deserializing). Cannot start with a '.'
343   * either. Also Family can not be an empty value or equal "recovered.edits".
344   */
345  public static byte[] isLegalColumnFamilyName(final byte[] b) {
346    if (b == null) {
347      return null;
348    }
349    Preconditions.checkArgument(b.length != 0, "Column Family name can not be empty");
350    if (b[0] == '.') {
351      throw new IllegalArgumentException("Column Family names cannot start with a "
352              + "period: " + Bytes.toString(b));
353    }
354    for (int i = 0; i < b.length; i++) {
355      if (Character.isISOControl(b[i]) || b[i] == ':' || b[i] == '\\' || b[i] == '/') {
356        throw new IllegalArgumentException("Illegal character <" + b[i]
357                + ">. Column Family names cannot contain control characters or colons: "
358                + Bytes.toString(b));
359      }
360    }
361    byte[] recoveredEdit = Bytes.toBytes(HConstants.RECOVERED_EDITS_DIR);
362    if (Bytes.equals(recoveredEdit, b)) {
363      throw new IllegalArgumentException("Column Family name cannot be: "
364              + HConstants.RECOVERED_EDITS_DIR);
365    }
366    return b;
367  }

723      Bytes value = values.get(key);

740    public ModifyableColumnFamilyDescriptor setMaxVersions(int maxVersions) {
741      if (maxVersions <= 0) {
742        // TODO: Allow maxVersion of 0 to be the way you say "Keep all versions".
743        // Until there is support, consider 0 or < 0 -- a configuration error.
744        throw new IllegalArgumentException("Maximum versions must be positive");
745      }
746      if (maxVersions < this.getMinVersions()) {
747        throw new IllegalArgumentException("Set MaxVersion to " + maxVersions
748                + " while minVersion is " + this.getMinVersions()
749                + ". Maximum versions must be >= minimum versions ");
750      }
751      setValue(MAX_VERSIONS_BYTES, Integer.toString(maxVersions));
752      return this;
753    }

1108    private StringBuilder getValues(boolean printDefaults) {
1109      StringBuilder s = new StringBuilder();
1110
1111      boolean hasConfigKeys = false;
1112
1113      // print all reserved keys first
1114      for (Map.Entry<Bytes, Bytes> entry : values.entrySet()) {
1115        if (!RESERVED_KEYWORDS.contains(entry.getKey())) {
1116          hasConfigKeys = true;
1117          continue;
1118        }
1119        String key = Bytes.toString(entry.getKey().get());
1120        String value = Bytes.toStringBinary(entry.getValue().get());
1121        if (printDefaults
1122                || !DEFAULT_VALUES.containsKey(key)
1123                || !DEFAULT_VALUES.get(key).equalsIgnoreCase(value)) {
1124          s.append(", ");
1125          s.append(key);
1126          s.append(" => ");
1127          s.append('\'').append(PrettyPrinter.format(value, getUnit(key))).append('\'');
1128        }
1129      }
1130
1131      // print all non-reserved, advanced config keys as a separate subset
1132      if (hasConfigKeys) {
1133        s.append(", ");
1134        s.append(HConstants.METADATA).append(" => ");
1135        s.append('{');
1136        boolean printComma = false;
1137        for (Map.Entry<Bytes, Bytes> entry : values.entrySet()) {
1138          Bytes k = entry.getKey();
1139          if (RESERVED_KEYWORDS.contains(k)) {
1140            continue;
1141          }
1142          String key = Bytes.toString(k.get());
1143          String value = Bytes.toStringBinary(entry.getValue().get());
1144          if (printComma) {
1145            s.append(", ");
1146          }
1147          printComma = true;
1148          s.append('\'').append(key).append('\'');
1149          s.append(" => ");
1150          s.append('\'').append(PrettyPrinter.format(value, getUnit(key))).append('\'');
1151        }
1152        s.append('}');
1153      }
1154
1155      if (!configuration.isEmpty()) {
1156        s.append(", ");
1157        s.append(HConstants.CONFIGURATION).append(" => ");
1158        s.append('{');
1159        boolean printCommaForConfiguration = false;
1160        for (Map.Entry<String, String> e : configuration.entrySet()) {
1161          if (printCommaForConfiguration) {
1162            s.append(", ");
1163          }
1164          printCommaForConfiguration = true;
1165          s.append('\'').append(e.getKey()).append('\'');
1166          s.append(" => ");
1167          s.append('\'').append(PrettyPrinter.format(e.getValue(), getUnit(e.getKey()))).append('\'');
1168        }
1169        s.append("}");
1170      }
1171      return s;
1172    }

1208    /**
1209     * @param bytes A pb serialized {@link ModifyableColumnFamilyDescriptor} instance with pb
1210     * magic prefix
1211     * @return An instance of {@link ModifyableColumnFamilyDescriptor} made from
1212     * <code>bytes</code>
1213     * @throws DeserializationException
1214     * @see #toByteArray()
1215     */
1216    private static ColumnFamilyDescriptor parseFrom(final byte[] bytes) throws DeserializationException {
1217      if (!ProtobufUtil.isPBMagicPrefix(bytes)) {
1218        throw new DeserializationException("No magic");
1219      }
1220      int pblen = ProtobufUtil.lengthOfPBMagic();
1221      ColumnFamilySchema.Builder builder = ColumnFamilySchema.newBuilder();
1222      ColumnFamilySchema cfs = null;
1223      try {
1224        ProtobufUtil.mergeFrom(builder, bytes, pblen, bytes.length - pblen);
1225        cfs = builder.build();
1226      } catch (IOException e) {
1227        throw new DeserializationException(e);
1228      }
1229      return ProtobufUtil.toColumnFamilyDescriptor(cfs);
1230    }
