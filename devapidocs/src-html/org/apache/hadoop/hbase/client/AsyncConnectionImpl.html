108  public AsyncConnectionImpl(Configuration conf, AsyncRegistry registry, String clusterId,
109      User user) {
110    this.conf = conf;
111    this.user = user;
112    if (user.isLoginFromKeytab()) {
113      spawnRenewalChore(user.getUGI());
114    }
115    this.connConf = new AsyncConnectionConfiguration(conf);
116    this.registry = registry;
117    this.rpcClient = RpcClientFactory.createClient(conf, clusterId);
118    this.rpcControllerFactory = RpcControllerFactory.instantiate(conf);
119    this.hostnameCanChange = conf.getBoolean(RESOLVE_HOSTNAME_ON_FAIL_KEY, true);
120    this.rpcTimeout =
121      (int) Math.min(Integer.MAX_VALUE, TimeUnit.NANOSECONDS.toMillis(connConf.getRpcTimeoutNs()));
122    this.locator = new AsyncRegionLocator(this, RETRY_TIMER);
123    this.callerFactory = new AsyncRpcRetryingCallerFactory(this, RETRY_TIMER);
124    if (conf.getBoolean(CLIENT_NONCES_ENABLED_KEY, true)) {
125      nonceGenerator = PerClientRandomNonceGenerator.get();
126    } else {
127      nonceGenerator = NO_NONCE_GENERATOR;
128    }
129  }
130
131  private void spawnRenewalChore(final UserGroupInformation user) {
132    authService = new ChoreService("Relogin service");
133    authService.scheduleChore(AuthUtil.getAuthRenewalChore(user));
134  }

190  private void makeMasterStub(CompletableFuture<MasterService.Interface> future) {
191    addListener(registry.getMasterAddress(), (sn, error) -> {
192      if (sn == null) {
193        String msg = "ZooKeeper available but no active master location found";
194        LOG.info(msg);
195        this.masterStubMakeFuture.getAndSet(null)
196          .completeExceptionally(new MasterNotRunningException(msg));
197        return;
198      }
199      try {
200        MasterService.Interface stub = createMasterStub(sn);
201        HBaseRpcController controller = getRpcController();
202        stub.isMasterRunning(controller, RequestConverter.buildIsMasterRunningRequest(),
203          new RpcCallback<IsMasterRunningResponse>() {
204            @Override
205            public void run(IsMasterRunningResponse resp) {
206              if (controller.failed() || resp == null ||
207                (resp != null && !resp.getIsMasterRunning())) {
208                masterStubMakeFuture.getAndSet(null).completeExceptionally(
209                  new MasterNotRunningException("Master connection is not running anymore"));
210              } else {
211                masterStub.set(stub);
212                masterStubMakeFuture.set(null);
213                future.complete(stub);
214              }
215            }
216          });
217      } catch (IOException e) {
218        this.masterStubMakeFuture.getAndSet(null)
219          .completeExceptionally(new IOException("Failed to create async master stub", e));
220      }
221    });
222  }
223
224  CompletableFuture<MasterService.Interface> getMasterStub() {
225    MasterService.Interface masterStub = this.masterStub.get();
226
227    if (masterStub == null) {
228      for (;;) {
229        if (this.masterStubMakeFuture.compareAndSet(null, new CompletableFuture<>())) {
230          CompletableFuture<MasterService.Interface> future = this.masterStubMakeFuture.get();
231          makeMasterStub(future);
232        } else {
233          CompletableFuture<MasterService.Interface> future = this.masterStubMakeFuture.get();
234          if (future != null) {
235            return future;
236          }
237        }
238      }
239    }
240
241    for (;;) {
242      if (masterStubMakeFuture.compareAndSet(null, new CompletableFuture<>())) {
243        CompletableFuture<MasterService.Interface> future = masterStubMakeFuture.get();
244        HBaseRpcController controller = getRpcController();
245        masterStub.isMasterRunning(controller, RequestConverter.buildIsMasterRunningRequest(),
246          new RpcCallback<IsMasterRunningResponse>() {
247            @Override
248            public void run(IsMasterRunningResponse resp) {
249              if (controller.failed() || resp == null ||
250                (resp != null && !resp.getIsMasterRunning())) {
251                makeMasterStub(future);
252              } else {
253                future.complete(masterStub);
254              }
255            }
256          });
257      } else {
258        CompletableFuture<MasterService.Interface> future = masterStubMakeFuture.get();
259        if (future != null) {
260          return future;
261        }
262      }
263    }
264  }
265
266  private HBaseRpcController getRpcController() {
267    HBaseRpcController controller = this.rpcControllerFactory.newController();
268    controller.setCallTimeout((int) TimeUnit.NANOSECONDS.toMillis(connConf.getRpcTimeoutNs()));
269    return controller;
270  }

331  @Override
332  public CompletableFuture<Hbck> getHbck() {
333    CompletableFuture<Hbck> future = new CompletableFuture<>();
334    addListener(registry.getMasterAddress(), (sn, error) -> {
335      if (error != null) {
336        future.completeExceptionally(error);
337      } else {
338        try {
339          future.complete(getHbck(sn));
340        } catch (IOException e) {
341          future.completeExceptionally(e);
342        }
343      }
344    });
345    return future;
346  }

348  @Override
349  public Hbck getHbck(ServerName masterServer) throws IOException {
350    // we will not create a new connection when creating a new protobuf stub, and for hbck there
351    // will be no performance consideration, so for simplification we will create a new stub every
352    // time instead of caching the stub here.
353    return new HBaseHbck(MasterProtos.HbckService.newBlockingStub(
354      rpcClient.createBlockingRpcChannel(masterServer, user, rpcTimeout)), rpcControllerFactory);
355  }
