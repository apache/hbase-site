106  public AsyncConnectionImpl(Configuration conf, AsyncRegistry registry, String clusterId,
107      User user) {
108    this.conf = conf;
109    this.user = user;
110    if (user.isLoginFromKeytab()) {
111      spawnRenewalChore(user.getUGI());
112    }
113    this.connConf = new AsyncConnectionConfiguration(conf);
114    this.registry = registry;
115    this.rpcClient = RpcClientFactory.createClient(conf, clusterId);
116    this.rpcControllerFactory = RpcControllerFactory.instantiate(conf);
117    this.hostnameCanChange = conf.getBoolean(RESOLVE_HOSTNAME_ON_FAIL_KEY, true);
118    this.rpcTimeout =
119      (int) Math.min(Integer.MAX_VALUE, TimeUnit.NANOSECONDS.toMillis(connConf.getRpcTimeoutNs()));
120    this.locator = new AsyncRegionLocator(this, RETRY_TIMER);
121    this.callerFactory = new AsyncRpcRetryingCallerFactory(this, RETRY_TIMER);
122    if (conf.getBoolean(CLIENT_NONCES_ENABLED_KEY, true)) {
123      nonceGenerator = PerClientRandomNonceGenerator.get();
124    } else {
125      nonceGenerator = NO_NONCE_GENERATOR;
126    }
127  }
128
129  private void spawnRenewalChore(final UserGroupInformation user) {
130    authService = new ChoreService("Relogin service");
131    authService.scheduleChore(AuthUtil.getAuthRenewalChore(user));
132  }

139  @Override
140  public void close() {
141    // As the code below is safe to be executed in parallel, here we do not use CAS or lock, just a
142    // simple volatile flag.
143    if (closed) {
144      return;
145    }
146    IOUtils.closeQuietly(rpcClient);
147    IOUtils.closeQuietly(registry);
148    if (authService != null) {
149      authService.shutdown();
150    }
151    closed = true;
152  }
153
154  @Override
155  public boolean isClosed() {
156    return closed;
157  }

199  CompletableFuture<MasterService.Interface> getMasterStub() {
200    return ConnectionUtils.getOrFetch(masterStub, masterStubMakeFuture, false, () -> {
201      CompletableFuture<MasterService.Interface> future = new CompletableFuture<>();
202      addListener(registry.getMasterAddress(), (addr, error) -> {
203        if (error != null) {
204          future.completeExceptionally(error);
205        } else if (addr == null) {
206          future.completeExceptionally(new MasterNotRunningException(
207            "ZooKeeper available but no active master location found"));
208        } else {
209          LOG.debug("The fetched master address is {}", addr);
210          try {
211            future.complete(createMasterStub(addr));
212          } catch (IOException e) {
213            future.completeExceptionally(e);
214          }
215        }
216
217      });
218      return future;
219    }, stub -> true, "master stub");
220  }
221
222  void clearMasterStubCache(MasterService.Interface stub) {

285  @Override
286  public CompletableFuture<Hbck> getHbck() {
287    CompletableFuture<Hbck> future = new CompletableFuture<>();
288    addListener(registry.getMasterAddress(), (sn, error) -> {
289      if (error != null) {
290        future.completeExceptionally(error);
291      } else {
292        try {
293          future.complete(getHbck(sn));
294        } catch (IOException e) {
295          future.completeExceptionally(e);
296        }
297      }
298    });
299    return future;
300  }
301
302  @Override
303  public Hbck getHbck(ServerName masterServer) throws IOException {
304    // we will not create a new connection when creating a new protobuf stub, and for hbck there
305    // will be no performance consideration, so for simplification we will create a new stub every
306    // time instead of caching the stub here.
307    return new HBaseHbck(MasterProtos.HbckService.newBlockingStub(
308      rpcClient.createBlockingRpcChannel(masterServer, user, rpcTimeout)), rpcControllerFactory);
309  }
