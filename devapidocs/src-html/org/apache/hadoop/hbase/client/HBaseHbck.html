018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021import java.util.List;
022import java.util.Map;
023import java.util.concurrent.Callable;
024import java.util.stream.Collectors;
025
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.TableName;
028import org.apache.hadoop.hbase.ipc.RpcControllerFactory;
029import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
030import org.apache.hadoop.hbase.shaded.protobuf.RequestConverter;
031import org.apache.hadoop.hbase.shaded.protobuf.generated.HBaseProtos;
032import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProtos;
033import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProtos.GetTableStateResponse;
034import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProtos.HbckService.BlockingInterface;
035
036import org.apache.hbase.thirdparty.com.google.protobuf.ServiceException;
037
038import org.apache.yetus.audience.InterfaceAudience;
039
040import org.slf4j.Logger;
041import org.slf4j.LoggerFactory;
042
043
044/**
045 * Use {@link ClusterConnection#getHbck()} to obtain an instance of {@link Hbck} instead of
046 * constructing an HBaseHbck directly.
047 *
048 * <p>Connection should be an <i>unmanaged</i> connection obtained via
049 * {@link ConnectionFactory#createConnection(Configuration)}.</p>
050 *
051 * <p>NOTE: The methods in here can do damage to a cluster if applied in the wrong sequence or at
052 * the wrong time. Use with caution. For experts only. These methods are only for the
053 * extreme case where the cluster has been damaged or has achieved an inconsistent state because
054 * of some unforeseen circumstance or bug and requires manual intervention.
055 *
056 * <p>An instance of this class is lightweight and not-thread safe. A new instance should be created
057 * by each thread. Pooling or caching of the instance is not recommended.</p>
058 *
059 * @see ConnectionFactory
060 * @see ClusterConnection
061 * @see Hbck
062 */
063@InterfaceAudience.Private
064public class HBaseHbck implements Hbck {
065  private static final Logger LOG = LoggerFactory.getLogger(HBaseHbck.class);
066
067  private boolean aborted;
068  private final BlockingInterface hbck;
069
070  private RpcControllerFactory rpcControllerFactory;
071
072  HBaseHbck(BlockingInterface hbck, RpcControllerFactory rpcControllerFactory) {
073    this.hbck = hbck;
074    this.rpcControllerFactory = rpcControllerFactory;
075  }
076
077  @Override
078  public void close() throws IOException {
079    // currently does nothing
080  }
081
082  @Override
083  public void abort(String why, Throwable e) {
084    this.aborted = true;
085    // Currently does nothing but throw the passed message and exception
086    throw new RuntimeException(why, e);
087  }
088
089  @Override
090  public boolean isAborted() {
091    return this.aborted;
092  }
093
094  @Override
095  public TableState setTableStateInMeta(TableState state) throws IOException {
096    try {
097      GetTableStateResponse response = hbck.setTableStateInMeta(
098          rpcControllerFactory.newController(),
099          RequestConverter.buildSetTableStateInMetaRequest(state));
100      return TableState.convert(state.getTableName(), response.getTableState());
101    } catch (ServiceException se) {
102      LOG.debug("table={}, state={}", state.getTableName(), state.getState(), se);
103      throw new IOException(se);
104    }
105  }
106
107  @Override
108  public List<Long> assigns(List<String> encodedRegionNames, boolean override)
109      throws IOException {
110    try {
111      MasterProtos.AssignsResponse response =
112          this.hbck.assigns(rpcControllerFactory.newController(),
113              RequestConverter.toAssignRegionsRequest(encodedRegionNames, override));
114      return response.getPidList();
115    } catch (ServiceException se) {
116      LOG.debug(toCommaDelimitedString(encodedRegionNames), se);
117      throw new IOException(se);
118    }
119  }
120
121  @Override
122  public List<Long> unassigns(List<String> encodedRegionNames, boolean override)
123      throws IOException {
124    try {
125      MasterProtos.UnassignsResponse response =
126          this.hbck.unassigns(rpcControllerFactory.newController(),
127              RequestConverter.toUnassignRegionsRequest(encodedRegionNames, override));
128      return response.getPidList();
129    } catch (ServiceException se) {
130      LOG.debug(toCommaDelimitedString(encodedRegionNames), se);
131      throw new IOException(se);
132    }
133  }
134
135  private static String toCommaDelimitedString(List<String> list) {
136    return list.stream().collect(Collectors.joining(", "));
137  }
138
139  @Override
140  public List<Boolean> bypassProcedure(List<Long> pids, long waitTime, boolean override,
141      boolean recursive)
142      throws IOException {
143    MasterProtos.BypassProcedureResponse response = ProtobufUtil.call(
144        new Callable<MasterProtos.BypassProcedureResponse>() {
145          @Override
146          public MasterProtos.BypassProcedureResponse call() throws Exception {
147            try {
148              return hbck.bypassProcedure(rpcControllerFactory.newController(),
149                  MasterProtos.BypassProcedureRequest.newBuilder().addAllProcId(pids).
150                      setWaitTime(waitTime).setOverride(override).setRecursive(recursive).build());
151            } catch (Throwable t) {
152              LOG.error(pids.stream().map(i -> i.toString()).
153                  collect(Collectors.joining(", ")), t);
154              throw t;
155            }
156          }
157        });
158    return response.getBypassedList();
159  }
160
161  @Override
162  public List<Long> scheduleServerCrashProcedure(List<HBaseProtos.ServerName> serverNames)
163      throws IOException {
164    try {
165      MasterProtos.ScheduleServerCrashProcedureResponse response =
166          this.hbck.scheduleServerCrashProcedure(rpcControllerFactory.newController(),
167            RequestConverter.toScheduleServerCrashProcedureRequest(serverNames));
168      return response.getPidList();
169    } catch (ServiceException se) {
170      LOG.debug(toCommaDelimitedString(
171        serverNames.stream().map(serverName -> ProtobufUtil.toServerName(serverName).toString())
172            .collect(Collectors.toList())),
173        se);
174      throw new IOException(se);
175    }
176  }
177
178  @Override
179  public Map<String, MasterProtos.REGION_ERROR_TYPE>
180      getFailedSplitMergeLegacyRegions(List<TableName> tableNames) throws IOException {
181    try {
182      MasterProtos.GetFailedSplitMergeLegacyRegionsResponse response =
183          this.hbck.getFailedSplitMergeLegacyRegions(rpcControllerFactory.newController(),
184            RequestConverter.toGetFailedSplitMergeLegacyRegionsRequest(tableNames));
185      return response.getErrorsMap();
186    } catch (ServiceException se) {
187      LOG.debug("get failed split/merge legacy regions failed", se);
188      throw new IOException(se);
189    }
190  }
191}
