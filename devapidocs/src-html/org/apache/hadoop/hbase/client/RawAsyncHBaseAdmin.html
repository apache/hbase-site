350  private <PREQ, PRESP, RESP> CompletableFuture<RESP> call(HBaseRpcController controller,

362            future.complete(respConverter.convert(resp));

364            future.completeExceptionally(e);
365          }
366        }
367      }
368    });

637      Optional<byte[][]> splitKeys) {
638    CompletableFuture<Boolean> future = new CompletableFuture<>();
639    isTableEnabled(tableName).whenComplete(
640      (enabled, error) -> {
641        if (error != null) {
642          future.completeExceptionally(error);
643          return;
644        }
645        if (!enabled) {
646          future.complete(false);
647        } else {
648          AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName))
649              .whenComplete(
650                (locations, error1) -> {
651                  if (error1 != null) {
652                    future.completeExceptionally(error1);
653                    return;
654                  }
655                  List<HRegionLocation> notDeployedRegions =
656                      locations.stream().filter(loc -> loc.getServerName() == null)
657                          .collect(Collectors.toList());
658                  if (notDeployedRegions.size() > 0) {
659                    if (LOG.isDebugEnabled()) {
660                      LOG.debug("Table " + tableName + " has " + notDeployedRegions.size()
661                          + " regions");
662                    }
663                    future.complete(false);
664                    return;
665                  }
666
667                  Optional<Boolean> available =
668                      splitKeys.map(keys -> compareRegionsWithSplitKeys(locations, keys));
669                  future.complete(available.orElse(true));
670                });
671        }
672      });
673    return future;
674  }

676  private boolean compareRegionsWithSplitKeys(List<HRegionLocation> locations, byte[][] splitKeys) {
677    int regionCount = 0;
678    for (HRegionLocation location : locations) {
679      RegionInfo info = location.getRegion();
680      if (Bytes.equals(info.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
681        regionCount++;
682        continue;
683      }
684      for (byte[] splitKey : splitKeys) {
685        // Just check if the splitkey is available
686        if (Bytes.equals(info.getStartKey(), splitKey)) {
687          regionCount++;
688          break;
689        }
690      }
691    }
692    return regionCount == splitKeys.length + 1;
693  }

1020      return AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName));

1024  /**
1025   * Compact column family of a table, Asynchronous operation even if CompletableFuture.get()
1026   */
1027  private CompletableFuture<Void> compact(TableName tableName, byte[] columnFamily,
1028      boolean major, CompactType compactType) {
1029    CompletableFuture<Void> future = new CompletableFuture<>();
1030
1031    switch (compactType) {
1032      case MOB:
1033        connection.registry.getMasterAddress().whenComplete((serverName, err) -> {
1034          if (err != null) {
1035            future.completeExceptionally(err);
1036            return;
1037          }
1038          RegionInfo regionInfo = RegionInfo.createMobRegionInfo(tableName);
1039          compact(serverName, regionInfo, major, columnFamily)
1040              .whenComplete((ret, err2) -> {
1041                if (err2 != null) {
1042                  future.completeExceptionally(err2);
1043                } else {
1044                  future.complete(ret);
1045                }
1046              });
1047        });
1048        break;
1049      case NORMAL:
1050        getTableHRegionLocations(tableName).whenComplete((locations, err) -> {
1051          if (err != null) {
1052            future.completeExceptionally(err);
1053            return;
1054          }
1055          CompletableFuture<?>[] compactFutures = locations.stream().filter(l -> l.getRegion() != null)
1056              .filter(l -> !l.getRegion().isOffline()).filter(l -> l.getServerName() != null)
1057              .map(l -> compact(l.getServerName(), l.getRegion(), major, columnFamily))
1058              .toArray(CompletableFuture<?>[]::new);
1059          // future complete unless all of the compact futures are completed.
1060          CompletableFuture.allOf(compactFutures).whenComplete((ret, err2) -> {
1061            if (err2 != null) {
1062              future.completeExceptionally(err2);
1063            } else {
1064              future.complete(ret);
1065            }
1066          });
1067        });
1068        break;
1069      default:
1070        throw new IllegalArgumentException("Unknown compactType: " + compactType);
1071    }
1072    return future;
1073  }

1220  public CompletableFuture<Void> split(TableName tableName) {
1221    CompletableFuture<Void> future = new CompletableFuture<>();
1222    tableExists(tableName).whenComplete((exist, error) -> {
1223      if (error != null) {
1224        future.completeExceptionally(error);
1225        return;
1226      }
1227      if (!exist) {
1228        future.completeExceptionally(new TableNotFoundException(tableName));
1229        return;
1230      }
1231      metaTable
1232          .scanAll(new Scan().setReadType(ReadType.PREAD).addFamily(HConstants.CATALOG_FAMILY)
1233              .withStartRow(MetaTableAccessor.getTableStartRowForMeta(tableName, QueryType.REGION))
1234              .withStopRow(MetaTableAccessor.getTableStopRowForMeta(tableName, QueryType.REGION)))
1235          .whenComplete((results, err2) -> {
1236            if (err2 != null) {
1237              future.completeExceptionally(err2);
1238              return;
1239            }
1240            if (results != null && !results.isEmpty()) {
1241              List<CompletableFuture<Void>> splitFutures = new ArrayList<>();
1242              for (Result r : results) {
1243                if (r.isEmpty() || MetaTableAccessor.getRegionInfo(r) == null) continue;
1244                RegionLocations rl = MetaTableAccessor.getRegionLocations(r);
1245                if (rl != null) {
1246                  for (HRegionLocation h : rl.getRegionLocations()) {
1247                    if (h != null && h.getServerName() != null) {
1248                      RegionInfo hri = h.getRegion();
1249                      if (hri == null || hri.isSplitParent()
1250                          || hri.getReplicaId() != RegionInfo.DEFAULT_REPLICA_ID)
1251                        continue;
1252                      splitFutures.add(split(hri, null));
1253                    }
1254                  }
1255                }
1256              }
1257              CompletableFuture
1258                  .allOf(splitFutures.toArray(new CompletableFuture<?>[splitFutures.size()]))
1259                  .whenComplete((ret, exception) -> {
1260                    if (exception != null) {
1261                      future.completeExceptionally(exception);
1262                      return;
1263                    }
1264                    future.complete(ret);
1265                  });
1266            } else {
1267              future.complete(null);
1268            }
1269          });
1270    });
1271    return future;
1272  }

1812    CompletableFuture<Void> future = new CompletableFuture<>();
1813    listSnapshots(Pattern.compile(snapshotName)).whenComplete(
1814      (snapshotDescriptions, err) -> {
1815        if (err != null) {
1816          future.completeExceptionally(err);
1817          return;
1818        }

1860  private CompletableFuture<Void> restoreSnapshot(String snapshotName, TableName tableName,
1861      boolean takeFailSafeSnapshot) {
1862    if (takeFailSafeSnapshot) {
1863      CompletableFuture<Void> future = new CompletableFuture<>();
1864      // Step.1 Take a snapshot of the current state
1865      String failSafeSnapshotSnapshotNameFormat = this.connection.getConfiguration().get(
1866        HConstants.SNAPSHOT_RESTORE_FAILSAFE_NAME,
1867        HConstants.DEFAULT_SNAPSHOT_RESTORE_FAILSAFE_NAME);
1868      final String failSafeSnapshotSnapshotName = failSafeSnapshotSnapshotNameFormat
1869          .replace("{snapshot.name}", snapshotName)
1870          .replace("{table.name}", tableName.toString().replace(TableName.NAMESPACE_DELIM, '.'))
1871          .replace("{restore.timestamp}", String.valueOf(EnvironmentEdgeManager.currentTime()));
1872      LOG.info("Taking restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
1873      snapshot(failSafeSnapshotSnapshotName, tableName).whenComplete((ret, err) -> {
1874        if (err != null) {
1875          future.completeExceptionally(err);
1876        } else {
1877          // Step.2 Restore snapshot
1878        internalRestoreSnapshot(snapshotName, tableName).whenComplete((void2, err2) -> {
1879          if (err2 != null) {
1880            // Step.3.a Something went wrong during the restore and try to rollback.
1881          internalRestoreSnapshot(failSafeSnapshotSnapshotName, tableName).whenComplete(
1882            (void3, err3) -> {
1883              if (err3 != null) {
1884                future.completeExceptionally(err3);
1885              } else {
1886                String msg = "Restore snapshot=" + snapshotName + " failed. Rollback to snapshot="
1887                    + failSafeSnapshotSnapshotName + " succeeded.";
1888                future.completeExceptionally(new RestoreSnapshotException(msg));
1889              }
1890            });
1891        } else {
1892          // Step.3.b If the restore is succeeded, delete the pre-restore snapshot.
1893          LOG.info("Deleting restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
1894          deleteSnapshot(failSafeSnapshotSnapshotName).whenComplete(
1895            (ret3, err3) -> {
1896              if (err3 != null) {
1897                LOG.error(
1898                  "Unable to remove the failsafe snapshot: " + failSafeSnapshotSnapshotName, err3);
1899                future.completeExceptionally(err3);
1900              } else {
1901                future.complete(ret3);
1902              }
1903            });
1904        }
1905      } );
1906      }
1907    } );
1908      return future;
1909    } else {
1910      return internalRestoreSnapshot(snapshotName, tableName);
1911    }
1912  }
