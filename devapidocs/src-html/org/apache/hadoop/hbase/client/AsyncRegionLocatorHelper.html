018package org.apache.hadoop.hbase.client;
019
020import static org.apache.hadoop.hbase.exceptions.ClientExceptionsUtil.findException;
021import static org.apache.hadoop.hbase.exceptions.ClientExceptionsUtil.isMetaClearingException;
022
023import java.util.Arrays;
024import java.util.Optional;
025import java.util.function.Consumer;
026import java.util.function.Function;
027import org.apache.commons.lang3.ObjectUtils;
028import org.apache.hadoop.hbase.HRegionLocation;
029import org.apache.hadoop.hbase.RegionLocations;
030import org.apache.hadoop.hbase.exceptions.RegionMovedException;
031import org.apache.yetus.audience.InterfaceAudience;
032import org.slf4j.Logger;
033import org.slf4j.LoggerFactory;
034
035/**
036 * Helper class for asynchronous region locator.
037 */
038@InterfaceAudience.Private
039final class AsyncRegionLocatorHelper {
040
041  private static final Logger LOG = LoggerFactory.getLogger(AsyncRegionLocatorHelper.class);
042
043  private AsyncRegionLocatorHelper() {
044  }
045
046  static boolean canUpdateOnError(HRegionLocation loc, HRegionLocation oldLoc) {
047    // Do not need to update if no such location, or the location is newer, or the location is not
048    // the same with us
049    return oldLoc != null && oldLoc.getSeqNum() <= loc.getSeqNum() &&
050      oldLoc.getServerName().equals(loc.getServerName());
051  }
052
053  static void updateCachedLocationOnError(HRegionLocation loc, Throwable exception,
054      Function<HRegionLocation, HRegionLocation> cachedLocationSupplier,
055      Consumer<HRegionLocation> addToCache, Consumer<HRegionLocation> removeFromCache,
056      Optional<MetricsConnection> metrics) {
057    HRegionLocation oldLoc = cachedLocationSupplier.apply(loc);
058    if (LOG.isDebugEnabled()) {
059      LOG.debug("Try updating {} , the old value is {}, error={}", loc, oldLoc,
060        exception != null ? exception.toString() : "none");
061    }
062    if (!canUpdateOnError(loc, oldLoc)) {
063      return;
064    }
065    Throwable cause = findException(exception);
066    if (LOG.isDebugEnabled()) {
067      LOG.debug("The actual exception when updating {} is {}", loc,
068        cause != null ? cause.toString() : "none");
069    }
070    if (cause == null || !isMetaClearingException(cause)) {
071      LOG.debug("Will not update {} because the exception is null or not the one we care about",
072        loc);
073      return;
074    }
075    if (cause instanceof RegionMovedException) {
076      RegionMovedException rme = (RegionMovedException) cause;
077      HRegionLocation newLoc =
078        new HRegionLocation(loc.getRegion(), rme.getServerName(), rme.getLocationSeqNum());
079      LOG.debug("Try updating {} with the new location {} constructed by {}", loc, newLoc,
080        rme.toString());
081      addToCache.accept(newLoc);
082    } else {
083      LOG.debug("Try removing {} from cache", loc);
084      metrics.ifPresent(m -> m.incrCacheDroppingExceptions(exception));
085      removeFromCache.accept(loc);
086    }
087  }
088
089  static RegionLocations createRegionLocations(HRegionLocation loc) {
090    int replicaId = loc.getRegion().getReplicaId();
091    HRegionLocation[] locs = new HRegionLocation[replicaId + 1];
092    locs[replicaId] = loc;
093    return new RegionLocations(locs);
094  }
095
096  /**
097   * Create a new {@link RegionLocations} based on the given {@code oldLocs}, and replace the
098   * location for the given {@code replicaId} with the given {@code loc}.
099   * <p/>
100   * All the {@link RegionLocations} in async locator related class are immutable because we want to
101   * access them concurrently, so here we need to create a new one, instead of calling
102   * {@link RegionLocations#updateLocation(HRegionLocation, boolean, boolean)}.
103   */
104  static RegionLocations replaceRegionLocation(RegionLocations oldLocs, HRegionLocation loc) {
105    int replicaId = loc.getRegion().getReplicaId();
106    HRegionLocation[] locs = oldLocs.getRegionLocations();
107    locs = Arrays.copyOf(locs, Math.max(replicaId + 1, locs.length));
108    locs[replicaId] = loc;
109    return new RegionLocations(locs);
110  }
111
112  /**
113   * Create a new {@link RegionLocations} based on the given {@code oldLocs}, and remove the
114   * location for the given {@code replicaId}.
115   * <p/>
116   * All the {@link RegionLocations} in async locator related class are immutable because we want to
117   * access them concurrently, so here we need to create a new one, instead of calling
118   * {@link RegionLocations#remove(int)}.
119   */
120  static RegionLocations removeRegionLocation(RegionLocations oldLocs, int replicaId) {
121    HRegionLocation[] locs = oldLocs.getRegionLocations();
122    if (locs.length < replicaId + 1) {
123      // Here we do not modify the oldLocs so it is safe to return it.
124      return oldLocs;
125    }
126    locs = Arrays.copyOf(locs, locs.length);
127    locs[replicaId] = null;
128    if (ObjectUtils.firstNonNull(locs) != null) {
129      return new RegionLocations(locs);
130    } else {
131      // if all the locations are null, just return null
132      return null;
133    }
134  }
135
136  static boolean isGood(RegionLocations locs, int replicaId) {
137    if (locs == null) {
138      return false;
139    }
140    HRegionLocation loc = locs.getRegionLocation(replicaId);
141    return loc != null && loc.getServerName() != null;
142  }
143}
