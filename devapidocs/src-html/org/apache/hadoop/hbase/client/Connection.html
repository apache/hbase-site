019package org.apache.hadoop.hbase.client;
020
021import java.io.Closeable;
022import java.io.IOException;
023import java.util.concurrent.ExecutorService;
024
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.Abortable;
027import org.apache.hadoop.hbase.TableName;
028import org.apache.hadoop.hbase.classification.InterfaceAudience;
029import org.apache.hadoop.hbase.classification.InterfaceStability;
030
031/**
032 * A cluster connection encapsulating lower level individual connections to actual servers and
033 * a connection to zookeeper. Connections are instantiated through the {@link ConnectionFactory}
034 * class. The lifecycle of the connection is managed by the caller, who has to {@link #close()}
035 * the connection to release the resources.
036 *
037 * <p> The connection object contains logic to find the master, locate regions out on the cluster,
038 * keeps a cache of locations and then knows how to re-calibrate after they move. The individual
039 * connections to servers, meta cache, zookeeper connection, etc are all shared by the
040 * {@link Table} and {@link Admin} instances obtained from this connection.
041 *
042 * <p> Connection creation is a heavy-weight operation. Connection implementations are thread-safe,
043 * so that the client can create a connection once, and share it with different threads.
044 * {@link Table} and {@link Admin} instances, on the other hand, are light-weight and are not
045 * thread-safe.  Typically, a single connection per client application is instantiated and every
046 * thread will obtain its own Table instance. Caching or pooling of {@link Table} and {@link Admin}
047 * is not recommended.
048 *
049 * @see ConnectionFactory
050 * @since 0.99.0
051 */
052@InterfaceAudience.Public
053@InterfaceStability.Evolving
054public interface Connection extends Abortable, Closeable {
055
056  /*
057   * Implementation notes:
058   *  - Only allow new style of interfaces:
059   *   -- All table names are passed as TableName. No more byte[] and string arguments
060   *   -- Most of the classes with names H is deprecated in favor of non-H versions
061   *   (Table, Connection, etc)
062   *   -- Only real client-facing public methods are allowed
063   *  - Connection should contain only getTable(), getAdmin() kind of general methods.
064   */
065
066  /**
067   * @return Configuration instance being used by this Connection instance.
068   */
069  Configuration getConfiguration();
070
071  /**
072   * Retrieve a Table implementation for accessing a table.
073   * The returned Table is not thread safe, a new instance should be created for each using thread.
074   * This is a lightweight operation, pooling or caching of the returned Table
075   * is neither required nor desired.
076   * <p>
077   * The caller is responsible for calling {@link Table#close()} on the returned
078   * table instance.
079   * <p>
080   * Since 0.98.1 this method no longer checks table existence. An exception
081   * will be thrown if the table does not exist only when the first operation is
082   * attempted.
083   * @param tableName the name of the table
084   * @return a Table to use for interactions with this table
085   */
086  default Table getTable(TableName tableName) throws IOException {
087    return getTable(tableName, null);
088  }
089
090  /**
091   * Retrieve a Table implementation for accessing a table.
092   * The returned Table is not thread safe, a new instance should be created for each using thread.
093   * This is a lightweight operation, pooling or caching of the returned Table
094   * is neither required nor desired.
095   * <p>
096   * The caller is responsible for calling {@link Table#close()} on the returned
097   * table instance.
098   * <p>
099   * Since 0.98.1 this method no longer checks table existence. An exception
100   * will be thrown if the table does not exist only when the first operation is
101   * attempted.
102   *
103   * @param tableName the name of the table
104   * @param pool The thread pool to use for batch operations, null to use a default pool.
105   * @return a Table to use for interactions with this table
106   */
107  default Table getTable(TableName tableName, ExecutorService pool) throws IOException {
108    return getTableBuilder(tableName, pool).build();
109  }
110
111  /**
112   * <p>
113   * Retrieve a {@link BufferedMutator} for performing client-side buffering of writes. The
114   * {@link BufferedMutator} returned by this method is thread-safe. This BufferedMutator will
115   * use the Connection's ExecutorService. This object can be used for long lived operations.
116   * </p>
117   * <p>
118   * The caller is responsible for calling {@link BufferedMutator#close()} on
119   * the returned {@link BufferedMutator} instance.
120   * </p>
121   * <p>
122   * This accessor will use the connection's ExecutorService and will throw an
123   * exception in the main thread when an asynchronous exception occurs.
124   *
125   * @param tableName the name of the table
126   *
127   * @return a {@link BufferedMutator} for the supplied tableName.
128   */
129  BufferedMutator getBufferedMutator(TableName tableName) throws IOException;
130
131  /**
132   * Retrieve a {@link BufferedMutator} for performing client-side buffering of writes. The
133   * {@link BufferedMutator} returned by this method is thread-safe. This object can be used for
134   * long lived table operations. The caller is responsible for calling
135   * {@link BufferedMutator#close()} on the returned {@link BufferedMutator} instance.
136   *
137   * @param params details on how to instantiate the {@code BufferedMutator}.
138   * @return a {@link BufferedMutator} for the supplied tableName.
139   */
140  BufferedMutator getBufferedMutator(BufferedMutatorParams params) throws IOException;
141
142  /**
143   * Retrieve a RegionLocator implementation to inspect region information on a table. The returned
144   * RegionLocator is not thread-safe, so a new instance should be created for each using thread.
145   *
146   * This is a lightweight operation.  Pooling or caching of the returned RegionLocator is neither
147   * required nor desired.
148   * <br>
149   * The caller is responsible for calling {@link RegionLocator#close()} on the returned
150   * RegionLocator instance.
151   *
152   * RegionLocator needs to be unmanaged
153   *
154   * @param tableName Name of the table who's region is to be examined
155   * @return A RegionLocator instance
156   */
157  RegionLocator getRegionLocator(TableName tableName) throws IOException;
158
159  /**
160   * Retrieve an Admin implementation to administer an HBase cluster.
161   * The returned Admin is not guaranteed to be thread-safe.  A new instance should be created for
162   * each using thread.  This is a lightweight operation.  Pooling or caching of the returned
163   * Admin is not recommended.
164   * <br>
165   * The caller is responsible for calling {@link Admin#close()} on the returned
166   * Admin instance.
167   *
168   * @return an Admin instance for cluster administration
169   */
170  Admin getAdmin() throws IOException;
171
172  @Override
173  void close() throws IOException;
174
175  /**
176   * Returns whether the connection is closed or not.
177   * @return true if this connection is closed
178   */
179  boolean isClosed();
180
181  /**
182   * Returns an {@link TableBuilder} for creating {@link Table}.
183   * @param tableName the name of the table
184   * @param pool the thread pool to use for requests like batch and scan
185   */
186  TableBuilder getTableBuilder(TableName tableName, ExecutorService pool);
187}
