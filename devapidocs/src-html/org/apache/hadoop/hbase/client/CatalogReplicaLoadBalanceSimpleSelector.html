042/**
043 * <p>CatalogReplicaLoadBalanceReplicaSimpleSelector implements a simple catalog replica load
044 * balancing algorithm. It maintains a stale location cache for each table. Whenever client looks
045 * up location, it first check if the row is the stale location cache. If yes, the location from
046 * catalog replica is stale, it will go to the primary region to look up update-to-date location;
047 * otherwise, it will randomly pick up a replica region or primary region for lookup. When clients
048 * receive RegionNotServedException from region servers, it will add these region locations to the
049 * stale location cache. The stale cache will be cleaned up periodically by a chore.</p>
050 *
051 * It follows a simple algorithm to choose a meta replica region (including primary meta) to go:
052 *
053 * <ol>
054 *  <li>If there is no stale location entry for rows it looks up, it will randomly
055 *     pick a meta replica region (including primary meta) to do lookup. </li>
056 *  <li>If the location from the replica region is stale, client gets RegionNotServedException
057 *     from region server, in this case, it will create StaleLocationCacheEntry in
058 *     CatalogReplicaLoadBalanceReplicaSimpleSelector.</li>
059 *  <li>When client tries to do location lookup, it checks StaleLocationCache first for rows it
060 *     tries to lookup, if entry exists, it will go with primary meta region to do lookup;
061 *     otherwise, it will follow step 1.</li>
062 *  <li>A chore will periodically run to clean up cache entries in the StaleLocationCache.</li>
063 * </ol>
064 */
065class CatalogReplicaLoadBalanceSimpleSelector implements
066  CatalogReplicaLoadBalanceSelector, Stoppable {
067  private static final Logger LOG =
068    LoggerFactory.getLogger(CatalogReplicaLoadBalanceSimpleSelector.class);
069  private final long STALE_CACHE_TIMEOUT_IN_MILLISECONDS = 3000; // 3 seconds
070  private final int STALE_CACHE_CLEAN_CHORE_INTERVAL_IN_MILLISECONDS = 1500; // 1.5 seconds
071  private final int REFRESH_REPLICA_COUNT_CHORE_INTERVAL_IN_MILLISECONDS = 60000; // 1 minute
072
073  /**
074   * StaleLocationCacheEntry is the entry when a stale location is reported by an client.
075   */
076  private static final class StaleLocationCacheEntry {
077    // timestamp in milliseconds
078    private final long timestamp;
079
080    private final byte[] endKey;
081
082    StaleLocationCacheEntry(final byte[] endKey) {
083      this.endKey = endKey;
084      timestamp = EnvironmentEdgeManager.currentTime();
085    }
086
087    public byte[] getEndKey() {
088      return this.endKey;
089    }
090
091    public long getTimestamp() {
092      return this.timestamp;
093    }
094
095    @Override
096    public String toString() {
097      return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE)
098        .append("endKey", endKey)
099        .append("timestamp", timestamp)
100        .toString();
101    }
102  }
103
104  private final ConcurrentMap<TableName, ConcurrentNavigableMap<byte[], StaleLocationCacheEntry>>
105    staleCache = new ConcurrentHashMap<>();
106  private volatile int numOfReplicas;
107  private final AsyncConnectionImpl conn;
108  private final TableName tableName;
109  private final IntSupplier getNumOfReplicas;
110  private volatile boolean isStopped = false;
111
112  CatalogReplicaLoadBalanceSimpleSelector(TableName tableName, AsyncConnectionImpl conn,
113    IntSupplier getNumOfReplicas) {
114    this.conn = conn;
115    this.tableName = tableName;
116    this.getNumOfReplicas = getNumOfReplicas;
117
118    // This numOfReplicas is going to be lazy initialized.
119    this.numOfReplicas = CatalogReplicaLoadBalanceSelector.UNINITIALIZED_NUM_OF_REPLICAS;
120    // Start chores
121    this.conn.getChoreService().scheduleChore(getCacheCleanupChore(this));
122    this.conn.getChoreService().scheduleChore(getRefreshReplicaCountChore(this));
123  }
124
125  /**
126   * When a client runs into RegionNotServingException, it will call this method to
127   * update Selector's internal state.
128   * @param loc the location which causes exception.
129   */
130  public void onError(HRegionLocation loc) {
131    ConcurrentNavigableMap<byte[], StaleLocationCacheEntry> tableCache =
132      computeIfAbsent(staleCache, loc.getRegion().getTable(),
133        () -> new ConcurrentSkipListMap<>(BYTES_COMPARATOR));
134    byte[] startKey = loc.getRegion().getStartKey();
135    tableCache.putIfAbsent(startKey,
136      new StaleLocationCacheEntry(loc.getRegion().getEndKey()));
137    LOG.debug("Add entry to stale cache for table {} with startKey {}, {}",
138      loc.getRegion().getTable(), startKey, loc.getRegion().getEndKey());
139  }
140
141  /**
142   * Select an random replica id (including the primary replica id). In case there is no replica region configured, return
143   * the primary replica id.
144   * @return Replica id
145   */
146  private int getRandomReplicaId() {
147    int cachedNumOfReplicas = this.numOfReplicas;
148    if (cachedNumOfReplicas == CatalogReplicaLoadBalanceSelector.UNINITIALIZED_NUM_OF_REPLICAS) {
149      cachedNumOfReplicas = refreshCatalogReplicaCount();
150      this.numOfReplicas = cachedNumOfReplicas;
151    }
152    // In case of no replica configured, return the primary region id.
153    if (cachedNumOfReplicas <= 1) {
154      return RegionInfo.DEFAULT_REPLICA_ID;
155    }
156    return ThreadLocalRandom.current().nextInt(cachedNumOfReplicas);
157  }
158
159  /**
160   * When it looks up a location, it will call this method to find a replica region to go.
161   * For a normal case, > 99% of region locations from catalog/meta replica will be up to date.
162   * In extreme cases such as region server crashes, it will depends on how fast replication
163   * catches up.
164   *
165   * @param tablename table name it looks up
166   * @param row key it looks up.
167   * @param locateType locateType, Only BEFORE and CURRENT will be passed in.
168   * @return catalog replica id
169   */
170  public int select(final TableName tablename, final byte[] row,
171    final RegionLocateType locateType) {
172    Preconditions.checkArgument(locateType == RegionLocateType.BEFORE ||
173        locateType == RegionLocateType.CURRENT,
174      "Expected type BEFORE or CURRENT but got: %s", locateType);
175
176    ConcurrentNavigableMap<byte[], StaleLocationCacheEntry> tableCache = staleCache.get(tablename);
177
178    // If there is no entry in StaleCache, select a random replica id.
179    if (tableCache == null) {
180      return getRandomReplicaId();
181    }
182
183    Map.Entry<byte[], StaleLocationCacheEntry> entry;
184    boolean isEmptyStopRow = isEmptyStopRow(row);
185    // Only BEFORE and CURRENT are passed in.
186    if (locateType == RegionLocateType.BEFORE) {
187      entry = isEmptyStopRow ? tableCache.lastEntry() : tableCache.lowerEntry(row);
188    } else {
189      entry = tableCache.floorEntry(row);
190    }
191
192    // It is not in the stale cache, return a random replica id.
193    if (entry == null) {
194      return getRandomReplicaId();
195    }
196
197    // The entry here is a possible match for the location. Check if the entry times out first as
198    // long comparing is faster than comparing byte arrays(in most cases). It could remove
199    // stale entries faster. If the possible match entry does not time out, it will check if
200    // the entry is a match for the row passed in and select the replica id accordingly.
201    if ((EnvironmentEdgeManager.currentTime() - entry.getValue().getTimestamp()) >=
202      STALE_CACHE_TIMEOUT_IN_MILLISECONDS) {
203      LOG.debug("Entry for table {} with startKey {}, {} times out", tablename, entry.getKey(),
204        entry);
205      tableCache.remove(entry.getKey());
206      return getRandomReplicaId();
207    }
208
209    byte[] endKey =  entry.getValue().getEndKey();
210
211    // The following logic is borrowed from AsyncNonMetaRegionLocator.
212    if (isEmptyStopRow(endKey)) {
213      LOG.debug("Lookup {} goes to primary region", row);
214      return RegionInfo.DEFAULT_REPLICA_ID;
215    }
216
217    if (locateType == RegionLocateType.BEFORE) {
218      if (!isEmptyStopRow && Bytes.compareTo(endKey, row) >= 0) {
219        LOG.debug("Lookup {} goes to primary meta", row);
220        return RegionInfo.DEFAULT_REPLICA_ID;
221      }
222    } else {
223      if (Bytes.compareTo(row, endKey) < 0) {
224        LOG.debug("Lookup {} goes to primary meta", row);
225        return RegionInfo.DEFAULT_REPLICA_ID;
226      }
227    }
228
229    // Not in stale cache, return a random replica id.
230    return getRandomReplicaId();
231  }

233  // This class implements the Stoppable interface as chores needs a Stopable object, there is
234  // no-op on this Stoppable object currently.
235  @Override
236  public void stop(String why) {
237    isStopped = true;
238  }
239
240  @Override
241  public boolean isStopped() {
242    return isStopped;
243  }
244
245  private void cleanupReplicaReplicaStaleCache() {
246    long curTimeInMills = EnvironmentEdgeManager.currentTime();
247    for (ConcurrentNavigableMap<byte[], StaleLocationCacheEntry> tableCache : staleCache.values()) {
248      Iterator<Map.Entry<byte[], StaleLocationCacheEntry>> it =
249        tableCache.entrySet().iterator();
250      while (it.hasNext()) {
251        Map.Entry<byte[], StaleLocationCacheEntry> entry = it.next();
252        if (curTimeInMills - entry.getValue().getTimestamp() >=
253          STALE_CACHE_TIMEOUT_IN_MILLISECONDS) {
254          LOG.debug("clean entry {}, {} from stale cache", entry.getKey(), entry.getValue());
255          it.remove();
256        }
257      }
258    }
259  }
260
261  private int refreshCatalogReplicaCount() {
262    int newNumOfReplicas = this.getNumOfReplicas.getAsInt();
263    LOG.debug("Refreshed replica count {}", newNumOfReplicas);
264    // If the returned number of replicas is -1, it is caused by failure to fetch the
265    // replica count. Do not update the numOfReplicas in this case.
266    if (newNumOfReplicas == CatalogReplicaLoadBalanceSelector.UNINITIALIZED_NUM_OF_REPLICAS) {
267      LOG.error("Failed to fetch Table {}'s region replica count", tableName);
268      return this.numOfReplicas;
269    }
270
271    int cachedNumOfReplicas = this.numOfReplicas;
272    if ((cachedNumOfReplicas == UNINITIALIZED_NUM_OF_REPLICAS) ||
273      (cachedNumOfReplicas != newNumOfReplicas)) {
274      this.numOfReplicas = newNumOfReplicas;
275    }
276    return newNumOfReplicas;
277  }
278
279  private ScheduledChore getCacheCleanupChore(
280    final CatalogReplicaLoadBalanceSimpleSelector selector) {
281    return new ScheduledChore("CleanupCatalogReplicaStaleCache", this,
282      STALE_CACHE_CLEAN_CHORE_INTERVAL_IN_MILLISECONDS) {
283      @Override
284      protected void chore() {
285        selector.cleanupReplicaReplicaStaleCache();
286      }
287    };
288  }
289
290  private ScheduledChore getRefreshReplicaCountChore(
291    final CatalogReplicaLoadBalanceSimpleSelector selector) {
292    return new ScheduledChore("RefreshReplicaCountChore", this,
293      REFRESH_REPLICA_COUNT_CHORE_INTERVAL_IN_MILLISECONDS) {
294      @Override
295      protected void chore() {
296        selector.refreshCatalogReplicaCount();
297      }
298    };
299  }
