019package org.apache.hadoop.hbase.client;
020
021import java.util.Collection;
022import java.util.Iterator;
023
024import org.apache.hadoop.hbase.util.Bytes;
025import org.apache.yetus.audience.InterfaceAudience;
026
027/**
028 * Utility methods which contain the logic for regions and replicas.
029 */
030@InterfaceAudience.Private
031public class RegionReplicaUtil {
032
033  /**
034   * Whether or not the secondary region will wait for observing a flush / region open event
035   * from the primary region via async wal replication before enabling read requests. Since replayed
036   * edits from async wal replication from primary is not persisted in WAL, the memstore of the
037   * secondary region might be non-empty at the time of close or crash. For ensuring seqId's not
038   * "going back in time" in the secondary region replica, this should be enabled. However, in some
039   * cases the above semantics might be ok for some application classes.
040   * See HBASE-11580 for more context.
041   */
042  public static final String REGION_REPLICA_WAIT_FOR_PRIMARY_FLUSH_CONF_KEY
043    = "hbase.region.replica.wait.for.primary.flush";
044  protected static final boolean DEFAULT_REGION_REPLICA_WAIT_FOR_PRIMARY_FLUSH = true;
045
046  /**
047   * The default replicaId for the region
048   */
049  static final int DEFAULT_REPLICA_ID = 0;
050
051  /**
052   * Returns the RegionInfo for the given replicaId.
053   * RegionInfo's correspond to a range of a table, but more than one
054   * "instance" of the same range can be deployed which are differentiated by
055   * the replicaId.
056   * @param regionInfo
057   * @param replicaId the replicaId to use
058   * @return an RegionInfo object corresponding to the same range (table, start and
059   * end key), but for the given replicaId.
060   */
061  public static RegionInfo getRegionInfoForReplica(RegionInfo regionInfo, int replicaId) {
062    if (regionInfo.getReplicaId() == replicaId) {
063      return regionInfo;
064    }
065
066    if (regionInfo.isMetaRegion()) {
067      return RegionInfoBuilder.newBuilder(regionInfo.getTable())
068          .setRegionId(regionInfo.getRegionId())
069          .setReplicaId(replicaId)
070          .setOffline(regionInfo.isOffline())
071          .build();
072    } else {
073      return RegionInfoBuilder.newBuilder(regionInfo.getTable())
074              .setStartKey(regionInfo.getStartKey())
075              .setEndKey(regionInfo.getEndKey())
076              .setSplit(regionInfo.isSplit())
077              .setRegionId(regionInfo.getRegionId())
078              .setReplicaId(replicaId)
079              .setOffline(regionInfo.isOffline())
080              .build();
081    }
082  }
083
084  /**
085   * Returns the RegionInfo for the default replicaId (0). RegionInfo's correspond to
086   * a range of a table, but more than one "instance" of the same range can be
087   * deployed which are differentiated by the replicaId.
088   * @return an RegionInfo object corresponding to the same range (table, start and
089   * end key), but for the default replicaId.
090   */
091  public static RegionInfo getRegionInfoForDefaultReplica(RegionInfo regionInfo) {
092    return getRegionInfoForReplica(regionInfo, DEFAULT_REPLICA_ID);
093  }
094
095  /** @return true if this replicaId corresponds to default replica for the region */
096  public static boolean isDefaultReplica(int replicaId) {
097    return DEFAULT_REPLICA_ID == replicaId;
098  }
099
100  /** @return true if this region is a default replica for the region */
101  public static boolean isDefaultReplica(RegionInfo hri) {
102    return  hri.getReplicaId() == DEFAULT_REPLICA_ID;
103  }
104
105  /**
106   * Removes the non-default replicas from the passed regions collection
107   * @param regions
108   */
109  public static void removeNonDefaultRegions(Collection<RegionInfo> regions) {
110    Iterator<RegionInfo> iterator = regions.iterator();
111    while (iterator.hasNext()) {
112      RegionInfo hri = iterator.next();
113      if (!RegionReplicaUtil.isDefaultReplica(hri)) {
114        iterator.remove();
115      }
116    }
117  }
118
119  public static boolean isReplicasForSameRegion(RegionInfo regionInfoA, RegionInfo regionInfoB) {
120    return compareRegionInfosWithoutReplicaId(regionInfoA, regionInfoB) == 0;
121  }
122
123  private static int compareRegionInfosWithoutReplicaId(RegionInfo regionInfoA,
124      RegionInfo regionInfoB) {
125    int result = regionInfoA.getTable().compareTo(regionInfoB.getTable());
126    if (result != 0) {
127      return result;
128    }
129
130    // Compare start keys.
131    result = Bytes.compareTo(regionInfoA.getStartKey(), regionInfoB.getStartKey());
132    if (result != 0) {
133      return result;
134    }
135
136    // Compare end keys.
137    result = Bytes.compareTo(regionInfoA.getEndKey(), regionInfoB.getEndKey());
138
139    if (result != 0) {
140      if (regionInfoA.getStartKey().length != 0
141              && regionInfoA.getEndKey().length == 0) {
142          return 1; // this is last region
143      }
144      if (regionInfoB.getStartKey().length != 0
145              && regionInfoB.getEndKey().length == 0) {
146          return -1; // o is the last region
147      }
148      return result;
149    }
150
151    // regionId is usually milli timestamp -- this defines older stamps
152    // to be "smaller" than newer stamps in sort order.
153    if (regionInfoA.getRegionId() > regionInfoB.getRegionId()) {
154      return 1;
155    } else if (regionInfoA.getRegionId() < regionInfoB.getRegionId()) {
156      return -1;
157    }
158    return 0;
159  }
160}
