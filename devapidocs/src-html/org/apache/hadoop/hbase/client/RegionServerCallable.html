019package org.apache.hadoop.hbase.client;
020
021import java.io.IOException;
022
023import org.apache.hadoop.hbase.CellScanner;
024import org.apache.hadoop.hbase.HRegionInfo;
025import org.apache.hadoop.hbase.HRegionLocation;
026import org.apache.hadoop.hbase.ServerName;
027import org.apache.hadoop.hbase.TableName;
028import org.apache.hadoop.hbase.TableNotEnabledException;
029import org.apache.hadoop.hbase.classification.InterfaceAudience;
030import org.apache.hadoop.hbase.ipc.HBaseRpcController;
031import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
032import org.apache.hadoop.hbase.shaded.com.google.protobuf.RpcController;
033import org.apache.hadoop.hbase.util.Bytes;
034
035/**
036 * Implementations make a RPC call against a RegionService via a protobuf Service.
037 * Implement rpcCall() and the parent class setClientByServiceName; this latter is where the
038 * RPC stub gets set (the appropriate protobuf 'Service'/Client). Be sure to make use of the
039 * RpcController that this instance is carrying via #getRpcController().
040 *
041 * <p>TODO: this class is actually tied to one region, because most of the paths make use of
042 *       the regioninfo part of location when building requests. The only reason it works for
043 *       multi-region requests (e.g. batch) is that they happen to not use the region parts.
044 *       This could be done cleaner (e.g. having a generic parameter and 2 derived classes,
045 *       RegionCallable and actual RegionServerCallable with ServerName.
046 *
047 * @param <T> The class that the ServerCallable handles.
048 * @param <S> The protocol to use (Admin or Client or even an Endpoint over in MetaTableAccessor).
049 */
050// TODO: MasterCallable and this Class have a lot in common. UNIFY!
051// Public but should be package private only it is used by MetaTableAccessor. FIX!!
052@InterfaceAudience.Private
053public abstract class RegionServerCallable<T, S> implements RetryingCallable<T> {
054  private final Connection connection;
055  private final TableName tableName;
056  private final byte[] row;
057  /**
058   * Some subclasses want to set their own location. Make it protected.
059   */
060  protected HRegionLocation location;
061  protected S stub;
062
063  /**
064   * This is 99% of the time a HBaseRpcController but also used doing Coprocessor Endpoints and in
065   * this case, it is a ServerRpcControllable which is not a HBaseRpcController.
066   * Can be null!
067   */
068  protected final RpcController rpcController;
069
070  /**
071   * @param connection Connection to use.
072   * @param rpcController Controller to use; can be shaded or non-shaded.
073   * @param tableName Table name to which <code>row</code> belongs.
074   * @param row The row we want in <code>tableName</code>.
075   */
076  public RegionServerCallable(Connection connection, TableName tableName, byte [] row,
077      RpcController rpcController) {
078    super();
079    this.connection = connection;
080    this.tableName = tableName;
081    this.row = row;
082    this.rpcController = rpcController;
083  }
084
085  protected RpcController getRpcController() {
086    return this.rpcController;
087  }
088
089  protected void setStub(S stub) {
090    this.stub = stub;
091  }
092
093  protected S getStub() {
094    return this.stub;
095  }
096
097  /**
098   * Override that changes call Exception from {@link Exception} to {@link IOException}.
099   * Also does set up of the rpcController.
100   */
101  public T call(int callTimeout) throws IOException {
102    try {
103      // Iff non-null and an instance of a SHADED rpcController, do config! Unshaded -- i.e.
104      // com.google.protobuf.RpcController or null -- will just skip over this config.
105      if (getRpcController() != null) {
106        RpcController shadedRpcController = (RpcController)getRpcController();
107        // Do a reset to clear previous states, such as CellScanner.
108        shadedRpcController.reset();
109        if (shadedRpcController instanceof HBaseRpcController) {
110          HBaseRpcController hrc = (HBaseRpcController)getRpcController();
111          // If it is an instance of HBaseRpcController, we can set priority on the controller based
112          // off the tableName. Set call timeout too.
113          hrc.setPriority(tableName);
114          hrc.setCallTimeout(callTimeout);
115        }
116      }
117      return rpcCall();
118    } catch (Exception e) {
119      throw ProtobufUtil.handleRemoteException(e);
120    }
121  }
122
123  /**
124   * Run the RPC call. Implement this method. To get at the rpcController that has been created
125   * and configured to make this rpc call, use getRpcController(). We are trying to contain
126   * rpcController references so we don't pollute codebase with protobuf references; keep the
127   * protobuf references contained and only present in a few classes rather than all about the
128   * code base.
129   * @throws Exception
130   */
131  protected abstract T rpcCall() throws Exception;
132
133  /**
134   * Get the RpcController CellScanner.
135   * If the RpcController is a HBaseRpcController, which it is in all cases except
136   * when we are processing Coprocessor Endpoint, then this method returns a reference to the
137   * CellScanner that the HBaseRpcController is carrying. Do it up here in this Callable
138   * so we don't have to scatter ugly instanceof tests around the codebase. Will return null
139   * if called in a Coproccessor Endpoint context. Should never happen.
140   */
141  protected CellScanner getRpcControllerCellScanner() {
142    return (getRpcController() != null && getRpcController() instanceof HBaseRpcController)?
143        ((HBaseRpcController)getRpcController()).cellScanner(): null;
144  }
145
146  protected void setRpcControllerCellScanner(CellScanner cellScanner) {
147    if (getRpcController() != null && getRpcController() instanceof HBaseRpcController) {
148      ((HBaseRpcController)this.rpcController).setCellScanner(cellScanner);
149    }
150  }
151
152  /**
153   * @return {@link ClusterConnection} instance used by this Callable.
154   */
155  protected ClusterConnection getConnection() {
156    return (ClusterConnection) this.connection;
157  }
158
159  protected HRegionLocation getLocation() {
160    return this.location;
161  }
162
163  protected void setLocation(final HRegionLocation location) {
164    this.location = location;
165  }
166
167  public TableName getTableName() {
168    return this.tableName;
169  }
170
171  public byte [] getRow() {
172    return this.row;
173  }
174
175  public void throwable(Throwable t, boolean retrying) {
176    if (location != null) {
177      getConnection().updateCachedLocations(tableName, location.getRegionInfo().getRegionName(),
178          row, t, location.getServerName());
179    }
180  }
181
182  public String getExceptionMessageAdditionalDetail() {
183    return "row '" + Bytes.toString(row) + "' on table '" + tableName + "' at " + location;
184  }
185
186  public long sleep(long pause, int tries) {
187    return ConnectionUtils.getPauseTime(pause, tries);
188  }
189
190  /**
191   * @return the HRegionInfo for the current region
192   */
193  public HRegionInfo getHRegionInfo() {
194    if (this.location == null) {
195      return null;
196    }
197    return this.location.getRegionInfo();
198  }
199
200  public void prepare(final boolean reload) throws IOException {
201    // check table state if this is a retry
202    if (reload && tableName != null && !tableName.equals(TableName.META_TABLE_NAME)
203        && getConnection().isTableDisabled(tableName)) {
204      throw new TableNotEnabledException(tableName.getNameAsString() + " is disabled.");
205    }
206    try (RegionLocator regionLocator = connection.getRegionLocator(tableName)) {
207      this.location = regionLocator.getRegionLocation(row);
208    }
209    if (this.location == null) {
210      throw new IOException("Failed to find location, tableName=" + tableName +
211          ", row=" + Bytes.toString(row) + ", reload=" + reload);
212    }
213    setStubByServiceName(this.location.getServerName());
214  }
215
216  /**
217   * Set the RCP client stub
218   * @param serviceName to get the rpc stub for
219   * @throws IOException When client could not be created
220   */
221  protected abstract void setStubByServiceName(ServerName serviceName) throws IOException;
222}
