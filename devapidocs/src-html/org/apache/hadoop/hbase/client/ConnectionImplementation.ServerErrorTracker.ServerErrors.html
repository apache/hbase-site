221  ConnectionImplementation(Configuration conf,
222                           ExecutorService pool, User user) throws IOException {
223    this.conf = conf;
224    this.user = user;
225    this.batchPool = pool;
226    this.connectionConfig = new ConnectionConfiguration(conf);
227    this.closed = false;
228    this.pause = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
229        HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
230    long configuredPauseForCQTBE = conf.getLong(HConstants.HBASE_CLIENT_PAUSE_FOR_CQTBE, pause);
231    if (configuredPauseForCQTBE < pause) {
232      LOG.warn("The " + HConstants.HBASE_CLIENT_PAUSE_FOR_CQTBE + " setting: "
233          + configuredPauseForCQTBE + " is smaller than " + HConstants.HBASE_CLIENT_PAUSE
234          + ", will use " + pause + " instead.");
235      this.pauseForCQTBE = pause;
236    } else {
237      this.pauseForCQTBE = configuredPauseForCQTBE;
238    }
239    this.useMetaReplicas = conf.getBoolean(HConstants.USE_META_REPLICAS,
240      HConstants.DEFAULT_USE_META_REPLICAS);
241    this.metaReplicaCallTimeoutScanInMicroSecond =
242        connectionConfig.getMetaReplicaCallTimeoutMicroSecondScan();
243
244    // how many times to try, one more than max *retry* time
245    this.numTries = retries2Attempts(connectionConfig.getRetriesNumber());
246    this.rpcTimeout = conf.getInt(
247        HConstants.HBASE_RPC_TIMEOUT_KEY,
248        HConstants.DEFAULT_HBASE_RPC_TIMEOUT);
249    if (conf.getBoolean(NonceGenerator.CLIENT_NONCES_ENABLED_KEY, true)) {
250      synchronized (nonceGeneratorCreateLock) {
251        if (nonceGenerator == null) {
252          nonceGenerator = PerClientRandomNonceGenerator.get();
253        }
254      }
255    } else {
256      nonceGenerator = NO_NONCE_GENERATOR;
257    }
258
259    this.stats = ServerStatisticTracker.create(conf);
260    this.interceptor = (new RetryingCallerInterceptorFactory(conf)).build();
261    this.rpcControllerFactory = RpcControllerFactory.instantiate(conf);
262    this.rpcCallerFactory = RpcRetryingCallerFactory.instantiate(conf, interceptor, this.stats);
263    this.backoffPolicy = ClientBackoffPolicyFactory.create(conf);
264    this.asyncProcess = new AsyncProcess(this, conf, rpcCallerFactory, false, rpcControllerFactory);
265    if (conf.getBoolean(CLIENT_SIDE_METRICS_ENABLED_KEY, false)) {
266      this.metrics = new MetricsConnection(this);
267    } else {
268      this.metrics = null;
269    }
270    this.metaCache = new MetaCache(this.metrics);
271
272    boolean shouldListen = conf.getBoolean(HConstants.STATUS_PUBLISHED,
273        HConstants.STATUS_PUBLISHED_DEFAULT);
274    this.hostnamesCanChange = conf.getBoolean(RESOLVE_HOSTNAME_ON_FAIL_KEY, true);
275    Class<? extends ClusterStatusListener.Listener> listenerClass =
276        conf.getClass(ClusterStatusListener.STATUS_LISTENER_CLASS,
277            ClusterStatusListener.DEFAULT_STATUS_LISTENER_CLASS,
278            ClusterStatusListener.Listener.class);
279
280    // Is there an alternate BufferedMutator to use?
281    this.alternateBufferedMutatorClassName =
282        this.conf.get(BufferedMutator.CLASSNAME_KEY);
283
284    try {
285      this.registry = AsyncRegistryFactory.getRegistry(conf);
286      retrieveClusterId();
287
288      this.rpcClient = RpcClientFactory.createClient(this.conf, this.clusterId, this.metrics);
289
290      // Do we publish the status?
291      if (shouldListen) {
292        if (listenerClass == null) {
293          LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
294              ClusterStatusListener.STATUS_LISTENER_CLASS + " is not set - not listening status");
295        } else {
296          clusterStatusListener = new ClusterStatusListener(
297              new ClusterStatusListener.DeadServerHandler() {
298                @Override
299                public void newDead(ServerName sn) {
300                  clearCaches(sn);
301                  rpcClient.cancelConnections(sn);
302                }
303              }, conf, listenerClass);
304        }
305      }
306    } catch (Throwable e) {
307      // avoid leaks: registry, rpcClient, ...
308      LOG.debug("connection construction failed", e);
309      close();
310      throw e;
311    }
312  }

359    }

361      params.pool(HTable.getDefaultExecutor(getConfiguration()));

364      params.writeBufferSize(connectionConfig.getWriteBufferSize());
365    }
366    if (params.getMaxKeyValueSize() == BufferedMutatorParams.UNSET) {
367      params.maxKeyValueSize(connectionConfig.getMaxKeyValueSize());
368    }
369    // Look to see if an alternate BufferedMutation implementation is wanted.
370    // Look in params and in config. If null, use default.
371    String implementationClassName = params.getImplementationClassName();
372    if (implementationClassName == null) {
373      implementationClassName = this.alternateBufferedMutatorClassName;
374    }
375    if (implementationClassName == null) {
376      return new BufferedMutatorImpl(this, rpcCallerFactory, rpcControllerFactory, params);
377    }
378    try {
379      return (BufferedMutator)ReflectionUtils.newInstance(Class.forName(implementationClassName),
380          this, rpcCallerFactory, rpcControllerFactory, params);
381    } catch (ClassNotFoundException e) {
382      throw new RuntimeException(e);
383    }
384  }

406  private ExecutorService getBatchPool() {
407    if (batchPool == null) {
408      synchronized (this) {
409        if (batchPool == null) {
410          int threads = conf.getInt("hbase.hconnection.threads.max", 256);
411          this.batchPool = getThreadPool(threads, threads, "-shared", null);
412          this.cleanupPool = true;
413        }
414      }
415    }
416    return this.batchPool;
417  }
418
419  private ExecutorService getThreadPool(int maxThreads, int coreThreads, String nameHint,
420      BlockingQueue<Runnable> passedWorkQueue) {
421    // shared HTable thread executor not yet initialized
422    if (maxThreads == 0) {
423      maxThreads = Runtime.getRuntime().availableProcessors() * 8;
424    }
425    if (coreThreads == 0) {
426      coreThreads = Runtime.getRuntime().availableProcessors() * 8;
427    }
428    long keepAliveTime = conf.getLong("hbase.hconnection.threads.keepalivetime", 60);
429    BlockingQueue<Runnable> workQueue = passedWorkQueue;
430    if (workQueue == null) {
431      workQueue =
432        new LinkedBlockingQueue<>(maxThreads *
433            conf.getInt(HConstants.HBASE_CLIENT_MAX_TOTAL_TASKS,
434                HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS));
435      coreThreads = maxThreads;
436    }
437    ThreadPoolExecutor tpe = new ThreadPoolExecutor(
438        coreThreads,
439        maxThreads,
440        keepAliveTime,
441        TimeUnit.SECONDS,
442        workQueue,
443        Threads.newDaemonThreadFactory(toString() + nameHint));
444    tpe.allowCoreThreadTimeOut(true);
445    return tpe;
446  }

448  private ExecutorService getMetaLookupPool() {
449    if (this.metaLookupPool == null) {
450      synchronized (this) {
451        if (this.metaLookupPool == null) {
452          //Some of the threads would be used for meta replicas
453          //To start with, threads.max.core threads can hit the meta (including replicas).
454          //After that, requests will get queued up in the passed queue, and only after
455          //the queue is full, a new thread will be started
456          int threads = conf.getInt("hbase.hconnection.meta.lookup.threads.max", 128);
457          this.metaLookupPool = getThreadPool(
458             threads,
459             threads,
460             "-metaLookup-shared-", new LinkedBlockingQueue<>());
461        }
462      }
463    }
464    return this.metaLookupPool;
465  }

475  private void shutdownPools() {
476    if (this.cleanupPool && this.batchPool != null && !this.batchPool.isShutdown()) {
477      shutdownBatchPool(this.batchPool);
478    }
479    if (this.metaLookupPool != null && !this.metaLookupPool.isShutdown()) {
480      shutdownBatchPool(this.metaLookupPool);
481    }
482  }

485    pool.shutdown();
486    try {

488        pool.shutdownNow();
489      }
490    } catch (InterruptedException e) {

568  @Override
569  public boolean isTableAvailable(final TableName tableName, @Nullable final byte[][] splitKeys)
570      throws IOException {
571    if (this.closed) {
572      throw new IOException(toString() + " closed");
573    }
574    try {
575      if (!isTableEnabled(tableName)) {
576        LOG.debug("Table " + tableName + " not enabled");
577        return false;
578      }
579      List<Pair<RegionInfo, ServerName>> locations =
580        MetaTableAccessor.getTableRegionsAndLocations(this, tableName, true);
581
582      int notDeployed = 0;
583      int regionCount = 0;
584      for (Pair<RegionInfo, ServerName> pair : locations) {
585        RegionInfo info = pair.getFirst();
586        if (pair.getSecond() == null) {
587          if (LOG.isDebugEnabled()) {
588            LOG.debug("Table " + tableName + " has not deployed region " + pair.getFirst()
589                .getEncodedName());
590          }
591          notDeployed++;
592        } else if (splitKeys != null
593            && !Bytes.equals(info.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
594          for (byte[] splitKey : splitKeys) {
595            // Just check if the splitkey is available
596            if (Bytes.equals(info.getStartKey(), splitKey)) {
597              regionCount++;
598              break;
599            }
600          }
601        } else {
602          // Always empty start row should be counted
603          regionCount++;
604        }
605      }
606      if (notDeployed > 0) {
607        if (LOG.isDebugEnabled()) {
608          LOG.debug("Table " + tableName + " has " + notDeployed + " regions");
609        }
610        return false;
611      } else if (splitKeys != null && regionCount != splitKeys.length + 1) {
612        if (LOG.isDebugEnabled()) {
613          LOG.debug("Table " + tableName + " expected to have " + (splitKeys.length + 1)
614              + " regions, but only " + regionCount + " available");
615        }
616        return false;
617      } else {
618        if (LOG.isDebugEnabled()) {
619          LOG.debug("Table " + tableName + " should be available");
620        }
621        return true;
622      }
623    } catch (TableNotFoundException tnfe) {
624      LOG.warn("Table " + tableName + " not enabled, it is not exists");
625      return false;
626    }
627  }

728  private RegionLocations locateMeta(final TableName tableName,
729      boolean useCache, int replicaId) throws IOException {
730    // HBASE-10785: We cache the location of the META itself, so that we are not overloading
731    // zookeeper with one request for every region lookup. We cache the META with empty row
732    // key in MetaCache.
733    byte[] metaCacheKey = HConstants.EMPTY_START_ROW; // use byte[0] as the row for meta
734    RegionLocations locations = null;
735    if (useCache) {
736      locations = getCachedLocation(tableName, metaCacheKey);
737      if (locations != null && locations.getRegionLocation(replicaId) != null) {
738        return locations;
739      }
740    }
741
742    // only one thread should do the lookup.
743    synchronized (metaRegionLock) {
744      // Check the cache again for a hit in case some other thread made the
745      // same query while we were waiting on the lock.
746      if (useCache) {
747        locations = getCachedLocation(tableName, metaCacheKey);
748        if (locations != null && locations.getRegionLocation(replicaId) != null) {
749          return locations;
750        }
751      }
752
753      // Look up from zookeeper
754      locations = get(this.registry.getMetaRegionLocation());
755      if (locations != null) {
756        cacheLocation(tableName, locations);
757      }
758    }
759    return locations;
760  }

762  /*
763    * Search the hbase:meta table for the HRegionLocation
764    * info that contains the table and row we're seeking.
765    */
766  private RegionLocations locateRegionInMeta(TableName tableName, byte[] row,
767                 boolean useCache, boolean retry, int replicaId) throws IOException {
768
769    // If we are supposed to be using the cache, look in the cache to see if
770    // we already have the region.
771    if (useCache) {
772      RegionLocations locations = getCachedLocation(tableName, row);
773      if (locations != null && locations.getRegionLocation(replicaId) != null) {
774        return locations;
775      }
776    }
777
778    // build the key of the meta region we should be looking for.
779    // the extra 9's on the end are necessary to allow "exact" matches
780    // without knowing the precise region names.
781    byte[] metaKey = RegionInfo.createRegionName(tableName, row, HConstants.NINES, false);
782
783    Scan s = new Scan();
784    s.setReversed(true);
785    s.withStartRow(metaKey);
786    s.addFamily(HConstants.CATALOG_FAMILY);
787
788    if (this.useMetaReplicas) {
789      s.setConsistency(Consistency.TIMELINE);
790    }
791
792    int maxAttempts = (retry ? numTries : 1);
793    for (int tries = 0; true; tries++) {
794      if (tries >= maxAttempts) {
795        throw new NoServerForRegionException("Unable to find region for "
796            + Bytes.toStringBinary(row) + " in " + tableName + " after " + tries + " tries.");
797      }
798      if (useCache) {
799        RegionLocations locations = getCachedLocation(tableName, row);
800        if (locations != null && locations.getRegionLocation(replicaId) != null) {
801          return locations;
802        }
803      } else {
804        // If we are not supposed to be using the cache, delete any existing cached location
805        // so it won't interfere.
806        // We are only supposed to clean the cache for the specific replicaId
807        metaCache.clearCache(tableName, row, replicaId);
808      }
809
810      // Query the meta region
811      long pauseBase = this.pause;
812      userRegionLock.lock();
813      try {
814        if (useCache) {// re-check cache after get lock
815          RegionLocations locations = getCachedLocation(tableName, row);
816          if (locations != null && locations.getRegionLocation(replicaId) != null) {
817            return locations;
818          }
819        }
820        Result regionInfoRow = null;
821        s.resetMvccReadPoint();
822        s.setOneRowLimit();
823        try (ReversedClientScanner rcs =
824            new ReversedClientScanner(conf, s, TableName.META_TABLE_NAME, this, rpcCallerFactory,
825                rpcControllerFactory, getMetaLookupPool(), metaReplicaCallTimeoutScanInMicroSecond)) {
826          regionInfoRow = rcs.next();
827        }
828
829        if (regionInfoRow == null) {
830          throw new TableNotFoundException(tableName);
831        }
832        // convert the row result into the HRegionLocation we need!
833        RegionLocations locations = MetaTableAccessor.getRegionLocations(regionInfoRow);
834        if (locations == null || locations.getRegionLocation(replicaId) == null) {
835          throw new IOException("HRegionInfo was null in " +
836            tableName + ", row=" + regionInfoRow);
837        }
838        RegionInfo regionInfo = locations.getRegionLocation(replicaId).getRegion();
839        if (regionInfo == null) {
840          throw new IOException("HRegionInfo was null or empty in " +
841            TableName.META_TABLE_NAME + ", row=" + regionInfoRow);
842        }
843
844        // possible we got a region of a different table...
845        if (!regionInfo.getTable().equals(tableName)) {
846          throw new TableNotFoundException(
847            "Region of '" + regionInfo.getRegionNameAsString() + "' is expected in the table of '" + tableName + "', " +
848            "but hbase:meta says it is in the table of '" + regionInfo.getTable() + "'. " +
849            "hbase:meta might be damaged.");
850        }
851        if (regionInfo.isSplit()) {
852          throw new RegionOfflineException(
853              "the only available region for the required row is a split parent,"
854                  + " the daughters should be online soon: " + regionInfo.getRegionNameAsString());
855        }
856        if (regionInfo.isOffline()) {
857          throw new RegionOfflineException("the region is offline, could"
858              + " be caused by a disable table call: " + regionInfo.getRegionNameAsString());
859        }
860
861        ServerName serverName = locations.getRegionLocation(replicaId).getServerName();
862        if (serverName == null) {
863          throw new NoServerForRegionException("No server address listed in "
864              + TableName.META_TABLE_NAME + " for region " + regionInfo.getRegionNameAsString()
865              + " containing row " + Bytes.toStringBinary(row));
866        }
867
868        if (isDeadServer(serverName)){
869          throw new RegionServerStoppedException("hbase:meta says the region "+
870              regionInfo.getRegionNameAsString()+" is managed by the server " + serverName +
871              ", but it is dead.");
872        }
873        // Instantiate the location
874        cacheLocation(tableName, locations);
875        return locations;
876      } catch (TableNotFoundException e) {
877        // if we got this error, probably means the table just plain doesn't
878        // exist. rethrow the error immediately. this should always be coming
879        // from the HTable constructor.
880        throw e;
881      } catch (IOException e) {
882        ExceptionUtil.rethrowIfInterrupt(e);
883        if (e instanceof RemoteException) {
884          e = ((RemoteException)e).unwrapRemoteException();
885        }
886        if (e instanceof CallQueueTooBigException) {
887          // Give a special check on CallQueueTooBigException, see #HBASE-17114
888          pauseBase = this.pauseForCQTBE;
889        }
890        if (tries < maxAttempts - 1) {
891          if (LOG.isDebugEnabled()) {
892            LOG.debug("locateRegionInMeta parentTable=" + TableName.META_TABLE_NAME
893                + ", metaLocation=" + ", attempt=" + tries + " of " + maxAttempts
894                + " failed; retrying after sleep of "
895                + ConnectionUtils.getPauseTime(pauseBase, tries) + " because: " + e.getMessage());
896          }
897        } else {
898          throw e;
899        }
900        // Only relocate the parent region if necessary
901        if(!(e instanceof RegionOfflineException ||
902            e instanceof NoServerForRegionException)) {
903          relocateRegion(TableName.META_TABLE_NAME, metaKey, replicaId);
904        }
905      } finally {
906        userRegionLock.unlock();
907      }
908      try{
909        Thread.sleep(ConnectionUtils.getPauseTime(pauseBase, tries));
910      } catch (InterruptedException e) {
911        throw new InterruptedIOException("Giving up trying to location region in " +
912          "meta: thread is interrupted.");
913      }
914    }
915  }

1022     * Constructor
1023     * @param timeout how long to wait before timeout, in unit of millisecond
1024     * @param maxTries how many times to try
1025     */
1026    public ServerErrorTracker(long timeout, int maxTries) {
1027      this.maxTries = maxTries;
1028      this.canRetryUntil = EnvironmentEdgeManager.currentTime() + timeout;
1029      this.startTrackingTime = new Date().getTime();

1032    /**
1033     * We stop to retry when we have exhausted BOTH the number of tries and the time allocated.
1034     * @param numAttempt how many times we have tried by now
1035     */
1036    boolean canTryMore(int numAttempt) {
1037      // If there is a single try we must not take into account the time.
1038      return numAttempt < maxTries || (maxTries > 1 &&
1039          EnvironmentEdgeManager.currentTime() < this.canRetryUntil);
1040    }

1044     *
1045     * @param server    The server in question.
1046     * @param basePause The default hci pause.
1047     * @return The time to wait before sending next request.
1048     */
1049    long calculateBackoffTime(ServerName server, long basePause) {
1050      long result;
1051      ServerErrors errorStats = errorsByServer.get(server);
1052      if (errorStats != null) {
1053        result = ConnectionUtils.getPauseTime(basePause, Math.max(0, errorStats.getCount() - 1));
1054      } else {
1055        result = 0; // yes, if the server is not in our list we don't wait before retrying.
1056      }
1057      return result;
1058    }

1727  private boolean isKeepAliveMasterConnectedAndRunning(MasterServiceState mss) {
1728    if (mss.getStub() == null){
1729      return false;
1730    }
1731    try {
1732      return mss.isMasterRunning();
1733    } catch (UndeclaredThrowableException e) {
1734      // It's somehow messy, but we can receive exceptions such as
1735      //  java.net.ConnectException but they're not declared. So we catch it...
1736      LOG.info("Master connection is not running anymore", e.getUndeclaredThrowable());
1737      return false;
1738    } catch (IOException se) {
1739      LOG.warn("Checking master connection", se);
1740      return false;
1741    }
1742  }

1781  /**
1782   * Update the location with the new value (if the exception is a RegionMovedException)
1783   * or delete it from the cache. Does nothing if we can be sure from the exception that
1784   * the location is still accurate, or if the cache has already been updated.
1785   * @param exception an object (to simplify user code) on which we will try to find a nested
1786   *   or wrapped or both RegionMovedException
1787   * @param source server that is the source of the location update.
1788   */
1789  @Override
1790  public void updateCachedLocations(final TableName tableName, byte[] regionName, byte[] rowkey,
1791    final Object exception, final ServerName source) {
1792    if (rowkey == null || tableName == null) {
1793      LOG.warn("Coding error, see method javadoc. row=" + (rowkey == null ? "null" : rowkey) +
1794          ", tableName=" + (tableName == null ? "null" : tableName));
1795      return;
1796    }
1797
1798    if (source == null) {
1799      // This should not happen, but let's secure ourselves.
1800      return;
1801    }
1802
1803    if (regionName == null) {
1804      // we do not know which region, so just remove the cache entry for the row and server
1805      if (metrics != null) {
1806        metrics.incrCacheDroppingExceptions(exception);
1807      }
1808      metaCache.clearCache(tableName, rowkey, source);
1809      return;
1810    }
1811
1812    // Is it something we have already updated?
1813    final RegionLocations oldLocations = getCachedLocation(tableName, rowkey);
1814    HRegionLocation oldLocation = null;
1815    if (oldLocations != null) {
1816      oldLocation = oldLocations.getRegionLocationByRegionName(regionName);
1817    }
1818    if (oldLocation == null || !source.equals(oldLocation.getServerName())) {
1819      // There is no such location in the cache (it's been removed already) or
1820      // the cache has already been refreshed with a different location.  => nothing to do
1821      return;
1822    }
1823
1824    RegionInfo regionInfo = oldLocation.getRegion();
1825    Throwable cause = ClientExceptionsUtil.findException(exception);
1826    if (cause != null) {
1827      if (!ClientExceptionsUtil.isMetaClearingException(cause)) {
1828        // We know that the region is still on this region server
1829        return;
1830      }
1831
1832      if (cause instanceof RegionMovedException) {
1833        RegionMovedException rme = (RegionMovedException) cause;
1834        if (LOG.isTraceEnabled()) {
1835          LOG.trace("Region " + regionInfo.getRegionNameAsString() + " moved to " +
1836              rme.getHostname() + ":" + rme.getPort() +
1837              " according to " + source.getAddress());
1838        }
1839        // We know that the region is not anymore on this region server, but we know
1840        //  the new location.
1841        updateCachedLocation(
1842            regionInfo, source, rme.getServerName(), rme.getLocationSeqNum());
1843        return;
1844      }
1845    }
1846
1847    if (metrics != null) {
1848      metrics.incrCacheDroppingExceptions(exception);
1849    }
1850
1851    // If we're here, it means that can cannot be sure about the location, so we remove it from
1852    // the cache. Do not send the source because source can be a new server in the same host:port
1853    metaCache.clearCache(regionInfo);
1854  }

1985  private static <T> T get(CompletableFuture<T> future) throws IOException {
1986    try {
1987      return future.get();
1988    } catch (InterruptedException e) {
1989      Thread.currentThread().interrupt();
1990      throw (IOException) new InterruptedIOException().initCause(e);
1991    } catch (ExecutionException e) {
1992      Throwable cause = e.getCause();
1993      Throwables.propagateIfPossible(cause, IOException.class);
1994      throw new IOException(cause);
1995    }
