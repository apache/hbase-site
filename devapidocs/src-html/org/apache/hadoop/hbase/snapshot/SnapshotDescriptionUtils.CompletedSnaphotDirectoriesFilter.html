051/**
052 * Utility class to help manage {@link SnapshotDescription SnapshotDesriptions}.
053 * <p>
054 * Snapshots are laid out on disk like this:
055 *
056 * <pre>
057 * /hbase/.snapshots
058 *          /.tmp                &lt;---- working directory
059 *          /[snapshot name]     &lt;----- completed snapshot
060 * </pre>
061 *
062 * A completed snapshot named 'completed' then looks like (multiple regions, servers, files, etc.
063 * signified by '...' on the same directory depth).
064 *
065 * <pre>
066 * /hbase/.snapshots/completed
067 *                   .snapshotinfo          &lt;--- Description of the snapshot
068 *                   .tableinfo             &lt;--- Copy of the tableinfo
069 *                    /.logs
070 *                        /[server_name]
071 *                            /... [log files]
072 *                         ...
073 *                   /[region name]           &lt;---- All the region's information
074 *                   .regioninfo              &lt;---- Copy of the HRegionInfo
075 *                      /[column family name]
076 *                          /[hfile name]     &lt;--- name of the hfile in the real region
077 *                          ...
078 *                      ...
079 *                    ...
080 * </pre>
081 *
082 * Utility methods in this class are useful for getting the correct locations for different parts of
083 * the snapshot, as well as moving completed snapshots into place (see
084 * {@link #completeSnapshot}, and writing the
085 * {@link SnapshotDescription} to the working snapshot directory.
086 */
087@InterfaceAudience.Private
088public final class SnapshotDescriptionUtils {

141  /**
142   * @param conf {@link Configuration} from which to check for the timeout
143   * @param type type of snapshot being taken
144   * @param defaultMaxWaitTime Default amount of time to wait, if none is in the configuration
145   * @return the max amount of time the master should wait for a snapshot to complete
146   */
147  public static long getMaxMasterTimeout(Configuration conf, SnapshotDescription.Type type,
148      long defaultMaxWaitTime) {
149    String confKey;
150    switch (type) {
151    case DISABLED:
152    default:
153      confKey = MASTER_SNAPSHOT_TIMEOUT_MILLIS;
154    }
155    return Math.max(conf.getLong(confKey, defaultMaxWaitTime),
156        conf.getLong(MASTER_SNAPSHOT_TIMEOUT_MILLIS, defaultMaxWaitTime));
157  }

280  /**
281   * Convert the passed snapshot description into a 'full' snapshot description based on default
282   * parameters, if none have been supplied. This resolves any 'optional' parameters that aren't
283   * supplied to their default values.
284   * @param snapshot general snapshot descriptor
285   * @param conf Configuration to read configured snapshot defaults if snapshot is not complete
286   * @return a valid snapshot description
287   * @throws IllegalArgumentException if the {@link SnapshotDescription} is not a complete
288   *           {@link SnapshotDescription}.
289   */
290  public static SnapshotDescription validate(SnapshotDescription snapshot, Configuration conf)
291      throws IllegalArgumentException, IOException {
292    if (!snapshot.hasTable()) {
293      throw new IllegalArgumentException(
294          "Descriptor doesn't apply to a table, so we can't build it.");
295    }
296
297    // set the creation time, if one hasn't been set
298    long time = snapshot.getCreationTime();
299    if (time == SnapshotDescriptionUtils.NO_SNAPSHOT_START_TIME_SPECIFIED) {
300      time = EnvironmentEdgeManager.currentTime();
301      LOG.debug("Creation time not specified, setting to:" + time + " (current time:"
302          + EnvironmentEdgeManager.currentTime() + ").");
303      SnapshotDescription.Builder builder = snapshot.toBuilder();
304      builder.setCreationTime(time);
305      snapshot = builder.build();
306    }
307
308    long ttl = snapshot.getTtl();
309    // set default ttl(sec) if it is not set already or the value is out of the range
310    if (ttl == SnapshotDescriptionUtils.NO_SNAPSHOT_TTL_SPECIFIED ||
311        ttl > TimeUnit.MILLISECONDS.toSeconds(Long.MAX_VALUE)) {
312      final long defaultSnapshotTtl = conf.getLong(HConstants.DEFAULT_SNAPSHOT_TTL_CONFIG_KEY,
313          HConstants.DEFAULT_SNAPSHOT_TTL);
314      if (LOG.isDebugEnabled()) {
315        LOG.debug("Snapshot current TTL value: {} resetting it to default value: {}", ttl,
316            defaultSnapshotTtl);
317      }
318      ttl = defaultSnapshotTtl;
319    }
320    SnapshotDescription.Builder builder = snapshot.toBuilder();
321    builder.setTtl(ttl);
322    snapshot = builder.build();
323
324    // set the acl to snapshot if security feature is enabled.
325    if (isSecurityAvailable(conf)) {
326      snapshot = writeAclToSnapshotDescription(snapshot, conf);
327    }
328    return snapshot;
329  }

331  /**
332   * Write the snapshot description into the working directory of a snapshot
333   * @param snapshot description of the snapshot being taken
334   * @param workingDir working directory of the snapshot
335   * @param fs {@link FileSystem} on which the snapshot should be taken
336   * @throws IOException if we can't reach the filesystem and the file cannot be cleaned up on
337   *           failure
338   */
339  public static void writeSnapshotInfo(SnapshotDescription snapshot, Path workingDir, FileSystem fs)
340      throws IOException {
341    FsPermission perms = CommonFSUtils.getFilePermissions(fs, fs.getConf(),
342      HConstants.DATA_FILE_UMASK_KEY);
343    Path snapshotInfo = new Path(workingDir, SnapshotDescriptionUtils.SNAPSHOTINFO_FILE);
344    try {
345      FSDataOutputStream out = CommonFSUtils.create(fs, snapshotInfo, perms, true);
346      try {
347        snapshot.writeTo(out);
348      } finally {
349        out.close();
350      }
351    } catch (IOException e) {
352      // if we get an exception, try to remove the snapshot info
353      if (!fs.delete(snapshotInfo, false)) {
354        String msg = "Couldn't delete snapshot info file: " + snapshotInfo;
355        LOG.error(msg);
356        throw new IOException(msg);
357      }
358    }
359  }

361  /**
362   * Read in the {@link SnapshotDescription} stored for the snapshot in the passed directory
363   * @param fs filesystem where the snapshot was taken
364   * @param snapshotDir directory where the snapshot was stored
365   * @return the stored snapshot description
366   * @throws CorruptedSnapshotException if the snapshot cannot be read
367   */
368  public static SnapshotDescription readSnapshotInfo(FileSystem fs, Path snapshotDir)
369      throws CorruptedSnapshotException {
370    Path snapshotInfo = new Path(snapshotDir, SNAPSHOTINFO_FILE);
371    try {
372      FSDataInputStream in = null;
373      try {
374        in = fs.open(snapshotInfo);
375        SnapshotDescription desc = SnapshotDescription.parseFrom(in);
376        return desc;
377      } finally {
378        if (in != null) in.close();
379      }
380    } catch (IOException e) {
381      throw new CorruptedSnapshotException("Couldn't read snapshot info from:" + snapshotInfo, e);
382    }
383  }

385  /**
386   * Move the finished snapshot to its final, publicly visible directory - this marks the snapshot
387   * as 'complete'.
388   * @param snapshot description of the snapshot being tabken
389   * @param rootdir root directory of the hbase installation
390   * @param workingDir directory where the in progress snapshot was built
391   * @param fs {@link FileSystem} where the snapshot was built
392   * @throws org.apache.hadoop.hbase.snapshot.SnapshotCreationException if the
393   * snapshot could not be moved
394   * @throws IOException the filesystem could not be reached
395   */
396  public static void completeSnapshot(SnapshotDescription snapshot, Path rootdir, Path workingDir,
397      FileSystem fs) throws SnapshotCreationException, IOException {
398    Path finishedDir = getCompletedSnapshotDir(snapshot, rootdir);
399    LOG.debug("Snapshot is done, just moving the snapshot from " + workingDir + " to "
400        + finishedDir);
401    if (!fs.rename(workingDir, finishedDir)) {
402      throw new SnapshotCreationException(
403          "Failed to move working directory(" + workingDir + ") to completed directory("
404              + finishedDir + ").", ProtobufUtil.createSnapshotDesc(snapshot));
405    }
406  }

408  /**
409   * Check if the user is this table snapshot's owner
410   * @param snapshot the table snapshot description
411   * @param user the user
412   * @return true if the user is the owner of the snapshot,
413   *         false otherwise or the snapshot owner field is not present.
414   */
415  public static boolean isSnapshotOwner(org.apache.hadoop.hbase.client.SnapshotDescription snapshot,
416      User user) {
417    if (user == null) return false;
418    return user.getShortName().equals(snapshot.getOwner());
419  }

421  public static boolean isSecurityAvailable(Configuration conf) throws IOException {
422    try (Connection conn = ConnectionFactory.createConnection(conf)) {
423      try (Admin admin = conn.getAdmin()) {
424        return admin.tableExists(PermissionStorage.ACL_TABLE_NAME);
425      }
426    }
427  }
428
429  private static SnapshotDescription writeAclToSnapshotDescription(SnapshotDescription snapshot,
430      Configuration conf) throws IOException {
431    ListMultimap<String, UserPermission> perms =
432        User.runAsLoginUser(new PrivilegedExceptionAction<ListMultimap<String, UserPermission>>() {
433          @Override
434          public ListMultimap<String, UserPermission> run() throws Exception {
435            return PermissionStorage.getTablePermissions(conf,
436              TableName.valueOf(snapshot.getTable()));
437          }
438        });
439    return snapshot.toBuilder()
440        .setUsersAndPermissions(ShadedAccessControlUtil.toUserTablePermissions(perms)).build();
441  }
