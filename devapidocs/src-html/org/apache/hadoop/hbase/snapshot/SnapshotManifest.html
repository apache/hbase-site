062/**
063 * Utility class to help read/write the Snapshot Manifest.
064 *
065 * The snapshot format is transparent for the users of this class,
066 * once the snapshot is written, it will never be modified.
067 * On open() the snapshot will be loaded to the current in-memory format.
068 */
069@InterfaceAudience.Private
070public final class SnapshotManifest {
071  private static final Log LOG = LogFactory.getLog(SnapshotManifest.class);
072
073  public static final String SNAPSHOT_MANIFEST_SIZE_LIMIT_CONF_KEY = "snapshot.manifest.size.limit";
074
075  public static final String DATA_MANIFEST_NAME = "data.manifest";

078  private SnapshotDescription desc;

081  private final ForeignExceptionSnare monitor;
082  private final Configuration conf;
083  private final Path workingDir;
084  private final FileSystem fs;
085  private int manifestSizeLimit;

087  private SnapshotManifest(final Configuration conf, final FileSystem fs,
088      final Path workingDir, final SnapshotDescription desc,
089      final ForeignExceptionSnare monitor) {
090    this.monitor = monitor;
091    this.desc = desc;
092    this.workingDir = workingDir;
093    this.conf = conf;
094    this.fs = fs;
095
096    this.manifestSizeLimit = conf.getInt(SNAPSHOT_MANIFEST_SIZE_LIMIT_CONF_KEY, 64 * 1024 * 1024);
097  }

099  /**
100   * Return a SnapshotManifest instance, used for writing a snapshot.
101   *
102   * There are two usage pattern:
103   *  - The Master will create a manifest, add the descriptor, offline regions
104   *    and consolidate the snapshot by writing all the pending stuff on-disk.
105   *      manifest = SnapshotManifest.create(...)
106   *      manifest.addRegion(tableDir, hri)
107   *      manifest.consolidate()
108   *  - The RegionServer will create a single region manifest
109   *      manifest = SnapshotManifest.create(...)
110   *      manifest.addRegion(region)
111   */
112  public static SnapshotManifest create(final Configuration conf, final FileSystem fs,
113      final Path workingDir, final SnapshotDescription desc,
114      final ForeignExceptionSnare monitor) {
115    return new SnapshotManifest(conf, fs, workingDir, desc, monitor);
116  }
117
118  /**
119   * Return a SnapshotManifest instance with the information already loaded in-memory.
120   *    SnapshotManifest manifest = SnapshotManifest.open(...)
121   *    HTableDescriptor htd = manifest.getTableDescriptor()
122   *    for (SnapshotRegionManifest regionManifest: manifest.getRegionManifests())
123   *      hri = regionManifest.getRegionInfo()
124   *      for (regionManifest.getFamilyFiles())
125   *        ...
126   */
127  public static SnapshotManifest open(final Configuration conf, final FileSystem fs,
128      final Path workingDir, final SnapshotDescription desc) throws IOException {
129    SnapshotManifest manifest = new SnapshotManifest(conf, fs, workingDir, desc, null);
130    manifest.load();
131    return manifest;
132  }

142  interface RegionVisitor<TRegion, TFamily> {
143    TRegion regionOpen(final HRegionInfo regionInfo) throws IOException;
144    void regionClose(final TRegion region) throws IOException;
145
146    TFamily familyOpen(final TRegion region, final byte[] familyName) throws IOException;
147    void familyClose(final TRegion region, final TFamily family) throws IOException;
148
149    void storeFile(final TRegion region, final TFamily family, final StoreFileInfo storeFile)
150      throws IOException;
151  }
152
153  private RegionVisitor createRegionVisitor(final SnapshotDescription desc) throws IOException {
154    switch (getSnapshotFormat(desc)) {
155      case SnapshotManifestV1.DESCRIPTOR_VERSION:
156        return new SnapshotManifestV1.ManifestBuilder(conf, fs, workingDir);
157      case SnapshotManifestV2.DESCRIPTOR_VERSION:
158        return new SnapshotManifestV2.ManifestBuilder(conf, fs, workingDir);
159      default:
160      throw new CorruptedSnapshotException("Invalid Snapshot version: " + desc.getVersion(),
161        ProtobufUtil.createSnapshotDesc(desc));
162    }
163  }

165  public void addMobRegion(HRegionInfo regionInfo) throws IOException {
166    // 0. Get the ManifestBuilder/RegionVisitor
167    RegionVisitor visitor = createRegionVisitor(desc);
168
169    // 1. dump region meta info into the snapshot directory
170    LOG.debug("Storing mob region '" + regionInfo + "' region-info for snapshot.");
171    Object regionData = visitor.regionOpen(regionInfo);
172    monitor.rethrowException();
173
174    // 2. iterate through all the stores in the region
175    LOG.debug("Creating references for mob files");
176
177    Path mobRegionPath = MobUtils.getMobRegionPath(conf, regionInfo.getTable());
178    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
179      // 2.1. build the snapshot reference for the store if it's a mob store
180      if (!hcd.isMobEnabled()) {
181        continue;
182      }
183      Object familyData = visitor.familyOpen(regionData, hcd.getName());
184      monitor.rethrowException();
185
186      Path storePath = MobUtils.getMobFamilyPath(mobRegionPath, hcd.getNameAsString());
187      List<StoreFileInfo> storeFiles = getStoreFiles(storePath);
188      if (storeFiles == null) {
189        if (LOG.isDebugEnabled()) {
190          LOG.debug("No mob files under family: " + hcd.getNameAsString());
191        }
192        continue;
193      }
194
195      addReferenceFiles(visitor, regionData, familyData, storeFiles, true);
196
197      visitor.familyClose(regionData, familyData);
198    }
199    visitor.regionClose(regionData);
200  }

202  /**
203   * Creates a 'manifest' for the specified region, by reading directly from the HRegion object.
204   * This is used by the "online snapshot" when the table is enabled.
205   */
206  public void addRegion(final HRegion region) throws IOException {
207    // 0. Get the ManifestBuilder/RegionVisitor
208    RegionVisitor visitor = createRegionVisitor(desc);
209
210    // 1. dump region meta info into the snapshot directory
211    LOG.debug("Storing '" + region + "' region-info for snapshot.");
212    Object regionData = visitor.regionOpen(region.getRegionInfo());
213    monitor.rethrowException();
214
215    // 2. iterate through all the stores in the region
216    LOG.debug("Creating references for hfiles");
217
218    for (Store store : region.getStores()) {
219      // 2.1. build the snapshot reference for the store
220      Object familyData = visitor.familyOpen(regionData, store.getFamily().getName());
221      monitor.rethrowException();
222
223      List<StoreFile> storeFiles = new ArrayList<>(store.getStorefiles());
224      if (LOG.isDebugEnabled()) {
225        LOG.debug("Adding snapshot references for " + storeFiles  + " hfiles");
226      }
227
228      // 2.2. iterate through all the store's files and create "references".
229      for (int i = 0, sz = storeFiles.size(); i < sz; i++) {
230        StoreFile storeFile = storeFiles.get(i);
231        monitor.rethrowException();
232
233        // create "reference" to this store file.
234        LOG.debug("Adding reference for file (" + (i+1) + "/" + sz + "): " + storeFile.getPath());
235        visitor.storeFile(regionData, familyData, storeFile.getFileInfo());
236      }
237      visitor.familyClose(regionData, familyData);
238    }
239    visitor.regionClose(regionData);
240  }

242  /**
243   * Creates a 'manifest' for the specified region, by reading directly from the disk.
244   * This is used by the "offline snapshot" when the table is disabled.
245   */
246  public void addRegion(final Path tableDir, final HRegionInfo regionInfo) throws IOException {
247    // 0. Get the ManifestBuilder/RegionVisitor
248    RegionVisitor visitor = createRegionVisitor(desc);
249
250    boolean isMobRegion = MobUtils.isMobRegionInfo(regionInfo);
251    try {
252      Path baseDir = tableDir;
253      // Open the RegionFS
254      if (isMobRegion) {
255        baseDir = FSUtils.getTableDir(MobUtils.getMobHome(conf), regionInfo.getTable());
256      }
257      HRegionFileSystem regionFs = HRegionFileSystem.openRegionFromFileSystem(conf, fs,
258        baseDir, regionInfo, true);
259      monitor.rethrowException();
260
261      // 1. dump region meta info into the snapshot directory
262      LOG.debug("Storing region-info for snapshot.");
263      Object regionData = visitor.regionOpen(regionInfo);
264      monitor.rethrowException();
265
266      // 2. iterate through all the stores in the region
267      LOG.debug("Creating references for hfiles");
268
269      // This ensures that we have an atomic view of the directory as long as we have < ls limit
270      // (batch size of the files in a directory) on the namenode. Otherwise, we get back the files
271      // in batches and may miss files being added/deleted. This could be more robust (iteratively
272      // checking to see if we have all the files until we are sure), but the limit is currently
273      // 1000 files/batch, far more than the number of store files under a single column family.
274      Collection<String> familyNames = regionFs.getFamilies();
275      if (familyNames != null) {
276        for (String familyName: familyNames) {
277          Object familyData = visitor.familyOpen(regionData, Bytes.toBytes(familyName));
278          monitor.rethrowException();
279
280          Collection<StoreFileInfo> storeFiles = regionFs.getStoreFiles(familyName);
281          if (storeFiles == null) {
282            if (LOG.isDebugEnabled()) {
283              LOG.debug("No files under family: " + familyName);
284            }
285            continue;
286          }
287
288          // 2.1. build the snapshot reference for the store
289          // iterate through all the store's files and create "references".
290          addReferenceFiles(visitor, regionData, familyData, storeFiles, false);
291
292          visitor.familyClose(regionData, familyData);
293        }
294      }
295      visitor.regionClose(regionData);
296    } catch (IOException e) {
297      // the mob directory might not be created yet, so do nothing when it is a mob region
298      if (!isMobRegion) {
299        throw e;
300      }
301    }
302  }
303
304  private List<StoreFileInfo> getStoreFiles(Path storeDir) throws IOException {
305    FileStatus[] stats = FSUtils.listStatus(fs, storeDir);
306    if (stats == null) return null;
307
308    ArrayList<StoreFileInfo> storeFiles = new ArrayList<>(stats.length);
309    for (int i = 0; i < stats.length; ++i) {
310      storeFiles.add(new StoreFileInfo(conf, fs, stats[i]));
311    }
312    return storeFiles;
313  }
314
315  private void addReferenceFiles(RegionVisitor visitor, Object regionData, Object familyData,
316      Collection<StoreFileInfo> storeFiles, boolean isMob) throws IOException {
317    final String fileType = isMob ? "mob file" : "hfile";
318
319    if (LOG.isDebugEnabled()) {
320      LOG.debug(String.format("Adding snapshot references for %s %ss", storeFiles, fileType));
321    }
322
323    int i = 0;
324    int sz = storeFiles.size();
325    for (StoreFileInfo storeFile: storeFiles) {
326      monitor.rethrowException();
327
328      LOG.debug(String.format("Adding reference for %s (%d/%d): %s",
329          fileType, ++i, sz, storeFile.getPath()));
330
331      // create "reference" to this store file.
332      visitor.storeFile(regionData, familyData, storeFile);
333    }
334  }

336  /**
337   * Load the information in the SnapshotManifest. Called by SnapshotManifest.open()
338   *
339   * If the format is v2 and there is no data-manifest, means that we are loading an
340   * in-progress snapshot. Since we support rolling-upgrades, we loook for v1 and v2
341   * regions format.
342   */
343  private void load() throws IOException {
344    switch (getSnapshotFormat(desc)) {
345      case SnapshotManifestV1.DESCRIPTOR_VERSION: {
346        this.htd = FSTableDescriptors.getTableDescriptorFromFs(fs, workingDir);
347        ThreadPoolExecutor tpool = createExecutor("SnapshotManifestLoader");
348        try {
349          this.regionManifests =
350            SnapshotManifestV1.loadRegionManifests(conf, tpool, fs, workingDir, desc);
351        } finally {
352          tpool.shutdown();
353        }
354        break;
355      }
356      case SnapshotManifestV2.DESCRIPTOR_VERSION: {
357        SnapshotDataManifest dataManifest = readDataManifest();
358        if (dataManifest != null) {
359          htd = ProtobufUtil.convertToHTableDesc(dataManifest.getTableSchema());
360          regionManifests = dataManifest.getRegionManifestsList();
361        } else {
362          // Compatibility, load the v1 regions
363          // This happens only when the snapshot is in-progress and the cache wants to refresh.
364          List<SnapshotRegionManifest> v1Regions, v2Regions;
365          ThreadPoolExecutor tpool = createExecutor("SnapshotManifestLoader");
366          try {
367            v1Regions = SnapshotManifestV1.loadRegionManifests(conf, tpool, fs, workingDir, desc);
368            v2Regions = SnapshotManifestV2.loadRegionManifests(conf, tpool, fs, workingDir, desc,
369                manifestSizeLimit);
370          } catch (InvalidProtocolBufferException e) {
371            throw new CorruptedSnapshotException("unable to parse region manifest " +
372                e.getMessage(), e);
373          } finally {
374            tpool.shutdown();
375          }
376          if (v1Regions != null && v2Regions != null) {
377            regionManifests = new ArrayList<>(v1Regions.size() + v2Regions.size());
378            regionManifests.addAll(v1Regions);
379            regionManifests.addAll(v2Regions);
380          } else if (v1Regions != null) {
381            regionManifests = v1Regions;
382          } else /* if (v2Regions != null) */ {
383            regionManifests = v2Regions;
384          }
385        }
386        break;
387      }
388      default:
389      throw new CorruptedSnapshotException("Invalid Snapshot version: " + desc.getVersion(),
390        ProtobufUtil.createSnapshotDesc(desc));
391    }
392  }

422  /**
423   * Get all the Region Manifest from the snapshot.
424   * This is an helper to get a map with the region encoded name
425   */
426  public Map<String, SnapshotRegionManifest> getRegionManifestsMap() {
427    if (regionManifests == null || regionManifests.isEmpty()) return null;
428
429    HashMap<String, SnapshotRegionManifest> regionsMap = new HashMap<>(regionManifests.size());
430    for (SnapshotRegionManifest manifest: regionManifests) {
431      String regionName = getRegionNameFromManifest(manifest);
432      regionsMap.put(regionName, manifest);
433    }
434    return regionsMap;
435  }

437  public void consolidate() throws IOException {
438    if (getSnapshotFormat(desc) == SnapshotManifestV1.DESCRIPTOR_VERSION) {
439      Path rootDir = FSUtils.getRootDir(conf);
440      LOG.info("Using old Snapshot Format");
441      // write a copy of descriptor to the snapshot directory
442      new FSTableDescriptors(conf, fs, rootDir)
443        .createTableDescriptorForTableDirectory(workingDir, htd, false);
444    } else {
445      LOG.debug("Convert to Single Snapshot Manifest");
446      convertToV2SingleManifest();
447    }
448  }

450  /*
451   * In case of rolling-upgrade, we try to read all the formats and build
452   * the snapshot with the latest format.
453   */
454  private void convertToV2SingleManifest() throws IOException {
455    // Try to load v1 and v2 regions
456    List<SnapshotRegionManifest> v1Regions, v2Regions;
457    ThreadPoolExecutor tpool = createExecutor("SnapshotManifestLoader");
458    try {
459      v1Regions = SnapshotManifestV1.loadRegionManifests(conf, tpool, fs, workingDir, desc);
460      v2Regions = SnapshotManifestV2.loadRegionManifests(conf, tpool, fs, workingDir, desc,
461          manifestSizeLimit);
462    } finally {
463      tpool.shutdown();
464    }
465
466    SnapshotDataManifest.Builder dataManifestBuilder = SnapshotDataManifest.newBuilder();
467    dataManifestBuilder.setTableSchema(ProtobufUtil.convertToTableSchema(htd));
468
469    if (v1Regions != null && v1Regions.size() > 0) {
470      dataManifestBuilder.addAllRegionManifests(v1Regions);
471    }
472    if (v2Regions != null && v2Regions.size() > 0) {
473      dataManifestBuilder.addAllRegionManifests(v2Regions);
474    }
475
476    // Write the v2 Data Manifest.
477    // Once the data-manifest is written, the snapshot can be considered complete.
478    // Currently snapshots are written in a "temporary" directory and later
479    // moved to the "complated" snapshot directory.
480    SnapshotDataManifest dataManifest = dataManifestBuilder.build();
481    writeDataManifest(dataManifest);
482    this.regionManifests = dataManifest.getRegionManifestsList();
483
484    // Remove the region manifests. Everything is now in the data-manifest.
485    // The delete operation is "relaxed", unless we get an exception we keep going.
486    // The extra files in the snapshot directory will not give any problem,
487    // since they have the same content as the data manifest, and even by re-reading
488    // them we will get the same information.
489    if (v1Regions != null && v1Regions.size() > 0) {
490      for (SnapshotRegionManifest regionManifest: v1Regions) {
491        SnapshotManifestV1.deleteRegionManifest(fs, workingDir, regionManifest);
492      }
493    }
494    if (v2Regions != null && v2Regions.size() > 0) {
495      for (SnapshotRegionManifest regionManifest: v2Regions) {
496        SnapshotManifestV2.deleteRegionManifest(fs, workingDir, regionManifest);
497      }
498    }
499  }

501  /*
502   * Write the SnapshotDataManifest file
503   */
504  private void writeDataManifest(final SnapshotDataManifest manifest)
505      throws IOException {
506    FSDataOutputStream stream = fs.create(new Path(workingDir, DATA_MANIFEST_NAME));
507    try {
508      manifest.writeTo(stream);
509    } finally {
510      stream.close();
511    }
512  }
513
514  /*
515   * Read the SnapshotDataManifest file
516   */
517  private SnapshotDataManifest readDataManifest() throws IOException {
518    FSDataInputStream in = null;
519    try {
520      in = fs.open(new Path(workingDir, DATA_MANIFEST_NAME));
521      CodedInputStream cin = CodedInputStream.newInstance(in);
522      cin.setSizeLimit(manifestSizeLimit);
523      return SnapshotDataManifest.parseFrom(cin);
524    } catch (FileNotFoundException e) {
525      return null;
526    } catch (InvalidProtocolBufferException e) {
527      throw new CorruptedSnapshotException("unable to parse data manifest " + e.getMessage(), e);
528    } finally {
529      if (in != null) in.close();
530    }
531  }
532
533  private ThreadPoolExecutor createExecutor(final String name) {
534    return createExecutor(conf, name);
535  }
536
537  public static ThreadPoolExecutor createExecutor(final Configuration conf, final String name) {
538    int maxThreads = conf.getInt("hbase.snapshot.thread.pool.max", 8);
539    return Threads.getBoundedCachedThreadPool(maxThreads, 30L, TimeUnit.SECONDS,
540              Threads.getNamedThreadFactory(name));
541  }

543  /**
544   * Extract the region encoded name from the region manifest
545   */
546  static String getRegionNameFromManifest(final SnapshotRegionManifest manifest) {
547    byte[] regionName = HRegionInfo.createRegionName(
548            ProtobufUtil.toTableName(manifest.getRegionInfo().getTableName()),
549            manifest.getRegionInfo().getStartKey().toByteArray(),
550            manifest.getRegionInfo().getRegionId(), true);
551    return HRegionInfo.encodeRegionName(regionName);
552  }
