072/**
073 * Helper to Restore/Clone a Snapshot
074 *
075 * <p>The helper assumes that a table is already created, and by calling restore()
076 * the content present in the snapshot will be restored as the new content of the table.
077 *
078 * <p>Clone from Snapshot: If the target table is empty, the restore operation
079 * is just a "clone operation", where the only operations are:
080 * <ul>
081 *  <li>for each region in the snapshot create a new region
082 *    (note that the region will have a different name, since the encoding contains the table name)
083 *  <li>for each file in the region create a new HFileLink to point to the original file.
084 *  <li>restore the logs, if any
085 * </ul>
086 *
087 * <p>Restore from Snapshot:
088 * <ul>
089 *  <li>for each region in the table verify which are available in the snapshot and which are not
090 *    <ul>
091 *    <li>if the region is not present in the snapshot, remove it.
092 *    <li>if the region is present in the snapshot
093 *      <ul>
094 *      <li>for each file in the table region verify which are available in the snapshot
095 *        <ul>
096 *          <li>if the hfile is not present in the snapshot, remove it
097 *          <li>if the hfile is present, keep it (nothing to do)
098 *        </ul>
099 *      <li>for each file in the snapshot region but not in the table
100 *        <ul>
101 *          <li>create a new HFileLink that point to the original file
102 *        </ul>
103 *      </ul>
104 *    </ul>
105 *  <li>for each region in the snapshot not present in the current table state
106 *    <ul>
107 *    <li>create a new region and for each file in the region create a new HFileLink
108 *      (This is the same as the clone operation)
109 *    </ul>
110 *  <li>restore the logs, if any
111 * </ul>
112 */

168  /**
169   * Restore the on-disk table to a specified snapshot state.
170   * @return the set of regions touched by the restore operation
171   */
172  public RestoreMetaChanges restoreHdfsRegions() throws IOException {
173    ThreadPoolExecutor exec = SnapshotManifest.createExecutor(conf, "RestoreSnapshot");
174    try {
175      return restoreHdfsRegions(exec);
176    } finally {
177      exec.shutdown();
178    }
179  }
180
181  private RestoreMetaChanges restoreHdfsRegions(final ThreadPoolExecutor exec) throws IOException {
182    LOG.info("starting restore table regions using snapshot=" + snapshotDesc);
183
184    Map<String, SnapshotRegionManifest> regionManifests = snapshotManifest.getRegionManifestsMap();
185    if (regionManifests == null) {
186      LOG.warn("Nothing to restore. Snapshot " + snapshotDesc + " looks empty");
187      return null;
188    }
189
190    RestoreMetaChanges metaChanges = new RestoreMetaChanges(tableDesc, parentsMap);
191
192    // Take a copy of the manifest.keySet() since we are going to modify
193    // this instance, by removing the regions already present in the restore dir.
194    Set<String> regionNames = new HashSet<>(regionManifests.keySet());
195
196    HRegionInfo mobRegion = MobUtils.getMobRegionInfo(snapshotManifest.getTableDescriptor()
197        .getTableName());
198    // Identify which region are still available and which not.
199    // NOTE: we rely upon the region name as: "table name, start key, end key"
200    List<HRegionInfo> tableRegions = getTableRegions();
201    if (tableRegions != null) {
202      monitor.rethrowException();
203      for (HRegionInfo regionInfo: tableRegions) {
204        String regionName = regionInfo.getEncodedName();
205        if (regionNames.contains(regionName)) {
206          LOG.info("region to restore: " + regionName);
207          regionNames.remove(regionName);
208          metaChanges.addRegionToRestore(regionInfo);
209        } else {
210          LOG.info("region to remove: " + regionName);
211          metaChanges.addRegionToRemove(regionInfo);
212        }
213      }
214
215      // Restore regions using the snapshot data
216      monitor.rethrowException();
217      status.setStatus("Restoring table regions...");
218      if (regionNames.contains(mobRegion.getEncodedName())) {
219        // restore the mob region in case
220        List<HRegionInfo> mobRegions = new ArrayList<>(1);
221        mobRegions.add(mobRegion);
222        restoreHdfsMobRegions(exec, regionManifests, mobRegions);
223        regionNames.remove(mobRegion.getEncodedName());
224      }
225      restoreHdfsRegions(exec, regionManifests, metaChanges.getRegionsToRestore());
226      status.setStatus("Finished restoring all table regions.");
227
228      // Remove regions from the current table
229      monitor.rethrowException();
230      status.setStatus("Starting to delete excess regions from table");
231      removeHdfsRegions(exec, metaChanges.getRegionsToRemove());
232      status.setStatus("Finished deleting excess regions from table.");
233    }
234
235    // Regions to Add: present in the snapshot but not in the current table
236    if (regionNames.size() > 0) {
237      List<HRegionInfo> regionsToAdd = new ArrayList<>(regionNames.size());
238
239      monitor.rethrowException();
240      // add the mob region
241      if (regionNames.contains(mobRegion.getEncodedName())) {
242        cloneHdfsMobRegion(regionManifests, mobRegion);
243        regionNames.remove(mobRegion.getEncodedName());
244      }
245      for (String regionName: regionNames) {
246        LOG.info("region to add: " + regionName);
247        regionsToAdd.add(HRegionInfo.convert(regionManifests.get(regionName).getRegionInfo()));
248      }
249
250      // Create new regions cloning from the snapshot
251      monitor.rethrowException();
252      status.setStatus("Cloning regions...");
253      HRegionInfo[] clonedRegions = cloneHdfsRegions(exec, regionManifests, regionsToAdd);
254      metaChanges.setNewRegions(clonedRegions);
255      status.setStatus("Finished cloning regions.");
256    }
257
258    LOG.info("finishing restore table regions using snapshot=" + snapshotDesc);
259
260    return metaChanges;
261  }

266  public static class RestoreMetaChanges {

356    void addRegionToRestore(final HRegionInfo hri) {

363    public void updateMetaParentRegions(Connection connection,
364        final List<HRegionInfo> regionInfos) throws IOException {
365      if (regionInfos == null || parentsMap.isEmpty()) return;
366
367      // Extract region names and offlined regions
368      Map<String, HRegionInfo> regionsByName = new HashMap<>(regionInfos.size());
369      List<HRegionInfo> parentRegions = new LinkedList<>();
370      for (HRegionInfo regionInfo: regionInfos) {
371        if (regionInfo.isSplitParent()) {
372          parentRegions.add(regionInfo);
373        } else {
374          regionsByName.put(regionInfo.getEncodedName(), regionInfo);
375        }
376      }
377
378      // Update Offline parents
379      for (HRegionInfo regionInfo: parentRegions) {
380        Pair<String, String> daughters = parentsMap.get(regionInfo.getEncodedName());
381        if (daughters == null) {
382          // The snapshot contains an unreferenced region.
383          // It will be removed by the CatalogJanitor.
384          LOG.warn("Skip update of unreferenced offline parent: " + regionInfo);
385          continue;
386        }
387
388        // One side of the split is already compacted
389        if (daughters.getSecond() == null) {
390          daughters.setSecond(daughters.getFirst());
391        }
392
393        LOG.debug("Update splits parent " + regionInfo.getEncodedName() + " -> " + daughters);
394        MetaTableAccessor.addRegionToMeta(connection, regionInfo,
395          regionsByName.get(daughters.getFirst()),
396          regionsByName.get(daughters.getSecond()));
397      }
398    }

478  /**
479   * Restore region by removing files not in the snapshot
480   * and adding the missing ones from the snapshot.
481   */
482  private void restoreRegion(final HRegionInfo regionInfo,
483      final SnapshotRegionManifest regionManifest, Path regionDir) throws IOException {
484    Map<String, List<SnapshotRegionManifest.StoreFile>> snapshotFiles =
485                getRegionHFileReferences(regionManifest);
486
487    String tableName = tableDesc.getTableName().getNameAsString();
488
489    // Restore families present in the table
490    for (Path familyDir: FSUtils.getFamilyDirs(fs, regionDir)) {
491      byte[] family = Bytes.toBytes(familyDir.getName());
492      Set<String> familyFiles = getTableRegionFamilyFiles(familyDir);
493      List<SnapshotRegionManifest.StoreFile> snapshotFamilyFiles =
494          snapshotFiles.remove(familyDir.getName());
495      if (snapshotFamilyFiles != null) {
496        List<SnapshotRegionManifest.StoreFile> hfilesToAdd = new ArrayList<>();
497        for (SnapshotRegionManifest.StoreFile storeFile: snapshotFamilyFiles) {
498          if (familyFiles.contains(storeFile.getName())) {
499            // HFile already present
500            familyFiles.remove(storeFile.getName());
501          } else {
502            // HFile missing
503            hfilesToAdd.add(storeFile);
504          }
505        }
506
507        // Remove hfiles not present in the snapshot
508        for (String hfileName: familyFiles) {
509          Path hfile = new Path(familyDir, hfileName);
510          LOG.trace("Removing hfile=" + hfileName +
511            " from region=" + regionInfo.getEncodedName() + " table=" + tableName);
512          HFileArchiver.archiveStoreFile(conf, fs, regionInfo, tableDir, family, hfile);
513        }
514
515        // Restore Missing files
516        for (SnapshotRegionManifest.StoreFile storeFile: hfilesToAdd) {
517          LOG.debug("Adding HFileLink " + storeFile.getName() +
518            " to region=" + regionInfo.getEncodedName() + " table=" + tableName);
519          restoreStoreFile(familyDir, regionInfo, storeFile, createBackRefs);
520        }
521      } else {
522        // Family doesn't exists in the snapshot
523        LOG.trace("Removing family=" + Bytes.toString(family) +
524          " from region=" + regionInfo.getEncodedName() + " table=" + tableName);
525        HFileArchiver.archiveFamilyByFamilyDir(fs, conf, regionInfo, familyDir, family);
526        fs.delete(familyDir, true);
527      }
528    }
529
530    // Add families not present in the table
531    for (Map.Entry<String, List<SnapshotRegionManifest.StoreFile>> familyEntry:
532                                                                      snapshotFiles.entrySet()) {
533      Path familyDir = new Path(regionDir, familyEntry.getKey());
534      if (!fs.mkdirs(familyDir)) {
535        throw new IOException("Unable to create familyDir=" + familyDir);
536      }
537
538      for (SnapshotRegionManifest.StoreFile storeFile: familyEntry.getValue()) {
539        LOG.trace("Adding HFileLink " + storeFile.getName() + " to table=" + tableName);
540        restoreStoreFile(familyDir, regionInfo, storeFile, createBackRefs);
541      }
542    }
543  }

545  /**
546   * @return The set of files in the specified family directory.
547   */
548  private Set<String> getTableRegionFamilyFiles(final Path familyDir) throws IOException {
549    FileStatus[] hfiles = FSUtils.listStatus(fs, familyDir);
550    if (hfiles == null) return Collections.emptySet();
551
552    Set<String> familyFiles = new HashSet<>(hfiles.length);
553    for (int i = 0; i < hfiles.length; ++i) {
554      String hfileName = hfiles[i].getPath().getName();
555      familyFiles.add(hfileName);
556    }
557
558    return familyFiles;
559  }

561  /**
562   * Clone specified regions. For each region create a new region
563   * and create a HFileLink for each hfile.
564   */
565  private HRegionInfo[] cloneHdfsRegions(final ThreadPoolExecutor exec,
566      final Map<String, SnapshotRegionManifest> regionManifests,
567      final List<HRegionInfo> regions) throws IOException {
568    if (regions == null || regions.isEmpty()) return null;
569
570    final Map<String, HRegionInfo> snapshotRegions = new HashMap<>(regions.size());
571
572    // clone region info (change embedded tableName with the new one)
573    HRegionInfo[] clonedRegionsInfo = new HRegionInfo[regions.size()];
574    for (int i = 0; i < clonedRegionsInfo.length; ++i) {
575      // clone the region info from the snapshot region info
576      HRegionInfo snapshotRegionInfo = regions.get(i);
577      clonedRegionsInfo[i] = cloneRegionInfo(snapshotRegionInfo);
578
579      // add the region name mapping between snapshot and cloned
580      String snapshotRegionName = snapshotRegionInfo.getEncodedName();
581      String clonedRegionName = clonedRegionsInfo[i].getEncodedName();
582      regionsMap.put(Bytes.toBytes(snapshotRegionName), Bytes.toBytes(clonedRegionName));
583      LOG.info("clone region=" + snapshotRegionName + " as " + clonedRegionName);
584
585      // Add mapping between cloned region name and snapshot region info
586      snapshotRegions.put(clonedRegionName, snapshotRegionInfo);
587    }
588
589    // create the regions on disk
590    ModifyRegionUtils.createRegions(exec, conf, rootDir,
591      tableDesc, clonedRegionsInfo, new ModifyRegionUtils.RegionFillTask() {
592        @Override
593        public void fillRegion(final HRegion region) throws IOException {
594          HRegionInfo snapshotHri = snapshotRegions.get(region.getRegionInfo().getEncodedName());
595          cloneRegion(region, snapshotHri, regionManifests.get(snapshotHri.getEncodedName()));
596        }
597      });
598
599    return clonedRegionsInfo;
600  }

606  private void cloneHdfsMobRegion(final Map<String, SnapshotRegionManifest> regionManifests,

613  /**
614   * Clone region directory content from the snapshot info.
615   *
616   * Each region is encoded with the table name, so the cloned region will have
617   * a different region name.
618   *
619   * Instead of copying the hfiles a HFileLink is created.
620   *
621   * @param regionDir {@link Path} cloned dir
622   * @param snapshotRegionInfo
623   */
624  private void cloneRegion(final Path regionDir, final HRegionInfo snapshotRegionInfo,
625      final SnapshotRegionManifest manifest) throws IOException {
626    final String tableName = tableDesc.getTableName().getNameAsString();
627    for (SnapshotRegionManifest.FamilyFiles familyFiles: manifest.getFamilyFilesList()) {
628      Path familyDir = new Path(regionDir, familyFiles.getFamilyName().toStringUtf8());
629      for (SnapshotRegionManifest.StoreFile storeFile: familyFiles.getStoreFilesList()) {
630        LOG.info("Adding HFileLink " + storeFile.getName() + " to table=" + tableName);
631        restoreStoreFile(familyDir, snapshotRegionInfo, storeFile, createBackRefs);
632      }
633    }
634  }

653  /**
654   * Create a new {@link HFileLink} to reference the store file.
655   * <p>The store file in the snapshot can be a simple hfile, an HFileLink or a reference.
656   * <ul>
657   *   <li>hfile: abc -> table=region-abc
658   *   <li>reference: abc.1234 -> table=region-abc.1234
659   *   <li>hfilelink: table=region-hfile -> table=region-hfile
660   * </ul>
661   * @param familyDir destination directory for the store file
662   * @param regionInfo destination region info for the table
663   * @param createBackRef - Whether back reference should be created. Defaults to true.
664   * @param storeFile store file name (can be a Reference, HFileLink or simple HFile)
665   */
666  private void restoreStoreFile(final Path familyDir, final HRegionInfo regionInfo,
667      final SnapshotRegionManifest.StoreFile storeFile, final boolean createBackRef)
668          throws IOException {
669    String hfileName = storeFile.getName();
670    if (HFileLink.isHFileLink(hfileName)) {
671      HFileLink.createFromHFileLink(conf, fs, familyDir, hfileName, createBackRef);
672    } else if (StoreFileInfo.isReference(hfileName)) {
673      restoreReferenceFile(familyDir, regionInfo, storeFile);
674    } else {
675      HFileLink.create(conf, fs, familyDir, regionInfo, hfileName, createBackRef);
676    }
677  }

679  /**
680   * Create a new {@link Reference} as copy of the source one.
681   * <p><blockquote><pre>
682   * The source table looks like:
683   *    1234/abc      (original file)
684   *    5678/abc.1234 (reference file)
685   *
686   * After the clone operation looks like:
687   *   wxyz/table=1234-abc
688   *   stuv/table=1234-abc.wxyz
689   *
690   * NOTE that the region name in the clone changes (md5 of regioninfo)
691   * and the reference should reflect that change.
692   * </pre></blockquote>
693   * @param familyDir destination directory for the store file
694   * @param regionInfo destination region info for the table
695   * @param storeFile reference file name
696   */
697  private void restoreReferenceFile(final Path familyDir, final HRegionInfo regionInfo,
698      final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
699    String hfileName = storeFile.getName();
700
701    // Extract the referred information (hfile name and parent region)
702    Path refPath =
703        StoreFileInfo.getReferredToFile(new Path(new Path(new Path(new Path(snapshotTable
704            .getNamespaceAsString(), snapshotTable.getQualifierAsString()), regionInfo
705            .getEncodedName()), familyDir.getName()), hfileName));
706    String snapshotRegionName = refPath.getParent().getParent().getName();
707    String fileName = refPath.getName();
708
709    // The new reference should have the cloned region name as parent, if it is a clone.
710    String clonedRegionName = Bytes.toString(regionsMap.get(Bytes.toBytes(snapshotRegionName)));
711    if (clonedRegionName == null) clonedRegionName = snapshotRegionName;
712
713    // The output file should be a reference link table=snapshotRegion-fileName.clonedRegionName
714    Path linkPath = null;
715    String refLink = fileName;
716    if (!HFileLink.isHFileLink(fileName)) {
717      refLink = HFileLink.createHFileLinkName(snapshotTable, snapshotRegionName, fileName);
718      linkPath = new Path(familyDir,
719        HFileLink.createHFileLinkName(snapshotTable, regionInfo.getEncodedName(), hfileName));
720    }
721
722    Path outPath = new Path(familyDir, refLink + '.' + clonedRegionName);
723
724    // Create the new reference
725    if (storeFile.hasReference()) {
726      Reference reference = Reference.convert(storeFile.getReference());
727      reference.write(fs, outPath);
728    } else {
729      InputStream in;
730      if (linkPath != null) {
731        in = HFileLink.buildFromHFileLinkPattern(conf, linkPath).open(fs);
732      } else {
733        linkPath = new Path(new Path(HRegion.getRegionDir(snapshotManifest.getSnapshotDir(),
734                        regionInfo.getEncodedName()), familyDir.getName()), hfileName);
735        in = fs.open(linkPath);
736      }
737      OutputStream out = fs.create(outPath);
738      IOUtils.copyBytes(in, out, conf);
739    }
740
741    // Add the daughter region to the map
742    String regionName = Bytes.toString(regionsMap.get(regionInfo.getEncodedNameAsBytes()));
743    LOG.debug("Restore reference " + regionName + " to " + clonedRegionName);
744    synchronized (parentsMap) {
745      Pair<String, String> daughters = parentsMap.get(clonedRegionName);
746      if (daughters == null) {
747        daughters = new Pair<>(regionName, null);
748        parentsMap.put(clonedRegionName, daughters);
749      } else if (!regionName.equals(daughters.getFirst())) {
750        daughters.setSecond(regionName);
751      }
752    }
753  }

775  /**
776   * @return the set of the regions contained in the table
777   */
778  private List<HRegionInfo> getTableRegions() throws IOException {
779    LOG.debug("get table regions: " + tableDir);
780    FileStatus[] regionDirs = FSUtils.listStatus(fs, tableDir, new FSUtils.RegionDirFilter(fs));
781    if (regionDirs == null) return null;
782
783    List<HRegionInfo> regions = new ArrayList<>(regionDirs.length);
784    for (int i = 0; i < regionDirs.length; ++i) {
785      HRegionInfo hri = HRegionFileSystem.loadRegionInfoFileContent(fs, regionDirs[i].getPath());
786      regions.add(hri);
787    }
788    LOG.debug("found " + regions.size() + " regions for table=" +
789        tableDesc.getTableName().getNameAsString());
790    return regions;
791  }

793  /**
794   * Copy the snapshot files for a snapshot scanner, discards meta changes.
795   * @param conf
796   * @param fs
797   * @param rootDir
798   * @param restoreDir
799   * @param snapshotName
800   * @throws IOException
801   */
802  public static RestoreMetaChanges copySnapshotForScanner(Configuration conf, FileSystem fs,
803      Path rootDir, Path restoreDir, String snapshotName) throws IOException {
804    // ensure that restore dir is not under root dir
805    if (!restoreDir.getFileSystem(conf).getUri().equals(rootDir.getFileSystem(conf).getUri())) {
806      throw new IllegalArgumentException("Filesystems for restore directory and HBase root " +
807          "directory should be the same");
808    }
809    if (restoreDir.toUri().getPath().startsWith(rootDir.toUri().getPath())) {
810      throw new IllegalArgumentException("Restore directory cannot be a sub directory of HBase " +
811          "root directory. RootDir: " + rootDir + ", restoreDir: " + restoreDir);
812    }
813
814    Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
815    SnapshotDescription snapshotDesc = SnapshotDescriptionUtils.readSnapshotInfo(fs, snapshotDir);
816    SnapshotManifest manifest = SnapshotManifest.open(conf, fs, snapshotDir, snapshotDesc);
817
818    MonitoredTask status = TaskMonitor.get().createStatus(
819        "Restoring  snapshot '" + snapshotName + "' to directory " + restoreDir);
820    ForeignExceptionDispatcher monitor = new ForeignExceptionDispatcher();
821
822    // we send createBackRefs=false so that restored hfiles do not create back reference links
823    // in the base hbase root dir.
824    RestoreSnapshotHelper helper = new RestoreSnapshotHelper(conf, fs,
825      manifest, manifest.getTableDescriptor(), restoreDir, monitor, status, false);
826    RestoreMetaChanges metaChanges = helper.restoreHdfsRegions(); // TODO: parallelize.
827
828    if (LOG.isDebugEnabled()) {
829      LOG.debug("Restored table dir:" + restoreDir);
830      FSUtils.logFileSystemState(fs, restoreDir, LOG);
831    }
832    return metaChanges;
833  }

835  public static void restoreSnapshotAcl(SnapshotDescription snapshot, TableName newTableName,
836      Configuration conf) throws IOException {
837    if (snapshot.hasUsersAndPermissions() && snapshot.getUsersAndPermissions() != null) {
838      LOG.info("Restore snapshot acl to table. snapshot: " + snapshot + ", table: " + newTableName);
839      ListMultimap<String, TablePermission> perms =
840          ShadedAccessControlUtil.toUserTablePermissions(snapshot.getUsersAndPermissions());
841      try (Connection conn = ConnectionFactory.createConnection(conf)) {
842        for (Entry<String, TablePermission> e : perms.entries()) {
843          String user = e.getKey();
844          TablePermission perm = e.getValue();
845          perm.setTableName(newTableName);
846          AccessControlClient.grant(conn, perm.getTableName(), user, perm.getFamily(),
847            perm.getQualifier(), perm.getActions());
848        }
849      } catch (Throwable e) {
850        throw new IOException("Grant acl into newly creatd table failed. snapshot: " + snapshot
851            + ", table: " + newTableName, e);
852      }
853    }
854  }
