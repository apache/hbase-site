018package org.apache.hadoop.hbase.regionserver;
019
020import org.apache.commons.logging.Log;
021import org.apache.commons.logging.LogFactory;
022import org.apache.hadoop.hbase.HBaseInterfaceAudience;
023import org.apache.yetus.audience.InterfaceAudience;
024
025/**
026 * A {@link FlushPolicy} that only flushes store larger a given threshold. If no store is large
027 * enough, then all stores will be flushed.
028 */
029@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
030public abstract class FlushLargeStoresPolicy extends FlushPolicy {
031
032  private static final Log LOG = LogFactory.getLog(FlushLargeStoresPolicy.class);
033
034  public static final String HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND =
035      "hbase.hregion.percolumnfamilyflush.size.lower.bound";
036
037  public static final String HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN =
038      "hbase.hregion.percolumnfamilyflush.size.lower.bound.min";
039
040  public static final long DEFAULT_HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN =
041      1024 * 1024 * 16L;
042
043  protected long flushSizeLowerBound = -1;
044
045  protected long getFlushSizeLowerBound(HRegion region) {
046    int familyNumber = region.getTableDescriptor().getColumnFamilyCount();
047    // For multiple families, lower bound is the "average flush size" by default
048    // unless setting in configuration is larger.
049    long flushSizeLowerBound = region.getMemstoreFlushSize() / familyNumber;
050    long minimumLowerBound =
051        getConf().getLong(HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN,
052          DEFAULT_HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN);
053    if (minimumLowerBound > flushSizeLowerBound) {
054      flushSizeLowerBound = minimumLowerBound;
055    }
056    // use the setting in table description if any
057    String flushedSizeLowerBoundString =
058        region.getTableDescriptor().getValue(HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND);
059    if (flushedSizeLowerBoundString == null) {
060      if (LOG.isDebugEnabled()) {
061        LOG.debug("No " + HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND
062            + " set in description of table " + region.getTableDescriptor().getTableName()
063            + ", use config (" + flushSizeLowerBound + ") instead");
064      }
065    } else {
066      try {
067        flushSizeLowerBound = Long.parseLong(flushedSizeLowerBoundString);
068      } catch (NumberFormatException nfe) {
069        // fall back for fault setting
070        LOG.warn("Number format exception when parsing "
071            + HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND + " for table "
072            + region.getTableDescriptor().getTableName() + ":" + flushedSizeLowerBoundString + ". " + nfe
073            + ", use config (" + flushSizeLowerBound + ") instead");
074
075      }
076    }
077    return flushSizeLowerBound;
078  }
079
080  protected boolean shouldFlush(Store store) {
081    if (store.getSizeOfMemStore().getDataSize() > this.flushSizeLowerBound) {
082      if (LOG.isDebugEnabled()) {
083        LOG.debug("Flush Column Family " + store.getColumnFamilyName() + " of " +
084            region.getRegionInfo().getEncodedName() + " because memstoreSize=" +
085            store.getSizeOfMemStore().getDataSize() + " > lower bound="
086            + this.flushSizeLowerBound);
087      }
088      return true;
089    }
090    return false;
091  }
092
093}
