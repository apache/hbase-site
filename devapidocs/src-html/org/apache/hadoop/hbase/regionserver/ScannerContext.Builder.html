022import org.apache.hadoop.hbase.HBaseInterfaceAudience;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.classification.InterfaceStability;

027/**
028 * ScannerContext instances encapsulate limit tracking AND progress towards those limits during
029 * invocations of {@link InternalScanner#next(java.util.List)} and
030 * {@link RegionScanner#next(java.util.List)}.
031 * <p>
032 * A ScannerContext instance should be updated periodically throughout execution whenever progress
033 * towards a limit has been made. Each limit can be checked via the appropriate checkLimit method.
034 * <p>
035 * Once a limit has been reached, the scan will stop. The invoker of
036 * {@link InternalScanner#next(java.util.List)} or {@link RegionScanner#next(java.util.List)} can
037 * use the appropriate check*Limit methods to see exactly which limits have been reached.
038 * Alternatively, {@link #checkAnyLimitReached(LimitScope)} is provided to see if ANY limit was
039 * reached
040 * <p>
041 * {@link NoLimitScannerContext#NO_LIMIT} is an immutable static definition that can be used
042 * whenever a {@link ScannerContext} is needed but limits do not need to be enforced.
043 * <p>
044 * NOTE: It is important that this class only ever expose setter methods that can be safely skipped
045 * when limits should be NOT enforced. This is because of the necessary immutability of the class
046 * {@link NoLimitScannerContext}. If a setter cannot be safely skipped, the immutable nature of
047 * {@link NoLimitScannerContext} will lead to incorrect behavior.
048 */
049@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
050@InterfaceStability.Evolving
051public class ScannerContext {
052
053  /**
054   * Two sets of the same fields. One for the limits, another for the progress towards those limits
055   */
056  LimitFields limits;
057  LimitFields progress;
058
059  /**
060   * The state of the scanner after the invocation of {@link InternalScanner#next(java.util.List)}
061   * or {@link RegionScanner#next(java.util.List)}.
062   */
063  NextState scannerState;
064  private static final NextState DEFAULT_STATE = NextState.MORE_VALUES;
065
066  /**
067   * Used as an indication to invocations of {@link InternalScanner#next(java.util.List)} and
068   * {@link RegionScanner#next(java.util.List)} that, if true, the progress tracked within this
069   * {@link ScannerContext} instance should be considered while evaluating the limits. Useful for
070   * enforcing a set of limits across multiple calls (i.e. the limit may not be reached in a single
071   * invocation, but any progress made should be considered in future invocations)
072   * <p>
073   * Defaulting this value to false means that, by default, any tracked progress will be wiped clean
074   * on invocations to {@link InternalScanner#next(java.util.List)} and
075   * {@link RegionScanner#next(java.util.List)} and the call will be treated as though no progress
076   * has been made towards the limits so far.
077   * <p>
078   * This is an important mechanism. Users of Internal/Region scanners expect that they can define
079   * some limits and then repeatedly invoke {@link InternalScanner#next(List)} or
080   * {@link RegionScanner#next(List)} where each invocation respects these limits separately.
081   * <p>
082   * For example: <pre> {@code
083   * ScannerContext context = new ScannerContext.newBuilder().setBatchLimit(5).build();
084   * RegionScanner scanner = ...
085   * List<Cell> results = new ArrayList<Cell>();
086   * while(scanner.next(results, context)) {
087   *   // Do something with a batch of 5 cells
088   * }
089   * }</pre> However, in the case of RPCs, the server wants to be able to define a set of
090   * limits for a particular RPC request and have those limits respected across multiple
091   * invocations. This means that the progress made towards the limits in earlier calls will be
092   * saved and considered in future invocations
093   */
094  boolean keepProgress;
095  private static boolean DEFAULT_KEEP_PROGRESS = false;
096
097  /**
098   * Tracks the relevant server side metrics during scans. null when metrics should not be tracked
099   */
100  final ServerSideScanMetrics metrics;
101
102  ScannerContext(boolean keepProgress, LimitFields limitsToCopy, boolean trackMetrics) {
103    this.limits = new LimitFields();
104    if (limitsToCopy != null) this.limits.copy(limitsToCopy);
105
106    // Progress fields are initialized to 0
107    progress = new LimitFields(0, LimitFields.DEFAULT_SCOPE, 0, LimitFields.DEFAULT_SCOPE, 0);
108
109    this.keepProgress = keepProgress;
110    this.scannerState = DEFAULT_STATE;
111    this.metrics = trackMetrics ? new ServerSideScanMetrics() : null;
112  }
113
114  boolean isTrackingMetrics() {
115    return this.metrics != null;
116  }
117
118  /**
119   * Get the metrics instance. Should only be called after a call to {@link #isTrackingMetrics()}
120   * has been made to confirm that metrics are indeed being tracked.
121   * @return {@link ServerSideScanMetrics} instance that is tracking metrics for this scan
122   */
123  ServerSideScanMetrics getMetrics() {
124    assert isTrackingMetrics();
125    return this.metrics;
126  }
127
128  /**
129   * @return true if the progress tracked so far in this instance will be considered during an
130   *         invocation of {@link InternalScanner#next(java.util.List)} or
131   *         {@link RegionScanner#next(java.util.List)}. false when the progress tracked so far
132   *         should not be considered and should instead be wiped away via {@link #clearProgress()}
133   */
134  boolean getKeepProgress() {
135    return keepProgress;
136  }
137
138  void setKeepProgress(boolean keepProgress) {
139    this.keepProgress = keepProgress;
140  }
141
142  /**
143   * Progress towards the batch limit has been made. Increment internal tracking of batch progress
144   */
145  void incrementBatchProgress(int batch) {
146    int currentBatch = progress.getBatch();
147    progress.setBatch(currentBatch + batch);
148  }
149
150  /**
151   * Progress towards the size limit has been made. Increment internal tracking of size progress
152   */
153  void incrementSizeProgress(long size) {
154    long currentSize = progress.getSize();
155    progress.setSize(currentSize + size);
156  }
157
158  /**
159   * Update the time progress with {@link System#currentTimeMillis()}
160   */
161  void updateTimeProgress() {
162    progress.setTime(System.currentTimeMillis());
163  }
164
165  int getBatchProgress() {
166    return progress.getBatch();
167  }
168
169  long getSizeProgress() {
170    return progress.getSize();
171  }
172
173  long getTimeProgress() {
174    return progress.getTime();
175  }
176
177  void setProgress(int batchProgress, long sizeProgress, long timeProgress) {
178    setBatchProgress(batchProgress);
179    setSizeProgress(sizeProgress);
180    setTimeProgress(timeProgress);
181  }
182
183  void setSizeProgress(long sizeProgress) {
184    progress.setSize(sizeProgress);
185  }
186
187  void setBatchProgress(int batchProgress) {
188    progress.setBatch(batchProgress);
189  }
190
191  void setTimeProgress(long timeProgress) {
192    progress.setTime(timeProgress);
193  }
194
195  /**
196   * Clear away any progress that has been made so far. All progress fields are reset to initial
197   * values
198   */
199  void clearProgress() {
200    progress.setFields(0, LimitFields.DEFAULT_SCOPE, 0, LimitFields.DEFAULT_SCOPE, 0);
201  }
202
203  /**
204   * Note that this is not a typical setter. This setter returns the {@link NextState} that was
205   * passed in so that methods can be invoked against the new state. Furthermore, this pattern
206   * allows the {@link NoLimitScannerContext} to cleanly override this setter and simply return the
207   * new state, thus preserving the immutability of {@link NoLimitScannerContext}
208   * @param state
209   * @return The state that was passed in.
210   */
211  NextState setScannerState(NextState state) {
212    if (!NextState.isValidState(state)) {
213      throw new IllegalArgumentException("Cannot set to invalid state: " + state);
214    }
215
216    this.scannerState = state;
217    return state;
218  }
219
220  /**
221   * @return true when we may have more cells for the current row. This usually because we have
222   *         reached a limit in the middle of a row
223   */
224  boolean mayHaveMoreCellsInRow() {
225    return scannerState == NextState.SIZE_LIMIT_REACHED_MID_ROW ||
226        scannerState == NextState.TIME_LIMIT_REACHED_MID_ROW ||
227        scannerState == NextState.BATCH_LIMIT_REACHED;
228  }
229
230  /**
231   * @param checkerScope
232   * @return true if the batch limit can be enforced in the checker's scope
233   */
234  boolean hasBatchLimit(LimitScope checkerScope) {
235    return limits.canEnforceBatchLimitFromScope(checkerScope) && limits.getBatch() > 0;
236  }
237
238  /**
239   * @param checkerScope
240   * @return true if the size limit can be enforced in the checker's scope
241   */
242  boolean hasSizeLimit(LimitScope checkerScope) {
243    return limits.canEnforceSizeLimitFromScope(checkerScope) && limits.getSize() > 0;
244  }
245
246  /**
247   * @param checkerScope
248   * @return true if the time limit can be enforced in the checker's scope
249   */
250  boolean hasTimeLimit(LimitScope checkerScope) {
251    return limits.canEnforceTimeLimitFromScope(checkerScope) && limits.getTime() > 0;
252  }
253
254  /**
255   * @param checkerScope
256   * @return true if any limit can be enforced within the checker's scope
257   */
258  boolean hasAnyLimit(LimitScope checkerScope) {
259    return hasBatchLimit(checkerScope) || hasSizeLimit(checkerScope) || hasTimeLimit(checkerScope);
260  }
261
262  /**
263   * @param scope The scope in which the size limit will be enforced
264   */
265  void setSizeLimitScope(LimitScope scope) {
266    limits.setSizeScope(scope);
267  }
268
269  /**
270   * @param scope The scope in which the time limit will be enforced
271   */
272  void setTimeLimitScope(LimitScope scope) {
273    limits.setTimeScope(scope);
274  }
275
276  int getBatchLimit() {
277    return limits.getBatch();
278  }
279
280  long getSizeLimit() {
281    return limits.getSize();
282  }
283
284  long getTimeLimit() {
285    return limits.getTime();
286  }
287
288  /**
289   * @param checkerScope The scope that the limit is being checked from
290   * @return true when the limit is enforceable from the checker's scope and it has been reached
291   */
292  boolean checkBatchLimit(LimitScope checkerScope) {
293    return hasBatchLimit(checkerScope) && progress.getBatch() >= limits.getBatch();
294  }
295
296  /**
297   * @param checkerScope The scope that the limit is being checked from
298   * @return true when the limit is enforceable from the checker's scope and it has been reached
299   */
300  boolean checkSizeLimit(LimitScope checkerScope) {
301    return hasSizeLimit(checkerScope) && progress.getSize() >= limits.getSize();
302  }
303
304  /**
305   * @param checkerScope The scope that the limit is being checked from. The time limit is always
306   *          checked against {@link System#currentTimeMillis()}
307   * @return true when the limit is enforceable from the checker's scope and it has been reached
308   */
309  boolean checkTimeLimit(LimitScope checkerScope) {
310    return hasTimeLimit(checkerScope) && progress.getTime() >= limits.getTime();
311  }
312
313  /**
314   * @param checkerScope The scope that the limits are being checked from
315   * @return true when some limit is enforceable from the checker's scope and it has been reached
316   */
317  boolean checkAnyLimitReached(LimitScope checkerScope) {
318    return checkSizeLimit(checkerScope) || checkBatchLimit(checkerScope)
319        || checkTimeLimit(checkerScope);
320  }

395  /**
396   * The possible states a scanner may be in following a call to {@link InternalScanner#next(List)}
397   */
398  public enum NextState {
399    MORE_VALUES(true, false),
400    NO_MORE_VALUES(false, false),
401    SIZE_LIMIT_REACHED(true, true),
402
403    /**
404     * Special case of size limit reached to indicate that the size limit was reached in the middle
405     * of a row and thus a partial results was formed
406     */
407    SIZE_LIMIT_REACHED_MID_ROW(true, true),
408    TIME_LIMIT_REACHED(true, true),
409
410    /**
411     * Special case of time limit reached to indicate that the time limit was reached in the middle
412     * of a row and thus a partial results was formed
413     */
414    TIME_LIMIT_REACHED_MID_ROW(true, true),
415    BATCH_LIMIT_REACHED(true, true);
416
417    private boolean moreValues;
418    private boolean limitReached;
419
420    private NextState(boolean moreValues, boolean limitReached) {
421      this.moreValues = moreValues;
422      this.limitReached = limitReached;
423    }
424
425    /**
426     * @return true when the state indicates that more values may follow those that have been
427     *         returned
428     */
429    public boolean hasMoreValues() {
430      return this.moreValues;
431    }
432
433    /**
434     * @return true when the state indicates that a limit has been reached and scan should stop
435     */
436    public boolean limitReached() {
437      return this.limitReached;
438    }
439
440    public static boolean isValidState(NextState state) {
441      return state != null;
442    }
443
444    public static boolean hasMoreValues(NextState state) {
445      return isValidState(state) && state.hasMoreValues();
446    }
447  }

449  /**
450   * The various scopes where a limit can be enforced. Used to differentiate when a limit should be
451   * enforced or not.
452   */
453  public enum LimitScope {
454    /**
455     * Enforcing a limit between rows means that the limit will not be considered until all the
456     * cells for a particular row have been retrieved
457     */
458    BETWEEN_ROWS(0),
459
460    /**
461     * Enforcing a limit between cells means that the limit will be considered after each full cell
462     * has been retrieved
463     */
464    BETWEEN_CELLS(1);
465
466    /**
467     * When enforcing a limit, we must check that the scope is appropriate for enforcement.
468     * <p>
469     * To communicate this concept, each scope has a depth. A limit will be enforced if the depth of
470     * the checker's scope is less than or equal to the limit's scope. This means that when checking
471     * limits, the checker must know their own scope (i.e. are they checking the limits between
472     * rows, between cells, etc...)
473     */
474    int depth;
475
476    LimitScope(int depth) {
477      this.depth = depth;
478    }
479
480    int depth() {
481      return depth;
482    }
483
484    /**
485     * @param checkerScope The scope in which the limit is being checked
486     * @return true when the checker is in a scope that indicates the limit can be enforced. Limits
487     *         can be enforced from "higher or equal" scopes (i.e. the checker's scope is at a
488     *         lesser depth than the limit)
489     */
490    boolean canEnforceLimitFromScope(LimitScope checkerScope) {
491      return checkerScope != null && checkerScope.depth() <= depth;
492    }
493  }

495  /**
496   * The different fields that can be used as limits in calls to
497   * {@link InternalScanner#next(java.util.List)} and {@link RegionScanner#next(java.util.List)}
498   */
499  private static class LimitFields {
500    /**
501     * Default values of the limit fields. Defined such that if a field does NOT change from its
502     * default, it will not be enforced
503     */
504    private static int DEFAULT_BATCH = -1;
505    private static long DEFAULT_SIZE = -1L;
506    private static long DEFAULT_TIME = -1L;
507
508    /**
509     * Default scope that is assigned to a limit if a scope is not specified.
510     */
511    private static final LimitScope DEFAULT_SCOPE = LimitScope.BETWEEN_ROWS;
512
513    // The batch limit will always be enforced between cells, thus, there isn't a field to hold the
514    // batch scope
515    int batch = DEFAULT_BATCH;
516
517    LimitScope sizeScope = DEFAULT_SCOPE;
518    long size = DEFAULT_SIZE;
519
520    LimitScope timeScope = DEFAULT_SCOPE;
521    long time = DEFAULT_TIME;
522
523    /**
524     * Fields keep their default values.
525     */
526    LimitFields() {
527    }
528
529    LimitFields(int batch, LimitScope sizeScope, long size, LimitScope timeScope, long time) {
530      setFields(batch, sizeScope, size, timeScope, time);
531    }
532
533    void copy(LimitFields limitsToCopy) {
534      if (limitsToCopy != null) {
535        setFields(limitsToCopy.getBatch(), limitsToCopy.getSizeScope(), limitsToCopy.getSize(),
536          limitsToCopy.getTimeScope(), limitsToCopy.getTime());
537      }
538    }
539
540    /**
541     * Set all fields together.
542     * @param batch
543     * @param sizeScope
544     * @param size
545     */
546    void setFields(int batch, LimitScope sizeScope, long size, LimitScope timeScope, long time) {
547      setBatch(batch);
548      setSizeScope(sizeScope);
549      setSize(size);
550      setTimeScope(timeScope);
551      setTime(time);
552    }
553
554    int getBatch() {
555      return this.batch;
556    }
557
558    void setBatch(int batch) {
559      this.batch = batch;
560    }
561
562    /**
563     * @param checkerScope
564     * @return true when the limit can be enforced from the scope of the checker
565     */
566    boolean canEnforceBatchLimitFromScope(LimitScope checkerScope) {
567      return LimitScope.BETWEEN_CELLS.canEnforceLimitFromScope(checkerScope);
568    }
569
570    long getSize() {
571      return this.size;
572    }
573
574    void setSize(long size) {
575      this.size = size;
576    }
577
578    /**
579     * @return {@link LimitScope} indicating scope in which the size limit is enforced
580     */
581    LimitScope getSizeScope() {
582      return this.sizeScope;
583    }
584
585    /**
586     * Change the scope in which the size limit is enforced
587     */
588    void setSizeScope(LimitScope scope) {
589      this.sizeScope = scope;
590    }
591
592    /**
593     * @param checkerScope
594     * @return true when the limit can be enforced from the scope of the checker
595     */
596    boolean canEnforceSizeLimitFromScope(LimitScope checkerScope) {
597      return this.sizeScope.canEnforceLimitFromScope(checkerScope);
598    }
599
600    long getTime() {
601      return this.time;
602    }
603
604    void setTime(long time) {
605      this.time = time;
606    }
607
608    /**
609     * @return {@link LimitScope} indicating scope in which the time limit is enforced
610     */
611    LimitScope getTimeScope() {
612      return this.timeScope;
613    }
614
615    /**
616     * Change the scope in which the time limit is enforced
617     */
618    void setTimeScope(LimitScope scope) {
619      this.timeScope = scope;
620    }
621
622    /**
623     * @param checkerScope
624     * @return true when the limit can be enforced from the scope of the checker
625     */
626    boolean canEnforceTimeLimitFromScope(LimitScope checkerScope) {
627      return this.timeScope.canEnforceLimitFromScope(checkerScope);
628    }

644      sb.append(", time:");

648      sb.append(timeScope);
