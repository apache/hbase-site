106  /**
107   * Create a view to the on-disk region
108   * @param conf       the {@link Configuration} to use
109   * @param fs         {@link FileSystem} that contains the region
110   * @param tableDir   {@link Path} to where the table is being stored
111   * @param regionInfo {@link RegionInfo} for region

113  HRegionFileSystem(final Configuration conf, final FileSystem fs, final Path tableDir,
114    final RegionInfo regionInfo) {
115    this.fs = fs;
116    this.conf = conf;
117    this.tableDir = Objects.requireNonNull(tableDir, "tableDir is null");
118    this.regionInfo = Objects.requireNonNull(regionInfo, "regionInfo is null");
119    this.regionInfoForFs = ServerRegionReplicaUtil.getRegionInfoForFs(regionInfo);
120    this.regionDir = FSUtils.getRegionDirFromTableDir(tableDir, regionInfo);
121    this.hdfsClientRetriesNumber =
122      conf.getInt("hdfs.client.retries.number", DEFAULT_HDFS_CLIENT_RETRIES_NUMBER);
123    this.baseSleepBeforeRetries =
124      conf.getInt("hdfs.client.sleep.before.retries", DEFAULT_BASE_SLEEP_BEFORE_RETRIES);
125  }

254  /**
255   * Returns the store files available for the family. This methods performs the filtering based on
256   * the valid store files.
257   * @param familyName Column Family Name
258   * @return a set of {@link StoreFileInfo} for the specified family.
259   */
260  public List<StoreFileInfo> getStoreFiles(final String familyName, final boolean validate)
261    throws IOException {
262    Path familyDir = getStoreDir(familyName);
263    FileStatus[] files = CommonFSUtils.listStatus(this.fs, familyDir);
264    if (files == null) {
265      if (LOG.isTraceEnabled()) {
266        LOG.trace("No StoreFiles for: " + familyDir);
267      }
268      return null;
269    }
270
271    ArrayList<StoreFileInfo> storeFiles = new ArrayList<>(files.length);
272    for (FileStatus status : files) {
273      if (validate && !StoreFileInfo.isValid(status)) {
274        // recovered.hfiles directory is expected inside CF path when hbase.wal.split.to.hfile to
275        // true, refer HBASE-23740
276        if (!HConstants.RECOVERED_HFILES_DIR.equals(status.getPath().getName())) {
277          LOG.warn("Invalid StoreFile: {}", status.getPath());
278        }
279        continue;
280      }
281      StoreFileInfo info = ServerRegionReplicaUtil.getStoreFileInfo(conf, fs, regionInfo,
282        regionInfoForFs, familyName, status.getPath());
283      storeFiles.add(info);
284
285    }
286    return storeFiles;
287  }

289  /**
290   * Returns the store files' LocatedFileStatus which available for the family. This methods
291   * performs the filtering based on the valid store files.
292   * @param familyName Column Family Name
293   * @return a list of store files' LocatedFileStatus for the specified family.
294   */
295  public static List<LocatedFileStatus> getStoreFilesLocatedStatus(final HRegionFileSystem regionfs,
296    final String familyName, final boolean validate) throws IOException {
297    Path familyDir = regionfs.getStoreDir(familyName);
298    List<LocatedFileStatus> locatedFileStatuses =
299      CommonFSUtils.listLocatedStatus(regionfs.getFileSystem(), familyDir);
300    if (locatedFileStatuses == null) {
301      if (LOG.isTraceEnabled()) {
302        LOG.trace("No StoreFiles for: " + familyDir);
303      }
304      return null;
305    }
306
307    List<LocatedFileStatus> validStoreFiles = Lists.newArrayList();
308    for (LocatedFileStatus status : locatedFileStatuses) {
309      if (validate && !StoreFileInfo.isValid(status)) {
310        // recovered.hfiles directory is expected inside CF path when hbase.wal.split.to.hfile to
311        // true, refer HBASE-23740
312        if (!HConstants.RECOVERED_HFILES_DIR.equals(status.getPath().getName())) {
313          LOG.warn("Invalid StoreFile: {}", status.getPath());
314        }
315      } else {
316        validStoreFiles.add(status);
317      }
318    }
319    return validStoreFiles;
320  }

346  /**
347   * Returns true if the specified family has reference files
348   * @param familyName Column Family Name
349   * @return true if family contains reference files n
350   */
351  public boolean hasReferences(final String familyName) throws IOException {
352    Path storeDir = getStoreDir(familyName);
353    FileStatus[] files = CommonFSUtils.listStatus(fs, storeDir);
354    if (files != null) {
355      for (FileStatus stat : files) {
356        if (stat.isDirectory()) {
357          continue;
358        }
359        if (StoreFileInfo.isReference(stat.getPath())) {
360          LOG.trace("Reference {}", stat.getPath());
361          return true;
362        }
363      }
364    }
365    return false;
366  }

461  /**
462   * Generate the filename in the main family store directory for moving the file from a build/temp
463   * location.
464   * @param familyName      Family that will gain the file
465   * @param buildPath       {@link Path} to the file to commit.
466   * @param seqNum          Sequence Number to append to the file name (less then 0 if no sequence
467   *                        number)
468   * @param generateNewName False if you want to keep the buildPath name
469   * @return The new {@link Path} of the to be committed file n
470   */
471  private Path preCommitStoreFile(final String familyName, final Path buildPath, final long seqNum,
472    final boolean generateNewName) throws IOException {
473    Path storeDir = getStoreDir(familyName);
474    if (!fs.exists(storeDir) && !createDir(storeDir))
475      throw new IOException("Failed creating " + storeDir);
476
477    String name = buildPath.getName();
478    if (generateNewName) {
479      name = generateUniqueName((seqNum < 0) ? null : "_SeqId_" + seqNum + "_");
480    }
481    Path dstPath = new Path(storeDir, name);
482    if (!fs.exists(buildPath)) {
483      throw new FileNotFoundException(buildPath.toString());
484    }
485    if (LOG.isDebugEnabled()) {
486      LOG.debug("Committing " + buildPath + " as " + dstPath);
487    }
488    return dstPath;
489  }

491  /*
492   * Moves file from staging dir to region dir
493   * @param buildPath {@link Path} to the file to commit.
494   * @param dstPath {@link Path} to the file under region dir
495   * @return The {@link Path} of the committed file n
496   */
497  Path commitStoreFile(final Path buildPath, Path dstPath) throws IOException {
498    // rename is not necessary in case of direct-insert stores
499    if (buildPath.equals(dstPath)) {
500      return dstPath;
501    }
502    // buildPath exists, therefore not doing an exists() check.
503    if (!rename(buildPath, dstPath)) {
504      throw new IOException("Failed rename of " + buildPath + " to " + dstPath);
505    }
506    return dstPath;
507  }

532  /**
533   * Bulk load: Add a specified store file to the specified family. If the source file is on the
534   * same different file-system is moved from the source location to the destination location,
535   * otherwise is copied over.
536   * @param familyName Family that will gain the file
537   * @param srcPath    {@link Path} to the file to import
538   * @param seqNum     Bulk Load sequence number
539   * @return The destination {@link Path} of the bulk loaded file n
540   */
541  Pair<Path, Path> bulkLoadStoreFile(final String familyName, Path srcPath, long seqNum)
542    throws IOException {
543    // Copy the file if it's on another filesystem
544    FileSystem srcFs = srcPath.getFileSystem(conf);
545    srcPath = srcFs.resolvePath(srcPath);
546    FileSystem realSrcFs = srcPath.getFileSystem(conf);
547    FileSystem desFs = fs instanceof HFileSystem ? ((HFileSystem) fs).getBackingFs() : fs;
548
549    // We can't compare FileSystem instances as equals() includes UGI instance
550    // as part of the comparison and won't work when doing SecureBulkLoad
551    // TODO deal with viewFS
552    if (!FSUtils.isSameHdfs(conf, realSrcFs, desFs)) {
553      LOG.info("Bulk-load file " + srcPath + " is on different filesystem than "
554        + "the destination store. Copying file over to destination filesystem.");
555      Path tmpPath = createTempName();
556      FileUtil.copy(realSrcFs, srcPath, fs, tmpPath, false, conf);
557      LOG.info("Copied " + srcPath + " to temporary path on destination filesystem: " + tmpPath);
558      srcPath = tmpPath;
559    }
560
561    return new Pair<>(srcPath, preCommitStoreFile(familyName, srcPath, seqNum, true));
562  }

583  /**
584   * Commit a daughter region, moving it from the split temporary directory to the proper location
585   * in the filesystem.
586   * @param regionInfo daughter {@link org.apache.hadoop.hbase.client.RegionInfo}
587   */
588  public Path commitDaughterRegion(final RegionInfo regionInfo, List<Path> allRegionFiles,
589    MasterProcedureEnv env) throws IOException {
590    Path regionDir = this.getSplitsDir(regionInfo);
591    if (fs.exists(regionDir)) {
592      // Write HRI to a file in case we need to recover hbase:meta
593      Path regionInfoFile = new Path(regionDir, REGION_INFO_FILE);
594      byte[] regionInfoContent = getRegionInfoFileContent(regionInfo);
595      writeRegionInfoFileContent(conf, fs, regionInfoFile, regionInfoContent);
596      HRegionFileSystem regionFs = HRegionFileSystem.openRegionFromFileSystem(
597        env.getMasterConfiguration(), fs, getTableDir(), regionInfo, false);
598      insertRegionFilesIntoStoreTracker(allRegionFiles, env, regionFs);
599    }
600    return regionDir;
601  }

603  private void insertRegionFilesIntoStoreTracker(List<Path> allFiles, MasterProcedureEnv env,
604    HRegionFileSystem regionFs) throws IOException {
605    TableDescriptor tblDesc =
606      env.getMasterServices().getTableDescriptors().get(regionInfo.getTable());
607    // we need to map trackers per store
608    Map<String, StoreFileTracker> trackerMap = new HashMap<>();
609    // we need to map store files per store
610    Map<String, List<StoreFileInfo>> fileInfoMap = new HashMap<>();
611    for (Path file : allFiles) {
612      String familyName = file.getParent().getName();
613      trackerMap.computeIfAbsent(familyName, t -> StoreFileTrackerFactory.create(conf, tblDesc,
614        tblDesc.getColumnFamily(Bytes.toBytes(familyName)), regionFs));
615      fileInfoMap.computeIfAbsent(familyName, l -> new ArrayList<>());
616      List<StoreFileInfo> infos = fileInfoMap.get(familyName);
617      infos.add(new StoreFileInfo(conf, fs, file, true));
618    }
619    for (Map.Entry<String, StoreFileTracker> entry : trackerMap.entrySet()) {
620      entry.getValue().add(fileInfoMap.get(entry.getKey()));

648  /**
649   * Write out a split reference. Package local so it doesnt leak out of regionserver.
650   * @param hri         {@link RegionInfo} of the destination
651   * @param familyName  Column Family Name
652   * @param f           File to split.
653   * @param splitRow    Split Row
654   * @param top         True if we are referring to the top half of the hfile.
655   * @param splitPolicy A split policy instance; be careful! May not be full populated; e.g. if this
656   *                    method is invoked on the Master side, then the RegionSplitPolicy will NOT
657   *                    have a reference to a Region.
658   * @return Path to created reference.
659   */
660  public Path splitStoreFile(RegionInfo hri, String familyName, HStoreFile f, byte[] splitRow,
661    boolean top, RegionSplitPolicy splitPolicy) throws IOException {
662    Path splitDir = new Path(getSplitsDir(hri), familyName);
663    // Add the referred-to regions name as a dot separated suffix.
664    // See REF_NAME_REGEX regex above. The referred-to regions name is
665    // up in the path of the passed in <code>f</code> -- parentdir is family,
666    // then the directory above is the region name.
667    String parentRegionName = regionInfoForFs.getEncodedName();
668    // Write reference with same file id only with the other region name as
669    // suffix and into the new region location (under same family).
670    Path p = new Path(splitDir, f.getPath().getName() + "." + parentRegionName);
671    if (fs.exists(p)) {
672      LOG.warn("Found an already existing split file for {}. Assuming this is a recovery.", p);
673      return p;
674    }
675    boolean createLinkFile = false;
676    if (splitPolicy == null || !splitPolicy.skipStoreFileRangeCheck(familyName)) {
677      // Check whether the split row lies in the range of the store file
678      // If it is outside the range, return directly.
679      f.initReader();
680      try {
681        Cell splitKey = PrivateCellUtil.createFirstOnRow(splitRow);
682        Optional<Cell> lastKey = f.getLastKey();
683        Optional<Cell> firstKey = f.getFirstKey();
684        if (top) {
685          // check if larger than last key.
686          // If lastKey is null means storefile is empty.
687          if (!lastKey.isPresent()) {
688            return null;
689          }
690          if (f.getComparator().compare(splitKey, lastKey.get()) > 0) {
691            return null;
692          }
693          if (firstKey.isPresent() && f.getComparator().compare(splitKey, firstKey.get()) <= 0) {
694            LOG.debug("Will create HFileLink file for {}, top=true", f.getPath());
695            createLinkFile = true;
696          }
697        } else {
698          // check if smaller than first key
699          // If firstKey is null means storefile is empty.
700          if (!firstKey.isPresent()) {
701            return null;
702          }
703          if (f.getComparator().compare(splitKey, firstKey.get()) < 0) {
704            return null;
705          }
706          if (lastKey.isPresent() && f.getComparator().compare(splitKey, lastKey.get()) >= 0) {
707            LOG.debug("Will create HFileLink file for {}, top=false", f.getPath());
708            createLinkFile = true;
709          }
710        }
711      } finally {
712        f.closeStoreFile(f.getCacheConf() != null ? f.getCacheConf().shouldEvictOnClose() : true);
713      }
714    }
715    if (createLinkFile) {
716      // create HFileLink file instead of Reference file for child
717      String hfileName = f.getPath().getName();
718      TableName linkedTable = regionInfoForFs.getTable();
719      String linkedRegion = regionInfoForFs.getEncodedName();
720      try {
721        if (HFileLink.isHFileLink(hfileName)) {
722          Matcher m = LINK_NAME_PATTERN.matcher(hfileName);
723          if (!m.matches()) {
724            throw new IllegalArgumentException(hfileName + " is not a valid HFileLink name!");
725          }
726          linkedTable = TableName.valueOf(m.group(1), m.group(2));
727          linkedRegion = m.group(3);
728          hfileName = m.group(4);
729        }
730        // must create back reference here
731        HFileLink.create(conf, fs, splitDir, familyName, hri.getTable().getNameAsString(),
732          hri.getEncodedName(), linkedTable, linkedRegion, hfileName, true);
733        Path path =
734          new Path(splitDir, HFileLink.createHFileLinkName(linkedTable, linkedRegion, hfileName));
735        LOG.info("Created linkFile:" + path.toString() + " for child: " + hri.getEncodedName()
736          + ", parent: " + regionInfoForFs.getEncodedName());
737        return path;
738      } catch (IOException e) {
739        // if create HFileLink file failed, then just skip the error and create Reference file
740        LOG.error("Create link file for " + hfileName + " for child " + hri.getEncodedName()
741          + "failed, will create Reference file", e);
742      }
743    }
744    // A reference to the bottom half of the hsf store file.
745    Reference r =
746      top ? Reference.createTopReference(splitRow) : Reference.createBottomReference(splitRow);
747    return r.write(fs, p);
748  }

862  /**
863   * Write out an info file under the stored region directory. Useful recovering mangled regions. If
864   * the regionInfo already exists on-disk, then we fast exit.
865   */
866  void checkRegionInfoOnFilesystem() throws IOException {
867    // Compose the content of the file so we can compare to length in filesystem. If not same,
868    // rewrite it (it may have been written in the old format using Writables instead of pb). The
869    // pb version is much shorter -- we write now w/o the toString version -- so checking length
870    // only should be sufficient. I don't want to read the file every time to check if it pb
871    // serialized.
872    byte[] content = getRegionInfoFileContent(regionInfoForFs);
873
874    // Verify if the region directory exists before opening a region. We need to do this since if
875    // the region directory doesn't exist we will re-create the region directory and a new HRI
876    // when HRegion.openHRegion() is called.
877    try {
878      FileStatus status = fs.getFileStatus(getRegionDir());
879    } catch (FileNotFoundException e) {
880      LOG.warn(getRegionDir() + " doesn't exist for region: " + regionInfoForFs.getEncodedName()
881        + " on table " + regionInfo.getTable());
882    }
883
884    try {
885      Path regionInfoFile = new Path(getRegionDir(), REGION_INFO_FILE);
886      FileStatus status = fs.getFileStatus(regionInfoFile);
887      if (status != null && status.getLen() == content.length) {
888        // Then assume the content good and move on.
889        // NOTE: that the length is not sufficient to define the the content matches.
890        return;
891      }
892
893      LOG.info("Rewriting .regioninfo file at: " + regionInfoFile);
894      if (!fs.delete(regionInfoFile, false)) {
895        throw new IOException("Unable to remove existing " + regionInfoFile);
896      }
897    } catch (FileNotFoundException e) {
898      LOG.warn(REGION_INFO_FILE + " file not found for region: " + regionInfoForFs.getEncodedName()
899        + " on table " + regionInfo.getTable());
900    }
901
902    // Write HRI to a file in case we need to recover hbase:meta
903    writeRegionInfoOnFilesystem(content, true);
904  }

915  /**
916   * Write out an info file under the region directory. Useful recovering mangled regions.
917   * @param regionInfoContent serialized version of the {@link RegionInfo}
918   * @param useTempDir        indicate whether or not using the region .tmp dir for a safer file
919   *                          creation.
920   */
921  private void writeRegionInfoOnFilesystem(final byte[] regionInfoContent, final boolean useTempDir)
922    throws IOException {
923    Path regionInfoFile = new Path(getRegionDir(), REGION_INFO_FILE);
924    if (useTempDir) {
925      // Create in tmpDir and then move into place in case we crash after
926      // create but before close. If we don't successfully close the file,
927      // subsequent region reopens will fail the below because create is
928      // registered in NN.
929
930      // And then create the file
931      Path tmpPath = new Path(getTempDir(), REGION_INFO_FILE);
932
933      // If datanode crashes or if the RS goes down just before the close is called while trying to
934      // close the created regioninfo file in the .tmp directory then on next
935      // creation we will be getting AlreadyCreatedException.
936      // Hence delete and create the file if exists.
937      if (CommonFSUtils.isExists(fs, tmpPath)) {
938        CommonFSUtils.delete(fs, tmpPath, true);
939      }
940
941      // Write HRI to a file in case we need to recover hbase:meta
942      writeRegionInfoFileContent(conf, fs, tmpPath, regionInfoContent);
943
944      // Move the created file to the original path
945      if (fs.exists(tmpPath) && !rename(tmpPath, regionInfoFile)) {
946        throw new IOException("Unable to rename " + tmpPath + " to " + regionInfoFile);
947      }
948    } else {
949      // Write HRI to a file in case we need to recover hbase:meta
950      writeRegionInfoFileContent(conf, fs, regionInfoFile, regionInfoContent);
951    }
952  }

1058  /**
1059   * Creates a directory. Assumes the user has already checked for this directory existence. n
1060   * * @return the result of fs.mkdirs(). In case underlying fs throws an IOException, it checks
1061   * whether the directory exists or not, and returns true if it exists. n
1062   */
1063  boolean createDir(Path dir) throws IOException {
1064    int i = 0;
1065    IOException lastIOE = null;
1066    do {
1067      try {
1068        return mkdirs(fs, conf, dir);
1069      } catch (IOException ioe) {
1070        lastIOE = ioe;
1071        if (fs.exists(dir)) return true; // directory is present
1072        try {
1073          sleepBeforeRetry("Create Directory", i + 1);
1074        } catch (InterruptedException e) {
1075          throw (InterruptedIOException) new InterruptedIOException().initCause(e);
1076        }
1077      }
1078    } while (++i <= hdfsClientRetriesNumber);
1079    throw new IOException("Exception in createDir", lastIOE);
1080  }

1086  boolean rename(Path srcpath, Path dstPath) throws IOException {

1090      try {
1091        return fs.rename(srcpath, dstPath);

1093        lastIOE = ioe;
1094        if (!fs.exists(srcpath) && fs.exists(dstPath)) return true; // successful move
1095        // dir is not there, retry after some time.
1096        try {
1097          sleepBeforeRetry("Rename Directory", i + 1);
1098        } catch (InterruptedException e) {
1099          throw (InterruptedIOException) new InterruptedIOException().initCause(e);
1100        }
1101      }
1102    } while (++i <= hdfsClientRetriesNumber);
1103
1104    throw new IOException("Exception in rename", lastIOE);
1105  }

1139  /**
1140   * Creates a directory for a filesystem and configuration object. Assumes the user has already
1141   * checked for this directory existence. nnn * @return the result of fs.mkdirs(). In case
1142   * underlying fs throws an IOException, it checks whether the directory exists or not, and returns
1143   * true if it exists. n
1144   */
1145  private static boolean createDirOnFileSystem(FileSystem fs, Configuration conf, Path dir)
1146    throws IOException {
1147    int i = 0;
1148    IOException lastIOE = null;
1149    int hdfsClientRetriesNumber =
1150      conf.getInt("hdfs.client.retries.number", DEFAULT_HDFS_CLIENT_RETRIES_NUMBER);
1151    int baseSleepBeforeRetries =
1152      conf.getInt("hdfs.client.sleep.before.retries", DEFAULT_BASE_SLEEP_BEFORE_RETRIES);
1153    do {
1154      try {
1155        return fs.mkdirs(dir);
1156      } catch (IOException ioe) {
1157        lastIOE = ioe;
1158        if (fs.exists(dir)) return true; // directory is present
1159        try {
1160          sleepBeforeRetry("Create Directory", i + 1, baseSleepBeforeRetries,
1161            hdfsClientRetriesNumber);
1162        } catch (InterruptedException e) {
1163          throw (InterruptedIOException) new InterruptedIOException().initCause(e);
1164        }
1165      }
1166    } while (++i <= hdfsClientRetriesNumber);
1167
1168    throw new IOException("Exception in createDir", lastIOE);
1169  }

1171  /**
1172   * sleeping logic for static methods; handles the interrupt exception. Keeping a static version
1173   * for this to avoid re-looking for the integer values.
1174   */
1175  private static void sleepBeforeRetry(String msg, int sleepMultiplier, int baseSleepBeforeRetries,
1176    int hdfsClientRetriesNumber) throws InterruptedException {
1177    if (sleepMultiplier > hdfsClientRetriesNumber) {
1178      if (LOG.isDebugEnabled()) {
1179        LOG.debug(msg + ", retries exhausted");
1180      }
1181      return;
1182    }
1183    if (LOG.isDebugEnabled()) {
1184      LOG.debug(msg + ", sleeping " + baseSleepBeforeRetries + " times " + sleepMultiplier);
1185    }
1186    Thread.sleep((long) baseSleepBeforeRetries * sleepMultiplier);
1187  }
