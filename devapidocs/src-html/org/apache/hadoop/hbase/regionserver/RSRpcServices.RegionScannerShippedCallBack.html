463  /**
464   * Instantiated as a scanner lease. If the lease times out, the scanner is
465   * closed
466   */
467  private class ScannerListener implements LeaseListener {
468    private final String scannerName;
469
470    ScannerListener(final String n) {
471      this.scannerName = n;
472    }
473
474    @Override
475    public void leaseExpired() {
476      RegionScannerHolder rsh = scanners.remove(this.scannerName);
477      if (rsh != null) {
478        RegionScanner s = rsh.s;
479        LOG.info("Scanner " + this.scannerName + " lease expired on region "
480          + s.getRegionInfo().getRegionNameAsString());
481        HRegion region = null;
482        try {
483          region = regionServer.getRegion(s.getRegionInfo().getRegionName());
484          if (region != null && region.getCoprocessorHost() != null) {
485            region.getCoprocessorHost().preScannerClose(s);
486          }
487        } catch (IOException e) {
488          LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
489        } finally {
490          try {
491            s.close();
492            if (region != null && region.getCoprocessorHost() != null) {
493              region.getCoprocessorHost().postScannerClose(s);
494            }
495          } catch (IOException e) {
496            LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
497          }
498        }
499      } else {
500        LOG.warn("Scanner " + this.scannerName + " lease expired, but no related" +
501          " scanner found, hence no chance to close that related scanner!");
502      }
503    }
504  }

606  /**
607   * Mutate a list of rows atomically.
608   * @param cellScanner if non-null, the mutation data -- the Cell content.
609   */
610  private boolean checkAndRowMutate(final HRegion region, final List<ClientProtos.Action> actions,
611    final CellScanner cellScanner, byte[] row, byte[] family, byte[] qualifier, CompareOperator op,
612    ByteArrayComparable comparator, TimeRange timeRange, RegionActionResult.Builder builder,
613    ActivePolicyEnforcement spaceQuotaEnforcement) throws IOException {
614    int countOfCompleteMutation = 0;
615    try {
616      if (!region.getRegionInfo().isMetaRegion()) {
617        regionServer.cacheFlusher.reclaimMemStoreMemory();
618      }
619      RowMutations rm = null;
620      int i = 0;
621      ClientProtos.ResultOrException.Builder resultOrExceptionOrBuilder =
622        ClientProtos.ResultOrException.newBuilder();
623      for (ClientProtos.Action action: actions) {
624        if (action.hasGet()) {
625          throw new DoNotRetryIOException("Atomic put and/or delete only, not a Get=" +
626            action.getGet());
627        }
628        MutationType type = action.getMutation().getMutateType();
629        if (rm == null) {
630          rm = new RowMutations(action.getMutation().getRow().toByteArray(), actions.size());
631        }
632        switch (type) {
633          case PUT:
634            Put put = ProtobufUtil.toPut(action.getMutation(), cellScanner);
635            ++countOfCompleteMutation;
636            checkCellSizeLimit(region, put);
637            spaceQuotaEnforcement.getPolicyEnforcement(region).check(put);
638            rm.add(put);
639            break;
640          case DELETE:
641            Delete del = ProtobufUtil.toDelete(action.getMutation(), cellScanner);
642            ++countOfCompleteMutation;
643            spaceQuotaEnforcement.getPolicyEnforcement(region).check(del);
644            rm.add(del);
645            break;
646          default:
647            throw new DoNotRetryIOException("Atomic put and/or delete only, not " + type.name());
648        }
649        // To unify the response format with doNonAtomicRegionMutation and read through client's
650        // AsyncProcess we have to add an empty result instance per operation
651        resultOrExceptionOrBuilder.clear();
652        resultOrExceptionOrBuilder.setIndex(i++);
653        builder.addResultOrException(
654          resultOrExceptionOrBuilder.build());
655      }
656      return region.checkAndRowMutate(row, family, qualifier, op, comparator, timeRange, rm);
657    } finally {
658      // Currently, the checkAndMutate isn't supported by batch so it won't mess up the cell scanner
659      // even if the malformed cells are not skipped.
660      for (int i = countOfCompleteMutation; i < actions.size(); ++i) {
661        skipCellsForMutation(actions.get(i), cellScanner);
662      }
663    }
664  }

666  /**
667   * Execute an append mutation.
668   *
669   * @return result to return to client if default operation should be
670   * bypassed as indicated by RegionObserver, null otherwise
671   */
672  private Result append(final HRegion region, final OperationQuota quota,
673      final MutationProto mutation, final CellScanner cellScanner, long nonceGroup,
674      ActivePolicyEnforcement spaceQuota)
675      throws IOException {
676    long before = EnvironmentEdgeManager.currentTime();
677    Append append = ProtobufUtil.toAppend(mutation, cellScanner);
678    checkCellSizeLimit(region, append);
679    spaceQuota.getPolicyEnforcement(region).check(append);
680    quota.addMutation(append);
681    Result r = null;
682    if (region.getCoprocessorHost() != null) {
683      r = region.getCoprocessorHost().preAppend(append);
684    }
685    if (r == null) {
686      boolean canProceed = startNonceOperation(mutation, nonceGroup);
687      boolean success = false;
688      try {
689        long nonce = mutation.hasNonce() ? mutation.getNonce() : HConstants.NO_NONCE;
690        if (canProceed) {
691          r = region.append(append, nonceGroup, nonce);
692        } else {
693          // convert duplicate append to get
694          List<Cell> results = region.get(ProtobufUtil.toGet(mutation, cellScanner), false,
695              nonceGroup, nonce);
696          r = Result.create(results);
697        }
698        success = true;
699      } finally {
700        if (canProceed) {
701          endNonceOperation(mutation, nonceGroup, success);
702        }
703      }
704      if (region.getCoprocessorHost() != null) {
705        r = region.getCoprocessorHost().postAppend(append, r);
706      }
707    }
708    if (regionServer.metricsRegionServer != null) {
709      regionServer.metricsRegionServer.updateAppend(
710          region.getTableDescriptor().getTableName(),
711        EnvironmentEdgeManager.currentTime() - before);
712    }
713    return r == null ? Result.EMPTY_RESULT : r;
714  }

768  /**
769   * Run through the regionMutation <code>rm</code> and per Mutation, do the work, and then when
770   * done, add an instance of a {@link ResultOrException} that corresponds to each Mutation.
771   * @param cellsToReturn  Could be null. May be allocated in this method.  This is what this
772   * method returns as a 'result'.
773   * @param closeCallBack the callback to be used with multigets
774   * @param context the current RpcCallContext
775   * @return Return the <code>cellScanner</code> passed
776   */
777  private List<CellScannable> doNonAtomicRegionMutation(final HRegion region,
778      final OperationQuota quota, final RegionAction actions, final CellScanner cellScanner,
779      final RegionActionResult.Builder builder, List<CellScannable> cellsToReturn, long nonceGroup,
780      final RegionScannersCloseCallBack closeCallBack, RpcCallContext context,
781      ActivePolicyEnforcement spaceQuotaEnforcement) {
782    // Gather up CONTIGUOUS Puts and Deletes in this mutations List.  Idea is that rather than do
783    // one at a time, we instead pass them in batch.  Be aware that the corresponding
784    // ResultOrException instance that matches each Put or Delete is then added down in the
785    // doNonAtomicBatchOp call.  We should be staying aligned though the Put and Delete are
786    // deferred/batched
787    List<ClientProtos.Action> mutations = null;
788    long maxQuotaResultSize = Math.min(maxScannerResultSize, quota.getReadAvailable());
789    IOException sizeIOE = null;
790    Object lastBlock = null;
791    ClientProtos.ResultOrException.Builder resultOrExceptionBuilder = ResultOrException.newBuilder();
792    boolean hasResultOrException = false;
793    for (ClientProtos.Action action : actions.getActionList()) {
794      hasResultOrException = false;
795      resultOrExceptionBuilder.clear();
796      try {
797        Result r = null;
798
799        if (context != null
800            && context.isRetryImmediatelySupported()
801            && (context.getResponseCellSize() > maxQuotaResultSize
802              || context.getResponseBlockSize() + context.getResponseExceptionSize()
803              > maxQuotaResultSize)) {
804
805          // We're storing the exception since the exception and reason string won't
806          // change after the response size limit is reached.
807          if (sizeIOE == null ) {
808            // We don't need the stack un-winding do don't throw the exception.
809            // Throwing will kill the JVM's JIT.
810            //
811            // Instead just create the exception and then store it.
812            sizeIOE = new MultiActionResultTooLarge("Max size exceeded"
813                + " CellSize: " + context.getResponseCellSize()
814                + " BlockSize: " + context.getResponseBlockSize());
815
816            // Only report the exception once since there's only one request that
817            // caused the exception. Otherwise this number will dominate the exceptions count.
818            rpcServer.getMetrics().exception(sizeIOE);
819          }
820
821          // Now that there's an exception is known to be created
822          // use it for the response.
823          //
824          // This will create a copy in the builder.
825          NameBytesPair pair = ResponseConverter.buildException(sizeIOE);
826          resultOrExceptionBuilder.setException(pair);
827          context.incrementResponseExceptionSize(pair.getSerializedSize());
828          resultOrExceptionBuilder.setIndex(action.getIndex());
829          builder.addResultOrException(resultOrExceptionBuilder.build());
830          skipCellsForMutation(action, cellScanner);
831          continue;
832        }
833        if (action.hasGet()) {
834          long before = EnvironmentEdgeManager.currentTime();
835          ClientProtos.Get pbGet = action.getGet();
836          // An asynchbase client, https://github.com/OpenTSDB/asynchbase, starts by trying to do
837          // a get closest before. Throwing the UnknownProtocolException signals it that it needs
838          // to switch and do hbase2 protocol (HBase servers do not tell clients what versions
839          // they are; its a problem for non-native clients like asynchbase. HBASE-20225.
840          if (pbGet.hasClosestRowBefore() && pbGet.getClosestRowBefore()) {
841            throw new UnknownProtocolException("Is this a pre-hbase-1.0.0 or asynchbase client? " +
842                "Client is invoking getClosestRowBefore removed in hbase-2.0.0 replaced by " +
843                "reverse Scan.");
844          }
845          try {
846            Get get = ProtobufUtil.toGet(pbGet);
847            if (context != null) {
848              r = get(get, (region), closeCallBack, context);
849            } else {
850              r = region.get(get);
851            }
852          } finally {
853            if (regionServer.metricsRegionServer != null) {
854              regionServer.metricsRegionServer.updateGet(
855                  region.getTableDescriptor().getTableName(),
856                  EnvironmentEdgeManager.currentTime() - before);
857            }
858          }
859        } else if (action.hasServiceCall()) {
860          hasResultOrException = true;
861          com.google.protobuf.Message result =
862            execServiceOnRegion(region, action.getServiceCall());
863          ClientProtos.CoprocessorServiceResult.Builder serviceResultBuilder =
864            ClientProtos.CoprocessorServiceResult.newBuilder();
865          resultOrExceptionBuilder.setServiceResult(
866            serviceResultBuilder.setValue(
867              serviceResultBuilder.getValueBuilder()
868                .setName(result.getClass().getName())
869                // TODO: Copy!!!
870                .setValue(UnsafeByteOperations.unsafeWrap(result.toByteArray()))));
871        } else if (action.hasMutation()) {
872          MutationType type = action.getMutation().getMutateType();
873          if (type != MutationType.PUT && type != MutationType.DELETE && mutations != null &&
874              !mutations.isEmpty()) {
875            // Flush out any Puts or Deletes already collected.
876            doNonAtomicBatchOp(builder, region, quota, mutations, cellScanner,
877              spaceQuotaEnforcement);
878            mutations.clear();
879          }
880          switch (type) {
881            case APPEND:
882              r = append(region, quota, action.getMutation(), cellScanner, nonceGroup,
883                  spaceQuotaEnforcement);
884              break;
885            case INCREMENT:
886              r = increment(region, quota, action.getMutation(), cellScanner, nonceGroup,
887                  spaceQuotaEnforcement);
888              break;
889            case PUT:
890            case DELETE:
891              // Collect the individual mutations and apply in a batch
892              if (mutations == null) {
893                mutations = new ArrayList<>(actions.getActionCount());
894              }
895              mutations.add(action);
896              break;
897            default:
898              throw new DoNotRetryIOException("Unsupported mutate type: " + type.name());
899          }
900        } else {
901          throw new HBaseIOException("Unexpected Action type");
902        }
903        if (r != null) {
904          ClientProtos.Result pbResult = null;
905          if (isClientCellBlockSupport(context)) {
906            pbResult = ProtobufUtil.toResultNoData(r);
907            //  Hard to guess the size here.  Just make a rough guess.
908            if (cellsToReturn == null) {
909              cellsToReturn = new ArrayList<>();
910            }
911            cellsToReturn.add(r);
912          } else {
913            pbResult = ProtobufUtil.toResult(r);
914          }
915          lastBlock = addSize(context, r, lastBlock);
916          hasResultOrException = true;
917          resultOrExceptionBuilder.setResult(pbResult);
918        }
919        // Could get to here and there was no result and no exception.  Presumes we added
920        // a Put or Delete to the collecting Mutations List for adding later.  In this
921        // case the corresponding ResultOrException instance for the Put or Delete will be added
922        // down in the doNonAtomicBatchOp method call rather than up here.
923      } catch (IOException ie) {
924        rpcServer.getMetrics().exception(ie);
925        hasResultOrException = true;
926        NameBytesPair pair = ResponseConverter.buildException(ie);
927        resultOrExceptionBuilder.setException(pair);
928        context.incrementResponseExceptionSize(pair.getSerializedSize());
929      }
930      if (hasResultOrException) {
931        // Propagate index.
932        resultOrExceptionBuilder.setIndex(action.getIndex());
933        builder.addResultOrException(resultOrExceptionBuilder.build());
934      }
935    }
936    // Finish up any outstanding mutations
937    if (!CollectionUtils.isEmpty(mutations)) {
938      doNonAtomicBatchOp(builder, region, quota, mutations, cellScanner, spaceQuotaEnforcement);
939    }
940    return cellsToReturn;
941  }

943  private void checkCellSizeLimit(final HRegion r, final Mutation m) throws IOException {
944    if (r.maxCellSize > 0) {
945      CellScanner cells = m.cellScanner();
946      while (cells.advance()) {
947        int size = PrivateCellUtil.estimatedSerializedSizeOf(cells.current());
948        if (size > r.maxCellSize) {
949          String msg = "Cell with size " + size + " exceeds limit of " + r.maxCellSize + " bytes";
950          if (LOG.isDebugEnabled()) {
951            LOG.debug(msg);
952          }
953          throw new DoNotRetryIOException(msg);
954        }
955      }
956    }

986  /**
987   * Execute a list of Put/Delete mutations.
988   *
989   * @param builder
990   * @param region
991   * @param mutations
992   */
993  private void doBatchOp(final RegionActionResult.Builder builder, final HRegion region,
994      final OperationQuota quota, final List<ClientProtos.Action> mutations,
995      final CellScanner cells, ActivePolicyEnforcement spaceQuotaEnforcement, boolean atomic)
996      throws IOException {
997    Mutation[] mArray = new Mutation[mutations.size()];
998    long before = EnvironmentEdgeManager.currentTime();
999    boolean batchContainsPuts = false, batchContainsDelete = false;
1000    try {
1001      /** HBASE-17924
1002       * mutationActionMap is a map to map the relation between mutations and actions
1003       * since mutation array may have been reoredered.In order to return the right
1004       * result or exception to the corresponding actions, We need to know which action
1005       * is the mutation belong to. We can't sort ClientProtos.Action array, since they
1006       * are bonded to cellscanners.
1007       */
1008      Map<Mutation, ClientProtos.Action> mutationActionMap = new HashMap<>();
1009      int i = 0;
1010      for (ClientProtos.Action action: mutations) {
1011        if (action.hasGet()) {
1012          throw new DoNotRetryIOException("Atomic put and/or delete only, not a Get=" +
1013            action.getGet());
1014        }
1015        MutationProto m = action.getMutation();
1016        Mutation mutation;
1017        if (m.getMutateType() == MutationType.PUT) {
1018          mutation = ProtobufUtil.toPut(m, cells);
1019          batchContainsPuts = true;
1020        } else {
1021          mutation = ProtobufUtil.toDelete(m, cells);
1022          batchContainsDelete = true;
1023        }
1024        mutationActionMap.put(mutation, action);
1025        mArray[i++] = mutation;
1026        checkCellSizeLimit(region, mutation);
1027        // Check if a space quota disallows this mutation
1028        spaceQuotaEnforcement.getPolicyEnforcement(region).check(mutation);
1029        quota.addMutation(mutation);
1030      }
1031
1032      if (!region.getRegionInfo().isMetaRegion()) {
1033        regionServer.cacheFlusher.reclaimMemStoreMemory();
1034      }
1035
1036      // HBASE-17924
1037      // Sort to improve lock efficiency for non-atomic batch of operations. If atomic
1038      // order is preserved as its expected from the client
1039      if (!atomic) {
1040        Arrays.sort(mArray, (v1, v2) -> Row.COMPARATOR.compare(v1, v2));
1041      }
1042
1043      OperationStatus[] codes = region.batchMutate(mArray, atomic, HConstants.NO_NONCE,
1044        HConstants.NO_NONCE);
1045      for (i = 0; i < codes.length; i++) {
1046        Mutation currentMutation = mArray[i];
1047        ClientProtos.Action currentAction = mutationActionMap.get(currentMutation);
1048        int index = currentAction.hasIndex() || !atomic ? currentAction.getIndex() : i;
1049        Exception e = null;
1050        switch (codes[i].getOperationStatusCode()) {
1051          case BAD_FAMILY:
1052            e = new NoSuchColumnFamilyException(codes[i].getExceptionMsg());
1053            builder.addResultOrException(getResultOrException(e, index));
1054            break;
1055
1056          case SANITY_CHECK_FAILURE:
1057            e = new FailedSanityCheckException(codes[i].getExceptionMsg());
1058            builder.addResultOrException(getResultOrException(e, index));
1059            break;
1060
1061          default:
1062            e = new DoNotRetryIOException(codes[i].getExceptionMsg());
1063            builder.addResultOrException(getResultOrException(e, index));
1064            break;
1065
1066          case SUCCESS:
1067            builder.addResultOrException(getResultOrException(
1068              ClientProtos.Result.getDefaultInstance(), index));
1069            break;
1070
1071          case STORE_TOO_BUSY:
1072            e = new RegionTooBusyException(codes[i].getExceptionMsg());
1073            builder.addResultOrException(getResultOrException(e, index));
1074            break;
1075        }
1076      }
1077    } finally {
1078      int processedMutationIndex = 0;
1079      for (Action mutation : mutations) {
1080        // The non-null mArray[i] means the cell scanner has been read.
1081        if (mArray[processedMutationIndex++] == null) {
1082          skipCellsForMutation(mutation, cells);
1083        }
1084      }
1085      updateMutationMetrics(region, before, batchContainsPuts, batchContainsDelete);
1086    }
1087  }

1104  /**
1105   * Execute a list of Put/Delete mutations. The function returns OperationStatus instead of
1106   * constructing MultiResponse to save a possible loop if caller doesn't need MultiResponse.
1107   * @param region
1108   * @param mutations
1109   * @param replaySeqId
1110   * @return an array of OperationStatus which internally contains the OperationStatusCode and the
1111   *         exceptionMessage if any
1112   * @throws IOException
1113   */
1114  private OperationStatus [] doReplayBatchOp(final HRegion region,
1115      final List<WALSplitter.MutationReplay> mutations, long replaySeqId) throws IOException {
1116    long before = EnvironmentEdgeManager.currentTime();
1117    boolean batchContainsPuts = false, batchContainsDelete = false;
1118    try {
1119      for (Iterator<WALSplitter.MutationReplay> it = mutations.iterator(); it.hasNext();) {
1120        WALSplitter.MutationReplay m = it.next();
1121
1122        if (m.type == MutationType.PUT) {
1123          batchContainsPuts = true;
1124        } else {
1125          batchContainsDelete = true;
1126        }
1127
1128        NavigableMap<byte[], List<Cell>> map = m.mutation.getFamilyCellMap();
1129        List<Cell> metaCells = map.get(WALEdit.METAFAMILY);
1130        if (metaCells != null && !metaCells.isEmpty()) {
1131          for (Cell metaCell : metaCells) {
1132            CompactionDescriptor compactionDesc = WALEdit.getCompaction(metaCell);
1133            boolean isDefaultReplica = RegionReplicaUtil.isDefaultReplica(region.getRegionInfo());
1134            HRegion hRegion = region;
1135            if (compactionDesc != null) {
1136              // replay the compaction. Remove the files from stores only if we are the primary
1137              // region replica (thus own the files)
1138              hRegion.replayWALCompactionMarker(compactionDesc, !isDefaultReplica, isDefaultReplica,
1139                replaySeqId);
1140              continue;
1141            }
1142            FlushDescriptor flushDesc = WALEdit.getFlushDescriptor(metaCell);
1143            if (flushDesc != null && !isDefaultReplica) {
1144              hRegion.replayWALFlushMarker(flushDesc, replaySeqId);
1145              continue;
1146            }
1147            RegionEventDescriptor regionEvent = WALEdit.getRegionEventDescriptor(metaCell);
1148            if (regionEvent != null && !isDefaultReplica) {
1149              hRegion.replayWALRegionEventMarker(regionEvent);
1150              continue;
1151            }
1152            BulkLoadDescriptor bulkLoadEvent = WALEdit.getBulkLoadDescriptor(metaCell);
1153            if (bulkLoadEvent != null) {
1154              hRegion.replayWALBulkLoadEventMarker(bulkLoadEvent);
1155              continue;
1156            }
1157          }
1158          it.remove();
1159        }
1160      }
1161      requestCount.increment();
1162      if (!region.getRegionInfo().isMetaRegion()) {
1163        regionServer.cacheFlusher.reclaimMemStoreMemory();
1164      }
1165      return region.batchReplay(mutations.toArray(
1166        new WALSplitter.MutationReplay[mutations.size()]), replaySeqId);
1167    } finally {
1168      updateMutationMetrics(region, before, batchContainsPuts, batchContainsDelete);
1169    }
1170  }

1369  /**
1370   * Method to account for the size of retained cells and retained data blocks.
1371   * @param context rpc call context
1372   * @param r result to add size.
1373   * @param lastBlock last block to check whether we need to add the block size in context.
1374   * @return an object that represents the last referenced block from this response.
1375   */
1376  Object addSize(RpcCallContext context, Result r, Object lastBlock) {
1377    if (context != null && r != null && !r.isEmpty()) {
1378      for (Cell c : r.rawCells()) {
1379        context.incrementResponseCellSize(PrivateCellUtil.estimatedSerializedSizeOf(c));
1380
1381        // Since byte buffers can point all kinds of crazy places it's harder to keep track
1382        // of which blocks are kept alive by what byte buffer.
1383        // So we make a guess.
1384        if (c instanceof ByteBufferExtendedCell) {
1385          ByteBufferExtendedCell bbCell = (ByteBufferExtendedCell) c;
1386          ByteBuffer bb = bbCell.getValueByteBuffer();
1387          if (bb != lastBlock) {
1388            context.incrementResponseBlockSize(bb.capacity());
1389            lastBlock = bb;
1390          }
1391        } else {
1392          // We're using the last block being the same as the current block as
1393          // a proxy for pointing to a new block. This won't be exact.
1394          // If there are multiple gets that bounce back and forth
1395          // Then it's possible that this will over count the size of
1396          // referenced blocks. However it's better to over count and
1397          // use two rpcs than to OOME the regionserver.
1398          byte[] valueArray = c.getValueArray();
1399          if (valueArray != lastBlock) {
1400            context.incrementResponseBlockSize(valueArray.length);
1401            lastBlock = valueArray;
1402          }
1403        }
1404
1405      }
1406    }
1407    return lastBlock;
1408  }

1637   * @param request the request

1640  @Override
1641  @QosPriority(priority = HConstants.ADMIN_QOS)
1642  public CompactRegionResponse compactRegion(final RpcController controller,
1643      final CompactRegionRequest request) throws ServiceException {
1644    try {
1645      checkOpen();
1646      requestCount.increment();
1647      HRegion region = getRegion(request.getRegion());
1648      // Quota support is enabled, the requesting user is not system/super user
1649      // and a quota policy is enforced that disables compactions.
1650      if (QuotaUtil.isQuotaEnabled(getConfiguration()) &&
1651          !Superusers.isSuperUser(RpcServer.getRequestUser().orElse(null)) &&
1652          this.regionServer.getRegionServerSpaceQuotaManager()
1653              .areCompactionsDisabled(region.getTableDescriptor().getTableName())) {
1654        throw new DoNotRetryIOException(
1655            "Compactions on this region are " + "disabled due to a space quota violation.");
1656      }
1657      region.startRegionOperation(Operation.COMPACT_REGION);
1658      LOG.info("Compacting " + region.getRegionInfo().getRegionNameAsString());
1659      boolean major = request.hasMajor() && request.getMajor();
1660      if (request.hasFamily()) {
1661        byte[] family = request.getFamily().toByteArray();
1662        String log = "User-triggered " + (major ? "major " : "") + "compaction for region " +
1663            region.getRegionInfo().getRegionNameAsString() + " and family " +
1664            Bytes.toString(family);
1665        LOG.trace(log);

1955      long serverStartCode = request.getServerStartCode();

1957        throw new ServiceException(new DoNotRetryIOException("This RPC was intended for a " +

2012            + region.getRegionNameAsString() + ", which is already online";

2020

2022          encodedNameBytes, Boolean.TRUE);

2026            // There is a close in progress. This should not happen any more.
2027            String error = "Received OPEN for the region:"
2028              + region.getRegionNameAsString() + ", which we are already trying to CLOSE";

2030            throw new IOException(error);
2031          }
2032          regionServer.regionsInTransitionInRS.put(encodedNameBytes, Boolean.TRUE);
2033        }
2034
2035        if (Boolean.TRUE.equals(previous)) {
2036          // An open is in progress. This is supported, but let's log this.
2037          LOG.info("Receiving OPEN for the region:" +
2038            region.getRegionNameAsString() + ", which we are already trying to OPEN"
2039              + " - ignoring this new request for this region.");
2040        }

2042        // We are opening this region. If it moves back and forth for whatever reason, we don't
2043        // want to keep returning the stale moved record while we are opening/if we close again.
2044        regionServer.removeFromMovedRegions(region.getEncodedName());

2046        if (previous == null || !previous.booleanValue()) {
2047          htd = htds.get(region.getTable());
2048          if (htd == null) {
2049            htd = regionServer.tableDescriptors.get(region.getTable());
2050            htds.put(region.getTable(), htd);
2051          }
2052          if (htd == null) {
2053            throw new IOException("Missing table descriptor for " + region.getEncodedName());
2054          }

2056          // Need to pass the expected version in the constructor.
2057          if (regionServer.executorService == null) {

2065                regionServer.updateRegionFavoredNodesMapping(region.getEncodedName(),
2066                regionOpenInfo.getFavoredNodesList());
2067              }

2069                regionServer.executorService.submit(new OpenPriorityRegionHandler(
2070                regionServer, regionServer, region, htd, masterSystemTime));
2071              } else {
2072                regionServer.executorService.submit(new OpenRegionHandler(
2073                regionServer, regionServer, region, htd, masterSystemTime));
2074              }
2075            }
2076          }
2077        }

2080      } catch (IOException ie) {

2083          builder.addOpeningState(RegionOpeningState.FAILED_OPENING);

2086        }

2712    HBaseRpcController controller = (HBaseRpcController)rpcc;

2714    if (controller != null) {

2723    RegionActionResult.Builder regionActionResultBuilder = RegionActionResult.newBuilder();
2724    Boolean processed = null;

2730      .getRegionActionCount());

2737      try {
2738        region = getRegion(regionSpecifier);

2740      } catch (IOException e) {
2741        failRegionAction(responseBuilder, regionActionResultBuilder, regionAction, cellScanner, e);
2742        continue;  // For this region it's a failure.
2743      }
2744      boolean rejectIfFromClient = shouldRejectRequestsFromClient(region);
2745      if (regionAction.hasAtomic() && regionAction.getAtomic()) {
2746        // We only allow replication in standby state and it will not set the atomic flag.
2747        if (rejectIfFromClient) {
2748          failRegionAction(responseBuilder, regionActionResultBuilder, regionAction, cellScanner,
2749            new DoNotRetryIOException(

2763              CompareOperator.valueOf(condition.getCompareType().name());

2765                ProtobufUtil.toComparator(condition.getComparator());
2766            TimeRange timeRange = condition.hasTimeRange() ?
2767              ProtobufUtil.toTimeRange(condition.getTimeRange()) :
2768              TimeRange.allTime();

2770              checkAndRowMutate(region, regionAction.getActionList(), cellScanner, row, family,
2771                qualifier, op, comparator, timeRange, regionActionResultBuilder,
2772                spaceQuotaEnforcement);
2773          } else {
2774            doAtomicBatchOp(regionActionResultBuilder, region, quota, regionAction.getActionList(),
2775              cellScanner, spaceQuotaEnforcement);
2776            processed = Boolean.TRUE;
2777          }
2778        } catch (IOException e) {
2779          rpcServer.getMetrics().exception(e);
2780          // As it's atomic, we may expect it's a global failure.
2781          regionActionResultBuilder.setException(ResponseConverter.buildException(e));
2782        }

2795          // An RpcCallBack that creates a list of scanners that needs to perform callBack
2796          // operation on completion of multiGets.
2797          // Set this only once
2798          closeCallBack = new RegionScannersCloseCallBack();
2799          context.setCallBack(closeCallBack);
2800        }
2801        cellsToReturn = doNonAtomicRegionMutation(region, quota, regionAction, cellScanner,
2802            regionActionResultBuilder, cellsToReturn, nonceGroup, closeCallBack, context,
2803            spaceQuotaEnforcement);
2804      }
2805      responseBuilder.addRegionActionResult(regionActionResultBuilder.build());
2806      quota.close();
2807      ClientProtos.RegionLoadStats regionLoadStats = region.getLoadStatistics();
2808      if (regionLoadStats != null) {
2809        regionStats.put(regionSpecifier, regionLoadStats);
2810      }

3143  private long getTimeLimit(HBaseRpcController controller, boolean allowHeartbeatMessages) {
3144    // Set the time limit to be half of the more restrictive timeout value (one of the
3145    // timeout values must be positive). In the event that both values are positive, the
3146    // more restrictive of the two is used to calculate the limit.
3147    if (allowHeartbeatMessages && (scannerLeaseTimeoutPeriod > 0 || rpcTimeout > 0)) {
3148      long timeLimitDelta;
3149      if (scannerLeaseTimeoutPeriod > 0 && rpcTimeout > 0) {
3150        timeLimitDelta = Math.min(scannerLeaseTimeoutPeriod, rpcTimeout);
3151      } else {
3152        timeLimitDelta = scannerLeaseTimeoutPeriod > 0 ? scannerLeaseTimeoutPeriod : rpcTimeout;
3153      }
3154      if (controller != null && controller.getCallTimeout() > 0) {
3155        timeLimitDelta = Math.min(timeLimitDelta, controller.getCallTimeout());
3156      }
3157      // Use half of whichever timeout value was more restrictive... But don't allow
3158      // the time limit to be less than the allowable minimum (could cause an
3159      // immediatate timeout before scanning any data).
3160      timeLimitDelta = Math.max(timeLimitDelta / 2, minimumScanTimeLimitDelta);
3161      // XXX: Can not use EnvironmentEdge here because TestIncrementTimeRange use a
3162      // ManualEnvironmentEdge. Consider using System.nanoTime instead.
3163      return System.currentTimeMillis() + timeLimitDelta;
3164    }
3165    // Default value of timeLimit is negative to indicate no timeLimit should be
3166    // enforced.
3167    return -1L;
3168  }

3193    }
3194    // This is cells inside a row. Default size is 10 so if many versions or many cfs,

3223        // create partial Results (in the event that the timeout occurs in the middle of a

3371   * @param request the scan request
3372   * @throws ServiceException

3374  @Override
3375  public ScanResponse scan(final RpcController controller, final ScanRequest request)
3376      throws ServiceException {
3377    if (controller != null && !(controller instanceof HBaseRpcController)) {
3378      throw new UnsupportedOperationException(
3379          "We only do " + "HBaseRpcControllers! FIX IF A PROBLEM: " + controller);
3380    }
3381    if (!request.hasScannerId() && !request.hasScan()) {
3382      throw new ServiceException(
3383          new DoNotRetryIOException("Missing required input: scannerId or scan"));
3384    }
3385    try {
3386      checkOpen();
3387    } catch (IOException e) {
3388      if (request.hasScannerId()) {
3389        String scannerName = Long.toString(request.getScannerId());
3390        if (LOG.isDebugEnabled()) {
3391          LOG.debug(
3392            "Server shutting down and client tried to access missing scanner " + scannerName);
3393        }
3394        if (regionServer.leases != null) {
3395          try {
3396            regionServer.leases.cancelLease(scannerName);
3397          } catch (LeaseException le) {
3398            // No problem, ignore
3399            if (LOG.isTraceEnabled()) {
3400              LOG.trace("Un-able to cancel lease of scanner. It could already be closed.");
3401            }
3402          }
3403        }
3404      }
3405      throw new ServiceException(e);
3406    }
3407    requestCount.increment();
3408    rpcScanRequestCount.increment();
3409    RegionScannerHolder rsh;
3410    ScanResponse.Builder builder = ScanResponse.newBuilder();
3411    try {
3412      if (request.hasScannerId()) {
3413        // The downstream projects such as AsyncHBase in OpenTSDB need this value. See HBASE-18000
3414        // for more details.
3415        builder.setScannerId(request.getScannerId());
3416        rsh = getRegionScanner(request);
3417      } else {
3418        rsh = newRegionScanner(request, builder);
3419      }
3420    } catch (IOException e) {
3421      if (e == SCANNER_ALREADY_CLOSED) {
3422        // Now we will close scanner automatically if there are no more results for this region but
3423        // the old client will still send a close request to us. Just ignore it and return.
3424        return builder.build();
3425      }
3426      throw new ServiceException(e);
3427    }
3428    HRegion region = rsh.r;
3429    String scannerName = rsh.scannerName;
3430    Leases.Lease lease;
3431    try {
3432      // Remove lease while its being processed in server; protects against case
3433      // where processing of request takes > lease expiration time.
3434      lease = regionServer.leases.removeLease(scannerName);
3435    } catch (LeaseException e) {
3436      throw new ServiceException(e);
3437    }
3438    if (request.hasRenew() && request.getRenew()) {
3439      // add back and return
3440      addScannerLeaseBack(lease);
3441      try {
3442        checkScanNextCallSeq(request, rsh);
3443      } catch (OutOfOrderScannerNextException e) {
3444        throw new ServiceException(e);
3445      }
3446      return builder.build();
3447    }
3448    OperationQuota quota;
3449    try {
3450      quota = getRpcQuotaManager().checkQuota(region, OperationQuota.OperationType.SCAN);
3451    } catch (IOException e) {
3452      addScannerLeaseBack(lease);
3453      throw new ServiceException(e);
3454    }
3455    try {
3456      checkScanNextCallSeq(request, rsh);
3457    } catch (OutOfOrderScannerNextException e) {
3458      addScannerLeaseBack(lease);
3459      throw new ServiceException(e);
3460    }
3461    // Now we have increased the next call sequence. If we give client an error, the retry will
3462    // never success. So we'd better close the scanner and return a DoNotRetryIOException to client
3463    // and then client will try to open a new scanner.
3464    boolean closeScanner = request.hasCloseScanner() ? request.getCloseScanner() : false;
3465    int rows; // this is scan.getCaching
3466    if (request.hasNumberOfRows()) {
3467      rows = request.getNumberOfRows();
3468    } else {
3469      rows = closeScanner ? 0 : 1;
3470    }
3471    RpcCallContext context = RpcServer.getCurrentCall().orElse(null);
3472    // now let's do the real scan.
3473    long maxQuotaResultSize = Math.min(maxScannerResultSize, quota.getReadAvailable());
3474    RegionScanner scanner = rsh.s;
3475    // this is the limit of rows for this scan, if we the number of rows reach this value, we will
3476    // close the scanner.
3477    int limitOfRows;
3478    if (request.hasLimitOfRows()) {
3479      limitOfRows = request.getLimitOfRows();
3480    } else {
3481      limitOfRows = -1;
3482    }
3483    MutableObject<Object> lastBlock = new MutableObject<>();
3484    boolean scannerClosed = false;
3485    try {
3486      List<Result> results = new ArrayList<>(Math.min(rows, 512));
3487      if (rows > 0) {
3488        boolean done = false;
3489        // Call coprocessor. Get region info from scanner.
3490        if (region.getCoprocessorHost() != null) {
3491          Boolean bypass = region.getCoprocessorHost().preScannerNext(scanner, results, rows);
3492          if (!results.isEmpty()) {
3493            for (Result r : results) {
3494              lastBlock.setValue(addSize(context, r, lastBlock.getValue()));
3495            }
3496          }
3497          if (bypass != null && bypass.booleanValue()) {
3498            done = true;
3499          }
3500        }
3501        if (!done) {
3502          scan((HBaseRpcController) controller, request, rsh, maxQuotaResultSize, rows, limitOfRows,
3503            results, builder, lastBlock, context);
3504        } else {
3505          builder.setMoreResultsInRegion(!results.isEmpty());
3506        }
3507      } else {
3508        // This is a open scanner call with numberOfRow = 0, so set more results in region to true.
3509        builder.setMoreResultsInRegion(true);
3510      }
3511
3512      quota.addScanResult(results);
3513      addResults(builder, results, (HBaseRpcController) controller,
3514        RegionReplicaUtil.isDefaultReplica(region.getRegionInfo()),
3515        isClientCellBlockSupport(context));
3516      if (scanner.isFilterDone() && results.isEmpty()) {
3517        // If the scanner's filter - if any - is done with the scan
3518        // only set moreResults to false if the results is empty. This is used to keep compatible
3519        // with the old scan implementation where we just ignore the returned results if moreResults
3520        // is false. Can remove the isEmpty check after we get rid of the old implementation.
3521        builder.setMoreResults(false);
3522      }
3523      // Later we may close the scanner depending on this flag so here we need to make sure that we
3524      // have already set this flag.
3525      assert builder.hasMoreResultsInRegion();
3526      // we only set moreResults to false in the above code, so set it to true if we haven't set it
3527      // yet.
3528      if (!builder.hasMoreResults()) {
3529        builder.setMoreResults(true);
3530      }
3531      if (builder.getMoreResults() && builder.getMoreResultsInRegion() && !results.isEmpty()) {
3532        // Record the last cell of the last result if it is a partial result
3533        // We need this to calculate the complete rows we have returned to client as the
3534        // mayHaveMoreCellsInRow is true does not mean that there will be extra cells for the
3535        // current row. We may filter out all the remaining cells for the current row and just
3536        // return the cells of the nextRow when calling RegionScanner.nextRaw. So here we need to
3537        // check for row change.
3538        Result lastResult = results.get(results.size() - 1);
3539        if (lastResult.mayHaveMoreCellsInRow()) {
3540          rsh.rowOfLastPartialResult = lastResult.getRow();
3541        } else {
3542          rsh.rowOfLastPartialResult = null;
3543        }
3544      }
3545      if (!builder.getMoreResults() || !builder.getMoreResultsInRegion() || closeScanner) {
3546        scannerClosed = true;
3547        closeScanner(region, scanner, scannerName, context);
3548      }
3549      return builder.build();
3550    } catch (IOException e) {
3551      try {
3552        // scanner is closed here
3553        scannerClosed = true;
3554        // The scanner state might be left in a dirty state, so we will tell the Client to
3555        // fail this RPC and close the scanner while opening up another one from the start of
3556        // row that the client has last seen.
3557        closeScanner(region, scanner, scannerName, context);
3558
3559        // If it is a DoNotRetryIOException already, throw as it is. Unfortunately, DNRIOE is
3560        // used in two different semantics.
3561        // (1) The first is to close the client scanner and bubble up the exception all the way
3562        // to the application. This is preferred when the exception is really un-recoverable
3563        // (like CorruptHFileException, etc). Plain DoNotRetryIOException also falls into this
3564        // bucket usually.
3565        // (2) Second semantics is to close the current region scanner only, but continue the
3566        // client scanner by overriding the exception. This is usually UnknownScannerException,
3567        // OutOfOrderScannerNextException, etc where the region scanner has to be closed, but the
3568        // application-level ClientScanner has to continue without bubbling up the exception to
3569        // the client. See ClientScanner code to see how it deals with these special exceptions.
3570        if (e instanceof DoNotRetryIOException) {
3571          throw e;
3572        }
3573
3574        // If it is a FileNotFoundException, wrap as a
3575        // DoNotRetryIOException. This can avoid the retry in ClientScanner.
3576        if (e instanceof FileNotFoundException) {
3577          throw new DoNotRetryIOException(e);
3578        }
3579
3580        // We closed the scanner already. Instead of throwing the IOException, and client
3581        // retrying with the same scannerId only to get USE on the next RPC, we directly throw
3582        // a special exception to save an RPC.
3583        if (VersionInfoUtil.hasMinimumVersion(context.getClientVersionInfo(), 1, 4)) {
3584          // 1.4.0+ clients know how to handle
3585          throw new ScannerResetException("Scanner is closed on the server-side", e);
3586        } else {
3587          // older clients do not know about SRE. Just throw USE, which they will handle
3588          throw new UnknownScannerException("Throwing UnknownScannerException to reset the client"
3589              + " scanner state for clients older than 1.3.", e);
3590        }
3591      } catch (IOException ioe) {
3592        throw new ServiceException(ioe);
3593      }
3594    } finally {
3595      if (!scannerClosed) {
3596        // Adding resets expiration time on lease.
3597        // the closeCallBack will be set in closeScanner so here we only care about shippedCallback
3598        if (context != null) {
3599          context.setCallBack(rsh.shippedCallback);
3600        } else {
3601          // When context != null, adding back the lease will be done in callback set above.
3602          addScannerLeaseBack(lease);
3603        }
3604      }
3605      quota.close();
3606    }
3607  }

3609  private void closeScanner(HRegion region, RegionScanner scanner, String scannerName,
3610      RpcCallContext context) throws IOException {
3611    if (region.getCoprocessorHost() != null) {
3612      if (region.getCoprocessorHost().preScannerClose(scanner)) {
3613        // bypass the actual close.
3614        return;
3615      }
3616    }
3617    RegionScannerHolder rsh = scanners.remove(scannerName);
3618    if (rsh != null) {
3619      if (context != null) {
3620        context.setCallBack(rsh.closeCallBack);
3621      } else {
3622        rsh.s.close();
3623      }
3624      if (region.getCoprocessorHost() != null) {
3625        region.getCoprocessorHost().postScannerClose(scanner);
3626      }
3627      closedScanners.put(scannerName, scannerName);
3628    }
3629  }
