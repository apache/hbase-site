417  /**
418   * Instantiated as a scanner lease. If the lease times out, the scanner is
419   * closed
420   */
421  private class ScannerListener implements LeaseListener {
422    private final String scannerName;
423
424    ScannerListener(final String n) {
425      this.scannerName = n;
426    }
427
428    @Override
429    public void leaseExpired() {
430      RegionScannerHolder rsh = scanners.remove(this.scannerName);
431      if (rsh != null) {
432        RegionScanner s = rsh.s;
433        LOG.info("Scanner " + this.scannerName + " lease expired on region "
434          + s.getRegionInfo().getRegionNameAsString());
435        Region region = null;
436        try {
437          region = regionServer.getRegion(s.getRegionInfo().getRegionName());
438          if (region != null && region.getCoprocessorHost() != null) {
439            region.getCoprocessorHost().preScannerClose(s);
440          }
441        } catch (IOException e) {
442          LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
443        } finally {
444          try {
445            s.close();
446            if (region != null && region.getCoprocessorHost() != null) {
447              region.getCoprocessorHost().postScannerClose(s);
448            }
449          } catch (IOException e) {
450            LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
451          }
452        }
453      } else {
454        LOG.warn("Scanner " + this.scannerName + " lease expired, but no related" +
455          " scanner found, hence no chance to close that related scanner!");
456      }
457    }
458  }

475   * Starts the nonce operation for a mutation, if needed.
476   * @param mutation Mutation.
477   * @param nonceGroup Nonce group from the request.
478   * @returns whether to proceed this mutation.
479   */
480  private boolean startNonceOperation(final MutationProto mutation, long nonceGroup)
481      throws IOException {
482    if (regionServer.nonceManager == null || !mutation.hasNonce()) return true;
483    boolean canProceed = false;
484    try {
485      canProceed = regionServer.nonceManager.startOperation(
486        nonceGroup, mutation.getNonce(), regionServer);
487    } catch (InterruptedException ex) {
488      throw new InterruptedIOException("Nonce start operation interrupted");
489    }
490    return canProceed;
491  }

651   *
652   * @param region
653   * @param m
654   * @param cellScanner
655   * @return result to return to client if default operation should be
656   * bypassed as indicated by RegionObserver, null otherwise
657   * @throws IOException
658   */
659  private Result append(final Region region, final OperationQuota quota,
660      final MutationProto mutation, final CellScanner cellScanner, long nonceGroup,
661      ActivePolicyEnforcement spaceQuota)
662      throws IOException {
663    long before = EnvironmentEdgeManager.currentTime();
664    Append append = ProtobufUtil.toAppend(mutation, cellScanner);
665    checkCellSizeLimit(region, append);
666    spaceQuota.getPolicyEnforcement(region).check(append);
667    quota.addMutation(append);
668    Result r = null;
669    if (region.getCoprocessorHost() != null) {
670      r = region.getCoprocessorHost().preAppend(append);
671    }
672    if (r == null) {
673      boolean canProceed = startNonceOperation(mutation, nonceGroup);
674      boolean success = false;
675      try {
676        long nonce = mutation.hasNonce() ? mutation.getNonce() : HConstants.NO_NONCE;
677        if (canProceed) {
678          r = region.append(append, nonceGroup, nonce);
679        } else {
680          // convert duplicate append to get
681          List<Cell> results = region.get(ProtobufUtil.toGet(mutation, cellScanner), false,
682            nonceGroup, nonce);
683          r = Result.create(results);
684        }
685        success = true;
686      } finally {
687        if (canProceed) {
688          endNonceOperation(mutation, nonceGroup, success);
689        }
690      }
691      if (region.getCoprocessorHost() != null) {
692        r = region.getCoprocessorHost().postAppend(append, r);
693      }
694    }
695    if (regionServer.metricsRegionServer != null) {
696      regionServer.metricsRegionServer.updateAppend(
697        EnvironmentEdgeManager.currentTime() - before);
698    }
699    return r;
700  }

753  /**
754   * Run through the regionMutation <code>rm</code> and per Mutation, do the work, and then when
755   * done, add an instance of a {@link ResultOrException} that corresponds to each Mutation.
756   * @param region
757   * @param actions
758   * @param cellScanner
759   * @param builder
760   * @param cellsToReturn  Could be null. May be allocated in this method.  This is what this
761   * method returns as a 'result'.
762   * @param closeCallBack the callback to be used with multigets
763   * @param context the current RpcCallContext
764   * @return Return the <code>cellScanner</code> passed
765   */
766  private List<CellScannable> doNonAtomicRegionMutation(final Region region,
767      final OperationQuota quota, final RegionAction actions, final CellScanner cellScanner,
768      final RegionActionResult.Builder builder, List<CellScannable> cellsToReturn, long nonceGroup,
769      final RegionScannersCloseCallBack closeCallBack, RpcCallContext context,
770      ActivePolicyEnforcement spaceQuotaEnforcement) {
771    // Gather up CONTIGUOUS Puts and Deletes in this mutations List.  Idea is that rather than do
772    // one at a time, we instead pass them in batch.  Be aware that the corresponding
773    // ResultOrException instance that matches each Put or Delete is then added down in the
774    // doBatchOp call.  We should be staying aligned though the Put and Delete are deferred/batched
775    List<ClientProtos.Action> mutations = null;
776    long maxQuotaResultSize = Math.min(maxScannerResultSize, quota.getReadAvailable());
777    IOException sizeIOE = null;
778    Object lastBlock = null;
779    ClientProtos.ResultOrException.Builder resultOrExceptionBuilder = ResultOrException.newBuilder();
780    boolean hasResultOrException = false;
781    for (ClientProtos.Action action : actions.getActionList()) {
782      hasResultOrException = false;
783      resultOrExceptionBuilder.clear();
784      try {
785        Result r = null;
786
787        if (context != null
788            && context.isRetryImmediatelySupported()
789            && (context.getResponseCellSize() > maxQuotaResultSize
790              || context.getResponseBlockSize() + context.getResponseExceptionSize()
791              > maxQuotaResultSize)) {
792
793          // We're storing the exception since the exception and reason string won't
794          // change after the response size limit is reached.
795          if (sizeIOE == null ) {
796            // We don't need the stack un-winding do don't throw the exception.
797            // Throwing will kill the JVM's JIT.
798            //
799            // Instead just create the exception and then store it.
800            sizeIOE = new MultiActionResultTooLarge("Max size exceeded"
801                + " CellSize: " + context.getResponseCellSize()
802                + " BlockSize: " + context.getResponseBlockSize());
803
804            // Only report the exception once since there's only one request that
805            // caused the exception. Otherwise this number will dominate the exceptions count.
806            rpcServer.getMetrics().exception(sizeIOE);
807          }
808
809          // Now that there's an exception is known to be created
810          // use it for the response.
811          //
812          // This will create a copy in the builder.
813          hasResultOrException = true;
814          NameBytesPair pair = ResponseConverter.buildException(sizeIOE);
815          resultOrExceptionBuilder.setException(pair);
816          context.incrementResponseExceptionSize(pair.getSerializedSize());
817          resultOrExceptionBuilder.setIndex(action.getIndex());
818          builder.addResultOrException(resultOrExceptionBuilder.build());
819          if (cellScanner != null) {
820            skipCellsForMutation(action, cellScanner);
821          }
822          continue;
823        }
824        if (action.hasGet()) {
825          long before = EnvironmentEdgeManager.currentTime();
826          try {
827            Get get = ProtobufUtil.toGet(action.getGet());
828            if (context != null) {
829              r = get(get, ((HRegion) region), closeCallBack, context);
830            } else {
831              r = region.get(get);
832            }
833          } finally {
834            if (regionServer.metricsRegionServer != null) {
835              regionServer.metricsRegionServer.updateGet(
836                EnvironmentEdgeManager.currentTime() - before);
837            }
838          }
839        } else if (action.hasServiceCall()) {
840          hasResultOrException = true;
841          try {
842            com.google.protobuf.Message result =
843                execServiceOnRegion(region, action.getServiceCall());
844            ClientProtos.CoprocessorServiceResult.Builder serviceResultBuilder =
845                ClientProtos.CoprocessorServiceResult.newBuilder();
846            resultOrExceptionBuilder.setServiceResult(
847                serviceResultBuilder.setValue(
848                  serviceResultBuilder.getValueBuilder()
849                    .setName(result.getClass().getName())
850                    // TODO: Copy!!!
851                    .setValue(UnsafeByteOperations.unsafeWrap(result.toByteArray()))));
852          } catch (IOException ioe) {
853            rpcServer.getMetrics().exception(ioe);
854            NameBytesPair pair = ResponseConverter.buildException(ioe);
855            resultOrExceptionBuilder.setException(pair);
856            context.incrementResponseExceptionSize(pair.getSerializedSize());
857          }
858        } else if (action.hasMutation()) {
859          MutationType type = action.getMutation().getMutateType();
860          if (type != MutationType.PUT && type != MutationType.DELETE && mutations != null &&
861              !mutations.isEmpty()) {
862            // Flush out any Puts or Deletes already collected.
863            doBatchOp(builder, region, quota, mutations, cellScanner, spaceQuotaEnforcement);
864            mutations.clear();
865          }
866          switch (type) {
867            case APPEND:
868              r = append(region, quota, action.getMutation(), cellScanner, nonceGroup,
869                  spaceQuotaEnforcement);
870              break;
871            case INCREMENT:
872              r = increment(region, quota, action.getMutation(), cellScanner, nonceGroup,
873                  spaceQuotaEnforcement);
874              break;
875            case PUT:
876            case DELETE:
877              // Collect the individual mutations and apply in a batch
878              if (mutations == null) {
879                mutations = new ArrayList<>(actions.getActionCount());
880              }
881              mutations.add(action);
882              break;
883            default:
884              throw new DoNotRetryIOException("Unsupported mutate type: " + type.name());
885          }
886        } else {
887          throw new HBaseIOException("Unexpected Action type");
888        }
889        if (r != null) {
890          ClientProtos.Result pbResult = null;
891          if (isClientCellBlockSupport(context)) {
892            pbResult = ProtobufUtil.toResultNoData(r);
893            //  Hard to guess the size here.  Just make a rough guess.
894            if (cellsToReturn == null) {
895              cellsToReturn = new ArrayList<>();
896            }
897            cellsToReturn.add(r);
898          } else {
899            pbResult = ProtobufUtil.toResult(r);
900          }
901          lastBlock = addSize(context, r, lastBlock);
902          hasResultOrException = true;
903          resultOrExceptionBuilder.setResult(pbResult);
904        }
905        // Could get to here and there was no result and no exception.  Presumes we added
906        // a Put or Delete to the collecting Mutations List for adding later.  In this
907        // case the corresponding ResultOrException instance for the Put or Delete will be added
908        // down in the doBatchOp method call rather than up here.
909      } catch (IOException ie) {
910        rpcServer.getMetrics().exception(ie);
911        hasResultOrException = true;
912        NameBytesPair pair = ResponseConverter.buildException(ie);
913        resultOrExceptionBuilder.setException(pair);
914        context.incrementResponseExceptionSize(pair.getSerializedSize());
915      }
916      if (hasResultOrException) {
917        // Propagate index.
918        resultOrExceptionBuilder.setIndex(action.getIndex());
919        builder.addResultOrException(resultOrExceptionBuilder.build());
920      }
921    }
922    // Finish up any outstanding mutations
923    if (mutations != null && !mutations.isEmpty()) {
924      doBatchOp(builder, region, quota, mutations, cellScanner, spaceQuotaEnforcement);
925    }
926    return cellsToReturn;
927  }

929  private void checkCellSizeLimit(final Region region, final Mutation m) throws IOException {
930    if (!(region instanceof HRegion)) {
931      return;
932    }
933    HRegion r = (HRegion)region;
934    if (r.maxCellSize > 0) {
935      CellScanner cells = m.cellScanner();
936      while (cells.advance()) {
937        int size = CellUtil.estimatedSerializedSizeOf(cells.current());
938        if (size > r.maxCellSize) {
939          String msg = "Cell with size " + size + " exceeds limit of " + r.maxCellSize + " bytes";
940          if (LOG.isDebugEnabled()) {
941            LOG.debug(msg);
942          }
943          throw new DoNotRetryIOException(msg);
944        }
945      }
946    }

949  /**
950   * Execute a list of Put/Delete mutations.
951   *
952   * @param builder
953   * @param region
954   * @param mutations
955   */
956  private void doBatchOp(final RegionActionResult.Builder builder, final Region region,
957      final OperationQuota quota, final List<ClientProtos.Action> mutations,
958      final CellScanner cells, ActivePolicyEnforcement spaceQuotaEnforcement) {
959    Mutation[] mArray = new Mutation[mutations.size()];
960    long before = EnvironmentEdgeManager.currentTime();
961    boolean batchContainsPuts = false, batchContainsDelete = false;
962    try {
963      /** HBASE-17924
964       * mutationActionMap is a map to map the relation between mutations and actions
965       * since mutation array may have been reoredered.In order to return the right
966       * result or exception to the corresponding actions, We need to know which action
967       * is the mutation belong to. We can't sort ClientProtos.Action array, since they
968       * are bonded to cellscanners.
969       */
970      Map<Mutation, ClientProtos.Action> mutationActionMap = new HashMap<Mutation, ClientProtos.Action>();
971      int i = 0;
972      for (ClientProtos.Action action: mutations) {
973        MutationProto m = action.getMutation();
974        Mutation mutation;
975        if (m.getMutateType() == MutationType.PUT) {
976          mutation = ProtobufUtil.toPut(m, cells);
977          batchContainsPuts = true;
978        } else {
979          mutation = ProtobufUtil.toDelete(m, cells);
980          batchContainsDelete = true;
981        }
982        mutationActionMap.put(mutation, action);
983        mArray[i++] = mutation;
984        checkCellSizeLimit(region, mutation);
985        // Check if a space quota disallows this mutation
986        spaceQuotaEnforcement.getPolicyEnforcement(region).check(mutation);
987        quota.addMutation(mutation);
988      }
989
990      if (!region.getRegionInfo().isMetaTable()) {
991        regionServer.cacheFlusher.reclaimMemStoreMemory();
992      }
993
994      // HBASE-17924
995      // sort to improve lock efficiency
996      Arrays.sort(mArray);
997
998      OperationStatus[] codes = region.batchMutate(mArray, HConstants.NO_NONCE,
999        HConstants.NO_NONCE);
1000      for (i = 0; i < codes.length; i++) {
1001        Mutation currentMutation = mArray[i];
1002        ClientProtos.Action currentAction = mutationActionMap.get(currentMutation);
1003        int index = currentAction.getIndex();
1004        Exception e = null;
1005        switch (codes[i].getOperationStatusCode()) {
1006          case BAD_FAMILY:
1007            e = new NoSuchColumnFamilyException(codes[i].getExceptionMsg());
1008            builder.addResultOrException(getResultOrException(e, index));
1009            break;
1010
1011          case SANITY_CHECK_FAILURE:
1012            e = new FailedSanityCheckException(codes[i].getExceptionMsg());
1013            builder.addResultOrException(getResultOrException(e, index));
1014            break;
1015
1016          default:
1017            e = new DoNotRetryIOException(codes[i].getExceptionMsg());
1018            builder.addResultOrException(getResultOrException(e, index));
1019            break;
1020
1021          case SUCCESS:
1022            builder.addResultOrException(getResultOrException(
1023              ClientProtos.Result.getDefaultInstance(), index));
1024            break;
1025        }
1026      }
1027    } catch (IOException ie) {
1028      for (int i = 0; i < mutations.size(); i++) {
1029        builder.addResultOrException(getResultOrException(ie, mutations.get(i).getIndex()));
1030      }
1031    }
1032    if (regionServer.metricsRegionServer != null) {
1033      long after = EnvironmentEdgeManager.currentTime();
1034      if (batchContainsPuts) {
1035        regionServer.metricsRegionServer.updatePutBatch(after - before);
1036      }
1037      if (batchContainsDelete) {
1038        regionServer.metricsRegionServer.updateDeleteBatch(after - before);
1039      }
1040    }
1041  }

1043  /**
1044   * Execute a list of Put/Delete mutations. The function returns OperationStatus instead of
1045   * constructing MultiResponse to save a possible loop if caller doesn't need MultiResponse.
1046   * @param region
1047   * @param mutations
1048   * @param replaySeqId
1049   * @return an array of OperationStatus which internally contains the OperationStatusCode and the
1050   *         exceptionMessage if any
1051   * @throws IOException
1052   */
1053  private OperationStatus [] doReplayBatchOp(final Region region,
1054      final List<WALSplitter.MutationReplay> mutations, long replaySeqId) throws IOException {
1055    long before = EnvironmentEdgeManager.currentTime();
1056    boolean batchContainsPuts = false, batchContainsDelete = false;
1057    try {
1058      for (Iterator<WALSplitter.MutationReplay> it = mutations.iterator(); it.hasNext();) {
1059        WALSplitter.MutationReplay m = it.next();
1060
1061        if (m.type == MutationType.PUT) {
1062          batchContainsPuts = true;
1063        } else {
1064          batchContainsDelete = true;
1065        }
1066
1067        NavigableMap<byte[], List<Cell>> map = m.mutation.getFamilyCellMap();
1068        List<Cell> metaCells = map.get(WALEdit.METAFAMILY);
1069        if (metaCells != null && !metaCells.isEmpty()) {
1070          for (Cell metaCell : metaCells) {
1071            CompactionDescriptor compactionDesc = WALEdit.getCompaction(metaCell);
1072            boolean isDefaultReplica = RegionReplicaUtil.isDefaultReplica(region.getRegionInfo());
1073            HRegion hRegion = (HRegion)region;
1074            if (compactionDesc != null) {
1075              // replay the compaction. Remove the files from stores only if we are the primary
1076              // region replica (thus own the files)
1077              hRegion.replayWALCompactionMarker(compactionDesc, !isDefaultReplica, isDefaultReplica,
1078                replaySeqId);
1079              continue;
1080            }
1081            FlushDescriptor flushDesc = WALEdit.getFlushDescriptor(metaCell);
1082            if (flushDesc != null && !isDefaultReplica) {
1083              hRegion.replayWALFlushMarker(flushDesc, replaySeqId);
1084              continue;
1085            }
1086            RegionEventDescriptor regionEvent = WALEdit.getRegionEventDescriptor(metaCell);
1087            if (regionEvent != null && !isDefaultReplica) {
1088              hRegion.replayWALRegionEventMarker(regionEvent);
1089              continue;
1090            }
1091            BulkLoadDescriptor bulkLoadEvent = WALEdit.getBulkLoadDescriptor(metaCell);
1092            if (bulkLoadEvent != null) {
1093              hRegion.replayWALBulkLoadEventMarker(bulkLoadEvent);
1094              continue;
1095            }
1096          }
1097          it.remove();
1098        }
1099      }
1100      requestCount.increment();
1101      requestRowActionCount.add(mutations.size());
1102      if (!region.getRegionInfo().isMetaTable()) {
1103        regionServer.cacheFlusher.reclaimMemStoreMemory();
1104      }
1105      return region.batchReplay(mutations.toArray(
1106        new WALSplitter.MutationReplay[mutations.size()]), replaySeqId);
1107    } finally {
1108      if (regionServer.metricsRegionServer != null) {
1109        long after = EnvironmentEdgeManager.currentTime();
1110          if (batchContainsPuts) {
1111          regionServer.metricsRegionServer.updatePutBatch(after - before);
1112        }
1113        if (batchContainsDelete) {
1114          regionServer.metricsRegionServer.updateDeleteBatch(after - before);
1115        }
1116      }
1117    }
1118  }

1120  private void closeAllScanners() {
1121    // Close any outstanding scanners. Means they'll get an UnknownScanner
1122    // exception next time they come in.
1123    for (Map.Entry<String, RegionScannerHolder> e : scanners.entrySet()) {
1124      try {
1125        e.getValue().s.close();
1126      } catch (IOException ioe) {

1299  /**
1300   * Method to account for the size of retained cells and retained data blocks.
1301   * @return an object that represents the last referenced block from this response.
1302   */
1303  Object addSize(RpcCallContext context, Result r, Object lastBlock) {
1304    if (context != null && r != null && !r.isEmpty()) {
1305      for (Cell c : r.rawCells()) {
1306        context.incrementResponseCellSize(CellUtil.estimatedSerializedSizeOf(c));
1307
1308        // Since byte buffers can point all kinds of crazy places it's harder to keep track
1309        // of which blocks are kept alive by what byte buffer.
1310        // So we make a guess.
1311        if (c instanceof ByteBufferCell) {
1312          ByteBufferCell bbCell = (ByteBufferCell) c;
1313          ByteBuffer bb = bbCell.getValueByteBuffer();
1314          if (bb != lastBlock) {
1315            context.incrementResponseBlockSize(bb.capacity());
1316            lastBlock = bb;
1317          }
1318        } else {
1319          // We're using the last block being the same as the current block as
1320          // a proxy for pointing to a new block. This won't be exact.
1321          // If there are multiple gets that bounce back and forth
1322          // Then it's possible that this will over count the size of
1323          // referenced blocks. However it's better to over count and
1324          // use two rpcs than to OOME the regionserver.
1325          byte[] valueArray = c.getValueArray();
1326          if (valueArray != lastBlock) {
1327            context.incrementResponseBlockSize(valueArray.length);
1328            lastBlock = valueArray;
1329          }
1330        }
1331
1332      }
1333    }
1334    return lastBlock;
1335  }

1614      LOG.info("Flushing " + region.getRegionInfo().getRegionNameAsString());
1615      boolean shouldFlush = true;
1616      if (request.hasIfOlderThanTs()) {

1622            request.getWriteFlushWalMarker() : false;

1624        HRegion.FlushResultImpl flushResult = (HRegion.FlushResultImpl)
1625            ((HRegion)region).flushcache(true, writeFlushWalMarker);
1626        boolean compactionNeeded = flushResult.isCompactionNeeded();
1627        if (compactionNeeded) {
1628          regionServer.compactSplitThread.requestSystemCompaction(region,
1629            "Compaction through user triggered flush");
1630        }
1631        builder.setFlushed(flushResult.isFlushSucceeded());

1822  /**
1823   * Open asynchronously a region or a set of regions on the region server.
1824   *
1825   * The opening is coordinated by ZooKeeper, and this method requires the znode to be created
1826   *  before being called. As a consequence, this method should be called only from the master.
1827   * <p>
1828   * Different manages states for the region are:
1829   * </p><ul>
1830   *  <li>region not opened: the region opening will start asynchronously.</li>
1831   *  <li>a close is already in progress: this is considered as an error.</li>
1832   *  <li>an open is already in progress: this new open request will be ignored. This is important
1833   *  because the Master can do multiple requests if it crashes.</li>
1834   *  <li>the region is already opened:  this new open request will be ignored.</li>
1835   *  </ul>
1836   * <p>
1837   * Bulk assign: If there are more than 1 region to open, it will be considered as a bulk assign.
1838   * For a single region opening, errors are sent through a ServiceException. For bulk assign,
1839   * errors are put in the response as FAILED_OPENING.
1840   * </p>
1841   * @param controller the RPC controller
1842   * @param request the request
1843   * @throws ServiceException
1844   */
1845  @Override
1846  @QosPriority(priority=HConstants.ADMIN_QOS)
1847  public OpenRegionResponse openRegion(final RpcController controller,
1848      final OpenRegionRequest request) throws ServiceException {
1849    requestCount.increment();
1850    if (request.hasServerStartCode()) {
1851      // check that we are the same server that this RPC is intended for.
1852      long serverStartCode = request.getServerStartCode();
1853      if (regionServer.serverName.getStartcode() !=  serverStartCode) {
1854        throw new ServiceException(new DoNotRetryIOException("This RPC was intended for a " +
1855            "different server with startCode: " + serverStartCode + ", this server is: "
1856            + regionServer.serverName));
1857      }
1858    }
1859
1860    OpenRegionResponse.Builder builder = OpenRegionResponse.newBuilder();
1861    final int regionCount = request.getOpenInfoCount();
1862    final Map<TableName, HTableDescriptor> htds = new HashMap<>(regionCount);
1863    final boolean isBulkAssign = regionCount > 1;
1864    try {
1865      checkOpen();
1866    } catch (IOException ie) {
1867      TableName tableName = null;
1868      if (regionCount == 1) {
1869        RegionInfo ri = request.getOpenInfo(0).getRegion();
1870        if (ri != null) {
1871          tableName = ProtobufUtil.toTableName(ri.getTableName());
1872        }
1873      }
1874      if (!TableName.META_TABLE_NAME.equals(tableName)) {
1875        throw new ServiceException(ie);
1876      }
1877      // We are assigning meta, wait a little for regionserver to finish initialization.
1878      int timeout = regionServer.conf.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY,
1879        HConstants.DEFAULT_HBASE_RPC_TIMEOUT) >> 2; // Quarter of RPC timeout
1880      long endTime = System.currentTimeMillis() + timeout;
1881      synchronized (regionServer.online) {
1882        try {
1883          while (System.currentTimeMillis() <= endTime
1884              && !regionServer.isStopped() && !regionServer.isOnline()) {
1885            regionServer.online.wait(regionServer.msgInterval);
1886          }
1887          checkOpen();
1888        } catch (InterruptedException t) {
1889          Thread.currentThread().interrupt();
1890          throw new ServiceException(t);
1891        } catch (IOException e) {
1892          throw new ServiceException(e);
1893        }
1894      }
1895    }
1896
1897    long masterSystemTime = request.hasMasterSystemTime() ? request.getMasterSystemTime() : -1;
1898
1899    for (RegionOpenInfo regionOpenInfo : request.getOpenInfoList()) {
1900      final HRegionInfo region = HRegionInfo.convert(regionOpenInfo.getRegion());
1901      HTableDescriptor htd;
1902      try {
1903        String encodedName = region.getEncodedName();
1904        byte[] encodedNameBytes = region.getEncodedNameAsBytes();
1905        final Region onlineRegion = regionServer.getFromOnlineRegions(encodedName);
1906        if (onlineRegion != null) {
1907          // The region is already online. This should not happen any more.
1908          String error = "Received OPEN for the region:"
1909            + region.getRegionNameAsString() + ", which is already online";
1910          LOG.warn(error);
1911          //regionServer.abort(error);
1912          //throw new IOException(error);
1913          builder.addOpeningState(RegionOpeningState.OPENED);
1914          continue;
1915        }
1916        LOG.info("Open " + region.getRegionNameAsString());
1917
1918        final Boolean previous = regionServer.regionsInTransitionInRS.putIfAbsent(
1919          encodedNameBytes, Boolean.TRUE);
1920
1921        if (Boolean.FALSE.equals(previous)) {
1922          if (regionServer.getFromOnlineRegions(encodedName) != null) {
1923            // There is a close in progress. This should not happen any more.
1924            String error = "Received OPEN for the region:"
1925              + region.getRegionNameAsString() + ", which we are already trying to CLOSE";
1926            regionServer.abort(error);
1927            throw new IOException(error);
1928          }
1929          regionServer.regionsInTransitionInRS.put(encodedNameBytes, Boolean.TRUE);
1930        }
1931
1932        if (Boolean.TRUE.equals(previous)) {
1933          // An open is in progress. This is supported, but let's log this.
1934          LOG.info("Receiving OPEN for the region:" +
1935            region.getRegionNameAsString() + ", which we are already trying to OPEN"
1936              + " - ignoring this new request for this region.");
1937        }
1938
1939        // We are opening this region. If it moves back and forth for whatever reason, we don't
1940        // want to keep returning the stale moved record while we are opening/if we close again.
1941        regionServer.removeFromMovedRegions(region.getEncodedName());
1942
1943        if (previous == null || !previous.booleanValue()) {
1944          // check if the region to be opened is marked in recovering state in ZK
1945          if (ZKSplitLog.isRegionMarkedRecoveringInZK(regionServer.getZooKeeper(),
1946              region.getEncodedName())) {
1947            // Check if current region open is for distributedLogReplay. This check is to support
1948            // rolling restart/upgrade where we want to Master/RS see same configuration
1949            if (!regionOpenInfo.hasOpenForDistributedLogReplay()
1950                  || regionOpenInfo.getOpenForDistributedLogReplay()) {
1951              regionServer.recoveringRegions.put(region.getEncodedName(), null);
1952            } else {
1953              // Remove stale recovery region from ZK when we open region not for recovering which
1954              // could happen when turn distributedLogReplay off from on.
1955              List<String> tmpRegions = new ArrayList<>();
1956              tmpRegions.add(region.getEncodedName());
1957              ZKSplitLog.deleteRecoveringRegionZNodes(regionServer.getZooKeeper(),
1958                tmpRegions);
1959            }
1960          }
1961          htd = htds.get(region.getTable());
1962          if (htd == null) {
1963            htd = regionServer.tableDescriptors.get(region.getTable());
1964            htds.put(region.getTable(), htd);
1965          }
1966          if (htd == null) {
1967            throw new IOException("Missing table descriptor for " + region.getEncodedName());
1968          }
1969          // If there is no action in progress, we can submit a specific handler.
1970          // Need to pass the expected version in the constructor.
1971          if (region.isMetaRegion()) {
1972            regionServer.service.submit(new OpenMetaHandler(
1973              regionServer, regionServer, region, htd, masterSystemTime));
1974          } else {
1975            if (regionOpenInfo.getFavoredNodesCount() > 0) {
1976              regionServer.updateRegionFavoredNodesMapping(region.getEncodedName(),
1977                  regionOpenInfo.getFavoredNodesList());
1978            }
1979            if (htd.getPriority() >= HConstants.ADMIN_QOS || region.getTable().isSystemTable()) {
1980              regionServer.service.submit(new OpenPriorityRegionHandler(
1981                regionServer, regionServer, region, htd, masterSystemTime));
1982            } else {
1983              regionServer.service.submit(new OpenRegionHandler(
1984                regionServer, regionServer, region, htd, masterSystemTime));
1985            }
1986          }
1987        }
1988
1989        builder.addOpeningState(RegionOpeningState.OPENED);
1990
1991      } catch (KeeperException zooKeeperEx) {
1992        LOG.error("Can't retrieve recovering state from zookeeper", zooKeeperEx);
1993        throw new ServiceException(zooKeeperEx);
1994      } catch (IOException ie) {
1995        LOG.warn("Failed opening region " + region.getRegionNameAsString(), ie);
1996        if (isBulkAssign) {
1997          builder.addOpeningState(RegionOpeningState.FAILED_OPENING);
1998        } else {
1999          throw new ServiceException(ie);
2000        }
2001      }
2002    }
2003    return builder.build();
2004  }

2083          ServerRegionReplicaUtil.isDefaultReplica(region.getRegionInfo())

2948    // Set the time limit to be half of the more restrictive timeout value (one of the

2962      // the time limit to be less than the allowable minimum (could cause an
