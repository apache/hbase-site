020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.Arrays;
023import java.util.Collection;
024import java.util.Collections;
025import java.util.List;
026
027import org.apache.commons.logging.Log;
028import org.apache.commons.logging.LogFactory;
029import org.apache.hadoop.hbase.Cell;
030import org.apache.hadoop.hbase.CellComparator;
031import org.apache.hadoop.hbase.CellUtil;
032import org.apache.hadoop.hbase.classification.InterfaceAudience;
033import org.apache.hadoop.hbase.util.Bytes;

035/**
036 * Base class for cell sink that separates the provided cells into multiple files for stripe
037 * compaction.
038 */
039@InterfaceAudience.Private
040public abstract class StripeMultiFileWriter extends AbstractMultiFileWriter {
041
042  private static final Log LOG = LogFactory.getLog(StripeMultiFileWriter.class);
043
044  protected final CellComparator comparator;
045  protected List<StoreFileWriter> existingWriters;
046  protected List<byte[]> boundaries;
047
048  /** Whether to write stripe metadata */
049  private boolean doWriteStripeMetadata = true;
050
051  public StripeMultiFileWriter(CellComparator comparator) {
052    this.comparator = comparator;
053  }
054
055  public void setNoStripeMetadata() {
056    this.doWriteStripeMetadata = false;
057  }
058
059  @Override
060  protected Collection<StoreFileWriter> writers() {
061    return existingWriters;
062  }
063
064  protected abstract void preCommitWritersInternal() throws IOException;
065
066  @Override
067  protected final void preCommitWriters() throws IOException {
068    // do some sanity check here.
069    assert this.existingWriters != null;
070    preCommitWritersInternal();
071    assert this.boundaries.size() == (this.existingWriters.size() + 1);
072  }
073
074  @Override
075  protected void preCloseWriter(StoreFileWriter writer) throws IOException {
076    if (doWriteStripeMetadata) {
077      if (LOG.isDebugEnabled()) {
078        LOG.debug("Write stripe metadata for " + writer.getPath().toString());
079      }
080      int index = existingWriters.indexOf(writer);
081      writer.appendFileInfo(StripeStoreFileManager.STRIPE_START_KEY, boundaries.get(index));
082      writer.appendFileInfo(StripeStoreFileManager.STRIPE_END_KEY, boundaries.get(index + 1));
083    } else {
084      if (LOG.isDebugEnabled()) {
085        LOG.debug("Skip writing stripe metadata for " + writer.getPath().toString());
086      }
087    }
088  }
089
090  /**
091   * Subclasses can call this method to make sure the first KV is within multi-writer range.
092   * @param left The left boundary of the writer.
093   * @param cell The cell whose row has to be checked.
094   */
095  protected void sanityCheckLeft(byte[] left, Cell cell) throws IOException {
096    if (!Arrays.equals(StripeStoreFileManager.OPEN_KEY, left)
097        && comparator.compareRows(cell, left, 0, left.length) < 0) {
098      String error =
099          "The first row is lower than the left boundary of [" + Bytes.toString(left) + "]: ["
100              + Bytes.toString(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()) + "]";
101      LOG.error(error);
102      throw new IOException(error);
103    }
104  }
105
106  /**
107   * Subclasses can call this method to make sure the last KV is within multi-writer range.
108   * @param right The right boundary of the writer.
109   */
110  protected void sanityCheckRight(byte[] right, Cell cell) throws IOException {
111    if (!Arrays.equals(StripeStoreFileManager.OPEN_KEY, right)
112        && comparator.compareRows(cell, right, 0, right.length) >= 0) {
113      String error =
114          "The last row is higher or equal than the right boundary of [" + Bytes.toString(right)
115              + "]: ["
116              + Bytes.toString(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()) + "]";
117      LOG.error(error);
118      throw new IOException(error);
119    }
120  }
121
122  /**
123   * MultiWriter that separates the cells based on fixed row-key boundaries. All the KVs between
124   * each pair of neighboring boundaries from the list supplied to ctor will end up in one file, and
125   * separate from all other such pairs.
126   */
127  public static class BoundaryMultiWriter extends StripeMultiFileWriter {
128    private StoreFileWriter currentWriter;
129    private byte[] currentWriterEndKey;
130
131    private Cell lastCell;
132    private long cellsInCurrentWriter = 0;
133    private int majorRangeFromIndex = -1, majorRangeToIndex = -1;
134    private boolean hasAnyWriter = false;
135
136    /**
137     * @param targetBoundaries The boundaries on which writers/files are separated.
138     * @param majorRangeFrom Major range is the range for which at least one file should be written
139     *          (because all files are included in compaction). majorRangeFrom is the left boundary.
140     * @param majorRangeTo The right boundary of majorRange (see majorRangeFrom).
141     */
142    public BoundaryMultiWriter(CellComparator comparator, List<byte[]> targetBoundaries,
143        byte[] majorRangeFrom, byte[] majorRangeTo) throws IOException {
144      super(comparator);
145      this.boundaries = targetBoundaries;
146      this.existingWriters = new ArrayList<StoreFileWriter>(this.boundaries.size() - 1);
147      // "major" range (range for which all files are included) boundaries, if any,
148      // must match some target boundaries, let's find them.
149      assert (majorRangeFrom == null) == (majorRangeTo == null);
150      if (majorRangeFrom != null) {
151        majorRangeFromIndex =
152            Arrays.equals(majorRangeFrom, StripeStoreFileManager.OPEN_KEY) ? 0 : Collections
153                .binarySearch(boundaries, majorRangeFrom, Bytes.BYTES_COMPARATOR);
154        majorRangeToIndex =
155            Arrays.equals(majorRangeTo, StripeStoreFileManager.OPEN_KEY) ? boundaries.size()
156                : Collections.binarySearch(boundaries, majorRangeTo, Bytes.BYTES_COMPARATOR);
157        if (this.majorRangeFromIndex < 0 || this.majorRangeToIndex < 0) {
158          throw new IOException("Major range does not match writer boundaries: ["
159              + Bytes.toString(majorRangeFrom) + "] [" + Bytes.toString(majorRangeTo) + "]; from "
160              + majorRangeFromIndex + " to " + majorRangeToIndex);
161        }
162      }
163    }
164
165    @Override
166    public void append(Cell cell) throws IOException {
167      if (currentWriter == null && existingWriters.isEmpty()) {
168        // First append ever, do a sanity check.
169        sanityCheckLeft(this.boundaries.get(0), cell);
170      }
171      prepareWriterFor(cell);
172      currentWriter.append(cell);
173      lastCell = cell; // for the sanity check
174      ++cellsInCurrentWriter;
175    }
176
177    private boolean isCellAfterCurrentWriter(Cell cell) {
178      return !Arrays.equals(currentWriterEndKey, StripeStoreFileManager.OPEN_KEY)
179        && (comparator.compareRows(cell, currentWriterEndKey, 0, currentWriterEndKey.length) >= 0);
180    }
181
182    @Override
183    protected void preCommitWritersInternal() throws IOException {
184      stopUsingCurrentWriter();
185      while (existingWriters.size() < boundaries.size() - 1) {
186        createEmptyWriter();
187      }
188      if (lastCell != null) {
189        sanityCheckRight(boundaries.get(boundaries.size() - 1), lastCell);
190      }
191    }
192
193    private void prepareWriterFor(Cell cell) throws IOException {
194      if (currentWriter != null && !isCellAfterCurrentWriter(cell)) return; // Use same writer.
195
196      stopUsingCurrentWriter();
197      // See if KV will be past the writer we are about to create; need to add another one.
198      while (isCellAfterCurrentWriter(cell)) {
199        checkCanCreateWriter();
200        createEmptyWriter();
201      }
202      checkCanCreateWriter();
203      hasAnyWriter = true;
204      currentWriter = writerFactory.createWriter();
205      existingWriters.add(currentWriter);
206    }
207
208    /**
209     * Called if there are no cells for some stripe. We need to have something in the writer list
210     * for this stripe, so that writer-boundary list indices correspond to each other. We can insert
211     * null in the writer list for that purpose, except in the following cases where we actually
212     * need a file: 1) If we are in range for which we are compacting all the files, we need to
213     * create an empty file to preserve stripe metadata. 2) If we have not produced any file at all
214     * for this compactions, and this is the last chance (the last stripe), we need to preserve last
215     * seqNum (see also HBASE-6059).
216     */
217    private void createEmptyWriter() throws IOException {
218      int index = existingWriters.size();
219      boolean isInMajorRange = (index >= majorRangeFromIndex) && (index < majorRangeToIndex);
220      // Stripe boundary count = stripe count + 1, so last stripe index is (#boundaries minus 2)
221      boolean isLastWriter = !hasAnyWriter && (index == (boundaries.size() - 2));
222      boolean needEmptyFile = isInMajorRange || isLastWriter;
223      existingWriters.add(needEmptyFile ? writerFactory.createWriter() : null);
224      hasAnyWriter |= needEmptyFile;
225      currentWriterEndKey =
226          (existingWriters.size() + 1 == boundaries.size()) ? null : boundaries.get(existingWriters
227              .size() + 1);
228    }
229
230    private void checkCanCreateWriter() throws IOException {
231      int maxWriterCount = boundaries.size() - 1;
232      assert existingWriters.size() <= maxWriterCount;
233      if (existingWriters.size() >= maxWriterCount) {
234        throw new IOException("Cannot create any more writers (created " + existingWriters.size()
235            + " out of " + maxWriterCount + " - row might be out of range of all valid writers");
236      }
237    }
238
239    private void stopUsingCurrentWriter() {
240      if (currentWriter != null) {
241        if (LOG.isDebugEnabled()) {
242          LOG.debug("Stopping to use a writer after [" + Bytes.toString(currentWriterEndKey)
243              + "] row; wrote out " + cellsInCurrentWriter + " kvs");
244        }
245        cellsInCurrentWriter = 0;
246      }
247      currentWriter = null;
248      currentWriterEndKey =
249          (existingWriters.size() + 1 == boundaries.size()) ? null : boundaries.get(existingWriters
250              .size() + 1);
251    }
252  }
253
254  /**
255   * MultiWriter that separates the cells based on target cell number per file and file count. New
256   * file is started every time the target number of KVs is reached, unless the fixed count of
257   * writers has already been created (in that case all the remaining KVs go into the last writer).
258   */
259  public static class SizeMultiWriter extends StripeMultiFileWriter {
260    private int targetCount;
261    private long targetCells;
262    private byte[] left;
263    private byte[] right;
264
265    private Cell lastCell;
266    private StoreFileWriter currentWriter;
267    protected byte[] lastRowInCurrentWriter = null;
268    private long cellsInCurrentWriter = 0;
269    private long cellsSeen = 0;
270    private long cellsSeenInPrevious = 0;
271
272    /**
273     * @param targetCount The maximum count of writers that can be created.
274     * @param targetKvs The number of KVs to read from source before starting each new writer.
275     * @param left The left boundary of the first writer.
276     * @param right The right boundary of the last writer.
277     */
278    public SizeMultiWriter(CellComparator comparator, int targetCount, long targetKvs, byte[] left,
279        byte[] right) {
280      super(comparator);
281      this.targetCount = targetCount;
282      this.targetCells = targetKvs;
283      this.left = left;
284      this.right = right;
285      int preallocate = Math.min(this.targetCount, 64);
286      this.existingWriters = new ArrayList<StoreFileWriter>(preallocate);
287      this.boundaries = new ArrayList<byte[]>(preallocate + 1);
288    }
289
290    @Override
291    public void append(Cell cell) throws IOException {
292      // If we are waiting for opportunity to close and we started writing different row,
293      // discard the writer and stop waiting.
294      boolean doCreateWriter = false;
295      if (currentWriter == null) {
296        // First append ever, do a sanity check.
297        sanityCheckLeft(left, cell);
298        doCreateWriter = true;
299      } else if (lastRowInCurrentWriter != null
300          && !CellUtil.matchingRow(cell, lastRowInCurrentWriter, 0,
301            lastRowInCurrentWriter.length)) {
302        if (LOG.isDebugEnabled()) {
303          LOG.debug("Stopping to use a writer after [" + Bytes.toString(lastRowInCurrentWriter)
304              + "] row; wrote out " + cellsInCurrentWriter + " kvs");
305        }
306        lastRowInCurrentWriter = null;
307        cellsInCurrentWriter = 0;
308        cellsSeenInPrevious += cellsSeen;
309        doCreateWriter = true;
310      }
311      if (doCreateWriter) {
312        // make a copy
313        byte[] boundary = existingWriters.isEmpty() ? left : CellUtil.cloneRow(cell);
314        if (LOG.isDebugEnabled()) {
315          LOG.debug("Creating new writer starting at [" + Bytes.toString(boundary) + "]");
316        }
317        currentWriter = writerFactory.createWriter();
318        boundaries.add(boundary);
319        existingWriters.add(currentWriter);
320      }
321
322      currentWriter.append(cell);
323      lastCell = cell; // for the sanity check
324      ++cellsInCurrentWriter;
325      cellsSeen = cellsInCurrentWriter;
326      if (this.sourceScanner != null) {
327        cellsSeen =
328            Math.max(cellsSeen, this.sourceScanner.getEstimatedNumberOfKvsScanned()
329                - cellsSeenInPrevious);
330      }
331
332      // If we are not already waiting for opportunity to close, start waiting if we can
333      // create any more writers and if the current one is too big.
334      if (lastRowInCurrentWriter == null && existingWriters.size() < targetCount
335          && cellsSeen >= targetCells) {
336        lastRowInCurrentWriter = CellUtil.cloneRow(cell); // make a copy
337        if (LOG.isDebugEnabled()) {
338          LOG.debug("Preparing to start a new writer after ["
339              + Bytes.toString(lastRowInCurrentWriter) + "] row; observed " + cellsSeen
340              + " kvs and wrote out " + cellsInCurrentWriter + " kvs");
341        }
342      }
343    }
344
345    @Override
346    protected void preCommitWritersInternal() throws IOException {
347      if (LOG.isDebugEnabled()) {
348        LOG.debug("Stopping with "
349            + cellsInCurrentWriter
350            + " kvs in last writer"
351            + ((this.sourceScanner == null) ? "" : ("; observed estimated "
352                + this.sourceScanner.getEstimatedNumberOfKvsScanned() + " KVs total")));
353      }
354      if (lastCell != null) {
355        sanityCheckRight(right, lastCell);
356      }
357
358      // When expired stripes were going to be merged into one, and if no writer was created during
359      // the compaction, we need to create an empty file to preserve metadata.
360      if (existingWriters.isEmpty() && 1 == targetCount) {
361        if (LOG.isDebugEnabled()) {
362          LOG.debug("Merge expired stripes into one, create an empty file to preserve metadata.");
363        }
364        boundaries.add(left);
365        existingWriters.add(writerFactory.createWriter());
366      }
367
368      this.boundaries.add(right);
369    }
370  }
