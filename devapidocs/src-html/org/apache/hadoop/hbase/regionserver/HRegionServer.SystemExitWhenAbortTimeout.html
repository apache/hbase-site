519      this.flushCheckFrequency = conf.getInt(PERIOD_FLUSH, this.threadWakeFrequency);

522      this.nonceManager = isNoncesEnabled ? new ServerNonceManager(this.conf) : null;

528        HConstants.DEFAULT_HBASE_RPC_SHORTOPERATION_RETRY_PAUSE_TIME);
529
530      regionServerAccounting = new RegionServerAccounting(conf);
531
532      blockCache = BlockCacheFactory.createBlockCache(conf);
533      mobFileCache = new MobFileCache(conf);
534
535      rsSnapshotVerifier = new RSSnapshotVerifier(conf);
536
537      uncaughtExceptionHandler =
538        (t, e) -> abort("Uncaught exception in executorService thread " + t.getName(), e);

540      // If no master in cluster, skip trying to track one or look for a cluster status.

547      this.rpcServices.start(zooKeeper);

551      // cause of failed startup is lost.
552      TraceUtil.setError(span, t);
553      LOG.error("Failed construction RegionServer", t);
554      throw t;
555    } finally {
556      span.end();
557    }

704  /**
705   * Bring up connection to zk ensemble and then wait until a master for this cluster and then after
706   * that, wait until cluster 'up' flag has been set. This is the order in which master does things.
707   * <p>
708   * Finally open long-living server short-circuit connection.
709   */
710  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "RV_RETURN_VALUE_IGNORED_BAD_PRACTICE",
711      justification = "cluster Id znode read would give us correct response")
712  private void initializeZooKeeper() throws IOException, InterruptedException {
713    // Nothing to do in here if no Master in the mix.
714    if (this.masterless) {
715      return;
716    }
717
718    // Create the master address tracker, register with zk, and start it. Then
719    // block until a master is available. No point in starting up if no master
720    // running.
721    blockAndCheckIfStopped(this.masterAddressTracker);
722
723    // Wait on cluster being up. Master will set this flag up in zookeeper
724    // when ready.
725    blockAndCheckIfStopped(this.clusterStatusTracker);
726
727    // If we are HMaster then the cluster id should have already been set.
728    if (clusterId == null) {
729      // Retrieve clusterId
730      // Since cluster status is now up
731      // ID should have already been set by HMaster
732      try {
733        clusterId = ZKClusterId.readClusterIdZNode(this.zooKeeper);
734        if (clusterId == null) {
735          this.abort("Cluster ID has not been set");
736        }
737        LOG.info("ClusterId : " + clusterId);
738      } catch (KeeperException e) {
739        this.abort("Failed to retrieve Cluster ID", e);
740      }
741    }
742
743    if (isStopped() || isAborted()) {
744      return; // No need for further initialization
745    }
746
747    // watch for snapshots and other procedures
748    try {
749      rspmHost = new RegionServerProcedureManagerHost();
750      rspmHost.loadProcedures(conf);
751      rspmHost.initialize(this);
752    } catch (KeeperException e) {
753      this.abort("Failed to reach coordination cluster when creating procedure handler.", e);
754    }
755  }

759   * is shut down
760   * @param tracker znode tracker to use
761   * @throws IOException          any IO exception, plus if the RS is stopped
762   * @throws InterruptedException if the waiting thread is interrupted

768        throw new IOException("Received the shutdown message while waiting.");
769      }
770    }

796  /**
797   * The HRegionServer sticks in this loop until closed.
798   */
799  @Override
800  public void run() {
801    if (isStopped()) {
802      LOG.info("Skipping run; stopped");
803      return;
804    }
805    try {
806      // Do pre-registration initializations; zookeeper, lease threads, etc.
807      preRegistrationInitialization();
808    } catch (Throwable e) {
809      abort("Fatal exception during initialization", e);
810    }
811
812    try {
813      if (!isStopped() && !isAborted()) {
814        installShutdownHook();
815        // Initialize the RegionServerCoprocessorHost now that our ephemeral
816        // node was created, in case any coprocessors want to use ZooKeeper
817        this.rsHost = new RegionServerCoprocessorHost(this, this.conf);
818
819        // Try and register with the Master; tell it we are here. Break if server is stopped or
820        // the clusterup flag is down or hdfs went wacky. Once registered successfully, go ahead and
821        // start up all Services. Use RetryCounter to get backoff in case Master is struggling to
822        // come up.
823        LOG.debug("About to register with Master.");
824        TraceUtil.trace(() -> {
825          RetryCounterFactory rcf =
826            new RetryCounterFactory(Integer.MAX_VALUE, this.sleeper.getPeriod(), 1000 * 60 * 5);
827          RetryCounter rc = rcf.create();
828          while (keepLooping()) {
829            RegionServerStartupResponse w = reportForDuty();
830            if (w == null) {
831              long sleepTime = rc.getBackoffTimeAndIncrementAttempts();
832              LOG.warn("reportForDuty failed; sleeping {} ms and then retrying.", sleepTime);
833              this.sleeper.sleep(sleepTime);
834            } else {
835              handleReportForDutyResponse(w);
836              break;
837            }
838          }
839        }, "HRegionServer.registerWithMaster");
840      }
841
842      if (!isStopped() && isHealthy()) {
843        TraceUtil.trace(() -> {
844          // start the snapshot handler and other procedure handlers,
845          // since the server is ready to run
846          if (this.rspmHost != null) {
847            this.rspmHost.start();
848          }
849          // Start the Quota Manager
850          if (this.rsQuotaManager != null) {
851            rsQuotaManager.start(getRpcServer().getScheduler());
852          }
853          if (this.rsSpaceQuotaManager != null) {
854            this.rsSpaceQuotaManager.start();
855          }
856        }, "HRegionServer.startup");
857      }
858
859      // We registered with the Master. Go into run mode.
860      long lastMsg = EnvironmentEdgeManager.currentTime();
861      long oldRequestCount = -1;
862      // The main run loop.
863      while (!isStopped() && isHealthy()) {
864        if (!isClusterUp()) {
865          if (onlineRegions.isEmpty()) {
866            stop("Exiting; cluster shutdown set and not carrying any regions");
867          } else if (!this.stopping) {
868            this.stopping = true;
869            LOG.info("Closing user regions");
870            closeUserRegions(isAborted());
871          } else {
872            boolean allUserRegionsOffline = areAllUserRegionsOffline();
873            if (allUserRegionsOffline) {
874              // Set stopped if no more write requests tp meta tables
875              // since last time we went around the loop. Any open
876              // meta regions will be closed on our way out.
877              if (oldRequestCount == getWriteRequestCount()) {
878                stop("Stopped; only catalog regions remaining online");
879                break;
880              }
881              oldRequestCount = getWriteRequestCount();
882            } else {
883              // Make sure all regions have been closed -- some regions may
884              // have not got it because we were splitting at the time of
885              // the call to closeUserRegions.
886              closeUserRegions(this.abortRequested.get());
887            }
888            LOG.debug("Waiting on " + getOnlineRegionsAsPrintableString());
889          }
890        }
891        long now = EnvironmentEdgeManager.currentTime();
892        if ((now - lastMsg) >= msgInterval) {
893          tryRegionServerReport(lastMsg, now);
894          lastMsg = EnvironmentEdgeManager.currentTime();
895        }
896        if (!isStopped() && !isAborted()) {
897          this.sleeper.sleep();
898        }
899      } // for
900    } catch (Throwable t) {
901      if (!rpcServices.checkOOME(t)) {
902        String prefix = t instanceof YouAreDeadException ? "" : "Unhandled: ";
903        abort(prefix + t.getMessage(), t);
904      }
905    }
906
907    final Span span = TraceUtil.createSpan("HRegionServer exiting main loop");
908    try (Scope ignored = span.makeCurrent()) {
909      if (this.leaseManager != null) {
910        this.leaseManager.closeAfterLeasesExpire();
911      }
912      if (this.splitLogWorker != null) {
913        splitLogWorker.stop();
914      }
915      stopInfoServer();
916      // Send cache a shutdown.
917      if (blockCache != null) {
918        blockCache.shutdown();
919      }
920      if (mobFileCache != null) {
921        mobFileCache.shutdown();
922      }
923
924      // Send interrupts to wake up threads if sleeping so they notice shutdown.
925      // TODO: Should we check they are alive? If OOME could have exited already
926      if (this.hMemManager != null) {
927        this.hMemManager.stop();
928      }
929      if (this.cacheFlusher != null) {
930        this.cacheFlusher.interruptIfNecessary();
931      }
932      if (this.compactSplitThread != null) {
933        this.compactSplitThread.interruptIfNecessary();
934      }
935
936      // Stop the snapshot and other procedure handlers, forcefully killing all running tasks
937      if (rspmHost != null) {
938        rspmHost.stop(this.abortRequested.get() || this.killed);
939      }
940
941      if (this.killed) {
942        // Just skip out w/o closing regions. Used when testing.
943      } else if (abortRequested.get()) {
944        if (this.dataFsOk) {
945          closeUserRegions(abortRequested.get()); // Don't leave any open file handles
946        }
947        LOG.info("aborting server " + this.serverName);
948      } else {
949        closeUserRegions(abortRequested.get());
950        LOG.info("stopping server " + this.serverName);
951      }
952      regionReplicationBufferManager.stop();
953      closeClusterConnection();
954      // Closing the compactSplit thread before closing meta regions
955      if (!this.killed && containsMetaTableRegions()) {
956        if (!abortRequested.get() || this.dataFsOk) {
957          if (this.compactSplitThread != null) {
958            this.compactSplitThread.join();
959            this.compactSplitThread = null;
960          }
961          closeMetaTableRegions(abortRequested.get());
962        }
963      }
964
965      if (!this.killed && this.dataFsOk) {
966        waitOnAllRegionsToClose(abortRequested.get());
967        LOG.info("stopping server " + this.serverName + "; all regions closed.");
968      }
969
970      // Stop the quota manager
971      if (rsQuotaManager != null) {
972        rsQuotaManager.stop();
973      }
974      if (rsSpaceQuotaManager != null) {
975        rsSpaceQuotaManager.stop();
976        rsSpaceQuotaManager = null;
977      }
978
979      // flag may be changed when closing regions throws exception.
980      if (this.dataFsOk) {
981        shutdownWAL(!abortRequested.get());
982      }
983
984      // Make sure the proxy is down.
985      if (this.rssStub != null) {
986        this.rssStub = null;
987      }
988      if (this.lockStub != null) {
989        this.lockStub = null;
990      }
991      if (this.rpcClient != null) {
992        this.rpcClient.close();
993      }
994      if (this.leaseManager != null) {
995        this.leaseManager.close();
996      }
997      if (this.pauseMonitor != null) {
998        this.pauseMonitor.stop();
999      }
1000
1001      if (!killed) {
1002        stopServiceThreads();
1003      }
1004
1005      if (this.rpcServices != null) {
1006        this.rpcServices.stop();
1007      }
1008
1009      try {
1010        deleteMyEphemeralNode();
1011      } catch (KeeperException.NoNodeException nn) {
1012        // pass
1013      } catch (KeeperException e) {
1014        LOG.warn("Failed deleting my ephemeral node", e);
1015      }
1016      // We may have failed to delete the znode at the previous step, but
1017      // we delete the file anyway: a second attempt to delete the znode is likely to fail again.
1018      ZNodeClearer.deleteMyEphemeralNodeOnDisk();
1019
1020      closeZooKeeper();
1021      closeTableDescriptors();
1022      LOG.info("Exiting; stopping=" + this.serverName + "; zookeeper connection closed.");
1023      span.setStatus(StatusCode.OK);
1024    } finally {
1025      span.end();
1026    }
1027  }

1056  @InterfaceAudience.Private
1057  protected void tryRegionServerReport(long reportStartTime, long reportEndTime)
1058    throws IOException {
1059    RegionServerStatusService.BlockingInterface rss = rssStub;
1060    if (rss == null) {
1061      // the current server could be stopping.
1062      return;
1063    }
1064    ClusterStatusProtos.ServerLoad sl = buildServerLoad(reportStartTime, reportEndTime);
1065    final Span span = TraceUtil.createSpan("HRegionServer.tryRegionServerReport");
1066    try (Scope ignored = span.makeCurrent()) {
1067      RegionServerReportRequest.Builder request = RegionServerReportRequest.newBuilder();
1068      request.setServer(ProtobufUtil.toServerName(this.serverName));
1069      request.setLoad(sl);
1070      rss.regionServerReport(null, request.build());
1071      span.setStatus(StatusCode.OK);
1072    } catch (ServiceException se) {
1073      IOException ioe = ProtobufUtil.getRemoteException(se);
1074      if (ioe instanceof YouAreDeadException) {
1075        // This will be caught and handled as a fatal error in run()
1076        TraceUtil.setError(span, ioe);
1077        throw ioe;
1078      }
1079      if (rssStub == rss) {
1080        rssStub = null;
1081      }
1082      TraceUtil.setError(span, se);
1083      // Couldn't connect to the master, get location from zk and reconnect
1084      // Method blocks until new master is found or we are stopped
1085      createRegionServerStatusStub(true);
1086    } finally {
1087      span.end();
1088    }
1089  }

1091  /**
1092   * Reports the given map of Regions and their size on the filesystem to the active Master.
1093   * @param regionSizeStore The store containing region sizes
1094   * @return false if FileSystemUtilizationChore should pause reporting to master. true otherwise
1095   */
1096  public boolean reportRegionSizesForQuotas(RegionSizeStore regionSizeStore) {
1097    RegionServerStatusService.BlockingInterface rss = rssStub;
1098    if (rss == null) {
1099      // the current server could be stopping.
1100      LOG.trace("Skipping Region size report to HMaster as stub is null");
1101      return true;
1102    }
1103    try {
1104      buildReportAndSend(rss, regionSizeStore);
1105    } catch (ServiceException se) {
1106      IOException ioe = ProtobufUtil.getRemoteException(se);
1107      if (ioe instanceof PleaseHoldException) {
1108        LOG.trace("Failed to report region sizes to Master because it is initializing."
1109          + " This will be retried.", ioe);
1110        // The Master is coming up. Will retry the report later. Avoid re-creating the stub.
1111        return true;
1112      }
1113      if (rssStub == rss) {
1114        rssStub = null;
1115      }
1116      createRegionServerStatusStub(true);
1117      if (ioe instanceof DoNotRetryIOException) {
1118        DoNotRetryIOException doNotRetryEx = (DoNotRetryIOException) ioe;
1119        if (doNotRetryEx.getCause() != null) {
1120          Throwable t = doNotRetryEx.getCause();
1121          if (t instanceof UnsupportedOperationException) {
1122            LOG.debug("master doesn't support ReportRegionSpaceUse, pause before retrying");
1123            return false;
1124          }
1125        }
1126      }
1127      LOG.debug("Failed to report region sizes to Master. This will be retried.", ioe);
1128    }
1129    return true;
1130  }

1133   * Builds the region size report and sends it to the master. Upon successful sending of the
1134   * report, the region sizes that were sent are marked as sent.
1135   * @param rss             The stub to send to the Master
1136   * @param regionSizeStore The store containing region sizes

1138  private void buildReportAndSend(RegionServerStatusService.BlockingInterface rss,

1140    RegionSpaceUseReportRequest request =
1141      buildRegionSpaceUseReportRequest(Objects.requireNonNull(regionSizeStore));

1290  /**
1291   * Wait on regions close.
1292   */
1293  private void waitOnAllRegionsToClose(final boolean abort) {
1294    // Wait till all regions are closed before going out.
1295    int lastCount = -1;
1296    long previousLogTime = 0;
1297    Set<String> closedRegions = new HashSet<>();
1298    boolean interrupted = false;
1299    try {
1300      while (!onlineRegions.isEmpty()) {
1301        int count = getNumberOfOnlineRegions();
1302        // Only print a message if the count of regions has changed.
1303        if (count != lastCount) {
1304          // Log every second at most
1305          if (EnvironmentEdgeManager.currentTime() > (previousLogTime + 1000)) {
1306            previousLogTime = EnvironmentEdgeManager.currentTime();
1307            lastCount = count;
1308            LOG.info("Waiting on " + count + " regions to close");
1309            // Only print out regions still closing if a small number else will
1310            // swamp the log.
1311            if (count < 10 && LOG.isDebugEnabled()) {
1312              LOG.debug("Online Regions=" + this.onlineRegions);
1313            }
1314          }
1315        }
1316        // Ensure all user regions have been sent a close. Use this to
1317        // protect against the case where an open comes in after we start the
1318        // iterator of onlineRegions to close all user regions.
1319        for (Map.Entry<String, HRegion> e : this.onlineRegions.entrySet()) {
1320          RegionInfo hri = e.getValue().getRegionInfo();
1321          if (
1322            !this.regionsInTransitionInRS.containsKey(hri.getEncodedNameAsBytes())
1323              && !closedRegions.contains(hri.getEncodedName())
1324          ) {
1325            closedRegions.add(hri.getEncodedName());
1326            // Don't update zk with this close transition; pass false.
1327            closeRegionIgnoreErrors(hri, abort);
1328          }
1329        }
1330        // No regions in RIT, we could stop waiting now.
1331        if (this.regionsInTransitionInRS.isEmpty()) {
1332          if (!onlineRegions.isEmpty()) {
1333            LOG.info("We were exiting though online regions are not empty,"
1334              + " because some regions failed closing");
1335          }
1336          break;
1337        } else {
1338          LOG.debug("Waiting on {}", this.regionsInTransitionInRS.keySet().stream()
1339            .map(e -> Bytes.toString(e)).collect(Collectors.joining(", ")));
1340        }
1341        if (sleepInterrupted(200)) {
1342          interrupted = true;
1343        }
1344      }
1345    } finally {
1346      if (interrupted) {
1347        Thread.currentThread().interrupt();
1348      }
1349    }
1350  }

1357      LOG.warn("Interrupted while sleeping");

1380   * Run init. Sets up wal and starts up all server threads.
1381   * @param c Extra configuration.
1382   */
1383  protected void handleReportForDutyResponse(final RegionServerStartupResponse c)
1384    throws IOException {
1385    try {
1386      boolean updateRootDir = false;
1387      for (NameStringPair e : c.getMapEntriesList()) {
1388        String key = e.getName();
1389        // The hostname the master sees us as.
1390        if (key.equals(HConstants.KEY_FOR_HOSTNAME_SEEN_BY_MASTER)) {
1391          String hostnameFromMasterPOV = e.getValue();
1392          this.serverName = ServerName.valueOf(hostnameFromMasterPOV,
1393            rpcServices.getSocketAddress().getPort(), this.startcode);
1394          String expectedHostName = rpcServices.getSocketAddress().getHostName();
1395          // if Master use-ip is enabled, RegionServer use-ip will be enabled by default even if it
1396          // is set to disable. so we will use the ip of the RegionServer to compare with the
1397          // hostname passed by the Master, see HBASE-27304 for details.
1398          if (
1399            StringUtils.isBlank(useThisHostnameInstead) && getActiveMaster().isPresent()
1400              && InetAddresses.isInetAddress(getActiveMaster().get().getHostname())
1401          ) {
1402            expectedHostName = rpcServices.getSocketAddress().getAddress().getHostAddress();
1403          }
1404          boolean isHostnameConsist = StringUtils.isBlank(useThisHostnameInstead)
1405            ? hostnameFromMasterPOV.equals(expectedHostName)
1406            : hostnameFromMasterPOV.equals(useThisHostnameInstead);
1407          if (!isHostnameConsist) {
1408            String msg = "Master passed us a different hostname to use; was="
1409              + (StringUtils.isBlank(useThisHostnameInstead)
1410                ? rpcServices.getSocketAddress().getHostName()
1411                : this.useThisHostnameInstead)
1412              + ", but now=" + hostnameFromMasterPOV;
1413            LOG.error(msg);
1414            throw new IOException(msg);
1415          }
1416          continue;
1417        }
1418
1419        String value = e.getValue();
1420        if (key.equals(HConstants.HBASE_DIR)) {
1421          if (value != null && !value.equals(conf.get(HConstants.HBASE_DIR))) {
1422            updateRootDir = true;
1423          }
1424        }
1425
1426        if (LOG.isDebugEnabled()) {
1427          LOG.debug("Config from master: " + key + "=" + value);
1428        }
1429        this.conf.set(key, value);
1430      }
1431      // Set our ephemeral znode up in zookeeper now we have a name.
1432      createMyEphemeralNode();
1433
1434      if (updateRootDir) {
1435        // initialize file system by the config fs.defaultFS and hbase.rootdir from master
1436        initializeFileSystem();
1437      }
1438
1439      // hack! Maps DFSClient => RegionServer for logs. HDFS made this
1440      // config param for task trackers, but we can piggyback off of it.
1441      if (this.conf.get("mapreduce.task.attempt.id") == null) {
1442        this.conf.set("mapreduce.task.attempt.id", "hb_rs_" + this.serverName.toString());
1443      }
1444
1445      // Save it in a file, this will allow to see if we crash
1446      ZNodeClearer.writeMyEphemeralNodeOnDisk(getMyEphemeralNodePath());
1447
1448      // This call sets up an initialized replication and WAL. Later we start it up.
1449      setupWALAndReplication();
1450      // Init in here rather than in constructor after thread name has been set
1451      final MetricsTable metricsTable =
1452        new MetricsTable(new MetricsTableWrapperAggregateImpl(this));
1453      this.metricsRegionServerImpl = new MetricsRegionServerWrapperImpl(this);
1454      this.metricsRegionServer =
1455        new MetricsRegionServer(metricsRegionServerImpl, conf, metricsTable);
1456      // Now that we have a metrics source, start the pause monitor
1457      this.pauseMonitor = new JvmPauseMonitor(conf, getMetrics().getMetricsSource());
1458      pauseMonitor.start();
1459
1460      // There is a rare case where we do NOT want services to start. Check config.
1461      if (getConfiguration().getBoolean("hbase.regionserver.workers", true)) {
1462        startServices();
1463      }
1464      // In here we start up the replication Service. Above we initialized it. TODO. Reconcile.
1465      // or make sense of it.
1466      startReplicationService();
1467
1468      // Set up ZK
1469      LOG.info("Serving as " + this.serverName + ", RpcServer on " + rpcServices.getSocketAddress()
1470        + ", sessionid=0x"
1471        + Long.toHexString(this.zooKeeper.getRecoverableZooKeeper().getSessionId()));
1472
1473      // Wake up anyone waiting for this server to online
1474      synchronized (online) {
1475        online.set(true);
1476        online.notifyAll();
1477      }
1478    } catch (Throwable e) {
1479      stop("Failed initialization");
1480      throw convertThrowableToIOE(cleanup(e, "Failed init"), "Region server startup failed");
1481    } finally {
1482      sleeper.skipSleepCycle();
1483    }
1484  }

1511  // Round the size with KB or MB.
1512  // A trick here is that if the sizeInBytes is less than sizeUnit, we will round the size to 1
1513  // instead of 0 if it is not 0, to avoid some schedulers think the region has no data. See
1514  // HBASE-26340 for more details on why this is important.
1515  private static int roundSize(long sizeInByte, int sizeUnit) {
1516    if (sizeInByte == 0) {
1517      return 0;
1518    } else if (sizeInByte < sizeUnit) {
1519      return 1;
1520    } else {
1521      return (int) Math.min(sizeInByte / sizeUnit, Integer.MAX_VALUE);
1522    }
1523  }

1599    HDFSBlocksDistribution hdfsBd = r.getHDFSBlocksDistribution();
1600    float dataLocality = hdfsBd.getBlockLocalityIndex(serverName.getHostname());
1601    float dataLocalityForSsd = hdfsBd.getBlockLocalityIndexForSsd(serverName.getHostname());
1602    long blocksTotalWeight = hdfsBd.getUniqueBlocksTotalWeight();
1603    long blocksLocalWeight = hdfsBd.getBlocksLocalWeight(serverName.getHostname());
1604    long blocksLocalWithSsdWeight = hdfsBd.getBlocksLocalWithSsdWeight(serverName.getHostname());
1605    if (regionLoadBldr == null) {
1606      regionLoadBldr = RegionLoad.newBuilder();
1607    }
1608    if (regionSpecifier == null) {
1609      regionSpecifier = RegionSpecifier.newBuilder();
1610    }

1613    regionSpecifier.setValue(UnsafeByteOperations.unsafeWrap(name));

1615      .setStorefiles(storefiles).setStoreRefCount(storeRefCount)
1616      .setMaxCompactedStoreFileRefCount(maxCompactedStoreFileRefCount)
1617      .setStoreUncompressedSizeMB(storeUncompressedSizeMB).setStorefileSizeMB(storefileSizeMB)
1618      .setMemStoreSizeMB(memstoreSizeMB).setStorefileIndexSizeKB(storefileIndexSizeKB)
1619      .setRootIndexSizeKB(rootLevelIndexSizeKB).setTotalStaticIndexSizeKB(totalStaticIndexSizeKB)
1620      .setTotalStaticBloomSizeKB(totalStaticBloomSizeKB)
1621      .setReadRequestsCount(r.getReadRequestsCount()).setCpRequestsCount(r.getCpRequestsCount())
1622      .setFilteredReadRequestsCount(r.getFilteredReadRequestsCount())
1623      .setWriteRequestsCount(r.getWriteRequestsCount()).setTotalCompactingKVs(totalCompactingKVs)
1624      .setCurrentCompactedKVs(currentCompactedKVs).setDataLocality(dataLocality)
1625      .setDataLocalityForSsd(dataLocalityForSsd).setBlocksLocalWeight(blocksLocalWeight)
1626      .setBlocksLocalWithSsdWeight(blocksLocalWithSsdWeight).setBlocksTotalWeight(blocksTotalWeight)
1627      .setCompactionState(ProtobufUtil.createCompactionStateForRegionLoad(r.getCompactionState()))
1628      .setLastMajorCompactionTs(r.getOldestHfileTs(true)).setRegionSizeMB(regionSizeMB)
1629      .setCurrentRegionCachedRatio(currentRegionCachedRatio.floatValue());
1630    r.setCompleteSequenceId(regionLoadBldr);
1631    return regionLoadBldr.build();

1670       */
1671      this.majorCompactPriority = this.instance.conf
1672        .getInt("hbase.regionserver.compactionChecker.majorCompactPriority", DEFAULT_PRIORITY);
1673    }
1674
1675    @Override
1676    protected void chore() {
1677      for (Region r : this.instance.onlineRegions.values()) {
1678        // If region is read only or compaction is disabled at table level, there's no need to
1679        // iterate through region's stores
1680        if (r == null || r.isReadOnly() || !r.getTableDescriptor().isCompactionEnabled()) {
1681          continue;
1682        }
1683
1684        HRegion hr = (HRegion) r;
1685        for (HStore s : hr.stores.values()) {
1686          try {
1687            long multiplier = s.getCompactionCheckMultiplier();
1688            assert multiplier > 0;
1689            if (iteration % multiplier != 0) {
1690              continue;
1691            }
1692            if (s.needsCompaction()) {
1693              // Queue a compaction. Will recognize if major is needed.
1694              this.instance.compactSplitThread.requestSystemCompaction(hr, s,
1695                getName() + " requests compaction");
1696            } else if (s.shouldPerformMajorCompaction()) {
1697              s.triggerMajorCompaction();
1698              if (
1699                majorCompactPriority == DEFAULT_PRIORITY
1700                  || majorCompactPriority > hr.getCompactPriority()
1701              ) {
1702                this.instance.compactSplitThread.requestCompaction(hr, s,
1703                  getName() + " requests major compaction; use default priority", Store.NO_PRIORITY,
1704                  CompactionLifeCycleTracker.DUMMY, null);
1705              } else {
1706                this.instance.compactSplitThread.requestCompaction(hr, s,
1707                  getName() + " requests major compaction; use configured priority",
1708                  this.majorCompactPriority, CompactionLifeCycleTracker.DUMMY, null);
1709              }
1710            }
1711          } catch (IOException e) {
1712            LOG.warn("Failed major compaction check on " + r, e);
1713          }
1714        }
1715      }
1716      iteration = (iteration == Long.MAX_VALUE) ? 0 : (iteration + 1);
1717    }
1718  }

1720  private static class PeriodicMemStoreFlusher extends ScheduledChore {
1721    private final HRegionServer server;
1722    private final static int RANGE_OF_DELAY = 5 * 60; // 5 min in seconds
1723    private final static int MIN_DELAY_TIME = 0; // millisec
1724    private final long rangeOfDelayMs;
1725
1726    PeriodicMemStoreFlusher(int cacheFlushInterval, final HRegionServer server) {
1727      super("MemstoreFlusherChore", server, cacheFlushInterval);
1728      this.server = server;
1729
1730      final long configuredRangeOfDelay = server.getConfiguration()
1731        .getInt("hbase.regionserver.periodicmemstoreflusher.rangeofdelayseconds", RANGE_OF_DELAY);
1732      this.rangeOfDelayMs = TimeUnit.SECONDS.toMillis(configuredRangeOfDelay);
1733    }
1734
1735    @Override
1736    protected void chore() {
1737      final StringBuilder whyFlush = new StringBuilder();
1738      for (HRegion r : this.server.onlineRegions.values()) {
1739        if (r == null) {
1740          continue;
1741        }
1742        if (r.shouldFlush(whyFlush)) {
1743          FlushRequester requester = server.getFlushRequester();
1744          if (requester != null) {
1745            long delay = ThreadLocalRandom.current().nextLong(rangeOfDelayMs) + MIN_DELAY_TIME;
1746            // Throttle the flushes by putting a delay. If we don't throttle, and there
1747            // is a balanced write-load on the regions in a table, we might end up
1748            // overwhelming the filesystem with too many flushes at once.
1749            if (requester.requestDelayedFlush(r, delay)) {
1750              LOG.info("{} requesting flush of {} because {} after random delay {} ms", getName(),
1751                r.getRegionInfo().getRegionNameAsString(), whyFlush.toString(), delay);
1752            }
1753          }
1754        }
1755      }
1756    }
1757  }

3011  /**
3012   * Close asynchronously a region, can be called from the master or internally by the regionserver
3013   * when stopping. If called from the master, the region will update the status.
3014   * <p>
3015   * If an opening was in progress, this method will cancel it, but will not start a new close. The
3016   * coprocessors are not called in this case. A NotServingRegionException exception is thrown.
3017   * </p>
3018   * <p>
3019   * If a close was in progress, this new request will be ignored, and an exception thrown.
3020   * </p>
3021   * <p>
3022   * Provides additional flag to indicate if this region blocks should be evicted from the cache.
3023   * </p>
3024   * @param encodedName Region to close
3025   * @param abort       True if we are aborting
3026   * @param destination Where the Region is being moved too... maybe null if unknown.
3027   * @return True if closed a region.
3028   * @throws NotServingRegionException if the region is not online
3029   */
3030  protected boolean closeRegion(String encodedName, final boolean abort,
3031    final ServerName destination) throws NotServingRegionException {
3032    // Check for permissions to close.
3033    HRegion actualRegion = this.getRegion(encodedName);
3034    // Can be null if we're calling close on a region that's not online
3035    if ((actualRegion != null) && (actualRegion.getCoprocessorHost() != null)) {
3036      try {
3037        actualRegion.getCoprocessorHost().preClose(false);
3038      } catch (IOException exp) {
3039        LOG.warn("Unable to close region: the coprocessor launched an error ", exp);
3040        return false;
3041      }
3042    }
3043
3044    // previous can come back 'null' if not in map.
3045    final Boolean previous =
3046      this.regionsInTransitionInRS.putIfAbsent(Bytes.toBytes(encodedName), Boolean.FALSE);
3047
3048    if (Boolean.TRUE.equals(previous)) {
3049      LOG.info("Received CLOSE for the region:" + encodedName + " , which we are already "
3050        + "trying to OPEN. Cancelling OPENING.");
3051      if (!regionsInTransitionInRS.replace(Bytes.toBytes(encodedName), previous, Boolean.FALSE)) {
3052        // The replace failed. That should be an exceptional case, but theoretically it can happen.
3053        // We're going to try to do a standard close then.
3054        LOG.warn("The opening for region " + encodedName + " was done before we could cancel it."
3055          + " Doing a standard close now");
3056        return closeRegion(encodedName, abort, destination);
3057      }
3058      // Let's get the region from the online region list again
3059      actualRegion = this.getRegion(encodedName);
3060      if (actualRegion == null) { // If already online, we still need to close it.
3061        LOG.info("The opening previously in progress has been cancelled by a CLOSE request.");
3062        // The master deletes the znode when it receives this exception.
3063        throw new NotServingRegionException(
3064          "The region " + encodedName + " was opening but not yet served. Opening is cancelled.");
3065      }
3066    } else if (previous == null) {
3067      LOG.info("Received CLOSE for {}", encodedName);
3068    } else if (Boolean.FALSE.equals(previous)) {
3069      LOG.info("Received CLOSE for the region: " + encodedName
3070        + ", which we are already trying to CLOSE, but not completed yet");
3071      return true;
3072    }
3073
3074    if (actualRegion == null) {
3075      LOG.debug("Received CLOSE for a region which is not online, and we're not opening.");
3076      this.regionsInTransitionInRS.remove(Bytes.toBytes(encodedName));
3077      // The master deletes the znode when it receives this exception.
3078      throw new NotServingRegionException(
3079        "The region " + encodedName + " is not online, and is not opening.");
3080    }
3081
3082    CloseRegionHandler crh;
3083    final RegionInfo hri = actualRegion.getRegionInfo();
3084    if (hri.isMetaRegion()) {
3085      crh = new CloseMetaHandler(this, this, hri, abort);
3086    } else {
3087      crh = new CloseRegionHandler(this, this, hri, abort, destination);
3088    }
3089    this.executorService.submit(crh);
3090    return true;
3091  }

3542  /**
3543   * Will ignore the open/close region procedures which already submitted or executed. When master
3544   * had unfinished open/close region procedure and restarted, new active master may send duplicate
3545   * open/close region request to regionserver. The open/close request is submitted to a thread pool
3546   * and execute. So first need a cache for submitted open/close region procedures. After the
3547   * open/close region request executed and report region transition succeed, cache it in executed
3548   * region procedures cache. See {@link #finishRegionProcedure(long)}. After report region
3549   * transition succeed, master will not send the open/close region request to regionserver again.
3550   * And we thought that the ongoing duplicate open/close region request should not be delayed more
3551   * than 600 seconds. So the executed region procedures cache will expire after 600 seconds. See
3552   * HBASE-22404 for more details.
3553   * @param procId the id of the open/close region procedure
3554   * @return true if the procedure can be submitted.
3555   */
3556  boolean submitRegionProcedure(long procId) {
3557    if (procId == -1) {
3558      return true;
3559    }
3560    // Ignore the region procedures which already submitted.
3561    Long previous = submittedRegionProcedures.putIfAbsent(procId, procId);
3562    if (previous != null) {
3563      LOG.warn("Received procedure pid={}, which already submitted, just ignore it", procId);
3564      return false;
3565    }
3566    // Ignore the region procedures which already executed.
3567    if (executedRegionProcedures.getIfPresent(procId) != null) {
3568      LOG.warn("Received procedure pid={}, which already executed, just ignore it", procId);
3569      return false;
3570    }
3571    return true;
3572  }
3573
3574  /**
3575   * See {@link #submitRegionProcedure(long)}.
3576   * @param procId the id of the open/close region procedure
3577   */
3578  public void finishRegionProcedure(long procId) {
3579    executedRegionProcedures.put(procId, procId);
3580    submittedRegionProcedures.remove(procId);
3581  }
