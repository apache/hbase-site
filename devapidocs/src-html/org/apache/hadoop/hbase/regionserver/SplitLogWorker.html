045/**
046 * This worker is spawned in every regionserver, including master. The Worker waits for log
047 * splitting tasks to be put up by the {@link org.apache.hadoop.hbase.master.SplitLogManager}
048 * running in the master and races with other workers in other serves to acquire those tasks.
049 * The coordination is done via coordination engine.
050 * <p>
051 * If a worker has successfully moved the task from state UNASSIGNED to OWNED then it owns the task.
052 * It keeps heart beating the manager by periodically moving the task from UNASSIGNED to OWNED
053 * state. On success it moves the task to TASK_DONE. On unrecoverable error it moves task state to
054 * ERR. If it cannot continue but wants the master to retry the task then it moves the task state to
055 * RESIGNED.
056 * <p>
057 * The manager can take a task away from a worker by moving the task from OWNED to UNASSIGNED. In
058 * the absence of a global lock there is a unavoidable race here - a worker might have just finished
059 * its task when it is stripped of its ownership. Here we rely on the idempotency of the log
060 * splitting task for correctness
061 */
062@InterfaceAudience.Private
063public class SplitLogWorker implements Runnable {
064
065  private static final Log LOG = LogFactory.getLog(SplitLogWorker.class);
066
067  Thread worker;
068  // thread pool which executes recovery work
069  private SplitLogWorkerCoordination coordination;
070  private Configuration conf;
071  private RegionServerServices server;
072
073  public SplitLogWorker(Server hserver, Configuration conf, RegionServerServices server,
074      TaskExecutor splitTaskExecutor) {
075    this.server = server;
076    this.conf = conf;
077    this.coordination = hserver.getCoordinatedStateManager().getSplitLogWorkerCoordination();
078    coordination.init(server, conf, splitTaskExecutor, this);
079  }
080
081  public SplitLogWorker(final Server hserver, final Configuration conf,
082      final RegionServerServices server, final LastSequenceId sequenceIdChecker,
083      final WALFactory factory) {
084    this(hserver, conf, server, new TaskExecutor() {
085      @Override
086      public Status exec(String filename, RecoveryMode mode, CancelableProgressable p) {
087        Path walDir;
088        FileSystem fs;
089        try {
090          walDir = FSUtils.getWALRootDir(conf);
091          fs = walDir.getFileSystem(conf);
092        } catch (IOException e) {
093          LOG.warn("could not find root dir or fs", e);
094          return Status.RESIGNED;
095        }
096        // TODO have to correctly figure out when log splitting has been
097        // interrupted or has encountered a transient error and when it has
098        // encountered a bad non-retry-able persistent error.
099        try {
100          if (!WALSplitter.splitLogFile(walDir, fs.getFileStatus(new Path(walDir, filename)),
101            fs, conf, p, sequenceIdChecker,
102              server.getCoordinatedStateManager().getSplitLogWorkerCoordination(),
103              server.getConnection(), mode, factory)) {
104            return Status.PREEMPTED;
105          }
106        } catch (InterruptedIOException iioe) {
107          LOG.warn("log splitting of " + filename + " interrupted, resigning", iioe);
108          return Status.RESIGNED;
109        } catch (IOException e) {
110          Throwable cause = e.getCause();
111          if (e instanceof RetriesExhaustedException && (cause instanceof NotServingRegionException
112                  || cause instanceof ConnectException
113                  || cause instanceof SocketTimeoutException)) {
114            LOG.warn("log replaying of " + filename + " can't connect to the target regionserver, "
115                + "resigning", e);
116            return Status.RESIGNED;
117          } else if (cause instanceof InterruptedException) {
118            LOG.warn("log splitting of " + filename + " interrupted, resigning", e);
119            return Status.RESIGNED;
120          }
121          LOG.warn("log splitting of " + filename + " failed, returning error", e);
122          return Status.ERR;
123        }
124        return Status.DONE;
125      }
126    });
127  }
128
129  @Override
130  public void run() {
131    try {
132      LOG.info("SplitLogWorker " + server.getServerName() + " starting");
133      coordination.registerListener();
134      // wait for Coordination Engine is ready
135      boolean res = false;
136      while (!res && !coordination.isStop()) {
137        res = coordination.isReady();
138      }
139      if (!coordination.isStop()) {
140        coordination.taskLoop();
141      }
142    } catch (Throwable t) {
143      if (ExceptionUtil.isInterrupt(t)) {
144        LOG.info("SplitLogWorker interrupted. Exiting. " + (coordination.isStop() ? "" :
145            " (ERROR: exitWorker is not set, exiting anyway)"));
146      } else {
147        // only a logical error can cause here. Printing it out
148        // to make debugging easier
149        LOG.error("unexpected error ", t);
150      }
151    } finally {
152      coordination.removeListener();
153      LOG.info("SplitLogWorker " + server.getServerName() + " exiting");
154    }
155  }
156
157  /**
158   * If the worker is doing a task i.e. splitting a log file then stop the task.
159   * It doesn't exit the worker thread.
160   */
161  public void stopTask() {
162    LOG.info("Sending interrupt to stop the worker thread");
163    worker.interrupt(); // TODO interrupt often gets swallowed, do what else?
164  }
165
166  /**
167   * start the SplitLogWorker thread
168   */
169  public void start() {
170    worker = new Thread(null, this, "SplitLogWorker-" + server.getServerName().toShortString());
171    worker.start();
172  }
173
174  /**
175   * stop the SplitLogWorker thread
176   */
177  public void stop() {
178    coordination.stopProcessingTasks();
179    stopTask();
180  }
181
182  /**
183   * Objects implementing this interface actually do the task that has been
184   * acquired by a {@link SplitLogWorker}. Since there isn't a water-tight
185   * guarantee that two workers will not be executing the same task therefore it
186   * is better to have workers prepare the task and then have the
187   * {@link org.apache.hadoop.hbase.master.SplitLogManager} commit the work in
188   * SplitLogManager.TaskFinisher
189   */
190  public interface TaskExecutor {
191    enum Status {
192      DONE(),
193      ERR(),
194      RESIGNED(),
195      PREEMPTED()
196    }
197    Status exec(String name, RecoveryMode mode, CancelableProgressable p);
198  }
199
200  /**
201   * Returns the number of tasks processed by coordination.
202   * This method is used by tests only
203   */
204  @VisibleForTesting
205  public int getTaskReadySeq() {
206    return coordination.getTaskReadySeq();
207  }
208}
