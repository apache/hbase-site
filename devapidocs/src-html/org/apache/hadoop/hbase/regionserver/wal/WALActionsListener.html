019package org.apache.hadoop.hbase.regionserver.wal;
020
021import java.io.IOException;
022
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.fs.Path;
025import org.apache.hadoop.hbase.HRegionInfo;
026
027import org.apache.hadoop.hbase.wal.WALEdit;
028import org.apache.hadoop.hbase.wal.WALKey;
029
030/**
031 * Get notification of WAL events. The invocations are inline
032 * so make sure your implementation is fast else you'll slow hbase.
033 */
034@InterfaceAudience.Private
035public interface WALActionsListener {
036
037  /**
038   * The WAL is going to be rolled. The oldPath can be null if this is
039   * the first log file from the regionserver.
040   * @param oldPath the path to the old wal
041   * @param newPath the path to the new wal
042   */
043  void preLogRoll(Path oldPath, Path newPath) throws IOException;
044
045  /**
046   * The WAL has been rolled. The oldPath can be null if this is
047   * the first log file from the regionserver.
048   * @param oldPath the path to the old wal
049   * @param newPath the path to the new wal
050   */
051  void postLogRoll(Path oldPath, Path newPath) throws IOException;
052
053  /**
054   * The WAL is going to be archived.
055   * @param oldPath the path to the old wal
056   * @param newPath the path to the new wal
057   */
058  void preLogArchive(Path oldPath, Path newPath) throws IOException;
059
060  /**
061   * The WAL has been archived.
062   * @param oldPath the path to the old wal
063   * @param newPath the path to the new wal
064   */
065  void postLogArchive(Path oldPath, Path newPath) throws IOException;
066
067  /**
068   * A request was made that the WAL be rolled.
069   */
070  void logRollRequested(boolean tooFewReplicas);
071
072  /**
073   * The WAL is about to close.
074   */
075  void logCloseRequested();
076
077  /**
078  * Called before each write.
079  * @param info
080  * @param logKey
081  * @param logEdit
082  */
083  void visitLogEntryBeforeWrite(
084    HRegionInfo info, WALKey logKey, WALEdit logEdit
085  );
086
087  /**
088   * @param logKey
089   * @param logEdit TODO: Retire this in favor of
090   *          {@link #visitLogEntryBeforeWrite(HRegionInfo, WALKey, WALEdit)} It only exists to get
091   *          scope when replicating. Scope should be in the WALKey and not need us passing in a
092   *          <code>htd</code>.
093   * @throws IOException If failed to parse the WALEdit
094   */
095  void visitLogEntryBeforeWrite(WALKey logKey, WALEdit logEdit) throws IOException;
096
097  /**
098   * For notification post append to the writer.  Used by metrics system at least.
099   * TODO: Combine this with above.
100   * @param entryLen approx length of cells in this append.
101   * @param elapsedTimeMillis elapsed time in milliseconds.
102   * @param logKey A WAL key
103   * @param logEdit A WAL edit containing list of cells.
104   * @throws IOException if any network or I/O error occurred
105   */
106  void postAppend(final long entryLen, final long elapsedTimeMillis, final WALKey logKey,
107      final WALEdit logEdit) throws IOException;
108
109  /**
110   * For notification post writer sync.  Used by metrics system at least.
111   * @param timeInNanos How long the filesystem sync took in nanoseconds.
112   * @param handlerSyncs How many sync handler calls were released by this call to filesystem
113   * sync.
114   */
115  void postSync(final long timeInNanos, final int handlerSyncs);
116
117  static class Base implements WALActionsListener {
118    @Override
119    public void preLogRoll(Path oldPath, Path newPath) throws IOException {}
120
121    @Override
122    public void postLogRoll(Path oldPath, Path newPath) throws IOException {}
123
124    @Override
125    public void preLogArchive(Path oldPath, Path newPath) throws IOException {}
126
127    @Override
128    public void postLogArchive(Path oldPath, Path newPath) throws IOException {}
129
130    @Override
131    public void logRollRequested(boolean tooFewReplicas) {}
132
133    @Override
134    public void logCloseRequested() {}
135
136    @Override
137    public void visitLogEntryBeforeWrite(HRegionInfo info, WALKey logKey, WALEdit logEdit) {}
138
139    @Override
140    public void visitLogEntryBeforeWrite(WALKey logKey, WALEdit logEdit) throws IOException {
141    }
142
143    @Override
144    public void postAppend(final long entryLen, final long elapsedTimeMillis, final WALKey logKey,
145        final WALEdit logEdit) throws IOException {
146    }
147
148    @Override
149    public void postSync(final long timeInNanos, final int handlerSyncs) {}
150  }
151}
