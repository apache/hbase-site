069  // IMPLEMENTATION NOTES:
070  //
071  // At the core is a ring buffer. Our ring buffer is the LMAX Disruptor. It tries to
072  // minimize synchronizations and volatile writes when multiple contending threads as is the case
073  // here appending and syncing on a single WAL. The Disruptor is configured to handle multiple
074  // producers but it has one consumer only (the producers in HBase are IPC Handlers calling append
075  // and then sync). The single consumer/writer pulls the appends and syncs off the ring buffer.
076  // When a handler calls sync, it is given back a future. The producer 'blocks' on the future so
077  // it does not return until the sync completes. The future is passed over the ring buffer from
078  // the producer/handler to the consumer thread where it does its best to batch up the producer
079  // syncs so one WAL sync actually spans multiple producer sync invocations. How well the
080  // batching works depends on the write rate; i.e. we tend to batch more in times of
081  // high writes/syncs.
082  //
083  // Calls to append now also wait until the append has been done on the consumer side of the
084  // disruptor. We used to not wait but it makes the implementation easier to grok if we have
085  // the region edit/sequence id after the append returns.
086  //
087  // TODO: Handlers need to coordinate appending AND syncing. Can we have the threads contend
088  // once only? Probably hard given syncs take way longer than an append.
089  //
090  // The consumer threads pass the syncs off to multiple syncing threads in a round robin fashion
091  // to ensure we keep up back-to-back FS sync calls (FS sync calls are the long poll writing the
092  // WAL). The consumer thread passes the futures to the sync threads for it to complete
093  // the futures when done.
094  //
095  // The 'sequence' in the below is the sequence of the append/sync on the ringbuffer. It
096  // acts as a sort-of transaction id. It is always incrementing.
097  //
098  // The RingBufferEventHandler class hosts the ring buffer consuming code. The threads that
099  // do the actual FS sync are implementations of SyncRunner. SafePointZigZagLatch is a
100  // synchronization class used to halt the consumer at a safe point -- just after all outstanding
101  // syncs and appends have completed -- so the log roller can swap the WAL out under it.
102  //
103  // We use ring buffer sequence as txid of FSWALEntry and SyncFuture.
104  private static final Logger LOG = LoggerFactory.getLogger(FSHLog.class);
105

192   * existing log. If there is a log at startup, it should have already been processed and deleted
193   * by the time the WAL object is started up.
194   * @param fs filesystem handle
195   * @param rootDir path to where logs and oldlogs
196   * @param logDir dir where wals are stored
197   * @param archiveDir dir where wals are archived
198   * @param conf configuration to use
199   * @param listeners Listeners on WAL events. Listeners passed here will be registered before we do
200   *          anything else; e.g. the Constructor {@link #rollWriter()}.
201   * @param failIfWALExists If true IOException will be thrown if files related to this wal already
202   *          exist.
203   * @param prefix should always be hostname and port in distributed env and it will be URL encoded
204   *          before being used. If prefix is null, "wal" will be used
205   * @param suffix will be url encoded. null is treated as empty. non-empty must start with
206   *          {@link org.apache.hadoop.hbase.wal.AbstractFSWALProvider#WAL_FILE_NAME_DELIMITER}
207   */
208  public FSHLog(final FileSystem fs, final Path rootDir, final String logDir,
209      final String archiveDir, final Configuration conf, final List<WALActionsListener> listeners,
210      final boolean failIfWALExists, final String prefix, final String suffix) throws IOException {
211    super(fs, rootDir, logDir, archiveDir, conf, listeners, failIfWALExists, prefix, suffix);
212    this.minTolerableReplication = conf.getInt("hbase.regionserver.hlog.tolerable.lowreplication",
213      FSUtils.getDefaultReplication(fs, this.walDir));
214    this.lowReplicationRollLimit = conf.getInt("hbase.regionserver.hlog.lowreplication.rolllimit",
215      5);
216    this.closeErrorsTolerated = conf.getInt("hbase.regionserver.logroll.errors.tolerated", 2);
217    this.useHsync = conf.getBoolean(HRegion.WAL_HSYNC_CONF_KEY, HRegion.DEFAULT_WAL_HSYNC);
218
219    // This is the 'writer' -- a single threaded executor. This single thread 'consumes' what is
220    // put on the ring buffer.
221    String hostingThreadName = Thread.currentThread().getName();
222    // Using BlockingWaitStrategy. Stuff that is going on here takes so long it makes no sense
223    // spinning as other strategies do.
224    this.disruptor = new Disruptor<>(RingBufferTruck::new,
225        getPreallocatedEventCount(), Threads.getNamedThreadFactory(hostingThreadName + ".append"),
226        ProducerType.MULTI, new BlockingWaitStrategy());
227    // Advance the ring buffer sequence so that it starts from 1 instead of 0,
228    // because SyncFuture.NOT_DONE = 0.
229    this.disruptor.getRingBuffer().next();
230    int maxHandlersCount = conf.getInt(HConstants.REGION_SERVER_HANDLER_COUNT, 200);
231    this.ringBufferEventHandler = new RingBufferEventHandler(
232        conf.getInt("hbase.regionserver.hlog.syncer.count", 5), maxHandlersCount);
233    this.disruptor.setDefaultExceptionHandler(new RingBufferExceptionHandler());
234    this.disruptor.handleEventsWith(new RingBufferEventHandler[] { this.ringBufferEventHandler });
235    // Starting up threads in constructor is a no no; Interface should have an init call.
236    this.disruptor.start();
237  }

254   * Run a sync after opening to set up the pipeline.
255   */
256  private void preemptiveSync(final ProtobufLogWriter nextWriter) {
257    long startTimeNanos = System.nanoTime();
258    try {
259      nextWriter.sync(useHsync);
260      postSync(System.nanoTime() - startTimeNanos, 0);
261    } catch (IOException e) {
262      // optimization failed, no need to abort here.
263      LOG.warn("pre-sync failed but an optimization so keep going", e);
264    }
265  }

301  @Override
302  protected void doReplaceWriter(Path oldPath, Path newPath, Writer nextWriter) throws IOException {
303    // Ask the ring buffer writer to pause at a safe point. Once we do this, the writer
304    // thread will eventually pause. An error hereafter needs to release the writer thread
305    // regardless -- hence the finally block below. Note, this method is called from the FSHLog
306    // constructor BEFORE the ring buffer is set running so it is null on first time through
307    // here; allow for that.
308    SyncFuture syncFuture = null;
309    SafePointZigZagLatch zigzagLatch = null;
310    long sequence = -1L;
311    if (this.ringBufferEventHandler != null) {
312      // Get sequence first to avoid dead lock when ring buffer is full
313      // Considering below sequence
314      // 1. replaceWriter is called and zigzagLatch is initialized
315      // 2. ringBufferEventHandler#onEvent is called and arrives at #attainSafePoint(long) then wait
316      // on safePointReleasedLatch
317      // 3. Since ring buffer is full, if we get sequence when publish sync, the replaceWriter
318      // thread will wait for the ring buffer to be consumed, but the only consumer is waiting
319      // replaceWriter thread to release safePointReleasedLatch, which causes a deadlock
320      sequence = getSequenceOnRingBuffer();
321      zigzagLatch = this.ringBufferEventHandler.attainSafePoint();
322    }
323    afterCreatingZigZagLatch();
324    try {
325      // Wait on the safe point to be achieved. Send in a sync in case nothing has hit the
326      // ring buffer between the above notification of writer that we want it to go to
327      // 'safe point' and then here where we are waiting on it to attain safe point. Use
328      // 'sendSync' instead of 'sync' because we do not want this thread to block waiting on it
329      // to come back. Cleanup this syncFuture down below after we are ready to run again.
330      try {
331        if (zigzagLatch != null) {
332          // use assert to make sure no change breaks the logic that
333          // sequence and zigzagLatch will be set together
334          assert sequence > 0L : "Failed to get sequence from ring buffer";
335          TraceUtil.addTimelineAnnotation("awaiting safepoint");
336          syncFuture = zigzagLatch.waitSafePoint(publishSyncOnRingBuffer(sequence, false));
337        }
338      } catch (FailedSyncBeforeLogCloseException e) {
339        // If unflushed/unsynced entries on close, it is reason to abort.
340        if (isUnflushedEntries()) {
341          throw e;
342        }
343        LOG.warn(
344          "Failed sync-before-close but no outstanding appends; closing WAL" + e.getMessage());
345      }
346      long oldFileLen = 0L;
347      // It is at the safe point. Swap out writer from under the blocked writer thread.
348      // TODO: This is close is inline with critical section. Should happen in background?
349      if (this.writer != null) {
350        oldFileLen = this.writer.getLength();
351        try {
352          TraceUtil.addTimelineAnnotation("closing writer");
353          this.writer.close();
354          TraceUtil.addTimelineAnnotation("writer closed");
355          this.closeErrorCount.set(0);
356        } catch (IOException ioe) {
357          int errors = closeErrorCount.incrementAndGet();
358          if (!isUnflushedEntries() && (errors <= this.closeErrorsTolerated)) {
359            LOG.warn("Riding over failed WAL close of " + oldPath + ", cause=\"" + ioe.getMessage()
360                + "\", errors=" + errors
361                + "; THIS FILE WAS NOT CLOSED BUT ALL EDITS SYNCED SO SHOULD BE OK");
362          } else {
363            throw ioe;
364          }
365        }
366      }
367      logRollAndSetupWalProps(oldPath, newPath, oldFileLen);
368      this.writer = nextWriter;
369      if (nextWriter != null && nextWriter instanceof ProtobufLogWriter) {
370        this.hdfs_out = ((ProtobufLogWriter) nextWriter).getStream();
371      } else {
372        this.hdfs_out = null;
373      }
374    } catch (InterruptedException ie) {
375      // Perpetuate the interrupt
376      Thread.currentThread().interrupt();
377    } catch (IOException e) {
378      long count = getUnflushedEntriesCount();
379      LOG.error("Failed close of WAL writer " + oldPath + ", unflushedEntries=" + count, e);
380      throw new FailedLogCloseException(oldPath + ", unflushedEntries=" + count, e);
381    } finally {
382      // Let the writer thread go regardless, whether error or not.
383      if (zigzagLatch != null) {
384        zigzagLatch.releaseSafePoint();
385        // syncFuture will be null if we failed our wait on safe point above. Otherwise, if
386        // latch was obtained successfully, the sync we threw in either trigger the latch or it
387        // got stamped with an exception because the WAL was damaged and we could not sync. Now
388        // the write pipeline has been opened up again by releasing the safe point, process the
389        // syncFuture we got above. This is probably a noop but it may be stale exception from
390        // when old WAL was in place. Catch it if so.
391        if (syncFuture != null) {
392          try {
393            blockOnSync(syncFuture);
394          } catch (IOException ioe) {
395            if (LOG.isTraceEnabled()) {
396              LOG.trace("Stale sync exception", ioe);
397            }
398          }
399        }
400      }
401    }
402  }

404  @Override
405  protected void doShutdown() throws IOException {
406    // Shutdown the disruptor. Will stop after all entries have been processed. Make sure we
407    // have stopped incoming appends before calling this else it will not shutdown. We are
408    // conservative below waiting a long time and if not elapsed, then halting.
409    if (this.disruptor != null) {
410      long timeoutms = conf.getLong("hbase.wal.disruptor.shutdown.timeout.ms", 60000);
411      try {
412        this.disruptor.shutdown(timeoutms, TimeUnit.MILLISECONDS);
413      } catch (TimeoutException e) {
414        LOG.warn("Timed out bringing down disruptor after " + timeoutms + "ms; forcing halt "
415            + "(It is a problem if this is NOT an ABORT! -- DATALOSS!!!!)");
416        this.disruptor.halt();
417        this.disruptor.shutdown();
418      }
419    }
420
421    if (LOG.isDebugEnabled()) {
422      LOG.debug("Closing WAL writer in " + FSUtils.getPath(walDir));
423    }
424    if (this.writer != null) {
425      this.writer.close();
426      this.writer = null;
427    }
428  }

439  /**
440   * Thread to runs the hdfs sync call. This call takes a while to complete. This is the longest
441   * pole adding edits to the WAL and this must complete to be sure all edits persisted. We run
442   * multiple threads sync'ng rather than one that just syncs in series so we have better latencies;
443   * otherwise, an edit that arrived just after a sync started, might have to wait almost the length
444   * of two sync invocations before it is marked done.
445   * <p>
446   * When the sync completes, it marks all the passed in futures done. On the other end of the sync
447   * future is a blocked thread, usually a regionserver Handler. There may be more than one future
448   * passed in the case where a few threads arrive at about the same time and all invoke 'sync'. In
449   * this case we'll batch up the invocations and run one filesystem sync only for a batch of
450   * Handler sync invocations. Do not confuse these Handler SyncFutures with the futures an
451   * ExecutorService returns when you call submit. We have no use for these in this model. These
452   * SyncFutures are 'artificial', something to hold the Handler until the filesystem sync
453   * completes.
454   */
455  private class SyncRunner extends HasThread {
456    private volatile long sequence;
457    // Keep around last exception thrown. Clear on successful sync.
458    private final BlockingQueue<SyncFuture> syncFutures;
459    private volatile SyncFuture takeSyncFuture = null;
460
461    SyncRunner(final String name, final int maxHandlersCount) {
462      super(name);
463      // LinkedBlockingQueue because of
464      // http://www.javacodegeeks.com/2010/09/java-best-practices-queue-battle-and.html
465      // Could use other blockingqueues here or concurrent queues.
466      //
467      // We could let the capacity be 'open' but bound it so we get alerted in pathological case
468      // where we cannot sync and we have a bunch of threads all backed up waiting on their syncs
469      // to come in. LinkedBlockingQueue actually shrinks when you remove elements so Q should
470      // stay neat and tidy in usual case. Let the max size be three times the maximum handlers.
471      // The passed in maxHandlerCount is the user-level handlers which is what we put up most of
472      // but HBase has other handlers running too -- opening region handlers which want to write
473      // the meta table when succesful (i.e. sync), closing handlers -- etc. These are usually
474      // much fewer in number than the user-space handlers so Q-size should be user handlers plus
475      // some space for these other handlers. Lets multiply by 3 for good-measure.
476      this.syncFutures = new LinkedBlockingQueue<>(maxHandlersCount * 3);
477    }
478
479    void offer(final long sequence, final SyncFuture[] syncFutures, final int syncFutureCount) {
480      // Set sequence first because the add to the queue will wake the thread if sleeping.
481      this.sequence = sequence;
482      for (int i = 0; i < syncFutureCount; ++i) {
483        this.syncFutures.add(syncFutures[i]);
484      }
485    }
486
487    /**
488     * Release the passed <code>syncFuture</code>
489     * @return Returns 1.
490     */
491    private int releaseSyncFuture(final SyncFuture syncFuture, final long currentSequence,
492        final Throwable t) {
493      if (!syncFuture.done(currentSequence, t)) {
494        throw new IllegalStateException();
495      }
496
497      // This function releases one sync future only.
498      return 1;
499    }
500
501    /**
502     * Release all SyncFutures whose sequence is <= <code>currentSequence</code>.
503     * @param t May be non-null if we are processing SyncFutures because an exception was thrown.
504     * @return Count of SyncFutures we let go.
505     */
506    private int releaseSyncFutures(final long currentSequence, final Throwable t) {
507      int syncCount = 0;
508      for (SyncFuture syncFuture; (syncFuture = this.syncFutures.peek()) != null;) {
509        if (syncFuture.getTxid() > currentSequence) {
510          break;
511        }
512        releaseSyncFuture(syncFuture, currentSequence, t);
513        if (!this.syncFutures.remove(syncFuture)) {
514          throw new IllegalStateException(syncFuture.toString());
515        }
516        syncCount++;
517      }
518      return syncCount;
519    }
520
521    /**
522     * @param sequence The sequence we ran the filesystem sync against.
523     * @return Current highest synced sequence.
524     */
525    private long updateHighestSyncedSequence(long sequence) {
526      long currentHighestSyncedSequence;
527      // Set the highestSyncedSequence IFF our current sequence id is the 'highest'.
528      do {
529        currentHighestSyncedSequence = highestSyncedTxid.get();
530        if (currentHighestSyncedSequence >= sequence) {
531          // Set the sync number to current highwater mark; might be able to let go more
532          // queued sync futures
533          sequence = currentHighestSyncedSequence;
534          break;
535        }
536      } while (!highestSyncedTxid.compareAndSet(currentHighestSyncedSequence, sequence));
537      return sequence;
538    }
539
540    boolean areSyncFuturesReleased() {
541      // check whether there is no sync futures offered, and no in-flight sync futures that is being
542      // processed.
543      return syncFutures.size() <= 0
544          && takeSyncFuture == null;
545    }
546
547    @Override
548    public void run() {
549      long currentSequence;
550      while (!isInterrupted()) {
551        int syncCount = 0;
552
553        try {
554          while (true) {
555            takeSyncFuture = null;
556            // We have to process what we 'take' from the queue
557            takeSyncFuture = this.syncFutures.take();
558            currentSequence = this.sequence;
559            long syncFutureSequence = takeSyncFuture.getTxid();
560            if (syncFutureSequence > currentSequence) {
561              throw new IllegalStateException("currentSequence=" + currentSequence
562                  + ", syncFutureSequence=" + syncFutureSequence);
563            }
564            // See if we can process any syncfutures BEFORE we go sync.
565            long currentHighestSyncedSequence = highestSyncedTxid.get();
566            if (currentSequence < currentHighestSyncedSequence) {
567              syncCount += releaseSyncFuture(takeSyncFuture, currentHighestSyncedSequence, null);
568              // Done with the 'take'. Go around again and do a new 'take'.
569              continue;
570            }
571            break;
572          }
573          // I got something. Lets run. Save off current sequence number in case it changes
574          // while we run.
575          //TODO handle htrace API change, see HBASE-18895
576          //TraceScope scope = Trace.continueSpan(takeSyncFuture.getSpan());
577          long start = System.nanoTime();
578          Throwable lastException = null;
579          try {
580            TraceUtil.addTimelineAnnotation("syncing writer");
581            writer.sync(useHsync);
582            TraceUtil.addTimelineAnnotation("writer synced");
583            currentSequence = updateHighestSyncedSequence(currentSequence);
584          } catch (IOException e) {
585            LOG.error("Error syncing, request close of WAL", e);
586            lastException = e;
587          } catch (Exception e) {
588            LOG.warn("UNEXPECTED", e);
589            lastException = e;
590          } finally {
591            // reattach the span to the future before releasing.
592            //TODO handle htrace API change, see HBASE-18895
593            // takeSyncFuture.setSpan(scope.getSpan());
594            // First release what we 'took' from the queue.
595            syncCount += releaseSyncFuture(takeSyncFuture, currentSequence, lastException);
596            // Can we release other syncs?
597            syncCount += releaseSyncFutures(currentSequence, lastException);
598            if (lastException != null) {
599              requestLogRoll();
600            } else {
601              checkLogRoll();
602            }
603          }
604          postSync(System.nanoTime() - start, syncCount);
605        } catch (InterruptedException e) {
606          // Presume legit interrupt.
607          Thread.currentThread().interrupt();
608        } catch (Throwable t) {
609          LOG.warn("UNEXPECTED, continuing", t);
610        }
611      }
612    }
613  }

615  /**
616   * Schedule a log roll if needed.
617   */
618  private void checkLogRoll() {
619    // Will return immediately if we are in the middle of a WAL log roll currently.
620    if (!rollWriterLock.tryLock()) {
621      return;
622    }
623    boolean lowReplication;
624    try {
625      lowReplication = doCheckLogLowReplication();
626    } finally {
627      rollWriterLock.unlock();
628    }
629    if (lowReplication || (writer != null && writer.getLength() > logrollsize)) {
630      requestLogRoll(lowReplication);
631    }
632  }

634  /**
635   * @return true if number of replicas for the WAL is lower than threshold
636   */
637  @Override
638  protected boolean doCheckLogLowReplication() {
639    boolean logRollNeeded = false;
640    // if the number of replicas in HDFS has fallen below the configured
641    // value, then roll logs.
642    try {
643      int numCurrentReplicas = getLogReplication();
644      if (numCurrentReplicas != 0 && numCurrentReplicas < this.minTolerableReplication) {
645        if (this.lowReplicationRollEnabled) {
646          if (this.consecutiveLogRolls.get() < this.lowReplicationRollLimit) {
647            LOG.warn("HDFS pipeline error detected. " + "Found " + numCurrentReplicas
648                + " replicas but expecting no less than " + this.minTolerableReplication
649                + " replicas. " + " Requesting close of WAL. current pipeline: "
650                + Arrays.toString(getPipeline()));
651            logRollNeeded = true;
652            // If rollWriter is requested, increase consecutiveLogRolls. Once it
653            // is larger than lowReplicationRollLimit, disable the
654            // LowReplication-Roller
655            this.consecutiveLogRolls.getAndIncrement();
656          } else {
657            LOG.warn("Too many consecutive RollWriter requests, it's a sign of "
658                + "the total number of live datanodes is lower than the tolerable replicas.");
659            this.consecutiveLogRolls.set(0);
660            this.lowReplicationRollEnabled = false;
661          }
662        }
663      } else if (numCurrentReplicas >= this.minTolerableReplication) {
664        if (!this.lowReplicationRollEnabled) {
665          // The new writer's log replicas is always the default value.
666          // So we should not enable LowReplication-Roller. If numEntries
667          // is lower than or equals 1, we consider it as a new writer.
668          if (this.numEntries.get() <= 1) {
669            return logRollNeeded;
670          }
671          // Once the live datanode number and the replicas return to normal,
672          // enable the LowReplication-Roller.
673          this.lowReplicationRollEnabled = true;
674          LOG.info("LowReplication-Roller was enabled.");
675        }
676      }
677    } catch (Exception e) {
678      LOG.warn("DFSOutputStream.getNumCurrentReplicas failed because of " + e + ", continuing...");
679    }
680    return logRollNeeded;
681  }

770  /**
771   * This class is used coordinating two threads holding one thread at a 'safe point' while the
772   * orchestrating thread does some work that requires the first thread paused: e.g. holding the WAL
773   * writer while its WAL is swapped out from under it by another thread.
774   * <p>
775   * Thread A signals Thread B to hold when it gets to a 'safe point'. Thread A wait until Thread B
776   * gets there. When the 'safe point' has been attained, Thread B signals Thread A. Thread B then
777   * holds at the 'safe point'. Thread A on notification that Thread B is paused, goes ahead and
778   * does the work it needs to do while Thread B is holding. When Thread A is done, it flags B and
779   * then Thread A and Thread B continue along on their merry way. Pause and signalling 'zigzags'
780   * between the two participating threads. We use two latches -- one the inverse of the other --
781   * pausing and signaling when states are achieved.
782   * <p>
783   * To start up the drama, Thread A creates an instance of this class each time it would do this
784   * zigzag dance and passes it to Thread B (these classes use Latches so it is one shot only).
785   * Thread B notices the new instance (via reading a volatile reference or how ever) and it starts
786   * to work toward the 'safe point'. Thread A calls {@link #waitSafePoint(SyncFuture)} when it cannot proceed
787   * until the Thread B 'safe point' is attained. Thread A will be held inside in
788   * {@link #waitSafePoint(SyncFuture)} until Thread B reaches the 'safe point'. Once there, Thread B frees
789   * Thread A by calling {@link #safePointAttained()}. Thread A now knows Thread B is at the 'safe
790   * point' and that it is holding there (When Thread B calls {@link #safePointAttained()} it blocks
791   * here until Thread A calls {@link #releaseSafePoint()}). Thread A proceeds to do what it needs
792   * to do while Thread B is paused. When finished, it lets Thread B lose by calling
793   * {@link #releaseSafePoint()} and away go both Threads again.
794   */
795  static class SafePointZigZagLatch {
796    /**
797     * Count down this latch when safe point attained.
798     */
799    private volatile CountDownLatch safePointAttainedLatch = new CountDownLatch(1);
800    /**
801     * Latch to wait on. Will be released when we can proceed.
802     */
803    private volatile CountDownLatch safePointReleasedLatch = new CountDownLatch(1);
804
805    private void checkIfSyncFailed(SyncFuture syncFuture) throws FailedSyncBeforeLogCloseException {
806      if (syncFuture.isThrowable()) {
807        throw new FailedSyncBeforeLogCloseException(syncFuture.getThrowable());
808      }
809    }
810
811    /**
812     * For Thread A to call when it is ready to wait on the 'safe point' to be attained. Thread A
813     * will be held in here until Thread B calls {@link #safePointAttained()}
814     * @param syncFuture We need this as barometer on outstanding syncs. If it comes home with an
815     *          exception, then something is up w/ our syncing.
816     * @return The passed <code>syncFuture</code>
817     */
818    SyncFuture waitSafePoint(SyncFuture syncFuture) throws InterruptedException,
819        FailedSyncBeforeLogCloseException {
820      while (!this.safePointAttainedLatch.await(1, TimeUnit.MILLISECONDS)) {
821        checkIfSyncFailed(syncFuture);
822      }
823      checkIfSyncFailed(syncFuture);
824      return syncFuture;
825    }
826
827    /**
828     * Called by Thread B when it attains the 'safe point'. In this method, Thread B signals Thread
829     * A it can proceed. Thread B will be held in here until {@link #releaseSafePoint()} is called
830     * by Thread A.
831     */
832    void safePointAttained() throws InterruptedException {
833      this.safePointAttainedLatch.countDown();
834      this.safePointReleasedLatch.await();
835    }
836
837    /**
838     * Called by Thread A when it is done with the work it needs to do while Thread B is halted.
839     * This will release the Thread B held in a call to {@link #safePointAttained()}
840     */
841    void releaseSafePoint() {
842      this.safePointReleasedLatch.countDown();
843    }
844
845    /**
846     * @return True is this is a 'cocked', fresh instance, and not one that has already fired.
847     */
848    boolean isCocked() {
849      return this.safePointAttainedLatch.getCount() > 0
850          && this.safePointReleasedLatch.getCount() > 0;
851    }
852  }

854  /**
855   * Handler that is run by the disruptor ringbuffer consumer. Consumer is a SINGLE
856   * 'writer/appender' thread. Appends edits and starts up sync runs. Tries its best to batch up
857   * syncs. There is no discernible benefit batching appends so we just append as they come in
858   * because it simplifies the below implementation. See metrics for batching effectiveness (In
859   * measurement, at 100 concurrent handlers writing 1k, we are batching > 10 appends and 10 handler
860   * sync invocations for every actual dfsclient sync call; at 10 concurrent handlers, YMMV).
861   * <p>
862   * Herein, we have an array into which we store the sync futures as they come in. When we have a
863   * 'batch', we'll then pass what we have collected to a SyncRunner thread to do the filesystem
864   * sync. When it completes, it will then call {@link SyncFuture#done(long, Throwable)} on each of
865   * SyncFutures in the batch to release blocked Handler threads.
866   * <p>
867   * I've tried various effects to try and make latencies low while keeping throughput high. I've
868   * tried keeping a single Queue of SyncFutures in this class appending to its tail as the syncs
869   * coming and having sync runner threads poll off the head to 'finish' completed SyncFutures. I've
870   * tried linkedlist, and various from concurrent utils whether LinkedBlockingQueue or
871   * ArrayBlockingQueue, etc. The more points of synchronization, the more 'work' (according to
872   * 'perf stats') that has to be done; small increases in stall percentages seem to have a big
873   * impact on throughput/latencies. The below model where we have an array into which we stash the
874   * syncs and then hand them off to the sync thread seemed like a decent compromise. See HBASE-8755
875   * for more detail.
876   */
877  class RingBufferEventHandler implements EventHandler<RingBufferTruck>, LifecycleAware {
878    private final SyncRunner[] syncRunners;
879    private final SyncFuture[] syncFutures;
880    // Had 'interesting' issues when this was non-volatile. On occasion, we'd not pass all
881    // syncFutures to the next sync'ing thread.
882    private AtomicInteger syncFuturesCount = new AtomicInteger();
883    private volatile SafePointZigZagLatch zigzagLatch;
884    /**
885     * Set if we get an exception appending or syncing so that all subsequence appends and syncs on
886     * this WAL fail until WAL is replaced.
887     */
888    private Exception exception = null;
889    /**
890     * Object to block on while waiting on safe point.
891     */
892    private final Object safePointWaiter = new Object();
893    private volatile boolean shutdown = false;
894
895    /**
896     * Which syncrunner to use next.
897     */
898    private int syncRunnerIndex;
899
900    RingBufferEventHandler(final int syncRunnerCount, final int maxHandlersCount) {
901      this.syncFutures = new SyncFuture[maxHandlersCount];
902      this.syncRunners = new SyncRunner[syncRunnerCount];
903      for (int i = 0; i < syncRunnerCount; i++) {
904        this.syncRunners[i] = new SyncRunner("sync." + i, maxHandlersCount);
905      }
906    }
907
908    private void cleanupOutstandingSyncsOnException(final long sequence, final Exception e) {
909      // There could be handler-count syncFutures outstanding.
910      for (int i = 0; i < this.syncFuturesCount.get(); i++) {
911        this.syncFutures[i].done(sequence, e);
912      }
913      this.syncFuturesCount.set(0);
914    }
915
916    /**
917     * @return True if outstanding sync futures still
918     */
919    private boolean isOutstandingSyncs() {
920      // Look at SyncFutures in the EventHandler
921      for (int i = 0; i < this.syncFuturesCount.get(); i++) {
922        if (!this.syncFutures[i].isDone()) {
923          return true;
924        }
925      }
926
927      return false;
928    }
929
930    private boolean isOutstandingSyncsFromRunners() {
931      // Look at SyncFutures in the SyncRunners
932      for (SyncRunner syncRunner: syncRunners) {
933        if(syncRunner.isAlive() && !syncRunner.areSyncFuturesReleased()) {
934          return true;
935        }
936      }
937      return false;
938    }
939
940    @Override
941    // We can set endOfBatch in the below method if at end of our this.syncFutures array
942    public void onEvent(final RingBufferTruck truck, final long sequence, boolean endOfBatch)
943        throws Exception {
944      // Appends and syncs are coming in order off the ringbuffer. We depend on this fact. We'll
945      // add appends to dfsclient as they come in. Batching appends doesn't give any significant
946      // benefit on measurement. Handler sync calls we will batch up. If we get an exception
947      // appending an edit, we fail all subsequent appends and syncs with the same exception until
948      // the WAL is reset. It is important that we not short-circuit and exit early this method.
949      // It is important that we always go through the attainSafePoint on the end. Another thread,
950      // the log roller may be waiting on a signal from us here and will just hang without it.
951
952      try {
953        if (truck.type() == RingBufferTruck.Type.SYNC) {
954          this.syncFutures[this.syncFuturesCount.getAndIncrement()] = truck.unloadSync();
955          // Force flush of syncs if we are carrying a full complement of syncFutures.
956          if (this.syncFuturesCount.get() == this.syncFutures.length) {
957            endOfBatch = true;
958          }
959        } else if (truck.type() == RingBufferTruck.Type.APPEND) {
960          FSWALEntry entry = truck.unloadAppend();
961          //TODO handle htrace API change, see HBASE-18895
962          //TraceScope scope = Trace.continueSpan(entry.detachSpan());
963          try {
964
965            if (this.exception != null) {
966              // Return to keep processing events coming off the ringbuffer
967              return;
968            }
969            append(entry);
970          } catch (Exception e) {
971            // Failed append. Record the exception.
972            this.exception = e;
973            // invoking cleanupOutstandingSyncsOnException when append failed with exception,
974            // it will cleanup existing sync requests recorded in syncFutures but not offered to SyncRunner yet,
975            // so there won't be any sync future left over if no further truck published to disruptor.
976            cleanupOutstandingSyncsOnException(sequence,
977                this.exception instanceof DamagedWALException ? this.exception
978                    : new DamagedWALException("On sync", this.exception));
979            // Return to keep processing events coming off the ringbuffer
980            return;
981          }
982        } else {
983          // What is this if not an append or sync. Fail all up to this!!!
984          cleanupOutstandingSyncsOnException(sequence,
985            new IllegalStateException("Neither append nor sync"));
986          // Return to keep processing.
987          return;
988        }
989
990        // TODO: Check size and if big go ahead and call a sync if we have enough data.
991        // This is a sync. If existing exception, fall through. Else look to see if batch.
992        if (this.exception == null) {
993          // If not a batch, return to consume more events from the ring buffer before proceeding;
994          // we want to get up a batch of syncs and appends before we go do a filesystem sync.
995          if (!endOfBatch || this.syncFuturesCount.get() <= 0) {
996            return;
997          }
998          // syncRunnerIndex is bound to the range [0, Integer.MAX_INT - 1] as follows:
999          //   * The maximum value possible for syncRunners.length is Integer.MAX_INT
1000          //   * syncRunnerIndex starts at 0 and is incremented only here
1001          //   * after the increment, the value is bounded by the '%' operator to
1002          //     [0, syncRunners.length), presuming the value was positive prior to
1003          //     the '%' operator.
1004          //   * after being bound to [0, Integer.MAX_INT - 1], the new value is stored in
1005          //     syncRunnerIndex ensuring that it can't grow without bound and overflow.
1006          //   * note that the value after the increment must be positive, because the most it
1007          //     could have been prior was Integer.MAX_INT - 1 and we only increment by 1.
1008          this.syncRunnerIndex = (this.syncRunnerIndex + 1) % this.syncRunners.length;
1009          try {
1010            // Below expects that the offer 'transfers' responsibility for the outstanding syncs to
1011            // the syncRunner. We should never get an exception in here.
1012            this.syncRunners[this.syncRunnerIndex].offer(sequence, this.syncFutures,
1013              this.syncFuturesCount.get());
1014          } catch (Exception e) {
1015            // Should NEVER get here.
1016            requestLogRoll();
1017            this.exception = new DamagedWALException("Failed offering sync", e);
1018          }
1019        }
1020        // We may have picked up an exception above trying to offer sync
1021        if (this.exception != null) {
1022          cleanupOutstandingSyncsOnException(sequence, this.exception instanceof DamagedWALException
1023              ? this.exception : new DamagedWALException("On sync", this.exception));
1024        }
1025        attainSafePoint(sequence);
1026        this.syncFuturesCount.set(0);
1027      } catch (Throwable t) {
1028        LOG.error("UNEXPECTED!!! syncFutures.length=" + this.syncFutures.length, t);
1029      }
1030    }

1032    SafePointZigZagLatch attainSafePoint() {
1033      this.zigzagLatch = new SafePointZigZagLatch();
1034      return this.zigzagLatch;
1035    }

1037    /**
1038     * Check if we should attain safe point. If so, go there and then wait till signalled before we
1039     * proceeding.
1040     */
1041    private void attainSafePoint(final long currentSequence) {
1042      if (this.zigzagLatch == null || !this.zigzagLatch.isCocked()) {
1043        return;
1044      }
1045      // If here, another thread is waiting on us to get to safe point. Don't leave it hanging.
1046      beforeWaitOnSafePoint();
1047      try {
1048        // Wait on outstanding syncers; wait for them to finish syncing (unless we've been
1049        // shutdown or unless our latch has been thrown because we have been aborted or unless
1050        // this WAL is broken and we can't get a sync/append to complete).
1051        while ((!this.shutdown && this.zigzagLatch.isCocked()
1052            && highestSyncedTxid.get() < currentSequence &&
1053            // We could be in here and all syncs are failing or failed. Check for this. Otherwise
1054            // we'll just be stuck here for ever. In other words, ensure there syncs running.
1055            isOutstandingSyncs())
1056            // Wait for all SyncRunners to finish their work so that we can replace the writer
1057            || isOutstandingSyncsFromRunners()) {
1058          synchronized (this.safePointWaiter) {
1059            this.safePointWaiter.wait(0, 1);
1060          }
1061        }
1062        // Tell waiting thread we've attained safe point. Can clear this.throwable if set here
1063        // because we know that next event through the ringbuffer will be going to a new WAL
1064        // after we do the zigzaglatch dance.
1065        this.exception = null;
1066        this.zigzagLatch.safePointAttained();
1067      } catch (InterruptedException e) {
1068        LOG.warn("Interrupted ", e);
1069        Thread.currentThread().interrupt();
1070      }
1071    }
1072
1073    /**
1074     * Append to the WAL. Does all CP and WAL listener calls.
1075     */
1076    void append(final FSWALEntry entry) throws Exception {
1077      try {
1078        FSHLog.this.append(writer, entry);
1079      } catch (Exception e) {
1080        String msg = "Append sequenceId=" + entry.getKey().getSequenceId()
1081            + ", requesting roll of WAL";
1082        LOG.warn(msg, e);
1083        requestLogRoll();
1084        throw new DamagedWALException(msg, e);
1085      }
1086    }

1089    public void onStart() {
1090      for (SyncRunner syncRunner : this.syncRunners) {
1091        syncRunner.start();
1092      }
1093    }

1095    @Override
1096    public void onShutdown() {
1097      for (SyncRunner syncRunner : this.syncRunners) {
1098        syncRunner.interrupt();
1099      }
1100    }
