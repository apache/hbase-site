054/**
055 * AsyncWriter for protobuf-based WAL.
056 */
057@InterfaceAudience.Private
058public class AsyncProtobufLogWriter extends AbstractProtobufLogWriter
059    implements AsyncFSWALProvider.AsyncWriter {
060
061  private static final Logger LOG = LoggerFactory.getLogger(AsyncProtobufLogWriter.class);
062
063  private final EventLoopGroup eventLoopGroup;
064
065  private final Class<? extends Channel> channelClass;
066
067  private volatile AsyncFSOutput output;
068  /**
069   * Save {@link AsyncFSOutput#getSyncedLength()} when {@link #output} is closed.
070   */
071  private volatile long finalSyncedLength = -1;
072
073  private static final class OutputStreamWrapper extends OutputStream
074      implements ByteBufferWriter {
075
076    private final AsyncFSOutput out;
077
078    private final byte[] oneByteBuf = new byte[1];
079
080    @Override
081    public void write(int b) throws IOException {
082      oneByteBuf[0] = (byte) b;
083      write(oneByteBuf);
084    }
085
086    public OutputStreamWrapper(AsyncFSOutput out) {
087      this.out = out;
088    }
089
090    @Override
091    public void write(ByteBuffer b, int off, int len) throws IOException {
092      ByteBuffer bb = b.duplicate();
093      bb.position(off);
094      bb.limit(off + len);
095      out.write(bb);
096    }
097
098    @Override
099    public void writeInt(int i) throws IOException {
100      out.writeInt(i);
101    }
102
103    @Override
104    public void write(byte[] b, int off, int len) throws IOException {
105      out.write(b, off, len);
106    }
107
108    @Override
109    public void close() throws IOException {
110      out.close();
111    }
112  }
113
114  private OutputStream asyncOutputWrapper;
115  private long waitTimeout;
116
117  public AsyncProtobufLogWriter(EventLoopGroup eventLoopGroup,
118      Class<? extends Channel> channelClass) {
119    this.eventLoopGroup = eventLoopGroup;
120    this.channelClass = channelClass;
121    // Reuse WAL_ROLL_WAIT_TIMEOUT here to avoid an infinite wait if somehow a wait on a future
122    // never completes. The objective is the same. We want to propagate an exception to trigger
123    // an abort if we seem to be hung.
124    if (this.conf == null) {
125      this.conf = HBaseConfiguration.create();
126    }
127    this.waitTimeout = this.conf.getLong(AbstractWALRoller.WAL_ROLL_WAIT_TIMEOUT,
128      AbstractWALRoller.DEFAULT_WAL_ROLL_WAIT_TIMEOUT);
129  }
130
131  /*
132   * @return class name which is recognized by hbase-1.x to avoid ProtobufLogReader throwing error:
133   *   IOException: Got unknown writer class: AsyncProtobufLogWriter
134   */
135  @Override
136  protected String getWriterClassName() {
137    return "ProtobufLogWriter";
138  }
139
140  @Override
141  public void append(Entry entry) {
142    int buffered = output.buffered();
143    try {
144      entry.getKey().
145        getBuilder(compressor).setFollowingKvCount(entry.getEdit().size()).build()
146          .writeDelimitedTo(asyncOutputWrapper);
147    } catch (IOException e) {
148      throw new AssertionError("should not happen", e);
149    }
150    try {
151      for (Cell cell : entry.getEdit().getCells()) {
152        cellEncoder.write(cell);
153      }
154    } catch (IOException e) {
155      throw new AssertionError("should not happen", e);
156    }
157    length.addAndGet(output.buffered() - buffered);
158  }
159
160  @Override
161  public CompletableFuture<Long> sync(boolean forceSync) {
162    return output.flush(forceSync);
163  }
164
165  @Override
166  public synchronized void close() throws IOException {
167    if (this.output == null) {
168      return;
169    }
170    try {
171      writeWALTrailer();
172      output.close();
173    } catch (Exception e) {
174      LOG.warn("normal close failed, try recover", e);
175      output.recoverAndClose(null);
176    }
177    /**
178     * We have to call {@link AsyncFSOutput#getSyncedLength()}
179     * after {@link AsyncFSOutput#close()} to get the final length
180     * synced to underlying filesystem because {@link AsyncFSOutput#close()}
181     * may also flush some data to underlying filesystem.
182     */
183    this.finalSyncedLength = this.output.getSyncedLength();
184    this.output = null;
185  }
186
187  public AsyncFSOutput getOutput() {
188    return this.output;
189  }
190
191  @Override
192  protected void initOutput(FileSystem fs, Path path, boolean overwritable, int bufferSize,
193      short replication, long blockSize, StreamSlowMonitor monitor) throws IOException,
194      StreamLacksCapabilityException {
195    this.output = AsyncFSOutputHelper.createOutput(fs, path, overwritable, false, replication,
196        blockSize, eventLoopGroup, channelClass, monitor);
197    this.asyncOutputWrapper = new OutputStreamWrapper(output);
198  }
199
200  private long writeWALMetadata(Consumer<CompletableFuture<Long>> action) throws IOException {
201    CompletableFuture<Long> future = new CompletableFuture<>();
202    action.accept(future);
203    try {
204      return future.get(waitTimeout, TimeUnit.MILLISECONDS).longValue();
205    } catch (InterruptedException e) {
206      InterruptedIOException ioe = new InterruptedIOException();
207      ioe.initCause(e);
208      throw ioe;
209    } catch (ExecutionException | TimeoutException e) {
210      Throwables.propagateIfPossible(e.getCause(), IOException.class);
211      throw new RuntimeException(e.getCause());
212    }
213  }
214
215  @Override
216  protected long writeMagicAndWALHeader(byte[] magic, WALHeader header) throws IOException {
217    return writeWALMetadata(future -> {
218      output.write(magic);
219      try {
220        header.writeDelimitedTo(asyncOutputWrapper);
221      } catch (IOException e) {
222        // should not happen
223        throw new AssertionError(e);
224      }
225      addListener(output.flush(false), (len, error) -> {
226        if (error != null) {
227          future.completeExceptionally(error);
228        } else {
229          future.complete(len);
230        }
231      });
232    });
233  }
234
235  @Override
236  protected long writeWALTrailerAndMagic(WALTrailer trailer, byte[] magic) throws IOException {
237    return writeWALMetadata(future -> {
238      try {
239        trailer.writeTo(asyncOutputWrapper);
240      } catch (IOException e) {
241        // should not happen
242        throw new AssertionError(e);
243      }
244      output.writeInt(trailer.getSerializedSize());
245      output.write(magic);
246      addListener(output.flush(false), (len, error) -> {
247        if (error != null) {
248          future.completeExceptionally(error);
249        } else {
250          future.complete(len);
251        }
252      });
253    });
254  }
255
256  @Override
257  protected OutputStream getOutputStreamForCellEncoder() {
258    return asyncOutputWrapper;
259  }
260
261  @Override
262  public long getSyncedLength() {
263   /**
264    * The statement "this.output = null;" in {@link AsyncProtobufLogWriter#close}
265    * is a sync point, if output is null, then finalSyncedLength must set,
266    * so we can return finalSyncedLength, else we return output.getSyncedLength
267    */
268    AsyncFSOutput outputToUse = this.output;
269    if(outputToUse == null) {
270        long finalSyncedLengthToUse = this.finalSyncedLength;
271        assert finalSyncedLengthToUse >= 0;
272        return finalSyncedLengthToUse;
273    }
274    return outputToUse.getSyncedLength();
275  }
