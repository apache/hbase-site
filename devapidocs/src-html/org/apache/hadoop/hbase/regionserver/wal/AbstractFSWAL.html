087/**
088 * Implementation of {@link WAL} to go against {@link FileSystem}; i.e. keep WALs in HDFS. Only one
089 * WAL is ever being written at a time. When a WAL hits a configured maximum size, it is rolled.
090 * This is done internal to the implementation.
091 * <p>
092 * As data is flushed from the MemStore to other on-disk structures (files sorted by key, hfiles), a
093 * WAL becomes obsolete. We can let go of all the log edits/entries for a given HRegion-sequence id.
094 * A bunch of work in the below is done keeping account of these region sequence ids -- what is
095 * flushed out to hfiles, and what is yet in WAL and in memory only.
096 * <p>
097 * It is only practical to delete entire files. Thus, we delete an entire on-disk file
098 * <code>F</code> when all of the edits in <code>F</code> have a log-sequence-id that's older
099 * (smaller) than the most-recent flush.
100 * <p>
101 * To read an WAL, call
102 * {@link WALFactory#createReader(org.apache.hadoop.fs.FileSystem, org.apache.hadoop.fs.Path)}. *
103 * <h2>Failure Semantic</h2> If an exception on append or sync, roll the WAL because the current WAL
104 * is now a lame duck; any more appends or syncs will fail also with the same original exception. If
105 * we have made successful appends to the WAL and we then are unable to sync them, our current
106 * semantic is to return error to the client that the appends failed but also to abort the current
107 * context, usually the hosting server. We need to replay the WALs. <br>
108 * TODO: Change this semantic. A roll of WAL may be sufficient as long as we have flagged client
109 * that the append failed. <br>
110 * TODO: replication may pick up these last edits though they have been marked as failed append
111 * (Need to keep our own file lengths, not rely on HDFS).
112 */

324  /**
325   * A log file has a creation timestamp (in ms) in its file name ({@link #filenum}. This helper
326   * method returns the creation timestamp from a given log file. It extracts the timestamp assuming
327   * the filename is created with the {@link #computeFilename(long filenum)} method.
328   * @return timestamp, as in the log file name.
329   */
330  protected long getFileNumFromFileName(Path fileName) {
331    checkNotNull(fileName, "file name can't be null");
332    if (!ourFiles.accept(fileName)) {
333      throw new IllegalArgumentException(
334          "The log file " + fileName + " doesn't belong to this WAL. (" + toString() + ")");
335    }
336    final String fileNameString = fileName.toString();
337    String chompedPath = fileNameString.substring(prefixPathStr.length(),
338      (fileNameString.length() - walFileSuffix.length()));
339    return Long.parseLong(chompedPath);
340  }

342  private int calculateMaxLogFiles(Configuration conf, long logRollSize) {
343    Pair<Long, MemoryType> globalMemstoreSize = MemorySizeUtil.getGlobalMemStoreSize(conf);
344    return (int) ((globalMemstoreSize.getFirst() * 2) / logRollSize);
345  }

347  // must be power of 2
348  protected final int getPreallocatedEventCount() {
349    // Preallocate objects to use on the ring buffer. The way that appends and syncs work, we will
350    // be stuck and make no progress if the buffer is filled with appends only and there is no
351    // sync. If no sync, then the handlers will be outstanding just waiting on sync completion
352    // before they return.
353    int preallocatedEventCount =
354      this.conf.getInt("hbase.regionserver.wal.disruptor.event.count", 1024 * 16);
355    checkArgument(preallocatedEventCount >= 0,
356      "hbase.regionserver.wal.disruptor.event.count must > 0");
357    int floor = Integer.highestOneBit(preallocatedEventCount);
358    if (floor == preallocatedEventCount) {
359      return floor;
360    }
361    // max capacity is 1 << 30
362    if (floor >= 1 << 29) {
363      return 1 << 30;
364    }
365    return floor << 1;
366  }

368  protected AbstractFSWAL(final FileSystem fs, final Path rootDir, final String logDir,
369      final String archiveDir, final Configuration conf, final List<WALActionsListener> listeners,
370      final boolean failIfWALExists, final String prefix, final String suffix)
371      throws FailedLogCloseException, IOException {
372    this.fs = fs;
373    this.walDir = new Path(rootDir, logDir);
374    this.walArchiveDir = new Path(rootDir, archiveDir);
375    this.conf = conf;
376
377    if (!fs.exists(walDir) && !fs.mkdirs(walDir)) {
378      throw new IOException("Unable to mkdir " + walDir);
379    }
380
381    if (!fs.exists(this.walArchiveDir)) {
382      if (!fs.mkdirs(this.walArchiveDir)) {
383        throw new IOException("Unable to mkdir " + this.walArchiveDir);
384      }
385    }
386
387    // If prefix is null||empty then just name it wal
388    this.walFilePrefix =
389      prefix == null || prefix.isEmpty() ? "wal" : URLEncoder.encode(prefix, "UTF8");
390    // we only correctly differentiate suffices when numeric ones start with '.'
391    if (suffix != null && !(suffix.isEmpty()) && !(suffix.startsWith(WAL_FILE_NAME_DELIMITER))) {
392      throw new IllegalArgumentException("WAL suffix must start with '" + WAL_FILE_NAME_DELIMITER +
393        "' but instead was '" + suffix + "'");
394    }
395    // Now that it exists, set the storage policy for the entire directory of wal files related to
396    // this FSHLog instance
397    String storagePolicy =
398        conf.get(HConstants.WAL_STORAGE_POLICY, HConstants.DEFAULT_WAL_STORAGE_POLICY);
399    CommonFSUtils.setStoragePolicy(fs, this.walDir, storagePolicy);
400    this.walFileSuffix = (suffix == null) ? "" : URLEncoder.encode(suffix, "UTF8");
401    this.prefixPathStr = new Path(walDir, walFilePrefix + WAL_FILE_NAME_DELIMITER).toString();
402
403    this.ourFiles = new PathFilter() {
404      @Override
405      public boolean accept(final Path fileName) {
406        // The path should start with dir/<prefix> and end with our suffix
407        final String fileNameString = fileName.toString();
408        if (!fileNameString.startsWith(prefixPathStr)) {
409          return false;
410        }
411        if (walFileSuffix.isEmpty()) {
412          // in the case of the null suffix, we need to ensure the filename ends with a timestamp.
413          return org.apache.commons.lang3.StringUtils
414              .isNumeric(fileNameString.substring(prefixPathStr.length()));
415        } else if (!fileNameString.endsWith(walFileSuffix)) {
416          return false;
417        }
418        return true;
419      }
420    };
421
422    if (failIfWALExists) {
423      final FileStatus[] walFiles = CommonFSUtils.listStatus(fs, walDir, ourFiles);
424      if (null != walFiles && 0 != walFiles.length) {
425        throw new IOException("Target WAL already exists within directory " + walDir);
426      }
427    }
428
429    // Register listeners. TODO: Should this exist anymore? We have CPs?
430    if (listeners != null) {
431      for (WALActionsListener i : listeners) {
432        registerWALActionsListener(i);
433      }
434    }
435    this.coprocessorHost = new WALCoprocessorHost(this, conf);
436
437    // Schedule a WAL roll when the WAL is 50% of the HDFS block size. Scheduling at 50% of block
438    // size should make it so WAL rolls before we get to the end-of-block (Block transitions cost
439    // some latency). In hbase-1 we did this differently. We scheduled a roll when we hit 95% of
440    // the block size but experience from the field has it that this was not enough time for the
441    // roll to happen before end-of-block. So the new accounting makes WALs of about the same
442    // size as those made in hbase-1 (to prevent surprise), we now have default block size as
443    // 2 times the DFS default: i.e. 2 * DFS default block size rolling at 50% full will generally
444    // make similar size logs to 1 * DFS default block size rolling at 95% full. See HBASE-19148.
445    this.blocksize = WALUtil.getWALBlockSize(this.conf, this.fs, this.walDir);
446    float multiplier = conf.getFloat("hbase.regionserver.logroll.multiplier", 0.5f);
447    this.logrollsize = (long)(this.blocksize * multiplier);
448    this.maxLogs = conf.getInt("hbase.regionserver.maxlogs",
449      Math.max(32, calculateMaxLogFiles(conf, logrollsize)));
450
451    LOG.info("WAL configuration: blocksize=" + StringUtils.byteDesc(blocksize) + ", rollsize=" +
452      StringUtils.byteDesc(this.logrollsize) + ", prefix=" + this.walFilePrefix + ", suffix=" +
453      walFileSuffix + ", logDir=" + this.walDir + ", archiveDir=" + this.walArchiveDir);
454    this.slowSyncNs = TimeUnit.MILLISECONDS.toNanos(conf.getInt(SLOW_SYNC_TIME_MS,
455      DEFAULT_SLOW_SYNC_TIME_MS));
456    this.rollOnSyncNs = TimeUnit.MILLISECONDS.toNanos(conf.getInt(ROLL_ON_SYNC_TIME_MS,
457      DEFAULT_ROLL_ON_SYNC_TIME_MS));
458    this.slowSyncRollThreshold = conf.getInt(SLOW_SYNC_ROLL_THRESHOLD,
459      DEFAULT_SLOW_SYNC_ROLL_THRESHOLD);
460    this.slowSyncCheckInterval = conf.getInt(SLOW_SYNC_ROLL_INTERVAL_MS,
461      DEFAULT_SLOW_SYNC_ROLL_INTERVAL_MS);
462    this.walSyncTimeoutNs = TimeUnit.MILLISECONDS.toNanos(conf.getLong(WAL_SYNC_TIMEOUT_MS,
463      DEFAULT_WAL_SYNC_TIMEOUT_MS));
464    this.cachedSyncFutures = new ThreadLocal<SyncFuture>() {
465      @Override
466      protected SyncFuture initialValue() {
467        return new SyncFuture();
468      }
469    };
470    this.implClassName = getClass().getSimpleName();
471    this.walTooOldNs = TimeUnit.SECONDS.toNanos(conf.getInt(
472            SURVIVED_TOO_LONG_SEC_KEY, SURVIVED_TOO_LONG_SEC_DEFAULT));
473  }

524  public long getEarliestMemStoreSeqNum(byte[] encodedRegionName, byte[] familyName) {
525    // This method is used by tests and for figuring if we should flush or not because our
526    // sequenceids are too old. It is also used reporting the master our oldest sequenceid for use
527    // figuring what edits can be skipped during log recovery. getEarliestMemStoreSequenceId
528    // from this.sequenceIdAccounting is looking first in flushingOldestStoreSequenceIds, the
529    // currently flushing sequence ids, and if anything found there, it is returning these. This is

531    // we crash during the flush. For figuring what to flush, we might get requeued if our sequence
532    // id is old even though we are currently flushing. This may mean we do too much flushing.
533    return this.sequenceIdAccounting.getLowestSequenceId(encodedRegionName, familyName);

563  /**
564   * retrieve the next path to use for writing. Increments the internal filenum.
565   */
566  private Path getNewPath() throws IOException {
567    this.filenum.set(System.currentTimeMillis());
568    Path newPath = getCurrentFileName();
569    while (fs.exists(newPath)) {
570      this.filenum.incrementAndGet();
571      newPath = getCurrentFileName();
572    }
573    return newPath;
574  }

656  /**
657   * Archive old logs. A WAL is eligible for archiving if all its WALEdits have been flushed.
658   */
659  private void cleanOldLogs() throws IOException {
660    List<Pair<Path, Long>> logsToArchive = null;
661    long now = System.nanoTime();
662    boolean mayLogTooOld = nextLogTooOldNs <= now;
663    ArrayList<byte[]> regionsBlockingWal = null;
664    // For each log file, look at its Map of regions to highest sequence id; if all sequence ids
665    // are older than what is currently in memory, the WAL can be GC'd.
666    for (Map.Entry<Path, WalProps> e : this.walFile2Props.entrySet()) {
667      Path log = e.getKey();
668      ArrayList<byte[]> regionsBlockingThisWal = null;
669      long ageNs = now - e.getValue().rollTimeNs;
670      if (ageNs > walTooOldNs) {
671        if (mayLogTooOld && regionsBlockingWal == null) {
672          regionsBlockingWal = new ArrayList<>();
673        }
674        regionsBlockingThisWal = regionsBlockingWal;
675      }
676      Map<byte[], Long> sequenceNums = e.getValue().encodedName2HighestSequenceId;
677      if (this.sequenceIdAccounting.areAllLower(sequenceNums, regionsBlockingThisWal)) {
678        if (logsToArchive == null) {
679          logsToArchive = new ArrayList<>();
680        }
681        logsToArchive.add(Pair.newPair(log, e.getValue().logSize));
682        if (LOG.isTraceEnabled()) {
683          LOG.trace("WAL file ready for archiving " + log);
684        }
685      } else if (regionsBlockingThisWal != null) {
686        StringBuilder sb = new StringBuilder(log.toString()).append(" has not been archived for ")
687          .append(TimeUnit.NANOSECONDS.toSeconds(ageNs)).append(" seconds; blocked by: ");
688        boolean isFirst = true;
689        for (byte[] region : regionsBlockingThisWal) {
690          if (!isFirst) {
691            sb.append("; ");
692          }
693          isFirst = false;
694          sb.append(Bytes.toString(region));
695        }
696        LOG.warn(sb.toString());
697        nextLogTooOldNs = now + SURVIVED_TOO_LONG_LOG_INTERVAL_NS;
698        regionsBlockingThisWal.clear();
699      }
700    }
701    if (logsToArchive != null) {
702      for (Pair<Path, Long> logAndSize : logsToArchive) {
703        this.totalLogSize.addAndGet(-logAndSize.getSecond());
704        archiveLogFile(logAndSize.getFirst());
705        this.walFile2Props.remove(logAndSize.getFirst());
706      }
707    }
708  }

777  protected final void blockOnSync(SyncFuture syncFuture) throws IOException {
778    // Now we have published the ringbuffer, halt the current thread until we get an answer back.
779    try {
780      if (syncFuture != null) {
781        if (closed) {
782          throw new IOException("WAL has been closed");
783        } else {
784          syncFuture.get(walSyncTimeoutNs);
785        }
786      }
787    } catch (TimeoutIOException tioe) {
788      // SyncFuture reuse by thread, if TimeoutIOException happens, ringbuffer
789      // still refer to it, so if this thread use it next time may get a wrong
790      // result.
791      this.cachedSyncFutures.remove();
792      throw tioe;
793    } catch (InterruptedException ie) {
794      LOG.warn("Interrupted", ie);
795      throw convertInterruptedExceptionToIOException(ie);
796    } catch (ExecutionException e) {
797      throw ensureIOException(e.getCause());
798    }
799  }

801  private static IOException ensureIOException(final Throwable t) {
802    return (t instanceof IOException) ? (IOException) t : new IOException(t);
803  }

806    Thread.currentThread().interrupt();
807    IOException ioe = new InterruptedIOException();
808    ioe.initCause(ie);
809    return ioe;
810  }

812  @Override
813  public byte[][] rollWriter(boolean force) throws FailedLogCloseException, IOException {
814    rollWriterLock.lock();
815    try {
816      if (this.closed) {
817        throw new WALClosedException("WAL has been closed");
818      }
819      // Return if nothing to flush.
820      if (!force && this.writer != null && this.numEntries.get() <= 0) {
821        return null;
822      }
823      byte[][] regionsToFlush = null;
824      try (TraceScope scope = TraceUtil.createTrace("FSHLog.rollWriter")) {
825        Path oldPath = getOldPath();
826        Path newPath = getNewPath();
827        // Any exception from here on is catastrophic, non-recoverable so we currently abort.
828        W nextWriter = this.createWriterInstance(newPath);
829        tellListenersAboutPreLogRoll(oldPath, newPath);
830        // NewPath could be equal to oldPath if replaceWriter fails.
831        newPath = replaceWriter(oldPath, newPath, nextWriter);
832        tellListenersAboutPostLogRoll(oldPath, newPath);
833        if (LOG.isDebugEnabled()) {
834          LOG.debug("Create new " + implClassName + " writer with pipeline: " +
835            Arrays.toString(getPipeline()));
836        }
837        // We got a new writer, so reset the slow sync count
838        lastTimeCheckSlowSync = EnvironmentEdgeManager.currentTime();
839        slowSyncCount.set(0);
840        // Can we delete any of the old log files?
841        if (getNumRolledLogFiles() > 0) {
842          cleanOldLogs();
843          regionsToFlush = findRegionsToForceFlush();
844        }
845      } catch (CommonFSUtils.StreamLacksCapabilityException exception) {
846        // If the underlying FileSystem can't do what we ask, treat as IO failure so
847        // we'll abort.
848        throw new IOException(
849            "Underlying FileSystem can't meet stream requirements. See RS log " + "for details.",
850            exception);
851      }
852      return regionsToFlush;
853    } finally {
854      rollWriterLock.unlock();
855    }
856  }

879  public void shutdown() throws IOException {
880    if (!shutdown.compareAndSet(false, true)) {
881      return;
882    }
883    closed = true;
884    // Tell our listeners that the log is closing
885    if (!this.listeners.isEmpty()) {
886      for (WALActionsListener i : this.listeners) {
887        i.logCloseRequested();
888      }
889    }
890    rollWriterLock.lock();
891    try {
892      doShutdown();
893    } finally {
894      rollWriterLock.unlock();
895    }

899  public void close() throws IOException {
900    shutdown();
901    final FileStatus[] files = getFiles();
902    if (null != files && 0 != files.length) {
903      for (FileStatus file : files) {
904        Path p = getWALArchivePath(this.walArchiveDir, file.getPath());
905        // Tell our listeners that a log is going to be archived.
906        if (!this.listeners.isEmpty()) {
907          for (WALActionsListener i : this.listeners) {
908            i.preLogArchive(file.getPath(), p);
909          }
910        }
911
912        if (!CommonFSUtils.renameAndSetModifyTime(fs, file.getPath(), p)) {
913          throw new IOException("Unable to rename " + file.getPath() + " to " + p);
914        }
915        // Tell our listeners that a log was archived.
916        if (!this.listeners.isEmpty()) {
917          for (WALActionsListener i : this.listeners) {
918            i.postLogArchive(file.getPath(), p);
919          }
920        }
921      }
922      LOG.debug(
923        "Moved " + files.length + " WAL file(s) to " + CommonFSUtils.getPath(this.walArchiveDir));
924    }
925    LOG.info("Closed WAL: " + toString());

978  protected final boolean append(W writer, FSWALEntry entry) throws IOException {
979    // TODO: WORK ON MAKING THIS APPEND FASTER. DOING WAY TOO MUCH WORK WITH CPs, PBing, etc.
980    atHeadOfRingBufferEventHandlerAppend();
981    long start = EnvironmentEdgeManager.currentTime();
982    byte[] encodedRegionName = entry.getKey().getEncodedRegionName();
983    long regionSequenceId = entry.getKey().getSequenceId();
984
985    // Edits are empty, there is nothing to append. Maybe empty when we are looking for a
986    // region sequence id only, a region edit/sequence id that is not associated with an actual
987    // edit. It has to go through all the rigmarole to be sure we have the right ordering.
988    if (entry.getEdit().isEmpty()) {
989      return false;
990    }
991
992    // Coprocessor hook.
993    coprocessorHost.preWALWrite(entry.getRegionInfo(), entry.getKey(), entry.getEdit());
994    if (!listeners.isEmpty()) {
995      for (WALActionsListener i : listeners) {
996        i.visitLogEntryBeforeWrite(entry.getKey(), entry.getEdit());
997      }
998    }
999    doAppend(writer, entry);
1000    assert highestUnsyncedTxid < entry.getTxid();
1001    highestUnsyncedTxid = entry.getTxid();
1002    sequenceIdAccounting.update(encodedRegionName, entry.getFamilyNames(), regionSequenceId,
1003      entry.isInMemStore());
1004    coprocessorHost.postWALWrite(entry.getRegionInfo(), entry.getKey(), entry.getEdit());
1005    // Update metrics.
1006    postAppend(entry, EnvironmentEdgeManager.currentTime() - start);
1007    numEntries.incrementAndGet();
1008    return true;
1009  }

1012    long len = 0;
1013    if (!listeners.isEmpty()) {

1020    }

1024  protected final void postSync(long timeInNanos, int handlerSyncs) {
1025    if (timeInNanos > this.slowSyncNs) {
1026      String msg = new StringBuilder().append("Slow sync cost: ")
1027          .append(TimeUnit.NANOSECONDS.toMillis(timeInNanos))
1028          .append(" ms, current pipeline: ")
1029          .append(Arrays.toString(getPipeline())).toString();
1030      TraceUtil.addTimelineAnnotation(msg);
1031      LOG.info(msg);
1032      if (timeInNanos > this.rollOnSyncNs) {
1033        // A single sync took too long.
1034        // Elsewhere in checkSlowSync, called from checkLogRoll, we will look at cumulative
1035        // effects. Here we have a single data point that indicates we should take immediate
1036        // action, so do so.
1037        LOG.warn("Requesting log roll because we exceeded slow sync threshold; time=" +
1038          TimeUnit.NANOSECONDS.toMillis(timeInNanos) + " ms, threshold=" +
1039          TimeUnit.NANOSECONDS.toMillis(rollOnSyncNs) + " ms, current pipeline: " +
1040          Arrays.toString(getPipeline()));
1041        requestLogRoll(SLOW_SYNC);
1042      }
1043      slowSyncCount.incrementAndGet(); // it's fine to unconditionally increment this
1044    }
1045    if (!listeners.isEmpty()) {
1046      for (WALActionsListener listener : listeners) {
1047        listener.postSync(timeInNanos, handlerSyncs);
1048      }
1049    }
1050  }

1052  protected final long stampSequenceIdAndPublishToRingBuffer(RegionInfo hri, WALKeyImpl key,
1053      WALEdit edits, boolean inMemstore, RingBuffer<RingBufferTruck> ringBuffer)
1054      throws IOException {
1055    if (this.closed) {
1056      throw new IOException(
1057          "Cannot append; log is closed, regionName = " + hri.getRegionNameAsString());
1058    }
1059    MutableLong txidHolder = new MutableLong();
1060    MultiVersionConcurrencyControl.WriteEntry we = key.getMvcc().begin(() -> {
1061      txidHolder.setValue(ringBuffer.next());
1062    });
1063    long txid = txidHolder.longValue();
1064    try (TraceScope scope = TraceUtil.createTrace(implClassName + ".append")) {
1065      FSWALEntry entry = new FSWALEntry(txid, key, edits, hri, inMemstore);
1066      entry.stampRegionSequenceId(we);
1067      ringBuffer.get(txid).load(entry);
1068    } finally {
1069      ringBuffer.publish(txid);
1070    }
1071    return txid;
1072  }

1136  /**
1137   * @return true if we exceeded the slow sync roll threshold over the last check
1138   *              interval
1139   */
1140  protected boolean doCheckSlowSync() {
1141    boolean result = false;
1142    long now = EnvironmentEdgeManager.currentTime();
1143    long elapsedTime = now - lastTimeCheckSlowSync;
1144    if (elapsedTime >= slowSyncCheckInterval) {
1145      if (slowSyncCount.get() >= slowSyncRollThreshold) {
1146        if (elapsedTime >= (2 * slowSyncCheckInterval)) {
1147          // If two or more slowSyncCheckInterval have elapsed this is a corner case
1148          // where a train of slow syncs almost triggered us but then there was a long
1149          // interval from then until the one more that pushed us over. If so, we
1150          // should do nothing and let the count reset.
1151          if (LOG.isDebugEnabled()) {
1152            LOG.debug("checkSlowSync triggered but we decided to ignore it; " +
1153                "count=" + slowSyncCount.get() + ", threshold=" + slowSyncRollThreshold +
1154                ", elapsedTime=" + elapsedTime +  " ms, slowSyncCheckInterval=" +
1155                slowSyncCheckInterval + " ms");
1156          }
1157          // Fall through to count reset below
1158        } else {
1159          LOG.warn("Requesting log roll because we exceeded slow sync threshold; count=" +
1160            slowSyncCount.get() + ", threshold=" + slowSyncRollThreshold +
1161            ", current pipeline: " + Arrays.toString(getPipeline()));
1162          result = true;
1163        }
1164      }
1165      lastTimeCheckSlowSync = now;
1166      slowSyncCount.set(0);
1167    }
1168    return result;
1169  }

1171  public void checkLogLowReplication(long checkInterval) {
1172    long now = EnvironmentEdgeManager.currentTime();
1173    if (now - lastTimeCheckLowReplication < checkInterval) {
1174      return;
1175    }
1176    // Will return immediately if we are in the middle of a WAL log roll currently.
1177    if (!rollWriterLock.tryLock()) {
1178      return;
1179    }
1180    try {
1181      lastTimeCheckLowReplication = now;
1182      if (doCheckLogLowReplication()) {
1183        requestLogRoll(LOW_REPLICATION);
1184      }
1185    } finally {
1186      rollWriterLock.unlock();
1187    }

1202  private static void split(final Configuration conf, final Path p) throws IOException {
1203    FileSystem fs = FSUtils.getWALFileSystem(conf);
1204    if (!fs.exists(p)) {
1205      throw new FileNotFoundException(p.toString());
1206    }
1207    if (!fs.getFileStatus(p).isDirectory()) {
1208      throw new IOException(p + " is not a directory");
1209    }
1210
1211    final Path baseDir = FSUtils.getWALRootDir(conf);
1212    Path archiveDir = new Path(baseDir, HConstants.HREGION_OLDLOGDIR_NAME);
1213    if (conf.getBoolean(AbstractFSWALProvider.SEPARATE_OLDLOGDIR,
1214      AbstractFSWALProvider.DEFAULT_SEPARATE_OLDLOGDIR)) {
1215      archiveDir = new Path(archiveDir, p.getName());
1216    }
1217    WALSplitter.split(baseDir, p, archiveDir, fs, conf, WALFactory.getInstance(conf));
1218  }

1231  /**
1232   * Pass one or more log file names and it will either dump out a text version on
1233   * <code>stdout</code> or split the specified log files.
1234   */
1235  public static void main(String[] args) throws IOException {
1236    if (args.length < 2) {
1237      usage();
1238      System.exit(-1);
1239    }
1240    // either dump using the WALPrettyPrinter or split, depending on args
1241    if (args[0].compareTo("--dump") == 0) {
1242      WALPrettyPrinter.run(Arrays.copyOfRange(args, 1, args.length));
1243    } else if (args[0].compareTo("--perf") == 0) {
1244      LOG.error(HBaseMarkers.FATAL, "Please use the WALPerformanceEvaluation tool instead. i.e.:");
1245      LOG.error(HBaseMarkers.FATAL,
1246        "\thbase org.apache.hadoop.hbase.wal.WALPerformanceEvaluation --iterations " + args[1]);
1247      System.exit(-1);
1248    } else if (args[0].compareTo("--split") == 0) {
1249      Configuration conf = HBaseConfiguration.create();
1250      for (int i = 1; i < args.length; i++) {
1251        try {
1252          Path logPath = new Path(args[i]);
1253          FSUtils.setFsDefault(conf, logPath);
1254          split(conf, logPath);
1255        } catch (IOException t) {
1256          t.printStackTrace(System.err);
1257          System.exit(-1);
1258        }
1259      }
1260    } else {
1261      usage();
1262      System.exit(-1);
1263    }
1264  }
