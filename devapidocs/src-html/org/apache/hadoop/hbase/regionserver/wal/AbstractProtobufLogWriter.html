080  private WALHeader buildWALHeader0(Configuration conf, WALHeader.Builder builder) {
081    if (!builder.hasWriterClsName()) {
082      builder.setWriterClsName(getWriterClassName());
083    }
084    if (!builder.hasCellCodecClsName()) {
085      builder.setCellCodecClsName(WALCellCodec.getWALCellCodecClass(conf).getName());
086    }
087    return builder.build();
088  }

090  protected WALHeader buildWALHeader(Configuration conf, WALHeader.Builder builder)

092    return buildWALHeader0(conf, builder);

095  // should be called in sub classes's buildWALHeader method to build WALHeader for secure
096  // environment. Do not forget to override the setEncryptor method as it will be called in this
097  // method to init your encryptor.
098  protected final WALHeader buildSecureWALHeader(Configuration conf, WALHeader.Builder builder)
099    throws IOException {
100    builder.setWriterClsName(getWriterClassName());
101    if (conf.getBoolean(HConstants.ENABLE_WAL_ENCRYPTION, false)) {
102      EncryptionTest.testKeyProvider(conf);
103      EncryptionTest.testCipherProvider(conf);
104
105      // Get an instance of our cipher
106      final String cipherName =
107        conf.get(HConstants.CRYPTO_WAL_ALGORITHM_CONF_KEY, HConstants.CIPHER_AES);
108      Cipher cipher = Encryption.getCipher(conf, cipherName);
109      if (cipher == null) {
110        throw new RuntimeException("Cipher '" + cipherName + "' is not available");
111      }
112
113      // Generate a random encryption key for this WAL
114      Key key = cipher.getRandomKey();
115      builder.setEncryptionKey(UnsafeByteOperations.unsafeWrap(EncryptionUtil.wrapKey(conf,
116        conf.get(HConstants.CRYPTO_WAL_KEY_NAME_CONF_KEY,
117          conf.get(HConstants.CRYPTO_MASTERKEY_NAME_CONF_KEY, User.getCurrent().getShortName())),
118        key)));
119
120      // Set up the encryptor
121      Encryptor encryptor = cipher.getEncryptor();
122      encryptor.setKey(key);
123      setEncryptor(encryptor);
124      if (LOG.isTraceEnabled()) {
125        LOG.trace("Initialized secure protobuf WAL: cipher=" + cipher.getName());
126      }
127    }
128    builder.setCellCodecClsName(SecureWALCellCodec.class.getName());
129    return buildWALHeader0(conf, builder);
130  }

140  private boolean initializeCompressionContext(Configuration conf, Path path) throws IOException {
141    boolean doCompress = conf.getBoolean(HConstants.ENABLE_WAL_COMPRESSION, false);
142    if (doCompress) {
143      try {
144        final boolean useTagCompression =
145          conf.getBoolean(CompressionContext.ENABLE_WAL_TAGS_COMPRESSION, true);
146        final boolean useValueCompression =
147          conf.getBoolean(CompressionContext.ENABLE_WAL_VALUE_COMPRESSION, false);
148        final Compression.Algorithm valueCompressionType = useValueCompression
149          ? CompressionContext.getValueCompressionAlgorithm(conf)
150          : Compression.Algorithm.NONE;
151        if (LOG.isTraceEnabled()) {
152          LOG.trace(
153            "Initializing compression context for {}: isRecoveredEdits={}"
154              + ", hasTagCompression={}, hasValueCompression={}, valueCompressionType={}",
155            path, CommonFSUtils.isRecoveredEdits(path), useTagCompression, useValueCompression,
156            valueCompressionType);
157        }
158        this.compressionContext =
159          new CompressionContext(LRUDictionary.class, CommonFSUtils.isRecoveredEdits(path),
160            useTagCompression, useValueCompression, valueCompressionType);
161      } catch (Exception e) {
162        throw new IOException("Failed to initiate CompressionContext", e);
163      }
164    }
165    return doCompress;
166  }
167
168  public void init(FileSystem fs, Path path, Configuration conf, boolean overwritable,
169    long blocksize, StreamSlowMonitor monitor) throws IOException, StreamLacksCapabilityException {
170    try {
171      this.conf = conf;
172      boolean doCompress = initializeCompressionContext(conf, path);
173      this.trailerWarnSize = conf.getInt(WAL_TRAILER_WARN_SIZE, DEFAULT_WAL_TRAILER_WARN_SIZE);
174      int bufferSize = CommonFSUtils.getDefaultBufferSize(fs);
175      short replication = (short) conf.getInt("hbase.regionserver.hlog.replication",
176        CommonFSUtils.getDefaultReplication(fs, path));
177
178      initOutput(fs, path, overwritable, bufferSize, replication, blocksize, monitor);
179
180      boolean doTagCompress =
181        doCompress && conf.getBoolean(CompressionContext.ENABLE_WAL_TAGS_COMPRESSION, true);
182      boolean doValueCompress =
183        doCompress && conf.getBoolean(CompressionContext.ENABLE_WAL_VALUE_COMPRESSION, false);
184      WALHeader.Builder headerBuilder = WALHeader.newBuilder().setHasCompression(doCompress)
185        .setHasTagCompression(doTagCompress).setHasValueCompression(doValueCompress);
186      if (doValueCompress) {
187        headerBuilder.setValueCompressionAlgorithm(
188          CompressionContext.getValueCompressionAlgorithm(conf).ordinal());
189      }
190      length.set(writeMagicAndWALHeader(PB_WAL_MAGIC, buildWALHeader(conf, headerBuilder)));
191
192      initAfterHeader(doCompress);
193
194      // instantiate trailer to default value.
195      trailer = WALTrailer.newBuilder().build();
196
197      if (LOG.isTraceEnabled()) {
198        LOG.trace("Initialized protobuf WAL={}, compression={}, tagCompression={}"
199          + ", valueCompression={}", path, doCompress, doTagCompress, doValueCompress);
200      }
201    } catch (Exception e) {
202      LOG.warn("Init output failed, path={}", path, e);
203      closeOutputIfNecessary();
204      throw e;
205    }
206  }
207
208  private void initAfterHeader0(boolean doCompress) throws IOException {
209    WALCellCodec codec = getCodec(conf, this.compressionContext);
210    this.cellEncoder = codec.getEncoder(getOutputStreamForCellEncoder());
211    if (doCompress) {
212      this.compressor = codec.getByteStringCompressor();
213    } else {
214      this.compressor = WALCellCodec.getNoneCompressor();
215    }
216  }
217
218  protected void initAfterHeader(boolean doCompress) throws IOException {
219    initAfterHeader0(doCompress);
220  }
221
222  // should be called in sub classes's initAfterHeader method to init SecureWALCellCodec.
223  protected final void secureInitAfterHeader(boolean doCompress, Encryptor encryptor)
224    throws IOException {
225    if (conf.getBoolean(HConstants.ENABLE_WAL_ENCRYPTION, false) && encryptor != null) {
226      WALCellCodec codec = SecureWALCellCodec.getCodec(this.conf, encryptor);
227      this.cellEncoder = codec.getEncoder(getOutputStreamForCellEncoder());
228      // We do not support compression
229      this.compressionContext = null;
230      this.compressor = WALCellCodec.getNoneCompressor();
231    } else {
232      initAfterHeader0(doCompress);
233    }
234  }

248  protected void writeWALTrailer() {
249    try {
250      int trailerSize = 0;
251      if (this.trailer == null) {
252        // use default trailer.
253        LOG.warn("WALTrailer is null. Continuing with default.");
254        this.trailer = buildWALTrailer(WALTrailer.newBuilder());
255        trailerSize = this.trailer.getSerializedSize();
256      } else if ((trailerSize = this.trailer.getSerializedSize()) > this.trailerWarnSize) {
257        // continue writing after warning the user.
258        LOG.warn("Please investigate WALTrailer usage. Trailer size > maximum size : " + trailerSize
259          + " > " + this.trailerWarnSize);
260      }
261      length.set(writeWALTrailerAndMagic(trailer, PB_WAL_COMPLETE_MAGIC));
262      this.trailerWritten = true;
263    } catch (IOException ioe) {
264      LOG.warn("Failed to write trailer, non-fatal, continuing...", ioe);
265    }
266  }
