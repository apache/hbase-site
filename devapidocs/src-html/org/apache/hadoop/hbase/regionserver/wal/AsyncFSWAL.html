067/**
068 * An asynchronous implementation of FSWAL.
069 * <p>
070 * Here 'waitingConsumePayloads' acts as the RingBuffer in FSHLog.
071 * <p>
072 * For append, we process it as follow:
073 * <ol>
074 * <li>In the caller thread(typically, in the rpc handler thread):
075 * <ol>
076 * <li>Insert the entry into 'waitingConsumePayloads'. Use ringbuffer sequence as txid.</li>
077 * <li>Schedule the consumer task if needed. See {@link #shouldScheduleConsumer()} for more details.
078 * </li>
079 * </ol>
080 * </li>
081 * <li>In the consumer task(executed in a single threaded thread pool)
082 * <ol>
083 * <li>Poll the entry from {@link #waitingConsumePayloads} and insert it into
084 * {@link #toWriteAppends}</li>
085 * <li>Poll the entry from {@link #toWriteAppends}, append it to the AsyncWriter, and insert it into
086 * {@link #unackedAppends}</li>
087 * <li>If the buffered size reaches {@link #batchSize}, or there is a sync request, then we call
088 * sync on the AsyncWriter.</li>
089 * <li>In the callback methods:
090 * <ul>
091 * <li>If succeeded, poll the entry from {@link #unackedAppends} and drop it.</li>
092 * <li>If failed, add all the entries in {@link #unackedAppends} back to {@link #toWriteAppends} and
093 * wait for writing them again.</li>
094 * </ul>
095 * </li>
096 * </ol>
097 * </li>
098 * </ol>
099 * For sync, the processing stages are almost same. And different from FSHLog, we will open a new
100 * writer and rewrite unacked entries to the new writer and sync again if we hit a sync error.
101 * <p>
102 * Here we only describe the logic of doReplaceWriter. The main logic of rollWriter is same with
103 * FSHLog.<br>
104 * For a normal roll request(for example, we have reached the log roll size):
105 * <ol>
106 * <li>In the log roller thread, we will set {@link #waitingRoll} to true and
107 * {@link #readyForRolling} to false, and then wait on {@link #readyForRolling}(see
108 * {@link #waitForSafePoint()}).</li>
109 * <li>In the consumer thread, we will stop polling entries from {@link #waitingConsumePayloads} if
110 * {@link #waitingRoll} is true, and also stop writing the entries in {@link #toWriteAppends} out.
111 * </li>
112 * <li>If there are unflush data in the writer, sync them.</li>
113 * <li>When all out-going sync request is finished, i.e, the {@link #unackedAppends} is empty,
114 * signal the {@link #readyForRollingCond}.</li>
115 * <li>Back to the log roller thread, now we can confirm that there are no out-going entries, i.e.,
116 * we reach a safe point. So it is safe to replace old writer with new writer now.</li>
117 * <li>Set {@link #writerBroken} and {@link #waitingRoll} to false.</li>
118 * <li>Schedule the consumer task.</li>
119 * <li>Schedule a background task to close the old writer.</li>
120 * </ol>
121 * For a broken writer roll request, the only difference is that we can bypass the wait for safe
122 * point stage.
123 */

204  public AsyncFSWAL(FileSystem fs, Path rootDir, String logDir, String archiveDir,
205      Configuration conf, List<WALActionsListener> listeners, boolean failIfWALExists,
206      String prefix, String suffix, EventLoopGroup eventLoopGroup,
207      Class<? extends Channel> channelClass) throws FailedLogCloseException, IOException {
208    super(fs, rootDir, logDir, archiveDir, conf, listeners, failIfWALExists, prefix, suffix);
209    this.eventLoopGroup = eventLoopGroup;
210    this.channelClass = channelClass;
211    Supplier<Boolean> hasConsumerTask;
212    if (conf.getBoolean(ASYNC_WAL_USE_SHARED_EVENT_LOOP, DEFAULT_ASYNC_WAL_USE_SHARED_EVENT_LOOP)) {
213      this.consumeExecutor = eventLoopGroup.next();
214      if (consumeExecutor instanceof SingleThreadEventExecutor) {
215        try {
216          Field field = SingleThreadEventExecutor.class.getDeclaredField("taskQueue");
217          field.setAccessible(true);
218          Queue<?> queue = (Queue<?>) field.get(consumeExecutor);
219          hasConsumerTask = () -> queue.peek() == consumer;
220        } catch (Exception e) {
221          LOG.warn("Can not get task queue of " + consumeExecutor +
222            ", this is not necessary, just give up", e);
223          hasConsumerTask = () -> false;
224        }
225      } else {
226        hasConsumerTask = () -> false;
227      }
228    } else {
229      ThreadPoolExecutor threadPool =
230        new ThreadPoolExecutor(1, 1, 0L, TimeUnit.MILLISECONDS, new LinkedBlockingQueue<Runnable>(),
231            new ThreadFactoryBuilder().setNameFormat("AsyncFSWAL-%d").setDaemon(true).build());
232      hasConsumerTask = () -> threadPool.getQueue().peek() == consumer;
233      this.consumeExecutor = threadPool;
234    }
235
236    this.hasConsumerTask = hasConsumerTask;
237    int preallocatedEventCount =
238      conf.getInt("hbase.regionserver.wal.disruptor.event.count", 1024 * 16);
239    waitingConsumePayloads =
240      RingBuffer.createMultiProducer(RingBufferTruck::new, preallocatedEventCount);
241    waitingConsumePayloadsGatingSequence = new Sequence(Sequencer.INITIAL_CURSOR_VALUE);
242    waitingConsumePayloads.addGatingSequences(waitingConsumePayloadsGatingSequence);
243
244    // inrease the ringbuffer sequence so our txid is start from 1
245    waitingConsumePayloads.publish(waitingConsumePayloads.next());
246    waitingConsumePayloadsGatingSequence.set(waitingConsumePayloads.getCursor());
247
248    batchSize = conf.getLong(WAL_BATCH_SIZE, DEFAULT_WAL_BATCH_SIZE);
249    waitOnShutdownInSeconds = conf.getInt(ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS,
250      DEFAULT_ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS);
251  }

253  private static boolean waitingRoll(int epochAndState) {
254    return (epochAndState & 1) != 0;
255  }
256
257  private static boolean writerBroken(int epochAndState) {
258    return ((epochAndState >>> 1) & 1) != 0;
259  }
260
261  private static int epoch(int epochAndState) {
262    return epochAndState >>> 2;
263  }

265  // return whether we have successfully set readyForRolling to true.
266  private boolean trySetReadyForRolling() {
267    // Check without holding lock first. Usually we will just return here.
268    // waitingRoll is volatile and unacedEntries is only accessed inside event loop so it is safe to
269    // check them outside the consumeLock.
270    if (!waitingRoll(epochAndState) || !unackedAppends.isEmpty()) {
271      return false;
272    }
273    consumeLock.lock();
274    try {
275      // 1. a roll is requested
276      // 2. all out-going entries have been acked(we have confirmed above).
277      if (waitingRoll(epochAndState)) {
278        readyForRolling = true;
279        readyForRollingCond.signalAll();
280        return true;
281      } else {
282        return false;
283      }
284    } finally {
285      consumeLock.unlock();
286    }
287  }

289  private void syncFailed(long epochWhenSync, Throwable error) {
290    LOG.warn("sync failed", error);
291    boolean shouldRequestLogRoll = true;
292    consumeLock.lock();
293    try {
294      int currentEpochAndState = epochAndState;
295      if (epoch(currentEpochAndState) != epochWhenSync || writerBroken(currentEpochAndState)) {
296        // this is not the previous writer which means we have already rolled the writer.
297        // or this is still the current writer, but we have already marked it as broken and request
298        // a roll.
299        return;
300      }
301      this.epochAndState = currentEpochAndState | 0b10;
302      if (waitingRoll(currentEpochAndState)) {
303        readyForRolling = true;
304        readyForRollingCond.signalAll();
305        // this means we have already in the middle of a rollWriter so just tell the roller thread
306        // that you can continue without requesting an extra log roll.
307        shouldRequestLogRoll = false;
308      }
309    } finally {
310      consumeLock.unlock();
311    }
312    for (Iterator<FSWALEntry> iter = unackedAppends.descendingIterator(); iter.hasNext();) {
313      toWriteAppends.addFirst(iter.next());
314    }
315    highestUnsyncedTxid = highestSyncedTxid.get();
316    if (shouldRequestLogRoll) {
317      // request a roll.
318      requestLogRoll();
319    }
320  }

322  private void syncCompleted(AsyncWriter writer, long processedTxid, long startTimeNs) {
323    highestSyncedTxid.set(processedTxid);
324    for (Iterator<FSWALEntry> iter = unackedAppends.iterator(); iter.hasNext();) {
325      if (iter.next().getTxid() <= processedTxid) {
326        iter.remove();
327      } else {
328        break;
329      }
330    }
331    postSync(System.nanoTime() - startTimeNs, finishSync(true));
332    if (trySetReadyForRolling()) {
333      // we have just finished a roll, then do not need to check for log rolling, the writer will be
334      // closed soon.
335      return;
336    }
337    if (writer.getLength() < logrollsize || rollRequested) {
338      return;
339    }
340    rollRequested = true;
341    requestLogRoll();
342  }

344  private void sync(AsyncWriter writer) {
345    fileLengthAtLastSync = writer.getLength();
346    long currentHighestProcessedAppendTxid = highestProcessedAppendTxid;
347    highestProcessedAppendTxidAtLastSync = currentHighestProcessedAppendTxid;
348    final long startTimeNs = System.nanoTime();
349    final long epoch = (long) epochAndState >>> 2L;
350    writer.sync().whenCompleteAsync((result, error) -> {
351      if (error != null) {
352        syncFailed(epoch, error);
353      } else {
354        syncCompleted(writer, currentHighestProcessedAppendTxid, startTimeNs);
355      }
356    }, consumeExecutor);
357  }

365  private int finishSyncLowerThanTxid(long txid, boolean addSyncTrace) {
366    int finished = 0;
367    for (Iterator<SyncFuture> iter = syncFutures.iterator(); iter.hasNext();) {
368      SyncFuture sync = iter.next();
369      if (sync.getTxid() <= txid) {
370        sync.done(txid, null);
371        iter.remove();
372        finished++;
373        if (addSyncTrace) {
374          addTimeAnnotation(sync, "writer synced");
375        }
376      } else {
377        break;
378      }
379    }
380    return finished;
381  }

383  // try advancing the highestSyncedTxid as much as possible
384  private int finishSync(boolean addSyncTrace) {
385    if (unackedAppends.isEmpty()) {
386      // All outstanding appends have been acked.
387      if (toWriteAppends.isEmpty()) {
388        // Also no appends that wait to be written out, then just finished all pending syncs.
389        long maxSyncTxid = highestSyncedTxid.get();
390        for (SyncFuture sync : syncFutures) {
391          maxSyncTxid = Math.max(maxSyncTxid, sync.getTxid());
392          sync.done(maxSyncTxid, null);
393          if (addSyncTrace) {
394            addTimeAnnotation(sync, "writer synced");
395          }
396        }
397        highestSyncedTxid.set(maxSyncTxid);
398        int finished = syncFutures.size();
399        syncFutures.clear();
400        return finished;
401      } else {
402        // There is no append between highestProcessedAppendTxid and lowestUnprocessedAppendTxid, so
403        // if highestSyncedTxid >= highestProcessedAppendTxid, then all syncs whose txid are between
404        // highestProcessedAppendTxid and lowestUnprocessedAppendTxid can be finished.
405        long lowestUnprocessedAppendTxid = toWriteAppends.peek().getTxid();
406        assert lowestUnprocessedAppendTxid > highestProcessedAppendTxid;
407        long doneTxid = lowestUnprocessedAppendTxid - 1;
408        highestSyncedTxid.set(doneTxid);
409        return finishSyncLowerThanTxid(doneTxid, addSyncTrace);
410      }
411    } else {
412      // There are still unacked appends. So let's move the highestSyncedTxid to the txid of the
413      // first unacked append minus 1.
414      long lowestUnackedAppendTxid = unackedAppends.peek().getTxid();
415      long doneTxid = Math.max(lowestUnackedAppendTxid - 1, highestSyncedTxid.get());
416      highestSyncedTxid.set(doneTxid);
417      return finishSyncLowerThanTxid(doneTxid, addSyncTrace);
418    }
419  }

421  private void appendAndSync() {
422    final AsyncWriter writer = this.writer;
423    // maybe a sync request is not queued when we issue a sync, so check here to see if we could
424    // finish some.
425    finishSync(false);
426    long newHighestProcessedAppendTxid = -1L;
427    for (Iterator<FSWALEntry> iter = toWriteAppends.iterator(); iter.hasNext();) {
428      FSWALEntry entry = iter.next();
429      boolean appended;
430      try {
431        appended = append(writer, entry);
432      } catch (IOException e) {
433        throw new AssertionError("should not happen", e);
434      }
435      newHighestProcessedAppendTxid = entry.getTxid();
436      iter.remove();
437      if (appended) {
438        unackedAppends.addLast(entry);
439        if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
440          break;
441        }
442      }
443    }
444    // if we have a newer transaction id, update it.
445    // otherwise, use the previous transaction id.
446    if (newHighestProcessedAppendTxid > 0) {
447      highestProcessedAppendTxid = newHighestProcessedAppendTxid;
448    } else {
449      newHighestProcessedAppendTxid = highestProcessedAppendTxid;
450    }
451
452    if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
453      // sync because buffer size limit.
454      sync(writer);
455      return;
456    }
457    if (writer.getLength() == fileLengthAtLastSync) {
458      // we haven't written anything out, just advance the highestSyncedSequence since we may only
459      // stamped some region sequence id.
460      if (unackedAppends.isEmpty()) {
461        highestSyncedTxid.set(highestProcessedAppendTxid);
462        finishSync(false);
463        trySetReadyForRolling();
464      }
465      return;
466    }
467    // reach here means that we have some unsynced data but haven't reached the batch size yet
468    // but we will not issue a sync directly here even if there are sync requests because we may
469    // have some new data in the ringbuffer, so let's just return here and delay the decision of
470    // whether to issue a sync in the caller method.
471  }

473  private void drainNonMarkerEditsAndFailSyncs() {
474    if (toWriteAppends.isEmpty()) {
475      return;
476    }
477    boolean hasNonMarkerEdits = false;
478    Iterator<FSWALEntry> iter = toWriteAppends.descendingIterator();
479    while (iter.hasNext()) {
480      FSWALEntry entry = iter.next();
481      if (!entry.getEdit().isMetaEdit()) {
482        hasNonMarkerEdits = true;
483        break;
484      }
485    }
486    if (hasNonMarkerEdits) {
487      for (;;) {
488        iter.remove();
489        if (!iter.hasNext()) {
490          break;
491        }
492        iter.next();
493      }
494      unackedAppends.clear();
495      // fail the sync futures which are under the txid of the first remaining edit, if none, fail
496      // all the sync futures.
497      long txid = toWriteAppends.isEmpty() ? Long.MAX_VALUE : toWriteAppends.peek().getTxid();
498      IOException error = new IOException("WAL is closing, only marker edit is allowed");
499      for (Iterator<SyncFuture> syncIter = syncFutures.iterator(); syncIter.hasNext();) {
500        SyncFuture future = syncIter.next();
501        if (future.getTxid() < txid) {
502          future.done(future.getTxid(), error);
503          syncIter.remove();
504        } else {
505          break;
506        }
507      }
508    }
509  }

511  private void consume() {
512    consumeLock.lock();
513    try {
514      int currentEpochAndState = epochAndState;
515      if (writerBroken(currentEpochAndState)) {
516        return;
517      }
518      if (waitingRoll(currentEpochAndState)) {
519        if (writer.getLength() > fileLengthAtLastSync) {
520          // issue a sync
521          sync(writer);
522        } else {
523          if (unackedAppends.isEmpty()) {
524            readyForRolling = true;
525            readyForRollingCond.signalAll();
526          }
527        }
528        return;
529      }
530    } finally {
531      consumeLock.unlock();
532    }
533    long nextCursor = waitingConsumePayloadsGatingSequence.get() + 1;
534    for (long cursorBound = waitingConsumePayloads.getCursor(); nextCursor <= cursorBound;
535      nextCursor++) {
536      if (!waitingConsumePayloads.isPublished(nextCursor)) {
537        break;
538      }
539      RingBufferTruck truck = waitingConsumePayloads.get(nextCursor);
540      switch (truck.type()) {
541        case APPEND:
542          toWriteAppends.addLast(truck.unloadAppend());
543          break;
544        case SYNC:
545          syncFutures.add(truck.unloadSync());
546          break;
547        default:
548          LOG.warn("RingBufferTruck with unexpected type: " + truck.type());
549          break;
550      }
551      waitingConsumePayloadsGatingSequence.set(nextCursor);
552    }
553    if (markerEditOnly()) {
554      drainNonMarkerEditsAndFailSyncs();
555    }
556    appendAndSync();
557    if (hasConsumerTask.get()) {
558      return;
559    }
560    if (toWriteAppends.isEmpty()) {
561      if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
562        consumerScheduled.set(false);
563        // recheck here since in append and sync we do not hold the consumeLock. Thing may
564        // happen like
565        // 1. we check cursor, no new entry
566        // 2. someone publishes a new entry to ringbuffer and the consumerScheduled is true and
567        // give up scheduling the consumer task.
568        // 3. we set consumerScheduled to false and also give up scheduling consumer task.
569        if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
570          // we will give up consuming so if there are some unsynced data we need to issue a sync.
571          if (writer.getLength() > fileLengthAtLastSync && !syncFutures.isEmpty() &&
572            syncFutures.last().getTxid() > highestProcessedAppendTxidAtLastSync) {
573            // no new data in the ringbuffer and we have at least one sync request
574            sync(writer);
575          }
576          return;
577        } else {
578          // maybe someone has grabbed this before us
579          if (!consumerScheduled.compareAndSet(false, true)) {
580            return;
581          }
582        }
583      }
584    }
585    // reschedule if we still have something to write.
586    consumeExecutor.execute(consumer);
587  }

589  private boolean shouldScheduleConsumer() {
590    int currentEpochAndState = epochAndState;
591    if (writerBroken(currentEpochAndState) || waitingRoll(currentEpochAndState)) {
592      return false;
593    }
594    return consumerScheduled.compareAndSet(false, true);
595  }

603  @Override
604  public long append(RegionInfo hri, WALKeyImpl key, WALEdit edits, boolean inMemstore)
605      throws IOException {
606    if (markerEditOnly() && !edits.isMetaEdit()) {
607      throw new IOException("WAL is closing, only marker edit is allowed");
608    }
609    long txid =
610      stampSequenceIdAndPublishToRingBuffer(hri, key, edits, inMemstore, waitingConsumePayloads);
611    if (shouldScheduleConsumer()) {
612      consumeExecutor.execute(consumer);
613    }
614    return txid;
615  }

617  @Override
618  public void sync() throws IOException {
619    try (TraceScope scope = TraceUtil.createTrace("AsyncFSWAL.sync")) {
620      long txid = waitingConsumePayloads.next();
621      SyncFuture future;
622      try {
623        future = getSyncFuture(txid);
624        RingBufferTruck truck = waitingConsumePayloads.get(txid);
625        truck.load(future);
626      } finally {
627        waitingConsumePayloads.publish(txid);
628      }
629      if (shouldScheduleConsumer()) {
630        consumeExecutor.execute(consumer);
631      }
632      blockOnSync(future);
633    }
634  }
635
636  @Override
637  public void sync(long txid) throws IOException {
638    if (highestSyncedTxid.get() >= txid) {
639      return;
640    }
641    try (TraceScope scope = TraceUtil.createTrace("AsyncFSWAL.sync")) {
642      // here we do not use ring buffer sequence as txid
643      long sequence = waitingConsumePayloads.next();
644      SyncFuture future;
645      try {
646        future = getSyncFuture(txid);
647        RingBufferTruck truck = waitingConsumePayloads.get(sequence);
648        truck.load(future);
649      } finally {
650        waitingConsumePayloads.publish(sequence);
651      }
652      if (shouldScheduleConsumer()) {
653        consumeExecutor.execute(consumer);
654      }
655      blockOnSync(future);
656    }
657  }

669  private void waitForSafePoint() {
670    consumeLock.lock();
671    try {
672      int currentEpochAndState = epochAndState;
673      if (writerBroken(currentEpochAndState) || this.writer == null) {
674        return;
675      }
676      consumerScheduled.set(true);
677      epochAndState = currentEpochAndState | 1;
678      readyForRolling = false;
679      consumeExecutor.execute(consumer);
680      while (!readyForRolling) {
681        readyForRollingCond.awaitUninterruptibly();
682      }
683    } finally {
684      consumeLock.unlock();
685    }
686  }

688  protected final long closeWriter(AsyncWriter writer) {
689    if (writer != null) {
690      long fileLength = writer.getLength();
691      closeExecutor.execute(() -> {
692        try {
693          writer.close();
694        } catch (IOException e) {
695          LOG.warn("close old writer failed", e);
696        }
697      });
698      return fileLength;
699    } else {
700      return 0L;
701    }
702  }

704  @Override
705  protected void doReplaceWriter(Path oldPath, Path newPath, AsyncWriter nextWriter)
706      throws IOException {
707    Preconditions.checkNotNull(nextWriter);
708    waitForSafePoint();
709    long oldFileLen = closeWriter(this.writer);
710    logRollAndSetupWalProps(oldPath, newPath, oldFileLen);
711    this.writer = nextWriter;
712    if (nextWriter instanceof AsyncProtobufLogWriter) {
713      this.fsOut = ((AsyncProtobufLogWriter) nextWriter).getOutput();
714    }
715    this.fileLengthAtLastSync = nextWriter.getLength();
716    this.rollRequested = false;
717    this.highestProcessedAppendTxidAtLastSync = 0L;
718    consumeLock.lock();
719    try {
720      consumerScheduled.set(true);
721      int currentEpoch = epochAndState >>> 2;
722      int nextEpoch = currentEpoch == MAX_EPOCH ? 0 : currentEpoch + 1;
723      // set a new epoch and also clear waitingRoll and writerBroken
724      this.epochAndState = nextEpoch << 2;
725      consumeExecutor.execute(consumer);
726    } finally {
727      consumeLock.unlock();
728    }
729  }

731  @Override
732  protected void doShutdown() throws IOException {
733    waitForSafePoint();
734    closeWriter(this.writer);
735    this.writer = null;
736    closeExecutor.shutdown();
737    try {
738      if (!closeExecutor.awaitTermination(waitOnShutdownInSeconds, TimeUnit.SECONDS)) {
739        LOG.error("We have waited " + waitOnShutdownInSeconds + " seconds but" +
740          " the close of async writer doesn't complete." +
741          "Please check the status of underlying filesystem" +
742          " or increase the wait time by the config \"" + ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS +
743          "\"");
744      }
745    } catch (InterruptedException e) {
746      LOG.error("The wait for close of async writer is interrupted");
747      Thread.currentThread().interrupt();
748    }
749    IOException error = new IOException("WAL has been closed");
750    long nextCursor = waitingConsumePayloadsGatingSequence.get() + 1;
751    // drain all the pending sync requests
752    for (long cursorBound = waitingConsumePayloads.getCursor(); nextCursor <= cursorBound;
753      nextCursor++) {
754      if (!waitingConsumePayloads.isPublished(nextCursor)) {
755        break;
756      }
757      RingBufferTruck truck = waitingConsumePayloads.get(nextCursor);
758      switch (truck.type()) {
759        case SYNC:
760          syncFutures.add(truck.unloadSync());
761          break;
762        default:
763          break;
764      }
765    }
766    // and fail them
767    syncFutures.forEach(f -> f.done(f.getTxid(), error));
768    if (!(consumeExecutor instanceof EventLoop)) {
769      consumeExecutor.shutdown();
770    }
771  }
