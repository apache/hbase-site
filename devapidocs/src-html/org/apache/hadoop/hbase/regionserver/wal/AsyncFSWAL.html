072/**
073 * An asynchronous implementation of FSWAL.
074 * <p>
075 * Here 'waitingConsumePayloads' acts as the RingBuffer in FSHLog.
076 * <p>
077 * For append, we process it as follow:
078 * <ol>
079 * <li>In the caller thread(typically, in the rpc handler thread):
080 * <ol>
081 * <li>Insert the entry into 'waitingConsumePayloads'. Use ringbuffer sequence as txid.</li>
082 * <li>Schedule the consumer task if needed. See {@link #shouldScheduleConsumer()} for more details.
083 * </li>
084 * </ol>
085 * </li>
086 * <li>In the consumer task(executed in a single threaded thread pool)
087 * <ol>
088 * <li>Poll the entry from {@link #waitingConsumePayloads} and insert it into
089 * {@link #toWriteAppends}</li>
090 * <li>Poll the entry from {@link #toWriteAppends}, append it to the AsyncWriter, and insert it into
091 * {@link #unackedAppends}</li>
092 * <li>If the buffered size reaches {@link #batchSize}, or there is a sync request, then we call
093 * sync on the AsyncWriter.</li>
094 * <li>In the callback methods:
095 * <ul>
096 * <li>If succeeded, poll the entry from {@link #unackedAppends} and drop it.</li>
097 * <li>If failed, add all the entries in {@link #unackedAppends} back to {@link #toWriteAppends} and
098 * wait for writing them again.</li>
099 * </ul>
100 * </li>
101 * </ol>
102 * </li>
103 * </ol>
104 * For sync, the processing stages are almost same. And different from FSHLog, we will open a new
105 * writer and rewrite unacked entries to the new writer and sync again if we hit a sync error.
106 * <p>
107 * Here we only describe the logic of doReplaceWriter. The main logic of rollWriter is same with
108 * FSHLog.<br>
109 * For a normal roll request(for example, we have reached the log roll size):
110 * <ol>
111 * <li>In the log roller thread, we will set {@link #waitingRoll} to true and
112 * {@link #readyForRolling} to false, and then wait on {@link #readyForRolling}(see
113 * {@link #waitForSafePoint()}).</li>
114 * <li>In the consumer thread, we will stop polling entries from {@link #waitingConsumePayloads} if
115 * {@link #waitingRoll} is true, and also stop writing the entries in {@link #toWriteAppends} out.
116 * </li>
117 * <li>If there are unflush data in the writer, sync them.</li>
118 * <li>When all out-going sync request is finished, i.e, the {@link #unackedAppends} is empty,
119 * signal the {@link #readyForRollingCond}.</li>
120 * <li>Back to the log roller thread, now we can confirm that there are no out-going entries, i.e.,
121 * we reach a safe point. So it is safe to replace old writer with new writer now.</li>
122 * <li>Set {@link #writerBroken} and {@link #waitingRoll} to false.</li>
123 * <li>Schedule the consumer task.</li>
124 * <li>Schedule a background task to close the old writer.</li>
125 * </ol>
126 * For a broken writer roll request, the only difference is that we can bypass the wait for safe
127 * point stage.
128 */

214  public AsyncFSWAL(FileSystem fs, Abortable abortable, Path rootDir, String logDir,
215      String archiveDir, Configuration conf, List<WALActionsListener> listeners,
216      boolean failIfWALExists, String prefix, String suffix, EventLoopGroup eventLoopGroup,
217      Class<? extends Channel> channelClass, StreamSlowMonitor monitor)
218      throws FailedLogCloseException, IOException {
219    super(fs, abortable, rootDir, logDir, archiveDir, conf, listeners, failIfWALExists, prefix,
220        suffix);
221    this.eventLoopGroup = eventLoopGroup;
222    this.channelClass = channelClass;
223    this.streamSlowMonitor = monitor;
224    Supplier<Boolean> hasConsumerTask;
225    if (conf.getBoolean(ASYNC_WAL_USE_SHARED_EVENT_LOOP, DEFAULT_ASYNC_WAL_USE_SHARED_EVENT_LOOP)) {
226      this.consumeExecutor = eventLoopGroup.next();
227      if (consumeExecutor instanceof SingleThreadEventExecutor) {
228        try {
229          Field field = SingleThreadEventExecutor.class.getDeclaredField("taskQueue");
230          field.setAccessible(true);
231          Queue<?> queue = (Queue<?>) field.get(consumeExecutor);
232          hasConsumerTask = () -> queue.peek() == consumer;
233        } catch (Exception e) {
234          LOG.warn("Can not get task queue of " + consumeExecutor +
235            ", this is not necessary, just give up", e);
236          hasConsumerTask = () -> false;
237        }
238      } else {
239        hasConsumerTask = () -> false;
240      }
241    } else {
242      ThreadPoolExecutor threadPool =
243        new ThreadPoolExecutor(1, 1, 0L,
244          TimeUnit.MILLISECONDS, new LinkedBlockingQueue<Runnable>(),
245          new ThreadFactoryBuilder().setNameFormat("AsyncFSWAL-%d-"+ rootDir.toString() +
246              "-prefix:" + (prefix == null ? "default" : prefix).replace("%", "%%"))
247            .setDaemon(true).build());
248      hasConsumerTask = () -> threadPool.getQueue().peek() == consumer;
249      this.consumeExecutor = threadPool;
250    }
251
252    this.hasConsumerTask = hasConsumerTask;
253    int preallocatedEventCount =
254      conf.getInt("hbase.regionserver.wal.disruptor.event.count", 1024 * 16);
255    waitingConsumePayloads =
256      RingBuffer.createMultiProducer(RingBufferTruck::new, preallocatedEventCount);
257    waitingConsumePayloadsGatingSequence = new Sequence(Sequencer.INITIAL_CURSOR_VALUE);
258    waitingConsumePayloads.addGatingSequences(waitingConsumePayloadsGatingSequence);
259
260    // inrease the ringbuffer sequence so our txid is start from 1
261    waitingConsumePayloads.publish(waitingConsumePayloads.next());
262    waitingConsumePayloadsGatingSequence.set(waitingConsumePayloads.getCursor());
263
264    batchSize = conf.getLong(WAL_BATCH_SIZE, DEFAULT_WAL_BATCH_SIZE);
265    waitOnShutdownInSeconds = conf.getInt(ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS,
266      DEFAULT_ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS);
267  }

277  private static boolean waitingRoll(int epochAndState) {
278    return (epochAndState & 1) != 0;

289  // return whether we have successfully set readyForRolling to true.
290  private boolean trySetReadyForRolling() {
291    // Check without holding lock first. Usually we will just return here.
292    // waitingRoll is volatile and unacedEntries is only accessed inside event loop so it is safe to
293    // check them outside the consumeLock.
294    if (!waitingRoll(epochAndState) || !unackedAppends.isEmpty()) {
295      return false;
296    }
297    consumeLock.lock();
298    try {
299      // 1. a roll is requested
300      // 2. all out-going entries have been acked(we have confirmed above).
301      if (waitingRoll(epochAndState)) {
302        readyForRolling = true;
303        readyForRollingCond.signalAll();
304        return true;
305      } else {
306        return false;
307      }
308    } finally {
309      consumeLock.unlock();
310    }
311  }

313  private void syncFailed(long epochWhenSync, Throwable error) {
314    LOG.warn("sync failed", error);
315    boolean shouldRequestLogRoll = true;
316    consumeLock.lock();
317    try {
318      int currentEpochAndState = epochAndState;
319      if (epoch(currentEpochAndState) != epochWhenSync || writerBroken(currentEpochAndState)) {
320        // this is not the previous writer which means we have already rolled the writer.
321        // or this is still the current writer, but we have already marked it as broken and request
322        // a roll.
323        return;
324      }
325      this.epochAndState = currentEpochAndState | 0b10;
326      if (waitingRoll(currentEpochAndState)) {
327        readyForRolling = true;
328        readyForRollingCond.signalAll();
329        // this means we have already in the middle of a rollWriter so just tell the roller thread
330        // that you can continue without requesting an extra log roll.
331        shouldRequestLogRoll = false;
332      }
333    } finally {
334      consumeLock.unlock();
335    }
336    for (Iterator<FSWALEntry> iter = unackedAppends.descendingIterator(); iter.hasNext();) {
337      toWriteAppends.addFirst(iter.next());
338    }
339    highestUnsyncedTxid = highestSyncedTxid.get();
340    if (shouldRequestLogRoll) {
341      // request a roll.
342      requestLogRoll(ERROR);
343    }
344  }

346  private void syncCompleted(AsyncWriter writer, long processedTxid, long startTimeNs) {
347    highestSyncedTxid.set(processedTxid);
348    for (Iterator<FSWALEntry> iter = unackedAppends.iterator(); iter.hasNext();) {
349      FSWALEntry entry = iter.next();
350      if (entry.getTxid() <= processedTxid) {
351        entry.release();
352        iter.remove();
353      } else {
354        break;
355      }
356    }
357    postSync(System.nanoTime() - startTimeNs, finishSync());
358    if (trySetReadyForRolling()) {
359      // we have just finished a roll, then do not need to check for log rolling, the writer will be
360      // closed soon.
361      return;
362    }
363    // If we haven't already requested a roll, check if we have exceeded logrollsize
364    if (!isLogRollRequested() && writer.getLength() > logrollsize) {
365      if (LOG.isDebugEnabled()) {
366        LOG.debug("Requesting log roll because of file size threshold; length=" +
367          writer.getLength() + ", logrollsize=" + logrollsize);
368      }
369      requestLogRoll(SIZE);
370    }
371  }

373  // find all the sync futures between these two txids to see if we need to issue a hsync, if no
374  // sync futures then just use the default one.
375  private boolean isHsync(long beginTxid, long endTxid) {
376    SortedSet<SyncFuture> futures =
377      syncFutures.subSet(new SyncFuture().reset(beginTxid, false),
378          new SyncFuture().reset(endTxid + 1, false));
379    if (futures.isEmpty()) {
380      return useHsync;
381    }
382    for (SyncFuture future : futures) {
383      if (future.isForceSync()) {
384        return true;
385      }
386    }
387    return false;
388  }

390  private void sync(AsyncWriter writer) {
391    fileLengthAtLastSync = writer.getLength();
392    long currentHighestProcessedAppendTxid = highestProcessedAppendTxid;
393    boolean shouldUseHsync =
394      isHsync(highestProcessedAppendTxidAtLastSync, currentHighestProcessedAppendTxid);
395    highestProcessedAppendTxidAtLastSync = currentHighestProcessedAppendTxid;
396    final long startTimeNs = System.nanoTime();
397    final long epoch = (long) epochAndState >>> 2L;
398    addListener(writer.sync(shouldUseHsync), (result, error) -> {
399      if (error != null) {
400        syncFailed(epoch, error);
401      } else {
402        syncCompleted(writer, currentHighestProcessedAppendTxid, startTimeNs);
403      }
404    }, consumeExecutor);
405  }

407  private int finishSyncLowerThanTxid(long txid) {
408    int finished = 0;
409    for (Iterator<SyncFuture> iter = syncFutures.iterator(); iter.hasNext();) {
410      SyncFuture sync = iter.next();
411      if (sync.getTxid() <= txid) {
412        markFutureDoneAndOffer(sync, txid, null);
413        iter.remove();
414        finished++;
415      } else {
416        break;
417      }
418    }
419    return finished;
420  }

422  // try advancing the highestSyncedTxid as much as possible
423  private int finishSync() {
424    if (unackedAppends.isEmpty()) {
425      // All outstanding appends have been acked.
426      if (toWriteAppends.isEmpty()) {
427        // Also no appends that wait to be written out, then just finished all pending syncs.
428        long maxSyncTxid = highestSyncedTxid.get();
429        for (SyncFuture sync : syncFutures) {
430          maxSyncTxid = Math.max(maxSyncTxid, sync.getTxid());
431          markFutureDoneAndOffer(sync, maxSyncTxid, null);
432        }
433        highestSyncedTxid.set(maxSyncTxid);
434        int finished = syncFutures.size();
435        syncFutures.clear();
436        return finished;
437      } else {
438        // There is no append between highestProcessedAppendTxid and lowestUnprocessedAppendTxid, so
439        // if highestSyncedTxid >= highestProcessedAppendTxid, then all syncs whose txid are between
440        // highestProcessedAppendTxid and lowestUnprocessedAppendTxid can be finished.
441        long lowestUnprocessedAppendTxid = toWriteAppends.peek().getTxid();
442        assert lowestUnprocessedAppendTxid > highestProcessedAppendTxid;
443        long doneTxid = lowestUnprocessedAppendTxid - 1;
444        highestSyncedTxid.set(doneTxid);
445        return finishSyncLowerThanTxid(doneTxid);
446      }
447    } else {
448      // There are still unacked appends. So let's move the highestSyncedTxid to the txid of the
449      // first unacked append minus 1.
450      long lowestUnackedAppendTxid = unackedAppends.peek().getTxid();
451      long doneTxid = Math.max(lowestUnackedAppendTxid - 1, highestSyncedTxid.get());
452      highestSyncedTxid.set(doneTxid);
453      return finishSyncLowerThanTxid(doneTxid);
454    }
455  }

457  // confirm non-empty before calling
458  private static long getLastTxid(Deque<FSWALEntry> queue) {
459    return queue.peekLast().getTxid();

462  private void appendAndSync() {
463    final AsyncWriter writer = this.writer;
464    // maybe a sync request is not queued when we issue a sync, so check here to see if we could
465    // finish some.
466    finishSync();
467    long newHighestProcessedAppendTxid = -1L;
468    // this is used to avoid calling peedLast every time on unackedAppends, appendAndAsync is single
469    // threaded, this could save us some cycles
470    boolean addedToUnackedAppends = false;
471    for (Iterator<FSWALEntry> iter = toWriteAppends.iterator(); iter.hasNext();) {
472      FSWALEntry entry = iter.next();
473      boolean appended;
474      try {
475        appended = appendEntry(writer, entry);
476      } catch (IOException e) {
477        throw new AssertionError("should not happen", e);
478      }
479      newHighestProcessedAppendTxid = entry.getTxid();
480      iter.remove();
481      if (appended) {
482        // This is possible, when we fail to sync, we will add the unackedAppends back to
483        // toWriteAppends, so here we may get an entry which is already in the unackedAppends.
484        if (addedToUnackedAppends || unackedAppends.isEmpty() ||
485          getLastTxid(unackedAppends) < entry.getTxid()) {
486          unackedAppends.addLast(entry);
487          addedToUnackedAppends = true;
488        }
489        // See HBASE-25905, here we need to make sure that, we will always write all the entries in
490        // unackedAppends out. As the code in the consume method will assume that, the entries in
491        // unackedAppends have all been sent out so if there is roll request and unackedAppends is
492        // not empty, we could just return as later there will be a syncCompleted call to clear the
493        // unackedAppends, or a syncFailed to lead us to another state.
494        // There could be other ways to fix, such as changing the logic in the consume method, but
495        // it will break the assumption and then (may) lead to a big refactoring. So here let's use
496        // this way to fix first, can optimize later.
497        if (writer.getLength() - fileLengthAtLastSync >= batchSize &&
498          (addedToUnackedAppends || entry.getTxid() >= getLastTxid(unackedAppends))) {
499          break;
500        }
501      }
502    }
503    // if we have a newer transaction id, update it.
504    // otherwise, use the previous transaction id.
505    if (newHighestProcessedAppendTxid > 0) {
506      highestProcessedAppendTxid = newHighestProcessedAppendTxid;
507    } else {
508      newHighestProcessedAppendTxid = highestProcessedAppendTxid;
509    }
510
511    if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
512      // sync because buffer size limit.
513      sync(writer);
514      return;
515    }
516    if (writer.getLength() == fileLengthAtLastSync) {
517      // we haven't written anything out, just advance the highestSyncedSequence since we may only
518      // stamped some region sequence id.
519      if (unackedAppends.isEmpty()) {
520        highestSyncedTxid.set(highestProcessedAppendTxid);
521        finishSync();
522        trySetReadyForRolling();
523      }
524      return;
525    }
526    // reach here means that we have some unsynced data but haven't reached the batch size yet
527    // but we will not issue a sync directly here even if there are sync requests because we may
528    // have some new data in the ringbuffer, so let's just return here and delay the decision of
529    // whether to issue a sync in the caller method.
530  }

532  private void drainNonMarkerEditsAndFailSyncs() {
533    if (toWriteAppends.isEmpty()) {
534      return;
535    }
536    boolean hasNonMarkerEdits = false;
537    Iterator<FSWALEntry> iter = toWriteAppends.descendingIterator();
538    while (iter.hasNext()) {
539      FSWALEntry entry = iter.next();
540      if (!entry.getEdit().isMetaEdit()) {
541        entry.release();
542        hasNonMarkerEdits = true;
543        break;
544      }
545    }
546    if (hasNonMarkerEdits) {
547      for (;;) {
548        iter.remove();
549        if (!iter.hasNext()) {
550          break;
551        }
552        iter.next().release();
553      }
554      for (FSWALEntry entry : unackedAppends) {
555        entry.release();
556      }
557      unackedAppends.clear();
558      // fail the sync futures which are under the txid of the first remaining edit, if none, fail
559      // all the sync futures.
560      long txid = toWriteAppends.isEmpty() ? Long.MAX_VALUE : toWriteAppends.peek().getTxid();
561      IOException error = new IOException("WAL is closing, only marker edit is allowed");
562      for (Iterator<SyncFuture> syncIter = syncFutures.iterator(); syncIter.hasNext();) {
563        SyncFuture future = syncIter.next();
564        if (future.getTxid() < txid) {
565          markFutureDoneAndOffer(future, future.getTxid(), error);
566          syncIter.remove();
567        } else {
568          break;
569        }
570      }
571    }
572  }

574  private void consume() {
575    consumeLock.lock();
576    try {
577      int currentEpochAndState = epochAndState;
578      if (writerBroken(currentEpochAndState)) {
579        return;
580      }
581      if (waitingRoll(currentEpochAndState)) {
582        if (writer.getLength() > fileLengthAtLastSync) {
583          // issue a sync
584          sync(writer);
585        } else {
586          if (unackedAppends.isEmpty()) {
587            readyForRolling = true;
588            readyForRollingCond.signalAll();
589          }
590        }
591        return;
592      }
593    } finally {
594      consumeLock.unlock();
595    }
596    long nextCursor = waitingConsumePayloadsGatingSequence.get() + 1;
597    for (long cursorBound = waitingConsumePayloads.getCursor(); nextCursor <= cursorBound;
598      nextCursor++) {
599      if (!waitingConsumePayloads.isPublished(nextCursor)) {
600        break;
601      }
602      RingBufferTruck truck = waitingConsumePayloads.get(nextCursor);
603      switch (truck.type()) {
604        case APPEND:
605          toWriteAppends.addLast(truck.unloadAppend());
606          break;
607        case SYNC:
608          syncFutures.add(truck.unloadSync());
609          break;
610        default:
611          LOG.warn("RingBufferTruck with unexpected type: " + truck.type());
612          break;
613      }
614      waitingConsumePayloadsGatingSequence.set(nextCursor);
615    }
616    if (markerEditOnly()) {
617      drainNonMarkerEditsAndFailSyncs();
618    }
619    appendAndSync();
620    if (hasConsumerTask.get()) {
621      return;
622    }
623    if (toWriteAppends.isEmpty()) {
624      if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
625        consumerScheduled.set(false);
626        // recheck here since in append and sync we do not hold the consumeLock. Thing may
627        // happen like
628        // 1. we check cursor, no new entry
629        // 2. someone publishes a new entry to ringbuffer and the consumerScheduled is true and
630        // give up scheduling the consumer task.
631        // 3. we set consumerScheduled to false and also give up scheduling consumer task.
632        if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
633          // we will give up consuming so if there are some unsynced data we need to issue a sync.
634          if (writer.getLength() > fileLengthAtLastSync && !syncFutures.isEmpty() &&
635            syncFutures.last().getTxid() > highestProcessedAppendTxidAtLastSync) {
636            // no new data in the ringbuffer and we have at least one sync request
637            sync(writer);
638          }
639          return;
640        } else {
641          // maybe someone has grabbed this before us
642          if (!consumerScheduled.compareAndSet(false, true)) {
643            return;
644          }
645        }
646      }
647    }
648    // reschedule if we still have something to write.
649    consumeExecutor.execute(consumer);
650  }

652  private boolean shouldScheduleConsumer() {
653    int currentEpochAndState = epochAndState;
654    if (writerBroken(currentEpochAndState) || waitingRoll(currentEpochAndState)) {
655      return false;
656    }
657    return consumerScheduled.compareAndSet(false, true);
658  }

666  @Override
667  protected long append(RegionInfo hri, WALKeyImpl key, WALEdit edits, boolean inMemstore)
668    throws IOException {
669    if (markerEditOnly() && !edits.isMetaEdit()) {
670      throw new IOException("WAL is closing, only marker edit is allowed");
671    }
672    long txid =
673      stampSequenceIdAndPublishToRingBuffer(hri, key, edits, inMemstore, waitingConsumePayloads);
674    if (shouldScheduleConsumer()) {
675      consumeExecutor.execute(consumer);
676    }
677    return txid;
678  }

680  @Override
681  protected void doSync(boolean forceSync) throws IOException {
682    long txid = waitingConsumePayloads.next();
683    SyncFuture future;
684    try {
685      future = getSyncFuture(txid, forceSync);
686      RingBufferTruck truck = waitingConsumePayloads.get(txid);
687      truck.load(future);
688    } finally {
689      waitingConsumePayloads.publish(txid);
690    }
691    if (shouldScheduleConsumer()) {
692      consumeExecutor.execute(consumer);
693    }
694    blockOnSync(future);
695  }

697  @Override
698  protected void doSync(long txid, boolean forceSync) throws IOException {
699    if (highestSyncedTxid.get() >= txid) {
700      return;
701    }
702    // here we do not use ring buffer sequence as txid
703    long sequence = waitingConsumePayloads.next();
704    SyncFuture future;
705    try {
706      future = getSyncFuture(txid, forceSync);
707      RingBufferTruck truck = waitingConsumePayloads.get(sequence);
708      truck.load(future);
709    } finally {
710      waitingConsumePayloads.publish(sequence);
711    }
712    if (shouldScheduleConsumer()) {
713      consumeExecutor.execute(consumer);
714    }
715    blockOnSync(future);
716  }

728  private void waitForSafePoint() {
729    consumeLock.lock();
730    try {
731      int currentEpochAndState = epochAndState;
732      if (writerBroken(currentEpochAndState) || this.writer == null) {
733        return;
734      }
735      consumerScheduled.set(true);
736      epochAndState = currentEpochAndState | 1;
737      readyForRolling = false;
738      consumeExecutor.execute(consumer);
739      while (!readyForRolling) {
740        readyForRollingCond.awaitUninterruptibly();
741      }
742    } finally {
743      consumeLock.unlock();
744    }
745  }

747  protected final long closeWriter(AsyncWriter writer, Path path) {
748    if (writer != null) {
749      inflightWALClosures.put(path.getName(), writer);
750      long fileLength = writer.getLength();
751      closeExecutor.execute(() -> {
752        try {
753          writer.close();
754        } catch (IOException e) {
755          LOG.warn("close old writer failed", e);
756        } finally {
757          inflightWALClosures.remove(path.getName());
758        }
759      });
760      return fileLength;
761    } else {
762      return 0L;
763    }
764  }

766  @Override
767  protected void doReplaceWriter(Path oldPath, Path newPath, AsyncWriter nextWriter)
768      throws IOException {
769    Preconditions.checkNotNull(nextWriter);
770    waitForSafePoint();
771    long oldFileLen = closeWriter(this.writer, oldPath);
772    logRollAndSetupWalProps(oldPath, newPath, oldFileLen);
773    this.writer = nextWriter;
774    if (nextWriter instanceof AsyncProtobufLogWriter) {
775      this.fsOut = ((AsyncProtobufLogWriter) nextWriter).getOutput();
776    }
777    this.fileLengthAtLastSync = nextWriter.getLength();
778    this.highestProcessedAppendTxidAtLastSync = 0L;
779    consumeLock.lock();
780    try {
781      consumerScheduled.set(true);
782      int currentEpoch = epochAndState >>> 2;
783      int nextEpoch = currentEpoch == MAX_EPOCH ? 0 : currentEpoch + 1;
784      // set a new epoch and also clear waitingRoll and writerBroken
785      this.epochAndState = nextEpoch << 2;
786      // Reset rollRequested status
787      rollRequested.set(false);
788      consumeExecutor.execute(consumer);
789    } finally {
790      consumeLock.unlock();
791    }
792  }

794  @Override
795  protected void doShutdown() throws IOException {
796    waitForSafePoint();
797    closeWriter(this.writer, getOldPath());
798    this.writer = null;
799    closeExecutor.shutdown();
800    try {
801      if (!closeExecutor.awaitTermination(waitOnShutdownInSeconds, TimeUnit.SECONDS)) {
802        LOG.error("We have waited " + waitOnShutdownInSeconds + " seconds but" +
803          " the close of async writer doesn't complete." +
804          "Please check the status of underlying filesystem" +
805          " or increase the wait time by the config \"" + ASYNC_WAL_WAIT_ON_SHUTDOWN_IN_SECONDS +
806          "\"");
807      }
808    } catch (InterruptedException e) {
809      LOG.error("The wait for close of async writer is interrupted");
810      Thread.currentThread().interrupt();
811    }
812    IOException error = new IOException("WAL has been closed");
813    long nextCursor = waitingConsumePayloadsGatingSequence.get() + 1;
814    // drain all the pending sync requests
815    for (long cursorBound = waitingConsumePayloads.getCursor(); nextCursor <= cursorBound;
816      nextCursor++) {
817      if (!waitingConsumePayloads.isPublished(nextCursor)) {
818        break;
819      }
820      RingBufferTruck truck = waitingConsumePayloads.get(nextCursor);
821      switch (truck.type()) {
822        case SYNC:
823          syncFutures.add(truck.unloadSync());
824          break;
825        default:
826          break;
827      }
828    }
829    // and fail them
830    syncFutures.forEach(f -> markFutureDoneAndOffer(f, f.getTxid(), error));
831    if (!(consumeExecutor instanceof EventLoop)) {
832      consumeExecutor.shutdown();
833    }
834  }
