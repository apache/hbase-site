049/**
050 * A Protobuf based WAL has the following structure:
051 * <p>
052 * &lt;PB_WAL_MAGIC&gt;&lt;WALHeader&gt;&lt;WALEdits&gt;...&lt;WALEdits&gt;&lt;Trailer&gt;
053 * &lt;TrailerSize&gt; &lt;PB_WAL_COMPLETE_MAGIC&gt;
054 * </p>
055 * The Reader reads meta information (WAL Compression state, WALTrailer, etc) in
056 * ProtobufLogReader#initReader(FSDataInputStream). A WALTrailer is an extensible structure which is
057 * appended at the end of the WAL. This is empty for now; it can contain some meta information such
058 * as Region level stats, etc in future.
059 */
060@InterfaceAudience.LimitedPrivate({ HBaseInterfaceAudience.COPROC, HBaseInterfaceAudience.PHOENIX,
061  HBaseInterfaceAudience.CONFIG })
062public class ProtobufLogReader extends ReaderBase {
063  private static final Logger LOG = LoggerFactory.getLogger(ProtobufLogReader.class);
064  // public for WALFactory until we move everything to o.a.h.h.wal
065  @InterfaceAudience.Private
066  public static final byte[] PB_WAL_MAGIC = Bytes.toBytes("PWAL");
067  // public for TestWALSplit
068  @InterfaceAudience.Private
069  public static final byte[] PB_WAL_COMPLETE_MAGIC = Bytes.toBytes("LAWP");
070  /**
071   * Configuration name of WAL Trailer's warning size. If a waltrailer's size is greater than the
072   * configured size, providers should log a warning. e.g. this is used with Protobuf reader/writer.
073   */
074  static final String WAL_TRAILER_WARN_SIZE = "hbase.regionserver.waltrailer.warn.size";
075  static final int DEFAULT_WAL_TRAILER_WARN_SIZE = 1024 * 1024; // 1MB

084  // walEditsStopOffset is the position of the last byte to read. After reading the last WALEdit

104  @InterfaceAudience.Private
105  public long trailerSize() {
106    if (trailerPresent) {
107      // sizeof PB_WAL_COMPLETE_MAGIC + sizof trailerSize + trailer
108      final long calculatedSize =
109        (long) PB_WAL_COMPLETE_MAGIC.length + Bytes.SIZEOF_INT + trailer.getSerializedSize();
110      final long expectedSize = fileLength - walEditsStopOffset;
111      if (expectedSize != calculatedSize) {
112        LOG.warn("After parsing the trailer, we expect the total footer to be {} bytes, but we "
113          + "calculate it as being {}", expectedSize, calculatedSize);
114      }
115      return expectedSize;
116    } else {
117      return -1L;
118    }
119  }

198  protected WALHdrContext readHeader(Builder builder, FSDataInputStream stream) throws IOException {
199    boolean res = builder.mergeDelimitedFrom(stream);
200    if (!res) return new WALHdrContext(WALHdrResult.EOF, null);
201    if (builder.hasWriterClsName() && !getWriterClsNames().contains(builder.getWriterClsName())) {
202      return new WALHdrContext(WALHdrResult.UNKNOWN_WRITER_CLS, null);
203    }
204    String clsName = null;
205    if (builder.hasCellCodecClsName()) {
206      clsName = builder.getCellCodecClsName();
207    }
208    return new WALHdrContext(WALHdrResult.SUCCESS, clsName);
209  }

211  private String initInternal(FSDataInputStream stream, boolean isFirst) throws IOException {
212    close();
213    if (!isFirst) {
214      // Re-compute the file length.
215      this.fileLength = fs.getFileStatus(path).getLen();
216    }
217    long expectedPos = PB_WAL_MAGIC.length;
218    if (stream == null) {
219      stream = fs.open(path);
220      stream.seek(expectedPos);
221    }
222    if (stream.getPos() != expectedPos) {
223      throw new IOException("The stream is at invalid position: " + stream.getPos());
224    }
225    // Initialize metadata or, when we reset, just skip the header.
226    WALProtos.WALHeader.Builder builder = WALProtos.WALHeader.newBuilder();
227    WALHdrContext hdrCtxt = readHeader(builder, stream);
228    WALHdrResult walHdrRes = hdrCtxt.getResult();
229    if (walHdrRes == WALHdrResult.EOF) {
230      throw new EOFException("Couldn't read WAL PB header");
231    }
232    if (walHdrRes == WALHdrResult.UNKNOWN_WRITER_CLS) {
233      throw new IOException("Got unknown writer class: " + builder.getWriterClsName());
234    }
235    if (isFirst) {
236      WALProtos.WALHeader header = builder.build();
237      this.hasCompression = header.hasHasCompression() && header.getHasCompression();
238      this.hasTagCompression = header.hasHasTagCompression() && header.getHasTagCompression();
239      this.hasValueCompression = header.hasHasValueCompression() && header.getHasValueCompression();
240      if (header.hasValueCompressionAlgorithm()) {
241        try {
242          this.valueCompressionType =
243            Compression.Algorithm.values()[header.getValueCompressionAlgorithm()];
244        } catch (ArrayIndexOutOfBoundsException e) {
245          throw new IOException("Invalid compression type", e);
246        }
247      }
248    }
249    this.inputStream = stream;
250    this.walEditsStopOffset = this.fileLength;
251    long currentPosition = stream.getPos();
252    trailerPresent = setTrailerIfPresent();
253    this.seekOnFs(currentPosition);
254    if (LOG.isTraceEnabled()) {
255      LOG.trace("After reading the trailer: walEditsStopOffset: " + this.walEditsStopOffset
256        + ", fileLength: " + this.fileLength + ", " + "trailerPresent: "
257        + (trailerPresent ? "true, size: " + trailer.getSerializedSize() : "false")
258        + ", currentPosition: " + currentPosition);
259    }
260
261    codecClsName = hdrCtxt.getCellCodecClsName();
262
263    return hdrCtxt.getCellCodecClsName();
264  }

266  /**
267   * To check whether a trailer is present in a WAL, it seeks to position (fileLength -
268   * PB_WAL_COMPLETE_MAGIC.size() - Bytes.SIZEOF_INT). It reads the int value to know the size of
269   * the trailer, and checks whether the trailer is present at the end or not by comparing the last
270   * PB_WAL_COMPLETE_MAGIC.size() bytes. In case trailer is not present, it returns false;
271   * otherwise, sets the trailer and sets this.walEditsStopOffset variable up to the point just
272   * before the trailer.
273   * <ul>
274   * The trailer is ignored in case:
275   * <li>fileLength is 0 or not correct (when file is under recovery, etc).
276   * <li>the trailer size is negative.
277   * </ul>
278   * <p>
279   * In case the trailer size > this.trailerMaxSize, it is read after a WARN message.
280   * @return true if a valid trailer is present
281   */
282  private boolean setTrailerIfPresent() {
283    try {
284      long trailerSizeOffset = this.fileLength - (PB_WAL_COMPLETE_MAGIC.length + Bytes.SIZEOF_INT);
285      if (trailerSizeOffset <= 0) return false;// no trailer possible.
286      this.seekOnFs(trailerSizeOffset);
287      // read the int as trailer size.
288      int trailerSize = this.inputStream.readInt();
289      ByteBuffer buf = ByteBuffer.allocate(ProtobufLogReader.PB_WAL_COMPLETE_MAGIC.length);
290      this.inputStream.readFully(buf.array(), buf.arrayOffset(), buf.capacity());
291      if (!Arrays.equals(buf.array(), PB_WAL_COMPLETE_MAGIC)) {
292        LOG.trace("No trailer found.");
293        return false;
294      }
295      if (trailerSize < 0) {
296        LOG.warn("Invalid trailer Size " + trailerSize + ", ignoring the trailer");
297        return false;
298      } else if (trailerSize > this.trailerWarnSize) {
299        // continue reading after warning the user.
300        LOG.warn("Please investigate WALTrailer usage. Trailer size > maximum configured size : "
301          + trailerSize + " > " + this.trailerWarnSize);
302      }
303      // seek to the position where trailer starts.
304      long positionOfTrailer = trailerSizeOffset - trailerSize;
305      this.seekOnFs(positionOfTrailer);
306      // read the trailer.
307      buf = ByteBuffer.allocate(trailerSize);// for trailer.
308      this.inputStream.readFully(buf.array(), buf.arrayOffset(), buf.capacity());
309      trailer = WALTrailer.parseFrom(buf.array());
310      this.walEditsStopOffset = positionOfTrailer;
311      return true;
312    } catch (IOException ioe) {
313      LOG.warn("Got IOE while reading the trailer. Continuing as if no trailer is present.", ioe);
314    }
315    return false;
316  }

359  @Override
360  protected boolean readNext(Entry entry) throws IOException {
361    // OriginalPosition might be < 0 on local fs; if so, it is useless to us.
362    long originalPosition = this.inputStream.getPos();
363    if (trailerPresent && originalPosition > 0 && originalPosition == this.walEditsStopOffset) {
364      LOG.trace("Reached end of expected edits area at offset {}", originalPosition);
365      return false;
366    }
367    WALKey.Builder builder = WALKey.newBuilder();
368    long size = 0;
369    boolean resetPosition = false;
370    // by default, we should reset the compression when seeking back after reading something
371    resetCompression = true;
372    try {
373      long available = -1;
374      try {
375        int firstByte = this.inputStream.read();
376        if (firstByte == -1) {
377          throw new EOFException();
378        }
379        size = CodedInputStream.readRawVarint32(firstByte, this.inputStream);
380        // available may be < 0 on local fs for instance. If so, can't depend on it.
381        available = this.inputStream.available();
382        if (available > 0 && available < size) {
383          // if we quit here, we have just read the length, no actual data yet, which means we
384          // haven't put anything into the compression dictionary yet, so when seeking back to the
385          // last good position, we do not need to reset compression context.
386          // This is very useful for saving the extra effort for reconstructing the compression
387          // dictionary, where we need to read from the beginning instead of just seek to the
388          // position, as DFSInputStream implement the available method, so in most cases we will
389          // reach here if there are not enough data.
390          resetCompression = false;
391          throw new EOFException("Available stream not enough for edit, "
392            + "inputStream.available()= " + this.inputStream.available() + ", " + "entry size= "
393            + size + " at offset = " + this.inputStream.getPos());
394        }
395        ProtobufUtil.mergeFrom(builder, ByteStreams.limit(this.inputStream, size), (int) size);
396      } catch (InvalidProtocolBufferException ipbe) {
397        resetPosition = true;
398        throw (EOFException) new EOFException("Invalid PB, EOF? Ignoring; originalPosition="
399          + originalPosition + ", currentPosition=" + this.inputStream.getPos() + ", messageSize="
400          + size + ", currentAvailable=" + available).initCause(ipbe);
401      }
402      if (!builder.isInitialized()) {
403        // TODO: not clear if we should try to recover from corrupt PB that looks semi-legit.
404        // If we can get the KV count, we could, theoretically, try to get next record.
405        throw new EOFException("Partial PB while reading WAL, "
406          + "probably an unexpected EOF, ignoring. current offset=" + this.inputStream.getPos());
407      }
408      WALKey walKey = builder.build();
409      entry.getKey().readFieldsFromPb(walKey, this.byteStringUncompressor);
410      if (!walKey.hasFollowingKvCount() || 0 == walKey.getFollowingKvCount()) {
411        LOG.trace("WALKey has no KVs that follow it; trying the next one. current offset={}",
412          this.inputStream.getPos());
413        seekOnFs(originalPosition);
414        return false;
415      }
416      int expectedCells = walKey.getFollowingKvCount();
417      long posBefore = this.inputStream.getPos();
418      try {
419        int actualCells = entry.getEdit().readFromCells(cellDecoder, expectedCells);
420        if (expectedCells != actualCells) {
421          resetPosition = true;
422          throw new EOFException("Only read " + actualCells); // other info added in catch
423        }
424      } catch (Exception ex) {
425        String posAfterStr = "<unknown>";
426        try {
427          posAfterStr = this.inputStream.getPos() + "";
428        } catch (Throwable t) {
429          LOG.trace("Error getting pos for error message - ignoring", t);
430        }
431        String message = " while reading " + expectedCells + " WAL KVs; started reading at "
432          + posBefore + " and read up to " + posAfterStr;
433        IOException realEofEx = extractHiddenEof(ex);
434        throw (EOFException) new EOFException("EOF " + message)
435          .initCause(realEofEx != null ? realEofEx : ex);
436      }
437      if (trailerPresent && this.inputStream.getPos() > this.walEditsStopOffset) {
438        LOG.error(
439          "Read WALTrailer while reading WALEdits. wal: " + this.path + ", inputStream.getPos(): "
440            + this.inputStream.getPos() + ", walEditsStopOffset: " + this.walEditsStopOffset);
441        throw new EOFException("Read WALTrailer while reading WALEdits");
442      }
443    } catch (EOFException eof) {
444      // If originalPosition is < 0, it is rubbish and we cannot use it (probably local fs)
445      if (originalPosition < 0) {
446        LOG.debug(
447          "Encountered a malformed edit, but can't seek back to last good position "
448            + "because originalPosition is negative. last offset={}",
449          this.inputStream.getPos(), eof);
450        throw eof;
451      }
452      // If stuck at the same place and we got an exception, lets go back at the beginning.
453      if (inputStream.getPos() == originalPosition) {
454        if (resetPosition) {
455          LOG.debug("Encountered a malformed edit, seeking to the beginning of the WAL since "
456            + "current position and original position match at {}", originalPosition);
457          seekOnFs(0);
458        } else {
459          LOG.debug("EOF at position {}", originalPosition);
460        }
461      } else {
462        // Else restore our position to original location in hope that next time through we will
463        // read successfully.
464        LOG.debug("Encountered a malformed edit, seeking back to last good position in file, "
465          + "from {} to {}", inputStream.getPos(), originalPosition, eof);
466        seekOnFs(originalPosition);
467      }
468      return false;
469    }
470    return true;
471  }

473  private IOException extractHiddenEof(Exception ex) {
474    // There are two problems we are dealing with here. Hadoop stream throws generic exception
475    // for EOF, not EOFException; and scanner further hides it inside RuntimeException.
476    IOException ioEx = null;
477    if (ex instanceof EOFException) {
478      return (EOFException) ex;
479    } else if (ex instanceof IOException) {
480      ioEx = (IOException) ex;
481    } else if (
482      ex instanceof RuntimeException && ex.getCause() != null
483        && ex.getCause() instanceof IOException
484    ) {
485      ioEx = (IOException) ex.getCause();
486    }
487    if ((ioEx != null) && (ioEx.getMessage() != null)) {
488      if (ioEx.getMessage().contains("EOF")) return ioEx;
489      return null;
490    }
491    return null;
492  }
