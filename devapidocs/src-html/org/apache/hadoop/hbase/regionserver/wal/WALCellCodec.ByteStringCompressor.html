047/**
048 * Compression in this class is lifted off Compressor/KeyValueCompression.
049 * This is a pure coincidence... they are independent and don't have to be compatible.
050 *
051 * This codec is used at server side for writing cells to WAL as well as for sending edits
052 * as part of the distributed splitting process.
053 */
054@InterfaceAudience.LimitedPrivate({HBaseInterfaceAudience.COPROC,
055  HBaseInterfaceAudience.PHOENIX, HBaseInterfaceAudience.CONFIG})
056public class WALCellCodec implements Codec {
057  /** Configuration key for the class to use when encoding cells in the WAL */
058  public static final String WAL_CELL_CODEC_CLASS_KEY = "hbase.regionserver.wal.codec";

064      return WALCellCodec.uncompressByteString(data, dict);

138  // TODO: it sucks that compression context is in WAL.Entry. It'd be nice if it was here.
139  //       Dictionary could be gotten by enum; initially, based on enum, context would create
140  //       an array of dictionaries.
141  static class BaosAndCompressor extends ByteArrayOutputStream implements ByteStringCompressor {
142    public ByteString toByteString() {
143      return ByteString.copyFrom(this.buf, 0, this.count);
144    }
145
146    @Override
147    public ByteString compress(byte[] data, Dictionary dict) throws IOException {
148      writeCompressed(data, dict);
149      ByteString result = ByteString.copyFrom(this.buf, 0, this.count);
150      reset(); // Only resets the count - we reuse the byte array.
151      return result;
152    }
153
154    private void writeCompressed(byte[] data, Dictionary dict) throws IOException {
155      assert dict != null;
156      short dictIdx = dict.findEntry(data, 0, data.length);
157      if (dictIdx == Dictionary.NOT_IN_DICTIONARY) {
158        write(Dictionary.NOT_IN_DICTIONARY);
159        StreamUtils.writeRawVInt32(this, data.length);
160        write(data, 0, data.length);
161      } else {
162        StreamUtils.writeShort(this, dictIdx);
163      }
164    }
165  }
166
167  private static byte[] uncompressByteString(ByteString bs, Dictionary dict) throws IOException {
168    InputStream in = bs.newInput();
169    byte status = (byte)in.read();
170    if (status == Dictionary.NOT_IN_DICTIONARY) {
171      byte[] arr = new byte[StreamUtils.readRawVarint32(in)];
172      int bytesRead = in.read(arr);
173      if (bytesRead != arr.length) {
174        throw new IOException("Cannot read; wanted " + arr.length + ", but got " + bytesRead);
175      }
176      if (dict != null) dict.addEntry(arr, 0, arr.length);
177      return arr;
178    } else {
179      // Status here is the higher-order byte of index of the dictionary entry.
180      short dictIdx = StreamUtils.toShort(status, (byte)in.read());
181      byte[] entry = dict.getEntry(dictIdx);
182      if (entry == null) {
183        throw new IOException("Missing dictionary entry for index " + dictIdx);
184      }
185      return entry;
186    }
187  }

189  static class CompressedKvEncoder extends BaseEncoder {
190    private final CompressionContext compression;
191    public CompressedKvEncoder(OutputStream out, CompressionContext compression) {
192      super(out);
193      this.compression = compression;
194    }
195
196    @Override
197    public void write(Cell cell) throws IOException {
198      // We first write the KeyValue infrastructure as VInts.
199      StreamUtils.writeRawVInt32(out, KeyValueUtil.keyLength(cell));
200      StreamUtils.writeRawVInt32(out, cell.getValueLength());
201      // To support tags
202      int tagsLength = cell.getTagsLength();
203      StreamUtils.writeRawVInt32(out, tagsLength);
204
205      // Write row, qualifier, and family; use dictionary
206      // compression as they're likely to have duplicates.
207      write(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(), compression.rowDict);
208      write(cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
209          compression.familyDict);
210      write(cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength(),
211          compression.qualifierDict);
212
213      // Write timestamp, type and value as uncompressed.
214      StreamUtils.writeLong(out, cell.getTimestamp());
215      out.write(cell.getTypeByte());
216      out.write(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength());
217      if (tagsLength > 0) {
218        if (compression.tagCompressionContext != null) {
219          // Write tags using Dictionary compression
220          compression.tagCompressionContext.compressTags(out, cell.getTagsArray(),
221              cell.getTagsOffset(), tagsLength);
222        } else {
223          // Tag compression is disabled within the WAL compression. Just write the tags bytes as
224          // it is.
225          out.write(cell.getTagsArray(), cell.getTagsOffset(), tagsLength);
226        }
227      }
228    }
229
230    private void write(byte[] data, int offset, int length, Dictionary dict) throws IOException {
231      short dictIdx = Dictionary.NOT_IN_DICTIONARY;
232      if (dict != null) {
233        dictIdx = dict.findEntry(data, offset, length);
234      }
235      if (dictIdx == Dictionary.NOT_IN_DICTIONARY) {
236        out.write(Dictionary.NOT_IN_DICTIONARY);
237        StreamUtils.writeRawVInt32(out, length);
238        out.write(data, offset, length);
239      } else {
240        StreamUtils.writeShort(out, dictIdx);
241      }
242    }
243  }

245  static class CompressedKvDecoder extends BaseDecoder {
246    private final CompressionContext compression;
247    public CompressedKvDecoder(InputStream in, CompressionContext compression) {
248      super(in);
249      this.compression = compression;
250    }
251
252    @Override
253    protected Cell parseCell() throws IOException {
254      int keylength = StreamUtils.readRawVarint32(in);
255      int vlength = StreamUtils.readRawVarint32(in);
256
257      int tagsLength = StreamUtils.readRawVarint32(in);
258      int length = 0;
259      if(tagsLength == 0) {
260        length = KeyValue.KEYVALUE_INFRASTRUCTURE_SIZE + keylength + vlength;
261      } else {
262        length = KeyValue.KEYVALUE_WITH_TAGS_INFRASTRUCTURE_SIZE + keylength + vlength + tagsLength;
263      }
264
265      byte[] backingArray = new byte[length];
266      int pos = 0;
267      pos = Bytes.putInt(backingArray, pos, keylength);
268      pos = Bytes.putInt(backingArray, pos, vlength);
269
270      // the row
271      int elemLen = readIntoArray(backingArray, pos + Bytes.SIZEOF_SHORT, compression.rowDict);
272      checkLength(elemLen, Short.MAX_VALUE);
273      pos = Bytes.putShort(backingArray, pos, (short)elemLen);
274      pos += elemLen;
275
276      // family
277      elemLen = readIntoArray(backingArray, pos + Bytes.SIZEOF_BYTE, compression.familyDict);
278      checkLength(elemLen, Byte.MAX_VALUE);
279      pos = Bytes.putByte(backingArray, pos, (byte)elemLen);
280      pos += elemLen;
281
282      // qualifier
283      elemLen = readIntoArray(backingArray, pos, compression.qualifierDict);
284      pos += elemLen;
285
286      // timestamp, type and value
287      int tsTypeValLen = length - pos;
288      if (tagsLength > 0) {
289        tsTypeValLen = tsTypeValLen - tagsLength - KeyValue.TAGS_LENGTH_SIZE;
290      }
291      IOUtils.readFully(in, backingArray, pos, tsTypeValLen);
292      pos += tsTypeValLen;
293
294      // tags
295      if (tagsLength > 0) {
296        pos = Bytes.putAsShort(backingArray, pos, tagsLength);
297        if (compression.tagCompressionContext != null) {
298          compression.tagCompressionContext.uncompressTags(in, backingArray, pos, tagsLength);
299        } else {
300          IOUtils.readFully(in, backingArray, pos, tagsLength);
301        }
302      }
303      return new KeyValue(backingArray, 0, length);
304    }
305
306    private int readIntoArray(byte[] to, int offset, Dictionary dict) throws IOException {
307      byte status = (byte)in.read();
308      if (status == Dictionary.NOT_IN_DICTIONARY) {
309        // status byte indicating that data to be read is not in dictionary.
310        // if this isn't in the dictionary, we need to add to the dictionary.
311        int length = StreamUtils.readRawVarint32(in);
312        IOUtils.readFully(in, to, offset, length);
313        dict.addEntry(to, offset, length);
314        return length;
315      } else {
316        // the status byte also acts as the higher order byte of the dictionary entry.
317        short dictIdx = StreamUtils.toShort(status, (byte)in.read());
318        byte[] entry = dict.getEntry(dictIdx);
319        if (entry == null) {
320          throw new IOException("Missing dictionary entry for index " + dictIdx);
321        }
322        // now we write the uncompressed value.
323        Bytes.putBytes(to, offset, entry, 0, entry.length);
324        return entry.length;
325      }
326    }
327
328    private static void checkLength(int len, int max) throws IOException {
329      if (len < 0 || len > max) {
330        throw new IOException("Invalid length for compresesed portion of keyvalue: " + len);
331      }
332    }
333  }

335  public static class EnsureKvEncoder extends BaseEncoder {
336    public EnsureKvEncoder(OutputStream out) {
337      super(out);
338    }
339    @Override
340    public void write(Cell cell) throws IOException {
341      checkFlushed();
342      // Make sure to write tags into WAL
343      ByteBufferUtils.putInt(this.out, KeyValueUtil.getSerializedSize(cell, true));
344      KeyValueUtil.oswrite(cell, this.out, true);
345    }
346  }

348  @Override
349  public Decoder getDecoder(InputStream is) {
350    return (compression == null)
351        ? new KeyValueCodecWithTags.KeyValueDecoder(is) : new CompressedKvDecoder(is, compression);
352  }

359  @Override

361    return (compression == null)
362        ? new EnsureKvEncoder(os) : new CompressedKvEncoder(os, compression);
363  }
