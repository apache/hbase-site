051/**
052 * WALEdit: Used in HBase's transaction log (WAL) to represent
053 * the collection of edits (KeyValue objects) corresponding to a
054 * single transaction. The class implements "Writable" interface
055 * for serializing/deserializing a set of KeyValue items.
056 *
057 * Previously, if a transaction contains 3 edits to c1, c2, c3 for a row R,
058 * the WAL would have three log entries as follows:
059 *
060 *    &lt;logseq1-for-edit1&gt;:&lt;eyValue-for-edit-c1&gt;
061 *    &lt;logseq2-for-edit2&gt;:&lt;KeyValue-for-edit-c2&gt;
062 *    &lt;logseq3-for-edit3&gt;:&lt;KeyValue-for-edit-c3&gt;
063 *
064 * This presents problems because row level atomicity of transactions
065 * was not guaranteed. If we crash after few of the above appends make
066 * it, then recovery will restore a partial transaction.
067 *
068 * In the new world, all the edits for a given transaction are written
069 * out as a single record, for example:
070 *
071 *   &lt;logseq#-for-entire-txn&gt;:&lt;WALEdit-for-entire-txn&gt;
072 *
073 * where, the WALEdit is serialized as:
074 *   &lt;-1, # of edits, &lt;KeyValue&gt;, &lt;KeyValue&gt;, ... &gt;
075 * For example:
076 *   &lt;-1, 3, &lt;KV-for-edit-c1&gt;, &lt;KV-for-edit-c2&gt;, &lt;KV-for-edit-c3&gt;&gt;
077 *
078 * The -1 marker is just a special way of being backward compatible with
079 * an old WAL which would have contained a single &lt;KeyValue&gt;.
080 *
081 * The deserializer for WALEdit backward compatibly detects if the record
082 * is an old style KeyValue or the new style WALEdit.
083 *
084 */
085@InterfaceAudience.LimitedPrivate({ HBaseInterfaceAudience.REPLICATION,
086    HBaseInterfaceAudience.COPROC })
087public class WALEdit implements Writable, HeapSize {
088  private static final Log LOG = LogFactory.getLog(WALEdit.class);

204  @Override
205  public void readFields(DataInput in) throws IOException {
206    cells.clear();
207    if (scopes != null) {
208      scopes.clear();
209    }
210    int versionOrLength = in.readInt();
211    // TODO: Change version when we protobuf.  Also, change way we serialize KV!  Pb it too.
212    if (versionOrLength == VERSION_2) {
213      // this is new style WAL entry containing multiple KeyValues.
214      int numEdits = in.readInt();
215      for (int idx = 0; idx < numEdits; idx++) {
216        if (compressionContext != null) {
217          this.add(KeyValueCompression.readKV(in, compressionContext));
218        } else {
219          this.add(KeyValueUtil.create(in));
220        }
221      }
222      int numFamilies = in.readInt();
223      if (numFamilies > 0) {
224        if (scopes == null) {
225          scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
226        }
227        for (int i = 0; i < numFamilies; i++) {
228          byte[] fam = Bytes.readByteArray(in);
229          int scope = in.readInt();
230          scopes.put(fam, scope);
231        }
232      }
233    } else {
234      // this is an old style WAL entry. The int that we just
235      // read is actually the length of a single KeyValue
236      this.add(KeyValueUtil.create(versionOrLength, in));
237    }
238  }
239
240  @Override
241  public void write(DataOutput out) throws IOException {
242    LOG.warn("WALEdit is being serialized to writable - only expected in test code");
243    out.writeInt(VERSION_2);
244    out.writeInt(cells.size());
245    // We interleave the two lists for code simplicity
246    for (Cell cell : cells) {
247      // This is not used in any of the core code flows so it is just fine to convert to KV
248      KeyValue kv = KeyValueUtil.ensureKeyValue(cell);
249      if (compressionContext != null) {
250        KeyValueCompression.writeKV(out, kv, compressionContext);
251      } else{
252        KeyValueUtil.write(kv, out);
253      }
254    }
255    if (scopes == null) {
256      out.writeInt(0);
257    } else {
258      out.writeInt(scopes.size());
259      for (byte[] key : scopes.keySet()) {
260        Bytes.writeByteArray(out, key);
261        out.writeInt(scopes.get(key));
262      }
263    }
264  }

266  /**
267   * Reads WALEdit from cells.
268   * @param cellDecoder Cell decoder.
269   * @param expectedCount Expected cell count.
270   * @return Number of KVs read.
271   */
272  public int readFromCells(Codec.Decoder cellDecoder, int expectedCount) throws IOException {
273    cells.clear();
274    cells.ensureCapacity(expectedCount);
275    while (cells.size() < expectedCount && cellDecoder.advance()) {
276      cells.add(cellDecoder.current());
277    }
278    return cells.size();
279  }

281  @Override
282  public long heapSize() {
283    long ret = ClassSize.ARRAYLIST;
284    for (Cell cell : cells) {
285      ret += CellUtil.estimatedHeapSizeOf(cell);
286    }
287    if (scopes != null) {
288      ret += ClassSize.TREEMAP;
289      ret += ClassSize.align(scopes.size() * ClassSize.MAP_ENTRY);
290      // TODO this isn't quite right, need help here
291    }
292    return ret;
293  }

295  @Override
296  public String toString() {
297    StringBuilder sb = new StringBuilder();
298
299    sb.append("[#edits: " + cells.size() + " = <");
300    for (Cell cell : cells) {
301      sb.append(cell);
302      sb.append("; ");
303    }
304    if (scopes != null) {
305      sb.append(" scopes: " + scopes.toString());
306    }
307    sb.append(">]");
308    return sb.toString();
309  }

311  public static WALEdit createFlushWALEdit(HRegionInfo hri, FlushDescriptor f) {
312    KeyValue kv = new KeyValue(getRowForRegion(hri), METAFAMILY, FLUSH,
313      EnvironmentEdgeManager.currentTime(), f.toByteArray());
314    return new WALEdit().add(kv);
315  }

338  /**
339   * Create a compaction WALEdit
340   * @param c
341   * @return A WALEdit that has <code>c</code> serialized as its value
342   */
343  public static WALEdit createCompaction(final HRegionInfo hri, final CompactionDescriptor c) {
344    byte [] pbbytes = c.toByteArray();
345    KeyValue kv = new KeyValue(getRowForRegion(hri), METAFAMILY, COMPACTION,
346      EnvironmentEdgeManager.currentTime(), pbbytes);
347    return new WALEdit().add(kv); //replication scope null so that this won't be replicated
348  }
349
350  public static byte[] getRowForRegion(HRegionInfo hri) {
351    byte[] startKey = hri.getStartKey();
352    if (startKey.length == 0) {
353      // empty row key is not allowed in mutations because it is both the start key and the end key
354      // we return the smallest byte[] that is bigger (in lex comparison) than byte[0].
355      return new byte[] {0};
356    }
357    return startKey;
358  }
