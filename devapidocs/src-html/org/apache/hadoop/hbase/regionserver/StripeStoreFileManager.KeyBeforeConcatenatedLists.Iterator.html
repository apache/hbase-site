051/**
052 * Stripe implementation of StoreFileManager.
053 * Not thread safe - relies on external locking (in HStore). Collections that this class
054 * returns are immutable or unique to the call, so they should be safe.
055 * Stripe store splits the key space of the region into non-overlapping stripes, as well as
056 * some recent files that have all the keys (level 0). Each stripe contains a set of files.
057 * When L0 is compacted, it's split into the files corresponding to existing stripe boundaries,
058 * that can thus be added to stripes.
059 * When scan or get happens, it only has to read the files from the corresponding stripes.
060 * See StripeCompationPolicy on how the stripes are determined; this class doesn't care.
061 *
062 * This class should work together with StripeCompactionPolicy and StripeCompactor.
063 * With regard to how they work, we make at least the following (reasonable) assumptions:
064 *  - Compaction produces one file per new stripe (if any); that is easy to change.
065 *  - Compaction has one contiguous set of stripes both in and out, except if L0 is involved.
066 */
067@InterfaceAudience.Private
068public class StripeStoreFileManager
069  implements StoreFileManager, StripeCompactionPolicy.StripeInformationProvider {
070  private static final Log LOG = LogFactory.getLog(StripeStoreFileManager.class);

092     * The end rows of each stripe. The last stripe end is always open-ended, so it's not stored

102     */

104      = new ArrayList<ImmutableList<StoreFile>>();
105    /** Level 0. The files are in reverse order by seqNum. */
106    public ImmutableList<StoreFile> level0Files = ImmutableList.<StoreFile>of();

109    public ImmutableList<StoreFile> allFilesCached = ImmutableList.<StoreFile>of();
110    private ImmutableList<StoreFile> allCompactedFilesCached = ImmutableList.<StoreFile>of();

179  /** See {@link StoreFileManager#getCandidateFilesForRowKeyBefore(KeyValue)}
180   * for details on this methods. */
181  @Override
182  public Iterator<StoreFile> getCandidateFilesForRowKeyBefore(final KeyValue targetKey) {
183    KeyBeforeConcatenatedLists result = new KeyBeforeConcatenatedLists();
184    // Order matters for this call.
185    result.addSublist(state.level0Files);
186    if (!state.stripeFiles.isEmpty()) {
187      int lastStripeIndex = findStripeForRow(CellUtil.cloneRow(targetKey), false);
188      for (int stripeIndex = lastStripeIndex; stripeIndex >= 0; --stripeIndex) {
189        result.addSublist(state.stripeFiles.get(stripeIndex));
190      }
191    }
192    return result.iterator();
193  }

195  /** See {@link StoreFileManager#getCandidateFilesForRowKeyBefore(KeyValue)} and
196   * {@link StoreFileManager#updateCandidateFilesForRowKeyBefore(Iterator, KeyValue, Cell)}
197   * for details on this methods. */
198  @Override
199  public Iterator<StoreFile> updateCandidateFilesForRowKeyBefore(
200      Iterator<StoreFile> candidateFiles, final KeyValue targetKey, final Cell candidate) {
201    KeyBeforeConcatenatedLists.Iterator original =
202        (KeyBeforeConcatenatedLists.Iterator)candidateFiles;
203    assert original != null;
204    ArrayList<List<StoreFile>> components = original.getComponents();
205    for (int firstIrrelevant = 0; firstIrrelevant < components.size(); ++firstIrrelevant) {
206      StoreFile sf = components.get(firstIrrelevant).get(0);
207      byte[] endKey = endOf(sf);
208      // Entries are ordered as such: L0, then stripes in reverse order. We never remove
209      // level 0; we remove the stripe, and all subsequent ones, as soon as we find the
210      // first one that cannot possibly have better candidates.
211      if (!isInvalid(endKey) && !isOpen(endKey)
212          && (nonOpenRowCompare(targetKey, endKey) >= 0)) {
213        original.removeComponents(firstIrrelevant);
214        break;
215      }
216    }
217    return original;
218  }

220  @Override
221  /**
222   * Override of getSplitPoint that determines the split point as the boundary between two
223   * stripes, unless it causes significant imbalance between split sides' sizes. In that
224   * case, the split boundary will be chosen from the middle of one of the stripes to
225   * minimize imbalance.
226   * @return The split point, or null if no split is possible.
227   */
228  public byte[] getSplitPoint() throws IOException {
229    if (this.getStorefileCount() == 0) return null;
230    if (state.stripeFiles.size() <= 1) {
231      return getSplitPointFromAllFiles();
232    }
233    int leftIndex = -1, rightIndex = state.stripeFiles.size();
234    long leftSize = 0, rightSize = 0;
235    long lastLeftSize = 0, lastRightSize = 0;
236    while (rightIndex - 1 != leftIndex) {
237      if (leftSize >= rightSize) {
238        --rightIndex;
239        lastRightSize = getStripeFilesSize(rightIndex);
240        rightSize += lastRightSize;
241      } else {
242        ++leftIndex;
243        lastLeftSize = getStripeFilesSize(leftIndex);
244        leftSize += lastLeftSize;
245      }
246    }
247    if (leftSize == 0 || rightSize == 0) {
248      String errMsg = String.format("Cannot split on a boundary - left index %d size %d, "
249          + "right index %d size %d", leftIndex, leftSize, rightIndex, rightSize);
250      debugDumpState(errMsg);
251      LOG.warn(errMsg);
252      return getSplitPointFromAllFiles();
253    }
254    double ratio = (double)rightSize / leftSize;
255    if (ratio < 1) {
256      ratio = 1 / ratio;
257    }
258    if (config.getMaxSplitImbalance() > ratio) return state.stripeEndRows[leftIndex];
259
260    // If the difference between the sides is too large, we could get the proportional key on
261    // the a stripe to equalize the difference, but there's no proportional key method at the
262    // moment, and it's not extremely important.
263    // See if we can achieve better ratio if we split the bigger side in half.
264    boolean isRightLarger = rightSize >= leftSize;
265    double newRatio = isRightLarger
266        ? getMidStripeSplitRatio(leftSize, rightSize, lastRightSize)
267        : getMidStripeSplitRatio(rightSize, leftSize, lastLeftSize);
268    if (newRatio < 1) {
269      newRatio = 1 / newRatio;
270    }
271    if (newRatio >= ratio)  return state.stripeEndRows[leftIndex];
272    LOG.debug("Splitting the stripe - ratio w/o split " + ratio + ", ratio with split "
273        + newRatio + " configured ratio " + config.getMaxSplitImbalance());
274    // Ok, we may get better ratio, get it.
275    return StoreUtils.getLargestFile(state.stripeFiles.get(
276        isRightLarger ? rightIndex : leftIndex)).getFileSplitPoint(this.cellComparator);
277  }
278
279  private byte[] getSplitPointFromAllFiles() throws IOException {
280    ConcatenatedLists<StoreFile> sfs = new ConcatenatedLists<StoreFile>();
281    sfs.addSublist(state.level0Files);
282    sfs.addAllSublists(state.stripeFiles);
283    if (sfs.isEmpty()) return null;
284    return StoreUtils.getLargestFile(sfs).getFileSplitPoint(this.cellComparator);
285  }
286
287  private double getMidStripeSplitRatio(long smallerSize, long largerSize, long lastLargerSize) {
288    return (double)(largerSize - lastLargerSize / 2f) / (smallerSize + lastLargerSize / 2f);
289  }

291  @Override
292  public Collection<StoreFile> getFilesForScanOrGet(
293      boolean isGet, byte[] startRow, byte[] stopRow) {
294    if (state.stripeFiles.isEmpty()) {
295      return state.level0Files; // There's just L0.
296    }
297
298    int firstStripe = findStripeForRow(startRow, true);
299    int lastStripe = findStripeForRow(stopRow, false);
300    assert firstStripe <= lastStripe;
301    if (firstStripe == lastStripe && state.level0Files.isEmpty()) {
302      return state.stripeFiles.get(firstStripe); // There's just one stripe we need.
303    }
304    if (firstStripe == 0 && lastStripe == (state.stripeFiles.size() - 1)) {
305      return state.allFilesCached; // We need to read all files.
306    }
307
308    ConcatenatedLists<StoreFile> result = new ConcatenatedLists<StoreFile>();
309    result.addAllSublists(state.stripeFiles.subList(firstStripe, lastStripe + 1));
310    result.addSublist(state.level0Files);
311    return result;
312  }

314  @Override
315  public void addCompactionResults(
316    Collection<StoreFile> compactedFiles, Collection<StoreFile> results) throws IOException {
317    // See class comment for the assumptions we make here.
318    LOG.debug("Attempting to merge compaction results: " + compactedFiles.size()
319        + " files replaced by " + results.size());
320    // In order to be able to fail in the middle of the operation, we'll operate on lazy
321    // copies and apply the result at the end.
322    CompactionOrFlushMergeCopy cmc = new CompactionOrFlushMergeCopy(false);
323    cmc.mergeResults(compactedFiles, results);
324    markCompactedAway(compactedFiles);
325    debugDumpState("Merged compaction results");
326  }

328  // Mark the files as compactedAway once the storefiles and compactedfiles list is finalised
329  // Let a background thread close the actual reader on these compacted files and also
330  // ensure to evict the blocks from block cache so that they are no longer in
331  // cache
332  private void markCompactedAway(Collection<StoreFile> compactedFiles) {
333    for (StoreFile file : compactedFiles) {
334      file.markCompactedAway();

349  @Override
350  public int getStoreCompactionPriority() {
351    // If there's only L0, do what the default store does.
352    // If we are in critical priority, do the same - we don't want to trump all stores all
353    // the time due to how many files we have.
354    int fc = getStorefileCount();
355    if (state.stripeFiles.isEmpty() || (this.blockingFileCount <= fc)) {
356      return this.blockingFileCount - fc;
357    }
358    // If we are in good shape, we don't want to be trumped by all other stores due to how
359    // many files we have, so do an approximate mapping to normal priority range; L0 counts
360    // for all stripes.
361    int l0 = state.level0Files.size(), sc = state.stripeFiles.size();
362    int priority = (int)Math.ceil(((double)(this.blockingFileCount - fc + l0) / sc) - l0);
363    return (priority <= HStore.PRIORITY_USER) ? (HStore.PRIORITY_USER + 1) : priority;
364  }

366  /**
367   * Gets the total size of all files in the stripe.
368   * @param stripeIndex Stripe index.
369   * @return Size.
370   */
371  private long getStripeFilesSize(int stripeIndex) {
372    long result = 0;
373    for (StoreFile sf : state.stripeFiles.get(stripeIndex)) {
374      result += sf.getReader().length();
375    }
376    return result;
377  }

379  /**
380   * Loads initial store files that were picked up from some physical location pertaining to
381   * this store (presumably). Unlike adding files after compaction, assumes empty initial
382   * sets, and is forgiving with regard to stripe constraints - at worst, many/all files will
383   * go to level 0.
384   * @param storeFiles Store files to add.
385   */
386  private void loadUnclassifiedStoreFiles(List<StoreFile> storeFiles) {
387    LOG.debug("Attempting to load " + storeFiles.size() + " store files.");
388    TreeMap<byte[], ArrayList<StoreFile>> candidateStripes =
389        new TreeMap<byte[], ArrayList<StoreFile>>(MAP_COMPARATOR);
390    ArrayList<StoreFile> level0Files = new ArrayList<StoreFile>();
391    // Separate the files into tentative stripes; then validate. Currently, we rely on metadata.
392    // If needed, we could dynamically determine the stripes in future.
393    for (StoreFile sf : storeFiles) {
394      byte[] startRow = startOf(sf), endRow = endOf(sf);
395      // Validate the range and put the files into place.
396      if (isInvalid(startRow) || isInvalid(endRow)) {
397        insertFileIntoStripe(level0Files, sf); // No metadata - goes to L0.
398        ensureLevel0Metadata(sf);
399      } else if (!isOpen(startRow) && !isOpen(endRow) &&
400          nonOpenRowCompare(startRow, endRow) >= 0) {
401        LOG.error("Unexpected metadata - start row [" + Bytes.toString(startRow) + "], end row ["
402          + Bytes.toString(endRow) + "] in file [" + sf.getPath() + "], pushing to L0");
403        insertFileIntoStripe(level0Files, sf); // Bad metadata - goes to L0 also.
404        ensureLevel0Metadata(sf);
405      } else {
406        ArrayList<StoreFile> stripe = candidateStripes.get(endRow);
407        if (stripe == null) {
408          stripe = new ArrayList<StoreFile>();
409          candidateStripes.put(endRow, stripe);
410        }
411        insertFileIntoStripe(stripe, sf);
412      }
413    }
414    // Possible improvement - for variable-count stripes, if all the files are in L0, we can
415    // instead create single, open-ended stripe with all files.
416
417    boolean hasOverlaps = false;
418    byte[] expectedStartRow = null; // first stripe can start wherever
419    Iterator<Map.Entry<byte[], ArrayList<StoreFile>>> entryIter =
420        candidateStripes.entrySet().iterator();
421    while (entryIter.hasNext()) {
422      Map.Entry<byte[], ArrayList<StoreFile>> entry = entryIter.next();
423      ArrayList<StoreFile> files = entry.getValue();
424      // Validate the file start rows, and remove the bad ones to level 0.
425      for (int i = 0; i < files.size(); ++i) {
426        StoreFile sf = files.get(i);
427        byte[] startRow = startOf(sf);
428        if (expectedStartRow == null) {
429          expectedStartRow = startRow; // ensure that first stripe is still consistent
430        } else if (!rowEquals(expectedStartRow, startRow)) {
431          hasOverlaps = true;
432          LOG.warn("Store file doesn't fit into the tentative stripes - expected to start at ["
433              + Bytes.toString(expectedStartRow) + "], but starts at [" + Bytes.toString(startRow)
434              + "], to L0 it goes");
435          StoreFile badSf = files.remove(i);
436          insertFileIntoStripe(level0Files, badSf);
437          ensureLevel0Metadata(badSf);
438          --i;
439        }
440      }
441      // Check if any files from the candidate stripe are valid. If so, add a stripe.
442      byte[] endRow = entry.getKey();
443      if (!files.isEmpty()) {
444        expectedStartRow = endRow; // Next stripe must start exactly at that key.
445      } else {
446        entryIter.remove();
447      }
448    }
449
450    // In the end, there must be open ends on two sides. If not, and there were no errors i.e.
451    // files are consistent, they might be coming from a split. We will treat the boundaries
452    // as open keys anyway, and log the message.
453    // If there were errors, we'll play it safe and dump everything into L0.
454    if (!candidateStripes.isEmpty()) {
455      StoreFile firstFile = candidateStripes.firstEntry().getValue().get(0);
456      boolean isOpen = isOpen(startOf(firstFile)) && isOpen(candidateStripes.lastKey());
457      if (!isOpen) {
458        LOG.warn("The range of the loaded files does not cover full key space: from ["
459            + Bytes.toString(startOf(firstFile)) + "], to ["
460            + Bytes.toString(candidateStripes.lastKey()) + "]");
461        if (!hasOverlaps) {
462          ensureEdgeStripeMetadata(candidateStripes.firstEntry().getValue(), true);
463          ensureEdgeStripeMetadata(candidateStripes.lastEntry().getValue(), false);
464        } else {
465          LOG.warn("Inconsistent files, everything goes to L0.");
466          for (ArrayList<StoreFile> files : candidateStripes.values()) {
467            for (StoreFile sf : files) {
468              insertFileIntoStripe(level0Files, sf);
469              ensureLevel0Metadata(sf);
470            }
471          }
472          candidateStripes.clear();
473        }
474      }
475    }
476
477    // Copy the results into the fields.
478    State state = new State();
479    state.level0Files = ImmutableList.copyOf(level0Files);
480    state.stripeFiles = new ArrayList<ImmutableList<StoreFile>>(candidateStripes.size());
481    state.stripeEndRows = new byte[Math.max(0, candidateStripes.size() - 1)][];
482    ArrayList<StoreFile> newAllFiles = new ArrayList<StoreFile>(level0Files);
483    int i = candidateStripes.size() - 1;
484    for (Map.Entry<byte[], ArrayList<StoreFile>> entry : candidateStripes.entrySet()) {
485      state.stripeFiles.add(ImmutableList.copyOf(entry.getValue()));
486      newAllFiles.addAll(entry.getValue());
487      if (i > 0) {
488        state.stripeEndRows[state.stripeFiles.size() - 1] = entry.getKey();
489      }
490      --i;
491    }
492    state.allFilesCached = ImmutableList.copyOf(newAllFiles);
493    this.state = state;
494    debugDumpState("Files loaded");
495  }

497  private void ensureEdgeStripeMetadata(ArrayList<StoreFile> stripe, boolean isFirst) {
498    HashMap<StoreFile, byte[]> targetMap = isFirst ? fileStarts : fileEnds;
499    for (StoreFile sf : stripe) {
500      targetMap.put(sf, OPEN_KEY);
501    }
502  }
503
504  private void ensureLevel0Metadata(StoreFile sf) {
505    if (!isInvalid(startOf(sf))) this.fileStarts.put(sf, INVALID_KEY_IN_MAP);
506    if (!isInvalid(endOf(sf))) this.fileEnds.put(sf, INVALID_KEY_IN_MAP);
507  }

509  private void debugDumpState(String string) {
510    if (!LOG.isDebugEnabled()) return;
511    StringBuilder sb = new StringBuilder();
512    sb.append("\n" + string + "; current stripe state is as such:");
513    sb.append("\n level 0 with ")
514        .append(state.level0Files.size())
515        .append(
516          " files: "
517              + TraditionalBinaryPrefix.long2String(
518                StripeCompactionPolicy.getTotalFileSize(state.level0Files), "", 1) + ";");
519    for (int i = 0; i < state.stripeFiles.size(); ++i) {
520      String endRow = (i == state.stripeEndRows.length)
521          ? "(end)" : "[" + Bytes.toString(state.stripeEndRows[i]) + "]";
522      sb.append("\n stripe ending in ")
523          .append(endRow)
524          .append(" with ")
525          .append(state.stripeFiles.get(i).size())
526          .append(
527            " files: "
528                + TraditionalBinaryPrefix.long2String(
529                  StripeCompactionPolicy.getTotalFileSize(state.stripeFiles.get(i)), "", 1) + ";");
530    }
531    sb.append("\n").append(state.stripeFiles.size()).append(" stripes total.");
532    sb.append("\n").append(getStorefileCount()).append(" files total.");
533    LOG.debug(sb.toString());
534  }

536  /**
537   * Checks whether the key indicates an open interval boundary (i.e. infinity).
538   */
539  private static final boolean isOpen(byte[] key) {
540    return key != null && key.length == 0;
541  }
542
543  private static final boolean isOpen(Cell key) {
544    return key != null && key.getRowLength() == 0;
545  }
546
547  /**
548   * Checks whether the key is invalid (e.g. from an L0 file, or non-stripe-compacted files).
549   */
550  private static final boolean isInvalid(byte[] key) {
551    // No need to use Arrays.equals because INVALID_KEY is null
552    return key == INVALID_KEY;
553  }
554
555  /**
556   * Compare two keys for equality.
557   */
558  private final boolean rowEquals(byte[] k1, byte[] k2) {
559    return Bytes.equals(k1, 0, k1.length, k2, 0, k2.length);
560  }
561
562  /**
563   * Compare two keys. Keys must not be open (isOpen(row) == false).
564   */
565  private final int nonOpenRowCompare(byte[] k1, byte[] k2) {
566    assert !isOpen(k1) && !isOpen(k2);
567    return cellComparator.compareRows(new KeyOnlyKeyValue(k1), k2, 0, k2.length);
568  }

570  private final int nonOpenRowCompare(Cell k1, byte[] k2) {
571    assert !isOpen(k1) && !isOpen(k2);
572    return cellComparator.compareRows(k1, k2, 0, k2.length);
573  }

575  /**
576   * Finds the stripe index by end row.
577   */
578  private final int findStripeIndexByEndRow(byte[] endRow) {
579    assert !isInvalid(endRow);
580    if (isOpen(endRow)) return state.stripeEndRows.length;
581    return Arrays.binarySearch(state.stripeEndRows, endRow, Bytes.BYTES_COMPARATOR);
582  }

584  /**
585   * Finds the stripe index for the stripe containing a row provided externally for get/scan.
586   */
587  private final int findStripeForRow(byte[] row, boolean isStart) {
588    if (isStart && Arrays.equals(row, HConstants.EMPTY_START_ROW)) return 0;
589    if (!isStart && Arrays.equals(row, HConstants.EMPTY_END_ROW)) {
590      return state.stripeFiles.size() - 1;
591    }
592    // If there's an exact match below, a stripe ends at "row". Stripe right boundary is
593    // exclusive, so that means the row is in the next stripe; thus, we need to add one to index.
594    // If there's no match, the return value of binarySearch is (-(insertion point) - 1), where
595    // insertion point is the index of the next greater element, or list size if none. The
596    // insertion point happens to be exactly what we need, so we need to add one to the result.
597    return Math.abs(Arrays.binarySearch(state.stripeEndRows, row, Bytes.BYTES_COMPARATOR) + 1);
598  }

612  private byte[] startOf(StoreFile sf) {
613    byte[] result = fileStarts.get(sf);
614
615    // result and INVALID_KEY_IN_MAP are compared _only_ by reference on purpose here as the latter
616    // serves only as a marker and is not to be confused with other empty byte arrays.
617    // See Javadoc of INVALID_KEY_IN_MAP for more information
618    return (result == null)
619             ? sf.getMetadataValue(STRIPE_START_KEY)
620             : result == INVALID_KEY_IN_MAP ? INVALID_KEY : result;
621  }

624    byte[] result = fileEnds.get(sf);
625
626    // result and INVALID_KEY_IN_MAP are compared _only_ by reference on purpose here as the latter
627    // serves only as a marker and is not to be confused with other empty byte arrays.
628    // See Javadoc of INVALID_KEY_IN_MAP for more information
629    return (result == null)
630             ? sf.getMetadataValue(STRIPE_END_KEY)
631             : result == INVALID_KEY_IN_MAP ? INVALID_KEY : result;
632  }

634  /**
635   * Inserts a file in the correct place (by seqnum) in a stripe copy.
636   * @param stripe Stripe copy to insert into.
637   * @param sf File to insert.
638   */
639  private static void insertFileIntoStripe(ArrayList<StoreFile> stripe, StoreFile sf) {
640    // The only operation for which sorting of the files matters is KeyBefore. Therefore,
641    // we will store the file in reverse order by seqNum from the outset.
642    for (int insertBefore = 0; ; ++insertBefore) {
643      if (insertBefore == stripe.size()
644          || (StoreFile.Comparators.SEQ_ID.compare(sf, stripe.get(insertBefore)) >= 0)) {
645        stripe.add(insertBefore, sf);
646        break;
647      }
648    }
649  }

651  /**
652   * An extension of ConcatenatedLists that has several peculiar properties.
653   * First, one can cut the tail of the logical list by removing last several sub-lists.
654   * Second, items can be removed thru iterator.
655   * Third, if the sub-lists are immutable, they are replaced with mutable copies when needed.
656   * On average KeyBefore operation will contain half the stripes as potential candidates,
657   * but will quickly cut down on them as it finds something in the more likely ones; thus,
658   * the above allow us to avoid unnecessary copying of a bunch of lists.
659   */
660  private static class KeyBeforeConcatenatedLists extends ConcatenatedLists<StoreFile> {
661    @Override
662    public java.util.Iterator<StoreFile> iterator() {
663      return new Iterator();
664    }
665
666    public class Iterator extends ConcatenatedLists<StoreFile>.Iterator {
667      public ArrayList<List<StoreFile>> getComponents() {
668        return components;
669      }
670
671      public void removeComponents(int startIndex) {
672        List<List<StoreFile>> subList = components.subList(startIndex, components.size());
673        for (List<StoreFile> entry : subList) {
674          size -= entry.size();
675        }
676        assert size >= 0;
677        subList.clear();
678      }
679
680      @Override
681      public void remove() {
682        if (!this.nextWasCalled) {
683          throw new IllegalStateException("No element to remove");
684        }
685        this.nextWasCalled = false;
686        List<StoreFile> src = components.get(currentComponent);
687        if (src instanceof ImmutableList<?>) {
688          src = new ArrayList<StoreFile>(src);
689          components.set(currentComponent, src);
690        }
691        src.remove(indexWithinComponent);
692        --size;
693        --indexWithinComponent;
694        if (src.isEmpty()) {
695          components.remove(currentComponent); // indexWithinComponent is already -1 here.
696        }
697      }
698    }
699  }

701  /**
702   * Non-static helper class for merging compaction or flush results.
703   * Since we want to merge them atomically (more or less), it operates on lazy copies,
704   * then creates a new state object and puts it in place.
705   */
706  private class CompactionOrFlushMergeCopy {
707    private ArrayList<List<StoreFile>> stripeFiles = null;
708    private ArrayList<StoreFile> level0Files = null;
709    private ArrayList<byte[]> stripeEndRows = null;
710
711    private Collection<StoreFile> compactedFiles = null;
712    private Collection<StoreFile> results = null;
713
714    private List<StoreFile> l0Results = new ArrayList<StoreFile>();
715    private final boolean isFlush;
716
717    public CompactionOrFlushMergeCopy(boolean isFlush) {
718      // Create a lazy mutable copy (other fields are so lazy they start out as nulls).
719      this.stripeFiles = new ArrayList<List<StoreFile>>(
720          StripeStoreFileManager.this.state.stripeFiles);
721      this.isFlush = isFlush;
722    }
723
724    private void mergeResults(Collection<StoreFile> compactedFiles, Collection<StoreFile> results)
725        throws IOException {
726      assert this.compactedFiles == null && this.results == null;
727      this.compactedFiles = compactedFiles;
728      this.results = results;
729      // Do logical processing.
730      if (!isFlush) removeCompactedFiles();
731      TreeMap<byte[], StoreFile> newStripes = processResults();
732      if (newStripes != null) {
733        processNewCandidateStripes(newStripes);
734      }
735      // Create new state and update parent.
736      State state = createNewState(false);
737      StripeStoreFileManager.this.state = state;
738      updateMetadataMaps();
739    }
740
741    private void deleteResults(Collection<StoreFile> compactedFiles) throws IOException {
742      this.compactedFiles = compactedFiles;
743      // Create new state and update parent.
744      State state = createNewState(true);
745      StripeStoreFileManager.this.state = state;
746      updateMetadataMaps();
747    }
748
749    private State createNewState(boolean delCompactedFiles) {
750      State oldState = StripeStoreFileManager.this.state;
751      // Stripe count should be the same unless the end rows changed.
752      assert oldState.stripeFiles.size() == this.stripeFiles.size() || this.stripeEndRows != null;
753      State newState = new State();
754      newState.level0Files = (this.level0Files == null) ? oldState.level0Files
755          : ImmutableList.copyOf(this.level0Files);
756      newState.stripeEndRows = (this.stripeEndRows == null) ? oldState.stripeEndRows
757          : this.stripeEndRows.toArray(new byte[this.stripeEndRows.size()][]);
758      newState.stripeFiles = new ArrayList<ImmutableList<StoreFile>>(this.stripeFiles.size());
759      for (List<StoreFile> newStripe : this.stripeFiles) {
760        newState.stripeFiles.add(newStripe instanceof ImmutableList<?>
761            ? (ImmutableList<StoreFile>)newStripe : ImmutableList.copyOf(newStripe));
762      }
763
764      List<StoreFile> newAllFiles = new ArrayList<StoreFile>(oldState.allFilesCached);
765      List<StoreFile> newAllCompactedFiles =
766          new ArrayList<StoreFile>(oldState.allCompactedFilesCached);
767      if (!isFlush) {
768        newAllFiles.removeAll(compactedFiles);
769        if (delCompactedFiles) {
770          newAllCompactedFiles.removeAll(compactedFiles);
771        } else {
772          newAllCompactedFiles.addAll(compactedFiles);
773        }
774      }
775      if (results != null) {
776        newAllFiles.addAll(results);
777      }
778      newState.allFilesCached = ImmutableList.copyOf(newAllFiles);
779      newState.allCompactedFilesCached = ImmutableList.copyOf(newAllCompactedFiles);
780      return newState;
781    }
782
783    private void updateMetadataMaps() {
784      StripeStoreFileManager parent = StripeStoreFileManager.this;
785      if (!isFlush) {
786        for (StoreFile sf : this.compactedFiles) {
787          parent.fileStarts.remove(sf);
788          parent.fileEnds.remove(sf);
789        }
790      }
791      if (this.l0Results != null) {
792        for (StoreFile sf : this.l0Results) {
793          parent.ensureLevel0Metadata(sf);
794        }
795      }
796    }

798    /**
799     * @param index Index of the stripe we need.
800     * @return A lazy stripe copy from current stripes.
801     */
802    private final ArrayList<StoreFile> getStripeCopy(int index) {
803      List<StoreFile> stripeCopy = this.stripeFiles.get(index);
804      ArrayList<StoreFile> result = null;
805      if (stripeCopy instanceof ImmutableList<?>) {
806        result = new ArrayList<StoreFile>(stripeCopy);
807        this.stripeFiles.set(index, result);
808      } else {
809        result = (ArrayList<StoreFile>)stripeCopy;
810      }
811      return result;
812    }
813
814    /**
815     * @return A lazy L0 copy from current state.
816     */
817    private final ArrayList<StoreFile> getLevel0Copy() {
818      if (this.level0Files == null) {
819        this.level0Files = new ArrayList<StoreFile>(StripeStoreFileManager.this.state.level0Files);
820      }
821      return this.level0Files;
822    }
823
824    /**
825     * Process new files, and add them either to the structure of existing stripes,
826     * or to the list of new candidate stripes.
827     * @return New candidate stripes.
828     */
829    private TreeMap<byte[], StoreFile> processResults() throws IOException {
830      TreeMap<byte[], StoreFile> newStripes = null;
831      for (StoreFile sf : this.results) {
832        byte[] startRow = startOf(sf), endRow = endOf(sf);
833        if (isInvalid(endRow) || isInvalid(startRow)) {
834          if (!isFlush) {
835            LOG.warn("The newly compacted file doesn't have stripes set: " + sf.getPath());
836          }
837          insertFileIntoStripe(getLevel0Copy(), sf);
838          this.l0Results.add(sf);
839          continue;
840        }
841        if (!this.stripeFiles.isEmpty()) {
842          int stripeIndex = findStripeIndexByEndRow(endRow);
843          if ((stripeIndex >= 0) && rowEquals(getStartRow(stripeIndex), startRow)) {
844            // Simple/common case - add file to an existing stripe.
845            insertFileIntoStripe(getStripeCopy(stripeIndex), sf);
846            continue;
847          }
848        }
849
850        // Make a new candidate stripe.
851        if (newStripes == null) {
852          newStripes = new TreeMap<byte[], StoreFile>(MAP_COMPARATOR);
853        }
854        StoreFile oldSf = newStripes.put(endRow, sf);
855        if (oldSf != null) {
856          throw new IOException("Compactor has produced multiple files for the stripe ending in ["
857              + Bytes.toString(endRow) + "], found " + sf.getPath() + " and " + oldSf.getPath());
858        }
859      }
860      return newStripes;
861    }
862
863    /**
864     * Remove compacted files.
865     * @param compactedFiles Compacted files.
866     */
867    private void removeCompactedFiles() throws IOException {
868      for (StoreFile oldFile : this.compactedFiles) {
869        byte[] oldEndRow = endOf(oldFile);
870        List<StoreFile> source = null;
871        if (isInvalid(oldEndRow)) {
872          source = getLevel0Copy();
873        } else {
874          int stripeIndex = findStripeIndexByEndRow(oldEndRow);
875          if (stripeIndex < 0) {
876            throw new IOException("An allegedly compacted file [" + oldFile + "] does not belong"
877                + " to a known stripe (end row - [" + Bytes.toString(oldEndRow) + "])");
878          }
879          source = getStripeCopy(stripeIndex);
880        }
881        if (!source.remove(oldFile)) {
882          throw new IOException("An allegedly compacted file [" + oldFile + "] was not found");
883        }
884      }
885    }

887    /**
888     * See {@link #addCompactionResults(Collection, Collection)} - updates the stripe list with
889     * new candidate stripes/removes old stripes; produces new set of stripe end rows.
890     * @param newStripes  New stripes - files by end row.
891     */
892    private void processNewCandidateStripes(
893        TreeMap<byte[], StoreFile> newStripes) throws IOException {
894      // Validate that the removed and added aggregate ranges still make for a full key space.
895      boolean hasStripes = !this.stripeFiles.isEmpty();
896      this.stripeEndRows = new ArrayList<byte[]>(
897          Arrays.asList(StripeStoreFileManager.this.state.stripeEndRows));
898      int removeFrom = 0;
899      byte[] firstStartRow = startOf(newStripes.firstEntry().getValue());
900      byte[] lastEndRow = newStripes.lastKey();
901      if (!hasStripes && (!isOpen(firstStartRow) || !isOpen(lastEndRow))) {
902        throw new IOException("Newly created stripes do not cover the entire key space.");
903      }
904
905      boolean canAddNewStripes = true;
906      Collection<StoreFile> filesForL0 = null;
907      if (hasStripes) {
908        // Determine which stripes will need to be removed because they conflict with new stripes.
909        // The new boundaries should match old stripe boundaries, so we should get exact matches.
910        if (isOpen(firstStartRow)) {
911          removeFrom = 0;
912        } else {
913          removeFrom = findStripeIndexByEndRow(firstStartRow);
914          if (removeFrom < 0) throw new IOException("Compaction is trying to add a bad range.");
915          ++removeFrom;
916        }
917        int removeTo = findStripeIndexByEndRow(lastEndRow);
918        if (removeTo < 0) throw new IOException("Compaction is trying to add a bad range.");
919        // See if there are files in the stripes we are trying to replace.
920        ArrayList<StoreFile> conflictingFiles = new ArrayList<StoreFile>();
921        for (int removeIndex = removeTo; removeIndex >= removeFrom; --removeIndex) {
922          conflictingFiles.addAll(this.stripeFiles.get(removeIndex));
923        }
924        if (!conflictingFiles.isEmpty()) {
925          // This can be caused by two things - concurrent flush into stripes, or a bug.
926          // Unfortunately, we cannot tell them apart without looking at timing or something
927          // like that. We will assume we are dealing with a flush and dump it into L0.
928          if (isFlush) {
929            long newSize = StripeCompactionPolicy.getTotalFileSize(newStripes.values());
930            LOG.warn("Stripes were created by a flush, but results of size " + newSize
931                + " cannot be added because the stripes have changed");
932            canAddNewStripes = false;
933            filesForL0 = newStripes.values();
934          } else {
935            long oldSize = StripeCompactionPolicy.getTotalFileSize(conflictingFiles);
936            LOG.info(conflictingFiles.size() + " conflicting files (likely created by a flush) "
937                + " of size " + oldSize + " are moved to L0 due to concurrent stripe change");
938            filesForL0 = conflictingFiles;
939          }
940          if (filesForL0 != null) {
941            for (StoreFile sf : filesForL0) {
942              insertFileIntoStripe(getLevel0Copy(), sf);
943            }
944            l0Results.addAll(filesForL0);
945          }
946        }
947
948        if (canAddNewStripes) {
949          // Remove old empty stripes.
950          int originalCount = this.stripeFiles.size();
951          for (int removeIndex = removeTo; removeIndex >= removeFrom; --removeIndex) {
952            if (removeIndex != originalCount - 1) {
953              this.stripeEndRows.remove(removeIndex);
954            }
955            this.stripeFiles.remove(removeIndex);
956          }
957        }
958      }
959
960      if (!canAddNewStripes) return; // Files were already put into L0.
961
962      // Now, insert new stripes. The total ranges match, so we can insert where we removed.
963      byte[] previousEndRow = null;
964      int insertAt = removeFrom;
965      for (Map.Entry<byte[], StoreFile> newStripe : newStripes.entrySet()) {
966        if (previousEndRow != null) {
967          // Validate that the ranges are contiguous.
968          assert !isOpen(previousEndRow);
969          byte[] startRow = startOf(newStripe.getValue());
970          if (!rowEquals(previousEndRow, startRow)) {
971            throw new IOException("The new stripes produced by "
972                + (isFlush ? "flush" : "compaction") + " are not contiguous");
973          }
974        }
975        // Add the new stripe.
976        ArrayList<StoreFile> tmp = new ArrayList<StoreFile>();
977        tmp.add(newStripe.getValue());
978        stripeFiles.add(insertAt, tmp);
979        previousEndRow = newStripe.getKey();
980        if (!isOpen(previousEndRow)) {
981          stripeEndRows.add(insertAt, previousEndRow);
982        }
983        ++insertAt;
984      }
985    }
986  }

1013  @Override
1014  public Collection<StoreFile> getUnneededFiles(long maxTs, List<StoreFile> filesCompacting) {
1015    // 1) We can never get rid of the last file which has the maximum seqid in a stripe.
1016    // 2) Files that are not the latest can't become one due to (1), so the rest are fair game.
1017    State state = this.state;
1018    Collection<StoreFile> expiredStoreFiles = null;
1019    for (ImmutableList<StoreFile> stripe : state.stripeFiles) {
1020      expiredStoreFiles = findExpiredFiles(stripe, maxTs, filesCompacting, expiredStoreFiles);
1021    }
1022    return findExpiredFiles(state.level0Files, maxTs, filesCompacting, expiredStoreFiles);
1023  }

1025  private Collection<StoreFile> findExpiredFiles(ImmutableList<StoreFile> stripe, long maxTs,
1026      List<StoreFile> filesCompacting, Collection<StoreFile> expiredStoreFiles) {
1027    // Order by seqnum is reversed.
1028    for (int i = 1; i < stripe.size(); ++i) {
1029      StoreFile sf = stripe.get(i);
1030      synchronized (sf) {
1031        long fileTs = sf.getReader().getMaxTimestamp();
1032        if (fileTs < maxTs && !filesCompacting.contains(sf)) {
1033          LOG.info("Found an expired store file: " + sf.getPath() + " whose maxTimeStamp is "
1034              + fileTs + ", which is below " + maxTs);
1035          if (expiredStoreFiles == null) {
1036            expiredStoreFiles = new ArrayList<StoreFile>();
1037          }
1038          expiredStoreFiles.add(sf);
1039        }
1040      }
1041    }
1042    return expiredStoreFiles;
1043  }

1045  @Override
1046  public double getCompactionPressure() {
1047    State stateLocal = this.state;
1048    if (stateLocal.allFilesCached.size() > blockingFileCount) {
1049      // just a hit to tell others that we have reached the blocking file count.
1050      return 2.0;
1051    }
1052    if (stateLocal.stripeFiles.isEmpty()) {
1053      return 0.0;
1054    }
1055    int blockingFilePerStripe = blockingFileCount / stateLocal.stripeFiles.size();
1056    // do not calculate L0 separately because data will be moved to stripe quickly and in most cases
1057    // we flush data to stripe directly.
1058    int delta = stateLocal.level0Files.isEmpty() ? 0 : 1;
1059    double max = 0.0;
1060    for (ImmutableList<StoreFile> stripeFile : stateLocal.stripeFiles) {
1061      int stripeFileCount = stripeFile.size();
1062      double normCount =
1063          (double) (stripeFileCount + delta - config.getStripeCompactMinFiles())
1064              / (blockingFilePerStripe - config.getStripeCompactMinFiles());
1065      if (normCount >= 1.0) {
1066        // This could happen if stripe is not split evenly. Do not return values that larger than
1067        // 1.0 because we have not reached the blocking file count actually.
1068        return 1.0;
1069      }
1070      if (normCount > max) {
1071        max = normCount;
1072      }
1073    }
1074    return max;
1075  }
