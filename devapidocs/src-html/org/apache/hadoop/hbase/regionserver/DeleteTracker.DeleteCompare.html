019package org.apache.hadoop.hbase.regionserver;
020
021import org.apache.hadoop.hbase.classification.InterfaceAudience;
022import org.apache.hadoop.hbase.Cell;
023
024/**
025 * This interface is used for the tracking and enforcement of Deletes
026 * during the course of a Get or Scan operation.
027 * <p>
028 * This class is utilized through three methods:
029 * <ul><li>{@link #add} when encountering a Delete</li>
030 * <li>{@link #isDeleted} when checking if a Put KeyValue has been deleted</li>
031 * <li>{@link #update} when reaching the end of a StoreFile</li>
032 * </ul>
033 */
034@InterfaceAudience.Private
035public interface DeleteTracker {
036
037  /**
038   * Add the specified cell to the list of deletes to check against for
039   * this row operation.
040   * <p>
041   * This is called when a Delete is encountered in a StoreFile.
042   * @param cell - the delete cell
043   */
044  void add(Cell cell);
045
046  /**
047   * Check if the specified cell buffer has been deleted by a previously
048   * seen delete.
049   * @param cell - current cell to check if deleted by a previously seen delete
050   * @return deleteResult The result tells whether the KeyValue is deleted and why
051   */
052  DeleteResult isDeleted(Cell cell);
053
054  /**
055   * @return true if there are no current delete, false otherwise
056   */
057  boolean isEmpty();
058
059  /**
060   * Called at the end of every StoreFile.
061   * <p>
062   * Many optimized implementations of Trackers will require an update at
063   * when the end of each StoreFile is reached.
064   */
065  void update();
066
067  /**
068   * Called between rows.
069   * <p>
070   * This clears everything as if a new DeleteTracker was instantiated.
071   */
072  void reset();
073
074
075  /**
076   * Return codes for comparison of two Deletes.
077   * <p>
078   * The codes tell the merging function what to do.
079   * <p>
080   * INCLUDE means add the specified Delete to the merged list.
081   * NEXT means move to the next element in the specified list(s).
082   */
083  enum DeleteCompare {
084    INCLUDE_OLD_NEXT_OLD,
085    INCLUDE_OLD_NEXT_BOTH,
086    INCLUDE_NEW_NEXT_NEW,
087    INCLUDE_NEW_NEXT_BOTH,
088    NEXT_OLD,
089    NEXT_NEW
090  }
091
092  /**
093   * Returns codes for delete result.
094   * The codes tell the ScanQueryMatcher whether the kv is deleted and why.
095   * Based on the delete result, the ScanQueryMatcher will decide the next
096   * operation
097   */
098  enum DeleteResult {
099    FAMILY_DELETED, // The KeyValue is deleted by a delete family.
100    FAMILY_VERSION_DELETED, // The KeyValue is deleted by a delete family version.
101    COLUMN_DELETED, // The KeyValue is deleted by a delete column.
102    VERSION_DELETED, // The KeyValue is deleted by a version delete.
103    NOT_DELETED
104  }
105
106}
