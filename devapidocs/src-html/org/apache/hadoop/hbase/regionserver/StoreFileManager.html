019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022import java.util.Collection;
023import java.util.Comparator;
024import java.util.Iterator;
025import java.util.List;
026
027import com.google.common.collect.ImmutableCollection;
028
029import org.apache.hadoop.hbase.Cell;
030import org.apache.hadoop.hbase.KeyValue;
031import org.apache.hadoop.hbase.classification.InterfaceAudience;
032
033/**
034 * Manages the store files and basic metadata about that that determines the logical structure
035 * (e.g. what files to return for scan, how to determine split point, and such).
036 * Does NOT affect the physical structure of files in HDFS.
037 * Example alternative structures - the default list of files by seqNum; levelDB one sorted
038 * by level and seqNum.
039 *
040 * Implementations are assumed to be not thread safe.
041 */
042@InterfaceAudience.Private
043public interface StoreFileManager {
044  /**
045   * Loads the initial store files into empty StoreFileManager.
046   * @param storeFiles The files to load.
047   */
048  void loadFiles(List<StoreFile> storeFiles);
049
050  /**
051   * Adds new files, either for from MemStore flush or bulk insert, into the structure.
052   * @param sfs New store files.
053   */
054  void insertNewFiles(Collection<StoreFile> sfs) throws IOException;
055
056  /**
057   * Adds only the new compaction results into the structure.
058   * @param compactedFiles The input files for the compaction.
059   * @param results The resulting files for the compaction.
060   */
061  void addCompactionResults(
062      Collection<StoreFile> compactedFiles, Collection<StoreFile> results) throws IOException;
063
064  /**
065   * Remove the compacted files
066   * @param compactedFiles the list of compacted files
067   * @throws IOException
068   */
069  void removeCompactedFiles(Collection<StoreFile> compactedFiles) throws IOException;
070
071  /**
072   * Clears all the files currently in use and returns them.
073   * @return The files previously in use.
074   */
075  ImmutableCollection<StoreFile> clearFiles();
076
077  /**
078   * Clears all the compacted files and returns them. This method is expected to be
079   * accessed single threaded.
080   * @return The files compacted previously.
081   */
082  Collection<StoreFile> clearCompactedFiles();
083
084  /**
085   * Gets the snapshot of the store files currently in use. Can be used for things like metrics
086   * and checks; should not assume anything about relations between store files in the list.
087   * @return The list of StoreFiles.
088   */
089  Collection<StoreFile> getStorefiles();
090
091  /**
092   * List of compacted files inside this store that needs to be excluded in reads
093   * because further new reads will be using only the newly created files out of compaction.
094   * These compacted files will be deleted/cleared once all the existing readers on these
095   * compacted files are done.
096   * @return the list of compacted files
097   */
098  Collection<StoreFile> getCompactedfiles();
099
100  /**
101   * Returns the number of files currently in use.
102   * @return The number of files.
103   */
104  int getStorefileCount();
105
106  /**
107   * Returns the number of compacted files.
108   * @return The number of files.
109   */
110  int getCompactedFilesCount();
111
112  /**
113   * Gets the store files to scan for a Scan or Get request.
114   * @param startRow Start row of the request.
115   * @param stopRow Stop row of the request.
116   * @return The list of files that are to be read for this request.
117   */
118  Collection<StoreFile> getFilesForScan(byte[] startRow, boolean includeStartRow, byte[] stopRow,
119      boolean includeStopRow);
120
121  /**
122   * Gets initial, full list of candidate store files to check for row-key-before.
123   * @param targetKey The key that is the basis of the search.
124   * @return The files that may have the key less than or equal to targetKey, in reverse
125   *         order of new-ness, and preference for target key.
126   */
127  Iterator<StoreFile> getCandidateFilesForRowKeyBefore(
128    KeyValue targetKey
129  );
130
131  /**
132   * Updates the candidate list for finding row key before. Based on the list of candidates
133   * remaining to check from getCandidateFilesForRowKeyBefore, targetKey and current candidate,
134   * may trim and reorder the list to remove the files where a better candidate cannot be found.
135   * @param candidateFiles The candidate files not yet checked for better candidates - return
136   *                       value from {@link #getCandidateFilesForRowKeyBefore(KeyValue)},
137   *                       with some files already removed.
138   * @param targetKey The key to search for.
139   * @param candidate The current best candidate found.
140   * @return The list to replace candidateFiles.
141   */
142  Iterator<StoreFile> updateCandidateFilesForRowKeyBefore(
143    Iterator<StoreFile> candidateFiles, KeyValue targetKey, Cell candidate
144  );
145
146
147  /**
148   * Gets the split point for the split of this set of store files (approx. middle).
149   * @return The mid-point, or null if no split is possible.
150   * @throws IOException
151   */
152  byte[] getSplitPoint() throws IOException;
153
154  /**
155   * @return The store compaction priority.
156   */
157  int getStoreCompactionPriority();
158
159  /**
160   * @param maxTs Maximum expired timestamp.
161   * @param filesCompacting Files that are currently compacting.
162   * @return The files which don't have any necessary data according to TTL and other criteria.
163   */
164  Collection<StoreFile> getUnneededFiles(long maxTs, List<StoreFile> filesCompacting);
165
166  /**
167   * @return the compaction pressure used for compaction throughput tuning.
168   * @see Store#getCompactionPressure()
169   */
170  double getCompactionPressure();
171
172  /**
173   * @return the comparator used to sort storefiles. Usually, the
174   *         {@link StoreFile#getMaxSequenceId()} is the first priority.
175   */
176  Comparator<StoreFile> getStoreFileComparator();
177}
