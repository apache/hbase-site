135    LOG.info("globalMemStoreLimit="
136        + TraditionalBinaryPrefix
137            .long2String(this.server.getRegionServerAccounting().getGlobalMemStoreLimit(), "", 1)
138        + ", globalMemStoreLimitLowMark="
139        + TraditionalBinaryPrefix.long2String(
140          this.server.getRegionServerAccounting().getGlobalMemStoreLimitLowMark(), "", 1)
141        + ", Offheap="
142        + (this.server.getRegionServerAccounting().isOffheap()));

149  /**
150   * The memstore across all regions has exceeded the low water mark. Pick
151   * one region to flush and flush it synchronously (this is called from the
152   * flush thread)
153   * @return true if successful
154   */
155  private boolean flushOneForGlobalPressure(FlushType flushType) {
156    SortedMap<Long, Collection<HRegion>> regionsBySize = null;
157    switch(flushType) {
158      case ABOVE_OFFHEAP_HIGHER_MARK:
159      case ABOVE_OFFHEAP_LOWER_MARK:
160        regionsBySize = server.getCopyOfOnlineRegionsSortedByOffHeapSize();
161        break;
162      case ABOVE_ONHEAP_HIGHER_MARK:
163      case ABOVE_ONHEAP_LOWER_MARK:
164      default:
165        regionsBySize = server.getCopyOfOnlineRegionsSortedByOnHeapSize();
166    }
167    Set<HRegion> excludedRegions = new HashSet<>();
168
169    double secondaryMultiplier
170      = ServerRegionReplicaUtil.getRegionReplicaStoreFileRefreshMultiplier(conf);
171
172    boolean flushedOne = false;
173    while (!flushedOne) {
174      // Find the biggest region that doesn't have too many storefiles (might be null!)
175      HRegion bestFlushableRegion =
176          getBiggestMemStoreRegion(regionsBySize, excludedRegions, true);
177      // Find the biggest region, total, even if it might have too many flushes.
178      HRegion bestAnyRegion = getBiggestMemStoreRegion(regionsBySize, excludedRegions, false);
179      // Find the biggest region that is a secondary region
180      HRegion bestRegionReplica = getBiggestMemStoreOfRegionReplica(regionsBySize, excludedRegions);
181      if (bestAnyRegion == null) {
182        // If bestAnyRegion is null, assign replica. It may be null too. Next step is check for null
183        bestAnyRegion = bestRegionReplica;
184      }
185      if (bestAnyRegion == null) {
186        LOG.error("Above memory mark but there are no flushable regions!");
187        return false;
188      }
189
190      HRegion regionToFlush;
191      long bestAnyRegionSize;
192      long bestFlushableRegionSize;
193      switch(flushType) {
194        case ABOVE_OFFHEAP_HIGHER_MARK:
195        case ABOVE_OFFHEAP_LOWER_MARK:
196          bestAnyRegionSize = bestAnyRegion.getMemStoreOffHeapSize();
197          bestFlushableRegionSize = getMemStoreOffHeapSize(bestFlushableRegion);
198          break;
199
200        case ABOVE_ONHEAP_HIGHER_MARK:
201        case ABOVE_ONHEAP_LOWER_MARK:
202          bestAnyRegionSize = bestAnyRegion.getMemStoreHeapSize();
203          bestFlushableRegionSize = getMemStoreHeapSize(bestFlushableRegion);
204          break;
205
206        default:
207          bestAnyRegionSize = bestAnyRegion.getMemStoreDataSize();
208          bestFlushableRegionSize = getMemStoreDataSize(bestFlushableRegion);
209      }
210      if (bestAnyRegionSize > 2 * bestFlushableRegionSize) {
211        // Even if it's not supposed to be flushed, pick a region if it's more than twice
212        // as big as the best flushable one - otherwise when we're under pressure we make
213        // lots of little flushes and cause lots of compactions, etc, which just makes
214        // life worse!
215        if (LOG.isDebugEnabled()) {
216          LOG.debug("Under global heap pressure: " + "Region "
217              + bestAnyRegion.getRegionInfo().getRegionNameAsString()
218              + " has too many " + "store files, but is "
219              + TraditionalBinaryPrefix.long2String(bestAnyRegionSize, "", 1)
220              + " vs best flushable region's "
221              + TraditionalBinaryPrefix.long2String(
222              bestFlushableRegionSize, "", 1)
223              + ". Choosing the bigger.");
224        }
225        regionToFlush = bestAnyRegion;
226      } else {
227        if (bestFlushableRegion == null) {
228          regionToFlush = bestAnyRegion;
229        } else {
230          regionToFlush = bestFlushableRegion;
231        }
232      }
233
234      long regionToFlushSize;
235      long bestRegionReplicaSize;
236      switch(flushType) {
237        case ABOVE_OFFHEAP_HIGHER_MARK:
238        case ABOVE_OFFHEAP_LOWER_MARK:
239          regionToFlushSize = regionToFlush.getMemStoreOffHeapSize();
240          bestRegionReplicaSize = getMemStoreOffHeapSize(bestRegionReplica);
241          break;
242
243        case ABOVE_ONHEAP_HIGHER_MARK:
244        case ABOVE_ONHEAP_LOWER_MARK:
245          regionToFlushSize = regionToFlush.getMemStoreHeapSize();
246          bestRegionReplicaSize = getMemStoreHeapSize(bestRegionReplica);
247          break;
248
249        default:
250          regionToFlushSize = regionToFlush.getMemStoreDataSize();
251          bestRegionReplicaSize = getMemStoreDataSize(bestRegionReplica);
252      }
253
254      if ((regionToFlush == null || regionToFlushSize == 0) && bestRegionReplicaSize == 0) {
255        // A concurrency issue (such as splitting region) may happen such that the online region
256        // seen by getCopyOfOnlineRegionsSortedByXX() method is no longer eligible to
257        // getBiggestMemStoreRegion(). This means that we can come out of the loop
258        LOG.debug("Above memory mark but there is no flushable region");
259        return false;
260      }
261
262      if (regionToFlush == null ||
263          (bestRegionReplica != null &&
264           ServerRegionReplicaUtil.isRegionReplicaStoreFileRefreshEnabled(conf) &&
265           (bestRegionReplicaSize > secondaryMultiplier * regionToFlushSize))) {
266        LOG.info("Refreshing storefiles of region " + bestRegionReplica +
267            " due to global heap pressure. Total memstore off heap size=" +
268            TraditionalBinaryPrefix.long2String(
269              server.getRegionServerAccounting().getGlobalMemStoreOffHeapSize(), "", 1) +
270            " memstore heap size=" + TraditionalBinaryPrefix.long2String(
271              server.getRegionServerAccounting().getGlobalMemStoreHeapSize(), "", 1));
272        flushedOne = refreshStoreFilesAndReclaimMemory(bestRegionReplica);
273        if (!flushedOne) {
274          LOG.info("Excluding secondary region " + bestRegionReplica +
275              " - trying to find a different region to refresh files.");
276          excludedRegions.add(bestRegionReplica);
277        }
278      } else {
279        LOG.info("Flush of region " + regionToFlush + " due to global heap pressure. " +
280            "Flush type=" + flushType.toString() +
281            ", Total Memstore Heap size=" +
282            TraditionalBinaryPrefix.long2String(
283                server.getRegionServerAccounting().getGlobalMemStoreHeapSize(), "", 1) +
284            ", Total Memstore Off-Heap size=" +
285            TraditionalBinaryPrefix.long2String(
286                server.getRegionServerAccounting().getGlobalMemStoreOffHeapSize(), "", 1) +
287            ", Region memstore size=" +
288            TraditionalBinaryPrefix.long2String(regionToFlushSize, "", 1));
289        flushedOne = flushRegion(regionToFlush, true, null, FlushLifeCycleTracker.DUMMY);
290
291        if (!flushedOne) {
292          LOG.info("Excluding unflushable region " + regionToFlush +
293              " - trying to find a different region to flush.");
294          excludedRegions.add(regionToFlush);
295        }
296      }
297    }
298    return true;
299  }

322  private class FlushHandler extends Thread {
323
324    private FlushHandler(String name) {
325      super(name);
326    }
327
328    @Override
329    public void run() {
330      while (!server.isStopped()) {
331        FlushQueueEntry fqe = null;
332        try {
333          wakeupPending.set(false); // allow someone to wake us up again
334          fqe = flushQueue.poll(threadWakeFrequency, TimeUnit.MILLISECONDS);
335          if (fqe == null || fqe == WAKEUPFLUSH_INSTANCE) {
336            FlushType type = isAboveLowWaterMark();
337            if (type != FlushType.NORMAL) {
338              LOG.debug("Flush thread woke up because memory above low water="
339                  + TraditionalBinaryPrefix.long2String(
340                    server.getRegionServerAccounting().getGlobalMemStoreLimitLowMark(), "", 1));
341              // For offheap memstore, even if the lower water mark was breached due to heap overhead
342              // we still select the regions based on the region's memstore data size.
343              // TODO : If we want to decide based on heap over head it can be done without tracking
344              // it per region.
345              if (!flushOneForGlobalPressure(type)) {
346                // Wasn't able to flush any region, but we're above low water mark
347                // This is unlikely to happen, but might happen when closing the
348                // entire server - another thread is flushing regions. We'll just
349                // sleep a little bit to avoid spinning, and then pretend that
350                // we flushed one, so anyone blocked will check again
351                Thread.sleep(1000);
352                wakeUpIfBlocking();
353              }
354              // Enqueue another one of these tokens so we'll wake up again
355              wakeupFlushThread();
356            }
357            continue;
358          }
359          FlushRegionEntry fre = (FlushRegionEntry) fqe;
360          if (!flushRegion(fre)) {
361            break;
362          }
363        } catch (InterruptedException ex) {
364          continue;
365        } catch (ConcurrentModificationException ex) {
366          continue;
367        } catch (Exception ex) {
368          LOG.error("Cache flusher failed for entry " + fqe, ex);
369          if (!server.checkFileSystem()) {
370            break;
371          }
372        }
373      }
374      synchronized (regionsInQueue) {
375        regionsInQueue.clear();
376        flushQueue.clear();
377      }
378
379      // Signal anyone waiting, so they see the close flag
380      wakeUpIfBlocking();
381      LOG.info(getName() + " exiting");
382    }
383  }
384
385
386  private void wakeupFlushThread() {
387    if (wakeupPending.compareAndSet(false, true)) {
388      flushQueue.add(WAKEUPFLUSH_INSTANCE);
389    }
390  }

392  private HRegion getBiggestMemStoreRegion(
393      SortedMap<Long, Collection<HRegion>> regionsBySize,
394      Set<HRegion> excludedRegions,
395      boolean checkStoreFileCount) {
396    synchronized (regionsInQueue) {
397      for (Map.Entry<Long, Collection<HRegion>> entry : regionsBySize.entrySet()) {
398        for (HRegion region : entry.getValue()) {
399          if (excludedRegions.contains(region)) {
400            continue;
401          }
402
403          if (region.writestate.flushing || !region.writestate.writesEnabled) {
404            continue;
405          }
406
407          if (checkStoreFileCount && isTooManyStoreFiles(region)) {
408            continue;
409          }
410          return region;
411        }
412      }
413    }
414    return null;
415  }
416
417  private HRegion getBiggestMemStoreOfRegionReplica(
418      SortedMap<Long, Collection<HRegion>> regionsBySize,
419      Set<HRegion> excludedRegions) {
420    synchronized (regionsInQueue) {
421      for (Map.Entry<Long, Collection<HRegion>> entry : regionsBySize.entrySet()) {
422        for (HRegion region : entry.getValue()) {
423          if (excludedRegions.contains(region)) {
424            continue;
425          }
426
427          if (RegionReplicaUtil.isDefaultReplica(region.getRegionInfo())) {
428            continue;
429          }
430          return region;
431        }
432      }
433    }
434    return null;
435  }

465  @Override
466  public boolean requestFlush(HRegion r, List<byte[]> families,
467      FlushLifeCycleTracker tracker) {
468    synchronized (regionsInQueue) {
469      if (!regionsInQueue.containsKey(r)) {
470        // This entry has no delay so it will be added at the top of the flush
471        // queue. It'll come out near immediately.
472        FlushRegionEntry fqe = new FlushRegionEntry(r, families, tracker);
473        this.regionsInQueue.put(r, fqe);
474        this.flushQueue.add(fqe);
475        r.incrementFlushesQueuedCount();
476        return true;
477      } else {
478        tracker.notExecuted("Flush already requested on " + r);
479        return false;
480      }
481    }
482  }
483
484  @Override
485  public boolean requestDelayedFlush(HRegion r, long delay) {
486    synchronized (regionsInQueue) {
487      if (!regionsInQueue.containsKey(r)) {
488        // This entry has some delay
489        FlushRegionEntry fqe =
490            new FlushRegionEntry(r, null, FlushLifeCycleTracker.DUMMY);
491        fqe.requeue(delay);
492        this.regionsInQueue.put(r, fqe);
493        this.flushQueue.add(fqe);
494        r.incrementFlushesQueuedCount();
495        return true;
496      }
497      return false;
498    }
499  }

505  /**
506   * Only interrupt once it's done with a run through the work loop.
507   */
508  void interruptIfNecessary() {
509    lock.writeLock().lock();
510    try {
511      for (FlushHandler flushHander : flushHandlers) {
512        if (flushHander != null) flushHander.interrupt();
513      }
514    } finally {
515      lock.writeLock().unlock();
516    }

519  synchronized void start(UncaughtExceptionHandler eh) {
520    ThreadFactory flusherThreadFactory = Threads.newDaemonThreadFactory(
521        server.getServerName().toShortString() + "-MemStoreFlusher", eh);
522    for (int i = 0; i < flushHandlers.length; i++) {
523      flushHandlers[i] = new FlushHandler("MemStoreFlusher." + i);
524      flusherThreadFactory.newThread(flushHandlers[i]);
525      flushHandlers[i].start();
526    }

546  /**
547   * A flushRegion that checks store file count.  If too many, puts the flush
548   * on delay queue to retry later.
549   * @param fqe
550   * @return true if the region was successfully flushed, false otherwise. If
551   * false, there will be accompanying log messages explaining why the region was
552   * not flushed.
553   */
554  private boolean flushRegion(final FlushRegionEntry fqe) {
555    HRegion region = fqe.region;
556    if (!region.getRegionInfo().isMetaRegion() && isTooManyStoreFiles(region)) {
557      if (fqe.isMaximumWait(this.blockingWaitTime)) {
558        LOG.info("Waited " + (EnvironmentEdgeManager.currentTime() - fqe.createTime) +
559          "ms on a compaction to clean up 'too many store files'; waited " +
560          "long enough... proceeding with flush of " +
561          region.getRegionInfo().getRegionNameAsString());
562      } else {
563        // If this is first time we've been put off, then emit a log message.
564        if (fqe.getRequeueCount() <= 0) {
565          // Note: We don't impose blockingStoreFiles constraint on meta regions
566          LOG.warn("{} has too many store files({}); delaying flush up to {} ms",
567              region.getRegionInfo().getEncodedName(), getStoreFileCount(region),
568              this.blockingWaitTime);
569          if (!this.server.compactSplitThread.requestSplit(region)) {
570            try {
571              this.server.compactSplitThread.requestSystemCompaction(region,
572                Thread.currentThread().getName());
573            } catch (IOException e) {
574              e = e instanceof RemoteException ?
575                      ((RemoteException)e).unwrapRemoteException() : e;
576              LOG.error("Cache flush failed for region " +
577                Bytes.toStringBinary(region.getRegionInfo().getRegionName()), e);
578            }
579          }
580        }
581
582        // Put back on the queue.  Have it come back out of the queue
583        // after a delay of this.blockingWaitTime / 100 ms.
584        this.flushQueue.add(fqe.requeue(this.blockingWaitTime / 100));
585        // Tell a lie, it's not flushed but it's ok
586        return true;
587      }
588    }
589    return flushRegion(region, false, fqe.families, fqe.getTracker());
590  }

592  /**
593   * Flush a region.
594   * @param region Region to flush.
595   * @param emergencyFlush Set if we are being force flushed. If true the region
596   * needs to be removed from the flush queue. If false, when we were called
597   * from the main flusher run loop and we got the entry to flush by calling
598   * poll on the flush queue (which removed it).
599   * @param families stores of region to flush.
600   * @return true if the region was successfully flushed, false otherwise. If
601   * false, there will be accompanying log messages explaining why the region was
602   * not flushed.
603   */
604  private boolean flushRegion(HRegion region, boolean emergencyFlush,
605      List<byte[]> families, FlushLifeCycleTracker tracker) {
606    synchronized (this.regionsInQueue) {
607      FlushRegionEntry fqe = this.regionsInQueue.remove(region);
608      // Use the start time of the FlushRegionEntry if available
609      if (fqe != null && emergencyFlush) {
610        // Need to remove from region from delay queue. When NOT an
611        // emergencyFlush, then item was removed via a flushQueue.poll.
612        flushQueue.remove(fqe);
613      }
614    }
615
616    tracker.beforeExecution();
617    lock.readLock().lock();
618    try {
619      notifyFlushRequest(region, emergencyFlush);
620      FlushResult flushResult = region.flushcache(families, false, tracker);
621      boolean shouldCompact = flushResult.isCompactionNeeded();
622      // We just want to check the size
623      boolean shouldSplit = region.checkSplit() != null;
624      if (shouldSplit) {
625        this.server.compactSplitThread.requestSplit(region);
626      } else if (shouldCompact) {
627        server.compactSplitThread.requestSystemCompaction(region, Thread.currentThread().getName());
628      }
629    } catch (DroppedSnapshotException ex) {
630      // Cache flush can fail in a few places. If it fails in a critical
631      // section, we get a DroppedSnapshotException and a replay of wal
632      // is required. Currently the only way to do this is a restart of
633      // the server. Abort because hdfs is probably bad (HBASE-644 is a case
634      // where hdfs was bad but passed the hdfs check).
635      server.abort("Replay of WAL required. Forcing server shutdown", ex);
636      return false;
637    } catch (IOException ex) {
638      ex = ex instanceof RemoteException ? ((RemoteException) ex).unwrapRemoteException() : ex;
639      LOG.error(
640        "Cache flush failed"
641            + (region != null ? (" for region " +
642                Bytes.toStringBinary(region.getRegionInfo().getRegionName()))
643              : ""), ex);
644      if (!server.checkFileSystem()) {
645        return false;
646      }
647    } finally {
648      lock.readLock().unlock();
649      wakeUpIfBlocking();
650      tracker.afterExecution();
651    }
652    return true;
653  }

655  private void notifyFlushRequest(Region region, boolean emergencyFlush) {
656    FlushType type = null;
657    if (emergencyFlush) {
658      type = isAboveHighWaterMark();
659      if (type == null) {
660        type = isAboveLowWaterMark();
661      }
662    }
663    for (FlushRequestListener listener : flushRequestListeners) {
664      listener.flushRequested(type, region);
665    }
666  }

668  private void wakeUpIfBlocking() {
669    synchronized (blockSignal) {
670      blockSignal.notifyAll();
671    }
672  }

674  private boolean isTooManyStoreFiles(Region region) {
675
676    // When compaction is disabled, the region is flushable
677    if (!region.getTableDescriptor().isCompactionEnabled()) {
678      return false;
679    }
680
681    for (Store store : region.getStores()) {
682      if (store.hasTooManyStoreFiles()) {
683        return true;
684      }
685    }
686    return false;
687  }

690    int count = 0;
691    for (Store store : region.getStores()) {
692      count += store.getStorefilesCount();
693    }
694    return count;
695  }

697  /**
698   * Check if the regionserver's memstore memory usage is greater than the
699   * limit. If so, flush regions with the biggest memstores until we're down
700   * to the lower limit. This method blocks callers until we're down to a safe
701   * amount of memstore consumption.
702   */
703  public void reclaimMemStoreMemory() {
704    try (TraceScope scope = TraceUtil.createTrace("MemStoreFluser.reclaimMemStoreMemory")) {
705      FlushType flushType = isAboveHighWaterMark();
706      if (flushType != FlushType.NORMAL) {
707        TraceUtil.addTimelineAnnotation("Force Flush. We're above high water mark.");
708        long start = EnvironmentEdgeManager.currentTime();
709        long nextLogTimeMs = start;
710        synchronized (this.blockSignal) {
711          boolean blocked = false;
712          long startTime = 0;
713          boolean interrupted = false;
714          try {
715            flushType = isAboveHighWaterMark();
716            while (flushType != FlushType.NORMAL && !server.isStopped()) {
717              if (!blocked) {
718                startTime = EnvironmentEdgeManager.currentTime();
719                if (!server.getRegionServerAccounting().isOffheap()) {
720                  logMsg("global memstore heapsize",
721                      server.getRegionServerAccounting().getGlobalMemStoreHeapSize(),
722                      server.getRegionServerAccounting().getGlobalMemStoreLimit());
723                } else {
724                  switch (flushType) {
725                    case ABOVE_OFFHEAP_HIGHER_MARK:
726                      logMsg("the global offheap memstore datasize",
727                          server.getRegionServerAccounting().getGlobalMemStoreOffHeapSize(),
728                          server.getRegionServerAccounting().getGlobalMemStoreLimit());
729                      break;
730                    case ABOVE_ONHEAP_HIGHER_MARK:
731                      logMsg("global memstore heapsize",
732                          server.getRegionServerAccounting().getGlobalMemStoreHeapSize(),
733                          server.getRegionServerAccounting().getGlobalOnHeapMemStoreLimit());
734                      break;
735                    default:
736                      break;
737                  }
738                }
739              }
740              blocked = true;
741              wakeupFlushThread();
742              try {
743                // we should be able to wait forever, but we've seen a bug where
744                // we miss a notify, so put a 5 second bound on it at least.
745                blockSignal.wait(5 * 1000);
746              } catch (InterruptedException ie) {
747                LOG.warn("Interrupted while waiting");
748                interrupted = true;
749              }
750              long nowMs = EnvironmentEdgeManager.currentTime();
751              if (nowMs >= nextLogTimeMs) {
752                LOG.warn("Memstore is above high water mark and block {} ms", nowMs - start);
753                nextLogTimeMs = nowMs + 1000;
754              }
755              flushType = isAboveHighWaterMark();
756            }
757          } finally {
758            if (interrupted) {
759              Thread.currentThread().interrupt();
760            }
761          }
762
763          if(blocked){
764            final long totalTime = EnvironmentEdgeManager.currentTime() - startTime;
765            if(totalTime > 0){
766              this.updatesBlockedMsHighWater.add(totalTime);
767            }
768            LOG.info("Unblocking updates for server " + server.toString());
769          }
770        }
771      } else {
772        flushType = isAboveLowWaterMark();
773        if (flushType != FlushType.NORMAL) {
774          wakeupFlushThread();
775        }
776      }
777    }
778  }

781    LOG.info("Blocking updates: {} {} is >= blocking {}", type,
782        TraditionalBinaryPrefix.long2String(val, "", 1),
783        TraditionalBinaryPrefix.long2String(max, "", 1));
784  }

835  interface FlushQueueEntry extends Delayed {
836  }

838  /**
839   * Datastructure used in the flush queue.  Holds region and retry count.
840   * Keeps tabs on how old this object is.  Implements {@link Delayed}.  On
841   * construction, the delay is zero. When added to a delay queue, we'll come
842   * out near immediately.  Call {@link #requeue(long)} passing delay in
843   * milliseconds before readding to delay queue if you want it to stay there
844   * a while.
845   */
846  static class FlushRegionEntry implements FlushQueueEntry {
847    private final HRegion region;
848
849    private final long createTime;
850    private long whenToExpire;
851    private int requeueCount = 0;
852
853    private final List<byte[]> families;
854
855    private final FlushLifeCycleTracker tracker;
856
857    FlushRegionEntry(final HRegion r, List<byte[]> families,
858        FlushLifeCycleTracker tracker) {
859      this.region = r;
860      this.createTime = EnvironmentEdgeManager.currentTime();
861      this.whenToExpire = this.createTime;
862      this.families = families;
863      this.tracker = tracker;
864    }
865
866    /**
867     * @param maximumWait
868     * @return True if we have been delayed > <code>maximumWait</code> milliseconds.
869     */
870    public boolean isMaximumWait(final long maximumWait) {
871      return (EnvironmentEdgeManager.currentTime() - this.createTime) > maximumWait;
872    }
873
874    /**
875     * @return Count of times {@link #requeue(long)} was called; i.e this is
876     * number of times we've been requeued.
877     */
878    public int getRequeueCount() {
879      return this.requeueCount;
880    }
881
882    public FlushLifeCycleTracker getTracker() {
883      return tracker;
884    }
885
886    /**
887     * @param when When to expire, when to come up out of the queue.
888     * Specify in milliseconds.  This method adds EnvironmentEdgeManager.currentTime()
889     * to whatever you pass.
890     * @return This.
891     */
892    public FlushRegionEntry requeue(final long when) {
893      this.whenToExpire = EnvironmentEdgeManager.currentTime() + when;
894      this.requeueCount++;
895      return this;
896    }
897
898    @Override
899    public long getDelay(TimeUnit unit) {
900      return unit.convert(this.whenToExpire - EnvironmentEdgeManager.currentTime(),
901          TimeUnit.MILLISECONDS);
902    }
903
904    @Override
905    public int compareTo(Delayed other) {
906      // Delay is compared first. If there is a tie, compare region's hash code
907      int ret = Long.valueOf(getDelay(TimeUnit.MILLISECONDS) -
908        other.getDelay(TimeUnit.MILLISECONDS)).intValue();
909      if (ret != 0) {
910        return ret;
911      }
912      FlushQueueEntry otherEntry = (FlushQueueEntry) other;
913      return hashCode() - otherEntry.hashCode();
914    }
915
916    @Override
917    public String toString() {
918      return "[flush region "+Bytes.toStringBinary(region.getRegionInfo().getRegionName())+"]";
919    }
920
921    @Override
922    public int hashCode() {
923      int hash = (int) getDelay(TimeUnit.MILLISECONDS);
924      return hash ^ region.hashCode();
925    }
926
927   @Override
928    public boolean equals(Object obj) {
929      if (this == obj) {
930        return true;
931      }
932      if (obj == null || getClass() != obj.getClass()) {
933        return false;
934      }
935      FlushRegionEntry other = (FlushRegionEntry) obj;
936      if (!Bytes.equals(this.region.getRegionInfo().getRegionName(),
937          other.region.getRegionInfo().getRegionName())) {
938        return false;
939      }
940      return compareTo(other) == 0;
941    }
942  }
