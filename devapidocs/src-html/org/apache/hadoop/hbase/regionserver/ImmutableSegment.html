019package org.apache.hadoop.hbase.regionserver;
020
021
022import org.apache.hadoop.hbase.Cell;
023import org.apache.hadoop.hbase.CellComparator;
024import org.apache.hadoop.hbase.ExtendedCell;
025import org.apache.hadoop.hbase.KeyValue;
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027import org.apache.hadoop.hbase.util.ClassSize;
028import org.apache.hadoop.hbase.client.Scan;
029import org.apache.hadoop.hbase.io.TimeRange;
030import org.apache.hadoop.hbase.util.CollectionBackedScanner;
031
032import java.io.IOException;
033
034/**
035 * ImmutableSegment is an abstract class that extends the API supported by a {@link Segment},
036 * and is not needed for a {@link MutableSegment}. Specifically, the method
037 * {@link ImmutableSegment#getKeyValueScanner()} builds a special scanner for the
038 * {@link MemStoreSnapshot} object.
039 */
040@InterfaceAudience.Private
041public class ImmutableSegment extends Segment {
042
043  private static final long DEEP_OVERHEAD = Segment.DEEP_OVERHEAD
044      + (2 * ClassSize.REFERENCE) // Refs to timeRange and type
045      + ClassSize.TIMERANGE;
046  public static final long DEEP_OVERHEAD_CSLM = DEEP_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP;
047  public static final long DEEP_OVERHEAD_CAM = DEEP_OVERHEAD + ClassSize.CELL_ARRAY_MAP;
048
049  /**
050   * Types of ImmutableSegment
051   */
052  public enum Type {
053    SKIPLIST_MAP_BASED,
054    ARRAY_MAP_BASED,
055  }
056
057  /**
058   * This is an immutable segment so use the read-only TimeRange rather than the heavy-weight
059   * TimeRangeTracker with all its synchronization when doing time range stuff.
060   */
061  private final TimeRange timeRange;
062
063  private Type type = Type.SKIPLIST_MAP_BASED;
064
065  // whether it is based on CellFlatMap or ConcurrentSkipListMap
066  private boolean isFlat(){
067    return (type != Type.SKIPLIST_MAP_BASED);
068  }
069
070  /////////////////////  CONSTRUCTORS  /////////////////////
071  /**------------------------------------------------------------------------
072   * Copy C-tor to be used when new ImmutableSegment is being built from a Mutable one.
073   * This C-tor should be used when active MutableSegment is pushed into the compaction
074   * pipeline and becomes an ImmutableSegment.
075   */
076  protected ImmutableSegment(Segment segment) {
077    super(segment);
078    this.type = Type.SKIPLIST_MAP_BASED;
079    this.timeRange = this.timeRangeTracker == null ? null : this.timeRangeTracker.toTimeRange();
080  }
081
082  /**------------------------------------------------------------------------
083   * C-tor to be used when new CELL_ARRAY BASED ImmutableSegment is a result of compaction of a
084   * list of older ImmutableSegments.
085   * The given iterator returns the Cells that "survived" the compaction.
086   * The input parameter "type" exists for future use when more types of flat ImmutableSegments
087   * are going to be introduced.
088   */
089  protected ImmutableSegment(CellComparator comparator, MemStoreSegmentsIterator iterator,
090      MemStoreLAB memStoreLAB, int numOfCells, Type type, boolean merge) {
091
092    super(null, // initiailize the CellSet with NULL
093        comparator, memStoreLAB);
094    this.type = type;
095    // build the true CellSet based on CellArrayMap
096    CellSet cs = createCellArrayMapSet(numOfCells, iterator, merge);
097
098    this.setCellSet(null, cs);            // update the CellSet of the new Segment
099    this.timeRange = this.timeRangeTracker == null ? null : this.timeRangeTracker.toTimeRange();
100  }
101
102  /**------------------------------------------------------------------------
103   * C-tor to be used when new SKIP-LIST BASED ImmutableSegment is a result of compaction of a
104   * list of older ImmutableSegments.
105   * The given iterator returns the Cells that "survived" the compaction.
106   */
107  protected ImmutableSegment(CellComparator comparator, MemStoreSegmentsIterator iterator,
108      MemStoreLAB memStoreLAB) {
109    super(new CellSet(comparator), // initiailize the CellSet with empty CellSet
110        comparator, memStoreLAB);
111    type = Type.SKIPLIST_MAP_BASED;
112    MemstoreSize memstoreSize = new MemstoreSize();
113    while (iterator.hasNext()) {
114      Cell c = iterator.next();
115      // The scanner is doing all the elimination logic
116      // now we just copy it to the new segment
117      Cell newKV = maybeCloneWithAllocator(c);
118      boolean usedMSLAB = (newKV != c);
119      internalAdd(newKV, usedMSLAB, memstoreSize);
120    }
121    this.timeRange = this.timeRangeTracker == null ? null : this.timeRangeTracker.toTimeRange();
122  }
123
124  /////////////////////  PUBLIC METHODS  /////////////////////
125  /**
126   * Builds a special scanner for the MemStoreSnapshot object that is different than the
127   * general segment scanner.
128   * @return a special scanner for the MemStoreSnapshot object
129   */
130  public KeyValueScanner getKeyValueScanner() {
131    return new CollectionBackedScanner(getCellSet(), getComparator());
132  }
133
134  @Override
135  public boolean shouldSeek(Scan scan, long oldestUnexpiredTS) {
136    return this.timeRange.includesTimeRange(scan.getTimeRange()) &&
137        this.timeRange.getMax() >= oldestUnexpiredTS;
138  }
139
140  @Override
141  public long getMinTimestamp() {
142    return this.timeRange.getMin();
143  }
144
145  /**------------------------------------------------------------------------
146   * Change the CellSet of this ImmutableSegment from one based on ConcurrentSkipListMap to one
147   * based on CellArrayMap.
148   * If this ImmutableSegment is not based on ConcurrentSkipListMap , this is NOOP
149   *
150   * Synchronization of the CellSet replacement:
151   * The reference to the CellSet is AtomicReference and is updated only when ImmutableSegment
152   * is constructed (single thread) or flattened. The flattening happens as part of a single
153   * thread of compaction, but to be on the safe side the initial CellSet is locally saved
154   * before the flattening and then replaced using CAS instruction.
155   */
156  public boolean flatten(MemstoreSize memstoreSize) {
157    if (isFlat()) return false;
158    CellSet oldCellSet = getCellSet();
159    int numOfCells = getCellsCount();
160
161    // build the new (CellSet CellArrayMap based)
162    CellSet  newCellSet = recreateCellArrayMapSet(numOfCells);
163    type = Type.ARRAY_MAP_BASED;
164    setCellSet(oldCellSet,newCellSet);
165
166    // arrange the meta-data size, decrease all meta-data sizes related to SkipList
167    // (recreateCellArrayMapSet doesn't take the care for the sizes)
168    long newSegmentSizeDelta = -(numOfCells * ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
169    // add size of CellArrayMap and meta-data overhead per Cell
170    newSegmentSizeDelta = newSegmentSizeDelta + numOfCells * ClassSize.CELL_ARRAY_MAP_ENTRY;
171    incSize(0, newSegmentSizeDelta);
172    if (memstoreSize != null) {
173      memstoreSize.incMemstoreSize(0, newSegmentSizeDelta);
174    }
175
176    return true;
177  }
178
179  /////////////////////  PRIVATE METHODS  /////////////////////
180  /*------------------------------------------------------------------------*/
181  // Create CellSet based on CellArrayMap from compacting iterator
182  private CellSet createCellArrayMapSet(int numOfCells, MemStoreSegmentsIterator iterator,
183      boolean merge) {
184
185    Cell[] cells = new Cell[numOfCells];   // build the Cell Array
186    int i = 0;
187    while (iterator.hasNext()) {
188      Cell c = iterator.next();
189      // The scanner behind the iterator is doing all the elimination logic
190      if (merge) {
191        // if this is merge we just move the Cell object without copying MSLAB
192        // the sizes still need to be updated in the new segment
193        cells[i] = c;
194      } else {
195        // now we just copy it to the new segment (also MSLAB copy)
196        cells[i] = maybeCloneWithAllocator(c);
197      }
198      boolean useMSLAB = (getMemStoreLAB()!=null);
199      // second parameter true, because in compaction addition of the cell to new segment
200      // is always successful
201      updateMetaInfo(c, true, useMSLAB, null); // updates the size per cell
202      i++;
203    }
204    // build the immutable CellSet
205    CellArrayMap cam = new CellArrayMap(getComparator(), cells, 0, i, false);
206    return new CellSet(cam);
207  }
208
209  @Override
210  protected long heapOverheadChange(Cell cell, boolean succ) {
211    if (succ) {
212      switch (this.type) {
213      case SKIPLIST_MAP_BASED:
214        return super.heapOverheadChange(cell, succ);
215      case ARRAY_MAP_BASED:
216        if (cell instanceof ExtendedCell) {
217          return ClassSize
218              .align(ClassSize.CELL_ARRAY_MAP_ENTRY + ((ExtendedCell) cell).heapOverhead());
219        }
220        return ClassSize.align(ClassSize.CELL_ARRAY_MAP_ENTRY + KeyValue.FIXED_OVERHEAD);
221      }
222    }
223    return 0;
224  }
225
226  /*------------------------------------------------------------------------*/
227  // Create CellSet based on CellArrayMap from current ConcurrentSkipListMap based CellSet
228  // (without compacting iterator)
229  private CellSet recreateCellArrayMapSet(int numOfCells) {
230
231    Cell[] cells = new Cell[numOfCells];   // build the Cell Array
232    Cell curCell;
233    int idx = 0;
234    // create this segment scanner with maximal possible read point, to go over all Cells
235    SegmentScanner segmentScanner = this.getScanner(Long.MAX_VALUE);
236
237    try {
238      while ((curCell = segmentScanner.next()) != null) {
239        cells[idx++] = curCell;
240      }
241    } catch (IOException ie) {
242      throw new IllegalStateException(ie);
243    } finally {
244      segmentScanner.close();
245    }
246
247    // build the immutable CellSet
248    CellArrayMap cam = new CellArrayMap(getComparator(), cells, 0, idx, false);
249    return new CellSet(cam);
250  }
