064/*
065 * The CompactionTool allows to execute a compaction specifying a:
066 * <ul>
067 *  <li>table folder (all regions and families will be compacted)
068 *  <li>region folder (all families in the region will be compacted)
069 *  <li>family folder (the store files will be compacted)
070 * </ul>
071 */
072@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
073public class CompactionTool extends Configured implements Tool {
074  private static final Log LOG = LogFactory.getLog(CompactionTool.class);
075
076  private final static String CONF_TMP_DIR = "hbase.tmp.dir";
077  private final static String CONF_COMPACT_ONCE = "hbase.compactiontool.compact.once";
078  private final static String CONF_COMPACT_MAJOR = "hbase.compactiontool.compact.major";
079  private final static String CONF_DELETE_COMPACTED = "hbase.compactiontool.delete";
080  private final static String CONF_COMPLETE_COMPACTION = "hbase.hstore.compaction.complete";
081
082  /**
083   * Class responsible to execute the Compaction on the specified path.
084   * The path can be a table, region or family directory.
085   */
086  private static class CompactionWorker {
087    private final boolean keepCompactedFiles;
088    private final boolean deleteCompacted;
089    private final Configuration conf;
090    private final FileSystem fs;
091    private final Path tmpDir;
092
093    public CompactionWorker(final FileSystem fs, final Configuration conf) {
094      this.conf = conf;
095      this.keepCompactedFiles = !conf.getBoolean(CONF_COMPLETE_COMPACTION, true);
096      this.deleteCompacted = conf.getBoolean(CONF_DELETE_COMPACTED, false);
097      this.tmpDir = new Path(conf.get(CONF_TMP_DIR));
098      this.fs = fs;
099    }
100
101    /**
102     * Execute the compaction on the specified path.
103     *
104     * @param path Directory path on which to run compaction.
105     * @param compactOnce Execute just a single step of compaction.
106     * @param major Request major compaction.
107     */
108    public void compact(final Path path, final boolean compactOnce, final boolean major) throws IOException {
109      if (isFamilyDir(fs, path)) {
110        Path regionDir = path.getParent();
111        Path tableDir = regionDir.getParent();
112        TableDescriptor htd = FSTableDescriptors.getTableDescriptorFromFs(fs, tableDir);
113        HRegionInfo hri = HRegionFileSystem.loadRegionInfoFileContent(fs, regionDir);
114        compactStoreFiles(tableDir, htd, hri,
115            path.getName(), compactOnce, major);
116      } else if (isRegionDir(fs, path)) {
117        Path tableDir = path.getParent();
118        TableDescriptor htd = FSTableDescriptors.getTableDescriptorFromFs(fs, tableDir);
119        compactRegion(tableDir, htd, path, compactOnce, major);
120      } else if (isTableDir(fs, path)) {
121        compactTable(path, compactOnce, major);
122      } else {
123        throw new IOException(
124          "Specified path is not a table, region or family directory. path=" + path);
125      }
126    }
127
128    private void compactTable(final Path tableDir, final boolean compactOnce, final boolean major)
129        throws IOException {
130      TableDescriptor htd = FSTableDescriptors.getTableDescriptorFromFs(fs, tableDir);
131      for (Path regionDir: FSUtils.getRegionDirs(fs, tableDir)) {
132        compactRegion(tableDir, htd, regionDir, compactOnce, major);
133      }
134    }
135
136    private void compactRegion(final Path tableDir, final TableDescriptor htd,
137        final Path regionDir, final boolean compactOnce, final boolean major)
138        throws IOException {
139      HRegionInfo hri = HRegionFileSystem.loadRegionInfoFileContent(fs, regionDir);
140      for (Path familyDir: FSUtils.getFamilyDirs(fs, regionDir)) {
141        compactStoreFiles(tableDir, htd, hri, familyDir.getName(), compactOnce, major);
142      }
143    }
144
145    /**
146     * Execute the actual compaction job.
147     * If the compact once flag is not specified, execute the compaction until
148     * no more compactions are needed. Uses the Configuration settings provided.
149     */
150    private void compactStoreFiles(final Path tableDir, final TableDescriptor htd,
151        final HRegionInfo hri, final String familyName, final boolean compactOnce,
152        final boolean major) throws IOException {
153      HStore store = getStore(conf, fs, tableDir, htd, hri, familyName, tmpDir);
154      LOG.info("Compact table=" + htd.getTableName() +
155        " region=" + hri.getRegionNameAsString() +
156        " family=" + familyName);
157      if (major) {
158        store.triggerMajorCompaction();
159      }
160      do {
161        CompactionContext compaction = store.requestCompaction(Store.PRIORITY_USER, null);
162        if (compaction == null) break;
163        List<StoreFile> storeFiles =
164            store.compact(compaction, NoLimitThroughputController.INSTANCE);
165        if (storeFiles != null && !storeFiles.isEmpty()) {
166          if (keepCompactedFiles && deleteCompacted) {
167            for (StoreFile storeFile: storeFiles) {
168              fs.delete(storeFile.getPath(), false);
169            }
170          }
171        }
172      } while (store.needsCompaction() && !compactOnce);
173    }
174
175    /**
176     * Create a "mock" HStore that uses the tmpDir specified by the user and
177     * the store dir to compact as source.
178     */
179    private static HStore getStore(final Configuration conf, final FileSystem fs,
180        final Path tableDir, final TableDescriptor htd, final HRegionInfo hri,
181        final String familyName, final Path tempDir) throws IOException {
182      HRegionFileSystem regionFs = new HRegionFileSystem(conf, fs, tableDir, hri) {
183        @Override
184        public Path getTempDir() {
185          return tempDir;
186        }
187      };
188      HRegion region = new HRegion(regionFs, null, conf, htd, null);
189      return new HStore(region, htd.getColumnFamily(Bytes.toBytes(familyName)), conf);
190    }
191  }
192
193  private static boolean isRegionDir(final FileSystem fs, final Path path) throws IOException {
194    Path regionInfo = new Path(path, HRegionFileSystem.REGION_INFO_FILE);
195    return fs.exists(regionInfo);
196  }
197
198  private static boolean isTableDir(final FileSystem fs, final Path path) throws IOException {
199    return FSTableDescriptors.getTableInfoPath(fs, path) != null;
200  }
201
202  private static boolean isFamilyDir(final FileSystem fs, final Path path) throws IOException {
203    return isRegionDir(fs, path.getParent());
204  }

206  private static class CompactionMapper
207      extends Mapper<LongWritable, Text, NullWritable, NullWritable> {
208    private CompactionWorker compactor = null;
209    private boolean compactOnce = false;
210    private boolean major = false;
211
212    @Override
213    public void setup(Context context) {
214      Configuration conf = context.getConfiguration();
215      compactOnce = conf.getBoolean(CONF_COMPACT_ONCE, false);
216      major = conf.getBoolean(CONF_COMPACT_MAJOR, false);
217
218      try {
219        FileSystem fs = FileSystem.get(conf);
220        this.compactor = new CompactionWorker(fs, conf);
221      } catch (IOException e) {
222        throw new RuntimeException("Could not get the input FileSystem", e);
223      }
224    }
225
226    @Override
227    public void map(LongWritable key, Text value, Context context)
228        throws InterruptedException, IOException {
229      Path path = new Path(value.toString());
230      this.compactor.compact(path, compactOnce, major);
231    }
232  }

234  /**
235   * Input format that uses store files block location as input split locality.
236   */
237  private static class CompactionInputFormat extends TextInputFormat {
238    @Override
239    protected boolean isSplitable(JobContext context, Path file) {
240      return true;
241    }
242
243    /**
244     * Returns a split for each store files directory using the block location
245     * of each file as locality reference.
246     */
247    @Override
248    public List<InputSplit> getSplits(JobContext job) throws IOException {
249      List<InputSplit> splits = new ArrayList<>();
250      List<FileStatus> files = listStatus(job);
251
252      Text key = new Text();
253      for (FileStatus file: files) {
254        Path path = file.getPath();
255        FileSystem fs = path.getFileSystem(job.getConfiguration());
256        LineReader reader = new LineReader(fs.open(path));
257        long pos = 0;
258        int n;
259        try {
260          while ((n = reader.readLine(key)) > 0) {
261            String[] hosts = getStoreDirHosts(fs, path);
262            splits.add(new FileSplit(path, pos, n, hosts));
263            pos += n;
264          }
265        } finally {
266          reader.close();
267        }
268      }
269
270      return splits;
271    }
272
273    /**
274     * return the top hosts of the store files, used by the Split
275     */
276    private static String[] getStoreDirHosts(final FileSystem fs, final Path path)
277        throws IOException {
278      FileStatus[] files = FSUtils.listStatus(fs, path);
279      if (files == null) {
280        return new String[] {};
281      }
282
283      HDFSBlocksDistribution hdfsBlocksDistribution = new HDFSBlocksDistribution();
284      for (FileStatus hfileStatus: files) {
285        HDFSBlocksDistribution storeFileBlocksDistribution =
286          FSUtils.computeHDFSBlocksDistribution(fs, hfileStatus, 0, hfileStatus.getLen());
287        hdfsBlocksDistribution.add(storeFileBlocksDistribution);
288      }
289
290      List<String> hosts = hdfsBlocksDistribution.getTopHosts();
291      return hosts.toArray(new String[hosts.size()]);
292    }

294    /**
295     * Create the input file for the given directories to compact.
296     * The file is a TextFile with each line corrisponding to a
297     * store files directory to compact.
298     */
299    public static void createInputFile(final FileSystem fs, final Path path,
300        final Set<Path> toCompactDirs) throws IOException {
301      // Extract the list of store dirs
302      List<Path> storeDirs = new LinkedList<>();
303      for (Path compactDir: toCompactDirs) {
304        if (isFamilyDir(fs, compactDir)) {
305          storeDirs.add(compactDir);
306        } else if (isRegionDir(fs, compactDir)) {
307          for (Path familyDir: FSUtils.getFamilyDirs(fs, compactDir)) {
308            storeDirs.add(familyDir);
309          }
310        } else if (isTableDir(fs, compactDir)) {
311          // Lookup regions
312          for (Path regionDir: FSUtils.getRegionDirs(fs, compactDir)) {
313            for (Path familyDir: FSUtils.getFamilyDirs(fs, regionDir)) {
314              storeDirs.add(familyDir);
315            }
316          }
317        } else {
318          throw new IOException(
319            "Specified path is not a table, region or family directory. path=" + compactDir);
320        }
321      }
322
323      // Write Input File
324      FSDataOutputStream stream = fs.create(path);
325      LOG.info("Create input file=" + path + " with " + storeDirs.size() + " dirs to compact.");
326      try {
327        final byte[] newLine = Bytes.toBytes("\n");
328        for (Path storeDir: storeDirs) {
329          stream.write(Bytes.toBytes(storeDir.toString()));
330          stream.write(newLine);
331        }
332      } finally {
333        stream.close();
334      }
335    }
336  }

338  /**
339   * Execute compaction, using a Map-Reduce job.
340   */
341  private int doMapReduce(final FileSystem fs, final Set<Path> toCompactDirs,
342      final boolean compactOnce, final boolean major) throws Exception {
343    Configuration conf = getConf();
344    conf.setBoolean(CONF_COMPACT_ONCE, compactOnce);
345    conf.setBoolean(CONF_COMPACT_MAJOR, major);
346
347    Job job = new Job(conf);
348    job.setJobName("CompactionTool");
349    job.setJarByClass(CompactionTool.class);
350    job.setMapperClass(CompactionMapper.class);
351    job.setInputFormatClass(CompactionInputFormat.class);
352    job.setOutputFormatClass(NullOutputFormat.class);
353    job.setMapSpeculativeExecution(false);
354    job.setNumReduceTasks(0);
355
356    // add dependencies (including HBase ones)
357    TableMapReduceUtil.addDependencyJars(job);
358
359    Path stagingDir = JobUtil.getStagingDir(conf);
360    try {
361      // Create input file with the store dirs
362      Path inputPath = new Path(stagingDir, "compact-"+ EnvironmentEdgeManager.currentTime());
363      CompactionInputFormat.createInputFile(fs, inputPath, toCompactDirs);
364      CompactionInputFormat.addInputPath(job, inputPath);
365
366      // Initialize credential for secure cluster
367      TableMapReduceUtil.initCredentials(job);
368
369      // Start the MR Job and wait
370      return job.waitForCompletion(true) ? 0 : 1;
371    } finally {
372      fs.delete(stagingDir, true);
373    }
374  }
375
376  /**
377   * Execute compaction, from this client, one path at the time.
378   */
379  private int doClient(final FileSystem fs, final Set<Path> toCompactDirs,
380      final boolean compactOnce, final boolean major) throws IOException {
381    CompactionWorker worker = new CompactionWorker(fs, getConf());
382    for (Path path: toCompactDirs) {
383      worker.compact(path, compactOnce, major);
384    }
385    return 0;
386  }

388  @Override
389  public int run(String[] args) throws Exception {
390    Set<Path> toCompactDirs = new HashSet<>();
391    boolean compactOnce = false;
392    boolean major = false;
393    boolean mapred = false;
394
395    Configuration conf = getConf();
396    FileSystem fs = FileSystem.get(conf);
397
398    try {
399      for (int i = 0; i < args.length; ++i) {
400        String opt = args[i];
401        if (opt.equals("-compactOnce")) {
402          compactOnce = true;
403        } else if (opt.equals("-major")) {
404          major = true;
405        } else if (opt.equals("-mapred")) {
406          mapred = true;
407        } else if (!opt.startsWith("-")) {
408          Path path = new Path(opt);
409          FileStatus status = fs.getFileStatus(path);
410          if (!status.isDirectory()) {
411            printUsage("Specified path is not a directory. path=" + path);
412            return 1;
413          }
414          toCompactDirs.add(path);
415        } else {
416          printUsage();
417        }
418      }
419    } catch (Exception e) {
420      printUsage(e.getMessage());
421      return 1;
422    }
423
424    if (toCompactDirs.isEmpty()) {
425      printUsage("No directories to compact specified.");
426      return 1;
427    }
428
429    // Execute compaction!
430    if (mapred) {
431      return doMapReduce(fs, toCompactDirs, compactOnce, major);
432    } else {
433      return doClient(fs, toCompactDirs, compactOnce, major);
434    }
435  }
436
437  private void printUsage() {
438    printUsage(null);
439  }
440
441  private void printUsage(final String message) {
442    if (message != null && message.length() > 0) {
443      System.err.println(message);
444    }
445    System.err.println("Usage: java " + this.getClass().getName() + " \\");
446    System.err.println("  [-compactOnce] [-major] [-mapred] [-D<property=value>]* files...");
447    System.err.println();
448    System.err.println("Options:");
449    System.err.println(" mapred         Use MapReduce to run compaction.");
450    System.err.println(" compactOnce    Execute just one compaction step. (default: while needed)");
451    System.err.println(" major          Trigger major compaction.");
452    System.err.println();
453    System.err.println("Note: -D properties will be applied to the conf used. ");
454    System.err.println("For example: ");
455    System.err.println(" To preserve input files, pass -D"+CONF_COMPLETE_COMPACTION+"=false");
456    System.err.println(" To stop delete of compacted file, pass -D"+CONF_DELETE_COMPACTED+"=false");
457    System.err.println(" To set tmp dir, pass -D"+CONF_TMP_DIR+"=ALTERNATE_DIR");
458    System.err.println();
459    System.err.println("Examples:");
460    System.err.println(" To compact the full 'TestTable' using MapReduce:");
461    System.err.println(" $ hbase " + this.getClass().getName() + " -mapred hdfs:///hbase/data/default/TestTable");
462    System.err.println();
463    System.err.println(" To compact column family 'x' of the table 'TestTable' region 'abc':");
464    System.err.println(" $ hbase " + this.getClass().getName() + " hdfs:///hbase/data/default/TestTable/abc/x");
465  }
