021import org.apache.hadoop.hbase.ByteBufferKeyValue;
022import org.apache.hadoop.hbase.Cell;
023import org.apache.hadoop.hbase.CellComparator;
024import org.apache.hadoop.hbase.KeyValue;
025import org.apache.hadoop.hbase.KeyValueUtil;
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027import org.apache.hadoop.hbase.util.ByteBufferUtils;
028import org.apache.hadoop.hbase.util.ClassSize;
029
030import java.io.IOException;
031import java.nio.ByteBuffer;

033/**
034 * CellChunkImmutableSegment extends the API supported by a {@link Segment},
035 * and {@link ImmutableSegment}. This immutable segment is working with CellSet with
036 * CellChunkMap delegatee.
037 */
038@InterfaceAudience.Private
039public class CellChunkImmutableSegment extends ImmutableSegment {
040
041  public static final long DEEP_OVERHEAD_CCM =
042      ImmutableSegment.DEEP_OVERHEAD + ClassSize.CELL_CHUNK_MAP;
043
044  /////////////////////  CONSTRUCTORS  /////////////////////
045  /**------------------------------------------------------------------------
046   * C-tor to be used when new CellChunkImmutableSegment is built as a result of compaction/merge
047   * of a list of older ImmutableSegments.
048   * The given iterator returns the Cells that "survived" the compaction.
049   */
050  protected CellChunkImmutableSegment(CellComparator comparator, MemStoreSegmentsIterator iterator,
051      MemStoreLAB memStoreLAB, int numOfCells, MemStoreCompactor.Action action) {
052    super(null, comparator, memStoreLAB); // initialize the CellSet with NULL
053    incSize(0, DEEP_OVERHEAD_CCM); // initiate the heapSize with the size of the segment metadata
054    // build the new CellSet based on CellArrayMap and update the CellSet of the new Segment
055    initializeCellSet(numOfCells, iterator, action);
056  }
057
058  /**------------------------------------------------------------------------
059   * C-tor to be used when new CellChunkImmutableSegment is built as a result of flattening
060   * of CSLMImmutableSegment
061   * The given iterator returns the Cells that "survived" the compaction.
062   */
063  protected CellChunkImmutableSegment(CSLMImmutableSegment segment, MemstoreSize memstoreSize) {
064    super(segment); // initiailize the upper class
065    incSize(0,-CSLMImmutableSegment.DEEP_OVERHEAD_CSLM+ CellChunkImmutableSegment.DEEP_OVERHEAD_CCM);
066    int numOfCells = segment.getCellsCount();
067    // build the new CellSet based on CellChunkMap
068    reinitializeCellSet(numOfCells, segment.getScanner(Long.MAX_VALUE), segment.getCellSet());
069    // arrange the meta-data size, decrease all meta-data sizes related to SkipList;
070    // add sizes of CellChunkMap entry, decrease also Cell object sizes
071    // (reinitializeCellSet doesn't take the care for the sizes)
072    long newSegmentSizeDelta = numOfCells*(indexEntrySize()-ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
073
074    incSize(0, newSegmentSizeDelta);
075    memstoreSize.incMemstoreSize(0, newSegmentSizeDelta);
076  }
077
078  @Override
079  protected long indexEntrySize() {
080    return (ClassSize.CELL_CHUNK_MAP_ENTRY - KeyValue.FIXED_OVERHEAD);
081  }
082
083  @Override
084  protected boolean canBeFlattened() {
085    return false;
086  }
087
088  /////////////////////  PRIVATE METHODS  /////////////////////
089  /*------------------------------------------------------------------------*/
090  // Create CellSet based on CellChunkMap from compacting iterator
091  private void initializeCellSet(int numOfCells, MemStoreSegmentsIterator iterator,
092      MemStoreCompactor.Action action) {
093
094    // calculate how many chunks we will need for index
095    int chunkSize = ChunkCreator.getInstance().getChunkSize();
096    int numOfCellsInChunk = CellChunkMap.NUM_OF_CELL_REPS_IN_CHUNK;
097    int numberOfChunks = calculateNumberOfChunks(numOfCells,numOfCellsInChunk);
098    int numOfCellsAfterCompaction = 0;
099    int currentChunkIdx = 0;
100    int offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
101    // all index Chunks are allocated from ChunkCreator
102    Chunk[] chunks = new Chunk[numberOfChunks];
103    for (int i=0; i < numberOfChunks; i++) {
104      chunks[i] = this.getMemStoreLAB().getNewExternalChunk();
105    }
106    while (iterator.hasNext()) {        // the iterator hides the elimination logic for compaction
107      Cell c = iterator.next();
108      numOfCellsAfterCompaction++;
109      assert (c instanceof ByteBufferKeyValue); // shouldn't get here anything but ByteBufferKeyValue
110      if (offsetInCurentChunk + ClassSize.CELL_CHUNK_MAP_ENTRY > chunkSize) {
111        currentChunkIdx++;              // continue to the next index chunk
112        offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
113      }
114      if (action == MemStoreCompactor.Action.COMPACT) {
115        c = maybeCloneWithAllocator(c); // for compaction copy cell to the new segment (MSLAB copy)
116      }
117      offsetInCurentChunk = // add the Cell reference to the index chunk
118          createCellReference((ByteBufferKeyValue)c, chunks[currentChunkIdx].getData(),
119              offsetInCurentChunk);
120      // the sizes still need to be updated in the new segment
121      // second parameter true, because in compaction/merge the addition of the cell to new segment
122      // is always successful
123      updateMetaInfo(c, true, null); // updates the size per cell
124    }
125    // build the immutable CellSet
126    CellChunkMap ccm =
127        new CellChunkMap(CellComparator.COMPARATOR,chunks,0,numOfCellsAfterCompaction,false);
128    this.setCellSet(null, new CellSet(ccm));  // update the CellSet of this Segment
129  }
130
131  /*------------------------------------------------------------------------*/
132  // Create CellSet based on CellChunkMap from current ConcurrentSkipListMap based CellSet
133  // (without compacting iterator)
134  // This is a service for not-flat immutable segments
135  // Assumption: cells do not exceed chunk size!
136  private void reinitializeCellSet(
137      int numOfCells, KeyValueScanner segmentScanner, CellSet oldCellSet) {
138    Cell curCell;
139    // calculate how many chunks we will need for metadata
140    int chunkSize = ChunkCreator.getInstance().getChunkSize();
141    int numOfCellsInChunk = CellChunkMap.NUM_OF_CELL_REPS_IN_CHUNK;
142    int numberOfChunks = calculateNumberOfChunks(numOfCells,numOfCellsInChunk);
143    // all index Chunks are allocated from ChunkCreator
144    Chunk[] chunks = new Chunk[numberOfChunks];
145    for (int i=0; i < numberOfChunks; i++) {
146      chunks[i] = this.getMemStoreLAB().getNewExternalChunk();
147    }
148
149    int currentChunkIdx = 0;
150    int offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
151
152    try {
153      while ((curCell = segmentScanner.next()) != null) {
154        assert (curCell instanceof ByteBufferKeyValue); // shouldn't get here anything but ByteBufferKeyValue
155        if (offsetInCurentChunk + ClassSize.CELL_CHUNK_MAP_ENTRY > chunkSize) {
156          // continue to the next metadata chunk
157          currentChunkIdx++;
158          offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
159        }
160        offsetInCurentChunk =
161            createCellReference((ByteBufferKeyValue) curCell, chunks[currentChunkIdx].getData(),
162                offsetInCurentChunk);
163      }
164    } catch (IOException ie) {
165      throw new IllegalStateException(ie);
166    } finally {
167      segmentScanner.close();
168    }
169
170    CellChunkMap ccm = new CellChunkMap(CellComparator.COMPARATOR,chunks,0,numOfCells,false);
171    this.setCellSet(oldCellSet, new CellSet(ccm)); // update the CellSet of this Segment
172  }
173
174  /*------------------------------------------------------------------------*/
175  // for a given cell, write the cell representation on the index chunk
176  private int createCellReference(ByteBufferKeyValue cell, ByteBuffer idxBuffer, int idxOffset) {
177    int offset = idxOffset;
178    int dataChunkID = cell.getChunkId();
179
180    offset = ByteBufferUtils.putInt(idxBuffer, offset, dataChunkID);    // write data chunk id
181    offset = ByteBufferUtils.putInt(idxBuffer, offset, cell.getOffset());          // offset
182    offset = ByteBufferUtils.putInt(idxBuffer, offset, KeyValueUtil.length(cell)); // length
183    offset = ByteBufferUtils.putLong(idxBuffer, offset, cell.getSequenceId());     // seqId
184
185    return offset;
186  }
187
188  private int calculateNumberOfChunks(int numOfCells, int numOfCellsInChunk) {
189    int numberOfChunks = numOfCells/numOfCellsInChunk;
190    if(numOfCells%numOfCellsInChunk!=0) { // if cells cannot be divided evenly between chunks
191      numberOfChunks++;                   // add one additional chunk
192    }
193    return numberOfChunks;
194  }
