035/**
036 * CellChunkImmutableSegment extends the API supported by a {@link Segment},
037 * and {@link ImmutableSegment}. This immutable segment is working with CellSet with
038 * CellChunkMap delegatee.
039 */
040@InterfaceAudience.Private
041public class CellChunkImmutableSegment extends ImmutableSegment {
042
043  public static final long DEEP_OVERHEAD_CCM =
044      ImmutableSegment.DEEP_OVERHEAD + ClassSize.CELL_CHUNK_MAP;
045  public static final float INDEX_CHUNK_UNUSED_SPACE_PRECENTAGE = 0.1f;
046
047  /////////////////////  CONSTRUCTORS  /////////////////////
048  /**------------------------------------------------------------------------
049   * C-tor to be used when new CellChunkImmutableSegment is built as a result of compaction/merge
050   * of a list of older ImmutableSegments.
051   * The given iterator returns the Cells that "survived" the compaction.
052   */
053  protected CellChunkImmutableSegment(CellComparator comparator, MemStoreSegmentsIterator iterator,
054      MemStoreLAB memStoreLAB, int numOfCells, MemStoreCompactionStrategy.Action action) {
055    super(null, comparator, memStoreLAB); // initialize the CellSet with NULL
056    long indexOverhead = DEEP_OVERHEAD_CCM;
057    // memStoreLAB cannot be null in this class
058    boolean onHeap = getMemStoreLAB().isOnHeap();
059    // initiate the heapSize with the size of the segment metadata
060    if(onHeap) {
061      incMemStoreSize(0, indexOverhead, 0);
062    } else {
063      incMemStoreSize(0, 0, indexOverhead);
064    }
065    // build the new CellSet based on CellArrayMap and update the CellSet of the new Segment
066    initializeCellSet(numOfCells, iterator, action);
067  }
068
069  /**------------------------------------------------------------------------
070   * C-tor to be used when new CellChunkImmutableSegment is built as a result of flattening
071   * of CSLMImmutableSegment
072   * The given iterator returns the Cells that "survived" the compaction.
073   */
074  protected CellChunkImmutableSegment(CSLMImmutableSegment segment,
075      MemStoreSizing memstoreSizing, MemStoreCompactionStrategy.Action action) {
076    super(segment); // initiailize the upper class
077    long indexOverhead = -CSLMImmutableSegment.DEEP_OVERHEAD_CSLM + DEEP_OVERHEAD_CCM;
078    // memStoreLAB cannot be null in this class
079    boolean onHeap = getMemStoreLAB().isOnHeap();
080    // initiate the heapSize with the size of the segment metadata
081    if(onHeap) {
082      incMemStoreSize(0, indexOverhead, 0);
083      memstoreSizing.incMemStoreSize(0, indexOverhead, 0);
084    } else {
085      incMemStoreSize(0, -CSLMImmutableSegment.DEEP_OVERHEAD_CSLM, DEEP_OVERHEAD_CCM);
086      memstoreSizing.incMemStoreSize(0, -CSLMImmutableSegment.DEEP_OVERHEAD_CSLM,
087          DEEP_OVERHEAD_CCM);
088    }
089    int numOfCells = segment.getCellsCount();
090    // build the new CellSet based on CellChunkMap
091    reinitializeCellSet(numOfCells, segment.getScanner(Long.MAX_VALUE), segment.getCellSet(),
092        memstoreSizing, action);
093    // arrange the meta-data size, decrease all meta-data sizes related to SkipList;
094    // add sizes of CellChunkMap entry, decrease also Cell object sizes
095    // (reinitializeCellSet doesn't take the care for the sizes)
096    long newSegmentSizeDelta = numOfCells*(indexEntrySize()-ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
097    if(onHeap) {
098      incMemStoreSize(0, newSegmentSizeDelta, 0);
099      memstoreSizing.incMemStoreSize(0, newSegmentSizeDelta, 0);
100    } else {
101      incMemStoreSize(0, 0, newSegmentSizeDelta);
102      memstoreSizing.incMemStoreSize(0, 0, newSegmentSizeDelta);
103
104    }
105  }
106
107  @Override
108  protected long indexEntryOnHeapSize(boolean onHeap) {
109    if(onHeap) {
110      return indexEntrySize();
111    }
112    // else the index is allocated off-heap
113    return 0;
114  }
115
116  @Override
117  protected long indexEntryOffHeapSize(boolean offHeap) {
118    if(offHeap) {
119      return indexEntrySize();
120    }
121    // else the index is allocated on-heap
122    return 0;
123  }
124
125  @Override
126  protected long indexEntrySize() {
127    return ((long) ClassSize.CELL_CHUNK_MAP_ENTRY - KeyValue.FIXED_OVERHEAD);
128  }
129
130  @Override
131  protected boolean canBeFlattened() {
132    return false;
133  }

136  /*------------------------------------------------------------------------*/
137  // Create CellSet based on CellChunkMap from compacting iterator
138  private void initializeCellSet(int numOfCells, MemStoreSegmentsIterator iterator,
139      MemStoreCompactionStrategy.Action action) {
140
141    int numOfCellsAfterCompaction = 0;
142    int currentChunkIdx = 0;
143    int offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
144    int numUniqueKeys=0;
145    Cell prev = null;
146    Chunk[] chunks = allocIndexChunks(numOfCells);
147    while (iterator.hasNext()) {        // the iterator hides the elimination logic for compaction
148      boolean alreadyCopied = false;
149      Cell c = iterator.next();
150      numOfCellsAfterCompaction++;
151      assert(c instanceof ExtendedCell);
152      if (((ExtendedCell)c).getChunkId() == ExtendedCell.CELL_NOT_BASED_ON_CHUNK) {
153        // CellChunkMap assumes all cells are allocated on MSLAB.
154        // Therefore, cells which are not allocated on MSLAB initially,
155        // are copied into MSLAB here.
156        c = copyCellIntoMSLAB(c, null); //no memstore sizing object to update
157        alreadyCopied = true;
158      }
159      if (offsetInCurentChunk + ClassSize.CELL_CHUNK_MAP_ENTRY > chunks[currentChunkIdx].size) {
160        currentChunkIdx++;              // continue to the next index chunk
161        offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
162      }
163      if (action == MemStoreCompactionStrategy.Action.COMPACT && !alreadyCopied) {
164        // for compaction copy cell to the new segment (MSLAB copy)
165        c = maybeCloneWithAllocator(c, false);
166      }
167      offsetInCurentChunk = // add the Cell reference to the index chunk
168          createCellReference((ByteBufferKeyValue)c, chunks[currentChunkIdx].getData(),
169              offsetInCurentChunk);
170      // the sizes still need to be updated in the new segment
171      // second parameter true, because in compaction/merge the addition of the cell to new segment
172      // is always successful
173      updateMetaInfo(c, true, null); // updates the size per cell
174      if(action == MemStoreCompactionStrategy.Action.MERGE_COUNT_UNIQUE_KEYS) {
175        //counting number of unique keys
176        if (prev != null) {
177          if (!CellUtil.matchingRowColumnBytes(prev, c)) {
178            numUniqueKeys++;
179          }
180        } else {
181          numUniqueKeys++;
182        }
183      }
184      prev = c;
185    }
186    if(action == MemStoreCompactionStrategy.Action.COMPACT) {
187      numUniqueKeys = numOfCells;
188    } else if(action != MemStoreCompactionStrategy.Action.MERGE_COUNT_UNIQUE_KEYS) {
189      numUniqueKeys = CellSet.UNKNOWN_NUM_UNIQUES;
190    }
191    // build the immutable CellSet
192    CellChunkMap ccm =
193        new CellChunkMap(getComparator(), chunks, 0, numOfCellsAfterCompaction, false);
194    this.setCellSet(null, new CellSet(ccm, numUniqueKeys));  // update the CellSet of this Segment
195  }

197  /*------------------------------------------------------------------------*/
198  // Create CellSet based on CellChunkMap from current ConcurrentSkipListMap based CellSet
199  // (without compacting iterator)
200  // This is a service for not-flat immutable segments
201  private void reinitializeCellSet(
202      int numOfCells, KeyValueScanner segmentScanner, CellSet oldCellSet,
203      MemStoreSizing memstoreSizing, MemStoreCompactionStrategy.Action action) {
204    Cell curCell;
205    Chunk[] chunks = allocIndexChunks(numOfCells);
206
207    int currentChunkIdx = 0;
208    int offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
209
210    int numUniqueKeys=0;
211    Cell prev = null;
212    try {
213      while ((curCell = segmentScanner.next()) != null) {
214        assert(curCell instanceof ExtendedCell);
215        if (((ExtendedCell)curCell).getChunkId() == ExtendedCell.CELL_NOT_BASED_ON_CHUNK) {
216          // CellChunkMap assumes all cells are allocated on MSLAB.
217          // Therefore, cells which are not allocated on MSLAB initially,
218          // are copied into MSLAB here.
219          curCell = copyCellIntoMSLAB(curCell, memstoreSizing);
220        }
221        if (offsetInCurentChunk + ClassSize.CELL_CHUNK_MAP_ENTRY > chunks[currentChunkIdx].size) {
222          // continue to the next metadata chunk
223          currentChunkIdx++;
224          offsetInCurentChunk = ChunkCreator.SIZEOF_CHUNK_HEADER;
225        }
226        offsetInCurentChunk =
227            createCellReference((ByteBufferKeyValue) curCell, chunks[currentChunkIdx].getData(),
228                offsetInCurentChunk);
229        if(action == MemStoreCompactionStrategy.Action.FLATTEN_COUNT_UNIQUE_KEYS) {
230          //counting number of unique keys
231          if (prev != null) {
232            if (!CellUtil.matchingRowColumn(prev, curCell)) {
233              numUniqueKeys++;
234            }
235          } else {
236            numUniqueKeys++;
237          }
238        }
239        prev = curCell;
240      }
241      if(action != MemStoreCompactionStrategy.Action.FLATTEN_COUNT_UNIQUE_KEYS) {
242        numUniqueKeys = CellSet.UNKNOWN_NUM_UNIQUES;
243      }
244    } catch (IOException ie) {
245      throw new IllegalStateException(ie);
246    } finally {
247      segmentScanner.close();
248    }
249
250    CellChunkMap ccm = new CellChunkMap(getComparator(), chunks, 0, numOfCells, false);
251    // update the CellSet of this Segment
252    this.setCellSet(oldCellSet, new CellSet(ccm, numUniqueKeys));
253  }

256  // for a given cell, write the cell representation on the index chunk
257  private int createCellReference(ByteBufferKeyValue cell, ByteBuffer idxBuffer, int idxOffset) {
258    int offset = idxOffset;
259    int dataChunkID = cell.getChunkId();
260
261    offset = ByteBufferUtils.putInt(idxBuffer, offset, dataChunkID);    // write data chunk id
262    offset = ByteBufferUtils.putInt(idxBuffer, offset, cell.getOffset());          // offset
263    offset = ByteBufferUtils.putInt(idxBuffer, offset, KeyValueUtil.length(cell)); // length
264    offset = ByteBufferUtils.putLong(idxBuffer, offset, cell.getSequenceId());     // seqId
265
266    return offset;
267  }

269  private int calculateNumberOfChunks(int numOfCells, int chunkSize) {
270    int numOfCellsInChunk = calcNumOfCellsInChunk(chunkSize);
271    int numberOfChunks = numOfCells / numOfCellsInChunk;
272    if(numOfCells % numOfCellsInChunk != 0) { // if cells cannot be divided evenly between chunks
273      numberOfChunks++;                   // add one additional chunk
274    }
275    return numberOfChunks;
276  }
277
278  // Assuming we are going to use regular data chunks as index chunks,
279  // we check here how much free space will remain in the last allocated chunk
280  // (the least occupied one).
281  // If the percentage of its remaining free space is above the INDEX_CHUNK_UNUSED_SPACE
282  // threshold, then we will use index chunks (which are smaller) instead.
283  private ChunkCreator.ChunkType useIndexChunks(int numOfCells) {
284    int dataChunkSize = ChunkCreator.getInstance().getChunkSize();
285    int numOfCellsInChunk = calcNumOfCellsInChunk(dataChunkSize);
286    int cellsInLastChunk = numOfCells % numOfCellsInChunk;
287    if (cellsInLastChunk == 0) { // There is no free space in the last chunk and thus,
288      return ChunkCreator.ChunkType.DATA_CHUNK;               // no need to use index chunks.
289    } else {
290      int chunkSpace = dataChunkSize - ChunkCreator.SIZEOF_CHUNK_HEADER;
291      int freeSpaceInLastChunk = chunkSpace - cellsInLastChunk * ClassSize.CELL_CHUNK_MAP_ENTRY;
292      if (freeSpaceInLastChunk > INDEX_CHUNK_UNUSED_SPACE_PRECENTAGE * chunkSpace) {
293        return ChunkCreator.ChunkType.INDEX_CHUNK;
294      }
295      return ChunkCreator.ChunkType.DATA_CHUNK;
296    }
297  }
298
299  private int calcNumOfCellsInChunk(int chunkSize) {
300    int chunkSpace = chunkSize - ChunkCreator.SIZEOF_CHUNK_HEADER;
301    int numOfCellsInChunk = chunkSpace / ClassSize.CELL_CHUNK_MAP_ENTRY;
302    return numOfCellsInChunk;
303  }
304
305  private Chunk[] allocIndexChunks(int numOfCells) {
306    // Decide whether to use regular or small chunks and then
307    // calculate how many chunks we will need for index
308
309    ChunkCreator.ChunkType chunkType = useIndexChunks(numOfCells);
310    int chunkSize = ChunkCreator.getInstance().getChunkSize(chunkType);
311    int numberOfChunks = calculateNumberOfChunks(numOfCells, chunkSize);
312    // all index Chunks are allocated from ChunkCreator
313    Chunk[] chunks = new Chunk[numberOfChunks];
314    // all index Chunks are allocated from ChunkCreator
315    for (int i = 0; i < numberOfChunks; i++) {
316      chunks[i] = this.getMemStoreLAB().getNewExternalChunk(chunkType);
317    }
318    return chunks;
319  }

321  private Cell copyCellIntoMSLAB(Cell cell, MemStoreSizing memstoreSizing) {
322    // Take care for a special case when a cell is copied from on-heap to (probably off-heap) MSLAB.
323    // The cell allocated as an on-heap JVM object (byte array) occupies slightly different
324    // amount of memory, than when the cell serialized and allocated on the MSLAB.
325    // Here, we update the heap size of the new segment only for the difference between object and
326    // serialized size. This is a decrease of the size as serialized cell is a bit smaller.
327    // The actual size of the cell is not added yet, and will be added (only in compaction)
328    // in initializeCellSet#updateMetaInfo().
329    long oldHeapSize = heapSizeChange(cell, true);
330    long oldOffHeapSize = offHeapSizeChange(cell, true);
331    long oldCellSize = getCellLength(cell);
332    cell = maybeCloneWithAllocator(cell, true);
333    long newHeapSize = heapSizeChange(cell, true);
334    long newOffHeapSize = offHeapSizeChange(cell, true);
335    long newCellSize = getCellLength(cell);
336    long heapOverhead = newHeapSize - oldHeapSize;
337    long offHeapOverhead = newOffHeapSize - oldOffHeapSize;
338    incMemStoreSize(newCellSize - oldCellSize, heapOverhead, offHeapOverhead);
339    if(memstoreSizing != null) {
340      memstoreSizing.incMemStoreSize(newCellSize - oldCellSize, heapOverhead, offHeapOverhead);
341    }
342    return cell;
343  }
