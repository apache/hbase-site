397      this.shipper.shipped();

402          regionServer.getLeaseManager().addLease(lease);
403        }
404      }

465  /**
466   * Instantiated as a scanner lease. If the lease times out, the scanner is
467   * closed
468   */
469  private class ScannerListener implements LeaseListener {
470    private final String scannerName;
471
472    ScannerListener(final String n) {
473      this.scannerName = n;
474    }
475
476    @Override
477    public void leaseExpired() {
478      RegionScannerHolder rsh = scanners.remove(this.scannerName);
479      if (rsh != null) {
480        RegionScanner s = rsh.s;
481        LOG.info("Scanner " + this.scannerName + " lease expired on region "
482          + s.getRegionInfo().getRegionNameAsString());
483        HRegion region = null;
484        try {
485          region = regionServer.getRegion(s.getRegionInfo().getRegionName());
486          if (region != null && region.getCoprocessorHost() != null) {
487            region.getCoprocessorHost().preScannerClose(s);
488          }
489        } catch (IOException e) {
490          LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
491        } finally {
492          try {
493            s.close();
494            if (region != null && region.getCoprocessorHost() != null) {
495              region.getCoprocessorHost().postScannerClose(s);
496            }
497          } catch (IOException e) {
498            LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
499          }
500        }
501      } else {
502        LOG.warn("Scanner " + this.scannerName + " lease expired, but no related" +
503          " scanner found, hence no chance to close that related scanner!");
504      }
505    }
506  }

608  /**
609   * Mutate a list of rows atomically.
610   * @param cellScanner if non-null, the mutation data -- the Cell content.
611   */
612  private boolean checkAndRowMutate(final HRegion region, final List<ClientProtos.Action> actions,
613    final CellScanner cellScanner, byte[] row, byte[] family, byte[] qualifier,
614    CompareOperator op, ByteArrayComparable comparator, Filter filter, TimeRange timeRange,
615    RegionActionResult.Builder builder, ActivePolicyEnforcement spaceQuotaEnforcement)
616    throws IOException {
617    int countOfCompleteMutation = 0;
618    try {
619      if (!region.getRegionInfo().isMetaRegion()) {
620        regionServer.getMemStoreFlusher().reclaimMemStoreMemory();
621      }
622      RowMutations rm = null;
623      int i = 0;
624      ClientProtos.ResultOrException.Builder resultOrExceptionOrBuilder =
625        ClientProtos.ResultOrException.newBuilder();
626      for (ClientProtos.Action action: actions) {
627        if (action.hasGet()) {
628          throw new DoNotRetryIOException("Atomic put and/or delete only, not a Get=" +
629            action.getGet());
630        }
631        MutationType type = action.getMutation().getMutateType();
632        if (rm == null) {
633          rm = new RowMutations(action.getMutation().getRow().toByteArray(), actions.size());
634        }
635        switch (type) {
636          case PUT:
637            Put put = ProtobufUtil.toPut(action.getMutation(), cellScanner);
638            ++countOfCompleteMutation;
639            checkCellSizeLimit(region, put);
640            spaceQuotaEnforcement.getPolicyEnforcement(region).check(put);
641            rm.add(put);
642            break;
643          case DELETE:
644            Delete del = ProtobufUtil.toDelete(action.getMutation(), cellScanner);
645            ++countOfCompleteMutation;
646            spaceQuotaEnforcement.getPolicyEnforcement(region).check(del);
647            rm.add(del);
648            break;
649          default:
650            throw new DoNotRetryIOException("Atomic put and/or delete only, not " + type.name());
651        }
652        // To unify the response format with doNonAtomicRegionMutation and read through client's
653        // AsyncProcess we have to add an empty result instance per operation
654        resultOrExceptionOrBuilder.clear();
655        resultOrExceptionOrBuilder.setIndex(i++);
656        builder.addResultOrException(
657          resultOrExceptionOrBuilder.build());
658      }
659
660      if (filter != null) {
661        return region.checkAndRowMutate(row, filter, timeRange, rm);
662      } else {
663        return region.checkAndRowMutate(row, family, qualifier, op, comparator, timeRange, rm);
664      }
665    } finally {
666      // Currently, the checkAndMutate isn't supported by batch so it won't mess up the cell scanner
667      // even if the malformed cells are not skipped.
668      for (int i = countOfCompleteMutation; i < actions.size(); ++i) {
669        skipCellsForMutation(actions.get(i), cellScanner);
670      }
671    }
672  }

674  /**
675   * Execute an append mutation.
676   *
677   * @return result to return to client if default operation should be
678   * bypassed as indicated by RegionObserver, null otherwise
679   */
680  private Result append(final HRegion region, final OperationQuota quota,
681      final MutationProto mutation, final CellScanner cellScanner, long nonceGroup,
682      ActivePolicyEnforcement spaceQuota)
683      throws IOException {
684    long before = EnvironmentEdgeManager.currentTime();
685    Append append = ProtobufUtil.toAppend(mutation, cellScanner);
686    checkCellSizeLimit(region, append);
687    spaceQuota.getPolicyEnforcement(region).check(append);
688    quota.addMutation(append);
689    Result r = null;
690    if (region.getCoprocessorHost() != null) {
691      r = region.getCoprocessorHost().preAppend(append);
692    }
693    if (r == null) {
694      boolean canProceed = startNonceOperation(mutation, nonceGroup);
695      boolean success = false;
696      try {
697        long nonce = mutation.hasNonce() ? mutation.getNonce() : HConstants.NO_NONCE;
698        if (canProceed) {
699          r = region.append(append, nonceGroup, nonce);
700        } else {
701          // convert duplicate append to get
702          List<Cell> results = region.get(ProtobufUtil.toGet(mutation, cellScanner), false,
703              nonceGroup, nonce);
704          r = Result.create(results);
705        }
706        success = true;
707      } finally {
708        if (canProceed) {
709          endNonceOperation(mutation, nonceGroup, success);
710        }
711      }
712      if (region.getCoprocessorHost() != null) {
713        r = region.getCoprocessorHost().postAppend(append, r);
714      }
715    }
716    if (regionServer.getMetrics() != null) {
717      regionServer.getMetrics().updateAppend(
718          region.getTableDescriptor().getTableName(),
719        EnvironmentEdgeManager.currentTime() - before);
720    }
721    return r == null ? Result.EMPTY_RESULT : r;
722  }

728      final MutationProto mutation, final CellScanner cells, long nonceGroup,
729      ActivePolicyEnforcement spaceQuota)

772  /**
773   * Run through the regionMutation <code>rm</code> and per Mutation, do the work, and then when
774   * done, add an instance of a {@link ResultOrException} that corresponds to each Mutation.
775   * @param cellsToReturn  Could be null. May be allocated in this method.  This is what this
776   * method returns as a 'result'.
777   * @param closeCallBack the callback to be used with multigets
778   * @param context the current RpcCallContext
779   * @return Return the <code>cellScanner</code> passed
780   */
781  private List<CellScannable> doNonAtomicRegionMutation(final HRegion region,
782      final OperationQuota quota, final RegionAction actions, final CellScanner cellScanner,
783      final RegionActionResult.Builder builder, List<CellScannable> cellsToReturn, long nonceGroup,
784      final RegionScannersCloseCallBack closeCallBack, RpcCallContext context,
785      ActivePolicyEnforcement spaceQuotaEnforcement) {
786    // Gather up CONTIGUOUS Puts and Deletes in this mutations List.  Idea is that rather than do
787    // one at a time, we instead pass them in batch.  Be aware that the corresponding
788    // ResultOrException instance that matches each Put or Delete is then added down in the
789    // doNonAtomicBatchOp call.  We should be staying aligned though the Put and Delete are
790    // deferred/batched
791    List<ClientProtos.Action> mutations = null;
792    long maxQuotaResultSize = Math.min(maxScannerResultSize, quota.getReadAvailable());
793    IOException sizeIOE = null;
794    Object lastBlock = null;
795    ClientProtos.ResultOrException.Builder resultOrExceptionBuilder = ResultOrException.newBuilder();
796    boolean hasResultOrException = false;
797    for (ClientProtos.Action action : actions.getActionList()) {
798      hasResultOrException = false;
799      resultOrExceptionBuilder.clear();
800      try {
801        Result r = null;
802
803        if (context != null
804            && context.isRetryImmediatelySupported()
805            && (context.getResponseCellSize() > maxQuotaResultSize
806              || context.getResponseBlockSize() + context.getResponseExceptionSize()
807              > maxQuotaResultSize)) {
808
809          // We're storing the exception since the exception and reason string won't
810          // change after the response size limit is reached.
811          if (sizeIOE == null ) {
812            // We don't need the stack un-winding do don't throw the exception.
813            // Throwing will kill the JVM's JIT.
814            //
815            // Instead just create the exception and then store it.
816            sizeIOE = new MultiActionResultTooLarge("Max size exceeded"
817                + " CellSize: " + context.getResponseCellSize()
818                + " BlockSize: " + context.getResponseBlockSize());
819
820            // Only report the exception once since there's only one request that
821            // caused the exception. Otherwise this number will dominate the exceptions count.
822            rpcServer.getMetrics().exception(sizeIOE);
823          }
824
825          // Now that there's an exception is known to be created
826          // use it for the response.
827          //
828          // This will create a copy in the builder.
829          NameBytesPair pair = ResponseConverter.buildException(sizeIOE);
830          resultOrExceptionBuilder.setException(pair);
831          context.incrementResponseExceptionSize(pair.getSerializedSize());
832          resultOrExceptionBuilder.setIndex(action.getIndex());
833          builder.addResultOrException(resultOrExceptionBuilder.build());
834          skipCellsForMutation(action, cellScanner);
835          continue;
836        }
837        if (action.hasGet()) {
838          long before = EnvironmentEdgeManager.currentTime();
839          ClientProtos.Get pbGet = action.getGet();
840          // An asynchbase client, https://github.com/OpenTSDB/asynchbase, starts by trying to do
841          // a get closest before. Throwing the UnknownProtocolException signals it that it needs
842          // to switch and do hbase2 protocol (HBase servers do not tell clients what versions
843          // they are; its a problem for non-native clients like asynchbase. HBASE-20225.
844          if (pbGet.hasClosestRowBefore() && pbGet.getClosestRowBefore()) {
845            throw new UnknownProtocolException("Is this a pre-hbase-1.0.0 or asynchbase client? " +
846                "Client is invoking getClosestRowBefore removed in hbase-2.0.0 replaced by " +
847                "reverse Scan.");
848          }
849          try {
850            Get get = ProtobufUtil.toGet(pbGet);
851            if (context != null) {
852              r = get(get, (region), closeCallBack, context);
853            } else {
854              r = region.get(get);
855            }
856          } finally {
857            final MetricsRegionServer metricsRegionServer = regionServer.getMetrics();
858            if (metricsRegionServer != null) {
859              metricsRegionServer.updateGet(
860                  region.getTableDescriptor().getTableName(),
861                  EnvironmentEdgeManager.currentTime() - before);
862            }
863          }
864        } else if (action.hasServiceCall()) {
865          hasResultOrException = true;
866          com.google.protobuf.Message result =
867            execServiceOnRegion(region, action.getServiceCall());
868          ClientProtos.CoprocessorServiceResult.Builder serviceResultBuilder =
869            ClientProtos.CoprocessorServiceResult.newBuilder();
870          resultOrExceptionBuilder.setServiceResult(
871            serviceResultBuilder.setValue(
872              serviceResultBuilder.getValueBuilder()
873                .setName(result.getClass().getName())
874                // TODO: Copy!!!
875                .setValue(UnsafeByteOperations.unsafeWrap(result.toByteArray()))));
876        } else if (action.hasMutation()) {
877          MutationType type = action.getMutation().getMutateType();
878          if (type != MutationType.PUT && type != MutationType.DELETE && mutations != null &&
879              !mutations.isEmpty()) {
880            // Flush out any Puts or Deletes already collected.
881            doNonAtomicBatchOp(builder, region, quota, mutations, cellScanner,
882              spaceQuotaEnforcement);
883            mutations.clear();
884          }
885          switch (type) {
886            case APPEND:
887              r = append(region, quota, action.getMutation(), cellScanner, nonceGroup,
888                  spaceQuotaEnforcement);
889              break;
890            case INCREMENT:
891              r = increment(region, quota, action.getMutation(), cellScanner, nonceGroup,
892                  spaceQuotaEnforcement);
893              break;
894            case PUT:
895            case DELETE:
896              // Collect the individual mutations and apply in a batch
897              if (mutations == null) {
898                mutations = new ArrayList<>(actions.getActionCount());
899              }
900              mutations.add(action);
901              break;
902            default:
903              throw new DoNotRetryIOException("Unsupported mutate type: " + type.name());
904          }
905        } else {
906          throw new HBaseIOException("Unexpected Action type");
907        }
908        if (r != null) {
909          ClientProtos.Result pbResult = null;
910          if (isClientCellBlockSupport(context)) {
911            pbResult = ProtobufUtil.toResultNoData(r);
912            //  Hard to guess the size here.  Just make a rough guess.
913            if (cellsToReturn == null) {
914              cellsToReturn = new ArrayList<>();
915            }
916            cellsToReturn.add(r);
917          } else {
918            pbResult = ProtobufUtil.toResult(r);
919          }
920          lastBlock = addSize(context, r, lastBlock);
921          hasResultOrException = true;
922          resultOrExceptionBuilder.setResult(pbResult);
923        }
924        // Could get to here and there was no result and no exception.  Presumes we added
925        // a Put or Delete to the collecting Mutations List for adding later.  In this
926        // case the corresponding ResultOrException instance for the Put or Delete will be added
927        // down in the doNonAtomicBatchOp method call rather than up here.
928      } catch (IOException ie) {
929        rpcServer.getMetrics().exception(ie);
930        hasResultOrException = true;
931        NameBytesPair pair = ResponseConverter.buildException(ie);
932        resultOrExceptionBuilder.setException(pair);
933        context.incrementResponseExceptionSize(pair.getSerializedSize());
934      }
935      if (hasResultOrException) {
936        // Propagate index.
937        resultOrExceptionBuilder.setIndex(action.getIndex());
938        builder.addResultOrException(resultOrExceptionBuilder.build());
939      }
940    }
941    // Finish up any outstanding mutations
942    if (!CollectionUtils.isEmpty(mutations)) {
943      doNonAtomicBatchOp(builder, region, quota, mutations, cellScanner, spaceQuotaEnforcement);
944    }
945    return cellsToReturn;
946  }

948  private void checkCellSizeLimit(final HRegion r, final Mutation m) throws IOException {
949    if (r.maxCellSize > 0) {
950      CellScanner cells = m.cellScanner();
951      while (cells.advance()) {
952        int size = PrivateCellUtil.estimatedSerializedSizeOf(cells.current());
953        if (size > r.maxCellSize) {
954          String msg = "Cell with size " + size + " exceeds limit of " + r.maxCellSize + " bytes";
955          if (LOG.isDebugEnabled()) {
956            LOG.debug(msg);
957          }
958          throw new DoNotRetryIOException(msg);
959        }
960      }
961    }

991  /**
992   * Execute a list of Put/Delete mutations.
993   *
994   * @param builder
995   * @param region
996   * @param mutations
997   */
998  private void doBatchOp(final RegionActionResult.Builder builder, final HRegion region,
999      final OperationQuota quota, final List<ClientProtos.Action> mutations,
1000      final CellScanner cells, ActivePolicyEnforcement spaceQuotaEnforcement, boolean atomic)
1001      throws IOException {
1002    Mutation[] mArray = new Mutation[mutations.size()];
1003    long before = EnvironmentEdgeManager.currentTime();
1004    boolean batchContainsPuts = false, batchContainsDelete = false;
1005    try {
1006      /** HBASE-17924
1007       * mutationActionMap is a map to map the relation between mutations and actions
1008       * since mutation array may have been reoredered.In order to return the right
1009       * result or exception to the corresponding actions, We need to know which action
1010       * is the mutation belong to. We can't sort ClientProtos.Action array, since they
1011       * are bonded to cellscanners.
1012       */
1013      Map<Mutation, ClientProtos.Action> mutationActionMap = new HashMap<>();
1014      int i = 0;
1015      for (ClientProtos.Action action: mutations) {
1016        if (action.hasGet()) {
1017          throw new DoNotRetryIOException("Atomic put and/or delete only, not a Get=" +
1018            action.getGet());
1019        }
1020        MutationProto m = action.getMutation();
1021        Mutation mutation;
1022        if (m.getMutateType() == MutationType.PUT) {
1023          mutation = ProtobufUtil.toPut(m, cells);
1024          batchContainsPuts = true;
1025        } else {
1026          mutation = ProtobufUtil.toDelete(m, cells);
1027          batchContainsDelete = true;
1028        }
1029        mutationActionMap.put(mutation, action);
1030        mArray[i++] = mutation;
1031        checkCellSizeLimit(region, mutation);
1032        // Check if a space quota disallows this mutation
1033        spaceQuotaEnforcement.getPolicyEnforcement(region).check(mutation);
1034        quota.addMutation(mutation);
1035      }
1036
1037      if (!region.getRegionInfo().isMetaRegion()) {
1038        regionServer.getMemStoreFlusher().reclaimMemStoreMemory();
1039      }
1040
1041      // HBASE-17924
1042      // Sort to improve lock efficiency for non-atomic batch of operations. If atomic
1043      // order is preserved as its expected from the client
1044      if (!atomic) {
1045        Arrays.sort(mArray, (v1, v2) -> Row.COMPARATOR.compare(v1, v2));
1046      }
1047
1048      OperationStatus[] codes = region.batchMutate(mArray, atomic, HConstants.NO_NONCE,
1049        HConstants.NO_NONCE);
1050      for (i = 0; i < codes.length; i++) {
1051        Mutation currentMutation = mArray[i];
1052        ClientProtos.Action currentAction = mutationActionMap.get(currentMutation);
1053        int index = currentAction.hasIndex() || !atomic ? currentAction.getIndex() : i;
1054        Exception e = null;
1055        switch (codes[i].getOperationStatusCode()) {
1056          case BAD_FAMILY:
1057            e = new NoSuchColumnFamilyException(codes[i].getExceptionMsg());
1058            builder.addResultOrException(getResultOrException(e, index));
1059            break;
1060
1061          case SANITY_CHECK_FAILURE:
1062            e = new FailedSanityCheckException(codes[i].getExceptionMsg());
1063            builder.addResultOrException(getResultOrException(e, index));
1064            break;
1065
1066          default:
1067            e = new DoNotRetryIOException(codes[i].getExceptionMsg());
1068            builder.addResultOrException(getResultOrException(e, index));
1069            break;
1070
1071          case SUCCESS:
1072            builder.addResultOrException(getResultOrException(
1073              ClientProtos.Result.getDefaultInstance(), index));
1074            break;
1075
1076          case STORE_TOO_BUSY:
1077            e = new RegionTooBusyException(codes[i].getExceptionMsg());
1078            builder.addResultOrException(getResultOrException(e, index));
1079            break;
1080        }
1081      }
1082    } finally {
1083      int processedMutationIndex = 0;
1084      for (Action mutation : mutations) {
1085        // The non-null mArray[i] means the cell scanner has been read.
1086        if (mArray[processedMutationIndex++] == null) {
1087          skipCellsForMutation(mutation, cells);
1088        }
1089      }
1090      updateMutationMetrics(region, before, batchContainsPuts, batchContainsDelete);
1091    }
1092  }

1110  /**
1111   * Execute a list of Put/Delete mutations. The function returns OperationStatus instead of
1112   * constructing MultiResponse to save a possible loop if caller doesn't need MultiResponse.
1113   * @param region
1114   * @param mutations
1115   * @param replaySeqId
1116   * @return an array of OperationStatus which internally contains the OperationStatusCode and the
1117   *         exceptionMessage if any
1118   * @throws IOException
1119   */
1120  private OperationStatus [] doReplayBatchOp(final HRegion region,
1121      final List<MutationReplay> mutations, long replaySeqId) throws IOException {
1122    long before = EnvironmentEdgeManager.currentTime();
1123    boolean batchContainsPuts = false, batchContainsDelete = false;
1124    try {
1125      for (Iterator<MutationReplay> it = mutations.iterator(); it.hasNext();) {
1126        MutationReplay m = it.next();
1127
1128        if (m.getType() == MutationType.PUT) {
1129          batchContainsPuts = true;
1130        } else {
1131          batchContainsDelete = true;
1132        }
1133
1134        NavigableMap<byte[], List<Cell>> map = m.mutation.getFamilyCellMap();
1135        List<Cell> metaCells = map.get(WALEdit.METAFAMILY);
1136        if (metaCells != null && !metaCells.isEmpty()) {
1137          for (Cell metaCell : metaCells) {
1138            CompactionDescriptor compactionDesc = WALEdit.getCompaction(metaCell);
1139            boolean isDefaultReplica = RegionReplicaUtil.isDefaultReplica(region.getRegionInfo());
1140            HRegion hRegion = region;
1141            if (compactionDesc != null) {
1142              // replay the compaction. Remove the files from stores only if we are the primary
1143              // region replica (thus own the files)
1144              hRegion.replayWALCompactionMarker(compactionDesc, !isDefaultReplica, isDefaultReplica,
1145                replaySeqId);
1146              continue;
1147            }
1148            FlushDescriptor flushDesc = WALEdit.getFlushDescriptor(metaCell);
1149            if (flushDesc != null && !isDefaultReplica) {
1150              hRegion.replayWALFlushMarker(flushDesc, replaySeqId);
1151              continue;
1152            }
1153            RegionEventDescriptor regionEvent = WALEdit.getRegionEventDescriptor(metaCell);
1154            if (regionEvent != null && !isDefaultReplica) {
1155              hRegion.replayWALRegionEventMarker(regionEvent);
1156              continue;
1157            }
1158            BulkLoadDescriptor bulkLoadEvent = WALEdit.getBulkLoadDescriptor(metaCell);
1159            if (bulkLoadEvent != null) {
1160              hRegion.replayWALBulkLoadEventMarker(bulkLoadEvent);
1161              continue;
1162            }
1163          }
1164          it.remove();
1165        }
1166      }
1167      requestCount.increment();
1168      if (!region.getRegionInfo().isMetaRegion()) {
1169        regionServer.getMemStoreFlusher().reclaimMemStoreMemory();
1170      }
1171      return region.batchReplay(mutations.toArray(
1172        new MutationReplay[mutations.size()]), replaySeqId);
1173    } finally {
1174      updateMutationMetrics(region, before, batchContainsPuts, batchContainsDelete);
1175    }
1176  }

1366  /**
1367   * Method to account for the size of retained cells and retained data blocks.
1368   * @param context rpc call context
1369   * @param r result to add size.
1370   * @param lastBlock last block to check whether we need to add the block size in context.
1371   * @return an object that represents the last referenced block from this response.
1372   */
1373  Object addSize(RpcCallContext context, Result r, Object lastBlock) {
1374    if (context != null && r != null && !r.isEmpty()) {
1375      for (Cell c : r.rawCells()) {
1376        context.incrementResponseCellSize(PrivateCellUtil.estimatedSerializedSizeOf(c));
1377
1378        // Since byte buffers can point all kinds of crazy places it's harder to keep track
1379        // of which blocks are kept alive by what byte buffer.
1380        // So we make a guess.
1381        if (c instanceof ByteBufferExtendedCell) {
1382          ByteBufferExtendedCell bbCell = (ByteBufferExtendedCell) c;
1383          ByteBuffer bb = bbCell.getValueByteBuffer();
1384          if (bb != lastBlock) {
1385            context.incrementResponseBlockSize(bb.capacity());
1386            lastBlock = bb;
1387          }
1388        } else {
1389          // We're using the last block being the same as the current block as
1390          // a proxy for pointing to a new block. This won't be exact.
1391          // If there are multiple gets that bounce back and forth
1392          // Then it's possible that this will over count the size of
1393          // referenced blocks. However it's better to over count and
1394          // use two rpcs than to OOME the regionserver.
1395          byte[] valueArray = c.getValueArray();
1396          if (valueArray != lastBlock) {
1397            context.incrementResponseBlockSize(valueArray.length);
1398            lastBlock = valueArray;
1399          }
1400        }
1401
1402      }
1403    }
1404    return lastBlock;
1405  }

3207  private long getTimeLimit(HBaseRpcController controller, boolean allowHeartbeatMessages) {
3208    // Set the time limit to be half of the more restrictive timeout value (one of the
3209    // timeout values must be positive). In the event that both values are positive, the
3210    // more restrictive of the two is used to calculate the limit.
3211    if (allowHeartbeatMessages && (scannerLeaseTimeoutPeriod > 0 || rpcTimeout > 0)) {
3212      long timeLimitDelta;
3213      if (scannerLeaseTimeoutPeriod > 0 && rpcTimeout > 0) {
3214        timeLimitDelta = Math.min(scannerLeaseTimeoutPeriod, rpcTimeout);
3215      } else {
3216        timeLimitDelta = scannerLeaseTimeoutPeriod > 0 ? scannerLeaseTimeoutPeriod : rpcTimeout;
3217      }
3218      if (controller != null && controller.getCallTimeout() > 0) {
3219        timeLimitDelta = Math.min(timeLimitDelta, controller.getCallTimeout());
3220      }
3221      // Use half of whichever timeout value was more restrictive... But don't allow
3222      // the time limit to be less than the allowable minimum (could cause an
3223      // immediatate timeout before scanning any data).
3224      timeLimitDelta = Math.max(timeLimitDelta / 2, minimumScanTimeLimitDelta);
3225      // XXX: Can not use EnvironmentEdge here because TestIncrementTimeRange use a
3226      // ManualEnvironmentEdge. Consider using System.nanoTime instead.
3227      return System.currentTimeMillis() + timeLimitDelta;
3228    }
3229    // Default value of timeLimit is negative to indicate no timeLimit should be
3230    // enforced.
3231    return -1L;

3375          limitReached = sizeLimitReached || timeLimitReached || resultsLimitReached;
3376
3377          if (limitReached || !moreRows) {
3378            // We only want to mark a ScanResponse as a heartbeat message in the event that
3379            // there are more values to be read server side. If there aren't more values,
3380            // marking it as a heartbeat is wasteful because the client will need to issue
3381            // another ScanRequest only to realize that they already have all the values
3382            if (moreRows && timeLimitReached) {
3383              // Heartbeat messages occur when the time limit has been reached.
3384              builder.setHeartbeatMessage(true);
3385              if (rsh.needCursor) {
3386                Cell cursorCell = scannerContext.getLastPeekedCell();
3387                if (cursorCell != null) {
3388                  builder.setCursor(ProtobufUtil.toCursor(cursorCell));
3389                }
3390              }
3391            }
3392            break;
3393          }
3394          values.clear();
3395        }
3396        builder.setMoreResultsInRegion(moreRows);
3397        // Check to see if the client requested that we track metrics server side. If the
3398        // client requested metrics, retrieve the metrics from the scanner context.
3399        if (trackMetrics) {
3400          Map<String, Long> metrics = scannerContext.getMetrics().getMetricsMap();
3401          ScanMetrics.Builder metricBuilder = ScanMetrics.newBuilder();
3402          NameInt64Pair.Builder pairBuilder = NameInt64Pair.newBuilder();
3403
3404          for (Entry<String, Long> entry : metrics.entrySet()) {
3405            pairBuilder.setName(entry.getKey());
3406            pairBuilder.setValue(entry.getValue());
3407            metricBuilder.addMetrics(pairBuilder.build());
3408          }
3409
3410          builder.setScanMetrics(metricBuilder.build());
3411        }
3412      }
3413      long end = EnvironmentEdgeManager.currentTime();
3414      long responseCellSize = context != null ? context.getResponseCellSize() : 0;
3415      region.getMetrics().updateScanTime(end - before);
3416      final MetricsRegionServer metricsRegionServer = regionServer.getMetrics();
3417      if (metricsRegionServer != null) {
3418        metricsRegionServer.updateScanSize(
3419            region.getTableDescriptor().getTableName(), responseCellSize);
3420        metricsRegionServer.updateScanTime(
3421            region.getTableDescriptor().getTableName(), end - before);
3422      }
3423    } finally {
3424      region.closeRegionOperation();
3425    }
3426    // coprocessor postNext hook
3427    if (region.getCoprocessorHost() != null) {
3428      region.getCoprocessorHost().postScannerNext(scanner, results, maxResults, true);
3429    }
3430  }

3432  /**
3433   * Scan data in a table.
3434   *
3435   * @param controller the RPC controller
3436   * @param request the scan request
3437   * @throws ServiceException
3438   */
3439  @Override
3440  public ScanResponse scan(final RpcController controller, final ScanRequest request)
3441      throws ServiceException {
3442    if (controller != null && !(controller instanceof HBaseRpcController)) {
3443      throw new UnsupportedOperationException(
3444          "We only do " + "HBaseRpcControllers! FIX IF A PROBLEM: " + controller);
3445    }
3446    if (!request.hasScannerId() && !request.hasScan()) {
3447      throw new ServiceException(
3448          new DoNotRetryIOException("Missing required input: scannerId or scan"));
3449    }
3450    try {
3451      checkOpen();
3452    } catch (IOException e) {
3453      if (request.hasScannerId()) {
3454        String scannerName = Long.toString(request.getScannerId());
3455        if (LOG.isDebugEnabled()) {
3456          LOG.debug(
3457            "Server shutting down and client tried to access missing scanner " + scannerName);
3458        }
3459        final LeaseManager leaseManager = regionServer.getLeaseManager();
3460        if (leaseManager != null) {
3461          try {
3462            leaseManager.cancelLease(scannerName);
3463          } catch (LeaseException le) {
3464            // No problem, ignore
3465            if (LOG.isTraceEnabled()) {
3466              LOG.trace("Un-able to cancel lease of scanner. It could already be closed.");
3467            }
3468          }
3469        }
3470      }
3471      throw new ServiceException(e);
3472    }
3473    requestCount.increment();
3474    rpcScanRequestCount.increment();
3475    RegionScannerHolder rsh;
3476    ScanResponse.Builder builder = ScanResponse.newBuilder();
3477    try {
3478      if (request.hasScannerId()) {
3479        // The downstream projects such as AsyncHBase in OpenTSDB need this value. See HBASE-18000
3480        // for more details.
3481        builder.setScannerId(request.getScannerId());
3482        rsh = getRegionScanner(request);
3483      } else {
3484        rsh = newRegionScanner(request, builder);
3485      }
3486    } catch (IOException e) {
3487      if (e == SCANNER_ALREADY_CLOSED) {
3488        // Now we will close scanner automatically if there are no more results for this region but
3489        // the old client will still send a close request to us. Just ignore it and return.
3490        return builder.build();
3491      }
3492      throw new ServiceException(e);
3493    }
3494    HRegion region = rsh.r;
3495    String scannerName = rsh.scannerName;
3496    LeaseManager.Lease lease;
3497    try {
3498      // Remove lease while its being processed in server; protects against case
3499      // where processing of request takes > lease expiration time.
3500      lease = regionServer.getLeaseManager().removeLease(scannerName);
3501    } catch (LeaseException e) {
3502      throw new ServiceException(e);
3503    }
3504    if (request.hasRenew() && request.getRenew()) {
3505      // add back and return
3506      addScannerLeaseBack(lease);
3507      try {
3508        checkScanNextCallSeq(request, rsh);
3509      } catch (OutOfOrderScannerNextException e) {
3510        throw new ServiceException(e);
3511      }
3512      return builder.build();
3513    }
3514    OperationQuota quota;
3515    try {
3516      quota = getRpcQuotaManager().checkQuota(region, OperationQuota.OperationType.SCAN);
3517    } catch (IOException e) {
3518      addScannerLeaseBack(lease);
3519      throw new ServiceException(e);
3520    }
3521    try {
3522      checkScanNextCallSeq(request, rsh);
3523    } catch (OutOfOrderScannerNextException e) {
3524      addScannerLeaseBack(lease);
3525      throw new ServiceException(e);
3526    }
3527    // Now we have increased the next call sequence. If we give client an error, the retry will
3528    // never success. So we'd better close the scanner and return a DoNotRetryIOException to client
3529    // and then client will try to open a new scanner.
3530    boolean closeScanner = request.hasCloseScanner() ? request.getCloseScanner() : false;
3531    int rows; // this is scan.getCaching
3532    if (request.hasNumberOfRows()) {
3533      rows = request.getNumberOfRows();
3534    } else {
3535      rows = closeScanner ? 0 : 1;
3536    }
3537    RpcCallContext context = RpcServer.getCurrentCall().orElse(null);
3538    // now let's do the real scan.
3539    long maxQuotaResultSize = Math.min(maxScannerResultSize, quota.getReadAvailable());
3540    RegionScanner scanner = rsh.s;
3541    // this is the limit of rows for this scan, if we the number of rows reach this value, we will
3542    // close the scanner.
3543    int limitOfRows;
3544    if (request.hasLimitOfRows()) {
3545      limitOfRows = request.getLimitOfRows();
3546    } else {
3547      limitOfRows = -1;
3548    }
3549    MutableObject<Object> lastBlock = new MutableObject<>();
3550    boolean scannerClosed = false;
3551    try {
3552      List<Result> results = new ArrayList<>(Math.min(rows, 512));
3553      if (rows > 0) {
3554        boolean done = false;
3555        // Call coprocessor. Get region info from scanner.
3556        if (region.getCoprocessorHost() != null) {
3557          Boolean bypass = region.getCoprocessorHost().preScannerNext(scanner, results, rows);
3558          if (!results.isEmpty()) {
3559            for (Result r : results) {
3560              lastBlock.setValue(addSize(context, r, lastBlock.getValue()));
3561            }
3562          }
3563          if (bypass != null && bypass.booleanValue()) {
3564            done = true;
3565          }
3566        }
3567        if (!done) {
3568          scan((HBaseRpcController) controller, request, rsh, maxQuotaResultSize, rows, limitOfRows,
3569            results, builder, lastBlock, context);
3570        } else {
3571          builder.setMoreResultsInRegion(!results.isEmpty());
3572        }
3573      } else {
3574        // This is a open scanner call with numberOfRow = 0, so set more results in region to true.
3575        builder.setMoreResultsInRegion(true);
3576      }
3577
3578      quota.addScanResult(results);
3579      addResults(builder, results, (HBaseRpcController) controller,
3580        RegionReplicaUtil.isDefaultReplica(region.getRegionInfo()),
3581        isClientCellBlockSupport(context));
3582      if (scanner.isFilterDone() && results.isEmpty()) {
3583        // If the scanner's filter - if any - is done with the scan
3584        // only set moreResults to false if the results is empty. This is used to keep compatible
3585        // with the old scan implementation where we just ignore the returned results if moreResults
3586        // is false. Can remove the isEmpty check after we get rid of the old implementation.
3587        builder.setMoreResults(false);
3588      }
3589      // Later we may close the scanner depending on this flag so here we need to make sure that we
3590      // have already set this flag.
3591      assert builder.hasMoreResultsInRegion();
3592      // we only set moreResults to false in the above code, so set it to true if we haven't set it
3593      // yet.
3594      if (!builder.hasMoreResults()) {
3595        builder.setMoreResults(true);
3596      }
3597      if (builder.getMoreResults() && builder.getMoreResultsInRegion() && !results.isEmpty()) {
3598        // Record the last cell of the last result if it is a partial result
3599        // We need this to calculate the complete rows we have returned to client as the
3600        // mayHaveMoreCellsInRow is true does not mean that there will be extra cells for the
3601        // current row. We may filter out all the remaining cells for the current row and just
3602        // return the cells of the nextRow when calling RegionScanner.nextRaw. So here we need to
3603        // check for row change.
3604        Result lastResult = results.get(results.size() - 1);
3605        if (lastResult.mayHaveMoreCellsInRow()) {
3606          rsh.rowOfLastPartialResult = lastResult.getRow();
3607        } else {
3608          rsh.rowOfLastPartialResult = null;
3609        }
3610      }
3611      if (!builder.getMoreResults() || !builder.getMoreResultsInRegion() || closeScanner) {
3612        scannerClosed = true;
3613        closeScanner(region, scanner, scannerName, context);
3614      }
3615      return builder.build();
3616    } catch (IOException e) {
3617      try {
3618        // scanner is closed here
3619        scannerClosed = true;
3620        // The scanner state might be left in a dirty state, so we will tell the Client to
3621        // fail this RPC and close the scanner while opening up another one from the start of
3622        // row that the client has last seen.
3623        closeScanner(region, scanner, scannerName, context);
3624
3625        // If it is a DoNotRetryIOException already, throw as it is. Unfortunately, DNRIOE is
3626        // used in two different semantics.
3627        // (1) The first is to close the client scanner and bubble up the exception all the way
3628        // to the application. This is preferred when the exception is really un-recoverable
3629        // (like CorruptHFileException, etc). Plain DoNotRetryIOException also falls into this
3630        // bucket usually.
3631        // (2) Second semantics is to close the current region scanner only, but continue the
3632        // client scanner by overriding the exception. This is usually UnknownScannerException,
3633        // OutOfOrderScannerNextException, etc where the region scanner has to be closed, but the
3634        // application-level ClientScanner has to continue without bubbling up the exception to
3635        // the client. See ClientScanner code to see how it deals with these special exceptions.
3636        if (e instanceof DoNotRetryIOException) {
3637          throw e;
3638        }
3639
3640        // If it is a FileNotFoundException, wrap as a
3641        // DoNotRetryIOException. This can avoid the retry in ClientScanner.
3642        if (e instanceof FileNotFoundException) {
3643          throw new DoNotRetryIOException(e);
3644        }
3645
3646        // We closed the scanner already. Instead of throwing the IOException, and client
3647        // retrying with the same scannerId only to get USE on the next RPC, we directly throw
3648        // a special exception to save an RPC.
3649        if (VersionInfoUtil.hasMinimumVersion(context.getClientVersionInfo(), 1, 4)) {
3650          // 1.4.0+ clients know how to handle
3651          throw new ScannerResetException("Scanner is closed on the server-side", e);
3652        } else {
3653          // older clients do not know about SRE. Just throw USE, which they will handle
3654          throw new UnknownScannerException("Throwing UnknownScannerException to reset the client"
3655              + " scanner state for clients older than 1.3.", e);
3656        }
3657      } catch (IOException ioe) {
3658        throw new ServiceException(ioe);
3659      }
3660    } finally {
3661      if (!scannerClosed) {
3662        // Adding resets expiration time on lease.
3663        // the closeCallBack will be set in closeScanner so here we only care about shippedCallback
3664        if (context != null) {
3665          context.setCallBack(rsh.shippedCallback);
3666        } else {
3667          // When context != null, adding back the lease will be done in callback set above.
3668          addScannerLeaseBack(lease);
3669        }
3670      }
3671      quota.close();
3672    }
3673  }

3675  private void closeScanner(HRegion region, RegionScanner scanner, String scannerName,
3676      RpcCallContext context) throws IOException {
3677    if (region.getCoprocessorHost() != null) {
3678      if (region.getCoprocessorHost().preScannerClose(scanner)) {
3679        // bypass the actual close.
3680        return;
3681      }
3682    }
3683    RegionScannerHolder rsh = scanners.remove(scannerName);
3684    if (rsh != null) {
3685      if (context != null) {
3686        context.setCallBack(rsh.closeCallBack);
3687      } else {
3688        rsh.s.close();
3689      }
3690      if (region.getCoprocessorHost() != null) {
3691        region.getCoprocessorHost().postScannerClose(scanner);
3692      }
3693      closedScanners.put(scannerName, scannerName);
3694    }
3695  }
