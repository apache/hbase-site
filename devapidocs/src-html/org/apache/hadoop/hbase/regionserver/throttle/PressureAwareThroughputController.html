018package org.apache.hadoop.hbase.regionserver.throttle;
019
020import java.util.concurrent.ConcurrentHashMap;
021import java.util.concurrent.ConcurrentMap;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.conf.Configured;
026import org.apache.hadoop.hbase.HBaseInterfaceAudience;
027import org.apache.hadoop.hbase.Stoppable;
028import org.apache.hadoop.hbase.classification.InterfaceAudience;
029import org.apache.hadoop.hbase.regionserver.RegionServerServices;
030import org.apache.hadoop.hbase.regionserver.compactions.OffPeakHours;
031import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
032
033@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
034public abstract class PressureAwareThroughputController extends Configured implements
035    ThroughputController, Stoppable {
036  private static final Log LOG = LogFactory.getLog(PressureAwareThroughputController.class);
037
038  /**
039   * Stores the information of one controlled compaction.
040   */
041  private static final class ActiveOperation {
042
043    private final long startTime;
044
045    private long lastControlTime;
046
047    private long lastControlSize;
048
049    private long totalSize;
050
051    private long numberOfSleeps;
052
053    private long totalSleepTime;
054
055    // prevent too many debug log
056    private long lastLogTime;
057
058    ActiveOperation() {
059      long currentTime = EnvironmentEdgeManager.currentTime();
060      this.startTime = currentTime;
061      this.lastControlTime = currentTime;
062      this.lastLogTime = currentTime;
063    }
064  }
065
066  protected long maxThroughputUpperBound;
067
068  protected long maxThroughputLowerBound;
069
070  protected OffPeakHours offPeakHours;
071
072  protected long controlPerSize;
073
074  protected int tuningPeriod;
075
076  private volatile double maxThroughput;
077
078  protected final ConcurrentMap<String, ActiveOperation> activeOperations =
079      new ConcurrentHashMap<String, ActiveOperation>();
080
081  @Override
082  public abstract void setup(final RegionServerServices server);
083
084  protected String throughputDesc(long deltaSize, long elapsedTime) {
085    return throughputDesc((double) deltaSize / elapsedTime * 1000);
086  }
087
088  protected String throughputDesc(double speed) {
089    if (speed >= 1E15) { // large enough to say it is unlimited
090      return "unlimited";
091    } else {
092      return String.format("%.2f MB/sec", speed / 1024 / 1024);
093    }
094  }
095
096  @Override
097  public void start(String opName) {
098    activeOperations.put(opName, new ActiveOperation());
099  }
100
101  @Override
102  public long control(String opName, long size) throws InterruptedException {
103    ActiveOperation operation = activeOperations.get(opName);
104    operation.totalSize += size;
105    long deltaSize = operation.totalSize - operation.lastControlSize;
106    if (deltaSize < controlPerSize) {
107      return 0;
108    }
109    long now = EnvironmentEdgeManager.currentTime();
110    double maxThroughputPerCompaction = this.getMaxThroughput() / activeOperations.size();
111    long minTimeAllowed = (long) (deltaSize / maxThroughputPerCompaction * 1000); // ms
112    long elapsedTime = now - operation.lastControlTime;
113    operation.lastControlSize = operation.totalSize;
114    if (elapsedTime >= minTimeAllowed) {
115      operation.lastControlTime = EnvironmentEdgeManager.currentTime();
116      return 0;
117    }
118    // too fast
119    long sleepTime = minTimeAllowed - elapsedTime;
120    if (LOG.isDebugEnabled()) {
121      // do not log too much
122      if (now - operation.lastLogTime > 5L * 1000) {
123        LOG.debug("deltaSize: " + deltaSize + " bytes; elapseTime: " + elapsedTime + " ns");
124        LOG.debug(opName + " sleep " + sleepTime + " ms because current throughput is "
125            + throughputDesc(deltaSize, elapsedTime) + ", max allowed is "
126            + throughputDesc(maxThroughputPerCompaction) + ", already slept "
127            + operation.numberOfSleeps + " time(s) and total slept time is "
128            + operation.totalSleepTime + " ms till now.");
129        operation.lastLogTime = now;
130      }
131    }
132    Thread.sleep(sleepTime);
133    operation.numberOfSleeps++;
134    operation.totalSleepTime += sleepTime;
135    operation.lastControlTime = EnvironmentEdgeManager.currentTime();
136    return sleepTime;
137  }
138
139  /**
140   * Check whether to skip control given delta size and control size
141   * @param deltaSize Delta size since last control
142   * @param controlSize Size limit to perform control
143   * @return a boolean indicates whether to skip this control
144   */
145  protected abstract boolean skipControl(long deltaSize, long controlSize);
146
147  @Override
148  public void finish(String opName) {
149    ActiveOperation operation = activeOperations.remove(opName);
150    long elapsedTime = EnvironmentEdgeManager.currentTime() - operation.startTime;
151    LOG.info(opName + " average throughput is "
152        + throughputDesc(operation.totalSize, elapsedTime) + ", slept "
153        + operation.numberOfSleeps + " time(s) and total slept time is "
154        + operation.totalSleepTime + " ms. " + activeOperations.size()
155        + " active operations remaining, total limit is " + throughputDesc(getMaxThroughput()));
156  }
157
158  private volatile boolean stopped = false;
159
160  @Override
161  public void stop(String why) {
162    stopped = true;
163  }
164
165  @Override
166  public boolean isStopped() {
167    return stopped;
168  }
169
170  public double getMaxThroughput() {
171    return maxThroughput;
172  }
173
174  public void setMaxThroughput(double maxThroughput) {
175    this.maxThroughput = maxThroughput;
176  }
177}
