042/**
043 * An abstract class, which implements the behaviour shared by all concrete memstore instances.
044 */
045@InterfaceAudience.Private
046public abstract class AbstractMemStore implements MemStore {
047
048  private static final long NO_SNAPSHOT_ID = -1;
049
050  private final Configuration conf;
051  private final CellComparator comparator;
052
053  // active segment absorbs write operations
054  private volatile MutableSegment active;
055  // Snapshot of memstore.  Made for flusher.
056  private volatile ImmutableSegment snapshot;
057  protected volatile long snapshotId;
058  // Used to track when to flush
059  private volatile long timeOfOldestEdit;
060
061  public final static long FIXED_OVERHEAD = ClassSize.align(
062      ClassSize.OBJECT +
063          (4 * ClassSize.REFERENCE) +
064          (2 * Bytes.SIZEOF_LONG));
065
066  public final static long DEEP_OVERHEAD = ClassSize.align(FIXED_OVERHEAD +
067      2 * (ClassSize.ATOMIC_LONG + ClassSize.TIMERANGE_TRACKER +
068      ClassSize.CELL_SKIPLIST_SET + ClassSize.CONCURRENT_SKIPLISTMAP));
069
070
071  protected AbstractMemStore(final Configuration conf, final CellComparator c) {
072    this.conf = conf;
073    this.comparator = c;
074    resetCellSet();
075    this.snapshot = SegmentFactory.instance().createImmutableSegment(conf, c, 0);
076    this.snapshotId = NO_SNAPSHOT_ID;
077  }
078
079  protected void resetCellSet() {
080    // Reset heap to not include any keys
081    this.active = SegmentFactory.instance().createMutableSegment(conf, comparator, DEEP_OVERHEAD);
082    this.timeOfOldestEdit = Long.MAX_VALUE;
083  }
084
085  /*
086  * Calculate how the MemStore size has changed.  Includes overhead of the
087  * backing Map.
088  * @param cell
089  * @param notPresent True if the cell was NOT present in the set.
090  * @return change in size
091  */
092  static long heapSizeChange(final Cell cell, final boolean notPresent) {
093    return notPresent ? ClassSize.align(ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY
094        + CellUtil.estimatedHeapSizeOf(cell)) : 0;
095  }

097  /**
098   * Updates the wal with the lowest sequence id (oldest entry) that is still in memory
099   * @param onlyIfMoreRecent a flag that marks whether to update the sequence id no matter what or
100   *                      only if it is greater than the previous sequence id
101   */
102  public abstract void updateLowestUnflushedSequenceIdInWal(boolean onlyIfMoreRecent);

104  /**
105   * Write an update
106   * @param cell the cell to be added
107   * @return approximate size of the passed cell & newly added cell which maybe different than the
108   *         passed-in cell
109   */
110  @Override
111  public long add(Cell cell) {
112    Cell toAdd = maybeCloneWithAllocator(cell);
113    return internalAdd(toAdd);
114  }
115
116  /**
117   * Update or insert the specified Cells.
118   * <p>
119   * For each Cell, insert into MemStore.  This will atomically upsert the
120   * value for that row/family/qualifier.  If a Cell did already exist,
121   * it will then be removed.
122   * <p>
123   * Currently the memstoreTS is kept at 0 so as each insert happens, it will
124   * be immediately visible.  May want to change this so it is atomic across
125   * all Cells.
126   * <p>
127   * This is called under row lock, so Get operations will still see updates
128   * atomically.  Scans will only see each Cell update as atomic.
129   *
130   * @param cells the cells to be updated
131   * @param readpoint readpoint below which we can safely remove duplicate KVs
132   * @return change in memstore size
133   */
134  @Override
135  public long upsert(Iterable<Cell> cells, long readpoint) {
136    long size = 0;
137    for (Cell cell : cells) {
138      size += upsert(cell, readpoint);
139    }
140    return size;
141  }

143  /**
144   * @return Oldest timestamp of all the Cells in the MemStore
145   */
146  @Override
147  public long timeOfOldestEdit() {
148    return timeOfOldestEdit;
149  }

152  /**
153   * Write a delete
154   * @param deleteCell the cell to be deleted
155   * @return approximate size of the passed key and value.
156   */
157  @Override
158  public long delete(Cell deleteCell) {
159    Cell toAdd = maybeCloneWithAllocator(deleteCell);
160    long s = internalAdd(toAdd);
161    return s;
162  }

164  /**
165   * An override on snapshot so the no arg version of the method implies zero seq num,
166   * like for cases without wal
167   */
168  public MemStoreSnapshot snapshot() {
169    return snapshot(0);
170  }
171
172  /**
173   * The passed snapshot was successfully persisted; it can be let go.
174   * @param id Id of the snapshot to clean out.
175   * @see MemStore#snapshot(long)
176   */
177  @Override
178  public void clearSnapshot(long id) throws UnexpectedStateException {
179    if (this.snapshotId != id) {
180      throw new UnexpectedStateException("Current snapshot id is " + this.snapshotId + ",passed "
181          + id);
182    }
183    // OK. Passed in snapshot is same as current snapshot. If not-empty,
184    // create a new snapshot and let the old one go.
185    Segment oldSnapshot = this.snapshot;
186    if (!this.snapshot.isEmpty()) {
187      this.snapshot = SegmentFactory.instance().createImmutableSegment(
188          getComparator(), 0);
189    }
190    this.snapshotId = NO_SNAPSHOT_ID;
191    oldSnapshot.close();
192  }

204   * On flush, how much memory we will clear from the active cell set.

228  @Override
229  public String toString() {
230    StringBuffer buf = new StringBuffer();
231    int i = 1;
232    try {
233      for (Segment segment : getListOfSegments()) {
234        buf.append("Segment (" + i + ") " + segment.toString() + "; ");

239    }
240    return buf.toString();

253  /**
254   * Inserts the specified Cell into MemStore and deletes any existing
255   * versions of the same row/family/qualifier as the specified Cell.
256   * <p>
257   * First, the specified Cell is inserted into the Memstore.
258   * <p>
259   * If there are any existing Cell in this MemStore with the same row,
260   * family, and qualifier, they are removed.
261   * <p>
262   * Callers must hold the read lock.
263   *
264   * @param cell the cell to be updated
265   * @param readpoint readpoint below which we can safely remove duplicate KVs
266   * @return change in size of MemStore
267   */
268  private long upsert(Cell cell, long readpoint) {
269    // Add the Cell to the MemStore
270    // Use the internalAdd method here since we (a) already have a lock
271    // and (b) cannot safely use the MSLAB here without potentially
272    // hitting OOME - see TestMemStore.testUpsertMSLAB for a
273    // test that triggers the pathological case if we don't avoid MSLAB
274    // here.
275    long addedSize = internalAdd(cell);
276
277    // Get the Cells for the row/family/qualifier regardless of timestamp.
278    // For this case we want to clean up any other puts
279    Cell firstCell = KeyValueUtil.createFirstOnRow(
280        cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
281        cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
282        cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength());
283    SortedSet<Cell> ss = active.tailSet(firstCell);
284    Iterator<Cell> it = ss.iterator();
285    // versions visible to oldest scanner
286    int versionsVisible = 0;
287    while (it.hasNext()) {
288      Cell cur = it.next();
289
290      if (cell == cur) {
291        // ignore the one just put in
292        continue;
293      }
294      // check that this is the row and column we are interested in, otherwise bail
295      if (CellUtil.matchingRow(cell, cur) && CellUtil.matchingQualifier(cell, cur)) {
296        // only remove Puts that concurrent scanners cannot possibly see
297        if (cur.getTypeByte() == KeyValue.Type.Put.getCode() &&
298            cur.getSequenceId() <= readpoint) {
299          if (versionsVisible >= 1) {
300            // if we get here we have seen at least one version visible to the oldest scanner,
301            // which means we can prove that no scanner will see this version
302
303            // false means there was a change, so give us the size.
304            long delta = heapSizeChange(cur, true);
305            addedSize -= delta;
306            active.incSize(-delta);
307            it.remove();
308            setOldestEditTimeToNow();
309          } else {
310            versionsVisible++;
311          }
312        }
313      } else {
314        // past the row or column, done
315        break;
316      }
317    }
318    return addedSize;
319  }

321  /*
322   * @param a
323   * @param b
324   * @return Return lowest of a or b or null if both a and b are null
325   */
326  protected Cell getLowest(final Cell a, final Cell b) {
327    if (a == null) {
328      return b;
329    }
330    if (b == null) {
331      return a;
332    }
333    return comparator.compareRows(a, b) <= 0? a: b;
334  }

336  /*
337   * @param key Find row that follows this one.  If null, return first.
338   * @param set Set to look in for a row beyond <code>row</code>.
339   * @return Next row or null if none found.  If one found, will be a new
340   * KeyValue -- can be destroyed by subsequent calls to this method.
341   */
342  protected Cell getNextRow(final Cell key,
343      final NavigableSet<Cell> set) {
344    Cell result = null;
345    SortedSet<Cell> tail = key == null? set: set.tailSet(key);
346    // Iterate until we fall into the next row; i.e. move off current row
347    for (Cell cell: tail) {
348      if (comparator.compareRows(cell, key) <= 0) {
349        continue;
350      }
351      // Note: Not suppressing deletes or expired cells.  Needs to be handled
352      // by higher up functions.
353      result = cell;
354      break;
355    }
356    return result;
357  }

359  /**
360   * Given the specs of a column, update it, first by inserting a new record,
361   * then removing the old one.  Since there is only 1 KeyValue involved, the memstoreTS
362   * will be set to 0, thus ensuring that they instantly appear to anyone. The underlying
363   * store will ensure that the insert/delete each are atomic. A scanner/reader will either
364   * get the new value, or the old value and all readers will eventually only see the new
365   * value after the old was removed.
366   */
367  @VisibleForTesting
368  @Override
369  public long updateColumnValue(byte[] row, byte[] family, byte[] qualifier,
370      long newValue, long now) {
371    Cell firstCell = KeyValueUtil.createFirstOnRow(row, family, qualifier);
372    // Is there a Cell in 'snapshot' with the same TS? If so, upgrade the timestamp a bit.
373    Cell snc = snapshot.getFirstAfter(firstCell);
374    if(snc != null) {
375      // is there a matching Cell in the snapshot?
376      if (CellUtil.matchingRow(snc, firstCell) && CellUtil.matchingQualifier(snc, firstCell)) {
377        if (snc.getTimestamp() == now) {
378          now += 1;
379        }
380      }
381    }
382    // logic here: the new ts MUST be at least 'now'. But it could be larger if necessary.
383    // But the timestamp should also be max(now, mostRecentTsInMemstore)
384
385    // so we cant add the new Cell w/o knowing what's there already, but we also
386    // want to take this chance to delete some cells. So two loops (sad)
387
388    SortedSet<Cell> ss = getActive().tailSet(firstCell);
389    for (Cell cell : ss) {
390      // if this isnt the row we are interested in, then bail:
391      if (!CellUtil.matchingColumn(cell, family, qualifier)
392          || !CellUtil.matchingRow(cell, firstCell)) {
393        break; // rows dont match, bail.
394      }
395
396      // if the qualifier matches and it's a put, just RM it out of the active.
397      if (cell.getTypeByte() == KeyValue.Type.Put.getCode() &&
398          cell.getTimestamp() > now && CellUtil.matchingQualifier(firstCell, cell)) {
399        now = cell.getTimestamp();
400      }
401    }
402
403    // create or update (upsert) a new Cell with
404    // 'now' and a 0 memstoreTS == immediately visible
405    List<Cell> cells = new ArrayList<Cell>(1);
406    cells.add(new KeyValue(row, family, qualifier, now, Bytes.toBytes(newValue)));
407    return upsert(cells, 1L);
408  }

410  private Cell maybeCloneWithAllocator(Cell cell) {
411    return active.maybeCloneWithAllocator(cell);
412  }

414  /**
415   * Internal version of add() that doesn't clone Cells with the
416   * allocator, and doesn't take the lock.
417   *
418   * Callers should ensure they already have the read lock taken
419   */
420  private long internalAdd(final Cell toAdd) {
421    long s = active.add(toAdd);
422    setOldestEditTimeToNow();
423    checkActiveSize();
424    return s;
425  }
426
427  private void setOldestEditTimeToNow() {
428    if (timeOfOldestEdit == Long.MAX_VALUE) {
429      timeOfOldestEdit = EnvironmentEdgeManager.currentTime();
430    }
431  }
