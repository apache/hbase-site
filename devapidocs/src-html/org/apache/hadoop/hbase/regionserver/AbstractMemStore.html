022import java.io.IOException;
023import java.util.List;
024import java.util.NavigableSet;
025import java.util.SortedSet;
026
027import org.apache.commons.logging.Log;
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.hbase.Cell;
030import org.apache.hadoop.hbase.CellComparator;
031import org.apache.hadoop.hbase.ExtendedCell;
032import org.apache.hadoop.hbase.classification.InterfaceAudience;
033import org.apache.hadoop.hbase.exceptions.UnexpectedStateException;

038/**
039 * An abstract class, which implements the behaviour shared by all concrete memstore instances.
040 */
041@InterfaceAudience.Private
042public abstract class AbstractMemStore implements MemStore {
043
044  private static final long NO_SNAPSHOT_ID = -1;
045
046  private final Configuration conf;
047  private final CellComparator comparator;
048
049  // active segment absorbs write operations
050  protected volatile MutableSegment active;
051  // Snapshot of memstore.  Made for flusher.
052  protected volatile ImmutableSegment snapshot;
053  protected volatile long snapshotId;
054  // Used to track when to flush
055  private volatile long timeOfOldestEdit;
056
057  public final static long FIXED_OVERHEAD = ClassSize
058      .align(ClassSize.OBJECT + (4 * ClassSize.REFERENCE) + (2 * Bytes.SIZEOF_LONG));
059
060  public final static long DEEP_OVERHEAD = FIXED_OVERHEAD;
061
062  protected AbstractMemStore(final Configuration conf, final CellComparator c) {
063    this.conf = conf;
064    this.comparator = c;
065    resetActive();
066    this.snapshot = SegmentFactory.instance().createImmutableSegment(c);
067    this.snapshotId = NO_SNAPSHOT_ID;
068  }
069
070  protected void resetActive() {
071    // Reset heap to not include any keys
072    this.active = SegmentFactory.instance().createMutableSegment(conf, comparator);
073    this.timeOfOldestEdit = Long.MAX_VALUE;
074  }
075
076  /**
077   * Updates the wal with the lowest sequence id (oldest entry) that is still in memory
078   * @param onlyIfMoreRecent a flag that marks whether to update the sequence id no matter what or
079   *                      only if it is greater than the previous sequence id
080   */
081  public abstract void updateLowestUnflushedSequenceIdInWAL(boolean onlyIfMoreRecent);
082
083  @Override
084  public void add(Iterable<Cell> cells, MemstoreSize memstoreSize) {
085    for (Cell cell : cells) {
086      add(cell, memstoreSize);
087    }
088  }
089
090  @Override
091  public void add(Cell cell, MemstoreSize memstoreSize) {
092    Cell toAdd = maybeCloneWithAllocator(cell);
093    boolean mslabUsed = (toAdd != cell);
094    // This cell data is backed by the same byte[] where we read request in RPC(See HBASE-15180). By
095    // default MSLAB is ON and we might have copied cell to MSLAB area. If not we must do below deep
096    // copy. Or else we will keep referring to the bigger chunk of memory and prevent it from
097    // getting GCed.
098    // Copy to MSLAB would not have happened if
099    // 1. MSLAB is turned OFF. See "hbase.hregion.memstore.mslab.enabled"
100    // 2. When the size of the cell is bigger than the max size supported by MSLAB. See
101    // "hbase.hregion.memstore.mslab.max.allocation". This defaults to 256 KB
102    // 3. When cells are from Append/Increment operation.
103    if (!mslabUsed) {
104      toAdd = deepCopyIfNeeded(toAdd);
105    }
106    internalAdd(toAdd, mslabUsed, memstoreSize);
107  }
108
109  private static Cell deepCopyIfNeeded(Cell cell) {
110    if (cell instanceof ExtendedCell) {
111      return ((ExtendedCell) cell).deepClone();
112    }
113    return cell;
114  }
115
116  @Override
117  public void upsert(Iterable<Cell> cells, long readpoint, MemstoreSize memstoreSize) {
118    for (Cell cell : cells) {
119      upsert(cell, readpoint, memstoreSize);
120    }
121  }
122
123  /**
124   * @return Oldest timestamp of all the Cells in the MemStore
125   */
126  @Override
127  public long timeOfOldestEdit() {
128    return timeOfOldestEdit;
129  }
130
131  /**
132   * The passed snapshot was successfully persisted; it can be let go.
133   * @param id Id of the snapshot to clean out.
134   * @see MemStore#snapshot()
135   */
136  @Override
137  public void clearSnapshot(long id) throws UnexpectedStateException {
138    if (this.snapshotId == -1) return;  // already cleared
139    if (this.snapshotId != id) {
140      throw new UnexpectedStateException("Current snapshot id is " + this.snapshotId + ",passed "
141          + id);
142    }
143    // OK. Passed in snapshot is same as current snapshot. If not-empty,
144    // create a new snapshot and let the old one go.
145    Segment oldSnapshot = this.snapshot;
146    if (!this.snapshot.isEmpty()) {
147      this.snapshot = SegmentFactory.instance().createImmutableSegment(this.comparator);
148    }
149    this.snapshotId = NO_SNAPSHOT_ID;
150    oldSnapshot.close();
151  }
152
153  @Override
154  public MemstoreSize getSnapshotSize() {
155    return new MemstoreSize(this.snapshot.keySize(), this.snapshot.heapOverhead());
156  }
157
158  @Override
159  public String toString() {
160    StringBuffer buf = new StringBuffer();
161    int i = 1;
162    try {
163      for (Segment segment : getSegments()) {
164        buf.append("Segment (" + i + ") " + segment.toString() + "; ");
165        i++;
166      }
167    } catch (IOException e){
168      return e.toString();
169    }
170    return buf.toString();
171  }
172
173  protected Configuration getConfiguration() {
174    return conf;
175  }
176
177  protected void dump(Log log) {
178    active.dump(log);
179    snapshot.dump(log);
180  }
181
182
183  /*
184   * Inserts the specified Cell into MemStore and deletes any existing
185   * versions of the same row/family/qualifier as the specified Cell.
186   * <p>
187   * First, the specified Cell is inserted into the Memstore.
188   * <p>
189   * If there are any existing Cell in this MemStore with the same row,
190   * family, and qualifier, they are removed.
191   * <p>
192   * Callers must hold the read lock.
193   *
194   * @param cell the cell to be updated
195   * @param readpoint readpoint below which we can safely remove duplicate KVs
196   * @param memstoreSize
197   */
198  private void upsert(Cell cell, long readpoint, MemstoreSize memstoreSize) {
199    // Add the Cell to the MemStore
200    // Use the internalAdd method here since we (a) already have a lock
201    // and (b) cannot safely use the MSLAB here without potentially
202    // hitting OOME - see TestMemStore.testUpsertMSLAB for a
203    // test that triggers the pathological case if we don't avoid MSLAB
204    // here.
205    // This cell data is backed by the same byte[] where we read request in RPC(See HBASE-15180). We
206    // must do below deep copy. Or else we will keep referring to the bigger chunk of memory and
207    // prevent it from getting GCed.
208    cell = deepCopyIfNeeded(cell);
209    this.active.upsert(cell, readpoint, memstoreSize);
210    setOldestEditTimeToNow();
211    checkActiveSize();
212  }
213
214  /*
215   * @param a
216   * @param b
217   * @return Return lowest of a or b or null if both a and b are null
218   */
219  protected Cell getLowest(final Cell a, final Cell b) {
220    if (a == null) {
221      return b;
222    }
223    if (b == null) {
224      return a;
225    }
226    return comparator.compareRows(a, b) <= 0? a: b;
227  }
228
229  /*
230   * @param key Find row that follows this one.  If null, return first.
231   * @param set Set to look in for a row beyond <code>row</code>.
232   * @return Next row or null if none found.  If one found, will be a new
233   * KeyValue -- can be destroyed by subsequent calls to this method.
234   */
235  protected Cell getNextRow(final Cell key,
236      final NavigableSet<Cell> set) {
237    Cell result = null;
238    SortedSet<Cell> tail = key == null? set: set.tailSet(key);
239    // Iterate until we fall into the next row; i.e. move off current row
240    for (Cell cell: tail) {
241      if (comparator.compareRows(cell, key) <= 0) {
242        continue;
243      }
244      // Note: Not suppressing deletes or expired cells.  Needs to be handled
245      // by higher up functions.
246      result = cell;
247      break;
248    }
249    return result;
250  }
251
252  private Cell maybeCloneWithAllocator(Cell cell) {
253    return active.maybeCloneWithAllocator(cell);
254  }
255
256  /*
257   * Internal version of add() that doesn't clone Cells with the
258   * allocator, and doesn't take the lock.
259   *
260   * Callers should ensure they already have the read lock taken
261   * @param toAdd the cell to add
262   * @param mslabUsed whether using MSLAB
263   * @param memstoreSize
264   */
265  private void internalAdd(final Cell toAdd, final boolean mslabUsed, MemstoreSize memstoreSize) {
266    active.add(toAdd, mslabUsed, memstoreSize);
267    setOldestEditTimeToNow();
268    checkActiveSize();
269  }
270
271  private void setOldestEditTimeToNow() {
272    if (timeOfOldestEdit == Long.MAX_VALUE) {
273      timeOfOldestEdit = EnvironmentEdgeManager.currentTime();
274    }
275  }
276
277  /**
278   * @return The total size of cells in this memstore. We will not consider cells in the snapshot
279   */
280  protected abstract long keySize();
281
282  /**
283   * @return The total heap overhead of cells in this memstore. We will not consider cells in the
284   *         snapshot
285   */
286  protected abstract long heapOverhead();
287
288  protected CellComparator getComparator() {
289    return comparator;
290  }
291
292  @VisibleForTesting
293  MutableSegment getActive() {
294    return active;
295  }
296
297  @VisibleForTesting
298  ImmutableSegment getSnapshot() {
299    return snapshot;
300  }
301
302  /**
303   * Check whether anything need to be done based on the current active set size
304   */
305  protected abstract void checkActiveSize();
306
307  /**
308   * @return an ordered list of segments from most recent to oldest in memstore
309   */
310  protected abstract List<Segment> getSegments() throws IOException;
