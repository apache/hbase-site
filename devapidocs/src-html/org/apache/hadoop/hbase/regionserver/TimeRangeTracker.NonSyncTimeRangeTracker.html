035/**
036 * Stores minimum and maximum timestamp values, it is [minimumTimestamp, maximumTimestamp] in
037 * interval notation.
038 * Use this class at write-time ONLY. Too much synchronization to use at read time
039 * Use {@link TimeRange} at read time instead of this. See toTimeRange() to make TimeRange to use.
040 * MemStores use this class to track minimum and maximum timestamps. The TimeRangeTracker made by
041 * the MemStore is passed to the StoreFile for it to write out as part a flush in the the file
042 * metadata. If no memstore involved -- i.e. a compaction -- then the StoreFile will calculate its
043 * own TimeRangeTracker as it appends. The StoreFile serialized TimeRangeTracker is used
044 * at read time via an instance of {@link TimeRange} to test if Cells fit the StoreFile TimeRange.
045 */
046@InterfaceAudience.Private
047public abstract class TimeRangeTracker implements Writable {
048
049  public enum Type {
050    // thread-unsafe
051    NON_SYNC,
052    // thread-safe
053    SYNC
054  }
055
056  static final long INITIAL_MIN_TIMESTAMP = Long.MAX_VALUE;
057  static final long INITIAL_MAX_TIMESTAMP = -1L;
058
059  public static TimeRangeTracker create(Type type) {
060    switch (type) {
061      case NON_SYNC:
062        return new NonSyncTimeRangeTracker();
063      case SYNC:
064        return new SyncTimeRangeTracker();
065      default:
066        throw new UnsupportedOperationException("The type:" + type + " is unsupported");
067    }
068  }
069
070  public static TimeRangeTracker create(Type type, TimeRangeTracker trt) {
071    switch (type) {
072      case NON_SYNC:
073        return new NonSyncTimeRangeTracker(trt);
074      case SYNC:
075        return new SyncTimeRangeTracker(trt);
076      default:
077        throw new UnsupportedOperationException("The type:" + type + " is unsupported");
078    }
079  }
080
081  public static TimeRangeTracker create(Type type, long minimumTimestamp, long maximumTimestamp) {
082    switch (type) {
083      case NON_SYNC:
084        return new NonSyncTimeRangeTracker(minimumTimestamp, maximumTimestamp);
085      case SYNC:
086        return new SyncTimeRangeTracker(minimumTimestamp, maximumTimestamp);
087      default:
088        throw new UnsupportedOperationException("The type:" + type + " is unsupported");
089    }
090  }
091
092  protected abstract void setMax(long ts);
093  protected abstract void setMin(long ts);
094  protected abstract boolean compareAndSetMin(long expect, long update);
095  protected abstract boolean compareAndSetMax(long expect, long update);
096  /**
097   * Update the current TimestampRange to include the timestamp from <code>cell</code>.
098   * If the Key is of type DeleteColumn or DeleteFamily, it includes the
099   * entire time range from 0 to timestamp of the key.
100   * @param cell the Cell to include
101   */
102  public void includeTimestamp(final Cell cell) {
103    includeTimestamp(cell.getTimestamp());
104    if (CellUtil.isDeleteColumnOrFamily(cell)) {
105      includeTimestamp(0);
106    }
107  }
108
109  /**
110   * If required, update the current TimestampRange to include timestamp
111   * @param timestamp the timestamp value to include
112   */
113  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="MT_CORRECTNESS",
114      justification="Intentional")
115  void includeTimestamp(final long timestamp) {
116    long initialMinTimestamp = getMin();
117    if (timestamp < initialMinTimestamp) {
118      long curMinTimestamp = initialMinTimestamp;
119      while (timestamp < curMinTimestamp) {
120        if (!compareAndSetMin(curMinTimestamp, timestamp)) {
121          curMinTimestamp = getMin();
122        } else {
123          // successfully set minimumTimestamp, break.
124          break;
125        }
126      }
127
128      // When it reaches here, there are two possibilities:
129      //  1). timestamp >= curMinTimestamp, someone already sets the minimumTimestamp. In this case,
130      //      it still needs to check if initialMinTimestamp == INITIAL_MIN_TIMESTAMP to see
131      //      if it needs to update minimumTimestamp. Someone may already set both
132      //      minimumTimestamp/minimumTimestamp to the same value(curMinTimestamp),
133      //      need to check if maximumTimestamp needs to be updated.
134      //  2). timestamp < curMinTimestamp, it sets the minimumTimestamp successfully.
135      //      In this case,it still needs to check if initialMinTimestamp == INITIAL_MIN_TIMESTAMP
136      //      to see if it needs to set maximumTimestamp.
137      if (initialMinTimestamp != INITIAL_MIN_TIMESTAMP) {
138        // Someone already sets minimumTimestamp and timestamp is less than minimumTimestamp.
139        // In this case, no need to set maximumTimestamp as it will be set to at least
140        // initialMinTimestamp.
141        return;
142      }
143    }
144
145    long curMaxTimestamp = getMax();
146
147    if (timestamp > curMaxTimestamp) {
148      while (timestamp > curMaxTimestamp) {
149        if (!compareAndSetMax(curMaxTimestamp, timestamp)) {
150          curMaxTimestamp = getMax();
151        } else {
152          // successfully set maximumTimestamp, break
153          break;
154        }
155      }
156    }
157  }
158
159  /**
160   * Check if the range has ANY overlap with TimeRange
161   * @param tr TimeRange, it expects [minStamp, maxStamp)
162   * @return True if there is overlap, false otherwise
163   */
164  public boolean includesTimeRange(final TimeRange tr) {
165    return (getMin() < tr.getMax() && getMax() >= tr.getMin());
166  }
167
168  /**
169   * @return the minimumTimestamp
170   */
171  public abstract long getMin();
172
173  /**
174   * @return the maximumTimestamp
175   */
176  public abstract long getMax();
177
178  public void write(final DataOutput out) throws IOException {
179    out.writeLong(getMin());
180    out.writeLong(getMax());
181  }
182
183  public void readFields(final DataInput in) throws IOException {
184    setMin(in.readLong());
185    setMax(in.readLong());
186  }
187
188  @Override
189  public String toString() {
190    return "[" + getMin() + "," + getMax() + "]";
191  }
192
193  /**
194   * @return An instance of NonSyncTimeRangeTracker filled w/ the content of serialized
195   * NonSyncTimeRangeTracker in <code>timeRangeTrackerBytes</code>.
196   * @throws IOException
197   */
198  public static TimeRangeTracker getTimeRangeTracker(final byte [] timeRangeTrackerBytes)
199  throws IOException {
200    if (timeRangeTrackerBytes == null) return null;
201    TimeRangeTracker trt = TimeRangeTracker.create(Type.NON_SYNC);
202    Writables.copyWritable(timeRangeTrackerBytes, trt);
203    return trt;
204  }
205
206  /**
207   * @return An instance of a TimeRange made from the serialized TimeRangeTracker passed in
208   * <code>timeRangeTrackerBytes</code>.
209   * @throws IOException
210   */
211  static TimeRange getTimeRange(final byte [] timeRangeTrackerBytes) throws IOException {
212    TimeRangeTracker trt = getTimeRangeTracker(timeRangeTrackerBytes);
213    return trt == null? null: trt.toTimeRange();
214  }
215
216  /**
217   * @return Make a TimeRange from current state of <code>this</code>.
218   */
219  TimeRange toTimeRange() {
220    long min = getMin();
221    long max = getMax();
222    // Initial TimeRangeTracker timestamps are the opposite of what you want for a TimeRange. Fix!
223    if (min == INITIAL_MIN_TIMESTAMP) {
224      min = TimeRange.INITIAL_MIN_TIMESTAMP;
225    }
226    if (max == INITIAL_MAX_TIMESTAMP) {
227      max = TimeRange.INITIAL_MAX_TIMESTAMP;
228    }
229    return new TimeRange(min, max);
230  }
231
232  private static class NonSyncTimeRangeTracker extends TimeRangeTracker {
233    private long minimumTimestamp = INITIAL_MIN_TIMESTAMP;
234    private long maximumTimestamp = INITIAL_MAX_TIMESTAMP;
235
236    NonSyncTimeRangeTracker() {
237    }
238
239    NonSyncTimeRangeTracker(final TimeRangeTracker trt) {
240      this.minimumTimestamp = trt.getMin();
241      this.maximumTimestamp = trt.getMax();
242    }
243
244    NonSyncTimeRangeTracker(long minimumTimestamp, long maximumTimestamp) {
245      this.minimumTimestamp = minimumTimestamp;
246      this.maximumTimestamp = maximumTimestamp;
247    }
248
249    @Override
250    protected void setMax(long ts) {
251      maximumTimestamp = ts;
252    }
253
254    @Override
255    protected void setMin(long ts) {
256      minimumTimestamp = ts;
257    }
258
259    @Override
260    protected boolean compareAndSetMin(long expect, long update) {
261      if (minimumTimestamp != expect) {
262        return false;
263      }
264      minimumTimestamp = update;
265      return true;
266    }
267
268    @Override
269    protected boolean compareAndSetMax(long expect, long update) {
270      if (maximumTimestamp != expect) {
271        return false;
272      }
273      maximumTimestamp = update;
274      return true;
275    }
276
277    @Override
278    public long getMin() {
279      return minimumTimestamp;
280    }
281
282    @Override
283    public long getMax() {
284      return maximumTimestamp;
285    }
286  }

288  @VisibleForTesting
289  //In order to estimate the heap size, this inner class need to be accessible to TestHeapSize.
290  public static class SyncTimeRangeTracker extends TimeRangeTracker {
291    private final AtomicLong minimumTimestamp = new AtomicLong(INITIAL_MIN_TIMESTAMP);
292    private final AtomicLong maximumTimestamp = new AtomicLong(INITIAL_MAX_TIMESTAMP);
293
294    private SyncTimeRangeTracker() {
295    }
296
297    SyncTimeRangeTracker(final TimeRangeTracker trt) {
298      this.minimumTimestamp.set(trt.getMin());
299      this.maximumTimestamp.set(trt.getMax());
300    }
301
302    SyncTimeRangeTracker(long minimumTimestamp, long maximumTimestamp) {
303      this.minimumTimestamp.set(minimumTimestamp);
304      this.maximumTimestamp.set(maximumTimestamp);
305    }
306
307    @Override
308    protected void setMax(long ts) {
309      maximumTimestamp.set(ts);
310    }
311
312    @Override
313    protected void setMin(long ts) {
314      minimumTimestamp.set(ts);
315    }
316
317    @Override
318    protected boolean compareAndSetMin(long expect, long update) {
319      return minimumTimestamp.compareAndSet(expect, update);
320    }
321
322    @Override
323    protected boolean compareAndSetMax(long expect, long update) {
324      return maximumTimestamp.compareAndSet(expect, update);
325    }
326
327    @Override
328    public long getMin() {
329      return minimumTimestamp.get();
330    }
331
332    @Override
333    public long getMax() {
334      return maximumTimestamp.get();
335    }
336  }
