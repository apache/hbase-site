018package org.apache.hadoop.hbase.regionserver;
019
020import java.util.Arrays;
021
022import org.apache.yetus.audience.InterfaceAudience;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026/**
027 * A custom RegionSplitPolicy implementing a SplitPolicy that groups
028 * rows by a prefix of the row-key
029 *
030 * This ensures that a region is not split "inside" a prefix of a row key.
031 * I.e. rows can be co-located in a region by their prefix.
032 *
033 * @deprecated since 3.0.0 and will be removed in 4.0.0. Use {@link RegionSplitRestriction},
034 *   instead.
035 */
036@Deprecated
037@InterfaceAudience.Private
038public class KeyPrefixRegionSplitPolicy extends IncreasingToUpperBoundRegionSplitPolicy {
039  private static final Logger LOG = LoggerFactory
040      .getLogger(KeyPrefixRegionSplitPolicy.class);
041  @Deprecated
042  public static final String PREFIX_LENGTH_KEY_DEPRECATED = "prefix_split_key_policy.prefix_length";
043  public static final String PREFIX_LENGTH_KEY = "KeyPrefixRegionSplitPolicy.prefix_length";
044
045  private int prefixLength = 0;
046
047  @Override
048  public String toString() {
049    return "KeyPrefixRegionSplitPolicy{" + "prefixLength=" + prefixLength + ", " +
050      super.toString() + '}';
051  }
052
053  @Override
054  protected void configureForRegion(HRegion region) {
055    super.configureForRegion(region);
056    prefixLength = 0;
057
058    // read the prefix length from the table descriptor
059    String prefixLengthString = region.getTableDescriptor().getValue(
060        PREFIX_LENGTH_KEY);
061    if (prefixLengthString == null) {
062      //read the deprecated value
063      prefixLengthString = region.getTableDescriptor().getValue(PREFIX_LENGTH_KEY_DEPRECATED);
064      if (prefixLengthString == null) {
065        LOG.error(PREFIX_LENGTH_KEY + " not specified for table "
066            + region.getTableDescriptor().getTableName()
067            + ". Using default RegionSplitPolicy");
068        return;
069      }
070    }
071    try {
072      prefixLength = Integer.parseInt(prefixLengthString);
073    } catch (NumberFormatException nfe) {
074      /* Differentiate NumberFormatException from an invalid value range reported below. */
075      LOG.error("Number format exception when parsing " + PREFIX_LENGTH_KEY + " for table "
076          + region.getTableDescriptor().getTableName() + ":"
077          + prefixLengthString + ". " + nfe);
078      return;
079    }
080    if (prefixLength <= 0) {
081      LOG.error("Invalid value for " + PREFIX_LENGTH_KEY + " for table "
082          + region.getTableDescriptor().getTableName() + ":"
083          + prefixLengthString + ". Using default RegionSplitPolicy");
084    }
085  }
086
087  @Override
088  protected byte[] getSplitPoint() {
089    byte[] splitPoint = super.getSplitPoint();
090    if (prefixLength > 0 && splitPoint != null && splitPoint.length > 0) {
091      // group split keys by a prefix
092      return Arrays.copyOf(splitPoint,
093          Math.min(prefixLength, splitPoint.length));
094    } else {
095      return splitPoint;
096    }
097  }
098}
