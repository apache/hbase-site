019package org.apache.hadoop.hbase.regionserver;
020
021/**
022 * Interface of class that will wrap an HRegion and export numbers so they can be
023 * used in MetricsRegionSource
024 */
025public interface MetricsRegionWrapper {
026
027  /**
028   * Get the name of the table the region belongs to.
029   *
030   * @return The string version of the table name.
031   */
032  String getTableName();
033
034  /**
035   * Get the name of the namespace this table is in.
036   * @return String version of the namespace.  Can't be empty.
037   */
038  String getNamespace();
039
040  /**
041   * Get the name of the region.
042   *
043   * @return The encoded name of the region.
044   */
045  String getRegionName();
046
047  /**
048   * Get the number of stores hosted on this region server.
049   */
050  long getNumStores();
051
052  /**
053   * Get the number of store files hosted on this region server.
054   */
055  long getNumStoreFiles();
056
057  /**
058   * Get the size of the memstore on this region server.
059   */
060  long getMemstoreSize();
061
062  /**
063   * Get the total size of the store files this region server is serving from.
064   */
065  long getStoreFileSize();
066
067  /**
068   * Get the total number of read requests that have been issued against this region
069   */
070  long getReadRequestCount();
071
072  /**
073   * Get the total number of filtered read requests that have been issued against this region
074   */
075  long getFilteredReadRequestCount();
076
077  /**
078   * @return Max age of store files under this region
079   */
080  long getMaxStoreFileAge();
081
082  /**
083   * @return Min age of store files under this region
084   */
085  long getMinStoreFileAge();
086
087  /**
088   *  @return Average age of store files under this region
089   */
090  long getAvgStoreFileAge();
091
092  /**
093   *  @return Number of reference files under this region
094   */
095  long getNumReferenceFiles();
096
097  /**
098   * Get the total number of mutations that have been issued against this region.
099   */
100  long getWriteRequestCount();
101
102  long getNumFilesCompacted();
103
104  long getNumBytesCompacted();
105
106  long getNumCompactionsCompleted();
107
108  /**
109   *  @return Age of the last major compaction
110   */
111  long getLastMajorCompactionAge();
112
113  /**
114   * Returns the total number of compactions that have been reported as failed on this region.
115   * Note that a given compaction can be reported as both completed and failed if an exception
116   * is thrown in the processing after {@code HRegion.compact()}.
117   */
118  long getNumCompactionsFailed();
119
120  /**
121   * @return the total number of compactions that are currently queued(or being executed) at point in
122   *  time
123   */
124  long getNumCompactionsQueued();
125
126  /**
127   * @return the total number of flushes currently queued(being executed) for this region at point in
128   *  time
129   */
130  long getNumFlushesQueued();
131
132  /**
133   * @return the max number of compactions queued for this region
134   * Note that this metric is updated periodically and hence might miss some data points
135   */
136  long getMaxCompactionQueueSize();
137
138  /**
139   * @return the max number of flushes queued for this region
140   * Note that this metric is updated periodically and hence might miss some data points
141   */
142  long getMaxFlushQueueSize();
143
144  int getRegionHashCode();
145
146  /**
147   * Get the replica id of this region.
148   */
149  int getReplicaId();
150}
