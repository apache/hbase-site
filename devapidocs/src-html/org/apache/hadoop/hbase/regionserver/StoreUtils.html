019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022import java.util.Collection;
023import java.util.Optional;
024import java.util.OptionalInt;
025
026import org.apache.commons.logging.Log;
027import org.apache.commons.logging.LogFactory;
028import org.apache.hadoop.hbase.Cell;
029import org.apache.hadoop.hbase.CellComparator;
030import org.apache.hadoop.hbase.CellUtil;
031import org.apache.hadoop.hbase.classification.InterfaceAudience;
032
033/**
034 * Utility functions for region server storage layer.
035 */
036@InterfaceAudience.Private
037public class StoreUtils {
038
039  private static final Log LOG = LogFactory.getLog(StoreUtils.class);
040
041  /**
042   * Creates a deterministic hash code for store file collection.
043   */
044  public static OptionalInt getDeterministicRandomSeed(Collection<StoreFile> files) {
045    return files.stream().mapToInt(f -> f.getPath().getName().hashCode()).findFirst();
046  }
047
048  /**
049   * Determines whether any files in the collection are references.
050   * @param files The files.
051   */
052  public static boolean hasReferences(final Collection<StoreFile> files) {
053    if (files != null) {
054      for (StoreFile hsf: files) {
055        if (hsf.isReference()) {
056          return true;
057        }
058      }
059    }
060    return false;
061  }
062
063  /**
064   * Gets lowest timestamp from candidate StoreFiles
065   */
066  public static long getLowestTimestamp(final Collection<StoreFile> candidates)
067    throws IOException {
068    long minTs = Long.MAX_VALUE;
069    for (StoreFile storeFile : candidates) {
070      minTs = Math.min(minTs, storeFile.getModificationTimeStamp());
071    }
072    return minTs;
073  }
074
075  /**
076   * Gets the largest file (with reader) out of the list of files.
077   * @param candidates The files to choose from.
078   * @return The largest file; null if no file has a reader.
079   */
080  static Optional<StoreFile> getLargestFile(Collection<StoreFile> candidates) {
081    return candidates.stream().filter(f -> f.getReader() != null)
082        .max((f1, f2) -> Long.compare(f1.getReader().length(), f2.getReader().length()));
083  }
084
085  /**
086   * Return the largest memstoreTS found across all storefiles in the given list. Store files that
087   * were created by a mapreduce bulk load are ignored, as they do not correspond to any specific
088   * put operation, and thus do not have a memstoreTS associated with them.
089   * @return 0 if no non-bulk-load files are provided or, this is Store that does not yet have any
090   *         store files.
091   */
092  public static long getMaxMemstoreTSInList(Collection<StoreFile> sfs) {
093    long max = 0;
094    for (StoreFile sf : sfs) {
095      if (!sf.isBulkLoadResult()) {
096        max = Math.max(max, sf.getMaxMemstoreTS());
097      }
098    }
099    return max;
100  }
101
102  /**
103   * Return the highest sequence ID found across all storefiles in
104   * the given list.
105   * @param sfs
106   * @return 0 if no non-bulk-load files are provided or, this is Store that
107   * does not yet have any store files.
108   */
109  public static long getMaxSequenceIdInList(Collection<StoreFile> sfs) {
110    long max = 0;
111    for (StoreFile sf : sfs) {
112      max = Math.max(max, sf.getMaxSequenceId());
113    }
114    return max;
115  }
116
117  /**
118   * Gets the approximate mid-point of the given file that is optimal for use in splitting it.
119   * @param file the store file
120   * @param comparator Comparator used to compare KVs.
121   * @return The split point row, or null if splitting is not possible, or reader is null.
122   */
123  static Optional<byte[]> getFileSplitPoint(StoreFile file, CellComparator comparator)
124      throws IOException {
125    StoreFileReader reader = file.getReader();
126    if (reader == null) {
127      LOG.warn("Storefile " + file + " Reader is null; cannot get split point");
128      return Optional.empty();
129    }
130    // Get first, last, and mid keys. Midkey is the key that starts block
131    // in middle of hfile. Has column and timestamp. Need to return just
132    // the row we want to split on as midkey.
133    Cell midkey = reader.midkey();
134    if (midkey != null) {
135      Cell firstKey = reader.getFirstKey();
136      Cell lastKey = reader.getLastKey();
137      // if the midkey is the same as the first or last keys, we cannot (ever) split this region.
138      if (comparator.compareRows(midkey, firstKey) == 0 ||
139          comparator.compareRows(midkey, lastKey) == 0) {
140        if (LOG.isDebugEnabled()) {
141          LOG.debug("cannot split because midkey is the same as first or last row");
142        }
143        return Optional.empty();
144      }
145      return Optional.of(CellUtil.cloneRow(midkey));
146    }
147    return Optional.empty();
148  }
149}
