114/**
115 * A Store holds a column family in a Region.  Its a memstore and a set of zero
116 * or more StoreFiles, which stretch backwards over time.
117 *
118 * <p>There's no reason to consider append-logging at this level; all logging
119 * and locking is handled at the HRegion level.  Store just provides
120 * services to manage sets of StoreFiles.  One of the most important of those
121 * services is compaction services where files are aggregated once they pass
122 * a configurable threshold.
123 *
124 * <p>Locking and transactions are handled at a higher level.  This API should
125 * not be called directly but by an HRegion manager.
126 */

220  /**
221   * Constructor
222   * @param region
223   * @param family HColumnDescriptor for this column
224   * @param confParam configuration object
225   * failed.  Can be null.
226   * @throws IOException
227   */
228  protected HStore(final HRegion region, final ColumnFamilyDescriptor family,
229      final Configuration confParam) throws IOException {
230
231    this.fs = region.getRegionFileSystem();
232
233    // Assemble the store's home directory and Ensure it exists.
234    fs.createStoreDir(family.getNameAsString());
235    this.region = region;
236    this.family = family;
237    // 'conf' renamed to 'confParam' b/c we use this.conf in the constructor
238    // CompoundConfiguration will look for keys in reverse order of addition, so we'd
239    // add global config first, then table and cf overrides, then cf metadata.
240    this.conf = new CompoundConfiguration()
241      .add(confParam)
242      .addBytesMap(region.getTableDescriptor().getValues())
243      .addStringMap(family.getConfiguration())
244      .addBytesMap(family.getValues());
245    this.blocksize = family.getBlocksize();
246
247    // set block storage policy for store directory
248    String policyName = family.getStoragePolicy();
249    if (null == policyName) {
250      policyName = this.conf.get(BLOCK_STORAGE_POLICY_KEY, DEFAULT_BLOCK_STORAGE_POLICY);
251    }
252    this.fs.setStoragePolicy(family.getNameAsString(), policyName.trim());
253
254    this.dataBlockEncoder =
255        new HFileDataBlockEncoderImpl(family.getDataBlockEncoding());
256
257    this.comparator = region.getCellComparator();
258    // used by ScanQueryMatcher
259    long timeToPurgeDeletes =
260        Math.max(conf.getLong("hbase.hstore.time.to.purge.deletes", 0), 0);
261    LOG.trace("Time to purge deletes set to {}ms in store {}", timeToPurgeDeletes, this);
262    // Get TTL
263    long ttl = determineTTLFromFamily(family);
264    // Why not just pass a HColumnDescriptor in here altogether?  Even if have
265    // to clone it?
266    scanInfo = new ScanInfo(conf, family, ttl, timeToPurgeDeletes, this.comparator);
267    MemoryCompactionPolicy inMemoryCompaction = null;
268    if (this.getTableName().isSystemTable()) {
269      inMemoryCompaction = MemoryCompactionPolicy
270          .valueOf(conf.get("hbase.systemtables.compacting.memstore.type", "NONE"));
271    } else {
272      inMemoryCompaction = family.getInMemoryCompaction();
273    }
274    if (inMemoryCompaction == null) {
275      inMemoryCompaction =
276          MemoryCompactionPolicy.valueOf(conf.get(CompactingMemStore.COMPACTING_MEMSTORE_TYPE_KEY,
277            CompactingMemStore.COMPACTING_MEMSTORE_TYPE_DEFAULT));
278    }
279    String className;
280    switch (inMemoryCompaction) {
281      case NONE:
282        className = DefaultMemStore.class.getName();
283        this.memstore = ReflectionUtils.newInstance(DefaultMemStore.class,
284            new Object[] { conf, this.comparator });
285        break;
286      default:
287        Class<? extends CompactingMemStore> clz = conf.getClass(MEMSTORE_CLASS_NAME,
288          CompactingMemStore.class, CompactingMemStore.class);
289        className = clz.getName();
290        this.memstore = ReflectionUtils.newInstance(clz, new Object[] { conf, this.comparator, this,
291            this.getHRegion().getRegionServicesForStores(), inMemoryCompaction });
292    }
293    LOG.info("Memstore class name is {}", className);
294    this.offPeakHours = OffPeakHours.getInstance(conf);
295
296    // Setting up cache configuration for this family
297    createCacheConf(family);
298
299    this.verifyBulkLoads = conf.getBoolean("hbase.hstore.bulkload.verify", false);
300
301    this.blockingFileCount =
302        conf.getInt(BLOCKING_STOREFILES_KEY, DEFAULT_BLOCKING_STOREFILE_COUNT);
303    this.compactionCheckMultiplier = conf.getInt(
304        COMPACTCHECKER_INTERVAL_MULTIPLIER_KEY, DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER);
305    if (this.compactionCheckMultiplier <= 0) {
306      LOG.error("Compaction check period multiplier must be positive, setting default: {}",
307          DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER);
308      this.compactionCheckMultiplier = DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER;
309    }
310
311    if (HStore.closeCheckInterval == 0) {
312      HStore.closeCheckInterval = conf.getInt(
313          "hbase.hstore.close.check.interval", 10*1000*1000 /* 10 MB */);
314    }
315
316    this.storeEngine = createStoreEngine(this, this.conf, this.comparator);
317    this.storeEngine.getStoreFileManager().loadFiles(loadStoreFiles());
318
319    // Initialize checksum type from name. The names are CRC32, CRC32C, etc.
320    this.checksumType = getChecksumType(conf);
321    // Initialize bytes per checksum
322    this.bytesPerChecksum = getBytesPerChecksum(conf);
323    flushRetriesNumber = conf.getInt(
324        "hbase.hstore.flush.retries.number", DEFAULT_FLUSH_RETRIES_NUMBER);
325    pauseTime = conf.getInt(HConstants.HBASE_SERVER_PAUSE, HConstants.DEFAULT_HBASE_SERVER_PAUSE);
326    if (flushRetriesNumber <= 0) {
327      throw new IllegalArgumentException(
328          "hbase.hstore.flush.retries.number must be > 0, not "
329              + flushRetriesNumber);
330    }
331    cryptoContext = EncryptionUtil.createEncryptionContext(conf, family);
332  }

355  /**
356   * @param family
357   * @return TTL in seconds of the specified family
358   */
359  public static long determineTTLFromFamily(final ColumnFamilyDescriptor family) {
360    // HCD.getTimeToLive returns ttl in seconds.  Convert to milliseconds.
361    long ttl = family.getTimeToLive();
362    if (ttl == HConstants.FOREVER) {
363      // Default is unlimited ttl.
364      ttl = Long.MAX_VALUE;
365    } else if (ttl == -1) {
366      ttl = Long.MAX_VALUE;
367    } else {
368      // Second -> ms adjust for user data
369      ttl *= 1000;
370    }
371    return ttl;
372  }

522  private List<HStoreFile> openStoreFiles(Collection<StoreFileInfo> files) throws IOException {
523    if (CollectionUtils.isEmpty(files)) {
524      return Collections.emptyList();
525    }
526    // initialize the thread pool for opening store files in parallel..
527    ThreadPoolExecutor storeFileOpenerThreadPool =
528      this.region.getStoreFileOpenAndCloseThreadPool("StoreFileOpenerThread-" +
529          this.getColumnFamilyName());
530    CompletionService<HStoreFile> completionService = new ExecutorCompletionService<>(storeFileOpenerThreadPool);
531
532    int totalValidStoreFile = 0;
533    for (StoreFileInfo storeFileInfo : files) {
534      // open each store file in parallel
535      completionService.submit(() -> this.createStoreFileAndReader(storeFileInfo));
536      totalValidStoreFile++;
537    }
538
539    ArrayList<HStoreFile> results = new ArrayList<>(files.size());
540    IOException ioe = null;
541    try {
542      for (int i = 0; i < totalValidStoreFile; i++) {
543        try {
544          HStoreFile storeFile = completionService.take().get();
545          if (storeFile != null) {
546            long length = storeFile.getReader().length();
547            this.storeSize += length;
548            this.totalUncompressedBytes += storeFile.getReader().getTotalUncompressedBytes();
549            LOG.debug("loaded {}", storeFile);
550            results.add(storeFile);
551          }
552        } catch (InterruptedException e) {
553          if (ioe == null) ioe = new InterruptedIOException(e.getMessage());
554        } catch (ExecutionException e) {
555          if (ioe == null) ioe = new IOException(e.getCause());
556        }
557      }
558    } finally {
559      storeFileOpenerThreadPool.shutdownNow();
560    }
561    if (ioe != null) {
562      // close StoreFile readers
563      boolean evictOnClose =
564          cacheConf != null? cacheConf.shouldEvictOnClose(): true;
565      for (HStoreFile file : results) {
566        try {
567          if (file != null) {
568            file.closeStoreFile(evictOnClose);
569          }
570        } catch (IOException e) {
571          LOG.warn("Could not close store file", e);
572        }
573      }
574      throw ioe;
575    }
576
577    return results;
578  }

599  /**
600   * Checks the underlying store files, and opens the files that  have not
601   * been opened, and removes the store file readers for store files no longer
602   * available. Mainly used by secondary region replicas to keep up to date with
603   * the primary region files.
604   * @throws IOException
605   */
606  private void refreshStoreFilesInternal(Collection<StoreFileInfo> newFiles) throws IOException {
607    StoreFileManager sfm = storeEngine.getStoreFileManager();
608    Collection<HStoreFile> currentFiles = sfm.getStorefiles();
609    Collection<HStoreFile> compactedFiles = sfm.getCompactedfiles();
610    if (currentFiles == null) currentFiles = Collections.emptySet();
611    if (newFiles == null) newFiles = Collections.emptySet();
612    if (compactedFiles == null) compactedFiles = Collections.emptySet();
613
614    HashMap<StoreFileInfo, HStoreFile> currentFilesSet = new HashMap<>(currentFiles.size());
615    for (HStoreFile sf : currentFiles) {
616      currentFilesSet.put(sf.getFileInfo(), sf);
617    }
618    HashMap<StoreFileInfo, HStoreFile> compactedFilesSet = new HashMap<>(compactedFiles.size());
619    for (HStoreFile sf : compactedFiles) {
620      compactedFilesSet.put(sf.getFileInfo(), sf);
621    }
622
623    Set<StoreFileInfo> newFilesSet = new HashSet<StoreFileInfo>(newFiles);
624    // Exclude the files that have already been compacted
625    newFilesSet = Sets.difference(newFilesSet, compactedFilesSet.keySet());
626    Set<StoreFileInfo> toBeAddedFiles = Sets.difference(newFilesSet, currentFilesSet.keySet());
627    Set<StoreFileInfo> toBeRemovedFiles = Sets.difference(currentFilesSet.keySet(), newFilesSet);
628
629    if (toBeAddedFiles.isEmpty() && toBeRemovedFiles.isEmpty()) {
630      return;
631    }
632
633    LOG.info("Refreshing store files for region " + this.getRegionInfo().getRegionNameAsString()
634      + " files to add: " + toBeAddedFiles + " files to remove: " + toBeRemovedFiles);
635
636    Set<HStoreFile> toBeRemovedStoreFiles = new HashSet<>(toBeRemovedFiles.size());
637    for (StoreFileInfo sfi : toBeRemovedFiles) {
638      toBeRemovedStoreFiles.add(currentFilesSet.get(sfi));
639    }
640
641    // try to open the files
642    List<HStoreFile> openedFiles = openStoreFiles(toBeAddedFiles);
643
644    // propogate the file changes to the underlying store file manager
645    replaceStoreFiles(toBeRemovedStoreFiles, openedFiles); //won't throw an exception
646
647    // Advance the memstore read point to be at least the new store files seqIds so that
648    // readers might pick it up. This assumes that the store is not getting any writes (otherwise
649    // in-flight transactions might be made visible)
650    if (!toBeAddedFiles.isEmpty()) {
651      // we must have the max sequence id here as we do have several store files
652      region.getMVCC().advanceTo(this.getMaxSequenceId().getAsLong());
653    }
654
655    completeCompaction(toBeRemovedStoreFiles);
656  }

730  /**
731   * This throws a WrongRegionException if the HFile does not fit in this region, or an
732   * InvalidHFileException if the HFile is not valid.
733   */
734  public void assertBulkLoadHFileOk(Path srcPath) throws IOException {
735    HFile.Reader reader  = null;
736    try {
737      LOG.info("Validating hfile at " + srcPath + " for inclusion in "
738          + "store " + this + " region " + this.getRegionInfo().getRegionNameAsString());
739      reader = HFile.createReader(srcPath.getFileSystem(conf), srcPath, cacheConf,
740        isPrimaryReplicaStore(), conf);
741      reader.loadFileInfo();
742
743      Optional<byte[]> firstKey = reader.getFirstRowKey();
744      Preconditions.checkState(firstKey.isPresent(), "First key can not be null");
745      Optional<Cell> lk = reader.getLastKey();
746      Preconditions.checkState(lk.isPresent(), "Last key can not be null");
747      byte[] lastKey =  CellUtil.cloneRow(lk.get());
748
749      if (LOG.isDebugEnabled()) {
750        LOG.debug("HFile bounds: first=" + Bytes.toStringBinary(firstKey.get()) +
751            " last=" + Bytes.toStringBinary(lastKey));
752        LOG.debug("Region bounds: first=" +
753            Bytes.toStringBinary(getRegionInfo().getStartKey()) +
754            " last=" + Bytes.toStringBinary(getRegionInfo().getEndKey()));
755      }
756
757      if (!this.getRegionInfo().containsRange(firstKey.get(), lastKey)) {
758        throw new WrongRegionException(
759            "Bulk load file " + srcPath.toString() + " does not fit inside region "
760            + this.getRegionInfo().getRegionNameAsString());
761      }
762
763      if(reader.length() > conf.getLong(HConstants.HREGION_MAX_FILESIZE,
764          HConstants.DEFAULT_MAX_FILE_SIZE)) {
765        LOG.warn("Trying to bulk load hfile " + srcPath + " with size: " +
766            reader.length() + " bytes can be problematic as it may lead to oversplitting.");
767      }
768
769      if (verifyBulkLoads) {
770        long verificationStartTime = EnvironmentEdgeManager.currentTime();
771        LOG.info("Full verification started for bulk load hfile: {}", srcPath);
772        Cell prevCell = null;
773        HFileScanner scanner = reader.getScanner(false, false, false);
774        scanner.seekTo();
775        do {
776          Cell cell = scanner.getCell();
777          if (prevCell != null) {
778            if (comparator.compareRows(prevCell, cell) > 0) {
779              throw new InvalidHFileException("Previous row is greater than"
780                  + " current row: path=" + srcPath + " previous="
781                  + CellUtil.getCellKeyAsString(prevCell) + " current="
782                  + CellUtil.getCellKeyAsString(cell));
783            }
784            if (CellComparator.getInstance().compareFamilies(prevCell, cell) != 0) {
785              throw new InvalidHFileException("Previous key had different"
786                  + " family compared to current key: path=" + srcPath
787                  + " previous="
788                  + Bytes.toStringBinary(prevCell.getFamilyArray(), prevCell.getFamilyOffset(),
789                      prevCell.getFamilyLength())
790                  + " current="
791                  + Bytes.toStringBinary(cell.getFamilyArray(), cell.getFamilyOffset(),
792                      cell.getFamilyLength()));
793            }
794          }
795          prevCell = cell;
796        } while (scanner.next());
797      LOG.info("Full verification complete for bulk load hfile: " + srcPath.toString()
798         + " took " + (EnvironmentEdgeManager.currentTime() - verificationStartTime)
799         + " ms");
800      }
801    } finally {
802      if (reader != null) reader.close();
803    }
804  }

871  /**
872   * Close all the readers We don't need to worry about subsequent requests because the Region holds
873   * a write lock that will prevent any more reads or writes.
874   * @return the {@link StoreFile StoreFiles} that were previously being used.
875   * @throws IOException on failure
876   */
877  public ImmutableCollection<HStoreFile> close() throws IOException {
878    this.archiveLock.lock();
879    this.lock.writeLock().lock();
880    try {
881      // Clear so metrics doesn't find them.
882      ImmutableCollection<HStoreFile> result = storeEngine.getStoreFileManager().clearFiles();
883      Collection<HStoreFile> compactedfiles =
884          storeEngine.getStoreFileManager().clearCompactedFiles();
885      // clear the compacted files
886      if (CollectionUtils.isNotEmpty(compactedfiles)) {
887        removeCompactedfiles(compactedfiles);
888      }
889      if (!result.isEmpty()) {
890        // initialize the thread pool for closing store files in parallel.
891        ThreadPoolExecutor storeFileCloserThreadPool = this.region
892            .getStoreFileOpenAndCloseThreadPool("StoreFileCloserThread-"
893                + this.getColumnFamilyName());
894
895        // close each store file in parallel
896        CompletionService<Void> completionService =
897          new ExecutorCompletionService<>(storeFileCloserThreadPool);
898        for (HStoreFile f : result) {
899          completionService.submit(new Callable<Void>() {
900            @Override
901            public Void call() throws IOException {
902              boolean evictOnClose =
903                  cacheConf != null? cacheConf.shouldEvictOnClose(): true;
904              f.closeStoreFile(evictOnClose);
905              return null;
906            }
907          });
908        }
909
910        IOException ioe = null;
911        try {
912          for (int i = 0; i < result.size(); i++) {
913            try {
914              Future<Void> future = completionService.take();
915              future.get();
916            } catch (InterruptedException e) {
917              if (ioe == null) {
918                ioe = new InterruptedIOException();
919                ioe.initCause(e);
920              }
921            } catch (ExecutionException e) {
922              if (ioe == null) ioe = new IOException(e.getCause());
923            }
924          }
925        } finally {
926          storeFileCloserThreadPool.shutdownNow();
927        }
928        if (ioe != null) throw ioe;
929      }
930      LOG.info("Closed {}", this);
931      return result;
932    } finally {
933      this.lock.writeLock().unlock();
934      this.archiveLock.unlock();
935    }
936  }

953  /**
954   * Write out current snapshot. Presumes {@link #snapshot()} has been called previously.
955   * @param logCacheFlushId flush sequence number
956   * @param snapshot
957   * @param status
958   * @param throughputController
959   * @return The path name of the tmp file to which the store was flushed
960   * @throws IOException if exception occurs during process
961   */
962  protected List<Path> flushCache(final long logCacheFlushId, MemStoreSnapshot snapshot,
963      MonitoredTask status, ThroughputController throughputController,
964      FlushLifeCycleTracker tracker) throws IOException {
965    // If an exception happens flushing, we let it out without clearing
966    // the memstore snapshot.  The old snapshot will be returned when we say
967    // 'snapshot', the next time flush comes around.
968    // Retry after catching exception when flushing, otherwise server will abort
969    // itself
970    StoreFlusher flusher = storeEngine.getStoreFlusher();
971    IOException lastException = null;
972    for (int i = 0; i < flushRetriesNumber; i++) {
973      try {
974        List<Path> pathNames =
975            flusher.flushSnapshot(snapshot, logCacheFlushId, status, throughputController, tracker);
976        Path lastPathName = null;
977        try {
978          for (Path pathName : pathNames) {
979            lastPathName = pathName;
980            validateStoreFile(pathName);
981          }
982          return pathNames;
983        } catch (Exception e) {
984          LOG.warn("Failed validating store file {}, retrying num={}", lastPathName, i, e);
985          if (e instanceof IOException) {
986            lastException = (IOException) e;
987          } else {
988            lastException = new IOException(e);
989          }
990        }
991      } catch (IOException e) {
992        LOG.warn("Failed flushing store file, retrying num={}", i, e);
993        lastException = e;
994      }
995      if (lastException != null && i < (flushRetriesNumber - 1)) {
996        try {
997          Thread.sleep(pauseTime);
998        } catch (InterruptedException e) {
999          IOException iie = new InterruptedIOException();
1000          iie.initCause(e);
1001          throw iie;
1002        }
1003      }
1004    }
1005    throw lastException;
1006  }

1293  /**
1294   * Compact the StoreFiles.  This method may take some time, so the calling
1295   * thread must be able to block for long periods.
1296   *
1297   * <p>During this time, the Store can work as usual, getting values from
1298   * StoreFiles and writing new StoreFiles from the memstore.
1299   *
1300   * Existing StoreFiles are not destroyed until the new compacted StoreFile is
1301   * completely written-out to disk.
1302   *
1303   * <p>The compactLock prevents multiple simultaneous compactions.
1304   * The structureLock prevents us from interfering with other write operations.
1305   *
1306   * <p>We don't want to hold the structureLock for the whole time, as a compact()
1307   * can be lengthy and we want to allow cache-flushes during this period.
1308   *
1309   * <p> Compaction event should be idempotent, since there is no IO Fencing for
1310   * the region directory in hdfs. A region server might still try to complete the
1311   * compaction after it lost the region. That is why the following events are carefully
1312   * ordered for a compaction:
1313   *  1. Compaction writes new files under region/.tmp directory (compaction output)
1314   *  2. Compaction atomically moves the temporary file under region directory
1315   *  3. Compaction appends a WAL edit containing the compaction input and output files.
1316   *  Forces sync on WAL.
1317   *  4. Compaction deletes the input files from the region directory.
1318   *
1319   * Failure conditions are handled like this:
1320   *  - If RS fails before 2, compaction wont complete. Even if RS lives on and finishes
1321   *  the compaction later, it will only write the new data file to the region directory.
1322   *  Since we already have this data, this will be idempotent but we will have a redundant
1323   *  copy of the data.
1324   *  - If RS fails between 2 and 3, the region will have a redundant copy of the data. The
1325   *  RS that failed won't be able to finish snyc() for WAL because of lease recovery in WAL.
1326   *  - If RS fails after 3, the region region server who opens the region will pick up the
1327   *  the compaction marker from the WAL and replay it by removing the compaction input files.
1328   *  Failed RS can also attempt to delete those files, but the operation will be idempotent
1329   *
1330   * See HBASE-2231 for details.
1331   *
1332   * @param compaction compaction details obtained from requestCompaction()
1333   * @throws IOException
1334   * @return Storefile we compacted into or null if we failed or opted out early.
1335   */
1336  public List<HStoreFile> compact(CompactionContext compaction,
1337    ThroughputController throughputController, User user) throws IOException {
1338    assert compaction != null;
1339    List<HStoreFile> sfs = null;
1340    CompactionRequestImpl cr = compaction.getRequest();
1341    try {
1342      // Do all sanity checking in here if we have a valid CompactionRequestImpl
1343      // because we need to clean up after it on the way out in a finally
1344      // block below
1345      long compactionStartTime = EnvironmentEdgeManager.currentTime();
1346      assert compaction.hasSelection();
1347      Collection<HStoreFile> filesToCompact = cr.getFiles();
1348      assert !filesToCompact.isEmpty();
1349      synchronized (filesCompacting) {
1350        // sanity check: we're compacting files that this store knows about
1351        // TODO: change this to LOG.error() after more debugging
1352        Preconditions.checkArgument(filesCompacting.containsAll(filesToCompact));
1353      }
1354
1355      // Ready to go. Have list of files to compact.
1356      LOG.info("Starting compaction of " + filesToCompact +
1357        " into tmpdir=" + fs.getTempDir() + ", totalSize=" +
1358          TraditionalBinaryPrefix.long2String(cr.getSize(), "", 1));
1359
1360      return doCompaction(cr, filesToCompact, user, compactionStartTime,
1361          compaction.compact(throughputController, user));
1362    } finally {
1363      finishCompactionRequest(cr);
1364    }
1365  }

1367  @VisibleForTesting
1368  protected List<HStoreFile> doCompaction(CompactionRequestImpl cr,
1369      Collection<HStoreFile> filesToCompact, User user, long compactionStartTime,
1370      List<Path> newFiles) throws IOException {
1371    // Do the steps necessary to complete the compaction.
1372    List<HStoreFile> sfs = moveCompactedFilesIntoPlace(cr, newFiles, user);
1373    writeCompactionWalRecord(filesToCompact, sfs);
1374    replaceStoreFiles(filesToCompact, sfs);
1375    if (cr.isMajor()) {
1376      majorCompactedCellsCount += getCompactionProgress().totalCompactingKVs;
1377      majorCompactedCellsSize += getCompactionProgress().totalCompactedSize;
1378    } else {
1379      compactedCellsCount += getCompactionProgress().totalCompactingKVs;
1380      compactedCellsSize += getCompactionProgress().totalCompactedSize;
1381    }
1382    long outputBytes = getTotalSize(sfs);
1383
1384    // At this point the store will use new files for all new scanners.
1385    completeCompaction(filesToCompact); // update store size.
1386
1387    long now = EnvironmentEdgeManager.currentTime();
1388    if (region.getRegionServerServices() != null
1389        && region.getRegionServerServices().getMetrics() != null) {
1390      region.getRegionServerServices().getMetrics().updateCompaction(cr.isMajor(),
1391          now - compactionStartTime, cr.getFiles().size(), newFiles.size(), cr.getSize(),
1392          outputBytes);
1393    }
1394
1395    logCompactionEndMessage(cr, sfs, now, compactionStartTime);
1396    return sfs;
1397  }

1497  /**
1498   * Call to complete a compaction. Its for the case where we find in the WAL a compaction
1499   * that was not finished.  We could find one recovering a WAL after a regionserver crash.
1500   * See HBASE-2231.
1501   * @param compaction
1502   */
1503  public void replayCompactionMarker(CompactionDescriptor compaction, boolean pickCompactionFiles,
1504      boolean removeFiles) throws IOException {
1505    LOG.debug("Completing compaction from the WAL marker");
1506    List<String> compactionInputs = compaction.getCompactionInputList();
1507    List<String> compactionOutputs = Lists.newArrayList(compaction.getCompactionOutputList());
1508
1509    // The Compaction Marker is written after the compaction is completed,
1510    // and the files moved into the region/family folder.
1511    //
1512    // If we crash after the entry is written, we may not have removed the
1513    // input files, but the output file is present.
1514    // (The unremoved input files will be removed by this function)
1515    //
1516    // If we scan the directory and the file is not present, it can mean that:
1517    //   - The file was manually removed by the user
1518    //   - The file was removed as consequence of subsequent compaction
1519    // so, we can't do anything with the "compaction output list" because those
1520    // files have already been loaded when opening the region (by virtue of
1521    // being in the store's folder) or they may be missing due to a compaction.
1522
1523    String familyName = this.getColumnFamilyName();
1524    Set<String> inputFiles = new HashSet<>();
1525    for (String compactionInput : compactionInputs) {
1526      Path inputPath = fs.getStoreFilePath(familyName, compactionInput);
1527      inputFiles.add(inputPath.getName());
1528    }
1529
1530    //some of the input files might already be deleted
1531    List<HStoreFile> inputStoreFiles = new ArrayList<>(compactionInputs.size());
1532    for (HStoreFile sf : this.getStorefiles()) {
1533      if (inputFiles.contains(sf.getPath().getName())) {
1534        inputStoreFiles.add(sf);
1535      }
1536    }
1537
1538    // check whether we need to pick up the new files
1539    List<HStoreFile> outputStoreFiles = new ArrayList<>(compactionOutputs.size());
1540
1541    if (pickCompactionFiles) {
1542      for (HStoreFile sf : this.getStorefiles()) {
1543        compactionOutputs.remove(sf.getPath().getName());
1544      }
1545      for (String compactionOutput : compactionOutputs) {
1546        StoreFileInfo storeFileInfo = fs.getStoreFileInfo(getColumnFamilyName(), compactionOutput);
1547        HStoreFile storeFile = createStoreFileAndReader(storeFileInfo);
1548        outputStoreFiles.add(storeFile);
1549      }
1550    }
1551
1552    if (!inputStoreFiles.isEmpty() || !outputStoreFiles.isEmpty()) {
1553      LOG.info("Replaying compaction marker, replacing input files: " +
1554          inputStoreFiles + " with output files : " + outputStoreFiles);
1555      this.replaceStoreFiles(inputStoreFiles, outputStoreFiles);
1556      this.completeCompaction(inputStoreFiles);
1557    }
1558  }

1649  public Optional<CompactionContext> requestCompaction(int priority,
1650      CompactionLifeCycleTracker tracker, User user) throws IOException {
1651    // don't even select for compaction if writes are disabled
1652    if (!this.areWritesEnabled()) {
1653      return Optional.empty();
1654    }
1655    // Before we do compaction, try to get rid of unneeded files to simplify things.
1656    removeUnneededFiles();
1657
1658    final CompactionContext compaction = storeEngine.createCompaction();
1659    CompactionRequestImpl request = null;
1660    this.lock.readLock().lock();
1661    try {
1662      synchronized (filesCompacting) {
1663        // First, see if coprocessor would want to override selection.
1664        if (this.getCoprocessorHost() != null) {
1665          final List<HStoreFile> candidatesForCoproc = compaction.preSelect(this.filesCompacting);
1666          boolean override = getCoprocessorHost().preCompactSelection(this,
1667              candidatesForCoproc, tracker, user);
1668          if (override) {
1669            // Coprocessor is overriding normal file selection.
1670            compaction.forceSelect(new CompactionRequestImpl(candidatesForCoproc));
1671          }
1672        }
1673
1674        // Normal case - coprocessor is not overriding file selection.
1675        if (!compaction.hasSelection()) {
1676          boolean isUserCompaction = priority == Store.PRIORITY_USER;
1677          boolean mayUseOffPeak = offPeakHours.isOffPeakHour() &&
1678              offPeakCompactionTracker.compareAndSet(false, true);
1679          try {
1680            compaction.select(this.filesCompacting, isUserCompaction,
1681              mayUseOffPeak, forceMajor && filesCompacting.isEmpty());
1682          } catch (IOException e) {
1683            if (mayUseOffPeak) {
1684              offPeakCompactionTracker.set(false);
1685            }
1686            throw e;
1687          }
1688          assert compaction.hasSelection();
1689          if (mayUseOffPeak && !compaction.getRequest().isOffPeak()) {
1690            // Compaction policy doesn't want to take advantage of off-peak.
1691            offPeakCompactionTracker.set(false);
1692          }
1693        }
1694        if (this.getCoprocessorHost() != null) {
1695          this.getCoprocessorHost().postCompactSelection(
1696              this, ImmutableList.copyOf(compaction.getRequest().getFiles()), tracker,
1697              compaction.getRequest(), user);
1698        }
1699        // Finally, we have the resulting files list. Check if we have any files at all.
1700        request = compaction.getRequest();
1701        Collection<HStoreFile> selectedFiles = request.getFiles();
1702        if (selectedFiles.isEmpty()) {
1703          return Optional.empty();
1704        }
1705
1706        addToCompactingFiles(selectedFiles);
1707
1708        // If we're enqueuing a major, clear the force flag.
1709        this.forceMajor = this.forceMajor && !request.isMajor();
1710
1711        // Set common request properties.
1712        // Set priority, either override value supplied by caller or from store.
1713        request.setPriority((priority != Store.NO_PRIORITY) ? priority : getCompactPriority());
1714        request.setDescription(getRegionInfo().getRegionNameAsString(), getColumnFamilyName());
1715        request.setTracker(tracker);
1716      }
1717    } finally {
1718      this.lock.readLock().unlock();
1719    }
1720
1721    if (LOG.isDebugEnabled()) {
1722      LOG.debug(getRegionInfo().getEncodedName() + " - " + getColumnFamilyName()
1723          + ": Initiating " + (request.isMajor() ? "major" : "minor") + " compaction"
1724          + (request.isAllFiles() ? " (all files)" : ""));
1725    }
1726    this.region.reportCompactionRequestStart(request.isMajor());
1727    return Optional.of(compaction);
1728  }

1732    if (CollectionUtils.isEmpty(filesToAdd)) {

1740    Collections.sort(filesCompacting, storeEngine.getStoreFileManager().getStoreFileComparator());

1743  private void removeUnneededFiles() throws IOException {
1744    if (!conf.getBoolean("hbase.store.delete.expired.storefile", true)) return;
1745    if (getColumnFamilyDescriptor().getMinVersions() > 0) {
1746      LOG.debug("Skipping expired store file removal due to min version being {}",
1747          getColumnFamilyDescriptor().getMinVersions());
1748      return;
1749    }
1750    this.lock.readLock().lock();
1751    Collection<HStoreFile> delSfs = null;
1752    try {
1753      synchronized (filesCompacting) {
1754        long cfTtl = getStoreFileTtl();
1755        if (cfTtl != Long.MAX_VALUE) {
1756          delSfs = storeEngine.getStoreFileManager().getUnneededFiles(
1757              EnvironmentEdgeManager.currentTime() - cfTtl, filesCompacting);
1758          addToCompactingFiles(delSfs);
1759        }
1760      }
1761    } finally {
1762      this.lock.readLock().unlock();
1763    }
1764
1765    if (CollectionUtils.isEmpty(delSfs)) {
1766      return;
1767    }
1768
1769    Collection<HStoreFile> newFiles = Collections.emptyList(); // No new files.
1770    writeCompactionWalRecord(delSfs, newFiles);
1771    replaceStoreFiles(delSfs, newFiles);

2460  public synchronized void closeAndArchiveCompactedFiles() throws IOException {
2461    // ensure other threads do not attempt to archive the same files on close()
2462    archiveLock.lock();
2463    try {
2464      lock.readLock().lock();
2465      Collection<HStoreFile> copyCompactedfiles = null;
2466      try {
2467        Collection<HStoreFile> compactedfiles =
2468            this.getStoreEngine().getStoreFileManager().getCompactedfiles();
2469        if (CollectionUtils.isNotEmpty(compactedfiles)) {
2470          // Do a copy under read lock
2471          copyCompactedfiles = new ArrayList<>(compactedfiles);
2472        } else {
2473          LOG.trace("No compacted files to archive");
2474        }
2475      } finally {
2476        lock.readLock().unlock();
2477      }
2478      if (CollectionUtils.isNotEmpty(copyCompactedfiles)) {
2479        removeCompactedfiles(copyCompactedfiles);
2480      }
2481    } finally {
2482      archiveLock.unlock();
2483    }

2486  /**
2487   * Archives and removes the compacted files
2488   * @param compactedfiles The compacted files in this store that are not active in reads
2489   * @throws IOException
2490   */
2491  private void removeCompactedfiles(Collection<HStoreFile> compactedfiles)
2492      throws IOException {
2493    final List<HStoreFile> filesToRemove = new ArrayList<>(compactedfiles.size());
2494    for (final HStoreFile file : compactedfiles) {
2495      synchronized (file) {
2496        try {
2497          StoreFileReader r = file.getReader();
2498          if (r == null) {
2499            LOG.debug("The file {} was closed but still not archived", file);
2500            filesToRemove.add(file);
2501            continue;
2502          }
2503          if (file.isCompactedAway() && !file.isReferencedInReads()) {
2504            // Even if deleting fails we need not bother as any new scanners won't be
2505            // able to use the compacted file as the status is already compactedAway
2506            LOG.trace("Closing and archiving the file {}", file);
2507            r.close(true);
2508            // Just close and return
2509            filesToRemove.add(file);
2510          }
2511        } catch (Exception e) {
2512          LOG.error("Exception while trying to close the compacted store file {}",
2513            file.getPath(), e);
2514        }
2515      }
2516    }
2517    if (this.isPrimaryReplicaStore()) {
2518      // Only the primary region is allowed to move the file to archive.
2519      // The secondary region does not move the files to archive. Any active reads from
2520      // the secondary region will still work because the file as such has active readers on it.
2521      if (!filesToRemove.isEmpty()) {
2522        LOG.debug("Moving the files {} to archive", filesToRemove);
2523        // Only if this is successful it has to be removed
2524        try {
2525          this.fs.removeStoreFiles(this.getColumnFamilyDescriptor().getNameAsString(), filesToRemove);
2526        } catch (FailedArchiveException fae) {
2527          // Even if archiving some files failed, we still need to clear out any of the
2528          // files which were successfully archived.  Otherwise we will receive a
2529          // FileNotFoundException when we attempt to re-archive them in the next go around.
2530          Collection<Path> failedFiles = fae.getFailedFiles();
2531          Iterator<HStoreFile> iter = filesToRemove.iterator();
2532          while (iter.hasNext()) {
2533            if (failedFiles.contains(iter.next().getPath())) {
2534              iter.remove();
2535            }
2536          }
2537          if (!filesToRemove.isEmpty()) {
2538            clearCompactedfiles(filesToRemove);
2539          }
2540          throw fae;
2541        }
2542      }
2543    }
2544    if (!filesToRemove.isEmpty()) {
2545      // Clear the compactedfiles from the store file manager
2546      clearCompactedfiles(filesToRemove);
2547    }
2548  }
