123/**
124 * A Store holds a column family in a Region.  Its a memstore and a set of zero
125 * or more StoreFiles, which stretch backwards over time.
126 *
127 * <p>There's no reason to consider append-logging at this level; all logging
128 * and locking is handled at the HRegion level.  Store just provides
129 * services to manage sets of StoreFiles.  One of the most important of those
130 * services is compaction services where files are aggregated once they pass
131 * a configurable threshold.
132 *
133 * <p>Locking and transactions are handled at a higher level.  This API should
134 * not be called directly but by an HRegion manager.
135 */

239   * Constructor
240   * @param family HColumnDescriptor for this column
241   * @param confParam configuration object failed.  Can be null.
242   */
243  protected HStore(final HRegion region, final ColumnFamilyDescriptor family,
244      final Configuration confParam, boolean warmup) throws IOException {
245
246    // 'conf' renamed to 'confParam' b/c we use this.conf in the constructor
247    // CompoundConfiguration will look for keys in reverse order of addition, so we'd
248    // add global config first, then table and cf overrides, then cf metadata.
249    this.conf = new CompoundConfiguration()
250      .add(confParam)
251      .addBytesMap(region.getTableDescriptor().getValues())
252      .addStringMap(family.getConfiguration())
253      .addBytesMap(family.getValues());
254
255    this.region = region;
256    this.storeContext = initializeStoreContext(family);
257
258    // Assemble the store's home directory and Ensure it exists.
259    region.getRegionFileSystem().createStoreDir(family.getNameAsString());
260
261    // set block storage policy for store directory
262    String policyName = family.getStoragePolicy();
263    if (null == policyName) {
264      policyName = this.conf.get(BLOCK_STORAGE_POLICY_KEY, DEFAULT_BLOCK_STORAGE_POLICY);
265    }
266    region.getRegionFileSystem().setStoragePolicy(family.getNameAsString(), policyName.trim());
267
268    this.dataBlockEncoder = new HFileDataBlockEncoderImpl(family.getDataBlockEncoding());
269
270    // used by ScanQueryMatcher
271    long timeToPurgeDeletes =
272        Math.max(conf.getLong("hbase.hstore.time.to.purge.deletes", 0), 0);
273    LOG.trace("Time to purge deletes set to {}ms in {}", timeToPurgeDeletes, this);
274    // Get TTL
275    long ttl = determineTTLFromFamily(family);
276    // Why not just pass a HColumnDescriptor in here altogether?  Even if have
277    // to clone it?
278    scanInfo = new ScanInfo(conf, family, ttl, timeToPurgeDeletes, region.getCellComparator());
279    this.memstore = getMemstore();
280
281    this.offPeakHours = OffPeakHours.getInstance(conf);
282
283    this.verifyBulkLoads = conf.getBoolean("hbase.hstore.bulkload.verify", false);
284
285    this.blockingFileCount =
286        conf.getInt(BLOCKING_STOREFILES_KEY, DEFAULT_BLOCKING_STOREFILE_COUNT);
287    this.compactionCheckMultiplier = conf.getInt(
288        COMPACTCHECKER_INTERVAL_MULTIPLIER_KEY, DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER);
289    if (this.compactionCheckMultiplier <= 0) {
290      LOG.error("Compaction check period multiplier must be positive, setting default: {}",
291          DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER);
292      this.compactionCheckMultiplier = DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER;
293    }
294
295    this.storeEngine = createStoreEngine(this, this.conf, region.getCellComparator());
296    List<HStoreFile> hStoreFiles = loadStoreFiles(warmup);
297    // Move the storeSize calculation out of loadStoreFiles() method, because the secondary read
298    // replica's refreshStoreFiles() will also use loadStoreFiles() to refresh its store files and
299    // update the storeSize in the refreshStoreSizeAndTotalBytes() finally (just like compaction) , so
300    // no need calculate the storeSize twice.
301    this.storeSize.addAndGet(getStorefilesSize(hStoreFiles, sf -> true));
302    this.totalUncompressedBytes.addAndGet(getTotalUncompressedBytes(hStoreFiles));
303    this.storeEngine.getStoreFileManager().loadFiles(hStoreFiles);
304
305    flushRetriesNumber = conf.getInt(
306        "hbase.hstore.flush.retries.number", DEFAULT_FLUSH_RETRIES_NUMBER);
307    pauseTime = conf.getInt(HConstants.HBASE_SERVER_PAUSE, HConstants.DEFAULT_HBASE_SERVER_PAUSE);
308    if (flushRetriesNumber <= 0) {
309      throw new IllegalArgumentException(
310          "hbase.hstore.flush.retries.number must be > 0, not "
311              + flushRetriesNumber);
312    }
313
314    int confPrintThreshold =
315        this.conf.getInt("hbase.region.store.parallel.put.print.threshold", 50);
316    if (confPrintThreshold < 10) {
317      confPrintThreshold = 10;
318    }
319    this.parallelPutCountPrintThreshold = confPrintThreshold;
320
321    LOG.info("Store={},  memstore type={}, storagePolicy={}, verifyBulkLoads={}, "
322            + "parallelPutCountPrintThreshold={}, encoding={}, compression={}",
323        this, memstore.getClass().getSimpleName(), policyName, verifyBulkLoads,
324        parallelPutCountPrintThreshold, family.getDataBlockEncoding(),
325        family.getCompressionType());
326    cacheOnWriteLogged = false;
327  }

355  /**
356   * @return MemStore Instance to use in this store.
357   */
358  private MemStore getMemstore() {
359    MemStore ms = null;
360    // Check if in-memory-compaction configured. Note MemoryCompactionPolicy is an enum!
361    MemoryCompactionPolicy inMemoryCompaction = null;
362    if (this.getTableName().isSystemTable()) {
363      inMemoryCompaction = MemoryCompactionPolicy.valueOf(
364          conf.get("hbase.systemtables.compacting.memstore.type", "NONE"));
365    } else {
366      inMemoryCompaction = getColumnFamilyDescriptor().getInMemoryCompaction();
367    }
368    if (inMemoryCompaction == null) {
369      inMemoryCompaction =
370          MemoryCompactionPolicy.valueOf(conf.get(CompactingMemStore.COMPACTING_MEMSTORE_TYPE_KEY,
371              CompactingMemStore.COMPACTING_MEMSTORE_TYPE_DEFAULT).toUpperCase());
372    }
373    switch (inMemoryCompaction) {
374      case NONE:
375        ms = ReflectionUtils.newInstance(DefaultMemStore.class,
376            new Object[] { conf, getComparator(),
377                this.getHRegion().getRegionServicesForStores()});
378        break;
379      default:
380        Class<? extends CompactingMemStore> clz = conf.getClass(MEMSTORE_CLASS_NAME,
381            CompactingMemStore.class, CompactingMemStore.class);
382        ms = ReflectionUtils.newInstance(clz, new Object[]{conf, getComparator(), this,
383            this.getHRegion().getRegionServicesForStores(), inMemoryCompaction});
384    }
385    return ms;
386  }

413  /**
414   * @return TTL in seconds of the specified family
415   */
416  public static long determineTTLFromFamily(final ColumnFamilyDescriptor family) {
417    // HCD.getTimeToLive returns ttl in seconds.  Convert to milliseconds.
418    long ttl = family.getTimeToLive();
419    if (ttl == HConstants.FOREVER) {
420      // Default is unlimited ttl.
421      ttl = Long.MAX_VALUE;
422    } else if (ttl == -1) {
423      ttl = Long.MAX_VALUE;
424    } else {
425      // Second -> ms adjust for user data
426      ttl *= 1000;
427    }
428    return ttl;
429  }

554  private List<HStoreFile> openStoreFiles(Collection<StoreFileInfo> files, boolean warmup)
555      throws IOException {
556    if (CollectionUtils.isEmpty(files)) {
557      return Collections.emptyList();
558    }
559    // initialize the thread pool for opening store files in parallel..
560    ThreadPoolExecutor storeFileOpenerThreadPool =
561      this.region.getStoreFileOpenAndCloseThreadPool("StoreFileOpener-"
562        + this.region.getRegionInfo().getEncodedName() + "-" + this.getColumnFamilyName());
563    CompletionService<HStoreFile> completionService =
564      new ExecutorCompletionService<>(storeFileOpenerThreadPool);
565
566    int totalValidStoreFile = 0;
567    for (StoreFileInfo storeFileInfo : files) {
568      // open each store file in parallel
569      completionService.submit(() -> this.createStoreFileAndReader(storeFileInfo));
570      totalValidStoreFile++;
571    }
572
573    Set<String> compactedStoreFiles = new HashSet<>();
574    ArrayList<HStoreFile> results = new ArrayList<>(files.size());
575    IOException ioe = null;
576    try {
577      for (int i = 0; i < totalValidStoreFile; i++) {
578        try {
579          HStoreFile storeFile = completionService.take().get();
580          if (storeFile != null) {
581            LOG.debug("loaded {}", storeFile);
582            results.add(storeFile);
583            compactedStoreFiles.addAll(storeFile.getCompactedStoreFiles());
584          }
585        } catch (InterruptedException e) {
586          if (ioe == null) {
587            ioe = new InterruptedIOException(e.getMessage());
588          }
589        } catch (ExecutionException e) {
590          if (ioe == null) {
591            ioe = new IOException(e.getCause());
592          }
593        }
594      }
595    } finally {
596      storeFileOpenerThreadPool.shutdownNow();
597    }
598    if (ioe != null) {
599      // close StoreFile readers
600      boolean evictOnClose =
601          getCacheConfig() != null? getCacheConfig().shouldEvictOnClose(): true;
602      for (HStoreFile file : results) {
603        try {
604          if (file != null) {
605            file.closeStoreFile(evictOnClose);
606          }
607        } catch (IOException e) {
608          LOG.warn("Could not close store file {}", file, e);
609        }
610      }
611      throw ioe;
612    }
613
614    // Should not archive the compacted store files when region warmup. See HBASE-22163.
615    if (!warmup) {
616      // Remove the compacted files from result
617      List<HStoreFile> filesToRemove = new ArrayList<>(compactedStoreFiles.size());
618      for (HStoreFile storeFile : results) {
619        if (compactedStoreFiles.contains(storeFile.getPath().getName())) {
620          LOG.warn("Clearing the compacted storefile {} from {}", storeFile, this);
621          storeFile.getReader().close(storeFile.getCacheConf() != null ?
622            storeFile.getCacheConf().shouldEvictOnClose() : true);
623          filesToRemove.add(storeFile);
624        }
625      }
626      results.removeAll(filesToRemove);
627      if (!filesToRemove.isEmpty() && this.isPrimaryReplicaStore()) {
628        LOG.debug("Moving the files {} to archive", filesToRemove);
629        getRegionFileSystem().removeStoreFiles(this.getColumnFamilyDescriptor().getNameAsString(),
630            filesToRemove);
631      }
632    }
633
634    return results;
635  }

655  /**
656   * Checks the underlying store files, and opens the files that  have not
657   * been opened, and removes the store file readers for store files no longer
658   * available. Mainly used by secondary region replicas to keep up to date with
659   * the primary region files.
660   */
661  private void refreshStoreFilesInternal(Collection<StoreFileInfo> newFiles) throws IOException {
662    StoreFileManager sfm = storeEngine.getStoreFileManager();
663    Collection<HStoreFile> currentFiles = sfm.getStorefiles();
664    Collection<HStoreFile> compactedFiles = sfm.getCompactedfiles();
665    if (currentFiles == null) {
666      currentFiles = Collections.emptySet();
667    }
668    if (newFiles == null) {
669      newFiles = Collections.emptySet();
670    }
671    if (compactedFiles == null) {
672      compactedFiles = Collections.emptySet();
673    }
674
675    HashMap<StoreFileInfo, HStoreFile> currentFilesSet = new HashMap<>(currentFiles.size());
676    for (HStoreFile sf : currentFiles) {
677      currentFilesSet.put(sf.getFileInfo(), sf);
678    }
679    HashMap<StoreFileInfo, HStoreFile> compactedFilesSet = new HashMap<>(compactedFiles.size());
680    for (HStoreFile sf : compactedFiles) {
681      compactedFilesSet.put(sf.getFileInfo(), sf);
682    }
683
684    Set<StoreFileInfo> newFilesSet = new HashSet<StoreFileInfo>(newFiles);
685    // Exclude the files that have already been compacted
686    newFilesSet = Sets.difference(newFilesSet, compactedFilesSet.keySet());
687    Set<StoreFileInfo> toBeAddedFiles = Sets.difference(newFilesSet, currentFilesSet.keySet());
688    Set<StoreFileInfo> toBeRemovedFiles = Sets.difference(currentFilesSet.keySet(), newFilesSet);
689
690    if (toBeAddedFiles.isEmpty() && toBeRemovedFiles.isEmpty()) {
691      return;
692    }
693
694    LOG.info("Refreshing store files for " + this + " files to add: "
695      + toBeAddedFiles + " files to remove: " + toBeRemovedFiles);
696
697    Set<HStoreFile> toBeRemovedStoreFiles = new HashSet<>(toBeRemovedFiles.size());
698    for (StoreFileInfo sfi : toBeRemovedFiles) {
699      toBeRemovedStoreFiles.add(currentFilesSet.get(sfi));
700    }
701
702    // try to open the files
703    List<HStoreFile> openedFiles = openStoreFiles(toBeAddedFiles, false);
704
705    // propogate the file changes to the underlying store file manager
706    replaceStoreFiles(toBeRemovedStoreFiles, openedFiles); //won't throw an exception
707
708    // Advance the memstore read point to be at least the new store files seqIds so that
709    // readers might pick it up. This assumes that the store is not getting any writes (otherwise
710    // in-flight transactions might be made visible)
711    if (!toBeAddedFiles.isEmpty()) {
712      // we must have the max sequence id here as we do have several store files
713      region.getMVCC().advanceTo(this.getMaxSequenceId().getAsLong());
714    }
715
716    refreshStoreSizeAndTotalBytes();
717  }

801  /**
802   * This throws a WrongRegionException if the HFile does not fit in this region, or an
803   * InvalidHFileException if the HFile is not valid.
804   */
805  public void assertBulkLoadHFileOk(Path srcPath) throws IOException {
806    HFile.Reader reader  = null;
807    try {
808      LOG.info("Validating hfile at " + srcPath + " for inclusion in " + this);
809      FileSystem srcFs = srcPath.getFileSystem(conf);
810      srcFs.access(srcPath, FsAction.READ_WRITE);
811      reader = HFile.createReader(srcFs, srcPath, getCacheConfig(), isPrimaryReplicaStore(), conf);
812
813      Optional<byte[]> firstKey = reader.getFirstRowKey();
814      Preconditions.checkState(firstKey.isPresent(), "First key can not be null");
815      Optional<Cell> lk = reader.getLastKey();
816      Preconditions.checkState(lk.isPresent(), "Last key can not be null");
817      byte[] lastKey =  CellUtil.cloneRow(lk.get());
818
819      if (LOG.isDebugEnabled()) {
820        LOG.debug("HFile bounds: first=" + Bytes.toStringBinary(firstKey.get()) +
821            " last=" + Bytes.toStringBinary(lastKey));
822        LOG.debug("Region bounds: first=" +
823            Bytes.toStringBinary(getRegionInfo().getStartKey()) +
824            " last=" + Bytes.toStringBinary(getRegionInfo().getEndKey()));
825      }
826
827      if (!this.getRegionInfo().containsRange(firstKey.get(), lastKey)) {
828        throw new WrongRegionException(
829            "Bulk load file " + srcPath.toString() + " does not fit inside region "
830            + this.getRegionInfo().getRegionNameAsString());
831      }
832
833      if(reader.length() > conf.getLong(HConstants.HREGION_MAX_FILESIZE,
834          HConstants.DEFAULT_MAX_FILE_SIZE)) {
835        LOG.warn("Trying to bulk load hfile " + srcPath + " with size: " +
836            reader.length() + " bytes can be problematic as it may lead to oversplitting.");
837      }
838
839      if (verifyBulkLoads) {
840        long verificationStartTime = EnvironmentEdgeManager.currentTime();
841        LOG.info("Full verification started for bulk load hfile: {}", srcPath);
842        Cell prevCell = null;
843        HFileScanner scanner = reader.getScanner(false, false, false);
844        scanner.seekTo();
845        do {
846          Cell cell = scanner.getCell();
847          if (prevCell != null) {
848            if (getComparator().compareRows(prevCell, cell) > 0) {
849              throw new InvalidHFileException("Previous row is greater than"
850                  + " current row: path=" + srcPath + " previous="
851                  + CellUtil.getCellKeyAsString(prevCell) + " current="
852                  + CellUtil.getCellKeyAsString(cell));
853            }
854            if (CellComparator.getInstance().compareFamilies(prevCell, cell) != 0) {
855              throw new InvalidHFileException("Previous key had different"
856                  + " family compared to current key: path=" + srcPath
857                  + " previous="
858                  + Bytes.toStringBinary(prevCell.getFamilyArray(), prevCell.getFamilyOffset(),
859                      prevCell.getFamilyLength())
860                  + " current="
861                  + Bytes.toStringBinary(cell.getFamilyArray(), cell.getFamilyOffset(),
862                      cell.getFamilyLength()));
863            }
864          }
865          prevCell = cell;
866        } while (scanner.next());
867        LOG.info("Full verification complete for bulk load hfile: " + srcPath.toString() +
868          " took " + (EnvironmentEdgeManager.currentTime() - verificationStartTime) + " ms");
869      }
870    } finally {
871      if (reader != null) {
872        reader.close();
873      }
874    }
875  }

941  /**
942   * Close all the readers We don't need to worry about subsequent requests because the Region holds
943   * a write lock that will prevent any more reads or writes.
944   * @return the {@link StoreFile StoreFiles} that were previously being used.
945   * @throws IOException on failure
946   */
947  public ImmutableCollection<HStoreFile> close() throws IOException {
948    this.archiveLock.lock();
949    this.lock.writeLock().lock();
950    try {
951      // Clear so metrics doesn't find them.
952      ImmutableCollection<HStoreFile> result = storeEngine.getStoreFileManager().clearFiles();
953      Collection<HStoreFile> compactedfiles =
954          storeEngine.getStoreFileManager().clearCompactedFiles();
955      // clear the compacted files
956      if (CollectionUtils.isNotEmpty(compactedfiles)) {
957        removeCompactedfiles(compactedfiles, getCacheConfig() != null ?
958            getCacheConfig().shouldEvictOnClose() : true);
959      }
960      if (!result.isEmpty()) {
961        // initialize the thread pool for closing store files in parallel.
962        ThreadPoolExecutor storeFileCloserThreadPool = this.region
963            .getStoreFileOpenAndCloseThreadPool("StoreFileCloser-"
964              + this.region.getRegionInfo().getEncodedName() + "-" + this.getColumnFamilyName());
965
966        // close each store file in parallel
967        CompletionService<Void> completionService =
968          new ExecutorCompletionService<>(storeFileCloserThreadPool);
969        for (HStoreFile f : result) {
970          completionService.submit(new Callable<Void>() {
971            @Override
972            public Void call() throws IOException {
973              boolean evictOnClose =
974                  getCacheConfig() != null? getCacheConfig().shouldEvictOnClose(): true;
975              f.closeStoreFile(evictOnClose);
976              return null;
977            }
978          });
979        }
980
981        IOException ioe = null;
982        try {
983          for (int i = 0; i < result.size(); i++) {
984            try {
985              Future<Void> future = completionService.take();
986              future.get();
987            } catch (InterruptedException e) {
988              if (ioe == null) {
989                ioe = new InterruptedIOException();
990                ioe.initCause(e);
991              }
992            } catch (ExecutionException e) {
993              if (ioe == null) {
994                ioe = new IOException(e.getCause());
995              }
996            }
997          }
998        } finally {
999          storeFileCloserThreadPool.shutdownNow();
1000        }
1001        if (ioe != null) {
1002          throw ioe;
1003        }
1004      }
1005      LOG.trace("Closed {}", this);
1006      return result;
1007    } finally {
1008      this.lock.writeLock().unlock();
1009      this.archiveLock.unlock();
1010    }
1011  }

1028  /**
1029   * Write out current snapshot. Presumes {@link #snapshot()} has been called previously.
1030   * @param logCacheFlushId flush sequence number
1031   * @return The path name of the tmp file to which the store was flushed
1032   * @throws IOException if exception occurs during process
1033   */
1034  protected List<Path> flushCache(final long logCacheFlushId, MemStoreSnapshot snapshot,
1035      MonitoredTask status, ThroughputController throughputController,
1036      FlushLifeCycleTracker tracker) throws IOException {
1037    // If an exception happens flushing, we let it out without clearing
1038    // the memstore snapshot.  The old snapshot will be returned when we say
1039    // 'snapshot', the next time flush comes around.
1040    // Retry after catching exception when flushing, otherwise server will abort
1041    // itself
1042    StoreFlusher flusher = storeEngine.getStoreFlusher();
1043    IOException lastException = null;
1044    for (int i = 0; i < flushRetriesNumber; i++) {
1045      try {
1046        List<Path> pathNames =
1047            flusher.flushSnapshot(snapshot, logCacheFlushId, status, throughputController, tracker);
1048        Path lastPathName = null;
1049        try {
1050          for (Path pathName : pathNames) {
1051            lastPathName = pathName;
1052            validateStoreFile(pathName);
1053          }
1054          return pathNames;
1055        } catch (Exception e) {
1056          LOG.warn("Failed validating store file {}, retrying num={}", lastPathName, i, e);
1057          if (e instanceof IOException) {
1058            lastException = (IOException) e;
1059          } else {
1060            lastException = new IOException(e);
1061          }
1062        }
1063      } catch (IOException e) {
1064        LOG.warn("Failed flushing store file for {}, retrying num={}", this, i, e);
1065        lastException = e;
1066      }
1067      if (lastException != null && i < (flushRetriesNumber - 1)) {
1068        try {
1069          Thread.sleep(pauseTime);
1070        } catch (InterruptedException e) {
1071          IOException iie = new InterruptedIOException();
1072          iie.initCause(e);
1073          throw iie;
1074        }
1075      }
1076    }
1077    throw lastException;
1078  }

1147   * @param compression Compression algorithm to use
1148   * @param isCompaction whether we are creating a new file in a compaction
1149   * @param includeMVCCReadpoint - whether to include MVCC or not
1150   * @param includesTag - includesTag or not
1151   * @return Writer for a new StoreFile in the tmp dir.
1152   */
1153  // TODO : allow the Writer factory to create Writers of ShipperListener type only in case of
1154  // compaction
1155  public StoreFileWriter createWriterInTmp(long maxKeyCount, Compression.Algorithm compression,
1156      boolean isCompaction, boolean includeMVCCReadpoint, boolean includesTag,
1157      boolean shouldDropBehind, long totalCompactedFilesSize, String fileStoragePolicy)
1158        throws IOException {
1159    // creating new cache config for each new writer
1160    final CacheConfig cacheConf = getCacheConfig();
1161    final CacheConfig writerCacheConf = new CacheConfig(cacheConf);
1162    if (isCompaction) {
1163      // Don't cache data on write on compactions, unless specifically configured to do so
1164      // Cache only when total file size remains lower than configured threshold
1165      final boolean cacheCompactedBlocksOnWrite =
1166        getCacheConfig().shouldCacheCompactedBlocksOnWrite();
1167      // if data blocks are to be cached on write
1168      // during compaction, we should forcefully
1169      // cache index and bloom blocks as well
1170      if (cacheCompactedBlocksOnWrite && totalCompactedFilesSize <= cacheConf
1171        .getCacheCompactedBlocksOnWriteThreshold()) {
1172        writerCacheConf.enableCacheOnWrite();
1173        if (!cacheOnWriteLogged) {
1174          LOG.info("For {} , cacheCompactedBlocksOnWrite is true, hence enabled " +
1175              "cacheOnWrite for Data blocks, Index blocks and Bloom filter blocks", this);
1176          cacheOnWriteLogged = true;
1177        }
1178      } else {
1179        writerCacheConf.setCacheDataOnWrite(false);
1180        if (totalCompactedFilesSize > cacheConf.getCacheCompactedBlocksOnWriteThreshold()) {
1181          // checking condition once again for logging
1182          LOG.debug(
1183            "For {}, setting cacheCompactedBlocksOnWrite as false as total size of compacted "
1184              + "files - {}, is greater than cacheCompactedBlocksOnWriteThreshold - {}",
1185            this, totalCompactedFilesSize,
1186            cacheConf.getCacheCompactedBlocksOnWriteThreshold());
1187        }
1188      }
1189    } else {
1190      final boolean shouldCacheDataOnWrite = cacheConf.shouldCacheDataOnWrite();
1191      if (shouldCacheDataOnWrite) {
1192        writerCacheConf.enableCacheOnWrite();
1193        if (!cacheOnWriteLogged) {
1194          LOG.info("For {} , cacheDataOnWrite is true, hence enabled cacheOnWrite for " +
1195            "Index blocks and Bloom filter blocks", this);
1196          cacheOnWriteLogged = true;
1197        }
1198      }
1199    }

1456  /**
1457   * Compact the StoreFiles.  This method may take some time, so the calling
1458   * thread must be able to block for long periods.
1459   *
1460   * <p>During this time, the Store can work as usual, getting values from
1461   * StoreFiles and writing new StoreFiles from the memstore.
1462   *
1463   * Existing StoreFiles are not destroyed until the new compacted StoreFile is
1464   * completely written-out to disk.
1465   *
1466   * <p>The compactLock prevents multiple simultaneous compactions.
1467   * The structureLock prevents us from interfering with other write operations.
1468   *
1469   * <p>We don't want to hold the structureLock for the whole time, as a compact()
1470   * can be lengthy and we want to allow cache-flushes during this period.
1471   *
1472   * <p> Compaction event should be idempotent, since there is no IO Fencing for
1473   * the region directory in hdfs. A region server might still try to complete the
1474   * compaction after it lost the region. That is why the following events are carefully
1475   * ordered for a compaction:
1476   *  1. Compaction writes new files under region/.tmp directory (compaction output)
1477   *  2. Compaction atomically moves the temporary file under region directory
1478   *  3. Compaction appends a WAL edit containing the compaction input and output files.
1479   *  Forces sync on WAL.
1480   *  4. Compaction deletes the input files from the region directory.
1481   *
1482   * Failure conditions are handled like this:
1483   *  - If RS fails before 2, compaction wont complete. Even if RS lives on and finishes
1484   *  the compaction later, it will only write the new data file to the region directory.
1485   *  Since we already have this data, this will be idempotent but we will have a redundant
1486   *  copy of the data.
1487   *  - If RS fails between 2 and 3, the region will have a redundant copy of the data. The
1488   *  RS that failed won't be able to finish sync() for WAL because of lease recovery in WAL.
1489   *  - If RS fails after 3, the region region server who opens the region will pick up the
1490   *  the compaction marker from the WAL and replay it by removing the compaction input files.
1491   *  Failed RS can also attempt to delete those files, but the operation will be idempotent
1492   *
1493   * See HBASE-2231 for details.
1494   *
1495   * @param compaction compaction details obtained from requestCompaction()
1496   * @return Storefile we compacted into or null if we failed or opted out early.
1497   */
1498  public List<HStoreFile> compact(CompactionContext compaction,
1499    ThroughputController throughputController, User user) throws IOException {
1500    assert compaction != null;
1501    CompactionRequestImpl cr = compaction.getRequest();
1502    try {
1503      // Do all sanity checking in here if we have a valid CompactionRequestImpl
1504      // because we need to clean up after it on the way out in a finally
1505      // block below
1506      long compactionStartTime = EnvironmentEdgeManager.currentTime();
1507      assert compaction.hasSelection();
1508      Collection<HStoreFile> filesToCompact = cr.getFiles();
1509      assert !filesToCompact.isEmpty();
1510      synchronized (filesCompacting) {
1511        // sanity check: we're compacting files that this store knows about
1512        // TODO: change this to LOG.error() after more debugging
1513        Preconditions.checkArgument(filesCompacting.containsAll(filesToCompact));
1514      }
1515
1516      // Ready to go. Have list of files to compact.
1517      LOG.info("Starting compaction of " + filesToCompact +
1518        " into tmpdir=" + getRegionFileSystem().getTempDir() + ", totalSize=" +
1519          TraditionalBinaryPrefix.long2String(cr.getSize(), "", 1));
1520
1521      return doCompaction(cr, filesToCompact, user, compactionStartTime,
1522          compaction.compact(throughputController, user));
1523    } finally {
1524      finishCompactionRequest(cr);
1525    }
1526  }

1528  protected List<HStoreFile> doCompaction(CompactionRequestImpl cr,
1529      Collection<HStoreFile> filesToCompact, User user, long compactionStartTime,
1530      List<Path> newFiles) throws IOException {
1531    // Do the steps necessary to complete the compaction.
1532    setStoragePolicyFromFileName(newFiles);
1533    List<HStoreFile> sfs = moveCompactedFilesIntoPlace(cr, newFiles, user);
1534    writeCompactionWalRecord(filesToCompact, sfs);
1535    replaceStoreFiles(filesToCompact, sfs);
1536    if (cr.isMajor()) {
1537      majorCompactedCellsCount.addAndGet(getCompactionProgress().getTotalCompactingKVs());
1538      majorCompactedCellsSize.addAndGet(getCompactionProgress().totalCompactedSize);
1539    } else {
1540      compactedCellsCount.addAndGet(getCompactionProgress().getTotalCompactingKVs());
1541      compactedCellsSize.addAndGet(getCompactionProgress().totalCompactedSize);
1542    }
1543    long outputBytes = getTotalSize(sfs);
1544
1545    // At this point the store will use new files for all new scanners.
1546    refreshStoreSizeAndTotalBytes(); // update store size.
1547
1548    long now = EnvironmentEdgeManager.currentTime();
1549    if (region.getRegionServerServices() != null
1550        && region.getRegionServerServices().getMetrics() != null) {
1551      region.getRegionServerServices().getMetrics().updateCompaction(
1552          region.getTableDescriptor().getTableName().getNameAsString(),
1553          cr.isMajor(), now - compactionStartTime, cr.getFiles().size(),
1554          newFiles.size(), cr.getSize(), outputBytes);
1555
1556    }
1557
1558    logCompactionEndMessage(cr, sfs, now, compactionStartTime);
1559    return sfs;
1560  }

1713  /**
1714   * Call to complete a compaction. Its for the case where we find in the WAL a compaction
1715   * that was not finished.  We could find one recovering a WAL after a regionserver crash.
1716   * See HBASE-2231.
1717   */
1718  public void replayCompactionMarker(CompactionDescriptor compaction, boolean pickCompactionFiles,
1719      boolean removeFiles) throws IOException {
1720    LOG.debug("Completing compaction from the WAL marker");
1721    List<String> compactionInputs = compaction.getCompactionInputList();
1722    List<String> compactionOutputs = Lists.newArrayList(compaction.getCompactionOutputList());
1723
1724    // The Compaction Marker is written after the compaction is completed,
1725    // and the files moved into the region/family folder.
1726    //
1727    // If we crash after the entry is written, we may not have removed the
1728    // input files, but the output file is present.
1729    // (The unremoved input files will be removed by this function)
1730    //
1731    // If we scan the directory and the file is not present, it can mean that:
1732    //   - The file was manually removed by the user
1733    //   - The file was removed as consequence of subsequent compaction
1734    // so, we can't do anything with the "compaction output list" because those
1735    // files have already been loaded when opening the region (by virtue of
1736    // being in the store's folder) or they may be missing due to a compaction.
1737
1738    String familyName = this.getColumnFamilyName();
1739    Set<String> inputFiles = new HashSet<>();
1740    for (String compactionInput : compactionInputs) {
1741      Path inputPath = getRegionFileSystem().getStoreFilePath(familyName, compactionInput);
1742      inputFiles.add(inputPath.getName());
1743    }
1744
1745    //some of the input files might already be deleted
1746    List<HStoreFile> inputStoreFiles = new ArrayList<>(compactionInputs.size());
1747    for (HStoreFile sf : this.getStorefiles()) {
1748      if (inputFiles.contains(sf.getPath().getName())) {
1749        inputStoreFiles.add(sf);
1750      }
1751    }
1752
1753    // check whether we need to pick up the new files
1754    List<HStoreFile> outputStoreFiles = new ArrayList<>(compactionOutputs.size());
1755
1756    if (pickCompactionFiles) {
1757      for (HStoreFile sf : this.getStorefiles()) {
1758        compactionOutputs.remove(sf.getPath().getName());
1759      }
1760      for (String compactionOutput : compactionOutputs) {
1761        StoreFileInfo storeFileInfo =
1762            getRegionFileSystem().getStoreFileInfo(getColumnFamilyName(), compactionOutput);
1763        HStoreFile storeFile = createStoreFileAndReader(storeFileInfo);
1764        outputStoreFiles.add(storeFile);
1765      }
1766    }
1767
1768    if (!inputStoreFiles.isEmpty() || !outputStoreFiles.isEmpty()) {
1769      LOG.info("Replaying compaction marker, replacing input files: " +
1770          inputStoreFiles + " with output files : " + outputStoreFiles);
1771      this.replaceStoreFiles(inputStoreFiles, outputStoreFiles);
1772      this.refreshStoreSizeAndTotalBytes();
1773    }
1774  }

1776  /**
1777   * This method tries to compact N recent files for testing.
1778   * Note that because compacting "recent" files only makes sense for some policies,
1779   * e.g. the default one, it assumes default policy is used. It doesn't use policy,
1780   * but instead makes a compaction candidate list by itself.
1781   * @param N Number of files.
1782   */
1783  public void compactRecentForTestingAssumingDefaultPolicy(int N) throws IOException {
1784    List<HStoreFile> filesToCompact;
1785    boolean isMajor;
1786
1787    this.lock.readLock().lock();
1788    try {
1789      synchronized (filesCompacting) {
1790        filesToCompact = Lists.newArrayList(storeEngine.getStoreFileManager().getStorefiles());
1791        if (!filesCompacting.isEmpty()) {
1792          // exclude all files older than the newest file we're currently
1793          // compacting. this allows us to preserve contiguity (HBASE-2856)
1794          HStoreFile last = filesCompacting.get(filesCompacting.size() - 1);
1795          int idx = filesToCompact.indexOf(last);
1796          Preconditions.checkArgument(idx != -1);
1797          filesToCompact.subList(0, idx + 1).clear();
1798        }
1799        int count = filesToCompact.size();
1800        if (N > count) {
1801          throw new RuntimeException("Not enough files");
1802        }
1803
1804        filesToCompact = filesToCompact.subList(count - N, count);
1805        isMajor = (filesToCompact.size() == storeEngine.getStoreFileManager().getStorefileCount());
1806        filesCompacting.addAll(filesToCompact);
1807        Collections.sort(filesCompacting, storeEngine.getStoreFileManager()
1808            .getStoreFileComparator());
1809      }
1810    } finally {
1811      this.lock.readLock().unlock();
1812    }
1813
1814    try {
1815      // Ready to go. Have list of files to compact.
1816      List<Path> newFiles = ((DefaultCompactor)this.storeEngine.getCompactor())
1817          .compactForTesting(filesToCompact, isMajor);
1818      for (Path newFile: newFiles) {
1819        // Move the compaction into place.
1820        HStoreFile sf = moveFileIntoPlace(newFile);
1821        if (this.getCoprocessorHost() != null) {
1822          this.getCoprocessorHost().postCompact(this, sf, null, null, null);
1823        }
1824        replaceStoreFiles(filesToCompact, Collections.singletonList(sf));
1825        refreshStoreSizeAndTotalBytes();
1826      }
1827    } finally {
1828      synchronized (filesCompacting) {
1829        filesCompacting.removeAll(filesToCompact);
1830      }
1831    }
1832  }

1874  public Optional<CompactionContext> requestCompaction(int priority,
1875      CompactionLifeCycleTracker tracker, User user) throws IOException {
1876    // don't even select for compaction if writes are disabled
1877    if (!this.areWritesEnabled()) {
1878      return Optional.empty();
1879    }
1880    // Before we do compaction, try to get rid of unneeded files to simplify things.
1881    removeUnneededFiles();
1882
1883    final CompactionContext compaction = storeEngine.createCompaction();
1884    CompactionRequestImpl request = null;
1885    this.lock.readLock().lock();
1886    try {
1887      synchronized (filesCompacting) {
1888        // First, see if coprocessor would want to override selection.
1889        if (this.getCoprocessorHost() != null) {
1890          final List<HStoreFile> candidatesForCoproc = compaction.preSelect(this.filesCompacting);
1891          boolean override = getCoprocessorHost().preCompactSelection(this,
1892              candidatesForCoproc, tracker, user);
1893          if (override) {
1894            // Coprocessor is overriding normal file selection.
1895            compaction.forceSelect(new CompactionRequestImpl(candidatesForCoproc));
1896          }
1897        }
1898
1899        // Normal case - coprocessor is not overriding file selection.
1900        if (!compaction.hasSelection()) {
1901          boolean isUserCompaction = priority == Store.PRIORITY_USER;
1902          boolean mayUseOffPeak = offPeakHours.isOffPeakHour() &&
1903              offPeakCompactionTracker.compareAndSet(false, true);
1904          try {
1905            compaction.select(this.filesCompacting, isUserCompaction,
1906              mayUseOffPeak, forceMajor && filesCompacting.isEmpty());
1907          } catch (IOException e) {
1908            if (mayUseOffPeak) {
1909              offPeakCompactionTracker.set(false);
1910            }
1911            throw e;
1912          }
1913          assert compaction.hasSelection();
1914          if (mayUseOffPeak && !compaction.getRequest().isOffPeak()) {
1915            // Compaction policy doesn't want to take advantage of off-peak.
1916            offPeakCompactionTracker.set(false);
1917          }
1918        }
1919        if (this.getCoprocessorHost() != null) {
1920          this.getCoprocessorHost().postCompactSelection(
1921              this, ImmutableList.copyOf(compaction.getRequest().getFiles()), tracker,
1922              compaction.getRequest(), user);
1923        }
1924        // Finally, we have the resulting files list. Check if we have any files at all.
1925        request = compaction.getRequest();
1926        Collection<HStoreFile> selectedFiles = request.getFiles();
1927        if (selectedFiles.isEmpty()) {
1928          return Optional.empty();
1929        }
1930
1931        addToCompactingFiles(selectedFiles);
1932
1933        // If we're enqueuing a major, clear the force flag.
1934        this.forceMajor = this.forceMajor && !request.isMajor();
1935
1936        // Set common request properties.
1937        // Set priority, either override value supplied by caller or from store.
1938        final int compactionPriority =
1939          (priority != Store.NO_PRIORITY) ? priority : getCompactPriority();
1940        request.setPriority(compactionPriority);
1941
1942        if (request.isAfterSplit()) {
1943          // If the store belongs to recently splitted daughter regions, better we consider
1944          // them with the higher priority in the compaction queue.
1945          // Override priority if it is lower (higher int value) than
1946          // SPLIT_REGION_COMPACTION_PRIORITY
1947          final int splitHousekeepingPriority =
1948            Math.min(compactionPriority, SPLIT_REGION_COMPACTION_PRIORITY);
1949          request.setPriority(splitHousekeepingPriority);
1950          LOG.info("Keeping/Overriding Compaction request priority to {} for CF {} since it"
1951              + " belongs to recently split daughter region {}", splitHousekeepingPriority,
1952            this.getColumnFamilyName(), getRegionInfo().getRegionNameAsString());
1953        }
1954        request.setDescription(getRegionInfo().getRegionNameAsString(), getColumnFamilyName());
1955        request.setTracker(tracker);
1956      }
1957    } finally {
1958      this.lock.readLock().unlock();
1959    }
1960
1961    if (LOG.isDebugEnabled()) {
1962      LOG.debug(this + " is initiating " + (request.isMajor() ? "major" : "minor") + " compaction"
1963          + (request.isAllFiles() ? " (all files)" : ""));
1964    }
1965    this.region.reportCompactionRequestStart(request.isMajor());
1966    return Optional.of(compaction);
1967  }

1969  /** Adds the files to compacting files. filesCompacting must be locked. */
1970  private void addToCompactingFiles(Collection<HStoreFile> filesToAdd) {
1971    if (CollectionUtils.isEmpty(filesToAdd)) {
1972      return;
1973    }
1974    // Check that we do not try to compact the same StoreFile twice.
1975    if (!Collections.disjoint(filesCompacting, filesToAdd)) {
1976      Preconditions.checkArgument(false, "%s overlaps with %s", filesToAdd, filesCompacting);
1977    }
1978    filesCompacting.addAll(filesToAdd);
1979    Collections.sort(filesCompacting, storeEngine.getStoreFileManager().getStoreFileComparator());
1980  }

1982  private void removeUnneededFiles() throws IOException {
1983    if (!conf.getBoolean("hbase.store.delete.expired.storefile", true)) {
1984      return;
1985    }
1986    if (getColumnFamilyDescriptor().getMinVersions() > 0) {
1987      LOG.debug("Skipping expired store file removal due to min version of {} being {}",
1988          this, getColumnFamilyDescriptor().getMinVersions());
1989      return;
1990    }
1991    this.lock.readLock().lock();
1992    Collection<HStoreFile> delSfs = null;
1993    try {
1994      synchronized (filesCompacting) {
1995        long cfTtl = getStoreFileTtl();
1996        if (cfTtl != Long.MAX_VALUE) {
1997          delSfs = storeEngine.getStoreFileManager().getUnneededFiles(
1998              EnvironmentEdgeManager.currentTime() - cfTtl, filesCompacting);
1999          addToCompactingFiles(delSfs);
2000        }
2001      }
2002    } finally {
2003      this.lock.readLock().unlock();
2004    }
2005
2006    if (CollectionUtils.isEmpty(delSfs)) {
2007      return;
2008    }
2009
2010    Collection<HStoreFile> newFiles = Collections.emptyList(); // No new files.
2011    writeCompactionWalRecord(delSfs, newFiles);
2012    replaceStoreFiles(delSfs, newFiles);
2013    refreshStoreSizeAndTotalBytes();
2014    LOG.info("Completed removal of " + delSfs.size() + " unnecessary (expired) file(s) in "
2015        + this + "; total size is "
2016        + TraditionalBinaryPrefix.long2String(storeSize.get(), "", 1));
2017  }

2408    /**

2412    @Override

2422    @Override

2425      ThroughputController throughputController =
2426          rsService == null ? null : rsService.getFlushThroughputController();

2431    @Override
2432    public boolean commit(MonitoredTask status) throws IOException {
2433      if (CollectionUtils.isEmpty(this.tempFiles)) {
2434        return false;
2435      }
2436      List<HStoreFile> storeFiles = new ArrayList<>(this.tempFiles.size());
2437      for (Path storeFilePath : tempFiles) {
2438        try {
2439          HStoreFile sf = HStore.this.commitFile(storeFilePath, cacheFlushSeqNum, status);
2440          outputFileSize += sf.getReader().length();
2441          storeFiles.add(sf);
2442        } catch (IOException ex) {
2443          LOG.error("Failed to commit store file {}", storeFilePath, ex);
2444          // Try to delete the files we have committed before.
2445          for (HStoreFile sf : storeFiles) {
2446            Path pathToDelete = sf.getPath();
2447            try {
2448              sf.deleteStoreFile();
2449            } catch (IOException deleteEx) {
2450              LOG.error(HBaseMarkers.FATAL, "Failed to delete store file we committed, "
2451                  + "halting {}", pathToDelete, ex);
2452              Runtime.getRuntime().halt(1);
2453            }
2454          }
2455          throw new IOException("Failed to commit the flush", ex);
2456        }
2457      }
2458
2459      for (HStoreFile sf : storeFiles) {
2460        if (HStore.this.getCoprocessorHost() != null) {
2461          HStore.this.getCoprocessorHost().postFlush(HStore.this, sf, tracker);
2462        }
2463        committedFiles.add(sf.getPath());
2464      }
2465
2466      HStore.this.flushedCellsCount.addAndGet(cacheFlushCount);
2467      HStore.this.flushedCellsSize.addAndGet(cacheFlushSize);
2468      HStore.this.flushedOutputFileSize.addAndGet(outputFileSize);
2469
2470      // Add new file to store files.  Clear snapshot too while we have the Store write lock.
2471      return HStore.this.updateStorefiles(storeFiles, snapshot.getId());
2472    }

2485     * Similar to commit, but called in secondary region replicas for replaying the
2486     * flush cache from primary region. Adds the new files to the store, and drops the
2487     * snapshot depending on dropMemstoreSnapshot argument.
2488     * @param fileNames names of the flushed files
2489     * @param dropMemstoreSnapshot whether to drop the prepared memstore snapshot

2493        throws IOException {
2494      List<HStoreFile> storeFiles = new ArrayList<>(fileNames.size());
2495      for (String file : fileNames) {
2496        // open the file as a store file (hfile link, etc)
2497        StoreFileInfo storeFileInfo =
2498          getRegionFileSystem().getStoreFileInfo(getColumnFamilyName(), file);

2500        storeFiles.add(storeFile);
2501        HStore.this.storeSize.addAndGet(storeFile.getReader().length());
2502        HStore.this.totalUncompressedBytes
2503            .addAndGet(storeFile.getReader().getTotalUncompressedBytes());
2504        if (LOG.isInfoEnabled()) {
2505          LOG.info(this + " added " + storeFile + ", entries=" + storeFile.getReader().getEntries() +
2506              ", sequenceid=" + storeFile.getReader().getSequenceID() + ", filesize="
2507              + TraditionalBinaryPrefix.long2String(storeFile.getReader().length(), "", 1));
2508        }
2509      }
2510
2511      long snapshotId = -1; // -1 means do not drop
2512      if (dropMemstoreSnapshot && snapshot != null) {
2513        snapshotId = snapshot.getId();
2514        snapshot.close();
2515      }
2516      HStore.this.updateStorefiles(storeFiles, snapshotId);
2517    }

2690   */
2691  public synchronized void closeAndArchiveCompactedFiles() throws IOException {
2692    // ensure other threads do not attempt to archive the same files on close()
2693    archiveLock.lock();
2694    try {
2695      lock.readLock().lock();
2696      Collection<HStoreFile> copyCompactedfiles = null;
2697      try {
2698        Collection<HStoreFile> compactedfiles =
2699            this.getStoreEngine().getStoreFileManager().getCompactedfiles();
2700        if (CollectionUtils.isNotEmpty(compactedfiles)) {
2701          // Do a copy under read lock
2702          copyCompactedfiles = new ArrayList<>(compactedfiles);
2703        } else {
2704          LOG.trace("No compacted files to archive");
2705        }
2706      } finally {
2707        lock.readLock().unlock();
2708      }
2709      if (CollectionUtils.isNotEmpty(copyCompactedfiles)) {
2710        removeCompactedfiles(copyCompactedfiles, true);
2711      }
2712    } finally {
2713      archiveLock.unlock();
2714    }
2715  }

2717  /**
2718   * Archives and removes the compacted files
2719   * @param compactedfiles The compacted files in this store that are not active in reads
2720   * @param evictOnClose true if blocks should be evicted from the cache when an HFile reader is
2721   *   closed, false if not
2722   */
2723  private void removeCompactedfiles(Collection<HStoreFile> compactedfiles, boolean evictOnClose)
2724      throws IOException {
2725    final List<HStoreFile> filesToRemove = new ArrayList<>(compactedfiles.size());
2726    final List<Long> storeFileSizes = new ArrayList<>(compactedfiles.size());
2727    for (final HStoreFile file : compactedfiles) {
2728      synchronized (file) {
2729        try {
2730          StoreFileReader r = file.getReader();
2731          if (r == null) {
2732            LOG.debug("The file {} was closed but still not archived", file);
2733            // HACK: Temporarily re-open the reader so we can get the size of the file. Ideally,
2734            // we should know the size of an HStoreFile without having to ask the HStoreFileReader
2735            // for that.
2736            long length = getStoreFileSize(file);
2737            filesToRemove.add(file);
2738            storeFileSizes.add(length);
2739            continue;
2740          }
2741
2742          if (file.isCompactedAway() && !file.isReferencedInReads()) {
2743            // Even if deleting fails we need not bother as any new scanners won't be
2744            // able to use the compacted file as the status is already compactedAway
2745            LOG.trace("Closing and archiving the file {}", file);
2746            // Copy the file size before closing the reader
2747            final long length = r.length();
2748            r.close(evictOnClose);
2749            // Just close and return
2750            filesToRemove.add(file);
2751            // Only add the length if we successfully added the file to `filesToRemove`
2752            storeFileSizes.add(length);
2753          } else {
2754            LOG.info("Can't archive compacted file " + file.getPath()
2755                + " because of either isCompactedAway=" + file.isCompactedAway()
2756                + " or file has reference, isReferencedInReads=" + file.isReferencedInReads()
2757                + ", refCount=" + r.getRefCount() + ", skipping for now.");
2758          }
2759        } catch (Exception e) {
2760          LOG.error("Exception while trying to close the compacted store file {}", file.getPath(),
2761              e);
2762        }
2763      }
2764    }
2765    if (this.isPrimaryReplicaStore()) {
2766      // Only the primary region is allowed to move the file to archive.
2767      // The secondary region does not move the files to archive. Any active reads from
2768      // the secondary region will still work because the file as such has active readers on it.
2769      if (!filesToRemove.isEmpty()) {
2770        LOG.debug("Moving the files {} to archive", filesToRemove);
2771        // Only if this is successful it has to be removed
2772        try {
2773          getRegionFileSystem()
2774            .removeStoreFiles(this.getColumnFamilyDescriptor().getNameAsString(), filesToRemove);
2775        } catch (FailedArchiveException fae) {
2776          // Even if archiving some files failed, we still need to clear out any of the
2777          // files which were successfully archived.  Otherwise we will receive a
2778          // FileNotFoundException when we attempt to re-archive them in the next go around.
2779          Collection<Path> failedFiles = fae.getFailedFiles();
2780          Iterator<HStoreFile> iter = filesToRemove.iterator();
2781          Iterator<Long> sizeIter = storeFileSizes.iterator();
2782          while (iter.hasNext()) {
2783            sizeIter.next();
2784            if (failedFiles.contains(iter.next().getPath())) {
2785              iter.remove();
2786              sizeIter.remove();
2787            }
2788          }
2789          if (!filesToRemove.isEmpty()) {
2790            clearCompactedfiles(filesToRemove);
2791          }
2792          throw fae;
2793        }
2794      }
2795    }
2796    if (!filesToRemove.isEmpty()) {
2797      // Clear the compactedfiles from the store file manager
2798      clearCompactedfiles(filesToRemove);
2799      // Try to send report of this archival to the Master for updating quota usage faster
2800      reportArchivedFilesForQuota(filesToRemove, storeFileSizes);
2801    }
2802  }

2850  void reportArchivedFilesForQuota(List<? extends StoreFile> archivedFiles, List<Long> fileSizes) {
2851    // Sanity check from the caller
2852    if (archivedFiles.size() != fileSizes.size()) {
2853      throw new RuntimeException("Coding error: should never see lists of varying size");
2854    }
2855    RegionServerServices rss = this.region.getRegionServerServices();
2856    if (rss == null) {
2857      return;
2858    }
2859    List<Entry<String,Long>> filesWithSizes = new ArrayList<>(archivedFiles.size());
2860    Iterator<Long> fileSizeIter = fileSizes.iterator();
2861    for (StoreFile storeFile : archivedFiles) {
2862      final long fileSize = fileSizeIter.next();
2863      if (storeFile.isHFile() && fileSize != 0) {
2864        filesWithSizes.add(Maps.immutableEntry(storeFile.getPath().getName(), fileSize));
2865      }
2866    }
2867    if (LOG.isTraceEnabled()) {
2868      LOG.trace("Files archived: " + archivedFiles + ", reporting the following to the Master: "
2869          + filesWithSizes);
2870    }
2871    boolean success = rss.reportFileArchivalForQuotas(getTableName(), filesWithSizes);
2872    if (!success) {
2873      LOG.warn("Failed to report archival of files: " + filesWithSizes);
2874    }
