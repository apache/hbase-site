094  public static HeapMemoryManager create(Configuration conf, FlushRequester memStoreFlusher,
095                Server server, RegionServerAccounting regionServerAccounting) {
096    BlockCache blockCache = CacheConfig.instantiateBlockCache(conf);
097    if (blockCache instanceof ResizableBlockCache) {
098      return new HeapMemoryManager((ResizableBlockCache) blockCache, memStoreFlusher, server,
099                 regionServerAccounting);
100    }
101    return null;
102  }

104  @VisibleForTesting
105  HeapMemoryManager(ResizableBlockCache blockCache, FlushRequester memStoreFlusher,
106                Server server, RegionServerAccounting regionServerAccounting) {
107    Configuration conf = server.getConfiguration();
108    this.blockCache = blockCache;
109    this.memStoreFlusher = memStoreFlusher;
110    this.server = server;
111    this.regionServerAccounting = regionServerAccounting;
112    this.tunerOn = doInit(conf);
113    this.defaultChorePeriod = conf.getInt(HBASE_RS_HEAP_MEMORY_TUNER_PERIOD,
114      HBASE_RS_HEAP_MEMORY_TUNER_DEFAULT_PERIOD);
115    this.heapOccupancyLowWatermark = conf.getFloat(HConstants.HEAP_OCCUPANCY_LOW_WATERMARK_KEY,
116      HConstants.DEFAULT_HEAP_OCCUPANCY_LOW_WATERMARK);
117    metricsHeapMemoryManager = new MetricsHeapMemoryManager();
118  }

120  private boolean doInit(Configuration conf) {
121    boolean tuningEnabled = true;
122    globalMemStorePercent = HeapMemorySizeUtil.getGlobalMemStorePercent(conf, false);
123    blockCachePercent = conf.getFloat(HFILE_BLOCK_CACHE_SIZE_KEY,
124        HConstants.HFILE_BLOCK_CACHE_SIZE_DEFAULT);
125    HeapMemorySizeUtil.checkForClusterFreeMemoryLimit(conf);
126    // Initialize max and min range for memstore heap space
127    globalMemStorePercentMinRange = conf.getFloat(MEMSTORE_SIZE_MIN_RANGE_KEY,
128        globalMemStorePercent);
129    globalMemStorePercentMaxRange = conf.getFloat(MEMSTORE_SIZE_MAX_RANGE_KEY,
130        globalMemStorePercent);
131    if (globalMemStorePercent < globalMemStorePercentMinRange) {
132      LOG.warn("Setting " + MEMSTORE_SIZE_MIN_RANGE_KEY + " to " + globalMemStorePercent
133          + ", same value as " + HeapMemorySizeUtil.MEMSTORE_SIZE_KEY
134          + " because supplied value greater than initial memstore size value.");
135      globalMemStorePercentMinRange = globalMemStorePercent;
136      conf.setFloat(MEMSTORE_SIZE_MIN_RANGE_KEY, globalMemStorePercentMinRange);
137    }
138    if (globalMemStorePercent > globalMemStorePercentMaxRange) {
139      LOG.warn("Setting " + MEMSTORE_SIZE_MAX_RANGE_KEY + " to " + globalMemStorePercent
140          + ", same value as " + HeapMemorySizeUtil.MEMSTORE_SIZE_KEY
141          + " because supplied value less than initial memstore size value.");
142      globalMemStorePercentMaxRange = globalMemStorePercent;
143      conf.setFloat(MEMSTORE_SIZE_MAX_RANGE_KEY, globalMemStorePercentMaxRange);
144    }
145    if (globalMemStorePercent == globalMemStorePercentMinRange
146        && globalMemStorePercent == globalMemStorePercentMaxRange) {
147      tuningEnabled = false;
148    }
149    // Initialize max and min range for block cache
150    blockCachePercentMinRange = conf.getFloat(BLOCK_CACHE_SIZE_MIN_RANGE_KEY, blockCachePercent);
151    blockCachePercentMaxRange = conf.getFloat(BLOCK_CACHE_SIZE_MAX_RANGE_KEY, blockCachePercent);
152    if (blockCachePercent < blockCachePercentMinRange) {
153      LOG.warn("Setting " + BLOCK_CACHE_SIZE_MIN_RANGE_KEY + " to " + blockCachePercent
154          + ", same value as " + HFILE_BLOCK_CACHE_SIZE_KEY
155          + " because supplied value greater than initial block cache size.");
156      blockCachePercentMinRange = blockCachePercent;
157      conf.setFloat(BLOCK_CACHE_SIZE_MIN_RANGE_KEY, blockCachePercentMinRange);
158    }
159    if (blockCachePercent > blockCachePercentMaxRange) {
160      LOG.warn("Setting " + BLOCK_CACHE_SIZE_MAX_RANGE_KEY + " to " + blockCachePercent
161          + ", same value as " + HFILE_BLOCK_CACHE_SIZE_KEY
162          + " because supplied value less than initial block cache size.");
163      blockCachePercentMaxRange = blockCachePercent;
164      conf.setFloat(BLOCK_CACHE_SIZE_MAX_RANGE_KEY, blockCachePercentMaxRange);
165    }
166    if (tuningEnabled && blockCachePercent == blockCachePercentMinRange
167        && blockCachePercent == blockCachePercentMaxRange) {
168      tuningEnabled = false;
169    }
170
171    int gml = (int) (globalMemStorePercentMaxRange * CONVERT_TO_PERCENTAGE);
172    this.l2BlockCachePercent = HeapMemorySizeUtil.getL2BlockCacheHeapPercent(conf);
173    int bcul = (int) ((blockCachePercentMinRange + l2BlockCachePercent) * CONVERT_TO_PERCENTAGE);
174    if (CONVERT_TO_PERCENTAGE - (gml + bcul) < CLUSTER_MINIMUM_MEMORY_THRESHOLD) {
175      throw new RuntimeException("Current heap configuration for MemStore and BlockCache exceeds "
176          + "the threshold required for successful cluster operation. "
177          + "The combined value cannot exceed 0.8. Please check the settings for "
178          + MEMSTORE_SIZE_MAX_RANGE_KEY + " and " + BLOCK_CACHE_SIZE_MIN_RANGE_KEY
179          + " in your configuration. " + MEMSTORE_SIZE_MAX_RANGE_KEY + " is "
180          + globalMemStorePercentMaxRange + " and " + BLOCK_CACHE_SIZE_MIN_RANGE_KEY + " is "
181          + blockCachePercentMinRange);
182    }
183    gml = (int) (globalMemStorePercentMinRange * CONVERT_TO_PERCENTAGE);
184    bcul = (int) ((blockCachePercentMaxRange + l2BlockCachePercent) * CONVERT_TO_PERCENTAGE);
185    if (CONVERT_TO_PERCENTAGE - (gml + bcul) < CLUSTER_MINIMUM_MEMORY_THRESHOLD) {
186      throw new RuntimeException("Current heap configuration for MemStore and BlockCache exceeds "
187          + "the threshold required for successful cluster operation. "
188          + "The combined value cannot exceed 0.8. Please check the settings for "
189          + MEMSTORE_SIZE_MIN_RANGE_KEY + " and " + BLOCK_CACHE_SIZE_MAX_RANGE_KEY
190          + " in your configuration. " + MEMSTORE_SIZE_MIN_RANGE_KEY + " is "
191          + globalMemStorePercentMinRange + " and " + BLOCK_CACHE_SIZE_MAX_RANGE_KEY + " is "
192          + blockCachePercentMaxRange);
193    }
194    return tuningEnabled;
195  }

197  public void start(ChoreService service) {
198      LOG.info("Starting HeapMemoryTuner chore.");
199      this.heapMemTunerChore = new HeapMemoryTunerChore();
200      service.scheduleChore(heapMemTunerChore);
201      if (tunerOn) {
202      // Register HeapMemoryTuner as a memstore flush listener
203      memStoreFlusher.registerFlushRequestListener(heapMemTunerChore);
204    }
205  }
206
207  public void stop() {
208    // The thread is Daemon. Just interrupting the ongoing process.
209    LOG.info("Stoping HeapMemoryTuner chore.");
210    this.heapMemTunerChore.cancel(true);
211  }

229  private class HeapMemoryTunerChore extends ScheduledChore implements FlushRequestListener {
230    private HeapMemoryTuner heapMemTuner;
231    private AtomicLong blockedFlushCount = new AtomicLong();
232    private AtomicLong unblockedFlushCount = new AtomicLong();
233    private long evictCount = 0L;
234    private long cacheMissCount = 0L;
235    private TunerContext tunerContext = new TunerContext();
236    private boolean alarming = false;
237
238    public HeapMemoryTunerChore() {
239      super(server.getServerName() + "-HeapMemoryTunerChore", server, defaultChorePeriod);
240      Class<? extends HeapMemoryTuner> tunerKlass = server.getConfiguration().getClass(
241          HBASE_RS_HEAP_MEMORY_TUNER_CLASS, DefaultHeapMemoryTuner.class, HeapMemoryTuner.class);
242      heapMemTuner = ReflectionUtils.newInstance(tunerKlass, server.getConfiguration());
243    }
244
245    @Override
246    protected void chore() {
247      // Sample heap occupancy
248      MemoryUsage memUsage = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
249      heapOccupancyPercent = (float)memUsage.getUsed() / (float)memUsage.getCommitted();
250      // If we are above the heap occupancy alarm low watermark, switch to short
251      // sleeps for close monitoring. Stop autotuning, we are in a danger zone.
252      if (heapOccupancyPercent >= heapOccupancyLowWatermark) {
253        if (!alarming) {
254          LOG.warn("heapOccupancyPercent " + heapOccupancyPercent +
255            " is above heap occupancy alarm watermark (" + heapOccupancyLowWatermark + ")");
256          alarming = true;
257        }
258        metricsHeapMemoryManager.increaseAboveHeapOccupancyLowWatermarkCounter();
259        triggerNow();
260        try {
261          // Need to sleep ourselves since we've told the chore's sleeper
262          // to skip the next sleep cycle.
263          Thread.sleep(1000);
264        } catch (InterruptedException e) {
265          // Interrupted, propagate
266          Thread.currentThread().interrupt();
267        }
268      } else {
269        if (alarming) {
270          LOG.info("heapOccupancyPercent " + heapOccupancyPercent +
271            " is now below the heap occupancy alarm watermark (" +
272            heapOccupancyLowWatermark + ")");
273          alarming = false;
274        }
275      }
276      // Autotune if tuning is enabled and allowed
277      if (tunerOn && !alarming) {
278        tune();
279      }
280    }
281
282    private void tune() {
283      // TODO check if we can increase the memory boundaries
284      // while remaining in the limits
285      long curEvictCount;
286      long curCacheMisCount;
287      long blockedFlushCnt;
288      long unblockedFlushCnt;
289      curEvictCount = blockCache.getStats().getEvictedCount();
290      tunerContext.setEvictCount(curEvictCount - evictCount);
291      evictCount = curEvictCount;
292      curCacheMisCount = blockCache.getStats().getMissCachingCount();
293      tunerContext.setCacheMissCount(curCacheMisCount-cacheMissCount);
294      cacheMissCount = curCacheMisCount;
295      blockedFlushCnt = blockedFlushCount.getAndSet(0);
296      tunerContext.setBlockedFlushCount(blockedFlushCnt);
297      metricsHeapMemoryManager.updateBlockedFlushCount(blockedFlushCnt);
298      unblockedFlushCnt = unblockedFlushCount.getAndSet(0);
299      tunerContext.setUnblockedFlushCount(unblockedFlushCnt);
300      metricsHeapMemoryManager.updateUnblockedFlushCount(unblockedFlushCnt);
301      tunerContext.setCurBlockCacheUsed((float) blockCache.getCurrentSize() / maxHeapSize);
302      metricsHeapMemoryManager.setCurBlockCacheSizeGauge(blockCache.getCurrentSize());
303      tunerContext.setCurMemStoreUsed((float)regionServerAccounting.getGlobalMemstoreSize() / maxHeapSize);
304      metricsHeapMemoryManager.setCurMemStoreSizeGauge(regionServerAccounting.getGlobalMemstoreSize());
305      tunerContext.setCurBlockCacheSize(blockCachePercent);
306      tunerContext.setCurMemStoreSize(globalMemStorePercent);
307      TunerResult result = null;
308      try {
309        result = this.heapMemTuner.tune(tunerContext);
310      } catch (Throwable t) {
311        LOG.error("Exception thrown from the HeapMemoryTuner implementation", t);
312      }
313      if (result != null && result.needsTuning()) {
314        float memstoreSize = result.getMemstoreSize();
315        float blockCacheSize = result.getBlockCacheSize();
316        LOG.debug("From HeapMemoryTuner new memstoreSize: " + memstoreSize
317            + ". new blockCacheSize: " + blockCacheSize);
318        if (memstoreSize < globalMemStorePercentMinRange) {
319          LOG.info("New memstoreSize from HeapMemoryTuner " + memstoreSize + " is below min level "
320              + globalMemStorePercentMinRange + ". Resetting memstoreSize to min size");
321          memstoreSize = globalMemStorePercentMinRange;
322        } else if (memstoreSize > globalMemStorePercentMaxRange) {
323          LOG.info("New memstoreSize from HeapMemoryTuner " + memstoreSize + " is above max level "
324              + globalMemStorePercentMaxRange + ". Resetting memstoreSize to max size");
325          memstoreSize = globalMemStorePercentMaxRange;
326        }
327        if (blockCacheSize < blockCachePercentMinRange) {
328          LOG.info("New blockCacheSize from HeapMemoryTuner " + blockCacheSize
329              + " is below min level " + blockCachePercentMinRange
330              + ". Resetting blockCacheSize to min size");
331          blockCacheSize = blockCachePercentMinRange;
332        } else if (blockCacheSize > blockCachePercentMaxRange) {
333          LOG.info("New blockCacheSize from HeapMemoryTuner " + blockCacheSize
334              + " is above max level " + blockCachePercentMaxRange
335              + ". Resetting blockCacheSize to min size");
336          blockCacheSize = blockCachePercentMaxRange;
337        }
338        int gml = (int) (memstoreSize * CONVERT_TO_PERCENTAGE);
339        int bcul = (int) ((blockCacheSize + l2BlockCachePercent) * CONVERT_TO_PERCENTAGE);
340        if (CONVERT_TO_PERCENTAGE - (gml + bcul) < CLUSTER_MINIMUM_MEMORY_THRESHOLD) {
341          LOG.info("Current heap configuration from HeapMemoryTuner exceeds "
342              + "the threshold required for successful cluster operation. "
343              + "The combined value cannot exceed 0.8. " + HeapMemorySizeUtil.MEMSTORE_SIZE_KEY
344              + " is " + memstoreSize + " and " + HFILE_BLOCK_CACHE_SIZE_KEY + " is "
345              + blockCacheSize);
346          // TODO can adjust the value so as not exceed 80%. Is that correct? may be.
347        } else {
348          int memStoreDeltaSize =
349              (int) ((memstoreSize - globalMemStorePercent) * CONVERT_TO_PERCENTAGE);
350          int blockCacheDeltaSize =
351              (int) ((blockCacheSize - blockCachePercent) * CONVERT_TO_PERCENTAGE);
352          metricsHeapMemoryManager.updateMemStoreDeltaSizeHistogram(memStoreDeltaSize);
353          metricsHeapMemoryManager.updateBlockCacheDeltaSizeHistogram(blockCacheDeltaSize);
354          long newBlockCacheSize = (long) (maxHeapSize * blockCacheSize);
355          long newMemstoreSize = (long) (maxHeapSize * memstoreSize);
356          LOG.info("Setting block cache heap size to " + newBlockCacheSize
357              + " and memstore heap size to " + newMemstoreSize);
358          blockCachePercent = blockCacheSize;
359          blockCache.setMaxSize(newBlockCacheSize);
360          globalMemStorePercent = memstoreSize;
361          memStoreFlusher.setGlobalMemstoreLimit(newMemstoreSize);
362          for (HeapMemoryTuneObserver observer : tuneObservers) {
363            observer.onHeapMemoryTune(newMemstoreSize, newBlockCacheSize);
364          }
365        }
366      } else {
367        metricsHeapMemoryManager.increaseTunerDoNothingCounter();
368        if (LOG.isDebugEnabled()) {
369          LOG.debug("No changes made by HeapMemoryTuner.");
370        }
371      }
372    }
373
374    @Override
375    public void flushRequested(FlushType type, Region region) {
376      switch (type) {
377      case ABOVE_HIGHER_MARK:
378        blockedFlushCount.incrementAndGet();
379        break;
380      case ABOVE_LOWER_MARK:
381        unblockedFlushCount.incrementAndGet();
382        break;
383      default:
384        // In case of normal flush don't do any action.
385        break;
386      }
387    }
