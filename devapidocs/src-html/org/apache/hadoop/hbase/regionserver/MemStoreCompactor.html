022import java.util.concurrent.atomic.AtomicBoolean;

036/**
037 * The ongoing MemStore Compaction manager, dispatches a solo running compaction and interrupts
038 * the compaction if requested. The compaction is interrupted and stopped by CompactingMemStore,
039 * for example when another compaction needs to be started.
040 * Prior to compaction the MemStoreCompactor evaluates
041 * the compacting ratio and aborts the compaction if it is not worthy.
042 * The MemStoreScanner is used to traverse the compaction pipeline. The MemStoreScanner
043 * is included in internal store scanner, where all compaction logic is implemented.
044 * Threads safety: It is assumed that the compaction pipeline is immutable,
045 * therefore no special synchronization is required.
046 */
047@InterfaceAudience.Private
048public class MemStoreCompactor {
049
050  public static final long DEEP_OVERHEAD = ClassSize
051      .align(ClassSize.OBJECT + 4 * ClassSize.REFERENCE
052          // compactingMemStore, versionedList, isInterrupted, strategy (the reference)
053          // "action" is an enum and thus it is a class with static final constants,
054          // so counting only the size of the reference to it and not the size of the internals
055          + Bytes.SIZEOF_INT        // compactionKVMax
056          + ClassSize.ATOMIC_BOOLEAN    // isInterrupted (the internals)
057      );
058
059  private static final Logger LOG = LoggerFactory.getLogger(MemStoreCompactor.class);
060  private CompactingMemStore compactingMemStore;
061
062  // a static version of the segment list from the pipeline
063  private VersionedSegmentsList versionedList;
064
065  // a flag raised when compaction is requested to stop
066  private final AtomicBoolean isInterrupted = new AtomicBoolean(false);
067
068  // the limit to the size of the groups to be later provided to MemStoreSegmentsIterator
069  private final int compactionKVMax;
070
071  private MemStoreCompactionStrategy strategy;
072
073  public MemStoreCompactor(CompactingMemStore compactingMemStore,
074      MemoryCompactionPolicy compactionPolicy) throws IllegalArgumentIOException {
075    this.compactingMemStore = compactingMemStore;
076    this.compactionKVMax = compactingMemStore.getConfiguration()
077        .getInt(HConstants.COMPACTION_KV_MAX, HConstants.COMPACTION_KV_MAX_DEFAULT);
078    initiateCompactionStrategy(compactionPolicy, compactingMemStore.getConfiguration(),
079        compactingMemStore.getFamilyName());
080  }
081
082  /**----------------------------------------------------------------------
083   * The request to dispatch the compaction asynchronous task.
084   * The method returns true if compaction was successfully dispatched, or false if there
085   * is already an ongoing compaction or no segments to compact.
086   */
087  public boolean start() throws IOException {
088    if (!compactingMemStore.hasImmutableSegments()) { // no compaction on empty pipeline
089      return false;
090    }
091
092    // get a snapshot of the list of the segments from the pipeline,
093    // this local copy of the list is marked with specific version
094    versionedList = compactingMemStore.getImmutableSegments();
095
096    if (LOG.isDebugEnabled()) {
097      LOG.debug("Starting the In-Memory Compaction for store "
098          + compactingMemStore.getStore().getColumnFamilyName());
099    }
100    HStore store = compactingMemStore.getStore();
101    RegionCoprocessorHost cpHost = store.getCoprocessorHost();
102    if (cpHost != null) {
103      cpHost.preMemStoreCompaction(store);
104    }
105    try {
106      doCompaction();
107    } finally {
108      if (cpHost != null) {
109        cpHost.postMemStoreCompaction(store);
110      }
111    }
112    return true;
113  }
114
115  /**----------------------------------------------------------------------
116  * The request to cancel the compaction asynchronous task
117  * The compaction may still happen if the request was sent too late
118  * Non-blocking request
119  */
120  public void stop() {
121      isInterrupted.compareAndSet(false, true);
122  }
123
124
125  public void resetStats() {
126    strategy.resetStats();
127  }
128
129  /**----------------------------------------------------------------------
130  * Reset the interruption indicator and clear the pointers in order to allow good
131  * garbage collection
132  */
133  private void releaseResources() {
134    isInterrupted.set(false);
135    versionedList = null;
136  }
137
138  /**----------------------------------------------------------------------
139  * The worker thread performs the compaction asynchronously.
140  * The solo (per compactor) thread only reads the compaction pipeline.
141  * There is at most one thread per memstore instance.
142  */
143  private void doCompaction() {
144    ImmutableSegment result = null;
145    boolean resultSwapped = false;
146    if (isInterrupted.get()) {      // if the entire process is interrupted cancel flattening
147      return;           // the compaction also doesn't start when interrupted
148    }
149
150    MemStoreCompactionStrategy.Action nextStep = strategy.getAction(versionedList);
151    boolean merge =
152        (nextStep == MemStoreCompactionStrategy.Action.MERGE ||
153            nextStep == MemStoreCompactionStrategy.Action.MERGE_COUNT_UNIQUE_KEYS);
154    try {
155      if (nextStep == MemStoreCompactionStrategy.Action.NOOP) {
156        return;
157      }
158      if (nextStep == MemStoreCompactionStrategy.Action.FLATTEN ||
159          nextStep == MemStoreCompactionStrategy.Action.FLATTEN_COUNT_UNIQUE_KEYS) {
160        // some Segment in the pipeline is with SkipList index, make it flat
161        compactingMemStore.flattenOneSegment(versionedList.getVersion(), nextStep);
162        return;
163      }
164
165      // Create one segment representing all segments in the compaction pipeline,
166      // either by compaction or by merge
167      if (!isInterrupted.get()) {
168        result = createSubstitution(nextStep);
169      }
170
171      // Substitute the pipeline with one segment
172      if (!isInterrupted.get()) {
173        if (resultSwapped = compactingMemStore.swapCompactedSegments(
174            versionedList, result, merge)) {
175          // update compaction strategy
176          strategy.updateStats(result);
177          // update the wal so it can be truncated and not get too long
178          compactingMemStore.updateLowestUnflushedSequenceIdInWAL(true); // only if greater
179        }
180      }
181    } catch (IOException e) {
182      LOG.debug("Interrupting the MemStore in-memory compaction for store "
183          + compactingMemStore.getFamilyName());
184      Thread.currentThread().interrupt();
185    } finally {
186      // For the MERGE case, if the result was created, but swap didn't happen,
187      // we DON'T need to close the result segment (meaning its MSLAB)!
188      // Because closing the result segment means closing the chunks of all segments
189      // in the compaction pipeline, which still have ongoing scans.
190      if (!merge && (result != null) && !resultSwapped) {
191        result.close();
192      }
193      releaseResources();
194    }
195
196  }
197
198  /**----------------------------------------------------------------------
199   * Creation of the ImmutableSegment either by merge or copy-compact of the segments of the
200   * pipeline, based on the Compactor Iterator. The new ImmutableSegment is returned.
201   */
202  private ImmutableSegment createSubstitution(MemStoreCompactionStrategy.Action action) throws
203      IOException {
204
205    ImmutableSegment result = null;
206    MemStoreSegmentsIterator iterator = null;
207
208    switch (action) {
209      case COMPACT:
210        iterator = new MemStoreCompactorSegmentsIterator(versionedList.getStoreSegments(),
211            compactingMemStore.getComparator(),
212            compactionKVMax, compactingMemStore.getStore());
213
214        result = SegmentFactory.instance().createImmutableSegmentByCompaction(
215          compactingMemStore.getConfiguration(), compactingMemStore.getComparator(), iterator,
216          versionedList.getNumOfCells(), compactingMemStore.getIndexType(), action);
217        iterator.close();
218        break;
219      case MERGE:
220      case MERGE_COUNT_UNIQUE_KEYS:
221        iterator =
222            new MemStoreMergerSegmentsIterator(versionedList.getStoreSegments(),
223            compactingMemStore.getComparator(), compactionKVMax);
224
225        result = SegmentFactory.instance().createImmutableSegmentByMerge(
226          compactingMemStore.getConfiguration(), compactingMemStore.getComparator(), iterator,
227          versionedList.getNumOfCells(), versionedList.getStoreSegments(),
228          compactingMemStore.getIndexType(), action);
229        iterator.close();
230        break;
231      default:
232        throw new RuntimeException("Unknown action " + action); // sanity check
233    }
234
235    return result;
236  }
237
238  @VisibleForTesting
239  void initiateCompactionStrategy(MemoryCompactionPolicy compType,
240      Configuration configuration, String cfName) throws IllegalArgumentIOException {
241
242    assert (compType !=MemoryCompactionPolicy.NONE);
243
244    switch (compType){
245      case BASIC: strategy = new BasicMemStoreCompactionStrategy(configuration, cfName);
246        break;
247      case EAGER: strategy = new EagerMemStoreCompactionStrategy(configuration, cfName);
248        break;
249      case ADAPTIVE: strategy = new AdaptiveMemStoreCompactionStrategy(configuration, cfName);
250        break;
251      default:
252        // sanity check
253        throw new IllegalArgumentIOException("Unknown memory compaction type " + compType);
254    }
255  }
