018package org.apache.hadoop.hbase.regionserver;
019
020import org.apache.hbase.thirdparty.com.google.common.annotations.VisibleForTesting;
021
022import java.util.Random;
023
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.HBaseInterfaceAudience;
026import org.apache.hadoop.hbase.HConstants;
027import org.apache.yetus.audience.InterfaceAudience;
028import org.apache.hadoop.hbase.client.TableDescriptor;
029import org.apache.hadoop.hbase.procedure2.util.StringUtils;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032
033/**
034 * A {@link RegionSplitPolicy} implementation which splits a region
035 * as soon as any of its store files exceeds a maximum configurable
036 * size.
037 * <p>
038 * This is the default split policy. From 0.94.0 on the default split policy has
039 * changed to {@link IncreasingToUpperBoundRegionSplitPolicy}
040 * </p>
041 */
042@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
043public class ConstantSizeRegionSplitPolicy extends RegionSplitPolicy {
044  private static final Logger LOG =
045    LoggerFactory.getLogger(ConstantSizeRegionSplitPolicy.class);
046  private static final Random RANDOM = new Random();
047
048  private long desiredMaxFileSize;
049  private double jitterRate;
050  protected boolean overallHRegionFiles;
051
052  @Override
053  protected void configureForRegion(HRegion region) {
054    super.configureForRegion(region);
055    Configuration conf = getConf();
056    TableDescriptor desc = region.getTableDescriptor();
057    if (desc != null) {
058      this.desiredMaxFileSize = desc.getMaxFileSize();
059    }
060    if (this.desiredMaxFileSize <= 0) {
061      this.desiredMaxFileSize = conf.getLong(HConstants.HREGION_MAX_FILESIZE,
062        HConstants.DEFAULT_MAX_FILE_SIZE);
063    }
064    this.overallHRegionFiles = conf.getBoolean(HConstants.OVERALL_HREGION_FILES,
065      HConstants.DEFAULT_OVERALL_HREGION_FILES);
066    double jitter = conf.getDouble("hbase.hregion.max.filesize.jitter", 0.25D);
067    this.jitterRate = (RANDOM.nextFloat() - 0.5D) * jitter;
068    long jitterValue = (long) (this.desiredMaxFileSize * this.jitterRate);
069    // make sure the long value won't overflow with jitter
070    if (this.jitterRate > 0 && jitterValue > (Long.MAX_VALUE - this.desiredMaxFileSize)) {
071      this.desiredMaxFileSize = Long.MAX_VALUE;
072    } else {
073      this.desiredMaxFileSize += jitterValue;
074    }
075  }
076
077  @Override
078  protected boolean shouldSplit() {
079    if (!canSplit()) {
080      return false;
081    }
082    return isExceedSize(desiredMaxFileSize);
083  }
084
085  long getDesiredMaxFileSize() {
086    return desiredMaxFileSize;
087  }
088
089  @VisibleForTesting
090  public boolean positiveJitterRate() {
091    return this.jitterRate > 0;
092  }
093
094  /**
095   * @return true if region size exceed the sizeToCheck
096   */
097  protected final boolean isExceedSize(long sizeToCheck) {
098    if (overallHRegionFiles) {
099      long sumSize = 0;
100      for (HStore store : region.getStores()) {
101        sumSize += store.getSize();
102      }
103      if (sumSize > sizeToCheck) {
104        LOG.debug("ShouldSplit because region size is big enough "
105            + "size={}, sizeToCheck={}{}", StringUtils.humanSize(sumSize),
106          StringUtils.humanSize(sizeToCheck));
107        return true;
108      }
109    } else {
110      for (HStore store : region.getStores()) {
111        long size = store.getSize();
112        if (size > sizeToCheck) {
113          LOG.debug("ShouldSplit because {} size={}, sizeToCheck={}{}",
114            store.getColumnFamilyName(), StringUtils.humanSize(size),
115            StringUtils.humanSize(sizeToCheck));
116          return true;
117        }
118      }
119    }
120    return false;
121  }
122}
