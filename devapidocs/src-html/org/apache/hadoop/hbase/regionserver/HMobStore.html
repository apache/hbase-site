063/**
064 * The store implementation to save MOBs (medium objects), it extends the HStore.
065 * When a descriptor of a column family has the value "IS_MOB", it means this column family
066 * is a mob one. When a HRegion instantiate a store for this column family, the HMobStore is
067 * created.
068 * HMobStore is almost the same with the HStore except using different types of scanners.
069 * In the method of getScanner, the MobStoreScanner and MobReversedStoreScanner are returned.
070 * In these scanners, a additional seeks in the mob files should be performed after the seek
071 * to HBase is done.
072 * The store implements how we save MOBs by extending HStore. When a descriptor
073 * of a column family has the value "IS_MOB", it means this column family is a mob one. When a
074 * HRegion instantiate a store for this column family, the HMobStore is created. HMobStore is
075 * almost the same with the HStore except using different types of scanners. In the method of
076 * getScanner, the MobStoreScanner and MobReversedStoreScanner are returned. In these scanners, a
077 * additional seeks in the mob files should be performed after the seek in HBase is done.
078 */
079@InterfaceAudience.Private
080public class HMobStore extends HStore {

098  // 1. A ref tag with type TagType.MOB_REFERENCE_TAG_TYPE. This just denote this this cell is not
099  // original one but a ref to another MOB Cell.
100  // 2. Table name tag. It's very useful in cloning the snapshot. When reading from the cloning
101  // table, we need to find the original mob files by this table name. For details please see
102  // cloning snapshot for mob files.
103  private final byte[] refCellTags;

105  public HMobStore(final HRegion region, final ColumnFamilyDescriptor family,
106      final Configuration confParam) throws IOException {
107    super(region, family, confParam);
108    this.family = family;
109    this.mobCacheConfig = (MobCacheConfig) cacheConf;
110    this.homePath = MobUtils.getMobHome(conf);
111    this.mobFamilyPath = MobUtils.getMobFamilyPath(conf, this.getTableName(),
112        family.getNameAsString());
113    List<Path> locations = new ArrayList<>(2);
114    locations.add(mobFamilyPath);
115    TableName tn = region.getTableDescriptor().getTableName();
116    locations.add(HFileArchiveUtil.getStoreArchivePath(conf, tn, MobUtils.getMobRegionInfo(tn)
117        .getEncodedName(), family.getNameAsString()));
118    map.put(Bytes.toString(tn.getName()), locations);
119    List<Tag> tags = new ArrayList<>(2);
120    tags.add(MobConstants.MOB_REF_TAG);
121    Tag tableNameTag = new ArrayBackedTag(TagType.MOB_TABLE_NAME_TAG_TYPE,
122        getTableName().getName());
123    tags.add(tableNameTag);
124    this.refCellTags = TagUtil.fromList(tags);
125  }

142  /**
143   * Gets the MobStoreScanner or MobReversedStoreScanner. In these scanners, a additional seeks in
144   * the mob files should be performed after the seek in HBase is done.
145   */
146  @Override
147  protected KeyValueScanner createScanner(Scan scan, ScanInfo scanInfo,
148      NavigableSet<byte[]> targetCols, long readPt) throws IOException {
149    if (MobUtils.isRefOnlyScan(scan)) {
150      Filter refOnlyFilter = new MobReferenceOnlyFilter();
151      Filter filter = scan.getFilter();
152      if (filter != null) {
153        scan.setFilter(new FilterList(filter, refOnlyFilter));
154      } else {
155        scan.setFilter(refOnlyFilter);
156      }
157    }
158    return scan.isReversed() ? new ReversedMobStoreScanner(this, scanInfo, scan, targetCols, readPt)
159        : new MobStoreScanner(this, scanInfo, scan, targetCols, readPt);
160  }

202  /**
203   * Creates the writer for the del file in temp directory.
204   * The del file keeps tracking the delete markers. Its name has a suffix _del,
205   * the format is [0-9a-f]+(_del)?.
206   * @param date The latest date of written cells.
207   * @param maxKeyCount The key count.
208   * @param compression The compression algorithm.
209   * @param startKey The start key.
210   * @return The writer for the del file.
211   * @throws IOException
212   */
213  public StoreFileWriter createDelFileWriterInTmp(Date date, long maxKeyCount,
214      Compression.Algorithm compression, byte[] startKey) throws IOException {
215    if (startKey == null) {
216      startKey = HConstants.EMPTY_START_ROW;
217    }
218    Path path = getTempDir();
219    String suffix = UUID
220        .randomUUID().toString().replaceAll("-", "") + "_del";
221    MobFileName mobFileName = MobFileName.create(startKey, MobUtils.formatDate(date), suffix);
222    return createWriterInTmp(mobFileName, path, maxKeyCount, compression, true);
223  }

262  /**
263   * Commits the mob file.
264   * @param sourceFile The source file.
265   * @param targetPath The directory path where the source file is renamed to.
266   * @throws IOException
267   */
268  public void commitFile(final Path sourceFile, Path targetPath) throws IOException {
269    if (sourceFile == null) {
270      return;
271    }
272    Path dstPath = new Path(targetPath, sourceFile.getName());
273    validateMobFile(sourceFile);
274    String msg = "Renaming flushed file from " + sourceFile + " to " + dstPath;
275    LOG.info(msg);
276    Path parent = dstPath.getParent();
277    if (!region.getFilesystem().exists(parent)) {
278      region.getFilesystem().mkdirs(parent);
279    }
280    if (!region.getFilesystem().rename(sourceFile, dstPath)) {
281      throw new IOException("Failed rename of " + sourceFile + " to " + dstPath);
282    }
283  }

285  /**
286   * Validates a mob file by opening and closing it.
287   *
288   * @param path the path to the mob file
289   */
290  private void validateMobFile(Path path) throws IOException {
291    HStoreFile storeFile = null;
292    try {
293      storeFile = new HStoreFile(region.getFilesystem(), path, conf, this.mobCacheConfig,
294          BloomType.NONE, isPrimaryReplicaStore());
295      storeFile.initReader();
296    } catch (IOException e) {
297      LOG.error("Fail to open mob file[" + path + "], keep it in temp directory.", e);
298      throw e;
299    } finally {
300      if (storeFile != null) {
301        storeFile.closeStoreFile(false);
302      }
303    }
304  }

318  /**
319   * Reads the cell from the mob file.
320   * @param reference The cell found in the HBase, its value is a path to a mob file.
321   * @param cacheBlocks Whether the scanner should cache blocks.
322   * @param readPt the read point.
323   * @param readEmptyValueOnMobCellMiss Whether return null value when the mob file is
324   *        missing or corrupt.
325   * @return The cell found in the mob file.
326   * @throws IOException
327   */
328  public Cell resolve(Cell reference, boolean cacheBlocks, long readPt,
329    boolean readEmptyValueOnMobCellMiss) throws IOException {
330    Cell result = null;
331    if (MobUtils.hasValidMobRefCellValue(reference)) {
332      String fileName = MobUtils.getMobFileName(reference);
333      Tag tableNameTag = MobUtils.getTableNameTag(reference);
334      if (tableNameTag != null) {
335        String tableNameString = Tag.getValueAsString(tableNameTag);
336        List<Path> locations = map.get(tableNameString);
337        if (locations == null) {
338          IdLock.Entry lockEntry = keyLock.getLockEntry(tableNameString.hashCode());
339          try {
340            locations = map.get(tableNameString);
341            if (locations == null) {
342              locations = new ArrayList<>(2);
343              TableName tn = TableName.valueOf(tableNameString);
344              locations.add(MobUtils.getMobFamilyPath(conf, tn, family.getNameAsString()));
345              locations.add(HFileArchiveUtil.getStoreArchivePath(conf, tn, MobUtils
346                  .getMobRegionInfo(tn).getEncodedName(), family.getNameAsString()));
347              map.put(tableNameString, locations);
348            }
349          } finally {
350            keyLock.releaseLockEntry(lockEntry);
351          }
352        }
353        result = readCell(locations, fileName, reference, cacheBlocks, readPt,
354          readEmptyValueOnMobCellMiss);
355      }
356    }
357    if (result == null) {
358      LOG.warn("The Cell result is null, assemble a new Cell with the same row,family,"
359          + "qualifier,timestamp,type and tags but with an empty value to return.");
360      result = ExtendedCellBuilderFactory.create(CellBuilderType.DEEP_COPY)
361              .setRow(reference.getRowArray(), reference.getRowOffset(), reference.getRowLength())
362              .setFamily(reference.getFamilyArray(), reference.getFamilyOffset(),
363                reference.getFamilyLength())
364              .setQualifier(reference.getQualifierArray(),
365                reference.getQualifierOffset(), reference.getQualifierLength())
366              .setTimestamp(reference.getTimestamp())
367              .setType(reference.getTypeByte())
368              .setValue(HConstants.EMPTY_BYTE_ARRAY)
369              .setTags(reference.getTagsArray(), reference.getTagsOffset(),
370                reference.getTagsLength())
371              .build();
372    }
373    return result;
374  }

376  /**
377   * Reads the cell from a mob file.
378   * The mob file might be located in different directories.
379   * 1. The working directory.
380   * 2. The archive directory.
381   * Reads the cell from the files located in both of the above directories.
382   * @param locations The possible locations where the mob files are saved.
383   * @param fileName The file to be read.
384   * @param search The cell to be searched.
385   * @param cacheMobBlocks Whether the scanner should cache blocks.
386   * @param readPt the read point.
387   * @param readEmptyValueOnMobCellMiss Whether return null value when the mob file is
388   *        missing or corrupt.
389   * @return The found cell. Null if there's no such a cell.
390   * @throws IOException
391   */
392  private Cell readCell(List<Path> locations, String fileName, Cell search, boolean cacheMobBlocks,
393    long readPt, boolean readEmptyValueOnMobCellMiss) throws IOException {
394    FileSystem fs = getFileSystem();
395    Throwable throwable = null;
396    for (Path location : locations) {
397      MobFile file = null;
398      Path path = new Path(location, fileName);
399      try {
400        file = mobCacheConfig.getMobFileCache().openFile(fs, path, mobCacheConfig);
401        return readPt != -1 ? file.readCell(search, cacheMobBlocks, readPt) : file.readCell(search,
402          cacheMobBlocks);
403      } catch (IOException e) {
404        mobCacheConfig.getMobFileCache().evictFile(fileName);
405        throwable = e;
406        if ((e instanceof FileNotFoundException) ||
407            (e.getCause() instanceof FileNotFoundException)) {
408          LOG.debug("Fail to read the cell, the mob file " + path + " doesn't exist", e);
409        } else if (e instanceof CorruptHFileException) {
410          LOG.error("The mob file " + path + " is corrupt", e);
411          break;
412        } else {
413          throw e;
414        }
415      } catch (NullPointerException e) { // HDFS 1.x - DFSInputStream.getBlockAt()
416        mobCacheConfig.getMobFileCache().evictFile(fileName);
417        LOG.debug("Fail to read the cell", e);
418        throwable = e;
419      } catch (AssertionError e) { // assert in HDFS 1.x - DFSInputStream.getBlockAt()
420        mobCacheConfig.getMobFileCache().evictFile(fileName);
421        LOG.debug("Fail to read the cell", e);
422        throwable = e;
423      } finally {
424        if (file != null) {
425          mobCacheConfig.getMobFileCache().closeFile(file);
426        }
427      }
428    }
429    LOG.error("The mob file " + fileName + " could not be found in the locations " + locations
430      + " or it is corrupt");
431    if (readEmptyValueOnMobCellMiss) {
432      return null;
433    } else if ((throwable instanceof FileNotFoundException)
434        || (throwable.getCause() instanceof FileNotFoundException)) {
435      // The region is re-opened when FileNotFoundException is thrown.
436      // This is not necessary when MOB files cannot be found, because the store files
437      // in a region only contain the references to MOB files and a re-open on a region
438      // doesn't help fix the lost MOB files.
439      throw new DoNotRetryIOException(throwable);
440    } else if (throwable instanceof IOException) {
441      throw (IOException) throwable;
442    } else {
443      throw new IOException(throwable);
444    }
445  }
