048/**
049 * Region is a subset of HRegion with operations required for the {@link RegionCoprocessor
050 * Coprocessors}. The operations include ability to do mutations, requesting compaction, getting
051 * different counters/sizes, locking rows and getting access to {@linkplain Store}s.
052 */
053@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
054@InterfaceStability.Evolving
055public interface Region extends ConfigurationObserver {
056
057  ///////////////////////////////////////////////////////////////////////////
058  // Region state
059
060  /** @return region information for this region */
061  RegionInfo getRegionInfo();
062
063  /** @return table descriptor for this region */
064  TableDescriptor getTableDescriptor();
065
066  /** @return true if region is available (not closed and not closing) */
067  boolean isAvailable();
068
069  /** @return true if region is closed */
070  boolean isClosed();
071
072  /** @return True if closing process has started */
073  boolean isClosing();
074
075  /** @return True if region is in recovering state */
076  boolean isRecovering();
077
078  /** @return True if region is read only */
079  boolean isReadOnly();
080
081  /** @return true if region is splittable */
082  boolean isSplittable();
083
084  /**
085   * @return true if region is mergeable
086   */
087  boolean isMergeable();
088
089  /**
090   * Return the list of Stores managed by this region
091   * <p>Use with caution.  Exposed for use of fixup utilities.
092   * @return a list of the Stores managed by this region
093   */
094  List<? extends Store> getStores();
095
096  /**
097   * Return the Store for the given family
098   * <p>Use with caution.  Exposed for use of fixup utilities.
099   * @return the Store for the given family
100   */
101  Store getStore(byte[] family);
102
103  /** @return list of store file names for the given families */
104  List<String> getStoreFileList(byte[][] columns);
105
106  /**
107   * Check the region's underlying store files, open the files that have not
108   * been opened yet, and remove the store file readers for store files no
109   * longer available.
110   * @throws IOException
111   */
112  boolean refreshStoreFiles() throws IOException;
113
114  /** @return the max sequence id of flushed data on this region; no edit in memory will have
115   * a sequence id that is less that what is returned here.
116   */
117  long getMaxFlushedSeqId();
118
119  /**
120   * This can be used to determine the last time all files of this region were major compacted.
121   * @param majorCompactionOnly Only consider HFile that are the result of major compaction
122   * @return the timestamp of the oldest HFile for all stores of this region
123   */
124  long getOldestHfileTs(boolean majorCompactionOnly) throws IOException;
125
126  /**
127   * @return map of column family names to max sequence id that was read from storage when this
128   * region was opened
129   */
130  public Map<byte[], Long> getMaxStoreSeqId();
131
132  /**
133   * @return The earliest time a store in the region was flushed. All
134   *         other stores in the region would have been flushed either at, or
135   *         after this time.
136   */
137  long getEarliestFlushTimeForAllStores();
138
139  ///////////////////////////////////////////////////////////////////////////
140  // Metrics
141
142  /** @return read requests count for this region */
143  long getReadRequestsCount();
144
145  /** @return filtered read requests count for this region */
146  long getFilteredReadRequestsCount();
147
148  /** @return write request count for this region */
149  long getWriteRequestsCount();
150
151  /**
152   * @return memstore size for this region, in bytes. It just accounts data size of cells added to
153   *         the memstores of this Region. Means size in bytes for key, value and tags within Cells.
154   *         It wont consider any java heap overhead for the cell objects or any other.
155   */
156  long getMemStoreSize();
157
158  /** @return the number of mutations processed bypassing the WAL */
159  long getNumMutationsWithoutWAL();
160
161  /** @return the size of data processed bypassing the WAL, in bytes */
162  long getDataInMemoryWithoutWAL();
163
164  /** @return the number of blocked requests */
165  long getBlockedRequestsCount();
166
167  /** @return the number of checkAndMutate guards that passed */
168  long getCheckAndMutateChecksPassed();
169
170  /** @return the number of failed checkAndMutate guards */
171  long getCheckAndMutateChecksFailed();
172
173  ///////////////////////////////////////////////////////////////////////////
174  // Locking
175
176  // Region read locks
177
178  /**
179   * Operation enum is used in {@link Region#startRegionOperation} and elsewhere to provide
180   * context for various checks.
181   */
182  enum Operation {
183    ANY, GET, PUT, DELETE, SCAN, APPEND, INCREMENT, SPLIT_REGION, MERGE_REGION, BATCH_MUTATE,
184    REPLAY_BATCH_MUTATE, COMPACT_REGION, REPLAY_EVENT, SNAPSHOT
185  }
186
187  /**
188   * This method needs to be called before any public call that reads or
189   * modifies data.
190   * Acquires a read lock and checks if the region is closing or closed.
191   * <p>{@link #closeRegionOperation} MUST then always be called after
192   * the operation has completed, whether it succeeded or failed.
193   * @throws IOException
194   */
195  // TODO Exposing this and closeRegionOperation() as we have getRowLock() exposed.
196  // Remove if we get rid of exposing getRowLock().
197  void startRegionOperation() throws IOException;
198
199  /**
200   * This method needs to be called before any public call that reads or
201   * modifies data.
202   * Acquires a read lock and checks if the region is closing or closed.
203   * <p>{@link #closeRegionOperation} MUST then always be called after
204   * the operation has completed, whether it succeeded or failed.
205   * @param op The operation is about to be taken on the region
206   * @throws IOException
207   */
208  void startRegionOperation(Operation op) throws IOException;
209
210  /**
211   * Closes the region operation lock.
212   * @throws IOException
213   */
214  void closeRegionOperation() throws IOException;
215
216  /**
217   * Closes the region operation lock. This needs to be called in the finally block corresponding
218   * to the try block of {@link #startRegionOperation(Operation)}
219   * @throws IOException
220   */
221  void closeRegionOperation(Operation op) throws IOException;
222
223  // Row write locks
224
225  /**
226   * Row lock held by a given thread.
227   * One thread may acquire multiple locks on the same row simultaneously.
228   * The locks must be released by calling release() from the same thread.
229   */
230  public interface RowLock {
231    /**
232     * Release the given lock.  If there are no remaining locks held by the current thread
233     * then unlock the row and allow other threads to acquire the lock.
234     * @throws IllegalArgumentException if called by a different thread than the lock owning
235     *     thread
236     */
237    void release();
238  }
239
240  /**
241   *
242   * Get a row lock for the specified row. All locks are reentrant.
243   *
244   * Before calling this function make sure that a region operation has already been
245   * started (the calling thread has already acquired the region-close-guard lock).
246   * <p>
247   * The obtained locks should be released after use by {@link RowLock#release()}
248   * <p>
249   * NOTE: the boolean passed here has changed. It used to be a boolean that
250   * stated whether or not to wait on the lock. Now it is whether it an exclusive
251   * lock is requested.
252   *
253   * @param row The row actions will be performed against
254   * @param readLock is the lock reader or writer. True indicates that a non-exclusive
255   * lock is requested
256   * @see #startRegionOperation()
257   * @see #startRegionOperation(Operation)
258   */
259  // TODO this needs to be exposed as we have RowProcessor now. If RowProcessor is removed, we can
260  // remove this too..
261  RowLock getRowLock(byte[] row, boolean readLock) throws IOException;
262
263  ///////////////////////////////////////////////////////////////////////////
264  // Region operations
265
266  /**
267   * Perform one or more append operations on a row.
268   * @param append
269   * @return result of the operation
270   * @throws IOException
271   */
272  Result append(Append append) throws IOException;
273
274  /**
275   * Perform a batch of mutations.
276   * <p>
277   * Note this supports only Put and Delete mutations and will ignore other types passed.
278   * @param mutations the list of mutations
279   * @return an array of OperationStatus which internally contains the
280   *         OperationStatusCode and the exceptionMessage if any.
281   * @throws IOException
282   */
283  OperationStatus[] batchMutate(Mutation[] mutations)
284      throws IOException;
285
286  /**
287   * Atomically checks if a row/family/qualifier value matches the expected value and if it does,
288   * it performs the mutation. If the passed value is null, the lack of column value
289   * (ie: non-existence) is used. See checkAndRowMutate to do many checkAndPuts at a time on a
290   * single row.
291   * @param row to check
292   * @param family column family to check
293   * @param qualifier column qualifier to check
294   * @param op the comparison operator
295   * @param comparator
296   * @param mutation
297   * @param writeToWAL
298   * @return true if mutation was applied, false otherwise
299   * @throws IOException
300   */
301  boolean checkAndMutate(byte [] row, byte [] family, byte [] qualifier, CompareOperator op,
302      ByteArrayComparable comparator, Mutation mutation, boolean writeToWAL) throws IOException;
303
304  /**
305   * Atomically checks if a row/family/qualifier value matches the expected values and if it does,
306   * it performs the row mutations. If the passed value is null, the lack of column value
307   * (ie: non-existence) is used. Use to do many mutations on a single row. Use checkAndMutate
308   * to do one checkAndMutate at a time.
309   * @param row to check
310   * @param family column family to check
311   * @param qualifier column qualifier to check
312   * @param op the comparison operator
313   * @param comparator
314   * @param mutations
315   * @param writeToWAL
316   * @return true if mutations were applied, false otherwise
317   * @throws IOException
318   */
319  boolean checkAndRowMutate(byte [] row, byte [] family, byte [] qualifier, CompareOperator op,
320      ByteArrayComparable comparator, RowMutations mutations, boolean writeToWAL)
321      throws IOException;
322
323  /**
324   * Deletes the specified cells/row.
325   * @param delete
326   * @throws IOException
327   */
328  void delete(Delete delete) throws IOException;
329
330  /**
331   * Do a get based on the get parameter.
332   * @param get query parameters
333   * @return result of the operation
334   */
335  Result get(Get get) throws IOException;
336
337  /**
338   * Do a get based on the get parameter.
339   * @param get query parameters
340   * @param withCoprocessor invoke coprocessor or not. We don't want to
341   * always invoke cp.
342   * @return list of cells resulting from the operation
343   */
344  List<Cell> get(Get get, boolean withCoprocessor) throws IOException;
345
346  /**
347   * Return an iterator that scans over the HRegion, returning the indicated
348   * columns and rows specified by the {@link Scan}.
349   * <p>
350   * This Iterator must be closed by the caller.
351   *
352   * @param scan configured {@link Scan}
353   * @return RegionScanner
354   * @throws IOException read exceptions
355   */
356  RegionScanner getScanner(Scan scan) throws IOException;
357
358  /**
359   * Return an iterator that scans over the HRegion, returning the indicated columns and rows
360   * specified by the {@link Scan}. The scanner will also include the additional scanners passed
361   * along with the scanners for the specified Scan instance. Should be careful with the usage to
362   * pass additional scanners only within this Region
363   * <p>
364   * This Iterator must be closed by the caller.
365   *
366   * @param scan configured {@link Scan}
367   * @param additionalScanners Any additional scanners to be used
368   * @return RegionScanner
369   * @throws IOException read exceptions
370   */
371  RegionScanner getScanner(Scan scan, List<KeyValueScanner> additionalScanners) throws IOException;
372
373  /** The comparator to be used with the region */
374  CellComparator getCellComparator();
375
376  /**
377   * Perform one or more increment operations on a row.
378   * @param increment
379   * @return result of the operation
380   * @throws IOException
381   */
382  Result increment(Increment increment) throws IOException;
383
384  /**
385   * Performs multiple mutations atomically on a single row. Currently
386   * {@link Put} and {@link Delete} are supported.
387   *
388   * @param mutations object that specifies the set of mutations to perform atomically
389   * @throws IOException
390   */
391  void mutateRow(RowMutations mutations) throws IOException;
392
393  /**
394   * Perform atomic mutations within the region.
395   *
396   * @param mutations The list of mutations to perform.
397   * <code>mutations</code> can contain operations for multiple rows.
398   * Caller has to ensure that all rows are contained in this region.
399   * @param rowsToLock Rows to lock
400   * @param nonceGroup Optional nonce group of the operation (client Id)
401   * @param nonce Optional nonce of the operation (unique random id to ensure "more idempotence")
402   * If multiple rows are locked care should be taken that
403   * <code>rowsToLock</code> is sorted in order to avoid deadlocks.
404   * @throws IOException
405   */
406  // TODO Should not be exposing with params nonceGroup, nonce. Change when doing the jira for
407  // Changing processRowsWithLocks and RowProcessor
408  void mutateRowsWithLocks(Collection<Mutation> mutations, Collection<byte[]> rowsToLock,
409      long nonceGroup, long nonce) throws IOException;
410
411  /**
412   * Performs atomic multiple reads and writes on a given row.
413   *
414   * @param processor The object defines the reads and writes to a row.
415   */
416  void processRowsWithLocks(RowProcessor<?,?> processor) throws IOException;
417
418  /**
419   * Performs atomic multiple reads and writes on a given row.
420   *
421   * @param processor The object defines the reads and writes to a row.
422   * @param nonceGroup Optional nonce group of the operation (client Id)
423   * @param nonce Optional nonce of the operation (unique random id to ensure "more idempotence")
424   */
425  // TODO Should not be exposing with params nonceGroup, nonce. Change when doing the jira for
426  // Changing processRowsWithLocks and RowProcessor
427  void processRowsWithLocks(RowProcessor<?,?> processor, long nonceGroup, long nonce)
428      throws IOException;
429
430  /**
431   * Performs atomic multiple reads and writes on a given row.
432   *
433   * @param processor The object defines the reads and writes to a row.
434   * @param timeout The timeout of the processor.process() execution
435   *                Use a negative number to switch off the time bound
436   * @param nonceGroup Optional nonce group of the operation (client Id)
437   * @param nonce Optional nonce of the operation (unique random id to ensure "more idempotence")
438   */
439  // TODO Should not be exposing with params nonceGroup, nonce. Change when doing the jira for
440  // Changing processRowsWithLocks and RowProcessor
441  void processRowsWithLocks(RowProcessor<?,?> processor, long timeout, long nonceGroup, long nonce)
442      throws IOException;
443
444  /**
445   * Puts some data in the table.
446   * @param put
447   * @throws IOException
448   */
449  void put(Put put) throws IOException;

469  void requestCompaction(byte[] family, String why, int priority, boolean major,

473   * Request flush on this region.

475  void requestFlush(FlushLifeCycleTracker tracker) throws IOException;
