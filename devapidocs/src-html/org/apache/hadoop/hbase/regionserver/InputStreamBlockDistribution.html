016package org.apache.hadoop.hbase.regionserver;
017
018import com.google.errorprone.annotations.RestrictedApi;
019import java.io.IOException;
020import org.apache.hadoop.conf.Configuration;
021import org.apache.hadoop.fs.FSDataInputStream;
022import org.apache.hadoop.hbase.HDFSBlocksDistribution;
023import org.apache.hadoop.hbase.io.FileLink;
024import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
025import org.apache.hadoop.hbase.util.FSUtils;
026import org.apache.hadoop.hdfs.client.HdfsDataInputStream;
027import org.apache.yetus.audience.InterfaceAudience;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031/**
032 * Computes the HDFSBlockDistribution for a file based on the underlying located blocks
033 * for an HdfsDataInputStream reading that file. The backing DFSInputStream.getAllBlocks involves
034 * allocating an array of numBlocks size per call. It may also involve calling the namenode, if
035 * the DFSInputStream has not fetched all the blocks yet. In order to avoid allocation pressure,
036 * we cache the computed distribution for a configurable period of time.
037 * <p>
038 * This class only gets instantiated for the <b>first</b> FSDataInputStream of each StoreFile (i.e.
039 * the one backing {@link HStoreFile#initialReader}). It's then used to dynamically update the
040 * value returned by {@link HStoreFile#getHDFSBlockDistribution()}.
041 * <p>
042 * Once the backing FSDataInputStream is closed, we should not expect the distribution result
043 * to change anymore. This is ok becuase the initialReader's InputStream is only closed when the
044 * StoreFile itself is closed, at which point nothing will be querying getHDFSBlockDistribution
045 * anymore. If/When the StoreFile is reopened, a new {@link InputStreamBlockDistribution} will
046 * be created for the new initialReader.
047 */
048@InterfaceAudience.Private
049public class InputStreamBlockDistribution {
050  private static final Logger LOG = LoggerFactory.getLogger(InputStreamBlockDistribution.class);
051
052  private static final String HBASE_LOCALITY_INPUTSTREAM_DERIVE_ENABLED =
053    "hbase.locality.inputstream.derive.enabled";
054  private static final boolean DEFAULT_HBASE_LOCALITY_INPUTSTREAM_DERIVE_ENABLED = false;
055
056  private static final String HBASE_LOCALITY_INPUTSTREAM_DERIVE_CACHE_PERIOD =
057    "hbase.locality.inputstream.derive.cache.period";
058  private static final int DEFAULT_HBASE_LOCALITY_INPUTSTREAM_DERIVE_CACHE_PERIOD = 60_000;
059
060  private final FSDataInputStream stream;
061  private final StoreFileInfo fileInfo;
062  private final int cachePeriodMs;
063
064  private HDFSBlocksDistribution hdfsBlocksDistribution;
065  private long lastCachedAt;
066  private boolean streamUnsupported;
067
068  /**
069   * This should only be called for the first FSDataInputStream of a StoreFile,
070   * in {@link HStoreFile#open()}.
071   *
072   * @see InputStreamBlockDistribution
073   * @param stream the input stream to derive locality from
074   * @param fileInfo the StoreFileInfo for the related store file
075   */
076  public InputStreamBlockDistribution(FSDataInputStream stream, StoreFileInfo fileInfo) {
077    this.stream = stream;
078    this.fileInfo = fileInfo;
079    this.cachePeriodMs = fileInfo.getConf().getInt(
080      HBASE_LOCALITY_INPUTSTREAM_DERIVE_CACHE_PERIOD,
081      DEFAULT_HBASE_LOCALITY_INPUTSTREAM_DERIVE_CACHE_PERIOD);
082    this.lastCachedAt = EnvironmentEdgeManager.currentTime();
083    this.streamUnsupported = false;
084    this.hdfsBlocksDistribution = fileInfo.getHDFSBlockDistribution();
085  }
086
087  /**
088   * True if we should derive StoreFile HDFSBlockDistribution from the underlying input stream
089   */
090  public static boolean isEnabled(Configuration conf) {
091    return conf.getBoolean(HBASE_LOCALITY_INPUTSTREAM_DERIVE_ENABLED,
092      DEFAULT_HBASE_LOCALITY_INPUTSTREAM_DERIVE_ENABLED);
093  }
094
095  /**
096   * Get the HDFSBlocksDistribution derived from the StoreFile input stream, re-computing if cache
097   * is expired.
098   */
099  public synchronized HDFSBlocksDistribution getHDFSBlockDistribution() {
100    if (EnvironmentEdgeManager.currentTime() - lastCachedAt > cachePeriodMs) {
101      try {
102        LOG.debug("Refreshing HDFSBlockDistribution for {}", fileInfo);
103        computeBlockDistribution();
104      } catch (IOException e) {
105        LOG.warn("Failed to recompute block distribution for {}. Falling back on cached value.",
106          fileInfo, e);
107      }
108    }
109    return hdfsBlocksDistribution;
110  }
111
112  private void computeBlockDistribution() throws IOException {
113    lastCachedAt = EnvironmentEdgeManager.currentTime();
114
115    FSDataInputStream stream;
116    if (fileInfo.isLink()) {
117      stream = FileLink.getUnderlyingFileLinkInputStream(this.stream);
118    } else {
119      stream = this.stream;
120    }
121
122    if (!(stream instanceof HdfsDataInputStream)) {
123      if (!streamUnsupported) {
124        LOG.warn("{} for storeFileInfo={}, isLink={}, is not an HdfsDataInputStream so cannot be "
125            + "used to derive locality. Falling back on cached value.",
126          stream, fileInfo, fileInfo.isLink());
127        streamUnsupported = true;
128      }
129      return;
130    }
131
132    streamUnsupported = false;
133    hdfsBlocksDistribution = FSUtils.computeHDFSBlocksDistribution((HdfsDataInputStream) stream);
134  }
135
136  /**
137   * For tests only, sets lastCachedAt so we can force a refresh
138   */
139  @RestrictedApi(explanation = "Should only be called in tests", link = "",
140    allowedOnPath = ".*/src/test/.*")
141  synchronized void setLastCachedAt(long timestamp) {
142    lastCachedAt = timestamp;
143  }
144
145  /**
146   * For tests only, returns the configured cache period
147   */
148  @RestrictedApi(explanation = "Should only be called in tests", link = "",
149    allowedOnPath = ".*/src/test/.*")
150  long getCachePeriodMs() {
151    return cachePeriodMs;
152  }
153
154  /**
155   * For tests only, returns whether the passed stream is supported
156   */
157  @RestrictedApi(explanation = "Should only be called in tests", link = "",
158    allowedOnPath = ".*/src/test/.*")
159  boolean isStreamUnsupported() {
160    return streamUnsupported;
161  }
162}
