559      this.msgInterval = conf.getInt("hbase.regionserver.msginterval", 3 * 1000);

619          this.csm = new ZkCoordinatedStateManager(this);

622          masterAddressTracker.start();
623

625          clusterStatusTracker.start();
626        } else {
627          masterAddressTracker = null;
628          clusterStatusTracker = null;
629        }
630      } else {
631        zooKeeper = null;
632        masterAddressTracker = null;
633        clusterStatusTracker = null;
634      }

637      // and executor being created and takes a different startup route. Lots of overlap between HRS
638      // and M (An M IS A HRS now). Need to refactor so less duplication between M and its super

646      // cause of failed startup is lost.

648      throw t;

839  /**
840   * Bring up connection to zk ensemble and then wait until a master for this cluster and then after
841   * that, wait until cluster 'up' flag has been set. This is the order in which master does things.
842   * <p>
843   * Finally open long-living server short-circuit connection.
844   */
845  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="RV_RETURN_VALUE_IGNORED_BAD_PRACTICE",
846    justification="cluster Id znode read would give us correct response")
847  private void initializeZooKeeper() throws IOException, InterruptedException {
848    // Nothing to do in here if no Master in the mix.
849    if (this.masterless) {
850      return;
851    }
852
853    // Create the master address tracker, register with zk, and start it.  Then
854    // block until a master is available.  No point in starting up if no master
855    // running.
856    blockAndCheckIfStopped(this.masterAddressTracker);
857
858    // Wait on cluster being up.  Master will set this flag up in zookeeper
859    // when ready.
860    blockAndCheckIfStopped(this.clusterStatusTracker);
861
862    // If we are HMaster then the cluster id should have already been set.
863    if (clusterId == null) {
864      // Retrieve clusterId
865      // Since cluster status is now up
866      // ID should have already been set by HMaster
867      try {
868        clusterId = ZKClusterId.readClusterIdZNode(this.zooKeeper);
869        if (clusterId == null) {
870          this.abort("Cluster ID has not been set");
871        }
872        LOG.info("ClusterId : " + clusterId);
873      } catch (KeeperException e) {
874        this.abort("Failed to retrieve Cluster ID", e);
875      }
876    }
877
878    // In case colocated master, wait here till it's active.
879    // So backup masters won't start as regionservers.
880    // This is to avoid showing backup masters as regionservers
881    // in master web UI, or assigning any region to them.
882    waitForMasterActive();
883    if (isStopped() || isAborted()) {
884      return; // No need for further initialization
885    }
886
887    // watch for snapshots and other procedures
888    try {
889      rspmHost = new RegionServerProcedureManagerHost();
890      rspmHost.loadProcedures(conf);
891      rspmHost.initialize(this);
892    } catch (KeeperException e) {
893      this.abort("Failed to reach coordination cluster when creating procedure handler.", e);
894    }
895  }

922  /**
923   * The HRegionServer sticks in this loop until closed.
924   */
925  @Override
926  public void run() {
927    try {
928      // Do pre-registration initializations; zookeeper, lease threads, etc.
929      preRegistrationInitialization();
930    } catch (Throwable e) {
931      abort("Fatal exception during initialization", e);
932    }
933
934    try {
935      if (!isStopped() && !isAborted()) {
936        ShutdownHook.install(conf, fs, this, Thread.currentThread());
937        // Initialize the RegionServerCoprocessorHost now that our ephemeral
938        // node was created, in case any coprocessors want to use ZooKeeper
939        this.rsHost = new RegionServerCoprocessorHost(this, this.conf);
940      }
941
942      // Try and register with the Master; tell it we are here.  Break if
943      // server is stopped or the clusterup flag is down or hdfs went wacky.
944      // Once registered successfully, go ahead and start up all Services.
945      while (keepLooping()) {
946        RegionServerStartupResponse w = reportForDuty();
947        if (w == null) {
948          LOG.warn("reportForDuty failed; sleeping and then retrying.");
949          this.sleeper.sleep();
950        } else {
951          handleReportForDutyResponse(w);
952          break;
953        }
954      }
955
956      if (!isStopped() && isHealthy()) {
957        // start the snapshot handler and other procedure handlers,
958        // since the server is ready to run
959        if (this.rspmHost != null) {
960          this.rspmHost.start();
961        }
962        // Start the Quota Manager
963        if (this.rsQuotaManager != null) {
964          rsQuotaManager.start(getRpcServer().getScheduler());
965        }
966        if (this.rsSpaceQuotaManager != null) {
967          this.rsSpaceQuotaManager.start();
968        }
969      }
970
971      // We registered with the Master.  Go into run mode.
972      long lastMsg = System.currentTimeMillis();
973      long oldRequestCount = -1;
974      // The main run loop.
975      while (!isStopped() && isHealthy()) {
976        if (!isClusterUp()) {
977          if (isOnlineRegionsEmpty()) {
978            stop("Exiting; cluster shutdown set and not carrying any regions");
979          } else if (!this.stopping) {
980            this.stopping = true;
981            LOG.info("Closing user regions");
982            closeUserRegions(this.abortRequested);
983          } else if (this.stopping) {
984            boolean allUserRegionsOffline = areAllUserRegionsOffline();
985            if (allUserRegionsOffline) {
986              // Set stopped if no more write requests tp meta tables
987              // since last time we went around the loop.  Any open
988              // meta regions will be closed on our way out.
989              if (oldRequestCount == getWriteRequestCount()) {
990                stop("Stopped; only catalog regions remaining online");
991                break;
992              }
993              oldRequestCount = getWriteRequestCount();
994            } else {
995              // Make sure all regions have been closed -- some regions may
996              // have not got it because we were splitting at the time of
997              // the call to closeUserRegions.
998              closeUserRegions(this.abortRequested);
999            }
1000            LOG.debug("Waiting on " + getOnlineRegionsAsPrintableString());
1001          }
1002        }
1003        long now = System.currentTimeMillis();
1004        if ((now - lastMsg) >= msgInterval) {
1005          tryRegionServerReport(lastMsg, now);
1006          lastMsg = System.currentTimeMillis();
1007        }
1008        if (!isStopped() && !isAborted()) {
1009          this.sleeper.sleep();
1010        }
1011      } // for
1012    } catch (Throwable t) {
1013      if (!rpcServices.checkOOME(t)) {
1014        String prefix = t instanceof YouAreDeadException? "": "Unhandled: ";
1015        abort(prefix + t.getMessage(), t);
1016      }
1017    }
1018    // Run shutdown.
1019    if (mxBean != null) {
1020      MBeans.unregister(mxBean);
1021      mxBean = null;
1022    }
1023    if (this.leases != null) this.leases.closeAfterLeasesExpire();
1024    if (this.splitLogWorker != null) {
1025      splitLogWorker.stop();
1026    }
1027    if (this.infoServer != null) {
1028      LOG.info("Stopping infoServer");
1029      try {
1030        this.infoServer.stop();
1031      } catch (Exception e) {
1032        LOG.error("Failed to stop infoServer", e);
1033      }
1034    }
1035    // Send cache a shutdown.
1036    if (cacheConfig != null && cacheConfig.isBlockCacheEnabled()) {
1037      cacheConfig.getBlockCache().shutdown();
1038    }
1039    mobCacheConfig.getMobFileCache().shutdown();
1040
1041    if (movedRegionsCleaner != null) {
1042      movedRegionsCleaner.stop("Region Server stopping");
1043    }
1044
1045    // Send interrupts to wake up threads if sleeping so they notice shutdown.
1046    // TODO: Should we check they are alive? If OOME could have exited already
1047    if (this.hMemManager != null) this.hMemManager.stop();
1048    if (this.cacheFlusher != null) this.cacheFlusher.interruptIfNecessary();
1049    if (this.compactSplitThread != null) this.compactSplitThread.interruptIfNecessary();
1050    if (this.compactionChecker != null) this.compactionChecker.cancel(true);
1051    if (this.healthCheckChore != null) this.healthCheckChore.cancel(true);
1052    if (this.nonceManagerChore != null) this.nonceManagerChore.cancel(true);
1053    if (this.storefileRefresher != null) this.storefileRefresher.cancel(true);
1054    sendShutdownInterrupt();
1055
1056    // Stop the quota manager
1057    if (rsQuotaManager != null) {
1058      rsQuotaManager.stop();
1059    }
1060    if (rsSpaceQuotaManager != null) {
1061      rsSpaceQuotaManager.stop();
1062      rsSpaceQuotaManager = null;
1063    }
1064
1065    // Stop the snapshot and other procedure handlers, forcefully killing all running tasks
1066    if (rspmHost != null) {
1067      rspmHost.stop(this.abortRequested || this.killed);
1068    }
1069
1070    if (this.killed) {
1071      // Just skip out w/o closing regions.  Used when testing.
1072    } else if (abortRequested) {
1073      if (this.fsOk) {
1074        closeUserRegions(abortRequested); // Don't leave any open file handles
1075      }
1076      LOG.info("aborting server " + this.serverName);
1077    } else {
1078      closeUserRegions(abortRequested);
1079      LOG.info("stopping server " + this.serverName);
1080    }
1081
1082    // so callers waiting for meta without timeout can stop
1083    if (this.metaTableLocator != null) this.metaTableLocator.stop();
1084    if (this.clusterConnection != null && !clusterConnection.isClosed()) {
1085      try {
1086        this.clusterConnection.close();
1087      } catch (IOException e) {
1088        // Although the {@link Closeable} interface throws an {@link
1089        // IOException}, in reality, the implementation would never do that.
1090        LOG.warn("Attempt to close server's short circuit ClusterConnection failed.", e);
1091      }
1092    }
1093
1094    // Closing the compactSplit thread before closing meta regions
1095    if (!this.killed && containsMetaTableRegions()) {
1096      if (!abortRequested || this.fsOk) {
1097        if (this.compactSplitThread != null) {
1098          this.compactSplitThread.join();
1099          this.compactSplitThread = null;
1100        }
1101        closeMetaTableRegions(abortRequested);
1102      }
1103    }
1104
1105    if (!this.killed && this.fsOk) {
1106      waitOnAllRegionsToClose(abortRequested);
1107      LOG.info("stopping server " + this.serverName + "; all regions closed.");
1108    }
1109
1110    //fsOk flag may be changed when closing regions throws exception.
1111    if (this.fsOk) {
1112      shutdownWAL(!abortRequested);
1113    }
1114
1115    // Make sure the proxy is down.
1116    if (this.rssStub != null) {
1117      this.rssStub = null;
1118    }
1119    if (this.lockStub != null) {
1120      this.lockStub = null;
1121    }
1122    if (this.rpcClient != null) {
1123      this.rpcClient.close();
1124    }
1125    if (this.leases != null) {
1126      this.leases.close();
1127    }
1128    if (this.pauseMonitor != null) {
1129      this.pauseMonitor.stop();
1130    }
1131
1132    if (!killed) {
1133      stopServiceThreads();
1134    }
1135
1136    if (this.rpcServices != null) {
1137      this.rpcServices.stop();
1138    }
1139
1140    try {
1141      deleteMyEphemeralNode();
1142    } catch (KeeperException.NoNodeException nn) {
1143    } catch (KeeperException e) {
1144      LOG.warn("Failed deleting my ephemeral node", e);
1145    }
1146    // We may have failed to delete the znode at the previous step, but
1147    //  we delete the file anyway: a second attempt to delete the znode is likely to fail again.
1148    ZNodeClearer.deleteMyEphemeralNodeOnDisk();
1149
1150    if (this.zooKeeper != null) {
1151      this.zooKeeper.close();
1152    }
1153    LOG.info("Exiting; stopping=" + this.serverName + "; zookeeper connection closed.");
1154  }

1302  ClusterStatusProtos.ServerLoad buildServerLoad(long reportStartTime, long reportEndTime)
1303      throws IOException {
1304    // We're getting the MetricsRegionServerWrapper here because the wrapper computes requests
1305    // per second, and other metrics  As long as metrics are part of ServerLoad it's best to use
1306    // the wrapper to compute those numbers in one place.
1307    // In the long term most of these should be moved off of ServerLoad and the heart beat.
1308    // Instead they should be stored in an HBase table so that external visibility into HBase is
1309    // improved; Additionally the load balancer will be able to take advantage of a more complete
1310    // history.
1311    MetricsRegionServerWrapper regionServerWrapper = metricsRegionServer.getRegionServerWrapper();
1312    Collection<HRegion> regions = getOnlineRegionsLocalContext();
1313    long usedMemory = -1L;
1314    long maxMemory = -1L;
1315    final MemoryUsage usage = MemorySizeUtil.safeGetHeapMemoryUsage();
1316    if (usage != null) {
1317      usedMemory = usage.getUsed();
1318      maxMemory = usage.getMax();
1319    }
1320
1321    ClusterStatusProtos.ServerLoad.Builder serverLoad = ClusterStatusProtos.ServerLoad.newBuilder();
1322    serverLoad.setNumberOfRequests((int) regionServerWrapper.getRequestsPerSecond());
1323    serverLoad.setTotalNumberOfRequests(regionServerWrapper.getTotalRequestCount());
1324    serverLoad.setUsedHeapMB((int)(usedMemory / 1024 / 1024));
1325    serverLoad.setMaxHeapMB((int) (maxMemory / 1024 / 1024));
1326    Set<String> coprocessors = getWAL(null).getCoprocessorHost().getCoprocessors();
1327    Builder coprocessorBuilder = Coprocessor.newBuilder();
1328    for (String coprocessor : coprocessors) {
1329      serverLoad.addCoprocessors(coprocessorBuilder.setName(coprocessor).build());
1330    }
1331    RegionLoad.Builder regionLoadBldr = RegionLoad.newBuilder();
1332    RegionSpecifier.Builder regionSpecifier = RegionSpecifier.newBuilder();
1333    for (HRegion region : regions) {
1334      if (region.getCoprocessorHost() != null) {
1335        Set<String> regionCoprocessors = region.getCoprocessorHost().getCoprocessors();
1336        Iterator<String> iterator = regionCoprocessors.iterator();
1337        while (iterator.hasNext()) {
1338          serverLoad.addCoprocessors(coprocessorBuilder.setName(iterator.next()).build());
1339        }
1340      }
1341      serverLoad.addRegionLoads(createRegionLoad(region, regionLoadBldr, regionSpecifier));
1342      for (String coprocessor : getWAL(region.getRegionInfo()).getCoprocessorHost()
1343          .getCoprocessors()) {
1344        serverLoad.addCoprocessors(coprocessorBuilder.setName(coprocessor).build());
1345      }
1346    }
1347    serverLoad.setReportStartTime(reportStartTime);
1348    serverLoad.setReportEndTime(reportEndTime);
1349    if (this.infoServer != null) {
1350      serverLoad.setInfoServerPort(this.infoServer.getPort());
1351    } else {
1352      serverLoad.setInfoServerPort(-1);
1353    }
1354
1355    // for the replicationLoad purpose. Only need to get from one executorService
1356    // either source or sink will get the same info
1357    ReplicationSourceService rsources = getReplicationSourceService();
1358
1359    if (rsources != null) {
1360      // always refresh first to get the latest value
1361      ReplicationLoad rLoad = rsources.refreshAndGetReplicationLoad();
1362      if (rLoad != null) {
1363        serverLoad.setReplLoadSink(rLoad.getReplicationLoadSink());
1364        for (ClusterStatusProtos.ReplicationLoadSource rLS : rLoad.getReplicationLoadSourceList()) {
1365          serverLoad.addReplLoadSource(rLS);
1366        }
1367      }
1368    }
1369
1370    return serverLoad.build();
1371  }

1382  /**
1383   * Wait on regions close.
1384   */
1385  private void waitOnAllRegionsToClose(final boolean abort) {
1386    // Wait till all regions are closed before going out.
1387    int lastCount = -1;
1388    long previousLogTime = 0;
1389    Set<String> closedRegions = new HashSet<>();
1390    boolean interrupted = false;
1391    try {
1392      while (!isOnlineRegionsEmpty()) {
1393        int count = getNumberOfOnlineRegions();
1394        // Only print a message if the count of regions has changed.
1395        if (count != lastCount) {
1396          // Log every second at most
1397          if (System.currentTimeMillis() > (previousLogTime + 1000)) {
1398            previousLogTime = System.currentTimeMillis();
1399            lastCount = count;
1400            LOG.info("Waiting on " + count + " regions to close");
1401            // Only print out regions still closing if a small number else will
1402            // swamp the log.
1403            if (count < 10 && LOG.isDebugEnabled()) {
1404              LOG.debug("Online Regions=" + this.onlineRegions);
1405            }
1406          }
1407        }
1408        // Ensure all user regions have been sent a close. Use this to
1409        // protect against the case where an open comes in after we start the
1410        // iterator of onlineRegions to close all user regions.
1411        for (Map.Entry<String, HRegion> e : this.onlineRegions.entrySet()) {
1412          RegionInfo hri = e.getValue().getRegionInfo();
1413          if (!this.regionsInTransitionInRS.containsKey(hri.getEncodedNameAsBytes()) &&
1414              !closedRegions.contains(hri.getEncodedName())) {
1415            closedRegions.add(hri.getEncodedName());
1416            // Don't update zk with this close transition; pass false.
1417            closeRegionIgnoreErrors(hri, abort);
1418          }
1419        }
1420        // No regions in RIT, we could stop waiting now.
1421        if (this.regionsInTransitionInRS.isEmpty()) {
1422          if (!isOnlineRegionsEmpty()) {
1423            LOG.info("We were exiting though online regions are not empty," +
1424                " because some regions failed closing");
1425          }
1426          break;
1427        }
1428        if (sleep(200)) {
1429          interrupted = true;
1430        }
1431      }
1432    } finally {
1433      if (interrupted) {
1434        Thread.currentThread().interrupt();
1435      }
1436    }
1437  }

1471  protected void handleReportForDutyResponse(final RegionServerStartupResponse c)
1472  throws IOException {
1473    try {
1474      boolean updateRootDir = false;
1475      for (NameStringPair e : c.getMapEntriesList()) {
1476        String key = e.getName();
1477        // The hostname the master sees us as.
1478        if (key.equals(HConstants.KEY_FOR_HOSTNAME_SEEN_BY_MASTER)) {
1479          String hostnameFromMasterPOV = e.getValue();
1480          this.serverName = ServerName.valueOf(hostnameFromMasterPOV, rpcServices.isa.getPort(),
1481              this.startcode);
1482          if (!StringUtils.isBlank(useThisHostnameInstead) &&
1483              !hostnameFromMasterPOV.equals(useThisHostnameInstead)) {
1484            String msg = "Master passed us a different hostname to use; was=" +
1485                this.useThisHostnameInstead + ", but now=" + hostnameFromMasterPOV;
1486            LOG.error(msg);
1487            throw new IOException(msg);
1488          }
1489          if (StringUtils.isBlank(useThisHostnameInstead) &&
1490              !hostnameFromMasterPOV.equals(rpcServices.isa.getHostName())) {
1491            String msg = "Master passed us a different hostname to use; was=" +
1492                rpcServices.isa.getHostName() + ", but now=" + hostnameFromMasterPOV;
1493            LOG.error(msg);
1494          }
1495          continue;
1496        }
1497
1498        String value = e.getValue();
1499        if (key.equals(HConstants.HBASE_DIR)) {
1500          if (value != null && !value.equals(conf.get(HConstants.HBASE_DIR))) {
1501            updateRootDir = true;
1502          }
1503        }
1504
1505        if (LOG.isDebugEnabled()) {
1506          LOG.debug("Config from master: " + key + "=" + value);
1507        }
1508        this.conf.set(key, value);
1509      }
1510      // Set our ephemeral znode up in zookeeper now we have a name.
1511      createMyEphemeralNode();
1512
1513      if (updateRootDir) {
1514        // initialize file system by the config fs.defaultFS and hbase.rootdir from master
1515        initializeFileSystem();
1516      }
1517
1518      // hack! Maps DFSClient => RegionServer for logs.  HDFS made this
1519      // config param for task trackers, but we can piggyback off of it.
1520      if (this.conf.get("mapreduce.task.attempt.id") == null) {
1521        this.conf.set("mapreduce.task.attempt.id", "hb_rs_" + this.serverName.toString());
1522      }
1523
1524      // Save it in a file, this will allow to see if we crash
1525      ZNodeClearer.writeMyEphemeralNodeOnDisk(getMyEphemeralNodePath());
1526
1527      // This call sets up an initialized replication and WAL. Later we start it up.
1528      setupWALAndReplication();
1529      // Init in here rather than in constructor after thread name has been set
1530      this.metricsRegionServer = new MetricsRegionServer(
1531          new MetricsRegionServerWrapperImpl(this), conf);
1532      this.metricsTable = new MetricsTable(new MetricsTableWrapperAggregateImpl(this));
1533      // Now that we have a metrics source, start the pause monitor
1534      this.pauseMonitor = new JvmPauseMonitor(conf, getMetrics().getMetricsSource());
1535      pauseMonitor.start();
1536
1537      // There is a rare case where we do NOT want services to start. Check config.
1538      if (getConfiguration().getBoolean("hbase.regionserver.workers", true)) {
1539        startServices();
1540      }
1541      // In here we start up the replication Service. Above we initialized it. TODO. Reconcile.
1542      // or make sense of it.
1543      startReplicationService();
1544
1545
1546      // Set up ZK
1547      LOG.info("Serving as " + this.serverName + ", RpcServer on " + rpcServices.isa +
1548          ", sessionid=0x" +
1549          Long.toHexString(this.zooKeeper.getRecoverableZooKeeper().getSessionId()));
1550
1551      // Wake up anyone waiting for this server to online
1552      synchronized (online) {
1553        online.set(true);
1554        online.notifyAll();
1555      }
1556    } catch (Throwable e) {
1557      stop("Failed initialization");
1558      throw convertThrowableToIOE(cleanup(e, "Failed init"),
1559          "Region server startup failed");
1560    } finally {
1561      sleeper.skipSleepCycle();
1562    }
1563  }

1692  /*
1693   * Inner class that runs on a long period checking if regions need compaction.
1694   */
1695  private static class CompactionChecker extends ScheduledChore {
1696    private final HRegionServer instance;
1697    private final int majorCompactPriority;
1698    private final static int DEFAULT_PRIORITY = Integer.MAX_VALUE;
1699    //Iteration is 1-based rather than 0-based so we don't check for compaction
1700    // immediately upon region server startup
1701    private long iteration = 1;
1702
1703    CompactionChecker(final HRegionServer h, final int sleepTime, final Stoppable stopper) {
1704      super("CompactionChecker", stopper, sleepTime);
1705      this.instance = h;
1706      LOG.info(this.getName() + " runs every " + Duration.ofMillis(sleepTime));
1707
1708      /* MajorCompactPriority is configurable.
1709       * If not set, the compaction will use default priority.
1710       */
1711      this.majorCompactPriority = this.instance.conf.
1712          getInt("hbase.regionserver.compactionChecker.majorCompactPriority",
1713              DEFAULT_PRIORITY);
1714    }
1715
1716    @Override
1717    protected void chore() {
1718      for (Region r : this.instance.onlineRegions.values()) {
1719        if (r == null) {
1720          continue;
1721        }
1722        HRegion hr = (HRegion) r;
1723        for (HStore s : hr.stores.values()) {
1724          try {
1725            long multiplier = s.getCompactionCheckMultiplier();
1726            assert multiplier > 0;
1727            if (iteration % multiplier != 0) {
1728              continue;
1729            }
1730            if (s.needsCompaction()) {
1731              // Queue a compaction. Will recognize if major is needed.
1732              this.instance.compactSplitThread.requestSystemCompaction(hr, s,
1733                getName() + " requests compaction");
1734            } else if (s.shouldPerformMajorCompaction()) {
1735              s.triggerMajorCompaction();
1736              if (majorCompactPriority == DEFAULT_PRIORITY ||
1737                  majorCompactPriority > hr.getCompactPriority()) {
1738                this.instance.compactSplitThread.requestCompaction(hr, s,
1739                    getName() + " requests major compaction; use default priority",
1740                    Store.NO_PRIORITY,
1741                CompactionLifeCycleTracker.DUMMY, null);
1742              } else {
1743                this.instance.compactSplitThread.requestCompaction(hr, s,
1744                    getName() + " requests major compaction; use configured priority",
1745                    this.majorCompactPriority, CompactionLifeCycleTracker.DUMMY, null);
1746              }
1747            }
1748          } catch (IOException e) {
1749            LOG.warn("Failed major compaction check on " + r, e);
1750          }
1751        }
1752      }
1753      iteration = (iteration == Long.MAX_VALUE) ? 0 : (iteration + 1);
1754    }

1757  static class PeriodicMemStoreFlusher extends ScheduledChore {
1758    final HRegionServer server;
1759    final static int RANGE_OF_DELAY = 5 * 60 * 1000; // 5 min in milliseconds
1760    final static int MIN_DELAY_TIME = 0; // millisec
1761    public PeriodicMemStoreFlusher(int cacheFlushInterval, final HRegionServer server) {
1762      super("MemstoreFlusherChore", server, cacheFlushInterval);
1763      this.server = server;
1764    }
1765
1766    @Override
1767    protected void chore() {
1768      final StringBuilder whyFlush = new StringBuilder();
1769      for (HRegion r : this.server.onlineRegions.values()) {
1770        if (r == null) continue;
1771        if (r.shouldFlush(whyFlush)) {
1772          FlushRequester requester = server.getFlushRequester();
1773          if (requester != null) {
1774            long randomDelay = (long) RandomUtils.nextInt(0, RANGE_OF_DELAY) + MIN_DELAY_TIME;
1775            LOG.info(getName() + " requesting flush of " +
1776              r.getRegionInfo().getRegionNameAsString() + " because " +
1777              whyFlush.toString() +
1778              " after random delay " + randomDelay + "ms");
1779            //Throttle the flushes by putting a delay. If we don't throttle, and there
1780            //is a balanced write-load on the regions in a table, we might end up
1781            //overwhelming the filesystem with too many flushes at once.
1782            requester.requestDelayedFlush(r, randomDelay, false);
1783          }
1784        }
1785      }
1786    }
1787  }

2244      ReportRegionStateTransitionRequest.newBuilder();

2246    RegionStateTransition.Builder transition = builder.addTransitionBuilder();
2247    transition.setTransitionCode(code);

2249      transition.setOpenSeqNum(openSeqNum);

2255    int tries = 0;
2256    long pauseTime = INIT_PAUSE_TIME_MS;
2257    // Keep looping till we get an error. We want to send reports even though server is going down.
2258    // Only go down if clusterConnection is null. It is set to null almost as last thing as the

2260    while (this.clusterConnection != null && !this.clusterConnection.isClosed()) {

2274        // know if were successful after an attempt showed in logs as failed.

2283            ioe instanceof PleaseHoldException;

2285          // Do backoff else we flood the Master with requests.
2286          pauseTime = ConnectionUtils.getPauseTime(INIT_PAUSE_TIME_MS, tries);
2287        } else {
2288          pauseTime = INIT_PAUSE_TIME_MS; // Reset.
2289        }

2291          TextFormat.shortDebugString(request) + "; retry (#" + tries + ")" +

2293                " after " + pauseTime + "ms delay (Master is coming online...).":
2294                " immediately."),
2295            ioe);

3110   * Close asynchronously a region, can be called from the master or internally by the regionserver
3111   * when stopping. If called from the master, the region will update the znode status.
3112   *
3113   * <p>
3114   * If an opening was in progress, this method will cancel it, but will not start a new close. The
3115   * coprocessors are not called in this case. A NotServingRegionException exception is thrown.
3116   * </p>
3117
3118   * <p>
3119   *   If a close was in progress, this new request will be ignored, and an exception thrown.
3120   * </p>
3121   *
3122   * @param encodedName Region to close
3123   * @param abort True if we are aborting
3124   * @return True if closed a region.
3125   * @throws NotServingRegionException if the region is not online
3126   */
3127  protected boolean closeRegion(String encodedName, final boolean abort, final ServerName sn)
3128      throws NotServingRegionException {
3129    //Check for permissions to close.
3130    HRegion actualRegion = this.getRegion(encodedName);
3131    // Can be null if we're calling close on a region that's not online
3132    if ((actualRegion != null) && (actualRegion.getCoprocessorHost() != null)) {
3133      try {
3134        actualRegion.getCoprocessorHost().preClose(false);
3135      } catch (IOException exp) {
3136        LOG.warn("Unable to close region: the coprocessor launched an error ", exp);
3137        return false;
3138      }
3139    }
3140
3141    final Boolean previous = this.regionsInTransitionInRS.putIfAbsent(Bytes.toBytes(encodedName),
3142        Boolean.FALSE);
3143
3144    if (Boolean.TRUE.equals(previous)) {
3145      LOG.info("Received CLOSE for the region:" + encodedName + " , which we are already " +
3146          "trying to OPEN. Cancelling OPENING.");
3147      if (!regionsInTransitionInRS.replace(Bytes.toBytes(encodedName), previous, Boolean.FALSE)) {
3148        // The replace failed. That should be an exceptional case, but theoretically it can happen.
3149        // We're going to try to do a standard close then.
3150        LOG.warn("The opening for region " + encodedName + " was done before we could cancel it." +
3151            " Doing a standard close now");
3152        return closeRegion(encodedName, abort, sn);
3153      }
3154      // Let's get the region from the online region list again
3155      actualRegion = this.getRegion(encodedName);
3156      if (actualRegion == null) { // If already online, we still need to close it.
3157        LOG.info("The opening previously in progress has been cancelled by a CLOSE request.");
3158        // The master deletes the znode when it receives this exception.
3159        throw new NotServingRegionException("The region " + encodedName +
3160          " was opening but not yet served. Opening is cancelled.");
3161      }
3162    } else if (Boolean.FALSE.equals(previous)) {
3163      LOG.info("Received CLOSE for the region: " + encodedName +
3164        ", which we are already trying to CLOSE, but not completed yet");
3165      return true;
3166    }
3167
3168    if (actualRegion == null) {
3169      LOG.debug("Received CLOSE for a region which is not online, and we're not opening.");
3170      this.regionsInTransitionInRS.remove(Bytes.toBytes(encodedName));
3171      // The master deletes the znode when it receives this exception.
3172      throw new NotServingRegionException("The region " + encodedName +
3173          " is not online, and is not opening.");
3174    }
3175
3176    CloseRegionHandler crh;
3177    final RegionInfo hri = actualRegion.getRegionInfo();
3178    if (hri.isMetaRegion()) {
3179      crh = new CloseMetaHandler(this, this, hri, abort);
3180    } else {
3181      crh = new CloseRegionHandler(this, this, hri, abort, sn);
3182    }
3183    this.executorService.submit(crh);
3184    return true;
3185  }

3701    RegionServerStatusService.BlockingInterface rss = rssStub;
3702    if (rss == null) {
3703      // the current server could be stopping.
3704      LOG.trace("Skipping file archival reporting to HMaster as stub is null");
3705      return false;
3706    }
3707    try {
3708      RegionServerStatusProtos.FileArchiveNotificationRequest request =
3709          rsSpaceQuotaManager.buildFileArchiveRequest(tableName, archivedFiles);
3710      rss.reportFileArchival(null, request);
3711    } catch (ServiceException se) {
3712      IOException ioe = ProtobufUtil.getRemoteException(se);
3713      if (ioe instanceof PleaseHoldException) {
3714        if (LOG.isTraceEnabled()) {
3715          LOG.trace("Failed to report file archival(s) to Master because it is initializing."
3716              + " This will be retried.", ioe);
3717        }
3718        // The Master is coming up. Will retry the report later. Avoid re-creating the stub.
3719        return false;
3720      }
3721      if (rssStub == rss) {
3722        rssStub = null;
3723      }
3724      // re-create the stub if we failed to report the archival
3725      createRegionServerStatusStub(true);
3726      LOG.debug("Failed to report file archival(s) to Master. This will be retried.", ioe);
3727      return false;
3728    }
3729    return true;
