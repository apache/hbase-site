369  /**
370   * Instantiated as a scanner lease. If the lease times out, the scanner is
371   * closed
372   */
373  private class ScannerListener implements LeaseListener {
374    private final String scannerName;
375
376    ScannerListener(final String n) {
377      this.scannerName = n;
378    }
379
380    @Override
381    public void leaseExpired() {
382      RegionScannerHolder rsh = scanners.remove(this.scannerName);
383      if (rsh != null) {
384        RegionScanner s = rsh.s;
385        LOG.info("Scanner " + this.scannerName + " lease expired on region "
386          + s.getRegionInfo().getRegionNameAsString());
387        Region region = null;
388        try {
389          region = regionServer.getRegion(s.getRegionInfo().getRegionName());
390          if (region != null && region.getCoprocessorHost() != null) {
391            region.getCoprocessorHost().preScannerClose(s);
392          }
393        } catch (IOException e) {
394          LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
395        } finally {
396          try {
397            s.close();
398            if (region != null && region.getCoprocessorHost() != null) {
399              region.getCoprocessorHost().postScannerClose(s);
400            }
401          } catch (IOException e) {
402            LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
403          }
404        }
405      } else {
406        LOG.warn("Scanner " + this.scannerName + " lease expired, but no related" +
407          " scanner found, hence no chance to close that related scanner!");
408      }
409    }
410  }

432  private boolean startNonceOperation(final MutationProto mutation, long nonceGroup)

434    if (regionServer.nonceManager == null || !mutation.hasNonce()) return true;
435    boolean canProceed = false;

441    }

494  /**
495   * Mutate a list of rows atomically.
496   *
497   * @param region
498   * @param actions
499   * @param cellScanner if non-null, the mutation data -- the Cell content.
500   * @throws IOException
501   */
502  private void mutateRows(final Region region,
503      final List<ClientProtos.Action> actions,
504      final CellScanner cellScanner, RegionActionResult.Builder builder) throws IOException {
505    if (!region.getRegionInfo().isMetaTable()) {
506      regionServer.cacheFlusher.reclaimMemStoreMemory();
507    }
508    RowMutations rm = null;
509    int i = 0;
510    ClientProtos.ResultOrException.Builder resultOrExceptionOrBuilder =
511        ClientProtos.ResultOrException.newBuilder();
512    for (ClientProtos.Action action: actions) {
513      if (action.hasGet()) {
514        throw new DoNotRetryIOException("Atomic put and/or delete only, not a Get=" +
515          action.getGet());
516      }
517      MutationType type = action.getMutation().getMutateType();
518      if (rm == null) {
519        rm = new RowMutations(action.getMutation().getRow().toByteArray(), actions.size());
520      }
521      switch (type) {
522        case PUT:
523          rm.add(ProtobufUtil.toPut(action.getMutation(), cellScanner));
524          break;
525        case DELETE:
526          rm.add(ProtobufUtil.toDelete(action.getMutation(), cellScanner));
527          break;
528        default:
529          throw new DoNotRetryIOException("Atomic put and/or delete only, not " + type.name());
530      }
531      // To unify the response format with doNonAtomicRegionMutation and read through client's
532      // AsyncProcess we have to add an empty result instance per operation
533      resultOrExceptionOrBuilder.clear();
534      resultOrExceptionOrBuilder.setIndex(i++);
535      builder.addResultOrException(
536          resultOrExceptionOrBuilder.build());
537    }
538    region.mutateRow(rm);
539  }

594  /**
595   * Execute an append mutation.
596   *
597   * @param region
598   * @param m
599   * @param cellScanner
600   * @return result to return to client if default operation should be
601   * bypassed as indicated by RegionObserver, null otherwise
602   * @throws IOException
603   */
604  private Result append(final Region region, final OperationQuota quota,
605      final MutationProto mutation, final CellScanner cellScanner, long nonceGroup)
606      throws IOException {
607    long before = EnvironmentEdgeManager.currentTime();
608    Append append = ProtobufUtil.toAppend(mutation, cellScanner);
609    quota.addMutation(append);
610    Result r = null;
611    if (region.getCoprocessorHost() != null) {
612      r = region.getCoprocessorHost().preAppend(append);
613    }
614    if (r == null) {
615      boolean canProceed = startNonceOperation(mutation, nonceGroup);
616      boolean success = false;
617      try {
618        long nonce = mutation.hasNonce() ? mutation.getNonce() : HConstants.NO_NONCE;
619        if (canProceed) {
620          r = region.append(append, nonceGroup, nonce);
621        } else {
622          // convert duplicate append to get
623          List<Cell> results = region.get(ProtobufUtil.toGet(mutation, cellScanner), false,
624            nonceGroup, nonce);
625          r = Result.create(results);
626        }
627        success = true;
628      } finally {
629        if (canProceed) {
630          endNonceOperation(mutation, nonceGroup, success);
631        }
632      }
633      if (region.getCoprocessorHost() != null) {
634        region.getCoprocessorHost().postAppend(append, r);
635      }
636    }
637    if (regionServer.metricsRegionServer != null) {
638      regionServer.metricsRegionServer.updateAppend(
639        EnvironmentEdgeManager.currentTime() - before);
640    }
641    return r;
642  }

692  /**
693   * Run through the regionMutation <code>rm</code> and per Mutation, do the work, and then when
694   * done, add an instance of a {@link ResultOrException} that corresponds to each Mutation.
695   * @param region
696   * @param actions
697   * @param cellScanner
698   * @param builder
699   * @param cellsToReturn  Could be null. May be allocated in this method.  This is what this
700   * method returns as a 'result'.
701   * @param closeCallBack the callback to be used with multigets
702   * @param context the current RpcCallContext
703   * @return Return the <code>cellScanner</code> passed
704   */
705  private List<CellScannable> doNonAtomicRegionMutation(final Region region,
706      final OperationQuota quota, final RegionAction actions, final CellScanner cellScanner,
707      final RegionActionResult.Builder builder, List<CellScannable> cellsToReturn, long nonceGroup,
708      final RegionScannersCloseCallBack closeCallBack, RpcCallContext context) {
709    // Gather up CONTIGUOUS Puts and Deletes in this mutations List.  Idea is that rather than do
710    // one at a time, we instead pass them in batch.  Be aware that the corresponding
711    // ResultOrException instance that matches each Put or Delete is then added down in the
712    // doBatchOp call.  We should be staying aligned though the Put and Delete are deferred/batched
713    List<ClientProtos.Action> mutations = null;
714    long maxQuotaResultSize = Math.min(maxScannerResultSize, quota.getReadAvailable());
715    IOException sizeIOE = null;
716    Object lastBlock = null;
717    ClientProtos.ResultOrException.Builder resultOrExceptionBuilder = ResultOrException.newBuilder();
718    boolean hasResultOrException = false;
719    for (ClientProtos.Action action : actions.getActionList()) {
720      hasResultOrException = false;
721      resultOrExceptionBuilder.clear();
722      try {
723        Result r = null;
724
725        if (context != null
726            && context.isRetryImmediatelySupported()
727            && (context.getResponseCellSize() > maxQuotaResultSize
728              || context.getResponseBlockSize() + context.getResponseExceptionSize()
729              > maxQuotaResultSize)) {
730
731          // We're storing the exception since the exception and reason string won't
732          // change after the response size limit is reached.
733          if (sizeIOE == null ) {
734            // We don't need the stack un-winding do don't throw the exception.
735            // Throwing will kill the JVM's JIT.
736            //
737            // Instead just create the exception and then store it.
738            sizeIOE = new MultiActionResultTooLarge("Max size exceeded"
739                + " CellSize: " + context.getResponseCellSize()
740                + " BlockSize: " + context.getResponseBlockSize());
741
742            // Only report the exception once since there's only one request that
743            // caused the exception. Otherwise this number will dominate the exceptions count.
744            rpcServer.getMetrics().exception(sizeIOE);
745          }
746
747          // Now that there's an exception is known to be created
748          // use it for the response.
749          //
750          // This will create a copy in the builder.
751          hasResultOrException = true;
752          NameBytesPair pair = ResponseConverter.buildException(sizeIOE);
753          resultOrExceptionBuilder.setException(pair);
754          context.incrementResponseExceptionSize(pair.getSerializedSize());
755          resultOrExceptionBuilder.setIndex(action.getIndex());
756          builder.addResultOrException(resultOrExceptionBuilder.build());
757          if (cellScanner != null) {
758            skipCellsForMutation(action, cellScanner);
759          }
760          continue;
761        }
762        if (action.hasGet()) {
763          long before = EnvironmentEdgeManager.currentTime();
764          try {
765            Get get = ProtobufUtil.toGet(action.getGet());
766            if (context != null) {
767              r = get(get, ((HRegion) region), closeCallBack, context);
768            } else {
769              r = region.get(get);
770            }
771          } finally {
772            if (regionServer.metricsRegionServer != null) {
773              regionServer.metricsRegionServer.updateGet(
774                EnvironmentEdgeManager.currentTime() - before);
775            }
776          }
777        } else if (action.hasServiceCall()) {
778          hasResultOrException = true;
779          try {
780            com.google.protobuf.Message result =
781                execServiceOnRegion(region, action.getServiceCall());
782            ClientProtos.CoprocessorServiceResult.Builder serviceResultBuilder =
783                ClientProtos.CoprocessorServiceResult.newBuilder();
784            resultOrExceptionBuilder.setServiceResult(
785                serviceResultBuilder.setValue(
786                  serviceResultBuilder.getValueBuilder()
787                    .setName(result.getClass().getName())
788                    // TODO: Copy!!!
789                    .setValue(UnsafeByteOperations.unsafeWrap(result.toByteArray()))));
790          } catch (IOException ioe) {
791            rpcServer.getMetrics().exception(ioe);
792            NameBytesPair pair = ResponseConverter.buildException(ioe);
793            resultOrExceptionBuilder.setException(pair);
794            context.incrementResponseExceptionSize(pair.getSerializedSize());
795          }
796        } else if (action.hasMutation()) {
797          MutationType type = action.getMutation().getMutateType();
798          if (type != MutationType.PUT && type != MutationType.DELETE && mutations != null &&
799              !mutations.isEmpty()) {
800            // Flush out any Puts or Deletes already collected.
801            doBatchOp(builder, region, quota, mutations, cellScanner);
802            mutations.clear();
803          }
804          switch (type) {
805            case APPEND:
806              r = append(region, quota, action.getMutation(), cellScanner, nonceGroup);
807              break;
808            case INCREMENT:
809              r = increment(region, quota, action.getMutation(), cellScanner, nonceGroup);
810              break;
811            case PUT:
812            case DELETE:
813              // Collect the individual mutations and apply in a batch
814              if (mutations == null) {
815                mutations = new ArrayList<>(actions.getActionCount());
816              }
817              mutations.add(action);
818              break;
819            default:
820              throw new DoNotRetryIOException("Unsupported mutate type: " + type.name());
821          }
822        } else {
823          throw new HBaseIOException("Unexpected Action type");
824        }
825        if (r != null) {
826          ClientProtos.Result pbResult = null;
827          if (isClientCellBlockSupport(context)) {
828            pbResult = ProtobufUtil.toResultNoData(r);
829            //  Hard to guess the size here.  Just make a rough guess.
830            if (cellsToReturn == null) {
831              cellsToReturn = new ArrayList<>();
832            }
833            cellsToReturn.add(r);
834          } else {
835            pbResult = ProtobufUtil.toResult(r);
836          }
837          lastBlock = addSize(context, r, lastBlock);
838          hasResultOrException = true;
839          resultOrExceptionBuilder.setResult(pbResult);
840        }
841        // Could get to here and there was no result and no exception.  Presumes we added
842        // a Put or Delete to the collecting Mutations List for adding later.  In this
843        // case the corresponding ResultOrException instance for the Put or Delete will be added
844        // down in the doBatchOp method call rather than up here.
845      } catch (IOException ie) {
846        rpcServer.getMetrics().exception(ie);
847        hasResultOrException = true;
848        NameBytesPair pair = ResponseConverter.buildException(ie);
849        resultOrExceptionBuilder.setException(pair);
850        context.incrementResponseExceptionSize(pair.getSerializedSize());
851      }
852      if (hasResultOrException) {
853        // Propagate index.
854        resultOrExceptionBuilder.setIndex(action.getIndex());
855        builder.addResultOrException(resultOrExceptionBuilder.build());
856      }
857    }
858    // Finish up any outstanding mutations
859    if (mutations != null && !mutations.isEmpty()) {
860      doBatchOp(builder, region, quota, mutations, cellScanner);
861    }
862    return cellsToReturn;
863  }

865  /**
866   * Execute a list of Put/Delete mutations.
867   *
868   * @param builder
869   * @param region
870   * @param mutations
871   */
872  private void doBatchOp(final RegionActionResult.Builder builder, final Region region,
873      final OperationQuota quota, final List<ClientProtos.Action> mutations,
874      final CellScanner cells) {
875    Mutation[] mArray = new Mutation[mutations.size()];
876    long before = EnvironmentEdgeManager.currentTime();
877    boolean batchContainsPuts = false, batchContainsDelete = false;
878    try {
879      /** HBASE-17924
880       * mutationActionMap is a map to map the relation between mutations and actions
881       * since mutation array may have been reoredered.In order to return the right
882       * result or exception to the corresponding actions, We need to know which action
883       * is the mutation belong to. We can't sort ClientProtos.Action array, since they
884       * are bonded to cellscanners.
885       */
886      Map<Mutation, ClientProtos.Action> mutationActionMap = new HashMap<Mutation, ClientProtos.Action>();
887      int i = 0;
888      for (ClientProtos.Action action: mutations) {
889        MutationProto m = action.getMutation();
890        Mutation mutation;
891        if (m.getMutateType() == MutationType.PUT) {
892          mutation = ProtobufUtil.toPut(m, cells);
893          batchContainsPuts = true;
894        } else {
895          mutation = ProtobufUtil.toDelete(m, cells);
896          batchContainsDelete = true;
897        }
898        mutationActionMap.put(mutation, action);
899        mArray[i++] = mutation;
900        quota.addMutation(mutation);
901      }
902
903      if (!region.getRegionInfo().isMetaTable()) {
904        regionServer.cacheFlusher.reclaimMemStoreMemory();
905      }
906      // HBASE-17924
907      // sort to improve lock efficiency
908      Arrays.sort(mArray);
909      OperationStatus[] codes = region.batchMutate(mArray, HConstants.NO_NONCE,
910        HConstants.NO_NONCE);
911      for (i = 0; i < codes.length; i++) {
912        Mutation currentMutation = mArray[i];
913        ClientProtos.Action currentAction = mutationActionMap.get(currentMutation);
914        int index = currentAction.getIndex();
915        Exception e = null;
916        switch (codes[i].getOperationStatusCode()) {
917          case BAD_FAMILY:
918            e = new NoSuchColumnFamilyException(codes[i].getExceptionMsg());
919            builder.addResultOrException(getResultOrException(e, index));
920            break;
921
922          case SANITY_CHECK_FAILURE:
923            e = new FailedSanityCheckException(codes[i].getExceptionMsg());
924            builder.addResultOrException(getResultOrException(e, index));
925            break;
926
927          default:
928            e = new DoNotRetryIOException(codes[i].getExceptionMsg());
929            builder.addResultOrException(getResultOrException(e, index));
930            break;
931
932          case SUCCESS:
933            builder.addResultOrException(getResultOrException(
934              ClientProtos.Result.getDefaultInstance(), index));
935            break;
936        }
937      }
938    } catch (IOException ie) {
939      for (int i = 0; i < mutations.size(); i++) {
940        builder.addResultOrException(getResultOrException(ie, mutations.get(i).getIndex()));
941      }
942    }
943    if (regionServer.metricsRegionServer != null) {
944      long after = EnvironmentEdgeManager.currentTime();
945      if (batchContainsPuts) {
946        regionServer.metricsRegionServer.updatePut(after - before);
947      }
948      if (batchContainsDelete) {
949        regionServer.metricsRegionServer.updateDelete(after - before);
950      }
951    }
952  }

954  /**
955   * Execute a list of Put/Delete mutations. The function returns OperationStatus instead of
956   * constructing MultiResponse to save a possible loop if caller doesn't need MultiResponse.
957   * @param region
958   * @param mutations
959   * @param replaySeqId
960   * @return an array of OperationStatus which internally contains the OperationStatusCode and the
961   *         exceptionMessage if any
962   * @throws IOException
963   */
964  private OperationStatus [] doReplayBatchOp(final Region region,
965      final List<WALSplitter.MutationReplay> mutations, long replaySeqId) throws IOException {
966    long before = EnvironmentEdgeManager.currentTime();
967    boolean batchContainsPuts = false, batchContainsDelete = false;
968    try {
969      for (Iterator<WALSplitter.MutationReplay> it = mutations.iterator(); it.hasNext();) {
970        WALSplitter.MutationReplay m = it.next();
971
972        if (m.type == MutationType.PUT) {
973          batchContainsPuts = true;
974        } else {
975          batchContainsDelete = true;
976        }
977
978        NavigableMap<byte[], List<Cell>> map = m.mutation.getFamilyCellMap();
979        List<Cell> metaCells = map.get(WALEdit.METAFAMILY);
980        if (metaCells != null && !metaCells.isEmpty()) {
981          for (Cell metaCell : metaCells) {
982            CompactionDescriptor compactionDesc = WALEdit.getCompaction(metaCell);
983            boolean isDefaultReplica = RegionReplicaUtil.isDefaultReplica(region.getRegionInfo());
984            HRegion hRegion = (HRegion)region;
985            if (compactionDesc != null) {
986              // replay the compaction. Remove the files from stores only if we are the primary
987              // region replica (thus own the files)
988              hRegion.replayWALCompactionMarker(compactionDesc, !isDefaultReplica, isDefaultReplica,
989                replaySeqId);
990              continue;
991            }
992            FlushDescriptor flushDesc = WALEdit.getFlushDescriptor(metaCell);
993            if (flushDesc != null && !isDefaultReplica) {
994              hRegion.replayWALFlushMarker(flushDesc, replaySeqId);
995              continue;
996            }
997            RegionEventDescriptor regionEvent = WALEdit.getRegionEventDescriptor(metaCell);
998            if (regionEvent != null && !isDefaultReplica) {
999              hRegion.replayWALRegionEventMarker(regionEvent);
1000              continue;
1001            }
1002            BulkLoadDescriptor bulkLoadEvent = WALEdit.getBulkLoadDescriptor(metaCell);
1003            if (bulkLoadEvent != null) {
1004              hRegion.replayWALBulkLoadEventMarker(bulkLoadEvent);
1005              continue;
1006            }
1007          }
1008          it.remove();
1009        }
1010      }
1011      requestCount.add(mutations.size());
1012      if (!region.getRegionInfo().isMetaTable()) {
1013        regionServer.cacheFlusher.reclaimMemStoreMemory();
1014      }
1015      return region.batchReplay(mutations.toArray(
1016        new WALSplitter.MutationReplay[mutations.size()]), replaySeqId);
1017    } finally {
1018      if (regionServer.metricsRegionServer != null) {
1019        long after = EnvironmentEdgeManager.currentTime();
1020          if (batchContainsPuts) {
1021          regionServer.metricsRegionServer.updatePut(after - before);
1022        }
1023        if (batchContainsDelete) {
1024          regionServer.metricsRegionServer.updateDelete(after - before);
1025        }
1026      }
1027    }
1028  }

1182  /**
1183   * Method to account for the size of retained cells and retained data blocks.
1184   * @return an object that represents the last referenced block from this response.
1185   */
1186  Object addSize(RpcCallContext context, Result r, Object lastBlock) {
1187    if (context != null && r != null && !r.isEmpty()) {
1188      for (Cell c : r.rawCells()) {
1189        context.incrementResponseCellSize(CellUtil.estimatedSerializedSizeOf(c));
1190
1191        // Since byte buffers can point all kinds of crazy places it's harder to keep track
1192        // of which blocks are kept alive by what byte buffer.
1193        // So we make a guess.
1194        if (c instanceof ByteBufferCell) {
1195          ByteBufferCell bbCell = (ByteBufferCell) c;
1196          ByteBuffer bb = bbCell.getValueByteBuffer();
1197          if (bb != lastBlock) {
1198            context.incrementResponseBlockSize(bb.capacity());
1199            lastBlock = bb;
1200          }
1201        } else {
1202          // We're using the last block being the same as the current block as
1203          // a proxy for pointing to a new block. This won't be exact.
1204          // If there are multiple gets that bounce back and forth
1205          // Then it's possible that this will over count the size of
1206          // referenced blocks. However it's better to over count and
1207          // use two rpcs than to OOME the regionserver.
1208          byte[] valueArray = c.getValueArray();
1209          if (valueArray != lastBlock) {
1210            context.incrementResponseBlockSize(valueArray.length);
1211            lastBlock = valueArray;
1212          }
1213        }
1214
1215      }
1216    }
1217    return lastBlock;
1218  }

1664   * Open asynchronously a region or a set of regions on the region server.
1665   *
1666   * The opening is coordinated by ZooKeeper, and this method requires the znode to be created
1667   *  before being called. As a consequence, this method should be called only from the master.
1668   * <p>
1669   * Different manages states for the region are:
1670   * </p><ul>
1671   *  <li>region not opened: the region opening will start asynchronously.</li>
1672   *  <li>a close is already in progress: this is considered as an error.</li>
1673   *  <li>an open is already in progress: this new open request will be ignored. This is important
1674   *  because the Master can do multiple requests if it crashes.</li>
1675   *  <li>the region is already opened:  this new open request will be ignored.</li>
1676   *  </ul>
1677   * <p>
1678   * Bulk assign: If there are more than 1 region to open, it will be considered as a bulk assign.
1679   * For a single region opening, errors are sent through a ServiceException. For bulk assign,
1680   * errors are put in the response as FAILED_OPENING.
1681   * </p>
1682   * @param controller the RPC controller
1683   * @param request the request
1684   * @throws ServiceException
1685   */
1686  @Override
1687  @QosPriority(priority=HConstants.ADMIN_QOS)
1688  public OpenRegionResponse openRegion(final RpcController controller,
1689      final OpenRegionRequest request) throws ServiceException {
1690    requestCount.increment();
1691    if (request.hasServerStartCode()) {
1692      // check that we are the same server that this RPC is intended for.
1693      long serverStartCode = request.getServerStartCode();
1694      if (regionServer.serverName.getStartcode() !=  serverStartCode) {
1695        throw new ServiceException(new DoNotRetryIOException("This RPC was intended for a " +
1696            "different server with startCode: " + serverStartCode + ", this server is: "
1697            + regionServer.serverName));
1698      }
1699    }
1700
1701    OpenRegionResponse.Builder builder = OpenRegionResponse.newBuilder();
1702    final int regionCount = request.getOpenInfoCount();
1703    final Map<TableName, HTableDescriptor> htds = new HashMap<>(regionCount);
1704    final boolean isBulkAssign = regionCount > 1;
1705    try {
1706      checkOpen();
1707    } catch (IOException ie) {
1708      TableName tableName = null;
1709      if (regionCount == 1) {
1710        RegionInfo ri = request.getOpenInfo(0).getRegion();
1711        if (ri != null) {
1712          tableName = ProtobufUtil.toTableName(ri.getTableName());
1713        }
1714      }
1715      if (!TableName.META_TABLE_NAME.equals(tableName)) {
1716        throw new ServiceException(ie);
1717      }
1718      // We are assigning meta, wait a little for regionserver to finish initialization.
1719      int timeout = regionServer.conf.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY,
1720        HConstants.DEFAULT_HBASE_RPC_TIMEOUT) >> 2; // Quarter of RPC timeout
1721      long endTime = System.currentTimeMillis() + timeout;
1722      synchronized (regionServer.online) {
1723        try {
1724          while (System.currentTimeMillis() <= endTime
1725              && !regionServer.isStopped() && !regionServer.isOnline()) {
1726            regionServer.online.wait(regionServer.msgInterval);
1727          }
1728          checkOpen();
1729        } catch (InterruptedException t) {
1730          Thread.currentThread().interrupt();
1731          throw new ServiceException(t);
1732        } catch (IOException e) {
1733          throw new ServiceException(e);
1734        }
1735      }
1736    }
1737
1738    long masterSystemTime = request.hasMasterSystemTime() ? request.getMasterSystemTime() : -1;
1739
1740    for (RegionOpenInfo regionOpenInfo : request.getOpenInfoList()) {
1741      final HRegionInfo region = HRegionInfo.convert(regionOpenInfo.getRegion());
1742      HTableDescriptor htd;
1743      try {
1744        String encodedName = region.getEncodedName();
1745        byte[] encodedNameBytes = region.getEncodedNameAsBytes();
1746        final Region onlineRegion = regionServer.getFromOnlineRegions(encodedName);
1747        if (onlineRegion != null) {
1748          // The region is already online. This should not happen any more.
1749          String error = "Received OPEN for the region:"
1750            + region.getRegionNameAsString() + ", which is already online";
1751          regionServer.abort(error);
1752          throw new IOException(error);
1753        }
1754        LOG.info("Open " + region.getRegionNameAsString());
1755
1756        final Boolean previous = regionServer.regionsInTransitionInRS.putIfAbsent(
1757          encodedNameBytes, Boolean.TRUE);
1758
1759        if (Boolean.FALSE.equals(previous)) {
1760          if (regionServer.getFromOnlineRegions(encodedName) != null) {
1761            // There is a close in progress. This should not happen any more.
1762            String error = "Received OPEN for the region:"
1763              + region.getRegionNameAsString() + ", which we are already trying to CLOSE";
1764            regionServer.abort(error);
1765            throw new IOException(error);
1766          }
1767          regionServer.regionsInTransitionInRS.put(encodedNameBytes, Boolean.TRUE);
1768        }
1769
1770        if (Boolean.TRUE.equals(previous)) {
1771          // An open is in progress. This is supported, but let's log this.
1772          LOG.info("Receiving OPEN for the region:" +
1773            region.getRegionNameAsString() + ", which we are already trying to OPEN"
1774              + " - ignoring this new request for this region.");
1775        }
1776
1777        // We are opening this region. If it moves back and forth for whatever reason, we don't
1778        // want to keep returning the stale moved record while we are opening/if we close again.
1779        regionServer.removeFromMovedRegions(region.getEncodedName());
1780
1781        if (previous == null || !previous.booleanValue()) {
1782          // check if the region to be opened is marked in recovering state in ZK
1783          if (ZKSplitLog.isRegionMarkedRecoveringInZK(regionServer.getZooKeeper(),
1784              region.getEncodedName())) {
1785            // Check if current region open is for distributedLogReplay. This check is to support
1786            // rolling restart/upgrade where we want to Master/RS see same configuration
1787            if (!regionOpenInfo.hasOpenForDistributedLogReplay()
1788                  || regionOpenInfo.getOpenForDistributedLogReplay()) {
1789              regionServer.recoveringRegions.put(region.getEncodedName(), null);
1790            } else {
1791              // Remove stale recovery region from ZK when we open region not for recovering which
1792              // could happen when turn distributedLogReplay off from on.
1793              List<String> tmpRegions = new ArrayList<>();
1794              tmpRegions.add(region.getEncodedName());
1795              ZKSplitLog.deleteRecoveringRegionZNodes(regionServer.getZooKeeper(),
1796                tmpRegions);
1797            }
1798          }
1799          htd = htds.get(region.getTable());
1800          if (htd == null) {
1801            htd = regionServer.tableDescriptors.get(region.getTable());
1802            htds.put(region.getTable(), htd);
1803          }
1804          if (htd == null) {
1805            throw new IOException("Missing table descriptor for " + region.getEncodedName());
1806          }
1807          // If there is no action in progress, we can submit a specific handler.
1808          // Need to pass the expected version in the constructor.
1809          if (region.isMetaRegion()) {
1810            regionServer.service.submit(new OpenMetaHandler(
1811              regionServer, regionServer, region, htd, masterSystemTime));
1812          } else {
1813            if (regionOpenInfo.getFavoredNodesCount() > 0) {
1814              regionServer.updateRegionFavoredNodesMapping(region.getEncodedName(),
1815                  regionOpenInfo.getFavoredNodesList());
1816            }
1817            if (htd.getPriority() >= HConstants.ADMIN_QOS || region.getTable().isSystemTable()) {
1818              regionServer.service.submit(new OpenPriorityRegionHandler(
1819                regionServer, regionServer, region, htd, masterSystemTime));
1820            } else {
1821              regionServer.service.submit(new OpenRegionHandler(
1822                regionServer, regionServer, region, htd, masterSystemTime));
1823            }
1824          }
1825        }
1826
1827        builder.addOpeningState(RegionOpeningState.OPENED);
1828
1829      } catch (KeeperException zooKeeperEx) {
1830        LOG.error("Can't retrieve recovering state from zookeeper", zooKeeperEx);
1831        throw new ServiceException(zooKeeperEx);
1832      } catch (IOException ie) {
1833        LOG.warn("Failed opening region " + region.getRegionNameAsString(), ie);
1834        if (isBulkAssign) {
1835          builder.addOpeningState(RegionOpeningState.FAILED_OPENING);
1836        } else {
1837          throw new ServiceException(ie);
1838        }
1839      }
1840    }
1841    return builder.build();
1842  }

1906  @QosPriority(priority = HConstants.REPLAY_QOS)
1907  public ReplicateWALEntryResponse replay(final RpcController controller,
1908      final ReplicateWALEntryRequest request) throws ServiceException {
1909    long before = EnvironmentEdgeManager.currentTime();
1910    CellScanner cells = ((HBaseRpcController) controller).cellScanner();
1911    try {
1912      checkOpen();
1913      List<WALEntry> entries = request.getEntryList();
1914      if (entries == null || entries.isEmpty()) {
1915        // empty input
1916        return ReplicateWALEntryResponse.newBuilder().build();
1917      }
1918      ByteString regionName = entries.get(0).getKey().getEncodedRegionName();
1919      Region region = regionServer.getRegionByEncodedName(regionName.toStringUtf8());
1920      RegionCoprocessorHost coprocessorHost =
1921          ServerRegionReplicaUtil.isDefaultReplica(region.getRegionInfo())
1922            ? region.getCoprocessorHost()
1923            : null; // do not invoke coprocessors if this is a secondary region replica
1924      List<Pair<WALKey, WALEdit>> walEntries = new ArrayList<>();
1925
1926      // Skip adding the edits to WAL if this is a secondary region replica
1927      boolean isPrimary = RegionReplicaUtil.isDefaultReplica(region.getRegionInfo());
1928      Durability durability = isPrimary ? Durability.USE_DEFAULT : Durability.SKIP_WAL;
1929
1930      for (WALEntry entry : entries) {
1931        if (!regionName.equals(entry.getKey().getEncodedRegionName())) {
1932          throw new NotServingRegionException("Replay request contains entries from multiple " +
1933              "regions. First region:" + regionName.toStringUtf8() + " , other region:"
1934              + entry.getKey().getEncodedRegionName());
1935        }
1936        if (regionServer.nonceManager != null && isPrimary) {
1937          long nonceGroup = entry.getKey().hasNonceGroup()
1938            ? entry.getKey().getNonceGroup() : HConstants.NO_NONCE;
1939          long nonce = entry.getKey().hasNonce() ? entry.getKey().getNonce() : HConstants.NO_NONCE;
1940          regionServer.nonceManager.reportOperationFromWal(
1941              nonceGroup,
1942              nonce,
1943              entry.getKey().getWriteTime());
1944        }
1945        Pair<WALKey, WALEdit> walEntry = (coprocessorHost == null) ? null : new Pair<>();
1946        List<WALSplitter.MutationReplay> edits = WALSplitter.getMutationsFromWALEntry(entry,
1947          cells, walEntry, durability);
1948        if (coprocessorHost != null) {
1949          // Start coprocessor replay here. The coprocessor is for each WALEdit instead of a
1950          // KeyValue.
1951          if (coprocessorHost.preWALRestore(region.getRegionInfo(), walEntry.getFirst(),
1952            walEntry.getSecond())) {
1953            // if bypass this log entry, ignore it ...
1954            continue;
1955          }
1956          walEntries.add(walEntry);
1957        }
1958        if(edits!=null && !edits.isEmpty()) {
1959          // HBASE-17924
1960          // sort to improve lock efficiency
1961          Collections.sort(edits);
1962          long replaySeqId = (entry.getKey().hasOrigSequenceNumber()) ?
1963            entry.getKey().getOrigSequenceNumber() : entry.getKey().getLogSequenceNumber();
1964          OperationStatus[] result = doReplayBatchOp(region, edits, replaySeqId);
1965          // check if it's a partial success
1966          for (int i = 0; result != null && i < result.length; i++) {
1967            if (result[i] != OperationStatus.SUCCESS) {
1968              throw new IOException(result[i].getExceptionMsg());
1969            }
1970          }
1971        }
1972      }
1973
1974      //sync wal at the end because ASYNC_WAL is used above
1975      WAL wal = getWAL(region);
1976      if (wal != null) {
1977        wal.sync();
1978      }
1979
1980      if (coprocessorHost != null) {
1981        for (Pair<WALKey, WALEdit> entry : walEntries) {
1982          coprocessorHost.postWALRestore(region.getRegionInfo(), entry.getFirst(),
1983            entry.getSecond());
1984        }
1985      }
1986      return ReplicateWALEntryResponse.newBuilder().build();
1987    } catch (IOException ie) {
1988      throw new ServiceException(ie);
1989    } finally {
1990      if (regionServer.metricsRegionServer != null) {
1991        regionServer.metricsRegionServer.updateReplay(
1992          EnvironmentEdgeManager.currentTime() - before);
1993      }
1994    }

2344      scanner.next(results);
2345    } finally {
2346      if (scanner != null) {
2347        if (closeCallBack == null) {
2348          // If there is a context then the scanner can be added to the current
2349          // RpcCallContext. The rpc callback will take care of closing the
2350          // scanner, for eg in case
2351          // of get()
2352          assert scanner instanceof org.apache.hadoop.hbase.ipc.RpcCallback;
2353          context.setCallBack((RegionScannerImpl) scanner);
2354        } else {
2355          // The call is from multi() where the results from the get() are
2356          // aggregated and then send out to the
2357          // rpc. The rpccall back will close all such scanners created as part
2358          // of multi().
2359          closeCallBack.addScanner(scanner);
2360        }
2361      }
2362    }
2363
2364    // post-get CP hook
2365    if (region.getCoprocessorHost() != null) {
2366      region.getCoprocessorHost().postGet(get, results);
2367    }
2368    region.metricsUpdateForGet(results, before);
2369    return Result.create(results, get.isCheckExistenceOnly() ? !results.isEmpty() : null, stale);
2370  }

2486      builder.addStat(stat.getValue());
2487    }

2489    return responseBuilder.build();
2490  }

2525      final MutateRequest request) throws ServiceException {
2526    // rpc controller is how we bring in data via the back door;  it is unprotobuf'ed data.
2527    // It is also the conduit via which we pass back data.
2528    HBaseRpcController controller = (HBaseRpcController)rpcc;
2529    CellScanner cellScanner = controller != null ? controller.cellScanner() : null;
2530    OperationQuota quota = null;
2531    RpcCallContext context = RpcServer.getCurrentCall();
2532    // Clear scanner so we are not holding on to reference across call.
2533    if (controller != null) {
2534      controller.setCellScanner(null);
2535    }

2537      checkOpen();

2539      rpcMutateRequestCount.increment();

2544        regionServer.cacheFlusher.reclaimMemStoreMemory();
2545      }
2546      long nonceGroup = request.hasNonceGroup() ? request.getNonceGroup() : HConstants.NO_NONCE;
2547      Result r = null;
2548      Boolean processed = null;
2549      MutationType type = mutation.getMutateType();
2550
2551      quota = getQuotaManager().checkQuota(region, OperationQuota.OperationType.MUTATE);
2552

2554      case APPEND:
2555        // TODO: this doesn't actually check anything.
2556        r = append(region, quota, mutation, cellScanner, nonceGroup);
2557        break;
2558      case INCREMENT:
2559        // TODO: this doesn't actually check anything.
2560        r = increment(region, quota, mutation, cellScanner, nonceGroup);
2561        break;
2562      case PUT:
2563        Put put = ProtobufUtil.toPut(mutation, cellScanner);
2564        quota.addMutation(put);

2566          Condition condition = request.getCondition();

2568          byte[] family = condition.getFamily().toByteArray();
2569          byte[] qualifier = condition.getQualifier().toByteArray();

2585          }
2586        } else {
2587          region.put(put);
2588          processed = Boolean.TRUE;
2589        }
2590        break;

2592        Delete delete = ProtobufUtil.toDelete(mutation, cellScanner);
2593        quota.addMutation(delete);
2594        if (request.hasCondition()) {
2595          Condition condition = request.getCondition();
2596          byte[] row = condition.getRow().toByteArray();
2597          byte[] family = condition.getFamily().toByteArray();
2598          byte[] qualifier = condition.getQualifier().toByteArray();
2599          CompareOp compareOp = CompareOp.valueOf(condition.getCompareType().name());
2600          ByteArrayComparable comparator =
2601            ProtobufUtil.toComparator(condition.getComparator());
2602          if (region.getCoprocessorHost() != null) {
2603            processed = region.getCoprocessorHost().preCheckAndDelete(
2604              row, family, qualifier, compareOp, comparator, delete);
2605          }
2606          if (processed == null) {
2607            boolean result = region.checkAndMutate(row, family,
2608              qualifier, compareOp, comparator, delete, true);
2609            if (region.getCoprocessorHost() != null) {
2610              result = region.getCoprocessorHost().postCheckAndDelete(row, family,
2611                qualifier, compareOp, comparator, delete, result);
2612            }
2613            processed = result;
2614          }
2615        } else {
2616          region.delete(delete);
2617          processed = Boolean.TRUE;
2618        }
2619        break;
2620      default:
2621          throw new DoNotRetryIOException(
2622            "Unsupported mutate type: " + type.name());
2623      }
2624      if (processed != null) {
2625        builder.setProcessed(processed.booleanValue());

2631      }
2632      return builder.build();
2633    } catch (IOException ie) {

2635      throw new ServiceException(ie);
2636    } finally {
2637      if (quota != null) {
2638        quota.close();
2639      }
2640    }

2757  private long getTimeLimit(HBaseRpcController controller, boolean allowHeartbeatMessages) {
2758    // Set the time limit to be half of the more restrictive timeout value (one of the
2759    // timeout values must be positive). In the event that both values are positive, the
2760    // more restrictive of the two is used to calculate the limit.
2761    if (allowHeartbeatMessages && (scannerLeaseTimeoutPeriod > 0 || rpcTimeout > 0)) {
2762      long timeLimitDelta;
2763      if (scannerLeaseTimeoutPeriod > 0 && rpcTimeout > 0) {
2764        timeLimitDelta = Math.min(scannerLeaseTimeoutPeriod, rpcTimeout);
2765      } else {
2766        timeLimitDelta = scannerLeaseTimeoutPeriod > 0 ? scannerLeaseTimeoutPeriod : rpcTimeout;
2767      }
2768      if (controller != null && controller.getCallTimeout() > 0) {
2769        timeLimitDelta = Math.min(timeLimitDelta, controller.getCallTimeout());
2770      }
2771      // Use half of whichever timeout value was more restrictive... But don't allow
2772      // the time limit to be less than the allowable minimum (could cause an
2773      // immediatate timeout before scanning any data).
2774      timeLimitDelta = Math.max(timeLimitDelta / 2, minimumScanTimeLimitDelta);
2775      // XXX: Can not use EnvironmentEdge here because TestIncrementTimeRange use a
2776      // ManualEnvironmentEdge. Consider using System.nanoTime instead.
2777      return System.currentTimeMillis() + timeLimitDelta;
2778    }
2779    // Default value of timeLimit is negative to indicate no timeLimit should be
2780    // enforced.
2781    return -1L;
2782  }

2795  // return whether we have more results in region.
2796  private void scan(HBaseRpcController controller, ScanRequest request, RegionScannerHolder rsh,
2797      long maxQuotaResultSize, int maxResults, int limitOfRows, List<Result> results,
2798      ScanResponse.Builder builder, MutableObject lastBlock, RpcCallContext context)
2799      throws IOException {
2800    Region region = rsh.r;
2801    RegionScanner scanner = rsh.s;
2802    long maxResultSize;
2803    if (scanner.getMaxResultSize() > 0) {
2804      maxResultSize = Math.min(scanner.getMaxResultSize(), maxQuotaResultSize);
2805    } else {
2806      maxResultSize = maxQuotaResultSize;
2807    }
2808    // This is cells inside a row. Default size is 10 so if many versions or many cfs,
2809    // then we'll resize. Resizings show in profiler. Set it higher than 10. For now
2810    // arbitrary 32. TODO: keep record of general size of results being returned.
2811    List<Cell> values = new ArrayList<>(32);
2812    region.startRegionOperation(Operation.SCAN);
2813    try {
2814      int numOfResults = 0;
2815      int numOfCompleteRows = 0;
2816      long before = EnvironmentEdgeManager.currentTime();
2817      synchronized (scanner) {
2818        boolean stale = (region.getRegionInfo().getReplicaId() != 0);
2819        boolean clientHandlesPartials =
2820            request.hasClientHandlesPartials() && request.getClientHandlesPartials();
2821        boolean clientHandlesHeartbeats =
2822            request.hasClientHandlesHeartbeats() && request.getClientHandlesHeartbeats();
2823
2824        // On the server side we must ensure that the correct ordering of partial results is
2825        // returned to the client to allow them to properly reconstruct the partial results.
2826        // If the coprocessor host is adding to the result list, we cannot guarantee the
2827        // correct ordering of partial results and so we prevent partial results from being
2828        // formed.
2829        boolean serverGuaranteesOrderOfPartials = results.isEmpty();
2830        boolean allowPartialResults = clientHandlesPartials && serverGuaranteesOrderOfPartials;
2831        boolean moreRows = false;
2832
2833        // Heartbeat messages occur when the processing of the ScanRequest is exceeds a
2834        // certain time threshold on the server. When the time threshold is exceeded, the
2835        // server stops the scan and sends back whatever Results it has accumulated within
2836        // that time period (may be empty). Since heartbeat messages have the potential to
2837        // create partial Results (in the event that the timeout occurs in the middle of a
2838        // row), we must only generate heartbeat messages when the client can handle both
2839        // heartbeats AND partials
2840        boolean allowHeartbeatMessages = clientHandlesHeartbeats && allowPartialResults;
2841
2842        long timeLimit = getTimeLimit(controller, allowHeartbeatMessages);
2843
2844        final LimitScope sizeScope =
2845            allowPartialResults ? LimitScope.BETWEEN_CELLS : LimitScope.BETWEEN_ROWS;
2846        final LimitScope timeScope =
2847            allowHeartbeatMessages ? LimitScope.BETWEEN_CELLS : LimitScope.BETWEEN_ROWS;
2848
2849        boolean trackMetrics = request.hasTrackScanMetrics() && request.getTrackScanMetrics();
2850
2851        // Configure with limits for this RPC. Set keep progress true since size progress
2852        // towards size limit should be kept between calls to nextRaw
2853        ScannerContext.Builder contextBuilder = ScannerContext.newBuilder(true);
2854        // maxResultSize - either we can reach this much size for all cells(being read) data or sum
2855        // of heap size occupied by cells(being read). Cell data means its key and value parts.
2856        contextBuilder.setSizeLimit(sizeScope, maxResultSize, maxResultSize);
2857        contextBuilder.setBatchLimit(scanner.getBatch());
2858        contextBuilder.setTimeLimit(timeScope, timeLimit);
2859        contextBuilder.setTrackMetrics(trackMetrics);
2860        ScannerContext scannerContext = contextBuilder.build();
2861        boolean limitReached = false;
2862        while (numOfResults < maxResults) {
2863          // Reset the batch progress to 0 before every call to RegionScanner#nextRaw. The
2864          // batch limit is a limit on the number of cells per Result. Thus, if progress is
2865          // being tracked (i.e. scannerContext.keepProgress() is true) then we need to
2866          // reset the batch progress between nextRaw invocations since we don't want the
2867          // batch progress from previous calls to affect future calls
2868          scannerContext.setBatchProgress(0);
2869
2870          // Collect values to be returned here
2871          moreRows = scanner.nextRaw(values, scannerContext);
2872
2873          if (!values.isEmpty()) {
2874            if (limitOfRows > 0) {
2875              // First we need to check if the last result is partial and we have a row change. If
2876              // so then we need to increase the numOfCompleteRows.
2877              if (results.isEmpty()) {
2878                if (rsh.rowOfLastPartialResult != null &&
2879                    !CellUtil.matchingRow(values.get(0), rsh.rowOfLastPartialResult)) {
2880                  numOfCompleteRows++;
2881                  checkLimitOfRows(numOfCompleteRows, limitOfRows, moreRows, scannerContext,
2882                    builder);
2883                }
2884              } else {
2885                Result lastResult = results.get(results.size() - 1);
2886                if (lastResult.mayHaveMoreCellsInRow() &&
2887                    !CellUtil.matchingRow(values.get(0), lastResult.getRow())) {
2888                  numOfCompleteRows++;
2889                  checkLimitOfRows(numOfCompleteRows, limitOfRows, moreRows, scannerContext,
2890                    builder);
2891                }
2892              }
2893              if (builder.hasMoreResults() && !builder.getMoreResults()) {
2894                break;
2895              }
2896            }

2938        }

2940        // client requested metrics, retrieve the metrics from the scanner context.
2941        if (trackMetrics) {
2942          Map<String, Long> metrics = scannerContext.getMetrics().getMetricsMap();

2947            pairBuilder.setName(entry.getKey());
2948            pairBuilder.setValue(entry.getValue());
2949            metricBuilder.addMetrics(pairBuilder.build());
2950          }

2956      long end = EnvironmentEdgeManager.currentTime();

2958      region.getMetrics().updateScanTime(end - before);
2959      if (regionServer.metricsRegionServer != null) {

2961        regionServer.metricsRegionServer.updateScanTime(end - before);

2963    } finally {

2967    if (region.getCoprocessorHost() != null) {
2968      region.getCoprocessorHost().postScannerNext(scanner, results, maxResults, true);
2969    }
