018package org.apache.hadoop.hbase.regionserver;
019
020import java.util.List;
021import java.util.concurrent.atomic.AtomicInteger;
022
023import org.apache.hadoop.hbase.Cell;
024import org.apache.yetus.audience.InterfaceAudience;
025
026/**
027 * A MemStoreLAB implementation which wraps N MemStoreLABs. Its main duty is in proper managing the
028 * close of the individual MemStoreLAB. This is treated as an immutable one and so do not allow to
029 * add any more Cells into it. {@link #copyCellInto(Cell)} throws Exception
030 */
031@InterfaceAudience.Private
032public class ImmutableMemStoreLAB implements MemStoreLAB {
033
034  private final AtomicInteger openScannerCount = new AtomicInteger();
035  private volatile boolean closed = false;
036
037  private final List<MemStoreLAB> mslabs;
038
039  public ImmutableMemStoreLAB(List<MemStoreLAB> mslabs) {
040    this.mslabs = mslabs;
041  }
042
043  @Override
044  public Cell copyCellInto(Cell cell) {
045    throw new IllegalStateException("This is an Immutable MemStoreLAB.");
046  }
047
048  /**
049   * The process of merging assumes all cells are allocated on mslab.
050   * There is a rare case in which the first immutable segment,
051   * participating in a merge, is a CSLM.
052   * Since the CSLM hasn't been flattened yet, and there is no point in flattening it (since it is
053   * going to be merged), its big cells (for whom size > maxAlloc) must be copied into mslab.
054   * This method copies the passed cell into the first mslab in the mslabs list,
055   * returning either a new cell instance over the copied data,
056   * or null when this cell cannt be copied.
057   */
058  @Override
059  public Cell forceCopyOfBigCellInto(Cell cell) {
060    MemStoreLAB mslab = this.mslabs.get(0);
061    return mslab.forceCopyOfBigCellInto(cell);
062  }
063
064  /* Creating chunk to be used as index chunk in CellChunkMap, part of the chunks array.
065  ** Returning a new chunk, without replacing current chunk,
066  ** meaning MSLABImpl does not make the returned chunk as CurChunk.
067  ** The space on this chunk will be allocated externally.
068  ** The interface is only for external callers
069  */
070  @Override
071  public Chunk getNewExternalChunk() {
072    MemStoreLAB mslab = this.mslabs.get(0);
073    return mslab.getNewExternalChunk();
074  }
075
076  /* Creating chunk to be used as data chunk in CellChunkMap.
077  ** This chunk is bigger the normal constant chunk size, and thus called JumboChunk it is used for
078  ** jumbo cells (which size is bigger than normal chunks).
079  ** Jumbo Chunks are needed only for CCM and thus are created only in
080  ** CompactingMemStore.IndexType.CHUNK_MAP type.
081  ** Returning a new chunk, without replacing current chunk,
082  ** meaning MSLABImpl does not make the returned chunk as CurChunk.
083  ** The space on this chunk will be allocated externally.
084  ** The interface is only for external callers
085  */
086  @Override
087  public Chunk getNewExternalJumboChunk(int size) {
088    MemStoreLAB mslab = this.mslabs.get(0);
089    return mslab.getNewExternalJumboChunk(size);
090  }
091
092  @Override
093  public void close() {
094    // 'openScannerCount' here tracks the scanners opened on segments which directly refer to this
095    // MSLAB. The individual MSLABs this refers also having its own 'openScannerCount'. The usage of
096    // the variable in close() and decScannerCount() is as as that in HeapMemstoreLAB. Here the
097    // close just delegates the call to the individual MSLABs. The actual return of the chunks to
098    // MSLABPool will happen within individual MSLABs only (which is at the leaf level).
099    // Say an ImmutableMemStoreLAB is created over 2 HeapMemStoreLABs at some point and at that time
100    // both of them were referred by ongoing scanners. So they have > 0 'openScannerCount'. Now over
101    // the new Segment some scanners come in and this MSLABs 'openScannerCount' also goes up and
102    // then come down on finish of scanners. Now a close() call comes to this Immutable MSLAB. As
103    // it's 'openScannerCount' is zero it will call close() on both of the Heap MSLABs. Say by that
104    // time the old scanners on one of the MSLAB got over where as on the other, still an old
105    // scanner is going on. The call close() on that MSLAB will not close it immediately but will
106    // just mark it for closure as it's 'openScannerCount' still > 0. Later once the old scan is
107    // over, the decScannerCount() call will do the actual close and return of the chunks.
108    this.closed = true;
109    // When there are still on going scanners over this MSLAB, we will defer the close until all
110    // scanners finish. We will just mark it for closure. See #decScannerCount(). This will be
111    // called at end of every scan. When it is marked for closure and scanner count reached 0, we
112    // will do the actual close then.
113    checkAndCloseMSLABs(openScannerCount.get());
114  }
115
116  private void checkAndCloseMSLABs(int openScanners) {
117    if (openScanners == 0) {
118      for (MemStoreLAB mslab : this.mslabs) {
119        mslab.close();
120      }
121    }
122  }
123
124  @Override
125  public void incScannerCount() {
126    this.openScannerCount.incrementAndGet();
127  }
128
129  @Override
130  public void decScannerCount() {
131    int count = this.openScannerCount.decrementAndGet();
132    if (this.closed) {
133      checkAndCloseMSLABs(count);
134    }
135  }
136
137  @Override
138  public boolean isOnHeap() {
139    return !isOffHeap();
140  }
141
142  @Override
143  public boolean isOffHeap() {
144    return ChunkCreator.getInstance().isOffheap();
145  }
146

148}
