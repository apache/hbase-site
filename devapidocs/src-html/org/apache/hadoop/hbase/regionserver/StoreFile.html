018package org.apache.hadoop.hbase.regionserver;
019
020import java.io.IOException;
021import java.util.Comparator;
022import java.util.Optional;
023import java.util.OptionalLong;
024
025import org.apache.hadoop.fs.Path;
026import org.apache.hadoop.hbase.Cell;
027import org.apache.hadoop.hbase.HBaseInterfaceAudience;
028import org.apache.hadoop.hbase.HDFSBlocksDistribution;
029import org.apache.yetus.audience.InterfaceAudience;
030import org.apache.yetus.audience.InterfaceStability;
031
032/**
033 * An interface to describe a store data file.
034 * <p>
035 * <strong>NOTICE: </strong>this interface is mainly designed for coprocessor, so it will not expose
036 * all the internal APIs for a 'store file'. If you are implementing something inside HBase, i.e,
037 * not a coprocessor hook, usually you should use {@link HStoreFile} directly as it is the only
038 * implementation of this interface.
039 */
040@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
041@InterfaceStability.Evolving
042public interface StoreFile {
043
044  /**
045   * Get the first key in this store file.
046   */
047  Optional<Cell> getFirstKey();
048
049  /**
050   * Get the last key in this store file.
051   */
052  Optional<Cell> getLastKey();
053
054  /**
055   * Get the comparator for comparing two cells.
056   */
057  Comparator<Cell> getComparator();
058
059  /**
060   * Get max of the MemstoreTS in the KV's in this store file.
061   */
062  long getMaxMemStoreTS();
063
064  /**
065   * @return Path or null if this StoreFile was made with a Stream.
066   */
067  Path getPath();
068
069  /**
070   * @return Returns the qualified path of this StoreFile
071   */
072  Path getQualifiedPath();
073
074  /**
075   * @return True if this is a StoreFile Reference.
076   */
077  boolean isReference();
078
079  /**
080   * @return True if this is HFile.
081   */
082  boolean isHFile();
083
084  /**
085   * @return True if this file was made by a major compaction.
086   */
087  boolean isMajorCompactionResult();
088
089  /**
090   * @return True if this file should not be part of a minor compaction.
091   */
092  boolean excludeFromMinorCompaction();
093
094  /**
095   * @return This files maximum edit sequence id.
096   */
097  long getMaxSequenceId();
098
099  /**
100   * Get the modification time of this store file. Usually will access the file system so throws
101   * IOException.
102   */
103  long getModificationTimeStamp() throws IOException;
104
105  /**
106   * Check if this storefile was created by bulk load. When a hfile is bulk loaded into HBase, we
107   * append {@code '_SeqId_<id-when-loaded>'} to the hfile name, unless
108   * "hbase.mapreduce.bulkload.assign.sequenceNumbers" is explicitly turned off. If
109   * "hbase.mapreduce.bulkload.assign.sequenceNumbers" is turned off, fall back to
110   * BULKLOAD_TIME_KEY.
111   * @return true if this storefile was created by bulk load.
112   */
113  boolean isBulkLoadResult();
114
115  /**
116   * Return the timestamp at which this bulk load file was generated.
117   */
118  OptionalLong getBulkLoadTimestamp();
119
120  /**
121   * @return the cached value of HDFS blocks distribution. The cached value is calculated when store
122   *         file is opened.
123   */
124  HDFSBlocksDistribution getHDFSBlockDistribution();
125
126  /**
127   * @return a length description of this StoreFile, suitable for debug output
128   */
129  String toStringDetailed();
130
131  /**
132   * Get the min timestamp of all the cells in the store file.
133   */
134  OptionalLong getMinimumTimestamp();
135
136  /**
137   * Get the max timestamp of all the cells in the store file.
138   */
139  OptionalLong getMaximumTimestamp();
140}
