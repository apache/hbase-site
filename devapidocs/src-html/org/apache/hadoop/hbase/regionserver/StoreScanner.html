166  /** An internal constructor. */
167  protected StoreScanner(Store store, Scan scan, final ScanInfo scanInfo,
168      final NavigableSet<byte[]> columns, long readPt, boolean cacheBlocks) {
169    this.readPt = readPt;
170    this.store = store;
171    this.cacheBlocks = cacheBlocks;
172    get = scan.isGetScan();
173    int numCol = columns == null ? 0 : columns.size();
174    explicitColumnQuery = numCol > 0;
175    this.scan = scan;
176    this.now = EnvironmentEdgeManager.currentTime();
177    this.oldestUnexpiredTS = scan.isRaw() ? 0L : now - scanInfo.getTtl();
178    this.minVersions = scanInfo.getMinVersions();
179
180     // We look up row-column Bloom filters for multi-column queries as part of
181     // the seek operation. However, we also look the row-column Bloom filter
182     // for multi-row (non-"get") scans because this is not done in
183     // StoreFile.passesBloomFilter(Scan, SortedSet<byte[]>).
184     this.useRowColBloom = numCol > 1 || (!get && numCol == 1);
185
186     this.maxRowSize = scanInfo.getTableMaxRowSize();
187    if (get) {
188      this.readType = Scan.ReadType.PREAD;
189      this.scanUsePread = true;
190    } else {
191      if (scan.getReadType() == Scan.ReadType.DEFAULT) {
192        this.readType = scanInfo.isUsePread() ? Scan.ReadType.PREAD : Scan.ReadType.DEFAULT;
193      } else {
194        this.readType = scan.getReadType();
195      }
196      // Always start with pread unless user specific stream. Will change to stream later if
197      // readType is default if the scan keeps running for a long time.
198      this.scanUsePread = this.readType != Scan.ReadType.STREAM;
199    }
200    this.preadMaxBytes = scanInfo.getPreadMaxBytes();
201    this.cellsPerHeartbeatCheck = scanInfo.getCellsPerTimeoutCheck();
202    // Parallel seeking is on if the config allows and more there is more than one store file.
203    if (this.store != null && this.store.getStorefilesCount() > 1) {
204      RegionServerServices rsService = ((HStore) store).getHRegion().getRegionServerServices();
205      if (rsService != null && scanInfo.isParallelSeekEnabled()) {
206        this.parallelSeekEnabled = true;
207        this.executor = rsService.getExecutorService();
208      }
209    }
210  }

216  /**
217   * Opens a scanner across memstore, snapshot, and all StoreFiles. Assumes we
218   * are not in a compaction.
219   *
220   * @param store who we scan
221   * @param scan the spec
222   * @param columns which columns we are scanning
223   * @throws IOException
224   */
225  public StoreScanner(Store store, ScanInfo scanInfo, Scan scan, final NavigableSet<byte[]> columns,
226      long readPt)
227  throws IOException {
228    this(store, scan, scanInfo, columns, readPt, scan.getCacheBlocks());
229    if (columns != null && scan.isRaw()) {
230      throw new DoNotRetryIOException("Cannot specify any column for a raw scan");
231    }
232    matcher = UserScanQueryMatcher.create(scan, scanInfo, columns, oldestUnexpiredTS, now,
233      store.getCoprocessorHost());
234
235    this.store.addChangedReaderObserver(this);
236
237    try {
238      // Pass columns to try to filter out unnecessary StoreFiles.
239      List<KeyValueScanner> scanners = getScannersNoCompaction();
240
241      // Seek all scanners to the start of the Row (or if the exact matching row
242      // key does not exist, then to the start of the next matching Row).
243      // Always check bloom filter to optimize the top row seek for delete
244      // family marker.
245      seekScanners(scanners, matcher.getStartKey(), explicitColumnQuery && lazySeekEnabledGlobally,
246        parallelSeekEnabled);
247
248      // set storeLimit
249      this.storeLimit = scan.getMaxResultsPerColumnFamily();
250
251      // set rowOffset
252      this.storeOffset = scan.getRowOffsetPerColumnFamily();
253      addCurrentScanners(scanners);
254      // Combine all seeked scanners with a heap
255      resetKVHeap(scanners, store.getComparator());
256    } catch (IOException e) {
257      // remove us from the HStore#changedReaderObservers here or we'll have no chance to
258      // and might cause memory leak
259      this.store.deleteChangedReaderObserver(this);
260      throw e;
261    }
262  }

298  private StoreScanner(Store store, ScanInfo scanInfo, Scan scan,
299      List<? extends KeyValueScanner> scanners, ScanType scanType, long smallestReadPoint,
300      long earliestPutTs, byte[] dropDeletesFromRow, byte[] dropDeletesToRow) throws IOException {
301    this(store, scan, scanInfo, null,
302        ((HStore) store).getHRegion().getReadPoint(IsolationLevel.READ_COMMITTED), false);
303    if (scan.hasFilter() || (scan.getStartRow() != null && scan.getStartRow().length > 0)
304        || (scan.getStopRow() != null && scan.getStopRow().length > 0)
305        || !scan.getTimeRange().isAllTime()) {
306      // use legacy query matcher since we do not consider the scan object in our code. Only used to
307      // keep compatibility for coprocessor.
308      matcher = LegacyScanQueryMatcher.create(scan, scanInfo, null, scanType, smallestReadPoint,
309        earliestPutTs, oldestUnexpiredTS, now, dropDeletesFromRow, dropDeletesToRow,
310        store.getCoprocessorHost());
311    } else {
312      matcher = CompactionScanQueryMatcher.create(scanInfo, scanType, smallestReadPoint,
313        earliestPutTs, oldestUnexpiredTS, now, dropDeletesFromRow, dropDeletesToRow,
314        store.getCoprocessorHost());
315    }
316
317    // Filter the list of scanners using Bloom filters, time range, TTL, etc.
318    scanners = selectScannersFrom(scanners);
319
320    // Seek all scanners to the initial key
321    seekScanners(scanners, matcher.getStartKey(), false, parallelSeekEnabled);
322    addCurrentScanners(scanners);
323    // Combine all seeked scanners with a heap
324    resetKVHeap(scanners, store.getComparator());
325  }

347  public StoreScanner(final Scan scan, ScanInfo scanInfo, ScanType scanType,
348      final NavigableSet<byte[]> columns, final List<? extends KeyValueScanner> scanners, long earliestPutTs,
349      long readPt) throws IOException {
350    this(null, scan, scanInfo, columns, readPt,
351        scanType == ScanType.USER_SCAN ? scan.getCacheBlocks() : false);
352    if (scanType == ScanType.USER_SCAN) {
353      this.matcher = UserScanQueryMatcher.create(scan, scanInfo, columns, oldestUnexpiredTS, now,
354        null);
355    } else {
356      if (scan.hasFilter() || (scan.getStartRow() != null && scan.getStartRow().length > 0)
357          || (scan.getStopRow() != null && scan.getStopRow().length > 0)
358          || !scan.getTimeRange().isAllTime() || columns != null) {
359        // use legacy query matcher since we do not consider the scan object in our code. Only used
360        // to keep compatibility for coprocessor.
361        matcher = LegacyScanQueryMatcher.create(scan, scanInfo, columns, scanType, Long.MAX_VALUE,
362          earliestPutTs, oldestUnexpiredTS, now, null, null, store.getCoprocessorHost());
363      } else {
364        this.matcher = CompactionScanQueryMatcher.create(scanInfo, scanType, Long.MAX_VALUE,
365          earliestPutTs, oldestUnexpiredTS, now, null, null, null);
366      }
367    }
368
369    // Seek all scanners to the initial key
370    seekScanners(scanners, matcher.getStartKey(), false, parallelSeekEnabled);
371    addCurrentScanners(scanners);
372    resetKVHeap(scanners, scanInfo.getComparator());
373  }

385  /**
386   * Seek the specified scanners with the given key
387   * @param scanners
388   * @param seekKey
389   * @param isLazy true if using lazy seek
390   * @param isParallelSeek true if using parallel seek
391   * @throws IOException
392   */
393  protected void seekScanners(List<? extends KeyValueScanner> scanners,
394      Cell seekKey, boolean isLazy, boolean isParallelSeek)
395      throws IOException {
396    // Seek all scanners to the start of the Row (or if the exact matching row
397    // key does not exist, then to the start of the next matching Row).
398    // Always check bloom filter to optimize the top row seek for delete
399    // family marker.
400    if (isLazy) {
401      for (KeyValueScanner scanner : scanners) {
402        scanner.requestSeek(seekKey, false, true);
403      }
404    } else {
405      if (!isParallelSeek) {
406        long totalScannersSoughtBytes = 0;
407        for (KeyValueScanner scanner : scanners) {
408          if (matcher.isUserScan() && totalScannersSoughtBytes >= maxRowSize) {
409            throw new RowTooBigException("Max row size allowed: " + maxRowSize
410              + ", but row is bigger than that");
411          }
412          scanner.seek(seekKey);
413          Cell c = scanner.peek();
414          if (c != null) {
415            totalScannersSoughtBytes += CellUtil.estimatedSerializedSizeOf(c);
416          }
417        }
418      } else {
419        parallelSeek(scanners, seekKey);
420      }
421    }
422  }

425      CellComparator comparator) throws IOException {
426    // Combine all seeked scanners with a heap
427    heap = new KeyValueHeap(scanners, comparator);

430  /**
431   * Filters the given list of scanners using Bloom filter, time range, and
432   * TTL.
433   * <p>
434   * Will be overridden by testcase so declared as protected.
435   */
436  @VisibleForTesting
437  protected List<KeyValueScanner> selectScannersFrom(
438      final List<? extends KeyValueScanner> allScanners) {
439    boolean memOnly;
440    boolean filesOnly;
441    if (scan instanceof InternalScan) {
442      InternalScan iscan = (InternalScan)scan;
443      memOnly = iscan.isCheckOnlyMemStore();
444      filesOnly = iscan.isCheckOnlyStoreFiles();
445    } else {
446      memOnly = false;
447      filesOnly = false;
448    }
449
450    List<KeyValueScanner> scanners = new ArrayList<>(allScanners.size());
451
452    // We can only exclude store files based on TTL if minVersions is set to 0.
453    // Otherwise, we might have to return KVs that have technically expired.
454    long expiredTimestampCutoff = minVersions == 0 ? oldestUnexpiredTS: Long.MIN_VALUE;
455
456    // include only those scan files which pass all filters
457    for (KeyValueScanner kvs : allScanners) {
458      boolean isFile = kvs.isFileScanner();
459      if ((!isFile && filesOnly) || (isFile && memOnly)) {
460        continue;
461      }
462
463      if (kvs.shouldUseScanner(scan, store, expiredTimestampCutoff)) {
464        scanners.add(kvs);
465      } else {
466        kvs.close();
467      }
468    }
469    return scanners;
470  }

488  private void close(boolean withHeapClose) {
489    if (this.closing) {
490      return;
491    }
492    if (withHeapClose) {
493      this.closing = true;
494    }
495    // Under test, we dont have a this.store
496    if (this.store != null) {
497      this.store.deleteChangedReaderObserver(this);
498    }
499    if (withHeapClose) {
500      clearAndClose(memStoreScannersAfterFlush);
501      for (KeyValueHeap h : this.heapsForDelayedClose) {
502        h.close();
503      }
504      this.heapsForDelayedClose.clear();
505      if (this.heap != null) {
506        this.heap.close();
507        this.currentScanners.clear();
508        this.heap = null; // CLOSED!
509      }
510    } else {
511      if (this.heap != null) {
512        this.heapsForDelayedClose.add(this.heap);
513        this.currentScanners.clear();
514        this.heap = null;
515      }
516    }
517  }

532  /**
533   * Get the next row of values from this Store.
534   * @param outResult
535   * @param scannerContext
536   * @return true if there are more rows, false if scanner is done
537   */
538  @Override
539  public boolean next(List<Cell> outResult, ScannerContext scannerContext) throws IOException {
540    if (scannerContext == null) {
541      throw new IllegalArgumentException("Scanner context cannot be null");
542    }
543    if (checkFlushed() && reopenAfterFlush()) {
544      return scannerContext.setScannerState(NextState.MORE_VALUES).hasMoreValues();
545    }
546
547    // if the heap was left null, then the scanners had previously run out anyways, close and
548    // return.
549    if (this.heap == null) {
550      // By this time partial close should happened because already heap is null
551      close(false);// Do all cleanup except heap.close()
552      return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
553    }
554
555    Cell cell = this.heap.peek();
556    if (cell == null) {
557      close(false);// Do all cleanup except heap.close()
558      return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
559    }
560
561    // only call setRow if the row changes; avoids confusing the query matcher
562    // if scanning intra-row
563
564    // If no limits exists in the scope LimitScope.Between_Cells then we are sure we are changing
565    // rows. Else it is possible we are still traversing the same row so we must perform the row
566    // comparison.
567    if (!scannerContext.hasAnyLimit(LimitScope.BETWEEN_CELLS) || matcher.currentRow() == null) {
568      this.countPerRow = 0;
569      matcher.setToNewRow(cell);
570    }
571
572    // Clear progress away unless invoker has indicated it should be kept.
573    if (!scannerContext.getKeepProgress()) {
574      scannerContext.clearProgress();
575    }
576
577    // Only do a sanity-check if store and comparator are available.
578    CellComparator comparator = store != null ? store.getComparator() : null;
579
580    int count = 0;
581    long totalBytesRead = 0;
582
583    LOOP: do {
584      // Update and check the time limit based on the configured value of cellsPerTimeoutCheck
585      if ((kvsScanned % cellsPerHeartbeatCheck == 0)) {
586        scannerContext.updateTimeProgress();
587        if (scannerContext.checkTimeLimit(LimitScope.BETWEEN_CELLS)) {
588          return scannerContext.setScannerState(NextState.TIME_LIMIT_REACHED).hasMoreValues();
589        }
590      }
591      // Do object compare - we set prevKV from the same heap.
592      if (prevCell != cell) {
593        ++kvsScanned;
594      }
595      checkScanOrder(prevCell, cell, comparator);
596      int cellSize = CellUtil.estimatedSerializedSizeOf(cell);
597      bytesRead += cellSize;
598      prevCell = cell;
599      ScanQueryMatcher.MatchCode qcode = matcher.match(cell);
600      switch (qcode) {
601        case INCLUDE:
602        case INCLUDE_AND_SEEK_NEXT_ROW:
603        case INCLUDE_AND_SEEK_NEXT_COL:
604
605          Filter f = matcher.getFilter();
606          if (f != null) {
607            cell = f.transformCell(cell);
608          }
609
610          this.countPerRow++;
611          if (storeLimit > -1 && this.countPerRow > (storeLimit + storeOffset)) {
612            // do what SEEK_NEXT_ROW does.
613            if (!matcher.moreRowsMayExistAfter(cell)) {
614              close(false);// Do all cleanup except heap.close()
615              return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
616            }
617            matcher.clearCurrentRow();
618            seekToNextRow(cell);
619            break LOOP;
620          }
621
622          // add to results only if we have skipped #storeOffset kvs
623          // also update metric accordingly
624          if (this.countPerRow > storeOffset) {
625            outResult.add(cell);
626
627            // Update local tracking information
628            count++;
629            totalBytesRead += cellSize;
630
631            // Update the progress of the scanner context
632            scannerContext.incrementSizeProgress(cellSize, CellUtil.estimatedHeapSizeOf(cell));
633            scannerContext.incrementBatchProgress(1);
634
635            if (matcher.isUserScan() && totalBytesRead > maxRowSize) {
636              throw new RowTooBigException(
637                  "Max row size allowed: " + maxRowSize + ", but the row is bigger than that.");
638            }
639          }
640
641          if (qcode == ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_ROW) {
642            if (!matcher.moreRowsMayExistAfter(cell)) {
643              close(false);// Do all cleanup except heap.close()
644              return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
645            }
646            matcher.clearCurrentRow();
647            seekOrSkipToNextRow(cell);
648          } else if (qcode == ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_COL) {
649            seekOrSkipToNextColumn(cell);
650          } else {
651            this.heap.next();
652          }
653
654          if (scannerContext.checkBatchLimit(LimitScope.BETWEEN_CELLS)) {
655            break LOOP;
656          }
657          if (scannerContext.checkSizeLimit(LimitScope.BETWEEN_CELLS)) {
658            break LOOP;
659          }
660          continue;
661
662        case DONE:
663          // Optimization for Gets! If DONE, no more to get on this row, early exit!
664          if (get) {
665            // Then no more to this row... exit.
666            close(false);// Do all cleanup except heap.close()
667            return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
668          }
669          matcher.clearCurrentRow();
670          return scannerContext.setScannerState(NextState.MORE_VALUES).hasMoreValues();
671
672        case DONE_SCAN:
673          close(false);// Do all cleanup except heap.close()
674          return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
675
676        case SEEK_NEXT_ROW:
677          // This is just a relatively simple end of scan fix, to short-cut end
678          // us if there is an endKey in the scan.
679          if (!matcher.moreRowsMayExistAfter(cell)) {
680            close(false);// Do all cleanup except heap.close()
681            return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
682          }
683          matcher.clearCurrentRow();
684          seekOrSkipToNextRow(cell);
685          break;
686
687        case SEEK_NEXT_COL:
688          seekOrSkipToNextColumn(cell);
689          break;
690
691        case SKIP:
692          this.heap.next();
693          break;
694
695        case SEEK_NEXT_USING_HINT:
696          Cell nextKV = matcher.getNextKeyHint(cell);
697          if (nextKV != null) {
698            seekAsDirection(nextKV);
699          } else {
700            heap.next();
701          }
702          break;
703
704        default:
705          throw new RuntimeException("UNEXPECTED");
706      }
707    } while ((cell = this.heap.peek()) != null);
708
709    if (count > 0) {
710      return scannerContext.setScannerState(NextState.MORE_VALUES).hasMoreValues();
711    }
712
713    // No more keys
714    close(false);// Do all cleanup except heap.close()
715    return scannerContext.setScannerState(NextState.NO_MORE_VALUES).hasMoreValues();
716  }

718  private void seekOrSkipToNextRow(Cell cell) throws IOException {
719    // If it is a Get Scan, then we know that we are done with this row; there are no more
720    // rows beyond the current one: don't try to optimize.
721    if (!get) {
722      if (trySkipToNextRow(cell)) {
723        return;
724      }
725    }
726    seekToNextRow(cell);
727  }

729  private void seekOrSkipToNextColumn(Cell cell) throws IOException {
730    if (!trySkipToNextColumn(cell)) {
731      seekAsDirection(matcher.getKeyForNextColumn(cell));
732    }

735  /**
736   * See if we should actually SEEK or rather just SKIP to the next Cell (see HBASE-13109).
737   * ScanQueryMatcher may issue SEEK hints, such as seek to next column, next row,
738   * or seek to an arbitrary seek key. This method decides whether a seek is the most efficient
739   * _actual_ way to get us to the requested cell (SEEKs are more expensive than SKIP, SKIP,
740   * SKIP inside the current, loaded block).
741   * It does this by looking at the next indexed key of the current HFile. This key
742   * is then compared with the _SEEK_ key, where a SEEK key is an artificial 'last possible key
743   * on the row' (only in here, we avoid actually creating a SEEK key; in the compare we work with
744   * the current Cell but compare as though it were a seek key; see down in
745   * matcher.compareKeyForNextRow, etc). If the compare gets us onto the
746   * next block we *_SEEK, otherwise we just SKIP to the next requested cell.
747   *
748   * <p>Other notes:
749   * <ul>
750   * <li>Rows can straddle block boundaries</li>
751   * <li>Versions of columns can straddle block boundaries (i.e. column C1 at T1 might be in a
752   * different block than column C1 at T2)</li>
753   * <li>We want to SKIP if the chance is high that we'll find the desired Cell after a
754   * few SKIPs...</li>
755   * <li>We want to SEEK when the chance is high that we'll be able to seek
756   * past many Cells, especially if we know we need to go to the next block.</li>
757   * </ul>
758   * <p>A good proxy (best effort) to determine whether SKIP is better than SEEK is whether
759   * we'll likely end up seeking to the next block (or past the next block) to get our next column.
760   * Example:
761   * <pre>
762   * |    BLOCK 1              |     BLOCK 2                   |
763   * |  r1/c1, r1/c2, r1/c3    |    r1/c4, r1/c5, r2/c1        |
764   *                                   ^         ^
765   *                                   |         |
766   *                           Next Index Key   SEEK_NEXT_ROW (before r2/c1)
767   *
768   *
769   * |    BLOCK 1                       |     BLOCK 2                      |
770   * |  r1/c1/t5, r1/c1/t4, r1/c1/t3    |    r1/c1/t2, r1/c1/T1, r1/c2/T3  |
771   *                                            ^              ^
772   *                                            |              |
773   *                                    Next Index Key        SEEK_NEXT_COL
774   * </pre>
775   * Now imagine we want columns c1 and c3 (see first diagram above), the 'Next Index Key' of r1/c4
776   * is > r1/c3 so we should seek to get to the c1 on the next row, r2. In second case, say we only
777   * want one version of c1, after we have it, a SEEK_COL will be issued to get to c2. Looking at
778   * the 'Next Index Key', it would land us in the next block, so we should SEEK. In other scenarios
779   * where the SEEK will not land us in the next block, it is very likely better to issues a series
780   * of SKIPs.
781   * @param cell current cell
782   * @return true means skip to next row, false means not
783   */
784  @VisibleForTesting
785  protected boolean trySkipToNextRow(Cell cell) throws IOException {
786    Cell nextCell = null;
787    do {
788      Cell nextIndexedKey = getNextIndexedKey();
789      if (nextIndexedKey != null && nextIndexedKey != KeyValueScanner.NO_NEXT_INDEXED_KEY
790          && matcher.compareKeyForNextRow(nextIndexedKey, cell) >= 0) {
791        this.heap.next();
792        ++kvsScanned;
793      } else {
794        return false;
795      }
796    } while ((nextCell = this.heap.peek()) != null && CellUtil.matchingRows(cell, nextCell));
797    return true;
798  }

854  /**
855   * @return if top of heap has changed (and KeyValueHeap has to try the next KV)
856   */
857  protected final boolean reopenAfterFlush() throws IOException {
858    Cell lastTop = heap.peek();
859    // When we have the scan object, should we not pass it to getScanners() to get a limited set of
860    // scanners? We did so in the constructor and we could have done it now by storing the scan
861    // object from the constructor
862    List<KeyValueScanner> scanners;
863    flushLock.lock();
864    try {
865      List<KeyValueScanner> allScanners = new ArrayList<>(flushedStoreFiles.size() + memStoreScannersAfterFlush.size());
866      allScanners.addAll(store.getScanners(flushedStoreFiles, cacheBlocks, get,
867        scanUsePread, false, matcher, scan.getStartRow(), scan.getStopRow(), this.readPt, false));
868      allScanners.addAll(memStoreScannersAfterFlush);
869      scanners = selectScannersFrom(allScanners);
870      // Clear the current set of flushed store files so that they don't get added again
871      flushedStoreFiles.clear();
872      memStoreScannersAfterFlush.clear();
873    } finally {
874      flushLock.unlock();
875    }
876
877    // Seek the new scanners to the last key
878    seekScanners(scanners, lastTop, false, parallelSeekEnabled);
879    // remove the older memstore scanner
880    for (int i = currentScanners.size() - 1; i >=0; i--) {
881      if (!currentScanners.get(i).isFileScanner()) {
882        currentScanners.remove(i).close();
883      } else {
884        // we add the memstore scanner to the end of currentScanners
885        break;
886      }
887    }
888    // add the newly created scanners on the flushed files and the current active memstore scanner
889    addCurrentScanners(scanners);
890    // Combine all seeked scanners with a heap
891    resetKVHeap(this.currentScanners, store.getComparator());
892    resetQueryMatcher(lastTop);
893    if (heap.peek() == null || store.getComparator().compareRows(lastTop, this.heap.peek()) != 0) {
894      if (LOG.isDebugEnabled()) {
895        LOG.debug("Storescanner.peek() is changed where before = " + lastTop.toString() +
896            ",and after = " + heap.peek());
897      }
898      return true;
899    }
900    return false;
901  }

903  private void resetQueryMatcher(Cell lastTopKey) {
904    // Reset the state of the Query Matcher and set to top row.
905    // Only reset and call setRow if the row changes; avoids confusing the
906    // query matcher if scanning intra-row.
907    Cell cell = heap.peek();
908    if (cell == null) {
909      cell = lastTopKey;
910    }
911    if ((matcher.currentRow() == null) || !CellUtil.matchingRows(cell, matcher.currentRow())) {
912      this.countPerRow = 0;
913      // The setToNewRow will call reset internally
914      matcher.setToNewRow(cell);
915    }
916  }

959  private void trySwitchToStreamRead() {
960    if (readType != Scan.ReadType.DEFAULT || !scanUsePread || closing || heap.peek() == null ||
961        bytesRead < preadMaxBytes) {
962      return;
963    }
964    if (LOG.isDebugEnabled()) {
965      LOG.debug("Switch to stream read because we have already read " + bytesRead +
966          " bytes from this scanner");
967    }
968    scanUsePread = false;
969    Cell lastTop = heap.peek();
970    Map<String, StoreFile> name2File = new HashMap<>(store.getStorefilesCount());
971    for (StoreFile file : store.getStorefiles()) {
972      name2File.put(file.getFileInfo().getActiveFileName(), file);
973    }
974    List<StoreFile> filesToReopen = new ArrayList<>();
975    List<KeyValueScanner> memstoreScanners = new ArrayList<>();
976    List<KeyValueScanner> scannersToClose = new ArrayList<>();
977    for (KeyValueScanner kvs : currentScanners) {
978      if (!kvs.isFileScanner()) {
979        memstoreScanners.add(kvs);
980      } else {
981        scannersToClose.add(kvs);
982        if (kvs.peek() == null) {
983          continue;
984        }
985        filesToReopen.add(name2File.get(kvs.getFilePath().getName()));
986      }
987    }
988    if (filesToReopen.isEmpty()) {
989      return;
990    }
991    List<KeyValueScanner> fileScanners = null;
992    List<KeyValueScanner> newCurrentScanners;
993    KeyValueHeap newHeap;
994    try {
995      fileScanners =
996          store.getScanners(filesToReopen, cacheBlocks, false, false, matcher, scan.getStartRow(),
997            scan.includeStartRow(), scan.getStopRow(), scan.includeStopRow(), readPt, false);
998      seekScanners(fileScanners, lastTop, false, parallelSeekEnabled);
999      newCurrentScanners = new ArrayList<>(fileScanners.size() + memstoreScanners.size());
1000      newCurrentScanners.addAll(fileScanners);
1001      newCurrentScanners.addAll(memstoreScanners);
1002      newHeap = new KeyValueHeap(newCurrentScanners, store.getComparator());
1003    } catch (Exception e) {
1004      LOG.warn("failed to switch to stream read", e);
1005      if (fileScanners != null) {
1006        fileScanners.forEach(KeyValueScanner::close);
1007      }
1008      return;
1009    }
1010    currentScanners.clear();
1011    addCurrentScanners(newCurrentScanners);
1012    this.heap = newHeap;
1013    resetQueryMatcher(lastTop);
1014    scannersToClose.forEach(KeyValueScanner::close);
1015  }

1017  protected final boolean checkFlushed() {
1018    // check the var without any lock. Suppose even if we see the old
1019    // value here still it is ok to continue because we will not be resetting
1020    // the heap but will continue with the referenced memstore's snapshot. For compactions
1021    // any way we don't need the updateReaders at all to happen as we still continue with
1022    // the older files
1023    if (flushed) {
1024      // If there is a flush and the current scan is notified on the flush ensure that the
1025      // scan's heap gets reset and we do a seek on the newly flushed file.
1026      if (this.closing) {
1027        return false;
1028      }
1029      // reset the flag
1030      flushed = false;
1031      return true;
1032    }
1033    return false;
1034  }

1044  /**
1045   * Seek storefiles in parallel to optimize IO latency as much as possible
1046   * @param scanners the list {@link KeyValueScanner}s to be read from
1047   * @param kv the KeyValue on which the operation is being requested
1048   * @throws IOException
1049   */
1050  private void parallelSeek(final List<? extends KeyValueScanner>
1051      scanners, final Cell kv) throws IOException {
1052    if (scanners.isEmpty()) return;
1053    int storeFileScannerCount = scanners.size();
1054    CountDownLatch latch = new CountDownLatch(storeFileScannerCount);
1055    List<ParallelSeekHandler> handlers = new ArrayList<>(storeFileScannerCount);
1056    for (KeyValueScanner scanner : scanners) {
1057      if (scanner instanceof StoreFileScanner) {
1058        ParallelSeekHandler seekHandler = new ParallelSeekHandler(scanner, kv,
1059          this.readPt, latch);
1060        executor.submit(seekHandler);
1061        handlers.add(seekHandler);
1062      } else {
1063        scanner.seek(kv);
1064        latch.countDown();
1065      }
1066    }
1067
1068    try {
1069      latch.await();
1070    } catch (InterruptedException ie) {
1071      throw (InterruptedIOException)new InterruptedIOException().initCause(ie);
1072    }
1073
1074    for (ParallelSeekHandler handler : handlers) {
1075      if (handler.getErr() != null) {
1076        throw new IOException(handler.getErr());
1077      }
1078    }
1079  }

1112  @Override
1113  public void shipped() throws IOException {
1114    if (prevCell != null) {
1115      // Do the copy here so that in case the prevCell ref is pointing to the previous
1116      // blocks we can safely release those blocks.
1117      // This applies to blocks that are got from Bucket cache, L1 cache and the blocks
1118      // fetched from HDFS. Copying this would ensure that we let go the references to these
1119      // blocks so that they can be GCed safely(in case of bucket cache)
1120      prevCell = KeyValueUtil.toNewKeyCell(this.prevCell);
1121    }
1122    matcher.beforeShipped();
1123    // There wont be further fetch of Cells from these scanners. Just close.
1124    this.heapsForDelayedClose.forEach(KeyValueHeap::close);
1125    this.heapsForDelayedClose.clear();
1126    if (this.heap != null) {
1127      this.heap.shipped();
1128      // When switching from pread to stream, we will open a new scanner for each store file, but
1129      // the old scanner may still track the HFileBlocks we have scanned but not sent back to client
1130      // yet. If we close the scanner immediately then the HFileBlocks may be messed up by others
1131      // before we serialize and send it back to client. The HFileBlocks will be released in shipped
1132      // method, so we here will also open new scanners and close old scanners in shipped method.
1133      // See HBASE-18055 for more details.
1134      trySwitchToStreamRead();
1135    }
1136  }
