019package org.apache.hadoop.hbase.regionserver.compactions;
020
021import org.apache.hadoop.hbase.HBaseInterfaceAudience;
022import org.apache.hadoop.hbase.regionserver.StoreFile;
023import org.apache.yetus.audience.InterfaceAudience;
024
025import java.util.Collection;
026
027/**
028 * Coprocessors use this interface to get details about compaction.
029 */
030@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
031public interface CompactionRequest {
032
033  /**
034   * @return unmodifiable collection of StoreFiles in compaction
035   */
036  Collection<? extends StoreFile> getFiles();
037
038  /**
039   * @return total size of all StoreFiles in compaction
040   */
041  long getSize();
042
043  /**
044   * @return <code>true</code> if major compaction or all files are compacted
045   */
046  boolean isAllFiles();
047
048  /**
049   * @return <code>true</code> if major compaction
050   */
051  boolean isMajor();
052
053  /**
054   * @return priority of compaction request
055   */
056  int getPriority();
057
058  /**
059   * @return <code>true</code> if compaction is Off-peak
060   */
061  boolean isOffPeak();
062
063  /**
064   * @return compaction request creation time in milliseconds
065   */
066  long getSelectionTime();
067
068}
