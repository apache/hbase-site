018package org.apache.hadoop.hbase.regionserver.compactions;
019
020import org.apache.commons.logging.Log;
021import org.apache.commons.logging.LogFactory;
022import org.apache.hadoop.hbase.ScheduledChore;
023import org.apache.hadoop.hbase.Stoppable;
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025import org.apache.hadoop.hbase.regionserver.Region;
026import org.apache.hadoop.hbase.regionserver.Store;
027
028/**
029 * A chore service that periodically cleans up the compacted files when there are no active readers
030 * using those compacted files and also helps in clearing the block cache with these compacted
031 * file entries
032 */
033@InterfaceAudience.Private
034public class CompactedHFilesDischarger extends ScheduledChore {
035  private static final Log LOG = LogFactory.getLog(CompactedHFilesDischarger.class);
036  private Region region;
037
038  /**
039   * @param period the period of time to sleep between each run
040   * @param stopper the stopper
041   * @param region the store to identify the family name
042   */
043  public CompactedHFilesDischarger(final int period, final Stoppable stopper, final Region region) {
044    // Need to add the config classes
045    super("CompactedHFilesCleaner", stopper, period);
046    this.region = region;
047  }
048
049  @Override
050  public void chore() {
051    if (LOG.isTraceEnabled()) {
052      LOG.trace(
053        "Started the compacted hfiles cleaner for the region " + this.region.getRegionInfo());
054    }
055    for (Store store : region.getStores()) {
056      try {
057        store.closeAndArchiveCompactedFiles();
058        if (LOG.isTraceEnabled()) {
059          LOG.trace("Completed archiving the compacted files for the region "
060              + this.region.getRegionInfo() + " under the store " + store.getColumnFamilyName());
061        }
062      } catch (Exception e) {
063        LOG.error(
064          "Exception while trying to close and archive the comapcted store files of the store  "
065              + store.getColumnFamilyName() + " in the region " + this.region.getRegionInfo(),
066          e);
067      }
068    }
069    if (LOG.isTraceEnabled()) {
070      LOG.trace(
071        "Completed the compacted hfiles cleaner for the region " + this.region.getRegionInfo());
072    }
073  }
074}
