067 * A compactor is a compaction algorithm associated a given policy. Base class also contains
068 * reusable parts for implementing compactors (what is common and what isn't is evolving).
069 */
070@InterfaceAudience.Private

094  //TODO: depending on Store is not good but, realistically, all compactors currently do.
095  Compactor(Configuration conf, HStore store) {
096    this.conf = conf;
097    this.store = store;
098    this.compactionKVMax =
099      this.conf.getInt(HConstants.COMPACTION_KV_MAX, HConstants.COMPACTION_KV_MAX_DEFAULT);
100    this.majorCompactionCompression = (store.getColumnFamilyDescriptor() == null) ?
101        Compression.Algorithm.NONE : store.getColumnFamilyDescriptor().getMajorCompactionCompressionType();
102    this.minorCompactionCompression = (store.getColumnFamilyDescriptor() == null) ?
103        Compression.Algorithm.NONE : store.getColumnFamilyDescriptor().getMinorCompactionCompressionType();
104    this.keepSeqIdPeriod = Math.max(this.conf.getInt(HConstants.KEEP_SEQID_PERIOD,
105      HConstants.MIN_KEEP_SEQID_PERIOD), HConstants.MIN_KEEP_SEQID_PERIOD);
106    this.dropCacheMajor = conf.getBoolean(MAJOR_COMPACTION_DROP_CACHE, true);
107    this.dropCacheMinor = conf.getBoolean(MINOR_COMPACTION_DROP_CACHE, true);
108  }

121  /** The sole reason this class exists is that java has no ref/out/pointer parameters. */
122  protected static class FileDetails {
123    /** Maximum key count after compaction (for blooms) */
124    public long maxKeyCount = 0;
125    /** Earliest put timestamp if major compaction */
126    public long earliestPutTs = HConstants.LATEST_TIMESTAMP;
127    /** Latest put timestamp */
128    public long latestPutTs = HConstants.LATEST_TIMESTAMP;
129    /** The last key in the files we're compacting. */
130    public long maxSeqId = 0;
131    /** Latest memstore read point found in any of the involved files */
132    public long maxMVCCReadpoint = 0;
133    /** Max tags length**/
134    public int maxTagsLength = 0;
135    /** Min SeqId to keep during a major compaction **/
136    public long minSeqIdToKeep = 0;
137    /** Total size of the compacted files **/
138    private long totalCompactedFilesSize = 0;
139  }

141  /**
142   * Extracts some details about the files to compact that are commonly needed by compactors.
143   * @param filesToCompact Files.
144   * @param allFiles Whether all files are included for compaction
145   * @parma major If major compaction
146   * @return The result.
147   */
148  private FileDetails getFileDetails(
149      Collection<HStoreFile> filesToCompact, boolean allFiles, boolean major) throws IOException {
150    FileDetails fd = new FileDetails();
151    long oldestHFileTimestampToKeepMVCC = EnvironmentEdgeManager.currentTime() -
152      (1000L * 60 * 60 * 24 * this.keepSeqIdPeriod);
153
154    for (HStoreFile file : filesToCompact) {
155      if(allFiles && (file.getModificationTimestamp() < oldestHFileTimestampToKeepMVCC)) {
156        // when isAllFiles is true, all files are compacted so we can calculate the smallest
157        // MVCC value to keep
158        if(fd.minSeqIdToKeep < file.getMaxMemStoreTS()) {
159          fd.minSeqIdToKeep = file.getMaxMemStoreTS();
160        }
161      }
162      long seqNum = file.getMaxSequenceId();
163      fd.maxSeqId = Math.max(fd.maxSeqId, seqNum);
164      StoreFileReader r = file.getReader();
165      if (r == null) {
166        LOG.warn("Null reader for " + file.getPath());
167        continue;
168      }
169      // NOTE: use getEntries when compacting instead of getFilterEntries, otherwise under-sized
170      // blooms can cause progress to be miscalculated or if the user switches bloom
171      // type (e.g. from ROW to ROWCOL)
172      long keyCount = r.getEntries();
173      fd.maxKeyCount += keyCount;
174      // calculate the latest MVCC readpoint in any of the involved store files
175      Map<byte[], byte[]> fileInfo = r.loadFileInfo();
176
177      // calculate the total size of the compacted files
178      fd.totalCompactedFilesSize += r.length();
179
180      byte[] tmp = null;
181      // Get and set the real MVCCReadpoint for bulk loaded files, which is the
182      // SeqId number.
183      if (r.isBulkLoaded()) {
184        fd.maxMVCCReadpoint = Math.max(fd.maxMVCCReadpoint, r.getSequenceID());
185      }
186      else {
187        tmp = fileInfo.get(HFile.Writer.MAX_MEMSTORE_TS_KEY);
188        if (tmp != null) {
189          fd.maxMVCCReadpoint = Math.max(fd.maxMVCCReadpoint, Bytes.toLong(tmp));
190        }
191      }
192      tmp = fileInfo.get(HFileInfo.MAX_TAGS_LEN);
193      if (tmp != null) {
194        fd.maxTagsLength = Math.max(fd.maxTagsLength, Bytes.toInt(tmp));
195      }
196      // If required, calculate the earliest put timestamp of all involved storefiles.
197      // This is used to remove family delete marker during compaction.
198      long earliestPutTs = 0;
199      if (allFiles) {
200        tmp = fileInfo.get(EARLIEST_PUT_TS);
201        if (tmp == null) {
202          // There's a file with no information, must be an old one
203          // assume we have very old puts
204          fd.earliestPutTs = earliestPutTs = PrivateConstants.OLDEST_TIMESTAMP;
205        } else {
206          earliestPutTs = Bytes.toLong(tmp);
207          fd.earliestPutTs = Math.min(fd.earliestPutTs, earliestPutTs);
208        }
209      }
210      tmp = fileInfo.get(TIMERANGE_KEY);
211      fd.latestPutTs = tmp == null ? HConstants.LATEST_TIMESTAMP: TimeRangeTracker.parseFrom(tmp).getMax();
212      LOG.debug("Compacting {}, keycount={}, bloomtype={}, size={}, "
213              + "encoding={}, compression={}, seqNum={}{}",
214          (file.getPath() == null? null: file.getPath().getName()),
215          keyCount,
216          r.getBloomFilterType().toString(),
217          TraditionalBinaryPrefix.long2String(r.length(), "", 1),
218          r.getHFileReader().getDataBlockEncoding(),
219          major ? majorCompactionCompression : minorCompactionCompression,
220          seqNum,
221          (allFiles? ", earliestPutTs=" + earliestPutTs: ""));
222    }
223    return fd;
224  }

270  protected final StoreFileWriter createTmpWriter(FileDetails fd, boolean shouldDropBehind, boolean major)
271      throws IOException {
272    // When all MVCC readpoints are 0, don't write them.
273    // See HBASE-8166, HBASE-12600, and HBASE-13389.
274    return store.createWriterInTmp(fd.maxKeyCount,
275        major ? majorCompactionCompression : minorCompactionCompression,
276        true, fd.maxMVCCReadpoint > 0,
277        fd.maxTagsLength > 0, shouldDropBehind, fd.totalCompactedFilesSize,
278        HConstants.EMPTY_STRING);
279  }

285      true, fd.maxMVCCReadpoint > 0,
286      fd.maxTagsLength > 0, shouldDropBehind, fd.totalCompactedFilesSize, fileStoragePolicy);

289  private ScanInfo preCompactScannerOpen(CompactionRequestImpl request, ScanType scanType,
290      User user) throws IOException {
291    if (store.getCoprocessorHost() == null) {
292      return store.getScanInfo();
293    }
294    return store.getCoprocessorHost().preCompactScannerOpen(store, scanType, request.getTracker(),

299   * Calls coprocessor, if any, to create scanners - after normal scanner creation.
300   * @param request Compaction request.
301   * @param scanType Scan type.
302   * @param scanner The default scanner created for compaction.
303   * @return Scanner scanner to use (usually the default); null if compaction should not proceed.
304   */
305  private InternalScanner postCompactScannerOpen(CompactionRequestImpl request, ScanType scanType,
306      InternalScanner scanner, User user) throws IOException {
307    if (store.getCoprocessorHost() == null) {
308      return scanner;
309    }
310    return store.getCoprocessorHost().preCompact(store, scanner, scanType, request.getTracker(),
311      request, user);

314  protected final List<Path> compact(final CompactionRequestImpl request,
315      InternalScannerFactory scannerFactory, CellSinkFactory<T> sinkFactory,
316      ThroughputController throughputController, User user) throws IOException {
317    FileDetails fd = getFileDetails(request.getFiles(), request.isAllFiles(), request.isMajor());
318    this.progress = new CompactionProgress(fd.maxKeyCount);
319
320    // Find the smallest read point across all the Scanners.
321    long smallestReadPoint = getSmallestReadPoint();
322
323    T writer = null;
324    boolean dropCache;
325    if (request.isMajor() || request.isAllFiles()) {
326      dropCache = this.dropCacheMajor;
327    } else {
328      dropCache = this.dropCacheMinor;
329    }
330
331    InternalScanner scanner = null;
332    boolean finished = false;
333    List<StoreFileScanner> scanners =
334      createFileScanners(request.getFiles(), smallestReadPoint, dropCache);
335    try {
336      /* Include deletes, unless we are doing a major compaction */
337      ScanType scanType = scannerFactory.getScanType(request);
338      ScanInfo scanInfo = preCompactScannerOpen(request, scanType, user);
339      scanner = postCompactScannerOpen(request, scanType,
340        scannerFactory.createScanner(scanInfo, scanners, scanType, fd, smallestReadPoint), user);
341      boolean cleanSeqId = false;
342      if (fd.minSeqIdToKeep > 0 && !store.getColumnFamilyDescriptor().isNewVersionBehavior()) {
343        // For mvcc-sensitive family, we never set mvcc to 0.
344        smallestReadPoint = Math.min(fd.minSeqIdToKeep, smallestReadPoint);
345        cleanSeqId = true;
346      }
347      writer = sinkFactory.createWriter(scanner, fd, dropCache, request.isMajor());
348      finished = performCompaction(fd, scanner, writer, smallestReadPoint, cleanSeqId,
349        throughputController, request.isAllFiles(), request.getFiles().size());
350      if (!finished) {
351        throw new InterruptedIOException("Aborting compaction of store " + store + " in region "
352            + store.getRegionInfo().getRegionNameAsString() + " because it was interrupted.");
353      }
354    } finally {
355      // createScanner may fail when seeking hfiles encounter Exception, e.g. even only one hfile
356      // reader encounters java.io.IOException: Invalid HFile block magic:
357      // \x00\x00\x00\x00\x00\x00\x00\x00
358      // and then scanner will be null, but scanners for all the hfiles should be closed,
359      // or else we will find leak of ESTABLISHED sockets.
360      if (scanner == null) {
361        for (StoreFileScanner sfs : scanners) {
362          sfs.close();
363        }
364      } else {
365        Closeables.close(scanner, true);
366      }
367      if (!finished && writer != null) {
368        abortWriter(writer);
369      }
370    }
371    assert finished : "We should have exited the method on all error paths";
372    assert writer != null : "Writer should be non-null if no error";
373    return commitWriter(writer, fd, request);
374  }

381  /**
382   * Performs the compaction.
383   * @param fd FileDetails of cell sink writer
384   * @param scanner Where to read from.
385   * @param writer Where to write to.
386   * @param smallestReadPoint Smallest read point.
387   * @param cleanSeqId When true, remove seqId(used to be mvcc) value which is &lt;=
388   *          smallestReadPoint
389   * @param major Is a major compaction.
390   * @param numofFilesToCompact the number of files to compact
391   * @return Whether compaction ended; false if it was interrupted for some reason.
392   */
393  protected boolean performCompaction(FileDetails fd, InternalScanner scanner, CellSink writer,
394      long smallestReadPoint, boolean cleanSeqId, ThroughputController throughputController,
395      boolean major, int numofFilesToCompact) throws IOException {
396    assert writer instanceof ShipperListener;
397    long bytesWrittenProgressForLog = 0;
398    long bytesWrittenProgressForShippedCall = 0;
399    // Since scanner.next() can return 'false' but still be delivering data,
400    // we have to use a do/while loop.
401    List<Cell> cells = new ArrayList<>();
402    long currentTime = EnvironmentEdgeManager.currentTime();
403    long lastMillis = 0;
404    if (LOG.isDebugEnabled()) {
405      lastMillis = currentTime;
406    }
407    CloseChecker closeChecker = new CloseChecker(conf, currentTime);
408    String compactionName = ThroughputControlUtil.getNameForThrottling(store, "compaction");
409    long now = 0;
410    boolean hasMore;
411    ScannerContext scannerContext =
412          ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
413
414    throughputController.start(compactionName);
415    KeyValueScanner kvs = (scanner instanceof KeyValueScanner) ? (KeyValueScanner) scanner : null;
416    long shippedCallSizeLimit =
417          (long) numofFilesToCompact * this.store.getColumnFamilyDescriptor().getBlocksize();
418    try {
419      do {
420        hasMore = scanner.next(cells, scannerContext);
421        currentTime = EnvironmentEdgeManager.currentTime();
422        if (LOG.isDebugEnabled()) {
423          now = currentTime;
424        }
425        if (closeChecker.isTimeLimit(store, currentTime)) {
426          progress.cancel();
427          return false;
428        }
429        // output to writer:
430        Cell lastCleanCell = null;
431        long lastCleanCellSeqId = 0;
432        for (Cell c : cells) {
433          if (cleanSeqId && c.getSequenceId() <= smallestReadPoint) {
434            lastCleanCell = c;
435            lastCleanCellSeqId = c.getSequenceId();
436            PrivateCellUtil.setSequenceId(c, 0);
437          } else {
438            lastCleanCell = null;
439            lastCleanCellSeqId = 0;
440          }
441          writer.append(c);
442          int len = c.getSerializedSize();
443          ++progress.currentCompactedKVs;
444          progress.totalCompactedSize += len;
445          bytesWrittenProgressForShippedCall += len;
446          if (LOG.isDebugEnabled()) {
447            bytesWrittenProgressForLog += len;
448          }
449          throughputController.control(compactionName, len);
450          if (closeChecker.isSizeLimit(store, len)) {
451            progress.cancel();
452            return false;
453          }
454        }
455        if (kvs != null && bytesWrittenProgressForShippedCall > shippedCallSizeLimit) {
456          if (lastCleanCell != null) {
457            // HBASE-16931, set back sequence id to avoid affecting scan order unexpectedly.
458            // ShipperListener will do a clone of the last cells it refer, so need to set back
459            // sequence id before ShipperListener.beforeShipped
460            PrivateCellUtil.setSequenceId(lastCleanCell, lastCleanCellSeqId);
461          }
462          // Clone the cells that are in the writer so that they are freed of references,
463          // if they are holding any.
464          ((ShipperListener) writer).beforeShipped();
465          // The SHARED block references, being read for compaction, will be kept in prevBlocks
466          // list(See HFileScannerImpl#prevBlocks). In case of scan flow, after each set of cells
467          // being returned to client, we will call shipped() which can clear this list. Here by
468          // we are doing the similar thing. In between the compaction (after every N cells
469          // written with collective size of 'shippedCallSizeLimit') we will call shipped which
470          // may clear prevBlocks list.
471          kvs.shipped();
472          bytesWrittenProgressForShippedCall = 0;
473        }
474        if (lastCleanCell != null) {
475          // HBASE-16931, set back sequence id to avoid affecting scan order unexpectedly
476          PrivateCellUtil.setSequenceId(lastCleanCell, lastCleanCellSeqId);
477        }
478        // Log the progress of long running compactions every minute if
479        // logging at DEBUG level
480        if (LOG.isDebugEnabled()) {
481          if ((now - lastMillis) >= COMPACTION_PROGRESS_LOG_INTERVAL) {
482            String rate = String.format("%.2f",
483              (bytesWrittenProgressForLog / 1024.0) / ((now - lastMillis) / 1000.0));
484            LOG.debug("Compaction progress: {} {}, rate={} KB/sec, throughputController is {}",
485              compactionName, progress, rate, throughputController);
486            lastMillis = now;
487            bytesWrittenProgressForLog = 0;
488          }
489        }
490        cells.clear();
491      } while (hasMore);
492    } catch (InterruptedException e) {
493      progress.cancel();
494      throw new InterruptedIOException(
495            "Interrupted while control throughput of compacting " + compactionName);
496    } finally {
497      // Clone last cell in the final because writer will append last cell when committing. If
498      // don't clone here and once the scanner get closed, then the memory of last cell will be
499      // released. (HBASE-22582)
500      ((ShipperListener) writer).beforeShipped();
501      throughputController.finish(compactionName);
502    }
503    progress.complete();
504    return true;
505  }
