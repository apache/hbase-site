018package org.apache.hadoop.hbase.regionserver.compactions;
019
020import java.io.IOException;
021import java.util.List;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.Path;
027import org.apache.hadoop.hbase.regionserver.HStore;
028import org.apache.hadoop.hbase.regionserver.InternalScanner;
029import org.apache.hadoop.hbase.regionserver.ScanType;
030import org.apache.hadoop.hbase.regionserver.StoreFileScanner;
031import org.apache.hadoop.hbase.regionserver.StripeMultiFileWriter;
032import org.apache.hadoop.hbase.regionserver.throttle.ThroughputController;
033import org.apache.hadoop.hbase.security.User;
034import org.apache.hadoop.hbase.util.Bytes;
035import org.apache.yetus.audience.InterfaceAudience;
036
037/**
038 * This is the placeholder for stripe compactor. The implementation, as well as the proper javadoc,
039 * will be added in HBASE-7967.
040 */
041@InterfaceAudience.Private
042public class StripeCompactor extends AbstractMultiOutputCompactor<StripeMultiFileWriter> {
043  private static final Log LOG = LogFactory.getLog(StripeCompactor.class);
044
045  public StripeCompactor(Configuration conf, HStore store) {
046    super(conf, store);
047  }
048
049  private final class StripeInternalScannerFactory implements InternalScannerFactory {
050
051    private final byte[] majorRangeFromRow;
052
053    private final byte[] majorRangeToRow;
054
055    public StripeInternalScannerFactory(byte[] majorRangeFromRow, byte[] majorRangeToRow) {
056      this.majorRangeFromRow = majorRangeFromRow;
057      this.majorRangeToRow = majorRangeToRow;
058    }
059
060    @Override
061    public ScanType getScanType(CompactionRequest request) {
062      // If majorRangeFromRow and majorRangeToRow are not null, then we will not use the return
063      // value to create InternalScanner. See the createScanner method below. The return value is
064      // also used when calling coprocessor hooks.
065      return ScanType.COMPACT_RETAIN_DELETES;
066    }
067
068    @Override
069    public InternalScanner createScanner(List<StoreFileScanner> scanners, ScanType scanType,
070        FileDetails fd, long smallestReadPoint) throws IOException {
071      return (majorRangeFromRow == null)
072          ? StripeCompactor.this.createScanner(store, scanners, scanType, smallestReadPoint,
073            fd.earliestPutTs)
074          : StripeCompactor.this.createScanner(store, scanners, smallestReadPoint, fd.earliestPutTs,
075            majorRangeFromRow, majorRangeToRow);
076    }
077  }
078
079  public List<Path> compact(CompactionRequest request, final List<byte[]> targetBoundaries,
080      final byte[] majorRangeFromRow, final byte[] majorRangeToRow,
081      ThroughputController throughputController, User user) throws IOException {
082    if (LOG.isDebugEnabled()) {
083      StringBuilder sb = new StringBuilder();
084      sb.append("Executing compaction with " + targetBoundaries.size() + " boundaries:");
085      for (byte[] tb : targetBoundaries) {
086        sb.append(" [").append(Bytes.toString(tb)).append("]");
087      }
088      LOG.debug(sb.toString());
089    }
090    return compact(request, new StripeInternalScannerFactory(majorRangeFromRow, majorRangeToRow),
091      new CellSinkFactory<StripeMultiFileWriter>() {
092
093        @Override
094        public StripeMultiFileWriter createWriter(InternalScanner scanner, FileDetails fd,
095            boolean shouldDropBehind) throws IOException {
096          StripeMultiFileWriter writer = new StripeMultiFileWriter.BoundaryMultiWriter(
097              store.getComparator(), targetBoundaries, majorRangeFromRow, majorRangeToRow);
098          initMultiWriter(writer, scanner, fd, shouldDropBehind);
099          return writer;
100        }
101      }, throughputController, user);
102  }
103
104  public List<Path> compact(CompactionRequest request, final int targetCount, final long targetSize,
105      final byte[] left, final byte[] right, byte[] majorRangeFromRow, byte[] majorRangeToRow,
106      ThroughputController throughputController, User user) throws IOException {
107    if (LOG.isDebugEnabled()) {
108      LOG.debug(
109        "Executing compaction with " + targetSize + " target file size, no more than " + targetCount
110            + " files, in [" + Bytes.toString(left) + "] [" + Bytes.toString(right) + "] range");
111    }
112    return compact(request, new StripeInternalScannerFactory(majorRangeFromRow, majorRangeToRow),
113      new CellSinkFactory<StripeMultiFileWriter>() {
114
115        @Override
116        public StripeMultiFileWriter createWriter(InternalScanner scanner, FileDetails fd,
117            boolean shouldDropBehind) throws IOException {
118          StripeMultiFileWriter writer = new StripeMultiFileWriter.SizeMultiWriter(
119              store.getComparator(), targetCount, targetSize, left, right);
120          initMultiWriter(writer, scanner, fd, shouldDropBehind);
121          return writer;
122        }
123      }, throughputController, user);
124  }
125
126  @Override
127  protected List<Path> commitWriter(StripeMultiFileWriter writer, FileDetails fd,
128      CompactionRequest request) throws IOException {
129    List<Path> newFiles = writer.commitWriters(fd.maxSeqId, request.isMajor());
130    assert !newFiles.isEmpty() : "Should have produced an empty file to preserve metadata.";
131    return newFiles;
132  }
133}
