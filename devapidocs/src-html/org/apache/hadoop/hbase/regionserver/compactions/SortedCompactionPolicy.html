033/**
034 * An abstract compaction policy that select files on seq id order.
035 */
036@InterfaceAudience.Private
037public abstract class SortedCompactionPolicy extends CompactionPolicy {
038
039  private static final Log LOG = LogFactory.getLog(SortedCompactionPolicy.class);
040
041  public SortedCompactionPolicy(Configuration conf, StoreConfigInformation storeConfigInfo) {
042    super(conf, storeConfigInfo);
043  }
044
045  public List<StoreFile> preSelectCompactionForCoprocessor(final Collection<StoreFile> candidates,
046      final List<StoreFile> filesCompacting) {
047    return getCurrentEligibleFiles(new ArrayList<>(candidates), filesCompacting);
048  }
049
050  /**
051   * @param candidateFiles candidate files, ordered from oldest to newest by seqId. We rely on
052   *   DefaultStoreFileManager to sort the files by seqId to guarantee contiguous compaction based
053   *   on seqId for data consistency.
054   * @return subset copy of candidate list that meets compaction criteria
055   */
056  public CompactionRequest selectCompaction(Collection<StoreFile> candidateFiles,
057      final List<StoreFile> filesCompacting, final boolean isUserCompaction,
058      final boolean mayUseOffPeak, final boolean forceMajor) throws IOException {
059    // Preliminary compaction subject to filters
060    ArrayList<StoreFile> candidateSelection = new ArrayList<>(candidateFiles);
061    // Stuck and not compacting enough (estimate). It is not guaranteed that we will be
062    // able to compact more if stuck and compacting, because ratio policy excludes some
063    // non-compacting files from consideration during compaction (see getCurrentEligibleFiles).
064    int futureFiles = filesCompacting.isEmpty() ? 0 : 1;
065    boolean mayBeStuck = (candidateFiles.size() - filesCompacting.size() + futureFiles)
066        >= storeConfigInfo.getBlockingFileCount();
067
068    candidateSelection = getCurrentEligibleFiles(candidateSelection, filesCompacting);
069    LOG.debug("Selecting compaction from " + candidateFiles.size() + " store files, " +
070        filesCompacting.size() + " compacting, " + candidateSelection.size() +
071        " eligible, " + storeConfigInfo.getBlockingFileCount() + " blocking");
072
073    // If we can't have all files, we cannot do major anyway
074    boolean isAllFiles = candidateFiles.size() == candidateSelection.size();
075    if (!(forceMajor && isAllFiles)) {
076      candidateSelection = skipLargeFiles(candidateSelection, mayUseOffPeak);
077      isAllFiles = candidateFiles.size() == candidateSelection.size();
078    }
079
080    // Try a major compaction if this is a user-requested major compaction,
081    // or if we do not have too many files to compact and this was requested as a major compaction
082    boolean isTryingMajor = (forceMajor && isAllFiles && isUserCompaction)
083        || (((forceMajor && isAllFiles) || shouldPerformMajorCompaction(candidateSelection))
084          && (candidateSelection.size() < comConf.getMaxFilesToCompact()));
085    // Or, if there are any references among the candidates.
086    boolean isAfterSplit = StoreUtils.hasReferences(candidateSelection);
087
088    CompactionRequest result = createCompactionRequest(candidateSelection,
089      isTryingMajor || isAfterSplit, mayUseOffPeak, mayBeStuck);
090
091    ArrayList<StoreFile> filesToCompact = Lists.newArrayList(result.getFiles());
092    removeExcessFiles(filesToCompact, isUserCompaction, isTryingMajor);
093    result.updateFiles(filesToCompact);
094
095    isAllFiles = (candidateFiles.size() == filesToCompact.size());
096    result.setOffPeak(!filesToCompact.isEmpty() && !isAllFiles && mayUseOffPeak);
097    result.setIsMajor(isTryingMajor && isAllFiles, isAllFiles);
098
099    return result;
100  }
101
102  protected abstract CompactionRequest createCompactionRequest(ArrayList<StoreFile>
103    candidateSelection, boolean tryingMajor, boolean mayUseOffPeak, boolean mayBeStuck)
104    throws IOException;
105
106  /*
107   * @param filesToCompact Files to compact. Can be null.
108   * @return True if we should run a major compaction.
109   */
110  public abstract boolean shouldPerformMajorCompaction(final Collection<StoreFile> filesToCompact)
111    throws IOException;
112
113  /**
114   * Used calculation jitter
115   */
116  private final Random random = new Random();
117
118  /**
119   * @param filesToCompact
120   * @return When to run next major compaction
121   */
122  public long getNextMajorCompactTime(Collection<StoreFile> filesToCompact) {
123    // default = 24hrs
124    long period = comConf.getMajorCompactionPeriod();
125    if (period <= 0) {
126      return period;
127    }
128    // default = 20% = +/- 4.8 hrs
129    double jitterPct = comConf.getMajorCompactionJitter();
130    if (jitterPct <= 0) {
131      return period;
132    }
133    // deterministic jitter avoids a major compaction storm on restart
134    OptionalInt seed = StoreUtils.getDeterministicRandomSeed(filesToCompact);
135    if (seed.isPresent()) {
136      // Synchronized to ensure one user of random instance at a time.
137      double rnd;
138      synchronized (this) {
139        this.random.setSeed(seed.getAsInt());
140        rnd = this.random.nextDouble();
141      }
142      long jitter = Math.round(period * jitterPct);
143      return period + jitter - Math.round(2L * jitter * rnd);
144    } else {
145      return 0L;
146    }
147  }
148
149  /**
150   * @param compactionSize Total size of some compaction
151   * @return whether this should be a large or small compaction
152   */
153  public boolean throttleCompaction(long compactionSize) {
154    return compactionSize > comConf.getThrottlePoint();
155  }
156
157  public abstract boolean needsCompaction(final Collection<StoreFile> storeFiles,
158    final List<StoreFile> filesCompacting);
159
160  protected ArrayList<StoreFile> getCurrentEligibleFiles(ArrayList<StoreFile> candidateFiles,
161      final List<StoreFile> filesCompacting) {
162    // candidates = all storefiles not already in compaction queue
163    if (!filesCompacting.isEmpty()) {
164      // exclude all files older than the newest file we're currently
165      // compacting. this allows us to preserve contiguity (HBASE-2856)
166      StoreFile last = filesCompacting.get(filesCompacting.size() - 1);
167      int idx = candidateFiles.indexOf(last);
168      Preconditions.checkArgument(idx != -1);
169      candidateFiles.subList(0, idx + 1).clear();
170    }
171    return candidateFiles;
172  }
173
174  /**
175   * @param candidates pre-filtrate
176   * @return filtered subset exclude all files above maxCompactSize
177   *   Also save all references. We MUST compact them
178   */
179  protected ArrayList<StoreFile> skipLargeFiles(ArrayList<StoreFile> candidates,
180    boolean mayUseOffpeak) {
181    int pos = 0;
182    while (pos < candidates.size() && !candidates.get(pos).isReference()
183      && (candidates.get(pos).getReader().length() > comConf.getMaxCompactSize(mayUseOffpeak))) {
184      ++pos;
185    }
186    if (pos > 0) {
187      LOG.debug("Some files are too large. Excluding " + pos
188          + " files from compaction candidates");
189      candidates.subList(0, pos).clear();
190    }
191    return candidates;
192  }
193
194  /**
195   * @param candidates pre-filtrate
196   * @return filtered subset exclude all bulk load files if configured
197   */
198  protected ArrayList<StoreFile> filterBulk(ArrayList<StoreFile> candidates) {
199    candidates.removeAll(Collections2.filter(candidates, new Predicate<StoreFile>() {
200      @Override
201      public boolean apply(StoreFile input) {
202        return input.excludeFromMinorCompaction();
203      }
204    }));
205    return candidates;
206  }
207
208  /**
209   * @param candidates pre-filtrate
210   */
211  protected void removeExcessFiles(ArrayList<StoreFile> candidates,
212      boolean isUserCompaction, boolean isMajorCompaction) {
213    int excess = candidates.size() - comConf.getMaxFilesToCompact();
214    if (excess > 0) {
215      if (isMajorCompaction && isUserCompaction) {
216        LOG.debug("Warning, compacting more than " + comConf.getMaxFilesToCompact()
217            + " files because of a user-requested major compaction");
218      } else {
219        LOG.debug("Too many admissible files. Excluding " + excess
220            + " files from compaction candidates");
221        candidates.subList(comConf.getMaxFilesToCompact(), candidates.size()).clear();
222      }
223    }
224  }
225
226  /**
227   * @param candidates pre-filtrate
228   * @return filtered subset forget the compactionSelection if we don't have enough files
229   */
230  protected ArrayList<StoreFile> checkMinFilesCriteria(ArrayList<StoreFile> candidates,
231    int minFiles) {
232    if (candidates.size() < minFiles) {
233      if (LOG.isDebugEnabled()) {
234        LOG.debug("Not compacting files because we only have " + candidates.size()
235            + " files ready for compaction. Need " + minFiles + " to initiate.");
236      }
237      candidates.clear();
238    }
239    return candidates;
240  }
