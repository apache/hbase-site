018package org.apache.hadoop.hbase.regionserver.compactions;
019
020import java.io.IOException;
021import java.util.Collection;
022import java.util.List;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.fs.Path;
028import org.apache.hadoop.hbase.classification.InterfaceAudience;
029import org.apache.hadoop.hbase.regionserver.InternalScanner;
030import org.apache.hadoop.hbase.regionserver.Store;
031import org.apache.hadoop.hbase.regionserver.StoreFile;
032import org.apache.hadoop.hbase.regionserver.StoreFileWriter;
033import org.apache.hadoop.hbase.regionserver.throttle.NoLimitThroughputController;
034import org.apache.hadoop.hbase.regionserver.throttle.ThroughputController;
035import org.apache.hadoop.hbase.security.User;
036
037import org.apache.hadoop.hbase.shaded.com.google.common.collect.Lists;
038
039/**
040 * Compact passed set of files. Create an instance and then call
041 * {@link #compact(CompactionRequest, ThroughputController, User)}
042 */
043@InterfaceAudience.Private
044public class DefaultCompactor extends Compactor<StoreFileWriter> {
045  private static final Log LOG = LogFactory.getLog(DefaultCompactor.class);
046
047  public DefaultCompactor(final Configuration conf, final Store store) {
048    super(conf, store);
049  }
050
051  private final CellSinkFactory<StoreFileWriter> writerFactory =
052      new CellSinkFactory<StoreFileWriter>() {
053        @Override
054        public StoreFileWriter createWriter(InternalScanner scanner,
055            org.apache.hadoop.hbase.regionserver.compactions.Compactor.FileDetails fd,
056            boolean shouldDropBehind) throws IOException {
057          return createTmpWriter(fd, shouldDropBehind);
058        }
059      };
060
061  /**
062   * Do a minor/major compaction on an explicit set of storefiles from a Store.
063   */
064  public List<Path> compact(final CompactionRequest request,
065      ThroughputController throughputController, User user) throws IOException {
066    return compact(request, defaultScannerFactory, writerFactory, throughputController, user);
067  }
068
069  /**
070   * Compact a list of files for testing. Creates a fake {@link CompactionRequest} to pass to
071   * {@link #compact(CompactionRequest, ThroughputController, User)};
072   * @param filesToCompact the files to compact. These are used as the compactionSelection for the
073   *          generated {@link CompactionRequest}.
074   * @param isMajor true to major compact (prune all deletes, max versions, etc)
075   * @return Product of compaction or an empty list if all cells expired or deleted and nothing \
076   *         made it through the compaction.
077   * @throws IOException
078   */
079  public List<Path> compactForTesting(final Collection<StoreFile> filesToCompact, boolean isMajor)
080      throws IOException {
081    CompactionRequest cr = new CompactionRequest(filesToCompact);
082    cr.setIsMajor(isMajor, isMajor);
083    return compact(cr, NoLimitThroughputController.INSTANCE, null);
084  }
085
086  @Override
087  protected List<Path> commitWriter(StoreFileWriter writer, FileDetails fd,
088      CompactionRequest request) throws IOException {
089    List<Path> newFiles = Lists.newArrayList(writer.getPath());
090    writer.appendMetadata(fd.maxSeqId, request.isAllFiles());
091    writer.close();
092    return newFiles;
093  }
094
095  @Override
096  protected void abortWriter(StoreFileWriter writer) throws IOException {
097    Path leftoverFile = writer.getPath();
098    try {
099      writer.close();
100    } catch (IOException e) {
101      LOG.warn("Failed to close the writer after an unfinished compaction.", e);
102    }
103    try {
104      store.getFileSystem().delete(leftoverFile, false);
105    } catch (IOException e) {
106      LOG.warn(
107        "Failed to delete the leftover file " + leftoverFile + " after an unfinished compaction.",
108        e);
109    }
110  }
111}
