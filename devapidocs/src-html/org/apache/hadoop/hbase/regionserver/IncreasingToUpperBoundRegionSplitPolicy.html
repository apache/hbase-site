018package org.apache.hadoop.hbase.regionserver;
019
020import java.io.IOException;
021import java.util.List;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.HConstants;
027import org.apache.hadoop.hbase.HTableDescriptor;
028import org.apache.hadoop.hbase.TableName;
029import org.apache.hadoop.hbase.classification.InterfaceAudience;
030
031/**
032 * Split size is the number of regions that are on this server that all are
033 * of the same table, cubed, times 2x the region flush size OR the maximum
034 * region split size, whichever is smaller.
035 * <p>
036 * For example, if the flush size is 128MB, then after two flushes (256MB) we
037 * will split which will make two regions that will split when their size is
038 * {@code 2^3 * 128MB*2 = 2048MB}.
039 * <p>
040 * If one of these regions splits, then there are three regions and now the
041 * split size is {@code 3^3 * 128MB*2 = 6912MB}, and so on until we reach the configured
042 * maximum file size and then from there on out, we'll use that.
043 */
044@InterfaceAudience.Private
045public class IncreasingToUpperBoundRegionSplitPolicy extends ConstantSizeRegionSplitPolicy {
046
047  private static final Log LOG = LogFactory.getLog(IncreasingToUpperBoundRegionSplitPolicy.class);
048  protected long initialSize;
049
050  @Override
051  protected void configureForRegion(HRegion region) {
052    super.configureForRegion(region);
053    Configuration conf = getConf();
054    initialSize = conf.getLong("hbase.increasing.policy.initial.size", -1);
055    if (initialSize > 0) {
056      return;
057    }
058    HTableDescriptor desc = region.getTableDesc();
059    if (desc != null) {
060      initialSize = 2 * desc.getMemStoreFlushSize();
061    }
062    if (initialSize <= 0) {
063      initialSize = 2 * conf.getLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE,
064                                     HTableDescriptor.DEFAULT_MEMSTORE_FLUSH_SIZE);
065    }
066  }
067
068  @Override
069  protected boolean shouldSplit() {
070    boolean force = region.shouldForceSplit();
071    boolean foundABigStore = false;
072    // Get count of regions that have the same common table as this.region
073    int tableRegionsCount = getCountOfCommonTableRegions();
074    // Get size to check
075    long sizeToCheck = getSizeToCheck(tableRegionsCount);
076
077    for (Store store : region.getStores()) {
078      // If any of the stores is unable to split (eg they contain reference files)
079      // then don't split
080      if (!store.canSplit()) {
081        return false;
082      }
083
084      // Mark if any store is big enough
085      long size = store.getSize();
086      if (size > sizeToCheck) {
087        LOG.debug("ShouldSplit because " + store.getColumnFamilyName() + " size=" + size
088                  + ", sizeToCheck=" + sizeToCheck + ", regionsWithCommonTable="
089                  + tableRegionsCount);
090        foundABigStore = true;
091      }
092    }
093
094    return foundABigStore | force;
095  }
096
097  /**
098   * @return Count of regions on this server that share the table this.region
099   * belongs to
100   */
101  private int getCountOfCommonTableRegions() {
102    RegionServerServices rss = region.getRegionServerServices();
103    // Can be null in tests
104    if (rss == null) {
105      return 0;
106    }
107    TableName tablename = region.getTableDesc().getTableName();
108    int tableRegionsCount = 0;
109    try {
110      List<Region> hri = rss.getOnlineRegions(tablename);
111      tableRegionsCount = hri == null || hri.isEmpty() ? 0 : hri.size();
112    } catch (IOException e) {
113      LOG.debug("Failed getOnlineRegions " + tablename, e);
114    }
115    return tableRegionsCount;
116  }
117
118  /**
119   * @return Region max size or {@code count of regions cubed * 2 * flushsize},
120   * which ever is smaller; guard against there being zero regions on this server.
121   */
122  protected long getSizeToCheck(final int tableRegionsCount) {
123    // safety check for 100 to avoid numerical overflow in extreme cases
124    return tableRegionsCount == 0 || tableRegionsCount > 100
125               ? getDesiredMaxFileSize()
126               : Math.min(getDesiredMaxFileSize(),
127                          initialSize * tableRegionsCount * tableRegionsCount * tableRegionsCount);
128  }
129}
