255   * Constructor
256   * @param region the region
257   * @param rsServices interface to available region server functionality
258   * @param conf the configuration
259   */
260  public RegionCoprocessorHost(final HRegion region,
261      final RegionServerServices rsServices, final Configuration conf) {
262    super(rsServices);
263    this.conf = conf;
264    this.rsServices = rsServices;
265    this.region = region;
266    this.pathPrefix = Integer.toString(this.region.getRegionInfo().hashCode());
267
268    // load system default cp's from configuration.
269    loadSystemCoprocessors(conf, REGION_COPROCESSOR_CONF_KEY);
270
271    // load system default cp's for user tables from configuration.
272    if (!region.getRegionInfo().getTable().isSystemTable()) {
273      loadSystemCoprocessors(conf, USER_REGION_COPROCESSOR_CONF_KEY);
274    }
275
276    // load Coprocessor From HDFS
277    loadTableCoprocessors(conf);
278
279    // now check whether any coprocessor implements postScannerFilterRow
280    boolean hasCustomPostScannerFilterRow = false;
281    out: for (RegionCoprocessorEnvironment env: coprocEnvironments) {
282      if (env.getInstance() instanceof RegionObserver) {
283        Class<?> clazz = env.getInstance().getClass();
284        for(;;) {
285          if (clazz == null) {
286            // we must have directly implemented RegionObserver
287            hasCustomPostScannerFilterRow = true;
288            break out;
289          }
290          try {
291            clazz.getDeclaredMethod("postScannerFilterRow", ObserverContext.class,
292              InternalScanner.class, Cell.class, boolean.class);
293            // this coprocessor has a custom version of postScannerFilterRow
294            hasCustomPostScannerFilterRow = true;
295            break out;
296          } catch (NoSuchMethodException ignore) {
297          }
298          // the deprecated signature still exists
299          try {
300            clazz.getDeclaredMethod("postScannerFilterRow", ObserverContext.class,
301              InternalScanner.class, byte[].class, int.class, short.class, boolean.class);
302            // this coprocessor has a custom version of postScannerFilterRow
303            hasCustomPostScannerFilterRow = true;
304            break out;
305          } catch (NoSuchMethodException ignore) {
306          }
307          clazz = clazz.getSuperclass();
308        }
309      }
310    }
311    this.hasCustomPostScannerFilterRow = hasCustomPostScannerFilterRow;
312  }

314  static List<TableCoprocessorAttribute> getTableCoprocessorAttrsFromSchema(Configuration conf,
315      TableDescriptor htd) {
316    List<TableCoprocessorAttribute> result = Lists.newArrayList();
317    for (Map.Entry<Bytes, Bytes> e: htd.getValues().entrySet()) {
318      String key = Bytes.toString(e.getKey().get()).trim();
319      if (HConstants.CP_HTD_ATTR_KEY_PATTERN.matcher(key).matches()) {
320        String spec = Bytes.toString(e.getValue().get()).trim();
321        // found one
322        try {
323          Matcher matcher = HConstants.CP_HTD_ATTR_VALUE_PATTERN.matcher(spec);
324          if (matcher.matches()) {
325            // jar file path can be empty if the cp class can be loaded
326            // from class loader.
327            Path path = matcher.group(1).trim().isEmpty() ?
328                null : new Path(matcher.group(1).trim());
329            String className = matcher.group(2).trim();
330            if (className.isEmpty()) {
331              LOG.error("Malformed table coprocessor specification: key=" +
332                key + ", spec: " + spec);
333              continue;
334            }
335            String priorityStr = matcher.group(3).trim();
336            int priority = priorityStr.isEmpty() ?
337                Coprocessor.PRIORITY_USER : Integer.parseInt(priorityStr);
338            String cfgSpec = null;
339            try {
340              cfgSpec = matcher.group(4);
341            } catch (IndexOutOfBoundsException ex) {
342              // ignore
343            }
344            Configuration ourConf;
345            if (cfgSpec != null && !cfgSpec.trim().equals("|")) {
346              cfgSpec = cfgSpec.substring(cfgSpec.indexOf('|') + 1);
347              // do an explicit deep copy of the passed configuration
348              ourConf = new Configuration(false);
349              HBaseConfiguration.merge(ourConf, conf);
350              Matcher m = HConstants.CP_HTD_ATTR_VALUE_PARAM_PATTERN.matcher(cfgSpec);
351              while (m.find()) {
352                ourConf.set(m.group(1), m.group(2));
353              }
354            } else {
355              ourConf = conf;
356            }
357            result.add(new TableCoprocessorAttribute(path, className, priority, ourConf));
358          } else {
359            LOG.error("Malformed table coprocessor specification: key=" + key +
360              ", spec: " + spec);
361          }
362        } catch (Exception ioe) {
363          LOG.error("Malformed table coprocessor specification: key=" + key +
364            ", spec: " + spec);
365        }
366      }
367    }
368    return result;
369  }

371  /**
372   * Sanity check the table coprocessor attributes of the supplied schema. Will
373   * throw an exception if there is a problem.
374   * @param conf
375   * @param htd
376   * @throws IOException
377   */
378  public static void testTableCoprocessorAttrs(final Configuration conf,
379      final TableDescriptor htd) throws IOException {
380    String pathPrefix = UUID.randomUUID().toString();
381    for (TableCoprocessorAttribute attr: getTableCoprocessorAttrsFromSchema(conf, htd)) {
382      if (attr.getPriority() < 0) {
383        throw new IOException("Priority for coprocessor " + attr.getClassName() +
384          " cannot be less than 0");
385      }
386      ClassLoader old = Thread.currentThread().getContextClassLoader();
387      try {
388        ClassLoader cl;
389        if (attr.getPath() != null) {
390          cl = CoprocessorClassLoader.getClassLoader(attr.getPath(),
391            CoprocessorHost.class.getClassLoader(), pathPrefix, conf);
392        } else {
393          cl = CoprocessorHost.class.getClassLoader();
394        }
395        Thread.currentThread().setContextClassLoader(cl);
396        cl.loadClass(attr.getClassName());
397      } catch (ClassNotFoundException e) {
398        throw new IOException("Class " + attr.getClassName() + " cannot be loaded", e);
399      } finally {
400        Thread.currentThread().setContextClassLoader(old);
401      }
402    }
403  }

405  void loadTableCoprocessors(final Configuration conf) {
406    boolean coprocessorsEnabled = conf.getBoolean(COPROCESSORS_ENABLED_CONF_KEY,
407      DEFAULT_COPROCESSORS_ENABLED);
408    boolean tableCoprocessorsEnabled = conf.getBoolean(USER_COPROCESSORS_ENABLED_CONF_KEY,
409      DEFAULT_USER_COPROCESSORS_ENABLED);
410    if (!(coprocessorsEnabled && tableCoprocessorsEnabled)) {
411      return;
412    }
413
414    // scan the table attributes for coprocessor load specifications
415    // initialize the coprocessors
416    List<RegionCoprocessorEnvironment> configured = new ArrayList<>();
417    for (TableCoprocessorAttribute attr: getTableCoprocessorAttrsFromSchema(conf,
418        region.getTableDescriptor())) {
419      // Load encompasses classloading and coprocessor initialization
420      try {
421        RegionCoprocessorEnvironment env = load(attr.getPath(), attr.getClassName(),
422            attr.getPriority(), attr.getConf());
423        if (env == null) {
424          continue;
425        }
426        configured.add(env);
427        LOG.info("Loaded coprocessor " + attr.getClassName() + " from HTD of " +
428            region.getTableDescriptor().getTableName().getNameAsString() + " successfully.");
429      } catch (Throwable t) {
430        // Coprocessor failed to load, do we abort on error?
431        if (conf.getBoolean(ABORT_ON_ERROR_KEY, DEFAULT_ABORT_ON_ERROR)) {
432          abortServer(attr.getClassName(), t);
433        } else {
434          LOG.error("Failed to load coprocessor " + attr.getClassName(), t);
435        }
436      }
437    }
438    // add together to coprocessor set for COW efficiency
439    coprocEnvironments.addAll(configured);
440  }

442  @Override
443  public RegionEnvironment createEnvironment(RegionCoprocessor instance, int priority, int seq,
444      Configuration conf) {
445    // If coprocessor exposes any services, register them.
446    for (Service service : instance.getServices()) {
447      region.registerService(service);
448    }
449    ConcurrentMap<String, Object> classData;
450    // make sure only one thread can add maps
451    synchronized (SHARED_DATA_MAP) {
452      // as long as at least one RegionEnvironment holds on to its classData it will
453      // remain in this map
454      classData =
455          SHARED_DATA_MAP.computeIfAbsent(instance.getClass().getName(),
456              k -> new ConcurrentHashMap<>());
457    }
458    // If a CoreCoprocessor, return a 'richer' environment, one laden with RegionServerServices.
459    return instance.getClass().isAnnotationPresent(CoreCoprocessor.class)?
460        new RegionEnvironmentForCoreCoprocessors(instance, priority, seq, conf, region,
461            rsServices, classData):
462        new RegionEnvironment(instance, priority, seq, conf, region, rsServices, classData);
463  }

465  @Override
466  public RegionCoprocessor checkAndGetInstance(Class<?> implClass)
467      throws InstantiationException, IllegalAccessException {
468    if (RegionCoprocessor.class.isAssignableFrom(implClass)) {
469      return (RegionCoprocessor)implClass.newInstance();
470    } else if (CoprocessorService.class.isAssignableFrom(implClass)) {
471      // For backward compatibility with old CoprocessorService impl which don't extend
472      // RegionCoprocessor.
473      return new CoprocessorServiceBackwardCompatiblity.RegionCoprocessorService(
474          (CoprocessorService)implClass.newInstance());
475    } else {
476      LOG.error(implClass.getName() + " is not of type RegionCoprocessor. Check the "
477          + "configuration " + CoprocessorHost.REGION_COPROCESSOR_CONF_KEY);
478      return null;
479    }
