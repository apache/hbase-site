018package org.apache.hadoop.hbase.regionserver;
019
020import java.io.IOException;
021import java.util.List;
022import java.util.Optional;
023
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.conf.Configured;
026import org.apache.hadoop.hbase.HBaseInterfaceAudience;
027import org.apache.hadoop.hbase.HConstants;
028import org.apache.hadoop.hbase.client.TableDescriptor;
029import org.apache.hadoop.util.ReflectionUtils;
030import org.apache.yetus.audience.InterfaceAudience;
031
032import org.apache.hbase.thirdparty.com.google.common.base.Preconditions;
033
034
035/**
036 * A split policy determines when a region should be split.
037 * @see SteppingSplitPolicy Default split policy since 2.0.0
038 * @see IncreasingToUpperBoundRegionSplitPolicy Default split policy since
039 *      0.94.0
040 * @see ConstantSizeRegionSplitPolicy Default split policy before 0.94.0
041 */
042@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
043public abstract class RegionSplitPolicy extends Configured {
044  private static final Class<? extends RegionSplitPolicy>
045    DEFAULT_SPLIT_POLICY_CLASS = SteppingSplitPolicy.class;
046
047  /**
048   * The region configured for this split policy.
049   */
050  protected HRegion region;
051
052  /**
053   * Upon construction, this method will be called with the region
054   * to be governed. It will be called once and only once.
055   */
056  protected void configureForRegion(HRegion region) {
057    Preconditions.checkState(
058        this.region == null,
059        "Policy already configured for region {}",
060        this.region);
061
062    this.region = region;
063  }
064
065  /**
066   * @return true if the specified region should be split.
067   */
068  protected abstract boolean shouldSplit();
069
070  /**
071   * @return the key at which the region should be split, or null
072   * if it cannot be split. This will only be called if shouldSplit
073   * previously returned true.
074   */
075  protected byte[] getSplitPoint() {
076    byte[] explicitSplitPoint = this.region.getExplicitSplitPoint();
077    if (explicitSplitPoint != null) {
078      return explicitSplitPoint;
079    }
080    List<HStore> stores = region.getStores();
081
082    byte[] splitPointFromLargestStore = null;
083    long largestStoreSize = 0;
084    for (HStore s : stores) {
085      Optional<byte[]> splitPoint = s.getSplitPoint();
086      // Store also returns null if it has references as way of indicating it is not splittable
087      long storeSize = s.getSize();
088      if (splitPoint.isPresent() && largestStoreSize < storeSize) {
089        splitPointFromLargestStore = splitPoint.get();
090        largestStoreSize = storeSize;
091      }
092    }
093
094    return splitPointFromLargestStore;
095  }
096
097  /**
098   * Create the RegionSplitPolicy configured for the given table.
099   * @param region
100   * @param conf
101   * @return a RegionSplitPolicy
102   * @throws IOException
103   */
104  public static RegionSplitPolicy create(HRegion region,
105      Configuration conf) throws IOException {
106    Class<? extends RegionSplitPolicy> clazz = getSplitPolicyClass(
107        region.getTableDescriptor(), conf);
108    RegionSplitPolicy policy = ReflectionUtils.newInstance(clazz, conf);
109    policy.configureForRegion(region);
110    return policy;
111  }
112
113  public static Class<? extends RegionSplitPolicy> getSplitPolicyClass(
114      TableDescriptor htd, Configuration conf) throws IOException {
115    String className = htd.getRegionSplitPolicyClassName();
116    if (className == null) {
117      className = conf.get(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
118          DEFAULT_SPLIT_POLICY_CLASS.getName());
119    }
120
121    try {
122      Class<? extends RegionSplitPolicy> clazz =
123        Class.forName(className).asSubclass(RegionSplitPolicy.class);
124      return clazz;
125    } catch (Exception  e) {
126      throw new IOException(
127          "Unable to load configured region split policy '" +
128          className + "' for table '" + htd.getTableName() + "'",
129          e);
130    }
131  }
132
133  /**
134   * In {@link HRegionFileSystem#splitStoreFile(org.apache.hadoop.hbase.client.RegionInfo, String,
135   * HStoreFile, byte[], boolean, RegionSplitPolicy)} we are not creating the split reference
136   * if split row not lies in the StoreFile range. But in some use cases we may need to create
137   * the split reference even when the split row not lies in the range. This method can be used
138   * to decide, whether to skip the the StoreFile range check or not.
139   * @return whether to skip the StoreFile range check or not
140   * @param familyName
141   * @return whether to skip the StoreFile range check or not
142   */
143  protected boolean skipStoreFileRangeCheck(String familyName) {
144    return false;
145  }
146}
