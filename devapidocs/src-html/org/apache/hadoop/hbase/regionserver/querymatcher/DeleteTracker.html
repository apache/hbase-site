018package org.apache.hadoop.hbase.regionserver.querymatcher;
019
020import org.apache.hadoop.hbase.classification.InterfaceAudience;
021import org.apache.hadoop.hbase.Cell;
022import org.apache.hadoop.hbase.regionserver.ShipperListener;
023
024/**
025 * This interface is used for the tracking and enforcement of Deletes during the course of a Get or
026 * Scan operation.
027 * <p>
028 * This class is utilized through three methods:
029 * <ul>
030 * <li>{@link #add} when encountering a Delete</li>
031 * <li>{@link #isDeleted} when checking if a Put KeyValue has been deleted</li>
032 * <li>{@link #update} when reaching the end of a StoreFile</li>
033 * </ul>
034 */
035@InterfaceAudience.Private
036public interface DeleteTracker extends ShipperListener {
037
038  /**
039   * Add the specified cell to the list of deletes to check against for this row operation.
040   * <p>
041   * This is called when a Delete is encountered in a StoreFile.
042   * @param cell - the delete cell
043   */
044  void add(Cell cell);
045
046  /**
047   * Check if the specified cell buffer has been deleted by a previously seen delete.
048   * @param cell - current cell to check if deleted by a previously seen delete
049   * @return deleteResult The result tells whether the KeyValue is deleted and why
050   */
051  DeleteResult isDeleted(Cell cell);
052
053  /**
054   * @return true if there are no current delete, false otherwise
055   */
056  boolean isEmpty();
057
058  /**
059   * Called at the end of every StoreFile.
060   * <p>
061   * Many optimized implementations of Trackers will require an update at when the end of each
062   * StoreFile is reached.
063   */
064  void update();
065
066  /**
067   * Called between rows.
068   * <p>
069   * This clears everything as if a new DeleteTracker was instantiated.
070   */
071  void reset();
072
073  /**
074   * Return codes for comparison of two Deletes.
075   * <p>
076   * The codes tell the merging function what to do.
077   * <p>
078   * INCLUDE means add the specified Delete to the merged list. NEXT means move to the next element
079   * in the specified list(s).
080   */
081  enum DeleteCompare {
082    INCLUDE_OLD_NEXT_OLD,
083    INCLUDE_OLD_NEXT_BOTH,
084    INCLUDE_NEW_NEXT_NEW,
085    INCLUDE_NEW_NEXT_BOTH,
086    NEXT_OLD,
087    NEXT_NEW
088  }
089
090  /**
091   * Returns codes for delete result. The codes tell the ScanQueryMatcher whether the kv is deleted
092   * and why. Based on the delete result, the ScanQueryMatcher will decide the next operation
093   */
094  enum DeleteResult {
095    FAMILY_DELETED, // The KeyValue is deleted by a delete family.
096    FAMILY_VERSION_DELETED, // The KeyValue is deleted by a delete family version.
097    COLUMN_DELETED, // The KeyValue is deleted by a delete column.
098    VERSION_DELETED, // The KeyValue is deleted by a version delete.
099    NOT_DELETED,
100    VERSION_MASKED  // The KeyValue is masked by max number of versions which is considered as
101                    // deleted in strong semantics of versions(See MvccTracker)
102  }
103
104}
