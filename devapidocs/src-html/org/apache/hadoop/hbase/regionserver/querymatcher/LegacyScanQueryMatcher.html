042/**
043 * The old query matcher implementation. Used to keep compatibility for coprocessor that could
044 * overwrite the StoreScanner before compaction. Should be removed once we find a better way to do
045 * filtering during compaction.
046 */
047@Deprecated
048@InterfaceAudience.Private
049public class LegacyScanQueryMatcher extends ScanQueryMatcher {
050
051  private final TimeRange tr;
052
053  private final Filter filter;
054
055  /** Keeps track of deletes */
056  private final DeleteTracker deletes;
057
058  /**
059   * The following three booleans define how we deal with deletes. There are three different
060   * aspects:
061   * <ol>
062   * <li>Whether to keep delete markers. This is used in compactions. Minor compactions always keep
063   * delete markers.</li>
064   * <li>Whether to keep deleted rows. This is also used in compactions, if the store is set to keep
065   * deleted rows. This implies keeping the delete markers as well.</li> In this case deleted rows
066   * are subject to the normal max version and TTL/min version rules just like "normal" rows.
067   * <li>Whether a scan can do time travel queries even before deleted marker to reach deleted
068   * rows.</li>
069   * </ol>
070   */
071  /** whether to retain delete markers */
072  private boolean retainDeletesInOutput;
073
074  /** whether to return deleted rows */
075  private final KeepDeletedCells keepDeletedCells;
076
077  // By default, when hbase.hstore.time.to.purge.deletes is 0ms, a delete
078  // marker is always removed during a major compaction. If set to non-zero
079  // value then major compaction will try to keep a delete marker around for
080  // the given number of milliseconds. We want to keep the delete markers
081  // around a bit longer because old puts might appear out-of-order. For
082  // example, during log replication between two clusters.
083  //
084  // If the delete marker has lived longer than its column-family's TTL then
085  // the delete marker will be removed even if time.to.purge.deletes has not
086  // passed. This is because all the Puts that this delete marker can influence
087  // would have also expired. (Removing of delete markers on col family TTL will
088  // not happen if min-versions is set to non-zero)
089  //
090  // But, if time.to.purge.deletes has not expired then a delete
091  // marker will not be removed just because there are no Puts that it is
092  // currently influencing. This is because Puts, that this delete can
093  // influence. may appear out of order.
094  private final long timeToPurgeDeletes;
095
096  /**
097   * This variable shows whether there is an null column in the query. There always exists a null
098   * column in the wildcard column query. There maybe exists a null column in the explicit column
099   * query based on the first column.
100   */
101  private final boolean hasNullColumn;
102
103  /** readPoint over which the KVs are unconditionally included */
104  private final long maxReadPointToTrackVersions;
105
106  /**
107   * Oldest put in any of the involved store files Used to decide whether it is ok to delete family
108   * delete marker of this store keeps deleted KVs.
109   */
110  protected final long earliestPutTs;
111
112  private final byte[] stopRow;
113
114  private byte[] dropDeletesFromRow = null, dropDeletesToRow = null;
115
116  private LegacyScanQueryMatcher(Scan scan, ScanInfo scanInfo, ColumnTracker columns,
117      boolean hasNullColumn, DeleteTracker deletes, ScanType scanType, long readPointToUse,
118      long earliestPutTs, long oldestUnexpiredTS, long now) {
119    super(createStartKeyFromRow(scan.getStartRow(), scanInfo), scanInfo, columns, oldestUnexpiredTS,
120        now);
121    TimeRange timeRange = scan.getColumnFamilyTimeRange().get(scanInfo.getFamily());
122    if (timeRange == null) {
123      this.tr = scan.getTimeRange();
124    } else {
125      this.tr = timeRange;
126    }
127    this.hasNullColumn = hasNullColumn;
128    this.deletes = deletes;
129    this.filter = scan.getFilter();
130    this.maxReadPointToTrackVersions = readPointToUse;
131    this.timeToPurgeDeletes = scanInfo.getTimeToPurgeDeletes();
132    this.earliestPutTs = earliestPutTs;
133
134    /* how to deal with deletes */
135    this.keepDeletedCells = scanInfo.getKeepDeletedCells();
136    this.retainDeletesInOutput = scanType == ScanType.COMPACT_RETAIN_DELETES;
137    this.stopRow = scan.getStopRow();
138  }
139
140  private LegacyScanQueryMatcher(Scan scan, ScanInfo scanInfo, ColumnTracker columns,
141      boolean hasNullColumn, DeleteTracker deletes, ScanType scanType, long readPointToUse,
142      long earliestPutTs, long oldestUnexpiredTS, long now, byte[] dropDeletesFromRow,
143      byte[] dropDeletesToRow) {
144    this(scan, scanInfo, columns, hasNullColumn, deletes, scanType, readPointToUse, earliestPutTs,
145        oldestUnexpiredTS, now);
146    this.dropDeletesFromRow = Preconditions.checkNotNull(dropDeletesFromRow);
147    this.dropDeletesToRow = Preconditions.checkNotNull(dropDeletesToRow);
148  }

150  @Override
151  public void beforeShipped() throws IOException {
152    super.beforeShipped();
153    deletes.beforeShipped();
154  }

156  @Override
157  public MatchCode match(Cell cell) throws IOException {
158    if (filter != null && filter.filterAllRemaining()) {
159      return MatchCode.DONE_SCAN;
160    }
161    MatchCode returnCode = preCheck(cell);
162    if (returnCode != null) {
163      return returnCode;
164    }
165    /*
166     * The delete logic is pretty complicated now.
167     * This is corroborated by the following:
168     * 1. The store might be instructed to keep deleted rows around.
169     * 2. A scan can optionally see past a delete marker now.
170     * 3. If deleted rows are kept, we have to find out when we can
171     *    remove the delete markers.
172     * 4. Family delete markers are always first (regardless of their TS)
173     * 5. Delete markers should not be counted as version
174     * 6. Delete markers affect puts of the *same* TS
175     * 7. Delete marker need to be version counted together with puts
176     *    they affect
177     */
178    long timestamp = cell.getTimestamp();
179    byte typeByte = cell.getTypeByte();
180    long mvccVersion = cell.getSequenceId();
181    if (CellUtil.isDelete(typeByte)) {
182      if (keepDeletedCells == KeepDeletedCells.FALSE
183          || (keepDeletedCells == KeepDeletedCells.TTL && timestamp < oldestUnexpiredTS)) {
184        // first ignore delete markers if the scanner can do so, and the
185        // range does not include the marker
186        //
187        // during flushes and compactions also ignore delete markers newer
188        // than the readpoint of any open scanner, this prevents deleted
189        // rows that could still be seen by a scanner from being collected
190        boolean includeDeleteMarker = tr.withinOrAfterTimeRange(timestamp);
191        if (includeDeleteMarker && mvccVersion <= maxReadPointToTrackVersions) {
192          this.deletes.add(cell);
193        }
194        // Can't early out now, because DelFam come before any other keys
195      }
196
197      if (timeToPurgeDeletes > 0
198          && (EnvironmentEdgeManager.currentTime() - timestamp) <= timeToPurgeDeletes) {
199        return MatchCode.INCLUDE;
200      } else if (retainDeletesInOutput || mvccVersion > maxReadPointToTrackVersions) {
201        // always include or it is not time yet to check whether it is OK
202        // to purge deltes or not
203        // if this is not a user scan (compaction), we can filter this deletemarker right here
204        // otherwise (i.e. a "raw" scan) we fall through to normal version and timerange checking
205        return MatchCode.INCLUDE;
206      } else if (keepDeletedCells == KeepDeletedCells.TRUE
207          || (keepDeletedCells == KeepDeletedCells.TTL && timestamp >= oldestUnexpiredTS)) {
208        if (timestamp < earliestPutTs) {
209          // keeping delete rows, but there are no puts older than
210          // this delete in the store files.
211          return columns.getNextRowOrNextColumn(cell);
212        }
213        // else: fall through and do version counting on the
214        // delete markers
215      } else {
216        return MatchCode.SKIP;
217      }
218      // note the following next else if...
219      // delete marker are not subject to other delete markers
220    } else if (!this.deletes.isEmpty()) {
221      DeleteResult deleteResult = deletes.isDeleted(cell);
222      switch (deleteResult) {
223        case FAMILY_DELETED:
224        case COLUMN_DELETED:
225          return columns.getNextRowOrNextColumn(cell);
226        case VERSION_DELETED:
227        case FAMILY_VERSION_DELETED:
228          return MatchCode.SKIP;
229        case NOT_DELETED:
230          break;
231        default:
232          throw new RuntimeException("UNEXPECTED");
233        }
234    }
235
236    int timestampComparison = tr.compare(timestamp);
237    if (timestampComparison >= 1) {
238      return MatchCode.SKIP;
239    } else if (timestampComparison <= -1) {
240      return columns.getNextRowOrNextColumn(cell);
241    }
242
243    // STEP 1: Check if the column is part of the requested columns
244    MatchCode colChecker = columns.checkColumn(cell, typeByte);
245    if (colChecker == MatchCode.INCLUDE) {
246      ReturnCode filterResponse = ReturnCode.SKIP;
247      // STEP 2: Yes, the column is part of the requested columns. Check if filter is present
248      if (filter != null) {
249        // STEP 3: Filter the key value and return if it filters out
250        filterResponse = filter.filterKeyValue(cell);
251        switch (filterResponse) {
252        case SKIP:
253          return MatchCode.SKIP;
254        case NEXT_COL:
255          return columns.getNextRowOrNextColumn(cell);
256        case NEXT_ROW:
257          return MatchCode.SEEK_NEXT_ROW;
258        case SEEK_NEXT_USING_HINT:
259          return MatchCode.SEEK_NEXT_USING_HINT;
260        default:
261          //It means it is either include or include and seek next
262          break;
263        }
264      }
265      /*
266       * STEP 4: Reaching this step means the column is part of the requested columns and either
267       * the filter is null or the filter has returned INCLUDE or INCLUDE_AND_NEXT_COL response.
268       * Now check the number of versions needed. This method call returns SKIP, INCLUDE,
269       * INCLUDE_AND_SEEK_NEXT_ROW, INCLUDE_AND_SEEK_NEXT_COL.
270       *
271       * FilterResponse            ColumnChecker               Desired behavior
272       * INCLUDE                   SKIP                        row has already been included, SKIP.
273       * INCLUDE                   INCLUDE                     INCLUDE
274       * INCLUDE                   INCLUDE_AND_SEEK_NEXT_COL   INCLUDE_AND_SEEK_NEXT_COL
275       * INCLUDE                   INCLUDE_AND_SEEK_NEXT_ROW   INCLUDE_AND_SEEK_NEXT_ROW
276       * INCLUDE_AND_SEEK_NEXT_COL SKIP                        row has already been included, SKIP.
277       * INCLUDE_AND_SEEK_NEXT_COL INCLUDE                     INCLUDE_AND_SEEK_NEXT_COL
278       * INCLUDE_AND_SEEK_NEXT_COL INCLUDE_AND_SEEK_NEXT_COL   INCLUDE_AND_SEEK_NEXT_COL
279       * INCLUDE_AND_SEEK_NEXT_COL INCLUDE_AND_SEEK_NEXT_ROW   INCLUDE_AND_SEEK_NEXT_ROW
280       *
281       * In all the above scenarios, we return the column checker return value except for
282       * FilterResponse (INCLUDE_AND_SEEK_NEXT_COL) and ColumnChecker(INCLUDE)
283       */
284      colChecker = columns.checkVersions(cell, timestamp, typeByte,
285          mvccVersion > maxReadPointToTrackVersions);
286      if (filterResponse == ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW) {
287        if (colChecker != MatchCode.SKIP) {
288          return MatchCode.INCLUDE_AND_SEEK_NEXT_ROW;
289        }
290        return MatchCode.SEEK_NEXT_ROW;
291      }
292      return (filterResponse == ReturnCode.INCLUDE_AND_NEXT_COL &&
293          colChecker == MatchCode.INCLUDE) ? MatchCode.INCLUDE_AND_SEEK_NEXT_COL
294          : colChecker;
295    }
296    return colChecker;
297  }

301    return hasNullColumn;
302  }

304  /**
305   * Handle partial-drop-deletes. As we match keys in order, when we have a range from which we can
306   * drop deletes, we can set retainDeletesInOutput to false for the duration of this range only,
307   * and maintain consistency.
308   */
309  private void checkPartialDropDeleteRange(Cell curCell) {
310    // If partial-drop-deletes are used, initially, dropDeletesFromRow and dropDeletesToRow
311    // are both set, and the matcher is set to retain deletes. We assume ordered keys. When
312    // dropDeletesFromRow is leq current kv, we start dropping deletes and reset
313    // dropDeletesFromRow; thus the 2nd "if" starts to apply.
314    if ((dropDeletesFromRow != null)
315        && (Arrays.equals(dropDeletesFromRow, HConstants.EMPTY_START_ROW)
316            || (CellComparator.COMPARATOR.compareRows(curCell, dropDeletesFromRow, 0,
317              dropDeletesFromRow.length) >= 0))) {
318      retainDeletesInOutput = false;
319      dropDeletesFromRow = null;
320    }
321    // If dropDeletesFromRow is null and dropDeletesToRow is set, we are inside the partial-
322    // drop-deletes range. When dropDeletesToRow is leq current kv, we stop dropping deletes,
323    // and reset dropDeletesToRow so that we don't do any more compares.
324    if ((dropDeletesFromRow == null) && (dropDeletesToRow != null)
325        && !Arrays.equals(dropDeletesToRow, HConstants.EMPTY_END_ROW) && (CellComparator.COMPARATOR
326            .compareRows(curCell, dropDeletesToRow, 0, dropDeletesToRow.length) >= 0)) {
327      retainDeletesInOutput = true;
328      dropDeletesToRow = null;
329    }
330  }
331
332  @Override
333  protected void reset() {
334    checkPartialDropDeleteRange(currentRow);
335  }

342  @Override
343  public boolean moreRowsMayExistAfter(Cell cell) {
344    if (this.stopRow == null || this.stopRow.length == 0) {
345      return true;
346    }
347    return rowComparator.compareRows(cell, stopRow, 0, stopRow.length) < 0;
348  }

364  public static LegacyScanQueryMatcher create(Scan scan, ScanInfo scanInfo,
365      NavigableSet<byte[]> columns, ScanType scanType, long readPointToUse, long earliestPutTs,
366      long oldestUnexpiredTS, long now, byte[] dropDeletesFromRow, byte[] dropDeletesToRow,
367      RegionCoprocessorHost regionCoprocessorHost) throws IOException {
368    int maxVersions = Math.min(scan.getMaxVersions(), scanInfo.getMaxVersions());
369    boolean hasNullColumn;
370    ColumnTracker columnTracker;
371    if (columns == null || columns.isEmpty()) {
372      // there is always a null column in the wildcard column query.
373      hasNullColumn = true;
374      // use a specialized scan for wildcard column tracker.
375      columnTracker = new ScanWildcardColumnTracker(scanInfo.getMinVersions(), maxVersions,
376          oldestUnexpiredTS);
377    } else {
378      // We can share the ExplicitColumnTracker, diff is we reset
379      // between rows, not between storefiles.
380      // whether there is null column in the explicit column query
381      hasNullColumn = columns.first().length == 0;
382      columnTracker = new ExplicitColumnTracker(columns, scanInfo.getMinVersions(), maxVersions,
383          oldestUnexpiredTS);
384    }
385    DeleteTracker deletes = instantiateDeleteTracker(regionCoprocessorHost);
386    if (dropDeletesFromRow == null) {
387      return new LegacyScanQueryMatcher(scan, scanInfo, columnTracker, hasNullColumn, deletes,
388          scanType, readPointToUse, earliestPutTs, oldestUnexpiredTS, now);
389    } else {
390      return new LegacyScanQueryMatcher(scan, scanInfo, columnTracker, hasNullColumn, deletes,
391          scanType, readPointToUse, earliestPutTs, oldestUnexpiredTS, now, dropDeletesFromRow,
392          dropDeletesToRow);
393    }
394  }
