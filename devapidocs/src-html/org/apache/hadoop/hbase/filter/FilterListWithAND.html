020package org.apache.hadoop.hbase.filter;
021
022import org.apache.hadoop.hbase.Cell;
023import org.apache.yetus.audience.InterfaceAudience;
024
025import java.io.IOException;
026import java.util.ArrayList;
027import java.util.List;
028
029/**
030 * FilterListWithAND represents an ordered list of filters which will be evaluated with an AND
031 * operator.
032 */
033@InterfaceAudience.Private
034public class FilterListWithAND extends FilterListBase {
035
036  private List<Filter> seekHintFilters = new ArrayList<>();
037
038  public FilterListWithAND(List<Filter> filters) {
039    super(filters);
040  }
041
042  @Override
043  public void addFilterLists(List<Filter> filters) {
044    if (checkAndGetReversed(filters, isReversed()) != isReversed()) {
045      throw new IllegalArgumentException("Filters in the list must have the same reversed flag");
046    }
047    this.filters.addAll(filters);
048  }
049
050  @Override
051  protected String formatLogFilters(List<Filter> logFilters) {
052    return String.format("FilterList AND (%d/%d): %s", logFilters.size(), this.size(),
053      logFilters.toString());
054  }
055
056  /**
057   * FilterList with MUST_PASS_ALL choose the maximal forward step among sub-filters in filter list.
058   * Let's call it: The Maximal Step Rule. So if filter-A in filter list return INCLUDE and filter-B
059   * in filter list return INCLUDE_AND_NEXT_COL, then the filter list should return
060   * INCLUDE_AND_NEXT_COL. For SEEK_NEXT_USING_HINT, it's more special, and in method
061   * filterCellWithMustPassAll(), if any sub-filter return SEEK_NEXT_USING_HINT, then our filter
062   * list will return SEEK_NEXT_USING_HINT. so we don't care about the SEEK_NEXT_USING_HINT here.
063   * <br/>
064   * <br/>
065   * The jump step will be:
066   *
067   * <pre>
068   * INCLUDE &lt; SKIP &lt; INCLUDE_AND_NEXT_COL &lt; NEXT_COL &lt; INCLUDE_AND_SEEK_NEXT_ROW &lt; NEXT_ROW &lt; SEEK_NEXT_USING_HINT
069   * </pre>
070   *
071   * Here, we have the following map to describe The Maximal Step Rule. if current return code (for
072   * previous sub-filters in filter list) is <strong>ReturnCode</strong>, and current filter returns
073   * <strong>localRC</strong>, then we should return map[ReturnCode][localRC] for the merged result,
074   * according to The Maximal Step Rule. <br/>
075   *
076   * <pre>
077   * LocalCode\ReturnCode       INCLUDE                    INCLUDE_AND_NEXT_COL      INCLUDE_AND_SEEK_NEXT_ROW  SKIP                  NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
078   * INCLUDE                    INCLUDE                    INCLUDE_AND_NEXT_COL      INCLUDE_AND_SEEK_NEXT_ROW  SKIP                  NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
079   * INCLUDE_AND_NEXT_COL       INCLUDE_AND_NEXT_COL       INCLUDE_AND_NEXT_COL      INCLUDE_AND_SEEK_NEXT_ROW  NEXT_COL              NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
080   * INCLUDE_AND_SEEK_NEXT_ROW  INCLUDE_AND_SEEK_NEXT_ROW  INCLUDE_AND_SEEK_NEXT_ROW INCLUDE_AND_SEEK_NEXT_ROW  NEXT_ROW              NEXT_ROW              NEXT_ROW              SEEK_NEXT_USING_HINT
081   * SKIP                       SKIP                       NEXT_COL                  NEXT_ROW                   SKIP                  NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
082   * NEXT_COL                   NEXT_COL                   NEXT_COL                  NEXT_ROW                   NEXT_COL              NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
083   * NEXT_ROW                   NEXT_ROW                   NEXT_ROW                  NEXT_ROW                   NEXT_ROW              NEXT_ROW              NEXT_ROW              SEEK_NEXT_USING_HINT
084   * SEEK_NEXT_USING_HINT       SEEK_NEXT_USING_HINT       SEEK_NEXT_USING_HINT      SEEK_NEXT_USING_HINT       SEEK_NEXT_USING_HINT  SEEK_NEXT_USING_HINT  SEEK_NEXT_USING_HINT  SEEK_NEXT_USING_HINT
085   * </pre>
086   *
087   * @param rc Return code which is calculated by previous sub-filter(s) in filter list.
088   * @param localRC Return code of the current sub-filter in filter list.
089   * @return Return code which is merged by the return code of previous sub-filter(s) and the return
090   *         code of current sub-filter.
091   */
092  private ReturnCode mergeReturnCode(ReturnCode rc, ReturnCode localRC) {
093    if (rc == ReturnCode.SEEK_NEXT_USING_HINT || localRC == ReturnCode.SEEK_NEXT_USING_HINT) {
094      return ReturnCode.SEEK_NEXT_USING_HINT;
095    }
096    switch (localRC) {
097    case INCLUDE:
098      return rc;
099    case INCLUDE_AND_NEXT_COL:
100      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL)) {
101        return ReturnCode.INCLUDE_AND_NEXT_COL;
102      }
103      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
104        return ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW;
105      }
106      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.NEXT_COL)) {
107        return ReturnCode.NEXT_COL;
108      }
109      if (isInReturnCodes(rc, ReturnCode.NEXT_ROW)) {
110        return ReturnCode.NEXT_ROW;
111      }
112      break;
113    case INCLUDE_AND_SEEK_NEXT_ROW:
114      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL,
115        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
116        return ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW;
117      }
118      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.NEXT_COL, ReturnCode.NEXT_ROW)) {
119        return ReturnCode.NEXT_ROW;
120      }
121      break;
122    case SKIP:
123      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.SKIP)) {
124        return ReturnCode.SKIP;
125      }
126      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_NEXT_COL, ReturnCode.NEXT_COL)) {
127        return ReturnCode.NEXT_COL;
128      }
129      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW, ReturnCode.NEXT_ROW)) {
130        return ReturnCode.NEXT_ROW;
131      }
132      break;
133    case NEXT_COL:
134      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL, ReturnCode.SKIP,
135        ReturnCode.NEXT_COL)) {
136        return ReturnCode.NEXT_COL;
137      }
138      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW, ReturnCode.NEXT_ROW)) {
139        return ReturnCode.NEXT_ROW;
140      }
141      break;
142    case NEXT_ROW:
143      return ReturnCode.NEXT_ROW;
144    }
145    throw new IllegalStateException(
146        "Received code is not valid. rc: " + rc + ", localRC: " + localRC);
147  }
148
149  @Override
150  ReturnCode internalFilterCell(Cell c, Cell transformedCell) throws IOException {
151    if (isEmpty()) {
152      return ReturnCode.INCLUDE;
153    }
154    ReturnCode rc = ReturnCode.INCLUDE;
155    Cell transformed = transformedCell;
156    this.referenceCell = c;
157    this.seekHintFilters.clear();
158    for (int i = 0, n = filters.size(); i < n; i++) {
159      Filter filter = filters.get(i);
160      if (filter.filterAllRemaining()) {
161        return ReturnCode.NEXT_ROW;
162      }
163      ReturnCode localRC;
164      if (filter instanceof FilterList) {
165        localRC = ((FilterList) filter).internalFilterCell(c, transformed);
166      } else {
167        localRC = filter.filterCell(c);
168      }
169      rc = mergeReturnCode(rc, localRC);
170
171      // For INCLUDE* case, we need to update the transformed cell.
172      if (isInReturnCodes(localRC, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL,
173        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
174        transformed = filter.transformCell(transformed);
175      }
176      if (localRC == ReturnCode.SEEK_NEXT_USING_HINT) {
177        seekHintFilters.add(filter);
178      }
179    }
180    this.transformedCell = transformed;
181    if (!seekHintFilters.isEmpty()) {
182      return ReturnCode.SEEK_NEXT_USING_HINT;
183    }
184    return rc;
185  }
186
187  @Override
188  public void reset() throws IOException {
189    for (int i = 0, n = filters.size(); i < n; i++) {
190      filters.get(i).reset();
191    }
192    seekHintFilters.clear();
193  }
194
195  @Override
196  public boolean filterRowKey(byte[] rowKey, int offset, int length) throws IOException {
197    if (isEmpty()) {
198      return super.filterRowKey(rowKey, offset, length);
199    }
200    boolean retVal = false;
201    for (int i = 0, n = filters.size(); i < n; i++) {
202      Filter filter = filters.get(i);
203      if (filter.filterAllRemaining() || filter.filterRowKey(rowKey, offset, length)) {
204        retVal = true;
205      }
206    }
207    return retVal;
208  }
209
210  @Override
211  public boolean filterRowKey(Cell firstRowCell) throws IOException {
212    if (isEmpty()) {
213      return super.filterRowKey(firstRowCell);
214    }
215    boolean retVal = false;
216    for (int i = 0, n = filters.size(); i < n; i++) {
217      Filter filter = filters.get(i);
218      if (filter.filterAllRemaining() || filter.filterRowKey(firstRowCell)) {
219        // Can't just return true here, because there are some filters (such as PrefixFilter) which
220        // will catch the row changed event by filterRowKey(). If we return early here, those
221        // filters will have no chance to update their row state.
222        retVal = true;
223      }
224    }
225    return retVal;
226  }
227
228  @Override
229  public boolean filterAllRemaining() throws IOException {
230    if (isEmpty()) {
231      return super.filterAllRemaining();
232    }
233    for (int i = 0, n = filters.size(); i < n; i++) {
234      if (filters.get(i).filterAllRemaining()) {
235        return true;
236      }
237    }
238    return false;
239  }
240
241  @Override
242  public boolean filterRow() throws IOException {
243    if (isEmpty()) {
244      return super.filterRow();
245    }
246    for (int i = 0, n = filters.size(); i < n; i++) {
247      Filter filter = filters.get(i);
248      if (filter.filterRow()) {
249        return true;
250      }
251    }
252    return false;
253  }
254
255  @Override
256  public Cell getNextCellHint(Cell currentCell) throws IOException {
257    if (isEmpty()) {
258      return super.getNextCellHint(currentCell);
259    }
260    Cell maxHint = null;
261    for (Filter filter : seekHintFilters) {
262      if (filter.filterAllRemaining()) {
263        continue;
264      }
265      Cell curKeyHint = filter.getNextCellHint(currentCell);
266      if (maxHint == null) {
267        maxHint = curKeyHint;
268        continue;
269      }
270      if (this.compareCell(maxHint, curKeyHint) < 0) {
271        maxHint = curKeyHint;
272      }
273    }
274    return maxHint;
275  }
