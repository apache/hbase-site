020package org.apache.hadoop.hbase.filter;
021
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026
027import org.apache.hadoop.hbase.Cell;
028import org.apache.hadoop.hbase.CellComparator;
029import org.apache.hadoop.hbase.CellUtil;
030import org.apache.hadoop.hbase.KeyValueUtil;
031import org.apache.yetus.audience.InterfaceAudience;
032
033/**
034 * Base class for FilterList. Currently, we have two sub-classes to extend this class:
035 * {@link FilterListWithOR}, {@link FilterListWithAND}.
036 */
037@InterfaceAudience.Private
038public abstract class FilterListBase extends FilterBase {
039  private static final int MAX_LOG_FILTERS = 5;
040  protected final ArrayList<Filter> filters;
041
042  /** Reference Cell used by {@link #transformCell(Cell)} for validation purpose. */
043  protected Cell referenceCell = null;
044
045  /**
046   * When filtering a given Cell in {@link #filterCell(Cell)}, this stores the transformed Cell
047   * to be returned by {@link #transformCell(Cell)}. Individual filters transformation are applied
048   * only when the filter includes the Cell. Transformations are composed in the order specified by
049   * {@link #filters}.
050   */
051  protected Cell transformedCell = null;
052
053  public FilterListBase(List<Filter> filters) {
054    reversed = checkAndGetReversed(filters, reversed);
055    this.filters = new ArrayList<>(filters);
056  }
057
058  protected static boolean isInReturnCodes(ReturnCode testRC, ReturnCode... returnCodes) {
059    return Arrays.stream(returnCodes).anyMatch(testRC::equals);
060  }
061
062  protected static boolean checkAndGetReversed(List<Filter> rowFilters, boolean defaultValue) {
063    if (rowFilters.isEmpty()) {
064      return defaultValue;
065    }
066    Boolean retValue = rowFilters.get(0).isReversed();
067    boolean allEqual = rowFilters.stream().map(Filter::isReversed).allMatch(retValue::equals);
068    if (!allEqual) {
069      throw new IllegalArgumentException("Filters in the list must have the same reversed flag");
070    }
071    return retValue;
072  }
073
074  public abstract void addFilterLists(List<Filter> filters);
075
076  public int size() {
077    return this.filters.size();
078  }
079
080  public boolean isEmpty() {
081    return this.filters.isEmpty();
082  }
083
084  public ArrayList<Filter> getFilters() {
085    return this.filters;
086  }
087
088  protected int compareCell(Cell a, Cell b) {
089    int cmp = CellComparator.getInstance().compare(a, b);
090    return reversed ? -1 * cmp : cmp;
091  }
092
093  @Override
094  public Cell transformCell(Cell c) throws IOException {
095    if (isEmpty()) {
096      return super.transformCell(c);
097    }
098    if (!CellUtil.equals(c, referenceCell)) {
099      throw new IllegalStateException(
100          "Reference Cell: " + this.referenceCell + " does not match: " + c);
101    }
102    // Copy transformedCell into a new cell and reset transformedCell & referenceCell to null for
103    // Java GC optimization
104    Cell cell = KeyValueUtil.copyToNewKeyValue(this.transformedCell);
105    this.transformedCell = null;
106    this.referenceCell = null;
107    return cell;
108  }
109
110  /**
111   * Internal implementation of {@link #filterCell(Cell)}
112   * @param c The cell in question.
113   * @param transformedCell The transformed cell of previous filter(s)
114   * @return ReturnCode of this filter operation.
115   * @throws IOException
116   * @see org.apache.hadoop.hbase.filter.FilterList#internalFilterCell(Cell, Cell)
117   */
118  abstract ReturnCode internalFilterCell(Cell c, Cell transformedCell) throws IOException;
119
120  @Override
121  public ReturnCode filterKeyValue(final Cell c) throws IOException {
122    return filterCell(c);
123  }
124
125  @Override
126  public ReturnCode filterCell(final Cell c) throws IOException {
127    return internalFilterCell(c, c);
128  }
129
130  /**
131   * Filters that never filter by modifying the returned List of Cells can inherit this
132   * implementation that does nothing. {@inheritDoc}
133   */
134  @Override
135  public void filterRowCells(List<Cell> cells) throws IOException {
136    for (int i = 0, n = filters.size(); i < n; i++) {
137      filters.get(i).filterRowCells(cells);
138    }
139  }
140
141  @Override
142  public boolean hasFilterRow() {
143    for (int i = 0, n = filters.size(); i < n; i++) {
144      if (filters.get(i).hasFilterRow()) {
145        return true;
146      }
147    }
148    return false;
149  }
150
151  @Override
152  public boolean isFamilyEssential(byte[] name) throws IOException {
153    if (this.filters.isEmpty()) {
154      return super.isFamilyEssential(name);
155    }
156    for (int i = 0, n = filters.size(); i < n; i++) {
157      if (filters.get(i).isFamilyEssential(name)) {
158        return true;
159      }
160    }
161    return false;
162  }
163
164  @Override
165  public void setReversed(boolean reversed) {
166    for (int i = 0, n = filters.size(); i < n; i++) {
167      filters.get(i).setReversed(reversed);
168    }
169    this.reversed = reversed;
170  }
171
172  @Override
173  public String toString() {
174    int endIndex = this.size() < MAX_LOG_FILTERS ? this.size() : MAX_LOG_FILTERS;
175    return formatLogFilters(filters.subList(0, endIndex));
176  }
177
178  protected abstract String formatLogFilters(List<Filter> logFilters);
179}
