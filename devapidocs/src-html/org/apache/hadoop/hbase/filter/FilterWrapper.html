018package org.apache.hadoop.hbase.filter;
019
020import java.io.IOException;
021import java.util.List;
022import java.util.Objects;
023import org.apache.hadoop.hbase.Cell;
024import org.apache.hadoop.hbase.exceptions.DeserializationException;
025import org.apache.yetus.audience.InterfaceAudience;
026
027import org.apache.hbase.thirdparty.com.google.protobuf.InvalidProtocolBufferException;
028
029import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
030import org.apache.hadoop.hbase.shaded.protobuf.generated.FilterProtos;
031
032/**
033 * This is a Filter wrapper class which is used in the server side. Some filter related hooks can be
034 * defined in this wrapper. The only way to create a FilterWrapper instance is passing a client side
035 * Filter instance through {@link org.apache.hadoop.hbase.client.Scan#getFilter()}.
036 */
037@InterfaceAudience.Private
038final public class FilterWrapper extends Filter {
039  Filter filter = null;
040
041  /**
042   * Constructor.
043   * @param filter filter to wrap
044   * @throws NullPointerException if {@code filter} is {@code null}
045   */
046  public FilterWrapper(Filter filter) {
047    this.filter = Objects.requireNonNull(filter, "Cannot create FilterWrapper with null Filter");
048  }
049
050  /**
051   * @return The filter serialized using pb
052   */
053  @Override
054  public byte[] toByteArray() throws IOException {
055    FilterProtos.FilterWrapper.Builder builder = FilterProtos.FilterWrapper.newBuilder();
056    builder.setFilter(ProtobufUtil.toFilter(this.filter));
057    return builder.build().toByteArray();
058  }
059
060  /**
061   * @param pbBytes A pb serialized {@link FilterWrapper} instance
062   * @return An instance of {@link FilterWrapper} made from <code>bytes</code>
063   * @throws org.apache.hadoop.hbase.exceptions.DeserializationException
064   * @see #toByteArray
065   */
066  public static FilterWrapper parseFrom(final byte[] pbBytes) throws DeserializationException {
067    FilterProtos.FilterWrapper proto;
068    try {
069      proto = FilterProtos.FilterWrapper.parseFrom(pbBytes);
070    } catch (InvalidProtocolBufferException e) {
071      throw new DeserializationException(e);
072    }
073    try {
074      return new FilterWrapper(ProtobufUtil.toFilter(proto.getFilter()));
075    } catch (IOException ioe) {
076      throw new DeserializationException(ioe);
077    }
078  }
079
080  @Override
081  public void reset() throws IOException {
082    this.filter.reset();
083  }
084
085  @Override
086  public boolean filterAllRemaining() throws IOException {
087    return this.filter.filterAllRemaining();
088  }
089
090  @Override
091  public boolean filterRow() throws IOException {
092    return this.filter.filterRow();
093  }
094
095  @Override
096  public Cell getNextCellHint(Cell currentCell) throws IOException {
097    return this.filter.getNextCellHint(currentCell);
098  }
099
100  @Override
101  public boolean filterRowKey(Cell cell) throws IOException {
102    if (filterAllRemaining()) return true;
103    return this.filter.filterRowKey(cell);
104  }
105
106  @Override
107  public ReturnCode filterCell(final Cell c) throws IOException {
108    return this.filter.filterCell(c);
109  }
110
111  @Override
112  public Cell transformCell(Cell v) throws IOException {
113    return this.filter.transformCell(v);
114  }
115
116  @Override
117  public boolean hasFilterRow() {
118    return this.filter.hasFilterRow();
119  }
120
121  @Override
122  public void filterRowCells(List<Cell> kvs) throws IOException {
123    filterRowCellsWithRet(kvs);
124  }
125
126  public enum FilterRowRetCode {
127    NOT_CALLED,
128    INCLUDE, // corresponds to filter.filterRow() returning false
129    EXCLUDE, // corresponds to filter.filterRow() returning true
130    INCLUDE_THIS_FAMILY // exclude other families
131  }
132
133  public FilterRowRetCode filterRowCellsWithRet(List<Cell> kvs) throws IOException {
134    // To fix HBASE-6429,
135    // Filter with filterRow() returning true is incompatible with scan with limit
136    // 1. hasFilterRow() returns true, if either filterRow() or filterRow(kvs) is implemented.
137    // 2. filterRow() is merged with filterRow(kvs),
138    // so that to make all those row related filtering stuff in the same function.
139    this.filter.filterRowCells(kvs);
140    if (!kvs.isEmpty()) {
141      if (this.filter.filterRow()) {
142        kvs.clear();
143        return FilterRowRetCode.EXCLUDE;
144      }
145      return FilterRowRetCode.INCLUDE;
146    }
147    return FilterRowRetCode.NOT_CALLED;
148  }
149
150  @Override
151  public boolean isFamilyEssential(byte[] name) throws IOException {
152    return filter.isFamilyEssential(name);
153  }
154
155  /**
156   * @param o the other filter to compare with
157   * @return true if and only if the fields of the filter that are serialized are equal to the
158   *         corresponding fields in other. Used for testing.
159   */
160  @Override
161  boolean areSerializedFieldsEqual(Filter o) {
162    if (o == this) return true;
163    if (!(o instanceof FilterWrapper)) return false;
164
165    FilterWrapper other = (FilterWrapper) o;
166    return this.filter.areSerializedFieldsEqual(other.filter);
167  }
168}
