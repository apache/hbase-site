045/**
046 * Runs periodically to determine if the WAL should be rolled.
047 * <p/>
048 * NOTE: This class extends Thread rather than Chore because the sleep time can be interrupted when
049 * there is something to do, rather than the Chore sleep time which is invariant.
050 * <p/>
051 * The {@link #scheduleFlush(String, List)} is abstract here,
052 * as sometimes we hold a region without a region server but we still want to roll its WAL.
053 * <p/>
054 * TODO: change to a pool of threads
055 */
056@InterfaceAudience.Private
057public abstract class AbstractWALRoller<T extends Abortable> extends Thread
058  implements Closeable {
059  private static final Logger LOG = LoggerFactory.getLogger(AbstractWALRoller.class);
060
061  protected static final String WAL_ROLL_PERIOD_KEY = "hbase.regionserver.logroll.period";
062
063  /**
064   * Configure for the timeout of log rolling retry.
065   */
066  protected static final String WAL_ROLL_WAIT_TIMEOUT = "hbase.regionserver.logroll.wait.timeout.ms";
067
068  /**
069   * Configure for the max count of log rolling retry.
070   * The real retry count is also limited by the timeout of log rolling
071   * via {@link #WAL_ROLL_WAIT_TIMEOUT}
072   */
073  protected static final String WAL_ROLL_RETRIES = "hbase.regionserver.logroll.retries";
074
075  protected final ConcurrentMap<WAL, RollController> wals = new ConcurrentHashMap<>();
076  protected final T abortable;
077  // Period to roll log.
078  private final long rollPeriod;
079  private final int threadWakeFrequency;
080  // The interval to check low replication on hlog's pipeline
081  private final long checkLowReplicationInterval;
082  // Wait period for roll log
083  private final long rollWaitTimeout;
084  // Max retry for roll log
085  private final int maxRollRetry;
086
087  private volatile boolean running = true;
088
089  public void addWAL(WAL wal) {
090    // check without lock first
091    if (wals.containsKey(wal)) {
092      return;
093    }
094    // this is to avoid race between addWAL and requestRollAll.
095    synchronized (this) {
096      if (wals.putIfAbsent(wal, new RollController(wal)) == null) {
097        wal.registerWALActionsListener(new WALActionsListener() {
098          @Override
099          public void logRollRequested(WALActionsListener.RollRequestReason reason) {
100            // TODO logs will contend with each other here, replace with e.g. DelayedQueue
101            synchronized (AbstractWALRoller.this) {
102              RollController controller = wals.computeIfAbsent(wal, rc -> new RollController(wal));
103              controller.requestRoll();
104              AbstractWALRoller.this.notifyAll();
105            }
106          }
107
108          @Override
109          public void postLogArchive(Path oldPath, Path newPath) throws IOException {
110            afterWALArchive(oldPath, newPath);
111          }
112        });
113      }
114    }
115  }
116
117  public void requestRollAll() {
118    synchronized (this) {
119      for (RollController controller : wals.values()) {
120        controller.requestRoll();
121      }
122      notifyAll();
123    }
124  }
125
126  protected AbstractWALRoller(String name, Configuration conf, T abortable) {
127    super(name);
128    this.abortable = abortable;
129    this.rollPeriod = conf.getLong(WAL_ROLL_PERIOD_KEY, 3600000);
130    this.threadWakeFrequency = conf.getInt(HConstants.THREAD_WAKE_FREQUENCY, 10 * 1000);
131    this.checkLowReplicationInterval =
132      conf.getLong("hbase.regionserver.hlog.check.lowreplication.interval", 30 * 1000);
133    this.rollWaitTimeout = conf.getLong(WAL_ROLL_WAIT_TIMEOUT, 30000);
134    // retry rolling does not have to be the default behavior, so the default value is 0 here
135    this.maxRollRetry = conf.getInt(WAL_ROLL_RETRIES, 0);
136  }
137
138  /**
139   * we need to check low replication in period, see HBASE-18132
140   */
141  private void checkLowReplication(long now) {
142    try {
143      for (Entry<WAL, RollController> entry : wals.entrySet()) {
144        WAL wal = entry.getKey();
145        boolean needRollAlready = entry.getValue().needsRoll(now);
146        if (needRollAlready || !(wal instanceof AbstractFSWAL)) {
147          continue;
148        }
149        ((AbstractFSWAL<?>) wal).checkLogLowReplication(checkLowReplicationInterval);
150      }
151    } catch (Throwable e) {
152      LOG.warn("Failed checking low replication", e);
153    }
154  }
155
156  private void abort(String reason, Throwable cause) {
157    // close all WALs before calling abort on RS.
158    // This is because AsyncFSWAL replies on us for rolling a new writer to make progress, and if we
159    // failed, AsyncFSWAL may be stuck, so we need to close it to let the upper layer know that it
160    // is already broken.
161    for (WAL wal : wals.keySet()) {
162      // shutdown rather than close here since we are going to abort the RS and the wals need to be
163      // split when recovery
164      try {
165        wal.shutdown();
166      } catch (IOException e) {
167        LOG.warn("Failed to shutdown wal", e);
168      }
169    }
170    abortable.abort(reason, cause);
171  }
172
173  @Override
174  public void run() {
175    while (running) {
176      long now = EnvironmentEdgeManager.currentTime();
177      checkLowReplication(now);
178      synchronized (this) {
179        if (wals.values().stream().noneMatch(rc -> rc.needsRoll(now))) {
180          try {
181            wait(this.threadWakeFrequency);
182          } catch (InterruptedException e) {
183            // restore the interrupt state
184            Thread.currentThread().interrupt();
185          }
186          // goto the beginning to check whether again whether we should fall through to roll
187          // several WALs, and also check whether we should quit.
188          continue;
189        }
190      }
191      try {
192        for (Iterator<Entry<WAL, RollController>> iter = wals.entrySet().iterator();
193             iter.hasNext();) {
194          Entry<WAL, RollController> entry = iter.next();
195          WAL wal = entry.getKey();
196          RollController controller = entry.getValue();
197          if (controller.isRollRequested()) {
198            // WAL roll requested, fall through
199            LOG.debug("WAL {} roll requested", wal);
200          } else if (controller.needsPeriodicRoll(now)){
201            // Time for periodic roll, fall through
202            LOG.debug("WAL {} roll period {} ms elapsed", wal, this.rollPeriod);
203          } else {
204            continue;
205          }
206          Map<byte[], List<byte[]>> regionsToFlush = null;
207          int nAttempts = 0;
208          long startWaiting = EnvironmentEdgeManager.currentTime();
209          do {
210            try {
211              // Force the roll if the logroll.period is elapsed or if a roll was requested.
212              // The returned value is an collection of actual region and family names.
213              regionsToFlush = controller.rollWal(EnvironmentEdgeManager.currentTime());
214              break;
215            } catch (IOException ioe) {
216              if (ioe instanceof WALClosedException) {
217                LOG.warn("WAL has been closed. Skipping rolling of writer and just remove it", ioe);
218                iter.remove();
219                break;
220              }
221              long waitingTime = EnvironmentEdgeManager.currentTime() - startWaiting;
222              if (waitingTime < rollWaitTimeout && nAttempts < maxRollRetry) {
223                nAttempts++;
224                LOG.warn("Retry to roll log, nAttempts={}, waiting time={}ms, sleeping 1s to retry,"
225                    + " last excepiton= {}", nAttempts, waitingTime,
226                  ioe.getCause().getClass().getSimpleName());
227                sleep(1000);
228              } else {
229                LOG.error("Roll wal failed and waiting timeout, will not retry", ioe);
230                throw ioe;
231              }
232            }
233          } while (EnvironmentEdgeManager.currentTime() - startWaiting < rollWaitTimeout);
234          if (regionsToFlush != null) {
235            for (Map.Entry<byte[], List<byte[]>> r : regionsToFlush.entrySet()) {
236              scheduleFlush(Bytes.toString(r.getKey()), r.getValue());
237            }
238          }
239        }
240      } catch (FailedLogCloseException | ConnectException e) {
241        abort("Failed log close in log roller", e);
242      } catch (IOException ex) {
243        // Abort if we get here. We probably won't recover an IOE. HBASE-1132
244        abort("IOE in log roller",
245          ex instanceof RemoteException ? ((RemoteException) ex).unwrapRemoteException() : ex);
246      } catch (Exception ex) {
247        LOG.error("Log rolling failed", ex);
248        abort("Log rolling failed", ex);
249      }
250    }
251    LOG.info("LogRoller exiting.");
252  }
253
254  protected void afterWALArchive(Path oldPath, Path newPath) {
255  }

257  /**
258   * @param encodedRegionName Encoded name of region to flush.
259   * @param families stores of region to flush.
260   */
261  protected abstract void scheduleFlush(String encodedRegionName, List<byte[]> families);
262
263  private boolean isWaiting() {
264    Thread.State state = getState();
265    return state == Thread.State.WAITING || state == Thread.State.TIMED_WAITING;
266  }
267
268  /**
269   * @return true if all WAL roll finished
270   */
271  public boolean walRollFinished() {
272    // TODO add a status field of roll in RollController
273    return wals.values().stream()
274        .noneMatch(rc -> rc.needsRoll(EnvironmentEdgeManager.currentTime()))
275      && isWaiting();
276  }
277
278  /**
279   * Wait until all wals have been rolled after calling {@link #requestRollAll()}.
280   */
281  public void waitUntilWalRollFinished() throws InterruptedException {
282    while (!walRollFinished()) {
283      Thread.sleep(100);
284    }
285  }
286
287  @Override
288  public void close() {
289    running = false;
290    interrupt();
291  }
292
293  /**
294   * Independently control the roll of each wal. When use multiwal,
295   * can avoid all wal roll together. see HBASE-24665 for detail
296   */
297  protected class RollController {
298    private final WAL wal;
299    private final AtomicBoolean rollRequest;
300    private long lastRollTime;
301
302    RollController(WAL wal) {
303      this.wal = wal;
304      this.rollRequest = new AtomicBoolean(false);
305      this.lastRollTime = EnvironmentEdgeManager.currentTime();
306    }
307
308    public void requestRoll() {
309      this.rollRequest.set(true);
310    }
311
312    public Map<byte[], List<byte[]>> rollWal(long now) throws IOException {
313      this.lastRollTime = now;
314      // reset the flag in front to avoid missing roll request before we return from rollWriter.
315      this.rollRequest.set(false);
316      return wal.rollWriter(true);
317    }
318
319    public boolean isRollRequested() {
320      return rollRequest.get();
321    }
322
323    public boolean needsPeriodicRoll(long now) {
324      return (now - this.lastRollTime) > rollPeriod;
325    }
326
327    public boolean needsRoll(long now) {
328      return isRollRequested() || needsPeriodicRoll(now);
329    }
330  }
