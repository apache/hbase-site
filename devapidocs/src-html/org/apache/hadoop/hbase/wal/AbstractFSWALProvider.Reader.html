046/**
047 * Base class of a WAL Provider that returns a single thread safe WAL that writes to Hadoop FS. By
048 * default, this implementation picks a directory in Hadoop FS based on a combination of
049 * <ul>
050 * <li>the HBase root directory
051 * <li>HConstants.HREGION_LOGDIR_NAME
052 * <li>the given factory's factoryId (usually identifying the regionserver by host:port)
053 * </ul>
054 * It also uses the providerId to differentiate among files.
055 */
056@InterfaceAudience.Private
057@InterfaceStability.Evolving
058public abstract class AbstractFSWALProvider<T extends AbstractFSWAL<?>> implements WALProvider {
059
060  private static final Logger LOG = LoggerFactory.getLogger(AbstractFSWALProvider.class);

064  public static final boolean DEFAULT_SEPARATE_OLDLOGDIR = false;

077  protected volatile T wal;
078  protected WALFactory factory = null;
079  protected Configuration conf = null;
080  protected List<WALActionsListener> listeners = null;
081  protected String providerId = null;
082  protected AtomicBoolean initialized = new AtomicBoolean(false);
083  // for default wal provider, logPrefix won't change
084  protected String logPrefix = null;
085
086  /**
087   * we synchronized on walCreateLock to prevent wal recreation in different threads
088   */
089  private final Object walCreateLock = new Object();
090
091  /**
092   * @param factory factory that made us, identity used for FS layout. may not be null
093   * @param conf may not be null
094   * @param listeners may be null
095   * @param providerId differentiate between providers from one factory, used for FS layout. may be
096   *          null
097   */
098  @Override
099  public void init(WALFactory factory, Configuration conf, List<WALActionsListener> listeners,
100      String providerId) throws IOException {
101    if (!initialized.compareAndSet(false, true)) {
102      throw new IllegalStateException("WALProvider.init should only be called once.");
103    }
104    this.factory = factory;
105    this.conf = conf;
106    this.listeners = listeners;
107    this.providerId = providerId;
108    // get log prefix
109    StringBuilder sb = new StringBuilder().append(factory.factoryId);
110    if (providerId != null) {
111      if (providerId.startsWith(WAL_FILE_NAME_DELIMITER)) {
112        sb.append(providerId);
113      } else {
114        sb.append(WAL_FILE_NAME_DELIMITER).append(providerId);
115      }
116    }
117    logPrefix = sb.toString();
118    doInit(conf);
119  }
120
121  @Override
122  public List<WAL> getWALs() {
123    if (wal == null) {
124      return Collections.emptyList();
125    }
126    List<WAL> wals = new ArrayList<>(1);
127    wals.add(wal);
128    return wals;
129  }
130
131  @Override
132  public T getWAL(RegionInfo region) throws IOException {
133    T walCopy = wal;
134    if (walCopy == null) {
135      // only lock when need to create wal, and need to lock since
136      // creating hlog on fs is time consuming
137      synchronized (walCreateLock) {
138        walCopy = wal;
139        if (walCopy == null) {
140          walCopy = createWAL();
141          wal = walCopy;
142        }
143      }
144    }
145    return walCopy;
146  }

148  protected abstract T createWAL() throws IOException;

152  @Override
153  public void shutdown() throws IOException {
154    T log = this.wal;

156      log.shutdown();
157    }
158  }

161  public void close() throws IOException {
162    T log = this.wal;
163    if (log != null) {
164      log.close();
165    }

231  /**
232   * It returns the file create timestamp from the file name. For name format see
233   * {@link #validateWALFilename(String)} public until remaining tests move to o.a.h.h.wal
234   * @param wal must not be null
235   * @return the file number that is part of the WAL file name
236   */
237  @VisibleForTesting
238  public static long extractFileNumFromWAL(final WAL wal) {
239    final Path walName = ((AbstractFSWAL<?>) wal).getCurrentFileName();
240    if (walName == null) {
241      throw new IllegalArgumentException("The WAL path couldn't be null");
242    }
243    final String[] walPathStrs = walName.toString().split("\\" + WAL_FILE_NAME_DELIMITER);
244    return Long.parseLong(walPathStrs[walPathStrs.length - (isMetaFile(walName) ? 2 : 1)]);
245  }

254  /**
255   * A WAL file name is of the format: &lt;wal-name&gt;{@link #WAL_FILE_NAME_DELIMITER}
256   * &lt;file-creation-timestamp&gt;[.meta]. provider-name is usually made up of a server-name and a
257   * provider-id
258   * @param filename name of the file to validate
259   * @return <tt>true</tt> if the filename matches an WAL, <tt>false</tt> otherwise
260   */
261  public static boolean validateWALFilename(String filename) {
262    return pattern.matcher(filename).matches();
263  }

279  /**
280   * Construct the directory name for all old WALs on a given server. The default old WALs dir
281   * looks like: <code>hbase/oldWALs</code>. If you config hbase.separate.oldlogdir.by.regionserver
282   * to true, it looks like <code>hbase//oldWALs/kalashnikov.att.net,61634,1486865297088</code>.
283   * @param conf
284   * @param serverName Server name formatted as described in {@link ServerName}
285   * @return the relative WAL directory name
286   */
287  public static String getWALArchiveDirectoryName(Configuration conf, final String serverName) {
288    StringBuilder dirName = new StringBuilder(HConstants.HREGION_OLDLOGDIR_NAME);
289    if (conf.getBoolean(SEPARATE_OLDLOGDIR, DEFAULT_SEPARATE_OLDLOGDIR)) {
290      dirName.append(Path.SEPARATOR);
291      dirName.append(serverName);
292    }
293    return dirName.toString();

296  /**
297   * Pulls a ServerName out of a Path generated according to our layout rules. In the below layouts,
298   * this method ignores the format of the logfile component. Current format: [base directory for
299   * hbase]/hbase/.logs/ServerName/logfile or [base directory for
300   * hbase]/hbase/.logs/ServerName-splitting/logfile Expected to work for individual log files and
301   * server-specific directories.
302   * @return null if it's not a log file. Returns the ServerName of the region server that created
303   *         this log file otherwise.
304   */
305  public static ServerName getServerNameFromWALDirectoryName(Configuration conf, String path)
306      throws IOException {
307    if (path == null || path.length() <= HConstants.HREGION_LOGDIR_NAME.length()) {
308      return null;
309    }
310
311    if (conf == null) {
312      throw new IllegalArgumentException("parameter conf must be set");
313    }
314
315    final String rootDir = conf.get(HConstants.HBASE_DIR);
316    if (rootDir == null || rootDir.isEmpty()) {
317      throw new IllegalArgumentException(HConstants.HBASE_DIR + " key not found in conf.");
318    }
319
320    final StringBuilder startPathSB = new StringBuilder(rootDir);
321    if (!rootDir.endsWith("/")) {
322      startPathSB.append('/');
323    }
324    startPathSB.append(HConstants.HREGION_LOGDIR_NAME);
325    if (!HConstants.HREGION_LOGDIR_NAME.endsWith("/")) {
326      startPathSB.append('/');
327    }
328    final String startPath = startPathSB.toString();
329
330    String fullPath;
331    try {
332      fullPath = FileSystem.get(conf).makeQualified(new Path(path)).toString();
333    } catch (IllegalArgumentException e) {
334      LOG.info("Call to makeQualified failed on " + path + " " + e.getMessage());
335      return null;
336    }
337
338    if (!fullPath.startsWith(startPath)) {
339      return null;
340    }
341
342    final String serverNameAndFile = fullPath.substring(startPath.length());
343
344    if (serverNameAndFile.indexOf('/') < "a,0,0".length()) {
345      // Either it's a file (not a directory) or it's not a ServerName format
346      return null;
347    }
348
349    Path p = new Path(path);
350    return getServerNameFromWALDirectoryName(p);
351  }

353  /**
354   * This function returns region server name from a log file name which is in one of the following
355   * formats:
356   * <ul>
357   * <li>hdfs://&lt;name node&gt;/hbase/.logs/&lt;server name&gt;-splitting/...</li>
358   * <li>hdfs://&lt;name node&gt;/hbase/.logs/&lt;server name&gt;/...</li>
359   * </ul>
360   * @return null if the passed in logFile isn't a valid WAL file path
361   */
362  public static ServerName getServerNameFromWALDirectoryName(Path logFile) {
363    String logDirName = logFile.getParent().getName();
364    // We were passed the directory and not a file in it.
365    if (logDirName.equals(HConstants.HREGION_LOGDIR_NAME)) {
366      logDirName = logFile.getName();
367    }
368    ServerName serverName = null;
369    if (logDirName.endsWith(SPLITTING_EXT)) {
370      logDirName = logDirName.substring(0, logDirName.length() - SPLITTING_EXT.length());
371    }
372    try {
373      serverName = ServerName.parseServerName(logDirName);
374    } catch (IllegalArgumentException|IllegalStateException ex) {
375      serverName = null;
376      LOG.warn("Cannot parse a server name from path=" + logFile + "; " + ex.getMessage());
377    }
378    if (serverName != null && serverName.getStartcode() < 0) {
379      LOG.warn("Invalid log file path=" + logFile);
380      serverName = null;
381    }
382    return serverName;
383  }

401  /**
402   * Get the archived WAL file path
403   * @param path - active WAL file path
404   * @param conf - configuration
405   * @return archived path if exists, path - otherwise
406   * @throws IOException exception
407   */
408  public static Path getArchivedLogPath(Path path, Configuration conf) throws IOException {
409    Path rootDir = FSUtils.getRootDir(conf);
410    Path oldLogDir = new Path(rootDir, HConstants.HREGION_OLDLOGDIR_NAME);
411    if (conf.getBoolean(SEPARATE_OLDLOGDIR, DEFAULT_SEPARATE_OLDLOGDIR)) {
412      ServerName serverName = getServerNameFromWALDirectoryName(path);
413      if (serverName == null) {
414        LOG.error("Couldn't locate log: " + path);
415        return path;
416      }
417      oldLogDir = new Path(oldLogDir, serverName.getServerName());
418    }
419    Path archivedLogLocation = new Path(oldLogDir, path.getName());
420    final FileSystem fs = FSUtils.getCurrentFileSystem(conf);
421
422    if (fs.exists(archivedLogLocation)) {
423      LOG.info("Log " + path + " was moved to " + archivedLogLocation);
424      return archivedLogLocation;
425    } else {
426      LOG.error("Couldn't locate log: " + path);
427      return path;
428    }
429  }

431  /**
432   * Opens WAL reader with retries and
433   * additional exception handling
434   * @param path path to WAL file
435   * @param conf configuration
436   * @return WAL Reader instance
437   * @throws IOException
438   */
439  public static org.apache.hadoop.hbase.wal.WAL.Reader
440    openReader(Path path, Configuration conf)
441        throws IOException
442
443  {
444    long retryInterval = 2000; // 2 sec
445    int maxAttempts = 30;
446    int attempt = 0;
447    Exception ee = null;
448    org.apache.hadoop.hbase.wal.WAL.Reader reader = null;
449    while (reader == null && attempt++ < maxAttempts) {
450      try {
451        // Detect if this is a new file, if so get a new reader else
452        // reset the current reader so that we see the new data
453        reader = WALFactory.createReader(path.getFileSystem(conf), path, conf);
454        return reader;
455      } catch (FileNotFoundException fnfe) {
456        // If the log was archived, continue reading from there
457        Path archivedLog = AbstractFSWALProvider.getArchivedLogPath(path, conf);
458        if (path != archivedLog) {
459          return openReader(archivedLog, conf);
460        } else {
461          throw fnfe;
462        }
463      } catch (LeaseNotRecoveredException lnre) {
464        // HBASE-15019 the WAL was not closed due to some hiccup.
465        LOG.warn("Try to recover the WAL lease " + path, lnre);
466        recoverLease(conf, path);
467        reader = null;
468        ee = lnre;
469      } catch (NullPointerException npe) {
470        // Workaround for race condition in HDFS-4380
471        // which throws a NPE if we open a file before any data node has the most recent block
472        // Just sleep and retry. Will require re-reading compressed WALs for compressionContext.
473        LOG.warn("Got NPE opening reader, will retry.");
474        reader = null;
475        ee = npe;
476      }
477      if (reader == null) {
478        // sleep before next attempt
479        try {
480          Thread.sleep(retryInterval);
481        } catch (InterruptedException e) {
482        }
483      }
484    }
485    throw new IOException("Could not open reader", ee);
486  }
487
488  // For HBASE-15019
489  private static void recoverLease(final Configuration conf, final Path path) {
490    try {
491      final FileSystem dfs = FSUtils.getCurrentFileSystem(conf);
492      FSUtils fsUtils = FSUtils.getInstance(dfs, conf);
493      fsUtils.recoverFileLease(dfs, path, conf, new CancelableProgressable() {
494        @Override
495        public boolean progress() {
496          LOG.debug("Still trying to recover WAL lease: " + path);
497          return true;
498        }
499      });
500    } catch (IOException e) {
501      LOG.warn("unable to recover lease for WAL: " + path, e);
502    }
503  }
