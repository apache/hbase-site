021
022import com.google.common.annotations.VisibleForTesting;
023
024import java.io.Closeable;
025import java.io.IOException;
026import java.util.Map;
027import java.util.Set;
028
029import org.apache.hadoop.hbase.HConstants;
030import org.apache.hadoop.hbase.HRegionInfo;
031import org.apache.hadoop.hbase.classification.InterfaceAudience;
032import org.apache.hadoop.hbase.classification.InterfaceStability;
033// imports we use from yet-to-be-moved regionsever.wal
034import org.apache.hadoop.hbase.regionserver.wal.CompressionContext;
035import org.apache.hadoop.hbase.regionserver.wal.FailedLogCloseException;
036import org.apache.hadoop.hbase.regionserver.wal.WALActionsListener;
037import org.apache.hadoop.hbase.regionserver.wal.WALCoprocessorHost;
038import org.apache.hadoop.hbase.regionserver.wal.WALEdit;

040/**
041 * A Write Ahead Log (WAL) provides service for reading, writing waledits. This interface provides
042 * APIs for WAL users (such as RegionServer) to use the WAL (do append, sync, etc).
043 *
044 * Note that some internals, such as log rolling and performance evaluation tools, will use
045 * WAL.equals to determine if they have already seen a given WAL.
046 */
047@InterfaceAudience.Private
048@InterfaceStability.Evolving
049public interface WAL extends Closeable {
050
051  /**
052   * Registers WALActionsListener
053   */
054  void registerWALActionsListener(final WALActionsListener listener);
055
056  /**
057   * Unregisters WALActionsListener
058   */
059  boolean unregisterWALActionsListener(final WALActionsListener listener);
060
061  /**
062   * Roll the log writer. That is, start writing log messages to a new file.
063   *
064   * <p>
065   * The implementation is synchronized in order to make sure there's one rollWriter
066   * running at any given time.
067   *
068   * @return If lots of logs, flush the returned regions so next time through we
069   *         can clean logs. Returns null if nothing to flush. Names are actual
070   *         region names as returned by {@link HRegionInfo#getEncodedName()}
071   */
072  byte[][] rollWriter() throws FailedLogCloseException, IOException;
073
074  /**
075   * Roll the log writer. That is, start writing log messages to a new file.
076   *
077   * <p>
078   * The implementation is synchronized in order to make sure there's one rollWriter
079   * running at any given time.
080   *
081   * @param force
082   *          If true, force creation of a new writer even if no entries have
083   *          been written to the current writer
084   * @return If lots of logs, flush the returned regions so next time through we
085   *         can clean logs. Returns null if nothing to flush. Names are actual
086   *         region names as returned by {@link HRegionInfo#getEncodedName()}
087   */
088  byte[][] rollWriter(boolean force) throws FailedLogCloseException, IOException;
089
090  /**
091   * Stop accepting new writes. If we have unsynced writes still in buffer, sync them.
092   * Extant edits are left in place in backing storage to be replayed later.
093   */
094  void shutdown() throws IOException;
095
096  /**
097   * Caller no longer needs any edits from this WAL. Implementers are free to reclaim
098   * underlying resources after this call; i.e. filesystem based WALs can archive or
099   * delete files.
100   */
101  void close() throws IOException;
102
103  /**
104   * Append a set of edits to the WAL. The WAL is not flushed/sync'd after this transaction
105   * completes BUT on return this edit must have its region edit/sequence id assigned
106   * else it messes up our unification of mvcc and sequenceid.  On return <code>key</code> will
107   * have the region edit/sequence id filled in.
108   * @param info the regioninfo associated with append
109   * @param key Modified by this call; we add to it this edits region edit/sequence id.
110   * @param edits Edits to append. MAY CONTAIN NO EDITS for case where we want to get an edit
111   * sequence id that is after all currently appended edits.
112   * @param inMemstore Always true except for case where we are writing a compaction completion
113   * record into the WAL; in this case the entry is just so we can finish an unfinished compaction
114   * -- it is not an edit for memstore.
115   * @return Returns a 'transaction id' and <code>key</code> will have the region edit/sequence id
116   * in it.
117   */
118  long append(HRegionInfo info, WALKey key, WALEdit edits, boolean inMemstore) throws IOException;
119
120  /**
121   * updates the seuence number of a specific store.
122   * depending on the flag: replaces current seq number if the given seq id is bigger,
123   * or even if it is lower than existing one
124   * @param encodedRegionName
125   * @param familyName
126   * @param sequenceid
127   * @param onlyIfGreater
128   */
129  void updateStore(byte[] encodedRegionName, byte[] familyName, Long sequenceid,
130      boolean onlyIfGreater);
131
132  /**
133   * Sync what we have in the WAL.
134   * @throws IOException
135   */
136  void sync() throws IOException;
137
138  /**
139   * Sync the WAL if the txId was not already sync'd.
140   * @param txid Transaction id to sync to.
141   * @throws IOException
142   */
143  void sync(long txid) throws IOException;
144
145  /**
146   * WAL keeps track of the sequence numbers that are as yet not flushed im memstores
147   * in order to be able to do accounting to figure which WALs can be let go. This method tells WAL
148   * that some region is about to flush. The flush can be the whole region or for a column family
149   * of the region only.
150   *
151   * <p>Currently, it is expected that the update lock is held for the region; i.e. no
152   * concurrent appends while we set up cache flush.
153   * @param families Families to flush. May be a subset of all families in the region.
154   * @return Returns {@link HConstants#NO_SEQNUM} if we are flushing the whole region OR if
155   * we are flushing a subset of all families but there are no edits in those families not
156   * being flushed; in other words, this is effectively same as a flush of all of the region
157   * though we were passed a subset of regions. Otherwise, it returns the sequence id of the
158   * oldest/lowest outstanding edit.
159   * @see #completeCacheFlush(byte[])
160   * @see #abortCacheFlush(byte[])
161   */
162  Long startCacheFlush(final byte[] encodedRegionName, Set<byte[]> families);
163
164  /**
165   * Complete the cache flush.
166   * @param encodedRegionName Encoded region name.
167   * @see #startCacheFlush(byte[], Set)
168   * @see #abortCacheFlush(byte[])
169   */
170  void completeCacheFlush(final byte[] encodedRegionName);
171
172  /**
173   * Abort a cache flush. Call if the flush fails. Note that the only recovery
174   * for an aborted flush currently is a restart of the regionserver so the
175   * snapshot content dropped by the failure gets restored to the memstore.
176   * @param encodedRegionName Encoded region name.
177   */
178  void abortCacheFlush(byte[] encodedRegionName);
179
180  /**
181   * @return Coprocessor host.
182   */
183  WALCoprocessorHost getCoprocessorHost();
184
185  /**
186   * Gets the earliest unflushed sequence id in the memstore for the region.
187   * @param encodedRegionName The region to get the number for.
188   * @return The earliest/lowest/oldest sequence id if present, HConstants.NO_SEQNUM if absent.
189   * @deprecated Since version 1.2.0. Removing because not used and exposes subtle internal
190   * workings. Use {@link #getEarliestMemstoreSeqNum(byte[], byte[])}
191   */
192  @VisibleForTesting
193  @Deprecated
194  long getEarliestMemstoreSeqNum(byte[] encodedRegionName);
195
196  /**
197   * Gets the earliest unflushed sequence id in the memstore for the store.
198   * @param encodedRegionName The region to get the number for.
199   * @param familyName The family to get the number for.
200   * @return The earliest/lowest/oldest sequence id if present, HConstants.NO_SEQNUM if absent.
201   */
202  long getEarliestMemstoreSeqNum(byte[] encodedRegionName, byte[] familyName);
203
204  /**
205   * Human readable identifying information about the state of this WAL.
206   * Implementors are encouraged to include information appropriate for debugging.
207   * Consumers are advised not to rely on the details of the returned String; it does
208   * not have a defined structure.
209   */
210  @Override
211  String toString();
212
213  /**
214   * When outside clients need to consume persisted WALs, they rely on a provided
215   * Reader.
216   */
217  interface Reader extends Closeable {
218    Entry next() throws IOException;
219    Entry next(Entry reuse) throws IOException;
220    void seek(long pos) throws IOException;
221    long getPosition() throws IOException;
222    void reset() throws IOException;
223  }
224
225  /**
226   * Utility class that lets us keep track of the edit with it's key.
227   */
228  class Entry {
229    private final WALEdit edit;
230    private final WALKey key;
231
232    public Entry() {
233      this(new WALKey(), new WALEdit());
234    }
235
236    /**
237     * Constructor for both params
238     *
239     * @param edit log's edit
240     * @param key log's key
241     */
242    public Entry(WALKey key, WALEdit edit) {
243      this.key = key;
244      this.edit = edit;
245    }
246
247    /**
248     * Gets the edit
249     *
250     * @return edit
251     */
252    public WALEdit getEdit() {
253      return edit;
254    }
255
256    /**
257     * Gets the key
258     *
259     * @return key
260     */
261    public WALKey getKey() {
262      return key;
263    }
264
265    /**
266     * Set compression context for this entry.
267     *
268     * @param compressionContext
269     *          Compression context
270     */
271    public void setCompressionContext(CompressionContext compressionContext) {
272      edit.setCompressionContext(compressionContext);
273      key.setCompressionContext(compressionContext);
274    }
275
276    public boolean hasSerialReplicationScope () {
277      if (getKey().getReplicationScopes() == null || getKey().getReplicationScopes().isEmpty()) {
278        return false;
279      }
280      for (Map.Entry<byte[], Integer> e:getKey().getReplicationScopes().entrySet()) {
281        if (e.getValue() == HConstants.REPLICATION_SCOPE_SERIAL){
282          return true;
283        }
284      }
285      return false;
286    }
287
288    @Override
289    public String toString() {
290      return this.key + "=" + this.edit;
291    }
292  }
