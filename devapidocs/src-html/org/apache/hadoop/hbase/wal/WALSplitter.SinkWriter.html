199  // A wrapper to split one log folder using the method used by distributed
200  // log splitting. Used by tools and unit tests. It should be package private.
201  // It is public only because TestWALObserver is in a different package,
202  // which uses this method to do log splitting.
203  @VisibleForTesting
204  public static List<Path> split(Path rootDir, Path logDir, Path oldLogDir,
205      FileSystem fs, Configuration conf, final WALFactory factory) throws IOException {
206    final FileStatus[] logfiles = SplitLogManager.getFileList(conf,
207        Collections.singletonList(logDir), null);
208    List<Path> splits = new ArrayList<>();
209    if (ArrayUtils.isNotEmpty(logfiles)) {
210      for (FileStatus logfile: logfiles) {
211        WALSplitter s = new WALSplitter(factory, conf, rootDir, fs, null, null);
212        if (s.splitLogFile(logfile, null)) {
213          finishSplitLogFile(rootDir, oldLogDir, logfile.getPath(), conf);
214          if (s.outputSink.splits != null) {
215            splits.addAll(s.outputSink.splits);
216          }
217        }
218      }
219    }
220    if (!fs.delete(logDir, true)) {
221      throw new IOException("Unable to delete src dir: " + logDir);
222    }
223    return splits;
224  }

226  /**
227   * log splitting implementation, splits one log file.
228   * @param logfile should be an actual log file.
229   */
230  @VisibleForTesting
231  boolean splitLogFile(FileStatus logfile, CancelableProgressable reporter) throws IOException {
232    Preconditions.checkState(status == null);
233    Preconditions.checkArgument(logfile.isFile(),
234        "passed in file status is for something other than a regular file.");
235    boolean isCorrupted = false;
236    boolean skipErrors = conf.getBoolean("hbase.hlog.split.skip.errors",
237      SPLIT_SKIP_ERRORS_DEFAULT);
238    int interval = conf.getInt("hbase.splitlog.report.interval.loglines", 1024);
239    Path logPath = logfile.getPath();
240    boolean outputSinkStarted = false;
241    boolean progress_failed = false;
242    int editsCount = 0;
243    int editsSkipped = 0;
244
245    status = TaskMonitor.get().createStatus(
246          "Splitting log file " + logfile.getPath() + "into a temporary staging area.");
247    Reader logFileReader = null;
248    this.fileBeingSplit = logfile;
249    try {
250      long logLength = logfile.getLen();
251      LOG.info("Splitting WAL={}, length={}", logPath, logLength);
252      status.setStatus("Opening log file");
253      if (reporter != null && !reporter.progress()) {
254        progress_failed = true;
255        return false;
256      }
257      logFileReader = getReader(logfile, skipErrors, reporter);
258      if (logFileReader == null) {
259        LOG.warn("Nothing to split in WAL={}", logPath);
260        return true;
261      }
262      int numOpenedFilesBeforeReporting = conf.getInt("hbase.splitlog.report.openedfiles", 3);
263      int numOpenedFilesLastCheck = 0;
264      outputSink.setReporter(reporter);
265      outputSink.startWriterThreads();
266      outputSinkStarted = true;
267      Entry entry;
268      Long lastFlushedSequenceId = -1L;
269      while ((entry = getNextLogLine(logFileReader, logPath, skipErrors)) != null) {
270        byte[] region = entry.getKey().getEncodedRegionName();
271        String encodedRegionNameAsStr = Bytes.toString(region);
272        lastFlushedSequenceId = lastFlushedSequenceIds.get(encodedRegionNameAsStr);
273        if (lastFlushedSequenceId == null) {
274          if (sequenceIdChecker != null) {
275            RegionStoreSequenceIds ids = sequenceIdChecker.getLastSequenceId(region);
276            Map<byte[], Long> maxSeqIdInStores = new TreeMap<>(Bytes.BYTES_COMPARATOR);
277            for (StoreSequenceId storeSeqId : ids.getStoreSequenceIdList()) {
278              maxSeqIdInStores.put(storeSeqId.getFamilyName().toByteArray(),
279                storeSeqId.getSequenceId());
280            }
281            regionMaxSeqIdInStores.put(encodedRegionNameAsStr, maxSeqIdInStores);
282            lastFlushedSequenceId = ids.getLastFlushedSequenceId();
283            if (LOG.isDebugEnabled()) {
284              LOG.debug("DLS Last flushed sequenceid for " + encodedRegionNameAsStr + ": " +
285                  TextFormat.shortDebugString(ids));
286            }
287          }
288          if (lastFlushedSequenceId == null) {
289            lastFlushedSequenceId = -1L;
290          }
291          lastFlushedSequenceIds.put(encodedRegionNameAsStr, lastFlushedSequenceId);
292        }
293        if (lastFlushedSequenceId >= entry.getKey().getSequenceId()) {
294          editsSkipped++;
295          continue;
296        }
297        // Don't send Compaction/Close/Open region events to recovered edit type sinks.
298        if (entry.getEdit().isMetaEdit() && !outputSink.keepRegionEvent(entry)) {
299          editsSkipped++;
300          continue;
301        }
302        entryBuffers.appendEntry(entry);
303        editsCount++;
304        int moreWritersFromLastCheck = this.getNumOpenWriters() - numOpenedFilesLastCheck;
305        // If sufficient edits have passed, check if we should report progress.
306        if (editsCount % interval == 0
307            || moreWritersFromLastCheck > numOpenedFilesBeforeReporting) {
308          numOpenedFilesLastCheck = this.getNumOpenWriters();
309          String countsStr = (editsCount - (editsSkipped + outputSink.getSkippedEdits()))
310              + " edits, skipped " + editsSkipped + " edits.";
311          status.setStatus("Split " + countsStr);
312          if (reporter != null && !reporter.progress()) {
313            progress_failed = true;
314            return false;
315          }
316        }
317      }
318    } catch (InterruptedException ie) {
319      IOException iie = new InterruptedIOException();
320      iie.initCause(ie);
321      throw iie;
322    } catch (CorruptedLogFileException e) {
323      LOG.warn("Could not parse, corrupted WAL={}", logPath, e);
324      if (splitLogWorkerCoordination != null) {
325        // Some tests pass in a csm of null.
326        splitLogWorkerCoordination.markCorrupted(walDir, logfile.getPath().getName(), fs);
327      } else {
328        // for tests only
329        ZKSplitLog.markCorrupted(walDir, logfile.getPath().getName(), fs);
330      }
331      isCorrupted = true;
332    } catch (IOException e) {
333      e = e instanceof RemoteException ? ((RemoteException) e).unwrapRemoteException() : e;
334      throw e;
335    } finally {
336      LOG.debug("Finishing writing output logs and closing down");
337      try {
338        if (null != logFileReader) {
339          logFileReader.close();
340        }
341      } catch (IOException exception) {
342        LOG.warn("Could not close WAL reader", exception);
343      }
344      try {
345        if (outputSinkStarted) {
346          // Set progress_failed to true as the immediate following statement will reset its value
347          // when finishWritingAndClose() throws exception, progress_failed has the right value
348          progress_failed = true;
349          progress_failed = outputSink.finishWritingAndClose() == null;
350        }
351      } finally {
352        String msg =
353            "Processed " + editsCount + " edits across " + outputSink.getNumberOfRecoveredRegions()
354                + " regions; edits skipped=" + editsSkipped + "; log file=" + logPath +
355                ", length=" + logfile.getLen() + // See if length got updated post lease recovery
356                ", corrupted=" + isCorrupted + ", progress failed=" + progress_failed;
357        LOG.info(msg);
358        status.markComplete(msg);
359      }
360    }
361    return !progress_failed;
362  }

364  /**
365   * Completes the work done by splitLogFile by archiving logs
366   * <p>
367   * It is invoked by SplitLogManager once it knows that one of the
368   * SplitLogWorkers have completed the splitLogFile() part. If the master
369   * crashes then this function might get called multiple times.
370   * <p>
371   * @param logfile
372   * @param conf
373   * @throws IOException
374   */
375  public static void finishSplitLogFile(String logfile,
376      Configuration conf)  throws IOException {
377    Path rootdir = FSUtils.getWALRootDir(conf);
378    Path oldLogDir = new Path(rootdir, HConstants.HREGION_OLDLOGDIR_NAME);
379    Path logPath;
380    if (FSUtils.isStartingWithPath(rootdir, logfile)) {
381      logPath = new Path(logfile);
382    } else {
383      logPath = new Path(rootdir, logfile);
384    }
385    finishSplitLogFile(rootdir, oldLogDir, logPath, conf);
386  }
387
388  private static void finishSplitLogFile(Path rootdir, Path oldLogDir,
389      Path logPath, Configuration conf) throws IOException {
390    List<Path> processedLogs = new ArrayList<>();
391    List<Path> corruptedLogs = new ArrayList<>();
392    FileSystem fs;
393    fs = rootdir.getFileSystem(conf);
394    if (ZKSplitLog.isCorrupted(rootdir, logPath.getName(), fs)) {
395      corruptedLogs.add(logPath);
396    } else {
397      processedLogs.add(logPath);
398    }
399    archiveLogs(corruptedLogs, processedLogs, oldLogDir, fs, conf);
400    Path stagingDir = ZKSplitLog.getSplitLogDir(rootdir, logPath.getName());
401    fs.delete(stagingDir, true);
402  }

404  /**
405   * Moves processed logs to a oldLogDir after successful processing Moves
406   * corrupted logs (any log that couldn't be successfully parsed to corruptDir
407   * (.corrupt) for later investigation
408   *
409   * @param corruptedLogs
410   * @param processedLogs
411   * @param oldLogDir
412   * @param fs
413   * @param conf
414   * @throws IOException
415   */
416  private static void archiveLogs(
417      final List<Path> corruptedLogs,
418      final List<Path> processedLogs, final Path oldLogDir,
419      final FileSystem fs, final Configuration conf) throws IOException {
420    final Path corruptDir = new Path(FSUtils.getWALRootDir(conf), HConstants.CORRUPT_DIR_NAME);
421    if (conf.get("hbase.regionserver.hlog.splitlog.corrupt.dir") != null) {
422      LOG.warn("hbase.regionserver.hlog.splitlog.corrupt.dir is deprecated. Default to {}",
423          corruptDir);
424    }
425    if (!fs.mkdirs(corruptDir)) {
426      LOG.info("Unable to mkdir {}", corruptDir);
427    }
428    fs.mkdirs(oldLogDir);
429
430    // this method can get restarted or called multiple times for archiving
431    // the same log files.
432    for (Path corrupted : corruptedLogs) {
433      Path p = new Path(corruptDir, corrupted.getName());
434      if (fs.exists(corrupted)) {
435        if (!fs.rename(corrupted, p)) {
436          LOG.warn("Unable to move corrupted log {} to {}", corrupted, p);
437        } else {
438          LOG.warn("Moved corrupted log {} to {}", corrupted, p);
439        }
440      }
441    }
442
443    for (Path p : processedLogs) {
444      Path newPath = AbstractFSWAL.getWALArchivePath(oldLogDir, p);
445      if (fs.exists(p)) {
446        if (!FSUtils.renameAndSetModifyTime(fs, p, newPath)) {
447          LOG.warn("Unable to move {} to {}", p, newPath);
448        } else {
449          LOG.info("Archived processed log {} to {}", p, newPath);
450        }
451      }
452    }
453  }

455  /**
456   * Path to a file under RECOVERED_EDITS_DIR directory of the region found in
457   * <code>logEntry</code> named for the sequenceid in the passed
458   * <code>logEntry</code>: e.g. /hbase/some_table/2323432434/recovered.edits/2332.
459   * This method also ensures existence of RECOVERED_EDITS_DIR under the region
460   * creating it if necessary.
461   * @param logEntry
462   * @param fileNameBeingSplit the file being split currently. Used to generate tmp file name.
463   * @param tmpDirName of the directory used to sideline old recovered edits file
464   * @param conf
465   * @return Path to file into which to dump split log edits.
466   * @throws IOException
467   */
468  @SuppressWarnings("deprecation")
469  @VisibleForTesting
470  static Path getRegionSplitEditsPath(final Entry logEntry, String fileNameBeingSplit,
471      String tmpDirName, Configuration conf) throws IOException {
472    FileSystem fs = FileSystem.get(conf);
473    Path rootDir = FSUtils.getRootDir(conf);
474    Path tableDir = FSUtils.getTableDir(rootDir, logEntry.getKey().getTableName());
475    String encodedRegionName = Bytes.toString(logEntry.getKey().getEncodedRegionName());
476    Path regiondir = HRegion.getRegionDir(tableDir, encodedRegionName);
477    Path dir = getRegionDirRecoveredEditsDir(regiondir);
478
479    if (!fs.exists(regiondir)) {
480      LOG.info("This region's directory does not exist: {}."
481          + "It is very likely that it was already split so it is "
482          + "safe to discard those edits.", regiondir);
483      return null;
484    }
485    if (fs.exists(dir) && fs.isFile(dir)) {
486      Path tmp = new Path(tmpDirName);
487      if (!fs.exists(tmp)) {
488        fs.mkdirs(tmp);
489      }
490      tmp = new Path(tmp,
491        HConstants.RECOVERED_EDITS_DIR + "_" + encodedRegionName);
492      LOG.warn("Found existing old file: {}. It could be some "
493        + "leftover of an old installation. It should be a folder instead. "
494        + "So moving it to {}", dir, tmp);
495      if (!fs.rename(dir, tmp)) {
496        LOG.warn("Failed to sideline old file {}", dir);
497      }
498    }
499
500    if (!fs.exists(dir) && !fs.mkdirs(dir)) {
501      LOG.warn("mkdir failed on {}", dir);
502    }
503    // Append fileBeingSplit to prevent name conflict since we may have duplicate wal entries now.
504    // Append file name ends with RECOVERED_LOG_TMPFILE_SUFFIX to ensure
505    // region's replayRecoveredEdits will not delete it
506    String fileName = formatRecoveredEditsFileName(logEntry.getKey().getSequenceId());
507    fileName = getTmpRecoveredEditsFileName(fileName + "-" + fileNameBeingSplit);
508    return new Path(dir, fileName);
509  }

555  public static boolean hasRecoveredEdits(final FileSystem fs,
556      final Configuration conf, final RegionInfo regionInfo) throws IOException {
557    // No recovered.edits for non default replica regions
558    if (regionInfo.getReplicaId() != RegionInfo.DEFAULT_REPLICA_ID) {

560    }

562    //Only default replica region can reach here, so we can use regioninfo
563    //directly without converting it to default replica's regioninfo.
564    Path regionDir = HRegion.getRegionDir(rootDir, regionInfo);
565    NavigableSet<Path> files = getSplitEditFilesSorted(fs, regionDir);
566    return files != null && !files.isEmpty();
567  }

582    Path editsdir = getRegionDirRecoveredEditsDir(regiondir);
583    if (!fs.exists(editsdir)) {
584      return filesSorted;
585    }
586    FileStatus[] files = FSUtils.listStatus(fs, editsdir, new PathFilter() {
587      @Override
588      public boolean accept(Path p) {
589        boolean result = false;
590        try {
591          // Return files and only files that match the editfile names pattern.
592          // There can be other files in this directory other than edit files.
593          // In particular, on error, we'll move aside the bad edit file giving
594          // it a timestamp suffix. See moveAsideBadEditsFile.
595          Matcher m = EDITFILES_NAME_PATTERN.matcher(p.getName());
596          result = fs.isFile(p) && m.matches();
597          // Skip the file whose name ends with RECOVERED_LOG_TMPFILE_SUFFIX,
598          // because it means splitwal thread is writting this file.
599          if (p.getName().endsWith(RECOVERED_LOG_TMPFILE_SUFFIX)) {
600            result = false;
601          }
602          // Skip SeqId Files
603          if (isSequenceIdFile(p)) {
604            result = false;
605          }
606        } catch (IOException e) {
607          LOG.warn("Failed isFile check on {}", p, e);
608        }
609        return result;
610      }
611    });
612    if (ArrayUtils.isNotEmpty(files)) {
613      Arrays.asList(files).forEach(status -> filesSorted.add(status.getPath()));
614    }
615    return filesSorted;
616  }

716  /**
717   * Create a new {@link Reader} for reading logs to split.
718   *
719   * @param file
720   * @return A new Reader instance, caller should close
721   * @throws IOException
722   * @throws CorruptedLogFileException
723   */
724  protected Reader getReader(FileStatus file, boolean skipErrors, CancelableProgressable reporter)
725      throws IOException, CorruptedLogFileException {
726    Path path = file.getPath();
727    long length = file.getLen();
728    Reader in;
729
730    // Check for possibly empty file. With appends, currently Hadoop reports a
731    // zero length even if the file has been sync'd. Revisit if HDFS-376 or
732    // HDFS-878 is committed.
733    if (length <= 0) {
734      LOG.warn("File {} might be still open, length is 0", path);
735    }
736
737    try {
738      FSUtils.getInstance(fs, conf).recoverFileLease(fs, path, conf, reporter);
739      try {
740        in = getReader(path, reporter);
741      } catch (EOFException e) {
742        if (length <= 0) {
743          // TODO should we ignore an empty, not-last log file if skip.errors
744          // is false? Either way, the caller should decide what to do. E.g.
745          // ignore if this is the last log in sequence.
746          // TODO is this scenario still possible if the log has been
747          // recovered (i.e. closed)
748          LOG.warn("Could not open {} for reading. File is empty", path, e);
749        }
750        // EOFException being ignored
751        return null;
752      }
753    } catch (IOException e) {
754      if (e instanceof FileNotFoundException) {
755        // A wal file may not exist anymore. Nothing can be recovered so move on
756        LOG.warn("File {} does not exist anymore", path, e);
757        return null;
758      }
759      if (!skipErrors || e instanceof InterruptedIOException) {
760        throw e; // Don't mark the file corrupted if interrupted, or not skipErrors
761      }
762      CorruptedLogFileException t =
763        new CorruptedLogFileException("skipErrors=true Could not open wal " +
764            path + " ignoring");
765      t.initCause(e);
766      throw t;
767    }
768    return in;
769  }

771  static private Entry getNextLogLine(Reader in, Path path, boolean skipErrors)
772  throws CorruptedLogFileException, IOException {
773    try {
774      return in.next();
775    } catch (EOFException eof) {
776      // truncated files are expected if a RS crashes (see HBASE-2643)
777      LOG.info("EOF from wal {}. Continuing.", path);
778      return null;
779    } catch (IOException e) {
780      // If the IOE resulted from bad file format,
781      // then this problem is idempotent and retrying won't help
782      if (e.getCause() != null &&
783          (e.getCause() instanceof ParseException ||
784           e.getCause() instanceof org.apache.hadoop.fs.ChecksumException)) {
785        LOG.warn("Parse exception from wal {}. Continuing", path, e);
786        return null;
787      }
788      if (!skipErrors) {
789        throw e;
790      }
791      CorruptedLogFileException t =
792        new CorruptedLogFileException("skipErrors=true Ignoring exception" +
793            " while parsing wal " + path + ". Marking as corrupted");
794      t.initCause(e);
795      throw t;
796    }
797  }

857  /**
858   * Class which accumulates edits and separates them into a buffer per region
859   * while simultaneously accounting RAM usage. Blocks if the RAM usage crosses
860   * a predefined threshold.
861   *
862   * Writer threads then pull region-specific buffers from this class.
863   */
864  public static class EntryBuffers {
865    PipelineController controller;
866
867    Map<byte[], RegionEntryBuffer> buffers = new TreeMap<>(Bytes.BYTES_COMPARATOR);
868
869    /* Track which regions are currently in the middle of writing. We don't allow
870       an IO thread to pick up bytes from a region if we're already writing
871       data for that region in a different IO thread. */
872    Set<byte[]> currentlyWriting = new TreeSet<>(Bytes.BYTES_COMPARATOR);
873
874    long totalBuffered = 0;
875    long maxHeapUsage;
876    boolean splitWriterCreationBounded;
877
878    public EntryBuffers(PipelineController controller, long maxHeapUsage) {
879      this(controller, maxHeapUsage, false);
880    }
881
882    public EntryBuffers(PipelineController controller, long maxHeapUsage,
883        boolean splitWriterCreationBounded){
884      this.controller = controller;
885      this.maxHeapUsage = maxHeapUsage;
886      this.splitWriterCreationBounded = splitWriterCreationBounded;
887    }
888
889    /**
890     * Append a log entry into the corresponding region buffer.
891     * Blocks if the total heap usage has crossed the specified threshold.
892     *
893     * @throws InterruptedException
894     * @throws IOException
895     */
896    public void appendEntry(Entry entry) throws InterruptedException, IOException {
897      WALKey key = entry.getKey();
898
899      RegionEntryBuffer buffer;
900      long incrHeap;
901      synchronized (this) {
902        buffer = buffers.get(key.getEncodedRegionName());
903        if (buffer == null) {
904          buffer = new RegionEntryBuffer(key.getTableName(), key.getEncodedRegionName());
905          buffers.put(key.getEncodedRegionName(), buffer);
906        }
907        incrHeap= buffer.appendEntry(entry);
908      }
909
910      // If we crossed the chunk threshold, wait for more space to be available
911      synchronized (controller.dataAvailable) {
912        totalBuffered += incrHeap;
913        while (totalBuffered > maxHeapUsage && controller.thrown.get() == null) {
914          LOG.debug("Used {} bytes of buffered edits, waiting for IO threads", totalBuffered);
915          controller.dataAvailable.wait(2000);
916        }
917        controller.dataAvailable.notifyAll();
918      }
919      controller.checkForErrors();
920    }
921
922    /**
923     * @return RegionEntryBuffer a buffer of edits to be written.
924     */
925    synchronized RegionEntryBuffer getChunkToWrite() {
926      // The core part of limiting opening writers is it doesn't return chunk only if the
927      // heap size is over maxHeapUsage. Thus it doesn't need to create a writer for each
928      // region during splitting. It will flush all the logs in the buffer after splitting
929      // through a threadpool, which means the number of writers it created is under control.
930      if (splitWriterCreationBounded && totalBuffered < maxHeapUsage) {
931        return null;
932      }
933      long biggestSize = 0;
934      byte[] biggestBufferKey = null;
935
936      for (Map.Entry<byte[], RegionEntryBuffer> entry : buffers.entrySet()) {
937        long size = entry.getValue().heapSize();
938        if (size > biggestSize && (!currentlyWriting.contains(entry.getKey()))) {
939          biggestSize = size;
940          biggestBufferKey = entry.getKey();
941        }
942      }
943      if (biggestBufferKey == null) {
944        return null;
945      }
946
947      RegionEntryBuffer buffer = buffers.remove(biggestBufferKey);
948      currentlyWriting.add(biggestBufferKey);
949      return buffer;
950    }
951
952    void doneWriting(RegionEntryBuffer buffer) {
953      synchronized (this) {
954        boolean removed = currentlyWriting.remove(buffer.encodedRegionName);
955        assert removed;
956      }
957      long size = buffer.heapSize();
958
959      synchronized (controller.dataAvailable) {
960        totalBuffered -= size;
961        // We may unblock writers
962        controller.dataAvailable.notifyAll();
963      }
964    }
965
966    synchronized boolean isRegionCurrentlyWriting(byte[] region) {
967      return currentlyWriting.contains(region);
968    }
969
970    public void waitUntilDrained() {
971      synchronized (controller.dataAvailable) {
972        while (totalBuffered > 0) {
973          try {
974            controller.dataAvailable.wait(2000);
975          } catch (InterruptedException e) {
976            LOG.warn("Got interrupted while waiting for EntryBuffers is drained");
977            Thread.interrupted();
978            break;
979          }
980        }
981      }
982    }
983  }

985  /**
986   * A buffer of some number of edits for a given region.
987   * This accumulates edits and also provides a memory optimization in order to
988   * share a single byte array instance for the table and region name.
989   * Also tracks memory usage of the accumulated edits.
990   */
991  public static class RegionEntryBuffer implements HeapSize {
992    long heapInBuffer = 0;
993    List<Entry> entryBuffer;
994    TableName tableName;
995    byte[] encodedRegionName;
996
997    RegionEntryBuffer(TableName tableName, byte[] region) {
998      this.tableName = tableName;
999      this.encodedRegionName = region;
1000      this.entryBuffer = new ArrayList<>();
1001    }
1002
1003    long appendEntry(Entry entry) {
1004      internify(entry);
1005      entryBuffer.add(entry);
1006      long incrHeap = entry.getEdit().heapSize() +
1007        ClassSize.align(2 * ClassSize.REFERENCE) + // WALKey pointers
1008        0; // TODO linkedlist entry
1009      heapInBuffer += incrHeap;
1010      return incrHeap;
1011    }
1012
1013    private void internify(Entry entry) {
1014      WALKeyImpl k = entry.getKey();
1015      k.internTableName(this.tableName);
1016      k.internEncodedRegionName(this.encodedRegionName);
1017    }
1018
1019    @Override
1020    public long heapSize() {
1021      return heapInBuffer;
1022    }
1023
1024    public byte[] getEncodedRegionName() {
1025      return encodedRegionName;
1026    }
1027
1028    public List<Entry> getEntryBuffer() {
1029      return entryBuffer;
1030    }
1031
1032    public TableName getTableName() {
1033      return tableName;
1034    }

1037  public static class WriterThread extends Thread {
1038    private volatile boolean shouldStop = false;
1039    private PipelineController controller;
1040    private EntryBuffers entryBuffers;
1041    private OutputSink outputSink = null;
1042
1043    WriterThread(PipelineController controller, EntryBuffers entryBuffers, OutputSink sink, int i){
1044      super(Thread.currentThread().getName() + "-Writer-" + i);
1045      this.controller = controller;
1046      this.entryBuffers = entryBuffers;
1047      outputSink = sink;
1048    }
1049
1050    @Override
1051    public void run()  {
1052      try {
1053        doRun();
1054      } catch (Throwable t) {
1055        LOG.error("Exiting thread", t);
1056        controller.writerThreadError(t);
1057      }
1058    }
1059
1060    private void doRun() throws IOException {
1061      LOG.trace("Writer thread starting");
1062      while (true) {
1063        RegionEntryBuffer buffer = entryBuffers.getChunkToWrite();
1064        if (buffer == null) {
1065          // No data currently available, wait on some more to show up
1066          synchronized (controller.dataAvailable) {
1067            if (shouldStop && !this.outputSink.flush()) {
1068              return;
1069            }
1070            try {
1071              controller.dataAvailable.wait(500);
1072            } catch (InterruptedException ie) {
1073              if (!shouldStop) {
1074                throw new RuntimeException(ie);
1075              }
1076            }
1077          }
1078          continue;
1079        }
1080
1081        assert buffer != null;
1082        try {
1083          writeBuffer(buffer);
1084        } finally {
1085          entryBuffers.doneWriting(buffer);
1086        }
1087      }
1088    }
1089
1090    private void writeBuffer(RegionEntryBuffer buffer) throws IOException {
1091      outputSink.append(buffer);
1092    }
1093
1094    void finish() {
1095      synchronized (controller.dataAvailable) {
1096        shouldStop = true;
1097        controller.dataAvailable.notifyAll();
1098      }
1099    }
1100  }

1318    /**
1319     * Close all of the output streams.
1320     * @return the list of paths written.
1321     */
1322    List<Path> close() throws IOException {
1323      Preconditions.checkState(!closeAndCleanCompleted);
1324
1325      final List<Path> paths = new ArrayList<>();
1326      final List<IOException> thrown = Lists.newArrayList();
1327      ThreadPoolExecutor closeThreadPool = Threads
1328          .getBoundedCachedThreadPool(numThreads, 30L, TimeUnit.SECONDS, new ThreadFactory() {
1329            private int count = 1;
1330
1331            @Override public Thread newThread(Runnable r) {
1332              Thread t = new Thread(r, "split-log-closeStream-" + count++);
1333              return t;
1334            }
1335          });
1336      CompletionService<Void> completionService = new ExecutorCompletionService<>(closeThreadPool);
1337      boolean progress_failed;
1338      try {
1339        progress_failed = executeCloseTask(completionService, thrown, paths);
1340      } catch (InterruptedException e) {
1341        IOException iie = new InterruptedIOException();
1342        iie.initCause(e);
1343        throw iie;
1344      } catch (ExecutionException e) {
1345        throw new IOException(e.getCause());
1346      } finally {
1347        closeThreadPool.shutdownNow();
1348      }
1349      if (!thrown.isEmpty()) {
1350        throw MultipleIOException.createIOException(thrown);
1351      }
1352      writersClosed = true;
1353      closeAndCleanCompleted = true;
1354      if (progress_failed) {
1355        return null;
1356      }
1357      return paths;
1358    }

1361     * @param completionService threadPool to execute the closing tasks
1362     * @param thrown store the exceptions
1363     * @param paths arrayList to store the paths written
1364     * @return if close tasks executed successful
1365     */
1366    boolean executeCloseTask(CompletionService<Void> completionService,
1367        List<IOException> thrown, List<Path> paths)
1368        throws InterruptedException, ExecutionException {
1369      for (final Map.Entry<String, SinkWriter> writersEntry : writers.entrySet()) {
1370        if (LOG.isTraceEnabled()) {
1371          LOG.trace("Submitting close of " + ((WriterAndPath) writersEntry.getValue()).p);
1372        }
1373        completionService.submit(new Callable<Void>() {
1374          @Override public Void call() throws Exception {
1375            WriterAndPath wap = (WriterAndPath) writersEntry.getValue();
1376            Path dst = closeWriter(writersEntry.getKey(), wap, thrown);
1377            paths.add(dst);
1378            return null;
1379          }
1380        });
1381      }
1382      boolean progress_failed = false;
1383      for (int i = 0, n = this.writers.size(); i < n; i++) {
1384        Future<Void> future = completionService.take();
1385        future.get();
1386        if (!progress_failed && reporter != null && !reporter.progress()) {
1387          progress_failed = true;
1388        }
1389      }
1390      return progress_failed;
1391    }

1441    private List<IOException> closeLogWriters(List<IOException> thrown) throws IOException {
1442      if (writersClosed) {
1443        return thrown;
1444      }
1445      if (thrown == null) {
1446        thrown = Lists.newArrayList();
1447      }
1448      try {
1449        for (WriterThread t : writerThreads) {
1450          while (t.isAlive()) {
1451            t.shouldStop = true;

1454              t.join(10);

1456              IOException iie = new InterruptedIOException();
1457              iie.initCause(e);
1458              throw iie;
1459            }
1460          }
1461        }
1462      } finally {

1475                  / 1000 / 1000) + "ms)");
1476        }
1477        writersClosed = true;
1478      }
1479
1480      return thrown;
1481    }

1536    void filterCellByStore(Entry logEntry) {
1537      Map<byte[], Long> maxSeqIdInStores =
1538          regionMaxSeqIdInStores.get(Bytes.toString(logEntry.getKey().getEncodedRegionName()));
1539      if (MapUtils.isEmpty(maxSeqIdInStores)) {
1540        return;
1541      }
1542      // Create the array list for the cells that aren't filtered.
1543      // We make the assumption that most cells will be kept.
1544      ArrayList<Cell> keptCells = new ArrayList<>(logEntry.getEdit().getCells().size());
1545      for (Cell cell : logEntry.getEdit().getCells()) {
1546        if (CellUtil.matchingFamily(cell, WALEdit.METAFAMILY)) {
1547          keptCells.add(cell);
1548        } else {
1549          byte[] family = CellUtil.cloneFamily(cell);
1550          Long maxSeqId = maxSeqIdInStores.get(family);
1551          // Do not skip cell even if maxSeqId is null. Maybe we are in a rolling upgrade,
1552          // or the master was crashed before and we can not get the information.
1553          if (maxSeqId == null || maxSeqId.longValue() < logEntry.getKey().getSequenceId()) {
1554            keptCells.add(cell);
1555          }
1556        }
1557      }
1558
1559      // Anything in the keptCells array list is still live.
1560      // So rather than removing the cells from the array list
1561      // which would be an O(n^2) operation, we just replace the list
1562      logEntry.getEdit().setCells(keptCells);
1563    }

1570    WriterAndPath appendBuffer(RegionEntryBuffer buffer, boolean reusable) throws IOException{
1571      List<Entry> entries = buffer.entryBuffer;
1572      if (entries.isEmpty()) {
1573        LOG.warn("got an empty buffer, skipping");
1574        return null;
1575      }
1576
1577      WriterAndPath wap = null;
1578
1579      long startTime = System.nanoTime();
1580      try {
1581        int editsCount = 0;
1582
1583        for (Entry logEntry : entries) {
1584          if (wap == null) {
1585            wap = getWriterAndPath(logEntry, reusable);
1586            if (wap == null) {
1587              // This log spews the full edit. Can be massive in the log. Enable only debugging
1588              // WAL lost edit issues.
1589              LOG.trace("getWriterAndPath decided we don't need to write edits for {}", logEntry);
1590              return null;
1591            }
1592          }
1593          filterCellByStore(logEntry);
1594          if (!logEntry.getEdit().isEmpty()) {
1595            wap.w.append(logEntry);
1596            this.updateRegionMaximumEditLogSeqNum(logEntry);
1597            editsCount++;
1598          } else {
1599            wap.incrementSkippedEdits(1);
1600          }
1601        }
1602        // Pass along summary statistics
1603        wap.incrementEdits(editsCount);
1604        wap.incrementNanoTime(System.nanoTime() - startTime);
1605      } catch (IOException e) {
1606          e = e instanceof RemoteException ?
1607                  ((RemoteException)e).unwrapRemoteException() : e;
1608        LOG.error(HBaseMarkers.FATAL, "Got while writing log entry to log", e);
1609        throw e;
1610      }
1611      return wap;
1612    }

1839  /**
1840   * This function is used to construct mutations from a WALEntry. It also
1841   * reconstructs WALKey &amp; WALEdit from the passed in WALEntry
1842   * @param entry
1843   * @param cells
1844   * @param logEntry pair of WALKey and WALEdit instance stores WALKey and WALEdit instances
1845   *          extracted from the passed in WALEntry.
1846   * @return list of Pair&lt;MutationType, Mutation&gt; to be replayed
1847   * @throws IOException
1848   */
1849  public static List<MutationReplay> getMutationsFromWALEntry(WALEntry entry, CellScanner cells,
1850      Pair<WALKey, WALEdit> logEntry, Durability durability) throws IOException {
1851    if (entry == null) {
1852      // return an empty array
1853      return Collections.emptyList();
1854    }
1855
1856    long replaySeqId = (entry.getKey().hasOrigSequenceNumber()) ?
1857      entry.getKey().getOrigSequenceNumber() : entry.getKey().getLogSequenceNumber();
1858    int count = entry.getAssociatedCellCount();
1859    List<MutationReplay> mutations = new ArrayList<>();
1860    Cell previousCell = null;
1861    Mutation m = null;
1862    WALKeyImpl key = null;
1863    WALEdit val = null;
1864    if (logEntry != null) {
1865      val = new WALEdit();
1866    }
1867
1868    for (int i = 0; i < count; i++) {
1869      // Throw index out of bounds if our cell count is off
1870      if (!cells.advance()) {
1871        throw new ArrayIndexOutOfBoundsException("Expected=" + count + ", index=" + i);
1872      }
1873      Cell cell = cells.current();
1874      if (val != null) val.add(cell);
1875
1876      boolean isNewRowOrType =
1877          previousCell == null || previousCell.getTypeByte() != cell.getTypeByte()
1878              || !CellUtil.matchingRows(previousCell, cell);
1879      if (isNewRowOrType) {
1880        // Create new mutation
1881        if (CellUtil.isDelete(cell)) {
1882          m = new Delete(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
1883          // Deletes don't have nonces.
1884          mutations.add(new MutationReplay(
1885              MutationType.DELETE, m, HConstants.NO_NONCE, HConstants.NO_NONCE));
1886        } else {
1887          m = new Put(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
1888          // Puts might come from increment or append, thus we need nonces.
1889          long nonceGroup = entry.getKey().hasNonceGroup()
1890              ? entry.getKey().getNonceGroup() : HConstants.NO_NONCE;
1891          long nonce = entry.getKey().hasNonce() ? entry.getKey().getNonce() : HConstants.NO_NONCE;
1892          mutations.add(new MutationReplay(MutationType.PUT, m, nonceGroup, nonce));
1893        }
1894      }
1895      if (CellUtil.isDelete(cell)) {
1896        ((Delete) m).add(cell);
1897      } else {
1898        ((Put) m).add(cell);
1899      }
1900      m.setDurability(durability);
1901      previousCell = cell;
1902    }
1903
1904    // reconstruct WALKey
1905    if (logEntry != null) {
1906      org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos.WALKey walKeyProto =
1907          entry.getKey();
1908      List<UUID> clusterIds = new ArrayList<>(walKeyProto.getClusterIdsCount());
1909      for (HBaseProtos.UUID uuid : entry.getKey().getClusterIdsList()) {
