019package org.apache.hadoop.hbase.wal;
020
021import java.io.IOException;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.FileSystem;
027import org.apache.hadoop.fs.Path;
028import org.apache.hadoop.hbase.HConstants;
029import org.apache.hadoop.hbase.classification.InterfaceAudience;
030import org.apache.hadoop.hbase.classification.InterfaceStability;
031// imports for things that haven't moved from regionserver.wal yet.
032import org.apache.hadoop.hbase.regionserver.wal.FSHLog;
033import org.apache.hadoop.hbase.regionserver.wal.ProtobufLogWriter;
034import org.apache.hadoop.hbase.util.FSUtils;
035
036/**
037 * A WAL provider that use {@link FSHLog}.
038 */
039@InterfaceAudience.Private
040@InterfaceStability.Evolving
041public class FSHLogProvider extends AbstractFSWALProvider<FSHLog> {
042
043  private static final Log LOG = LogFactory.getLog(FSHLogProvider.class);
044
045  // Only public so classes back in regionserver.wal can access
046  public interface Writer extends WALProvider.Writer {
047    void init(FileSystem fs, Path path, Configuration c, boolean overwritable) throws IOException;
048  }
049
050  /**
051   * public because of FSHLog. Should be package-private
052   */
053  public static Writer createWriter(final Configuration conf, final FileSystem fs, final Path path,
054      final boolean overwritable) throws IOException {
055    // Configuration already does caching for the Class lookup.
056    Class<? extends Writer> logWriterClass = conf.getClass("hbase.regionserver.hlog.writer.impl",
057      ProtobufLogWriter.class, Writer.class);
058    try {
059      Writer writer = logWriterClass.newInstance();
060      writer.init(fs, path, conf, overwritable);
061      return writer;
062    } catch (Exception e) {
063      LOG.debug("Error instantiating log writer.", e);
064      throw new IOException("cannot get log writer", e);
065    }
066  }
067
068  @Override
069  protected FSHLog createWAL() throws IOException {
070    return new FSHLog(FileSystem.get(conf), FSUtils.getRootDir(conf),
071        getWALDirectoryName(factory.factoryId), HConstants.HREGION_OLDLOGDIR_NAME, conf, listeners,
072        true, logPrefix, META_WAL_PROVIDER_ID.equals(providerId) ? META_WAL_PROVIDER_ID : null);
073  }
074
075  @Override
076  protected void doInit(Configuration conf) throws IOException {
077  }
078}
