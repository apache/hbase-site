047/**
048 * Used in HBase's transaction log (WAL) to represent a collection of edits (Cell/KeyValue objects)
049 * that came in as a single transaction. All the edits for a given transaction are written out as a
050 * single record, in PB format, followed (optionally) by Cells written via the WALCellEncoder.
051 * <p>This class is LimitedPrivate for CPs to read-only. The {@link #add} methods are
052 * classified as private methods, not for use by CPs.</p>
053 * <p>WALEdit will accumulate a Set of all column family names referenced by the Cells
054 * {@link #add(Cell)}'d. This is an optimization. Usually when loading a WALEdit, we have the
055 * column family name to-hand.. just shove it into the WALEdit if available. Doing this, we can
056 * save on a parse of each Cell to figure column family down the line when we go to add the
057 * WALEdit to the WAL file. See the hand-off in FSWALEntry Constructor.
058 */
059// TODO: Do not expose this class to Coprocessors. It has set methods. A CP might meddle.
060@InterfaceAudience.LimitedPrivate({ HBaseInterfaceAudience.REPLICATION,
061    HBaseInterfaceAudience.COPROC })
062public class WALEdit implements HeapSize {
063
064  // TODO: Get rid of this; see HBASE-8457
065  public static final byte [] METAFAMILY = Bytes.toBytes("METAFAMILY");
066  @VisibleForTesting
067  public static final byte [] METAROW = Bytes.toBytes("METAROW");
068  @VisibleForTesting
069  public static final byte[] COMPACTION = Bytes.toBytes("HBASE::COMPACTION");
070  @VisibleForTesting
071  public static final byte [] FLUSH = Bytes.toBytes("HBASE::FLUSH");
072  @VisibleForTesting
073  public static final byte [] REGION_EVENT = Bytes.toBytes("HBASE::REGION_EVENT");
074  @VisibleForTesting
075  public static final byte [] BULK_LOAD = Bytes.toBytes("HBASE::BULK_LOAD");
076
077  private final boolean replay;
078
079  private ArrayList<Cell> cells = null;
080
081  /**
082   * All the Cell families in <code>cells</code>. Updated by {@link #add(Cell)} and
083   * {@link #add(Map)}. This Set is passed to the FSWALEntry so it does not have
084   * to recalculate the Set of families in a transaction; makes for a bunch of CPU savings.
085   * An optimization that saves on CPU-expensive Cell-parsing.
086   */
087  private Set<byte []> families = null;

118  public WALEdit(int cellCount, boolean isReplay) {
119    this.replay = isReplay;

123  private Set<byte[]> getOrCreateFamilies() {
124    if (this.families == null) {
125      this.families = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
126    }
127    return this.families;
128  }
129
130  /**
131   * For use by FSWALEntry ONLY. An optimization.
132   * @return All families in {@link #getCells()}; may be null.
133   */
134  public Set<byte []> getFamilies() {
135    return this.families;
136  }
137
138  /**
139   * @return True is <code>f</code> is {@link #METAFAMILY}
140   */
141  public static boolean isMetaEditFamily(final byte [] f) {
142    return Bytes.equals(METAFAMILY, f);
143  }
144
145  public static boolean isMetaEditFamily(Cell cell) {
146    return CellUtil.matchingFamily(cell, METAFAMILY);
147  }
148
149  public boolean isMetaEdit() {
150    for (Cell cell: cells) {
151      if (!isMetaEditFamily(cell)) {
152        return false;
153      }
154    }
155    return true;
156  }
157
158  /**
159   * @return True when current WALEdit is created by log replay. Replication skips WALEdits from
160   *         replay.
161   */
162  public boolean isReplay() {
163    return this.replay;
164  }
165
166  @InterfaceAudience.Private
167  public WALEdit add(Cell cell, byte [] family) {
168    getOrCreateFamilies().add(family);
169    return addCell(cell);
170  }
171
172  @InterfaceAudience.Private
173  public WALEdit add(Cell cell) {
174    // We clone Family each time we add a Cell. Expensive but safe. For CPU savings, use
175    // add(Map) or add(Cell, family).
176    return add(cell, CellUtil.cloneFamily(cell));
177  }
178
179  public boolean isEmpty() {
180    return cells.isEmpty();
181  }
182
183  public int size() {
184    return cells.size();
185  }
186
187  public ArrayList<Cell> getCells() {
188    return cells;
189  }

191  /**
192   * This is not thread safe.
193   * This will change the WALEdit and shouldn't be used unless you are sure that nothing
194   * else depends on the contents being immutable.
195   *
196   * @param cells the list of cells that this WALEdit now contains.
197   */
198  @InterfaceAudience.Private
199  // Used by replay.
200  public void setCells(ArrayList<Cell> cells) {
201    this.cells = cells;
202    this.families = null;
203  }
204
205  /**
206   * Reads WALEdit from cells.
207   * @param cellDecoder Cell decoder.
208   * @param expectedCount Expected cell count.
209   * @return Number of KVs read.
210   */
211  public int readFromCells(Codec.Decoder cellDecoder, int expectedCount) throws IOException {
212    cells.clear();
213    cells.ensureCapacity(expectedCount);
214    while (cells.size() < expectedCount && cellDecoder.advance()) {
215      cells.add(cellDecoder.current());
216    }
217    return cells.size();
218  }
219
220  @Override
221  public long heapSize() {
222    long ret = ClassSize.ARRAYLIST;
223    for (Cell cell : cells) {
224      ret += cell.heapSize();
225    }
226    return ret;
227  }
228
229  public long estimatedSerializedSizeOf() {
230    long ret = 0;
231    for (Cell cell: cells) {
232      ret += PrivateCellUtil.estimatedSerializedSizeOf(cell);
233    }
234    return ret;
235  }
236
237  @Override
238  public String toString() {
239    StringBuilder sb = new StringBuilder();
240
241    sb.append("[#edits: " + cells.size() + " = <");
242    for (Cell cell : cells) {
243      sb.append(cell);
244      sb.append("; ");
245    }
246    sb.append(">]");
247    return sb.toString();
248  }

250  public static WALEdit createFlushWALEdit(RegionInfo hri, FlushDescriptor f) {
251    KeyValue kv = new KeyValue(getRowForRegion(hri), METAFAMILY, FLUSH,
252      EnvironmentEdgeManager.currentTime(), f.toByteArray());
253    return new WALEdit().add(kv, METAFAMILY);
254  }

260    return null;

277  /**
278   * Create a compaction WALEdit
279   * @param c
280   * @return A WALEdit that has <code>c</code> serialized as its value
281   */
282  public static WALEdit createCompaction(final RegionInfo hri, final CompactionDescriptor c) {
283    byte [] pbbytes = c.toByteArray();
284    KeyValue kv = new KeyValue(getRowForRegion(hri), METAFAMILY, COMPACTION,
285      EnvironmentEdgeManager.currentTime(), pbbytes);
286    return new WALEdit().add(kv, METAFAMILY); //replication scope null so this won't be replicated
287  }
288
289  public static byte[] getRowForRegion(RegionInfo hri) {
290    byte[] startKey = hri.getStartKey();
291    if (startKey.length == 0) {
292      // empty row key is not allowed in mutations because it is both the start key and the end key
293      // we return the smallest byte[] that is bigger (in lex comparison) than byte[0].
294      return new byte[] {0};
295    }
296    return startKey;
297  }

305    if (isCompactionMarker(kv)) {
306      return CompactionDescriptor.parseFrom(CellUtil.cloneValue(kv));

349  /**
350   * Append the given map of family->edits to a WALEdit data structure.
351   * This does not write to the WAL itself.
352   * Note that as an optimization, we will stamp the Set of column families into the WALEdit
353   * to save on our having to calculate it subsequently way down in the actual WAL writing.
354   *
355   * @param familyMap map of family->edits
356   */
357  public void add(Map<byte[], List<Cell>> familyMap) {
358    for (Map.Entry<byte [], List<Cell>> e: familyMap.entrySet()) {
359      // 'foreach' loop NOT used. See HBASE-12023 "...creates too many iterator objects."
360      int listSize = e.getValue().size();
361      // Add all Cells first and then at end, add the family rather than call {@link #add(Cell)}
362      // and have it clone family each time. Optimization!
363      for (int i = 0; i < listSize; i++) {
364        addCell(e.getValue().get(i));
365      }
366      addFamily(e.getKey());
367    }
368  }
369
370  private void addFamily(byte [] family) {
371    getOrCreateFamilies().add(family);
372  }
373
374  private WALEdit addCell(Cell cell) {
375    this.cells.add(cell);
376    return this;
377  }
