044/**
045 * Used in HBase's transaction log (WAL) to represent a collection of edits (Cell/KeyValue objects)
046 * that came in as a single transaction. All the edits for a given transaction are written out as a
047 * single record, in PB format, followed (optionally) by Cells written via the WALCellEncoder.
048 * <p>This class is LimitedPrivate for CPs to read-only. The {@link #add} methods are
049 * classified as private methods, not for use by CPs.</p>
050 *
051 * <p>A particular WALEdit 'type' is the 'meta' type used to mark key operational
052 * events in the WAL such as compaction, flush, or region open. These meta types do not traverse
053 * hbase memstores. They are edits made by the hbase system rather than edit data submitted by
054 * clients. They only show in the WAL. These 'Meta' types have not been formally specified
055 * (or made into an explicit class type). They evolved organically. HBASE-8457 suggests codifying
056 * a WALEdit 'type' by adding a type field to WALEdit that gets serialized into the WAL. TODO.
057 * Would have to work on the consumption-side. Reading WALs on replay we seem to consume
058 * a Cell-at-a-time rather than by WALEdit. We are already in the below going out of our
059 * way to figure particular types --  e.g. if a compaction, replay, or close meta Marker -- during
060 * normal processing so would make sense to do this. Current system is an awkward marking of Cell
061 * columnfamily as {@link #METAFAMILY} and then setting qualifier based off meta edit type. For
062 * replay-time where we read Cell-at-a-time, there are utility methods below for figuring
063 * meta type. See also
064 * {@link #createBulkLoadEvent(RegionInfo, WALProtos.BulkLoadDescriptor)}, etc., for where we
065 * create meta WALEdit instances.</p>
066 *
067 * <p>WALEdit will accumulate a Set of all column family names referenced by the Cells
068 * {@link #add(Cell)}'d. This is an optimization. Usually when loading a WALEdit, we have the
069 * column family name to-hand.. just shove it into the WALEdit if available. Doing this, we can
070 * save on a parse of each Cell to figure column family down the line when we go to add the
071 * WALEdit to the WAL file. See the hand-off in FSWALEntry Constructor.
072 * @see WALKey
073 */
074// TODO: Do not expose this class to Coprocessors. It has set methods. A CP might meddle.
075@InterfaceAudience.LimitedPrivate({ HBaseInterfaceAudience.REPLICATION,
076    HBaseInterfaceAudience.COPROC })
077public class WALEdit implements HeapSize {

104  /**
105   * Qualifier for region event meta 'Marker' WALEdits start with the
106   * {@link #REGION_EVENT_PREFIX} prefix ('HBASE::REGION_EVENT::'). After the prefix,
107   * we note the type of the event which we get from the RegionEventDescriptor protobuf
108   * instance type (A RegionEventDescriptor protobuf instance is written as the meta Marker
109   * Cell value). Adding a type suffix means we do not have to deserialize the protobuf to
110   * figure out what type of event this is.. .just read the qualifier suffix. For example,
111   * a close region event descriptor will have a qualifier of HBASE::REGION_EVENT::REGION_CLOSE.
112   * See WAL.proto and the EventType in RegionEventDescriptor protos for all possible
113   * event types.
114   */
115  private static final String REGION_EVENT_STR = "HBASE::REGION_EVENT";
116  private static final String REGION_EVENT_PREFIX_STR = REGION_EVENT_STR + "::";
117  private static final byte [] REGION_EVENT_PREFIX = Bytes.toBytes(REGION_EVENT_PREFIX_STR);

234    // We clone Family each time we add a Cell. Expensive but safe. For CPU savings, use

265  /**
266   * Reads WALEdit from cells.
267   * @param cellDecoder Cell decoder.
268   * @param expectedCount Expected cell count.
269   * @return Number of KVs read.
270   */
271  public int readFromCells(Codec.Decoder cellDecoder, int expectedCount) throws IOException {
272    cells.clear();
273    cells.ensureCapacity(expectedCount);
274    while (cells.size() < expectedCount && cellDecoder.advance()) {
275      add(cellDecoder.current());
276    }
277    return cells.size();
278  }
279
280  @Override
281  public long heapSize() {
282    long ret = ClassSize.ARRAYLIST;
283    for (Cell cell : cells) {
284      ret += cell.heapSize();
285    }
286    return ret;
287  }
288
289  public long estimatedSerializedSizeOf() {
290    long ret = 0;
291    for (Cell cell: cells) {
292      ret += PrivateCellUtil.estimatedSerializedSizeOf(cell);
293    }
294    return ret;
295  }

297  @Override
298  public String toString() {
299    StringBuilder sb = new StringBuilder();
300
301    sb.append("[#edits: ").append(cells.size()).append(" = <");
302    for (Cell cell : cells) {
303      sb.append(cell);
304      sb.append("; ");
305    }
306    sb.append(">]");
307    return sb.toString();
308  }

380  public static byte[] getRowForRegion(RegionInfo hri) {
381    byte[] startKey = hri.getStartKey();
382    if (startKey.length == 0) {
383      // empty row key is not allowed in mutations because it is both the start key and the end key
384      // we return the smallest byte[] that is bigger (in lex comparison) than byte[0].
385      return new byte[] {0};
386    }
387    return startKey;
388  }

432  /**
433   * Append the given map of family->edits to a WALEdit data structure.
434   * This does not write to the WAL itself.
435   * Note that as an optimization, we will stamp the Set of column families into the WALEdit
436   * to save on our having to calculate column families subsequently down in the actual WAL
437   * writing.
438   *
439   * @param familyMap map of family->edits
440   */
441  public void add(Map<byte[], List<Cell>> familyMap) {
442    for (Map.Entry<byte [], List<Cell>> e: familyMap.entrySet()) {
443      // 'foreach' loop NOT used. See HBASE-12023 "...creates too many iterator objects."
444      int listSize = e.getValue().size();
445      // Add all Cells first and then at end, add the family rather than call {@link #add(Cell)}
446      // and have it clone family each time. Optimization!
447      for (int i = 0; i < listSize; i++) {
448        addCell(e.getValue().get(i));
449      }
450      addFamily(e.getKey());
451    }
452  }
453
454  private void addFamily(byte [] family) {
455    getOrCreateFamilies().add(family);
456  }
457
458  private WALEdit addCell(Cell cell) {
459    this.cells.add(cell);
460    return this;
461  }
