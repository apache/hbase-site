052/**
053 * A Key for an entry in the WAL.
054 *
055 * The log intermingles edits to many tables and rows, so each log entry
056 * identifies the appropriate table and row.  Within a table and row, they're
057 * also sorted.
058 *
059 * <p>Some Transactional edits (START, COMMIT, ABORT) will not have an associated row.
060 *
061 */
062// TODO: Key and WALEdit are never used separately, or in one-to-many relation, for practical
063//       purposes. They need to be merged into WALEntry.
064@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.REPLICATION)
065public class WALKey implements SequenceId, Comparable<WALKey> {

493  @Override
494  public int compareTo(WALKey o) {
495    int result = Bytes.compareTo(this.encodedRegionName, o.encodedRegionName);
496    if (result == 0) {
497      long sid = getSequenceId();
498      long otherSid = o.getSequenceId();
499      if (sid < otherSid) {
500        result = -1;
501      } else if (sid  > otherSid) {
502        result = 1;
503      }
504      if (result == 0) {
505        if (this.writeTime < o.writeTime) {
506          result = -1;
507        } else if (this.writeTime > o.writeTime) {
508          return 1;
509        }
510      }
511    }
512    // why isn't cluster id accounted for?
513    return result;
514  }

542  public WALProtos.WALKey.Builder getBuilder(
543      WALCellCodec.ByteStringCompressor compressor) throws IOException {
544    WALProtos.WALKey.Builder builder = WALProtos.WALKey.newBuilder();
545    if (compressionContext == null) {
546      builder.setEncodedRegionName(UnsafeByteOperations.unsafeWrap(this.encodedRegionName));
547      builder.setTableName(UnsafeByteOperations.unsafeWrap(this.tablename.getName()));
548    } else {
549      builder.setEncodedRegionName(compressor.compress(this.encodedRegionName,
550          compressionContext.regionDict));
551      builder.setTableName(compressor.compress(this.tablename.getName(),
552          compressionContext.tableDict));
553    }
554    builder.setLogSequenceNumber(getSequenceId());
555    builder.setWriteTime(writeTime);
556    if (this.origLogSeqNum > 0) {
557      builder.setOrigSequenceNumber(this.origLogSeqNum);
558    }
559    if (this.nonce != HConstants.NO_NONCE) {
560      builder.setNonce(nonce);
561    }
562    if (this.nonceGroup != HConstants.NO_NONCE) {
563      builder.setNonceGroup(nonceGroup);
564    }
565    HBaseProtos.UUID.Builder uuidBuilder = HBaseProtos.UUID.newBuilder();
566    for (UUID clusterId : clusterIds) {
567      uuidBuilder.setLeastSigBits(clusterId.getLeastSignificantBits());
568      uuidBuilder.setMostSigBits(clusterId.getMostSignificantBits());
569      builder.addClusterIds(uuidBuilder.build());
570    }
571    if (replicationScope != null) {
572      for (Map.Entry<byte[], Integer> e : replicationScope.entrySet()) {
573        ByteString family = (compressionContext == null)
574            ? UnsafeByteOperations.unsafeWrap(e.getKey())
575            : compressor.compress(e.getKey(), compressionContext.familyDict);
576        builder.addScopes(FamilyScope.newBuilder()
577            .setFamily(family).setScopeType(ScopeType.forNumber(e.getValue())));
578      }
579    }
580    return builder;
581  }

583  public void readFieldsFromPb(WALProtos.WALKey walKey,
584                               WALCellCodec.ByteStringUncompressor uncompressor)
585      throws IOException {
586    if (this.compressionContext != null) {
587      this.encodedRegionName = uncompressor.uncompress(
588          walKey.getEncodedRegionName(), compressionContext.regionDict);
589      byte[] tablenameBytes = uncompressor.uncompress(
590          walKey.getTableName(), compressionContext.tableDict);
591      this.tablename = TableName.valueOf(tablenameBytes);
592    } else {
593      this.encodedRegionName = walKey.getEncodedRegionName().toByteArray();
594      this.tablename = TableName.valueOf(walKey.getTableName().toByteArray());
595    }
596    clusterIds.clear();
597    for (HBaseProtos.UUID clusterId : walKey.getClusterIdsList()) {
598      clusterIds.add(new UUID(clusterId.getMostSigBits(), clusterId.getLeastSigBits()));
599    }
600    if (walKey.hasNonceGroup()) {
601      this.nonceGroup = walKey.getNonceGroup();
602    }
603    if (walKey.hasNonce()) {
604      this.nonce = walKey.getNonce();
605    }
606    this.replicationScope = null;
607    if (walKey.getScopesCount() > 0) {
608      this.replicationScope = new TreeMap<>(Bytes.BYTES_COMPARATOR);
609      for (FamilyScope scope : walKey.getScopesList()) {
610        byte[] family = (compressionContext == null) ? scope.getFamily().toByteArray() :
611          uncompressor.uncompress(scope.getFamily(), compressionContext.familyDict);
612        this.replicationScope.put(family, scope.getScopeType().getNumber());
613      }
614    }
615    setSequenceId(walKey.getLogSequenceNumber());
616    this.writeTime = walKey.getWriteTime();
617    if(walKey.hasOrigSequenceNumber()) {
618      this.origLogSeqNum = walKey.getOrigSequenceNumber();
619    }
620  }

622  public long estimatedSerializedSizeOf() {
623    long size = encodedRegionName != null ? encodedRegionName.length : 0;
624    size += tablename != null ? tablename.toBytes().length : 0;
625    if (clusterIds != null) {
626      size += 16 * clusterIds.size();
627    }
628    if (nonceGroup != HConstants.NO_NONCE) {
629      size += Bytes.SIZEOF_LONG; // nonce group
630    }
631    if (nonce != HConstants.NO_NONCE) {
632      size += Bytes.SIZEOF_LONG; // nonce
633    }
634    if (replicationScope != null) {
635      for (Map.Entry<byte[], Integer> scope: replicationScope.entrySet()) {
636        size += scope.getKey().length;
637        size += Bytes.SIZEOF_INT;
638      }
639    }
640    size += Bytes.SIZEOF_LONG; // sequence number
641    size += Bytes.SIZEOF_LONG; // write time
642    if (origLogSeqNum > 0) {
643      size += Bytes.SIZEOF_LONG; // original sequence number
644    }
645    return size;
