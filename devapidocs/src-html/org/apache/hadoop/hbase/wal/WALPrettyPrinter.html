052/**
053 * WALPrettyPrinter prints the contents of a given WAL with a variety of
054 * options affecting formatting and extent of content.
055 *
056 * It targets two usage cases: pretty printing for ease of debugging directly by
057 * humans, and JSON output for consumption by monitoring and/or maintenance
058 * scripts.
059 *
060 * It can filter by row, region, or sequence id.
061 *
062 * It can also toggle output of values.
063 *
064 */
065@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
066@InterfaceStability.Evolving
067public class WALPrettyPrinter {
068  private boolean outputValues;
069  private boolean outputJSON;
070  // The following enable filtering by sequence, region, and row, respectively
071  private long sequence;
072  private String region;
073  private String row;
074  // enable in order to output a single list of transactions from several files
075  private boolean persistentOutput;
076  private boolean firstTxn;
077  // useful for programmatic capture of JSON output
078  private PrintStream out;
079  // for JSON encoding
080  private static final ObjectMapper MAPPER = new ObjectMapper();
081
082  /**
083   * Basic constructor that simply initializes values to reasonable defaults.
084   */
085  public WALPrettyPrinter() {
086    outputValues = false;
087    outputJSON = false;
088    sequence = -1;
089    region = null;
090    row = null;
091    persistentOutput = false;
092    firstTxn = true;
093    out = System.out;
094  }
095
096  /**
097   * Fully specified constructor.
098   *
099   * @param outputValues
100   *          when true, enables output of values along with other log
101   *          information
102   * @param outputJSON
103   *          when true, enables output in JSON format rather than a
104   *          "pretty string"
105   * @param sequence
106   *          when nonnegative, serves as a filter; only log entries with this
107   *          sequence id will be printed
108   * @param region
109   *          when not null, serves as a filter; only log entries from this
110   *          region will be printed
111   * @param row
112   *          when not null, serves as a filter; only log entries from this row
113   *          will be printed
114   * @param persistentOutput
115   *          keeps a single list running for multiple files. if enabled, the
116   *          endPersistentOutput() method must be used!
117   * @param out
118   *          Specifies an alternative to stdout for the destination of this
119   *          PrettyPrinter's output.
120   */
121  public WALPrettyPrinter(boolean outputValues, boolean outputJSON,
122      long sequence, String region, String row, boolean persistentOutput,
123      PrintStream out) {
124    this.outputValues = outputValues;
125    this.outputJSON = outputJSON;
126    this.sequence = sequence;
127    this.region = region;
128    this.row = row;
129    this.persistentOutput = persistentOutput;
130    if (persistentOutput) {
131      beginPersistentOutput();
132    }
133    this.out = out;
134    this.firstTxn = true;
135  }
136
137  /**
138   * turns value output on
139   */
140  public void enableValues() {
141    outputValues = true;
142  }
143
144  /**
145   * turns value output off
146   */
147  public void disableValues() {
148    outputValues = false;
149  }
150
151  /**
152   * turns JSON output on
153   */
154  public void enableJSON() {
155    outputJSON = true;
156  }
157
158  /**
159   * turns JSON output off, and turns on "pretty strings" for human consumption
160   */
161  public void disableJSON() {
162    outputJSON = false;
163  }
164
165  /**
166   * sets the region by which output will be filtered
167   *
168   * @param sequence
169   *          when nonnegative, serves as a filter; only log entries with this
170   *          sequence id will be printed
171   */
172  public void setSequenceFilter(long sequence) {
173    this.sequence = sequence;
174  }
175
176  /**
177   * sets the region by which output will be filtered
178   *
179   * @param region
180   *          when not null, serves as a filter; only log entries from this
181   *          region will be printed
182   */
183  public void setRegionFilter(String region) {
184    this.region = region;
185  }
186
187  /**
188   * sets the region by which output will be filtered
189   *
190   * @param row
191   *          when not null, serves as a filter; only log entries from this row
192   *          will be printed
193   */
194  public void setRowFilter(String row) {
195    this.row = row;
196  }
197
198  /**
199   * enables output as a single, persistent list. at present, only relevant in
200   * the case of JSON output.
201   */
202  public void beginPersistentOutput() {
203    if (persistentOutput)
204      return;
205    persistentOutput = true;
206    firstTxn = true;
207    if (outputJSON)
208      out.print("[");
209  }
210
211  /**
212   * ends output of a single, persistent list. at present, only relevant in the
213   * case of JSON output.
214   */
215  public void endPersistentOutput() {
216    if (!persistentOutput)
217      return;
218    persistentOutput = false;
219    if (outputJSON)
220      out.print("]");
221  }

223  /**
224   * reads a log file and outputs its contents, one transaction at a time, as
225   * specified by the currently configured options
226   *
227   * @param conf
228   *          the HBase configuration relevant to this log file
229   * @param p
230   *          the path of the log file to be read
231   * @throws IOException
232   *           may be unable to access the configured filesystem or requested
233   *           file.
234   */
235  public void processFile(final Configuration conf, final Path p)
236      throws IOException {
237    FileSystem fs = p.getFileSystem(conf);
238    if (!fs.exists(p)) {
239      throw new FileNotFoundException(p.toString());
240    }
241    if (!fs.isFile(p)) {
242      throw new IOException(p + " is not a file");
243    }
244
245    WAL.Reader log = WALFactory.createReader(fs, p, conf);
246
247    if (log instanceof ProtobufLogReader) {
248      List<String> writerClsNames = ((ProtobufLogReader) log).getWriterClsNames();
249      if (writerClsNames != null && writerClsNames.size() > 0) {
250        out.print("Writer Classes: ");
251        for (int i = 0; i < writerClsNames.size(); i++) {
252          out.print(writerClsNames.get(i));
253          if (i != writerClsNames.size() - 1) {
254            out.print(" ");
255          }
256        }
257        out.println();
258      }
259
260      String cellCodecClsName = ((ProtobufLogReader) log).getCodecClsName();
261      if (cellCodecClsName != null) {
262        out.println("Cell Codec Class: " + cellCodecClsName);
263      }
264    }
265
266    if (outputJSON && !persistentOutput) {
267      out.print("[");
268      firstTxn = true;
269    }
270
271    try {
272      WAL.Entry entry;
273      while ((entry = log.next()) != null) {
274        WALKey key = entry.getKey();
275        WALEdit edit = entry.getEdit();
276        // begin building a transaction structure
277        Map<String, Object> txn = key.toStringMap();
278        long writeTime = key.getWriteTime();
279        // check output filters
280        if (sequence >= 0 && ((Long) txn.get("sequence")) != sequence)
281          continue;
282        if (region != null && !((String) txn.get("region")).equals(region))
283          continue;
284        // initialize list into which we will store atomic actions
285        List<Map> actions = new ArrayList<>();
286        for (Cell cell : edit.getCells()) {
287          // add atomic operation to txn
288          Map<String, Object> op = new HashMap<>(toStringMap(cell));
289          if (outputValues) op.put("value", Bytes.toStringBinary(CellUtil.cloneValue(cell)));
290          // check row output filter
291          if (row == null || ((String) op.get("row")).equals(row)) {
292            actions.add(op);
293          }
294        }
295        if (actions.isEmpty())
296          continue;
297        txn.put("actions", actions);
298        if (outputJSON) {
299          // JSON output is a straightforward "toString" on the txn object
300          if (firstTxn)
301            firstTxn = false;
302          else
303            out.print(",");
304          // encode and print JSON
305          out.print(MAPPER.writeValueAsString(txn));
306        } else {
307          // Pretty output, complete with indentation by atomic action
308          out.println("Sequence=" + txn.get("sequence") + " "
309              + ", region=" + txn.get("region") + " at write timestamp=" + new Date(writeTime));
310          for (int i = 0; i < actions.size(); i++) {
311            Map op = actions.get(i);
312            out.println("row=" + op.get("row") +
313                ", column=" + op.get("family") + ":" + op.get("qualifier"));
314            if (op.get("tag") != null) {
315              out.println("    tag: " + op.get("tag"));
316            }
317            if (outputValues) out.println("    value: " + op.get("value"));
318          }
319        }
320      }
321    } finally {
322      log.close();
323    }
324    if (outputJSON && !persistentOutput) {
325      out.print("]");
326    }
327  }

329  private static Map<String, Object> toStringMap(Cell cell) {
330    Map<String, Object> stringMap = new HashMap<>();
331    stringMap.put("row",
332        Bytes.toStringBinary(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()));
333    stringMap.put("family", Bytes.toStringBinary(cell.getFamilyArray(), cell.getFamilyOffset(),
334                cell.getFamilyLength()));
335    stringMap.put("qualifier",
336        Bytes.toStringBinary(cell.getQualifierArray(), cell.getQualifierOffset(),
337            cell.getQualifierLength()));
338    stringMap.put("timestamp", cell.getTimestamp());
339    stringMap.put("vlen", cell.getValueLength());
340    if (cell.getTagsLength() > 0) {
341      List<String> tagsString = new ArrayList<>();
342      Iterator<Tag> tagsIterator = CellUtil.tagsIterator(cell);
343      while (tagsIterator.hasNext()) {
344        Tag tag = tagsIterator.next();
345        tagsString.add((tag.getType()) + ":" + Bytes.toStringBinary(TagUtil.cloneValue(tag)));
346      }
347      stringMap.put("tag", tagsString);
348    }
349    return stringMap;
350  }

352  public static void main(String[] args) throws IOException {
353    run(args);

356  /**
357   * Pass one or more log file names and formatting options and it will dump out
358   * a text version of the contents on <code>stdout</code>.
359   *
360   * @param args
361   *          Command line arguments
362   * @throws IOException
363   *           Thrown upon file system errors etc.
364   */
365  public static void run(String[] args) throws IOException {
366    // create options
367    Options options = new Options();
368    options.addOption("h", "help", false, "Output help message");
369    options.addOption("j", "json", false, "Output JSON");
370    options.addOption("p", "printvals", false, "Print values");
371    options.addOption("r", "region", true,
372        "Region to filter by. Pass encoded region name; e.g. '9192caead6a5a20acb4454ffbc79fa14'");
373    options.addOption("s", "sequence", true,
374        "Sequence to filter by. Pass sequence number.");
375    options.addOption("w", "row", true, "Row to filter by. Pass row name.");
376
377    WALPrettyPrinter printer = new WALPrettyPrinter();
378    CommandLineParser parser = new PosixParser();
379    List<?> files = null;
380    try {
381      CommandLine cmd = parser.parse(options, args);
382      files = cmd.getArgList();
383      if (files.isEmpty() || cmd.hasOption("h")) {
384        HelpFormatter formatter = new HelpFormatter();
385        formatter.printHelp("WAL <filename...>", options, true);
386        System.exit(-1);
387      }
388      // configure the pretty printer using command line options
389      if (cmd.hasOption("p"))
390        printer.enableValues();
391      if (cmd.hasOption("j"))
392        printer.enableJSON();
393      if (cmd.hasOption("r"))
394        printer.setRegionFilter(cmd.getOptionValue("r"));
395      if (cmd.hasOption("s"))
396        printer.setSequenceFilter(Long.parseLong(cmd.getOptionValue("s")));
397      if (cmd.hasOption("w"))
398        printer.setRowFilter(cmd.getOptionValue("w"));
399    } catch (ParseException e) {
400      e.printStackTrace();
401      HelpFormatter formatter = new HelpFormatter();
402      formatter.printHelp("HFile filename(s) ", options, true);
403      System.exit(-1);
404    }
405    // get configuration, file system, and process the given files
406    Configuration conf = HBaseConfiguration.create();
407    FSUtils.setFsDefault(conf, FSUtils.getRootDir(conf));
408
409    // begin output
410    printer.beginPersistentOutput();
411    for (Object f : files) {
412      Path file = new Path((String) f);
413      FileSystem fs = file.getFileSystem(conf);
414      if (!fs.exists(file)) {
415        System.err.println("ERROR, file doesnt exist: " + file);
416        return;
417      }
418      printer.processFile(conf, file);
419    }
420    printer.endPersistentOutput();
421  }
