082  /**
083   * Completes the work done by splitLogFile by archiving logs
084   * <p>
085   * It is invoked by SplitLogManager once it knows that one of the SplitLogWorkers have completed
086   * the splitLogFile() part. If the master crashes then this function might get called multiple
087   * times.
088   * <p>
089   * @param logfile
090   * @param conf
091   * @throws IOException
092   */
093  public static void finishSplitLogFile(String logfile, Configuration conf) throws IOException {
094    Path walDir = FSUtils.getWALRootDir(conf);
095    Path oldLogDir = new Path(walDir, HConstants.HREGION_OLDLOGDIR_NAME);
096    Path walPath;
097    if (FSUtils.isStartingWithPath(walDir, logfile)) {
098      walPath = new Path(logfile);
099    } else {
100      walPath = new Path(walDir, logfile);
101    }
102    finishSplitLogFile(walDir, oldLogDir, walPath, conf);
103  }
104
105  static void finishSplitLogFile(Path walDir, Path oldWALDir, Path walPath,
106      Configuration conf) throws IOException {
107    List<Path> processedLogs = new ArrayList<>();
108    List<Path> corruptedLogs = new ArrayList<>();
109    FileSystem walFS = walDir.getFileSystem(conf);
110    if (ZKSplitLog.isCorrupted(walDir, walPath.getName(), walFS)) {
111      corruptedLogs.add(walPath);
112    } else {
113      processedLogs.add(walPath);
114    }
115    archiveWALs(corruptedLogs, processedLogs, oldWALDir, walFS, conf);
116    Path stagingDir = ZKSplitLog.getSplitLogDir(walDir, walPath.getName());
117    walFS.delete(stagingDir, true);
118  }

120  /**
121   * Moves processed logs to a oldLogDir after successful processing Moves corrupted logs (any log
122   * that couldn't be successfully parsed to corruptDir (.corrupt) for later investigation
123   */
124  private static void archiveWALs(final List<Path> corruptedWALs, final List<Path> processedWALs,
125      final Path oldWALDir, final FileSystem walFS, final Configuration conf) throws IOException {
126    final Path corruptDir = new Path(FSUtils.getWALRootDir(conf), HConstants.CORRUPT_DIR_NAME);
127    if (conf.get("hbase.regionserver.hlog.splitlog.corrupt.dir") != null) {
128      LOG.warn("hbase.regionserver.hlog.splitlog.corrupt.dir is deprecated. Default to {}",
129        corruptDir);
130    }
131    if (!walFS.mkdirs(corruptDir)) {
132      LOG.info("Unable to mkdir {}", corruptDir);
133    }
134    walFS.mkdirs(oldWALDir);
135
136    // this method can get restarted or called multiple times for archiving
137    // the same log files.
138    for (Path corruptedWAL : corruptedWALs) {
139      Path p = new Path(corruptDir, corruptedWAL.getName());
140      if (walFS.exists(corruptedWAL)) {
141        if (!walFS.rename(corruptedWAL, p)) {
142          LOG.warn("Unable to move corrupted log {} to {}", corruptedWAL, p);
143        } else {
144          LOG.warn("Moved corrupted log {} to {}", corruptedWAL, p);
145        }
146      }
147    }
148
149    for (Path p : processedWALs) {
150      Path newPath = AbstractFSWAL.getWALArchivePath(oldWALDir, p);
151      if (walFS.exists(p)) {
152        if (!FSUtils.renameAndSetModifyTime(walFS, p, newPath)) {
153          LOG.warn("Unable to move {} to {}", p, newPath);
154        } else {
155          LOG.info("Archived processed log {} to {}", p, newPath);
156        }
157      }
158    }
159  }

161  /**
162   * Path to a file under RECOVERED_EDITS_DIR directory of the region found in <code>logEntry</code>
163   * named for the sequenceid in the passed <code>logEntry</code>: e.g.
164   * /hbase/some_table/2323432434/recovered.edits/2332. This method also ensures existence of
165   * RECOVERED_EDITS_DIR under the region creating it if necessary.
166   * @param walEntry walEntry to recover
167   * @param fileNameBeingSplit the file being split currently. Used to generate tmp file name.
168   * @param tmpDirName of the directory used to sideline old recovered edits file
169   * @param conf configuration
170   * @return Path to file into which to dump split log edits.
171   * @throws IOException
172   */
173  @SuppressWarnings("deprecation")
174  @VisibleForTesting
175  static Path getRegionSplitEditsPath(final WAL.Entry walEntry, String fileNameBeingSplit,
176      String tmpDirName, Configuration conf) throws IOException {
177    FileSystem walFS = FSUtils.getWALFileSystem(conf);
178    Path tableDir = FSUtils.getWALTableDir(conf, walEntry.getKey().getTableName());
179    String encodedRegionName = Bytes.toString(walEntry.getKey().getEncodedRegionName());
180    Path regionDir = HRegion.getRegionDir(tableDir, encodedRegionName);
181    Path dir = getRegionDirRecoveredEditsDir(regionDir);
182
183    if (walFS.exists(dir) && walFS.isFile(dir)) {
184      Path tmp = new Path(tmpDirName);
185      if (!walFS.exists(tmp)) {
186        walFS.mkdirs(tmp);
187      }
188      tmp = new Path(tmp, HConstants.RECOVERED_EDITS_DIR + "_" + encodedRegionName);
189      LOG.warn("Found existing old file: {}. It could be some "
190          + "leftover of an old installation. It should be a folder instead. "
191          + "So moving it to {}",
192        dir, tmp);
193      if (!walFS.rename(dir, tmp)) {
194        LOG.warn("Failed to sideline old file {}", dir);
195      }
196    }
197
198    if (!walFS.exists(dir) && !walFS.mkdirs(dir)) {
199      LOG.warn("mkdir failed on {}", dir);
200    }
201    // Append fileBeingSplit to prevent name conflict since we may have duplicate wal entries now.
202    // Append file name ends with RECOVERED_LOG_TMPFILE_SUFFIX to ensure
203    // region's replayRecoveredEdits will not delete it
204    String fileName = formatRecoveredEditsFileName(walEntry.getKey().getSequenceId());
205    fileName = getTmpRecoveredEditsFileName(fileName + "-" + fileNameBeingSplit);
206    return new Path(dir, fileName);
207  }
208
209  private static String getTmpRecoveredEditsFileName(String fileName) {
210    return fileName + RECOVERED_LOG_TMPFILE_SUFFIX;
211  }

237  /**
238   * Check whether there is recovered.edits in the region dir
239   * @param conf conf
240   * @param regionInfo the region to check
241   * @return true if recovered.edits exist in the region dir
242   */
243  public static boolean hasRecoveredEdits(final Configuration conf, final RegionInfo regionInfo)
244      throws IOException {
245    // No recovered.edits for non default replica regions
246    if (regionInfo.getReplicaId() != RegionInfo.DEFAULT_REPLICA_ID) {
247      return false;
248    }
249    // Only default replica region can reach here, so we can use regioninfo
250    // directly without converting it to default replica's regioninfo.
251    Path regionWALDir =
252        FSUtils.getWALRegionDir(conf, regionInfo.getTable(), regionInfo.getEncodedName());
253    Path regionDir = FSUtils.getRegionDirFromRootDir(FSUtils.getRootDir(conf), regionInfo);
254    Path wrongRegionWALDir =
255        FSUtils.getWrongWALRegionDir(conf, regionInfo.getTable(), regionInfo.getEncodedName());
256    FileSystem walFs = FSUtils.getWALFileSystem(conf);
257    FileSystem rootFs = FSUtils.getRootDirFileSystem(conf);
258    NavigableSet<Path> files = getSplitEditFilesSorted(walFs, regionWALDir);
259    if (!files.isEmpty()) {
260      return true;
261    }
262    files = getSplitEditFilesSorted(rootFs, regionDir);
263    if (!files.isEmpty()) {
264      return true;
265    }
266    files = getSplitEditFilesSorted(walFs, wrongRegionWALDir);
267    return !files.isEmpty();
268  }

270  /**
271   * This method will check 3 places for finding the max sequence id file. One is the expected
272   * place, another is the old place under the region directory, and the last one is the wrong one
273   * we introduced in HBASE-20734. See HBASE-22617 for more details.
274   * <p/>
275   * Notice that, you should always call this method instead of
276   * {@link #getMaxRegionSequenceId(FileSystem, Path)} until 4.0.0 release.
277   * @deprecated Only for compatibility, will be removed in 4.0.0.
278   */
279  @Deprecated
280  public static long getMaxRegionSequenceId(Configuration conf, RegionInfo region,
281      IOExceptionSupplier<FileSystem> rootFsSupplier, IOExceptionSupplier<FileSystem> walFsSupplier)
282      throws IOException {
283    FileSystem rootFs = rootFsSupplier.get();
284    FileSystem walFs = walFsSupplier.get();
285    Path regionWALDir = FSUtils.getWALRegionDir(conf, region.getTable(), region.getEncodedName());
286    // This is the old place where we store max sequence id file
287    Path regionDir = FSUtils.getRegionDirFromRootDir(FSUtils.getRootDir(conf), region);
288    // This is for HBASE-20734, where we use a wrong directory, see HBASE-22617 for more details.
289    Path wrongRegionWALDir =
290      FSUtils.getWrongWALRegionDir(conf, region.getTable(), region.getEncodedName());
291    long maxSeqId = getMaxRegionSequenceId(walFs, regionWALDir);
292    maxSeqId = Math.max(maxSeqId, getMaxRegionSequenceId(rootFs, regionDir));
293    maxSeqId = Math.max(maxSeqId, getMaxRegionSequenceId(walFs, wrongRegionWALDir));
294    return maxSeqId;
295  }

297  /**
298   * Returns sorted set of edit files made by splitter, excluding files with '.temp' suffix.
299   * @param walFS WAL FileSystem used to retrieving split edits files.
300   * @param regionDir WAL region dir to look for recovered edits files under.
301   * @return Files in passed <code>regionDir</code> as a sorted set.
302   * @throws IOException
303   */
304  public static NavigableSet<Path> getSplitEditFilesSorted(final FileSystem walFS,
305      final Path regionDir) throws IOException {
306    NavigableSet<Path> filesSorted = new TreeSet<>();
307    Path editsdir = getRegionDirRecoveredEditsDir(regionDir);
308    if (!walFS.exists(editsdir)) {
309      return filesSorted;
310    }
311    FileStatus[] files = FSUtils.listStatus(walFS, editsdir, new PathFilter() {
312      @Override
313      public boolean accept(Path p) {
314        boolean result = false;
315        try {
316          // Return files and only files that match the editfile names pattern.
317          // There can be other files in this directory other than edit files.
318          // In particular, on error, we'll move aside the bad edit file giving
319          // it a timestamp suffix. See moveAsideBadEditsFile.
320          Matcher m = EDITFILES_NAME_PATTERN.matcher(p.getName());
321          result = walFS.isFile(p) && m.matches();
322          // Skip the file whose name ends with RECOVERED_LOG_TMPFILE_SUFFIX,
323          // because it means splitwal thread is writting this file.
324          if (p.getName().endsWith(RECOVERED_LOG_TMPFILE_SUFFIX)) {
325            result = false;
326          }
327          // Skip SeqId Files
328          if (isSequenceIdFile(p)) {
329            result = false;
330          }
331        } catch (IOException e) {
332          LOG.warn("Failed isFile check on {}", p, e);
333        }
334        return result;
335      }
336    });
337    if (ArrayUtils.isNotEmpty(files)) {
338      Arrays.asList(files).forEach(status -> filesSorted.add(status.getPath()));
339    }
340    return filesSorted;
341  }

343  /**
344   * Move aside a bad edits file.
345   * @param walFS WAL FileSystem used to rename bad edits file.
346   * @param edits Edits file to move aside.
347   * @return The name of the moved aside file.
348   * @throws IOException
349   */
350  public static Path moveAsideBadEditsFile(final FileSystem walFS, final Path edits)
351      throws IOException {
352    Path moveAsideName =
353        new Path(edits.getParent(), edits.getName() + "." + System.currentTimeMillis());
354    if (!walFS.rename(edits, moveAsideName)) {
355      LOG.warn("Rename failed from {} to {}", edits, moveAsideName);
356    }
357    return moveAsideName;
358  }

369  private static FileStatus[] getSequenceIdFiles(FileSystem walFS, Path regionDir)
370      throws IOException {
371    // TODO: Why are we using a method in here as part of our normal region open where
372    // there is no splitting involved? Fix. St.Ack 01/20/2017.
373    Path editsDir = getRegionDirRecoveredEditsDir(regionDir);
374    try {
375      FileStatus[] files = walFS.listStatus(editsDir, WALSplitUtil::isSequenceIdFile);
376      return files != null ? files : new FileStatus[0];
377    } catch (FileNotFoundException e) {
378      return new FileStatus[0];
379    }
380  }

382  private static long getMaxSequenceId(FileStatus[] files) {
383    long maxSeqId = -1L;
384    for (FileStatus file : files) {
385      String fileName = file.getPath().getName();
386      try {
387        maxSeqId = Math.max(maxSeqId, Long
388            .parseLong(fileName.substring(0, fileName.length() - SEQUENCE_ID_FILE_SUFFIX_LENGTH)));
389      } catch (NumberFormatException ex) {
390        LOG.warn("Invalid SeqId File Name={}", fileName);
391      }
392    }
393    return maxSeqId;
394  }

403  /**
404   * Create a file with name as region's max sequence id
405   */
406  public static void writeRegionSequenceIdFile(FileSystem walFS, Path regionDir, long newMaxSeqId)
407      throws IOException {
408    FileStatus[] files = getSequenceIdFiles(walFS, regionDir);
409    long maxSeqId = getMaxSequenceId(files);
410    if (maxSeqId > newMaxSeqId) {
411      throw new IOException("The new max sequence id " + newMaxSeqId
412          + " is less than the old max sequence id " + maxSeqId);
413    }
414    // write a new seqId file
415    Path newSeqIdFile =
416        new Path(getRegionDirRecoveredEditsDir(regionDir), newMaxSeqId + SEQUENCE_ID_FILE_SUFFIX);
417    if (newMaxSeqId != maxSeqId) {
418      try {
419        if (!walFS.createNewFile(newSeqIdFile) && !walFS.exists(newSeqIdFile)) {
420          throw new IOException("Failed to create SeqId file:" + newSeqIdFile);
421        }
422        LOG.debug("Wrote file={}, newMaxSeqId={}, maxSeqId={}", newSeqIdFile, newMaxSeqId,
423          maxSeqId);
424      } catch (FileAlreadyExistsException ignored) {
425        // latest hdfs throws this exception. it's all right if newSeqIdFile already exists
426      }
427    }
428    // remove old ones
429    for (FileStatus status : files) {
430      if (!newSeqIdFile.equals(status.getPath())) {
431        walFS.delete(status.getPath(), false);
432      }
433    }
434  }

436  /** A struct used by getMutationsFromWALEntry */
437  public static class MutationReplay implements Comparable<MutationReplay> {
438    public MutationReplay(ClientProtos.MutationProto.MutationType type, Mutation mutation,
439        long nonceGroup, long nonce) {
440      this.type = type;
441      this.mutation = mutation;
442      if (this.mutation.getDurability() != Durability.SKIP_WAL) {
443        // using ASYNC_WAL for relay
444        this.mutation.setDurability(Durability.ASYNC_WAL);
445      }
446      this.nonceGroup = nonceGroup;
447      this.nonce = nonce;
448    }
449
450    private final ClientProtos.MutationProto.MutationType type;
451    public final Mutation mutation;
452    public final long nonceGroup;
453    public final long nonce;
454
455    @Override
456    public int compareTo(final MutationReplay d) {
457      return Row.COMPARATOR.compare(mutation, d.mutation);
458    }
459
460    @Override
461    public boolean equals(Object obj) {
462      if (!(obj instanceof MutationReplay)) {
463        return false;
464      } else {
465        return this.compareTo((MutationReplay) obj) == 0;
466      }
467    }
468
469    @Override
470    public int hashCode() {
471      return this.mutation.hashCode();
472    }
473
474    public ClientProtos.MutationProto.MutationType getType() {
475      return type;
476    }
477  }

479  /**
480   * This function is used to construct mutations from a WALEntry. It also reconstructs WALKey &amp;
481   * WALEdit from the passed in WALEntry
482   * @param entry
483   * @param cells
484   * @param logEntry pair of WALKey and WALEdit instance stores WALKey and WALEdit instances
485   *          extracted from the passed in WALEntry.
486   * @return list of Pair&lt;MutationType, Mutation&gt; to be replayed
487   * @throws IOException
488   */
489  public static List<MutationReplay> getMutationsFromWALEntry(AdminProtos.WALEntry entry,
490      CellScanner cells, Pair<WALKey, WALEdit> logEntry, Durability durability) throws IOException {
491    if (entry == null) {
492      // return an empty array
493      return Collections.emptyList();
494    }
495
496    long replaySeqId =
497        (entry.getKey().hasOrigSequenceNumber()) ? entry.getKey().getOrigSequenceNumber()
498            : entry.getKey().getLogSequenceNumber();
499    int count = entry.getAssociatedCellCount();
500    List<MutationReplay> mutations = new ArrayList<>();
501    Cell previousCell = null;
502    Mutation m = null;
503    WALKeyImpl key = null;
504    WALEdit val = null;
505    if (logEntry != null) {
506      val = new WALEdit();
507    }
508
509    for (int i = 0; i < count; i++) {
510      // Throw index out of bounds if our cell count is off
511      if (!cells.advance()) {
512        throw new ArrayIndexOutOfBoundsException("Expected=" + count + ", index=" + i);
513      }
514      Cell cell = cells.current();
515      if (val != null) val.add(cell);
516
517      boolean isNewRowOrType =
518          previousCell == null || previousCell.getTypeByte() != cell.getTypeByte()
519              || !CellUtil.matchingRows(previousCell, cell);
520      if (isNewRowOrType) {
521        // Create new mutation
522        if (CellUtil.isDelete(cell)) {
523          m = new Delete(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
524          // Deletes don't have nonces.
525          mutations.add(new MutationReplay(ClientProtos.MutationProto.MutationType.DELETE, m,
526              HConstants.NO_NONCE, HConstants.NO_NONCE));
527        } else {
528          m = new Put(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
529          // Puts might come from increment or append, thus we need nonces.
530          long nonceGroup =
531              entry.getKey().hasNonceGroup() ? entry.getKey().getNonceGroup() : HConstants.NO_NONCE;
532          long nonce = entry.getKey().hasNonce() ? entry.getKey().getNonce() : HConstants.NO_NONCE;
533          mutations.add(
534            new MutationReplay(ClientProtos.MutationProto.MutationType.PUT, m, nonceGroup, nonce));
535        }
536      }
537      if (CellUtil.isDelete(cell)) {
538        ((Delete) m).add(cell);
539      } else {
540        ((Put) m).add(cell);
541      }
542      m.setDurability(durability);
543      previousCell = cell;
544    }
545
546    // reconstruct WALKey
547    if (logEntry != null) {
548      org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos.WALKey walKeyProto =
549          entry.getKey();
550      List<UUID> clusterIds = new ArrayList<>(walKeyProto.getClusterIdsCount());
551      for (HBaseProtos.UUID uuid : entry.getKey().getClusterIdsList()) {
552        clusterIds.add(new UUID(uuid.getMostSigBits(), uuid.getLeastSigBits()));
553      }
554      key = new WALKeyImpl(walKeyProto.getEncodedRegionName().toByteArray(),
555          TableName.valueOf(walKeyProto.getTableName().toByteArray()), replaySeqId,
556          walKeyProto.getWriteTime(), clusterIds, walKeyProto.getNonceGroup(),
557          walKeyProto.getNonce(), null);
558      logEntry.setFirst(key);
559      logEntry.setSecond(val);
560    }
561
562    return mutations;
563  }
