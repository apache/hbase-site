083  /**
084   * Completes the work done by splitLogFile by archiving logs
085   * <p>
086   * It is invoked by SplitLogManager once it knows that one of the SplitLogWorkers have completed
087   * the splitLogFile() part. If the master crashes then this function might get called multiple
088   * times.
089   * <p>
090   * @param logfile
091   * @param conf
092   * @throws IOException
093   */
094  public static void finishSplitLogFile(String logfile, Configuration conf) throws IOException {
095    Path walDir = FSUtils.getWALRootDir(conf);
096    Path oldLogDir = new Path(walDir, HConstants.HREGION_OLDLOGDIR_NAME);
097    Path walPath;
098    if (FSUtils.isStartingWithPath(walDir, logfile)) {
099      walPath = new Path(logfile);
100    } else {
101      walPath = new Path(walDir, logfile);
102    }
103    finishSplitLogFile(walDir, oldLogDir, walPath, conf);
104  }
105
106  static void finishSplitLogFile(Path walDir, Path oldWALDir, Path walPath,
107      Configuration conf) throws IOException {
108    List<Path> processedLogs = new ArrayList<>();
109    List<Path> corruptedLogs = new ArrayList<>();
110    FileSystem walFS = walDir.getFileSystem(conf);
111    if (ZKSplitLog.isCorrupted(walDir, walPath.getName(), walFS)) {
112      corruptedLogs.add(walPath);
113    } else {
114      processedLogs.add(walPath);
115    }
116    archiveWALs(corruptedLogs, processedLogs, oldWALDir, walFS, conf);
117    Path stagingDir = ZKSplitLog.getSplitLogDir(walDir, walPath.getName());
118    walFS.delete(stagingDir, true);
119  }

121  /**
122   * Moves processed logs to a oldLogDir after successful processing Moves corrupted logs (any log
123   * that couldn't be successfully parsed to corruptDir (.corrupt) for later investigation
124   */
125  private static void archiveWALs(final List<Path> corruptedWALs, final List<Path> processedWALs,
126      final Path oldWALDir, final FileSystem walFS, final Configuration conf) throws IOException {
127    final Path corruptDir = new Path(FSUtils.getWALRootDir(conf), HConstants.CORRUPT_DIR_NAME);
128    if (conf.get("hbase.regionserver.hlog.splitlog.corrupt.dir") != null) {
129      LOG.warn("hbase.regionserver.hlog.splitlog.corrupt.dir is deprecated. Default to {}",
130        corruptDir);
131    }
132    if (!walFS.mkdirs(corruptDir)) {
133      LOG.info("Unable to mkdir {}", corruptDir);
134    }
135    walFS.mkdirs(oldWALDir);
136
137    // this method can get restarted or called multiple times for archiving
138    // the same log files.
139    for (Path corruptedWAL : corruptedWALs) {
140      Path p = new Path(corruptDir, corruptedWAL.getName());
141      if (walFS.exists(corruptedWAL)) {
142        if (!walFS.rename(corruptedWAL, p)) {
143          LOG.warn("Unable to move corrupted log {} to {}", corruptedWAL, p);
144        } else {
145          LOG.warn("Moved corrupted log {} to {}", corruptedWAL, p);
146        }
147      }
148    }
149
150    for (Path p : processedWALs) {
151      Path newPath = AbstractFSWAL.getWALArchivePath(oldWALDir, p);
152      if (walFS.exists(p)) {
153        if (!FSUtils.renameAndSetModifyTime(walFS, p, newPath)) {
154          LOG.warn("Unable to move {} to {}", p, newPath);
155        } else {
156          LOG.info("Archived processed log {} to {}", p, newPath);
157        }
158      }
159    }
160  }

162  /**
163   * Path to a file under RECOVERED_EDITS_DIR directory of the region found in <code>logEntry</code>
164   * named for the sequenceid in the passed <code>logEntry</code>: e.g.
165   * /hbase/some_table/2323432434/recovered.edits/2332. This method also ensures existence of
166   * RECOVERED_EDITS_DIR under the region creating it if necessary.
167   * @param tableName the table name
168   * @param encodedRegionName the encoded region name
169   * @param seqId the sequence id which used to generate file name
170   * @param fileNameBeingSplit the file being split currently. Used to generate tmp file name.
171   * @param tmpDirName of the directory used to sideline old recovered edits file
172   * @param conf configuration
173   * @return Path to file into which to dump split log edits.
174   * @throws IOException
175   */
176  @SuppressWarnings("deprecation")
177  @VisibleForTesting
178  static Path getRegionSplitEditsPath(TableName tableName, byte[] encodedRegionName, long seqId,
179      String fileNameBeingSplit, String tmpDirName, Configuration conf) throws IOException {
180    FileSystem walFS = FSUtils.getWALFileSystem(conf);
181    Path tableDir = FSUtils.getWALTableDir(conf, tableName);
182    String encodedRegionNameStr = Bytes.toString(encodedRegionName);
183    Path regionDir = HRegion.getRegionDir(tableDir, encodedRegionNameStr);
184    Path dir = getRegionDirRecoveredEditsDir(regionDir);
185
186    if (walFS.exists(dir) && walFS.isFile(dir)) {
187      Path tmp = new Path(tmpDirName);
188      if (!walFS.exists(tmp)) {
189        walFS.mkdirs(tmp);
190      }
191      tmp = new Path(tmp, HConstants.RECOVERED_EDITS_DIR + "_" + encodedRegionNameStr);
192      LOG.warn("Found existing old file: {}. It could be some "
193          + "leftover of an old installation. It should be a folder instead. "
194          + "So moving it to {}",
195        dir, tmp);
196      if (!walFS.rename(dir, tmp)) {
197        LOG.warn("Failed to sideline old file {}", dir);
198      }
199    }
200
201    if (!walFS.exists(dir) && !walFS.mkdirs(dir)) {
202      LOG.warn("mkdir failed on {}", dir);
203    }
204    // Append fileBeingSplit to prevent name conflict since we may have duplicate wal entries now.
205    // Append file name ends with RECOVERED_LOG_TMPFILE_SUFFIX to ensure
206    // region's replayRecoveredEdits will not delete it
207    String fileName = formatRecoveredEditsFileName(seqId);
208    fileName = getTmpRecoveredEditsFileName(fileName + "-" + fileNameBeingSplit);
209    return new Path(dir, fileName);
210  }

240  /**
241   * Check whether there is recovered.edits in the region dir
242   * @param conf conf
243   * @param regionInfo the region to check
244   * @return true if recovered.edits exist in the region dir
245   */
246  public static boolean hasRecoveredEdits(final Configuration conf, final RegionInfo regionInfo)
247      throws IOException {
248    // No recovered.edits for non default replica regions
249    if (regionInfo.getReplicaId() != RegionInfo.DEFAULT_REPLICA_ID) {
250      return false;
251    }
252    // Only default replica region can reach here, so we can use regioninfo
253    // directly without converting it to default replica's regioninfo.
254    Path regionWALDir =
255        FSUtils.getWALRegionDir(conf, regionInfo.getTable(), regionInfo.getEncodedName());
256    Path regionDir = FSUtils.getRegionDirFromRootDir(FSUtils.getRootDir(conf), regionInfo);
257    Path wrongRegionWALDir =
258        FSUtils.getWrongWALRegionDir(conf, regionInfo.getTable(), regionInfo.getEncodedName());
259    FileSystem walFs = FSUtils.getWALFileSystem(conf);
260    FileSystem rootFs = FSUtils.getRootDirFileSystem(conf);
261    NavigableSet<Path> files = getSplitEditFilesSorted(walFs, regionWALDir);
262    if (!files.isEmpty()) {
263      return true;
264    }
265    files = getSplitEditFilesSorted(rootFs, regionDir);
266    if (!files.isEmpty()) {
267      return true;
268    }
269    files = getSplitEditFilesSorted(walFs, wrongRegionWALDir);
270    return !files.isEmpty();
271  }

273  /**
274   * This method will check 3 places for finding the max sequence id file. One is the expected
275   * place, another is the old place under the region directory, and the last one is the wrong one
276   * we introduced in HBASE-20734. See HBASE-22617 for more details.
277   * <p/>
278   * Notice that, you should always call this method instead of
279   * {@link #getMaxRegionSequenceId(FileSystem, Path)} until 4.0.0 release.
280   * @deprecated Only for compatibility, will be removed in 4.0.0.
281   */
282  @Deprecated
283  public static long getMaxRegionSequenceId(Configuration conf, RegionInfo region,
284      IOExceptionSupplier<FileSystem> rootFsSupplier, IOExceptionSupplier<FileSystem> walFsSupplier)
285      throws IOException {
286    FileSystem rootFs = rootFsSupplier.get();
287    FileSystem walFs = walFsSupplier.get();
288    Path regionWALDir = FSUtils.getWALRegionDir(conf, region.getTable(), region.getEncodedName());
289    // This is the old place where we store max sequence id file
290    Path regionDir = FSUtils.getRegionDirFromRootDir(FSUtils.getRootDir(conf), region);
291    // This is for HBASE-20734, where we use a wrong directory, see HBASE-22617 for more details.
292    Path wrongRegionWALDir =
293      FSUtils.getWrongWALRegionDir(conf, region.getTable(), region.getEncodedName());
294    long maxSeqId = getMaxRegionSequenceId(walFs, regionWALDir);
295    maxSeqId = Math.max(maxSeqId, getMaxRegionSequenceId(rootFs, regionDir));
296    maxSeqId = Math.max(maxSeqId, getMaxRegionSequenceId(walFs, wrongRegionWALDir));
297    return maxSeqId;
298  }

300  /**
301   * Returns sorted set of edit files made by splitter, excluding files with '.temp' suffix.
302   * @param walFS WAL FileSystem used to retrieving split edits files.
303   * @param regionDir WAL region dir to look for recovered edits files under.
304   * @return Files in passed <code>regionDir</code> as a sorted set.
305   * @throws IOException
306   */
307  public static NavigableSet<Path> getSplitEditFilesSorted(final FileSystem walFS,
308      final Path regionDir) throws IOException {
309    NavigableSet<Path> filesSorted = new TreeSet<>();
310    Path editsdir = getRegionDirRecoveredEditsDir(regionDir);
311    if (!walFS.exists(editsdir)) {
312      return filesSorted;
313    }
314    FileStatus[] files = FSUtils.listStatus(walFS, editsdir, new PathFilter() {
315      @Override
316      public boolean accept(Path p) {
317        boolean result = false;
318        try {
319          // Return files and only files that match the editfile names pattern.
320          // There can be other files in this directory other than edit files.
321          // In particular, on error, we'll move aside the bad edit file giving
322          // it a timestamp suffix. See moveAsideBadEditsFile.
323          Matcher m = EDITFILES_NAME_PATTERN.matcher(p.getName());
324          result = walFS.isFile(p) && m.matches();
325          // Skip the file whose name ends with RECOVERED_LOG_TMPFILE_SUFFIX,
326          // because it means splitwal thread is writting this file.
327          if (p.getName().endsWith(RECOVERED_LOG_TMPFILE_SUFFIX)) {
328            result = false;
329          }
330          // Skip SeqId Files
331          if (isSequenceIdFile(p)) {
332            result = false;
333          }
334        } catch (IOException e) {
335          LOG.warn("Failed isFile check on {}", p, e);
336        }
337        return result;
338      }
339    });
340    if (ArrayUtils.isNotEmpty(files)) {
341      Arrays.asList(files).forEach(status -> filesSorted.add(status.getPath()));
342    }
343    return filesSorted;
344  }

347   * Move aside a bad edits file.
348   * @param fs the file system used to rename bad edits file.
349   * @param edits Edits file to move aside.
350   * @return The name of the moved aside file.
351   * @throws IOException

353  public static Path moveAsideBadEditsFile(final FileSystem fs, final Path edits)
354      throws IOException {
355    Path moveAsideName =
356        new Path(edits.getParent(), edits.getName() + "." + System.currentTimeMillis());

358      LOG.warn("Rename failed from {} to {}", edits, moveAsideName);
359    }
360    return moveAsideName;

372  private static FileStatus[] getSequenceIdFiles(FileSystem walFS, Path regionDir)
373      throws IOException {
374    // TODO: Why are we using a method in here as part of our normal region open where
375    // there is no splitting involved? Fix. St.Ack 01/20/2017.
376    Path editsDir = getRegionDirRecoveredEditsDir(regionDir);
377    try {
378      FileStatus[] files = walFS.listStatus(editsDir, WALSplitUtil::isSequenceIdFile);
379      return files != null ? files : new FileStatus[0];
380    } catch (FileNotFoundException e) {
381      return new FileStatus[0];
382    }
383  }

385  private static long getMaxSequenceId(FileStatus[] files) {
386    long maxSeqId = -1L;
387    for (FileStatus file : files) {
388      String fileName = file.getPath().getName();
389      try {
390        maxSeqId = Math.max(maxSeqId, Long
391            .parseLong(fileName.substring(0, fileName.length() - SEQUENCE_ID_FILE_SUFFIX_LENGTH)));
392      } catch (NumberFormatException ex) {
393        LOG.warn("Invalid SeqId File Name={}", fileName);
394      }
395    }
396    return maxSeqId;
397  }

406  /**
407   * Create a file with name as region's max sequence id
408   */
409  public static void writeRegionSequenceIdFile(FileSystem walFS, Path regionDir, long newMaxSeqId)
410      throws IOException {
411    FileStatus[] files = getSequenceIdFiles(walFS, regionDir);
412    long maxSeqId = getMaxSequenceId(files);
413    if (maxSeqId > newMaxSeqId) {
414      throw new IOException("The new max sequence id " + newMaxSeqId
415          + " is less than the old max sequence id " + maxSeqId);
416    }
417    // write a new seqId file
418    Path newSeqIdFile =
419        new Path(getRegionDirRecoveredEditsDir(regionDir), newMaxSeqId + SEQUENCE_ID_FILE_SUFFIX);
420    if (newMaxSeqId != maxSeqId) {
421      try {
422        if (!walFS.createNewFile(newSeqIdFile) && !walFS.exists(newSeqIdFile)) {
423          throw new IOException("Failed to create SeqId file:" + newSeqIdFile);
424        }
425        LOG.debug("Wrote file={}, newMaxSeqId={}, maxSeqId={}", newSeqIdFile, newMaxSeqId,
426          maxSeqId);
427      } catch (FileAlreadyExistsException ignored) {
428        // latest hdfs throws this exception. it's all right if newSeqIdFile already exists
429      }
430    }
431    // remove old ones
432    for (FileStatus status : files) {
433      if (!newSeqIdFile.equals(status.getPath())) {
434        walFS.delete(status.getPath(), false);
435      }
436    }
437  }

439  /** A struct used by getMutationsFromWALEntry */
440  public static class MutationReplay implements Comparable<MutationReplay> {
441    public MutationReplay(ClientProtos.MutationProto.MutationType type, Mutation mutation,
442        long nonceGroup, long nonce) {
443      this.type = type;
444      this.mutation = mutation;
445      if (this.mutation.getDurability() != Durability.SKIP_WAL) {
446        // using ASYNC_WAL for relay
447        this.mutation.setDurability(Durability.ASYNC_WAL);
448      }
449      this.nonceGroup = nonceGroup;
450      this.nonce = nonce;
451    }
452
453    private final ClientProtos.MutationProto.MutationType type;
454    public final Mutation mutation;
455    public final long nonceGroup;
456    public final long nonce;
457
458    @Override
459    public int compareTo(final MutationReplay d) {
460      return Row.COMPARATOR.compare(mutation, d.mutation);
461    }
462
463    @Override
464    public boolean equals(Object obj) {
465      if (!(obj instanceof MutationReplay)) {
466        return false;
467      } else {
468        return this.compareTo((MutationReplay) obj) == 0;
469      }
470    }
471
472    @Override
473    public int hashCode() {
474      return this.mutation.hashCode();
475    }
476
477    public ClientProtos.MutationProto.MutationType getType() {
478      return type;
479    }
480  }

482  /**
483   * This function is used to construct mutations from a WALEntry. It also reconstructs WALKey &amp;
484   * WALEdit from the passed in WALEntry
485   * @param entry
486   * @param cells
487   * @param logEntry pair of WALKey and WALEdit instance stores WALKey and WALEdit instances
488   *          extracted from the passed in WALEntry.
489   * @return list of Pair&lt;MutationType, Mutation&gt; to be replayed
490   * @throws IOException
491   */
492  public static List<MutationReplay> getMutationsFromWALEntry(AdminProtos.WALEntry entry,
493      CellScanner cells, Pair<WALKey, WALEdit> logEntry, Durability durability) throws IOException {
494    if (entry == null) {
495      // return an empty array
496      return Collections.emptyList();
497    }
498
499    long replaySeqId =
500        (entry.getKey().hasOrigSequenceNumber()) ? entry.getKey().getOrigSequenceNumber()
501            : entry.getKey().getLogSequenceNumber();
502    int count = entry.getAssociatedCellCount();
503    List<MutationReplay> mutations = new ArrayList<>();
504    Cell previousCell = null;
505    Mutation m = null;
506    WALKeyImpl key = null;
507    WALEdit val = null;
508    if (logEntry != null) {
509      val = new WALEdit();
510    }
511
512    for (int i = 0; i < count; i++) {
513      // Throw index out of bounds if our cell count is off
514      if (!cells.advance()) {
515        throw new ArrayIndexOutOfBoundsException("Expected=" + count + ", index=" + i);
516      }
517      Cell cell = cells.current();
518      if (val != null) val.add(cell);
519
520      boolean isNewRowOrType =
521          previousCell == null || previousCell.getTypeByte() != cell.getTypeByte()
522              || !CellUtil.matchingRows(previousCell, cell);
523      if (isNewRowOrType) {
524        // Create new mutation
525        if (CellUtil.isDelete(cell)) {
526          m = new Delete(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
527          // Deletes don't have nonces.
528          mutations.add(new MutationReplay(ClientProtos.MutationProto.MutationType.DELETE, m,
529              HConstants.NO_NONCE, HConstants.NO_NONCE));
530        } else {
531          m = new Put(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength());
532          // Puts might come from increment or append, thus we need nonces.
533          long nonceGroup =
534              entry.getKey().hasNonceGroup() ? entry.getKey().getNonceGroup() : HConstants.NO_NONCE;
535          long nonce = entry.getKey().hasNonce() ? entry.getKey().getNonce() : HConstants.NO_NONCE;
536          mutations.add(
537            new MutationReplay(ClientProtos.MutationProto.MutationType.PUT, m, nonceGroup, nonce));
538        }
539      }
540      if (CellUtil.isDelete(cell)) {
541        ((Delete) m).add(cell);
542      } else {
543        ((Put) m).add(cell);
544      }
545      m.setDurability(durability);
546      previousCell = cell;
547    }
548
549    // reconstruct WALKey
550    if (logEntry != null) {
551      org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos.WALKey walKeyProto =
552          entry.getKey();
553      List<UUID> clusterIds = new ArrayList<>(walKeyProto.getClusterIdsCount());
554      for (HBaseProtos.UUID uuid : entry.getKey().getClusterIdsList()) {
555        clusterIds.add(new UUID(uuid.getMostSigBits(), uuid.getLeastSigBits()));
556      }
557      key = new WALKeyImpl(walKeyProto.getEncodedRegionName().toByteArray(),
558          TableName.valueOf(walKeyProto.getTableName().toByteArray()), replaySeqId,
559          walKeyProto.getWriteTime(), clusterIds, walKeyProto.getNonceGroup(),
560          walKeyProto.getNonce(), null);
561      logEntry.setFirst(key);
562      logEntry.setSecond(val);
563    }
564
565    return mutations;
566  }

569   * Path to a file under recovered.hfiles directory of the region's column family: e.g.
570   * /hbase/some_table/2323432434/cf/recovered.hfiles/2332-wal. This method also ensures existence
571   * of recovered.hfiles directory under the region's column family, creating it if necessary.
572   *
573   * @param tableName          the table name
574   * @param encodedRegionName  the encoded region name
575   * @param familyName         the column family name
576   * @param seqId              the sequence id which used to generate file name
577   * @param fileNameBeingSplit the file being split currently. Used to generate tmp file name
578   * @param conf               configuration
579   * @param rootFS             the root file system
580   * @return Path to file into which to dump split log edits.
581   */
582  static Path getRegionRecoveredHFilePath(TableName tableName, String encodedRegionName,
583    String familyName, long seqId, String fileNameBeingSplit, Configuration conf, FileSystem rootFS)
584    throws IOException {
585    Path rootDir = FSUtils.getRootDir(conf);
586    Path regionDir =
587      FSUtils.getRegionDirFromTableDir(FSUtils.getTableDir(rootDir, tableName), encodedRegionName);
588    Path dir = getStoreDirRecoveredHFilesDir(regionDir, familyName);
589
590    if (!rootFS.exists(dir) && !rootFS.mkdirs(dir)) {
591      LOG.warn("mkdir failed on {}, region {}, column family {}", dir, encodedRegionName,
592        familyName);
593    }
594
595    String fileName = formatRecoveredHFileName(seqId, fileNameBeingSplit);
596    return new Path(dir, fileName);
597  }
598
599  private static String formatRecoveredHFileName(long seqId, String fileNameBeingSplit) {
600    return String.format("%019d", seqId) + "-" + fileNameBeingSplit;
601  }
602
603  public static long getSeqIdForRecoveredHFile(String fileName) {
604    return Long.parseLong(fileName.split("-")[0]);
605  }
606
607  /**
608   * @param regionDir  This regions directory in the filesystem
609   * @param familyName The column family name
610   * @return The directory that holds recovered hfiles for the region's column family
611   */
612  private static Path getStoreDirRecoveredHFilesDir(final Path regionDir, String familyName) {
613    return new Path(new Path(regionDir, familyName), HConstants.RECOVERED_HFILES_DIR);
614  }
615
616  public static FileStatus[] getRecoveredHFiles(final FileSystem rootFS,
617      final Path regionDir, String familyName) throws IOException {
618    Path dir = getStoreDirRecoveredHFilesDir(regionDir, familyName);
619    return FSUtils.listStatus(rootFS, dir);
620  }
