018package org.apache.hadoop.hbase.io;
019
020import org.apache.hadoop.hbase.metrics.BaseSource;
021import org.apache.yetus.audience.InterfaceAudience;
022
023@InterfaceAudience.Private
024public interface MetricsIOSource extends BaseSource {
025
026  /**
027   * The name of the metrics
028   */
029  String METRICS_NAME = "IO";
030
031  /**
032   * The name of the metrics context that metrics will be under.
033   */
034  String METRICS_CONTEXT = "regionserver";
035
036  /**
037   * Description
038   */
039  String METRICS_DESCRIPTION = "Metrics about FileSystem IO";
040
041  /**
042   * The name of the metrics context that metrics will be under in jmx
043   */
044  String METRICS_JMX_CONTEXT = "RegionServer,sub=" + METRICS_NAME;
045
046  String FS_READ_TIME_HISTO_KEY = "fsReadTime";
047  String FS_PREAD_TIME_HISTO_KEY = "fsPReadTime";
048  String FS_WRITE_HISTO_KEY = "fsWriteTime";
049
050  String SLOW_FS_READS_KEY = "fsSlowReadsCount";
051  String SLOW_FS_READS_DESC = "Number of HFile reads which were slower than a configured threshold";
052
053  String CHECKSUM_FAILURES_KEY = "fsChecksumFailureCount";
054
055  String FS_READ_TIME_HISTO_DESC =
056    "Latency of HFile's sequential reads on this region server in milliseconds";
057  String FS_PREAD_TIME_HISTO_DESC =
058    "Latency of HFile's positional reads on this region server in milliseconds";
059  String FS_WRITE_TIME_HISTO_DESC =
060    "Latency of HFile's writes on this region server in milliseconds";
061
062  String CHECKSUM_FAILURES_DESC = "Number of checksum failures for the HBase HFile checksums at the"
063    + " HBase level (separate from HDFS checksums)";
064
065  /**
066   * Update the fs sequential read time histogram
067   * @param t time it took, in milliseconds
068   */
069  void updateFsReadTime(long t);
070
071  /**
072   * Update the fs positional read time histogram
073   * @param t time it took, in milliseconds
074   */
075  void updateFsPReadTime(long t);
076
077  /**
078   * Update the fs write time histogram
079   * @param t time it took, in milliseconds
080   */
081  void updateFsWriteTime(long t);
082
083  void incrSlowFsRead();
084}
