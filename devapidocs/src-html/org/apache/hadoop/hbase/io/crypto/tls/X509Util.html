094  private static String[] getGCMCiphers() {
095    return new String[] { "TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256",
096      "TLS_ECDHE_RSA_WITH_AES_128_GCM_SHA256", "TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384",
097      "TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384" };
098  }

100  private static String[] getCBCCiphers() {
101    return new String[] { "TLS_ECDHE_ECDSA_WITH_AES_128_CBC_SHA256",
102      "TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA256", "TLS_ECDHE_ECDSA_WITH_AES_128_CBC_SHA",
103      "TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA", "TLS_ECDHE_ECDSA_WITH_AES_256_CBC_SHA384",
104      "TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA384", "TLS_ECDHE_ECDSA_WITH_AES_256_CBC_SHA",
105      "TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA" };
106  }

108  // On Java 8, prefer CBC ciphers since AES-NI support is lacking and GCM is slower than CBC.
109  private static final String[] DEFAULT_CIPHERS_JAVA8 =
110    ObjectArrays.concat(getCBCCiphers(), getGCMCiphers(), String.class);
111  // On Java 9 and later, prefer GCM ciphers due to improved AES-NI support.
112  // Note that this performance assumption might not hold true for architectures other than x86_64.
113  private static final String[] DEFAULT_CIPHERS_JAVA9 =
114    ObjectArrays.concat(getGCMCiphers(), getCBCCiphers(), String.class);

120  static String[] getDefaultCipherSuites() {
121    return getDefaultCipherSuitesForJavaVersion(System.getProperty("java.specification.version"));
122  }
123
124  static String[] getDefaultCipherSuitesForJavaVersion(String javaVersion) {
125    Objects.requireNonNull(javaVersion);
126    if (javaVersion.matches("\\d+")) {
127      // Must be Java 9 or later
128      LOG.debug("Using Java9+ optimized cipher suites for Java version {}", javaVersion);
129      return DEFAULT_CIPHERS_JAVA9;
130    } else if (javaVersion.startsWith("1.")) {
131      // Must be Java 1.8 or earlier
132      LOG.debug("Using Java8 optimized cipher suites for Java version {}", javaVersion);
133      return DEFAULT_CIPHERS_JAVA8;
134    } else {
135      LOG.debug("Could not parse java version {}, using Java8 optimized cipher suites",
136        javaVersion);
137      return DEFAULT_CIPHERS_JAVA8;
138    }
139  }

141  public static SslContext createSslContextForClient(Configuration config)
142    throws X509Exception, IOException {
143
144    SslContextBuilder sslContextBuilder = SslContextBuilder.forClient();
145
146    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
147    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
148    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
149
150    if (keyStoreLocation.isEmpty()) {
151      LOG.warn(TLS_CONFIG_KEYSTORE_LOCATION + " not specified");
152    } else {
153      sslContextBuilder
154        .keyManager(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
155    }
156
157    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
158    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
159    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
160
161    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
162    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
163
164    if (trustStoreLocation.isEmpty()) {
165      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
166    } else {
167      sslContextBuilder.trustManager(createTrustManager(trustStoreLocation, trustStorePassword,
168        trustStoreType, sslCrlEnabled, sslOcspEnabled));
169    }
170
171    sslContextBuilder.enableOcsp(sslOcspEnabled);
172    sslContextBuilder.protocols(getEnabledProtocols(config));
173    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config)));
174
175    return sslContextBuilder.build();
176  }

178  public static SslContext createSslContextForServer(Configuration config)
179    throws X509Exception, IOException {
180    String keyStoreLocation = config.get(TLS_CONFIG_KEYSTORE_LOCATION, "");
181    char[] keyStorePassword = config.getPassword(TLS_CONFIG_KEYSTORE_PASSWORD);
182    String keyStoreType = config.get(TLS_CONFIG_KEYSTORE_TYPE, "");
183
184    if (keyStoreLocation.isEmpty()) {
185      throw new SSLContextException(
186        "Keystore is required for SSL server: " + TLS_CONFIG_KEYSTORE_LOCATION);
187    }
188
189    SslContextBuilder sslContextBuilder;
190
191    sslContextBuilder = SslContextBuilder
192      .forServer(createKeyManager(keyStoreLocation, keyStorePassword, keyStoreType));
193
194    String trustStoreLocation = config.get(TLS_CONFIG_TRUSTSTORE_LOCATION, "");
195    char[] trustStorePassword = config.getPassword(TLS_CONFIG_TRUSTSTORE_PASSWORD);
196    String trustStoreType = config.get(TLS_CONFIG_TRUSTSTORE_TYPE, "");
197
198    boolean sslCrlEnabled = config.getBoolean(TLS_CONFIG_CLR, false);
199    boolean sslOcspEnabled = config.getBoolean(TLS_CONFIG_OCSP, false);
200
201    if (trustStoreLocation.isEmpty()) {
202      LOG.warn(TLS_CONFIG_TRUSTSTORE_LOCATION + " not specified");
203    } else {
204      sslContextBuilder.trustManager(createTrustManager(trustStoreLocation, trustStorePassword,
205        trustStoreType, sslCrlEnabled, sslOcspEnabled));
206    }
207
208    sslContextBuilder.enableOcsp(sslOcspEnabled);
209    sslContextBuilder.protocols(getEnabledProtocols(config));
210    sslContextBuilder.ciphers(Arrays.asList(getCipherSuites(config)));
211
212    return sslContextBuilder.build();
213  }

215  /**
216   * Creates a key manager by loading the key store from the given file of the given type,
217   * optionally decrypting it using the given password.
218   * @param keyStoreLocation the location of the key store file.
219   * @param keyStorePassword optional password to decrypt the key store. If empty, assumes the key
220   *                         store is not encrypted.
221   * @param keyStoreType     must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
222   *                         autodetect the key store type from the file extension (e.g. .jks /
223   *                         .pem).
224   * @return the key manager.
225   * @throws KeyManagerException if something goes wrong.
226   */
227  static X509KeyManager createKeyManager(String keyStoreLocation, char[] keyStorePassword,
228    String keyStoreType) throws KeyManagerException {
229
230    if (keyStoreType == null) {
231      keyStoreType = "jks";
232    }
233
234    if (keyStorePassword == null) {
235      keyStorePassword = EMPTY_CHAR_ARRAY;
236    }
237
238    try {
239      KeyStore ks = KeyStore.getInstance(keyStoreType);
240      try (InputStream inputStream = Files.newInputStream(new File(keyStoreLocation).toPath())) {
241        ks.load(inputStream, keyStorePassword);
242      }
243
244      KeyManagerFactory kmf = KeyManagerFactory.getInstance("PKIX");
245      kmf.init(ks, keyStorePassword);
246
247      for (KeyManager km : kmf.getKeyManagers()) {
248        if (km instanceof X509KeyManager) {
249          return (X509KeyManager) km;
250        }
251      }
252      throw new KeyManagerException("Couldn't find X509KeyManager");
253    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
254      throw new KeyManagerException(e);
255    }
256  }

258  /**
259   * Creates a trust manager by loading the trust store from the given file of the given type,
260   * optionally decrypting it using the given password.
261   * @param trustStoreLocation the location of the trust store file.
262   * @param trustStorePassword optional password to decrypt the trust store (only applies to JKS
263   *                           trust stores). If empty, assumes the trust store is not encrypted.
264   * @param trustStoreType     must be JKS, PEM, PKCS12, BCFKS or null. If null, attempts to
265   *                           autodetect the trust store type from the file extension (e.g. .jks /
266   *                           .pem).
267   * @param crlEnabled         enable CRL (certificate revocation list) checks.
268   * @param ocspEnabled        enable OCSP (online certificate status protocol) checks.
269   * @return the trust manager.
270   * @throws TrustManagerException if something goes wrong.
271   */
272  static X509TrustManager createTrustManager(String trustStoreLocation, char[] trustStorePassword,
273    String trustStoreType, boolean crlEnabled, boolean ocspEnabled) throws TrustManagerException {
274
275    if (trustStoreType == null) {
276      trustStoreType = "jks";
277    }
278
279    if (trustStorePassword == null) {
280      trustStorePassword = EMPTY_CHAR_ARRAY;
281    }
282
283    try {
284      KeyStore ts = KeyStore.getInstance(trustStoreType);
285      try (InputStream inputStream = Files.newInputStream(new File(trustStoreLocation).toPath())) {
286        ts.load(inputStream, trustStorePassword);
287      }
288
289      PKIXBuilderParameters pbParams = new PKIXBuilderParameters(ts, new X509CertSelector());
290      if (crlEnabled || ocspEnabled) {
291        pbParams.setRevocationEnabled(true);
292        System.setProperty("com.sun.net.ssl.checkRevocation", "true");
293        if (crlEnabled) {
294          System.setProperty("com.sun.security.enableCRLDP", "true");
295        }
296        if (ocspEnabled) {
297          Security.setProperty("ocsp.enable", "true");
298        }
299      } else {
300        pbParams.setRevocationEnabled(false);
301      }
302
303      // Revocation checking is only supported with the PKIX algorithm
304      TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
305      tmf.init(new CertPathTrustManagerParameters(pbParams));
306
307      for (final TrustManager tm : tmf.getTrustManagers()) {
308        if (tm instanceof X509ExtendedTrustManager) {
309          return (X509ExtendedTrustManager) tm;
310        }
311      }
312      throw new TrustManagerException("Couldn't find X509TrustManager");
313    } catch (IOException | GeneralSecurityException | IllegalArgumentException e) {
314      throw new TrustManagerException(e);
315    }
316  }

318  private static String[] getEnabledProtocols(Configuration config) {
319    String enabledProtocolsInput = config.get(TLS_ENABLED_PROTOCOLS);
320    if (enabledProtocolsInput == null) {
321      return new String[] { config.get(TLS_CONFIG_PROTOCOL, DEFAULT_PROTOCOL) };
322    }
323    return enabledProtocolsInput.split(",");
324  }
325
326  private static String[] getCipherSuites(Configuration config) {
327    String cipherSuitesInput = config.get(TLS_CIPHER_SUITES);
328    if (cipherSuitesInput == null) {
329      return getDefaultCipherSuites();
330    } else {
331      return cipherSuitesInput.split(",");
332    }
