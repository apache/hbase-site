037/**
038 * Wrapper for input stream(s) that takes care of the interaction of FS and HBase checksums,
039 * as well as closing streams. Initialization is not thread-safe, but normal operation is;
040 * see method comments.
041 */
042@InterfaceAudience.Private
043public class FSDataInputStreamWrapper implements Closeable {
044  private static final Logger LOG = LoggerFactory.getLogger(FSDataInputStreamWrapper.class);
045  private static final boolean isLogTraceEnabled = LOG.isTraceEnabled();
046
047  private final HFileSystem hfs;
048  private final Path path;
049  private final FileLink link;
050  private final boolean doCloseStreams;
051  private final boolean dropBehind;
052  private final long readahead;
053
054  /** Two stream handles, one with and one without FS-level checksum.
055   * HDFS checksum setting is on FS level, not single read level, so you have to keep two
056   * FS objects and two handles open to interleave different reads freely, which is very sad.
057   * This is what we do:
058   * 1) First, we need to read the trailer of HFile to determine checksum parameters.
059   *  We always use FS checksum to do that, so ctor opens {@link #stream}.
060   * 2.1) After that, if HBase checksum is not used, we'd just always use {@link #stream};
061   * 2.2) If HBase checksum can be used, we'll open {@link #streamNoFsChecksum},
062   *  and close {@link #stream}. User MUST call prepareForBlockReader for that to happen;
063   *  if they don't, (2.1) will be the default.
064   * 3) The users can call {@link #shouldUseHBaseChecksum()}, and pass its result to
065   *  {@link #getStream(boolean)} to get stream (if Java had out/pointer params we could
066   *  return both in one call). This stream is guaranteed to be set.
067   * 4) The first time HBase checksum fails, one would call {@link #fallbackToFsChecksum(int)}.
068   * That will take lock, and open {@link #stream}. While this is going on, others will
069   * continue to use the old stream; if they also want to fall back, they'll also call
070   * {@link #fallbackToFsChecksum(int)}, and block until {@link #stream} is set.
071   * 5) After some number of checksumOk() calls, we will go back to using HBase checksum.
072   * We will have 2 handles; however we presume checksums fail so rarely that we don't care.
073   */
074  private volatile FSDataInputStream stream = null;
075  private volatile FSDataInputStream streamNoFsChecksum = null;
076  private Object streamNoFsChecksumFirstCreateLock = new Object();
077
078  // The configuration states that we should validate hbase checksums
079  private boolean useHBaseChecksumConfigured;
080
081  // Record the current state of this reader with respect to
082  // validating checkums in HBase. This is originally set the same
083  // value as useHBaseChecksumConfigured, but can change state as and when
084  // we encounter checksum verification failures.
085  private volatile boolean useHBaseChecksum;
086
087  // In the case of a checksum failure, do these many succeeding
088  // reads without hbase checksum verification.
089  private volatile int hbaseChecksumOffCount = -1;
090
091  private Boolean instanceOfCanUnbuffer = null;
092  // Using reflection to get org.apache.hadoop.fs.CanUnbuffer#unbuffer method to avoid compilation
093  // errors against Hadoop pre 2.6.4 and 2.7.1 versions.
094  private Method unbuffer = null;
095
096  public FSDataInputStreamWrapper(FileSystem fs, Path path) throws IOException {
097    this(fs, path, false, -1L);
098  }
099
100  public FSDataInputStreamWrapper(FileSystem fs, Path path, boolean dropBehind, long readahead) throws IOException {
101    this(fs, null, path, dropBehind, readahead);
102  }
103
104  public FSDataInputStreamWrapper(FileSystem fs, FileLink link,
105                                  boolean dropBehind, long readahead) throws IOException {
106    this(fs, link, null, dropBehind, readahead);
107  }
108
109  private FSDataInputStreamWrapper(FileSystem fs, FileLink link, Path path, boolean dropBehind,
110      long readahead) throws IOException {
111    assert (path == null) != (link == null);
112    this.path = path;
113    this.link = link;
114    this.doCloseStreams = true;
115    this.dropBehind = dropBehind;
116    this.readahead = readahead;
117    // If the fs is not an instance of HFileSystem, then create an instance of HFileSystem
118    // that wraps over the specified fs. In this case, we will not be able to avoid
119    // checksumming inside the filesystem.
120    this.hfs = (fs instanceof HFileSystem) ? (HFileSystem) fs : new HFileSystem(fs);
121
122    // Initially we are going to read the tail block. Open the reader w/FS checksum.
123    this.useHBaseChecksumConfigured = this.useHBaseChecksum = false;
124    this.stream = (link != null) ? link.open(hfs) : hfs.open(path);
125    setStreamOptions(stream);
126  }
127
128  private void setStreamOptions(FSDataInputStream in) {
129    try {
130      in.setDropBehind(dropBehind);
131    } catch (Exception e) {
132      // Skipped.
133    }
134    if (readahead >= 0) {
135      try {
136        in.setReadahead(readahead);
137      } catch (Exception e) {
138        // Skipped.
139      }
140    }
141  }
142
143  /**
144   * Prepares the streams for block reader. NOT THREAD SAFE. Must be called once, after any
145   * reads finish and before any other reads start (what happens in reality is we read the
146   * tail, then call this based on what's in the tail, then read blocks).
147   * @param forceNoHBaseChecksum Force not using HBase checksum.
148   */
149  public void prepareForBlockReader(boolean forceNoHBaseChecksum) throws IOException {
150    if (hfs == null) return;
151    assert this.stream != null && !this.useHBaseChecksumConfigured;
152    boolean useHBaseChecksum =
153        !forceNoHBaseChecksum && hfs.useHBaseChecksum() && (hfs.getNoChecksumFs() != hfs);
154
155    if (useHBaseChecksum) {
156      FileSystem fsNc = hfs.getNoChecksumFs();
157      this.streamNoFsChecksum = (link != null) ? link.open(fsNc) : fsNc.open(path);
158      setStreamOptions(streamNoFsChecksum);
159      this.useHBaseChecksumConfigured = this.useHBaseChecksum = useHBaseChecksum;
160      // Close the checksum stream; we will reopen it if we get an HBase checksum failure.
161      this.stream.close();
162      this.stream = null;
163    }
164  }
165
166  /** For use in tests. */
167  @VisibleForTesting
168  public FSDataInputStreamWrapper(FSDataInputStream fsdis) {
169    this(fsdis, fsdis);
170  }
171
172  /** For use in tests. */
173  @VisibleForTesting
174  public FSDataInputStreamWrapper(FSDataInputStream fsdis, FSDataInputStream noChecksum) {
175    doCloseStreams = false;
176    stream = fsdis;
177    streamNoFsChecksum = noChecksum;
178    path = null;
179    link = null;
180    hfs = null;
181    useHBaseChecksumConfigured = useHBaseChecksum = false;
182    dropBehind = false;
183    readahead = 0;
184  }
185
186  /**
187   * @return Whether we are presently using HBase checksum.
188   */
189  public boolean shouldUseHBaseChecksum() {
190    return this.useHBaseChecksum;
191  }
192
193  /**
194   * Get the stream to use. Thread-safe.
195   * @param useHBaseChecksum must be the value that shouldUseHBaseChecksum has returned
196   *  at some point in the past, otherwise the result is undefined.
197   */
198  public FSDataInputStream getStream(boolean useHBaseChecksum) {
199    return useHBaseChecksum ? this.streamNoFsChecksum : this.stream;
200  }
201
202  /**
203   * Read from non-checksum stream failed, fall back to FS checksum. Thread-safe.
204   * @param offCount For how many checksumOk calls to turn off the HBase checksum.
205   */
206  public FSDataInputStream fallbackToFsChecksum(int offCount) throws IOException {
207    // checksumOffCount is speculative, but let's try to reset it less.
208    boolean partOfConvoy = false;
209    if (this.stream == null) {
210      synchronized (streamNoFsChecksumFirstCreateLock) {
211        partOfConvoy = (this.stream != null);
212        if (!partOfConvoy) {
213          this.stream = (link != null) ? link.open(hfs) : hfs.open(path);
214        }
215      }
216    }
217    if (!partOfConvoy) {
218      this.useHBaseChecksum = false;
219      this.hbaseChecksumOffCount = offCount;
220    }
221    return this.stream;
222  }
223
224  /** Report that checksum was ok, so we may ponder going back to HBase checksum. */
225  public void checksumOk() {
226    if (this.useHBaseChecksumConfigured && !this.useHBaseChecksum
227        && (this.hbaseChecksumOffCount-- < 0)) {
228      // The stream we need is already open (because we were using HBase checksum in the past).
229      assert this.streamNoFsChecksum != null;
230      this.useHBaseChecksum = true;
231    }
232  }
233
234  /** Close stream(s) if necessary. */
235  @Override
236  public void close() {
237    if (!doCloseStreams) {
238      return;
239    }
240    // we do not care about the close exception as it is for reading, no data loss issue.
241    IOUtils.closeQuietly(streamNoFsChecksum);
242    IOUtils.closeQuietly(stream);
243  }
244
245  public HFileSystem getHfs() {
246    return this.hfs;
247  }
248
249  /**
250   * This will free sockets and file descriptors held by the stream only when the stream implements
251   * org.apache.hadoop.fs.CanUnbuffer. NOT THREAD SAFE. Must be called only when all the clients
252   * using this stream to read the blocks have finished reading. If by chance the stream is
253   * unbuffered and there are clients still holding this stream for read then on next client read
254   * request a new socket will be opened by Datanode without client knowing about it and will serve
255   * its read request. Note: If this socket is idle for some time then the DataNode will close the
256   * socket and the socket will move into CLOSE_WAIT state and on the next client request on this
257   * stream, the current socket will be closed and a new socket will be opened to serve the
258   * requests.
259   */
260  @SuppressWarnings({ "rawtypes" })
261  public void unbuffer() {
262    FSDataInputStream stream = this.getStream(this.shouldUseHBaseChecksum());
263    if (stream != null) {
264      InputStream wrappedStream = stream.getWrappedStream();
265      // CanUnbuffer interface was added as part of HDFS-7694 and the fix is available in Hadoop
266      // 2.6.4+ and 2.7.1+ versions only so check whether the stream object implements the
267      // CanUnbuffer interface or not and based on that call the unbuffer api.
268      final Class<? extends InputStream> streamClass = wrappedStream.getClass();
269      if (this.instanceOfCanUnbuffer == null) {
270        // To ensure we compute whether the stream is instance of CanUnbuffer only once.
271        this.instanceOfCanUnbuffer = false;
272        Class<?>[] streamInterfaces = streamClass.getInterfaces();
273        for (Class c : streamInterfaces) {
274          if (c.getCanonicalName().toString().equals("org.apache.hadoop.fs.CanUnbuffer")) {
275            try {
276              this.unbuffer = streamClass.getDeclaredMethod("unbuffer");
277            } catch (NoSuchMethodException | SecurityException e) {
278              if (isLogTraceEnabled) {
279                LOG.trace("Failed to find 'unbuffer' method in class " + streamClass
280                    + " . So there may be a TCP socket connection "
281                    + "left open in CLOSE_WAIT state.", e);
282              }
283              return;
284            }
285            this.instanceOfCanUnbuffer = true;
286            break;
287          }
288        }
289      }
290      if (this.instanceOfCanUnbuffer) {
291        try {
292          this.unbuffer.invoke(wrappedStream);
293        } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
294          if (isLogTraceEnabled) {
295            LOG.trace("Failed to invoke 'unbuffer' method in class " + streamClass
296                + " . So there may be a TCP socket connection left open in CLOSE_WAIT state.", e);
297          }
298        }
299      } else {
300        if (isLogTraceEnabled) {
301          LOG.trace("Failed to find 'unbuffer' method in class " + streamClass
302              + " . So there may be a TCP socket connection "
303              + "left open in CLOSE_WAIT state. For more details check "
304              + "https://issues.apache.org/jira/browse/HBASE-9393");
305        }
306      }
307    }
308  }
