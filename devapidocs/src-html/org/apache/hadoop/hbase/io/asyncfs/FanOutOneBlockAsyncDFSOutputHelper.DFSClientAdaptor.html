270  private static PipelineAckStatusGetter createPipelineAckStatusGetter27()
271      throws NoSuchMethodException {
272    Method getFlagListMethod = PipelineAckProto.class.getMethod("getFlagList");
273    @SuppressWarnings("rawtypes")
274    Class<? extends Enum> ecnClass;
275    try {
276      ecnClass = Class.forName("org.apache.hadoop.hdfs.protocol.datatransfer.PipelineAck$ECN")
277          .asSubclass(Enum.class);
278    } catch (ClassNotFoundException e) {
279      String msg = "Couldn't properly initialize the PipelineAck.ECN class. Please "
280          + "update your WAL Provider to not make use of the 'asyncfs' provider. See "
281          + "HBASE-16110 for more information.";
282      LOG.error(msg, e);
283      throw new Error(msg, e);
284    }
285    @SuppressWarnings("unchecked")
286    Enum<?> disabledECN = Enum.valueOf(ecnClass, "DISABLED");
287    Method getReplyMethod = PipelineAckProto.class.getMethod("getReply", int.class);
288    Method combineHeaderMethod =
289        PipelineAck.class.getMethod("combineHeader", ecnClass, Status.class);
290    Method getStatusFromHeaderMethod =
291        PipelineAck.class.getMethod("getStatusFromHeader", int.class);
292    return new PipelineAckStatusGetter() {
293
294      @Override
295      public Status get(PipelineAckProto ack) {
296        try {
297          @SuppressWarnings("unchecked")
298          List<Integer> flagList = (List<Integer>) getFlagListMethod.invoke(ack);
299          Integer headerFlag;
300          if (flagList.isEmpty()) {
301            Status reply = (Status) getReplyMethod.invoke(ack, 0);
302            headerFlag = (Integer) combineHeaderMethod.invoke(null, disabledECN, reply);
303          } else {
304            headerFlag = flagList.get(0);
305          }
306          return (Status) getStatusFromHeaderMethod.invoke(null, headerFlag);
307        } catch (IllegalAccessException | InvocationTargetException e) {
308          throw new RuntimeException(e);
309        }
310      }
311    };
312  }

364    for (Method method : ClientProtocol.class.getMethods()) {
365      if (method.getName().equals("addBlock")) {
366        Method addBlockMethod = method;
367        Class<?>[] paramTypes = addBlockMethod.getParameterTypes();
368        if (paramTypes[paramTypes.length - 1] == String[].class) {
369          return new BlockAdder() {
370
371            @Override
372            public LocatedBlock addBlock(ClientProtocol namenode, String src, String clientName,
373                ExtendedBlock previous, DatanodeInfo[] excludeNodes, long fileId,
374                String[] favoredNodes) throws IOException {
375              try {
376                return (LocatedBlock) addBlockMethod.invoke(namenode, src, clientName, previous,
377                  excludeNodes, fileId, favoredNodes);
378              } catch (IllegalAccessException e) {
379                throw new RuntimeException(e);
380              } catch (InvocationTargetException e) {
381                Throwables.propagateIfPossible(e.getTargetException(), IOException.class);
382                throw new RuntimeException(e);
383              }
384            }
385          };
386        } else {
387          return new BlockAdder() {
388
389            @Override
390            public LocatedBlock addBlock(ClientProtocol namenode, String src, String clientName,
391                ExtendedBlock previous, DatanodeInfo[] excludeNodes, long fileId,
392                String[] favoredNodes) throws IOException {
393              try {
394                return (LocatedBlock) addBlockMethod.invoke(namenode, src, clientName, previous,
395                  excludeNodes, fileId, favoredNodes, null);
396              } catch (IllegalAccessException e) {
397                throw new RuntimeException(e);
398              } catch (InvocationTargetException e) {
399                Throwables.propagateIfPossible(e.getTargetException(), IOException.class);
400                throw new RuntimeException(e);
401              }
402            }
403          };
404        }
405      }
406    }
407    throw new NoSuchMethodException("Can not find addBlock method in ClientProtocol");
408  }

576  private static void processWriteBlockResponse(Channel channel, DatanodeInfo dnInfo,
577      Promise<Channel> promise, int timeoutMs) {
578    channel.pipeline().addLast(new IdleStateHandler(timeoutMs, 0, 0, TimeUnit.MILLISECONDS),
579      new ProtobufVarint32FrameDecoder(),
580      new ProtobufDecoder(BlockOpResponseProto.getDefaultInstance()),
581      new SimpleChannelInboundHandler<BlockOpResponseProto>() {
582
583        @Override
584        protected void channelRead0(ChannelHandlerContext ctx, BlockOpResponseProto resp)
585            throws Exception {
586          Status pipelineStatus = resp.getStatus();
587          if (PipelineAck.isRestartOOBStatus(pipelineStatus)) {
588            throw new IOException("datanode " + dnInfo + " is restarting");
589          }
590          String logInfo = "ack with firstBadLink as " + resp.getFirstBadLink();
591          if (resp.getStatus() != Status.SUCCESS) {
592            if (resp.getStatus() == Status.ERROR_ACCESS_TOKEN) {
593              throw new InvalidBlockTokenException("Got access token error" + ", status message "
594                  + resp.getMessage() + ", " + logInfo);
595            } else {
596              throw new IOException("Got error" + ", status=" + resp.getStatus().name()
597                  + ", status message " + resp.getMessage() + ", " + logInfo);
598            }
599          }
600          // success
601          ChannelPipeline p = ctx.pipeline();
602          for (ChannelHandler handler; (handler = p.removeLast()) != null;) {
603            // do not remove all handlers because we may have wrap or unwrap handlers at the header
604            // of pipeline.
605            if (handler instanceof IdleStateHandler) {
606              break;
607            }
608          }
609          // Disable auto read here. Enable it after we setup the streaming pipeline in
610          // FanOutOneBLockAsyncDFSOutput.
611          ctx.channel().config().setAutoRead(false);
612          promise.trySuccess(ctx.channel());
613        }
614
615        @Override
616        public void channelInactive(ChannelHandlerContext ctx) throws Exception {
617          promise.tryFailure(new IOException("connection to " + dnInfo + " is closed"));
618        }
619
620        @Override
621        public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
622          if (evt instanceof IdleStateEvent && ((IdleStateEvent) evt).state() == READER_IDLE) {
623            promise
624                .tryFailure(new IOException("Timeout(" + timeoutMs + "ms) waiting for response"));
625          } else {
626            super.userEventTriggered(ctx, evt);
627          }
628        }
629
630        @Override
631        public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
632          promise.tryFailure(cause);
633        }
634      });
635  }

637  private static void requestWriteBlock(Channel channel, Enum<?> storageType,
638      OpWriteBlockProto.Builder writeBlockProtoBuilder) throws IOException {
639    OpWriteBlockProto proto = STORAGE_TYPE_SETTER.set(writeBlockProtoBuilder, storageType).build();
640    int protoLen = proto.getSerializedSize();
641    ByteBuf buffer =
642        channel.alloc().buffer(3 + CodedOutputStream.computeRawVarint32Size(protoLen) + protoLen);
643    buffer.writeShort(DataTransferProtocol.DATA_TRANSFER_VERSION);
644    buffer.writeByte(Op.WRITE_BLOCK.code);
645    proto.writeDelimitedTo(new ByteBufOutputStream(buffer));
646    channel.writeAndFlush(buffer);
647  }

649  private static void initialize(Configuration conf, Channel channel, DatanodeInfo dnInfo,
650      Enum<?> storageType, OpWriteBlockProto.Builder writeBlockProtoBuilder, int timeoutMs,
651      DFSClient client, Token<BlockTokenIdentifier> accessToken, Promise<Channel> promise)
652      throws IOException {
653    Promise<Void> saslPromise = channel.eventLoop().newPromise();
654    trySaslNegotiate(conf, channel, dnInfo, timeoutMs, client, accessToken, saslPromise);
655    saslPromise.addListener(new FutureListener<Void>() {
656
657      @Override
658      public void operationComplete(Future<Void> future) throws Exception {
659        if (future.isSuccess()) {
660          // setup response processing pipeline first, then send request.
661          processWriteBlockResponse(channel, dnInfo, promise, timeoutMs);
662          requestWriteBlock(channel, storageType, writeBlockProtoBuilder);
663        } else {
664          promise.tryFailure(future.cause());
665        }
666      }
667    });
668  }

670  private static List<Future<Channel>> connectToDataNodes(Configuration conf,
671      DFSClient client, String clientName, LocatedBlock locatedBlock, long maxBytesRcvd,
672      long latestGS, BlockConstructionStage stage, DataChecksum summer,
673      EventLoopGroup eventLoopGroup, Class<? extends Channel> channelClass) {
674    Enum<?>[] storageTypes = locatedBlock.getStorageTypes();
675    DatanodeInfo[] datanodeInfos = locatedBlock.getLocations();
676    boolean connectToDnViaHostname =
677        conf.getBoolean(DFS_CLIENT_USE_DN_HOSTNAME, DFS_CLIENT_USE_DN_HOSTNAME_DEFAULT);
678    int timeoutMs = conf.getInt(DFS_CLIENT_SOCKET_TIMEOUT_KEY, READ_TIMEOUT);
679    ExtendedBlock blockCopy = new ExtendedBlock(locatedBlock.getBlock());
680    blockCopy.setNumBytes(locatedBlock.getBlockSize());
681    ClientOperationHeaderProto header = ClientOperationHeaderProto.newBuilder()
682        .setBaseHeader(BaseHeaderProto.newBuilder().setBlock(PB_HELPER.convert(blockCopy))
683            .setToken(PB_HELPER.convert(locatedBlock.getBlockToken())))
684        .setClientName(clientName).build();
685    ChecksumProto checksumProto = DataTransferProtoUtil.toProto(summer);
686    OpWriteBlockProto.Builder writeBlockProtoBuilder = OpWriteBlockProto.newBuilder()
687        .setHeader(header).setStage(OpWriteBlockProto.BlockConstructionStage.valueOf(stage.name()))
688        .setPipelineSize(1).setMinBytesRcvd(locatedBlock.getBlock().getNumBytes())
689        .setMaxBytesRcvd(maxBytesRcvd).setLatestGenerationStamp(latestGS)
690        .setRequestedChecksum(checksumProto)
691        .setCachingStrategy(CachingStrategyProto.newBuilder().setDropBehind(true).build());
692    List<Future<Channel>> futureList = new ArrayList<>(datanodeInfos.length);
693    for (int i = 0; i < datanodeInfos.length; i++) {
694      DatanodeInfo dnInfo = datanodeInfos[i];
695      Enum<?> storageType = storageTypes[i];
696      Promise<Channel> promise = eventLoopGroup.next().newPromise();
697      futureList.add(promise);
698      String dnAddr = dnInfo.getXferAddr(connectToDnViaHostname);
699      new Bootstrap().group(eventLoopGroup).channel(channelClass)
700          .option(CONNECT_TIMEOUT_MILLIS, timeoutMs).handler(new ChannelInitializer<Channel>() {
701
702            @Override
703            protected void initChannel(Channel ch) throws Exception {
704              // we need to get the remote address of the channel so we can only move on after
705              // channel connected. Leave an empty implementation here because netty does not allow
706              // a null handler.
707            }
708          }).connect(NetUtils.createSocketAddr(dnAddr)).addListener(new ChannelFutureListener() {
709
710            @Override
711            public void operationComplete(ChannelFuture future) throws Exception {
712              if (future.isSuccess()) {
713                initialize(conf, future.channel(), dnInfo, storageType, writeBlockProtoBuilder,
714                  timeoutMs, client, locatedBlock.getBlockToken(), promise);
715              } else {
716                promise.tryFailure(future.cause());
717              }
718            }
719          });
720    }
721    return futureList;
722  }

736  private static FanOutOneBlockAsyncDFSOutput createOutput(DistributedFileSystem dfs, String src,
737      boolean overwrite, boolean createParent, short replication, long blockSize,
738      EventLoopGroup eventLoopGroup, Class<? extends Channel> channelClass) throws IOException {
739    Configuration conf = dfs.getConf();
740    FSUtils fsUtils = FSUtils.getInstance(dfs, conf);
741    DFSClient client = dfs.getClient();
742    String clientName = client.getClientName();
743    ClientProtocol namenode = client.getNamenode();
744    HdfsFileStatus stat;
745    try {
746      stat = FILE_CREATOR.create(namenode, src,
747        FsPermission.getFileDefault().applyUMask(FsPermission.getUMask(conf)), clientName,
748        new EnumSetWritable<>(overwrite ? EnumSet.of(CREATE, OVERWRITE) : EnumSet.of(CREATE)),
749        createParent, replication, blockSize, CryptoProtocolVersion.supported());
750    } catch (Exception e) {
751      if (e instanceof RemoteException) {
752        throw (RemoteException) e;
753      } else {
754        throw new NameNodeException(e);
755      }
756    }
757    beginFileLease(client, stat.getFileId());
758    boolean succ = false;
759    LocatedBlock locatedBlock = null;
760    List<Future<Channel>> futureList = null;
761    try {
762      DataChecksum summer = createChecksum(client);
763      locatedBlock = BLOCK_ADDER.addBlock(namenode, src, client.getClientName(), null, null,
764        stat.getFileId(), null);
765      List<Channel> datanodeList = new ArrayList<>();
766      futureList = connectToDataNodes(conf, client, clientName, locatedBlock, 0L, 0L,
767        PIPELINE_SETUP_CREATE, summer, eventLoopGroup, channelClass);
768      for (Future<Channel> future : futureList) {
769        // fail the creation if there are connection failures since we are fail-fast. The upper
770        // layer should retry itself if needed.
771        datanodeList.add(future.syncUninterruptibly().getNow());
772      }
773      Encryptor encryptor = createEncryptor(conf, stat, client);
774      FanOutOneBlockAsyncDFSOutput output =
775          new FanOutOneBlockAsyncDFSOutput(conf, fsUtils, dfs, client, namenode, clientName, src,
776              stat.getFileId(), locatedBlock, encryptor, datanodeList, summer, ALLOC);
777      succ = true;
778      return output;
779    } finally {
780      if (!succ) {
781        if (futureList != null) {
782          for (Future<Channel> f : futureList) {
783            f.addListener(new FutureListener<Channel>() {
784
785              @Override
786              public void operationComplete(Future<Channel> future) throws Exception {
787                if (future.isSuccess()) {
788                  future.getNow().close();
789                }
790              }
791            });
792          }
793        }
794        endFileLease(client, stat.getFileId());
795        fsUtils.recoverFileLease(dfs, new Path(src), conf, new CancelOnClose(client));
796      }
797    }
798  }

800  /**
801   * Create a {@link FanOutOneBlockAsyncDFSOutput}. The method maybe blocked so do not call it
802   * inside an {@link EventLoop}.
803   */
804  public static FanOutOneBlockAsyncDFSOutput createOutput(DistributedFileSystem dfs, Path f,
805      boolean overwrite, boolean createParent, short replication, long blockSize,
806      EventLoopGroup eventLoopGroup, Class<? extends Channel> channelClass) throws IOException {
807    return new FileSystemLinkResolver<FanOutOneBlockAsyncDFSOutput>() {
808
809      @Override
810      public FanOutOneBlockAsyncDFSOutput doCall(Path p)
811          throws IOException, UnresolvedLinkException {
812        return createOutput(dfs, p.toUri().getPath(), overwrite, createParent, replication,
813          blockSize, eventLoopGroup, channelClass);
814      }
815
816      @Override
817      public FanOutOneBlockAsyncDFSOutput next(FileSystem fs, Path p) throws IOException {
818        throw new UnsupportedOperationException();
819      }
820    }.resolve(dfs, f);
821  }

824    // RetryStartFileException is introduced in HDFS 2.6+, so here we can only use the class name.

830  static void completeFile(DFSClient client, ClientProtocol namenode, String src, String clientName,
831      ExtendedBlock block, long fileId) {
832    for (int retry = 0;; retry++) {
833      try {
834        if (namenode.complete(src, clientName, block, fileId)) {
835          endFileLease(client, fileId);
836          return;
837        } else {
838          LOG.warn("complete file " + src + " not finished, retry = " + retry);
839        }
840      } catch (RemoteException e) {
841        IOException ioe = e.unwrapRemoteException();
842        if (ioe instanceof LeaseExpiredException) {
843          LOG.warn("lease for file " + src + " is expired, give up", e);
844          return;
845        } else {
846          LOG.warn("complete file " + src + " failed, retry = " + retry, e);
847        }
848      } catch (Exception e) {
849        LOG.warn("complete file " + src + " failed, retry = " + retry, e);
850      }
851      sleepIgnoreInterrupt(retry);
852    }
853  }
854
855  static void sleepIgnoreInterrupt(int retry) {
856    try {
857      Thread.sleep(ConnectionUtils.getPauseTime(100, retry));
858    } catch (InterruptedException e) {
859    }
860  }
