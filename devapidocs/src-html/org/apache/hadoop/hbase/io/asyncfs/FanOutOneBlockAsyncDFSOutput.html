079/**
080 * An asynchronous HDFS output stream implementation which fans out data to datanode and only
081 * supports writing file with only one block.
082 * <p>
083 * Use the createOutput method in {@link FanOutOneBlockAsyncDFSOutputHelper} to create. The mainly
084 * usage of this class is implementing WAL, so we only expose a little HDFS configurations in the
085 * method. And we place it here under io package because we want to make it independent of WAL
086 * implementation thus easier to move it to HDFS project finally.
087 * <p>
088 * Note that, although we support pipelined flush, i.e, write new data and then flush before the
089 * previous flush succeeds, the implementation is not thread safe, so you should not call its
090 * methods concurrently.
091 * <p>
092 * Advantages compare to DFSOutputStream:
093 * <ol>
094 * <li>The fan out mechanism. This will reduce the latency.</li>
095 * <li>Fail-fast when connection to datanode error. The WAL implementation could open new writer
096 * ASAP.</li>
097 * <li>We could benefit from netty's ByteBuf management mechanism.</li>
098 * </ol>
099 */
100@InterfaceAudience.Private

135  private static final class Callback {
136
137    private final CompletableFuture<Long> future;
138
139    private final long ackedLength;
140
141    // should be backed by a thread safe collection
142    private final Set<ChannelId> unfinishedReplicas;
143    private final long packetDataLen;
144    private final long flushTimestamp;
145    private long lastAckTimestamp = -1;
146
147    public Callback(CompletableFuture<Long> future, long ackedLength,
148      final Collection<Channel> replicas, long packetDataLen) {
149      this.future = future;
150      this.ackedLength = ackedLength;
151      this.packetDataLen = packetDataLen;
152      this.flushTimestamp = EnvironmentEdgeManager.currentTime();
153      if (replicas.isEmpty()) {
154        this.unfinishedReplicas = Collections.emptySet();
155      } else {
156        this.unfinishedReplicas =
157          Collections.newSetFromMap(new ConcurrentHashMap<ChannelId, Boolean>(replicas.size()));
158        replicas.stream().map(Channel::id).forEachOrdered(unfinishedReplicas::add);
159      }
160    }
161  }

163  private final ConcurrentLinkedDeque<Callback> waitingAckQueue = new ConcurrentLinkedDeque<>();

165  private volatile long ackedBlockLength = 0L;

167  // this could be different from acked block length because a packet can not start at the middle of
168  // a chunk.
169  private long nextPacketOffsetInBlock = 0L;

171  // the length of the trailing partial chunk, this is because the packet start offset must be
172  // aligned with the length of checksum chunk so we need to resend the same data.
173  private int trailingPartialChunkLength = 0;

181  // State for connections to DN
182  private enum State {
183    STREAMING,
184    CLOSING,
185    BROKEN,
186    CLOSED
187  }

189  private volatile State state;

191  private final StreamSlowMonitor streamSlowMonitor;

193  // all lock-free to make it run faster
194  private void completed(Channel channel) {
195    for (Iterator<Callback> iter = waitingAckQueue.iterator(); iter.hasNext();) {
196      Callback c = iter.next();
197      // if the current unfinished replicas does not contain us then it means that we have already
198      // acked this one, let's iterate to find the one we have not acked yet.
199      if (c.unfinishedReplicas.remove(channel.id())) {
200        long current = EnvironmentEdgeManager.currentTime();
201        streamSlowMonitor.checkProcessTimeAndSpeed(datanodeInfoMap.get(channel), c.packetDataLen,
202          current - c.flushTimestamp, c.lastAckTimestamp, c.unfinishedReplicas.size());
203        c.lastAckTimestamp = current;
204        if (c.unfinishedReplicas.isEmpty()) {
205          // we need to remove first before complete the future. It is possible that after we
206          // complete the future the upper layer will call close immediately before we remove the
207          // entry from waitingAckQueue and lead to an IllegalStateException. And also set the
208          // ackedBlockLength first otherwise we may use a wrong length to commit the block. This
209          // may lead to multiple remove and assign but is OK. The semantic of iter.remove is
210          // removing the entry returned by calling previous next, so if the entry has already been
211          // removed then it is a no-op, and for the assign, the values are the same so no problem.
212          iter.remove();
213          ackedBlockLength = c.ackedLength;
214          // the future.complete check is to confirm that we are the only one who grabbed the work,
215          // otherwise just give up and return.
216          if (c.future.complete(c.ackedLength)) {
217            // also wake up flush requests which have the same length.
218            while (iter.hasNext()) {
219              Callback maybeDummyCb = iter.next();
220              if (maybeDummyCb.ackedLength == c.ackedLength) {
221                iter.remove();
222                maybeDummyCb.future.complete(c.ackedLength);
223              } else {
224                break;
225              }
226            }
227          }
228        }
229        return;
230      }
231    }
232  }

234  // this usually does not happen which means it is not on the critical path so make it synchronized
235  // so that the implementation will not burn up our brain as there are multiple state changes and
236  // checks.
237  private synchronized void failed(Channel channel, Supplier<Throwable> errorSupplier) {
238    if (state == State.CLOSED) {
239      return;
240    }
241    if (state == State.BROKEN) {
242      failWaitingAckQueue(channel, errorSupplier);
243      return;
244    }
245    if (state == State.CLOSING) {
246      Callback c = waitingAckQueue.peekFirst();
247      if (c == null || !c.unfinishedReplicas.contains(channel.id())) {
248        // nothing, the endBlock request has already finished.
249        return;
250      }
251    }
252    // disable further write, and fail all pending ack.
253    state = State.BROKEN;
254    failWaitingAckQueue(channel, errorSupplier);
255    datanodeInfoMap.keySet().forEach(ChannelOutboundInvoker::close);
256  }
257
258  private void failWaitingAckQueue(Channel channel, Supplier<Throwable> errorSupplier) {
259    Throwable error = errorSupplier.get();
260    for (Iterator<Callback> iter = waitingAckQueue.iterator(); iter.hasNext();) {
261      Callback c = iter.next();
262      // find the first sync request which we have not acked yet and fail all the request after it.
263      if (!c.unfinishedReplicas.contains(channel.id())) {
264        continue;
265      }
266      for (;;) {
267        c.future.completeExceptionally(error);
268        if (!iter.hasNext()) {
269          break;
270        }
271        c = iter.next();
272      }
273      break;
274    }
275  }

277  @Sharable
278  private final class AckHandler extends SimpleChannelInboundHandler<PipelineAckProto> {
279
280    private final int timeoutMs;
281
282    public AckHandler(int timeoutMs) {
283      this.timeoutMs = timeoutMs;
284    }
285
286    @Override
287    protected void channelRead0(ChannelHandlerContext ctx, PipelineAckProto ack) throws Exception {
288      Status reply = getStatus(ack);
289      if (reply != Status.SUCCESS) {
290        failed(ctx.channel(), () -> new IOException("Bad response " + reply + " for block " + block
291          + " from datanode " + ctx.channel().remoteAddress()));
292        return;
293      }
294      if (PipelineAck.isRestartOOBStatus(reply)) {
295        failed(ctx.channel(), () -> new IOException("Restart response " + reply + " for block "
296          + block + " from datanode " + ctx.channel().remoteAddress()));
297        return;
298      }
299      if (ack.getSeqno() == HEART_BEAT_SEQNO) {
300        return;
301      }
302      completed(ctx.channel());
303    }
304
305    @Override
306    public void channelInactive(ChannelHandlerContext ctx) throws Exception {
307      if (state == State.CLOSED) {
308        return;
309      }
310      failed(ctx.channel(),
311        () -> new IOException("Connection to " + ctx.channel().remoteAddress() + " closed"));
312    }
313
314    @Override
315    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
316      failed(ctx.channel(), () -> cause);
317    }
318
319    @Override
320    public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
321      if (evt instanceof IdleStateEvent) {
322        IdleStateEvent e = (IdleStateEvent) evt;
323        if (e.state() == READER_IDLE) {
324          failed(ctx.channel(),
325            () -> new IOException("Timeout(" + timeoutMs + "ms) waiting for response"));
326        } else if (e.state() == WRITER_IDLE) {
327          PacketHeader heartbeat = new PacketHeader(4, 0, HEART_BEAT_SEQNO, false, 0, false);
328          int len = heartbeat.getSerializedSize();
329          ByteBuf buf = alloc.buffer(len);
330          heartbeat.putInBuffer(buf.nioBuffer(0, len));
331          buf.writerIndex(len);
332          ctx.channel().writeAndFlush(buf);
333        }
334        return;
335      }
336      super.userEventTriggered(ctx, evt);
337    }
338  }

340  private void setupReceiver(int timeoutMs) {
341    AckHandler ackHandler = new AckHandler(timeoutMs);
342    for (Channel ch : datanodeInfoMap.keySet()) {
343      ch.pipeline().addLast(
344        new IdleStateHandler(timeoutMs, timeoutMs / 2, 0, TimeUnit.MILLISECONDS),
345        new ProtobufVarint32FrameDecoder(),
346        new ProtobufDecoder(PipelineAckProto.getDefaultInstance()), ackHandler);
347      ch.config().setAutoRead(true);
348    }
349  }

408  private void flushBuffer(CompletableFuture<Long> future, ByteBuf dataBuf,
409    long nextPacketOffsetInBlock, boolean syncBlock) {
410    int dataLen = dataBuf.readableBytes();
411    int chunkLen = summer.getBytesPerChecksum();
412    int trailingPartialChunkLen = dataLen % chunkLen;
413    int numChecks = dataLen / chunkLen + (trailingPartialChunkLen != 0 ? 1 : 0);
414    int checksumLen = numChecks * summer.getChecksumSize();
415    ByteBuf checksumBuf = alloc.directBuffer(checksumLen);
416    summer.calculateChunkedSums(dataBuf.nioBuffer(), checksumBuf.nioBuffer(0, checksumLen));
417    checksumBuf.writerIndex(checksumLen);
418    PacketHeader header = new PacketHeader(4 + checksumLen + dataLen, nextPacketOffsetInBlock,
419      nextPacketSeqno, false, dataLen, syncBlock);
420    int headerLen = header.getSerializedSize();
421    ByteBuf headerBuf = alloc.buffer(headerLen);
422    header.putInBuffer(headerBuf.nioBuffer(0, headerLen));
423    headerBuf.writerIndex(headerLen);
424    Callback c =
425      new Callback(future, nextPacketOffsetInBlock + dataLen, datanodeInfoMap.keySet(), dataLen);
426    waitingAckQueue.addLast(c);
427    // recheck again after we pushed the callback to queue
428    if (state != State.STREAMING && waitingAckQueue.peekFirst() == c) {
429      future.completeExceptionally(new IOException("stream already broken"));
430      // it's the one we have just pushed or just a no-op
431      waitingAckQueue.removeFirst();
432
433      checksumBuf.release();
434      headerBuf.release();
435
436      // This method takes ownership of the dataBuf so we need release it before returning.
437      dataBuf.release();
438      return;
439    }
440    // TODO: we should perhaps measure time taken per DN here;
441    // we could collect statistics per DN, and/or exclude bad nodes in createOutput.
442    datanodeInfoMap.keySet().forEach(ch -> {
443      ch.write(headerBuf.retainedDuplicate());
444      ch.write(checksumBuf.retainedDuplicate());
445      ch.writeAndFlush(dataBuf.retainedDuplicate());
446    });
447    checksumBuf.release();
448    headerBuf.release();
449    dataBuf.release();
450    nextPacketSeqno++;
451  }

453  private void flush0(CompletableFuture<Long> future, boolean syncBlock) {
454    if (state != State.STREAMING) {
455      future.completeExceptionally(new IOException("stream already broken"));
456      return;
457    }
458    int dataLen = buf.readableBytes();
459    if (dataLen == trailingPartialChunkLength) {
460      // no new data
461      long lengthAfterFlush = nextPacketOffsetInBlock + dataLen;
462      Callback lastFlush = waitingAckQueue.peekLast();
463      if (lastFlush != null) {
464        Callback c = new Callback(future, lengthAfterFlush, Collections.emptySet(), dataLen);
465        waitingAckQueue.addLast(c);
466        // recheck here if we have already removed the previous callback from the queue
467        if (waitingAckQueue.peekFirst() == c) {
468          // all previous callbacks have been removed
469          // notice that this does mean we will always win here because the background thread may
470          // have already started to mark the future here as completed in the completed or failed
471          // methods but haven't removed it from the queue yet. That's also why the removeFirst
472          // call below may be a no-op.
473          if (state != State.STREAMING) {
474            future.completeExceptionally(new IOException("stream already broken"));
475          } else {
476            future.complete(lengthAfterFlush);
477          }
478          // it's the one we have just pushed or just a no-op
479          waitingAckQueue.removeFirst();
480        }
481      } else {
482        // we must have acked all the data so the ackedBlockLength must be same with
483        // lengthAfterFlush
484        future.complete(lengthAfterFlush);
485      }
486      return;
487    }
488
489    if (encryptor != null) {
490      ByteBuf encryptBuf = alloc.directBuffer(dataLen);
491      buf.readBytes(encryptBuf, trailingPartialChunkLength);
492      int toEncryptLength = dataLen - trailingPartialChunkLength;
493      try {
494        encryptor.encrypt(buf.nioBuffer(trailingPartialChunkLength, toEncryptLength),
495          encryptBuf.nioBuffer(trailingPartialChunkLength, toEncryptLength));
496      } catch (IOException e) {
497        encryptBuf.release();
498        future.completeExceptionally(e);
499        return;
500      }
501      encryptBuf.writerIndex(dataLen);
502      buf.release();
503      buf = encryptBuf;
504    }
505
506    if (dataLen > maxDataLen) {
507      // We need to write out the data by multiple packets as the max packet allowed is 16M.
508      long nextSubPacketOffsetInBlock = nextPacketOffsetInBlock;
509      for (int remaining = dataLen;;) {
510        if (remaining < maxDataLen) {
511          flushBuffer(future, buf.readRetainedSlice(remaining), nextSubPacketOffsetInBlock,
512            syncBlock);
513          break;
514        } else {
515          flushBuffer(new CompletableFuture<>(), buf.readRetainedSlice(maxDataLen),
516            nextSubPacketOffsetInBlock, syncBlock);
517          remaining -= maxDataLen;
518          nextSubPacketOffsetInBlock += maxDataLen;
519        }
520      }
521    } else {
522      flushBuffer(future, buf.retain(), nextPacketOffsetInBlock, syncBlock);
523    }
524    trailingPartialChunkLength = dataLen % summer.getBytesPerChecksum();
525    ByteBuf newBuf = alloc.directBuffer(sendBufSizePRedictor.guess(dataLen))
526      .ensureWritable(trailingPartialChunkLength);
527    if (trailingPartialChunkLength != 0) {
528      buf.readerIndex(dataLen - trailingPartialChunkLength).readBytes(newBuf,
529        trailingPartialChunkLength);
530    }
531    buf.release();
532    this.buf = newBuf;
533    nextPacketOffsetInBlock += dataLen - trailingPartialChunkLength;
534  }

536  /**
537   * Flush the buffer out to datanodes.
538   * @param syncBlock will call hsync if true, otherwise hflush.
539   * @return A CompletableFuture that hold the acked length after flushing.
540   */
541  @Override
542  public CompletableFuture<Long> flush(boolean syncBlock) {
543    CompletableFuture<Long> future = new CompletableFuture<>();
544    flush0(future, syncBlock);
545    return future;
546  }

548  private void endBlock() throws IOException {
549    Preconditions.checkState(waitingAckQueue.isEmpty(),
550      "should call flush first before calling close");
551    if (state != State.STREAMING) {
552      throw new IOException("stream already broken");
553    }
554    state = State.CLOSING;
555    long finalizedLength = ackedBlockLength;
556    PacketHeader header = new PacketHeader(4, finalizedLength, nextPacketSeqno, true, 0, false);
557    buf.release();
558    buf = null;
559    int headerLen = header.getSerializedSize();
560    ByteBuf headerBuf = alloc.directBuffer(headerLen);
561    header.putInBuffer(headerBuf.nioBuffer(0, headerLen));
562    headerBuf.writerIndex(headerLen);
563    CompletableFuture<Long> future = new CompletableFuture<>();
564    waitingAckQueue.add(new Callback(future, finalizedLength, datanodeInfoMap.keySet(), 0));
565    datanodeInfoMap.keySet().forEach(ch -> ch.writeAndFlush(headerBuf.retainedDuplicate()));
566    headerBuf.release();
567    try {
568      future.get();
569    } catch (InterruptedException e) {
570      throw (IOException) new InterruptedIOException().initCause(e);
571    } catch (ExecutionException e) {
572      Throwable cause = e.getCause();
573      Throwables.propagateIfPossible(cause, IOException.class);
574      throw new IOException(cause);
575    }
576  }

578  /**
579   * The close method when error occurred. Now we just call recoverFileLease.
580   */
581  @Override
582  @SuppressWarnings("FutureReturnValueIgnored")
583  public void recoverAndClose(CancelableProgressable reporter) throws IOException {
584    if (buf != null) {
585      buf.release();
586      buf = null;
587    }
588    datanodeInfoMap.keySet().forEach(ChannelOutboundInvoker::close);
589    datanodeInfoMap.keySet().forEach(ch -> ch.closeFuture().awaitUninterruptibly());
590    endFileLease(client, fileId);
591    RecoverLeaseFSUtils.recoverFileLease(dfs, new Path(src), conf,
592      reporter == null ? new CancelOnClose(client) : reporter);
593  }

595  /**
596   * End the current block and complete file at namenode. You should call
597   * {@link #recoverAndClose(CancelableProgressable)} if this method throws an exception.
598   */
599  @Override
600  @SuppressWarnings("FutureReturnValueIgnored")
601  public void close() throws IOException {
602    endBlock();
603    state = State.CLOSED;
604    datanodeInfoMap.keySet().forEach(ChannelOutboundInvoker::close);
605    datanodeInfoMap.keySet().forEach(ch -> ch.closeFuture().awaitUninterruptibly());
606    block.setNumBytes(ackedBlockLength);
607    completeFile(client, namenode, src, clientName, block, fileId);
608  }
