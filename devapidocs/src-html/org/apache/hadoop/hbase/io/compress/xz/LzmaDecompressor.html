018package org.apache.hadoop.hbase.io.compress.xz;
019
020import java.io.IOException;
021import java.nio.ByteBuffer;
022import org.apache.hadoop.hbase.io.ByteBufferInputStream;
023import org.apache.hadoop.hbase.io.compress.CompressionUtil;
024import org.apache.hadoop.io.compress.Decompressor;
025import org.apache.yetus.audience.InterfaceAudience;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.tukaani.xz.ArrayCache;
029import org.tukaani.xz.BasicArrayCache;
030import org.tukaani.xz.LZMAInputStream;
031
032/**
033 * Hadoop decompressor glue for XZ for Java.
034 */
035@InterfaceAudience.Private
036public class LzmaDecompressor implements Decompressor {
037
038  protected static final Logger LOG = LoggerFactory.getLogger(LzmaDecompressor.class);
039  protected static final ArrayCache ARRAY_CACHE = new BasicArrayCache() {
040    @Override
041    public byte[] getByteArray(int size, boolean fillWithZeros) {
042      // Work around a bug in XZ decompression if cached byte arrays are not cleared by
043      // always clearing them.
044      return super.getByteArray(size, true);
045    }
046  };
047  protected ByteBuffer inBuf, outBuf;
048  protected int inLen;
049  protected boolean finished;
050
051  LzmaDecompressor(int bufferSize) {
052    this.inBuf = ByteBuffer.allocate(bufferSize);
053    this.outBuf = ByteBuffer.allocate(bufferSize);
054    this.outBuf.position(bufferSize);
055  }
056
057  @Override
058  public int decompress(byte[] b, int off, int len) throws IOException {
059    if (outBuf.hasRemaining()) {
060      int remaining = outBuf.remaining(), n = Math.min(remaining, len);
061      outBuf.get(b, off, n);
062      LOG.trace("decompress: read {} remaining bytes from outBuf", n);
063      return n;
064    }
065    if (inBuf.position() > 0) {
066      inBuf.flip();
067      int remaining = inBuf.remaining();
068      inLen -= remaining;
069      // This is pretty ugly. I don't see how to do it better. Stream to byte buffers back to
070      // stream back to byte buffers... if only XZ for Java had a public block compression API.
071      // It does not. LZMA decompression speed is reasonably good, so inefficiency here is a
072      // shame.
073      // Perhaps we could look at using reflection to make package protected classes for block
074      // compression in XZ for Java accessible here, that library can be expected to rarely
075      // change, if at all.
076      outBuf.clear();
077      try (ByteBufferInputStream lowerIn = new ByteBufferInputStream(inBuf)) {
078        final byte[] buf = new byte[8192];
079        try (LZMAInputStream in = new LZMAInputStream(lowerIn, ARRAY_CACHE)) {
080          int read;
081          do {
082            read = in.read(buf);
083            if (read > 0) {
084              outBuf.put(buf, 0, read);
085            }
086          } while (read > 0);
087        }
088      }
089      int written = outBuf.position();
090      outBuf.flip();
091      inBuf.clear();
092      LOG.trace("decompress: decompressed {} -> {}", remaining, written);
093      int n = Math.min(written, len);
094      outBuf.get(b, off, n);
095      LOG.trace("decompress: {} bytes", n);
096      return n;
097    }
098    LOG.trace("decompress: No output, finished");
099    finished = true;
100    return 0;
101  }
102
103  @Override
104  public void end() {
105    LOG.trace("end");
106  }
107
108  @Override
109  public boolean finished() {
110    LOG.trace("finished");
111    return finished;
112  }
113
114  @Override
115  public int getRemaining() {
116    LOG.trace("getRemaining: {}", inLen);
117    return inLen;
118  }
119
120  @Override
121  public boolean needsDictionary() {
122    LOG.trace("needsDictionary");
123    return false;
124  }
125
126  @Override
127  public void reset() {
128    LOG.trace("reset");
129    inBuf.clear();
130    inLen = 0;
131    outBuf.clear();
132    outBuf.position(outBuf.capacity());
133    finished = false;
134  }
135
136  @Override
137  public boolean needsInput() {
138    boolean b = (inBuf.position() == 0);
139    LOG.trace("needsInput: {}", b);
140    return b;
141  }
142
143  @Override
144  public void setDictionary(byte[] b, int off, int len) {
145    throw new UnsupportedOperationException("setDictionary is not supported");
146  }
147
148  @Override
149  public void setInput(byte[] b, int off, int len) {
150    LOG.trace("setInput: off={} len={}", off, len);
151    if (inBuf.remaining() < len) {
152      // Get a new buffer that can accomodate the accumulated input plus the additional
153      // input that would cause a buffer overflow without reallocation.
154      // This condition should be fortunately rare, because it is expensive.
155      int needed = CompressionUtil.roundInt2(inBuf.capacity() + len);
156      LOG.trace("setInput: resize inBuf {}", needed);
157      ByteBuffer newBuf = ByteBuffer.allocate(needed);
158      inBuf.flip();
159      newBuf.put(inBuf);
160      inBuf = newBuf;
161    }
162    inBuf.put(b, off, len);
163    inLen += len;
164    finished = false;
165  }
166
167}
