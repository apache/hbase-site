018package org.apache.hadoop.hbase.io.compress.brotli;
019
020import com.aayushatharva.brotli4j.Brotli4jLoader;
021import com.aayushatharva.brotli4j.encoder.Encoder;
022import com.aayushatharva.brotli4j.encoder.Encoders;
023import java.io.IOException;
024import java.nio.ByteBuffer;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.io.compress.CanReinit;
027import org.apache.hadoop.hbase.io.compress.CompressionUtil;
028import org.apache.hadoop.io.compress.Compressor;
029import org.apache.yetus.audience.InterfaceAudience;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032
033/**
034 * Hadoop compressor glue for Brotli4j
035 */
036@InterfaceAudience.Private
037public class BrotliCompressor implements CanReinit, Compressor {
038
039  protected static final Logger LOG = LoggerFactory.getLogger(BrotliCompressor.class);
040  protected ByteBuffer inBuf, outBuf;
041  protected int bufferSize;
042  protected boolean finish, finished;
043  protected long bytesRead, bytesWritten;
044  protected Encoder.Parameters params;
045
046  static {
047    Brotli4jLoader.ensureAvailability();
048  }
049
050  BrotliCompressor(int level, int window, int bufferSize) {
051    this.bufferSize = bufferSize;
052    this.inBuf = ByteBuffer.allocate(bufferSize);
053    this.outBuf = ByteBuffer.allocate(bufferSize);
054    this.outBuf.position(bufferSize);
055    params = new Encoder.Parameters();
056    params.setQuality(level);
057    params.setWindow(window);
058  }
059
060  @Override
061  public int compress(byte[] b, int off, int len) throws IOException {
062    // If we have previously compressed our input and still have some buffered bytes
063    // remaining, provide them to the caller.
064    if (outBuf.hasRemaining()) {
065      int remaining = outBuf.remaining(), n = Math.min(remaining, len);
066      outBuf.get(b, off, n);
067      LOG.trace("compress: read {} remaining bytes from outBuf", n);
068      return n;
069    }
070    // We don't actually begin compression until our caller calls finish().
071    if (finish) {
072      if (inBuf.position() > 0) {
073        inBuf.flip();
074        int uncompressed = inBuf.remaining();
075        // If we don't have enough capacity in our currently allocated output buffer,
076        // allocate a new one which does.
077        int needed = maxCompressedLength(uncompressed);
078        // Can we compress directly into the provided array?
079        boolean direct = false;
080        ByteBuffer writeBuf;
081        if (len <= needed) {
082          direct = true;
083          writeBuf = ByteBuffer.wrap(b, off, len);
084        } else {
085          if (outBuf.capacity() < needed) {
086            needed = CompressionUtil.roundInt2(needed);
087            LOG.trace("compress: resize outBuf {}", needed);
088            outBuf = ByteBuffer.allocate(needed);
089          } else {
090            outBuf.clear();
091          }
092          writeBuf = outBuf;
093        }
094        final int oldPos = writeBuf.position();
095        Encoders.compress(inBuf, writeBuf, params);
096        final int written = writeBuf.position() - oldPos;
097        bytesWritten += written;
098        inBuf.clear();
099        LOG.trace("compress: compressed {} -> {}", uncompressed, written);
100        finished = true;
101        if (!direct) {
102          outBuf.flip();
103          int n = Math.min(written, len);
104          outBuf.get(b, off, n);
105          LOG.trace("compress: {} bytes", n);
106          return n;
107        } else {
108          LOG.trace("compress: {} bytes direct", written);
109          return written;
110        }
111      } else {
112        finished = true;
113      }
114    }
115    LOG.trace("No output");
116    return 0;
117  }
118
119  @Override
120  public void end() {
121    LOG.trace("end");
122  }
123
124  @Override
125  public void finish() {
126    LOG.trace("finish");
127    finish = true;
128  }
129
130  @Override
131  public boolean finished() {
132    boolean b = finished && !outBuf.hasRemaining();
133    LOG.trace("finished: {}", b);
134    return b;
135  }
136
137  @Override
138  public long getBytesRead() {
139    return bytesRead;
140  }
141
142  @Override
143  public long getBytesWritten() {
144    return bytesWritten;
145  }
146
147  @Override
148  public boolean needsInput() {
149    boolean b = !finished();
150    LOG.trace("needsInput: {}", b);
151    return b;
152  }
153
154  @Override
155  public void reinit(Configuration conf) {
156    LOG.trace("reinit");
157    if (conf != null) {
158      // Quality or window settings might have changed
159      params.setQuality(BrotliCodec.getLevel(conf));
160      params.setWindow(BrotliCodec.getWindow(conf));
161      // Buffer size might have changed
162      int newBufferSize = BrotliCodec.getBufferSize(conf);
163      if (bufferSize != newBufferSize) {
164        bufferSize = newBufferSize;
165        this.inBuf = ByteBuffer.allocate(bufferSize);
166        this.outBuf = ByteBuffer.allocate(bufferSize);
167      }
168    }
169    reset();
170  }
171
172  @Override
173  public void reset() {
174    LOG.trace("reset");
175    inBuf.clear();
176    outBuf.clear();
177    outBuf.position(outBuf.capacity());
178    bytesRead = 0;
179    bytesWritten = 0;
180    finish = false;
181    finished = false;
182  }
183
184  @Override
185  public void setDictionary(byte[] b, int off, int len) {
186    throw new UnsupportedOperationException("setDictionary is not supported");
187  }
188
189  @Override
190  public void setInput(byte[] b, int off, int len) {
191    LOG.trace("setInput: off={} len={}", off, len);
192    if (inBuf.remaining() < len) {
193      // Get a new buffer that can accomodate the accumulated input plus the additional
194      // input that would cause a buffer overflow without reallocation.
195      // This condition should be fortunately rare, because it is expensive.
196      int needed = CompressionUtil.roundInt2(inBuf.capacity() + len);
197      LOG.trace("setInput: resize inBuf {}", needed);
198      ByteBuffer newBuf = ByteBuffer.allocate(needed);
199      inBuf.flip();
200      newBuf.put(inBuf);
201      inBuf = newBuf;
202    }
203    inBuf.put(b, off, len);
204    bytesRead += len;
205    finished = false;
206  }
207
208  // Package private
209
210  int maxCompressedLength(int len) {
211    return len + CompressionUtil.compressionOverhead(len);
212  }
213
214}
