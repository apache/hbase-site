017package org.apache.hadoop.hbase.io.compress.lz4;
018
019import java.io.IOException;
020import java.nio.ByteBuffer;
021import org.apache.hadoop.conf.Configuration;
022import org.apache.hadoop.hbase.io.compress.CompressionUtil;
023import org.apache.hadoop.io.compress.Compressor;
024import org.apache.yetus.audience.InterfaceAudience;
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027import net.jpountz.lz4.LZ4Compressor;
028import net.jpountz.lz4.LZ4Factory;
029
030/**
031 * Hadoop compressor glue for lz4-java.
032 */
033@InterfaceAudience.Private
034public class Lz4Compressor implements Compressor {
035
036  protected static final Logger LOG = LoggerFactory.getLogger(Lz4Compressor.class);
037  protected LZ4Compressor compressor;
038  protected ByteBuffer inBuf, outBuf;
039  protected int bufferSize;
040  protected boolean finish, finished;
041  protected long bytesRead, bytesWritten;
042
043  Lz4Compressor(int bufferSize) {
044    compressor = LZ4Factory.fastestInstance().fastCompressor();
045    this.bufferSize = bufferSize;
046    this.inBuf = ByteBuffer.allocate(bufferSize);
047    this.outBuf = ByteBuffer.allocate(bufferSize);
048    this.outBuf.position(bufferSize);
049  }
050
051  @Override
052  public int compress(byte[] b, int off, int len) throws IOException {
053    // If we have previously compressed our input and still have some buffered bytes
054    // remaining, provide them to the caller.
055    if (outBuf.hasRemaining()) {
056      int remaining = outBuf.remaining(), n = Math.min(remaining, len);
057      outBuf.get(b, off, n);
058      LOG.trace("compress: {} bytes from outBuf", n);
059      return n;
060    }
061    // We don't actually begin compression until our caller calls finish().
062    if (finish) {
063      if (inBuf.position() > 0) {
064        inBuf.flip();
065        int uncompressed = inBuf.remaining();
066        int needed = maxCompressedLength(uncompressed);
067        // Can we decompress directly into the provided array?
068        ByteBuffer writeBuffer;
069        boolean direct = false;
070        if (len <= needed) {
071          writeBuffer = ByteBuffer.wrap(b, off, len);
072          direct = true;
073        } else {
074          // If we don't have enough capacity in our currently allocated output buffer,
075          // allocate a new one which does.
076          if (outBuf.capacity() < needed) {
077            needed = CompressionUtil.roundInt2(needed);
078            LOG.trace("compress: resize outBuf {}", needed);
079            outBuf = ByteBuffer.allocate(needed);
080          } else {
081            outBuf.clear();
082          }
083          writeBuffer = outBuf;
084        }
085        final int oldPos = writeBuffer.position();
086        compressor.compress(inBuf, writeBuffer);
087        final int written = writeBuffer.position() - oldPos;
088        bytesWritten += written;
089        inBuf.clear();
090        LOG.trace("compress: compressed {} -> {}", uncompressed, written);
091        finished = true;
092        if (!direct) {
093          outBuf.flip();
094          int n = Math.min(written, len);
095          outBuf.get(b, off, n);
096          LOG.trace("compress: {} bytes", n);
097          return n;
098        } else {
099          LOG.trace("compress: {} bytes direct", written);
100          return written;
101        }
102      } else {
103        finished = true;
104      }
105    }
106    LOG.trace("No output");
107    return 0;
108  }
109
110  @Override
111  public void end() {
112    LOG.trace("end");
113  }
114
115  @Override
116  public void finish() {
117    LOG.trace("finish");
118    finish = true;
119  }
120
121  @Override
122  public boolean finished() {
123    boolean b = finished && !outBuf.hasRemaining();
124    LOG.trace("finished: {}", b);
125    return b;
126  }
127
128  @Override
129  public long getBytesRead() {
130    return bytesRead;
131  }
132
133  @Override
134  public long getBytesWritten() {
135    return bytesWritten;
136  }
137
138  @Override
139  public boolean needsInput() {
140    boolean b = !finished();
141    LOG.trace("needsInput: {}", b);
142    return b;
143  }
144
145  @Override
146  public void reinit(Configuration conf) {
147    LOG.trace("reinit");
148    if (conf != null) {
149      // Buffer size might have changed
150      int newBufferSize = Lz4Codec.getBufferSize(conf);
151      if (bufferSize != newBufferSize) {
152        bufferSize = newBufferSize;
153        this.inBuf = ByteBuffer.allocate(bufferSize);
154        this.outBuf = ByteBuffer.allocate(bufferSize);
155      }
156    }
157    reset();
158  }
159
160  @Override
161  public void reset() {
162    LOG.trace("reset");
163    compressor = LZ4Factory.fastestInstance().fastCompressor();
164    inBuf.clear();
165    outBuf.clear();
166    outBuf.position(outBuf.capacity());
167    bytesRead = 0;
168    bytesWritten = 0;
169    finish = false;
170    finished = false;
171  }
172
173  @Override
174  public void setDictionary(byte[] b, int off, int len) {
175    throw new UnsupportedOperationException("setDictionary is not supported");
176  }
177
178  @Override
179  public void setInput(byte[] b, int off, int len) {
180    LOG.trace("setInput: off={} len={}", off, len);
181    if (inBuf.remaining() < len) {
182      // Get a new buffer that can accomodate the accumulated input plus the additional
183      // input that would cause a buffer overflow without reallocation.
184      // This condition should be fortunately rare, because it is expensive.
185      int needed = CompressionUtil.roundInt2(inBuf.capacity() + len);
186      LOG.trace("setInput: resize inBuf {}", needed);
187      ByteBuffer newBuf = ByteBuffer.allocate(needed);
188      inBuf.flip();
189      newBuf.put(inBuf);
190      inBuf = newBuf;
191    }
192    inBuf.put(b, off, len);
193    bytesRead += len;
194    finished = false;
195  }
196
197  // Package private
198
199  int maxCompressedLength(int len) {
200    return compressor.maxCompressedLength(len);
201  }
202
203}
