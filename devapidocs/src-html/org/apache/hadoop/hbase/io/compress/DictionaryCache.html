017package org.apache.hadoop.hbase.io.compress;
018
019import java.io.ByteArrayOutputStream;
020import java.io.FileNotFoundException;
021import java.io.IOException;
022import java.io.InputStream;
023import java.util.concurrent.ExecutionException;
024import java.util.concurrent.TimeUnit;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.FSDataInputStream;
027import org.apache.hadoop.fs.FileStatus;
028import org.apache.hadoop.fs.FileSystem;
029import org.apache.hadoop.fs.Path;
030import org.apache.yetus.audience.InterfaceAudience;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033import org.apache.hbase.thirdparty.com.google.common.cache.CacheBuilder;
034import org.apache.hbase.thirdparty.com.google.common.cache.CacheLoader;
035import org.apache.hbase.thirdparty.com.google.common.cache.LoadingCache;
036
037/**
038 * A utility class for managing compressor/decompressor dictionary loading and caching of load
039 * results. Useful for any codec that can support changing dictionaries at runtime,
040 * such as ZStandard.
041 */
042@InterfaceAudience.Private
043public class DictionaryCache {
044
045  public static final String DICTIONARY_MAX_SIZE_KEY = "hbase.io.compress.dictionary.max.size";
046  public static final int DEFAULT_DICTIONARY_MAX_SIZE = 10 * 1024 * 1024;
047  public static final String RESOURCE_SCHEME = "resource://";
048
049  private static final Logger LOG = LoggerFactory.getLogger(DictionaryCache.class);
050  private static volatile LoadingCache<String, byte[]> CACHE;
051
052  private DictionaryCache() { }
053
054  /**
055   * Load a dictionary or return a previously cached load.
056   * @param conf configuration
057   * @param path the hadoop Path where the dictionary is located, as a String
058   * @return the dictionary bytes if successful, null otherwise
059   */
060  public static byte[] getDictionary(final Configuration conf, final String path)
061      throws IOException {
062    if (path == null || path.isEmpty()) {
063      return null;
064    }
065    // Create the dictionary loading cache if we haven't already
066    if (CACHE == null) {
067      synchronized (DictionaryCache.class) {
068        if (CACHE == null) {
069          final int maxSize = conf.getInt(DICTIONARY_MAX_SIZE_KEY, DEFAULT_DICTIONARY_MAX_SIZE);
070          CACHE = CacheBuilder.newBuilder()
071            .maximumSize(100)
072            .expireAfterAccess(10, TimeUnit.MINUTES)
073            .build(
074              new CacheLoader<String, byte[]>() {
075                public byte[] load(String s) throws Exception {
076                  byte[] bytes;
077                  if (path.startsWith(RESOURCE_SCHEME)) {
078                    bytes = loadFromResource(conf, path, maxSize);
079                  } else {
080                    bytes = loadFromHadoopFs(conf, path, maxSize);
081                  }
082                  LOG.info("Loaded dictionary from {} (size {})", s, bytes.length);
083                  return bytes;
084                }
085              });
086        }
087      }
088    }
089
090    // Get or load the dictionary for the given path
091    try {
092      return CACHE.get(path);
093    } catch (ExecutionException e) {
094      throw new RuntimeException(e);
095    }
096  }
097
098  // Visible for testing
099  public static byte[] loadFromResource(final Configuration conf, final String s,
100      final int maxSize) throws IOException {
101    if (!s.startsWith(RESOURCE_SCHEME)) {
102      throw new IllegalArgumentException("Path does not start with " + RESOURCE_SCHEME);
103    }
104    final String path = s.substring(RESOURCE_SCHEME.length(), s.length());
105    final InputStream in = DictionaryCache.class.getClassLoader().getResourceAsStream(path);
106    if (in == null) {
107      throw new FileNotFoundException("Resource " + path + " not found");
108    }
109    final ByteArrayOutputStream baos = new ByteArrayOutputStream();
110    final byte[] buffer = new byte[8192];
111    try {
112      int n, len = 0;
113      do {
114        n = in.read(buffer);
115        if (n > 0) {
116          len += n;
117          if (len > maxSize) {
118            throw new IllegalArgumentException("Dictionary " + s + " is too large" +
119              ", limit=" + maxSize);
120          }
121          baos.write(buffer, 0, n);
122        }
123      } while (n > 0);
124    } finally {
125      in.close();
126    }
127    return baos.toByteArray();
128  }
129
130  private static byte[] loadFromHadoopFs(final Configuration conf, final String s,
131      final int maxSize) throws IOException {
132    final Path path = new Path(s);
133    final FileSystem fs = FileSystem.get(path.toUri(), conf);
134    final FileStatus stat = fs.getFileStatus(path);
135    if (!stat.isFile()) {
136      throw new IllegalArgumentException(s + " is not a file");
137    }
138    if (stat.getLen() > maxSize) {
139      throw new IllegalArgumentException("Dictionary " + s + " is too large" +
140        ", size=" + stat.getLen() + ", limit=" + maxSize);
141    }
142    final ByteArrayOutputStream baos = new ByteArrayOutputStream();
143    final byte[] buffer = new byte[8192];
144    try (final FSDataInputStream in = fs.open(path)) {
145      int n;
146      do {
147        n = in.read(buffer);
148        if (n > 0) {
149          baos.write(buffer, 0, n);
150        }
151      } while (n > 0);
152    }
153    return baos.toByteArray();
154  }
155
156  // Visible for testing
157  public static boolean contains(String dictionaryPath) {
158    if (CACHE != null) {
159      return CACHE.asMap().containsKey(dictionaryPath);
160    }
161    return false;
162  }
