049/**
050 * Class to store blocks into memcached.
051 * This should only be used on a cluster of Memcached daemons that are tuned well and have a
052 * good network connection to the HBase regionservers. Any other use will likely slow down HBase
053 * greatly.
054 */
055@InterfaceAudience.Private
056public class MemcachedBlockCache implements BlockCache {
057  private static final Log LOG = LogFactory.getLog(MemcachedBlockCache.class.getName());
058
059  // Some memcache versions won't take more than 1024 * 1024. So set the limit below
060  // that just in case this client is used with those versions.
061  public static final int MAX_SIZE = 1020 * 1024;
062
063  // Config key for what memcached servers to use.
064  // They should be specified in a comma sperated list with ports.
065  // like:
066  //
067  // host1:11211,host3:8080,host4:11211
068  public static final String MEMCACHED_CONFIG_KEY = "hbase.cache.memcached.servers";
069  public static final String MEMCACHED_TIMEOUT_KEY = "hbase.cache.memcached.timeout";
070  public static final String MEMCACHED_OPTIMEOUT_KEY = "hbase.cache.memcached.optimeout";
071  public static final long MEMCACHED_DEFAULT_TIMEOUT = 500;
072
073  private final MemcachedClient client;
074  private final HFileBlockTranscoder tc = new HFileBlockTranscoder();
075  private final CacheStats cacheStats = new CacheStats("MemcachedBlockCache");
076
077  public MemcachedBlockCache(Configuration c) throws IOException {
078    LOG.info("Creating MemcachedBlockCache");
079
080    long opTimeout = c.getLong(MEMCACHED_OPTIMEOUT_KEY, MEMCACHED_DEFAULT_TIMEOUT);
081    long queueTimeout = c.getLong(MEMCACHED_TIMEOUT_KEY, opTimeout + MEMCACHED_DEFAULT_TIMEOUT);
082
083    ConnectionFactoryBuilder builder = new ConnectionFactoryBuilder()
084        .setOpTimeout(opTimeout)
085        .setOpQueueMaxBlockTime(queueTimeout) // Cap the max time before anything times out
086        .setFailureMode(FailureMode.Redistribute)
087        .setShouldOptimize(true)              // When regions move lots of reads happen together
088                                              // So combining them into single requests is nice.
089        .setDaemon(true)                      // Don't keep threads around past the end of days.
090        .setUseNagleAlgorithm(false)          // Ain't nobody got time for that
091        .setReadBufferSize(HConstants.DEFAULT_BLOCKSIZE * 4 * 1024);  // 4 times larger than the
092                                                                      // default block just in case
093
094
095    // Assume only the localhost is serving memecached.
096    // A la mcrouter or co-locating memcached with split regionservers.
097    //
098    // If this config is a pool of memecached servers they will all be used according to the
099    // default hashing scheme defined by the memcache client. Spy Memecache client in this
100    // case.
101    String serverListString = c.get(MEMCACHED_CONFIG_KEY,"localhost:11211");
102    String[] servers = serverListString.split(",");
103    List<InetSocketAddress> serverAddresses = new ArrayList<InetSocketAddress>(servers.length);
104    for (String s:servers) {
105      serverAddresses.add(Addressing.createInetSocketAddressFromHostAndPortStr(s));
106    }
107
108    client = new MemcachedClient(builder.build(), serverAddresses);
109  }
110
111  @Override
112  public void cacheBlock(BlockCacheKey cacheKey,
113                         Cacheable buf,
114                         boolean inMemory,
115                         boolean cacheDataInL1) {
116    cacheBlock(cacheKey, buf);
117  }
118
119  @Override
120  public void cacheBlock(BlockCacheKey cacheKey, Cacheable buf) {
121    if (buf instanceof HFileBlock) {
122      client.add(cacheKey.toString(), MAX_SIZE, (HFileBlock) buf, tc);
123    } else {
124      if (LOG.isDebugEnabled()) {
125        LOG.debug("MemcachedBlockCache can not cache Cacheable's of type "
126            + buf.getClass().toString());
127      }
128    }
129  }
130
131  @Override
132  public Cacheable getBlock(BlockCacheKey cacheKey, boolean caching,
133                            boolean repeat, boolean updateCacheMetrics) {
134    // Assume that nothing is the block cache
135    HFileBlock result = null;
136
137    try (TraceScope traceScope = Trace.startSpan("MemcachedBlockCache.getBlock")) {
138      result = client.get(cacheKey.toString(), tc);
139    } catch (Exception e) {
140      // Catch a pretty broad set of exceptions to limit any changes in the memecache client
141      // and how it handles failures from leaking into the read path.
142      if (LOG.isDebugEnabled()) {
143        LOG.debug("Exception pulling from memcached [ "
144            + cacheKey.toString()
145            + " ]. Treating as a miss.", e);
146      }
147      result = null;
148    } finally {
149      // Update stats if this request doesn't have it turned off 100% of the time
150      if (updateCacheMetrics) {
151        if (result == null) {
152          cacheStats.miss(caching, cacheKey.isPrimary());
153        } else {
154          cacheStats.hit(caching, cacheKey.isPrimary());
155        }
156      }
157    }
158
159
160    return result;
161  }
162
163  @Override
164  public boolean evictBlock(BlockCacheKey cacheKey) {
165    try {
166      cacheStats.evict();
167      return client.delete(cacheKey.toString()).get();
168    } catch (InterruptedException e) {
169      LOG.warn("Error deleting " + cacheKey.toString(), e);
170      Thread.currentThread().interrupt();
171    } catch (ExecutionException e) {
172      if (LOG.isDebugEnabled()) {
173        LOG.debug("Error deleting " + cacheKey.toString(), e);
174      }
175    }
176    return false;
177  }

217  @Override
218  public Iterator<CachedBlock> iterator() {
219    return new Iterator<CachedBlock>() {
220      @Override
221      public boolean hasNext() {
222        return false;
223      }
224
225      @Override
226      public CachedBlock next() {
227        throw new NoSuchElementException("MemcachedBlockCache can't iterate over blocks.");
228      }
229
230      @Override

232
233      }
234    };
235  }

239    return null;
240  }

242  /**
243   * Class to encode and decode an HFileBlock to and from memecached's resulting byte arrays.
244   */
245  private static class HFileBlockTranscoder implements Transcoder<HFileBlock> {
246
247    @Override
248    public boolean asyncDecode(CachedData d) {
249      return false;
250    }
251
252    @Override
253    public CachedData encode(HFileBlock block) {
254      ByteBuffer bb = ByteBuffer.allocate(block.getSerializedLength());
255      block.serialize(bb);
256      return new CachedData(0, bb.array(), CachedData.MAX_SIZE);
257    }
258
259    @Override
260    public HFileBlock decode(CachedData d) {
261      try {
262        ByteBuff buf = new SingleByteBuff(ByteBuffer.wrap(d.getData()));
263        return (HFileBlock) HFileBlock.blockDeserializer.deserialize(buf, true,
264          MemoryType.EXCLUSIVE);
265      } catch (IOException e) {
266        LOG.warn("Error deserializing data from memcached",e);
267      }
268      return null;
269    }
270
271    @Override
272    public int getMaxSize() {
273      return MAX_SIZE;
274    }
275  }
276
277  @Override
278  public void returnBlock(BlockCacheKey cacheKey, Cacheable block) {
279    // Not doing reference counting. All blocks here are EXCLUSIVE
280  }
