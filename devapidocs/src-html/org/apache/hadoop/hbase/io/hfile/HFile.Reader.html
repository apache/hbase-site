072/**
073 * File format for hbase.
074 * A file of sorted key/value pairs. Both keys and values are byte arrays.
075 * <p>
076 * The memory footprint of a HFile includes the following (below is taken from the
077 * <a
078 * href=https://issues.apache.org/jira/browse/HADOOP-3315>TFile</a> documentation
079 * but applies also to HFile):
080 * <ul>
081 * <li>Some constant overhead of reading or writing a compressed block.
082 * <ul>
083 * <li>Each compressed block requires one compression/decompression codec for
084 * I/O.
085 * <li>Temporary space to buffer the key.
086 * <li>Temporary space to buffer the value.
087 * </ul>
088 * <li>HFile index, which is proportional to the total number of Data Blocks.
089 * The total amount of memory needed to hold the index can be estimated as
090 * (56+AvgKeySize)*NumBlocks.
091 * </ul>
092 * Suggestions on performance optimization.
093 * <ul>
094 * <li>Minimum block size. We recommend a setting of minimum block size between
095 * 8KB to 1MB for general usage. Larger block size is preferred if files are
096 * primarily for sequential access. However, it would lead to inefficient random
097 * access (because there are more data to decompress). Smaller blocks are good
098 * for random access, but require more memory to hold the block index, and may
099 * be slower to create (because we must flush the compressor stream at the
100 * conclusion of each data block, which leads to an FS I/O flush). Further, due
101 * to the internal caching in Compression codec, the smallest possible block
102 * size would be around 20KB-30KB.
103 * <li>The current implementation does not offer true multi-threading for
104 * reading. The implementation uses FSDataInputStream seek()+read(), which is
105 * shown to be much faster than positioned-read call in single thread mode.
106 * However, it also means that if multiple threads attempt to access the same
107 * HFile (using multiple scanners) simultaneously, the actual I/O is carried out
108 * sequentially even if they access different DFS blocks (Reexamine! pread seems
109 * to be 10% faster than seek+read in my testing -- stack).
110 * <li>Compression codec. Use "none" if the data is not very compressable (by
111 * compressable, I mean a compression ratio at least 2:1). Generally, use "lzo"
112 * as the starting point for experimenting. "gz" overs slightly better
113 * compression ratio over "lzo" but requires 4x CPU to compress and 2x CPU to
114 * decompress, comparing to "lzo".
115 * </ul>
116 *
117 * For more on the background behind HFile, see <a
118 * href=https://issues.apache.org/jira/browse/HBASE-61>HBASE-61</a>.
119 * <p>
120 * File is made of data blocks followed by meta data blocks (if any), a fileinfo
121 * block, data block index, meta data block index, and a fixed size trailer
122 * which records the offsets at which file changes content type.
123 * <pre>&lt;data blocks&gt;&lt;meta blocks&gt;&lt;fileinfo&gt;&lt;
124 * data index&gt;&lt;meta index&gt;&lt;trailer&gt;</pre>
125 * Each block has a bit of magic at its start.  Block are comprised of
126 * key/values.  In data blocks, they are both byte arrays.  Metadata blocks are
127 * a String key and a byte array value.  An empty file looks like this:
128 * <pre>&lt;fileinfo&gt;&lt;trailer&gt;</pre>.  That is, there are not data nor meta
129 * blocks present.
130 * <p>
131 * TODO: Do scanners need to be able to take a start and end row?
132 * TODO: Should BlockIndex know the name of its file?  Should it have a Path
133 * that points at its file say for the case where an index lives apart from
134 * an HFile instance?
135 */

252        CellComparator.COMPARATOR;

254    private HFileContext fileContext;

266      this.path = path;
267      return this;

271      Preconditions.checkNotNull(ostream);
272      this.ostream = ostream;
273      return this;
274    }

276    public WriterFactory withComparator(CellComparator comparator) {
277      Preconditions.checkNotNull(comparator);
278      this.comparator = comparator;
279      return this;
280    }
281
282    public WriterFactory withFavoredNodes(InetSocketAddress[] favoredNodes) {
283      // Deliberately not checking for null here.
284      this.favoredNodes = favoredNodes;
285      return this;
286    }
287
288    public WriterFactory withFileContext(HFileContext fileContext) {
289      this.fileContext = fileContext;
290      return this;
291    }
292

294      this.shouldDropBehind = shouldDropBehind;
295      return this;
296    }

298
299    public Writer create() throws IOException {
300      if ((path != null ? 1 : 0) + (ostream != null ? 1 : 0) != 1) {
301        throw new AssertionError("Please specify exactly one of " +
302            "filesystem/path or path");
303      }
304      if (path != null) {
305        ostream = HFileWriterImpl.createOutputStream(conf, fs, path, favoredNodes);
306        try {
307          ostream.setDropBehind(shouldDropBehind && cacheConf.shouldDropBehindCompaction());
308        } catch (UnsupportedOperationException uoe) {
309          if (LOG.isTraceEnabled()) LOG.trace("Unable to set drop behind on " + path, uoe);
310          else if (LOG.isDebugEnabled()) LOG.debug("Unable to set drop behind on " + path);
311        }
312      }
313      return new HFileWriterImpl(conf, cacheConf, path, ostream, comparator, fileContext);
314    }

480  /**
481   * Method returns the reader given the specified arguments.
482   * TODO This is a bad abstraction.  See HBASE-6635.
483   *
484   * @param path hfile's path
485   * @param fsdis stream of path's file
486   * @param size max size of the trailer.
487   * @param cacheConf Cache configuation values, cannot be null.
488   * @param hfs
489   * @return an appropriate instance of HFileReader
490   * @throws IOException If file is invalid, will throw CorruptHFileException flavored IOException
491   */
492  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="SF_SWITCH_FALLTHROUGH",
493      justification="Intentional")
494  private static Reader pickReaderVersion(Path path, FSDataInputStreamWrapper fsdis,
495      long size, CacheConfig cacheConf, HFileSystem hfs, Configuration conf) throws IOException {
496    FixedFileTrailer trailer = null;
497    try {
498      boolean isHBaseChecksum = fsdis.shouldUseHBaseChecksum();
499      assert !isHBaseChecksum; // Initially we must read with FS checksum.
500      trailer = FixedFileTrailer.readFromStream(fsdis.getStream(isHBaseChecksum), size);
501      switch (trailer.getMajorVersion()) {
502      case 2:
503        LOG.debug("Opening HFile v2 with v3 reader");
504        // Fall through. FindBugs: SF_SWITCH_FALLTHROUGH
505      case 3 :
506        return new HFileReaderImpl(path, trailer, fsdis, size, cacheConf, hfs, conf);
507      default:
508        throw new IllegalArgumentException("Invalid HFile version " + trailer.getMajorVersion());
509      }
510    } catch (Throwable t) {
511      try {
512        fsdis.close();
513      } catch (Throwable t2) {
514        LOG.warn("Error closing fsdis FSDataInputStreamWrapper", t2);
515      }
516      throw new CorruptHFileException("Problem reading HFile Trailer from file " + path, t);
517    }
518  }

521   * @param fs A file system
522   * @param path Path to HFile
523   * @param fsdis a stream of path's file
524   * @param size max size of the trailer.
525   * @param cacheConf Cache configuration for hfile's contents
526   * @param conf Configuration
527   * @return A version specific Hfile Reader
528   * @throws IOException If file is invalid, will throw CorruptHFileException flavored IOException
529   */
530  @SuppressWarnings("resource")
531  public static Reader createReader(FileSystem fs, Path path,
532      FSDataInputStreamWrapper fsdis, long size, CacheConfig cacheConf, Configuration conf)
533      throws IOException {
534    HFileSystem hfs = null;
535
536    // If the fs is not an instance of HFileSystem, then create an
537    // instance of HFileSystem that wraps over the specified fs.
538    // In this case, we will not be able to avoid checksumming inside
539    // the filesystem.
540    if (!(fs instanceof HFileSystem)) {
541      hfs = new HFileSystem(fs);
542    } else {
543      hfs = (HFileSystem)fs;
544    }
545    return pickReaderVersion(path, fsdis, size, cacheConf, hfs, conf);
546  }

599   * Returns true if the specified file has a valid HFile Trailer.
600   * @param fs filesystem
601   * @param fileStatus the file to verify
602   * @return true if the file has a valid HFile Trailer, otherwise false
603   * @throws IOException if failed to read from the underlying stream
604   */
605  public static boolean isHFileFormat(final FileSystem fs, final FileStatus fileStatus)
606      throws IOException {
607    final Path path = fileStatus.getPath();
608    final long size = fileStatus.getLen();
609    FSDataInputStreamWrapper fsdis = new FSDataInputStreamWrapper(fs, path);
610    try {
611      boolean isHBaseChecksum = fsdis.shouldUseHBaseChecksum();
612      assert !isHBaseChecksum; // Initially we must read with FS checksum.
613      FixedFileTrailer.readFromStream(fsdis.getStream(isHBaseChecksum), size);
614      return true;
615    } catch (IllegalArgumentException e) {
616      return false;
617    } catch (IOException e) {
618      throw e;
619    } finally {
620      try {
621        fsdis.close();
622      } catch (Throwable t) {
623        LOG.warn("Error closing fsdis FSDataInputStreamWrapper: " + path, t);
624      }
625    }

648     * Append the given key/value pair to the file info, optionally checking the
649     * key prefix.
650     *
651     * @param k key to add
652     * @param v value to add
653     * @param checkPrefix whether to check that the provided key does not start
654     *          with the reserved prefix
655     * @return this file info object
656     * @throws IOException if the key or value is invalid
657     */
658    public FileInfo append(final byte[] k, final byte[] v,
659        final boolean checkPrefix) throws IOException {
660      if (k == null || v == null) {
661        throw new NullPointerException("Key nor value may be null");
662      }
663      if (checkPrefix && isReservedFileInfoKey(k)) {
664        throw new IOException("Keys with a " + FileInfo.RESERVED_PREFIX
665            + " are reserved");
666      }
667      put(k, v);
668      return this;
669    }

771    /**
772     * Write out this instance on the passed in <code>out</code> stream.
773     * We write it as a protobuf.
774     * @param out
775     * @throws IOException
776     * @see #read(DataInputStream)
777     */
778    void write(final DataOutputStream out) throws IOException {
779      HFileProtos.FileInfoProto.Builder builder = HFileProtos.FileInfoProto.newBuilder();
780      for (Map.Entry<byte [], byte[]> e: this.map.entrySet()) {
781        HBaseProtos.BytesBytesPair.Builder bbpBuilder = HBaseProtos.BytesBytesPair.newBuilder();
782        bbpBuilder.setFirst(UnsafeByteOperations.unsafeWrap(e.getKey()));
783        bbpBuilder.setSecond(UnsafeByteOperations.unsafeWrap(e.getValue()));
784        builder.addMapEntry(bbpBuilder.build());
785      }
786      out.write(ProtobufMagic.PB_MAGIC);
787      builder.build().writeDelimitedTo(out);
788    }

790    /**
791     * Populate this instance with what we find on the passed in <code>in</code> stream.
792     * Can deserialize protobuf of old Writables format.
793     * @param in
794     * @throws IOException
795     * @see #write(DataOutputStream)
796     */
797    void read(final DataInputStream in) throws IOException {
798      // This code is tested over in TestHFileReaderV1 where we read an old hfile w/ this new code.
799      int pblen = ProtobufUtil.lengthOfPBMagic();
800      byte [] pbuf = new byte[pblen];
801      if (in.markSupported()) in.mark(pblen);
802      int read = in.read(pbuf);
803      if (read != pblen) throw new IOException("read=" + read + ", wanted=" + pblen);
804      if (ProtobufUtil.isPBMagicPrefix(pbuf)) {
805        parsePB(HFileProtos.FileInfoProto.parseDelimitedFrom(in));
806      } else {
807        if (in.markSupported()) {
808          in.reset();
809          parseWritable(in);
810        } else {
811          // We cannot use BufferedInputStream, it consumes more than we read from the underlying IS
812          ByteArrayInputStream bais = new ByteArrayInputStream(pbuf);
813          SequenceInputStream sis = new SequenceInputStream(bais, in); // Concatenate input streams
814          // TODO: Am I leaking anything here wrapping the passed in stream?  We are not calling close on the wrapped
815          // streams but they should be let go after we leave this context?  I see that we keep a reference to the
816          // passed in inputstream but since we no longer have a reference to this after we leave, we should be ok.
817          parseWritable(new DataInputStream(sis));
818        }
819      }
820    }
821
822    /** Now parse the old Writable format.  It was a list of Map entries.  Each map entry was a key and a value of
823     * a byte [].  The old map format had a byte before each entry that held a code which was short for the key or
824     * value type.  We know it was a byte [] so in below we just read and dump it.
825     * @throws IOException
826     */
827    void parseWritable(final DataInputStream in) throws IOException {
828      // First clear the map.  Otherwise we will just accumulate entries every time this method is called.
829      this.map.clear();
830      // Read the number of entries in the map
831      int entries = in.readInt();
832      // Then read each key/value pair
833      for (int i = 0; i < entries; i++) {
834        byte [] key = Bytes.readByteArray(in);
835        // We used to read a byte that encoded the class type.  Read and ignore it because it is always byte [] in hfile
836        in.readByte();
837        byte [] value = Bytes.readByteArray(in);
838        this.map.put(key, value);
839      }
840    }

842    /**
843     * Fill our map with content of the pb we read off disk
844     * @param fip protobuf message to read
845     */
846    void parsePB(final HFileProtos.FileInfoProto fip) {
847      this.map.clear();
848      for (BytesBytesPair pair: fip.getMapEntryList()) {
849        this.map.put(pair.getFirst().toByteArray(), pair.getSecond().toByteArray());
850      }
851    }

881    // Expecting the size() of a block not exceeding 4GB. Assuming the

886  /**
887   * Returns all HFiles belonging to the given region directory. Could return an
888   * empty list.
889   *
890   * @param fs  The file system reference.
891   * @param regionDir  The region directory to scan.
892   * @return The list of files found.
893   * @throws IOException When scanning the files fails.
894   */
895  static List<Path> getStoreFiles(FileSystem fs, Path regionDir)
896      throws IOException {
897    List<Path> regionHFiles = new ArrayList<Path>();
898    PathFilter dirFilter = new FSUtils.DirFilter(fs);
899    FileStatus[] familyDirs = fs.listStatus(regionDir, dirFilter);
900    for(FileStatus dir : familyDirs) {
901      FileStatus[] files = fs.listStatus(dir.getPath());
902      for (FileStatus file : files) {
903        if (!file.isDirectory() &&
904            (!file.getPath().toString().contains(HConstants.HREGION_OLDLOGDIR_NAME)) &&
905            (!file.getPath().toString().contains(HConstants.RECOVERED_EDITS_DIR))) {
906          regionHFiles.add(file.getPath());
907        }
908      }
909    }
910    return regionHFiles;
911  }

915   * invalid. Note that if the version number comes from an input file and has
916   * not been verified, the caller needs to re-throw an {@link IOException} to
917   * indicate that this is not a software error, but corrupted input.

925      throw new IllegalArgumentException("Invalid HFile version: " + version
926          + " (expected to be " + "between " + MIN_FORMAT_VERSION + " and "
927          + MAX_FORMAT_VERSION + ")");
928    }
929  }
