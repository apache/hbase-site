085/**
086 * BucketCache uses {@link BucketAllocator} to allocate/free blocks, and uses
087 * BucketCache#ramCache and BucketCache#backingMap in order to
088 * determine if a given element is in the cache. The bucket cache can use on-heap or
089 * off-heap memory {@link ByteBufferIOEngine} or in a file {@link FileIOEngine} to
090 * store/read the block data.
091 *
092 * <p>Eviction is via a similar algorithm as used in
093 * {@link org.apache.hadoop.hbase.io.hfile.LruBlockCache}
094 *
095 * <p>BucketCache can be used as mainly a block cache (see
096 * {@link org.apache.hadoop.hbase.io.hfile.CombinedBlockCache}), combined with
097 * LruBlockCache to decrease CMS GC and heap fragmentation.
098 *
099 * <p>It also can be used as a secondary cache (e.g. using a file on ssd/fusionio to store
100 * blocks) to enlarge cache space via
101 * {@link org.apache.hadoop.hbase.io.hfile.LruBlockCache#setVictimCache}
102 */

256  public BucketCache(String ioEngineName, long capacity, int blockSize, int[] bucketSizes,
257                     int writerThreadNum, int writerQLen, String persistencePath, int ioErrorsTolerationDuration,
258                     Configuration conf)
259      throws FileNotFoundException, IOException {
260    this.ioEngine = getIOEngineFromName(ioEngineName, capacity, persistencePath);
261    this.writerThreads = new WriterThread[writerThreadNum];
262    long blockNumCapacity = capacity / blockSize;
263    if (blockNumCapacity >= Integer.MAX_VALUE) {
264      // Enough for about 32TB of cache!
265      throw new IllegalArgumentException("Cache capacity is too large, only support 32TB now");
266    }
267
268    this.acceptableFactor = conf.getFloat(ACCEPT_FACTOR_CONFIG_NAME, DEFAULT_ACCEPT_FACTOR);
269    this.minFactor = conf.getFloat(MIN_FACTOR_CONFIG_NAME, DEFAULT_MIN_FACTOR);
270    this.extraFreeFactor = conf.getFloat(EXTRA_FREE_FACTOR_CONFIG_NAME, DEFAULT_EXTRA_FREE_FACTOR);
271    this.singleFactor = conf.getFloat(SINGLE_FACTOR_CONFIG_NAME, DEFAULT_SINGLE_FACTOR);
272    this.multiFactor = conf.getFloat(MULTI_FACTOR_CONFIG_NAME, DEFAULT_MULTI_FACTOR);
273    this.memoryFactor = conf.getFloat(MEMORY_FACTOR_CONFIG_NAME, DEFAULT_MEMORY_FACTOR);
274
275    sanityCheckConfigs();
276
277    LOG.info("Instantiating BucketCache with acceptableFactor: " + acceptableFactor + ", minFactor: " + minFactor +
278        ", extraFreeFactor: " + extraFreeFactor + ", singleFactor: " + singleFactor + ", multiFactor: " + multiFactor +
279        ", memoryFactor: " + memoryFactor);
280
281    this.cacheCapacity = capacity;
282    this.persistencePath = persistencePath;
283    this.blockSize = blockSize;
284    this.ioErrorsTolerationDuration = ioErrorsTolerationDuration;
285
286    bucketAllocator = new BucketAllocator(capacity, bucketSizes);
287    for (int i = 0; i < writerThreads.length; ++i) {
288      writerQueues.add(new ArrayBlockingQueue<>(writerQLen));
289    }
290
291    assert writerQueues.size() == writerThreads.length;
292    this.ramCache = new ConcurrentHashMap<>();
293
294    this.backingMap = new ConcurrentHashMap<>((int) blockNumCapacity);
295
296    if (ioEngine.isPersistent() && persistencePath != null) {
297      try {
298        retrieveFromFile(bucketSizes);
299      } catch (IOException ioex) {
300        LOG.error("Can't restore from file[" + persistencePath + "] because of ", ioex);
301      }
302    }
303    final String threadName = Thread.currentThread().getName();
304    this.cacheEnabled = true;
305    for (int i = 0; i < writerThreads.length; ++i) {
306      writerThreads[i] = new WriterThread(writerQueues.get(i));
307      writerThreads[i].setName(threadName + "-BucketCacheWriter-" + i);
308      writerThreads[i].setDaemon(true);
309    }
310    startWriterThreads();
311
312    // Run the statistics thread periodically to print the cache statistics log
313    // TODO: Add means of turning this off.  Bit obnoxious running thread just to make a log
314    // every five minutes.
315    this.scheduleThreadPool.scheduleAtFixedRate(new StatisticsThread(this),
316        statThreadPeriod, statThreadPeriod, TimeUnit.SECONDS);
317    LOG.info("Started bucket cache; ioengine=" + ioEngineName +
318        ", capacity=" + StringUtils.byteDesc(capacity) +
319      ", blockSize=" + StringUtils.byteDesc(blockSize) + ", writerThreadNum=" +
320        writerThreadNum + ", writerQLen=" + writerQLen + ", persistencePath=" +
321      persistencePath + ", bucketAllocator=" + this.bucketAllocator.getClass().getName());
322  }

369  private IOEngine getIOEngineFromName(String ioEngineName, long capacity, String persistencePath)
370      throws IOException {
371    if (ioEngineName.startsWith("file:") || ioEngineName.startsWith("files:")) {
372      // In order to make the usage simple, we only need the prefix 'files:' in
373      // document whether one or multiple file(s), but also support 'file:' for
374      // the compatibility
375      String[] filePaths = ioEngineName.substring(ioEngineName.indexOf(":") + 1)
376          .split(FileIOEngine.FILE_DELIMITER);
377      return new FileIOEngine(capacity, persistencePath != null, filePaths);
378    } else if (ioEngineName.startsWith("offheap")) {
379      return new ByteBufferIOEngine(capacity);
380    } else if (ioEngineName.startsWith("mmap:")) {
381      return new FileMmapEngine(ioEngineName.substring(5), capacity);
382    } else {
383      throw new IllegalArgumentException(
384          "Don't understand io engine name for cache- prefix with file:, files:, mmap: or offheap");
385    }
386  }

409  /**
410   * Cache the block to ramCache
411   * @param cacheKey block's cache key
412   * @param cachedItem block buffer
413   * @param inMemory if block is in-memory
414   * @param wait if true, blocking wait when queue is full
415   */
416  private void cacheBlockWithWait(BlockCacheKey cacheKey, Cacheable cachedItem, boolean inMemory,
417      boolean wait) {
418    if (cacheEnabled) {
419      if (backingMap.containsKey(cacheKey) || ramCache.containsKey(cacheKey)) {
420        if (BlockCacheUtil.shouldReplaceExistingCacheBlock(this, cacheKey, cachedItem)) {
421          cacheBlockWithWaitInternal(cacheKey, cachedItem, inMemory, wait);
422        }
423      } else {
424        cacheBlockWithWaitInternal(cacheKey, cachedItem, inMemory, wait);
425      }
426    }
427  }
428
429  private void cacheBlockWithWaitInternal(BlockCacheKey cacheKey, Cacheable cachedItem,
430      boolean inMemory, boolean wait) {
431    if (!cacheEnabled) {
432      return;
433    }
434    LOG.trace("Caching key={}, item={}", cacheKey, cachedItem);
435    // Stuff the entry into the RAM cache so it can get drained to the persistent store
436    RAMQueueEntry re =
437        new RAMQueueEntry(cacheKey, cachedItem, accessCount.incrementAndGet(), inMemory);
438    /**
439     * Don't use ramCache.put(cacheKey, re) here. because there may be a existing entry with same
440     * key in ramCache, the heap size of bucket cache need to update if replacing entry from
441     * ramCache. But WriterThread will also remove entry from ramCache and update heap size, if
442     * using ramCache.put(), It's possible that the removed entry in WriterThread is not the correct
443     * one, then the heap size will mess up (HBASE-20789)
444     */
445    if (ramCache.putIfAbsent(cacheKey, re) != null) {
446      return;
447    }
448    int queueNum = (cacheKey.hashCode() & 0x7FFFFFFF) % writerQueues.size();
449    BlockingQueue<RAMQueueEntry> bq = writerQueues.get(queueNum);
450    boolean successfulAddition = false;
451    if (wait) {
452      try {
453        successfulAddition = bq.offer(re, DEFAULT_CACHE_WAIT_TIME, TimeUnit.MILLISECONDS);
454      } catch (InterruptedException e) {
455        Thread.currentThread().interrupt();
456      }
457    } else {
458      successfulAddition = bq.offer(re);
459    }
460    if (!successfulAddition) {
461      ramCache.remove(cacheKey);
462      cacheStats.failInsert();
463    } else {
464      this.blockNumber.increment();
465      this.heapSize.add(cachedItem.heapSize());
466      blocksByHFile.add(cacheKey);
467    }
468  }

470  /**
471   * Get the buffer of the block with the specified key.
472   * @param key block's cache key
473   * @param caching true if the caller caches blocks on cache misses
474   * @param repeat Whether this is a repeat lookup for the same block
475   * @param updateCacheMetrics Whether we should update cache metrics or not
476   * @return buffer of specified cache key, or null if not in cache
477   */
478  @Override
479  public Cacheable getBlock(BlockCacheKey key, boolean caching, boolean repeat,
480      boolean updateCacheMetrics) {
481    if (!cacheEnabled) {
482      return null;
483    }
484    RAMQueueEntry re = ramCache.get(key);
485    if (re != null) {
486      if (updateCacheMetrics) {
487        cacheStats.hit(caching, key.isPrimary(), key.getBlockType());
488      }
489      re.access(accessCount.incrementAndGet());
490      return re.getData();
491    }
492    BucketEntry bucketEntry = backingMap.get(key);
493    if (bucketEntry != null) {
494      long start = System.nanoTime();
495      ReentrantReadWriteLock lock = offsetLock.getLock(bucketEntry.offset());
496      try {
497        lock.readLock().lock();
498        // We can not read here even if backingMap does contain the given key because its offset
499        // maybe changed. If we lock BlockCacheKey instead of offset, then we can only check
500        // existence here.
501        if (bucketEntry.equals(backingMap.get(key))) {
502          // TODO : change this area - should be removed after server cells and
503          // 12295 are available
504          int len = bucketEntry.getLength();
505          if (LOG.isTraceEnabled()) {
506            LOG.trace("Read offset=" + bucketEntry.offset() + ", len=" + len);
507          }
508          Cacheable cachedBlock = ioEngine.read(bucketEntry.offset(), len,
509              bucketEntry.deserializerReference());
510          long timeTaken = System.nanoTime() - start;
511          if (updateCacheMetrics) {
512            cacheStats.hit(caching, key.isPrimary(), key.getBlockType());
513            cacheStats.ioHit(timeTaken);
514          }
515          if (cachedBlock.getMemoryType() == MemoryType.SHARED) {
516            bucketEntry.incrementRefCountAndGet();
517          }
518          bucketEntry.access(accessCount.incrementAndGet());
519          if (this.ioErrorStartTime > 0) {
520            ioErrorStartTime = -1;
521          }
522          return cachedBlock;
523        }
524      } catch (IOException ioex) {
525        LOG.error("Failed reading block " + key + " from bucket cache", ioex);
526        checkIOErrorIsTolerated();
527      } finally {
528        lock.readLock().unlock();
529      }
530    }
531    if (!repeat && updateCacheMetrics) {
532      cacheStats.miss(caching, key.isPrimary(), key.getBlockType());
533    }
534    return null;
535  }

552  // does not check for the ref count. Just tries to evict it if found in the
553  // bucket map
554  private boolean forceEvict(BlockCacheKey cacheKey) {
555    if (!cacheEnabled) {
556      return false;
557    }
558    RAMQueueEntry removedBlock = checkRamCache(cacheKey);
559    BucketEntry bucketEntry = backingMap.get(cacheKey);
560    if (bucketEntry == null) {
561      if (removedBlock != null) {
562        cacheStats.evicted(0, cacheKey.isPrimary());
563        return true;
564      } else {
565        return false;
566      }
567    }
568    ReentrantReadWriteLock lock = offsetLock.getLock(bucketEntry.offset());
569    try {
570      lock.writeLock().lock();
571      if (backingMap.remove(cacheKey, bucketEntry)) {
572        blockEvicted(cacheKey, bucketEntry, removedBlock == null);
573      } else {
574        return false;
575      }
576    } finally {
577      lock.writeLock().unlock();
578    }
579    cacheStats.evicted(bucketEntry.getCachedTime(), cacheKey.isPrimary());
580    return true;
581  }

583  private RAMQueueEntry checkRamCache(BlockCacheKey cacheKey) {
584    RAMQueueEntry removedBlock = ramCache.remove(cacheKey);
585    if (removedBlock != null) {
586      this.blockNumber.decrement();
587      this.heapSize.add(-1 * removedBlock.getData().heapSize());
588    }
589    return removedBlock;
590  }

594      return false;
595    }
596    RAMQueueEntry removedBlock = checkRamCache(cacheKey);
597    BucketEntry bucketEntry = backingMap.get(cacheKey);
598    if (bucketEntry == null) {
599      if (removedBlock != null) {
600        cacheStats.evicted(0, cacheKey.isPrimary());
601        return true;
602      } else {
603        return false;
604      }
605    }
606    ReentrantReadWriteLock lock = offsetLock.getLock(bucketEntry.offset());
607    try {
608      lock.writeLock().lock();
609      int refCount = bucketEntry.getRefCount();
610      if (refCount == 0) {
611        if (backingMap.remove(cacheKey, bucketEntry)) {
612          blockEvicted(cacheKey, bucketEntry, removedBlock == null);
613        } else {
614          return false;
615        }
616      } else {
617        if(!deletedBlock) {
618          if (LOG.isDebugEnabled()) {
619            LOG.debug("This block " + cacheKey + " is still referred by " + refCount
620                + " readers. Can not be freed now");
621          }
622          return false;
623        } else {
624          if (LOG.isDebugEnabled()) {
625            LOG.debug("This block " + cacheKey + " is still referred by " + refCount
626                + " readers. Can not be freed now. Hence will mark this"
627                + " for evicting at a later point");
628          }
629          bucketEntry.markForEvict();
630        }
631      }
632    } finally {
633      lock.writeLock().unlock();
634    }
635    cacheStats.evicted(bucketEntry.getCachedTime(), cacheKey.isPrimary());
636    return true;
637  }

698   */
699  private int bucketSizesAboveThresholdCount(float minFactor) {
700    BucketAllocator.IndexStatistics[] stats = bucketAllocator.getIndexStatistics();
701    int fullCount = 0;
702    for (int i = 0; i < stats.length; i++) {
703      long freeGoal = (long) Math.floor(stats[i].totalCount() * (1 - minFactor));

712  /**
713   * This method will find the buckets that are minimally occupied
714   * and are not reference counted and will free them completely
715   * without any constraint on the access times of the elements,
716   * and as a process will completely free at most the number of buckets
717   * passed, sometimes it might not due to changing refCounts
718   *
719   * @param completelyFreeBucketsNeeded number of buckets to free
720   **/
721  private void freeEntireBuckets(int completelyFreeBucketsNeeded) {
722    if (completelyFreeBucketsNeeded != 0) {
723      // First we will build a set where the offsets are reference counted, usually
724      // this set is small around O(Handler Count) unless something else is wrong
725      Set<Integer> inUseBuckets = new HashSet<Integer>();
726      for (BucketEntry entry : backingMap.values()) {
727        if (entry.getRefCount() != 0) {
728          inUseBuckets.add(bucketAllocator.getBucketIndex(entry.offset()));
729        }
730      }
731
732      Set<Integer> candidateBuckets = bucketAllocator.getLeastFilledBuckets(
733          inUseBuckets, completelyFreeBucketsNeeded);
734      for (Map.Entry<BlockCacheKey, BucketEntry> entry : backingMap.entrySet()) {
735        if (candidateBuckets.contains(bucketAllocator
736            .getBucketIndex(entry.getValue().offset()))) {
737          evictBlock(entry.getKey(), false);
738        }
739      }
740    }
741  }

743  /**
744   * Free the space if the used size reaches acceptableSize() or one size block
745   * couldn't be allocated. When freeing the space, we use the LRU algorithm and
746   * ensure there must be some blocks evicted
747   * @param why Why we are being called
748   */
749  private void freeSpace(final String why) {
750    // Ensure only one freeSpace progress at a time
751    if (!freeSpaceLock.tryLock()) {
752      return;
753    }
754    try {
755      freeInProgress = true;
756      long bytesToFreeWithoutExtra = 0;
757      // Calculate free byte for each bucketSizeinfo
758      StringBuilder msgBuffer = LOG.isDebugEnabled()? new StringBuilder(): null;
759      BucketAllocator.IndexStatistics[] stats = bucketAllocator.getIndexStatistics();
760      long[] bytesToFreeForBucket = new long[stats.length];
761      for (int i = 0; i < stats.length; i++) {
762        bytesToFreeForBucket[i] = 0;
763        long freeGoal = (long) Math.floor(stats[i].totalCount() * (1 - minFactor));
764        freeGoal = Math.max(freeGoal, 1);
765        if (stats[i].freeCount() < freeGoal) {
766          bytesToFreeForBucket[i] = stats[i].itemSize() * (freeGoal - stats[i].freeCount());
767          bytesToFreeWithoutExtra += bytesToFreeForBucket[i];
768          if (msgBuffer != null) {
769            msgBuffer.append("Free for bucketSize(" + stats[i].itemSize() + ")="
770              + StringUtils.byteDesc(bytesToFreeForBucket[i]) + ", ");
771          }
772        }
773      }
774      if (msgBuffer != null) {
775        msgBuffer.append("Free for total=" + StringUtils.byteDesc(bytesToFreeWithoutExtra) + ", ");
776      }
777
778      if (bytesToFreeWithoutExtra <= 0) {
779        return;
780      }
781      long currentSize = bucketAllocator.getUsedSize();
782      long totalSize = bucketAllocator.getTotalSize();
783      if (LOG.isDebugEnabled() && msgBuffer != null) {
784        LOG.debug("Free started because \"" + why + "\"; " + msgBuffer.toString() +
785          " of current used=" + StringUtils.byteDesc(currentSize) + ", actual cacheSize=" +
786          StringUtils.byteDesc(realCacheSize.sum()) + ", total=" + StringUtils.byteDesc(totalSize));
787      }
788
789      long bytesToFreeWithExtra = (long) Math.floor(bytesToFreeWithoutExtra
790          * (1 + extraFreeFactor));
791
792      // Instantiate priority buckets
793      BucketEntryGroup bucketSingle = new BucketEntryGroup(bytesToFreeWithExtra,
794          blockSize, getPartitionSize(singleFactor));
795      BucketEntryGroup bucketMulti = new BucketEntryGroup(bytesToFreeWithExtra,
796          blockSize, getPartitionSize(multiFactor));
797      BucketEntryGroup bucketMemory = new BucketEntryGroup(bytesToFreeWithExtra,
798          blockSize, getPartitionSize(memoryFactor));
799
800      // Scan entire map putting bucket entry into appropriate bucket entry
801      // group
802      for (Map.Entry<BlockCacheKey, BucketEntry> bucketEntryWithKey : backingMap.entrySet()) {
803        switch (bucketEntryWithKey.getValue().getPriority()) {
804          case SINGLE: {
805            bucketSingle.add(bucketEntryWithKey);
806            break;
807          }
808          case MULTI: {
809            bucketMulti.add(bucketEntryWithKey);
810            break;
811          }
812          case MEMORY: {
813            bucketMemory.add(bucketEntryWithKey);
814            break;
815          }
816        }
817      }
818
819      PriorityQueue<BucketEntryGroup> bucketQueue = new PriorityQueue<>(3,
820          Comparator.comparingLong(BucketEntryGroup::overflow));
821
822      bucketQueue.add(bucketSingle);
823      bucketQueue.add(bucketMulti);
824      bucketQueue.add(bucketMemory);
825
826      int remainingBuckets = 3;
827      long bytesFreed = 0;
828
829      BucketEntryGroup bucketGroup;
830      while ((bucketGroup = bucketQueue.poll()) != null) {
831        long overflow = bucketGroup.overflow();
832        if (overflow > 0) {
833          long bucketBytesToFree = Math.min(overflow,
834              (bytesToFreeWithoutExtra - bytesFreed) / remainingBuckets);
835          bytesFreed += bucketGroup.free(bucketBytesToFree);
836        }
837        remainingBuckets--;
838      }
839
840      // Check and free if there are buckets that still need freeing of space
841      if (bucketSizesAboveThresholdCount(minFactor) > 0) {
842        bucketQueue.clear();
843        remainingBuckets = 3;
844
845        bucketQueue.add(bucketSingle);
846        bucketQueue.add(bucketMulti);
847        bucketQueue.add(bucketMemory);
848
849        while ((bucketGroup = bucketQueue.poll()) != null) {
850          long bucketBytesToFree = (bytesToFreeWithExtra - bytesFreed) / remainingBuckets;
851          bytesFreed += bucketGroup.free(bucketBytesToFree);
852          remainingBuckets--;
853        }
854      }
855
856      // Even after the above free we might still need freeing because of the
857      // De-fragmentation of the buckets (also called Slab Calcification problem), i.e
858      // there might be some buckets where the occupancy is very sparse and thus are not
859      // yielding the free for the other bucket sizes, the fix for this to evict some
860      // of the buckets, we do this by evicting the buckets that are least fulled
861      freeEntireBuckets(DEFAULT_FREE_ENTIRE_BLOCK_FACTOR *
862          bucketSizesAboveThresholdCount(1.0f));
863
864      if (LOG.isDebugEnabled()) {
865        long single = bucketSingle.totalSize();
866        long multi = bucketMulti.totalSize();
867        long memory = bucketMemory.totalSize();
868        if (LOG.isDebugEnabled()) {
869          LOG.debug("Bucket cache free space completed; " + "freed="
870            + StringUtils.byteDesc(bytesFreed) + ", " + "total="
871            + StringUtils.byteDesc(totalSize) + ", " + "single="
872            + StringUtils.byteDesc(single) + ", " + "multi="
873            + StringUtils.byteDesc(multi) + ", " + "memory="
874            + StringUtils.byteDesc(memory));
875        }
876      }
877
878    } catch (Throwable t) {
879      LOG.warn("Failed freeing space", t);
880    } finally {
881      cacheStats.evict();
882      freeInProgress = false;
883      freeSpaceLock.unlock();
884    }
885  }

887  // This handles flushing the RAM cache to IOEngine.
888  @VisibleForTesting
889  class WriterThread extends HasThread {
890    private final BlockingQueue<RAMQueueEntry> inputQueue;
891    private volatile boolean writerEnabled = true;
892
893    WriterThread(BlockingQueue<RAMQueueEntry> queue) {
894      super("BucketCacheWriterThread");
895      this.inputQueue = queue;
896    }
897
898    // Used for test
899    @VisibleForTesting
900    void disableWriter() {
901      this.writerEnabled = false;
902    }
903
904    @Override
905    public void run() {
906      List<RAMQueueEntry> entries = new ArrayList<>();
907      try {
908        while (cacheEnabled && writerEnabled) {
909          try {
910            try {
911              // Blocks
912              entries = getRAMQueueEntries(inputQueue, entries);
913            } catch (InterruptedException ie) {
914              if (!cacheEnabled) break;
915            }
916            doDrain(entries);
917          } catch (Exception ioe) {
918            LOG.error("WriterThread encountered error", ioe);
919          }
920        }
921      } catch (Throwable t) {
922        LOG.warn("Failed doing drain", t);
923      }
924      LOG.info(this.getName() + " exiting, cacheEnabled=" + cacheEnabled);
925    }
926
927    /**
928     * Put the new bucket entry into backingMap. Notice that we are allowed to replace the existing
929     * cache with a new block for the same cache key. there's a corner case: one thread cache a
930     * block in ramCache, copy to io-engine and add a bucket entry to backingMap. Caching another
931     * new block with the same cache key do the same thing for the same cache key, so if not evict
932     * the previous bucket entry, then memory leak happen because the previous bucketEntry is gone
933     * but the bucketAllocator do not free its memory.
934     * @see BlockCacheUtil#shouldReplaceExistingCacheBlock(BlockCache blockCache,BlockCacheKey
935     *      cacheKey, Cacheable newBlock)
936     * @param key Block cache key
937     * @param bucketEntry Bucket entry to put into backingMap.
938     */
939    private void putIntoBackingMap(BlockCacheKey key, BucketEntry bucketEntry) {
940      BucketEntry previousEntry = backingMap.put(key, bucketEntry);
941      if (previousEntry != null && previousEntry != bucketEntry) {
942        ReentrantReadWriteLock lock = offsetLock.getLock(previousEntry.offset());
943        lock.writeLock().lock();
944        try {
945          blockEvicted(key, previousEntry, false);
946        } finally {
947          lock.writeLock().unlock();
948        }
949      }
950    }
951
952    /**
953     * Flush the entries in ramCache to IOEngine and add bucket entry to backingMap.
954     * Process all that are passed in even if failure being sure to remove from ramCache else we'll
955     * never undo the references and we'll OOME.
956     * @param entries Presumes list passed in here will be processed by this invocation only. No
957     *   interference expected.
958     * @throws InterruptedException
959     */
960    @VisibleForTesting
961    void doDrain(final List<RAMQueueEntry> entries) throws InterruptedException {
962      if (entries.isEmpty()) {
963        return;
964      }
965      // This method is a little hard to follow. We run through the passed in entries and for each
966      // successful add, we add a non-null BucketEntry to the below bucketEntries.  Later we must
967      // do cleanup making sure we've cleared ramCache of all entries regardless of whether we
968      // successfully added the item to the bucketcache; if we don't do the cleanup, we'll OOME by
969      // filling ramCache.  We do the clean up by again running through the passed in entries
970      // doing extra work when we find a non-null bucketEntries corresponding entry.
971      final int size = entries.size();
972      BucketEntry[] bucketEntries = new BucketEntry[size];
973      // Index updated inside loop if success or if we can't succeed. We retry if cache is full
974      // when we go to add an entry by going around the loop again without upping the index.
975      int index = 0;
976      while (cacheEnabled && index < size) {
977        RAMQueueEntry re = null;
978        try {
979          re = entries.get(index);
980          if (re == null) {
981            LOG.warn("Couldn't get entry or changed on us; who else is messing with it?");
982            index++;
983            continue;
984          }
985          BucketEntry bucketEntry =
986            re.writeToCache(ioEngine, bucketAllocator, realCacheSize);
987          // Successfully added.  Up index and add bucketEntry. Clear io exceptions.
988          bucketEntries[index] = bucketEntry;
989          if (ioErrorStartTime > 0) {
990            ioErrorStartTime = -1;
991          }
992          index++;
993        } catch (BucketAllocatorException fle) {
994          LOG.warn("Failed allocation for " + (re == null ? "" : re.getKey()) + "; " + fle);
995          // Presume can't add. Too big? Move index on. Entry will be cleared from ramCache below.
996          bucketEntries[index] = null;
997          index++;
998        } catch (CacheFullException cfe) {
999          // Cache full when we tried to add. Try freeing space and then retrying (don't up index)
1000          if (!freeInProgress) {
1001            freeSpace("Full!");
1002          } else {
1003            Thread.sleep(50);
1004          }
1005        } catch (IOException ioex) {
1006          // Hopefully transient. Retry. checkIOErrorIsTolerated disables cache if problem.
1007          LOG.error("Failed writing to bucket cache", ioex);
1008          checkIOErrorIsTolerated();
1009        }
1010      }
1011
1012      // Make sure data pages are written on media before we update maps.
1013      try {
1014        ioEngine.sync();
1015      } catch (IOException ioex) {
1016        LOG.error("Failed syncing IO engine", ioex);
1017        checkIOErrorIsTolerated();
1018        // Since we failed sync, free the blocks in bucket allocator
1019        for (int i = 0; i < entries.size(); ++i) {
1020          if (bucketEntries[i] != null) {
1021            bucketAllocator.freeBlock(bucketEntries[i].offset());
1022            bucketEntries[i] = null;
1023          }
1024        }
1025      }
1026
1027      // Now add to backingMap if successfully added to bucket cache.  Remove from ramCache if
1028      // success or error.
1029      for (int i = 0; i < size; ++i) {
1030        BlockCacheKey key = entries.get(i).getKey();
1031        // Only add if non-null entry.
1032        if (bucketEntries[i] != null) {
1033          putIntoBackingMap(key, bucketEntries[i]);
1034        }
1035        // Always remove from ramCache even if we failed adding it to the block cache above.
1036        RAMQueueEntry ramCacheEntry = ramCache.remove(key);
1037        if (ramCacheEntry != null) {
1038          heapSize.add(-1 * entries.get(i).getData().heapSize());
1039        } else if (bucketEntries[i] != null){
1040          // Block should have already been evicted. Remove it and free space.
1041          ReentrantReadWriteLock lock = offsetLock.getLock(bucketEntries[i].offset());
1042          try {
1043            lock.writeLock().lock();
1044            if (backingMap.remove(key, bucketEntries[i])) {
1045              blockEvicted(key, bucketEntries[i], false);
1046            }
1047          } finally {
1048            lock.writeLock().unlock();
1049          }
1050        }
1051      }
1052
1053      long used = bucketAllocator.getUsedSize();
1054      if (used > acceptableSize()) {
1055        freeSpace("Used=" + used + " > acceptable=" + acceptableSize());
1056      }
1057      return;
1058    }
1059  }

1296  /**
1297   * Item in cache. We expect this to be where most memory goes. Java uses 8
1298   * bytes just for object headers; after this, we want to use as little as
1299   * possible - so we only use 8 bytes, but in order to do so we end up messing
1300   * around with all this Java casting stuff. Offset stored as 5 bytes that make
1301   * up the long. Doubt we'll see devices this big for ages. Offsets are divided
1302   * by 256. So 5 bytes gives us 256TB or so.
1303   */
1304  static class BucketEntry implements Serializable {
1305    private static final long serialVersionUID = -6741504807982257534L;
1306
1307    // access counter comparator, descending order
1308    static final Comparator<BucketEntry> COMPARATOR = Comparator
1309        .comparingLong(BucketEntry::getAccessCounter).reversed();
1310
1311    private int offsetBase;
1312    private int length;
1313    private byte offset1;
1314
1315    /**
1316     * The index of the deserializer that can deserialize this BucketEntry content.
1317     * See {@link CacheableDeserializerIdManager} for hosting of index to serializers.
1318     */
1319    byte deserialiserIndex;
1320
1321    private volatile long accessCounter;
1322    private BlockPriority priority;
1323
1324    /**
1325     * Time this block was cached.  Presumes we are created just before we are added to the cache.
1326     */
1327    private final long cachedTime = System.nanoTime();
1328
1329    BucketEntry(long offset, int length, long accessCounter, boolean inMemory) {
1330      setOffset(offset);
1331      this.length = length;
1332      this.accessCounter = accessCounter;
1333      if (inMemory) {
1334        this.priority = BlockPriority.MEMORY;
1335      } else {
1336        this.priority = BlockPriority.SINGLE;
1337      }
1338    }
1339
1340    long offset() { // Java has no unsigned numbers
1341      long o = ((long) offsetBase) & 0xFFFFFFFFL; //This needs the L cast otherwise it will be sign extended as a negative number.
1342      o += (((long) (offset1)) & 0xFF) << 32; //The 0xFF here does not need the L cast because it is treated as a positive int.
1343      return o << 8;
1344    }
1345
1346    private void setOffset(long value) {
1347      assert (value & 0xFF) == 0;
1348      value >>= 8;
1349      offsetBase = (int) value;
1350      offset1 = (byte) (value >> 32);
1351    }
1352
1353    public int getLength() {
1354      return length;
1355    }
1356

1358      return CacheableDeserializerIdManager.getDeserializer(deserialiserIndex);
1359    }
1360
1361    protected void setDeserialiserReference(CacheableDeserializer<Cacheable> deserializer) {
1362      this.deserialiserIndex = (byte) deserializer.getDeserialiserIdentifier();

1445  /**
1446   * Used to group bucket entries into priority buckets. There will be a
1447   * BucketEntryGroup for each priority (single, multi, memory). Once bucketed,
1448   * the eviction algorithm takes the appropriate number of elements out of each
1449   * according to configuration parameters and their relative sizes.
1450   */
1451  private class BucketEntryGroup {
1452
1453    private CachedEntryQueue queue;
1454    private long totalSize = 0;
1455    private long bucketSize;
1456
1457    public BucketEntryGroup(long bytesToFree, long blockSize, long bucketSize) {
1458      this.bucketSize = bucketSize;
1459      queue = new CachedEntryQueue(bytesToFree, blockSize);
1460      totalSize = 0;
1461    }
1462
1463    public void add(Map.Entry<BlockCacheKey, BucketEntry> block) {
1464      totalSize += block.getValue().getLength();
1465      queue.add(block);
1466    }
1467
1468    public long free(long toFree) {
1469      Map.Entry<BlockCacheKey, BucketEntry> entry;
1470      long freedBytes = 0;
1471      // TODO avoid a cycling siutation. We find no block which is not in use and so no way to free
1472      // What to do then? Caching attempt fail? Need some changes in cacheBlock API?
1473      while ((entry = queue.pollLast()) != null) {
1474        if (evictBlock(entry.getKey(), false)) {
1475          freedBytes += entry.getValue().getLength();
1476        }
1477        if (freedBytes >= toFree) {
1478          return freedBytes;
1479        }
1480      }
1481      return freedBytes;
1482    }
1483
1484    public long overflow() {
1485      return totalSize - bucketSize;
1486    }
1487
1488    public long totalSize() {
1489      return totalSize;
1490    }
1491  }

1523    public BucketEntry writeToCache(final IOEngine ioEngine,
1524        final BucketAllocator bucketAllocator,
1525        final LongAdder realCacheSize) throws CacheFullException, IOException,
1526        BucketAllocatorException {
1527      int len = data.getSerializedLength();
1528      // This cacheable thing can't be serialized
1529      if (len == 0) return null;
1530      long offset = bucketAllocator.allocateBlock(len);
1531      BucketEntry bucketEntry = ioEngine.usesSharedMemory()
1532          ? UnsafeAvailChecker.isAvailable()
1533              ? new UnsafeSharedMemoryBucketEntry(offset, len, accessCounter, inMemory)
1534              : new SharedMemoryBucketEntry(offset, len, accessCounter, inMemory)
1535          : new BucketEntry(offset, len, accessCounter, inMemory);
1536      bucketEntry.setDeserialiserReference(data.getDeserializer());
1537      try {
1538        if (data instanceof HFileBlock) {
1539          // If an instance of HFileBlock, save on some allocations.
1540          HFileBlock block = (HFileBlock)data;
1541          ByteBuff sliceBuf = block.getBufferReadOnly();
1542          ByteBuffer metadata = block.getMetaData();
1543          if (LOG.isTraceEnabled()) {
1544            LOG.trace("Write offset=" + offset + ", len=" + len);
1545          }
1546          ioEngine.write(sliceBuf, offset);
1547          ioEngine.write(metadata, offset + len - metadata.limit());
1548        } else {
1549          ByteBuffer bb = ByteBuffer.allocate(len);
1550          data.serialize(bb, true);
1551          ioEngine.write(bb, offset);
1552        }
1553      } catch (IOException ioe) {
1554        // free it in bucket allocator
1555        bucketAllocator.freeBlock(offset);
1556        throw ioe;
1557      }
1558
1559      realCacheSize.add(len);
1560      return bucketEntry;
1561    }
