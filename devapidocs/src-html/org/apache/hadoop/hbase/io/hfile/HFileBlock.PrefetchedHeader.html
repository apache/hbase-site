059/**
060 * Reads {@link HFile} version 2 blocks to HFiles and via {@link Cacheable} Interface to caches.
061 * Version 2 was introduced in hbase-0.92.0. No longer has support for version 1 blocks since
062 * hbase-1.3.0.
063 *
064 * <p>Version 1 was the original file block. Version 2 was introduced when we changed the hbase file
065 * format to support multi-level block indexes and compound bloom filters (HBASE-3857).
066 *
067 * <h3>HFileBlock: Version 2</h3>
068 * In version 2, a block is structured as follows:
069 * <ul>
070 * <li><b>Header:</b> See Writer#putHeader() for where header is written; header total size is
071 * HFILEBLOCK_HEADER_SIZE
072 * <ul>
073 * <li>0. blockType: Magic record identifying the {@link BlockType} (8 bytes):
074 * e.g. <code>DATABLK*</code>
075 * <li>1. onDiskSizeWithoutHeader: Compressed -- a.k.a 'on disk' -- block size, excluding header,
076 * but including tailing checksum bytes (4 bytes)
077 * <li>2. uncompressedSizeWithoutHeader: Uncompressed block size, excluding header, and excluding
078 * checksum bytes (4 bytes)
079 * <li>3. prevBlockOffset: The offset of the previous block of the same type (8 bytes). This is
080 * used to navigate to the previous block without having to go to the block index
081 * <li>4: For minorVersions &gt;=1, the ordinal describing checksum type (1 byte)
082 * <li>5: For minorVersions &gt;=1, the number of data bytes/checksum chunk (4 bytes)
083 * <li>6: onDiskDataSizeWithHeader: For minorVersions &gt;=1, the size of data 'on disk', including
084 * header, excluding checksums (4 bytes)
085 * </ul>
086 * </li>
087 * <li><b>Raw/Compressed/Encrypted/Encoded data:</b> The compression
088 * algorithm is the same for all the blocks in an {@link HFile}. If compression is NONE, this is
089 * just raw, serialized Cells.
090 * <li><b>Tail:</b> For minorVersions &gt;=1, a series of 4 byte checksums, one each for
091 * the number of bytes specified by bytesPerChecksum.
092 * </ul>
093 *
094 * <h3>Caching</h3>
095 * Caches cache whole blocks with trailing checksums if any. We then tag on some metadata, the
096 * content of BLOCK_METADATA_SPACE which will be flag on if we are doing 'hbase'
097 * checksums and then the offset into the file which is needed when we re-make a cache key
098 * when we return the block to the cache as 'done'. See {@link Cacheable#serialize(ByteBuffer)} and
099 * {@link Cacheable#getDeserializer()}.
100 *
101 * <p>TODO: Should we cache the checksums? Down in Writer#getBlockForCaching(CacheConfig) where
102 * we make a block to cache-on-write, there is an attempt at turning off checksums. This is not the
103 * only place we get blocks to cache. We also will cache the raw return from an hdfs read. In this
104 * case, the checksums may be present. If the cache is backed by something that doesn't do ECC,
105 * say an SSD, we might want to preserve checksums. For now this is open question.
106 * <p>TODO: Over in BucketCache, we save a block allocation by doing a custom serialization.
107 * Be sure to change it if serialization changes in here. Could we add a method here that takes an
108 * IOEngine and that then serializes to it rather than expose our internals over in BucketCache?
109 * IOEngine is in the bucket subpackage. Pull it up? Then this class knows about bucketcache. Ugh.
110 */

222   * Used deserializing blocks from Cache.
223   *
224   * <code>
225   * ++++++++++++++
226   * + HFileBlock +
227   * ++++++++++++++
228   * + Checksums  + <= Optional
229   * ++++++++++++++
230   * + Metadata!  +
231   * ++++++++++++++
232   * </code>
233   * @see #serialize(ByteBuffer)

235  static final CacheableDeserializer<Cacheable> BLOCK_DESERIALIZER =
236      new CacheableDeserializer<Cacheable>() {
237        public HFileBlock deserialize(ByteBuff buf, boolean reuse, MemoryType memType)
238        throws IOException {
239          // The buf has the file block followed by block metadata.
240          // Set limit to just before the BLOCK_METADATA_SPACE then rewind.
241          buf.limit(buf.limit() - BLOCK_METADATA_SPACE).rewind();
242          // Get a new buffer to pass the HFileBlock for it to 'own'.
243          ByteBuff newByteBuff;
244          if (reuse) {
245            newByteBuff = buf.slice();
246          } else {
247            int len = buf.limit();
248            newByteBuff = new SingleByteBuff(ByteBuffer.allocate(len));
249            newByteBuff.put(0, buf, buf.position(), len);
250          }
251          // Read out the BLOCK_METADATA_SPACE content and shove into our HFileBlock.
252          buf.position(buf.limit());
253          buf.limit(buf.limit() + HFileBlock.BLOCK_METADATA_SPACE);
254          boolean usesChecksum = buf.get() == (byte)1;
255          long offset = buf.getLong();
256          int nextBlockOnDiskSize = buf.getInt();
257          HFileBlock hFileBlock =
258              new HFileBlock(newByteBuff, usesChecksum, memType, offset, nextBlockOnDiskSize, null);
259          return hFileBlock;
260        }
261
262        @Override
263        public int getDeserialiserIdentifier() {
264          return DESERIALIZER_IDENTIFIER;
265        }
266
267        @Override
268        public HFileBlock deserialize(ByteBuff b) throws IOException {
269          // Used only in tests
270          return deserialize(b, false, MemoryType.EXCLUSIVE);
271        }
272      };

359  /**
360   * Creates a block from an existing buffer starting with a header. Rewinds
361   * and takes ownership of the buffer. By definition of rewind, ignores the
362   * buffer position, but if you slice the buffer beforehand, it will rewind
363   * to that point.
364   * @param buf Has header, content, and trailing checksums if present.
365   */
366  HFileBlock(ByteBuff buf, boolean usesHBaseChecksum, MemoryType memType, final long offset,
367      final int nextBlockOnDiskSize, HFileContext fileContext) throws IOException {
368    buf.rewind();
369    final BlockType blockType = BlockType.read(buf);
370    final int onDiskSizeWithoutHeader = buf.getInt(Header.ON_DISK_SIZE_WITHOUT_HEADER_INDEX);
371    final int uncompressedSizeWithoutHeader =
372        buf.getInt(Header.UNCOMPRESSED_SIZE_WITHOUT_HEADER_INDEX);
373    final long prevBlockOffset = buf.getLong(Header.PREV_BLOCK_OFFSET_INDEX);
374    // This constructor is called when we deserialize a block from cache and when we read a block in
375    // from the fs. fileCache is null when deserialized from cache so need to make up one.
376    HFileContextBuilder fileContextBuilder = fileContext != null?
377        new HFileContextBuilder(fileContext): new HFileContextBuilder();
378    fileContextBuilder.withHBaseCheckSum(usesHBaseChecksum);
379    int onDiskDataSizeWithHeader;
380    if (usesHBaseChecksum) {
381      byte checksumType = buf.get(Header.CHECKSUM_TYPE_INDEX);
382      int bytesPerChecksum = buf.getInt(Header.BYTES_PER_CHECKSUM_INDEX);
383      onDiskDataSizeWithHeader = buf.getInt(Header.ON_DISK_DATA_SIZE_WITH_HEADER_INDEX);
384      // Use the checksum type and bytes per checksum from header, not from filecontext.
385      fileContextBuilder.withChecksumType(ChecksumType.codeToType(checksumType));
386      fileContextBuilder.withBytesPerCheckSum(bytesPerChecksum);
387    } else {
388      fileContextBuilder.withChecksumType(ChecksumType.NULL);
389      fileContextBuilder.withBytesPerCheckSum(0);
390      // Need to fix onDiskDataSizeWithHeader; there are not checksums after-block-data
391      onDiskDataSizeWithHeader = onDiskSizeWithoutHeader + headerSize(usesHBaseChecksum);
392    }
393    fileContext = fileContextBuilder.build();
394    assert usesHBaseChecksum == fileContext.isUseHBaseChecksum();
395    init(blockType, onDiskSizeWithoutHeader, uncompressedSizeWithoutHeader,
396        prevBlockOffset, offset, onDiskDataSizeWithHeader, nextBlockOnDiskSize, fileContext);
397    this.memType = memType;
398    this.offset = offset;
399    this.buf = buf;
400    this.buf.rewind();
401  }

483  /**
484   * Rewinds {@code buf} and writes first 4 header fields. {@code buf} position
485   * is modified as side-effect.
486   */
487  private void overwriteHeader() {
488    buf.rewind();
489    blockType.write(buf);
490    buf.putInt(onDiskSizeWithoutHeader);
491    buf.putInt(uncompressedSizeWithoutHeader);
492    buf.putLong(prevBlockOffset);
493    if (this.fileContext.isUseHBaseChecksum()) {
494      buf.put(fileContext.getChecksumType().getCode());
495      buf.putInt(fileContext.getBytesPerChecksum());
496      buf.putInt(onDiskDataSizeWithHeader);
497    }
498  }

544  /**
545   * Checks if the block is internally consistent, i.e. the first
546   * {@link HConstants#HFILEBLOCK_HEADER_SIZE} bytes of the buffer contain a
547   * valid header consistent with the fields. Assumes a packed block structure.
548   * This function is primary for testing and debugging, and is not
549   * thread-safe, because it alters the internal buffer pointer.
550   * Used by tests only.
551   */
552  @VisibleForTesting
553  void sanityCheck() throws IOException {
554    // Duplicate so no side-effects
555    ByteBuff dup = this.buf.duplicate().rewind();
556    sanityCheckAssertion(BlockType.read(dup), blockType);
557
558    sanityCheckAssertion(dup.getInt(), onDiskSizeWithoutHeader, "onDiskSizeWithoutHeader");
559
560    sanityCheckAssertion(dup.getInt(), uncompressedSizeWithoutHeader,
561        "uncompressedSizeWithoutHeader");
562
563    sanityCheckAssertion(dup.getLong(), prevBlockOffset, "prevBlockOffset");
564    if (this.fileContext.isUseHBaseChecksum()) {
565      sanityCheckAssertion(dup.get(), this.fileContext.getChecksumType().getCode(), "checksumType");
566      sanityCheckAssertion(dup.getInt(), this.fileContext.getBytesPerChecksum(),
567          "bytesPerChecksum");
568      sanityCheckAssertion(dup.getInt(), onDiskDataSizeWithHeader, "onDiskDataSizeWithHeader");
569    }
570
571    int cksumBytes = totalChecksumBytes();
572    int expectedBufLimit = onDiskDataSizeWithHeader + cksumBytes;
573    if (dup.limit() != expectedBufLimit) {
574      throw new AssertionError("Expected limit " + expectedBufLimit + ", got " + dup.limit());
575    }
576
577    // We might optionally allocate HFILEBLOCK_HEADER_SIZE more bytes to read the next
578    // block's header, so there are two sensible values for buffer capacity.
579    int hdrSize = headerSize();
580    if (dup.capacity() != expectedBufLimit && dup.capacity() != expectedBufLimit + hdrSize) {
581      throw new AssertionError("Invalid buffer capacity: " + dup.capacity() +
582          ", expected " + expectedBufLimit + " or " + (expectedBufLimit + hdrSize));
583    }
584  }

627  /**
628   * Retrieves the decompressed/decrypted view of this block. An encoded block remains in its
629   * encoded structure. Internal structures are shared between instances where applicable.
630   */
631  HFileBlock unpack(HFileContext fileContext, FSReader reader) throws IOException {
632    if (!fileContext.isCompressedOrEncrypted()) {
633      // TODO: cannot use our own fileContext here because HFileBlock(ByteBuffer, boolean),
634      // which is used for block serialization to L2 cache, does not preserve encoding and
635      // encryption details.
636      return this;
637    }
638
639    HFileBlock unpacked = new HFileBlock(this);
640    unpacked.allocateBuffer(); // allocates space for the decompressed block
641
642    HFileBlockDecodingContext ctx = blockType == BlockType.ENCODED_DATA ?
643      reader.getBlockDecodingContext() : reader.getDefaultBlockDecodingContext();
644
645    ByteBuff dup = this.buf.duplicate();
646    dup.position(this.headerSize());
647    dup = dup.slice();
648    ctx.prepareDecoding(unpacked.getOnDiskSizeWithoutHeader(),
649      unpacked.getUncompressedSizeWithoutHeader(), unpacked.getBufferWithoutHeader(),
650      dup);
651    return unpacked;
652  }

654  /**
655   * Always allocates a new buffer of the correct size. Copies header bytes
656   * from the existing buffer. Does not change header fields.
657   * Reserve room to keep checksum bytes too.
658   */
659  private void allocateBuffer() {
660    int cksumBytes = totalChecksumBytes();
661    int headerSize = headerSize();
662    int capacityNeeded = headerSize + uncompressedSizeWithoutHeader + cksumBytes;
663
664    // TODO we need consider allocating offheap here?
665    ByteBuffer newBuf = ByteBuffer.allocate(capacityNeeded);
666
667    // Copy header bytes into newBuf.
668    // newBuf is HBB so no issue in calling array()
669    buf.position(0);
670    buf.get(newBuf.array(), newBuf.arrayOffset(), headerSize);
671
672    buf = new SingleByteBuff(newBuf);
673    // set limit to exclude next block's header
674    buf.limit(headerSize + uncompressedSizeWithoutHeader + cksumBytes);
675  }

677  /**
678   * Return true when this block's buffer has been unpacked, false otherwise. Note this is a
679   * calculated heuristic, not tracked attribute of the block.
680   */
681  public boolean isUnpacked() {
682    final int cksumBytes = totalChecksumBytes();
683    final int headerSize = headerSize();
684    final int expectedCapacity = headerSize + uncompressedSizeWithoutHeader + cksumBytes;
685    final int bufCapacity = buf.capacity();
686    return bufCapacity == expectedCapacity || bufCapacity == expectedCapacity + headerSize;
687  }

720  @Override
721  public long heapSize() {
722    long size = ClassSize.align(
723        ClassSize.OBJECT +
724        // Block type, multi byte buffer, MemoryType and meta references
725        4 * ClassSize.REFERENCE +
726        // On-disk size, uncompressed size, and next block's on-disk size
727        // bytePerChecksum and onDiskDataSize
728        4 * Bytes.SIZEOF_INT +
729        // This and previous block offset
730        2 * Bytes.SIZEOF_LONG +
731        // Heap size of the meta object. meta will be always not null.
732        fileContext.heapSize()
733    );
734
735    if (buf != null) {
736      // Deep overhead of the byte buffer. Needs to be aligned separately.
737      size += ClassSize.align(buf.capacity() + MULTI_BYTE_BUFFER_HEAP_SIZE);
738    }
739
740    return ClassSize.align(size);
741  }

743  /**
744   * Read from an input stream at least <code>necessaryLen</code> and if possible,
745   * <code>extraLen</code> also if available. Analogous to
746   * {@link IOUtils#readFully(InputStream, byte[], int, int)}, but specifies a
747   * number of "extra" bytes to also optionally read.
748   *
749   * @param in the input stream to read from
750   * @param buf the buffer to read into
751   * @param bufOffset the destination offset in the buffer
752   * @param necessaryLen the number of bytes that are absolutely necessary to read
753   * @param extraLen the number of extra bytes that would be nice to read
754   * @return true if succeeded reading the extra bytes
755   * @throws IOException if failed to read the necessary bytes
756   */
757  static boolean readWithExtra(InputStream in, byte[] buf,
758      int bufOffset, int necessaryLen, int extraLen) throws IOException {
759    int bytesRemaining = necessaryLen + extraLen;
760    while (bytesRemaining > 0) {
761      int ret = in.read(buf, bufOffset, bytesRemaining);
762      if (ret == -1 && bytesRemaining <= extraLen) {
763        // We could not read the "extra data", but that is OK.
764        break;
765      }
766      if (ret < 0) {
767        throw new IOException("Premature EOF from inputStream (read "
768            + "returned " + ret + ", was trying to read " + necessaryLen
769            + " necessary bytes and " + extraLen + " extra bytes, "
770            + "successfully read "
771            + (necessaryLen + extraLen - bytesRemaining));
772      }
773      bufOffset += ret;
774      bytesRemaining -= ret;
775    }
776    return bytesRemaining <= 0;
777  }

779  /**
780   * Read from an input stream at least <code>necessaryLen</code> and if possible,
781   * <code>extraLen</code> also if available. Analogous to
782   * {@link IOUtils#readFully(InputStream, byte[], int, int)}, but uses
783   * positional read and specifies a number of "extra" bytes that would be
784   * desirable but not absolutely necessary to read.
785   *
786   * @param in the input stream to read from
787   * @param position the position within the stream from which to start reading
788   * @param buf the buffer to read into
789   * @param bufOffset the destination offset in the buffer
790   * @param necessaryLen the number of bytes that are absolutely necessary to
791   *     read
792   * @param extraLen the number of extra bytes that would be nice to read
793   * @return true if and only if extraLen is > 0 and reading those extra bytes
794   *     was successful
795   * @throws IOException if failed to read the necessary bytes
796   */
797  @VisibleForTesting
798  static boolean positionalReadWithExtra(FSDataInputStream in,
799      long position, byte[] buf, int bufOffset, int necessaryLen, int extraLen)
800      throws IOException {
801    int bytesRemaining = necessaryLen + extraLen;
802    int bytesRead = 0;
803    while (bytesRead < necessaryLen) {
804      int ret = in.read(position, buf, bufOffset, bytesRemaining);
805      if (ret < 0) {
806        throw new IOException("Premature EOF from inputStream (positional read "
807            + "returned " + ret + ", was trying to read " + necessaryLen
808            + " necessary bytes and " + extraLen + " extra bytes, "
809            + "successfully read " + bytesRead);
810      }
811      position += ret;
812      bufOffset += ret;
813      bytesRemaining -= ret;
814      bytesRead += ret;
815    }
816    return bytesRead != necessaryLen && bytesRemaining <= 0;
817  }

819  /**
820   * Unified version 2 {@link HFile} block writer. The intended usage pattern
821   * is as follows:
822   * <ol>
823   * <li>Construct an {@link HFileBlock.Writer}, providing a compression algorithm.
824   * <li>Call {@link Writer#startWriting} and get a data stream to write to.
825   * <li>Write your data into the stream.
826   * <li>Call Writer#writeHeaderAndData(FSDataOutputStream) as many times as you need to.
827   * store the serialized block into an external stream.
828   * <li>Repeat to write more blocks.
829   * </ol>
830   * <p>
831   */
832  static class Writer {
833    private enum State {
834      INIT,
835      WRITING,
836      BLOCK_READY
837    };
838
839    /** Writer state. Used to ensure the correct usage protocol. */
840    private State state = State.INIT;
841
842    /** Data block encoder used for data blocks */
843    private final HFileDataBlockEncoder dataBlockEncoder;
844
845    private HFileBlockEncodingContext dataBlockEncodingCtx;
846
847    /** block encoding context for non-data blocks*/
848    private HFileBlockDefaultEncodingContext defaultBlockEncodingCtx;
849
850    /**
851     * The stream we use to accumulate data into a block in an uncompressed format.
852     * We reset this stream at the end of each block and reuse it. The
853     * header is written as the first {@link HConstants#HFILEBLOCK_HEADER_SIZE} bytes into this
854     * stream.
855     */
856    private ByteArrayOutputStream baosInMemory;
857
858    /**
859     * Current block type. Set in {@link #startWriting(BlockType)}. Could be
860     * changed in {@link #finishBlock()} from {@link BlockType#DATA}
861     * to {@link BlockType#ENCODED_DATA}.
862     */
863    private BlockType blockType;
864
865    /**
866     * A stream that we write uncompressed bytes to, which compresses them and
867     * writes them to {@link #baosInMemory}.
868     */
869    private DataOutputStream userDataStream;
870
871    // Size of actual data being written. Not considering the block encoding/compression. This
872    // includes the header size also.
873    private int unencodedDataSizeWritten;
874
875    // Size of actual data being written. considering the block encoding. This
876    // includes the header size also.
877    private int encodedDataSizeWritten;
878
879    /**
880     * Bytes to be written to the file system, including the header. Compressed
881     * if compression is turned on. It also includes the checksum data that
882     * immediately follows the block data. (header + data + checksums)
883     */
884    private ByteArrayOutputStream onDiskBlockBytesWithHeader;
885
886    /**
887     * The size of the checksum data on disk. It is used only if data is
888     * not compressed. If data is compressed, then the checksums are already
889     * part of onDiskBytesWithHeader. If data is uncompressed, then this
890     * variable stores the checksum data for this block.
891     */
892    private byte[] onDiskChecksum = HConstants.EMPTY_BYTE_ARRAY;
893
894    /**
895     * Current block's start offset in the {@link HFile}. Set in
896     * {@link #writeHeaderAndData(FSDataOutputStream)}.
897     */
898    private long startOffset;
899
900    /**
901     * Offset of previous block by block type. Updated when the next block is
902     * started.
903     */
904    private long[] prevOffsetByType;
905
906    /** The offset of the previous block of the same type */
907    private long prevOffset;
908    /** Meta data that holds information about the hfileblock**/
909    private HFileContext fileContext;

911    /**
912     * @param dataBlockEncoder data block encoding algorithm to use
913     */
914    public Writer(HFileDataBlockEncoder dataBlockEncoder, HFileContext fileContext) {
915      if (fileContext.getBytesPerChecksum() < HConstants.HFILEBLOCK_HEADER_SIZE) {
916        throw new RuntimeException("Unsupported value of bytesPerChecksum. " +
917            " Minimum is " + HConstants.HFILEBLOCK_HEADER_SIZE + " but the configured value is " +
918            fileContext.getBytesPerChecksum());
919      }
920      this.dataBlockEncoder = dataBlockEncoder != null?
921          dataBlockEncoder: NoOpDataBlockEncoder.INSTANCE;
922      this.dataBlockEncodingCtx = this.dataBlockEncoder.
923          newDataBlockEncodingContext(HConstants.HFILEBLOCK_DUMMY_HEADER, fileContext);
924      // TODO: This should be lazily instantiated since we usually do NOT need this default encoder
925      this.defaultBlockEncodingCtx = new HFileBlockDefaultEncodingContext(null,
926          HConstants.HFILEBLOCK_DUMMY_HEADER, fileContext);
927      // TODO: Set BAOS initial size. Use fileContext.getBlocksize() and add for header/checksum
928      baosInMemory = new ByteArrayOutputStream();
929      prevOffsetByType = new long[BlockType.values().length];
930      for (int i = 0; i < prevOffsetByType.length; ++i) {
931        prevOffsetByType[i] = UNSET;
932      }
933      // TODO: Why fileContext saved away when we have dataBlockEncoder and/or
934      // defaultDataBlockEncoder?
935      this.fileContext = fileContext;
936    }

938    /**
939     * Starts writing into the block. The previous block's data is discarded.
940     *
941     * @return the stream the user can write their data into
942     * @throws IOException
943     */
944    DataOutputStream startWriting(BlockType newBlockType)
945        throws IOException {
946      if (state == State.BLOCK_READY && startOffset != -1) {
947        // We had a previous block that was written to a stream at a specific
948        // offset. Save that offset as the last offset of a block of that type.
949        prevOffsetByType[blockType.getId()] = startOffset;
950      }
951
952      startOffset = -1;
953      blockType = newBlockType;
954
955      baosInMemory.reset();
956      baosInMemory.write(HConstants.HFILEBLOCK_DUMMY_HEADER);
957
958      state = State.WRITING;
959
960      // We will compress it later in finishBlock()
961      userDataStream = new ByteBufferWriterDataOutputStream(baosInMemory);
962      if (newBlockType == BlockType.DATA) {
963        this.dataBlockEncoder.startBlockEncoding(dataBlockEncodingCtx, userDataStream);
964      }
965      this.unencodedDataSizeWritten = 0;
966      this.encodedDataSizeWritten = 0;
967      return userDataStream;
968    }

970    /**
971     * Writes the Cell to this block
972     * @param cell
973     * @throws IOException
974     */
975    void write(Cell cell) throws IOException{
976      expectState(State.WRITING);
977      int posBeforeEncode = this.userDataStream.size();
978      this.unencodedDataSizeWritten +=
979          this.dataBlockEncoder.encode(cell, dataBlockEncodingCtx, this.userDataStream);
980      this.encodedDataSizeWritten += this.userDataStream.size() - posBeforeEncode;
981    }

985     * of handling compression and buffering for caching on write. Can only be
986     * called in the "writing" state.
987     *
988     * @return the data output stream for the user to write to

991      expectState(State.WRITING);
992      return userDataStream;

995    /**
996     * Transitions the block writer from the "writing" state to the "block
997     * ready" state.  Does nothing if a block is already finished.
998     */
999    void ensureBlockReady() throws IOException {
1000      Preconditions.checkState(state != State.INIT,
1001          "Unexpected state: " + state);
1002
1003      if (state == State.BLOCK_READY) {
1004        return;
1005      }
1006
1007      // This will set state to BLOCK_READY.
1008      finishBlock();
1009    }

1011    /**
1012     * Finish up writing of the block.
1013     * Flushes the compressing stream (if using compression), fills out the header,
1014     * does any compression/encryption of bytes to flush out to disk, and manages
1015     * the cache on write content, if applicable. Sets block write state to "block ready".
1016     */
1017    private void finishBlock() throws IOException {
1018      if (blockType == BlockType.DATA) {
1019        this.dataBlockEncoder.endBlockEncoding(dataBlockEncodingCtx, userDataStream,
1020            baosInMemory.getBuffer(), blockType);
1021        blockType = dataBlockEncodingCtx.getBlockType();
1022      }
1023      userDataStream.flush();
1024      prevOffset = prevOffsetByType[blockType.getId()];
1025
1026      // We need to set state before we can package the block up for cache-on-write. In a way, the
1027      // block is ready, but not yet encoded or compressed.
1028      state = State.BLOCK_READY;
1029      Bytes compressAndEncryptDat;
1030      if (blockType == BlockType.DATA || blockType == BlockType.ENCODED_DATA) {
1031        compressAndEncryptDat = dataBlockEncodingCtx.
1032            compressAndEncrypt(baosInMemory.getBuffer(), 0, baosInMemory.size());
1033      } else {
1034        compressAndEncryptDat = defaultBlockEncodingCtx.
1035            compressAndEncrypt(baosInMemory.getBuffer(), 0, baosInMemory.size());
1036      }
1037      if (compressAndEncryptDat == null) {
1038        compressAndEncryptDat = new Bytes(baosInMemory.getBuffer(), 0, baosInMemory.size());
1039      }
1040      if (onDiskBlockBytesWithHeader == null) {
1041        onDiskBlockBytesWithHeader = new ByteArrayOutputStream(compressAndEncryptDat.getLength());
1042      }
1043      onDiskBlockBytesWithHeader.reset();
1044      onDiskBlockBytesWithHeader.write(compressAndEncryptDat.get(),
1045            compressAndEncryptDat.getOffset(), compressAndEncryptDat.getLength());
1046      // Calculate how many bytes we need for checksum on the tail of the block.
1047      int numBytes = (int) ChecksumUtil.numBytes(
1048          onDiskBlockBytesWithHeader.size(),
1049          fileContext.getBytesPerChecksum());
1050
1051      // Put the header for the on disk bytes; header currently is unfilled-out
1052      putHeader(onDiskBlockBytesWithHeader,
1053          onDiskBlockBytesWithHeader.size() + numBytes,
1054          baosInMemory.size(), onDiskBlockBytesWithHeader.size());
1055      if (onDiskChecksum.length != numBytes) {
1056        onDiskChecksum = new byte[numBytes];
1057      }
1058      ChecksumUtil.generateChecksums(
1059          onDiskBlockBytesWithHeader.getBuffer(), 0,onDiskBlockBytesWithHeader.size(),
1060          onDiskChecksum, 0, fileContext.getChecksumType(), fileContext.getBytesPerChecksum());
1061    }

1063    /**
1064     * Put the header into the given byte array at the given offset.
1065     * @param onDiskSize size of the block on disk header + data + checksum
1066     * @param uncompressedSize size of the block after decompression (but
1067     *          before optional data block decoding) including header
1068     * @param onDiskDataSize size of the block on disk with header
1069     *        and data but not including the checksums
1070     */
1071    private void putHeader(byte[] dest, int offset, int onDiskSize,
1072        int uncompressedSize, int onDiskDataSize) {
1073      offset = blockType.put(dest, offset);
1074      offset = Bytes.putInt(dest, offset, onDiskSize - HConstants.HFILEBLOCK_HEADER_SIZE);
1075      offset = Bytes.putInt(dest, offset, uncompressedSize - HConstants.HFILEBLOCK_HEADER_SIZE);
1076      offset = Bytes.putLong(dest, offset, prevOffset);
1077      offset = Bytes.putByte(dest, offset, fileContext.getChecksumType().getCode());
1078      offset = Bytes.putInt(dest, offset, fileContext.getBytesPerChecksum());
1079      Bytes.putInt(dest, offset, onDiskDataSize);
1080    }

1083        int uncompressedSize, int onDiskDataSize) {
1084      putHeader(dest.getBuffer(),0, onDiskSize, uncompressedSize, onDiskDataSize);
1085    }

1087    /**
1088     * Similar to {@link #writeHeaderAndData(FSDataOutputStream)}, but records
1089     * the offset of this block so that it can be referenced in the next block
1090     * of the same type.
1091     *
1092     * @param out
1093     * @throws IOException
1094     */
1095    void writeHeaderAndData(FSDataOutputStream out) throws IOException {
1096      long offset = out.getPos();
1097      if (startOffset != UNSET && offset != startOffset) {
1098        throw new IOException("A " + blockType + " block written to a "
1099            + "stream twice, first at offset " + startOffset + ", then at "
1100            + offset);
1101      }
1102      startOffset = offset;
1103
1104      finishBlockAndWriteHeaderAndData((DataOutputStream) out);
1105    }

1108     * Writes the header and the compressed data of this block (or uncompressed
1109     * data when not using compression) into the given stream. Can be called in
1110     * the "writing" state or in the "block ready" state. If called in the
1111     * "writing" state, transitions the writer to the "block ready" state.
1112     *
1113     * @param out the output stream to write the
1114     * @throws IOException
1115     */
1116    protected void finishBlockAndWriteHeaderAndData(DataOutputStream out)
1117      throws IOException {
1118      ensureBlockReady();
1119      long startTime = System.currentTimeMillis();
1120      out.write(onDiskBlockBytesWithHeader.getBuffer(), 0, onDiskBlockBytesWithHeader.size());
1121      out.write(onDiskChecksum);
1122      HFile.updateWriteLatency(System.currentTimeMillis() - startTime);
1123    }

1522     *

1524     * @param destOffset offset into the destination buffer at where to put the bytes we read
1525     * @param size size of read
1526     * @param peekIntoNextBlock whether to read the next block's on-disk size
1527     * @param fileOffset position in the stream to read at
1528     * @param pread whether we should do a positional read
1529     * @param istream The input source of data
1530     * @return the on-disk size of the next block with header size included, or

1534     */

1536    protected int readAtOffset(FSDataInputStream istream, byte[] dest, int destOffset, int size,
1537        boolean peekIntoNextBlock, long fileOffset, boolean pread)
1538        throws IOException {
1539      if (peekIntoNextBlock && destOffset + size + hdrSize > dest.length) {
1540        // We are asked to read the next block's header as well, but there is
1541        // not enough room in the array.
1542        throw new IOException("Attempted to read " + size + " bytes and " + hdrSize +
1543            " bytes of next header into a " + dest.length + "-byte array at offset " + destOffset);
1544      }
1545
1546      if (!pread) {
1547        // Seek + read. Better for scanning.
1548        HFileUtil.seekOnMultipleSources(istream, fileOffset);
1549        // TODO: do we need seek time latencies?
1550        long realOffset = istream.getPos();
1551        if (realOffset != fileOffset) {
1552          throw new IOException("Tried to seek to " + fileOffset + " to " + "read " + size +
1553              " bytes, but pos=" + realOffset + " after seek");
1554        }
1555
1556        if (!peekIntoNextBlock) {
1557          IOUtils.readFully(istream, dest, destOffset, size);
1558          return -1;
1559        }
1560
1561        // Try to read the next block header.
1562        if (!readWithExtra(istream, dest, destOffset, size, hdrSize)) {
1563          return -1;
1564        }
1565      } else {
1566        // Positional read. Better for random reads; or when the streamLock is already locked.
1567        int extraSize = peekIntoNextBlock ? hdrSize : 0;
1568        if (!positionalReadWithExtra(istream, fileOffset, dest, destOffset, size, extraSize)) {
1569          return -1;
1570        }
1571      }
1572      assert peekIntoNextBlock;
1573      return Bytes.toInt(dest, destOffset + size + BlockType.MAGIC_LENGTH) + hdrSize;
1574    }

1576    /**
1577     * Reads a version 2 block (version 1 blocks not supported and not expected). Tries to do as
1578     * little memory allocation as possible, using the provided on-disk size.
1579     *
1580     * @param offset the offset in the stream to read at
1581     * @param onDiskSizeWithHeaderL the on-disk size of the block, including
1582     *          the header, or -1 if unknown; i.e. when iterating over blocks reading
1583     *          in the file metadata info.
1584     * @param pread whether to use a positional read
1585     */
1586    @Override
1587    public HFileBlock readBlockData(long offset, long onDiskSizeWithHeaderL, boolean pread,
1588                                    boolean updateMetrics) throws IOException {
1589      // Get a copy of the current state of whether to validate
1590      // hbase checksums or not for this read call. This is not
1591      // thread-safe but the one constaint is that if we decide
1592      // to skip hbase checksum verification then we are
1593      // guaranteed to use hdfs checksum verification.
1594      boolean doVerificationThruHBaseChecksum = streamWrapper.shouldUseHBaseChecksum();
1595      FSDataInputStream is = streamWrapper.getStream(doVerificationThruHBaseChecksum);
1596
1597      HFileBlock blk = readBlockDataInternal(is, offset,
1598                         onDiskSizeWithHeaderL, pread,
1599                         doVerificationThruHBaseChecksum, updateMetrics);
1600      if (blk == null) {
1601        HFile.LOG.warn("HBase checksum verification failed for file " +
1602                       pathName + " at offset " +
1603                       offset + " filesize " + fileSize +
1604                       ". Retrying read with HDFS checksums turned on...");
1605
1606        if (!doVerificationThruHBaseChecksum) {
1607          String msg = "HBase checksum verification failed for file " +
1608                       pathName + " at offset " +
1609                       offset + " filesize " + fileSize +
1610                       " but this cannot happen because doVerify is " +
1611                       doVerificationThruHBaseChecksum;
1612          HFile.LOG.warn(msg);
1613          throw new IOException(msg); // cannot happen case here
1614        }
1615        HFile.CHECKSUM_FAILURES.increment(); // update metrics
1616
1617        // If we have a checksum failure, we fall back into a mode where
1618        // the next few reads use HDFS level checksums. We aim to make the
1619        // next CHECKSUM_VERIFICATION_NUM_IO_THRESHOLD reads avoid
1620        // hbase checksum verification, but since this value is set without
1621        // holding any locks, it can so happen that we might actually do
1622        // a few more than precisely this number.
1623        is = this.streamWrapper.fallbackToFsChecksum(CHECKSUM_VERIFICATION_NUM_IO_THRESHOLD);
1624        doVerificationThruHBaseChecksum = false;
1625        blk = readBlockDataInternal(is, offset, onDiskSizeWithHeaderL, pread,
1626                                    doVerificationThruHBaseChecksum, updateMetrics);
1627        if (blk != null) {
1628          HFile.LOG.warn("HDFS checksum verification succeeded for file " +
1629                         pathName + " at offset " +
1630                         offset + " filesize " + fileSize);
1631        }
1632      }
1633      if (blk == null && !doVerificationThruHBaseChecksum) {
1634        String msg = "readBlockData failed, possibly due to " +
1635                     "checksum verification failed for file " + pathName +
1636                     " at offset " + offset + " filesize " + fileSize;
1637        HFile.LOG.warn(msg);
1638        throw new IOException(msg);
1639      }
1640
1641      // If there is a checksum mismatch earlier, then retry with
1642      // HBase checksums switched off and use HDFS checksum verification.
1643      // This triggers HDFS to detect and fix corrupt replicas. The
1644      // next checksumOffCount read requests will use HDFS checksums.
1645      // The decrementing of this.checksumOffCount is not thread-safe,
1646      // but it is harmless because eventually checksumOffCount will be
1647      // a negative number.
1648      streamWrapper.checksumOk();
1649      return blk;
1650    }

1738      if (LOG.isTraceEnabled()) {
1739        LOG.trace("Reading " + this.fileContext.getHFileName() + " at offset=" + offset +
1740          ", pread=" + pread + ", verifyChecksum=" + verifyChecksum + ", cachedHeader=" +
1741          headerBuf + ", onDiskSizeWithHeader=" + onDiskSizeWithHeader);
1742      }
1743      long startTime = System.currentTimeMillis();
1744      if (onDiskSizeWithHeader <= 0) {
1745        // We were not passed the block size. Need to get it from the header. If header was not in
1746        // cache, need to seek to pull it in. This is costly and should happen very rarely.
1747        // Currently happens on open of a hfile reader where we read the trailer blocks for
1748        // indices. Otherwise, we are reading block sizes out of the hfile index. To check,
1749        // enable TRACE in this file and you'll get an exception in a LOG every time we seek.
1750        // See HBASE-17072 for more detail.
1751        if (headerBuf == null) {
1752          if (LOG.isTraceEnabled()) {
1753            LOG.trace("Extra see to get block size!", new RuntimeException());
1754          }
1755          headerBuf = ByteBuffer.allocate(hdrSize);
1756          readAtOffset(is, headerBuf.array(), headerBuf.arrayOffset(), hdrSize, false,
1757              offset, pread);
1758        }
1759        onDiskSizeWithHeader = getOnDiskSizeWithHeader(headerBuf,
1760          this.fileContext.isUseHBaseChecksum());
1761      }
1762      int preReadHeaderSize = headerBuf == null? 0 : hdrSize;
1763      // Allocate enough space to fit the next block's header too; saves a seek next time through.
1764      // onDiskBlock is whole block + header + checksums then extra hdrSize to read next header;
1765      // onDiskSizeWithHeader is header, body, and any checksums if present. preReadHeaderSize
1766      // says where to start reading. If we have the header cached, then we don't need to read
1767      // it again and we can likely read from last place we left off w/o need to backup and reread
1768      // the header we read last time through here. TODO: Review this overread of the header. Is it necessary
1769      // when we get the block size from the hfile index? See note on PrefetchedHeader class above.
1770      // TODO: Make this ByteBuffer-based. Will make it easier to go to HDFS with BBPool (offheap).
1771      byte [] onDiskBlock = new byte[onDiskSizeWithHeader + hdrSize];
1772      int nextBlockOnDiskSize = readAtOffset(is, onDiskBlock, preReadHeaderSize,
1773          onDiskSizeWithHeader - preReadHeaderSize, true, offset + preReadHeaderSize, pread);
1774      if (headerBuf != null) {
1775        // The header has been read when reading the previous block OR in a distinct header-only
1776        // read. Copy to this block's header.
1777        System.arraycopy(headerBuf.array(), headerBuf.arrayOffset(), onDiskBlock, 0, hdrSize);
1778      } else {
1779        headerBuf = ByteBuffer.wrap(onDiskBlock, 0, hdrSize);
1780      }
1781      // Do a few checks before we go instantiate HFileBlock.
1782      assert onDiskSizeWithHeader > this.hdrSize;
1783      verifyOnDiskSizeMatchesHeader(onDiskSizeWithHeader, headerBuf, offset,
1784        this.fileContext.isUseHBaseChecksum());
1785      ByteBuffer onDiskBlockByteBuffer = ByteBuffer.wrap(onDiskBlock, 0, onDiskSizeWithHeader);
1786      // Verify checksum of the data before using it for building HFileBlock.
1787      if (verifyChecksum &&
1788          !validateChecksum(offset, onDiskBlockByteBuffer, hdrSize)) {
1789        return null;
1790      }
1791      long duration = System.currentTimeMillis() - startTime;
1792      if (updateMetrics) {
1793        HFile.updateReadLatency(duration, pread);
1794      }
1795      // The onDiskBlock will become the headerAndDataBuffer for this block.
1796      // If nextBlockOnDiskSizeWithHeader is not zero, the onDiskBlock already
1797      // contains the header of next block, so no need to set next block's header in it.
1798      HFileBlock hFileBlock =
1799          new HFileBlock(new SingleByteBuff(onDiskBlockByteBuffer),
1800              this.fileContext.isUseHBaseChecksum(), MemoryType.EXCLUSIVE, offset,
1801              nextBlockOnDiskSize, fileContext);
1802      // Run check on uncompressed sizings.
1803      if (!fileContext.isCompressedOrEncrypted()) {
1804        hFileBlock.sanityCheckUncompressed();
1805      }
1806      if (LOG.isTraceEnabled()) {
1807        LOG.trace("Read " + hFileBlock + " in " + duration + " ns");
1808      }
1809      // Cache next block header if we read it for the next time through here.
1810      if (nextBlockOnDiskSize != -1) {
1811        cacheNextBlockHeader(offset + hFileBlock.getOnDiskSizeWithHeader(),
1812            onDiskBlock, onDiskSizeWithHeader, hdrSize);
1813      }
1814      return hFileBlock;
1815    }

1942  public int hashCode() {
1943    int result = 1;
1944    result = result * 31 + blockType.hashCode();
1945    result = result * 31 + nextBlockOnDiskSize;
1946    result = result * 31 + (int) (offset ^ (offset >>> 32));
1947    result = result * 31 + onDiskSizeWithoutHeader;
1948    result = result * 31 + (int) (prevBlockOffset ^ (prevBlockOffset >>> 32));
1949    result = result * 31 + uncompressedSizeWithoutHeader;
1950    result = result * 31 + buf.hashCode();
1951    return result;

1954  @Override
1955  public boolean equals(Object comparison) {
1956    if (this == comparison) {
1957      return true;
1958    }
1959    if (comparison == null) {
1960      return false;
1961    }
1962    if (comparison.getClass() != this.getClass()) {
1963      return false;
1964    }
1965
1966    HFileBlock castedComparison = (HFileBlock) comparison;
1967
1968    if (castedComparison.blockType != this.blockType) {
1969      return false;
1970    }
1971    if (castedComparison.nextBlockOnDiskSize != this.nextBlockOnDiskSize) {
1972      return false;
1973    }
1974    // Offset is important. Needed when we have to remake cachekey when block is returned to cache.
1975    if (castedComparison.offset != this.offset) {
1976      return false;
1977    }
1978    if (castedComparison.onDiskSizeWithoutHeader != this.onDiskSizeWithoutHeader) {
1979      return false;
1980    }
1981    if (castedComparison.prevBlockOffset != this.prevBlockOffset) {
1982      return false;
1983    }
1984    if (castedComparison.uncompressedSizeWithoutHeader != this.uncompressedSizeWithoutHeader) {
1985      return false;
1986    }
1987    if (ByteBuff.compareTo(this.buf, 0, this.buf.limit(), castedComparison.buf, 0,
1988        castedComparison.buf.limit()) != 0) {
1989      return false;
1990    }
1991    return true;
1992  }

2016   * for this block. Each checksum value is a 4 byte integer.

2020    // data to validate. Similarly, a zero value in this.bytesPerChecksum
2021    // indicates that cached blocks do not have checksum data because

2024      return 0;

2027        this.fileContext.getBytesPerChecksum());
