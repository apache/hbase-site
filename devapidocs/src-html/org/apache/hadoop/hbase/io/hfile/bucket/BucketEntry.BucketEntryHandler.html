023import java.io.IOException;
024import java.nio.ByteBuffer;
025import java.util.Comparator;
026import java.util.concurrent.atomic.AtomicBoolean;
027import java.util.concurrent.locks.ReentrantReadWriteLock;
028import java.util.function.Function;
029
030import org.apache.hadoop.hbase.io.ByteBuffAllocator;
031import org.apache.hadoop.hbase.io.ByteBuffAllocator.Recycler;
032import org.apache.hadoop.hbase.io.hfile.BlockPriority;
033import org.apache.hadoop.hbase.io.hfile.Cacheable;
034import org.apache.hadoop.hbase.io.hfile.CacheableDeserializer;
035import org.apache.hadoop.hbase.io.hfile.CacheableDeserializerIdManager;
036import org.apache.hadoop.hbase.nio.ByteBuff;
037import org.apache.hadoop.hbase.nio.HBaseReferenceCounted;
038import org.apache.hadoop.hbase.nio.RefCnt;
039import org.apache.hadoop.hbase.util.IdReadWriteLock;
040import org.apache.yetus.audience.InterfaceAudience;

042/**
043 * Item in cache. We expect this to be where most memory goes. Java uses 8 bytes just for object
044 * headers; after this, we want to use as little as possible - so we only use 8 bytes, but in order
045 * to do so we end up messing around with all this Java casting stuff. Offset stored as 5 bytes that
046 * make up the long. Doubt we'll see devices this big for ages. Offsets are divided by 256. So 5
047 * bytes gives us 256TB or so.
048 */
049@InterfaceAudience.Private
050class BucketEntry implements HBaseReferenceCounted {
051  // access counter comparator, descending order
052  static final Comparator<BucketEntry> COMPARATOR =
053      Comparator.comparingLong(BucketEntry::getAccessCounter).reversed();
054
055  private int offsetBase;
056  private int length;
057  private byte offset1;
058
059  /**
060   * The index of the deserializer that can deserialize this BucketEntry content. See
061   * {@link CacheableDeserializerIdManager} for hosting of index to serializers.
062   */
063  byte deserializerIndex;
064
065  private volatile long accessCounter;
066  private BlockPriority priority;
067
068  /**
069   * The RefCnt means how many paths are referring the {@link BucketEntry}, each RPC reading path is
070   * considering as one path, the {@link BucketCache#backingMap} reference is also considered a
071   * path. NOTICE that if two read RPC path hit the same {@link BucketEntry}, then the HFileBlocks
072   * the two RPC referred will share the same refCnt instance with the BucketEntry. so the refCnt
073   * will increase or decrease as the following: <br>
074   * 1. when writerThread flush the block into IOEngine and add the bucketEntry into backingMap, the
075   * refCnt ++; <br>
076   * 2. If BucketCache evict the block and move the bucketEntry out of backingMap, the refCnt--; it
077   * usually happen when HFile is closing or someone call the clearBucketCache by force. <br>
078   * 3. The read RPC path start to refer the block which is backend by the memory area in
079   * bucketEntry, then refCnt ++ ; <br>
080   * 4. The read RPC patch shipped the response, and release the block. then refCnt--; <br>
081   * Once the refCnt decrease to zero, then the {@link BucketAllocator} will free the block area.
082   */
083  private final RefCnt refCnt;
084  final AtomicBoolean markedAsEvicted;
085  final ByteBuffAllocator allocator;
086
087  /**
088   * Time this block was cached. Presumes we are created just before we are added to the cache.
089   */
090  private final long cachedTime = System.nanoTime();
091
092  BucketEntry(long offset, int length, long accessCounter, boolean inMemory) {
093    this(offset, length, accessCounter, inMemory, null, ByteBuffAllocator.HEAP);
094  }
095
096  BucketEntry(long offset, int length, long accessCounter, boolean inMemory,
097      Function<BucketEntry, Recycler> createRecycler,
098      ByteBuffAllocator allocator) {
099    setOffset(offset);
100    this.length = length;
101    this.accessCounter = accessCounter;
102    this.priority = inMemory ? BlockPriority.MEMORY : BlockPriority.MULTI;
103    if (createRecycler == null) {
104      this.refCnt = RefCnt.create();
105    } else {
106      this.refCnt = RefCnt.create(createRecycler.apply(this));
107    }
108    this.markedAsEvicted = new AtomicBoolean(false);
109    this.allocator = allocator;
110  }
111
112  long offset() {
113    // Java has no unsigned numbers, so this needs the L cast otherwise it will be sign extended
114    // as a negative number.
115    long o = ((long) offsetBase) & 0xFFFFFFFFL;
116    // The 0xFF here does not need the L cast because it is treated as a positive int.
117    o += (((long) (offset1)) & 0xFF) << 32;
118    return o << 8;
119  }
120
121  private void setOffset(long value) {
122    assert (value & 0xFF) == 0;
123    value >>= 8;
124    offsetBase = (int) value;
125    offset1 = (byte) (value >> 32);
126  }
127
128  public int getLength() {
129    return length;
130  }
131
132  CacheableDeserializer<Cacheable> deserializerReference() {
133    return CacheableDeserializerIdManager.getDeserializer(deserializerIndex);
134  }
135
136  void setDeserializerReference(CacheableDeserializer<Cacheable> deserializer) {
137    this.deserializerIndex = (byte) deserializer.getDeserializerIdentifier();
138  }
139
140  long getAccessCounter() {
141    return accessCounter;
142  }
143
144  /**
145   * Block has been accessed. Update its local access counter.
146   */
147  void access(long accessCounter) {
148    this.accessCounter = accessCounter;
149    if (this.priority == BlockPriority.SINGLE) {
150      this.priority = BlockPriority.MULTI;
151    }
152  }
153
154  public BlockPriority getPriority() {
155    return this.priority;
156  }
157
158  long getCachedTime() {
159    return cachedTime;
160  }
161
162  /**
163   * The {@link BucketCache} will try to release its reference to this BucketEntry many times. we
164   * must make sure the idempotent, otherwise it'll decrease the RPC's reference count in advance,
165   * then for RPC memory leak happen.
166   * @return true if we deallocate this entry successfully.
167   */
168  boolean markAsEvicted() {
169    if (markedAsEvicted.compareAndSet(false, true)) {
170      return this.release();
171    }
172    return false;
173  }
174
175  /**
176   * Check whether have some RPC patch referring this block. There're two case: <br>
177   * 1. If current refCnt is greater than 1, there must be at least one referring RPC path; <br>
178   * 2. If current refCnt is equal to 1 and the markedAtEvicted is true, the it means backingMap has
179   * released its reference, the remaining reference can only be from RPC path. <br>
180   * We use this check to decide whether we can free the block area: when cached size exceed the
181   * acceptable size, our eviction policy will choose those stale blocks without any RPC reference
182   * and the RPC referred block will be excluded.
183   * @return true to indicate there're some RPC referring the block.
184   */
185  boolean isRpcRef() {
186    boolean evicted = markedAsEvicted.get();
187    return this.refCnt() > 1 || (evicted && refCnt() == 1);
188  }
189
190  Cacheable wrapAsCacheable(ByteBuffer[] buffers) throws IOException {
191    return wrapAsCacheable(ByteBuff.wrap(buffers, this.refCnt));
192  }
193
194  Cacheable wrapAsCacheable(ByteBuff buf) throws IOException {
195    return this.deserializerReference().deserialize(buf, allocator);
196  }
197
198  interface BucketEntryHandler<T> {
199    T handle();
200  }
201
202  <T> T withWriteLock(IdReadWriteLock<Long> offsetLock, BucketEntryHandler<T> handler) {
203    ReentrantReadWriteLock lock = offsetLock.getLock(this.offset());
204    try {
205      lock.writeLock().lock();
206      return handler.handle();
207    } finally {
208      lock.writeLock().unlock();
209    }
210  }
211
212  @Override
213  public int refCnt() {
214    return this.refCnt.refCnt();
215  }
216
217  @Override
218  public BucketEntry retain() {
219    refCnt.retain();
220    return this;
221  }
222
223  /**
224   * We've three cases to release refCnt now: <br>
225   * 1. BucketCache#evictBlock, it will release the backingMap's reference by force because we're
226   * closing file or clear the bucket cache or some corruption happen. when all rpc references gone,
227   * then free the area in bucketAllocator. <br>
228   * 2. BucketCache#returnBlock . when rpc shipped, we'll release the block, only when backingMap
229   * also release its refCnt (case.1 will do this) and no other rpc reference, then it will free the
230   * area in bucketAllocator. <br>
231   * 3.evict those block without any rpc reference if cache size exceeded. we'll only free those
232   * blocks with zero rpc reference count, as the {@link BucketEntry#markStaleAsEvicted()} do.
233   * @return true to indicate we've decreased to zero and do the de-allocation.
234   */
235  @Override
236  public boolean release() {
237    return refCnt.release();
238  }
239}
