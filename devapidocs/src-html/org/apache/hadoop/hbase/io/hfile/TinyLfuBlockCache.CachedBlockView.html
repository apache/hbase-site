046/**
047 * A block cache that is memory-aware using {@link HeapSize}, memory bounded using the W-TinyLFU
048 * eviction algorithm, and concurrent. This implementation delegates to a Caffeine cache to provide
049 * O(1) read and write operations.
050 * <ul>
051 *   <li>W-TinyLFU: http://arxiv.org/pdf/1512.00727.pdf</li>
052 *   <li>Caffeine: https://github.com/ben-manes/caffeine</li>
053 *   <li>Cache design: http://highscalability.com/blog/2016/1/25/design-of-a-modern-cache.html</li>
054 * </ul>
055 */
056@InterfaceAudience.Private
057public final class TinyLfuBlockCache implements FirstLevelBlockCache {
058  private static final Logger LOG = LoggerFactory.getLogger(TinyLfuBlockCache.class);
059
060  private static final String MAX_BLOCK_SIZE = "hbase.tinylfu.max.block.size";
061  private static final long DEFAULT_MAX_BLOCK_SIZE = 16L * 1024L * 1024L;
062  private static final int STAT_THREAD_PERIOD_SECONDS = 5 * 60;
063
064  private final Eviction<BlockCacheKey, Cacheable> policy;
065  private final ScheduledExecutorService statsThreadPool;
066  private final long maxBlockSize;
067  private final CacheStats stats;
068
069  private BlockCache victimCache;
070
071  final Cache<BlockCacheKey, Cacheable> cache;

073  /**
074   * Creates a block cache.
075   *
076   * @param maximumSizeInBytes maximum size of this cache, in bytes
077   * @param avgBlockSize expected average size of blocks, in bytes
078   * @param executor the cache's executor
079   * @param conf additional configuration
080   */
081  public TinyLfuBlockCache(long maximumSizeInBytes, long avgBlockSize,
082      Executor executor, Configuration conf) {
083    this(maximumSizeInBytes, avgBlockSize,
084        conf.getLong(MAX_BLOCK_SIZE, DEFAULT_MAX_BLOCK_SIZE), executor);
085  }
086
087  /**
088   * Creates a block cache.
089   *
090   * @param maximumSizeInBytes maximum size of this cache, in bytes
091   * @param avgBlockSize expected average size of blocks, in bytes
092   * @param maxBlockSize maximum size of a block, in bytes
093   * @param executor the cache's executor
094   */
095  public TinyLfuBlockCache(long maximumSizeInBytes,
096      long avgBlockSize, long maxBlockSize, Executor executor) {
097    this.cache = Caffeine.newBuilder()
098        .executor(executor)
099        .maximumWeight(maximumSizeInBytes)
100        .removalListener(new EvictionListener())
101        .weigher((BlockCacheKey key, Cacheable value) ->
102            (int) Math.min(value.heapSize(), Integer.MAX_VALUE))
103        .initialCapacity((int) Math.ceil((1.2 * maximumSizeInBytes) / avgBlockSize))
104        .build();
105    this.maxBlockSize = maxBlockSize;
106    this.policy = cache.policy().eviction().get();
107    this.stats = new CacheStats(getClass().getSimpleName());
108
109    statsThreadPool = Executors.newSingleThreadScheduledExecutor(new ThreadFactoryBuilder()
110        .setNameFormat("TinyLfuBlockCacheStatsExecutor").setDaemon(true).build());
111    statsThreadPool.scheduleAtFixedRate(this::logStats,
112        STAT_THREAD_PERIOD_SECONDS, STAT_THREAD_PERIOD_SECONDS, TimeUnit.SECONDS);
113  }

158  @Override
159  public Cacheable getBlock(BlockCacheKey cacheKey,
160      boolean caching, boolean repeat, boolean updateCacheMetrics) {
161    Cacheable value = cache.asMap().computeIfPresent(cacheKey, (blockCacheKey, cacheable) -> {
162      // It will be referenced by RPC path, so increase here. NOTICE: Must do the retain inside
163      // this block. because if retain outside the map#computeIfPresent, the evictBlock may remove
164      // the block and release, then we're retaining a block with refCnt=0 which is disallowed.
165      cacheable.retain();
166      return cacheable;
167    });
168    if (value == null) {
169      if (repeat) {
170        return null;
171      }
172      if (updateCacheMetrics) {
173        stats.miss(caching, cacheKey.isPrimary(), cacheKey.getBlockType());
174      }
175      if (victimCache != null) {
176        value = victimCache.getBlock(cacheKey, caching, repeat, updateCacheMetrics);
177        if ((value != null) && caching) {
178          cacheBlock(cacheKey, value);
179        }
180      }
181    } else if (updateCacheMetrics) {
182      stats.hit(caching, cacheKey.isPrimary(), cacheKey.getBlockType());
183    }
184    return value;
185  }

192  @Override
193  public void cacheBlock(BlockCacheKey key, Cacheable value) {
194    if (value.heapSize() > maxBlockSize) {
195      // If there are a lot of blocks that are too big this can make the logs too noisy (2% logged)
196      if (stats.failInsert() % 50 == 0) {
197        LOG.warn(String.format(
198          "Trying to cache too large a block %s @ %,d is %,d which is larger than %,d",
199          key.getHfileName(), key.getOffset(), value.heapSize(), DEFAULT_MAX_BLOCK_SIZE));
200      }
201    } else {
202      value = asReferencedHeapBlock(value);
203      cache.put(key, value);
204    }
205  }
206
207  /**
208   * The block cached in TinyLfuBlockCache will always be an heap block: on the one side, the heap
209   * access will be more faster then off-heap, the small index block or meta block cached in
210   * CombinedBlockCache will benefit a lot. on other side, the TinyLfuBlockCache size is always
211   * calculated based on the total heap size, if caching an off-heap block in TinyLfuBlockCache, the
212   * heap size will be messed up. Here we will clone the block into an heap block if it's an
213   * off-heap block, otherwise just use the original block. The key point is maintain the refCnt of
214   * the block (HBASE-22127): <br>
215   * 1. if cache the cloned heap block, its refCnt is an totally new one, it's easy to handle; <br>
216   * 2. if cache the original heap block, we're sure that it won't be tracked in ByteBuffAllocator's
217   * reservoir, if both RPC and TinyLfuBlockCache release the block, then it can be
218   * garbage collected by JVM, so need a retain here.
219   *
220   * @param buf the original block
221   * @return an block with an heap memory backend.
222   */
223  private Cacheable asReferencedHeapBlock(Cacheable buf) {
224    if (buf instanceof HFileBlock) {
225      HFileBlock blk = ((HFileBlock) buf);
226      if (blk.isSharedMem()) {
227        return HFileBlock.deepCloneOnHeap(blk);
228      }
229    }
230    // The block will be referenced by this TinyLfuBlockCache, so should increase its refCnt here.
231    return buf.retain();
232  }
233
234  @Override
235  public boolean evictBlock(BlockCacheKey cacheKey) {
236    Cacheable value = cache.asMap().remove(cacheKey);
237    if (value != null) {
238      value.release();
239    }
240    return (value != null);
241  }
242
243  @Override
244  public int evictBlocksByHfileName(String hfileName) {
245    int evicted = 0;
246    for (BlockCacheKey key : cache.asMap().keySet()) {
247      if (key.getHfileName().equals(hfileName) && evictBlock(key)) {
248        evicted++;
249      }
250    }
251    if (victimCache != null) {
252      evicted += victimCache.evictBlocksByHfileName(hfileName);
253    }
254    return evicted;
255  }

275  @Override
276  public Iterator<CachedBlock> iterator() {
277    long now = System.nanoTime();
278    return cache.asMap().entrySet().stream()
279        .map(entry -> (CachedBlock) new CachedBlockView(entry.getKey(), entry.getValue(), now))

283  private void logStats() {
284    LOG.info(
285        "totalSize=" + StringUtils.byteDesc(heapSize()) + ", " +
286        "freeSize=" + StringUtils.byteDesc(getFreeSize()) + ", " +
287        "max=" + StringUtils.byteDesc(size()) + ", " +
288        "blockCount=" + getBlockCount() + ", " +
289        "accesses=" + stats.getRequestCount() + ", " +
290        "hits=" + stats.getHitCount() + ", " +
291        "hitRatio=" + (stats.getHitCount() == 0 ?
292          "0," : StringUtils.formatPercent(stats.getHitRatio(), 2) + ", ") +
293        "cachingAccesses=" + stats.getRequestCachingCount() + ", " +
294        "cachingHits=" + stats.getHitCachingCount() + ", " +
295        "cachingHitsRatio=" + (stats.getHitCachingCount() == 0 ?
296          "0,": (StringUtils.formatPercent(stats.getHitCachingRatio(), 2) + ", ")) +
297        "evictions=" + stats.getEvictionCount() + ", " +
298        "evicted=" + stats.getEvictedCount());
299  }

313  /** A removal listener to asynchronously record evictions and populate the victim cache. */
314  private final class EvictionListener implements RemovalListener<BlockCacheKey, Cacheable> {
315
316    @Override
317    public void onRemoval(BlockCacheKey key, Cacheable value, RemovalCause cause) {
318      if (!cause.wasEvicted()) {
319        // An explicit eviction (invalidation) is not added to the victim cache as the data may
320        // no longer be valid for subsequent queries.
321        return;
322      }
323
324      recordEviction();
325
326      if (victimCache == null) {
327        return;
328      } else if (victimCache instanceof BucketCache) {
329        BucketCache victimBucketCache = (BucketCache) victimCache;
330        victimBucketCache.cacheBlockWithWait(key, value, /* inMemory */ true, /* wait */ true);
331      } else {
332        victimCache.cacheBlock(key, value);
333      }
334    }
335  }
336
337  /**
338   * Records an eviction. The number of eviction operations and evicted blocks are identical, as
339   * an eviction is triggered immediately when the capacity has been exceeded. An eviction is
340   * performed asynchronously. See the library's documentation for details on write buffers,
341   * batching, and maintenance behavior.
342   */
343  private void recordEviction() {
344    // FIXME: Currently does not capture the insertion time
345    stats.evicted(Long.MAX_VALUE, true);
346    stats.evict();
347  }

349  private static final class CachedBlockView implements CachedBlock {
350    private static final Comparator<CachedBlock> COMPARATOR = Comparator
351        .comparing(CachedBlock::getFilename)
352        .thenComparing(CachedBlock::getOffset)
353        .thenComparing(CachedBlock::getCachedTime);
354
355    private final BlockCacheKey key;
356    private final Cacheable value;
357    private final long now;
358
359    public CachedBlockView(BlockCacheKey key, Cacheable value, long now) {
360      this.now = now;
361      this.key = key;
362      this.value = value;
363    }
364
365    @Override
366    public BlockPriority getBlockPriority() {
367      // This does not appear to be used in any meaningful way and is irrelevant to this cache
368      return BlockPriority.MEMORY;
369    }
370
371    @Override
372    public BlockType getBlockType() {
373      return value.getBlockType();
374    }
375
376    @Override
377    public long getOffset() {
378      return key.getOffset();
379    }
380
381    @Override
382    public long getSize() {
383      return value.heapSize();
384    }
385
386    @Override
387    public long getCachedTime() {
388      // This does not appear to be used in any meaningful way, so not captured
389      return 0L;
390    }
391
392    @Override
393    public String getFilename() {
394      return key.getHfileName();
395    }
396
397    @Override
398    public int compareTo(CachedBlock other) {
399      return COMPARATOR.compare(this, other);
400    }
401
402    @Override
403    public boolean equals(Object obj) {
404      if (obj == this) {
405        return true;
406      } else if (!(obj instanceof CachedBlock)) {
407        return false;
408      }
409      CachedBlock other = (CachedBlock) obj;
410      return compareTo(other) == 0;
411    }
412
413    @Override
414    public int hashCode() {
415      return key.hashCode();
416    }
417
418    @Override
419    public String toString() {
420      return BlockCacheUtil.toString(this, now);
421    }
422  }

426    return size();
427  }

431    return getCurrentSize();
432  }
