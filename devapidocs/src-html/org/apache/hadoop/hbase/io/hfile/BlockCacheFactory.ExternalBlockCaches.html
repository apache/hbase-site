051  /**
052   * If the chosen ioengine can persist its state across restarts, the path to the file to persist
053   * to. This file is NOT the data file. It is a file into which we will serialize the map of
054   * what is in the data file. For example, if you pass the following argument as
055   * BUCKET_CACHE_IOENGINE_KEY ("hbase.bucketcache.ioengine"),
056   * <code>file:/tmp/bucketcache.data </code>, then we will write the bucketcache data to the file

059   * in-memory state is what you supply here: e.g. <code>/tmp/bucketcache.map</code>.
060   */

107  public static BlockCache createBlockCache(Configuration conf) {
108    if (conf.get(DEPRECATED_BLOCKCACHE_BLOCKSIZE_KEY) != null) {
109      LOG.warn("The config key {} is deprecated now, instead please use {}. In future release "
110          + "we will remove the deprecated config.", DEPRECATED_BLOCKCACHE_BLOCKSIZE_KEY,
111        BLOCKCACHE_BLOCKSIZE_KEY);
112    }
113    FirstLevelBlockCache l1Cache = createFirstLevelCache(conf);
114    if (l1Cache == null) {
115      return null;
116    }
117    boolean useExternal = conf.getBoolean(EXTERNAL_BLOCKCACHE_KEY, EXTERNAL_BLOCKCACHE_DEFAULT);
118    if (useExternal) {
119      BlockCache l2CacheInstance = createExternalBlockcache(conf);
120      return l2CacheInstance == null ?
121          l1Cache :
122          new InclusiveCombinedBlockCache(l1Cache, l2CacheInstance);
123    } else {
124      // otherwise use the bucket cache.
125      BucketCache bucketCache = createBucketCache(conf);
126      if (!conf.getBoolean("hbase.bucketcache.combinedcache.enabled", true)) {
127        // Non combined mode is off from 2.0
128        LOG.warn(
129            "From HBase 2.0 onwards only combined mode of LRU cache and bucket cache is available");
130      }
131      return bucketCache == null ? l1Cache : new CombinedBlockCache(l1Cache, bucketCache);
132    }
133  }
134
135  private static FirstLevelBlockCache createFirstLevelCache(final Configuration c) {
136    final long cacheSize = MemorySizeUtil.getOnHeapCacheSize(c);
137    if (cacheSize < 0) {
138      return null;
139    }
140    String policy = c.get(BLOCKCACHE_POLICY_KEY, BLOCKCACHE_POLICY_DEFAULT);
141    int blockSize = c.getInt(BLOCKCACHE_BLOCKSIZE_KEY, HConstants.DEFAULT_BLOCKSIZE);
142    LOG.info("Allocating BlockCache size=" +
143        StringUtils.byteDesc(cacheSize) + ", blockSize=" + StringUtils.byteDesc(blockSize));
144    if (policy.equalsIgnoreCase("LRU")) {
145      return new LruBlockCache(cacheSize, blockSize, true, c);
146    } else if (policy.equalsIgnoreCase("TinyLFU")) {
147      return new TinyLfuBlockCache(cacheSize, blockSize, ForkJoinPool.commonPool(), c);
148    } else if (policy.equalsIgnoreCase("AdaptiveLRU")) {
149      return new LruAdaptiveBlockCache(cacheSize, blockSize, true, c);
150    } else {
151      throw new IllegalArgumentException("Unknown policy: " + policy);
152    }
153  }

155  /**
156   * Enum of all built in external block caches.
157   * This is used for config.
158   */
159  private static enum ExternalBlockCaches {
160    memcached("org.apache.hadoop.hbase.io.hfile.MemcachedBlockCache");
161    // TODO(eclark): Consider more. Redis, etc.
162    Class<? extends BlockCache> clazz;
163    ExternalBlockCaches(String clazzName) {
164      try {
165        clazz = (Class<? extends BlockCache>) Class.forName(clazzName);
166      } catch (ClassNotFoundException cnef) {
167        clazz = null;
168      }
169    }
170    ExternalBlockCaches(Class<? extends BlockCache> clazz) {
171      this.clazz = clazz;
172    }
173  }
174
175  private static BlockCache createExternalBlockcache(Configuration c) {
176    if (LOG.isDebugEnabled()) {
177      LOG.debug("Trying to use External l2 cache");
178    }
179    Class klass = null;
180
181    // Get the class, from the config. s
182    try {
183      klass = ExternalBlockCaches
184          .valueOf(c.get(EXTERNAL_BLOCKCACHE_CLASS_KEY, "memcache")).clazz;
185    } catch (IllegalArgumentException exception) {
186      try {
187        klass = c.getClass(EXTERNAL_BLOCKCACHE_CLASS_KEY, Class.forName(
188            "org.apache.hadoop.hbase.io.hfile.MemcachedBlockCache"));
189      } catch (ClassNotFoundException e) {
190        return null;
191      }
192    }
193
194    // Now try and create an instance of the block cache.
195    try {
196      LOG.info("Creating external block cache of type: " + klass);
197      return (BlockCache) ReflectionUtils.newInstance(klass, c);
198    } catch (Exception e) {
199      LOG.warn("Error creating external block cache", e);
200    }
201    return null;
202
203  }

205  private static BucketCache createBucketCache(Configuration c) {
206    // Check for L2.  ioengine name must be non-null.
207    String bucketCacheIOEngineName = c.get(BUCKET_CACHE_IOENGINE_KEY, null);
208    if (bucketCacheIOEngineName == null || bucketCacheIOEngineName.length() <= 0) {
209      return null;
210    }
211
212    int blockSize = c.getInt(BLOCKCACHE_BLOCKSIZE_KEY, HConstants.DEFAULT_BLOCKSIZE);
213    final long bucketCacheSize = MemorySizeUtil.getBucketCacheSize(c);
214    if (bucketCacheSize <= 0) {
215      throw new IllegalStateException("bucketCacheSize <= 0; Check " +
216          BUCKET_CACHE_SIZE_KEY + " setting and/or server java heap size");
217    }
218    if (c.get("hbase.bucketcache.percentage.in.combinedcache") != null) {
219      LOG.warn("Configuration 'hbase.bucketcache.percentage.in.combinedcache' is no longer "
220          + "respected. See comments in http://hbase.apache.org/book.html#_changes_of_note");
221    }
222    int writerThreads = c.getInt(BUCKET_CACHE_WRITER_THREADS_KEY,
223        DEFAULT_BUCKET_CACHE_WRITER_THREADS);
224    int writerQueueLen = c.getInt(BUCKET_CACHE_WRITER_QUEUE_KEY,
225        DEFAULT_BUCKET_CACHE_WRITER_QUEUE);
226    String persistentPath = c.get(BUCKET_CACHE_PERSISTENT_PATH_KEY);
227    String[] configuredBucketSizes = c.getStrings(BUCKET_CACHE_BUCKETS_KEY);
228    int [] bucketSizes = null;
229    if (configuredBucketSizes != null) {
230      bucketSizes = new int[configuredBucketSizes.length];
231      for (int i = 0; i < configuredBucketSizes.length; i++) {
232        int bucketSize = Integer.parseInt(configuredBucketSizes[i].trim());
233        if (bucketSize % 256 != 0) {
234          // We need all the bucket sizes to be multiples of 256. Having all the configured bucket
235          // sizes to be multiples of 256 will ensure that the block offsets within buckets,
236          // that are calculated, will also be multiples of 256.
237          // See BucketEntry where offset to each block is represented using 5 bytes (instead of 8
238          // bytes long). We would like to save heap overhead as less as possible.
239          throw new IllegalArgumentException("Illegal value: " + bucketSize + " configured for '"
240              + BUCKET_CACHE_BUCKETS_KEY + "'. All bucket sizes to be multiples of 256");
241        }
242        bucketSizes[i] = bucketSize;
243      }
244    }
245    BucketCache bucketCache = null;
246    try {
247      int ioErrorsTolerationDuration = c.getInt(
248          "hbase.bucketcache.ioengine.errors.tolerated.duration",
249          BucketCache.DEFAULT_ERROR_TOLERATION_DURATION);
250      // Bucket cache logs its stats on creation internal to the constructor.
251      bucketCache = new BucketCache(bucketCacheIOEngineName,
252          bucketCacheSize, blockSize, bucketSizes, writerThreads, writerQueueLen, persistentPath,
253          ioErrorsTolerationDuration, c);
254    } catch (IOException ioex) {
255      LOG.error("Can't instantiate bucket cache", ioex); throw new RuntimeException(ioex);
256    }
257    return bucketCache;
258  }
