168   * Opens a HFile. You must load the index before you can use it by calling
169   * {@link #loadFileInfo()}.
170   * @param path
171   *          Path to HFile.
172   * @param trailer
173   *          File trailer.
174   * @param fsdis
175   *          input stream.
176   * @param fileSize
177   *          Length of the stream.
178   * @param cacheConf
179   *          Cache configuration.
180   * @param hfs
181   *          The file system.
182   * @param conf
183   *          Configuration
184   */
185  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="URF_UNREAD_PUBLIC_OR_PROTECTED_FIELD")
186  public HFileReaderImpl(Path path, FixedFileTrailer trailer, FSDataInputStreamWrapper fsdis,
187      long fileSize, CacheConfig cacheConf, HFileSystem hfs, boolean primaryReplicaReader,
188      Configuration conf) throws IOException {
189    this.trailer = trailer;
190    this.compressAlgo = trailer.getCompressionCodec();
191    this.cacheConf = cacheConf;
192    this.fileSize = fileSize;
193    this.path = path;
194    this.name = path.getName();
195    this.conf = conf;
196    this.primaryReplicaReader = primaryReplicaReader;
197    checkFileVersion();
198    this.hfileContext = createHFileContext(fsdis, fileSize, hfs, path, trailer);
199    this.fsBlockReader = new HFileBlock.FSReaderImpl(fsdis, fileSize, hfs, path, hfileContext);
200
201    // Comparator class name is stored in the trailer in version 2.
202    comparator = trailer.createComparator();
203    dataBlockIndexReader = new HFileBlockIndex.CellBasedKeyBlockIndexReader(comparator,
204        trailer.getNumDataIndexLevels(), this);
205    metaBlockIndexReader = new HFileBlockIndex.ByteArrayKeyBlockIndexReader(1);
206
207    // Parse load-on-open data.
208
209    HFileBlock.BlockIterator blockIter = fsBlockReader.blockRange(
210        trailer.getLoadOnOpenDataOffset(),
211        fileSize - trailer.getTrailerSize());
212
213    // Data index. We also read statistics about the block index written after
214    // the root level.
215    dataBlockIndexReader.readMultiLevelIndexRoot(
216        blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX),
217        trailer.getDataIndexCount());
218
219    // Meta index.
220    metaBlockIndexReader.readRootIndex(
221        blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX),
222        trailer.getMetaIndexCount());
223
224    // File info
225    fileInfo = new FileInfo();
226    fileInfo.read(blockIter.nextBlockWithBlockType(BlockType.FILE_INFO).getByteStream());
227    byte[] creationTimeBytes = fileInfo.get(FileInfo.CREATE_TIME_TS);
228    this.hfileContext.setFileCreateTime(creationTimeBytes == null?  0:
229        Bytes.toLong(creationTimeBytes));
230    if (fileInfo.get(FileInfo.LASTKEY) != null) {
231      lastKeyCell = new KeyValue.KeyOnlyKeyValue(fileInfo.get(FileInfo.LASTKEY));
232    }
233    avgKeyLen = Bytes.toInt(fileInfo.get(FileInfo.AVG_KEY_LEN));
234    avgValueLen = Bytes.toInt(fileInfo.get(FileInfo.AVG_VALUE_LEN));
235    byte [] keyValueFormatVersion = fileInfo.get(HFileWriterImpl.KEY_VALUE_VERSION);
236    includesMemstoreTS = keyValueFormatVersion != null &&
237        Bytes.toInt(keyValueFormatVersion) == HFileWriterImpl.KEY_VALUE_VER_WITH_MEMSTORE;
238    fsBlockReader.setIncludesMemstoreTS(includesMemstoreTS);
239    if (includesMemstoreTS) {
240      decodeMemstoreTS = Bytes.toLong(fileInfo.get(HFileWriterImpl.MAX_MEMSTORE_TS_KEY)) > 0;
241    }
242
243    // Read data block encoding algorithm name from file info.
244    dataBlockEncoder = HFileDataBlockEncoderImpl.createFromFileInfo(fileInfo);
245    fsBlockReader.setDataBlockEncoder(dataBlockEncoder);
246
247    // Store all other load-on-open blocks for further consumption.
248    HFileBlock b;
249    while ((b = blockIter.nextBlock()) != null) {
250      loadOnOpenBlocks.add(b);
251    }
252
253    // Prefetch file blocks upon open if requested
254    if (cacheConf.shouldPrefetchOnOpen()) {
255      PrefetchExecutor.request(path, new Runnable() {
256        public void run() {
257          long offset = 0;
258          long end = 0;
259          try {
260            end = getTrailer().getLoadOnOpenDataOffset();
261            if (LOG.isTraceEnabled()) {
262              LOG.trace("Prefetch start " + getPathOffsetEndStr(path, offset, end));
263            }
264            // TODO: Could we use block iterator in here? Would that get stuff into the cache?
265            HFileBlock prevBlock = null;
266            while (offset < end) {
267              if (Thread.interrupted()) {
268                break;
269              }
270              // Perhaps we got our block from cache? Unlikely as this may be, if it happens, then
271              // the internal-to-hfileblock thread local which holds the overread that gets the
272              // next header, will not have happened...so, pass in the onDiskSize gotten from the
273              // cached block. This 'optimization' triggers extremely rarely I'd say.
274              long onDiskSize = prevBlock != null? prevBlock.getNextBlockOnDiskSize(): -1;
275              HFileBlock block = readBlock(offset, onDiskSize, true, false, false, false,
276                  null, null);
277              // Need not update the current block. Ideally here the readBlock won't find the
278              // block in cache. We call this readBlock so that block data is read from FS and
279              // cached in BC. So there is no reference count increment that happens here.
280              // The return will ideally be a noop because the block is not of MemoryType SHARED.
281              returnBlock(block);
282              prevBlock = block;
283              offset += block.getOnDiskSizeWithHeader();
284            }
285          } catch (IOException e) {
286            // IOExceptions are probably due to region closes (relocation, etc.)
287            if (LOG.isTraceEnabled()) {
288              LOG.trace("Prefetch " + getPathOffsetEndStr(path, offset, end), e);
289            }
290          } catch (NullPointerException e) {
291            LOG.warn("Stream moved/closed or prefetch cancelled?" +
292                getPathOffsetEndStr(path, offset, end), e);
293          } catch (Exception e) {
294            // Other exceptions are interesting
295            LOG.warn("Prefetch " + getPathOffsetEndStr(path, offset, end), e);
296          } finally {
297            PrefetchExecutor.complete(path);
298          }
299        }
300      });
301    }
302
303    byte[] tmp = fileInfo.get(FileInfo.MAX_TAGS_LEN);
304    // max tag length is not present in the HFile means tags were not at all written to file.
305    if (tmp != null) {
306      hfileContext.setIncludesTags(true);
307      tmp = fileInfo.get(FileInfo.TAGS_COMPRESSED);
308      if (tmp != null && Bytes.toBoolean(tmp)) {
309        hfileContext.setCompressTags(true);
310      }
311    }
312  }

318  /**
319   * File version check is a little sloppy. We read v3 files but can also read v2 files if their
320   * content has been pb'd; files written with 0.98.
321   */
322  private void checkFileVersion() {
323    int majorVersion = trailer.getMajorVersion();
324    if (majorVersion == getMajorVersion()) return;
325    int minorVersion = trailer.getMinorVersion();
326    if (majorVersion == 2 && minorVersion >= MIN_V2_MINOR_VERSION_WITH_PB) return;
327    // We can read v3 or v2 versions of hfile.
328    throw new IllegalArgumentException("Invalid HFile version: major=" +
329      trailer.getMajorVersion() + ", minor=" + trailer.getMinorVersion() + ": expected at least " +
330      "major=2 and minor=" + MAX_MINOR_VERSION);
331  }

598    // Returns the #bytes in HFile for the current cell. Used to skip these many bytes in current
599    // HFile block's buffer so as to position to the next cell.

601      int curCellSize =  KEY_VALUE_LEN_SIZE + currKeyLen + currValueLen

609    protected void readKeyValueLen() {
610      // This is a hot method. We go out of our way to make this method short so it can be
611      // inlined and is not too big to compile. We also manage position in ByteBuffer ourselves
612      // because it is faster than going via range-checked ByteBuffer methods or going through a
613      // byte buffer array a byte at a time.
614      // Get a long at a time rather than read two individual ints. In micro-benchmarking, even
615      // with the extra bit-fiddling, this is order-of-magnitude faster than getting two ints.
616      // Trying to imitate what was done - need to profile if this is better or
617      // earlier way is better by doing mark and reset?
618      // But ensure that you read long instead of two ints
619      long ll = blockBuffer.getLongAfterPosition(0);
620      // Read top half as an int of key length and bottom int as value length
621      this.currKeyLen = (int)(ll >> Integer.SIZE);
622      this.currValueLen = (int)(Bytes.MASK_FOR_LOWER_INT_IN_LONG ^ ll);
623      checkKeyValueLen();
624      // Move position past the key and value lengths and then beyond the key and value
625      int p = (Bytes.SIZEOF_LONG + currKeyLen + currValueLen);
626      if (reader.getFileContext().isIncludesTags()) {
627        // Tags length is a short.
628        this.currTagsLen = blockBuffer.getShortAfterPosition(p);
629        checkTagsLen();
630        p += (Bytes.SIZEOF_SHORT + currTagsLen);
631      }
632      readMvccVersion(p);
633    }

638          ". Block offset: " + curBlock.getOffset() + ", block length: " +
639            this.blockBuffer.limit() +
640          ", position: " + this.blockBuffer.position() + " (without header).");
641      }

644    /**
645     * Read mvcc. Does checks to see if we even need to read the mvcc at all.
646     * @param offsetFromPos
647     */
648    protected void readMvccVersion(final int offsetFromPos) {
649      // See if we even need to decode mvcc.
650      if (!this.reader.shouldIncludeMemstoreTS()) return;
651      if (!this.reader.isDecodeMemstoreTS()) {
652        currMemstoreTS = 0;
653        currMemstoreTSLen = 1;
654        return;
655      }
656      _readMvccVersion(offsetFromPos);
657    }

659    /**
660     * Actually do the mvcc read. Does no checks.
661     * @param offsetFromPos
662     */
663    private void _readMvccVersion(int offsetFromPos) {
664      // This is Bytes#bytesToVint inlined so can save a few instructions in this hot method; i.e.
665      // previous if one-byte vint, we'd redo the vint call to find int size.
666      // Also the method is kept small so can be inlined.
667      byte firstByte = blockBuffer.getByteAfterPosition(offsetFromPos);
668      int len = WritableUtils.decodeVIntSize(firstByte);
669      if (len == 1) {
670        this.currMemstoreTS = firstByte;
671      } else {
672        int remaining = len -1;
673        long i = 0;
674        offsetFromPos++;
675        if (remaining >= Bytes.SIZEOF_INT) {
676          // The int read has to be converted to unsigned long so the & op
677          i = (blockBuffer.getIntAfterPosition(offsetFromPos) & 0x00000000ffffffffL);
678          remaining -= Bytes.SIZEOF_INT;
679          offsetFromPos += Bytes.SIZEOF_INT;
680        }
681        if (remaining >= Bytes.SIZEOF_SHORT) {
682          short s = blockBuffer.getShortAfterPosition(offsetFromPos);
683          i = i << 16;
684          i = i | (s & 0xFFFF);
685          remaining -= Bytes.SIZEOF_SHORT;
686          offsetFromPos += Bytes.SIZEOF_SHORT;
687        }
688        for (int idx = 0; idx < remaining; idx++) {
689          byte b = blockBuffer.getByteAfterPosition(offsetFromPos + idx);
690          i = i << 8;
691          i = i | (b & 0xFF);
692        }
693        currMemstoreTS = (WritableUtils.isNegativeVInt(firstByte) ? ~i : i);
694      }
695      this.currMemstoreTSLen = len;
696    }

698    /**
699     * Within a loaded block, seek looking for the last key that is smaller than
700     * (or equal to?) the key we are interested in.
701     * A note on the seekBefore: if you have seekBefore = true, AND the first
702     * key in the block = key, then you'll get thrown exceptions. The caller has
703     * to check for that case and load the previous block as appropriate.
704     * @param key
705     *          the key to find
706     * @param seekBefore
707     *          find the key before the given key in case of exact match.
708     * @return 0 in case of an exact key match, 1 in case of an inexact match,
709     *         -2 in case of an inexact match and furthermore, the input key
710     *         less than the first key of current block(e.g. using a faked index
711     *         key)
712     */
713    protected int blockSeek(Cell key, boolean seekBefore) {
714      int klen, vlen, tlen = 0;
715      int lastKeyValueSize = -1;
716      int offsetFromPos;
717      do {
718        offsetFromPos = 0;
719        // Better to ensure that we use the BB Utils here
720        long ll = blockBuffer.getLongAfterPosition(offsetFromPos);
721        klen = (int)(ll >> Integer.SIZE);
722        vlen = (int)(Bytes.MASK_FOR_LOWER_INT_IN_LONG ^ ll);
723        if (checkKeyLen(klen) || checkLen(vlen)) {
724          throw new IllegalStateException("Invalid klen " + klen + " or vlen "
725              + vlen + ". Block offset: "
726              + curBlock.getOffset() + ", block length: " + blockBuffer.limit() + ", position: "
727              + blockBuffer.position() + " (without header).");
728        }
729        offsetFromPos += Bytes.SIZEOF_LONG;
730        blockBuffer.asSubByteBuffer(blockBuffer.position() + offsetFromPos, klen, pair);
731        bufBackedKeyOnlyKv.setKey(pair.getFirst(), pair.getSecond(), klen);
732        int comp = reader.getComparator().compareKeyIgnoresMvcc(key, bufBackedKeyOnlyKv);
733        offsetFromPos += klen + vlen;
734        if (this.reader.getFileContext().isIncludesTags()) {
735          // Read short as unsigned, high byte first
736          tlen = ((blockBuffer.getByteAfterPosition(offsetFromPos) & 0xff) << 8)
737              ^ (blockBuffer.getByteAfterPosition(offsetFromPos + 1) & 0xff);
738          if (checkLen(tlen)) {
739            throw new IllegalStateException("Invalid tlen " + tlen + ". Block offset: "
740                + curBlock.getOffset() + ", block length: " + blockBuffer.limit() + ", position: "
741                + blockBuffer.position() + " (without header).");
742          }
743          // add the two bytes read for the tags.
744          offsetFromPos += tlen + (Bytes.SIZEOF_SHORT);
745        }
746        if (this.reader.shouldIncludeMemstoreTS()) {
747          // Directly read the mvcc based on current position
748          readMvccVersion(offsetFromPos);
749        }
750        if (comp == 0) {
751          if (seekBefore) {
752            if (lastKeyValueSize < 0) {
753              throw new IllegalStateException("blockSeek with seekBefore "
754                  + "at the first key of the block: key=" + CellUtil.getCellKeyAsString(key)
755                  + ", blockOffset=" + curBlock.getOffset() + ", onDiskSize="
756                  + curBlock.getOnDiskSizeWithHeader());
757            }
758            blockBuffer.moveBack(lastKeyValueSize);
759            readKeyValueLen();
760            return 1; // non exact match.
761          }
762          currKeyLen = klen;
763          currValueLen = vlen;
764          currTagsLen = tlen;
765          return 0; // indicate exact match
766        } else if (comp < 0) {
767          if (lastKeyValueSize > 0) {
768            blockBuffer.moveBack(lastKeyValueSize);
769          }
770          readKeyValueLen();
771          if (lastKeyValueSize == -1 && blockBuffer.position() == 0) {
772            return HConstants.INDEX_KEY_MAGIC;
773          }
774          return 1;
775        }
776        // The size of this key/value tuple, including key/value length fields.
777        lastKeyValueSize = klen + vlen + currMemstoreTSLen + KEY_VALUE_LEN_SIZE;
778        // include tag length also if tags included with KV
779        if (reader.getFileContext().isIncludesTags()) {
780          lastKeyValueSize += tlen + Bytes.SIZEOF_SHORT;
781        }
782        blockBuffer.skip(lastKeyValueSize);
783      } while (blockBuffer.hasRemaining());
784
785      // Seek to the last key we successfully read. This will happen if this is
786      // the last key/value pair in the file, in which case the following call
787      // to next() has to return false.
788      blockBuffer.moveBack(lastKeyValueSize);
789      readKeyValueLen();
790      return 1; // didn't exactly find it.
791    }

803    @Override
804    public int reseekTo(Cell key) throws IOException {
805      int compared;
806      if (isSeeked()) {
807        compared = compareKey(reader.getComparator(), key);
808        if (compared < 1) {
809          // If the required key is less than or equal to current key, then
810          // don't do anything.
811          return compared;
812        } else {
813          // The comparison with no_next_index_key has to be checked
814          if (this.nextIndexedKey != null &&
815              (this.nextIndexedKey == KeyValueScanner.NO_NEXT_INDEXED_KEY || reader
816              .getComparator().compareKeyIgnoresMvcc(key, nextIndexedKey) < 0)) {
817            // The reader shall continue to scan the current data block instead
818            // of querying the
819            // block index as long as it knows the target key is strictly
820            // smaller than
821            // the next indexed key or the current data block is the last data
822            // block.
823            return loadBlockAndSeekToKey(this.curBlock, nextIndexedKey, false, key,
824                false);
825          }
826
827        }
828      }
829      // Don't rewind on a reseek operation, because reseek implies that we are
830      // always going forward in the file.
831      return seekTo(key, false);
832    }

835     * An internal API function. Seek to the given key, optionally rewinding to
836     * the first key of the block before doing the seek.
837     *
838     * @param key - a cell representing the key that we need to fetch
839     * @param rewind whether to rewind to the first key of the block before
840     *        doing the seek. If this is false, we are assuming we never go
841     *        back, otherwise the result is undefined.
842     * @return -1 if the key is earlier than the first key of the file,
843     *         0 if we are at the given key, 1 if we are past the given key
844     *         -2 if the key is earlier than the first key of the file while
845     *         using a faked index key
846     * @throws IOException
847     */
848    public int seekTo(Cell key, boolean rewind) throws IOException {
849      HFileBlockIndex.BlockIndexReader indexReader = reader.getDataBlockIndexReader();
850      BlockWithScanInfo blockWithScanInfo = indexReader.loadDataBlockWithScanInfo(key, curBlock,
851          cacheBlocks, pread, isCompaction, getEffectiveDataBlockEncoding());
852      if (blockWithScanInfo == null || blockWithScanInfo.getHFileBlock() == null) {
853        // This happens if the key e.g. falls before the beginning of the
854        // file.
855        return -1;
856      }
857      return loadBlockAndSeekToKey(blockWithScanInfo.getHFileBlock(),
858          blockWithScanInfo.getNextIndexedKey(), rewind, key, false);
859    }

861    @Override
862    public boolean seekBefore(Cell key) throws IOException {
863      HFileBlock seekToBlock = reader.getDataBlockIndexReader().seekToDataBlock(key, curBlock,
864          cacheBlocks, pread, isCompaction, reader.getEffectiveEncodingInCache(isCompaction));
865      if (seekToBlock == null) {
866        return false;
867      }
868      Cell firstKey = getFirstKeyCellInBlock(seekToBlock);
869      if (reader.getComparator()
870           .compareKeyIgnoresMvcc(firstKey, key) >= 0) {
871        long previousBlockOffset = seekToBlock.getPrevBlockOffset();
872        // The key we are interested in
873        if (previousBlockOffset == -1) {
874          // we have a 'problem', the key we want is the first of the file.
875          return false;
876        }
877
878        // The first key in the current block 'seekToBlock' is greater than the given 
879        // seekBefore key. We will go ahead by reading the next block that satisfies the
880        // given key. Return the current block before reading the next one.
881        reader.returnBlock(seekToBlock);
882        // It is important that we compute and pass onDiskSize to the block
883        // reader so that it does not have to read the header separately to
884        // figure out the size.  Currently, we do not have a way to do this
885        // correctly in the general case however.
886        // TODO: See https://issues.apache.org/jira/browse/HBASE-14576
887        int prevBlockSize = -1;
888        seekToBlock = reader.readBlock(previousBlockOffset,
889            prevBlockSize, cacheBlocks,
890            pread, isCompaction, true, BlockType.DATA, getEffectiveDataBlockEncoding());
891        // TODO shortcut: seek forward in this block to the last key of the
892        // block.
893      }
894      loadBlockAndSeekToKey(seekToBlock, firstKey, true, key, true);
895      return true;
896    }

898    /**
899     * Scans blocks in the "scanned" section of the {@link HFile} until the next
900     * data block is found.
901     *
902     * @return the next block, or null if there are no more data blocks
903     * @throws IOException
904     */
905    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="NP_NULL_ON_SOME_PATH",
906        justification="Yeah, unnecessary null check; could do w/ clean up")
907    protected HFileBlock readNextDataBlock() throws IOException {
908      long lastDataBlockOffset = reader.getTrailer().getLastDataBlockOffset();
909      if (curBlock == null)
910        return null;
911
912      HFileBlock block = this.curBlock;
913
914      do {
915        if (block.getOffset() >= lastDataBlockOffset) {
916          return null;
917        }
918
919        if (block.getOffset() < 0) {
920          throw new IOException("Invalid block file offset: " + block);
921        }
922
923        // We are reading the next block without block type validation, because
924        // it might turn out to be a non-data block.
925        block = reader.readBlock(block.getOffset() + block.getOnDiskSizeWithHeader(),
926            block.getNextBlockOnDiskSize(), cacheBlocks, pread,
927            isCompaction, true, null, getEffectiveDataBlockEncoding());
928        if (block != null && !block.getBlockType().isData()) { // Findbugs: NP_NULL_ON_SOME_PATH
929          // Whatever block we read we will be returning it unless
930          // it is a datablock. Just in case the blocks are non data blocks
931          reader.returnBlock(block);
932        }
933      } while (!block.getBlockType().isData());
934
935      return block;
936    }

943    public Cell getCell() {
944      if (!isSeeked())
945        return null;
946
947      Cell ret;
948      int cellBufSize = getKVBufSize();
949      long seqId = 0l;
950      if (this.reader.shouldIncludeMemstoreTS()) {
951        seqId = currMemstoreTS;
952      }
953      if (blockBuffer.hasArray()) {
954        // TODO : reduce the varieties of KV here. Check if based on a boolean
955        // we can handle the 'no tags' case.
956        if (currTagsLen > 0) {
957          ret = new SizeCachedKeyValue(blockBuffer.array(),
958              blockBuffer.arrayOffset() + blockBuffer.position(), cellBufSize, seqId);
959        } else {
960          ret = new SizeCachedNoTagsKeyValue(blockBuffer.array(),
961              blockBuffer.arrayOffset() + blockBuffer.position(), cellBufSize, seqId);
962        }
963      } else {
964        ByteBuffer buf = blockBuffer.asSubByteBuffer(cellBufSize);
965        if (buf.isDirect()) {
966          ret = new ByteBufferKeyValue(buf, buf.position(), cellBufSize, seqId);
967        } else {
968          if (currTagsLen > 0) {
969            ret = new SizeCachedKeyValue(buf.array(), buf.arrayOffset() + buf.position(),
970                cellBufSize, seqId);
971          } else {
972            ret = new SizeCachedNoTagsKeyValue(buf.array(), buf.arrayOffset() + buf.position(),
973                cellBufSize, seqId);
974          }
975        }
976      }
977      return ret;
978    }

981    public Cell getKey() {
982      assertSeeked();
983      // Create a new object so that this getKey is cached as firstKey, lastKey
984      ObjectIntPair<ByteBuffer> keyPair = new ObjectIntPair<>();
985      blockBuffer.asSubByteBuffer(blockBuffer.position() + KEY_VALUE_LEN_SIZE, currKeyLen, keyPair);
986      ByteBuffer keyBuf = keyPair.getFirst();
987      if (keyBuf.hasArray()) {
988        return new KeyValue.KeyOnlyKeyValue(keyBuf.array(), keyBuf.arrayOffset()
989            + keyPair.getSecond(), currKeyLen);
990      } else {
991        // Better to do a copy here instead of holding on to this BB so that
992        // we could release the blocks referring to this key. This key is specifically used 
993        // in HalfStoreFileReader to get the firstkey and lastkey by creating a new scanner
994        // every time. So holding onto the BB (incase of DBB) is not advised here.
995        byte[] key = new byte[currKeyLen];
996        ByteBufferUtils.copyFromBufferToArray(key, keyBuf, keyPair.getSecond(), 0, currKeyLen);
997        return new KeyValue.KeyOnlyKeyValue(key, 0, currKeyLen);
998      }

1288  /**
1289   * Retrieve block from cache. Validates the retrieved block's type vs {@code expectedBlockType}
1290   * and its encoding vs. {@code expectedDataBlockEncoding}. Unpacks the block as necessary.
1291   */
1292   private HFileBlock getCachedBlock(BlockCacheKey cacheKey, boolean cacheBlock, boolean useLock,
1293       boolean isCompaction, boolean updateCacheMetrics, BlockType expectedBlockType,
1294       DataBlockEncoding expectedDataBlockEncoding) throws IOException {
1295     // Check cache for block. If found return.
1296     if (cacheConf.isBlockCacheEnabled()) {
1297       BlockCache cache = cacheConf.getBlockCache();
1298       HFileBlock cachedBlock = (HFileBlock) cache.getBlock(cacheKey, cacheBlock, useLock,
1299         updateCacheMetrics);
1300       if (cachedBlock != null) {
1301         if (cacheConf.shouldCacheCompressed(cachedBlock.getBlockType().getCategory())) {
1302           HFileBlock compressedBlock = cachedBlock;
1303           cachedBlock = compressedBlock.unpack(hfileContext, fsBlockReader);
1304           // In case of compressed block after unpacking we can return the compressed block
1305          if (compressedBlock != cachedBlock) {
1306            cache.returnBlock(cacheKey, compressedBlock);
1307          }
1308        }
1309         validateBlockType(cachedBlock, expectedBlockType);
1310
1311         if (expectedDataBlockEncoding == null) {
1312           return cachedBlock;
1313         }
1314         DataBlockEncoding actualDataBlockEncoding =
1315                 cachedBlock.getDataBlockEncoding();
1316         // Block types other than data blocks always have
1317         // DataBlockEncoding.NONE. To avoid false negative cache misses, only
1318         // perform this check if cached block is a data block.
1319         if (cachedBlock.getBlockType().isData() &&
1320                 !actualDataBlockEncoding.equals(expectedDataBlockEncoding)) {
1321           // This mismatch may happen if a Scanner, which is used for say a
1322           // compaction, tries to read an encoded block from the block cache.
1323           // The reverse might happen when an EncodedScanner tries to read
1324           // un-encoded blocks which were cached earlier.
1325           //
1326           // Because returning a data block with an implicit BlockType mismatch
1327           // will cause the requesting scanner to throw a disk read should be
1328           // forced here. This will potentially cause a significant number of
1329           // cache misses, so update so we should keep track of this as it might
1330           // justify the work on a CompoundScanner.
1331           if (!expectedDataBlockEncoding.equals(DataBlockEncoding.NONE) &&
1332                   !actualDataBlockEncoding.equals(DataBlockEncoding.NONE)) {
1333             // If the block is encoded but the encoding does not match the
1334             // expected encoding it is likely the encoding was changed but the
1335             // block was not yet evicted. Evictions on file close happen async
1336             // so blocks with the old encoding still linger in cache for some
1337             // period of time. This event should be rare as it only happens on
1338             // schema definition change.
1339             LOG.info("Evicting cached block with key " + cacheKey +
1340                     " because of a data block encoding mismatch" +
1341                     "; expected: " + expectedDataBlockEncoding +
1342                     ", actual: " + actualDataBlockEncoding);
1343             // This is an error scenario. so here we need to decrement the
1344             // count.
1345             cache.returnBlock(cacheKey, cachedBlock);
1346             cache.evictBlock(cacheKey);
1347           }
1348           return null;
1349         }
1350         return cachedBlock;
1351       }
1352     }
1353     return null;
1354   }

1356  /**
1357   * @param metaBlockName
1358   * @param cacheBlock Add block to cache, if found
1359   * @return block wrapped in a ByteBuffer, with header skipped
1360   * @throws IOException
1361   */
1362  @Override
1363  public HFileBlock getMetaBlock(String metaBlockName, boolean cacheBlock)
1364      throws IOException {
1365    if (trailer.getMetaIndexCount() == 0) {
1366      return null; // there are no meta blocks
1367    }
1368    if (metaBlockIndexReader == null) {
1369      throw new IOException("Meta index not loaded");
1370    }
1371
1372    byte[] mbname = Bytes.toBytes(metaBlockName);
1373    int block = metaBlockIndexReader.rootBlockContainingKey(mbname,
1374        0, mbname.length);
1375    if (block == -1)
1376      return null;
1377    long blockSize = metaBlockIndexReader.getRootBlockDataSize(block);
1378
1379    // Per meta key from any given file, synchronize reads for said block. This
1380    // is OK to do for meta blocks because the meta block index is always
1381    // single-level.
1382    synchronized (metaBlockIndexReader
1383        .getRootBlockKey(block)) {
1384      // Check cache for block. If found return.
1385      long metaBlockOffset = metaBlockIndexReader.getRootBlockOffset(block);
1386      BlockCacheKey cacheKey = new BlockCacheKey(name, metaBlockOffset,
1387        this.isPrimaryReplicaReader(), BlockType.META);
1388
1389      cacheBlock &= cacheConf.shouldCacheBlockOnRead(BlockType.META.getCategory());
1390      if (cacheConf.isBlockCacheEnabled()) {
1391        HFileBlock cachedBlock = getCachedBlock(cacheKey, cacheBlock, false, true, true,
1392          BlockType.META, null);
1393        if (cachedBlock != null) {
1394          assert cachedBlock.isUnpacked() : "Packed block leak.";
1395          // Return a distinct 'shallow copy' of the block,
1396          // so pos does not get messed by the scanner
1397          return cachedBlock;
1398        }
1399        // Cache Miss, please load.
1400      }
1401
1402      HFileBlock metaBlock = fsBlockReader.readBlockData(metaBlockOffset, blockSize, true, false).
1403          unpack(hfileContext, fsBlockReader);
1404
1405      // Cache the block
1406      if (cacheBlock) {
1407        cacheConf.getBlockCache().cacheBlock(cacheKey, metaBlock,
1408            cacheConf.isInMemory(), this.cacheConf.isCacheDataInL1());
1409      }
1410
1411      return metaBlock;
1412    }

1415  @Override
1416  public HFileBlock readBlock(long dataBlockOffset, long onDiskBlockSize,
1417      final boolean cacheBlock, boolean pread, final boolean isCompaction,
1418      boolean updateCacheMetrics, BlockType expectedBlockType,
1419      DataBlockEncoding expectedDataBlockEncoding)
1420      throws IOException {
1421    if (dataBlockIndexReader == null) {
1422      throw new IOException("Block index not loaded");
1423    }
1424    long trailerOffset = trailer.getLoadOnOpenDataOffset();
1425    if (dataBlockOffset < 0 || dataBlockOffset >= trailerOffset) {
1426      throw new IOException("Requested block is out of range: " + dataBlockOffset +
1427        ", lastDataBlockOffset: " + trailer.getLastDataBlockOffset() +
1428        ", trailer.getLoadOnOpenDataOffset: " + trailerOffset);
1429    }
1430    // For any given block from any given file, synchronize reads for said
1431    // block.
1432    // Without a cache, this synchronizing is needless overhead, but really
1433    // the other choice is to duplicate work (which the cache would prevent you
1434    // from doing).
1435
1436    BlockCacheKey cacheKey = new BlockCacheKey(name, dataBlockOffset,
1437      this.isPrimaryReplicaReader(), expectedBlockType);
1438
1439    boolean useLock = false;
1440    IdLock.Entry lockEntry = null;
1441    TraceScope traceScope = Trace.startSpan("HFileReaderImpl.readBlock");
1442    try {
1443      while (true) {
1444        // Check cache for block. If found return.
1445        if (cacheConf.shouldReadBlockFromCache(expectedBlockType)) {
1446          if (useLock) {
1447            lockEntry = offsetLock.getLockEntry(dataBlockOffset);
1448          }
1449          // Try and get the block from the block cache. If the useLock variable is true then this
1450          // is the second time through the loop and it should not be counted as a block cache miss.
1451          HFileBlock cachedBlock = getCachedBlock(cacheKey, cacheBlock, useLock, isCompaction,
1452            updateCacheMetrics, expectedBlockType, expectedDataBlockEncoding);
1453          if (cachedBlock != null) {
1454            if (LOG.isTraceEnabled()) {
1455              LOG.trace("From Cache " + cachedBlock);
1456            }
1457            if (Trace.isTracing()) {
1458              traceScope.getSpan().addTimelineAnnotation("blockCacheHit");
1459            }
1460            assert cachedBlock.isUnpacked() : "Packed block leak.";
1461            if (cachedBlock.getBlockType().isData()) {
1462              if (updateCacheMetrics) {
1463                HFile.DATABLOCK_READ_COUNT.increment();
1464              }
1465              // Validate encoding type for data blocks. We include encoding
1466              // type in the cache key, and we expect it to match on a cache hit.
1467              if (cachedBlock.getDataBlockEncoding() != dataBlockEncoder.getDataBlockEncoding()) {
1468                throw new IOException("Cached block under key " + cacheKey + " "
1469                  + "has wrong encoding: " + cachedBlock.getDataBlockEncoding() + " (expected: "
1470                  + dataBlockEncoder.getDataBlockEncoding() + ")");
1471              }
1472            }
1473            // Cache-hit. Return!
1474            return cachedBlock;
1475          }
1476
1477          if (!useLock && cacheBlock && cacheConf.shouldLockOnCacheMiss(expectedBlockType)) {
1478            // check cache again with lock
1479            useLock = true;
1480            continue;
1481          }
1482          // Carry on, please load.
1483        }
1484
1485        if (Trace.isTracing()) {
1486          traceScope.getSpan().addTimelineAnnotation("blockCacheMiss");
1487        }
1488        // Load block from filesystem.
1489        HFileBlock hfileBlock =
1490            fsBlockReader.readBlockData(dataBlockOffset, onDiskBlockSize, pread, !isCompaction);
1491        validateBlockType(hfileBlock, expectedBlockType);
1492        HFileBlock unpacked = hfileBlock.unpack(hfileContext, fsBlockReader);
1493        BlockType.BlockCategory category = hfileBlock.getBlockType().getCategory();
1494
1495        // Cache the block if necessary
1496        if (cacheBlock && cacheConf.shouldCacheBlockOnRead(category)) {
1497          cacheConf.getBlockCache().cacheBlock(cacheKey,
1498            cacheConf.shouldCacheCompressed(category) ? hfileBlock : unpacked,
1499            cacheConf.isInMemory(), this.cacheConf.isCacheDataInL1());
1500        }
1501
1502        if (updateCacheMetrics && hfileBlock.getBlockType().isData()) {
1503          HFile.DATABLOCK_READ_COUNT.increment();
1504        }
1505
1506        return unpacked;
1507      }
1508    } finally {
1509      traceScope.close();
1510      if (lockEntry != null) {
1511        offsetLock.releaseLockEntry(lockEntry);
1512      }
1513    }
1514  }

1522   * Compares the actual type of a block retrieved from cache or disk with its
1523   * expected type and throws an exception in case of a mismatch. Expected
1524   * block type of {@link BlockType#DATA} is considered to match the actual
1525   * block type [@link {@link BlockType#ENCODED_DATA} as well.
1526   * @param block a block retrieved from cache or disk
1527   * @param expectedBlockType the expected block type, or null to skip the
1528   *          check
1529   */

1531      BlockType expectedBlockType) throws IOException {
1532    if (expectedBlockType == null) {
1533      return;
1534    }
1535    BlockType actualBlockType = block.getBlockType();
1536    if (expectedBlockType.isData() && actualBlockType.isData()) {
1537      // We consider DATA to match ENCODED_DATA for the purpose of this
1538      // verification.
1539      return;
1540    }
1541    if (actualBlockType != expectedBlockType) {
1542      throw new IOException("Expected block type " + expectedBlockType + ", " +
1543          "but got " + actualBlockType + ": " + block);
1544    }

1620    /**
1621     * Updates the current block to be the given {@link HFileBlock}. Seeks to
1622     * the the first key/value pair.
1623     *
1624     * @param newBlock the block to make current
1625     * @throws CorruptHFileException
1626     */
1627    @Override
1628    protected void updateCurrentBlock(HFileBlock newBlock) throws CorruptHFileException {
1629
1630      // sanity checks
1631      if (newBlock.getBlockType() != BlockType.ENCODED_DATA) {
1632        throw new IllegalStateException("EncodedScanner works only on encoded data blocks");
1633      }
1634      short dataBlockEncoderId = newBlock.getDataBlockEncodingId();
1635      if (!DataBlockEncoding.isCorrectEncoder(dataBlockEncoder, dataBlockEncoderId)) {
1636        String encoderCls = dataBlockEncoder.getClass().getName();
1637        throw new CorruptHFileException("Encoder " + encoderCls
1638          + " doesn't support data block encoding "
1639          + DataBlockEncoding.getNameFromId(dataBlockEncoderId));
1640      }
1641      updateCurrBlockRef(newBlock);
1642      ByteBuff encodedBuffer = getEncodedBuffer(newBlock);
1643      seeker.setCurrentBuffer(encodedBuffer);
1644      blockFetches.incrementAndGet();
1645
1646      // Reset the next indexed key
1647      this.nextIndexedKey = null;
1648    }

1650    private ByteBuff getEncodedBuffer(HFileBlock newBlock) {
1651      ByteBuff origBlock = newBlock.getBufferReadOnly();
1652      int pos = newBlock.headerSize() + DataBlockEncoding.ID_SIZE;
1653      origBlock.position(pos);
1654      origBlock
1655          .limit(pos + newBlock.getUncompressedSizeWithoutHeader() - DataBlockEncoding.ID_SIZE);
1656      return origBlock.slice();
1657    }

1661      seeker.rewind();
1662      return true;
1663    }

1665    @Override
1666    public boolean next() throws IOException {
1667      boolean isValid = seeker.next();
1668      if (!isValid) {
1669        HFileBlock newBlock = readNextDataBlock();
1670        isValid = newBlock != null;
1671        if (isValid) {
1672          updateCurrentBlock(newBlock);
1673        } else {
1674          setNonSeekedState();
1675        }
1676      }
1677      return isValid;
1678    }

1803      }

1805      cryptoContext.setKey(key);
1806      builder.withEncryptionContext(cryptoContext);
1807    }
1808
1809    HFileContext context = builder.build();
1810
1811    if (LOG.isTraceEnabled()) {
1812      LOG.trace("Reader" + (path != null? " for " + path: "") +
1813        " initialized with cacheConf: " + cacheConf +
1814        " comparator: " + comparator.getClass().getSimpleName() +
1815        " fileContext: " + context);
1816    }
