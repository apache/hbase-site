160  public HFileWriterImpl(final Configuration conf, CacheConfig cacheConf, Path path,
161      FSDataOutputStream outputStream,
162      CellComparator comparator, HFileContext fileContext) {
163    this.outputStream = outputStream;
164    this.path = path;
165    this.name = path != null ? path.getName() : outputStream.toString();
166    this.hFileContext = fileContext;
167    DataBlockEncoding encoding = hFileContext.getDataBlockEncoding();
168    if (encoding != DataBlockEncoding.NONE) {
169      this.blockEncoder = new HFileDataBlockEncoderImpl(encoding);
170    } else {
171      this.blockEncoder = NoOpDataBlockEncoder.INSTANCE;
172    }
173    this.comparator = comparator != null? comparator: CellComparator.COMPARATOR;
174
175    closeOutputStream = path != null;
176    this.cacheConf = cacheConf;
177    finishInit(conf);
178    if (LOG.isTraceEnabled()) {
179      LOG.trace("Writer" + (path != null ? " for " + path : "") +
180        " initialized with cacheConf: " + cacheConf +
181        " comparator: " + comparator.getClass().getSimpleName() +
182        " fileContext: " + fileContext);
183    }
184  }

217  /**
218   * Checks that the given Cell's key does not violate the key order.
219   *
220   * @param cell Cell whose key to check.
221   * @return true if the key is duplicate
222   * @throws IOException if the key or the key order is wrong
223   */
224  protected boolean checkKey(final Cell cell) throws IOException {
225    boolean isDuplicateKey = false;
226
227    if (cell == null) {
228      throw new IOException("Key cannot be null or empty");
229    }
230    if (lastCell != null) {
231      int keyComp = comparator.compareKeyIgnoresMvcc(lastCell, cell);
232
233      if (keyComp > 0) {
234        throw new IOException("Added a key not lexically larger than"
235            + " previous. Current cell = " + cell + ", lastCell = " + lastCell);
236      } else if (keyComp == 0) {
237        isDuplicateKey = true;
238      }
239    }
240    return isDuplicateKey;
241  }

243  /** Checks the given value for validity. */
244  protected void checkValue(final byte[] value, final int offset,
245      final int length) throws IOException {
246    if (value == null) {
247      throw new IOException("Value cannot be null");
248    }
249  }

279  /** Additional initialization steps */
280  protected void finishInit(final Configuration conf) {
281    if (blockWriter != null) {
282      throw new IllegalStateException("finishInit called twice");
283    }
284
285    blockWriter = new HFileBlock.Writer(blockEncoder, hFileContext);
286
287    // Data block index writer
288    boolean cacheIndexesOnWrite = cacheConf.shouldCacheIndexesOnWrite();
289    dataBlockIndexWriter = new HFileBlockIndex.BlockIndexWriter(blockWriter,
290        cacheIndexesOnWrite ? cacheConf : null,
291        cacheIndexesOnWrite ? name : null);
292    dataBlockIndexWriter.setMaxChunkSize(
293        HFileBlockIndex.getMaxChunkSize(conf));
294    dataBlockIndexWriter.setMinIndexNumEntries(
295        HFileBlockIndex.getMinIndexNumEntries(conf));
296    inlineBlockWriters.add(dataBlockIndexWriter);
297
298    // Meta data block index writer
299    metaBlockIndexWriter = new HFileBlockIndex.BlockIndexWriter();
300    if (LOG.isTraceEnabled()) LOG.trace("Initialized with " + cacheConf);
301  }

304   * At a block boundary, write all the inline blocks and opens new block.
305   *
306   * @throws IOException
307   */
308  protected void checkBlockBoundary() throws IOException {
309    if (blockWriter.blockSizeWritten() < hFileContext.getBlocksize()) return;
310    finishBlock();
311    writeInlineBlocks(false);
312    newBlock();
313  }

315  /** Clean up the data block that is currently being written.*/
316  private void finishBlock() throws IOException {
317    if (!blockWriter.isWriting() || blockWriter.blockSizeWritten() == 0) return;
318
319    // Update the first data block offset if UNSET; used scanning.
320    if (firstDataBlockOffset == UNSET) {
321      firstDataBlockOffset = outputStream.getPos();
322    }
323    // Update the last data block offset each time through here.
324    lastDataBlockOffset = outputStream.getPos();
325    blockWriter.writeHeaderAndData(outputStream);
326    int onDiskSize = blockWriter.getOnDiskSizeWithHeader();
327    Cell indexEntry =
328      getMidpoint(this.comparator, lastCellOfPreviousBlock, firstCellInBlock);
329    dataBlockIndexWriter.addEntry(CellUtil.getCellKeySerializedAsKeyValueKey(indexEntry),
330      lastDataBlockOffset, onDiskSize);
331    totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
332    if (cacheConf.shouldCacheDataOnWrite()) {
333      doCacheOnWrite(lastDataBlockOffset);
334    }
335  }

337  /**
338   * Try to return a Cell that falls between <code>left</code> and
339   * <code>right</code> but that is shorter; i.e. takes up less space. This
340   * trick is used building HFile block index. Its an optimization. It does not
341   * always work. In this case we'll just return the <code>right</code> cell.
342   *
343   * @param comparator
344   *          Comparator to use.
345   * @param left
346   * @param right
347   * @return A cell that sorts between <code>left</code> and <code>right</code>.
348   */
349  public static Cell getMidpoint(final CellComparator comparator, final Cell left,
350      final Cell right) {
351    // TODO: Redo so only a single pass over the arrays rather than one to
352    // compare and then a second composing midpoint.
353    if (right == null) {
354      throw new IllegalArgumentException("right cell can not be null");
355    }
356    if (left == null) {
357      return right;
358    }
359    // If Cells from meta table, don't mess around. meta table Cells have schema
360    // (table,startrow,hash) so can't be treated as plain byte arrays. Just skip
361    // out without trying to do this optimization.
362    if (comparator instanceof MetaCellComparator) {
363      return right;
364    }
365    int diff = comparator.compareRows(left, right);
366    if (diff > 0) {
367      throw new IllegalArgumentException("Left row sorts after right row; left="
368          + CellUtil.getCellKeyAsString(left) + ", right=" + CellUtil.getCellKeyAsString(right));
369    }
370    byte[] midRow;
371    boolean bufferBacked = left instanceof ByteBufferCell && right instanceof ByteBufferCell;
372    if (diff < 0) {
373      // Left row is < right row.
374      if (bufferBacked) {
375        midRow = getMinimumMidpointArray(((ByteBufferCell) left).getRowByteBuffer(),
376            ((ByteBufferCell) left).getRowPosition(), left.getRowLength(),
377            ((ByteBufferCell) right).getRowByteBuffer(),
378            ((ByteBufferCell) right).getRowPosition(), right.getRowLength());
379      } else {
380        midRow = getMinimumMidpointArray(left.getRowArray(), left.getRowOffset(),
381            left.getRowLength(), right.getRowArray(), right.getRowOffset(), right.getRowLength());
382      }
383      // If midRow is null, just return 'right'. Can't do optimization.
384      if (midRow == null) return right;
385      return CellUtil.createFirstOnRow(midRow);
386    }
387    // Rows are same. Compare on families.
388    diff = CellComparator.compareFamilies(left, right);
389    if (diff > 0) {
390      throw new IllegalArgumentException("Left family sorts after right family; left="
391          + CellUtil.getCellKeyAsString(left) + ", right=" + CellUtil.getCellKeyAsString(right));
392    }
393    if (diff < 0) {
394      if (bufferBacked) {
395        midRow = getMinimumMidpointArray(((ByteBufferCell) left).getFamilyByteBuffer(),
396            ((ByteBufferCell) left).getFamilyPosition(), left.getFamilyLength(),
397            ((ByteBufferCell) right).getFamilyByteBuffer(),
398            ((ByteBufferCell) right).getFamilyPosition(), right.getFamilyLength());
399      } else {
400        midRow = getMinimumMidpointArray(left.getFamilyArray(), left.getFamilyOffset(),
401            left.getFamilyLength(), right.getFamilyArray(), right.getFamilyOffset(),
402            right.getFamilyLength());
403      }
404      // If midRow is null, just return 'right'. Can't do optimization.
405      if (midRow == null) return right;
406      // Return new Cell where we use right row and then a mid sort family.
407      return CellUtil.createFirstOnRowFamily(right, midRow, 0, midRow.length);
408    }
409    // Families are same. Compare on qualifiers.
410    diff = CellComparator.compareQualifiers(left, right);
411    if (diff > 0) {
412      throw new IllegalArgumentException("Left qualifier sorts after right qualifier; left="
413          + CellUtil.getCellKeyAsString(left) + ", right=" + CellUtil.getCellKeyAsString(right));
414    }
415    if (diff < 0) {
416      if (bufferBacked) {
417        midRow = getMinimumMidpointArray(((ByteBufferCell) left).getQualifierByteBuffer(),
418            ((ByteBufferCell) left).getQualifierPosition(), left.getQualifierLength(),
419            ((ByteBufferCell) right).getQualifierByteBuffer(),
420            ((ByteBufferCell) right).getQualifierPosition(), right.getQualifierLength());
421      } else {
422        midRow = getMinimumMidpointArray(left.getQualifierArray(), left.getQualifierOffset(),
423            left.getQualifierLength(), right.getQualifierArray(), right.getQualifierOffset(),
424            right.getQualifierLength());
425      }
426      // If midRow is null, just return 'right'. Can't do optimization.
427      if (midRow == null) return right;
428      // Return new Cell where we use right row and family and then a mid sort qualifier.
429      return CellUtil.createFirstOnRowCol(right, midRow, 0, midRow.length);
430    }
431    // No opportunity for optimization. Just return right key.
432    return right;
433  }

435  /**
436   * @param leftArray
437   * @param leftOffset
438   * @param leftLength
439   * @param rightArray
440   * @param rightOffset
441   * @param rightLength
442   * @return Return a new array that is between left and right and minimally
443   *         sized else just return null as indicator that we could not create a
444   *         mid point.
445   */
446  private static byte[] getMinimumMidpointArray(final byte[] leftArray, final int leftOffset,
447      final int leftLength, final byte[] rightArray, final int rightOffset, final int rightLength) {
448    // rows are different
449    int minLength = leftLength < rightLength ? leftLength : rightLength;
450    int diffIdx = 0;
451    while (diffIdx < minLength
452        && leftArray[leftOffset + diffIdx] == rightArray[rightOffset + diffIdx]) {
453      diffIdx++;
454    }
455    byte[] minimumMidpointArray = null;
456    if (diffIdx >= minLength) {
457      // leftKey's row is prefix of rightKey's.
458      minimumMidpointArray = new byte[diffIdx + 1];
459      System.arraycopy(rightArray, rightOffset, minimumMidpointArray, 0, diffIdx + 1);
460    } else {
461      int diffByte = leftArray[leftOffset + diffIdx];
462      if ((0xff & diffByte) < 0xff && (diffByte + 1) < (rightArray[rightOffset + diffIdx] & 0xff)) {
463        minimumMidpointArray = new byte[diffIdx + 1];
464        System.arraycopy(leftArray, leftOffset, minimumMidpointArray, 0, diffIdx);
465        minimumMidpointArray[diffIdx] = (byte) (diffByte + 1);
466      } else {
467        minimumMidpointArray = new byte[diffIdx + 1];
468        System.arraycopy(rightArray, rightOffset, minimumMidpointArray, 0, diffIdx + 1);
469      }
470    }
471    return minimumMidpointArray;
472  }
473
474  private static byte[] getMinimumMidpointArray(ByteBuffer left, int leftOffset, int leftLength,
475      ByteBuffer right, int rightOffset, int rightLength) {
476    // rows are different
477    int minLength = leftLength < rightLength ? leftLength : rightLength;
478    int diffIdx = 0;
479    while (diffIdx < minLength && ByteBufferUtils.toByte(left,
480        leftOffset + diffIdx) == ByteBufferUtils.toByte(right, rightOffset + diffIdx)) {
481      diffIdx++;
482    }
483    byte[] minMidpoint = null;
484    if (diffIdx >= minLength) {
485      // leftKey's row is prefix of rightKey's.
486      minMidpoint = new byte[diffIdx + 1];
487      ByteBufferUtils.copyFromBufferToArray(minMidpoint, right, rightOffset, 0, diffIdx + 1);
488    } else {
489      int diffByte = ByteBufferUtils.toByte(left, leftOffset + diffIdx);
490      if ((0xff & diffByte) < 0xff
491          && (diffByte + 1) < (ByteBufferUtils.toByte(right, rightOffset + diffIdx) & 0xff)) {
492        minMidpoint = new byte[diffIdx + 1];
493        ByteBufferUtils.copyFromBufferToArray(minMidpoint, left, leftOffset, 0, diffIdx);
494        minMidpoint[diffIdx] = (byte) (diffByte + 1);
495      } else {
496        minMidpoint = new byte[diffIdx + 1];
497        ByteBufferUtils.copyFromBufferToArray(minMidpoint, right, rightOffset, 0, diffIdx + 1);
498      }
499    }
500    return minMidpoint;
501  }

503  /** Gives inline block writers an opportunity to contribute blocks. */
504  private void writeInlineBlocks(boolean closing) throws IOException {
505    for (InlineBlockWriter ibw : inlineBlockWriters) {
506      while (ibw.shouldWriteBlock(closing)) {
507        long offset = outputStream.getPos();
508        boolean cacheThisBlock = ibw.getCacheOnWrite();
509        ibw.writeInlineBlock(blockWriter.startWriting(
510            ibw.getInlineBlockType()));
511        blockWriter.writeHeaderAndData(outputStream);
512        ibw.blockWritten(offset, blockWriter.getOnDiskSizeWithHeader(),
513            blockWriter.getUncompressedSizeWithoutHeader());
514        totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
515
516        if (cacheThisBlock) {
517          doCacheOnWrite(offset);
518        }
519      }
520    }
521  }

523  /**
524   * Caches the last written HFile block.
525   * @param offset the offset of the block we want to cache. Used to determine
526   *          the cache key.
527   */
528  private void doCacheOnWrite(long offset) {
529    HFileBlock cacheFormatBlock = blockWriter.getBlockForCaching(cacheConf);
530    cacheConf.getBlockCache().cacheBlock(
531        new BlockCacheKey(name, offset, true, cacheFormatBlock.getBlockType()),
532        cacheFormatBlock);
533  }

549  /**
550   * Add a meta block to the end of the file. Call before close(). Metadata
551   * blocks are expensive. Fill one with a bunch of serialized data rather than
552   * do a metadata block per metadata instance. If metadata is small, consider
553   * adding to file info using {@link #appendFileInfo(byte[], byte[])}
554   *
555   * @param metaBlockName
556   *          name of the block
557   * @param content
558   *          will call readFields to get data later (DO NOT REUSE)
559   */
560  @Override
561  public void appendMetaBlock(String metaBlockName, Writable content) {
562    byte[] key = Bytes.toBytes(metaBlockName);
563    int i;
564    for (i = 0; i < metaNames.size(); ++i) {
565      // stop when the current key is greater than our own
566      byte[] cur = metaNames.get(i);
567      if (Bytes.BYTES_RAWCOMPARATOR.compare(cur, 0, cur.length, key, 0,
568          key.length) > 0) {
569        break;
570      }
571    }
572    metaNames.add(i, key);
573    metaData.add(i, content);
574  }

576  @Override
577  public void close() throws IOException {
578    if (outputStream == null) {
579      return;
580    }
581    // Save data block encoder metadata in the file info.
582    blockEncoder.saveMetadata(this);
583    // Write out the end of the data blocks, then write meta data blocks.
584    // followed by fileinfo, data block index and meta block index.
585
586    finishBlock();
587    writeInlineBlocks(true);
588
589    FixedFileTrailer trailer = new FixedFileTrailer(getMajorVersion(), getMinorVersion());
590
591    // Write out the metadata blocks if any.
592    if (!metaNames.isEmpty()) {
593      for (int i = 0; i < metaNames.size(); ++i) {
594        // store the beginning offset
595        long offset = outputStream.getPos();
596        // write the metadata content
597        DataOutputStream dos = blockWriter.startWriting(BlockType.META);
598        metaData.get(i).write(dos);
599
600        blockWriter.writeHeaderAndData(outputStream);
601        totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
602
603        // Add the new meta block to the meta index.
604        metaBlockIndexWriter.addEntry(metaNames.get(i), offset,
605            blockWriter.getOnDiskSizeWithHeader());
606      }
607    }
608
609    // Load-on-open section.
610
611    // Data block index.
612    //
613    // In version 2, this section of the file starts with the root level data
614    // block index. We call a function that writes intermediate-level blocks
615    // first, then root level, and returns the offset of the root level block
616    // index.
617
618    long rootIndexOffset = dataBlockIndexWriter.writeIndexBlocks(outputStream);
619    trailer.setLoadOnOpenOffset(rootIndexOffset);
620
621    // Meta block index.
622    metaBlockIndexWriter.writeSingleLevelIndex(blockWriter.startWriting(
623        BlockType.ROOT_INDEX), "meta");
624    blockWriter.writeHeaderAndData(outputStream);
625    totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
626
627    if (this.hFileContext.isIncludesMvcc()) {
628      appendFileInfo(MAX_MEMSTORE_TS_KEY, Bytes.toBytes(maxMemstoreTS));
629      appendFileInfo(KEY_VALUE_VERSION, Bytes.toBytes(KEY_VALUE_VER_WITH_MEMSTORE));
630    }
631
632    // File info
633    writeFileInfo(trailer, blockWriter.startWriting(BlockType.FILE_INFO));
634    blockWriter.writeHeaderAndData(outputStream);
635    totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
636
637    // Load-on-open data supplied by higher levels, e.g. Bloom filters.
638    for (BlockWritable w : additionalLoadOnOpenData){
639      blockWriter.writeBlock(w, outputStream);
640      totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
641    }
642
643    // Now finish off the trailer.
644    trailer.setNumDataIndexLevels(dataBlockIndexWriter.getNumLevels());
645    trailer.setUncompressedDataIndexSize(
646        dataBlockIndexWriter.getTotalUncompressedSize());
647    trailer.setFirstDataBlockOffset(firstDataBlockOffset);
648    trailer.setLastDataBlockOffset(lastDataBlockOffset);
649    trailer.setComparatorClass(comparator.getClass());
650    trailer.setDataIndexCount(dataBlockIndexWriter.getNumRootEntries());
651
652
653    finishClose(trailer);
654
655    blockWriter.release();
656  }

673  private void addBloomFilter(final BloomFilterWriter bfw,
674      final BlockType blockType) {
675    if (bfw.getKeyCount() <= 0)
676      return;
677
678    if (blockType != BlockType.GENERAL_BLOOM_META &&
679        blockType != BlockType.DELETE_FAMILY_BLOOM_META) {
680      throw new RuntimeException("Block Type: " + blockType.toString() +
681          "is not supported");
682    }
683    additionalLoadOnOpenData.add(new BlockWritable() {
684      @Override
685      public BlockType getBlockType() {
686        return blockType;
687      }
688
689      @Override
690      public void writeToBlock(DataOutput out) throws IOException {
691        bfw.getMetaWriter().write(out);
692        Writable dataWriter = bfw.getDataWriter();
693        if (dataWriter != null)
694          dataWriter.write(out);
695      }
696    });
697  }

704  /**
705   * Add key/value to file. Keys must be added in an order that agrees with the
706   * Comparator passed on construction.
707   *
708   * @param cell
709   *          Cell to add. Cannot be empty nor null.
710   * @throws IOException
711   */
712  @Override
713  public void append(final Cell cell) throws IOException {
714    // checkKey uses comparator to check we are writing in order.
715    boolean dupKey = checkKey(cell);
716    if (!dupKey) {
717      checkBlockBoundary();
718    }
719
720    if (!blockWriter.isWriting()) {
721      newBlock();
722    }
723
724    blockWriter.write(cell);
725
726    totalKeyLength += CellUtil.estimatedSerializedSizeOfKey(cell);
727    totalValueLength += cell.getValueLength();
728
729    // Are we the first key in this block?
730    if (firstCellInBlock == null) {
731      // If cell is big, block will be closed and this firstCellInBlock reference will only last
732      // a short while.
733      firstCellInBlock = cell;
734    }
735
736    // TODO: What if cell is 10MB and we write infrequently? We hold on to cell here indefinitely?
737    lastCell = cell;
738    entryCount++;
739    this.maxMemstoreTS = Math.max(this.maxMemstoreTS, cell.getSequenceId());
740    int tagsLength = cell.getTagsLength();
741    if (tagsLength > this.maxTagsLength) {
742      this.maxTagsLength = tagsLength;
743    }
744  }

746  @Override
747  public void beforeShipped() throws IOException {
748    // Add clone methods for every cell
749    if (this.lastCell != null) {
750      this.lastCell = KeyValueUtil.toNewKeyCell(this.lastCell);
751    }
752    if (this.firstCellInBlock != null) {
753      this.firstCellInBlock = KeyValueUtil.toNewKeyCell(this.firstCellInBlock);
754    }
755    if (this.lastCellOfPreviousBlock != null) {
756      this.lastCellOfPreviousBlock = KeyValueUtil.toNewKeyCell(this.lastCellOfPreviousBlock);
757    }
758  }

760  protected void finishFileInfo() throws IOException {
761    if (lastCell != null) {
762      // Make a copy. The copy is stuffed into our fileinfo map. Needs a clean
763      // byte buffer. Won't take a tuple.
764      byte [] lastKey = CellUtil.getCellKeySerializedAsKeyValueKey(this.lastCell);
765      fileInfo.append(FileInfo.LASTKEY, lastKey, false);
766    }
767
768    // Average key length.
769    int avgKeyLen =
770        entryCount == 0 ? 0 : (int) (totalKeyLength / entryCount);
771    fileInfo.append(FileInfo.AVG_KEY_LEN, Bytes.toBytes(avgKeyLen), false);
772    fileInfo.append(FileInfo.CREATE_TIME_TS, Bytes.toBytes(hFileContext.getFileCreateTime()),
773      false);
774
775    // Average value length.
776    int avgValueLen =
777        entryCount == 0 ? 0 : (int) (totalValueLength / entryCount);
778    fileInfo.append(FileInfo.AVG_VALUE_LEN, Bytes.toBytes(avgValueLen), false);
779    if (hFileContext.getDataBlockEncoding() == DataBlockEncoding.PREFIX_TREE) {
780      // In case of Prefix Tree encoding, we always write tags information into HFiles even if all
781      // KVs are having no tags.
782      fileInfo.append(FileInfo.MAX_TAGS_LEN, Bytes.toBytes(this.maxTagsLength), false);
783    } else if (hFileContext.isIncludesTags()) {
784      // When tags are not being written in this file, MAX_TAGS_LEN is excluded
785      // from the FileInfo
786      fileInfo.append(FileInfo.MAX_TAGS_LEN, Bytes.toBytes(this.maxTagsLength), false);
787      boolean tagsCompressed = (hFileContext.getDataBlockEncoding() != DataBlockEncoding.NONE)
788        && hFileContext.isCompressTags();
789      fileInfo.append(FileInfo.TAGS_COMPRESSED, Bytes.toBytes(tagsCompressed), false);
790    }
791  }

801  protected void finishClose(FixedFileTrailer trailer) throws IOException {
802    // Write out encryption metadata before finalizing if we have a valid crypto context
803    Encryption.Context cryptoContext = hFileContext.getEncryptionContext();
804    if (cryptoContext != Encryption.Context.NONE) {
805      // Wrap the context's key and write it as the encryption metadata, the wrapper includes
806      // all information needed for decryption
807      trailer.setEncryptionKey(EncryptionUtil.wrapKey(cryptoContext.getConf(),
808        cryptoContext.getConf().get(HConstants.CRYPTO_MASTERKEY_NAME_CONF_KEY,
809          User.getCurrent().getShortName()),
810        cryptoContext.getKey()));
811    }
812    // Now we can finish the close
813    trailer.setMetaIndexCount(metaNames.size());
814    trailer.setTotalUncompressedBytes(totalUncompressedBytes+ trailer.getTrailerSize());
815    trailer.setEntryCount(entryCount);
816    trailer.setCompressionCodec(hFileContext.getCompression());
817
818    trailer.serialize(outputStream);
819
820    if (closeOutputStream) {
821      outputStream.close();
822      outputStream = null;
823    }
824  }
