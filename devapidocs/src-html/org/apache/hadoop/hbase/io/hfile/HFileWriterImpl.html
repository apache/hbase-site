161  public HFileWriterImpl(final Configuration conf, CacheConfig cacheConf, Path path,
162    FSDataOutputStream outputStream, HFileContext fileContext) {
163    this.outputStream = outputStream;
164    this.path = path;
165    this.name = path != null ? path.getName() : outputStream.toString();
166    this.hFileContext = fileContext;
167    DataBlockEncoding encoding = hFileContext.getDataBlockEncoding();
168    if (encoding != DataBlockEncoding.NONE) {
169      this.blockEncoder = new HFileDataBlockEncoderImpl(encoding);
170    } else {
171      this.blockEncoder = NoOpDataBlockEncoder.INSTANCE;
172    }
173    closeOutputStream = path != null;
174    this.cacheConf = cacheConf;
175    float encodeBlockSizeRatio = conf.getFloat(UNIFIED_ENCODED_BLOCKSIZE_RATIO, 0f);
176    this.encodedBlockSizeLimit = (int) (hFileContext.getBlocksize() * encodeBlockSizeRatio);
177
178    finishInit(conf);
179    if (LOG.isTraceEnabled()) {
180      LOG.trace("Writer" + (path != null ? " for " + path : "") + " initialized with cacheConf: "
181        + cacheConf + " fileContext: " + fileContext);
182    }
183  }

218  /**
219   * Checks that the given Cell's key does not violate the key order.
220   * @param cell Cell whose key to check.
221   * @return true if the key is duplicate
222   * @throws IOException if the key or the key order is wrong
223   */
224  protected boolean checkKey(final Cell cell) throws IOException {
225    boolean isDuplicateKey = false;
226
227    if (cell == null) {
228      throw new IOException("Key cannot be null or empty");
229    }
230    if (lastCell != null) {
231      int keyComp = PrivateCellUtil.compareKeyIgnoresMvcc(this.hFileContext.getCellComparator(),
232        lastCell, cell);
233      if (keyComp > 0) {
234        String message = getLexicalErrorMessage(cell);
235        throw new IOException(message);
236      } else if (keyComp == 0) {
237        isDuplicateKey = true;
238      }
239    }
240    return isDuplicateKey;
241  }

243  private String getLexicalErrorMessage(Cell cell) {
244    StringBuilder sb = new StringBuilder();
245    sb.append("Added a key not lexically larger than previous. Current cell = ");
246    sb.append(cell);
247    sb.append(", lastCell = ");
248    sb.append(lastCell);
249    // file context includes HFile path and optionally table and CF of file being written
250    sb.append("fileContext=");
251    sb.append(hFileContext);
252    return sb.toString();
253  }

289  /** Additional initialization steps */
290  protected void finishInit(final Configuration conf) {
291    if (blockWriter != null) {
292      throw new IllegalStateException("finishInit called twice");
293    }
294    blockWriter =
295      new HFileBlock.Writer(conf, blockEncoder, hFileContext, cacheConf.getByteBuffAllocator());

298    dataBlockIndexWriter = new HFileBlockIndex.BlockIndexWriter(blockWriter,

305    metaBlockIndexWriter = new HFileBlockIndex.BlockIndexWriter();
306    LOG.trace("Initialized with {}", cacheConf);

309  /**
310   * At a block boundary, write all the inline blocks and opens new block.
311   */
312  protected void checkBlockBoundary() throws IOException {
313    boolean shouldFinishBlock = false;
314    // This means hbase.writer.unified.encoded.blocksize.ratio was set to something different from 0
315    // and we should use the encoding ratio
316    if (encodedBlockSizeLimit > 0) {
317      shouldFinishBlock = blockWriter.encodedBlockSizeWritten() >= encodedBlockSizeLimit;
318    } else {
319      shouldFinishBlock = blockWriter.encodedBlockSizeWritten() >= hFileContext.getBlocksize()
320        || blockWriter.blockSizeWritten() >= hFileContext.getBlocksize();
321    }
322    if (shouldFinishBlock) {
323      finishBlock();
324      writeInlineBlocks(false);
325      newBlock();
326    }
327  }

329  /** Clean up the data block that is currently being written. */
330  private void finishBlock() throws IOException {
331    if (!blockWriter.isWriting() || blockWriter.blockSizeWritten() == 0) {
332      return;
333    }
334
335    // Update the first data block offset if UNSET; used scanning.
336    if (firstDataBlockOffset == UNSET) {
337      firstDataBlockOffset = outputStream.getPos();
338    }
339    // Update the last data block offset each time through here.
340    lastDataBlockOffset = outputStream.getPos();
341    blockWriter.writeHeaderAndData(outputStream);
342    int onDiskSize = blockWriter.getOnDiskSizeWithHeader();
343    Cell indexEntry =
344      getMidpoint(this.hFileContext.getCellComparator(), lastCellOfPreviousBlock, firstCellInBlock);
345    dataBlockIndexWriter.addEntry(PrivateCellUtil.getCellKeySerializedAsKeyValueKey(indexEntry),
346      lastDataBlockOffset, onDiskSize);
347    totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
348    if (cacheConf.shouldCacheDataOnWrite()) {
349      doCacheOnWrite(lastDataBlockOffset);
350    }
351  }

353  /**
354   * Try to return a Cell that falls between <code>left</code> and <code>right</code> but that is
355   * shorter; i.e. takes up less space. This trick is used building HFile block index. Its an
356   * optimization. It does not always work. In this case we'll just return the <code>right</code>
357   * cell.
358   * @return A cell that sorts between <code>left</code> and <code>right</code>.
359   */
360  public static Cell getMidpoint(final CellComparator comparator, final Cell left,
361    final Cell right) {
362    if (right == null) {
363      throw new IllegalArgumentException("right cell can not be null");
364    }
365    if (left == null) {
366      return right;
367    }
368    // If Cells from meta table, don't mess around. meta table Cells have schema
369    // (table,startrow,hash) so can't be treated as plain byte arrays. Just skip
370    // out without trying to do this optimization.
371    if (comparator instanceof MetaCellComparator) {
372      return right;
373    }
374    byte[] midRow;
375    boolean bufferBacked =
376      left instanceof ByteBufferExtendedCell && right instanceof ByteBufferExtendedCell;
377    if (bufferBacked) {
378      midRow = getMinimumMidpointArray(((ByteBufferExtendedCell) left).getRowByteBuffer(),
379        ((ByteBufferExtendedCell) left).getRowPosition(), left.getRowLength(),
380        ((ByteBufferExtendedCell) right).getRowByteBuffer(),
381        ((ByteBufferExtendedCell) right).getRowPosition(), right.getRowLength());
382    } else {
383      midRow = getMinimumMidpointArray(left.getRowArray(), left.getRowOffset(), left.getRowLength(),
384        right.getRowArray(), right.getRowOffset(), right.getRowLength());
385    }
386    if (midRow != null) {
387      return PrivateCellUtil.createFirstOnRow(midRow);
388    }
389    // Rows are same. Compare on families.
390    if (bufferBacked) {
391      midRow = getMinimumMidpointArray(((ByteBufferExtendedCell) left).getFamilyByteBuffer(),
392        ((ByteBufferExtendedCell) left).getFamilyPosition(), left.getFamilyLength(),
393        ((ByteBufferExtendedCell) right).getFamilyByteBuffer(),
394        ((ByteBufferExtendedCell) right).getFamilyPosition(), right.getFamilyLength());
395    } else {
396      midRow = getMinimumMidpointArray(left.getFamilyArray(), left.getFamilyOffset(),
397        left.getFamilyLength(), right.getFamilyArray(), right.getFamilyOffset(),
398        right.getFamilyLength());
399    }
400    if (midRow != null) {
401      return PrivateCellUtil.createFirstOnRowFamily(right, midRow, 0, midRow.length);
402    }
403    // Families are same. Compare on qualifiers.
404    if (bufferBacked) {
405      midRow = getMinimumMidpointArray(((ByteBufferExtendedCell) left).getQualifierByteBuffer(),
406        ((ByteBufferExtendedCell) left).getQualifierPosition(), left.getQualifierLength(),
407        ((ByteBufferExtendedCell) right).getQualifierByteBuffer(),
408        ((ByteBufferExtendedCell) right).getQualifierPosition(), right.getQualifierLength());
409    } else {
410      midRow = getMinimumMidpointArray(left.getQualifierArray(), left.getQualifierOffset(),
411        left.getQualifierLength(), right.getQualifierArray(), right.getQualifierOffset(),
412        right.getQualifierLength());
413    }
414    if (midRow != null) {
415      return PrivateCellUtil.createFirstOnRowCol(right, midRow, 0, midRow.length);
416    }
417    // No opportunity for optimization. Just return right key.
418    return right;
419  }

421  /**
422   * Try to get a byte array that falls between left and right as short as possible with
423   * lexicographical order;
424   * @return Return a new array that is between left and right and minimally sized else just return
425   *         null if left == right.
426   */
427  private static byte[] getMinimumMidpointArray(final byte[] leftArray, final int leftOffset,
428    final int leftLength, final byte[] rightArray, final int rightOffset, final int rightLength) {
429    int minLength = leftLength < rightLength ? leftLength : rightLength;
430    int diffIdx = 0;
431    for (; diffIdx < minLength; diffIdx++) {
432      byte leftByte = leftArray[leftOffset + diffIdx];
433      byte rightByte = rightArray[rightOffset + diffIdx];
434      if ((leftByte & 0xff) > (rightByte & 0xff)) {
435        throw new IllegalArgumentException("Left byte array sorts after right row; left="
436          + Bytes.toStringBinary(leftArray, leftOffset, leftLength) + ", right="
437          + Bytes.toStringBinary(rightArray, rightOffset, rightLength));
438      } else if (leftByte != rightByte) {
439        break;
440      }
441    }
442    if (diffIdx == minLength) {
443      if (leftLength > rightLength) {
444        // right is prefix of left
445        throw new IllegalArgumentException("Left byte array sorts after right row; left="
446          + Bytes.toStringBinary(leftArray, leftOffset, leftLength) + ", right="
447          + Bytes.toStringBinary(rightArray, rightOffset, rightLength));
448      } else if (leftLength < rightLength) {
449        // left is prefix of right.
450        byte[] minimumMidpointArray = new byte[minLength + 1];
451        System.arraycopy(rightArray, rightOffset, minimumMidpointArray, 0, minLength + 1);
452        minimumMidpointArray[minLength] = 0x00;
453        return minimumMidpointArray;
454      } else {
455        // left == right
456        return null;
457      }
458    }
459    // Note that left[diffIdx] can never be equal to 0xff since left < right
460    byte[] minimumMidpointArray = new byte[diffIdx + 1];
461    System.arraycopy(leftArray, leftOffset, minimumMidpointArray, 0, diffIdx + 1);
462    minimumMidpointArray[diffIdx] = (byte) (minimumMidpointArray[diffIdx] + 1);
463    return minimumMidpointArray;
464  }

466  /**
467   * Try to create a new byte array that falls between left and right as short as possible with
468   * lexicographical order.
469   * @return Return a new array that is between left and right and minimally sized else just return
470   *         null if left == right.
471   */
472  private static byte[] getMinimumMidpointArray(ByteBuffer left, int leftOffset, int leftLength,
473    ByteBuffer right, int rightOffset, int rightLength) {
474    int minLength = leftLength < rightLength ? leftLength : rightLength;
475    int diffIdx = 0;
476    for (; diffIdx < minLength; diffIdx++) {
477      int leftByte = ByteBufferUtils.toByte(left, leftOffset + diffIdx);
478      int rightByte = ByteBufferUtils.toByte(right, rightOffset + diffIdx);
479      if ((leftByte & 0xff) > (rightByte & 0xff)) {
480        throw new IllegalArgumentException("Left byte array sorts after right row; left="
481          + ByteBufferUtils.toStringBinary(left, leftOffset, leftLength) + ", right="
482          + ByteBufferUtils.toStringBinary(right, rightOffset, rightLength));
483      } else if (leftByte != rightByte) {
484        break;
485      }
486    }
487    if (diffIdx == minLength) {
488      if (leftLength > rightLength) {
489        // right is prefix of left
490        throw new IllegalArgumentException("Left byte array sorts after right row; left="
491          + ByteBufferUtils.toStringBinary(left, leftOffset, leftLength) + ", right="
492          + ByteBufferUtils.toStringBinary(right, rightOffset, rightLength));
493      } else if (leftLength < rightLength) {
494        // left is prefix of right.
495        byte[] minimumMidpointArray = new byte[minLength + 1];
496        ByteBufferUtils.copyFromBufferToArray(minimumMidpointArray, right, rightOffset, 0,
497          minLength + 1);
498        minimumMidpointArray[minLength] = 0x00;
499        return minimumMidpointArray;
500      } else {
501        // left == right
502        return null;
503      }
504    }
505    // Note that left[diffIdx] can never be equal to 0xff since left < right
506    byte[] minimumMidpointArray = new byte[diffIdx + 1];
507    ByteBufferUtils.copyFromBufferToArray(minimumMidpointArray, left, leftOffset, 0, diffIdx + 1);
508    minimumMidpointArray[diffIdx] = (byte) (minimumMidpointArray[diffIdx] + 1);
509    return minimumMidpointArray;
510  }

512  /** Gives inline block writers an opportunity to contribute blocks. */
513  private void writeInlineBlocks(boolean closing) throws IOException {
514    for (InlineBlockWriter ibw : inlineBlockWriters) {
515      while (ibw.shouldWriteBlock(closing)) {
516        long offset = outputStream.getPos();
517        boolean cacheThisBlock = ibw.getCacheOnWrite();
518        ibw.writeInlineBlock(blockWriter.startWriting(ibw.getInlineBlockType()));
519        blockWriter.writeHeaderAndData(outputStream);
520        ibw.blockWritten(offset, blockWriter.getOnDiskSizeWithHeader(),
521          blockWriter.getUncompressedSizeWithoutHeader());
522        totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
523
524        if (cacheThisBlock) {
525          doCacheOnWrite(offset);
526        }
527      }
528    }
529  }

531  /**
532   * Caches the last written HFile block.
533   * @param offset the offset of the block we want to cache. Used to determine the cache key.
534   */
535  private void doCacheOnWrite(long offset) {
536    cacheConf.getBlockCache().ifPresent(cache -> {
537      HFileBlock cacheFormatBlock = blockWriter.getBlockForCaching(cacheConf);
538      try {
539        cache.cacheBlock(new BlockCacheKey(name, offset, true, cacheFormatBlock.getBlockType()),
540          cacheFormatBlock);
541      } finally {
542        // refCnt will auto increase when block add to Cache, see RAMCache#putIfAbsent
543        cacheFormatBlock.release();
544      }
545    });
546  }

560  /**
561   * Add a meta block to the end of the file. Call before close(). Metadata blocks are expensive.
562   * Fill one with a bunch of serialized data rather than do a metadata block per metadata instance.
563   * If metadata is small, consider adding to file info using
564   * {@link #appendFileInfo(byte[], byte[])} n * name of the block n * will call readFields to get
565   * data later (DO NOT REUSE)
566   */
567  @Override
568  public void appendMetaBlock(String metaBlockName, Writable content) {
569    byte[] key = Bytes.toBytes(metaBlockName);
570    int i;
571    for (i = 0; i < metaNames.size(); ++i) {
572      // stop when the current key is greater than our own
573      byte[] cur = metaNames.get(i);
574      if (Bytes.BYTES_RAWCOMPARATOR.compare(cur, 0, cur.length, key, 0, key.length) > 0) {
575        break;
576      }
577    }
578    metaNames.add(i, key);
579    metaData.add(i, content);
580  }

582  @Override
583  public void close() throws IOException {
584    if (outputStream == null) {
585      return;
586    }
587    // Save data block encoder metadata in the file info.
588    blockEncoder.saveMetadata(this);
589    // Write out the end of the data blocks, then write meta data blocks.
590    // followed by fileinfo, data block index and meta block index.
591
592    finishBlock();
593    writeInlineBlocks(true);
594
595    FixedFileTrailer trailer = new FixedFileTrailer(getMajorVersion(), getMinorVersion());
596
597    // Write out the metadata blocks if any.
598    if (!metaNames.isEmpty()) {
599      for (int i = 0; i < metaNames.size(); ++i) {
600        // store the beginning offset
601        long offset = outputStream.getPos();
602        // write the metadata content
603        DataOutputStream dos = blockWriter.startWriting(BlockType.META);
604        metaData.get(i).write(dos);
605
606        blockWriter.writeHeaderAndData(outputStream);
607        totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
608
609        // Add the new meta block to the meta index.
610        metaBlockIndexWriter.addEntry(metaNames.get(i), offset,
611          blockWriter.getOnDiskSizeWithHeader());
612      }
613    }
614
615    // Load-on-open section.
616
617    // Data block index.
618    //
619    // In version 2, this section of the file starts with the root level data
620    // block index. We call a function that writes intermediate-level blocks
621    // first, then root level, and returns the offset of the root level block
622    // index.
623
624    long rootIndexOffset = dataBlockIndexWriter.writeIndexBlocks(outputStream);
625    trailer.setLoadOnOpenOffset(rootIndexOffset);
626
627    // Meta block index.
628    metaBlockIndexWriter.writeSingleLevelIndex(blockWriter.startWriting(BlockType.ROOT_INDEX),
629      "meta");
630    blockWriter.writeHeaderAndData(outputStream);
631    totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
632
633    if (this.hFileContext.isIncludesMvcc()) {
634      appendFileInfo(MAX_MEMSTORE_TS_KEY, Bytes.toBytes(maxMemstoreTS));
635      appendFileInfo(KEY_VALUE_VERSION, Bytes.toBytes(KEY_VALUE_VER_WITH_MEMSTORE));
636    }
637
638    // File info
639    writeFileInfo(trailer, blockWriter.startWriting(BlockType.FILE_INFO));
640    blockWriter.writeHeaderAndData(outputStream);
641    totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
642
643    // Load-on-open data supplied by higher levels, e.g. Bloom filters.
644    for (BlockWritable w : additionalLoadOnOpenData) {
645      blockWriter.writeBlock(w, outputStream);
646      totalUncompressedBytes += blockWriter.getUncompressedSizeWithHeader();
647    }
648
649    // Now finish off the trailer.
650    trailer.setNumDataIndexLevels(dataBlockIndexWriter.getNumLevels());
651    trailer.setUncompressedDataIndexSize(dataBlockIndexWriter.getTotalUncompressedSize());
652    trailer.setFirstDataBlockOffset(firstDataBlockOffset);
653    trailer.setLastDataBlockOffset(lastDataBlockOffset);
654    trailer.setComparatorClass(this.hFileContext.getCellComparator().getClass());
655    trailer.setDataIndexCount(dataBlockIndexWriter.getNumRootEntries());
656
657    finishClose(trailer);
658
659    blockWriter.release();
660  }

677  private void addBloomFilter(final BloomFilterWriter bfw, final BlockType blockType) {
678    if (bfw.getKeyCount() <= 0) {
679      return;
680    }
681
682    if (
683      blockType != BlockType.GENERAL_BLOOM_META && blockType != BlockType.DELETE_FAMILY_BLOOM_META
684    ) {
685      throw new RuntimeException("Block Type: " + blockType.toString() + "is not supported");
686    }
687    additionalLoadOnOpenData.add(new BlockWritable() {
688      @Override
689      public BlockType getBlockType() {
690        return blockType;
691      }
692
693      @Override
694      public void writeToBlock(DataOutput out) throws IOException {
695        bfw.getMetaWriter().write(out);
696        Writable dataWriter = bfw.getDataWriter();
697        if (dataWriter != null) {
698          dataWriter.write(out);
699        }
700      }
701    });
702  }

709  /**
710   * Add key/value to file. Keys must be added in an order that agrees with the Comparator passed on
711   * construction. n * Cell to add. Cannot be empty nor null.
712   */
713  @Override
714  public void append(final Cell cell) throws IOException {
715    // checkKey uses comparator to check we are writing in order.
716    boolean dupKey = checkKey(cell);
717    if (!dupKey) {
718      checkBlockBoundary();
719    }
720
721    if (!blockWriter.isWriting()) {
722      newBlock();
723    }
724
725    blockWriter.write(cell);
726
727    totalKeyLength += PrivateCellUtil.estimatedSerializedSizeOfKey(cell);
728    totalValueLength += cell.getValueLength();
729
730    // Are we the first key in this block?
731    if (firstCellInBlock == null) {
732      // If cell is big, block will be closed and this firstCellInBlock reference will only last
733      // a short while.
734      firstCellInBlock = cell;
735    }
736
737    // TODO: What if cell is 10MB and we write infrequently? We hold on to cell here indefinitely?
738    lastCell = cell;
739    entryCount++;
740    this.maxMemstoreTS = Math.max(this.maxMemstoreTS, cell.getSequenceId());
741    int tagsLength = cell.getTagsLength();
742    if (tagsLength > this.maxTagsLength) {
743      this.maxTagsLength = tagsLength;
744    }
745  }

747  @Override
748  public void beforeShipped() throws IOException {
749    this.blockWriter.beforeShipped();
750    // Add clone methods for every cell
751    if (this.lastCell != null) {
752      this.lastCell = KeyValueUtil.toNewKeyCell(this.lastCell);
753    }
754    if (this.firstCellInBlock != null) {
755      this.firstCellInBlock = KeyValueUtil.toNewKeyCell(this.firstCellInBlock);
756    }
757    if (this.lastCellOfPreviousBlock != null) {
758      this.lastCellOfPreviousBlock = KeyValueUtil.toNewKeyCell(this.lastCellOfPreviousBlock);
759    }
760  }

766  protected void finishFileInfo() throws IOException {
767    if (lastCell != null) {
768      // Make a copy. The copy is stuffed into our fileinfo map. Needs a clean
769      // byte buffer. Won't take a tuple.
770      byte[] lastKey = PrivateCellUtil.getCellKeySerializedAsKeyValueKey(this.lastCell);
771      fileInfo.append(HFileInfo.LASTKEY, lastKey, false);
772    }
773
774    // Average key length.
775    int avgKeyLen = entryCount == 0 ? 0 : (int) (totalKeyLength / entryCount);
776    fileInfo.append(HFileInfo.AVG_KEY_LEN, Bytes.toBytes(avgKeyLen), false);
777    fileInfo.append(HFileInfo.CREATE_TIME_TS, Bytes.toBytes(hFileContext.getFileCreateTime()),
778      false);
779
780    // Average value length.
781    int avgValueLen = entryCount == 0 ? 0 : (int) (totalValueLength / entryCount);
782    fileInfo.append(HFileInfo.AVG_VALUE_LEN, Bytes.toBytes(avgValueLen), false);
783    if (hFileContext.isIncludesTags()) {
784      // When tags are not being written in this file, MAX_TAGS_LEN is excluded
785      // from the FileInfo
786      fileInfo.append(HFileInfo.MAX_TAGS_LEN, Bytes.toBytes(this.maxTagsLength), false);
787      boolean tagsCompressed = (hFileContext.getDataBlockEncoding() != DataBlockEncoding.NONE)
788        && hFileContext.isCompressTags();
789      fileInfo.append(HFileInfo.TAGS_COMPRESSED, Bytes.toBytes(tagsCompressed), false);
790    }
791  }

801  protected void finishClose(FixedFileTrailer trailer) throws IOException {
802    // Write out encryption metadata before finalizing if we have a valid crypto context
803    Encryption.Context cryptoContext = hFileContext.getEncryptionContext();
804    if (cryptoContext != Encryption.Context.NONE) {
805      // Wrap the context's key and write it as the encryption metadata, the wrapper includes
806      // all information needed for decryption
807      trailer.setEncryptionKey(EncryptionUtil.wrapKey(
808        cryptoContext.getConf(), cryptoContext.getConf()
809          .get(HConstants.CRYPTO_MASTERKEY_NAME_CONF_KEY, User.getCurrent().getShortName()),
810        cryptoContext.getKey()));
811    }
812    // Now we can finish the close
813    trailer.setMetaIndexCount(metaNames.size());
814    trailer.setTotalUncompressedBytes(totalUncompressedBytes + trailer.getTrailerSize());
815    trailer.setEntryCount(entryCount);
816    trailer.setCompressionCodec(hFileContext.getCompression());
817
818    long startTime = EnvironmentEdgeManager.currentTime();
819    trailer.serialize(outputStream);
820    HFile.updateWriteLatency(EnvironmentEdgeManager.currentTime() - startTime);
821
822    if (closeOutputStream) {
823      outputStream.close();
824      outputStream = null;
825    }
826  }
