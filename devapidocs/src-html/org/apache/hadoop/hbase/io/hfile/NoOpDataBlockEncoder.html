017package org.apache.hadoop.hbase.io.hfile;
018
019import java.io.DataOutputStream;
020import java.io.IOException;
021
022import org.apache.hadoop.hbase.classification.InterfaceAudience;
023import org.apache.hadoop.hbase.Cell;
024import org.apache.hadoop.hbase.CellUtil;
025import org.apache.hadoop.hbase.KeyValue;
026import org.apache.hadoop.hbase.KeyValueUtil;
027import org.apache.hadoop.hbase.io.encoding.DataBlockEncoding;
028import org.apache.hadoop.hbase.io.encoding.HFileBlockDecodingContext;
029import org.apache.hadoop.hbase.io.encoding.HFileBlockDefaultDecodingContext;
030import org.apache.hadoop.hbase.io.encoding.HFileBlockDefaultEncodingContext;
031import org.apache.hadoop.hbase.io.encoding.HFileBlockEncodingContext;
032import org.apache.hadoop.io.WritableUtils;
033
034/**
035 * Does not perform any kind of encoding/decoding.
036 */
037@InterfaceAudience.Private
038public class NoOpDataBlockEncoder implements HFileDataBlockEncoder {
039
040  public static final NoOpDataBlockEncoder INSTANCE =
041      new NoOpDataBlockEncoder();
042
043  /** Cannot be instantiated. Use {@link #INSTANCE} instead. */
044  private NoOpDataBlockEncoder() {
045  }
046
047  @Override
048  public int encode(Cell cell, HFileBlockEncodingContext encodingCtx, DataOutputStream out)
049      throws IOException {
050    int klength = KeyValueUtil.keyLength(cell);
051    int vlength = cell.getValueLength();
052
053    out.writeInt(klength);
054    out.writeInt(vlength);
055    CellUtil.writeFlatKey(cell, out);
056    CellUtil.writeValue(out, cell, vlength);
057    int encodedKvSize = klength + vlength + KeyValue.KEYVALUE_INFRASTRUCTURE_SIZE;
058    // Write the additional tag into the stream
059    if (encodingCtx.getHFileContext().isIncludesTags()) {
060      int tagsLength = cell.getTagsLength();
061      out.writeShort(tagsLength);
062      if (tagsLength > 0) {
063        CellUtil.writeTags(out, cell, tagsLength);
064      }
065      encodedKvSize += tagsLength + KeyValue.TAGS_LENGTH_SIZE;
066    }
067    if (encodingCtx.getHFileContext().isIncludesMvcc()) {
068      WritableUtils.writeVLong(out, cell.getSequenceId());
069      encodedKvSize += WritableUtils.getVIntSize(cell.getSequenceId());
070    }
071    return encodedKvSize;
072  }
073
074  @Override
075  public boolean useEncodedScanner() {
076    return false;
077  }
078
079  @Override
080  public void saveMetadata(HFile.Writer writer) {
081  }
082
083  @Override
084  public DataBlockEncoding getDataBlockEncoding() {
085    return DataBlockEncoding.NONE;
086  }
087
088  @Override
089  public DataBlockEncoding getEffectiveEncodingInCache(boolean isCompaction) {
090    return DataBlockEncoding.NONE;
091  }
092  
093  @Override
094  public String toString() {
095    return getClass().getSimpleName();
096  }
097
098  @Override
099  public HFileBlockEncodingContext newDataBlockEncodingContext(
100      byte[] dummyHeader, HFileContext meta) {
101    return new HFileBlockDefaultEncodingContext(null, dummyHeader, meta);
102  }
103
104  @Override
105  public HFileBlockDecodingContext newDataBlockDecodingContext(HFileContext meta) {
106    return new HFileBlockDefaultDecodingContext(meta);
107  }
108
109  @Override
110  public void startBlockEncoding(HFileBlockEncodingContext encodingCtx, DataOutputStream out)
111      throws IOException {
112  }
113
114  @Override
115  public void endBlockEncoding(HFileBlockEncodingContext encodingCtx, DataOutputStream out,
116      byte[] uncompressedBytesWithHeader, BlockType blockType) throws IOException {
117    encodingCtx.postEncoding(BlockType.DATA);
118  }
119}
