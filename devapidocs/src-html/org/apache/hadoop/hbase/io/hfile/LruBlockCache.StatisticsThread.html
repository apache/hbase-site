056/**
057 * A block cache implementation that is memory-aware using {@link HeapSize},
058 * memory-bound using an LRU eviction algorithm, and concurrent: backed by a
059 * {@link ConcurrentHashMap} and with a non-blocking eviction thread giving
060 * constant-time {@link #cacheBlock} and {@link #getBlock} operations.<p>
061 *
062 * Contains three levels of block priority to allow for scan-resistance and in-memory families
063 * {@link org.apache.hadoop.hbase.HColumnDescriptor#setInMemory(boolean)} (An in-memory column
064 * family is a column family that should be served from memory if possible):
065 * single-access, multiple-accesses, and in-memory priority.
066 * A block is added with an in-memory priority flag if
067 * {@link org.apache.hadoop.hbase.HColumnDescriptor#isInMemory()}, otherwise a block becomes a
068 * single access priority the first time it is read into this block cache.  If a block is
069 * accessed again while in cache, it is marked as a multiple access priority block.  This
070 * delineation of blocks is used to prevent scans from thrashing the cache adding a
071 * least-frequently-used element to the eviction algorithm.<p>
072 *
073 * Each priority is given its own chunk of the total cache to ensure
074 * fairness during eviction.  Each priority will retain close to its maximum
075 * size, however, if any priority is not using its entire chunk the others
076 * are able to grow beyond their chunk size.<p>
077 *
078 * Instantiated at a minimum with the total size and average block size.
079 * All sizes are in bytes.  The block size is not especially important as this
080 * cache is fully dynamic in its sizing of blocks.  It is only used for
081 * pre-allocating data structures and in initial heap estimation of the map.<p>
082 *
083 * The detailed constructor defines the sizes for the three priorities (they
084 * should total to the <code>maximum size</code> defined).  It also sets the levels that
085 * trigger and control the eviction thread.<p>
086 *
087 * The <code>acceptable size</code> is the cache size level which triggers the eviction
088 * process to start.  It evicts enough blocks to get the size below the
089 * minimum size specified.<p>
090 *
091 * Eviction happens in a separate thread and involves a single full-scan
092 * of the map.  It determines how many bytes must be freed to reach the minimum
093 * size, and then while scanning determines the fewest least-recently-used
094 * blocks necessary from each of the three priorities (would be 3 times bytes
095 * to free).  It then uses the priority chunk sizes to evict fairly according
096 * to the relative sizes and usage.
097 */

299  public LruBlockCache(long maxSize, long blockSize, boolean evictionThread,
300      int mapInitialSize, float mapLoadFactor, int mapConcurrencyLevel,
301      float minFactor, float acceptableFactor, float singleFactor,
302      float multiFactor, float memoryFactor, float hardLimitFactor,

305    if(singleFactor + multiFactor + memoryFactor != 1 ||
306        singleFactor < 0 || multiFactor < 0 || memoryFactor < 0) {

309    }
310    if (minFactor >= acceptableFactor) {
311      throw new IllegalArgumentException("minFactor must be smaller than acceptableFactor");
312    }
313    if (minFactor >= 1.0f || acceptableFactor >= 1.0f) {
314      throw new IllegalArgumentException("all factors must be < 1");
315    }
316    this.maxSize = maxSize;
317    this.blockSize = blockSize;
318    this.forceInMemory = forceInMemory;
319    map = new ConcurrentHashMap<>(mapInitialSize, mapLoadFactor, mapConcurrencyLevel);
320    this.minFactor = minFactor;
321    this.acceptableFactor = acceptableFactor;
322    this.singleFactor = singleFactor;
323    this.multiFactor = multiFactor;
324    this.memoryFactor = memoryFactor;
325    this.stats = new CacheStats(this.getClass().getSimpleName());
326    this.count = new AtomicLong(0);
327    this.elements = new AtomicLong(0);
328    this.dataBlockElements = new LongAdder();
329    this.dataBlockSize = new LongAdder();
330    this.overhead = calculateOverhead(maxSize, blockSize, mapConcurrencyLevel);
331    this.size = new AtomicLong(this.overhead);
332    this.hardCapacityLimitFactor = hardLimitFactor;
333    if (evictionThread) {
334      this.evictionThread = new EvictionThread(this);
335      this.evictionThread.start(); // FindBugs SC_START_IN_CTOR
336    } else {
337      this.evictionThread = null;
338    }
339    // TODO: Add means of turning this off.  Bit obnoxious running thread just to make a log
340    // every five minutes.
341    this.scheduleThreadPool.scheduleAtFixedRate(new StatisticsThread(this), STAT_THREAD_PERIOD,
342                                                STAT_THREAD_PERIOD, TimeUnit.SECONDS);
343  }

355  /**
356   * Cache the block with the specified name and buffer.
357   * <p>
358   * It is assumed this will NOT be called on an already cached block. In rare cases (HBASE-8547)
359   * this can happen, for which we compare the buffer contents.
360   *
361   * @param cacheKey block's cache key
362   * @param buf      block buffer
363   * @param inMemory if block is in-memory
364   */
365  @Override
366  public void cacheBlock(BlockCacheKey cacheKey, Cacheable buf, boolean inMemory) {
367    if (buf.heapSize() > maxBlockSize) {
368      // If there are a lot of blocks that are too
369      // big this can make the logs way too noisy.
370      // So we log 2%
371      if (stats.failInsert() % 50 == 0) {
372        LOG.warn("Trying to cache too large a block "
373            + cacheKey.getHfileName() + " @ "
374            + cacheKey.getOffset()
375            + " is " + buf.heapSize()
376            + " which is larger than " + maxBlockSize);
377      }
378      return;
379    }
380
381    LruCachedBlock cb = map.get(cacheKey);
382    if (cb != null) {
383      int comparison = BlockCacheUtil.validateBlockAddition(cb.getBuffer(), buf, cacheKey);
384      if (comparison != 0) {
385        if (comparison < 0) {
386          LOG.warn("Cached block contents differ by nextBlockOnDiskSize. Keeping cached block.");
387          return;
388        } else {
389          LOG.warn("Cached block contents differ by nextBlockOnDiskSize. Caching new block.");
390        }
391      } else {
392        String msg = "Cached an already cached block: " + cacheKey + " cb:" + cb.getCacheKey();
393        msg += ". This is harmless and can happen in rare cases (see HBASE-8547)";
394        LOG.warn(msg);
395        return;
396      }
397    }
398    long currentSize = size.get();
399    long currentAcceptableSize = acceptableSize();
400    long hardLimitSize = (long) (hardCapacityLimitFactor * currentAcceptableSize);
401    if (currentSize >= hardLimitSize) {
402      stats.failInsert();
403      if (LOG.isTraceEnabled()) {
404        LOG.trace("LruBlockCache current size " + StringUtils.byteDesc(currentSize)
405          + " has exceeded acceptable size " + StringUtils.byteDesc(currentAcceptableSize) + "."
406          + " The hard limit size is " + StringUtils.byteDesc(hardLimitSize)
407          + ", failed to put cacheKey:" + cacheKey + " into LruBlockCache.");
408      }
409      if (!evictionInProgress) {
410        runEviction();
411      }
412      return;
413    }
414    cb = new LruCachedBlock(cacheKey, buf, count.incrementAndGet(), inMemory);
415    long newSize = updateSizeMetrics(cb, false);
416    map.put(cacheKey, cb);
417    long val = elements.incrementAndGet();
418    if (buf.getBlockType().isData()) {
419       dataBlockElements.increment();
420    }
421    if (LOG.isTraceEnabled()) {
422      long size = map.size();
423      assertCounterSanity(size, val);
424    }
425    if (newSize > currentAcceptableSize && !evictionInProgress) {
426      runEviction();
427    }
428  }

430  /**
431   * Sanity-checking for parity between actual block cache content and metrics.
432   * Intended only for use with TRACE level logging and -ea JVM.
433   */
434  private static void assertCounterSanity(long mapSize, long counterVal) {
435    if (counterVal < 0) {
436      LOG.trace("counterVal overflow. Assertions unreliable. counterVal=" + counterVal +
437        ", mapSize=" + mapSize);
438      return;
439    }
440    if (mapSize < Integer.MAX_VALUE) {
441      double pct_diff = Math.abs((((double) counterVal) / ((double) mapSize)) - 1.);
442      if (pct_diff > 0.05) {
443        LOG.trace("delta between reported and actual size > 5%. counterVal=" + counterVal +
444          ", mapSize=" + mapSize);
445      }
446    }
447  }

461  /**
462   * Helper function that updates the local size counter and also updates any
463   * per-cf or per-blocktype metrics it can discern from given
464   * {@link LruCachedBlock}
465   */
466  private long updateSizeMetrics(LruCachedBlock cb, boolean evict) {
467    long heapsize = cb.heapSize();
468    BlockType bt = cb.getBuffer().getBlockType();
469    if (evict) {
470      heapsize *= -1;
471    }
472    if (bt != null && bt.isData()) {
473       dataBlockSize.add(heapsize);
474    }
475    return size.addAndGet(heapsize);
476  }

478  /**
479   * Get the buffer of the block with the specified name.
480   *
481   * @param cacheKey           block's cache key
482   * @param caching            true if the caller caches blocks on cache misses
483   * @param repeat             Whether this is a repeat lookup for the same block
484   *                           (used to avoid double counting cache misses when doing double-check
485   *                           locking)
486   * @param updateCacheMetrics Whether to update cache metrics or not
487   *
488   * @return buffer of specified cache key, or null if not in cache
489   */
490  @Override
491  public Cacheable getBlock(BlockCacheKey cacheKey, boolean caching, boolean repeat,
492      boolean updateCacheMetrics) {
493    LruCachedBlock cb = map.get(cacheKey);
494    if (cb == null) {
495      if (!repeat && updateCacheMetrics) {
496        stats.miss(caching, cacheKey.isPrimary(), cacheKey.getBlockType());
497      }
498      // If there is another block cache then try and read there.
499      // However if this is a retry ( second time in double checked locking )
500      // And it's already a miss then the l2 will also be a miss.
501      if (victimHandler != null && !repeat) {
502        Cacheable result = victimHandler.getBlock(cacheKey, caching, repeat, updateCacheMetrics);
503
504        // Promote this to L1.
505        if (result != null && caching) {
506          if (result instanceof HFileBlock && ((HFileBlock) result).usesSharedMemory()) {
507            result = ((HFileBlock) result).deepClone();
508          }
509          cacheBlock(cacheKey, result, /* inMemory = */ false);
510        }
511        return result;
512      }
513      return null;
514    }
515    if (updateCacheMetrics) stats.hit(caching, cacheKey.isPrimary(), cacheKey.getBlockType());
516    cb.access(count.incrementAndGet());
517    return cb.getBuffer();
518  }

535  /**
536   * Evicts all blocks for a specific HFile. This is an
537   * expensive operation implemented as a linear-time search through all blocks
538   * in the cache. Ideally this should be a search in a log-access-time map.
539   *
540   * <p>
541   * This is used for evict-on-close to remove all blocks of a specific HFile.
542   *
543   * @return the number of blocks evicted
544   */
545  @Override
546  public int evictBlocksByHfileName(String hfileName) {
547    int numEvicted = 0;
548    for (BlockCacheKey key : map.keySet()) {
549      if (key.getHfileName().equals(hfileName)) {
550        if (evictBlock(key))
551          ++numEvicted;
552      }
553    }
554    if (victimHandler != null) {
555      numEvicted += victimHandler.evictBlocksByHfileName(hfileName);
556    }
557    return numEvicted;
558  }

560  /**
561   * Evict the block, and it will be cached by the victim handler if exists &amp;&amp;
562   * block may be read again later
563   *
564   * @param evictedByEvictionProcess true if the given block is evicted by
565   *          EvictionThread
566   * @return the heap size of evicted block
567   */
568  protected long evictBlock(LruCachedBlock block, boolean evictedByEvictionProcess) {
569    boolean found = map.remove(block.getCacheKey()) != null;
570    if (!found) {
571      return 0;
572    }
573    updateSizeMetrics(block, true);
574    long val = elements.decrementAndGet();
575    if (LOG.isTraceEnabled()) {
576      long size = map.size();
577      assertCounterSanity(size, val);
578    }
579    if (block.getBuffer().getBlockType().isData()) {
580       dataBlockElements.decrement();
581    }
582    if (evictedByEvictionProcess) {
583      // When the eviction of the block happened because of invalidation of HFiles, no need to
584      // update the stats counter.
585      stats.evicted(block.getCachedTime(), block.getCacheKey().isPrimary());
586      if (victimHandler != null) {
587        victimHandler.cacheBlock(block.getCacheKey(), block.getBuffer());
588      }
589    }
590    return block.heapSize();
591  }

593  /**
594   * Multi-threaded call to run the eviction process.
595   */
596  private void runEviction() {
597    if (evictionThread == null) {
598      evict();
599    } else {
600      evictionThread.evict();
601    }
602  }

614  /**
615   * Eviction method.
616   */
617  void evict() {
618
619    // Ensure only one eviction at a time
620    if(!evictionLock.tryLock()) return;
621
622    try {
623      evictionInProgress = true;
624      long currentSize = this.size.get();
625      long bytesToFree = currentSize - minSize();
626
627      if (LOG.isTraceEnabled()) {
628        LOG.trace("Block cache LRU eviction started; Attempting to free " +
629          StringUtils.byteDesc(bytesToFree) + " of total=" +
630          StringUtils.byteDesc(currentSize));
631      }
632
633      if (bytesToFree <= 0) return;
634
635      // Instantiate priority buckets
636      BlockBucket bucketSingle = new BlockBucket("single", bytesToFree, blockSize, singleSize());
637      BlockBucket bucketMulti = new BlockBucket("multi", bytesToFree, blockSize, multiSize());
638      BlockBucket bucketMemory = new BlockBucket("memory", bytesToFree, blockSize, memorySize());
639
640      // Scan entire map putting into appropriate buckets
641      for (LruCachedBlock cachedBlock : map.values()) {
642        switch (cachedBlock.getPriority()) {
643          case SINGLE: {
644            bucketSingle.add(cachedBlock);
645            break;
646          }
647          case MULTI: {
648            bucketMulti.add(cachedBlock);
649            break;
650          }
651          case MEMORY: {
652            bucketMemory.add(cachedBlock);
653            break;
654          }
655        }
656      }
657
658      long bytesFreed = 0;
659      if (forceInMemory || memoryFactor > 0.999f) {
660        long s = bucketSingle.totalSize();
661        long m = bucketMulti.totalSize();
662        if (bytesToFree > (s + m)) {
663          // this means we need to evict blocks in memory bucket to make room,
664          // so the single and multi buckets will be emptied
665          bytesFreed = bucketSingle.free(s);
666          bytesFreed += bucketMulti.free(m);
667          if (LOG.isTraceEnabled()) {
668            LOG.trace("freed " + StringUtils.byteDesc(bytesFreed) +
669              " from single and multi buckets");
670          }
671          bytesFreed += bucketMemory.free(bytesToFree - bytesFreed);
672          if (LOG.isTraceEnabled()) {
673            LOG.trace("freed " + StringUtils.byteDesc(bytesFreed) +
674              " total from all three buckets ");
675          }
676        } else {
677          // this means no need to evict block in memory bucket,
678          // and we try best to make the ratio between single-bucket and
679          // multi-bucket is 1:2
680          long bytesRemain = s + m - bytesToFree;
681          if (3 * s <= bytesRemain) {
682            // single-bucket is small enough that no eviction happens for it
683            // hence all eviction goes from multi-bucket
684            bytesFreed = bucketMulti.free(bytesToFree);
685          } else if (3 * m <= 2 * bytesRemain) {
686            // multi-bucket is small enough that no eviction happens for it
687            // hence all eviction goes from single-bucket
688            bytesFreed = bucketSingle.free(bytesToFree);
689          } else {
690            // both buckets need to evict some blocks
691            bytesFreed = bucketSingle.free(s - bytesRemain / 3);
692            if (bytesFreed < bytesToFree) {
693              bytesFreed += bucketMulti.free(bytesToFree - bytesFreed);
694            }
695          }
696        }
697      } else {
698        PriorityQueue<BlockBucket> bucketQueue = new PriorityQueue<>(3);
699
700        bucketQueue.add(bucketSingle);
701        bucketQueue.add(bucketMulti);
702        bucketQueue.add(bucketMemory);
703
704        int remainingBuckets = 3;
705
706        BlockBucket bucket;
707        while ((bucket = bucketQueue.poll()) != null) {
708          long overflow = bucket.overflow();
709          if (overflow > 0) {
710            long bucketBytesToFree =
711                Math.min(overflow, (bytesToFree - bytesFreed) / remainingBuckets);
712            bytesFreed += bucket.free(bucketBytesToFree);
713          }
714          remainingBuckets--;
715        }
716      }
717      if (LOG.isTraceEnabled()) {
718        long single = bucketSingle.totalSize();
719        long multi = bucketMulti.totalSize();
720        long memory = bucketMemory.totalSize();
721        LOG.trace("Block cache LRU eviction completed; " +
722          "freed=" + StringUtils.byteDesc(bytesFreed) + ", " +
723          "total=" + StringUtils.byteDesc(this.size.get()) + ", " +
724          "single=" + StringUtils.byteDesc(single) + ", " +
725          "multi=" + StringUtils.byteDesc(multi) + ", " +
726          "memory=" + StringUtils.byteDesc(memory));
727      }
728    } finally {
729      stats.evict();
730      evictionInProgress = false;
731      evictionLock.unlock();
732    }
733  }

752  /**
753   * Used to group blocks into priority buckets.  There will be a BlockBucket
754   * for each priority (single, multi, memory).  Once bucketed, the eviction
755   * algorithm takes the appropriate number of elements out of each according
756   * to configuration parameters and their relatives sizes.
757   */
758  private class BlockBucket implements Comparable<BlockBucket> {
759
760    private final String name;
761    private LruCachedBlockQueue queue;
762    private long totalSize = 0;
763    private long bucketSize;
764
765    public BlockBucket(String name, long bytesToFree, long blockSize, long bucketSize) {
766      this.name = name;
767      this.bucketSize = bucketSize;
768      queue = new LruCachedBlockQueue(bytesToFree, blockSize);
769      totalSize = 0;
770    }
771
772    public void add(LruCachedBlock block) {
773      totalSize += block.heapSize();
774      queue.add(block);
775    }
776
777    public long free(long toFree) {
778      if (LOG.isTraceEnabled()) {
779        LOG.trace("freeing " + StringUtils.byteDesc(toFree) + " from " + this);
780      }
781      LruCachedBlock cb;
782      long freedBytes = 0;
783      while ((cb = queue.pollLast()) != null) {
784        freedBytes += evictBlock(cb, true);
785        if (freedBytes >= toFree) {
786          return freedBytes;
787        }
788      }
789      if (LOG.isTraceEnabled()) {
790        LOG.trace("freed " + StringUtils.byteDesc(freedBytes) + " from " + this);
791      }
792      return freedBytes;
793    }
794
795    public long overflow() {
796      return totalSize - bucketSize;
797    }
798
799    public long totalSize() {
800      return totalSize;
801    }
802
803    @Override
804    public int compareTo(BlockBucket that) {
805      return Long.compare(this.overflow(), that.overflow());
806    }
807
808    @Override
809    public boolean equals(Object that) {
810      if (that == null || !(that instanceof BlockBucket)) {
811        return false;
812      }
813      return compareTo((BlockBucket)that) == 0;
814    }
815
816    @Override
817    public int hashCode() {
818      return Objects.hashCode(name, bucketSize, queue, totalSize);
819    }
820
821    @Override
822    public String toString() {
823      return MoreObjects.toStringHelper(this)
824        .add("name", name)
825        .add("totalSize", StringUtils.byteDesc(totalSize))
826        .add("bucketSize", StringUtils.byteDesc(bucketSize))
827        .toString();
828    }
829  }

876  /*
877   * Eviction thread.  Sits in waiting state until an eviction is triggered
878   * when the cache size grows above the acceptable level.<p>
879   *
880   * Thread is triggered into action by {@link LruBlockCache#runEviction()}
881   */
882  static class EvictionThread extends HasThread {
883
884    private WeakReference<LruBlockCache> cache;
885    private volatile boolean go = true;
886    // flag set after enter the run method, used for test
887    private boolean enteringRun = false;
888
889    public EvictionThread(LruBlockCache cache) {
890      super(Thread.currentThread().getName() + ".LruBlockCache.EvictionThread");
891      setDaemon(true);
892      this.cache = new WeakReference<>(cache);
893    }
894
895    @Override
896    public void run() {
897      enteringRun = true;
898      while (this.go) {
899        synchronized (this) {
900          try {
901            this.wait(1000 * 10/*Don't wait for ever*/);
902          } catch (InterruptedException e) {
903            LOG.warn("Interrupted eviction thread ", e);
904            Thread.currentThread().interrupt();
905          }
906        }
907        LruBlockCache cache = this.cache.get();
908        if (cache == null) break;
909        cache.evict();
910      }
911    }
912
913    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="NN_NAKED_NOTIFY",
914        justification="This is what we want")
915    public void evict() {
916      synchronized (this) {
917        this.notifyAll();
918      }
919    }
920
921    synchronized void shutdown() {
922      this.go = false;
923      this.notifyAll();
924    }
925
926    /**
927     * Used for the test.
928     */
929    boolean isEnteringRun() {
930      return this.enteringRun;
931    }
932  }

986      (4 * Bytes.SIZEOF_LONG) + (11 * ClassSize.REFERENCE) +

995  private static long calculateOverhead(long maxSize, long blockSize, int concurrency) {
996    // FindBugs ICAST_INTEGER_MULTIPLY_CAST_TO_LONG
997    return CACHE_FIXED_OVERHEAD + ClassSize.CONCURRENT_HASHMAP
998           + ((long) Math.ceil(maxSize * 1.2 / blockSize) * ClassSize.CONCURRENT_HASHMAP_ENTRY)
999           + ((long) concurrency * ClassSize.CONCURRENT_HASHMAP_SEGMENT);
1000  }

1107  @Override
1108  public void shutdown() {
1109    if (victimHandler != null) {
1110      victimHandler.shutdown();
1111    }
1112    this.scheduleThreadPool.shutdown();
1113    for (int i = 0; i < 10; i++) {
1114      if (!this.scheduleThreadPool.isShutdown()) {
1115        try {
1116          Thread.sleep(10);
1117        } catch (InterruptedException e) {
1118          LOG.warn("Interrupted while sleeping");
1119          Thread.currentThread().interrupt();
1120          break;
1121        }
1122      }
1123    }
1124
1125    if (!this.scheduleThreadPool.isShutdown()) {
1126      List<Runnable> runnables = this.scheduleThreadPool.shutdownNow();
1127      LOG.debug("Still running " + runnables);
1128    }
1129    this.evictionThread.shutdown();
1130  }
