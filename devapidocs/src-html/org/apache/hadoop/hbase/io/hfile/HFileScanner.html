019package org.apache.hadoop.hbase.io.hfile;
020
021import java.io.IOException;
022import java.nio.ByteBuffer;
023
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025import org.apache.hadoop.hbase.regionserver.Shipper;
026import org.apache.hadoop.hbase.Cell;
027
028/**
029 * A scanner allows you to position yourself within a HFile and
030 * scan through it.  It allows you to reposition yourself as well.
031 *
032 * <p>A scanner doesn't always have a key/value that it is pointing to
033 * when it is first created and before
034 * {@link #seekTo()}/{@link #seekTo(Cell)} are called.
035 * In this case, {@link #getKey()}/{@link #getValue()} returns null.  At most
036 * other times, a key and value will be available.  The general pattern is that
037 * you position the Scanner using the seekTo variants and then getKey and
038 * getValue.
039 */
040@InterfaceAudience.Private
041public interface HFileScanner extends Shipper {
042  /**
043   * SeekTo or just before the passed <code>cell</code>.  Examine the return
044   * code to figure whether we found the cell or not.
045   * Consider the cell stream of all the cells in the file,
046   * <code>c[0] .. c[n]</code>, where there are n cells in the file.
047   * @param cell
048   * @return -1, if cell &lt; c[0], no position;
049   * 0, such that c[i] = cell and scanner is left in position i; and
050   * 1, such that c[i] &lt; cell, and scanner is left in position i.
051   * The scanner will position itself between c[i] and c[i+1] where
052   * c[i] &lt; cell &lt;= c[i+1].
053   * If there is no cell c[i+1] greater than or equal to the input cell, then the
054   * scanner will position itself at the end of the file and next() will return
055   * false when it is called.
056   * @throws IOException
057   */
058  int seekTo(Cell cell) throws IOException;
059
060  /**
061   * Reseek to or just before the passed <code>cell</code>. Similar to seekTo
062   * except that this can be called even if the scanner is not at the beginning
063   * of a file.
064   * This can be used to seek only to cells which come after the current position
065   * of the scanner.
066   * Consider the cell stream of all the cells in the file,
067   * <code>c[0] .. c[n]</code>, where there are n cellc in the file after
068   * current position of HFileScanner.
069   * The scanner will position itself between c[i] and c[i+1] where
070   * c[i] &lt; cell &lt;= c[i+1].
071   * If there is no cell c[i+1] greater than or equal to the input cell, then the
072   * scanner will position itself at the end of the file and next() will return
073   * false when it is called.
074   * @param cell Cell to find (should be non-null)
075   * @return -1, if cell &lt; c[0], no position;
076   * 0, such that c[i] = cell and scanner is left in position i; and
077   * 1, such that c[i] &lt; cell, and scanner is left in position i.
078   * @throws IOException
079   */
080  int reseekTo(Cell cell) throws IOException;
081
082  /**
083   * Consider the cell stream of all the cells in the file,
084   * <code>c[0] .. c[n]</code>, where there are n cells in the file.
085   * @param cell Cell to find
086   * @return false if cell &lt;= c[0] or true with scanner in position 'i' such
087   * that: c[i] &lt; cell.  Furthermore: there may be a c[i+1], such that
088   * c[i] &lt; cell &lt;= c[i+1] but there may also NOT be a c[i+1], and next() will
089   * return false (EOF).
090   * @throws IOException
091   */
092  boolean seekBefore(Cell cell) throws IOException;
093
094  /**
095   * Positions this scanner at the start of the file.
096   * @return False if empty file; i.e. a call to next would return false and
097   * the current key and value are undefined.
098   * @throws IOException
099   */
100  boolean seekTo() throws IOException;
101  /**
102   * Scans to the next entry in the file.
103   * @return Returns false if you are at the end otherwise true if more in file.
104   * @throws IOException
105   */
106  boolean next() throws IOException;
107  /**
108   * Gets the current key in the form of a cell. You must call
109   * {@link #seekTo(Cell)} before this method.
110   * @return gets the current key as a Cell.
111   */
112  Cell getKey();
113  /**
114   * Gets a buffer view to the current value.  You must call
115   * {@link #seekTo(Cell)} before this method.
116   *
117   * @return byte buffer for the value. The limit is set to the value size, and
118   * the position is 0, the start of the buffer view.
119   */
120  ByteBuffer getValue();
121  /**
122   * @return Instance of {@link org.apache.hadoop.hbase.Cell}.
123   */
124  Cell getCell();
125  /**
126   * Convenience method to get a copy of the key as a string - interpreting the
127   * bytes as UTF8. You must call {@link #seekTo(Cell)} before this method.
128   * @return key as a string
129   */
130  String getKeyString();
131  /**
132   * Convenience method to get a copy of the value as a string - interpreting
133   * the bytes as UTF8. You must call {@link #seekTo(Cell)} before this method.
134   * @return value as a string
135   */
136  String getValueString();
137  /**
138   * @return Reader that underlies this Scanner instance.
139   */
140  HFile.Reader getReader();
141  /**
142   * @return True is scanner has had one of the seek calls invoked; i.e.
143   * {@link #seekBefore(Cell)} or {@link #seekTo()} or {@link #seekTo(Cell)}.
144   * Otherwise returns false.
145   */
146  boolean isSeeked();
147
148  /**
149   * @return the next key in the index (the key to seek to the next block)
150   */
151  Cell getNextIndexedKey();
152
153  /**
154   * Close this HFile scanner and do necessary cleanup.
155   */
156  void close();
157}
