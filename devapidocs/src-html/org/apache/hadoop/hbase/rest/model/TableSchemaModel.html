055/**
056 * A representation of HBase table descriptors.
057 *
058 * <pre>
059 * &lt;complexType name="TableSchema"&gt;
060 *   &lt;sequence&gt;
061 *     &lt;element name="column" type="tns:ColumnSchema"
062 *       maxOccurs="unbounded" minOccurs="1"&gt;&lt;/element&gt;
063 *   &lt;/sequence&gt;
064 *   &lt;attribute name="name" type="string"&gt;&lt;/attribute&gt;
065 *   &lt;anyAttribute&gt;&lt;/anyAttribute&gt;
066 * &lt;/complexType&gt;
067 * </pre>
068 */
069@XmlRootElement(name="TableSchema")
070@InterfaceAudience.Private
071public class TableSchemaModel implements Serializable, ProtobufMessageHandler {
072  private static final long serialVersionUID = 1L;
073  private static final QName IS_META = new QName(HTableDescriptor.IS_META);
074  private static final QName IS_ROOT = new QName(HTableDescriptor.IS_ROOT);
075  private static final QName READONLY = new QName(HTableDescriptor.READONLY);
076  private static final QName TTL = new QName(HColumnDescriptor.TTL);
077  private static final QName VERSIONS = new QName(HConstants.VERSIONS);
078  private static final QName COMPRESSION =
079    new QName(HColumnDescriptor.COMPRESSION);
080
081  private String name;
082  private Map<QName,Object> attrs = new LinkedHashMap<>();
083  private List<ColumnSchemaModel> columns = new ArrayList<>();
084
085  /**
086   * Default constructor.
087   */
088  public TableSchemaModel() {}
089
090  /**
091   * Constructor
092   * @param tableDescriptor the table descriptor
093   */
094  public TableSchemaModel(TableDescriptor tableDescriptor) {
095    setName(tableDescriptor.getTableName().getNameAsString());
096    for (Map.Entry<Bytes, Bytes> e : tableDescriptor.getValues().entrySet()) {
097      addAttribute(Bytes.toString(e.getKey().get()),
098        Bytes.toString(e.getValue().get()));
099    }
100    for (ColumnFamilyDescriptor hcd : tableDescriptor.getColumnFamilies()) {
101      ColumnSchemaModel columnModel = new ColumnSchemaModel();
102      columnModel.setName(hcd.getNameAsString());
103      for (Map.Entry<Bytes, Bytes> e:
104          hcd.getValues().entrySet()) {
105        columnModel.addAttribute(Bytes.toString(e.getKey().get()),
106            Bytes.toString(e.getValue().get()));
107      }
108      addColumnFamily(columnModel);
109    }
110  }

189  /* (non-Javadoc)
190   * @see java.lang.Object#toString()
191   */
192  @Override
193  public String toString() {
194    StringBuilder sb = new StringBuilder();
195    sb.append("{ NAME=> '");
196    sb.append(name);
197    sb.append('\'');
198    for (Map.Entry<QName,Object> e : attrs.entrySet()) {
199      sb.append(", ");
200      sb.append(e.getKey().getLocalPart());
201      sb.append(" => '");
202      sb.append(e.getValue().toString());
203      sb.append('\'');
204    }
205    sb.append(", COLUMNS => [ ");
206    Iterator<ColumnSchemaModel> i = columns.iterator();
207    while (i.hasNext()) {
208      ColumnSchemaModel family = i.next();
209      sb.append(family.toString());
210      if (i.hasNext()) {
211        sb.append(',');
212      }
213      sb.append(' ');
214    }
215    sb.append("] }");
216    return sb.toString();
217  }

269  @Override
270  public byte[] createProtobufOutput() {
271    TableSchema.Builder builder = TableSchema.newBuilder();
272    builder.setName(name);
273    for (Map.Entry<QName, Object> e : attrs.entrySet()) {
274      TableSchema.Attribute.Builder attrBuilder =
275        TableSchema.Attribute.newBuilder();
276      attrBuilder.setName(e.getKey().getLocalPart());
277      attrBuilder.setValue(e.getValue().toString());
278      builder.addAttrs(attrBuilder);
279    }
280    for (ColumnSchemaModel family : columns) {
281      Map<QName, Object> familyAttrs = family.getAny();
282      ColumnSchema.Builder familyBuilder = ColumnSchema.newBuilder();
283      familyBuilder.setName(family.getName());
284      for (Map.Entry<QName, Object> e : familyAttrs.entrySet()) {
285        ColumnSchema.Attribute.Builder attrBuilder =
286          ColumnSchema.Attribute.newBuilder();
287        attrBuilder.setName(e.getKey().getLocalPart());
288        attrBuilder.setValue(e.getValue().toString());
289        familyBuilder.addAttrs(attrBuilder);
290      }
291      if (familyAttrs.containsKey(TTL)) {
292        familyBuilder.setTtl(Integer.parseInt(familyAttrs.get(TTL).toString()));
293      }
294      if (familyAttrs.containsKey(VERSIONS)) {
295        familyBuilder.setMaxVersions(Integer.parseInt(familyAttrs.get(VERSIONS).toString()));
296      }
297      if (familyAttrs.containsKey(COMPRESSION)) {
298        familyBuilder.setCompression(familyAttrs.get(COMPRESSION).toString());
299      }
300      builder.addColumns(familyBuilder);
301    }
302    if (attrs.containsKey(READONLY)) {
303      builder.setReadOnly(Boolean.parseBoolean(attrs.get(READONLY).toString()));
304    }
305    return builder.build().toByteArray();
306  }
307
308  @Override
309  public ProtobufMessageHandler getObjectFromMessage(byte[] message)
310      throws IOException {
311    TableSchema.Builder builder = TableSchema.newBuilder();
312    ProtobufUtil.mergeFrom(builder, message);
313    this.setName(builder.getName());
314    for (TableSchema.Attribute attr : builder.getAttrsList()) {
315      this.addAttribute(attr.getName(), attr.getValue());
316    }
317    if (builder.hasReadOnly()) {
318      this.addAttribute(HTableDescriptor.READONLY, builder.getReadOnly());
319    }
320    for (ColumnSchema family : builder.getColumnsList()) {
321      ColumnSchemaModel familyModel = new ColumnSchemaModel();
322      familyModel.setName(family.getName());
323      for (ColumnSchema.Attribute attr : family.getAttrsList()) {
324        familyModel.addAttribute(attr.getName(), attr.getValue());
325      }
326      if (family.hasTtl()) {
327        familyModel.addAttribute(HColumnDescriptor.TTL, family.getTtl());
328      }
329      if (family.hasMaxVersions()) {
330        familyModel.addAttribute(HConstants.VERSIONS,
331          family.getMaxVersions());
332      }
333      if (family.hasCompression()) {
334        familyModel.addAttribute(HColumnDescriptor.COMPRESSION,
335          family.getCompression());
336      }
337      this.addColumnFamily(familyModel);
338    }
339    return this;
340  }

342  /**
343   * @return a table descriptor
344   */
345  @JsonIgnore
346  public TableDescriptor getTableDescriptor() {
347    TableDescriptorBuilder tableDescriptorBuilder =
348      TableDescriptorBuilder.newBuilder(TableName.valueOf(getName()));
349    for (Map.Entry<QName, Object> e : getAny().entrySet()) {
350      tableDescriptorBuilder.setValue(e.getKey().getLocalPart(), e.getValue().toString());
351    }
352    for (ColumnSchemaModel column : getColumns()) {
353      ColumnFamilyDescriptorBuilder cfdb = ColumnFamilyDescriptorBuilder
354        .newBuilder(Bytes.toBytes(column.getName()));
355      for (Map.Entry<QName, Object> e : column.getAny().entrySet()) {
356        cfdb.setValue(e.getKey().getLocalPart(), e.getValue().toString());
357      }
358      tableDescriptorBuilder.setColumnFamily(cfdb.build());
359    }
360    return tableDescriptorBuilder.build();
361  }
