653  @Override
654  public String toString() {
655    StringBuilder sb = new StringBuilder();
656    sb.append(String.format("%d live servers, %d dead servers, " + "%.4f average load%n%n",
657      liveNodes.size(), deadNodes.size(), averageLoad));
658    if (!liveNodes.isEmpty()) {
659      sb.append(liveNodes.size());
660      sb.append(" live servers\n");
661      for (Node node : liveNodes) {
662        sb.append("    ");
663        sb.append(node.name);
664        sb.append(' ');
665        sb.append(node.startCode);
666        sb.append("\n        requests=");
667        sb.append(node.requests);
668        sb.append(", regions=");
669        sb.append(node.regions.size());
670        sb.append("\n        heapSizeMB=");
671        sb.append(node.heapSizeMB);
672        sb.append("\n        maxHeapSizeMB=");
673        sb.append(node.maxHeapSizeMB);
674        sb.append("\n\n");
675        for (Node.Region region : node.regions) {
676          sb.append("        ");
677          sb.append(Bytes.toString(region.name));
678          sb.append("\n            stores=");
679          sb.append(region.stores);
680          sb.append("\n            storefiless=");
681          sb.append(region.storefiles);
682          sb.append("\n            storefileSizeMB=");
683          sb.append(region.storefileSizeMB);
684          sb.append("\n            memstoreSizeMB=");
685          sb.append(region.memstoreSizeMB);
686          sb.append("\n            storefileIndexSizeKB=");
687          sb.append(region.storefileIndexSizeKB);
688          sb.append("\n            readRequestsCount=");
689          sb.append(region.readRequestsCount);
690          sb.append("\n            cpRequestsCount=");
691          sb.append(region.cpRequestsCount);
692          sb.append("\n            writeRequestsCount=");
693          sb.append(region.writeRequestsCount);
694          sb.append("\n            rootIndexSizeKB=");
695          sb.append(region.rootIndexSizeKB);
696          sb.append("\n            totalStaticIndexSizeKB=");
697          sb.append(region.totalStaticIndexSizeKB);
698          sb.append("\n            totalStaticBloomSizeKB=");
699          sb.append(region.totalStaticBloomSizeKB);
700          sb.append("\n            totalCompactingKVs=");
701          sb.append(region.totalCompactingKVs);
702          sb.append("\n            currentCompactedKVs=");
703          sb.append(region.currentCompactedKVs);
704          sb.append('\n');
705        }
706        sb.append('\n');
707      }
708    }
709    if (!deadNodes.isEmpty()) {
710      sb.append('\n');
711      sb.append(deadNodes.size());
712      sb.append(" dead servers\n");
713      for (String node : deadNodes) {
714        sb.append("    ");
715        sb.append(node);
716        sb.append('\n');
717      }
718    }
719    return sb.toString();
720  }

722  @Override
723  public byte[] createProtobufOutput() {
724    StorageClusterStatus.Builder builder = StorageClusterStatus.newBuilder();
725    builder.setRegions(regions);
726    builder.setRequests(requests);
727    builder.setAverageLoad(averageLoad);
728    for (Node node : liveNodes) {
729      StorageClusterStatus.Node.Builder nodeBuilder = StorageClusterStatus.Node.newBuilder();
730      nodeBuilder.setName(node.name);
731      nodeBuilder.setStartCode(node.startCode);
732      nodeBuilder.setRequests(node.requests);
733      nodeBuilder.setHeapSizeMB(node.heapSizeMB);
734      nodeBuilder.setMaxHeapSizeMB(node.maxHeapSizeMB);
735      for (Node.Region region : node.regions) {
736        StorageClusterStatus.Region.Builder regionBuilder =
737          StorageClusterStatus.Region.newBuilder();
738        regionBuilder.setName(UnsafeByteOperations.unsafeWrap(region.name));
739        regionBuilder.setStores(region.stores);
740        regionBuilder.setStorefiles(region.storefiles);
741        regionBuilder.setStorefileSizeMB(region.storefileSizeMB);
742        regionBuilder.setMemStoreSizeMB(region.memstoreSizeMB);
743        regionBuilder.setStorefileIndexSizeKB(region.storefileIndexSizeKB);
744        regionBuilder.setReadRequestsCount(region.readRequestsCount);
745        regionBuilder.setCpRequestsCount(region.cpRequestsCount);
746        regionBuilder.setWriteRequestsCount(region.writeRequestsCount);
