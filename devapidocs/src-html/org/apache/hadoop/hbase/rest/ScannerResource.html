042import org.apache.hadoop.hbase.classification.InterfaceAudience;
043import org.apache.hadoop.hbase.TableNotFoundException;

045import org.apache.hadoop.hbase.rest.model.ScannerModel;
046
047@InterfaceAudience.Private
048public class ScannerResource extends ResourceBase {
049
050  private static final Log LOG = LogFactory.getLog(ScannerResource.class);
051
052  static final Map<String,ScannerInstanceResource> scanners =
053   Collections.synchronizedMap(new HashMap<String,ScannerInstanceResource>());
054
055  TableResource tableResource;
056
057  /**
058   * Constructor
059   * @param tableResource
060   * @throws IOException
061   */
062  public ScannerResource(TableResource tableResource)throws IOException {
063    super();
064    this.tableResource = tableResource;
065  }
066
067  static boolean delete(final String id) {
068    ScannerInstanceResource instance = scanners.remove(id);
069    if (instance != null) {
070      instance.generator.close();
071      return true;
072    } else {
073      return false;
074    }
075  }
076
077  Response update(final ScannerModel model, final boolean replace,
078      final UriInfo uriInfo) {
079    servlet.getMetrics().incrementRequests(1);
080    if (servlet.isReadOnly()) {
081      return Response.status(Response.Status.FORBIDDEN)
082        .type(MIMETYPE_TEXT).entity("Forbidden" + CRLF)
083        .build();
084    }
085    byte[] endRow = model.hasEndRow() ? model.getEndRow() : null;
086    RowSpec spec = null;
087    if (model.getLabels() != null) {
088      spec = new RowSpec(model.getStartRow(), endRow, model.getColumns(), model.getStartTime(),
089          model.getEndTime(), model.getMaxVersions(), model.getLabels());
090    } else {
091      spec = new RowSpec(model.getStartRow(), endRow, model.getColumns(), model.getStartTime(),
092          model.getEndTime(), model.getMaxVersions());
093    }
094    
095    try {
096      Filter filter = ScannerResultGenerator.buildFilterFromModel(model);
097      String tableName = tableResource.getName();
098      ScannerResultGenerator gen =
099        new ScannerResultGenerator(tableName, spec, filter, model.getCaching(),
100          model.getCacheBlocks());
101      String id = gen.getID();
102      ScannerInstanceResource instance =
103        new ScannerInstanceResource(tableName, id, gen, model.getBatch());
104      scanners.put(id, instance);
105      if (LOG.isDebugEnabled()) {
106        LOG.debug("new scanner: " + id);
107      }
108      UriBuilder builder = uriInfo.getAbsolutePathBuilder();
109      URI uri = builder.path(id).build();
110      servlet.getMetrics().incrementSucessfulPutRequests(1);
111      return Response.created(uri).build();
112    } catch (Exception e) {
113      servlet.getMetrics().incrementFailedPutRequests(1);
114      if (e instanceof TableNotFoundException) {
115        return Response.status(Response.Status.NOT_FOUND)
116          .type(MIMETYPE_TEXT).entity("Not found" + CRLF)
117          .build();
118      } else if (e instanceof RuntimeException) {
119        return Response.status(Response.Status.BAD_REQUEST)
120          .type(MIMETYPE_TEXT).entity("Bad request" + CRLF)
121          .build();
122      }
123      return Response.status(Response.Status.SERVICE_UNAVAILABLE)
124        .type(MIMETYPE_TEXT).entity("Unavailable" + CRLF)
125        .build();
126    }
127  }
128
129  @PUT
130  @Consumes({MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF,
131    MIMETYPE_PROTOBUF_IETF})
132  public Response put(final ScannerModel model, 
133      final @Context UriInfo uriInfo) {
134    if (LOG.isDebugEnabled()) {
135      LOG.debug("PUT " + uriInfo.getAbsolutePath());
136    }
137    return update(model, true, uriInfo);
138  }
139
140  @POST
141  @Consumes({MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF,
142    MIMETYPE_PROTOBUF_IETF})
143  public Response post(final ScannerModel model,
144      final @Context UriInfo uriInfo) {
145    if (LOG.isDebugEnabled()) {
146      LOG.debug("POST " + uriInfo.getAbsolutePath());
147    }
148    return update(model, false, uriInfo);
149  }
150
151  @Path("{scanner: .+}")
152  public ScannerInstanceResource getScannerInstanceResource(
153      final @PathParam("scanner") String id) throws IOException {
154    ScannerInstanceResource instance = scanners.get(id);
155    if (instance == null) {
156      servlet.getMetrics().incrementFailedGetRequests(1);
157      return new ScannerInstanceResource();
158    } else {
159      servlet.getMetrics().incrementSucessfulGetRequests(1);
160    }
161    return instance;
162  }
163}
