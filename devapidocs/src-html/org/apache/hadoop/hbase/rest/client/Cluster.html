020package org.apache.hadoop.hbase.rest.client;
021
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.List;
025
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027import org.apache.hadoop.hbase.classification.InterfaceStability;
028
029/**
030 * A list of 'host:port' addresses of HTTP servers operating as a single
031 * entity, for example multiple redundant web service gateways.
032 */
033@InterfaceAudience.Public
034@InterfaceStability.Stable
035public class Cluster {
036  protected List<String> nodes = 
037    Collections.synchronizedList(new ArrayList<String>());
038  protected String lastHost;
039
040  /**
041   * Constructor
042   */
043  public Cluster() {}
044
045  /**
046   * Constructor
047   * @param nodes a list of service locations, in 'host:port' format
048   */
049  public Cluster(List<String> nodes) {
050    this.nodes.addAll(nodes);
051  }
052
053  /**
054   * @return true if no locations have been added, false otherwise
055   */
056  public boolean isEmpty() {
057    return nodes.isEmpty();
058  }
059
060  /**
061   * Add a node to the cluster
062   * @param node the service location in 'host:port' format
063   */
064  public Cluster add(String node) {
065    nodes.add(node);
066    return this;
067  }
068
069  /**
070   * Add a node to the cluster
071   * @param name host name
072   * @param port service port
073   */
074  public Cluster add(String name, int port) {
075    StringBuilder sb = new StringBuilder();
076    sb.append(name);
077    sb.append(':');
078    sb.append(port);
079    return add(sb.toString());
080  }
081
082  /**
083   * Remove a node from the cluster
084   * @param node the service location in 'host:port' format
085   */
086  public Cluster remove(String node) {
087    nodes.remove(node);
088    return this;
089  }
090
091  /**
092   * Remove a node from the cluster
093   * @param name host name
094   * @param port service port
095   */
096  public Cluster remove(String name, int port) {
097    StringBuilder sb = new StringBuilder();
098    sb.append(name);
099    sb.append(':');
100    sb.append(port);
101    return remove(sb.toString());
102  }
103
104  @Override public String toString() {
105    return "Cluster{" +
106        "nodes=" + nodes +
107        ", lastHost='" + lastHost + '\'' +
108        '}';
109  }
110}
