019package org.apache.hadoop.hbase.rest;
020
021import org.apache.hadoop.hbase.metrics.BaseSource;
022import org.apache.hadoop.hbase.metrics.JvmPauseMonitorSource;
023
024/**
025 * Interface of the Metrics Source that will export data to Hadoop's Metrics2 system.
026 */
027public interface MetricsRESTSource extends BaseSource, JvmPauseMonitorSource {
028
029  String METRICS_NAME = "REST";
030
031  String CONTEXT = "rest";
032
033  String JMX_CONTEXT = "REST";
034
035  String METRICS_DESCRIPTION = "Metrics about the HBase REST server";
036
037  String REQUEST_KEY = "requests";
038
039  String SUCCESSFUL_GET_KEY = "successfulGet";
040
041  String SUCCESSFUL_PUT_KEY = "successfulPut";
042
043  String SUCCESSFUL_DELETE_KEY = "successfulDelete";
044
045  String FAILED_GET_KEY = "failedGet";
046
047  String FAILED_PUT_KEY = "failedPut";
048
049  String FAILED_DELETE_KEY = "failedDelete";
050  
051  String SUCCESSFUL_SCAN_KEY = "successfulScanCount";
052  
053  String FAILED_SCAN_KEY = "failedScanCount";
054
055  String SUCCESSFUL_APPEND_KEY = "successfulAppendCount";
056
057  String FAILED_APPEND_KEY = "failedAppendCount";
058
059  String SUCCESSFUL_INCREMENT_KEY = "successfulIncrementCount";
060
061  String FAILED_INCREMENT_KEY = "failedIncrementCount";
062
063  /**
064   * Increment the number of requests
065   *
066   * @param inc Ammount to increment by
067   */
068  void incrementRequests(int inc);
069
070  /**
071   * Increment the number of successful Get requests.
072   *
073   * @param inc Number of successful get requests.
074   */
075  void incrementSucessfulGetRequests(int inc);
076
077  /**
078   * Increment the number of successful Put requests.
079   *
080   * @param inc Number of successful put requests.
081   */
082  void incrementSucessfulPutRequests(int inc);
083
084  /**
085   * Increment the number of successful Delete requests.
086   *
087   * @param inc number of successful delete requests
088   */
089  void incrementSucessfulDeleteRequests(int inc);
090
091  /**
092   * Increment the number of failed Put Requests.
093   *
094   * @param inc Number of failed Put requests.
095   */
096  void incrementFailedPutRequests(int inc);
097
098  /**
099   * Increment the number of failed Get requests.
100   *
101   * @param inc The number of failed Get Requests.
102   */
103  void incrementFailedGetRequests(int inc);
104
105  /**
106   * Increment the number of failed Delete requests.
107   *
108   * @param inc The number of failed delete requests.
109   */
110  void incrementFailedDeleteRequests(int inc);
111  
112  /**
113   * Increment the number of successful scan requests.
114   *
115   * @param inc Number of successful scan requests.
116   */
117  void incrementSucessfulScanRequests(final int inc);
118  
119  /**
120   * Increment the number failed scan requests.
121   *
122   * @param inc Number of failed scan requests.
123   */
124  void incrementFailedScanRequests(final int inc);
125
126  /**
127   * Increment the number of successful append requests.
128   *
129   * @param inc Number of successful append requests.
130   */
131  void incrementSucessfulAppendRequests(final int inc);
132
133  /**
134   * Increment the number failed append requests.
135   *
136   * @param inc Number of failed append requests.
137   */
138  void incrementFailedAppendRequests(final int inc);
139
140  /**
141   * Increment the number of successful increment requests.
142   *
143   * @param inc Number of successful increment requests.
144   */
145  void incrementSucessfulIncrementRequests(final int inc);
146
147  /**
148   * Increment the number failed increment requests.
149   *
150   * @param inc Number of failed increment requests.
151   */
152  void incrementFailedIncrementRequests(final int inc);
153}
