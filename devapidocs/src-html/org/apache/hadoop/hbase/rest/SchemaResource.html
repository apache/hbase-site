049@InterfaceAudience.Private
050public class SchemaResource extends ResourceBase {
051  private static final Logger LOG = LoggerFactory.getLogger(SchemaResource.class);
052
053  static CacheControl cacheControl;
054  static {
055    cacheControl = new CacheControl();
056    cacheControl.setNoCache(true);
057    cacheControl.setNoTransform(false);
058  }
059
060  TableResource tableResource;
061
062  /**
063   * Constructor
064   */
065  public SchemaResource(TableResource tableResource) throws IOException {
066    super();
067    this.tableResource = tableResource;
068  }
069
070  private HTableDescriptor getTableSchema() throws IOException, TableNotFoundException {
071    try (Table table = servlet.getTable(tableResource.getName())) {
072      return new HTableDescriptor(table.getDescriptor());
073    }
074  }
075
076  @GET
077  @Produces({MIMETYPE_TEXT, MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF,
078    MIMETYPE_PROTOBUF_IETF})
079  public Response get(final @Context UriInfo uriInfo) {
080    if (LOG.isTraceEnabled()) {
081      LOG.trace("GET " + uriInfo.getAbsolutePath());
082    }
083    servlet.getMetrics().incrementRequests(1);
084    try {
085      ResponseBuilder response =
086        Response.ok(new TableSchemaModel(getTableSchema()));
087      response.cacheControl(cacheControl);
088      servlet.getMetrics().incrementSucessfulGetRequests(1);
089      return response.build();
090    } catch (Exception e) {
091      servlet.getMetrics().incrementFailedGetRequests(1);
092      return processException(e);
093    }
094  }
095
096  private Response replace(final TableName name, final TableSchemaModel model,
097      final UriInfo uriInfo, final Admin admin) {
098    if (servlet.isReadOnly()) {
099      return Response.status(Response.Status.FORBIDDEN)
100        .type(MIMETYPE_TEXT).entity("Forbidden" + CRLF)
101        .build();
102    }
103    try {
104      HTableDescriptor htd = new HTableDescriptor(name);
105      for (Map.Entry<QName,Object> e: model.getAny().entrySet()) {
106        htd.setValue(e.getKey().getLocalPart(), e.getValue().toString());
107      }
108      for (ColumnSchemaModel family: model.getColumns()) {
109        HColumnDescriptor hcd = new HColumnDescriptor(family.getName());
110        for (Map.Entry<QName,Object> e: family.getAny().entrySet()) {
111          hcd.setValue(e.getKey().getLocalPart(), e.getValue().toString());
112        }
113        htd.addFamily(hcd);
114      }
115      if (admin.tableExists(name)) {
116        admin.disableTable(name);
117        admin.modifyTable(htd);
118        admin.enableTable(name);
119        servlet.getMetrics().incrementSucessfulPutRequests(1);
120      } else try {
121        admin.createTable(htd);
122        servlet.getMetrics().incrementSucessfulPutRequests(1);
123      } catch (TableExistsException e) {
124        // race, someone else created a table with the same name
125        return Response.status(Response.Status.NOT_MODIFIED)
126          .type(MIMETYPE_TEXT).entity("Not modified" + CRLF)
127          .build();
128      }
129      return Response.created(uriInfo.getAbsolutePath()).build();
130    } catch (Exception e) {
131      LOG.info("Caught exception", e);
132      servlet.getMetrics().incrementFailedPutRequests(1);
133      return processException(e);
134    }
135  }
136
137  private Response update(final TableName name, final TableSchemaModel model,
138      final UriInfo uriInfo, final Admin admin) {
139    if (servlet.isReadOnly()) {
140      return Response.status(Response.Status.FORBIDDEN)
141        .type(MIMETYPE_TEXT).entity("Forbidden" + CRLF)
142        .build();
143    }
144    try {
145      HTableDescriptor htd = new HTableDescriptor(admin.getDescriptor(name));
146      admin.disableTable(name);
147      try {
148        for (ColumnSchemaModel family: model.getColumns()) {
149          HColumnDescriptor hcd = new HColumnDescriptor(family.getName());
150          for (Map.Entry<QName,Object> e: family.getAny().entrySet()) {
151            hcd.setValue(e.getKey().getLocalPart(), e.getValue().toString());
152          }
153          if (htd.hasFamily(hcd.getName())) {
154            admin.modifyColumnFamily(name, hcd);
155          } else {
156            admin.addColumnFamily(name, hcd);
157          }
158        }
159      } catch (IOException e) {
160        return Response.status(Response.Status.SERVICE_UNAVAILABLE)
161          .type(MIMETYPE_TEXT).entity("Unavailable" + CRLF)
162          .build();
163      } finally {
164        admin.enableTable(TableName.valueOf(tableResource.getName()));
165      }
166      servlet.getMetrics().incrementSucessfulPutRequests(1);
167      return Response.ok().build();
168    } catch (Exception e) {
169      servlet.getMetrics().incrementFailedPutRequests(1);
170      return processException(e);
171    }
172  }
173
174  private Response update(final TableSchemaModel model, final boolean replace,
175      final UriInfo uriInfo) {
176    try {
177      TableName name = TableName.valueOf(tableResource.getName());
178      Admin admin = servlet.getAdmin();
179      if (replace || !admin.tableExists(name)) {
180        return replace(name, model, uriInfo, admin);
181      } else {
182        return update(name, model, uriInfo, admin);
183      }
184    } catch (Exception e) {
185      servlet.getMetrics().incrementFailedPutRequests(1);
186      // Avoid re-unwrapping the exception
187      if (e instanceof WebApplicationException) {
188        throw (WebApplicationException) e;
189      }
190      return processException(e);
191    }
192  }
193
194  @PUT
195  @Consumes({MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF,
196    MIMETYPE_PROTOBUF_IETF})
197  public Response put(final TableSchemaModel model,
198      final @Context UriInfo uriInfo) {
199    if (LOG.isTraceEnabled()) {
200      LOG.trace("PUT " + uriInfo.getAbsolutePath());
201    }
202    servlet.getMetrics().incrementRequests(1);
203    return update(model, true, uriInfo);
204  }
205
206  @POST
207  @Consumes({MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF,
208    MIMETYPE_PROTOBUF_IETF})
209  public Response post(final TableSchemaModel model,
210      final @Context UriInfo uriInfo) {
211    if (LOG.isTraceEnabled()) {
212      LOG.trace("PUT " + uriInfo.getAbsolutePath());
213    }
214    servlet.getMetrics().incrementRequests(1);
215    return update(model, false, uriInfo);
216  }
217
218  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="DE_MIGHT_IGNORE",
219      justification="Expected")
220  @DELETE
221  public Response delete(final @Context UriInfo uriInfo) {
222    if (LOG.isTraceEnabled()) {
223      LOG.trace("DELETE " + uriInfo.getAbsolutePath());
224    }
225    servlet.getMetrics().incrementRequests(1);
226    if (servlet.isReadOnly()) {
227      return Response.status(Response.Status.FORBIDDEN).type(MIMETYPE_TEXT)
228          .entity("Forbidden" + CRLF).build();
229    }
230    try {
231      Admin admin = servlet.getAdmin();
232      try {
233        admin.disableTable(TableName.valueOf(tableResource.getName()));
234      } catch (TableNotEnabledException e) { /* this is what we want anyway */ }
235      admin.deleteTable(TableName.valueOf(tableResource.getName()));
236      servlet.getMetrics().incrementSucessfulDeleteRequests(1);
237      return Response.ok().build();
238    } catch (Exception e) {
239      servlet.getMetrics().incrementFailedDeleteRequests(1);
240      return processException(e);
241    }
242  }
