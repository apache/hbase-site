018package org.apache.hadoop.hbase.hbtop;
019
020import java.util.ArrayList;
021import java.util.List;
022import java.util.Objects;
023import java.util.Optional;
024import java.util.stream.Collectors;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.conf.Configured;
027import org.apache.hadoop.hbase.HBaseConfiguration;
028import org.apache.hadoop.hbase.HBaseInterfaceAudience;
029import org.apache.hadoop.hbase.hbtop.field.Field;
030import org.apache.hadoop.hbase.hbtop.field.FieldInfo;
031import org.apache.hadoop.hbase.hbtop.mode.Mode;
032import org.apache.hadoop.hbase.hbtop.screen.Screen;
033import org.apache.hadoop.util.Tool;
034import org.apache.hadoop.util.ToolRunner;
035import org.apache.yetus.audience.InterfaceAudience;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038
039import org.apache.hbase.thirdparty.org.apache.commons.cli.CommandLine;
040import org.apache.hbase.thirdparty.org.apache.commons.cli.DefaultParser;
041import org.apache.hbase.thirdparty.org.apache.commons.cli.HelpFormatter;
042import org.apache.hbase.thirdparty.org.apache.commons.cli.Options;
043
044/**
045 * A real-time monitoring tool for HBase like Unix top command.
046 */
047@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
048public class HBTop extends Configured implements Tool {
049
050  private static final Logger LOGGER = LoggerFactory.getLogger(HBTop.class);
051
052  public HBTop() {
053    this(HBaseConfiguration.create());
054  }
055
056  public HBTop(Configuration conf) {
057    super(Objects.requireNonNull(conf));
058  }
059
060  @Override
061  public int run(String[] args) throws Exception {
062    long initialRefreshDelay = 3 * 1000;
063    Mode initialMode = Mode.REGION;
064    List<Field> initialFields = null;
065    Field initialSortField = null;
066    Boolean initialAscendingSort = null;
067    List<RecordFilter> initialFilters = null;
068    long numberOfIterations = Long.MAX_VALUE;
069    boolean batchMode = false;
070    try {
071      Options opts = getOptions();
072      CommandLine commandLine = new DefaultParser().parse(opts, args);
073
074      if (commandLine.hasOption("help")) {
075        printUsage(opts);
076        return 0;
077      }
078
079      if (commandLine.hasOption("mode")) {
080        String mode = commandLine.getOptionValue("mode");
081        switch (mode) {
082          case "n":
083            initialMode = Mode.NAMESPACE;
084            break;
085
086          case "t":
087            initialMode = Mode.TABLE;
088            break;
089
090          case "r":
091            initialMode = Mode.REGION;
092            break;
093
094          case "s":
095            initialMode = Mode.REGION_SERVER;
096            break;
097
098          case "u":
099            initialMode = Mode.USER;
100            break;
101
102          case "c":
103            initialMode = Mode.CLIENT;
104            break;
105
106          default:
107            LOGGER.warn("Mode set invalid, using default");
108            break;
109        }
110      }
111
112      if (commandLine.hasOption("outputFieldNames")) {
113        initialMode.getFieldInfos().forEach(f -> System.out.println(f.getField().getHeader()));
114        return 0;
115      }
116
117      if (commandLine.hasOption("delay")) {
118        int delay = 0;
119        try {
120          delay = Integer.parseInt(commandLine.getOptionValue("delay"));
121        } catch (NumberFormatException ignored) {
122        }
123
124        if (delay < 1) {
125          LOGGER.warn("Delay set too low or invalid, using default");
126        } else {
127          initialRefreshDelay = delay * 1000L;
128        }
129      }
130
131      if (commandLine.hasOption("numberOfIterations")) {
132        try {
133          numberOfIterations = Long.parseLong(commandLine.getOptionValue("numberOfIterations"));
134        } catch (NumberFormatException ignored) {
135          LOGGER.warn("The number of iterations set invalid, ignoring");
136        }
137      }
138
139      if (commandLine.hasOption("sortField")) {
140        String sortField = commandLine.getOptionValue("sortField");
141
142        String field;
143        boolean ascendingSort;
144        if (sortField.startsWith("+")) {
145          field = sortField.substring(1);
146          ascendingSort = false;
147        } else if (sortField.startsWith("-")) {
148          field = sortField.substring(1);
149          ascendingSort = true;
150        } else {
151          field = sortField;
152          ascendingSort = false;
153        }
154
155        Optional<FieldInfo> fieldInfo = initialMode.getFieldInfos().stream()
156          .filter(f -> f.getField().getHeader().equals(field)).findFirst();
157        if (fieldInfo.isPresent()) {
158          initialSortField = fieldInfo.get().getField();
159          initialAscendingSort = ascendingSort;
160        } else {
161          LOGGER.warn("The specified sort field " + field + " is not found, using default");
162        }
163      }
164
165      if (commandLine.hasOption("fields")) {
166        String[] fields = commandLine.getOptionValue("fields").split(",");
167        initialFields = new ArrayList<>();
168        for (String field : fields) {
169          Optional<FieldInfo> fieldInfo = initialMode.getFieldInfos().stream()
170            .filter(f -> f.getField().getHeader().equals(field)).findFirst();
171          if (fieldInfo.isPresent()) {
172            initialFields.add(fieldInfo.get().getField());
173          } else {
174            LOGGER.warn("The specified field " + field + " is not found, ignoring");
175          }
176        }
177      }
178
179      if (commandLine.hasOption("filters")) {
180        String[] filters = commandLine.getOptionValue("filters").split(",");
181        List<Field> fields = initialMode.getFieldInfos().stream().map(FieldInfo::getField)
182          .collect(Collectors.toList());
183        for (String filter : filters) {
184          RecordFilter f = RecordFilter.parse(filter, fields, false);
185          if (f != null) {
186            if (initialFilters == null) {
187              initialFilters = new ArrayList<>();
188            }
189            initialFilters.add(f);
190          } else {
191            LOGGER.warn("The specified filter " + filter + " is invalid, ignoring");
192          }
193        }
194      }
195
196      if (commandLine.hasOption("batchMode")) {
197        batchMode = true;
198      }
199    } catch (Exception e) {
200      LOGGER.error("Unable to parse options", e);
201      return 1;
202    }
203
204    try (Screen screen = new Screen(getConf(), initialRefreshDelay, initialMode, initialFields,
205      initialSortField, initialAscendingSort, initialFilters, numberOfIterations, batchMode)) {
206      screen.run();
207    }
208
209    return 0;
210  }
211
212  private Options getOptions() {
213    Options opts = new Options();
214    opts.addOption("h", "help", false, "Print usage; for help while the tool is running press 'h'");
215    opts.addOption("d", "delay", true, "The refresh delay (in seconds); default is 3 seconds");
216    opts.addOption("m", "mode", true,
217      "The mode; n (Namespace)|t (Table)|r (Region)|s (RegionServer)|u (User)"
218        + "|c (Client), default is r");
219    opts.addOption("n", "numberOfIterations", true, "The number of iterations");
220    opts.addOption("s", "sortField", true,
221      "The initial sort field. You can prepend a `+' or `-' to the field name to also override"
222        + " the sort direction. A leading `+' will force sorting high to low, whereas a `-' will"
223        + " ensure a low to high ordering");
224    opts.addOption("O", "outputFieldNames", false,
225      "Print each of the available field names on a separate line, then quit");
226    opts.addOption("f", "fields", true,
227      "Show only the given fields. Specify comma separated fields to show multiple fields");
228    opts.addOption("i", "filters", true,
229      "The initial filters. Specify comma separated filters to set multiple filters");
230    opts.addOption("b", "batchMode", false,
231      "Starts hbtop in Batch mode, which could be useful for sending output from hbtop to other"
232        + " programs or to a file. In this mode, hbtop will not accept input and runs until the"
233        + " iterations limit you've set with the `-n' command-line option or until killed");
234    return opts;
235  }
236
237  private void printUsage(Options opts) {
238    new HelpFormatter().printHelp("hbase hbtop [opts] [-D<property=value>]*", opts);
239    System.out.println("");
240    System.out.println(" Note: -D properties will be applied to the conf used.");
241    System.out.println("  For example:");
242    System.out.println("   -Dhbase.client.zookeeper.quorum=<zookeeper quorum>");
243    System.out.println("   -Dzookeeper.znode.parent=<znode parent>");
244    System.out.println("");
245  }
246
247  public static void main(String[] args) throws Exception {
248    int res = ToolRunner.run(new HBTop(), args);
249    System.exit(res);
250  }
251}
