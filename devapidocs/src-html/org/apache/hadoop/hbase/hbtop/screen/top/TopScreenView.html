037/**
038 * The screen that provides a dynamic real-time view for the HBase metrics.
039 *
040 * This shows the metric {@link Summary} and the metric {@link Record}s. The summary and the
041 * metrics are updated periodically (3 seconds by default).
042 */
043@InterfaceAudience.Private
044public class TopScreenView extends AbstractScreenView {
045
046  private static final int SUMMARY_START_ROW = 0;
047  private static final int SUMMARY_ROW_NUM = 7;
048  private static final int MESSAGE_ROW = 7;
049  private static final int RECORD_HEADER_ROW = 8;
050  private static final int RECORD_START_ROW = 9;
051
052  private final TopScreenPresenter topScreenPresenter;
053  private int pageSize;
054
055  public TopScreenView(Screen screen, Terminal terminal, long initialRefreshDelay, Admin admin,
056    Mode initialMode) {
057    super(screen, terminal);
058    this.topScreenPresenter = new TopScreenPresenter(this, initialRefreshDelay,
059      new TopScreenModel(admin, initialMode));
060  }
061
062  @Override
063  public void init() {
064    topScreenPresenter.init();
065    long delay = topScreenPresenter.refresh(true);
066    setTimer(delay);
067  }
068
069  @Override
070  public ScreenView handleTimer() {
071    long delay = topScreenPresenter.refresh(false);
072    setTimer(delay);
073    return this;
074  }
075
076  @Nullable
077  @Override
078  public ScreenView handleKeyPress(KeyPress keyPress) {
079    switch (keyPress.getType()) {
080      case Enter:
081        topScreenPresenter.refresh(true);
082        return this;
083
084      case ArrowUp:
085        topScreenPresenter.arrowUp();
086        return this;
087
088      case ArrowDown:
089        topScreenPresenter.arrowDown();
090        return this;
091
092      case ArrowLeft:
093        topScreenPresenter.arrowLeft();
094        return this;
095
096      case ArrowRight:
097        topScreenPresenter.arrowRight();
098        return this;
099
100      case PageUp:
101        topScreenPresenter.pageUp();
102        return this;
103
104      case PageDown:
105        topScreenPresenter.pageDown();
106        return this;
107
108      case Home:
109        topScreenPresenter.home();
110        return this;
111
112      case End:
113        topScreenPresenter.end();
114        return this;
115
116      case Escape:
117        return null;
118    }
119
120    if (keyPress.getType() != KeyPress.Type.Character) {
121      return unknownCommandMessage();
122    }
123
124    assert keyPress.getCharacter() != null;
125    switch (keyPress.getCharacter()) {
126      case 'R':
127        topScreenPresenter.switchSortOrder();
128        break;
129
130      case 'f':
131        cancelTimer();
132        return topScreenPresenter.transitionToFieldScreen(getScreen(), getTerminal());
133
134      case 'm':
135        cancelTimer();
136        return topScreenPresenter.transitionToModeScreen(getScreen(), getTerminal());
137
138      case 'h':
139        cancelTimer();
140        return topScreenPresenter.transitionToHelpScreen(getScreen(), getTerminal());
141
142      case 'd':
143        cancelTimer();
144        return topScreenPresenter.goToInputModeForRefreshDelay(getScreen(), getTerminal(),
145          MESSAGE_ROW);
146
147      case 'o':
148        cancelTimer();
149        if (keyPress.isCtrl()) {
150          return topScreenPresenter.goToFilterDisplayMode(getScreen(), getTerminal(), MESSAGE_ROW);
151        }
152        return topScreenPresenter.goToInputModeForFilter(getScreen(), getTerminal(), MESSAGE_ROW,
153          true);
154
155      case 'O':
156        cancelTimer();
157        return topScreenPresenter.goToInputModeForFilter(getScreen(), getTerminal(), MESSAGE_ROW,
158          false);
159
160      case '=':
161        topScreenPresenter.clearFilters();
162        break;
163
164      case 'X':
165        topScreenPresenter.adjustFieldLength();
166        break;
167
168      case 'i':
169        topScreenPresenter.drillDown();
170        break;
171
172      case 'q':
173        return null;
174
175      default:
176        return unknownCommandMessage();
177    }
178    return this;
179  }
180
181  public TerminalSize getTerminalSize() {
182    TerminalSize terminalSize = super.getTerminalSize();
183    updatePageSize(terminalSize);
184    return terminalSize;
185  }
186
187  public TerminalSize doResizeIfNecessary() {
188    TerminalSize terminalSize = super.doResizeIfNecessary();
189    if (terminalSize == null) {
190      return null;
191    }
192    updatePageSize(terminalSize);
193    return terminalSize;
194  }
195
196  private void updatePageSize(TerminalSize terminalSize) {
197    pageSize = terminalSize.getRows() - SUMMARY_ROW_NUM - 2;
198    if (pageSize < 0) {
199      pageSize = 0;
200    }
201  }
202
203  public int getPageSize() {
204    return pageSize;
205  }
206
207  public void showTopScreen(Summary summary, List<Header> headers, List<Record> records,
208    Record selectedRecord) {
209    showSummary(summary);
210    clearMessage();
211    showHeaders(headers);
212    showRecords(headers, records, selectedRecord);
213  }
214
215  private void showSummary(Summary summary) {
216    TerminalPrinter printer = getTerminalPrinter(SUMMARY_START_ROW);
217    printer.print(String.format("HBase hbtop - %s", summary.getCurrentTime())).endOfLine();
218    printer.print(String.format("Version: %s", summary.getVersion())).endOfLine();
219    printer.print(String.format("Cluster ID: %s", summary.getClusterId())).endOfLine();
220    printer.print("RegionServer(s): ")
221      .startBold().print(Integer.toString(summary.getServers())).stopBold()
222      .print(" total, ")
223      .startBold().print(Integer.toString(summary.getLiveServers())).stopBold()
224      .print(" live, ")
225      .startBold().print(Integer.toString(summary.getDeadServers())).stopBold()
226      .print(" dead").endOfLine();
227    printer.print("RegionCount: ")
228      .startBold().print(Integer.toString(summary.getRegionCount())).stopBold()
229      .print(" total, ")
230      .startBold().print(Integer.toString(summary.getRitCount())).stopBold()
231      .print(" rit").endOfLine();
232    printer.print("Average Cluster Load: ")
233      .startBold().print(String.format("%.2f", summary.getAverageLoad())).stopBold().endOfLine();
234    printer.print("Aggregate Request/s: ")
235      .startBold().print(Long.toString(summary.getAggregateRequestPerSecond())).stopBold()
236      .endOfLine();
237  }
238
239  private void showRecords(List<Header> headers, List<Record> records, Record selectedRecord) {
240    TerminalPrinter printer = getTerminalPrinter(RECORD_START_ROW);
241    List<String> buf = new ArrayList<>(headers.size());
242    for (int i = 0; i < pageSize; i++) {
243      if(i < records.size()) {
244        Record record = records.get(i);
245        buf.clear();
246        for (Header header : headers) {
247          String value = "";
248          if (record.containsKey(header.getField())) {
249            value = record.get(header.getField()).asString();
250          }
251
252          buf.add(limitLineLength(String.format(header.format(), value), header.getLength()));
253        }
254
255        String recordString = String.join(" ", buf);
256        if (!recordString.isEmpty()) {
257          recordString += " ";
258        }
259
260        if (record == selectedRecord) {
261          printer.startHighlight().print(recordString).stopHighlight().endOfLine();
262        } else {
263          printer.print(recordString).endOfLine();
264        }
265      } else {
266        printer.endOfLine();
267      }
268    }
269  }
270
271  private void showHeaders(List<Header> headers) {
272    String header = headers.stream()
273      .map(h -> String.format(h.format(), h.getField().getHeader()))
274      .collect(Collectors.joining(" "));
275
276    if (!header.isEmpty()) {
277      header += " ";
278    }
279
280    getTerminalPrinter(RECORD_HEADER_ROW).startHighlight().print(header).stopHighlight()
281      .endOfLine();
282  }
283
284  private String limitLineLength(String line, int length) {
285    if (line.length() > length) {
286      return line.substring(0, length - 1) + "+";
287    }
288    return line;
289  }
290
291  private void clearMessage() {
292    getTerminalPrinter(MESSAGE_ROW).print("").endOfLine();
293  }
294
295  private ScreenView unknownCommandMessage() {
296    cancelTimer();
297    return topScreenPresenter.goToMessageMode(getScreen(), getTerminal(), MESSAGE_ROW,
298      "Unknown command - try 'h' for help");
299  }
