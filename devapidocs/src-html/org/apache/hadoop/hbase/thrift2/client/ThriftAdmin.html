225  @Override
226  public void createTable(TableDescriptor desc, byte[] startKey, byte[] endKey, int numRegions)
227      throws IOException {
228    if(numRegions < 3) {
229      throw new IllegalArgumentException("Must create at least three regions");
230    } else if(Bytes.compareTo(startKey, endKey) >= 0) {
231      throw new IllegalArgumentException("Start key must be smaller than end key");
232    }
233    if (numRegions == 3) {
234      createTable(desc, new byte[][]{startKey, endKey});
235      return;
236    }
237    byte [][] splitKeys = Bytes.split(startKey, endKey, numRegions - 3);
238    if(splitKeys == null || splitKeys.length != numRegions - 1) {
239      throw new IllegalArgumentException("Unable to split key range into enough regions");
240    }
241    createTable(desc, splitKeys);
242  }

246    TTableDescriptor tTableDescriptor = ThriftUtilities.tableDescriptorFromHBase(desc);
247    List<ByteBuffer> splitKeyInBuffer = ThriftUtilities.splitKeyFromHBase(splitKeys);
248    try {
249      client.createTable(tTableDescriptor, splitKeyInBuffer);
250    } catch (TException e) {
251      throw new IOException(e);
252    }
