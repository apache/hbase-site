018package org.apache.hadoop.hbase.namequeues;
019
020import com.lmax.disruptor.BlockingWaitStrategy;
021import com.lmax.disruptor.RingBuffer;
022import com.lmax.disruptor.dsl.Disruptor;
023import com.lmax.disruptor.dsl.ProducerType;
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.namequeues.request.NamedQueueGetRequest;
026import org.apache.hadoop.hbase.namequeues.response.NamedQueueGetResponse;
027import org.apache.hadoop.hbase.util.Threads;
028import org.apache.yetus.audience.InterfaceAudience;
029import org.apache.yetus.audience.InterfaceStability;
030
031import org.apache.hbase.thirdparty.com.google.common.base.Preconditions;
032import org.apache.hbase.thirdparty.com.google.common.util.concurrent.ThreadFactoryBuilder;
033
034/**
035 * NamedQueue recorder that maintains various named queues. The service uses LMAX Disruptor to save
036 * queue records which are then consumed by a queue and based on the ring buffer size, the available
037 * records are then fetched from the queue in thread-safe manner.
038 */
039@InterfaceAudience.Private
040@InterfaceStability.Evolving
041public class NamedQueueRecorder {
042
043  private final Disruptor<RingBufferEnvelope> disruptor;
044  private final LogEventHandler logEventHandler;
045
046  private static volatile NamedQueueRecorder namedQueueRecorder;
047  private static boolean isInit = false;
048  private static final Object LOCK = new Object();
049
050  /**
051   * Initialize disruptor with configurable ringbuffer size
052   */
053  private NamedQueueRecorder(Configuration conf) {
054
055    // This is the 'writer' -- a single threaded executor. This single thread consumes what is
056    // put on the ringbuffer.
057    final String hostingThreadName = Thread.currentThread().getName();
058
059    int eventCount = conf.getInt("hbase.namedqueue.ringbuffer.size", 1024);
060
061    // disruptor initialization with BlockingWaitStrategy
062    this.disruptor = new Disruptor<>(RingBufferEnvelope::new, getEventCount(eventCount),
063      new ThreadFactoryBuilder().setNameFormat(hostingThreadName + ".slowlog.append-pool-%d")
064        .setDaemon(true).setUncaughtExceptionHandler(Threads.LOGGING_EXCEPTION_HANDLER).build(),
065      ProducerType.MULTI, new BlockingWaitStrategy());
066    this.disruptor.setDefaultExceptionHandler(new DisruptorExceptionHandler());
067
068    // initialize ringbuffer event handler
069    this.logEventHandler = new LogEventHandler(conf);
070    this.disruptor.handleEventsWith(new LogEventHandler[] { this.logEventHandler });
071    this.disruptor.start();
072  }
073
074  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "MS_EXPOSE_REP",
075      justification = "singleton pattern")
076  public static NamedQueueRecorder getInstance(Configuration conf) {
077    if (namedQueueRecorder != null) {
078      return namedQueueRecorder;
079    }
080    synchronized (LOCK) {
081      if (!isInit) {
082        namedQueueRecorder = new NamedQueueRecorder(conf);
083        isInit = true;
084      }
085    }
086    return namedQueueRecorder;
087  }
088
089  // must be power of 2 for disruptor ringbuffer
090  private int getEventCount(int eventCount) {
091    Preconditions.checkArgument(eventCount >= 0, "hbase.namedqueue.ringbuffer.size must be > 0");
092    int floor = Integer.highestOneBit(eventCount);
093    if (floor == eventCount) {
094      return floor;
095    }
096    // max capacity is 1 << 30
097    if (floor >= 1 << 29) {
098      return 1 << 30;
099    }
100    return floor << 1;
101  }
102
103  /**
104   * Retrieve in memory queue records from ringbuffer
105   * @param request namedQueue request with event type
106   * @return queue records from ringbuffer after filter (if applied)
107   */
108  public NamedQueueGetResponse getNamedQueueRecords(NamedQueueGetRequest request) {
109    return this.logEventHandler.getNamedQueueRecords(request);
110  }
111
112  /**
113   * clears queue records from ringbuffer
114   * @param namedQueueEvent type of queue to clear
115   * @return true if slow log payloads are cleaned up or hbase.regionserver.slowlog.buffer.enabled
116   *         is not set to true, false if failed to clean up slow logs
117   */
118  public boolean clearNamedQueue(NamedQueuePayload.NamedQueueEvent namedQueueEvent) {
119    return this.logEventHandler.clearNamedQueue(namedQueueEvent);
120  }
121
122  /**
123   * Add various NamedQueue records to ringbuffer. Based on the type of the event (e.g slowLog),
124   * consumer of disruptor ringbuffer will have specific logic. This method is producer of disruptor
125   * ringbuffer which is initialized in NamedQueueRecorder constructor.
126   * @param namedQueuePayload namedQueue payload sent by client of ring buffer service
127   */
128  public void addRecord(NamedQueuePayload namedQueuePayload) {
129    RingBuffer<RingBufferEnvelope> ringBuffer = this.disruptor.getRingBuffer();
130    long seqId = ringBuffer.next();
131    try {
132      ringBuffer.get(seqId).load(namedQueuePayload);
133    } finally {
134      ringBuffer.publish(seqId);
135    }
136  }
137
138  /**
139   * Add all in memory queue records to system table. The implementors can use system table or
140   * direct HDFS file or ZK as persistence system.
141   */
142  public void persistAll(NamedQueuePayload.NamedQueueEvent namedQueueEvent) {
143    if (this.logEventHandler != null) {
144      this.logEventHandler.persistAll(namedQueueEvent);
145    }
146  }
147
148}
