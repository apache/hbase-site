053@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
054public class MasterFlushTableProcedureManager extends MasterProcedureManager {
055
056  public static final String FLUSH_TABLE_PROCEDURE_SIGNATURE = "flush-table-proc";
057
058  private static final String FLUSH_TIMEOUT_MILLIS_KEY = "hbase.flush.master.timeoutMillis";
059  private static final int FLUSH_TIMEOUT_MILLIS_DEFAULT = 60000;
060  private static final String FLUSH_WAKE_MILLIS_KEY = "hbase.flush.master.wakeMillis";
061  private static final int FLUSH_WAKE_MILLIS_DEFAULT = 500;
062
063  private static final String FLUSH_PROC_POOL_THREADS_KEY =
064      "hbase.flush.procedure.master.threads";
065  private static final int FLUSH_PROC_POOL_THREADS_DEFAULT = 1;
066
067  private static final Log LOG = LogFactory.getLog(MasterFlushTableProcedureManager.class);
068
069  private MasterServices master;
070  private ProcedureCoordinator coordinator;
071  private Map<TableName, Procedure> procMap = new HashMap<>();
072  private boolean stopped;
073
074  public MasterFlushTableProcedureManager() {};
075
076  @Override
077  public void stop(String why) {
078    LOG.info("stop: " + why);
079    this.stopped = true;
080  }
081
082  @Override
083  public boolean isStopped() {
084    return this.stopped;
085  }
086
087  @Override
088  public void initialize(MasterServices master, MetricsMaster metricsMaster)
089      throws KeeperException, IOException, UnsupportedOperationException {
090    this.master = master;
091
092    // get the configuration for the coordinator
093    Configuration conf = master.getConfiguration();
094    long wakeFrequency = conf.getInt(FLUSH_WAKE_MILLIS_KEY, FLUSH_WAKE_MILLIS_DEFAULT);
095    long timeoutMillis = conf.getLong(FLUSH_TIMEOUT_MILLIS_KEY, FLUSH_TIMEOUT_MILLIS_DEFAULT);
096    int threads = conf.getInt(FLUSH_PROC_POOL_THREADS_KEY, FLUSH_PROC_POOL_THREADS_DEFAULT);
097
098    // setup the procedure coordinator
099    String name = master.getServerName().toString();
100    ThreadPoolExecutor tpool = ProcedureCoordinator.defaultPool(name, threads);
101    ProcedureCoordinatorRpcs comms = new ZKProcedureCoordinator(
102        master.getZooKeeper(), getProcedureSignature(), name);
103
104    this.coordinator = new ProcedureCoordinator(comms, tpool, timeoutMillis, wakeFrequency);
105  }
106
107  @Override
108  public String getProcedureSignature() {
109    return FLUSH_TABLE_PROCEDURE_SIGNATURE;
110  }
111
112  @Override
113  public void execProcedure(ProcedureDescription desc) throws IOException {
114
115    TableName tableName = TableName.valueOf(desc.getInstance());
116
117    // call pre coproc hook
118    MasterCoprocessorHost cpHost = master.getMasterCoprocessorHost();
119    if (cpHost != null) {
120      cpHost.preTableFlush(tableName);
121    }
122
123    // Get the list of region servers that host the online regions for table.
124    // We use the procedure instance name to carry the table name from the client.
125    // It is possible that regions may move after we get the region server list.
126    // Each region server will get its own online regions for the table.
127    // We may still miss regions that need to be flushed.
128    List<Pair<HRegionInfo, ServerName>> regionsAndLocations;
129
130    if (TableName.META_TABLE_NAME.equals(tableName)) {
131      regionsAndLocations = new MetaTableLocator().getMetaRegionsAndLocations(
132        master.getZooKeeper());
133    } else {
134      regionsAndLocations = MetaTableAccessor.getTableRegionsAndLocations(
135        master.getConnection(), tableName, false);
136    }
137
138    Set<String> regionServers = new HashSet<>(regionsAndLocations.size());
139    for (Pair<HRegionInfo, ServerName> region : regionsAndLocations) {
140      if (region != null && region.getFirst() != null && region.getSecond() != null) {
141        HRegionInfo hri = region.getFirst();
142        if (hri.isOffline() && (hri.isSplit() || hri.isSplitParent())) continue;
143        regionServers.add(region.getSecond().toString());
144      }
145    }
146
147    ForeignExceptionDispatcher monitor = new ForeignExceptionDispatcher(desc.getInstance());
148
149    // Kick of the global procedure from the master coordinator to the region servers.
150    // We rely on the existing Distributed Procedure framework to prevent any concurrent
151    // procedure with the same name.
152    Procedure proc = coordinator.startProcedure(monitor, desc.getInstance(),
153      new byte[0], Lists.newArrayList(regionServers));
154    monitor.rethrowException();
155    if (proc == null) {
156      String msg = "Failed to submit distributed procedure " + desc.getSignature() + " for '"
157          + desc.getInstance() + "'. " + "Another flush procedure is running?";
158      LOG.error(msg);
159      throw new IOException(msg);
160    }
161
162    procMap.put(tableName, proc);
163
164    try {
165      // wait for the procedure to complete.  A timer thread is kicked off that should cancel this
166      // if it takes too long.
167      proc.waitForCompleted();
168      LOG.info("Done waiting - exec procedure " + desc.getSignature() + " for '"
169          + desc.getInstance() + "'");
170      LOG.info("Master flush table procedure is successful!");
171    } catch (InterruptedException e) {
172      ForeignException ee =
173          new ForeignException("Interrupted while waiting for flush table procdure to finish", e);
174      monitor.receive(ee);
175      Thread.currentThread().interrupt();
176    } catch (ForeignException e) {
177      ForeignException ee =
178          new ForeignException("Exception while waiting for flush table procdure to finish", e);
179      monitor.receive(ee);
180    }
181    monitor.rethrowException();
182  }
183
184  @Override
185  public synchronized boolean isProcedureDone(ProcedureDescription desc) throws IOException {
186    // Procedure instance name is the table name.
187    TableName tableName = TableName.valueOf(desc.getInstance());
188    Procedure proc = procMap.get(tableName);
189    if (proc == null) {
190      // The procedure has not even been started yet.
191      // The client would request the procedure and call isProcedureDone().
192      // The HBaseAdmin.execProcedure() wraps both request and isProcedureDone().
193      return false;
194    }
195    // We reply on the existing Distributed Procedure framework to give us the status.
196    return proc.isCompleted();
197  }
