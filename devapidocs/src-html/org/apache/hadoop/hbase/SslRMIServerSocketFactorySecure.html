011package org.apache.hadoop.hbase;
012
013import java.io.IOException;
014import java.net.ServerSocket;
015import java.net.Socket;
016import java.util.ArrayList;
017
018import javax.net.ssl.SSLSocket;
019import javax.net.ssl.SSLSocketFactory;
020import javax.rmi.ssl.SslRMIServerSocketFactory;
021
022/**
023 * Avoid SSL V3.0 "Poodle" Vulnerability - CVE-2014-3566
024 */
025public class SslRMIServerSocketFactorySecure extends SslRMIServerSocketFactory {
026  // If you add more constructors, you may have to change the rest of this implementation,
027  // which assumes an empty constructor, i.e. there are no specially enabled protocols or
028  // cipher suites on this RMI factory nor a provided SSLContext
029  public SslRMIServerSocketFactorySecure() {
030    super();
031  }
032
033  @Override
034  public ServerSocket createServerSocket(int port) throws IOException {
035    return new ServerSocket(port) {
036      public Socket accept() throws IOException {
037        Socket socket = super.accept();
038        SSLSocketFactory sslSocketFactory = (SSLSocketFactory) SSLSocketFactory.getDefault();
039        SSLSocket sslSocket =
040            (SSLSocket) sslSocketFactory.createSocket(socket,
041              socket.getInetAddress().getHostName(), socket.getPort(), true);
042        sslSocket.setUseClientMode(false);
043        sslSocket.setNeedClientAuth(false);
044
045        ArrayList<String> secureProtocols = new ArrayList<>();
046        for (String p : sslSocket.getEnabledProtocols()) {
047          if (!p.contains("SSLv3")) {
048            secureProtocols.add(p);
049          }
050        }
051        sslSocket.setEnabledProtocols(secureProtocols.toArray(new String[secureProtocols.size()]));
052
053        return sslSocket;
054      }
055    };
056  }
057}
