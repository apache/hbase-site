019package org.apache.hadoop.hbase.codec.prefixtree.encode.column;
020
021import java.io.IOException;
022import java.io.OutputStream;
023
024import org.apache.yetus.audience.InterfaceAudience;
025import org.apache.hadoop.hbase.codec.prefixtree.PrefixTreeBlockMeta;
026import org.apache.hadoop.hbase.codec.prefixtree.encode.other.ColumnNodeType;
027import org.apache.hadoop.hbase.codec.prefixtree.encode.tokenize.TokenizerNode;
028import org.apache.hadoop.hbase.util.ByteRange;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.apache.hadoop.hbase.util.Strings;
031import org.apache.hadoop.hbase.util.vint.UFIntTool;
032import org.apache.hadoop.hbase.util.vint.UVIntTool;
033
034/**
035 * <p>
036 * Column nodes can be either family nodes or qualifier nodes, as both sections encode similarly.
037 * The family and qualifier sections of the data block are made of 1 or more of these nodes.
038 * </p>
039 * Each node is composed of 3 sections:<br>
040 * <ul>
041 * <li>tokenLength: UVInt (normally 1 byte) indicating the number of token bytes
042 * <li>token[]: the actual token bytes
043 * <li>parentStartPosition: the offset of the next node from the start of the family or qualifier
044 * section
045 * </ul>
046 */
047@InterfaceAudience.Private
048public class ColumnNodeWriter{
049
050  /************* fields ****************************/
051
052  protected TokenizerNode builderNode;
053  protected PrefixTreeBlockMeta blockMeta;
054
055  protected int tokenLength;
056  protected byte[] token;
057  protected int parentStartPosition;
058  protected ColumnNodeType nodeType;
059
060
061  /*************** construct **************************/
062
063  public ColumnNodeWriter(PrefixTreeBlockMeta blockMeta, TokenizerNode builderNode,
064      ColumnNodeType nodeType) {
065    this.blockMeta = blockMeta;
066    this.builderNode = builderNode;
067    this.nodeType = nodeType;
068    calculateTokenLength();
069  }
070
071
072  /************* methods *******************************/
073
074  public boolean isRoot() {
075    return parentStartPosition == 0;
076  }
077
078  private void calculateTokenLength() {
079    tokenLength = builderNode.getTokenLength();
080    token = new byte[tokenLength];
081  }
082
083  /**
084   * This method is called before blockMeta.qualifierOffsetWidth is known, so we pass in a
085   * placeholder.
086   * @param offsetWidthPlaceholder the placeholder
087   * @return node width
088   */
089  public int getWidthUsingPlaceholderForOffsetWidth(int offsetWidthPlaceholder) {
090    int width = 0;
091    width += UVIntTool.numBytes(tokenLength);
092    width += token.length;
093    width += offsetWidthPlaceholder;
094    return width;
095  }
096
097  public void writeBytes(OutputStream os) throws IOException {
098    int parentOffsetWidth;
099    if (this.nodeType == ColumnNodeType.FAMILY) {
100      parentOffsetWidth = blockMeta.getFamilyOffsetWidth();
101    } else if (this.nodeType == ColumnNodeType.QUALIFIER) {
102      parentOffsetWidth = blockMeta.getQualifierOffsetWidth();
103    } else {
104      parentOffsetWidth = blockMeta.getTagsOffsetWidth();
105    }
106    UVIntTool.writeBytes(tokenLength, os);
107    os.write(token);
108    UFIntTool.writeBytes(parentOffsetWidth, parentStartPosition, os);
109  }
110
111  public void setTokenBytes(ByteRange source) {
112    source.deepCopySubRangeTo(0, tokenLength, token, 0);
113  }
114
115
116  /****************** standard methods ************************/
117
118  @Override
119  public String toString() {
120    StringBuilder sb = new StringBuilder();
121    sb.append(Strings.padFront(builderNode.getOutputArrayOffset() + "", ' ', 3) + ",");
122    sb.append("[");
123    sb.append(Bytes.toString(token));
124    sb.append("]->");
125    sb.append(parentStartPosition);
126    return sb.toString();
127  }
128
129
130  /************************** get/set ***********************/
131
132  public void setParentStartPosition(int parentStartPosition) {
133    this.parentStartPosition = parentStartPosition;
134  }
135
136}
