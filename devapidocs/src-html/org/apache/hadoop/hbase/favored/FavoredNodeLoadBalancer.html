051/**
052 * An implementation of the {@link org.apache.hadoop.hbase.master.LoadBalancer} that
053 * assigns favored nodes for each region. There is a Primary RegionServer that hosts
054 * the region, and then there is Secondary and Tertiary RegionServers. Currently, the
055 * favored nodes information is used in creating HDFS files - the Primary RegionServer
056 * passes the primary, secondary, tertiary node addresses as hints to the
057 * DistributedFileSystem API for creating files on the filesystem. These nodes are
058 * treated as hints by the HDFS to place the blocks of the file. This alleviates the
059 * problem to do with reading from remote nodes (since we can make the Secondary
060 * RegionServer as the new Primary RegionServer) after a region is recovered. This
061 * should help provide consistent read latencies for the regions even when their
062 * primary region servers die.
063 *
064 */
065@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
066public class FavoredNodeLoadBalancer extends BaseLoadBalancer implements FavoredNodesPromoter {
067  private static final Log LOG = LogFactory.getLog(FavoredNodeLoadBalancer.class);
068
069  private RackManager rackManager;
070  private Configuration conf;
071  private FavoredNodesManager fnm;

087  @Override
088  public List<RegionPlan> balanceCluster(Map<ServerName, List<HRegionInfo>> clusterState)  {
089    //TODO. Look at is whether Stochastic loadbalancer can be integrated with this
090    List<RegionPlan> plans = new ArrayList<>();
091    //perform a scan of the meta to get the latest updates (if any)
092    SnapshotOfRegionAssignmentFromMeta snaphotOfRegionAssignment =
093        new SnapshotOfRegionAssignmentFromMeta(super.services.getConnection());
094    try {
095      snaphotOfRegionAssignment.initialize();
096    } catch (IOException ie) {
097      LOG.warn("Not running balancer since exception was thrown " + ie);
098      return plans;
099    }
100    // This is not used? Findbugs says so: Map<ServerName, ServerName> serverNameToServerNameWithoutCode = new HashMap<>();
101    Map<ServerName, ServerName> serverNameWithoutCodeToServerName = new HashMap<>();
102    ServerManager serverMgr = super.services.getServerManager();
103    for (ServerName sn: serverMgr.getOnlineServersList()) {
104      ServerName s = ServerName.valueOf(sn.getHostname(), sn.getPort(), ServerName.NON_STARTCODE);
105      // FindBugs complains about useless store! serverNameToServerNameWithoutCode.put(sn, s);
106      serverNameWithoutCodeToServerName.put(s, sn);
107    }
108    for (Map.Entry<ServerName, List<HRegionInfo>> entry : clusterState.entrySet()) {
109      ServerName currentServer = entry.getKey();
110      //get a server without the startcode for the currentServer
111      ServerName currentServerWithoutStartCode = ServerName.valueOf(currentServer.getHostname(),
112          currentServer.getPort(), ServerName.NON_STARTCODE);
113      List<HRegionInfo> list = entry.getValue();
114      for (HRegionInfo region : list) {
115        if(!FavoredNodesManager.isFavoredNodeApplicable(region)) {
116          continue;
117        }
118        List<ServerName> favoredNodes = fnm.getFavoredNodes(region);
119        if (favoredNodes == null || favoredNodes.get(0).equals(currentServerWithoutStartCode)) {
120          continue; //either favorednodes does not exist or we are already on the primary node
121        }
122        ServerName destination = null;
123        //check whether the primary is available
124        destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(0));
125        if (destination == null) {
126          //check whether the region is on secondary/tertiary
127          if (currentServerWithoutStartCode.equals(favoredNodes.get(1)) ||
128              currentServerWithoutStartCode.equals(favoredNodes.get(2))) {
129            continue;
130          }
131          //the region is currently on none of the favored nodes
132          //get it on one of them if possible
133          ServerLoad l1 = super.services.getServerManager().getLoad(
134              serverNameWithoutCodeToServerName.get(favoredNodes.get(1)));
135          ServerLoad l2 = super.services.getServerManager().getLoad(
136              serverNameWithoutCodeToServerName.get(favoredNodes.get(2)));
137          if (l1 != null && l2 != null) {
138            if (l1.getLoad() > l2.getLoad()) {
139              destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(2));
140            } else {
141              destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(1));
142            }
143          } else if (l1 != null) {
144            destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(1));
145          } else if (l2 != null) {
146            destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(2));
147          }
148        }
149
150        if (destination != null) {
151          RegionPlan plan = new RegionPlan(region, currentServer, destination);
152          plans.add(plan);
153        }
154      }
155    }
156    return plans;
157  }

159  @Override
160  public Map<ServerName, List<HRegionInfo>> roundRobinAssignment(List<HRegionInfo> regions,
161      List<ServerName> servers) throws HBaseIOException {
162    Map<ServerName, List<HRegionInfo>> assignmentMap;
163    try {
164      FavoredNodeAssignmentHelper assignmentHelper =
165          new FavoredNodeAssignmentHelper(servers, rackManager);
166      assignmentHelper.initialize();
167      if (!assignmentHelper.canPlaceFavoredNodes()) {
168        return super.roundRobinAssignment(regions, servers);
169      }
170      // Segregate the regions into two types:
171      // 1. The regions that have favored node assignment, and where at least
172      //    one of the favored node is still alive. In this case, try to adhere
173      //    to the current favored nodes assignment as much as possible - i.e.,
174      //    if the current primary is gone, then make the secondary or tertiary
175      //    as the new host for the region (based on their current load).
176      //    Note that we don't change the favored
177      //    node assignments here (even though one or more favored node is currently
178      //    down). It is up to the balanceCluster to do this hard work. The HDFS
179      //    can handle the fact that some nodes in the favored nodes hint is down
180      //    It'd allocate some other DNs. In combination with stale settings for HDFS,
181      //    we should be just fine.
182      // 2. The regions that currently don't have favored node assignment. We will
183      //    need to come up with favored nodes assignments for them. The corner case
184      //    in (1) above is that all the nodes are unavailable and in that case, we
185      //    will note that this region doesn't have favored nodes.
186      Pair<Map<ServerName,List<HRegionInfo>>, List<HRegionInfo>> segregatedRegions =
187          segregateRegionsAndAssignRegionsWithFavoredNodes(regions, servers);
188      Map<ServerName,List<HRegionInfo>> regionsWithFavoredNodesMap = segregatedRegions.getFirst();
189      List<HRegionInfo> regionsWithNoFavoredNodes = segregatedRegions.getSecond();
190      assignmentMap = new HashMap<>();
191      roundRobinAssignmentImpl(assignmentHelper, assignmentMap, regionsWithNoFavoredNodes,
192          servers);
193      // merge the assignment maps
194      assignmentMap.putAll(regionsWithFavoredNodesMap);
195    } catch (Exception ex) {
196      LOG.warn("Encountered exception while doing favored-nodes assignment " + ex +
197          " Falling back to regular assignment");
198      assignmentMap = super.roundRobinAssignment(regions, servers);
199    }
200    return assignmentMap;
201  }

203  @Override
204  public ServerName randomAssignment(HRegionInfo regionInfo, List<ServerName> servers)
205      throws HBaseIOException {
206    try {
207      FavoredNodeAssignmentHelper assignmentHelper =
208          new FavoredNodeAssignmentHelper(servers, rackManager);
209      assignmentHelper.initialize();
210      ServerName primary = super.randomAssignment(regionInfo, servers);
211      if (!FavoredNodesManager.isFavoredNodeApplicable(regionInfo)
212          || !assignmentHelper.canPlaceFavoredNodes()) {
213        return primary;
214      }
215      List<ServerName> favoredNodes = fnm.getFavoredNodes(regionInfo);
216      // check if we have a favored nodes mapping for this region and if so, return
217      // a server from the favored nodes list if the passed 'servers' contains this
218      // server as well (available servers, that is)
219      if (favoredNodes != null) {
220        for (ServerName s : favoredNodes) {
221          ServerName serverWithLegitStartCode = availableServersContains(servers, s);
222          if (serverWithLegitStartCode != null) {
223            return serverWithLegitStartCode;
224          }
225        }
226      }
227      List<HRegionInfo> regions = new ArrayList<>(1);
228      regions.add(regionInfo);
229      Map<HRegionInfo, ServerName> primaryRSMap = new HashMap<>(1);
230      primaryRSMap.put(regionInfo, primary);
231      assignSecondaryAndTertiaryNodesForRegion(assignmentHelper, regions, primaryRSMap);
232      return primary;
233    } catch (Exception ex) {
234      LOG.warn("Encountered exception while doing favored-nodes (random)assignment " + ex +
235          " Falling back to regular assignment");
236      return super.randomAssignment(regionInfo, servers);
237    }
238  }

240  private Pair<Map<ServerName, List<HRegionInfo>>, List<HRegionInfo>>
241  segregateRegionsAndAssignRegionsWithFavoredNodes(List<HRegionInfo> regions,
242      List<ServerName> availableServers) {
243    Map<ServerName, List<HRegionInfo>> assignmentMapForFavoredNodes = new HashMap<>(regions.size() / 2);
244    List<HRegionInfo> regionsWithNoFavoredNodes = new ArrayList<>(regions.size()/2);
245    for (HRegionInfo region : regions) {
246      List<ServerName> favoredNodes = fnm.getFavoredNodes(region);
247      ServerName primaryHost = null;
248      ServerName secondaryHost = null;
249      ServerName tertiaryHost = null;
250      if (favoredNodes != null) {
251        for (ServerName s : favoredNodes) {
252          ServerName serverWithLegitStartCode = availableServersContains(availableServers, s);
253          if (serverWithLegitStartCode != null) {
254            FavoredNodesPlan.Position position =
255                FavoredNodesPlan.getFavoredServerPosition(favoredNodes, s);
256            if (Position.PRIMARY.equals(position)) {
257              primaryHost = serverWithLegitStartCode;
258            } else if (Position.SECONDARY.equals(position)) {
259              secondaryHost = serverWithLegitStartCode;
260            } else if (Position.TERTIARY.equals(position)) {
261              tertiaryHost = serverWithLegitStartCode;
262            }
263          }
264        }
265        assignRegionToAvailableFavoredNode(assignmentMapForFavoredNodes, region,
266              primaryHost, secondaryHost, tertiaryHost);
267      }
268      if (primaryHost == null && secondaryHost == null && tertiaryHost == null) {
269        //all favored nodes unavailable
270        regionsWithNoFavoredNodes.add(region);
271      }
272    }
273    return new Pair<>(assignmentMapForFavoredNodes, regionsWithNoFavoredNodes);
274  }

276  // Do a check of the hostname and port and return the servername from the servers list
277  // that matched (the favoredNode will have a startcode of -1 but we want the real
278  // server with the legit startcode
279  private ServerName availableServersContains(List<ServerName> servers, ServerName favoredNode) {
280    for (ServerName server : servers) {
281      if (ServerName.isSameAddress(favoredNode, server)) {
282        return server;
283      }
284    }
285    return null;
286  }

288  private void assignRegionToAvailableFavoredNode(Map<ServerName,
289      List<HRegionInfo>> assignmentMapForFavoredNodes, HRegionInfo region, ServerName primaryHost,
290      ServerName secondaryHost, ServerName tertiaryHost) {
291    if (primaryHost != null) {
292      addRegionToMap(assignmentMapForFavoredNodes, region, primaryHost);
293    } else if (secondaryHost != null && tertiaryHost != null) {
294      // assign the region to the one with a lower load
295      // (both have the desired hdfs blocks)
296      ServerName s;
297      ServerLoad tertiaryLoad = super.services.getServerManager().getLoad(tertiaryHost);
298      ServerLoad secondaryLoad = super.services.getServerManager().getLoad(secondaryHost);
299      if (secondaryLoad.getLoad() < tertiaryLoad.getLoad()) {
300        s = secondaryHost;
301      } else {
302        s = tertiaryHost;
303      }
304      addRegionToMap(assignmentMapForFavoredNodes, region, s);
305    } else if (secondaryHost != null) {
306      addRegionToMap(assignmentMapForFavoredNodes, region, secondaryHost);
307    } else if (tertiaryHost != null) {
308      addRegionToMap(assignmentMapForFavoredNodes, region, tertiaryHost);
309    }
310  }
311
312  private void addRegionToMap(Map<ServerName, List<HRegionInfo>> assignmentMapForFavoredNodes,
313      HRegionInfo region, ServerName host) {
314    List<HRegionInfo> regionsOnServer = null;
315    if ((regionsOnServer = assignmentMapForFavoredNodes.get(host)) == null) {
316      regionsOnServer = new ArrayList<>();
317      assignmentMapForFavoredNodes.put(host, regionsOnServer);
318    }
319    regionsOnServer.add(region);
320  }

326  private void roundRobinAssignmentImpl(FavoredNodeAssignmentHelper assignmentHelper,
327      Map<ServerName, List<HRegionInfo>> assignmentMap,
328      List<HRegionInfo> regions, List<ServerName> servers) throws IOException {
329    Map<HRegionInfo, ServerName> primaryRSMap = new HashMap<>();
330    // figure the primary RSs
331    assignmentHelper.placePrimaryRSAsRoundRobin(assignmentMap, primaryRSMap, regions);
332    assignSecondaryAndTertiaryNodesForRegion(assignmentHelper, regions, primaryRSMap);
333  }
334
335  private void assignSecondaryAndTertiaryNodesForRegion(
336      FavoredNodeAssignmentHelper assignmentHelper,
337      List<HRegionInfo> regions, Map<HRegionInfo, ServerName> primaryRSMap) throws IOException {
338    // figure the secondary and tertiary RSs
339    Map<HRegionInfo, ServerName[]> secondaryAndTertiaryRSMap =
340        assignmentHelper.placeSecondaryAndTertiaryRS(primaryRSMap);
341
342    Map<HRegionInfo, List<ServerName>> regionFNMap = Maps.newHashMap();
343    // now record all the assignments so that we can serve queries later
344    for (HRegionInfo region : regions) {
345      // Store the favored nodes without startCode for the ServerName objects
346      // We don't care about the startcode; but only the hostname really
347      List<ServerName> favoredNodesForRegion = new ArrayList<>(3);
348      ServerName sn = primaryRSMap.get(region);
349      favoredNodesForRegion.add(ServerName.valueOf(sn.getHostname(), sn.getPort(),
350          ServerName.NON_STARTCODE));
351      ServerName[] secondaryAndTertiaryNodes = secondaryAndTertiaryRSMap.get(region);
352      if (secondaryAndTertiaryNodes != null) {
353        favoredNodesForRegion.add(ServerName.valueOf(secondaryAndTertiaryNodes[0].getHostname(),
354            secondaryAndTertiaryNodes[0].getPort(), ServerName.NON_STARTCODE));
355        favoredNodesForRegion.add(ServerName.valueOf(secondaryAndTertiaryNodes[1].getHostname(),
356            secondaryAndTertiaryNodes[1].getPort(), ServerName.NON_STARTCODE));
357      }
358      regionFNMap.put(region, favoredNodesForRegion);
359    }
360    fnm.updateFavoredNodes(regionFNMap);
361  }

363  /*
364   * Generate Favored Nodes for daughters during region split.
365   *
366   * If the parent does not have FN, regenerates them for the daughters.
367   *
368   * If the parent has FN, inherit two FN from parent for each daughter and generate the remaining.
369   * The primary FN for both the daughters should be the same as parent. Inherit the secondary
370   * FN from the parent but keep it different for each daughter. Choose the remaining FN
371   * randomly. This would give us better distribution over a period of time after enough splits.
372   */
373  @Override
374  public void generateFavoredNodesForDaughter(List<ServerName> servers, HRegionInfo parent,
375      HRegionInfo regionA, HRegionInfo regionB) throws IOException {
376
377    Map<HRegionInfo, List<ServerName>> result = new HashMap<>();
378    FavoredNodeAssignmentHelper helper = new FavoredNodeAssignmentHelper(servers, rackManager);
379    helper.initialize();
380
381    List<ServerName> parentFavoredNodes = getFavoredNodes(parent);
382    if (parentFavoredNodes == null) {
383      LOG.debug("Unable to find favored nodes for parent, " + parent
384          + " generating new favored nodes for daughter");
385      result.put(regionA, helper.generateFavoredNodes(regionA));
386      result.put(regionB, helper.generateFavoredNodes(regionB));
387
388    } else {
389
390      // Lets get the primary and secondary from parent for regionA
391      Set<ServerName> regionAFN =
392          getInheritedFNForDaughter(helper, parentFavoredNodes, PRIMARY, SECONDARY);
393      result.put(regionA, Lists.newArrayList(regionAFN));
394
395      // Lets get the primary and tertiary from parent for regionB
396      Set<ServerName> regionBFN =
397          getInheritedFNForDaughter(helper, parentFavoredNodes, PRIMARY, TERTIARY);
398      result.put(regionB, Lists.newArrayList(regionBFN));
399    }
400
401    fnm.updateFavoredNodes(result);
402  }
403
404  private Set<ServerName> getInheritedFNForDaughter(FavoredNodeAssignmentHelper helper,
405      List<ServerName> parentFavoredNodes, Position primary, Position secondary)
406      throws IOException {
407
408    Set<ServerName> daughterFN = Sets.newLinkedHashSet();
409    if (parentFavoredNodes.size() >= primary.ordinal()) {
410      daughterFN.add(parentFavoredNodes.get(primary.ordinal()));
411    }
412
413    if (parentFavoredNodes.size() >= secondary.ordinal()) {
414      daughterFN.add(parentFavoredNodes.get(secondary.ordinal()));
415    }
416
417    while (daughterFN.size() < FavoredNodeAssignmentHelper.FAVORED_NODES_NUM) {
418      ServerName newNode = helper.generateMissingFavoredNode(Lists.newArrayList(daughterFN));
419      daughterFN.add(newNode);
420    }
421    return daughterFN;
422  }

424  /*
425   * Generate favored nodes for a region during merge. Choose the FN from one of the sources to
426   * keep it simple.
427   */
428  @Override
429  public void generateFavoredNodesForMergedRegion(HRegionInfo merged, HRegionInfo regionA,
430      HRegionInfo regionB) throws IOException {
431    Map<HRegionInfo, List<ServerName>> regionFNMap = Maps.newHashMap();
432    regionFNMap.put(merged, getFavoredNodes(regionA));
433    fnm.updateFavoredNodes(regionFNMap);
434  }
