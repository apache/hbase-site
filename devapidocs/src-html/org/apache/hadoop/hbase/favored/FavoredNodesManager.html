054/**
055 * FavoredNodesManager is responsible for maintaining favored nodes info in internal cache and
056 * META table. Its the centralized store for all favored nodes information. All reads and updates
057 * should be done through this class. There should only be one instance of
058 * {@link FavoredNodesManager} in Master. {@link FavoredNodesPlan} and favored node information
059 * from {@link SnapshotOfRegionAssignmentFromMeta} should not be used outside this class (except
060 * for may be tools that only read or test cases). All other classes including Favored balancers
061 * and {@link FavoredNodeAssignmentHelper} should use {@link FavoredNodesManager} for any
062 * read/write/deletes to favored nodes.
063 */
064@InterfaceAudience.Private
065public class FavoredNodesManager {
066
067  private static final Log LOG = LogFactory.getLog(FavoredNodesManager.class);
068
069  private FavoredNodesPlan globalFavoredNodesAssignmentPlan;
070  private Map<ServerName, List<RegionInfo>> primaryRSToRegionMap;
071  private Map<ServerName, List<RegionInfo>> secondaryRSToRegionMap;
072  private Map<ServerName, List<RegionInfo>> teritiaryRSToRegionMap;
073
074  private MasterServices masterServices;
075  private RackManager rackManager;
076
077  /**
078   * Datanode port to be used for Favored Nodes.
079   */
080  private int datanodeDataTransferPort;
081
082  public FavoredNodesManager(MasterServices masterServices) {
083    this.masterServices = masterServices;
084    this.globalFavoredNodesAssignmentPlan = new FavoredNodesPlan();
085    this.primaryRSToRegionMap = new HashMap<>();
086    this.secondaryRSToRegionMap = new HashMap<>();
087    this.teritiaryRSToRegionMap = new HashMap<>();
088    this.rackManager = new RackManager(masterServices.getConfiguration());
089  }
090
091  public void initialize(SnapshotOfRegionAssignmentFromMeta snapshotOfRegionAssignment)
092      throws HBaseIOException {
093    globalFavoredNodesAssignmentPlan = snapshotOfRegionAssignment.getExistingAssignmentPlan();
094    primaryRSToRegionMap = snapshotOfRegionAssignment.getPrimaryToRegionInfoMap();
095    secondaryRSToRegionMap = snapshotOfRegionAssignment.getSecondaryToRegionInfoMap();
096    teritiaryRSToRegionMap = snapshotOfRegionAssignment.getTertiaryToRegionInfoMap();
097    datanodeDataTransferPort = getDataNodePort();
098  }
099
100  public int getDataNodePort() {
101    HdfsConfiguration.init();
102
103    Configuration dnConf = new HdfsConfiguration(masterServices.getConfiguration());
104
105    int dnPort = NetUtils.createSocketAddr(
106        dnConf.get(DFSConfigKeys.DFS_DATANODE_ADDRESS_KEY,
107            DFSConfigKeys.DFS_DATANODE_ADDRESS_DEFAULT)).getPort();
108    LOG.debug("Loaded default datanode port for FN: " + datanodeDataTransferPort);
109    return dnPort;
110  }
111
112  public synchronized List<ServerName> getFavoredNodes(RegionInfo regionInfo) {
113    return this.globalFavoredNodesAssignmentPlan.getFavoredNodes(regionInfo);
114  }
115
116  /*
117   * Favored nodes are not applicable for system tables. We will use this to check before
118   * we apply any favored nodes logic on a region.
119   */
120  public static boolean isFavoredNodeApplicable(RegionInfo regionInfo) {
121    return !regionInfo.getTable().isSystemTable();
122  }
123
124  /**
125   * Filter and return regions for which favored nodes is not applicable.
126   *
127   * @param regions - collection of regions
128   * @return set of regions for which favored nodes is not applicable
129   */
130  public static Set<RegionInfo> filterNonFNApplicableRegions(Collection<RegionInfo> regions) {
131    Set<RegionInfo> fnRegions = Sets.newHashSet();
132    for (RegionInfo regionInfo : regions) {
133      if (!isFavoredNodeApplicable(regionInfo)) {
134        fnRegions.add(regionInfo);
135      }
136    }
137    return fnRegions;
138  }
139
140  /*
141   * This should only be used when sending FN information to the region servers. Instead of
142   * sending the region server port, we use the datanode port. This helps in centralizing the DN
143   * port logic in Master. The RS uses the port from the favored node list as hints.
144   */
145  public synchronized List<ServerName> getFavoredNodesWithDNPort(RegionInfo regionInfo) {
146    if (getFavoredNodes(regionInfo) == null) {
147      return null;
148    }
149
150    List<ServerName> fnWithDNPort = Lists.newArrayList();
151    for (ServerName sn : getFavoredNodes(regionInfo)) {
152      fnWithDNPort.add(ServerName.valueOf(sn.getHostname(), datanodeDataTransferPort,
153        NON_STARTCODE));
154    }
155    return fnWithDNPort;
156  }

158  public synchronized void updateFavoredNodes(Map<RegionInfo, List<ServerName>> regionFNMap)
159      throws IOException {
160
161    Map<RegionInfo, List<ServerName>> regionToFavoredNodes = new HashMap<>();
162    for (Map.Entry<RegionInfo, List<ServerName>> entry : regionFNMap.entrySet()) {
163      RegionInfo regionInfo = entry.getKey();
164      List<ServerName> servers = entry.getValue();
165
166      /*
167       * None of the following error conditions should happen. If it does, there is an issue with
168       * favored nodes generation or the regions its called on.
169       */
170      if (servers.size() != Sets.newHashSet(servers).size()) {
171        throw new IOException("Duplicates found: " + servers);
172      }
173
174      if (!isFavoredNodeApplicable(regionInfo)) {
175        throw new IOException("Can't update FN for a un-applicable region: "
176            + regionInfo.getRegionNameAsString() + " with " + servers);
177      }
178
179      if (servers.size() != FAVORED_NODES_NUM) {
180        throw new IOException("At least " + FAVORED_NODES_NUM
181            + " favored nodes should be present for region : " + regionInfo.getEncodedName()
182            + " current FN servers:" + servers);
183      }
184
185      List<ServerName> serversWithNoStartCodes = Lists.newArrayList();
186      for (ServerName sn : servers) {
187        if (sn.getStartcode() == NON_STARTCODE) {
188          serversWithNoStartCodes.add(sn);
189        } else {
190          serversWithNoStartCodes.add(ServerName.valueOf(sn.getHostname(), sn.getPort(),
191              NON_STARTCODE));
192        }
193      }
194      regionToFavoredNodes.put(regionInfo, serversWithNoStartCodes);
195    }
196
197    // Lets do a bulk update to meta since that reduces the RPC's
198    FavoredNodeAssignmentHelper.updateMetaWithFavoredNodesInfo(
199        regionToFavoredNodes,
200        masterServices.getConnection());
201    deleteFavoredNodesForRegions(regionToFavoredNodes.keySet());
202
203    for (Map.Entry<RegionInfo, List<ServerName>> entry : regionToFavoredNodes.entrySet()) {
204      RegionInfo regionInfo = entry.getKey();
205      List<ServerName> serversWithNoStartCodes = entry.getValue();
206      globalFavoredNodesAssignmentPlan.updateFavoredNodesMap(regionInfo, serversWithNoStartCodes);
207      addToReplicaLoad(regionInfo, serversWithNoStartCodes);
208    }
209  }
210
211  private synchronized void addToReplicaLoad(RegionInfo hri, List<ServerName> servers) {
212    ServerName serverToUse = ServerName.valueOf(servers.get(PRIMARY.ordinal()).getHostAndPort(),
213        NON_STARTCODE);
214    List<RegionInfo> regionList = primaryRSToRegionMap.get(serverToUse);
215    if (regionList == null) {
216      regionList = new ArrayList<>();
217    }
218    regionList.add(hri);
219    primaryRSToRegionMap.put(serverToUse, regionList);
220
221    serverToUse = ServerName
222        .valueOf(servers.get(SECONDARY.ordinal()).getHostAndPort(), NON_STARTCODE);
223    regionList = secondaryRSToRegionMap.get(serverToUse);
224    if (regionList == null) {
225      regionList = new ArrayList<>();
226    }
227    regionList.add(hri);
228    secondaryRSToRegionMap.put(serverToUse, regionList);
229
230    serverToUse = ServerName.valueOf(servers.get(TERTIARY.ordinal()).getHostAndPort(),
231      NON_STARTCODE);
232    regionList = teritiaryRSToRegionMap.get(serverToUse);
233    if (regionList == null) {
234      regionList = new ArrayList<>();
235    }
236    regionList.add(hri);
237    teritiaryRSToRegionMap.put(serverToUse, regionList);
238  }

240  /*
241   * Get the replica count for the servers provided.
242   *
243   * For each server, replica count includes three counts for primary, secondary and tertiary.
244   * If a server is the primary favored node for 10 regions, secondary for 5 and tertiary
245   * for 1, then the list would be [10, 5, 1]. If the server is newly added to the cluster is
246   * not a favored node for any region, the replica count would be [0, 0, 0].
247   */
248  public synchronized Map<ServerName, List<Integer>> getReplicaLoad(List<ServerName> servers) {
249    Map<ServerName, List<Integer>> result = Maps.newHashMap();
250    for (ServerName sn : servers) {
251      ServerName serverWithNoStartCode = ServerName.valueOf(sn.getHostAndPort(), NON_STARTCODE);
252      List<Integer> countList = Lists.newArrayList();
253      if (primaryRSToRegionMap.containsKey(serverWithNoStartCode)) {
254        countList.add(primaryRSToRegionMap.get(serverWithNoStartCode).size());
255      } else {
256        countList.add(0);
257      }
258      if (secondaryRSToRegionMap.containsKey(serverWithNoStartCode)) {
259        countList.add(secondaryRSToRegionMap.get(serverWithNoStartCode).size());
260      } else {
261        countList.add(0);
262      }
263      if (teritiaryRSToRegionMap.containsKey(serverWithNoStartCode)) {
264        countList.add(teritiaryRSToRegionMap.get(serverWithNoStartCode).size());
265      } else {
266        countList.add(0);
267      }
268      result.put(sn, countList);
269    }
270    return result;
271  }
272
273  public synchronized void deleteFavoredNodesForRegions(Collection<RegionInfo> regionInfoList) {
274    for (RegionInfo hri : regionInfoList) {
275      List<ServerName> favNodes = getFavoredNodes(hri);
276      if (favNodes != null) {
277        if (primaryRSToRegionMap.containsKey(favNodes.get(PRIMARY.ordinal()))) {
278          primaryRSToRegionMap.get(favNodes.get(PRIMARY.ordinal())).remove(hri);
279        }
280        if (secondaryRSToRegionMap.containsKey(favNodes.get(SECONDARY.ordinal()))) {
281          secondaryRSToRegionMap.get(favNodes.get(SECONDARY.ordinal())).remove(hri);
282        }
283        if (teritiaryRSToRegionMap.containsKey(favNodes.get(TERTIARY.ordinal()))) {
284          teritiaryRSToRegionMap.get(favNodes.get(TERTIARY.ordinal())).remove(hri);
285        }
286        globalFavoredNodesAssignmentPlan.removeFavoredNodes(hri);
287      }
288    }
289  }

291  @VisibleForTesting
292  public synchronized Set<RegionInfo> getRegionsOfFavoredNode(ServerName serverName) {
293    Set<RegionInfo> regionInfos = Sets.newHashSet();
294
295    ServerName serverToUse = ServerName.valueOf(serverName.getHostAndPort(), NON_STARTCODE);
296    if (primaryRSToRegionMap.containsKey(serverToUse)) {
297      regionInfos.addAll(primaryRSToRegionMap.get(serverToUse));
298    }
299    if (secondaryRSToRegionMap.containsKey(serverToUse)) {
300      regionInfos.addAll(secondaryRSToRegionMap.get(serverToUse));
301    }
302    if (teritiaryRSToRegionMap.containsKey(serverToUse)) {
303      regionInfos.addAll(teritiaryRSToRegionMap.get(serverToUse));
304    }
305    return regionInfos;
306  }

309    return rackManager;
