050/**
051 * ProcedureWALPrettyPrinter prints the contents of a given ProcedureWAL file
052 * @see WALProcedureStore#main(String[]) if you want to check parse of a directory of WALs.
053 */
054@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
055@InterfaceStability.Evolving
056public class ProcedureWALPrettyPrinter extends Configured implements Tool {
057  private static final Logger LOG = LoggerFactory.getLogger(ProcedureWALPrettyPrinter.class);
058
059  private final PrintStream out;
060
061  public ProcedureWALPrettyPrinter() {
062    out = System.out;
063  }
064
065  /**
066   * Reads a log file and outputs its contents.
067   *
068   * @param conf   HBase configuration relevant to this log file
069   * @param p       path of the log file to be read
070   * @throws IOException  IOException
071   */
072  public void processFile(final Configuration conf, final Path p)
073      throws IOException {
074
075    FileSystem fs = p.getFileSystem(conf);
076    if (!fs.exists(p)) {
077      System.err.println("ERROR, file doesnt exist: " + p);
078      return;
079    }
080    if (!fs.isFile(p)) {
081      System.err.println(p + " is not a file");
082      return;
083    }
084
085    FileStatus logFile = fs.getFileStatus(p);
086    if (logFile.getLen() == 0) {
087      out.println("Zero length file: " + p);
088      return;
089    }
090
091    out.println("Opening procedure state-log: " + p);
092    ProcedureWALFile log = new ProcedureWALFile(fs, logFile);
093    processProcedureWALFile(log);
094  }
095
096  public void processProcedureWALFile(ProcedureWALFile log) throws IOException {
097    log.open();
098    ProcedureWALHeader header = log.getHeader();
099    printHeader(header);
100
101    FSDataInputStream stream = log.getStream();
102    try {
103      boolean hasMore = true;
104      while (hasMore) {
105        ProcedureWALEntry entry = ProcedureWALFormat.readEntry(stream);
106        if (entry == null) {
107          out.println("No more entry, exiting with missing EOF");
108          hasMore = false;
109          break;
110        }
111        switch (entry.getType()) {
112          case PROCEDURE_WAL_EOF:
113            hasMore = false;
114            break;
115          default:
116            printEntry(entry);
117        }
118      }
119    } catch (IOException e) {
120      out.println("got an exception while reading the procedure WAL " + e.getMessage());
121    }
122    finally {
123      log.close();
124    }
125  }
126
127  private void printEntry(final ProcedureWALEntry entry) throws IOException {
128    out.println("EntryType=" + entry.getType());
129    int procCount = entry.getProcedureCount();
130    for (int i = 0; i < procCount; i++) {
131      Procedure<?> proc = ProcedureUtil.convertToProcedure(entry.getProcedure(i));
132      printProcedure(proc);
133    }
134  }
135
136  private void printProcedure(Procedure<?> proc) {
137    out.println(proc.toStringDetails());
138  }
139
140  private void printHeader(ProcedureWALHeader header) {
141    out.println("ProcedureWALHeader: ");
142    out.println("  Version: " + header.getVersion());
143    out.println("  Type: " + header.getType());
144    out.println("  LogId: " + header.getLogId());
145    out.println("  MinProcId: " + header.getMinProcId());
146    out.println();
147  }
148
149  /**
150   * Pass one or more log file names and formatting options and it will dump out
151   * a text version of the contents on <code>stdout</code>.
152   *
153   * @param args
154   *          Command line arguments
155   * @throws IOException
156   *           Thrown upon file system errors etc.
157   */
158  @Override
159  public int run(final String[] args) throws IOException {
160    // create options
161    Options options = new Options();
162    options.addOption("h", "help", false, "Output help message");
163    options.addOption("f", "file", true, "File to print");
164
165    final List<Path> files = new ArrayList<>();
166    try {
167      CommandLine cmd = new PosixParser().parse(options, args);
168
169      if (cmd.hasOption("f")) {
170        files.add(new Path(cmd.getOptionValue("f")));
171      }
172
173      if (files.isEmpty() || cmd.hasOption("h")) {
174        HelpFormatter formatter = new HelpFormatter();
175        formatter.printHelp("ProcedureWALPrettyPrinter ", options, true);
176        return(-1);
177      }
178    } catch (ParseException e) {
179      LOG.error("Failed to parse commandLine arguments", e);
180      HelpFormatter formatter = new HelpFormatter();
181      formatter.printHelp("ProcedureWALPrettyPrinter ", options, true);
182      return(-1);
183    }
184    // get configuration, file system, and process the given files
185    for (Path file : files) {
186      processFile(getConf(), file);
187    }
188    return(0);
189  }
190
191  public static void main(String[] args) throws Exception {
192    final Configuration conf = HBaseConfiguration.create();
193    int exitCode = ToolRunner.run(conf, new ProcedureWALPrettyPrinter(), args);
194    System.exit(exitCode);
195  }
