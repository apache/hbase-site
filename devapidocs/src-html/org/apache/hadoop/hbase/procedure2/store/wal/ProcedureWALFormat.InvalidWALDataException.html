041/**
042 * Helper class that contains the WAL serialization utils.
043 */
044@InterfaceAudience.Private
045@InterfaceStability.Evolving
046public final class ProcedureWALFormat {
047  static final byte LOG_TYPE_STREAM = 0;
048  static final byte LOG_TYPE_COMPACTED = 1;
049  static final byte LOG_TYPE_MAX_VALID = 1;
050
051  static final byte HEADER_VERSION = 1;
052  static final byte TRAILER_VERSION = 1;
053  static final long HEADER_MAGIC = 0x31764c4157637250L;
054  static final long TRAILER_MAGIC = 0x50726357414c7631L;
055
056  @InterfaceAudience.Private
057  public static class InvalidWALDataException extends IOException {
058    public InvalidWALDataException(String s) {
059      super(s);
060    }
061
062    public InvalidWALDataException(Throwable t) {
063      super(t);
064    }
065  }
066
067  interface Loader extends ProcedureLoader {
068    void markCorruptedWAL(ProcedureWALFile log, IOException e);
069  }
070
071  private ProcedureWALFormat() {}
072
073  public static void load(final Iterator<ProcedureWALFile> logs,
074      final ProcedureStoreTracker tracker, final Loader loader) throws IOException {
075    ProcedureWALFormatReader reader = new ProcedureWALFormatReader(tracker);
076    tracker.setKeepDeletes(true);
077    try {
078      while (logs.hasNext()) {
079        ProcedureWALFile log = logs.next();
080        log.open();
081        try {
082          reader.read(log, loader);
083        } finally {
084          log.close();
085        }
086      }
087      reader.finalize(loader);
088      // The tracker is now updated with all the procedures read from the logs
089      tracker.setPartialFlag(false);
090      tracker.resetUpdates();
091    } finally {
092      tracker.setKeepDeletes(false);
093    }
094  }
095
096  public static void writeHeader(OutputStream stream, ProcedureWALHeader header)
097      throws IOException {
098    header.writeDelimitedTo(stream);
099  }
100
101  /*
102   * +-----------------+
103   * | END OF WAL DATA | <---+
104   * +-----------------+     |
105   * |                 |     |
106   * |     Tracker     |     |
107   * |                 |     |
108   * +-----------------+     |
109   * |     version     |     |
110   * +-----------------+     |
111   * |  TRAILER_MAGIC  |     |
112   * +-----------------+     |
113   * |      offset     |-----+
114   * +-----------------+
115   */
116  public static long writeTrailer(FSDataOutputStream stream, ProcedureStoreTracker tracker)
117      throws IOException {
118    long offset = stream.getPos();
119
120    // Write EOF Entry
121    ProcedureWALEntry.newBuilder()
122      .setType(ProcedureWALEntry.Type.PROCEDURE_WAL_EOF)
123      .build().writeDelimitedTo(stream);
124
125    // Write Tracker
126    tracker.toProto().writeDelimitedTo(stream);
127
128    stream.write(TRAILER_VERSION);
129    StreamUtils.writeLong(stream, TRAILER_MAGIC);
130    StreamUtils.writeLong(stream, offset);
131    return stream.getPos() - offset;
132  }
133
134  public static ProcedureWALHeader readHeader(InputStream stream)
135      throws IOException {
136    ProcedureWALHeader header;
137    try {
138      header = ProcedureWALHeader.parseDelimitedFrom(stream);
139    } catch (InvalidProtocolBufferException e) {
140      throw new InvalidWALDataException(e);
141    }
142
143    if (header == null) {
144      throw new InvalidWALDataException("No data available to read the Header");
145    }
146
147    if (header.getVersion() < 0 || header.getVersion() != HEADER_VERSION) {
148      throw new InvalidWALDataException("Invalid Header version. got " + header.getVersion() +
149          " expected " + HEADER_VERSION);
150    }
151
152    if (header.getType() < 0 || header.getType() > LOG_TYPE_MAX_VALID) {
153      throw new InvalidWALDataException("Invalid header type. got " + header.getType());
154    }
155
156    return header;
157  }
158
159  public static ProcedureWALTrailer readTrailer(FSDataInputStream stream, long startPos, long size)
160      throws IOException {
161    // Beginning of the Trailer Jump. 17 = 1 byte version + 8 byte magic + 8 byte offset
162    long trailerPos = size - 17;
163
164    if (trailerPos < startPos) {
165      throw new InvalidWALDataException("Missing trailer: size=" + size + " startPos=" + startPos);
166    }
167
168    stream.seek(trailerPos);
169    int version = stream.read();
170    if (version != TRAILER_VERSION) {
171      throw new InvalidWALDataException("Invalid Trailer version. got " + version +
172          " expected " + TRAILER_VERSION);
173    }
174
175    long magic = StreamUtils.readLong(stream);
176    if (magic != TRAILER_MAGIC) {
177      throw new InvalidWALDataException("Invalid Trailer magic. got " + magic +
178          " expected " + TRAILER_MAGIC);
179    }
180
181    long trailerOffset = StreamUtils.readLong(stream);
182    stream.seek(trailerOffset);
183
184    ProcedureWALEntry entry = readEntry(stream);
185    if (entry.getType() != ProcedureWALEntry.Type.PROCEDURE_WAL_EOF) {
186      throw new InvalidWALDataException("Invalid Trailer begin");
187    }
188
189    ProcedureWALTrailer trailer = ProcedureWALTrailer.newBuilder()
190      .setVersion(version)
191      .setTrackerPos(stream.getPos())
192      .build();
193    return trailer;
194  }
195
196  public static ProcedureWALEntry readEntry(InputStream stream) throws IOException {
197    return ProcedureWALEntry.parseDelimitedFrom(stream);
198  }
199
200  public static void writeEntry(ByteSlot slot, ProcedureWALEntry.Type type,
201      Procedure proc, Procedure[] subprocs) throws IOException {
202    ProcedureWALEntry.Builder builder = ProcedureWALEntry.newBuilder();
203    builder.setType(type);
204    builder.addProcedure(Procedure.convert(proc));
205    if (subprocs != null) {
206      for (int i = 0; i < subprocs.length; ++i) {
207        builder.addProcedure(Procedure.convert(subprocs[i]));
208      }
209    }
210    builder.build().writeDelimitedTo(slot);
211  }
212
213  public static void writeInsert(ByteSlot slot, Procedure proc)
214      throws IOException {
215    writeEntry(slot, ProcedureWALEntry.Type.PROCEDURE_WAL_INIT, proc, null);
216  }
217
218  public static void writeInsert(ByteSlot slot, Procedure proc, Procedure[] subprocs)
219      throws IOException {
220    writeEntry(slot, ProcedureWALEntry.Type.PROCEDURE_WAL_INSERT, proc, subprocs);
221  }
222
223  public static void writeUpdate(ByteSlot slot, Procedure proc)
224      throws IOException {
225    writeEntry(slot, ProcedureWALEntry.Type.PROCEDURE_WAL_UPDATE, proc, null);
226  }
227
228  public static void writeDelete(ByteSlot slot, long procId)
229      throws IOException {
230    ProcedureWALEntry.Builder builder = ProcedureWALEntry.newBuilder();
231    builder.setType(ProcedureWALEntry.Type.PROCEDURE_WAL_DELETE);
232    builder.setProcId(procId);
233    builder.build().writeDelimitedTo(slot);
234  }
235
236  public static void writeDelete(ByteSlot slot, Procedure proc, long[] subprocs)
237      throws IOException {
238    ProcedureWALEntry.Builder builder = ProcedureWALEntry.newBuilder();
239    builder.setType(ProcedureWALEntry.Type.PROCEDURE_WAL_DELETE);
240    builder.setProcId(proc.getProcId());
241    if (subprocs != null) {
242      builder.addProcedure(Procedure.convert(proc));
243      for (int i = 0; i < subprocs.length; ++i) {
244        builder.addChildId(subprocs[i]);
245      }
246    }
247    builder.build().writeDelimitedTo(slot);
248  }
