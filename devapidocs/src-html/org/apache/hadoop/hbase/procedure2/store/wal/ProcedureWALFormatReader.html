018package org.apache.hadoop.hbase.procedure2.store.wal;
019
020import java.io.IOException;
021import org.apache.hadoop.fs.FSDataInputStream;
022import org.apache.hadoop.hbase.procedure2.store.ProcedureStore.ProcedureIterator;
023import org.apache.hadoop.hbase.procedure2.store.ProcedureStoreTracker;
024import org.apache.yetus.audience.InterfaceAudience;
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028import org.apache.hbase.thirdparty.com.google.protobuf.InvalidProtocolBufferException;
029
030import org.apache.hadoop.hbase.shaded.protobuf.generated.ProcedureProtos;
031import org.apache.hadoop.hbase.shaded.protobuf.generated.ProcedureProtos.ProcedureWALEntry;
032
033/**
034 * Helper class that loads the procedures stored in a WAL
035 */
036@InterfaceAudience.Private
037public class ProcedureWALFormatReader {
038  private static final Logger LOG = LoggerFactory.getLogger(ProcedureWALFormatReader.class);
039
040  // ==============================================================================================
041  //  We read the WALs in reverse order from the newest to the oldest.
042  //  We have different entry types:
043  //   - INIT: Procedure submitted by the user (also known as 'root procedure')
044  //   - INSERT: Children added to the procedure <parentId>:[<childId>, ...]
045  //   - UPDATE: The specified procedure was updated
046  //   - DELETE: The procedure was removed (finished/rolledback and result TTL expired)
047  //
048  // In the WAL we can find multiple times the same procedure as UPDATE or INSERT.
049  // We read the WAL from top to bottom, so every time we find an entry of the
050  // same procedure, that will be the "latest" update (Caveat: with multiple threads writing
051  // the store, this assumption does not hold).
052  //
053  // We keep two in-memory maps:
054  //  - localProcedureMap: is the map containing the entries in the WAL we are processing
055  //  - procedureMap: is the map containing all the procedures we found up to the WAL in process.
056  // localProcedureMap is merged with the procedureMap once we reach the WAL EOF.
057  //
058  // Since we are reading the WALs in reverse order (newest to oldest),
059  // if we find an entry related to a procedure we already have in 'procedureMap' we can discard it.
060  //
061  // The WAL is append-only so the last procedure in the WAL is the one that
062  // was in execution at the time we crashed/closed the server.
063  // Given that, the procedure replay order can be inferred by the WAL order.
064  //
065  // Example:
066  //    WAL-2: [A, B, A, C, D]
067  //    WAL-1: [F, G, A, F, B]
068  //    Replay-Order: [D, C, A, B, F, G]
069  //
070  // The "localProcedureMap" keeps a "replayOrder" list. Every time we add the
071  // record to the map that record is moved to the head of the "replayOrder" list.
072  // Using the example above:
073  //    WAL-2 localProcedureMap.replayOrder is [D, C, A, B]
074  //    WAL-1 localProcedureMap.replayOrder is [F, G]
075  //
076  // Each time we reach the WAL-EOF, the "replayOrder" list is merged/appended in 'procedureMap'
077  // so using the example above we end up with: [D, C, A, B] + [F, G] as replay order.
078  //
079  //  Fast Start: INIT/INSERT record and StackIDs
080  // ---------------------------------------------
081  // We have two special records, INIT and INSERT, that track the first time
082  // the procedure was added to the WAL. We can use this information to be able
083  // to start procedures before reaching the end of the WAL, or before reading all WALs.
084  // But in some cases, the WAL with that record can be already gone.
085  // As an alternative, we can use the stackIds on each procedure,
086  // to identify when a procedure is ready to start.
087  // If there are gaps in the sum of the stackIds we need to read more WALs.
088  //
089  // Example (all procs child of A):
090  //   WAL-2: [A, B]                   A stackIds = [0, 4], B stackIds = [1, 5]
091  //   WAL-1: [A, B, C, D]
092  //
093  // In the case above we need to read one more WAL to be able to consider
094  // the root procedure A and all children as ready.
095  // ==============================================================================================
096  private final WALProcedureMap localProcedureMap = new WALProcedureMap(1024);
097  private final WALProcedureMap procedureMap = new WALProcedureMap(1024);
098
099  private final ProcedureWALFormat.Loader loader;
100
101  /**
102   * Global tracker that will be used by the WALProcedureStore after load.
103   * If the last WAL was closed cleanly we already have a full tracker ready to be used.
104   * If the last WAL was truncated (e.g. master killed) the tracker will be empty
105   * and the 'partial' flag will be set. In this case, on WAL replay we are going
106   * to rebuild the tracker.
107   */
108  private final ProcedureStoreTracker tracker;
109
110  /**
111   * If tracker for a log file is partial (see {@link ProcedureStoreTracker#partial}), we re-build
112   * the list of procedures modified in that WAL because we need it for log cleaning purposes. If
113   * all procedures modified in a WAL are found to be obsolete, it can be safely deleted. (see
114   * {@link WALProcedureStore#removeInactiveLogs()}).
115   * <p/>
116   * Notice that, the deleted part for this tracker will not be global valid as we can only count
117   * the deletes in the current file, but it is not big problem as finally, the above tracker will
118   * have the global state of deleted, and it will also be used to build the cleanup tracker.
119   */
120  private ProcedureStoreTracker localTracker;
121
122  private long maxProcId = 0;
123
124  public ProcedureWALFormatReader(final ProcedureStoreTracker tracker,
125      ProcedureWALFormat.Loader loader) {
126    this.tracker = tracker;
127    this.loader = loader;
128  }
129
130  public void read(ProcedureWALFile log) throws IOException {
131    localTracker = log.getTracker();
132    if (localTracker.isPartial()) {
133      LOG.info("Rebuilding tracker for {}", log);
134    }
135
136    long count = 0;
137    FSDataInputStream stream = log.getStream();
138    try {
139      boolean hasMore = true;
140      while (hasMore) {
141        ProcedureWALEntry entry = ProcedureWALFormat.readEntry(stream);
142        if (entry == null) {
143          LOG.warn("Nothing left to decode. Exiting with missing EOF, log={}", log);
144          break;
145        }
146        count++;
147        switch (entry.getType()) {
148          case PROCEDURE_WAL_INIT:
149            readInitEntry(entry);
150            break;
151          case PROCEDURE_WAL_INSERT:
152            readInsertEntry(entry);
153            break;
154          case PROCEDURE_WAL_UPDATE:
155          case PROCEDURE_WAL_COMPACT:
156            readUpdateEntry(entry);
157            break;
158          case PROCEDURE_WAL_DELETE:
159            readDeleteEntry(entry);
160            break;
161          case PROCEDURE_WAL_EOF:
162            hasMore = false;
163            break;
164          default:
165            throw new CorruptedWALProcedureStoreException("Invalid entry: " + entry);
166        }
167      }
168      LOG.info("Read {} entries in {}", count, log);
169    } catch (InvalidProtocolBufferException e) {
170      LOG.error("While reading entry #{} in {}", count, log, e);
171      loader.markCorruptedWAL(log, e);
172    }
173
174    if (!localProcedureMap.isEmpty()) {
175      log.setProcIds(localProcedureMap.getMinModifiedProcId(),
176        localProcedureMap.getMaxModifiedProcId());
177      if (localTracker.isPartial()) {
178        localTracker.setMinMaxModifiedProcIds(localProcedureMap.getMinModifiedProcId(),
179          localProcedureMap.getMaxModifiedProcId());
180      }
181      procedureMap.mergeTail(localProcedureMap);
182    }
183    if (localTracker.isPartial()) {
184      localTracker.setPartialFlag(false);
185    }
186  }
187
188  public void finish() throws IOException {
189    // notify the loader about the max proc ID
190    loader.setMaxProcId(maxProcId);
191
192    // fetch the procedure ready to run.
193    ProcedureIterator procIter = procedureMap.fetchReady();
194    if (procIter != null) {
195      loader.load(procIter);
196    }
197
198    // remaining procedures have missing link or dependencies
199    // consider them as corrupted, manual fix is probably required.
200    procIter = procedureMap.fetchAll();
201    if (procIter != null) {
202      loader.handleCorrupted(procIter);
203    }
204  }
205
206  private void setDeletedIfPartial(ProcedureStoreTracker tracker, long procId) {
207    if (tracker.isPartial()) {
208      tracker.setDeleted(procId, true);
209    }
210  }
211
212  private void insertIfPartial(ProcedureStoreTracker tracker, ProcedureProtos.Procedure proc) {
213    if (tracker.isPartial()) {
214      tracker.insert(proc.getProcId());
215    }
216  }
217
218  private void loadProcedure(ProcedureWALEntry entry, ProcedureProtos.Procedure proc) {
219    maxProcId = Math.max(maxProcId, proc.getProcId());
220    if (isRequired(proc.getProcId())) {
221      LOG.trace("Read {} entry {}", entry.getType(), proc.getProcId());
222      localProcedureMap.add(proc);
223      insertIfPartial(tracker, proc);
224    }
225    insertIfPartial(localTracker, proc);
226  }
227
228  private void readInitEntry(ProcedureWALEntry entry) {
229    assert entry.getProcedureCount() == 1 : "Expected only one procedure";
230    loadProcedure(entry, entry.getProcedure(0));
231  }
232
233  private void readInsertEntry(ProcedureWALEntry entry) {
234    assert entry.getProcedureCount() >= 1 : "Expected one or more procedures";
235    loadProcedure(entry, entry.getProcedure(0));
236    for (int i = 1; i < entry.getProcedureCount(); ++i) {
237      loadProcedure(entry, entry.getProcedure(i));
238    }
239  }
240
241  private void readUpdateEntry(ProcedureWALEntry entry) {
242    assert entry.getProcedureCount() == 1 : "Expected only one procedure";
243    loadProcedure(entry, entry.getProcedure(0));
244  }
245
246  private void readDeleteEntry(ProcedureWALEntry entry) {
247    assert entry.hasProcId() : "expected ProcID";
248
249    if (entry.getChildIdCount() > 0) {
250      assert entry.getProcedureCount() == 1 : "Expected only one procedure";
251
252      // update the parent procedure
253      loadProcedure(entry, entry.getProcedure(0));
254
255      // remove the child procedures of entry.getProcId()
256      for (int i = 0, count = entry.getChildIdCount(); i < count; ++i) {
257        deleteEntry(entry.getChildId(i));
258      }
259    } else {
260      assert entry.getProcedureCount() == 0 : "Expected no procedures";
261
262      // delete the procedure
263      deleteEntry(entry.getProcId());
264    }
265  }
266
267  private void deleteEntry(final long procId) {
268    LOG.trace("delete entry {}", procId);
269    maxProcId = Math.max(maxProcId, procId);
270    localProcedureMap.remove(procId);
271    assert !procedureMap.contains(procId);
272    setDeletedIfPartial(tracker, procId);
273    setDeletedIfPartial(localTracker, procId);
274  }
275
276  private boolean isDeleted(long procId) {
277    return tracker.isDeleted(procId) == ProcedureStoreTracker.DeleteState.YES;
278  }
279
280  private boolean isRequired(long procId) {
281    return !isDeleted(procId) && !procedureMap.contains(procId);
282  }
