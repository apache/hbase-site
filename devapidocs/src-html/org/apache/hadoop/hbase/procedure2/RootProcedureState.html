019package org.apache.hadoop.hbase.procedure2;
020
021import java.util.ArrayList;
022import java.util.HashSet;
023import java.util.List;
024import java.util.Set;
025import org.apache.yetus.audience.InterfaceAudience;
026import org.apache.yetus.audience.InterfaceStability;
027
028import org.apache.hadoop.hbase.shaded.protobuf.generated.ProcedureProtos.ProcedureState;
029
030/**
031 * Internal state of the ProcedureExecutor that describes the state of a "Root Procedure".
032 * A "Root Procedure" is a Procedure without parent, each subprocedure will be
033 * added to the "Root Procedure" stack (or rollback-stack).
034 *
035 * RootProcedureState is used and managed only by the ProcedureExecutor.
036 *    Long rootProcId = getRootProcedureId(proc);
037 *    rollbackStack.get(rootProcId).acquire(proc)
038 *    rollbackStack.get(rootProcId).release(proc)
039 *    ...
040 */
041@InterfaceAudience.Private
042@InterfaceStability.Evolving
043class RootProcedureState<TEnvironment> {
044
045  private enum State {
046    RUNNING,         // The Procedure is running or ready to run
047    FAILED,          // The Procedure failed, waiting for the rollback executing
048    ROLLINGBACK,     // The Procedure failed and the execution was rolledback
049  }
050
051  private Set<Procedure<TEnvironment>> subprocs = null;
052  private ArrayList<Procedure<TEnvironment>> subprocStack = null;
053  private State state = State.RUNNING;
054  private int running = 0;
055
056  public synchronized boolean isFailed() {
057    switch (state) {
058      case ROLLINGBACK:
059      case FAILED:
060        return true;
061      default:
062        break;
063    }
064    return false;
065  }
066
067  public synchronized boolean isRollingback() {
068    return state == State.ROLLINGBACK;
069  }
070
071  /**
072   * Called by the ProcedureExecutor to mark rollback execution
073   */
074  protected synchronized boolean setRollback() {
075    if (running == 0 && state == State.FAILED) {
076      state = State.ROLLINGBACK;
077      return true;
078    }
079    return false;
080  }
081
082  /**
083   * Called by the ProcedureExecutor to mark rollback execution
084   */
085  protected synchronized void unsetRollback() {
086    assert state == State.ROLLINGBACK;
087    state = State.FAILED;
088  }
089
090  protected synchronized long[] getSubprocedureIds() {
091    if (subprocs == null) {
092      return null;
093    }
094    return subprocs.stream().mapToLong(Procedure::getProcId).toArray();
095  }
096
097  protected synchronized List<Procedure<TEnvironment>> getSubproceduresStack() {
098    return subprocStack;
099  }
100
101  protected synchronized RemoteProcedureException getException() {
102    if (subprocStack != null) {
103      for (Procedure<TEnvironment> proc: subprocStack) {
104        if (proc.hasException()) {
105          return proc.getException();
106        }
107      }
108    }
109    return null;
110  }
111
112  /**
113   * Called by the ProcedureExecutor to mark the procedure step as running.
114   */
115  protected synchronized boolean acquire(Procedure<TEnvironment> proc) {
116    if (state != State.RUNNING) {
117      return false;
118    }
119
120    running++;
121    return true;
122  }
123
124  /**
125   * Called by the ProcedureExecutor to mark the procedure step as finished.
126   */
127  protected synchronized void release(Procedure<TEnvironment> proc) {
128    running--;
129  }
130
131  protected synchronized void abort() {
132    if (state == State.RUNNING) {
133      state = State.FAILED;
134    }
135  }
136
137  /**
138   * Called by the ProcedureExecutor after the procedure step is completed,
139   * to add the step to the rollback list (or procedure stack)
140   */
141  protected synchronized void addRollbackStep(Procedure<TEnvironment> proc) {
142    if (proc.isFailed()) {
143      state = State.FAILED;
144    }
145    if (subprocStack == null) {
146      subprocStack = new ArrayList<>();
147    }
148    proc.addStackIndex(subprocStack.size());
149    subprocStack.add(proc);
150  }
151
152  protected synchronized void addSubProcedure(Procedure<TEnvironment> proc) {
153    if (!proc.hasParent()) {
154      return;
155    }
156    if (subprocs == null) {
157      subprocs = new HashSet<>();
158    }
159    subprocs.add(proc);
160  }
161
162  /**
163   * Called on store load by the ProcedureExecutor to load part of the stack.
164   *
165   * Each procedure has its own stack-positions. Which means we have to write
166   * to the store only the Procedure we executed, and nothing else.
167   * on load we recreate the full stack by aggregating each procedure stack-positions.
168   */
169  protected synchronized void loadStack(Procedure<TEnvironment> proc) {
170    addSubProcedure(proc);
171    int[] stackIndexes = proc.getStackIndexes();
172    if (stackIndexes != null) {
173      if (subprocStack == null) {
174        subprocStack = new ArrayList<>();
175      }
176      int diff = (1 + stackIndexes[stackIndexes.length - 1]) - subprocStack.size();
177      if (diff > 0) {
178        subprocStack.ensureCapacity(1 + stackIndexes[stackIndexes.length - 1]);
179        while (diff-- > 0) subprocStack.add(null);
180      }
181      for (int i = 0; i < stackIndexes.length; ++i) {
182        subprocStack.set(stackIndexes[i], proc);
183      }
184    }
185    if (proc.getState() == ProcedureState.ROLLEDBACK) {
186      state = State.ROLLINGBACK;
187    } else if (proc.isFailed()) {
188      state = State.FAILED;
189    }
190  }
191
192  /**
193   * Called on store load by the ProcedureExecutor to validate the procedure stack.
194   */
195  protected synchronized boolean isValid() {
196    if (subprocStack != null) {
197      for (Procedure<TEnvironment> proc : subprocStack) {
198        if (proc == null) {
199          return false;
200        }
201      }
202    }
203    return true;
204  }
205}
