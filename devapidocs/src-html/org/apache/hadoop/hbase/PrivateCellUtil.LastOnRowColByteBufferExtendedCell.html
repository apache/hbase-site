591    static int write(OutputStream out, boolean withTags, Cell cell, byte[] value, byte[] tags)
592        throws IOException {
593      int valLen = value == null ? 0 : value.length;
594      ByteBufferUtils.putInt(out, KeyValueUtil.keyLength(cell));// Key length
595      ByteBufferUtils.putInt(out, valLen);// Value length
596      int len = 2 * Bytes.SIZEOF_INT;
597      len += writeFlatKey(cell, out);// Key
598      if (valLen > 0) {
599        out.write(value);// Value
600      }
601      len += valLen;
602      if (withTags && tags != null) {
603        // Write the tagsLength 2 bytes
604        out.write((byte) (0xff & (tags.length >> 8)));
605        out.write((byte) (0xff & tags.length));
606        out.write(tags);
607        len += KeyValue.TAGS_LENGTH_SIZE + tags.length;
608      }
609      return len;
610    }

1023  /**
1024   * Find length of common prefix in keys of the cells, considering key as byte[] if serialized in
1025   * {@link KeyValue}. The key format is &lt;2 bytes rk len&gt;&lt;rk&gt;&lt;1 byte cf
1026   * len&gt;&lt;cf&gt;&lt;qualifier&gt;&lt;8 bytes timestamp&gt;&lt;1 byte type&gt;
1027   * @param c1 the cell
1028   * @param c2 the cell
1029   * @param bypassFamilyCheck when true assume the family bytes same in both cells. Pass it as true
1030   *          when dealing with Cells in same CF so as to avoid some checks
1031   * @param withTsType when true check timestamp and type bytes also.
1032   * @return length of common prefix
1033   */
1034  public static int findCommonPrefixInFlatKey(Cell c1, Cell c2, boolean bypassFamilyCheck,
1035      boolean withTsType) {
1036    // Compare the 2 bytes in RK length part
1037    short rLen1 = c1.getRowLength();
1038    short rLen2 = c2.getRowLength();
1039    int commonPrefix = KeyValue.ROW_LENGTH_SIZE;
1040    if (rLen1 != rLen2) {
1041      // early out when the RK length itself is not matching
1042      return ByteBufferUtils
1043          .findCommonPrefix(Bytes.toBytes(rLen1), 0, KeyValue.ROW_LENGTH_SIZE, Bytes.toBytes(rLen2),
1044              0, KeyValue.ROW_LENGTH_SIZE);
1045    }
1046    // Compare the RKs
1047    int rkCommonPrefix = 0;
1048    if (c1 instanceof ByteBufferExtendedCell && c2 instanceof ByteBufferExtendedCell) {
1049      rkCommonPrefix = ByteBufferUtils
1050          .findCommonPrefix(((ByteBufferExtendedCell) c1).getRowByteBuffer(),
1051              ((ByteBufferExtendedCell) c1).getRowPosition(), rLen1,
1052              ((ByteBufferExtendedCell) c2).getRowByteBuffer(),
1053              ((ByteBufferExtendedCell) c2).getRowPosition(), rLen2);
1054    } else {
1055      // There cannot be a case where one cell is BBCell and other is KeyValue. This flow comes
1056      // either
1057      // in flush or compactions. In flushes both cells are KV and in case of compaction it will be
1058      // either
1059      // KV or BBCell
1060      rkCommonPrefix = ByteBufferUtils
1061          .findCommonPrefix(c1.getRowArray(), c1.getRowOffset(), rLen1, c2.getRowArray(),
1062              c2.getRowOffset(), rLen2);
1063    }
1064    commonPrefix += rkCommonPrefix;
1065    if (rkCommonPrefix != rLen1) {
1066      // Early out when RK is not fully matching.
1067      return commonPrefix;
1068    }
1069    // Compare 1 byte CF length part
1070    byte fLen1 = c1.getFamilyLength();
1071    if (bypassFamilyCheck) {
1072      // This flag will be true when caller is sure that the family will be same for both the cells
1073      // Just make commonPrefix to increment by the family part
1074      commonPrefix += KeyValue.FAMILY_LENGTH_SIZE + fLen1;
1075    } else {
1076      byte fLen2 = c2.getFamilyLength();
1077      if (fLen1 != fLen2) {
1078        // early out when the CF length itself is not matching
1079        return commonPrefix;
1080      }
1081      // CF lengths are same so there is one more byte common in key part
1082      commonPrefix += KeyValue.FAMILY_LENGTH_SIZE;
1083      // Compare the CF names
1084      int fCommonPrefix;
1085      if (c1 instanceof ByteBufferExtendedCell && c2 instanceof ByteBufferExtendedCell) {
1086        fCommonPrefix = ByteBufferUtils
1087            .findCommonPrefix(((ByteBufferExtendedCell) c1).getFamilyByteBuffer(),
1088                ((ByteBufferExtendedCell) c1).getFamilyPosition(), fLen1,
1089                ((ByteBufferExtendedCell) c2).getFamilyByteBuffer(),
1090                ((ByteBufferExtendedCell) c2).getFamilyPosition(), fLen2);
1091      } else {
1092        fCommonPrefix = ByteBufferUtils
1093            .findCommonPrefix(c1.getFamilyArray(), c1.getFamilyOffset(), fLen1, c2.getFamilyArray(),
1094                c2.getFamilyOffset(), fLen2);
1095      }
1096      commonPrefix += fCommonPrefix;
1097      if (fCommonPrefix != fLen1) {
1098        return commonPrefix;
1099      }
1100    }
1101    // Compare the Qualifiers
1102    int qLen1 = c1.getQualifierLength();
1103    int qLen2 = c2.getQualifierLength();
1104    int qCommon;
1105    if (c1 instanceof ByteBufferExtendedCell && c2 instanceof ByteBufferExtendedCell) {
1106      qCommon = ByteBufferUtils
1107          .findCommonPrefix(((ByteBufferExtendedCell) c1).getQualifierByteBuffer(),
1108              ((ByteBufferExtendedCell) c1).getQualifierPosition(), qLen1,
1109              ((ByteBufferExtendedCell) c2).getQualifierByteBuffer(),
1110              ((ByteBufferExtendedCell) c2).getQualifierPosition(), qLen2);
1111    } else {
1112      qCommon = ByteBufferUtils
1113          .findCommonPrefix(c1.getQualifierArray(), c1.getQualifierOffset(), qLen1,
1114              c2.getQualifierArray(), c2.getQualifierOffset(), qLen2);
1115    }
1116    commonPrefix += qCommon;
1117    if (!withTsType || Math.max(qLen1, qLen2) != qCommon) {
1118      return commonPrefix;
1119    }
1120    // Compare the timestamp parts
1121    int tsCommonPrefix = ByteBufferUtils
1122        .findCommonPrefix(Bytes.toBytes(c1.getTimestamp()), 0, KeyValue.TIMESTAMP_SIZE,
1123            Bytes.toBytes(c2.getTimestamp()), 0, KeyValue.TIMESTAMP_SIZE);
1124    commonPrefix += tsCommonPrefix;
1125    if (tsCommonPrefix != KeyValue.TIMESTAMP_SIZE) {
1126      return commonPrefix;
1127    }
1128    // Compare the type
1129    if (c1.getTypeByte() == c2.getTypeByte()) {
1130      commonPrefix += KeyValue.TYPE_SIZE;
1131    }
1132    return commonPrefix;
1133  }

1135  /**
1136   * Used to compare two cells based on the column hint provided. This is specifically used when we
1137   * need to optimize the seeks based on the next indexed key. This is an advanced usage API
1138   * specifically needed for some optimizations.
1139   * @param nextIndexedCell the next indexed cell
1140   * @param currentCell the cell to be compared
1141   * @param foff the family offset of the currentCell
1142   * @param flen the family length of the currentCell
1143   * @param colHint the column hint provided - could be null
1144   * @param coff the offset of the column hint if provided, if not offset of the currentCell's
1145   *          qualifier
1146   * @param clen the length of the column hint if provided, if not length of the currentCell's
1147   *          qualifier
1148   * @param ts the timestamp to be seeked
1149   * @param type the type to be seeked
1150   * @return an int based on the given column hint TODO : To be moved out of here because this is a
1151   *         special API used in scan optimization.
1152   */
1153  // compare a key against row/fam/qual/ts/type
1154  public static final int compareKeyBasedOnColHint(CellComparator comparator, Cell nextIndexedCell,
1155      Cell currentCell, int foff, int flen, byte[] colHint, int coff, int clen, long ts,
1156      byte type) {
1157    int compare = comparator.compareRows(nextIndexedCell, currentCell);
1158    if (compare != 0) {
1159      return compare;
1160    }
1161    // If the column is not specified, the "minimum" key type appears the
1162    // latest in the sorted order, regardless of the timestamp. This is used
1163    // for specifying the last key/value in a given row, because there is no
1164    // "lexicographically last column" (it would be infinitely long). The
1165    // "maximum" key type does not need this behavior.
1166    if (nextIndexedCell.getFamilyLength() + nextIndexedCell.getQualifierLength() == 0
1167        && nextIndexedCell.getTypeByte() == KeyValue.Type.Minimum.getCode()) {
1168      // left is "bigger", i.e. it appears later in the sorted order
1169      return 1;
1170    }
1171    if (flen + clen == 0 && type == KeyValue.Type.Minimum.getCode()) {
1172      return -1;
1173    }
1174
1175    compare = comparator.compareFamilies(nextIndexedCell, currentCell);
1176    if (compare != 0) {
1177      return compare;
1178    }
1179    if (colHint == null) {
1180      compare = comparator.compareQualifiers(nextIndexedCell, currentCell);
1181    } else {
1182      compare = CellUtil.compareQualifiers(nextIndexedCell, colHint, coff, clen);
1183    }
1184    if (compare != 0) {
1185      return compare;
1186    }
1187    // Next compare timestamps.
1188    compare = comparator.compareTimestamps(nextIndexedCell.getTimestamp(), ts);
1189    if (compare != 0) {
1190      return compare;
1191    }
1192
1193    // Compare types. Let the delete types sort ahead of puts; i.e. types
1194    // of higher numbers sort before those of lesser numbers. Maximum (255)
1195    // appears ahead of everything, and minimum (0) appears after
1196    // everything.
1197    return (0xff & type) - (0xff & nextIndexedCell.getTypeByte());
1198  }

2145    byte fLen = cell.getFamilyLength();
2146    int qLen = cell.getQualifierLength();

2155        ((ByteBufferExtendedCell) cell).getFamilyPosition(), fLen);

2158          ((ByteBufferExtendedCell) cell).getQualifierPosition(), qLen);

2590  public static final int compare(CellComparator comparator, Cell left, byte[] key, int offset,

2593    short rrowlength = Bytes.toShort(key, offset);

2595    if (c != 0) return c;

2603  /**
2604   * Compare columnFamily, qualifier, timestamp, and key type (everything except the row). This
2605   * method is used both in the normal comparator and the "same-prefix" comparator. Note that we are
2606   * assuming that row portions of both KVs have already been parsed and found identical, and we
2607   * don't validate that assumption here.
2608   * @param comparator the {@link CellComparator} to use for comparison
2609   * @param left the cell to be compared
2610   * @param right the serialized key part of a key-value
2611   * @param roffset the offset in the key byte[]
2612   * @param rlength the length of the key byte[]
2613   * @param rowlength the row length
2614   * @return greater than 0 if left cell is bigger, less than 0 if right cell is bigger, 0 if both
2615   *         cells are equal
2616   */
2617  static final int compareWithoutRow(CellComparator comparator, Cell left, byte[] right,
2618      int roffset, int rlength, short rowlength) {
2619    /***
2620     * KeyValue Format and commonLength:
2621     * |_keyLen_|_valLen_|_rowLen_|_rowKey_|_famiLen_|_fami_|_Quali_|....
2622     * ------------------|-------commonLength--------|--------------
2623     */
2624    int commonLength = KeyValue.ROW_LENGTH_SIZE + KeyValue.FAMILY_LENGTH_SIZE + rowlength;
2625
2626    // commonLength + TIMESTAMP_TYPE_SIZE
2627    int commonLengthWithTSAndType = KeyValue.TIMESTAMP_TYPE_SIZE + commonLength;
2628    // ColumnFamily + Qualifier length.
2629    int lcolumnlength = left.getFamilyLength() + left.getQualifierLength();
2630    int rcolumnlength = rlength - commonLengthWithTSAndType;
2631
2632    byte ltype = left.getTypeByte();
2633    byte rtype = right[roffset + (rlength - 1)];
2634
2635    // If the column is not specified, the "minimum" key type appears the
2636    // latest in the sorted order, regardless of the timestamp. This is used
2637    // for specifying the last key/value in a given row, because there is no
2638    // "lexicographically last column" (it would be infinitely long). The
2639    // "maximum" key type does not need this behavior.
2640    if (lcolumnlength == 0 && ltype == KeyValue.Type.Minimum.getCode()) {
2641      // left is "bigger", i.e. it appears later in the sorted order
2642      return 1;
2643    }
2644    if (rcolumnlength == 0 && rtype == KeyValue.Type.Minimum.getCode()) {
2645      return -1;
2646    }
2647
2648    int rfamilyoffset = commonLength + roffset;
2649
2650    // Column family length.
2651    int lfamilylength = left.getFamilyLength();
2652    int rfamilylength = right[rfamilyoffset - 1];
2653    // If left family size is not equal to right family size, we need not
2654    // compare the qualifiers.
2655    boolean sameFamilySize = (lfamilylength == rfamilylength);
2656    if (!sameFamilySize) {
2657      // comparing column family is enough.
2658      return CellUtil.compareFamilies(left, right, rfamilyoffset, rfamilylength);
2659    }
2660    // Compare family & qualifier together.
2661    // Families are same. Compare on qualifiers.
2662    int comparison = CellUtil.compareColumns(left, right, rfamilyoffset, rfamilylength,
2663      rfamilyoffset + rfamilylength, (rcolumnlength - rfamilylength));
2664    if (comparison != 0) {
2665      return comparison;
2666    }
2667
2668    // //
2669    // Next compare timestamps.
2670    long rtimestamp = Bytes.toLong(right, roffset + (rlength - KeyValue.TIMESTAMP_TYPE_SIZE));
2671    int compare = comparator.compareTimestamps(left.getTimestamp(), rtimestamp);
2672    if (compare != 0) {
2673      return compare;
2674    }
2675
2676    // Compare types. Let the delete types sort ahead of puts; i.e. types
2677    // of higher numbers sort before those of lesser numbers. Maximum (255)
2678    // appears ahead of everything, and minimum (0) appears after
2679    // everything.
2680    return (0xff & rtype) - (0xff & ltype);
2681  }
