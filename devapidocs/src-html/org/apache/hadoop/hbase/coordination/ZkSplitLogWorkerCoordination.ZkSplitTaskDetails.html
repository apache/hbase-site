096  public void nodeChildrenChanged(String path) {

103        taskReadySeq.notify();
104      }

109   * Override handler from {@link ZooKeeperListener}
110   */
111  @Override
112  public void nodeDataChanged(String path) {
113    // there will be a self generated dataChanged event every time attemptToOwnTask()
114    // heartbeats the task znode by upping its version
115    synchronized (grabTaskLock) {
116      if (workerInGrabTask) {
117        // currentTask can change
118        String taskpath = currentTask;
119        if (taskpath != null && taskpath.equals(path)) {
120          getDataSetWatchAsync();
121        }
122      }
123    }
124  }

144  void getDataSetWatchFailure(String path) {
145    synchronized (grabTaskLock) {
146      if (workerInGrabTask) {
147        // currentTask can change but that's ok
148        String taskpath = currentTask;
149        if (taskpath != null && taskpath.equals(path)) {
150          LOG.info("retrying data watch on " + path);
151          SplitLogCounters.tot_wkr_get_data_retry.increment();
152          getDataSetWatchAsync();
153        } else {
154          // no point setting a watch on the task which this worker is not
155          // working upon anymore
156        }
157      }
158    }
159  }

161  public void getDataSetWatchAsync() {
162    watcher.getRecoverableZooKeeper().getZooKeeper()
163        .getData(currentTask, watcher, new GetDataAsyncCallback(), null);
164    SplitLogCounters.tot_wkr_get_data_queued.increment();
165  }

167  void getDataSetWatchSuccess(String path, byte[] data) {
168    SplitLogTask slt;
169    try {
170      slt = SplitLogTask.parseFrom(data);
171    } catch (DeserializationException e) {
172      LOG.warn("Failed parse", e);
173      return;
174    }
175    synchronized (grabTaskLock) {
176      if (workerInGrabTask) {
177        // currentTask can change but that's ok
178        String taskpath = currentTask;
179        if (taskpath != null && taskpath.equals(path)) {
180          // have to compare data. cannot compare version because then there
181          // will be race with attemptToOwnTask()
182          // cannot just check whether the node has been transitioned to
183          // UNASSIGNED because by the time this worker sets the data watch
184          // the node might have made two transitions - from owned by this
185          // worker to unassigned to owned by another worker
186          if (!slt.isOwned(serverName) && !slt.isDone(serverName) && !slt.isErr(serverName)
187              && !slt.isResigned(serverName)) {
188            LOG.info("task " + taskpath + " preempted from " + serverName
189                + ", current task state and owner=" + slt.toString());
190            worker.stopTask();
191          }
192        }
193      }
194    }
195  }

197  /**
198   * try to grab a 'lock' on the task zk node to own and execute the task.
199   * <p>
200   * @param path zk node for the task
201   */
202  private void grabTask(String path) {
203    Stat stat = new Stat();
204    byte[] data;
205    synchronized (grabTaskLock) {
206      currentTask = path;
207      workerInGrabTask = true;
208      if (Thread.interrupted()) {
209        return;
210      }
211    }
212    try {
213      try {
214        if ((data = ZKUtil.getDataNoWatch(watcher, path, stat)) == null) {
215          SplitLogCounters.tot_wkr_failed_to_grab_task_no_data.increment();
216          return;
217        }
218      } catch (KeeperException e) {
219        LOG.warn("Failed to get data for znode " + path, e);
220        SplitLogCounters.tot_wkr_failed_to_grab_task_exception.increment();
221        return;
222      }
223      SplitLogTask slt;
224      try {
225        slt = SplitLogTask.parseFrom(data);
226      } catch (DeserializationException e) {
227        LOG.warn("Failed parse data for znode " + path, e);
228        SplitLogCounters.tot_wkr_failed_to_grab_task_exception.increment();
229        return;
230      }
231      if (!slt.isUnassigned()) {
232        SplitLogCounters.tot_wkr_failed_to_grab_task_owned.increment();
233        return;
234      }
235
236      currentVersion =
237          attemptToOwnTask(true, watcher, server.getServerName(), path, stat.getVersion());
238      if (currentVersion < 0) {
239        SplitLogCounters.tot_wkr_failed_to_grab_task_lost_race.increment();
240        return;
241      }
242
243      if (ZKSplitLog.isRescanNode(watcher, currentTask)) {
244        ZkSplitLogWorkerCoordination.ZkSplitTaskDetails splitTaskDetails =
245            new ZkSplitLogWorkerCoordination.ZkSplitTaskDetails();
246        splitTaskDetails.setTaskNode(currentTask);
247        splitTaskDetails.setCurTaskZKVersion(new MutableInt(currentVersion));
248
249        endTask(new SplitLogTask.Done(server.getServerName()),
250          SplitLogCounters.tot_wkr_task_acquired_rescan, splitTaskDetails);
251        return;
252      }
253
254      LOG.info("worker " + server.getServerName() + " acquired task " + path);
255      SplitLogCounters.tot_wkr_task_acquired.increment();
256      getDataSetWatchAsync();
257
258      submitTask(path, currentVersion, reportPeriod);
259
260      // after a successful submit, sleep a little bit to allow other RSs to grab the rest tasks
261      try {
262        int sleepTime = RandomUtils.nextInt(0, 500) + 500;
263        Thread.sleep(sleepTime);
264      } catch (InterruptedException e) {
265        LOG.warn("Interrupted while yielding for other region servers", e);
266        Thread.currentThread().interrupt();
267      }
268    } finally {
269      synchronized (grabTaskLock) {
270        workerInGrabTask = false;
271        // clear the interrupt from stopTask() otherwise the next task will
272        // suffer
273        Thread.interrupted();
274      }
275    }
276  }

278  /**
279   * Submit a log split task to executor service
280   * @param curTask task to submit
281   * @param curTaskZKVersion current version of task
282   */
283  void submitTask(final String curTask, final int curTaskZKVersion, final int reportPeriod) {
284    final MutableInt zkVersion = new MutableInt(curTaskZKVersion);
285
286    CancelableProgressable reporter = new CancelableProgressable() {
287      private long last_report_at = 0;
288
289      @Override
290      public boolean progress() {
291        long t = EnvironmentEdgeManager.currentTime();
292        if ((t - last_report_at) > reportPeriod) {
293          last_report_at = t;
294          int latestZKVersion =
295              attemptToOwnTask(false, watcher, server.getServerName(), curTask,
296                zkVersion.intValue());
297          if (latestZKVersion < 0) {
298            LOG.warn("Failed to heartbeat the task" + curTask);
299            return false;
300          }
301          zkVersion.setValue(latestZKVersion);
302        }
303        return true;
304      }
305    };
306    ZkSplitLogWorkerCoordination.ZkSplitTaskDetails splitTaskDetails =
307        new ZkSplitLogWorkerCoordination.ZkSplitTaskDetails();
308    splitTaskDetails.setTaskNode(curTask);
309    splitTaskDetails.setCurTaskZKVersion(zkVersion);
310
311    WALSplitterHandler hsh =
312        new WALSplitterHandler(server, this, splitTaskDetails, reporter,
313            this.tasksInProgress, splitTaskExecutor);
314    server.getExecutorService().submit(hsh);
315  }

317  /**
318   * This function calculates how many splitters it could create based on expected average tasks per
319   * RS and the hard limit upper bound(maxConcurrentTasks) set by configuration. <br>
320   * At any given time, a RS allows spawn MIN(Expected Tasks/RS, Hard Upper Bound)
321   * @param numTasks current total number of available tasks
322   */
323  private int calculateAvailableSplitters(int numTasks) {
324    // at lease one RS(itself) available
325    int availableRSs = 1;
326    try {
327      List<String> regionServers =
328          ZKUtil.listChildrenNoWatch(watcher, watcher.znodePaths.rsZNode);
329      availableRSs = Math.max(availableRSs, (regionServers == null) ? 0 : regionServers.size());
330    } catch (KeeperException e) {
331      // do nothing
332      LOG.debug("getAvailableRegionServers got ZooKeeper exception", e);
333    }
334
335    int expectedTasksPerRS = (numTasks / availableRSs) + ((numTasks % availableRSs == 0) ? 0 : 1);
336    expectedTasksPerRS = Math.max(1, expectedTasksPerRS); // at least be one
337    // calculate how many more splitters we could spawn
338    return Math.min(expectedTasksPerRS, maxConcurrentTasks)
339        - this.tasksInProgress.get();
340  }

342  /**
343   * Try to own the task by transitioning the zk node data from UNASSIGNED to OWNED.
344   * <p>
345   * This method is also used to periodically heartbeat the task progress by transitioning the node
346   * from OWNED to OWNED.
347   * <p>
348   * @param isFirstTime shows whther it's the first attempt.
349   * @param zkw zk wathcer
350   * @param server name
351   * @param task to own
352   * @param taskZKVersion version of the task in zk
353   * @return non-negative integer value when task can be owned by current region server otherwise -1
354   */
355  protected static int attemptToOwnTask(boolean isFirstTime, ZooKeeperWatcher zkw,
356      ServerName server, String task, int taskZKVersion) {
357    int latestZKVersion = FAILED_TO_OWN_TASK;
358    try {
359      SplitLogTask slt = new SplitLogTask.Owned(server);
360      Stat stat = zkw.getRecoverableZooKeeper().setData(task, slt.toByteArray(), taskZKVersion);
361      if (stat == null) {
362        LOG.warn("zk.setData() returned null for path " + task);
363        SplitLogCounters.tot_wkr_task_heartbeat_failed.increment();
364        return FAILED_TO_OWN_TASK;
365      }
366      latestZKVersion = stat.getVersion();
367      SplitLogCounters.tot_wkr_task_heartbeat.increment();
368      return latestZKVersion;
369    } catch (KeeperException e) {
370      if (!isFirstTime) {
371        if (e.code().equals(KeeperException.Code.NONODE)) {
372          LOG.warn("NONODE failed to assert ownership for " + task, e);
373        } else if (e.code().equals(KeeperException.Code.BADVERSION)) {
374          LOG.warn("BADVERSION failed to assert ownership for " + task, e);
375        } else {
376          LOG.warn("failed to assert ownership for " + task, e);
377        }
378      }
379    } catch (InterruptedException e1) {
380      LOG.warn("Interrupted while trying to assert ownership of " + task + " "
381          + StringUtils.stringifyException(e1));
382      Thread.currentThread().interrupt();
383    }
384    SplitLogCounters.tot_wkr_task_heartbeat_failed.increment();
385    return FAILED_TO_OWN_TASK;
386  }

388  /**
389   * Wait for tasks to become available at /hbase/splitlog zknode. Grab a task one at a time. This
390   * policy puts an upper-limit on the number of simultaneous log splitting that could be happening
391   * in a cluster.
392   * <p>
393   * Synchronization using <code>taskReadySeq</code> ensures that it will try to grab every task
394   * that has been put up
395   * @throws InterruptedException
396   */
397  @Override
398  public void taskLoop() throws InterruptedException {
399    while (!shouldStop) {
400      int seq_start = taskReadySeq.get();
401      List<String> paths;
402      paths = getTaskList();
403      if (paths == null) {
404        LOG.warn("Could not get tasks, did someone remove " + watcher.znodePaths.splitLogZNode
405            + " ... worker thread exiting.");
406        return;
407      }
408      // pick meta wal firstly
409      int offset = (int) (Math.random() * paths.size());
410      for (int i = 0; i < paths.size(); i++) {
411        if (AbstractFSWALProvider.isMetaFile(paths.get(i))) {
412          offset = i;
413          break;
414        }
415      }
416      int numTasks = paths.size();
417      for (int i = 0; i < numTasks; i++) {
418        int idx = (i + offset) % paths.size();
419        // don't call ZKSplitLog.getNodeName() because that will lead to
420        // double encoding of the path name
421        if (this.calculateAvailableSplitters(numTasks) > 0) {
422          grabTask(ZKUtil.joinZNode(watcher.znodePaths.splitLogZNode, paths.get(idx)));
423        } else {
424          LOG.debug("Current region server " + server.getServerName() + " has "
425              + this.tasksInProgress.get() + " tasks in progress and can't take more.");
426          break;
427        }
428        if (shouldStop) {
429          return;
430        }
431      }
432      SplitLogCounters.tot_wkr_task_grabing.increment();
433      synchronized (taskReadySeq) {
434        while (seq_start == taskReadySeq.get()) {
435          taskReadySeq.wait(checkInterval);
436        }
437      }
438    }
439  }

441  private List<String> getTaskList() throws InterruptedException {
442    List<String> childrenPaths = null;
443    long sleepTime = 1000;
444    // It will be in loop till it gets the list of children or
445    // it will come out if worker thread exited.
446    while (!shouldStop) {
447      try {
448        childrenPaths = ZKUtil.listChildrenAndWatchForNewChildren(watcher,
449          watcher.znodePaths.splitLogZNode);
450        if (childrenPaths != null) {
451          return childrenPaths;
452        }
453      } catch (KeeperException e) {
454        LOG.warn("Could not get children of znode " + watcher.znodePaths.splitLogZNode, e);
455      }
456      LOG.debug("Retry listChildren of znode " + watcher.znodePaths.splitLogZNode
457          + " after sleep for " + sleepTime + "ms!");
458      Thread.sleep(sleepTime);
459    }
460    return childrenPaths;
461  }

468  @Override
469  public boolean isReady() throws InterruptedException {
470    int result = -1;
471    try {
472      result = ZKUtil.checkExists(watcher, watcher.znodePaths.splitLogZNode);
473    } catch (KeeperException e) {
474      // ignore
475      LOG.warn("Exception when checking for " + watcher.znodePaths.splitLogZNode
476          + " ... retrying", e);
477    }
478    if (result == -1) {
479      LOG.info(watcher.znodePaths.splitLogZNode
480          + " znode does not exist, waiting for master to create");
481      Thread.sleep(1000);
482    }
483    return (result != -1);
484  }

513  /**
514   * Asynchronous handler for zk get-data-set-watch on node results.
515   */
516  class GetDataAsyncCallback implements AsyncCallback.DataCallback {
517    private final Log LOG = LogFactory.getLog(GetDataAsyncCallback.class);
518
519    @Override
520    public void processResult(int rc, String path, Object ctx, byte[] data, Stat stat) {
521      SplitLogCounters.tot_wkr_get_data_result.increment();
522      if (rc != 0) {
523        LOG.warn("getdata rc = " + KeeperException.Code.get(rc) + " " + path);
524        getDataSetWatchFailure(path);
525        return;
526      }
527      data = RecoverableZooKeeper.removeMetaData(data);
528      getDataSetWatchSuccess(path, data);
529    }
530  }

535  /**
536   * endTask() can fail and the only way to recover out of it is for the
537   * {@link org.apache.hadoop.hbase.master.SplitLogManager} to timeout the task node.
538   * @param slt
539   * @param ctr
540   */
541  @Override
542  public void endTask(SplitLogTask slt, LongAdder ctr, SplitTaskDetails details) {
543    ZkSplitTaskDetails zkDetails = (ZkSplitTaskDetails) details;
544    String task = zkDetails.getTaskNode();
545    int taskZKVersion = zkDetails.getCurTaskZKVersion().intValue();
546    try {
547      if (ZKUtil.setData(watcher, task, slt.toByteArray(), taskZKVersion)) {
548        LOG.info("successfully transitioned task " + task + " to final state " + slt);
549        ctr.increment();
550        return;
551      }
552      LOG.warn("failed to transistion task " + task + " to end state " + slt
553          + " because of version mismatch ");
554    } catch (KeeperException.BadVersionException bve) {
555      LOG.warn("transisition task " + task + " to " + slt + " failed because of version mismatch",
556        bve);
557    } catch (KeeperException.NoNodeException e) {
558      LOG.fatal(
559        "logic error - end task " + task + " " + slt + " failed because task doesn't exist", e);
560    } catch (KeeperException e) {
561      LOG.warn("failed to end task, " + task + " " + slt, e);
562    }
563    SplitLogCounters.tot_wkr_final_transition_failed.increment();
564  }
