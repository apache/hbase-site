049/**
050 * MapReduce implementation of {@link BackupMergeJob}
051 * Must be initialized with configuration of a backup destination cluster
052 *
053 */
054
055@InterfaceAudience.Private
056public class MapReduceBackupMergeJob implements BackupMergeJob {
057  public static final Log LOG = LogFactory.getLog(MapReduceBackupMergeJob.class);
058
059  protected Tool player;
060  protected Configuration conf;
061
062  public MapReduceBackupMergeJob() {
063  }
064
065  @Override
066  public Configuration getConf() {
067    return conf;
068  }
069
070  @Override
071  public void setConf(Configuration conf) {
072    this.conf = conf;
073  }
074
075  @Override
076  public void run(String[] backupIds) throws IOException {
077    String bulkOutputConfKey;
078
079    // TODO : run player on remote cluster
080    player = new MapReduceHFileSplitterJob();
081    bulkOutputConfKey = MapReduceHFileSplitterJob.BULK_OUTPUT_CONF_KEY;
082    // Player reads all files in arbitrary directory structure and creates
083    // a Map task for each file
084    String bids = StringUtils.join(backupIds, ",");
085
086    if (LOG.isDebugEnabled()) {
087      LOG.debug("Merge backup images " + bids);
088    }
089
090    List<Pair<TableName, Path>> processedTableList = new ArrayList<Pair<TableName, Path>>();
091    boolean finishedTables = false;
092    Connection conn = ConnectionFactory.createConnection(getConf());
093    BackupSystemTable table = new BackupSystemTable(conn);
094    FileSystem fs = FileSystem.get(getConf());
095
096    try {
097
098      // Get exclusive lock on backup system
099      table.startBackupExclusiveOperation();
100      // Start merge operation
101      table.startMergeOperation(backupIds);
102
103      // Select most recent backup id
104      String mergedBackupId = findMostRecentBackupId(backupIds);
105
106      TableName[] tableNames = getTableNamesInBackupImages(backupIds);
107      String backupRoot = null;
108
109      BackupInfo bInfo = table.readBackupInfo(backupIds[0]);
110      backupRoot = bInfo.getBackupRootDir();
111
112      for (int i = 0; i < tableNames.length; i++) {
113
114        LOG.info("Merge backup images for " + tableNames[i]);
115
116        // Find input directories for table
117
118        Path[] dirPaths = findInputDirectories(fs, backupRoot, tableNames[i], backupIds);
119        String dirs = StringUtils.join(dirPaths, ",");
120        Path bulkOutputPath =
121            BackupUtils.getBulkOutputDir(BackupUtils.getFileNameCompatibleString(tableNames[i]),
122              getConf(), false);
123        // Delete content if exists
124        if (fs.exists(bulkOutputPath)) {
125          if (!fs.delete(bulkOutputPath, true)) {
126            LOG.warn("Can not delete: " + bulkOutputPath);
127          }
128        }
129        Configuration conf = getConf();
130        conf.set(bulkOutputConfKey, bulkOutputPath.toString());
131        String[] playerArgs = { dirs, tableNames[i].getNameAsString() };
132
133        int result = 0;
134
135        player.setConf(getConf());
136        result = player.run(playerArgs);
137        if (!succeeded(result)) {
138          throw new IOException("Can not merge backup images for " + dirs
139              + " (check Hadoop/MR and HBase logs). Player return code =" + result);
140        }
141        // Add to processed table list
142        processedTableList.add(new Pair<TableName, Path>(tableNames[i], bulkOutputPath));
143        LOG.debug("Merge Job finished:" + result);
144      }
145      List<TableName> tableList = toTableNameList(processedTableList);
146      table.updateProcessedTablesForMerge(tableList);
147      finishedTables = true;
148
149      // Move data
150      for (Pair<TableName, Path> tn : processedTableList) {
151        moveData(fs, backupRoot, tn.getSecond(), tn.getFirst(), mergedBackupId);
152      }
153
154      // Delete old data and update manifest
155      List<String> backupsToDelete = getBackupIdsToDelete(backupIds, mergedBackupId);
156      deleteBackupImages(backupsToDelete, conn, fs, backupRoot);
157      updateBackupManifest(backupRoot, mergedBackupId, backupsToDelete);
158      // Finish merge session
159      table.finishMergeOperation();
160      // Release lock
161      table.finishBackupExclusiveOperation();
162    } catch (RuntimeException e) {
163
164      throw e;
165    } catch (Exception e) {
166      LOG.error(e);
167      if (!finishedTables) {
168        // cleanup bulk directories and finish merge
169        // merge MUST be repeated (no need for repair)
170        cleanupBulkLoadDirs(fs, toPathList(processedTableList));
171        table.finishMergeOperation();
172        table.finishBackupExclusiveOperation();
173        throw new IOException("Backup merge operation failed, you should try it again", e);
174      } else {
175        // backup repair must be run
176        throw new IOException(
177            "Backup merge operation failed, run backup repair tool to restore system's integrity",
178            e);
179      }
180    } finally {
181      table.close();
182      conn.close();
183    }
184  }
185
186  protected List<Path> toPathList(List<Pair<TableName, Path>> processedTableList) {
187    ArrayList<Path> list = new ArrayList<Path>();
188    for (Pair<TableName, Path> p : processedTableList) {
189      list.add(p.getSecond());
190    }
191    return list;
192  }
193
194  protected List<TableName> toTableNameList(List<Pair<TableName, Path>> processedTableList) {
195    ArrayList<TableName> list = new ArrayList<TableName>();
196    for (Pair<TableName, Path> p : processedTableList) {
197      list.add(p.getFirst());
198    }
199    return list;
200  }
201
202  protected void cleanupBulkLoadDirs(FileSystem fs, List<Path> pathList) throws IOException {
203    for (Path path : pathList) {
204
205      if (!fs.delete(path, true)) {
206        LOG.warn("Can't delete " + path);
207      }
208    }
209  }
210
211  protected void updateBackupManifest(String backupRoot, String mergedBackupId,
212      List<String> backupsToDelete) throws IllegalArgumentException, IOException {
213
214    BackupManifest manifest =
215        HBackupFileSystem.getManifest(conf, new Path(backupRoot), mergedBackupId);
216    manifest.getBackupImage().removeAncestors(backupsToDelete);
217    // save back
218    manifest.store(conf);
219
220  }
221
222  protected void deleteBackupImages(List<String> backupIds, Connection conn, FileSystem fs,
223      String backupRoot) throws IOException {
224
225    // Delete from backup system table
226    try (BackupSystemTable table = new BackupSystemTable(conn);) {
227      for (String backupId : backupIds) {
228        table.deleteBackupInfo(backupId);
229      }
230    }
231
232    // Delete from file system
233    for (String backupId : backupIds) {
234      Path backupDirPath = HBackupFileSystem.getBackupPath(backupRoot, backupId);
235
236      if (!fs.delete(backupDirPath, true)) {
237        LOG.warn("Could not delete " + backupDirPath);
238      }
239    }
240  }
241
242  protected List<String> getBackupIdsToDelete(String[] backupIds, String mergedBackupId) {
243    List<String> list = new ArrayList<String>();
244    for (String id : backupIds) {
245      if (id.equals(mergedBackupId)) {
246        continue;
247      }
248      list.add(id);
249    }
250    return list;
251  }
252
253  protected void moveData(FileSystem fs, String backupRoot, Path bulkOutputPath, TableName tableName,
254      String mergedBackupId) throws IllegalArgumentException, IOException {
255
256    Path dest =
257        new Path(HBackupFileSystem.getTableBackupDataDir(backupRoot, mergedBackupId, tableName));
258
259    // Delete all in dest
260    if (!fs.delete(dest, true)) {
261      throw new IOException("Could not delete " + dest);
262    }
263
264    FileStatus[] fsts = fs.listStatus(bulkOutputPath);
265    for (FileStatus fst : fsts) {
266      if (fst.isDirectory()) {
267        fs.rename(fst.getPath().getParent(), dest);
268      }
269    }
270
271  }
272
273  protected String findMostRecentBackupId(String[] backupIds) {
274    long recentTimestamp = Long.MIN_VALUE;
275    for (String backupId : backupIds) {
276      long ts = Long.parseLong(backupId.split("_")[1]);
277      if (ts > recentTimestamp) {
278        recentTimestamp = ts;
279      }
280    }
281    return BackupRestoreConstants.BACKUPID_PREFIX + recentTimestamp;
282  }
283
284  protected TableName[] getTableNamesInBackupImages(String[] backupIds) throws IOException {
285
286    Set<TableName> allSet = new HashSet<TableName>();
287
288    try (Connection conn = ConnectionFactory.createConnection(conf);
289        BackupSystemTable table = new BackupSystemTable(conn);) {
290      for (String backupId : backupIds) {
291        BackupInfo bInfo = table.readBackupInfo(backupId);
292
293        allSet.addAll(bInfo.getTableNames());
294      }
295    }
296
297    TableName[] ret = new TableName[allSet.size()];
298    return allSet.toArray(ret);
299  }
300
301  protected Path[] findInputDirectories(FileSystem fs, String backupRoot, TableName tableName,
302      String[] backupIds) throws IOException {
303
304    List<Path> dirs = new ArrayList<Path>();
305
306    for (String backupId : backupIds) {
307      Path fileBackupDirPath =
308          new Path(HBackupFileSystem.getTableBackupDataDir(backupRoot, backupId, tableName));
309      if (fs.exists(fileBackupDirPath)) {
310        dirs.add(fileBackupDirPath);
311      } else {
312        if (LOG.isTraceEnabled()) {
313          LOG.trace("File: " + fileBackupDirPath + " does not exist.");
314        }
315      }
316    }
317    Path[] ret = new Path[dirs.size()];
318    return dirs.toArray(ret);
319  }
