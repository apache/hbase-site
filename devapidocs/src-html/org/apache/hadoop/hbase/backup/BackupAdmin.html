019package org.apache.hadoop.hbase.backup;
020
021import java.io.Closeable;
022import java.io.IOException;
023import java.util.List;
024
025import org.apache.hadoop.hbase.TableName;
026import org.apache.hadoop.hbase.backup.util.BackupSet;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028
029/**
030 * The administrative API for HBase Backup. Construct an instance and call {@link #close()}
031 * afterwards.
032 * <p>
033 * BackupAdmin can be used to create backups, restore data from backups and for other
034 * backup-related operations.
035 * @since 2.0
036 */
037@InterfaceAudience.Private
038public interface BackupAdmin extends Closeable {
039
040  /**
041   * Backup given list of tables fully. This is a synchronous operation.
042   * It returns backup id on success or throw exception on failure.
043   * @param userRequest BackupRequest instance
044   * @return the backup Id
045   */
046
047  String backupTables(final BackupRequest userRequest) throws IOException;
048
049  /**
050   * Restore backup
051   * @param request restore request
052   * @throws IOException exception
053   */
054  void restore(RestoreRequest request) throws IOException;
055
056  /**
057   * Describe backup image command
058   * @param backupId backup id
059   * @return backup info
060   * @throws IOException exception
061   */
062  BackupInfo getBackupInfo(String backupId) throws IOException;
063
064
065  /**
066   * Delete backup image command
067   * @param backupIds backup id list
068   * @return total number of deleted sessions
069   * @throws IOException exception
070   */
071  int deleteBackups(String[] backupIds) throws IOException;
072
073  /**
074   * Show backup history command
075   * @param n last n backup sessions
076   * @return list of backup info objects
077   * @throws IOException exception
078   */
079  List<BackupInfo> getHistory(int n) throws IOException;
080
081  /**
082   * Show backup history command with filters
083   * @param n last n backup sessions
084   * @param f list of filters
085   * @return list of backup info objects
086   * @throws IOException exception
087   */
088  List<BackupInfo> getHistory(int n, BackupInfo.Filter... f) throws IOException;
089
090  /**
091   * Backup sets list command - list all backup sets. Backup set is a named group of tables.
092   * @return all registered backup sets
093   * @throws IOException exception
094   */
095  List<BackupSet> listBackupSets() throws IOException;
096
097  /**
098   * Backup set describe command. Shows list of tables in this particular backup set.
099   * @param name set name
100   * @return backup set description or null
101   * @throws IOException exception
102   */
103  BackupSet getBackupSet(String name) throws IOException;
104
105  /**
106   * Delete backup set command
107   * @param name backup set name
108   * @return true, if success, false - otherwise
109   * @throws IOException exception
110   */
111  boolean deleteBackupSet(String name) throws IOException;
112
113  /**
114   * Add tables to backup set command
115   * @param name name of backup set.
116   * @param tables list of tables to be added to this set.
117   * @throws IOException exception
118   */
119  void addToBackupSet(String name, TableName[] tables) throws IOException;
120
121  /**
122   * Remove tables from backup set
123   * @param name name of backup set.
124   * @param tables list of tables to be removed from this set.
125   * @throws IOException exception
126   */
127  void removeFromBackupSet(String name, TableName[] tables) throws IOException;
128}
