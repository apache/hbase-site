062/**
063 *
064 * Command-line entry point for restore operation
065 *
066 */
067@InterfaceAudience.Private
068public class RestoreDriver extends AbstractHBaseTool {
069  private static final Logger LOG = LoggerFactory.getLogger(RestoreDriver.class);
070  private CommandLine cmd;
071
072  private static final String USAGE_STRING =
073      "Usage: hbase restore <backup_path> <backup_id> [options]\n"
074          + "  backup_path     Path to a backup destination root\n"
075          + "  backup_id       Backup image ID to restore\n"
076          + "  table(s)        Comma-separated list of tables to restore\n";
077
078  private static final String USAGE_FOOTER = "";
079
080  protected RestoreDriver() throws IOException {
081    init();
082  }
083
084  protected void init() {
085    // disable irrelevant loggers to avoid it mess up command output
086    LogUtils.disableZkAndClientLoggers();
087  }
088
089  private int parseAndRun(String[] args) throws IOException {
090    // Check if backup is enabled
091    if (!BackupManager.isBackupEnabled(getConf())) {
092      System.err.println(BackupRestoreConstants.ENABLE_BACKUP);
093      return -1;
094    }
095
096    System.out.println(BackupRestoreConstants.VERIFY_BACKUP);
097
098    // enable debug logging
099    if (cmd.hasOption(OPTION_DEBUG)) {
100      LogManager.getLogger("org.apache.hadoop.hbase.backup").setLevel(Level.DEBUG);
101    }
102
103    // whether to overwrite to existing table if any, false by default
104    boolean overwrite = cmd.hasOption(OPTION_OVERWRITE);
105    if (overwrite) {
106      LOG.debug("Found -overwrite option in restore command, "
107          + "will overwrite to existing table if any in the restore target");
108    }
109
110    // whether to only check the dependencies, false by default
111    boolean check = cmd.hasOption(OPTION_CHECK);
112    if (check) {
113      LOG.debug("Found -check option in restore command, "
114          + "will check and verify the dependencies");
115    }
116
117    if (cmd.hasOption(OPTION_SET) && cmd.hasOption(OPTION_TABLE)) {
118      System.err.println("Options -s and -t are mutaully exclusive,"+
119          " you can not specify both of them.");
120      printToolUsage();
121      return -1;
122    }
123
124    if (!cmd.hasOption(OPTION_SET) && !cmd.hasOption(OPTION_TABLE)) {
125      System.err.println("You have to specify either set name or table list to restore");
126      printToolUsage();
127      return -1;
128    }
129
130    if (cmd.hasOption(OPTION_YARN_QUEUE_NAME)) {
131      String queueName = cmd.getOptionValue(OPTION_YARN_QUEUE_NAME);
132      // Set system property value for MR job
133      System.setProperty("mapreduce.job.queuename", queueName);
134    }
135
136    // parse main restore command options
137    String[] remainArgs = cmd.getArgs();
138    if (remainArgs.length != 2) {
139      printToolUsage();
140      return -1;
141    }
142
143    String backupRootDir = remainArgs[0];
144    String backupId = remainArgs[1];
145    String tables;
146    String tableMapping =
147        cmd.hasOption(OPTION_TABLE_MAPPING) ? cmd.getOptionValue(OPTION_TABLE_MAPPING) : null;
148    try (final Connection conn = ConnectionFactory.createConnection(conf);
149        BackupAdmin client = new BackupAdminImpl(conn)) {
150      // Check backup set
151      if (cmd.hasOption(OPTION_SET)) {
152        String setName = cmd.getOptionValue(OPTION_SET);
153        try {
154          tables = getTablesForSet(conn, setName, conf);
155        } catch (IOException e) {
156          System.out.println("ERROR: " + e.getMessage() + " for setName=" + setName);
157          printToolUsage();
158          return -2;
159        }
160        if (tables == null) {
161          System.out.println("ERROR: Backup set '" + setName
162              + "' is either empty or does not exist");
163          printToolUsage();
164          return -3;
165        }
166      } else {
167        tables = cmd.getOptionValue(OPTION_TABLE);
168      }
169
170      TableName[] sTableArray = BackupUtils.parseTableNames(tables);
171      TableName[] tTableArray = BackupUtils.parseTableNames(tableMapping);
172
173      if (sTableArray != null && tTableArray != null &&
174          (sTableArray.length != tTableArray.length)) {
175        System.out.println("ERROR: table mapping mismatch: " + tables + " : " + tableMapping);
176        printToolUsage();
177        return -4;
178      }
179
180      client.restore(BackupUtils.createRestoreRequest(backupRootDir, backupId, check,
181        sTableArray, tTableArray, overwrite));
182    } catch (Exception e) {
183      LOG.error("Error while running restore backup", e);
184      return -5;
185    }
186    return 0;
187  }
188
189  private String getTablesForSet(Connection conn, String name, Configuration conf)
190      throws IOException {
191    try (final BackupSystemTable table = new BackupSystemTable(conn)) {
192      List<TableName> tables = table.describeBackupSet(name);
193
194      if (tables == null) {
195        return null;
196      }
197
198      return StringUtils.join(tables, BackupRestoreConstants.TABLENAME_DELIMITER_IN_COMMAND);
199    }
200  }
201
202  @Override
203  protected void addOptions() {
204    // define supported options
205    addOptNoArg(OPTION_OVERWRITE, OPTION_OVERWRITE_DESC);
206    addOptNoArg(OPTION_CHECK, OPTION_CHECK_DESC);
207    addOptNoArg(OPTION_DEBUG, OPTION_DEBUG_DESC);
208    addOptWithArg(OPTION_SET, OPTION_SET_RESTORE_DESC);
209    addOptWithArg(OPTION_TABLE, OPTION_TABLE_LIST_DESC);
210    addOptWithArg(OPTION_TABLE_MAPPING, OPTION_TABLE_MAPPING_DESC);
211    addOptWithArg(OPTION_YARN_QUEUE_NAME, OPTION_YARN_QUEUE_NAME_RESTORE_DESC);
212  }
213
214  @Override
215  protected void processOptions(CommandLine cmd) {
216    this.cmd = cmd;
217  }
218
219  @Override
220  protected int doWork() throws Exception {
221    return parseAndRun(cmd.getArgs());
222  }
223
224  public static void main(String[] args) throws Exception {
225    Configuration conf = HBaseConfiguration.create();
226    Path hbasedir = FSUtils.getRootDir(conf);
227    URI defaultFs = hbasedir.getFileSystem(conf).getUri();
228    FSUtils.setFsDefault(conf, new Path(defaultFs));
229    int ret = ToolRunner.run(conf, new RestoreDriver(), args);
230    System.exit(ret);
231  }
232
233  @Override
234  public int run(String[] args) {
235    if (conf == null) {
236      LOG.error("Tool configuration is not initialized");
237      throw new NullPointerException("conf");
238    }
239
240    CommandLine cmd;
241    try {
242      // parse the command line arguments
243      cmd = parseArgs(args);
244      cmdLineArgs = args;
245    } catch (Exception e) {
246      System.out.println("Error when parsing command-line arguments: " + e.getMessage());
247      printToolUsage();
248      return EXIT_FAILURE;
249    }
250
251    if (cmd.hasOption(SHORT_HELP_OPTION) || cmd.hasOption(LONG_HELP_OPTION)) {
252      printToolUsage();
253      return EXIT_FAILURE;
254    }
255
256    processOptions(cmd);
257
258    int ret = EXIT_FAILURE;
259    try {
260      ret = doWork();
261    } catch (Exception e) {
262      LOG.error("Error running command-line tool", e);
263      return EXIT_FAILURE;
264    }
265    return ret;
266  }
267
268  protected void printToolUsage() {
269    System.out.println(USAGE_STRING);
270    HelpFormatter helpFormatter = new HelpFormatter();
271    helpFormatter.setLeftPadding(2);
272    helpFormatter.setDescPadding(8);
273    helpFormatter.setWidth(100);
274    helpFormatter.setSyntaxPrefix("Options:");
275    helpFormatter.printHelp(" ", null, options, USAGE_FOOTER);
276  }
