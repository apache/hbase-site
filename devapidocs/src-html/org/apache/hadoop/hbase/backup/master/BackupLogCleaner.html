049/**
050 * Implementation of a log cleaner that checks if a log is still scheduled for incremental backup
051 * before deleting it when its TTL is over.
052 */
053@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
054public class BackupLogCleaner extends BaseLogCleanerDelegate {
055  private static final Logger LOG = LoggerFactory.getLogger(BackupLogCleaner.class);
056
057  private boolean stopped = false;
058  private Connection conn;
059
060  public BackupLogCleaner() {
061  }
062
063  @Override
064  public void init(Map<String, Object> params) {
065    MasterServices master = (MasterServices) MapUtils.getObject(params,
066      HMaster.MASTER);
067    if (master != null) {
068      conn = master.getConnection();
069      if (getConf() == null) {
070        super.setConf(conn.getConfiguration());
071      }
072    }
073    if (conn == null) {
074      try {
075        conn = ConnectionFactory.createConnection(getConf());
076      } catch (IOException ioe) {
077        throw new RuntimeException("Failed to create connection", ioe);
078      }
079    }
080  }
081
082
083  private Map<Address, Long> getServersToOldestBackupMapping(List<BackupInfo> backups)
084    throws IOException {
085    Map<Address, Long> serverAddressToLastBackupMap = new HashMap<>();
086
087    Map<TableName, Long> tableNameBackupInfoMap = new HashMap<>();
088    for (BackupInfo backupInfo : backups) {
089      for (TableName table : backupInfo.getTables()) {
090        tableNameBackupInfoMap.putIfAbsent(table, backupInfo.getStartTs());
091        if (tableNameBackupInfoMap.get(table) <= backupInfo.getStartTs()) {
092          tableNameBackupInfoMap.put(table, backupInfo.getStartTs());
093          for (Map.Entry<String, Long> entry : backupInfo.getTableSetTimestampMap().get(table)
094            .entrySet()) {
095            serverAddressToLastBackupMap.put(Address.fromString(entry.getKey()), entry.getValue());
096          }
097        }
098      }
099    }
100
101    return serverAddressToLastBackupMap;
102  }
103
104  @Override
105  public Iterable<FileStatus> getDeletableFiles(Iterable<FileStatus> files) {
106    List<FileStatus> filteredFiles = new ArrayList<>();
107
108    // all members of this class are null if backup is disabled,
109    // so we cannot filter the files
110    if (this.getConf() == null || !BackupManager.isBackupEnabled(getConf())) {
111      LOG.debug("Backup is not enabled. Check your {} setting",
112        BackupRestoreConstants.BACKUP_ENABLE_KEY);
113      return files;
114    }
115
116    Map<Address, Long> addressToLastBackupMap;
117    try {
118      try (BackupManager backupManager = new BackupManager(conn, getConf())) {
119        addressToLastBackupMap =
120          getServersToOldestBackupMapping(backupManager.getBackupHistory(true));
121      }
122    } catch (IOException ex) {
123      LOG.error("Failed to analyse backup history with exception: {}. Retaining all logs",
124        ex.getMessage(), ex);
125      return Collections.emptyList();
126    }
127    for (FileStatus file : files) {
128      String fn = file.getPath().getName();
129      if (fn.startsWith(WALProcedureStore.LOG_PREFIX)) {
130        filteredFiles.add(file);
131        continue;
132      }
133
134      try {
135        Address walServerAddress =
136          Address.fromString(BackupUtils.parseHostNameFromLogFile(file.getPath()));
137        long walTimestamp = AbstractFSWALProvider.getTimestamp(file.getPath().getName());
138
139        if (!addressToLastBackupMap.containsKey(walServerAddress)
140          || addressToLastBackupMap.get(walServerAddress) >= walTimestamp) {
141          filteredFiles.add(file);
142        }
143      } catch (Exception ex) {
144        LOG.warn(
145          "Error occurred while filtering file: {} with error: {}. Ignoring cleanup of this log",
146          file.getPath(), ex.getMessage());
147      }
148    }
149
150    LOG
151      .info("Total files: {}, Filtered Files: {}", IterableUtils.size(files), filteredFiles.size());
152    return filteredFiles;
153  }
154
155  @Override
156  public void setConf(Configuration config) {
157    // If backup is disabled, keep all members null
158    super.setConf(config);
159    if (!config.getBoolean(BackupRestoreConstants.BACKUP_ENABLE_KEY,
160      BackupRestoreConstants.BACKUP_ENABLE_DEFAULT)) {
161      LOG.warn("Backup is disabled - allowing all wals to be deleted");
162    }
163  }
164
165  @Override
166  public void stop(String why) {
167    if (!this.stopped) {
168      this.stopped = true;
169      LOG.info("Stopping BackupLogCleaner");
170    }
171  }
172
173  @Override
174  public boolean isStopped() {
175    return this.stopped;
176  }
177}
