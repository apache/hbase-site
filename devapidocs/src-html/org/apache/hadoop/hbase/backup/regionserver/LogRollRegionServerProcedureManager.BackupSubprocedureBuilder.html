019package org.apache.hadoop.hbase.backup.regionserver;
020
021import java.io.IOException;
022import java.util.concurrent.ThreadPoolExecutor;
023
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.backup.BackupRestoreConstants;
026import org.apache.hadoop.hbase.backup.impl.BackupManager;
027import org.apache.hadoop.hbase.backup.master.LogRollMasterProcedureManager;
028import org.apache.hadoop.hbase.errorhandling.ForeignExceptionDispatcher;
029import org.apache.hadoop.hbase.procedure.ProcedureCoordinationManager;
030import org.apache.hadoop.hbase.procedure.ProcedureMember;
031import org.apache.hadoop.hbase.procedure.ProcedureMemberRpcs;
032import org.apache.hadoop.hbase.procedure.RegionServerProcedureManager;
033import org.apache.hadoop.hbase.procedure.Subprocedure;
034import org.apache.hadoop.hbase.procedure.SubprocedureFactory;
035import org.apache.hadoop.hbase.procedure.ZKProcedureCoordinationManager;
036import org.apache.hadoop.hbase.regionserver.RegionServerServices;
037import org.apache.yetus.audience.InterfaceAudience;
038import org.apache.zookeeper.KeeperException;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041
042/**
043 * This manager class handles the work dealing with distributed WAL roll request.
044 * <p>
045 * This provides the mechanism necessary to kick off a backup specific {@link Subprocedure} that is
046 * responsible by this region server. If any failures occur with the sub-procedure, the manager's
047 * procedure member notifies the procedure coordinator to abort all others.
048 * <p>
049 * On startup, requires {@link #start()} to be called.
050 * <p>
051 * On shutdown, requires org.apache.hadoop.hbase.procedure.ProcedureMember.close() to be called
052 */
053@InterfaceAudience.Private
054public class LogRollRegionServerProcedureManager extends RegionServerProcedureManager {
055  private static final Logger LOG =
056      LoggerFactory.getLogger(LogRollRegionServerProcedureManager.class);
057
058  /** Conf key for number of request threads to start backup on region servers */
059  public static final String BACKUP_REQUEST_THREADS_KEY = "hbase.backup.region.pool.threads";
060  /** # of threads for backup work on the rs. */
061  public static final int BACKUP_REQUEST_THREADS_DEFAULT = 10;
062
063  public static final String BACKUP_TIMEOUT_MILLIS_KEY = "hbase.backup.timeout";
064  public static final long BACKUP_TIMEOUT_MILLIS_DEFAULT = 60000;
065
066  /** Conf key for millis between checks to see if backup work completed or if there are errors */
067  public static final String BACKUP_REQUEST_WAKE_MILLIS_KEY = "hbase.backup.region.wakefrequency";
068  /** Default amount of time to check for errors while regions finish backup work */
069  private static final long BACKUP_REQUEST_WAKE_MILLIS_DEFAULT = 500;
070
071  private RegionServerServices rss;
072  private ProcedureMemberRpcs memberRpcs;
073  private ProcedureMember member;
074  private boolean started = false;
075
076  /**
077   * Create a default backup procedure manager
078   */
079  public LogRollRegionServerProcedureManager() {
080  }
081
082  /**
083   * Start accepting backup procedure requests.
084   */
085  @Override
086  public void start() {
087    if (!BackupManager.isBackupEnabled(rss.getConfiguration())) {
088      LOG.warn("Backup is not enabled. Check your " + BackupRestoreConstants.BACKUP_ENABLE_KEY
089          + " setting");
090      return;
091    }
092    this.memberRpcs.start(rss.getServerName().toString(), member);
093    started = true;
094    LOG.info("Started region server backup manager.");
095  }
096
097  /**
098   * Close <tt>this</tt> and all running backup procedure tasks
099   * @param force forcefully stop all running tasks
100   * @throws IOException exception
101   */
102  @Override
103  public void stop(boolean force) throws IOException {
104    if (!started) {
105      return;
106    }
107    String mode = force ? "abruptly" : "gracefully";
108    LOG.info("Stopping RegionServerBackupManager " + mode + ".");
109
110    try {
111      this.member.close();
112    } finally {
113      this.memberRpcs.close();
114    }
115  }
116
117  /**
118   * If in a running state, creates the specified subprocedure for handling a backup procedure.
119   * @return Subprocedure to submit to the ProcedureMemeber.
120   */
121  public Subprocedure buildSubprocedure(byte[] data) {
122    // don't run a backup if the parent is stop(ping)
123    if (rss.isStopping() || rss.isStopped()) {
124      throw new IllegalStateException("Can't start backup procedure on RS: " + rss.getServerName()
125          + ", because stopping/stopped!");
126    }
127
128    LOG.info("Attempting to run a roll log procedure for backup.");
129    ForeignExceptionDispatcher errorDispatcher = new ForeignExceptionDispatcher();
130    Configuration conf = rss.getConfiguration();
131    long timeoutMillis = conf.getLong(BACKUP_TIMEOUT_MILLIS_KEY, BACKUP_TIMEOUT_MILLIS_DEFAULT);
132    long wakeMillis =
133        conf.getLong(BACKUP_REQUEST_WAKE_MILLIS_KEY, BACKUP_REQUEST_WAKE_MILLIS_DEFAULT);
134
135    LogRollBackupSubprocedurePool taskManager =
136        new LogRollBackupSubprocedurePool(rss.getServerName().toString(), conf);
137    return new LogRollBackupSubprocedure(rss, member, errorDispatcher, wakeMillis, timeoutMillis,
138        taskManager, data);
139  }
140
141  /**
142   * Build the actual backup procedure runner that will do all the 'hard' work
143   */
144  public class BackupSubprocedureBuilder implements SubprocedureFactory {
145    @Override
146    public Subprocedure buildSubprocedure(String name, byte[] data) {
147      return LogRollRegionServerProcedureManager.this.buildSubprocedure(data);
148    }
149  }
150
151  @Override
152  public void initialize(RegionServerServices rss) throws KeeperException {
153    this.rss = rss;
154    if (!BackupManager.isBackupEnabled(rss.getConfiguration())) {
155      LOG.warn("Backup is not enabled. Check your " + BackupRestoreConstants.BACKUP_ENABLE_KEY
156          + " setting");
157      return;
158    }
159    ProcedureCoordinationManager coordManager = new ZKProcedureCoordinationManager(rss);
160    this.memberRpcs = coordManager
161            .getProcedureMemberRpcs(LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_SIGNATURE);
162
163    // read in the backup handler configuration properties
164    Configuration conf = rss.getConfiguration();
165    long keepAlive = conf.getLong(BACKUP_TIMEOUT_MILLIS_KEY, BACKUP_TIMEOUT_MILLIS_DEFAULT);
166    int opThreads = conf.getInt(BACKUP_REQUEST_THREADS_KEY, BACKUP_REQUEST_THREADS_DEFAULT);
167    // create the actual cohort member
168    ThreadPoolExecutor pool =
169        ProcedureMember.defaultPool(rss.getServerName().toString(), opThreads, keepAlive);
170    this.member = new ProcedureMember(memberRpcs, pool, new BackupSubprocedureBuilder());
171  }
172
173  @Override
174  public String getProcedureSignature() {
175    return "backup-proc";
176  }
177}
