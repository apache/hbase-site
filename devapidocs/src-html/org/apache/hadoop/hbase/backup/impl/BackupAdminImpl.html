092  @Override
093  public int deleteBackups(String[] backupIds) throws IOException {
094
095    int totalDeleted = 0;
096    Map<String, HashSet<TableName>> allTablesMap = new HashMap<String, HashSet<TableName>>();
097
098    boolean deleteSessionStarted = false;
099    boolean snapshotDone = false;
100    try (final BackupSystemTable sysTable = new BackupSystemTable(conn)) {
101
102      // Step 1: Make sure there is no active session
103      // is running by using startBackupSession API
104      // If there is an active session in progress, exception will be thrown
105      try {
106        sysTable.startBackupExclusiveOperation();
107        deleteSessionStarted = true;
108      } catch (IOException e) {
109        LOG.warn("You can not run delete command while active backup session is in progress. \n"
110            + "If there is no active backup session running, run backup repair utility to restore \n"
111            + "backup system integrity.");
112        return -1;
113      }
114
115      // Step 2: Make sure there is no failed session
116      List<BackupInfo> list = sysTable.getBackupInfos(BackupState.RUNNING);
117      if (list.size() != 0) {
118        // ailed sessions found
119        LOG.warn("Failed backup session found. Run backup repair tool first.");
120        return -1;
121      }
122
123      // Step 3: Record delete session
124      sysTable.startDeleteOperation(backupIds);
125      // Step 4: Snapshot backup system table
126      if (!BackupSystemTable.snapshotExists(conn)) {
127        BackupSystemTable.snapshot(conn);
128      } else {
129        LOG.warn("Backup system table snapshot exists");
130      }
131      snapshotDone = true;
132      try {
133        for (int i = 0; i < backupIds.length; i++) {
134          BackupInfo info = sysTable.readBackupInfo(backupIds[i]);
135          if (info != null) {
136            String rootDir = info.getBackupRootDir();
137            HashSet<TableName> allTables = allTablesMap.get(rootDir);
138            if (allTables == null) {
139              allTables = new HashSet<TableName>();
140              allTablesMap.put(rootDir, allTables);
141            }
142            allTables.addAll(info.getTableNames());
143            totalDeleted += deleteBackup(backupIds[i], sysTable);
144          }
145        }
146        finalizeDelete(allTablesMap, sysTable);
147        // Finish
148        sysTable.finishDeleteOperation();
149        // delete snapshot
150        BackupSystemTable.deleteSnapshot(conn);
151      } catch (IOException e) {
152        // Fail delete operation
153        // Step 1
154        if (snapshotDone) {
155          if (BackupSystemTable.snapshotExists(conn)) {
156            BackupSystemTable.restoreFromSnapshot(conn);
157            // delete snapshot
158            BackupSystemTable.deleteSnapshot(conn);
159            // We still have record with unfinished delete operation
160            LOG.error("Delete operation failed, please run backup repair utility to restore "
161                + "backup system integrity", e);
162            throw e;
163          } else {
164            LOG.warn("Delete operation succeeded, there were some errors: ", e);
165          }
166        }
167
168      } finally {
169        if (deleteSessionStarted) {
170          sysTable.finishBackupExclusiveOperation();
171        }
172      }
173    }
174    return totalDeleted;
175  }

178   * Updates incremental backup set for every backupRoot
179   * @param tablesMap map [backupRoot: Set<TableName>]
180   * @param table backup system table
181   * @throws IOException
182   */
183
184  private void finalizeDelete(Map<String, HashSet<TableName>> tablesMap, BackupSystemTable table)
185      throws IOException {
186    for (String backupRoot : tablesMap.keySet()) {
187      Set<TableName> incrTableSet = table.getIncrementalBackupTableSet(backupRoot);
188      Map<TableName, ArrayList<BackupInfo>> tableMap =
189          table.getBackupHistoryForTableSet(incrTableSet, backupRoot);
190      for (Map.Entry<TableName, ArrayList<BackupInfo>> entry : tableMap.entrySet()) {
191        if (entry.getValue() == null) {
192          // No more backups for a table
193          incrTableSet.remove(entry.getKey());
194        }
195      }
196      if (!incrTableSet.isEmpty()) {
197        table.addIncrementalBackupTableSet(incrTableSet, backupRoot);
198      } else { // empty
199        table.deleteIncrementalBackupTableSet(backupRoot);
200      }
201    }
202  }

204  /**
205   * Delete single backup and all related backups <br>
206   * Algorithm:<br>
207   * Backup type: FULL or INCREMENTAL <br>
208   * Is this last backup session for table T: YES or NO <br>
209   * For every table T from table list 'tables':<br>
210   * if(FULL, YES) deletes only physical data (PD) <br>
211   * if(FULL, NO), deletes PD, scans all newer backups and removes T from backupInfo,<br>
212   * until we either reach the most recent backup for T in the system or FULL backup<br>
213   * which includes T<br>
214   * if(INCREMENTAL, YES) deletes only physical data (PD) if(INCREMENTAL, NO) deletes physical data
215   * and for table T scans all backup images between last<br>
216   * FULL backup, which is older than the backup being deleted and the next FULL backup (if exists) <br>
217   * or last one for a particular table T and removes T from list of backup tables.
218   * @param backupId backup id
219   * @param sysTable backup system table
220   * @return total number of deleted backup images
221   * @throws IOException
222   */
223  private int deleteBackup(String backupId, BackupSystemTable sysTable) throws IOException {
224
225    BackupInfo backupInfo = sysTable.readBackupInfo(backupId);
226
227    int totalDeleted = 0;
228    if (backupInfo != null) {
229      LOG.info("Deleting backup " + backupInfo.getBackupId() + " ...");
230      // Step 1: clean up data for backup session (idempotent)
231      BackupUtils.cleanupBackupData(backupInfo, conn.getConfiguration());
232      // List of tables in this backup;
233      List<TableName> tables = backupInfo.getTableNames();
234      long startTime = backupInfo.getStartTs();
235      for (TableName tn : tables) {
236        boolean isLastBackupSession = isLastBackupSession(sysTable, tn, startTime);
237        if (isLastBackupSession) {
238          continue;
239        }
240        // else
241        List<BackupInfo> affectedBackups = getAffectedBackupSessions(backupInfo, tn, sysTable);
242        for (BackupInfo info : affectedBackups) {
243          if (info.equals(backupInfo)) {
244            continue;
245          }
246          removeTableFromBackupImage(info, tn, sysTable);
247        }
248      }
249      Map<byte[], String> map = sysTable.readBulkLoadedFiles(backupId);
250      FileSystem fs = FileSystem.get(conn.getConfiguration());
251      boolean success = true;
252      int numDeleted = 0;
253      for (String f : map.values()) {
254        Path p = new Path(f);
255        try {
256          LOG.debug("Delete backup info " + p + " for " + backupInfo.getBackupId());
257          if (!fs.delete(p)) {
258            if (fs.exists(p)) {
259              LOG.warn(f + " was not deleted");
260              success = false;
261            }
262          } else {
263            numDeleted++;
264          }
265        } catch (IOException ioe) {
266          LOG.warn(f + " was not deleted", ioe);
267          success = false;
268        }
269      }
270      if (LOG.isDebugEnabled()) {
271        LOG.debug(numDeleted + " bulk loaded files out of " + map.size() + " were deleted");
272      }
273      if (success) {
274        sysTable.deleteBulkLoadedFiles(map);
275      }
276
277      sysTable.deleteBackupInfo(backupInfo.getBackupId());
278      LOG.info("Delete backup " + backupInfo.getBackupId() + " completed.");
279      totalDeleted++;
280    } else {
281      LOG.warn("Delete backup failed: no information found for backupID=" + backupId);
282    }
283    return totalDeleted;
284  }

286  private void
287      removeTableFromBackupImage(BackupInfo info, TableName tn, BackupSystemTable sysTable)
288          throws IOException {
289    List<TableName> tables = info.getTableNames();
290    LOG.debug("Remove " + tn + " from " + info.getBackupId() + " tables="
291        + info.getTableListAsString());
292    if (tables.contains(tn)) {
293      tables.remove(tn);
294
295      if (tables.isEmpty()) {
296        LOG.debug("Delete backup info " + info.getBackupId());
297
298        sysTable.deleteBackupInfo(info.getBackupId());
299        // Idempotent operation
300        BackupUtils.cleanupBackupData(info, conn.getConfiguration());
301      } else {
302        info.setTables(tables);
303        sysTable.updateBackupInfo(info);
304        // Now, clean up directory for table (idempotent)
305        cleanupBackupDir(info, tn, conn.getConfiguration());
306      }
307    }
308  }

310  private List<BackupInfo> getAffectedBackupSessions(BackupInfo backupInfo, TableName tn,
311      BackupSystemTable table) throws IOException {
312    LOG.debug("GetAffectedBackupInfos for: " + backupInfo.getBackupId() + " table=" + tn);
313    long ts = backupInfo.getStartTs();
314    List<BackupInfo> list = new ArrayList<BackupInfo>();
315    List<BackupInfo> history = table.getBackupHistory(backupInfo.getBackupRootDir());
316    // Scan from most recent to backupInfo
317    // break when backupInfo reached
318    for (BackupInfo info : history) {
319      if (info.getStartTs() == ts) {
320        break;
321      }
322      List<TableName> tables = info.getTableNames();
323      if (tables.contains(tn)) {
324        BackupType bt = info.getType();
325        if (bt == BackupType.FULL) {
326          // Clear list if we encounter FULL backup
327          list.clear();
328        } else {
329          LOG.debug("GetAffectedBackupInfos for: " + backupInfo.getBackupId() + " table=" + tn
330              + " added " + info.getBackupId() + " tables=" + info.getTableListAsString());
331          list.add(info);
332        }
333      }
334    }
335    return list;
336  }

338  /**
339   * Clean up the data at target directory
340   * @throws IOException
341   */
342  private void cleanupBackupDir(BackupInfo backupInfo, TableName table, Configuration conf)
343      throws IOException {
344    try {
345      // clean up the data at target directory
346      String targetDir = backupInfo.getBackupRootDir();
347      if (targetDir == null) {
348        LOG.warn("No target directory specified for " + backupInfo.getBackupId());
349        return;
350      }
351
352      FileSystem outputFs = FileSystem.get(new Path(backupInfo.getBackupRootDir()).toUri(), conf);
353
354      Path targetDirPath =
355          new Path(BackupUtils.getTableBackupDir(backupInfo.getBackupRootDir(),
356            backupInfo.getBackupId(), table));
357      if (outputFs.delete(targetDirPath, true)) {
358        LOG.info("Cleaning up backup data at " + targetDirPath.toString() + " done.");
359      } else {
360        LOG.info("No data has been found in " + targetDirPath.toString() + ".");
361      }
362
363    } catch (IOException e1) {
364      LOG.error("Cleaning up backup data of " + backupInfo.getBackupId() + " for table " + table
365          + "at " + backupInfo.getBackupRootDir() + " failed due to " + e1.getMessage() + ".");
366      throw e1;
367    }
368  }

370  private boolean isLastBackupSession(BackupSystemTable table, TableName tn, long startTime)
371      throws IOException {
372    List<BackupInfo> history = table.getBackupHistory();
373    for (BackupInfo info : history) {
374      List<TableName> tables = info.getTableNames();
375      if (!tables.contains(tn)) {
376        continue;
377      }
378      if (info.getStartTs() <= startTime) {
379        return true;
380      } else {
381        return false;
382      }
383    }
384    return false;
385  }

400  @Override
401  public List<BackupInfo> getHistory(int n, BackupInfo.Filter... filters) throws IOException {
402    if (filters.length == 0) return getHistory(n);
403    try (final BackupSystemTable table = new BackupSystemTable(conn)) {
404      List<BackupInfo> history = table.getBackupHistory();
405      List<BackupInfo> result = new ArrayList<BackupInfo>();
406      for (BackupInfo bi : history) {
407        if (result.size() == n) break;
408        boolean passed = true;
409        for (int i = 0; i < filters.length; i++) {
410          if (!filters[i].apply(bi)) {
411            passed = false;
412            break;
413          }
414        }
415        if (passed) {
416          result.add(bi);
417        }
418      }
419      return result;
420    }
421  }

493  @Override
494  public void restore(RestoreRequest request) throws IOException {
495    if (request.isCheck()) {
496      HashMap<TableName, BackupManifest> backupManifestMap = new HashMap<>();
497      // check and load backup image manifest for the tables
498      Path rootPath = new Path(request.getBackupRootDir());
499      String backupId = request.getBackupId();
500      TableName[] sTableArray = request.getFromTables();
501      HBackupFileSystem.checkImageManifestExist(backupManifestMap, sTableArray,
502        conn.getConfiguration(), rootPath, backupId);
503
504      // Check and validate the backup image and its dependencies
505
506      if (BackupUtils.validate(backupManifestMap, conn.getConfiguration())) {
507        LOG.info(CHECK_OK);
508      } else {
509        LOG.error(CHECK_FAILED);
510      }
511      return;
512    }
513    // Execute restore request
514    new RestoreTablesClient(conn, request).execute();
515  }

517  @Override
518  public String backupTables(BackupRequest request) throws IOException {
519    BackupType type = request.getBackupType();
520    String targetRootDir = request.getTargetRootDir();
521    List<TableName> tableList = request.getTableList();
522
523    String backupId = BackupRestoreConstants.BACKUPID_PREFIX + EnvironmentEdgeManager.currentTime();
524    if (type == BackupType.INCREMENTAL) {
525      Set<TableName> incrTableSet = null;
526      try (BackupSystemTable table = new BackupSystemTable(conn)) {
527        incrTableSet = table.getIncrementalBackupTableSet(targetRootDir);
528      }
529
530      if (incrTableSet.isEmpty()) {
531        String msg =
532            "Incremental backup table set contains no tables. "
533                + "You need to run full backup first "
534                + (tableList != null ? "on " + StringUtils.join(tableList, ",") : "");
535
536        throw new IOException(msg);
537      }
538      if (tableList != null) {
539        tableList.removeAll(incrTableSet);
540        if (!tableList.isEmpty()) {
541          String extraTables = StringUtils.join(tableList, ",");
542          String msg =
543              "Some tables (" + extraTables + ") haven't gone through full backup. "
544                  + "Perform full backup on " + extraTables + " first, " + "then retry the command";
545          throw new IOException(msg);
546        }
547      }
548      tableList = Lists.newArrayList(incrTableSet);
549    }
550    if (tableList != null && !tableList.isEmpty()) {
551      for (TableName table : tableList) {
552        String targetTableBackupDir =
553            HBackupFileSystem.getTableBackupDir(targetRootDir, backupId, table);
554        Path targetTableBackupDirPath = new Path(targetTableBackupDir);
555        FileSystem outputFs =
556            FileSystem.get(targetTableBackupDirPath.toUri(), conn.getConfiguration());
557        if (outputFs.exists(targetTableBackupDirPath)) {
558          throw new IOException("Target backup directory " + targetTableBackupDir
559              + " exists already.");
560        }
561        outputFs.mkdirs(targetTableBackupDirPath);
562      }
563      ArrayList<TableName> nonExistingTableList = null;
564      try (Admin admin = conn.getAdmin();) {
565        for (TableName tableName : tableList) {
566          if (!admin.tableExists(tableName)) {
567            if (nonExistingTableList == null) {
568              nonExistingTableList = new ArrayList<>();
569            }
570            nonExistingTableList.add(tableName);
571          }
572        }
573      }
574      if (nonExistingTableList != null) {
575        if (type == BackupType.INCREMENTAL) {
576          // Update incremental backup set
577          tableList = excludeNonExistingTables(tableList, nonExistingTableList);
578        } else {
579          // Throw exception only in full mode - we try to backup non-existing table
580          throw new IOException("Non-existing tables found in the table list: "
581              + nonExistingTableList);
582        }
583      }
584    }
585
586    // update table list
587    BackupRequest.Builder builder = new BackupRequest.Builder();
588    request =
589        builder.withBackupType(request.getBackupType()).withTableList(tableList)
590            .withTargetRootDir(request.getTargetRootDir())
591            .withBackupSetName(request.getBackupSetName()).withTotalTasks(request.getTotalTasks())
592            .withBandwidthPerTasks((int) request.getBandwidth()).build();
593
594    TableBackupClient client = null;
595    try {
596      client = BackupClientFactory.create(conn, backupId, request);
597    } catch (IOException e) {
598      LOG.error("There is an active session already running");
599      throw e;
600    }
601
602    client.execute();
603
604    return backupId;
605  }

607  private List<TableName> excludeNonExistingTables(List<TableName> tableList,
608      List<TableName> nonExistingTableList) {
609
610    for (TableName table : nonExistingTableList) {
611      tableList.remove(table);
612    }
613    return tableList;

625  /**
626   * Verifies that backup images are valid for merge.
627   *
628   * <ul>
629   * <li>All backups MUST be in the same destination
630   * <li>No FULL backups are allowed - only INCREMENTAL
631   * <li>All backups must be in COMPLETE state
632   * <li>No holes in backup list are allowed
633   * </ul>
634   * <p>
635   * @param backupIds list of backup ids
636   * @param table backup system table
637   * @throws IOException
638   */
639  private void checkIfValidForMerge(String[] backupIds, BackupSystemTable table) throws IOException {
640    String backupRoot = null;
641
642    final Set<TableName> allTables = new HashSet<TableName>();
643    final Set<String> allBackups = new HashSet<String>();
644    long minTime = Long.MAX_VALUE, maxTime = Long.MIN_VALUE;
645    for (String backupId : backupIds) {
646      BackupInfo bInfo = table.readBackupInfo(backupId);
647      if (bInfo == null) {
648        String msg = "Backup session " + backupId + " not found";
649        throw new IOException(msg);
650      }
651      if (backupRoot == null) {
652        backupRoot = bInfo.getBackupRootDir();
653      } else if (!bInfo.getBackupRootDir().equals(backupRoot)) {
654        throw new IOException("Found different backup destinations in a list of a backup sessions \n"
655            + "1. " + backupRoot + "\n" + "2. " + bInfo.getBackupRootDir());
656      }
657      if (bInfo.getType() == BackupType.FULL) {
658        throw new IOException("FULL backup image can not be merged for: \n" + bInfo);
659      }
660
661      if (bInfo.getState() != BackupState.COMPLETE) {
662        throw new IOException("Backup image " + backupId
663            + " can not be merged becuase of its state: " + bInfo.getState());
664      }
665      allBackups.add(backupId);
666      allTables.addAll(bInfo.getTableNames());
667      long time = bInfo.getStartTs();
668      if (time < minTime) {
669        minTime = time;
670      }
671      if (time > maxTime) {
672        maxTime = time;
673      }
674    }
675
676
677    final long startRangeTime  = minTime;
678    final long endRangeTime = maxTime;
679    final String backupDest = backupRoot;
680    // Check we have no 'holes' in backup id list
681    // Filter 1 : backupRoot
682    // Filter 2 : time range filter
683    // Filter 3 : table filter
684
685    BackupInfo.Filter destinationFilter = new  BackupInfo.Filter() {
686
687      @Override
688      public boolean apply(BackupInfo info) {
689        return info.getBackupRootDir().equals(backupDest);
690      }
691    };
692
693    BackupInfo.Filter timeRangeFilter = new  BackupInfo.Filter() {
694
695      @Override
696      public boolean apply(BackupInfo info) {
697        long time = info.getStartTs();
698        return time >= startRangeTime && time <= endRangeTime ;
699      }
700    };
701
702    BackupInfo.Filter tableFilter = new  BackupInfo.Filter() {
703
704      @Override
705      public boolean apply(BackupInfo info) {
706        List<TableName> tables = info.getTableNames();
707        return !Collections.disjoint(allTables, tables);
708      }
709    };
710
711    BackupInfo.Filter typeFilter = new  BackupInfo.Filter() {
712
713      @Override
714      public boolean apply(BackupInfo info) {
715        return info.getType() == BackupType.INCREMENTAL;
716      }
717    };
718
719    BackupInfo.Filter stateFilter = new  BackupInfo.Filter() {
720      @Override
721      public boolean apply(BackupInfo info) {
722        return info.getState() == BackupState.COMPLETE;
723      }
724    };
725
726    List<BackupInfo> allInfos =
727        table.getBackupHistory( -1, destinationFilter,
728          timeRangeFilter, tableFilter, typeFilter, stateFilter);
729    if (allInfos.size() != allBackups.size()) {
730      // Yes we have at least one  hole in backup image sequence
731      List<String> missingIds = new ArrayList<String>();
732      for(BackupInfo info: allInfos) {
733        if(allBackups.contains(info.getBackupId())) {
734          continue;
735        }
736        missingIds.add(info.getBackupId());
737      }
738      String errMsg =
739          "Sequence of backup ids has 'holes'. The following backup images must be added:" +
740           org.apache.hadoop.util.StringUtils.join(",", missingIds);
741      throw new IOException(errMsg);
742    }
743  }
744}
