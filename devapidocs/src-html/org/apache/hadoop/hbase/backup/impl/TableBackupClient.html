048/**
049 * Base class for backup operation. Concrete implementation for
050 * full and incremental backup are delegated to corresponding sub-classes:
051 * {@link FullTableBackupClient} and {@link IncrementalTableBackupClient}
052 *
053 */
054@InterfaceAudience.Private
055public abstract class TableBackupClient {

082  public void init(final Connection conn, final String backupId, BackupRequest request)
083      throws IOException
084  {
085    if (request.getBackupType() == BackupType.FULL) {
086      backupManager = new BackupManager(conn, conn.getConfiguration());
087    } else {
088      backupManager = new IncrementalBackupManager(conn, conn.getConfiguration());
089    }
090    this.backupId = backupId;
091    this.tableList = request.getTableList();
092    this.conn = conn;
093    this.conf = conn.getConfiguration();
094    this.fs = FSUtils.getCurrentFileSystem(conf);
095    backupInfo =
096        backupManager.createBackupInfo(backupId, request.getBackupType(), tableList,
097          request.getTargetRootDir(), request.getTotalTasks(), request.getBandwidth());
098    if (tableList == null || tableList.isEmpty()) {
099      this.tableList = new ArrayList<>(backupInfo.getTables());
100    }
101    // Start new session
102    backupManager.startBackupSession();
103  }
104
105  /**
106   * Begin the overall backup.
107   * @param backupInfo backup info
108   * @throws IOException exception
109   */
110  protected void beginBackup(BackupManager backupManager, BackupInfo backupInfo)
111      throws IOException {
112
113    BackupSystemTable.snapshot(conn);
114    backupManager.setBackupInfo(backupInfo);
115    // set the start timestamp of the overall backup
116    long startTs = EnvironmentEdgeManager.currentTime();
117    backupInfo.setStartTs(startTs);
118    // set overall backup status: ongoing
119    backupInfo.setState(BackupState.RUNNING);
120    backupInfo.setPhase(BackupPhase.REQUEST);
121    LOG.info("Backup " + backupInfo.getBackupId() + " started at " + startTs + ".");
122
123    backupManager.updateBackupInfo(backupInfo);
124    if (LOG.isDebugEnabled()) {
125      LOG.debug("Backup session " + backupInfo.getBackupId() + " has been started.");
126    }
127  }
128
129  protected String getMessage(Exception e) {
130    String msg = e.getMessage();
131    if (msg == null || msg.equals("")) {
132      msg = e.getClass().getName();
133    }
134    return msg;
135  }

137  /**
138   * Delete HBase snapshot for backup.
139   * @param backupInfo backup info
140   * @throws Exception exception
141   */
142  protected static void deleteSnapshots(final Connection conn, BackupInfo backupInfo, Configuration conf)
143      throws IOException {
144    LOG.debug("Trying to delete snapshot for full backup.");
145    for (String snapshotName : backupInfo.getSnapshotNames()) {
146      if (snapshotName == null) {
147        continue;
148      }
149      LOG.debug("Trying to delete snapshot: " + snapshotName);
150
151      try (Admin admin = conn.getAdmin()) {
152        admin.deleteSnapshot(snapshotName);
153      }
154      LOG.debug("Deleting the snapshot " + snapshotName + " for backup " + backupInfo.getBackupId()
155          + " succeeded.");
156    }
157  }
158
159  /**
160   * Clean up directories with prefix "exportSnapshot-", which are generated when exporting
161   * snapshots.
162   * @throws IOException exception
163   */
164  protected static void cleanupExportSnapshotLog(Configuration conf) throws IOException {
165    FileSystem fs = FSUtils.getCurrentFileSystem(conf);
166    Path stagingDir =
167        new Path(conf.get(BackupRestoreConstants.CONF_STAGING_ROOT, fs.getWorkingDirectory()
168            .toString()));
169    FileStatus[] files = FSUtils.listStatus(fs, stagingDir);
170    if (files == null) {
171      return;
172    }
173    for (FileStatus file : files) {
174      if (file.getPath().getName().startsWith("exportSnapshot-")) {
175        LOG.debug("Delete log files of exporting snapshot: " + file.getPath().getName());
176        if (FSUtils.delete(fs, file.getPath(), true) == false) {
177          LOG.warn("Can not delete " + file.getPath());
178        }
179      }
180    }
181  }
182
183  /**
184   * Clean up the uncompleted data at target directory if the ongoing backup has already entered
185   * the copy phase.
186   */
187  protected static void cleanupTargetDir(BackupInfo backupInfo, Configuration conf) {
188    try {
189      // clean up the uncompleted data at target directory if the ongoing backup has already entered
190      // the copy phase
191      LOG.debug("Trying to cleanup up target dir. Current backup phase: "
192          + backupInfo.getPhase());
193      if (backupInfo.getPhase().equals(BackupPhase.SNAPSHOTCOPY)
194          || backupInfo.getPhase().equals(BackupPhase.INCREMENTAL_COPY)
195          || backupInfo.getPhase().equals(BackupPhase.STORE_MANIFEST)) {
196        FileSystem outputFs =
197            FileSystem.get(new Path(backupInfo.getBackupRootDir()).toUri(), conf);
198
199        // now treat one backup as a transaction, clean up data that has been partially copied at
200        // table level
201        for (TableName table : backupInfo.getTables()) {
202          Path targetDirPath =
203              new Path(HBackupFileSystem.getTableBackupDir(backupInfo.getBackupRootDir(),
204                backupInfo.getBackupId(), table));
205          if (outputFs.delete(targetDirPath, true)) {
206            LOG.debug("Cleaning up uncompleted backup data at " + targetDirPath.toString()
207                + " done.");
208          } else {
209            LOG.debug("No data has been copied to " + targetDirPath.toString() + ".");
210          }
211
212          Path tableDir = targetDirPath.getParent();
213          FileStatus[] backups = FSUtils.listStatus(outputFs, tableDir);
214          if (backups == null || backups.length == 0) {
215            outputFs.delete(tableDir, true);
216            LOG.debug(tableDir.toString() + " is empty, remove it.");
217          }
218        }
219      }
220
221    } catch (IOException e1) {
222      LOG.error("Cleaning up uncompleted backup data of " + backupInfo.getBackupId() + " at "
223          + backupInfo.getBackupRootDir() + " failed due to " + e1.getMessage() + ".");
224    }
225  }

227  /**
228   * Fail the overall backup.
229   * @param backupInfo backup info
230   * @param e exception
231   * @throws Exception exception
232   */
233  protected void failBackup(Connection conn, BackupInfo backupInfo, BackupManager backupManager,
234      Exception e, String msg, BackupType type, Configuration conf) throws IOException {
235
236    try {
237      LOG.error(msg + getMessage(e), e);
238      // If this is a cancel exception, then we've already cleaned.
239      // set the failure timestamp of the overall backup
240      backupInfo.setCompleteTs(EnvironmentEdgeManager.currentTime());
241      // set failure message
242      backupInfo.setFailedMsg(e.getMessage());
243      // set overall backup status: failed
244      backupInfo.setState(BackupState.FAILED);
245      // compose the backup failed data
246      String backupFailedData =
247          "BackupId=" + backupInfo.getBackupId() + ",startts=" + backupInfo.getStartTs()
248              + ",failedts=" + backupInfo.getCompleteTs() + ",failedphase=" + backupInfo.getPhase()
249              + ",failedmessage=" + backupInfo.getFailedMsg();
250      LOG.error(backupFailedData);
251      cleanupAndRestoreBackupSystem(conn, backupInfo, conf);
252      // If backup session is updated to FAILED state - means we
253      // processed recovery already.
254      backupManager.updateBackupInfo(backupInfo);
255      backupManager.finishBackupSession();
256      LOG.error("Backup " + backupInfo.getBackupId() + " failed.");
257    } catch (IOException ee) {
258      LOG.error("Please run backup repair tool manually to restore backup system integrity");
259      throw ee;
260    }
261  }
262
263  public static void cleanupAndRestoreBackupSystem(Connection conn, BackupInfo backupInfo,
264      Configuration conf) throws IOException {
265    BackupType type = backupInfo.getType();
266    // if full backup, then delete HBase snapshots if there already are snapshots taken
267    // and also clean up export snapshot log files if exist
268    if (type == BackupType.FULL) {
269      deleteSnapshots(conn, backupInfo, conf);
270      cleanupExportSnapshotLog(conf);
271    }
272    BackupSystemTable.restoreFromSnapshot(conn);
273    BackupSystemTable.deleteSnapshot(conn);
274    // clean up the uncompleted data at target directory if the ongoing backup has already entered
275    // the copy phase
276    // For incremental backup, DistCp logs will be cleaned with the targetDir.
277    cleanupTargetDir(backupInfo, conf);
278  }

282  /**
283   * Add manifest for the current backup. The manifest is stored within the table backup directory.
284   * @param backupInfo The current backup info
285   * @throws IOException exception
286   * @throws BackupException exception
287   */
288  protected void addManifest(BackupInfo backupInfo, BackupManager backupManager, BackupType type,
289      Configuration conf) throws IOException, BackupException {
290    // set the overall backup phase : store manifest
291    backupInfo.setPhase(BackupPhase.STORE_MANIFEST);
292
293    BackupManifest manifest;
294
295    // Since we have each table's backup in its own directory structure,
296    // we'll store its manifest with the table directory.
297    for (TableName table : backupInfo.getTables()) {
298      manifest = new BackupManifest(backupInfo, table);
299      ArrayList<BackupImage> ancestors = backupManager.getAncestors(backupInfo, table);
300      for (BackupImage image : ancestors) {
301        manifest.addDependentImage(image);
302      }
303
304      if (type == BackupType.INCREMENTAL) {
305        // We'll store the log timestamps for this table only in its manifest.
306        HashMap<TableName, HashMap<String, Long>> tableTimestampMap =
307            new HashMap<TableName, HashMap<String, Long>>();
308        tableTimestampMap.put(table, backupInfo.getIncrTimestampMap().get(table));
309        manifest.setIncrTimestampMap(tableTimestampMap);
310        ArrayList<BackupImage> ancestorss = backupManager.getAncestors(backupInfo);
311        for (BackupImage image : ancestorss) {
312          manifest.addDependentImage(image);
313        }
314      }
315      manifest.store(conf);
316    }
317
318    // For incremental backup, we store a overall manifest in
319    // <backup-root-dir>/WALs/<backup-id>
320    // This is used when created the next incremental backup
321    if (type == BackupType.INCREMENTAL) {
322      manifest = new BackupManifest(backupInfo);
323      // set the table region server start and end timestamps for incremental backup
324      manifest.setIncrTimestampMap(backupInfo.getIncrTimestampMap());
325      ArrayList<BackupImage> ancestors = backupManager.getAncestors(backupInfo);
326      for (BackupImage image : ancestors) {
327        manifest.addDependentImage(image);
328      }
329      manifest.store(conf);
330    }
331  }

333  /**
334   * Get backup request meta data dir as string.
335   * @param backupInfo backup info
336   * @return meta data dir
337   */
338  protected String obtainBackupMetaDataStr(BackupInfo backupInfo) {
339    StringBuffer sb = new StringBuffer();
340    sb.append("type=" + backupInfo.getType() + ",tablelist=");
341    for (TableName table : backupInfo.getTables()) {
342      sb.append(table + ";");
343    }
344    if (sb.lastIndexOf(";") > 0) {
345      sb.delete(sb.lastIndexOf(";"), sb.lastIndexOf(";") + 1);
346    }
347    sb.append(",targetRootDir=" + backupInfo.getBackupRootDir());
348
349    return sb.toString();
350  }

352  /**
353   * Clean up directories with prefix "_distcp_logs-", which are generated when DistCp copying
354   * hlogs.
355   * @throws IOException exception
356   */
357  protected void cleanupDistCpLog(BackupInfo backupInfo, Configuration conf) throws IOException {
358    Path rootPath = new Path(backupInfo.getHLogTargetDir()).getParent();
359    FileStatus[] files = FSUtils.listStatus(fs, rootPath);
360    if (files == null) {
361      return;
362    }
363    for (FileStatus file : files) {
364      if (file.getPath().getName().startsWith("_distcp_logs")) {
365        LOG.debug("Delete log files of DistCp: " + file.getPath().getName());
366        FSUtils.delete(fs, file.getPath(), true);
367      }
368    }
369  }

371  /**
372   * Complete the overall backup.
373   * @param backupInfo backup info
374   * @throws Exception exception
375   */
376  protected void completeBackup(final Connection conn, BackupInfo backupInfo,
377      BackupManager backupManager, BackupType type, Configuration conf) throws IOException {
378    // set the complete timestamp of the overall backup
379    backupInfo.setCompleteTs(EnvironmentEdgeManager.currentTime());
380    // set overall backup status: complete
381    backupInfo.setState(BackupState.COMPLETE);
382    backupInfo.setProgress(100);
383    // add and store the manifest for the backup
384    addManifest(backupInfo, backupManager, type, conf);
385
386    // compose the backup complete data
387    String backupCompleteData =
388        obtainBackupMetaDataStr(backupInfo) + ",startts=" + backupInfo.getStartTs()
389            + ",completets=" + backupInfo.getCompleteTs() + ",bytescopied="
390            + backupInfo.getTotalBytesCopied();
391    if (LOG.isDebugEnabled()) {
392      LOG.debug("Backup " + backupInfo.getBackupId() + " finished: " + backupCompleteData);
393    }
394
395    // when full backup is done:
396    // - delete HBase snapshot
397    // - clean up directories with prefix "exportSnapshot-", which are generated when exporting
398    // snapshots
399    if (type == BackupType.FULL) {
400      deleteSnapshots(conn, backupInfo, conf);
401      cleanupExportSnapshotLog(conf);
402    } else if (type == BackupType.INCREMENTAL) {
403      cleanupDistCpLog(backupInfo, conf);
404    }
405    BackupSystemTable.deleteSnapshot(conn);
406    backupManager.updateBackupInfo(backupInfo);
407
408    // Finish active session
409    backupManager.finishBackupSession();
410
411    LOG.info("Backup " + backupInfo.getBackupId() + " completed.");
412  }
