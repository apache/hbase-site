105  /**
106   * Remove an entire region from the table directory via archiving the region's hfiles.
107   * @param fs {@link FileSystem} from which to remove the region
108   * @param rootdir {@link Path} to the root directory where hbase files are stored (for building
109   *          the archive path)
110   * @param tableDir {@link Path} to where the table is being stored (for building the archive path)
111   * @param regionDir {@link Path} to where a region is being stored (for building the archive path)
112   * @return <tt>true</tt> if the region was successfully deleted. <tt>false</tt> if the filesystem
113   *         operations could not complete.
114   * @throws IOException if the request cannot be completed
115   */
116  public static boolean archiveRegion(FileSystem fs, Path rootdir, Path tableDir, Path regionDir)
117      throws IOException {
118    // otherwise, we archive the files
119    // make sure we can archive
120    if (tableDir == null || regionDir == null) {
121      LOG.error("No archive directory could be found because tabledir (" + tableDir
122          + ") or regiondir (" + regionDir + "was null. Deleting files instead.");
123      if (regionDir != null) {
124        deleteRegionWithoutArchiving(fs, regionDir);
125      }
126      // we should have archived, but failed to. Doesn't matter if we deleted
127      // the archived files correctly or not.
128      return false;
129    }
130
131    LOG.debug("ARCHIVING {}", regionDir);
132
133    // make sure the regiondir lives under the tabledir
134    Preconditions.checkArgument(regionDir.toString().startsWith(tableDir.toString()));
135    Path regionArchiveDir = HFileArchiveUtil.getRegionArchiveDir(rootdir,
136        FSUtils.getTableName(tableDir),
137        regionDir.getName());
138
139    FileStatusConverter getAsFile = new FileStatusConverter(fs);
140    // otherwise, we attempt to archive the store files
141
142    // build collection of just the store directories to archive
143    Collection<File> toArchive = new ArrayList<>();
144    final PathFilter dirFilter = new FSUtils.DirFilter(fs);
145    PathFilter nonHidden = new PathFilter() {
146      @Override
147      public boolean accept(Path file) {
148        return dirFilter.accept(file) && !file.getName().startsWith(".");
149      }
150    };
151    FileStatus[] storeDirs = FSUtils.listStatus(fs, regionDir, nonHidden);
152    // if there no files, we can just delete the directory and return;
153    if (storeDirs == null) {
154      LOG.debug("Directory {} empty.", regionDir);
155      return deleteRegionWithoutArchiving(fs, regionDir);
156    }
157
158    // convert the files in the region to a File
159    Stream.of(storeDirs).map(getAsFile).forEachOrdered(toArchive::add);
160    LOG.debug("Archiving " + toArchive);
161    List<File> failedArchive = resolveAndArchive(fs, regionArchiveDir, toArchive,
162        EnvironmentEdgeManager.currentTime());
163    if (!failedArchive.isEmpty()) {
164      throw new FailedArchiveException(
165        "Failed to archive/delete all the files for region:" + regionDir.getName() + " into " +
166          regionArchiveDir + ". Something is probably awry on the filesystem.",
167        failedArchive.stream().map(FUNC_FILE_TO_PATH).collect(Collectors.toList()));
168    }
169    // if that was successful, then we delete the region
170    return deleteRegionWithoutArchiving(fs, regionDir);
171  }

173  /**
174   * Archive the specified regions in parallel.
175   * @param conf the configuration to use
176   * @param fs {@link FileSystem} from which to remove the region
177   * @param rootDir {@link Path} to the root directory where hbase files are stored (for building
178   *                            the archive path)
179   * @param tableDir {@link Path} to where the table is being stored (for building the archive
180   *                             path)
181   * @param regionDirList {@link Path} to where regions are being stored (for building the archive
182   *                                  path)
183   * @throws IOException if the request cannot be completed
184   */
185  public static void archiveRegions(Configuration conf, FileSystem fs, Path rootDir, Path tableDir,
186    List<Path> regionDirList) throws IOException {
187    List<Future<Void>> futures = new ArrayList<>(regionDirList.size());
188    for (Path regionDir: regionDirList) {
189      Future<Void> future = getArchiveExecutor(conf).submit(() -> {
190        archiveRegion(fs, rootDir, tableDir, regionDir);
191        return null;
192      });
193      futures.add(future);
194    }
195    try {
196      for (Future<Void> future: futures) {
197        future.get();
198      }
199    } catch (InterruptedException e) {
200      throw new InterruptedIOException(e.getMessage());
201    } catch (ExecutionException e) {
202      throw new IOException(e.getCause());
203    }
204  }
205
206  private static synchronized ThreadPoolExecutor getArchiveExecutor(final Configuration conf) {
207    if (archiveExecutor == null) {
208      int maxThreads = conf.getInt("hbase.hfilearchiver.thread.pool.max", 8);
209      archiveExecutor = Threads.getBoundedCachedThreadPool(maxThreads, 30L, TimeUnit.SECONDS,
210        getThreadFactory());
211
212      // Shutdown this ThreadPool in a shutdown hook
213      Runtime.getRuntime().addShutdownHook(new Thread(() -> archiveExecutor.shutdown()));
214    }
215    return archiveExecutor;
216  }

218  // We need this method instead of Threads.getNamedThreadFactory() to pass some tests.
219  // The difference from Threads.getNamedThreadFactory() is that it doesn't fix ThreadGroup for
220  // new threads. If we use Threads.getNamedThreadFactory(), we will face ThreadGroup related
221  // issues in some tests.
222  private static ThreadFactory getThreadFactory() {
223    return new ThreadFactory() {
224      final AtomicInteger threadNumber = new AtomicInteger(1);
225
226      @Override
227      public Thread newThread(Runnable r) {
228        final String name = "HFileArchiver-" + threadNumber.getAndIncrement();
229        Thread t = new Thread(r, name);
230        t.setDaemon(true);
231        return t;
232      }
233    };
234  }

252  /**
253   * Removes from the specified region the store files of the specified column family,
254   * either by archiving them or outright deletion
255   * @param fs the filesystem where the store files live
256   * @param conf {@link Configuration} to examine to determine the archive directory
257   * @param parent Parent region hosting the store files
258   * @param familyDir {@link Path} to where the family is being stored
259   * @param family the family hosting the store files
260   * @throws IOException if the files could not be correctly disposed.
261   */
262  public static void archiveFamilyByFamilyDir(FileSystem fs, Configuration conf,
263      RegionInfo parent, Path familyDir, byte[] family) throws IOException {
264    FileStatus[] storeFiles = FSUtils.listStatus(fs, familyDir);
265    if (storeFiles == null) {
266      LOG.debug("No files to dispose of in {}, family={}", parent.getRegionNameAsString(),
267          Bytes.toString(family));
268      return;
269    }
270
271    FileStatusConverter getAsFile = new FileStatusConverter(fs);
272    Collection<File> toArchive = Stream.of(storeFiles).map(getAsFile).collect(Collectors.toList());
273    Path storeArchiveDir = HFileArchiveUtil.getStoreArchivePath(conf, parent, family);
274
275    // do the actual archive
276    List<File> failedArchive = resolveAndArchive(fs, storeArchiveDir, toArchive,
277        EnvironmentEdgeManager.currentTime());
278    if (!failedArchive.isEmpty()){
279      throw new FailedArchiveException("Failed to archive/delete all the files for region:"
280          + Bytes.toString(parent.getRegionName()) + ", family:" + Bytes.toString(family)
281          + " into " + storeArchiveDir + ". Something is probably awry on the filesystem.",
282          failedArchive.stream().map(FUNC_FILE_TO_PATH).collect(Collectors.toList()));
283    }
284  }

286  /**
287   * Remove the store files, either by archiving them or outright deletion
288   * @param conf {@link Configuration} to examine to determine the archive directory
289   * @param fs the filesystem where the store files live
290   * @param regionInfo {@link RegionInfo} of the region hosting the store files
291   * @param family the family hosting the store files
292   * @param compactedFiles files to be disposed of. No further reading of these files should be
293   *          attempted; otherwise likely to cause an {@link IOException}
294   * @throws IOException if the files could not be correctly disposed.
295   */
296  public static void archiveStoreFiles(Configuration conf, FileSystem fs, RegionInfo regionInfo,
297      Path tableDir, byte[] family, Collection<HStoreFile> compactedFiles)
298      throws IOException, FailedArchiveException {
299
300    // sometimes in testing, we don't have rss, so we need to check for that
301    if (fs == null) {
302      LOG.warn("Passed filesystem is null, so just deleting files without archiving for {}," +
303              "family={}", Bytes.toString(regionInfo.getRegionName()), Bytes.toString(family));
304      deleteStoreFilesWithoutArchiving(compactedFiles);
305      return;
306    }
307
308    // short circuit if we don't have any files to delete
309    if (compactedFiles.isEmpty()) {
310      LOG.debug("No files to dispose of, done!");
311      return;
312    }
313
314    // build the archive path
315    if (regionInfo == null || family == null) throw new IOException(
316        "Need to have a region and a family to archive from.");
317
318    Path storeArchiveDir = HFileArchiveUtil.getStoreArchivePath(conf, regionInfo, tableDir, family);
319
320    // make sure we don't archive if we can't and that the archive dir exists
321    if (!fs.mkdirs(storeArchiveDir)) {
322      throw new IOException("Could not make archive directory (" + storeArchiveDir + ") for store:"
323          + Bytes.toString(family) + ", deleting compacted files instead.");
324    }
325
326    // otherwise we attempt to archive the store files
327    LOG.debug("Archiving compacted files.");
328
329    // Wrap the storefile into a File
330    StoreToFile getStorePath = new StoreToFile(fs);
331    Collection<File> storeFiles =
332      compactedFiles.stream().map(getStorePath).collect(Collectors.toList());
333
334    // do the actual archive
335    List<File> failedArchive =
336      resolveAndArchive(fs, storeArchiveDir, storeFiles, EnvironmentEdgeManager.currentTime());
337
338    if (!failedArchive.isEmpty()){
339      throw new FailedArchiveException("Failed to archive/delete all the files for region:"
340          + Bytes.toString(regionInfo.getRegionName()) + ", family:" + Bytes.toString(family)
341          + " into " + storeArchiveDir + ". Something is probably awry on the filesystem.",
342          failedArchive.stream().map(FUNC_FILE_TO_PATH).collect(Collectors.toList()));
343    }
344  }

346  /**
347   * Archive the store file
348   * @param fs the filesystem where the store files live
349   * @param regionInfo region hosting the store files
350   * @param conf {@link Configuration} to examine to determine the archive directory
351   * @param tableDir {@link Path} to where the table is being stored (for building the archive path)
352   * @param family the family hosting the store files
353   * @param storeFile file to be archived
354   * @throws IOException if the files could not be correctly disposed.
355   */
356  public static void archiveStoreFile(Configuration conf, FileSystem fs, RegionInfo regionInfo,
357      Path tableDir, byte[] family, Path storeFile) throws IOException {
358    Path storeArchiveDir = HFileArchiveUtil.getStoreArchivePath(conf, regionInfo, tableDir, family);
359    // make sure we don't archive if we can't and that the archive dir exists
360    if (!fs.mkdirs(storeArchiveDir)) {
361      throw new IOException("Could not make archive directory (" + storeArchiveDir + ") for store:"
362          + Bytes.toString(family) + ", deleting compacted files instead.");
363    }
364
365    // do the actual archive
366    long start = EnvironmentEdgeManager.currentTime();
367    File file = new FileablePath(fs, storeFile);
368    if (!resolveAndArchiveFile(storeArchiveDir, file, Long.toString(start))) {
369      throw new IOException("Failed to archive/delete the file for region:"
370          + regionInfo.getRegionNameAsString() + ", family:" + Bytes.toString(family)
371          + " into " + storeArchiveDir + ". Something is probably awry on the filesystem.");
372    }
373  }

375  /**
376   * Resolve any conflict with an existing archive file via timestamp-append
377   * renaming of the existing file and then archive the passed in files.
378   * @param fs {@link FileSystem} on which to archive the files
379   * @param baseArchiveDir base archive directory to store the files. If any of
380   *          the files to archive are directories, will append the name of the
381   *          directory to the base archive directory name, creating a parallel
382   *          structure.
383   * @param toArchive files/directories that need to be archvied
384   * @param start time the archiving started - used for resolving archive
385   *          conflicts.
386   * @return the list of failed to archive files.
387   * @throws IOException if an unexpected file operation exception occurred
388   */
389  private static List<File> resolveAndArchive(FileSystem fs, Path baseArchiveDir,
390      Collection<File> toArchive, long start) throws IOException {
391    // short circuit if no files to move
392    if (toArchive.isEmpty()) {
393      return Collections.emptyList();
394    }
395
396    LOG.trace("Moving files to the archive directory {}", baseArchiveDir);
397
398    // make sure the archive directory exists
399    if (!fs.exists(baseArchiveDir)) {
400      if (!fs.mkdirs(baseArchiveDir)) {
401        throw new IOException("Failed to create the archive directory:" + baseArchiveDir
402            + ", quitting archive attempt.");
403      }
404      LOG.trace("Created archive directory {}", baseArchiveDir);
405    }
406
407    List<File> failures = new ArrayList<>();
408    String startTime = Long.toString(start);
409    for (File file : toArchive) {
410      // if its a file archive it
411      try {
412        LOG.trace("Archiving {}", file);
413        if (file.isFile()) {
414          // attempt to archive the file
415          if (!resolveAndArchiveFile(baseArchiveDir, file, startTime)) {
416            LOG.warn("Couldn't archive " + file + " into backup directory: " + baseArchiveDir);
417            failures.add(file);
418          }
419        } else {
420          // otherwise its a directory and we need to archive all files
421          LOG.trace("{} is a directory, archiving children files", file);
422          // so we add the directory name to the one base archive
423          Path parentArchiveDir = new Path(baseArchiveDir, file.getName());
424          // and then get all the files from that directory and attempt to
425          // archive those too
426          Collection<File> children = file.getChildren();
427          failures.addAll(resolveAndArchive(fs, parentArchiveDir, children, start));
428        }
429      } catch (IOException e) {
430        LOG.warn("Failed to archive {}", file, e);
431        failures.add(file);
432      }
433    }
434    return failures;
435  }

437  /**
438   * Attempt to archive the passed in file to the archive directory.
439   * <p>
440   * If the same file already exists in the archive, it is moved to a timestamped directory under
441   * the archive directory and the new file is put in its place.
442   * @param archiveDir {@link Path} to the directory that stores the archives of the hfiles
443   * @param currentFile {@link Path} to the original HFile that will be archived
444   * @param archiveStartTime time the archiving started, to resolve naming conflicts
445   * @return <tt>true</tt> if the file is successfully archived. <tt>false</tt> if there was a
446   *         problem, but the operation still completed.
447   * @throws IOException on failure to complete {@link FileSystem} operations.
448   */
449  private static boolean resolveAndArchiveFile(Path archiveDir, File currentFile,
450      String archiveStartTime) throws IOException {
451    // build path as it should be in the archive
452    String filename = currentFile.getName();
453    Path archiveFile = new Path(archiveDir, filename);
454    FileSystem fs = currentFile.getFileSystem();
455
456    // if the file already exists in the archive, move that one to a timestamped backup. This is a
457    // really, really unlikely situtation, where we get the same name for the existing file, but
458    // is included just for that 1 in trillion chance.
459    if (fs.exists(archiveFile)) {
460      LOG.debug("{} already exists in archive, moving to timestamped backup and " +
461          "overwriting current.", archiveFile);
462
463      // move the archive file to the stamped backup
464      Path backedupArchiveFile = new Path(archiveDir, filename + SEPARATOR + archiveStartTime);
465      if (!fs.rename(archiveFile, backedupArchiveFile)) {
466        LOG.error("Could not rename archive file to backup: " + backedupArchiveFile
467            + ", deleting existing file in favor of newer.");
468        // try to delete the exisiting file, if we can't rename it
469        if (!fs.delete(archiveFile, false)) {
470          throw new IOException("Couldn't delete existing archive file (" + archiveFile
471              + ") or rename it to the backup file (" + backedupArchiveFile
472              + ") to make room for similarly named file.");
473        }
474      }
475      LOG.debug("Backed up archive file from " + archiveFile);
476    }
477
478    LOG.trace("No existing file in archive for {}, free to archive original file.", archiveFile);
479
480    // at this point, we should have a free spot for the archive file
481    boolean success = false;
482    for (int i = 0; !success && i < DEFAULT_RETRIES_NUMBER; ++i) {
483      if (i > 0) {
484        // Ensure that the archive directory exists.
485        // The previous "move to archive" operation has failed probably because
486        // the cleaner has removed our archive directory (HBASE-7643).
487        // (we're in a retry loop, so don't worry too much about the exception)
488        try {
489          if (!fs.exists(archiveDir)) {
490            if (fs.mkdirs(archiveDir)) {
491              LOG.debug("Created archive directory {}", archiveDir);
492            }
493          }
494        } catch (IOException e) {
495          LOG.warn("Failed to create directory {}", archiveDir, e);
496        }
497      }
498
499      try {
500        success = currentFile.moveAndClose(archiveFile);
501      } catch (FileNotFoundException fnfe) {
502        LOG.warn("Failed to archive " + currentFile +
503            " because it does not exist! Skipping and continuing on.", fnfe);
504        success = true;
505      } catch (IOException e) {
506        LOG.warn("Failed to archive " + currentFile + " on try #" + i, e);
507        success = false;
508      }
509    }
510
511    if (!success) {
512      LOG.error("Failed to archive " + currentFile);
513      return false;
514    }
515
516    LOG.debug("Archived from {} to {}", currentFile, archiveFile);
517    return true;
518  }

520  /**
521   * Without regard for backup, delete a region. Should be used with caution.
522   * @param regionDir {@link Path} to the region to be deleted.
523   * @param fs FileSystem from which to delete the region
524   * @return <tt>true</tt> on successful deletion, <tt>false</tt> otherwise
525   * @throws IOException on filesystem operation failure
526   */
527  private static boolean deleteRegionWithoutArchiving(FileSystem fs, Path regionDir)
528      throws IOException {
529    if (fs.delete(regionDir, true)) {
530      LOG.debug("Deleted {}", regionDir);
531      return true;
532    }
533    LOG.debug("Failed to delete directory {}", regionDir);
534    return false;
535  }

537  /**
538   * Just do a simple delete of the given store files
539   * <p>
540   * A best effort is made to delete each of the files, rather than bailing on the first failure.
541   * <p>
542   * @param compactedFiles store files to delete from the file system.
543   * @throws IOException if a file cannot be deleted. All files will be attempted to deleted before
544   *           throwing the exception, rather than failing at the first file.
545   */
546  private static void deleteStoreFilesWithoutArchiving(Collection<HStoreFile> compactedFiles)
547      throws IOException {
548    LOG.debug("Deleting files without archiving.");
549    List<IOException> errors = new ArrayList<>(0);
550    for (HStoreFile hsf : compactedFiles) {
551      try {
552        hsf.deleteStoreFile();
553      } catch (IOException e) {
554        LOG.error("Failed to delete {}", hsf.getPath());
555        errors.add(e);
556      }
557    }
558    if (errors.size() > 0) {
559      throw MultipleIOException.createIOException(errors);
560    }
561  }

611    public File(FileSystem fs) {

615    /**

617     * @throws IOException on failure

621    /**
622     * Check to see if this is a file or a directory
623     * @return <tt>true</tt> if it is a file, <tt>false</tt> otherwise

628    /**
629     * @return if this is a directory, returns all the children in the
630     *         directory, otherwise returns an empty list
631     * @throws IOException
632     */
633    abstract Collection<File> getChildren() throws IOException;
634
635    /**
636     * close any outside readers of the file
637     * @throws IOException
638     */
639    abstract void close() throws IOException;
640
641    /**
642     * @return the name of the file (not the full fs path, just the individual
643     *         file name)
644     */
645    abstract String getName();
646
647    /**
648     * @return the path to this file
649     */
650    abstract Path getPath();
651
652    /**
653     * Move the file to the given destination
654     * @param dest
655     * @return <tt>true</tt> on success
656     * @throws IOException
657     */
658    public boolean moveAndClose(Path dest) throws IOException {
659      this.close();
660      Path p = this.getPath();
661      return FSUtils.renameAndSetModifyTime(fs, p, dest);
662    }
663
664    /**
665     * @return the {@link FileSystem} on which this file resides
666     */
667    public FileSystem getFileSystem() {
668      return this.fs;
669    }
670
671    @Override
672    public String toString() {
673      return this.getClass().getSimpleName() + ", " + getPath().toString();
674    }
675  }

677  /**
678   * A {@link File} that wraps a simple {@link Path} on a {@link FileSystem}.
679   */
680  private static class FileablePath extends File {
681    private final Path file;
682    private final FileStatusConverter getAsFile;
683
684    public FileablePath(FileSystem fs, Path file) {
685      super(fs);
686      this.file = file;
687      this.getAsFile = new FileStatusConverter(fs);
688    }
689
690    @Override
691    public void delete() throws IOException {
692      if (!fs.delete(file, true)) throw new IOException("Failed to delete:" + this.file);
693    }
694
695    @Override
696    public String getName() {
697      return file.getName();
698    }
699
700    @Override
701    public Collection<File> getChildren() throws IOException {
702      if (fs.isFile(file)) {
703        return Collections.emptyList();
704      }
705      return Stream.of(fs.listStatus(file)).map(getAsFile).collect(Collectors.toList());
706    }
707
708    @Override
709    public boolean isFile() throws IOException {
710      return fs.isFile(file);
711    }
712
713    @Override
714    public void close() throws IOException {
715      // NOOP - files are implicitly closed on removal
716    }
717
718    @Override
719    Path getPath() {
720      return file;
721    }
722  }
