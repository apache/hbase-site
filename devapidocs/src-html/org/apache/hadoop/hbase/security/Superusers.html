020package org.apache.hadoop.hbase.security;
021
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.List;
025
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.AuthUtil;
028import org.apache.yetus.audience.InterfaceAudience;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032/**
033 * Keeps lists of superusers and super groups loaded from HBase configuration,
034 * checks if certain user is regarded as superuser.
035 */
036@InterfaceAudience.Private
037public final class Superusers {
038  private static final Logger LOG = LoggerFactory.getLogger(Superusers.class);
039
040  /** Configuration key for superusers */
041  public static final String SUPERUSER_CONF_KEY = "hbase.superuser"; // Not getting a name
042
043  private static List<String> superUsers;
044  private static List<String> superGroups;
045  private static User systemUser;
046
047  private Superusers(){}
048
049  /**
050   * Should be called only once to pre-load list of super users and super
051   * groups from Configuration. This operation is idempotent.
052   * @param conf configuration to load users from
053   * @throws IOException if unable to initialize lists of superusers or super groups
054   * @throws IllegalStateException if current user is null
055   */
056  public static void initialize(Configuration conf) throws IOException {
057    superUsers = new ArrayList<>();
058    superGroups = new ArrayList<>();
059    systemUser = User.getCurrent();
060
061    if (systemUser == null) {
062      throw new IllegalStateException("Unable to obtain the current user, "
063        + "authorization checks for internal operations will not work correctly!");
064    }
065
066    if (LOG.isTraceEnabled()) {
067      LOG.trace("Current user name is " + systemUser.getShortName());
068    }
069    String currentUser = systemUser.getShortName();
070    String[] superUserList = conf.getStrings(SUPERUSER_CONF_KEY, new String[0]);
071    for (String name : superUserList) {
072      if (AuthUtil.isGroupPrincipal(name)) {
073        superGroups.add(AuthUtil.getGroupName(name));
074      } else {
075        superUsers.add(name);
076      }
077    }
078    superUsers.add(currentUser);
079  }
080
081  /**
082   * @return true if current user is a super user (whether as user running process,
083   * declared as individual superuser or member of supergroup), false otherwise.
084   * @param user to check
085   * @throws IllegalStateException if lists of superusers/super groups
086   *   haven't been initialized properly
087   */
088  public static boolean isSuperUser(User user) {
089    if (superUsers == null) {
090      throw new IllegalStateException("Super users/super groups lists"
091        + " haven't been initialized properly.");
092    }
093    if (superUsers.contains(user.getShortName())) {
094      return true;
095    }
096
097    for (String group : user.getGroupNames()) {
098      if (superGroups.contains(group)) {
099        return true;
100      }
101    }
102    return false;
103  }
104
105  public static List<String> getSuperUsers() {
106    return superUsers;
107  }
108
109  public static User getSystemUser() {
110    return systemUser;
111  }
112}
