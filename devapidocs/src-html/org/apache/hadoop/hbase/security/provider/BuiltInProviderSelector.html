018package org.apache.hadoop.hbase.security.provider;
019
020import static java.util.Objects.requireNonNull;
021
022import java.util.Collection;
023import java.util.Objects;
024
025import net.jcip.annotations.NotThreadSafe;
026
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.hbase.HBaseInterfaceAudience;
029import org.apache.hadoop.hbase.security.User;
030import org.apache.hadoop.hbase.util.Pair;
031import org.apache.hadoop.io.Text;
032import org.apache.hadoop.security.token.Token;
033import org.apache.hadoop.security.token.TokenIdentifier;
034import org.apache.yetus.audience.InterfaceAudience;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037
038/**
039 * Default implementation of {@link AuthenticationProviderSelector} which can choose from the
040 * authentication implementations which HBase provides out of the box: Simple, Kerberos, and
041 * Delegation Token authentication.
042 *
043 * This implementation will ignore any {@link SaslAuthenticationProvider}'s which are available
044 * on the classpath or specified in the configuration because HBase cannot correctly choose which
045 * token should be returned to a client when multiple are present. It is expected that users
046 * implement their own {@link AuthenticationProviderSelector} when writing a custom provider.
047 *
048 * This implementation is not thread-safe. {@link #configure(Configuration, Collection)} and
049 * {@link #selectProvider(String, User)} is not safe if they are called concurrently.
050 */
051@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.AUTHENTICATION)
052@NotThreadSafe
053public class BuiltInProviderSelector implements AuthenticationProviderSelector {
054  private static final Logger LOG = LoggerFactory.getLogger(BuiltInProviderSelector.class);
055
056  Configuration conf;
057  SimpleSaslClientAuthenticationProvider simpleAuth = null;
058  GssSaslClientAuthenticationProvider krbAuth = null;
059  DigestSaslClientAuthenticationProvider digestAuth = null;
060  Text digestAuthTokenKind = null;
061
062  @Override
063  public void configure(
064      Configuration conf, Collection<SaslClientAuthenticationProvider> providers) {
065    if (this.conf != null) {
066      throw new IllegalStateException("configure() should only be called once");
067    }
068    this.conf = Objects.requireNonNull(conf);
069
070    for (SaslClientAuthenticationProvider provider : Objects.requireNonNull(providers)) {
071      final String name = provider.getSaslAuthMethod().getName();
072      if (SimpleSaslAuthenticationProvider.SASL_AUTH_METHOD.getName().contentEquals(name)) {
073        if (simpleAuth != null) {
074          throw new IllegalStateException(
075              "Encountered multiple SimpleSaslClientAuthenticationProvider instances");
076        }
077        simpleAuth = (SimpleSaslClientAuthenticationProvider) provider;
078      } else if (GssSaslAuthenticationProvider.SASL_AUTH_METHOD.getName().equals(name)) {
079        if (krbAuth != null) {
080          throw new IllegalStateException(
081              "Encountered multiple GssSaslClientAuthenticationProvider instances");
082        }
083        krbAuth = (GssSaslClientAuthenticationProvider) provider;
084      } else if (DigestSaslAuthenticationProvider.SASL_AUTH_METHOD.getName().equals(name)) {
085        if (digestAuth != null) {
086          throw new IllegalStateException(
087              "Encountered multiple DigestSaslClientAuthenticationProvider instances");
088        }
089        digestAuth = (DigestSaslClientAuthenticationProvider) provider;
090        digestAuthTokenKind = new Text(digestAuth.getTokenKind());
091      } else {
092        LOG.warn("Ignoring unknown SaslClientAuthenticationProvider: {}", provider.getClass());
093      }
094    }
095    if (simpleAuth == null || krbAuth == null || digestAuth == null) {
096      throw new IllegalStateException("Failed to load SIMPLE, KERBEROS, and DIGEST authentication "
097          + "providers. Classpath is not sane.");
098    }
099  }
100
101  @Override
102  public Pair<SaslClientAuthenticationProvider, Token<? extends TokenIdentifier>> selectProvider(
103      String clusterId, User user) {
104    requireNonNull(clusterId, "Null clusterId was given");
105    requireNonNull(user, "Null user was given");
106
107    // Superfluous: we don't do SIMPLE auth over SASL, but we should to simplify.
108    if (!User.isHBaseSecurityEnabled(conf)) {
109      return new Pair<>(simpleAuth, null);
110    }
111
112    final Text clusterIdAsText = new Text(clusterId);
113
114    // Must be digest auth, look for a token.
115    // TestGenerateDelegationToken is written expecting DT is used when DT and Krb are both present.
116    // (for whatever that's worth).
117    for (Token<? extends TokenIdentifier> token : user.getTokens()) {
118      // We need to check for two things:
119      //   1. This token is for the HBase cluster we want to talk to
120      //   2. We have suppporting client implementation to handle the token (the "kind" of token)
121      if (clusterIdAsText.equals(token.getService()) &&
122          digestAuthTokenKind.equals(token.getKind())) {
123        return new Pair<>(digestAuth, token);
124      }
125    }
126    // Unwrap PROXY auth'n method if that's what we have coming in.
127    if (user.getUGI().hasKerberosCredentials() ||
128        user.getUGI().getRealUser().hasKerberosCredentials()) {
129      return new Pair<>(krbAuth, null);
130    }
131    // This indicates that a client is requesting some authentication mechanism which the servers
132    // don't know how to process (e.g. there is no provider which can support it). This may be
133    // a bug or simply a misconfiguration of client *or* server.
134    LOG.warn("No matching SASL authentication provider and supporting token found from providers"
135        + " for user: {}", user);
136    return null;
137  }
138
139}
