168  @Override
169  public void start(CoprocessorEnvironment env) throws IOException {
170    this.conf = env.getConfiguration();
171
172    authorizationEnabled = isAuthorizationSupported(conf);
173    if (!authorizationEnabled) {
174      LOG.warn("The VisibilityController has been loaded with authorization checks disabled.");
175    }
176
177    if (HFile.getFormatVersion(conf) < HFile.MIN_FORMAT_VERSION_WITH_TAGS) {
178      throw new RuntimeException("A minimum HFile version of " + HFile.MIN_FORMAT_VERSION_WITH_TAGS
179        + " is required to persist visibility labels. Consider setting " + HFile.FORMAT_VERSION_KEY
180        + " accordingly.");
181    }
182
183    // Do not create for master CPs
184    if (!(env instanceof MasterCoprocessorEnvironment)) {
185      visibilityLabelService = VisibilityLabelServiceManager.getInstance()
186          .getVisibilityLabelService(this.conf);
187    }
188  }

214  @Override
215  public void postStartMaster(ObserverContext<MasterCoprocessorEnvironment> ctx) throws IOException {
216    // Need to create the new system table for labels here
217    if (!MetaTableAccessor.tableExists(ctx.getEnvironment().getConnection(), LABELS_TABLE_NAME)) {
218      HTableDescriptor labelsTable = new HTableDescriptor(LABELS_TABLE_NAME);
219      HColumnDescriptor labelsColumn = new HColumnDescriptor(LABELS_TABLE_FAMILY);
220      labelsColumn.setBloomFilterType(BloomType.NONE);
221      labelsColumn.setBlockCacheEnabled(false); // We will cache all the labels. No need of normal
222                                                 // table block cache.
223      labelsTable.addFamily(labelsColumn);
224      // Let the "labels" table having only one region always. We are not expecting too many labels in
225      // the system.
226      labelsTable.setValue(HTableDescriptor.SPLIT_POLICY,
227          DisabledRegionSplitPolicy.class.getName());
228      labelsTable.setValue(Bytes.toBytes(HConstants.DISALLOW_WRITES_IN_RECOVERING),
229          Bytes.toBytes(true));
230      try (Admin admin = ctx.getEnvironment().getConnection().getAdmin()) {
231        admin.createTable(labelsTable);
232      }
233    }
234  }

262    // Read the entire labels table and populate the zk
263    if (e.getEnvironment().getRegion().getRegionInfo().getTable().equals(LABELS_TABLE_NAME)) {

275          .getBoolean(VisibilityConstants.CHECK_AUTHS_FOR_MUTATION, false);
276      initVisibilityLabelService(e.getEnvironment());
277    }

309  @Override
310  public void preBatchMutate(ObserverContext<RegionCoprocessorEnvironment> c,
311      MiniBatchOperationInProgress<Mutation> miniBatchOp) throws IOException {
312    if (c.getEnvironment().getRegion().getRegionInfo().getTable().isSystemTable()) {
313      return;
314    }
315    // TODO this can be made as a global LRU cache at HRS level?
316    Map<String, List<Tag>> labelCache = new HashMap<>();
317    for (int i = 0; i < miniBatchOp.size(); i++) {
318      Mutation m = miniBatchOp.getOperation(i);
319      CellVisibility cellVisibility = null;
320      try {
321        cellVisibility = m.getCellVisibility();
322      } catch (DeserializationException de) {
323        miniBatchOp.setOperationStatus(i,
324            new OperationStatus(SANITY_CHECK_FAILURE, de.getMessage()));
325        continue;
326      }
327      boolean sanityFailure = false;
328      boolean modifiedTagFound = false;
329      Pair<Boolean, Tag> pair = new Pair<>(false, null);
330      for (CellScanner cellScanner = m.cellScanner(); cellScanner.advance();) {
331        pair = checkForReservedVisibilityTagPresence(cellScanner.current(), pair);
332        if (!pair.getFirst()) {
333          // Don't disallow reserved tags if authorization is disabled
334          if (authorizationEnabled) {
335            miniBatchOp.setOperationStatus(i, new OperationStatus(SANITY_CHECK_FAILURE,
336              "Mutation contains cell with reserved type tag"));
337            sanityFailure = true;
338          }
339          break;
340        } else {
341          // Indicates that the cell has a the tag which was modified in the src replication cluster
342          Tag tag = pair.getSecond();
343          if (cellVisibility == null && tag != null) {
344            // May need to store only the first one
345            cellVisibility = new CellVisibility(TagUtil.getValueAsString(tag));
346            modifiedTagFound = true;
347          }
348        }
349      }
350      if (!sanityFailure) {
351        if (cellVisibility != null) {
352          String labelsExp = cellVisibility.getExpression();
353          List<Tag> visibilityTags = labelCache.get(labelsExp);
354          if (visibilityTags == null) {
355            // Don't check user auths for labels with Mutations when the user is super user
356            boolean authCheck = authorizationEnabled && checkAuths && !(isSystemOrSuperUser());
357            try {
358              visibilityTags = this.visibilityLabelService.createVisibilityExpTags(labelsExp, true,
359                  authCheck);
360            } catch (InvalidLabelException e) {
361              miniBatchOp.setOperationStatus(i,
362                  new OperationStatus(SANITY_CHECK_FAILURE, e.getMessage()));
363            }
364            if (visibilityTags != null) {
365              labelCache.put(labelsExp, visibilityTags);
366            }
367          }
368          if (visibilityTags != null) {
369            List<Cell> updatedCells = new ArrayList<>();
370            for (CellScanner cellScanner = m.cellScanner(); cellScanner.advance();) {
371              Cell cell = cellScanner.current();
372              List<Tag> tags = CellUtil.getTags(cell);
373              if (modifiedTagFound) {
374                // Rewrite the tags by removing the modified tags.
375                removeReplicationVisibilityTag(tags);
376              }
377              tags.addAll(visibilityTags);
378              Cell updatedCell = CellUtil.createCell(cell, tags);
379              updatedCells.add(updatedCell);
380            }
381            m.getFamilyCellMap().clear();
382            // Clear and add new Cells to the Mutation.
383            for (Cell cell : updatedCells) {
384              if (m instanceof Put) {
385                Put p = (Put) m;
386                p.add(cell);
387              } else if (m instanceof Delete) {
388                Delete d = (Delete) m;
389                d.add(cell);
390              }
391            }
392          }
393        }
394      }
395    }
396  }

398  @Override
399  public void prePrepareTimeStampForDeleteVersion(
400      ObserverContext<RegionCoprocessorEnvironment> ctx, Mutation delete, Cell cell,
401      byte[] byteNow, Get get) throws IOException {
402    // Nothing to do if we are not filtering by visibility
403    if (!authorizationEnabled) {
404      return;
405    }
406
407    CellVisibility cellVisibility = null;
408    try {
409      cellVisibility = delete.getCellVisibility();
410    } catch (DeserializationException de) {
411      throw new IOException("Invalid cell visibility specified " + delete, de);
412    }
413    // The check for checkForReservedVisibilityTagPresence happens in preBatchMutate happens.
414    // It happens for every mutation and that would be enough.
415    List<Tag> visibilityTags = new ArrayList<>();
416    if (cellVisibility != null) {
417      String labelsExp = cellVisibility.getExpression();
418      try {
419        visibilityTags = this.visibilityLabelService.createVisibilityExpTags(labelsExp, false,
420            false);
421      } catch (InvalidLabelException e) {
422        throw new IOException("Invalid cell visibility specified " + labelsExp, e);
423      }
424    }
425    get.setFilter(new DeleteVersionVisibilityExpressionFilter(visibilityTags,
426        VisibilityConstants.SORTED_ORDINAL_SERIALIZATION_FORMAT));
427    List<Cell> result = ctx.getEnvironment().getRegion().get(get, false);
428
429    if (result.size() < get.getMaxVersions()) {
430      // Nothing to delete
431      CellUtil.updateLatestStamp(cell, byteNow, 0);
432      return;
433    }
434    if (result.size() > get.getMaxVersions()) {
435      throw new RuntimeException("Unexpected size: " + result.size()
436          + ". Results more than the max versions obtained.");
437    }
438    Cell getCell = result.get(get.getMaxVersions() - 1);
439    CellUtil.setTimestamp(cell, getCell.getTimestamp());
440
441    // We are bypassing here because in the HRegion.updateDeleteLatestVersionTimeStamp we would
442    // update with the current timestamp after again doing a get. As the hook as already determined
443    // the needed timestamp we need to bypass here.
444    // TODO : See if HRegion.updateDeleteLatestVersionTimeStamp() could be
445    // called only if the hook is not called.
446    ctx.bypass();
447  }

449  /**
450   * Checks whether cell contains any tag with type as VISIBILITY_TAG_TYPE. This
451   * tag type is reserved and should not be explicitly set by user.
452   *
453   * @param cell The cell under consideration
454   * @param pair An optional pair of type {@code <Boolean, Tag>} which would be reused if already
455   *     set and new one will be created if NULL is passed
456   * @return If the boolean is false then it indicates that the cell has a RESERVERD_VIS_TAG and
457   *     with boolean as true, not null tag indicates that a string modified tag was found.
458   */
459  private Pair<Boolean, Tag> checkForReservedVisibilityTagPresence(Cell cell,
460      Pair<Boolean, Tag> pair) throws IOException {
461    if (pair == null) {
462      pair = new Pair<>(false, null);
463    } else {
464      pair.setFirst(false);
465      pair.setSecond(null);
466    }
467    // Bypass this check when the operation is done by a system/super user.
468    // This is done because, while Replication, the Cells coming to the peer cluster with reserved
469    // typed tags and this is fine and should get added to the peer cluster table
470    if (isSystemOrSuperUser()) {
471      // Does the cell contain special tag which indicates that the replicated
472      // cell visiblilty tags
473      // have been modified
474      Tag modifiedTag = null;
475      Iterator<Tag> tagsIterator = CellUtil.tagsIterator(cell);
476      while (tagsIterator.hasNext()) {
477        Tag tag = tagsIterator.next();
478        if (tag.getType() == TagType.STRING_VIS_TAG_TYPE) {
479          modifiedTag = tag;
480          break;
481        }
482      }
483      pair.setFirst(true);
484      pair.setSecond(modifiedTag);
485      return pair;
486    }
487    Iterator<Tag> tagsItr = CellUtil.tagsIterator(cell);
488    while (tagsItr.hasNext()) {
489      if (RESERVED_VIS_TAG_TYPES.contains(tagsItr.next().getType())) {
490        return pair;
491      }
492    }
493    pair.setFirst(true);
494    return pair;
495  }

497  /**
498   * Checks whether cell contains any tag with type as VISIBILITY_TAG_TYPE. This
499   * tag type is reserved and should not be explicitly set by user. There are
500   * two versions of this method one that accepts pair and other without pair.
501   * In case of preAppend and preIncrement the additional operations are not
502   * needed like checking for STRING_VIS_TAG_TYPE and hence the API without pair
503   * could be used.
504   *
505   * @param cell
506   * @throws IOException
507   */
508  private boolean checkForReservedVisibilityTagPresence(Cell cell) throws IOException {
509    // Bypass this check when the operation is done by a system/super user.
510    // This is done because, while Replication, the Cells coming to the peer
511    // cluster with reserved
512    // typed tags and this is fine and should get added to the peer cluster
513    // table
514    if (isSystemOrSuperUser()) {
515      return true;
516    }
517    Iterator<Tag> tagsItr = CellUtil.tagsIterator(cell);
518    while (tagsItr.hasNext()) {
519      if (RESERVED_VIS_TAG_TYPES.contains(tagsItr.next().getType())) {
520        return false;
521      }
522    }
523    return true;
524  }
525
526  private void removeReplicationVisibilityTag(List<Tag> tags) throws IOException {
527    Iterator<Tag> iterator = tags.iterator();
528    while (iterator.hasNext()) {
529      Tag tag = iterator.next();
530      if (tag.getType() == TagType.STRING_VIS_TAG_TYPE) {
531        iterator.remove();
532        break;
533      }
534    }
535  }

537  @Override
538  public RegionScanner preScannerOpen(ObserverContext<RegionCoprocessorEnvironment> e, Scan scan,
539      RegionScanner s) throws IOException {
540    if (!initialized) {
541      throw new VisibilityControllerNotReadyException("VisibilityController not yet initialized!");
542    }
543    // Nothing to do if authorization is not enabled
544    if (!authorizationEnabled) {
545      return s;
546    }
547    Region region = e.getEnvironment().getRegion();
548    Authorizations authorizations = null;
549    try {
550      authorizations = scan.getAuthorizations();
551    } catch (DeserializationException de) {
552      throw new IOException(de);
553    }
554    if (authorizations == null) {
555      // No Authorizations present for this scan/Get!
556      // In case of system tables other than "labels" just scan with out visibility check and
557      // filtering. Checking visibility labels for META and NAMESPACE table is not needed.
558      TableName table = region.getRegionInfo().getTable();
559      if (table.isSystemTable() && !table.equals(LABELS_TABLE_NAME)) {
560        return s;
561      }
562    }
563
564    Filter visibilityLabelFilter = VisibilityUtils.createVisibilityLabelFilter(region,
565        authorizations);
566    if (visibilityLabelFilter != null) {
567      Filter filter = scan.getFilter();
568      if (filter != null) {
569        scan.setFilter(new FilterList(filter, visibilityLabelFilter));
570      } else {
571        scan.setFilter(visibilityLabelFilter);
572      }
573    }
574    return s;
575  }

577  @Override
578  public DeleteTracker postInstantiateDeleteTracker(
579      ObserverContext<RegionCoprocessorEnvironment> ctx, DeleteTracker delTracker)
580      throws IOException {
581    // Nothing to do if we are not filtering by visibility
582    if (!authorizationEnabled) {
583      return delTracker;
584    }
585    Region region = ctx.getEnvironment().getRegion();
586    TableName table = region.getRegionInfo().getTable();
587    if (table.isSystemTable()) {
588      return delTracker;
589    }
590    // We are creating a new type of delete tracker here which is able to track
591    // the timestamps and also the
592    // visibility tags per cell. The covering cells are determined not only
593    // based on the delete type and ts
594    // but also on the visibility expression matching.
595    return new VisibilityScanDeleteTracker(delTracker.getCellComparator());
596  }

608  @Override
609  public boolean preScannerNext(final ObserverContext<RegionCoprocessorEnvironment> c,

630   * Verify, when servicing an RPC, that the caller is the scanner owner. If so, we assume that
631   * access control is correctly enforced based on the checks performed in preScannerOpen()
632   */
633  private void requireScannerOwner(InternalScanner s) throws AccessDeniedException {
634    if (!RpcServer.isInRpcCallContext())
635      return;
636    String requestUName = RpcServer.getRequestUserName().orElse(null);
637    String owner = scannerOwners.get(s);
638    if (authorizationEnabled && owner != null && !owner.equals(requestUName)) {
639      throw new AccessDeniedException("User '" + requestUName + "' is not the scanner owner!");
640    }

643  @Override
644  public void preGetOp(ObserverContext<RegionCoprocessorEnvironment> e, Get get,
645      List<Cell> results) throws IOException {
646    if (!initialized) {
647      throw new VisibilityControllerNotReadyException("VisibilityController not yet initialized");
648    }
649    // Nothing useful to do if authorization is not enabled
650    if (!authorizationEnabled) {
651      return;
652    }
653    Region region = e.getEnvironment().getRegion();
654    Authorizations authorizations = null;
655    try {
656      authorizations = get.getAuthorizations();
657    } catch (DeserializationException de) {
658      throw new IOException(de);
659    }
660    if (authorizations == null) {
661      // No Authorizations present for this scan/Get!
662      // In case of system tables other than "labels" just scan with out visibility check and
663      // filtering. Checking visibility labels for META and NAMESPACE table is not needed.
664      TableName table = region.getRegionInfo().getTable();
665      if (table.isSystemTable() && !table.equals(LABELS_TABLE_NAME)) {
666        return;
667      }
668    }
669    Filter visibilityLabelFilter = VisibilityUtils.createVisibilityLabelFilter(e.getEnvironment()
670        .getRegion(), authorizations);
671    if (visibilityLabelFilter != null) {
672      Filter filter = get.getFilter();
673      if (filter != null) {
674        get.setFilter(new FilterList(filter, visibilityLabelFilter));
675      } else {
676        get.setFilter(visibilityLabelFilter);
677      }
678    }
679  }

715  @Override
716  public Cell postMutationBeforeWAL(ObserverContext<RegionCoprocessorEnvironment> ctx,
717      MutationType opType, Mutation mutation, Cell oldCell, Cell newCell) throws IOException {
718    List<Tag> tags = Lists.newArrayList();
719    CellVisibility cellVisibility = null;
720    try {
721      cellVisibility = mutation.getCellVisibility();
722    } catch (DeserializationException e) {
723      throw new IOException(e);
724    }
725    if (cellVisibility == null) {
726      return newCell;
727    }
728    // Prepend new visibility tags to a new list of tags for the cell
729    // Don't check user auths for labels with Mutations when the user is super user
730    boolean authCheck = authorizationEnabled && checkAuths && !(isSystemOrSuperUser());
731    tags.addAll(this.visibilityLabelService.createVisibilityExpTags(cellVisibility.getExpression(),
732        true, authCheck));
733    // Carry forward all other tags
734    Iterator<Tag> tagsItr = CellUtil.tagsIterator(newCell);
735    while (tagsItr.hasNext()) {
736      Tag tag = tagsItr.next();
737      if (tag.getType() != TagType.VISIBILITY_TAG_TYPE
738          && tag.getType() != TagType.VISIBILITY_EXP_SERIALIZATION_FORMAT_TAG_TYPE) {
739        tags.add(tag);
740      }
741    }
742
743    Cell rewriteCell = CellUtil.createCell(newCell, tags);
744    return rewriteCell;
745  }

755  @Override
756  public synchronized void addLabels(RpcController controller, VisibilityLabelsRequest request,
757      RpcCallback<VisibilityLabelsResponse> done) {
758    VisibilityLabelsResponse.Builder response = VisibilityLabelsResponse.newBuilder();
759    List<VisibilityLabel> visLabels = request.getVisLabelList();
760    if (!initialized) {
761      setExceptionResults(visLabels.size(),
762        new VisibilityControllerNotReadyException("VisibilityController not yet initialized!"),
763        response);
764    } else {
765      List<byte[]> labels = new ArrayList<>(visLabels.size());
766      try {
767        if (authorizationEnabled) {
768          checkCallingUserAuth();
769        }
770        RegionActionResult successResult = RegionActionResult.newBuilder().build();
771        for (VisibilityLabel visLabel : visLabels) {
772          byte[] label = visLabel.getLabel().toByteArray();
773          labels.add(label);
774          response.addResult(successResult); // Just mark as success. Later it will get reset
775                                             // based on the result from
776                                             // visibilityLabelService.addLabels ()
777        }
778        if (!labels.isEmpty()) {
779          OperationStatus[] opStatus = this.visibilityLabelService.addLabels(labels);
780          logResult(true, "addLabels", "Adding labels allowed", null, labels, null);
781          int i = 0;
782          for (OperationStatus status : opStatus) {
783            while (response.getResult(i) != successResult)
784              i++;
785            if (status.getOperationStatusCode() != SUCCESS) {
786              RegionActionResult.Builder failureResultBuilder = RegionActionResult.newBuilder();
787              failureResultBuilder.setException(buildException(new DoNotRetryIOException(
788                  status.getExceptionMsg())));
789              response.setResult(i, failureResultBuilder.build());
790            }
791            i++;
792          }
793        }
794      } catch (AccessDeniedException e) {
795        logResult(false, "addLabels", e.getMessage(), null, labels, null);
796        LOG.error("User is not having required permissions to add labels", e);
797        setExceptionResults(visLabels.size(), e, response);
798      } catch (IOException e) {
799        LOG.error(e);
800        setExceptionResults(visLabels.size(), e, response);
801      }
802    }
803    done.run(response.build());
804  }

806  private void setExceptionResults(int size, IOException e,
807      VisibilityLabelsResponse.Builder response) {
808    RegionActionResult.Builder failureResultBuilder = RegionActionResult.newBuilder();
809    failureResultBuilder.setException(buildException(e));
810    RegionActionResult failureResult = failureResultBuilder.build();
811    for (int i = 0; i < size; i++) {
812      response.addResult(i, failureResult);
813    }

861  private void logResult(boolean isAllowed, String request, String reason, byte[] user,
862      List<byte[]> labelAuths, String regex) {
863    if (AUDITLOG.isTraceEnabled()) {
864      // This is more duplicated code!
865      List<String> labelAuthsStr = new ArrayList<>();
866      if (labelAuths != null) {
867        int labelAuthsSize = labelAuths.size();
868        labelAuthsStr = new ArrayList<>(labelAuthsSize);
869        for (int i = 0; i < labelAuthsSize; i++) {
870          labelAuthsStr.add(Bytes.toString(labelAuths.get(i)));
871        }
872      }
873
874      User requestingUser = null;

876        requestingUser = VisibilityUtils.getActiveUser();

880      }
881      AUDITLOG.trace("Access " + (isAllowed ? "allowed" : "denied") + " for user " +
882          (requestingUser != null ? requestingUser.getShortName() : "UNKNOWN") + "; reason: " +
883          reason + "; remote address: " +
884          RpcServer.getRemoteAddress().map(InetAddress::toString).orElse("") + "; request: " +
885          request + "; user: " + (user != null ? Bytes.toShort(user) : "null") + "; labels: " +
886          labelAuthsStr + "; regex: " + regex);
887    }

890  @Override
891  public synchronized void getAuths(RpcController controller, GetAuthsRequest request,
892      RpcCallback<GetAuthsResponse> done) {
893    GetAuthsResponse.Builder response = GetAuthsResponse.newBuilder();
894    if (!initialized) {
895      controller.setFailed("VisibilityController not yet initialized");
896    } else {
897      byte[] user = request.getUser().toByteArray();
898      List<String> labels = null;
899      try {
900        // We do ACL check here as we create scanner directly on region. It will not make calls to
901        // AccessController CP methods.
902        if (authorizationEnabled && accessControllerAvailable && !isSystemOrSuperUser()) {
903          User requestingUser = VisibilityUtils.getActiveUser();
904          throw new AccessDeniedException("User '"
905              + (requestingUser != null ? requestingUser.getShortName() : "null")
906              + "' is not authorized to perform this action.");
907        }
908        if (AuthUtil.isGroupPrincipal(Bytes.toString(user))) {
909          String group = AuthUtil.getGroupName(Bytes.toString(user));
910          labels = this.visibilityLabelService.getGroupAuths(new String[]{group}, false);
911        }
912        else {
913          labels = this.visibilityLabelService.getUserAuths(user, false);
914        }
915        logResult(true, "getAuths", "Get authorizations for user allowed", user, null, null);
916      } catch (AccessDeniedException e) {
917        logResult(false, "getAuths", e.getMessage(), user, null, null);
918        CoprocessorRpcUtils.setControllerException(controller, e);
919      } catch (IOException e) {
920        CoprocessorRpcUtils.setControllerException(controller, e);
921      }
922      response.setUser(request.getUser());
923      if (labels != null) {
924        for (String label : labels) {
925          response.addAuth(ByteString.copyFrom(Bytes.toBytes(label)));
926        }
927      }
928    }
929    done.run(response.build());
930  }

934      RpcCallback<VisibilityLabelsResponse> done) {
935    VisibilityLabelsResponse.Builder response = VisibilityLabelsResponse.newBuilder();
936    List<ByteString> auths = request.getAuthList();
937    if (!initialized) {
938      setExceptionResults(auths.size(), new CoprocessorException(
939          "VisibilityController not yet initialized"), response);
940    } else {
941      byte[] requestUser = request.getUser().toByteArray();
942      List<byte[]> labelAuths = new ArrayList<>(auths.size());
943      try {
944        // When AC is ON, do AC based user auth check
945        if (authorizationEnabled && accessControllerAvailable && !isSystemOrSuperUser()) {
946          User user = VisibilityUtils.getActiveUser();
947          throw new AccessDeniedException("User '" + (user != null ? user.getShortName() : "null")
948              + " is not authorized to perform this action.");
949        }
950        if (authorizationEnabled) {
951          checkCallingUserAuth(); // When AC is not in place the calling user should have
952                                  // SYSTEM_LABEL auth to do this action.
953        }
954        for (ByteString authBS : auths) {
955          labelAuths.add(authBS.toByteArray());
956        }
957
958        OperationStatus[] opStatus =
959            this.visibilityLabelService.clearAuths(requestUser, labelAuths);
960        logResult(true, "clearAuths", "Removing authorization for labels allowed", requestUser,
961          labelAuths, null);
962        RegionActionResult successResult = RegionActionResult.newBuilder().build();
963        for (OperationStatus status : opStatus) {
964          if (status.getOperationStatusCode() == SUCCESS) {
965            response.addResult(successResult);
966          } else {
967            RegionActionResult.Builder failureResultBuilder = RegionActionResult.newBuilder();
968            failureResultBuilder.setException(buildException(new DoNotRetryIOException(
969                status.getExceptionMsg())));
970            response.addResult(failureResultBuilder.build());
971          }
972        }
973      } catch (AccessDeniedException e) {
974        logResult(false, "clearAuths", e.getMessage(), requestUser, labelAuths, null);
975        LOG.error("User is not having required permissions to clear authorization", e);
976        setExceptionResults(auths.size(), e, response);
977      } catch (IOException e) {
978        LOG.error(e);
979        setExceptionResults(auths.size(), e, response);
980      }
981    }
982    done.run(response.build());
983  }

1020  private void checkCallingUserAuth() throws IOException {
1021    if (!authorizationEnabled) { // Redundant, but just in case
1022      return;
1023    }
1024    if (!accessControllerAvailable) {
1025      User user = VisibilityUtils.getActiveUser();
1026      if (user == null) {
1027        throw new IOException("Unable to retrieve calling user");
1028      }
1029      if (!(this.visibilityLabelService.havingSystemAuth(user))) {
1030        throw new AccessDeniedException("User '" + user.getShortName()
1031            + "' is not authorized to perform this action.");
1032      }
1033    }

1036  private static class DeleteVersionVisibilityExpressionFilter extends FilterBase {
1037    private List<Tag> deleteCellVisTags;
1038    private Byte deleteCellVisTagsFormat;
1039
1040    public DeleteVersionVisibilityExpressionFilter(List<Tag> deleteCellVisTags,
1041        Byte deleteCellVisTagsFormat) {
1042      this.deleteCellVisTags = deleteCellVisTags;
1043      this.deleteCellVisTagsFormat = deleteCellVisTagsFormat;
1044    }
1045
1046    @Override
1047    public boolean filterRowKey(Cell cell) throws IOException {
1048      // Impl in FilterBase might do unnecessary copy for Off heap backed Cells.
1049      return false;
1050    }
1051
1052    @Override
1053    public ReturnCode filterKeyValue(Cell cell) throws IOException {
1054      List<Tag> putVisTags = new ArrayList<>();
1055      Byte putCellVisTagsFormat = VisibilityUtils.extractVisibilityTags(cell, putVisTags);
1056      if (putVisTags.isEmpty() && deleteCellVisTags.isEmpty()) {
1057        // Early out if there are no tags in the cell
1058        return ReturnCode.INCLUDE;
1059      }
1060      boolean matchFound = VisibilityLabelServiceManager
1061          .getInstance().getVisibilityLabelService()
1062          .matchVisibility(putVisTags, putCellVisTagsFormat, deleteCellVisTags,
1063              deleteCellVisTagsFormat);
1064      return matchFound ? ReturnCode.INCLUDE : ReturnCode.SKIP;
1065    }
