019package org.apache.hadoop.hbase.security.access;
020
021import java.io.IOException;
022import java.util.Map;
023
024import org.apache.yetus.audience.InterfaceAudience;
025import org.apache.hadoop.hbase.Cell;
026import org.apache.hadoop.hbase.CellUtil;
027import org.apache.hadoop.hbase.PrivateCellUtil;
028import org.apache.hadoop.hbase.TableName;
029import org.apache.hadoop.hbase.exceptions.DeserializationException;
030import org.apache.hadoop.hbase.filter.FilterBase;
031import org.apache.hadoop.hbase.security.User;
032import org.apache.hadoop.hbase.util.ByteRange;
033import org.apache.hadoop.hbase.util.SimpleMutableByteRange;
034
035/**
036 * <strong>NOTE: for internal use only by AccessController implementation</strong>
037 *
038 * <p>
039 * TODO: There is room for further performance optimization here.
040 * Calling TableAuthManager.authorize() per KeyValue imposes a fair amount of
041 * overhead.  A more optimized solution might look at the qualifiers where
042 * permissions are actually granted and explicitly limit the scan to those.
043 * </p>
044 * <p>
045 * We should aim to use this _only_ when access to the requested column families
046 * is not granted at the column family levels.  If table or column family
047 * access succeeds, then there is no need to impose the overhead of this filter.
048 * </p>
049 */
050@InterfaceAudience.Private
051class AccessControlFilter extends FilterBase {
052
053  public static enum Strategy {
054    /** Filter only by checking the table or CF permissions */
055    CHECK_TABLE_AND_CF_ONLY,
056    /** Cell permissions can override table or CF permissions */
057    CHECK_CELL_DEFAULT,
058  }
059
060  private TableAuthManager authManager;
061  private TableName table;
062  private User user;
063  private boolean isSystemTable;
064  private Strategy strategy;
065  private Map<ByteRange, Integer> cfVsMaxVersions;
066  private int familyMaxVersions;
067  private int currentVersions;
068  private ByteRange prevFam;
069  private ByteRange prevQual;
070
071  /**
072   * For Writable
073   */
074  AccessControlFilter() {
075  }
076
077  AccessControlFilter(TableAuthManager mgr, User ugi, TableName tableName,
078      Strategy strategy, Map<ByteRange, Integer> cfVsMaxVersions) {
079    authManager = mgr;
080    table = tableName;
081    user = ugi;
082    isSystemTable = tableName.isSystemTable();
083    this.strategy = strategy;
084    this.cfVsMaxVersions = cfVsMaxVersions;
085    this.prevFam = new SimpleMutableByteRange();
086    this.prevQual = new SimpleMutableByteRange();
087  }
088
089  @Override
090  public boolean filterRowKey(Cell cell) throws IOException {
091    // Impl in FilterBase might do unnecessary copy for Off heap backed Cells.
092    return false;
093  }
094
095  @Override
096  public ReturnCode filterCell(final Cell cell) {
097    if (isSystemTable) {
098      return ReturnCode.INCLUDE;
099    }
100    if (prevFam.getBytes() == null
101        || !(PrivateCellUtil.matchingFamily(cell, prevFam.getBytes(), prevFam.getOffset(),
102            prevFam.getLength()))) {
103      prevFam.set(cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength());
104      // Similar to VisibilityLabelFilter
105      familyMaxVersions = cfVsMaxVersions.get(prevFam);
106      // Family is changed. Just unset curQualifier.
107      prevQual.unset();
108    }
109    if (prevQual.getBytes() == null
110        || !(PrivateCellUtil.matchingQualifier(cell, prevQual.getBytes(), prevQual.getOffset(),
111            prevQual.getLength()))) {
112      prevQual.set(cell.getQualifierArray(), cell.getQualifierOffset(),
113          cell.getQualifierLength());
114      currentVersions = 0;
115    }
116    currentVersions++;
117    if (currentVersions > familyMaxVersions) {
118      return ReturnCode.SKIP;
119    }
120    // XXX: Compare in place, don't clone
121    byte[] family = CellUtil.cloneFamily(cell);
122    byte[] qualifier = CellUtil.cloneQualifier(cell);
123    switch (strategy) {
124      // Filter only by checking the table or CF permissions
125      case CHECK_TABLE_AND_CF_ONLY: {
126        if (authManager.authorize(user, table, family, qualifier, Permission.Action.READ)) {
127          return ReturnCode.INCLUDE;
128        }
129      }
130      break;
131      // Cell permissions can override table or CF permissions
132      case CHECK_CELL_DEFAULT: {
133        if (authManager.authorize(user, table, family, qualifier, Permission.Action.READ) ||
134            authManager.authorize(user, table, cell, Permission.Action.READ)) {
135          return ReturnCode.INCLUDE;
136        }
137      }
138      break;
139      default:
140        throw new RuntimeException("Unhandled strategy " + strategy);
141    }
142
143    return ReturnCode.SKIP;
144  }
145
146  @Override
147  public void reset() throws IOException {
148    this.prevFam.unset();
149    this.prevQual.unset();
150    this.familyMaxVersions = 0;
151    this.currentVersions = 0;
152  }
153
154  /**
155   * @return The filter serialized using pb
156   */
157  @Override
158  public byte [] toByteArray() {
159    // no implementation, server-side use only
160    throw new UnsupportedOperationException(
161      "Serialization not supported.  Intended for server-side use only.");
162  }
163
164  /**
165   * @param pbBytes A pb serialized {@link AccessControlFilter} instance
166   * @return An instance of {@link AccessControlFilter} made from <code>bytes</code>
167   * @throws org.apache.hadoop.hbase.exceptions.DeserializationException
168   * @see {@link #toByteArray()}
169   */
170  public static AccessControlFilter parseFrom(final byte [] pbBytes)
171  throws DeserializationException {
172    // no implementation, server-side use only
173    throw new UnsupportedOperationException(
174      "Serialization not supported.  Intended for server-side use only.");
175  }
176}
