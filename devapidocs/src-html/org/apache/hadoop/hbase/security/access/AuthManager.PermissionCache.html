049/**
050 * Performs authorization checks for a given user's assigned permissions.
051 * <p>
052 *   There're following scopes: <b>Global</b>, <b>Namespace</b>, <b>Table</b>, <b>Family</b>,
053 *   <b>Qualifier</b>, <b>Cell</b>.
054 *   Generally speaking, higher scopes can overrides lower scopes,
055 *   except for Cell permission can be granted even a user has not permission on specified table,
056 *   which means the user can get/scan only those granted cells parts.
057 * </p>
058 * e.g, if user A has global permission R(ead), he can
059 * read table T without checking table scope permission, so authorization checks alway starts from
060 * Global scope.
061 * <p>
062 *   For each scope, not only user but also groups he belongs to will be checked.
063 * </p>
064 */
065@InterfaceAudience.Private
066public final class AuthManager implements Closeable {
067
068  /**
069   * Cache of permissions, it is thread safe.
070   * @param <T> T extends Permission
071   */
072  private static class PermissionCache<T extends Permission> {
073    private final Object mutex = new Object();
074    private Map<String, Set<T>> cache = new HashMap<>();
075
076    void put(String name, T perm) {
077      synchronized (mutex) {
078        Set<T> perms = cache.getOrDefault(name, new HashSet<>());
079        perms.add(perm);
080        cache.put(name, perms);
081      }
082    }
083
084    Set<T> get(String name) {
085      synchronized (mutex) {
086        return cache.get(name);
087      }
088    }
089
090    void clear() {
091      synchronized (mutex) {
092        for (Map.Entry<String, Set<T>> entry : cache.entrySet()) {
093          entry.getValue().clear();
094        }
095        cache.clear();
096      }
097    }
098  }
099  PermissionCache<NamespacePermission> NS_NO_PERMISSION = new PermissionCache<>();
100  PermissionCache<TablePermission> TBL_NO_PERMISSION = new PermissionCache<>();
101
102  /**
103   * Cache for global permission excluding superuser and supergroup.
104   * Since every user/group can only have one global permission, no need to use PermissionCache.
105   */
106  private Map<String, GlobalPermission> globalCache = new ConcurrentHashMap<>();
107  /** Cache for namespace permission. */
108  private ConcurrentHashMap<String, PermissionCache<NamespacePermission>> namespaceCache =
109    new ConcurrentHashMap<>();
110  /** Cache for table permission. */
111  private ConcurrentHashMap<TableName, PermissionCache<TablePermission>> tableCache =
112    new ConcurrentHashMap<>();
113
114  private static final Logger LOG = LoggerFactory.getLogger(AuthManager.class);

118  private final AtomicLong mtime = new AtomicLong(0L);

120  private AuthManager(ZKWatcher watcher, Configuration conf)
121      throws IOException {
122    this.conf = conf;
123
124    this.zkperms = new ZKPermissionWatcher(watcher, this, conf);
125    try {
126      this.zkperms.start();
127    } catch (KeeperException ke) {
128      LOG.error("ZooKeeper initialization failed", ke);
129    }
130  }
131
132  @Override
133  public void close() {
134    this.zkperms.close();
135  }

137  public ZKPermissionWatcher getZKPermissionWatcher() {
138    return this.zkperms;
139  }

141  /**
142   * Update acl info for table.
143   * @param table name of table
144   * @param data updated acl data
145   * @throws IOException exception when deserialize data
146   */
147  public void refreshTableCacheFromWritable(TableName table, byte[] data) throws IOException {
148    if (data != null && data.length > 0) {
149      try {
150        ListMultimap<String, Permission> perms = PermissionStorage.readPermissions(data, conf);
151        if (perms != null) {
152          if (Bytes.equals(table.getName(), PermissionStorage.ACL_GLOBAL_NAME)) {
153            updateGlobalCache(perms);
154          } else {
155            updateTableCache(table, perms);
156          }
157        }
158      } catch (DeserializationException e) {
159        throw new IOException(e);
160      }
161    } else {
162      LOG.info("Skipping permission cache refresh because writable data is empty");
163    }
164  }
165
166  /**
167   * Update acl info for namespace.
168   * @param namespace namespace
169   * @param data updated acl data
170   * @throws IOException exception when deserialize data
171   */
172  public void refreshNamespaceCacheFromWritable(String namespace, byte[] data) throws IOException {
173    if (data != null && data.length > 0) {
174      try {
175        ListMultimap<String, Permission> perms = PermissionStorage.readPermissions(data, conf);
176        if (perms != null) {
177          updateNamespaceCache(namespace, perms);
178        }
179      } catch (DeserializationException e) {
180        throw new IOException(e);
181      }
182    } else {
183      LOG.debug("Skipping permission cache refresh because writable data is empty");
184    }
185  }

187  /**
188   * Updates the internal global permissions cache.
189   * @param globalPerms new global permissions
190   */
191  private void updateGlobalCache(ListMultimap<String, Permission> globalPerms) {
192    globalCache.clear();
193    for (String name : globalPerms.keySet()) {
194      for (Permission permission : globalPerms.get(name)) {
195        globalCache.put(name, (GlobalPermission) permission);
196      }
197    }
198    mtime.incrementAndGet();
199  }
200
201  /**
202   * Updates the internal table permissions cache for specified table.
203   * @param table updated table name
204   * @param tablePerms new table permissions
205   */
206  private void updateTableCache(TableName table, ListMultimap<String, Permission> tablePerms) {
207    PermissionCache<TablePermission> cacheToUpdate =
208      tableCache.getOrDefault(table, new PermissionCache<>());
209    clearCache(cacheToUpdate);
210    updateCache(tablePerms, cacheToUpdate);
211    tableCache.put(table, cacheToUpdate);
212    mtime.incrementAndGet();
213  }
214
215  /**
216   * Updates the internal namespace permissions cache for specified namespace.
217   * @param namespace updated namespace
218   * @param nsPerms new namespace permissions
219   */
220  private void updateNamespaceCache(String namespace,
221      ListMultimap<String, Permission> nsPerms) {
222    PermissionCache<NamespacePermission> cacheToUpdate =
223      namespaceCache.getOrDefault(namespace, new PermissionCache<>());
224    clearCache(cacheToUpdate);
225    updateCache(nsPerms, cacheToUpdate);
226    namespaceCache.put(namespace, cacheToUpdate);
227    mtime.incrementAndGet();
228  }
229
230  private void clearCache(PermissionCache cacheToUpdate) {
231    cacheToUpdate.clear();
232  }
233
234  @SuppressWarnings("unchecked")
235  private void updateCache(ListMultimap<String, ? extends Permission> newPermissions,
236      PermissionCache cacheToUpdate) {
237    for (String name : newPermissions.keySet()) {
238      for (Permission permission : newPermissions.get(name)) {
239        cacheToUpdate.put(name, permission);
240      }
241    }
242  }

244  /**
245   * Check if user has given action privilige in global scope.
246   * @param user user name
247   * @param action one of action in [Read, Write, Create, Exec, Admin]
248   * @return true if user has, false otherwise
249   */
250  public boolean authorizeUserGlobal(User user, Permission.Action action) {
251    if (user == null) {
252      return false;
253    }
254    if (Superusers.isSuperUser(user)) {
255      return true;
256    }
257    if (authorizeGlobal(globalCache.get(user.getShortName()), action)) {
258      return true;
259    }
260    for (String group : user.getGroupNames()) {
261      if (authorizeGlobal(globalCache.get(AuthUtil.toGroupEntry(group)), action)) {
262        return true;
263      }
264    }
265    return false;
266  }
267
268  private boolean authorizeGlobal(GlobalPermission permissions, Permission.Action action) {
269    return permissions != null && permissions.implies(action);
270  }
271
272  /**
273   * Check if user has given action privilige in namespace scope.
274   * @param user user name
275   * @param namespace namespace
276   * @param action one of action in [Read, Write, Create, Exec, Admin]
277   * @return true if user has, false otherwise
278   */
279  public boolean authorizeUserNamespace(User user, String namespace, Permission.Action action) {
280    if (user == null) {
281      return false;
282    }
283    if (authorizeUserGlobal(user, action)) {
284      return true;
285    }
286    PermissionCache<NamespacePermission> nsPermissions = namespaceCache.getOrDefault(namespace,
287      NS_NO_PERMISSION);
288    if (authorizeNamespace(nsPermissions.get(user.getShortName()), namespace, action)) {
289      return true;
290    }
291    for (String group : user.getGroupNames()) {
292      if (authorizeNamespace(nsPermissions.get(AuthUtil.toGroupEntry(group)), namespace, action)) {
293        return true;
294      }
295    }
296    return false;
297  }
298
299  private boolean authorizeNamespace(Set<NamespacePermission> permissions,
300      String namespace, Permission.Action action) {
301    if (permissions == null) {
302      return false;
303    }
304    for (NamespacePermission permission : permissions) {
305      if (permission.implies(namespace, action)) {
306        return true;
307      }
308    }
309    return false;
310  }

312  /**
313   * Checks if the user has access to the full table or at least a family/qualifier
314   * for the specified action.
315   * @param user user name
316   * @param table table name
317   * @param action action in one of [Read, Write, Create, Exec, Admin]
318   * @return true if the user has access to the table, false otherwise
319   */
320  public boolean accessUserTable(User user, TableName table, Permission.Action action) {
321    if (user == null) {
322      return false;
323    }
324    if (table == null) {
325      table = PermissionStorage.ACL_TABLE_NAME;
326    }
327    if (authorizeUserNamespace(user, table.getNamespaceAsString(), action)) {
328      return true;
329    }
330    PermissionCache<TablePermission> tblPermissions = tableCache.getOrDefault(table,
331      TBL_NO_PERMISSION);
332    if (hasAccessTable(tblPermissions.get(user.getShortName()), action)) {
333      return true;
334    }
335    for (String group : user.getGroupNames()) {
336      if (hasAccessTable(tblPermissions.get(AuthUtil.toGroupEntry(group)), action)) {
337        return true;
338      }
339    }
340    return false;
341  }
342
343  private boolean hasAccessTable(Set<TablePermission> permissions, Permission.Action action) {
344    if (permissions == null) {
345      return false;
346    }
347    for (TablePermission permission : permissions) {
348      if (permission.implies(action)) {
349        return true;
350      }
351    }
352    return false;
353  }

379  /**
380   * Check if user has given action privilige in table:family:qualifier scope.
381   * @param user user name
382   * @param table table name
383   * @param family family name
384   * @param qualifier qualifier name
385   * @param action one of action in [Read, Write, Create, Exec, Admin]
386   * @return true if user has, false otherwise
387   */
388  public boolean authorizeUserTable(User user, TableName table, byte[] family,
389      byte[] qualifier, Permission.Action action) {
390    if (user == null) {
391      return false;
392    }
393    if (table == null) {
394      table = PermissionStorage.ACL_TABLE_NAME;
395    }
396    if (authorizeUserNamespace(user, table.getNamespaceAsString(), action)) {
397      return true;
398    }
399    PermissionCache<TablePermission> tblPermissions = tableCache.getOrDefault(table,
400      TBL_NO_PERMISSION);
401    if (authorizeTable(tblPermissions.get(user.getShortName()), table, family, qualifier, action)) {
402      return true;
403    }
404    for (String group : user.getGroupNames()) {
405      if (authorizeTable(tblPermissions.get(AuthUtil.toGroupEntry(group)),
406          table, family, qualifier, action)) {
407        return true;
408      }
409    }
410    return false;
411  }

464  /**
465   * Check if user has given action privilige in cell scope.
466   * @param user user name
467   * @param table table name
468   * @param cell cell to be checked
469   * @param action one of action in [Read, Write, Create, Exec, Admin]
470   * @return true if user has, false otherwise
471   */
472  public boolean authorizeCell(User user, TableName table, Cell cell, Permission.Action action) {
473    try {
474      List<Permission> perms = PermissionStorage.getCellPermissionsForUser(user, cell);
475      if (LOG.isTraceEnabled()) {
476        LOG.trace("Perms for user {} in table {} in cell {}: {}",
477          user.getShortName(), table, cell, (perms != null ? perms : ""));
478      }
479      if (perms != null) {
480        for (Permission p: perms) {
481          if (p.implies(action)) {
482            return true;
483          }
484        }
485      }
486    } catch (IOException e) {
487      // We failed to parse the KV tag
488      LOG.error("Failed parse of ACL tag in cell " + cell);
489      // Fall through to check with the table and CF perms we were able
490      // to collect regardless
491    }
492    return false;
493  }

519  private static Map<ZKWatcher, AuthManager> managerMap = new HashMap<>();
520
521  private static Map<AuthManager, Integer> refCount = new HashMap<>();
522
523  /**
524   * Returns a AuthManager from the cache. If not cached, constructs a new one.
525   * Returned instance should be released back by calling {@link #release(AuthManager)}.
526   * @param watcher zk watcher
527   * @param conf configuration
528   * @return an AuthManager
529   * @throws IOException zookeeper initialization failed
530   */
531  public synchronized static AuthManager getOrCreate(
532      ZKWatcher watcher, Configuration conf) throws IOException {
533    AuthManager instance = managerMap.get(watcher);
534    if (instance == null) {
535      instance = new AuthManager(watcher, conf);
536      managerMap.put(watcher, instance);
537    }
538    int ref = refCount.get(instance) == null ? 0 : refCount.get(instance);
539    refCount.put(instance, ref + 1);
540    return instance;
541  }
542
543  @VisibleForTesting
544  public static int getTotalRefCount() {
545    int total = 0;
546    for (int count : refCount.values()) {
547      total += count;
548    }
549    return total;
550  }
551
552  /**
553   * Releases the resources for the given AuthManager if the reference count is down to 0.
554   * @param instance AuthManager to be released
555   */
556  public synchronized static void release(AuthManager instance) {
557    if (refCount.get(instance) == null || refCount.get(instance) < 1) {
558      String msg = "Something wrong with the AuthManager reference counting: " + instance
559          + " whose count is " + refCount.get(instance);
560      LOG.error(HBaseMarkers.FATAL, msg);
561      instance.close();
562      managerMap.remove(instance.getZKPermissionWatcher().getWatcher());
563      instance.getZKPermissionWatcher().getWatcher().abort(msg, null);
564    } else {
565      int ref = refCount.get(instance);
566      --ref;
567      refCount.put(instance, ref);
568      if (ref == 0) {
569        instance.close();
570        managerMap.remove(instance.getZKPermissionWatcher().getWatcher());
571        refCount.remove(instance);
572      }
573    }
574  }
