048  private static final Logger AUDITLOG =

050  // TODO: we should move to a design where we don't even instantiate an AccessChecker if

058  /**
059   * if we are active, usually false, only true if "hbase.security.authorization"
060   * has been set to true in site configuration.see HBASE-19483.
061   */
062  private boolean authorizationEnabled;

064  public static boolean isAuthorizationSupported(Configuration conf) {
065    return conf.getBoolean(User.HBASE_SECURITY_AUTHORIZATION_CONF_KEY, false);
066  }

068  /**
069   * Constructor with existing configuration
070   *
071   * @param conf Existing configuration to use
072   * @param zkw reference to the {@link ZKWatcher}
073   */
074  public AccessChecker(final Configuration conf, final ZKWatcher zkw)
075      throws RuntimeException {
076    if (zkw != null) {
077      try {
078        this.authManager = TableAuthManager.getOrCreate(zkw, conf);
079      } catch (IOException ioe) {
080        throw new RuntimeException("Error obtaining AccessChecker", ioe);
081      }
082    } else {
083      throw new NullPointerException("Error obtaining AccessChecker, zk found null.");
084    }
085    authorizationEnabled = isAuthorizationSupported(conf);
086    initGroupService(conf);
087  }

100  /**
101   * Authorizes that the current user has any of the given permissions to access the table.
102   *
103   * @param user Active user to which authorization checks should be applied
104   * @param request Request type.
105   * @param tableName   Table requested
106   * @param permissions Actions being requested
107   * @throws IOException if obtaining the current user fails
108   * @throws AccessDeniedException if user has no authorization
109   */
110  public void requireAccess(User user, String request, TableName tableName,
111      Action... permissions) throws IOException {
112    if (!authorizationEnabled) {
113      return;
114    }
115    AuthResult result = null;
116
117    for (Action permission : permissions) {
118      if (authManager.hasAccess(user, tableName, permission)) {
119        result = AuthResult.allow(request, "Table permission granted",
120            user, permission, tableName, null, null);
121        break;
122      } else {
123        // rest of the world
124        result = AuthResult.deny(request, "Insufficient permissions",
125            user, permission, tableName, null, null);
126      }
127    }
128    logResult(result);
129    if (!result.isAllowed()) {
130      throw new AccessDeniedException("Insufficient permissions " + result.toContextString());
131    }
132  }

134  /**
135   * Authorizes that the current user has global privileges for the given action.
136   * @param user Active user to which authorization checks should be applied
137   * @param request Request type
138   * @param filterUser User name to be filtered from permission as requested
139   * @param perm The action being requested
140   * @throws IOException if obtaining the current user fails
141   * @throws AccessDeniedException if authorization is denied
142   */
143  public void requirePermission(User user, String request, String filterUser, Action perm)
144      throws IOException {
145    requireGlobalPermission(user, request, perm, null, null, filterUser);
146  }

148  /**
149   * Checks that the user has the given global permission. The generated
150   * audit log message will contain context information for the operation
151   * being authorized, based on the given parameters.
152   *
153   * @param user Active user to which authorization checks should be applied
154   * @param request Request type
155   * @param perm      Action being requested
156   * @param tableName Affected table name.
157   * @param familyMap Affected column families.
158   * @param filterUser User name to be filtered from permission as requested
159   */
160  public void requireGlobalPermission(User user, String request,
161      Action perm, TableName tableName,
162      Map<byte[], ? extends Collection<byte[]>> familyMap, String filterUser) throws IOException {
163    if (!authorizationEnabled) {
164      return;
165    }
166    AuthResult result;
167    if (authManager.authorize(user, perm)) {
168      result = AuthResult.allow(request, "Global check allowed", user, perm, tableName, familyMap);
169    } else {
170      result = AuthResult.deny(request, "Global check failed", user, perm, tableName, familyMap);
171    }
172    result.getParams().setTableName(tableName).setFamilies(familyMap);
173    result.getParams().addExtraParam("filterUser", filterUser);
174    logResult(result);
175    if (!result.isAllowed()) {
176      throw new AccessDeniedException(
177          "Insufficient permissions for user '" + (user != null ? user.getShortName() : "null")
178              + "' (global, action=" + perm.toString() + ")");
179    }
180  }

182  /**
183   * Checks that the user has the given global permission. The generated
184   * audit log message will contain context information for the operation
185   * being authorized, based on the given parameters.
186   *
187   * @param user Active user to which authorization checks should be applied
188   * @param request Request type
189   * @param perm      Action being requested
190   * @param namespace The given namespace
191   */
192  public void requireGlobalPermission(User user, String request, Action perm,
193      String namespace) throws IOException {
194    if (!authorizationEnabled) {
195      return;
196    }
197    AuthResult authResult;
198    if (authManager.authorize(user, perm)) {
199      authResult = AuthResult.allow(request, "Global check allowed", user, perm, null);
200      authResult.getParams().setNamespace(namespace);
201      logResult(authResult);
202    } else {
203      authResult = AuthResult.deny(request, "Global check failed", user, perm, null);
204      authResult.getParams().setNamespace(namespace);
205      logResult(authResult);
206      throw new AccessDeniedException(
207          "Insufficient permissions for user '" + (user != null ? user.getShortName() : "null")
208              + "' (global, action=" + perm.toString() + ")");
209    }
210  }

358  public void checkLockPermissions(User user, String namespace,
359      TableName tableName, RegionInfo[] regionInfos, String reason)
360      throws IOException {
361    if (namespace != null && !namespace.isEmpty()) {
362      requireNamespacePermission(user, reason, namespace, null, Action.ADMIN, Action.CREATE);
363    } else if (tableName != null || (regionInfos != null && regionInfos.length > 0)) {
364      // So, either a table or regions op. If latter, check perms ons table.
365      TableName tn = tableName != null? tableName: regionInfos[0].getTable();
366      requireTablePermission(user, reason, tn, null, null,
367          Action.ADMIN, Action.CREATE);
368    } else {
369      throw new DoNotRetryIOException("Invalid lock level when requesting permissions.");
370    }
371  }

373  public static void logResult(AuthResult result) {
374    if (AUDITLOG.isTraceEnabled()) {
375      AUDITLOG.trace(
376        "Access {} for user {}; reason: {}; remote address: {}; request: {}; context: {}",
377        (result.isAllowed() ? "allowed" : "denied"),
378        (result.getUser() != null ? result.getUser().getShortName() : "UNKNOWN"),
379        result.getReason(), RpcServer.getRemoteAddress().map(InetAddress::toString).orElse(""),
380        result.getRequest(), result.toContextString());
381    }
382  }

384  /*
385   * Validate the hasPermission operation caller with the filter user. Self check doesn't require
386   * any privilege but for others caller must have ADMIN privilege.
387   */
388  public User validateCallerWithFilterUser(User caller, TablePermission tPerm, String inputUserName)
389      throws IOException {
390    User filterUser = null;
391    if (!caller.getShortName().equals(inputUserName)) {
392      // User should have admin privilege if checking permission for other users
393      requirePermission(caller, "hasPermission", tPerm.getTableName(), tPerm.getFamily(),
394        tPerm.getQualifier(), inputUserName, Action.ADMIN);
395      // Initialize user instance for the input user name
396      List<String> groups = getUserGroups(inputUserName);
397      filterUser = new InputUser(inputUserName, groups.toArray(new String[groups.size()]));
398    } else {
399      // User don't need ADMIN privilege for self check.
400      // Setting action as null in AuthResult to display empty action in audit log
401      AuthResult result = AuthResult.allow("hasPermission", "Self user validation allowed", caller,
402        null, tPerm.getTableName(), tPerm.getFamily(), tPerm.getQualifier());
403      logResult(result);
404      filterUser = caller;
405    }
406    return filterUser;
407  }

409  /**
410   * A temporary user class to instantiate User instance based on the name and groups.
411   */
412  public static class InputUser extends User {
413    private String name;
414    private String shortName = null;
415    private String[] groups;
416
417    public InputUser(String name, String[] groups) {
418      this.name = name;
419      this.groups = groups;
420    }
421
422    @Override
423    public String getShortName() {
424      if (this.shortName == null) {
425        try {
426          this.shortName = new HadoopKerberosName(this.name).getShortName();
427        } catch (IOException ioe) {
428          throw new IllegalArgumentException(
429              "Illegal principal name " + this.name + ": " + ioe.toString(), ioe);
430        }
431      }
432      return shortName;
433    }
434
435    @Override
436    public String getName() {
437      return this.name;
438    }
439
440    @Override
441    public String[] getGroupNames() {
442      return this.groups;
443    }
444
445    @Override
446    public <T> T runAs(PrivilegedAction<T> action) {
447      throw new UnsupportedOperationException(
448          "Method not supported, this class has limited implementation");
449    }
450
451    @Override
452    public <T> T runAs(PrivilegedExceptionAction<T> action)
453        throws IOException, InterruptedException {
454      throw new UnsupportedOperationException(
455          "Method not supported, this class has limited implementation");
456    }
457
458    @Override
459    public String toString() {
460      return this.name;
461    }
462  }

464  /*
465   * Initialize the group service.
466   */
467  private void initGroupService(Configuration conf) {
468    if (groupService == null) {
469      groupService = Groups.getUserToGroupsMappingService(conf);
470    }
471  }
472
473  /**
474   * Retrieve the groups of the given user.
475   * @param user User name
476   * @return Groups
477   */
478  public static List<String> getUserGroups(String user) {
479    try {
480      return groupService.getGroups(user);
481    } catch (IOException e) {
482      LOG.error("Error occured while retrieving group for " + user, e);
483      return new ArrayList<String>();
484    }
485  }
