182  /**
183   * List all the userPermissions matching the given pattern. If pattern is null, the behavior is
184   * dependent on whether user has global admin privileges or not. If yes, the global permissions
185   * along with the list of superusers would be returned. Else, no rows get returned.
186   * @param connection The Connection instance to use
187   * @param tableRegex The regular expression string to match against
188   * @return List of UserPermissions n
189   */
190  public static List<UserPermission> getUserPermissions(Connection connection, String tableRegex)
191    throws Throwable {
192    return getUserPermissions(connection, tableRegex, HConstants.EMPTY_STRING);
193  }
194
195  /**
196   * List all the userPermissions matching the given table pattern and user name.
197   * @param connection Connection
198   * @param tableRegex The regular expression string to match against
199   * @param userName   User name, if empty then all user permissions will be retrieved.
200   * @return List of UserPermissions
201   * @throws Throwable on failure
202   */
203  public static List<UserPermission> getUserPermissions(Connection connection, String tableRegex,
204    String userName) throws Throwable {
205    List<UserPermission> permList = new ArrayList<>();
206    try (Admin admin = connection.getAdmin()) {
207      if (tableRegex == null || tableRegex.isEmpty()) {
208        permList = admin.getUserPermissions(
209          GetUserPermissionsRequest.newBuilder().withUserName(userName).build());
210      } else if (tableRegex.charAt(0) == '@') { // Namespaces
211        String namespaceRegex = tableRegex.substring(1);
212        for (NamespaceDescriptor nsds : admin.listNamespaceDescriptors()) { // Read out all
213                                                                            // namespaces
214          String namespace = nsds.getName();
215          if (namespace.matches(namespaceRegex)) { // Match the given namespace regex?
216            permList.addAll(admin.getUserPermissions(
217              GetUserPermissionsRequest.newBuilder(namespace).withUserName(userName).build()));
218          }
219        }
220      } else { // Tables
221        List<TableDescriptor> htds = admin.listTableDescriptors(Pattern.compile(tableRegex), true);
222        for (TableDescriptor htd : htds) {
223          permList.addAll(admin.getUserPermissions(GetUserPermissionsRequest
224            .newBuilder(htd.getTableName()).withUserName(userName).build()));
225        }
226      }
227    }
228    return permList;
229  }

277  /**
278   * List all the userPermissions matching the given table pattern, column family and column
279   * qualifier.
280   * @param connection      Connection
281   * @param tableRegex      The regular expression string to match against. It shouldn't be null,
282   *                        empty or a namespace regular expression.
283   * @param columnFamily    Column family
284   * @param columnQualifier Column qualifier
285   * @param userName        User name, if empty then all user permissions will be retrieved.
286   * @return List of UserPermissions
287   * @throws Throwable on failure
288   */
289  public static List<UserPermission> getUserPermissions(Connection connection, String tableRegex,
290    byte[] columnFamily, byte[] columnQualifier, String userName) throws Throwable {
291    if (tableRegex == null || tableRegex.isEmpty() || tableRegex.charAt(0) == '@') {
292      throw new IllegalArgumentException("Table name can't be null or empty or a namespace.");
293    }
294    List<UserPermission> permList = new ArrayList<UserPermission>();
295    try (Admin admin = connection.getAdmin()) {
296      List<TableDescriptor> htds = admin.listTableDescriptors(Pattern.compile(tableRegex), true);
297      // Retrieve table permissions
298      for (TableDescriptor htd : htds) {
299        permList.addAll(admin.getUserPermissions(
300          GetUserPermissionsRequest.newBuilder(htd.getTableName()).withFamily(columnFamily)
301            .withQualifier(columnQualifier).withUserName(userName).build()));
302      }
303    }
304    return permList;
305  }

307  /**
308   * Validates whether specified user has permission to perform actions on the mentioned table,
309   * column family or column qualifier.
310   * @param connection      Connection
311   * @param tableName       Table name, it shouldn't be null or empty.
312   * @param columnFamily    The column family. Optional argument, can be empty. If empty then
313   *                        validation will happen at table level.
314   * @param columnQualifier The column qualifier. Optional argument, can be empty. If empty then
315   *                        validation will happen at table and column family level. columnQualifier
316   *                        will not be considered if columnFamily is passed as null or empty.
317   * @param userName        User name, it shouldn't be null or empty.
318   * @param actions         Actions
319   * @return true if access allowed to the specified user, otherwise false.
320   * @throws Throwable on failure
321   */
322  public static boolean hasPermission(Connection connection, String tableName, String columnFamily,
323    String columnQualifier, String userName, Permission.Action... actions) throws Throwable {
324    return hasPermission(connection, tableName, Bytes.toBytes(columnFamily),
325      Bytes.toBytes(columnQualifier), userName, actions);
326  }
327
328  /**
329   * Validates whether specified user has permission to perform actions on the mentioned table,
330   * column family or column qualifier.
331   * @param connection      Connection
332   * @param tableName       Table name, it shouldn't be null or empty.
333   * @param columnFamily    The column family. Optional argument, can be empty. If empty then
334   *                        validation will happen at table level.
335   * @param columnQualifier The column qualifier. Optional argument, can be empty. If empty then
336   *                        validation will happen at table and column family level. columnQualifier
337   *                        will not be considered if columnFamily is passed as null or empty.
338   * @param userName        User name, it shouldn't be null or empty.
339   * @param actions         Actions
340   * @return true if access allowed to the specified user, otherwise false.
341   * @throws Throwable on failure
342   */
343  public static boolean hasPermission(Connection connection, String tableName, byte[] columnFamily,
344    byte[] columnQualifier, String userName, Permission.Action... actions) throws Throwable {
345    if (StringUtils.isEmpty(tableName) || StringUtils.isEmpty(userName)) {
346      throw new IllegalArgumentException("Table and user name can't be null or empty.");
347    }
348    List<Permission> permissions = new ArrayList<>(1);
349    permissions.add(Permission.newBuilder(TableName.valueOf(tableName)).withFamily(columnFamily)
350      .withQualifier(columnQualifier).withActions(actions).build());
351    return connection.getAdmin().hasUserPermissions(userName, permissions).get(0);
352  }
