141/**
142 * Provides basic authorization checks for data access and administrative
143 * operations.
144 *
145 * <p>
146 * {@code AccessController} performs authorization checks for HBase operations
147 * based on:
148 * </p>
149 * <ul>
150 *   <li>the identity of the user performing the operation</li>
151 *   <li>the scope over which the operation is performed, in increasing
152 *   specificity: global, table, column family, or qualifier</li>
153 *   <li>the type of action being performed (as mapped to
154 *   {@link Permission.Action} values)</li>
155 * </ul>
156 * <p>
157 * If the authorization check fails, an {@link AccessDeniedException}
158 * will be thrown for the operation.
159 * </p>
160 *
161 * <p>
162 * To perform authorization checks, {@code AccessController} relies on the
163 * RpcServerEngine being loaded to provide
164 * the user identities for remote requests.
165 * </p>
166 *
167 * <p>
168 * The access control lists used for authorization can be manipulated via the
169 * exposed {@link AccessControlService} Interface implementation, and the associated
170 * {@code grant}, {@code revoke}, and {@code user_permission} HBase shell
171 * commands.
172 * </p>
173 */

243  void initialize(RegionCoprocessorEnvironment e) throws IOException {
244    final Region region = e.getRegion();
245    Configuration conf = e.getConfiguration();
246    Map<byte[], ListMultimap<String,TablePermission>> tables =
247        AccessControlLists.loadAll(region);
248    // For each table, write out the table's permissions to the respective
249    // znode for that table.
250    for (Map.Entry<byte[], ListMultimap<String,TablePermission>> t:
251      tables.entrySet()) {
252      byte[] entry = t.getKey();
253      ListMultimap<String,TablePermission> perms = t.getValue();
254      byte[] serialized = AccessControlLists.writePermissionsAsBytes(perms, conf);
255      this.authManager.getZKPermissionWatcher().writeToZookeeper(entry, serialized);
256    }
257    initialized = true;
258  }

260  /**
261   * Writes all table ACLs for the tables in the given Map up into ZooKeeper
262   * znodes.  This is called to synchronize ACL changes following {@code _acl_}
263   * table updates.
264   */
265  void updateACL(RegionCoprocessorEnvironment e,
266      final Map<byte[], List<Cell>> familyMap) {
267    Set<byte[]> entries = new TreeSet<>(Bytes.BYTES_RAWCOMPARATOR);
268    for (Map.Entry<byte[], List<Cell>> f : familyMap.entrySet()) {
269      List<Cell> cells = f.getValue();
270      for (Cell cell: cells) {
271        if (CellUtil.matchingFamily(cell, AccessControlLists.ACL_LIST_FAMILY)) {
272          entries.add(CellUtil.cloneRow(cell));
273        }
274      }
275    }
276    ZKPermissionWatcher zkw = this.authManager.getZKPermissionWatcher();
277    Configuration conf = regionEnv.getConfiguration();
278    for (byte[] entry: entries) {
279      try {
280        try (Table t = regionEnv.getTable(AccessControlLists.ACL_TABLE_NAME)) {
281          ListMultimap<String,TablePermission> perms =
282              AccessControlLists.getPermissions(conf, entry, t);
283          byte[] serialized = AccessControlLists.writePermissionsAsBytes(perms, conf);
284          zkw.writeToZookeeper(entry, serialized);
285        }
286      } catch (IOException ex) {
287        LOG.error("Failed updating permissions mirror for '" + Bytes.toString(entry) + "'",
288            ex);
289      }
290    }
291  }

293  /**
294   * Check the current user for authorization to perform a specific action
295   * against the given set of row data.
296   *
297   * <p>Note: Ordering of the authorization checks
298   * has been carefully optimized to short-circuit the most common requests
299   * and minimize the amount of processing required.</p>
300   *
301   * @param permRequest the action being requested
302   * @param e the coprocessor environment
303   * @param families the map of column families to qualifiers present in
304   * the request
305   * @return an authorization result
306   */
307  AuthResult permissionGranted(String request, User user, Action permRequest,
308      RegionCoprocessorEnvironment e,
309      Map<byte [], ? extends Collection<?>> families) {
310    RegionInfo hri = e.getRegion().getRegionInfo();
311    TableName tableName = hri.getTable();
312
313    // 1. All users need read access to hbase:meta table.
314    // this is a very common operation, so deal with it quickly.
315    if (hri.isMetaRegion()) {
316      if (permRequest == Action.READ) {
317        return AuthResult.allow(request, "All users allowed", user,
318          permRequest, tableName, families);
319      }
320    }
321
322    if (user == null) {
323      return AuthResult.deny(request, "No user associated with request!", null,
324        permRequest, tableName, families);
325    }
326
327    // 2. check for the table-level, if successful we can short-circuit
328    if (authManager.authorize(user, tableName, (byte[])null, permRequest)) {
329      return AuthResult.allow(request, "Table permission granted", user,
330        permRequest, tableName, families);
331    }
332
333    // 3. check permissions against the requested families
334    if (families != null && families.size() > 0) {
335      // all families must pass
336      for (Map.Entry<byte [], ? extends Collection<?>> family : families.entrySet()) {
337        // a) check for family level access
338        if (authManager.authorize(user, tableName, family.getKey(),
339            permRequest)) {
340          continue;  // family-level permission overrides per-qualifier
341        }
342
343        // b) qualifier level access can still succeed
344        if ((family.getValue() != null) && (family.getValue().size() > 0)) {
345          if (family.getValue() instanceof Set) {
346            // for each qualifier of the family
347            Set<byte[]> familySet = (Set<byte[]>)family.getValue();
348            for (byte[] qualifier : familySet) {
349              if (!authManager.authorize(user, tableName, family.getKey(),
350                                         qualifier, permRequest)) {
351                return AuthResult.deny(request, "Failed qualifier check", user,
352                    permRequest, tableName, makeFamilyMap(family.getKey(), qualifier));
353              }
354            }
355          } else if (family.getValue() instanceof List) { // List<Cell>
356            List<Cell> cellList = (List<Cell>)family.getValue();
357            for (Cell cell : cellList) {
358              if (!authManager.authorize(user, tableName, family.getKey(),
359                CellUtil.cloneQualifier(cell), permRequest)) {
360                return AuthResult.deny(request, "Failed qualifier check", user, permRequest,
361                  tableName, makeFamilyMap(family.getKey(), CellUtil.cloneQualifier(cell)));
362              }
363            }
364          }
365        } else {
366          // no qualifiers and family-level check already failed
367          return AuthResult.deny(request, "Failed family check", user, permRequest,
368              tableName, makeFamilyMap(family.getKey(), null));
369        }
370      }
371
372      // all family checks passed
373      return AuthResult.allow(request, "All family checks passed", user, permRequest,
374          tableName, families);
375    }
376
377    // 4. no families to check and table level access failed
378    return AuthResult.deny(request, "No families to check and table permission failed",
379        user, permRequest, tableName, families);
380  }

382  /**
383   * Check the current user for authorization to perform a specific action
384   * against the given set of row data.
385   * @param opType the operation type
386   * @param user the user
387   * @param e the coprocessor environment
388   * @param families the map of column families to qualifiers present in
389   * the request
390   * @param actions the desired actions
391   * @return an authorization result
392   */
393  AuthResult permissionGranted(OpType opType, User user, RegionCoprocessorEnvironment e,
394      Map<byte [], ? extends Collection<?>> families, Action... actions) {
395    AuthResult result = null;
396    for (Action action: actions) {
397      result = permissionGranted(opType.toString(), user, action, e, families);
398      if (!result.isAllowed()) {
399        return result;
400      }
401    }
402    return result;
403  }

413          "; context: " + result.toContextString());

431  /**
432   * Authorizes that the current user has any of the given permissions for the
433   * given table, column family and column qualifier.
434   * @param tableName Table requested
435   * @param family Column family requested
436   * @param qualifier Column qualifier requested
437   * @throws IOException if obtaining the current user fails
438   * @throws AccessDeniedException if user has no authorization
439   */
440  private void requirePermission(User user, String request, TableName tableName, byte[] family,
441      byte[] qualifier, Action... permissions) throws IOException {
442    AuthResult result = null;
443
444    for (Action permission : permissions) {
445      if (authManager.authorize(user, tableName, family, qualifier, permission)) {
446        result = AuthResult.allow(request, "Table permission granted", user,
447                                  permission, tableName, family, qualifier);
448        break;
449      } else {
450        // rest of the world
451        result = AuthResult.deny(request, "Insufficient permissions", user,
452                                 permission, tableName, family, qualifier);
453      }
454    }
455    logResult(result);
456    if (authorizationEnabled && !result.isAllowed()) {
457      throw new AccessDeniedException("Insufficient permissions " + result.toContextString());
458    }
459  }

600        // rest of the world
601        result = AuthResult.deny(request, "Insufficient permissions", user,
602            permission, namespace);
603      }
604    }
605    logResult(result);
606    if (authorizationEnabled && !result.isAllowed()) {
607      throw new AccessDeniedException("Insufficient permissions "
608          + result.toContextString());
609    }
610  }

639          + result.toContextString());

644   * Returns <code>true</code> if the current user is allowed the given action
645   * over at least one of the column qualifiers in the given column families.
646   */
647  private boolean hasFamilyQualifierPermission(User user,
648      Action perm,
649      RegionCoprocessorEnvironment env,
650      Map<byte[], ? extends Collection<byte[]>> familyMap)
651    throws IOException {
652    RegionInfo hri = env.getRegion().getRegionInfo();
653    TableName tableName = hri.getTable();
654
655    if (user == null) {
656      return false;
657    }
658
659    if (familyMap != null && familyMap.size() > 0) {
660      // at least one family must be allowed
661      for (Map.Entry<byte[], ? extends Collection<byte[]>> family :
662          familyMap.entrySet()) {
663        if (family.getValue() != null && !family.getValue().isEmpty()) {
664          for (byte[] qualifier : family.getValue()) {
665            if (authManager.matchPermission(user, tableName,
666                family.getKey(), qualifier, perm)) {
667              return true;
668            }
669          }
670        } else {
671          if (authManager.matchPermission(user, tableName, family.getKey(),
672              perm)) {
673            return true;
674          }
675        }
676      }
677    } else if (LOG.isDebugEnabled()) {
678      LOG.debug("Empty family map passed for permission check");
679    }
680
681    return false;
682  }

708  /**
709   * Determine if cell ACLs covered by the operation grant access. This is expensive.
710   * @return false if cell ACLs failed to grant access, true otherwise
711   * @throws IOException
712   */
713  private boolean checkCoveringPermission(User user, OpType request, RegionCoprocessorEnvironment e,
714      byte[] row, Map<byte[], ? extends Collection<?>> familyMap, long opTs, Action... actions)
715      throws IOException {
716    if (!cellFeaturesEnabled) {
717      return false;
718    }
719    long cellGrants = 0;
720    long latestCellTs = 0;
721    Get get = new Get(row);
722    // Only in case of Put/Delete op, consider TS within cell (if set for individual cells).
723    // When every cell, within a Mutation, can be linked with diff TS we can not rely on only one
724    // version. We have to get every cell version and check its TS against the TS asked for in
725    // Mutation and skip those Cells which is outside this Mutation TS.In case of Put, we have to
726    // consider only one such passing cell. In case of Delete we have to consider all the cell
727    // versions under this passing version. When Delete Mutation contains columns which are a
728    // version delete just consider only one version for those column cells.
729    boolean considerCellTs  = (request == OpType.PUT || request == OpType.DELETE);
730    if (considerCellTs) {
731      get.setMaxVersions();
732    } else {
733      get.setMaxVersions(1);
734    }
735    boolean diffCellTsFromOpTs = false;
736    for (Map.Entry<byte[], ? extends Collection<?>> entry: familyMap.entrySet()) {
737      byte[] col = entry.getKey();
738      // TODO: HBASE-7114 could possibly unify the collection type in family
739      // maps so we would not need to do this
740      if (entry.getValue() instanceof Set) {
741        Set<byte[]> set = (Set<byte[]>)entry.getValue();
742        if (set == null || set.isEmpty()) {
743          get.addFamily(col);
744        } else {
745          for (byte[] qual: set) {
746            get.addColumn(col, qual);
747          }
748        }
749      } else if (entry.getValue() instanceof List) {
750        List<Cell> list = (List<Cell>)entry.getValue();
751        if (list == null || list.isEmpty()) {
752          get.addFamily(col);
753        } else {
754          // In case of family delete, a Cell will be added into the list with Qualifier as null.
755          for (Cell cell : list) {
756            if (cell.getQualifierLength() == 0
757                && (cell.getTypeByte() == Type.DeleteFamily.getCode()
758                || cell.getTypeByte() == Type.DeleteFamilyVersion.getCode())) {
759              get.addFamily(col);
760            } else {
761              get.addColumn(col, CellUtil.cloneQualifier(cell));
762            }
763            if (considerCellTs) {
764              long cellTs = cell.getTimestamp();
765              latestCellTs = Math.max(latestCellTs, cellTs);
766              diffCellTsFromOpTs = diffCellTsFromOpTs || (opTs != cellTs);
767            }
768          }
769        }
770      } else if (entry.getValue() == null) {
771        get.addFamily(col);
772      } else {
773        throw new RuntimeException("Unhandled collection type " +
774          entry.getValue().getClass().getName());
775      }
776    }
777    // We want to avoid looking into the future. So, if the cells of the
778    // operation specify a timestamp, or the operation itself specifies a
779    // timestamp, then we use the maximum ts found. Otherwise, we bound
780    // the Get to the current server time. We add 1 to the timerange since
781    // the upper bound of a timerange is exclusive yet we need to examine
782    // any cells found there inclusively.
783    long latestTs = Math.max(opTs, latestCellTs);
784    if (latestTs == 0 || latestTs == HConstants.LATEST_TIMESTAMP) {
785      latestTs = EnvironmentEdgeManager.currentTime();
786    }
787    get.setTimeRange(0, latestTs + 1);
788    // In case of Put operation we set to read all versions. This was done to consider the case
789    // where columns are added with TS other than the Mutation TS. But normally this wont be the
790    // case with Put. There no need to get all versions but get latest version only.
791    if (!diffCellTsFromOpTs && request == OpType.PUT) {
792      get.setMaxVersions(1);
793    }
794    if (LOG.isTraceEnabled()) {
795      LOG.trace("Scanning for cells with " + get);
796    }
797    // This Map is identical to familyMap. The key is a BR rather than byte[].
798    // It will be easy to do gets over this new Map as we can create get keys over the Cell cf by
799    // new SimpleByteRange(cell.familyArray, cell.familyOffset, cell.familyLen)
800    Map<ByteRange, List<Cell>> familyMap1 = new HashMap<>();
801    for (Entry<byte[], ? extends Collection<?>> entry : familyMap.entrySet()) {
802      if (entry.getValue() instanceof List) {
803        familyMap1.put(new SimpleMutableByteRange(entry.getKey()), (List<Cell>) entry.getValue());
804      }
805    }
806    RegionScanner scanner = getRegion(e).getScanner(new Scan(get));
807    List<Cell> cells = Lists.newArrayList();
808    Cell prevCell = null;
809    ByteRange curFam = new SimpleMutableByteRange();
810    boolean curColAllVersions = (request == OpType.DELETE);
811    long curColCheckTs = opTs;
812    boolean foundColumn = false;
813    try {
814      boolean more = false;
815      ScannerContext scannerContext = ScannerContext.newBuilder().setBatchLimit(1).build();
816
817      do {
818        cells.clear();
819        // scan with limit as 1 to hold down memory use on wide rows
820        more = scanner.next(cells, scannerContext);
821        for (Cell cell: cells) {
822          if (LOG.isTraceEnabled()) {
823            LOG.trace("Found cell " + cell);
824          }
825          boolean colChange = prevCell == null || !CellUtil.matchingColumn(prevCell, cell);
826          if (colChange) foundColumn = false;
827          prevCell = cell;
828          if (!curColAllVersions && foundColumn) {
829            continue;
830          }
831          if (colChange && considerCellTs) {
832            curFam.set(cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength());
833            List<Cell> cols = familyMap1.get(curFam);
834            for (Cell col : cols) {
835              // null/empty qualifier is used to denote a Family delete. The TS and delete type
836              // associated with this is applicable for all columns within the family. That is
837              // why the below (col.getQualifierLength() == 0) check.
838              if ((col.getQualifierLength() == 0 && request == OpType.DELETE)
839                  || CellUtil.matchingQualifier(cell, col)) {
840                byte type = col.getTypeByte();
841                if (considerCellTs) {
842                  curColCheckTs = col.getTimestamp();
843                }
844                // For a Delete op we pass allVersions as true. When a Delete Mutation contains
845                // a version delete for a column no need to check all the covering cells within
846                // that column. Check all versions when Type is DeleteColumn or DeleteFamily
847                // One version delete types are Delete/DeleteFamilyVersion
848                curColAllVersions = (KeyValue.Type.DeleteColumn.getCode() == type)
849                    || (KeyValue.Type.DeleteFamily.getCode() == type);
850                break;
851              }
852            }
853          }
854          if (cell.getTimestamp() > curColCheckTs) {
855            // Just ignore this cell. This is not a covering cell.
856            continue;
857          }
858          foundColumn = true;
859          for (Action action: actions) {
860            // Are there permissions for this user for the cell?
861            if (!authManager.authorize(user, getTableName(e), cell, action)) {
862              // We can stop if the cell ACL denies access
863              return false;
864            }
865          }
866          cellGrants++;
867        }
868      } while (more);
869    } catch (AccessDeniedException ex) {
870      throw ex;
871    } catch (IOException ex) {
872      LOG.error("Exception while getting cells to calculate covering permission", ex);
873    } finally {
874      scanner.close();
875    }
876    // We should not authorize unless we have found one or more cell ACLs that
877    // grant access. This code is used to check for additional permissions
878    // after no table or CF grants are found.
879    return cellGrants > 0;
880  }

882  private static void addCellPermissions(final byte[] perms, Map<byte[], List<Cell>> familyMap) {
883    // Iterate over the entries in the familyMap, replacing the cells therein
884    // with new cells including the ACL data
885    for (Map.Entry<byte[], List<Cell>> e: familyMap.entrySet()) {
886      List<Cell> newCells = Lists.newArrayList();
887      for (Cell cell: e.getValue()) {
888        // Prepend the supplied perms in a new ACL tag to an update list of tags for the cell
889        List<Tag> tags = new ArrayList<>();
890        tags.add(new ArrayBackedTag(AccessControlLists.ACL_TAG_TYPE, perms));
891        Iterator<Tag> tagIterator = CellUtil.tagsIterator(cell);
892        while (tagIterator.hasNext()) {
893          tags.add(tagIterator.next());
894        }
895        newCells.add(CellUtil.createCell(cell, tags));
896      }
897      // This is supposed to be safe, won't CME
898      e.setValue(newCells);
899    }
900  }

902  // Checks whether incoming cells contain any tag with type as ACL_TAG_TYPE. This tag
903  // type is reserved and should not be explicitly set by user.
904  private void checkForReservedTagPresence(User user, Mutation m) throws IOException {
905    // No need to check if we're not going to throw
906    if (!authorizationEnabled) {
907      m.setAttribute(TAG_CHECK_PASSED, TRUE);
908      return;
909    }
910    // Superusers are allowed to store cells unconditionally.
911    if (Superusers.isSuperUser(user)) {
912      m.setAttribute(TAG_CHECK_PASSED, TRUE);
913      return;
914    }
915    // We already checked (prePut vs preBatchMutation)
916    if (m.getAttribute(TAG_CHECK_PASSED) != null) {
917      return;
918    }
919    for (CellScanner cellScanner = m.cellScanner(); cellScanner.advance();) {
920      Iterator<Tag> tagsItr = CellUtil.tagsIterator(cellScanner.current());
921      while (tagsItr.hasNext()) {
922        if (tagsItr.next().getType() == AccessControlLists.ACL_TAG_TYPE) {
923          throw new AccessDeniedException("Mutation contains cell with reserved type tag");
924        }
925      }
926    }
927    m.setAttribute(TAG_CHECK_PASSED, TRUE);
928  }

979        throw new RuntimeException("Error obtaining TableAuthManager", ioe);
980      }
981    } else {
982      throw new RuntimeException("Error obtaining TableAuthManager, zk found null.");
983    }
984

1041  public void postCompletedCreateTableAction(
1042      final ObserverContext<MasterCoprocessorEnvironment> c,
1043      final TableDescriptor desc,
1044      final RegionInfo[] regions) throws IOException {
1045    // When AC is used, it should be configured as the 1st CP.
1046    // In Master, the table operations like create, are handled by a Thread pool but the max size
1047    // for this pool is 1. So if multiple CPs create tables on startup, these creations will happen
1048    // sequentially only.
1049    // Related code in HMaster#startServiceThreads
1050    // {code}
1051    //   // We depend on there being only one instance of this executor running
1052    //   // at a time. To do concurrency, would need fencing of enable/disable of
1053    //   // tables.
1054    //   this.service.startExecutorService(ExecutorType.MASTER_TABLE_OPERATIONS, 1);
1055    // {code}
1056    // In future if we change this pool to have more threads, then there is a chance for thread,
1057    // creating acl table, getting delayed and by that time another table creation got over and
1058    // this hook is getting called. In such a case, we will need a wait logic here which will
1059    // wait till the acl table is created.
1060    if (AccessControlLists.isAclTable(desc)) {
1061      this.aclTabAvailable = true;
1062    } else if (!(TableName.NAMESPACE_TABLE_NAME.equals(desc.getTableName()))) {
1063      if (!aclTabAvailable) {
1064        LOG.warn("Not adding owner permission for table " + desc.getTableName() + ". "
1065            + AccessControlLists.ACL_TABLE_NAME + " is not yet created. "
1066            + getClass().getSimpleName() + " should be configured as the first Coprocessor");
1067      } else {
1068        String owner = desc.getOwnerString();
1069        // default the table owner to current user, if not specified.
1070        if (owner == null)
1071          owner = getActiveUser(c).getShortName();
1072        final UserPermission userperm = new UserPermission(Bytes.toBytes(owner),
1073            desc.getTableName(), null, Action.values());
1074        // switch to the real hbase master user for doing the RPC on the ACL table
1075        User.runAsLoginUser(new PrivilegedExceptionAction<Void>() {
1076          @Override
1077          public Void run() throws Exception {
1078            AccessControlLists.addUserPermission(c.getEnvironment().getConfiguration(),
1079                userperm, c.getEnvironment().getTable(AccessControlLists.ACL_TABLE_NAME));
1080            return null;
1081          }
1082        });
1083      }
1084    }

1582  private void internalPreRead(final ObserverContext<RegionCoprocessorEnvironment> c,
1583      final Query query, OpType opType) throws IOException {
1584    Filter filter = query.getFilter();
1585    // Don't wrap an AccessControlFilter
1586    if (filter != null && filter instanceof AccessControlFilter) {
1587      return;
1588    }
1589    User user = getActiveUser(c);
1590    RegionCoprocessorEnvironment env = c.getEnvironment();
1591    Map<byte[],? extends Collection<byte[]>> families = null;
1592    switch (opType) {
1593    case GET:
1594    case EXISTS:
1595      families = ((Get)query).getFamilyMap();
1596      break;
1597    case SCAN:
1598      families = ((Scan)query).getFamilyMap();
1599      break;
1600    default:
1601      throw new RuntimeException("Unhandled operation " + opType);
1602    }
1603    AuthResult authResult = permissionGranted(opType, user, env, families, Action.READ);
1604    Region region = getRegion(env);
1605    TableName table = getTableName(region);
1606    Map<ByteRange, Integer> cfVsMaxVersions = Maps.newHashMap();
1607    for (ColumnFamilyDescriptor hcd : region.getTableDescriptor().getColumnFamilies()) {
1608      cfVsMaxVersions.put(new SimpleMutableByteRange(hcd.getName()), hcd.getMaxVersions());
1609    }
1610    if (!authResult.isAllowed()) {
1611      if (!cellFeaturesEnabled || compatibleEarlyTermination) {
1612        // Old behavior: Scan with only qualifier checks if we have partial
1613        // permission. Backwards compatible behavior is to throw an
1614        // AccessDeniedException immediately if there are no grants for table
1615        // or CF or CF+qual. Only proceed with an injected filter if there are
1616        // grants for qualifiers. Otherwise we will fall through below and log
1617        // the result and throw an ADE. We may end up checking qualifier
1618        // grants three times (permissionGranted above, here, and in the
1619        // filter) but that's the price of backwards compatibility.
1620        if (hasFamilyQualifierPermission(user, Action.READ, env, families)) {
1621          authResult.setAllowed(true);
1622          authResult.setReason("Access allowed with filter");
1623          // Only wrap the filter if we are enforcing authorizations
1624          if (authorizationEnabled) {
1625            Filter ourFilter = new AccessControlFilter(authManager, user, table,
1626              AccessControlFilter.Strategy.CHECK_TABLE_AND_CF_ONLY,
1627              cfVsMaxVersions);
1628            // wrap any existing filter
1629            if (filter != null) {
1630              ourFilter = new FilterList(FilterList.Operator.MUST_PASS_ALL,
1631                Lists.newArrayList(ourFilter, filter));
1632            }
1633            switch (opType) {
1634              case GET:
1635              case EXISTS:
1636                ((Get)query).setFilter(ourFilter);
1637                break;
1638              case SCAN:
1639                ((Scan)query).setFilter(ourFilter);
1640                break;
1641              default:
1642                throw new RuntimeException("Unhandled operation " + opType);
1643            }
1644          }
1645        }
1646      } else {
1647        // New behavior: Any access we might be granted is more fine-grained
1648        // than whole table or CF. Simply inject a filter and return what is
1649        // allowed. We will not throw an AccessDeniedException. This is a
1650        // behavioral change since 0.96.
1651        authResult.setAllowed(true);
1652        authResult.setReason("Access allowed with filter");
1653        // Only wrap the filter if we are enforcing authorizations
1654        if (authorizationEnabled) {
1655          Filter ourFilter = new AccessControlFilter(authManager, user, table,
1656            AccessControlFilter.Strategy.CHECK_CELL_DEFAULT, cfVsMaxVersions);
1657          // wrap any existing filter
1658          if (filter != null) {
1659            ourFilter = new FilterList(FilterList.Operator.MUST_PASS_ALL,
1660              Lists.newArrayList(ourFilter, filter));
1661          }
1662          switch (opType) {
1663            case GET:
1664            case EXISTS:
1665              ((Get)query).setFilter(ourFilter);
1666              break;
1667            case SCAN:
1668              ((Scan)query).setFilter(ourFilter);
1669              break;
1670            default:
1671              throw new RuntimeException("Unhandled operation " + opType);
1672          }
1673        }
1674      }
1675    }
1676
1677    logResult(authResult);
1678    if (authorizationEnabled && !authResult.isAllowed()) {
1679      throw new AccessDeniedException("Insufficient permissions for user '"
1680          + (user != null ? user.getShortName() : "null")
1681          + "' (table=" + table + ", action=READ)");
1682    }
1683  }

1698  @Override
1699  public void prePut(final ObserverContext<RegionCoprocessorEnvironment> c,
1700      final Put put, final WALEdit edit, final Durability durability)
1701      throws IOException {
1702    User user = getActiveUser(c);
1703    checkForReservedTagPresence(user, put);
1704
1705    // Require WRITE permission to the table, CF, or top visible value, if any.
1706    // NOTE: We don't need to check the permissions for any earlier Puts
1707    // because we treat the ACLs in each Put as timestamped like any other
1708    // HBase value. A new ACL in a new Put applies to that Put. It doesn't
1709    // change the ACL of any previous Put. This allows simple evolution of
1710    // security policy over time without requiring expensive updates.
1711    RegionCoprocessorEnvironment env = c.getEnvironment();
1712    Map<byte[],? extends Collection<Cell>> families = put.getFamilyCellMap();
1713    AuthResult authResult = permissionGranted(OpType.PUT, user, env, families, Action.WRITE);
1714    logResult(authResult);
1715    if (!authResult.isAllowed()) {
1716      if (cellFeaturesEnabled && !compatibleEarlyTermination) {
1717        put.setAttribute(CHECK_COVERING_PERM, TRUE);
1718      } else if (authorizationEnabled) {
1719        throw new AccessDeniedException("Insufficient permissions " + authResult.toContextString());
1720      }
1721    }
1722
1723    // Add cell ACLs from the operation to the cells themselves
1724    byte[] bytes = put.getAttribute(AccessControlConstants.OP_ATTRIBUTE_ACL);
1725    if (bytes != null) {
1726      if (cellFeaturesEnabled) {
1727        addCellPermissions(bytes, put.getFamilyCellMap());
1728      } else {
1729        throw new DoNotRetryIOException("Cell ACLs cannot be persisted");
1730      }
1731    }
1732  }

1744      final Delete delete, final WALEdit edit, final Durability durability)

1770  @Override
1771  public void preBatchMutate(ObserverContext<RegionCoprocessorEnvironment> c,
1772      MiniBatchOperationInProgress<Mutation> miniBatchOp) throws IOException {
1773    if (cellFeaturesEnabled && !compatibleEarlyTermination) {
1774      TableName table = c.getEnvironment().getRegion().getRegionInfo().getTable();
1775      User user = getActiveUser(c);
1776      for (int i = 0; i < miniBatchOp.size(); i++) {
1777        Mutation m = miniBatchOp.getOperation(i);
1778        if (m.getAttribute(CHECK_COVERING_PERM) != null) {
1779          // We have a failure with table, cf and q perm checks and now giving a chance for cell
1780          // perm check
1781          OpType opType;
1782          if (m instanceof Put) {
1783            checkForReservedTagPresence(user, m);
1784            opType = OpType.PUT;
1785          } else {
1786            opType = OpType.DELETE;
1787          }
1788          AuthResult authResult = null;
1789          if (checkCoveringPermission(user, opType, c.getEnvironment(), m.getRow(),
1790            m.getFamilyCellMap(), m.getTimeStamp(), Action.WRITE)) {
1791            authResult = AuthResult.allow(opType.toString(), "Covering cell set",
1792              user, Action.WRITE, table, m.getFamilyCellMap());
1793          } else {
1794            authResult = AuthResult.deny(opType.toString(), "Covering cell set",
1795              user, Action.WRITE, table, m.getFamilyCellMap());
1796          }
1797          logResult(authResult);
1798          if (authorizationEnabled && !authResult.isAllowed()) {
1799            throw new AccessDeniedException("Insufficient permissions "
1800              + authResult.toContextString());
1801          }
1802        }
1803      }
1804    }

2058  @Override
2059  public Cell postMutationBeforeWAL(ObserverContext<RegionCoprocessorEnvironment> ctx,
2060      MutationType opType, Mutation mutation, Cell oldCell, Cell newCell) throws IOException {
2061    // If the HFile version is insufficient to persist tags, we won't have any
2062    // work to do here
2063    if (!cellFeaturesEnabled) {
2064      return newCell;
2065    }
2066
2067    // Collect any ACLs from the old cell
2068    List<Tag> tags = Lists.newArrayList();
2069    List<Tag> aclTags = Lists.newArrayList();
2070    ListMultimap<String,Permission> perms = ArrayListMultimap.create();
2071    if (oldCell != null) {
2072      Iterator<Tag> tagIterator = CellUtil.tagsIterator(oldCell);
2073      while (tagIterator.hasNext()) {
2074        Tag tag = tagIterator.next();
2075        if (tag.getType() != AccessControlLists.ACL_TAG_TYPE) {
2076          // Not an ACL tag, just carry it through
2077          if (LOG.isTraceEnabled()) {
2078            LOG.trace("Carrying forward tag from " + oldCell + ": type " + tag.getType()
2079                + " length " + tag.getValueLength());
2080          }
2081          tags.add(tag);
2082        } else {
2083          aclTags.add(tag);
2084        }
2085      }
2086    }
2087
2088    // Do we have an ACL on the operation?
2089    byte[] aclBytes = mutation.getACL();
2090    if (aclBytes != null) {
2091      // Yes, use it
2092      tags.add(new ArrayBackedTag(AccessControlLists.ACL_TAG_TYPE, aclBytes));
2093    } else {
2094      // No, use what we carried forward
2095      if (perms != null) {
2096        // TODO: If we collected ACLs from more than one tag we may have a
2097        // List<Permission> of size > 1, this can be collapsed into a single
2098        // Permission
2099        if (LOG.isTraceEnabled()) {
2100          LOG.trace("Carrying forward ACLs from " + oldCell + ": " + perms);
2101        }
2102        tags.addAll(aclTags);
2103      }
2104    }
2105
2106    // If we have no tags to add, just return
2107    if (tags.isEmpty()) {
2108      return newCell;
2109    }
2110
2111    Cell rewriteCell = CellUtil.createCell(newCell, tags);
2112    return rewriteCell;
2113  }

2401      } else {
2402        throw new CoprocessorException(AccessController.class, "This method "
2403            + "can only execute at " + AccessControlLists.ACL_TABLE_NAME + " table.");
2404      }
2405    } catch (IOException ioe) {
2406      // pass exception back up
2407      CoprocessorRpcUtils.setControllerException(controller, ioe);
2408    }

2412  @Override
2413  public void checkPermissions(RpcController controller,
2414                               AccessControlProtos.CheckPermissionsRequest request,
2415                               RpcCallback<AccessControlProtos.CheckPermissionsResponse> done) {
2416    Permission[] permissions = new Permission[request.getPermissionCount()];
2417    for (int i=0; i < request.getPermissionCount(); i++) {
2418      permissions[i] = AccessControlUtil.toPermission(request.getPermission(i));
2419    }
2420    AccessControlProtos.CheckPermissionsResponse response = null;
2421    try {
2422      User user = RpcServer.getRequestUser();
2423      TableName tableName = regionEnv.getRegion().getTableDescriptor().getTableName();
2424      for (Permission permission : permissions) {
2425        if (permission instanceof TablePermission) {
2426          // Check table permissions
2427
2428          TablePermission tperm = (TablePermission) permission;
2429          for (Action action : permission.getActions()) {
2430            if (!tperm.getTableName().equals(tableName)) {
2431              throw new CoprocessorException(AccessController.class, String.format("This method "
2432                  + "can only execute at the table specified in TablePermission. " +
2433                  "Table of the region:%s , requested table:%s", tableName,
2434                  tperm.getTableName()));
2435            }
2436
2437            Map<byte[], Set<byte[]>> familyMap = new TreeMap<>(Bytes.BYTES_COMPARATOR);
2438            if (tperm.getFamily() != null) {
2439              if (tperm.getQualifier() != null) {
2440                Set<byte[]> qualifiers = Sets.newTreeSet(Bytes.BYTES_COMPARATOR);
2441                qualifiers.add(tperm.getQualifier());
2442                familyMap.put(tperm.getFamily(), qualifiers);
2443              } else {
2444                familyMap.put(tperm.getFamily(), null);
2445              }
2446            }
2447
2448            AuthResult result = permissionGranted("checkPermissions", user, action, regionEnv,
2449              familyMap);
2450            logResult(result);
2451            if (!result.isAllowed()) {
2452              // Even if passive we need to throw an exception here, we support checking
2453              // effective permissions, so throw unconditionally
2454              throw new AccessDeniedException("Insufficient permissions (table=" + tableName +
2455                (familyMap.size() > 0 ? ", family: " + result.toFamilyString() : "") +
2456                ", action=" + action.toString() + ")");
2457            }
2458          }
2459
2460        } else {
2461          // Check global permissions
2462
2463          for (Action action : permission.getActions()) {
2464            AuthResult result;
2465            if (authManager.authorize(user, action)) {
2466              result = AuthResult.allow("checkPermissions", "Global action allowed", user,
2467                action, null, null);
2468            } else {
2469              result = AuthResult.deny("checkPermissions", "Global action denied", user, action,
2470                null, null);
2471            }
2472            logResult(result);
2473            if (!result.isAllowed()) {
2474              // Even if passive we need to throw an exception here, we support checking
2475              // effective permissions, so throw unconditionally
2476              throw new AccessDeniedException("Insufficient permissions (action=" +
2477                action.toString() + ")");
2478            }
2479          }
2480        }
2481      }
2482      response = AccessControlProtos.CheckPermissionsResponse.getDefaultInstance();
2483    } catch (IOException ioe) {
2484      CoprocessorRpcUtils.setControllerException(controller, ioe);
2485    }
2486    done.run(response);
2487  }

2566      List<TableName> tableNamesList, List<TableDescriptor> descriptors,

2765  private void checkLockPermissions(User user, String namespace,
2766      TableName tableName, RegionInfo[] regionInfos, String reason)
2767  throws IOException {
2768    if (namespace != null && !namespace.isEmpty()) {
2769      requireNamespacePermission(user, reason, namespace, Action.ADMIN, Action.CREATE);
2770    } else if (tableName != null || (regionInfos != null && regionInfos.length > 0)) {
2771      // So, either a table or regions op. If latter, check perms ons table.
2772      TableName tn = tableName != null? tableName: regionInfos[0].getTable();
2773      requireTablePermission(user, reason, tn, null, null,
2774          Action.ADMIN, Action.CREATE);
2775    } else {
2776      throw new DoNotRetryIOException("Invalid lock level when requesting permissions.");
2777    }
2778  }
