021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.Collections;
024import java.util.List;
025
026import org.apache.hadoop.fs.FileStatus;
027import org.apache.hadoop.fs.Path;

033/**
034 * An implementation of {@link MobCompactionRequest} that is used in
035 * {@link PartitionedMobCompactor}.
036 * The mob files that have the same start key and date in their names belong to
037 * the same partition.
038 */
039@InterfaceAudience.Private
040public class PartitionedMobCompactionRequest extends MobCompactionRequest {
041
042  protected List<CompactionDelPartition> delPartitions;
043  protected Collection<CompactionPartition> compactionPartitions;
044
045  public PartitionedMobCompactionRequest(Collection<CompactionPartition> compactionPartitions,
046    List<CompactionDelPartition> delPartitions) {
047    this.selectionTime = EnvironmentEdgeManager.currentTime();
048    this.compactionPartitions = compactionPartitions;
049    this.delPartitions = delPartitions;
050  }
051
052  /**
053   * Gets the compaction partitions.
054   * @return The compaction partitions.
055   */
056  public Collection<CompactionPartition> getCompactionPartitions() {
057    return this.compactionPartitions;
058  }
059
060  /**
061   * Gets the del files.
062   * @return The del files.
063   */
064  public List<CompactionDelPartition> getDelPartitions() {
065    return this.delPartitions;
066  }
067
068  /**
069   * The partition in the mob compaction.
070   * The mob files that have the same start key and date in their names belong to
071   * the same partition.
072   */
073  protected static class CompactionPartition {
074    private List<FileStatus> files = new ArrayList<>();
075    private CompactionPartitionId partitionId;
076
077    // The startKey and endKey of this partition, both are inclusive.
078    private byte[] startKey;
079    private byte[] endKey;
080
081    public CompactionPartition(CompactionPartitionId partitionId) {
082      this.partitionId = partitionId;
083    }
084
085    public CompactionPartitionId getPartitionId() {
086      return this.partitionId;
087    }
088
089    public void addFile(FileStatus file) {
090      files.add(file);
091    }
092
093    public List<FileStatus> listFiles() {
094      return Collections.unmodifiableList(files);
095    }
096
097    public int getFileCount () {
098      return files.size();
099    }
100
101    public byte[] getStartKey() {
102      return startKey;
103    }
104
105    /**
106     * Set start key of this partition, only if the input startKey is less than
107     * the current start key.
108     */
109    public void setStartKey(final byte[] startKey)
110    {
111      if ((this.startKey == null) || (Bytes.compareTo(startKey, this.startKey) < 0)) {
112        this.startKey = startKey;
113      }
114    }
115
116    public byte[] getEndKey() {
117      return endKey;
118    }
119
120    /**
121     * Set end key of this partition, only if the input endKey is greater than
122     * the current end key.
123     */
124    public void setEndKey(final byte[] endKey) {
125      if ((this.endKey == null) || (Bytes.compareTo(endKey, this.endKey) > 0)) {
126        this.endKey = endKey;
127      }
128    }
129  }
130
131  /**
132   * The partition id that consists of start key and date of the mob file name.
133   */
134  public static class CompactionPartitionId {
135    private String startKey;
136    private String date;
137    private String latestDate;
138    private long threshold;
139
140    public CompactionPartitionId() {
141      // initialize these fields to empty string
142      this.startKey = "";
143      this.date = "";
144      this.latestDate = "";
145      this.threshold = 0;
146    }
147
148    public CompactionPartitionId(String startKey, String date) {
149      if (startKey == null || date == null) {
150        throw new IllegalArgumentException("Neither of start key and date could be null");
151      }
152      this.startKey = startKey;
153      this.date = date;
154      this.latestDate = "";
155      this.threshold = 0;
156    }
157
158    public void setThreshold (final long threshold) {
159      this.threshold = threshold;
160    }
161
162    public long getThreshold () {
163      return this.threshold;
164    }
165
166    public String getStartKey() {
167      return this.startKey;
168    }
169
170    public void setStartKey(final String startKey) {
171      this.startKey = startKey;
172    }
173
174    public String getDate() {
175      return this.date;
176    }
177
178    public void setDate(final String date) {
179      this.date = date;
180    }
181
182    public String getLatestDate () { return this.latestDate; }
183
184    public void updateLatestDate(final String latestDate) {
185      if (this.latestDate.compareTo(latestDate) < 0) {
186        this.latestDate = latestDate;
187      }
188    }
189
190    @Override
191    public int hashCode() {
192      int result = 17;
193      result = 31 * result + startKey.hashCode();
194      result = 31 * result + date.hashCode();
195      return result;
196    }
197
198    @Override
199    public boolean equals(Object obj) {
200      if (this == obj) {
201        return true;
202      }
203      if (!(obj instanceof CompactionPartitionId)) {
204        return false;
205      }
206      CompactionPartitionId another = (CompactionPartitionId) obj;
207      if (!this.startKey.equals(another.startKey)) {
208        return false;
209      }
210      if (!this.date.equals(another.date)) {
211        return false;
212      }
213      return true;
214    }
215
216    @Override
217    public String toString() {
218      return new StringBuilder(startKey).append(date).toString();
219    }
220  }
221
222  /**
223   * The delete file partition in the mob compaction.
224   * The delete partition is defined as [startKey, endKey] pair.
225   * The mob delete files that have the same start key and end key belong to
226   * the same partition.
227   */
228  protected static class CompactionDelPartition {
229    private List<Path> delFiles = new ArrayList<Path>();
230    private List<StoreFile> storeFiles = new ArrayList<>();
231    private CompactionDelPartitionId id;
232
233    public CompactionDelPartition(CompactionDelPartitionId id) {
234      this.id = id;
235    }
236
237    public CompactionDelPartitionId getId() {
238      return this.id;
239    }
240
241    void addDelFile(FileStatus file) {
242      delFiles.add(file.getPath());
243    }
244    public void addStoreFile(final StoreFile file) {
245      storeFiles.add(file);
246    }
247
248    public List<StoreFile> getStoreFiles() {
249      return storeFiles;
250    }
251
252    List<Path> listDelFiles() {
253      return Collections.unmodifiableList(delFiles);
254    }
255
256    void addDelFileList(final Collection<Path> list) {
257      delFiles.addAll(list);
258    }
259
260    int getDelFileCount () {
261      return delFiles.size();
262    }
263
264    void cleanDelFiles() {
265      delFiles.clear();
266    }
267  }
268
269  /**
270   * The delete partition id that consists of start key and end key
271   */
272  public static class CompactionDelPartitionId implements Comparable<CompactionDelPartitionId> {
273    private byte[] startKey;
274    private byte[] endKey;
275
276    public CompactionDelPartitionId() {
277    }
278
279    public CompactionDelPartitionId(final byte[] startKey, final byte[] endKey) {
280      this.startKey = startKey;
281      this.endKey = endKey;
282    }
283
284    public byte[] getStartKey() {
285      return this.startKey;
286    }
287    public void setStartKey(final byte[] startKey) {
288      this.startKey = startKey;
289    }
290
291    public byte[] getEndKey() {
292      return this.endKey;
293    }
294    public void setEndKey(final byte[] endKey) {
295      this.endKey = endKey;
296    }
297
298    public int compareTo(CompactionDelPartitionId o) {
299      /*
300       * 1). Compare the start key, if the k1 < k2, then k1 is less
301       * 2). If start Key is same, check endKey, k1 < k2, k1 is less
302       *     If both are same, then they are equal.
303       */
304      int result = Bytes.compareTo(this.startKey, o.getStartKey());
305      if (result != 0) {
306        return result;
307      }
308
309      return Bytes.compareTo(this.endKey, o.getEndKey());
310    }
311
312    @Override
313    public int hashCode() {
314      int result = 17;
315      result = 31 * result + java.util.Arrays.hashCode(startKey);
316      result = 31 * result + java.util.Arrays.hashCode(endKey);
317      return result;
318    }
319
320    @Override
321    public boolean equals(Object obj) {
322      if (this == obj) {
323        return true;
324      }
325      if (!(obj instanceof CompactionDelPartitionId)) {
326        return false;
327      }
328      CompactionDelPartitionId another = (CompactionDelPartitionId) obj;
329
330      return (this.compareTo(another) == 0);
331    }
332  }
