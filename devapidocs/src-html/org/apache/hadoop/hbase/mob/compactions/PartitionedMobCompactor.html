136  @Override
137  public List<Path> compact(List<FileStatus> files, boolean allFiles) throws IOException {
138    if (files == null || files.isEmpty()) {
139      LOG.info("No candidate mob files");
140      return null;
141    }
142    LOG.info("is allFiles: " + allFiles);
143
144    // find the files to compact.
145    PartitionedMobCompactionRequest request = select(files, allFiles);
146    // compact the files.
147    return performCompaction(request);
148  }

150  /**
151   * Selects the compacted mob/del files.
152   * Iterates the candidates to find out all the del files and small mob files.
153   * @param candidates All the candidates.
154   * @param allFiles Whether add all mob files into the compaction.
155   * @return A compaction request.
156   * @throws IOException if IO failure is encountered
157   */
158  protected PartitionedMobCompactionRequest select(List<FileStatus> candidates,
159    boolean allFiles) throws IOException {
160    final Map<CompactionPartitionId, CompactionPartition> filesToCompact = new HashMap<>();
161    final CompactionPartitionId id = new CompactionPartitionId();
162    final NavigableMap<CompactionDelPartitionId, CompactionDelPartition> delFilesToCompact = new TreeMap<>();
163    final CompactionDelPartitionId delId = new CompactionDelPartitionId();
164    final ArrayList<CompactionDelPartition> allDelPartitions = new ArrayList<>();
165    int selectedFileCount = 0;
166    int irrelevantFileCount = 0;
167    int totalDelFiles = 0;
168    MobCompactPartitionPolicy policy = column.getMobCompactPartitionPolicy();
169
170    Calendar calendar =  Calendar.getInstance();
171    Date currentDate = new Date();
172    Date firstDayOfCurrentMonth = null;
173    Date firstDayOfCurrentWeek = null;
174
175    if (policy == MobCompactPartitionPolicy.MONTHLY) {
176      firstDayOfCurrentMonth = MobUtils.getFirstDayOfMonth(calendar, currentDate);
177      firstDayOfCurrentWeek = MobUtils.getFirstDayOfWeek(calendar, currentDate);
178    } else if (policy == MobCompactPartitionPolicy.WEEKLY) {
179      firstDayOfCurrentWeek = MobUtils.getFirstDayOfWeek(calendar, currentDate);
180    }
181
182    // We check if there is any del files so the logic can be optimized for the following processing
183    // First step is to check if there is any delete files. If there is any delete files,
184    // For each Partition, it needs to read its startKey and endKey from files.
185    // If there is no delete file, there is no need to read startKey and endKey from files, this
186    // is an optimization.
187    boolean withDelFiles = false;
188    for (FileStatus file : candidates) {
189      if (!file.isFile()) {
190        continue;
191      }
192      // group the del files and small files.
193      FileStatus linkedFile = file;
194      if (HFileLink.isHFileLink(file.getPath())) {
195        HFileLink link = HFileLink.buildFromHFileLinkPattern(conf, file.getPath());
196        linkedFile = getLinkedFileStatus(link);
197        if (linkedFile == null) {
198          continue;
199        }
200      }
201      if (StoreFileInfo.isDelFile(linkedFile.getPath())) {
202        withDelFiles = true;
203        break;
204      }
205    }
206
207    for (FileStatus file : candidates) {
208      if (!file.isFile()) {
209        irrelevantFileCount++;
210        continue;
211      }
212      // group the del files and small files.
213      FileStatus linkedFile = file;
214      if (HFileLink.isHFileLink(file.getPath())) {
215        HFileLink link = HFileLink.buildFromHFileLinkPattern(conf, file.getPath());
216        linkedFile = getLinkedFileStatus(link);
217        if (linkedFile == null) {
218          // If the linked file cannot be found, regard it as an irrelevantFileCount file
219          irrelevantFileCount++;
220          continue;
221        }
222      }
223      if (withDelFiles && StoreFileInfo.isDelFile(linkedFile.getPath())) {
224        // File in the Del Partition List
225
226        // Get delId from the file
227        try (Reader reader = HFile.createReader(fs, linkedFile.getPath(), conf)) {
228          delId.setStartKey(reader.getFirstRowKey());
229          delId.setEndKey(reader.getLastRowKey());
230        }
231        CompactionDelPartition delPartition = delFilesToCompact.get(delId);
232        if (delPartition == null) {
233          CompactionDelPartitionId newDelId =
234              new CompactionDelPartitionId(delId.getStartKey(), delId.getEndKey());
235          delPartition = new CompactionDelPartition(newDelId);
236          delFilesToCompact.put(newDelId, delPartition);
237        }
238        delPartition.addDelFile(file);
239        totalDelFiles ++;
240      } else {
241        String fileName = linkedFile.getPath().getName();
242        String date = MobFileName.getDateFromName(fileName);
243        boolean skipCompaction = MobUtils
244            .fillPartitionId(id, firstDayOfCurrentMonth, firstDayOfCurrentWeek, date, policy,
245                calendar, mergeableSize);
246        if (allFiles || (!skipCompaction && (linkedFile.getLen() < id.getThreshold()))) {
247          // add all files if allFiles is true,
248          // otherwise add the small files to the merge pool
249          // filter out files which are not supposed to be compacted with the
250          // current policy
251
252          id.setStartKey(MobFileName.getStartKeyFromName(fileName));
253          CompactionPartition compactionPartition = filesToCompact.get(id);
254          if (compactionPartition == null) {
255            CompactionPartitionId newId = new CompactionPartitionId(id.getStartKey(), id.getDate());
256            compactionPartition = new CompactionPartition(newId);
257            compactionPartition.addFile(file);
258            filesToCompact.put(newId, compactionPartition);
259            newId.updateLatestDate(date);
260          } else {
261            compactionPartition.addFile(file);
262            compactionPartition.getPartitionId().updateLatestDate(date);
263          }
264
265          if (withDelFiles) {
266            // get startKey and endKey from the file and update partition
267            // TODO: is it possible to skip read of most hfiles?
268            try (Reader reader = HFile.createReader(fs, linkedFile.getPath(), conf)) {
269              compactionPartition.setStartKey(reader.getFirstRowKey());
270              compactionPartition.setEndKey(reader.getLastRowKey());
271            }
272          }
273
274          selectedFileCount++;
275        }
276      }
277    }
278
279    /*
280     * Merge del files so there are only non-overlapped del file lists
281     */
282    for(Map.Entry<CompactionDelPartitionId, CompactionDelPartition> entry : delFilesToCompact.entrySet()) {
283      if (allDelPartitions.size() > 0) {
284        // check if the current key range overlaps the previous one
285        CompactionDelPartition prev = allDelPartitions.get(allDelPartitions.size() - 1);
286        if (Bytes.compareTo(prev.getId().getEndKey(), entry.getKey().getStartKey()) >= 0) {
287          // merge them together
288          prev.getId().setEndKey(entry.getValue().getId().getEndKey());
289          prev.addDelFileList(entry.getValue().listDelFiles());
290
291        } else {
292          allDelPartitions.add(entry.getValue());
293        }
294      } else {
295        allDelPartitions.add(entry.getValue());
296      }
297    }
298
299    PartitionedMobCompactionRequest request = new PartitionedMobCompactionRequest(
300      filesToCompact.values(), allDelPartitions);
301    if (candidates.size() == (totalDelFiles + selectedFileCount + irrelevantFileCount)) {
302      // all the files are selected
303      request.setCompactionType(CompactionType.ALL_FILES);
304    }
305    LOG.info("The compaction type is " + request.getCompactionType() + ", the request has "
306      + totalDelFiles + " del files, " + selectedFileCount + " selected files, and "
307      + irrelevantFileCount + " irrelevant files");
308    return request;
309  }

311  /**
312   * Performs the compaction on the selected files.
313   * <ol>
314   * <li>Compacts the del files.</li>
315   * <li>Compacts the selected small mob files and all the del files.</li>
316   * <li>If all the candidates are selected, delete the del files.</li>
317   * </ol>
318   * @param request The compaction request.
319   * @return The paths of new mob files generated in the compaction.
320   * @throws IOException if IO failure is encountered
321   */
322  protected List<Path> performCompaction(PartitionedMobCompactionRequest request)
323    throws IOException {
324
325    // merge the del files, it is per del partition
326    for (CompactionDelPartition delPartition : request.getDelPartitions()) {
327      if (delPartition.getDelFileCount() <= 1) continue;
328      List<Path> newDelPaths = compactDelFiles(request, delPartition.listDelFiles());
329      delPartition.cleanDelFiles();
330      delPartition.addDelFileList(newDelPaths);
331    }
332
333    List<Path> paths = null;
334    int totalDelFileCount = 0;
335    try {
336      for (CompactionDelPartition delPartition : request.getDelPartitions()) {
337        for (Path newDelPath : delPartition.listDelFiles()) {
338          StoreFile sf =
339              new HStoreFile(fs, newDelPath, conf, compactionCacheConfig, BloomType.NONE, true);
340          // pre-create reader of a del file to avoid race condition when opening the reader in each
341          // partition.
342          sf.initReader();
343          delPartition.addStoreFile(sf);
344          totalDelFileCount++;
345        }
346      }
347      LOG.info("After merging, there are " + totalDelFileCount + " del files");
348      // compact the mob files by partitions.
349      paths = compactMobFiles(request);
350      LOG.info("After compaction, there are " + paths.size() + " mob files");
351    } finally {
352      for (CompactionDelPartition delPartition : request.getDelPartitions()) {
353        closeStoreFileReaders(delPartition.getStoreFiles());
354      }
355    }
356
357    // archive the del files if all the mob files are selected.
358    if (request.type == CompactionType.ALL_FILES && !request.getDelPartitions().isEmpty()) {
359      LOG.info(
360          "After a mob compaction with all files selected, archiving the del files ");
361      for (CompactionDelPartition delPartition : request.getDelPartitions()) {
362        LOG.info(delPartition.listDelFiles());
363        try {
364          MobUtils.removeMobFiles(conf, fs, tableName, mobTableDir, column.getName(), delPartition.getStoreFiles());
365        } catch (IOException e) {
366          LOG.error("Failed to archive the del files " + delPartition.getStoreFiles(), e);
367        }
368      }
369    }
370    return paths;
371  }

373  static class DelPartitionComparator implements Comparator<CompactionDelPartition> {
374    private boolean compareStartKey;
375
376    DelPartitionComparator(boolean compareStartKey) {
377      this.compareStartKey = compareStartKey;
378    }
379
380    public boolean getCompareStartKey() {
381      return this.compareStartKey;
382    }
383
384    public void setCompareStartKey(final boolean compareStartKey) {
385      this.compareStartKey = compareStartKey;
386    }
387
388    @Override
389    public int compare(CompactionDelPartition o1, CompactionDelPartition o2) {
390
391      if (compareStartKey) {
392        return Bytes.compareTo(o1.getId().getStartKey(), o2.getId().getStartKey());
393      } else {
394        return Bytes.compareTo(o1.getId().getEndKey(), o2.getId().getEndKey());
395      }
396    }
397  }

399  @VisibleForTesting
400  List<StoreFile> getListOfDelFilesForPartition(final CompactionPartition partition,
401      final List<CompactionDelPartition> delPartitions) {
402    // Binary search for startKey and endKey
403
404    List<StoreFile> result = new ArrayList<>();
405
406    DelPartitionComparator comparator = new DelPartitionComparator(false);
407    CompactionDelPartitionId id = new CompactionDelPartitionId(null, partition.getStartKey());
408    CompactionDelPartition target = new CompactionDelPartition(id);
409    int start = Collections.binarySearch(delPartitions, target, comparator);
410
411    // Get the start index for partition
412    if (start < 0) {
413      // Calculate the insert point
414      start = (start + 1) * (-1);
415      if (start == delPartitions.size()) {
416        // no overlap
417        return result;
418      } else {
419        // Check another case which has no overlap
420        if (Bytes.compareTo(partition.getEndKey(), delPartitions.get(start).getId().getStartKey()) < 0) {
421          return result;
422        }
423      }
424    }
425
426    // Search for end index for the partition
427    comparator.setCompareStartKey(true);
428    id.setStartKey(partition.getEndKey());
429    int end = Collections.binarySearch(delPartitions, target, comparator);
430
431    if (end < 0) {
432      end = (end + 1) * (-1);
433      if (end == 0) {
434        return result;
435      } else {
436        --end;
437        if (Bytes.compareTo(partition.getStartKey(), delPartitions.get(end).getId().getEndKey()) > 0) {
438          return result;
439        }
440      }
441    }
442
443    for (int i = start; i <= end; ++i) {
444        result.addAll(delPartitions.get(i).getStoreFiles());
445    }
446
447    return result;
448  }

450  /**
451   * Compacts the selected small mob files and all the del files.
452   * @param request The compaction request.
453   * @return The paths of new mob files after compactions.
454   * @throws IOException if IO failure is encountered
455   */
456  protected List<Path> compactMobFiles(final PartitionedMobCompactionRequest request)
457      throws IOException {
458    Collection<CompactionPartition> partitions = request.compactionPartitions;
459    if (partitions == null || partitions.isEmpty()) {
460      LOG.info("No partitions of mob files");
461      return Collections.emptyList();
462    }
463    List<Path> paths = new ArrayList<>();
464    final Connection c = ConnectionFactory.createConnection(conf);
465    final Table table = c.getTable(tableName);
466
467    try {
468      Map<CompactionPartitionId, Future<List<Path>>> results = new HashMap<>();
469      // compact the mob files by partitions in parallel.
470      for (final CompactionPartition partition : partitions) {
471
472        // How to efficiently come up a list of delFiles for one partition?
473        // Search the delPartitions and collect all the delFiles for the partition
474        // One optimization can do is that if there is no del file, we do not need to
475        // come up with startKey/endKey.
476        List<StoreFile> delFiles = getListOfDelFilesForPartition(partition,
477            request.getDelPartitions());
478
479        results.put(partition.getPartitionId(), pool.submit(new Callable<List<Path>>() {
480          @Override
481          public List<Path> call() throws Exception {
482            LOG.info("Compacting mob files for partition " + partition.getPartitionId());
483            return compactMobFilePartition(request, partition, delFiles, c, table);
484          }
485        }));
486      }
487      // compact the partitions in parallel.
488      List<CompactionPartitionId> failedPartitions = new ArrayList<>();
489      for (Entry<CompactionPartitionId, Future<List<Path>>> result : results.entrySet()) {
490        try {
491          paths.addAll(result.getValue().get());
492        } catch (Exception e) {
493          // just log the error
494          LOG.error("Failed to compact the partition " + result.getKey(), e);
495          failedPartitions.add(result.getKey());
496        }
497      }
498      if (!failedPartitions.isEmpty()) {
499        // if any partition fails in the compaction, directly throw an exception.
500        throw new IOException("Failed to compact the partitions " + failedPartitions);
501      }
502    } finally {
503      try {
504        table.close();
505      } catch (IOException e) {
506        LOG.error("Failed to close the Table", e);
507      }
508    }
509    return paths;
510  }

512  /**
513   * Compacts a partition of selected small mob files and all the del files.
514   * @param request The compaction request.
515   * @param partition A compaction partition.
516   * @param delFiles The del files.
517   * @param connection The connection to use.
518   * @param table The current table.
519   * @return The paths of new mob files after compactions.
520   * @throws IOException if IO failure is encountered
521   */
522  private List<Path> compactMobFilePartition(PartitionedMobCompactionRequest request,
523                                             CompactionPartition partition,
524                                             List<StoreFile> delFiles,
525                                             Connection connection,
526                                             Table table) throws IOException {
527    if (MobUtils.isMobFileExpired(column, EnvironmentEdgeManager.currentTime(),
528      partition.getPartitionId().getDate())) {
529      // If the files in the partition are expired, do not compact them and directly
530      // return an empty list.
531      return Collections.emptyList();
532    }
533    List<Path> newFiles = new ArrayList<>();
534    List<FileStatus> files = partition.listFiles();
535    int offset = 0;
536    Path bulkloadPathOfPartition = new Path(bulkloadPath, partition.getPartitionId().toString());
537    Path bulkloadColumnPath = new Path(bulkloadPathOfPartition, column.getNameAsString());
538    while (offset < files.size()) {
539      int batch = compactionBatchSize;
540      if (files.size() - offset < compactionBatchSize) {
541        batch = files.size() - offset;
542      }
543      if (batch == 1 && delFiles.isEmpty()) {
544        // only one file left and no del files, do not compact it,
545        // and directly add it to the new files.
546        newFiles.add(files.get(offset).getPath());
547        offset++;
548        continue;
549      }
550      // clean the bulkload directory to avoid loading old files.
551      fs.delete(bulkloadPathOfPartition, true);
552      // add the selected mob files and del files into filesToCompact
553      List<StoreFile> filesToCompact = new ArrayList<>();
554      for (int i = offset; i < batch + offset; i++) {
555        StoreFile sf = new HStoreFile(fs, files.get(i).getPath(), conf, compactionCacheConfig,
556            BloomType.NONE, true);
557        filesToCompact.add(sf);
558      }
559      filesToCompact.addAll(delFiles);
560      // compact the mob files in a batch.
561      compactMobFilesInBatch(request, partition, connection, table, filesToCompact, batch,
562        bulkloadPathOfPartition, bulkloadColumnPath, newFiles);
563      // move to the next batch.
564      offset += batch;
565    }
566    LOG.info("Compaction is finished. The number of mob files is changed from " + files.size()
567      + " to " + newFiles.size());
568    return newFiles;
569  }

571  /**
572   * Closes the readers of store files.
573   * @param storeFiles The store files to be closed.
574   */
575  private void closeStoreFileReaders(List<StoreFile> storeFiles) {
576    for (StoreFile storeFile : storeFiles) {
577      try {
578        storeFile.closeReader(true);
579      } catch (IOException e) {
580        LOG.warn("Failed to close the reader on store file " + storeFile.getPath(), e);
581      }
582    }
583  }

585  /**
586   * Compacts a partition of selected small mob files and all the del files in a batch.
587   * @param request The compaction request.
588   * @param partition A compaction partition.
589   * @param connection To use for transport
590   * @param table The current table.
591   * @param filesToCompact The files to be compacted.
592   * @param batch The number of mob files to be compacted in a batch.
593   * @param bulkloadPathOfPartition The directory where the bulkload column of the current
594   *   partition is saved.
595   * @param bulkloadColumnPath The directory where the bulkload files of current partition
596   *   are saved.
597   * @param newFiles The paths of new mob files after compactions.
598   * @throws IOException if IO failure is encountered
599   */
600  private void compactMobFilesInBatch(PartitionedMobCompactionRequest request,
601                                      CompactionPartition partition,
602                                      Connection connection, Table table,
603                                      List<StoreFile> filesToCompact, int batch,
604                                      Path bulkloadPathOfPartition, Path bulkloadColumnPath,
605                                      List<Path> newFiles)
606      throws IOException {
607    // open scanner to the selected mob files and del files.
608    StoreScanner scanner = createScanner(filesToCompact, ScanType.COMPACT_DROP_DELETES);
609    // the mob files to be compacted, not include the del files.
610    List<StoreFile> mobFilesToCompact = filesToCompact.subList(0, batch);
611    // Pair(maxSeqId, cellsCount)
612    Pair<Long, Long> fileInfo = getFileInfo(mobFilesToCompact);
613    // open writers for the mob files and new ref store files.
614    StoreFileWriter writer = null;
615    StoreFileWriter refFileWriter = null;
616    Path filePath = null;
617    long mobCells = 0;
618    boolean cleanupTmpMobFile = false;
619    boolean cleanupBulkloadDirOfPartition = false;
620    boolean cleanupCommittedMobFile = false;
621    boolean closeReaders= true;
622
623    try {
624      try {
625        writer = MobUtils
626            .createWriter(conf, fs, column, partition.getPartitionId().getLatestDate(), tempPath,
627                Long.MAX_VALUE, column.getCompactionCompressionType(),
628                partition.getPartitionId().getStartKey(), compactionCacheConfig, cryptoContext,
629                true);
630        cleanupTmpMobFile = true;
631        filePath = writer.getPath();
632        byte[] fileName = Bytes.toBytes(filePath.getName());
633        // create a temp file and open a writer for it in the bulkloadPath
634        refFileWriter = MobUtils.createRefFileWriter(conf, fs, column, bulkloadColumnPath,
635            fileInfo.getSecond().longValue(), compactionCacheConfig, cryptoContext, true);
636        cleanupBulkloadDirOfPartition = true;
637        List<Cell> cells = new ArrayList<>();
638        boolean hasMore;
639        ScannerContext scannerContext =
640            ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
641        do {
642          hasMore = scanner.next(cells, scannerContext);
643          for (Cell cell : cells) {
644            // write the mob cell to the mob file.
645            writer.append(cell);
646            // write the new reference cell to the store file.
647            Cell reference = MobUtils.createMobRefCell(cell, fileName, this.refCellTags);
648            refFileWriter.append(reference);
649            mobCells++;
650          }
651          cells.clear();
652        } while (hasMore);
653      } finally {
654        // close the scanner.
655        scanner.close();
656
657        if (cleanupTmpMobFile) {
658          // append metadata to the mob file, and close the mob file writer.
659          closeMobFileWriter(writer, fileInfo.getFirst(), mobCells);
660        }
661
662        if (cleanupBulkloadDirOfPartition) {
663          // append metadata and bulkload info to the ref mob file, and close the writer.
664          closeRefFileWriter(refFileWriter, fileInfo.getFirst(), request.selectionTime);
665        }
666      }
667
668      if (mobCells > 0) {
669        // commit mob file
670        MobUtils.commitFile(conf, fs, filePath, mobFamilyDir, compactionCacheConfig);
671        cleanupTmpMobFile = false;
672        cleanupCommittedMobFile = true;
673        // bulkload the ref file
674        bulkloadRefFile(connection, table, bulkloadPathOfPartition, filePath.getName());
675        cleanupCommittedMobFile = false;
676        newFiles.add(new Path(mobFamilyDir, filePath.getName()));
677      }
678
679      // archive the old mob files, do not archive the del files.
680      try {
681        closeStoreFileReaders(mobFilesToCompact);
682        closeReaders = false;
683        MobUtils.removeMobFiles(conf, fs, tableName, mobTableDir, column.getName(), mobFilesToCompact);
684      } catch (IOException e) {
685        LOG.error("Failed to archive the files " + mobFilesToCompact, e);
686      }
687    } finally {
688      if (closeReaders) {
689        closeStoreFileReaders(mobFilesToCompact);
690      }
691
692      if (cleanupTmpMobFile) {
693        deletePath(filePath);
694      }
695
696      if (cleanupBulkloadDirOfPartition) {
697        // delete the bulkload files in bulkloadPath
698        deletePath(bulkloadPathOfPartition);
699      }
700
701      if (cleanupCommittedMobFile) {
702        deletePath(new Path(mobFamilyDir, filePath.getName()));
703      }
704    }
705  }

707  /**
708   * Compacts the del files in batches which avoids opening too many files.
709   * @param request The compaction request.
710   * @param delFilePaths Del file paths to compact
711   * @return The paths of new del files after merging or the original files if no merging
712   *         is necessary.
713   * @throws IOException if IO failure is encountered
714   */
715  protected List<Path> compactDelFiles(PartitionedMobCompactionRequest request,
716    List<Path> delFilePaths) throws IOException {
717    if (delFilePaths.size() <= delFileMaxCount) {
718      return delFilePaths;
719    }
720    // when there are more del files than the number that is allowed, merge it firstly.
721    int offset = 0;
722    List<Path> paths = new ArrayList<>();
723    while (offset < delFilePaths.size()) {
724      // get the batch
725      int batch = compactionBatchSize;
726      if (delFilePaths.size() - offset < compactionBatchSize) {
727        batch = delFilePaths.size() - offset;
728      }
729      List<StoreFile> batchedDelFiles = new ArrayList<>();
730      if (batch == 1) {
731        // only one file left, do not compact it, directly add it to the new files.
732        paths.add(delFilePaths.get(offset));
733        offset++;
734        continue;
735      }
736      for (int i = offset; i < batch + offset; i++) {
737        batchedDelFiles.add(new HStoreFile(fs, delFilePaths.get(i), conf, compactionCacheConfig,
738          BloomType.NONE, true));
739      }
740      // compact the del files in a batch.
741      paths.add(compactDelFilesInBatch(request, batchedDelFiles));
742      // move to the next batch.
743      offset += batch;
744    }
745    return compactDelFiles(request, paths);
746  }

748  /**
749   * Compacts the del file in a batch.
750   * @param request The compaction request.
751   * @param delFiles The del files.
752   * @return The path of new del file after merging.
753   * @throws IOException if IO failure is encountered
754   */
755  private Path compactDelFilesInBatch(PartitionedMobCompactionRequest request,
756    List<StoreFile> delFiles) throws IOException {
757    // create a scanner for the del files.
758    StoreScanner scanner = createScanner(delFiles, ScanType.COMPACT_RETAIN_DELETES);
759    StoreFileWriter writer = null;
760    Path filePath = null;
761    try {
762      writer = MobUtils.createDelFileWriter(conf, fs, column,
763        MobUtils.formatDate(new Date(request.selectionTime)), tempPath, Long.MAX_VALUE,
764        column.getCompactionCompressionType(), HConstants.EMPTY_START_ROW, compactionCacheConfig,
765          cryptoContext);
766      filePath = writer.getPath();
767      List<Cell> cells = new ArrayList<>();
768      boolean hasMore;
769      ScannerContext scannerContext =
770              ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
771      do {
772        hasMore = scanner.next(cells, scannerContext);
773        for (Cell cell : cells) {
774          writer.append(cell);
775        }
776        cells.clear();
777      } while (hasMore);
778    } finally {
779      scanner.close();
780      if (writer != null) {
781        try {
782          writer.close();
783        } catch (IOException e) {
784          LOG.error("Failed to close the writer of the file " + filePath, e);
785        }
786      }
787    }
788    // commit the new del file
789    Path path = MobUtils.commitFile(conf, fs, filePath, mobFamilyDir, compactionCacheConfig);
790    // archive the old del files
791    try {
792      MobUtils.removeMobFiles(conf, fs, tableName, mobTableDir, column.getName(), delFiles);
793    } catch (IOException e) {
794      LOG.error("Failed to archive the old del files " + delFiles, e);
795    }
796    return path;
797  }

799  /**
800   * Creates a store scanner.
801   * @param filesToCompact The files to be compacted.
802   * @param scanType The scan type.
803   * @return The store scanner.
804   * @throws IOException if IO failure is encountered
805   */
806  private StoreScanner createScanner(List<StoreFile> filesToCompact, ScanType scanType)
807      throws IOException {
808    List<StoreFileScanner> scanners = StoreFileScanner.getScannersForStoreFiles(filesToCompact,
809      false, true, false, false, HConstants.LATEST_TIMESTAMP);
810    long ttl = HStore.determineTTLFromFamily(column);
811    ScanInfo scanInfo = new ScanInfo(conf, column, ttl, 0, CellComparator.COMPARATOR);
812    return new StoreScanner(scanInfo, scanType, scanners);
813  }

837  /**
838   * Closes the mob file writer.
839   * @param writer The mob file writer.
840   * @param maxSeqId Maximum sequence id.
841   * @param mobCellsCount The number of mob cells.
842   * @throws IOException if IO failure is encountered
843   */
844  private void closeMobFileWriter(StoreFileWriter writer, long maxSeqId, long mobCellsCount)
845    throws IOException {
846    if (writer != null) {
847      writer.appendMetadata(maxSeqId, false, mobCellsCount);
848      try {
849        writer.close();
850      } catch (IOException e) {
851        LOG.error("Failed to close the writer of the file " + writer.getPath(), e);
852      }
853    }
854  }

856  /**
857   * Closes the ref file writer.
858   * @param writer The ref file writer.
859   * @param maxSeqId Maximum sequence id.
860   * @param bulkloadTime The timestamp at which the bulk load file is created.
861   * @throws IOException if IO failure is encountered
862   */
863  private void closeRefFileWriter(StoreFileWriter writer, long maxSeqId, long bulkloadTime)
864    throws IOException {
865    if (writer != null) {
866      writer.appendMetadata(maxSeqId, false);
867      writer.appendFileInfo(StoreFile.BULKLOAD_TIME_KEY, Bytes.toBytes(bulkloadTime));
868      writer.appendFileInfo(StoreFile.SKIP_RESET_SEQ_ID, Bytes.toBytes(true));
869      try {
870        writer.close();
871      } catch (IOException e) {
872        LOG.error("Failed to close the writer of the ref file " + writer.getPath(), e);
873      }
874    }
875  }

877  /**
878   * Gets the max seqId and number of cells of the store files.
879   * @param storeFiles The store files.
880   * @return The pair of the max seqId and number of cells of the store files.
881   * @throws IOException if IO failure is encountered
882   */
883  private Pair<Long, Long> getFileInfo(List<StoreFile> storeFiles) throws IOException {
884    long maxSeqId = 0;
885    long maxKeyCount = 0;
886    for (StoreFile sf : storeFiles) {
887      // the readers will be closed later after the merge.
888      maxSeqId = Math.max(maxSeqId, sf.getMaxSequenceId());
889      sf.initReader();
890      byte[] count = sf.getReader().loadFileInfo().get(StoreFile.MOB_CELLS_COUNT);
891      if (count != null) {
892        maxKeyCount += Bytes.toLong(count);
893      }
894    }
895    return new Pair<>(maxSeqId, maxKeyCount);
896  }

912  private FileStatus getLinkedFileStatus(HFileLink link) throws IOException {
913    Path[] locations = link.getLocations();
914    FileStatus file;
915    for (Path location : locations) {
916
917      if (location != null) {
918        try {
919          file = fs.getFileStatus(location);
920          if (file != null) {
921            return file;
922          }
923        }  catch (FileNotFoundException e) {
924        }
925      }
926    }
927    LOG.warn("The file " + link + " links to can not be found");
928    return null;
929  }
