147   * @param cell to extract tag from

151    Optional<Tag> tag = getTableNameTag(cell);

155    }

159  /**
160   * Get the table name from when this cell was written into a mob hfile as a TableName.
161   * @param cell to extract tag from
162   * @return name of table as a TableName. empty if the tag is not found.
163   */
164  public static Optional<TableName> getTableName(Cell cell) {
165    Optional<Tag> maybe = getTableNameTag(cell);
166    Optional<TableName> name = Optional.empty();
167    if (maybe.isPresent()) {
168      final Tag tag = maybe.get();
169      if (tag.hasArray()) {
170        name = Optional
171          .of(TableName.valueOf(tag.getValueArray(), tag.getValueOffset(), tag.getValueLength()));
172      } else {
173        // TODO ByteBuffer handling in tags looks busted. revisit.
174        ByteBuffer buffer = tag.getValueByteBuffer().duplicate();
175        buffer.mark();
176        buffer.position(tag.getValueOffset());
177        buffer.limit(tag.getValueOffset() + tag.getValueLength());
178        name = Optional.of(TableName.valueOf(buffer));
179      }
180    }
181    return name;
182  }

259  /**
260   * Cleans the expired mob files. Cleans the files whose creation date is older than (current -
261   * columnFamily.ttl), and the minVersions of that column family is 0.
262   * @param fs               The current file system.
263   * @param conf             The current configuration.
264   * @param tableName        The current table name.
265   * @param columnDescriptor The descriptor of the current column family.
266   * @param cacheConfig      The cacheConfig that disables the block cache.
267   * @param current          The current time.
268   */
269  public static void cleanExpiredMobFiles(FileSystem fs, Configuration conf, TableName tableName,
270    ColumnFamilyDescriptor columnDescriptor, CacheConfig cacheConfig, long current)
271    throws IOException {
272    long timeToLive = columnDescriptor.getTimeToLive();
273    if (Integer.MAX_VALUE == timeToLive) {
274      // no need to clean, because the TTL is not set.
275      return;
276    }
277
278    Calendar calendar = Calendar.getInstance();
279    calendar.setTimeInMillis(current - timeToLive * 1000);
280    calendar.set(Calendar.HOUR_OF_DAY, 0);
281    calendar.set(Calendar.MINUTE, 0);
282    calendar.set(Calendar.SECOND, 0);
283
284    Date expireDate = calendar.getTime();
285
286    LOG.info("MOB HFiles older than " + expireDate.toGMTString() + " will be deleted!");
287
288    FileStatus[] stats = null;
289    Path mobTableDir = CommonFSUtils.getTableDir(getMobHome(conf), tableName);
290    Path path = getMobFamilyPath(conf, tableName, columnDescriptor.getNameAsString());
291    try {
292      stats = fs.listStatus(path);
293    } catch (FileNotFoundException e) {
294      LOG.warn("Failed to find the mob file " + path, e);
295    }
296    if (null == stats) {
297      // no file found
298      return;
299    }
300    List<HStoreFile> filesToClean = new ArrayList<>();
301    int deletedFileCount = 0;
302    for (FileStatus file : stats) {
303      String fileName = file.getPath().getName();
304      try {
305        if (HFileLink.isHFileLink(file.getPath())) {
306          HFileLink hfileLink = HFileLink.buildFromHFileLinkPattern(conf, file.getPath());
307          fileName = hfileLink.getOriginPath().getName();
308        }
309
310        Date fileDate = parseDate(MobFileName.getDateFromName(fileName));
311
312        if (LOG.isDebugEnabled()) {
313          LOG.debug("Checking file {}", fileName);
314        }
315        if (fileDate.getTime() < expireDate.getTime()) {
316          if (LOG.isDebugEnabled()) {
317            LOG.debug("{} is an expired file", fileName);
318          }
319          filesToClean
320            .add(new HStoreFile(fs, file.getPath(), conf, cacheConfig, BloomType.NONE, true));
321          if (
322            filesToClean.size() >= conf.getInt(MOB_CLEANER_BATCH_SIZE_UPPER_BOUND,
323              DEFAULT_MOB_CLEANER_BATCH_SIZE_UPPER_BOUND)
324          ) {
325            if (
326              removeMobFiles(conf, fs, tableName, mobTableDir, columnDescriptor.getName(),
327                filesToClean)
328            ) {
329              deletedFileCount += filesToClean.size();
330            }
331            filesToClean.clear();
332          }
333        }
334      } catch (Exception e) {
335        LOG.error("Cannot parse the fileName " + fileName, e);
336      }
337    }
338    if (
339      !filesToClean.isEmpty() && removeMobFiles(conf, fs, tableName, mobTableDir,
340        columnDescriptor.getName(), filesToClean)
341    ) {
342      deletedFileCount += filesToClean.size();
343    }
344    LOG.info("Table {} {} expired mob files in total are deleted", tableName, deletedFileCount);
345  }

490  /**
491   * Creates a mob reference KeyValue. The value of the mob reference KeyValue is mobCellValueSize +
492   * mobFileName.
493   * @param cell         The original Cell.
494   * @param fileName     The mob file name where the mob reference KeyValue is written.
495   * @param tableNameTag The tag of the current table name. It's very important in cloning the
496   *                     snapshot.
497   * @return The mob reference KeyValue.
498   */
499  public static Cell createMobRefCell(Cell cell, byte[] fileName, Tag tableNameTag) {
500    // Append the tags to the KeyValue.
501    // The key is same, the value is the filename of the mob file
502    List<Tag> tags = new ArrayList<>();
503    // Add the ref tag as the 1st one.
504    tags.add(MobConstants.MOB_REF_TAG);
505    // Add the tag of the source table name, this table is where this mob file is flushed
506    // from.
507    // It's very useful in cloning the snapshot. When reading from the cloning table, we need to
508    // find the original mob files by this table name. For details please see cloning
509    // snapshot for mob files.
510    tags.add(tableNameTag);
511    return createMobRefCell(cell, fileName, TagUtil.fromList(tags));
512  }

605   * @param blocksize             The HFile block size.
606   * @param bloomType             The bloom filter type.
607   * @param isCompaction          If the writer is used in compaction.
608   * @param writerCreationTracker to track the current writer in the store
609   * @return The writer for the mob file.
610   */
611  public static StoreFileWriter createWriter(Configuration conf, FileSystem fs,
612    ColumnFamilyDescriptor family, Path path, long maxKeyCount, Compression.Algorithm compression,
613    CacheConfig cacheConfig, Encryption.Context cryptoContext, ChecksumType checksumType,
614    int bytesPerChecksum, int blocksize, BloomType bloomType, boolean isCompaction,
615    Consumer<Path> writerCreationTracker) throws IOException {
616    if (compression == null) {
617      compression = HFile.DEFAULT_COMPRESSION_ALGORITHM;
618    }
619    final CacheConfig writerCacheConf;
620    if (isCompaction) {
621      writerCacheConf = new CacheConfig(cacheConfig);
622      writerCacheConf.setCacheDataOnWrite(false);
623    } else {
624      writerCacheConf = cacheConfig;
625    }
626    HFileContext hFileContext = new HFileContextBuilder().withCompression(compression)
627      .withIncludesMvcc(true).withIncludesTags(true).withCompressTags(family.isCompressTags())
628      .withChecksumType(checksumType).withBytesPerCheckSum(bytesPerChecksum)
629      .withBlockSize(blocksize).withHBaseCheckSum(true)
630      .withDataBlockEncoding(family.getDataBlockEncoding()).withEncryptionContext(cryptoContext)
631      .withCreateTime(EnvironmentEdgeManager.currentTime()).build();
632
633    StoreFileWriter w = new StoreFileWriter.Builder(conf, writerCacheConf, fs).withFilePath(path)
634      .withBloomType(bloomType).withMaxKeyCount(maxKeyCount).withFileContext(hFileContext)
635      .withWriterCreationTracker(writerCreationTracker).build();
636    return w;
637  }

721  /**
722   * Checks if the mob file is expired.
723   * @param column   The descriptor of the current column family.
724   * @param current  The current time.
725   * @param fileDate The date string parsed from the mob file name.
726   * @return True if the mob file is expired.
727   */
728  public static boolean isMobFileExpired(ColumnFamilyDescriptor column, long current,
729    String fileDate) {
730    if (column.getMinVersions() > 0) {
731      return false;
732    }
733    long timeToLive = column.getTimeToLive();
734    if (Integer.MAX_VALUE == timeToLive) {
735      return false;
736    }
737
738    Date expireDate = new Date(current - timeToLive * 1000);
739    expireDate = new Date(expireDate.getYear(), expireDate.getMonth(), expireDate.getDate());
740    try {
741      Date date = parseDate(fileDate);
742      if (date.getTime() < expireDate.getTime()) {
743        return true;
744      }
745    } catch (ParseException e) {
746      LOG.warn("Failed to parse the date " + fileDate, e);
747      return false;
748    }
749    return false;
750  }

752  /**
753   * Serialize a set of referenced mob hfiles
754   * @param mobRefSet to serialize, may be null
755   * @return byte array to i.e. put into store file metadata. will not be null
756   */
757  public static byte[] serializeMobFileRefs(SetMultimap<TableName, String> mobRefSet) {
758    if (mobRefSet != null && mobRefSet.size() > 0) {
759      // Here we rely on the fact that '/' and ',' are not allowed in either table names nor hfile
760      // names for serialization.
761      //
762      // exampleTable/filename1,filename2//example:table/filename5//otherTable/filename3,filename4
763      //
764      // to approximate the needed capacity we use the fact that there will usually be 1 table name
765      // and each mob filename is around 105 bytes. we pick an arbitrary number to cover "most"
766      // single table name lengths
767      StringBuilder sb = new StringBuilder(100 + mobRefSet.size() * 105);
768      boolean doubleSlash = false;
769      for (TableName tableName : mobRefSet.keySet()) {
770        if (doubleSlash) {
771          sb.append("//");
772        } else {
773          doubleSlash = true;
774        }
775        sb.append(tableName).append("/");
776        boolean comma = false;
777        for (String refs : mobRefSet.get(tableName)) {
778          if (comma) {
779            sb.append(",");
780          } else {
781            comma = true;
782          }
783          sb.append(refs);
784        }
785      }
786      return Bytes.toBytes(sb.toString());
787    } else {
788      return HStoreFile.NULL_VALUE;
789    }
790  }

792  /**
793   * Deserialize the set of referenced mob hfiles from store file metadata.
794   * @param bytes compatibly serialized data. can not be null
795   * @return a setmultimap of original table to list of hfile names. will be empty if no values.
796   * @throws IllegalStateException if there are values but no table name
797   */
798  public static ImmutableSetMultimap.Builder<TableName, String> deserializeMobFileRefs(byte[] bytes)
799    throws IllegalStateException {
800    ImmutableSetMultimap.Builder<TableName, String> map = ImmutableSetMultimap.builder();
801    if (bytes.length > 1) {
802      // TODO avoid turning the tablename pieces in to strings.
803      String s = Bytes.toString(bytes);
804      String[] tables = s.split("//");
805      for (String tableEnc : tables) {
806        final int delim = tableEnc.indexOf('/');
807        if (delim <= 0) {
808          throw new IllegalStateException("MOB reference data does not match expected encoding: "
809            + "no table name included before list of mob refs.");
810        }
811        TableName table = TableName.valueOf(tableEnc.substring(0, delim));
812        String[] refs = tableEnc.substring(delim + 1).split(",");
813        map.putAll(table, refs);
814      }
815    } else {
816      if (LOG.isDebugEnabled()) {
817        // array length 1 should be the NULL_VALUE.
818        if (!Arrays.equals(HStoreFile.NULL_VALUE, bytes)) {
819          LOG.debug(
820            "Serialized MOB file refs array was treated as the placeholder 'no entries' but"
821              + " didn't have the expected placeholder byte. expected={} and actual={}",
822            Arrays.toString(HStoreFile.NULL_VALUE), Arrays.toString(bytes));
823        }
824      }
825    }
826    return map;
827  }
