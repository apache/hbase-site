049/**
050 * An implementation of the StoreFlusher. It extends the DefaultStoreFlusher.
051 * If the store is not a mob store, the flusher flushes the MemStore the same with
052 * DefaultStoreFlusher,
053 * If the store is a mob store, the flusher flushes the MemStore into two places.
054 * One is the store files of HBase, the other is the mob files.
055 * <ol>
056 * <li>Cells that are not PUT type or have the delete mark will be directly flushed to HBase.</li>
057 * <li>If the size of a cell value is larger than a threshold, it'll be flushed
058 * to a mob file, another cell with the path of this file will be flushed to HBase.</li>
059 * <li>If the size of a cell value is smaller than or equal with a threshold, it'll be flushed to
060 * HBase directly.</li>
061 * </ol>
062 *
063 */
064@InterfaceAudience.Private
065public class DefaultMobStoreFlusher extends DefaultStoreFlusher {
066
067  private static final Log LOG = LogFactory.getLog(DefaultMobStoreFlusher.class);
068  private final Object flushLock = new Object();
069  private long mobCellValueSizeThreshold = 0;
070  private Path targetPath;
071  private HMobStore mobStore;
072
073  public DefaultMobStoreFlusher(Configuration conf, Store store) throws IOException {
074    super(conf, store);
075    mobCellValueSizeThreshold = store.getFamily().getMobThreshold();
076    this.targetPath = MobUtils.getMobFamilyPath(conf, store.getTableName(),
077        store.getColumnFamilyName());
078    if (!this.store.getFileSystem().exists(targetPath)) {
079      this.store.getFileSystem().mkdirs(targetPath);
080    }
081    this.mobStore = (HMobStore) store;
082  }
083
084  /**
085   * Flushes the snapshot of the MemStore.
086   * If this store is not a mob store, flush the cells in the snapshot to store files of HBase.
087   * If the store is a mob one, the flusher flushes the MemStore into two places.
088   * One is the store files of HBase, the other is the mob files.
089   * <ol>
090   * <li>Cells that are not PUT type or have the delete mark will be directly flushed to
091   * HBase.</li>
092   * <li>If the size of a cell value is larger than a threshold, it'll be
093   * flushed to a mob file, another cell with the path of this file will be flushed to HBase.</li>
094   * <li>If the size of a cell value is smaller than or equal with a threshold, it'll be flushed to
095   * HBase directly.</li>
096   * </ol>
097   */
098  @Override
099  public List<Path> flushSnapshot(MemStoreSnapshot snapshot, long cacheFlushId,
100      MonitoredTask status, ThroughputController throughputController) throws IOException {
101    ArrayList<Path> result = new ArrayList<Path>();
102    int cellsCount = snapshot.getCellsCount();
103    if (cellsCount == 0) return result; // don't flush if there are no entries
104
105    // Use a store scanner to find which rows to flush.
106    long smallestReadPoint = store.getSmallestReadPoint();
107    InternalScanner scanner = createScanner(snapshot.getScanner(), smallestReadPoint);
108    if (scanner == null) {
109      return result; // NULL scanner returned from coprocessor hooks means skip normal processing
110    }
111    StoreFileWriter writer;
112    try {
113      // TODO: We can fail in the below block before we complete adding this flush to
114      // list of store files. Add cleanup of anything put on filesystem if we fail.
115      synchronized (flushLock) {
116        status.setStatus("Flushing " + store + ": creating writer");
117        // Write the map out to the disk
118        writer = store.createWriterInTmp(cellsCount, store.getFamily().getCompression(),
119            false, true, true);
120        writer.setTimeRangeTracker(snapshot.getTimeRangeTracker());
121        try {
122          // It's a mob store, flush the cells in a mob way. This is the difference of flushing
123          // between a normal and a mob store.
124          performMobFlush(snapshot, cacheFlushId, scanner, writer, status);
125        } finally {
126          finalizeWriter(writer, cacheFlushId, status);
127        }
128      }
129    } finally {
130      scanner.close();
131    }
132    LOG.info("Mob store is flushed, sequenceid=" + cacheFlushId + ", memsize="
133        + StringUtils.TraditionalBinaryPrefix.long2String(snapshot.getSize(), "", 1) +
134        ", hasBloomFilter=" + writer.hasGeneralBloom() +
135        ", into tmp file " + writer.getPath());
136    result.add(writer.getPath());
137    return result;
138  }
139
140  /**
141   * Flushes the cells in the mob store.
142   * <ol>In the mob store, the cells with PUT type might have or have no mob tags.
143   * <li>If a cell does not have a mob tag, flushing the cell to different files depends
144   * on the value length. If the length is larger than a threshold, it's flushed to a
145   * mob file and the mob file is flushed to a store file in HBase. Otherwise, directly
146   * flush the cell to a store file in HBase.</li>
147   * <li>If a cell have a mob tag, its value is a mob file name, directly flush it
148   * to a store file in HBase.</li>
149   * </ol>
150   * @param snapshot Memstore snapshot.
151   * @param cacheFlushId Log cache flush sequence number.
152   * @param scanner The scanner of memstore snapshot.
153   * @param writer The store file writer.
154   * @param status Task that represents the flush operation and may be updated with status.
155   * @throws IOException
156   */
157  protected void performMobFlush(MemStoreSnapshot snapshot, long cacheFlushId,
158      InternalScanner scanner, StoreFileWriter writer, MonitoredTask status) throws IOException {
159    StoreFileWriter mobFileWriter = null;
160    int compactionKVMax = conf.getInt(HConstants.COMPACTION_KV_MAX,
161        HConstants.COMPACTION_KV_MAX_DEFAULT);
162    long mobCount = 0;
163    long mobSize = 0;
164    long time = snapshot.getTimeRangeTracker().getMaximumTimestamp();
165    mobFileWriter = mobStore.createWriterInTmp(new Date(time), snapshot.getCellsCount(),
166        store.getFamily().getCompression(), store.getRegionInfo().getStartKey());
167    // the target path is {tableName}/.mob/{cfName}/mobFiles
168    // the relative path is mobFiles
169    byte[] fileName = Bytes.toBytes(mobFileWriter.getPath().getName());
170    try {
171      Tag tableNameTag = new ArrayBackedTag(TagType.MOB_TABLE_NAME_TAG_TYPE,
172          store.getTableName().getName());
173      List<Cell> cells = new ArrayList<Cell>();
174      boolean hasMore;
175      ScannerContext scannerContext =
176              ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
177
178      do {
179        hasMore = scanner.next(cells, scannerContext);
180        if (!cells.isEmpty()) {
181          for (Cell c : cells) {
182            // If we know that this KV is going to be included always, then let us
183            // set its memstoreTS to 0. This will help us save space when writing to
184            // disk.
185            if (c.getValueLength() <= mobCellValueSizeThreshold || MobUtils.isMobReferenceCell(c)
186                || c.getTypeByte() != KeyValue.Type.Put.getCode()) {
187              writer.append(c);
188            } else {
189              // append the original keyValue in the mob file.
190              mobFileWriter.append(c);
191              mobSize += c.getValueLength();
192              mobCount++;
193
194              // append the tags to the KeyValue.
195              // The key is same, the value is the filename of the mob file
196              KeyValue reference = MobUtils.createMobRefKeyValue(c, fileName, tableNameTag);
197              writer.append(reference);
198            }
199          }
200          cells.clear();
201        }
202      } while (hasMore);
203    } finally {
204      status.setStatus("Flushing mob file " + store + ": appending metadata");
205      mobFileWriter.appendMetadata(cacheFlushId, false, mobCount);
206      status.setStatus("Flushing mob file " + store + ": closing flushed file");
207      mobFileWriter.close();
208    }
209
210    if (mobCount > 0) {
211      // commit the mob file from temp folder to target folder.
212      // If the mob file is committed successfully but the store file is not,
213      // the committed mob file will be handled by the sweep tool as an unused
214      // file.
215      mobStore.commitFile(mobFileWriter.getPath(), targetPath);
216      mobStore.updateMobFlushCount();
217      mobStore.updateMobFlushedCellsCount(mobCount);
218      mobStore.updateMobFlushedCellsSize(mobSize);
219    } else {
220      try {
221        // If the mob file is empty, delete it instead of committing.
222        store.getFileSystem().delete(mobFileWriter.getPath(), true);
223      } catch (IOException e) {
224        LOG.error("Failed to delete the temp mob file", e);
225      }
226    }
227  }
