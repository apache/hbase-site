020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.List;
024import java.util.Map;
025import java.util.concurrent.TimeUnit;

027import org.apache.hadoop.conf.Configuration;

044/**
045 * Periodic MOB compaction chore.
046 * It runs MOB compaction on region servers in parallel, thus
047 * utilizing distributed cluster resources. To avoid possible major
048 * compaction storms, one can specify maximum number regions to be compacted
049 * in parallel by setting configuration parameter: <br>
050 * 'hbase.mob.major.compaction.region.batch.size', which by default is 0 (unlimited).
051 *
052 */
053@InterfaceAudience.Private
054public class MobFileCompactionChore extends ScheduledChore {
055
056  private static final Logger LOG = LoggerFactory.getLogger(MobFileCompactionChore.class);
057  private HMaster master;
058  private int regionBatchSize = 0;// not set - compact all
059
060  public MobFileCompactionChore(HMaster master) {
061    super(master.getServerName() + "-MobFileCompactionChore", master,
062        master.getConfiguration().getInt(MobConstants.MOB_COMPACTION_CHORE_PERIOD,
063          MobConstants.DEFAULT_MOB_COMPACTION_CHORE_PERIOD),
064        master.getConfiguration().getInt(MobConstants.MOB_COMPACTION_CHORE_PERIOD,
065          MobConstants.DEFAULT_MOB_COMPACTION_CHORE_PERIOD),
066        TimeUnit.SECONDS);
067    this.master = master;
068    this.regionBatchSize =
069        master.getConfiguration().getInt(MobConstants.MOB_MAJOR_COMPACTION_REGION_BATCH_SIZE,
070          MobConstants.DEFAULT_MOB_MAJOR_COMPACTION_REGION_BATCH_SIZE);
071
072  }
073
074  public MobFileCompactionChore(Configuration conf, int batchSize) {
075    this.regionBatchSize = batchSize;
076  }
077
078  @Override
079  protected void chore() {
080
081    boolean reported = false;
082
083    try (Admin admin = master.getConnection().getAdmin()) {
084      TableDescriptors htds = master.getTableDescriptors();
085      Map<String, TableDescriptor> map = htds.getAll();
086      for (TableDescriptor htd : map.values()) {
087        if (!master.getTableStateManager().isTableState(htd.getTableName(),
088          TableState.State.ENABLED)) {
089          LOG.info("Skipping MOB compaction on table {} because it is not ENABLED",
090            htd.getTableName());
091          continue;
092        } else {
093          LOG.info("Starting MOB compaction on table {}, checking {} column families",
094            htd.getTableName(), htd.getColumnFamilyCount());
095        }
096        for (ColumnFamilyDescriptor hcd : htd.getColumnFamilies()) {
097          try {
098            if (hcd.isMobEnabled()) {
099              if (!reported) {
100                master.reportMobCompactionStart(htd.getTableName());
101                reported = true;
102              }
103              LOG.info("Major MOB compacting table={} cf={}", htd.getTableName(),
104                hcd.getNameAsString());
105              if (regionBatchSize == MobConstants.DEFAULT_MOB_MAJOR_COMPACTION_REGION_BATCH_SIZE) {
106                LOG.debug("Table={} cf ={}: batch MOB compaction is disabled, {}=0 -"+
107                  " all regions will be compacted in parallel", htd.getTableName(),
108                  hcd.getNameAsString(), "hbase.mob.compaction.batch.size");
109                admin.majorCompact(htd.getTableName(), hcd.getName());
110              } else {
111                LOG.info("Table={} cf={}: performing MOB major compaction in batches "+
112                    "'hbase.mob.compaction.batch.size'={}", htd.getTableName(),
113                    hcd.getNameAsString(), regionBatchSize);
114                performMajorCompactionInBatches(admin, htd, hcd);
115              }
116            } else {
117              LOG.debug("Skipping table={} column family={} because it is not MOB-enabled",
118                htd.getTableName(), hcd.getNameAsString());
119            }
120          } catch (IOException e) {
121            LOG.error("Failed to compact table={} cf={}",
122              htd.getTableName(), hcd.getNameAsString(), e);
123          } catch (InterruptedException ee) {
124            Thread.currentThread().interrupt();
125            master.reportMobCompactionEnd(htd.getTableName());
126            LOG.warn("Failed to compact table={} cf={}",
127              htd.getTableName(), hcd.getNameAsString(), ee);
128            // Quit the chore
129            return;
130          }
131        }
132        if (reported) {
133          master.reportMobCompactionEnd(htd.getTableName());
134          reported = false;
135        }
136      }
137    } catch (IOException e) {
138      LOG.error("Failed to compact", e);
139    }
140  }
141
142  public void performMajorCompactionInBatches(Admin admin, TableDescriptor htd,
143      ColumnFamilyDescriptor hcd) throws IOException, InterruptedException {
144
145    List<RegionInfo> regions = admin.getRegions(htd.getTableName());
146    if (regions.size() <= this.regionBatchSize) {
147      LOG.debug(
148        "Table={} cf={} - performing major MOB compaction in non-batched mode,"
149            + "regions={}, batch size={}",
150        htd.getTableName(), hcd.getNameAsString(), regions.size(), regionBatchSize);
151      admin.majorCompact(htd.getTableName(), hcd.getName());
152      return;
153    }
154    // Shuffle list of regions in case if they come ordered by region server
155    Collections.shuffle(regions);
156    // Create first batch
157    List<RegionInfo> toCompact = new ArrayList<RegionInfo>(this.regionBatchSize);
158    for (int i = 0; i < this.regionBatchSize; i++) {
159      toCompact.add(regions.remove(0));
160    }
161
162    // Start compaction now
163    for (RegionInfo ri : toCompact) {
164      startCompaction(admin, htd.getTableName(), ri, hcd.getName());
165    }
166
167    List<RegionInfo> compacted = new ArrayList<RegionInfo>(toCompact.size());
168    List<RegionInfo> failed = new ArrayList<RegionInfo>();
169    int totalCompacted = 0;
170    while (!toCompact.isEmpty()) {
171      // Check status of active compactions
172      for (RegionInfo ri : toCompact) {
173        try {
174          if (admin.getCompactionStateForRegion(ri.getRegionName()) == CompactionState.NONE) {
175            totalCompacted++;
176            LOG.info(
177              "Finished major MOB compaction: table={} cf={} region={} compacted regions={}",
178              htd.getTableName(), hcd.getNameAsString(), ri.getRegionNameAsString(),
179              totalCompacted);
180            compacted.add(ri);
181          }
182        } catch (IOException e) {
183          LOG.error("Could not get compaction state for table={} cf={} region={}, compaction will"+
184            " aborted for the region.",
185            htd.getTableName(), hcd.getNameAsString(), ri.getEncodedName());
186          LOG.error("Because of:", e);
187          failed.add(ri);
188        }
189      }
190      // Remove failed regions to avoid
191      // endless compaction loop
192      toCompact.removeAll(failed);
193      failed.clear();
194      // Update batch: remove compacted regions and add new ones
195      for (RegionInfo ri : compacted) {
196        toCompact.remove(ri);
197        if (regions.size() > 0) {
198          RegionInfo region = regions.remove(0);
199          toCompact.add(region);
200          startCompaction(admin, htd.getTableName(), region, hcd.getName());
201        }
202      }
203      compacted.clear();
204
205      LOG.debug(
206        "Table={}  cf={}. Wait for 10 sec, toCompact size={} regions left={}"
207        + " compacted so far={}", htd.getTableName(), hcd.getNameAsString(), toCompact.size(),
208        regions.size(), totalCompacted);
209      Thread.sleep(10000);
210    }
211    LOG.info("Finished major MOB compacting table={}. cf={}", htd.getTableName(),
212      hcd.getNameAsString());
213
214  }
215
216  private void startCompaction(Admin admin, TableName table, RegionInfo region, byte[] cf)
217      throws IOException, InterruptedException {
218
219    LOG.info("Started major compaction: table={} cf={} region={}", table,
220      Bytes.toString(cf), region.getRegionNameAsString());
221    admin.majorCompactRegion(region.getRegionName(), cf);
222    // Wait until it really starts
223    // but with finite timeout
224    long waitTime = 300000; // 5 min
225    long startTime = EnvironmentEdgeManager.currentTime();
226    while (admin.getCompactionStateForRegion(region.getRegionName()) == CompactionState.NONE) {
227      // Is 1 second too aggressive?
228      Thread.sleep(1000);
229      if (EnvironmentEdgeManager.currentTime() - startTime > waitTime) {
230        LOG.warn("Waited for {} ms to start major MOB compaction on table={} cf={} region={}."+
231          " Stopped waiting for request confirmation. This is not an ERROR, continue next region."
232          , waitTime, table.getNameAsString(), Bytes.toString(cf),region.getRegionNameAsString());
233        break;
234      }
235    }
236  }
