043/**
044 * Used by the HMaster on startup to split meta logs and assign the meta table.
045 */
046@InterfaceAudience.Private
047public class MasterMetaBootstrap {
048  private static final Log LOG = LogFactory.getLog(MasterMetaBootstrap.class);
049
050  private final MonitoredTask status;
051  private final HMaster master;
052
053  private Set<ServerName> previouslyFailedServers;
054  private Set<ServerName> previouslyFailedMetaRSs;
055
056  public MasterMetaBootstrap(final HMaster master, final MonitoredTask status) {
057    this.master = master;
058    this.status = status;
059  }
060
061  public void splitMetaLogsBeforeAssignment() throws IOException, KeeperException {
062    // get a list for previously failed RS which need log splitting work
063    // we recover hbase:meta region servers inside master initialization and
064    // handle other failed servers in SSH in order to start up master node ASAP
065    previouslyFailedServers = master.getMasterWalManager().getFailedServersFromLogFolders();
066
067    // log splitting for hbase:meta server
068    ServerName oldMetaServerLocation = master.getMetaTableLocator()
069        .getMetaRegionLocation(master.getZooKeeper());
070    if (oldMetaServerLocation != null && previouslyFailedServers.contains(oldMetaServerLocation)) {
071      splitMetaLogBeforeAssignment(oldMetaServerLocation);
072      // Note: we can't remove oldMetaServerLocation from previousFailedServers list because it
073      // may also host user regions
074    }
075    previouslyFailedMetaRSs = getPreviouselyFailedMetaServersFromZK();
076    // need to use union of previouslyFailedMetaRSs recorded in ZK and previouslyFailedServers
077    // instead of previouslyFailedMetaRSs alone to address the following two situations:
078    // 1) the chained failure situation(recovery failed multiple times in a row).
079    // 2) master get killed right before it could delete the recovering hbase:meta from ZK while the
080    // same server still has non-meta wals to be replayed so that
081    // removeStaleRecoveringRegionsFromZK can't delete the stale hbase:meta region
082    // Passing more servers into splitMetaLog is all right. If a server doesn't have hbase:meta wal,
083    // there is no op for the server.
084    previouslyFailedMetaRSs.addAll(previouslyFailedServers);
085  }
086
087  public void assignMeta() throws InterruptedException, IOException, KeeperException {
088    assignMeta(previouslyFailedMetaRSs, HRegionInfo.DEFAULT_REPLICA_ID);
089  }
090
091  public void processDeadServers() throws IOException {
092    // Master has recovered hbase:meta region server and we put
093    // other failed region servers in a queue to be handled later by SSH
094    for (ServerName tmpServer : previouslyFailedServers) {
095      master.getServerManager().processDeadServer(tmpServer, true);
096    }
097  }
098
099  public void assignMetaReplicas()
100      throws IOException, InterruptedException, KeeperException {
101    int numReplicas = master.getConfiguration().getInt(HConstants.META_REPLICAS_NUM,
102           HConstants.DEFAULT_META_REPLICA_NUM);
103    final Set<ServerName> EMPTY_SET = new HashSet<ServerName>();
104    for (int i = 1; i < numReplicas; i++) {
105      assignMeta(EMPTY_SET, i);
106    }
107    unassignExcessMetaReplica(numReplicas);
108  }
109
110  private void splitMetaLogBeforeAssignment(ServerName currentMetaServer) throws IOException {
111    if (RecoveryMode.LOG_REPLAY == master.getMasterWalManager().getLogRecoveryMode()) {
112      // In log replay mode, we mark hbase:meta region as recovering in ZK
113      master.getMasterWalManager().prepareLogReplay(currentMetaServer,
114        Collections.<HRegionInfo>singleton(HRegionInfo.FIRST_META_REGIONINFO));
115    } else {
116      // In recovered.edits mode: create recovered edits file for hbase:meta server
117      master.getMasterWalManager().splitMetaLog(currentMetaServer);
118    }
119  }
120
121  private void unassignExcessMetaReplica(int numMetaReplicasConfigured) {
122    final ZooKeeperWatcher zooKeeper = master.getZooKeeper();
123    // unassign the unneeded replicas (for e.g., if the previous master was configured
124    // with a replication of 3 and now it is 2, we need to unassign the 1 unneeded replica)
125    try {
126      List<String> metaReplicaZnodes = zooKeeper.getMetaReplicaNodes();
127      for (String metaReplicaZnode : metaReplicaZnodes) {
128        int replicaId = zooKeeper.znodePaths.getMetaReplicaIdFromZnode(metaReplicaZnode);
129        if (replicaId >= numMetaReplicasConfigured) {
130          RegionState r = MetaTableLocator.getMetaRegionState(zooKeeper, replicaId);
131          LOG.info("Closing excess replica of meta region " + r.getRegion());
132          // send a close and wait for a max of 30 seconds
133          ServerManager.closeRegionSilentlyAndWait(master.getClusterConnection(),
134              r.getServerName(), r.getRegion(), 30000);
135          ZKUtil.deleteNode(zooKeeper, zooKeeper.znodePaths.getZNodeForReplica(replicaId));
136        }
137      }
138    } catch (Exception ex) {
139      // ignore the exception since we don't want the master to be wedged due to potential
140      // issues in the cleanup of the extra regions. We can do that cleanup via hbck or manually
141      LOG.warn("Ignoring exception " + ex);
142    }
143  }
144
145  /**
146   * Check <code>hbase:meta</code> is assigned. If not, assign it.
147   */
148  protected void assignMeta(Set<ServerName> previouslyFailedMetaRSs, int replicaId)
149      throws InterruptedException, IOException, KeeperException {
150    final AssignmentManager assignmentManager = master.getAssignmentManager();
151
152    // Work on meta region
153    int assigned = 0;
154    long timeout = master.getConfiguration().getLong("hbase.catalog.verification.timeout", 1000);
155    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
156      status.setStatus("Assigning hbase:meta region");
157    } else {
158      status.setStatus("Assigning hbase:meta region, replicaId " + replicaId);
159    }
160
161    // Get current meta state from zk.
162    RegionState metaState = MetaTableLocator.getMetaRegionState(master.getZooKeeper(), replicaId);
163    HRegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(HRegionInfo.FIRST_META_REGIONINFO,
164        replicaId);
165    RegionStates regionStates = assignmentManager.getRegionStates();
166    regionStates.createRegionState(hri, metaState.getState(),
167        metaState.getServerName(), null);
168
169    if (!metaState.isOpened() || !master.getMetaTableLocator().verifyMetaRegionLocation(
170        master.getClusterConnection(), master.getZooKeeper(), timeout, replicaId)) {
171      ServerName currentMetaServer = metaState.getServerName();
172      if (master.getServerManager().isServerOnline(currentMetaServer)) {
173        if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
174          LOG.info("Meta was in transition on " + currentMetaServer);
175        } else {
176          LOG.info("Meta with replicaId " + replicaId + " was in transition on " +
177                    currentMetaServer);
178        }
179        assignmentManager.processRegionsInTransition(Collections.singletonList(metaState));
180      } else {
181        if (currentMetaServer != null) {
182          if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
183            splitMetaLogBeforeAssignment(currentMetaServer);
184            regionStates.logSplit(HRegionInfo.FIRST_META_REGIONINFO);
185            previouslyFailedMetaRSs.add(currentMetaServer);
186          }
187        }
188        LOG.info("Re-assigning hbase:meta with replicaId, " + replicaId +
189            " it was on " + currentMetaServer);
190        assignmentManager.assignMeta(hri);
191      }
192      assigned++;
193    }
194
195    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
196      // TODO: should we prevent from using state manager before meta was initialized?
197      // tableStateManager.start();
198      master.getTableStateManager()
199        .setTableState(TableName.META_TABLE_NAME, TableState.State.ENABLED);
200    }
201
202    if ((RecoveryMode.LOG_REPLAY == master.getMasterWalManager().getLogRecoveryMode())
203        && (!previouslyFailedMetaRSs.isEmpty())) {
204      // replay WAL edits mode need new hbase:meta RS is assigned firstly
205      status.setStatus("replaying log for Meta Region");
206      master.getMasterWalManager().splitMetaLog(previouslyFailedMetaRSs);
207    }
208
209    assignmentManager.setEnabledTable(TableName.META_TABLE_NAME);
210    master.getTableStateManager().start();
211
212    // Make sure a hbase:meta location is set. We need to enable SSH here since
213    // if the meta region server is died at this time, we need it to be re-assigned
214    // by SSH so that system tables can be assigned.
215    // No need to wait for meta is assigned = 0 when meta is just verified.
216    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) enableCrashedServerProcessing(assigned != 0);
217    LOG.info("hbase:meta with replicaId " + replicaId + " assigned=" + assigned + ", location="
218      + master.getMetaTableLocator().getMetaRegionLocation(master.getZooKeeper(), replicaId));
219    status.setStatus("META assigned.");
220  }
221
222  private void enableCrashedServerProcessing(final boolean waitForMeta)
223      throws IOException, InterruptedException {
224    // If crashed server processing is disabled, we enable it and expire those dead but not expired
225    // servers. This is required so that if meta is assigning to a server which dies after
226    // assignMeta starts assignment, ServerCrashProcedure can re-assign it. Otherwise, we will be
227    // stuck here waiting forever if waitForMeta is specified.
228    if (!master.isServerCrashProcessingEnabled()) {
229      master.setServerCrashProcessingEnabled(true);
230      master.getServerManager().processQueuedDeadServers();
231    }
232
233    if (waitForMeta) {
234      master.getMetaTableLocator().waitMetaRegionLocation(master.getZooKeeper());
235    }
236  }

238  /**
239   * This function returns a set of region server names under hbase:meta recovering region ZK node
240   * @return Set of meta server names which were recorded in ZK
241   */
242  private Set<ServerName> getPreviouselyFailedMetaServersFromZK() throws KeeperException {
243    final ZooKeeperWatcher zooKeeper = master.getZooKeeper();
244    Set<ServerName> result = new HashSet<ServerName>();
245    String metaRecoveringZNode = ZKUtil.joinZNode(zooKeeper.znodePaths.recoveringRegionsZNode,
246      HRegionInfo.FIRST_META_REGIONINFO.getEncodedName());
247    List<String> regionFailedServers = ZKUtil.listChildrenNoWatch(zooKeeper, metaRecoveringZNode);
248    if (regionFailedServers == null) return result;
249
250    for (String failedServer : regionFailedServers) {
251      ServerName server = ServerName.parseServerName(failedServer);
252      result.add(server);
253    }
254    return result;
255  }
