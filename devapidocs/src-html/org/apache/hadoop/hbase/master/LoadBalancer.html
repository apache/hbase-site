019package org.apache.hadoop.hbase.master;
020
021import edu.umd.cs.findbugs.annotations.Nullable;
022import java.io.IOException;
023import java.util.List;
024import java.util.Map;
025import org.apache.hadoop.conf.Configurable;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.ClusterMetrics;
028import org.apache.hadoop.hbase.ServerName;
029import org.apache.hadoop.hbase.Stoppable;
030import org.apache.hadoop.hbase.TableName;
031import org.apache.hadoop.hbase.client.RegionInfo;
032import org.apache.hadoop.hbase.conf.ConfigurationObserver;
033import org.apache.yetus.audience.InterfaceAudience;
034
035/**
036 * Makes decisions about the placement and movement of Regions across
037 * RegionServers.
038 *
039 * <p>Cluster-wide load balancing will occur only when there are no regions in
040 * transition and according to a fixed period of a time using {@link #balanceCluster(Map)}.
041 *
042 * <p>On cluster startup, bulk assignment can be used to determine
043 * locations for all Regions in a cluster.
044 *
045 * <p>This class produces plans for the
046 * {@link org.apache.hadoop.hbase.master.assignment.AssignmentManager}
047 * to execute.
048 */
049@InterfaceAudience.Private
050public interface LoadBalancer extends Configurable, Stoppable, ConfigurationObserver {
051  /**
052   * Master can carry regions as of hbase-2.0.0.
053   * By default, it carries no tables.
054   * TODO: Add any | system as flags to indicate what it can do.
055   */
056  String TABLES_ON_MASTER = "hbase.balancer.tablesOnMaster";
057
058  /**
059   * Master carries system tables.
060   */
061  String SYSTEM_TABLES_ON_MASTER = "hbase.balancer.tablesOnMaster.systemTablesOnly";
062
063  // Used to signal to the caller that the region(s) cannot be assigned
064  // We deliberately use 'localhost' so the operation will fail fast
065  ServerName BOGUS_SERVER_NAME = ServerName.valueOf("localhost,1,1");
066
067  /**
068   * Config for pluggable load balancers.
069   * @deprecated since 3.0.0, will be removed in 4.0.0. In the new implementation, as the base load
070   *             balancer will always be the rs group based one, you should just use
071   *             {@link org.apache.hadoop.hbase.HConstants#HBASE_MASTER_LOADBALANCER_CLASS} to
072   *             config the per group load balancer.
073   */
074  @Deprecated
075  String HBASE_RSGROUP_LOADBALANCER_CLASS = "hbase.rsgroup.grouploadbalancer.class";
076  /**
077   * Set the current cluster status. This allows a LoadBalancer to map host name to a server
078   */
079  void setClusterMetrics(ClusterMetrics st);
080
081
082  /**
083   * Set the master service.
084   */
085  void setMasterServices(MasterServices masterServices);
086
087  /**
088   * Perform the major balance operation for cluster, will invoke {@link #balanceTable} to do actual
089   * balance. Normally not need override this method, except
090   * {@link org.apache.hadoop.hbase.master.balancer.SimpleLoadBalancer} and
091   * {@link org.apache.hadoop.hbase.rsgroup.RSGroupBasedLoadBalancer}
092   * @param loadOfAllTable region load of servers for all table
093   * @return a list of regions to be moved, including source and destination, or null if cluster is
094   *         already balanced
095   */
096  List<RegionPlan> balanceCluster(Map<TableName,
097      Map<ServerName, List<RegionInfo>>> loadOfAllTable) throws IOException;
098
099  /**
100   * Perform the major balance operation for table, all class implement of {@link LoadBalancer}
101   * should override this method
102   * @param tableName the table to be balanced
103   * @param loadOfOneTable region load of servers for the specific one table
104   * @return List of plans
105   */
106  List<RegionPlan> balanceTable(TableName tableName,
107      Map<ServerName, List<RegionInfo>> loadOfOneTable);
108
109  /**
110   * Perform a Round Robin assignment of regions.
111   * @return Map of servername to regioninfos
112   */
113  Map<ServerName, List<RegionInfo>> roundRobinAssignment(List<RegionInfo> regions,
114      List<ServerName> servers) throws IOException;
115
116  /**
117   * Assign regions to the previously hosting region server
118   * @return List of plans
119   */
120  @Nullable
121  Map<ServerName, List<RegionInfo>> retainAssignment(Map<RegionInfo, ServerName> regions,
122      List<ServerName> servers) throws IOException;
123
124  /**
125   * Get a random region server from the list
126   * @param regionInfo Region for which this selection is being done.
127   */
128  ServerName randomAssignment(RegionInfo regionInfo, List<ServerName> servers) throws IOException;
129
130  /**
131   * Initialize the load balancer. Must be called after setters.
132   */
133  void initialize() throws IOException;
134
135  /**
136   * Marks the region as online at balancer.
137   */
138  void regionOnline(RegionInfo regionInfo, ServerName sn);
139
140  /**
141   * Marks the region as offline at balancer.
142   */
143  void regionOffline(RegionInfo regionInfo);
144
145  /**
146   * Notification that config has changed
147   */
148  @Override
149  void onConfigurationChange(Configuration conf);
150
151  /**
152   * If balancer needs to do initialization after Master has started up, lets do that here.
153   */
154  void postMasterStartupInitialize();
155
156  /*Updates balancer status tag reported to JMX*/
157  void updateBalancerStatus(boolean status);
158
159  /**
160   * @return true if Master carries regions
161   */
162  static boolean isTablesOnMaster(Configuration conf) {
163    return conf.getBoolean(TABLES_ON_MASTER, false);
164  }
165
166  static boolean isSystemTablesOnlyOnMaster(Configuration conf) {
167    return conf.getBoolean(SYSTEM_TABLES_ON_MASTER, false);
168  }
169
170  static boolean isMasterCanHostUserRegions(Configuration conf) {
171    return isTablesOnMaster(conf) && !isSystemTablesOnlyOnMaster(conf);
172  }
173}
