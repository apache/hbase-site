019package org.apache.hadoop.hbase.master;
020
021import edu.umd.cs.findbugs.annotations.NonNull;
022import java.io.IOException;
023import java.util.List;
024import java.util.Map;
025import org.apache.hadoop.conf.Configurable;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.ClusterMetrics;
028import org.apache.hadoop.hbase.ServerName;
029import org.apache.hadoop.hbase.Stoppable;
030import org.apache.hadoop.hbase.TableName;
031import org.apache.hadoop.hbase.client.RegionInfo;
032import org.apache.hadoop.hbase.conf.ConfigurationObserver;
033import org.apache.hadoop.hbase.master.balancer.ClusterInfoProvider;
034import org.apache.yetus.audience.InterfaceAudience;
035
036/**
037 * Makes decisions about the placement and movement of Regions across RegionServers.
038 * <p/>
039 * Cluster-wide load balancing will occur only when there are no regions in transition and according
040 * to a fixed period of a time using {@link #balanceCluster(Map)}.
041 * <p/>
042 * On cluster startup, bulk assignment can be used to determine locations for all Regions in a
043 * cluster.
044 * <p/>
045 * This class produces plans for the {@code AssignmentManager} to execute.
046 */
047@InterfaceAudience.Private
048public interface LoadBalancer extends Configurable, Stoppable, ConfigurationObserver {
049
050  // Used to signal to the caller that the region(s) cannot be assigned
051  // We deliberately use 'localhost' so the operation will fail fast
052  ServerName BOGUS_SERVER_NAME = ServerName.valueOf("localhost,1,1");
053
054  /**
055   * Config for pluggable load balancers.
056   * @deprecated since 3.0.0, will be removed in 4.0.0. In the new implementation, as the base load
057   *             balancer will always be the rs group based one, you should just use
058   *             {@link org.apache.hadoop.hbase.HConstants#HBASE_MASTER_LOADBALANCER_CLASS} to
059   *             config the per group load balancer.
060   */
061  @Deprecated
062  String HBASE_RSGROUP_LOADBALANCER_CLASS = "hbase.rsgroup.grouploadbalancer.class";
063  /**
064   * Set the current cluster status. This allows a LoadBalancer to map host name to a server
065   */
066  void setClusterMetrics(ClusterMetrics st);
067
068
069  /**
070   * Set the cluster info provider. Usually it is just a wrapper of master.
071   */
072  void setClusterInfoProvider(ClusterInfoProvider provider);
073
074  /**
075   * Perform the major balance operation for cluster, will invoke {@link #balanceTable} to do actual
076   * balance. Normally not need override this method, except {@link SimpleLoadBalancer} and
077   * {@code RSGroupBasedLoadBalancer}
078   * @param loadOfAllTable region load of servers for all table
079   * @return a list of regions to be moved, including source and destination, or null if cluster is
080   *         already balanced
081   */
082  List<RegionPlan> balanceCluster(Map<TableName,
083      Map<ServerName, List<RegionInfo>>> loadOfAllTable) throws IOException;
084
085  /**
086   * Perform the major balance operation for table, all class implement of {@link LoadBalancer}
087   * should override this method
088   * @param tableName the table to be balanced
089   * @param loadOfOneTable region load of servers for the specific one table
090   * @return List of plans
091   */
092  List<RegionPlan> balanceTable(TableName tableName,
093      Map<ServerName, List<RegionInfo>> loadOfOneTable);
094
095  /**
096   * Perform a Round Robin assignment of regions.
097   * @return Map of servername to regioninfos
098   */
099  @NonNull
100  Map<ServerName, List<RegionInfo>> roundRobinAssignment(List<RegionInfo> regions,
101      List<ServerName> servers) throws IOException;
102
103  /**
104   * Assign regions to the previously hosting region server
105   * @return List of plans
106   */
107  @NonNull
108  Map<ServerName, List<RegionInfo>> retainAssignment(Map<RegionInfo, ServerName> regions,
109      List<ServerName> servers) throws IOException;
110
111  /**
112   * Get a random region server from the list
113   * @param regionInfo Region for which this selection is being done.
114   */
115  ServerName randomAssignment(RegionInfo regionInfo, List<ServerName> servers) throws IOException;
116
117  /**
118   * Initialize the load balancer. Must be called after setters.
119   */
120  void initialize() throws IOException;
121
122  /**
123   * Marks the region as online at balancer.
124   */
125  void regionOnline(RegionInfo regionInfo, ServerName sn);
126
127  /**
128   * Marks the region as offline at balancer.
129   */
130  void regionOffline(RegionInfo regionInfo);
131
132  /**
133   * Notification that config has changed
134   */
135  @Override
136  void onConfigurationChange(Configuration conf);
137
138  /**
139   * If balancer needs to do initialization after Master has started up, lets do that here.
140   */
141  void postMasterStartupInitialize();
142
143  /*Updates balancer status tag reported to JMX*/
144  void updateBalancerStatus(boolean status);
145}
