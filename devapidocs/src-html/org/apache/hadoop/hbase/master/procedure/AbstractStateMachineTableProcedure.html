019package org.apache.hadoop.hbase.master.procedure;
020
021import java.io.IOException;
022import org.apache.hadoop.fs.Path;
023import org.apache.hadoop.hbase.HBaseIOException;
024import org.apache.hadoop.hbase.MetaTableAccessor;
025import org.apache.hadoop.hbase.TableName;
026import org.apache.hadoop.hbase.TableNotDisabledException;
027import org.apache.hadoop.hbase.TableNotEnabledException;
028import org.apache.hadoop.hbase.TableNotFoundException;
029import org.apache.hadoop.hbase.client.RegionInfo;
030import org.apache.hadoop.hbase.client.TableState;
031import org.apache.hadoop.hbase.master.MasterFileSystem;
032import org.apache.hadoop.hbase.master.MasterServices;
033import org.apache.hadoop.hbase.master.TableStateManager;
034import org.apache.hadoop.hbase.procedure2.StateMachineProcedure;
035import org.apache.hadoop.hbase.security.User;
036import org.apache.yetus.audience.InterfaceAudience;
037
038/**
039 * Base class for all the Table procedures that want to use a StateMachineProcedure.
040 * It provides helpers like basic locking, sync latch, and toStringClassDetails().
041 */
042@InterfaceAudience.Private
043public abstract class AbstractStateMachineTableProcedure<TState>
044    extends StateMachineProcedure<MasterProcedureEnv, TState>
045    implements TableProcedureInterface {
046
047  // used for compatibility with old clients
048  private final ProcedurePrepareLatch syncLatch;
049
050  private User user;
051
052  protected AbstractStateMachineTableProcedure() {
053    // Required by the Procedure framework to create the procedure on replay
054    syncLatch = null;
055  }
056
057  protected AbstractStateMachineTableProcedure(final MasterProcedureEnv env) {
058    this(env, null);
059  }
060
061  /**
062   * @param env Uses this to set Procedure Owner at least.
063   */
064  protected AbstractStateMachineTableProcedure(final MasterProcedureEnv env,
065      final ProcedurePrepareLatch latch) {
066    if (env != null) {
067      this.user = env.getRequestUser();
068      this.setOwner(user);
069    }
070    // used for compatibility with clients without procedures
071    // they need a sync TableExistsException, TableNotFoundException, TableNotDisabledException, ...
072    this.syncLatch = latch;
073  }
074
075  @Override
076  public abstract TableName getTableName();
077
078  @Override
079  public abstract TableOperationType getTableOperationType();
080
081  @Override
082  public void toStringClassDetails(final StringBuilder sb) {
083    sb.append(getClass().getSimpleName());
084    sb.append(" table=");
085    sb.append(getTableName());
086  }
087
088  @Override
089  protected LockState acquireLock(final MasterProcedureEnv env) {
090    if (env.waitInitialized(this)) return LockState.LOCK_EVENT_WAIT;
091    if (env.getProcedureScheduler().waitTableExclusiveLock(this, getTableName())) {
092      return LockState.LOCK_EVENT_WAIT;
093    }
094    return LockState.LOCK_ACQUIRED;
095  }
096
097  @Override
098  protected void releaseLock(final MasterProcedureEnv env) {
099    env.getProcedureScheduler().wakeTableExclusiveLock(this, getTableName());
100  }
101
102  protected User getUser() {
103    return user;
104  }
105
106  protected void setUser(final User user) {
107    this.user = user;
108  }
109
110  protected void releaseSyncLatch() {
111    ProcedurePrepareLatch.releaseLatch(syncLatch, this);
112  }
113
114  /**
115   * Check whether a table is modifiable - exists and either offline or online with config set
116   * @param env MasterProcedureEnv
117   * @throws IOException
118   */
119  protected void checkTableModifiable(final MasterProcedureEnv env) throws IOException {
120    // Checks whether the table exists
121    if (!MetaTableAccessor.tableExists(env.getMasterServices().getConnection(), getTableName())) {
122      throw new TableNotFoundException(getTableName());
123    }
124  }
125
126  protected final Path getRegionDir(MasterProcedureEnv env, RegionInfo region) throws IOException {
127    return env.getMasterServices().getMasterFileSystem().getRegionDir(region);
128  }
129
130  /**
131   * Check that cluster is up and master is running. Check table is modifiable.
132   * If <code>enabled</code>, check table is enabled else check it is disabled.
133   * Call in Procedure constructor so can pass any exception to caller.
134   * @param enabled If true, check table is enabled and throw exception if not. If false, do the
135   *                inverse. If null, do no table checks.
136   */
137  protected void preflightChecks(MasterProcedureEnv env, Boolean enabled) throws HBaseIOException {
138    MasterServices master = env.getMasterServices();
139    if (!master.isClusterUp()) {
140      throw new HBaseIOException("Cluster not up!");
141    }
142    if (master.isStopping() || master.isStopped()) {
143      throw new HBaseIOException("Master stopping=" + master.isStopping() +
144          ", stopped=" + master.isStopped());
145    }
146    if (enabled == null) {
147      // Don't do any table checks.
148      return;
149    }
150    try {
151      // Checks table exists and is modifiable.
152      checkTableModifiable(env);
153      TableName tn = getTableName();
154      TableStateManager tsm = master.getTableStateManager();
155      TableState ts = tsm.getTableState(tn);
156      if (enabled) {
157        if (!ts.isEnabledOrEnabling()) {
158          throw new TableNotEnabledException(tn);
159        }
160      } else {
161        if (!ts.isDisabledOrDisabling()) {
162          throw new TableNotDisabledException(tn);
163        }
164      }
165    } catch (IOException ioe) {
166      if (ioe instanceof HBaseIOException) {
167        throw (HBaseIOException)ioe;
168      }
169      throw new HBaseIOException(ioe);
170    }
171  }
172}
