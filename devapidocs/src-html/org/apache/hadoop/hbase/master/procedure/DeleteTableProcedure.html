058@InterfaceAudience.Private
059public class DeleteTableProcedure
060    extends AbstractStateMachineTableProcedure<DeleteTableState> {
061  private static final Logger LOG = LoggerFactory.getLogger(DeleteTableProcedure.class);
062
063  private List<RegionInfo> regions;
064  private TableName tableName;
065
066  public DeleteTableProcedure() {
067    // Required by the Procedure framework to create the procedure on replay
068    super();
069  }

071  public DeleteTableProcedure(final MasterProcedureEnv env, final TableName tableName) {
072    this(env, tableName, null);
073  }

075  public DeleteTableProcedure(final MasterProcedureEnv env, final TableName tableName,
076      final ProcedurePrepareLatch syncLatch) {
077    super(env, syncLatch);
078    this.tableName = tableName;
079  }

081  @Override
082  protected Flow executeFromState(final MasterProcedureEnv env, DeleteTableState state)
083      throws InterruptedException {
084    if (LOG.isTraceEnabled()) {
085      LOG.trace(this + " execute state=" + state);
086    }
087    try {
088      switch (state) {
089        case DELETE_TABLE_PRE_OPERATION:
090          // Verify if we can delete the table
091          boolean deletable = prepareDelete(env);
092          releaseSyncLatch();
093          if (!deletable) {
094            assert isFailed() : "the delete should have an exception here";
095            return Flow.NO_MORE_STATE;
096          }
097
098          // TODO: Move out... in the acquireLock()
099          LOG.debug("Waiting for RIT for {}", this);
100          regions = env.getAssignmentManager().getRegionStates().getRegionsOfTable(getTableName());
101          assert regions != null && !regions.isEmpty() : "unexpected 0 regions";
102          ProcedureSyncWait.waitRegionInTransition(env, regions);
103
104          // Call coprocessors
105          preDelete(env);
106
107          setNextState(DeleteTableState.DELETE_TABLE_REMOVE_FROM_META);
108          break;
109        case DELETE_TABLE_REMOVE_FROM_META:
110          LOG.debug("Deleting regions from META for {}", this);
111          DeleteTableProcedure.deleteFromMeta(env, getTableName(), regions);
112          setNextState(DeleteTableState.DELETE_TABLE_CLEAR_FS_LAYOUT);
113          break;
114        case DELETE_TABLE_CLEAR_FS_LAYOUT:
115          LOG.debug("Deleting regions from filesystem for {}", this);
116          DeleteTableProcedure.deleteFromFs(env, getTableName(), regions, true);
117          setNextState(DeleteTableState.DELETE_TABLE_UPDATE_DESC_CACHE);
118          regions = null;
119          break;
120        case DELETE_TABLE_UPDATE_DESC_CACHE:
121          LOG.debug("Deleting descriptor for {}", this);
122          DeleteTableProcedure.deleteTableDescriptorCache(env, getTableName());
123          setNextState(DeleteTableState.DELETE_TABLE_UNASSIGN_REGIONS);
124          break;
125        case DELETE_TABLE_UNASSIGN_REGIONS:
126          LOG.debug("Deleting assignment state for {}", this);
127          DeleteTableProcedure.deleteAssignmentState(env, getTableName());
128          setNextState(DeleteTableState.DELETE_TABLE_POST_OPERATION);
129          break;
130        case DELETE_TABLE_POST_OPERATION:
131          postDelete(env);
132          LOG.debug("Finished {}", this);
133          return Flow.NO_MORE_STATE;
134        default:
135          throw new UnsupportedOperationException("unhandled state=" + state);
136      }
137    } catch (IOException e) {
138      if (isRollbackSupported(state)) {
139        setFailure("master-delete-table", e);
140      } else {
141        LOG.warn("Retriable error trying to delete table=" + getTableName() + " state=" + state, e);
142      }
143    }
144    return Flow.HAS_MORE_STATE;
145  }

147  @Override
148  protected boolean abort(MasterProcedureEnv env) {
149    // TODO: Current behavior is: with no rollback and no abort support, procedure may get stuck
150    // looping in retrying failing a step forever. Default behavior of abort is changed to support
151    // aborting all procedures. Override the default wisely. Following code retains the current
152    // behavior. Revisit it later.
153    return isRollbackSupported(getCurrentState()) ? super.abort(env) : false;
154  }
155
156  @Override
157  protected void rollbackState(final MasterProcedureEnv env, final DeleteTableState state) {
158    if (state == DeleteTableState.DELETE_TABLE_PRE_OPERATION) {
159      // nothing to rollback, pre-delete is just table-state checks.
160      // We can fail if the table does not exist or is not disabled.
161      // TODO: coprocessor rollback semantic is still undefined.
162      releaseSyncLatch();
163      return;
164    }
165
166    // The delete doesn't have a rollback. The execution will succeed, at some point.
167    throw new UnsupportedOperationException("unhandled state=" + state);
168  }
169
170  @Override
171  protected boolean isRollbackSupported(final DeleteTableState state) {
172    switch (state) {
173      case DELETE_TABLE_PRE_OPERATION:
174        return true;
175      default:
176        return false;
177    }
178  }

210  @Override
211  protected void serializeStateData(ProcedureStateSerializer serializer)
212      throws IOException {
213    super.serializeStateData(serializer);
214
215    MasterProcedureProtos.DeleteTableStateData.Builder state =
216      MasterProcedureProtos.DeleteTableStateData.newBuilder()
217        .setUserInfo(MasterProcedureUtil.toProtoUserInfo(getUser()))
218        .setTableName(ProtobufUtil.toProtoTableName(tableName));
219    if (regions != null) {
220      for (RegionInfo hri: regions) {
221        state.addRegionInfo(ProtobufUtil.toRegionInfo(hri));
222      }
223    }
224    serializer.serialize(state.build());
225  }
226
227  @Override
228  protected void deserializeStateData(ProcedureStateSerializer serializer)
229      throws IOException {
230    super.deserializeStateData(serializer);
231
232    MasterProcedureProtos.DeleteTableStateData state =
233        serializer.deserialize(MasterProcedureProtos.DeleteTableStateData.class);
234    setUser(MasterProcedureUtil.toUserInfo(state.getUserInfo()));
235    tableName = ProtobufUtil.toTableName(state.getTableName());
236    if (state.getRegionInfoCount() == 0) {
237      regions = null;
238    } else {
239      regions = new ArrayList<>(state.getRegionInfoCount());
240      for (HBaseProtos.RegionInfo hri: state.getRegionInfoList()) {
241        regions.add(ProtobufUtil.toRegionInfo(hri));
242      }
243    }
244  }
245
246  private boolean prepareDelete(final MasterProcedureEnv env) throws IOException {
247    try {
248      env.getMasterServices().checkTableModifiable(tableName);
249    } catch (TableNotFoundException|TableNotDisabledException e) {
250      setFailure("master-delete-table", e);
251      return false;
252    }
253    return true;
254  }
255
256  private boolean preDelete(final MasterProcedureEnv env)
257      throws IOException, InterruptedException {
258    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
259    if (cpHost != null) {
260      final TableName tableName = this.tableName;
261      cpHost.preDeleteTableAction(tableName, getUser());
262    }
263    return true;
264  }
265
266  private void postDelete(final MasterProcedureEnv env)
267      throws IOException, InterruptedException {
268    deleteTableStates(env, tableName);
269
270    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
271    if (cpHost != null) {
272      final TableName tableName = this.tableName;
273      cpHost.postCompletedDeleteTableAction(tableName, getUser());
274    }
275  }

277  protected static void deleteFromFs(final MasterProcedureEnv env,
278      final TableName tableName, final List<RegionInfo> regions,
279      final boolean archive) throws IOException {
280    final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
281    final FileSystem fs = mfs.getFileSystem();
282    final Path tempdir = mfs.getTempDir();
283
284    final Path tableDir = FSUtils.getTableDir(mfs.getRootDir(), tableName);
285    final Path tempTableDir = FSUtils.getTableDir(tempdir, tableName);
286
287    if (fs.exists(tableDir)) {
288      // Ensure temp exists
289      if (!fs.exists(tempdir) && !fs.mkdirs(tempdir)) {
290        throw new IOException("HBase temp directory '" + tempdir + "' creation failure.");
291      }
292
293      // Ensure parent exists
294      if (!fs.exists(tempTableDir.getParent()) && !fs.mkdirs(tempTableDir.getParent())) {
295        throw new IOException("HBase temp directory '" + tempdir + "' creation failure.");
296      }
297
298      if (fs.exists(tempTableDir)) {
299        // TODO
300        // what's in this dir? something old? probably something manual from the user...
301        // let's get rid of this stuff...
302        FileStatus[] files = fs.listStatus(tempTableDir);
303        if (files != null && files.length > 0) {
304          List<Path> regionDirList = Arrays.stream(files)
305            .filter(FileStatus::isDirectory)
306            .map(FileStatus::getPath)
307            .collect(Collectors.toList());
308          HFileArchiver.archiveRegions(env.getMasterConfiguration(), fs, mfs.getRootDir(),
309            tempTableDir, regionDirList);
310        }
311        fs.delete(tempTableDir, true);
312      }
313
314      // Move the table in /hbase/.tmp
315      if (!fs.rename(tableDir, tempTableDir)) {
316        throw new IOException("Unable to move '" + tableDir + "' to temp '" + tempTableDir + "'");
317      }
318    }
319
320    // Archive regions from FS (temp directory)
321    if (archive) {
322      List<Path> regionDirList = regions.stream().filter(RegionReplicaUtil::isDefaultReplica)
323        .map(region -> FSUtils.getRegionDirFromTableDir(tempTableDir, region))
324        .collect(Collectors.toList());
325      HFileArchiver.archiveRegions(env.getMasterConfiguration(), fs, mfs.getRootDir(), tempTableDir,
326        regionDirList);
327      LOG.debug("Table '{}' archived!", tableName);
328    }
329
330    // Archive mob data
331    Path mobTableDir = FSUtils.getTableDir(new Path(mfs.getRootDir(), MobConstants.MOB_DIR_NAME),
332            tableName);
333    Path regionDir =
334            new Path(mobTableDir, MobUtils.getMobRegionInfo(tableName).getEncodedName());
335    if (fs.exists(regionDir)) {
336      HFileArchiver.archiveRegion(fs, mfs.getRootDir(), mobTableDir, regionDir);
337    }
338
339    // Delete table directory from FS (temp directory)
340    if (!fs.delete(tempTableDir, true) && fs.exists(tempTableDir)) {
341      throw new IOException("Couldn't delete " + tempTableDir);
342    }
343
344    // Delete the table directory where the mob files are saved
345    if (mobTableDir != null && fs.exists(mobTableDir)) {
346      if (!fs.delete(mobTableDir, true)) {
347        throw new IOException("Couldn't delete mob dir " + mobTableDir);
348      }
349    }
350
351    // Delete the directory on wal filesystem
352    FileSystem walFs = mfs.getWALFileSystem();
353    Path tableWALDir = FSUtils.getWALTableDir(env.getMasterConfiguration(), tableName);
354    if (walFs.exists(tableWALDir) && !walFs.delete(tableWALDir, true)) {
355      throw new IOException("Couldn't delete table dir on wal filesystem" + tableWALDir);
356    }
357  }

359  /**
360   * There may be items for this table still up in hbase:meta in the case where the
361   * info:regioninfo column was empty because of some write error. Remove ALL rows from hbase:meta
362   * that have to do with this table. See HBASE-12980.
363   */
364  private static void cleanAnyRemainingRows(final MasterProcedureEnv env,
365      final TableName tableName) throws IOException {
366    Connection connection = env.getMasterServices().getConnection();
367    Scan tableScan = MetaTableAccessor.getScanForTableName(connection, tableName);
368    try (Table metaTable = connection.getTable(TableName.META_TABLE_NAME)) {
369      List<Delete> deletes = new ArrayList<>();
370      try (ResultScanner resScanner = metaTable.getScanner(tableScan)) {
371        for (Result result : resScanner) {
372          deletes.add(new Delete(result.getRow()));
373        }
374      }
375      if (!deletes.isEmpty()) {
376        LOG.warn("Deleting some vestigial " + deletes.size() + " rows of " + tableName +
377          " from " + TableName.META_TABLE_NAME);
378        metaTable.delete(deletes);
379      }
380    }
381  }
382
383  protected static void deleteFromMeta(final MasterProcedureEnv env,
384      final TableName tableName, List<RegionInfo> regions) throws IOException {
385    MetaTableAccessor.deleteRegions(env.getMasterServices().getConnection(), regions);
386
387    // Clean any remaining rows for this table.
388    cleanAnyRemainingRows(env, tableName);
389
390    // clean region references from the server manager
391    env.getMasterServices().getServerManager().removeRegions(regions);
392
393    // Clear Favored Nodes for this table
394    FavoredNodesManager fnm = env.getMasterServices().getFavoredNodesManager();
395    if (fnm != null) {
396      fnm.deleteFavoredNodesForRegions(regions);
397    }
398  }
399
400  protected static void deleteAssignmentState(final MasterProcedureEnv env,
401      final TableName tableName) throws IOException {
402    // Clean up regions of the table in RegionStates.
403    LOG.debug("Removing '" + tableName + "' from region states.");
404    env.getMasterServices().getAssignmentManager().deleteTable(tableName);
405
406    // If entry for this table states, remove it.
407    LOG.debug("Marking '" + tableName + "' as deleted.");
408    env.getMasterServices().getTableStateManager().setDeletedTable(tableName);
409  }
410
411  protected static void deleteTableDescriptorCache(final MasterProcedureEnv env,
412      final TableName tableName) throws IOException {
413    LOG.debug("Removing '" + tableName + "' descriptor.");
414    env.getMasterServices().getTableDescriptors().remove(tableName);
415  }
416
417  protected static void deleteTableStates(final MasterProcedureEnv env, final TableName tableName)
418      throws IOException {
419    if (!tableName.isSystemTable()) {
420      ProcedureSyncWait.getMasterQuotaManager(env).removeTableFromNamespaceQuota(tableName);
421    }
422  }
