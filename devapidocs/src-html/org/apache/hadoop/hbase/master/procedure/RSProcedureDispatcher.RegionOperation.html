057/**
058 * A remote procecdure dispatcher for regionservers.
059 */
060@InterfaceAudience.Private
061public class RSProcedureDispatcher
062    extends RemoteProcedureDispatcher<MasterProcedureEnv, ServerName>
063    implements ServerListener {
064  private static final Logger LOG = LoggerFactory.getLogger(RSProcedureDispatcher.class);
065
066  public static final String RS_RPC_STARTUP_WAIT_TIME_CONF_KEY =
067      "hbase.regionserver.rpc.startup.waittime";
068  private static final int DEFAULT_RS_RPC_STARTUP_WAIT_TIME = 60000;
069
070  private static final int RS_VERSION_WITH_EXEC_PROCS = 0x0200000; // 2.0
071
072  protected final MasterServices master;
073  private final long rsStartupWaitTime;
074  private MasterProcedureEnv procedureEnv;
075
076  public RSProcedureDispatcher(final MasterServices master) {
077    super(master.getConfiguration());
078
079    this.master = master;
080    this.rsStartupWaitTime = master.getConfiguration().getLong(
081      RS_RPC_STARTUP_WAIT_TIME_CONF_KEY, DEFAULT_RS_RPC_STARTUP_WAIT_TIME);
082  }
083
084  @Override
085  public boolean start() {
086    if (!super.start()) {
087      return false;
088    }
089
090    master.getServerManager().registerListener(this);
091    procedureEnv = master.getMasterProcedureExecutor().getEnvironment();
092    for (ServerName serverName: master.getServerManager().getOnlineServersList()) {
093      addNode(serverName);
094    }
095    return true;
096  }
097
098  @Override
099  public boolean stop() {
100    if (!super.stop()) {
101      return false;
102    }
103
104    master.getServerManager().unregisterListener(this);
105    return true;
106  }
107
108  @Override
109  protected void remoteDispatch(final ServerName serverName,
110      final Set<RemoteProcedure> remoteProcedures) {
111    final int rsVersion = master.getServerManager().getVersionNumber(serverName);
112    if (rsVersion >= RS_VERSION_WITH_EXEC_PROCS) {
113      LOG.trace("Using procedure batch rpc execution for serverName={} version={}", serverName,
114        rsVersion);
115      submitTask(new ExecuteProceduresRemoteCall(serverName, remoteProcedures));
116    } else if (rsVersion == 0 && !master.getServerManager().isServerOnline(serverName)) {
117      submitTask(new DeadRSRemoteCall(serverName, remoteProcedures));
118    } else {
119      LOG.info(String.format(
120        "Fallback to compat rpc execution for serverName=%s version=%s",
121        serverName, rsVersion));
122      submitTask(new CompatRemoteProcedureResolver(serverName, remoteProcedures));
123    }
124  }
125
126  @Override
127  protected void abortPendingOperations(final ServerName serverName,
128      final Set<RemoteProcedure> operations) {
129    // TODO: Replace with a ServerNotOnlineException()
130    final IOException e = new DoNotRetryIOException("server not online " + serverName);
131    for (RemoteProcedure proc: operations) {
132      proc.remoteCallFailed(procedureEnv, serverName, e);
133    }
134  }
135
136  @Override
137  public void serverAdded(final ServerName serverName) {
138    addNode(serverName);
139  }
140
141  @Override
142  public void serverRemoved(final ServerName serverName) {
143    removeNode(serverName);
144  }

148   */
149  protected abstract class AbstractRSRemoteCall implements Callable<Void> {
150    @Override
151    public abstract Void call();
152
153    private final ServerName serverName;
154
155    private int numberOfAttemptsSoFar = 0;
156    private long maxWaitTime = -1;
157
158    public AbstractRSRemoteCall(final ServerName serverName) {
159      this.serverName = serverName;
160    }
161
162    protected AdminService.BlockingInterface getRsAdmin() throws IOException {
163      final AdminService.BlockingInterface admin = master.getServerManager().getRsAdmin(serverName);
164      if (admin == null) {
165        throw new IOException("Attempting to send OPEN RPC to server " + getServerName() +
166          " failed because no RPC connection found to this server");
167      }
168      return admin;
169    }
170
171    protected ServerName getServerName() {
172      return serverName;
173    }
174
175    protected boolean scheduleForRetry(final IOException e) {
176      // Should we wait a little before retrying? If the server is starting it's yes.
177      final boolean hold = (e instanceof ServerNotRunningYetException);
178      if (hold) {
179        LOG.warn(String.format("waiting a little before trying on the same server=%s try=%d",
180            serverName, numberOfAttemptsSoFar), e);
181        long now = EnvironmentEdgeManager.currentTime();
182        if (now < getMaxWaitTime()) {
183          if (LOG.isDebugEnabled()) {
184            LOG.debug(String.format("server is not yet up; waiting up to %dms",
185              (getMaxWaitTime() - now)), e);
186          }
187          submitTask(this, 100, TimeUnit.MILLISECONDS);
188          return true;
189        }
190
191        LOG.warn(String.format("server %s is not up for a while; try a new one", serverName), e);
192        return false;
193      }
194
195      // In case socket is timed out and the region server is still online,
196      // the openRegion RPC could have been accepted by the server and
197      // just the response didn't go through.  So we will retry to
198      // open the region on the same server.
199      final boolean retry = !hold && (e instanceof SocketTimeoutException
200          && master.getServerManager().isServerOnline(serverName));
201      if (retry) {
202        // we want to retry as many times as needed as long as the RS is not dead.
203        if (LOG.isDebugEnabled()) {
204          LOG.debug(String.format("Retrying to same RegionServer %s because: %s",
205              serverName, e.getMessage()), e);
206        }
207        submitTask(this);
208        return true;
209      }
210
211      // trying to send the request elsewhere instead
212      LOG.warn(String.format("Failed dispatch to server=%s try=%d",
213                  serverName, numberOfAttemptsSoFar), e);
214      return false;
215    }
216
217    private long getMaxWaitTime() {
218      if (this.maxWaitTime < 0) {
219        // This is the max attempts, not retries, so it should be at least 1.
220        this.maxWaitTime = EnvironmentEdgeManager.currentTime() + rsStartupWaitTime;
221      }
222      return this.maxWaitTime;
223    }
224
225    protected IOException unwrapException(IOException e) {
226      if (e instanceof RemoteException) {
227        e = ((RemoteException)e).unwrapRemoteException();
228      }
229      return e;
230    }
231  }

233  private interface RemoteProcedureResolver {
234    void dispatchOpenRequests(MasterProcedureEnv env, List<RegionOpenOperation> operations);
235
236    void dispatchCloseRequests(MasterProcedureEnv env, List<RegionCloseOperation> operations);
237
238    void dispatchServerOperations(MasterProcedureEnv env, List<ServerOperation> operations);
239  }
240
241  /**
242   * Fetches {@link org.apache.hadoop.hbase.procedure2.RemoteProcedureDispatcher.RemoteOperation}s
243   * from the given {@code remoteProcedures} and groups them by class of the returned operation.
244   * Then {@code resolver} is used to dispatch {@link RegionOpenOperation}s and
245   * {@link RegionCloseOperation}s.
246   * @param serverName RegionServer to which the remote operations are sent
247   * @param operations Remote procedures which are dispatched to the given server
248   * @param resolver Used to dispatch remote procedures to given server.
249   */
250  public void splitAndResolveOperation(ServerName serverName, Set<RemoteProcedure> operations,
251      RemoteProcedureResolver resolver) {
252    MasterProcedureEnv env = master.getMasterProcedureExecutor().getEnvironment();
253    ArrayListMultimap<Class<?>, RemoteOperation> reqsByType =
254      buildAndGroupRequestByType(env, serverName, operations);
255
256    List<RegionOpenOperation> openOps = fetchType(reqsByType, RegionOpenOperation.class);
257    if (!openOps.isEmpty()) {
258      resolver.dispatchOpenRequests(env, openOps);
259    }
260
261    List<RegionCloseOperation> closeOps = fetchType(reqsByType, RegionCloseOperation.class);
262    if (!closeOps.isEmpty()) {
263      resolver.dispatchCloseRequests(env, closeOps);
264    }
265
266    List<ServerOperation> refreshOps = fetchType(reqsByType, ServerOperation.class);
267    if (!refreshOps.isEmpty()) {
268      resolver.dispatchServerOperations(env, refreshOps);
269    }
270
271    if (!reqsByType.isEmpty()) {
272      LOG.warn("unknown request type in the queue: " + reqsByType);
273    }
274  }

276  private class DeadRSRemoteCall extends ExecuteProceduresRemoteCall {
277
278    public DeadRSRemoteCall(ServerName serverName, Set<RemoteProcedure> remoteProcedures) {
279      super(serverName, remoteProcedures);
280    }
281
282    @Override
283    public Void call() {
284      remoteCallFailed(procedureEnv,
285        new RegionServerStoppedException("Server " + getServerName() + " is not online"));
286      return null;
287    }
288  }

293  protected class ExecuteProceduresRemoteCall extends AbstractRSRemoteCall
294      implements RemoteProcedureResolver {
295    protected final Set<RemoteProcedure> remoteProcedures;
296
297    private ExecuteProceduresRequest.Builder request = null;
298
299    public ExecuteProceduresRemoteCall(final ServerName serverName,
300        final Set<RemoteProcedure> remoteProcedures) {
301      super(serverName);
302      this.remoteProcedures = remoteProcedures;
303    }
304
305    @Override
306    public Void call() {
307      request = ExecuteProceduresRequest.newBuilder();
308      if (LOG.isTraceEnabled()) {
309        LOG.trace("Building request with operations count=" + remoteProcedures.size());
310      }
311      splitAndResolveOperation(getServerName(), remoteProcedures, this);
312
313      try {
314        sendRequest(getServerName(), request.build());
315      } catch (IOException e) {
316        e = unwrapException(e);
317        // TODO: In the future some operation may want to bail out early.
318        // TODO: How many times should we retry (use numberOfAttemptsSoFar)
319        if (!scheduleForRetry(e)) {
320          remoteCallFailed(procedureEnv, e);
321        }
322      }
323      return null;
324    }
325
326    @Override
327    public void dispatchOpenRequests(final MasterProcedureEnv env,
328        final List<RegionOpenOperation> operations) {
329      request.addOpenRegion(buildOpenRegionRequest(env, getServerName(), operations));
330    }
331
332    @Override
333    public void dispatchCloseRequests(final MasterProcedureEnv env,
334        final List<RegionCloseOperation> operations) {
335      for (RegionCloseOperation op: operations) {
336        request.addCloseRegion(op.buildCloseRegionRequest(getServerName()));
337      }
338    }
339
340    @Override
341    public void dispatchServerOperations(MasterProcedureEnv env, List<ServerOperation> operations) {
342      operations.stream().map(o -> o.buildRequest()).forEachOrdered(request::addProc);
343    }
344
345    protected ExecuteProceduresResponse sendRequest(final ServerName serverName,
346        final ExecuteProceduresRequest request) throws IOException {
347      try {
348        return getRsAdmin().executeProcedures(null, request);
349      } catch (ServiceException se) {
350        throw ProtobufUtil.getRemoteException(se);
351      }
352    }
353
354    protected void remoteCallFailed(final MasterProcedureEnv env, final IOException e) {
355      for (RemoteProcedure proc : remoteProcedures) {
356        proc.remoteCallFailed(env, getServerName(), e);
357      }
358    }
359  }

361  private static OpenRegionRequest buildOpenRegionRequest(final MasterProcedureEnv env,
362      final ServerName serverName, final List<RegionOpenOperation> operations) {
363    final OpenRegionRequest.Builder builder = OpenRegionRequest.newBuilder();
364    builder.setServerStartCode(serverName.getStartcode());
365    builder.setMasterSystemTime(EnvironmentEdgeManager.currentTime());
366    for (RegionOpenOperation op: operations) {
367      builder.addOpenInfo(op.buildRegionOpenInfoRequest(env));
368    }
369    return builder.build();
370  }

377  /**
378   * Compatibility class used by {@link CompatRemoteProcedureResolver} to open regions using old
379   * {@link AdminService#openRegion(RpcController, OpenRegionRequest, RpcCallback)} rpc.
380   */
381  private final class OpenRegionRemoteCall extends AbstractRSRemoteCall {
382    private final List<RegionOpenOperation> operations;
383
384    public OpenRegionRemoteCall(final ServerName serverName,
385        final List<RegionOpenOperation> operations) {
386      super(serverName);
387      this.operations = operations;
388    }
389
390    @Override
391    public Void call() {
392      final OpenRegionRequest request =
393          buildOpenRegionRequest(procedureEnv, getServerName(), operations);
394
395      try {
396        sendRequest(getServerName(), request);
397      } catch (IOException e) {
398        e = unwrapException(e);
399        // TODO: In the future some operation may want to bail out early.
400        // TODO: How many times should we retry (use numberOfAttemptsSoFar)
401        if (!scheduleForRetry(e)) {
402          remoteCallFailed(procedureEnv, e);
403        }
404      }
405      return null;
406    }
407
408    private OpenRegionResponse sendRequest(final ServerName serverName,
409        final OpenRegionRequest request) throws IOException {
410      try {
411        return getRsAdmin().openRegion(null, request);
412      } catch (ServiceException se) {
413        throw ProtobufUtil.getRemoteException(se);
414      }
415    }
416
417    private void remoteCallFailed(final MasterProcedureEnv env, final IOException e) {
418      for (RegionOpenOperation op: operations) {
419        op.getRemoteProcedure().remoteCallFailed(env, getServerName(), e);
420      }
421    }
422  }

424  /**
425   * Compatibility class used by {@link CompatRemoteProcedureResolver} to close regions using old
426   * {@link AdminService#closeRegion(RpcController, CloseRegionRequest, RpcCallback)} rpc.
427   */
428  private final class CloseRegionRemoteCall extends AbstractRSRemoteCall {
429    private final RegionCloseOperation operation;
430
431    public CloseRegionRemoteCall(final ServerName serverName,
432        final RegionCloseOperation operation) {
433      super(serverName);
434      this.operation = operation;
435    }
436
437    @Override
438    public Void call() {
439      final CloseRegionRequest request = operation.buildCloseRegionRequest(getServerName());
440      try {
441        CloseRegionResponse response = sendRequest(getServerName(), request);
442        remoteCallCompleted(procedureEnv, response);
443      } catch (IOException e) {
444        e = unwrapException(e);
445        // TODO: In the future some operation may want to bail out early.
446        // TODO: How many times should we retry (use numberOfAttemptsSoFar)
447        if (!scheduleForRetry(e)) {
448          remoteCallFailed(procedureEnv, e);
449        }
450      }
451      return null;
452    }
453
454    private CloseRegionResponse sendRequest(final ServerName serverName,
455        final CloseRegionRequest request) throws IOException {
456      try {
457        return getRsAdmin().closeRegion(null, request);
458      } catch (ServiceException se) {
459        throw ProtobufUtil.getRemoteException(se);
460      }
461    }
462
463    private void remoteCallCompleted(final MasterProcedureEnv env,
464        final CloseRegionResponse response) {
465      operation.setClosed(response.getClosed());
466    }
467
468    private void remoteCallFailed(final MasterProcedureEnv env, final IOException e) {
469      operation.getRemoteProcedure().remoteCallFailed(env, getServerName(), e);
470    }
471  }

473  /**
474   * Compatibility class to open and close regions using old endpoints (openRegion/closeRegion) in
475   * {@link AdminService}.
476   */
477  protected class CompatRemoteProcedureResolver implements Callable<Void>, RemoteProcedureResolver {
478    private final Set<RemoteProcedure> operations;
479    private final ServerName serverName;
480
481    public CompatRemoteProcedureResolver(final ServerName serverName,
482        final Set<RemoteProcedure> operations) {
483      this.serverName = serverName;
484      this.operations = operations;
485    }
486
487    @Override
488    public Void call() {
489      splitAndResolveOperation(serverName, operations, this);
490      return null;
491    }
492
493    @Override
494    public void dispatchOpenRequests(final MasterProcedureEnv env,
495        final List<RegionOpenOperation> operations) {
496      submitTask(new OpenRegionRemoteCall(serverName, operations));
497    }
498
499    @Override
500    public void dispatchCloseRequests(final MasterProcedureEnv env,
501        final List<RegionCloseOperation> operations) {
502      for (RegionCloseOperation op: operations) {
503        submitTask(new CloseRegionRemoteCall(serverName, op));
504      }
505    }
506
507    @Override
508    public void dispatchServerOperations(MasterProcedureEnv env, List<ServerOperation> operations) {
509      throw new UnsupportedOperationException();
510    }
511  }

519  public static final class ServerOperation extends RemoteOperation {
520
521    private final long procId;
522
523    private final Class<?> rsProcClass;
524
525    private final byte[] rsProcData;
526
527    public ServerOperation(RemoteProcedure remoteProcedure, long procId, Class<?> rsProcClass,
528        byte[] rsProcData) {
529      super(remoteProcedure);
530      this.procId = procId;
531      this.rsProcClass = rsProcClass;
532      this.rsProcData = rsProcData;
533    }
534
535    public RemoteProcedureRequest buildRequest() {
536      return RemoteProcedureRequest.newBuilder().setProcId(procId)
537          .setProcClass(rsProcClass.getName()).setProcData(ByteString.copyFrom(rsProcData)).build();
538    }
539  }
