018package org.apache.hadoop.hbase.master.procedure;
019
020import java.io.IOException;
021import java.util.Collections;
022import java.util.List;
023import java.util.stream.Collectors;
024import org.apache.hadoop.hbase.HRegionLocation;
025import org.apache.hadoop.hbase.TableName;
026import org.apache.hadoop.hbase.master.assignment.RegionStateNode;
027import org.apache.hadoop.hbase.master.assignment.TransitRegionStateProcedure;
028import org.apache.hadoop.hbase.procedure2.ProcedureStateSerializer;
029import org.apache.hadoop.hbase.procedure2.ProcedureSuspendedException;
030import org.apache.hadoop.hbase.procedure2.ProcedureUtil;
031import org.apache.hadoop.hbase.procedure2.ProcedureYieldException;
032import org.apache.yetus.audience.InterfaceAudience;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
037import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.ReopenTableRegionsState;
038import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.ReopenTableRegionsStateData;
039import org.apache.hadoop.hbase.shaded.protobuf.generated.ProcedureProtos;
040
041/**
042 * Used for reopening the regions for a table.
043 */
044@InterfaceAudience.Private
045public class ReopenTableRegionsProcedure
046    extends AbstractStateMachineTableProcedure<ReopenTableRegionsState> {
047
048  private static final Logger LOG = LoggerFactory.getLogger(ReopenTableRegionsProcedure.class);
049
050  private TableName tableName;
051
052  private List<HRegionLocation> regions = Collections.emptyList();
053
054  private int attempt;
055
056  public ReopenTableRegionsProcedure() {
057  }
058
059  public ReopenTableRegionsProcedure(TableName tableName) {
060    this.tableName = tableName;
061  }
062
063  @Override
064  public TableName getTableName() {
065    return tableName;
066  }
067
068  @Override
069  public TableOperationType getTableOperationType() {
070    return TableOperationType.REGION_EDIT;
071  }
072
073  @Override
074  protected Flow executeFromState(MasterProcedureEnv env, ReopenTableRegionsState state)
075      throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
076    switch (state) {
077      case REOPEN_TABLE_REGIONS_GET_REGIONS:
078        if (!env.getAssignmentManager().isTableEnabled(tableName)) {
079          LOG.info("Table {} is disabled, give up reopening its regions");
080          return Flow.NO_MORE_STATE;
081        }
082        regions =
083          env.getAssignmentManager().getRegionStates().getRegionsOfTableForReopen(tableName);
084        setNextState(ReopenTableRegionsState.REOPEN_TABLE_REGIONS_REOPEN_REGIONS);
085        return Flow.HAS_MORE_STATE;
086      case REOPEN_TABLE_REGIONS_REOPEN_REGIONS:
087        for (HRegionLocation loc : regions) {
088          RegionStateNode regionNode = env.getAssignmentManager().getRegionStates()
089            .getOrCreateRegionStateNode(loc.getRegion());
090          TransitRegionStateProcedure proc;
091          regionNode.lock();
092          try {
093            if (regionNode.getProcedure() != null) {
094              continue;
095            }
096            proc = TransitRegionStateProcedure.reopen(env, regionNode.getRegionInfo());
097            regionNode.setProcedure(proc);
098          } finally {
099            regionNode.unlock();
100          }
101          addChildProcedure(proc);
102        }
103        setNextState(ReopenTableRegionsState.REOPEN_TABLE_REGIONS_CONFIRM_REOPENED);
104        return Flow.HAS_MORE_STATE;
105      case REOPEN_TABLE_REGIONS_CONFIRM_REOPENED:
106        regions = regions.stream().map(env.getAssignmentManager().getRegionStates()::checkReopened)
107          .filter(l -> l != null).collect(Collectors.toList());
108        if (regions.isEmpty()) {
109          return Flow.NO_MORE_STATE;
110        }
111        if (regions.stream().anyMatch(l -> l.getSeqNum() >= 0)) {
112          attempt = 0;
113          setNextState(ReopenTableRegionsState.REOPEN_TABLE_REGIONS_REOPEN_REGIONS);
114          return Flow.HAS_MORE_STATE;
115        }
116        // All the regions need to reopen are in OPENING state which means we can not schedule any
117        // MRPs.
118        long backoff = ProcedureUtil.getBackoffTimeMs(this.attempt++);
119        LOG.info(
120          "There are still {} region(s) which need to be reopened for table {} are in " +
121            "OPENING state, suspend {}secs and try again later",
122          regions.size(), tableName, backoff / 1000);
123        setTimeout(Math.toIntExact(backoff));
124        setState(ProcedureProtos.ProcedureState.WAITING_TIMEOUT);
125        skipPersistence();
126        throw new ProcedureSuspendedException();
127      default:
128        throw new UnsupportedOperationException("unhandled state=" + state);
129    }
130  }
131
132  /**
133   * At end of timeout, wake ourselves up so we run again.
134   */
135  @Override
136  protected synchronized boolean setTimeoutFailure(MasterProcedureEnv env) {
137    setState(ProcedureProtos.ProcedureState.RUNNABLE);
138    env.getProcedureScheduler().addFront(this);
139    return false; // 'false' means that this procedure handled the timeout
140  }
141  @Override
142  protected void rollbackState(MasterProcedureEnv env, ReopenTableRegionsState state)
143      throws IOException, InterruptedException {
144    throw new UnsupportedOperationException("unhandled state=" + state);
145  }
146
147  @Override
148  protected ReopenTableRegionsState getState(int stateId) {
149    return ReopenTableRegionsState.forNumber(stateId);
150  }
151
152  @Override
153  protected int getStateId(ReopenTableRegionsState state) {
154    return state.getNumber();
155  }
156
157  @Override
158  protected ReopenTableRegionsState getInitialState() {
159    return ReopenTableRegionsState.REOPEN_TABLE_REGIONS_GET_REGIONS;
160  }
161
162  @Override
163  protected void serializeStateData(ProcedureStateSerializer serializer) throws IOException {
164    super.serializeStateData(serializer);
165    ReopenTableRegionsStateData.Builder builder = ReopenTableRegionsStateData.newBuilder()
166      .setTableName(ProtobufUtil.toProtoTableName(tableName));
167    regions.stream().map(ProtobufUtil::toRegionLocation).forEachOrdered(builder::addRegion);
168    serializer.serialize(builder.build());
169  }
170
171  @Override
172  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
173    super.deserializeStateData(serializer);
174    ReopenTableRegionsStateData data = serializer.deserialize(ReopenTableRegionsStateData.class);
175    tableName = ProtobufUtil.toTableName(data.getTableName());
176    regions = data.getRegionList().stream().map(ProtobufUtil::toRegionLocation)
177      .collect(Collectors.toList());
178  }
179}
