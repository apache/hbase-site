103  public RestoreSnapshotProcedure(
104      final MasterProcedureEnv env,
105      final TableDescriptor tableDescriptor,
106      final SnapshotDescription snapshot,
107      final boolean restoreAcl)
108  throws HBaseIOException {
109    super(env);
110    // This is the new schema we are going to write out as this modification.
111    this.modifiedTableDescriptor = tableDescriptor;
112    preflightChecks(env, null/*Table can be online when restore is called?*/);
113    // Snapshot information
114    this.snapshot = snapshot;
115    this.restoreAcl = restoreAcl;
116
117    // Monitor
118    getMonitorStatus();
119  }

121  /**
122   * Set up monitor status if it is not created.
123   */
124  private MonitoredTask getMonitorStatus() {
125    if (monitorStatus == null) {
126      monitorStatus = TaskMonitor.get().createStatus("Restoring  snapshot '" + snapshot.getName()
127        + "' to table " + getTableName());

132  @Override
133  protected Flow executeFromState(final MasterProcedureEnv env, final RestoreSnapshotState state)
134      throws InterruptedException {
135    LOG.trace("{} execute state={}", this, state);
136
137    // Make sure that the monitor status is set up
138    getMonitorStatus();
139
140    try {
141      switch (state) {
142        case RESTORE_SNAPSHOT_PRE_OPERATION:
143          // Verify if we can restore the table
144          prepareRestore(env);
145          setNextState(RestoreSnapshotState.RESTORE_SNAPSHOT_UPDATE_TABLE_DESCRIPTOR);
146          break;
147        case RESTORE_SNAPSHOT_UPDATE_TABLE_DESCRIPTOR:
148          updateTableDescriptor(env);
149          setNextState(RestoreSnapshotState.RESTORE_SNAPSHOT_WRITE_FS_LAYOUT);
150          break;
151        case RESTORE_SNAPSHOT_WRITE_FS_LAYOUT:
152          restoreSnapshot(env);
153          setNextState(RestoreSnapshotState.RESTORE_SNAPSHOT_UPDATE_META);
154          break;
155        case RESTORE_SNAPSHOT_UPDATE_META:
156          updateMETA(env);
157          setNextState(RestoreSnapshotState.RESTORE_SNAPSHOT_RESTORE_ACL);
158          break;
159        case RESTORE_SNAPSHOT_RESTORE_ACL:
160          restoreSnapshotAcl(env);
161          return Flow.NO_MORE_STATE;
162        default:
163          throw new UnsupportedOperationException("unhandled state=" + state);
164      }
165    } catch (IOException e) {
166      if (isRollbackSupported(state)) {
167        setFailure("master-restore-snapshot", e);
168      } else {
169        LOG.warn("Retriable error trying to restore snapshot=" + snapshot.getName() +
170          " to table=" + getTableName() + " (in state=" + state + ")", e);
171      }
172    }
173    return Flow.HAS_MORE_STATE;
174  }
175
176  @Override
177  protected void rollbackState(final MasterProcedureEnv env, final RestoreSnapshotState state)
178      throws IOException {
179    if (state == RestoreSnapshotState.RESTORE_SNAPSHOT_PRE_OPERATION) {
180      // nothing to rollback
181      return;
182    }
183
184    // The restore snapshot doesn't have a rollback. The execution will succeed, at some point.
185    throw new UnsupportedOperationException("unhandled state=" + state);
186  }
187
188  @Override
189  protected boolean isRollbackSupported(final RestoreSnapshotState state) {
190    switch (state) {
191      case RESTORE_SNAPSHOT_PRE_OPERATION:
192        return true;
193      default:
194        return false;
195    }
196  }

239  @Override
240  protected void serializeStateData(ProcedureStateSerializer serializer)
241      throws IOException {
242    super.serializeStateData(serializer);
243
244    RestoreSnapshotStateData.Builder restoreSnapshotMsg = RestoreSnapshotStateData.newBuilder()
245      .setUserInfo(MasterProcedureUtil.toProtoUserInfo(getUser()))
246      .setSnapshot(this.snapshot)
247      .setModifiedTableSchema(ProtobufUtil.toTableSchema(modifiedTableDescriptor));
248
249    if (regionsToRestore != null) {
250      for (RegionInfo hri: regionsToRestore) {
251        restoreSnapshotMsg.addRegionInfoForRestore(ProtobufUtil.toRegionInfo(hri));
252      }
253    }
254    if (regionsToRemove != null) {
255      for (RegionInfo hri: regionsToRemove) {
256        restoreSnapshotMsg.addRegionInfoForRemove(ProtobufUtil.toRegionInfo(hri));
257      }
258    }
259    if (regionsToAdd != null) {
260      for (RegionInfo hri: regionsToAdd) {
261        restoreSnapshotMsg.addRegionInfoForAdd(ProtobufUtil.toRegionInfo(hri));
262      }
263    }
264    if (!parentsToChildrenPairMap.isEmpty()) {
265      final Iterator<Map.Entry<String, Pair<String, String>>> it =
266        parentsToChildrenPairMap.entrySet().iterator();
267      while (it.hasNext()) {
268        final Map.Entry<String, Pair<String, String>> entry = it.next();
269
270        RestoreParentToChildRegionsPair.Builder parentToChildrenPair =
271          RestoreParentToChildRegionsPair.newBuilder()
272            .setParentRegionName(entry.getKey())
273            .setChild1RegionName(entry.getValue().getFirst())
274            .setChild2RegionName(entry.getValue().getSecond());
275        restoreSnapshotMsg.addParentToChildRegionsPairList (parentToChildrenPair);
276      }
277    }
278    restoreSnapshotMsg.setRestoreAcl(restoreAcl);
279    serializer.serialize(restoreSnapshotMsg.build());
280  }

282  @Override
283  protected void deserializeStateData(ProcedureStateSerializer serializer)
284      throws IOException {
285    super.deserializeStateData(serializer);
286
287    RestoreSnapshotStateData restoreSnapshotMsg =
288        serializer.deserialize(RestoreSnapshotStateData.class);
289    setUser(MasterProcedureUtil.toUserInfo(restoreSnapshotMsg.getUserInfo()));
290    snapshot = restoreSnapshotMsg.getSnapshot();
291    modifiedTableDescriptor =
292      ProtobufUtil.toTableDescriptor(restoreSnapshotMsg.getModifiedTableSchema());
293
294    if (restoreSnapshotMsg.getRegionInfoForRestoreCount() == 0) {
295      regionsToRestore = null;
296    } else {
297      regionsToRestore = new ArrayList<>(restoreSnapshotMsg.getRegionInfoForRestoreCount());
298      for (HBaseProtos.RegionInfo hri: restoreSnapshotMsg.getRegionInfoForRestoreList()) {
299        regionsToRestore.add(ProtobufUtil.toRegionInfo(hri));
300      }
301    }
302    if (restoreSnapshotMsg.getRegionInfoForRemoveCount() == 0) {
303      regionsToRemove = null;
304    } else {
305      regionsToRemove = new ArrayList<>(restoreSnapshotMsg.getRegionInfoForRemoveCount());
306      for (HBaseProtos.RegionInfo hri: restoreSnapshotMsg.getRegionInfoForRemoveList()) {
307        regionsToRemove.add(ProtobufUtil.toRegionInfo(hri));
308      }
309    }
310    if (restoreSnapshotMsg.getRegionInfoForAddCount() == 0) {
311      regionsToAdd = null;
312    } else {
313      regionsToAdd = new ArrayList<>(restoreSnapshotMsg.getRegionInfoForAddCount());
314      for (HBaseProtos.RegionInfo hri: restoreSnapshotMsg.getRegionInfoForAddList()) {
315        regionsToAdd.add(ProtobufUtil.toRegionInfo(hri));
316      }
317    }
318    if (restoreSnapshotMsg.getParentToChildRegionsPairListCount() > 0) {
319      for (RestoreParentToChildRegionsPair parentToChildrenPair : restoreSnapshotMsg
320        .getParentToChildRegionsPairListList()) {
321        parentsToChildrenPairMap.put(
322          parentToChildrenPair.getParentRegionName(),
323          new Pair<>(
324            parentToChildrenPair.getChild1RegionName(),
325            parentToChildrenPair.getChild2RegionName()));
326      }
327    }
328    if (restoreSnapshotMsg.hasRestoreAcl()) {
329      restoreAcl = restoreSnapshotMsg.getRestoreAcl();
330    }
331  }

333  /**
334   * Action before any real action of restoring from snapshot.
335   * @param env MasterProcedureEnv
336   * @throws IOException
337   */
338  private void prepareRestore(final MasterProcedureEnv env) throws IOException {
339    final TableName tableName = getTableName();
340    // Checks whether the table exists
341    if (!env.getMasterServices().getTableDescriptors().exists(tableName)) {
342      throw new TableNotFoundException(tableName);
343    }
344
345    // Check whether table is disabled.
346    env.getMasterServices().checkTableModifiable(tableName);
347
348    // Check that we have at least 1 CF
349    if (modifiedTableDescriptor.getColumnFamilyCount() == 0) {
350      throw new DoNotRetryIOException("Table " + getTableName().toString() +
351        " should have at least one column family.");
352    }
353
354    if (!getTableName().isSystemTable()) {
355      // Table already exist. Check and update the region quota for this table namespace.
356      final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
357      SnapshotManifest manifest = SnapshotManifest.open(
358        env.getMasterConfiguration(),
359        mfs.getFileSystem(),
360        SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshot, mfs.getRootDir()),
361        snapshot);
362      int snapshotRegionCount = manifest.getRegionManifestsMap().size();
363      int tableRegionCount =
364          ProcedureSyncWait.getMasterQuotaManager(env).getRegionCountOfTable(tableName);
365
366      if (snapshotRegionCount > 0 && tableRegionCount != snapshotRegionCount) {
367        ProcedureSyncWait.getMasterQuotaManager(env).checkAndUpdateNamespaceRegionQuota(
368          tableName, snapshotRegionCount);
369      }
370    }
371  }

382  /**
383   * Execute the on-disk Restore
384   * @param env MasterProcedureEnv
385   * @throws IOException
386   **/
387  private void restoreSnapshot(final MasterProcedureEnv env) throws IOException {
388    MasterFileSystem fileSystemManager = env.getMasterServices().getMasterFileSystem();
389    FileSystem fs = fileSystemManager.getFileSystem();
390    Path rootDir = fileSystemManager.getRootDir();
391    final ForeignExceptionDispatcher monitorException = new ForeignExceptionDispatcher();
392    final Configuration conf = new Configuration(env.getMasterConfiguration());
393
394    LOG.info("Starting restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot));
395    try {
396      Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshot, rootDir);
397      SnapshotManifest manifest = SnapshotManifest.open(
398        conf, fs, snapshotDir, snapshot);
399      RestoreSnapshotHelper restoreHelper = new RestoreSnapshotHelper(
400        conf,
401        fs,
402        manifest,
403        modifiedTableDescriptor,
404        rootDir,
405        monitorException,
406        getMonitorStatus());
407
408      RestoreSnapshotHelper.RestoreMetaChanges metaChanges = restoreHelper.restoreHdfsRegions();
409      regionsToRestore = metaChanges.getRegionsToRestore();
410      regionsToRemove = metaChanges.getRegionsToRemove();
411      regionsToAdd = metaChanges.getRegionsToAdd();
412      parentsToChildrenPairMap = metaChanges.getParentToChildrenPairMap();
413    } catch (IOException e) {
414      String msg = "restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot)
415        + " failed in on-disk restore. Try re-running the restore command.";
416      LOG.error(msg, e);
417      monitorException.receive(
418        new ForeignException(env.getMasterServices().getServerName().toString(), e));
419      throw new IOException(msg, e);
420    }
421  }

423  /**
424   * Apply changes to hbase:meta
425   **/
426  private void updateMETA(final MasterProcedureEnv env) throws IOException {
427    try {
428      Connection conn = env.getMasterServices().getConnection();
429      RegionStateStore regionStateStore = env.getAssignmentManager().getRegionStateStore();
430      int regionReplication = modifiedTableDescriptor.getRegionReplication();
431
432      // 1. Prepare to restore
433      getMonitorStatus().setStatus("Preparing to restore each region");
434
435      // 2. Applies changes to hbase:meta and in-memory states
436      // (2.1). Removes the current set of regions from META and in-memory states
437      //
438      // By removing also the regions to restore (the ones present both in the snapshot
439      // and in the current state) we ensure that no extra fields are present in META
440      // e.g. with a simple add addRegionToMeta() the splitA and splitB attributes
441      // not overwritten/removed, so you end up with old informations
442      // that are not correct after the restore.
443      if (regionsToRemove != null) {
444        regionStateStore.deleteRegions(regionsToRemove);
445        deleteRegionsFromInMemoryStates(regionsToRemove, env, regionReplication);
446      }
447
448      // (2.2). Add the new set of regions to META and in-memory states
449      //
450      // At this point the old regions are no longer present in META.
451      // and the set of regions present in the snapshot will be written to META.
452      // All the information in hbase:meta are coming from the .regioninfo of each region present
453      // in the snapshot folder.
454      if (regionsToAdd != null) {
455        MetaTableAccessor.addRegionsToMeta(conn, regionsToAdd, regionReplication);
456        addRegionsToInMemoryStates(regionsToAdd, env, regionReplication);
457      }
458
459      if (regionsToRestore != null) {
460        regionStateStore.overwriteRegions(regionsToRestore, regionReplication);
461
462        deleteRegionsFromInMemoryStates(regionsToRestore, env, regionReplication);
463        addRegionsToInMemoryStates(regionsToRestore, env, regionReplication);
464      }
465
466      RestoreSnapshotHelper.RestoreMetaChanges metaChanges =
467        new RestoreSnapshotHelper.RestoreMetaChanges(
468                modifiedTableDescriptor, parentsToChildrenPairMap);
469      metaChanges.updateMetaParentRegions(conn, regionsToAdd);
470
471      // At this point the restore is complete.
472      LOG.info("Restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot) +
473        " on table=" + getTableName() + " completed!");
474    } catch (IOException e) {
475      final ForeignExceptionDispatcher monitorException = new ForeignExceptionDispatcher();
476      String msg = "restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot)
477          + " failed in meta update. Try re-running the restore command.";
478      LOG.error(msg, e);
479      monitorException.receive(
480        new ForeignException(env.getMasterServices().getServerName().toString(), e));
481      throw new IOException(msg, e);
482    }
483
484    monitorStatus.markComplete("Restore snapshot '"+ snapshot.getName() +"'!");
485    MetricsSnapshot metricsSnapshot = new MetricsSnapshot();
486    metricsSnapshot.addSnapshotRestore(
487      monitorStatus.getCompletionTimestamp() - monitorStatus.getStartTime());
488  }

490  /**
491   * Delete regions from in-memory states
492   * @param regionInfos regions to delete
493   * @param env MasterProcedureEnv
494   * @param regionReplication the number of region replications
495   */
496  private void deleteRegionsFromInMemoryStates(List<RegionInfo> regionInfos,
497      MasterProcedureEnv env, int regionReplication) {
498    FavoredNodesManager fnm = env.getMasterServices().getFavoredNodesManager();
499
500    env.getAssignmentManager().getRegionStates().deleteRegions(regionInfos);
501    env.getMasterServices().getServerManager().removeRegions(regionInfos);
502    if (fnm != null) {
503      fnm.deleteFavoredNodesForRegions(regionInfos);
504    }
505
506    // For region replicas
507    if (regionReplication > 1) {
508      for (RegionInfo regionInfo : regionInfos) {
509        for (int i = 1; i < regionReplication; i++) {
510          RegionInfo regionInfoForReplica =
511              RegionReplicaUtil.getRegionInfoForReplica(regionInfo, i);
512          env.getAssignmentManager().getRegionStates().deleteRegion(regionInfoForReplica);
513          env.getMasterServices().getServerManager().removeRegion(regionInfoForReplica);
514          if (fnm != null) {
515            fnm.deleteFavoredNodesForRegion(regionInfoForReplica);
516          }
517        }
518      }
519    }
520  }

522  /**
523   * Add regions to in-memory states
524   * @param regionInfos regions to add
525   * @param env MasterProcedureEnv
526   * @param regionReplication the number of region replications
527   */
528  private void addRegionsToInMemoryStates(List<RegionInfo> regionInfos, MasterProcedureEnv env,
529      int regionReplication) {
530    AssignmentManager am = env.getAssignmentManager();
531    for (RegionInfo regionInfo : regionInfos) {
532      if (regionInfo.isSplit()) {
533        am.getRegionStates().updateRegionState(regionInfo, RegionState.State.SPLIT);
534      } else {
535        am.getRegionStates().updateRegionState(regionInfo, RegionState.State.CLOSED);
536
537        // For region replicas
538        for (int i = 1; i < regionReplication; i++) {
539          RegionInfo regionInfoForReplica =
540              RegionReplicaUtil.getRegionInfoForReplica(regionInfo, i);
541          am.getRegionStates().updateRegionState(regionInfoForReplica, RegionState.State.CLOSED);
542        }
543      }
544    }
545  }

547  private void restoreSnapshotAcl(final MasterProcedureEnv env) throws IOException {
548    if (restoreAcl && snapshot.hasUsersAndPermissions() && snapshot.getUsersAndPermissions() != null
549        && SnapshotDescriptionUtils
550            .isSecurityAvailable(env.getMasterServices().getConfiguration())) {
551      // restore acl of snapshot to table.
552      RestoreSnapshotHelper.restoreSnapshotAcl(snapshot, TableName.valueOf(snapshot.getTable()),
553        env.getMasterServices().getConfiguration());
554    }
555  }
