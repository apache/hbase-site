056/**
057 * ProcedureScheduler for the Master Procedures.
058 * This ProcedureScheduler tries to provide to the ProcedureExecutor procedures
059 * that can be executed without having to wait on a lock.
060 * Most of the master operations can be executed concurrently, if they
061 * are operating on different tables (e.g. two create table procedures can be performed
062 * at the same time) or against two different servers; say two servers that crashed at
063 * about the same time.
064 *
065 * <p>Each procedure should implement an Interface providing information for this queue.
066 * For example table related procedures should implement TableProcedureInterface.
067 * Each procedure will be pushed in its own queue, and based on the operation type
068 * we may make smarter decisions: e.g. we can abort all the operations preceding
069 * a delete table, or similar.
070 *
071 * <h4>Concurrency control</h4>
072 * Concurrent access to member variables (tableRunQueue, serverRunQueue, locking, tableMap,
073 * serverBuckets) is controlled by schedLock(). This mainly includes:<br>
074 * <ul>
075 *   <li>
076 *     {@link #push(Procedure, boolean, boolean)}: A push will add a Queue back to run-queue
077 *     when:
078 *     <ol>
079 *       <li>Queue was empty before push (so must have been out of run-queue)</li>
080 *       <li>Child procedure is added (which means parent procedure holds exclusive lock, and it
081 *           must have moved Queue out of run-queue)</li>
082 *     </ol>
083 *   </li>
084 *   <li>
085 *     {@link #poll(long)}: A poll will remove a Queue from run-queue when:
086 *     <ol>
087 *       <li>Queue becomes empty after poll</li>
088 *       <li>Exclusive lock is requested by polled procedure and lock is available (returns the
089 *           procedure)</li>
090 *       <li>Exclusive lock is requested but lock is not available (returns null)</li>
091 *       <li>Polled procedure is child of parent holding exclusive lock and the next procedure is
092 *           not a child</li>
093 *     </ol>
094 *   </li>
095 *   <li>
096 *     Namespace/table/region locks: Queue is added back to run-queue when lock being released is:
097 *     <ol>
098 *       <li>Exclusive lock</li>
099 *       <li>Last shared lock (in case queue was removed because next procedure in queue required
100 *           exclusive lock)</li>
101 *     </ol>
102 *   </li>
103 * </ul>
104 */

157  @Override
158  protected void enqueue(final Procedure proc, final boolean addFront) {
159    if (isTableProcedure(proc)) {
160      doAdd(tableRunQueue, getTableQueue(getTableName(proc)), proc, addFront);
161    } else if (isServerProcedure(proc)) {
162      doAdd(serverRunQueue, getServerQueue(getServerName(proc)), proc, addFront);
163    } else {
164      // TODO: at the moment we only have Table and Server procedures
165      // if you are implementing a non-table/non-server procedure, you have two options: create
166      // a group for all the non-table/non-server procedures or try to find a key for your
167      // non-table/non-server procedures and implement something similar to the TableRunQueue.
168      throw new UnsupportedOperationException(
169        "RQs for non-table/non-server procedures are not implemented yet: " + proc);
170    }
171  }

173  private <T extends Comparable<T>> void doAdd(final FairQueue<T> fairq,
174      final Queue<T> queue, final Procedure proc, final boolean addFront) {
175    queue.add(proc, addFront);
176    if (!queue.getLockStatus().hasExclusiveLock() || queue.getLockStatus().isLockOwner(proc.getProcId())) {
177      // if the queue was not remove for an xlock execution
178      // or the proc is the lock owner, put the queue back into execution
179      addToRunQueue(fairq, queue);
180    } else if (queue.getLockStatus().hasParentLock(proc)) {
181      assert addFront : "expected to add a child in the front";
182      // our (proc) parent has the xlock,
183      // so the queue is not in the fairq (run-queue)
184      // add it back to let the child run (inherit the lock)
185      addToRunQueue(fairq, queue);
186    }
187  }

194  @Override
195  protected Procedure dequeue() {
196    // For now, let server handling have precedence over table handling; presumption is that it
197    // is more important handling crashed servers than it is running the
198    // enabling/disabling tables, etc.
199    Procedure pollResult = doPoll(serverRunQueue);
200    if (pollResult == null) {
201      pollResult = doPoll(tableRunQueue);
202    }
203    return pollResult;
204  }
205
206  private <T extends Comparable<T>> Procedure doPoll(final FairQueue<T> fairq) {
207    final Queue<T> rq = fairq.poll();
208    if (rq == null || !rq.isAvailable()) {
209      return null;
210    }
211
212    final Procedure pollResult = rq.peek();
213    final boolean xlockReq = rq.requireExclusiveLock(pollResult);
214    if (xlockReq && rq.getLockStatus().isLocked() && !rq.getLockStatus().hasLockAccess(pollResult)) {
215      // someone is already holding the lock (e.g. shared lock). avoid a yield
216      removeFromRunQueue(fairq, rq);
217      return null;
218    }
219
220    rq.poll();
221    if (rq.isEmpty() || xlockReq) {
222      removeFromRunQueue(fairq, rq);
223    } else if (rq.getLockStatus().hasParentLock(pollResult)) {
224      // if the rq is in the fairq because of runnable child
225      // check if the next procedure is still a child.
226      // if not, remove the rq from the fairq and go back to the xlock state
227      Procedure nextProc = rq.peek();
228      if (nextProc != null && !Procedure.haveSameParent(nextProc, pollResult)) {
229        removeFromRunQueue(fairq, rq);
230      }
231    }
232
233    return pollResult;
234  }

236  private LockInfo createLockInfo(LockInfo.ResourceType resourceType,
237      String resourceName, LockAndQueue queue) {
238    LockInfo info = new LockInfo();
239
240    info.setResourceType(resourceType);
241    info.setResourceName(resourceName);
242
243    if (queue.hasExclusiveLock()) {
244      info.setLockType(LockInfo.LockType.EXCLUSIVE);
245
246      Procedure<?> exclusiveLockOwnerProcedure = queue.getExclusiveLockOwnerProcedure();
247      ProcedureInfo exclusiveLockOwnerProcedureInfo =
248          ProcedureUtil.convertToProcedureInfo(exclusiveLockOwnerProcedure);
249      info.setExclusiveLockOwnerProcedure(exclusiveLockOwnerProcedureInfo);
250    } else if (queue.getSharedLockCount() > 0) {
251      info.setLockType(LockInfo.LockType.SHARED);
252      info.setSharedLockCount(queue.getSharedLockCount());
253    }
254
255    for (Procedure<?> procedure : queue) {
256      if (!(procedure instanceof LockProcedure)) {
257        continue;
258      }
259
260      LockProcedure lockProcedure = (LockProcedure)procedure;
261      LockInfo.WaitingProcedure waitingProcedure = new LockInfo.WaitingProcedure();
262
263      switch (lockProcedure.getType()) {
264      case EXCLUSIVE:
265        waitingProcedure.setLockType(LockInfo.LockType.EXCLUSIVE);
266        break;
267      case SHARED:
268        waitingProcedure.setLockType(LockInfo.LockType.SHARED);
269        break;
270      }
271
272      ProcedureInfo procedureInfo = ProcedureUtil.convertToProcedureInfo(lockProcedure);
273      waitingProcedure.setProcedure(procedureInfo);
274
275      info.addWaitingProcedure(waitingProcedure);
276    }
277
278    return info;
279  }

281  @Override
282  public List<LockInfo> listLocks() {
283    schedLock();
284
285    try {
286      List<LockInfo> lockInfos = new ArrayList<>();
287
288      for (Entry<ServerName, LockAndQueue> entry : locking.serverLocks
289          .entrySet()) {
290        String serverName = entry.getKey().getServerName();
291        LockAndQueue queue = entry.getValue();
292
293        if (queue.isLocked()) {
294          LockInfo lockInfo = createLockInfo(LockInfo.ResourceType.SERVER,
295              serverName, queue);
296          lockInfos.add(lockInfo);
297        }
298      }
299
300      for (Entry<String, LockAndQueue> entry : locking.namespaceLocks
301          .entrySet()) {
302        String namespaceName = entry.getKey();
303        LockAndQueue queue = entry.getValue();
304
305        if (queue.isLocked()) {
306          LockInfo lockInfo = createLockInfo(LockInfo.ResourceType.NAMESPACE,
307              namespaceName, queue);
308          lockInfos.add(lockInfo);
309        }
310      }
311
312      for (Entry<TableName, LockAndQueue> entry : locking.tableLocks
313          .entrySet()) {
314        String tableName = entry.getKey().getNameAsString();
315        LockAndQueue queue = entry.getValue();
316
317        if (queue.isLocked()) {
318          LockInfo lockInfo = createLockInfo(LockInfo.ResourceType.TABLE,
319              tableName, queue);
320          lockInfos.add(lockInfo);
321        }
322      }
323
324      for (Entry<String, LockAndQueue> entry : locking.regionLocks.entrySet()) {
325        String regionName = entry.getKey();
326        LockAndQueue queue = entry.getValue();
327
328        if (queue.isLocked()) {
329          LockInfo lockInfo = createLockInfo(LockInfo.ResourceType.REGION,
330              regionName, queue);
331          lockInfos.add(lockInfo);
332        }
333      }
334
335      return lockInfos;
336    } finally {
337      schedUnlock();
338    }
339  }

341  @Override
342  public LockInfo getLockInfoForResource(LockInfo.ResourceType resourceType, String resourceName) {
343    LockAndQueue queue = null;
344    schedLock();
345    try {
346      switch (resourceType) {
347        case SERVER:
348          queue = locking.serverLocks.get(ServerName.valueOf(resourceName));
349          break;
350        case NAMESPACE:
351          queue = locking.namespaceLocks.get(resourceName);
352          break;
353        case TABLE:
354          queue = locking.tableLocks.get(TableName.valueOf(resourceName));
355          break;
356        case REGION:
357          queue = locking.regionLocks.get(resourceName);
358          break;
359      }
360
361      return queue != null ? createLockInfo(resourceType, resourceName, queue) : null;
362    } finally {
363      schedUnlock();
364    }
365  }

369    schedLock();

371      clearQueue();
372      locking.clear();

374      schedUnlock();
375    }

378  protected void clearQueue() {
379    // Remove Servers
380    for (int i = 0; i < serverBuckets.length; ++i) {
381      clear(serverBuckets[i], serverRunQueue, SERVER_QUEUE_KEY_COMPARATOR);
382      serverBuckets[i] = null;
383    }
384
385    // Remove Tables
386    clear(tableMap, tableRunQueue, TABLE_QUEUE_KEY_COMPARATOR);
387    tableMap = null;
388
389    assert size() == 0 : "expected queue size to be 0, got " + size();
390  }
391
392  private <T extends Comparable<T>, TNode extends Queue<T>> void clear(TNode treeMap,
393      final FairQueue<T> fairq, final AvlKeyComparator<TNode> comparator) {
394    while (treeMap != null) {
395      Queue<T> node = AvlTree.getFirst(treeMap);
396      treeMap = AvlTree.remove(treeMap, node.getKey(), comparator);
397      if (fairq != null) removeFromRunQueue(fairq, node);
398    }
399  }

401  @Override
402  protected int queueSize() {
403    int count = 0;
404
405    // Server queues
406    final AvlTreeIterator<ServerQueue> serverIter = new AvlTreeIterator<>();
407    for (int i = 0; i < serverBuckets.length; ++i) {
408      serverIter.seekFirst(serverBuckets[i]);
409      while (serverIter.hasNext()) {
410        count += serverIter.next().size();
411      }
412    }
413
414    // Table queues
415    final AvlTreeIterator<TableQueue> tableIter = new AvlTreeIterator<>(tableMap);
416    while (tableIter.hasNext()) {
417      count += tableIter.next().size();
418    }
419    return count;
420  }

422  @Override
423  public void completionCleanup(final Procedure proc) {
424    if (proc instanceof TableProcedureInterface) {
425      TableProcedureInterface iProcTable = (TableProcedureInterface)proc;
426      boolean tableDeleted;
427      if (proc.hasException()) {
428        Exception procEx = proc.getException().unwrapRemoteException();
429        if (iProcTable.getTableOperationType() == TableOperationType.CREATE) {
430          // create failed because the table already exist
431          tableDeleted = !(procEx instanceof TableExistsException);
432        } else {
433          // the operation failed because the table does not exist
434          tableDeleted = (procEx instanceof TableNotFoundException);
435        }
436      } else {
437        // the table was deleted
438        tableDeleted = (iProcTable.getTableOperationType() == TableOperationType.DELETE);
439      }
440      if (tableDeleted) {
441        markTableAsDeleted(iProcTable.getTableName(), proc);
442        return;
443      }
444    } else {
445      // No cleanup for ServerProcedureInterface types, yet.
446      return;
447    }
448  }

450  private static <T extends Comparable<T>> void addToRunQueue(FairQueue<T> fairq, Queue<T> queue) {
451    if (!AvlIterableList.isLinked(queue) && !queue.isEmpty()) {
452      fairq.add(queue);
453    }
454  }
455
456  private static <T extends Comparable<T>> void removeFromRunQueue(
457      FairQueue<T> fairq, Queue<T> queue) {
458    if (AvlIterableList.isLinked(queue)) {
459      fairq.remove(queue);
460    }
461  }

466  private TableQueue getTableQueue(TableName tableName) {
467    TableQueue node = AvlTree.get(tableMap, tableName, TABLE_QUEUE_KEY_COMPARATOR);
468    if (node != null) return node;
469
470    node = new TableQueue(tableName, tablePriorities.getPriority(tableName),
471        locking.getTableLock(tableName), locking.getNamespaceLock(tableName.getNamespaceAsString()));
472    tableMap = AvlTree.insert(tableMap, node);
473    return node;
474  }

476  private void removeTableQueue(TableName tableName) {
477    tableMap = AvlTree.remove(tableMap, tableName, TABLE_QUEUE_KEY_COMPARATOR);
478    locking.removeTableLock(tableName);
479  }

493  private ServerQueue getServerQueue(ServerName serverName) {
494    final int index = getBucketIndex(serverBuckets, serverName.hashCode());
495    ServerQueue node = AvlTree.get(serverBuckets[index], serverName, SERVER_QUEUE_KEY_COMPARATOR);
496    if (node != null) return node;
497
498    node = new ServerQueue(serverName, locking.getServerLock(serverName));
499    serverBuckets[index] = AvlTree.insert(serverBuckets[index], node);
500    return node;
501  }

550  public static class TableQueue extends Queue<TableName> {
551    private final LockStatus namespaceLockStatus;
552
553    public TableQueue(TableName tableName, int priority, LockStatus tableLock,
554        LockStatus namespaceLockStatus) {
555      super(tableName, priority, tableLock);
556      this.namespaceLockStatus = namespaceLockStatus;
557    }
558
559    @Override
560    public boolean isAvailable() {
561      // if there are no items in the queue, or the namespace is locked.
562      // we can't execute operation on this table
563      if (isEmpty() || namespaceLockStatus.hasExclusiveLock()) {
564        return false;
565      }
566
567      if (getLockStatus().hasExclusiveLock()) {
568        // if we have an exclusive lock already taken
569        // only child of the lock owner can be executed
570        final Procedure nextProc = peek();
571        return nextProc != null && getLockStatus().hasLockAccess(nextProc);
572      }
573
574      // no xlock
575      return true;
576    }
577
578    @Override
579    public boolean requireExclusiveLock(Procedure proc) {
580      return requireTableExclusiveLock((TableProcedureInterface)proc);
581    }
582  }

585  //  Table Locking Helpers

587  private static boolean requireTableExclusiveLock(TableProcedureInterface proc) {
588    switch (proc.getTableOperationType()) {
589      case CREATE:
590      case DELETE:
591      case DISABLE:
592      case ENABLE:
593        return true;
594      case EDIT:
595        // we allow concurrent edit on the NS table
596        return !proc.getTableName().equals(TableName.NAMESPACE_TABLE_NAME);
597      case READ:
598        return false;
599      // region operations are using the shared-lock on the table
600      // and then they will grab an xlock on the region.
601      case REGION_SPLIT:
602      case REGION_MERGE:
603      case REGION_ASSIGN:
604      case REGION_UNASSIGN:
605      case REGION_EDIT:
606      case REGION_GC:
607      case MERGED_REGIONS_GC:
608        return false;
609      default:
610        break;
611    }
612    throw new UnsupportedOperationException("unexpected type " +
613        proc.getTableOperationType());
614  }

616  /**
617   * Get lock info for a resource of specified type and name and log details
618   */
619  protected void logLockInfoForResource(LockInfo.ResourceType resourceType, String resourceName) {
620    if (!LOG.isDebugEnabled()) {
621      return;
622    }
623
624    LockInfo lockInfo = getLockInfoForResource(resourceType, resourceName);
625    if (lockInfo != null) {
626      String msg = resourceType.toString() + " '" + resourceName + "', shared lock count=" +
627          lockInfo.getSharedLockCount();
628
629      ProcedureInfo proc = lockInfo.getExclusiveLockOwnerProcedure();
630      if (proc != null) {
631        msg += ", exclusively locked by procId=" + proc.getProcId();
632      }
633      LOG.debug(msg);
634    }

637  /**
638   * Suspend the procedure if the specified table is already locked.
639   * Other operations in the table-queue will be executed after the lock is released.
640   * @param procedure the procedure trying to acquire the lock
641   * @param table Table to lock
642   * @return true if the procedure has to wait for the table to be available
643   */
644  public boolean waitTableExclusiveLock(final Procedure procedure, final TableName table) {
645    schedLock();
646    try {
647      final String namespace = table.getNamespaceAsString();
648      final LockAndQueue namespaceLock = locking.getNamespaceLock(namespace);
649      final LockAndQueue tableLock = locking.getTableLock(table);
650      if (!namespaceLock.trySharedLock()) {
651        waitProcedure(namespaceLock, procedure);
652        logLockInfoForResource(LockInfo.ResourceType.NAMESPACE, namespace);
653        return true;
654      }
655      if (!tableLock.tryExclusiveLock(procedure)) {
656        namespaceLock.releaseSharedLock();
657        waitProcedure(tableLock, procedure);
658        logLockInfoForResource(LockInfo.ResourceType.TABLE, table.getNameAsString());
659        return true;
660      }
661      removeFromRunQueue(tableRunQueue, getTableQueue(table));
662      return false;
663    } finally {
664      schedUnlock();
665    }
666  }

668  /**
669   * Wake the procedures waiting for the specified table
670   * @param procedure the procedure releasing the lock
671   * @param table the name of the table that has the exclusive lock
672   */
673  public void wakeTableExclusiveLock(final Procedure procedure, final TableName table) {
674    schedLock();
675    try {
676      final LockAndQueue namespaceLock = locking.getNamespaceLock(table.getNamespaceAsString());
677      final LockAndQueue tableLock = locking.getTableLock(table);
678      int waitingCount = 0;
679
680      if (!tableLock.hasParentLock(procedure)) {
681        tableLock.releaseExclusiveLock(procedure);
682        waitingCount += wakeWaitingProcedures(tableLock);
683      }
684      if (namespaceLock.releaseSharedLock()) {
685        waitingCount += wakeWaitingProcedures(namespaceLock);
686      }
687      addToRunQueue(tableRunQueue, getTableQueue(table));
688      wakePollIfNeeded(waitingCount);
689    } finally {
690      schedUnlock();
691    }
692  }

694  /**
695   * Suspend the procedure if the specified table is already locked.
696   * other "read" operations in the table-queue may be executed concurrently,
697   * @param procedure the procedure trying to acquire the lock
698   * @param table Table to lock
699   * @return true if the procedure has to wait for the table to be available
700   */
701  public boolean waitTableSharedLock(final Procedure procedure, final TableName table) {
702    return waitTableQueueSharedLock(procedure, table) == null;
703  }
704
705  private TableQueue waitTableQueueSharedLock(final Procedure procedure, final TableName table) {
706    schedLock();
707    try {
708      final LockAndQueue namespaceLock = locking.getNamespaceLock(table.getNamespaceAsString());
709      final LockAndQueue tableLock = locking.getTableLock(table);
710      if (!namespaceLock.trySharedLock()) {
711        waitProcedure(namespaceLock, procedure);
712        return null;
713      }
714
715      if (!tableLock.trySharedLock()) {
716        namespaceLock.releaseSharedLock();
717        waitProcedure(tableLock, procedure);
718        return null;
719      }
720
721      return getTableQueue(table);
722    } finally {
723      schedUnlock();
724    }
725  }

727  /**
728   * Wake the procedures waiting for the specified table
729   * @param procedure the procedure releasing the lock
730   * @param table the name of the table that has the shared lock
731   */
732  public void wakeTableSharedLock(final Procedure procedure, final TableName table) {
733    schedLock();
734    try {
735      final LockAndQueue namespaceLock = locking.getNamespaceLock(table.getNamespaceAsString());
736      final LockAndQueue tableLock = locking.getTableLock(table);
737      int waitingCount = 0;
738      if (tableLock.releaseSharedLock()) {
739        addToRunQueue(tableRunQueue, getTableQueue(table));
740        waitingCount += wakeWaitingProcedures(tableLock);
741      }
742      if (namespaceLock.releaseSharedLock()) {
743        waitingCount += wakeWaitingProcedures(namespaceLock);
744      }
745      wakePollIfNeeded(waitingCount);
746    } finally {
747      schedUnlock();
748    }
749  }

751  /**
752   * Tries to remove the queue and the table-lock of the specified table.
753   * If there are new operations pending (e.g. a new create),
754   * the remove will not be performed.
755   * @param table the name of the table that should be marked as deleted
756   * @param procedure the procedure that is removing the table
757   * @return true if deletion succeeded, false otherwise meaning that there are
758   *     other new operations pending for that table (e.g. a new create).
759   */
760  @VisibleForTesting
761  protected boolean markTableAsDeleted(final TableName table, final Procedure procedure) {
762    schedLock();
763    try {
764      final TableQueue queue = getTableQueue(table);
765      final LockAndQueue tableLock = locking.getTableLock(table);
766      if (queue == null) return true;
767
768      if (queue.isEmpty() && tableLock.tryExclusiveLock(procedure)) {
769        // remove the table from the run-queue and the map
770        if (AvlIterableList.isLinked(queue)) {
771          tableRunQueue.remove(queue);
772        }
773        removeTableQueue(table);
774      } else {
775        // TODO: If there are no create, we can drop all the other ops
776        return false;
777      }
778    } finally {
779      schedUnlock();
780    }
781    return true;
782  }

797  /**
798   * Suspend the procedure if the specified set of regions are already locked.
799   * @param procedure the procedure trying to acquire the lock on the regions
800   * @param table the table name of the regions we are trying to lock
801   * @param regionInfo the list of regions we are trying to lock
802   * @return true if the procedure has to wait for the regions to be available
803   */
804  public boolean waitRegions(final Procedure procedure, final TableName table,
805      final HRegionInfo... regionInfo) {
806    Arrays.sort(regionInfo);
807    schedLock();
808    try {
809      // If there is parent procedure, it would have already taken xlock, so no need to take
810      // shared lock here. Otherwise, take shared lock.
811      if (!procedure.hasParent()
812          && waitTableQueueSharedLock(procedure, table) == null) {
813          return true;
814      }
815
816      // acquire region xlocks or wait
817      boolean hasLock = true;
818      final LockAndQueue[] regionLocks = new LockAndQueue[regionInfo.length];
819      for (int i = 0; i < regionInfo.length; ++i) {
820        LOG.info(procedure + " " + table + " " + regionInfo[i].getRegionNameAsString());
821        assert table != null;
822        assert regionInfo[i] != null;
823        assert regionInfo[i].getTable() != null;
824        assert regionInfo[i].getTable().equals(table): regionInfo[i] + " " + procedure;
825        assert i == 0 || regionInfo[i] != regionInfo[i - 1] : "duplicate region: " + regionInfo[i];
826
827        regionLocks[i] = locking.getRegionLock(regionInfo[i].getEncodedName());
828        if (!regionLocks[i].tryExclusiveLock(procedure)) {
829          waitProcedure(regionLocks[i], procedure);
830          hasLock = false;
831          while (i-- > 0) {
832            regionLocks[i].releaseExclusiveLock(procedure);
833          }
834          break;
835        }
836      }
837
838      if (!hasLock && !procedure.hasParent()) {
839        wakeTableSharedLock(procedure, table);
840      }
841      return !hasLock;
842    } finally {
843      schedUnlock();
844    }
845  }

849   * @param procedure the procedure that was holding the region

856  /**
857   * Wake the procedures waiting for the specified regions
858   * @param procedure the procedure that was holding the regions
859   * @param regionInfo the list of regions the procedure was holding
860   */
861  public void wakeRegions(final Procedure procedure,final TableName table,
862      final HRegionInfo... regionInfo) {
863    Arrays.sort(regionInfo);
864    schedLock();
865    try {
866      int numProcs = 0;
867      final Procedure[] nextProcs = new Procedure[regionInfo.length];
868      for (int i = 0; i < regionInfo.length; ++i) {
869        assert regionInfo[i].getTable().equals(table);
870        assert i == 0 || regionInfo[i] != regionInfo[i - 1] : "duplicate region: " + regionInfo[i];
871
872        LockAndQueue regionLock = locking.getRegionLock(regionInfo[i].getEncodedName());
873        if (regionLock.releaseExclusiveLock(procedure)) {
874          if (!regionLock.isEmpty()) {
875            // release one procedure at the time since regions has an xlock
876            nextProcs[numProcs++] = regionLock.removeFirst();
877          } else {
878            locking.removeRegionLock(regionInfo[i].getEncodedName());
879          }
880        }
881      }
882
883      // awake procedures if any
884      for (int i = numProcs - 1; i >= 0; --i) {
885        wakeProcedure(nextProcs[i]);
886      }
887      wakePollIfNeeded(numProcs);
888      if (!procedure.hasParent()) {
889        // release the table shared-lock.
890        // (if we have a parent, it is holding an xlock so we didn't take the shared-lock)
891        wakeTableSharedLock(procedure, table);
892      }
893    } finally {
894      schedUnlock();
895    }
896  }

901  /**
902   * Suspend the procedure if the specified namespace is already locked.
903   * @see #wakeNamespaceExclusiveLock(Procedure,String)
904   * @param procedure the procedure trying to acquire the lock
905   * @param namespace Namespace to lock
906   * @return true if the procedure has to wait for the namespace to be available
907   */
908  public boolean waitNamespaceExclusiveLock(final Procedure procedure, final String namespace) {
909    schedLock();
910    try {
911      final LockAndQueue systemNamespaceTableLock =
912          locking.getTableLock(TableName.NAMESPACE_TABLE_NAME);
913      if (!systemNamespaceTableLock.trySharedLock()) {
914        waitProcedure(systemNamespaceTableLock, procedure);
915        logLockInfoForResource(LockInfo.ResourceType.TABLE,
916            TableName.NAMESPACE_TABLE_NAME.getNameAsString());
917        return true;
918      }
919
920      final LockAndQueue namespaceLock = locking.getNamespaceLock(namespace);
921      if (!namespaceLock.tryExclusiveLock(procedure)) {
922        systemNamespaceTableLock.releaseSharedLock();
923        waitProcedure(namespaceLock, procedure);
924        logLockInfoForResource(LockInfo.ResourceType.NAMESPACE, namespace);
925        return true;
926      }
927      return false;
928    } finally {
929      schedUnlock();
930    }
931  }

933  /**
934   * Wake the procedures waiting for the specified namespace
935   * @see #waitNamespaceExclusiveLock(Procedure,String)
936   * @param procedure the procedure releasing the lock
937   * @param namespace the namespace that has the exclusive lock
938   */
939  public void wakeNamespaceExclusiveLock(final Procedure procedure, final String namespace) {
940    schedLock();
941    try {
942      final LockAndQueue namespaceLock = locking.getNamespaceLock(namespace);
943      final LockAndQueue systemNamespaceTableLock =
944          locking.getTableLock(TableName.NAMESPACE_TABLE_NAME);
945      namespaceLock.releaseExclusiveLock(procedure);
946      int waitingCount = 0;
947      if (systemNamespaceTableLock.releaseSharedLock()) {
948        addToRunQueue(tableRunQueue, getTableQueue(TableName.NAMESPACE_TABLE_NAME));
949        waitingCount += wakeWaitingProcedures(systemNamespaceTableLock);
950      }
951      waitingCount += wakeWaitingProcedures(namespaceLock);
952      wakePollIfNeeded(waitingCount);
953    } finally {
954      schedUnlock();
955    }
956  }

963   * @see #wakeServerExclusiveLock(Procedure,ServerName)

972      if (lock.tryExclusiveLock(procedure)) {
973        removeFromRunQueue(serverRunQueue, getServerQueue(serverName));
974        return false;
975      }

977      logLockInfoForResource(LockInfo.ResourceType.SERVER, serverName.getServerName());
978      return true;

1006  private static abstract class Queue<TKey extends Comparable<TKey>>
1007      extends AvlLinkedNode<Queue<TKey>> {
1008    abstract boolean requireExclusiveLock(Procedure proc);
1009
1010    private final TKey key;
1011    private final int priority;
1012    private final ProcedureDeque runnables = new ProcedureDeque();
1013    // Reference to status of lock on entity this queue represents.
1014    private final LockStatus lockStatus;
1015
1016    public Queue(TKey key, LockStatus lockStatus) {
1017      this(key, 1, lockStatus);
1018    }
1019
1020    public Queue(TKey key, int priority, LockStatus lockStatus) {
1021      this.key = key;
1022      this.priority = priority;
1023      this.lockStatus = lockStatus;
1024    }
1025
1026    protected TKey getKey() {
1027      return key;
1028    }
1029
1030    protected int getPriority() {
1031      return priority;
1032    }
1033
1034    protected LockStatus getLockStatus() {
1035      return lockStatus;
1036    }
1037
1038    // This should go away when we have the new AM and its events
1039    // and we move xlock to the lock-event-queue.
1040    public boolean isAvailable() {
1041      return !lockStatus.hasExclusiveLock() && !isEmpty();
1042    }
1043
1044    // ======================================================================
1045    //  Functions to handle procedure queue
1046    // ======================================================================
1047    public void add(final Procedure proc, final boolean addToFront) {
1048      if (addToFront) {
1049        runnables.addFirst(proc);
1050      } else {
1051        runnables.addLast(proc);
1052      }
1053    }
1054
1055    public Procedure peek() {
1056      return runnables.peek();
1057    }
1058
1059    public Procedure poll() {
1060      return runnables.poll();
1061    }
1062
1063    public boolean isEmpty() {
1064      return runnables.isEmpty();
1065    }
1066
1067    public int size() {
1068      return runnables.size();
1069    }
1070
1071    // ======================================================================
1072    //  Generic Helpers
1073    // ======================================================================
1074    public int compareKey(TKey cmpKey) {
1075      return key.compareTo(cmpKey);
1076    }
1077
1078    @Override
1079    public int compareTo(Queue<TKey> other) {
1080      return compareKey(other.key);
1081    }
1082
1083    @Override
1084    public String toString() {
1085      return String.format("%s(%s, xlock=%s sharedLock=%s size=%s)",
1086          getClass().getSimpleName(), key,
1087          lockStatus.hasExclusiveLock() ?
1088              "true (" + lockStatus.getExclusiveLockProcIdOwner() + ")" : "false",
1089          lockStatus.getSharedLockCount(), size());
1090    }
1091  }

1093  /**
1094   * Locks on namespaces, tables, and regions.
1095   * Since LockAndQueue implementation is NOT thread-safe, schedLock() guards all calls to these
1096   * locks.
1097   */
1098  private static class SchemaLocking {
1099    final Map<ServerName, LockAndQueue> serverLocks = new HashMap<>();
1100    final Map<String, LockAndQueue> namespaceLocks = new HashMap<>();
1101    final Map<TableName, LockAndQueue> tableLocks = new HashMap<>();
1102    // Single map for all regions irrespective of tables. Key is encoded region name.
1103    final Map<String, LockAndQueue> regionLocks = new HashMap<>();
1104
1105    private <T> LockAndQueue getLock(Map<T, LockAndQueue> map, T key) {
1106      LockAndQueue lock = map.get(key);
1107      if (lock == null) {
1108        lock = new LockAndQueue();
1109        map.put(key, lock);
1110      }
1111      return lock;
1112    }
1113
1114    LockAndQueue getTableLock(TableName tableName) {
1115      return getLock(tableLocks, tableName);
1116    }
1117
1118    LockAndQueue removeTableLock(TableName tableName) {
1119      return tableLocks.remove(tableName);
1120    }
1121
1122    LockAndQueue getNamespaceLock(String namespace) {
1123      return getLock(namespaceLocks, namespace);
1124    }
1125
1126    LockAndQueue getRegionLock(String encodedRegionName) {
1127      return getLock(regionLocks, encodedRegionName);
1128    }
1129
1130    LockAndQueue removeRegionLock(String encodedRegionName) {
1131      return regionLocks.remove(encodedRegionName);
1132    }
1133
1134    LockAndQueue getServerLock(ServerName serverName) {
1135      return getLock(serverLocks, serverName);
1136    }
1137
1138    /**
1139     * Removes all locks by clearing the maps.
1140     * Used when procedure executor is stopped for failure and recovery testing.
1141     */
1142    @VisibleForTesting
1143    void clear() {
1144      serverLocks.clear();
1145      namespaceLocks.clear();
1146      tableLocks.clear();
1147      regionLocks.clear();
1148    }
1149
1150    @Override
1151    public String toString() {
1152      return "serverLocks=" + filterUnlocked(this.serverLocks) +
1153        ", namespaceLocks=" + filterUnlocked(this.namespaceLocks) +
1154        ", tableLocks=" + filterUnlocked(this.tableLocks) +
1155        ", regionLocks=" + filterUnlocked(this.regionLocks);
1156    }
1157
1158    private String filterUnlocked(Map<?, LockAndQueue> locks) {
1159      StringBuilder sb = new StringBuilder("{");
1160      int initialLength = sb.length();
1161      for (Map.Entry<?, LockAndQueue> entry: locks.entrySet()) {
1162        if (!entry.getValue().isLocked()) continue;
1163        if (sb.length() > initialLength) sb.append(", ");
1164          sb.append("{");
1165          sb.append(entry.getKey());
1166          sb.append("=");
1167          sb.append(entry.getValue());
1168          sb.append("}");
1169        }
1170        sb.append("}");
1171        return sb.toString();
1172     }
1173  }

1179  private static class FairQueue<T extends Comparable<T>> {
1180    private final int quantum;
1181
1182    private Queue<T> currentQueue = null;
1183    private Queue<T> queueHead = null;
1184    private int currentQuantum = 0;
1185    private int size = 0;
1186
1187    public FairQueue() {
1188      this(1);
1189    }
1190
1191    public FairQueue(int quantum) {
1192      this.quantum = quantum;
1193    }
1194
1195    public boolean hasRunnables() {
1196      return size > 0;
1197    }
1198
1199    public void add(Queue<T> queue) {
1200      queueHead = AvlIterableList.append(queueHead, queue);
1201      if (currentQueue == null) setNextQueue(queueHead);
1202      size++;
1203    }
1204
1205    public void remove(Queue<T> queue) {
1206      Queue<T> nextQueue = AvlIterableList.readNext(queue);
1207      queueHead = AvlIterableList.remove(queueHead, queue);
1208      if (currentQueue == queue) {
1209        setNextQueue(queueHead != null ? nextQueue : null);
1210      }
1211      size--;
1212    }
1213
1214    public Queue<T> poll() {
1215      if (currentQuantum == 0) {
1216        if (!nextQueue()) {
1217          return null; // nothing here
1218        }
1219        currentQuantum = calculateQuantum(currentQueue) - 1;
1220      } else {
1221        currentQuantum--;
1222      }
1223
1224      // This should go away when we have the new AM and its events
1225      if (!currentQueue.isAvailable()) {
1226        Queue<T> lastQueue = currentQueue;
1227        do {
1228          if (!nextQueue())
1229            return null;
1230        } while (currentQueue != lastQueue && !currentQueue.isAvailable());
1231
1232        currentQuantum = calculateQuantum(currentQueue) - 1;
1233      }
1234      return currentQueue;
1235    }
1236
1237    private boolean nextQueue() {
1238      if (currentQueue == null) return false;
1239      currentQueue = AvlIterableList.readNext(currentQueue);
1240      return currentQueue != null;
1241    }
1242
1243    private void setNextQueue(Queue<T> queue) {
1244      currentQueue = queue;
1245      if (queue != null) {
1246        currentQuantum = calculateQuantum(currentQueue);
1247      } else {
1248        currentQuantum = 0;
1249      }
1250    }
1251
1252    private int calculateQuantum(final Queue queue) {
1253      return Math.max(1, queue.getPriority() * quantum); // TODO
1254    }
1255  }
