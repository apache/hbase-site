048/**
049 * This procedure recovers meta from prior shutdown/ crash of a server, and brings meta online by
050 * assigning meta region/s. Any place where meta is accessed and requires meta to be online, need to
051 * submit this procedure instead of duplicating steps to recover meta in the code.
052 */
053public class RecoverMetaProcedure
054    extends StateMachineProcedure<MasterProcedureEnv, MasterProcedureProtos.RecoverMetaState>
055    implements TableProcedureInterface {
056  private static final Logger LOG = LoggerFactory.getLogger(RecoverMetaProcedure.class);
057
058  private ServerName failedMetaServer;
059  private boolean shouldSplitWal;
060  private int replicaId;
061
062  private final ProcedurePrepareLatch syncLatch;
063  private MasterServices master;
064
065  /**
066   * Call this constructor to queue up a {@link RecoverMetaProcedure} in response to meta
067   * carrying server crash
068   * @param failedMetaServer failed/ crashed region server that was carrying meta
069   * @param shouldSplitLog split log file of meta region
070   */
071  public RecoverMetaProcedure(final ServerName failedMetaServer, final boolean shouldSplitLog) {
072    this(failedMetaServer, shouldSplitLog, null);
073  }
074
075  /**
076   * Constructor with latch, for blocking/ sync usage
077   */
078  public RecoverMetaProcedure(final ServerName failedMetaServer, final boolean shouldSplitLog,
079                              final ProcedurePrepareLatch latch) {
080    this.failedMetaServer = failedMetaServer;
081    this.shouldSplitWal = shouldSplitLog;
082    this.replicaId = RegionInfo.DEFAULT_REPLICA_ID;
083    this.syncLatch = latch;
084  }
085
086  /**
087   * This constructor is also used when deserializing from a procedure store; we'll construct one
088   * of these then call #deserializeStateData(InputStream). Do not use directly.
089   */
090  public RecoverMetaProcedure() {
091    this(null, false);
092  }
093
094  @Override
095  protected Flow executeFromState(MasterProcedureEnv env,
096      MasterProcedureProtos.RecoverMetaState state)
097      throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
098    prepare(env);
099
100    if (!isRunRequired()) {
101      LOG.info(this + "; Meta already initialized. Skipping run");
102      return Flow.NO_MORE_STATE;
103    }
104
105    try {
106      switch (state) {
107        case RECOVER_META_PREPARE:
108          // If Master is going down or cluster is up, skip this assign by returning NO_MORE_STATE
109          if (!master.isClusterUp()) {
110            String msg = "Cluster not up! Skipping hbase:meta assign.";
111            LOG.warn(msg);
112            return Flow.NO_MORE_STATE;
113          }
114          if (master.isStopping() || master.isStopped()) {
115            String msg = "Master stopping=" + master.isStopping() + ", stopped=" +
116                master.isStopped() + "; skipping hbase:meta assign.";
117            LOG.warn(msg);
118            return Flow.NO_MORE_STATE;
119          }
120          setNextState(RecoverMetaState.RECOVER_META_SPLIT_LOGS);
121          break;
122        case RECOVER_META_SPLIT_LOGS:
123          LOG.info("Start " + this);
124          if (shouldSplitWal) {
125            // TODO: Matteo. We BLOCK here but most important thing to be doing at this moment.
126            if (failedMetaServer != null) {
127              master.getMasterWalManager().splitMetaLog(failedMetaServer);
128            } else {
129              ServerName serverName =
130                  master.getMetaTableLocator().getMetaRegionLocation(master.getZooKeeper());
131              Set<ServerName> previouslyFailedServers =
132                  master.getMasterWalManager().getFailedServersFromLogFolders();
133              if (serverName != null && previouslyFailedServers.contains(serverName)) {
134                master.getMasterWalManager().splitMetaLog(serverName);
135              }
136            }
137          }
138          setNextState(RecoverMetaState.RECOVER_META_ASSIGN_REGIONS);
139          break;
140
141        case RECOVER_META_ASSIGN_REGIONS:
142          RegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(
143              RegionInfoBuilder.FIRST_META_REGIONINFO, this.replicaId);
144
145          AssignProcedure metaAssignProcedure;
146          AssignmentManager am = master.getAssignmentManager();
147          if (failedMetaServer != null) {
148            handleRIT(env, hri, this.failedMetaServer);
149            LOG.info(this + "; Assigning meta with new plan; previous server=" + failedMetaServer);
150            metaAssignProcedure = am.createAssignProcedure(hri);
151          } else {
152            // get server carrying meta from zk
153            ServerName metaServer =
154                MetaTableLocator.getMetaRegionState(master.getZooKeeper()).getServerName();
155            LOG.info(this + "; Retaining meta assignment to server=" + metaServer);
156            metaAssignProcedure = am.createAssignProcedure(hri, metaServer);
157          }
158
159          addChildProcedure(metaAssignProcedure);
160          return Flow.NO_MORE_STATE;
161
162        default:
163          throw new UnsupportedOperationException("unhandled state=" + state);
164      }
165    } catch (IOException|KeeperException e) {
166      LOG.warn(this + "; Failed state=" + state + ", retry " + this + "; cycles=" +
167          getCycles(), e);
168    }
169    return Flow.HAS_MORE_STATE;
170  }
171
172  /**
173   * Is the region stuck assigning to this failedMetaServer? If so, cancel the call
174   * just as we do over in ServerCrashProcedure#handleRIT except less to do here; less context
175   * to carry.
176   */
177  // NOTE: Make sure any fix or improvement done here is also done in SCP#handleRIT; the methods
178  // have overlap.
179  private void handleRIT(MasterProcedureEnv env, RegionInfo ri, ServerName crashedServerName) {
180    AssignmentManager am = env.getAssignmentManager();
181    RegionTransitionProcedure rtp = am.getRegionStates().getRegionTransitionProcedure(ri);
182    if (rtp == null) {
183      return; // Nothing to do. Not in RIT.
184    }
185    // Make sure the RIT is against this crashed server. In the case where there are many
186    // processings of a crashed server -- backed up for whatever reason (slow WAL split)
187    // -- then a previous SCP may have already failed an assign, etc., and it may have a
188    // new location target; DO NOT fail these else we make for assign flux.
189    ServerName rtpServerName = rtp.getServer(env);
190    if (rtpServerName == null) {
191      LOG.warn("RIT with ServerName null! " + rtp);
192    } else if (rtpServerName.equals(crashedServerName)) {
193      LOG.info("pid=" + getProcId() + " found RIT " + rtp + "; " +
194          rtp.getRegionState(env).toShortString());
195      rtp.remoteCallFailed(env, crashedServerName,
196          new ServerCrashException(getProcId(), crashedServerName));
197    }
198  }

200  @Override
201  protected void rollbackState(MasterProcedureEnv env,
202      MasterProcedureProtos.RecoverMetaState recoverMetaState)
203      throws IOException, InterruptedException {
204    // Can't rollback
205    throw new UnsupportedOperationException("unhandled state=" + recoverMetaState);
206  }

208  @Override
209  protected MasterProcedureProtos.RecoverMetaState getState(int stateId) {
210    return RecoverMetaState.forNumber(stateId);
211  }
212
213  @Override
214  protected int getStateId(MasterProcedureProtos.RecoverMetaState recoverMetaState) {
215    return recoverMetaState.getNumber();
216  }
217
218  @Override
219  protected MasterProcedureProtos.RecoverMetaState getInitialState() {
220    return RecoverMetaState.RECOVER_META_PREPARE;
221  }
222
223  @Override
224  protected void toStringClassDetails(StringBuilder sb) {
225    sb.append(getClass().getSimpleName());
226    sb.append(" failedMetaServer=");
227    sb.append(failedMetaServer);
228    sb.append(", splitWal=");
229    sb.append(shouldSplitWal);
230  }
231
232  @Override
233  protected void serializeStateData(ProcedureStateSerializer serializer)
234      throws IOException {
235    super.serializeStateData(serializer);
236    MasterProcedureProtos.RecoverMetaStateData.Builder state =
237        MasterProcedureProtos.RecoverMetaStateData.newBuilder().setShouldSplitWal(shouldSplitWal);
238    if (failedMetaServer != null) {
239      state.setFailedMetaServer(ProtobufUtil.toServerName(failedMetaServer));
240    }
241    state.setReplicaId(replicaId);
242    serializer.serialize(state.build());
243  }
244
245  @Override
246  protected void deserializeStateData(ProcedureStateSerializer serializer)
247      throws IOException {
248    super.deserializeStateData(serializer);
249    MasterProcedureProtos.RecoverMetaStateData state =
250        serializer.deserialize(MasterProcedureProtos.RecoverMetaStateData.class);
251    this.shouldSplitWal = state.hasShouldSplitWal() && state.getShouldSplitWal();
252    this.failedMetaServer = state.hasFailedMetaServer() ?
253        ProtobufUtil.toServerName(state.getFailedMetaServer()) : null;
254    this.replicaId = state.hasReplicaId() ? state.getReplicaId() : RegionInfo.DEFAULT_REPLICA_ID;
255  }
256
257  @Override
258  protected LockState acquireLock(MasterProcedureEnv env) {
259    if (env.getProcedureScheduler().waitTableExclusiveLock(this, TableName.META_TABLE_NAME)) {
260      return LockState.LOCK_EVENT_WAIT;
261    }
262    return LockState.LOCK_ACQUIRED;
263  }
264
265  @Override
266  protected void releaseLock(MasterProcedureEnv env) {
267    env.getProcedureScheduler().wakeTableExclusiveLock(this, TableName.META_TABLE_NAME);
268  }
269
270  @Override
271  protected void completionCleanup(MasterProcedureEnv env) {
272    ProcedurePrepareLatch.releaseLatch(syncLatch, this);
273  }
274
275  @Override
276  public TableName getTableName() {
277    return TableName.META_TABLE_NAME;
278  }
279
280  @Override
281  public TableOperationType getTableOperationType() {
282    return TableOperationType.ENABLE;
283  }
284
285  /**
286   * @return true if failedMetaServer is not null (meta carrying server crashed) or meta is
287   * already initialized
288   */
289  private boolean isRunRequired() {
290    return failedMetaServer != null || !master.getAssignmentManager().isMetaInitialized();
291  }
292
293  /**
294   * Prepare for execution
295   */
296  private void prepare(MasterProcedureEnv env) {
297    if (master == null) {
298      master = env.getMasterServices();
299      Preconditions.checkArgument(master != null);
300    }
301  }
