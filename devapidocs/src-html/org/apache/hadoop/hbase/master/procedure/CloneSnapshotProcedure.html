101  /**
102   * Set up monitor status if it is not created.
103   */
104  private MonitoredTask getMonitorStatus() {
105    if (monitorStatus == null) {
106      monitorStatus = TaskMonitor.get().createStatus("Cloning  snapshot '" + snapshot.getName() +
107        "' to table " + getTableName());
108    }
109    return monitorStatus;
110  }

112  @Override
113  protected Flow executeFromState(final MasterProcedureEnv env, final CloneSnapshotState state)
114      throws InterruptedException {
115    if (isTraceEnabled()) {
116      LOG.trace(this + " execute state=" + state);
117    }
118    try {
119      switch (state) {
120        case CLONE_SNAPSHOT_PRE_OPERATION:
121          // Verify if we can clone the table
122          prepareClone(env);
123
124          preCloneSnapshot(env);
125          setNextState(CloneSnapshotState.CLONE_SNAPSHOT_WRITE_FS_LAYOUT);
126          break;
127        case CLONE_SNAPSHOT_WRITE_FS_LAYOUT:
128          newRegions = createFilesystemLayout(env, hTableDescriptor, newRegions);
129          setNextState(CloneSnapshotState.CLONE_SNAPSHOT_ADD_TO_META);
130          break;
131        case CLONE_SNAPSHOT_ADD_TO_META:
132          addRegionsToMeta(env);
133          setNextState(CloneSnapshotState.CLONE_SNAPSHOT_ASSIGN_REGIONS);
134          break;
135        case CLONE_SNAPSHOT_ASSIGN_REGIONS:
136          CreateTableProcedure.assignRegions(env, getTableName(), newRegions);
137          setNextState(CloneSnapshotState.CLONE_SNAPSHOT_UPDATE_DESC_CACHE);
138          break;
139        case CLONE_SNAPSHOT_UPDATE_DESC_CACHE:
140          CreateTableProcedure.updateTableDescCache(env, getTableName());
141          setNextState(CloneSnapshotState.CLONE_SNAPSHOT_POST_OPERATION);
142          break;
143        case CLONE_SNAPSHOT_POST_OPERATION:
144          postCloneSnapshot(env);
145
146          MetricsSnapshot metricsSnapshot = new MetricsSnapshot();
147          metricsSnapshot.addSnapshotClone(
148            getMonitorStatus().getCompletionTimestamp() - getMonitorStatus().getStartTime());
149          getMonitorStatus().markComplete("Clone snapshot '"+ snapshot.getName() +"' completed!");
150          return Flow.NO_MORE_STATE;
151        default:
152          throw new UnsupportedOperationException("unhandled state=" + state);
153      }
154    } catch (IOException e) {
155      if (isRollbackSupported(state)) {
156        setFailure("master-clone-snapshot", e);
157      } else {
158        LOG.warn("Retriable error trying to clone snapshot=" + snapshot.getName() +
159          " to table=" + getTableName() + " state=" + state, e);
160      }
161    }
162    return Flow.HAS_MORE_STATE;
163  }
164
165  @Override
166  protected void rollbackState(final MasterProcedureEnv env, final CloneSnapshotState state)
167      throws IOException {
168    if (state == CloneSnapshotState.CLONE_SNAPSHOT_PRE_OPERATION) {
169      DeleteTableProcedure.deleteTableStates(env, getTableName());
170      // TODO-MAYBE: call the deleteTable coprocessor event?
171      return;
172    }
173
174    // The procedure doesn't have a rollback. The execution will succeed, at some point.
175    throw new UnsupportedOperationException("unhandled state=" + state);
176  }
177
178  @Override
179  protected boolean isRollbackSupported(final CloneSnapshotState state) {
180    switch (state) {
181      case CLONE_SNAPSHOT_PRE_OPERATION:
182        return true;
183      default:
184        return false;
185    }
186  }

223  @Override
224  public void serializeStateData(final OutputStream stream) throws IOException {
225    super.serializeStateData(stream);
226
227    MasterProcedureProtos.CloneSnapshotStateData.Builder cloneSnapshotMsg =
228      MasterProcedureProtos.CloneSnapshotStateData.newBuilder()
229        .setUserInfo(MasterProcedureUtil.toProtoUserInfo(getUser()))
230        .setSnapshot(this.snapshot)
231        .setTableSchema(ProtobufUtil.convertToTableSchema(hTableDescriptor));
232    if (newRegions != null) {
233      for (HRegionInfo hri: newRegions) {
234        cloneSnapshotMsg.addRegionInfo(HRegionInfo.convert(hri));
235      }
236    }
237    if (!parentsToChildrenPairMap.isEmpty()) {
238      final Iterator<Map.Entry<String, Pair<String, String>>> it =
239        parentsToChildrenPairMap.entrySet().iterator();
240      while (it.hasNext()) {
241        final Map.Entry<String, Pair<String, String>> entry = it.next();
242
243        MasterProcedureProtos.RestoreParentToChildRegionsPair.Builder parentToChildrenPair =
244          MasterProcedureProtos.RestoreParentToChildRegionsPair.newBuilder()
245          .setParentRegionName(entry.getKey())
246          .setChild1RegionName(entry.getValue().getFirst())
247          .setChild2RegionName(entry.getValue().getSecond());
248        cloneSnapshotMsg.addParentToChildRegionsPairList(parentToChildrenPair);
249      }
250    }
251    cloneSnapshotMsg.build().writeDelimitedTo(stream);
252  }
253
254  @Override
255  public void deserializeStateData(final InputStream stream) throws IOException {
256    super.deserializeStateData(stream);
257
258    MasterProcedureProtos.CloneSnapshotStateData cloneSnapshotMsg =
259      MasterProcedureProtos.CloneSnapshotStateData.parseDelimitedFrom(stream);
260    setUser(MasterProcedureUtil.toUserInfo(cloneSnapshotMsg.getUserInfo()));
261    snapshot = cloneSnapshotMsg.getSnapshot();
262    hTableDescriptor = ProtobufUtil.convertToHTableDesc(cloneSnapshotMsg.getTableSchema());
263    if (cloneSnapshotMsg.getRegionInfoCount() == 0) {
264      newRegions = null;
265    } else {
266      newRegions = new ArrayList<>(cloneSnapshotMsg.getRegionInfoCount());
267      for (HBaseProtos.RegionInfo hri: cloneSnapshotMsg.getRegionInfoList()) {
268        newRegions.add(HRegionInfo.convert(hri));
269      }
270    }
271    if (cloneSnapshotMsg.getParentToChildRegionsPairListCount() > 0) {
272      parentsToChildrenPairMap = new HashMap<>();
273      for (MasterProcedureProtos.RestoreParentToChildRegionsPair parentToChildrenPair:
274        cloneSnapshotMsg.getParentToChildRegionsPairListList()) {
275        parentsToChildrenPairMap.put(
276          parentToChildrenPair.getParentRegionName(),
277          new Pair<>(
278            parentToChildrenPair.getChild1RegionName(),
279            parentToChildrenPair.getChild2RegionName()));
280      }
281    }
282    // Make sure that the monitor status is set up
283    getMonitorStatus();
284  }

286  /**
287   * Action before any real action of cloning from snapshot.
288   * @param env MasterProcedureEnv
289   * @throws IOException
290   */
291  private void prepareClone(final MasterProcedureEnv env) throws IOException {
292    final TableName tableName = getTableName();
293    if (MetaTableAccessor.tableExists(env.getMasterServices().getConnection(), tableName)) {
294      throw new TableExistsException(getTableName());
295    }
296  }
297
298  /**
299   * Action before cloning from snapshot.
300   * @param env MasterProcedureEnv
301   * @throws IOException
302   * @throws InterruptedException
303   */
304  private void preCloneSnapshot(final MasterProcedureEnv env)
305      throws IOException, InterruptedException {
306    if (!getTableName().isSystemTable()) {
307      // Check and update namespace quota
308      final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
309
310      SnapshotManifest manifest = SnapshotManifest.open(
311        env.getMasterConfiguration(),
312        mfs.getFileSystem(),
313        SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshot, mfs.getRootDir()),
314        snapshot);
315
316      ProcedureSyncWait.getMasterQuotaManager(env)
317        .checkNamespaceTableAndRegionQuota(getTableName(), manifest.getRegionManifestsMap().size());
318    }
319
320    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
321    if (cpHost != null) {
322      cpHost.preCreateTableAction(hTableDescriptor, null, getUser());
323    }
324  }
325
326  /**
327   * Action after cloning from snapshot.
328   * @param env MasterProcedureEnv
329   * @throws IOException
330   * @throws InterruptedException
331   */
332  private void postCloneSnapshot(final MasterProcedureEnv env)
333      throws IOException, InterruptedException {
334    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
335    if (cpHost != null) {
336      final HRegionInfo[] regions = (newRegions == null) ? null :
337        newRegions.toArray(new HRegionInfo[newRegions.size()]);
338      cpHost.postCompletedCreateTableAction(hTableDescriptor, regions, getUser());
339    }
340  }

342  /**
343   * Create regions in file system.
344   * @param env MasterProcedureEnv
345   * @throws IOException
346   */
347  private List<HRegionInfo> createFilesystemLayout(
348    final MasterProcedureEnv env,
349    final HTableDescriptor hTableDescriptor,
350    final List<HRegionInfo> newRegions) throws IOException {
351    return createFsLayout(env, hTableDescriptor, newRegions, new CreateHdfsRegions() {
352      @Override
353      public List<HRegionInfo> createHdfsRegions(
354        final MasterProcedureEnv env,
355        final Path tableRootDir, final TableName tableName,
356        final List<HRegionInfo> newRegions) throws IOException {
357
358        final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
359        final FileSystem fs = mfs.getFileSystem();
360        final Path rootDir = mfs.getRootDir();
361        final Configuration conf = env.getMasterConfiguration();
362        final ForeignExceptionDispatcher monitorException = new ForeignExceptionDispatcher();
363
364        getMonitorStatus().setStatus("Clone snapshot - creating regions for table: " + tableName);
365
366        try {
367          // 1. Execute the on-disk Clone
368          Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshot, rootDir);
369          SnapshotManifest manifest = SnapshotManifest.open(conf, fs, snapshotDir, snapshot);
370          RestoreSnapshotHelper restoreHelper = new RestoreSnapshotHelper(
371            conf, fs, manifest, hTableDescriptor, tableRootDir, monitorException, monitorStatus);
372          RestoreSnapshotHelper.RestoreMetaChanges metaChanges = restoreHelper.restoreHdfsRegions();
373
374          // Clone operation should not have stuff to restore or remove
375          Preconditions.checkArgument(
376            !metaChanges.hasRegionsToRestore(), "A clone should not have regions to restore");
377          Preconditions.checkArgument(
378            !metaChanges.hasRegionsToRemove(), "A clone should not have regions to remove");
379
380          // At this point the clone is complete. Next step is enabling the table.
381          String msg =
382            "Clone snapshot="+ snapshot.getName() +" on table=" + tableName + " completed!";
383          LOG.info(msg);
384          monitorStatus.setStatus(msg + " Waiting for table to be enabled...");
385
386          // 2. Let the next step to add the regions to meta
387          return metaChanges.getRegionsToAdd();
388        } catch (Exception e) {
389          String msg = "clone snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot) +
390            " failed because " + e.getMessage();
391          LOG.error(msg, e);
392          IOException rse = new RestoreSnapshotException(msg, e,
393              ProtobufUtil.createSnapshotDesc(snapshot));
394
395          // these handlers aren't futures so we need to register the error here.
396          monitorException.receive(new ForeignException("Master CloneSnapshotProcedure", rse));
397          throw rse;
398        }
399      }
400    });
401  }

403  /**
404   * Create region layout in file system.
405   * @param env MasterProcedureEnv
406   * @throws IOException
407   */
408  private List<HRegionInfo> createFsLayout(
409    final MasterProcedureEnv env,
410    final HTableDescriptor hTableDescriptor,
411    List<HRegionInfo> newRegions,
412    final CreateHdfsRegions hdfsRegionHandler) throws IOException {
413    final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
414    final Path tempdir = mfs.getTempDir();
415
416    // 1. Create Table Descriptor
417    // using a copy of descriptor, table will be created enabling first
418    HTableDescriptor underConstruction = new HTableDescriptor(hTableDescriptor);
419    final Path tempTableDir = FSUtils.getTableDir(tempdir, hTableDescriptor.getTableName());
420    ((FSTableDescriptors)(env.getMasterServices().getTableDescriptors()))
421      .createTableDescriptorForTableDirectory(tempTableDir, underConstruction, false);
422
423    // 2. Create Regions
424    newRegions = hdfsRegionHandler.createHdfsRegions(
425      env, tempdir, hTableDescriptor.getTableName(), newRegions);
426
427    // 3. Move Table temp directory to the hbase root location
428    CreateTableProcedure.moveTempDirectoryToHBaseRoot(env, hTableDescriptor, tempTableDir);
429
430    return newRegions;
431  }
432
433  /**
434   * Add regions to hbase:meta table.
435   * @param env MasterProcedureEnv
436   * @throws IOException
437   */
438  private void addRegionsToMeta(final MasterProcedureEnv env) throws IOException {
439    newRegions = CreateTableProcedure.addTableToMeta(env, hTableDescriptor, newRegions);
440
441    RestoreSnapshotHelper.RestoreMetaChanges metaChanges =
442        new RestoreSnapshotHelper.RestoreMetaChanges(
443          hTableDescriptor, parentsToChildrenPairMap);
444    metaChanges.updateMetaParentRegions(env.getMasterServices().getConnection(), newRegions);
445  }
