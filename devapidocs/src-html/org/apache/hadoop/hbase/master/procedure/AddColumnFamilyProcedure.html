040/**
041 * The procedure to add a column family to an existing table.
042 */
043@InterfaceAudience.Private
044public class AddColumnFamilyProcedure
045    extends AbstractStateMachineTableProcedure<AddColumnFamilyState> {
046  private static final Log LOG = LogFactory.getLog(AddColumnFamilyProcedure.class);
047
048  private TableName tableName;
049  private TableDescriptor unmodifiedTableDescriptor;
050  private ColumnFamilyDescriptor cfDescriptor;
051
052  private List<RegionInfo> regionInfoList;
053  private Boolean traceEnabled;
054
055  public AddColumnFamilyProcedure() {
056    super();
057    this.unmodifiedTableDescriptor = null;
058    this.regionInfoList = null;
059    this.traceEnabled = null;
060  }

062  public AddColumnFamilyProcedure(final MasterProcedureEnv env, final TableName tableName,
063      final ColumnFamilyDescriptor cfDescriptor) throws IOException {
064    this(env, tableName, cfDescriptor, null);
065  }

067  public AddColumnFamilyProcedure(final MasterProcedureEnv env, final TableName tableName,
068      final ColumnFamilyDescriptor cfDescriptor, final ProcedurePrepareLatch latch) {
069    super(env, latch);
070    this.tableName = tableName;
071    this.cfDescriptor = cfDescriptor;
072    this.unmodifiedTableDescriptor = null;
073    this.regionInfoList = null;
074    this.traceEnabled = null;
075  }

077  @Override
078  protected Flow executeFromState(final MasterProcedureEnv env, final AddColumnFamilyState state)
079      throws InterruptedException {
080    if (isTraceEnabled()) {
081      LOG.trace(this + " execute state=" + state);
082    }
083
084    try {
085      switch (state) {
086      case ADD_COLUMN_FAMILY_PREPARE:
087        prepareAdd(env);
088        setNextState(AddColumnFamilyState.ADD_COLUMN_FAMILY_PRE_OPERATION);
089        break;
090      case ADD_COLUMN_FAMILY_PRE_OPERATION:
091        preAdd(env, state);
092        setNextState(AddColumnFamilyState.ADD_COLUMN_FAMILY_UPDATE_TABLE_DESCRIPTOR);
093        break;
094      case ADD_COLUMN_FAMILY_UPDATE_TABLE_DESCRIPTOR:
095        updateTableDescriptor(env);
096        setNextState(AddColumnFamilyState.ADD_COLUMN_FAMILY_POST_OPERATION);
097        break;
098      case ADD_COLUMN_FAMILY_POST_OPERATION:
099        postAdd(env, state);
100        setNextState(AddColumnFamilyState.ADD_COLUMN_FAMILY_REOPEN_ALL_REGIONS);
101        break;
102      case ADD_COLUMN_FAMILY_REOPEN_ALL_REGIONS:
103        if (env.getAssignmentManager().isTableEnabled(getTableName())) {
104          addChildProcedure(env.getAssignmentManager()
105            .createReopenProcedures(getRegionInfoList(env)));
106        }
107        return Flow.NO_MORE_STATE;
108      default:
109        throw new UnsupportedOperationException(this + " unhandled state=" + state);
110      }
111    } catch (IOException e) {
112      if (isRollbackSupported(state)) {
113        setFailure("master-add-columnfamily", e);
114      } else {
115        LOG.warn("Retriable error trying to add the column family " + getColumnFamilyName() +
116          " to the table " + tableName + " (in state=" + state + ")", e);
117      }
118    }
119    return Flow.HAS_MORE_STATE;
120  }
121
122  @Override
123  protected void rollbackState(final MasterProcedureEnv env, final AddColumnFamilyState state)
124      throws IOException {
125    if (state == AddColumnFamilyState.ADD_COLUMN_FAMILY_PREPARE ||
126        state == AddColumnFamilyState.ADD_COLUMN_FAMILY_PRE_OPERATION) {
127      // nothing to rollback, pre is just table-state checks.
128      // We can fail if the table does not exist or is not disabled.
129      // TODO: coprocessor rollback semantic is still undefined.
130      return;
131    }
132
133    // The procedure doesn't have a rollback. The execution will succeed, at some point.
134    throw new UnsupportedOperationException("unhandled state=" + state);
135  }
136
137  @Override
138  protected boolean isRollbackSupported(final AddColumnFamilyState state) {
139    switch (state) {
140      case ADD_COLUMN_FAMILY_PREPARE:
141      case ADD_COLUMN_FAMILY_PRE_OPERATION:
142        return true;
143      default:
144        return false;
145    }
146  }

168  @Override
169  protected void serializeStateData(ProcedureStateSerializer serializer)
170      throws IOException {
171    super.serializeStateData(serializer);
172
173    MasterProcedureProtos.AddColumnFamilyStateData.Builder addCFMsg =
174        MasterProcedureProtos.AddColumnFamilyStateData.newBuilder()
175            .setUserInfo(MasterProcedureUtil.toProtoUserInfo(getUser()))
176            .setTableName(ProtobufUtil.toProtoTableName(tableName))
177            .setColumnfamilySchema(ProtobufUtil.toColumnFamilySchema(cfDescriptor));
178    if (unmodifiedTableDescriptor != null) {
179      addCFMsg
180          .setUnmodifiedTableSchema(ProtobufUtil.toTableSchema(unmodifiedTableDescriptor));
181    }
182
183    serializer.serialize(addCFMsg.build());
184  }
185
186  @Override
187  protected void deserializeStateData(ProcedureStateSerializer serializer)
188      throws IOException {
189    super.deserializeStateData(serializer);
190
191    MasterProcedureProtos.AddColumnFamilyStateData addCFMsg =
192        serializer.deserialize(MasterProcedureProtos.AddColumnFamilyStateData.class);
193    setUser(MasterProcedureUtil.toUserInfo(addCFMsg.getUserInfo()));
194    tableName = ProtobufUtil.toTableName(addCFMsg.getTableName());
195    cfDescriptor = ProtobufUtil.toColumnFamilyDescriptor(addCFMsg.getColumnfamilySchema());
196    if (addCFMsg.hasUnmodifiedTableSchema()) {
197      unmodifiedTableDescriptor = ProtobufUtil.toTableDescriptor(addCFMsg.getUnmodifiedTableSchema());
198    }
199  }

225  /**
226   * Action before any real action of adding column family.
227   * @param env MasterProcedureEnv
228   * @throws IOException
229   */
230  private void prepareAdd(final MasterProcedureEnv env) throws IOException {
231    // Checks whether the table is allowed to be modified.
232    checkTableModifiable(env);
233
234    // In order to update the descriptor, we need to retrieve the old descriptor for comparison.
235    unmodifiedTableDescriptor = env.getMasterServices().getTableDescriptors().get(tableName);
236    if (unmodifiedTableDescriptor == null) {
237      throw new IOException("TableDescriptor missing for " + tableName);
238    }
239    if (unmodifiedTableDescriptor.hasColumnFamily(cfDescriptor.getName())) {
240      throw new InvalidFamilyOperationException("Column family '" + getColumnFamilyName()
241          + "' in table '" + tableName + "' already exists so cannot be added");
242    }
243  }
244
245  /**
246   * Action before adding column family.
247   * @param env MasterProcedureEnv
248   * @param state the procedure state
249   * @throws IOException
250   * @throws InterruptedException
251   */
252  private void preAdd(final MasterProcedureEnv env, final AddColumnFamilyState state)
253      throws IOException, InterruptedException {
254    runCoprocessorAction(env, state);
255  }
256
257  /**
258   * Add the column family to the file system
259   */
260  private void updateTableDescriptor(final MasterProcedureEnv env) throws IOException {
261    // Update table descriptor
262    LOG.info("AddColumn. Table = " + tableName + " HCD = " + cfDescriptor.toString());
263
264    TableDescriptor htd = env.getMasterServices().getTableDescriptors().get(tableName);
265
266    if (htd.hasColumnFamily(cfDescriptor.getName())) {
267      // It is possible to reach this situation, as we could already add the column family
268      // to table descriptor, but the master failover happens before we complete this state.
269      // We should be able to handle running this function multiple times without causing problem.
270      return;
271    }
272
273    env.getMasterServices().getTableDescriptors().add(
274            TableDescriptorBuilder.newBuilder(htd)
275            .addColumnFamily(cfDescriptor).build());
276  }
277
278  /**
279   * Restore the table descriptor back to pre-add
280   * @param env MasterProcedureEnv
281   * @throws IOException
282   **/
283  private void restoreTableDescriptor(final MasterProcedureEnv env) throws IOException {
284    TableDescriptor htd = env.getMasterServices().getTableDescriptors().get(tableName);
285    if (htd.hasColumnFamily(cfDescriptor.getName())) {
286      // Remove the column family from file system and update the table descriptor to
287      // the before-add-column-family-state
288      MasterDDLOperationHelper.deleteColumnFamilyFromFileSystem(env, tableName,
289        getRegionInfoList(env), cfDescriptor.getName(), cfDescriptor.isMobEnabled());
290
291      env.getMasterServices().getTableDescriptors().add(unmodifiedTableDescriptor);
292
293      // Make sure regions are opened after table descriptor is updated.
294      //reOpenAllRegionsIfTableIsOnline(env);
295      // TODO: NUKE ROLLBACK!!!!
296    }
297  }

299  /**
300   * Action after adding column family.
301   * @param env MasterProcedureEnv
302   * @param state the procedure state
303   * @throws IOException
304   * @throws InterruptedException
305   */
306  private void postAdd(final MasterProcedureEnv env, final AddColumnFamilyState state)
307      throws IOException, InterruptedException {
308    runCoprocessorAction(env, state);
309  }

311  /**
312   * The procedure could be restarted from a different machine. If the variable is null, we need to
313   * retrieve it.
314   * @return traceEnabled
315   */
316  private Boolean isTraceEnabled() {
317    if (traceEnabled == null) {
318      traceEnabled = LOG.isTraceEnabled();
319    }
320    return traceEnabled;
321  }
322
323  private String getColumnFamilyName() {
324    return cfDescriptor.getNameAsString();
325  }
326
327  /**
328   * Coprocessor Action.
329   * @param env MasterProcedureEnv
330   * @param state the procedure state
331   * @throws IOException
332   * @throws InterruptedException
333   */
334  private void runCoprocessorAction(final MasterProcedureEnv env, final AddColumnFamilyState state)
335      throws IOException, InterruptedException {
336    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
337    if (cpHost != null) {
338      switch (state) {
339        case ADD_COLUMN_FAMILY_PRE_OPERATION:
340          cpHost.preAddColumnFamilyAction(tableName, cfDescriptor, getUser());
341          break;
342        case ADD_COLUMN_FAMILY_POST_OPERATION:
343          cpHost.postCompletedAddColumnFamilyAction(tableName, cfDescriptor, getUser());
344          break;
345        default:
346          throw new UnsupportedOperationException(this + " unhandled state=" + state);
347      }
348    }
349  }
350
351  private List<RegionInfo> getRegionInfoList(final MasterProcedureEnv env) throws IOException {
352    if (regionInfoList == null) {
353      regionInfoList = env.getAssignmentManager().getRegionStates()
354          .getRegionsOfTable(getTableName());
355    }
356    return regionInfoList;
357  }
