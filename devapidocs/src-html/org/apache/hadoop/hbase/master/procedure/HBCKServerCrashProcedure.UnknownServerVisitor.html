040/**
041 * Acts like the super class in all cases except when no Regions found in the
042 * current Master in-memory context. In this latter case, when the call to
043 * super#getRegionsOnCrashedServer returns nothing, this SCP will scan
044 * hbase:meta for references to the passed ServerName. If any found, we'll
045 * clean them up.
046 *
047 * <p>This version of SCP is for external invocation as part of fix-up (e.g. HBCK2's
048 * scheduleRecoveries); the super class is used during normal recovery operations.
049 * It is for the case where meta has references to 'Unknown Servers',
050 * servers that are in hbase:meta but not in live-server or dead-server lists; i.e. Master
051 * and hbase:meta content have deviated. It should never happen in normal running
052 * cluster but if we do drop accounting of servers, we need a means of fix-up.
053 * Eventually, as part of normal CatalogJanitor task, rather than just identify
054 * these 'Unknown Servers', it would make repair, queuing something like this
055 * HBCKSCP to do cleanup, reassigning them so Master and hbase:meta are aligned again.
056 *
057 * <p>NOTE that this SCP is costly to run; does a full scan of hbase:meta.</p>
058 */
059@InterfaceAudience.Private
060public class HBCKServerCrashProcedure extends ServerCrashProcedure {
061  private static final Logger LOG = LoggerFactory.getLogger(HBCKServerCrashProcedure.class);
062
063  /**
064   * @param serverName Name of the crashed server.
065   * @param shouldSplitWal True if we should split WALs as part of crashed server processing.
066   * @param carryingMeta True if carrying hbase:meta table region.
067   */
068  public HBCKServerCrashProcedure(final MasterProcedureEnv env, final ServerName serverName,
069                              final boolean shouldSplitWal, final boolean carryingMeta) {
070    super(env, serverName, shouldSplitWal, carryingMeta);
071  }
072
073  /**
074   * Used when deserializing from a procedure store; we'll construct one of these then call
075   * #deserializeStateData(InputStream). Do not use directly.
076   */
077  public HBCKServerCrashProcedure() {}
078
079  /**
080   * If no Regions found in Master context, then we will search hbase:meta for references
081   * to the passed server. Operator may have passed ServerName because they have found
082   * references to 'Unknown Servers'. They are using HBCKSCP to clear them out.
083   */
084  @Override
085  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="NP_NULL_ON_SOME_PATH_EXCEPTION",
086    justification="FindBugs seems confused on ps in below.")
087  List<RegionInfo> getRegionsOnCrashedServer(MasterProcedureEnv env) {
088    // Super will return an immutable list (empty if nothing on this server).
089    List<RegionInfo> ris = super.getRegionsOnCrashedServer(env);
090    if (!ris.isEmpty()) {
091      return ris;
092    }
093    // Nothing in in-master context. Check for Unknown Server! in hbase:meta.
094    // If super list is empty, then allow that an operator scheduled an SCP because they are trying
095    // to purge 'Unknown Servers' -- servers that are neither online nor in dead servers
096    // list but that ARE in hbase:meta and so showing as unknown in places like 'HBCK Report'.
097    // This mis-accounting does not happen in normal circumstance but may arise in-extremis
098    // when cluster has been damaged in operation.
099    UnknownServerVisitor visitor =
100        new UnknownServerVisitor(env.getMasterServices().getConnection(), getServerName());
101    try {
102      MetaTableAccessor.scanMetaForTableRegions(env.getMasterServices().getConnection(),
103          visitor, null);
104    } catch (IOException ioe) {
105      LOG.warn("Failed scan of hbase:meta for 'Unknown Servers'", ioe);
106      return ris;
107    }
108    LOG.info("Found {} mentions of {} in hbase:meta of OPEN/OPENING Regions: {}",
109        visitor.getReassigns().size(), getServerName(),
110        visitor.getReassigns().stream().map(RegionInfo::getEncodedName).
111            collect(Collectors.joining(",")));
112    return visitor.getReassigns();
113  }
114
115  /**
116   * Visitor for hbase:meta that 'fixes' Unknown Server issues. Collects a List of Regions to
117   * reassign as 'result'.
118   */
119  private static final class UnknownServerVisitor implements ClientMetaTableAccessor.Visitor {
120    private final List<RegionInfo> reassigns = new ArrayList<>();
121    private final ServerName unknownServerName;
122    private final Connection connection;
123
124    private UnknownServerVisitor(Connection connection, ServerName unknownServerName) {
125      this.connection = connection;
126      this.unknownServerName = unknownServerName;
127    }
128
129    @Override
130    public boolean visit(Result result) throws IOException {
131      RegionLocations rls = CatalogFamilyFormat.getRegionLocations(result);
132      if (rls == null) {
133        return true;
134      }
135      for (HRegionLocation hrl: rls.getRegionLocations()) {
136        if (hrl == null) {
137          continue;
138        }
139        if (hrl.getRegion() == null) {
140          continue;
141        }
142        if (hrl.getServerName() == null) {
143          continue;
144        }
145        if (!hrl.getServerName().equals(this.unknownServerName)) {
146          continue;
147        }
148        RegionState.State state = RegionStateStore.getRegionState(result, hrl.getRegion());
149        RegionState rs = new RegionState(hrl.getRegion(), state, hrl.getServerName());
150        if (rs.isClosing()) {
151          // Move region to CLOSED in hbase:meta.
152          LOG.info("Moving {} from CLOSING to CLOSED in hbase:meta",
153              hrl.getRegion().getRegionNameAsString());
154          try {
155            MetaTableAccessor.updateRegionState(this.connection, hrl.getRegion(),
156                RegionState.State.CLOSED);
157          } catch (IOException ioe) {
158            LOG.warn("Failed moving {} from CLOSING to CLOSED",
159              hrl.getRegion().getRegionNameAsString(), ioe);
160          }
161        } else if (rs.isOpening() || rs.isOpened()) {
162          this.reassigns.add(hrl.getRegion());
163        } else {
164          LOG.info("Passing {}", rs);
165        }
166      }
167      return true;
168    }
169
170    private List<RegionInfo> getReassigns() {
171      return this.reassigns;
172    }
173  }
174
175  /**
176   * The RegionStateNode will not have a location if a confirm of an OPEN fails. On fail,
177   * the RegionStateNode regionLocation is set to null. This is 'looser' than the test done
178   * in the superclass. The HBCKSCP has been scheduled by an operator via hbck2 probably at the
179   * behest of a report of an 'Unknown Server' in the 'HBCK Report'. Let the operators operation
180   * succeed even in case where the region location in the RegionStateNode is null.
181   */
182  @Override
183  protected boolean isMatchingRegionLocation(RegionStateNode rsn) {
184    return super.isMatchingRegionLocation(rsn) || rsn.getRegionLocation() == null;
185  }
186}
