019package org.apache.hadoop.hbase.master.procedure;
020
021import java.io.IOException;
022
023import org.apache.hadoop.hbase.NamespaceDescriptor;
024import org.apache.hadoop.hbase.NamespaceExistException;
025import org.apache.yetus.audience.InterfaceAudience;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.apache.hadoop.hbase.master.MasterFileSystem;
029import org.apache.hadoop.hbase.master.TableNamespaceManager;
030import org.apache.hadoop.hbase.procedure2.ProcedureStateSerializer;
031import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
032import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos;
033import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.CreateNamespaceState;
034import org.apache.hadoop.hbase.util.FSUtils;
035
036/**
037 * The procedure to create a new namespace.
038 */
039@InterfaceAudience.Private
040public class CreateNamespaceProcedure
041    extends AbstractStateMachineNamespaceProcedure<CreateNamespaceState> {
042  private static final Logger LOG = LoggerFactory.getLogger(CreateNamespaceProcedure.class);
043
044  private NamespaceDescriptor nsDescriptor;
045  private Boolean traceEnabled;
046
047  public CreateNamespaceProcedure() {
048    this.traceEnabled = null;
049  }
050
051  public CreateNamespaceProcedure(final MasterProcedureEnv env,
052      final NamespaceDescriptor nsDescriptor) {
053    super(env);
054    this.nsDescriptor = nsDescriptor;
055    this.traceEnabled = null;
056  }
057
058  @Override
059  protected Flow executeFromState(final MasterProcedureEnv env, final CreateNamespaceState state)
060      throws InterruptedException {
061    if (isTraceEnabled()) {
062      LOG.trace(this + " execute state=" + state);
063    }
064    try {
065      switch (state) {
066      case CREATE_NAMESPACE_PREPARE:
067        prepareCreate(env);
068        setNextState(CreateNamespaceState.CREATE_NAMESPACE_CREATE_DIRECTORY);
069        break;
070      case CREATE_NAMESPACE_CREATE_DIRECTORY:
071        createDirectory(env, nsDescriptor);
072        setNextState(CreateNamespaceState.CREATE_NAMESPACE_INSERT_INTO_NS_TABLE);
073        break;
074      case CREATE_NAMESPACE_INSERT_INTO_NS_TABLE:
075        insertIntoNSTable(env, nsDescriptor);
076        setNextState(CreateNamespaceState.CREATE_NAMESPACE_UPDATE_ZK);
077        break;
078      case CREATE_NAMESPACE_UPDATE_ZK:
079        updateZKNamespaceManager(env, nsDescriptor);
080        setNextState(CreateNamespaceState.CREATE_NAMESPACE_SET_NAMESPACE_QUOTA);
081        break;
082      case CREATE_NAMESPACE_SET_NAMESPACE_QUOTA:
083        setNamespaceQuota(env, nsDescriptor);
084        return Flow.NO_MORE_STATE;
085      default:
086        throw new UnsupportedOperationException(this + " unhandled state=" + state);
087      }
088    } catch (IOException e) {
089      if (isRollbackSupported(state)) {
090        setFailure("master-create-namespace", e);
091      } else {
092        LOG.warn("Retriable error trying to create namespace=" + nsDescriptor.getName() +
093          " (in state=" + state + ")", e);
094      }
095    }
096    return Flow.HAS_MORE_STATE;
097  }
098
099  @Override
100  protected void rollbackState(final MasterProcedureEnv env, final CreateNamespaceState state)
101      throws IOException {
102    if (state == CreateNamespaceState.CREATE_NAMESPACE_PREPARE) {
103      // nothing to rollback, pre-create is just state checks.
104      // TODO: coprocessor rollback semantic is still undefined.
105      return;
106    }
107    // The procedure doesn't have a rollback. The execution will succeed, at some point.
108    throw new UnsupportedOperationException("unhandled state=" + state);
109  }
110
111  @Override
112  protected boolean isRollbackSupported(final CreateNamespaceState state) {
113    switch (state) {
114      case CREATE_NAMESPACE_PREPARE:
115        return true;
116      default:
117        return false;
118    }
119  }
120
121  @Override
122  protected CreateNamespaceState getState(final int stateId) {
123    return CreateNamespaceState.valueOf(stateId);
124  }
125
126  @Override
127  protected int getStateId(final CreateNamespaceState state) {
128    return state.getNumber();
129  }
130
131  @Override
132  protected CreateNamespaceState getInitialState() {
133    return CreateNamespaceState.CREATE_NAMESPACE_PREPARE;
134  }
135
136  @Override
137  protected void serializeStateData(ProcedureStateSerializer serializer)
138      throws IOException {
139    super.serializeStateData(serializer);
140
141    MasterProcedureProtos.CreateNamespaceStateData.Builder createNamespaceMsg =
142        MasterProcedureProtos.CreateNamespaceStateData.newBuilder().setNamespaceDescriptor(
143          ProtobufUtil.toProtoNamespaceDescriptor(this.nsDescriptor));
144    serializer.serialize(createNamespaceMsg.build());
145  }
146
147  @Override
148  protected void deserializeStateData(ProcedureStateSerializer serializer)
149      throws IOException {
150    super.deserializeStateData(serializer);
151
152    MasterProcedureProtos.CreateNamespaceStateData createNamespaceMsg =
153        serializer.deserialize(MasterProcedureProtos.CreateNamespaceStateData.class);
154    nsDescriptor = ProtobufUtil.toNamespaceDescriptor(createNamespaceMsg.getNamespaceDescriptor());
155  }
156
157  private boolean isBootstrapNamespace() {
158    return nsDescriptor.equals(NamespaceDescriptor.DEFAULT_NAMESPACE) ||
159        nsDescriptor.equals(NamespaceDescriptor.SYSTEM_NAMESPACE);
160  }
161
162  @Override
163  protected LockState acquireLock(final MasterProcedureEnv env) {
164    if (!env.getMasterServices().isInitialized()) {
165      // Namespace manager might not be ready if master is not fully initialized,
166      // return false to reject user namespace creation; return true for default
167      // and system namespace creation (this is part of master initialization).
168      if (!isBootstrapNamespace() && env.waitInitialized(this)) {
169        return LockState.LOCK_EVENT_WAIT;
170      }
171    }
172    if (env.getProcedureScheduler().waitNamespaceExclusiveLock(this, getNamespaceName())) {
173      return LockState.LOCK_EVENT_WAIT;
174    }
175    return LockState.LOCK_ACQUIRED;
176  }
177
178  @Override
179  public TableOperationType getTableOperationType() {
180    return TableOperationType.EDIT;
181  }
182
183  @Override
184  protected String getNamespaceName() {
185    return nsDescriptor.getName();
186  }
187
188  /**
189   * Action before any real action of creating namespace.
190   * @param env MasterProcedureEnv
191   * @throws IOException
192   */
193  private void prepareCreate(final MasterProcedureEnv env) throws IOException {
194    if (getTableNamespaceManager(env).doesNamespaceExist(nsDescriptor.getName())) {
195      throw new NamespaceExistException(nsDescriptor.getName());
196    }
197    getTableNamespaceManager(env).validateTableAndRegionCount(nsDescriptor);
198  }
199
200  /**
201   * Create the namespace directory
202   * @param env MasterProcedureEnv
203   * @param nsDescriptor NamespaceDescriptor
204   * @throws IOException
205   */
206  protected static void createDirectory(
207      final MasterProcedureEnv env,
208      final NamespaceDescriptor nsDescriptor) throws IOException {
209    MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
210    mfs.getFileSystem().mkdirs(
211      FSUtils.getNamespaceDir(mfs.getRootDir(), nsDescriptor.getName()));
212  }
213
214  /**
215   * Insert the row into ns table
216   * @param env MasterProcedureEnv
217   * @param nsDescriptor NamespaceDescriptor
218   * @throws IOException
219   */
220  protected static void insertIntoNSTable(
221      final MasterProcedureEnv env,
222      final NamespaceDescriptor nsDescriptor) throws IOException {
223    getTableNamespaceManager(env).insertIntoNSTable(nsDescriptor);
224  }
225
226  /**
227   * Update ZooKeeper.
228   * @param env MasterProcedureEnv
229   * @param nsDescriptor NamespaceDescriptor
230   * @throws IOException
231   */
232  protected static void updateZKNamespaceManager(
233      final MasterProcedureEnv env,
234      final NamespaceDescriptor nsDescriptor) throws IOException {
235    getTableNamespaceManager(env).updateZKNamespaceManager(nsDescriptor);
236  }
237
238  /**
239   * Set quota for the namespace
240   * @param env MasterProcedureEnv
241   * @param nsDescriptor NamespaceDescriptor
242   * @throws IOException
243   **/
244  protected static void setNamespaceQuota(
245      final MasterProcedureEnv env,
246      final NamespaceDescriptor nsDescriptor) throws IOException {
247    if (env.getMasterServices().isInitialized()) {
248      env.getMasterServices().getMasterQuotaManager().setNamespaceQuota(nsDescriptor);
249    }
250  }
251
252  /**
253   * remove quota for the namespace if exists
254   * @param env MasterProcedureEnv
255   * @throws IOException
256   **/
257  private void rollbackSetNamespaceQuota(final MasterProcedureEnv env) throws IOException {
258    try {
259      DeleteNamespaceProcedure.removeNamespaceQuota(env, nsDescriptor.getName());
260    } catch (Exception e) {
261      // Ignore exception
262      LOG.debug("Rollback of setNamespaceQuota throws exception: " + e);
263    }
264  }
265
266  private static TableNamespaceManager getTableNamespaceManager(final MasterProcedureEnv env) {
267    return env.getMasterServices().getClusterSchema().getTableNamespaceManager();
268  }
269
270  /**
271   * The procedure could be restarted from a different machine. If the variable is null, we need to
272   * retrieve it.
273   * @return traceEnabled
274   */
275  private Boolean isTraceEnabled() {
276    if (traceEnabled == null) {
277      traceEnabled = LOG.isTraceEnabled();
278    }
279    return traceEnabled;
280  }
281
282  @Override
283  protected boolean shouldWaitClientAck(MasterProcedureEnv env) {
284    // hbase and default namespaces are created on bootstrap internally by the system
285    // the client does not know about this procedures.
286    return !isBootstrapNamespace();
287  }
