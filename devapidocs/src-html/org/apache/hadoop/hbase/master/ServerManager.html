087/**
088 * The ServerManager class manages info about region servers.
089 * <p>
090 * Maintains lists of online and dead servers.  Processes the startups,
091 * shutdowns, and deaths of region servers.
092 * <p>
093 * Servers are distinguished in two different ways.  A given server has a
094 * location, specified by hostname and port, and of which there can only be one
095 * online at any given time.  A server instance is specified by the location
096 * (hostname and port) as well as the startcode (timestamp from when the server
097 * was started).  This is used to differentiate a restarted instance of a given
098 * server from the original instance.
099 * <p>
100 * If a sever is known not to be running any more, it is called dead. The dead
101 * server needs to be handled by a ServerShutdownHandler.  If the handler is not
102 * enabled yet, the server can't be handled right away so it is queued up.
103 * After the handler is enabled, the server will be submitted to a handler to handle.
104 * However, the handler may be just partially enabled.  If so,
105 * the server cannot be fully processed, and be queued up for further processing.
106 * A server is fully processed only after the handler is fully enabled
107 * and has completed the handling.
108 */

247  /**
248   * Let the server manager know a new regionserver has come online
249   * @param request the startup request
250   * @param ia the InetAddress from which request is received
251   * @return The ServerName we know this server as.
252   * @throws IOException
253   */
254  ServerName regionServerStartup(RegionServerStartupRequest request, InetAddress ia)
255      throws IOException {
256    // Test for case where we get a region startup message from a regionserver
257    // that has been quickly restarted but whose znode expiration handler has
258    // not yet run, or from a server whose fail we are currently processing.
259    // Test its host+port combo is present in serverAddresstoServerInfo.  If it
260    // is, reject the server and trigger its expiration. The next time it comes
261    // in, it should have been removed from serverAddressToServerInfo and queued
262    // for processing by ProcessServerShutdown.
263
264    final String hostname = request.hasUseThisHostnameInstead() ?
265        request.getUseThisHostnameInstead() :ia.getHostName();
266    ServerName sn = ServerName.valueOf(hostname, request.getPort(),
267      request.getServerStartCode());
268    checkClockSkew(sn, request.getServerCurrentTime());
269    checkIsDead(sn, "STARTUP");
270    if (!checkAndRecordNewServer(sn, ServerLoad.EMPTY_SERVERLOAD)) {
271      LOG.warn("THIS SHOULD NOT HAPPEN, RegionServerStartup"
272        + " could not record the server: " + sn);
273    }
274    return sn;
275  }

277  /**
278   * Updates last flushed sequence Ids for the regions on server sn
279   * @param sn
280   * @param hsl
281   */
282  private void updateLastFlushedSequenceIds(ServerName sn, ServerLoad hsl) {
283    Map<byte[], RegionLoad> regionsLoad = hsl.getRegionsLoad();
284    for (Entry<byte[], RegionLoad> entry : regionsLoad.entrySet()) {
285      byte[] encodedRegionName = Bytes.toBytes(HRegionInfo.encodeRegionName(entry.getKey()));
286      Long existingValue = flushedSequenceIdByRegion.get(encodedRegionName);
287      long l = entry.getValue().getCompleteSequenceId();
288      // Don't let smaller sequence ids override greater sequence ids.
289      if (LOG.isTraceEnabled()) {
290        LOG.trace(Bytes.toString(encodedRegionName) + ", existingValue=" + existingValue +
291          ", completeSequenceId=" + l);
292      }
293      if (existingValue == null || (l != HConstants.NO_SEQNUM && l > existingValue)) {
294        flushedSequenceIdByRegion.put(encodedRegionName, l);
295      } else if (l != HConstants.NO_SEQNUM && l < existingValue) {
296        LOG.warn("RegionServer " + sn + " indicates a last flushed sequence id ("
297            + l + ") that is less than the previous last flushed sequence id ("
298            + existingValue + ") for region " + Bytes.toString(entry.getKey()) + " Ignoring.");
299      }
300      ConcurrentNavigableMap<byte[], Long> storeFlushedSequenceId =
301          computeIfAbsent(storeFlushedSequenceIdsByRegion, encodedRegionName,
302            () -> new ConcurrentSkipListMap<>(Bytes.BYTES_COMPARATOR));
303      for (StoreSequenceId storeSeqId : entry.getValue().getStoreCompleteSequenceId()) {
304        byte[] family = storeSeqId.getFamilyName().toByteArray();
305        existingValue = storeFlushedSequenceId.get(family);
306        l = storeSeqId.getSequenceId();
307        if (LOG.isTraceEnabled()) {
308          LOG.trace(Bytes.toString(encodedRegionName) + ", family=" + Bytes.toString(family) +
309            ", existingValue=" + existingValue + ", completeSequenceId=" + l);
310        }
311        // Don't let smaller sequence ids override greater sequence ids.
312        if (existingValue == null || (l != HConstants.NO_SEQNUM && l > existingValue.longValue())) {
313          storeFlushedSequenceId.put(family, l);
314        }
315      }
316    }
317  }

319  void regionServerReport(ServerName sn,
320      ServerLoad sl) throws YouAreDeadException {
321    checkIsDead(sn, "REPORT");
322    if (null == this.onlineServers.replace(sn, sl)) {
323      // Already have this host+port combo and its just different start code?
324      // Just let the server in. Presume master joining a running cluster.
325      // recordNewServer is what happens at the end of reportServerStartup.
326      // The only thing we are skipping is passing back to the regionserver
327      // the ServerName to use. Here we presume a master has already done
328      // that so we'll press on with whatever it gave us for ServerName.
329      if (!checkAndRecordNewServer(sn, sl)) {
330        LOG.info("RegionServerReport ignored, could not record the server: " + sn);
331        return; // Not recorded, so no need to move on
332      }
333    }
334    updateLastFlushedSequenceIds(sn, sl);
335  }

337  /**
338   * Check is a server of same host and port already exists,
339   * if not, or the existed one got a smaller start code, record it.
340   *
341   * @param serverName the server to check and record
342   * @param sl the server load on the server
343   * @return true if the server is recorded, otherwise, false
344   */
345  boolean checkAndRecordNewServer(
346      final ServerName serverName, final ServerLoad sl) {
347    ServerName existingServer = null;
348    synchronized (this.onlineServers) {
349      existingServer = findServerWithSameHostnamePortWithLock(serverName);
350      if (existingServer != null && (existingServer.getStartcode() > serverName.getStartcode())) {
351        LOG.info("Server serverName=" + serverName + " rejected; we already have "
352            + existingServer.toString() + " registered with same hostname and port");
353        return false;
354      }
355      recordNewServerWithLock(serverName, sl);
356    }
357
358    // Tell our listeners that a server was added
359    if (!this.listeners.isEmpty()) {
360      for (ServerListener listener : this.listeners) {
361        listener.serverAdded(serverName);
362      }
363    }
364
365    // Note that we assume that same ts means same server, and don't expire in that case.
366    //  TODO: ts can theoretically collide due to clock shifts, so this is a bit hacky.
367    if (existingServer != null && (existingServer.getStartcode() < serverName.getStartcode())) {
368      LOG.info("Triggering server recovery; existingServer " +
369          existingServer + " looks stale, new server:" + serverName);
370      expireServer(existingServer);
371    }
372    return true;
373  }

375  /**
376   * Checks if the clock skew between the server and the master. If the clock skew exceeds the
377   * configured max, it will throw an exception; if it exceeds the configured warning threshold,
378   * it will log a warning but start normally.
379   * @param serverName Incoming servers's name
380   * @param serverCurrentTime
381   * @throws ClockOutOfSyncException if the skew exceeds the configured max value
382   */
383  private void checkClockSkew(final ServerName serverName, final long serverCurrentTime)
384  throws ClockOutOfSyncException {
385    long skew = Math.abs(System.currentTimeMillis() - serverCurrentTime);
386    if (skew > maxSkew) {
387      String message = "Server " + serverName + " has been " +
388        "rejected; Reported time is too far out of sync with master.  " +
389        "Time difference of " + skew + "ms > max allowed of " + maxSkew + "ms";
390      LOG.warn(message);
391      throw new ClockOutOfSyncException(message);
392    } else if (skew > warningSkew){
393      String message = "Reported time for server " + serverName + " is out of sync with master " +
394        "by " + skew + "ms. (Warning threshold is " + warningSkew + "ms; " +
395        "error threshold is " + maxSkew + "ms)";
396      LOG.warn(message);
397    }
398  }

400  /**
401   * If this server is on the dead list, reject it with a YouAreDeadException.
402   * If it was dead but came back with a new start code, remove the old entry
403   * from the dead list.
404   * @param serverName
405   * @param what START or REPORT
406   * @throws org.apache.hadoop.hbase.YouAreDeadException
407   */
408  private void checkIsDead(final ServerName serverName, final String what)
409      throws YouAreDeadException {
410    if (this.deadservers.isDeadServer(serverName)) {
411      // host name, port and start code all match with existing one of the
412      // dead servers. So, this server must be dead.
413      String message = "Server " + what + " rejected; currently processing " +
414          serverName + " as dead server";
415      LOG.debug(message);
416      throw new YouAreDeadException(message);
417    }
418    // remove dead server with same hostname and port of newly checking in rs after master
419    // initialization.See HBASE-5916 for more information.
420    if ((this.master == null || this.master.isInitialized())
421        && this.deadservers.cleanPreviousInstance(serverName)) {
422      // This server has now become alive after we marked it as dead.
423      // We removed it's previous entry from the dead list to reflect it.
424      LOG.debug(what + ":" + " Server " + serverName + " came back up," +
425          " removed it from the dead servers list");
426    }
427  }

429  /**
430   * Assumes onlineServers is locked.
431   * @return ServerName with matching hostname and port.
432   */
433  private ServerName findServerWithSameHostnamePortWithLock(
434      final ServerName serverName) {
435    ServerName end = ServerName.valueOf(serverName.getHostname(), serverName.getPort(),
436        Long.MAX_VALUE);
437
438    ServerName r = onlineServers.lowerKey(end);
439    if (r != null) {
440      if (ServerName.isSameHostnameAndPort(r, serverName)) {
441        return r;
442      }
443    }
444    return null;
445  }

459  public RegionStoreSequenceIds getLastFlushedSequenceId(byte[] encodedRegionName) {
460    RegionStoreSequenceIds.Builder builder = RegionStoreSequenceIds.newBuilder();
461    Long seqId = flushedSequenceIdByRegion.get(encodedRegionName);
462    builder.setLastFlushedSequenceId(seqId != null ? seqId.longValue() : HConstants.NO_SEQNUM);
463    Map<byte[], Long> storeFlushedSequenceId =
464        storeFlushedSequenceIdsByRegion.get(encodedRegionName);
465    if (storeFlushedSequenceId != null) {
466      for (Map.Entry<byte[], Long> entry : storeFlushedSequenceId.entrySet()) {
467        builder.addStoreSequenceId(StoreSequenceId.newBuilder()
468            .setFamilyName(UnsafeByteOperations.unsafeWrap(entry.getKey()))
469            .setSequenceId(entry.getValue().longValue()).build());
470      }
471    }
472    return builder.build();
473  }

484   * Compute the average load across all region servers.
485   * Currently, this uses a very naive computation - just uses the number of
486   * regions being served, ignoring stats about number of requests.
487   * @return the average load
488   */
489  public double getAverageLoad() {
490    int totalLoad = 0;
491    int numServers = 0;
492    for (ServerLoad sl: this.onlineServers.values()) {
493        numServers++;
494        totalLoad += sl.getNumberOfRegions();
495    }
496    return numServers == 0 ? 0 :
497      (double)totalLoad / (double)numServers;
498  }

528  void letRegionServersShutdown() {
529    long previousLogTime = 0;
530    ServerName sn = master.getServerName();
531    ZooKeeperWatcher zkw = master.getZooKeeper();
532    int onlineServersCt;
533    while ((onlineServersCt = onlineServers.size()) > 0){
534
535      if (System.currentTimeMillis() > (previousLogTime + 1000)) {
536        Set<ServerName> remainingServers = onlineServers.keySet();
537        synchronized (onlineServers) {
538          if (remainingServers.size() == 1 && remainingServers.contains(sn)) {
539            // Master will delete itself later.
540            return;
541          }
542        }
543        StringBuilder sb = new StringBuilder();
544        // It's ok here to not sync on onlineServers - merely logging
545        for (ServerName key : remainingServers) {
546          if (sb.length() > 0) {
547            sb.append(", ");
548          }
549          sb.append(key);
550        }
551        LOG.info("Waiting on regionserver(s) to go down " + sb.toString());
552        previousLogTime = System.currentTimeMillis();
553      }
554
555      try {
556        List<String> servers = ZKUtil.listChildrenNoWatch(zkw, zkw.znodePaths.rsZNode);
557        if (servers == null || servers.isEmpty() || (servers.size() == 1
558            && servers.contains(sn.toString()))) {
559          LOG.info("ZK shows there is only the master self online, exiting now");
560          // Master could have lost some ZK events, no need to wait more.
561          break;
562        }
563      } catch (KeeperException ke) {
564        LOG.warn("Failed to list regionservers", ke);
565        // ZK is malfunctioning, don't hang here
566        break;
567      }
568      synchronized (onlineServers) {
569        try {
570          if (onlineServersCt == onlineServers.size()) onlineServers.wait(100);
571        } catch (InterruptedException ignored) {
572          // continue
573        }
574      }
575    }
576  }

578  /*
579   * Expire the passed server.  Add it to list of dead servers and queue a
580   * shutdown processing.
581   */
582  public synchronized void expireServer(final ServerName serverName) {
583    if (serverName.equals(master.getServerName())) {
584      if (!(master.isAborted() || master.isStopped())) {
585        master.stop("We lost our znode?");
586      }
587      return;
588    }
589    if (!master.isServerCrashProcessingEnabled()) {
590      LOG.info("Master doesn't enable ServerShutdownHandler during initialization, "
591          + "delay expiring server " + serverName);
592      this.queuedDeadServers.add(serverName);
593      return;
594    }
595    if (this.deadservers.isDeadServer(serverName)) {
596      // TODO: Can this happen?  It shouldn't be online in this case?
597      LOG.warn("Expiration of " + serverName +
598          " but server shutdown already in progress");
599      return;
600    }
601    moveFromOnelineToDeadServers(serverName);
602
603    // If cluster is going down, yes, servers are going to be expiring; don't
604    // process as a dead server
605    if (this.clusterShutdown) {
606      LOG.info("Cluster shutdown set; " + serverName +
607        " expired; onlineServers=" + this.onlineServers.size());
608      if (this.onlineServers.isEmpty()) {
609        master.stop("Cluster shutdown set; onlineServer=0");
610      }
611      return;
612    }
613
614    boolean carryingMeta = master.getAssignmentManager().isCarryingMeta(serverName);
615    ProcedureExecutor<MasterProcedureEnv> procExec = this.master.getMasterProcedureExecutor();
616    procExec.submitProcedure(new ServerCrashProcedure(
617      procExec.getEnvironment(), serverName, true, carryingMeta));
618    LOG.debug("Added=" + serverName +
619      " to dead servers, submitted shutdown handler to be executed meta=" + carryingMeta);
620
621    // Tell our listeners that a server was removed
622    if (!this.listeners.isEmpty()) {
623      for (ServerListener listener : this.listeners) {
624        listener.serverRemoved(serverName);
625      }
626    }
627  }

629  @VisibleForTesting
630  public void moveFromOnelineToDeadServers(final ServerName sn) {
631    synchronized (onlineServers) {
632      if (!this.onlineServers.containsKey(sn)) {
633        LOG.warn("Expiration of " + sn + " but server not online");
634      }
635      // Remove the server from the known servers lists and update load info BUT
636      // add to deadservers first; do this so it'll show in dead servers list if
637      // not in online servers list.
638      this.deadservers.add(sn);
639      this.onlineServers.remove(sn);
640      onlineServers.notifyAll();
641    }
642    this.rsAdmins.remove(sn);
643  }

645  public synchronized void processDeadServer(final ServerName serverName, boolean shouldSplitWal) {
646    // When assignment manager is cleaning up the zookeeper nodes and rebuilding the
647    // in-memory region states, region servers could be down. Meta table can and
648    // should be re-assigned, log splitting can be done too. However, it is better to
649    // wait till the cleanup is done before re-assigning user regions.
650    //
651    // We should not wait in the server shutdown handler thread since it can clog
652    // the handler threads and meta table could not be re-assigned in case
653    // the corresponding server is down. So we queue them up here instead.
654    if (!master.getAssignmentManager().isFailoverCleanupDone()) {
655      requeuedDeadServers.put(serverName, shouldSplitWal);
656      return;
657    }
658
659    this.deadservers.add(serverName);
660    ProcedureExecutor<MasterProcedureEnv> procExec = this.master.getMasterProcedureExecutor();
661    procExec.submitProcedure(new ServerCrashProcedure(
662      procExec.getEnvironment(), serverName, shouldSplitWal, false));
663  }

665  /**
666   * Process the servers which died during master's initialization. It will be
667   * called after HMaster#assignMeta and AssignmentManager#joinCluster.
668   * */
669  synchronized void processQueuedDeadServers() {
670    if (!master.isServerCrashProcessingEnabled()) {
671      LOG.info("Master hasn't enabled ServerShutdownHandler");
672    }
673    Iterator<ServerName> serverIterator = queuedDeadServers.iterator();
674    while (serverIterator.hasNext()) {
675      ServerName tmpServerName = serverIterator.next();
676      expireServer(tmpServerName);
677      serverIterator.remove();
678      requeuedDeadServers.remove(tmpServerName);
679    }
680
681    if (!master.getAssignmentManager().isFailoverCleanupDone()) {
682      LOG.info("AssignmentManager hasn't finished failover cleanup; waiting");
683    }
684
685    for (Map.Entry<ServerName, Boolean> entry : requeuedDeadServers.entrySet()) {
686      processDeadServer(entry.getKey(), entry.getValue());
687    }
688    requeuedDeadServers.clear();
689  }

867   * Contacts a region server and waits up to timeout ms
868   * to close the region.  This bypasses the active hmaster.
869   */
870  public static void closeRegionSilentlyAndWait(ClusterConnection connection,
871    ServerName server, HRegionInfo region, long timeout) throws IOException, InterruptedException {
872    AdminService.BlockingInterface rs = connection.getAdmin(server);
873    HBaseRpcController controller = connection.getRpcControllerFactory().newController();
874    try {
875      ProtobufUtil.closeRegion(controller, rs, server, region.getRegionName());
876    } catch (IOException e) {
877      LOG.warn("Exception when closing region: " + region.getRegionNameAsString(), e);
878    }
879    long expiration = timeout + System.currentTimeMillis();
880    while (System.currentTimeMillis() < expiration) {
881      controller.reset();
882      try {
883        HRegionInfo rsRegion =
884          ProtobufUtil.getRegionInfo(controller, rs, region.getRegionName());
885        if (rsRegion == null) return;
886      } catch (IOException ioe) {
887        if (ioe instanceof NotServingRegionException) // no need to retry again
888          return;
889        LOG.warn("Exception when retrieving regioninfo from: "
890          + region.getRegionNameAsString(), ioe);
891      }
892      Thread.sleep(1000);
893    }
894    throw new IOException("Region " + region + " failed to close within"
895        + " timeout " + timeout);
896  }

898  /**
899   * Check if a region server is reachable and has the expected start code
900   */
901  public boolean isServerReachable(ServerName server) {
902    if (server == null) throw new NullPointerException("Passed server is null");
903
904
905    RetryCounter retryCounter = pingRetryCounterFactory.create();
906    while (retryCounter.shouldRetry()) {
907      try {
908        HBaseRpcController controller = newRpcController();
909        AdminService.BlockingInterface admin = getRsAdmin(server);
910        if (admin != null) {
911          ServerInfo info = ProtobufUtil.getServerInfo(controller, admin);
912          return info != null && info.hasServerName()
913            && server.getStartcode() == info.getServerName().getStartCode();
914        }
915      } catch (IOException ioe) {
916        LOG.debug("Couldn't reach " + server + ", try=" + retryCounter.getAttemptTimes()
917          + " of " + retryCounter.getMaxAttempts(), ioe);
918        try {
919          retryCounter.sleepUntilNextRetry();
920        } catch(InterruptedException ie) {
921          Thread.currentThread().interrupt();
922        }
923      }
924    }
925    return false;
926  }

950  /**
951   * Wait for the region servers to report in.
952   * We will wait until one of this condition is met:
953   *  - the master is stopped
954   *  - the 'hbase.master.wait.on.regionservers.maxtostart' number of
955   *    region servers is reached
956   *  - the 'hbase.master.wait.on.regionservers.mintostart' is reached AND
957   *   there have been no new region server in for
958   *      'hbase.master.wait.on.regionservers.interval' time AND
959   *   the 'hbase.master.wait.on.regionservers.timeout' is reached
960   *
961   * @throws InterruptedException
962   */
963  public void waitForRegionServers(MonitoredTask status)
964  throws InterruptedException {
965    final long interval = this.master.getConfiguration().
966      getLong(WAIT_ON_REGIONSERVERS_INTERVAL, 1500);
967    final long timeout = this.master.getConfiguration().
968      getLong(WAIT_ON_REGIONSERVERS_TIMEOUT, 4500);
969    int defaultMinToStart = 1;
970    if (BaseLoadBalancer.tablesOnMaster(master.getConfiguration())) {
971      // If we assign regions to master, we'd like to start
972      // at least another region server so that we don't
973      // assign all regions to master if other region servers
974      // don't come up in time.
975      defaultMinToStart = 2;
976    }
977    int minToStart = this.master.getConfiguration().
978      getInt(WAIT_ON_REGIONSERVERS_MINTOSTART, defaultMinToStart);
979    if (minToStart < 1) {
980      LOG.warn(String.format(
981        "The value of '%s' (%d) can not be less than 1, ignoring.",
982        WAIT_ON_REGIONSERVERS_MINTOSTART, minToStart));
983      minToStart = 1;
984    }
985    int maxToStart = this.master.getConfiguration().
986      getInt(WAIT_ON_REGIONSERVERS_MAXTOSTART, Integer.MAX_VALUE);
987    if (maxToStart < minToStart) {
988        LOG.warn(String.format(
989            "The value of '%s' (%d) is set less than '%s' (%d), ignoring.",
990            WAIT_ON_REGIONSERVERS_MAXTOSTART, maxToStart,
991            WAIT_ON_REGIONSERVERS_MINTOSTART, minToStart));
992        maxToStart = Integer.MAX_VALUE;
993    }
994
995    long now =  System.currentTimeMillis();
996    final long startTime = now;
997    long slept = 0;
998    long lastLogTime = 0;
999    long lastCountChange = startTime;
1000    int count = countOfRegionServers();
1001    int oldCount = 0;
1002    while (!this.master.isStopped() && count < maxToStart
1003        && (lastCountChange+interval > now || timeout > slept || count < minToStart)) {
1004      // Log some info at every interval time or if there is a change
1005      if (oldCount != count || lastLogTime+interval < now){
1006        lastLogTime = now;
1007        String msg =
1008          "Waiting for region servers count to settle; currently"+
1009            " checked in " + count + ", slept for " + slept + " ms," +
1010            " expecting minimum of " + minToStart + ", maximum of "+ maxToStart+
1011            ", timeout of "+timeout+" ms, interval of "+interval+" ms.";
1012        LOG.info(msg);
1013        status.setStatus(msg);
1014      }
1015
1016      // We sleep for some time
1017      final long sleepTime = 50;
1018      Thread.sleep(sleepTime);
1019      now =  System.currentTimeMillis();
1020      slept = now - startTime;
1021
1022      oldCount = count;
1023      count = countOfRegionServers();
1024      if (count != oldCount) {
1025        lastCountChange = now;
1026      }
1027    }
1028
1029    LOG.info("Finished waiting for region servers count to settle;" +
1030      " checked in " + count + ", slept for " + slept + " ms," +
1031      " expecting minimum of " + minToStart + ", maximum of "+ maxToStart+","+
1032      " master is "+ (this.master.isStopped() ? "stopped.": "running")
1033    );
1034  }

1216          + " failed because no RPC connection found to this server");
1217    } else {
1218      List<Pair<HRegionInfo, List<ServerName>>> regionUpdateInfos =
1219          new ArrayList<Pair<HRegionInfo, List<ServerName>>>();
1220      for (Entry<HRegionInfo, List<ServerName>> entry : favoredNodes.entrySet()) {
1221        regionUpdateInfos.add(new Pair<HRegionInfo, List<ServerName>>(entry.getKey(),
1222          entry.getValue()));
1223      }
1224      UpdateFavoredNodesRequest request =
1225        RequestConverter.buildUpdateFavoredNodesRequest(regionUpdateInfos);
1226      try {
