082/**
083 * The ServerManager class manages info about region servers.
084 * <p>
085 * Maintains lists of online and dead servers.  Processes the startups,
086 * shutdowns, and deaths of region servers.
087 * <p>
088 * Servers are distinguished in two different ways.  A given server has a
089 * location, specified by hostname and port, and of which there can only be one
090 * online at any given time.  A server instance is specified by the location
091 * (hostname and port) as well as the startcode (timestamp from when the server
092 * was started).  This is used to differentiate a restarted instance of a given
093 * server from the original instance.
094 * <p>
095 * If a sever is known not to be running any more, it is called dead. The dead
096 * server needs to be handled by a ServerShutdownHandler.  If the handler is not
097 * enabled yet, the server can't be handled right away so it is queued up.
098 * After the handler is enabled, the server will be submitted to a handler to handle.
099 * However, the handler may be just partially enabled.  If so,
100 * the server cannot be fully processed, and be queued up for further processing.
101 * A server is fully processed only after the handler is fully enabled
102 * and has completed the handling.
103 */

214  /**
215   * Let the server manager know a new regionserver has come online
216   * @param request the startup request
217   * @param versionNumber the version number of the new regionserver
218   * @param version the version of the new regionserver, could contain strings like "SNAPSHOT"
219   * @param ia the InetAddress from which request is received
220   * @return The ServerName we know this server as.
221   * @throws IOException
222   */
223  ServerName regionServerStartup(RegionServerStartupRequest request, int versionNumber,
224      String version, InetAddress ia) throws IOException {
225    // Test for case where we get a region startup message from a regionserver
226    // that has been quickly restarted but whose znode expiration handler has
227    // not yet run, or from a server whose fail we are currently processing.
228    // Test its host+port combo is present in serverAddressToServerInfo. If it
229    // is, reject the server and trigger its expiration. The next time it comes
230    // in, it should have been removed from serverAddressToServerInfo and queued
231    // for processing by ProcessServerShutdown.
232
233    final String hostname =
234      request.hasUseThisHostnameInstead() ? request.getUseThisHostnameInstead() : ia.getHostName();
235    ServerName sn = ServerName.valueOf(hostname, request.getPort(), request.getServerStartCode());
236    checkClockSkew(sn, request.getServerCurrentTime());
237    checkIsDead(sn, "STARTUP");
238    if (!checkAndRecordNewServer(sn, ServerMetricsBuilder.of(sn, versionNumber, version))) {
239      LOG.warn(
240        "THIS SHOULD NOT HAPPEN, RegionServerStartup" + " could not record the server: " + sn);
241    }
242    return sn;
243  }

245  /**
246   * Updates last flushed sequence Ids for the regions on server sn
247   * @param sn
248   * @param hsl
249   */
250  private void updateLastFlushedSequenceIds(ServerName sn, ServerMetrics hsl) {
251    for (Entry<byte[], RegionMetrics> entry : hsl.getRegionMetrics().entrySet()) {
252      byte[] encodedRegionName = Bytes.toBytes(RegionInfo.encodeRegionName(entry.getKey()));
253      Long existingValue = flushedSequenceIdByRegion.get(encodedRegionName);
254      long l = entry.getValue().getCompletedSequenceId();
255      // Don't let smaller sequence ids override greater sequence ids.
256      if (LOG.isTraceEnabled()) {
257        LOG.trace(Bytes.toString(encodedRegionName) + ", existingValue=" + existingValue +
258          ", completeSequenceId=" + l);
259      }
260      if (existingValue == null || (l != HConstants.NO_SEQNUM && l > existingValue)) {
261        flushedSequenceIdByRegion.put(encodedRegionName, l);
262      } else if (l != HConstants.NO_SEQNUM && l < existingValue) {
263        LOG.warn("RegionServer " + sn + " indicates a last flushed sequence id ("
264            + l + ") that is less than the previous last flushed sequence id ("
265            + existingValue + ") for region " + Bytes.toString(entry.getKey()) + " Ignoring.");
266      }
267      ConcurrentNavigableMap<byte[], Long> storeFlushedSequenceId =
268          computeIfAbsent(storeFlushedSequenceIdsByRegion, encodedRegionName,
269            () -> new ConcurrentSkipListMap<>(Bytes.BYTES_COMPARATOR));
270      for (Entry<byte[], Long> storeSeqId : entry.getValue().getStoreSequenceId().entrySet()) {
271        byte[] family = storeSeqId.getKey();
272        existingValue = storeFlushedSequenceId.get(family);
273        l = storeSeqId.getValue();
274        if (LOG.isTraceEnabled()) {
275          LOG.trace(Bytes.toString(encodedRegionName) + ", family=" + Bytes.toString(family) +
276            ", existingValue=" + existingValue + ", completeSequenceId=" + l);
277        }
278        // Don't let smaller sequence ids override greater sequence ids.
279        if (existingValue == null || (l != HConstants.NO_SEQNUM && l > existingValue.longValue())) {
280          storeFlushedSequenceId.put(family, l);
281        }
282      }
283    }
284  }

286  @VisibleForTesting
287  public void regionServerReport(ServerName sn,
288    ServerMetrics sl) throws YouAreDeadException {
289    checkIsDead(sn, "REPORT");
290    if (null == this.onlineServers.replace(sn, sl)) {
291      // Already have this host+port combo and its just different start code?
292      // Just let the server in. Presume master joining a running cluster.
293      // recordNewServer is what happens at the end of reportServerStartup.
294      // The only thing we are skipping is passing back to the regionserver
295      // the ServerName to use. Here we presume a master has already done
296      // that so we'll press on with whatever it gave us for ServerName.
297      if (!checkAndRecordNewServer(sn, sl)) {
298        LOG.info("RegionServerReport ignored, could not record the server: " + sn);
299        return; // Not recorded, so no need to move on
300      }
301    }
302    updateLastFlushedSequenceIds(sn, sl);
303  }

305  /**
306   * Check is a server of same host and port already exists,
307   * if not, or the existed one got a smaller start code, record it.
308   *
309   * @param serverName the server to check and record
310   * @param sl the server load on the server
311   * @return true if the server is recorded, otherwise, false
312   */
313  boolean checkAndRecordNewServer(final ServerName serverName, final ServerMetrics sl) {
314    ServerName existingServer = null;
315    synchronized (this.onlineServers) {
316      existingServer = findServerWithSameHostnamePortWithLock(serverName);
317      if (existingServer != null && (existingServer.getStartcode() > serverName.getStartcode())) {
318        LOG.info("Server serverName=" + serverName + " rejected; we already have "
319            + existingServer.toString() + " registered with same hostname and port");
320        return false;
321      }
322      recordNewServerWithLock(serverName, sl);
323    }
324
325    // Tell our listeners that a server was added
326    if (!this.listeners.isEmpty()) {
327      for (ServerListener listener : this.listeners) {
328        listener.serverAdded(serverName);
329      }
330    }
331
332    // Note that we assume that same ts means same server, and don't expire in that case.
333    //  TODO: ts can theoretically collide due to clock shifts, so this is a bit hacky.
334    if (existingServer != null &&
335        (existingServer.getStartcode() < serverName.getStartcode())) {
336      LOG.info("Triggering server recovery; existingServer " +
337          existingServer + " looks stale, new server:" + serverName);
338      expireServer(existingServer);
339    }
340    return true;
341  }

343  /**
344   * Find out the region servers crashed between the crash of the previous master instance and the
345   * current master instance and schedule SCP for them.
346   * <p/>
347   * Since the {@code RegionServerTracker} has already helped us to construct the online servers set
348   * by scanning zookeeper, now we can compare the online servers with {@code liveServersFromWALDir}
349   * to find out whether there are servers which are already dead.
350   * <p/>
351   * Must be called inside the initialization method of {@code RegionServerTracker} to avoid
352   * concurrency issue.
353   * @param deadServersFromPE the region servers which already have SCP associated.
354   * @param liveServersFromWALDir the live region servers from wal directory.
355   */
356  void findOutDeadServersAndProcess(Set<ServerName> deadServersFromPE,
357      Set<ServerName> liveServersFromWALDir) {
358    deadServersFromPE.forEach(deadservers::add);
359    liveServersFromWALDir.stream().filter(sn -> !onlineServers.containsKey(sn))
360      .forEach(this::expireServer);
361  }

363  /**
364   * Checks if the clock skew between the server and the master. If the clock skew exceeds the
365   * configured max, it will throw an exception; if it exceeds the configured warning threshold,
366   * it will log a warning but start normally.
367   * @param serverName Incoming servers's name
368   * @param serverCurrentTime
369   * @throws ClockOutOfSyncException if the skew exceeds the configured max value
370   */
371  private void checkClockSkew(final ServerName serverName, final long serverCurrentTime)
372      throws ClockOutOfSyncException {
373    long skew = Math.abs(System.currentTimeMillis() - serverCurrentTime);
374    if (skew > maxSkew) {
375      String message = "Server " + serverName + " has been " +
376        "rejected; Reported time is too far out of sync with master.  " +
377        "Time difference of " + skew + "ms > max allowed of " + maxSkew + "ms";
378      LOG.warn(message);
379      throw new ClockOutOfSyncException(message);
380    } else if (skew > warningSkew){
381      String message = "Reported time for server " + serverName + " is out of sync with master " +
382        "by " + skew + "ms. (Warning threshold is " + warningSkew + "ms; " +
383        "error threshold is " + maxSkew + "ms)";
384      LOG.warn(message);
385    }
386  }

388  /**
389   * If this server is on the dead list, reject it with a YouAreDeadException.
390   * If it was dead but came back with a new start code, remove the old entry
391   * from the dead list.
392   * @param what START or REPORT
393   */
394  private void checkIsDead(final ServerName serverName, final String what)
395      throws YouAreDeadException {
396    if (this.deadservers.isDeadServer(serverName)) {
397      // host name, port and start code all match with existing one of the
398      // dead servers. So, this server must be dead.
399      String message = "Server " + what + " rejected; currently processing " +
400          serverName + " as dead server";
401      LOG.debug(message);
402      throw new YouAreDeadException(message);
403    }
404    // remove dead server with same hostname and port of newly checking in rs after master
405    // initialization.See HBASE-5916 for more information.
406    if ((this.master == null || this.master.isInitialized())
407        && this.deadservers.cleanPreviousInstance(serverName)) {
408      // This server has now become alive after we marked it as dead.
409      // We removed it's previous entry from the dead list to reflect it.
410      LOG.debug(what + ":" + " Server " + serverName + " came back up," +
411          " removed it from the dead servers list");
412    }
413  }

415  /**
416   * Assumes onlineServers is locked.
417   * @return ServerName with matching hostname and port.
418   */
419  private ServerName findServerWithSameHostnamePortWithLock(
420      final ServerName serverName) {
421    ServerName end = ServerName.valueOf(serverName.getHostname(), serverName.getPort(),
422        Long.MAX_VALUE);
423
424    ServerName r = onlineServers.lowerKey(end);
425    if (r != null) {
426      if (ServerName.isSameAddress(r, serverName)) {
427        return r;
428      }
429    }
430    return null;
431  }

449  public RegionStoreSequenceIds getLastFlushedSequenceId(byte[] encodedRegionName) {
450    RegionStoreSequenceIds.Builder builder = RegionStoreSequenceIds.newBuilder();
451    Long seqId = flushedSequenceIdByRegion.get(encodedRegionName);
452    builder.setLastFlushedSequenceId(seqId != null ? seqId.longValue() : HConstants.NO_SEQNUM);
453    Map<byte[], Long> storeFlushedSequenceId =
454        storeFlushedSequenceIdsByRegion.get(encodedRegionName);
455    if (storeFlushedSequenceId != null) {
456      for (Map.Entry<byte[], Long> entry : storeFlushedSequenceId.entrySet()) {
457        builder.addStoreSequenceId(StoreSequenceId.newBuilder()
458            .setFamilyName(UnsafeByteOperations.unsafeWrap(entry.getKey()))
459            .setSequenceId(entry.getValue().longValue()).build());
460      }
461    }
462    return builder.build();
463  }

477   * @return the average load
478   */
479  public double getAverageLoad() {
480    int totalLoad = 0;
481    int numServers = 0;
482    for (ServerMetrics sl : this.onlineServers.values()) {
483      numServers++;
484      totalLoad += sl.getRegionMetrics().size();
485    }
486    return numServers == 0 ? 0 :
487      (double)totalLoad / (double)numServers;
488  }

518  void letRegionServersShutdown() {
519    long previousLogTime = 0;
520    ServerName sn = master.getServerName();
521    ZKWatcher zkw = master.getZooKeeper();
522    int onlineServersCt;
523    while ((onlineServersCt = onlineServers.size()) > 0){
524
525      if (System.currentTimeMillis() > (previousLogTime + 1000)) {
526        Set<ServerName> remainingServers = onlineServers.keySet();
527        synchronized (onlineServers) {
528          if (remainingServers.size() == 1 && remainingServers.contains(sn)) {
529            // Master will delete itself later.
530            return;
531          }
532        }
533        StringBuilder sb = new StringBuilder();
534        // It's ok here to not sync on onlineServers - merely logging
535        for (ServerName key : remainingServers) {
536          if (sb.length() > 0) {
537            sb.append(", ");
538          }
539          sb.append(key);
540        }
541        LOG.info("Waiting on regionserver(s) " + sb.toString());
542        previousLogTime = System.currentTimeMillis();
543      }
544
545      try {
546        List<String> servers = getRegionServersInZK(zkw);
547        if (servers == null || servers.isEmpty() || (servers.size() == 1
548            && servers.contains(sn.toString()))) {
549          LOG.info("ZK shows there is only the master self online, exiting now");
550          // Master could have lost some ZK events, no need to wait more.
551          break;
552        }
553      } catch (KeeperException ke) {
554        LOG.warn("Failed to list regionservers", ke);
555        // ZK is malfunctioning, don't hang here
556        break;
557      }
558      synchronized (onlineServers) {
559        try {
560          if (onlineServersCt == onlineServers.size()) onlineServers.wait(100);
561        } catch (InterruptedException ignored) {
562          // continue
563        }
564      }
565    }
566  }

573  /**
574   * Expire the passed server. Add it to list of dead servers and queue a shutdown processing.
575   * @return True if we queued a ServerCrashProcedure else false if we did not (could happen for
576   *         many reasons including the fact that its this server that is going down or we already
577   *         have queued an SCP for this server or SCP processing is currently disabled because we
578   *         are in startup phase).
579   */
580  public synchronized boolean expireServer(final ServerName serverName) {
581    // THIS server is going down... can't handle our own expiration.
582    if (serverName.equals(master.getServerName())) {
583      if (!(master.isAborted() || master.isStopped())) {
584        master.stop("We lost our znode?");
585      }
586      return false;
587    }
588    if (this.deadservers.isDeadServer(serverName)) {
589      LOG.warn("Expiration called on {} but crash processing already in progress", serverName);
590      return false;
591    }
592    moveFromOnlineToDeadServers(serverName);
593
594    // If cluster is going down, yes, servers are going to be expiring; don't
595    // process as a dead server
596    if (isClusterShutdown()) {
597      LOG.info("Cluster shutdown set; " + serverName +
598        " expired; onlineServers=" + this.onlineServers.size());
599      if (this.onlineServers.isEmpty()) {
600        master.stop("Cluster shutdown set; onlineServer=0");
601      }
602      return false;
603    }
604    LOG.info("Processing expiration of " + serverName + " on " + this.master.getServerName());
605    master.getAssignmentManager().submitServerCrash(serverName, true);
606
607    // Tell our listeners that a server was removed
608    if (!this.listeners.isEmpty()) {
609      for (ServerListener listener : this.listeners) {
610        listener.serverRemoved(serverName);
611      }
612    }
613    // trigger a persist of flushedSeqId
614    if (flushedSeqIdFlusher != null) {
615      flushedSeqIdFlusher.triggerNow();
616    }
617    return true;
618  }

620  @VisibleForTesting
621  public void moveFromOnlineToDeadServers(final ServerName sn) {
622    synchronized (onlineServers) {
623      if (!this.onlineServers.containsKey(sn)) {
624        LOG.warn("Expiration of " + sn + " but server not online");
625      }
626      // Remove the server from the known servers lists and update load info BUT
627      // add to deadservers first; do this so it'll show in dead servers list if
628      // not in online servers list.
629      this.deadservers.add(sn);
630      this.onlineServers.remove(sn);
631      onlineServers.notifyAll();
632    }
633    this.rsAdmins.remove(sn);
634  }

657    // Warn if the server (sn) is not online.  ServerName is of the form:

673    return this.drainingServers.add(sn);

704  /**
705   * Contacts a region server and waits up to timeout ms
706   * to close the region.  This bypasses the active hmaster.
707   */
708  public static void closeRegionSilentlyAndWait(ClusterConnection connection,
709    ServerName server, RegionInfo region, long timeout) throws IOException, InterruptedException {
710    AdminService.BlockingInterface rs = connection.getAdmin(server);
711    HBaseRpcController controller = connection.getRpcControllerFactory().newController();
712    try {
713      ProtobufUtil.closeRegion(controller, rs, server, region.getRegionName());
714    } catch (IOException e) {
715      LOG.warn("Exception when closing region: " + region.getRegionNameAsString(), e);
716    }
717    long expiration = timeout + System.currentTimeMillis();
718    while (System.currentTimeMillis() < expiration) {
719      controller.reset();
720      try {
721        RegionInfo rsRegion =
722          ProtobufUtil.getRegionInfo(controller, rs, region.getRegionName());
723        if (rsRegion == null) return;
724      } catch (IOException ioe) {
725        if (ioe instanceof NotServingRegionException) // no need to retry again
726          return;
727        LOG.warn("Exception when retrieving regioninfo from: "
728          + region.getRegionNameAsString(), ioe);
729      }
730      Thread.sleep(1000);
731    }
732    throw new IOException("Region " + region + " failed to close within"
733        + " timeout " + timeout);
734  }

739   * @throws IOException

742  public AdminService.BlockingInterface getRsAdmin(final ServerName sn)

758  /**
759   * Calculate min necessary to start. This is not an absolute. It is just
760   * a friction that will cause us hang around a bit longer waiting on
761   * RegionServers to check-in.
762   */
763  private int getMinToStart() {
764    // One server should be enough to get us off the ground.
765    int requiredMinToStart = 1;
766    if (LoadBalancer.isTablesOnMaster(master.getConfiguration())) {
767      if (LoadBalancer.isSystemTablesOnlyOnMaster(master.getConfiguration())) {
768        // If Master is carrying regions but NOT user-space regions, it
769        // still shows as a 'server'. We need at least one more server to check
770        // in before we can start up so set defaultMinToStart to 2.
771        requiredMinToStart = requiredMinToStart + 1;
772      }
773    }
774    int minToStart = this.master.getConfiguration().getInt(WAIT_ON_REGIONSERVERS_MINTOSTART, -1);
775    // Ensure we are never less than requiredMinToStart else stuff won't work.
776    return minToStart == -1 || minToStart < requiredMinToStart? requiredMinToStart: minToStart;
777  }

779  /**
780   * Wait for the region servers to report in.
781   * We will wait until one of this condition is met:
782   *  - the master is stopped
783   *  - the 'hbase.master.wait.on.regionservers.maxtostart' number of
784   *    region servers is reached
785   *  - the 'hbase.master.wait.on.regionservers.mintostart' is reached AND
786   *   there have been no new region server in for
787   *      'hbase.master.wait.on.regionservers.interval' time AND
788   *   the 'hbase.master.wait.on.regionservers.timeout' is reached
789   *
790   * @throws InterruptedException
791   */
792  public void waitForRegionServers(MonitoredTask status) throws InterruptedException {
793    final long interval = this.master.getConfiguration().
794        getLong(WAIT_ON_REGIONSERVERS_INTERVAL, 1500);
795    final long timeout = this.master.getConfiguration().
796        getLong(WAIT_ON_REGIONSERVERS_TIMEOUT, 4500);
797    // Min is not an absolute; just a friction making us wait longer on server checkin.
798    int minToStart = getMinToStart();
799    int maxToStart = this.master.getConfiguration().
800        getInt(WAIT_ON_REGIONSERVERS_MAXTOSTART, Integer.MAX_VALUE);
801    if (maxToStart < minToStart) {
802      LOG.warn(String.format("The value of '%s' (%d) is set less than '%s' (%d), ignoring.",
803          WAIT_ON_REGIONSERVERS_MAXTOSTART, maxToStart,
804          WAIT_ON_REGIONSERVERS_MINTOSTART, minToStart));
805      maxToStart = Integer.MAX_VALUE;
806    }
807
808    long now =  System.currentTimeMillis();
809    final long startTime = now;
810    long slept = 0;
811    long lastLogTime = 0;
812    long lastCountChange = startTime;
813    int count = countOfRegionServers();
814    int oldCount = 0;
815    // This while test is a little hard to read. We try to comment it in below but in essence:
816    // Wait if Master is not stopped and the number of regionservers that have checked-in is
817    // less than the maxToStart. Both of these conditions will be true near universally.
818    // Next, we will keep cycling if ANY of the following three conditions are true:
819    // 1. The time since a regionserver registered is < interval (means servers are actively checking in).
820    // 2. We are under the total timeout.
821    // 3. The count of servers is < minimum.
822    for (ServerListener listener: this.listeners) {
823      listener.waiting();
824    }
825    while (!this.master.isStopped() && !isClusterShutdown() && count < maxToStart &&
826        ((lastCountChange + interval) > now || timeout > slept || count < minToStart)) {
827      // Log some info at every interval time or if there is a change
828      if (oldCount != count || lastLogTime + interval < now) {
829        lastLogTime = now;
830        String msg =
831            "Waiting on regionserver count=" + count + "; waited="+
832                slept + "ms, expecting min=" + minToStart + " server(s), max="+ getStrForMax(maxToStart) +
833                " server(s), " + "timeout=" + timeout + "ms, lastChange=" + (lastCountChange - now) + "ms";
834        LOG.info(msg);
835        status.setStatus(msg);
836      }
837
838      // We sleep for some time
839      final long sleepTime = 50;
840      Thread.sleep(sleepTime);
841      now =  System.currentTimeMillis();
842      slept = now - startTime;
843
844      oldCount = count;
845      count = countOfRegionServers();
846      if (count != oldCount) {
847        lastCountChange = now;
848      }
849    }
850    // Did we exit the loop because cluster is going down?
851    if (isClusterShutdown()) {
852      this.master.stop("Cluster shutdown");
853    }
854    LOG.info("Finished waiting on RegionServer count=" + count + "; waited=" + slept + "ms," +
855        " expected min=" + minToStart + " server(s), max=" +  getStrForMax(maxToStart) + " server(s),"+
856        " master is "+ (this.master.isStopped() ? "stopped.": "running"));
857  }

1042  /**
1043   * Persist last flushed sequence id of each region to HDFS
1044   * @throws IOException if persit to HDFS fails
1045   */
1046  private void persistRegionLastFlushedSequenceIds() throws IOException {
1047    if (isFlushSeqIdPersistInProgress) {
1048      return;
1049    }
1050    isFlushSeqIdPersistInProgress = true;
1051    try {
1052      Configuration conf = master.getConfiguration();
1053      Path rootDir = FSUtils.getRootDir(conf);
1054      Path lastFlushedSeqIdPath = new Path(rootDir, LAST_FLUSHED_SEQ_ID_FILE);
1055      FileSystem fs = FileSystem.get(conf);
1056      if (fs.exists(lastFlushedSeqIdPath)) {
1057        LOG.info("Rewriting .lastflushedseqids file at: "
1058            + lastFlushedSeqIdPath);
1059        if (!fs.delete(lastFlushedSeqIdPath, false)) {
1060          throw new IOException("Unable to remove existing "
1061              + lastFlushedSeqIdPath);
1062        }
1063      } else {
1064        LOG.info("Writing .lastflushedseqids file at: " + lastFlushedSeqIdPath);
1065      }
1066      FSDataOutputStream out = fs.create(lastFlushedSeqIdPath);
1067      FlushedSequenceId.Builder flushedSequenceIdBuilder =
1068          FlushedSequenceId.newBuilder();
1069      try {
1070        for (Entry<byte[], Long> entry : flushedSequenceIdByRegion.entrySet()) {
1071          FlushedRegionSequenceId.Builder flushedRegionSequenceIdBuilder =
1072              FlushedRegionSequenceId.newBuilder();
1073          flushedRegionSequenceIdBuilder.setRegionEncodedName(
1074              ByteString.copyFrom(entry.getKey()));
1075          flushedRegionSequenceIdBuilder.setSeqId(entry.getValue());
1076          ConcurrentNavigableMap<byte[], Long> storeSeqIds =
1077              storeFlushedSequenceIdsByRegion.get(entry.getKey());
1078          if (storeSeqIds != null) {
1079            for (Entry<byte[], Long> store : storeSeqIds.entrySet()) {
1080              FlushedStoreSequenceId.Builder flushedStoreSequenceIdBuilder =
1081                  FlushedStoreSequenceId.newBuilder();
1082              flushedStoreSequenceIdBuilder.setFamily(ByteString.copyFrom(store.getKey()));
1083              flushedStoreSequenceIdBuilder.setSeqId(store.getValue());
1084              flushedRegionSequenceIdBuilder.addStores(flushedStoreSequenceIdBuilder);
1085            }
1086          }
1087          flushedSequenceIdBuilder.addRegionSequenceId(flushedRegionSequenceIdBuilder);
1088        }
1089        flushedSequenceIdBuilder.build().writeDelimitedTo(out);
1090      } finally {
1091        if (out != null) {
1092          out.close();
1093        }
1094      }
1095    } finally {
1096      isFlushSeqIdPersistInProgress = false;
1097    }
1098  }

1100  /**
1101   * Load last flushed sequence id of each region from HDFS, if persisted
1102   */
1103  public void loadLastFlushedSequenceIds() throws IOException {
1104    if (!persistFlushedSequenceId) {
1105      return;
1106    }
1107    Configuration conf = master.getConfiguration();
1108    Path rootDir = FSUtils.getRootDir(conf);
1109    Path lastFlushedSeqIdPath = new Path(rootDir, LAST_FLUSHED_SEQ_ID_FILE);
1110    FileSystem fs = FileSystem.get(conf);
1111    if (!fs.exists(lastFlushedSeqIdPath)) {
1112      LOG.info("No .lastflushedseqids found at" + lastFlushedSeqIdPath
1113          + " will record last flushed sequence id"
1114          + " for regions by regionserver report all over again");
1115      return;
1116    } else {
1117      LOG.info("begin to load .lastflushedseqids at " + lastFlushedSeqIdPath);
1118    }
1119    FSDataInputStream in = fs.open(lastFlushedSeqIdPath);
1120    try {
1121      FlushedSequenceId flushedSequenceId =
1122          FlushedSequenceId.parseDelimitedFrom(in);
1123      for (FlushedRegionSequenceId flushedRegionSequenceId : flushedSequenceId
1124          .getRegionSequenceIdList()) {
1125        byte[] encodedRegionName = flushedRegionSequenceId
1126            .getRegionEncodedName().toByteArray();
1127        flushedSequenceIdByRegion
1128            .putIfAbsent(encodedRegionName, flushedRegionSequenceId.getSeqId());
1129        if (flushedRegionSequenceId.getStoresList() != null
1130            && flushedRegionSequenceId.getStoresList().size() != 0) {
1131          ConcurrentNavigableMap<byte[], Long> storeFlushedSequenceId =
1132              computeIfAbsent(storeFlushedSequenceIdsByRegion, encodedRegionName,
1133                () -> new ConcurrentSkipListMap<>(Bytes.BYTES_COMPARATOR));
1134          for (FlushedStoreSequenceId flushedStoreSequenceId : flushedRegionSequenceId
1135              .getStoresList()) {
1136            storeFlushedSequenceId
1137                .put(flushedStoreSequenceId.getFamily().toByteArray(),
1138                    flushedStoreSequenceId.getSeqId());
1139          }
1140        }
1141      }
1142    } finally {
1143      in.close();
1144    }
1145  }

1147  /**
1148   * Regions may have been removed between latest persist of FlushedSequenceIds
1149   * and master abort. So after loading FlushedSequenceIds from file, and after
1150   * meta loaded, we need to remove the deleted region according to RegionStates.
1151   */
1152  public void removeDeletedRegionFromLoadedFlushedSequenceIds() {
1153    RegionStates regionStates = master.getAssignmentManager().getRegionStates();
1154    Iterator<byte[]> it = flushedSequenceIdByRegion.keySet().iterator();
1155    while(it.hasNext()) {
1156      byte[] regionEncodedName = it.next();
1157      if (regionStates.getRegionState(Bytes.toStringBinary(regionEncodedName)) == null) {
1158        it.remove();
1159        storeFlushedSequenceIdsByRegion.remove(regionEncodedName);
1160      }
1161    }
1162  }
