037/**
038 * This is an optional Cost function designed to allow region count skew across RegionServers. A
039 * rule file is loaded from the local FS or HDFS before balancing. It contains lines of rules. A
040 * rule is composed of a regexp for hostname, and a limit. For example, we could have:
041 * <p>
042 * * rs[0-9] 200 * rs1[0-9] 50
043 * </p>
044 * RegionServers with hostname matching the first rules will have a limit of 200, and the others 50.
045 * If there's no match, a default is set. The costFunction is trying to fill all RegionServers
046 * linearly, meaning that if the global usage is at 50%, then all RegionServers should hold half of
047 * their capacity in terms of regions. In order to use this CostFunction, you need to set the
048 * following options:
049 * <ul>
050 * <li>hbase.master.balancer.stochastic.additionalCostFunctions</li>
051 * <li>hbase.master.balancer.stochastic.heterogeneousRegionCountRulesFile</li>
052 * <li>hbase.master.balancer.stochastic.heterogeneousRegionCountDefault</li>
053 * </ul>
054 * The rule file can be located on local FS or HDFS, depending on the prefix (file//: or hdfs://).
055 */
056@InterfaceAudience.Private
057public class HeterogeneousRegionCountCostFunction extends CostFunction {
058
059  /**
060   * configuration used for the path where the rule file is stored.
061   */
062  static final String HBASE_MASTER_BALANCER_HETEROGENEOUS_RULES_FILE =
063      "hbase.master.balancer.heterogeneousRegionCountRulesFile";
064  private static final Logger LOG =
065      LoggerFactory.getLogger(HeterogeneousRegionCountCostFunction.class);
066  /**
067   * Default rule to apply when the rule file is not found. Default to 200.
068   */
069  private static final String HBASE_MASTER_BALANCER_HETEROGENEOUS_RULES_DEFAULT =
070      "hbase.master.balancer.heterogeneousRegionCountDefault";
071  /**
072   * Cost for the function. Default to 500, can be changed.
073   */
074  private static final String REGION_COUNT_SKEW_COST_KEY =
075      "hbase.master.balancer.stochastic.heterogeneousRegionCountCost";
076  private static final float DEFAULT_REGION_COUNT_SKEW_COST = 500;
077  private final String rulesPath;
078
079  /**
080   * Contains the rules, key is the regexp for ServerName, value is the limit
081   */
082  private final Map<Pattern, Integer> limitPerRule;
083
084  /**
085   * This is a cache, used to not go through all the limitPerRule map when searching for limit
086   */
087  private final Map<ServerName, Integer> limitPerRS;
088  private final Configuration conf;
089  private int defaultNumberOfRegions;
090
091  /**
092   * Total capacity of regions for the cluster, based on the online RS and their associated rules
093   */
094  private int totalCapacity = 0;
095  double overallUsage;
096
097  HeterogeneousRegionCountCostFunction(final Configuration conf) {
098    this.conf = conf;
099    this.limitPerRS = new HashMap<>();
100    this.limitPerRule = new HashMap<>();
101    this.setMultiplier(conf.getFloat(REGION_COUNT_SKEW_COST_KEY, DEFAULT_REGION_COUNT_SKEW_COST));
102    this.rulesPath = conf.get(HBASE_MASTER_BALANCER_HETEROGENEOUS_RULES_FILE);
103    this.defaultNumberOfRegions =
104        conf.getInt(HBASE_MASTER_BALANCER_HETEROGENEOUS_RULES_DEFAULT, 200);
105
106    if (this.defaultNumberOfRegions < 0) {
107      LOG.warn("invalid configuration '" + HBASE_MASTER_BALANCER_HETEROGENEOUS_RULES_DEFAULT
108          + "'. Setting default to 200");
109      this.defaultNumberOfRegions = 200;
110    }
111    if (conf.getFloat(RegionCountSkewCostFunction.REGION_COUNT_SKEW_COST_KEY,
112      RegionCountSkewCostFunction.DEFAULT_REGION_COUNT_SKEW_COST) > 0) {
113      LOG.warn("regionCountCost is not set to 0, "
114          + " this will interfere with the HeterogeneousRegionCountCostFunction!");
115    }
116  }
117
118  /**
119   * Called once per LB invocation to give the cost function to initialize it's state, and perform
120   * any costly calculation.
121   */
122  @Override
123  void prepare(final BalancerClusterState cluster) {
124    this.cluster = cluster;
125    this.loadRules();
126  }
127
128  @Override
129  protected double cost() {
130    double cost = 0;
131    final double targetUsage = ((double) this.cluster.numRegions / (double) this.totalCapacity);
132
133    for (int i = 0; i < this.cluster.numServers; i++) {
134      // retrieve capacity for each RS
135      final ServerName sn = this.cluster.servers[i];
136      final double limit = this.limitPerRS.getOrDefault(sn, defaultNumberOfRegions);
137      final double nbrRegions = this.cluster.regionsPerServer[i].length;
138      final double usage = nbrRegions / limit;
139      if (usage > targetUsage) {
140        // cost is the number of regions above the local limit
141        final double localCost = (nbrRegions - Math.round(limit * targetUsage)) / limit;
142        cost += localCost;
143      }
144    }
145
146    return cost / (double) this.cluster.numServers;
147  }
148
149  /**
150   * used to load the rule files.
151   */
152  @RestrictedApi(explanation = "Should only be called in tests", link = "",
153    allowedOnPath = ".*(/src/test/.*|HeterogeneousRegionCountCostFunction).java")
154  void loadRules() {
155    final List<String> lines = readFile(this.rulesPath);
156    if (null == lines) {
157      LOG.warn("cannot load rules file, keeping latest rules file which has "
158          + this.limitPerRule.size() + " rules");
159      return;
160    }
161
162    LOG.info("loading rules file '" + this.rulesPath + "'");
163    this.limitPerRule.clear();
164    for (final String line : lines) {
165      try {
166        if (line.length() == 0) {
167          continue;
168        }
169        if (line.startsWith("#")) {
170          continue;
171        }
172        final String[] splits = line.split(" ");
173        if (splits.length != 2) {
174          throw new IOException(
175              "line '" + line + "' is malformated, " + "expected [regexp] [limit]. Skipping line");
176        }
177
178        final Pattern pattern = Pattern.compile(splits[0]);
179        final Integer limit = Integer.parseInt(splits[1]);
180        this.limitPerRule.put(pattern, limit);
181      } catch (final IOException | NumberFormatException | PatternSyntaxException e) {
182        LOG.error("error on line: " + e);
183      }
184    }
185    this.rebuildCache();
186  }
187
188  /**
189   * used to read the rule files from either HDFS or local FS
190   */
191  private List<String> readFile(final String filename) {
192    if (null == filename) {
193      return null;
194    }
195    try {
196      if (filename.startsWith("file:")) {
197        return readFileFromLocalFS(filename);
198      }
199      return readFileFromHDFS(filename);
200    } catch (IOException e) {
201      LOG.error("cannot read rules file located at ' " + filename + " ':" + e.getMessage());
202      return null;
203    }
204  }
205
206  /**
207   * used to read the rule files from HDFS
208   */
209  private List<String> readFileFromHDFS(final String filename) throws IOException {
210    final Path path = new Path(filename);
211    final FileSystem fs = FileSystem.get(this.conf);
212    final BufferedReader reader = new BufferedReader(new InputStreamReader(fs.open(path)));
213    return readLines(reader);
214  }
215
216  /**
217   * used to read the rule files from local FS
218   */
219  private List<String> readFileFromLocalFS(final String filename) throws IOException {
220    BufferedReader reader = new BufferedReader(new FileReader(filename));
221    return readLines(reader);
222  }
223
224  private List<String> readLines(BufferedReader reader) throws IOException {
225    final List<String> records = new ArrayList<>();
226    try {
227      String line;
228      while ((line = reader.readLine()) != null) {
229        records.add(line);
230      }
231    } finally {
232      reader.close();
233    }
234    return records;
235  }
236
237  /**
238   * Rebuild cache matching ServerNames and their capacity.
239   */
240  private void rebuildCache() {
241    LOG.debug("Rebuilding cache of capacity for each RS");
242    this.limitPerRS.clear();
243    this.totalCapacity = 0;
244    if (null == this.cluster) {
245      return;
246    }
247    for (int i = 0; i < this.cluster.numServers; i++) {
248      final ServerName sn = this.cluster.servers[i];
249      final int capacity = this.findLimitForRS(sn);
250      LOG.debug(sn.getHostname() + " can hold " + capacity + " regions");
251      this.totalCapacity += capacity;
252    }
253    overallUsage = (double) this.cluster.numRegions / (double) this.totalCapacity;
254    LOG.info("Cluster can hold " + this.cluster.numRegions + "/" + this.totalCapacity + " regions ("
255        + Math.round(overallUsage * 100) + "%)");
256    if (overallUsage >= 1) {
257      LOG.warn("Cluster is overused, {}", overallUsage);
258    }
259  }
260
261  /**
262   * Find the limit for a ServerName. If not found then return the default value
263   * @param serverName the server we are looking for
264   * @return the limit
265   */
266  int findLimitForRS(final ServerName serverName) {
267    boolean matched = false;
268    int limit = -1;
269    for (final Map.Entry<Pattern, Integer> entry : this.limitPerRule.entrySet()) {
270      if (entry.getKey().matcher(serverName.getHostname()).matches()) {
271        matched = true;
272        limit = entry.getValue();
273        break;
274      }
275    }
276    if (!matched) {
277      limit = this.defaultNumberOfRegions;
278    }
279    // Feeding cache
280    this.limitPerRS.put(serverName, limit);
281    return limit;
282  }
283
284  int getNumberOfRulesLoaded() {
285    return this.limitPerRule.size();
286  }
