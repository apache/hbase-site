054/**
055 * An implementation of the {@link org.apache.hadoop.hbase.master.LoadBalancer} that
056 * assigns favored nodes for each region. There is a Primary RegionServer that hosts
057 * the region, and then there is Secondary and Tertiary RegionServers. Currently, the
058 * favored nodes information is used in creating HDFS files - the Primary RegionServer
059 * passes the primary, secondary, tertiary node addresses as hints to the
060 * DistributedFileSystem API for creating files on the filesystem. These nodes are
061 * treated as hints by the HDFS to place the blocks of the file. This alleviates the
062 * problem to do with reading from remote nodes (since we can make the Secondary
063 * RegionServer as the new Primary RegionServer) after a region is recovered. This
064 * should help provide consistent read latencies for the regions even when their
065 * primary region servers die. This provides two
066 * {@link org.apache.hadoop.hbase.master.balancer.StochasticLoadBalancer.CandidateGenerator}
067 *
068 */
069public class FavoredStochasticBalancer extends StochasticLoadBalancer implements
070    FavoredNodesPromoter {
071
072  private static final Log LOG = LogFactory.getLog(FavoredStochasticBalancer.class);
073  private FavoredNodesManager fnm;

094  /*
095   * Round robin assignment: Segregate the regions into two types:
096   *
097   * 1. The regions that have favored node assignment where at least one of the favored node
098   * is still alive. In this case, try to adhere to the current favored nodes assignment as
099   * much as possible - i.e., if the current primary is gone, then make the secondary or
100   * tertiary as the new host for the region (based on their current load). Note that we don't
101   * change the favored node assignments here (even though one or more favored node is
102   * currently down). That will be done by the admin operations.
103   *
104   * 2. The regions that currently don't have favored node assignments. Generate favored nodes
105   * for them and then assign. Generate the primary fn in round robin fashion and generate
106   * secondary and tertiary as per favored nodes constraints.
107   */
108  @Override
109  public Map<ServerName, List<HRegionInfo>> roundRobinAssignment(List<HRegionInfo> regions,
110      List<ServerName> servers) throws HBaseIOException {
111
112    metricsBalancer.incrMiscInvocations();
113
114    Set<HRegionInfo> regionSet = Sets.newHashSet(regions);
115    Map<ServerName, List<HRegionInfo>> assignmentMap = assignMasterRegions(regions, servers);
116    if (assignmentMap != null && !assignmentMap.isEmpty()) {
117      servers = new ArrayList<>(servers);
118      // Guarantee not to put other regions on master
119      servers.remove(masterServerName);
120      List<HRegionInfo> masterRegions = assignmentMap.get(masterServerName);
121      if (!masterRegions.isEmpty()) {
122        for (HRegionInfo region: masterRegions) {
123          regionSet.remove(region);
124        }
125      }
126    }
127
128    if (regionSet.isEmpty()) {
129      return assignmentMap;
130    }
131
132    try {
133      FavoredNodeAssignmentHelper helper =
134          new FavoredNodeAssignmentHelper(servers, fnm.getRackManager());
135      helper.initialize();
136
137      Set<HRegionInfo> systemRegions = FavoredNodesManager.filterNonFNApplicableRegions(regionSet);
138      regionSet.removeAll(systemRegions);
139
140      // Assign all system regions
141      Map<ServerName, List<HRegionInfo>> systemAssignments =
142        super.roundRobinAssignment(Lists.newArrayList(systemRegions), servers);
143
144      // Segregate favored and non-favored nodes regions and assign accordingly.
145      Pair<Map<ServerName,List<HRegionInfo>>, List<HRegionInfo>> segregatedRegions =
146        segregateRegionsAndAssignRegionsWithFavoredNodes(regionSet, servers);
147      Map<ServerName, List<HRegionInfo>> regionsWithFavoredNodesMap = segregatedRegions.getFirst();
148      Map<ServerName, List<HRegionInfo>> regionsWithoutFN =
149        generateFNForRegionsWithoutFN(helper, segregatedRegions.getSecond());
150
151      // merge the assignment maps
152      mergeAssignmentMaps(assignmentMap, systemAssignments);
153      mergeAssignmentMaps(assignmentMap, regionsWithFavoredNodesMap);
154      mergeAssignmentMaps(assignmentMap, regionsWithoutFN);
155
156    } catch (Exception ex) {
157      throw new HBaseIOException("Encountered exception while doing favored-nodes assignment "
158        + ex + " Falling back to regular assignment", ex);
159    }
160    return assignmentMap;
161  }
162
163  private void mergeAssignmentMaps(Map<ServerName, List<HRegionInfo>> assignmentMap,
164      Map<ServerName, List<HRegionInfo>> otherAssignments) {
165
166    if (otherAssignments == null || otherAssignments.isEmpty()) {
167      return;
168    }
169
170    for (Entry<ServerName, List<HRegionInfo>> entry : otherAssignments.entrySet()) {
171      ServerName sn = entry.getKey();
172      List<HRegionInfo> regionsList = entry.getValue();
173      if (assignmentMap.get(sn) == null) {
174        assignmentMap.put(sn, Lists.newArrayList(regionsList));
175      } else {
176        assignmentMap.get(sn).addAll(regionsList);
177      }
178    }
179  }
180
181  private Map<ServerName, List<HRegionInfo>> generateFNForRegionsWithoutFN(
182      FavoredNodeAssignmentHelper helper, List<HRegionInfo> regions) throws IOException {
183
184    Map<ServerName, List<HRegionInfo>> assignmentMap = Maps.newHashMap();
185    Map<HRegionInfo, List<ServerName>> regionsNoFNMap;
186
187    if (regions.size() > 0) {
188      regionsNoFNMap = helper.generateFavoredNodesRoundRobin(assignmentMap, regions);
189      fnm.updateFavoredNodes(regionsNoFNMap);
190    }
191    return assignmentMap;
192  }

194  /*
195   * Return a pair - one with assignments when favored nodes are present and another with regions
196   * without favored nodes.
197   */
198  private Pair<Map<ServerName, List<HRegionInfo>>, List<HRegionInfo>>
199  segregateRegionsAndAssignRegionsWithFavoredNodes(Collection<HRegionInfo> regions,
200      List<ServerName> onlineServers) throws HBaseIOException {
201
202    // Since we expect FN to be present most of the time, lets create map with same size
203    Map<ServerName, List<HRegionInfo>> assignmentMapForFavoredNodes =
204        new HashMap<>(onlineServers.size());
205    List<HRegionInfo> regionsWithNoFavoredNodes = new ArrayList<>();
206
207    for (HRegionInfo region : regions) {
208      List<ServerName> favoredNodes = fnm.getFavoredNodes(region);
209      ServerName primaryHost = null;
210      ServerName secondaryHost = null;
211      ServerName tertiaryHost = null;
212
213      if (favoredNodes != null && !favoredNodes.isEmpty()) {
214        for (ServerName s : favoredNodes) {
215          ServerName serverWithLegitStartCode = getServerFromFavoredNode(onlineServers, s);
216          if (serverWithLegitStartCode != null) {
217            FavoredNodesPlan.Position position =
218                FavoredNodesPlan.getFavoredServerPosition(favoredNodes, s);
219            if (Position.PRIMARY.equals(position)) {
220              primaryHost = serverWithLegitStartCode;
221            } else if (Position.SECONDARY.equals(position)) {
222              secondaryHost = serverWithLegitStartCode;
223            } else if (Position.TERTIARY.equals(position)) {
224              tertiaryHost = serverWithLegitStartCode;
225            }
226          }
227        }
228        assignRegionToAvailableFavoredNode(assignmentMapForFavoredNodes, region, primaryHost,
229            secondaryHost, tertiaryHost);
230      } else {
231        regionsWithNoFavoredNodes.add(region);
232      }
233    }
234    return new Pair<>(assignmentMapForFavoredNodes, regionsWithNoFavoredNodes);
235  }

237  private void addRegionToMap(Map<ServerName, List<HRegionInfo>> assignmentMapForFavoredNodes,
238      HRegionInfo region, ServerName host) {
239
240    List<HRegionInfo> regionsOnServer;
241    if ((regionsOnServer = assignmentMapForFavoredNodes.get(host)) == null) {
242      regionsOnServer = Lists.newArrayList();
243      assignmentMapForFavoredNodes.put(host, regionsOnServer);
244    }
245    regionsOnServer.add(region);
246  }

248  /*
249   * Get the ServerName for the FavoredNode. Since FN's startcode is -1, we could want to get the
250   * ServerName with the correct start code from the list of provided servers.
251   */
252  private ServerName getServerFromFavoredNode(List<ServerName> servers, ServerName fn) {
253    for (ServerName server : servers) {
254      if (ServerName.isSameHostnameAndPort(fn, server)) {
255        return server;
256      }
257    }
258    return null;
259  }

261  /*
262   * Assign the region to primary if its available. If both secondary and tertiary are available,
263   * assign to the host which has less load. Else assign to secondary or tertiary whichever is
264   * available (in that order).
265   */
266  private void assignRegionToAvailableFavoredNode(
267      Map<ServerName, List<HRegionInfo>> assignmentMapForFavoredNodes, HRegionInfo region,
268      ServerName primaryHost, ServerName secondaryHost, ServerName tertiaryHost) {
269
270    if (primaryHost != null) {
271      addRegionToMap(assignmentMapForFavoredNodes, region, primaryHost);
272
273    } else if (secondaryHost != null && tertiaryHost != null) {
274
275      // Assign the region to the one with a lower load (both have the desired hdfs blocks)
276      ServerName s;
277      ServerLoad tertiaryLoad = super.services.getServerManager().getLoad(tertiaryHost);
278      ServerLoad secondaryLoad = super.services.getServerManager().getLoad(secondaryHost);
279      if (secondaryLoad != null && tertiaryLoad != null) {
280        if (secondaryLoad.getLoad() < tertiaryLoad.getLoad()) {
281          s = secondaryHost;
282        } else {
283          s = tertiaryHost;
284        }
285      } else {
286        // We don't have one/more load, lets just choose a random node
287        s = RANDOM.nextBoolean() ? secondaryHost : tertiaryHost;
288      }
289      addRegionToMap(assignmentMapForFavoredNodes, region, s);
290    } else if (secondaryHost != null) {
291      addRegionToMap(assignmentMapForFavoredNodes, region, secondaryHost);
292    } else if (tertiaryHost != null) {
293      addRegionToMap(assignmentMapForFavoredNodes, region, tertiaryHost);
294    } else {
295      // No favored nodes are online, lets assign to BOGUS server
296      addRegionToMap(assignmentMapForFavoredNodes, region, BOGUS_SERVER_NAME);
297    }
298  }

300  /*
301   * If we have favored nodes for a region, we will return one of the FN as destination. If
302   * favored nodes are not present for a region, we will generate and return one of the FN as
303   * destination. If we can't generate anything, lets fallback.
304   */
305  @Override
306  public ServerName randomAssignment(HRegionInfo regionInfo, List<ServerName> servers)
307      throws HBaseIOException {
308
309    if (servers != null && servers.contains(masterServerName)) {
310      if (shouldBeOnMaster(regionInfo)) {
311        metricsBalancer.incrMiscInvocations();
312        return masterServerName;
313      }
314      servers = new ArrayList<>(servers);
315      // Guarantee not to put other regions on master
316      servers.remove(masterServerName);
317    }
318
319    ServerName destination = null;
320    if (!FavoredNodesManager.isFavoredNodeApplicable(regionInfo)) {
321      return super.randomAssignment(regionInfo, servers);
322    }
323
324    metricsBalancer.incrMiscInvocations();
325
326    List<ServerName> favoredNodes = fnm.getFavoredNodes(regionInfo);
327    if (favoredNodes == null || favoredNodes.isEmpty()) {
328      // Generate new favored nodes and return primary
329      FavoredNodeAssignmentHelper helper = new FavoredNodeAssignmentHelper(servers, getConf());
330      helper.initialize();
331      try {
332        favoredNodes = helper.generateFavoredNodes(regionInfo);
333        updateFavoredNodesForRegion(regionInfo, favoredNodes);
334
335      } catch (IOException e) {
336        LOG.warn("Encountered exception while doing favored-nodes (random)assignment " + e);
337        throw new HBaseIOException(e);
338      }
339    }
340
341    List<ServerName> onlineServers = getOnlineFavoredNodes(servers, favoredNodes);
342    if (onlineServers.size() > 0) {
343      destination = onlineServers.get(RANDOM.nextInt(onlineServers.size()));
344    }
345
346    boolean alwaysAssign = getConf().getBoolean(FAVORED_ALWAYS_ASSIGN_REGIONS, true);
347    if (destination == null && alwaysAssign) {
348      LOG.warn("Can't generate FN for region: " + regionInfo + " falling back");
349      destination = super.randomAssignment(regionInfo, servers);
350    }
351    return destination;
352  }
353
354  private void updateFavoredNodesForRegion(HRegionInfo regionInfo, List<ServerName> newFavoredNodes)
355      throws IOException {
356    Map<HRegionInfo, List<ServerName>> regionFNMap = Maps.newHashMap();
357    regionFNMap.put(regionInfo, newFavoredNodes);
358    fnm.updateFavoredNodes(regionFNMap);
359  }

361  /*
362   * Reuse BaseLoadBalancer's retainAssignment, but generate favored nodes when its missing.
363   */
364  @Override
365  public Map<ServerName, List<HRegionInfo>> retainAssignment(Map<HRegionInfo, ServerName> regions,
366      List<ServerName> servers) throws HBaseIOException {
367
368    Map<ServerName, List<HRegionInfo>> assignmentMap = Maps.newHashMap();
369    Map<ServerName, List<HRegionInfo>> result = super.retainAssignment(regions, servers);
370    if (result == null || result.isEmpty()) {
371      LOG.warn("Nothing to assign to, probably no servers or no regions");
372      return null;
373    }
374
375    // Guarantee not to put other regions on master
376    if (servers != null && servers.contains(masterServerName)) {
377      servers = new ArrayList<>(servers);
378      servers.remove(masterServerName);
379    }
380
381    // Lets check if favored nodes info is in META, if not generate now.
382    FavoredNodeAssignmentHelper helper = new FavoredNodeAssignmentHelper(servers, getConf());
383    helper.initialize();
384
385    LOG.debug("Generating favored nodes for regions missing them.");
386    Map<HRegionInfo, List<ServerName>> regionFNMap = Maps.newHashMap();
387
388    try {
389      for (Entry<ServerName, List<HRegionInfo>> entry : result.entrySet()) {
390
391        ServerName sn = entry.getKey();
392        ServerName primary = ServerName.valueOf(sn.getHostname(), sn.getPort(), NON_STARTCODE);
393
394        for (HRegionInfo hri : entry.getValue()) {
395
396          if (FavoredNodesManager.isFavoredNodeApplicable(hri)) {
397            List<ServerName> favoredNodes = fnm.getFavoredNodes(hri);
398            if (favoredNodes == null || favoredNodes.size() < FAVORED_NODES_NUM) {
399
400              LOG.debug("Generating favored nodes for: " + hri + " with primary: " + primary);
401              ServerName[] secondaryAndTertiaryNodes = helper.getSecondaryAndTertiary(hri, primary);
402              if (secondaryAndTertiaryNodes != null && secondaryAndTertiaryNodes.length == 2) {
403                List<ServerName> newFavoredNodes = Lists.newArrayList();
404                newFavoredNodes.add(primary);
405                newFavoredNodes.add(ServerName.valueOf(secondaryAndTertiaryNodes[0].getHostname(),
406                    secondaryAndTertiaryNodes[0].getPort(), NON_STARTCODE));
407                newFavoredNodes.add(ServerName.valueOf(secondaryAndTertiaryNodes[1].getHostname(),
408                    secondaryAndTertiaryNodes[1].getPort(), NON_STARTCODE));
409                regionFNMap.put(hri, newFavoredNodes);
410                addRegionToMap(assignmentMap, hri, sn);
411
412              } else {
413                throw new HBaseIOException("Cannot generate secondary/tertiary FN for " + hri
414                  + " generated "
415                  + (secondaryAndTertiaryNodes != null ? secondaryAndTertiaryNodes : " nothing"));
416              }
417            } else {
418              List<ServerName> onlineFN = getOnlineFavoredNodes(servers, favoredNodes);
419              if (onlineFN.isEmpty()) {
420                // All favored nodes are dead, lets assign it to BOGUS
421                addRegionToMap(assignmentMap, hri, BOGUS_SERVER_NAME);
422              } else {
423                // Is primary not on FN? Less likely, but we can still take care of this.
424                if (FavoredNodesPlan.getFavoredServerPosition(favoredNodes, sn) != null) {
425                  addRegionToMap(assignmentMap, hri, sn);
426                } else {
427                  ServerName destination = onlineFN.get(RANDOM.nextInt(onlineFN.size()));
428                  LOG.warn("Region: " + hri + " not hosted on favored nodes: " + favoredNodes
429                    + " current: " + sn + " moving to: " + destination);
430                  addRegionToMap(assignmentMap, hri, destination);
431                }
432              }
433            }
434          } else {
435            addRegionToMap(assignmentMap, hri, sn);
436          }
437        }
438      }
439
440      if (!regionFNMap.isEmpty()) {
441        LOG.debug("Updating FN in meta for missing regions, count: " + regionFNMap.size());
442        fnm.updateFavoredNodes(regionFNMap);
443      }
444
445    } catch (IOException e) {
446      throw new HBaseIOException("Cannot generate/update FN for regions: " + regionFNMap.keySet());
447    }
448
449    return assignmentMap;
450  }

452  /*
453   * Return list of favored nodes that are online.
454   */
455  private List<ServerName> getOnlineFavoredNodes(List<ServerName> onlineServers,
456      List<ServerName> serversWithoutStartCodes) {
457    if (serversWithoutStartCodes == null) {
458      return null;
459    } else {
460      List<ServerName> result = Lists.newArrayList();
461      for (ServerName sn : serversWithoutStartCodes) {
462        for (ServerName online : onlineServers) {
463          if (ServerName.isSameHostnameAndPort(sn, online)) {
464            result.add(online);
465          }
466        }
467      }
468      return result;
469    }
470  }

476  /*
477   * Generate Favored Nodes for daughters during region split.
478   *
479   * If the parent does not have FN, regenerates them for the daughters.
480   *
481   * If the parent has FN, inherit two FN from parent for each daughter and generate the remaining.
482   * The primary FN for both the daughters should be the same as parent. Inherit the secondary
483   * FN from the parent but keep it different for each daughter. Choose the remaining FN
484   * randomly. This would give us better distribution over a period of time after enough splits.
485   */
486  @Override
487  public void generateFavoredNodesForDaughter(List<ServerName> servers, HRegionInfo parent,
488      HRegionInfo regionA, HRegionInfo regionB) throws IOException {
489
490    Map<HRegionInfo, List<ServerName>> result = new HashMap<>();
491    FavoredNodeAssignmentHelper helper = new FavoredNodeAssignmentHelper(servers, rackManager);
492    helper.initialize();
493
494    List<ServerName> parentFavoredNodes = fnm.getFavoredNodes(parent);
495    if (parentFavoredNodes == null) {
496      LOG.debug("Unable to find favored nodes for parent, " + parent
497          + " generating new favored nodes for daughter");
498      result.put(regionA, helper.generateFavoredNodes(regionA));
499      result.put(regionB, helper.generateFavoredNodes(regionB));
500
501    } else {
502
503      // Lets get the primary and secondary from parent for regionA
504      Set<ServerName> regionAFN =
505          getInheritedFNForDaughter(helper, parentFavoredNodes, PRIMARY, SECONDARY);
506      result.put(regionA, Lists.newArrayList(regionAFN));
507
508      // Lets get the primary and tertiary from parent for regionB
509      Set<ServerName> regionBFN =
510          getInheritedFNForDaughter(helper, parentFavoredNodes, PRIMARY, TERTIARY);
511      result.put(regionB, Lists.newArrayList(regionBFN));
512    }
513
514    fnm.updateFavoredNodes(result);
515  }
516
517  private Set<ServerName> getInheritedFNForDaughter(FavoredNodeAssignmentHelper helper,
518      List<ServerName> parentFavoredNodes, Position primary, Position secondary)
519      throws IOException {
520
521    Set<ServerName> daughterFN = Sets.newLinkedHashSet();
522    if (parentFavoredNodes.size() >= primary.ordinal()) {
523      daughterFN.add(parentFavoredNodes.get(primary.ordinal()));
524    }
525
526    if (parentFavoredNodes.size() >= secondary.ordinal()) {
527      daughterFN.add(parentFavoredNodes.get(secondary.ordinal()));
528    }
529
530    while (daughterFN.size() < FAVORED_NODES_NUM) {
531      ServerName newNode = helper.generateMissingFavoredNode(Lists.newArrayList(daughterFN));
532      daughterFN.add(newNode);
533    }
534    return daughterFN;
535  }

547  /*
548   * Pick favored nodes with the highest locality for a region with lowest locality.
549   */
550  private class FavoredNodeLocalityPicker extends CandidateGenerator {
551
552    @Override
553    protected Cluster.Action generate(Cluster cluster) {
554
555      int thisServer = pickRandomServer(cluster);
556      int thisRegion;
557      if (thisServer == -1) {
558        LOG.trace("Could not pick lowest local region server");
559        return Cluster.NullAction;
560      } else {
561        // Pick lowest local region on this server
562        thisRegion = pickLowestLocalRegionOnServer(cluster, thisServer);
563      }
564      if (thisRegion == -1) {
565        if (cluster.regionsPerServer[thisServer].length > 0) {
566          LOG.trace("Could not pick lowest local region even when region server held "
567            + cluster.regionsPerServer[thisServer].length + " regions");
568        }
569        return Cluster.NullAction;
570      }
571
572      HRegionInfo hri = cluster.regions[thisRegion];
573      List<ServerName> favoredNodes = fnm.getFavoredNodes(hri);
574      int otherServer;
575      if (favoredNodes == null) {
576        if (!FavoredNodesManager.isFavoredNodeApplicable(hri)) {
577          otherServer = pickOtherRandomServer(cluster, thisServer);
578        } else {
579          // No FN, ignore
580          LOG.trace("Ignoring, no favored nodes for region: " + hri);
581          return Cluster.NullAction;
582        }
583      } else {
584        // Pick other favored node with the highest locality
585        otherServer = getDifferentFavoredNode(cluster, favoredNodes, thisServer);
586      }
587      return getAction(thisServer, thisRegion, otherServer, -1);
588    }
589
590    private int getDifferentFavoredNode(Cluster cluster, List<ServerName> favoredNodes,
591        int currentServer) {
592      List<Integer> fnIndex = new ArrayList<>();
593      for (ServerName sn : favoredNodes) {
594        if (cluster.serversToIndex.containsKey(sn.getHostAndPort())) {
595          fnIndex.add(cluster.serversToIndex.get(sn.getHostAndPort()));
596        }
597      }
598      float locality = 0;
599      int highestLocalRSIndex = -1;
600      for (Integer index : fnIndex) {
601        if (index != currentServer) {
602          float temp = cluster.localityPerServer[index];
603          if (temp >= locality) {
604            locality = temp;
605            highestLocalRSIndex = index;
606          }
607        }
608      }
609      return highestLocalRSIndex;
610    }
611
612    private int pickLowestLocalRegionOnServer(Cluster cluster, int server) {
613      return cluster.getLowestLocalityRegionOnServer(server);
614    }
615  }
616
617  /*
618   * This is like LoadCandidateGenerator, but we choose appropriate FN for the region on the
619   * most loaded server.
620   */
621  class FavoredNodeLoadPicker extends CandidateGenerator {
622
623    @Override
624    Cluster.Action generate(Cluster cluster) {
625      cluster.sortServersByRegionCount();
626      int thisServer = pickMostLoadedServer(cluster);
627      int thisRegion = pickRandomRegion(cluster, thisServer, 0);
628      HRegionInfo hri = cluster.regions[thisRegion];
629      int otherServer;
630      List<ServerName> favoredNodes = fnm.getFavoredNodes(hri);
631      if (favoredNodes == null) {
632        if (!FavoredNodesManager.isFavoredNodeApplicable(hri)) {
633          otherServer = pickLeastLoadedServer(cluster, thisServer);
634        } else {
635          return Cluster.NullAction;
636        }
637      } else {
638        otherServer = pickLeastLoadedFNServer(cluster, favoredNodes, thisServer);
639      }
640      return getAction(thisServer, thisRegion, otherServer, -1);
641    }
642
643    private int pickLeastLoadedServer(final Cluster cluster, int thisServer) {
644      Integer[] servers = cluster.serverIndicesSortedByRegionCount;
645      int index;
646      for (index = 0; index < servers.length ; index++) {
647        if ((servers[index] != null) && servers[index] != thisServer) {
648          break;
649        }
650      }
651      return servers[index];
652    }
653
654    private int pickLeastLoadedFNServer(final Cluster cluster, List<ServerName> favoredNodes,
655        int currentServerIndex) {
656      List<Integer> fnIndex = new ArrayList<>();
657      for (ServerName sn : favoredNodes) {
658        if (cluster.serversToIndex.containsKey(sn.getHostAndPort())) {
659          fnIndex.add(cluster.serversToIndex.get(sn.getHostAndPort()));
660        }
661      }
662      int leastLoadedFN = -1;
663      int load = Integer.MAX_VALUE;
664      for (Integer index : fnIndex) {
665        if (index != currentServerIndex) {
666          int temp = cluster.getNumRegions(index);
667          if (temp < load) {
668            load = temp;
669            leastLoadedFN = index;
670          }
671        }
672      }
673      return leastLoadedFN;
674    }
675
676    private int pickMostLoadedServer(final Cluster cluster) {
677      Integer[] servers = cluster.serverIndicesSortedByRegionCount;
678      int index;
679      for (index = servers.length - 1; index > 0 ; index--) {
680        if (servers[index] != null) {
681          break;
682        }
683      }
684      return servers[index];
685    }
686  }

688  /*
689   * For all regions correctly assigned to favored nodes, we just use the stochastic balancer
690   * implementation. For the misplaced regions, we assign a bogus server to it and AM takes care.
691   */
692  @Override
693  public List<RegionPlan> balanceCluster(Map<ServerName, List<HRegionInfo>> clusterState) {
694
695    if (this.services != null) {
696
697      List<RegionPlan> regionPlans = Lists.newArrayList();
698      Map<ServerName, List<HRegionInfo>> correctAssignments = new HashMap<>();
699      int misplacedRegions = 0;
700
701      for (Entry<ServerName, List<HRegionInfo>> entry : clusterState.entrySet()) {
702        ServerName current = entry.getKey();
703        List<HRegionInfo> regions = Lists.newArrayList();
704        correctAssignments.put(current, regions);
705
706        for (HRegionInfo hri : entry.getValue()) {
707          List<ServerName> favoredNodes = fnm.getFavoredNodes(hri);
708          if (FavoredNodesPlan.getFavoredServerPosition(favoredNodes, current) != null ||
709              !FavoredNodesManager.isFavoredNodeApplicable(hri)) {
710            regions.add(hri);
711
712          } else {
713            // No favored nodes, lets unassign.
714            LOG.warn("Region not on favored nodes, unassign. Region: " + hri
715              + " current: " + current + " favored nodes: " + favoredNodes);
716            try {
717              this.services.getAssignmentManager().unassign(hri);
718            } catch (IOException e) {
719              LOG.warn("Failed unassign", e);
720              continue;
721            }
722            RegionPlan rp = new RegionPlan(hri, null, null);
723            regionPlans.add(rp);
724            misplacedRegions++;
725          }
726        }
727      }
728      LOG.debug("Found misplaced regions: " + misplacedRegions + ", not on favored nodes.");
729      List<RegionPlan> regionPlansFromBalance = super.balanceCluster(correctAssignments);
730      if (regionPlansFromBalance != null) {
731        regionPlans.addAll(regionPlansFromBalance);
732      }
733      return regionPlans;
734    } else {
735      return super.balanceCluster(clusterState);
736    }
737  }
