053/**
054 * <p>This is a best effort load balancer. Given a Cost function F(C) =&gt; x It will
055 * randomly try and mutate the cluster to Cprime. If F(Cprime) &lt; F(C) then the
056 * new cluster state becomes the plan. It includes costs functions to compute the cost of:</p>
057 * <ul>
058 * <li>Region Load</li>
059 * <li>Table Load</li>
060 * <li>Data Locality</li>
061 * <li>Memstore Sizes</li>
062 * <li>Storefile Sizes</li>
063 * </ul>
064 *
065 *
066 * <p>Every cost function returns a number between 0 and 1 inclusive; where 0 is the lowest cost
067 * best solution, and 1 is the highest possible cost and the worst solution.  The computed costs are
068 * scaled by their respective multipliers:</p>
069 *
070 * <ul>
071 *   <li>hbase.master.balancer.stochastic.regionLoadCost</li>
072 *   <li>hbase.master.balancer.stochastic.moveCost</li>
073 *   <li>hbase.master.balancer.stochastic.tableLoadCost</li>
074 *   <li>hbase.master.balancer.stochastic.localityCost</li>
075 *   <li>hbase.master.balancer.stochastic.memstoreSizeCost</li>
076 *   <li>hbase.master.balancer.stochastic.storefileSizeCost</li>
077 * </ul>
078 *
079 * <p>In addition to the above configurations, the balancer can be tuned by the following
080 * configuration values:</p>
081 * <ul>
082 *   <li>hbase.master.balancer.stochastic.maxMoveRegions which
083 *   controls what the max number of regions that can be moved in a single invocation of this
084 *   balancer.</li>
085 *   <li>hbase.master.balancer.stochastic.stepsPerRegion is the coefficient by which the number of
086 *   regions is multiplied to try and get the number of times the balancer will
087 *   mutate all servers.</li>
088 *   <li>hbase.master.balancer.stochastic.maxSteps which controls the maximum number of times that
089 *   the balancer will try and mutate all the servers. The balancer will use the minimum of this
090 *   value and the above computation.</li>
091 * </ul>
092 *
093 * <p>This balancer is best used with hbase.master.loadbalance.bytable set to false
094 * so that the balancer gets the full picture of all loads on the cluster.</p>
095 */
096@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)

265  /**
266   * Given the cluster state this will try and approach an optimal balance. This
267   * should always approach the optimal state given enough steps.
268   */
269  @Override
270  public synchronized List<RegionPlan> balanceCluster(Map<ServerName,
271    List<HRegionInfo>> clusterState) {
272    List<RegionPlan> plans = balanceMasterRegions(clusterState);
273    if (plans != null || clusterState == null || clusterState.size() <= 1) {
274      return plans;
275    }
276
277    if (masterServerName != null && clusterState.containsKey(masterServerName)) {
278      if (clusterState.size() <= 2) {
279        return null;
280      }
281      clusterState = new HashMap<ServerName, List<HRegionInfo>>(clusterState);
282      clusterState.remove(masterServerName);
283    }
284
285    // On clusters with lots of HFileLinks or lots of reference files,
286    // instantiating the storefile infos can be quite expensive.
287    // Allow turning this feature off if the locality cost is not going to
288    // be used in any computations.
289    RegionLocationFinder finder = null;
290    if (this.localityCost != null && this.localityCost.getMultiplier() > 0) {
291      finder = this.regionFinder;
292    }
293
294    //The clusterState that is given to this method contains the state
295    //of all the regions in the table(s) (that's true today)
296    // Keep track of servers to iterate through them.
297    Cluster cluster = new Cluster(clusterState, loads, finder, rackManager);
298
299    if (!needsBalance(cluster)) {
300      return null;
301    }
302
303    long startTime = EnvironmentEdgeManager.currentTime();
304
305    initCosts(cluster);
306
307    double currentCost = computeCost(cluster, Double.MAX_VALUE);
308    curOverallCost = currentCost;
309    for (int i = 0; i < this.curFunctionCosts.length; i++) {
310      curFunctionCosts[i] = tempFunctionCosts[i];
311    }
312
313    double initCost = currentCost;
314    double newCost = currentCost;
315
316    long computedMaxSteps = Math.min(this.maxSteps,
317        ((long)cluster.numRegions * (long)this.stepsPerRegion * (long)cluster.numServers));
318    // Perform a stochastic walk to see if we can get a good fit.
319    long step;
320
321    for (step = 0; step < computedMaxSteps; step++) {
322      int generatorIdx = RANDOM.nextInt(candidateGenerators.length);
323      CandidateGenerator p = candidateGenerators[generatorIdx];
324      Cluster.Action action = p.generate(cluster);
325
326      if (action.type == Type.NULL) {
327        continue;
328      }
329
330      cluster.doAction(action);
331      updateCostsWithAction(cluster, action);
332
333      newCost = computeCost(cluster, currentCost);
334
335      // Should this be kept?
336      if (newCost < currentCost) {
337        currentCost = newCost;
338
339        // save for JMX
340        curOverallCost = currentCost;
341        for (int i = 0; i < this.curFunctionCosts.length; i++) {
342          curFunctionCosts[i] = tempFunctionCosts[i];
343        }
344      } else {
345        // Put things back the way they were before.
346        // TODO: undo by remembering old values
347        Action undoAction = action.undoAction();
348        cluster.doAction(undoAction);
349        updateCostsWithAction(cluster, undoAction);
350      }
351
352      if (EnvironmentEdgeManager.currentTime() - startTime >
353          maxRunningTime) {
354        break;
355      }
356    }
357    long endTime = EnvironmentEdgeManager.currentTime();
358
359    metricsBalancer.balanceCluster(endTime - startTime);
360
361    // update costs metrics
362    updateStochasticCosts(tableName, curOverallCost, curFunctionCosts);
363    if (initCost > currentCost) {
364      plans = createRegionPlans(cluster);
365      if (LOG.isDebugEnabled()) {
366        LOG.debug("Finished computing new load balance plan.  Computation took "
367            + (endTime - startTime) + "ms to try " + step
368            + " different iterations.  Found a solution that moves "
369            + plans.size() + " regions; Going from a computed cost of "
370            + initCost + " to a new cost of " + currentCost);
371      }
372      
373      return plans;
374    }
375    if (LOG.isDebugEnabled()) {
376      LOG.debug("Could not find a better load balance plan.  Tried "
377          + step + " different configurations in " + (endTime - startTime)
378          + "ms, and did not find anything with a computed cost less than " + initCost);
379    }
380    return null;
381  }

383  /**
384   * update costs to JMX
385   */
386  private void updateStochasticCosts(TableName tableName, Double overall, Double[] subCosts) {
387    if (tableName == null) return;
388
389    // check if the metricsBalancer is MetricsStochasticBalancer before casting
390    if (metricsBalancer instanceof MetricsStochasticBalancer) {
391      MetricsStochasticBalancer balancer = (MetricsStochasticBalancer) metricsBalancer;
392      // overall cost
393      balancer.updateStochasticCost(tableName.getNameAsString(), 
394        "Overall", "Overall cost", overall);
395
396      // each cost function
397      for (int i = 0; i < costFunctions.length; i++) {
398        CostFunction costFunction = costFunctions[i];
399        String costFunctionName = costFunction.getClass().getSimpleName();
400        Double costPercent = (overall == 0) ? 0 : (subCosts[i] / overall);
401        // TODO: cost function may need a specific description
402        balancer.updateStochasticCost(tableName.getNameAsString(), costFunctionName,
403          "The percent of " + costFunctionName, costPercent);
404      }
405    }
406  }

409  /**
410   * Create all of the RegionPlan's needed to move from the initial cluster state to the desired
411   * state.
412   *
413   * @param cluster The state of the cluster
414   * @return List of RegionPlan's that represent the moves needed to get to desired final state.
415   */
416  private List<RegionPlan> createRegionPlans(Cluster cluster) {
417    List<RegionPlan> plans = new LinkedList<RegionPlan>();
418    for (int regionIndex = 0;
419         regionIndex < cluster.regionIndexToServerIndex.length; regionIndex++) {
420      int initialServerIndex = cluster.initialRegionIndexToServerIndex[regionIndex];
421      int newServerIndex = cluster.regionIndexToServerIndex[regionIndex];
422
423      if (initialServerIndex != newServerIndex) {
424        HRegionInfo region = cluster.regions[regionIndex];
425        ServerName initialServer = cluster.servers[initialServerIndex];
426        ServerName newServer = cluster.servers[newServerIndex];
427
428        if (LOG.isTraceEnabled()) {
429          LOG.trace("Moving Region " + region.getEncodedName() + " from server "
430              + initialServer.getHostname() + " to " + newServer.getHostname());
431        }
432        RegionPlan rp = new RegionPlan(region, initialServer, newServer);
433        plans.add(rp);
434      }
435    }
436    return plans;
437  }

439  /**
440   * Store the current region loads.
441   */
442  private synchronized void updateRegionLoad() {
443    // We create a new hashmap so that regions that are no longer there are removed.
444    // However we temporarily need the old loads so we can use them to keep the rolling average.
445    Map<String, Deque<RegionLoad>> oldLoads = loads;
446    loads = new HashMap<String, Deque<RegionLoad>>();
447
448    for (ServerName sn : clusterStatus.getServers()) {
449      ServerLoad sl = clusterStatus.getLoad(sn);
450      if (sl == null) {
451        continue;
452      }
453      for (Entry<byte[], RegionLoad> entry : sl.getRegionsLoad().entrySet()) {
454        Deque<RegionLoad> rLoads = oldLoads.get(Bytes.toString(entry.getKey()));
455        if (rLoads == null) {
456          // There was nothing there
457          rLoads = new ArrayDeque<RegionLoad>();
458        } else if (rLoads.size() >= numRegionLoadsToRemember) {
459          rLoads.remove();
460        }
461        rLoads.add(entry.getValue());
462        loads.put(Bytes.toString(entry.getKey()), rLoads);
463
464      }
465    }
466
467    for(CostFromRegionLoadFunction cost : regionLoadFunctions) {
468      cost.setLoads(loads);
469    }
470  }

498  /**
499   * This is the main cost function.  It will compute a cost associated with a proposed cluster
500   * state.  All different costs will be combined with their multipliers to produce a double cost.
501   *
502   * @param cluster The state of the cluster
503   * @param previousCost the previous cost. This is used as an early out.
504   * @return a double of a cost associated with the proposed cluster state.  This cost is an
505   *         aggregate of all individual cost functions.
506   */
507  protected double computeCost(Cluster cluster, double previousCost) {
508    double total = 0;
509
510    for (int i = 0; i < costFunctions.length; i++) {
511      CostFunction c = costFunctions[i];
512      this.tempFunctionCosts[i] = 0.0;
513      
514      if (c.getMultiplier() <= 0) {
515        continue;
516      }
517
518      Float multiplier = c.getMultiplier();
519      Double cost = c.cost();
520
521      this.tempFunctionCosts[i] = multiplier*cost;
522      total += this.tempFunctionCosts[i];
523
524      if (total > previousCost) {
525        break;
526      }
527    }
528   
529    return total;
530  }

532  /** Generates a candidate action to be applied to the cluster for cost function search */
533  abstract static class CandidateGenerator {
534    abstract Cluster.Action generate(Cluster cluster);
535
536    /**
537     * From a list of regions pick a random one. Null can be returned which
538     * {@link StochasticLoadBalancer#balanceCluster(Map)} recognize as signal to try a region move
539     * rather than swap.
540     *
541     * @param cluster        The state of the cluster
542     * @param server         index of the server
543     * @param chanceOfNoSwap Chance that this will decide to try a move rather
544     *                       than a swap.
545     * @return a random {@link HRegionInfo} or null if an asymmetrical move is
546     *         suggested.
547     */
548    protected int pickRandomRegion(Cluster cluster, int server, double chanceOfNoSwap) {
549      // Check to see if this is just a move.
550      if (cluster.regionsPerServer[server].length == 0 || RANDOM.nextFloat() < chanceOfNoSwap) {
551        // signal a move only.
552        return -1;
553      }
554      int rand = RANDOM.nextInt(cluster.regionsPerServer[server].length);
555      return cluster.regionsPerServer[server][rand];
556
557    }
558    protected int pickRandomServer(Cluster cluster) {
559      if (cluster.numServers < 1) {
560        return -1;
561      }
562
563      return RANDOM.nextInt(cluster.numServers);
564    }
565
566    protected int pickRandomRack(Cluster cluster) {
567      if (cluster.numRacks < 1) {
568        return -1;
569      }
570
571      return RANDOM.nextInt(cluster.numRacks);
572    }
573
574    protected int pickOtherRandomServer(Cluster cluster, int serverIndex) {
575      if (cluster.numServers < 2) {
576        return -1;
577      }
578      while (true) {
579        int otherServerIndex = pickRandomServer(cluster);
580        if (otherServerIndex != serverIndex) {
581          return otherServerIndex;
582        }
583      }
584    }
585
586    protected int pickOtherRandomRack(Cluster cluster, int rackIndex) {
587      if (cluster.numRacks < 2) {
588        return -1;
589      }
590      while (true) {
591        int otherRackIndex = pickRandomRack(cluster);
592        if (otherRackIndex != rackIndex) {
593          return otherRackIndex;
594        }
595      }
596    }
597
598    protected Cluster.Action pickRandomRegions(Cluster cluster,
599                                                       int thisServer,
600                                                       int otherServer) {
601      if (thisServer < 0 || otherServer < 0) {
602        return Cluster.NullAction;
603      }
604
605      // Decide who is most likely to need another region
606      int thisRegionCount = cluster.getNumRegions(thisServer);
607      int otherRegionCount = cluster.getNumRegions(otherServer);
608
609      // Assign the chance based upon the above
610      double thisChance = (thisRegionCount > otherRegionCount) ? 0 : 0.5;
611      double otherChance = (thisRegionCount <= otherRegionCount) ? 0 : 0.5;
612
613      int thisRegion = pickRandomRegion(cluster, thisServer, thisChance);
614      int otherRegion = pickRandomRegion(cluster, otherServer, otherChance);
615
616      return getAction(thisServer, thisRegion, otherServer, otherRegion);
617    }
618
619    protected Cluster.Action getAction(int fromServer, int fromRegion,
620        int toServer, int toRegion) {
621      if (fromServer < 0 || toServer < 0) {
622        return Cluster.NullAction;
623      }
624      if (fromRegion > 0 && toRegion > 0) {
625        return new Cluster.SwapRegionsAction(fromServer, fromRegion,
626          toServer, toRegion);
627      } else if (fromRegion > 0) {
628        return new Cluster.MoveRegionAction(fromRegion, fromServer, toServer);
629      } else if (toRegion > 0) {
630        return new Cluster.MoveRegionAction(toRegion, toServer, fromServer);
631      } else {
632        return Cluster.NullAction;
633      }
634    }
635  }

637  static class RandomCandidateGenerator extends CandidateGenerator {
638
639    @Override
640    Cluster.Action generate(Cluster cluster) {
641
642      int thisServer = pickRandomServer(cluster);
643
644      // Pick the other server
645      int otherServer = pickOtherRandomServer(cluster, thisServer);
646
647      return pickRandomRegions(cluster, thisServer, otherServer);
648    }
649  }

651  static class LoadCandidateGenerator extends CandidateGenerator {
652
653    @Override
654    Cluster.Action generate(Cluster cluster) {
655      cluster.sortServersByRegionCount();
656      int thisServer = pickMostLoadedServer(cluster, -1);
657      int otherServer = pickLeastLoadedServer(cluster, thisServer);
658
659      return pickRandomRegions(cluster, thisServer, otherServer);
660    }
661
662    private int pickLeastLoadedServer(final Cluster cluster, int thisServer) {
663      Integer[] servers = cluster.serverIndicesSortedByRegionCount;
664
665      int index = 0;
666      while (servers[index] == null || servers[index] == thisServer) {
667        index++;
668        if (index == servers.length) {
669          return -1;
670        }
671      }
672      return servers[index];
673    }

676      Integer[] servers = cluster.serverIndicesSortedByRegionCount;
677
678      int index = servers.length - 1;
679      while (servers[index] == null || servers[index] == thisServer) {
680        index--;
681        if (index < 0) {
682          return -1;
683        }
684      }
685      return servers[index];
686    }

702        int otherServer = pickOtherRandomServer(cluster, thisServer);

726        return Cluster.NullAction;
727      }
728

730      int otherRegion = -1;
731

733    }

748  /**
749   * Generates candidates which moves the replicas out of the region server for
750   * co-hosted region replicas
751   */
752  static class RegionReplicaCandidateGenerator extends CandidateGenerator {
753
754    RandomCandidateGenerator randomGenerator = new RandomCandidateGenerator();
755
756    /**
757     * Randomly select one regionIndex out of all region replicas co-hosted in the same group
758     * (a group is a server, host or rack)
759     * @param primariesOfRegionsPerGroup either Cluster.primariesOfRegionsPerServer,
760     * primariesOfRegionsPerHost or primariesOfRegionsPerRack
761     * @param regionsPerGroup either Cluster.regionsPerServer, regionsPerHost or regionsPerRack
762     * @param regionIndexToPrimaryIndex Cluster.regionsIndexToPrimaryIndex
763     * @return a regionIndex for the selected primary or -1 if there is no co-locating
764     */
765    int selectCoHostedRegionPerGroup(int[] primariesOfRegionsPerGroup, int[] regionsPerGroup
766        , int[] regionIndexToPrimaryIndex) {
767      int currentPrimary = -1;
768      int currentPrimaryIndex = -1;
769      int selectedPrimaryIndex = -1;
770      double currentLargestRandom = -1;
771      // primariesOfRegionsPerGroup is a sorted array. Since it contains the primary region
772      // ids for the regions hosted in server, a consecutive repetition means that replicas
773      // are co-hosted
774      for (int j = 0; j <= primariesOfRegionsPerGroup.length; j++) {
775        int primary = j < primariesOfRegionsPerGroup.length
776            ? primariesOfRegionsPerGroup[j] : -1;
777        if (primary != currentPrimary) { // check for whether we see a new primary
778          int numReplicas = j - currentPrimaryIndex;
779          if (numReplicas > 1) { // means consecutive primaries, indicating co-location
780            // decide to select this primary region id or not
781            double currentRandom = RANDOM.nextDouble();
782            // we don't know how many region replicas are co-hosted, we will randomly select one
783            // using reservoir sampling (http://gregable.com/2007/10/reservoir-sampling.html)
784            if (currentRandom > currentLargestRandom) {
785              selectedPrimaryIndex = currentPrimary;
786              currentLargestRandom = currentRandom;
787            }
788          }
789          currentPrimary = primary;
790          currentPrimaryIndex = j;
791        }
792      }
793
794      // we have found the primary id for the region to move. Now find the actual regionIndex
795      // with the given primary, prefer to move the secondary region.
796      for (int j = 0; j < regionsPerGroup.length; j++) {
797        int regionIndex = regionsPerGroup[j];
798        if (selectedPrimaryIndex == regionIndexToPrimaryIndex[regionIndex]) {
799          // always move the secondary, not the primary
800          if (selectedPrimaryIndex != regionIndex) {
801            return regionIndex;
802          }
803        }
804      }
805      return -1;
806    }
807
808    @Override
809    Cluster.Action generate(Cluster cluster) {
810      int serverIndex = pickRandomServer(cluster);
811      if (cluster.numServers <= 1 || serverIndex == -1) {
812        return Cluster.NullAction;
813      }
814
815      int regionIndex = selectCoHostedRegionPerGroup(
816        cluster.primariesOfRegionsPerServer[serverIndex],
817        cluster.regionsPerServer[serverIndex],
818        cluster.regionIndexToPrimaryIndex);
819
820      // if there are no pairs of region replicas co-hosted, default to random generator
821      if (regionIndex == -1) {
822        // default to randompicker
823        return randomGenerator.generate(cluster);
824      }
825
826      int toServerIndex = pickOtherRandomServer(cluster, serverIndex);
827      int toRegionIndex = pickRandomRegion(cluster, toServerIndex, 0.9f);
828      return getAction(serverIndex, regionIndex, toServerIndex, toRegionIndex);
829    }
830  }

865  /**
866   * Base class of StochasticLoadBalancer's Cost Functions.
867   */
868  abstract static class CostFunction {
869
870    private float multiplier = 0;
871
872    protected Cluster cluster;
873
874    CostFunction(Configuration c) {
875
876    }
877
878    float getMultiplier() {
879      return multiplier;
880    }
881
882    void setMultiplier(float m) {
883      this.multiplier = m;
884    }
885
886    /** Called once per LB invocation to give the cost function
887     * to initialize it's state, and perform any costly calculation.
888     */
889    void init(Cluster cluster) {
890      this.cluster = cluster;
891    }
892
893    /** Called once per cluster Action to give the cost function
894     * an opportunity to update it's state. postAction() is always
895     * called at least once before cost() is called with the cluster
896     * that this action is performed on. */
897    void postAction(Action action) {
898      switch (action.type) {
899      case NULL: break;
900      case ASSIGN_REGION:
901        AssignRegionAction ar = (AssignRegionAction) action;
902        regionMoved(ar.region, -1, ar.server);
903        break;
904      case MOVE_REGION:
905        MoveRegionAction mra = (MoveRegionAction) action;
906        regionMoved(mra.region, mra.fromServer, mra.toServer);
907        break;
908      case SWAP_REGIONS:
909        SwapRegionsAction a = (SwapRegionsAction) action;
910        regionMoved(a.fromRegion, a.fromServer, a.toServer);
911        regionMoved(a.toRegion, a.toServer, a.fromServer);
912        break;
913      default:
914        throw new RuntimeException("Uknown action:" + action.type);
915      }
916    }
917
918    protected void regionMoved(int region, int oldServer, int newServer) {
919    }
920
921    abstract double cost();
922
923    /**
924     * Function to compute a scaled cost using {@link DescriptiveStatistics}. It
925     * assumes that this is a zero sum set of costs.  It assumes that the worst case
926     * possible is all of the elements in one region server and the rest having 0.
927     *
928     * @param stats the costs
929     * @return a scaled set of costs.
930     */
931    protected double costFromArray(double[] stats) {
932      double totalCost = 0;
933      double total = getSum(stats);
934
935      double count = stats.length;
936      double mean = total/count;
937
938      // Compute max as if all region servers had 0 and one had the sum of all costs.  This must be
939      // a zero sum cost for this to make sense.
940      double max = ((count - 1) * mean) + (total - mean);
941
942      // It's possible that there aren't enough regions to go around
943      double min;
944      if (count > total) {
945        min = ((count - total) * mean) + ((1 - mean) * total);
946      } else {
947        // Some will have 1 more than everything else.
948        int numHigh = (int) (total - (Math.floor(mean) * count));
949        int numLow = (int) (count - numHigh);
950
951        min = (numHigh * (Math.ceil(mean) - mean)) + (numLow * (mean - Math.floor(mean)));
952
953      }
954      min = Math.max(0, min);
955      for (int i=0; i<stats.length; i++) {
956        double n = stats[i];
957        double diff = Math.abs(mean - n);
958        totalCost += diff;
959      }
960
961      double scaled =  scale(min, max, totalCost);
962      return scaled;
963    }
964
965    private double getSum(double[] stats) {
966      double total = 0;
967      for(double s:stats) {
968        total += s;
969      }
970      return total;
971    }
972
973    /**
974     * Scale the value between 0 and 1.
975     *
976     * @param min   Min value
977     * @param max   The Max value
978     * @param value The value to be scaled.
979     * @return The scaled value.
980     */
981    protected double scale(double min, double max, double value) {
982      if (max <= min || value <= min) {
983        return 0;
984      }
985      if ((max - min) == 0) return 0;
986
987      return Math.max(0d, Math.min(1d, (value - min) / (max - min)));
988    }
989  }

992   * Given the starting state of the regions and a potential ending state
993   * compute cost based upon the number of regions that have moved.
994   */
995  static class MoveCostFunction extends CostFunction {
996    private static final String MOVE_COST_KEY = "hbase.master.balancer.stochastic.moveCost";
997    private static final String MAX_MOVES_PERCENT_KEY =
998        "hbase.master.balancer.stochastic.maxMovePercent";
999    private static final float DEFAULT_MOVE_COST = 100;
1000    private static final int DEFAULT_MAX_MOVES = 600;
1001    private static final float DEFAULT_MAX_MOVE_PERCENT = 0.25f;
1002
1003    private final float maxMovesPercent;
1004
1005    MoveCostFunction(Configuration conf) {
1006      super(conf);
1007
1008      // Move cost multiplier should be the same cost or higher than the rest of the costs to ensure
1009      // that large benefits are need to overcome the cost of a move.
1010      this.setMultiplier(conf.getFloat(MOVE_COST_KEY, DEFAULT_MOVE_COST));
1011      // What percent of the number of regions a single run of the balancer can move.
1012      maxMovesPercent = conf.getFloat(MAX_MOVES_PERCENT_KEY, DEFAULT_MAX_MOVE_PERCENT);
1013    }
1014
1015    @Override
1016    double cost() {
1017      // Try and size the max number of Moves, but always be prepared to move some.
1018      int maxMoves = Math.max((int) (cluster.numRegions * maxMovesPercent),
1019          DEFAULT_MAX_MOVES);
1020
1021      double moveCost = cluster.numMovedRegions;
1022
1023      // Don't let this single balance move more than the max moves.
1024      // This allows better scaling to accurately represent the actual cost of a move.
1025      if (moveCost > maxMoves) {
1026        return 1000000;   // return a number much greater than any of the other cost
1027      }
1028
1029      return scale(0, Math.min(cluster.numRegions, maxMoves), moveCost);
1030    }

1196   */
1197  abstract static class CostFromRegionLoadFunction extends CostFunction {
1198
1199    private ClusterStatus clusterStatus = null;
1200    private Map<String, Deque<RegionLoad>> loads = null;
1201    private double[] stats = null;
1202    CostFromRegionLoadFunction(Configuration conf) {
1203      super(conf);
1204    }

1206    void setClusterStatus(ClusterStatus status) {
1207      this.clusterStatus = status;
1208    }

1210    void setLoads(Map<String, Deque<RegionLoad>> l) {
1211      this.loads = l;
1212    }

1214    @Override
1215    double cost() {
1216      if (clusterStatus == null || loads == null) {
1217        return 0;
1218      }
1219
1220      if (stats == null || stats.length != cluster.numServers) {
1221        stats = new double[cluster.numServers];
1222      }
1223
1224      for (int i =0; i < stats.length; i++) {

1226        long cost = 0;
1227

1229        for(int regionIndex:cluster.regionsPerServer[i]) {
1230          Collection<RegionLoad> regionLoadList =  cluster.regionLoads[regionIndex];
1231

1234            cost += getRegionLoadCost(regionLoadList);
1235          }
1236        }
1237
1238        // Add the total cost to the stats.
1239        stats[i] = cost;

1251
1252        if (cost == 0) {
1253          cost = toAdd;
1254        } else {
1255          cost = (.5 * cost) + (.5 * toAdd);
1256        }
1257      }
1258
1259      return cost;
1260    }

1311   * replicas of the same region in the same host. We do not prevent the case

1317        "hbase.master.balancer.stochastic.regionReplicaHostCostKey";
1318    private static final float DEFAULT_REGION_REPLICA_HOST_COST_KEY = 100000;
1319
1320    long maxCost = 0;
1321    long[] costsPerGroup; // group is either server, host or rack
1322    int[][] primariesOfRegionsPerGroup;
1323
1324    public RegionReplicaHostCostFunction(Configuration conf) {
1325      super(conf);
1326      this.setMultiplier(conf.getFloat(REGION_REPLICA_HOST_COST_KEY,
1327        DEFAULT_REGION_REPLICA_HOST_COST_KEY));

1332      super.init(cluster);
1333      // max cost is the case where every region replica is hosted together regardless of host
1334      maxCost = cluster.numHosts > 1 ? getMaxCost(cluster) : 0;
1335      costsPerGroup = new long[cluster.numHosts];
1336      primariesOfRegionsPerGroup = cluster.multiServersPerHost // either server based or host based

1347      }
1348      // max cost is the case where every region replica is hosted together regardless of host
1349      int[] primariesOfRegions = new int[cluster.numRegions];
1350      System.arraycopy(cluster.regionIndexToPrimaryIndex, 0, primariesOfRegions, 0,
1351          cluster.regions.length);
1352
1353      Arrays.sort(primariesOfRegions);
1354
1355      // compute numReplicas from the sorted array
1356      return costPerGroup(primariesOfRegions);
1357    }
1358
1359    @Override
1360    double cost() {
1361      if (maxCost <= 0) {
1362        return 0;
1363      }
1364
1365      long totalCost = 0;
1366      for (int i = 0 ; i < costsPerGroup.length; i++) {
1367        totalCost += costsPerGroup[i];
1368      }
1369      return scale(0, maxCost, totalCost);
1370    }
1371
1372    /**
1373     * For each primary region, it computes the total number of replicas in the array (numReplicas)
1374     * and returns a sum of numReplicas-1 squared. For example, if the server hosts
1375     * regions a, b, c, d, e, f where a and b are same replicas, and c,d,e are same replicas, it
1376     * returns (2-1) * (2-1) + (3-1) * (3-1) + (1-1) * (1-1).
1377     * @param primariesOfRegions a sorted array of primary regions ids for the regions hosted
1378     * @return a sum of numReplicas-1 squared for each primary region in the group.
1379     */
1380    protected long costPerGroup(int[] primariesOfRegions) {
1381      long cost = 0;
1382      int currentPrimary = -1;
1383      int currentPrimaryIndex = -1;
1384      // primariesOfRegions is a sorted array of primary ids of regions. Replicas of regions
1385      // sharing the same primary will have consecutive numbers in the array.
1386      for (int j = 0 ; j <= primariesOfRegions.length; j++) {
1387        int primary = j < primariesOfRegions.length ? primariesOfRegions[j] : -1;
1388        if (primary != currentPrimary) { // we see a new primary
1389          int numReplicas = j - currentPrimaryIndex;
1390          // square the cost
1391          if (numReplicas > 1) { // means consecutive primaries, indicating co-location
1392            cost += (numReplicas - 1) * (numReplicas - 1);
1393          }
1394          currentPrimary = primary;
1395          currentPrimaryIndex = j;
1396        }
1397      }
1398
1399      return cost;
1400    }
1401
1402    @Override
1403    protected void regionMoved(int region, int oldServer, int newServer) {
1404      if (maxCost <= 0) {
1405        return; // no need to compute
1406      }
1407      if (cluster.multiServersPerHost) {
1408        int oldHost = cluster.serverIndexToHostIndex[oldServer];
1409        int newHost = cluster.serverIndexToHostIndex[newServer];
1410        if (newHost != oldHost) {
1411          costsPerGroup[oldHost] = costPerGroup(cluster.primariesOfRegionsPerHost[oldHost]);
1412          costsPerGroup[newHost] = costPerGroup(cluster.primariesOfRegionsPerHost[newHost]);
1413        }
1414      } else {
1415        costsPerGroup[oldServer] = costPerGroup(cluster.primariesOfRegionsPerServer[oldServer]);
1416        costsPerGroup[newServer] = costPerGroup(cluster.primariesOfRegionsPerServer[newServer]);
1417      }
1418    }
1419  }

1421  /**
1422   * A cost function for region replicas for the rack distribution. We give a relatively high
1423   * cost to hosting replicas of the same region in the same rack. We do not prevent the case
1424   * though.
1425   */
1426  static class RegionReplicaRackCostFunction extends RegionReplicaHostCostFunction {
1427    private static final String REGION_REPLICA_RACK_COST_KEY =
1428        "hbase.master.balancer.stochastic.regionReplicaRackCostKey";
1429    private static final float DEFAULT_REGION_REPLICA_RACK_COST_KEY = 10000;
1430
1431    public RegionReplicaRackCostFunction(Configuration conf) {
1432      super(conf);
1433      this.setMultiplier(conf.getFloat(REGION_REPLICA_RACK_COST_KEY, 
1434        DEFAULT_REGION_REPLICA_RACK_COST_KEY));
1435    }
1436
1437    @Override
1438    void init(Cluster cluster) {
1439      this.cluster = cluster;
1440      if (cluster.numRacks <= 1) {
1441        maxCost = 0;
1442        return; // disabled for 1 rack
1443      }
1444      // max cost is the case where every region replica is hosted together regardless of rack
1445      maxCost = getMaxCost(cluster);
1446      costsPerGroup = new long[cluster.numRacks];
1447      for (int i = 0 ; i < cluster.primariesOfRegionsPerRack.length; i++) {
1448        costsPerGroup[i] = costPerGroup(cluster.primariesOfRegionsPerRack[i]);
1449      }
1450    }
1451
1452    @Override
1453    protected void regionMoved(int region, int oldServer, int newServer) {
1454      if (maxCost <= 0) {
1455        return; // no need to compute
1456      }
1457      int oldRack = cluster.serverIndexToRackIndex[oldServer];
1458      int newRack = cluster.serverIndexToRackIndex[newServer];
1459      if (newRack != oldRack) {
1460        costsPerGroup[oldRack] = costPerGroup(cluster.primariesOfRegionsPerRack[oldRack]);
1461        costsPerGroup[newRack] = costPerGroup(cluster.primariesOfRegionsPerRack[newRack]);
1462      }
1463    }
1464  }
