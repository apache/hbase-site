059/**
060 * <p>This is a best effort load balancer. Given a Cost function F(C) =&gt; x It will
061 * randomly try and mutate the cluster to Cprime. If F(Cprime) &lt; F(C) then the
062 * new cluster state becomes the plan. It includes costs functions to compute the cost of:</p>
063 * <ul>
064 * <li>Region Load</li>
065 * <li>Table Load</li>
066 * <li>Data Locality</li>
067 * <li>Memstore Sizes</li>
068 * <li>Storefile Sizes</li>
069 * </ul>
070 *
071 *
072 * <p>Every cost function returns a number between 0 and 1 inclusive; where 0 is the lowest cost
073 * best solution, and 1 is the highest possible cost and the worst solution.  The computed costs are
074 * scaled by their respective multipliers:</p>
075 *
076 * <ul>
077 *   <li>hbase.master.balancer.stochastic.regionLoadCost</li>
078 *   <li>hbase.master.balancer.stochastic.moveCost</li>
079 *   <li>hbase.master.balancer.stochastic.tableLoadCost</li>
080 *   <li>hbase.master.balancer.stochastic.localityCost</li>
081 *   <li>hbase.master.balancer.stochastic.memstoreSizeCost</li>
082 *   <li>hbase.master.balancer.stochastic.storefileSizeCost</li>
083 * </ul>
084 *
085 * <p>You can also add custom Cost function by setting the the following configuration value:</p>
086 * <ul>
087 *     <li>hbase.master.balancer.stochastic.additionalCostFunctions</li>
088 * </ul>
089 *
090 * <p>All custom Cost Functions needs to extends {@link StochasticLoadBalancer.CostFunction}</p>
091 *
092 * <p>In addition to the above configurations, the balancer can be tuned by the following
093 * configuration values:</p>
094 * <ul>
095 *   <li>hbase.master.balancer.stochastic.maxMoveRegions which
096 *   controls what the max number of regions that can be moved in a single invocation of this
097 *   balancer.</li>
098 *   <li>hbase.master.balancer.stochastic.stepsPerRegion is the coefficient by which the number of
099 *   regions is multiplied to try and get the number of times the balancer will
100 *   mutate all servers.</li>
101 *   <li>hbase.master.balancer.stochastic.maxSteps which controls the maximum number of times that
102 *   the balancer will try and mutate all the servers. The balancer will use the minimum of this
103 *   value and the above computation.</li>
104 * </ul>
105 *
106 * <p>This balancer is best used with hbase.master.loadbalance.bytable set to false
107 * so that the balancer gets the full picture of all loads on the cluster.</p>
108 */

242  private void loadCustomCostFunctions(Configuration conf) {
243    String[] functionsNames = conf.getStrings(COST_FUNCTIONS_COST_FUNCTIONS_KEY);
244
245    if (null == functionsNames) {
246      return;
247    }
248
249    costFunctions.addAll(Arrays.stream(functionsNames).map(c -> {
250      Class<? extends CostFunction> klass = null;
251      try {
252        klass = Class.forName(c).asSubclass(CostFunction.class);
253      } catch (ClassNotFoundException e) {
254        LOG.warn("Cannot load class " + c + "': " + e.getMessage());
255      }
256      if (null == klass) {
257        return null;
258      }
259      CostFunction reflected = ReflectionUtils.newInstance(klass, conf);
260      LOG.info(
261        "Successfully loaded custom CostFunction '" + reflected.getClass().getSimpleName() + "'");
262      return reflected;
263    }).filter(Objects::nonNull).collect(Collectors.toList()));
264  }

324  @Override
325  protected boolean needsBalance(TableName tableName, Cluster cluster) {
326    ClusterLoadState cs = new ClusterLoadState(cluster.clusterState);
327    if (cs.getNumServers() < MIN_SERVER_BALANCE) {
328      if (LOG.isDebugEnabled()) {
329        LOG.debug("Not running balancer because only " + cs.getNumServers()
330            + " active regionserver(s)");
331      }
332      return false;
333    }
334    if (areSomeRegionReplicasColocated(cluster)) {
335      return true;
336    }
337
338    if (idleRegionServerExist(cluster)){
339      return true;
340    }
341
342    double total = 0.0;
343    float sumMultiplier = 0.0f;
344    for (CostFunction c : costFunctions) {
345      float multiplier = c.getMultiplier();
346      if (multiplier <= 0) {
347        LOG.trace("{} not needed because multiplier is <= 0", c.getClass().getSimpleName());
348        continue;
349      }
350      if (!c.isNeeded()) {
351        LOG.trace("{} not needed", c.getClass().getSimpleName());
352        continue;
353      }
354      sumMultiplier += multiplier;
355      total += c.cost() * multiplier;
356    }
357
358    boolean balanced = total <= 0 || sumMultiplier <= 0 ||
359        (sumMultiplier > 0 && (total / sumMultiplier) < minCostNeedBalance);
360    if (LOG.isDebugEnabled()) {
361      LOG.debug("{} {}; total cost={}, sum multiplier={}; cost/multiplier to need a balance is {}",
362          balanced ? "Skipping load balancing because balanced" : "We need to load balance",
363          isByTable ? String.format("table (%s)", tableName) : "cluster",
364          total, sumMultiplier, minCostNeedBalance);
365      if (LOG.isTraceEnabled()) {
366        LOG.trace("Balance decision detailed function costs={}", functionCost());
367      }
368    }
369    return !balanced;
370  }

377  /**
378   * Given the cluster state this will try and approach an optimal balance. This
379   * should always approach the optimal state given enough steps.
380   */
381  @Override
382  public synchronized List<RegionPlan> balanceTable(TableName tableName, Map<ServerName,
383    List<RegionInfo>> loadOfOneTable) {
384    List<RegionPlan> plans = balanceMasterRegions(loadOfOneTable);
385    if (plans != null || loadOfOneTable == null || loadOfOneTable.size() <= 1) {
386      return plans;
387    }
388
389    if (masterServerName != null && loadOfOneTable.containsKey(masterServerName)) {
390      if (loadOfOneTable.size() <= 2) {
391        return null;
392      }
393      loadOfOneTable = new HashMap<>(loadOfOneTable);
394      loadOfOneTable.remove(masterServerName);
395    }
396
397    // On clusters with lots of HFileLinks or lots of reference files,
398    // instantiating the storefile infos can be quite expensive.
399    // Allow turning this feature off if the locality cost is not going to
400    // be used in any computations.
401    RegionHDFSBlockLocationFinder finder = null;
402    if ((this.localityCost != null && this.localityCost.getMultiplier() > 0)
403        || (this.rackLocalityCost != null && this.rackLocalityCost.getMultiplier() > 0)) {
404      finder = this.regionFinder;
405    }
406
407    //The clusterState that is given to this method contains the state
408    //of all the regions in the table(s) (that's true today)
409    // Keep track of servers to iterate through them.
410    Cluster cluster = new Cluster(loadOfOneTable, loads, finder, rackManager);
411
412    long startTime = EnvironmentEdgeManager.currentTime();
413
414    initCosts(cluster);
415
416    if (!needsBalance(tableName, cluster)) {
417      return null;
418    }
419
420    double currentCost = computeCost(cluster, Double.MAX_VALUE);
421    curOverallCost = currentCost;
422    System.arraycopy(tempFunctionCosts, 0, curFunctionCosts, 0, curFunctionCosts.length);
423    double initCost = currentCost;
424    double newCost;
425
426    long computedMaxSteps;
427    if (runMaxSteps) {
428      computedMaxSteps = Math.max(this.maxSteps,
429          ((long)cluster.numRegions * (long)this.stepsPerRegion * (long)cluster.numServers));
430    } else {
431      long calculatedMaxSteps = (long)cluster.numRegions * (long)this.stepsPerRegion *
432          (long)cluster.numServers;
433      computedMaxSteps = Math.min(this.maxSteps, calculatedMaxSteps);
434      if (calculatedMaxSteps > maxSteps) {
435        LOG.warn("calculatedMaxSteps:{} for loadbalancer's stochastic walk is larger than "
436            + "maxSteps:{}. Hence load balancing may not work well. Setting parameter "
437            + "\"hbase.master.balancer.stochastic.runMaxSteps\" to true can overcome this issue."
438            + "(This config change does not require service restart)", calculatedMaxSteps,
439            maxSteps);
440      }
441    }
442    LOG.info("start StochasticLoadBalancer.balancer, initCost=" + currentCost + ", functionCost="
443        + functionCost() + " computedMaxSteps: " + computedMaxSteps);
444
445    final String initFunctionTotalCosts = totalCostsPerFunc();
446    // Perform a stochastic walk to see if we can get a good fit.
447    long step;
448
449    for (step = 0; step < computedMaxSteps; step++) {
450      Cluster.Action action = nextAction(cluster);
451
452      if (action.type == Type.NULL) {
453        continue;
454      }
455
456      cluster.doAction(action);
457      updateCostsWithAction(cluster, action);
458
459      newCost = computeCost(cluster, currentCost);
460
461      // Should this be kept?
462      if (newCost < currentCost) {
463        currentCost = newCost;
464
465        // save for JMX
466        curOverallCost = currentCost;
467        System.arraycopy(tempFunctionCosts, 0, curFunctionCosts, 0, curFunctionCosts.length);
468      } else {
469        // Put things back the way they were before.
470        // TODO: undo by remembering old values
471        Action undoAction = action.undoAction();
472        cluster.doAction(undoAction);
473        updateCostsWithAction(cluster, undoAction);
474      }
475
476      if (EnvironmentEdgeManager.currentTime() - startTime >
477          maxRunningTime) {
478        break;
479      }
480    }
481    long endTime = EnvironmentEdgeManager.currentTime();
482
483    metricsBalancer.balanceCluster(endTime - startTime);
484
485    // update costs metrics
486    updateStochasticCosts(tableName, curOverallCost, curFunctionCosts);
487    if (initCost > currentCost) {
488      plans = createRegionPlans(cluster);
489      LOG.info("Finished computing new load balance plan. Computation took {}" +
490        " to try {} different iterations.  Found a solution that moves " +
491        "{} regions; Going from a computed cost of {}" +
492        " to a new cost of {}", java.time.Duration.ofMillis(endTime - startTime),
493        step, plans.size(), initCost, currentCost);
494      sendRegionPlansToRingBuffer(plans, currentCost, initCost, initFunctionTotalCosts, step);
495      return plans;
496    }
497    LOG.info("Could not find a better load balance plan.  Tried {} different configurations in " +
498      "{}, and did not find anything with a computed cost less than {}", step,
499      java.time.Duration.ofMillis(endTime - startTime), initCost);
500    return null;
501  }

503  private void sendRegionPlansToRingBuffer(List<RegionPlan> plans, double currentCost,
504      double initCost, String initFunctionTotalCosts, long step) {
505    if (this.namedQueueRecorder != null) {
506      List<String> regionPlans = new ArrayList<>();
507      for (RegionPlan plan : plans) {
508        regionPlans.add(
509          "table: " + plan.getRegionInfo().getTable() + " , region: " + plan.getRegionName()
510            + " , source: " + plan.getSource() + " , destination: " + plan.getDestination());
511      }
512      BalancerDecision balancerDecision =
513        new BalancerDecision.Builder()
514          .setInitTotalCost(initCost)
515          .setInitialFunctionCosts(initFunctionTotalCosts)
516          .setComputedTotalCost(currentCost)
517          .setFinalFunctionCosts(totalCostsPerFunc())
518          .setComputedSteps(step)
519          .setRegionPlans(regionPlans).build();
520      namedQueueRecorder.addRecord(new BalancerDecisionDetails(balancerDecision));
521    }
522  }

524  /**
525   * update costs to JMX
526   */
527  private void updateStochasticCosts(TableName tableName, Double overall, Double[] subCosts) {
528    if (tableName == null) return;
529
530    // check if the metricsBalancer is MetricsStochasticBalancer before casting
531    if (metricsBalancer instanceof MetricsStochasticBalancer) {
532      MetricsStochasticBalancer balancer = (MetricsStochasticBalancer) metricsBalancer;
533      // overall cost
534      balancer.updateStochasticCost(tableName.getNameAsString(),
535        "Overall", "Overall cost", overall);
536
537      // each cost function
538      for (int i = 0; i < costFunctions.size(); i++) {
539        CostFunction costFunction = costFunctions.get(i);
540        String costFunctionName = costFunction.getClass().getSimpleName();
541        Double costPercent = (overall == 0) ? 0 : (subCosts[i] / overall);
542        // TODO: cost function may need a specific description
543        balancer.updateStochasticCost(tableName.getNameAsString(), costFunctionName,
544          "The percent of " + costFunctionName, costPercent);
545      }
546    }
547  }

555  private String functionCost() {
556    StringBuilder builder = new StringBuilder();
557    for (CostFunction c:costFunctions) {
558      builder.append(c.getClass().getSimpleName());
559      builder.append(" : (");
560      builder.append(c.getMultiplier());
561      builder.append(", ");
562      builder.append(c.cost());
563      builder.append("); ");
564    }
565    return builder.toString();
566  }
567
568  private String totalCostsPerFunc() {
569    StringBuilder builder = new StringBuilder();
570    for (CostFunction c : costFunctions) {
571      if (c.getMultiplier() * c.cost() > 0.0) {
572        builder.append(" ");
573        builder.append(c.getClass().getSimpleName());
574        builder.append(" : ");
575        builder.append(c.getMultiplier() * c.cost());
576        builder.append(";");
577      }
578    }
579    if (builder.length() > 0) {
580      builder.deleteCharAt(builder.length() - 1);
581    }
582    return builder.toString();
583  }

585  /**
586   * Create all of the RegionPlan's needed to move from the initial cluster state to the desired
587   * state.
588   *
589   * @param cluster The state of the cluster
590   * @return List of RegionPlan's that represent the moves needed to get to desired final state.
591   */
592  private List<RegionPlan> createRegionPlans(Cluster cluster) {
593    List<RegionPlan> plans = new LinkedList<>();
594    for (int regionIndex = 0;
595         regionIndex < cluster.regionIndexToServerIndex.length; regionIndex++) {
596      int initialServerIndex = cluster.initialRegionIndexToServerIndex[regionIndex];
597      int newServerIndex = cluster.regionIndexToServerIndex[regionIndex];
598
599      if (initialServerIndex != newServerIndex) {
600        RegionInfo region = cluster.regions[regionIndex];
601        ServerName initialServer = cluster.servers[initialServerIndex];
602        ServerName newServer = cluster.servers[newServerIndex];
603
604        if (LOG.isTraceEnabled()) {
605          LOG.trace("Moving Region " + region.getEncodedName() + " from server "
606              + initialServer.getHostname() + " to " + newServer.getHostname());
607        }
608        RegionPlan rp = new RegionPlan(region, initialServer, newServer);
609        plans.add(rp);
610      }
611    }

616   * Store the current region loads.
617   */
618  private synchronized void updateRegionLoad() {
619    // We create a new hashmap so that regions that are no longer there are removed.
620    // However we temporarily need the old loads so we can use them to keep the rolling average.
621    Map<String, Deque<BalancerRegionLoad>> oldLoads = loads;
622    loads = new HashMap<>();
623
624    clusterStatus.getLiveServerMetrics().forEach((ServerName sn, ServerMetrics sm) -> {
625      sm.getRegionMetrics().forEach((byte[] regionName, RegionMetrics rm) -> {
626        String regionNameAsString = RegionInfo.getRegionNameAsString(regionName);
627        Deque<BalancerRegionLoad> rLoads = oldLoads.get(regionNameAsString);
628        if (rLoads == null) {
629          rLoads = new ArrayDeque<>(numRegionLoadsToRemember + 1);
630        } else if (rLoads.size() >= numRegionLoadsToRemember) {
631          rLoads.remove();
632        }
633        rLoads.add(new BalancerRegionLoad(rm));
634        loads.put(regionNameAsString, rLoads);
635      });
636    });
637
638    for(CostFromRegionLoadFunction cost : regionLoadFunctions) {
639      cost.setLoads(loads);
640    }
641  }

669  /**
670   * This is the main cost function.  It will compute a cost associated with a proposed cluster
671   * state.  All different costs will be combined with their multipliers to produce a double cost.
672   *
673   * @param cluster The state of the cluster
674   * @param previousCost the previous cost. This is used as an early out.
675   * @return a double of a cost associated with the proposed cluster state.  This cost is an
676   *         aggregate of all individual cost functions.
677   */
678  protected double computeCost(Cluster cluster, double previousCost) {
679    double total = 0;
680
681    for (int i = 0; i < costFunctions.size(); i++) {
682      CostFunction c = costFunctions.get(i);
683      this.tempFunctionCosts[i] = 0.0;
684
685      if (c.getMultiplier() <= 0) {
686        continue;
687      }
688
689      Float multiplier = c.getMultiplier();
690      Double cost = c.cost();
691
692      this.tempFunctionCosts[i] = multiplier*cost;
693      total += this.tempFunctionCosts[i];
694
695      if (total > previousCost) {
696        break;
697      }
698    }
699
700    return total;
701  }

750  /**
751   * Base class of StochasticLoadBalancer's Cost Functions.
752   */
753  public abstract static class CostFunction {
754
755    private float multiplier = 0;
756
757    protected Cluster cluster;
758
759    public CostFunction(Configuration c) {
760    }
761
762    boolean isNeeded() {
763      return true;
764    }
765    float getMultiplier() {
766      return multiplier;
767    }
768
769    void setMultiplier(float m) {
770      this.multiplier = m;
771    }
772
773    /** Called once per LB invocation to give the cost function
774     * to initialize it's state, and perform any costly calculation.
775     */
776    void init(Cluster cluster) {
777      this.cluster = cluster;
778    }
779
780    /** Called once per cluster Action to give the cost function
781     * an opportunity to update it's state. postAction() is always
782     * called at least once before cost() is called with the cluster
783     * that this action is performed on. */
784    void postAction(Action action) {
785      switch (action.type) {
786      case NULL: break;
787      case ASSIGN_REGION:
788        AssignRegionAction ar = (AssignRegionAction) action;
789        regionMoved(ar.region, -1, ar.server);
790        break;
791      case MOVE_REGION:
792        MoveRegionAction mra = (MoveRegionAction) action;
793        regionMoved(mra.region, mra.fromServer, mra.toServer);
794        break;
795      case SWAP_REGIONS:
796        SwapRegionsAction a = (SwapRegionsAction) action;
797        regionMoved(a.fromRegion, a.fromServer, a.toServer);
798        regionMoved(a.toRegion, a.toServer, a.fromServer);
799        break;
800      default:
801        throw new RuntimeException("Uknown action:" + action.type);
802      }
803    }
804
805    protected void regionMoved(int region, int oldServer, int newServer) {
806    }
807
808    protected abstract double cost();
809
810    @SuppressWarnings("checkstyle:linelength")
811    /**
812     * Function to compute a scaled cost using
813     * {@link org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#DescriptiveStatistics()}.
814     * It assumes that this is a zero sum set of costs.  It assumes that the worst case
815     * possible is all of the elements in one region server and the rest having 0.
816     *
817     * @param stats the costs
818     * @return a scaled set of costs.
819     */
820    protected double costFromArray(double[] stats) {
821      double totalCost = 0;
822      double total = getSum(stats);
823
824      double count = stats.length;
825      double mean = total/count;
826
827      // Compute max as if all region servers had 0 and one had the sum of all costs.  This must be
828      // a zero sum cost for this to make sense.
829      double max = ((count - 1) * mean) + (total - mean);
830
831      // It's possible that there aren't enough regions to go around
832      double min;
833      if (count > total) {
834        min = ((count - total) * mean) + ((1 - mean) * total);
835      } else {
836        // Some will have 1 more than everything else.
837        int numHigh = (int) (total - (Math.floor(mean) * count));
838        int numLow = (int) (count - numHigh);
839
840        min = (numHigh * (Math.ceil(mean) - mean)) + (numLow * (mean - Math.floor(mean)));
841
842      }
843      min = Math.max(0, min);
844      for (int i=0; i<stats.length; i++) {
845        double n = stats[i];
846        double diff = Math.abs(mean - n);
847        totalCost += diff;
848      }
849
850      double scaled =  scale(min, max, totalCost);
851      return scaled;
852    }
853
854    private double getSum(double[] stats) {
855      double total = 0;
856      for(double s:stats) {
857        total += s;
858      }
859      return total;
860    }
861
862    /**
863     * Scale the value between 0 and 1.
864     *
865     * @param min   Min value
866     * @param max   The Max value
867     * @param value The value to be scaled.
868     * @return The scaled value.
869     */
870    protected double scale(double min, double max, double value) {
871      if (max <= min || value <= min) {
872        return 0;
873      }
874      if ((max - min) == 0) return 0;
875
876      return Math.max(0d, Math.min(1d, (value - min) / (max - min)));
877    }
878  }

880  /**
881   * Given the starting state of the regions and a potential ending state
882   * compute cost based upon the number of regions that have moved.
883   */
884  static class MoveCostFunction extends CostFunction {
885    private static final String MOVE_COST_KEY = "hbase.master.balancer.stochastic.moveCost";
886    private static final String MOVE_COST_OFFPEAK_KEY =
887      "hbase.master.balancer.stochastic.moveCost.offpeak";
888    private static final String MAX_MOVES_PERCENT_KEY =
889        "hbase.master.balancer.stochastic.maxMovePercent";
890    static final float DEFAULT_MOVE_COST = 7;
891    static final float DEFAULT_MOVE_COST_OFFPEAK = 3;
892    private static final int DEFAULT_MAX_MOVES = 600;
893    private static final float DEFAULT_MAX_MOVE_PERCENT = 0.25f;
894
895    private final float maxMovesPercent;
896    private final Configuration conf;
897
898    MoveCostFunction(Configuration conf) {
899      super(conf);
900      this.conf = conf;
901      // What percent of the number of regions a single run of the balancer can move.
902      maxMovesPercent = conf.getFloat(MAX_MOVES_PERCENT_KEY, DEFAULT_MAX_MOVE_PERCENT);
903
904      // Initialize the multiplier so that addCostFunction will add this cost function.
905      // It may change during later evaluations, due to OffPeakHours.
906      this.setMultiplier(conf.getFloat(MOVE_COST_KEY, DEFAULT_MOVE_COST));
907    }
908
909    @Override
910    protected double cost() {
911      // Move cost multiplier should be the same cost or higher than the rest of the costs to ensure
912      // that large benefits are need to overcome the cost of a move.
913      if (OffPeakHours.getInstance(conf).isOffPeakHour()) {
914        this.setMultiplier(conf.getFloat(MOVE_COST_OFFPEAK_KEY, DEFAULT_MOVE_COST_OFFPEAK));
915      } else {
916        this.setMultiplier(conf.getFloat(MOVE_COST_KEY, DEFAULT_MOVE_COST));
917      }
918      // Try and size the max number of Moves, but always be prepared to move some.
919      int maxMoves = Math.max((int) (cluster.numRegions * maxMovesPercent),
920          DEFAULT_MAX_MOVES);
921
922      double moveCost = cluster.numMovedRegions;
923
924      // Don't let this single balance move more than the max moves.
925      // This allows better scaling to accurately represent the actual cost of a move.
926      if (moveCost > maxMoves) {
927        return 1000000;   // return a number much greater than any of the other cost
928      }
929
930      return scale(0, Math.min(cluster.numRegions, maxMoves), moveCost);
931    }
932  }

1050  /**
1051   * Compute a cost of a potential cluster configuration based upon where
1052   * {@link org.apache.hadoop.hbase.regionserver.HStoreFile}s are located.
1053   */
1054  static abstract class LocalityBasedCostFunction extends CostFunction {
1055
1056    private final LocalityType type;
1057
1058    private double bestLocality; // best case locality across cluster weighted by local data size
1059    private double locality; // current locality across cluster weighted by local data size
1060
1061    LocalityBasedCostFunction(Configuration conf, LocalityType type, String localityCostKey,
1062      float defaultLocalityCost) {
1063      super(conf);
1064      this.type = type;
1065      this.setMultiplier(conf.getFloat(localityCostKey, defaultLocalityCost));
1066      this.locality = 0.0;
1067      this.bestLocality = 0.0;
1068    }
1069
1070    /**
1071     * Maps region to the current entity (server or rack) on which it is stored
1072     */
1073    abstract int regionIndexToEntityIndex(int region);
1074
1075    @Override
1076    void init(Cluster cluster) {
1077      super.init(cluster);
1078      locality = 0.0;
1079      bestLocality = 0.0;
1080
1081      for (int region = 0; region < cluster.numRegions; region++) {
1082        locality += getWeightedLocality(region, regionIndexToEntityIndex(region));
1083        bestLocality += getWeightedLocality(region, getMostLocalEntityForRegion(region));
1084      }
1085
1086      // We normalize locality to be a score between 0 and 1.0 representing how good it
1087      // is compared to how good it could be. If bestLocality is 0, assume locality is 100
1088      // (and the cost is 0)
1089      locality = bestLocality == 0 ? 1.0 : locality / bestLocality;
1090    }
1091
1092    @Override
1093    protected void regionMoved(int region, int oldServer, int newServer) {
1094      int oldEntity = type == LocalityType.SERVER ? oldServer : cluster.serverIndexToRackIndex[oldServer];
1095      int newEntity = type == LocalityType.SERVER ? newServer : cluster.serverIndexToRackIndex[newServer];
1096      double localityDelta = getWeightedLocality(region, newEntity) - getWeightedLocality(region, oldEntity);
1097      double normalizedDelta = bestLocality == 0 ? 0.0 : localityDelta / bestLocality;
1098      locality += normalizedDelta;
1099    }
1100
1101    @Override
1102    protected double cost() {
1103      return 1 - locality;
1104    }
1105
1106    private int getMostLocalEntityForRegion(int region) {
1107      return cluster.getOrComputeRegionsToMostLocalEntities(type)[region];
1108    }
1109
1110    private double getWeightedLocality(int region, int entity) {
1111      return cluster.getOrComputeWeightedLocality(region, entity, type);
1112    }
1113

1147   * Base class the allows writing costs functions from rolling average of some
1148   * number from RegionLoad.
1149   */
1150  abstract static class CostFromRegionLoadFunction extends CostFunction {
1151
1152    private ClusterMetrics clusterStatus = null;
1153    private Map<String, Deque<BalancerRegionLoad>> loads = null;
1154    private double[] stats = null;
1155    CostFromRegionLoadFunction(Configuration conf) {
1156      super(conf);
1157    }
1158
1159    void setClusterMetrics(ClusterMetrics status) {
1160      this.clusterStatus = status;
1161    }
1162
1163    void setLoads(Map<String, Deque<BalancerRegionLoad>> l) {
1164      this.loads = l;
1165    }
1166
1167    @Override
1168    protected double cost() {
1169      if (clusterStatus == null || loads == null) {
1170        return 0;
1171      }
1172
1173      if (stats == null || stats.length != cluster.numServers) {
1174        stats = new double[cluster.numServers];
1175      }
1176
1177      for (int i =0; i < stats.length; i++) {
1178        //Cost this server has from RegionLoad
1179        long cost = 0;
1180
1181        // for every region on this server get the rl
1182        for(int regionIndex:cluster.regionsPerServer[i]) {
1183          Collection<BalancerRegionLoad> regionLoadList =  cluster.regionLoads[regionIndex];
1184
1185          // Now if we found a region load get the type of cost that was requested.
1186          if (regionLoadList != null) {
1187            cost = (long) (cost + getRegionLoadCost(regionLoadList));
1188          }
1189        }
1190
1191        // Add the total cost to the stats.
1192        stats[i] = cost;
1193      }
1194
1195      // Now return the scaled cost from data held in the stats object.
1196      return costFromArray(stats);
1197    }
1198
1199    protected double getRegionLoadCost(Collection<BalancerRegionLoad> regionLoadList) {
1200      double cost = 0;
1201      for (BalancerRegionLoad rl : regionLoadList) {
1202        cost += getCostFromRl(rl);
1203      }
1204      return cost / regionLoadList.size();
1205    }

1208  }

1211   * Class to be used for the subset of RegionLoad costs that should be treated as rates.
1212   * We do not compare about the actual rate in requests per second but rather the rate relative
1213   * to the rest of the regions.
1214   */

1223      double cost = 0;
1224      double previous = 0;

1226      for (BalancerRegionLoad rl : regionLoadList) {
1227        double current = getCostFromRl(rl);
1228        if (isFirst) {
1229          isFirst = false;
1230        } else {
1231          cost += current - previous;
1232        }
1233        previous = current;
1234      }
1235      return Math.max(0, cost / (regionLoadList.size() - 1));
1236    }

1304  /**
1305   * A cost function for region replicas. We give a very high cost to hosting
1306   * replicas of the same region in the same host. We do not prevent the case
1307   * though, since if numReplicas > numRegionServers, we still want to keep the
1308   * replica open.
1309   */
1310  static class RegionReplicaHostCostFunction extends CostFunction {
1311    private static final String REGION_REPLICA_HOST_COST_KEY =
1312        "hbase.master.balancer.stochastic.regionReplicaHostCostKey";
1313    private static final float DEFAULT_REGION_REPLICA_HOST_COST_KEY = 100000;
1314
1315    long maxCost = 0;
1316    long[] costsPerGroup; // group is either server, host or rack
1317    int[][] primariesOfRegionsPerGroup;
1318
1319    public RegionReplicaHostCostFunction(Configuration conf) {
1320      super(conf);
1321      this.setMultiplier(conf.getFloat(REGION_REPLICA_HOST_COST_KEY,
1322        DEFAULT_REGION_REPLICA_HOST_COST_KEY));
1323    }
1324
1325    @Override
1326    void init(Cluster cluster) {
1327      super.init(cluster);
1328      // max cost is the case where every region replica is hosted together regardless of host
1329      maxCost = cluster.numHosts > 1 ? getMaxCost(cluster) : 0;
1330      costsPerGroup = new long[cluster.numHosts];
1331      primariesOfRegionsPerGroup = cluster.multiServersPerHost // either server based or host based
1332          ? cluster.primariesOfRegionsPerHost
1333          : cluster.primariesOfRegionsPerServer;
1334      for (int i = 0 ; i < primariesOfRegionsPerGroup.length; i++) {
1335        costsPerGroup[i] = costPerGroup(primariesOfRegionsPerGroup[i]);
1336      }
1337    }
1338
1339    long getMaxCost(Cluster cluster) {
1340      if (!cluster.hasRegionReplicas) {
1341        return 0; // short circuit
1342      }
1343      // max cost is the case where every region replica is hosted together regardless of host
1344      int[] primariesOfRegions = new int[cluster.numRegions];
1345      System.arraycopy(cluster.regionIndexToPrimaryIndex, 0, primariesOfRegions, 0,
1346          cluster.regions.length);
1347
1348      Arrays.sort(primariesOfRegions);
1349
1350      // compute numReplicas from the sorted array
1351      return costPerGroup(primariesOfRegions);
1352    }
1353
1354    @Override
1355    boolean isNeeded() {
1356      return cluster.hasRegionReplicas;
1357    }
1358
1359    @Override
1360    protected double cost() {
1361      if (maxCost <= 0) {
1362        return 0;
1363      }
1364
1365      long totalCost = 0;
1366      for (int i = 0 ; i < costsPerGroup.length; i++) {
1367        totalCost += costsPerGroup[i];
1368      }
1369      return scale(0, maxCost, totalCost);
1370    }
1371
1372    /**
1373     * For each primary region, it computes the total number of replicas in the array (numReplicas)
1374     * and returns a sum of numReplicas-1 squared. For example, if the server hosts
1375     * regions a, b, c, d, e, f where a and b are same replicas, and c,d,e are same replicas, it
1376     * returns (2-1) * (2-1) + (3-1) * (3-1) + (1-1) * (1-1).
1377     * @param primariesOfRegions a sorted array of primary regions ids for the regions hosted
1378     * @return a sum of numReplicas-1 squared for each primary region in the group.
1379     */
1380    protected long costPerGroup(int[] primariesOfRegions) {
1381      long cost = 0;
1382      int currentPrimary = -1;
1383      int currentPrimaryIndex = -1;
1384      // primariesOfRegions is a sorted array of primary ids of regions. Replicas of regions
1385      // sharing the same primary will have consecutive numbers in the array.
1386      for (int j = 0 ; j <= primariesOfRegions.length; j++) {
1387        int primary = j < primariesOfRegions.length ? primariesOfRegions[j] : -1;
1388        if (primary != currentPrimary) { // we see a new primary
1389          int numReplicas = j - currentPrimaryIndex;
1390          // square the cost
1391          if (numReplicas > 1) { // means consecutive primaries, indicating co-location
1392            cost += (numReplicas - 1) * (numReplicas - 1);
1393          }
1394          currentPrimary = primary;
1395          currentPrimaryIndex = j;
1396        }
1397      }
1398
1399      return cost;
1400    }
1401
1402    @Override
1403    protected void regionMoved(int region, int oldServer, int newServer) {
1404      if (maxCost <= 0) {
1405        return; // no need to compute
1406      }
1407      if (cluster.multiServersPerHost) {
1408        int oldHost = cluster.serverIndexToHostIndex[oldServer];
1409        int newHost = cluster.serverIndexToHostIndex[newServer];
1410        if (newHost != oldHost) {
1411          costsPerGroup[oldHost] = costPerGroup(cluster.primariesOfRegionsPerHost[oldHost]);
1412          costsPerGroup[newHost] = costPerGroup(cluster.primariesOfRegionsPerHost[newHost]);
1413        }
1414      } else {
1415        costsPerGroup[oldServer] = costPerGroup(cluster.primariesOfRegionsPerServer[oldServer]);
1416        costsPerGroup[newServer] = costPerGroup(cluster.primariesOfRegionsPerServer[newServer]);
1417      }
1418    }
1419  }

1421  /**
1422   * A cost function for region replicas for the rack distribution. We give a relatively high
1423   * cost to hosting replicas of the same region in the same rack. We do not prevent the case
1424   * though.
1425   */
1426  static class RegionReplicaRackCostFunction extends RegionReplicaHostCostFunction {
1427    private static final String REGION_REPLICA_RACK_COST_KEY =
1428        "hbase.master.balancer.stochastic.regionReplicaRackCostKey";
1429    private static final float DEFAULT_REGION_REPLICA_RACK_COST_KEY = 10000;
1430
1431    public RegionReplicaRackCostFunction(Configuration conf) {
1432      super(conf);
1433      this.setMultiplier(conf.getFloat(REGION_REPLICA_RACK_COST_KEY,
1434        DEFAULT_REGION_REPLICA_RACK_COST_KEY));
1435    }
1436
1437    @Override
1438    void init(Cluster cluster) {
1439      this.cluster = cluster;
1440      if (cluster.numRacks <= 1) {
1441        maxCost = 0;
1442        return; // disabled for 1 rack
1443      }
1444      // max cost is the case where every region replica is hosted together regardless of rack
1445      maxCost = getMaxCost(cluster);
1446      costsPerGroup = new long[cluster.numRacks];
1447      for (int i = 0 ; i < cluster.primariesOfRegionsPerRack.length; i++) {
1448        costsPerGroup[i] = costPerGroup(cluster.primariesOfRegionsPerRack[i]);
1449      }
1450    }
1451
1452    @Override
1453    protected void regionMoved(int region, int oldServer, int newServer) {
1454      if (maxCost <= 0) {
1455        return; // no need to compute
1456      }
1457      int oldRack = cluster.serverIndexToRackIndex[oldServer];
1458      int newRack = cluster.serverIndexToRackIndex[newServer];
1459      if (newRack != oldRack) {
1460        costsPerGroup[oldRack] = costPerGroup(cluster.primariesOfRegionsPerRack[oldRack]);
1461        costsPerGroup[newRack] = costPerGroup(cluster.primariesOfRegionsPerRack[newRack]);
1462      }
1463    }
