018package org.apache.hadoop.hbase.master.balancer;
019
020import org.apache.hadoop.conf.Configuration;
021import org.apache.yetus.audience.InterfaceAudience;
022
023/**
024 * Compute the cost of a potential cluster configuration based upon how evenly distributed tables
025 * are.
026 */
027@InterfaceAudience.Private
028class TableSkewCostFunction extends CostFunction {
029  private static final String TABLE_SKEW_COST_KEY =
030    "hbase.master.balancer.stochastic.tableSkewCost";
031  private static final float DEFAULT_TABLE_SKEW_COST = 35;
032  DoubleArrayCost[] costsPerTable;
033
034  TableSkewCostFunction(Configuration conf) {
035    this.setMultiplier(conf.getFloat(TABLE_SKEW_COST_KEY, DEFAULT_TABLE_SKEW_COST));
036  }
037
038  @Override
039  void prepare(BalancerClusterState cluster) {
040    super.prepare(cluster);
041    costsPerTable = new DoubleArrayCost[cluster.numTables];
042    for (int tableIdx = 0; tableIdx < cluster.numTables; tableIdx++) {
043      costsPerTable[tableIdx] = new DoubleArrayCost();
044      costsPerTable[tableIdx].prepare(cluster.numServers);
045      final int tableIndex = tableIdx;
046      costsPerTable[tableIdx].applyCostsChange(costs -> {
047        // Keep a cached deep copy for change-only recomputation
048        for (int i = 0; i < cluster.numServers; i++) {
049          costs[i] = cluster.numRegionsPerServerPerTable[tableIndex][i];
050        }
051      });
052    }
053  }
054
055  @Override
056  protected void regionMoved(int region, int oldServer, int newServer) {
057    int tableIdx = cluster.regionIndexToTableIndex[region];
058    costsPerTable[tableIdx].applyCostsChange(costs -> {
059      costs[oldServer] = cluster.numRegionsPerServerPerTable[tableIdx][oldServer];
060      costs[newServer] = cluster.numRegionsPerServerPerTable[tableIdx][newServer];
061    });
062  }
063
064  @Override
065  protected double cost() {
066    double cost = 0;
067    for (int tableIdx = 0; tableIdx < cluster.numTables; tableIdx++) {
068      cost += costsPerTable[tableIdx].cost();
069    }
070    return cost;
071  }
072}
