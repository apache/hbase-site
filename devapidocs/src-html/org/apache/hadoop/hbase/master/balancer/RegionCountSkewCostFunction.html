018package org.apache.hadoop.hbase.master.balancer;
019
020import org.apache.hadoop.conf.Configuration;
021import org.apache.yetus.audience.InterfaceAudience;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 * Compute the cost of a potential cluster state from skew in number of regions on a cluster.
027 */
028@InterfaceAudience.Private
029class RegionCountSkewCostFunction extends CostFunction {
030
031  private static final Logger LOG = LoggerFactory.getLogger(RegionCountSkewCostFunction.class);
032
033  static final String REGION_COUNT_SKEW_COST_KEY =
034    "hbase.master.balancer.stochastic.regionCountCost";
035  static final float DEFAULT_REGION_COUNT_SKEW_COST = 500;
036
037  private final DoubleArrayCost cost = new DoubleArrayCost();
038
039  RegionCountSkewCostFunction(Configuration conf) {
040    // Load multiplier should be the greatest as it is the most general way to balance data.
041    this.setMultiplier(conf.getFloat(REGION_COUNT_SKEW_COST_KEY, DEFAULT_REGION_COUNT_SKEW_COST));
042  }
043
044  @Override
045  void prepare(BalancerClusterState cluster) {
046    super.prepare(cluster);
047    cost.prepare(cluster.numServers);
048    cost.setCosts(costs -> {
049      for (int i = 0; i < cluster.numServers; i++) {
050        costs[i] = cluster.regionsPerServer[i].length;
051      }
052    });
053    LOG.debug("{} sees a total of {} servers and {} regions.", getClass().getSimpleName(),
054      cluster.numServers, cluster.numRegions);
055    if (LOG.isTraceEnabled()) {
056      for (int i = 0; i < cluster.numServers; i++) {
057        LOG.trace("{} sees server '{}' has {} regions", getClass().getSimpleName(),
058          cluster.servers[i], cluster.regionsPerServer[i].length);
059      }
060    }
061  }
062
063  @Override
064  protected double cost() {
065    return cost.cost();
066  }
067
068  @Override
069  protected void regionMoved(int region, int oldServer, int newServer) {
070    cost.setCosts(costs -> {
071      costs[oldServer] = cluster.regionsPerServer[oldServer].length;
072      costs[newServer] = cluster.regionsPerServer[newServer].length;
073    });
074  }
075}
