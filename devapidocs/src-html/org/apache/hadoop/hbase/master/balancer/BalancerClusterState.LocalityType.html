041/**
042 * An efficient array based implementation similar to ClusterState for keeping the status of the
043 * cluster in terms of region assignment and distribution. LoadBalancers, such as
044 * StochasticLoadBalancer uses this Cluster object because of hundreds of thousands of hashmap
045 * manipulations are very costly, which is why this class uses mostly indexes and arrays.
046 * <p/>
047 * BalancerClusterState tracks a list of unassigned regions, region assignments, and the server
048 * topology in terms of server names, hostnames and racks.
049 */
050@InterfaceAudience.Private
051class BalancerClusterState {
052
053  private static final Logger LOG = LoggerFactory.getLogger(BalancerClusterState.class);
054
055  ServerName[] servers;
056  // ServerName uniquely identifies a region server. multiple RS can run on the same host
057  String[] hosts;
058  String[] racks;
059  boolean multiServersPerHost = false; // whether or not any host has more than one server
060
061  ArrayList<String> tables;
062  RegionInfo[] regions;
063  Deque<BalancerRegionLoad>[] regionLoads;
064  private RegionHDFSBlockLocationFinder regionFinder;
065
066  int[][] regionLocations; // regionIndex -> list of serverIndex sorted by locality
067
068  int[] serverIndexToHostIndex; // serverIndex -> host index
069  int[] serverIndexToRackIndex; // serverIndex -> rack index
070
071  int[][] regionsPerServer; // serverIndex -> region list
072  int[] serverIndexToRegionsOffset; // serverIndex -> offset of region list
073  int[][] regionsPerHost; // hostIndex -> list of regions
074  int[][] regionsPerRack; // rackIndex -> region list
075  Int2IntCounterMap[] colocatedReplicaCountsPerServer; // serverIndex -> counts of colocated
076                                       // replicas by primary region index
077  Int2IntCounterMap[] colocatedReplicaCountsPerHost; // hostIndex -> counts of colocated replicas by
078                                      // primary region index
079  Int2IntCounterMap[] colocatedReplicaCountsPerRack; // rackIndex -> counts of colocated replicas by
080                                      // primary region index
081
082  int[][] serversPerHost; // hostIndex -> list of server indexes
083  int[][] serversPerRack; // rackIndex -> list of server indexes
084  int[] regionIndexToServerIndex; // regionIndex -> serverIndex
085  int[] initialRegionIndexToServerIndex; // regionIndex -> serverIndex (initial cluster state)
086  int[] regionIndexToTableIndex; // regionIndex -> tableIndex
087  int[][] numRegionsPerServerPerTable; // serverIndex -> tableIndex -> # regions
088  int[] numRegionsPerTable; // tableIndex -> region count
089  double[] meanRegionsPerTable; // mean region count per table
090  double[] regionSkewByTable;       // skew on RS per by table
091  double[] minRegionSkewByTable;       // min skew on RS per by table
092  double[] maxRegionSkewByTable;       // max skew on RS per by table
093  int[] regionIndexToPrimaryIndex; // regionIndex -> regionIndex of the primary
094  boolean hasRegionReplicas = false; // whether there is regions with replicas
095
096  Integer[] serverIndicesSortedByRegionCount;
097  Integer[] serverIndicesSortedByLocality;
098
099  Map<Address, Integer> serversToIndex;
100  Map<String, Integer> hostsToIndex;
101  Map<String, Integer> racksToIndex;
102  Map<String, Integer> tablesToIndex;
103  Map<RegionInfo, Integer> regionsToIndex;
104  float[] localityPerServer;
105
106  int numServers;
107  int numHosts;
108  int numRacks;
109  int numTables;
110  int numRegions;
111
112  int numMovedRegions = 0; // num moved regions from the initial configuration
113  Map<ServerName, List<RegionInfo>> clusterState;
114
115  private final RackManager rackManager;
116  // Maps region -> rackIndex -> locality of region on rack
117  private float[][] rackLocalities;
118  // Maps localityType -> region -> [server|rack]Index with highest locality
119  private int[][] regionsToMostLocalEntities;

134  @SuppressWarnings("unchecked")
135  BalancerClusterState(Collection<RegionInfo> unassignedRegions,
136    Map<ServerName, List<RegionInfo>> clusterState, Map<String, Deque<BalancerRegionLoad>> loads,
137    RegionHDFSBlockLocationFinder regionFinder, RackManager rackManager) {
138    if (unassignedRegions == null) {
139      unassignedRegions = Collections.emptyList();
140    }
141
142    serversToIndex = new HashMap<>();
143    hostsToIndex = new HashMap<>();
144    racksToIndex = new HashMap<>();
145    tablesToIndex = new HashMap<>();
146
147    // TODO: We should get the list of tables from master
148    tables = new ArrayList<>();
149    this.rackManager = rackManager != null ? rackManager : new DefaultRackManager();
150
151    numRegions = 0;
152
153    List<List<Integer>> serversPerHostList = new ArrayList<>();
154    List<List<Integer>> serversPerRackList = new ArrayList<>();
155    this.clusterState = clusterState;
156    this.regionFinder = regionFinder;
157
158    // Use servername and port as there can be dead servers in this list. We want everything with
159    // a matching hostname and port to have the same index.
160    for (ServerName sn : clusterState.keySet()) {
161      if (sn == null) {
162        LOG.warn("TODO: Enable TRACE on BaseLoadBalancer. Empty servername); " +
163          "skipping; unassigned regions?");
164        if (LOG.isTraceEnabled()) {
165          LOG.trace("EMPTY SERVERNAME " + clusterState.toString());
166        }
167        continue;
168      }
169      if (serversToIndex.get(sn.getAddress()) == null) {
170        serversToIndex.put(sn.getAddress(), numServers++);
171      }
172      if (!hostsToIndex.containsKey(sn.getHostname())) {
173        hostsToIndex.put(sn.getHostname(), numHosts++);
174        serversPerHostList.add(new ArrayList<>(1));
175      }
176
177      int serverIndex = serversToIndex.get(sn.getAddress());
178      int hostIndex = hostsToIndex.get(sn.getHostname());
179      serversPerHostList.get(hostIndex).add(serverIndex);
180
181      String rack = this.rackManager.getRack(sn);
182      if (!racksToIndex.containsKey(rack)) {
183        racksToIndex.put(rack, numRacks++);
184        serversPerRackList.add(new ArrayList<>());
185      }
186      int rackIndex = racksToIndex.get(rack);
187      serversPerRackList.get(rackIndex).add(serverIndex);
188    }
189
190    // Count how many regions there are.
191    for (Map.Entry<ServerName, List<RegionInfo>> entry : clusterState.entrySet()) {
192      numRegions += entry.getValue().size();
193    }
194    numRegions += unassignedRegions.size();
195
196    regionsToIndex = new HashMap<>(numRegions);
197    servers = new ServerName[numServers];
198    serversPerHost = new int[numHosts][];
199    serversPerRack = new int[numRacks][];
200    regions = new RegionInfo[numRegions];
201    regionIndexToServerIndex = new int[numRegions];
202    initialRegionIndexToServerIndex = new int[numRegions];
203    regionIndexToTableIndex = new int[numRegions];
204    regionIndexToPrimaryIndex = new int[numRegions];
205    regionLoads = new Deque[numRegions];
206
207    regionLocations = new int[numRegions][];
208    serverIndicesSortedByRegionCount = new Integer[numServers];
209    serverIndicesSortedByLocality = new Integer[numServers];
210    localityPerServer = new float[numServers];
211
212    serverIndexToHostIndex = new int[numServers];
213    serverIndexToRackIndex = new int[numServers];
214    regionsPerServer = new int[numServers][];
215    serverIndexToRegionsOffset = new int[numServers];
216    regionsPerHost = new int[numHosts][];
217    regionsPerRack = new int[numRacks][];
218    colocatedReplicaCountsPerServer = new Int2IntCounterMap[numServers];
219    colocatedReplicaCountsPerHost = new Int2IntCounterMap[numHosts];
220    colocatedReplicaCountsPerRack = new Int2IntCounterMap[numRacks];
221
222    int tableIndex = 0, regionIndex = 0, regionPerServerIndex = 0;
223
224    for (Map.Entry<ServerName, List<RegionInfo>> entry : clusterState.entrySet()) {
225      if (entry.getKey() == null) {
226        LOG.warn("SERVERNAME IS NULL, skipping " + entry.getValue());
227        continue;
228      }
229      int serverIndex = serversToIndex.get(entry.getKey().getAddress());
230
231      // keep the servername if this is the first server name for this hostname
232      // or this servername has the newest startcode.
233      if (servers[serverIndex] == null ||
234        servers[serverIndex].getStartcode() < entry.getKey().getStartcode()) {
235        servers[serverIndex] = entry.getKey();
236      }
237
238      if (regionsPerServer[serverIndex] != null) {
239        // there is another server with the same hostAndPort in ClusterState.
240        // allocate the array for the total size
241        regionsPerServer[serverIndex] =
242          new int[entry.getValue().size() + regionsPerServer[serverIndex].length];
243      } else {
244        regionsPerServer[serverIndex] = new int[entry.getValue().size()];
245      }
246      colocatedReplicaCountsPerServer[serverIndex] = new Int2IntCounterMap(
247        regionsPerServer[serverIndex].length, Hashing.DEFAULT_LOAD_FACTOR, 0);
248      serverIndicesSortedByRegionCount[serverIndex] = serverIndex;
249      serverIndicesSortedByLocality[serverIndex] = serverIndex;
250    }
251
252    hosts = new String[numHosts];
253    for (Map.Entry<String, Integer> entry : hostsToIndex.entrySet()) {
254      hosts[entry.getValue()] = entry.getKey();
255    }
256    racks = new String[numRacks];
257    for (Map.Entry<String, Integer> entry : racksToIndex.entrySet()) {
258      racks[entry.getValue()] = entry.getKey();
259    }
260
261    for (Map.Entry<ServerName, List<RegionInfo>> entry : clusterState.entrySet()) {
262      int serverIndex = serversToIndex.get(entry.getKey().getAddress());
263      regionPerServerIndex = serverIndexToRegionsOffset[serverIndex];
264
265      int hostIndex = hostsToIndex.get(entry.getKey().getHostname());
266      serverIndexToHostIndex[serverIndex] = hostIndex;
267
268      int rackIndex = racksToIndex.get(this.rackManager.getRack(entry.getKey()));
269      serverIndexToRackIndex[serverIndex] = rackIndex;
270
271      for (RegionInfo region : entry.getValue()) {
272        registerRegion(region, regionIndex, serverIndex, loads, regionFinder);
273        regionsPerServer[serverIndex][regionPerServerIndex++] = regionIndex;
274        regionIndex++;
275      }
276      serverIndexToRegionsOffset[serverIndex] = regionPerServerIndex;
277    }
278
279    for (RegionInfo region : unassignedRegions) {
280      registerRegion(region, regionIndex, -1, loads, regionFinder);
281      regionIndex++;
282    }
283
284    for (int i = 0; i < serversPerHostList.size(); i++) {
285      serversPerHost[i] = new int[serversPerHostList.get(i).size()];
286      for (int j = 0; j < serversPerHost[i].length; j++) {
287        serversPerHost[i][j] = serversPerHostList.get(i).get(j);
288      }
289      if (serversPerHost[i].length > 1) {
290        multiServersPerHost = true;
291      }
292    }
293
294    for (int i = 0; i < serversPerRackList.size(); i++) {
295      serversPerRack[i] = new int[serversPerRackList.get(i).size()];
296      for (int j = 0; j < serversPerRack[i].length; j++) {
297        serversPerRack[i][j] = serversPerRackList.get(i).get(j);
298      }
299    }
300
301    numTables = tables.size();
302    LOG.debug("Number of tables={}", numTables);
303    numRegionsPerServerPerTable = new int[numServers][numTables];
304    numRegionsPerTable = new int[numTables];
305
306    for (int i = 0; i < numServers; i++) {
307      for (int j = 0; j < numTables; j++) {
308        numRegionsPerServerPerTable[i][j] = 0;
309      }
310    }
311
312    for (int i = 0; i < regionIndexToServerIndex.length; i++) {
313      if (regionIndexToServerIndex[i] >= 0) {
314        numRegionsPerServerPerTable[regionIndexToServerIndex[i]][regionIndexToTableIndex[i]]++;
315        numRegionsPerTable[regionIndexToTableIndex[i]]++;
316      }
317    }
318
319    // Avoid repeated computation for planning
320    meanRegionsPerTable = new double[numTables];
321    regionSkewByTable = new double[numTables];
322    maxRegionSkewByTable  = new double[numTables];
323    minRegionSkewByTable = new double[numTables];
324
325    for (int i = 0; i < numTables; i++) {
326      meanRegionsPerTable[i] = Double.valueOf(numRegionsPerTable[i]) / numServers;
327      minRegionSkewByTable[i] += DoubleArrayCost.getMinSkew(numRegionsPerTable[i], numServers);
328      maxRegionSkewByTable[i] += DoubleArrayCost.getMaxSkew(numRegionsPerTable[i], numServers);
329    }
330
331    for (int[] aNumRegionsPerServerPerTable : numRegionsPerServerPerTable) {
332      for (int tableIdx = 0; tableIdx < aNumRegionsPerServerPerTable.length; tableIdx++) {
333        regionSkewByTable[tableIdx] += Math.abs(aNumRegionsPerServerPerTable[tableIdx] -
334          meanRegionsPerTable[tableIdx]);
335      }
336    }
337
338    for (int i = 0; i < regions.length; i++) {
339      RegionInfo info = regions[i];
340      if (RegionReplicaUtil.isDefaultReplica(info)) {
341        regionIndexToPrimaryIndex[i] = i;
342      } else {
343        hasRegionReplicas = true;
344        RegionInfo primaryInfo = RegionReplicaUtil.getRegionInfoForDefaultReplica(info);
345        regionIndexToPrimaryIndex[i] = regionsToIndex.getOrDefault(primaryInfo, -1);
346      }
347    }
348
349    for (int i = 0; i < regionsPerServer.length; i++) {
350      colocatedReplicaCountsPerServer[i] = new Int2IntCounterMap(
351        regionsPerServer[i].length, Hashing.DEFAULT_LOAD_FACTOR, 0);
352      for (int j = 0; j < regionsPerServer[i].length; j++) {
353        int primaryIndex = regionIndexToPrimaryIndex[regionsPerServer[i][j]];
354        colocatedReplicaCountsPerServer[i].getAndIncrement(primaryIndex);
355      }
356    }
357    // compute regionsPerHost
358    if (multiServersPerHost) {
359      populateRegionPerLocationFromServer(regionsPerHost, colocatedReplicaCountsPerHost,
360        serversPerHost);
361    }
362
363    // compute regionsPerRack
364    if (numRacks > 1) {
365      populateRegionPerLocationFromServer(regionsPerRack, colocatedReplicaCountsPerRack,
366        serversPerRack);
367    }
368  }
369
370  private void populateRegionPerLocationFromServer(int[][] regionsPerLocation,
371    Int2IntCounterMap[] colocatedReplicaCountsPerLocation,
372    int[][] serversPerLocation) {
373    for (int i = 0; i < serversPerLocation.length; i++) {
374      int numRegionsPerLocation = 0;
375      for (int j = 0; j < serversPerLocation[i].length; j++) {
376        numRegionsPerLocation += regionsPerServer[serversPerLocation[i][j]].length;
377      }
378      regionsPerLocation[i] = new int[numRegionsPerLocation];
379      colocatedReplicaCountsPerLocation[i] = new Int2IntCounterMap(numRegionsPerLocation,
380        Hashing.DEFAULT_LOAD_FACTOR, 0);
381    }
382
383    for (int i = 0; i < serversPerLocation.length; i++) {
384      int numRegionPerLocationIndex = 0;
385      for (int j = 0; j < serversPerLocation[i].length; j++) {
386        for (int k = 0; k < regionsPerServer[serversPerLocation[i][j]].length; k++) {
387          int region = regionsPerServer[serversPerLocation[i][j]][k];
388          regionsPerLocation[i][numRegionPerLocationIndex] = region;
389          int primaryIndex = regionIndexToPrimaryIndex[region];
390          colocatedReplicaCountsPerLocation[i].getAndIncrement(primaryIndex);
391          numRegionPerLocationIndex++;
392        }
393      }
394    }
395
396  }

398  /** Helper for Cluster constructor to handle a region */
399  private void registerRegion(RegionInfo region, int regionIndex, int serverIndex,
400    Map<String, Deque<BalancerRegionLoad>> loads, RegionHDFSBlockLocationFinder regionFinder) {
401    String tableName = region.getTable().getNameAsString();
402    if (!tablesToIndex.containsKey(tableName)) {
403      tables.add(tableName);
404      tablesToIndex.put(tableName, tablesToIndex.size());
405    }
406    int tableIndex = tablesToIndex.get(tableName);
407
408    regionsToIndex.put(region, regionIndex);
409    regions[regionIndex] = region;
410    regionIndexToServerIndex[regionIndex] = serverIndex;
411    initialRegionIndexToServerIndex[regionIndex] = serverIndex;
412    regionIndexToTableIndex[regionIndex] = tableIndex;
413
414    // region load
415    if (loads != null) {
416      Deque<BalancerRegionLoad> rl = loads.get(region.getRegionNameAsString());
417      // That could have failed if the RegionLoad is using the other regionName
418      if (rl == null) {
419        // Try getting the region load using encoded name.
420        rl = loads.get(region.getEncodedName());
421      }
422      regionLoads[regionIndex] = rl;
423    }
424
425    if (regionFinder != null) {
426      // region location
427      List<ServerName> loc = regionFinder.getTopBlockLocations(region);
428      regionLocations[regionIndex] = new int[loc.size()];
429      for (int i = 0; i < loc.size(); i++) {
430        regionLocations[regionIndex][i] = loc.get(i) == null ? -1 :
431          (serversToIndex.get(loc.get(i).getAddress()) == null ? -1 :
432            serversToIndex.get(loc.get(i).getAddress()));
433      }
434    }
435  }

437  /**
438   * Returns true iff a given server has less regions than the balanced amount
439   */
440  public boolean serverHasTooFewRegions(int server) {
441    int minLoad = this.numRegions / numServers;
442    int numRegions = getNumRegions(server);
443    return numRegions < minLoad;

446  /**
447   * Retrieves and lazily initializes a field storing the locality of every region/server
448   * combination
449   */
450  public float[][] getOrComputeRackLocalities() {
451    if (rackLocalities == null || regionsToMostLocalEntities == null) {
452      computeCachedLocalities();
453    }
454    return rackLocalities;
455  }

457  /**
458   * Lazily initializes and retrieves a mapping of region -> server for which region has the highest
459   * the locality
460   */
461  public int[] getOrComputeRegionsToMostLocalEntities(BalancerClusterState.LocalityType type) {
462    if (rackLocalities == null || regionsToMostLocalEntities == null) {
463      computeCachedLocalities();
464    }
465    return regionsToMostLocalEntities[type.ordinal()];
466  }

468  /**
469   * Looks up locality from cache of localities. Will create cache if it does not already exist.
470   */
471  public float getOrComputeLocality(int region, int entity,
472    BalancerClusterState.LocalityType type) {
473    switch (type) {
474      case SERVER:
475        return getLocalityOfRegion(region, entity);
476      case RACK:
477        return getOrComputeRackLocalities()[region][entity];
478      default:
479        throw new IllegalArgumentException("Unsupported LocalityType: " + type);
480    }
481  }

483  /**
484   * Returns locality weighted by region size in MB. Will create locality cache if it does not
485   * already exist.
486   */
487  public double getOrComputeWeightedLocality(int region, int server,
488    BalancerClusterState.LocalityType type) {
489    return getRegionSizeMB(region) * getOrComputeLocality(region, server, type);
490  }

492  /**
493   * Returns the size in MB from the most recent RegionLoad for region
494   */
495  public int getRegionSizeMB(int region) {
496    Deque<BalancerRegionLoad> load = regionLoads[region];
497    // This means regions have no actual data on disk
498    if (load == null) {
499      return 0;
500    }
501    return regionLoads[region].getLast().getStorefileSizeMB();
502  }

504  /**
505   * Computes and caches the locality for each region/rack combinations, as well as storing a
506   * mapping of region -> server and region -> rack such that server and rack have the highest
507   * locality for region
508   */
509  private void computeCachedLocalities() {
510    rackLocalities = new float[numRegions][numRacks];
511    regionsToMostLocalEntities = new int[LocalityType.values().length][numRegions];
512
513    // Compute localities and find most local server per region
514    for (int region = 0; region < numRegions; region++) {
515      int serverWithBestLocality = 0;
516      float bestLocalityForRegion = 0;
517      for (int server = 0; server < numServers; server++) {
518        // Aggregate per-rack locality
519        float locality = getLocalityOfRegion(region, server);
520        int rack = serverIndexToRackIndex[server];
521        int numServersInRack = serversPerRack[rack].length;
522        rackLocalities[region][rack] += locality / numServersInRack;
523
524        if (locality > bestLocalityForRegion) {
525          serverWithBestLocality = server;
526          bestLocalityForRegion = locality;
527        }
528      }
529      regionsToMostLocalEntities[LocalityType.SERVER.ordinal()][region] = serverWithBestLocality;
530
531      // Find most local rack per region
532      int rackWithBestLocality = 0;
533      float bestRackLocalityForRegion = 0.0f;
534      for (int rack = 0; rack < numRacks; rack++) {
535        float rackLocality = rackLocalities[region][rack];
536        if (rackLocality > bestRackLocalityForRegion) {
537          bestRackLocalityForRegion = rackLocality;
538          rackWithBestLocality = rack;
539        }
540      }
541      regionsToMostLocalEntities[LocalityType.RACK.ordinal()][region] = rackWithBestLocality;
542    }
543
544  }

546  /**
547   * Maps region index to rack index

549  public int getRackForRegion(int region) {
550    return serverIndexToRackIndex[regionIndexToServerIndex[region]];

557  public void doAction(BalanceAction action) {
558    switch (action.getType()) {
559      case NULL:
560        break;
561      case ASSIGN_REGION:
562        // FindBugs: Having the assert quietens FB BC_UNCONFIRMED_CAST warnings
563        assert action instanceof AssignRegionAction : action.getClass();
564        AssignRegionAction ar = (AssignRegionAction) action;
565        regionsPerServer[ar.getServer()] =
566          addRegion(regionsPerServer[ar.getServer()], ar.getRegion());
567        regionMoved(ar.getRegion(), -1, ar.getServer());
568        break;
569      case MOVE_REGION:
570        assert action instanceof MoveRegionAction : action.getClass();
571        MoveRegionAction mra = (MoveRegionAction) action;
572        regionsPerServer[mra.getFromServer()] =
573          removeRegion(regionsPerServer[mra.getFromServer()], mra.getRegion());
574        regionsPerServer[mra.getToServer()] =
575          addRegion(regionsPerServer[mra.getToServer()], mra.getRegion());
576        regionMoved(mra.getRegion(), mra.getFromServer(), mra.getToServer());
577        break;
578      case SWAP_REGIONS:
579        assert action instanceof SwapRegionsAction : action.getClass();
580        SwapRegionsAction a = (SwapRegionsAction) action;
581        regionsPerServer[a.getFromServer()] =
582          replaceRegion(regionsPerServer[a.getFromServer()], a.getFromRegion(), a.getToRegion());
583        regionsPerServer[a.getToServer()] =
584          replaceRegion(regionsPerServer[a.getToServer()], a.getToRegion(), a.getFromRegion());
585        regionMoved(a.getFromRegion(), a.getFromServer(), a.getToServer());
586        regionMoved(a.getToRegion(), a.getToServer(), a.getFromServer());
587        break;
588      default:
589        throw new RuntimeException("Uknown action:" + action.getType());
590    }
591  }

593  /**
594   * Return true if the placement of region on server would lower the availability of the region in
595   * question
596   * @return true or false
597   */
598  boolean wouldLowerAvailability(RegionInfo regionInfo, ServerName serverName) {
599    if (!serversToIndex.containsKey(serverName.getAddress())) {
600      return false; // safeguard against race between cluster.servers and servers from LB method
601      // args
602    }
603    int server = serversToIndex.get(serverName.getAddress());
604    int region = regionsToIndex.get(regionInfo);
605
606    // Region replicas for same region should better assign to different servers
607    for (int i : regionsPerServer[server]) {
608      RegionInfo otherRegionInfo = regions[i];
609      if (RegionReplicaUtil.isReplicasForSameRegion(regionInfo, otherRegionInfo)) {
610        return true;
611      }
612    }
613
614    int primary = regionIndexToPrimaryIndex[region];
615    if (primary == -1) {
616      return false;
617    }
618    // there is a subset relation for server < host < rack
619    // check server first
620    int result = checkLocationForPrimary(server, colocatedReplicaCountsPerServer, primary);
621    if (result != 0) {
622      return result > 0;
623    }
624
625    // check host
626    if (multiServersPerHost) {
627      result = checkLocationForPrimary(serverIndexToHostIndex[server],
628        colocatedReplicaCountsPerHost, primary);
629      if (result != 0) {
630        return result > 0;
631      }
632    }
633
634    // check rack
635    if (numRacks > 1) {
636      result = checkLocationForPrimary(serverIndexToRackIndex[server],
637        colocatedReplicaCountsPerRack, primary);
638      if (result != 0) {
639        return result > 0;
640      }
641    }
642    return false;
643  }

645  /**
646   * Common method for better solution check.
647   * @param colocatedReplicaCountsPerLocation colocatedReplicaCountsPerHost or
648   *                                          colocatedReplicaCountsPerRack
649   * @return 1 for better, -1 for no better, 0 for unknown
650   */
651  private int checkLocationForPrimary(int location,
652    Int2IntCounterMap[] colocatedReplicaCountsPerLocation, int primary) {
653    if (colocatedReplicaCountsPerLocation[location].containsKey(primary)) {
654      // check for whether there are other Locations that we can place this region
655      for (int i = 0; i < colocatedReplicaCountsPerLocation.length; i++) {
656        if (i != location && !colocatedReplicaCountsPerLocation[i].containsKey(primary)) {
657          return 1; // meaning there is a better Location
658        }
659      }
660      return -1; // there is not a better Location to place this
661    }
662    return 0;
663  }

674  void regionMoved(int region, int oldServer, int newServer) {
675    regionIndexToServerIndex[region] = newServer;
676    if (initialRegionIndexToServerIndex[region] == newServer) {
677      numMovedRegions--; // region moved back to original location
678    } else if (oldServer >= 0 && initialRegionIndexToServerIndex[region] == oldServer) {
679      numMovedRegions++; // region moved from original location
680    }
681    int tableIndex = regionIndexToTableIndex[region];
682    if (oldServer >= 0) {
683      numRegionsPerServerPerTable[oldServer][tableIndex]--;
684      // update regionSkewPerTable for the move from old server
685      regionSkewByTable[tableIndex] += getSkewChangeFor(oldServer, tableIndex, -1);
686    }
687    numRegionsPerServerPerTable[newServer][tableIndex]++;
688
689    // update regionSkewPerTable for the move to new server
690    regionSkewByTable[tableIndex] += getSkewChangeFor(newServer, tableIndex, 1);
691
692    // update for servers
693    int primary = regionIndexToPrimaryIndex[region];
694    if (oldServer >= 0) {
695      colocatedReplicaCountsPerServer[oldServer].getAndDecrement(primary);
696    }
697    colocatedReplicaCountsPerServer[newServer].getAndIncrement(primary);
698
699    // update for hosts
700    if (multiServersPerHost) {
701      updateForLocation(serverIndexToHostIndex, regionsPerHost, colocatedReplicaCountsPerHost,
702        oldServer, newServer, primary, region);
703    }
704
705    // update for racks
706    if (numRacks > 1) {
707      updateForLocation(serverIndexToRackIndex, regionsPerRack, colocatedReplicaCountsPerRack,
708        oldServer, newServer, primary, region);
709    }
710  }
711  /**
712   * Common method for per host and per Location region index updates when a region is moved.
713   * @param serverIndexToLocation serverIndexToHostIndex or serverIndexToLocationIndex
714   * @param regionsPerLocation regionsPerHost or regionsPerLocation
715   * @param colocatedReplicaCountsPerLocation colocatedReplicaCountsPerHost or
716   *                                          colocatedReplicaCountsPerRack
717   */
718  private void updateForLocation(int[] serverIndexToLocation,
719    int[][] regionsPerLocation,
720    Int2IntCounterMap[] colocatedReplicaCountsPerLocation,
721    int oldServer, int newServer, int primary, int region) {
722    int oldLocation = oldServer >= 0 ? serverIndexToLocation[oldServer] : -1;
723    int newLocation = serverIndexToLocation[newServer];
724    if (newLocation != oldLocation) {
725      regionsPerLocation[newLocation] = addRegion(regionsPerLocation[newLocation], region);
726      colocatedReplicaCountsPerLocation[newLocation].getAndIncrement(primary);
727      if (oldLocation >= 0) {
728        regionsPerLocation[oldLocation] = removeRegion(regionsPerLocation[oldLocation], region);
729        colocatedReplicaCountsPerLocation[oldLocation].getAndDecrement(primary);
730      }
731    }
732
733  }
734  int[] removeRegion(int[] regions, int regionIndex) {
735    // TODO: this maybe costly. Consider using linked lists
736    int[] newRegions = new int[regions.length - 1];
737    int i = 0;
738    for (i = 0; i < regions.length; i++) {
739      if (regions[i] == regionIndex) {
740        break;
741      }
742      newRegions[i] = regions[i];
743    }
744    System.arraycopy(regions, i + 1, newRegions, i, newRegions.length - i);
745    return newRegions;
746  }

748  int[] addRegion(int[] regions, int regionIndex) {
749    int[] newRegions = new int[regions.length + 1];
750    System.arraycopy(regions, 0, newRegions, 0, regions.length);
751    newRegions[newRegions.length - 1] = regionIndex;
752    return newRegions;
753  }

755  int[] addRegionSorted(int[] regions, int regionIndex) {
756    int[] newRegions = new int[regions.length + 1];
757    int i = 0;
758    for (i = 0; i < regions.length; i++) { // find the index to insert
759      if (regions[i] > regionIndex) {
760        break;
761      }
762    }
763    System.arraycopy(regions, 0, newRegions, 0, i); // copy first half
764    System.arraycopy(regions, i, newRegions, i + 1, regions.length - i); // copy second half
765    newRegions[i] = regionIndex;
766
767    return newRegions;
768  }

770  int[] replaceRegion(int[] regions, int regionIndex, int newRegionIndex) {
771    int i = 0;
772    for (i = 0; i < regions.length; i++) {
773      if (regions[i] == regionIndex) {
774        regions[i] = newRegionIndex;
775        break;
776      }
777    }
778    return regions;

795  int getLowestLocalityRegionOnServer(int serverIndex) {
796    if (regionFinder != null) {
797      float lowestLocality = 1.0f;
798      int lowestLocalityRegionIndex = -1;
799      if (regionsPerServer[serverIndex].length == 0) {
800        // No regions on that region server
801        return -1;
802      }
803      for (int j = 0; j < regionsPerServer[serverIndex].length; j++) {
804        int regionIndex = regionsPerServer[serverIndex][j];
805        HDFSBlocksDistribution distribution =
806          regionFinder.getBlockDistribution(regions[regionIndex]);
807        float locality = distribution.getBlockLocalityIndex(servers[serverIndex].getHostname());
808        // skip empty region
809        if (distribution.getUniqueBlocksTotalWeight() == 0) {
810          continue;
811        }
812        if (locality < lowestLocality) {
813          lowestLocality = locality;
814          lowestLocalityRegionIndex = j;
815        }
816      }
817      if (lowestLocalityRegionIndex == -1) {
818        return -1;
819      }
820      if (LOG.isTraceEnabled()) {
821        LOG.trace("Lowest locality region is " +
822          regions[regionsPerServer[serverIndex][lowestLocalityRegionIndex]]
823            .getRegionNameAsString() +
824          " with locality " + lowestLocality + " and its region server contains " +
825          regionsPerServer[serverIndex].length + " regions");
826      }
827      return regionsPerServer[serverIndex][lowestLocalityRegionIndex];
828    } else {
829      return -1;
830    }
831  }

867  private double getSkewChangeFor(int serverIndex, int tableIndex, int regionCountChange) {
868    double curSkew = Math.abs(numRegionsPerServerPerTable[serverIndex][tableIndex] -
869      meanRegionsPerTable[tableIndex]);
870    double oldSkew = Math.abs(numRegionsPerServerPerTable[serverIndex][tableIndex] -
871      regionCountChange - meanRegionsPerTable[tableIndex]);
872    return curSkew - oldSkew;
873  }
