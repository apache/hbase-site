018package org.apache.hadoop.hbase.master.replication;
019
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.List;
023
024import org.apache.hadoop.hbase.ServerName;
025import org.apache.hadoop.hbase.master.procedure.MasterProcedureEnv;
026import org.apache.hadoop.hbase.master.procedure.PeerProcedureInterface;
027import org.apache.hadoop.hbase.procedure2.ProcedureStateSerializer;
028import org.apache.hadoop.hbase.procedure2.ProcedureSuspendedException;
029import org.apache.hadoop.hbase.procedure2.ProcedureYieldException;
030import org.apache.hadoop.hbase.procedure2.StateMachineProcedure;
031import org.apache.hadoop.hbase.replication.ReplicationException;
032import org.apache.yetus.audience.InterfaceAudience;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.SyncReplicationReplayWALState;
037import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.SyncReplicationReplayWALStateData;
038
039@InterfaceAudience.Private
040public class SyncReplicationReplayWALProcedure
041    extends StateMachineProcedure<MasterProcedureEnv, SyncReplicationReplayWALState>
042    implements PeerProcedureInterface {
043
044  private static final Logger LOG =
045      LoggerFactory.getLogger(SyncReplicationReplayWALProcedure.class);
046
047  private String peerId;
048
049  private ServerName worker = null;
050
051  private List<String> wals;
052
053  public SyncReplicationReplayWALProcedure() {
054  }
055
056  public SyncReplicationReplayWALProcedure(String peerId, List<String> wals) {
057    this.peerId = peerId;
058    this.wals = wals;
059  }
060
061  @Override protected Flow executeFromState(MasterProcedureEnv env,
062      SyncReplicationReplayWALState state)
063      throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
064    SyncReplicationReplayWALManager syncReplicationReplayWALManager =
065        env.getMasterServices().getSyncReplicationReplayWALManager();
066    switch (state) {
067      case ASSIGN_WORKER:
068        try {
069          worker = syncReplicationReplayWALManager.getPeerWorker(peerId);
070        } catch (ReplicationException e) {
071          LOG.info("Failed to get worker to replay wals {} for peer id={}, retry", wals, peerId);
072          throw new ProcedureYieldException();
073        }
074        if (worker == null) {
075          LOG.info("No worker to replay wals {} for peer id={}, retry", wals, peerId);
076          setNextState(SyncReplicationReplayWALState.ASSIGN_WORKER);
077        } else {
078          setNextState(SyncReplicationReplayWALState.DISPATCH_WALS_TO_WORKER);
079        }
080        return Flow.HAS_MORE_STATE;
081      case DISPATCH_WALS_TO_WORKER:
082        addChildProcedure(new SyncReplicationReplayWALRemoteProcedure(peerId, wals, worker));
083        setNextState(SyncReplicationReplayWALState.RELEASE_WORKER);
084        return Flow.HAS_MORE_STATE;
085      case RELEASE_WORKER:
086        boolean finished = false;
087        try {
088          finished = syncReplicationReplayWALManager.isReplayWALFinished(wals.get(0));
089        } catch (IOException e) {
090          LOG.info("Failed to check whether replay wals {} finished for peer id={}", wals, peerId);
091          throw new ProcedureYieldException();
092        }
093        try {
094          syncReplicationReplayWALManager.removePeerWorker(peerId, worker);
095        } catch (ReplicationException e) {
096          LOG.info("Failed to remove worker for peer id={}, retry", peerId);
097          throw new ProcedureYieldException();
098        }
099        if (!finished) {
100          LOG.info("Failed to replay wals {} for peer id={}, retry", wals, peerId);
101          setNextState(SyncReplicationReplayWALState.ASSIGN_WORKER);
102          return Flow.HAS_MORE_STATE;
103        }
104        return Flow.NO_MORE_STATE;
105      default:
106        throw new UnsupportedOperationException("unhandled state=" + state);
107    }
108  }
109
110  @Override
111  protected void rollbackState(MasterProcedureEnv env,
112      SyncReplicationReplayWALState state)
113      throws IOException, InterruptedException {
114    if (state == getInitialState()) {
115      return;
116    }
117    throw new UnsupportedOperationException();
118  }
119
120  @Override
121  protected SyncReplicationReplayWALState getState(int state) {
122    return SyncReplicationReplayWALState.forNumber(state);
123  }
124
125  @Override
126  protected int getStateId(
127      SyncReplicationReplayWALState state) {
128    return state.getNumber();
129  }
130
131  @Override
132  protected SyncReplicationReplayWALState getInitialState() {
133    return SyncReplicationReplayWALState.ASSIGN_WORKER;
134  }
135
136  @Override
137  protected void serializeStateData(ProcedureStateSerializer serializer)
138      throws IOException {
139    SyncReplicationReplayWALStateData.Builder builder =
140        SyncReplicationReplayWALStateData.newBuilder();
141    builder.setPeerId(peerId);
142    wals.stream().forEach(builder::addWal);
143    serializer.serialize(builder.build());
144  }
145
146  @Override
147  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
148    SyncReplicationReplayWALStateData data =
149        serializer.deserialize(SyncReplicationReplayWALStateData.class);
150    peerId = data.getPeerId();
151    wals = new ArrayList<>();
152    data.getWalList().forEach(wals::add);
153  }
154
155  @Override
156  public String getPeerId() {
157    return peerId;
158  }
159
160  @Override
161  public PeerOperationType getPeerOperationType() {
162    return PeerOperationType.SYNC_REPLICATION_REPLAY_WAL;
163  }
164}
