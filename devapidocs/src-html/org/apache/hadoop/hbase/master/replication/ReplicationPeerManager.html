050/**
051 * Manages and performs all replication admin operations.
052 * <p>
053 * Used to add/remove a replication peer.
054 */
055@InterfaceAudience.Private
056public class ReplicationPeerManager {
057
058  private final ReplicationPeerStorage peerStorage;
059
060  private final ReplicationQueueStorage queueStorage;
061
062  private final ConcurrentMap<String, ReplicationPeerDescription> peers;
063
064  ReplicationPeerManager(ReplicationPeerStorage peerStorage, ReplicationQueueStorage queueStorage,
065      ConcurrentMap<String, ReplicationPeerDescription> peers) {
066    this.peerStorage = peerStorage;
067    this.queueStorage = queueStorage;
068    this.peers = peers;
069  }
070
071  private void checkQueuesDeleted(String peerId)
072      throws ReplicationException, DoNotRetryIOException {
073    for (ServerName replicator : queueStorage.getListOfReplicators()) {
074      List<String> queueIds = queueStorage.getAllQueues(replicator);
075      for (String queueId : queueIds) {
076        ReplicationQueueInfo queueInfo = new ReplicationQueueInfo(queueId);
077        if (queueInfo.getPeerId().equals(peerId)) {
078          throw new DoNotRetryIOException("undeleted queue for peerId: " + peerId +
079            ", replicator: " + replicator + ", queueId: " + queueId);
080        }
081      }
082    }
083    if (queueStorage.getAllPeersFromHFileRefsQueue().contains(peerId)) {
084      throw new DoNotRetryIOException("Undeleted queue for peer " + peerId + " in hfile-refs");
085    }
086  }
087
088  public void preAddPeer(String peerId, ReplicationPeerConfig peerConfig)
089      throws DoNotRetryIOException, ReplicationException {
090    if (peerId.contains("-")) {
091      throw new DoNotRetryIOException("Found invalid peer name: " + peerId);
092    }
093    checkPeerConfig(peerConfig);
094    if (peers.containsKey(peerId)) {
095      throw new DoNotRetryIOException("Replication peer " + peerId + " already exists");
096    }
097    // make sure that there is no queues with the same peer id. This may happen when we create a
098    // peer with the same id with a old deleted peer. If the replication queues for the old peer
099    // have not been cleaned up yet then we should not create the new peer, otherwise the old wal
100    // file may also be replicated.
101    checkQueuesDeleted(peerId);
102  }
103
104  private ReplicationPeerDescription checkPeerExists(String peerId) throws DoNotRetryIOException {
105    ReplicationPeerDescription desc = peers.get(peerId);
106    if (desc == null) {
107      throw new DoNotRetryIOException("Replication peer " + peerId + " does not exist");
108    }
109    return desc;
110  }
111
112  public void preRemovePeer(String peerId) throws DoNotRetryIOException {
113    checkPeerExists(peerId);
114  }
115
116  public void preEnablePeer(String peerId) throws DoNotRetryIOException {
117    ReplicationPeerDescription desc = checkPeerExists(peerId);
118    if (desc.isEnabled()) {
119      throw new DoNotRetryIOException("Replication peer " + peerId + " has already been enabled");
120    }
121  }
122
123  public void preDisablePeer(String peerId) throws DoNotRetryIOException {
124    ReplicationPeerDescription desc = checkPeerExists(peerId);
125    if (!desc.isEnabled()) {
126      throw new DoNotRetryIOException("Replication peer " + peerId + " has already been disabled");
127    }
128  }
129
130  public void preUpdatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
131      throws DoNotRetryIOException {
132    checkPeerConfig(peerConfig);
133    ReplicationPeerDescription desc = checkPeerExists(peerId);
134    ReplicationPeerConfig oldPeerConfig = desc.getPeerConfig();
135    if (!StringUtils.isBlank(peerConfig.getClusterKey()) &&
136      !peerConfig.getClusterKey().equals(oldPeerConfig.getClusterKey())) {
137      throw new DoNotRetryIOException(
138          "Changing the cluster key on an existing peer is not allowed. Existing key '" +
139            oldPeerConfig.getClusterKey() + "' for peer " + peerId + " does not match new key '" +
140            peerConfig.getClusterKey() + "'");
141    }
142
143    if (!StringUtils.isBlank(peerConfig.getReplicationEndpointImpl()) &&
144      !peerConfig.getReplicationEndpointImpl().equals(oldPeerConfig.getReplicationEndpointImpl())) {
145      throw new DoNotRetryIOException("Changing the replication endpoint implementation class " +
146        "on an existing peer is not allowed. Existing class '" +
147        oldPeerConfig.getReplicationEndpointImpl() + "' for peer " + peerId +
148        " does not match new class '" + peerConfig.getReplicationEndpointImpl() + "'");
149    }
150  }
151
152  public void addPeer(String peerId, ReplicationPeerConfig peerConfig, boolean enabled)
153      throws ReplicationException {
154    if (peers.containsKey(peerId)) {
155      // this should be a retry, just return
156      return;
157    }
158    ReplicationPeerConfig copiedPeerConfig = ReplicationPeerConfig.newBuilder(peerConfig).build();
159    peerStorage.addPeer(peerId, copiedPeerConfig, enabled);
160    peers.put(peerId, new ReplicationPeerDescription(peerId, enabled, copiedPeerConfig));
161  }
162
163  public void removePeer(String peerId) throws ReplicationException {
164    if (!peers.containsKey(peerId)) {
165      // this should be a retry, just return
166      return;
167    }
168    peerStorage.removePeer(peerId);
169    peers.remove(peerId);
170  }
171
172  private void setPeerState(String peerId, boolean enabled) throws ReplicationException {
173    ReplicationPeerDescription desc = peers.get(peerId);
174    if (desc.isEnabled() == enabled) {
175      // this should be a retry, just return
176      return;
177    }
178    peerStorage.setPeerState(peerId, enabled);
179    peers.put(peerId, new ReplicationPeerDescription(peerId, enabled, desc.getPeerConfig()));
180  }
181
182  public void enablePeer(String peerId) throws ReplicationException {
183    setPeerState(peerId, true);
184  }
185
186  public void disablePeer(String peerId) throws ReplicationException {
187    setPeerState(peerId, false);
188  }
189
190  public void updatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
191      throws ReplicationException {
192    // the checking rules are too complicated here so we give up checking whether this is a retry.
193    ReplicationPeerDescription desc = peers.get(peerId);
194    ReplicationPeerConfig oldPeerConfig = desc.getPeerConfig();
195    ReplicationPeerConfigBuilder newPeerConfigBuilder =
196        ReplicationPeerConfig.newBuilder(peerConfig);
197    // we need to use the new conf to overwrite the old one.
198    newPeerConfigBuilder.putAllConfiguration(oldPeerConfig.getConfiguration());
199    newPeerConfigBuilder.putAllConfiguration(peerConfig.getConfiguration());
200    newPeerConfigBuilder.putAllConfiguration(oldPeerConfig.getConfiguration());
201    newPeerConfigBuilder.putAllConfiguration(peerConfig.getConfiguration());
202    ReplicationPeerConfig newPeerConfig = newPeerConfigBuilder.build();
203    peerStorage.updatePeerConfig(peerId, newPeerConfig);
204    peers.put(peerId, new ReplicationPeerDescription(peerId, desc.isEnabled(), newPeerConfig));
205  }

207  public List<ReplicationPeerDescription> listPeers(Pattern pattern) {
208    if (pattern == null) {
209      return new ArrayList<>(peers.values());
210    }
211    return peers.values().stream().filter(r -> pattern.matcher(r.getPeerId()).matches())
212        .collect(Collectors.toList());
213  }
214
215  public Optional<ReplicationPeerConfig> getPeerConfig(String peerId) {
216    ReplicationPeerDescription desc = peers.get(peerId);
217    return desc != null ? Optional.of(desc.getPeerConfig()) : Optional.empty();
218  }

220  public void removeAllQueuesAndHFileRefs(String peerId) throws ReplicationException {
221    // Here we need two passes to address the problem of claimQueue. Maybe a claimQueue is still
222    // on-going when the refresh peer config procedure is done, if a RS which has already been
223    // scanned claims the queue of a RS which has not been scanned yet, we will miss that queue in
224    // the scan here, and if the RS who has claimed the queue crashed before creating recovered
225    // source, then the queue will leave there until the another RS detects the crash and helps
226    // removing the queue.
227    // A two pass scan can solve the problem. Anyway, the queue will not disappear during the
228    // claiming, it will either under the old RS or under the new RS, and a queue can only be
229    // claimed once after the refresh peer procedure done(as the next claim queue will just delete
230    // it), so we can make sure that a two pass scan will finally find the queue and remove it,
231    // unless it has already been removed by others.
232    ReplicationUtils.removeAllQueues(queueStorage, peerId);
233    ReplicationUtils.removeAllQueues(queueStorage, peerId);
234    queueStorage.removePeerFromHFileRefs(peerId);
235  }

237  private void checkPeerConfig(ReplicationPeerConfig peerConfig) throws DoNotRetryIOException {
238    checkClusterKey(peerConfig.getClusterKey());
239
240    if (peerConfig.replicateAllUserTables()) {
241      // If replicate_all flag is true, it means all user tables will be replicated to peer cluster.
242      // Then allow config exclude namespaces or exclude table-cfs which can't be replicated to peer
243      // cluster.
244      if ((peerConfig.getNamespaces() != null && !peerConfig.getNamespaces().isEmpty())
245          || (peerConfig.getTableCFsMap() != null && !peerConfig.getTableCFsMap().isEmpty())) {
246        throw new DoNotRetryIOException("Need clean namespaces or table-cfs config firstly "
247            + "when you want replicate all cluster");
248      }
249      checkNamespacesAndTableCfsConfigConflict(peerConfig.getExcludeNamespaces(),
250        peerConfig.getExcludeTableCFsMap());
251    } else {
252      // If replicate_all flag is false, it means all user tables can't be replicated to peer
253      // cluster. Then allow to config namespaces or table-cfs which will be replicated to peer
254      // cluster.
255      if ((peerConfig.getExcludeNamespaces() != null
256          && !peerConfig.getExcludeNamespaces().isEmpty())
257          || (peerConfig.getExcludeTableCFsMap() != null
258              && !peerConfig.getExcludeTableCFsMap().isEmpty())) {
259        throw new DoNotRetryIOException(
260            "Need clean exclude-namespaces or exclude-table-cfs config firstly"
261                + " when replicate_all flag is false");
262      }
263      checkNamespacesAndTableCfsConfigConflict(peerConfig.getNamespaces(),
264        peerConfig.getTableCFsMap());
265    }
266
267    checkConfiguredWALEntryFilters(peerConfig);
268  }
269
270  /**
271   * Set a namespace in the peer config means that all tables in this namespace will be replicated
272   * to the peer cluster.
273   * <ol>
274   * <li>If peer config already has a namespace, then not allow set any table of this namespace to
275   * the peer config.</li>
276   * <li>If peer config already has a table, then not allow set this table's namespace to the peer
277   * config.</li>
278   * </ol>
279   * <p>
280   * Set a exclude namespace in the peer config means that all tables in this namespace can't be
281   * replicated to the peer cluster.
282   * <ol>
283   * <li>If peer config already has a exclude namespace, then not allow set any exclude table of
284   * this namespace to the peer config.</li>
285   * <li>If peer config already has a exclude table, then not allow set this table's namespace as a
286   * exclude namespace.</li>
287   * </ol>
288   */
289  private void checkNamespacesAndTableCfsConfigConflict(Set<String> namespaces,
290      Map<TableName, ? extends Collection<String>> tableCfs) throws DoNotRetryIOException {
291    if (namespaces == null || namespaces.isEmpty()) {
292      return;
293    }
294    if (tableCfs == null || tableCfs.isEmpty()) {
295      return;
296    }
297    for (Map.Entry<TableName, ? extends Collection<String>> entry : tableCfs.entrySet()) {
298      TableName table = entry.getKey();
299      if (namespaces.contains(table.getNamespaceAsString())) {
300        throw new DoNotRetryIOException("Table-cfs " + table + " is conflict with namespaces " +
301          table.getNamespaceAsString() + " in peer config");
302      }
303    }
304  }
305
306  private void checkConfiguredWALEntryFilters(ReplicationPeerConfig peerConfig)
307      throws DoNotRetryIOException {
308    String filterCSV = peerConfig.getConfiguration()
309        .get(BaseReplicationEndpoint.REPLICATION_WALENTRYFILTER_CONFIG_KEY);
310    if (filterCSV != null && !filterCSV.isEmpty()) {
311      String[] filters = filterCSV.split(",");
312      for (String filter : filters) {
313        try {
314          Class.forName(filter).newInstance();
315        } catch (Exception e) {
316          throw new DoNotRetryIOException("Configured WALEntryFilter " + filter +
317            " could not be created. Failing add/update " + "peer operation.", e);
318        }
319      }
320    }
321  }
322
323  private void checkClusterKey(String clusterKey) throws DoNotRetryIOException {
324    try {
325      ZKConfig.validateClusterKey(clusterKey);
326    } catch (IOException e) {
327      throw new DoNotRetryIOException("Invalid cluster key: " + clusterKey, e);
328    }
329  }

331  public static ReplicationPeerManager create(ZKWatcher zk, Configuration conf)
332      throws ReplicationException {
333    ReplicationPeerStorage peerStorage =
334        ReplicationStorageFactory.getReplicationPeerStorage(zk, conf);
335    ConcurrentMap<String, ReplicationPeerDescription> peers = new ConcurrentHashMap<>();
336    for (String peerId : peerStorage.listPeerIds()) {
337      ReplicationPeerConfig peerConfig = peerStorage.getPeerConfig(peerId);
338      boolean enabled = peerStorage.isPeerEnabled(peerId);
339      peers.put(peerId, new ReplicationPeerDescription(peerId, enabled, peerConfig));
340    }
341    return new ReplicationPeerManager(peerStorage,
342        ReplicationStorageFactory.getReplicationQueueStorage(zk, conf), peers);
343  }
