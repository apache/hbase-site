071  private final ImmutableMap<SyncReplicationState, EnumSet<SyncReplicationState>>
072    allowedTransition = Maps.immutableEnumMap(ImmutableMap.of(SyncReplicationState.ACTIVE,
073      EnumSet.of(SyncReplicationState.DOWNGRADE_ACTIVE, SyncReplicationState.STANDBY),
074      SyncReplicationState.STANDBY, EnumSet.of(SyncReplicationState.DOWNGRADE_ACTIVE),
075      SyncReplicationState.DOWNGRADE_ACTIVE,
076      EnumSet.of(SyncReplicationState.STANDBY, SyncReplicationState.ACTIVE)));

085  private void checkQueuesDeleted(String peerId)
086      throws ReplicationException, DoNotRetryIOException {
087    for (ServerName replicator : queueStorage.getListOfReplicators()) {
088      List<String> queueIds = queueStorage.getAllQueues(replicator);
089      for (String queueId : queueIds) {
090        ReplicationQueueInfo queueInfo = new ReplicationQueueInfo(queueId);
091        if (queueInfo.getPeerId().equals(peerId)) {
092          throw new DoNotRetryIOException("undeleted queue for peerId: " + peerId +
093            ", replicator: " + replicator + ", queueId: " + queueId);
094        }
095      }
096    }
097    if (queueStorage.getAllPeersFromHFileRefsQueue().contains(peerId)) {
098      throw new DoNotRetryIOException("Undeleted queue for peer " + peerId + " in hfile-refs");
099    }
100  }

102  void preAddPeer(String peerId, ReplicationPeerConfig peerConfig)
103      throws DoNotRetryIOException, ReplicationException {
104    if (peerId.contains("-")) {
105      throw new DoNotRetryIOException("Found invalid peer name: " + peerId);
106    }
107    checkPeerConfig(peerConfig);
108    if (peers.containsKey(peerId)) {
109      throw new DoNotRetryIOException("Replication peer " + peerId + " already exists");
110    }
111    // make sure that there is no queues with the same peer id. This may happen when we create a
112    // peer with the same id with a old deleted peer. If the replication queues for the old peer
113    // have not been cleaned up yet then we should not create the new peer, otherwise the old wal
114    // file may also be replicated.
115    checkQueuesDeleted(peerId);
116  }

118  private ReplicationPeerDescription checkPeerExists(String peerId) throws DoNotRetryIOException {
119    ReplicationPeerDescription desc = peers.get(peerId);
120    if (desc == null) {
121      throw new DoNotRetryIOException("Replication peer " + peerId + " does not exist");
122    }
123    return desc;
124  }
125
126  private void checkPeerInDAStateIfSyncReplication(String peerId) throws DoNotRetryIOException {
127    ReplicationPeerDescription desc = peers.get(peerId);
128    if (desc != null && desc.getPeerConfig().isSyncReplication()
129        && !SyncReplicationState.DOWNGRADE_ACTIVE.equals(desc.getSyncReplicationState())) {
130      throw new DoNotRetryIOException("Couldn't remove synchronous replication peer with state="
131          + desc.getSyncReplicationState()
132          + ", Transit the synchronous replication state to be DOWNGRADE_ACTIVE firstly.");
133    }
134  }

136  ReplicationPeerConfig preRemovePeer(String peerId) throws DoNotRetryIOException {
137    ReplicationPeerDescription pd = checkPeerExists(peerId);
138    checkPeerInDAStateIfSyncReplication(peerId);
139    return pd.getPeerConfig();
140  }

142  void preEnablePeer(String peerId) throws DoNotRetryIOException {
143    ReplicationPeerDescription desc = checkPeerExists(peerId);
144    if (desc.isEnabled()) {
145      throw new DoNotRetryIOException("Replication peer " + peerId + " has already been enabled");
146    }

149  void preDisablePeer(String peerId) throws DoNotRetryIOException {
150    ReplicationPeerDescription desc = checkPeerExists(peerId);
151    if (!desc.isEnabled()) {
152      throw new DoNotRetryIOException("Replication peer " + peerId + " has already been disabled");
153    }

156  /**
157   * Return the old peer description. Can never be null.
158   */
159  ReplicationPeerDescription preUpdatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
160      throws DoNotRetryIOException {
161    checkPeerConfig(peerConfig);
162    ReplicationPeerDescription desc = checkPeerExists(peerId);
163    ReplicationPeerConfig oldPeerConfig = desc.getPeerConfig();
164    if (!isStringEquals(peerConfig.getClusterKey(), oldPeerConfig.getClusterKey())) {
165      throw new DoNotRetryIOException(
166        "Changing the cluster key on an existing peer is not allowed. Existing key '" +
167          oldPeerConfig.getClusterKey() + "' for peer " + peerId + " does not match new key '" +
168          peerConfig.getClusterKey() + "'");
169    }
170
171    if (!isStringEquals(peerConfig.getReplicationEndpointImpl(),
172      oldPeerConfig.getReplicationEndpointImpl())) {
173      throw new DoNotRetryIOException("Changing the replication endpoint implementation class " +
174        "on an existing peer is not allowed. Existing class '" +
175        oldPeerConfig.getReplicationEndpointImpl() + "' for peer " + peerId +
176        " does not match new class '" + peerConfig.getReplicationEndpointImpl() + "'");
177    }
178
179    if (!isStringEquals(peerConfig.getRemoteWALDir(), oldPeerConfig.getRemoteWALDir())) {
180      throw new DoNotRetryIOException(
181        "Changing the remote wal dir on an existing peer is not allowed. Existing remote wal " +
182          "dir '" + oldPeerConfig.getRemoteWALDir() + "' for peer " + peerId +
183          " does not match new remote wal dir '" + peerConfig.getRemoteWALDir() + "'");
184    }
185
186    if (oldPeerConfig.isSyncReplication()) {
187      if (!ReplicationUtils.isNamespacesAndTableCFsEqual(oldPeerConfig, peerConfig)) {
188        throw new DoNotRetryIOException(
189          "Changing the replicated namespace/table config on a synchronous replication " +
190            "peer(peerId: " + peerId + ") is not allowed.");
191      }
192    }
193    return desc;
194  }

196  /**
197   * @return the old desciption of the peer
198   */
199  ReplicationPeerDescription preTransitPeerSyncReplicationState(String peerId,
200      SyncReplicationState state) throws DoNotRetryIOException {
201    ReplicationPeerDescription desc = checkPeerExists(peerId);
202    SyncReplicationState fromState = desc.getSyncReplicationState();
203    EnumSet<SyncReplicationState> allowedToStates = allowedTransition.get(fromState);
204    if (allowedToStates == null || !allowedToStates.contains(state)) {
205      throw new DoNotRetryIOException("Can not transit current cluster state from " + fromState +
206        " to " + state + " for peer id=" + peerId);
207    }
208    return desc;
209  }

211  public void addPeer(String peerId, ReplicationPeerConfig peerConfig, boolean enabled)
212      throws ReplicationException {
213    if (peers.containsKey(peerId)) {
214      // this should be a retry, just return
215      return;
216    }
217    ReplicationPeerConfig copiedPeerConfig = ReplicationPeerConfig.newBuilder(peerConfig).build();
218    SyncReplicationState syncReplicationState =
219      copiedPeerConfig.isSyncReplication() ? SyncReplicationState.DOWNGRADE_ACTIVE
220        : SyncReplicationState.NONE;
221    peerStorage.addPeer(peerId, copiedPeerConfig, enabled, syncReplicationState);
222    peers.put(peerId,
223      new ReplicationPeerDescription(peerId, enabled, copiedPeerConfig, syncReplicationState));
224  }
225
226  public void removePeer(String peerId) throws ReplicationException {
227    if (!peers.containsKey(peerId)) {
228      // this should be a retry, just return
229      return;
230    }
231    peerStorage.removePeer(peerId);
232    peers.remove(peerId);
233  }

235  private void setPeerState(String peerId, boolean enabled) throws ReplicationException {
236    ReplicationPeerDescription desc = peers.get(peerId);
237    if (desc.isEnabled() == enabled) {
238      // this should be a retry, just return
239      return;
240    }
241    peerStorage.setPeerState(peerId, enabled);
242    peers.put(peerId, new ReplicationPeerDescription(peerId, enabled, desc.getPeerConfig(),
243      desc.getSyncReplicationState()));
244  }

246  public void enablePeer(String peerId) throws ReplicationException {
247    setPeerState(peerId, true);
248  }

251    setPeerState(peerId, false);

254  public void updatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
255      throws ReplicationException {
256    // the checking rules are too complicated here so we give up checking whether this is a retry.
257    ReplicationPeerDescription desc = peers.get(peerId);
258    ReplicationPeerConfig oldPeerConfig = desc.getPeerConfig();
259    ReplicationPeerConfigBuilder newPeerConfigBuilder =
260      ReplicationPeerConfig.newBuilder(peerConfig);
261    // we need to use the new conf to overwrite the old one.
262    newPeerConfigBuilder.putAllConfiguration(oldPeerConfig.getConfiguration());
263    newPeerConfigBuilder.putAllConfiguration(peerConfig.getConfiguration());
264    newPeerConfigBuilder.putAllConfiguration(oldPeerConfig.getConfiguration());
265    newPeerConfigBuilder.putAllConfiguration(peerConfig.getConfiguration());
266    ReplicationPeerConfig newPeerConfig = newPeerConfigBuilder.build();
267    peerStorage.updatePeerConfig(peerId, newPeerConfig);
268    peers.put(peerId, new ReplicationPeerDescription(peerId, desc.isEnabled(), newPeerConfig,
269      desc.getSyncReplicationState()));
270  }

272  public List<ReplicationPeerDescription> listPeers(Pattern pattern) {
273    if (pattern == null) {
274      return new ArrayList<>(peers.values());
275    }
276    return peers.values().stream().filter(r -> pattern.matcher(r.getPeerId()).matches())
277      .collect(Collectors.toList());

294  public void transitPeerSyncReplicationState(String peerId, SyncReplicationState newState)
295      throws ReplicationException {
296    if (peerStorage.getPeerNewSyncReplicationState(peerId) != SyncReplicationState.NONE) {
297      // Only transit if this is not a retry
298      peerStorage.transitPeerSyncReplicationState(peerId);
299    }
300    ReplicationPeerDescription desc = peers.get(peerId);
301    if (desc.getSyncReplicationState() != newState) {
302      // Only recreate the desc if this is not a retry
303      peers.put(peerId,
304        new ReplicationPeerDescription(peerId, desc.isEnabled(), desc.getPeerConfig(), newState));
305    }
306  }

308  public void removeAllQueues(String peerId) throws ReplicationException {
309    // Here we need two passes to address the problem of claimQueue. Maybe a claimQueue is still
310    // on-going when the refresh peer config procedure is done, if a RS which has already been
311    // scanned claims the queue of a RS which has not been scanned yet, we will miss that queue in
312    // the scan here, and if the RS who has claimed the queue crashed before creating recovered
313    // source, then the queue will leave there until the another RS detects the crash and helps
314    // removing the queue.
315    // A two pass scan can solve the problem. Anyway, the queue will not disappear during the
316    // claiming, it will either under the old RS or under the new RS, and a queue can only be
317    // claimed once after the refresh peer procedure done(as the next claim queue will just delete
318    // it), so we can make sure that a two pass scan will finally find the queue and remove it,
319    // unless it has already been removed by others.
320    ReplicationUtils.removeAllQueues(queueStorage, peerId);
321    ReplicationUtils.removeAllQueues(queueStorage, peerId);
322  }

326    queueStorage.removePeerFromHFileRefs(peerId);

329  private void checkPeerConfig(ReplicationPeerConfig peerConfig) throws DoNotRetryIOException {
330    checkClusterKey(peerConfig.getClusterKey());
331
332    if (peerConfig.replicateAllUserTables()) {
333      // If replicate_all flag is true, it means all user tables will be replicated to peer cluster.
334      // Then allow config exclude namespaces or exclude table-cfs which can't be replicated to peer
335      // cluster.
336      if ((peerConfig.getNamespaces() != null && !peerConfig.getNamespaces().isEmpty()) ||
337        (peerConfig.getTableCFsMap() != null && !peerConfig.getTableCFsMap().isEmpty())) {
338        throw new DoNotRetryIOException("Need clean namespaces or table-cfs config firstly " +
339          "when you want replicate all cluster");
340      }
341      checkNamespacesAndTableCfsConfigConflict(peerConfig.getExcludeNamespaces(),
342        peerConfig.getExcludeTableCFsMap());
343    } else {
344      // If replicate_all flag is false, it means all user tables can't be replicated to peer
345      // cluster. Then allow to config namespaces or table-cfs which will be replicated to peer
346      // cluster.
347      if ((peerConfig.getExcludeNamespaces() != null &&
348        !peerConfig.getExcludeNamespaces().isEmpty()) ||
349        (peerConfig.getExcludeTableCFsMap() != null &&
350          !peerConfig.getExcludeTableCFsMap().isEmpty())) {
351        throw new DoNotRetryIOException(
352          "Need clean exclude-namespaces or exclude-table-cfs config firstly" +
353            " when replicate_all flag is false");
354      }
355      checkNamespacesAndTableCfsConfigConflict(peerConfig.getNamespaces(),
356        peerConfig.getTableCFsMap());
357    }
358
359    if (peerConfig.isSyncReplication()) {
360      checkPeerConfigForSyncReplication(peerConfig);
361    }
362
363    checkConfiguredWALEntryFilters(peerConfig);
364  }
365
366  private void checkPeerConfigForSyncReplication(ReplicationPeerConfig peerConfig)
367      throws DoNotRetryIOException {
368    // This is used to reduce the difficulty for implementing the sync replication state transition
369    // as we need to reopen all the related regions.
370    // TODO: Add namespace, replicat_all flag back
371    if (peerConfig.replicateAllUserTables()) {
372      throw new DoNotRetryIOException(
373        "Only support replicated table config for sync replication peer");
374    }
375    if (peerConfig.getNamespaces() != null && !peerConfig.getNamespaces().isEmpty()) {
376      throw new DoNotRetryIOException(
377        "Only support replicated table config for sync replication peer");
378    }
379    if (peerConfig.getTableCFsMap() == null || peerConfig.getTableCFsMap().isEmpty()) {
380      throw new DoNotRetryIOException("Need config replicated tables for sync replication peer");
381    }
382    for (List<String> cfs : peerConfig.getTableCFsMap().values()) {
383      if (cfs != null && !cfs.isEmpty()) {
384        throw new DoNotRetryIOException(
385          "Only support replicated table config for sync replication peer");
386      }
387    }
388    Path remoteWALDir = new Path(peerConfig.getRemoteWALDir());
389    if (!remoteWALDir.isAbsolute()) {
390      throw new DoNotRetryIOException(
391        "The remote WAL directory " + peerConfig.getRemoteWALDir() + " is not absolute");
392    }
393    URI remoteWALDirUri = remoteWALDir.toUri();
394    if (remoteWALDirUri.getScheme() == null || remoteWALDirUri.getAuthority() == null) {
395      throw new DoNotRetryIOException("The remote WAL directory " + peerConfig.getRemoteWALDir() +
396        " is not qualified, you must provide scheme and authority");
397    }
398  }

400  /**
401   * Set a namespace in the peer config means that all tables in this namespace will be replicated
402   * to the peer cluster.
403   * <ol>
404   * <li>If peer config already has a namespace, then not allow set any table of this namespace to
405   * the peer config.</li>
406   * <li>If peer config already has a table, then not allow set this table's namespace to the peer
407   * config.</li>
408   * </ol>
409   * <p>
410   * Set a exclude namespace in the peer config means that all tables in this namespace can't be
411   * replicated to the peer cluster.
412   * <ol>
413   * <li>If peer config already has a exclude namespace, then not allow set any exclude table of
414   * this namespace to the peer config.</li>
415   * <li>If peer config already has a exclude table, then not allow set this table's namespace as a
416   * exclude namespace.</li>
417   * </ol>
418   */
419  private void checkNamespacesAndTableCfsConfigConflict(Set<String> namespaces,
420      Map<TableName, ? extends Collection<String>> tableCfs) throws DoNotRetryIOException {
421    if (namespaces == null || namespaces.isEmpty()) {
422      return;
423    }
424    if (tableCfs == null || tableCfs.isEmpty()) {
425      return;
426    }
427    for (Map.Entry<TableName, ? extends Collection<String>> entry : tableCfs.entrySet()) {
428      TableName table = entry.getKey();
429      if (namespaces.contains(table.getNamespaceAsString())) {
430        throw new DoNotRetryIOException("Table-cfs " + table + " is conflict with namespaces " +
431          table.getNamespaceAsString() + " in peer config");
432      }
433    }
434  }

436  private void checkConfiguredWALEntryFilters(ReplicationPeerConfig peerConfig)
437      throws DoNotRetryIOException {
438    String filterCSV = peerConfig.getConfiguration()
439      .get(BaseReplicationEndpoint.REPLICATION_WALENTRYFILTER_CONFIG_KEY);
440    if (filterCSV != null && !filterCSV.isEmpty()) {
441      String[] filters = filterCSV.split(",");
442      for (String filter : filters) {
443        try {
444          Class.forName(filter).getDeclaredConstructor().newInstance();
445        } catch (Exception e) {
446          throw new DoNotRetryIOException("Configured WALEntryFilter " + filter +
447            " could not be created. Failing add/update peer operation.", e);
448        }
449      }
450    }
451  }

471  public static ReplicationPeerManager create(ZKWatcher zk, Configuration conf)
472      throws ReplicationException {
473    ReplicationPeerStorage peerStorage =
474      ReplicationStorageFactory.getReplicationPeerStorage(zk, conf);
475    ConcurrentMap<String, ReplicationPeerDescription> peers = new ConcurrentHashMap<>();
476    for (String peerId : peerStorage.listPeerIds()) {
477      ReplicationPeerConfig peerConfig = peerStorage.getPeerConfig(peerId);
478      boolean enabled = peerStorage.isPeerEnabled(peerId);
479      SyncReplicationState state = peerStorage.getPeerSyncReplicationState(peerId);
480      peers.put(peerId, new ReplicationPeerDescription(peerId, enabled, peerConfig, state));
481    }
482    return new ReplicationPeerManager(peerStorage,
483      ReplicationStorageFactory.getReplicationQueueStorage(zk, conf), peers);
484  }

486  /**
487   * For replication peer cluster key or endpoint class, null and empty string is same. So here
488   * don't use {@link StringUtils#equals(CharSequence, CharSequence)} directly.
489   */
490  private boolean isStringEquals(String s1, String s2) {
491    if (StringUtils.isBlank(s1)) {
492      return StringUtils.isBlank(s2);
493    }
494    return s1.equals(s2);
495  }
