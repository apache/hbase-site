018package org.apache.hadoop.hbase.master.replication;
019
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.List;
023
024import org.apache.hadoop.hbase.ServerName;
025import org.apache.hadoop.hbase.master.procedure.MasterProcedureEnv;
026import org.apache.hadoop.hbase.master.procedure.PeerProcedureInterface;
027import org.apache.hadoop.hbase.master.procedure.RSProcedureDispatcher.ServerOperation;
028import org.apache.hadoop.hbase.procedure2.FailedRemoteDispatchException;
029import org.apache.hadoop.hbase.procedure2.Procedure;
030import org.apache.hadoop.hbase.procedure2.ProcedureEvent;
031import org.apache.hadoop.hbase.procedure2.ProcedureStateSerializer;
032import org.apache.hadoop.hbase.procedure2.ProcedureSuspendedException;
033import org.apache.hadoop.hbase.procedure2.ProcedureYieldException;
034import org.apache.hadoop.hbase.procedure2.RemoteProcedureDispatcher.RemoteOperation;
035import org.apache.hadoop.hbase.procedure2.RemoteProcedureDispatcher.RemoteProcedure;
036import org.apache.hadoop.hbase.procedure2.RemoteProcedureException;
037import org.apache.hadoop.hbase.replication.regionserver.ReplaySyncReplicationWALCallable;
038import org.apache.yetus.audience.InterfaceAudience;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041
042import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
043import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.ReplaySyncReplicationWALParameter;
044import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.SyncReplicationReplayWALRemoteStateData;
045
046@InterfaceAudience.Private
047public class SyncReplicationReplayWALRemoteProcedure extends Procedure<MasterProcedureEnv>
048    implements RemoteProcedure<MasterProcedureEnv, ServerName>, PeerProcedureInterface {
049
050  private static final Logger LOG =
051      LoggerFactory.getLogger(SyncReplicationReplayWALRemoteProcedure.class);
052
053  private String peerId;
054
055  private ServerName targetServer;
056
057  private List<String> wals;
058
059  private boolean dispatched;
060
061  private ProcedureEvent<?> event;
062
063  private boolean succ;
064
065  public SyncReplicationReplayWALRemoteProcedure() {
066  }
067
068  public SyncReplicationReplayWALRemoteProcedure(String peerId, List<String> wals,
069      ServerName targetServer) {
070    this.peerId = peerId;
071    this.wals = wals;
072    this.targetServer = targetServer;
073  }
074
075  @Override
076  public RemoteOperation remoteCallBuild(MasterProcedureEnv env, ServerName remote) {
077    ReplaySyncReplicationWALParameter.Builder builder =
078        ReplaySyncReplicationWALParameter.newBuilder();
079    builder.setPeerId(peerId);
080    wals.stream().forEach(builder::addWal);
081    return new ServerOperation(this, getProcId(), ReplaySyncReplicationWALCallable.class,
082        builder.build().toByteArray());
083  }
084
085  @Override
086  public void remoteCallFailed(MasterProcedureEnv env, ServerName remote, IOException exception) {
087    complete(env, exception);
088  }
089
090  @Override
091  public void remoteOperationCompleted(MasterProcedureEnv env) {
092    complete(env, null);
093  }
094
095  @Override
096  public void remoteOperationFailed(MasterProcedureEnv env, RemoteProcedureException error) {
097    complete(env, error);
098  }
099
100  private void complete(MasterProcedureEnv env, Throwable error) {
101    if (event == null) {
102      LOG.warn("procedure event for {} is null, maybe the procedure is created when recovery",
103        getProcId());
104      return;
105    }
106    if (error != null) {
107      LOG.warn("Replay wals {} on {} failed for peer id={}", wals, targetServer, peerId, error);
108      this.succ = false;
109    } else {
110      truncateWALs(env);
111      LOG.info("Replay wals {} on {} succeed for peer id={}", wals, targetServer, peerId);
112      this.succ = true;
113    }
114    event.wake(env.getProcedureScheduler());
115    event = null;
116  }
117
118  /**
119   * Only truncate wals one by one when task succeed. The parent procedure will check the first
120   * wal length to know whether this task succeed.
121   */
122  private void truncateWALs(MasterProcedureEnv env) {
123    String firstWal = wals.get(0);
124    try {
125      env.getMasterServices().getSyncReplicationReplayWALManager().finishReplayWAL(firstWal);
126    } catch (IOException e) {
127      // As it is idempotent to rerun this task. Just ignore this exception and return.
128      LOG.warn("Failed to truncate wal {} for peer id={}", firstWal, peerId, e);
129      return;
130    }
131    for (int i = 1; i < wals.size(); i++) {
132      String wal = wals.get(i);
133      try {
134        env.getMasterServices().getSyncReplicationReplayWALManager().finishReplayWAL(wal);
135      } catch (IOException e1) {
136        try {
137          // retry
138          env.getMasterServices().getSyncReplicationReplayWALManager().finishReplayWAL(wal);
139        } catch (IOException e2) {
140          // As the parent procedure only check the first wal length. Just ignore this exception.
141          LOG.warn("Failed to truncate wal {} for peer id={}", wal, peerId, e2);
142        }
143      }
144    }
145  }
146
147  @Override
148  protected Procedure<MasterProcedureEnv>[] execute(MasterProcedureEnv env)
149      throws ProcedureYieldException, ProcedureSuspendedException, InterruptedException {
150    if (dispatched) {
151      if (succ) {
152        return null;
153      }
154      // retry
155      dispatched = false;
156    }
157
158    // Dispatch task to target server
159    try {
160      env.getRemoteDispatcher().addOperationToNode(targetServer, this);
161    } catch (FailedRemoteDispatchException e) {
162      LOG.warn(
163          "Can not add remote operation for replay wals {} on {} for peer id={}, "
164              + "this usually because the server is already dead",
165          wals, targetServer, peerId);
166      // Return directly and the parent procedure will assign a new worker to replay wals
167      return null;
168    }
169    dispatched = true;
170    event = new ProcedureEvent<>(this);
171    event.suspendIfNotReady(this);
172    throw new ProcedureSuspendedException();
173  }
174
175  @Override
176  protected void rollback(MasterProcedureEnv env) throws IOException, InterruptedException {
177    throw new UnsupportedOperationException();
178  }
179
180  @Override
181  protected boolean abort(MasterProcedureEnv env) {
182    return false;
183  }
184
185  @Override
186  protected void serializeStateData(ProcedureStateSerializer serializer)
187      throws IOException {
188    SyncReplicationReplayWALRemoteStateData.Builder builder =
189        SyncReplicationReplayWALRemoteStateData.newBuilder().setPeerId(peerId)
190            .setTargetServer(ProtobufUtil.toServerName(targetServer));
191    wals.stream().forEach(builder::addWal);
192    serializer.serialize(builder.build());
193  }
194
195  @Override
196  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
197    SyncReplicationReplayWALRemoteStateData data =
198        serializer.deserialize(SyncReplicationReplayWALRemoteStateData.class);
199    peerId = data.getPeerId();
200    wals = new ArrayList<>();
201    data.getWalList().forEach(wals::add);
202    targetServer = ProtobufUtil.toServerName(data.getTargetServer());
203  }
204
205  @Override
206  public String getPeerId() {
207    return peerId;
208  }
209
210  @Override
211  public PeerOperationType getPeerOperationType() {
212    return PeerOperationType.SYNC_REPLICATION_REPLAY_WAL_REMOTE;
213  }
214}
