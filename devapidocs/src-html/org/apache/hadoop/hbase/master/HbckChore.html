045/**
046 * Used to do the hbck checking job at master side.
047 */
048@InterfaceAudience.Private
049@InterfaceStability.Evolving
050public class HbckChore extends ScheduledChore {
051  private static final Logger LOG = LoggerFactory.getLogger(HbckChore.class.getName());
052
053  private static final String HBCK_CHORE_INTERVAL = "hbase.master.hbck.chore.interval";
054  private static final int DEFAULT_HBCK_CHORE_INTERVAL = 60 * 60 * 1000;
055
056  private final MasterServices master;
057
058  /**
059   * This map contains the state of all hbck items.  It maps from encoded region
060   * name to HbckRegionInfo structure.  The information contained in HbckRegionInfo is used
061   * to detect and correct consistency (hdfs/meta/deployment) problems.
062   */
063  private final Map<String, HbckRegionInfo> regionInfoMap = new HashMap<>();
064
065  private final Set<String> disabledTableRegions = new HashSet<>();
066  private final Set<String> splitParentRegions = new HashSet<>();
067
068  /**
069   * The regions only opened on RegionServers, but no region info in meta.
070   */
071  private final Map<String, ServerName> orphanRegionsOnRS = new HashMap<>();
072  /**
073   * The regions have directory on FileSystem, but no region info in meta.
074   */
075  private final Map<String, Path> orphanRegionsOnFS = new HashMap<>();
076  /**
077   * The inconsistent regions. There are three case:
078   * case 1. Master thought this region opened, but no regionserver reported it.
079   * case 2. Master thought this region opened on Server1, but regionserver reported Server2
080   * case 3. More than one regionservers reported opened this region
081   */
082  private final Map<String, Pair<ServerName, List<ServerName>>> inconsistentRegions =
083      new HashMap<>();
084
085  /**
086   * The "snapshot" is used to save the last round's HBCK checking report.
087   */
088  private final Map<String, ServerName> orphanRegionsOnRSSnapshot = new HashMap<>();
089  private final Map<String, Path> orphanRegionsOnFSSnapshot = new HashMap<>();
090  private final Map<String, Pair<ServerName, List<ServerName>>> inconsistentRegionsSnapshot =
091      new HashMap<>();
092
093  /**
094   * The "snapshot" may be changed after checking. And this checking report "snapshot" may be
095   * accessed by web ui. Use this rwLock to synchronize.
096   */
097  ReentrantReadWriteLock rwLock = new ReentrantReadWriteLock();
098
099  /**
100   * When running, the "snapshot" may be changed when this round's checking finish.
101   */
102  private volatile boolean running = false;
103  private volatile long checkingStartTimestamp = 0;
104  private volatile long checkingEndTimestamp = 0;
105
106  private boolean disabled = false;
107
108  public HbckChore(MasterServices master) {
109    super("HbckChore-", master,
110        master.getConfiguration().getInt(HBCK_CHORE_INTERVAL, DEFAULT_HBCK_CHORE_INTERVAL));
111    this.master = master;
112    int interval =
113        master.getConfiguration().getInt(HBCK_CHORE_INTERVAL, DEFAULT_HBCK_CHORE_INTERVAL);
114    if (interval <= 0) {
115      LOG.warn(HBCK_CHORE_INTERVAL + " is <=0 hence disabling hbck chore");
116      disableChore();
117    }
118  }
119
120  @Override
121  protected synchronized void chore() {
122    if (isDisabled() || isRunning()) {
123      LOG.warn("hbckChore is either disabled or is already running. Can't run the chore");
124      return;
125    }
126    regionInfoMap.clear();
127    disabledTableRegions.clear();
128    splitParentRegions.clear();
129    orphanRegionsOnRS.clear();
130    orphanRegionsOnFS.clear();
131    inconsistentRegions.clear();
132    checkingStartTimestamp = EnvironmentEdgeManager.currentTime();
133    running = true;
134    try {
135      loadRegionsFromInMemoryState();
136      loadRegionsFromRSReport();
137      try {
138        loadRegionsFromFS(scanForMergedParentRegions());
139      } catch (IOException e) {
140        LOG.warn("Failed to load the regions from filesystem", e);
141      }
142      saveCheckResultToSnapshot();
143    } catch (Throwable t) {
144      LOG.warn("Unexpected", t);
145    }
146    running = false;
147    updateAssignmentManagerMetrics();
148  }
149
150  // This function does the sanity checks of making sure the chore is not run when it is
151  // disabled or when it's already running. It returns whether the chore was actually run or not.
152  protected boolean runChore() {
153    if (isDisabled() || isRunning()) {
154      if (isDisabled()) {
155        LOG.warn("hbck chore is disabled! Set " + HBCK_CHORE_INTERVAL + " > 0 to enable it.");
156      } else {
157        LOG.warn("hbck chore already running. Can't run till it finishes.");
158      }
159      return false;
160    }
161    chore();
162    return true;
163  }
164
165  private void disableChore() {
166    this.disabled = true;
167  }
168
169  public boolean isDisabled() {
170    return this.disabled;
171  }
172
173  private void saveCheckResultToSnapshot() {
174    // Need synchronized here, as this "snapshot" may be access by web ui.
175    rwLock.writeLock().lock();
176    try {
177      orphanRegionsOnRSSnapshot.clear();
178      orphanRegionsOnRS.entrySet()
179          .forEach(e -> orphanRegionsOnRSSnapshot.put(e.getKey(), e.getValue()));
180      orphanRegionsOnFSSnapshot.clear();
181      orphanRegionsOnFS.entrySet()
182          .forEach(e -> orphanRegionsOnFSSnapshot.put(e.getKey(), e.getValue()));
183      inconsistentRegionsSnapshot.clear();
184      inconsistentRegions.entrySet()
185          .forEach(e -> inconsistentRegionsSnapshot.put(e.getKey(), e.getValue()));
186      checkingEndTimestamp = EnvironmentEdgeManager.currentTime();
187    } finally {
188      rwLock.writeLock().unlock();
189    }
190  }
191
192  /**
193   * Scan hbase:meta to get set of merged parent regions, this is a very heavy scan.
194   *
195   * @return Return generated {@link HashSet}
196   */
197  private HashSet<String> scanForMergedParentRegions() throws IOException {
198    HashSet<String> mergedParentRegions = new HashSet<>();
199    // Null tablename means scan all of meta.
200    MetaTableAccessor.scanMetaForTableRegions(this.master.getConnection(),
201      r -> {
202        List<RegionInfo> mergeParents = CatalogFamilyFormat.getMergeRegions(r.rawCells());
203        if (mergeParents != null) {
204          for (RegionInfo mergeRegion : mergeParents) {
205            if (mergeRegion != null) {
206              // This region is already being merged
207              mergedParentRegions.add(mergeRegion.getEncodedName());
208            }
209          }
210        }
211        return true;
212        },
213      null);
214    return mergedParentRegions;
215  }
216
217  private void loadRegionsFromInMemoryState() {
218    List<RegionState> regionStates =
219        master.getAssignmentManager().getRegionStates().getRegionStates();
220    for (RegionState regionState : regionStates) {
221      RegionInfo regionInfo = regionState.getRegion();
222      if (master.getTableStateManager()
223          .isTableState(regionInfo.getTable(), TableState.State.DISABLED)) {
224        disabledTableRegions.add(regionInfo.getRegionNameAsString());
225      }
226      // Check both state and regioninfo for split status, see HBASE-26383
227      if (regionState.isSplit() || regionInfo.isSplit()) {
228        splitParentRegions.add(regionInfo.getRegionNameAsString());
229      }
230      HbckRegionInfo.MetaEntry metaEntry =
231          new HbckRegionInfo.MetaEntry(regionInfo, regionState.getServerName(),
232              regionState.getStamp());
233      regionInfoMap.put(regionInfo.getEncodedName(), new HbckRegionInfo(metaEntry));
234    }
235    LOG.info("Loaded {} regions ({} disabled, {} split parents) from in-memory state",
236      regionStates.size(), disabledTableRegions.size(), splitParentRegions.size());
237    if (LOG.isDebugEnabled()) {
238      Map<RegionState.State,Integer> stateCountMap = new HashMap<>();
239      for (RegionState regionState : regionStates) {
240        stateCountMap.compute(regionState.getState(), (k, v) -> (v == null) ? 1 : v + 1);
241      }
242      StringBuffer sb = new StringBuffer();
243      sb.append("Regions by state: ");
244      stateCountMap.entrySet().forEach(e -> {
245          sb.append(e.getKey());
246          sb.append('=');
247          sb.append(e.getValue());
248          sb.append(' ');
249        }
250      );
251      LOG.debug(sb.toString());
252    }
253    if (LOG.isTraceEnabled()) {
254      for (RegionState regionState : regionStates) {
255        LOG.trace("{}: {}, serverName=", regionState.getRegion(), regionState.getState(),
256          regionState.getServerName());
257      }
258    }
259  }
260
261  private void loadRegionsFromRSReport() {
262    int numRegions = 0;
263    Map<ServerName, Set<byte[]>> rsReports = master.getAssignmentManager().getRSReports();
264    for (Map.Entry<ServerName, Set<byte[]>> entry : rsReports.entrySet()) {
265      ServerName serverName = entry.getKey();
266      for (byte[] regionName : entry.getValue()) {
267        String encodedRegionName = RegionInfo.encodeRegionName(regionName);
268        HbckRegionInfo hri = regionInfoMap.get(encodedRegionName);
269        if (hri == null) {
270          orphanRegionsOnRS.put(RegionInfo.getRegionNameAsString(regionName), serverName);
271          continue;
272        }
273        hri.addServer(hri.getMetaEntry().getRegionInfo(), serverName);
274      }
275      numRegions += entry.getValue().size();
276    }
277    LOG.info("Loaded {} regions from {} regionservers' reports and found {} orphan regions",
278        numRegions, rsReports.size(), orphanRegionsOnRS.size());
279
280    for (Map.Entry<String, HbckRegionInfo> entry : regionInfoMap.entrySet()) {
281      HbckRegionInfo hri = entry.getValue();
282      ServerName locationInMeta = hri.getMetaEntry().getRegionServer();
283      if (locationInMeta == null) {
284        continue;
285      }
286      if (hri.getDeployedOn().size() == 0) {
287        // skip the offline region which belong to disabled table.
288        if (disabledTableRegions.contains(hri.getRegionNameAsString())) {
289          continue;
290        }
291        // skip the split parent regions
292        if (splitParentRegions.contains(hri.getRegionNameAsString())) {
293          continue;
294        }
295        // Master thought this region opened, but no regionserver reported it.
296        inconsistentRegions.put(hri.getRegionNameAsString(),
297            new Pair<>(locationInMeta, new LinkedList<>()));
298      } else if (hri.getDeployedOn().size() > 1) {
299        // More than one regionserver reported opened this region
300        inconsistentRegions.put(hri.getRegionNameAsString(),
301            new Pair<>(locationInMeta, hri.getDeployedOn()));
302      } else if (!hri.getDeployedOn().get(0).equals(locationInMeta)) {
303        // Master thought this region opened on Server1, but regionserver reported Server2
304        inconsistentRegions.put(hri.getRegionNameAsString(),
305            new Pair<>(locationInMeta, hri.getDeployedOn()));
306      }
307    }
308  }

310  private void loadRegionsFromFS(final HashSet<String> mergedParentRegions) throws IOException {
311    Path rootDir = master.getMasterFileSystem().getRootDir();
312    FileSystem fs = master.getMasterFileSystem().getFileSystem();
313
314    int numRegions = 0;
315    List<Path> tableDirs = FSUtils.getTableDirs(fs, rootDir);
316    for (Path tableDir : tableDirs) {
317      List<Path> regionDirs = FSUtils.getRegionDirs(fs, tableDir);
318      for (Path regionDir : regionDirs) {
319        String encodedRegionName = regionDir.getName();
320        if (encodedRegionName == null) {
321          LOG.warn("Failed get of encoded name from {}", regionDir);
322          continue;
323        }
324        HbckRegionInfo hri = regionInfoMap.get(encodedRegionName);
325        // If it is not in in-memory database and not a merged region,
326        // report it as an orphan region.
327        if (hri == null && !mergedParentRegions.contains(encodedRegionName)) {
328          orphanRegionsOnFS.put(encodedRegionName, regionDir);
329          continue;
330        }
331      }
332      numRegions += regionDirs.size();
333    }
334    LOG.info("Loaded {} tables {} regions from filesystem and found {} orphan regions",
335        tableDirs.size(), numRegions, orphanRegionsOnFS.size());
336  }
337
338  private void updateAssignmentManagerMetrics() {
339    master.getAssignmentManager().getAssignmentManagerMetrics()
340        .updateOrphanRegionsOnRs(getOrphanRegionsOnRS().size());
341    master.getAssignmentManager().getAssignmentManagerMetrics()
342        .updateOrphanRegionsOnFs(getOrphanRegionsOnFS().size());
343    master.getAssignmentManager().getAssignmentManagerMetrics()
344        .updateInconsistentRegions(getInconsistentRegions().size());
345  }

347  /**
348   * When running, the HBCK report may be changed later.
349   */
350  public boolean isRunning() {
351    return running;
352  }

354  /**
355   * @return the regions only opened on RegionServers, but no region info in meta.
356   */
357  public Map<String, ServerName> getOrphanRegionsOnRS() {
358    // Need synchronized here, as this "snapshot" may be changed after checking.
359    rwLock.readLock().lock();
360    try {
361      return this.orphanRegionsOnRSSnapshot;
362    } finally {
363      rwLock.readLock().unlock();
364    }
365  }
366
367  /**
368   * @return the regions have directory on FileSystem, but no region info in meta.
369   */
370  public Map<String, Path> getOrphanRegionsOnFS() {
371    // Need synchronized here, as this "snapshot" may be changed after checking.
372    rwLock.readLock().lock();
373    try {
374      return this.orphanRegionsOnFSSnapshot;
375    } finally {
376      rwLock.readLock().unlock();
377    }
378  }

380  /**
381   * Found the inconsistent regions. There are three case:
382   * case 1. Master thought this region opened, but no regionserver reported it.
383   * case 2. Master thought this region opened on Server1, but regionserver reported Server2
384   * case 3. More than one regionservers reported opened this region
385   *
386   * @return the map of inconsistent regions. Key is the region name. Value is a pair of location in
387   *         meta and the regionservers which reported opened this region.
388   */
389  public Map<String, Pair<ServerName, List<ServerName>>> getInconsistentRegions() {
390    // Need synchronized here, as this "snapshot" may be changed after checking.
391    rwLock.readLock().lock();
392    try {
393      return this.inconsistentRegionsSnapshot;
394    } finally {
395      rwLock.readLock().unlock();
396    }
397  }
