019
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.HashMap;
023import java.util.HashSet;
024import java.util.List;
025import java.util.Map;
026import java.util.Set;
027import java.util.Map.Entry;
028
029import org.apache.commons.logging.Log;
030import org.apache.commons.logging.LogFactory;
031import org.apache.hadoop.conf.Configuration;
032import org.apache.hadoop.hbase.Abortable;
033import org.apache.hadoop.hbase.classification.InterfaceAudience;
034import org.apache.hadoop.hbase.replication.ReplicationFactory;
035import org.apache.hadoop.hbase.replication.ReplicationPeers;
036import org.apache.hadoop.hbase.replication.ReplicationQueueInfo;
037import org.apache.hadoop.hbase.replication.ReplicationQueuesClient;
038import org.apache.hadoop.hbase.replication.ReplicationQueuesClientArguments;
039import org.apache.hadoop.hbase.replication.ReplicationStateZKBase;
040import org.apache.hadoop.hbase.zookeeper.ZKUtil;
041import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
042import org.apache.zookeeper.KeeperException;
043
044/**
045 * Used to clean the replication queues belonging to the peer which does not exist.
046 */
047@InterfaceAudience.Private
048public class ReplicationZKNodeCleaner {
049  private static final Log LOG = LogFactory.getLog(ReplicationZKNodeCleaner.class);
050  private final ZooKeeperWatcher zkw;
051  private final ReplicationQueuesClient queuesClient;
052  private final ReplicationPeers replicationPeers;
053  private final ReplicationQueueDeletor queueDeletor;
054
055  public ReplicationZKNodeCleaner(Configuration conf, ZooKeeperWatcher zkw, Abortable abortable)
056      throws IOException {
057    try {
058      this.zkw = zkw;
059      this.queuesClient = ReplicationFactory
060          .getReplicationQueuesClient(new ReplicationQueuesClientArguments(conf, abortable, zkw));
061      this.queuesClient.init();
062      this.replicationPeers = ReplicationFactory.getReplicationPeers(zkw, conf, this.queuesClient,
063        abortable);
064      this.replicationPeers.init();
065      this.queueDeletor = new ReplicationQueueDeletor(zkw, conf, abortable);
066    } catch (Exception e) {
067      throw new IOException("failed to construct ReplicationZKNodeCleaner", e);
068    }
069  }
070
071  /**
072   * @return undeletedQueues replicator with its queueIds for removed peers
073   * @throws IOException
074   */
075  public Map<String, List<String>> getUnDeletedQueues() throws IOException {
076    Map<String, List<String>> undeletedQueues = new HashMap<>();
077    Set<String> peerIds = new HashSet<>(this.replicationPeers.getAllPeerIds());
078    try {
079      List<String> replicators = this.queuesClient.getListOfReplicators();
080      if (replicators == null || replicators.isEmpty()) {
081        return undeletedQueues;
082      }
083      for (String replicator : replicators) {
084        List<String> queueIds = this.queuesClient.getAllQueues(replicator);
085        for (String queueId : queueIds) {
086          ReplicationQueueInfo queueInfo = new ReplicationQueueInfo(queueId);
087          if (!peerIds.contains(queueInfo.getPeerId())) {
088            undeletedQueues.computeIfAbsent(replicator, (key) -> new ArrayList<>()).add(
089              queueId);
090            if (LOG.isDebugEnabled()) {
091              LOG.debug("Undeleted replication queue for removed peer found: "
092                  + String.format("[removedPeerId=%s, replicator=%s, queueId=%s]",
093                    queueInfo.getPeerId(), replicator, queueId));
094            }
095          }
096        }
097      }
098    } catch (KeeperException ke) {
099      throw new IOException("Failed to get the replication queues of all replicators", ke);
100    }
101    return undeletedQueues;
102  }
103
104  /**
105   * @return undeletedHFileRefsQueue replicator with its undeleted queueIds for removed peers in
106   *         hfile-refs queue
107   * @throws IOException
108   */
109  public Set<String> getUnDeletedHFileRefsQueues() throws IOException {
110    Set<String> undeletedHFileRefsQueue = new HashSet<>();
111    Set<String> peerIds = new HashSet<>(this.replicationPeers.getAllPeerIds());
112    String hfileRefsZNode = queueDeletor.getHfileRefsZNode();
113    try {
114      if (-1 == ZKUtil.checkExists(zkw, hfileRefsZNode)) {
115        return null;
116      }
117      List<String> listOfPeers = this.queuesClient.getAllPeersFromHFileRefsQueue();
118      Set<String> peers = new HashSet<>(listOfPeers);
119      peers.removeAll(peerIds);
120      if (!peers.isEmpty()) {
121        undeletedHFileRefsQueue.addAll(peers);
122      }
123    } catch (KeeperException e) {
124      throw new IOException("Failed to get list of all peers from hfile-refs znode "
125          + hfileRefsZNode, e);
126    }
127    return undeletedHFileRefsQueue;
128  }
129
130  private class ReplicationQueueDeletor extends ReplicationStateZKBase {
131
132    public ReplicationQueueDeletor(ZooKeeperWatcher zk, Configuration conf, Abortable abortable) {
133      super(zk, conf, abortable);
134    }
135
136    /**
137     * @param replicator The regionserver which has undeleted queue
138     * @param queueId The undeleted queue id
139     * @throws IOException
140     */
141    public void removeQueue(final String replicator, final String queueId) throws IOException {
142      String queueZnodePath = ZKUtil.joinZNode(ZKUtil.joinZNode(this.queuesZNode, replicator),
143        queueId);
144      try {
145        ReplicationQueueInfo queueInfo = new ReplicationQueueInfo(queueId);
146        if (!replicationPeers.getAllPeerIds().contains(queueInfo.getPeerId())) {
147          ZKUtil.deleteNodeRecursively(this.zookeeper, queueZnodePath);
148          LOG.info("Successfully removed replication queue, replicator: " + replicator
149              + ", queueId: " + queueId);
150        }
151      } catch (KeeperException e) {
152        throw new IOException("Failed to delete queue, replicator: " + replicator + ", queueId: "
153            + queueId);
154      }
155    }
156
157    /**
158     * @param hfileRefsQueueId The undeleted hfile-refs queue id
159     * @throws IOException
160     */
161    public void removeHFileRefsQueue(final String hfileRefsQueueId) throws IOException {
162      String node = ZKUtil.joinZNode(this.hfileRefsZNode, hfileRefsQueueId);
163      try {
164        if (!replicationPeers.getAllPeerIds().contains(hfileRefsQueueId)) {
165          ZKUtil.deleteNodeRecursively(this.zookeeper, node);
166          LOG.info("Successfully removed hfile-refs queue " + hfileRefsQueueId + " from path "
167              + hfileRefsZNode);
168        }
169      } catch (KeeperException e) {
170        throw new IOException("Failed to delete hfile-refs queue " + hfileRefsQueueId
171            + " from path " + hfileRefsZNode);
172      }
173    }
174
175    String getHfileRefsZNode() {
176      return this.hfileRefsZNode;
177    }
178  }
179
180  /**
181   * Remove the undeleted replication queue's zk node for removed peers.
182   * @param undeletedQueues replicator with its queueIds for removed peers
183   * @throws IOException
184   */
185  public void removeQueues(final Map<String, List<String>> undeletedQueues) throws IOException {
186    for (Entry<String, List<String>> replicatorAndQueueIds : undeletedQueues.entrySet()) {
187      String replicator = replicatorAndQueueIds.getKey();
188      for (String queueId : replicatorAndQueueIds.getValue()) {
189        queueDeletor.removeQueue(replicator, queueId);
190      }
191    }
192  }
193
194  /**
195   * Remove the undeleted hfile-refs queue's zk node for removed peers.
196   * @param undeletedHFileRefsQueues replicator with its undeleted queueIds for removed peers in
197   *          hfile-refs queue
198   * @throws IOException
199   */
200  public void removeHFileRefsQueues(final Set<String> undeletedHFileRefsQueues) throws IOException {
201    for (String hfileRefsQueueId : undeletedHFileRefsQueues) {
202      queueDeletor.removeHFileRefsQueue(hfileRefsQueueId);
203    }
204  }
