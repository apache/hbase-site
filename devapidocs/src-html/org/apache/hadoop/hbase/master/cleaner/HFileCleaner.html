040/**
041 * This Chore, every time it runs, will clear the HFiles in the hfile archive
042 * folder that are deletable for each HFile cleaner in the chain.
043 */
044@InterfaceAudience.Private
045public class HFileCleaner extends CleanerChore<BaseHFileCleanerDelegate> implements
046    ConfigurationObserver {
047
048  public static final String MASTER_HFILE_CLEANER_PLUGINS = "hbase.master.hfilecleaner.plugins";
049
050  public HFileCleaner(final int period, final Stoppable stopper, Configuration conf, FileSystem fs,
051      Path directory) {
052    this(period, stopper, conf, fs, directory, null);
053  }
054
055  // Configuration key for large/small throttle point
056  public final static String HFILE_DELETE_THROTTLE_THRESHOLD =
057      "hbase.regionserver.thread.hfilecleaner.throttle";
058  public final static int DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD = 64 * 1024 * 1024;// 64M
059
060  // Configuration key for large queue initial size
061  public final static String LARGE_HFILE_QUEUE_INIT_SIZE =
062      "hbase.regionserver.hfilecleaner.large.queue.size";
063  public final static int DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE = 10240;
064
065  // Configuration key for small queue initial size
066  public final static String SMALL_HFILE_QUEUE_INIT_SIZE =
067      "hbase.regionserver.hfilecleaner.small.queue.size";
068  public final static int DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE = 10240;
069
070  private static final Log LOG = LogFactory.getLog(HFileCleaner.class);
071
072  StealJobQueue<HFileDeleteTask> largeFileQueue;
073  BlockingQueue<HFileDeleteTask> smallFileQueue;
074  private int throttlePoint;
075  private int largeQueueInitSize;
076  private int smallQueueInitSize;
077  private List<Thread> threads = new ArrayList<Thread>();
078  private boolean running;
079
080  private long deletedLargeFiles = 0L;
081  private long deletedSmallFiles = 0L;
082
083  /**
084   * @param period the period of time to sleep between each run
085   * @param stopper the stopper
086   * @param conf configuration to use
087   * @param fs handle to the FS
088   * @param directory directory to be cleaned
089   * @param params params could be used in subclass of BaseHFileCleanerDelegate
090   */
091  public HFileCleaner(final int period, final Stoppable stopper, Configuration conf, FileSystem fs,
092                      Path directory, Map<String, Object> params) {
093    super("HFileCleaner", period, stopper, conf, fs,
094      directory, MASTER_HFILE_CLEANER_PLUGINS, params);
095    throttlePoint =
096        conf.getInt(HFILE_DELETE_THROTTLE_THRESHOLD, DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD);
097    largeQueueInitSize =
098        conf.getInt(LARGE_HFILE_QUEUE_INIT_SIZE, DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE);
099    smallQueueInitSize =
100        conf.getInt(SMALL_HFILE_QUEUE_INIT_SIZE, DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE);
101    largeFileQueue = new StealJobQueue<>(largeQueueInitSize, smallQueueInitSize);
102    smallFileQueue = largeFileQueue.getStealFromQueue();
103    startHFileDeleteThreads();
104  }
105
106  @Override
107  protected boolean validate(Path file) {
108    if (HFileLink.isBackReferencesDir(file) || HFileLink.isBackReferencesDir(file.getParent())) {
109      return true;
110    }
111    return StoreFileInfo.validateStoreFileName(file.getName());
112  }

121  @Override
122  public int deleteFiles(Iterable<FileStatus> filesToDelete) {
123    int deletedFiles = 0;
124    List<HFileDeleteTask> tasks = new ArrayList<HFileDeleteTask>();
125    // construct delete tasks and add into relative queue
126    for (FileStatus file : filesToDelete) {
127      HFileDeleteTask task = deleteFile(file);
128      if (task != null) {
129        tasks.add(task);
130      }
131    }
132    // wait for each submitted task to finish
133    for (HFileDeleteTask task : tasks) {
134      if (task.getResult()) {
135        deletedFiles++;
136      }
137    }
138    return deletedFiles;
139  }
140
141  /**
142   * Construct an {@link HFileDeleteTask} for each file to delete and add into the correct queue
143   * @param file the file to delete
144   * @return HFileDeleteTask to track progress
145   */
146  private HFileDeleteTask deleteFile(FileStatus file) {
147    HFileDeleteTask task = new HFileDeleteTask(file);
148    boolean enqueued = dispatch(task);
149    return enqueued ? task : null;
150  }
151
152  private boolean dispatch(HFileDeleteTask task) {
153    if (task.fileLength >= this.throttlePoint) {
154      if (!this.largeFileQueue.offer(task)) {
155        // should never arrive here as long as we use PriorityQueue
156        if (LOG.isTraceEnabled()) {
157          LOG.trace("Large file deletion queue is full");
158        }
159        return false;
160      }
161    } else {
162      if (!this.smallFileQueue.offer(task)) {
163        // should never arrive here as long as we use PriorityQueue
164        if (LOG.isTraceEnabled()) {
165          LOG.trace("Small file deletion queue is full");
166        }
167        return false;
168      }
169    }
170    return true;
171  }
172
173  @Override
174  public void cleanup() {
175    super.cleanup();
176    stopHFileDeleteThreads();
177  }
178
179  /**
180   * Start threads for hfile deletion
181   */
182  private void startHFileDeleteThreads() {
183    final String n = Thread.currentThread().getName();
184    running = true;
185    // start thread for large file deletion
186    Thread large = new Thread() {
187      @Override
188      public void run() {
189        consumerLoop(largeFileQueue);
190      }
191    };
192    large.setDaemon(true);
193    large.setName(n + "-HFileCleaner.large-" + System.currentTimeMillis());
194    large.start();
195    LOG.debug("Starting hfile cleaner for large files: " + large.getName());
196    threads.add(large);
197
198    // start thread for small file deletion
199    Thread small = new Thread() {
200      @Override
201      public void run() {
202        consumerLoop(smallFileQueue);
203      }
204    };
205    small.setDaemon(true);
206    small.setName(n + "-HFileCleaner.small-" + System.currentTimeMillis());
207    small.start();
208    LOG.debug("Starting hfile cleaner for small files: " + small.getName());
209    threads.add(small);
210  }
211
212  protected void consumerLoop(BlockingQueue<HFileDeleteTask> queue) {
213    try {
214      while (running) {
215        HFileDeleteTask task = null;
216        try {
217          task = queue.take();
218        } catch (InterruptedException e) {
219          if (LOG.isDebugEnabled()) {
220            LOG.debug("Interrupted while trying to take a task from queue", e);
221          }
222          break;
223        }
224        if (task != null) {
225          if (LOG.isDebugEnabled()) {
226            LOG.debug("Removing: " + task.filePath + " from archive");
227          }
228          boolean succeed;
229          try {
230            succeed = this.fs.delete(task.filePath, false);
231          } catch (IOException e) {
232            LOG.warn("Failed to delete file " + task.filePath, e);
233            succeed = false;
234          }
235          task.setResult(succeed);
236          if (succeed) {
237            countDeletedFiles(task.fileLength >= throttlePoint, queue == largeFileQueue);
238          }
239        }
240      }
241    } finally {
242      if (LOG.isDebugEnabled()) {
243        LOG.debug("Exit thread: " + Thread.currentThread());
244      }
245    }
246  }
247
248  // Currently only for testing purpose
249  private void countDeletedFiles(boolean isLargeFile, boolean fromLargeQueue) {
250    if (isLargeFile) {
251      if (deletedLargeFiles == Long.MAX_VALUE) {
252        LOG.info("Deleted more than Long.MAX_VALUE large files, reset counter to 0");
253        deletedLargeFiles = 0L;
254      }
255      deletedLargeFiles++;
256    } else {
257      if (deletedSmallFiles == Long.MAX_VALUE) {
258        LOG.info("Deleted more than Long.MAX_VALUE small files, reset counter to 0");
259        deletedSmallFiles = 0L;
260      }
261      if (fromLargeQueue && LOG.isTraceEnabled()) {
262        LOG.trace("Stolen a small file deletion task in large file thread");
263      }
264      deletedSmallFiles++;
265    }
266  }
267
268  /**
269   * Stop threads for hfile deletion
270   */
271  private void stopHFileDeleteThreads() {
272    running = false;
273    if (LOG.isDebugEnabled()) {
274      LOG.debug("Stopping file delete threads");
275    }
276    for(Thread thread: threads){
277      thread.interrupt();
278    }
279  }
280
281  static class HFileDeleteTask implements Comparable<HFileDeleteTask> {
282    private static final long MAX_WAIT = 60 * 1000L;
283    private static final long WAIT_UNIT = 1000L;
284
285    boolean done = false;
286    boolean result;
287    final Path filePath;
288    final long fileLength;
289
290    public HFileDeleteTask(FileStatus file) {
291      this.filePath = file.getPath();
292      this.fileLength = file.getLen();
293    }
294
295    public synchronized void setResult(boolean result) {
296      this.done = true;
297      this.result = result;
298      notify();
299    }
300
301    public synchronized boolean getResult() {
302      long waitTime = 0;
303      try {
304        while (!done) {
305          wait(WAIT_UNIT);
306          waitTime += WAIT_UNIT;
307          if (done) {
308            return this.result;
309          }
310          if (waitTime > MAX_WAIT) {
311            LOG.warn("Wait more than " + MAX_WAIT + " ms for deleting " + this.filePath
312                + ", exit...");
313            return false;
314          }
315        }
316      } catch (InterruptedException e) {
317        LOG.warn("Interrupted while waiting for result of deleting " + filePath
318            + ", will return false", e);
319        return false;
320      }
321      return this.result;
322    }
323
324    @Override
325    public int compareTo(HFileDeleteTask o) {
326      long sub = this.fileLength - o.fileLength;
327      // smaller value with higher priority in PriorityQueue, and we intent to delete the larger
328      // file first.
329      return (sub > 0) ? -1 : (sub < 0 ? 1 : 0);
330    }
331
332    @Override
333    public boolean equals(Object o) {
334      if (this == o) {
335        return true;
336      }
337      if (o == null || !(o instanceof HFileDeleteTask)) {
338        return false;
339      }
340      HFileDeleteTask otherTask = (HFileDeleteTask) o;
341      return this.filePath.equals(otherTask.filePath) && (this.fileLength == otherTask.fileLength);
342    }
343
344    @Override
345    public int hashCode() {
346      return filePath.hashCode();
347    }
348  }

380  @Override
381  public void onConfigurationChange(Configuration conf) {
382    StringBuilder builder = new StringBuilder();
383    builder.append("Updating configuration for HFileCleaner, previous throttle point: ")
384        .append(throttlePoint).append(", largeQueueInitSize: ").append(largeQueueInitSize)
385        .append(", smallQueueInitSize: ").append(smallQueueInitSize);
386    stopHFileDeleteThreads();
387    this.throttlePoint =
388        conf.getInt(HFILE_DELETE_THROTTLE_THRESHOLD, DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD);
389    this.largeQueueInitSize =
390        conf.getInt(LARGE_HFILE_QUEUE_INIT_SIZE, DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE);
391    this.smallQueueInitSize =
392        conf.getInt(SMALL_HFILE_QUEUE_INIT_SIZE, DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE);
393    // record the left over tasks
394    List<HFileDeleteTask> leftOverTasks = new ArrayList<>();
395    for (HFileDeleteTask task : largeFileQueue) {
396      leftOverTasks.add(task);
397    }
398    for (HFileDeleteTask task : smallFileQueue) {
399      leftOverTasks.add(task);
400    }
401    largeFileQueue = new StealJobQueue<>(largeQueueInitSize, smallQueueInitSize);
402    smallFileQueue = largeFileQueue.getStealFromQueue();
403    threads.clear();
404    builder.append("; new throttle point: ").append(throttlePoint).append(", largeQueueInitSize: ")
405        .append(largeQueueInitSize).append(", smallQueueInitSize: ").append(smallQueueInitSize);
406    LOG.debug(builder.toString());
407    startHFileDeleteThreads();
408    // re-dispatch the left over tasks
409    for (HFileDeleteTask task : leftOverTasks) {
410      dispatch(task);
411    }
412  }
