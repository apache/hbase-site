041 * This Chore, every time it runs, will clear the HFiles in the hfile archive
042 * folder that are deletable for each HFile cleaner in the chain.
043 */
044@InterfaceAudience.Private
045public class HFileCleaner extends CleanerChore<BaseHFileCleanerDelegate> {

047  public static final String MASTER_HFILE_CLEANER_PLUGINS = "hbase.master.hfilecleaner.plugins";

054  // Configuration key for large/small throttle point
055  public final static String HFILE_DELETE_THROTTLE_THRESHOLD =
056      "hbase.regionserver.thread.hfilecleaner.throttle";
057  public final static int DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD = 64 * 1024 * 1024;// 64M
058
059  // Configuration key for large queue initial size
060  public final static String LARGE_HFILE_QUEUE_INIT_SIZE =
061      "hbase.regionserver.hfilecleaner.large.queue.size";
062  public final static int DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE = 10240;
063
064  // Configuration key for small queue initial size
065  public final static String SMALL_HFILE_QUEUE_INIT_SIZE =
066      "hbase.regionserver.hfilecleaner.small.queue.size";
067  public final static int DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE = 10240;
068
069  // Configuration key for large file delete thread number
070  public final static String LARGE_HFILE_DELETE_THREAD_NUMBER =
071      "hbase.regionserver.hfilecleaner.large.thread.count";
072  public final static int DEFAULT_LARGE_HFILE_DELETE_THREAD_NUMBER = 1;
073
074  // Configuration key for small file delete thread number
075  public final static String SMALL_HFILE_DELETE_THREAD_NUMBER =
076      "hbase.regionserver.hfilecleaner.small.thread.count";
077  public final static int DEFAULT_SMALL_HFILE_DELETE_THREAD_NUMBER = 1;

079  private static final Logger LOG = LoggerFactory.getLogger(HFileCleaner.class);

081  StealJobQueue<HFileDeleteTask> largeFileQueue;
082  BlockingQueue<HFileDeleteTask> smallFileQueue;
083  private int throttlePoint;
084  private int largeQueueInitSize;
085  private int smallQueueInitSize;
086  private int largeFileDeleteThreadNumber;
087  private int smallFileDeleteThreadNumber;
088  private List<Thread> threads = new ArrayList<Thread>();
089  private boolean running;
090
091  private AtomicLong deletedLargeFiles = new AtomicLong();
092  private AtomicLong deletedSmallFiles = new AtomicLong();

094  /**
095   * @param period the period of time to sleep between each run
096   * @param stopper the stopper
097   * @param conf configuration to use
098   * @param fs handle to the FS
099   * @param directory directory to be cleaned
100   * @param params params could be used in subclass of BaseHFileCleanerDelegate
101   */
102  public HFileCleaner(final int period, final Stoppable stopper, Configuration conf, FileSystem fs,
103                      Path directory, Map<String, Object> params) {
104    super("HFileCleaner", period, stopper, conf, fs,
105      directory, MASTER_HFILE_CLEANER_PLUGINS, params);
106    throttlePoint =
107        conf.getInt(HFILE_DELETE_THROTTLE_THRESHOLD, DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD);
108    largeQueueInitSize =
109        conf.getInt(LARGE_HFILE_QUEUE_INIT_SIZE, DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE);
110    smallQueueInitSize =
111        conf.getInt(SMALL_HFILE_QUEUE_INIT_SIZE, DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE);
112    largeFileQueue = new StealJobQueue<>(largeQueueInitSize, smallQueueInitSize, COMPARATOR);
113    smallFileQueue = largeFileQueue.getStealFromQueue();
114    largeFileDeleteThreadNumber =
115        conf.getInt(LARGE_HFILE_DELETE_THREAD_NUMBER, DEFAULT_LARGE_HFILE_DELETE_THREAD_NUMBER);
116    smallFileDeleteThreadNumber =
117        conf.getInt(SMALL_HFILE_DELETE_THREAD_NUMBER, DEFAULT_SMALL_HFILE_DELETE_THREAD_NUMBER);
118    startHFileDeleteThreads();
119  }

121  @Override
122  protected boolean validate(Path file) {
123    if (HFileLink.isBackReferencesDir(file) || HFileLink.isBackReferencesDir(file.getParent())) {
124      return true;
125    }
126    return StoreFileInfo.validateStoreFileName(file.getName());
127  }

136  @Override
137  public int deleteFiles(Iterable<FileStatus> filesToDelete) {
138    int deletedFiles = 0;
139    List<HFileDeleteTask> tasks = new ArrayList<HFileDeleteTask>();
140    // construct delete tasks and add into relative queue
141    for (FileStatus file : filesToDelete) {
142      HFileDeleteTask task = deleteFile(file);
143      if (task != null) {
144        tasks.add(task);
145      }
146    }
147    // wait for each submitted task to finish
148    for (HFileDeleteTask task : tasks) {
149      if (task.getResult()) {
150        deletedFiles++;
151      }
152    }
153    return deletedFiles;
154  }
155
156  /**
157   * Construct an {@link HFileDeleteTask} for each file to delete and add into the correct queue
158   * @param file the file to delete
159   * @return HFileDeleteTask to track progress
160   */
161  private HFileDeleteTask deleteFile(FileStatus file) {
162    HFileDeleteTask task = new HFileDeleteTask(file);
163    boolean enqueued = dispatch(task);
164    return enqueued ? task : null;
165  }
166
167  private boolean dispatch(HFileDeleteTask task) {
168    if (task.fileLength >= this.throttlePoint) {
169      if (!this.largeFileQueue.offer(task)) {
170        // should never arrive here as long as we use PriorityQueue
171        LOG.trace("Large file deletion queue is full");
172        return false;
173      }
174    } else {
175      if (!this.smallFileQueue.offer(task)) {
176        // should never arrive here as long as we use PriorityQueue
177        LOG.trace("Small file deletion queue is full");
178        return false;
179      }
180    }
181    return true;
182  }

184  @Override
185  public void cleanup() {
186    super.cleanup();
187    stopHFileDeleteThreads();
188  }
189
190  /**
191   * Start threads for hfile deletion
192   */
193  private void startHFileDeleteThreads() {
194    final String n = Thread.currentThread().getName();
195    running = true;
196    // start thread for large file deletion
197    for (int i = 0; i < largeFileDeleteThreadNumber; i++) {
198      Thread large = new Thread() {
199        @Override
200        public void run() {
201          consumerLoop(largeFileQueue);
202        }
203      };
204      large.setDaemon(true);
205      large.setName(n + "-HFileCleaner.large." + i + "-" + System.currentTimeMillis());
206      large.start();
207      LOG.debug("Starting hfile cleaner for large files: {}", large);
208      threads.add(large);
209    }
210
211    // start thread for small file deletion
212    for (int i = 0; i < smallFileDeleteThreadNumber; i++) {
213      Thread small = new Thread() {
214        @Override
215        public void run() {
216          consumerLoop(smallFileQueue);
217        }
218      };
219      small.setDaemon(true);
220      small.setName(n + "-HFileCleaner.small." + i + "-" + System.currentTimeMillis());
221      small.start();
222      LOG.debug("Starting hfile cleaner for small files: {}", small);
223      threads.add(small);
224    }
225  }
226
227  protected void consumerLoop(BlockingQueue<HFileDeleteTask> queue) {
228    try {
229      while (running) {
230        HFileDeleteTask task = null;
231        try {
232          task = queue.take();
233        } catch (InterruptedException e) {
234          LOG.debug("Interrupted while trying to take a task from queue", e);
235          break;
236        }
237        if (task != null) {
238          LOG.debug("Removing: {} from archive", task.filePath);
239          boolean succeed;
240          try {
241            succeed = this.fs.delete(task.filePath, false);
242          } catch (IOException e) {
243            LOG.warn("Failed to delete file {}", task.filePath, e);
244            succeed = false;
245          }
246          task.setResult(succeed);
247          if (succeed) {
248            countDeletedFiles(task.fileLength >= throttlePoint, queue == largeFileQueue);
249          }
250        }
251      }
252    } finally {
253      LOG.debug("Exit thread: {}", Thread.currentThread());
254    }
255  }
256
257  // Currently only for testing purpose
258  private void countDeletedFiles(boolean isLargeFile, boolean fromLargeQueue) {
259    if (isLargeFile) {
260      if (deletedLargeFiles.get() == Long.MAX_VALUE) {
261        LOG.info("Deleted more than Long.MAX_VALUE large files, reset counter to 0");
262        deletedLargeFiles.set(0L);
263      }
264      deletedLargeFiles.incrementAndGet();
265    } else {
266      if (deletedSmallFiles.get() == Long.MAX_VALUE) {
267        LOG.info("Deleted more than Long.MAX_VALUE small files, reset counter to 0");
268        deletedSmallFiles.set(0L);
269      }
270      if (fromLargeQueue) {
271        LOG.trace("Stolen a small file deletion task in large file thread");
272      }
273      deletedSmallFiles.incrementAndGet();
274    }
275  }
276
277  /**
278   * Stop threads for hfile deletion
279   */
280  private void stopHFileDeleteThreads() {
281    running = false;
282    LOG.debug("Stopping file delete threads");
283    for(Thread thread: threads){
284      thread.interrupt();
285    }
286  }

288  private static final Comparator<HFileDeleteTask> COMPARATOR = new Comparator<HFileDeleteTask>() {
289
290    @Override
291    public int compare(HFileDeleteTask o1, HFileDeleteTask o2) {
292      // larger file first so reverse compare
293      int cmp = Long.compare(o2.fileLength, o1.fileLength);
294      if (cmp != 0) {
295        return cmp;
296      }
297      // just use hashCode to generate a stable result.
298      return System.identityHashCode(o1) - System.identityHashCode(o2);
299    }
300  };
301
302  private static final class HFileDeleteTask {
303    private static final long MAX_WAIT = 60 * 1000L;
304    private static final long WAIT_UNIT = 1000L;
305
306    boolean done = false;
307    boolean result;
308    final Path filePath;
309    final long fileLength;
310
311    public HFileDeleteTask(FileStatus file) {
312      this.filePath = file.getPath();
313      this.fileLength = file.getLen();
314    }
315
316    public synchronized void setResult(boolean result) {
317      this.done = true;
318      this.result = result;
319      notify();
320    }
321
322    public synchronized boolean getResult() {
323      long waitTime = 0;
324      try {
325        while (!done) {
326          wait(WAIT_UNIT);
327          waitTime += WAIT_UNIT;
328          if (done) {
329            return this.result;
330          }
331          if (waitTime > MAX_WAIT) {
332            LOG.warn("Wait more than " + MAX_WAIT + " ms for deleting " + this.filePath
333                + ", exit...");
334            return false;
335          }
336        }
337      } catch (InterruptedException e) {
338        LOG.warn("Interrupted while waiting for result of deleting " + filePath
339            + ", will return false", e);
340        return false;
341      }
342      return this.result;
343    }
344  }

376  @Override
377  public void onConfigurationChange(Configuration conf) {
378    super.onConfigurationChange(conf);
379
380    if (!checkAndUpdateConfigurations(conf)) {
381      LOG.debug("Update configuration triggered but nothing changed for this cleaner");
382      return;
383    }
384    stopHFileDeleteThreads();
385    // record the left over tasks
386    List<HFileDeleteTask> leftOverTasks =
387        new ArrayList<>(largeFileQueue.size() + smallFileQueue.size());
388    leftOverTasks.addAll(largeFileQueue);
389    leftOverTasks.addAll(smallFileQueue);
390    largeFileQueue = new StealJobQueue<>(largeQueueInitSize, smallQueueInitSize, COMPARATOR);
391    smallFileQueue = largeFileQueue.getStealFromQueue();
392    threads.clear();
393    startHFileDeleteThreads();
394    // re-dispatch the left over tasks
395    for (HFileDeleteTask task : leftOverTasks) {
396      dispatch(task);
397    }
398  }

400  /**
401   * Check new configuration and update settings if value changed
402   * @param conf The new configuration
403   * @return true if any configuration for HFileCleaner changes, false if no change
404   */
405  private boolean checkAndUpdateConfigurations(Configuration conf) {
406    boolean updated = false;
407    int throttlePoint =
408        conf.getInt(HFILE_DELETE_THROTTLE_THRESHOLD, DEFAULT_HFILE_DELETE_THROTTLE_THRESHOLD);
409    if (throttlePoint != this.throttlePoint) {
410      LOG.debug("Updating throttle point, from {} to {}", this.throttlePoint, throttlePoint);
411      this.throttlePoint = throttlePoint;
412      updated = true;
413    }
414    int largeQueueInitSize =
415        conf.getInt(LARGE_HFILE_QUEUE_INIT_SIZE, DEFAULT_LARGE_HFILE_QUEUE_INIT_SIZE);
416    if (largeQueueInitSize != this.largeQueueInitSize) {
417      LOG.debug("Updating largeQueueInitSize, from {} to {}", this.largeQueueInitSize,
418          largeQueueInitSize);
419      this.largeQueueInitSize = largeQueueInitSize;
420      updated = true;
421    }
422    int smallQueueInitSize =
423        conf.getInt(SMALL_HFILE_QUEUE_INIT_SIZE, DEFAULT_SMALL_HFILE_QUEUE_INIT_SIZE);
424    if (smallQueueInitSize != this.smallQueueInitSize) {
425      LOG.debug("Updating smallQueueInitSize, from {} to {}", this.smallQueueInitSize,
426          smallQueueInitSize);
427      this.smallQueueInitSize = smallQueueInitSize;
428      updated = true;
429    }
430    int largeFileDeleteThreadNumber =
431        conf.getInt(LARGE_HFILE_DELETE_THREAD_NUMBER, DEFAULT_LARGE_HFILE_DELETE_THREAD_NUMBER);
432    if (largeFileDeleteThreadNumber != this.largeFileDeleteThreadNumber) {
433      LOG.debug("Updating largeFileDeleteThreadNumber, from {} to {}",
434          this.largeFileDeleteThreadNumber, largeFileDeleteThreadNumber);
435      this.largeFileDeleteThreadNumber = largeFileDeleteThreadNumber;
436      updated = true;
437    }
438    int smallFileDeleteThreadNumber =
439        conf.getInt(SMALL_HFILE_DELETE_THREAD_NUMBER, DEFAULT_SMALL_HFILE_DELETE_THREAD_NUMBER);
440    if (smallFileDeleteThreadNumber != this.smallFileDeleteThreadNumber) {
441      LOG.debug("Updating smallFileDeleteThreadNumber, from {} to {}",
442          this.smallFileDeleteThreadNumber, smallFileDeleteThreadNumber);
443      this.smallFileDeleteThreadNumber = smallFileDeleteThreadNumber;
444      updated = true;
445    }
446    return updated;
447  }
