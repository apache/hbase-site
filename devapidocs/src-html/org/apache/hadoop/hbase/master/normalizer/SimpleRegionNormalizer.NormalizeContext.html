049/**
050 * Simple implementation of region normalizer. Logic in use:
051 * <ol>
052 *   <li>Get all regions of a given table</li>
053 *   <li>Get avg size S of the regions in the table (by total size of store files reported in
054 *     RegionMetrics)</li>
055 *   <li>For each region R0, if R0 is bigger than S * 2, it is kindly requested to split.</li>
056 *   <li>Otherwise, for the next region in the chain R1, if R0 + R1 is smaller then S, R0 and R1
057 *     are kindly requested to merge.</li>
058 * </ol>
059 */
060@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
061class SimpleRegionNormalizer implements RegionNormalizer, ConfigurationObserver {
062  private static final Logger LOG = LoggerFactory.getLogger(SimpleRegionNormalizer.class);
063

104  private static int parseMinRegionCount(final Configuration conf) {
105    final int parsedValue = conf.getInt(MIN_REGION_COUNT_KEY, DEFAULT_MIN_REGION_COUNT);
106    final int settledValue = Math.max(1, parsedValue);
107    if (parsedValue != settledValue) {
108      warnInvalidValue(MIN_REGION_COUNT_KEY, parsedValue, settledValue);
109    }
110    return settledValue;
111  }
112
113  private static Period parseMergeMinRegionAge(final Configuration conf) {
114    final int parsedValue =
115      conf.getInt(MERGE_MIN_REGION_AGE_DAYS_KEY, DEFAULT_MERGE_MIN_REGION_AGE_DAYS);
116    final int settledValue = Math.max(0, parsedValue);
117    if (parsedValue != settledValue) {
118      warnInvalidValue(MERGE_MIN_REGION_AGE_DAYS_KEY, parsedValue, settledValue);
119    }
120    return Period.ofDays(settledValue);
121  }
122
123  private static long parseMergeMinRegionSizeMb(final Configuration conf) {
124    final long parsedValue =
125      conf.getLong(MERGE_MIN_REGION_SIZE_MB_KEY, DEFAULT_MERGE_MIN_REGION_SIZE_MB);
126    final long settledValue = Math.max(0, parsedValue);
127    if (parsedValue != settledValue) {
128      warnInvalidValue(MERGE_MIN_REGION_SIZE_MB_KEY, parsedValue, settledValue);
129    }
130    return settledValue;
131  }

136      key, parsedValue, settledValue);

186  @Override
187  public List<NormalizationPlan> computePlansForTable(final TableName table) {
188    if (table == null) {
189      return Collections.emptyList();
190    }
191    if (table.isSystemTable()) {
192      LOG.debug("Normalization of system table {} isn't allowed", table);
193      return Collections.emptyList();
194    }
195
196    final boolean proceedWithSplitPlanning = proceedWithSplitPlanning();
197    final boolean proceedWithMergePlanning = proceedWithMergePlanning();
198    if (!proceedWithMergePlanning && !proceedWithSplitPlanning) {
199      LOG.debug("Both split and merge are disabled. Skipping normalization of table: {}", table);
200      return Collections.emptyList();
201    }
202
203    final NormalizeContext ctx = new NormalizeContext(table);
204    if (isEmpty(ctx.getTableRegions())) {
205      return Collections.emptyList();
206    }
207
208    LOG.debug("Computing normalization plan for table:  {}, number of regions: {}", table,
209      ctx.getTableRegions().size());
210
211    final List<NormalizationPlan> plans = new ArrayList<>();
212    int splitPlansCount = 0;
213    if (proceedWithSplitPlanning) {
214      List<NormalizationPlan> splitPlans = computeSplitNormalizationPlans(ctx);
215      splitPlansCount = splitPlans.size();
216      plans.addAll(splitPlans);
217    }
218    int mergePlansCount = 0;
219    if (proceedWithMergePlanning) {
220      List<NormalizationPlan> mergePlans = computeMergeNormalizationPlans(ctx);
221      mergePlansCount = mergePlans.size();
222      plans.addAll(mergePlans);
223    }
224
225    LOG.debug("Computed normalization plans for table {}. Total plans: {}, split plans: {}, " +
226        "merge plans: {}", table, plans.size(), splitPlansCount, mergePlansCount);
227    return plans;
228  }

230  /**
231   * @return size of region in MB and if region is not found than -1
232   */
233  private long getRegionSizeMB(RegionInfo hri) {
234    ServerName sn =
235      masterServices.getAssignmentManager().getRegionStates().getRegionServerOfRegion(hri);
236    if (sn == null) {
237      LOG.debug("{} region was not found on any Server", hri.getRegionNameAsString());
238      return -1;
239    }
240    ServerMetrics serverMetrics = masterServices.getServerManager().getLoad(sn);
241    if (serverMetrics == null) {
242      LOG.debug("server {} was not found in ServerManager", sn.getServerName());
243      return -1;
244    }
245    RegionMetrics regionLoad = serverMetrics.getRegionMetrics().get(hri.getRegionName());
246    if (regionLoad == null) {
247      LOG.debug("{} was not found in RegionsLoad", hri.getRegionNameAsString());
248      return -1;
249    }
250    return (long) regionLoad.getStoreFileSize().get(Size.Unit.MEGABYTE);
251  }

254    return masterServices.isSplitOrMergeEnabled(masterSwitchType);

257  private boolean proceedWithSplitPlanning() {
258    return isSplitEnabled() && isMasterSwitchEnabled(MasterSwitchType.SPLIT);

261  private boolean proceedWithMergePlanning() {

265  /**
266   * @param tableRegions regions of table to normalize
267   * @return average region size depending on
268   * @see org.apache.hadoop.hbase.client.TableDescriptor#getNormalizerTargetRegionCount()
269   * Also make sure tableRegions contains regions of the same table
270   */
271  private double getAverageRegionSizeMb(final List<RegionInfo> tableRegions) {
272    if (isEmpty(tableRegions)) {
273      throw new IllegalStateException(
274        "Cannot calculate average size of a table without any regions.");
275    }
276    TableName table = tableRegions.get(0).getTable();
277    int targetRegionCount = -1;
278    long targetRegionSize = -1;
279    double avgRegionSize;
280    try {
281      TableDescriptor tableDescriptor = masterServices.getTableDescriptors().get(table);
282      if (tableDescriptor != null) {
283        targetRegionCount = tableDescriptor.getNormalizerTargetRegionCount();
284        targetRegionSize = tableDescriptor.getNormalizerTargetRegionSize();
285        LOG.debug("Table {} configured with target region count {}, target region size {}", table,
286          targetRegionCount, targetRegionSize);
287      }
288    } catch (IOException e) {
289      LOG.warn("TableDescriptor for {} unavailable, table-level target region count and size"
290        + " configurations cannot be considered.", table, e);
291    }
292    if (targetRegionSize > 0) {
293      avgRegionSize = targetRegionSize;
294    } else {
295      final int regionCount = tableRegions.size();
296      final long totalSizeMb = tableRegions.stream()
297        .mapToLong(this::getRegionSizeMB)
298        .sum();
299      if (targetRegionCount > 0) {
300        avgRegionSize = totalSizeMb / (double) targetRegionCount;
301      } else {
302        avgRegionSize = totalSizeMb / (double) regionCount;
303      }
304      LOG.debug("Table {}, total aggregated regions size: {} and average region size {}", table,
305        totalSizeMb, avgRegionSize);
306    }
307
308    return avgRegionSize;
309  }

311  /**
312   * Determine if a {@link RegionInfo} should be considered for a merge operation.
313   * </p>
314   * Callers beware: for safe concurrency, be sure to pass in the local instance of
315   * {@link NormalizerConfiguration}, don't use {@code this}'s instance.
316   */
317  private boolean skipForMerge(
318    final NormalizerConfiguration normalizerConfiguration,
319    final RegionStates regionStates,
320    final RegionInfo regionInfo
321  ) {
322    final RegionState state = regionStates.getRegionState(regionInfo);
323    final String name = regionInfo.getEncodedName();
324    return
325      logTraceReason(
326        () -> state == null,
327        "skipping merge of region {} because no state information is available.", name)
328        || logTraceReason(
329          () -> !Objects.equals(state.getState(), RegionState.State.OPEN),
330          "skipping merge of region {} because it is not open.", name)
331        || logTraceReason(
332          () -> !isOldEnoughForMerge(normalizerConfiguration, regionInfo),
333          "skipping merge of region {} because it is not old enough.", name)
334        || logTraceReason(
335          () -> !isLargeEnoughForMerge(normalizerConfiguration, regionInfo),
336          "skipping merge region {} because it is not large enough.", name);
337  }

339  /**
340   * Computes the merge plans that should be executed for this table to converge average region
341   * towards target average or target region count.
342   */
343  private List<NormalizationPlan> computeMergeNormalizationPlans(final NormalizeContext ctx) {
344    final NormalizerConfiguration configuration = normalizerConfiguration;
345    if (ctx.getTableRegions().size() < configuration.getMinRegionCount()) {
346      LOG.debug("Table {} has {} regions, required min number of regions for normalizer to run"
347          + " is {}, not computing merge plans.", ctx.getTableName(),
348        ctx.getTableRegions().size(), configuration.getMinRegionCount());
349      return Collections.emptyList();
350    }
351
352    final long avgRegionSizeMb = (long) ctx.getAverageRegionSizeMb();
353    if (avgRegionSizeMb < configuration.getMergeMinRegionSizeMb()) {
354      return Collections.emptyList();
355    }
356    LOG.debug("Computing normalization plan for table {}. average region size: {}, number of"
357      + " regions: {}.", ctx.getTableName(), avgRegionSizeMb, ctx.getTableRegions().size());
358
359    // this nested loop walks the table's region chain once, looking for contiguous sequences of
360    // regions that meet the criteria for merge. The outer loop tracks the starting point of the
361    // next sequence, the inner loop looks for the end of that sequence. A single sequence becomes
362    // an instance of MergeNormalizationPlan.
363
364    final List<NormalizationPlan> plans = new LinkedList<>();
365    final List<NormalizationTarget> rangeMembers = new LinkedList<>();
366    long sumRangeMembersSizeMb;
367    int current = 0;
368    for (int rangeStart = 0;
369         rangeStart < ctx.getTableRegions().size() - 1 && current < ctx.getTableRegions().size();) {
370      // walk the region chain looking for contiguous sequences of regions that can be merged.
371      rangeMembers.clear();
372      sumRangeMembersSizeMb = 0;
373      for (current = rangeStart; current < ctx.getTableRegions().size(); current++) {
374        final RegionInfo regionInfo = ctx.getTableRegions().get(current);
375        final long regionSizeMb = getRegionSizeMB(regionInfo);
376        if (skipForMerge(configuration, ctx.getRegionStates(), regionInfo)) {
377          // this region cannot participate in a range. resume the outer loop.
378          rangeStart = Math.max(current, rangeStart + 1);
379          break;
380        }
381        if (rangeMembers.isEmpty() // when there are no range members, seed the range with whatever
382                                   // we have. this way we're prepared in case the next region is
383                                   // 0-size.
384          || (rangeMembers.size() == 1 && sumRangeMembersSizeMb == 0) // when there is only one
385                                                                      // region and the size is 0,
386                                                                      // seed the range with
387                                                                      // whatever we have.
388          || regionSizeMb == 0 // always add an empty region to the current range.
389          || (regionSizeMb + sumRangeMembersSizeMb <= avgRegionSizeMb)) { // add the current region
390                                                                          // to the range when
391                                                                          // there's capacity
392                                                                          // remaining.
393          rangeMembers.add(new NormalizationTarget(regionInfo, regionSizeMb));
394          sumRangeMembersSizeMb += regionSizeMb;
395          continue;
396        }
397        // we have accumulated enough regions to fill a range. resume the outer loop.
398        rangeStart = Math.max(current, rangeStart + 1);
399        break;
400      }
401      if (rangeMembers.size() > 1) {
402        plans.add(new MergeNormalizationPlan.Builder().setTargets(rangeMembers).build());
403      }
404    }
405    return plans;
406  }

408  /**
409   * Determine if a region in {@link RegionState} should be considered for a split operation.
410   */
411  private static boolean skipForSplit(final RegionState state, final RegionInfo regionInfo) {
412    final String name = regionInfo.getEncodedName();
413    return
414      logTraceReason(
415        () -> state == null,
416        "skipping split of region {} because no state information is available.", name)
417        || logTraceReason(
418          () -> !Objects.equals(state.getState(), RegionState.State.OPEN),
419          "skipping merge of region {} because it is not open.", name);
420  }

422  /**
423   * Computes the split plans that should be executed for this table to converge average region size
424   * towards target average or target region count.
425   * <br />
426   * if the region is > 2 times larger than average, we split it. split
427   * is more high priority normalization action than merge.
428   */
429  private List<NormalizationPlan> computeSplitNormalizationPlans(final NormalizeContext ctx) {
430    final double avgRegionSize = ctx.getAverageRegionSizeMb();
431    LOG.debug("Table {}, average region size: {}", ctx.getTableName(), avgRegionSize);
432
433    final List<NormalizationPlan> plans = new ArrayList<>();
434    for (final RegionInfo hri : ctx.getTableRegions()) {
435      if (skipForSplit(ctx.getRegionStates().getRegionState(hri), hri)) {
436        continue;
437      }
438      final long regionSizeMb = getRegionSizeMB(hri);
439      if (regionSizeMb > 2 * avgRegionSize) {
440        LOG.info("Table {}, large region {} has size {}, more than twice avg size {}, splitting",
441          ctx.getTableName(), hri.getRegionNameAsString(), regionSizeMb, avgRegionSize);
442        plans.add(new SplitNormalizationPlan(hri, regionSizeMb));
443      }
444    }
445    return plans;
446  }

448  /**
449   * Return {@code true} when {@code regionInfo} has a creation date that is old
450   * enough to be considered for a merge operation, {@code false} otherwise.
451   */
452  private static boolean isOldEnoughForMerge(
453    final NormalizerConfiguration normalizerConfiguration,
454    final RegionInfo regionInfo
455  ) {
456    final Instant currentTime = Instant.ofEpochMilli(EnvironmentEdgeManager.currentTime());
457    final Instant regionCreateTime = Instant.ofEpochMilli(regionInfo.getRegionId());
458    return currentTime.isAfter(
459      regionCreateTime.plus(normalizerConfiguration.getMergeMinRegionAge()));
460  }

462  /**
463   * Return {@code true} when {@code regionInfo} has a size that is sufficient
464   * to be considered for a merge operation, {@code false} otherwise.
465   * </p>
466   * Callers beware: for safe concurrency, be sure to pass in the local instance of
467   * {@link NormalizerConfiguration}, don't use {@code this}'s instance.
468   */
469  private boolean isLargeEnoughForMerge(
470    final NormalizerConfiguration normalizerConfiguration,
471    final RegionInfo regionInfo
472  ) {
473    return getRegionSizeMB(regionInfo) >= normalizerConfiguration.getMergeMinRegionSizeMb();
474  }

485  /**
486   * Holds the configuration values read from {@link Configuration}. Encapsulation in a POJO
487   * enables atomic hot-reloading of configs without locks.
488   */
489  private static final class NormalizerConfiguration {
490    private final Configuration conf;
491    private final boolean splitEnabled;
492    private final boolean mergeEnabled;
493    private final int minRegionCount;
494    private final Period mergeMinRegionAge;
495    private final long mergeMinRegionSizeMb;
496
497    private NormalizerConfiguration() {
498      conf = null;
499      splitEnabled = DEFAULT_SPLIT_ENABLED;
500      mergeEnabled = DEFAULT_MERGE_ENABLED;
501      minRegionCount = DEFAULT_MIN_REGION_COUNT;
502      mergeMinRegionAge = Period.ofDays(DEFAULT_MERGE_MIN_REGION_AGE_DAYS);
503      mergeMinRegionSizeMb = DEFAULT_MERGE_MIN_REGION_SIZE_MB;
504    }
505
506    private NormalizerConfiguration(
507      final Configuration conf,
508      final NormalizerConfiguration currentConfiguration
509    ) {
510      this.conf = conf;
511      splitEnabled = conf.getBoolean(SPLIT_ENABLED_KEY, DEFAULT_SPLIT_ENABLED);
512      mergeEnabled = conf.getBoolean(MERGE_ENABLED_KEY, DEFAULT_MERGE_ENABLED);
513      minRegionCount = parseMinRegionCount(conf);
514      mergeMinRegionAge = parseMergeMinRegionAge(conf);
515      mergeMinRegionSizeMb = parseMergeMinRegionSizeMb(conf);
516      logConfigurationUpdated(SPLIT_ENABLED_KEY, currentConfiguration.isSplitEnabled(),
517        splitEnabled);
518      logConfigurationUpdated(MERGE_ENABLED_KEY, currentConfiguration.isMergeEnabled(),
519        mergeEnabled);
520      logConfigurationUpdated(MIN_REGION_COUNT_KEY, currentConfiguration.getMinRegionCount(),
521        minRegionCount);
522      logConfigurationUpdated(MERGE_MIN_REGION_AGE_DAYS_KEY,
523        currentConfiguration.getMergeMinRegionAge(), mergeMinRegionAge);
524      logConfigurationUpdated(MERGE_MIN_REGION_SIZE_MB_KEY,
525        currentConfiguration.getMergeMinRegionSizeMb(), mergeMinRegionSizeMb);
526    }
527
528    public Configuration getConf() {
529      return conf;
530    }
531
532    public boolean isSplitEnabled() {
533      return splitEnabled;
534    }
535
536    public boolean isMergeEnabled() {
537      return mergeEnabled;
538    }
539
540    public int getMinRegionCount() {
541      return minRegionCount;
542    }
543
544    public Period getMergeMinRegionAge() {
545      return mergeMinRegionAge;
546    }
547
548    public long getMergeMinRegionSizeMb() {
549      return mergeMinRegionSizeMb;
550    }
551  }

553  /**
554   * Inner class caries the state necessary to perform a single invocation of
555   * {@link #computePlansForTable(TableName)}. Grabbing this data from the assignment manager
556   * up-front allows any computed values to be realized just once.
557   */
558  private class NormalizeContext {
559    private final TableName tableName;
560    private final RegionStates regionStates;
561    private final List<RegionInfo> tableRegions;
562    private final double averageRegionSizeMb;
563
564    public NormalizeContext(final TableName tableName) {
565      this.tableName = tableName;
566      regionStates = SimpleRegionNormalizer.this.masterServices
567        .getAssignmentManager()
568        .getRegionStates();
569      tableRegions = regionStates.getRegionsOfTable(tableName);
570      // The list of regionInfo from getRegionsOfTable() is ordered by regionName.
571      // regionName does not necessary guarantee the order by STARTKEY (let's say 'aa1', 'aa1!',
572      // in order by regionName, it will be 'aa1!' followed by 'aa1').
573      // This could result in normalizer merging non-adjacent regions into one and creates overlaps.
574      // In order to avoid that, sort the list by RegionInfo.COMPARATOR.
575      // See HBASE-24376
576      tableRegions.sort(RegionInfo.COMPARATOR);
577      averageRegionSizeMb = SimpleRegionNormalizer.this.getAverageRegionSizeMb(this.tableRegions);
578    }
579
580    public TableName getTableName() {
581      return tableName;
582    }
583
584    public RegionStates getRegionStates() {
585      return regionStates;
586    }
587
588    public List<RegionInfo> getTableRegions() {
589      return tableRegions;
590    }
591
592    public double getAverageRegionSizeMb() {
593      return averageRegionSizeMb;
594    }
595  }
