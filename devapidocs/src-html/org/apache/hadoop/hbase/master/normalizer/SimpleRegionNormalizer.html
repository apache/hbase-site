019package org.apache.hadoop.hbase.master.normalizer;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.hbase.HBaseIOException;
024import org.apache.hadoop.hbase.HRegionInfo;
025import org.apache.hadoop.hbase.RegionLoad;
026import org.apache.hadoop.hbase.ServerName;
027import org.apache.hadoop.hbase.TableName;
028import org.apache.hadoop.hbase.classification.InterfaceAudience;
029import org.apache.hadoop.hbase.master.MasterServices;
030import org.apache.hadoop.hbase.util.Pair;
031
032import java.util.List;
033
034/**
035 * Simple implementation of region normalizer.
036 *
037 * Logic in use:
038 *
039 *  <ol>
040 *  <li> get all regions of a given table
041 *  <li> get avg size S of each region (by total size of store files reported in RegionLoad)
042 *  <li> If biggest region is bigger than S * 2, it is kindly requested to split,
043 *    and normalization stops
044 *  <li> Otherwise, two smallest region R1 and its smallest neighbor R2 are kindly requested
045 *    to merge, if R1 + R1 &lt;  S, and normalization stops
046 *  <li> Otherwise, no action is performed
047 * </ol>
048 * <p>
049 * Region sizes are coarse and approximate on the order of megabytes. Additionally,
050 * "empty" regions (less than 1MB, with the previous note) are not merged away. This
051 * is by design to prevent normalization from undoing the pre-splitting of a table.
052 */
053@InterfaceAudience.Private
054public class SimpleRegionNormalizer implements RegionNormalizer {
055
056  private static final Log LOG = LogFactory.getLog(SimpleRegionNormalizer.class);
057  private static final int MIN_REGION_COUNT = 3;
058  private MasterServices masterServices;
059
060  /**
061   * Set the master service.
062   * @param masterServices inject instance of MasterServices
063   */
064  @Override
065  public void setMasterServices(MasterServices masterServices) {
066    this.masterServices = masterServices;
067  }
068
069  /**
070   * Computes next most "urgent" normalization action on the table.
071   * Action may be either a split, or a merge, or no action.
072   *
073   * @param table table to normalize
074   * @return normalization plan to execute
075   */
076  @Override
077  public NormalizationPlan computePlanForTable(TableName table) throws HBaseIOException {
078    if (table == null || table.isSystemTable()) {
079      LOG.debug("Normalization of table " + table + " isn't allowed");
080      return EmptyNormalizationPlan.getInstance();
081    }
082
083    List<HRegionInfo> tableRegions = masterServices.getAssignmentManager().getRegionStates().
084      getRegionsOfTable(table);
085
086    //TODO: should we make min number of regions a config param?
087    if (tableRegions == null || tableRegions.size() < MIN_REGION_COUNT) {
088      int nrRegions = tableRegions == null ? 0 : tableRegions.size();
089      LOG.debug("Table " + table + " has " + nrRegions + " regions, required min number"
090        + " of regions for normalizer to run is " + MIN_REGION_COUNT + ", not running normalizer");
091      return EmptyNormalizationPlan.getInstance();
092    }
093
094    LOG.debug("Computing normalization plan for table: " + table +
095      ", number of regions: " + tableRegions.size());
096
097    long totalSizeMb = 0;
098    Pair<HRegionInfo, Long> largestRegion = new Pair<>();
099
100    // A is a smallest region, B is it's smallest neighbor
101    Pair<HRegionInfo, Long> smallestRegion = new Pair<>();
102    int smallestRegionIndex = 0;
103
104    for (int i = 0; i < tableRegions.size(); i++) {
105      HRegionInfo hri = tableRegions.get(i);
106      long regionSize = getRegionSize(hri);
107      totalSizeMb += regionSize;
108
109      if (largestRegion.getFirst() == null || regionSize > largestRegion.getSecond()) {
110        largestRegion.setFirst(hri);
111        largestRegion.setSecond(regionSize);
112      }
113
114      if (smallestRegion.getFirst() == null || regionSize < smallestRegion.getSecond()) {
115        smallestRegion.setFirst(hri);
116        smallestRegion.setSecond(regionSize);
117        smallestRegionIndex = i;
118      }
119    }
120
121    // now get smallest neighbor of smallest region
122    long leftNeighborSize = -1;
123
124    if (smallestRegionIndex > 0) {
125      leftNeighborSize = getRegionSize(tableRegions.get(smallestRegionIndex - 1));
126    }
127
128    long rightNeighborSize = -1;
129    if (smallestRegionIndex < tableRegions.size() - 1) {
130      rightNeighborSize = getRegionSize(tableRegions.get(smallestRegionIndex + 1));
131    }
132
133    Pair<HRegionInfo, Long> smallestNeighborOfSmallestRegion;
134    if (leftNeighborSize == -1) {
135      smallestNeighborOfSmallestRegion =
136        new Pair<>(tableRegions.get(smallestRegionIndex + 1), rightNeighborSize);
137    } else if (rightNeighborSize == -1) {
138      smallestNeighborOfSmallestRegion =
139        new Pair<>(tableRegions.get(smallestRegionIndex - 1), leftNeighborSize);
140    } else {
141      if (leftNeighborSize < rightNeighborSize) {
142        smallestNeighborOfSmallestRegion =
143          new Pair<>(tableRegions.get(smallestRegionIndex - 1), leftNeighborSize);
144      } else {
145        smallestNeighborOfSmallestRegion =
146          new Pair<>(tableRegions.get(smallestRegionIndex + 1), rightNeighborSize);
147      }
148    }
149
150    double avgRegionSize = totalSizeMb / (double) tableRegions.size();
151
152    LOG.debug("Table " + table + ", total aggregated regions size: " + totalSizeMb);
153    LOG.debug("Table " + table + ", average region size: " + avgRegionSize);
154
155    // now; if the largest region is >2 times large than average, we split it, split
156    // is more high priority normalization action than merge.
157    if (largestRegion.getSecond() > 2 * avgRegionSize) {
158      LOG.debug("Table " + table + ", largest region "
159        + largestRegion.getFirst().getRegionNameAsString() + " has size "
160        + largestRegion.getSecond() + ", more than 2 times than avg size, splitting");
161      return new SplitNormalizationPlan(largestRegion.getFirst(), null);
162    } else {
163      if (smallestRegion.getSecond() + smallestNeighborOfSmallestRegion.getSecond()
164          < avgRegionSize) {
165        LOG.debug("Table " + table + ", smallest region size: " + smallestRegion.getSecond()
166          + " and its smallest neighbor size: " + smallestNeighborOfSmallestRegion.getSecond()
167          + ", less than the avg size, merging them");
168        return new MergeNormalizationPlan(smallestRegion.getFirst(),
169          smallestNeighborOfSmallestRegion.getFirst());
170      } else {
171        LOG.debug("No normalization needed, regions look good for table: " + table);
172        return EmptyNormalizationPlan.getInstance();
173      }
174    }
175  }
176
177  private long getRegionSize(HRegionInfo hri) {
178    ServerName sn = masterServices.getAssignmentManager().getRegionStates().
179      getRegionServerOfRegion(hri);
180    RegionLoad regionLoad = masterServices.getServerManager().getLoad(sn).
181      getRegionsLoad().get(hri.getRegionName());
182    return regionLoad.getStorefileSizeMB();
183  }
184}
