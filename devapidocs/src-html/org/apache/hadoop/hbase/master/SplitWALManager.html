018package org.apache.hadoop.hbase.master;
019import static org.apache.hadoop.hbase.HConstants.DEFAULT_HBASE_SPLIT_WAL_MAX_SPLITTER;
020import static org.apache.hadoop.hbase.HConstants.HBASE_SPLIT_WAL_MAX_SPLITTER;
021import static org.apache.hadoop.hbase.master.MasterWalManager.META_FILTER;
022import static org.apache.hadoop.hbase.master.MasterWalManager.NON_META_FILTER;
023import java.io.IOException;
024import java.util.Arrays;
025import java.util.Collections;
026import java.util.List;
027import java.util.Optional;
028import java.util.stream.Collectors;
029import org.apache.hadoop.conf.Configuration;
030import org.apache.hadoop.fs.FileStatus;
031import org.apache.hadoop.fs.FileSystem;
032import org.apache.hadoop.fs.Path;
033import org.apache.hadoop.fs.PathIsNotEmptyDirectoryException;
034import org.apache.hadoop.hbase.HConstants;
035import org.apache.hadoop.hbase.ServerName;
036import org.apache.hadoop.hbase.master.procedure.MasterProcedureScheduler;
037import org.apache.hadoop.hbase.master.procedure.SplitWALProcedure;
038import org.apache.hadoop.hbase.procedure2.Procedure;
039import org.apache.hadoop.hbase.procedure2.ProcedureEvent;
040import org.apache.hadoop.hbase.procedure2.ProcedureSuspendedException;

044import org.apache.yetus.audience.InterfaceAudience;
045import org.slf4j.Logger;

048/**
049 * Create {@link SplitWALProcedure} for each WAL which need to split. Manage the workers for each
050 * {@link SplitWALProcedure}.
051 * Total number of workers is (number of online servers) * (HBASE_SPLIT_WAL_MAX_SPLITTER).
052 * Helps assign and release workers for split tasks.
053 * Provide helper method to delete split WAL file and directory.
054 *
055 * The user can get the SplitWALProcedures via splitWALs(crashedServer, splitMeta)
056 * can get the files that need to split via getWALsToSplit(crashedServer, splitMeta)
057 * can delete the splitting WAL and directory via deleteSplitWAL(wal)
058 * and deleteSplitWAL(crashedServer)
059 * can check if splitting WALs of a crashed server is success via isSplitWALFinished(walPath)
060 * can acquire and release a worker for splitting WAL via acquireSplitWALWorker(procedure)
061 * and releaseSplitWALWorker(worker, scheduler)
062 *
063 * This class is to replace the zk-based WAL splitting related code, {@link MasterWalManager},
064 * {@link SplitLogManager}, {@link org.apache.hadoop.hbase.zookeeper.ZKSplitLog} and
065 * {@link org.apache.hadoop.hbase.coordination.ZKSplitLogManagerCoordination} can be removed
066 * after we switch to procedure-based WAL splitting.
067 * @see SplitLogManager for the original distributed split WAL manager.
068 */
069@InterfaceAudience.Private
070public class SplitWALManager {
071  private static final Logger LOG = LoggerFactory.getLogger(SplitWALManager.class);
072
073  private final MasterServices master;
074  private final WorkerAssigner splitWorkerAssigner;
075  private final Path rootDir;
076  private final FileSystem fs;
077  private final Configuration conf;
078  private final Path walArchiveDir;
079
080  public SplitWALManager(MasterServices master) throws IOException {
081    this.master = master;
082    this.conf = master.getConfiguration();
083    this.splitWorkerAssigner = new WorkerAssigner(this.master,
084        conf.getInt(HBASE_SPLIT_WAL_MAX_SPLITTER, DEFAULT_HBASE_SPLIT_WAL_MAX_SPLITTER),
085        new ProcedureEvent<>("split-WAL-worker-assigning"));
086    this.rootDir = master.getMasterFileSystem().getWALRootDir();
087    this.fs = master.getMasterFileSystem().getWALFileSystem();
088    this.walArchiveDir = new Path(this.rootDir, HConstants.HREGION_OLDLOGDIR_NAME);
089  }
090
091  public List<Procedure> splitWALs(ServerName crashedServer, boolean splitMeta)
092      throws IOException {
093    try {
094      // 1. list all splitting files
095      List<FileStatus> splittingFiles = getWALsToSplit(crashedServer, splitMeta);
096      // 2. create corresponding procedures
097      return createSplitWALProcedures(splittingFiles, crashedServer);
098    } catch (IOException e) {
099      LOG.error("Failed to create procedures for splitting WALs of {}", crashedServer, e);
100      throw e;
101    }
102  }
103
104  public List<FileStatus> getWALsToSplit(ServerName serverName, boolean splitMeta)
105      throws IOException {
106    List<Path> logDirs = master.getMasterWalManager().getLogDirs(Collections.singleton(serverName));
107    List<FileStatus> fileStatuses =
108      SplitLogManager.getFileList(this.conf, logDirs, splitMeta ? META_FILTER : NON_META_FILTER);
109    LOG.info("{} WAL count={}, meta={}", serverName, fileStatuses.size(), splitMeta);
110    return fileStatuses;
111  }
112
113  private Path getWALSplitDir(ServerName serverName) {
114    Path logDir =
115        new Path(this.rootDir, AbstractFSWALProvider.getWALDirectoryName(serverName.toString()));
116    return logDir.suffix(AbstractFSWALProvider.SPLITTING_EXT);
117  }
118
119  /**
120   * Archive processed WAL
121   */
122  public void archive(String wal) throws IOException {
123    WALSplitUtil.moveWAL(this.fs, new Path(wal), this.walArchiveDir);
124  }
125
126  public void deleteWALDir(ServerName serverName) throws IOException {
127    Path splitDir = getWALSplitDir(serverName);
128    try {
129      if (!fs.delete(splitDir, false)) {
130        LOG.warn("Failed delete {}, contains {}", splitDir, fs.listFiles(splitDir, true));
131      }
132    } catch (PathIsNotEmptyDirectoryException e) {
133      FileStatus [] files = CommonFSUtils.listStatus(fs, splitDir);
134      LOG.warn("PathIsNotEmptyDirectoryException {}",
135        Arrays.stream(files).map(f -> f.getPath()).collect(Collectors.toList()));
136      throw e;
137    }
138  }
139
140  public boolean isSplitWALFinished(String walPath) throws IOException {
141    return !fs.exists(new Path(rootDir, walPath));
142  }
143
144  List<Procedure> createSplitWALProcedures(List<FileStatus> splittingWALs,
145      ServerName crashedServer) {
146    return splittingWALs.stream()
147        .map(wal -> new SplitWALProcedure(wal.getPath().toString(), crashedServer))
148        .collect(Collectors.toList());
149  }
150
151  /**
152   * Acquire a split WAL worker
153   * @param procedure split WAL task
154   * @return an available region server which could execute this task
155   * @throws ProcedureSuspendedException if there is no available worker,
156   *         it will throw this exception to WAIT the procedure.
157   */
158  public ServerName acquireSplitWALWorker(Procedure<?> procedure)
159      throws ProcedureSuspendedException {
160    Optional<ServerName> worker = splitWorkerAssigner.acquire();
161    if (worker.isPresent()) {
162      LOG.debug("Acquired split WAL worker={}", worker.get());
163      return worker.get();
164    }
165    splitWorkerAssigner.suspend(procedure);
166    throw new ProcedureSuspendedException();
167  }
168
169  /**
170   * After the worker finished the split WAL task, it will release the worker, and wake up all the
171   * suspend procedures in the ProcedureEvent
172   * @param worker worker which is about to release
173   * @param scheduler scheduler which is to wake up the procedure event
174   */
175  public void releaseSplitWALWorker(ServerName worker, MasterProcedureScheduler scheduler) {
176    LOG.debug("Release split WAL worker={}", worker);
177    splitWorkerAssigner.release(worker);
178    splitWorkerAssigner.wake(scheduler);
179  }
180
181  /**
182   * When master restart, there will be a new splitWorkerAssigner. But if there are splitting WAL
183   * tasks running on the region server side, they will not be count by the new splitWorkerAssigner.
184   * Thus we should add the workers of running tasks to the assigner when we load the procedures
185   * from MasterProcWALs.
186   * @param worker region server which is executing a split WAL task
187   */
188  public void addUsedSplitWALWorker(ServerName worker){
189    splitWorkerAssigner.addUsedWorker(worker);
190  }
191}
