051/**
052 * Create {@link SplitWALProcedure} for each WAL which need to split. Manage the workers for each
053 * {@link SplitWALProcedure}.
054 * Total number of workers is (number of online servers) * (HBASE_SPLIT_WAL_MAX_SPLITTER).
055 * Helps assign and release workers for split tasks.
056 * Provide helper method to delete split WAL file and directory.
057 *
058 * The user can get the SplitWALProcedures via splitWALs(crashedServer, splitMeta)
059 * can get the files that need to split via getWALsToSplit(crashedServer, splitMeta)
060 * can delete the splitting WAL and directory via deleteSplitWAL(wal)
061 * and deleteSplitWAL(crashedServer)
062 * can check if splitting WALs of a crashed server is success via isSplitWALFinished(walPath)
063 * can acquire and release a worker for splitting WAL via acquireSplitWALWorker(procedure)
064 * and releaseSplitWALWorker(worker, scheduler)
065 *
066 * This class is to replace the zk-based WAL splitting related code, {@link MasterWalManager},
067 * {@link SplitLogManager}, {@link org.apache.hadoop.hbase.zookeeper.ZKSplitLog} and
068 * {@link org.apache.hadoop.hbase.coordination.ZKSplitLogManagerCoordination} can be removed
069 * after we switch to procedure-based WAL splitting.
070 * @see SplitLogManager for the original distributed split WAL manager.
071 */
072@InterfaceAudience.Private
073public class SplitWALManager {
074  private static final Logger LOG = LoggerFactory.getLogger(SplitWALManager.class);
075
076  private final MasterServices master;
077  private final SplitWorkerAssigner splitWorkerAssigner;
078  private final Path rootDir;
079  private final FileSystem fs;
080  private final Configuration conf;
081  private final Path walArchiveDir;
082
083  public SplitWALManager(MasterServices master) throws IOException {
084    this.master = master;
085    this.conf = master.getConfiguration();
086    this.splitWorkerAssigner = new SplitWorkerAssigner(this.master,
087        conf.getInt(HBASE_SPLIT_WAL_MAX_SPLITTER, DEFAULT_HBASE_SPLIT_WAL_MAX_SPLITTER));
088    this.rootDir = master.getMasterFileSystem().getWALRootDir();
089    // TODO: This should be the WAL FS, not the Master FS?
090    this.fs = master.getMasterFileSystem().getFileSystem();
091    this.walArchiveDir = new Path(this.rootDir, HConstants.HREGION_OLDLOGDIR_NAME);
092  }
093
094  public List<Procedure> splitWALs(ServerName crashedServer, boolean splitMeta)
095      throws IOException {
096    try {
097      // 1. list all splitting files
098      List<FileStatus> splittingFiles = getWALsToSplit(crashedServer, splitMeta);
099      // 2. create corresponding procedures
100      return createSplitWALProcedures(splittingFiles, crashedServer);
101    } catch (IOException e) {
102      LOG.error("Failed to create procedures for splitting WALs of {}", crashedServer, e);
103      throw e;
104    }
105  }
106
107  public List<FileStatus> getWALsToSplit(ServerName serverName, boolean splitMeta)
108      throws IOException {
109    List<Path> logDirs = master.getMasterWalManager().getLogDirs(Collections.singleton(serverName));
110    FileStatus[] fileStatuses =
111        SplitLogManager.getFileList(this.conf, logDirs, splitMeta ? META_FILTER : NON_META_FILTER);
112    LOG.info("{} WAL count={}, meta={}", serverName, fileStatuses.length, splitMeta);
113    return Lists.newArrayList(fileStatuses);
114  }
115
116  private Path getWALSplitDir(ServerName serverName) {
117    Path logDir =
118        new Path(this.rootDir, AbstractFSWALProvider.getWALDirectoryName(serverName.toString()));
119    return logDir.suffix(AbstractFSWALProvider.SPLITTING_EXT);
120  }
121
122  /**
123   * Archive processed WAL
124   */
125  public void archive(String wal) throws IOException {
126    WALSplitUtil.moveWAL(this.fs, new Path(wal), this.walArchiveDir);
127  }
128
129  public void deleteWALDir(ServerName serverName) throws IOException {
130    Path splitDir = getWALSplitDir(serverName);
131    try {
132      if (!fs.delete(splitDir, false)) {
133        LOG.warn("Failed delete {}, contains {}", splitDir, fs.listFiles(splitDir, true));
134      }
135    } catch (PathIsNotEmptyDirectoryException e) {
136      FileStatus [] files = CommonFSUtils.listStatus(fs, splitDir);
137      LOG.warn("PathIsNotEmptyDirectoryException {}",
138        Arrays.stream(files).map(f -> f.getPath()).collect(Collectors.toList()));
139      throw e;
140    }
141  }
142
143  public boolean isSplitWALFinished(String walPath) throws IOException {
144    return !fs.exists(new Path(rootDir, walPath));
145  }
146
147  List<Procedure> createSplitWALProcedures(List<FileStatus> splittingWALs,
148      ServerName crashedServer) {
149    return splittingWALs.stream()
150        .map(wal -> new SplitWALProcedure(wal.getPath().toString(), crashedServer))
151        .collect(Collectors.toList());
152  }
153
154  /**
155   * Acquire a split WAL worker
156   * @param procedure split WAL task
157   * @return an available region server which could execute this task
158   * @throws ProcedureSuspendedException if there is no available worker,
159   *         it will throw this exception to WAIT the procedure.
160   */
161  public ServerName acquireSplitWALWorker(Procedure<?> procedure)
162      throws ProcedureSuspendedException {
163    Optional<ServerName> worker = splitWorkerAssigner.acquire();
164    if (worker.isPresent()) {
165      LOG.debug("Acquired split WAL worker={}", worker.get());
166      return worker.get();
167    }
168    splitWorkerAssigner.suspend(procedure);
169    throw new ProcedureSuspendedException();
170  }
171
172  /**
173   * After the worker finished the split WAL task, it will release the worker, and wake up all the
174   * suspend procedures in the ProcedureEvent
175   * @param worker worker which is about to release
176   * @param scheduler scheduler which is to wake up the procedure event
177   */
178  public void releaseSplitWALWorker(ServerName worker, MasterProcedureScheduler scheduler) {
179    LOG.debug("Release split WAL worker={}", worker);
180    splitWorkerAssigner.release(worker);
181    splitWorkerAssigner.wake(scheduler);
182  }
183
184  /**
185   * When master restart, there will be a new splitWorkerAssigner. But if there are splitting WAL
186   * tasks running on the region server side, they will not be count by the new splitWorkerAssigner.
187   * Thus we should add the workers of running tasks to the assigner when we load the procedures
188   * from MasterProcWALs.
189   * @param worker region server which is executing a split WAL task
190   */
191  public void addUsedSplitWALWorker(ServerName worker){
192    splitWorkerAssigner.addUsedWorker(worker);
193  }
194
195  /**
196   * help assign and release a worker for each WAL splitting task
197   * For each worker, concurrent running splitting task should be no more than maxSplitTasks
198   * If a task failed to acquire a worker, it will suspend and wait for workers available
199   *
200   */
201  private static final class SplitWorkerAssigner implements ServerListener {
202    private int maxSplitTasks;
203    private final ProcedureEvent<?> event;
204    private Map<ServerName, Integer> currentWorkers = new HashMap<>();
205    private MasterServices master;
206
207    public SplitWorkerAssigner(MasterServices master, int maxSplitTasks) {
208      this.maxSplitTasks = maxSplitTasks;
209      this.master = master;
210      this.event = new ProcedureEvent<>("split-WAL-worker-assigning");
211      // ServerManager might be null in a test context where we are mocking; allow for this
212      ServerManager sm = this.master.getServerManager();
213      if (sm != null) {
214        sm.registerListener(this);
215      }
216    }
217
218    public synchronized Optional<ServerName> acquire() {
219      List<ServerName> serverList = master.getServerManager().getOnlineServersList();
220      Collections.shuffle(serverList);
221      Optional<ServerName> worker = serverList.stream().filter(
222        serverName -> !currentWorkers.containsKey(serverName) || currentWorkers.get(serverName) > 0)
223          .findAny();
224      if (worker.isPresent()) {
225        currentWorkers.compute(worker.get(), (serverName,
226            availableWorker) -> availableWorker == null ? maxSplitTasks - 1 : availableWorker - 1);
227      }
228      return worker;
229    }
230
231    public synchronized void release(ServerName serverName) {
232      currentWorkers.compute(serverName, (k, v) -> v == null ? null : v + 1);
233    }
234
235    public void suspend(Procedure<?> proc) {
236      event.suspend();
237      event.suspendIfNotReady(proc);
238    }
239
240    public void wake(MasterProcedureScheduler scheduler) {
241      if (!event.isReady()) {
242        event.wake(scheduler);
243      }
244    }
245
246    @Override
247    public void serverAdded(ServerName worker) {
248      this.wake(master.getMasterProcedureExecutor().getEnvironment().getProcedureScheduler());
249    }
250
251    public synchronized void addUsedWorker(ServerName worker) {
252      // load used worker when master restart
253      currentWorkers.compute(worker, (serverName,
254          availableWorker) -> availableWorker == null ? maxSplitTasks - 1 : availableWorker - 1);
255    }
256  }
