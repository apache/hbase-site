057/**
058 * This is a helper class used internally to manage the namespace metadata that is stored in
059 * TableName.NAMESPACE_TABLE_NAME. It also mirrors updates to the ZK store by forwarding updates to
060 * {@link org.apache.hadoop.hbase.ZKNamespaceManager}.
061 *
062 * WARNING: Do not use. Go via the higher-level {@link ClusterSchema} API instead. This manager
063 * is likely to go aways anyways.
064 */
065@InterfaceAudience.Private
066@edu.umd.cs.findbugs.annotations.SuppressWarnings(value="IS2_INCONSISTENT_SYNC",
067  justification="TODO: synchronize access on nsTable but it is done in tiers above and this " +
068    "class is going away/shrinking")
069public class TableNamespaceManager {
070  private static final Log LOG = LogFactory.getLog(TableNamespaceManager.class);
071
072  private Configuration conf;
073  private MasterServices masterServices;
074  private Table nsTable = null; // FindBugs: IS2_INCONSISTENT_SYNC TODO: Access is not synchronized
075  private ZKNamespaceManager zkNamespaceManager;
076  private boolean initialized;
077
078  public static final String KEY_MAX_REGIONS = "hbase.namespace.quota.maxregions";
079  public static final String KEY_MAX_TABLES = "hbase.namespace.quota.maxtables";
080  static final String NS_INIT_TIMEOUT = "hbase.master.namespace.init.timeout";
081  static final int DEFAULT_NS_INIT_TIMEOUT = 300000;
082
083  TableNamespaceManager(MasterServices masterServices) {
084    this.masterServices = masterServices;
085    this.conf = masterServices.getConfiguration();
086  }
087
088  public void start() throws IOException {
089    if (!MetaTableAccessor.tableExists(masterServices.getConnection(),
090        TableName.NAMESPACE_TABLE_NAME)) {
091      LOG.info("Namespace table not found. Creating...");
092      createNamespaceTable(masterServices);
093    }
094
095    try {
096      // Wait for the namespace table to be initialized.
097      long startTime = EnvironmentEdgeManager.currentTime();
098      int timeout = conf.getInt(NS_INIT_TIMEOUT, DEFAULT_NS_INIT_TIMEOUT);
099      while (!isTableAvailableAndInitialized()) {
100        if (EnvironmentEdgeManager.currentTime() - startTime + 100 > timeout) {
101          // We can't do anything if ns is not online.
102          throw new IOException("Timedout " + timeout + "ms waiting for namespace table to "
103              + "be assigned and enabled: " + getTableState());
104        }
105        Thread.sleep(100);
106      }
107    } catch (InterruptedException e) {
108      throw (InterruptedIOException) new InterruptedIOException().initCause(e);
109    }
110  }
111
112  private synchronized Table getNamespaceTable() throws IOException {
113    if (!isTableNamespaceManagerInitialized()) {
114      throw new IOException(this.getClass().getName() + " isn't ready to serve");
115    }
116    return nsTable;
117  }
118
119  /*
120   * check whether a namespace has already existed.
121   */
122  public boolean doesNamespaceExist(final String namespaceName) throws IOException {
123    if (nsTable == null) {
124      throw new IOException(this.getClass().getName() + " isn't ready to serve");
125    }
126    return (get(nsTable, namespaceName) != null);
127  }
128
129  public synchronized NamespaceDescriptor get(String name) throws IOException {
130    if (!isTableNamespaceManagerInitialized()) {
131      return null;
132    }
133    return zkNamespaceManager.get(name);
134  }
135
136  private NamespaceDescriptor get(Table table, String name) throws IOException {
137    Result res = table.get(new Get(Bytes.toBytes(name)));
138    if (res.isEmpty()) {
139      return null;
140    }
141    byte[] val = CellUtil.cloneValue(res.getColumnLatestCell(
142        HTableDescriptor.NAMESPACE_FAMILY_INFO_BYTES, HTableDescriptor.NAMESPACE_COL_DESC_BYTES));
143    return
144        ProtobufUtil.toNamespaceDescriptor(
145            HBaseProtos.NamespaceDescriptor.parseFrom(val));
146  }
147
148  public void insertIntoNSTable(final NamespaceDescriptor ns) throws IOException {
149    if (nsTable == null) {
150      throw new IOException(this.getClass().getName() + " isn't ready to serve");
151    }
152    Put p = new Put(Bytes.toBytes(ns.getName()));
153    p.addImmutable(HTableDescriptor.NAMESPACE_FAMILY_INFO_BYTES,
154        HTableDescriptor.NAMESPACE_COL_DESC_BYTES,
155        ProtobufUtil.toProtoNamespaceDescriptor(ns).toByteArray());
156    nsTable.put(p);
157  }
158
159  public void updateZKNamespaceManager(final NamespaceDescriptor ns) throws IOException {
160    try {
161      zkNamespaceManager.update(ns);
162    } catch (IOException ex) {
163      String msg = "Failed to update namespace information in ZK.";
164      LOG.error(msg, ex);
165      throw new IOException(msg, ex);
166    }
167  }
168
169  public void removeFromNSTable(final String namespaceName) throws IOException {
170    if (nsTable == null) {
171      throw new IOException(this.getClass().getName() + " isn't ready to serve");
172    }
173    Delete d = new Delete(Bytes.toBytes(namespaceName));
174    nsTable.delete(d);
175  }
176
177  public void removeFromZKNamespaceManager(final String namespaceName) throws IOException {
178    zkNamespaceManager.remove(namespaceName);
179  }
180
181  public synchronized NavigableSet<NamespaceDescriptor> list() throws IOException {
182    NavigableSet<NamespaceDescriptor> ret =
183        Sets.newTreeSet(NamespaceDescriptor.NAMESPACE_DESCRIPTOR_COMPARATOR);
184    ResultScanner scanner =
185        getNamespaceTable().getScanner(HTableDescriptor.NAMESPACE_FAMILY_INFO_BYTES);
186    try {
187      for(Result r : scanner) {
188        byte[] val = CellUtil.cloneValue(r.getColumnLatestCell(
189          HTableDescriptor.NAMESPACE_FAMILY_INFO_BYTES,
190          HTableDescriptor.NAMESPACE_COL_DESC_BYTES));
191        ret.add(ProtobufUtil.toNamespaceDescriptor(
192            HBaseProtos.NamespaceDescriptor.parseFrom(val)));
193      }
194    } finally {
195      scanner.close();
196    }
197    return ret;
198  }
199
200  private void createNamespaceTable(MasterServices masterServices) throws IOException {
201    masterServices.createSystemTable(HTableDescriptor.NAMESPACE_TABLEDESC);
202  }
203
204  @SuppressWarnings("deprecation")
205  private boolean isTableNamespaceManagerInitialized() throws IOException {
206    if (initialized) {
207      this.nsTable = this.masterServices.getConnection().getTable(TableName.NAMESPACE_TABLE_NAME);
208      return true;
209    }
210    return false;
211  }

213  /**
214   * Create Namespace in a blocking manner. Keeps trying until
215   * {@link ClusterSchema#HBASE_MASTER_CLUSTER_SCHEMA_OPERATION_TIMEOUT_KEY} expires.
216   * Note, by-passes notifying coprocessors and name checks. Use for system namespaces only.
217   */
218  private void blockingCreateNamespace(final NamespaceDescriptor namespaceDescriptor)
219      throws IOException {
220    ClusterSchema clusterSchema = this.masterServices.getClusterSchema();
221    long procId = clusterSchema.createNamespace(namespaceDescriptor, null);
222    block(this.masterServices, procId);
223  }
224
225
226  /**
227   * An ugly utility to be removed when refactor TableNamespaceManager.
228   * @throws TimeoutIOException
229   */
230  private static void block(final MasterServices services, final long procId)
231  throws TimeoutIOException {
232    int timeoutInMillis = services.getConfiguration().
233        getInt(ClusterSchema.HBASE_MASTER_CLUSTER_SCHEMA_OPERATION_TIMEOUT_KEY,
234            ClusterSchema.DEFAULT_HBASE_MASTER_CLUSTER_SCHEMA_OPERATION_TIMEOUT);
235    long deadlineTs = EnvironmentEdgeManager.currentTime() + timeoutInMillis;
236    ProcedureExecutor<MasterProcedureEnv> procedureExecutor =
237        services.getMasterProcedureExecutor();
238    while(EnvironmentEdgeManager.currentTime() < deadlineTs) {
239      if (procedureExecutor.isFinished(procId)) return;
240      // Sleep some
241      Threads.sleep(10);
242    }
243    throw new TimeoutIOException("Procedure pid=" + procId + " is still running");
244  }

246  /**
247   * This method checks if the namespace table is assigned and then
248   * tries to create its Table reference. If it was already created before, it also makes
249   * sure that the connection isn't closed.
250   * @return true if the namespace table manager is ready to serve, false otherwise
251   */
252  @SuppressWarnings("deprecation")
253  public synchronized boolean isTableAvailableAndInitialized()
254  throws IOException {
255    // Did we already get a table? If so, still make sure it's available
256    if (isTableNamespaceManagerInitialized()) {
257      return true;
258    }
259
260    // Now check if the table is assigned, if not then fail fast
261    if (isTableAssigned() && isTableEnabled()) {
262      try {
263        boolean initGoodSofar = true;
264        nsTable = this.masterServices.getConnection().getTable(TableName.NAMESPACE_TABLE_NAME);
265        zkNamespaceManager = new ZKNamespaceManager(masterServices.getZooKeeper());
266        zkNamespaceManager.start();
267
268        if (get(nsTable, NamespaceDescriptor.DEFAULT_NAMESPACE.getName()) == null) {
269          blockingCreateNamespace(NamespaceDescriptor.DEFAULT_NAMESPACE);
270        }
271        if (get(nsTable, NamespaceDescriptor.SYSTEM_NAMESPACE.getName()) == null) {
272          blockingCreateNamespace(NamespaceDescriptor.SYSTEM_NAMESPACE);
273        }
274
275        if (!initGoodSofar) {
276          // some required namespace is created asynchronized. We should complete init later.
277          return false;
278        }
279
280        ResultScanner scanner = nsTable.getScanner(HTableDescriptor.NAMESPACE_FAMILY_INFO_BYTES);
281        try {
282          for (Result result : scanner) {
283            byte[] val =  CellUtil.cloneValue(result.getColumnLatestCell(
284                HTableDescriptor.NAMESPACE_FAMILY_INFO_BYTES,
285                HTableDescriptor.NAMESPACE_COL_DESC_BYTES));
286            NamespaceDescriptor ns =
287                ProtobufUtil.toNamespaceDescriptor(
288                    HBaseProtos.NamespaceDescriptor.parseFrom(val));
289            zkNamespaceManager.update(ns);
290          }
291        } finally {
292          scanner.close();
293        }
294        initialized = true;
295        return true;
296      } catch (IOException ie) {
297        LOG.warn("Caught exception in initializing namespace table manager", ie);
298        if (nsTable != null) {
299          nsTable.close();
300        }
301        throw ie;
302      }
303    }
304    return false;
305  }

307  private TableState.State getTableState() throws IOException {
308    return masterServices.getTableStateManager().getTableState(TableName.NAMESPACE_TABLE_NAME);
309  }
310
311  private boolean isTableEnabled() throws IOException {
312    return getTableState().equals(TableState.State.ENABLED);
313  }
314
315  private boolean isTableAssigned() {
316    // TODO: we have a better way now (wait on event)
317    return masterServices.getAssignmentManager()
318        .getRegionStates().hasTableRegionStates(TableName.NAMESPACE_TABLE_NAME);
319  }

321  public void validateTableAndRegionCount(NamespaceDescriptor desc) throws IOException {
322    if (getMaxRegions(desc) <= 0) {
323      throw new ConstraintException("The max region quota for " + desc.getName()
324          + " is less than or equal to zero.");
325    }
326    if (getMaxTables(desc) <= 0) {
327      throw new ConstraintException("The max tables quota for " + desc.getName()
328          + " is less than or equal to zero.");
329    }
330  }
331
332  public static long getMaxTables(NamespaceDescriptor ns) throws IOException {
333    String value = ns.getConfigurationValue(KEY_MAX_TABLES);
334    long maxTables = 0;
335    if (StringUtils.isNotEmpty(value)) {
336      try {
337        maxTables = Long.parseLong(value);
338      } catch (NumberFormatException exp) {
339        throw new DoNotRetryIOException("NumberFormatException while getting max tables.", exp);
340      }
341    } else {
342      // The property is not set, so assume its the max long value.
343      maxTables = Long.MAX_VALUE;
344    }
345    return maxTables;
346  }
347
348  public static long getMaxRegions(NamespaceDescriptor ns) throws IOException {
349    String value = ns.getConfigurationValue(KEY_MAX_REGIONS);
350    long maxRegions = 0;
351    if (StringUtils.isNotEmpty(value)) {
352      try {
353        maxRegions = Long.parseLong(value);
354      } catch (NumberFormatException exp) {
355        throw new DoNotRetryIOException("NumberFormatException while getting max regions.", exp);
356      }
357    } else {
358      // The property is not set, so assume its the max long value.
359      maxRegions = Long.MAX_VALUE;
360    }
361    return maxRegions;
362  }
