238/**
239 * HMaster is the "master server" for HBase. An HBase cluster has one active
240 * master.  If many masters are started, all compete.  Whichever wins goes on to
241 * run the cluster.  All others park themselves in their constructor until
242 * master or cluster shutdown or until the active master loses its lease in
243 * zookeeper.  Thereafter, all running master jostle to take over master role.
244 *
245 * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
246 * this case it will tell all regionservers to go down and then wait on them
247 * all reporting in that they are down.  This master will then shut itself down.
248 *
249 * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
250 *
251 * @see org.apache.zookeeper.Watcher
252 */
253@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)

255public class HMaster extends HRegionServer implements MasterServices {

258  /**
259   * Protection against zombie master. Started once Master accepts active responsibility and
260   * starts taking over responsibilities. Allows a finite time window before giving up ownership.
261   */
262  private static class InitializationMonitor extends HasThread {
263    /** The amount of time in milliseconds to sleep before checking initialization status. */
264    public static final String TIMEOUT_KEY = "hbase.master.initializationmonitor.timeout";
265    public static final long TIMEOUT_DEFAULT = TimeUnit.MILLISECONDS.convert(15, TimeUnit.MINUTES);
266
267    /**
268     * When timeout expired and initialization has not complete, call {@link System#exit(int)} when
269     * true, do nothing otherwise.
270     */
271    public static final String HALT_KEY = "hbase.master.initializationmonitor.haltontimeout";
272    public static final boolean HALT_DEFAULT = false;
273
274    private final HMaster master;
275    private final long timeout;
276    private final boolean haltOnTimeout;
277
278    /** Creates a Thread that monitors the {@link #isInitialized()} state. */
279    InitializationMonitor(HMaster master) {
280      super("MasterInitializationMonitor");
281      this.master = master;
282      this.timeout = master.getConfiguration().getLong(TIMEOUT_KEY, TIMEOUT_DEFAULT);
283      this.haltOnTimeout = master.getConfiguration().getBoolean(HALT_KEY, HALT_DEFAULT);
284      this.setDaemon(true);
285    }
286
287    @Override
288    public void run() {
289      try {
290        while (!master.isStopped() && master.isActiveMaster()) {
291          Thread.sleep(timeout);
292          if (master.isInitialized()) {
293            LOG.debug("Initialization completed within allotted tolerance. Monitor exiting.");
294          } else {
295            LOG.error("Master failed to complete initialization after " + timeout + "ms. Please"
296                + " consider submitting a bug report including a thread dump of this process.");
297            if (haltOnTimeout) {
298              LOG.error("Zombie Master exiting. Thread dump to stdout");
299              Threads.printThreadInfo(System.out, "Zombie HMaster");
300              System.exit(-1);
301            }
302          }
303        }
304      } catch (InterruptedException ie) {
305        LOG.trace("InitMonitor thread interrupted. Existing.");
306      }
307    }
308  }

526          ", hbase.cluster.distributed=" + this.conf.getBoolean(HConstants.CLUSTER_DISTRIBUTED, false));
527
528      // Disable usage of meta replicas in the master
529      this.conf.setBoolean(HConstants.USE_META_REPLICAS, false);
530
531      decorateMasterConfiguration(this.conf);
532
533      // Hack! Maps DFSClient => Master for logs.  HDFS made this
534      // config param for task trackers, but we can piggyback off of it.
535      if (this.conf.get("mapreduce.task.attempt.id") == null) {
536        this.conf.set("mapreduce.task.attempt.id", "hb_m_" + this.serverName.toString());
537      }
538
539      // should we check the compression codec type at master side, default true, HBASE-6370
540      this.masterCheckCompression = conf.getBoolean("hbase.master.check.compression", true);
541
542      // should we check encryption settings at master side, default true
543      this.masterCheckEncryption = conf.getBoolean("hbase.master.check.encryption", true);

550      this.maxBlancingTime = getMaxBalancingTime();
551      this.maxRitPercent = conf.getDouble(HConstants.HBASE_MASTER_BALANCER_MAX_RIT_PERCENT,
552          HConstants.DEFAULT_HBASE_MASTER_BALANCER_MAX_RIT_PERCENT);
553
554      // Do we publish the status?
555
556      boolean shouldPublish = conf.getBoolean(HConstants.STATUS_PUBLISHED,
557          HConstants.STATUS_PUBLISHED_DEFAULT);
558      Class<? extends ClusterStatusPublisher.Publisher> publisherClass =
559          conf.getClass(ClusterStatusPublisher.STATUS_PUBLISHER_CLASS,
560              ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS,
561              ClusterStatusPublisher.Publisher.class);
562
563      if (shouldPublish) {
564        if (publisherClass == null) {
565          LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
566              ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS +
567              " is not set - not publishing status");
568        } else {
569          clusterStatusPublisherChore = new ClusterStatusPublisher(this, conf, publisherClass);
570          getChoreService().scheduleChore(clusterStatusPublisherChore);
571        }
572      }
573
574      // Some unit tests don't need a cluster, so no zookeeper at all
575      if (!conf.getBoolean("hbase.testing.nocluster", false)) {
576        this.activeMasterManager = new ActiveMasterManager(zooKeeper, this.serverName, this);
577      } else {
578        this.activeMasterManager = null;
579      }
580    } catch (Throwable t) {
581      // Make sure we log the exception. HMaster is often started via reflection and the
582      // cause of failed startup is lost.
583      LOG.error("Failed construction of Master", t);
584      throw t;
585    }

593  // Main run loop. Calls through to the regionserver run loop AFTER becoming active Master; will
594  // block in here until then.
595  @Override
596  public void run() {
597    try {
598      if (!conf.getBoolean("hbase.testing.nocluster", false)) {
599        Threads.setDaemonThreadRunning(new Thread(() -> {
600          try {
601            int infoPort = putUpJettyServer();
602            startActiveMasterManager(infoPort);
603          } catch (Throwable t) {
604            // Make sure we log the exception.
605            String error = "Failed to become Active Master";
606            LOG.error(error, t);
607            // Abort should have been called already.
608            if (!isAborted()) {
609              abort(error, t);
610            }
611          }
612        }), getName() + ":becomeActiveMaster");
613      }
614      // Fall in here even if we have been aborted. Need to run the shutdown services and
615      // the super run call will do this for us.
616      super.run();
617    } finally {
618      if (this.clusterSchemaService != null) {
619        // If on way out, then we are no longer active master.
620        this.clusterSchemaService.stopAsync();
621        try {
622          this.clusterSchemaService.awaitTerminated(
623              getConfiguration().getInt(HBASE_MASTER_WAIT_ON_SERVICE_IN_SECONDS,
624              DEFAULT_HBASE_MASTER_WAIT_ON_SERVICE_IN_SECONDS), TimeUnit.SECONDS);
625        } catch (TimeoutException te) {
626          LOG.warn("Failed shutdown of clusterSchemaService", te);
627        }
628      }
629      this.activeMaster = false;
630    }
631  }

633  // return the actual infoPort, -1 means disable info server.
634  private int putUpJettyServer() throws IOException {
635    if (!conf.getBoolean("hbase.master.infoserver.redirect", true)) {
636      return -1;
637    }
638    final int infoPort = conf.getInt("hbase.master.info.port.orig",
639      HConstants.DEFAULT_MASTER_INFOPORT);
640    // -1 is for disabling info server, so no redirecting
641    if (infoPort < 0 || infoServer == null) {
642      return -1;
643    }
644    if(infoPort == infoServer.getPort()) {
645      return infoPort;
646    }
647    final String addr = conf.get("hbase.master.info.bindAddress", "0.0.0.0");
648    if (!Addressing.isLocalAddress(InetAddress.getByName(addr))) {
649      String msg =
650          "Failed to start redirecting jetty server. Address " + addr
651              + " does not belong to this host. Correct configuration parameter: "
652              + "hbase.master.info.bindAddress";
653      LOG.error(msg);
654      throw new IOException(msg);
655    }
656
657    // TODO I'm pretty sure we could just add another binding to the InfoServer run by
658    // the RegionServer and have it run the RedirectServlet instead of standing up
659    // a second entire stack here.
660    masterJettyServer = new Server();
661    final ServerConnector connector = new ServerConnector(masterJettyServer);
662    connector.setHost(addr);
663    connector.setPort(infoPort);
664    masterJettyServer.addConnector(connector);
665    masterJettyServer.setStopAtShutdown(true);
666
667    final String redirectHostname =
668        StringUtils.isBlank(useThisHostnameInstead) ? null : useThisHostnameInstead;
669
670    final RedirectServlet redirect = new RedirectServlet(infoServer, redirectHostname);
671    final WebAppContext context = new WebAppContext(null, "/", null, null, null, null, WebAppContext.NO_SESSIONS);
672    context.addServlet(new ServletHolder(redirect), "/*");
673    context.setServer(masterJettyServer);
674
675    try {
676      masterJettyServer.start();
677    } catch (Exception e) {
678      throw new IOException("Failed to start redirecting jetty server", e);
679    }
680    return connector.getLocalPort();
681  }

830  private static final ImmutableSet<Class<? extends Procedure>> UNSUPPORTED_PROCEDURES =
831    ImmutableSet.of(RecoverMetaProcedure.class, AssignProcedure.class, UnassignProcedure.class,
832      MoveRegionProcedure.class);
833
834  /**
835   * In HBASE-20811, we have introduced a new TRSP to assign/unassign/move regions, and it is
836   * incompatible with the old AssignProcedure/UnassignProcedure/MoveRegionProcedure. So we need to
837   * make sure that there are none these procedures when upgrading. If there are, the master will
838   * quit, you need to go back to the old version to finish these procedures first before upgrading.
839   */
840  private void checkUnsupportedProcedure(
841      Map<Class<? extends Procedure>, List<Procedure<MasterProcedureEnv>>> procsByType)
842      throws HBaseIOException {
843    // Confirm that we do not have unfinished assign/unassign related procedures. It is not easy to
844    // support both the old assign/unassign procedures and the new TransitRegionStateProcedure as
845    // there will be conflict in the code for AM. We should finish all these procedures before
846    // upgrading.
847    for (Class<? extends Procedure> clazz : UNSUPPORTED_PROCEDURES) {
848      List<Procedure<MasterProcedureEnv>> procs = procsByType.get(clazz);
849      if (procs != null) {
850        LOG.error(
851          "Unsupported procedure type {} found, please rollback your master to the old" +
852            " version to finish them, and then try to upgrade again. The full procedure list: {}",
853          clazz, procs);
854        throw new HBaseIOException("Unsupported procedure type " + clazz + " found");
855      }
856    }
857    // A special check for SCP, as we do not support RecoverMetaProcedure any more so we need to
858    // make sure that no one will try to schedule it but SCP does have a state which will schedule
859    // it.
860    if (procsByType.getOrDefault(ServerCrashProcedure.class, Collections.emptyList()).stream()
861      .map(p -> (ServerCrashProcedure) p).anyMatch(ServerCrashProcedure::isInRecoverMetaState)) {
862      LOG.error("At least one ServerCrashProcedure is going to schedule a RecoverMetaProcedure," +
863        " which is not supported any more. Please rollback your master to the old version to" +
864        " finish them, and then try to upgrade again.");
865      throw new HBaseIOException("Unsupported procedure state found for ServerCrashProcedure");
866    }
867  }

875  /**
876   * Finish initialization of HMaster after becoming the primary master.
877   * <p/>
878   * The startup order is a bit complicated but very important, do not change it unless you know
879   * what you are doing.
880   * <ol>
881   * <li>Initialize file system based components - file system manager, wal manager, table
882   * descriptors, etc</li>
883   * <li>Publish cluster id</li>
884   * <li>Here comes the most complicated part - initialize server manager, assignment manager and
885   * region server tracker
886   * <ol type='i'>
887   * <li>Create server manager</li>
888   * <li>Create procedure executor, load the procedures, but do not start workers. We will start it
889   * later after we finish scheduling SCPs to avoid scheduling duplicated SCPs for the same
890   * server</li>
891   * <li>Create assignment manager and start it, load the meta region state, but do not load data
892   * from meta region</li>
893   * <li>Start region server tracker, construct the online servers set and find out dead servers and
894   * schedule SCP for them. The online servers will be constructed by scanning zk, and we will also
895   * scan the wal directory to find out possible live region servers, and the differences between
896   * these two sets are the dead servers</li>
897   * </ol>
898   * </li>
899   * <li>If this is a new deploy, schedule a InitMetaProcedure to initialize meta</li>
900   * <li>Start necessary service threads - balancer, catalog janior, executor services, and also the
901   * procedure executor, etc. Notice that the balancer must be created first as assignment manager
902   * may use it when assigning regions.</li>
903   * <li>Wait for meta to be initialized if necesssary, start table state manager.</li>
904   * <li>Wait for enough region servers to check-in</li>
905   * <li>Let assignment manager load data from meta and construct region states</li>
906   * <li>Start all other things such as chore services, etc</li>
907   * </ol>
908   * <p/>
909   * Notice that now we will not schedule a special procedure to make meta online(unless the first
910   * time where meta has not been created yet), we will rely on SCP to bring meta online.
911   */
912  private void finishActiveMasterInitialization(MonitoredTask status) throws IOException,
913          InterruptedException, KeeperException, ReplicationException {
914    /*
915     * We are active master now... go initialize components we need to run.
916     */
917    status.setStatus("Initializing Master file system");
918
919    this.masterActiveTime = System.currentTimeMillis();
920    // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
921
922    // Only initialize the MemStoreLAB when master carry table
923    if (LoadBalancer.isTablesOnMaster(conf)) {
924      initializeMemStoreChunkCreator();
925    }
926    this.fileSystemManager = new MasterFileSystem(conf);
927    this.walManager = new MasterWalManager(this);
928
929    // enable table descriptors cache
930    this.tableDescriptors.setCacheOn();
931
932    // warm-up HTDs cache on master initialization
933    if (preLoadTableDescriptors) {
934      status.setStatus("Pre-loading table descriptors");
935      this.tableDescriptors.getAll();
936    }
937
938    // Publish cluster ID; set it in Master too. The superclass RegionServer does this later but
939    // only after it has checked in with the Master. At least a few tests ask Master for clusterId
940    // before it has called its run method and before RegionServer has done the reportForDuty.
941    ClusterId clusterId = fileSystemManager.getClusterId();
942    status.setStatus("Publishing Cluster ID " + clusterId + " in ZooKeeper");
943    ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
944    this.clusterId = clusterId.toString();
945
946    // Precaution. Put in place the old hbck1 lock file to fence out old hbase1s running their
947    // hbck1s against an hbase2 cluster; it could do damage. To skip this behavior, set
948    // hbase.write.hbck1.lock.file to false.
949    if (this.conf.getBoolean("hbase.write.hbck1.lock.file", true)) {
950      HBaseFsck.checkAndMarkRunningHbck(this.conf,
951          HBaseFsck.createLockRetryCounterFactory(this.conf).create());
952    }
953
954    status.setStatus("Initialize ServerManager and schedule SCP for crash servers");
955    // The below two managers must be created before loading procedures, as they will be used during
956    // loading.
957    this.serverManager = createServerManager(this);
958    this.syncReplicationReplayWALManager = new SyncReplicationReplayWALManager(this);
959    if (!conf.getBoolean(HBASE_SPLIT_WAL_COORDINATED_BY_ZK,
960      DEFAULT_HBASE_SPLIT_COORDINATED_BY_ZK)) {
961      this.splitWALManager = new SplitWALManager(this);
962    }
963    createProcedureExecutor();
964    @SuppressWarnings("rawtypes")
965    Map<Class<? extends Procedure>, List<Procedure<MasterProcedureEnv>>> procsByType =
966      procedureExecutor.getActiveProceduresNoCopy().stream()
967        .collect(Collectors.groupingBy(p -> p.getClass()));
968
969    checkUnsupportedProcedure(procsByType);
970
971    // Create Assignment Manager
972    this.assignmentManager = createAssignmentManager(this);
973    this.assignmentManager.start();
974    // TODO: TRSP can perform as the sub procedure for other procedures, so even if it is marked as
975    // completed, it could still be in the procedure list. This is a bit strange but is another
976    // story, need to verify the implementation for ProcedureExecutor and ProcedureStore.
977    List<TransitRegionStateProcedure> ritList =
978      procsByType.getOrDefault(TransitRegionStateProcedure.class, Collections.emptyList()).stream()
979        .filter(p -> !p.isFinished()).map(p -> (TransitRegionStateProcedure) p)
980        .collect(Collectors.toList());
981    this.assignmentManager.setupRIT(ritList);
982
983    // Start RegionServerTracker with listing of servers found with exiting SCPs -- these should
984    // be registered in the deadServers set -- and with the list of servernames out on the
985    // filesystem that COULD BE 'alive' (we'll schedule SCPs for each and let SCP figure it out).
986    // We also pass dirs that are already 'splitting'... so we can do some checks down in tracker.
987    // TODO: Generate the splitting and live Set in one pass instead of two as we currently do.
988    this.regionServerTracker = new RegionServerTracker(zooKeeper, this, this.serverManager);
989    this.regionServerTracker.start(
990      procsByType.getOrDefault(ServerCrashProcedure.class, Collections.emptyList()).stream()
991        .map(p -> (ServerCrashProcedure) p).map(p -> p.getServerName()).collect(Collectors.toSet()),
992      walManager.getLiveServersFromWALDir(), walManager.getSplittingServersFromWALDir());
993    // This manager will be started AFTER hbase:meta is confirmed on line.
994    // hbase.mirror.table.state.to.zookeeper is so hbase1 clients can connect. They read table
995    // state from zookeeper while hbase2 reads it from hbase:meta. Disable if no hbase1 clients.
996    this.tableStateManager =
997      this.conf.getBoolean(MirroringTableStateManager.MIRROR_TABLE_STATE_TO_ZK_KEY, true)
998        ?
999        new MirroringTableStateManager(this):
1000        new TableStateManager(this);
1001
1002    status.setStatus("Initializing ZK system trackers");
1003    initializeZKBasedSystemTrackers();
1004    status.setStatus("Loading last flushed sequence id of regions");
1005    try {
1006      this.serverManager.loadLastFlushedSequenceIds();
1007    } catch (IOException e) {
1008      LOG.debug("Failed to load last flushed sequence id of regions"
1009          + " from file system", e);
1010    }
1011    // Set ourselves as active Master now our claim has succeeded up in zk.
1012    this.activeMaster = true;
1013
1014    // Start the Zombie master detector after setting master as active, see HBASE-21535
1015    Thread zombieDetector = new Thread(new InitializationMonitor(this),
1016        "ActiveMasterInitializationMonitor-" + System.currentTimeMillis());
1017    zombieDetector.setDaemon(true);
1018    zombieDetector.start();
1019
1020    // This is for backwards compatibility
1021    // See HBASE-11393
1022    status.setStatus("Update TableCFs node in ZNode");
1023    ReplicationPeerConfigUpgrader tableCFsUpdater =
1024        new ReplicationPeerConfigUpgrader(zooKeeper, conf);
1025    tableCFsUpdater.copyTableCFs();
1026
1027    if (!maintenanceMode) {
1028      // Add the Observer to delete quotas on table deletion before starting all CPs by
1029      // default with quota support, avoiding if user specifically asks to not load this Observer.
1030      if (QuotaUtil.isQuotaEnabled(conf)) {
1031        updateConfigurationForQuotasObserver(conf);
1032      }
1033      // initialize master side coprocessors before we start handling requests
1034      status.setStatus("Initializing master coprocessors");
1035      this.cpHost = new MasterCoprocessorHost(this, this.conf);
1036    }
1037
1038    // Checking if meta needs initializing.
1039    status.setStatus("Initializing meta table if this is a new deploy");
1040    InitMetaProcedure initMetaProc = null;
1041    // Print out state of hbase:meta on startup; helps debugging.
1042    RegionState rs = this.assignmentManager.getRegionStates().
1043        getRegionState(RegionInfoBuilder.FIRST_META_REGIONINFO);
1044    LOG.info("hbase:meta {}", rs);
1045    if (rs.isOffline()) {
1046      Optional<InitMetaProcedure> optProc = procedureExecutor.getProcedures().stream()
1047        .filter(p -> p instanceof InitMetaProcedure).map(o -> (InitMetaProcedure) o).findAny();
1048      initMetaProc = optProc.orElseGet(() -> {
1049        // schedule an init meta procedure if meta has not been deployed yet
1050        InitMetaProcedure temp = new InitMetaProcedure();
1051        procedureExecutor.submitProcedure(temp);
1052        return temp;
1053      });
1054    }
1055    if (this.balancer instanceof FavoredNodesPromoter) {
1056      favoredNodesManager = new FavoredNodesManager(this);
1057    }
1058
1059    // initialize load balancer
1060    this.balancer.setMasterServices(this);
1061    this.balancer.setClusterMetrics(getClusterMetricsWithoutCoprocessor());
1062    this.balancer.initialize();
1063
1064    // start up all service threads.
1065    status.setStatus("Initializing master service threads");
1066    startServiceThreads();
1067    // wait meta to be initialized after we start procedure executor
1068    if (initMetaProc != null) {
1069      initMetaProc.await();
1070    }
1071    // Wake up this server to check in
1072    sleeper.skipSleepCycle();
1073
1074    // Wait for region servers to report in.
1075    // With this as part of master initialization, it precludes our being able to start a single
1076    // server that is both Master and RegionServer. Needs more thought. TODO.
1077    String statusStr = "Wait for region servers to report in";
1078    status.setStatus(statusStr);
1079    LOG.info(Objects.toString(status));
1080    waitForRegionServers(status);
1081
1082    // Check if master is shutting down because issue initializing regionservers or balancer.
1083    if (isStopped()) {
1084      return;
1085    }

1087    status.setStatus("Starting assignment manager");
1088    // FIRST HBASE:META READ!!!!
1089    // The below cannot make progress w/o hbase:meta being online.
1090    // This is the FIRST attempt at going to hbase:meta. Meta on-lining is going on in background
1091    // as procedures run -- in particular SCPs for crashed servers... One should put up hbase:meta
1092    // if it is down. It may take a while to come online. So, wait here until meta if for sure
1093    // available. That's what waitForMetaOnline does.
1094    if (!waitForMetaOnline()) {
1095      return;
1096    }

1099    this.tableStateManager.start();

1101    // is being started over an hbase-1.x dataset. tablestatemanager runs a migration as part
1102    // of its 'start' moving table state from zookeeper to hbase:meta. This migration needs to
1103    // complete before we do this next step processing offline regions else it fails reading
1104    // table states messing up master launch (namespace table, etc., are not assigned).
1105    this.assignmentManager.processOfflineRegions();

1107    if (favoredNodesManager != null && !maintenanceMode) {
1108      SnapshotOfRegionAssignmentFromMeta snapshotOfRegionAssignment =
1109          new SnapshotOfRegionAssignmentFromMeta(getConnection());
1110      snapshotOfRegionAssignment.initialize();

1112    }

1114    // set cluster status again after user regions are assigned
1115    this.balancer.setClusterMetrics(getClusterMetricsWithoutCoprocessor());
1116
1117    // Start balancer and meta catalog janitor after meta and regions have been assigned.
1118    status.setStatus("Starting balancer and catalog janitor");
1119    this.clusterStatusChore = new ClusterStatusChore(this, balancer);
1120    getChoreService().scheduleChore(clusterStatusChore);
1121    this.balancerChore = new BalancerChore(this);
1122    getChoreService().scheduleChore(balancerChore);
1123    this.normalizerChore = new RegionNormalizerChore(this);
1124    getChoreService().scheduleChore(normalizerChore);
1125    this.catalogJanitorChore = new CatalogJanitor(this);
1126    getChoreService().scheduleChore(catalogJanitorChore);
1127    this.serverManager.startChore();
1128
1129    // Only for rolling upgrade, where we need to migrate the data in namespace table to meta table.
1130    if (!waitForNamespaceOnline()) {
1131      return;
1132    }
1133    status.setStatus("Starting cluster schema service");
1134    initClusterSchemaService();
1135
1136    if (this.cpHost != null) {
1137      try {
1138        this.cpHost.preMasterInitialization();
1139      } catch (IOException e) {
1140        LOG.error("Coprocessor preMasterInitialization() hook failed", e);
1141      }
1142    }
1143
1144    status.markComplete("Initialization successful");

1146       (System.currentTimeMillis() - masterActiveTime) / 1000.0f));
1147    this.masterFinishedInitializationTime = System.currentTimeMillis();
1148    configurationManager.registerObserver(this.balancer);
1149    configurationManager.registerObserver(this.hfileCleaner);
1150    configurationManager.registerObserver(this.logCleaner);
1151    // Set master as 'initialized'.
1152    setInitialized(true);
1153
1154    if (maintenanceMode) {
1155      LOG.info("Detected repair mode, skipping final initialization steps.");
1156      return;
1157    }
1158
1159    assignmentManager.checkIfShouldMoveSystemRegionAsync();
1160    status.setStatus("Assign meta replicas");
1161    MasterMetaBootstrap metaBootstrap = createMetaBootstrap();
1162    metaBootstrap.assignMetaReplicas();
1163    status.setStatus("Starting quota manager");
1164    initQuotaManager();
1165    if (QuotaUtil.isQuotaEnabled(conf)) {
1166      // Create the quota snapshot notifier
1167      spaceQuotaSnapshotNotifier = createQuotaSnapshotNotifier();
1168      spaceQuotaSnapshotNotifier.initialize(getClusterConnection());
1169      this.quotaObserverChore = new QuotaObserverChore(this, getMasterMetrics());
1170      // Start the chore to read the region FS space reports and act on them
1171      getChoreService().scheduleChore(quotaObserverChore);
1172
1173      this.snapshotQuotaChore = new SnapshotQuotaObserverChore(this, getMasterMetrics());
1174      // Start the chore to read snapshots and add their usage to table/NS quotas
1175      getChoreService().scheduleChore(snapshotQuotaChore);
1176    }
1177
1178    // clear the dead servers with same host name and port of online server because we are not
1179    // removing dead server with same hostname and port of rs which is trying to check in before
1180    // master initialization. See HBASE-5916.
1181    this.serverManager.clearDeadServersWithSameHostNameAndPortOfOnlineServer();
1182
1183    // Check and set the znode ACLs if needed in case we are overtaking a non-secure configuration
1184    status.setStatus("Checking ZNode ACLs");
1185    zooKeeper.checkAndSetZNodeAcls();
1186
1187    status.setStatus("Initializing MOB Cleaner");
1188    initMobCleaner();
1189

1192      // don't let cp initialization errors kill the master

1194        this.cpHost.postStartMaster();
1195      } catch (IOException ioe) {
1196        LOG.error("Coprocessor postStartMaster() hook failed", ioe);
1197      }
1198    }

1201
1202    /*
1203     * After master has started up, lets do balancer post startup initialization. Since this runs
1204     * in activeMasterManager thread, it should be fine.
1205     */
1206    long start = System.currentTimeMillis();
1207    this.balancer.postMasterStartupInitialize();
1208    if (LOG.isDebugEnabled()) {
1209      LOG.debug("Balancer post startup initialization complete, took " + (
1210          (System.currentTimeMillis() - start) / 1000) + " seconds");
1211    }

1224  /**
1225   * @return True if region is online and scannable else false if an error or shutdown (Otherwise
1226   *   we just block in here holding up all forward-progess).
1227   */
1228  private boolean isRegionOnline(RegionInfo ri) throws InterruptedException {
1229    RetryCounter rc = null;
1230    while (!isStopped()) {
1231      RegionState rs = this.assignmentManager.getRegionStates().getRegionState(ri);
1232      if (rs.isOpened()) {
1233        if (this.getServerManager().isServerOnline(rs.getServerName())) {
1234          return true;
1235        }
1236      }
1237      // Region is not OPEN.
1238      Optional<Procedure<MasterProcedureEnv>> optProc = this.procedureExecutor.getProcedures().
1239          stream().filter(p -> p instanceof ServerCrashProcedure).findAny();
1240      // TODO: Add a page to refguide on how to do repair. Have this log message point to it.
1241      // Page will talk about loss of edits, how to schedule at least the meta WAL recovery, and
1242      // then how to assign including how to break region lock if one held.
1243      LOG.warn("{} is NOT online; state={}; ServerCrashProcedures={}. Master startup cannot " +
1244          "progress, in holding-pattern until region onlined.",
1245          ri.getRegionNameAsString(), rs, optProc.isPresent());
1246      // Check once-a-minute.
1247      if (rc == null) {
1248        rc = new RetryCounterFactory(1000).create();
1249      }
1250      Threads.sleep(rc.getBackoffTimeAndIncrementAttempts());
1251    }
1252    return false;
1253  }

1257   * <p/>
1258   * This is for rolling upgrading, later we will migrate the data in ns table to the ns family of
1259   * meta table. And if this is a new clsuter, this method will return immediately as there will be
1260   * no namespace table/region.

1265      MetaTableAccessor.getTableState(getClusterConnection(), TableName.NAMESPACE_TABLE_NAME);

1267      // this means we have already migrated the data and disabled or deleted the namespace table,

1269      return true;

1272      this.assignmentManager.getRegionStates().getRegionsOfTable(TableName.NAMESPACE_TABLE_NAME);

1639  /**
1640   * It first sleep to the next balance plan start time. Meanwhile, throttling by the max
1641   * number regions in transition to protect availability.
1642   * @param nextBalanceStartTime The next balance plan start time
1643   * @param maxRegionsInTransition max number of regions in transition
1644   * @param cutoffTime when to exit balancer
1645   */
1646  private void balanceThrottling(long nextBalanceStartTime, int maxRegionsInTransition,
1647      long cutoffTime) {
1648    boolean interrupted = false;
1649
1650    // Sleep to next balance plan start time
1651    // But if there are zero regions in transition, it can skip sleep to speed up.
1652    while (!interrupted && System.currentTimeMillis() < nextBalanceStartTime
1653        && this.assignmentManager.getRegionStates().hasRegionsInTransition()) {
1654      try {
1655        Thread.sleep(100);
1656      } catch (InterruptedException ie) {
1657        interrupted = true;
1658      }
1659    }
1660
1661    // Throttling by max number regions in transition
1662    while (!interrupted
1663        && maxRegionsInTransition > 0

1681  public boolean balance(boolean force) throws IOException {
1682    // if master not initialized, don't run balancer.
1683    if (!isInitialized()) {
1684      LOG.debug("Master has not been initialized, don't run balancer.");
1685      return false;
1686    }
1687
1688    if (isInMaintenanceMode()) {
1689      LOG.info("Master is in maintenanceMode mode, don't run balancer.");
1690      return false;
1691    }
1692
1693    int maxRegionsInTransition = getMaxRegionsInTransition();
1694    synchronized (this.balancer) {
1695      // If balance not true, don't run balancer.
1696      if (!this.loadBalancerTracker.isBalancerOn()) return false;
1697        // Only allow one balance run at at time.
1698      if (this.assignmentManager.hasRegionsInTransition()) {
1699        List<RegionStateNode> regionsInTransition = assignmentManager.getRegionsInTransition();
1700        // if hbase:meta region is in transition, result of assignment cannot be recorded
1701        // ignore the force flag in that case
1702        boolean metaInTransition = assignmentManager.isMetaRegionInTransition();
1703        String prefix = force && !metaInTransition ? "R" : "Not r";
1704        List<RegionStateNode> toPrint = regionsInTransition;
1705        int max = 5;
1706        boolean truncated = false;
1707        if (regionsInTransition.size() > max) {
1708          toPrint = regionsInTransition.subList(0, max);
1709          truncated = true;
1710        }
1711        LOG.info(prefix + "unning balancer because " + regionsInTransition.size() +
1712          " region(s) in transition: " + toPrint + (truncated? "(truncated list)": ""));
1713        if (!force || metaInTransition) return false;
1714      }
1715      if (this.serverManager.areDeadServersInProgress()) {
1716        LOG.info("Not running balancer because processing dead regionserver(s): " +
1717          this.serverManager.getDeadServers());
1718        return false;
1719      }
1720
1721      if (this.cpHost != null) {
1722        try {
1723          if (this.cpHost.preBalance()) {
1724            LOG.debug("Coprocessor bypassing balancer request");
1725            return false;
1726          }
1727        } catch (IOException ioe) {
1728          LOG.error("Error invoking master coprocessor preBalance()", ioe);
1729          return false;
1730        }
1731      }
1732
1733      boolean isByTable = getConfiguration().getBoolean("hbase.master.loadbalance.bytable", false);
1734      Map<TableName, Map<ServerName, List<RegionInfo>>> assignmentsByTable =
1735        this.assignmentManager.getRegionStates().getAssignmentsByTable(!isByTable);
1736
1737      List<RegionPlan> plans = new ArrayList<>();
1738
1739      //Give the balancer the current cluster state.
1740      this.balancer.setClusterMetrics(getClusterMetricsWithoutCoprocessor());
1741      this.balancer.setClusterLoad(assignmentsByTable);
1742
1743      for (Map<ServerName, List<RegionInfo>> serverMap : assignmentsByTable.values()) {
1744        serverMap.keySet().removeAll(this.serverManager.getDrainingServersList());
1745      }
1746      for (Entry<TableName, Map<ServerName, List<RegionInfo>>> e : assignmentsByTable.entrySet()) {
1747        List<RegionPlan> partialPlans = this.balancer.balanceCluster(e.getKey(), e.getValue());
1748        if (partialPlans != null) plans.addAll(partialPlans);
1749      }
1750
1751      long balanceStartTime = System.currentTimeMillis();
1752      long cutoffTime = balanceStartTime + this.maxBlancingTime;
1753      int rpCount = 0;  // number of RegionPlans balanced so far
1754      if (plans != null && !plans.isEmpty()) {
1755        int balanceInterval = this.maxBlancingTime / plans.size();
1756        LOG.info("Balancer plans size is " + plans.size() + ", the balance interval is "
1757            + balanceInterval + " ms, and the max number regions in transition is "
1758            + maxRegionsInTransition);
1759
1760        for (RegionPlan plan: plans) {
1761          LOG.info("balance " + plan);
1762          //TODO: bulk assign
1763          try {
1764            this.assignmentManager.moveAsync(plan);
1765          } catch (HBaseIOException hioe) {
1766            //should ignore failed plans here, avoiding the whole balance plans be aborted
1767            //later calls of balance() can fetch up the failed and skipped plans
1768            LOG.warn("Failed balance plan: {}, just skip it", plan, hioe);
1769          }
1770          //rpCount records balance plans processed, does not care if a plan succeeds
1771          rpCount++;
1772
1773          balanceThrottling(balanceStartTime + rpCount * balanceInterval, maxRegionsInTransition,
1774            cutoffTime);
1775
1776          // if performing next balance exceeds cutoff time, exit the loop
1777          if (rpCount < plans.size() && System.currentTimeMillis() > cutoffTime) {
1778            // TODO: After balance, there should not be a cutoff time (keeping it as
1779            // a security net for now)
1780            LOG.debug("No more balancing till next balance run; maxBalanceTime="
1781                + this.maxBlancingTime);
1782            break;
1783          }
1784        }
1785      }
1786
1787      if (this.cpHost != null) {
1788        try {
1789          this.cpHost.postBalance(rpCount < plans.size() ? plans.subList(0, rpCount) : plans);
1790        } catch (IOException ioe) {
1791          // balancing already succeeded so don't change the result
1792          LOG.error("Error invoking master coprocessor postBalance()", ioe);
1793        }
1794      }
1795    }
1796    // If LoadBalancer did not generate any plans, it means the cluster is already balanced.
1797    // Return true indicating a success.
1798    return true;
1799  }

1808   * Perform normalization of cluster (invoked by {@link RegionNormalizerChore}).
1809   *
1810   * @return true if normalization step was performed successfully, false otherwise
1811   *    (specifically, if HMaster hasn't been initialized properly or normalization
1812   *    is globally disabled)
1813   */
1814  public boolean normalizeRegions() throws IOException {
1815    if (!isInitialized()) {
1816      LOG.debug("Master has not been initialized, don't run region normalizer.");
1817      return false;
1818    }
1819    if (this.getServerManager().isClusterShutdown()) {
1820      LOG.info("Cluster is shutting down, don't run region normalizer.");
1821      return false;
1822    }
1823    if (isInMaintenanceMode()) {
1824      LOG.info("Master is in maintenance mode, don't run region normalizer.");
1825      return false;
1826    }
1827    if (!this.regionNormalizerTracker.isNormalizerOn()) {
1828      LOG.debug("Region normalization is disabled, don't run region normalizer.");
1829      return false;
1830    }
1831
1832    synchronized (this.normalizer) {
1833      // Don't run the normalizer concurrently
1834      List<TableName> allEnabledTables = new ArrayList<>(
1835        this.tableStateManager.getTablesInStates(TableState.State.ENABLED));
1836
1837      Collections.shuffle(allEnabledTables);
1838
1839      for (TableName table : allEnabledTables) {
1840        if (isInMaintenanceMode()) {
1841          LOG.debug("Master is in maintenance mode, stop running region normalizer.");
1842          return false;
1843        }
1844
1845        TableDescriptor tblDesc = getTableDescriptors().get(table);
1846        if (table.isSystemTable() || (tblDesc != null &&
1847            !tblDesc.isNormalizationEnabled())) {
1848          LOG.trace("Skipping normalization for {}, as it's either system"
1849              + " table or doesn't have auto normalization turned on", table);
1850          continue;
1851        }
1852        List<NormalizationPlan> plans = this.normalizer.computePlanForTable(table);
1853        if (plans != null) {
1854          for (NormalizationPlan plan : plans) {
1855            plan.execute(clusterConnection.getAdmin());
1856            if (plan.getType() == PlanType.SPLIT) {
1857              splitPlanCount++;
1858            } else if (plan.getType() == PlanType.MERGE) {
1859              mergePlanCount++;
1860            }
1861          }
1862        }
1863      }
1864    }
1865    // If Region did not generate any plans, it means the cluster is already balanced.
1866    // Return true indicating a success.
1867    return true;
1868  }

2260  private void checkCompactionPolicy(Configuration conf, TableDescriptor htd)
2261      throws IOException {
2262    // FIFO compaction has some requirements
2263    // Actually FCP ignores periodic major compactions
2264    String className = htd.getValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
2265    if (className == null) {
2266      className =
2267          conf.get(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
2268            ExploringCompactionPolicy.class.getName());
2269    }
2270
2271    int blockingFileCount = HStore.DEFAULT_BLOCKING_STOREFILE_COUNT;
2272    String sv = htd.getValue(HStore.BLOCKING_STOREFILES_KEY);
2273    if (sv != null) {
2274      blockingFileCount = Integer.parseInt(sv);
2275    } else {
2276      blockingFileCount = conf.getInt(HStore.BLOCKING_STOREFILES_KEY, blockingFileCount);
2277    }
2278
2279    for (ColumnFamilyDescriptor hcd : htd.getColumnFamilies()) {
2280      String compactionPolicy =
2281          hcd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
2282      if (compactionPolicy == null) {
2283        compactionPolicy = className;
2284      }
2285      if (!compactionPolicy.equals(FIFOCompactionPolicy.class.getName())) {
2286        continue;
2287      }
2288      // FIFOCompaction
2289      String message = null;
2290
2291      // 1. Check TTL
2292      if (hcd.getTimeToLive() == ColumnFamilyDescriptorBuilder.DEFAULT_TTL) {
2293        message = "Default TTL is not supported for FIFO compaction";
2294        throw new IOException(message);
2295      }
2296
2297      // 2. Check min versions
2298      if (hcd.getMinVersions() > 0) {
2299        message = "MIN_VERSION > 0 is not supported for FIFO compaction";
2300        throw new IOException(message);
2301      }
2302
2303      // 3. blocking file count
2304      sv = hcd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
2305      if (sv != null) {
2306        blockingFileCount = Integer.parseInt(sv);
2307      }
2308      if (blockingFileCount < 1000) {
2309        message =
2310            "Blocking file count '" + HStore.BLOCKING_STOREFILES_KEY + "' " + blockingFileCount
2311                + " is below recommended minimum of 1000 for column family "+ hcd.getNameAsString();
2312        throw new IOException(message);
2313      }
2314    }
2315  }

2337  private void startActiveMasterManager(int infoPort) throws KeeperException {
2338    String backupZNode = ZNodePaths.joinZNode(
2339      zooKeeper.getZNodePaths().backupMasterAddressesZNode, serverName.toString());
2340    /*
2341    * Add a ZNode for ourselves in the backup master directory since we
2342    * may not become the active master. If so, we want the actual active
2343    * master to know we are backup masters, so that it won't assign
2344    * regions to us if so configured.
2345    *
2346    * If we become the active master later, ActiveMasterManager will delete
2347    * this node explicitly.  If we crash before then, ZooKeeper will delete
2348    * this node for us since it is ephemeral.
2349    */
2350    LOG.info("Adding backup master ZNode " + backupZNode);
2351    if (!MasterAddressTracker.setMasterAddress(zooKeeper, backupZNode, serverName, infoPort)) {
2352      LOG.warn("Failed create of " + backupZNode + " by " + serverName);
2353    }
2354    this.activeMasterManager.setInfoPort(infoPort);
2355    int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT, HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
2356    // If we're a backup master, stall until a primary to write this address
2357    if (conf.getBoolean(HConstants.MASTER_TYPE_BACKUP, HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
2358      LOG.debug("HMaster started in backup mode. Stalling until master znode is written.");
2359      // This will only be a minute or so while the cluster starts up,
2360      // so don't worry about setting watches on the parent znode
2361      while (!activeMasterManager.hasActiveMaster()) {
2362        LOG.debug("Waiting for master address and cluster state znode to be written.");
2363        Threads.sleep(timeout);
2364      }
2365    }
2366    MonitoredTask status = TaskMonitor.get().createStatus("Master startup");
2367    status.setDescription("Master startup");
2368    try {
2369      if (activeMasterManager.blockUntilBecomingActiveMaster(timeout, status)) {
2370        finishActiveMasterInitialization(status);
2371      }
2372    } catch (Throwable t) {
2373      status.setStatus("Failed to become active: " + t.getMessage());
2374      LOG.error(HBaseMarkers.FATAL, "Failed to become active master", t);
2375      // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
2376      if (t instanceof NoClassDefFoundError && t.getMessage().
2377          contains("org/apache/hadoop/hdfs/protocol/HdfsConstants$SafeModeAction")) {
2378        // improved error message for this special case
2379        abort("HBase is having a problem with its Hadoop jars.  You may need to recompile " +
2380          "HBase against Hadoop version " + org.apache.hadoop.util.VersionInfo.getVersion() +
2381          " or change your hadoop jars to start properly", t);
2382      } else {
2383        abort("Unhandled exception. Starting shutdown.", t);
2384      }
2385    } finally {
2386      status.cleanup();
2387    }
2388  }

3012  /**
3013   * Shutdown the cluster.
3014   * Master runs a coordinated stop of all RegionServers and then itself.
3015   */
3016  public void shutdown() throws IOException {
3017    if (cpHost != null) {
3018      cpHost.preShutdown();
3019    }
3020    // Tell the servermanager cluster shutdown has been called. This makes it so when Master is
3021    // last running server, it'll stop itself. Next, we broadcast the cluster shutdown by setting
3022    // the cluster status as down. RegionServers will notice this change in state and will start
3023    // shutting themselves down. When last has exited, Master can go down.
3024    if (this.serverManager != null) {
3025      this.serverManager.shutdownCluster();
3026    }
3027    if (this.clusterStatusTracker != null) {
3028      try {
3029        this.clusterStatusTracker.setClusterDown();
3030      } catch (KeeperException e) {
3031        LOG.error("ZooKeeper exception trying to set cluster as down in ZK", e);
3032      }
3033    }
3034    // Stop the procedure executor. Will stop any ongoing assign, unassign, server crash etc.,
3035    // processing so we can go down.
3036    if (this.procedureExecutor != null) {
3037      this.procedureExecutor.stop();
3038    }
3039    // Shutdown our cluster connection. This will kill any hosted RPCs that might be going on;
3040    // this is what we want especially if the Master is in startup phase doing call outs to
3041    // hbase:meta, etc. when cluster is down. Without ths connection close, we'd have to wait on
3042    // the rpc to timeout.
3043    if (this.clusterConnection != null) {
3044      this.clusterConnection.close();
3045    }
3046  }

3502      final String namespace, final String regex, final List<TableName> tableNameList,

3782  /**
3783   * Mark region server(s) as decommissioned (previously called 'draining') to prevent additional
3784   * regions from getting assigned to them. Also unload the regions on the servers asynchronously.0
3785   * @param servers Region servers to decommission.
3786   */
3787  public void decommissionRegionServers(final List<ServerName> servers, final boolean offload)
3788      throws HBaseIOException {
3789    List<ServerName> serversAdded = new ArrayList<>(servers.size());
3790    // Place the decommission marker first.
3791    String parentZnode = getZooKeeper().getZNodePaths().drainingZNode;
3792    for (ServerName server : servers) {
3793      try {
3794        String node = ZNodePaths.joinZNode(parentZnode, server.getServerName());
3795        ZKUtil.createAndFailSilent(getZooKeeper(), node);
3796      } catch (KeeperException ke) {
3797        throw new HBaseIOException(
3798          this.zooKeeper.prefix("Unable to decommission '" + server.getServerName() + "'."), ke);
3799      }
3800      if (this.serverManager.addServerToDrainList(server)) {
3801        serversAdded.add(server);
3802      }
3803    }
3804    // Move the regions off the decommissioned servers.
3805    if (offload) {
3806      final List<ServerName> destServers = this.serverManager.createDestinationServersList();
3807      for (ServerName server : serversAdded) {
3808        final List<RegionInfo> regionsOnServer = this.assignmentManager.getRegionsOnServer(server);
3809        for (RegionInfo hri : regionsOnServer) {
3810          ServerName dest = balancer.randomAssignment(hri, destServers);
3811          if (dest == null) {
3812            throw new HBaseIOException("Unable to determine a plan to move " + hri);
3813          }
3814          RegionPlan rp = new RegionPlan(hri, server, dest);
3815          this.assignmentManager.moveAsync(rp);
3816        }
3817      }
3818    }
3819  }

3830  /**
3831   * Remove decommission marker (previously called 'draining') from a region server to allow regions
3832   * assignments. Load regions onto the server asynchronously if a list of regions is given
3833   * @param server Region server to remove decommission marker from.
3834   */
3835  public void recommissionRegionServer(final ServerName server,
3836      final List<byte[]> encodedRegionNames) throws IOException {
3837    // Remove the server from decommissioned (draining) server list.
3838    String parentZnode = getZooKeeper().getZNodePaths().drainingZNode;
3839    String node = ZNodePaths.joinZNode(parentZnode, server.getServerName());
3840    try {
3841      ZKUtil.deleteNodeFailSilent(getZooKeeper(), node);
3842    } catch (KeeperException ke) {
3843      throw new HBaseIOException(
3844        this.zooKeeper.prefix("Unable to recommission '" + server.getServerName() + "'."), ke);
3845    }
3846    this.serverManager.removeServerFromDrainList(server);
3847
3848    // Load the regions onto the server if we are given a list of regions.
3849    if (encodedRegionNames == null || encodedRegionNames.isEmpty()) {
3850      return;
3851    }
3852    if (!this.serverManager.isServerOnline(server)) {
3853      return;
3854    }
3855    for (byte[] encodedRegionName : encodedRegionNames) {
3856      RegionState regionState =
3857        assignmentManager.getRegionStates().getRegionState(Bytes.toString(encodedRegionName));
3858      if (regionState == null) {
3859        LOG.warn("Unknown region " + Bytes.toStringBinary(encodedRegionName));
3860        continue;
3861      }
3862      RegionInfo hri = regionState.getRegion();
3863      if (server.equals(regionState.getServerName())) {
3864        LOG.info("Skipping move of region " + hri.getRegionNameAsString() +
3865          " because region already assigned to the same server " + server + ".");
3866        continue;
3867      }
3868      RegionPlan rp = new RegionPlan(hri, regionState.getServerName(), server);
3869      this.assignmentManager.moveAsync(rp);
3870    }
3871  }
