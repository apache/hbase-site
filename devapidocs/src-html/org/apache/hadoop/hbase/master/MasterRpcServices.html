498  enum BalanceSwitchMode {
499    SYNC,
500    ASYNC
501  }
502
503  /**
504   * Assigns balancer switch according to BalanceSwitchMode
505   * @param b new balancer switch
506   * @param mode BalanceSwitchMode
507   * @return old balancer switch
508   */
509  boolean switchBalancer(final boolean b, BalanceSwitchMode mode) throws IOException {
510    boolean oldValue = master.loadBalancerTracker.isBalancerOn();
511    boolean newValue = b;
512    try {
513      if (master.cpHost != null) {
514        master.cpHost.preBalanceSwitch(newValue);
515      }
516      try {
517        if (mode == BalanceSwitchMode.SYNC) {
518          synchronized (master.getLoadBalancer()) {
519            master.loadBalancerTracker.setBalancerOn(newValue);
520          }
521        } else {
522          master.loadBalancerTracker.setBalancerOn(newValue);
523        }
524      } catch (KeeperException ke) {
525        throw new IOException(ke);
526      }
527      LOG.info(master.getClientIdAuditPrefix() + " set balanceSwitch=" + newValue);
528      if (master.cpHost != null) {
529        master.cpHost.postBalanceSwitch(oldValue, newValue);
530      }
531      master.getLoadBalancer().updateBalancerStatus(newValue);
532    } catch (IOException ioe) {
533      LOG.warn("Error flipping balance switch", ioe);
534    }
535    return oldValue;
536  }

538  boolean synchronousBalanceSwitch(final boolean b) throws IOException {
539    return switchBalancer(b, BalanceSwitchMode.SYNC);

591      ClusterStatusProtos.ServerLoad sl = request.getLoad();
592      ServerName serverName = ProtobufUtil.toServerName(request.getServer());

594      ServerMetrics newLoad =
595        ServerMetricsBuilder.toServerMetrics(serverName, versionNumber, version, sl);

597      master.getAssignmentManager().reportOnlineRegions(serverName,
598        newLoad.getRegionMetrics().keySet());
599      if (sl != null && master.metricsMaster != null) {
600        // Up our metrics.
601        master.metricsMaster.incrementRequests(
602          sl.getTotalNumberOfRequests() - (oldLoad != null ? oldLoad.getRequestCount() : 0));
603      }
604    } catch (IOException ioe) {
605      throw new ServiceException(ioe);
606    }
607    return RegionServerReportResponse.newBuilder().build();

1958    rpcPreCheck("setNormalizerRunning");

1960    // Sets normalizer on/off flag in ZK.
1961    // TODO: this method is totally broken in terms of atomicity of actions and values read.
1962    //  1. The contract has this RPC returning the previous value. There isn't a ZKUtil method
1963    //     that lets us retrieve the previous value as part of setting a new value, so we simply
1964    //     perform a read before issuing the update. Thus we have a data race opportunity, between
