446  /**
447   * Assigns balancer switch according to BalanceSwitchMode
448   * @param b new balancer switch
449   * @param mode BalanceSwitchMode
450   * @return old balancer switch
451   */
452  boolean switchBalancer(final boolean b, BalanceSwitchMode mode) throws IOException {
453    boolean oldValue = master.loadBalancerTracker.isBalancerOn();
454    boolean newValue = b;
455    try {
456      if (master.cpHost != null) {
457        master.cpHost.preBalanceSwitch(newValue);
458      }
459      try {
460        if (mode == BalanceSwitchMode.SYNC) {
461          synchronized (master.getLoadBalancer()) {
462            master.loadBalancerTracker.setBalancerOn(newValue);
463          }
464        } else {
465          master.loadBalancerTracker.setBalancerOn(newValue);
466        }
467      } catch (KeeperException ke) {
468        throw new IOException(ke);
469      }
470      LOG.info(master.getClientIdAuditPrefix() + " set balanceSwitch=" + newValue);
471      if (master.cpHost != null) {
472        master.cpHost.postBalanceSwitch(oldValue, newValue);
473      }
474      master.getLoadBalancer().updateBalancerStatus(newValue);
475    } catch (IOException ioe) {
476      LOG.warn("Error flipping balance switch", ioe);
477    }
478    return oldValue;
479  }

896        (message) -> {

902      if (execController.getFailedOn() != null) {

1798  /**
1799   * Compacts the mob files in the current table.
1800   * @param request the request.
1801   * @param tableName the current table name.
1802   * @return The response of the mob file compaction.
1803   * @throws IOException
1804   */
1805  private CompactRegionResponse compactMob(final CompactRegionRequest request,
1806    TableName tableName) throws IOException {
1807    if (!master.getTableStateManager().isTableState(tableName, TableState.State.ENABLED)) {
1808      throw new DoNotRetryIOException("Table " + tableName + " is not enabled");
1809    }
1810    boolean allFiles = false;
1811    List<ColumnFamilyDescriptor> compactedColumns = new ArrayList<>();
1812    ColumnFamilyDescriptor[] hcds = master.getTableDescriptors().get(tableName).getColumnFamilies();
1813    byte[] family = null;
1814    if (request.hasFamily()) {
1815      family = request.getFamily().toByteArray();
1816      for (ColumnFamilyDescriptor hcd : hcds) {
1817        if (Bytes.equals(family, hcd.getName())) {
1818          if (!hcd.isMobEnabled()) {
1819            LOG.error("Column family " + hcd.getNameAsString() + " is not a mob column family");
1820            throw new DoNotRetryIOException("Column family " + hcd.getNameAsString()
1821                    + " is not a mob column family");
1822          }
1823          compactedColumns.add(hcd);
1824        }
1825      }
1826    } else {
1827      for (ColumnFamilyDescriptor hcd : hcds) {
1828        if (hcd.isMobEnabled()) {
1829          compactedColumns.add(hcd);
1830        }
1831      }
1832    }
1833    if (compactedColumns.isEmpty()) {
1834      LOG.error("No mob column families are assigned in the mob compaction");
1835      throw new DoNotRetryIOException(
1836              "No mob column families are assigned in the mob compaction");
1837    }
1838    if (request.hasMajor() && request.getMajor()) {
1839      allFiles = true;
1840    }
1841    String familyLogMsg = (family != null) ? Bytes.toString(family) : "";
1842    if (LOG.isTraceEnabled()) {
1843      LOG.trace("User-triggered mob compaction requested for table: "
1844              + tableName.getNameAsString() + " for column family: " + familyLogMsg);
1845    }
1846    master.requestMobCompaction(tableName, compactedColumns, allFiles);
1847    return CompactRegionResponse.newBuilder().build();
1848  }

2766        if (permissionType == Type.Table) {
2767          boolean filter = (cf != null || userName != null) ? true : false;
2768          perms = PermissionStorage.getUserTablePermissions(master.getConfiguration(), table, cf,
2769            cq, userName, filter);
2770        } else if (permissionType == Type.Namespace) {
2771          perms = PermissionStorage.getUserNamespacePermissions(master.getConfiguration(),
2772            namespace, userName, userName != null ? true : false);
2773        } else {
2774          perms = PermissionStorage.getUserPermissions(master.getConfiguration(), null, null, null,
2775            userName, userName != null ? true : false);
2776          // Skip super users when filter user is specified
2777          if (userName == null) {
2778            // Adding superusers explicitly to the result set as PermissionStorage do not store
2779            // them. Also using acl as table name to be inline with the results of global admin and
2780            // will help in avoiding any leakage of information about being superusers.
2781            for (String user : Superusers.getSuperUsers()) {
2782              perms.add(new UserPermission(user,
2783                  Permission.newBuilder().withActions(Action.values()).build()));
2784            }
2785          }
2786        }
2787
2788        master.getMasterCoprocessorHost().postGetUserPermissions(userName, namespace, table, cf,
2789          cq);
2790        AccessControlProtos.GetUserPermissionsResponse response =
2791            ShadedAccessControlUtil.buildGetUserPermissionsResponse(perms);
2792        return response;
2793      } else {
2794        throw new DoNotRetryIOException(
2795            new UnsupportedOperationException(AccessController.class.getName() + " is not loaded"));
2796      }
2797    } catch (IOException ioe) {
2798      throw new ServiceException(ioe);
2799    }
2800  }

2804      HasUserPermissionsRequest request) throws ServiceException {
2805    try {
2806      master.checkInitialized();
2807      if (master.cpHost != null && hasAccessControlServiceCoprocessor(master.cpHost)) {
2808        User caller = RpcServer.getRequestUser().orElse(null);
2809        String userName =
2810            request.hasUserName() ? request.getUserName().toStringUtf8() : caller.getShortName();
2811        List<Permission> permissions = new ArrayList<>();
2812        for (int i = 0; i < request.getPermissionCount(); i++) {
2813          permissions.add(ShadedAccessControlUtil.toPermission(request.getPermission(i)));
2814        }
2815        master.getMasterCoprocessorHost().preHasUserPermissions(userName, permissions);
2816        if (!caller.getShortName().equals(userName)) {
2817          List<String> groups = AccessChecker.getUserGroups(userName);
2818          caller = new InputUser(userName, groups.toArray(new String[groups.size()]));
2819        }
2820        List<Boolean> hasUserPermissions = new ArrayList<>();
2821        if (getAccessChecker() != null) {
2822          for (Permission permission : permissions) {
2823            boolean hasUserPermission =
2824                getAccessChecker().hasUserPermission(caller, "hasUserPermissions", permission);
2825            hasUserPermissions.add(hasUserPermission);
2826          }
2827        } else {
2828          for (int i = 0; i < permissions.size(); i++) {
2829            hasUserPermissions.add(true);
2830          }
2831        }
2832        master.getMasterCoprocessorHost().postHasUserPermissions(userName, permissions);
2833        HasUserPermissionsResponse.Builder builder =
2834            HasUserPermissionsResponse.newBuilder().addAllHasUserPermission(hasUserPermissions);
2835        return builder.build();
2836      } else {
2837        throw new DoNotRetryIOException(
2838            new UnsupportedOperationException(AccessController.class.getName() + " is not loaded"));
2839      }
2840    } catch (IOException ioe) {
2841      throw new ServiceException(ioe);

2860  private boolean shouldSubmitSCP(ServerName serverName) {
2861    // check if there is already a SCP of this server running
2862    List<Procedure<MasterProcedureEnv>> procedures =
2863        master.getMasterProcedureExecutor().getProcedures();
2864    for (Procedure<MasterProcedureEnv> procedure : procedures) {
2865      if (procedure instanceof ServerCrashProcedure) {
2866        if (serverName.compareTo(((ServerCrashProcedure) procedure).getServerName()) == 0
2867            && !procedure.isFinished()) {
2868          LOG.info("there is already a SCP of this server {} running, pid {}", serverName,
2869            procedure.getProcId());
2870          return false;
2871        }
2872      }
2873    }
2874    return true;
