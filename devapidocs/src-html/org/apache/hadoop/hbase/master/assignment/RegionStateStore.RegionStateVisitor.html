083  public interface RegionStateVisitor {
084    void visitRegionState(RegionInfo regionInfo, State state,
085      ServerName regionLocation, ServerName lastHost, long openSeqNum);
086  }
087
088  public void visitMeta(final RegionStateVisitor visitor) throws IOException {
089    MetaTableAccessor.fullScanRegions(master.getConnection(), new MetaTableAccessor.Visitor() {
090      final boolean isDebugEnabled = LOG.isDebugEnabled();
091
092      @Override
093      public boolean visit(final Result r) throws IOException {
094        if (r !=  null && !r.isEmpty()) {
095          long st = 0;
096          if (LOG.isTraceEnabled()) {
097            st = System.currentTimeMillis();
098          }
099          visitMetaEntry(visitor, r);
100          if (LOG.isTraceEnabled()) {
101            long et = System.currentTimeMillis();
102            LOG.trace("[T] LOAD META PERF " + StringUtils.humanTimeDiff(et - st));
103          }
104        } else if (isDebugEnabled) {
105          LOG.debug("NULL result from meta - ignoring but this is strange.");
106        }
107        return true;
108      }
109    });
110  }
111
112  private void visitMetaEntry(final RegionStateVisitor visitor, final Result result)
113      throws IOException {
114    final RegionLocations rl = MetaTableAccessor.getRegionLocations(result);
115    if (rl == null) return;
116
117    final HRegionLocation[] locations = rl.getRegionLocations();
118    if (locations == null) return;
119
120    for (int i = 0; i < locations.length; ++i) {
121      final HRegionLocation hrl = locations[i];
122      if (hrl == null) continue;
123
124      final RegionInfo regionInfo = hrl.getRegionInfo();
125      if (regionInfo == null) continue;
126
127      final int replicaId = regionInfo.getReplicaId();
128      final State state = getRegionState(result, replicaId);
129
130      final ServerName lastHost = hrl.getServerName();
131      final ServerName regionLocation = getRegionServer(result, replicaId);
132      final long openSeqNum = -1;
133
134      // TODO: move under trace, now is visible for debugging
135      LOG.info(String.format("Load hbase:meta entry region=%s regionState=%s lastHost=%s regionLocation=%s",
136        regionInfo, state, lastHost, regionLocation));
137
138      visitor.visitRegionState(regionInfo, state, regionLocation, lastHost, openSeqNum);
139    }
140  }
141
142  public void updateRegionLocation(RegionStates.RegionStateNode regionStateNode)
143      throws IOException {
144    if (regionStateNode.getRegionInfo().isMetaRegion()) {
145      updateMetaLocation(regionStateNode.getRegionInfo(), regionStateNode.getRegionLocation());
146    } else {
147      long openSeqNum = regionStateNode.getState() == State.OPEN ?
148          regionStateNode.getOpenSeqNum() : HConstants.NO_SEQNUM;
149      updateUserRegionLocation(regionStateNode.getRegionInfo(), regionStateNode.getState(),
150          regionStateNode.getRegionLocation(), regionStateNode.getLastHost(), openSeqNum,
151          regionStateNode.getProcedure().getProcId());
152    }
153  }
154
155  protected void updateMetaLocation(final RegionInfo regionInfo, final ServerName serverName)
156      throws IOException {
157    try {
158      MetaTableLocator.setMetaLocation(master.getZooKeeper(), serverName,
159        regionInfo.getReplicaId(), State.OPEN);
160    } catch (KeeperException e) {
161      throw new IOException(e);
162    }
163  }
164
165  protected void updateUserRegionLocation(final RegionInfo regionInfo, final State state,
166      final ServerName regionLocation, final ServerName lastHost, final long openSeqNum,
167      final long pid)
168      throws IOException {
169    final int replicaId = regionInfo.getReplicaId();
170    final Put put = new Put(MetaTableAccessor.getMetaKeyForRegion(regionInfo));
171    MetaTableAccessor.addRegionInfo(put, regionInfo);
172    final StringBuilder info = new StringBuilder("pid=" + pid + " updating hbase:meta row=");
173    info.append(regionInfo.getRegionNameAsString()).append(", regionState=").append(state);
174    if (openSeqNum >= 0) {
175      Preconditions.checkArgument(state == State.OPEN && regionLocation != null,
176          "Open region should be on a server");
177      MetaTableAccessor.addLocation(put, regionLocation, openSeqNum, -1, replicaId);
178      info.append(", openSeqNum=").append(openSeqNum);
179      info.append(", regionLocation=").append(regionLocation);
180    } else if (regionLocation != null && !regionLocation.equals(lastHost)) {
181      // Ideally, if no regionLocation, write null to the hbase:meta but this will confuse clients
182      // currently; they want a server to hit. TODO: Make clients wait if no location.
183      put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
184          .setRow(put.getRow())
185          .setFamily(HConstants.CATALOG_FAMILY)
186          .setQualifier(getServerNameColumn(replicaId))
187          .setTimestamp(put.getTimeStamp())
188          .setType(DataType.Put)
189          .setValue(Bytes.toBytes(regionLocation.getServerName()))
190          .build());
191      info.append(", regionLocation=").append(regionLocation);
192    }
193    put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY)
194        .setRow(put.getRow())
195        .setFamily(HConstants.CATALOG_FAMILY)
196        .setQualifier(getStateColumn(replicaId))
197        .setTimestamp(put.getTimeStamp())
198        .setType(DataType.Put)
199        .setValue(Bytes.toBytes(state.name()))
200        .build());
201    LOG.info(info);
202
203    final boolean serialReplication = hasSerialReplicationScope(regionInfo.getTable());
204    if (serialReplication && state == State.OPEN) {
205      Put barrierPut = MetaTableAccessor.makeBarrierPut(regionInfo.getEncodedNameAsBytes(),
206          openSeqNum, regionInfo.getTable().getName());
207      updateRegionLocation(regionInfo, state, put, barrierPut);
208    } else {
209      updateRegionLocation(regionInfo, state, put);
210    }
211  }
212
213  protected void updateRegionLocation(final RegionInfo regionInfo, final State state,
214      final Put... put) throws IOException {
215    synchronized (this) {
216      if (multiHConnection == null) {
217        multiHConnection = new MultiHConnection(master.getConfiguration(), 1);
218      }
219    }
220
221    try {
222      multiHConnection.processBatchCallback(Arrays.asList(put), TableName.META_TABLE_NAME, null, null);
223    } catch (IOException e) {
224      // TODO: Revist!!!! Means that if a server is loaded, then we will abort our host!
225      // In tests we abort the Master!
226      String msg = String.format("FAILED persisting region=%s state=%s",
227          regionInfo.getShortNameToLog(), state);
228      LOG.error(msg, e);
229      master.abort(msg, e);
230      throw e;
231    }
232  }

237  public void splitRegion(final RegionInfo parent, final RegionInfo hriA,
238      final RegionInfo hriB, final ServerName serverName)  throws IOException {
239    final TableDescriptor htd = getTableDescriptor(parent.getTable());
240    MetaTableAccessor.splitRegion(master.getConnection(), parent, hriA, hriB, serverName,
241        getRegionReplication(htd), hasSerialReplicationScope(htd));
242  }

244  // ============================================================================================
245  //  Update Region Merging State helpers
246  // ============================================================================================
247  public void mergeRegions(final RegionInfo parent, final RegionInfo hriA,
248      final RegionInfo hriB, final ServerName serverName)  throws IOException {
249    final TableDescriptor htd = getTableDescriptor(parent.getTable());
250    MetaTableAccessor.mergeRegions(master.getConnection(), parent, hriA, hriB, serverName,
251        getRegionReplication(htd), EnvironmentEdgeManager.currentTime(),
252        hasSerialReplicationScope(htd));
253  }

266  // ==========================================================================
267  //  Table Descriptors helpers
268  // ==========================================================================
269  private boolean hasSerialReplicationScope(final TableName tableName) throws IOException {
270    return hasSerialReplicationScope(getTableDescriptor(tableName));
271  }
272
273  private boolean hasSerialReplicationScope(final TableDescriptor htd) {
274    return (htd != null)? htd.hasSerialReplicationScope(): false;
275  }
276
277  private int getRegionReplication(final TableDescriptor htd) {
278    return (htd != null) ? htd.getRegionReplication() : 1;
279  }
280
281  private TableDescriptor getTableDescriptor(final TableName tableName) throws IOException {
282    return master.getTableDescriptors().get(tableName);
283  }

289  /**
290   * Returns the {@link ServerName} from catalog table {@link Result}
291   * where the region is transitioning. It should be the same as
292   * {@link MetaTableAccessor#getServerName(Result,int)} if the server is at OPEN state.
293   * @param r Result to pull the transitioning server name from
294   * @return A ServerName instance or {@link MetaTableAccessor#getServerName(Result,int)}
295   * if necessary fields not found or empty.
296   */
297  static ServerName getRegionServer(final Result r, int replicaId) {
298    final Cell cell = r.getColumnLatestCell(HConstants.CATALOG_FAMILY,
299        getServerNameColumn(replicaId));
300    if (cell == null || cell.getValueLength() == 0) {
301      RegionLocations locations = MetaTableAccessor.getRegionLocations(r);
302      if (locations != null) {
303        HRegionLocation location = locations.getRegionLocation(replicaId);
304        if (location != null) {
305          return location.getServerName();
306        }
307      }
308      return null;
309    }
310    return ServerName.parseServerName(Bytes.toString(cell.getValueArray(),
311      cell.getValueOffset(), cell.getValueLength()));
312  }
313
314  private static byte[] getServerNameColumn(int replicaId) {
315    return replicaId == 0
316        ? HConstants.SERVERNAME_QUALIFIER
317        : Bytes.toBytes(HConstants.SERVERNAME_QUALIFIER_STR + META_REPLICA_ID_DELIMITER
318          + String.format(RegionInfo.REPLICA_ID_FORMAT, replicaId));
319  }

325  /**
326   * Pull the region state from a catalog table {@link Result}.
327   * @param r Result to pull the region state from
328   * @return the region state, or null if unknown.
329   */
330  @VisibleForTesting
331  public static State getRegionState(final Result r, int replicaId) {
332    Cell cell = r.getColumnLatestCell(HConstants.CATALOG_FAMILY, getStateColumn(replicaId));
333    if (cell == null || cell.getValueLength() == 0) {
334      return null;
335    }
336    return State.valueOf(Bytes.toString(cell.getValueArray(), cell.getValueOffset(),
337        cell.getValueLength()));
338  }
339
340  private static byte[] getStateColumn(int replicaId) {
341    return replicaId == 0
342        ? HConstants.STATE_QUALIFIER
343        : Bytes.toBytes(HConstants.STATE_QUALIFIER_STR + META_REPLICA_ID_DELIMITER
344          + String.format(RegionInfo.REPLICA_ID_FORMAT, replicaId));
345  }
