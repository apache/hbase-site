065  static GetRegionInfoResponse getRegionInfoResponse(final MasterProcedureEnv env,
066      final ServerName regionLocation, final RegionInfo hri, boolean includeBestSplitRow)
067      throws IOException {
068    AsyncRegionServerAdmin admin =
069      env.getMasterServices().getAsyncClusterConnection().getRegionServerAdmin(regionLocation);
070    GetRegionInfoRequest request = null;
071    if (includeBestSplitRow) {
072      request = RequestConverter.buildGetRegionInfoRequest(hri.getRegionName(), false, true);
073    } else {
074      request = RequestConverter.buildGetRegionInfoRequest(hri.getRegionName());
075    }
076    return FutureUtils.get(admin.getRegionInfo(request));
077  }

079  private static void lock(List<RegionStateNode> regionNodes) {
080    regionNodes.iterator().forEachRemaining(RegionStateNode::lock);
081  }
082
083  private static void unlock(List<RegionStateNode> regionNodes) {
084    for (ListIterator<RegionStateNode> iter = regionNodes.listIterator(regionNodes.size()); iter
085      .hasPrevious();) {
086      iter.previous().unlock();
087    }
088  }

090  static TransitRegionStateProcedure[] createUnassignProceduresForSplitOrMerge(
091      MasterProcedureEnv env, Stream<RegionInfo> regions, int regionReplication)
092      throws IOException {
093    List<RegionStateNode> regionNodes = regions
094      .flatMap(hri -> IntStream.range(0, regionReplication)
095        .mapToObj(i -> RegionReplicaUtil.getRegionInfoForReplica(hri, i)))
096      .map(env.getAssignmentManager().getRegionStates()::getOrCreateRegionStateNode)
097      .collect(Collectors.toList());
098    TransitRegionStateProcedure[] procs = new TransitRegionStateProcedure[regionNodes.size()];
099    boolean rollback = true;
100    int i = 0;
101    // hold the lock at once, and then release it in finally. This is important as SCP may jump in
102    // if we release the lock in the middle when we want to do rollback, and cause problems.
103    lock(regionNodes);
104    try {
105      for (; i < procs.length; i++) {
106        RegionStateNode regionNode = regionNodes.get(i);
107        TransitRegionStateProcedure proc =
108          TransitRegionStateProcedure.unassign(env, regionNode.getRegionInfo());
109        if (regionNode.getProcedure() != null) {
110          throw new HBaseIOException(
111            "The parent region " + regionNode + " is currently in transition, give up");
112        }
113        regionNode.setProcedure(proc);
114        procs[i] = proc;
115      }
116      // all succeeded, set rollback to false
117      rollback = false;
118    } finally {
119      if (rollback) {
120        for (;;) {
121          i--;
122          if (i < 0) {
123            break;
124          }
125          RegionStateNode regionNode = regionNodes.get(i);
126          regionNode.unsetProcedure(procs[i]);
127        }
128      }
129      unlock(regionNodes);
130    }
131    return procs;
132  }

134  /**
135   * Create assign procedures for the give regions, according to the {@code regionReplication}.
136   * <p/>
137   * For rolling back, we will submit procedures directly to the {@code ProcedureExecutor}, so it is
138   * possible that we persist the newly scheduled procedures, and then crash before persisting the
139   * rollback state, so when we arrive here the second time, it is possible that some regions have
140   * already been associated with a TRSP.
141   * @param ignoreIfInTransition if true, will skip creating TRSP for the given region if it is
142   *          already in transition, otherwise we will add an assert that it should not in
143   *          transition.
144   */
145  private static TransitRegionStateProcedure[] createAssignProcedures(MasterProcedureEnv env,
146      List<RegionInfo> regions, int regionReplication, ServerName targetServer,
147      boolean ignoreIfInTransition) {
148    // create the assign procs only for the primary region using the targetServer
149    TransitRegionStateProcedure[] primaryRegionProcs =
150      regions.stream().map(env.getAssignmentManager().getRegionStates()::getOrCreateRegionStateNode)
151        .map(regionNode -> {
152          TransitRegionStateProcedure proc =
153            TransitRegionStateProcedure.assign(env, regionNode.getRegionInfo(), targetServer);
154          regionNode.lock();
155          try {
156            if (ignoreIfInTransition) {
157              if (regionNode.isInTransition()) {
158                return null;
159              }
160            } else {
161              // should never fail, as we have the exclusive region lock, and the region is newly
162              // created, or has been successfully closed so should not be on any servers, so SCP
163              // will
164              // not process it either.
165              assert !regionNode.isInTransition();
166            }
167            regionNode.setProcedure(proc);
168          } finally {
169            regionNode.unlock();
170          }
171          return proc;
172        }).filter(p -> p != null).toArray(TransitRegionStateProcedure[]::new);
173    if (regionReplication == DEFAULT_REGION_REPLICA) {
174      // this is the default case
175      return primaryRegionProcs;
176    }
177    // collect the replica region infos
178    List<RegionInfo> replicaRegionInfos =
179        new ArrayList<RegionInfo>(regions.size() * (regionReplication - 1));
180    for (RegionInfo hri : regions) {
181      // start the index from 1
182      for (int i = 1; i < regionReplication; i++) {
183        RegionInfo ri = RegionReplicaUtil.getRegionInfoForReplica(hri, i);
184        // apply ignoreRITs to replica regions as well.
185        if (!ignoreIfInTransition || !env.getAssignmentManager().getRegionStates().
186          getOrCreateRegionStateNode(ri).isInTransition()) {
187          replicaRegionInfos.add(ri);
188        }
189      }
190    }
191
192    // create round robin procs. Note that we exclude the primary region's target server
193    TransitRegionStateProcedure[] replicaRegionAssignProcs =
194        env.getAssignmentManager().createRoundRobinAssignProcedures(replicaRegionInfos,
195          Collections.singletonList(targetServer));
196    // combine both the procs and return the result
197    return ArrayUtils.addAll(primaryRegionProcs, replicaRegionAssignProcs);
198  }

200  /**
201   * Create round robin assign procedures for the given regions,
202   * according to the {@code regionReplication}.
203   * <p/>
204   * For rolling back, we will submit procedures directly to the {@code ProcedureExecutor}, so it is
205   * possible that we persist the newly scheduled procedures, and then crash before persisting the
206   * rollback state, so when we arrive here the second time, it is possible that some regions have
207   * already been associated with a TRSP.
208   * @param ignoreIfInTransition if true, will skip creating TRSP for the given region if it is
209   *          already in transition, otherwise we will add an assert that it should not in
210   *          transition.
211   */
212  private static TransitRegionStateProcedure[] createRoundRobinAssignProcedures(
213    MasterProcedureEnv env, List<RegionInfo> regions, int regionReplication,
214    List<ServerName> serversToExclude, boolean ignoreIfInTransition) {
215    List<RegionInfo> regionsAndReplicas = new ArrayList<>(regions);
216    if (regionReplication != DEFAULT_REGION_REPLICA) {
217
218      // collect the replica region infos
219      List<RegionInfo> replicaRegionInfos =
220        new ArrayList<RegionInfo>(regions.size() * (regionReplication - 1));
221      for (RegionInfo hri : regions) {
222        // start the index from 1
223        for (int i = 1; i < regionReplication; i++) {
224          replicaRegionInfos.add(RegionReplicaUtil.getRegionInfoForReplica(hri, i));
225        }
226      }
227      regionsAndReplicas.addAll(replicaRegionInfos);
228    }
229    if (ignoreIfInTransition) {
230      for (RegionInfo region : regionsAndReplicas) {
231        if (env.getAssignmentManager().getRegionStates().getOrCreateRegionStateNode(region)
232          .isInTransition()) {
233          return null;
234        }
235      }
236    }
237    // create round robin procs. Note that we exclude the primary region's target server
238    return env.getAssignmentManager()
239      .createRoundRobinAssignProcedures(regionsAndReplicas, serversToExclude);
240  }

242  static TransitRegionStateProcedure[] createAssignProceduresForSplitDaughters(
243    MasterProcedureEnv env, List<RegionInfo> daughters, int regionReplication,
244    ServerName parentServer) {
245    if(env.getMasterConfiguration().getBoolean(HConstants.HBASE_ENABLE_SEPARATE_CHILD_REGIONS,
246      DEFAULT_HBASE_ENABLE_SEPARATE_CHILD_REGIONS)){
247      // keep one daughter on the parent region server
248      TransitRegionStateProcedure[] daughterOne =
249        createAssignProcedures(env, Collections.singletonList(daughters.get(0)),
250          regionReplication, parentServer, false);
251      // round robin assign the other daughter
252      TransitRegionStateProcedure[] daughterTwo =
253        createRoundRobinAssignProcedures(env, Collections.singletonList(daughters.get(1)),
254          regionReplication, Collections.singletonList(parentServer), false);
255      return ArrayUtils.addAll(daughterOne, daughterTwo);
256    }
257    return createAssignProceduresForOpeningNewRegions(env, daughters, regionReplication,
258      parentServer);
259  }
260
261  static TransitRegionStateProcedure[] createAssignProceduresForOpeningNewRegions(
262      MasterProcedureEnv env, List<RegionInfo> regions, int regionReplication,
263      ServerName targetServer) {
264    return createAssignProcedures(env, regions, regionReplication, targetServer, false);
265  }

267  static void reopenRegionsForRollback(MasterProcedureEnv env, List<RegionInfo> regions,
268      int regionReplication, ServerName targetServer) {
269    TransitRegionStateProcedure[] procs =
270        createAssignProcedures(env, regions, regionReplication, targetServer, true);
271    if (procs.length > 0) {
272      env.getMasterServices().getMasterProcedureExecutor().submitProcedures(procs);
273    }
274  }

276  static void removeNonDefaultReplicas(MasterProcedureEnv env, Stream<RegionInfo> regions,
277      int regionReplication) {
278    // Remove from in-memory states
279    regions.flatMap(hri -> IntStream.range(1, regionReplication)
280      .mapToObj(i -> RegionReplicaUtil.getRegionInfoForReplica(hri, i))).forEach(hri -> {
281        env.getAssignmentManager().getRegionStates().deleteRegion(hri);
282        env.getMasterServices().getServerManager().removeRegion(hri);
283        FavoredNodesManager fnm = env.getMasterServices().getFavoredNodesManager();
284        if (fnm != null) {
285          fnm.deleteFavoredNodesForRegions(Collections.singletonList(hri));
286        }
287      });
288  }

290  static void checkClosedRegion(MasterProcedureEnv env, RegionInfo regionInfo) throws IOException {
291    if (WALSplitUtil.hasRecoveredEdits(env.getMasterConfiguration(), regionInfo)) {
292      throw new IOException("Recovered.edits are found in Region: " + regionInfo +
293        ", abort split/merge to prevent data loss");
294    }
295  }
