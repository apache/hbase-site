047/**
048 * Procedure that describe the assignment of a single region.
049 * There can only be one RegionTransitionProcedure per region running at a time
050 * since each procedure takes a lock on the region.
051 *
052 * <p>The Assign starts by pushing the "assign" operation to the AssignmentManager
053 * and then will go in a "waiting" state.
054 * The AM will batch the "assign" requests and ask the Balancer where to put
055 * the region (the various policies will be respected: retain, round-robin, random).
056 * Once the AM and the balancer have found a place for the region the procedure
057 * will be resumed and an "open region" request will be placed in the Remote Dispatcher
058 * queue, and the procedure once again will go in a "waiting state".
059 * The Remote Dispatcher will batch the various requests for that server and
060 * they will be sent to the RS for execution.
061 * The RS will complete the open operation by calling master.reportRegionStateTransition().
062 * The AM will intercept the transition report, and notify the procedure.
063 * The procedure will finish the assignment by publishing to new state on meta
064 * or it will retry the assignment.
065 *
066 * <p>This procedure does not rollback when beyond the first
067 * REGION_TRANSITION_QUEUE step; it will press on trying to assign in the face of
068 * failure. Should we ignore rollback calls to Assign/Unassign then? Or just
069 * remove rollback here?
070 */
071// TODO: Add being able to assign a region to open read-only.
072@InterfaceAudience.Private
073public class AssignProcedure extends RegionTransitionProcedure {
074  private static final Logger LOG = LoggerFactory.getLogger(AssignProcedure.class);
075
076  /**
077   * Set to true when we need recalibrate -- choose a new target -- because original assign failed.
078   */
079  private boolean forceNewPlan = false;
080
081  /**
082   * Gets set as desired target on move, merge, etc., when we want to go to a particular server.
083   * We may not be able to respect this request but will try. When it is NOT set, then we ask
084   * the balancer to assign. This value is used below in startTransition to set regionLocation if
085   * non-null. Setting regionLocation in regionServerNode is how we override balancer setting
086   * destination.
087   */
088  protected volatile ServerName targetServer;
089
090  /**
091   * Comparator that will sort AssignProcedures so meta assigns come first, then system table
092   * assigns and finally user space assigns.
093   */
094  public static final CompareAssignProcedure COMPARATOR = new CompareAssignProcedure();

126  @Override
127  protected void serializeStateData(ProcedureStateSerializer serializer)
128      throws IOException {
129    final AssignRegionStateData.Builder state = AssignRegionStateData.newBuilder()
130        .setTransitionState(getTransitionState())
131        .setRegionInfo(ProtobufUtil.toRegionInfo(getRegionInfo()));
132    if (forceNewPlan) {
133      state.setForceNewPlan(true);
134    }
135    if (this.targetServer != null) {
136      state.setTargetServer(ProtobufUtil.toServerName(this.targetServer));
137    }
138    if (getAttempt() > 0) {
139      state.setAttempt(getAttempt());
140    }
141    serializer.serialize(state.build());
142  }
143
144  @Override
145  protected void deserializeStateData(ProcedureStateSerializer serializer)
146      throws IOException {
147    final AssignRegionStateData state = serializer.deserialize(AssignRegionStateData.class);
148    setTransitionState(state.getTransitionState());
149    setRegionInfo(ProtobufUtil.toRegionInfo(state.getRegionInfo()));
150    forceNewPlan = state.getForceNewPlan();
151    if (state.hasTargetServer()) {
152      this.targetServer = ProtobufUtil.toServerName(state.getTargetServer());
153    }
154    if (state.hasAttempt()) {
155      setAttempt(state.getAttempt());
156    }
157  }

159  @Override
160  protected boolean startTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
161      throws IOException {
162    // If the region is already open we can't do much...
163    if (regionNode.isInState(State.OPEN) && isServerOnline(env, regionNode)) {
164      LOG.info("Assigned, not reassigning; " + this + "; " + regionNode.toShortString());
165      return false;
166    }
167    // Don't assign if table is in disabling or disabled state.
168    TableStateManager tsm = env.getMasterServices().getTableStateManager();
169    TableName tn = regionNode.getRegionInfo().getTable();
170    if (tsm.getTableState(tn).isDisabledOrDisabling()) {
171      LOG.info("Table " + tn + " state=" + tsm.getTableState(tn) + ", skipping " + this);
172      return false;
173    }
174    // If the region is SPLIT, we can't assign it. But state might be CLOSED, rather than
175    // SPLIT which is what a region gets set to when unassigned as part of SPLIT. FIX.
176    if (regionNode.isInState(State.SPLIT) ||
177        (regionNode.getRegionInfo().isOffline() && regionNode.getRegionInfo().isSplit())) {
178      LOG.info("SPLIT, cannot be assigned; " + this + "; " + regionNode +
179        "; hri=" + regionNode.getRegionInfo());
180      return false;
181    }
182
183    // If we haven't started the operation yet, we can abort
184    if (aborted.get() && regionNode.isInState(State.CLOSED, State.OFFLINE)) {
185      if (incrementAndCheckMaxAttempts(env, regionNode)) {
186        regionNode.setState(State.FAILED_OPEN);
187        setFailure(getClass().getSimpleName(),
188          new RetriesExhaustedException("Max attempts exceeded"));
189      } else {
190        setAbortFailure(getClass().getSimpleName(), "Abort requested");
191      }
192      return false;
193    }
194
195    // Send assign (add into assign-pool). We call regionNode.offline below to set state to
196    // OFFLINE and to clear the region location. Setting a new regionLocation here is how we retain
197    // old assignment or specify target server if a move or merge. See
198    // AssignmentManager#processAssignQueue. Otherwise, balancer gives us location.
199    // TODO: Region will be set into OFFLINE state below regardless of what its previous state was
200    // This is dangerous? Wrong? What if region was in an unexpected state?
201    ServerName lastRegionLocation = regionNode.offline();
202    boolean retain = false;
203    if (!forceNewPlan) {
204      if (this.targetServer != null) {
205        retain = targetServer.equals(lastRegionLocation);
206        regionNode.setRegionLocation(targetServer);
207      } else {
208        if (lastRegionLocation != null) {
209          // Try and keep the location we had before we offlined.
210          retain = true;
211          regionNode.setRegionLocation(lastRegionLocation);
212        } else if (regionNode.getLastHost() != null) {
213          retain = true;
214          LOG.info("Setting lastHost as the region location " + regionNode.getLastHost());
215          regionNode.setRegionLocation(regionNode.getLastHost());
216        }
217      }
218    }
219    LOG.info("Starting " + this + "; " + regionNode.toShortString() +
220        "; forceNewPlan=" + this.forceNewPlan +
221        ", retain=" + retain);
222    env.getAssignmentManager().queueAssign(regionNode);
223    return true;
224  }
225
226  @Override
227  protected boolean updateTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
228  throws IOException, ProcedureSuspendedException {
229    // TODO: crash if destinationServer is specified and not online
230    // which is also the case when the balancer provided us with a different location.
231    if (LOG.isTraceEnabled()) {
232      LOG.trace("Update " + this + "; " + regionNode.toShortString());
233    }
234    if (regionNode.getRegionLocation() == null) {
235      setTransitionState(RegionTransitionState.REGION_TRANSITION_QUEUE);
236      return true;
237    }
238
239    if (!isServerOnline(env, regionNode)) {
240      // TODO: is this correct? should we wait the chore/ssh?
241      LOG.info("Server not online, re-queuing " + this + "; " + regionNode.toShortString());
242      setTransitionState(RegionTransitionState.REGION_TRANSITION_QUEUE);
243      return true;
244    }
245
246    if (env.getAssignmentManager().waitServerReportEvent(regionNode.getRegionLocation(), this)) {
247      LOG.info("Early suspend! " + this + "; " + regionNode.toShortString());
248      throw new ProcedureSuspendedException();
249    }
250
251    if (regionNode.isInState(State.OPEN)) {
252      LOG.info("Already assigned: " + this + "; " + regionNode.toShortString());
253      return false;
254    }
255
256    // Transition regionNode State. Set it to OPENING. Update hbase:meta, and add
257    // region to list of regions on the target regionserver. Need to UNDO if failure!
258    env.getAssignmentManager().markRegionAsOpening(regionNode);
259
260    // TODO: Requires a migration to be open by the RS?
261    // regionNode.getFormatVersion()
262
263    if (!addToRemoteDispatcher(env, regionNode.getRegionLocation())) {
264      // Failed the dispatch BUT addToRemoteDispatcher internally does
265      // cleanup on failure -- even the undoing of markRegionAsOpening above --
266      // so nothing more to do here; in fact we need to get out of here
267      // fast since we've been put back on the scheduler.
268    }
269
270    // We always return true, even if we fail dispatch because addToRemoteDispatcher
271    // failure processing sets state back to REGION_TRANSITION_QUEUE so we try again;
272    // i.e. return true to keep the Procedure running; it has been reset to startover.
273    return true;
274  }

276  @Override
277  protected void finishTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
278      throws IOException {
279    env.getAssignmentManager().markRegionAsOpened(regionNode);
280    // This success may have been after we failed open a few times. Be sure to cleanup any
281    // failed open references. See #incrementAndCheckMaxAttempts and where it is called.
282    env.getAssignmentManager().getRegionStates().removeFromFailedOpen(regionNode.getRegionInfo());
283  }

285  @Override
286  protected void reportTransition(final MasterProcedureEnv env, final RegionStateNode regionNode,
287      final TransitionCode code, final long openSeqNum) throws UnexpectedStateException {
288    switch (code) {
289      case OPENED:
290        if (openSeqNum < 0) {
291          throw new UnexpectedStateException("Received report unexpected " + code +
292              " transition openSeqNum=" + openSeqNum + ", " + regionNode);
293        }
294        if (openSeqNum < regionNode.getOpenSeqNum()) {
295          // Don't bother logging if openSeqNum == 0
296          if (openSeqNum != 0) {
297            LOG.warn("Skipping update of open seqnum with " + openSeqNum +
298                " because current seqnum=" + regionNode.getOpenSeqNum());
299          }
300        } else {
301          regionNode.setOpenSeqNum(openSeqNum);
302        }
303        // Leave the state here as OPENING for now. We set it to OPEN in
304        // REGION_TRANSITION_FINISH section where we do a bunch of checks.
305        // regionNode.setState(RegionState.State.OPEN, RegionState.State.OPENING);
306        setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
307        break;
308      case FAILED_OPEN:
309        handleFailure(env, regionNode);
310        break;
311      default:
312        throw new UnexpectedStateException("Received report unexpected " + code +
313            " transition openSeqNum=" + openSeqNum + ", " + regionNode.toShortString() +
314            ", " + this + ", expected OPENED or FAILED_OPEN.");
315    }
316  }

318  /**
319   * Called when dispatch or subsequent OPEN request fail. Can be run by the
320   * inline dispatch call or later by the ServerCrashProcedure. Our state is
321   * generally OPENING. Cleanup and reset to OFFLINE and put our Procedure
322   * State back to REGION_TRANSITION_QUEUE so the Assign starts over.
323   */
324  private void handleFailure(final MasterProcedureEnv env, final RegionStateNode regionNode) {
325    if (incrementAndCheckMaxAttempts(env, regionNode)) {
326      aborted.set(true);
327    }
328    this.forceNewPlan = true;
329    this.targetServer = null;
330    regionNode.offline();
331    // We were moved to OPENING state before dispatch. Undo. It is safe to call
332    // this method because it checks for OPENING first.
333    env.getAssignmentManager().undoRegionAsOpening(regionNode);
334    setTransitionState(RegionTransitionState.REGION_TRANSITION_QUEUE);
335  }
336
337  private boolean incrementAndCheckMaxAttempts(final MasterProcedureEnv env,
338      final RegionStateNode regionNode) {
339    final int retries = env.getAssignmentManager().getRegionStates().
340        addToFailedOpen(regionNode).incrementAndGetRetries();
341    int max = env.getAssignmentManager().getAssignMaxAttempts();
342    LOG.info("Retry=" + retries + " of max=" + max + "; " +
343        this + "; " + regionNode.toShortString());
344    return retries >= max;
345  }

347  @Override
348  public RemoteOperation remoteCallBuild(final MasterProcedureEnv env, final ServerName serverName) {
349    assert serverName.equals(getRegionState(env).getRegionLocation());
350    return new RegionOpenOperation(this, getRegionInfo(),
351        env.getAssignmentManager().getFavoredNodes(getRegionInfo()), false);
352  }
353
354  @Override
355  protected boolean remoteCallFailed(final MasterProcedureEnv env, final RegionStateNode regionNode,
356      final IOException exception) {
357    handleFailure(env, regionNode);
358    return true;
359  }

380  /**
381   * Sort AssignProcedures such that meta and system assigns come first before user-space assigns.
382   * Have to do it this way w/ distinct Comparator because Procedure is already Comparable on
383   * 'Env'(?).
384   */
385  public static class CompareAssignProcedure implements Comparator<AssignProcedure> {
386    @Override
387    public int compare(AssignProcedure left, AssignProcedure right) {
388      if (left.getRegionInfo().isMetaRegion()) {
389        if (right.getRegionInfo().isMetaRegion()) {
390          return RegionInfo.COMPARATOR.compare(left.getRegionInfo(), right.getRegionInfo());
391        }
392        return -1;
393      } else if (right.getRegionInfo().isMetaRegion()) {
394        return +1;
395      }
396      if (left.getRegionInfo().getTable().isSystemTable()) {
397        if (right.getRegionInfo().getTable().isSystemTable()) {
398          return RegionInfo.COMPARATOR.compare(left.getRegionInfo(), right.getRegionInfo());
399        }
400        return -1;
401      } else if (right.getRegionInfo().getTable().isSystemTable()) {
402        return +1;
403      }
404      return RegionInfo.COMPARATOR.compare(left.getRegionInfo(), right.getRegionInfo());
405    }
406  }
