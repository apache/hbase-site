088  /**
089   * Current Region State.
090   * In-memory only. Not persisted.
091   */
092  // Mutable/Immutable? Changes have to be synchronized or not?
093  // Data members are volatile which seems to say multi-threaded access is fine.
094  // In the below we do check and set but the check state could change before
095  // we do the set because no synchronization....which seems dodgy. Clear up
096  // understanding here... how many threads accessing? Do locks make it so one
097  // thread at a time working on a single Region's RegionStateNode? Lets presume

099  // the RegionStateNode instance. TODO.
100  public static class RegionStateNode implements Comparable<RegionStateNode> {
101    private final RegionInfo regionInfo;
102    private final ProcedureEvent<?> event;
103
104    private volatile RegionTransitionProcedure procedure = null;
105    private volatile ServerName regionLocation = null;
106    private volatile ServerName lastHost = null;
107    /**
108     * A Region-in-Transition (RIT) moves through states.
109     * See {@link State} for complete list. A Region that
110     * is opened moves from OFFLINE => OPENING => OPENED.
111     */
112    private volatile State state = State.OFFLINE;
113
114    /**
115     * Updated whenever a call to {@link #setRegionLocation(ServerName)}
116     * or {@link #setState(State, State...)}.
117     */
118    private volatile long lastUpdate = 0;
119
120    private volatile long openSeqNum = HConstants.NO_SEQNUM;
121
122    public RegionStateNode(final RegionInfo regionInfo) {
123      this.regionInfo = regionInfo;
124      this.event = new AssignmentProcedureEvent(regionInfo);
125    }
126
127    /**
128     * @param update new region state this node should be assigned.
129     * @param expected current state should be in this given list of expected states
130     * @return true, if current state is in expected list; otherwise false.
131     */
132    public boolean setState(final State update, final State... expected) {
133      if (!isInState(expected)) {
134        return false;
135      }
136      this.state = update;
137      this.lastUpdate = EnvironmentEdgeManager.currentTime();
138      return true;
139    }
140
141    /**
142     * Put region into OFFLINE mode (set state and clear location).
143     * @return Last recorded server deploy
144     */
145    public ServerName offline() {
146      setState(State.OFFLINE);
147      return setRegionLocation(null);
148    }
149
150    /**
151     * Set new {@link State} but only if currently in <code>expected</code> State
152     * (if not, throw {@link UnexpectedStateException}.
153     */
154    public void transitionState(final State update, final State... expected)
155    throws UnexpectedStateException {
156      if (!setState(update, expected)) {
157        throw new UnexpectedStateException("Expected " + Arrays.toString(expected) +
158          " so could move to " + update + " but current state=" + getState());
159      }
160    }
161
162    public boolean isInState(final State... expected) {
163      if (expected != null && expected.length > 0) {
164        boolean expectedState = false;
165        for (int i = 0; i < expected.length; ++i) {
166          expectedState |= (getState() == expected[i]);
167        }
168        return expectedState;
169      }
170      return true;
171    }
172
173    public boolean isStuck() {
174      return isInState(State.FAILED_OPEN) && getProcedure() != null;
175    }
176
177    public boolean isInTransition() {
178      return getProcedure() != null;
179    }
180
181    public long getLastUpdate() {
182      return procedure != null ? procedure.getLastUpdate() : lastUpdate;
183    }
184
185    public void setLastHost(final ServerName serverName) {
186      this.lastHost = serverName;
187    }
188
189    public void setOpenSeqNum(final long seqId) {
190      this.openSeqNum = seqId;
191    }
192
193    public ServerName setRegionLocation(final ServerName serverName) {
194      ServerName lastRegionLocation = this.regionLocation;
195      if (LOG.isTraceEnabled() && serverName == null) {
196        LOG.trace("Tracking when we are set to null " + this, new Throwable("TRACE"));
197      }
198      this.regionLocation = serverName;
199      this.lastUpdate = EnvironmentEdgeManager.currentTime();
200      return lastRegionLocation;
201    }
202
203    public boolean setProcedure(final RegionTransitionProcedure proc) {
204      if (this.procedure != null && this.procedure != proc) {
205        return false;
206      }
207      this.procedure = proc;
208      return true;
209    }
210
211    public boolean unsetProcedure(final RegionTransitionProcedure proc) {
212      if (this.procedure != null && this.procedure != proc) {
213        return false;
214      }
215      this.procedure = null;
216      return true;
217    }
218
219    public RegionTransitionProcedure getProcedure() {
220      return procedure;
221    }
222
223    public ProcedureEvent<?> getProcedureEvent() {
224      return event;
225    }
226
227    public RegionInfo getRegionInfo() {
228      return regionInfo;
229    }
230
231    public TableName getTable() {
232      return getRegionInfo().getTable();
233    }
234
235    public boolean isSystemTable() {
236      return getTable().isSystemTable();
237    }
238
239    public ServerName getLastHost() {
240      return lastHost;
241    }
242
243    public ServerName getRegionLocation() {
244      return regionLocation;
245    }
246
247    public State getState() {
248      return state;
249    }
250
251    public long getOpenSeqNum() {
252      return openSeqNum;
253    }
254
255    public int getFormatVersion() {
256      // we don't have any format for now
257      // it should probably be in regionInfo.getFormatVersion()
258      return 0;
259    }
260
261    public RegionState toRegionState() {
262      return new RegionState(getRegionInfo(), getState(), getLastUpdate(), getRegionLocation());
263    }
264
265    @Override
266    public int compareTo(final RegionStateNode other) {
267      // NOTE: RegionInfo sort by table first, so we are relying on that.
268      // we have a TestRegionState#testOrderedByTable() that check for that.
269      return RegionInfo.COMPARATOR.compare(getRegionInfo(), other.getRegionInfo());
270    }
271
272    @Override
273    public int hashCode() {
274      return getRegionInfo().hashCode();
275    }
276
277    @Override
278    public boolean equals(final Object other) {
279      if (this == other) return true;
280      if (!(other instanceof RegionStateNode)) return false;
281      return compareTo((RegionStateNode)other) == 0;
282    }
283
284    @Override
285    public String toString() {
286      return toDescriptiveString();
287    }
288
289    public String toShortString() {
290      // rit= is the current Region-In-Transition State -- see State enum.
291      return String.format("rit=%s, location=%s", getState(), getRegionLocation());
292    }
293
294    public String toDescriptiveString() {
295      return String.format("%s, table=%s, region=%s",
296        toShortString(), getTable(), getRegionInfo().getEncodedName());
297    }

300  // This comparator sorts the RegionStates by time stamp then Region name.
301  // Comparing by timestamp alone can lead us to discard different RegionStates that happen
302  // to share a timestamp.
303  private static class RegionStateStampComparator implements Comparator<RegionState> {
304    @Override
305    public int compare(final RegionState l, final RegionState r) {
306      int stampCmp = Long.compare(l.getStamp(), r.getStamp());
307      return stampCmp != 0 ? stampCmp : RegionInfo.COMPARATOR.compare(l.getRegion(), r.getRegion());
308    }

572   * {@link State#SPLIT} state.

574  public List<RegionInfo> getRegionsOfTable(final TableName table, final boolean offline) {
575    final ArrayList<RegionStateNode> nodes = getTableRegionStateNodes(table);
576    final ArrayList<RegionInfo> hris = new ArrayList<RegionInfo>(nodes.size());
577    for (RegionStateNode node: nodes) {
578      if (include(node, offline)) hris.add(node.getRegionInfo());
579    }
580    return hris;

583  /**
584   * Utility. Whether to include region in list of regions. Default is to
585   * weed out split and offline regions.
586   * @return True if we should include the <code>node</code> (do not include
587   * if split or offline unless <code>offline</code> is set to true.
588   */
589  boolean include(final RegionStateNode node, final boolean offline) {
590    if (LOG.isTraceEnabled()) {
591      LOG.trace("WORKING ON " + node + " " + node.getRegionInfo());
592    }
593    if (node.isInState(State.SPLIT)) return false;
594    if (node.isInState(State.OFFLINE) && !offline) return false;
595    final RegionInfo hri = node.getRegionInfo();
596    return (!hri.isOffline() && !hri.isSplit()) ||
597        ((hri.isOffline() || hri.isSplit()) && offline);
598  }

689  public Map<ServerName, List<RegionInfo>> getSnapShotOfAssignment(
690      final Collection<RegionInfo> regions) {
691    final Map<ServerName, List<RegionInfo>> result = new HashMap<ServerName, List<RegionInfo>>();
692    for (RegionInfo hri: regions) {
693      final RegionStateNode node = getRegionStateNode(hri);
694      if (node == null) continue;
695
696      // TODO: State.OPEN
697      final ServerName serverName = node.getRegionLocation();
698      if (serverName == null) continue;
699
700      List<RegionInfo> serverRegions = result.get(serverName);
701      if (serverRegions == null) {
702        serverRegions = new ArrayList<RegionInfo>();
703        result.put(serverName, serverRegions);
704      }
705
706      serverRegions.add(node.getRegionInfo());
707    }
708    return result;
709  }

719  public Map<RegionState.State, List<RegionInfo>> getRegionByStateOfTable(TableName tableName) {
720    final State[] states = State.values();
721    final Map<RegionState.State, List<RegionInfo>> tableRegions =
722        new HashMap<State, List<RegionInfo>>(states.length);
723    for (int i = 0; i < states.length; ++i) {
724      tableRegions.put(states[i], new ArrayList<RegionInfo>());
725    }
726
727    for (RegionStateNode node: regionsMap.values()) {
728      if (node.getTable().equals(tableName)) {
729        tableRegions.get(node.getState()).add(node.getRegionInfo());
730      }
731    }
732    return tableRegions;
733  }

746  /**
747   * This is an EXPENSIVE clone.  Cloning though is the safest thing to do.
748   * Can't let out original since it can change and at least the load balancer
749   * wants to iterate this exported list.  We need to synchronize on regions
750   * since all access to this.servers is under a lock on this.regions.
751   * @param forceByCluster a flag to force to aggregate the server-load to the cluster level
752   * @return A clone of current assignments by table.
753   */
754  public Map<TableName, Map<ServerName, List<RegionInfo>>> getAssignmentsByTable(
755      final boolean forceByCluster) {
756    if (!forceByCluster) return getAssignmentsByTable();
757
758    final HashMap<ServerName, List<RegionInfo>> ensemble =
759      new HashMap<ServerName, List<RegionInfo>>(serverMap.size());
760    for (ServerStateNode serverNode: serverMap.values()) {
761      ensemble.put(serverNode.getServerName(), serverNode.getRegionInfoList());
762    }
763
764    // TODO: can we use Collections.singletonMap(HConstants.ENSEMBLE_TABLE_NAME, ensemble)?
765    final Map<TableName, Map<ServerName, List<RegionInfo>>> result =
766      new HashMap<TableName, Map<ServerName, List<RegionInfo>>>(1);
767    result.put(HConstants.ENSEMBLE_TABLE_NAME, ensemble);
768    return result;
769  }
770
771  public Map<TableName, Map<ServerName, List<RegionInfo>>> getAssignmentsByTable() {
772    final Map<TableName, Map<ServerName, List<RegionInfo>>> result = new HashMap<>();
773    for (RegionStateNode node: regionsMap.values()) {
774      Map<ServerName, List<RegionInfo>> tableResult = result.get(node.getTable());
775      if (tableResult == null) {
776        tableResult = new HashMap<ServerName, List<RegionInfo>>();
777        result.put(node.getTable(), tableResult);
778      }
779
780      final ServerName serverName = node.getRegionLocation();
781      if (serverName == null) {
782        LOG.info("Skipping, no server for " + node);
783        continue;
784      }
785      List<RegionInfo> serverResult = tableResult.get(serverName);
786      if (serverResult == null) {
787        serverResult = new ArrayList<RegionInfo>();
788        tableResult.put(serverName, serverResult);
789      }
790
791      serverResult.add(node.getRegionInfo());
792    }
793    // Add online servers with no assignment for the table.
794    for (Map<ServerName, List<RegionInfo>> table: result.values()) {
795        for (ServerName svr : serverMap.keySet()) {
796          if (!table.containsKey(svr)) {
797            table.put(svr, new ArrayList<RegionInfo>());
798          }
799        }
800    }
801    return result;
802  }
