060  // This comparator sorts the RegionStates by time stamp then Region name.
061  // Comparing by timestamp alone can lead us to discard different RegionStates that happen
062  // to share a timestamp.
063  private static class RegionStateStampComparator implements Comparator<RegionState> {
064    @Override
065    public int compare(final RegionState l, final RegionState r) {
066      int stampCmp = Long.compare(l.getStamp(), r.getStamp());
067      return stampCmp != 0 ? stampCmp : RegionInfo.COMPARATOR.compare(l.getRegion(), r.getRegion());
068    }
069  }

071  public final static RegionStateStampComparator REGION_STATE_STAMP_COMPARATOR =
072      new RegionStateStampComparator();

135  public void deleteRegion(final RegionInfo regionInfo) {
136    regionsMap.remove(regionInfo.getRegionName());
137    // See HBASE-20860
138    // After master restarts, merged regions' RIT state may not be cleaned,
139    // making sure they are cleaned here
140    if (regionInTransition.containsKey(regionInfo)) {
141      regionInTransition.remove(regionInfo);
142    }
143    // Remove from the offline regions map too if there.
144    if (this.regionOffline.containsKey(regionInfo)) {
145      if (LOG.isTraceEnabled()) LOG.trace("Removing from regionOffline Map: " + regionInfo);
146      this.regionOffline.remove(regionInfo);
147    }
148  }

226  private HRegionLocation createRegionForReopen(RegionStateNode node) {
227    node.lock();
228    try {
229      if (!include(node, false)) {
230        return null;
231      }
232      if (node.isInState(State.OPEN)) {
233        return new HRegionLocation(node.getRegionInfo(), node.getRegionLocation(),
234          node.getOpenSeqNum());
235      } else if (node.isInState(State.OPENING)) {
236        return new HRegionLocation(node.getRegionInfo(), node.getRegionLocation(), -1);
237      } else {
238        return null;
239      }
240    } finally {
241      node.unlock();
242    }
243  }

245  /**
246   * Get the regions to be reopened when modifying a table.
247   * <p/>
248   * Notice that the {@code openSeqNum} in the returned HRegionLocation is also used to indicate the
249   * state of this region, positive means the region is in {@link State#OPEN}, -1 means
250   * {@link State#OPENING}. And for regions in other states we do not need reopen them.
251   */
252  public List<HRegionLocation> getRegionsOfTableForReopen(TableName tableName) {
253    return getTableRegionStateNodes(tableName).stream().map(this::createRegionForReopen)
254      .filter(r -> r != null).collect(Collectors.toList());
255  }

257  /**
258   * Check whether the region has been reopened. The meaning of the {@link HRegionLocation} is the
259   * same with {@link #getRegionsOfTableForReopen(TableName)}.
260   * <p/>
261   * For a region which is in {@link State#OPEN} before, if the region state is changed or the open
262   * seq num is changed, we can confirm that it has been reopened.
263   * <p/>
264   * For a region which is in {@link State#OPENING} before, usually it will be in {@link State#OPEN}
265   * now and we will schedule a MRP to reopen it. But there are several exceptions:
266   * <ul>
267   * <li>The region is in state other than {@link State#OPEN} or {@link State#OPENING}.</li>
268   * <li>The location of the region has been changed</li>
269   * </ul>
270   * Of course the region could still be in {@link State#OPENING} state and still on the same
271   * server, then here we will still return a {@link HRegionLocation} for it, just like
272   * {@link #getRegionsOfTableForReopen(TableName)}.
273   * @param oldLoc the previous state/location of this region
274   * @return null if the region has been reopened, otherwise a new {@link HRegionLocation} which
275   *         means we still need to reopen the region.
276   * @see #getRegionsOfTableForReopen(TableName)
277   */
278  public HRegionLocation checkReopened(HRegionLocation oldLoc) {
279    RegionStateNode node = getRegionStateNode(oldLoc.getRegion());
280    // HBASE-20921
281    // if the oldLoc's state node does not exist, that means the region is
282    // merged or split, no need to check it
283    if (node == null) {
284      return null;
285    }
286    node.lock();
287    try {
288      if (oldLoc.getSeqNum() >= 0) {
289        // in OPEN state before
290        if (node.isInState(State.OPEN)) {
291          if (node.getOpenSeqNum() > oldLoc.getSeqNum()) {
292            // normal case, the region has been reopened
293            return null;
294          } else {
295            // the open seq num does not change, need to reopen again
296            return new HRegionLocation(node.getRegionInfo(), node.getRegionLocation(),
297              node.getOpenSeqNum());
298          }
299        } else {
300          // the state has been changed so we can make sure that the region has been reopened(not
301          // finished maybe, but not a problem).
302          return null;
303        }
304      } else {
305        // in OPENING state before
306        if (!node.isInState(State.OPEN, State.OPENING)) {
307          // not in OPEN or OPENING state, then we can make sure that the region has been
308          // reopened(not finished maybe, but not a problem)
309          return null;
310        } else {
311          if (!node.getRegionLocation().equals(oldLoc.getServerName())) {
312            // the region has been moved, so we can make sure that the region has been reopened.
313            return null;
314          }
315          // normal case, we are still in OPENING state, or the reopen has been opened and the state
316          // is changed to OPEN.
317          long openSeqNum = node.isInState(State.OPEN) ? node.getOpenSeqNum() : -1;
318          return new HRegionLocation(node.getRegionInfo(), node.getRegionLocation(), openSeqNum);
319        }
320      }
321    } finally {
322      node.unlock();
323    }
324  }

343  /**
344   * Utility. Whether to include region in list of regions. Default is to
345   * weed out split and offline regions.
346   * @return True if we should include the <code>node</code> (do not include
347   * if split or offline unless <code>offline</code> is set to true.
348   */
349  boolean include(final RegionStateNode node, final boolean offline) {
350    if (LOG.isTraceEnabled()) {
351      LOG.trace("WORKING ON " + node + " " + node.getRegionInfo());
352    }
353    if (node.isInState(State.SPLIT)) return false;
354    if (node.isInState(State.OFFLINE) && !offline) return false;
355    final RegionInfo hri = node.getRegionInfo();
356    return (!hri.isOffline() && !hri.isSplit()) ||
357        ((hri.isOffline() || hri.isSplit()) && offline);
358  }

452  public Map<ServerName, List<RegionInfo>> getSnapShotOfAssignment(
453      final Collection<RegionInfo> regions) {
454    final Map<ServerName, List<RegionInfo>> result = new HashMap<ServerName, List<RegionInfo>>();
455    if (regions != null) {
456      for (RegionInfo hri : regions) {
457        final RegionStateNode node = getRegionStateNode(hri);
458        if (node == null) {
459          continue;
460        }
461        createSnapshot(node, result);
462      }
463    } else {
464      for (RegionStateNode node : regionsMap.values()) {
465        if (node == null) {
466          continue;
467        }
468        createSnapshot(node, result);
469      }
470    }
471    return result;
472  }
473
474  private void createSnapshot(RegionStateNode node, Map<ServerName, List<RegionInfo>> result) {
475    final ServerName serverName = node.getRegionLocation();
476    if (serverName == null) {
477      return;
478    }
479
480    List<RegionInfo> serverRegions = result.get(serverName);
481    if (serverRegions == null) {
482      serverRegions = new ArrayList<RegionInfo>();
483      result.put(serverName, serverRegions);
484    }
485    serverRegions.add(node.getRegionInfo());
486  }

526  /**
527   * This is an EXPENSIVE clone.  Cloning though is the safest thing to do.
528   * Can't let out original since it can change and at least the load balancer
529   * wants to iterate this exported list.  We need to synchronize on regions
530   * since all access to this.servers is under a lock on this.regions.
531   * @param forceByCluster a flag to force to aggregate the server-load to the cluster level
532   * @return A clone of current assignments by table.
533   */
534  public Map<TableName, Map<ServerName, List<RegionInfo>>> getAssignmentsByTable(
535      final boolean forceByCluster) {
536    if (!forceByCluster) return getAssignmentsByTable();
537
538    final HashMap<ServerName, List<RegionInfo>> ensemble =
539      new HashMap<ServerName, List<RegionInfo>>(serverMap.size());
540    for (ServerStateNode serverNode: serverMap.values()) {
541      ensemble.put(serverNode.getServerName(), serverNode.getRegionInfoList());
542    }
543
544    // TODO: can we use Collections.singletonMap(HConstants.ENSEMBLE_TABLE_NAME, ensemble)?
545    final Map<TableName, Map<ServerName, List<RegionInfo>>> result =
546      new HashMap<TableName, Map<ServerName, List<RegionInfo>>>(1);
547    result.put(HConstants.ENSEMBLE_TABLE_NAME, ensemble);
548    return result;
549  }
550
551  public Map<TableName, Map<ServerName, List<RegionInfo>>> getAssignmentsByTable() {
552    final Map<TableName, Map<ServerName, List<RegionInfo>>> result = new HashMap<>();
553    for (RegionStateNode node: regionsMap.values()) {
554      Map<ServerName, List<RegionInfo>> tableResult = result.get(node.getTable());
555      if (tableResult == null) {
556        tableResult = new HashMap<ServerName, List<RegionInfo>>();
557        result.put(node.getTable(), tableResult);
558      }
559
560      final ServerName serverName = node.getRegionLocation();
561      if (serverName == null) {
562        LOG.info("Skipping, no server for " + node);
563        continue;
564      }
565      List<RegionInfo> serverResult = tableResult.get(serverName);
566      if (serverResult == null) {
567        serverResult = new ArrayList<RegionInfo>();
568        tableResult.put(serverName, serverResult);
569      }
570
571      serverResult.add(node.getRegionInfo());
572    }
573    // Add online servers with no assignment for the table.
574    for (Map<ServerName, List<RegionInfo>> table: result.values()) {
575        for (ServerName svr : serverMap.keySet()) {
576          if (!table.containsKey(svr)) {
577            table.put(svr, new ArrayList<RegionInfo>());
578          }
579        }
580    }
581    return result;
582  }

690    final byte[] key = regionNode.getRegionInfo().getRegionName();
691    RegionFailedOpen node = regionFailedOpen.get(key);
692    if (node == null) {
693      RegionFailedOpen newNode = new RegionFailedOpen(regionNode);
694      RegionFailedOpen oldNode = regionFailedOpen.putIfAbsent(key, newNode);
695      node = oldNode != null ? oldNode : newNode;
696    }
697    return node;
698  }

745  public double getAverageLoad() {
746    int numServers = 0;
747    int totalLoad = 0;
748    for (ServerStateNode node: serverMap.values()) {
749      totalLoad += node.getRegionCount();
750      numServers++;
751    }
752    return numServers == 0 ? 0.0: (double)totalLoad / (double)numServers;
753  }

761  public boolean isReplicaAvailableForRegion(final RegionInfo info) {
762    // if the region info itself is a replica return true.
763    if (!RegionReplicaUtil.isDefaultReplica(info)) {
764      return true;
765    }
766    // iterate the regionsMap for the given region name. If there are replicas it should
767    // list them in order.
768    for (RegionStateNode node : regionsMap.tailMap(info.getRegionName()).values()) {
769      if (!node.getTable().equals(info.getTable())
770          || !ServerRegionReplicaUtil.isReplicasForSameRegion(info, node.getRegionInfo())) {
771        break;
772      } else if (!RegionReplicaUtil.isDefaultReplica(node.getRegionInfo())) {
773        // we have replicas
774        return true;
775      }
776    }
777    // we don have replicas
778    return false;
779  }

791  public static String regionNamesToString(final Collection<byte[]> regions) {
792    final StringBuilder sb = new StringBuilder();
793    final Iterator<byte[]> it = regions.iterator();
794    sb.append("[");
795    if (it.hasNext()) {
796      sb.append(Bytes.toStringBinary(it.next()));
797      while (it.hasNext()) {
798        sb.append(", ");
799        sb.append(Bytes.toStringBinary(it.next()));
800      }
801    }
802    sb.append("]");
803    return sb.toString();
804  }
