088/**
089 * The procedure to split a region in a table.
090 * Takes lock on the parent region.
091 * It holds the lock for the life of the procedure.
092 * <p>Throws exception on construction if determines context hostile to spllt (cluster going
093 * down or master is shutting down or table is disabled).</p>
094 */
095@InterfaceAudience.Private
096public class SplitTableRegionProcedure
097    extends AbstractStateMachineRegionProcedure<SplitTableRegionState> {
098  private static final Logger LOG = LoggerFactory.getLogger(SplitTableRegionProcedure.class);
099  private RegionInfo daughterOneRI;
100  private RegionInfo daughterTwoRI;
101  private byte[] bestSplitRow;
102  private RegionSplitPolicy splitPolicy;

104  public SplitTableRegionProcedure() {
105    // Required by the Procedure framework to create the procedure on replay
106  }

108  public SplitTableRegionProcedure(final MasterProcedureEnv env,
109      final RegionInfo regionToSplit, final byte[] splitRow) throws IOException {
110    super(env, regionToSplit);
111    preflightChecks(env, true);
112    // When procedure goes to run in its prepare step, it also does these checkOnline checks. Here
113    // we fail-fast on construction. There it skips the split with just a warning.
114    checkOnline(env, regionToSplit);
115    this.bestSplitRow = splitRow;
116    TableDescriptor tableDescriptor = env.getMasterServices().getTableDescriptors()
117      .get(getTableName());
118    Configuration conf = env.getMasterConfiguration();
119    if (hasBestSplitRow()) {
120      // Apply the split restriction for the table to the user-specified split point
121      RegionSplitRestriction splitRestriction =
122        RegionSplitRestriction.create(tableDescriptor, conf);
123      byte[] restrictedSplitRow = splitRestriction.getRestrictedSplitPoint(bestSplitRow);
124      if (!Bytes.equals(bestSplitRow, restrictedSplitRow)) {
125        LOG.warn("The specified split point {} violates the split restriction of the table. "
126            + "Using {} as a split point.", Bytes.toStringBinary(bestSplitRow),
127          Bytes.toStringBinary(restrictedSplitRow));
128        bestSplitRow = restrictedSplitRow;
129      }
130    }
131    checkSplittable(env, regionToSplit);
132    final TableName table = regionToSplit.getTable();
133    final long rid = getDaughterRegionIdTimestamp(regionToSplit);
134    this.daughterOneRI = RegionInfoBuilder.newBuilder(table)
135        .setStartKey(regionToSplit.getStartKey())
136        .setEndKey(bestSplitRow)
137        .setSplit(false)
138        .setRegionId(rid)
139        .build();
140    this.daughterTwoRI = RegionInfoBuilder.newBuilder(table)
141        .setStartKey(bestSplitRow)
142        .setEndKey(regionToSplit.getEndKey())
143        .setSplit(false)
144        .setRegionId(rid)
145        .build();
146
147    if (tableDescriptor.getRegionSplitPolicyClassName() != null) {
148      // Since we don't have region reference here, creating the split policy instance without it.
149      // This can be used to invoke methods which don't require Region reference. This instantiation
150      // of a class on Master-side though it only makes sense on the RegionServer-side is
151      // for Phoenix Local Indexing. Refer HBASE-12583 for more information.
152      Class<? extends RegionSplitPolicy> clazz =
153          RegionSplitPolicy.getSplitPolicyClass(tableDescriptor, conf);
154      this.splitPolicy = ReflectionUtils.newInstance(clazz, conf);
155    }
156  }

186  private boolean hasBestSplitRow() {

190  /**
191   * Check whether the region is splittable
192   * @param env MasterProcedureEnv
193   * @param regionToSplit parent Region to be split
194   */
195  private void checkSplittable(final MasterProcedureEnv env,
196      final RegionInfo regionToSplit) throws IOException {
197    // Ask the remote RS if this region is splittable.
198    // If we get an IOE, report it along w/ the failure so can see why we are not splittable at
199    // this time.
200    if(regionToSplit.getReplicaId() != RegionInfo.DEFAULT_REPLICA_ID) {
201      throw new IllegalArgumentException("Can't invoke split on non-default regions directly");
202    }
203    RegionStateNode node =
204        env.getAssignmentManager().getRegionStates().getRegionStateNode(getParentRegion());
205    IOException splittableCheckIOE = null;
206    boolean splittable = false;
207    if (node != null) {
208      try {
209        GetRegionInfoResponse response;
210        if (!hasBestSplitRow()) {
211          LOG.info(
212            "{} splitKey isn't explicitly specified, will try to find a best split key from RS {}",
213            node.getRegionInfo().getRegionNameAsString(), node.getRegionLocation());
214          response = AssignmentManagerUtil.getRegionInfoResponse(env, node.getRegionLocation(),
215            node.getRegionInfo(), true);
216          bestSplitRow =
217            response.hasBestSplitRow() ? response.getBestSplitRow().toByteArray() : null;
218        } else {
219          response = AssignmentManagerUtil.getRegionInfoResponse(env, node.getRegionLocation(),
220            node.getRegionInfo(), false);
221        }
222        splittable = response.hasSplittable() && response.getSplittable();
223        if (LOG.isDebugEnabled()) {
224          LOG.debug("Splittable=" + splittable + " " + node.toShortString());
225        }
226      } catch (IOException e) {
227        splittableCheckIOE = e;
228      }
229    }
230
231    if (!splittable) {
232      IOException e =
233        new DoNotRetryIOException(regionToSplit.getShortNameToLog() + " NOT splittable");
234      if (splittableCheckIOE != null) {
235        e.initCause(splittableCheckIOE);
236      }
237      throw e;
238    }
239
240    if (!hasBestSplitRow()) {
241      throw new DoNotRetryIOException("Region not splittable because bestSplitPoint = null, " +
242        "maybe table is too small for auto split. For force split, try specifying split row");
243    }
244
245    if (Bytes.equals(regionToSplit.getStartKey(), bestSplitRow)) {
246      throw new DoNotRetryIOException(
247        "Split row is equal to startkey: " + Bytes.toStringBinary(bestSplitRow));
248    }
249
250    if (!regionToSplit.containsRow(bestSplitRow)) {
251      throw new DoNotRetryIOException("Split row is not inside region key range splitKey:" +
252        Bytes.toStringBinary(bestSplitRow) + " region: " + regionToSplit);
253    }
254  }

256  /**
257   * Calculate daughter regionid to use.
258   * @param hri Parent {@link RegionInfo}
259   * @return Daughter region id (timestamp) to use.
260   */
261  private static long getDaughterRegionIdTimestamp(final RegionInfo hri) {
262    long rid = EnvironmentEdgeManager.currentTime();
263    // Regionid is timestamp.  Can't be less than that of parent else will insert
264    // at wrong location in hbase:meta (See HBASE-710).
265    if (rid < hri.getRegionId()) {
266      LOG.warn("Clock skew; parent regions id is " + hri.getRegionId() +
267        " but current time here is " + rid);
268      rid = hri.getRegionId() + 1;
269    }
270    return rid;
271  }

284  @Override
285  protected Flow executeFromState(MasterProcedureEnv env, SplitTableRegionState state)
286      throws InterruptedException {
287    LOG.trace("{} execute state={}", this, state);
288
289    try {
290      switch (state) {
291        case SPLIT_TABLE_REGION_PREPARE:
292          if (prepareSplitRegion(env)) {
293            setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION);
294            break;
295          } else {
296            return Flow.NO_MORE_STATE;
297          }
298        case SPLIT_TABLE_REGION_PRE_OPERATION:
299          preSplitRegion(env);
300          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_CLOSE_PARENT_REGION);
301          break;
302        case SPLIT_TABLE_REGION_CLOSE_PARENT_REGION:
303          addChildProcedure(createUnassignProcedures(env));
304          setNextState(SplitTableRegionState.SPLIT_TABLE_REGIONS_CHECK_CLOSED_REGIONS);
305          break;
306        case SPLIT_TABLE_REGIONS_CHECK_CLOSED_REGIONS:
307          checkClosedRegions(env);
308          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS);
309          break;
310        case SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS:
311          removeNonDefaultReplicas(env);
312          createDaughterRegions(env);
313          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_WRITE_MAX_SEQUENCE_ID_FILE);
314          break;
315        case SPLIT_TABLE_REGION_WRITE_MAX_SEQUENCE_ID_FILE:
316          writeMaxSequenceIdFile(env);
317          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_META);
318          break;
319        case SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_META:
320          preSplitRegionBeforeMETA(env);
321          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_UPDATE_META);
322          break;
323        case SPLIT_TABLE_REGION_UPDATE_META:
324          updateMeta(env);
325          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META);
326          break;
327        case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META:
328          preSplitRegionAfterMETA(env);
329          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS);
330          break;
331        case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
332          addChildProcedure(createAssignProcedures(env));
333          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_POST_OPERATION);
334          break;
335        case SPLIT_TABLE_REGION_POST_OPERATION:
336          postSplitRegion(env);
337          return Flow.NO_MORE_STATE;
338        default:
339          throw new UnsupportedOperationException(this + " unhandled state=" + state);
340      }
341    } catch (IOException e) {
342      String msg = "Splitting " + getParentRegion().getEncodedName() + ", " + this;
343      if (!isRollbackSupported(state)) {
344        // We reach a state that cannot be rolled back. We just need to keep retrying.
345        LOG.warn(msg, e);
346      } else {
347        LOG.error(msg, e);
348        setFailure("master-split-regions", e);
349      }
350    }
351    // if split fails,  need to call ((HRegion)parent).clearSplit() when it is a force split
352    return Flow.HAS_MORE_STATE;
353  }

355  /**
356   * To rollback {@link SplitTableRegionProcedure}, an AssignProcedure is asynchronously
357   * submitted for parent region to be split (rollback doesn't wait on the completion of the
358   * AssignProcedure) . This can be improved by changing rollback() to support sub-procedures.
359   * See HBASE-19851 for details.
360   */
361  @Override
362  protected void rollbackState(final MasterProcedureEnv env, final SplitTableRegionState state)
363      throws IOException, InterruptedException {
364    LOG.trace("{} rollback state={}", this, state);
365
366    try {
367      switch (state) {
368        case SPLIT_TABLE_REGION_POST_OPERATION:
369        case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
370        case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META:
371        case SPLIT_TABLE_REGION_UPDATE_META:
372          // PONR
373          throw new UnsupportedOperationException(this + " unhandled state=" + state);
374        case SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_META:
375          break;
376        case SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS:
377        case SPLIT_TABLE_REGION_WRITE_MAX_SEQUENCE_ID_FILE:
378          deleteDaughterRegions(env);
379          break;
380        case SPLIT_TABLE_REGIONS_CHECK_CLOSED_REGIONS:
381          // Doing nothing, in SPLIT_TABLE_REGION_CLOSE_PARENT_REGION,
382          // we will bring parent region online
383          break;
384        case SPLIT_TABLE_REGION_CLOSE_PARENT_REGION:
385          openParentRegion(env);
386          break;
387        case SPLIT_TABLE_REGION_PRE_OPERATION:
388          postRollBackSplitRegion(env);
389          break;
390        case SPLIT_TABLE_REGION_PREPARE:
391          break; // nothing to do
392        default:
393          throw new UnsupportedOperationException(this + " unhandled state=" + state);
394      }
395    } catch (IOException e) {
396      // This will be retried. Unless there is a bug in the code,
397      // this should be just a "temporary error" (e.g. network down)
398      LOG.warn("pid=" + getProcId() + " failed rollback attempt step " + state +
399          " for splitting the region "
400        + getParentRegion().getEncodedName() + " in table " + getTableName(), e);
401      throw e;
402    }
403  }

407   */
408  @Override
409  protected boolean isRollbackSupported(final SplitTableRegionState state) {
410    switch (state) {
411      case SPLIT_TABLE_REGION_POST_OPERATION:
412      case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
413      case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META:
414      case SPLIT_TABLE_REGION_UPDATE_META:
415        // It is not safe to rollback if we reach to these states.
416        return false;
417      default:
418        break;
419    }
420    return true;
421  }

497  private static final State[] EXPECTED_SPLIT_STATES = new State[] { State.OPEN, State.CLOSED };

499  /**
500   * Prepare to Split region.
501   * @param env MasterProcedureEnv
502   */
503  public boolean prepareSplitRegion(final MasterProcedureEnv env) throws IOException {
504    // Fail if we are taking snapshot for the given table
505    if (env.getMasterServices().getSnapshotManager()
506      .isTakingSnapshot(getParentRegion().getTable())) {
507      setFailure(new IOException("Skip splitting region " + getParentRegion().getShortNameToLog() +
508        ", because we are taking snapshot for the table " + getParentRegion().getTable()));
509      return false;
510    }
511    // Check whether the region is splittable
512    RegionStateNode node =
513        env.getAssignmentManager().getRegionStates().getRegionStateNode(getParentRegion());
514
515    if (node == null) {
516      throw new UnknownRegionException(getParentRegion().getRegionNameAsString());
517    }
518
519    RegionInfo parentHRI = node.getRegionInfo();
520    if (parentHRI == null) {
521      LOG.info("Unsplittable; parent region is null; node={}", node);
522      return false;
523    }
524    // Lookup the parent HRI state from the AM, which has the latest updated info.
525    // Protect against the case where concurrent SPLIT requests came in and succeeded
526    // just before us.
527    if (node.isInState(State.SPLIT)) {
528      LOG.info("Split of " + parentHRI + " skipped; state is already SPLIT");
529      return false;
530    }
531    if (parentHRI.isSplit() || parentHRI.isOffline()) {
532      LOG.info("Split of " + parentHRI + " skipped because offline/split.");
533      return false;
534    }
535
536    // expected parent to be online or closed
537    if (!node.isInState(EXPECTED_SPLIT_STATES)) {
538      // We may have SPLIT already?
539      setFailure(new IOException("Split " + parentHRI.getRegionNameAsString() +
540          " FAILED because state=" + node.getState() + "; expected " +
541          Arrays.toString(EXPECTED_SPLIT_STATES)));
542      return false;
543    }
544
545    // Mostly this check is not used because we already check the switch before submit a split
546    // procedure. Just for safe, check the switch again. This procedure can be rollbacked if
547    // the switch was set to false after submit.
548    if (!env.getMasterServices().isSplitOrMergeEnabled(MasterSwitchType.SPLIT)) {
549      LOG.warn("pid=" + getProcId() + " split switch is off! skip split of " + parentHRI);
550      setFailure(new IOException("Split region " + parentHRI.getRegionNameAsString() +
551          " failed due to split switch off"));
552      return false;
553    }
554
555    if (!env.getMasterServices().getTableDescriptors().get(getTableName()).isSplitEnabled()) {
556      LOG.warn("pid={}, split is disabled for the table! Skipping split of {}", getProcId(),
557        parentHRI);
558      setFailure(new IOException("Split region " + parentHRI.getRegionNameAsString()
559          + " failed as region split is disabled for the table"));
560      return false;
561    }
562
563    // set node state as SPLITTING
564    node.setState(State.SPLITTING);
565
566    // Since we have the lock and the master is coordinating the operation
567    // we are always able to split the region
568    return true;
569  }

571  /**
572   * Action before splitting region in a table.
573   * @param env MasterProcedureEnv
574   */
575  private void preSplitRegion(final MasterProcedureEnv env)
576      throws IOException, InterruptedException {
577    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
578    if (cpHost != null) {
579      cpHost.preSplitRegionAction(getTableName(), getSplitRow(), getUser());
580    }
581
582    // TODO: Clean up split and merge. Currently all over the place.
583    // Notify QuotaManager and RegionNormalizer
584    try {
585      env.getMasterServices().getMasterQuotaManager().onRegionSplit(this.getParentRegion());
586    } catch (QuotaExceededException e) {
587      // TODO: why is this here? split requests can be submitted by actors other than the normalizer
588      env.getMasterServices()
589        .getRegionNormalizerManager()
590        .planSkipped(NormalizationPlan.PlanType.SPLIT);
591      throw e;
592    }
593  }

651  /**
652   * Create Split directory
653   * @param env MasterProcedureEnv
654   */
655  private Pair<Integer, Integer> splitStoreFiles(final MasterProcedureEnv env,
656      final HRegionFileSystem regionFs) throws IOException {
657    final Configuration conf = env.getMasterConfiguration();
658    TableDescriptor htd = env.getMasterServices().getTableDescriptors().get(getTableName());
659    // The following code sets up a thread pool executor with as many slots as
660    // there's files to split. It then fires up everything, waits for
661    // completion and finally checks for any exception
662    //
663    // Note: From HBASE-26187, splitStoreFiles now creates daughter region dirs straight under the
664    // table dir. In case of failure, the proc would go through this again, already existing
665    // region dirs and split files would just be ignored, new split files should get created.
666    int nbFiles = 0;
667    final Map<String, Collection<StoreFileInfo>> files =
668        new HashMap<String, Collection<StoreFileInfo>>(htd.getColumnFamilyCount());
669    for (ColumnFamilyDescriptor cfd : htd.getColumnFamilies()) {
670      String family = cfd.getNameAsString();
671      Collection<StoreFileInfo> sfis = regionFs.getStoreFiles(family);
672      if (sfis == null) {
673        continue;
674      }
675      Collection<StoreFileInfo> filteredSfis = null;
676      for (StoreFileInfo sfi : sfis) {
677        // Filter. There is a lag cleaning up compacted reference files. They get cleared
678        // after a delay in case outstanding Scanners still have references. Because of this,
679        // the listing of the Store content may have straggler reference files. Skip these.
680        // It should be safe to skip references at this point because we checked above with
681        // the region if it thinks it is splittable and if we are here, it thinks it is
682        // splitable.
683        if (sfi.isReference()) {
684          LOG.info("Skipping split of " + sfi + "; presuming ready for archiving.");
685          continue;
686        }
687        if (filteredSfis == null) {
688          filteredSfis = new ArrayList<StoreFileInfo>(sfis.size());
689          files.put(family, filteredSfis);
690        }
691        filteredSfis.add(sfi);
692        nbFiles++;
693      }
694    }
695    if (nbFiles == 0) {
696      // no file needs to be splitted.
697      return new Pair<Integer, Integer>(0, 0);
698    }
699    // Max #threads is the smaller of the number of storefiles or the default max determined above.
700    int maxThreads = Math.min(
701      conf.getInt(HConstants.REGION_SPLIT_THREADS_MAX,
702        conf.getInt(HStore.BLOCKING_STOREFILES_KEY, HStore.DEFAULT_BLOCKING_STOREFILE_COUNT)),
703      nbFiles);
704    LOG.info("pid=" + getProcId() + " splitting " + nbFiles + " storefiles, region=" +
705        getParentRegion().getShortNameToLog() + ", threads=" + maxThreads);
706    final ExecutorService threadPool = Executors.newFixedThreadPool(maxThreads,
707      new ThreadFactoryBuilder().setNameFormat("StoreFileSplitter-pool-%d").setDaemon(true)
708        .setUncaughtExceptionHandler(Threads.LOGGING_EXCEPTION_HANDLER).build());
709    final List<Future<Pair<Path, Path>>> futures = new ArrayList<Future<Pair<Path, Path>>>(nbFiles);
710
711    // Split each store file.
712    for (Map.Entry<String, Collection<StoreFileInfo>> e : files.entrySet()) {
713      byte[] familyName = Bytes.toBytes(e.getKey());
714      final ColumnFamilyDescriptor hcd = htd.getColumnFamily(familyName);
715      final Collection<StoreFileInfo> storeFiles = e.getValue();
716      if (storeFiles != null && storeFiles.size() > 0) {
717        final Configuration storeConfiguration =
718          StoreUtils.createStoreConfiguration(env.getMasterConfiguration(), htd, hcd);
719        for (StoreFileInfo storeFileInfo : storeFiles) {
720          // As this procedure is running on master, use CacheConfig.DISABLED means
721          // don't cache any block.
722          // We also need to pass through a suitable CompoundConfiguration as if this
723          // is running in a regionserver's Store context, or we might not be able
724          // to read the hfiles.
725          storeFileInfo.setConf(storeConfiguration);
726          StoreFileSplitter sfs = new StoreFileSplitter(regionFs, familyName,
727            new HStoreFile(storeFileInfo, hcd.getBloomFilterType(), CacheConfig.DISABLED));
728          futures.add(threadPool.submit(sfs));
729        }
730      }
731    }
732    // Shutdown the pool
733    threadPool.shutdown();
734
735    // Wait for all the tasks to finish.
736    // When splits ran on the RegionServer, how-long-to-wait-configuration was named
737    // hbase.regionserver.fileSplitTimeout. If set, use its value.
738    long fileSplitTimeout = conf.getLong("hbase.master.fileSplitTimeout",
739      conf.getLong("hbase.regionserver.fileSplitTimeout", 600000));
740    try {
741      boolean stillRunning = !threadPool.awaitTermination(fileSplitTimeout, TimeUnit.MILLISECONDS);
742      if (stillRunning) {
743        threadPool.shutdownNow();
744        // wait for the thread to shutdown completely.
745        while (!threadPool.isTerminated()) {
746          Thread.sleep(50);
747        }
748        throw new IOException(
749            "Took too long to split the" + " files and create the references, aborting split");
750      }
751    } catch (InterruptedException e) {
752      throw (InterruptedIOException) new InterruptedIOException().initCause(e);
753    }
754
755    int daughterA = 0;
756    int daughterB = 0;
757    // Look for any exception
758    for (Future<Pair<Path, Path>> future : futures) {
759      try {
760        Pair<Path, Path> p = future.get();
761        daughterA += p.getFirst() != null ? 1 : 0;
762        daughterB += p.getSecond() != null ? 1 : 0;
763      } catch (InterruptedException e) {
764        throw (InterruptedIOException) new InterruptedIOException().initCause(e);
765      } catch (ExecutionException e) {
766        throw new IOException(e);
767      }
768    }
769
770    if (LOG.isDebugEnabled()) {
771      LOG.debug("pid=" + getProcId() + " split storefiles for region " +
772          getParentRegion().getShortNameToLog() + " Daughter A: " + daughterA +
773          " storefiles, Daughter B: " + daughterB + " storefiles.");
774    }
775    return new Pair<Integer, Integer>(daughterA, daughterB);
776  }

778  private void assertSplitResultFilesCount(final FileSystem fs,
779      final int expectedSplitResultFileCount, Path dir)
780    throws IOException {
781    if (expectedSplitResultFileCount != 0) {
782      int resultFileCount = FSUtils.getRegionReferenceAndLinkFileCount(fs, dir);
783      if (expectedSplitResultFileCount != resultFileCount) {
784        throw new IOException("Failing split. Didn't have expected reference and HFileLink files"
785            + ", expected=" + expectedSplitResultFileCount + ", actual=" + resultFileCount);
786      }
787    }
788  }

790  private Pair<Path, Path> splitStoreFile(HRegionFileSystem regionFs, byte[] family, HStoreFile sf)
791    throws IOException {
792    if (LOG.isDebugEnabled()) {
793      LOG.debug("pid=" + getProcId() + " splitting started for store file: " +
794          sf.getPath() + " for region: " + getParentRegion().getShortNameToLog());
795    }
796
797    final byte[] splitRow = getSplitRow();
798    final String familyName = Bytes.toString(family);
799    final Path path_first = regionFs.splitStoreFile(this.daughterOneRI, familyName, sf, splitRow,
800        false, splitPolicy);
801    final Path path_second = regionFs.splitStoreFile(this.daughterTwoRI, familyName, sf, splitRow,
802       true, splitPolicy);
803    if (LOG.isDebugEnabled()) {
804      LOG.debug("pid=" + getProcId() + " splitting complete for store file: " +
805          sf.getPath() + " for region: " + getParentRegion().getShortNameToLog());
806    }
807    return new Pair<Path,Path>(path_first, path_second);
808  }

810  /**
811   * Utility class used to do the file splitting / reference writing
812   * in parallel instead of sequentially.
813   */
814  private class StoreFileSplitter implements Callable<Pair<Path,Path>> {
815    private final HRegionFileSystem regionFs;
816    private final byte[] family;
817    private final HStoreFile sf;
818
819    /**
820     * Constructor that takes what it needs to split
821     * @param regionFs the file system
822     * @param family Family that contains the store file
823     * @param sf which file
824     */
825    public StoreFileSplitter(HRegionFileSystem regionFs, byte[] family, HStoreFile sf) {
826      this.regionFs = regionFs;
827      this.sf = sf;
828      this.family = family;
829    }
830
831    @Override
832    public Pair<Path,Path> call() throws IOException {
833      return splitStoreFile(regionFs, family, sf);
834    }
835  }

853            + "region name."
854            + "Mutations from coprocessor should only for hbase:meta table.");
855        throw e;
856      }
857    }
