103  public SplitTableRegionProcedure(final MasterProcedureEnv env,
104      final RegionInfo regionToSplit, final byte[] splitRow) throws IOException {
105    super(env, regionToSplit);
106    this.bestSplitRow = splitRow;
107    checkSplittable(env, regionToSplit, bestSplitRow);
108    final TableName table = regionToSplit.getTable();
109    final long rid = getDaughterRegionIdTimestamp(regionToSplit);
110    this.daughter_1_RI = RegionInfoBuilder.newBuilder(table)
111        .setStartKey(regionToSplit.getStartKey())
112        .setEndKey(bestSplitRow)
113        .setSplit(false)
114        .setRegionId(rid)
115        .build();
116    this.daughter_2_RI = RegionInfoBuilder.newBuilder(table)
117        .setStartKey(bestSplitRow)
118        .setEndKey(regionToSplit.getEndKey())
119        .setSplit(false)
120        .setRegionId(rid)
121        .build();
122    TableDescriptor htd = env.getMasterServices().getTableDescriptors().get(getTableName());
123    if(htd.getRegionSplitPolicyClassName() != null) {
124      // Since we don't have region reference here, creating the split policy instance without it.
125      // This can be used to invoke methods which don't require Region reference. This instantiation
126      // of a class on Master-side though it only makes sense on the RegionServer-side is
127      // for Phoenix Local Indexing. Refer HBASE-12583 for more information.
128      Class<? extends RegionSplitPolicy> clazz =
129          RegionSplitPolicy.getSplitPolicyClass(htd, env.getMasterConfiguration());
130      this.splitPolicy = ReflectionUtils.newInstance(clazz, env.getMasterConfiguration());
131    }
132  }

134  /**
135   * Check whether the region is splittable
136   * @param env MasterProcedureEnv
137   * @param regionToSplit parent Region to be split
138   * @param splitRow if splitRow is not specified, will first try to get bestSplitRow from RS
139   * @throws IOException
140   */
141  private void checkSplittable(final MasterProcedureEnv env,
142      final RegionInfo regionToSplit, final byte[] splitRow) throws IOException {
143    // Ask the remote RS if this region is splittable.
144    // If we get an IOE, report it along w/ the failure so can see why we are not splittable at this time.
145    if(regionToSplit.getReplicaId() != RegionInfo.DEFAULT_REPLICA_ID) {
146      throw new IllegalArgumentException ("Can't invoke split on non-default regions directly");
147    }
148    RegionStateNode node =
149        env.getAssignmentManager().getRegionStates().getRegionStateNode(getParentRegion());
150    IOException splittableCheckIOE = null;
151    boolean splittable = false;
152    if (node != null) {
153      try {
154        if (bestSplitRow == null || bestSplitRow.length == 0) {
155          LOG.info("splitKey isn't explicitly specified, " + " will try to find a best split key from RS");
156        }
157        // Always set bestSplitRow request as true here,
158        // need to call Region#checkSplit to check it splittable or not
159        GetRegionInfoResponse response =
160            Util.getRegionInfoResponse(env, node.getRegionLocation(), node.getRegionInfo(), true);
161        if(bestSplitRow == null || bestSplitRow.length == 0) {
162          bestSplitRow = response.hasBestSplitRow() ? response.getBestSplitRow().toByteArray() : null;
163        }
164        splittable = response.hasSplittable() && response.getSplittable();
165
166        if (LOG.isDebugEnabled()) {
167          LOG.debug("Splittable=" + splittable + " " + node.toShortString());
168        }
169      } catch (IOException e) {
170        splittableCheckIOE = e;
171      }
172    }
173
174    if (!splittable) {
175      IOException e = new IOException(regionToSplit.getShortNameToLog() + " NOT splittable");
176      if (splittableCheckIOE != null) e.initCause(splittableCheckIOE);
177      throw e;
178    }
179
180    if(bestSplitRow == null || bestSplitRow.length == 0) {
181      throw new DoNotRetryIOException("Region not splittable because bestSplitPoint = null, "
182          + "maybe table is too small for auto split. For force split, try specifying split row");
183    }
184
185    if (Bytes.equals(regionToSplit.getStartKey(), bestSplitRow)) {
186      throw new DoNotRetryIOException(
187        "Split row is equal to startkey: " + Bytes.toStringBinary(splitRow));
188    }
189
190    if (!regionToSplit.containsRow(bestSplitRow)) {
191      throw new DoNotRetryIOException(
192        "Split row is not inside region key range splitKey:" + Bytes.toStringBinary(splitRow) +
193        " region: " + regionToSplit);
194    }
195  }

197  /**
198   * Calculate daughter regionid to use.
199   * @param hri Parent {@link RegionInfo}
200   * @return Daughter region id (timestamp) to use.
201   */
202  private static long getDaughterRegionIdTimestamp(final RegionInfo hri) {
203    long rid = EnvironmentEdgeManager.currentTime();
204    // Regionid is timestamp.  Can't be less than that of parent else will insert
205    // at wrong location in hbase:meta (See HBASE-710).
206    if (rid < hri.getRegionId()) {
207      LOG.warn("Clock skew; parent regions id is " + hri.getRegionId() +
208        " but current time here is " + rid);
209      rid = hri.getRegionId() + 1;
210    }
211    return rid;
212  }

214  @Override
215  protected Flow executeFromState(final MasterProcedureEnv env, final SplitTableRegionState state)
216      throws InterruptedException {
217    if (isTraceEnabled()) {
218      LOG.trace(this + " execute state=" + state);
219    }
220
221    try {
222      switch (state) {
223      case SPLIT_TABLE_REGION_PREPARE:
224        if (prepareSplitRegion(env)) {
225          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION);
226          break;
227        } else {
228          assert isFailed() : "split region should have an exception here";
229          return Flow.NO_MORE_STATE;
230        }
231      case SPLIT_TABLE_REGION_PRE_OPERATION:
232        preSplitRegion(env);
233        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_CLOSE_PARENT_REGION);
234        break;
235      case SPLIT_TABLE_REGION_CLOSE_PARENT_REGION:
236        addChildProcedure(createUnassignProcedures(env, getRegionReplication(env)));
237        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS);
238        break;
239      case SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS:
240        createDaughterRegions(env);
241        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_META);
242        break;
243      case SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_META:
244        preSplitRegionBeforeMETA(env);
245        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_UPDATE_META);
246        break;
247      case SPLIT_TABLE_REGION_UPDATE_META:
248        updateMetaForDaughterRegions(env);
249        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META);
250        break;
251      case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META:
252        preSplitRegionAfterMETA(env);
253        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS);
254        break;
255      case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
256        addChildProcedure(createAssignProcedures(env, getRegionReplication(env)));
257        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_POST_OPERATION);
258        break;
259      case SPLIT_TABLE_REGION_POST_OPERATION:
260        postSplitRegion(env);
261        return Flow.NO_MORE_STATE;
262      default:
263        throw new UnsupportedOperationException(this + " unhandled state=" + state);
264      }
265    } catch (IOException e) {
266      String msg = "Error trying to split region " + getParentRegion().getEncodedName() + " in the table "
267          + getTableName() + " (in state=" + state + ")";
268      if (!isRollbackSupported(state)) {
269        // We reach a state that cannot be rolled back. We just need to keep retry.
270        LOG.warn(msg, e);
271      } else {
272        LOG.error(msg, e);
273        setFailure(e);
274      }
275    }
276    // if split fails,  need to call ((HRegion)parent).clearSplit() when it is a force split
277    return Flow.HAS_MORE_STATE;
278  }

280  /**
281   * To rollback {@link SplitTableRegionProcedure}, an AssignProcedure is asynchronously
282   * submitted for parent region to be split (rollback doesn't wait on the completion of the
283   * AssignProcedure) . This can be improved by changing rollback() to support sub-procedures.
284   * See HBASE-19851 for details.
285   */
286  @Override
287  protected void rollbackState(final MasterProcedureEnv env, final SplitTableRegionState state)
288      throws IOException, InterruptedException {
289    if (isTraceEnabled()) {
290      LOG.trace(this + " rollback state=" + state);
291    }
292
293    try {
294      switch (state) {
295      case SPLIT_TABLE_REGION_POST_OPERATION:
296      case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
297      case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META:
298      case SPLIT_TABLE_REGION_UPDATE_META:
299        // PONR
300        throw new UnsupportedOperationException(this + " unhandled state=" + state);
301      case SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_META:
302        break;
303      case SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS:
304        // Doing nothing, as re-open parent region would clean up daughter region directories.
305        break;
306      case SPLIT_TABLE_REGION_CLOSE_PARENT_REGION:
307        openParentRegion(env);
308        break;
309      case SPLIT_TABLE_REGION_PRE_OPERATION:
310        postRollBackSplitRegion(env);
311        break;
312      case SPLIT_TABLE_REGION_PREPARE:
313        break; // nothing to do
314      default:
315        throw new UnsupportedOperationException(this + " unhandled state=" + state);
316      }
317    } catch (IOException e) {
318      // This will be retried. Unless there is a bug in the code,
319      // this should be just a "temporary error" (e.g. network down)
320      LOG.warn("pid=" + getProcId() + " failed rollback attempt step " + state +
321          " for splitting the region "
322        + getParentRegion().getEncodedName() + " in table " + getTableName(), e);
323      throw e;
324    }
325  }

327  /*
328   * Check whether we are in the state that can be rollback
329   */
330  @Override
331  protected boolean isRollbackSupported(final SplitTableRegionState state) {
332    switch (state) {
333      case SPLIT_TABLE_REGION_POST_OPERATION:
334      case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
335      case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_META:
336      case SPLIT_TABLE_REGION_UPDATE_META:
337        // It is not safe to rollback if we reach to these states.
338        return false;
339      default:
340        break;
341    }
342    return true;
343  }

419  private static State [] EXPECTED_SPLIT_STATES = new State [] {State.OPEN, State.CLOSED};
420  /**
421   * Prepare to Split region.
422   * @param env MasterProcedureEnv
423   * @throws IOException
424   */
425  @VisibleForTesting
426  public boolean prepareSplitRegion(final MasterProcedureEnv env) throws IOException {
427    // Check whether the region is splittable
428    RegionStateNode node =
429        env.getAssignmentManager().getRegionStates().getRegionStateNode(getParentRegion());
430
431    if (node == null) {
432      throw new UnknownRegionException(getParentRegion().getRegionNameAsString());
433    }
434
435    RegionInfo parentHRI = node.getRegionInfo();
436    // Lookup the parent HRI state from the AM, which has the latest updated info.
437    // Protect against the case where concurrent SPLIT requests came in and succeeded
438    // just before us.
439    if (node.isInState(State.SPLIT)) {
440      LOG.info("Split of " + parentHRI + " skipped; state is already SPLIT");
441      return false;
442    }
443    if (parentHRI.isSplit() || parentHRI.isOffline()) {
444      LOG.info("Split of " + parentHRI + " skipped because offline/split.");
445      return false;
446    }
447
448    // expected parent to be online or closed
449    if (!node.isInState(EXPECTED_SPLIT_STATES)) {
450      // We may have SPLIT already?
451      setFailure(new IOException("Split " + parentHRI.getRegionNameAsString() +
452          " FAILED because state=" + node.getState() + "; expected " +
453          Arrays.toString(EXPECTED_SPLIT_STATES)));
454      return false;
455    }
456
457    // Since we have the lock and the master is coordinating the operation
458    // we are always able to split the region
459    if (!env.getMasterServices().isSplitOrMergeEnabled(MasterSwitchType.SPLIT)) {
460      LOG.warn("pid=" + getProcId() + " split switch is off! skip split of " + parentHRI);
461      setFailure(new IOException("Split region " +
462          (parentHRI == null? "null": parentHRI.getRegionNameAsString()) +
463          " failed due to split switch off"));
464      return false;
465    }
466
467    // set node state as SPLITTING
468    node.setState(State.SPLITTING);
469
470    return true;
471  }

473  /**
474   * Action before splitting region in a table.
475   * @param env MasterProcedureEnv
476   * @throws IOException
477   * @throws InterruptedException
478   */
479  private void preSplitRegion(final MasterProcedureEnv env)
480      throws IOException, InterruptedException {
481    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
482    if (cpHost != null) {
483      cpHost.preSplitRegionAction(getTableName(), getSplitRow(), getUser());
484    }
485
486    // TODO: Clean up split and merge. Currently all over the place.
487    // Notify QuotaManager and RegionNormalizer
488    try {
489      env.getMasterServices().getMasterQuotaManager().onRegionSplit(this.getParentRegion());
490    } catch (QuotaExceededException e) {
491      env.getAssignmentManager().getRegionNormalizer().planSkipped(this.getParentRegion(),
492          NormalizationPlan.PlanType.SPLIT);
493      throw e;
494    }
495  }

509  /**
510   * Rollback close parent region
511   * @param env MasterProcedureEnv
512   **/
513  private void openParentRegion(final MasterProcedureEnv env) throws IOException {
514    // Check whether the region is closed; if so, open it in the same server
515    final int regionReplication = getRegionReplication(env);
516    final ServerName serverName = getParentRegionServerName(env);
517
518    final AssignProcedure[] procs = new AssignProcedure[regionReplication];
519    for (int i = 0; i < regionReplication; ++i) {
520      final RegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(getParentRegion(), i);
521      procs[i] = env.getAssignmentManager().createAssignProcedure(hri, serverName);
522    }
523    env.getMasterServices().getMasterProcedureExecutor().submitProcedures(procs);
524  }

527   * Create daughter regions
528   * @param env MasterProcedureEnv
529   * @throws IOException
530   */
531  @VisibleForTesting
532  public void createDaughterRegions(final MasterProcedureEnv env) throws IOException {
533    final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
534    final Path tabledir = FSUtils.getTableDir(mfs.getRootDir(), getTableName());
535    final FileSystem fs = mfs.getFileSystem();
536    HRegionFileSystem regionFs = HRegionFileSystem.openRegionFromFileSystem(
537      env.getMasterConfiguration(), fs, tabledir, getParentRegion(), false);
538    regionFs.createSplitsDir();
539
540    Pair<Integer, Integer> expectedReferences = splitStoreFiles(env, regionFs);
541
542    assertReferenceFileCount(fs, expectedReferences.getFirst(),
543      regionFs.getSplitsDir(daughter_1_RI));
544    //Move the files from the temporary .splits to the final /table/region directory
545    regionFs.commitDaughterRegion(daughter_1_RI);
546    assertReferenceFileCount(fs, expectedReferences.getFirst(),
547      new Path(tabledir, daughter_1_RI.getEncodedName()));
548
549    assertReferenceFileCount(fs, expectedReferences.getSecond(),
550      regionFs.getSplitsDir(daughter_2_RI));
551    regionFs.commitDaughterRegion(daughter_2_RI);
552    assertReferenceFileCount(fs, expectedReferences.getSecond(),
553      new Path(tabledir, daughter_2_RI.getEncodedName()));
554  }

556  /**
557   * Create Split directory
558   * @param env MasterProcedureEnv
559   * @throws IOException
560   */
561  private Pair<Integer, Integer> splitStoreFiles(final MasterProcedureEnv env,
562      final HRegionFileSystem regionFs) throws IOException {
563    final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
564    final Configuration conf = env.getMasterConfiguration();
565    // The following code sets up a thread pool executor with as many slots as
566    // there's files to split. It then fires up everything, waits for
567    // completion and finally checks for any exception
568    //
569    // Note: splitStoreFiles creates daughter region dirs under the parent splits dir
570    // Nothing to unroll here if failure -- re-run createSplitsDir will
571    // clean this up.
572    int nbFiles = 0;
573    final Map<String, Collection<StoreFileInfo>> files =
574      new HashMap<String, Collection<StoreFileInfo>>(regionFs.getFamilies().size());
575    for (String family: regionFs.getFamilies()) {
576      Collection<StoreFileInfo> sfis = regionFs.getStoreFiles(family);
577      if (sfis == null) continue;
578      Collection<StoreFileInfo> filteredSfis = null;
579      for (StoreFileInfo sfi: sfis) {
580        // Filter. There is a lag cleaning up compacted reference files. They get cleared
581        // after a delay in case outstanding Scanners still have references. Because of this,
582        // the listing of the Store content may have straggler reference files. Skip these.
583        // It should be safe to skip references at this point because we checked above with
584        // the region if it thinks it is splittable and if we are here, it thinks it is
585        // splitable.
586        if (sfi.isReference()) {
587          LOG.info("Skipping split of " + sfi + "; presuming ready for archiving.");
588          continue;
589        }
590        if (filteredSfis == null) {
591          filteredSfis = new ArrayList<StoreFileInfo>(sfis.size());
592          files.put(family, filteredSfis);
593        }
594        filteredSfis.add(sfi);
595        nbFiles++;
596      }
597    }
598    if (nbFiles == 0) {
599      // no file needs to be splitted.
600      return new Pair<Integer, Integer>(0,0);
601    }
602    // Max #threads is the smaller of the number of storefiles or the default max determined above.
603    int maxThreads = Math.min(
604      conf.getInt(HConstants.REGION_SPLIT_THREADS_MAX,
605        conf.getInt(HStore.BLOCKING_STOREFILES_KEY, HStore.DEFAULT_BLOCKING_STOREFILE_COUNT)),
606      nbFiles);
607    LOG.info("pid=" + getProcId() + " splitting " + nbFiles + " storefiles, region=" +
608      getParentRegion().getShortNameToLog() + ", threads=" + maxThreads);
609    final ExecutorService threadPool = Executors.newFixedThreadPool(
610      maxThreads, Threads.getNamedThreadFactory("StoreFileSplitter-%1$d"));
611    final List<Future<Pair<Path,Path>>> futures = new ArrayList<Future<Pair<Path,Path>>>(nbFiles);
612
613    TableDescriptor htd = env.getMasterServices().getTableDescriptors().get(getTableName());
614    // Split each store file.
615    for (Map.Entry<String, Collection<StoreFileInfo>>e: files.entrySet()) {
616      byte [] familyName = Bytes.toBytes(e.getKey());
617      final ColumnFamilyDescriptor hcd = htd.getColumnFamily(familyName);
618      final Collection<StoreFileInfo> storeFiles = e.getValue();
619      if (storeFiles != null && storeFiles.size() > 0) {
620        final CacheConfig cacheConf = new CacheConfig(conf, hcd);
621        for (StoreFileInfo storeFileInfo: storeFiles) {
622          StoreFileSplitter sfs =
623              new StoreFileSplitter(regionFs, familyName, new HStoreFile(mfs.getFileSystem(),
624                  storeFileInfo, conf, cacheConf, hcd.getBloomFilterType(), true));
625          futures.add(threadPool.submit(sfs));
626        }
627      }
628    }
629    // Shutdown the pool
630    threadPool.shutdown();
631
632    // Wait for all the tasks to finish.
633    // When splits ran on the RegionServer, how-long-to-wait-configuration was named
634    // hbase.regionserver.fileSplitTimeout. If set, use its value.
635    long fileSplitTimeout = conf.getLong("hbase.master.fileSplitTimeout",
636        conf.getLong("hbase.regionserver.fileSplitTimeout", 600000));
637    try {
638      boolean stillRunning = !threadPool.awaitTermination(fileSplitTimeout, TimeUnit.MILLISECONDS);
639      if (stillRunning) {
640        threadPool.shutdownNow();
641        // wait for the thread to shutdown completely.
642        while (!threadPool.isTerminated()) {
643          Thread.sleep(50);
644        }
645        throw new IOException("Took too long to split the" +
646            " files and create the references, aborting split");
647      }
648    } catch (InterruptedException e) {
649      throw (InterruptedIOException)new InterruptedIOException().initCause(e);
650    }
651
652    int daughterA = 0;
653    int daughterB = 0;
654    // Look for any exception
655    for (Future<Pair<Path, Path>> future : futures) {
656      try {
657        Pair<Path, Path> p = future.get();
658        daughterA += p.getFirst() != null ? 1 : 0;
659        daughterB += p.getSecond() != null ? 1 : 0;
660      } catch (InterruptedException e) {
661        throw (InterruptedIOException) new InterruptedIOException().initCause(e);
662      } catch (ExecutionException e) {
663        throw new IOException(e);
664      }
665    }
666
667    if (LOG.isDebugEnabled()) {
668      LOG.debug("pid=" + getProcId() + " split storefiles for region " +
669        getParentRegion().getShortNameToLog() +
670          " Daughter A: " + daughterA + " storefiles, Daughter B: " +
671          daughterB + " storefiles.");
672    }
673    return new Pair<Integer, Integer>(daughterA, daughterB);
674  }

676  private void assertReferenceFileCount(final FileSystem fs, final int expectedReferenceFileCount,
677      final Path dir) throws IOException {
678    if (expectedReferenceFileCount != 0 &&
679        expectedReferenceFileCount != FSUtils.getRegionReferenceFileCount(fs, dir)) {
680      throw new IOException("Failing split. Expected reference file count isn't equal.");
681    }

684  private Pair<Path, Path> splitStoreFile(HRegionFileSystem regionFs, byte[] family, HStoreFile sf)
685    throws IOException {
686    if (LOG.isDebugEnabled()) {
687      LOG.debug("pid=" + getProcId() + " splitting started for store file: " +
688          sf.getPath() + " for region: " + getParentRegion().getShortNameToLog());
689    }
690
691    final byte[] splitRow = getSplitRow();
692    final String familyName = Bytes.toString(family);
693    final Path path_first = regionFs.splitStoreFile(this.daughter_1_RI, familyName, sf, splitRow,
694        false, splitPolicy);
695    final Path path_second = regionFs.splitStoreFile(this.daughter_2_RI, familyName, sf, splitRow,
696       true, splitPolicy);
697    if (LOG.isDebugEnabled()) {
698      LOG.debug("pid=" + getProcId() + " splitting complete for store file: " +
699          sf.getPath() + " for region: " + getParentRegion().getShortNameToLog());
700    }
701    return new Pair<Path,Path>(path_first, path_second);
702  }

704  /**
705   * Utility class used to do the file splitting / reference writing
706   * in parallel instead of sequentially.
707   */
708  private class StoreFileSplitter implements Callable<Pair<Path,Path>> {
709    private final HRegionFileSystem regionFs;
710    private final byte[] family;
711    private final HStoreFile sf;
712
713    /**
714     * Constructor that takes what it needs to split
715     * @param regionFs the file system
716     * @param family Family that contains the store file
717     * @param sf which file
718     */
719    public StoreFileSplitter(HRegionFileSystem regionFs, byte[] family, HStoreFile sf) {
720      this.regionFs = regionFs;
721      this.sf = sf;
722      this.family = family;
723    }
724
725    @Override
726    public Pair<Path,Path> call() throws IOException {
727      return splitStoreFile(regionFs, family, sf);
728    }
729  }

733   * @param env MasterProcedureEnv

735  private void preSplitRegionBeforeMETA(final MasterProcedureEnv env)
736      throws IOException, InterruptedException {
737    final List<Mutation> metaEntries = new ArrayList<Mutation>();
738    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
739    if (cpHost != null) {
740      cpHost.preSplitBeforeMETAAction(getSplitRow(), metaEntries, getUser());
741      try {
742        for (Mutation p : metaEntries) {
743          RegionInfo.parseRegionName(p.getRow());
744        }
745      } catch (IOException e) {
746        LOG.error("pid=" + getProcId() + " row key of mutation from coprocessor not parsable as "
747            + "region name."
748            + "Mutations from coprocessor should only for hbase:meta table.");
749        throw e;
750      }
751    }
752  }

792  private UnassignProcedure[] createUnassignProcedures(final MasterProcedureEnv env,
793      final int regionReplication) {
794    final UnassignProcedure[] procs = new UnassignProcedure[regionReplication];
795    for (int i = 0; i < procs.length; ++i) {
796      final RegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(getParentRegion(), i);
797      procs[i] = env.getAssignmentManager().createUnassignProcedure(hri, null, true);
798    }
799    return procs;
800  }

802  private AssignProcedure[] createAssignProcedures(final MasterProcedureEnv env,
803      final int regionReplication) {
804    final ServerName targetServer = getParentRegionServerName(env);
805    final AssignProcedure[] procs = new AssignProcedure[regionReplication * 2];
806    int procsIdx = 0;
807    for (int i = 0; i < regionReplication; ++i) {
808      final RegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(daughter_1_RI, i);
809      procs[procsIdx++] = env.getAssignmentManager().createAssignProcedure(hri, targetServer);
810    }
811    for (int i = 0; i < regionReplication; ++i) {
812      final RegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(daughter_2_RI, i);
813      procs[procsIdx++] = env.getAssignmentManager().createAssignProcedure(hri, targetServer);
814    }
815    return procs;
816  }
