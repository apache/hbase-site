279  private void mirrorMetaLocations() throws IOException, KeeperException {
280    // For compatibility, mirror the meta region state to zookeeper
281    // And we still need to use zookeeper to publish the meta region locations to region
282    // server, so they can serve as ClientMetaService
283    ZKWatcher zk = master.getZooKeeper();
284    if (zk == null || !zk.getRecoverableZooKeeper().getState().isAlive()) {
285      // this is possible in tests, we do not provide a zk watcher or the zk watcher has been closed
286      return;
287    }
288    Collection<RegionStateNode> metaStates = regionStates.getRegionStateNodes();
289    for (RegionStateNode metaState : metaStates) {
290      MetaTableLocator.setMetaLocation(zk, metaState.getRegionLocation(),
291        metaState.getRegionInfo().getReplicaId(), metaState.getState());
292    }
293    int replicaCount = metaStates.size();
294    // remove extra mirror locations
295    for (String znode : zk.getMetaReplicaNodes()) {
296      int replicaId = zk.getZNodePaths().getMetaReplicaIdFromZNode(znode);
297      if (replicaId >= replicaCount) {
298        MetaTableLocator.deleteMetaLocation(zk, replicaId);
299      }
300    }
301  }

303  public void start() throws IOException, KeeperException {
304    if (!running.compareAndSet(false, true)) {
305      return;
306    }
307
308    LOG.trace("Starting assignment manager");
309
310    // Start the Assignment Thread
311    startAssignmentThread();
312    // load meta region states.
313    // here we are still in the early steps of active master startup. There is only one thread(us)
314    // can access AssignmentManager and create region node, so here we do not need to lock the
315    // region node.
316    try (ResultScanner scanner =
317      masterRegion.getScanner(new Scan().addFamily(HConstants.CATALOG_FAMILY))) {
318      for (;;) {
319        Result result = scanner.next();
320        if (result == null) {
321          break;
322        }
323        RegionStateStore
324          .visitMetaEntry((r, regionInfo, state, regionLocation, lastHost, openSeqNum) -> {
325            RegionStateNode regionNode = regionStates.getOrCreateRegionStateNode(regionInfo);
326            regionNode.setState(state);
327            regionNode.setLastHost(lastHost);
328            regionNode.setRegionLocation(regionLocation);
329            regionNode.setOpenSeqNum(openSeqNum);
330            if (regionNode.getProcedure() != null) {
331              regionNode.getProcedure().stateLoaded(this, regionNode);
332            }
333            if (regionLocation != null) {
334              regionStates.addRegionToServer(regionNode);
335            }
336            if (RegionReplicaUtil.isDefaultReplica(regionInfo.getReplicaId())) {
337              setMetaAssigned(regionInfo, state == State.OPEN);
338            }
339            LOG.debug("Loaded hbase:meta {}", regionNode);
340          }, result);
341      }
342    }
343    mirrorMetaLocations();
344  }

346  /**
347   * Create RegionStateNode based on the TRSP list, and attach the TRSP to the RegionStateNode.
348   * <p>
349   * This is used to restore the RIT region list, so we do not need to restore it in the loadingMeta
350   * method below. And it is also very important as now before submitting a TRSP, we need to attach
351   * it to the RegionStateNode, which acts like a guard, so we need to restore this information at
352   * the very beginning, before we start processing any procedures.
353   */
354  public void setupRIT(List<TransitRegionStateProcedure> procs) {
355    procs.forEach(proc -> {
356      RegionInfo regionInfo = proc.getRegion();
357      RegionStateNode regionNode = regionStates.getOrCreateRegionStateNode(regionInfo);
358      TransitRegionStateProcedure existingProc = regionNode.getProcedure();
359      if (existingProc != null) {
360        // This is possible, as we will detach the procedure from the RSN before we
361        // actually finish the procedure. This is because that, we will detach the TRSP from the RSN
362        // during execution, at that time, the procedure has not been marked as done in the pv2
363        // framework yet, so it is possible that we schedule a new TRSP immediately and when
364        // arriving here, we will find out that there are multiple TRSPs for the region. But we can
365        // make sure that, only the last one can take the charge, the previous ones should have all
366        // been finished already. So here we will compare the proc id, the greater one will win.
367        if (existingProc.getProcId() < proc.getProcId()) {
368          // the new one wins, unset and set it to the new one below
369          regionNode.unsetProcedure(existingProc);
370        } else {
371          // the old one wins, skip
372          return;
373        }
374      }
375      LOG.info("Attach {} to {} to restore RIT", proc, regionNode);
376      regionNode.setProcedure(proc);
377    });
378  }

380  public void stop() {
381    if (!running.compareAndSet(true, false)) {
382      return;
383    }
384
385    LOG.info("Stopping assignment manager");
386
387    // The AM is started before the procedure executor,
388    // but the actual work will be loaded/submitted only once we have the executor
389    final boolean hasProcExecutor = master.getMasterProcedureExecutor() != null;
390
391    // Remove the RIT chore
392    if (hasProcExecutor) {
393      master.getMasterProcedureExecutor().removeChore(this.ritChore);
394      if (this.deadMetricChore != null) {
395        master.getMasterProcedureExecutor().removeChore(this.deadMetricChore);
396      }
397    }
398
399    // Stop the Assignment Thread
400    stopAssignmentThread();
401
402    // Stop the RegionStateStore
403    regionStates.clear();
404
405    // Update meta events (for testing)
406    if (hasProcExecutor) {
407      metaLoadEvent.suspend();
408      for (RegionInfo hri : getMetaRegionSet()) {
409        setMetaAssigned(hri, false);
410      }
411    }
412  }

664   * Start a new thread to check if there are region servers whose versions are higher than others.
665   * If so, move all system table regions to RS with the highest version to keep compatibility. The
666   * reason is, RS in new version may not be able to access RS in old version when there are some
667   * incompatible changes.
668   * <p>
669   * This method is called when a new RegionServer is added to cluster only.
670   * </p>
671   */
672  public void checkIfShouldMoveSystemRegionAsync() {
673    // TODO: Fix this thread. If a server is killed and a new one started, this thread thinks that
674    // it should 'move' the system tables from the old server to the new server but
675    // ServerCrashProcedure is on it; and it will take care of the assign without dataloss.
676    if (this.master.getServerManager().countOfRegionServers() <= 1) {
677      return;
678    }
679    // This thread used to run whenever there was a change in the cluster. The ZooKeeper
680    // childrenChanged notification came in before the nodeDeleted message and so this method
681    // cold run before a ServerCrashProcedure could run. That meant that this thread could see
682    // a Crashed Server before ServerCrashProcedure and it could find system regions on the
683    // crashed server and go move them before ServerCrashProcedure had a chance; could be
684    // dataloss too if WALs were not recovered.
685    new Thread(() -> {
686      try {
687        synchronized (checkIfShouldMoveSystemRegionLock) {
688          List<RegionPlan> plans = new ArrayList<>();
689          // TODO: I don't think this code does a good job if all servers in cluster have same
690          // version. It looks like it will schedule unnecessary moves.
691          for (ServerName server : getExcludedServersForSystemTable()) {
692            if (master.getServerManager().isServerDead(server)) {
693              // TODO: See HBASE-18494 and HBASE-18495. Though getExcludedServersForSystemTable()
694              // considers only online servers, the server could be queued for dead server
695              // processing. As region assignments for crashed server is handled by
696              // ServerCrashProcedure, do NOT handle them here. The goal is to handle this through
697              // regular flow of LoadBalancer as a favored node and not to have this special
698              // handling.
699              continue;
700            }
701            List<RegionInfo> regionsShouldMove = getSystemTables(server);
702            if (!regionsShouldMove.isEmpty()) {
703              for (RegionInfo regionInfo : regionsShouldMove) {
704                // null value for dest forces destination server to be selected by balancer
705                RegionPlan plan = new RegionPlan(regionInfo, server, null);
706                if (regionInfo.isMetaRegion()) {
707                  // Must move meta region first.
708                  LOG.info("Async MOVE of {} to newer Server={}", regionInfo.getEncodedName(),
709                    server);
710                  moveAsync(plan);
711                } else {
712                  plans.add(plan);
713                }
714              }
715            }
716            for (RegionPlan plan : plans) {
717              LOG.info("Async MOVE of {} to newer Server={}", plan.getRegionInfo().getEncodedName(),
718                server);
719              moveAsync(plan);
720            }
721          }
722        }
723      } catch (Throwable t) {
724        LOG.error(t.toString(), t);
725      }
726    }).start();
727  }

753   * not appropriate UNLESS override is set. Used by hbck2 but also by straightline
754   * {@link #assign(RegionInfo, ServerName)} and {@link #assignAsync(RegionInfo, ServerName)}.
755   * @see #createAssignProcedure(RegionStateNode, ServerName) for a version that does NO checking
756   *      used when only when no checking needed.
757   * @param override If false, check RegionState is appropriate for assign; if not throw exception.

759  private TransitRegionStateProcedure createAssignProcedure(RegionInfo regionInfo, ServerName sn,
760    boolean override) throws IOException {
761    RegionStateNode regionNode = regionStates.getOrCreateRegionStateNode(regionInfo);
762    regionNode.lock();
763    try {

765        if (regionNode.getProcedure() != null) {
766          regionNode.unsetProcedure(regionNode.getProcedure());
767        }
768      } else {
769        preTransitCheck(regionNode, STATES_EXPECTED_ON_ASSIGN);
770      }
771      assert regionNode.getProcedure() == null;
772      return regionNode.setProcedure(
773        TransitRegionStateProcedure.assign(getProcedureEnvironment(), regionInfo, sn));
774    } finally {
775      regionNode.unlock();
776    }
777  }

889   * @return AssignProcedures made out of the passed in <code>hris</code> and a call to the balancer
890   *         to populate the assigns with targets chosen using round-robin (default balancer
891   *         scheme). If at assign-time, the target chosen is no longer up, thats fine, the
892   *         AssignProcedure will ask the balancer for a new target, and so on.
893   */
894  public TransitRegionStateProcedure[] createRoundRobinAssignProcedures(List<RegionInfo> hris,
895    List<ServerName> serversToExclude) {
896    if (hris.isEmpty()) {
897      return new TransitRegionStateProcedure[0];
898    }
899
900    if (
901      serversToExclude != null && this.master.getServerManager().getOnlineServersList().size() == 1
902    ) {
903      LOG.debug("Only one region server found and hence going ahead with the assignment");
904      serversToExclude = null;
905    }
906    try {
907      // Ask the balancer to assign our regions. Pass the regions en masse. The balancer can do
908      // a better job if it has all the assignments in the one lump.
909      Map<ServerName, List<RegionInfo>> assignments = getBalancer().roundRobinAssignment(hris,
910        this.master.getServerManager().createDestinationServersList(serversToExclude));
911      // Return mid-method!
912      return createAssignProcedures(assignments);
913    } catch (IOException hioe) {
914      LOG.warn("Failed roundRobinAssignment", hioe);
915    }
916    // If an error above, fall-through to this simpler assign. Last resort.
917    return createAssignProcedures(hris);
918  }

931  static int compare(TransitRegionStateProcedure left, TransitRegionStateProcedure right) {
932    if (left.getRegion().isMetaRegion()) {
933      if (right.getRegion().isMetaRegion()) {
934        return RegionInfo.COMPARATOR.compare(left.getRegion(), right.getRegion());
935      }
936      return -1;
937    } else if (right.getRegion().isMetaRegion()) {
938      return +1;
939    }
940    if (left.getRegion().getTable().isSystemTable()) {
941      if (right.getRegion().getTable().isSystemTable()) {
942        return RegionInfo.COMPARATOR.compare(left.getRegion(), right.getRegion());
943      }
944      return -1;
945    } else if (right.getRegion().getTable().isSystemTable()) {
946      return +1;
947    }
948    return RegionInfo.COMPARATOR.compare(left.getRegion(), right.getRegion());
949  }

1034  // for creating unassign TRSP when disabling a table or closing excess region replicas
1035  private TransitRegionStateProcedure forceCreateUnssignProcedure(RegionStateNode regionNode) {
1036    regionNode.lock();
1037    try {
1038      if (regionNode.isInState(State.OFFLINE, State.CLOSED, State.SPLIT)) {
1039        return null;
1040      }
1041      // in general, a split parent should be in CLOSED or SPLIT state, but anyway, let's check it
1042      // here for safety
1043      if (regionNode.getRegionInfo().isSplit()) {
1044        LOG.warn("{} is a split parent but not in CLOSED or SPLIT state", regionNode);
1045        return null;
1046      }
1047      // As in DisableTableProcedure or ModifyTableProcedure, we will hold the xlock for table, so
1048      // we can make sure that this procedure has not been executed yet, as TRSP will hold the
1049      // shared lock for table all the time. So here we will unset it and when it is actually
1050      // executed, it will find that the attach procedure is not itself and quit immediately.
1051      if (regionNode.getProcedure() != null) {
1052        regionNode.unsetProcedure(regionNode.getProcedure());
1053      }
1054      return regionNode.setProcedure(TransitRegionStateProcedure.unassign(getProcedureEnvironment(),
1055        regionNode.getRegionInfo()));
1056    } finally {
1057      regionNode.unlock();
1058    }
1059  }

1110  private void reportRegionStateTransition(ReportRegionStateTransitionResponse.Builder builder,
1111    ServerName serverName, List<RegionStateTransition> transitionList) throws IOException {
1112    for (RegionStateTransition transition : transitionList) {
1113      switch (transition.getTransitionCode()) {
1114        case OPENED:
1115        case FAILED_OPEN:
1116        case CLOSED:
1117          assert transition.getRegionInfoCount() == 1 : transition;
1118          final RegionInfo hri = ProtobufUtil.toRegionInfo(transition.getRegionInfo(0));
1119          long procId =
1120            transition.getProcIdCount() > 0 ? transition.getProcId(0) : Procedure.NO_PROC_ID;
1121          updateRegionTransition(serverName, transition.getTransitionCode(), hri,
1122            transition.hasOpenSeqNum() ? transition.getOpenSeqNum() : HConstants.NO_SEQNUM, procId);
1123          break;
1124        case READY_TO_SPLIT:
1125        case SPLIT:
1126        case SPLIT_REVERTED:
1127          assert transition.getRegionInfoCount() == 3 : transition;
1128          final RegionInfo parent = ProtobufUtil.toRegionInfo(transition.getRegionInfo(0));
1129          final RegionInfo splitA = ProtobufUtil.toRegionInfo(transition.getRegionInfo(1));
1130          final RegionInfo splitB = ProtobufUtil.toRegionInfo(transition.getRegionInfo(2));
1131          updateRegionSplitTransition(serverName, transition.getTransitionCode(), parent, splitA,
1132            splitB);
1133          break;
1134        case READY_TO_MERGE:
1135        case MERGED:
1136        case MERGE_REVERTED:
1137          assert transition.getRegionInfoCount() == 3 : transition;
1138          final RegionInfo merged = ProtobufUtil.toRegionInfo(transition.getRegionInfo(0));
1139          final RegionInfo mergeA = ProtobufUtil.toRegionInfo(transition.getRegionInfo(1));
1140          final RegionInfo mergeB = ProtobufUtil.toRegionInfo(transition.getRegionInfo(2));

1148  public ReportRegionStateTransitionResponse reportRegionStateTransition(
1149    final ReportRegionStateTransitionRequest req) throws PleaseHoldException {
1150    ReportRegionStateTransitionResponse.Builder builder =
1151      ReportRegionStateTransitionResponse.newBuilder();
1152    ServerName serverName = ProtobufUtil.toServerName(req.getServer());
1153    ServerStateNode serverNode = regionStates.getOrCreateServer(serverName);
1154    // here we have to acquire a read lock instead of a simple exclusive lock. This is because that
1155    // we should not block other reportRegionStateTransition call from the same region server. This
1156    // is not only about performance, but also to prevent dead lock. Think of the meta region is
1157    // also on the same region server and you hold the lock which blocks the
1158    // reportRegionStateTransition for meta, and since meta is not online, you will block inside the
1159    // lock protection to wait for meta online...
1160    serverNode.readLock().lock();
1161    try {
1162      // we only accept reportRegionStateTransition if the region server is online, see the comment
1163      // above in submitServerCrash method and HBASE-21508 for more details.
1164      if (serverNode.isInState(ServerState.ONLINE)) {
1165        try {
1166          reportRegionStateTransition(builder, serverName, req.getTransitionList());
1167        } catch (PleaseHoldException e) {
1168          LOG.trace("Failed transition ", e);
1169          throw e;
1170        } catch (UnsupportedOperationException | IOException e) {
1171          // TODO: at the moment we have a single error message and the RS will abort
1172          // if the master says that one of the region transitions failed.
1173          LOG.warn("Failed transition", e);
1174          builder.setErrorMessage("Failed transition " + e.getMessage());
1175        }
1176      } else {
1177        LOG.warn("The region server {} is already dead, skip reportRegionStateTransition call",
1178          serverName);
1179        builder.setErrorMessage("You are dead");
1180      }
1181    } finally {
1182      serverNode.readLock().unlock();
1183    }
1184
1185    return builder.build();
1186  }

1189    RegionInfo regionInfo, long seqId, long procId) throws IOException {
1190    checkMetaLoaded(regionInfo);
1191
1192    RegionStateNode regionNode = regionStates.getRegionStateNode(regionInfo);
1193    if (regionNode == null) {
1194      // the table/region is gone. maybe a delete, split, merge
1195      throw new UnexpectedStateException(String.format(
1196        "Server %s was trying to transition region %s to %s. but Region is not known.", serverName,
1197        regionInfo, state));
1198    }
1199    LOG.trace("Update region transition serverName={} region={} regionState={}", serverName,
1200      regionNode, state);
1201
1202    ServerStateNode serverNode = regionStates.getOrCreateServer(serverName);
1203    regionNode.lock();
1204    try {
1205      if (!reportTransition(regionNode, serverNode, state, seqId, procId)) {
1206        // Don't log WARN if shutting down cluster; during shutdown. Avoid the below messages:
1207        // 2018-08-13 10:45:10,551 WARN ...AssignmentManager: No matching procedure found for

1209        // table=IntegrationTestBigLinkedList, region=65ab289e2fc1530df65f6c3d7cde7aa5 transition
1210        // to CLOSED

1212        // regions. This is the only time that region close is not run by the Master (so cluster

1214        if (
1215          this.master.getServerManager().isClusterShutdown() && state.equals(TransitionCode.CLOSED)

1217          LOG.info("RegionServer {} {}", state, regionNode.getRegionInfo().getEncodedName());
1218        } else {
1219          LOG.warn("No matching procedure found for {} transition on {} to {}", serverName,

1240  private void updateRegionSplitTransition(final ServerName serverName, final TransitionCode state,
1241    final RegionInfo parent, final RegionInfo hriA, final RegionInfo hriB) throws IOException {
1242    checkMetaLoaded(parent);
1243
1244    if (state != TransitionCode.READY_TO_SPLIT) {
1245      throw new UnexpectedStateException(
1246        "unsupported split regionState=" + state + " for parent region " + parent
1247          + " maybe an old RS (< 2.0) had the operation in progress");
1248    }
1249
1250    // sanity check on the request
1251    if (!Bytes.equals(hriA.getEndKey(), hriB.getStartKey())) {
1252      throw new UnsupportedOperationException("unsupported split request with bad keys: parent="
1253        + parent + " hriA=" + hriA + " hriB=" + hriB);
1254    }
1255
1256    if (!master.isSplitOrMergeEnabled(MasterSwitchType.SPLIT)) {
1257      LOG.warn("Split switch is off! skip split of " + parent);
1258      throw new DoNotRetryIOException(
1259        "Split region " + parent.getRegionNameAsString() + " failed due to split switch off");
1260    }
1261
1262    // Submit the Split procedure
1263    final byte[] splitKey = hriB.getStartKey();
1264    if (LOG.isDebugEnabled()) {
1265      LOG.debug("Split request from " + serverName + ", parent=" + parent + " splitKey="
1266        + Bytes.toStringBinary(splitKey));
1267    }
1268    // Processing this report happens asynchronously from other activities which can mutate
1269    // the region state. For example, a split procedure may already be running for this parent.
1270    // A split procedure cannot succeed if the parent region is no longer open, so we can
1271    // ignore it in that case.
1272    // Note that submitting more than one split procedure for a given region is
1273    // harmless -- the split is fenced in the procedure handling -- but it would be noisy in
1274    // the logs. Only one procedure can succeed. The other procedure(s) would abort during
1275    // initialization and report failure with WARN level logging.
1276    RegionState parentState = regionStates.getRegionState(parent);
1277    if (parentState != null && parentState.isOpened()) {
1278      master.getMasterProcedureExecutor().submitProcedure(createSplitProcedure(parent, splitKey));
1279    } else {
1280      LOG.info("Ignoring split request from " + serverName + ", parent=" + parent
1281        + " because parent is unknown or not open");
1282      return;
1283    }
1284
1285    // If the RS is < 2.0 throw an exception to abort the operation, we are handling the split
1286    if (master.getServerManager().getVersionNumber(serverName) < 0x0200000) {
1287      throw new UnsupportedOperationException(
1288        String.format("Split handled by the master: " + "parent=%s hriA=%s hriB=%s",
1289          parent.getShortNameToLog(), hriA, hriB));
1290    }
1291  }

1327   * The master will call this method when the RS send the regionServerReport(). The report will
1328   * contains the "online regions". This method will check the the online regions against the
1329   * in-memory state of the AM, and we will log a warn message if there is a mismatch. This is
1330   * because that there is no fencing between the reportRegionStateTransition method and
1331   * regionServerReport method, so there could be race and introduce inconsistency here, but
1332   * actually there is no problem.
1333   * <p/>
1334   * Please see HBASE-21421 and HBASE-21463 for more details.
1335   */
1336  public void reportOnlineRegions(ServerName serverName, Set<byte[]> regionNames) {
1337    if (!isRunning()) {
1338      return;
1339    }
1340    if (LOG.isTraceEnabled()) {
1341      LOG.trace("ReportOnlineRegions {} regionCount={}, metaLoaded={} {}", serverName,
1342        regionNames.size(), isMetaLoaded(),
1343        regionNames.stream().map(Bytes::toStringBinary).collect(Collectors.toList()));
1344    }
1345
1346    ServerStateNode serverNode = regionStates.getOrCreateServer(serverName);
1347    synchronized (serverNode) {
1348      if (!serverNode.isInState(ServerState.ONLINE)) {
1349        LOG.warn("Got a report from a server result in state " + serverNode.getState());
1350        return;
1351      }
1352    }
1353
1354    // Track the regionserver reported online regions in memory.
1355    synchronized (rsReports) {
1356      rsReports.put(serverName, regionNames);
1357    }
1358
1359    if (regionNames.isEmpty()) {
1360      // nothing to do if we don't have regions
1361      LOG.trace("no online region found on {}", serverName);
1362      return;
1363    }
1364    if (!isMetaLoaded()) {
1365      // we are still on startup, skip checking
1366      return;
1367    }
1368    // The Heartbeat tells us of what regions are on the region serve, check the state.
1369    checkOnlineRegionsReport(serverNode, regionNames);

1388  /**
1389   * Check that what the RegionServer reports aligns with the Master's image. If disagreement, we
1390   * will tell the RegionServer to expediently close a Region we do not think it should have.
1391   */
1392  private void checkOnlineRegionsReport(ServerStateNode serverNode, Set<byte[]> regionNames) {
1393    ServerName serverName = serverNode.getServerName();
1394    for (byte[] regionName : regionNames) {
1395      if (!isRunning()) {
1396        return;
1397      }
1398      RegionStateNode regionNode = regionStates.getRegionStateNodeFromName(regionName);
1399      if (regionNode == null) {
1400        String regionNameAsStr = Bytes.toStringBinary(regionName);
1401        LOG.warn("No RegionStateNode for {} but reported as up on {}; closing...", regionNameAsStr,
1402          serverName);
1403        closeRegionSilently(serverNode.getServerName(), regionName);
1404        continue;
1405      }
1406      final long lag = 1000;
1407      // This is just a fallback check designed to identify unexpected data inconsistencies, so we
1408      // use tryLock to attempt to acquire the lock, and if the lock cannot be acquired, we skip the
1409      // check. This will not cause any additional problems and also prevents the regionServerReport
1410      // call from being stuck for too long which may cause deadlock on region assignment.
1411      if (regionNode.tryLock()) {
1412        try {
1413          long diff = EnvironmentEdgeManager.currentTime() - regionNode.getLastUpdate();
1414          if (regionNode.isInState(State.OPENING, State.OPEN)) {
1415            // This is possible as a region server has just closed a region but the region server
1416            // report is generated before the closing, but arrive after the closing. Make sure
1417            // there
1418            // is some elapsed time so less false alarms.
1419            if (!regionNode.getRegionLocation().equals(serverName) && diff > lag) {
1420              LOG.warn("Reporting {} server does not match {} (time since last "
1421                + "update={}ms); closing...", serverName, regionNode, diff);
1422              closeRegionSilently(serverNode.getServerName(), regionName);
1423            }
1424          } else if (!regionNode.isInState(State.CLOSING, State.SPLITTING)) {
1425            // So, we can get report that a region is CLOSED or SPLIT because a heartbeat
1426            // came in at about same time as a region transition. Make sure there is some
1427            // elapsed time so less false alarms.
1428            if (diff > lag) {
1429              LOG.warn("Reporting {} state does not match {} (time since last update={}ms)",
1430                serverName, regionNode, diff);
1431            }
1432          }
1433        } finally {
1434          regionNode.unlock();
1435        }
1436      } else {
1437        LOG.warn(
1438          "Unable to acquire lock for regionNode {}. It is likely that another thread is currently holding the lock. To avoid deadlock, skip execution for now.",
1439          regionNode);
1440      }
1441    }
1442  }

1468  private static class DeadServerMetricRegionChore
1469    extends ProcedureInMemoryChore<MasterProcedureEnv> {
1470    public DeadServerMetricRegionChore(final int timeoutMsec) {
1471      super(timeoutMsec);
1472    }
1473
1474    @Override
1475    protected void periodicExecute(final MasterProcedureEnv env) {
1476      final ServerManager sm = env.getMasterServices().getServerManager();
1477      final AssignmentManager am = env.getAssignmentManager();
1478      // To minimize inconsistencies we are not going to snapshot live servers in advance in case
1479      // new servers are added; OTOH we don't want to add heavy sync for a consistent view since
1480      // this is for metrics. Instead, we're going to check each regions as we go; to avoid making
1481      // too many checks, we maintain a local lists of server, limiting us to false negatives. If
1482      // we miss some recently-dead server, we'll just see it next time.
1483      Set<ServerName> recentlyLiveServers = new HashSet<>();
1484      int deadRegions = 0, unknownRegions = 0;
1485      for (RegionStateNode rsn : am.getRegionStates().getRegionStateNodes()) {
1486        if (rsn.getState() != State.OPEN) {
1487          continue; // Opportunistic check, should quickly skip RITs, offline tables, etc.
1488        }
1489        // Do not need to acquire region state lock as this is only for showing metrics.
1490        ServerName sn = rsn.getRegionLocation();
1491        State state = rsn.getState();
1492        if (state != State.OPEN) {
1493          continue; // Mostly skipping RITs that are already being take care of.
1494        }
1495        if (sn == null) {
1496          ++unknownRegions; // Opened on null?
1497          continue;
1498        }
1499        if (recentlyLiveServers.contains(sn)) {
1500          continue;
1501        }
1502        ServerManager.ServerLiveState sls = sm.isServerKnownAndOnline(sn);
1503        switch (sls) {
1504          case LIVE:
1505            recentlyLiveServers.add(sn);
1506            break;
1507          case DEAD:
1508            ++deadRegions;
1509            break;
1510          case UNKNOWN:
1511            ++unknownRegions;
1512            break;
1513          default:
1514            throw new AssertionError("Unexpected " + sls);
1515        }
1516      }
1517      if (deadRegions > 0 || unknownRegions > 0) {
1518        LOG.info("Found {} OPEN regions on dead servers and {} OPEN regions on unknown servers",
1519          deadRegions, unknownRegions);
1520      }
1521
1522      am.updateDeadServerRegionMetrics(deadRegions, unknownRegions);
1523    }
1524  }

1712  /*
1713   * AM internal RegionStateStore.RegionStateVisitor implementation. To be used when scanning META
1714   * table for region rows, using RegionStateStore utility methods. RegionStateStore methods will
1715   * convert Result into proper RegionInfo instances, but those would still need to be added into
1716   * AssignmentManager.regionStates in-memory cache. RegionMetaLoadingVisitor.visitRegionState
1717   * method provides the logic for adding RegionInfo instances as loaded from latest META scan into
1718   * AssignmentManager.regionStates.
1719   */
1720  private class RegionMetaLoadingVisitor implements RegionStateStore.RegionStateVisitor {
1721
1722    @Override
1723    public void visitRegionState(Result result, final RegionInfo regionInfo, final State state,
1724      final ServerName regionLocation, final ServerName lastHost, final long openSeqNum) {
1725      if (
1726        state == null && regionLocation == null && lastHost == null
1727          && openSeqNum == SequenceId.NO_SEQUENCE_ID
1728      ) {
1729        // This is a row with nothing in it.
1730        LOG.warn("Skipping empty row={}", result);
1731        return;
1732      }
1733      State localState = state;
1734      if (localState == null) {
1735        // No region state column data in hbase:meta table! Are I doing a rolling upgrade from
1736        // hbase1 to hbase2? Am I restoring a SNAPSHOT or otherwise adding a region to hbase:meta?
1737        // In any of these cases, state is empty. For now, presume OFFLINE but there are probably
1738        // cases where we need to probe more to be sure this correct; TODO informed by experience.
1739        LOG.info(regionInfo.getEncodedName() + " regionState=null; presuming " + State.OFFLINE);
1740        localState = State.OFFLINE;
1741      }
1742      RegionStateNode regionNode = regionStates.getOrCreateRegionStateNode(regionInfo);
1743      // Do not need to lock on regionNode, as we can make sure that before we finish loading
1744      // meta, all the related procedures can not be executed. The only exception is for meta
1745      // region related operations, but here we do not load the informations for meta region.
1746      regionNode.setState(localState);
1747      regionNode.setLastHost(lastHost);
1748      regionNode.setRegionLocation(regionLocation);
1749      regionNode.setOpenSeqNum(openSeqNum);
1750
1751      // Note: keep consistent with other methods, see region(Opening|Opened|Closing)
1752      // RIT/ServerCrash handling should take care of the transiting regions.
1753      if (
1754        localState.matches(State.OPEN, State.OPENING, State.CLOSING, State.SPLITTING, State.MERGING)
1755      ) {
1756        assert regionLocation != null : "found null region location for " + regionNode;
1757        regionStates.addRegionToServer(regionNode);
1758      } else if (localState == State.OFFLINE || regionInfo.isOffline()) {
1759        regionStates.addToOfflineRegions(regionNode);
1760      }
1761      if (regionNode.getProcedure() != null) {
1762        regionNode.getProcedure().stateLoaded(AssignmentManager.this, regionNode);
1763      }
1764    }
1765  };

1825  /**
1826   * Usually run by the Master in reaction to server crash during normal processing. Can also be
1827   * invoked via external RPC to effect repair; in the latter case, the 'force' flag is set so we
1828   * push through the SCP though context may indicate already-running-SCP (An old SCP may have
1829   * exited abnormally, or damaged cluster may still have references in hbase:meta to 'Unknown
1830   * Servers' -- servers that are not online or in dead servers list, etc.)
1831   * @param force Set if the request came in externally over RPC (via hbck2). Force means run the
1832   *              SCP even if it seems as though there might be an outstanding SCP running.
1833   * @return pid of scheduled SCP or {@link Procedure#NO_PROC_ID} if none scheduled.
1834   */
1835  public long submitServerCrash(ServerName serverName, boolean shouldSplitWal, boolean force) {
1836    // May be an 'Unknown Server' so handle case where serverNode is null.
1837    ServerStateNode serverNode = regionStates.getServerNode(serverName);
1838    // Remove the in-memory rsReports result
1839    synchronized (rsReports) {
1840      rsReports.remove(serverName);
1841    }
1842
1843    // We hold the write lock here for fencing on reportRegionStateTransition. Once we set the
1844    // server state to CRASHED, we will no longer accept the reportRegionStateTransition call from
1845    // this server. This is used to simplify the implementation for TRSP and SCP, where we can make
1846    // sure that, the region list fetched by SCP will not be changed any more.
1847    if (serverNode != null) {
1848      serverNode.writeLock().lock();
1849    }
1850    boolean carryingMeta;
1851    long pid;
1852    try {
1853      ProcedureExecutor<MasterProcedureEnv> procExec = this.master.getMasterProcedureExecutor();
1854      carryingMeta = isCarryingMeta(serverName);
1855      if (!force && serverNode != null && !serverNode.isInState(ServerState.ONLINE)) {
1856        LOG.info("Skip adding ServerCrashProcedure for {} (meta={}) -- running?", serverNode,
1857          carryingMeta);
1858        return Procedure.NO_PROC_ID;
1859      } else {
1860        MasterProcedureEnv mpe = procExec.getEnvironment();
1861        // If serverNode == null, then 'Unknown Server'. Schedule HBCKSCP instead.
1862        // HBCKSCP scours Master in-memory state AND hbase;meta for references to
1863        // serverName just-in-case. An SCP that is scheduled when the server is
1864        // 'Unknown' probably originated externally with HBCK2 fix-it tool.
1865        ServerState oldState = null;
1866        if (serverNode != null) {
1867          oldState = serverNode.getState();
1868          serverNode.setState(ServerState.CRASHED);
1869        }
1870
1871        if (force) {
1872          pid = procExec.submitProcedure(
1873            new HBCKServerCrashProcedure(mpe, serverName, shouldSplitWal, carryingMeta));
1874        } else {
1875          pid = procExec.submitProcedure(
1876            new ServerCrashProcedure(mpe, serverName, shouldSplitWal, carryingMeta));
1877        }
1878        LOG.info("Scheduled ServerCrashProcedure pid={} for {} (carryingMeta={}){}.", pid,
1879          serverName, carryingMeta,
1880          serverNode == null ? "" : " " + serverNode.toString() + ", oldState=" + oldState);
1881      }
1882    } finally {
1883      if (serverNode != null) {
1884        serverNode.writeLock().unlock();
1885      }
1886    }
1887    return pid;
1888  }

2279  private void processAssignQueue() {
2280    final HashMap<RegionInfo, RegionStateNode> regions = waitOnAssignQueue();
2281    if (regions == null || regions.size() == 0 || !isRunning()) {
2282      return;
2283    }
2284
2285    if (LOG.isTraceEnabled()) {
2286      LOG.trace("PROCESS ASSIGN QUEUE regionCount=" + regions.size());
2287    }
2288
2289    // TODO: Optimize balancer. pass a RegionPlan?
2290    final HashMap<RegionInfo, ServerName> retainMap = new HashMap<>();
2291    final List<RegionInfo> userHRIs = new ArrayList<>(regions.size());
2292    // Regions for system tables requiring reassignment
2293    final List<RegionInfo> systemHRIs = new ArrayList<>();
2294    for (RegionStateNode regionStateNode : regions.values()) {
2295      boolean sysTable = regionStateNode.isSystemTable();
2296      final List<RegionInfo> hris = sysTable ? systemHRIs : userHRIs;
2297      if (regionStateNode.getRegionLocation() != null) {
2298        retainMap.put(regionStateNode.getRegionInfo(), regionStateNode.getRegionLocation());
2299      } else {
2300        hris.add(regionStateNode.getRegionInfo());
2301      }
2302    }
2303
2304    // TODO: connect with the listener to invalidate the cache
2305
2306    // TODO use events
2307    List<ServerName> servers = master.getServerManager().createDestinationServersList();
2308    for (int i = 0; servers.size() < 1; ++i) {
2309      // Report every fourth time around this loop; try not to flood log.
2310      if (i % 4 == 0) {
2311        LOG.warn("No servers available; cannot place " + regions.size() + " unassigned regions.");
2312      }
2313
2314      if (!isRunning()) {
2315        LOG.debug("Stopped! Dropping assign of " + regions.size() + " queued regions.");
2316        return;
2317      }
2318      Threads.sleep(250);
2319      servers = master.getServerManager().createDestinationServersList();
2320    }
2321
2322    if (!systemHRIs.isEmpty()) {
2323      // System table regions requiring reassignment are present, get region servers
2324      // not available for system table regions
2325      final List<ServerName> excludeServers = getExcludedServersForSystemTable();
2326      List<ServerName> serversForSysTables =
2327        servers.stream().filter(s -> !excludeServers.contains(s)).collect(Collectors.toList());
2328      if (serversForSysTables.isEmpty()) {
2329        LOG.warn("Filtering old server versions and the excluded produced an empty set; "
2330          + "instead considering all candidate servers!");
2331      }
2332      LOG.debug("Processing assignQueue; systemServersCount=" + serversForSysTables.size()
2333        + ", allServersCount=" + servers.size());
2334      processAssignmentPlans(regions, null, systemHRIs,
2335        serversForSysTables.isEmpty() && !containsBogusAssignments(regions, systemHRIs)
2336          ? servers
2337          : serversForSysTables);
2338    }
2339
2340    processAssignmentPlans(regions, retainMap, userHRIs, servers);
2341  }

2356  private void processAssignmentPlans(final HashMap<RegionInfo, RegionStateNode> regions,
2357    final HashMap<RegionInfo, ServerName> retainMap, final List<RegionInfo> hris,
2358    final List<ServerName> servers) {
2359    boolean isTraceEnabled = LOG.isTraceEnabled();
2360    if (isTraceEnabled) {
2361      LOG.trace("Available servers count=" + servers.size() + ": " + servers);
2362    }
2363
2364    final LoadBalancer balancer = getBalancer();
2365    // ask the balancer where to place regions
2366    if (retainMap != null && !retainMap.isEmpty()) {
2367      if (isTraceEnabled) {
2368        LOG.trace("retain assign regions=" + retainMap);
2369      }
2370      try {
2371        acceptPlan(regions, balancer.retainAssignment(retainMap, servers));
2372      } catch (IOException e) {
2373        LOG.warn("unable to retain assignment", e);
2374        addToPendingAssignment(regions, retainMap.keySet());
2375      }
2376    }
2377
2378    // TODO: Do we need to split retain and round-robin?
2379    // the retain seems to fallback to round-robin/random if the region is not in the map.
2380    if (!hris.isEmpty()) {
2381      Collections.sort(hris, RegionInfo.COMPARATOR);
2382      if (isTraceEnabled) {
2383        LOG.trace("round robin regions=" + hris);
2384      }
2385      try {
2386        acceptPlan(regions, balancer.roundRobinAssignment(hris, servers));
2387      } catch (IOException e) {
2388        LOG.warn("unable to round-robin assignment", e);
2389        addToPendingAssignment(regions, hris);
2390      }
2391    }
2392  }

2394  private void acceptPlan(final HashMap<RegionInfo, RegionStateNode> regions,
2395    final Map<ServerName, List<RegionInfo>> plan) throws HBaseIOException {
2396    final ProcedureEvent<?>[] events = new ProcedureEvent[regions.size()];
2397    final long st = EnvironmentEdgeManager.currentTime();
2398
2399    if (plan.isEmpty()) {
2400      throw new HBaseIOException("unable to compute plans for regions=" + regions.size());
2401    }
2402
2403    int evcount = 0;
2404    for (Map.Entry<ServerName, List<RegionInfo>> entry : plan.entrySet()) {
2405      final ServerName server = entry.getKey();
2406      for (RegionInfo hri : entry.getValue()) {
2407        final RegionStateNode regionNode = regions.get(hri);
2408        regionNode.setRegionLocation(server);
2409        if (server.equals(LoadBalancer.BOGUS_SERVER_NAME) && regionNode.isSystemTable()) {
2410          assignQueueLock.lock();
2411          try {
2412            pendingAssignQueue.add(regionNode);
2413          } finally {
2414            assignQueueLock.unlock();
2415          }
2416        } else {
2417          events[evcount++] = regionNode.getProcedureEvent();
2418        }
2419      }
2420    }
2421    ProcedureEvent.wakeEvents(getProcedureScheduler(), events);
2422
2423    final long et = EnvironmentEdgeManager.currentTime();
2424    if (LOG.isTraceEnabled()) {
2425      LOG.trace("ASSIGN ACCEPT " + events.length + " -> " + StringUtils.humanTimeDiff(et - st));
2426    }
2427  }

2441  /**
2442   * For a given cluster with mixed versions of servers, get a list of servers with lower versions,
2443   * where system table regions should not be assigned to. For system table, we must assign regions
2444   * to a server with highest version. However, we can disable this exclusion using config:
2445   * "hbase.min.version.move.system.tables" if checkForMinVersion is true. Detailed explanation
2446   * available with definition of minVersionToMoveSysTables.
2447   * @return List of Excluded servers for System table regions.
2448   */
2449  public List<ServerName> getExcludedServersForSystemTable() {
2450    // TODO: This should be a cached list kept by the ServerManager rather than calculated on each
2451    // move or system region assign. The RegionServerTracker keeps list of online Servers with
2452    // RegionServerInfo that includes Version.
2453    List<Pair<ServerName, String>> serverList =
2454      master.getServerManager().getOnlineServersList().stream()
2455        .map(s -> new Pair<>(s, master.getRegionServerVersion(s))).collect(Collectors.toList());
2456    if (serverList.isEmpty()) {
2457      return new ArrayList<>();
2458    }
2459    String highestVersion = Collections
2460      .max(serverList, (o1, o2) -> VersionInfo.compareVersion(o1.getSecond(), o2.getSecond()))
2461      .getSecond();
2462    if (!DEFAULT_MIN_VERSION_MOVE_SYS_TABLES_CONFIG.equals(minVersionToMoveSysTables)) {
2463      int comparedValue = VersionInfo.compareVersion(minVersionToMoveSysTables, highestVersion);
2464      if (comparedValue > 0) {
2465        return new ArrayList<>();
2466      }
2467    }
2468    return serverList.stream().filter(pair -> !pair.getSecond().equals(highestVersion))
2469      .map(Pair::getFirst).collect(Collectors.toList());
2470  }

2497    if (!isTableDisabled(tableName)) {
