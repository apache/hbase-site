087   * Current Region State.
088   * In-memory only. Not persisted.
089   */
090  // Mutable/Immutable? Changes have to be synchronized or not?

098  public static class RegionStateNode implements Comparable<RegionStateNode> {
099    private final RegionInfo regionInfo;
100    private final ProcedureEvent<?> event;
101
102    private volatile RegionTransitionProcedure procedure = null;
103    private volatile ServerName regionLocation = null;
104    private volatile ServerName lastHost = null;
105    /**
106     * A Region-in-Transition (RIT) moves through states.
107     * See {@link State} for complete list. A Region that
108     * is opened moves from OFFLINE => OPENING => OPENED.
109     */
110    private volatile State state = State.OFFLINE;
111
112    /**
113     * Updated whenever a call to {@link #setRegionLocation(ServerName)}
114     * or {@link #setState(State, State...)}.
115     */
116    private volatile long lastUpdate = 0;
117
118    private volatile long openSeqNum = HConstants.NO_SEQNUM;
119
120    public RegionStateNode(final RegionInfo regionInfo) {
121      this.regionInfo = regionInfo;
122      this.event = new AssignmentProcedureEvent(regionInfo);
123    }
124
125    /**
126     * @param update new region state this node should be assigned.
127     * @param expected current state should be in this given list of expected states
128     * @return true, if current state is in expected list; otherwise false.
129     */
130    public boolean setState(final State update, final State... expected) {
131      if (!isInState(expected)) {
132        return false;
133      }
134      this.state = update;
135      this.lastUpdate = EnvironmentEdgeManager.currentTime();
136      return true;
137    }
138
139    /**
140     * Put region into OFFLINE mode (set state and clear location).
141     * @return Last recorded server deploy
142     */
143    public ServerName offline() {
144      setState(State.OFFLINE);
145      return setRegionLocation(null);
146    }
147
148    /**
149     * Set new {@link State} but only if currently in <code>expected</code> State
150     * (if not, throw {@link UnexpectedStateException}.
151     */
152    public void transitionState(final State update, final State... expected)
153    throws UnexpectedStateException {
154      if (!setState(update, expected)) {
155        throw new UnexpectedStateException("Expected " + Arrays.toString(expected) +
156          " so could move to " + update + " but current state=" + getState());
157      }
158    }
159
160    public boolean isInState(final State... expected) {
161      if (expected != null && expected.length > 0) {
162        boolean expectedState = false;
163        for (int i = 0; i < expected.length; ++i) {
164          expectedState |= (getState() == expected[i]);
165        }
166        return expectedState;
167      }
168      return true;
169    }
170
171    public boolean isStuck() {
172      return isInState(State.FAILED_OPEN) && getProcedure() != null;
173    }
174
175    public boolean isInTransition() {
176      return getProcedure() != null;
177    }
178
179    public long getLastUpdate() {
180      return procedure != null ? procedure.getLastUpdate() : lastUpdate;
181    }
182
183    public void setLastHost(final ServerName serverName) {
184      this.lastHost = serverName;
185    }
186
187    public void setOpenSeqNum(final long seqId) {
188      this.openSeqNum = seqId;
189    }
190
191    public ServerName setRegionLocation(final ServerName serverName) {
192      ServerName lastRegionLocation = this.regionLocation;
193      if (LOG.isTraceEnabled() && serverName == null) {
194        LOG.trace("Tracking when we are set to null " + this, new Throwable("TRACE"));
195      }
196      this.regionLocation = serverName;
197      this.lastUpdate = EnvironmentEdgeManager.currentTime();
198      return lastRegionLocation;
199    }
200
201    public boolean setProcedure(final RegionTransitionProcedure proc) {
202      if (this.procedure != null && this.procedure != proc) {
203        return false;
204      }
205      this.procedure = proc;
206      return true;
207    }
208
209    public boolean unsetProcedure(final RegionTransitionProcedure proc) {
210      if (this.procedure != null && this.procedure != proc) {
211        return false;
212      }
213      this.procedure = null;
214      return true;
215    }
216
217    public RegionTransitionProcedure getProcedure() {
218      return procedure;
219    }
220
221    public ProcedureEvent<?> getProcedureEvent() {
222      return event;
223    }
224
225    public RegionInfo getRegionInfo() {
226      return regionInfo;
227    }
228
229    public TableName getTable() {
230      return getRegionInfo().getTable();
231    }
232
233    public boolean isSystemTable() {
234      return getTable().isSystemTable();
235    }
236
237    public ServerName getLastHost() {
238      return lastHost;
239    }
240
241    public ServerName getRegionLocation() {
242      return regionLocation;
243    }
244
245    public State getState() {
246      return state;
247    }
248
249    public long getOpenSeqNum() {
250      return openSeqNum;
251    }
252
253    public int getFormatVersion() {
254      // we don't have any format for now
255      // it should probably be in regionInfo.getFormatVersion()
256      return 0;
257    }
258
259    public RegionState toRegionState() {
260      return new RegionState(getRegionInfo(), getState(), getLastUpdate(), getRegionLocation());
261    }
262
263    @Override
264    public int compareTo(final RegionStateNode other) {
265      // NOTE: RegionInfo sort by table first, so we are relying on that.
266      // we have a TestRegionState#testOrderedByTable() that check for that.
267      return RegionInfo.COMPARATOR.compare(getRegionInfo(), other.getRegionInfo());
268    }
269
270    @Override
271    public int hashCode() {
272      return getRegionInfo().hashCode();
273    }
274
275    @Override
276    public boolean equals(final Object other) {
277      if (this == other) return true;
278      if (!(other instanceof RegionStateNode)) return false;
279      return compareTo((RegionStateNode)other) == 0;
280    }
281
282    @Override
283    public String toString() {
284      return toDescriptiveString();
285    }
286
287    public String toShortString() {
288      // rit= is the current Region-In-Transition State -- see State enum.
289      return String.format("rit=%s, location=%s", getState(), getRegionLocation());
290    }
291
292    public String toDescriptiveString() {
293      return String.format("%s, table=%s, region=%s",
294        toShortString(), getTable(), getRegionInfo().getEncodedName());
295    }

298  // This comparator sorts the RegionStates by time stamp then Region name.
299  // Comparing by timestamp alone can lead us to discard different RegionStates that happen
300  // to share a timestamp.
301  private static class RegionStateStampComparator implements Comparator<RegionState> {
302    @Override
303    public int compare(final RegionState l, final RegionState r) {
304      int stampCmp = Long.compare(l.getStamp(), r.getStamp());
305      return stampCmp != 0 ? stampCmp : RegionInfo.COMPARATOR.compare(l.getRegion(), r.getRegion());
306    }

555  public List<RegionInfo> getRegionsOfTable(final TableName table, final boolean offline) {
556    final ArrayList<RegionStateNode> nodes = getTableRegionStateNodes(table);
557    final ArrayList<RegionInfo> hris = new ArrayList<RegionInfo>(nodes.size());
558    for (RegionStateNode node: nodes) {
559      if (include(node, offline)) hris.add(node.getRegionInfo());
560    }
561    return hris;
562  }

564  /**
565   * Utility. Whether to include region in list of regions. Default is to
566   * weed out split and offline regions.
567   * @return True if we should include the <code>node</code> (do not include
568   * if split or offline unless <code>offline</code> is set to true.
569   */
570  boolean include(final RegionStateNode node, final boolean offline) {
571    if (LOG.isTraceEnabled()) {
572      LOG.trace("WORKING ON " + node + " " + node.getRegionInfo());
573    }
574    if (node.isInState(State.SPLIT)) return false;
575    if (node.isInState(State.OFFLINE) && !offline) return false;
576    final RegionInfo hri = node.getRegionInfo();
577    return (!hri.isOffline() && !hri.isSplit()) ||
578        ((hri.isOffline() || hri.isSplit()) && offline);
579  }

661  public Map<ServerName, List<RegionInfo>> getSnapShotOfAssignment(
662      final Collection<RegionInfo> regions) {
663    final Map<ServerName, List<RegionInfo>> result = new HashMap<ServerName, List<RegionInfo>>();
664    for (RegionInfo hri: regions) {
665      final RegionStateNode node = getRegionStateNode(hri);
666      if (node == null) continue;
667
668      // TODO: State.OPEN
669      final ServerName serverName = node.getRegionLocation();
670      if (serverName == null) continue;
671
672      List<RegionInfo> serverRegions = result.get(serverName);
673      if (serverRegions == null) {
674        serverRegions = new ArrayList<RegionInfo>();
675        result.put(serverName, serverRegions);
676      }
677
678      serverRegions.add(node.getRegionInfo());
679    }
680    return result;
681  }

691  public Map<RegionState.State, List<RegionInfo>> getRegionByStateOfTable(TableName tableName) {
692    final State[] states = State.values();
693    final Map<RegionState.State, List<RegionInfo>> tableRegions =
694        new HashMap<State, List<RegionInfo>>(states.length);
695    for (int i = 0; i < states.length; ++i) {
696      tableRegions.put(states[i], new ArrayList<RegionInfo>());
697    }
698
699    for (RegionStateNode node: regionsMap.values()) {
700      if (node.getTable().equals(tableName)) {
701        tableRegions.get(node.getState()).add(node.getRegionInfo());
702      }
703    }
704    return tableRegions;

719   * This is an EXPENSIVE clone.  Cloning though is the safest thing to do.
720   * Can't let out original since it can change and at least the load balancer
721   * wants to iterate this exported list.  We need to synchronize on regions
722   * since all access to this.servers is under a lock on this.regions.
723   * @param forceByCluster a flag to force to aggregate the server-load to the cluster level
724   * @return A clone of current assignments by table.
725   */
726  public Map<TableName, Map<ServerName, List<RegionInfo>>> getAssignmentsByTable(
727      final boolean forceByCluster) {
728    if (!forceByCluster) return getAssignmentsByTable();
729
730    final HashMap<ServerName, List<RegionInfo>> ensemble =
731      new HashMap<ServerName, List<RegionInfo>>(serverMap.size());
732    for (ServerStateNode serverNode: serverMap.values()) {
733      ensemble.put(serverNode.getServerName(), serverNode.getRegionInfoList());
734    }
735
736    // TODO: can we use Collections.singletonMap(HConstants.ENSEMBLE_TABLE_NAME, ensemble)?
737    final Map<TableName, Map<ServerName, List<RegionInfo>>> result =
738      new HashMap<TableName, Map<ServerName, List<RegionInfo>>>(1);
739    result.put(HConstants.ENSEMBLE_TABLE_NAME, ensemble);
740    return result;
741  }
742
743  public Map<TableName, Map<ServerName, List<RegionInfo>>> getAssignmentsByTable() {
744    final Map<TableName, Map<ServerName, List<RegionInfo>>> result = new HashMap<>();
745    for (RegionStateNode node: regionsMap.values()) {
746      Map<ServerName, List<RegionInfo>> tableResult = result.get(node.getTable());
747      if (tableResult == null) {
748        tableResult = new HashMap<ServerName, List<RegionInfo>>();
749        result.put(node.getTable(), tableResult);
750      }
751
752      final ServerName serverName = node.getRegionLocation();
753      if (serverName == null) {
754        LOG.info("Skipping, no server for " + node);
755        continue;
756      }
757      List<RegionInfo> serverResult = tableResult.get(serverName);
758      if (serverResult == null) {
759        serverResult = new ArrayList<RegionInfo>();
760        tableResult.put(serverName, serverResult);
761      }
762
763      serverResult.add(node.getRegionInfo());
764    }
765    // Add online servers with no assignment for the table.
766    for (Map<ServerName, List<RegionInfo>> table: result.values()) {
767        for (ServerName svr : serverMap.keySet()) {
768          if (!table.containsKey(svr)) {
769            table.put(svr, new ArrayList<RegionInfo>());
770          }
771        }
772    }
773    return result;
774  }

899  public RegionFailedOpen addToFailedOpen(final RegionStateNode regionNode) {
900    final byte[] key = regionNode.getRegionInfo().getRegionName();
901    RegionFailedOpen node = regionFailedOpen.get(key);

903      RegionFailedOpen newNode = new RegionFailedOpen(regionNode);
904      RegionFailedOpen oldNode = regionFailedOpen.putIfAbsent(key, newNode);
905      node = oldNode != null ? oldNode : newNode;
906    }

949  public double getAverageLoad() {
950    int numServers = 0;
951    int totalLoad = 0;
952    for (ServerStateNode node: serverMap.values()) {

975  public static String regionNamesToString(final Collection<byte[]> regions) {
976    final StringBuilder sb = new StringBuilder();
977    final Iterator<byte[]> it = regions.iterator();
978    sb.append("[");
979    if (it.hasNext()) {
980      sb.append(Bytes.toStringBinary(it.next()));
981      while (it.hasNext()) {
982        sb.append(", ");
983        sb.append(Bytes.toStringBinary(it.next()));
984      }
985    }
986    sb.append("]");
987    return sb.toString();
988  }
