105/**
106 * HBase Canary Tool for "canary monitoring" of a running HBase cluster.
107 *
108 * There are three modes:
109 * <ol>
110 * <li>region mode (Default): For each region, try to get one row per column family outputting
111 * information on failure (ERROR) or else the latency.
112 * </li>
113 *
114 * <li>regionserver mode: For each regionserver try to get one row from one table selected
115 * randomly outputting information on failure (ERROR) or else the latency.
116 * </li>
117 *
118 * <li>zookeeper mode: for each zookeeper instance, selects a znode outputting information on
119 * failure (ERROR) or else the latency.
120 * </li>
121 * </ol>
122 */
123@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
124public class CanaryTool implements Tool, Canary {

356   * Run a single zookeeper Task and then exit.

358  static class ZookeeperTask implements Callable<Void> {
359    private final Connection connection;
360    private final String host;
361    private String znode;
362    private final int timeout;
363    private ZookeeperStdOutSink sink;
364
365    public ZookeeperTask(Connection connection, String host, String znode, int timeout,
366        ZookeeperStdOutSink sink) {
367      this.connection = connection;
368      this.host = host;
369      this.znode = znode;
370      this.timeout = timeout;
371      this.sink = sink;
372    }
373
374    @Override public Void call() throws Exception {
375      ZooKeeper zooKeeper = null;
376      try {
377        zooKeeper = new ZooKeeper(host, timeout, EmptyWatcher.instance);
378        Stat exists = zooKeeper.exists(znode, false);
379        StopWatch stopwatch = new StopWatch();
380        stopwatch.start();
381        zooKeeper.getData(znode, false, exists);
382        stopwatch.stop();
383        sink.publishReadTiming(znode, host, stopwatch.getTime());
384      } catch (KeeperException | InterruptedException e) {
385        sink.publishReadFailure(znode, host);
386      } finally {
387        if (zooKeeper != null) {
388          zooKeeper.close();
389        }
390      }
391      return null;
392    }
393  }

395  /**
396   * Run a single Region Task and then exit. For each column family of the Region, get one row and
397   * output latency or failure.
398   */
399  static class RegionTask implements Callable<Void> {
400    public enum TaskType{
401      READ, WRITE
402    }
403    private Connection connection;
404    private RegionInfo region;
405    private RegionStdOutSink sink;
406    private TaskType taskType;
407    private boolean rawScanEnabled;
408    private ServerName serverName;
409    private LongAdder readWriteLatency;
410
411    RegionTask(Connection connection, RegionInfo region, ServerName serverName,
412        RegionStdOutSink sink, TaskType taskType, boolean rawScanEnabled, LongAdder rwLatency) {
413      this.connection = connection;
414      this.region = region;
415      this.serverName = serverName;
416      this.sink = sink;
417      this.taskType = taskType;
418      this.rawScanEnabled = rawScanEnabled;
419      this.readWriteLatency = rwLatency;
420    }
421
422    @Override
423    public Void call() {
424      switch (taskType) {
425      case READ:
426        return read();
427      case WRITE:
428        return write();
429      default:
430        return read();
431      }
432    }
433
434    public Void read() {
435      Table table = null;
436      TableDescriptor tableDesc = null;
437      try {
438        LOG.debug("Reading table descriptor for table {}", region.getTable());
439        table = connection.getTable(region.getTable());
440        tableDesc = table.getDescriptor();
441      } catch (IOException e) {
442        LOG.debug("sniffRegion {} of {} failed", region.getEncodedName(), e);
443        sink.publishReadFailure(serverName, region, e);
444        if (table != null) {
445          try {
446            table.close();
447          } catch (IOException ioe) {
448            LOG.error("Close table failed", e);
449          }
450        }
451        return null;
452      }
453
454      byte[] startKey = null;
455      Get get = null;
456      Scan scan = null;
457      ResultScanner rs = null;
458      StopWatch stopWatch = new StopWatch();
459      for (ColumnFamilyDescriptor column : tableDesc.getColumnFamilies()) {
460        stopWatch.reset();
461        startKey = region.getStartKey();
462        // Can't do a get on empty start row so do a Scan of first element if any instead.
463        if (startKey.length > 0) {
464          get = new Get(startKey);
465          get.setCacheBlocks(false);
466          get.setFilter(new FirstKeyOnlyFilter());
467          get.addFamily(column.getName());
468        } else {
469          scan = new Scan();
470          LOG.debug("rawScan {} for {}", rawScanEnabled, tableDesc.getTableName());
471          scan.setRaw(rawScanEnabled);
472          scan.setCaching(1);
473          scan.setCacheBlocks(false);
474          scan.setFilter(new FirstKeyOnlyFilter());
475          scan.addFamily(column.getName());
476          scan.setMaxResultSize(1L);
477          scan.setOneRowLimit();
478        }
479        LOG.debug("Reading from {} {} {} {}", tableDesc.getTableName(),
480            region.getRegionNameAsString(), column.getNameAsString(),
481            Bytes.toStringBinary(startKey));
482        try {
483          stopWatch.start();
484          if (startKey.length > 0) {
485            table.get(get);
486          } else {
487            rs = table.getScanner(scan);
488            rs.next();
489          }
490          stopWatch.stop();
491          this.readWriteLatency.add(stopWatch.getTime());
492          sink.publishReadTiming(serverName, region, column, stopWatch.getTime());
493        } catch (Exception e) {
494          sink.publishReadFailure(serverName, region, column, e);
495          sink.updateReadFailures(region.getRegionNameAsString(), serverName.getHostname());
496        } finally {
497          if (rs != null) {
498            rs.close();
499          }
500          scan = null;
501          get = null;
502        }
503      }
504      try {
505        table.close();
506      } catch (IOException e) {
507        LOG.error("Close table failed", e);
508      }
509      return null;
510    }
511
512    /**
513     * Check writes for the canary table
514     */
515    private Void write() {
516      Table table = null;
517      TableDescriptor tableDesc = null;
518      try {
519        table = connection.getTable(region.getTable());
520        tableDesc = table.getDescriptor();
521        byte[] rowToCheck = region.getStartKey();
522        if (rowToCheck.length == 0) {
523          rowToCheck = new byte[]{0x0};
524        }
525        int writeValueSize =
526            connection.getConfiguration().getInt(HConstants.HBASE_CANARY_WRITE_VALUE_SIZE_KEY, 10);
527        for (ColumnFamilyDescriptor column : tableDesc.getColumnFamilies()) {
528          Put put = new Put(rowToCheck);
529          byte[] value = new byte[writeValueSize];
530          Bytes.random(value);
531          put.addColumn(column.getName(), HConstants.EMPTY_BYTE_ARRAY, value);
532
533          LOG.debug("Writing to {} {} {} {}",
534            tableDesc.getTableName(), region.getRegionNameAsString(), column.getNameAsString(),
535            Bytes.toStringBinary(rowToCheck));
536          try {
537            long startTime = System.currentTimeMillis();
538            table.put(put);
539            long time = System.currentTimeMillis() - startTime;
540            this.readWriteLatency.add(time);
541            sink.publishWriteTiming(serverName, region, column, time);
542          } catch (Exception e) {
543            sink.publishWriteFailure(serverName, region, column, e);
544          }
545        }
546        table.close();
547      } catch (IOException e) {
548        sink.publishWriteFailure(serverName, region, e);
549        sink.updateWriteFailures(region.getRegionNameAsString(), serverName.getHostname() );
550      }
551      return null;
552    }
553  }

555  /**
556   * Run a single RegionServer Task and then exit.
557   * Get one row from a region on the regionserver and output latency or the failure.
558   */
559  static class RegionServerTask implements Callable<Void> {
560    private Connection connection;
561    private String serverName;
562    private RegionInfo region;
563    private RegionServerStdOutSink sink;
564    private AtomicLong successes;
565
566    RegionServerTask(Connection connection, String serverName, RegionInfo region,
567        RegionServerStdOutSink sink, AtomicLong successes) {
568      this.connection = connection;
569      this.serverName = serverName;
570      this.region = region;
571      this.sink = sink;
572      this.successes = successes;
573    }
574
575    @Override
576    public Void call() {
577      TableName tableName = null;
578      Table table = null;
579      Get get = null;
580      byte[] startKey = null;
581      Scan scan = null;
582      StopWatch stopWatch = new StopWatch();
583      // monitor one region on every region server
584      stopWatch.reset();
585      try {
586        tableName = region.getTable();
587        table = connection.getTable(tableName);
588        startKey = region.getStartKey();
589        // Can't do a get on empty start row so do a Scan of first element if any instead.
590        LOG.debug("Reading from {} {} {} {}",
591          serverName, region.getTable(), region.getRegionNameAsString(),
592          Bytes.toStringBinary(startKey));
593        if (startKey.length > 0) {
594          get = new Get(startKey);
595          get.setCacheBlocks(false);
596          get.setFilter(new FirstKeyOnlyFilter());
597          stopWatch.start();
598          table.get(get);
599          stopWatch.stop();
600        } else {
601          scan = new Scan();
602          scan.setCacheBlocks(false);
603          scan.setFilter(new FirstKeyOnlyFilter());
604          scan.setCaching(1);
605          scan.setMaxResultSize(1L);
606          scan.setOneRowLimit();
607          stopWatch.start();
608          ResultScanner s = table.getScanner(scan);
609          s.next();
610          s.close();
611          stopWatch.stop();
612        }
613        successes.incrementAndGet();
614        sink.publishReadTiming(tableName.getNameAsString(), serverName, stopWatch.getTime());
615      } catch (TableNotFoundException tnfe) {
616        LOG.error("Table may be deleted", tnfe);
617        // This is ignored because it doesn't imply that the regionserver is dead
618      } catch (TableNotEnabledException tnee) {
619        // This is considered a success since we got a response.
620        successes.incrementAndGet();
621        LOG.debug("The targeted table was disabled.  Assuming success.");
622      } catch (DoNotRetryIOException dnrioe) {
623        sink.publishReadFailure(tableName.getNameAsString(), serverName);
624        LOG.error(dnrioe.toString(), dnrioe);
625      } catch (IOException e) {
626        sink.publishReadFailure(tableName.getNameAsString(), serverName);
627        LOG.error(e.toString(), e);
628      } finally {
629        if (table != null) {
630          try {
631            table.close();
632          } catch (IOException e) {/* DO NOTHING */
633            LOG.error("Close table failed", e);
634          }
635        }
636        scan = null;
637        get = null;
638        startKey = null;
639      }
640      return null;
641    }
642  }

741  private int parseArgs(String[] args) {
742    int index = -1;
743    long permittedFailures = 0;
744    boolean regionServerAllRegions = false, writeSniffing = false;
745    String readTableTimeoutsStr = null;
746
747    // Process command line args
748    for (int i = 0; i < args.length; i++) {
749      String cmd = args[i];
750
751      if (cmd.startsWith("-")) {
752        if (index >= 0) {
753          // command line args must be in the form: [opts] [table 1 [table 2 ...]]
754          System.err.println("Invalid command line options");
755          printUsageAndExit();
756        }
757
758        if (cmd.equals("-help") || cmd.equals("-h")) {
759          // user asked for help, print the help and quit.
760          printUsageAndExit();
761        } else if (cmd.equals("-daemon") && interval == 0) {
762          // user asked for daemon mode, set a default interval between checks
763          interval = DEFAULT_INTERVAL;
764        } else if (cmd.equals("-interval")) {
765          // user has specified an interval for canary breaths (-interval N)
766          i++;
767
768          if (i == args.length) {
769            System.err.println("-interval takes a numeric seconds value argument.");
770            printUsageAndExit();
771          }
772
773          try {
774            interval = Long.parseLong(args[i]) * 1000;
775          } catch (NumberFormatException e) {
776            System.err.println("-interval needs a numeric value argument.");
777            printUsageAndExit();
778          }
779        } else if (cmd.equals("-zookeeper")) {
780          this.zookeeperMode = true;
781        } else if(cmd.equals("-regionserver")) {
782          this.regionServerMode = true;
783        } else if(cmd.equals("-allRegions")) {
784          conf.setBoolean(HBASE_CANARY_REGIONSERVER_ALL_REGIONS, true);
785          regionServerAllRegions = true;
786        } else if(cmd.equals("-writeSniffing")) {
787          writeSniffing = true;
788          conf.setBoolean(HBASE_CANARY_REGION_WRITE_SNIFFING, true);
789        } else if(cmd.equals("-treatFailureAsError") || cmd.equals("-failureAsError")) {
790          conf.setBoolean(HBASE_CANARY_FAIL_ON_ERROR, true);
791        } else if (cmd.equals("-e")) {
792          conf.setBoolean(HBASE_CANARY_USE_REGEX, true);
793        } else if (cmd.equals("-t")) {
794          i++;
795
796          if (i == args.length) {
797            System.err.println("-t takes a numeric milliseconds value argument.");
798            printUsageAndExit();
799          }
800          long timeout = 0;
801          try {
802            timeout = Long.parseLong(args[i]);
803          } catch (NumberFormatException e) {
804            System.err.println("-t takes a numeric milliseconds value argument.");
805            printUsageAndExit();
806          }
807          conf.setLong(HBASE_CANARY_TIMEOUT, timeout);
808        } else if(cmd.equals("-writeTableTimeout")) {
809          i++;
810
811          if (i == args.length) {
812            System.err.println("-writeTableTimeout takes a numeric milliseconds value argument.");
813            printUsageAndExit();
814          }
815          long configuredWriteTableTimeout = 0;
816          try {
817            configuredWriteTableTimeout = Long.parseLong(args[i]);
818          } catch (NumberFormatException e) {
819            System.err.println("-writeTableTimeout takes a numeric milliseconds value argument.");
820            printUsageAndExit();
821          }
822          conf.setLong(HBASE_CANARY_REGION_WRITE_TABLE_TIMEOUT, configuredWriteTableTimeout);
823        } else if (cmd.equals("-writeTable")) {
824          i++;
825
826          if (i == args.length) {
827            System.err.println("-writeTable takes a string tablename value argument.");
828            printUsageAndExit();
829          }
830          conf.set(HBASE_CANARY_REGION_WRITE_TABLE_NAME, args[i]);
831        } else if (cmd.equals("-f")) {
832          i++;
833
834          if (i == args.length) {
835            System.err
836                .println("-f needs a boolean value argument (true|false).");
837            printUsageAndExit();
838          }
839
840          conf.setBoolean(HBASE_CANARY_FAIL_ON_ERROR, Boolean.parseBoolean(args[i]));
841        } else if (cmd.equals("-readTableTimeouts")) {
842          i++;
843
844          if (i == args.length) {
845            System.err.println("-readTableTimeouts needs a comma-separated list of read " +
846                "millisecond timeouts per table (without spaces).");
847            printUsageAndExit();
848          }
849          readTableTimeoutsStr = args[i];
850          conf.set(HBASE_CANARY_REGION_READ_TABLE_TIMEOUT, readTableTimeoutsStr);
851        } else if (cmd.equals("-permittedZookeeperFailures")) {
852          i++;
853
854          if (i == args.length) {
855            System.err.println("-permittedZookeeperFailures needs a numeric value argument.");
856            printUsageAndExit();
857          }
858          try {
859            permittedFailures = Long.parseLong(args[i]);
860          } catch (NumberFormatException e) {
861            System.err.println("-permittedZookeeperFailures needs a numeric value argument.");
862            printUsageAndExit();
863          }
864          conf.setLong(HBASE_CANARY_ZOOKEEPER_PERMITTED_FAILURES, permittedFailures);
865        } else {
866          // no options match
867          System.err.println(cmd + " options is invalid.");
868          printUsageAndExit();
869        }
870      } else if (index < 0) {
871        // keep track of first table name specified by the user
872        index = i;
873      }
874    }
875    if (regionServerAllRegions && !this.regionServerMode) {
876      System.err.println("-allRegions can only be specified in regionserver mode.");
877      printUsageAndExit();
878    }
879    if (this.zookeeperMode) {
880      if (this.regionServerMode || regionServerAllRegions || writeSniffing) {
881        System.err.println("-zookeeper is exclusive and cannot be combined with "
882            + "other modes.");
883        printUsageAndExit();
884      }
885    }
886    if (permittedFailures != 0 && !this.zookeeperMode) {
887      System.err.println("-permittedZookeeperFailures requires -zookeeper mode.");
888      printUsageAndExit();
889    }
890    if (readTableTimeoutsStr != null && (this.regionServerMode || this.zookeeperMode)) {
891      System.err.println("-readTableTimeouts can only be configured in region mode.");
892      printUsageAndExit();
893    }
894    return index;
895  }

917  private int runMonitor(String[] monitorTargets) throws Exception {
918    ChoreService choreService = null;
919
920    // Launches chore for refreshing kerberos credentials if security is enabled.
921    // Please see http://hbase.apache.org/book.html#_running_canary_in_a_kerberos_enabled_cluster
922    // for more details.
923    final ScheduledChore authChore = AuthUtil.getAuthChore(conf);
924    if (authChore != null) {
925      choreService = new ChoreService("CANARY_TOOL");
926      choreService.scheduleChore(authChore);
927    }
928
929    // Start to prepare the stuffs
930    Monitor monitor = null;
931    Thread monitorThread;
932    long startTime = 0;
933    long currentTimeLength = 0;
934    boolean failOnError = conf.getBoolean(HBASE_CANARY_FAIL_ON_ERROR, true);
935    long timeout = conf.getLong(HBASE_CANARY_TIMEOUT, DEFAULT_TIMEOUT);
936    // Get a connection to use in below.
937    try (Connection connection = ConnectionFactory.createConnection(this.conf)) {
938      do {
939        // Do monitor !!
940        try {
941          monitor = this.newMonitor(connection, monitorTargets);
942          monitorThread = new Thread(monitor, "CanaryMonitor-" + System.currentTimeMillis());
943          startTime = System.currentTimeMillis();
944          monitorThread.start();
945          while (!monitor.isDone()) {
946            // wait for 1 sec
947            Thread.sleep(1000);
948            // exit if any error occurs
949            if (failOnError && monitor.hasError()) {
950              monitorThread.interrupt();
951              if (monitor.initialized) {
952                return monitor.errorCode;
953              } else {
954                return INIT_ERROR_EXIT_CODE;
955              }
956            }
957            currentTimeLength = System.currentTimeMillis() - startTime;
958            if (currentTimeLength > timeout) {
959              LOG.error("The monitor is running too long (" + currentTimeLength
960                  + ") after timeout limit:" + timeout
961                  + " will be killed itself !!");
962              if (monitor.initialized) {
963                return TIMEOUT_ERROR_EXIT_CODE;
964              } else {
965                return INIT_ERROR_EXIT_CODE;
966              }
967            }
968          }
969
970          if (failOnError && monitor.finalCheckForErrors()) {
971            monitorThread.interrupt();
972            return monitor.errorCode;
973          }
974        } finally {
975          if (monitor != null) monitor.close();
976        }
977
978        Thread.sleep(interval);
979      } while (interval > 0);
980    } // try-with-resources close
981
982    if (choreService != null) {
983      choreService.shutdown();
984    }
985    return monitor.errorCode;
986  }

1328          conf.getFloat(HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_LOWERLIMIT_KEY, 1.0f);
1329      this.regionsUpperLimit =
1330          conf.getFloat(HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_UPPERLIMIT_KEY, 1.5f);
1331      this.checkPeriod =
1332          conf.getInt(HConstants.HBASE_CANARY_WRITE_TABLE_CHECK_PERIOD_KEY,
1333            DEFAULT_WRITE_TABLE_CHECK_PERIOD);
1334      this.rawScanEnabled = conf.getBoolean(HConstants.HBASE_CANARY_READ_RAW_SCAN_KEY, false);
1335      this.configuredReadTableTimeouts = new HashMap<>(configuredReadTableTimeouts);
1336      this.configuredWriteTableTimeout = configuredWriteTableTimeout;

1346    @Override
1347    public void run() {
1348      if (this.initAdmin()) {
1349        try {
1350          List<Future<Void>> taskFutures = new LinkedList<>();
1351          RegionStdOutSink regionSink = this.getSink();
1352          if (this.targets != null && this.targets.length > 0) {
1353            String[] tables = generateMonitorTables(this.targets);
1354            // Check to see that each table name passed in the -readTableTimeouts argument is also
1355            // passed as a monitor target.
1356            if (!new HashSet<>(Arrays.asList(tables)).
1357                containsAll(this.configuredReadTableTimeouts.keySet())) {
1358              LOG.error("-readTableTimeouts can only specify read timeouts for monitor targets " +
1359                  "passed via command line.");
1360              this.errorCode = USAGE_EXIT_CODE;
1361              return;
1362            }
1363            this.initialized = true;
1364            for (String table : tables) {
1365              LongAdder readLatency = regionSink.initializeAndGetReadLatencyForTable(table);
1366              taskFutures.addAll(CanaryTool.sniff(admin, regionSink, table, executor, TaskType.READ,
1367                this.rawScanEnabled, readLatency));
1368            }
1369          } else {
1370            taskFutures.addAll(sniff(TaskType.READ, regionSink));
1371          }
1372
1373          if (writeSniffing) {
1374            if (EnvironmentEdgeManager.currentTime() - lastCheckTime > checkPeriod) {
1375              try {
1376                checkWriteTableDistribution();
1377              } catch (IOException e) {
1378                LOG.error("Check canary table distribution failed!", e);
1379              }
1380              lastCheckTime = EnvironmentEdgeManager.currentTime();
1381            }
1382            // sniff canary table with write operation
1383            regionSink.initializeWriteLatency();
1384            LongAdder writeTableLatency = regionSink.getWriteLatency();
1385            taskFutures.addAll(CanaryTool.sniff(admin, regionSink, admin.getDescriptor(writeTableName),
1386              executor, TaskType.WRITE, this.rawScanEnabled, writeTableLatency));
1387          }
1388
1389          for (Future<Void> future : taskFutures) {
1390            try {
1391              future.get();
1392            } catch (ExecutionException e) {
1393              LOG.error("Sniff region failed!", e);
1394            }
1395          }
1396          Map<String, LongAdder> actualReadTableLatency = regionSink.getReadLatencyMap();
1397          for (Map.Entry<String, Long> entry : configuredReadTableTimeouts.entrySet()) {
1398            String tableName = entry.getKey();
1399            if (actualReadTableLatency.containsKey(tableName)) {
1400              Long actual = actualReadTableLatency.get(tableName).longValue();
1401              Long configured = entry.getValue();
1402              if (actual > configured) {
1403                LOG.error("Read operation for {} took {}ms exceeded the configured read timeout." +
1404                    "(Configured read timeout {}ms.", tableName, actual, configured);
1405              } else {
1406                LOG.info("Read operation for {} took {}ms (Configured read timeout {}ms.",
1407                    tableName, actual, configured);
1408              }
1409            } else {
1410              LOG.error("Read operation for {} failed!", tableName);
1411            }
1412          }
1413          if (this.writeSniffing) {
1414            String writeTableStringName = this.writeTableName.getNameAsString();
1415            long actualWriteLatency = regionSink.getWriteLatency().longValue();
1416            LOG.info("Write operation for {} took {}ms. Configured write timeout {}ms.",
1417                writeTableStringName, actualWriteLatency, this.configuredWriteTableTimeout);
1418            // Check that the writeTable write operation latency does not exceed the configured timeout.
1419            if (actualWriteLatency > this.configuredWriteTableTimeout) {
1420              LOG.error("Write operation for {} exceeded the configured write timeout.",
1421                  writeTableStringName);
1422            }
1423          }
1424        } catch (Exception e) {
1425          LOG.error("Run regionMonitor failed", e);
1426          this.errorCode = ERROR_EXIT_CODE;
1427        } finally {
1428          this.done = true;
1429        }
1430      }
1431      this.done = true;
1432    }

1435     * @return List of tables to use in test.
1436     */
1437    private String[] generateMonitorTables(String[] monitorTargets) throws IOException {
1438      String[] returnTables = null;
1439
1440      if (this.useRegExp) {
1441        Pattern pattern = null;
1442        List<TableDescriptor> tds = null;
1443        Set<String> tmpTables = new TreeSet<>();
1444        try {
1445          LOG.debug(String.format("reading list of tables"));
1446          tds = this.admin.listTableDescriptors(pattern);
1447          if (tds == null) {
1448            tds = Collections.emptyList();
1449          }
1450          for (String monitorTarget : monitorTargets) {
1451            pattern = Pattern.compile(monitorTarget);
1452            for (TableDescriptor td : tds) {
1453              if (pattern.matcher(td.getTableName().getNameAsString()).matches()) {
1454                tmpTables.add(td.getTableName().getNameAsString());
1455              }
1456            }
1457          }
1458        } catch (IOException e) {
1459          LOG.error("Communicate with admin failed", e);
1460          throw e;
1461        }
1462
1463        if (tmpTables.size() > 0) {
1464          returnTables = tmpTables.toArray(new String[tmpTables.size()]);
1465        } else {
1466          String msg = "No HTable found, tablePattern:" + Arrays.toString(monitorTargets);
1467          LOG.error(msg);
1468          this.errorCode = INIT_ERROR_EXIT_CODE;
1469          throw new TableNotFoundException(msg);
1470        }
1471      } else {
1472        returnTables = monitorTargets;
1473      }
1474
1475      return returnTables;
1476    }

1479     * Canary entry point to monitor all the tables.
1480     */
1481    private List<Future<Void>> sniff(TaskType taskType, RegionStdOutSink regionSink)
1482        throws Exception {
1483      LOG.debug("Reading list of tables");
1484      List<Future<Void>> taskFutures = new LinkedList<>();
1485      for (TableDescriptor td: admin.listTableDescriptors()) {
1486        if (admin.tableExists(td.getTableName()) && admin.isTableEnabled(td.getTableName()) &&
1487            (!td.getTableName().equals(writeTableName))) {
1488          LongAdder readLatency =
1489              regionSink.initializeAndGetReadLatencyForTable(td.getTableName().getNameAsString());
1490          taskFutures.addAll(CanaryTool.sniff(admin, sink, td, executor, taskType, this.rawScanEnabled,
1491              readLatency));
1492        }
1493      }
1494      return taskFutures;
1495    }

1497    private void checkWriteTableDistribution() throws IOException {
1498      if (!admin.tableExists(writeTableName)) {
1499        int numberOfServers = admin.getRegionServers().size();
1500        if (numberOfServers == 0) {
1501          throw new IllegalStateException("No live regionservers");
1502        }
1503        createWriteTable(numberOfServers);
1504      }
1505
1506      if (!admin.isTableEnabled(writeTableName)) {
1507        admin.enableTable(writeTableName);
1508      }
1509
1510      ClusterMetrics status =
1511          admin.getClusterMetrics(EnumSet.of(Option.SERVERS_NAME, Option.MASTER));
1512      int numberOfServers = status.getServersName().size();
1513      if (status.getServersName().contains(status.getMasterName())) {
1514        numberOfServers -= 1;
1515      }
1516
1517      List<Pair<RegionInfo, ServerName>> pairs =
1518          MetaTableAccessor.getTableRegionsAndLocations(connection, writeTableName);
1519      int numberOfRegions = pairs.size();
1520      if (numberOfRegions < numberOfServers * regionsLowerLimit
1521          || numberOfRegions > numberOfServers * regionsUpperLimit) {
1522        admin.disableTable(writeTableName);
1523        admin.deleteTable(writeTableName);
1524        createWriteTable(numberOfServers);
1525      }
1526      HashSet<ServerName> serverSet = new HashSet<>();
1527      for (Pair<RegionInfo, ServerName> pair : pairs) {
1528        serverSet.add(pair.getSecond());
1529      }
1530      int numberOfCoveredServers = serverSet.size();
1531      if (numberOfCoveredServers < numberOfServers) {
1532        admin.balance();
1533      }
1534    }
1535
1536    private void createWriteTable(int numberOfServers) throws IOException {
1537      int numberOfRegions = (int)(numberOfServers * regionsLowerLimit);
1538      LOG.info("Number of live regionservers {}, pre-splitting the canary table into {} regions " +
1539        "(current lower limit of regions per server is {} and you can change it with config {}).",
1540          numberOfServers, numberOfRegions, regionsLowerLimit,
1541          HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_LOWERLIMIT_KEY);
1542      HTableDescriptor desc = new HTableDescriptor(writeTableName);
1543      HColumnDescriptor family = new HColumnDescriptor(CANARY_TABLE_FAMILY_NAME);
1544      family.setMaxVersions(1);
1545      family.setTimeToLive(writeDataTTL);
1546
1547      desc.addFamily(family);
1548      byte[][] splits = new RegionSplitter.HexStringSplit().split(numberOfRegions);
1549      admin.createTable(desc, splits);
1550    }

1670    public RegionServerMonitor(Connection connection, String[] monitorTargets, boolean useRegExp,

1686    public void run() {

1690          regionServerSink = this.getSink();

1702    private boolean checkNoTableNames() {

1705      LOG.debug("Reading list of tables");
1706      try {
1707        tableNames = this.admin.listTableNames();

1732    private void monitorRegionServers(Map<String, List<RegionInfo>> rsAndRMap, RegionServerStdOutSink regionServerSink) {
1733      List<RegionServerTask> tasks = new ArrayList<>();
1734      Map<String, AtomicLong> successMap = new HashMap<>();
1735      Random rand = new Random();
1736      for (Map.Entry<String, List<RegionInfo>> entry : rsAndRMap.entrySet()) {
1737        String serverName = entry.getKey();
1738        AtomicLong successes = new AtomicLong(0);
1739        successMap.put(serverName, successes);
1740        if (entry.getValue().isEmpty()) {
1741          LOG.error("Regionserver not serving any regions - {}", serverName);
1742        } else if (this.allRegions) {
1743          for (RegionInfo region : entry.getValue()) {
1744            tasks.add(new RegionServerTask(this.connection,
1745                serverName,
1746                region,
1747                regionServerSink,
1748                successes));
1749          }
1750        } else {
1751          // random select a region if flag not set
1752          RegionInfo region = entry.getValue().get(rand.nextInt(entry.getValue().size()));
1753          tasks.add(new RegionServerTask(this.connection,
1754              serverName,
1755              region,
1756              regionServerSink,
1757              successes));
1758        }
1759      }
1760      try {
1761        for (Future<Void> future : this.executor.invokeAll(tasks)) {
1762          try {
1763            future.get();
1764          } catch (ExecutionException e) {
1765            LOG.error("Sniff regionserver failed!", e);
1766            this.errorCode = ERROR_EXIT_CODE;
1767          }
1768        }
1769        if (this.allRegions) {
1770          for (Map.Entry<String, List<RegionInfo>> entry : rsAndRMap.entrySet()) {
1771            String serverName = entry.getKey();
1772            LOG.info("Successfully read {} regions out of {} on regionserver {}",
1773                successMap.get(serverName), entry.getValue().size(), serverName);
1774          }
1775        }
1776      } catch (InterruptedException e) {
1777        this.errorCode = ERROR_EXIT_CODE;
1778        LOG.error("Sniff regionserver interrupted!", e);
1779      }
1780    }

1788    private Map<String, List<RegionInfo>> getAllRegionServerByName() {
1789      Map<String, List<RegionInfo>> rsAndRMap = new HashMap<>();
1790      try {
1791        LOG.debug("Reading list of tables and locations");
1792        List<TableDescriptor> tableDescs = this.admin.listTableDescriptors();
1793        List<RegionInfo> regions = null;
1794        for (TableDescriptor tableDesc: tableDescs) {
1795          try (RegionLocator regionLocator =
1796                   this.admin.getConnection().getRegionLocator(tableDesc.getTableName())) {
1797            for (HRegionLocation location : regionLocator.getAllRegionLocations()) {
1798              ServerName rs = location.getServerName();
1799              String rsName = rs.getHostname();
1800              RegionInfo r = location.getRegion();
1801              if (rsAndRMap.containsKey(rsName)) {
1802                regions = rsAndRMap.get(rsName);
1803              } else {
1804                regions = new ArrayList<>();
1805                rsAndRMap.put(rsName, regions);
1806              }
1807              regions.add(r);
1808            }
1809          }
1810        }
1811
1812        // get any live regionservers not serving any regions
1813        for (ServerName rs: this.admin.getRegionServers()) {
1814          String rsName = rs.getHostname();
1815          if (!rsAndRMap.containsKey(rsName)) {
1816            rsAndRMap.put(rsName, Collections.<RegionInfo> emptyList());
1817          }
1818        }
1819      } catch (IOException e) {
1820        LOG.error("Get HTables info failed", e);
1821        this.errorCode = INIT_ERROR_EXIT_CODE;
1822      }
1823      return rsAndRMap;
1824    }

1826    private Map<String, List<RegionInfo>> doFilterRegionServerByName(
1827        Map<String, List<RegionInfo>> fullRsAndRMap) {
1828
1829      Map<String, List<RegionInfo>> filteredRsAndRMap = null;
1830
1831      if (this.targets != null && this.targets.length > 0) {
1832        filteredRsAndRMap = new HashMap<>();
1833        Pattern pattern = null;
1834        Matcher matcher = null;
1835        boolean regExpFound = false;
1836        for (String rsName : this.targets) {
1837          if (this.useRegExp) {
1838            regExpFound = false;
1839            pattern = Pattern.compile(rsName);
1840            for (Map.Entry<String, List<RegionInfo>> entry : fullRsAndRMap.entrySet()) {
1841              matcher = pattern.matcher(entry.getKey());
1842              if (matcher.matches()) {
1843                filteredRsAndRMap.put(entry.getKey(), entry.getValue());
1844                regExpFound = true;
1845              }
1846            }
1847            if (!regExpFound) {
1848              LOG.info("No RegionServerInfo found, regionServerPattern {}", rsName);
1849            }
1850          } else {
1851            if (fullRsAndRMap.containsKey(rsName)) {
1852              filteredRsAndRMap.put(rsName, fullRsAndRMap.get(rsName));
1853            } else {
1854              LOG.info("No RegionServerInfo found, regionServerName {}", rsName);
1855            }
1856          }
1857        }
1858      } else {
1859        filteredRsAndRMap = fullRsAndRMap;
1860      }
1861      return filteredRsAndRMap;
1862    }
