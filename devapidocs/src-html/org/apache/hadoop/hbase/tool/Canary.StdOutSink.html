083/**
084 * HBase Canary Tool, that that can be used to do
085 * "canary monitoring" of a running HBase cluster.
086 *
087 * Here are two modes
088 * 1. region mode - Foreach region tries to get one row per column family
089 * and outputs some information about failure or latency.
090 *
091 * 2. regionserver mode - Foreach regionserver tries to get one row from one table
092 * selected randomly and outputs some information about failure or latency.
093 */
094public final class Canary implements Tool {
095  // Sink interface used by the canary to outputs information
096  public interface Sink {
097    public long getReadFailureCount();
098    public long incReadFailureCount();
099    public void publishReadFailure(HRegionInfo region, Exception e);
100    public void publishReadFailure(HRegionInfo region, HColumnDescriptor column, Exception e);
101    public void publishReadTiming(HRegionInfo region, HColumnDescriptor column, long msTime);
102    public long getWriteFailureCount();
103    public void publishWriteFailure(HRegionInfo region, Exception e);
104    public void publishWriteFailure(HRegionInfo region, HColumnDescriptor column, Exception e);
105    public void publishWriteTiming(HRegionInfo region, HColumnDescriptor column, long msTime);
106  }
107  // new extended sink for output regionserver mode info
108  // do not change the Sink interface directly due to maintaining the API
109  public interface ExtendedSink extends Sink {
110    public void publishReadFailure(String table, String server);
111    public void publishReadTiming(String table, String server, long msTime);
112  }

189  /**
190   * For each column family of the region tries to get one row and outputs the latency, or the
191   * failure.
192   */
193  static class RegionTask implements Callable<Void> {
194    public enum TaskType{
195      READ, WRITE
196    }
197    private Connection connection;
198    private HRegionInfo region;
199    private Sink sink;
200    private TaskType taskType;
201
202    RegionTask(Connection connection, HRegionInfo region, Sink sink, TaskType taskType) {
203      this.connection = connection;
204      this.region = region;
205      this.sink = sink;
206      this.taskType = taskType;
207    }
208
209    @Override
210    public Void call() {
211      switch (taskType) {
212      case READ:
213        return read();
214      case WRITE:
215        return write();
216      default:
217        return read();
218      }
219    }
220
221    public Void read() {
222      Table table = null;
223      HTableDescriptor tableDesc = null;
224      try {
225        if (LOG.isDebugEnabled()) {
226          LOG.debug(String.format("reading table descriptor for table %s",
227            region.getTable()));
228        }
229        table = connection.getTable(region.getTable());
230        tableDesc = table.getTableDescriptor();
231      } catch (IOException e) {
232        LOG.debug("sniffRegion failed", e);
233        sink.publishReadFailure(region, e);
234        if (table != null) {
235          try {
236            table.close();
237          } catch (IOException ioe) {
238            LOG.error("Close table failed", e);
239          }
240        }
241        return null;
242      }
243
244      byte[] startKey = null;
245      Get get = null;
246      Scan scan = null;
247      ResultScanner rs = null;
248      StopWatch stopWatch = new StopWatch();
249      for (HColumnDescriptor column : tableDesc.getColumnFamilies()) {
250        stopWatch.reset();
251        startKey = region.getStartKey();
252        // Can't do a get on empty start row so do a Scan of first element if any instead.
253        if (startKey.length > 0) {
254          get = new Get(startKey);
255          get.setCacheBlocks(false);
256          get.setFilter(new FirstKeyOnlyFilter());
257          get.addFamily(column.getName());
258        } else {
259          scan = new Scan();
260          scan.setRaw(true);
261          scan.setCaching(1);
262          scan.setCacheBlocks(false);
263          scan.setFilter(new FirstKeyOnlyFilter());
264          scan.addFamily(column.getName());
265          scan.setMaxResultSize(1L);
266          scan.setSmall(true);
267        }
268
269        if (LOG.isDebugEnabled()) {
270          LOG.debug(String.format("reading from table %s region %s column family %s and key %s",
271            tableDesc.getTableName(), region.getRegionNameAsString(), column.getNameAsString(),
272            Bytes.toStringBinary(startKey)));
273        }
274        try {
275          stopWatch.start();
276          if (startKey.length > 0) {
277            table.get(get);
278          } else {
279            rs = table.getScanner(scan);
280            rs.next();
281          }
282          stopWatch.stop();
283          sink.publishReadTiming(region, column, stopWatch.getTime());
284        } catch (Exception e) {
285          sink.publishReadFailure(region, column, e);
286        } finally {
287          if (rs != null) {
288            rs.close();
289          }
290          scan = null;
291          get = null;
292          startKey = null;
293        }
294      }
295      try {
296        table.close();
297      } catch (IOException e) {
298        LOG.error("Close table failed", e);
299      }
300      return null;
301    }
302
303    /**
304     * Check writes for the canary table
305     * @return
306     */
307    private Void write() {
308      Table table = null;
309      HTableDescriptor tableDesc = null;
310      try {
311        table = connection.getTable(region.getTable());
312        tableDesc = table.getTableDescriptor();
313        byte[] rowToCheck = region.getStartKey();
314        if (rowToCheck.length == 0) {
315          rowToCheck = new byte[]{0x0};
316        }
317        int writeValueSize =
318            connection.getConfiguration().getInt(HConstants.HBASE_CANARY_WRITE_VALUE_SIZE_KEY, 10);
319        for (HColumnDescriptor column : tableDesc.getColumnFamilies()) {
320          Put put = new Put(rowToCheck);
321          byte[] value = new byte[writeValueSize];
322          Bytes.random(value);
323          put.addColumn(column.getName(), HConstants.EMPTY_BYTE_ARRAY, value);
324
325          if (LOG.isDebugEnabled()) {
326            LOG.debug(String.format("writing to table %s region %s column family %s and key %s",
327              tableDesc.getTableName(), region.getRegionNameAsString(), column.getNameAsString(),
328              Bytes.toStringBinary(rowToCheck)));
329          }
330          try {
331            long startTime = System.currentTimeMillis();
332            table.put(put);
333            long time = System.currentTimeMillis() - startTime;
334            sink.publishWriteTiming(region, column, time);
335          } catch (Exception e) {
336            sink.publishWriteFailure(region, column, e);
337          }
338        }
339        table.close();
340      } catch (IOException e) {
341        sink.publishWriteFailure(region, e);
342      }
343      return null;
344    }
345  }

347  /**
348   * Get one row from a region on the regionserver and outputs the latency, or the failure.
349   */
350  static class RegionServerTask implements Callable<Void> {
351    private Connection connection;
352    private String serverName;
353    private HRegionInfo region;
354    private ExtendedSink sink;
355    private AtomicLong successes;
356
357    RegionServerTask(Connection connection, String serverName, HRegionInfo region,
358        ExtendedSink sink, AtomicLong successes) {
359      this.connection = connection;
360      this.serverName = serverName;
361      this.region = region;
362      this.sink = sink;
363      this.successes = successes;
364    }
365
366    @Override
367    public Void call() {
368      TableName tableName = null;
369      Table table = null;
370      Get get = null;
371      byte[] startKey = null;
372      Scan scan = null;
373      StopWatch stopWatch = new StopWatch();
374      // monitor one region on every region server
375      stopWatch.reset();
376      try {
377        tableName = region.getTable();
378        table = connection.getTable(tableName);
379        startKey = region.getStartKey();
380        // Can't do a get on empty start row so do a Scan of first element if any instead.
381        if (LOG.isDebugEnabled()) {
382          LOG.debug(String.format("reading from region server %s table %s region %s and key %s",
383            serverName, region.getTable(), region.getRegionNameAsString(),
384            Bytes.toStringBinary(startKey)));
385        }
386        if (startKey.length > 0) {
387          get = new Get(startKey);
388          get.setCacheBlocks(false);
389          get.setFilter(new FirstKeyOnlyFilter());
390          stopWatch.start();
391          table.get(get);
392          stopWatch.stop();
393        } else {
394          scan = new Scan();
395          scan.setCacheBlocks(false);
396          scan.setFilter(new FirstKeyOnlyFilter());
397          scan.setCaching(1);
398          scan.setMaxResultSize(1L);
399          scan.setSmall(true);
400          stopWatch.start();
401          ResultScanner s = table.getScanner(scan);
402          s.next();
403          s.close();
404          stopWatch.stop();
405        }
406        successes.incrementAndGet();
407        sink.publishReadTiming(tableName.getNameAsString(), serverName, stopWatch.getTime());
408      } catch (TableNotFoundException tnfe) {
409        LOG.error("Table may be deleted", tnfe);
410        // This is ignored because it doesn't imply that the regionserver is dead
411      } catch (TableNotEnabledException tnee) {
412        // This is considered a success since we got a response.
413        successes.incrementAndGet();
414        LOG.debug("The targeted table was disabled.  Assuming success.");
415      } catch (DoNotRetryIOException dnrioe) {
416        sink.publishReadFailure(tableName.getNameAsString(), serverName);
417        LOG.error(dnrioe);
418      } catch (IOException e) {
419        sink.publishReadFailure(tableName.getNameAsString(), serverName);
420        LOG.error(e);
421      } finally {
422        if (table != null) {
423          try {
424            table.close();
425          } catch (IOException e) {/* DO NOTHING */
426            LOG.error("Close table failed", e);
427          }
428        }
429        scan = null;
430        get = null;
431        startKey = null;
432      }
433      return null;
434    }
435  }

489  private int parseArgs(String[] args) {
490    int index = -1;
491    // Process command line args
492    for (int i = 0; i < args.length; i++) {
493      String cmd = args[i];
494
495      if (cmd.startsWith("-")) {
496        if (index >= 0) {
497          // command line args must be in the form: [opts] [table 1 [table 2 ...]]
498          System.err.println("Invalid command line options");
499          printUsageAndExit();
500        }
501
502        if (cmd.equals("-help")) {
503          // user asked for help, print the help and quit.
504          printUsageAndExit();
505        } else if (cmd.equals("-daemon") && interval == 0) {
506          // user asked for daemon mode, set a default interval between checks
507          interval = DEFAULT_INTERVAL;
508        } else if (cmd.equals("-interval")) {
509          // user has specified an interval for canary breaths (-interval N)
510          i++;
511
512          if (i == args.length) {
513            System.err.println("-interval needs a numeric value argument.");
514            printUsageAndExit();
515          }
516
517          try {
518            interval = Long.parseLong(args[i]) * 1000;
519          } catch (NumberFormatException e) {
520            System.err.println("-interval needs a numeric value argument.");
521            printUsageAndExit();
522          }
523        } else if(cmd.equals("-regionserver")) {
524          this.regionServerMode = true;
525        } else if(cmd.equals("-allRegions")) {
526          this.regionServerAllRegions = true;
527        } else if(cmd.equals("-writeSniffing")) {
528          this.writeSniffing = true;
529        } else if(cmd.equals("-treatFailureAsError")) {
530          this.treatFailureAsError = true;
531        } else if (cmd.equals("-e")) {
532          this.useRegExp = true;
533        } else if (cmd.equals("-t")) {
534          i++;
535
536          if (i == args.length) {
537            System.err.println("-t needs a numeric value argument.");
538            printUsageAndExit();
539          }
540
541          try {
542            this.timeout = Long.parseLong(args[i]);
543          } catch (NumberFormatException e) {
544            System.err.println("-t needs a numeric value argument.");
545            printUsageAndExit();
546          }
547        } else if (cmd.equals("-writeTable")) {
548          i++;
549
550          if (i == args.length) {
551            System.err.println("-writeTable needs a string value argument.");
552            printUsageAndExit();
553          }
554          this.writeTableName = TableName.valueOf(args[i]);
555        } else if (cmd.equals("-f")) {
556          i++;
557
558          if (i == args.length) {
559            System.err
560                .println("-f needs a boolean value argument (true|false).");
561            printUsageAndExit();
562          }
563
564          this.failOnError = Boolean.parseBoolean(args[i]);
565        } else {
566          // no options match
567          System.err.println(cmd + " options is invalid.");
568          printUsageAndExit();
569        }
570      } else if (index < 0) {
571        // keep track of first table name specified by the user
572        index = i;
573      }
574    }
575    if (this.regionServerAllRegions && !this.regionServerMode) {
576      System.err.println("-allRegions can only be specified in regionserver mode.");
577      printUsageAndExit();
578    }
579    return index;
580  }

582  @Override
583  public int run(String[] args) throws Exception {
584    int index = parseArgs(args);
585    ChoreService choreService = null;
586
587    // Launches chore for refreshing kerberos credentials if security is enabled.
588    // Please see http://hbase.apache.org/book.html#_running_canary_in_a_kerberos_enabled_cluster
589    // for more details.
590    final ScheduledChore authChore = AuthUtil.getAuthChore(conf);
591    if (authChore != null) {
592      choreService = new ChoreService("CANARY_TOOL");
593      choreService.scheduleChore(authChore);
594    }
595
596    // Start to prepare the stuffs
597    Monitor monitor = null;
598    Thread monitorThread = null;
599    long startTime = 0;
600    long currentTimeLength = 0;
601    // Get a connection to use in below.
602    try (Connection connection = ConnectionFactory.createConnection(this.conf)) {
603      do {
604        // Do monitor !!
605        try {
606          monitor = this.newMonitor(connection, index, args);
607          monitorThread = new Thread(monitor);
608          startTime = System.currentTimeMillis();
609          monitorThread.start();
610          while (!monitor.isDone()) {
611            // wait for 1 sec
612            Thread.sleep(1000);
613            // exit if any error occurs
614            if (this.failOnError && monitor.hasError()) {
615              monitorThread.interrupt();
616              if (monitor.initialized) {
617                return monitor.errorCode;
618              } else {
619                return INIT_ERROR_EXIT_CODE;
620              }
621            }
622            currentTimeLength = System.currentTimeMillis() - startTime;
623            if (currentTimeLength > this.timeout) {
624              LOG.error("The monitor is running too long (" + currentTimeLength
625                  + ") after timeout limit:" + this.timeout
626                  + " will be killed itself !!");
627              if (monitor.initialized) {
628                return TIMEOUT_ERROR_EXIT_CODE;
629              } else {
630                return INIT_ERROR_EXIT_CODE;
631              }
632            }
633          }
634
635          if (this.failOnError && monitor.finalCheckForErrors()) {
636            monitorThread.interrupt();
637            return monitor.errorCode;
638          }
639        } finally {
640          if (monitor != null) monitor.close();
641        }
642
643        Thread.sleep(interval);
644      } while (interval > 0);
645    } // try-with-resources close
646
647    if (choreService != null) {
648      choreService.shutdown();
649    }
650    return monitor.errorCode;
651  }

653  private void printUsageAndExit() {
654    System.err.printf(
655      "Usage: bin/hbase %s [opts] [table1 [table2]...] | [regionserver1 [regionserver2]..]%n",
656        getClass().getName());
657    System.err.println(" where [opts] are:");
658    System.err.println("   -help          Show this help and exit.");
659    System.err.println("   -regionserver  replace the table argument to regionserver,");
660    System.err.println("      which means to enable regionserver mode");
661    System.err.println("   -allRegions    Tries all regions on a regionserver,");
662    System.err.println("      only works in regionserver mode.");
663    System.err.println("   -daemon        Continuous check at defined intervals.");
664    System.err.println("   -interval <N>  Interval between checks (sec)");
665    System.err.println("   -e             Use table/regionserver as regular expression");
666    System.err.println("      which means the table/regionserver is regular expression pattern");
667    System.err.println("   -f <B>         stop whole program if first error occurs," +
668        " default is true");
669    System.err.println("   -t <N>         timeout for a check, default is 600000 (milisecs)");
670    System.err.println("   -writeSniffing enable the write sniffing in canary");
671    System.err.println("   -treatFailureAsError treats read / write failure as error");
672    System.err.println("   -writeTable    The table used for write sniffing."
673        + " Default is hbase:canary");
674    System.err
675        .println("   -D<configProperty>=<value> assigning or override the configuration params");
676    System.exit(USAGE_EXIT_CODE);
677  }

679  /**
680   * A Factory method for {@link Monitor}.
681   * Can be overridden by user.
682   * @param index a start index for monitor target
683   * @param args args passed from user
684   * @return a Monitor instance
685   */
686  public Monitor newMonitor(final Connection connection, int index, String[] args) {
687    Monitor monitor = null;
688    String[] monitorTargets = null;
689
690    if(index >= 0) {
691      int length = args.length - index;
692      monitorTargets = new String[length];
693      System.arraycopy(args, index, monitorTargets, 0, length);
694    }
695
696    if (this.regionServerMode) {
697      monitor =
698          new RegionServerMonitor(connection, monitorTargets, this.useRegExp,
699              (ExtendedSink) this.sink, this.executor, this.regionServerAllRegions,
700              this.treatFailureAsError);
701    } else {
702      monitor =
703          new RegionMonitor(connection, monitorTargets, this.useRegExp, this.sink, this.executor,
704              this.writeSniffing, this.writeTableName, this.treatFailureAsError);
705    }
706    return monitor;
707  }

709  // a Monitor super-class can be extended by users
710  public static abstract class Monitor implements Runnable, Closeable {
711
712    protected Connection connection;
713    protected Admin admin;
714    protected String[] targets;
715    protected boolean useRegExp;
716    protected boolean treatFailureAsError;
717    protected boolean initialized = false;
718
719    protected boolean done = false;
720    protected int errorCode = 0;
721    protected Sink sink;
722    protected ExecutorService executor;
723
724    public boolean isDone() {
725      return done;
726    }
727
728    public boolean hasError() {
729      return errorCode != 0;
730    }
731
732    public boolean finalCheckForErrors() {
733      if (errorCode != 0) {
734        return true;
735      }
736      if (treatFailureAsError &&
737          (sink.getReadFailureCount() > 0 || sink.getWriteFailureCount() > 0)) {
738        errorCode = FAILURE_EXIT_CODE;
739        return true;
740      }
741      return false;
742    }
743
744    @Override
745    public void close() throws IOException {
746      if (this.admin != null) this.admin.close();
747    }
748
749    protected Monitor(Connection connection, String[] monitorTargets, boolean useRegExp, Sink sink,
750        ExecutorService executor, boolean treatFailureAsError) {
751      if (null == connection) throw new IllegalArgumentException("connection shall not be null");
752
753      this.connection = connection;
754      this.targets = monitorTargets;
755      this.useRegExp = useRegExp;
756      this.treatFailureAsError = treatFailureAsError;
757      this.sink = sink;
758      this.executor = executor;
759    }
760
761    @Override
762    public abstract void run();
763
764    protected boolean initAdmin() {
765      if (null == this.admin) {
766        try {
767          this.admin = this.connection.getAdmin();
768        } catch (Exception e) {
769          LOG.error("Initial HBaseAdmin failed...", e);
770          this.errorCode = INIT_ERROR_EXIT_CODE;
771        }
772      } else if (admin.isAborted()) {
773        LOG.error("HBaseAdmin aborted");
774        this.errorCode = INIT_ERROR_EXIT_CODE;
775      }
776      return !this.hasError();
777    }
778  }

780  // a monitor for region mode
781  private static class RegionMonitor extends Monitor {
782    // 10 minutes
783    private static final int DEFAULT_WRITE_TABLE_CHECK_PERIOD = 10 * 60 * 1000;
784    // 1 days
785    private static final int DEFAULT_WRITE_DATA_TTL = 24 * 60 * 60;
786
787    private long lastCheckTime = -1;
788    private boolean writeSniffing;
789    private TableName writeTableName;
790    private int writeDataTTL;
791    private float regionsLowerLimit;
792    private float regionsUpperLimit;
793    private int checkPeriod;
794
795    public RegionMonitor(Connection connection, String[] monitorTargets, boolean useRegExp,
796        Sink sink, ExecutorService executor, boolean writeSniffing, TableName writeTableName,
797        boolean treatFailureAsError) {
798      super(connection, monitorTargets, useRegExp, sink, executor, treatFailureAsError);
799      Configuration conf = connection.getConfiguration();
800      this.writeSniffing = writeSniffing;
801      this.writeTableName = writeTableName;
802      this.writeDataTTL =
803          conf.getInt(HConstants.HBASE_CANARY_WRITE_DATA_TTL_KEY, DEFAULT_WRITE_DATA_TTL);
804      this.regionsLowerLimit =
805          conf.getFloat(HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_LOWERLIMIT_KEY, 1.0f);
806      this.regionsUpperLimit =
807          conf.getFloat(HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_UPPERLIMIT_KEY, 1.5f);
808      this.checkPeriod =
809          conf.getInt(HConstants.HBASE_CANARY_WRITE_TABLE_CHECK_PERIOD_KEY,
810            DEFAULT_WRITE_TABLE_CHECK_PERIOD);
811    }
812
813    @Override
814    public void run() {
815      if (this.initAdmin()) {
816        try {
817          List<Future<Void>> taskFutures = new LinkedList<Future<Void>>();
818          if (this.targets != null && this.targets.length > 0) {
819            String[] tables = generateMonitorTables(this.targets);
820            this.initialized = true;
821            for (String table : tables) {
822              taskFutures.addAll(Canary.sniff(admin, sink, table, executor, TaskType.READ));
823            }
824          } else {
825            taskFutures.addAll(sniff(TaskType.READ));
826          }
827
828          if (writeSniffing) {
829            if (EnvironmentEdgeManager.currentTime() - lastCheckTime > checkPeriod) {
830              try {
831                checkWriteTableDistribution();
832              } catch (IOException e) {
833                LOG.error("Check canary table distribution failed!", e);
834              }
835              lastCheckTime = EnvironmentEdgeManager.currentTime();
836            }
837            // sniff canary table with write operation
838            taskFutures.addAll(Canary.sniff(admin, sink,
839              admin.getTableDescriptor(writeTableName), executor, TaskType.WRITE));
840          }
841
842          for (Future<Void> future : taskFutures) {
843            try {
844              future.get();
845            } catch (ExecutionException e) {
846              LOG.error("Sniff region failed!", e);
847            }
848          }
849        } catch (Exception e) {
850          LOG.error("Run regionMonitor failed", e);
851          this.errorCode = ERROR_EXIT_CODE;
852        }
853      }
854      this.done = true;
855    }
856
857    private String[] generateMonitorTables(String[] monitorTargets) throws IOException {
858      String[] returnTables = null;
859
860      if (this.useRegExp) {
861        Pattern pattern = null;
862        HTableDescriptor[] tds = null;
863        Set<String> tmpTables = new TreeSet<String>();
864        try {
865          if (LOG.isDebugEnabled()) {
866            LOG.debug(String.format("reading list of tables"));
867          }
868          tds = this.admin.listTables(pattern);
869          if (tds == null) {
870            tds = new HTableDescriptor[0];
871          }
872          for (String monitorTarget : monitorTargets) {
873            pattern = Pattern.compile(monitorTarget);
874            for (HTableDescriptor td : tds) {
875              if (pattern.matcher(td.getNameAsString()).matches()) {
876                tmpTables.add(td.getNameAsString());
877              }
878            }
879          }
880        } catch (IOException e) {
881          LOG.error("Communicate with admin failed", e);
882          throw e;
883        }
884
885        if (tmpTables.size() > 0) {
886          returnTables = tmpTables.toArray(new String[tmpTables.size()]);
887        } else {
888          String msg = "No HTable found, tablePattern:" + Arrays.toString(monitorTargets);
889          LOG.error(msg);
890          this.errorCode = INIT_ERROR_EXIT_CODE;
891          throw new TableNotFoundException(msg);
892        }
893      } else {
894        returnTables = monitorTargets;
895      }
896
897      return returnTables;
898    }

900    /*
901     * canary entry point to monitor all the tables.
902     */
903    private List<Future<Void>> sniff(TaskType taskType) throws Exception {
904      if (LOG.isDebugEnabled()) {
905        LOG.debug(String.format("reading list of tables"));
906      }
907      List<Future<Void>> taskFutures = new LinkedList<Future<Void>>();
908      for (HTableDescriptor table : admin.listTables()) {
909        if (admin.isTableEnabled(table.getTableName())
910            && (!table.getTableName().equals(writeTableName))) {
911          taskFutures.addAll(Canary.sniff(admin, sink, table, executor, taskType));
912        }
913      }
914      return taskFutures;
915    }
916
917    private void checkWriteTableDistribution() throws IOException {
918      if (!admin.tableExists(writeTableName)) {
919        int numberOfServers = admin.getClusterStatus().getServers().size();
920        if (numberOfServers == 0) {
921          throw new IllegalStateException("No live regionservers");
922        }
923        createWriteTable(numberOfServers);
924      }
925
926      if (!admin.isTableEnabled(writeTableName)) {
927        admin.enableTable(writeTableName);
928      }
929
930      int numberOfServers = admin.getClusterStatus().getServers().size();
931      List<Pair<HRegionInfo, ServerName>> pairs =
932          MetaTableAccessor.getTableRegionsAndLocations(connection, writeTableName);
933      int numberOfRegions = pairs.size();
934      if (numberOfRegions < numberOfServers * regionsLowerLimit
935          || numberOfRegions > numberOfServers * regionsUpperLimit) {
936        admin.disableTable(writeTableName);
937        admin.deleteTable(writeTableName);
938        createWriteTable(numberOfServers);
939      }
940      HashSet<ServerName> serverSet = new HashSet<ServerName>();
941      for (Pair<HRegionInfo, ServerName> pair : pairs) {
942        serverSet.add(pair.getSecond());
943      }
944      int numberOfCoveredServers = serverSet.size();
945      if (numberOfCoveredServers < numberOfServers) {
946        admin.balancer();
947      }
948    }
949
950    private void createWriteTable(int numberOfServers) throws IOException {
951      int numberOfRegions = (int)(numberOfServers * regionsLowerLimit);
952      LOG.info("Number of live regionservers: " + numberOfServers + ", "
953          + "pre-splitting the canary table into " + numberOfRegions + " regions "
954          + "(current lower limit of regions per server is " + regionsLowerLimit
955          + " and you can change it by config: "
956          + HConstants.HBASE_CANARY_WRITE_PERSERVER_REGIONS_LOWERLIMIT_KEY + " )");
957      HTableDescriptor desc = new HTableDescriptor(writeTableName);
958      HColumnDescriptor family = new HColumnDescriptor(CANARY_TABLE_FAMILY_NAME);
959      family.setMaxVersions(1);
960      family.setTimeToLive(writeDataTTL);
961
962      desc.addFamily(family);
963      byte[][] splits = new RegionSplitter.HexStringSplit().split(numberOfRegions);
964      admin.createTable(desc, splits);
965    }

1011   * Loops over regions that owns this table, and output some information abouts the state.
1012   */
1013  private static List<Future<Void>> sniff(final Admin admin, final Sink sink,
1014      HTableDescriptor tableDesc, ExecutorService executor, TaskType taskType) throws Exception {
1015
1016    if (LOG.isDebugEnabled()) {
1017      LOG.debug(String.format("reading list of regions for table %s", tableDesc.getTableName()));
1018    }
1019
1020    Table table = null;
1021    try {
1022      table = admin.getConnection().getTable(tableDesc.getTableName());
1023    } catch (TableNotFoundException e) {
1024      return new ArrayList<Future<Void>>();
1025    }
1026    List<RegionTask> tasks = new ArrayList<RegionTask>();
1027    try {
1028      for (HRegionInfo region : admin.getTableRegions(tableDesc.getTableName())) {
1029        tasks.add(new RegionTask(admin.getConnection(), region, sink, taskType));
1030      }
1031    } finally {
1032      table.close();
1033    }
1034    return executor.invokeAll(tasks);

1036  // a monitor for regionserver mode
1037  private static class RegionServerMonitor extends Monitor {
1038
1039    private boolean allRegions;
1040
1041    public RegionServerMonitor(Connection connection, String[] monitorTargets, boolean useRegExp,
1042        ExtendedSink sink, ExecutorService executor, boolean allRegions,
1043        boolean treatFailureAsError) {
1044      super(connection, monitorTargets, useRegExp, sink, executor, treatFailureAsError);
1045      this.allRegions = allRegions;
1046    }
1047
1048    private ExtendedSink getSink() {
1049      return (ExtendedSink) this.sink;
1050    }
1051
1052    @Override
1053    public void run() {
1054      if (this.initAdmin() && this.checkNoTableNames()) {
1055        Map<String, List<HRegionInfo>> rsAndRMap = this.filterRegionServerByName();
1056        this.initialized = true;
1057        this.monitorRegionServers(rsAndRMap);
1058      }
1059      this.done = true;
1060    }
1061
1062    private boolean checkNoTableNames() {
1063      List<String> foundTableNames = new ArrayList<String>();
1064      TableName[] tableNames = null;
1065
1066      if (LOG.isDebugEnabled()) {
1067        LOG.debug(String.format("reading list of tables"));
1068      }
1069      try {
1070        tableNames = this.admin.listTableNames();
1071      } catch (IOException e) {
1072        LOG.error("Get listTableNames failed", e);
1073        this.errorCode = INIT_ERROR_EXIT_CODE;
1074        return false;
1075      }
1076
1077      if (this.targets == null || this.targets.length == 0) return true;
1078
1079      for (String target : this.targets) {
1080        for (TableName tableName : tableNames) {
1081          if (target.equals(tableName.getNameAsString())) {
1082            foundTableNames.add(target);
1083          }
1084        }
1085      }
1086
1087      if (foundTableNames.size() > 0) {
1088        System.err.println("Cannot pass a tablename when using the -regionserver " +
1089            "option, tablenames:" + foundTableNames.toString());
1090        this.errorCode = USAGE_EXIT_CODE;
1091      }
1092      return foundTableNames.size() == 0;
1093    }
1094
1095    private void monitorRegionServers(Map<String, List<HRegionInfo>> rsAndRMap) {
1096      List<RegionServerTask> tasks = new ArrayList<RegionServerTask>();
1097      Map<String, AtomicLong> successMap = new HashMap<String, AtomicLong>();
1098      Random rand = new Random();
1099      for (Map.Entry<String, List<HRegionInfo>> entry : rsAndRMap.entrySet()) {
1100        String serverName = entry.getKey();
1101        AtomicLong successes = new AtomicLong(0);
1102        successMap.put(serverName, successes);
1103        if (this.allRegions) {
1104          for (HRegionInfo region : entry.getValue()) {
1105            tasks.add(new RegionServerTask(this.connection,
1106                serverName,
1107                region,
1108                getSink(),
1109                successes));
1110          }
1111        } else {
1112          // random select a region if flag not set
1113          HRegionInfo region = entry.getValue().get(rand.nextInt(entry.getValue().size()));
1114          tasks.add(new RegionServerTask(this.connection,
1115              serverName,
1116              region,
1117              getSink(),
1118              successes));
1119        }
1120      }
1121      try {
1122        for (Future<Void> future : this.executor.invokeAll(tasks)) {
1123          try {
1124            future.get();
1125          } catch (ExecutionException e) {
1126            LOG.error("Sniff regionserver failed!", e);
1127            this.errorCode = ERROR_EXIT_CODE;
1128          }
1129        }
1130        if (this.allRegions) {
1131          for (Map.Entry<String, List<HRegionInfo>> entry : rsAndRMap.entrySet()) {
1132            String serverName = entry.getKey();
1133            LOG.info("Successfully read " + successMap.get(serverName) + " regions out of "
1134                    + entry.getValue().size() + " on regionserver:" + serverName);
1135          }
1136        }
1137      } catch (InterruptedException e) {
1138        this.errorCode = ERROR_EXIT_CODE;
1139        LOG.error("Sniff regionserver interrupted!", e);
1140      }
1141    }
1142
1143    private Map<String, List<HRegionInfo>> filterRegionServerByName() {
1144      Map<String, List<HRegionInfo>> regionServerAndRegionsMap = this.getAllRegionServerByName();
1145      regionServerAndRegionsMap = this.doFilterRegionServerByName(regionServerAndRegionsMap);
1146      return regionServerAndRegionsMap;
1147    }
1148
1149    private Map<String, List<HRegionInfo>> getAllRegionServerByName() {
1150      Map<String, List<HRegionInfo>> rsAndRMap = new HashMap<String, List<HRegionInfo>>();
1151      Table table = null;
1152      RegionLocator regionLocator = null;
1153      try {
1154        if (LOG.isDebugEnabled()) {
1155          LOG.debug(String.format("reading list of tables and locations"));
1156        }
1157        HTableDescriptor[] tableDescs = this.admin.listTables();
1158        List<HRegionInfo> regions = null;
1159        for (HTableDescriptor tableDesc : tableDescs) {
1160          table = this.admin.getConnection().getTable(tableDesc.getTableName());
1161          regionLocator = this.admin.getConnection().getRegionLocator(tableDesc.getTableName());
1162
1163          for (HRegionLocation location : regionLocator.getAllRegionLocations()) {
1164            ServerName rs = location.getServerName();
1165            String rsName = rs.getHostname();
1166            HRegionInfo r = location.getRegionInfo();
1167
1168            if (rsAndRMap.containsKey(rsName)) {
1169              regions = rsAndRMap.get(rsName);
1170            } else {
1171              regions = new ArrayList<HRegionInfo>();
1172              rsAndRMap.put(rsName, regions);
1173            }
1174            regions.add(r);
1175          }
1176          table.close();
1177        }
1178
1179      } catch (IOException e) {
1180        String msg = "Get HTables info failed";
1181        LOG.error(msg, e);
1182        this.errorCode = INIT_ERROR_EXIT_CODE;
1183      } finally {
1184        if (table != null) {
1185          try {
1186            table.close();
1187          } catch (IOException e) {
1188            LOG.warn("Close table failed", e);
1189          }
1190        }
1191      }
1192
1193      return rsAndRMap;
1194    }

1196    private Map<String, List<HRegionInfo>> doFilterRegionServerByName(
1197        Map<String, List<HRegionInfo>> fullRsAndRMap) {
1198
1199      Map<String, List<HRegionInfo>> filteredRsAndRMap = null;
1200
1201      if (this.targets != null && this.targets.length > 0) {
1202        filteredRsAndRMap = new HashMap<String, List<HRegionInfo>>();
1203        Pattern pattern = null;
1204        Matcher matcher = null;
1205        boolean regExpFound = false;
1206        for (String rsName : this.targets) {
1207          if (this.useRegExp) {
1208            regExpFound = false;
1209            pattern = Pattern.compile(rsName);
1210            for (Map.Entry<String, List<HRegionInfo>> entry : fullRsAndRMap.entrySet()) {
1211              matcher = pattern.matcher(entry.getKey());
1212              if (matcher.matches()) {
1213                filteredRsAndRMap.put(entry.getKey(), entry.getValue());
1214                regExpFound = true;
1215              }
1216            }
1217            if (!regExpFound) {
1218              LOG.info("No RegionServerInfo found, regionServerPattern:" + rsName);
1219            }
1220          } else {
1221            if (fullRsAndRMap.containsKey(rsName)) {
1222              filteredRsAndRMap.put(rsName, fullRsAndRMap.get(rsName));
1223            } else {
1224              LOG.info("No RegionServerInfo found, regionServerName:" + rsName);
1225            }
1226          }
1227        }
1228      } else {
1229        filteredRsAndRMap = fullRsAndRMap;
1230      }
1231      return filteredRsAndRMap;
1232    }
