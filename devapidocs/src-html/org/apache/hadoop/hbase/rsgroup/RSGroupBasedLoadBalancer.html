058/**
059 * GroupBasedLoadBalancer, used when Region Server Grouping is configured (HBase-6721)
060 * It does region balance based on a table's group membership.
061 *
062 * Most assignment methods contain two exclusive code paths: Online - when the group
063 * table is online and Offline - when it is unavailable.
064 *
065 * During Offline, assignments are assigned based on cached information in zookeeper.
066 * If unavailable (ie bootstrap) then regions are assigned randomly.
067 *
068 * Once the GROUP table has been assigned, the balancer switches to Online and will then
069 * start providing appropriate assignments for user tables.
070 *
071 */
072@InterfaceAudience.Private
073public class RSGroupBasedLoadBalancer implements RSGroupableBalancer {
074  private static final Log LOG = LogFactory.getLog(RSGroupBasedLoadBalancer.class);
075

114  @Override
115  public List<RegionPlan> balanceCluster(Map<ServerName, List<RegionInfo>> clusterState)
116      throws HBaseIOException {
117    if (!isOnline()) {
118      throw new ConstraintException(RSGroupInfoManager.RSGROUP_TABLE_NAME +
119          " is not online, unable to perform balance");
120    }
121
122    Map<ServerName,List<RegionInfo>> correctedState = correctAssignments(clusterState);
123    List<RegionPlan> regionPlans = new ArrayList<>();
124
125    List<RegionInfo> misplacedRegions = correctedState.get(LoadBalancer.BOGUS_SERVER_NAME);
126    for (RegionInfo regionInfo : misplacedRegions) {
127      ServerName serverName = findServerForRegion(clusterState, regionInfo);
128      regionPlans.add(new RegionPlan(regionInfo, serverName, null));
129    }
130    try {
131      List<RSGroupInfo> rsgi = rsGroupInfoManager.listRSGroups();
132      for (RSGroupInfo info: rsgi) {
133        Map<ServerName, List<RegionInfo>> groupClusterState = new HashMap<>();
134        Map<TableName, Map<ServerName, List<RegionInfo>>> groupClusterLoad = new HashMap<>();
135        for (Address sName : info.getServers()) {
136          for(ServerName curr: clusterState.keySet()) {
137            if(curr.getAddress().equals(sName)) {
138              groupClusterState.put(curr, correctedState.get(curr));
139            }
140          }
141        }
142        groupClusterLoad.put(HConstants.ENSEMBLE_TABLE_NAME, groupClusterState);
143        this.internalBalancer.setClusterLoad(groupClusterLoad);
144        List<RegionPlan> groupPlans = this.internalBalancer
145            .balanceCluster(groupClusterState);
146        if (groupPlans != null) {
147          regionPlans.addAll(groupPlans);
148        }
149      }
150    } catch (IOException exp) {
151      LOG.warn("Exception while balancing cluster.", exp);
152      regionPlans.clear();
153    }
154    return regionPlans;
155  }

157  @Override
158  public Map<ServerName, List<RegionInfo>> roundRobinAssignment(
159      List<RegionInfo> regions, List<ServerName> servers) throws HBaseIOException {
160    Map<ServerName, List<RegionInfo>> assignments = Maps.newHashMap();
161    ListMultimap<String,RegionInfo> regionMap = ArrayListMultimap.create();
162    ListMultimap<String,ServerName> serverMap = ArrayListMultimap.create();
163    generateGroupMaps(regions, servers, regionMap, serverMap);
164    for(String groupKey : regionMap.keySet()) {
165      if (regionMap.get(groupKey).size() > 0) {
166        Map<ServerName, List<RegionInfo>> result =
167            this.internalBalancer.roundRobinAssignment(
168                regionMap.get(groupKey),
169                serverMap.get(groupKey));
170        if(result != null) {
171          if(result.containsKey(LoadBalancer.BOGUS_SERVER_NAME) &&
172              assignments.containsKey(LoadBalancer.BOGUS_SERVER_NAME)){
173            assignments.get(LoadBalancer.BOGUS_SERVER_NAME).addAll(
174              result.get(LoadBalancer.BOGUS_SERVER_NAME));
175          } else {
176            assignments.putAll(result);
177          }
178        }
179      }
180    }
181    return assignments;
182  }

184  @Override
185  public Map<ServerName, List<RegionInfo>> retainAssignment(
186      Map<RegionInfo, ServerName> regions, List<ServerName> servers) throws HBaseIOException {
187    try {
188      Map<ServerName, List<RegionInfo>> assignments = new TreeMap<>();
189      ListMultimap<String, RegionInfo> groupToRegion = ArrayListMultimap.create();
190      Set<RegionInfo> misplacedRegions = getMisplacedRegions(regions);
191      for (RegionInfo region : regions.keySet()) {
192        if (!misplacedRegions.contains(region)) {
193          String groupName = rsGroupInfoManager.getRSGroupOfTable(region.getTable());
194          groupToRegion.put(groupName, region);
195        }
196      }
197      // Now the "groupToRegion" map has only the regions which have correct
198      // assignments.
199      for (String key : groupToRegion.keySet()) {
200        Map<RegionInfo, ServerName> currentAssignmentMap = new TreeMap<RegionInfo, ServerName>();
201        List<RegionInfo> regionList = groupToRegion.get(key);
202        RSGroupInfo info = rsGroupInfoManager.getRSGroup(key);
203        List<ServerName> candidateList = filterOfflineServers(info, servers);
204        for (RegionInfo region : regionList) {
205          currentAssignmentMap.put(region, regions.get(region));
206        }
207        if(candidateList.size() > 0) {
208          assignments.putAll(this.internalBalancer.retainAssignment(
209              currentAssignmentMap, candidateList));
210        }
211      }
212
213      for (RegionInfo region : misplacedRegions) {
214        String groupName = rsGroupInfoManager.getRSGroupOfTable(region.getTable());;
215        RSGroupInfo info = rsGroupInfoManager.getRSGroup(groupName);
216        List<ServerName> candidateList = filterOfflineServers(info, servers);
217        ServerName server = this.internalBalancer.randomAssignment(region,
218            candidateList);
219        if (server != null) {
220          if (!assignments.containsKey(server)) {
221            assignments.put(server, new ArrayList<>());
222          }
223          assignments.get(server).add(region);
224        } else {
225          //if not server is available assign to bogus so it ends up in RIT
226          if(!assignments.containsKey(LoadBalancer.BOGUS_SERVER_NAME)) {
227            assignments.put(LoadBalancer.BOGUS_SERVER_NAME, new ArrayList<>());
228          }
229          assignments.get(LoadBalancer.BOGUS_SERVER_NAME).add(region);
230        }
231      }
232      return assignments;
233    } catch (IOException e) {
234      throw new HBaseIOException("Failed to do online retain assignment", e);
235    }
236  }

238  @Override
239  public ServerName randomAssignment(RegionInfo region,
240      List<ServerName> servers) throws HBaseIOException {
241    ListMultimap<String,RegionInfo> regionMap = LinkedListMultimap.create();
242    ListMultimap<String,ServerName> serverMap = LinkedListMultimap.create();
243    generateGroupMaps(Lists.newArrayList(region), servers, regionMap, serverMap);
244    List<ServerName> filteredServers = serverMap.get(regionMap.keySet().iterator().next());
245    return this.internalBalancer.randomAssignment(region, filteredServers);
246  }
247
248  private void generateGroupMaps(
249    List<RegionInfo> regions,
250    List<ServerName> servers,
251    ListMultimap<String, RegionInfo> regionMap,
252    ListMultimap<String, ServerName> serverMap) throws HBaseIOException {
253    try {
254      for (RegionInfo region : regions) {
255        String groupName = rsGroupInfoManager.getRSGroupOfTable(region.getTable());
256        if (groupName == null) {
257          LOG.warn("Group for table "+region.getTable()+" is null");
258        }
259        regionMap.put(groupName, region);
260      }
261      for (String groupKey : regionMap.keySet()) {
262        RSGroupInfo info = rsGroupInfoManager.getRSGroup(groupKey);
263        serverMap.putAll(groupKey, filterOfflineServers(info, servers));
264        if(serverMap.get(groupKey).size() < 1) {
265          serverMap.put(groupKey, LoadBalancer.BOGUS_SERVER_NAME);
266        }
267      }
268    } catch(IOException e) {
269      throw new HBaseIOException("Failed to generate group maps", e);
270    }
271  }
272
273  private List<ServerName> filterOfflineServers(RSGroupInfo RSGroupInfo,
274                                                List<ServerName> onlineServers) {
275    if (RSGroupInfo != null) {
276      return filterServers(RSGroupInfo.getServers(), onlineServers);
277    } else {
278      LOG.warn("RSGroup Information found to be null. Some regions might be unassigned.");
279      return Collections.EMPTY_LIST;
280    }
281  }

283  /**
284   * Filter servers based on the online servers.
285   *
286   * @param servers
287   *          the servers
288   * @param onlineServers
289   *          List of servers which are online.
290   * @return the list
291   */
292  private List<ServerName> filterServers(Collection<Address> servers,
293      Collection<ServerName> onlineServers) {
294    ArrayList<ServerName> finalList = new ArrayList<ServerName>();
295    for (Address server : servers) {
296      for(ServerName curr: onlineServers) {
297        if(curr.getAddress().equals(server)) {
298          finalList.add(curr);
299        }
300      }
301    }
302    return finalList;
303  }

305  private Set<RegionInfo> getMisplacedRegions(
306      Map<RegionInfo, ServerName> regions) throws IOException {
307    Set<RegionInfo> misplacedRegions = new HashSet<>();
308    for(Map.Entry<RegionInfo, ServerName> region : regions.entrySet()) {
309      RegionInfo regionInfo = region.getKey();
310      ServerName assignedServer = region.getValue();
311      RSGroupInfo info = rsGroupInfoManager.getRSGroup(rsGroupInfoManager.
312              getRSGroupOfTable(regionInfo.getTable()));
313      if (assignedServer != null &&
314          (info == null || !info.containsServer(assignedServer.getAddress()))) {
315        RSGroupInfo otherInfo = null;
316        otherInfo = rsGroupInfoManager.getRSGroupOfServer(assignedServer.getAddress());
317        LOG.debug("Found misplaced region: " + regionInfo.getRegionNameAsString() +
318            " on server: " + assignedServer +
319            " found in group: " +  otherInfo +
320            " outside of group: " + (info == null ? "UNKNOWN" : info.getName()));
321        misplacedRegions.add(regionInfo);
322      }
323    }
324    return misplacedRegions;
325  }

327  private ServerName findServerForRegion(
328      Map<ServerName, List<RegionInfo>> existingAssignments, RegionInfo region)
329  {
330    for (Map.Entry<ServerName, List<RegionInfo>> entry : existingAssignments.entrySet()) {
331      if (entry.getValue().contains(region)) {
332        return entry.getKey();
333      }
334    }
335
336    throw new IllegalStateException("Could not find server for region "
337        + region.getShortNameToLog());
338  }

340  private Map<ServerName, List<RegionInfo>> correctAssignments(
341       Map<ServerName, List<RegionInfo>> existingAssignments)
342  throws HBaseIOException{
343    Map<ServerName, List<RegionInfo>> correctAssignments = new TreeMap<>();
344    List<RegionInfo> misplacedRegions = new LinkedList<>();
345    correctAssignments.put(LoadBalancer.BOGUS_SERVER_NAME, new LinkedList<>());
346    for (Map.Entry<ServerName, List<RegionInfo>> assignments : existingAssignments.entrySet()){
347      ServerName sName = assignments.getKey();
348      correctAssignments.put(sName, new LinkedList<>());
349      List<RegionInfo> regions = assignments.getValue();
350      for (RegionInfo region : regions) {
351        RSGroupInfo info = null;
352        try {
353          info = rsGroupInfoManager.getRSGroup(
354              rsGroupInfoManager.getRSGroupOfTable(region.getTable()));
355        } catch (IOException exp) {
356          LOG.debug("RSGroup information null for region of table " + region.getTable(),
357              exp);
358        }
359        if ((info == null) || (!info.containsServer(sName.getAddress()))) {
360          correctAssignments.get(LoadBalancer.BOGUS_SERVER_NAME).add(region);
361        } else {
362          correctAssignments.get(sName).add(region);
363        }
364      }
365    }
366
367    //TODO bulk unassign?
368    //unassign misplaced regions, so that they are assigned to correct groups.
369    for(RegionInfo info: misplacedRegions) {
370      try {
371        this.masterServices.getAssignmentManager().unassign(info);
372      } catch (IOException e) {
373        throw new HBaseIOException(e);
374      }
375    }
376    return correctAssignments;
377  }

379  @Override
380  public void initialize() throws HBaseIOException {
381    try {
382      if (rsGroupInfoManager == null) {
383        List<RSGroupAdminEndpoint> cps =
384          masterServices.getMasterCoprocessorHost().findCoprocessors(RSGroupAdminEndpoint.class);
385        if (cps.size() != 1) {
386          String msg = "Expected one implementation of GroupAdminEndpoint but found " + cps.size();
387          LOG.error(msg);
388          throw new HBaseIOException(msg);
389        }
390        rsGroupInfoManager = cps.get(0).getGroupInfoManager();
391      }
392    } catch (IOException e) {
393      throw new HBaseIOException("Failed to initialize GroupInfoManagerImpl", e);
394    }
395
396    // Create the balancer
397    Class<? extends LoadBalancer> balancerKlass = config.getClass(HBASE_RSGROUP_LOADBALANCER_CLASS,
398        StochasticLoadBalancer.class, LoadBalancer.class);
399    internalBalancer = ReflectionUtils.newInstance(balancerKlass, config);
400    internalBalancer.setMasterServices(masterServices);
401    internalBalancer.setClusterStatus(clusterStatus);
402    internalBalancer.setConf(config);
403    internalBalancer.initialize();
404  }

406  public boolean isOnline() {
407    if (this.rsGroupInfoManager == null) return false;
408    return this.rsGroupInfoManager.isOnline();
409  }
