059/**
060 * GroupBasedLoadBalancer, used when Region Server Grouping is configured (HBase-6721) It does
061 * region balance based on a table's group membership.
062 * <p/>
063 * Most assignment methods contain two exclusive code paths: Online - when the group table is online
064 * and Offline - when it is unavailable.
065 * <p/>
066 * During Offline, assignments are assigned based on cached information in zookeeper. If unavailable
067 * (ie bootstrap) then regions are assigned randomly.
068 * <p/>
069 * Once the GROUP table has been assigned, the balancer switches to Online and will then start
070 * providing appropriate assignments for user tables.
071 */
072@InterfaceAudience.Private
073public class RSGroupBasedLoadBalancer implements LoadBalancer {
074  private static final Logger LOG = LoggerFactory.getLogger(RSGroupBasedLoadBalancer.class);
075
076  private MasterServices masterServices;
077  private ClusterInfoProvider provider;
078  private FavoredNodesManager favoredNodesManager;
079  private volatile RSGroupInfoManager rsGroupInfoManager;
080  private volatile LoadBalancer internalBalancer;

110  /**
111   * Balance by RSGroup.
112   */
113  @Override
114  public synchronized List<RegionPlan> balanceCluster(
115      Map<TableName, Map<ServerName, List<RegionInfo>>> loadOfAllTable) throws IOException {
116    if (!isOnline()) {
117      throw new ConstraintException(
118          RSGroupInfoManager.class.getSimpleName() + " is not online, unable to perform balance");
119    }
120
121    // Calculate correct assignments and a list of RegionPlan for mis-placed regions
122    Pair<Map<TableName, Map<ServerName, List<RegionInfo>>>, List<RegionPlan>>
123      correctedStateAndRegionPlans = correctAssignments(loadOfAllTable);
124    Map<TableName, Map<ServerName, List<RegionInfo>>> correctedLoadOfAllTable =
125        correctedStateAndRegionPlans.getFirst();
126    List<RegionPlan> regionPlans = correctedStateAndRegionPlans.getSecond();
127    RSGroupInfo defaultInfo = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
128    // Add RegionPlan
129    // for the regions which have been placed according to the region server group assignment
130    // into the movement list
131    try {
132      // For each rsgroup
133      for (RSGroupInfo rsgroup : rsGroupInfoManager.listRSGroups()) {
134        Map<TableName, Map<ServerName, List<RegionInfo>>> loadOfTablesInGroup = new HashMap<>();
135        for (Map.Entry<TableName, Map<ServerName, List<RegionInfo>>> entry : correctedLoadOfAllTable
136            .entrySet()) {
137          TableName tableName = entry.getKey();
138          RSGroupInfo targetRSGInfo = RSGroupUtil
139              .getRSGroupInfo(masterServices, rsGroupInfoManager, tableName).orElse(defaultInfo);
140          if (targetRSGInfo.getName().equals(rsgroup.getName())) {
141            loadOfTablesInGroup.put(tableName, entry.getValue());
142          }
143        }
144        List<RegionPlan> groupPlans = null;
145        if (!loadOfTablesInGroup.isEmpty()) {
146          LOG.info("Start Generate Balance plan for group: " + rsgroup.getName());
147          groupPlans = this.internalBalancer.balanceCluster(loadOfTablesInGroup);
148        }
149        if (groupPlans != null) {
150          regionPlans.addAll(groupPlans);
151        }
152      }
153    } catch (IOException exp) {
154      LOG.warn("Exception while balancing cluster.", exp);
155      regionPlans.clear();
156    }
157
158    // Return the whole movement list
159    return regionPlans;
160  }

162  @Override
163  @NonNull
164  public Map<ServerName, List<RegionInfo>> roundRobinAssignment(List<RegionInfo> regions,
165      List<ServerName> servers) throws IOException {
166    Map<ServerName, List<RegionInfo>> assignments = Maps.newHashMap();
167    List<Pair<List<RegionInfo>, List<ServerName>>> pairs =
168        generateGroupAssignments(regions, servers);
169    for (Pair<List<RegionInfo>, List<ServerName>> pair : pairs) {
170      Map<ServerName, List<RegionInfo>> result =
171          this.internalBalancer.roundRobinAssignment(pair.getFirst(), pair.getSecond());
172      result.forEach((server, regionInfos) -> assignments
173          .computeIfAbsent(server, s -> Lists.newArrayList()).addAll(regionInfos));
174    }
175    return assignments;
176  }

178  @Override
179  @NonNull
180  public Map<ServerName, List<RegionInfo>> retainAssignment(Map<RegionInfo, ServerName> regions,
181      List<ServerName> servers) throws HBaseIOException {
182    try {
183      Map<ServerName, List<RegionInfo>> assignments = new TreeMap<>();
184      List<Pair<List<RegionInfo>, List<ServerName>>> pairs =
185          generateGroupAssignments(Lists.newArrayList(regions.keySet()), servers);
186      for (Pair<List<RegionInfo>, List<ServerName>> pair : pairs) {
187        List<RegionInfo> regionList = pair.getFirst();
188        Map<RegionInfo, ServerName> currentAssignmentMap = Maps.newTreeMap();
189        regionList.forEach(r -> currentAssignmentMap.put(r, regions.get(r)));
190        Map<ServerName, List<RegionInfo>> pairResult =
191            this.internalBalancer.retainAssignment(currentAssignmentMap, pair.getSecond());
192        pairResult.forEach((server, rs) -> assignments
193            .computeIfAbsent(server, s -> Lists.newArrayList()).addAll(rs));
194      }
195      return assignments;
196    } catch (IOException e) {
197      throw new HBaseIOException("Failed to do online retain assignment", e);
198    }
199  }

201  @Override
202  public ServerName randomAssignment(RegionInfo region,
203      List<ServerName> servers) throws IOException {
204    List<Pair<List<RegionInfo>, List<ServerName>>> pairs =
205        generateGroupAssignments(Lists.newArrayList(region), servers);
206    List<ServerName> filteredServers = pairs.iterator().next().getSecond();
207    return this.internalBalancer.randomAssignment(region, filteredServers);
208  }

210  private List<Pair<List<RegionInfo>, List<ServerName>>> generateGroupAssignments(
211      List<RegionInfo> regions, List<ServerName> servers) throws HBaseIOException {
212    try {
213      ListMultimap<String, RegionInfo> regionMap = ArrayListMultimap.create();
214      ListMultimap<String, ServerName> serverMap = ArrayListMultimap.create();
215      RSGroupInfo defaultInfo = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
216      for (RegionInfo region : regions) {
217        String groupName =
218            RSGroupUtil.getRSGroupInfo(masterServices, rsGroupInfoManager, region.getTable())
219                .orElse(defaultInfo).getName();
220        regionMap.put(groupName, region);
221      }
222      for (String groupKey : regionMap.keySet()) {
223        RSGroupInfo info = rsGroupInfoManager.getRSGroup(groupKey);
224        serverMap.putAll(groupKey, filterOfflineServers(info, servers));
225      }
226
227      List<Pair<List<RegionInfo>, List<ServerName>>> result = Lists.newArrayList();
228      List<RegionInfo> fallbackRegions = Lists.newArrayList();
229      for (String groupKey : regionMap.keySet()) {
230        if (serverMap.get(groupKey).isEmpty()) {
231          fallbackRegions.addAll(regionMap.get(groupKey));
232        } else {
233          result.add(Pair.newPair(regionMap.get(groupKey), serverMap.get(groupKey)));
234        }
235      }
236      if (!fallbackRegions.isEmpty()) {
237        List<ServerName> candidates = null;
238        if (isFallbackEnabled()) {
239          candidates = getFallBackCandidates(servers);
240        }
241        candidates = (candidates == null || candidates.isEmpty()) ?
242          Lists.newArrayList(BOGUS_SERVER_NAME) : candidates;
243        result.add(Pair.newPair(fallbackRegions, candidates));
244      }
245      return result;
246    } catch(IOException e) {
247      throw new HBaseIOException("Failed to generate group assignments", e);
248    }
249  }

251  private List<ServerName> filterOfflineServers(RSGroupInfo RSGroupInfo,
252                                                List<ServerName> onlineServers) {
253    if (RSGroupInfo != null) {
254      return filterServers(RSGroupInfo.getServers(), onlineServers);
255    } else {
256      LOG.warn("RSGroup Information found to be null. Some regions might be unassigned.");
257      return Collections.emptyList();
258    }
259  }

261  /**
262   * Filter servers based on the online servers.
263   * <p/>
264   * servers is actually a TreeSet (see {@link org.apache.hadoop.hbase.rsgroup.RSGroupInfo}), having
265   * its contains()'s time complexity as O(logn), which is good enough.
266   * <p/>
267   * TODO: consider using HashSet to pursue O(1) for contains() throughout the calling chain if
268   * needed.
269   * @param servers the servers
270   * @param onlineServers List of servers which are online.
271   * @return the list
272   */
273  private List<ServerName> filterServers(Set<Address> servers, List<ServerName> onlineServers) {
274    ArrayList<ServerName> finalList = new ArrayList<>();
275    for (ServerName onlineServer : onlineServers) {
276      if (servers.contains(onlineServer.getAddress())) {
277        finalList.add(onlineServer);
278      }
279    }
280    return finalList;
281  }

283  private Pair<Map<TableName, Map<ServerName, List<RegionInfo>>>, List<RegionPlan>>
284      correctAssignments(Map<TableName, Map<ServerName, List<RegionInfo>>> existingAssignments)
285          throws IOException {
286    // To return
287    Map<TableName, Map<ServerName, List<RegionInfo>>> correctAssignments = new HashMap<>();
288    List<RegionPlan> regionPlansForMisplacedRegions = new ArrayList<>();
289    RSGroupInfo defaultInfo = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
290    for (Map.Entry<TableName, Map<ServerName, List<RegionInfo>>> assignments : existingAssignments
291        .entrySet()) {
292      TableName tableName = assignments.getKey();
293      Map<ServerName, List<RegionInfo>> clusterLoad = assignments.getValue();
294      RSGroupInfo targetRSGInfo = null;
295      Map<ServerName, List<RegionInfo>> correctServerRegion = new TreeMap<>();
296      try {
297        targetRSGInfo = RSGroupUtil.getRSGroupInfo(masterServices, rsGroupInfoManager, tableName)
298            .orElse(defaultInfo);
299      } catch (IOException exp) {
300        LOG.debug("RSGroup information null for region of table " + tableName, exp);
301      }
302      for (Map.Entry<ServerName, List<RegionInfo>> serverRegionMap : clusterLoad.entrySet()) {
303        ServerName currentHostServer = serverRegionMap.getKey();
304        List<RegionInfo> regionInfoList = serverRegionMap.getValue();
305        if (targetRSGInfo == null
306            || !targetRSGInfo.containsServer(currentHostServer.getAddress())) {
307          regionInfoList.forEach(regionInfo -> {
308            regionPlansForMisplacedRegions.add(new RegionPlan(regionInfo, currentHostServer, null));
309          });
310        } else {
311          correctServerRegion.put(currentHostServer, regionInfoList);
312        }
313      }
314      correctAssignments.put(tableName, correctServerRegion);
315    }
316
317    // Return correct assignments and region movement plan for mis-placed regions together
318    return new Pair<Map<TableName, Map<ServerName, List<RegionInfo>>>, List<RegionPlan>>(
319        correctAssignments, regionPlansForMisplacedRegions);
320  }

322  @Override
323  public void initialize() throws IOException {
324    if (rsGroupInfoManager == null) {
325      rsGroupInfoManager = masterServices.getRSGroupInfoManager();
326      if (rsGroupInfoManager == null) {
327        String msg = "RSGroupInfoManager hasn't been initialized";
328        LOG.error(msg);
329        throw new HBaseIOException(msg);
330      }
331      rsGroupInfoManager.start();
332    }
333
334    // Create the balancer
335    Configuration conf = masterServices.getConfiguration();
336    Class<? extends LoadBalancer> balancerClass;
337    @SuppressWarnings("deprecation")
338    String balancerClassName = conf.get(HBASE_RSGROUP_LOADBALANCER_CLASS);
339    if (balancerClassName == null) {
340      balancerClass = conf.getClass(HConstants.HBASE_MASTER_LOADBALANCER_CLASS,
341        LoadBalancerFactory.getDefaultLoadBalancerClass(), LoadBalancer.class);
342    } else {
343      try {
344        balancerClass = Class.forName(balancerClassName).asSubclass(LoadBalancer.class);
345      } catch (ClassNotFoundException e) {
346        throw new IOException(e);
347      }
348    }
349    this.provider = new MasterClusterInfoProvider(masterServices);
350    // avoid infinite nesting
351    if (getClass().isAssignableFrom(balancerClass)) {
352      balancerClass = LoadBalancerFactory.getDefaultLoadBalancerClass();
353    }
354    internalBalancer = ReflectionUtils.newInstance(balancerClass);
355    internalBalancer.setClusterInfoProvider(provider);
356    // special handling for favor node balancers
357    if (internalBalancer instanceof FavoredNodesPromoter) {
358      favoredNodesManager = new FavoredNodesManager(provider);
359      if (internalBalancer instanceof FavoredNodeLoadBalancer) {
360        ((FavoredNodeLoadBalancer) internalBalancer).setMasterServices(masterServices);
361      }
362      if (internalBalancer instanceof FavoredStochasticBalancer) {
363        ((FavoredStochasticBalancer) internalBalancer).setMasterServices(masterServices);
364      }
365    }
366    internalBalancer.initialize();
367    // init fallback groups
368    this.fallbackEnabled = conf.getBoolean(FALLBACK_GROUP_ENABLE_KEY, false);
369  }
370
371  public boolean isOnline() {
372    if (this.rsGroupInfoManager == null) {
373      return false;
374    }
375
376    return this.rsGroupInfoManager.isOnline();
377  }

391  @Override
392  public synchronized void onConfigurationChange(Configuration conf) {
393    boolean newFallbackEnabled = conf.getBoolean(FALLBACK_GROUP_ENABLE_KEY, false);
394    if (fallbackEnabled != newFallbackEnabled) {
395      LOG.info("Changing the value of {} from {} to {}", FALLBACK_GROUP_ENABLE_KEY,
396        fallbackEnabled, newFallbackEnabled);
397      fallbackEnabled = newFallbackEnabled;
398    }
399    provider.onConfigurationChange(conf);
400    internalBalancer.onConfigurationChange(conf);
401  }

430  private List<ServerName> getFallBackCandidates(List<ServerName> servers) {
431    List<ServerName> serverNames = null;
432    try {
433      RSGroupInfo info = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
434      serverNames = filterOfflineServers(info, servers);
435    } catch (IOException e) {
436      LOG.error("Failed to get default rsgroup info to fallback", e);
437    }
438    return serverNames == null || serverNames.isEmpty() ? servers : serverNames;
439  }
