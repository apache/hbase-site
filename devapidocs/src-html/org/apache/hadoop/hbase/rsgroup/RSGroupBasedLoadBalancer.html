058/**
059 * GroupBasedLoadBalancer, used when Region Server Grouping is configured (HBase-6721)
060 * It does region balance based on a table's group membership.
061 *
062 * Most assignment methods contain two exclusive code paths: Online - when the group
063 * table is online and Offline - when it is unavailable.
064 *
065 * During Offline, assignments are assigned based on cached information in zookeeper.
066 * If unavailable (ie bootstrap) then regions are assigned randomly.
067 *
068 * Once the GROUP table has been assigned, the balancer switches to Online and will then
069 * start providing appropriate assignments for user tables.
070 *
071 */
072@InterfaceAudience.Private
073public class RSGroupBasedLoadBalancer implements RSGroupableBalancer {
074  private static final Logger LOG = LoggerFactory.getLogger(RSGroupBasedLoadBalancer.class);
075

120  @Override
121  public List<RegionPlan> balanceCluster(Map<ServerName, List<RegionInfo>> clusterState)
122      throws HBaseIOException {
123    if (!isOnline()) {
124      throw new ConstraintException(RSGroupInfoManager.RSGROUP_TABLE_NAME +
125          " is not online, unable to perform balance");
126    }
127
128    // Calculate correct assignments and a list of RegionPlan for mis-placed regions
129    Pair<Map<ServerName,List<RegionInfo>>, List<RegionPlan>> correctedStateAndRegionPlans =
130        correctAssignments(clusterState);
131    Map<ServerName,List<RegionInfo>> correctedState = correctedStateAndRegionPlans.getFirst();
132    List<RegionPlan> regionPlans = correctedStateAndRegionPlans.getSecond();
133
134    // Add RegionPlan
135    // for the regions which have been placed according to the region server group assignment
136    // into the movement list
137    try {
138      // Record which region servers have been processedï¼Œso as to skip them after processed
139      HashSet<ServerName> processedServers = new HashSet<>();
140
141      // For each rsgroup
142      for (RSGroupInfo rsgroup : rsGroupInfoManager.listRSGroups()) {
143        Map<ServerName, List<RegionInfo>> groupClusterState = new HashMap<>();
144        Map<TableName, Map<ServerName, List<RegionInfo>>> groupClusterLoad = new HashMap<>();
145        for (ServerName server : clusterState.keySet()) { // for each region server
146          if (!processedServers.contains(server) // server is not processed yet
147              && rsgroup.containsServer(server.getAddress())) { // server belongs to this rsgroup
148            List<RegionInfo> regionsOnServer = correctedState.get(server);
149            groupClusterState.put(server, regionsOnServer);
150
151            processedServers.add(server);
152          }
153        }
154
155        groupClusterLoad.put(HConstants.ENSEMBLE_TABLE_NAME, groupClusterState);
156        this.internalBalancer.setClusterLoad(groupClusterLoad);
157        List<RegionPlan> groupPlans = this.internalBalancer
158            .balanceCluster(groupClusterState);
159        if (groupPlans != null) {
160          regionPlans.addAll(groupPlans);
161        }
162      }
163    } catch (IOException exp) {
164      LOG.warn("Exception while balancing cluster.", exp);
165      regionPlans.clear();
166    }
167
168    // Return the whole movement list
169    return regionPlans;
170  }

172  @Override
173  public Map<ServerName, List<RegionInfo>> roundRobinAssignment(
174      List<RegionInfo> regions, List<ServerName> servers) throws HBaseIOException {
175    Map<ServerName, List<RegionInfo>> assignments = Maps.newHashMap();
176    ListMultimap<String,RegionInfo> regionMap = ArrayListMultimap.create();
177    ListMultimap<String,ServerName> serverMap = ArrayListMultimap.create();
178    generateGroupMaps(regions, servers, regionMap, serverMap);
179    for(String groupKey : regionMap.keySet()) {
180      if (regionMap.get(groupKey).size() > 0) {
181        Map<ServerName, List<RegionInfo>> result =
182            this.internalBalancer.roundRobinAssignment(
183                regionMap.get(groupKey),
184                serverMap.get(groupKey));
185        if(result != null) {
186          if(result.containsKey(LoadBalancer.BOGUS_SERVER_NAME) &&
187              assignments.containsKey(LoadBalancer.BOGUS_SERVER_NAME)){
188            assignments.get(LoadBalancer.BOGUS_SERVER_NAME).addAll(
189              result.get(LoadBalancer.BOGUS_SERVER_NAME));
190          } else {
191            assignments.putAll(result);
192          }
193        }
194      }
195    }
196    return assignments;
197  }

199  @Override
200  public Map<ServerName, List<RegionInfo>> retainAssignment(
201      Map<RegionInfo, ServerName> regions, List<ServerName> servers) throws HBaseIOException {
202    try {
203      Map<ServerName, List<RegionInfo>> assignments = new TreeMap<>();
204      ListMultimap<String, RegionInfo> groupToRegion = ArrayListMultimap.create();
205      Set<RegionInfo> misplacedRegions = getMisplacedRegions(regions);
206      for (RegionInfo region : regions.keySet()) {
207        if (!misplacedRegions.contains(region)) {
208          String groupName = rsGroupInfoManager.getRSGroupOfTable(region.getTable());
209          if (groupName == null) {
210            LOG.info("Group not found for table " + region.getTable() + ", using default");
211            groupName = RSGroupInfo.DEFAULT_GROUP;
212          }
213          groupToRegion.put(groupName, region);
214        }
215      }
216      // Now the "groupToRegion" map has only the regions which have correct
217      // assignments.
218      for (String key : groupToRegion.keySet()) {
219        Map<RegionInfo, ServerName> currentAssignmentMap = new TreeMap<RegionInfo, ServerName>();
220        List<RegionInfo> regionList = groupToRegion.get(key);
221        RSGroupInfo info = rsGroupInfoManager.getRSGroup(key);
222        List<ServerName> candidateList = filterOfflineServers(info, servers);
223        for (RegionInfo region : regionList) {
224          currentAssignmentMap.put(region, regions.get(region));
225        }
226        if(candidateList.size() > 0) {
227          assignments.putAll(this.internalBalancer.retainAssignment(
228              currentAssignmentMap, candidateList));
229        }
230      }
231
232      for (RegionInfo region : misplacedRegions) {
233        String groupName = rsGroupInfoManager.getRSGroupOfTable(region.getTable());
234        if (groupName == null) {
235          LOG.info("Group not found for table " + region.getTable() + ", using default");
236          groupName = RSGroupInfo.DEFAULT_GROUP;
237        }
238        RSGroupInfo info = rsGroupInfoManager.getRSGroup(groupName);
239        List<ServerName> candidateList = filterOfflineServers(info, servers);
240        ServerName server = this.internalBalancer.randomAssignment(region,
241            candidateList);
242        if (server != null) {
243          if (!assignments.containsKey(server)) {
244            assignments.put(server, new ArrayList<>());
245          }
246          assignments.get(server).add(region);
247        } else {
248          //if not server is available assign to bogus so it ends up in RIT
249          if(!assignments.containsKey(LoadBalancer.BOGUS_SERVER_NAME)) {
250            assignments.put(LoadBalancer.BOGUS_SERVER_NAME, new ArrayList<>());
251          }
252          assignments.get(LoadBalancer.BOGUS_SERVER_NAME).add(region);
253        }
254      }
255      return assignments;
256    } catch (IOException e) {
257      throw new HBaseIOException("Failed to do online retain assignment", e);
258    }
259  }

261  @Override
262  public ServerName randomAssignment(RegionInfo region,
263      List<ServerName> servers) throws HBaseIOException {
264    ListMultimap<String,RegionInfo> regionMap = LinkedListMultimap.create();
265    ListMultimap<String,ServerName> serverMap = LinkedListMultimap.create();
266    generateGroupMaps(Lists.newArrayList(region), servers, regionMap, serverMap);
267    List<ServerName> filteredServers = serverMap.get(regionMap.keySet().iterator().next());
268    return this.internalBalancer.randomAssignment(region, filteredServers);
269  }
270
271  private void generateGroupMaps(
272    List<RegionInfo> regions,
273    List<ServerName> servers,
274    ListMultimap<String, RegionInfo> regionMap,
275    ListMultimap<String, ServerName> serverMap) throws HBaseIOException {
276    try {
277      for (RegionInfo region : regions) {
278        String groupName = rsGroupInfoManager.getRSGroupOfTable(region.getTable());
279        if (groupName == null) {
280          LOG.info("Group not found for table " + region.getTable() + ", using default");
281          groupName = RSGroupInfo.DEFAULT_GROUP;
282        }
283        regionMap.put(groupName, region);
284      }
285      for (String groupKey : regionMap.keySet()) {
286        RSGroupInfo info = rsGroupInfoManager.getRSGroup(groupKey);
287        serverMap.putAll(groupKey, filterOfflineServers(info, servers));
288        if(serverMap.get(groupKey).size() < 1) {
289          serverMap.put(groupKey, LoadBalancer.BOGUS_SERVER_NAME);
290        }
291      }
292    } catch(IOException e) {
293      throw new HBaseIOException("Failed to generate group maps", e);
294    }
295  }

297  private List<ServerName> filterOfflineServers(RSGroupInfo RSGroupInfo,
298                                                List<ServerName> onlineServers) {
299    if (RSGroupInfo != null) {
300      return filterServers(RSGroupInfo.getServers(), onlineServers);
301    } else {
302      LOG.warn("RSGroup Information found to be null. Some regions might be unassigned.");
303      return Collections.emptyList();
304    }
305  }

307  /**
308   * Filter servers based on the online servers.
309   *
310   * @param servers
311   *          the servers
312   * @param onlineServers
313   *          List of servers which are online.
314   * @return the list
315   */
316  private List<ServerName> filterServers(Set<Address> servers,
317                                         List<ServerName> onlineServers) {
318    /**
319     * servers is actually a TreeSet (see {@link org.apache.hadoop.hbase.rsgroup.RSGroupInfo}),
320     * having its contains()'s time complexity as O(logn), which is good enough.
321     * TODO: consider using HashSet to pursue O(1) for contains() throughout the calling chain
322     * if needed. */
323    ArrayList<ServerName> finalList = new ArrayList<>();
324    for (ServerName onlineServer : onlineServers) {
325      if (servers.contains(onlineServer.getAddress())) {
326        finalList.add(onlineServer);
327      }
328    }
329
330    return finalList;
331  }

333  @VisibleForTesting
334  public Set<RegionInfo> getMisplacedRegions(
335      Map<RegionInfo, ServerName> regions) throws IOException {
336    Set<RegionInfo> misplacedRegions = new HashSet<>();
337    for(Map.Entry<RegionInfo, ServerName> region : regions.entrySet()) {
338      RegionInfo regionInfo = region.getKey();
339      ServerName assignedServer = region.getValue();
340      String groupName = rsGroupInfoManager.getRSGroupOfTable(regionInfo.getTable());
341      if (groupName == null) {
342        LOG.info("Group not found for table " + regionInfo.getTable() + ", using default");
343        groupName = RSGroupInfo.DEFAULT_GROUP;
344      }
345      RSGroupInfo info = rsGroupInfoManager.getRSGroup(groupName);
346      if (assignedServer == null) {
347        LOG.debug("There is no assigned server for {}", region);
348        continue;
349      }
350      RSGroupInfo otherInfo = rsGroupInfoManager.getRSGroupOfServer(assignedServer.getAddress());
351      if (info == null && otherInfo == null) {
352        LOG.warn("Couldn't obtain rs group information for {} on {}", region, assignedServer);
353        continue;
354      }
355      if ((info == null || !info.containsServer(assignedServer.getAddress()))) {
356        LOG.debug("Found misplaced region: " + regionInfo.getRegionNameAsString() +
357            " on server: " + assignedServer +
358            " found in group: " +  otherInfo +
359            " outside of group: " + (info == null ? "UNKNOWN" : info.getName()));
360        misplacedRegions.add(regionInfo);
361      }
362    }
363    return misplacedRegions;
364  }

366  private Pair<Map<ServerName, List<RegionInfo>>, List<RegionPlan>> correctAssignments(
367      Map<ServerName, List<RegionInfo>> existingAssignments) throws HBaseIOException{
368    // To return
369    Map<ServerName, List<RegionInfo>> correctAssignments = new TreeMap<>();
370    List<RegionPlan> regionPlansForMisplacedRegions = new ArrayList<>();
371
372    for (Map.Entry<ServerName, List<RegionInfo>> assignments : existingAssignments.entrySet()){
373      ServerName currentHostServer = assignments.getKey();
374      correctAssignments.put(currentHostServer, new LinkedList<>());
375      List<RegionInfo> regions = assignments.getValue();
376      for (RegionInfo region : regions) {
377        RSGroupInfo targetRSGInfo = null;
378        try {
379          String groupName = rsGroupInfoManager.getRSGroupOfTable(region.getTable());
380          if (groupName == null) {
381            LOG.info("Group not found for table " + region.getTable() + ", using default");
382            groupName = RSGroupInfo.DEFAULT_GROUP;
383          }
384          targetRSGInfo = rsGroupInfoManager.getRSGroup(groupName);
385        } catch (IOException exp) {
386          LOG.debug("RSGroup information null for region of table " + region.getTable(),
387              exp);
388        }
389        if (targetRSGInfo == null ||
390            !targetRSGInfo.containsServer(currentHostServer.getAddress())) { // region is mis-placed
391          regionPlansForMisplacedRegions.add(new RegionPlan(region, currentHostServer, null));
392        } else { // region is placed as expected
393          correctAssignments.get(currentHostServer).add(region);
394        }
395      }
396    }
397
398    // Return correct assignments and region movement plan for mis-placed regions together
399    return new Pair<Map<ServerName, List<RegionInfo>>, List<RegionPlan>>(
400        correctAssignments, regionPlansForMisplacedRegions);
401  }

403  @Override
404  public void initialize() throws HBaseIOException {
405    try {
406      if (rsGroupInfoManager == null) {
407        List<RSGroupAdminEndpoint> cps =
408          masterServices.getMasterCoprocessorHost().findCoprocessors(RSGroupAdminEndpoint.class);
409        if (cps.size() != 1) {
410          String msg = "Expected one implementation of GroupAdminEndpoint but found " + cps.size();
411          LOG.error(msg);
412          throw new HBaseIOException(msg);
413        }
414        rsGroupInfoManager = cps.get(0).getGroupInfoManager();
415        if(rsGroupInfoManager == null){
416          String msg = "RSGroupInfoManager hasn't been initialized";
417          LOG.error(msg);
418          throw new HBaseIOException(msg);
419        }
420        rsGroupInfoManager.start();
421      }
422    } catch (IOException e) {
423      throw new HBaseIOException("Failed to initialize GroupInfoManagerImpl", e);
424    }
425
426    // Create the balancer
427    Class<? extends LoadBalancer> balancerKlass = config.getClass(HBASE_RSGROUP_LOADBALANCER_CLASS,
428        StochasticLoadBalancer.class, LoadBalancer.class);
429    internalBalancer = ReflectionUtils.newInstance(balancerKlass, config);
430    internalBalancer.setMasterServices(masterServices);
431    if(clusterStatus != null) {
432      internalBalancer.setClusterMetrics(clusterStatus);
433    }
434    internalBalancer.setConf(config);
435    internalBalancer.initialize();
436  }
