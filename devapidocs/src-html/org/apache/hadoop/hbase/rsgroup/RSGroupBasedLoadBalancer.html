059/**
060 * GroupBasedLoadBalancer, used when Region Server Grouping is configured (HBase-6721) It does
061 * region balance based on a table's group membership.
062 * <p/>
063 * Most assignment methods contain two exclusive code paths: Online - when the group table is online
064 * and Offline - when it is unavailable.
065 * <p/>
066 * During Offline, assignments are assigned based on cached information in zookeeper. If unavailable
067 * (ie bootstrap) then regions are assigned randomly.
068 * <p/>
069 * Once the GROUP table has been assigned, the balancer switches to Online and will then start
070 * providing appropriate assignments for user tables.
071 */
072@InterfaceAudience.Private
073public class RSGroupBasedLoadBalancer implements LoadBalancer {
074  private static final Logger LOG = LoggerFactory.getLogger(RSGroupBasedLoadBalancer.class);
075
076  private Configuration config;
077  private ClusterMetrics clusterStatus;
078  private MasterServices masterServices;
079  private FavoredNodesManager favoredNodesManager;
080  private volatile RSGroupInfoManager rsGroupInfoManager;
081  private LoadBalancer internalBalancer;

128  @Override
129  public List<RegionPlan> balanceCluster(
130      Map<TableName, Map<ServerName, List<RegionInfo>>> loadOfAllTable) throws IOException {
131    if (!isOnline()) {
132      throw new ConstraintException(
133          RSGroupInfoManager.class.getSimpleName() + " is not online, unable to perform balance");
134    }
135
136    // Calculate correct assignments and a list of RegionPlan for mis-placed regions
137    Pair<Map<TableName, Map<ServerName, List<RegionInfo>>>, List<RegionPlan>>
138      correctedStateAndRegionPlans = correctAssignments(loadOfAllTable);
139    Map<TableName, Map<ServerName, List<RegionInfo>>> correctedLoadOfAllTable =
140        correctedStateAndRegionPlans.getFirst();
141    List<RegionPlan> regionPlans = correctedStateAndRegionPlans.getSecond();
142    RSGroupInfo defaultInfo = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
143    // Add RegionPlan
144    // for the regions which have been placed according to the region server group assignment
145    // into the movement list
146    try {
147      // For each rsgroup
148      for (RSGroupInfo rsgroup : rsGroupInfoManager.listRSGroups()) {
149        Map<TableName, Map<ServerName, List<RegionInfo>>> loadOfTablesInGroup = new HashMap<>();
150        for (Map.Entry<TableName, Map<ServerName, List<RegionInfo>>> entry : correctedLoadOfAllTable
151            .entrySet()) {
152          TableName tableName = entry.getKey();
153          RSGroupInfo targetRSGInfo = RSGroupUtil
154              .getRSGroupInfo(masterServices, rsGroupInfoManager, tableName).orElse(defaultInfo);
155          if (targetRSGInfo.getName().equals(rsgroup.getName())) {
156            loadOfTablesInGroup.put(tableName, entry.getValue());
157          }
158        }
159        List<RegionPlan> groupPlans = null;
160        if (!loadOfTablesInGroup.isEmpty()) {
161          LOG.info("Start Generate Balance plan for group: " + rsgroup.getName());
162          groupPlans = this.internalBalancer.balanceCluster(loadOfTablesInGroup);
163        }
164        if (groupPlans != null) {
165          regionPlans.addAll(groupPlans);
166        }
167      }
168    } catch (IOException exp) {
169      LOG.warn("Exception while balancing cluster.", exp);
170      regionPlans.clear();
171    }
172
173    // Return the whole movement list
174    return regionPlans;
175  }

177  @Override
178  @NonNull
179  public Map<ServerName, List<RegionInfo>> roundRobinAssignment(List<RegionInfo> regions,
180      List<ServerName> servers) throws IOException {
181    Map<ServerName, List<RegionInfo>> assignments = Maps.newHashMap();
182    List<Pair<List<RegionInfo>, List<ServerName>>> pairs =
183        generateGroupAssignments(regions, servers);
184    for (Pair<List<RegionInfo>, List<ServerName>> pair : pairs) {
185      Map<ServerName, List<RegionInfo>> result =
186          this.internalBalancer.roundRobinAssignment(pair.getFirst(), pair.getSecond());
187      result.forEach((server, regionInfos) -> assignments
188          .computeIfAbsent(server, s -> Lists.newArrayList()).addAll(regionInfos));
189    }
190    return assignments;
191  }

193  @Override
194  @NonNull
195  public Map<ServerName, List<RegionInfo>> retainAssignment(Map<RegionInfo, ServerName> regions,
196      List<ServerName> servers) throws HBaseIOException {
197    try {
198      Map<ServerName, List<RegionInfo>> assignments = new TreeMap<>();
199      List<Pair<List<RegionInfo>, List<ServerName>>> pairs =
200          generateGroupAssignments(Lists.newArrayList(regions.keySet()), servers);
201      for (Pair<List<RegionInfo>, List<ServerName>> pair : pairs) {
202        List<RegionInfo> regionList = pair.getFirst();
203        Map<RegionInfo, ServerName> currentAssignmentMap = Maps.newTreeMap();
204        regionList.forEach(r -> currentAssignmentMap.put(r, regions.get(r)));
205        Map<ServerName, List<RegionInfo>> pairResult =
206            this.internalBalancer.retainAssignment(currentAssignmentMap, pair.getSecond());
207        pairResult.forEach((server, rs) -> assignments
208            .computeIfAbsent(server, s -> Lists.newArrayList()).addAll(rs));
209      }
210      return assignments;
211    } catch (IOException e) {
212      throw new HBaseIOException("Failed to do online retain assignment", e);
213    }
214  }

216  @Override
217  public ServerName randomAssignment(RegionInfo region,
218      List<ServerName> servers) throws IOException {
219    List<Pair<List<RegionInfo>, List<ServerName>>> pairs =
220        generateGroupAssignments(Lists.newArrayList(region), servers);
221    List<ServerName> filteredServers = pairs.iterator().next().getSecond();
222    return this.internalBalancer.randomAssignment(region, filteredServers);
223  }

225  private List<Pair<List<RegionInfo>, List<ServerName>>> generateGroupAssignments(
226      List<RegionInfo> regions, List<ServerName> servers) throws HBaseIOException {
227    try {
228      ListMultimap<String, RegionInfo> regionMap = ArrayListMultimap.create();
229      ListMultimap<String, ServerName> serverMap = ArrayListMultimap.create();
230      RSGroupInfo defaultInfo = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
231      for (RegionInfo region : regions) {
232        String groupName =
233            RSGroupUtil.getRSGroupInfo(masterServices, rsGroupInfoManager, region.getTable())
234                .orElse(defaultInfo).getName();
235        regionMap.put(groupName, region);
236      }
237      for (String groupKey : regionMap.keySet()) {
238        RSGroupInfo info = rsGroupInfoManager.getRSGroup(groupKey);
239        serverMap.putAll(groupKey, filterOfflineServers(info, servers));
240      }
241
242      List<Pair<List<RegionInfo>, List<ServerName>>> result = Lists.newArrayList();
243      List<RegionInfo> fallbackRegions = Lists.newArrayList();
244      for (String groupKey : regionMap.keySet()) {
245        if (serverMap.get(groupKey).isEmpty()) {
246          fallbackRegions.addAll(regionMap.get(groupKey));
247        } else {
248          result.add(Pair.newPair(regionMap.get(groupKey), serverMap.get(groupKey)));
249        }
250      }
251      if (!fallbackRegions.isEmpty()) {
252        List<ServerName> candidates = null;
253        if (isFallbackEnabled()) {
254          candidates = getFallBackCandidates(servers);
255        }
256        candidates = (candidates == null || candidates.isEmpty()) ?
257          Lists.newArrayList(BOGUS_SERVER_NAME) : candidates;
258        result.add(Pair.newPair(fallbackRegions, candidates));
259      }
260      return result;
261    } catch(IOException e) {
262      throw new HBaseIOException("Failed to generate group assignments", e);
263    }
264  }
265
266  private List<ServerName> filterOfflineServers(RSGroupInfo RSGroupInfo,
267                                                List<ServerName> onlineServers) {
268    if (RSGroupInfo != null) {
269      return filterServers(RSGroupInfo.getServers(), onlineServers);
270    } else {
271      LOG.warn("RSGroup Information found to be null. Some regions might be unassigned.");
272      return Collections.emptyList();
273    }
274  }

276  /**
277   * Filter servers based on the online servers.
278   * <p/>
279   * servers is actually a TreeSet (see {@link org.apache.hadoop.hbase.rsgroup.RSGroupInfo}), having
280   * its contains()'s time complexity as O(logn), which is good enough.
281   * <p/>
282   * TODO: consider using HashSet to pursue O(1) for contains() throughout the calling chain if
283   * needed.
284   * @param servers the servers
285   * @param onlineServers List of servers which are online.
286   * @return the list
287   */
288  private List<ServerName> filterServers(Set<Address> servers, List<ServerName> onlineServers) {
289    ArrayList<ServerName> finalList = new ArrayList<>();
290    for (ServerName onlineServer : onlineServers) {
291      if (servers.contains(onlineServer.getAddress())) {
292        finalList.add(onlineServer);
293      }
294    }
295    return finalList;
296  }

298  private Pair<Map<TableName, Map<ServerName, List<RegionInfo>>>, List<RegionPlan>>
299      correctAssignments(Map<TableName, Map<ServerName, List<RegionInfo>>> existingAssignments)
300          throws IOException {
301    // To return
302    Map<TableName, Map<ServerName, List<RegionInfo>>> correctAssignments = new HashMap<>();
303    List<RegionPlan> regionPlansForMisplacedRegions = new ArrayList<>();
304    RSGroupInfo defaultInfo = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
305    for (Map.Entry<TableName, Map<ServerName, List<RegionInfo>>> assignments : existingAssignments
306        .entrySet()) {
307      TableName tableName = assignments.getKey();
308      Map<ServerName, List<RegionInfo>> clusterLoad = assignments.getValue();
309      RSGroupInfo targetRSGInfo = null;
310      Map<ServerName, List<RegionInfo>> correctServerRegion = new TreeMap<>();
311      try {
312        targetRSGInfo = RSGroupUtil.getRSGroupInfo(masterServices, rsGroupInfoManager, tableName)
313            .orElse(defaultInfo);
314      } catch (IOException exp) {
315        LOG.debug("RSGroup information null for region of table " + tableName, exp);
316      }
317      for (Map.Entry<ServerName, List<RegionInfo>> serverRegionMap : clusterLoad.entrySet()) {
318        ServerName currentHostServer = serverRegionMap.getKey();
319        List<RegionInfo> regionInfoList = serverRegionMap.getValue();
320        if (targetRSGInfo == null
321            || !targetRSGInfo.containsServer(currentHostServer.getAddress())) {
322          regionInfoList.forEach(regionInfo -> {
323            regionPlansForMisplacedRegions.add(new RegionPlan(regionInfo, currentHostServer, null));
324          });
325        } else {
326          correctServerRegion.put(currentHostServer, regionInfoList);
327        }
328      }
329      correctAssignments.put(tableName, correctServerRegion);
330    }
331
332    // Return correct assignments and region movement plan for mis-placed regions together
333    return new Pair<Map<TableName, Map<ServerName, List<RegionInfo>>>, List<RegionPlan>>(
334        correctAssignments, regionPlansForMisplacedRegions);
335  }

337  @Override
338  public void initialize() throws IOException {
339    if (rsGroupInfoManager == null) {
340      rsGroupInfoManager = masterServices.getRSGroupInfoManager();
341      if (rsGroupInfoManager == null) {
342        String msg = "RSGroupInfoManager hasn't been initialized";
343        LOG.error(msg);
344        throw new HBaseIOException(msg);
345      }
346      rsGroupInfoManager.start();
347    }
348
349    // Create the balancer
350    Class<? extends LoadBalancer> balancerClass;
351    String balancerClassName = config.get(HBASE_RSGROUP_LOADBALANCER_CLASS);
352    if (balancerClassName == null) {
353      balancerClass = config.getClass(HConstants.HBASE_MASTER_LOADBALANCER_CLASS,
354        LoadBalancerFactory.getDefaultLoadBalancerClass(), LoadBalancer.class);
355    } else {
356      try {
357        balancerClass = Class.forName(balancerClassName).asSubclass(LoadBalancer.class);
358      } catch (ClassNotFoundException e) {
359        throw new IOException(e);
360      }
361    }
362    // avoid infinite nesting
363    if (getClass().isAssignableFrom(balancerClass)) {
364      balancerClass = LoadBalancerFactory.getDefaultLoadBalancerClass();
365    }
366    internalBalancer = ReflectionUtils.newInstance(balancerClass);
367    internalBalancer.setConf(config);
368    internalBalancer.setClusterInfoProvider(new MasterClusterInfoProvider(masterServices));
369    if(clusterStatus != null) {
370      internalBalancer.setClusterMetrics(clusterStatus);
371    }
372    // special handling for favor node balancers
373    if (internalBalancer instanceof FavoredNodesPromoter) {
374      favoredNodesManager = new FavoredNodesManager(masterServices);
375      if (internalBalancer instanceof FavoredNodeLoadBalancer) {
376        ((FavoredNodeLoadBalancer) internalBalancer).setMasterServices(masterServices);
377      }
378      if (internalBalancer instanceof FavoredStochasticBalancer) {
379        ((FavoredStochasticBalancer) internalBalancer).setMasterServices(masterServices);
380      }
381    }
382
383
384    internalBalancer.initialize();
385    // init fallback groups
386    this.fallbackEnabled = config.getBoolean(FALLBACK_GROUP_ENABLE_KEY, false);
387  }
388
389  public boolean isOnline() {
390    if (this.rsGroupInfoManager == null) {
391      return false;
392    }
393
394    return this.rsGroupInfoManager.isOnline();
395  }

409  @Override
410  public void onConfigurationChange(Configuration conf) {
411    this.config = conf;
412    boolean newFallbackEnabled = conf.getBoolean(FALLBACK_GROUP_ENABLE_KEY, false);
413    if (fallbackEnabled != newFallbackEnabled) {
414      LOG.info("Changing the value of {} from {} to {}", FALLBACK_GROUP_ENABLE_KEY,
415        fallbackEnabled, newFallbackEnabled);
416      fallbackEnabled = newFallbackEnabled;
417    }
418    internalBalancer.onConfigurationChange(conf);
419  }

451  private List<ServerName> getFallBackCandidates(List<ServerName> servers) {
452    List<ServerName> serverNames = null;
453    try {
454      RSGroupInfo info = rsGroupInfoManager.getRSGroup(RSGroupInfo.DEFAULT_GROUP);
455      serverNames = filterOfflineServers(info, servers);
456    } catch (IOException e) {
457      LOG.error("Failed to get default rsgroup info to fallback", e);
458    }
459    return serverNames == null || serverNames.isEmpty() ? servers : serverNames;
460  }
