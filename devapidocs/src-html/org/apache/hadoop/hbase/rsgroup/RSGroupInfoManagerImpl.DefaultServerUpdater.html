092 * use of an HBase table as the persistence store for the group information.
093 * It also makes use of zookeeper to store group information needed
094 * for bootstrapping during offline mode.

096public class RSGroupInfoManagerImpl implements RSGroupInfoManager, ServerListener {

148  @Override
149  public synchronized void addRSGroup(RSGroupInfo rsGroupInfo) throws IOException {
150    checkGroupName(rsGroupInfo.getName());
151    if (rsGroupMap.get(rsGroupInfo.getName()) != null ||
152        rsGroupInfo.getName().equals(rsGroupInfo.DEFAULT_GROUP)) {
153      throw new DoNotRetryIOException("Group already exists: "+ rsGroupInfo.getName());
154    }
155    Map<String, RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
156    newGroupMap.put(rsGroupInfo.getName(), rsGroupInfo);
157    flushConfig(newGroupMap);
158  }

160  @Override
161  public synchronized boolean moveServers(Set<HostAndPort> hostPorts, String srcGroup,
162                                          String dstGroup) throws IOException {
163    if (!rsGroupMap.containsKey(srcGroup)) {
164      throw new DoNotRetryIOException("Group "+srcGroup+" does not exist");
165    }
166    if (!rsGroupMap.containsKey(dstGroup)) {
167      throw new DoNotRetryIOException("Group "+dstGroup+" does not exist");
168    }
169
170    RSGroupInfo src = new RSGroupInfo(getRSGroup(srcGroup));
171    RSGroupInfo dst = new RSGroupInfo(getRSGroup(dstGroup));
172    boolean foundOne = false;
173    for(HostAndPort el: hostPorts) {
174      foundOne = src.removeServer(el) || foundOne;
175      dst.addServer(el);
176    }
177
178    Map<String,RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
179    newGroupMap.put(src.getName(), src);
180    newGroupMap.put(dst.getName(), dst);
181
182    flushConfig(newGroupMap);
183    return foundOne;
184  }

222  @Override
223  public synchronized void moveTables(
224      Set<TableName> tableNames, String groupName) throws IOException {
225    if (groupName != null && !rsGroupMap.containsKey(groupName)) {
226      throw new DoNotRetryIOException("Group "+groupName+" does not exist or is a special group");
227    }
228
229    Map<String,RSGroupInfo> newGroupMap = Maps.newHashMap(rsGroupMap);
230    for(TableName tableName: tableNames) {
231      if (tableMap.containsKey(tableName)) {
232        RSGroupInfo src = new RSGroupInfo(rsGroupMap.get(tableMap.get(tableName)));
233        src.removeTable(tableName);
234        newGroupMap.put(src.getName(), src);
235      }
236      if(groupName != null) {
237        RSGroupInfo dst = new RSGroupInfo(newGroupMap.get(groupName));
238        dst.addTable(tableName);
239        newGroupMap.put(dst.getName(), dst);
240      }
241    }
242
243    flushConfig(newGroupMap);
244  }

279  private synchronized void refresh(boolean forceOnline) throws IOException {
280    List<RSGroupInfo> groupList = new LinkedList<RSGroupInfo>();
281
282    // overwrite anything read from zk, group table is source of truth
283    // if online read from GROUP table
284    if (forceOnline || isOnline()) {
285      LOG.debug("Refreshing in Online mode.");
286      if (rsGroupTable == null) {
287        rsGroupTable = conn.getTable(RSGROUP_TABLE_NAME);
288      }
289      groupList.addAll(rsGroupSerDe.retrieveGroupList(rsGroupTable));
290    } else {
291      LOG.debug("Refershing in Offline mode.");
292      String groupBasePath = ZKUtil.joinZNode(watcher.baseZNode, rsGroupZNode);
293      groupList.addAll(rsGroupSerDe.retrieveGroupList(watcher, groupBasePath));
294    }
295
296    // refresh default group, prune
297    NavigableSet<TableName> orphanTables = new TreeSet<TableName>();
298    for(String entry: master.getTableDescriptors().getAll().keySet()) {
299      orphanTables.add(TableName.valueOf(entry));
300    }
301
302    List<TableName> specialTables;
303    if(!master.isInitialized()) {
304      specialTables = new ArrayList<TableName>();
305      specialTables.add(AccessControlLists.ACL_TABLE_NAME);
306      specialTables.add(TableName.META_TABLE_NAME);
307      specialTables.add(TableName.NAMESPACE_TABLE_NAME);
308      specialTables.add(RSGROUP_TABLE_NAME);
309    } else {
310      specialTables =
311          master.listTableNamesByNamespace(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR);
312    }
313
314    for(TableName table : specialTables) {
315      orphanTables.add(table);
316    }
317    for(RSGroupInfo group: groupList) {
318      if(!group.getName().equals(RSGroupInfo.DEFAULT_GROUP)) {
319        orphanTables.removeAll(group.getTables());
320      }
321    }
322
323    // This is added to the last of the list
324    // so it overwrites the default group loaded
325    // from region group table or zk
326    groupList.add(new RSGroupInfo(RSGroupInfo.DEFAULT_GROUP,
327        Sets.newHashSet(getDefaultServers()),
328        orphanTables));
329
330
331    // populate the data
332    HashMap<String, RSGroupInfo> newGroupMap = Maps.newHashMap();
333    HashMap<TableName, String> newTableMap = Maps.newHashMap();
334    for (RSGroupInfo group : groupList) {
335      newGroupMap.put(group.getName(), group);
336      for(TableName table: group.getTables()) {
337        newTableMap.put(table, group.getName());
338      }
339    }
340    rsGroupMap = Collections.unmodifiableMap(newGroupMap);
341    tableMap = Collections.unmodifiableMap(newTableMap);
342
343    prevRSGroups.clear();
344    prevRSGroups.addAll(rsGroupMap.keySet());
345  }

347  private synchronized Map<TableName,String> flushConfigTable(Map<String,RSGroupInfo> newGroupMap)
348      throws IOException {
349    Map<TableName,String> newTableMap = Maps.newHashMap();
350    List<Mutation> mutations = Lists.newArrayList();
351
352    // populate deletes
353    for(String groupName : prevRSGroups) {
354      if(!newGroupMap.containsKey(groupName)) {
355        Delete d = new Delete(Bytes.toBytes(groupName));
356        mutations.add(d);
357      }
358    }
359
360    // populate puts
361    for(RSGroupInfo RSGroupInfo : newGroupMap.values()) {
362      RSGroupProtos.RSGroupInfo proto = ProtobufUtil.toProtoGroupInfo(RSGroupInfo);
363      Put p = new Put(Bytes.toBytes(RSGroupInfo.getName()));
364      p.addColumn(META_FAMILY_BYTES,
365          META_QUALIFIER_BYTES,
366          proto.toByteArray());
367      mutations.add(p);
368      for(TableName entry: RSGroupInfo.getTables()) {
369        newTableMap.put(entry, RSGroupInfo.getName());
370      }
371    }
372
373    if(mutations.size() > 0) {
374      multiMutate(mutations);
375    }
376    return newTableMap;
377  }

379  private synchronized void flushConfig(Map<String, RSGroupInfo> newGroupMap) throws IOException {
380    Map<TableName, String> newTableMap;
381
382    // For offline mode persistence is still unavailable
383    // We're refreshing in-memory state but only for default servers
384    if (!isOnline()) {
385      Map<String, RSGroupInfo> m = Maps.newHashMap(rsGroupMap);
386      RSGroupInfo oldDefaultGroup = m.remove(RSGroupInfo.DEFAULT_GROUP);
387      RSGroupInfo newDefaultGroup = newGroupMap.remove(RSGroupInfo.DEFAULT_GROUP);
388      if (!m.equals(newGroupMap) ||
389          !oldDefaultGroup.getTables().equals(newDefaultGroup.getTables())) {
390        throw new IOException("Only default servers can be updated during offline mode");
391      }
392      newGroupMap.put(RSGroupInfo.DEFAULT_GROUP, newDefaultGroup);
393      rsGroupMap = newGroupMap;
394      return;
395    }
396
397    newTableMap = flushConfigTable(newGroupMap);
398
399    // make changes visible since it has been
400    // persisted in the source of truth
401    rsGroupMap = Collections.unmodifiableMap(newGroupMap);
402    tableMap = Collections.unmodifiableMap(newTableMap);
403
404
405    try {
406      String groupBasePath = ZKUtil.joinZNode(watcher.baseZNode, rsGroupZNode);
407      ZKUtil.createAndFailSilent(watcher, groupBasePath, ProtobufMagic.PB_MAGIC);
408
409      List<ZKUtil.ZKUtilOp> zkOps = new ArrayList<ZKUtil.ZKUtilOp>(newGroupMap.size());
410      for(String groupName : prevRSGroups) {
411        if(!newGroupMap.containsKey(groupName)) {
412          String znode = ZKUtil.joinZNode(groupBasePath, groupName);
413          zkOps.add(ZKUtil.ZKUtilOp.deleteNodeFailSilent(znode));
414        }
415      }
416
417
418      for(RSGroupInfo RSGroupInfo : newGroupMap.values()) {
419        String znode = ZKUtil.joinZNode(groupBasePath, RSGroupInfo.getName());
420        RSGroupProtos.RSGroupInfo proto = ProtobufUtil.toProtoGroupInfo(RSGroupInfo);
421        LOG.debug("Updating znode: "+znode);
422        ZKUtil.createAndFailSilent(watcher, znode);
423        zkOps.add(ZKUtil.ZKUtilOp.deleteNodeFailSilent(znode));
424        zkOps.add(ZKUtil.ZKUtilOp.createAndFailSilent(znode,
425            ProtobufUtil.prependPBMagic(proto.toByteArray())));
426      }
427      LOG.debug("Writing ZK GroupInfo count: " + zkOps.size());
428
429      ZKUtil.multiOrSequential(watcher, zkOps, false);
430    } catch (KeeperException e) {
431      LOG.error("Failed to write to rsGroupZNode", e);
432      master.abort("Failed to write to rsGroupZNode", e);
433      throw new IOException("Failed to write to rsGroupZNode",e);
434    }
435
436    prevRSGroups.clear();
437    prevRSGroups.addAll(newGroupMap.keySet());
438  }

440  private List<ServerName> getOnlineRS() throws IOException {
441    if (master != null) {
442      return master.getServerManager().getOnlineServersList();
443    }
444    try {
445      LOG.debug("Reading online RS from zookeeper");
446      List<ServerName> servers = new LinkedList<ServerName>();
447      for (String el: ZKUtil.listChildrenNoWatch(watcher, watcher.rsZNode)) {
448        servers.add(ServerName.parseServerName(el));
449      }
450      return servers;
451    } catch (KeeperException e) {
452      throw new IOException("Failed to retrieve server list from zookeeper", e);
453    }
454  }

456  private List<HostAndPort> getDefaultServers() throws IOException {
457    List<HostAndPort> defaultServers = new LinkedList<HostAndPort>();
458    for(ServerName server : getOnlineRS()) {
459      HostAndPort hostPort = HostAndPort.fromParts(server.getHostname(), server.getPort());
460      boolean found = false;
461      for(RSGroupInfo RSGroupInfo : rsGroupMap.values()) {
462        if(!RSGroupInfo.DEFAULT_GROUP.equals(RSGroupInfo.getName()) &&
463            RSGroupInfo.containsServer(hostPort)) {
464          found = true;
465          break;
466        }
467      }
468      if(!found) {
469        defaultServers.add(hostPort);
470      }
471    }
472    return defaultServers;
473  }

494  private static class DefaultServerUpdater extends Thread {
495    private static final Log LOG = LogFactory.getLog(DefaultServerUpdater.class);
496    private RSGroupInfoManagerImpl mgr;
497    private boolean hasChanged = false;
498
499    public DefaultServerUpdater(RSGroupInfoManagerImpl mgr) {
500      this.mgr = mgr;
501    }
502
503    @Override
504    public void run() {
505      List<HostAndPort> prevDefaultServers = new LinkedList<HostAndPort>();
506      while(!mgr.master.isAborted() || !mgr.master.isStopped()) {
507        try {
508          LOG.info("Updating default servers.");
509          List<HostAndPort> servers = mgr.getDefaultServers();
510          Collections.sort(servers, new Comparator<HostAndPort>() {
511            @Override
512            public int compare(HostAndPort o1, HostAndPort o2) {
513              int diff = o1.getHostText().compareTo(o2.getHostText());
514              if (diff != 0) {
515                return diff;
516              }
517              return o1.getPort() - o2.getPort();
518            }
519          });
520          if(!servers.equals(prevDefaultServers)) {
521            mgr.updateDefaultServers(Sets.<HostAndPort>newHashSet(servers));
522            prevDefaultServers = servers;
523            LOG.info("Updated with servers: "+servers.size());
524          }
525          try {
526            synchronized (this) {
527              if(!hasChanged) {
528                wait();
529              }
530              hasChanged = false;
531            }
532          } catch (InterruptedException e) {
533          }
534        } catch (IOException e) {
535          LOG.warn("Failed to update default servers", e);
536        }
537      }
538    }
539
540    public void serverChanged() {
541      synchronized (this) {
542        hasChanged = true;
543        this.notify();
544      }
545    }
546  }

549  private static class RSGroupStartupWorker extends Thread {
550    private static final Log LOG = LogFactory.getLog(RSGroupStartupWorker.class);
551
552    private Configuration conf;
553    private volatile boolean isOnline = false;
554    private MasterServices masterServices;
555    private RSGroupInfoManagerImpl groupInfoManager;
556    private ClusterConnection conn;
557
558    public RSGroupStartupWorker(RSGroupInfoManagerImpl groupInfoManager,
559                                MasterServices masterServices,
560                                ClusterConnection conn) {
561      this.conf = masterServices.getConfiguration();
562      this.masterServices = masterServices;
563      this.groupInfoManager = groupInfoManager;
564      this.conn = conn;
565      setName(RSGroupStartupWorker.class.getName()+"-"+masterServices.getServerName());
566      setDaemon(true);
567    }
568
569    @Override
570    public void run() {
571      if(waitForGroupTableOnline()) {
572        LOG.info("GroupBasedLoadBalancer is now online");
573      }
574    }
575
576    public boolean waitForGroupTableOnline() {
577      final List<HRegionInfo> foundRegions = new LinkedList<HRegionInfo>();
578      final List<HRegionInfo> assignedRegions = new LinkedList<HRegionInfo>();
579      final AtomicBoolean found = new AtomicBoolean(false);
580      final TableStateManager tsm = masterServices.getTableStateManager();
581      boolean createSent = false;
582      while (!found.get() && isMasterRunning()) {
583        foundRegions.clear();
584        assignedRegions.clear();
585        found.set(true);
586        try {
587          final Table nsTable = conn.getTable(TableName.NAMESPACE_TABLE_NAME);
588          final Table groupTable = conn.getTable(RSGROUP_TABLE_NAME);
589          boolean rootMetaFound =
590              masterServices.getMetaTableLocator().verifyMetaRegionLocation(
591                  conn,
592                  masterServices.getZooKeeper(),
593                  1);
594          final AtomicBoolean nsFound = new AtomicBoolean(false);
595          if (rootMetaFound) {
596
597            MetaTableAccessor.Visitor visitor = new DefaultVisitorBase() {
598              @Override
599              public boolean visitInternal(Result row) throws IOException {
600
601                HRegionInfo info = MetaTableAccessor.getHRegionInfo(row);
602                if (info != null) {
603                  Cell serverCell =
604                      row.getColumnLatestCell(HConstants.CATALOG_FAMILY,
605                          HConstants.SERVER_QUALIFIER);
606                  if (RSGROUP_TABLE_NAME.equals(info.getTable()) && serverCell != null) {
607                    ServerName sn =
608                        ServerName.parseVersionedServerName(CellUtil.cloneValue(serverCell));
609                    if (sn == null) {
610                      found.set(false);
611                    } else if (tsm.isTableState(RSGROUP_TABLE_NAME, TableState.State.ENABLED)) {
612                      try {
613                        ClientProtos.ClientService.BlockingInterface rs = conn.getClient(sn);
614                        ClientProtos.GetRequest request =
615                            RequestConverter.buildGetRequest(info.getRegionName(),
616                                new Get(ROW_KEY));
617                        rs.get(null, request);
618                        assignedRegions.add(info);
619                      } catch(Exception ex) {
620                        LOG.debug("Caught exception while verifying group region", ex);
621                      }
622                    }
623                    foundRegions.add(info);
624                  }
625                  if (TableName.NAMESPACE_TABLE_NAME.equals(info.getTable())) {
626                    Cell cell = row.getColumnLatestCell(HConstants.CATALOG_FAMILY,
627                        HConstants.SERVER_QUALIFIER);
628                    ServerName sn = null;
629                    if(cell != null) {
630                      sn = ServerName.parseVersionedServerName(CellUtil.cloneValue(cell));
631                    }
632                    if (tsm.isTableState(TableName.NAMESPACE_TABLE_NAME,
633                        TableState.State.ENABLED)) {
634                      try {
635                        ClientProtos.ClientService.BlockingInterface rs = conn.getClient(sn);
636                        ClientProtos.GetRequest request =
637                            RequestConverter.buildGetRequest(info.getRegionName(),
638                                new Get(ROW_KEY));
639                        rs.get(null, request);
640                        nsFound.set(true);
641                      } catch(Exception ex) {
642                        LOG.debug("Caught exception while verifying group region", ex);
643                      }
644                    }
645                  }
646                }
647                return true;
648              }
649            };
650            MetaTableAccessor.fullScanRegions(conn, visitor);
651            // if no regions in meta then we have to create the table
652            if (foundRegions.size() < 1 && rootMetaFound && !createSent && nsFound.get()) {
653              groupInfoManager.createGroupTable(masterServices);
654              createSent = true;
655            }
656            LOG.info("Group table: " + RSGROUP_TABLE_NAME + " isOnline: " + found.get()
657                + ", regionCount: " + foundRegions.size() + ", assignCount: "
658                + assignedRegions.size() + ", rootMetaFound: "+rootMetaFound);
659            found.set(found.get() && assignedRegions.size() == foundRegions.size()
660                && foundRegions.size() > 0);
661          } else {
662            LOG.info("Waiting for catalog tables to come online");
663            found.set(false);
664          }
665          if (found.get()) {
666            LOG.debug("With group table online, refreshing cached information.");
667            groupInfoManager.refresh(true);
668            isOnline = true;
669            //flush any inconsistencies between ZK and HTable
670            groupInfoManager.flushConfig(groupInfoManager.rsGroupMap);
671          }
672        } catch(Exception e) {
673          found.set(false);
674          LOG.warn("Failed to perform check", e);
675        }
676        try {
677          Thread.sleep(100);
678        } catch (InterruptedException e) {
679          LOG.info("Sleep interrupted", e);
680        }
681      }
682      return found.get();
683    }

685    public boolean isOnline() {
686      return isOnline;
687    }

694  private void createGroupTable(MasterServices masterServices) throws IOException {
695    HRegionInfo[] newRegions =
696        ModifyRegionUtils.createHRegionInfos(RSGROUP_TABLE_DESC, null);
697    ProcedurePrepareLatch latch = ProcedurePrepareLatch.createLatch();
698    masterServices.getMasterProcedureExecutor().submitProcedure(
699        new CreateTableProcedure(
700            masterServices.getMasterProcedureExecutor().getEnvironment(),
701            RSGROUP_TABLE_DESC,
702            newRegions,
703            latch),
704        HConstants.NO_NONCE,
705        HConstants.NO_NONCE);
706    latch.await();
707    // wait for region to be online
708    int tries = 600;
709    while(masterServices.getAssignmentManager().getRegionStates()
710        .getRegionServerOfRegion(newRegions[0]) == null && tries > 0) {
711      try {
712        Thread.sleep(100);
713      } catch (InterruptedException e) {
714        throw new IOException("Wait interrupted", e);
715      }
716      tries--;
717    }
718    if(tries <= 0) {
719      throw new IOException("Failed to create group table.");
720    }
721  }

723  private void multiMutate(List<Mutation> mutations)
724      throws IOException {
725    CoprocessorRpcChannel channel = rsGroupTable.coprocessorService(ROW_KEY);
726    MultiRowMutationProtos.MutateRowsRequest.Builder mmrBuilder
727      = MultiRowMutationProtos.MutateRowsRequest.newBuilder();
728    for (Mutation mutation : mutations) {
729      if (mutation instanceof Put) {
730        mmrBuilder.addMutationRequest(ProtobufUtil.toMutation(
731          ClientProtos.MutationProto.MutationType.PUT, mutation));
732      } else if (mutation instanceof Delete) {
733        mmrBuilder.addMutationRequest(ProtobufUtil.toMutation(
734          ClientProtos.MutationProto.MutationType.DELETE, mutation));
735      } else {
736        throw new DoNotRetryIOException("multiMutate doesn't support "
737          + mutation.getClass().getName());
738      }
739    }
740
741    MultiRowMutationProtos.MultiRowMutationService.BlockingInterface service =
742      MultiRowMutationProtos.MultiRowMutationService.newBlockingStub(channel);
743    try {
744      service.mutateRows(null, mmrBuilder.build());
745    } catch (ServiceException ex) {
746      ProtobufUtil.toIOException(ex);
747    }
748  }
