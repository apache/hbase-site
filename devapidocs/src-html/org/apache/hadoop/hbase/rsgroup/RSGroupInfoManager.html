019package org.apache.hadoop.hbase.rsgroup;
020
021import java.io.IOException;
022import java.util.List;
023import java.util.Set;
024
025import org.apache.hadoop.hbase.NamespaceDescriptor;
026import org.apache.hadoop.hbase.TableName;
027import org.apache.hadoop.hbase.net.Address;
028import org.apache.hadoop.hbase.util.Bytes;
029import org.apache.yetus.audience.InterfaceAudience;
030
031/**
032 * Interface used to manage RSGroupInfo storage. An implementation
033 * has the option to support offline mode.
034 * See {@link RSGroupBasedLoadBalancer}
035 */
036@InterfaceAudience.Private
037public interface RSGroupInfoManager {
038
039  String REASSIGN_WAIT_INTERVAL_KEY = "hbase.rsgroup.reassign.wait";
040  long DEFAULT_REASSIGN_WAIT_INTERVAL = 30 * 1000L;
041
042  //Assigned before user tables
043  TableName RSGROUP_TABLE_NAME =
044      TableName.valueOf(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR, "rsgroup");
045  String rsGroupZNode = "rsgroup";
046  byte[] META_FAMILY_BYTES = Bytes.toBytes("m");
047  byte[] META_QUALIFIER_BYTES = Bytes.toBytes("i");
048  byte[] ROW_KEY = {0};
049
050  /**
051   * Add given RSGroupInfo to existing list of group infos.
052   */
053  void addRSGroup(RSGroupInfo rsGroupInfo) throws IOException;
054
055  /**
056   * Remove a region server group.
057   */
058  void removeRSGroup(String groupName) throws IOException;
059
060  /**
061   * Move servers to a new group.
062   * @param servers list of servers, must be part of the same group
063   * @param srcGroup groupName being moved from
064   * @param dstGroup groupName being moved to
065   * @return Set of servers moved (May be a subset of {@code servers}).
066   */
067  Set<Address> moveServers(Set<Address> servers, String srcGroup, String dstGroup)
068      throws IOException;
069
070  /**
071   * Gets the group info of server.
072   */
073  RSGroupInfo getRSGroupOfServer(Address serverHostPort) throws IOException;
074
075  /**
076   * Gets {@code RSGroupInfo} for the given group name.
077   */
078  RSGroupInfo getRSGroup(String groupName) throws IOException;
079
080  /**
081   * Get the group membership of a table
082   */
083  String getRSGroupOfTable(TableName tableName) throws IOException;
084
085  /**
086   * Set the group membership of a set of tables
087   *
088   * @param tableNames set of tables to move
089   * @param groupName name of group of tables to move to
090   */
091  void moveTables(Set<TableName> tableNames, String groupName) throws IOException;
092
093  /**
094   * List the existing {@code RSGroupInfo}s.
095   */
096  List<RSGroupInfo> listRSGroups() throws IOException;
097
098  /**
099   * Refresh/reload the group information from the persistent store
100   */
101  void refresh() throws IOException;
102
103  /**
104   * Whether the manager is able to fully return group metadata
105   *
106   * @return whether the manager is in online mode
107   */
108  boolean isOnline();
109
110  /**
111   * Move servers and tables to a new group.
112   * @param servers list of servers, must be part of the same group
113   * @param tables set of tables to move
114   * @param srcGroup groupName being moved from
115   * @param dstGroup groupName being moved to
116   */
117  void moveServersAndTables(Set<Address> servers, Set<TableName> tables,
118      String srcGroup, String dstGroup) throws IOException;
119
120  /**
121   * Remove decommissioned servers from rsgroup
122   * @param servers set of servers to remove
123   */
124  void removeServers(Set<Address> servers) throws IOException;
125}
