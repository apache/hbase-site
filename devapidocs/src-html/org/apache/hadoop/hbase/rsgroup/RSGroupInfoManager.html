021package org.apache.hadoop.hbase.rsgroup;
022
023import com.google.common.net.HostAndPort;
024
025import java.io.IOException;
026import java.util.List;
027import java.util.Set;
028
029import org.apache.hadoop.hbase.NamespaceDescriptor;
030import org.apache.hadoop.hbase.TableName;
031import org.apache.hadoop.hbase.util.Bytes;
032
033/**
034 * Interface used to manage RSGroupInfo storage. An implementation
035 * has the option to support offline mode.
036 * See {@link RSGroupBasedLoadBalancer}
037 */
038public interface RSGroupInfoManager {
039  //Assigned before user tables
040  public static final TableName RSGROUP_TABLE_NAME =
041      TableName.valueOf(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR, "rsgroup");
042  public static final byte[] RSGROUP_TABLE_NAME_BYTES = RSGROUP_TABLE_NAME.toBytes();
043  public static final String rsGroupZNode = "rsgroup";
044  public static final byte[] META_FAMILY_BYTES = Bytes.toBytes("m");
045  public static final byte[] META_QUALIFIER_BYTES = Bytes.toBytes("i");
046  public static final byte[] ROW_KEY = {0};
047
048
049  /**
050   * Adds the group.
051   *
052   * @param rsGroupInfo the group name
053   * @throws java.io.IOException Signals that an I/O exception has occurred.
054   */
055  void addRSGroup(RSGroupInfo rsGroupInfo) throws IOException;
056
057  /**
058   * Remove a region server group.
059   *
060   * @param groupName the group name
061   * @throws java.io.IOException Signals that an I/O exception has occurred.
062   */
063  void removeRSGroup(String groupName) throws IOException;
064
065  /**
066   * move servers to a new group.
067   * @param hostPorts list of servers, must be part of the same group
068   * @param srcGroup groupName being moved from
069   * @param dstGroup groupName being moved to
070   * @return true if move was successful
071   * @throws java.io.IOException on move failure
072   */
073  boolean moveServers(Set<HostAndPort> hostPorts,
074                      String srcGroup, String dstGroup) throws IOException;
075
076  /**
077   * Gets the group info of server.
078   *
079   * @param hostPort the server
080   * @return An instance of RSGroupInfo
081   */
082  RSGroupInfo getRSGroupOfServer(HostAndPort hostPort) throws IOException;
083
084  /**
085   * Gets the group information.
086   *
087   * @param groupName the group name
088   * @return An instance of RSGroupInfo
089   */
090  RSGroupInfo getRSGroup(String groupName) throws IOException;
091
092  /**
093   * Get the group membership of a table
094   * @param tableName name of table to get group membership
095   * @return Group name of table
096   * @throws java.io.IOException on failure to retrive information
097   */
098  String getRSGroupOfTable(TableName tableName) throws IOException;
099
100  /**
101   * Set the group membership of a set of tables
102   *
103   * @param tableNames set of tables to move
104   * @param groupName name of group of tables to move to
105   * @throws java.io.IOException on failure to move
106   */
107  void moveTables(Set<TableName> tableNames, String groupName) throws IOException;
108
109  /**
110   * List the groups
111   *
112   * @return list of RSGroupInfo
113   * @throws java.io.IOException on failure
114   */
115  List<RSGroupInfo> listRSGroups() throws IOException;
116
117  /**
118   * Refresh/reload the group information from
119   * the persistent store
120   *
121   * @throws java.io.IOException on failure to refresh
122   */
123  void refresh() throws IOException;
124
125  /**
126   * Whether the manager is able to fully
127   * return group metadata
128   *
129   * @return whether the manager is in online mode
130   */
131  boolean isOnline();
132}
