066@InterfaceAudience.Private
067public class HttpDoAsClient {
068  private static final Logger LOG = LoggerFactory.getLogger(HttpDoAsClient.class);
069
070  static protected int port;
071  static protected String host;
072  CharsetDecoder decoder = null;
073  private static boolean secure = false;
074  static protected String doAsUser = null;
075  static protected String principal = null;
076  static protected String keyTab = null;
077
078  public static void main(String[] args) throws Exception {
079    if (args.length < 3 || args.length > 6) {
080      System.out.println("Invalid arguments!");
081      System.out.println(
082          "Usage: HttpDoAsClient host port doAsUserName [security=true] [principal] [keytab]");
083      System.exit(-1);
084    }
085
086    host = args[0];
087    port = Integer.parseInt(args[1]);
088    doAsUser = args[2];
089    if (args.length > 3) {
090      secure = Boolean.parseBoolean(args[3]);
091      if (args.length > 4) {
092        principal = args[4];
093        keyTab = args[5];
094        if (!new File(keyTab).exists()) {
095          System.err.printf("ERROR: KeyTab File %s not found %n", keyTab);
096          System.exit(-1);
097        }
098      } else {
099        principal = getSubject().getPrincipals().iterator().next().getName();
100      }
101    }
102
103    final HttpDoAsClient client = new HttpDoAsClient();
104    Subject.doAs(getSubject(),
105        new PrivilegedExceptionAction<Void>() {
106          @Override
107          public Void run() throws Exception {
108            client.run();
109            return null;
110          }
111        });
112  }
113
114  HttpDoAsClient() {
115    decoder = Charset.forName("UTF-8").newDecoder();
116  }
117
118  // Helper to translate byte[]'s to UTF8 strings
119  private String utf8(byte[] buf) {
120    try {
121      return decoder.decode(ByteBuffer.wrap(buf)).toString();
122    } catch (CharacterCodingException e) {
123      return "[INVALID UTF-8]";
124    }
125  }
126
127  // Helper to translate strings to UTF8 bytes
128  private byte[] bytes(String s) {
129    return Bytes.toBytes(s);
130  }
131
132  private void run() throws Exception {
133    TTransport transport = new TSocket(host, port);
134
135    transport.open();
136    String url = "http://" + host + ":" + port;
137    THttpClient httpClient = new THttpClient(url);
138    httpClient.open();
139    TProtocol protocol = new TBinaryProtocol(httpClient);
140    Hbase.Client client = new Hbase.Client(protocol);
141
142    byte[] t = bytes("demo_table");
143
144    //
145    // Scan all tables, look for the demo table and delete it.
146    //
147    System.out.println("scanning tables...");
148    for (ByteBuffer name : refresh(client, httpClient).getTableNames()) {
149      System.out.println("  found: " + utf8(name.array()));
150      if (utf8(name.array()).equals(utf8(t))) {
151        if (refresh(client, httpClient).isTableEnabled(name)) {
152          System.out.println("    disabling table: " + utf8(name.array()));
153          refresh(client, httpClient).disableTable(name);
154        }
155        System.out.println("    deleting table: " + utf8(name.array()));
156        refresh(client, httpClient).deleteTable(name);
157      }
158    }
159
160    //
161    // Create the demo table with two column families, entry: and unused:
162    //
163    ArrayList<ColumnDescriptor> columns = new ArrayList<>(2);
164    ColumnDescriptor col;
165    col = new ColumnDescriptor();
166    col.name = ByteBuffer.wrap(bytes("entry:"));
167    col.timeToLive = Integer.MAX_VALUE;
168    col.maxVersions = 10;
169    columns.add(col);
170    col = new ColumnDescriptor();
171    col.name = ByteBuffer.wrap(bytes("unused:"));
172    col.timeToLive = Integer.MAX_VALUE;
173    columns.add(col);
174
175    System.out.println("creating table: " + utf8(t));
176    try {
177
178      refresh(client, httpClient).createTable(ByteBuffer.wrap(t), columns);
179    } catch (AlreadyExists ae) {
180      System.out.println("WARN: " + ae.message);
181    }
182
183    System.out.println("column families in " + utf8(t) + ": ");
184    Map<ByteBuffer, ColumnDescriptor> columnMap = refresh(client, httpClient)
185        .getColumnDescriptors(ByteBuffer.wrap(t));
186    for (ColumnDescriptor col2 : columnMap.values()) {
187      System.out.println("  column: " + utf8(col2.name.array()) + ", maxVer: " + col2.maxVersions);
188    }
189
190    transport.close();
191    httpClient.close();
192  }
193
194  private Hbase.Client refresh(Hbase.Client client, THttpClient httpClient) {
195    httpClient.setCustomHeader("doAs", doAsUser);
196    if (secure) {
197      try {
198        httpClient.setCustomHeader("Authorization", generateTicket());
199      } catch (GSSException e) {
200        LOG.error("Kerberos authentication failed", e);
201      }
202    }
203    return client;
204  }
205
206  private String generateTicket() throws GSSException {
207    final GSSManager manager = GSSManager.getInstance();
208    // Oid for kerberos principal name
209    Oid krb5PrincipalOid = new Oid("1.2.840.113554.1.2.2.1");
210    Oid KERB_V5_OID = new Oid("1.2.840.113554.1.2.2");
211    final GSSName clientName = manager.createName(principal,
212        krb5PrincipalOid);
213    final GSSCredential clientCred = manager.createCredential(clientName,
214        8 * 3600,
215        KERB_V5_OID,
216        GSSCredential.INITIATE_ONLY);
217
218    final GSSName serverName = manager.createName(principal, krb5PrincipalOid);
219
220    final GSSContext context = manager.createContext(serverName,
221        KERB_V5_OID,
222        clientCred,
223        GSSContext.DEFAULT_LIFETIME);
224    context.requestMutualAuth(true);
225    context.requestConf(false);
226    context.requestInteg(true);
227
228    final byte[] outToken = context.initSecContext(new byte[0], 0, 0);
229    StringBuffer outputBuffer = new StringBuffer();
230    outputBuffer.append("Negotiate ");
231    outputBuffer.append(Bytes.toString(Base64.getEncoder().encode(outToken)));
232    System.out.print("Ticket is: " + outputBuffer);
233    return outputBuffer.toString();
234  }

236  private void printVersions(ByteBuffer row, List<TCell> versions) {
237    StringBuilder rowStr = new StringBuilder();
238    for (TCell cell : versions) {
239      rowStr.append(utf8(cell.value.array()));
240      rowStr.append("; ");
241    }
242    System.out.println("row: " + utf8(row.array()) + ", values: " + rowStr);
243  }
244
245  private void printRow(TRowResult rowResult) {
246    // copy values into a TreeMap to get them in sorted order
247    TreeMap<String, TCell> sorted = new TreeMap<>();
248    for (Map.Entry<ByteBuffer, TCell> column : rowResult.columns.entrySet()) {
249      sorted.put(utf8(column.getKey().array()), column.getValue());
250    }
251
252    StringBuilder rowStr = new StringBuilder();
253    for (SortedMap.Entry<String, TCell> entry : sorted.entrySet()) {
254      rowStr.append(entry.getKey());
255      rowStr.append(" => ");
256      rowStr.append(utf8(entry.getValue().value.array()));
257      rowStr.append("; ");
258    }
259    System.out.println("row: " + utf8(rowResult.row.array()) + ", cols: " + rowStr);
260  }

262  static Subject getSubject() throws Exception {
263    if (!secure) {
264      return new Subject();
265    }
266
267    /*
268     * To authenticate the DemoClient, kinit should be invoked ahead.
269     * Here we try to get the Kerberos credential from the ticket cache.
270     */
271    LoginContext context;
272
273    if (keyTab != null) {
274      // To authenticate the HttpDoAsClient using principal and keyTab
275      Set<Principal> principals = new HashSet<>();
276      principals.add(new KerberosPrincipal(principal));
277      Subject subject =
278          new Subject(false, principals, new HashSet<>(), new HashSet<>());
279
280      context = new LoginContext("", subject, null, new KerberosConfiguration(principal, keyTab));
281    } else {
282      /*
283       * To authenticate the HttpDoAsClient, kinit should be invoked ahead. Here we try to
284       * get the Kerberos credential from the ticket cache.
285       */
286      context = new LoginContext("", new Subject(), null, new KerberosConfiguration());
287    }
288    context.login();
289    return context.getSubject();
290  }

292  private static class KerberosConfiguration extends Configuration {
293    private String principal;
294    private String keyTab;
295
296    public KerberosConfiguration() {
297      // Empty constructor will have no principal or keyTab values
298    }
299
300    public KerberosConfiguration(String principal, String keyTab) {
301      this.principal = principal;
302      this.keyTab = keyTab;
303    }
304
305    @Override
306    public AppConfigurationEntry[] getAppConfigurationEntry(String name) {
307      Map<String, String> options = new HashMap<>();
308      if (principal != null && keyTab != null) {
309        options.put("principal", principal);
310        options.put("keyTab", keyTab);
311        options.put("useKeyTab", "true");
312        options.put("storeKey", "true");
313      } else {
314        options.put("useKeyTab", "false");
315        options.put("storeKey", "false");
316      }
317      options.put("doNotPrompt", "true");
318      options.put("useTicketCache", "true");
319      options.put("renewTGT", "true");
320      options.put("refreshKrb5Config", "true");
321      options.put("isInitiator", "true");
322      String ticketCache = System.getenv("KRB5CCNAME");
323      if (ticketCache != null) {
324        options.put("ticketCache", ticketCache);
325      }
326      options.put("debug", "true");
327
328      return new AppConfigurationEntry[] {
329        new AppConfigurationEntry("com.sun.security.auth.module.Krb5LoginModule",
330          AppConfigurationEntry.LoginModuleControlFlag.REQUIRED, options) };
331    }
332  }
