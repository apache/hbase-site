104  /**
105   * Returns a list of all the column families for a given Table.
106   */
107  byte[][] getAllColumns(Table table) throws IOException {
108    ColumnFamilyDescriptor[] cds = table.getDescriptor().getColumnFamilies();
109    byte[][] columns = new byte[cds.length][];
110    for (int i = 0; i < cds.length; i++) {
111      columns[i] = Bytes.add(cds[i].getName(), KeyValue.COLUMN_FAMILY_DELIM_ARRAY);
112    }
113    return columns;
114  }

117  /**
118   * Assigns a unique ID to the scanner and adds the mapping to an internal
119   * hash-map.
120   *
121   * @param scanner the {@link ResultScanner} to add
122   * @return integer scanner id
123   */
124  protected synchronized int addScanner(ResultScanner scanner, boolean sortColumns) {
125    int id = nextScannerId++;
126    ResultScannerWrapper resultScannerWrapper =
127        new ResultScannerWrapper(scanner, sortColumns);
128    scannerMap.put(id, resultScannerWrapper);
129    return id;
130  }
131
132  /**
133   * Returns the scanner associated with the specified ID.
134   *
135   * @param id the ID of the scanner to get
136   * @return a Scanner, or null if ID was invalid.
137   */
138  private synchronized ResultScannerWrapper getScanner(int id) {
139    return scannerMap.get(id);
140  }
141
142  /**
143   * Removes the scanner associated with the specified ID from the internal
144   * id-&gt;scanner hash-map.
145   *
146   * @param id the ID of the scanner to remove
147   * @return a Scanner, or null if ID was invalid.
148   */
149  private synchronized ResultScannerWrapper removeScanner(int id) {
150    return scannerMap.remove(id);
151  }

240  /**
241   * @return the list of regions in the given table, or an empty list if the table does not exist
242   */
243  @Override
244  public List<TRegionInfo> getTableRegions(ByteBuffer tableName) throws IOError {
245    try (RegionLocator locator = connectionCache.getRegionLocator(getBytes(tableName))) {
246      List<HRegionLocation> regionLocations = locator.getAllRegionLocations();
247      List<TRegionInfo> results = new ArrayList<>(regionLocations.size());
248      for (HRegionLocation regionLocation : regionLocations) {
249        RegionInfo info = regionLocation.getRegion();
250        ServerName serverName = regionLocation.getServerName();
251        TRegionInfo region = new TRegionInfo();
252        region.serverName = ByteBuffer.wrap(
253            Bytes.toBytes(serverName.getHostname()));
254        region.port = serverName.getPort();
255        region.startKey = ByteBuffer.wrap(info.getStartKey());
256        region.endKey = ByteBuffer.wrap(info.getEndKey());
257        region.id = info.getRegionId();
258        region.name = ByteBuffer.wrap(info.getRegionName());
259        region.version = HREGION_VERSION; // HRegion now not versioned, PB encoding used
260        results.add(region);
261      }
262      return results;
263    } catch (TableNotFoundException e) {
264      // Return empty list for non-existing table
265      return Collections.emptyList();
266    } catch (IOException e){
267      LOG.warn(e.getMessage(), e);
268      throw getIOError(e);
269    }
270  }

273  public List<TCell> get(
274      ByteBuffer tableName, ByteBuffer row, ByteBuffer column,
275      Map<ByteBuffer, ByteBuffer> attributes)
276      throws IOError {
277    byte [][] famAndQf = CellUtil.parseColumn(getBytes(column));
278    if (famAndQf.length == 1) {
279      return get(tableName, row, famAndQf[0], null, attributes);
280    }
281    if (famAndQf.length == 2) {
282      return get(tableName, row, famAndQf[0], famAndQf[1], attributes);
283    }
284    throw new IllegalArgumentException("Invalid familyAndQualifier provided.");
285  }

287  /**
288   * Note: this internal interface is slightly different from public APIs in regard to handling
289   * of the qualifier. Here we differ from the public Java API in that null != byte[0]. Rather,
290   * we respect qual == null as a request for the entire column family. The caller (
291   * {@link #get(ByteBuffer, ByteBuffer, ByteBuffer, Map)}) interface IS consistent in that the
292   * column is parse like normal.
293   */
294  protected List<TCell> get(ByteBuffer tableName,
295      ByteBuffer row,
296      byte[] family,
297      byte[] qualifier,
298      Map<ByteBuffer, ByteBuffer> attributes) throws IOError {
299    Table table = null;
300    try {
301      table = getTable(tableName);
302      Get get = new Get(getBytes(row));
303      addAttributes(get, attributes);
304      if (qualifier == null) {
305        get.addFamily(family);
306      } else {
307        get.addColumn(family, qualifier);
308      }
309      Result result = table.get(get);
310      return ThriftUtilities.cellFromHBase(result.rawCells());
311    } catch (IOException e) {
312      LOG.warn(e.getMessage(), e);
313      throw getIOError(e);
314    } finally {
315      closeTable(table);
316    }
317  }

437  @Override
438  public List<TRowResult> getRowWithColumnsTs(
439      ByteBuffer tableName, ByteBuffer row, List<ByteBuffer> columns,
440      long timestamp, Map<ByteBuffer, ByteBuffer> attributes) throws IOError {
441
442    Table table = null;
443    try {
444      table = getTable(tableName);
445      if (columns == null) {
446        Get get = new Get(getBytes(row));
447        addAttributes(get, attributes);
448        get.setTimeRange(0, timestamp);
449        Result result = table.get(get);
450        return ThriftUtilities.rowResultFromHBase(result);
451      }
452      Get get = new Get(getBytes(row));
453      addAttributes(get, attributes);
454      for(ByteBuffer column : columns) {
455        byte [][] famAndQf = CellUtil.parseColumn(getBytes(column));
456        if (famAndQf.length == 1) {
457          get.addFamily(famAndQf[0]);
458        } else {
459          get.addColumn(famAndQf[0], famAndQf[1]);
460        }
461      }
462      get.setTimeRange(0, timestamp);
463      Result result = table.get(get);
464      return ThriftUtilities.rowResultFromHBase(result);
465    } catch (IOException e) {
466      LOG.warn(e.getMessage(), e);
467      throw getIOError(e);
468    } finally{
469      closeTable(table);
470    }

502  @Override
503  public List<TRowResult> getRowsWithColumnsTs(ByteBuffer tableName,
504      List<ByteBuffer> rows,
505      List<ByteBuffer> columns, long timestamp,
506      Map<ByteBuffer, ByteBuffer> attributes) throws IOError {
507
508    Table table= null;
509    try {
510      List<Get> gets = new ArrayList<>(rows.size());
511      table = getTable(tableName);
512      if (metrics != null) {
513        metrics.incNumRowKeysInBatchGet(rows.size());
514      }
515      for (ByteBuffer row : rows) {
516        Get get = new Get(getBytes(row));
517        addAttributes(get, attributes);
518        if (columns != null) {
519
520          for(ByteBuffer column : columns) {
521            byte [][] famAndQf = CellUtil.parseColumn(getBytes(column));
522            if (famAndQf.length == 1) {
523              get.addFamily(famAndQf[0]);
524            } else {
525              get.addColumn(famAndQf[0], famAndQf[1]);
526            }
527          }
528        }
529        get.setTimeRange(0, timestamp);
530        gets.add(get);
531      }
532      Result[] result = table.get(gets);
533      return ThriftUtilities.rowResultFromHBase(result);
534    } catch (IOException e) {
535      LOG.warn(e.getMessage(), e);
536      throw getIOError(e);
537    } finally{
538      closeTable(table);
539    }

602  @Override
603  public void createTable(ByteBuffer in_tableName,
604      List<ColumnDescriptor> columnFamilies) throws IOError, IllegalArgument, AlreadyExists {
605    TableName tableName = getTableName(in_tableName);
606    try {
607      if (getAdmin().tableExists(tableName)) {
608        throw new AlreadyExists("table name already in use");
609      }
610      HTableDescriptor desc = new HTableDescriptor(tableName);
611      for (ColumnDescriptor col : columnFamilies) {
612        HColumnDescriptor colDesc = ThriftUtilities.colDescFromThrift(col);
613        desc.addFamily(colDesc);
614      }
615      getAdmin().createTable(desc);
616    } catch (IOException e) {
617      LOG.warn(e.getMessage(), e);
618      throw getIOError(e);
619    } catch (IllegalArgumentException e) {
620      LOG.warn(e.getMessage(), e);
621      throw new IllegalArgument(Throwables.getStackTraceAsString(e));
622    }
623  }
