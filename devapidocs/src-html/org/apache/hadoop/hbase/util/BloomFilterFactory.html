018package org.apache.hadoop.hbase.util;
019
020import java.io.DataInput;
021import java.io.IOException;
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.CellComparatorImpl;
024import org.apache.hadoop.hbase.io.hfile.CacheConfig;
025import org.apache.hadoop.hbase.io.hfile.CompoundBloomFilter;
026import org.apache.hadoop.hbase.io.hfile.CompoundBloomFilterBase;
027import org.apache.hadoop.hbase.io.hfile.CompoundBloomFilterWriter;
028import org.apache.hadoop.hbase.io.hfile.HFile;
029import org.apache.hadoop.hbase.regionserver.BloomType;
030import org.apache.yetus.audience.InterfaceAudience;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034/**
035 * Handles Bloom filter initialization based on configuration and serialized metadata in the reader
036 * and writer of {@link org.apache.hadoop.hbase.regionserver.HStoreFile}.
037 */
038@InterfaceAudience.Private
039public final class BloomFilterFactory {
040
041  private static final Logger LOG = LoggerFactory.getLogger(BloomFilterFactory.class.getName());
042
043  /** This class should not be instantiated. */
044  private BloomFilterFactory() {
045  }
046
047  /**
048   * Specifies the target error rate to use when selecting the number of keys per Bloom filter.
049   */
050  public static final String IO_STOREFILE_BLOOM_ERROR_RATE = "io.storefile.bloom.error.rate";
051
052  /**
053   * Maximum folding factor allowed. The Bloom filter will be shrunk by the factor of up to 2 **
054   * this times if we oversize it initially.
055   */
056  public static final String IO_STOREFILE_BLOOM_MAX_FOLD = "io.storefile.bloom.max.fold";
057
058  /**
059   * For default (single-block) Bloom filters this specifies the maximum number of keys.
060   */
061  public static final String IO_STOREFILE_BLOOM_MAX_KEYS = "io.storefile.bloom.max.keys";
062
063  /** Master switch to enable Bloom filters */
064  public static final String IO_STOREFILE_BLOOM_ENABLED = "io.storefile.bloom.enabled";
065
066  /** Master switch to enable Delete Family Bloom filters */
067  public static final String IO_STOREFILE_DELETEFAMILY_BLOOM_ENABLED =
068    "io.storefile.delete.family.bloom.enabled";
069
070  /**
071   * Target Bloom block size. Bloom filter blocks of approximately this size are interleaved with
072   * data blocks.
073   */
074  public static final String IO_STOREFILE_BLOOM_BLOCK_SIZE = "io.storefile.bloom.block.size";
075
076  /** Maximum number of times a Bloom filter can be "folded" if oversized */
077  private static final int MAX_ALLOWED_FOLD_FACTOR = 7;
078
079  /**
080   * Instantiates the correct Bloom filter class based on the version provided in the meta block
081   * data.
082   * @param meta   the byte array holding the Bloom filter's metadata, including version information
083   * @param reader the {@link HFile} reader to use to lazily load Bloom filter blocks
084   * @return an instance of the correct type of Bloom filter n
085   */
086  public static BloomFilter createFromMeta(DataInput meta, HFile.Reader reader)
087    throws IllegalArgumentException, IOException {
088    int version = meta.readInt();
089    switch (version) {
090      case CompoundBloomFilterBase.VERSION:
091        return new CompoundBloomFilter(meta, reader);
092
093      default:
094        throw new IllegalArgumentException("Bad bloom filter format version " + version);
095    }
096  }
097
098  /**
099   * Returns true if general Bloom (Row or RowCol) filters are enabled in the given configuration
100   */
101  public static boolean isGeneralBloomEnabled(Configuration conf) {
102    return conf.getBoolean(IO_STOREFILE_BLOOM_ENABLED, true);
103  }
104
105  /** Returns true if Delete Family Bloom filters are enabled in the given configuration */
106  public static boolean isDeleteFamilyBloomEnabled(Configuration conf) {
107    return conf.getBoolean(IO_STOREFILE_DELETEFAMILY_BLOOM_ENABLED, true);
108  }
109
110  /** Returns the Bloom filter error rate in the given configuration */
111  public static float getErrorRate(Configuration conf) {
112    return conf.getFloat(IO_STOREFILE_BLOOM_ERROR_RATE, (float) 0.01);
113  }
114
115  /** Returns the value for Bloom filter max fold in the given configuration */
116  public static int getMaxFold(Configuration conf) {
117    return conf.getInt(IO_STOREFILE_BLOOM_MAX_FOLD, MAX_ALLOWED_FOLD_FACTOR);
118  }
119
120  /** Returns the compound Bloom filter block size from the configuration */
121  public static int getBloomBlockSize(Configuration conf) {
122    return conf.getInt(IO_STOREFILE_BLOOM_BLOCK_SIZE, 128 * 1024);
123  }
124
125  /** Returns max key for the Bloom filter from the configuration */
126  public static int getMaxKeys(Configuration conf) {
127    return conf.getInt(IO_STOREFILE_BLOOM_MAX_KEYS, 128 * 1000 * 1000);
128  }
129
130  /**
131   * Creates a new general (Row or RowCol) Bloom filter at the time of
132   * {@link org.apache.hadoop.hbase.regionserver.HStoreFile} writing. nnn * @param maxKeys an
133   * estimate of the number of keys we expect to insert. Irrelevant if compound Bloom filters are
134   * enabled.
135   * @param writer the HFile writer
136   * @return the new Bloom filter, or null in case Bloom filters are disabled or when failed to
137   *         create one.
138   */
139  public static BloomFilterWriter createGeneralBloomAtWrite(Configuration conf,
140    CacheConfig cacheConf, BloomType bloomType, int maxKeys, HFile.Writer writer) {
141    if (!isGeneralBloomEnabled(conf)) {
142      LOG.trace("Bloom filters are disabled by configuration for " + writer.getPath()
143        + (conf == null ? " (configuration is null)" : ""));
144      return null;
145    } else if (bloomType == BloomType.NONE) {
146      LOG.trace("Bloom filter is turned off for the column family");
147      return null;
148    }
149
150    float err = getErrorRate(conf);
151
152    // In case of row/column Bloom filter lookups, each lookup is an OR if two
153    // separate lookups. Therefore, if each lookup's false positive rate is p,
154    // the resulting false positive rate is err = 1 - (1 - p)^2, and
155    // p = 1 - sqrt(1 - err).
156    if (bloomType == BloomType.ROWCOL) {
157      err = (float) (1 - Math.sqrt(1 - err));
158    }
159
160    int maxFold = conf.getInt(IO_STOREFILE_BLOOM_MAX_FOLD, MAX_ALLOWED_FOLD_FACTOR);
161
162    // Do we support compound bloom filters?
163    // In case of compound Bloom filters we ignore the maxKeys hint.
164    CompoundBloomFilterWriter bloomWriter = new CompoundBloomFilterWriter(getBloomBlockSize(conf),
165      err, Hash.getHashType(conf), maxFold, cacheConf.shouldCacheBloomsOnWrite(),
166      bloomType == BloomType.ROWCOL ? CellComparatorImpl.COMPARATOR : null, bloomType);
167    writer.addInlineBlockWriter(bloomWriter);
168    return bloomWriter;
169  }
170
171  /**
172   * Creates a new Delete Family Bloom filter at the time of
173   * {@link org.apache.hadoop.hbase.regionserver.HStoreFile} writing. nn * @param maxKeys an
174   * estimate of the number of keys we expect to insert. Irrelevant if compound Bloom filters are
175   * enabled.
176   * @param writer the HFile writer
177   * @return the new Bloom filter, or null in case Bloom filters are disabled or when failed to
178   *         create one.
179   */
180  public static BloomFilterWriter createDeleteBloomAtWrite(Configuration conf,
181    CacheConfig cacheConf, int maxKeys, HFile.Writer writer) {
182    if (!isDeleteFamilyBloomEnabled(conf)) {
183      LOG.info("Delete Bloom filters are disabled by configuration for " + writer.getPath()
184        + (conf == null ? " (configuration is null)" : ""));
185      return null;
186    }
187
188    float err = getErrorRate(conf);
189
190    int maxFold = getMaxFold(conf);
191    // In case of compound Bloom filters we ignore the maxKeys hint.
192    CompoundBloomFilterWriter bloomWriter =
193      new CompoundBloomFilterWriter(getBloomBlockSize(conf), err, Hash.getHashType(conf), maxFold,
194        cacheConf.shouldCacheBloomsOnWrite(), null, BloomType.ROW);
195    writer.addInlineBlockWriter(bloomWriter);
196    return bloomWriter;
197  }
198}
