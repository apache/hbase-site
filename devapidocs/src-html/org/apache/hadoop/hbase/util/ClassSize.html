141  /**
142   * MemoryLayout abstracts details about the JVM object layout. Default implementation is used in
143   * case Unsafe is not available.
144   */
145  private static class MemoryLayout {
146    int headerSize() {
147      return 2 * oopSize();
148    }
149
150    int arrayHeaderSize() {
151      return (int) align(3 * oopSize());
152    }
153
154    /**
155     * Return the size of an "ordinary object pointer". Either 4 or 8, depending on 32/64 bit,
156     * and CompressedOops
157     */
158    int oopSize() {
159      return is32BitJVM() ? 4 : 8;
160    }
161
162    /**
163     * Aligns a number to 8.
164     * @param num number to align to 8
165     * @return smallest number &gt;= input that is a multiple of 8
166     */
167    public long align(long num) {
168      //The 7 comes from that the alignSize is 8 which is the number of bytes
169      //stored and sent together
170      return  ((num + 7) >> 3) << 3;
171    }
172
173    long sizeOfByteArray(int len) {
174      return align(ARRAY + len);
175    }
176  }
177
178  /**
179   * UnsafeLayout uses Unsafe to guesstimate the object-layout related parameters like object header
180   * sizes and oop sizes
181   * See HBASE-15950.
182   */
183  private static class UnsafeLayout extends MemoryLayout {
184    @SuppressWarnings("unused")
185    private static final class HeaderSize {
186      private byte a;
187    }
188
189    public UnsafeLayout() {
190    }
191
192    @Override
193    int headerSize() {
194      try {
195        return (int) UnsafeAccess.theUnsafe.objectFieldOffset(
196          HeaderSize.class.getDeclaredField("a"));
197      } catch (NoSuchFieldException | SecurityException e) {
198        LOG.error(e.toString(), e);
199      }
200      return super.headerSize();
201    }
202
203    @Override
204    int arrayHeaderSize() {
205      return UnsafeAccess.theUnsafe.arrayBaseOffset(byte[].class);
206    }
207
208    @Override
209    @SuppressWarnings("static-access")
210    int oopSize() {
211      // Unsafe.addressSize() returns 8, even with CompressedOops. This is how many bytes each
212      // element is allocated in an Object[].
213      return UnsafeAccess.theUnsafe.ARRAY_OBJECT_INDEX_SCALE;
214    }
215
216    @Override
217    @SuppressWarnings("static-access")
218    long sizeOfByteArray(int len) {
219      return align(ARRAY + len * UnsafeAccess.theUnsafe.ARRAY_BYTE_INDEX_SCALE);
220    }
221  }

223  private static MemoryLayout getMemoryLayout() {
224    // Have a safeguard in case Unsafe estimate is wrong. This is static context, there is
225    // no configuration, so we look at System property.
226    String enabled = System.getProperty("hbase.memorylayout.use.unsafe");
227    if (UnsafeAvailChecker.isAvailable() && (enabled == null || Boolean.parseBoolean(enabled))) {
228      LOG.debug("Using Unsafe to estimate memory layout");
229      return new UnsafeLayout();
230    }
231    LOG.debug("Not using Unsafe to estimate memory layout");
232    return new MemoryLayout();
233  }

235  private static final MemoryLayout memoryLayout = getMemoryLayout();
236  private static final boolean USE_UNSAFE_LAYOUT = (memoryLayout instanceof UnsafeLayout);
237
238  public static boolean useUnsafeLayout() {
239    return USE_UNSAFE_LAYOUT;
240  }

242  /**
243   * Method for reading the arc settings and setting overheads according
244   * to 32-bit or 64-bit architecture.
245   */
246  static {
247    REFERENCE = memoryLayout.oopSize();
248
249    OBJECT = memoryLayout.headerSize();
250
251    ARRAY = memoryLayout.arrayHeaderSize();
252
253    ARRAYLIST = align(OBJECT + REFERENCE + (2 * Bytes.SIZEOF_INT)) + align(ARRAY);
254
255    LINKEDLIST = align(OBJECT + (2 * Bytes.SIZEOF_INT) + (2 * REFERENCE));
256
257    LINKEDLIST_ENTRY = align(OBJECT + (2 * REFERENCE));
258
259    //noinspection PointlessArithmeticExpression
260    BYTE_BUFFER = align(OBJECT + REFERENCE +
261        (5 * Bytes.SIZEOF_INT) +
262        (3 * Bytes.SIZEOF_BOOLEAN) + Bytes.SIZEOF_LONG) + align(ARRAY);
263
264    INTEGER = align(OBJECT + Bytes.SIZEOF_INT);
265
266    MAP_ENTRY = align(OBJECT + 5 * REFERENCE + Bytes.SIZEOF_BOOLEAN);
267
268    TREEMAP = align(OBJECT + (2 * Bytes.SIZEOF_INT) + 7 * REFERENCE);
269
270    // STRING is different size in jdk6 and jdk7. Just use what we estimate as size rather than
271    // have a conditional on whether jdk7.
272    STRING = (int) estimateBase(String.class, false);
273
274    // CONCURRENT_HASHMAP is different size in jdk6 and jdk7; it looks like its different between
275    // 23.6-b03 and 23.0-b21. Just use what we estimate as size rather than have a conditional on
276    // whether jdk7.
277    CONCURRENT_HASHMAP = (int) estimateBase(ConcurrentHashMap.class, false);
278
279    CONCURRENT_HASHMAP_ENTRY = align(REFERENCE + OBJECT + (3 * REFERENCE) +
280        (2 * Bytes.SIZEOF_INT));
281
282    CONCURRENT_HASHMAP_SEGMENT = align(REFERENCE + OBJECT +
283        (3 * Bytes.SIZEOF_INT) + Bytes.SIZEOF_FLOAT + ARRAY);
284
285    // The size changes from jdk7 to jdk8, estimate the size rather than use a conditional
286    CONCURRENT_SKIPLISTMAP = (int) estimateBase(ConcurrentSkipListMap.class, false);
287
288    // CellFlatMap object contains two integers, one boolean and one reference to object, so
289    // 2*INT + BOOLEAN + REFERENCE
290    CELL_FLAT_MAP = OBJECT + 2*Bytes.SIZEOF_INT + Bytes.SIZEOF_BOOLEAN + REFERENCE;
291
292    // CELL_ARRAY_MAP is the size of an instance of CellArrayMap class, which extends
293    // CellFlatMap class. CellArrayMap object containing a ref to an Array of Cells
294    CELL_ARRAY_MAP = align(CELL_FLAT_MAP + REFERENCE + ARRAY);
295
296    // CELL_CHUNK_MAP is the size of an instance of CellChunkMap class, which extends
297    // CellFlatMap class. CellChunkMap object containing a ref to an Array of Chunks
298    CELL_CHUNK_MAP = align(CELL_FLAT_MAP + REFERENCE + ARRAY);
299
300    CONCURRENT_SKIPLISTMAP_ENTRY = align(
301        align(OBJECT + (3 * REFERENCE)) + /* one node per entry */
302        align((OBJECT + (3 * REFERENCE))/2)); /* one index per two entries */
303
304    // REFERENCE in the CellArrayMap all the rest is counted in KeyValue.heapSize()
305    CELL_ARRAY_MAP_ENTRY = align(REFERENCE);
306
307    // The Cell Representation in the CellChunkMap, the Cell object size shouldn't be counted
308    // in KeyValue.heapSize()
309    // each cell-representation requires three integers for chunkID (reference to the ByteBuffer),
310    // offset and length, and one long for seqID
311    CELL_CHUNK_MAP_ENTRY = 3*Bytes.SIZEOF_INT + Bytes.SIZEOF_LONG;
312
313    REENTRANT_LOCK = align(OBJECT + (3 * REFERENCE));
314
315    ATOMIC_LONG = align(OBJECT + Bytes.SIZEOF_LONG);
316
317    ATOMIC_INTEGER = align(OBJECT + Bytes.SIZEOF_INT);
318
319    ATOMIC_BOOLEAN = align(OBJECT + Bytes.SIZEOF_BOOLEAN);
320
321    ATOMIC_REFERENCE = align(OBJECT + REFERENCE);
322
323    COPYONWRITE_ARRAYSET = align(OBJECT + REFERENCE);
324
325    COPYONWRITE_ARRAYLIST = align(OBJECT + (2 * REFERENCE) + ARRAY);
326
327    TIMERANGE = align(ClassSize.OBJECT + Bytes.SIZEOF_LONG * 2 + Bytes.SIZEOF_BOOLEAN);
328
329    SYNC_TIMERANGE_TRACKER = align(ClassSize.OBJECT + 2 * REFERENCE);
330
331    NON_SYNC_TIMERANGE_TRACKER = align(ClassSize.OBJECT + 2 * Bytes.SIZEOF_LONG);
332
333    CELL_SET = align(OBJECT + REFERENCE + Bytes.SIZEOF_INT);
334
335    STORE_SERVICES = align(OBJECT + REFERENCE + ATOMIC_LONG);
336  }

338  /**
339   * The estimate of the size of a class instance depends on whether the JVM
340   * uses 32 or 64 bit addresses, that is it depends on the size of an object
341   * reference. It is a linear function of the size of a reference, e.g.
342   * 24 + 5*r where r is the size of a reference (usually 4 or 8 bytes).
343   *
344   * This method returns the coefficients of the linear function, e.g. {24, 5}
345   * in the above example.
346   *
347   * @param cl A class whose instance size is to be estimated
348   * @param debug debug flag
349   * @return an array of 3 integers. The first integer is the size of the
350   * primitives, the second the number of arrays and the third the number of
351   * references.
352   */
353  @SuppressWarnings("unchecked")
354  private static int [] getSizeCoefficients(Class cl, boolean debug) {
355    int primitives = 0;
356    int arrays = 0;
357    int references = 0;
358    int index = 0;
359
360    for ( ; null != cl; cl = cl.getSuperclass()) {
361      Field[] field = cl.getDeclaredFields();
362      if (null != field) {
363        for (Field aField : field) {
364          if (Modifier.isStatic(aField.getModifiers())) continue;
365          Class fieldClass = aField.getType();
366          if (fieldClass.isArray()) {
367            arrays++;
368            references++;
369          } else if (!fieldClass.isPrimitive()) {
370            references++;
371          } else {// Is simple primitive
372            String name = fieldClass.getName();
373
374            if (name.equals("int") || name.equals("I"))
375              primitives += Bytes.SIZEOF_INT;
376            else if (name.equals("long") || name.equals("J"))
377              primitives += Bytes.SIZEOF_LONG;
378            else if (name.equals("boolean") || name.equals("Z"))
379              primitives += Bytes.SIZEOF_BOOLEAN;
380            else if (name.equals("short") || name.equals("S"))
381              primitives += Bytes.SIZEOF_SHORT;
382            else if (name.equals("byte") || name.equals("B"))
383              primitives += Bytes.SIZEOF_BYTE;
384            else if (name.equals("char") || name.equals("C"))
385              primitives += Bytes.SIZEOF_CHAR;
386            else if (name.equals("float") || name.equals("F"))
387              primitives += Bytes.SIZEOF_FLOAT;
388            else if (name.equals("double") || name.equals("D"))
389              primitives += Bytes.SIZEOF_DOUBLE;
390          }
391          if (debug) {
392            if (LOG.isDebugEnabled()) {
393              LOG.debug("" + index + " " + aField.getName() + " " + aField.getType());
394            }
395          }
396          index++;
397        }
398      }
399    }
400    return new int [] {primitives, arrays, references};
401  }

403  /**
404   * Estimate the static space taken up by a class instance given the
405   * coefficients returned by getSizeCoefficients.
406   *
407   * @param coeff the coefficients
408   *
409   * @param debug debug flag
410   * @return the size estimate, in bytes
411   */
412  private static long estimateBaseFromCoefficients(int [] coeff, boolean debug) {
413    long prealign_size = OBJECT + coeff[0] + coeff[2] * REFERENCE;
414
415    // Round up to a multiple of 8
416    long size = align(prealign_size) + align(coeff[1] * ARRAY);
417    if (debug) {
418      if (LOG.isDebugEnabled()) {
419        LOG.debug("Primitives=" + coeff[0] + ", arrays=" + coeff[1] +
420            ", references=" + coeff[2] + ", refSize " + REFERENCE +
421            ", size=" + size + ", prealign_size=" + prealign_size);
422      }
423    }
424    return size;
425  }

427  /**
428   * Estimate the static space taken up by the fields of a class. This includes
429   * the space taken up by by references (the pointer) but not by the referenced
430   * object. So the estimated size of an array field does not depend on the size
431   * of the array. Similarly the size of an object (reference) field does not
432   * depend on the object.
433   *
434   * @param cl class
435   * @param debug debug flag
436   * @return the size estimate in bytes.
437   */
438  @SuppressWarnings("unchecked")
439  public static long estimateBase(Class cl, boolean debug) {
440    return estimateBaseFromCoefficients( getSizeCoefficients(cl, debug), debug);
441  }

461  /**
462   * Determines if we are running in a 32-bit JVM. Some unit tests need to
463   * know this too.
464   */
465  public static boolean is32BitJVM() {
466    final String model = System.getProperty("sun.arch.data.model");
467    return model != null && model.equals("32");
468  }

470  /**
471   * Calculate the memory consumption (in byte) of a byte array,
472   * including the array header and the whole backing byte array.
473   *
474   * If the whole byte array is occupied (not shared with other objects), please use this function.
475   * If not, please use {@link #sizeOfByteArray(int)} instead.
476   *
477   * @param b the byte array
478   * @return the memory consumption (in byte) of the whole byte array
479   */
480  public static long sizeOf(byte[] b) {
481    return memoryLayout.sizeOfByteArray(b.length);
482  }

484  /**
485   * Calculate the memory consumption (in byte) of a part of a byte array,
486   * including the array header and the part of the backing byte array.
487   *
488   * This function is used when the byte array backs multiple objects.
489   * For example, in {@link org.apache.hadoop.hbase.KeyValue},
490   * multiple KeyValue objects share a same backing byte array ({@link org.apache.hadoop.hbase.KeyValue#bytes}).
491   * Also see {@link org.apache.hadoop.hbase.KeyValue#heapSize()}.
492   *
493   * @param len the length (in byte) used partially in the backing byte array
494   * @return the memory consumption (in byte) of the part of the byte array
495   */
496  public static long sizeOfByteArray(int len) {
497    return memoryLayout.sizeOfByteArray(len);
498  }
