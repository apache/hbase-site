059/**
060 * Implementation of {@link TableDescriptors} that reads descriptors from the
061 * passed filesystem.  It expects descriptors to be in a file in the
062 * {@link #TABLEINFO_DIR} subdir of the table's directory in FS.  Can be read-only
063 *  -- i.e. does not modify the filesystem or can be read and write.
064 *
065 * <p>Also has utility for keeping up the table descriptors tableinfo file.
066 * The table schema file is kept in the {@link #TABLEINFO_DIR} subdir
067 * of the table directory in the filesystem.
068 * It has a {@link #TABLEINFO_FILE_PREFIX} and then a suffix that is the
069 * edit sequenceid: e.g. <code>.tableinfo.0000000003</code>.  This sequenceid
070 * is always increasing.  It starts at zero.  The table schema file with the
071 * highest sequenceid has the most recent schema edit. Usually there is one file
072 * only, the most recent but there may be short periods where there are more
073 * than one file. Old files are eventually cleaned.  Presumption is that there
074 * will not be lots of concurrent clients making table schema edits.  If so,
075 * the below needs a bit of a reworking and perhaps some supporting api in hdfs.
076 */
077@InterfaceAudience.Private
078public class FSTableDescriptors implements TableDescriptors {
079  private static final Logger LOG = LoggerFactory.getLogger(FSTableDescriptors.class);
080  private final FileSystem fs;
081  private final Path rootdir;

083  private final boolean usecache;
084  private volatile boolean fsvisited;

125  public static TableDescriptor tryUpdateAndGetMetaTableDescriptor(Configuration conf,
126    FileSystem fs, Path rootdir) throws IOException {
127    // see if we already have meta descriptor on fs. Write one if not.
128    Optional<Pair<FileStatus, TableDescriptor>> opt = getTableDescriptorFromFs(fs,
129      CommonFSUtils.getTableDir(rootdir, TableName.META_TABLE_NAME), false);
130    if (opt.isPresent()) {
131      return opt.get().getSecond();
132    }
133    TableDescriptorBuilder builder = createMetaTableDescriptorBuilder(conf);
134    TableDescriptor td = StoreFileTrackerFactory.updateWithTrackerConfigs(conf, builder.build());
135    LOG.info("Creating new hbase:meta table descriptor {}", td);
136    TableName tableName = td.getTableName();
137    Path tableDir = CommonFSUtils.getTableDir(rootdir, tableName);
138    Path p = writeTableDescriptor(fs, td, tableDir, null);
139    if (p == null) {
140      throw new IOException("Failed update hbase:meta table descriptor");
141    }
142    LOG.info("Updated hbase:meta table descriptor to {}", p);
143    return td;
144  }

209  /**
210   * Get the current table descriptor for the given table, or null if none exists.
211   * <p/>
212   * Uses a local cache of the descriptor but still checks the filesystem on each call if
213   * {@link #fsvisited} is not {@code true}, i.e, we haven't done a full scan yet, to see if a newer
214   * file has been created since the cached one was read.
215   */
216  @Override
217  @Nullable
218  public TableDescriptor get(TableName tableName) {
219    invocations++;
220    if (usecache) {
221      // Look in cache of descriptors.
222      TableDescriptor cachedtdm = this.cache.get(tableName);
223      if (cachedtdm != null) {
224        cachehits++;
225        return cachedtdm;
226      }
227      // we do not need to go to fs any more
228      if (fsvisited) {
229        return null;
230      }
231    }
232    TableDescriptor tdmt = null;
233    try {
234      tdmt = getTableDescriptorFromFs(fs, getTableDir(tableName), fsreadonly).map(Pair::getSecond)
235        .orElse(null);
236    } catch (IOException ioe) {
237      LOG.debug("Exception during readTableDecriptor. Current table name = " + tableName, ioe);
238    }
239    // last HTD written wins
240    if (usecache && tdmt != null) {
241      this.cache.put(tableName, tdmt);
242    }
243
244    return tdmt;
245  }

247  /**
248   * Returns a map from table name to table descriptor for all tables.
249   */
250  @Override
251  public Map<String, TableDescriptor> getAll() throws IOException {
252    Map<String, TableDescriptor> tds = new TreeMap<>();
253    if (fsvisited) {
254      for (Map.Entry<TableName, TableDescriptor> entry: this.cache.entrySet()) {
255        tds.put(entry.getKey().getNameWithNamespaceInclAsString(), entry.getValue());
256      }
257    } else {
258      LOG.trace("Fetching table descriptors from the filesystem.");
259      boolean allvisited = usecache;
260      for (Path d : FSUtils.getTableDirs(fs, rootdir)) {
261        TableDescriptor htd = get(CommonFSUtils.getTableName(d));
262        if (htd == null) {
263          allvisited = false;
264        } else {
265          tds.put(htd.getTableName().getNameWithNamespaceInclAsString(), htd);
266        }
267      }
268      fsvisited = allvisited;
269    }
270    return tds;
271  }

273  /**
274    * Find descriptors by namespace.
275    * @see #get(org.apache.hadoop.hbase.TableName)
276    */
277  @Override
278  public Map<String, TableDescriptor> getByNamespace(String name) throws IOException {
279    Map<String, TableDescriptor> htds = new TreeMap<>();
280    List<Path> tableDirs =
281      FSUtils.getLocalTableDirs(fs, CommonFSUtils.getNamespaceDir(rootdir, name));
282    for (Path d : tableDirs) {
283      TableDescriptor htd = get(CommonFSUtils.getTableName(d));
284      if (htd == null) {
285        continue;
286      }
287      htds.put(CommonFSUtils.getTableName(d).getNameAsString(), htd);
288    }
289    return htds;
290  }

292  @Override
293  public void update(TableDescriptor td, boolean cacheOnly) throws IOException {
294    // TODO: in fact this method will only be called at master side, so fsreadonly and usecache will
295    // always be true. In general, we'd better have a ReadOnlyFSTableDesciptors for HRegionServer
296    // but now, HMaster extends HRegionServer, so unless making use of generic, we can not have
297    // different implementations for HMaster and HRegionServer. Revisit this when we make HMaster
298    // not extend HRegionServer in the future.
299    if (fsreadonly) {
300      throw new UnsupportedOperationException("Cannot add a table descriptor - in read only mode");
301    }
302    if (!cacheOnly) {
303      updateTableDescriptor(td);
304    }
305    if (usecache) {
306      this.cache.put(td.getTableName(), td);
307    }
308  }

310  @RestrictedApi(explanation = "Should only be called in tests or self", link = "",
311    allowedOnPath = ".*/src/test/.*|.*/FSTableDescriptors\\.java")
312  Path updateTableDescriptor(TableDescriptor td) throws IOException {
313    TableName tableName = td.getTableName();
314    Path tableDir = getTableDir(tableName);
315    Path p = writeTableDescriptor(fs, td, tableDir,
316      getTableDescriptorFromFs(fs, tableDir, fsreadonly).map(Pair::getFirst).orElse(null));
317    if (p == null) {
318      throw new IOException("Failed update");
319    }
320    LOG.info("Updated tableinfo=" + p);
321    return p;
322  }

324  /**
325   * Removes the table descriptor from the local cache and returns it.
326   * If not in read only mode, it also deletes the entire table directory(!)
327   * from the FileSystem.
328   */
329  @Override
330  public TableDescriptor remove(final TableName tablename) throws IOException {
331    if (fsreadonly) {
332      throw new NotImplementedException("Cannot remove a table descriptor - in read only mode");
333    }
334    Path tabledir = getTableDir(tablename);
335    if (this.fs.exists(tabledir)) {
336      if (!this.fs.delete(tabledir, true)) {
337        throw new IOException("Failed delete of " + tabledir.toString());
338      }
339    }
340    TableDescriptor descriptor = this.cache.remove(tablename);
341    return descriptor;
342  }

376  /**
377   * Width of the sequenceid that is a suffix on a tableinfo file.
378   */
379  static final int WIDTH_OF_SEQUENCE_ID = 10;
380
381  /**
382   * @param number Number to use as suffix.
383   * @return Returns zero-prefixed decimal version of passed number (Does absolute in case number is
384   *         negative).
385   */
386  private static String formatTableInfoSequenceId(final int number) {
387    byte [] b = new byte[WIDTH_OF_SEQUENCE_ID];
388    int d = Math.abs(number);
389    for (int i = b.length - 1; i >= 0; i--) {
390      b[i] = (byte)((d % 10) + '0');
391      d /= 10;
392    }
393    return Bytes.toString(b);
394  }

396  static final class SequenceIdAndFileLength {
397
398    final int sequenceId;
399
400    final int fileLength;
401
402    SequenceIdAndFileLength(int sequenceId, int fileLength) {
403      this.sequenceId = sequenceId;
404      this.fileLength = fileLength;
405    }
406  }

408  /**
409   * Returns the current sequence id and file length or 0 if none found.
410   * @param p Path to a <code>.tableinfo</code> file.
411   */
412  @RestrictedApi(explanation = "Should only be called in tests or self", link = "",
413    allowedOnPath = ".*/src/test/.*|.*/FSTableDescriptors\\.java")
414  static SequenceIdAndFileLength getTableInfoSequenceIdAndFileLength(Path p) {
415    String name = p.getName();
416    if (!name.startsWith(TABLEINFO_FILE_PREFIX)) {
417      throw new IllegalArgumentException("Invalid table descriptor file name: " + name);
418    }
419    int firstDot = name.indexOf('.', TABLEINFO_FILE_PREFIX.length());
420    if (firstDot < 0) {
421      // oldest style where we do not have both sequence id and file length
422      return new SequenceIdAndFileLength(0, 0);
423    }
424    int secondDot = name.indexOf('.', firstDot + 1);
425    if (secondDot < 0) {
426      // old stype where we do not have file length
427      int sequenceId = Integer.parseInt(name.substring(firstDot + 1));
428      return new SequenceIdAndFileLength(sequenceId, 0);
429    }
430    int sequenceId = Integer.parseInt(name.substring(firstDot + 1, secondDot));
431    int fileLength = Integer.parseInt(name.substring(secondDot + 1));
432    return new SequenceIdAndFileLength(sequenceId, fileLength);
433  }

472  private static Optional<Pair<FileStatus, TableDescriptor>> getTableDescriptorFromFs(FileSystem fs,
473    Path tableDir, boolean readonly) throws IOException {
474    Path tableInfoDir = new Path(tableDir, TABLEINFO_DIR);
475    FileStatus[] descFiles = CommonFSUtils.listStatus(fs, tableInfoDir, TABLEINFO_PATHFILTER);
476    if (descFiles == null || descFiles.length < 1) {
477      return Optional.empty();
478    }
479    Arrays.sort(descFiles, TABLEINFO_FILESTATUS_COMPARATOR);
480    int i = 0;
481    TableDescriptor td = null;
482    FileStatus descFile = null;
483    for (; i < descFiles.length; i++) {
484      descFile = descFiles[i];
485      Path file = descFile.getPath();
486      // get file length from file name if present
487      int fileLength = getTableInfoSequenceIdAndFileLength(file).fileLength;
488      byte[] content = new byte[fileLength > 0 ? fileLength : Ints.checkedCast(descFile.getLen())];
489      try (FSDataInputStream in = fs.open(file)) {
490        in.readFully(content);
491      } catch (EOFException e) {
492        LOG.info("Failed to load file {} due to EOF, it should be half written: {}", file,
493          e.toString());
494        if (!readonly) {
495          deleteMalformedFile(fs, file);
496        }
497        continue;
498      }
499      try {
500        td = TableDescriptorBuilder.parseFrom(content);
501        break;
502      } catch (DeserializationException e) {
503        LOG.info("Failed to parse file {} due to malformed protobuf message: {}", file,
504          e.toString());
505        if (!readonly) {
506          deleteMalformedFile(fs, file);
507        }
508      }
509    }
510    if (!readonly) {
511      // i + 1 to skip the one we load
512      for (i = i + 1; i < descFiles.length; i++) {
513        Path file = descFiles[i].getPath();
514        LOG.info("Delete old table descriptor file {}", file);
515        if (!fs.delete(file, false)) {
516          LOG.info("Failed to delete old table descriptor file {}", file);
517        }
518      }
519    }
520    return td != null ? Optional.of(Pair.newPair(descFile, td)) : Optional.empty();
521  }

523  /**
524   * Deletes files matching the table info file pattern within the given directory whose sequenceId
525   * is at most the given max sequenceId.
526   */
527  private static void deleteTableDescriptorFiles(FileSystem fs, Path dir, int maxSequenceId)
528    throws IOException {
529    FileStatus[] status = CommonFSUtils.listStatus(fs, dir, TABLEINFO_PATHFILTER);
530    for (FileStatus file : status) {
531      Path path = file.getPath();
532      int sequenceId = getTableInfoSequenceIdAndFileLength(path).sequenceId;
533      if (sequenceId <= maxSequenceId) {
534        boolean success = CommonFSUtils.delete(fs, path, false);
535        if (success) {
536          LOG.debug("Deleted {}", path);
537        } else {
538          LOG.error("Failed to delete table descriptor at {}", path);
539        }
540      }
541    }
542  }

544  /**
545   * Attempts to write a new table descriptor to the given table's directory. It begins at the
546   * currentSequenceId + 1 and tries 10 times to find a new sequence number not already in use.
547   * <p/>
548   * Removes the current descriptor file if passed in.
549   * @return Descriptor file or null if we failed write.
550   */
551  private static Path writeTableDescriptor(final FileSystem fs, final TableDescriptor td,
552    final Path tableDir, final FileStatus currentDescriptorFile) throws IOException {
553    // Here we will write to the final directory directly to avoid renaming as on OSS renaming is
554    // not atomic and has performance issue. The reason why we could do this is that, in the below
555    // code we will not overwrite existing files, we will write a new file instead. And when
556    // loading, we will skip the half written file, please see the code in getTableDescriptorFromFs
557    Path tableInfoDir = new Path(tableDir, TABLEINFO_DIR);
558
559    // In proc v2 we have table lock so typically, there will be no concurrent writes. Keep the
560    // retry logic here since we may still want to write the table descriptor from for example,
561    // HBCK2?
562    int currentSequenceId = currentDescriptorFile == null ? 0 :
563      getTableInfoSequenceIdAndFileLength(currentDescriptorFile.getPath()).sequenceId;
564
565    // Put arbitrary upperbound on how often we retry
566    int maxAttempts = 10;
567    int maxSequenceId = currentSequenceId + maxAttempts;
568    byte[] bytes = TableDescriptorBuilder.toByteArray(td);
569    for (int newSequenceId =
570      currentSequenceId + 1; newSequenceId <= maxSequenceId; newSequenceId++) {
571      String fileName = getTableInfoFileName(newSequenceId, bytes);
572      Path filePath = new Path(tableInfoDir, fileName);
573      try (FSDataOutputStream out = fs.create(filePath, false)) {
574        out.write(bytes);
575      } catch (FileAlreadyExistsException e) {
576        LOG.debug("{} exists; retrying up to {} times", filePath, maxAttempts, e);
577        continue;
578      } catch (IOException e) {
579        LOG.debug("Failed write {}; retrying up to {} times", filePath, maxAttempts, e);
580        continue;
581      }
582      deleteTableDescriptorFiles(fs, tableInfoDir, newSequenceId - 1);
583      return filePath;
584    }
585    return null;
586  }

602   * @return True if we successfully created file.
603   */
604  public boolean createTableDescriptor(TableDescriptor htd, boolean forceCreation)
605  throws IOException {
606    Path tableDir = getTableDir(htd.getTableName());
607    return createTableDescriptorForTableDirectory(tableDir, htd, forceCreation);
608  }

610  /**
611   * Create a new TableDescriptor in HDFS in the specified table directory. Happens when we create
612   * a new table during cluster start or in Clone and Create Table Procedures. Checks readOnly flag
613   * passed on construction.
614   * @param tableDir table directory under which we should write the file
615   * @param htd description of the table to write
616   * @param forceCreation if <tt>true</tt>,then even if previous table descriptor is present it will
617   *          be overwritten
618   * @return <tt>true</tt> if the we successfully created the file, <tt>false</tt> if the file
619   *         already exists and we weren't forcing the descriptor creation.
620   * @throws IOException if a filesystem error occurs
621   */
622  public boolean createTableDescriptorForTableDirectory(Path tableDir, TableDescriptor htd,
623      boolean forceCreation) throws IOException {
624    if (this.fsreadonly) {
625      throw new NotImplementedException("Cannot create a table descriptor - in read only mode");
626    }
627    return createTableDescriptorForTableDirectory(this.fs, tableDir, htd, forceCreation);
628  }

630  /**
631   * Create a new TableDescriptor in HDFS in the specified table directory. Happens when we create
632   * a new table snapshoting. Does not enforce read-only. That is for caller to determine.
633   * @param fs Filesystem to use.
634   * @param tableDir table directory under which we should write the file
635   * @param htd description of the table to write
636   * @param forceCreation if <tt>true</tt>,then even if previous table descriptor is present it will
637   *          be overwritten
638   * @return <tt>true</tt> if the we successfully created the file, <tt>false</tt> if the file
639   *         already exists and we weren't forcing the descriptor creation.
640   * @throws IOException if a filesystem error occurs
641   */
642  public static boolean createTableDescriptorForTableDirectory(FileSystem fs, Path tableDir,
643    TableDescriptor htd, boolean forceCreation) throws IOException {
644    Optional<Pair<FileStatus, TableDescriptor>> opt = getTableDescriptorFromFs(fs, tableDir, false);
645    if (opt.isPresent()) {
646      LOG.debug("Current path={}", opt.get().getFirst());
647      if (!forceCreation) {
648        if (htd.equals(opt.get().getSecond())) {
649          LOG.trace("TableInfo already exists.. Skipping creation");
650          return false;
651        }
652      }
653    }
654    return writeTableDescriptor(fs, htd, tableDir, opt.map(Pair::getFirst).orElse(null)) != null;
655  }
