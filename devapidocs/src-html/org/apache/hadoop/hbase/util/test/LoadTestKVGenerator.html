017package org.apache.hadoop.hbase.util.test;
018
019import java.util.Random;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.util.Bytes;
025import org.apache.hadoop.hbase.util.MD5Hash;
026
027/**
028 * A generator of random keys and values for load testing. Keys are generated
029 * by converting numeric indexes to strings and prefixing them with an MD5
030 * hash. Values are generated by selecting value size in the configured range
031 * and generating a pseudo-random sequence of bytes seeded by key, column
032 * qualifier, and value size.
033 */
034@InterfaceAudience.Private
035public class LoadTestKVGenerator {
036
037  private static final Log LOG = LogFactory.getLog(LoadTestKVGenerator.class);
038  private static int logLimit = 10;
039
040  /** A random number generator for determining value size */
041  private Random randomForValueSize = new Random();
042
043  private final int minValueSize;
044  private final int maxValueSize;
045
046  public LoadTestKVGenerator(int minValueSize, int maxValueSize) {
047    if (minValueSize <= 0 || maxValueSize <= 0) {
048      throw new IllegalArgumentException("Invalid min/max value sizes: " +
049          minValueSize + ", " + maxValueSize);
050    }
051    this.minValueSize = minValueSize;
052    this.maxValueSize = maxValueSize;
053  }
054
055  /**
056   * Verifies that the given byte array is the same as what would be generated
057   * for the given seed strings (row/cf/column/...). We are assuming that the
058   * value size is correct, and only verify the actual bytes. However, if the
059   * min/max value sizes are set sufficiently high, an accidental match should be
060   * extremely improbable.
061   */
062  public static boolean verify(byte[] value, byte[]... seedStrings) {
063    byte[] expectedData = getValueForRowColumn(value.length, seedStrings);
064    boolean equals = Bytes.equals(expectedData, value);
065    if (!equals && LOG.isDebugEnabled() && logLimit > 0) {
066      LOG.debug("verify failed, expected value: " + Bytes.toStringBinary(expectedData)
067        + " actual value: "+ Bytes.toStringBinary(value));
068      logLimit--; // this is not thread safe, but at worst we will have more logging
069    }
070    return equals;
071  }
072
073  /**
074   * Converts the given key to string, and prefixes it with the MD5 hash of
075   * the index's string representation.
076   */
077  public static String md5PrefixedKey(long key) {
078    String stringKey = Long.toString(key);
079    String md5hash = MD5Hash.getMD5AsHex(Bytes.toBytes(stringKey));
080
081    // flip the key to randomize
082    return md5hash + "-" + stringKey;
083  }
084
085  /**
086   * Generates a value for the given key index and column qualifier. Size is
087   * selected randomly in the configured range. The generated value depends
088   * only on the combination of the strings passed (key/cf/column/...) and the selected
089   * value size. This allows to verify the actual value bytes when reading, as done
090   * in {#verify(byte[], byte[]...)}
091   * This method is as thread-safe as Random class. It appears that the worst bug ever
092   * found with the latter is that multiple threads will get some duplicate values, which
093   * we don't care about.
094   */
095  public byte[] generateRandomSizeValue(byte[]... seedStrings) {
096    int dataSize = minValueSize;
097    if(minValueSize != maxValueSize) {
098      dataSize = minValueSize + randomForValueSize.nextInt(Math.abs(maxValueSize - minValueSize));
099    }
100    return getValueForRowColumn(dataSize, seedStrings);
101  }
102
103  /**
104   * Generates random bytes of the given size for the given row and column
105   * qualifier. The random seed is fully determined by these parameters.
106   */
107  private static byte[] getValueForRowColumn(int dataSize, byte[]... seedStrings) {
108    long seed = dataSize;
109    for (byte[] str : seedStrings) {
110      seed += Bytes.toString(str).hashCode();
111    }
112    Random seededRandom = new Random(seed);
113    byte[] randomBytes = new byte[dataSize];
114    seededRandom.nextBytes(randomBytes);
115    return randomBytes;
116  }
117
118}
