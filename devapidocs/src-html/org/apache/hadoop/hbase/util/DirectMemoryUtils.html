020package org.apache.hadoop.hbase.util;
021
022import java.lang.management.ManagementFactory;
023import java.lang.management.RuntimeMXBean;
024import java.lang.reflect.InvocationTargetException;
025import java.lang.reflect.Method;
026import java.nio.ByteBuffer;
027import java.util.List;
028import java.util.Locale;
029
030import javax.management.JMException;
031import javax.management.MBeanServer;
032import javax.management.MalformedObjectNameException;
033import javax.management.ObjectName;
034
035import org.apache.commons.logging.Log;
036import org.apache.commons.logging.LogFactory;
037import org.apache.hadoop.hbase.classification.InterfaceAudience;
038import org.apache.hadoop.hbase.classification.InterfaceStability;
039
040import com.google.common.base.Preconditions;
041
042/**
043 * Utilities for interacting with and monitoring DirectByteBuffer allocations.
044 */
045@InterfaceAudience.Private
046@InterfaceStability.Evolving
047public class DirectMemoryUtils {
048  private static final Log LOG = LogFactory.getLog(DirectMemoryUtils.class);
049  private static final String MEMORY_USED = "MemoryUsed";
050  private static final MBeanServer BEAN_SERVER;
051  private static final ObjectName NIO_DIRECT_POOL;
052  private static final boolean HAS_MEMORY_USED_ATTRIBUTE;
053
054  static {
055    // initialize singletons. Only maintain a reference to the MBeanServer if
056    // we're able to consume it -- hence convoluted logic.
057    ObjectName n = null;
058    MBeanServer s = null;
059    Object a = null;
060    try {
061      n = new ObjectName("java.nio:type=BufferPool,name=direct");
062    } catch (MalformedObjectNameException e) {
063      LOG.warn("Unable to initialize ObjectName for DirectByteBuffer allocations.");
064    } finally {
065      NIO_DIRECT_POOL = n;
066    }
067    if (NIO_DIRECT_POOL != null) {
068      s = ManagementFactory.getPlatformMBeanServer();
069    }
070    BEAN_SERVER = s;
071    if (BEAN_SERVER != null) {
072      try {
073        a = BEAN_SERVER.getAttribute(NIO_DIRECT_POOL, MEMORY_USED);
074      } catch (JMException e) {
075        LOG.debug("Failed to retrieve nio.BufferPool direct MemoryUsed attribute: " + e);
076      }
077    }
078    HAS_MEMORY_USED_ATTRIBUTE = a != null;
079  }
080
081  /**
082   * @return the setting of -XX:MaxDirectMemorySize as a long. Returns 0 if
083   *         -XX:MaxDirectMemorySize is not set.
084   */
085  public static long getDirectMemorySize() {
086    RuntimeMXBean runtimemxBean = ManagementFactory.getRuntimeMXBean();
087    List<String> arguments = runtimemxBean.getInputArguments();
088    long multiplier = 1; //for the byte case.
089    for (String s : arguments) {
090      if (s.contains("-XX:MaxDirectMemorySize=")) {
091        String memSize = s.toLowerCase(Locale.ROOT)
092            .replace("-xx:maxdirectmemorysize=", "").trim();
093
094        if (memSize.contains("k")) {
095          multiplier = 1024;
096        }
097
098        else if (memSize.contains("m")) {
099          multiplier = 1048576;
100        }
101
102        else if (memSize.contains("g")) {
103          multiplier = 1073741824;
104        }
105        memSize = memSize.replaceAll("[^\\d]", "");
106
107        long retValue = Long.parseLong(memSize);
108        return retValue * multiplier;
109      }
110    }
111    return 0;
112  }
113
114  /**
115   * @return the current amount of direct memory used.
116   */
117  public static long getDirectMemoryUsage() {
118    if (BEAN_SERVER == null || NIO_DIRECT_POOL == null || !HAS_MEMORY_USED_ATTRIBUTE) return 0;
119    try {
120      Long value = (Long) BEAN_SERVER.getAttribute(NIO_DIRECT_POOL, MEMORY_USED);
121      return value == null ? 0 : value;
122    } catch (JMException e) {
123      // should print further diagnostic information?
124      return 0;
125    }
126  }
127
128  /**
129   * DirectByteBuffers are garbage collected by using a phantom reference and a
130   * reference queue. Every once a while, the JVM checks the reference queue and
131   * cleans the DirectByteBuffers. However, as this doesn't happen
132   * immediately after discarding all references to a DirectByteBuffer, it's
133   * easy to OutOfMemoryError yourself using DirectByteBuffers. This function
134   * explicitly calls the Cleaner method of a DirectByteBuffer.
135   * 
136   * @param toBeDestroyed
137   *          The DirectByteBuffer that will be "cleaned". Utilizes reflection.
138   *          
139   */
140  public static void destroyDirectByteBuffer(ByteBuffer toBeDestroyed)
141      throws IllegalArgumentException, IllegalAccessException,
142      InvocationTargetException, SecurityException, NoSuchMethodException {
143
144    Preconditions.checkArgument(toBeDestroyed.isDirect(),
145        "toBeDestroyed isn't direct!");
146
147    Method cleanerMethod = toBeDestroyed.getClass().getMethod("cleaner");
148    cleanerMethod.setAccessible(true);
149    Object cleaner = cleanerMethod.invoke(toBeDestroyed);
150    Method cleanMethod = cleaner.getClass().getMethod("clean");
151    cleanMethod.setAccessible(true);
152    cleanMethod.invoke(cleaner);
153  }
154}
