036@InterfaceAudience.Private
037public class ReflectionUtils {
038  @SuppressWarnings("unchecked")
039  public static <T> T instantiateWithCustomCtor(String className,
040      Class<? >[] ctorArgTypes, Object[] ctorArgs) {
041    try {
042      Class<? extends T> resultType = (Class<? extends T>) Class.forName(className);
043      Constructor<? extends T> ctor = resultType.getDeclaredConstructor(ctorArgTypes);
044      return instantiate(className, ctor, ctorArgs);
045    } catch (ClassNotFoundException e) {
046      throw new UnsupportedOperationException(
047          "Unable to find " + className, e);
048    } catch (NoSuchMethodException e) {
049      throw new UnsupportedOperationException(
050          "Unable to find suitable constructor for class " + className, e);
051    }
052  }
053
054  private static <T> T instantiate(final String className, Constructor<T> ctor, Object[] ctorArgs) {
055    try {
056      ctor.setAccessible(true);
057      return ctor.newInstance(ctorArgs);
058    } catch (IllegalAccessException e) {
059      throw new UnsupportedOperationException(
060          "Unable to access specified class " + className, e);
061    } catch (InstantiationException e) {
062      throw new UnsupportedOperationException(
063          "Unable to instantiate specified class " + className, e);
064    } catch (InvocationTargetException e) {
065      throw new UnsupportedOperationException(
066          "Constructor threw an exception for " + className, e);
067    }
068  }
069
070  public static <T> T newInstance(Class<T> type, Object... params) {
071    return instantiate(type.getName(), findConstructor(type, params), params);
072  }
073
074  @SuppressWarnings("unchecked")
075  public static <T> Constructor<T> findConstructor(Class<T> type, Object... paramTypes) {
076    Constructor<T>[] constructors = (Constructor<T>[]) type.getDeclaredConstructors();
077    for (Constructor<T> ctor : constructors) {
078      Class<?>[] ctorParamTypes = ctor.getParameterTypes();
079      if (ctorParamTypes.length != paramTypes.length) {
080        continue;
081      }
082
083      boolean match = true;
084      for (int i = 0; i < ctorParamTypes.length && match; ++i) {
085        Class<?> paramType = paramTypes[i].getClass();
086        match = (!ctorParamTypes[i].isPrimitive()) ? ctorParamTypes[i].isAssignableFrom(paramType) :
087                  ((int.class.equals(ctorParamTypes[i]) && Integer.class.equals(paramType)) ||
088                   (long.class.equals(ctorParamTypes[i]) && Long.class.equals(paramType)) ||
089                   (double.class.equals(ctorParamTypes[i]) && Double.class.equals(paramType)) ||
090                   (char.class.equals(ctorParamTypes[i]) && Character.class.equals(paramType)) ||
091                   (short.class.equals(ctorParamTypes[i]) && Short.class.equals(paramType)) ||
092                   (boolean.class.equals(ctorParamTypes[i]) && Boolean.class.equals(paramType)) ||
093                   (byte.class.equals(ctorParamTypes[i]) && Byte.class.equals(paramType)));
094      }
095
096      if (match) {
097        return ctor;
098      }
099    }
100    throw new UnsupportedOperationException(
101      "Unable to find suitable constructor for class " + type.getName());
102  }

104  /* synchronized on ReflectionUtils.class */
105  private static long previousLogTime = 0;
106  private static final ThreadMXBean threadBean = ManagementFactory.getThreadMXBean();
107
108  /**
109   * Log the current thread stacks at INFO level.
110   * @param log the logger that logs the stack trace
111   * @param title a descriptive title for the call stacks
112   * @param minInterval the minimum time from the last
113   */
114  public static void logThreadInfo(Log log,
115                                   String title,
116                                   long minInterval) {
117    boolean dumpStack = false;
118    if (log.isInfoEnabled()) {
119      synchronized (ReflectionUtils.class) {
120        long now = System.currentTimeMillis();
121        if (now - previousLogTime >= minInterval * 1000) {
122          previousLogTime = now;
123          dumpStack = true;
124        }
125      }
126      if (dumpStack) {
127        try {
128          ByteArrayOutputStream buffer = new ByteArrayOutputStream();
129          printThreadInfo(new PrintStream(buffer, false, "UTF-8"), title);
130          log.info(buffer.toString(Charset.defaultCharset().name()));
131        } catch (UnsupportedEncodingException ignored) {
132          log.warn("Could not write thread info about '" + title +
133              "' due to a string encoding issue.");
134        }
135      }
136    }
137  }
138
139  /**
140   * Print all of the thread's information and stack traces.
141   *
142   * @param stream the stream to
143   * @param title a string title for the stack trace
144   */
145  private static void printThreadInfo(PrintStream stream,
146                                     String title) {
147    final int STACK_DEPTH = 20;
148    boolean contention = threadBean.isThreadContentionMonitoringEnabled();
149    long[] threadIds = threadBean.getAllThreadIds();
150    stream.println("Process Thread Dump: " + title);
151    stream.println(threadIds.length + " active threads");
152    for (long tid: threadIds) {
153      ThreadInfo info = threadBean.getThreadInfo(tid, STACK_DEPTH);
154      if (info == null) {
155        stream.println("  Inactive");
156        continue;
157      }
158      stream.println("Thread " +
159                     getTaskName(info.getThreadId(),
160                                 info.getThreadName()) + ":");
161      Thread.State state = info.getThreadState();
162      stream.println("  State: " + state);
163      stream.println("  Blocked count: " + info.getBlockedCount());
164      stream.println("  Waited count: " + info.getWaitedCount());
165      if (contention) {
166        stream.println("  Blocked time: " + info.getBlockedTime());
167        stream.println("  Waited time: " + info.getWaitedTime());
168      }
169      if (state == Thread.State.WAITING) {
170        stream.println("  Waiting on " + info.getLockName());
171      } else  if (state == Thread.State.BLOCKED) {
172        stream.println("  Blocked on " + info.getLockName());
173        stream.println("  Blocked by " +
174                       getTaskName(info.getLockOwnerId(),
175                                   info.getLockOwnerName()));
176      }
177      stream.println("  Stack:");
178      for (StackTraceElement frame: info.getStackTrace()) {
179        stream.println("    " + frame.toString());
180      }
181    }
182    stream.flush();
183  }
184
185  private static String getTaskName(long id, String name) {
186    if (name == null) {
187      return Long.toString(id);
188    }
189    return id + " (" + name + ")";
190  }

192  /**
193   * Get and invoke the target method from the given object with given parameters
194   * @param obj the object to get and invoke method from
195   * @param methodName the name of the method to invoke
196   * @param params the parameters for the method to invoke
197   * @return the return value of the method invocation
198   */
199  @NonNull
200  public static Object invokeMethod(Object obj, String methodName, Object... params) {
201    Method m;
202    try {
203      m = obj.getClass().getMethod(methodName, getParameterTypes(params));
204      m.setAccessible(true);
205      return m.invoke(obj, params);
206    } catch (NoSuchMethodException e) {
207      throw new UnsupportedOperationException("Cannot find specified method " + methodName, e);
208    } catch (IllegalAccessException e) {
209      throw new UnsupportedOperationException("Unable to access specified method " + methodName, e);
210    } catch (IllegalArgumentException e) {
211      throw new UnsupportedOperationException("Illegal arguments supplied for method " + methodName,
212          e);
213    } catch (InvocationTargetException e) {
214      throw new UnsupportedOperationException("Method threw an exception for " + methodName, e);
215    }
216  }
217
218  private static Class<?>[] getParameterTypes(Object[] params) {
219    Class<?>[] parameterTypes = new Class<?>[params.length];
220    for (int i = 0; i < params.length; i++) {
221      parameterTypes[i] = params[i].getClass();
222    }
223    return parameterTypes;
224  }
