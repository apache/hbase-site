018package org.apache.hadoop.hbase.util;
019
020import java.lang.ref.Reference;
021import java.lang.ref.SoftReference;
022
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.util.ObjectPool.ObjectFactory;
025
026/**
027 * A {@code SoftReference} based shared object pool.
028 * The objects are kept in soft references and
029 * associated with keys which are identified by the {@code equals} method.
030 * The objects are created by {@link ObjectFactory} on demand.
031 * The object creation is expected to be lightweight,
032 * and the objects may be excessively created and discarded.
033 * Thread safe.
034 */
035@InterfaceAudience.Private
036public class SoftObjectPool<K, V> extends ObjectPool<K, V> {
037
038  public SoftObjectPool(ObjectFactory<K, V> objectFactory) {
039    super(objectFactory);
040  }
041
042  public SoftObjectPool(ObjectFactory<K, V> objectFactory, int initialCapacity) {
043    super(objectFactory, initialCapacity);
044  }
045
046  public SoftObjectPool(ObjectFactory<K, V> objectFactory, int initialCapacity,
047      int concurrencyLevel) {
048    super(objectFactory, initialCapacity, concurrencyLevel);
049  }
050
051  @Override
052  public Reference<V> createReference(K key, V obj) {
053    return new SoftObjectReference(key, obj);
054  }
055
056  private class SoftObjectReference extends SoftReference<V> {
057    final K key;
058
059    SoftObjectReference(K key, V obj) {
060      super(obj, staleRefQueue);
061      this.key = key;
062    }
063  }
064
065  @Override
066  public K getReferenceKey(Reference<V> ref) {
067    return ((SoftObjectReference) ref).key;
068  }
069
070}
