018package org.apache.hadoop.hbase.util;
019
020import java.io.IOException;
021import java.lang.reflect.Constructor;
022import java.lang.reflect.InvocationTargetException;
023import java.util.ArrayList;
024import java.util.List;
025
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027import org.apache.hadoop.hbase.classification.InterfaceStability;
028import org.apache.hadoop.hbase.shaded.protobuf.generated.ErrorHandlingProtos.ForeignExceptionMessage;
029import org.apache.hadoop.hbase.shaded.protobuf.generated.ErrorHandlingProtos.GenericExceptionMessage;
030import org.apache.hadoop.hbase.shaded.protobuf.generated.ErrorHandlingProtos.StackTraceElementMessage;
031
032/**
033 * Helper to convert Exceptions and StackTraces from/to protobuf.
034 * (see ErrorHandling.proto for the internal of the proto messages)
035 */
036@InterfaceAudience.Private
037@InterfaceStability.Evolving
038public final class ForeignExceptionUtil {
039  private ForeignExceptionUtil() { }
040
041  public static Exception toException(final ForeignExceptionMessage eem) {
042    Exception re;
043    try {
044      re = createException(Exception.class, eem);
045    } catch (Throwable e) {
046      re = new Exception(eem.getGenericException().getMessage());
047    }
048    return setExceptionDetails(re, eem);
049  }
050
051  public static IOException toIOException(final ForeignExceptionMessage eem) {
052    IOException re;
053    try {
054      re = createException(IOException.class, eem);
055    } catch (Throwable e) {
056      re = new IOException(eem.getGenericException().getMessage());
057    }
058    return setExceptionDetails(re, eem);
059  }
060
061  private static <T extends Exception> T createException(final Class<T> clazz,
062      final ForeignExceptionMessage eem) throws ClassNotFoundException, NoSuchMethodException,
063        InstantiationException, IllegalAccessException, InvocationTargetException {
064    final GenericExceptionMessage gem = eem.getGenericException();
065    final Class<?> realClass = Class.forName(gem.getClassName());
066    final Class<? extends T> cls = realClass.asSubclass(clazz);
067    final Constructor<? extends T> cn = cls.getConstructor(String.class);
068    cn.setAccessible(true);
069    return cn.newInstance(gem.getMessage());
070  }
071
072  private static <T extends Exception> T setExceptionDetails(final T exception,
073      final ForeignExceptionMessage eem) {
074    final GenericExceptionMessage gem = eem.getGenericException();
075    final StackTraceElement[] trace = toStackTrace(gem.getTraceList());
076    exception.setStackTrace(trace);
077    return exception;
078  }
079
080  public static ForeignExceptionMessage toProtoForeignException(final Throwable t) {
081    return toProtoForeignException(null, t);
082  }
083
084  public static ForeignExceptionMessage toProtoForeignException(String source, Throwable t) {
085    GenericExceptionMessage.Builder gemBuilder = GenericExceptionMessage.newBuilder();
086    gemBuilder.setClassName(t.getClass().getName());
087    if (t.getMessage() != null) {
088      gemBuilder.setMessage(t.getMessage());
089    }
090    // set the stack trace, if there is one
091    List<StackTraceElementMessage> stack = toProtoStackTraceElement(t.getStackTrace());
092    if (stack != null) {
093      gemBuilder.addAllTrace(stack);
094    }
095    GenericExceptionMessage payload = gemBuilder.build();
096    ForeignExceptionMessage.Builder exception = ForeignExceptionMessage.newBuilder();
097    exception.setGenericException(payload);
098    if (source != null) exception.setSource(source);
099    return exception.build();
100  }
101
102  /**
103   * Convert a stack trace to list of {@link StackTraceElement}.
104   * @param trace the stack trace to convert to protobuf message
105   * @return <tt>null</tt> if the passed stack is <tt>null</tt>.
106   */
107  public static List<StackTraceElementMessage> toProtoStackTraceElement(StackTraceElement[] trace) {
108    // if there is no stack trace, ignore it and just return the message
109    if (trace == null) return null;
110    // build the stack trace for the message
111    List<StackTraceElementMessage> pbTrace = new ArrayList<>(trace.length);
112    for (StackTraceElement elem : trace) {
113      StackTraceElementMessage.Builder stackBuilder = StackTraceElementMessage.newBuilder();
114      stackBuilder.setDeclaringClass(elem.getClassName());
115      if (elem.getFileName() != null) {
116        stackBuilder.setFileName(elem.getFileName());
117      }
118      stackBuilder.setLineNumber(elem.getLineNumber());
119      stackBuilder.setMethodName(elem.getMethodName());
120      pbTrace.add(stackBuilder.build());
121    }
122    return pbTrace;
123  }
124
125  /**
126   * Unwind a serialized array of {@link StackTraceElementMessage}s to a
127   * {@link StackTraceElement}s.
128   * @param traceList list that was serialized
129   * @return the deserialized list or <tt>null</tt> if it couldn't be unwound (e.g. wasn't set on
130   *         the sender).
131   */
132  public static StackTraceElement[] toStackTrace(List<StackTraceElementMessage> traceList) {
133    if (traceList == null || traceList.isEmpty()) {
134      return new StackTraceElement[0]; // empty array
135    }
136    StackTraceElement[] trace = new StackTraceElement[traceList.size()];
137    for (int i = 0; i < traceList.size(); i++) {
138      StackTraceElementMessage elem = traceList.get(i);
139      trace[i] = new StackTraceElement(
140          elem.getDeclaringClass(), elem.getMethodName(),
141          elem.hasFileName() ? elem.getFileName() : null,
142          elem.getLineNumber());
143    }
144    return trace;
145  }
146}
