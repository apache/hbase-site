018package org.apache.hadoop.hbase.util;
019
020import java.io.IOException;
021import java.io.InterruptedIOException;
022import java.util.concurrent.CompletableFuture;
023import java.util.concurrent.CompletionException;
024import java.util.concurrent.ExecutionException;
025import java.util.concurrent.Executor;
026import java.util.concurrent.Future;
027import java.util.concurrent.TimeUnit;
028import java.util.concurrent.TimeoutException;
029import java.util.function.BiConsumer;
030import org.apache.hadoop.hbase.exceptions.TimeoutIOException;
031import org.apache.yetus.audience.InterfaceAudience;
032import org.slf4j.Logger;
033import org.slf4j.LoggerFactory;
034
035import org.apache.hbase.thirdparty.com.google.common.base.Throwables;
036
037/**
038 * Helper class for processing futures.
039 */
040@InterfaceAudience.Private
041public final class FutureUtils {
042
043  private static final Logger LOG = LoggerFactory.getLogger(FutureUtils.class);
044
045  private FutureUtils() {
046  }
047
048  /**
049   * This is method is used when you just want to add a listener to the given future. We will call
050   * {@link CompletableFuture#whenComplete(BiConsumer)} to register the {@code action} to the
051   * {@code future}. Ignoring the return value of a Future is considered as a bad practice as it may
052   * suppress exceptions thrown from the code that completes the future, and this method will catch
053   * all the exception thrown from the {@code action} to catch possible code bugs.
054   * <p/>
055   * And the error phone check will always report FutureReturnValueIgnored because every method in
056   * the {@link CompletableFuture} class will return a new {@link CompletableFuture}, so you always
057   * have one future that has not been checked. So we introduce this method and add a suppress
058   * warnings annotation here.
059   */
060  @SuppressWarnings("FutureReturnValueIgnored")
061  public static <T> void addListener(CompletableFuture<T> future,
062      BiConsumer<? super T, ? super Throwable> action) {
063    future.whenComplete((resp, error) -> {
064      try {
065        // See this post on stack overflow(shorten since the url is too long),
066        // https://s.apache.org/completionexception
067        // For a chain of CompleableFuture, only the first child CompletableFuture can get the
068        // original exception, others will get a CompletionException, which wraps the original
069        // exception. So here we unwrap it before passing it to the callback action.
070        action.accept(resp, unwrapCompletionException(error));
071      } catch (Throwable t) {
072        LOG.error("Unexpected error caught when processing CompletableFuture", t);
073      }
074    });
075  }
076
077  /**
078   * Almost the same with {@link #addListener(CompletableFuture, BiConsumer)} method above, the only
079   * exception is that we will call
080   * {@link CompletableFuture#whenCompleteAsync(BiConsumer, Executor)}.
081   * @see #addListener(CompletableFuture, BiConsumer)
082   */
083  @SuppressWarnings("FutureReturnValueIgnored")
084  public static <T> void addListener(CompletableFuture<T> future,
085      BiConsumer<? super T, ? super Throwable> action, Executor executor) {
086    future.whenCompleteAsync((resp, error) -> {
087      try {
088        action.accept(resp, unwrapCompletionException(error));
089      } catch (Throwable t) {
090        LOG.error("Unexpected error caught when processing CompletableFuture", t);
091      }
092    }, executor);
093  }
094
095  /**
096   * Return a {@link CompletableFuture} which is same with the given {@code future}, but execute all
097   * the callbacks in the given {@code executor}.
098   */
099  public static <T> CompletableFuture<T> wrapFuture(CompletableFuture<T> future,
100      Executor executor) {
101    CompletableFuture<T> wrappedFuture = new CompletableFuture<>();
102    addListener(future, (r, e) -> {
103      if (e != null) {
104        wrappedFuture.completeExceptionally(e);
105      } else {
106        wrappedFuture.complete(r);
107      }
108    }, executor);
109    return wrappedFuture;
110  }
111
112  /**
113   * Get the cause of the {@link Throwable} if it is a {@link CompletionException}.
114   */
115  public static Throwable unwrapCompletionException(Throwable error) {
116    if (error instanceof CompletionException) {
117      Throwable cause = error.getCause();
118      if (cause != null) {
119        return cause;
120      }
121    }
122    return error;
123  }
124
125  /**
126   * A helper class for getting the result of a Future, and convert the error to an
127   * {@link IOException}.
128   */
129  public static <T> T get(Future<T> future) throws IOException {
130    try {
131      return future.get();
132    } catch (InterruptedException e) {
133      throw (IOException) new InterruptedIOException().initCause(e);
134    } catch (ExecutionException e) {
135      Throwable cause = e.getCause();
136      Throwables.propagateIfPossible(cause, IOException.class);
137      throw new IOException(cause);
138    }
139  }
140
141  /**
142   * A helper class for getting the result of a Future, and convert the error to an
143   * {@link IOException}.
144   */
145  public static <T> T get(Future<T> future, long timeout, TimeUnit unit) throws IOException {
146    try {
147      return future.get(timeout, unit);
148    } catch (InterruptedException e) {
149      throw (IOException) new InterruptedIOException().initCause(e);
150    } catch (ExecutionException e) {
151      Throwable cause = e.getCause();
152      Throwables.propagateIfPossible(cause, IOException.class);
153      throw new IOException(cause);
154    } catch (TimeoutException e) {
155      throw new TimeoutIOException(e);
156    }
157  }
158
159  /**
160   * Returns a CompletableFuture that is already completed exceptionally with the given exception.
161   */
162  public static <T> CompletableFuture<T> failedFuture(Throwable e) {
163    CompletableFuture<T> future = new CompletableFuture<>();
164    future.completeExceptionally(e);
165    return future;
166  }
167}
