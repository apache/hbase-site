090  /**
091   * Compare path component of the Path URI; e.g. if hdfs://a/b/c and /a/b/c, it will compare the
092   * '/a/b/c' part. Does not consider schema; i.e. if schemas different but path or subpath matches,
093   * the two will equate.
094   * @param pathToSearch Path we will be trying to match against.
095   * @param pathTail what to match
096   * @return True if <code>pathTail</code> is tail on the path of <code>pathToSearch</code>
097   */
098  public static boolean isMatchingTail(final Path pathToSearch, String pathTail) {
099    return isMatchingTail(pathToSearch, new Path(pathTail));
100  }

102  /**
103   * Compare path component of the Path URI; e.g. if hdfs://a/b/c and /a/b/c, it will compare the
104   * '/a/b/c' part. If you passed in 'hdfs://a/b/c and b/c, it would return true.  Does not consider
105   * schema; i.e. if schemas different but path or subpath matches, the two will equate.
106   * @param pathToSearch Path we will be trying to match agains against
107   * @param pathTail what to match
108   * @return True if <code>pathTail</code> is tail on the path of <code>pathToSearch</code>
109   */
110  public static boolean isMatchingTail(final Path pathToSearch, final Path pathTail) {
111    if (pathToSearch.depth() != pathTail.depth()) {
112      return false;
113    }
114    Path tailPath = pathTail;
115    String tailName;
116    Path toSearch = pathToSearch;
117    String toSearchName;
118    boolean result = false;
119    do {
120      tailName = tailPath.getName();
121      if (tailName == null || tailName.length() <= 0) {
122        result = true;
123        break;
124      }
125      toSearchName = toSearch.getName();
126      if (toSearchName == null || toSearchName.length() <= 0) {
127        break;
128      }
129      // Move up a parent on each path for next go around.  Path doesn't let us go off the end.
130      tailPath = tailPath.getParent();
131      toSearch = toSearch.getParent();
132    } while(tailName.equals(toSearchName));
133    return result;
134  }

209  /**
210   * Get the file permissions specified in the configuration, if they are
211   * enabled.
212   *
213   * @param fs filesystem that the file will be created on.
214   * @param conf configuration to read for determining if permissions are
215   *          enabled and which to use
216   * @param permssionConfKey property key in the configuration to use when
217   *          finding the permission
218   * @return the permission to use when creating a new file on the fs. If
219   *         special permissions are not specified in the configuration, then
220   *         the default permissions on the the fs will be returned.
221   */
222  public static FsPermission getFilePermissions(final FileSystem fs,
223      final Configuration conf, final String permssionConfKey) {
224    boolean enablePermissions = conf.getBoolean(
225        HConstants.ENABLE_DATA_FILE_UMASK, false);
226
227    if (enablePermissions) {
228      try {
229        FsPermission perm = new FsPermission(FULL_RWX_PERMISSIONS);
230        // make sure that we have a mask, if not, go default.
231        String mask = conf.get(permssionConfKey);
232        if (mask == null) {
233          return FsPermission.getFileDefault();
234        }
235        // appy the umask
236        FsPermission umask = new FsPermission(mask);
237        return perm.applyUMask(umask);
238      } catch (IllegalArgumentException e) {
239        LOG.warn(
240            "Incorrect umask attempted to be created: "
241                + conf.get(permssionConfKey)
242                + ", using default file permissions.", e);
243        return FsPermission.getFileDefault();
244      }
245    }
246    return FsPermission.getFileDefault();
247  }

360  private static boolean isValidWALRootDir(Path walDir, final Configuration c) throws IOException {
361    Path rootDir = getRootDir(c);
362    FileSystem fs = walDir.getFileSystem(c);
363    Path qualifiedWalDir = walDir.makeQualified(fs.getUri(), fs.getWorkingDirectory());
364    if (!qualifiedWalDir.equals(rootDir)) {
365      if (qualifiedWalDir.toString().startsWith(rootDir.toString() + "/")) {
366        throw new IllegalStateException("Illegal WAL directory specified. " +
367            "WAL directories are not permitted to be under the root directory if set.");
368      }
369    }
370    return true;
371  }

456  /**
457   * Sets storage policy for given path.
458   * If the passed path is a directory, we'll set the storage policy for all files
459   * created in the future in said directory. Note that this change in storage
460   * policy takes place at the FileSystem level; it will persist beyond this RS's lifecycle.
461   * If we're running on a version of FileSystem that doesn't support the given storage policy
462   * (or storage policies at all), then we'll issue a log message and continue.
463   *
464   * See http://hadoop.apache.org/docs/r2.6.0/hadoop-project-dist/hadoop-hdfs/ArchivalStorage.html
465   *
466   * @param fs We only do anything it implements a setStoragePolicy method
467   * @param path the Path whose storage policy is to be set
468   * @param storagePolicy Policy to set on <code>path</code>; see hadoop 2.6+
469   *   org.apache.hadoop.hdfs.protocol.HdfsConstants for possible list e.g
470   *   'COLD', 'WARM', 'HOT', 'ONE_SSD', 'ALL_SSD', 'LAZY_PERSIST'.
471   */
472  public static void setStoragePolicy(final FileSystem fs, final Path path,
473      final String storagePolicy) {
474    try {
475      setStoragePolicy(fs, path, storagePolicy, false);
476    } catch (IOException e) {
477      // should never arrive here
478      LOG.warn("We have chosen not to throw exception but some unexpectedly thrown out", e);
479    }
480  }
481
482  static void setStoragePolicy(final FileSystem fs, final Path path, final String storagePolicy,
483      boolean throwException) throws IOException {
484    if (storagePolicy == null) {
485      if (LOG.isTraceEnabled()) {
486        LOG.trace("We were passed a null storagePolicy, exiting early.");
487      }
488      return;
489    }
490    String trimmedStoragePolicy = storagePolicy.trim();
491    if (trimmedStoragePolicy.isEmpty()) {
492      LOG.trace("We were passed an empty storagePolicy, exiting early.");
493      return;
494    } else {
495      trimmedStoragePolicy = trimmedStoragePolicy.toUpperCase(Locale.ROOT);
496    }
497    if (trimmedStoragePolicy.equals(HConstants.DEFER_TO_HDFS_STORAGE_POLICY)) {
498      LOG.trace("We were passed the defer-to-hdfs policy {}, exiting early.", trimmedStoragePolicy);
499      return;
500    }
501    try {
502      invokeSetStoragePolicy(fs, path, trimmedStoragePolicy);
503    } catch (IOException e) {
504      LOG.trace("Failed to invoke set storage policy API on FS", e);
505      if (throwException) {
506        throw e;
507      }
508    }
509  }

511  /*
512   * All args have been checked and are good. Run the setStoragePolicy invocation.
513   */
514  private static void invokeSetStoragePolicy(final FileSystem fs, final Path path,
515      final String storagePolicy) throws IOException {
516    Exception toThrow = null;
517
518    try {
519      fs.setStoragePolicy(path, storagePolicy);
520      LOG.debug("Set storagePolicy={} for path={}", storagePolicy, path);
521    } catch (Exception e) {
522      toThrow = e;
523      // This swallows FNFE, should we be throwing it? seems more likely to indicate dev
524      // misuse than a runtime problem with HDFS.
525      if (!warningMap.containsKey(fs)) {
526        warningMap.put(fs, true);
527        LOG.warn("Unable to set storagePolicy=" + storagePolicy + " for path=" + path + ". " +
528            "DEBUG log level might have more details.", e);
529      } else if (LOG.isDebugEnabled()) {
530        LOG.debug("Unable to set storagePolicy=" + storagePolicy + " for path=" + path, e);
531      }
532
533      // Hadoop 2.8+, 3.0-a1+ added FileSystem.setStoragePolicy with a default implementation
534      // that throws UnsupportedOperationException
535      if (e instanceof UnsupportedOperationException) {
536        if (LOG.isDebugEnabled()) {
537          LOG.debug("The underlying FileSystem implementation doesn't support " +
538              "setStoragePolicy. This is probably intentional on their part, since HDFS-9345 " +
539              "appears to be present in your version of Hadoop. For more information check " +
540              "the Hadoop documentation on 'ArchivalStorage', the Hadoop FileSystem " +
541              "specification docs from HADOOP-11981, and/or related documentation from the " +
542              "provider of the underlying FileSystem (its name should appear in the " +
543              "stacktrace that accompanies this message). Note in particular that Hadoop's " +
544              "local filesystem implementation doesn't support storage policies.", e);
545        }
546      }
547    }
548
549    if (toThrow != null) {
550      throw new IOException(toThrow);
551    }
552  }

583  /**
584   * Calls fs.listStatus() and treats FileNotFoundException as non-fatal
585   * This accommodates differences between hadoop versions, where hadoop 1
586   * does not throw a FileNotFoundException, and return an empty FileStatus[]
587   * while Hadoop 2 will throw FileNotFoundException.
588   *
589   * Where possible, prefer FSUtils#listStatusWithStatusFilter(FileSystem,
590   * Path, FileStatusFilter) instead.
591   *
592   * @param fs file system
593   * @param dir directory
594   * @param filter path filter
595   * @return null if dir is empty or doesn't exist, otherwise FileStatus array
596   */
597  public static FileStatus[] listStatus(final FileSystem fs,
598      final Path dir, final PathFilter filter) throws IOException {
599    FileStatus [] status = null;
600    try {
601      status = filter == null ? fs.listStatus(dir) : fs.listStatus(dir, filter);
602    } catch (FileNotFoundException fnfe) {
603      // if directory doesn't exist, return null
604      if (LOG.isTraceEnabled()) {
605        LOG.trace("{} doesn't exist", dir);
606      }
607    }
608    if (status == null || status.length < 1) {
609      return null;
610    }
611    return status;
612  }

626  /**
627   * Calls fs.listFiles() to get FileStatus and BlockLocations together for reducing rpc call
628   *
629   * @param fs file system
630   * @param dir directory
631   * @return LocatedFileStatus list
632   */
633  public static List<LocatedFileStatus> listLocatedStatus(final FileSystem fs,
634      final Path dir) throws IOException {
635    List<LocatedFileStatus> status = null;
636    try {
637      RemoteIterator<LocatedFileStatus> locatedFileStatusRemoteIterator = fs
638          .listFiles(dir, false);
639      while (locatedFileStatusRemoteIterator.hasNext()) {
640        if (status == null) {
641          status = Lists.newArrayList();
642        }
643        status.add(locatedFileStatusRemoteIterator.next());
644      }
645    } catch (FileNotFoundException fnfe) {
646      // if directory doesn't exist, return null
647      if (LOG.isTraceEnabled()) {
648        LOG.trace("{} doesn't exist", dir);
649      }
650    }
651    return status;
652  }

693  /**
694   * Recursive helper to log the state of the FS
695   *
696   * @see #logFileSystemState(FileSystem, Path, Logger)
697   */
698  private static void logFSTree(Logger log, final FileSystem fs, final Path root, String prefix)
699      throws IOException {
700    FileStatus[] files = listStatus(fs, root, null);
701    if (files == null) {
702      return;
703    }
704
705    for (FileStatus file : files) {
706      if (file.isDirectory()) {
707        log.debug(prefix + file.getPath().getName() + "/");
708        logFSTree(log, fs, file.getPath(), prefix + "---");
709      } else {
710        log.debug(prefix + file.getPath().getName());
711      }
712    }
713  }

722  /**
723   * Check if short circuit read buffer size is set and if not, set it to hbase value.
724   * @param conf must not be null
725   */
726  public static void checkShortCircuitReadBufferSize(final Configuration conf) {
727    final int defaultSize = HConstants.DEFAULT_BLOCKSIZE * 2;
728    final int notSet = -1;
729    // DFSConfigKeys.DFS_CLIENT_READ_SHORTCIRCUIT_BUFFER_SIZE_KEY is only defined in h2
730    final String dfsKey = "dfs.client.read.shortcircuit.buffer.size";
731    int size = conf.getInt(dfsKey, notSet);
732    // If a size is set, return -- we will use it.
733    if (size != notSet) {
734      return;
735    }
736    // But short circuit buffer size is normally not set.  Put in place the hbase wanted size.
737    int hbaseSize = conf.getInt("hbase." + dfsKey, defaultSize);
738    conf.setIfUnset(dfsKey, Integer.toString(hbaseSize));
739  }

741  private static final class DfsBuilderUtility {
742    private static final Class<?> BUILDER;
743    private static final Method REPLICATE;
744
745    static {
746      String builderName = "org.apache.hadoop.hdfs.DistributedFileSystem$HdfsDataOutputStreamBuilder";
747      Class<?> builderClass = null;
748      try {
749        builderClass = Class.forName(builderName);
750      } catch (ClassNotFoundException e) {
751        LOG.debug("{} not available, will not set replicate when creating output stream", builderName);
752      }
753      Method replicateMethod = null;
754      if (builderClass != null) {
755        try {
756          replicateMethod = builderClass.getMethod("replicate");
757          LOG.debug("Using builder API via reflection for DFS file creation.");
758        } catch (NoSuchMethodException e) {
759          LOG.debug("Could not find replicate method on builder; will not set replicate when" +
760            " creating output stream", e);
761        }
762      }
763      BUILDER = builderClass;
764      REPLICATE = replicateMethod;
765    }
766
767    /**
768     * Attempt to use builder API via reflection to call the replicate method on the given builder.
769     */
770    static void replicate(FSDataOutputStreamBuilder<?, ?> builder) {
771      if (BUILDER != null && REPLICATE != null && BUILDER.isAssignableFrom(builder.getClass())) {
772        try {
773          REPLICATE.invoke(builder);
774        } catch (IllegalAccessException | InvocationTargetException e) {
775          // Should have caught this failure during initialization, so log full trace here
776          LOG.warn("Couldn't use reflection with builder API", e);
777        }
778      }
779    }
780  }

782  /**
783   * Attempt to use builder API via reflection to create a file with the given parameters and
784   * replication enabled.
785   * <p/>
786   * Will not attempt to enable replication when passed an HFileSystem.
787   */
788  public static FSDataOutputStream createForWal(FileSystem fs, Path path, boolean overwrite)
789    throws IOException {
790    FSDataOutputStreamBuilder<?, ?> builder = fs.createFile(path).overwrite(overwrite);
791    DfsBuilderUtility.replicate(builder);
792    return builder.build();
793  }

795  /**
796   * Attempt to use builder API via reflection to create a file with the given parameters and
797   * replication enabled.
798   * <p/>
799   * Will not attempt to enable replication when passed an HFileSystem.
800   */
801  public static FSDataOutputStream createForWal(FileSystem fs, Path path, boolean overwrite,
802    int bufferSize, short replication, long blockSize, boolean isRecursive) throws IOException {
803    FSDataOutputStreamBuilder<?, ?> builder = fs.createFile(path).overwrite(overwrite)
804      .bufferSize(bufferSize).replication(replication).blockSize(blockSize);
805    if (isRecursive) {
806      builder.recursive();
807    }
808    DfsBuilderUtility.replicate(builder);
809    return builder.build();
810  }
