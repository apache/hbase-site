044/**
045 * Similar to {@link RegionReplicaUtil} but for the server side
046 */
047public class ServerRegionReplicaUtil extends RegionReplicaUtil {
048
049  private static final Log LOG = LogFactory.getLog(ServerRegionReplicaUtil.class);
050
051  /**
052   * Whether asynchronous WAL replication to the secondary region replicas is enabled or not.
053   * If this is enabled, a replication peer named "region_replica_replication" will be created
054   * which will tail the logs and replicate the mutatations to region replicas for tables that
055   * have region replication &gt; 1. If this is enabled once, disabling this replication also
056   * requires disabling the replication peer using shell or ReplicationAdmin java class.
057   * Replication to secondary region replicas works over standard inter-cluster replication.Â·
058   */
059  public static final String REGION_REPLICA_REPLICATION_CONF_KEY
060    = "hbase.region.replica.replication.enabled";
061  private static final boolean DEFAULT_REGION_REPLICA_REPLICATION = false;
062  private static final String REGION_REPLICA_REPLICATION_PEER = "region_replica_replication";
063
064  /**
065   * Enables or disables refreshing store files of secondary region replicas when the memory is
066   * above the global memstore lower limit. Refreshing the store files means that we will do a file
067   * list of the primary regions store files, and pick up new files. Also depending on the store
068   * files, we can drop some memstore contents which will free up memory.
069   */
070  public static final String REGION_REPLICA_STORE_FILE_REFRESH
071    = "hbase.region.replica.storefile.refresh";
072  private static final boolean DEFAULT_REGION_REPLICA_STORE_FILE_REFRESH = true;
073
074  /**
075   * The multiplier to use when we want to refresh a secondary region instead of flushing a primary
076   * region. Default value assumes that for doing the file refresh, the biggest secondary should be
077   * 4 times bigger than the biggest primary.
078   */
079  public static final String REGION_REPLICA_STORE_FILE_REFRESH_MEMSTORE_MULTIPLIER
080    = "hbase.region.replica.storefile.refresh.memstore.multiplier";
081  private static final double DEFAULT_REGION_REPLICA_STORE_FILE_REFRESH_MEMSTORE_MULTIPLIER = 4;
082
083  /**
084   * Returns the regionInfo object to use for interacting with the file system.
085   * @return An HRegionInfo object to interact with the filesystem
086   */
087  public static HRegionInfo getRegionInfoForFs(HRegionInfo regionInfo) {
088    if (regionInfo == null) {
089      return null;
090    }
091    return RegionReplicaUtil.getRegionInfoForDefaultReplica(regionInfo);
092  }
093
094  /**
095   * Returns whether this region replica can accept writes.
096   * @param region the HRegion object
097   * @return whether the replica is read only
098   */
099  public static boolean isReadOnly(HRegion region) {
100    return region.getTableDesc().isReadOnly()
101      || !isDefaultReplica(region.getRegionInfo());
102  }
103
104  /**
105   * Returns whether to replay the recovered edits to flush the results.
106   * Currently secondary region replicas do not replay the edits, since it would
107   * cause flushes which might affect the primary region. Primary regions even opened
108   * in read only mode should replay the edits.
109   * @param region the HRegion object
110   * @return whether recovered edits should be replayed.
111   */
112  public static boolean shouldReplayRecoveredEdits(HRegion region) {
113    return isDefaultReplica(region.getRegionInfo());
114  }
115
116  /**
117   * Returns a StoreFileInfo from the given FileStatus. Secondary replicas refer to the
118   * files of the primary region, so an HFileLink is used to construct the StoreFileInfo. This
119   * way ensures that the secondary will be able to continue reading the store files even if
120   * they are moved to archive after compaction
121   * @throws IOException
122   */
123  public static StoreFileInfo getStoreFileInfo(Configuration conf, FileSystem fs,
124      HRegionInfo regionInfo, HRegionInfo regionInfoForFs, String familyName, Path path)
125      throws IOException {
126
127    // if this is a primary region, just return the StoreFileInfo constructed from path
128    if (regionInfo.equals(regionInfoForFs)) {
129      return new StoreFileInfo(conf, fs, path);
130    }
131
132    // else create a store file link. The link file does not exists on filesystem though.
133    HFileLink link = HFileLink.build(conf, regionInfoForFs.getTable(),
134            regionInfoForFs.getEncodedName(), familyName, path.getName());
135
136    if (StoreFileInfo.isReference(path)) {
137      Reference reference = Reference.read(fs, path);
138      return new StoreFileInfo(conf, fs, link.getFileStatus(fs), reference);
139    }
140
141    return new StoreFileInfo(conf, fs, link.getFileStatus(fs), link);
142  }
143
144  /**
145   * Create replication peer for replicating to region replicas if needed.
146   * @param conf configuration to use
147   * @throws IOException
148   */
149  public static void setupRegionReplicaReplication(Configuration conf) throws IOException {
150    if (!isRegionReplicaReplicationEnabled(conf)) {
151      return;
152    }
153    Admin admin = ConnectionFactory.createConnection(conf).getAdmin();
154    ReplicationPeerConfig peerConfig = null;
155    try {
156      peerConfig = admin.getReplicationPeerConfig(REGION_REPLICA_REPLICATION_PEER);
157    } catch (ReplicationPeerNotFoundException e) {
158      LOG.warn("Region replica replication peer id=" + REGION_REPLICA_REPLICATION_PEER
159          + " not exist", e);
160    }
161    try {
162      if (peerConfig == null) {
163        LOG.info("Region replica replication peer id=" + REGION_REPLICA_REPLICATION_PEER
164            + " not exist. Creating...");
165        peerConfig = new ReplicationPeerConfig();
166        peerConfig.setClusterKey(ZKConfig.getZooKeeperClusterKey(conf));
167        peerConfig.setReplicationEndpointImpl(RegionReplicaReplicationEndpoint.class.getName());
168        admin.addReplicationPeer(REGION_REPLICA_REPLICATION_PEER, peerConfig);
169      }
170    } finally {
171      admin.close();
172    }
173  }
174
175  public static boolean isRegionReplicaReplicationEnabled(Configuration conf) {
176    return conf.getBoolean(REGION_REPLICA_REPLICATION_CONF_KEY,
177      DEFAULT_REGION_REPLICA_REPLICATION);
178  }
179
180  public static boolean isRegionReplicaWaitForPrimaryFlushEnabled(Configuration conf) {
181    return conf.getBoolean(REGION_REPLICA_WAIT_FOR_PRIMARY_FLUSH_CONF_KEY,
182      DEFAULT_REGION_REPLICA_WAIT_FOR_PRIMARY_FLUSH);
183  }
184
185  public static boolean isRegionReplicaStoreFileRefreshEnabled(Configuration conf) {
186    return conf.getBoolean(REGION_REPLICA_STORE_FILE_REFRESH,
187      DEFAULT_REGION_REPLICA_STORE_FILE_REFRESH);
188  }
189
190  public static double getRegionReplicaStoreFileRefreshMultiplier(Configuration conf) {
191    return conf.getDouble(REGION_REPLICA_STORE_FILE_REFRESH_MEMSTORE_MULTIPLIER,
192      DEFAULT_REGION_REPLICA_STORE_FILE_REFRESH_MEMSTORE_MULTIPLIER);
193  }
194
195  /**
196   * Return the peer id used for replicating to secondary region replicas
197   */
198  public static String getReplicationPeerId() {
199    return REGION_REPLICA_REPLICATION_PEER;
200  }
