132  public static int copyRowTo(Cell cell, byte[] destination, int destinationOffset) {
133    short rowLen = cell.getRowLength();
134    if (cell instanceof ByteBufferedCell) {
135      ByteBufferUtils.copyFromBufferToArray(destination,
136          ((ByteBufferedCell) cell).getRowByteBuffer(),
137          ((ByteBufferedCell) cell).getRowPosition(), destinationOffset, rowLen);
138    } else {
139      System.arraycopy(cell.getRowArray(), cell.getRowOffset(), destination, destinationOffset,
140          rowLen);
141    }
142    return destinationOffset + rowLen;
143  }

375  /**
376   * This can be used when a Cell has to change with addition/removal of one or more tags. This is an
377   * efficient way to do so in which only the tags bytes part need to recreated and copied. All other
378   * parts, refer to the original Cell.
379   */
380  @InterfaceAudience.Private
381  private static class TagRewriteCell implements ExtendedCell {
382    protected Cell cell;
383    protected byte[] tags;
384    private static final long HEAP_SIZE_OVERHEAD = 2 * ClassSize.REFERENCE + ClassSize.ARRAY;
385
386    /**
387     * @param cell The original Cell which it rewrites
388     * @param tags the tags bytes. The array suppose to contain the tags bytes alone.
389     */
390    public TagRewriteCell(Cell cell, byte[] tags) {
391      assert cell instanceof ExtendedCell;
392      assert tags != null;
393      this.cell = cell;
394      this.tags = tags;
395      // tag offset will be treated as 0 and length this.tags.length
396      if (this.cell instanceof TagRewriteCell) {
397        // Cleaning the ref so that the byte[] can be GCed
398        ((TagRewriteCell) this.cell).tags = null;
399      }
400    }
401
402    @Override
403    public byte[] getRowArray() {
404      return cell.getRowArray();
405    }
406
407    @Override
408    public int getRowOffset() {
409      return cell.getRowOffset();
410    }
411
412    @Override
413    public short getRowLength() {
414      return cell.getRowLength();
415    }
416
417    @Override
418    public byte[] getFamilyArray() {
419      return cell.getFamilyArray();
420    }
421
422    @Override
423    public int getFamilyOffset() {
424      return cell.getFamilyOffset();
425    }
426
427    @Override
428    public byte getFamilyLength() {
429      return cell.getFamilyLength();
430    }
431
432    @Override
433    public byte[] getQualifierArray() {
434      return cell.getQualifierArray();
435    }
436
437    @Override
438    public int getQualifierOffset() {
439      return cell.getQualifierOffset();
440    }
441
442    @Override
443    public int getQualifierLength() {
444      return cell.getQualifierLength();
445    }
446
447    @Override
448    public long getTimestamp() {
449      return cell.getTimestamp();
450    }
451
452    @Override
453    public byte getTypeByte() {
454      return cell.getTypeByte();
455    }
456
457    @Override
458    public long getSequenceId() {
459      return cell.getSequenceId();
460    }
461
462    @Override
463    public byte[] getValueArray() {
464      return cell.getValueArray();
465    }
466
467    @Override
468    public int getValueOffset() {
469      return cell.getValueOffset();
470    }
471
472    @Override
473    public int getValueLength() {
474      return cell.getValueLength();
475    }
476
477    @Override
478    public byte[] getTagsArray() {
479      return this.tags;
480    }
481
482    @Override
483    public int getTagsOffset() {
484      return 0;
485    }
486
487    @Override
488    public int getTagsLength() {
489      if (null == this.tags) {
490        // Nulled out tags array optimization in constructor
491        return 0;
492      }
493      return this.tags.length;
494    }
495
496    @Override
497    public long heapSize() {
498      long sum = CellUtil.estimatedHeapSizeOf(cell) - cell.getTagsLength();
499      sum += ClassSize.OBJECT;// this object itself
500      sum += (2 * ClassSize.REFERENCE);// pointers to cell and tags array
501      if (this.tags != null) {
502        sum += ClassSize.align(ClassSize.ARRAY);// "tags"
503        sum += this.tags.length;
504      }
505      return sum;
506    }
507
508    @Override
509    public void setTimestamp(long ts) throws IOException {
510      // The incoming cell is supposed to be SettableTimestamp type.
511      CellUtil.setTimestamp(cell, ts);
512    }
513
514    @Override
515    public void setTimestamp(byte[] ts, int tsOffset) throws IOException {
516      // The incoming cell is supposed to be SettableTimestamp type.
517      CellUtil.setTimestamp(cell, ts, tsOffset);
518    }
519
520    @Override
521    public void setSequenceId(long seqId) throws IOException {
522      // The incoming cell is supposed to be SettableSequenceId type.
523      CellUtil.setSequenceId(cell, seqId);
524    }
525
526    @Override
527    public int write(OutputStream out, boolean withTags) throws IOException {
528      int len = ((ExtendedCell) this.cell).write(out, false);
529      if (withTags && this.tags != null) {
530        // Write the tagsLength 2 bytes
531        out.write((byte) (0xff & (this.tags.length >> 8)));
532        out.write((byte) (0xff & this.tags.length));
533        out.write(this.tags);
534        len += KeyValue.TAGS_LENGTH_SIZE + this.tags.length;
535      }
536      return len;
537    }
538
539    @Override
540    public int getSerializedSize(boolean withTags) {
541      int len = ((ExtendedCell) this.cell).getSerializedSize(false);
542      if (withTags && this.tags != null) {
543        len += KeyValue.TAGS_LENGTH_SIZE + this.tags.length;
544      }
545      return len;
546    }
547
548    @Override
549    public void write(byte[] buf, int offset) {
550      offset = KeyValueUtil.appendToByteArray(this.cell, buf, offset, false);
551      int tagsLen = this.tags.length;
552      assert tagsLen > 0;
553      offset = Bytes.putAsShort(buf, offset, tagsLen);
554      System.arraycopy(this.tags, 0, buf, offset, tagsLen);
555    }
556
557    @Override
558    public long heapOverhead() {
559      return ((ExtendedCell) this.cell).heapOverhead() + HEAP_SIZE_OVERHEAD;
560    }
561  }

582   * @param cellScannerables
583   * @return CellScanner interface over <code>cellIterables</code>
584   */
585  public static CellScanner createCellScanner(
586      final List<? extends CellScannable> cellScannerables) {
587    return new CellScanner() {
588      private final Iterator<? extends CellScannable> iterator = cellScannerables.iterator();
589      private CellScanner cellScanner = null;
590
591      @Override
592      public Cell current() {
593        return this.cellScanner != null? this.cellScanner.current(): null;
594      }
595
596      @Override
597      public boolean advance() throws IOException {
598        while (true) {
599          if (this.cellScanner == null) {
600            if (!this.iterator.hasNext()) return false;
601            this.cellScanner = this.iterator.next().cellScanner();
602          }
603          if (this.cellScanner.advance()) return true;
604          this.cellScanner = null;
605        }
606      }
607    };
608  }

667  /**
668   * Flatten the map of cells out under the CellScanner
669   * @param map Map of Cell Lists; for example, the map of families to Cells that is used
670   * inside Put, etc., keeping Cells organized by family.
671   * @return CellScanner interface over <code>cellIterable</code>
672   */
673  public static CellScanner createCellScanner(final NavigableMap<byte [], List<Cell>> map) {
674    return new CellScanner() {
675      private final Iterator<Entry<byte[], List<Cell>>> entries = map.entrySet().iterator();
676      private Iterator<Cell> currentIterator = null;
677      private Cell currentCell;
678
679      @Override
680      public Cell current() {
681        return this.currentCell;
682      }
683
684      @Override
685      public boolean advance() {
686        while(true) {
687          if (this.currentIterator == null) {
688            if (!this.entries.hasNext()) return false;
689            this.currentIterator = this.entries.next().getValue().iterator();
690          }
691          if (this.currentIterator.hasNext()) {
692            this.currentCell = this.currentIterator.next();
693            return true;
694          }
695          this.currentCell = null;
696          this.currentIterator = null;
697        }
698      }
699    };
700  }

1131   * @param cell The Cell
1132   * @param type Type of the Tag to retrieve
1133   * @return null if there is no tag of the passed in tag type
1134   */
1135  public static Tag getTag(Cell cell, byte type){
1136    boolean bufferBacked = cell instanceof ByteBufferedCell;
1137    int length = cell.getTagsLength();
1138    int offset = bufferBacked? ((ByteBufferedCell)cell).getTagsPosition():cell.getTagsOffset();
1139    int pos = offset;
1140    while (pos < offset + length) {
1141      int tagLen;
1142      if (bufferBacked) {
1143        ByteBuffer tagsBuffer = ((ByteBufferedCell)cell).getTagsByteBuffer();
1144        tagLen = ByteBufferUtils.readAsInt(tagsBuffer, pos, TAG_LENGTH_SIZE);
1145        if (ByteBufferUtils.toByte(tagsBuffer, pos + TAG_LENGTH_SIZE) == type) {
1146          return new OffheapTag(tagsBuffer, pos, tagLen + TAG_LENGTH_SIZE);

1151          return new ArrayBackedTag(cell.getTagsArray(), pos, tagLen + TAG_LENGTH_SIZE);
1152        }
1153      }
1154      pos += TAG_LENGTH_SIZE + tagLen;
1155    }
1156    return null;

1469  /**
1470   * Find length of common prefix in keys of the cells, considering key as byte[] if serialized in
1471   * {@link KeyValue}. The key format is &lt;2 bytes rk len&gt;&lt;rk&gt;&lt;1 byte cf
1472   * len&gt;&lt;cf&gt;&lt;qualifier&gt;&lt;8 bytes timestamp&gt;&lt;1 byte type&gt;
1473   * @param c1
1474   *          the cell
1475   * @param c2
1476   *          the cell
1477   * @param bypassFamilyCheck
1478   *          when true assume the family bytes same in both cells. Pass it as true when dealing
1479   *          with Cells in same CF so as to avoid some checks
1480   * @param withTsType
1481   *          when true check timestamp and type bytes also.
1482   * @return length of common prefix
1483   */
1484  public static int findCommonPrefixInFlatKey(Cell c1, Cell c2, boolean bypassFamilyCheck,
1485      boolean withTsType) {
1486    // Compare the 2 bytes in RK length part
1487    short rLen1 = c1.getRowLength();
1488    short rLen2 = c2.getRowLength();
1489    int commonPrefix = KeyValue.ROW_LENGTH_SIZE;
1490    if (rLen1 != rLen2) {
1491      // early out when the RK length itself is not matching
1492      return ByteBufferUtils.findCommonPrefix(Bytes.toBytes(rLen1), 0, KeyValue.ROW_LENGTH_SIZE,
1493          Bytes.toBytes(rLen2), 0, KeyValue.ROW_LENGTH_SIZE);
1494    }
1495    // Compare the RKs
1496    int rkCommonPrefix = 0;
1497    if (c1 instanceof ByteBufferedCell && c2 instanceof ByteBufferedCell) {
1498      rkCommonPrefix = ByteBufferUtils.findCommonPrefix(((ByteBufferedCell) c1).getRowByteBuffer(),
1499        ((ByteBufferedCell) c1).getRowPosition(), rLen1, ((ByteBufferedCell) c2).getRowByteBuffer(),
1500        ((ByteBufferedCell) c2).getRowPosition(), rLen2);
1501    } else {
1502      // There cannot be a case where one cell is BBCell and other is KeyValue. This flow comes either
1503      // in flush or compactions. In flushes both cells are KV and in case of compaction it will be either
1504      // KV or BBCell
1505      rkCommonPrefix = ByteBufferUtils.findCommonPrefix(c1.getRowArray(), c1.getRowOffset(),
1506        rLen1, c2.getRowArray(), c2.getRowOffset(), rLen2);
1507    }
1508    commonPrefix += rkCommonPrefix;
1509    if (rkCommonPrefix != rLen1) {
1510      // Early out when RK is not fully matching.
1511      return commonPrefix;
1512    }
1513    // Compare 1 byte CF length part
1514    byte fLen1 = c1.getFamilyLength();
1515    if (bypassFamilyCheck) {
1516      // This flag will be true when caller is sure that the family will be same for both the cells
1517      // Just make commonPrefix to increment by the family part
1518      commonPrefix += KeyValue.FAMILY_LENGTH_SIZE + fLen1;
1519    } else {
1520      byte fLen2 = c2.getFamilyLength();
1521      if (fLen1 != fLen2) {
1522        // early out when the CF length itself is not matching
1523        return commonPrefix;
1524      }
1525      // CF lengths are same so there is one more byte common in key part
1526      commonPrefix += KeyValue.FAMILY_LENGTH_SIZE;
1527      // Compare the CF names
1528      int fCommonPrefix;
1529      if (c1 instanceof ByteBufferedCell && c2 instanceof ByteBufferedCell) {
1530        fCommonPrefix =
1531            ByteBufferUtils.findCommonPrefix(((ByteBufferedCell) c1).getFamilyByteBuffer(),
1532              ((ByteBufferedCell) c1).getFamilyPosition(), fLen1,
1533              ((ByteBufferedCell) c2).getFamilyByteBuffer(),
1534              ((ByteBufferedCell) c2).getFamilyPosition(), fLen2);
1535      } else {
1536        fCommonPrefix = ByteBufferUtils.findCommonPrefix(c1.getFamilyArray(), c1.getFamilyOffset(),
1537          fLen1, c2.getFamilyArray(), c2.getFamilyOffset(), fLen2);
1538      }
1539      commonPrefix += fCommonPrefix;
1540      if (fCommonPrefix != fLen1) {
1541        return commonPrefix;
1542      }
1543    }
1544    // Compare the Qualifiers
1545    int qLen1 = c1.getQualifierLength();
1546    int qLen2 = c2.getQualifierLength();
1547    int qCommon;
1548    if (c1 instanceof ByteBufferedCell && c2 instanceof ByteBufferedCell) {
1549      qCommon = ByteBufferUtils.findCommonPrefix(((ByteBufferedCell) c1).getQualifierByteBuffer(),
1550        ((ByteBufferedCell) c1).getQualifierPosition(), qLen1,
1551        ((ByteBufferedCell) c2).getQualifierByteBuffer(),
1552        ((ByteBufferedCell) c2).getQualifierPosition(), qLen2);
1553    } else {
1554      qCommon = ByteBufferUtils.findCommonPrefix(c1.getQualifierArray(), c1.getQualifierOffset(),
1555        qLen1, c2.getQualifierArray(), c2.getQualifierOffset(), qLen2);
1556    }
1557    commonPrefix += qCommon;
1558    if (!withTsType || Math.max(qLen1, qLen2) != qCommon) {
1559      return commonPrefix;
1560    }
1561    // Compare the timestamp parts
1562    int tsCommonPrefix = ByteBufferUtils.findCommonPrefix(Bytes.toBytes(c1.getTimestamp()), 0,
1563        KeyValue.TIMESTAMP_SIZE, Bytes.toBytes(c2.getTimestamp()), 0, KeyValue.TIMESTAMP_SIZE);
1564    commonPrefix += tsCommonPrefix;
1565    if (tsCommonPrefix != KeyValue.TIMESTAMP_SIZE) {
1566      return commonPrefix;
1567    }
1568    // Compare the type
1569    if (c1.getTypeByte() == c2.getTypeByte()) {
1570      commonPrefix += KeyValue.TYPE_SIZE;
1571    }
1572    return commonPrefix;
1573  }

1652   *
1653   * @param left
1654   * @param right
1655   * @return True if rows match.
1656   */
1657  public static boolean matchingRows(final Cell left, final Cell right) {
1658    short lrowlength = left.getRowLength();
1659    short rrowlength = right.getRowLength();
1660    if (lrowlength != rrowlength) return false;
1661    if (left instanceof ByteBufferedCell && right instanceof ByteBufferedCell) {
1662      return ByteBufferUtils.equals(((ByteBufferedCell) left).getRowByteBuffer(),
1663          ((ByteBufferedCell) left).getRowPosition(), lrowlength,
1664          ((ByteBufferedCell) right).getRowByteBuffer(),
1665          ((ByteBufferedCell) right).getRowPosition(), rrowlength);
1666    }
1667    if (left instanceof ByteBufferedCell) {
1668      return ByteBufferUtils.equals(((ByteBufferedCell) left).getRowByteBuffer(),
1669          ((ByteBufferedCell) left).getRowPosition(), lrowlength, right.getRowArray(),
1670          right.getRowOffset(), rrowlength);
1671    }
1672    if (right instanceof ByteBufferedCell) {
1673      return ByteBufferUtils.equals(((ByteBufferedCell) right).getRowByteBuffer(),
1674          ((ByteBufferedCell) right).getRowPosition(), rrowlength, left.getRowArray(),
1675          left.getRowOffset(), lrowlength);
1676    }
1677    return Bytes.equals(left.getRowArray(), left.getRowOffset(), lrowlength,
1678        right.getRowArray(), right.getRowOffset(), rrowlength);
1679  }
