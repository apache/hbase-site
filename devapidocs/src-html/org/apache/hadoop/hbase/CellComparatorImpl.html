031/**
032 * Compare two HBase cells.  Do not use this method comparing <code>-ROOT-</code> or
033 * <code>hbase:meta</code> cells.  Cells from these tables need a specialized comparator, one that
034 * takes account of the special formatting of the row where we have commas to delimit table from
035 * regionname, from row.  See KeyValue for how it has a special comparator to do hbase:meta cells
036 * and yet another for -ROOT-.
037 * While using this comparator for {{@link #compareRows(Cell, Cell)} et al, the hbase:meta cells
038 * format should be taken into consideration, for which the instance of this comparator
039 * should be used.  In all other cases the static APIs in this comparator would be enough
040 */
041@edu.umd.cs.findbugs.annotations.SuppressWarnings(
042    value="UNKNOWN",
043    justification="Findbugs doesn't like the way we are negating the result of a compare in below")
044@InterfaceAudience.Private
045@InterfaceStability.Evolving
046public class CellComparatorImpl implements CellComparator {
047  static final Logger LOG = LoggerFactory.getLogger(CellComparatorImpl.class);
048  /**
049   * Comparator for plain key/values; i.e. non-catalog table key/values. Works on Key portion
050   * of KeyValue only.
051   */
052  public static final CellComparatorImpl COMPARATOR = new CellComparatorImpl();
053  /**
054   * A {@link CellComparatorImpl} for <code>hbase:meta</code> catalog table
055   * {@link KeyValue}s.
056   */
057  public static final CellComparatorImpl META_COMPARATOR = new MetaCellComparator();

059  @Override
060  public int compare(Cell a, Cell b) {
061    return compare(a, b, false);
062  }
063
064  /**
065   * Compare cells.
066   * @param a
067   * @param b
068   * @param ignoreSequenceid True if we are to compare the key portion only and ignore
069   * the sequenceid. Set to false to compare key and consider sequenceid.
070   * @return 0 if equal, -1 if a &lt; b, and +1 if a &gt; b.
071   */
072  public final int compare(final Cell a, final Cell b, boolean ignoreSequenceid) {
073    int diff = 0;
074    if (a instanceof ByteBufferKeyValue && b instanceof ByteBufferKeyValue) {
075      diff = compareByteBufferKeyValue((ByteBufferKeyValue)a, (ByteBufferKeyValue)b);
076      if (diff != 0) {
077        return diff;
078      }
079    } else {
080      diff = compareRows(a, b);
081      if (diff != 0) {
082        return diff;
083      }
084
085      diff = compareWithoutRow(a, b);
086      if (diff != 0) {
087        return diff;
088      }
089    }
090
091    // Negate following comparisons so later edits show up first mvccVersion: later sorts first
092    return ignoreSequenceid? diff: Longs.compare(b.getSequenceId(), a.getSequenceId());
093  }

095  /**
096   * Specialized comparator for the ByteBufferKeyValue type exclusivesly.
097   * Caches deserialized lengths of rows and families, etc., and reuses them where it can
098   * (ByteBufferKeyValue has been changed to be amenable to our providing pre-made lengths, etc.)
099   */
100  private final int compareByteBufferKeyValue(ByteBufferKeyValue left, ByteBufferKeyValue right) {
101    // Compare Rows. Cache row length.
102    int leftRowLength = left.getRowLength();
103    int rightRowLength = right.getRowLength();
104    int diff = ByteBufferUtils.compareTo(
105        left.getRowByteBuffer(), left.getRowPosition(), leftRowLength,
106        right.getRowByteBuffer(), right.getRowPosition(), rightRowLength);
107    if (diff != 0) {
108      return diff;
109    }
110
111    // If the column is not specified, the "minimum" key type appears the
112    // latest in the sorted order, regardless of the timestamp. This is used
113    // for specifying the last key/value in a given row, because there is no
114    // "lexicographically last column" (it would be infinitely long). The
115    // "maximum" key type does not need this behavior.
116    // Copied from KeyValue. This is bad in that we can't do memcmp w/ special rules like this.
117    // I tried to get rid of the above but scanners depend on it. TODO.
118    int leftFamilyLengthPosition = left.getFamilyLengthPosition(leftRowLength);
119    int leftFamilyLength = left.getFamilyLength(leftFamilyLengthPosition);
120    int rightFamilyLengthPosition = right.getFamilyLengthPosition(rightRowLength);
121    int rightFamilyLength = right.getFamilyLength(rightFamilyLengthPosition);
122    int leftKeyLength = left.getKeyLength();
123    int leftQualifierLength = left.getQualifierLength(leftKeyLength, leftRowLength,
124        leftFamilyLength);
125    byte leftType = left.getTypeByte(leftKeyLength);
126    if (leftFamilyLength + leftQualifierLength == 0 && leftType == Type.Minimum.getCode()) {
127      // left is "bigger", i.e. it appears later in the sorted order
128      return 1;
129    }
130    int rightKeyLength = right.getKeyLength();
131    int rightQualifierLength = right.getQualifierLength(rightKeyLength, rightRowLength,
132        rightFamilyLength);
133    byte rightType = right.getTypeByte(rightKeyLength);
134    if (rightFamilyLength + rightQualifierLength == 0 && rightType == Type.Minimum.getCode()) {
135      return -1;
136    }
137    // Compare families.
138    int leftFamilyPosition = left.getFamilyPosition(leftFamilyLengthPosition);
139    int rightFamilyPosition = right.getFamilyPosition(rightFamilyLengthPosition);
140    diff = ByteBufferUtils.compareTo(
141        left.getFamilyByteBuffer(), leftFamilyPosition, leftFamilyLength,
142        right.getFamilyByteBuffer(), rightFamilyPosition, rightFamilyLength);
143    if (diff != 0) {
144      return diff;
145    }
146    // Compare qualifiers
147    diff = ByteBufferUtils.compareTo(left.getQualifierByteBuffer(),
148        left.getQualifierPosition(leftFamilyPosition, leftFamilyLength), leftQualifierLength,
149        right.getQualifierByteBuffer(),
150        right.getQualifierPosition(rightFamilyPosition, rightFamilyLength),
151        rightQualifierLength);
152    if (diff != 0) {
153      return diff;
154    }
155    // Timestamps.
156    diff = compareTimestamps(left.getTimestamp(leftKeyLength), right.getTimestamp(rightKeyLength));
157    if (diff != 0) {
158      return diff;
159    }
160    // Compare types. Let the delete types sort ahead of puts; i.e. types
161    // of higher numbers sort before those of lesser numbers. Maximum (255)
162    // appears ahead of everything, and minimum (0) appears after
163    // everything.
164    return (0xff & rightType) - (0xff & leftType);
165  }

167  /**
168   * Compares the family and qualifier part of the cell
169   * @param left the left cell
170   * @param right the right cell
171   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
172   */
173  public final int compareColumns(final Cell left, final Cell right) {
174    int diff = compareFamilies(left, right);
175    if (diff != 0) {
176      return diff;
177    }
178    return compareQualifiers(left, right);
179  }
180
181  /**
182   * Compare the families of left and right cell
183   * @param left
184   * @param right
185   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
186   */
187  @Override
188  public final int compareFamilies(Cell left, Cell right) {
189    if (left instanceof ByteBufferExtendedCell && right instanceof ByteBufferExtendedCell) {
190      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getFamilyByteBuffer(),
191          ((ByteBufferExtendedCell) left).getFamilyPosition(), left.getFamilyLength(),
192          ((ByteBufferExtendedCell) right).getFamilyByteBuffer(),
193          ((ByteBufferExtendedCell) right).getFamilyPosition(), right.getFamilyLength());
194    }
195    if (left instanceof ByteBufferExtendedCell) {
196      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getFamilyByteBuffer(),
197          ((ByteBufferExtendedCell) left).getFamilyPosition(), left.getFamilyLength(),
198          right.getFamilyArray(), right.getFamilyOffset(), right.getFamilyLength());
199    }
200    if (right instanceof ByteBufferExtendedCell) {
201      // Notice how we flip the order of the compare here. We used to negate the return value but
202      // see what FindBugs says
203      // http://findbugs.sourceforge.net/bugDescriptions.html#RV_NEGATING_RESULT_OF_COMPARETO
204      // It suggest flipping the order to get same effect and 'safer'.
205      return ByteBufferUtils.compareTo(
206          left.getFamilyArray(), left.getFamilyOffset(), left.getFamilyLength(),
207          ((ByteBufferExtendedCell)right).getFamilyByteBuffer(),
208          ((ByteBufferExtendedCell)right).getFamilyPosition(), right.getFamilyLength());
209    }
210    return Bytes.compareTo(left.getFamilyArray(), left.getFamilyOffset(), left.getFamilyLength(),
211        right.getFamilyArray(), right.getFamilyOffset(), right.getFamilyLength());
212  }

214  /**
215   * Compare the qualifiers part of the left and right cells.
216   * @param left
217   * @param right
218   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
219   */
220  @Override
221  public final int compareQualifiers(Cell left, Cell right) {
222    if (left instanceof ByteBufferExtendedCell && right instanceof ByteBufferExtendedCell) {
223      return ByteBufferUtils
224          .compareTo(((ByteBufferExtendedCell) left).getQualifierByteBuffer(),
225              ((ByteBufferExtendedCell) left).getQualifierPosition(),
226              left.getQualifierLength(), ((ByteBufferExtendedCell) right).getQualifierByteBuffer(),
227              ((ByteBufferExtendedCell) right).getQualifierPosition(),
228              right.getQualifierLength());
229    }
230    if (left instanceof ByteBufferExtendedCell) {
231      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getQualifierByteBuffer(),
232          ((ByteBufferExtendedCell) left).getQualifierPosition(), left.getQualifierLength(),
233          right.getQualifierArray(), right.getQualifierOffset(), right.getQualifierLength());
234    }
235    if (right instanceof ByteBufferExtendedCell) {
236      // Notice how we flip the order of the compare here. We used to negate the return value but
237      // see what FindBugs says
238      // http://findbugs.sourceforge.net/bugDescriptions.html#RV_NEGATING_RESULT_OF_COMPARETO
239      // It suggest flipping the order to get same effect and 'safer'.
240      return ByteBufferUtils.compareTo(left.getQualifierArray(),
241          left.getQualifierOffset(), left.getQualifierLength(),
242          ((ByteBufferExtendedCell)right).getQualifierByteBuffer(),
243          ((ByteBufferExtendedCell)right).getQualifierPosition(), right.getQualifierLength());
244    }
245    return Bytes.compareTo(left.getQualifierArray(), left.getQualifierOffset(),
246        left.getQualifierLength(), right.getQualifierArray(), right.getQualifierOffset(),
247        right.getQualifierLength());
248  }

250  /**
251   * Compares the rows of the left and right cell.
252   * For the hbase:meta case this method is overridden such that it can handle hbase:meta cells.
253   * The caller should ensure using the appropriate comparator for hbase:meta.
254   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
255   */
256  @Override
257  public int compareRows(final Cell left, final Cell right) {
258    return compareRows(left, left.getRowLength(), right, right.getRowLength());
259  }
260
261  int compareRows(final Cell left, int leftRowLength, final Cell right, int rightRowLength) {
262    // left and right can be exactly the same at the beginning of a row
263    if (left == right) {
264      return 0;
265    }
266    if (left instanceof ByteBufferExtendedCell && right instanceof ByteBufferExtendedCell) {
267      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getRowByteBuffer(),
268          ((ByteBufferExtendedCell) left).getRowPosition(), leftRowLength,
269          ((ByteBufferExtendedCell) right).getRowByteBuffer(),
270          ((ByteBufferExtendedCell) right).getRowPosition(), rightRowLength);
271    }
272    if (left instanceof ByteBufferExtendedCell) {
273      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getRowByteBuffer(),
274          ((ByteBufferExtendedCell) left).getRowPosition(), leftRowLength,
275          right.getRowArray(), right.getRowOffset(), rightRowLength);
276    }
277    if (right instanceof ByteBufferExtendedCell) {
278      // Notice how we flip the order of the compare here. We used to negate the return value but
279      // see what FindBugs says
280      // http://findbugs.sourceforge.net/bugDescriptions.html#RV_NEGATING_RESULT_OF_COMPARETO
281      // It suggest flipping the order to get same effect and 'safer'.
282      return ByteBufferUtils.compareTo(left.getRowArray(), left.getRowOffset(), leftRowLength,
283          ((ByteBufferExtendedCell)right).getRowByteBuffer(),
284          ((ByteBufferExtendedCell)right).getRowPosition(), rightRowLength);
285    }
286    return Bytes.compareTo(left.getRowArray(), left.getRowOffset(), left.getRowLength(),
287        right.getRowArray(), right.getRowOffset(), right.getRowLength());
288  }

290  /**
291   * Compares the row part of the cell with a simple plain byte[] like the
292   * stopRow in Scan. This should be used with context where for hbase:meta
293   * cells the {{@link #META_COMPARATOR} should be used
294   *
295   * @param left
296   *          the cell to be compared
297   * @param right
298   *          the kv serialized byte[] to be compared with
299   * @param roffset
300   *          the offset in the byte[]
301   * @param rlength
302   *          the length in the byte[]
303   * @return 0 if both cell and the byte[] are equal, 1 if the cell is bigger
304   *         than byte[], -1 otherwise
305   */
306  @Override
307  public int compareRows(Cell left, byte[] right, int roffset, int rlength) {
308    if (left instanceof ByteBufferExtendedCell) {
309      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getRowByteBuffer(),
310          ((ByteBufferExtendedCell) left).getRowPosition(), left.getRowLength(), right,
311          roffset, rlength);
312    }
313    return Bytes.compareTo(left.getRowArray(), left.getRowOffset(), left.getRowLength(), right,
314        roffset, rlength);
315  }

317  @Override
318  public final int compareWithoutRow(final Cell left, final Cell right) {
319    // If the column is not specified, the "minimum" key type appears the
320    // latest in the sorted order, regardless of the timestamp. This is used
321    // for specifying the last key/value in a given row, because there is no
322    // "lexicographically last column" (it would be infinitely long). The
323    // "maximum" key type does not need this behavior.
324    // Copied from KeyValue. This is bad in that we can't do memcmp w/ special rules like this.
325    int lFamLength = left.getFamilyLength();
326    int rFamLength = right.getFamilyLength();
327    int lQualLength = left.getQualifierLength();
328    int rQualLength = right.getQualifierLength();
329    if (lFamLength + lQualLength == 0
330          && left.getTypeByte() == Type.Minimum.getCode()) {
331      // left is "bigger", i.e. it appears later in the sorted order
332      return 1;
333    }
334    if (rFamLength + rQualLength == 0
335        && right.getTypeByte() == Type.Minimum.getCode()) {
336      return -1;
337    }
338    if (lFamLength != rFamLength) {
339      // comparing column family is enough.
340      return compareFamilies(left, right);
341    }
342    // Compare cf:qualifier
343    int diff = compareColumns(left, right);
344    if (diff != 0) {
345      return diff;
346    }
347
348    diff = compareTimestamps(left, right);
349    if (diff != 0) {
350      return diff;
351    }
352
353    // Compare types. Let the delete types sort ahead of puts; i.e. types
354    // of higher numbers sort before those of lesser numbers. Maximum (255)
355    // appears ahead of everything, and minimum (0) appears after
356    // everything.
357    return (0xff & right.getTypeByte()) - (0xff & left.getTypeByte());
358  }

360  /**
361   * Compares cell's timestamps in DESCENDING order.
362   * The below older timestamps sorting ahead of newer timestamps looks
363   * wrong but it is intentional. This way, newer timestamps are first
364   * found when we iterate over a memstore and newer versions are the
365   * first we trip over when reading from a store file.
366   * @return 1 if left's timestamp &lt; right's timestamp
367   *         -1 if left's timestamp &gt; right's timestamp
368   *         0 if both timestamps are equal
369   */
370  @Override
371  public int compareTimestamps(final Cell left, final Cell right) {
372    return compareTimestamps(left.getTimestamp(), right.getTimestamp());
373  }

376  /**
377   * Compares timestamps in DESCENDING order.
378   * The below older timestamps sorting ahead of newer timestamps looks
379   * wrong but it is intentional. This way, newer timestamps are first
380   * found when we iterate over a memstore and newer versions are the
381   * first we trip over when reading from a store file.
382   * @return 1 if left timestamp &lt; right timestamp
383   *         -1 if left timestamp &gt; right timestamp
384   *         0 if both timestamps are equal
385   */
386  @Override
387  public int compareTimestamps(final long ltimestamp, final long rtimestamp) {
388    if (ltimestamp < rtimestamp) {
389      return 1;
390    } else if (ltimestamp > rtimestamp) {
391      return -1;
392    }
393    return 0;
394  }

396  /**
397   * A {@link CellComparatorImpl} for <code>hbase:meta</code> catalog table
398   * {@link KeyValue}s.
399   */
400  public static class MetaCellComparator extends CellComparatorImpl {
401
402    @Override
403    public int compareRows(final Cell left, final Cell right) {
404      return compareRows(left.getRowArray(), left.getRowOffset(), left.getRowLength(),
405          right.getRowArray(), right.getRowOffset(), right.getRowLength());
406    }
407
408    @Override
409    public int compareRows(Cell left, byte[] right, int roffset, int rlength) {
410      return compareRows(left.getRowArray(), left.getRowOffset(), left.getRowLength(), right,
411          roffset, rlength);
412    }
413
414    private int compareRows(byte[] left, int loffset, int llength, byte[] right, int roffset,
415        int rlength) {
416      int leftDelimiter = Bytes.searchDelimiterIndex(left, loffset, llength, HConstants.DELIMITER);
417      int rightDelimiter = Bytes
418          .searchDelimiterIndex(right, roffset, rlength, HConstants.DELIMITER);
419      // Compare up to the delimiter
420      int lpart = (leftDelimiter < 0 ? llength : leftDelimiter - loffset);
421      int rpart = (rightDelimiter < 0 ? rlength : rightDelimiter - roffset);
422      int result = Bytes.compareTo(left, loffset, lpart, right, roffset, rpart);
423      if (result != 0) {
424        return result;
425      } else {
426        if (leftDelimiter < 0 && rightDelimiter >= 0) {
427          return -1;
428        } else if (rightDelimiter < 0 && leftDelimiter >= 0) {
429          return 1;
430        } else if (leftDelimiter < 0 && rightDelimiter < 0) {
431          return 0;
432        }
433      }
434      // Compare middle bit of the row.
435      // Move past delimiter
436      leftDelimiter++;
437      rightDelimiter++;
438      int leftFarDelimiter = Bytes.searchDelimiterIndexInReverse(left, leftDelimiter, llength
439          - (leftDelimiter - loffset), HConstants.DELIMITER);
440      int rightFarDelimiter = Bytes.searchDelimiterIndexInReverse(right, rightDelimiter, rlength
441          - (rightDelimiter - roffset), HConstants.DELIMITER);
442      // Now compare middlesection of row.
443      lpart = (leftFarDelimiter < 0 ? llength + loffset : leftFarDelimiter) - leftDelimiter;
444      rpart = (rightFarDelimiter < 0 ? rlength + roffset : rightFarDelimiter) - rightDelimiter;
445      result = Bytes.compareTo(left, leftDelimiter, lpart, right, rightDelimiter, rpart);
446      if (result != 0) {
447        return result;
448      } else {
449        if (leftDelimiter < 0 && rightDelimiter >= 0) {
450          return -1;
451        } else if (rightDelimiter < 0 && leftDelimiter >= 0) {
452          return 1;
453        } else if (leftDelimiter < 0 && rightDelimiter < 0) {
454          return 0;
455        }
456      }
457      // Compare last part of row, the rowid.
458      leftFarDelimiter++;
459      rightFarDelimiter++;
460      result = Bytes.compareTo(left, leftFarDelimiter, llength - (leftFarDelimiter - loffset),
461          right, rightFarDelimiter, rlength - (rightFarDelimiter - roffset));
462      return result;
463    }
464  }
