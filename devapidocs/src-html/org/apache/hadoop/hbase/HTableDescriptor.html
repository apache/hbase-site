677  /**
678   * Return a HColumnDescriptor for user to keep the compatibility as much as possible.
679   * @param desc read-only ColumnFamilyDescriptor
680   * @return The older implementation of ColumnFamilyDescriptor
681   */
682  protected HColumnDescriptor toHColumnDescriptor(ColumnFamilyDescriptor desc) {
683    if (desc == null) {
684      return null;
685    } else if (desc instanceof ModifyableColumnFamilyDescriptor) {
686      return new HColumnDescriptor((ModifyableColumnFamilyDescriptor) desc);
687    } else if (desc instanceof HColumnDescriptor) {
688      return (HColumnDescriptor) desc;
689    } else {
690      return new HColumnDescriptor(new ModifyableColumnFamilyDescriptor(desc));
691    }
692  }

714   * @param jarFilePath Path of the jar file. If it's null, the class will be
715   * loaded from default classloader.
716   * @param className Full class name.
717   * @param priority Priority
718   * @param kvs Arbitrary key-value parameter pairs passed into the coprocessor.
719   * @throws IOException
720   */
721  public HTableDescriptor addCoprocessor(String className, Path jarFilePath,
722                             int priority, final Map<String, String> kvs)
723  throws IOException {
724    getDelegateeForModification().setCoprocessor(
725      CoprocessorDescriptorBuilder.newBuilder(className)
726        .setJarPath(jarFilePath == null ? null : jarFilePath.toString())
727        .setPriority(priority)
728        .setProperties(kvs == null ? Collections.emptyMap() : kvs)
729        .build());
730    return this;
731  }
