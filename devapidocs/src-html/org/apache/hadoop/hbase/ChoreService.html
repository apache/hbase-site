020import com.google.errorprone.annotations.RestrictedApi;
021import java.util.ArrayList;
022import java.util.HashMap;
023import java.util.LinkedHashMap;
024import java.util.Map.Entry;
025import java.util.concurrent.ScheduledFuture;
026import java.util.concurrent.ScheduledThreadPoolExecutor;
027import java.util.concurrent.ThreadFactory;
028import java.util.concurrent.atomic.AtomicInteger;
029import org.apache.hadoop.hbase.trace.TraceUtil;
030import org.apache.yetus.audience.InterfaceAudience;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;

034/**
035 * ChoreService is a service that can be used to schedule instances of {@link ScheduledChore} to run
036 * periodically while sharing threads. The ChoreService is backed by a
037 * {@link ScheduledThreadPoolExecutor} whose core pool size changes dynamically depending on the
038 * number of {@link ScheduledChore} scheduled. All of the threads in the core thread pool of the
039 * underlying {@link ScheduledThreadPoolExecutor} are set to be daemon threads.
040 * <p>
041 * The ChoreService provides the ability to schedule, cancel, and trigger instances of
042 * {@link ScheduledChore}. The ChoreService also provides the ability to check on the status of
043 * scheduled chores. The number of threads used by the ChoreService changes based on the scheduling
044 * load and whether or not the scheduled chores are executing on time. As more chores are scheduled,
045 * there may be a need to increase the number of threads if it is noticed that chores are no longer
046 * meeting their scheduled start times. On the other hand, as chores are cancelled, an attempt is
047 * made to reduce the number of running threads to see if chores can still meet their start times
048 * with a smaller thread pool.
049 * <p>
050 * When finished with a ChoreService it is good practice to call {@link ChoreService#shutdown()}.
051 * Calling this method ensures that all scheduled chores are cancelled and cleaned up properly.
052 */
053@InterfaceAudience.Private
054public class ChoreService {
055  private static final Logger LOG = LoggerFactory.getLogger(ChoreService.class);
056
057  /**
058   * The minimum number of threads in the core pool of the underlying ScheduledThreadPoolExecutor
059   */
060  @InterfaceAudience.Private
061  public final static int MIN_CORE_POOL_SIZE = 1;
062
063  /**
064   * This thread pool is used to schedule all of the Chores
065   */
066  private final ScheduledThreadPoolExecutor scheduler;
067
068  /**
069   * Maps chores to their futures. Futures are used to control a chore's schedule
070   */
071  private final HashMap<ScheduledChore, ScheduledFuture<?>> scheduledChores;
072
073  /**
074   * Maps chores to Booleans which indicate whether or not a chore has caused an increase in the
075   * core pool size of the ScheduledThreadPoolExecutor. Each chore should only be allowed to
076   * increase the core pool size by 1 (otherwise a single long running chore whose execution is
077   * longer than its period would be able to spawn too many threads).
078   */
079  private final HashMap<ScheduledChore, Boolean> choresMissingStartTime;
080
081  /**
082   * The coreThreadPoolPrefix is the prefix that will be applied to all threads within the
083   * ScheduledThreadPoolExecutor. The prefix is typically related to the Server that the service is
084   * running on. The prefix is useful because it allows us to monitor how the thread pool of a
085   * particular service changes over time VIA thread dumps.
086   */
087  private final String coreThreadPoolPrefix;
088
089  /**
090   * @param coreThreadPoolPrefix Prefix that will be applied to the Thread name of all threads
091   *                             spawned by this service
092   */
093  @InterfaceAudience.Private
094  public ChoreService(final String coreThreadPoolPrefix) {
095    this(coreThreadPoolPrefix, MIN_CORE_POOL_SIZE, false);
096  }
097
098  /**
099   * @param coreThreadPoolPrefix Prefix that will be applied to the Thread name of all threads
100   *                             spawned by this service
101   * @param jitter               Should chore service add some jitter for all of the scheduled
102   *                             chores. When set to true this will add -10% to 10% jitter.
103   */
104  public ChoreService(final String coreThreadPoolPrefix, final boolean jitter) {
105    this(coreThreadPoolPrefix, MIN_CORE_POOL_SIZE, jitter);
106  }
107
108  /**
109   * @param coreThreadPoolPrefix Prefix that will be applied to the Thread name of all threads
110   *                             spawned by this service
111   * @param corePoolSize         The initial size to set the core pool of the
112   *                             ScheduledThreadPoolExecutor to during initialization. The default
113   *                             size is 1, but specifying a larger size may be beneficial if you
114   *                             know that 1 thread will not be enough.
115   * @param jitter               Should chore service add some jitter for all of the scheduled
116   *                             chores. When set to true this will add -10% to 10% jitter.
117   */
118  public ChoreService(final String coreThreadPoolPrefix, int corePoolSize, boolean jitter) {
119    this.coreThreadPoolPrefix = coreThreadPoolPrefix;
120    if (corePoolSize < MIN_CORE_POOL_SIZE) {
121      corePoolSize = MIN_CORE_POOL_SIZE;
122    }
123
124    final ThreadFactory threadFactory = new ChoreServiceThreadFactory(coreThreadPoolPrefix);
125    if (jitter) {
126      scheduler = new JitterScheduledThreadPoolExecutorImpl(corePoolSize, threadFactory, 0.1);
127    } else {
128      scheduler = new ScheduledThreadPoolExecutor(corePoolSize, threadFactory);
129    }
130
131    scheduler.setRemoveOnCancelPolicy(true);
132    scheduledChores = new HashMap<>();
133    choresMissingStartTime = new HashMap<>();
134  }
135
136  /**
137   * @param chore Chore to be scheduled. If the chore is already scheduled with another ChoreService
138   *              instance, that schedule will be cancelled (i.e. a Chore can only ever be scheduled
139   *              with a single ChoreService instance).
140   * @return true when the chore was successfully scheduled. false when the scheduling failed
141   *         (typically occurs when a chore is scheduled during shutdown of service)
142   */
143  public boolean scheduleChore(ScheduledChore chore) {
144    if (chore == null) {
145      return false;
146    }
147    // always lock chore first to prevent dead lock
148    synchronized (chore) {
149      synchronized (this) {
150        try {
151          // Chores should only ever be scheduled with a single ChoreService. If the choreService
152          // is changing, cancel any existing schedules of this chore.
153          if (chore.getChoreService() == this) {
154            LOG.warn("Chore {} has already been scheduled with us", chore);
155            return false;
156          }
157          if (chore.getPeriod() <= 0) {
158            LOG.info("Chore {} is disabled because its period is not positive.", chore);
159            return false;
160          }
161          LOG.info("Chore {} is enabled.", chore);
162          if (chore.getChoreService() != null) {
163            LOG.info("Cancel chore {} from its previous service", chore);
164            chore.getChoreService().cancelChore(chore);
165          }
166          chore.setChoreService(this);
167          ScheduledFuture<?> future =
168            scheduler.scheduleAtFixedRate(TraceUtil.tracedRunnable(chore, chore.getName()),
169              chore.getInitialDelay(), chore.getPeriod(), chore.getTimeUnit());
170          scheduledChores.put(chore, future);
171          return true;
172        } catch (Exception e) {
173          LOG.error("Could not successfully schedule chore: {}", chore.getName(), e);
174          return false;
175        }
176      }
177    }
178  }
179
180  /**
181   * @param chore The Chore to be rescheduled. If the chore is not scheduled with this ChoreService
182   *              yet then this call is equivalent to a call to scheduleChore.
183   */
184  private void rescheduleChore(ScheduledChore chore) {
185    if (scheduledChores.containsKey(chore)) {
186      ScheduledFuture<?> future = scheduledChores.get(chore);
187      future.cancel(false);
188    }
189    ScheduledFuture<?> future = scheduler.scheduleAtFixedRate(chore, chore.getInitialDelay(),
190      chore.getPeriod(), chore.getTimeUnit());
191    scheduledChores.put(chore, future);
192  }
193
194  /**
195   * Cancel any ongoing schedules that this chore has with the implementer of this interface.
196   * <p/>
197   * Call {@link ScheduledChore#cancel()} to cancel a {@link ScheduledChore}, in
198   * {@link ScheduledChore#cancel()} method we will call this method to remove the
199   * {@link ScheduledChore} from this {@link ChoreService}.
200   */
201  @RestrictedApi(explanation = "Should only be called in ScheduledChore", link = "",
202      allowedOnPath = ".*/org/apache/hadoop/hbase/(ScheduledChore|ChoreService).java")
203  synchronized void cancelChore(ScheduledChore chore) {
204    cancelChore(chore, true);
205  }
206
207  /**
208   * Cancel any ongoing schedules that this chore has with the implementer of this interface.
209   * <p/>
210   * Call {@link ScheduledChore#cancel(boolean)} to cancel a {@link ScheduledChore}, in
211   * {@link ScheduledChore#cancel(boolean)} method we will call this method to remove the
212   * {@link ScheduledChore} from this {@link ChoreService}.
213   */
214  @RestrictedApi(explanation = "Should only be called in ScheduledChore", link = "",
215      allowedOnPath = ".*/org/apache/hadoop/hbase/(ScheduledChore|ChoreService).java")
216  synchronized void cancelChore(ScheduledChore chore, boolean mayInterruptIfRunning) {
217    if (scheduledChores.containsKey(chore)) {
218      ScheduledFuture<?> future = scheduledChores.get(chore);
219      future.cancel(mayInterruptIfRunning);
220      scheduledChores.remove(chore);
221
222      // Removing a chore that was missing its start time means it may be possible
223      // to reduce the number of threads
224      if (choresMissingStartTime.containsKey(chore)) {
225        choresMissingStartTime.remove(chore);
226        requestCorePoolDecrease();
227      }
228    }
229  }
230
231  /** Returns true when the chore is scheduled with the implementer of this interface */
232  @InterfaceAudience.Private
233  public synchronized boolean isChoreScheduled(ScheduledChore chore) {
234    return chore != null && scheduledChores.containsKey(chore)
235      && !scheduledChores.get(chore).isDone();
236  }
237
238  /**
239   * This method tries to execute the chore immediately. If the chore is executing at the time of
240   * this call, the chore will begin another execution as soon as the current execution finishes
241   */
242  @RestrictedApi(explanation = "Should only be called in ScheduledChore", link = "",
243      allowedOnPath = ".*/org/apache/hadoop/hbase/ScheduledChore.java")
244  synchronized void triggerNow(ScheduledChore chore) {
245    assert chore.getChoreService() == this;
246    rescheduleChore(chore);
247  }
248
249  /** Returns number of chores that this service currently has scheduled */
250  int getNumberOfScheduledChores() {
251    return scheduledChores.size();
252  }
253
254  /**
255   * @return number of chores that this service currently has scheduled that are missing their
256   *         scheduled start time
257   */
258  int getNumberOfChoresMissingStartTime() {
259    return choresMissingStartTime.size();
260  }
261
262  /** Returns number of threads in the core pool of the underlying ScheduledThreadPoolExecutor */
263  int getCorePoolSize() {
264    return scheduler.getCorePoolSize();
265  }
266
267  /**
268   * Custom ThreadFactory used with the ScheduledThreadPoolExecutor so that all the threads are
269   * daemon threads, and thus, don't prevent the JVM from shutting down
270   */
271  static class ChoreServiceThreadFactory implements ThreadFactory {
272    private final String threadPrefix;
273    private final static String THREAD_NAME_SUFFIX = ".Chore.";
274    private AtomicInteger threadNumber = new AtomicInteger(1);
275
276    /**
277     * @param threadPrefix The prefix given to all threads created by this factory
278     */
279    public ChoreServiceThreadFactory(final String threadPrefix) {
280      this.threadPrefix = threadPrefix;
281    }
282
283    @Override
284    public Thread newThread(Runnable r) {
285      Thread thread =
286        new Thread(r, threadPrefix + THREAD_NAME_SUFFIX + threadNumber.getAndIncrement());
287      thread.setDaemon(true);
288      return thread;
289    }
290  }
291
292  /**
293   * Represents a request to increase the number of core pool threads. Typically a request
294   * originates from the fact that the current core pool size is not sufficient to service all of
295   * the currently running Chores
296   * @return true when the request to increase the core pool size succeeds
297   */
298  private synchronized boolean requestCorePoolIncrease() {
299    // There is no point in creating more threads than scheduledChores.size since scheduled runs
300    // of the same chore cannot run concurrently (i.e. happen-before behavior is enforced
301    // amongst occurrences of the same chore).
302    if (scheduler.getCorePoolSize() < scheduledChores.size()) {
303      scheduler.setCorePoolSize(scheduler.getCorePoolSize() + 1);
304      printChoreServiceDetails("requestCorePoolIncrease");
305      return true;
306    }
307    return false;
308  }
309
310  /**
311   * Represents a request to decrease the number of core pool threads. Typically a request
312   * originates from the fact that the current core pool size is more than sufficient to service the
313   * running Chores.
314   */
315  private synchronized void requestCorePoolDecrease() {
316    if (scheduler.getCorePoolSize() > MIN_CORE_POOL_SIZE) {
317      scheduler.setCorePoolSize(scheduler.getCorePoolSize() - 1);
318      printChoreServiceDetails("requestCorePoolDecrease");
319    }
320  }
321
322  /**
323   * A callback that tells the implementer of this interface that one of the scheduled chores is
324   * missing its start time. The implication of a chore missing its start time is that the service's
325   * current means of scheduling may not be sufficient to handle the number of ongoing chores (the
326   * other explanation is that the chore's execution time is greater than its scheduled period). The
327   * service should try to increase its concurrency when this callback is received.
328   * @param chore The chore that missed its start time
329   */
330  @RestrictedApi(explanation = "Should only be called in ScheduledChore", link = "",
331      allowedOnPath = ".*/org/apache/hadoop/hbase/ScheduledChore.java")
332  synchronized void onChoreMissedStartTime(ScheduledChore chore) {
333    if (!scheduledChores.containsKey(chore)) {
334      return;
335    }
336
337    // If the chore has not caused an increase in the size of the core thread pool then request an
338    // increase. This allows each chore missing its start time to increase the core pool size by
339    // at most 1.
340    if (!choresMissingStartTime.containsKey(chore) || !choresMissingStartTime.get(chore)) {
341      choresMissingStartTime.put(chore, requestCorePoolIncrease());
342    }
343
344    // Must reschedule the chore to prevent unnecessary delays of chores in the scheduler. If
345    // the chore is NOT rescheduled, future executions of this chore will be delayed more and
346    // more on each iteration. This hurts us because the ScheduledThreadPoolExecutor allocates
347    // idle threads to chores based on how delayed they are.
348    rescheduleChore(chore);
349    printChoreDetails("onChoreMissedStartTime", chore);
350  }
351
352  /**
353   * shutdown the service. Any chores that are scheduled for execution will be cancelled. Any chores
354   * in the middle of execution will be interrupted and shutdown. This service will be unusable
355   * after this method has been called (i.e. future scheduling attempts will fail).
356   * <p/>
357   * Notice that, this will only clean the chore from this ChoreService but you could still schedule
358   * the chore with other ChoreService.
359   */
360  public synchronized void shutdown() {
361    if (isShutdown()) {
362      return;
363    }
364    scheduler.shutdownNow();
365    LOG.info("Chore service for: {} had {} on shutdown", coreThreadPoolPrefix,
366      scheduledChores.keySet());
367    cancelAllChores(true);
368    scheduledChores.clear();
369    choresMissingStartTime.clear();
370  }
371
372  /** Returns true when the service is shutdown and thus cannot be used anymore */
373  public boolean isShutdown() {
374    return scheduler.isShutdown();
375  }
376
377  /** Returns true when the service is shutdown and all threads have terminated */
378  public boolean isTerminated() {
379    return scheduler.isTerminated();
380  }
381
382  private void cancelAllChores(final boolean mayInterruptIfRunning) {
383    // Build list of chores to cancel so we can iterate through a set that won't change
384    // as chores are cancelled. If we tried to cancel each chore while iterating through
385    // keySet the results would be undefined because the keySet would be changing
386    ArrayList<ScheduledChore> choresToCancel = new ArrayList<>(scheduledChores.keySet());
387
388    for (ScheduledChore chore : choresToCancel) {
389      cancelChore(chore, mayInterruptIfRunning);
390    }
391  }

393  /**
394   * Prints a summary of important details about the chore. Used for debugging purposes
395   */
396  private void printChoreDetails(final String header, ScheduledChore chore) {
397    if (!LOG.isTraceEnabled()) {
398      return;
399    }
400    LinkedHashMap<String, String> output = new LinkedHashMap<>();
401    output.put(header, "");
402    output.put("Chore name: ", chore.getName());
403    output.put("Chore period: ", Integer.toString(chore.getPeriod()));
404    output.put("Chore timeBetweenRuns: ", Long.toString(chore.getTimeBetweenRuns()));
405
406    for (Entry<String, String> entry : output.entrySet()) {
407      LOG.trace(entry.getKey() + entry.getValue());
408    }
409  }
410
411  /**
412   * Prints a summary of important details about the service. Used for debugging purposes
413   */
414  private void printChoreServiceDetails(final String header) {
415    if (!LOG.isTraceEnabled()) {
416      return;
417    }
418    LinkedHashMap<String, String> output = new LinkedHashMap<>();
419    output.put(header, "");
420    output.put("ChoreService corePoolSize: ", Integer.toString(getCorePoolSize()));
421    output.put("ChoreService scheduledChores: ", Integer.toString(getNumberOfScheduledChores()));
422    output.put("ChoreService missingStartTimeCount: ",
423      Integer.toString(getNumberOfChoresMissingStartTime()));
424
425    for (Entry<String, String> entry : output.entrySet()) {
426      LOG.trace(entry.getKey() + entry.getValue());
427    }
428  }
