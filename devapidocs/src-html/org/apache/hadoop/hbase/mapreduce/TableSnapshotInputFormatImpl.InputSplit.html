089  /**
090   * Implementation class for InputSplit logic common between mapred and mapreduce.
091   */
092  public static class InputSplit implements Writable {
093
094    private TableDescriptor htd;
095    private HRegionInfo regionInfo;
096    private String[] locations;
097    private String scan;
098    private String restoreDir;
099
100    // constructor for mapreduce framework / Writable
101    public InputSplit() {}
102
103    public InputSplit(TableDescriptor htd, HRegionInfo regionInfo, List<String> locations,
104        Scan scan, Path restoreDir) {
105      this.htd = htd;
106      this.regionInfo = regionInfo;
107      if (locations == null || locations.isEmpty()) {
108        this.locations = new String[0];
109      } else {
110        this.locations = locations.toArray(new String[locations.size()]);
111      }
112      try {
113        this.scan = scan != null ? TableMapReduceUtil.convertScanToString(scan) : "";
114      } catch (IOException e) {
115        LOG.warn("Failed to convert Scan to String", e);
116      }
117
118      this.restoreDir = restoreDir.toString();
119    }
120
121    public TableDescriptor getHtd() {
122      return htd;
123    }
124
125    public String getScan() {
126      return scan;
127    }
128
129    public String getRestoreDir() {
130      return restoreDir;
131    }
132
133    public long getLength() {
134      //TODO: We can obtain the file sizes of the snapshot here.
135      return 0;
136    }
137
138    public String[] getLocations() {
139      return locations;
140    }
141
142    public TableDescriptor getTableDescriptor() {
143      return htd;
144    }
145
146    public HRegionInfo getRegionInfo() {
147      return regionInfo;
148    }
149
150    // TODO: We should have ProtobufSerialization in Hadoop, and directly use PB objects instead of
151    // doing this wrapping with Writables.
152    @Override
153    public void write(DataOutput out) throws IOException {
154      TableSnapshotRegionSplit.Builder builder = TableSnapshotRegionSplit.newBuilder()
155          .setTable(ProtobufUtil.toTableSchema(htd))
156          .setRegion(HRegionInfo.convert(regionInfo));
157
158      for (String location : locations) {
159        builder.addLocations(location);
160      }
161
162      TableSnapshotRegionSplit split = builder.build();
163
164      ByteArrayOutputStream baos = new ByteArrayOutputStream();
165      split.writeTo(baos);
166      baos.close();
167      byte[] buf = baos.toByteArray();
168      out.writeInt(buf.length);
169      out.write(buf);
170
171      Bytes.writeByteArray(out, Bytes.toBytes(scan));
172      Bytes.writeByteArray(out, Bytes.toBytes(restoreDir));
173
174    }
175
176    @Override
177    public void readFields(DataInput in) throws IOException {
178      int len = in.readInt();
179      byte[] buf = new byte[len];
180      in.readFully(buf);
181      TableSnapshotRegionSplit split = TableSnapshotRegionSplit.PARSER.parseFrom(buf);
182      this.htd = ProtobufUtil.toTableDescriptor(split.getTable());
183      this.regionInfo = HRegionInfo.convert(split.getRegion());
184      List<String> locationsList = split.getLocationsList();
185      this.locations = locationsList.toArray(new String[locationsList.size()]);
186
187      this.scan = Bytes.toString(Bytes.readByteArray(in));
188      this.restoreDir = Bytes.toString(Bytes.readByteArray(in));
189    }
190  }

192  /**
193   * Implementation class for RecordReader logic common between mapred and mapreduce.
194   */
195  public static class RecordReader {
196    private InputSplit split;
197    private Scan scan;
198    private Result result = null;
199    private ImmutableBytesWritable row = null;
200    private ClientSideRegionScanner scanner;
201
202    public ClientSideRegionScanner getScanner() {
203      return scanner;
204    }
205
206    public void initialize(InputSplit split, Configuration conf) throws IOException {
207      this.scan = TableMapReduceUtil.convertStringToScan(split.getScan());
208      this.split = split;
209      TableDescriptor htd = split.htd;
210      HRegionInfo hri = this.split.getRegionInfo();
211      FileSystem fs = FSUtils.getCurrentFileSystem(conf);
212
213
214      // region is immutable, this should be fine,
215      // otherwise we have to set the thread read point
216      scan.setIsolationLevel(IsolationLevel.READ_UNCOMMITTED);
217      // disable caching of data blocks
218      scan.setCacheBlocks(false);
219
220      scanner =
221          new ClientSideRegionScanner(conf, fs, new Path(split.restoreDir), htd, hri, scan, null);
222    }
223
224    public boolean nextKeyValue() throws IOException {
225      result = scanner.next();
226      if (result == null) {
227        //we are done
228        return false;
229      }
230
231      if (this.row == null) {
232        this.row = new ImmutableBytesWritable();
233      }
234      this.row.set(result.getRow());
235      return true;
236    }
237
238    public ImmutableBytesWritable getCurrentKey() {
239      return row;
240    }
241
242    public Result getCurrentValue() {
243      return result;
244    }
245
246    public long getPos() {
247      return 0;
248    }
249
250    public float getProgress() {
251      return 0; // TODO: use total bytes to estimate
252    }
253
254    public void close() {
255      if (this.scanner != null) {
256        this.scanner.close();
257      }
258    }
259  }

261  public static List<InputSplit> getSplits(Configuration conf) throws IOException {
262    String snapshotName = getSnapshotName(conf);
263
264    Path rootDir = FSUtils.getRootDir(conf);
265    FileSystem fs = rootDir.getFileSystem(conf);
266
267    SnapshotManifest manifest = getSnapshotManifest(conf, snapshotName, rootDir, fs);
268
269    List<HRegionInfo> regionInfos = getRegionInfosFromManifest(manifest);
270
271    // TODO: mapred does not support scan as input API. Work around for now.
272    Scan scan = extractScanFromConf(conf);
273    // the temp dir where the snapshot is restored
274    Path restoreDir = new Path(conf.get(RESTORE_DIR_KEY));
275
276    RegionSplitter.SplitAlgorithm splitAlgo = getSplitAlgo(conf);
277
278    int numSplits = conf.getInt(NUM_SPLITS_PER_REGION, 1);
279
280    return getSplits(scan, manifest, regionInfos, restoreDir, conf, splitAlgo, numSplits);
281  }
282
283  public static RegionSplitter.SplitAlgorithm getSplitAlgo(Configuration conf) throws IOException{
284    String splitAlgoClassName = conf.get(SPLIT_ALGO);
285    if (splitAlgoClassName == null)
286      return null;
287    try {
288      return ((Class<? extends RegionSplitter.SplitAlgorithm>)
289              Class.forName(splitAlgoClassName)).newInstance();
290    } catch (ClassNotFoundException e) {
291      throw new IOException("SplitAlgo class " + splitAlgoClassName +
292              " is not found", e);
293    } catch (InstantiationException e) {
294      throw new IOException("SplitAlgo class " + splitAlgoClassName +
295              " is not instantiable", e);
296    } catch (IllegalAccessException e) {
297      throw new IOException("SplitAlgo class " + splitAlgoClassName +
298              " is not instantiable", e);
299    }
300  }

303  public static List<HRegionInfo> getRegionInfosFromManifest(SnapshotManifest manifest) {
304    List<SnapshotRegionManifest> regionManifests = manifest.getRegionManifests();
305    if (regionManifests == null) {
306      throw new IllegalArgumentException("Snapshot seems empty");
307    }
308
309    List<HRegionInfo> regionInfos = Lists.newArrayListWithCapacity(regionManifests.size());
310
311    for (SnapshotRegionManifest regionManifest : regionManifests) {
312      HRegionInfo hri = HRegionInfo.convert(regionManifest.getRegionInfo());
313      if (hri.isOffline() && (hri.isSplit() || hri.isSplitParent())) {
314        continue;
315      }
316      regionInfos.add(hri);
317    }
318    return regionInfos;
319  }

321  public static SnapshotManifest getSnapshotManifest(Configuration conf, String snapshotName,
322      Path rootDir, FileSystem fs) throws IOException {
323    Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
324    SnapshotDescription snapshotDesc = SnapshotDescriptionUtils.readSnapshotInfo(fs, snapshotDir);
325    return SnapshotManifest.open(conf, fs, snapshotDir, snapshotDesc);

328  public static Scan extractScanFromConf(Configuration conf) throws IOException {
329    Scan scan = null;
330    if (conf.get(TableInputFormat.SCAN) != null) {
331      scan = TableMapReduceUtil.convertStringToScan(conf.get(TableInputFormat.SCAN));
332    } else if (conf.get(org.apache.hadoop.hbase.mapred.TableInputFormat.COLUMN_LIST) != null) {
333      String[] columns =
334        conf.get(org.apache.hadoop.hbase.mapred.TableInputFormat.COLUMN_LIST).split(" ");
335      scan = new Scan();
336      for (String col : columns) {
337        scan.addFamily(Bytes.toBytes(col));
338      }
339    } else {
340      throw new IllegalArgumentException("Unable to create scan");
341    }
342    return scan;
343  }

345  public static List<InputSplit> getSplits(Scan scan, SnapshotManifest manifest,
346      List<HRegionInfo> regionManifests, Path restoreDir, Configuration conf) throws IOException {
347    return getSplits(scan, manifest, regionManifests, restoreDir, conf, null, 1);

350  public static List<InputSplit> getSplits(Scan scan, SnapshotManifest manifest,
351      List<HRegionInfo> regionManifests, Path restoreDir,
352      Configuration conf, RegionSplitter.SplitAlgorithm sa, int numSplits) throws IOException {
353    // load table descriptor
354    TableDescriptor htd = manifest.getTableDescriptor();
355
356    Path tableDir = FSUtils.getTableDir(restoreDir, htd.getTableName());
357
358    List<InputSplit> splits = new ArrayList<>();
359    for (HRegionInfo hri : regionManifests) {
360      // load region descriptor
361
362      if (numSplits > 1) {
363        byte[][] sp = sa.split(hri.getStartKey(), hri.getEndKey(), numSplits, true);
364        for (int i = 0; i < sp.length - 1; i++) {
365          if (CellUtil.overlappingKeys(scan.getStartRow(), scan.getStopRow(), sp[i],
366                  sp[i + 1])) {
367            // compute HDFS locations from snapshot files (which will get the locations for
368            // referred hfiles)
369            List<String> hosts = getBestLocations(conf,
370                    HRegion.computeHDFSBlocksDistribution(conf, htd, hri, tableDir));
371
372            int len = Math.min(3, hosts.size());
373            hosts = hosts.subList(0, len);
374            Scan boundedScan = new Scan(scan);
375            boundedScan.setStartRow(sp[i]);
376            boundedScan.setStopRow(sp[i + 1]);
377            splits.add(new InputSplit(htd, hri, hosts, boundedScan, restoreDir));
378          }
379        }
380      } else {
381        if (CellUtil.overlappingKeys(scan.getStartRow(), scan.getStopRow(), hri.getStartKey(),
382                hri.getEndKey())) {
383          // compute HDFS locations from snapshot files (which will get the locations for
384          // referred hfiles)
385          List<String> hosts = getBestLocations(conf,
386                  HRegion.computeHDFSBlocksDistribution(conf, htd, hri, tableDir));
387
388          int len = Math.min(3, hosts.size());
389          hosts = hosts.subList(0, len);
390          splits.add(new InputSplit(htd, hri, hosts, scan, restoreDir));
391        }
392      }
393    }
394
395    return splits;
396
397  }

399  /**
400   * This computes the locations to be passed from the InputSplit. MR/Yarn schedulers does not take
401   * weights into account, thus will treat every location passed from the input split as equal. We
402   * do not want to blindly pass all the locations, since we are creating one split per region, and
403   * the region's blocks are all distributed throughout the cluster unless favorite node assignment
404   * is used. On the expected stable case, only one location will contain most of the blocks as
405   * local.
406   * On the other hand, in favored node assignment, 3 nodes will contain highly local blocks. Here
407   * we are doing a simple heuristic, where we will pass all hosts which have at least 80%
408   * (hbase.tablesnapshotinputformat.locality.cutoff.multiplier) as much block locality as the top
409   * host with the best locality.
410   */
411  public static List<String> getBestLocations(
412      Configuration conf, HDFSBlocksDistribution blockDistribution) {
413    List<String> locations = new ArrayList<>(3);
414
415    HostAndWeight[] hostAndWeights = blockDistribution.getTopHostsWithWeights();
416
417    if (hostAndWeights.length == 0) {
418      return locations;
419    }
420
421    HostAndWeight topHost = hostAndWeights[0];
422    locations.add(topHost.getHost());
423
424    // Heuristic: filter all hosts which have at least cutoffMultiplier % of block locality
425    double cutoffMultiplier
426            = conf.getFloat(LOCALITY_CUTOFF_MULTIPLIER, DEFAULT_LOCALITY_CUTOFF_MULTIPLIER);
427
428    double filterWeight = topHost.getWeight() * cutoffMultiplier;
429
430    for (int i = 1; i < hostAndWeights.length; i++) {
431      if (hostAndWeights[i].getWeight() >= filterWeight) {
432        locations.add(hostAndWeights[i].getHost());
433      } else {
434        break;
435      }
436    }
437
438    return locations;
439  }

450   * Configures the job to use TableSnapshotInputFormat to read from a snapshot.
451   * @param conf the job to configuration
452   * @param snapshotName the name of the snapshot to read from
453   * @param restoreDir a temporary directory to restore the snapshot into. Current user should
454   * have write permissions to this directory, and this should not be a subdirectory of rootdir.
455   * After the job is finished, restoreDir can be deleted.
456   * @throws IOException if an error occurs

458  public static void setInput(Configuration conf, String snapshotName, Path restoreDir)
459      throws IOException {
460    setInput(conf, snapshotName, restoreDir, null, 1);
461  }

463  /**
464   * Configures the job to use TableSnapshotInputFormat to read from a snapshot.
465   * @param conf the job to configure
466   * @param snapshotName the name of the snapshot to read from
467   * @param restoreDir a temporary directory to restore the snapshot into. Current user should
468   * have write permissions to this directory, and this should not be a subdirectory of rootdir.
469   * After the job is finished, restoreDir can be deleted.
470   * @param numSplitsPerRegion how many input splits to generate per one region
471   * @param splitAlgo SplitAlgorithm to be used when generating InputSplits
472   * @throws IOException if an error occurs
473   */
474  public static void setInput(Configuration conf, String snapshotName, Path restoreDir,
475                              RegionSplitter.SplitAlgorithm splitAlgo, int numSplitsPerRegion)
476          throws IOException {
477    conf.set(SNAPSHOT_NAME_KEY, snapshotName);
478    if (numSplitsPerRegion < 1) {
479      throw new IllegalArgumentException("numSplits must be >= 1, " +
480              "illegal numSplits : " + numSplitsPerRegion);
481    }
482    if (splitAlgo == null && numSplitsPerRegion > 1) {
483      throw new IllegalArgumentException("Split algo can't be null when numSplits > 1");
484    }
485    if (splitAlgo != null) {
486      conf.set(SPLIT_ALGO, splitAlgo.getClass().getName());
487    }
488    conf.setInt(NUM_SPLITS_PER_REGION, numSplitsPerRegion);
489    Path rootDir = FSUtils.getRootDir(conf);
490    FileSystem fs = rootDir.getFileSystem(conf);
491
492    restoreDir = new Path(restoreDir, UUID.randomUUID().toString());
493
494    // TODO: restore from record readers to parallelize.
495    RestoreSnapshotHelper.copySnapshotForScanner(conf, fs, rootDir, restoreDir, snapshotName);
496
497    conf.set(RESTORE_DIR_KEY, restoreDir.toString());
498  }
