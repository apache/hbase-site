095 * Writes HFiles. Passed Cells must arrive in order.
096 * Writes current time as the sequence id for the file. Sets the major compacted
097 * attribute on created @{link {@link HFile}s. Calling write(null,null) will forcibly roll

186  static <V extends Cell> RecordWriter<ImmutableBytesWritable, V> createRecordWriter(
187    final TaskAttemptContext context, final OutputCommitter committer) throws IOException {
188
189    // Get the path of the temporary output file
190    final Path outputDir = ((FileOutputCommitter)committer).getWorkPath();
191    final Configuration conf = context.getConfiguration();
192    final boolean writeMultipleTables =
193      conf.getBoolean(MULTI_TABLE_HFILEOUTPUTFORMAT_CONF_KEY, false);
194    final String writeTableNames = conf.get(OUTPUT_TABLE_NAME_CONF_KEY);
195    if (writeTableNames == null || writeTableNames.isEmpty()) {
196      throw new IllegalArgumentException("" + OUTPUT_TABLE_NAME_CONF_KEY + " cannot be empty");
197    }
198    final FileSystem fs = outputDir.getFileSystem(conf);
199    // These configs. are from hbase-*.xml
200    final long maxsize = conf.getLong(HConstants.HREGION_MAX_FILESIZE,
201        HConstants.DEFAULT_MAX_FILE_SIZE);
202    // Invented config.  Add to hbase-*.xml if other than default compression.
203    final String defaultCompressionStr = conf.get("hfile.compression",
204        Compression.Algorithm.NONE.getName());
205    final Algorithm defaultCompression = HFileWriterImpl.compressionByName(defaultCompressionStr);
206    String compressionStr = conf.get(COMPRESSION_OVERRIDE_CONF_KEY);
207    final Algorithm overriddenCompression = compressionStr != null ?
208      Compression.getCompressionAlgorithmByName(compressionStr): null;
209    final boolean compactionExclude = conf.getBoolean(
210        "hbase.mapreduce.hfileoutputformat.compaction.exclude", false);
211    final Set<String> allTableNames = Arrays.stream(writeTableNames.split(
212            Bytes.toString(tableSeparator))).collect(Collectors.toSet());
213
214    // create a map from column family to the compression algorithm
215    final Map<byte[], Algorithm> compressionMap = createFamilyCompressionMap(conf);
216    final Map<byte[], BloomType> bloomTypeMap = createFamilyBloomTypeMap(conf);
217    final Map<byte[], String> bloomParamMap = createFamilyBloomParamMap(conf);
218    final Map<byte[], Integer> blockSizeMap = createFamilyBlockSizeMap(conf);
219
220    String dataBlockEncodingStr = conf.get(DATABLOCK_ENCODING_OVERRIDE_CONF_KEY);
221    final Map<byte[], DataBlockEncoding> datablockEncodingMap
222        = createFamilyDataBlockEncodingMap(conf);
223    final DataBlockEncoding overriddenEncoding = dataBlockEncodingStr != null ?
224      DataBlockEncoding.valueOf(dataBlockEncodingStr) : null;
225
226    return new RecordWriter<ImmutableBytesWritable, V>() {
227      // Map of families to writers and how much has been output on the writer.
228      private final Map<byte[], WriterLength> writers = new TreeMap<>(Bytes.BYTES_COMPARATOR);
229      private final Map<byte[], byte[]> previousRows = new TreeMap<>(Bytes.BYTES_COMPARATOR);
230      private final long now = EnvironmentEdgeManager.currentTime();
231      private byte[] tableNameBytes = writeMultipleTables ? null : Bytes.toBytes(writeTableNames);
232
233      @Override
234      public void write(ImmutableBytesWritable row, V cell) throws IOException {
235        Cell kv = cell;
236        // null input == user explicitly wants to flush
237        if (row == null && kv == null) {
238          rollWriters(null);
239          return;
240        }
241
242        byte[] rowKey = CellUtil.cloneRow(kv);
243        int length = (PrivateCellUtil.estimatedSerializedSizeOf(kv)) - Bytes.SIZEOF_INT;
244        byte[] family = CellUtil.cloneFamily(kv);
245        if (writeMultipleTables) {
246          tableNameBytes = MultiTableHFileOutputFormat.getTableName(row.get());
247          tableNameBytes = TableName.valueOf(tableNameBytes).getNameWithNamespaceInclAsString()
248              .getBytes(Charset.defaultCharset());
249          if (!allTableNames.contains(Bytes.toString(tableNameBytes))) {
250            throw new IllegalArgumentException("TableName " + Bytes.toString(tableNameBytes) +
251              " not expected");
252          }
253        }
254        byte[] tableAndFamily = getTableNameSuffixedWithFamily(tableNameBytes, family);
255
256        WriterLength wl = this.writers.get(tableAndFamily);
257
258        // If this is a new column family, verify that the directory exists
259        if (wl == null) {
260          Path writerPath = null;
261          if (writeMultipleTables) {
262            Path tableRelPath = getTableRelativePath(tableNameBytes);
263            writerPath = new Path(outputDir, new Path(tableRelPath, Bytes.toString(family)));
264          } else {
265            writerPath = new Path(outputDir, Bytes.toString(family));
266          }
267          fs.mkdirs(writerPath);
268          configureStoragePolicy(conf, fs, tableAndFamily, writerPath);
269        }
270
271        // This can only happen once a row is finished though
272        if (wl != null && wl.written + length >= maxsize
273                && Bytes.compareTo(this.previousRows.get(family), rowKey) != 0) {
274          rollWriters(wl);
275        }
276
277        // create a new WAL writer, if necessary
278        if (wl == null || wl.writer == null) {
279          InetSocketAddress[] favoredNodes = null;
280          if (conf.getBoolean(LOCALITY_SENSITIVE_CONF_KEY, DEFAULT_LOCALITY_SENSITIVE)) {
281            HRegionLocation loc = null;
282            String tableName = Bytes.toString(tableNameBytes);
283            if (tableName != null) {
284              try (Connection connection = ConnectionFactory.createConnection(
285                createRemoteClusterConf(conf));
286                  RegionLocator locator =
287                      connection.getRegionLocator(TableName.valueOf(tableName))) {
288                loc = locator.getRegionLocation(rowKey);
289              } catch (Throwable e) {
290                LOG.warn("Something wrong locating rowkey {} in {}", Bytes.toString(rowKey),
291                  tableName, e);
292                loc = null;
293              }
294            }
295            if (null == loc) {
296              LOG.trace("Failed get of location, use default writer {}", Bytes.toString(rowKey));
297            } else {
298              LOG.debug("First rowkey: [{}]", Bytes.toString(rowKey));
299              InetSocketAddress initialIsa =
300                  new InetSocketAddress(loc.getHostname(), loc.getPort());
301              if (initialIsa.isUnresolved()) {
302                LOG.trace("Failed resolve address {}, use default writer", loc.getHostnamePort());
303              } else {
304                LOG.debug("Use favored nodes writer: {}", initialIsa.getHostString());
305                favoredNodes = new InetSocketAddress[] { initialIsa };
306              }
307            }
308          }
309          wl = getNewWriter(tableNameBytes, family, conf, favoredNodes);
310
311        }
312
313        // we now have the proper WAL writer. full steam ahead
314        PrivateCellUtil.updateLatestStamp(cell, this.now);
315        wl.writer.append(kv);
316        wl.written += length;
317
318        // Copy the row so we know when a row transition.
319        this.previousRows.put(family, rowKey);
320      }
321
322      private Path getTableRelativePath(byte[] tableNameBytes) {
323        String tableName = Bytes.toString(tableNameBytes);
324        String[] tableNameParts = tableName.split(":");
325        Path tableRelPath = new Path(tableNameParts[0]);
326        if (tableNameParts.length > 1) {
327          tableRelPath = new Path(tableRelPath, tableNameParts[1]);
328        }
329        return tableRelPath;
330      }
331      private void rollWriters(WriterLength writerLength) throws IOException {
332        if (writerLength != null) {
333          closeWriter(writerLength);
334        } else {
335          for (WriterLength wl : this.writers.values()) {
336            closeWriter(wl);
337          }
338        }
339      }
340
341      private void closeWriter(WriterLength wl) throws IOException {
342        if (wl.writer != null) {
343          LOG.info("Writer=" + wl.writer.getPath() +
344            ((wl.written == 0)? "": ", wrote=" + wl.written));
345          close(wl.writer);
346          wl.writer = null;
347        }
348        wl.written = 0;
349      }
350
351      private Configuration createRemoteClusterConf(Configuration conf) {
352        final Configuration newConf = new Configuration(conf);
353
354        final String quorum = conf.get(REMOTE_CLUSTER_ZOOKEEPER_QUORUM_CONF_KEY);
355        final String clientPort = conf.get(REMOTE_CLUSTER_ZOOKEEPER_CLIENT_PORT_CONF_KEY);
356        final String parent = conf.get(REMOTE_CLUSTER_ZOOKEEPER_ZNODE_PARENT_CONF_KEY);
357
358        if (quorum != null && clientPort != null && parent != null) {
359          newConf.set(HConstants.ZOOKEEPER_QUORUM, quorum);
360          newConf.setInt(HConstants.ZOOKEEPER_CLIENT_PORT, Integer.parseInt(clientPort));
361          newConf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, parent);
362        }
363
364        return newConf;
365      }
366
367      /*
368       * Create a new StoreFile.Writer.
369       * @return A WriterLength, containing a new StoreFile.Writer.
370       */
371      @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="BX_UNBOXING_IMMEDIATELY_REBOXED",
372          justification="Not important")
373      private WriterLength getNewWriter(byte[] tableName, byte[] family, Configuration conf,
374          InetSocketAddress[] favoredNodes) throws IOException {
375        byte[] tableAndFamily = getTableNameSuffixedWithFamily(tableName, family);
376        Path familydir = new Path(outputDir, Bytes.toString(family));
377        if (writeMultipleTables) {
378          familydir = new Path(outputDir,
379            new Path(getTableRelativePath(tableName), Bytes.toString(family)));
380        }
381        WriterLength wl = new WriterLength();
382        Algorithm compression = overriddenCompression;
383        compression = compression == null ? compressionMap.get(tableAndFamily) : compression;
384        compression = compression == null ? defaultCompression : compression;
385        BloomType bloomType = bloomTypeMap.get(tableAndFamily);
386        bloomType = bloomType == null ? BloomType.NONE : bloomType;
387        String bloomParam = bloomParamMap.get(tableAndFamily);
388        if (bloomType == BloomType.ROWPREFIX_FIXED_LENGTH) {
389          conf.set(BloomFilterUtil.PREFIX_LENGTH_KEY, bloomParam);
390        }
391        Integer blockSize = blockSizeMap.get(tableAndFamily);
392        blockSize = blockSize == null ? HConstants.DEFAULT_BLOCKSIZE : blockSize;
393        DataBlockEncoding encoding = overriddenEncoding;
394        encoding = encoding == null ? datablockEncodingMap.get(tableAndFamily) : encoding;
395        encoding = encoding == null ? DataBlockEncoding.NONE : encoding;
396        HFileContextBuilder contextBuilder = new HFileContextBuilder().withCompression(compression)
397            .withDataBlockEncoding(encoding).withChecksumType(StoreUtils.getChecksumType(conf))
398            .withBytesPerCheckSum(StoreUtils.getBytesPerChecksum(conf)).withBlockSize(blockSize)
399            .withColumnFamily(family).withTableName(tableName);
400
401        if (HFile.getFormatVersion(conf) >= HFile.MIN_FORMAT_VERSION_WITH_TAGS) {
402          contextBuilder.withIncludesTags(true);
403        }
404
405        HFileContext hFileContext = contextBuilder.build();
406        if (null == favoredNodes) {
407          wl.writer = new StoreFileWriter.Builder(conf, CacheConfig.DISABLED, fs)
408            .withOutputDir(familydir).withBloomType(bloomType)
409            .withFileContext(hFileContext).build();
410        } else {
411          wl.writer = new StoreFileWriter.Builder(conf, CacheConfig.DISABLED, new HFileSystem(fs))
412            .withOutputDir(familydir).withBloomType(bloomType)
413            .withFileContext(hFileContext).withFavoredNodes(favoredNodes).build();
414        }
415
416        this.writers.put(tableAndFamily, wl);
417        return wl;
418      }
419
420      private void close(final StoreFileWriter w) throws IOException {
421        if (w != null) {
422          w.appendFileInfo(BULKLOAD_TIME_KEY, Bytes.toBytes(System.currentTimeMillis()));
423          w.appendFileInfo(BULKLOAD_TASK_KEY, Bytes.toBytes(context.getTaskAttemptID().toString()));
424          w.appendFileInfo(MAJOR_COMPACTION_KEY, Bytes.toBytes(true));
425          w.appendFileInfo(EXCLUDE_FROM_MINOR_COMPACTION_KEY, Bytes.toBytes(compactionExclude));
426          w.appendTrackedTimestampsToMetadata();
427          w.close();
428        }
429      }
430
431      @Override
432      public void close(TaskAttemptContext c) throws IOException, InterruptedException {
433        for (WriterLength wl: this.writers.values()) {
434          close(wl.writer);
435        }
436      }
437    };

463  /**
464   * Return the start keys of all of the regions in this table,
465   * as a list of ImmutableBytesWritable.
466   */
467  private static List<ImmutableBytesWritable> getRegionStartKeys(List<RegionLocator> regionLocators,
468                                                                 boolean writeMultipleTables)
469          throws IOException {
470
471    ArrayList<ImmutableBytesWritable> ret = new ArrayList<>();
472    for(RegionLocator regionLocator : regionLocators) {
473      TableName tableName = regionLocator.getName();
474      LOG.info("Looking up current regions for table " + tableName);
475      byte[][] byteKeys = regionLocator.getStartKeys();
476      for (byte[] byteKey : byteKeys) {
477        byte[] fullKey = byteKey; //HFileOutputFormat2 use case
478        if (writeMultipleTables) {
479          //MultiTableHFileOutputFormat use case
480          fullKey = combineTableNameSuffix(tableName.getName(), byteKey);
481        }
482        if (LOG.isDebugEnabled()) {
483          LOG.debug("SplitPoint startkey for " + tableName + ": " + Bytes.toStringBinary(fullKey));
484        }
485        ret.add(new ImmutableBytesWritable(fullKey));
486      }
487    }
488    return ret;
489  }

491  /**
492   * Write out a {@link SequenceFile} that can be read by
493   * {@link TotalOrderPartitioner} that contains the split points in startKeys.
494   */
495  @SuppressWarnings("deprecation")
496  private static void writePartitions(Configuration conf, Path partitionsPath,
497      List<ImmutableBytesWritable> startKeys, boolean writeMultipleTables) throws IOException {
498    LOG.info("Writing partition information to " + partitionsPath);
499    if (startKeys.isEmpty()) {
500      throw new IllegalArgumentException("No regions passed");
501    }
502
503    // We're generating a list of split points, and we don't ever
504    // have keys < the first region (which has an empty start key)
505    // so we need to remove it. Otherwise we would end up with an
506    // empty reducer with index 0
507    TreeSet<ImmutableBytesWritable> sorted = new TreeSet<>(startKeys);
508    ImmutableBytesWritable first = sorted.first();
509    if (writeMultipleTables) {
510      first =
511        new ImmutableBytesWritable(MultiTableHFileOutputFormat.getSuffix(sorted.first().get()));
512    }
513    if (!first.equals(HConstants.EMPTY_BYTE_ARRAY)) {
514      throw new IllegalArgumentException(
515          "First region of table should have empty start key. Instead has: "
516          + Bytes.toStringBinary(first.get()));
517    }
518    sorted.remove(sorted.first());
519
520    // Write the actual file
521    FileSystem fs = partitionsPath.getFileSystem(conf);
522    SequenceFile.Writer writer = SequenceFile.createWriter(
523      fs, conf, partitionsPath, ImmutableBytesWritable.class,
524      NullWritable.class);
525
526    try {
527      for (ImmutableBytesWritable startKey : sorted) {
528        writer.append(startKey, NullWritable.get());
529      }
530    } finally {
531      writer.close();
532    }
533  }

536   * Configure a MapReduce Job to perform an incremental load into the given
537   * table. This
538   * <ul>
539   *   <li>Inspects the table to configure a total order partitioner</li>
540   *   <li>Uploads the partitions file to the cluster and adds it to the DistributedCache</li>
541   *   <li>Sets the number of reduce tasks to match the current number of regions</li>
542   *   <li>Sets the output key/value class to match HFileOutputFormat2's requirements</li>
543   *   <li>Sets the reducer up to perform the appropriate sorting (either KeyValueSortReducer or
544   *     PutSortReducer)</li>
545   *   <li>Sets the HBase cluster key to load region locations for locality-sensitive</li>
546   * </ul>
547   * The user should be sure to set the map output value class to either KeyValue or Put before
548   * running this function.
549   */
550  public static void configureIncrementalLoad(Job job, Table table, RegionLocator regionLocator)
551      throws IOException {
552    configureIncrementalLoad(job, table.getDescriptor(), regionLocator);
553    configureRemoteCluster(job, table.getConfiguration());

557   * Configure a MapReduce Job to perform an incremental load into the given
558   * table. This
559   * <ul>
560   *   <li>Inspects the table to configure a total order partitioner</li>
561   *   <li>Uploads the partitions file to the cluster and adds it to the DistributedCache</li>
562   *   <li>Sets the number of reduce tasks to match the current number of regions</li>
563   *   <li>Sets the output key/value class to match HFileOutputFormat2's requirements</li>
564   *   <li>Sets the reducer up to perform the appropriate sorting (either KeyValueSortReducer or
565   *     PutSortReducer)</li>
566   * </ul>
567   * The user should be sure to set the map output value class to either KeyValue or Put before
568   * running this function.
569   */
570  public static void configureIncrementalLoad(Job job, TableDescriptor tableDescriptor,
571      RegionLocator regionLocator) throws IOException {
572    ArrayList<TableInfo> singleTableInfo = new ArrayList<>();
573    singleTableInfo.add(new TableInfo(tableDescriptor, regionLocator));
574    configureIncrementalLoad(job, singleTableInfo, HFileOutputFormat2.class);
575  }

577  static void configureIncrementalLoad(Job job, List<TableInfo> multiTableInfo,
578      Class<? extends OutputFormat<?, ?>> cls) throws IOException {
579    Configuration conf = job.getConfiguration();
580    job.setOutputKeyClass(ImmutableBytesWritable.class);
581    job.setOutputValueClass(MapReduceExtendedCell.class);
582    job.setOutputFormatClass(cls);
583
584    if (multiTableInfo.stream().distinct().count() != multiTableInfo.size()) {
585      throw new IllegalArgumentException("Duplicate entries found in TableInfo argument");
586    }
587    boolean writeMultipleTables = false;
588    if (MultiTableHFileOutputFormat.class.equals(cls)) {
589      writeMultipleTables = true;
590      conf.setBoolean(MULTI_TABLE_HFILEOUTPUTFORMAT_CONF_KEY, true);
591    }
592    // Based on the configured map output class, set the correct reducer to properly
593    // sort the incoming values.
594    // TODO it would be nice to pick one or the other of these formats.
595    if (KeyValue.class.equals(job.getMapOutputValueClass())
596        || MapReduceExtendedCell.class.equals(job.getMapOutputValueClass())) {
597      job.setReducerClass(CellSortReducer.class);
598    } else if (Put.class.equals(job.getMapOutputValueClass())) {
599      job.setReducerClass(PutSortReducer.class);
600    } else if (Text.class.equals(job.getMapOutputValueClass())) {
601      job.setReducerClass(TextSortReducer.class);
602    } else {
603      LOG.warn("Unknown map output value type:" + job.getMapOutputValueClass());
604    }
605
606    conf.setStrings("io.serializations", conf.get("io.serializations"),
607        MutationSerialization.class.getName(), ResultSerialization.class.getName(),
608        CellSerialization.class.getName());
609
610    if (conf.getBoolean(LOCALITY_SENSITIVE_CONF_KEY, DEFAULT_LOCALITY_SENSITIVE)) {
611      LOG.info("bulkload locality sensitive enabled");
612    }
613
614    /* Now get the region start keys for every table required */
615    List<String> allTableNames = new ArrayList<>(multiTableInfo.size());
616    List<RegionLocator> regionLocators = new ArrayList<>(multiTableInfo.size());
617    List<TableDescriptor> tableDescriptors = new ArrayList<>(multiTableInfo.size());
618
619    for(TableInfo tableInfo : multiTableInfo) {
620      regionLocators.add(tableInfo.getRegionLocator());
621      String tn = writeMultipleTables?
622        tableInfo.getRegionLocator().getName().getNameWithNamespaceInclAsString():
623        tableInfo.getRegionLocator().getName().getNameAsString();
624      allTableNames.add(tn);
625      tableDescriptors.add(tableInfo.getTableDescriptor());
626    }
627    // Record tablenames for creating writer by favored nodes, and decoding compression,
628    // block size and other attributes of columnfamily per table
629    conf.set(OUTPUT_TABLE_NAME_CONF_KEY, StringUtils.join(allTableNames, Bytes
630            .toString(tableSeparator)));
631    List<ImmutableBytesWritable> startKeys =
632      getRegionStartKeys(regionLocators, writeMultipleTables);
633    // Use table's region boundaries for TOP split points.
634    LOG.info("Configuring " + startKeys.size() + " reduce partitions " +
635        "to match current region count for all tables");
636    job.setNumReduceTasks(startKeys.size());
637
638    configurePartitioner(job, startKeys, writeMultipleTables);
639    // Set compression algorithms based on column families
640
641    conf.set(COMPRESSION_FAMILIES_CONF_KEY, serializeColumnFamilyAttribute(compressionDetails,
642            tableDescriptors));
643    conf.set(BLOCK_SIZE_FAMILIES_CONF_KEY, serializeColumnFamilyAttribute(blockSizeDetails,
644            tableDescriptors));
645    conf.set(BLOOM_TYPE_FAMILIES_CONF_KEY, serializeColumnFamilyAttribute(bloomTypeDetails,
646            tableDescriptors));
647    conf.set(BLOOM_PARAM_FAMILIES_CONF_KEY, serializeColumnFamilyAttribute(bloomParamDetails,
648        tableDescriptors));
649    conf.set(DATABLOCK_ENCODING_FAMILIES_CONF_KEY,
650            serializeColumnFamilyAttribute(dataBlockEncodingDetails, tableDescriptors));
651
652    TableMapReduceUtil.addDependencyJars(job);
653    TableMapReduceUtil.initCredentials(job);
654    LOG.info("Incremental output configured for tables: " + StringUtils.join(allTableNames, ","));
655  }

657  public static void configureIncrementalLoadMap(Job job, TableDescriptor tableDescriptor) throws
658      IOException {
659    Configuration conf = job.getConfiguration();
660
661    job.setOutputKeyClass(ImmutableBytesWritable.class);
662    job.setOutputValueClass(MapReduceExtendedCell.class);
663    job.setOutputFormatClass(HFileOutputFormat2.class);
664
665    ArrayList<TableDescriptor> singleTableDescriptor = new ArrayList<>(1);
666    singleTableDescriptor.add(tableDescriptor);
667
668    conf.set(OUTPUT_TABLE_NAME_CONF_KEY, tableDescriptor.getTableName().getNameAsString());
669    // Set compression algorithms based on column families
670    conf.set(COMPRESSION_FAMILIES_CONF_KEY,
671        serializeColumnFamilyAttribute(compressionDetails, singleTableDescriptor));
672    conf.set(BLOCK_SIZE_FAMILIES_CONF_KEY,
673        serializeColumnFamilyAttribute(blockSizeDetails, singleTableDescriptor));
674    conf.set(BLOOM_TYPE_FAMILIES_CONF_KEY,
675        serializeColumnFamilyAttribute(bloomTypeDetails, singleTableDescriptor));
676    conf.set(BLOOM_PARAM_FAMILIES_CONF_KEY,
677        serializeColumnFamilyAttribute(bloomParamDetails, singleTableDescriptor));
678    conf.set(DATABLOCK_ENCODING_FAMILIES_CONF_KEY,
679        serializeColumnFamilyAttribute(dataBlockEncodingDetails, singleTableDescriptor));
680
681    TableMapReduceUtil.addDependencyJars(job);
682    TableMapReduceUtil.initCredentials(job);
683    LOG.info("Incremental table " + tableDescriptor.getTableName() + " output configured.");
684  }

686  /**
687   * Configure HBase cluster key for remote cluster to load region location for locality-sensitive
688   * if it's enabled.
689   * It's not necessary to call this method explicitly when the cluster key for HBase cluster to be
690   * used to load region location is configured in the job configuration.
691   * Call this method when another HBase cluster key is configured in the job configuration.
692   * For example, you should call when you load data from HBase cluster A using
693   * {@link TableInputFormat} and generate hfiles for HBase cluster B.
694   * Otherwise, HFileOutputFormat2 fetch location from cluster A and locality-sensitive won't
695   * working correctly.
696   * {@link #configureIncrementalLoad(Job, Table, RegionLocator)} calls this method using
697   * {@link Table#getConfiguration} as clusterConf.
698   * See HBASE-25608.
699   *
700   * @param job which has configuration to be updated
701   * @param clusterConf which contains cluster key of the HBase cluster to be locality-sensitive
702   *
703   * @see #configureIncrementalLoad(Job, Table, RegionLocator)
704   * @see #LOCALITY_SENSITIVE_CONF_KEY
705   * @see #REMOTE_CLUSTER_ZOOKEEPER_QUORUM_CONF_KEY
706   * @see #REMOTE_CLUSTER_ZOOKEEPER_CLIENT_PORT_CONF_KEY
707   * @see #REMOTE_CLUSTER_ZOOKEEPER_ZNODE_PARENT_CONF_KEY
708   */
709  public static void configureRemoteCluster(Job job, Configuration clusterConf) {
710    Configuration conf = job.getConfiguration();
711
712    if (!conf.getBoolean(LOCALITY_SENSITIVE_CONF_KEY, DEFAULT_LOCALITY_SENSITIVE)) {
713      return;
714    }
715
716    final String quorum = clusterConf.get(HConstants.ZOOKEEPER_QUORUM);
717    final int clientPort = clusterConf.getInt(
718      HConstants.ZOOKEEPER_CLIENT_PORT, HConstants.DEFAULT_ZOOKEEPER_CLIENT_PORT);
719    final String parent = clusterConf.get(
720      HConstants.ZOOKEEPER_ZNODE_PARENT, HConstants.DEFAULT_ZOOKEEPER_ZNODE_PARENT);
721
722    conf.set(REMOTE_CLUSTER_ZOOKEEPER_QUORUM_CONF_KEY, quorum);
723    conf.setInt(REMOTE_CLUSTER_ZOOKEEPER_CLIENT_PORT_CONF_KEY, clientPort);
724    conf.set(REMOTE_CLUSTER_ZOOKEEPER_ZNODE_PARENT_CONF_KEY, parent);
725
726    LOG.info("ZK configs for remote cluster of bulkload is configured: " +
727      quorum + ":" + clientPort + "/" + parent);
728  }

731   * Runs inside the task to deserialize column family to compression algorithm
732   * map from the configuration.

734   * @param conf to read the serialized values from
735   * @return a map from column family to the configured compression algorithm

737  @InterfaceAudience.Private
738  static Map<byte[], Algorithm> createFamilyCompressionMap(Configuration
739      conf) {
740    Map<byte[], String> stringMap = createFamilyConfValueMap(conf,
741        COMPRESSION_FAMILIES_CONF_KEY);
742    Map<byte[], Algorithm> compressionMap = new TreeMap<>(Bytes.BYTES_COMPARATOR);
743    for (Map.Entry<byte[], String> e : stringMap.entrySet()) {
744      Algorithm algorithm = HFileWriterImpl.compressionByName(e.getValue());
745      compressionMap.put(e.getKey(), algorithm);
746    }
747    return compressionMap;
748  }

820  /**
821   * Run inside the task to deserialize column family to given conf value map.
822   *
823   * @param conf to read the serialized values from
824   * @param confName conf key to read from the configuration
825   * @return a map of column family to the given configuration value
826   */
827  private static Map<byte[], String> createFamilyConfValueMap(
828      Configuration conf, String confName) {
829    Map<byte[], String> confValMap = new TreeMap<>(Bytes.BYTES_COMPARATOR);
830    String confVal = conf.get(confName, "");
831    for (String familyConf : confVal.split("&")) {
832      String[] familySplit = familyConf.split("=");
833      if (familySplit.length != 2) {
834        continue;
835      }
836      try {
837        confValMap.put(Bytes.toBytes(URLDecoder.decode(familySplit[0], "UTF-8")),
838            URLDecoder.decode(familySplit[1], "UTF-8"));
839      } catch (UnsupportedEncodingException e) {
840        // will not happen with UTF-8 encoding
841        throw new AssertionError(e);
842      }
843    }
844    return confValMap;
845  }

847  /**
848   * Configure <code>job</code> with a TotalOrderPartitioner, partitioning against
849   * <code>splitPoints</code>. Cleans up the partitions file after job exists.
850   */
851  static void configurePartitioner(Job job, List<ImmutableBytesWritable> splitPoints, boolean
852          writeMultipleTables)
853      throws IOException {
854    Configuration conf = job.getConfiguration();
855    // create the partitions file
856    FileSystem fs = FileSystem.get(conf);
857    String hbaseTmpFsDir =
858        conf.get(HConstants.TEMPORARY_FS_DIRECTORY_KEY,
859            fs.getHomeDirectory() + "/hbase-staging");
860    Path partitionsPath = new Path(hbaseTmpFsDir, "partitions_" + UUID.randomUUID());
861    fs.makeQualified(partitionsPath);
862    writePartitions(conf, partitionsPath, splitPoints, writeMultipleTables);
863    fs.deleteOnExit(partitionsPath);
864
865    // configure job to use it
866    job.setPartitionerClass(TotalOrderPartitioner.class);
867    TotalOrderPartitioner.setPartitionFile(conf, partitionsPath);
868  }

870  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value =
871    "RCN_REDUNDANT_NULLCHECK_OF_NONNULL_VALUE")
872  @InterfaceAudience.Private
873  static String serializeColumnFamilyAttribute(Function<ColumnFamilyDescriptor, String> fn,
874        List<TableDescriptor> allTables)
875      throws UnsupportedEncodingException {
876    StringBuilder attributeValue = new StringBuilder();
877    int i = 0;
878    for (TableDescriptor tableDescriptor : allTables) {
879      if (tableDescriptor == null) {
880        // could happen with mock table instance
881        // CODEREVIEW: Can I set an empty string in conf if mock table instance?
882        return "";
883      }
884      for (ColumnFamilyDescriptor familyDescriptor : tableDescriptor.getColumnFamilies()) {
885        if (i++ > 0) {
886          attributeValue.append('&');
887        }
888        attributeValue.append(URLEncoder.encode(
889          Bytes.toString(combineTableNameSuffix(tableDescriptor.getTableName().getName(),
890            familyDescriptor.getName())), "UTF-8"));
891        attributeValue.append('=');
892        attributeValue.append(URLEncoder.encode(fn.apply(familyDescriptor), "UTF-8"));
893      }
894    }
895    // Get rid of the last ampersand
896    return attributeValue.toString();
897  }
