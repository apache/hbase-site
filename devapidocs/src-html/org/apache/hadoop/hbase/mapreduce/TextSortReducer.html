021import java.util.ArrayList;
022import java.util.Iterator;
023import java.util.List;
024import java.util.Set;
025import java.util.TreeSet;

027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.hbase.ArrayBackedTag;
029import org.apache.hadoop.hbase.Cell;
030import org.apache.hadoop.hbase.CellComparator;
031import org.apache.hadoop.hbase.KeyValue;
032import org.apache.hadoop.hbase.KeyValueUtil;
033import org.apache.hadoop.hbase.Tag;
034import org.apache.hadoop.hbase.TagType;
035import org.apache.hadoop.hbase.classification.InterfaceAudience;
036import org.apache.hadoop.hbase.classification.InterfaceStability;
037import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
038import org.apache.hadoop.hbase.security.visibility.InvalidLabelException;
039import org.apache.hadoop.hbase.util.Base64;
040import org.apache.hadoop.hbase.util.Bytes;
041import org.apache.hadoop.io.Text;
042import org.apache.hadoop.mapreduce.Counter;
043import org.apache.hadoop.mapreduce.Reducer;
044import org.apache.hadoop.util.StringUtils;

046/**
047 * Emits Sorted KeyValues. Parse the passed text and creates KeyValues. Sorts them before emit.
048 * @see HFileOutputFormat2
049 * @see KeyValueSortReducer
050 * @see PutSortReducer
051 */
052@InterfaceAudience.Public
053@InterfaceStability.Evolving
054public class TextSortReducer extends
055    Reducer<ImmutableBytesWritable, Text, ImmutableBytesWritable, KeyValue> {
056  
057  /** Timestamp for all inserted rows */
058  private long ts;
059
060  /** Column seperator */
061  private String separator;
062
063  /** Should skip bad lines */
064  private boolean skipBadLines;
065  
066  private Counter badLineCount;
067
068  private ImportTsv.TsvParser parser;
069
070  /** Cell visibility expr **/
071  private String cellVisibilityExpr;
072
073  /** Cell TTL */
074  private long ttl;
075
076  private CellCreator kvCreator;
077
078  public long getTs() {
079    return ts;
080  }
081
082  public boolean getSkipBadLines() {
083    return skipBadLines;
084  }
085
086  public Counter getBadLineCount() {
087    return badLineCount;
088  }
089
090  public void incrementBadLineCount(int count) {
091    this.badLineCount.increment(count);
092  }
093
094  /**
095   * Handles initializing this class with objects specific to it (i.e., the parser).
096   * Common initialization that might be leveraged by a subsclass is done in
097   * <code>doSetup</code>. Hence a subclass may choose to override this method
098   * and call <code>doSetup</code> as well before handling it's own custom params.
099   *
100   * @param context
101   */
102  @Override
103  protected void setup(Context context) {
104    Configuration conf = context.getConfiguration();
105    doSetup(context, conf);
106
107    parser = new ImportTsv.TsvParser(conf.get(ImportTsv.COLUMNS_CONF_KEY), separator);
108    if (parser.getRowKeyColumnIndex() == -1) {
109      throw new RuntimeException("No row key column specified");
110    }
111    this.kvCreator = new CellCreator(conf);
112  }
113
114  /**
115   * Handles common parameter initialization that a subclass might want to leverage.
116   * @param context
117   * @param conf
118   */
119  protected void doSetup(Context context, Configuration conf) {
120    // If a custom separator has been used,
121    // decode it back from Base64 encoding.
122    separator = conf.get(ImportTsv.SEPARATOR_CONF_KEY);
123    if (separator == null) {
124      separator = ImportTsv.DEFAULT_SEPARATOR;
125    } else {
126      separator = new String(Base64.decode(separator));
127    }
128
129    // Should never get 0 as we are setting this to a valid value in job configuration.
130    ts = conf.getLong(ImportTsv.TIMESTAMP_CONF_KEY, 0);
131
132    skipBadLines = context.getConfiguration().getBoolean(ImportTsv.SKIP_LINES_CONF_KEY, true);
133    badLineCount = context.getCounter("ImportTsv", "Bad Lines");
134  }
135  
136  @Override
137  protected void reduce(
138      ImmutableBytesWritable rowKey,
139      java.lang.Iterable<Text> lines,
140      Reducer<ImmutableBytesWritable, Text,
141              ImmutableBytesWritable, KeyValue>.Context context)
142      throws java.io.IOException, InterruptedException
143  {
144    // although reduce() is called per-row, handle pathological case
145    long threshold = context.getConfiguration().getLong(
146        "reducer.row.threshold", 1L * (1<<30));
147    Iterator<Text> iter = lines.iterator();
148    while (iter.hasNext()) {
149      Set<KeyValue> kvs = new TreeSet<KeyValue>(CellComparator.COMPARATOR);
150      long curSize = 0;
151      // stop at the end or the RAM threshold
152      while (iter.hasNext() && curSize < threshold) {
153        Text line = iter.next();
154        byte[] lineBytes = line.getBytes();
155        try {
156          ImportTsv.TsvParser.ParsedLine parsed = parser.parse(lineBytes, line.getLength());
157          // Retrieve timestamp if exists
158          ts = parsed.getTimestamp(ts);
159          cellVisibilityExpr = parsed.getCellVisibility();
160          ttl = parsed.getCellTTL();
161          
162          // create tags for the parsed line
163          List<Tag> tags = new ArrayList<Tag>();
164          if (cellVisibilityExpr != null) {
165            tags.addAll(kvCreator.getVisibilityExpressionResolver().createVisibilityExpTags(
166              cellVisibilityExpr));
167          }
168          // Add TTL directly to the KV so we can vary them when packing more than one KV
169          // into puts
170          if (ttl > 0) {
171            tags.add(new ArrayBackedTag(TagType.TTL_TAG_TYPE, Bytes.toBytes(ttl)));
172          }
173          for (int i = 0; i < parsed.getColumnCount(); i++) {
174            if (i == parser.getRowKeyColumnIndex() || i == parser.getTimestampKeyColumnIndex()
175                || i == parser.getAttributesKeyColumnIndex() || i == parser.getCellVisibilityColumnIndex()
176                || i == parser.getCellTTLColumnIndex()) {
177              continue;
178            }
179            // Creating the KV which needs to be directly written to HFiles. Using the Facade
180            // KVCreator for creation of kvs.
181            Cell cell = this.kvCreator.create(lineBytes, parsed.getRowKeyOffset(),
182                parsed.getRowKeyLength(), parser.getFamily(i), 0, parser.getFamily(i).length,
183                parser.getQualifier(i), 0, parser.getQualifier(i).length, ts, lineBytes,
184                parsed.getColumnOffset(i), parsed.getColumnLength(i), tags);
185            KeyValue kv = KeyValueUtil.ensureKeyValue(cell);
186            kvs.add(kv);
187            curSize += kv.heapSize();
188          }
189        } catch (ImportTsv.TsvParser.BadTsvLineException | IllegalArgumentException
190            | InvalidLabelException badLine) {
191          if (skipBadLines) {
192            System.err.println("Bad line." + badLine.getMessage());
193            incrementBadLineCount(1);
194            continue;
195          }
196          throw new IOException(badLine);
197        }
198      }
199      context.setStatus("Read " + kvs.size() + " entries of " + kvs.getClass()
200          + "(" + StringUtils.humanReadableInt(curSize) + ")");
201      int index = 0;
202      for (KeyValue kv : kvs) {
203        context.write(rowKey, kv);
204        if (++index > 0 && index % 100 == 0)
205          context.setStatus("Wrote " + index + " key values.");
206      }
207
208      // if we have more entries to process
209      if (iter.hasNext()) {
210        // force flush because we cannot guarantee intra-row sorted order
211        context.write(null, null);
212      }
213    }
214  }
215}
