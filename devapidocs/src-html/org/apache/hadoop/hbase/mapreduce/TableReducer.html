019package org.apache.hadoop.hbase.mapreduce;
020
021import org.apache.hadoop.hbase.classification.InterfaceAudience;
022import org.apache.hadoop.hbase.classification.InterfaceStability;
023import org.apache.hadoop.hbase.client.Mutation;
024import org.apache.hadoop.mapreduce.Reducer;
025
026/**
027 * Extends the basic <code>Reducer</code> class to add the required key and
028 * value input/output classes. While the input key and value as well as the
029 * output key can be anything handed in from the previous map phase the output
030 * value <u>must</u> be either a {@link org.apache.hadoop.hbase.client.Put Put}
031 * or a {@link org.apache.hadoop.hbase.client.Delete Delete} instance when
032 * using the {@link TableOutputFormat} class.
033 * <p>
034 * This class is extended by {@link IdentityTableReducer} but can also be
035 * subclassed to implement similar features or any custom code needed. It has
036 * the advantage to enforce the output value to a specific basic type.
037 *
038 * @param <KEYIN>  The type of the input key.
039 * @param <VALUEIN>  The type of the input value.
040 * @param <KEYOUT>  The type of the output key.
041 * @see org.apache.hadoop.mapreduce.Reducer
042 */
043@InterfaceAudience.Public
044@InterfaceStability.Stable
045public abstract class TableReducer<KEYIN, VALUEIN, KEYOUT>
046extends Reducer<KEYIN, VALUEIN, KEYOUT, Mutation> {
047}
