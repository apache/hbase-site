019package org.apache.hadoop.hbase.mapreduce;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.classification.InterfaceStability;
025import org.apache.hadoop.conf.Configurable;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
028import org.apache.hadoop.hbase.util.Base64;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.apache.hadoop.mapreduce.Partitioner;
031
032/**
033 * A partitioner that takes start and end keys and uses bigdecimal to figure
034 * which reduce a key belongs to.  Pass the start and end
035 * keys in the Configuration using <code>hbase.simpletotalorder.start</code>
036 * and <code>hbase.simpletotalorder.end</code>.  The end key needs to be
037 * exclusive; i.e. one larger than the biggest key in your key space.
038 * You may be surprised at how this class partitions the space; it may not
039 * align with preconceptions; e.g. a start key of zero and an end key of 100
040 * divided in ten will not make regions whose range is 0-10, 10-20, and so on.
041 * Make your own partitioner if you need the region spacing to come out a
042 * particular way.
043 * @param <VALUE>
044 * @see #START
045 * @see #END
046 */
047@InterfaceAudience.Public
048@InterfaceStability.Stable
049public class SimpleTotalOrderPartitioner<VALUE> extends Partitioner<ImmutableBytesWritable, VALUE>
050implements Configurable {
051  private final static Log LOG = LogFactory.getLog(SimpleTotalOrderPartitioner.class);
052
053  @Deprecated
054  public static final String START = "hbase.simpletotalorder.start";
055  @Deprecated
056  public static final String END = "hbase.simpletotalorder.end";
057  
058  static final String START_BASE64 = "hbase.simpletotalorder.start.base64";
059  static final String END_BASE64 = "hbase.simpletotalorder.end.base64";
060  
061  private Configuration c;
062  private byte [] startkey;
063  private byte [] endkey;
064  private byte [][] splits;
065  private int lastReduces = -1;
066
067  public static void setStartKey(Configuration conf, byte[] startKey) {
068    conf.set(START_BASE64, Base64.encodeBytes(startKey));
069  }
070  
071  public static void setEndKey(Configuration conf, byte[] endKey) {
072    conf.set(END_BASE64, Base64.encodeBytes(endKey));
073  }
074  
075  @SuppressWarnings("deprecation")
076  static byte[] getStartKey(Configuration conf) {
077    return getKeyFromConf(conf, START_BASE64, START);
078  }
079  
080  @SuppressWarnings("deprecation")
081  static byte[] getEndKey(Configuration conf) {
082    return getKeyFromConf(conf, END_BASE64, END);
083  }
084  
085  private static byte[] getKeyFromConf(Configuration conf,
086      String base64Key, String deprecatedKey) {
087    String encoded = conf.get(base64Key);
088    if (encoded != null) {
089      return Base64.decode(encoded);
090    }
091    String oldStyleVal = conf.get(deprecatedKey);
092    if (oldStyleVal == null) {
093      return null;
094    }
095    LOG.warn("Using deprecated configuration " + deprecatedKey +
096        " - please use static accessor methods instead.");
097    return Bytes.toBytesBinary(oldStyleVal);
098  }
099  
100  @Override
101  public int getPartition(final ImmutableBytesWritable key, final VALUE value,
102      final int reduces) {
103    if (reduces == 1) return 0;
104    if (this.lastReduces != reduces) {
105      this.splits = Bytes.split(this.startkey, this.endkey, reduces - 1);
106      for (int i = 0; i < splits.length; i++) {
107        LOG.info(Bytes.toStringBinary(splits[i]));
108      }
109      this.lastReduces = reduces;
110    }
111    int pos = Bytes.binarySearch(this.splits, key.get(), key.getOffset(),
112      key.getLength());
113    // Below code is from hfile index search.
114    if (pos < 0) {
115      pos++;
116      pos *= -1;
117      if (pos == 0) {
118        // falls before the beginning of the file.
119        throw new RuntimeException("Key outside start/stop range: " +
120          key.toString());
121      }
122      pos--;
123    }
124    return pos;
125  }
126
127  @Override
128  public Configuration getConf() {
129    return this.c;
130  }
131
132  @Override
133  public void setConf(Configuration conf) {
134    this.c = conf;
135    this.startkey = getStartKey(conf);
136    this.endkey = getEndKey(conf);
137    if (startkey == null || endkey == null) {
138      throw new RuntimeException(this.getClass() + " not configured");
139    }
140    LOG.info("startkey=" + Bytes.toStringBinary(startkey) +
141        ", endkey=" + Bytes.toStringBinary(endkey));
142    // Reset last reduces count on change of Start / End key
143    this.lastReduces = -1;
144  }
145}
