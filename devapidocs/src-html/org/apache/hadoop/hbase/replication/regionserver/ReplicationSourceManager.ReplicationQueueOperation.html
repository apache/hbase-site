071/**
072 * This class is responsible to manage all the replication sources. There are two classes of
073 * sources:
074 * <ul>
075 * <li>Normal sources are persistent and one per peer cluster</li>
076 * <li>Old sources are recovered from a failed region server and our only goal is to finish
077 * replicating the WAL queue it had</li>
078 * </ul>
079 * <p>
080 * When a region server dies, this class uses a watcher to get notified and it tries to grab a lock
081 * in order to transfer all the queues in a local old source.
082 * <p>
083 * Synchronization specification:
084 * <ul>
085 * <li>No need synchronized on {@link #sources}. {@link #sources} is a ConcurrentHashMap and there
086 * is a Lock for peer id in {@link PeerProcedureHandlerImpl}. So there is no race for peer
087 * operations.</li>
088 * <li>Need synchronized on {@link #walsById}. There are four methods which modify it,
089 * {@link #addPeer(String)}, {@link #removePeer(String)},
090 * {@link #cleanOldLogs(SortedSet, String, String)} and {@link #preLogRoll(Path)}.
091 * {@link #walsById} is a ConcurrentHashMap and there is a Lock for peer id in
092 * {@link PeerProcedureHandlerImpl}. So there is no race between {@link #addPeer(String)} and
093 * {@link #removePeer(String)}. {@link #cleanOldLogs(SortedSet, String, String)} is called by
094 * {@link ReplicationSourceInterface}. So no race with {@link #addPeer(String)}.
095 * {@link #removePeer(String)} will terminate the {@link ReplicationSourceInterface} firstly, then
096 * remove the wals from {@link #walsById}. So no race with {@link #removePeer(String)}. The only
097 * case need synchronized is {@link #cleanOldLogs(SortedSet, String, String)} and
098 * {@link #preLogRoll(Path)}.</li>
099 * <li>No need synchronized on {@link #walsByIdRecoveredQueues}. There are three methods which
100 * modify it, {@link #removePeer(String)} , {@link #cleanOldLogs(SortedSet, String, String)} and
101 * {@link ReplicationSourceManager.NodeFailoverWorker#run()}.
102 * {@link #cleanOldLogs(SortedSet, String, String)} is called by {@link ReplicationSourceInterface}.
103 * {@link #removePeer(String)} will terminate the {@link ReplicationSourceInterface} firstly, then
104 * remove the wals from {@link #walsByIdRecoveredQueues}. And
105 * {@link ReplicationSourceManager.NodeFailoverWorker#run()} will add the wals to
106 * {@link #walsByIdRecoveredQueues} firstly, then start up a {@link ReplicationSourceInterface}. So
107 * there is no race here. For {@link ReplicationSourceManager.NodeFailoverWorker#run()} and
108 * {@link #removePeer(String)}, there is already synchronized on {@link #oldsources}. So no need
109 * synchronized on {@link #walsByIdRecoveredQueues}.</li>
110 * <li>Need synchronized on {@link #latestPaths} to avoid the new open source miss new log.</li>
111 * <li>Need synchronized on {@link #oldsources} to avoid adding recovered source for the
112 * to-be-removed peer.</li>
113 * </ul>
114 */

160  /**
161   * Creates a replication manager and sets the watch on all the other registered region servers
162   * @param queueStorage the interface for manipulating replication queues
163   * @param replicationPeers
164   * @param replicationTracker
165   * @param conf the configuration to use
166   * @param server the server for this region server
167   * @param fs the file system to use
168   * @param logDir the directory that contains all wal directories of live RSs
169   * @param oldLogDir the directory where old logs are archived
170   * @param clusterId
171   */
172  public ReplicationSourceManager(ReplicationQueueStorage queueStorage,
173      ReplicationPeers replicationPeers, ReplicationTracker replicationTracker, Configuration conf,
174      Server server, FileSystem fs, Path logDir, Path oldLogDir, UUID clusterId,
175      WALFileLengthProvider walFileLengthProvider) throws IOException {
176    // CopyOnWriteArrayList is thread-safe.
177    // Generally, reading is more than modifying.
178    this.sources = new ConcurrentHashMap<>();
179    this.queueStorage = queueStorage;
180    this.replicationPeers = replicationPeers;
181    this.replicationTracker = replicationTracker;
182    this.server = server;
183    this.walsById = new ConcurrentHashMap<>();
184    this.walsByIdRecoveredQueues = new ConcurrentHashMap<>();
185    this.oldsources = new ArrayList<>();
186    this.conf = conf;
187    this.fs = fs;
188    this.logDir = logDir;
189    this.oldLogDir = oldLogDir;
190    this.sleepBeforeFailover = conf.getLong("replication.sleep.before.failover", 30000); // 30
191                                                                                         // seconds
192    this.clusterId = clusterId;
193    this.walFileLengthProvider = walFileLengthProvider;
194    this.replicationTracker.registerListener(this);
195    // It's preferable to failover 1 RS at a time, but with good zk servers
196    // more could be processed at the same time.
197    int nbWorkers = conf.getInt("replication.executor.workers", 1);
198    // use a short 100ms sleep since this could be done inline with a RS startup
199    // even if we fail, other region servers can take care of it
200    this.executor = new ThreadPoolExecutor(nbWorkers, nbWorkers, 100, TimeUnit.MILLISECONDS,
201        new LinkedBlockingQueue<>());
202    ThreadFactoryBuilder tfb = new ThreadFactoryBuilder();
203    tfb.setNameFormat("ReplicationExecutor-%d");
204    tfb.setDaemon(true);
205    this.executor.setThreadFactory(tfb.build());
206    this.latestPaths = new HashSet<Path>();
207    replicationForBulkLoadDataEnabled = conf.getBoolean(HConstants.REPLICATION_BULKLOAD_ENABLE_KEY,
208      HConstants.REPLICATION_BULKLOAD_ENABLE_DEFAULT);
209    this.replicationWaitTime = conf.getLong(HConstants.REPLICATION_SERIALLY_WAITING_KEY,
210      HConstants.REPLICATION_SERIALLY_WAITING_DEFAULT);
211    connection = ConnectionFactory.createConnection(conf);
212  }

214  /**
215   * Adds a normal source per registered peer cluster and tries to process all old region server wal
216   * queues
217   * <p>
218   * The returned future is for adoptAbandonedQueues task.
219   */
220  Future<?> init() throws IOException {
221    for (String id : this.replicationPeers.getAllPeerIds()) {
222      addSource(id);
223      if (replicationForBulkLoadDataEnabled) {
224        // Check if peer exists in hfile-refs queue, if not add it. This can happen in the case
225        // when a peer was added before replication for bulk loaded data was enabled.
226        throwIOExceptionWhenFail(() -> this.queueStorage.addPeerToHFileRefs(id));
227      }
228    }
229    return this.executor.submit(this::adoptAbandonedQueues);
230  }
231
232  private void adoptAbandonedQueues() {
233    List<ServerName> currentReplicators = null;
234    try {
235      currentReplicators = queueStorage.getListOfReplicators();
236    } catch (ReplicationException e) {
237      server.abort("Failed to get all replicators", e);
238      return;
239    }
240    if (currentReplicators == null || currentReplicators.isEmpty()) {
241      return;
242    }
243    List<ServerName> otherRegionServers = replicationTracker.getListOfRegionServers().stream()
244        .map(ServerName::valueOf).collect(Collectors.toList());
245    LOG.info(
246      "Current list of replicators: " + currentReplicators + " other RSs: " + otherRegionServers);
247
248    // Look if there's anything to process after a restart
249    for (ServerName rs : currentReplicators) {
250      if (!otherRegionServers.contains(rs)) {
251        transferQueues(rs);
252      }
253    }
254  }

256  /**
257   * 1. Add peer to replicationPeers 2. Add the normal source and related replication queue 3. Add
258   * HFile Refs
259   * @param peerId the id of replication peer
260   */
261  public void addPeer(String peerId) throws IOException {
262    boolean added = false;
263    try {
264      added = this.replicationPeers.addPeer(peerId);
265    } catch (ReplicationException e) {
266      throw new IOException(e);
267    }
268    if (added) {
269      addSource(peerId);
270      if (replicationForBulkLoadDataEnabled) {
271        throwIOExceptionWhenFail(() -> this.queueStorage.addPeerToHFileRefs(peerId));
272      }
273    }
274  }

276  /**
277   * 1. Remove peer for replicationPeers 2. Remove all the recovered sources for the specified id
278   * and related replication queues 3. Remove the normal source and related replication queue 4.
279   * Remove HFile Refs
280   * @param peerId the id of the replication peer
281   */
282  public void removePeer(String peerId) {
283    replicationPeers.removePeer(peerId);
284    String terminateMessage = "Replication stream was removed by a user";
285    List<ReplicationSourceInterface> oldSourcesToDelete = new ArrayList<>();
286    // synchronized on oldsources to avoid adding recovered source for the to-be-removed peer
287    // see NodeFailoverWorker.run
288    synchronized (this.oldsources) {
289      // First close all the recovered sources for this peer
290      for (ReplicationSourceInterface src : oldsources) {
291        if (peerId.equals(src.getPeerId())) {
292          oldSourcesToDelete.add(src);
293        }
294      }
295      for (ReplicationSourceInterface src : oldSourcesToDelete) {
296        src.terminate(terminateMessage);
297        removeRecoveredSource(src);
298      }
299    }
300    LOG.info(
301      "Number of deleted recovered sources for " + peerId + ": " + oldSourcesToDelete.size());
302    // Now close the normal source for this peer
303    ReplicationSourceInterface srcToRemove = this.sources.get(peerId);
304    if (srcToRemove != null) {
305      srcToRemove.terminate(terminateMessage);
306      removeSource(srcToRemove);
307    } else {
308      // This only happened in unit test TestReplicationSourceManager#testPeerRemovalCleanup
309      // Delete queue from storage and memory and queue id is same with peer id for normal
310      // source
311      deleteQueue(peerId);
312      this.walsById.remove(peerId);
313    }
314
315    // Remove HFile Refs
316    abortWhenFail(() -> this.queueStorage.removePeerFromHFileRefs(peerId));
317  }

321   * @param queueId the id of the replication queue
322   * @return the created source
323   */
324  private ReplicationSourceInterface createSource(String queueId, ReplicationPeer replicationPeer)

326    ReplicationSourceInterface src = ReplicationSourceFactory.create(conf, queueId);
327
328    MetricsSource metrics = new MetricsSource(queueId);

335  /**
336   * Add a normal source for the given peer on this region server. Meanwhile, add new replication
337   * queue to storage. For the newly added peer, we only need to enqueue the latest log of each wal
338   * group and do replication
339   * @param peerId the id of the replication peer
340   * @return the source that was created
341   */
342  @VisibleForTesting
343  ReplicationSourceInterface addSource(String peerId) throws IOException {
344    ReplicationPeer peer = replicationPeers.getPeer(peerId);
345    ReplicationSourceInterface src = createSource(peerId, peer);
346    // synchronized on latestPaths to avoid missing the new log
347    synchronized (this.latestPaths) {
348      this.sources.put(peerId, src);
349      Map<String, SortedSet<String>> walsByGroup = new HashMap<>();
350      this.walsById.put(peerId, walsByGroup);
351      // Add the latest wal to that source's queue
352      if (this.latestPaths.size() > 0) {
353        for (Path logPath : latestPaths) {
354          String name = logPath.getName();
355          String walPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(name);
356          SortedSet<String> logs = new TreeSet<>();
357          logs.add(name);
358          walsByGroup.put(walPrefix, logs);
359          // Abort RS and throw exception to make add peer failed
360          abortAndThrowIOExceptionWhenFail(
361            () -> this.queueStorage.addWAL(server.getServerName(), peerId, name));
362          src.enqueueLog(logPath);
363        }
364      }
365    }
366    src.startup();
367    return src;
368  }

370  /**
371   * Close the previous replication sources of this peer id and open new sources to trigger the new
372   * replication state changes or new replication config changes. Here we don't need to change
373   * replication queue storage and only to enqueue all logs to the new replication source
374   * @param peerId the id of the replication peer
375   * @throws IOException
376   */
377  public void refreshSources(String peerId) throws IOException {
378    String terminateMessage = "Peer " + peerId +
379      " state or config changed. Will close the previous replication source and open a new one";
380    ReplicationPeer peer = replicationPeers.getPeer(peerId);
381    ReplicationSourceInterface src = createSource(peerId, peer);
382    // synchronized on latestPaths to avoid missing the new log
383    synchronized (this.latestPaths) {
384      ReplicationSourceInterface toRemove = this.sources.put(peerId, src);
385      if (toRemove != null) {
386        LOG.info("Terminate replication source for " + toRemove.getPeerId());
387        toRemove.terminate(terminateMessage);
388      }
389      for (SortedSet<String> walsByGroup : walsById.get(peerId).values()) {
390        walsByGroup.forEach(wal -> src.enqueueLog(new Path(this.logDir, wal)));
391      }
392    }
393    LOG.info("Startup replication source for " + src.getPeerId());
394    src.startup();
395
396    List<ReplicationSourceInterface> toStartup = new ArrayList<>();
397    // synchronized on oldsources to avoid race with NodeFailoverWorker
398    synchronized (this.oldsources) {
399      List<String> previousQueueIds = new ArrayList<>();
400      for (ReplicationSourceInterface oldSource : this.oldsources) {
401        if (oldSource.getPeerId().equals(peerId)) {
402          previousQueueIds.add(oldSource.getQueueId());
403          oldSource.terminate(terminateMessage);
404          this.oldsources.remove(oldSource);
405        }
406      }
407      for (String queueId : previousQueueIds) {
408        ReplicationSourceInterface replicationSource = createSource(queueId, peer);
409        this.oldsources.add(replicationSource);
410        for (SortedSet<String> walsByGroup : walsByIdRecoveredQueues.get(queueId).values()) {
411          walsByGroup.forEach(wal -> src.enqueueLog(new Path(wal)));
412        }
413        toStartup.add(replicationSource);
414      }
415    }
416    for (ReplicationSourceInterface replicationSource : oldsources) {
417      replicationSource.startup();
418    }
419  }

460  private void abortWhenFail(ReplicationQueueOperation op) {
461    try {
462      op.exec();
463    } catch (ReplicationException e) {
464      server.abort("Failed to operate on replication queue", e);
465    }
466  }
467
468  private void throwIOExceptionWhenFail(ReplicationQueueOperation op) throws IOException {
469    try {
470      op.exec();
471    } catch (ReplicationException e) {
472      throw new IOException(e);
473    }
474  }

476  private void abortAndThrowIOExceptionWhenFail(ReplicationQueueOperation op) throws IOException {
477    try {
478      op.exec();
479    } catch (ReplicationException e) {
480      server.abort("Failed to operate on replication queue", e);
481      throw new IOException(e);
482    }
483  }

501  /**
502   * Cleans a log file and all older logs from replication queue. Called when we are sure that a log
503   * file is closed and has no more entries.
504   * @param log Path to the log
505   * @param queueId id of the replication queue
506   * @param queueRecovered Whether this is a recovered queue
507   */
508  @VisibleForTesting
509  void cleanOldLogs(String log, String queueId, boolean queueRecovered) {
510    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(log);
511    if (queueRecovered) {
512      SortedSet<String> wals = walsByIdRecoveredQueues.get(queueId).get(logPrefix);
513      if (wals != null && !wals.first().equals(log)) {
514        cleanOldLogs(wals, log, queueId);
515      }
516    } else {
517      // synchronized on walsById to avoid race with preLogRoll
518      synchronized (this.walsById) {
519        SortedSet<String> wals = walsById.get(queueId).get(logPrefix);
520        if (wals != null && !wals.first().equals(log)) {
521          cleanOldLogs(wals, log, queueId);
522        }
523      }
524    }
525  }
526
527  private void cleanOldLogs(SortedSet<String> wals, String key, String id) {
528    SortedSet<String> walSet = wals.headSet(key);
529    if (LOG.isDebugEnabled()) {
530      LOG.debug("Removing " + walSet.size() + " logs in the list: " + walSet);
531    }
532    for (String wal : walSet) {
533      abortWhenFail(() -> this.queueStorage.removeWAL(server.getServerName(), id, wal));
534    }
535    walSet.clear();
536  }

538  // public because of we call it in TestReplicationEmptyWALRecovery
539  @VisibleForTesting
540  public void preLogRoll(Path newLog) throws IOException {
541    String logName = newLog.getName();
542    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(logName);
543    // synchronized on latestPaths to avoid the new open source miss the new log
544    synchronized (this.latestPaths) {
545      // Add log to queue storage
546      for (ReplicationSourceInterface source : this.sources.values()) {
547        // If record log to queue storage failed, abort RS and throw exception to make log roll
548        // failed
549        abortAndThrowIOExceptionWhenFail(
550          () -> this.queueStorage.addWAL(server.getServerName(), source.getQueueId(), logName));
551      }
552
553      // synchronized on walsById to avoid race with cleanOldLogs
554      synchronized (this.walsById) {
555        // Update walsById map
556        for (Map.Entry<String, Map<String, SortedSet<String>>> entry : this.walsById.entrySet()) {
557          String peerId = entry.getKey();
558          Map<String, SortedSet<String>> walsByPrefix = entry.getValue();
559          boolean existingPrefix = false;
560          for (Map.Entry<String, SortedSet<String>> walsEntry : walsByPrefix.entrySet()) {
561            SortedSet<String> wals = walsEntry.getValue();
562            if (this.sources.isEmpty()) {
563              // If there's no slaves, don't need to keep the old wals since
564              // we only consider the last one when a new slave comes in
565              wals.clear();
566            }
567            if (logPrefix.equals(walsEntry.getKey())) {
568              wals.add(logName);
569              existingPrefix = true;
570            }
571          }
572          if (!existingPrefix) {
573            // The new log belongs to a new group, add it into this peer
574            LOG.debug("Start tracking logs for wal group " + logPrefix + " for peer " + peerId);
575            SortedSet<String> wals = new TreeSet<>();
576            wals.add(logName);
577            walsByPrefix.put(logPrefix, wals);
578          }
579        }
580      }
581
582      // Add to latestPaths
583      Iterator<Path> iterator = latestPaths.iterator();
584      while (iterator.hasNext()) {
585        Path path = iterator.next();
586        if (path.getName().contains(logPrefix)) {
587          iterator.remove();
588          break;
589        }
590      }
591      this.latestPaths.add(newLog);
592    }
593  }

609  /**
610   * Transfer all the queues of the specified to this region server. First it tries to grab a lock
611   * and if it works it will move the old queues and finally will delete the old queues.
612   * <p>
613   * It creates one old source for any type of source of the old rs.
614   */
615  private void transferQueues(ServerName deadRS) {
616    if (server.getServerName().equals(deadRS)) {
617      // it's just us, give up
618      return;
619    }
620    NodeFailoverWorker transfer = new NodeFailoverWorker(deadRS);
621    try {
622      this.executor.execute(transfer);
623    } catch (RejectedExecutionException ex) {
624      LOG.info("Cancelling the transfer of " + deadRS + " because of " + ex.getMessage());
625    }
626  }

628  /**
629   * Class responsible to setup new ReplicationSources to take care of the queues from dead region
630   * servers.
631   */
632  class NodeFailoverWorker extends Thread {
633
634    private final ServerName deadRS;
635
636    @VisibleForTesting
637    public NodeFailoverWorker(ServerName deadRS) {
638      super("Failover-for-" + deadRS);
639      this.deadRS = deadRS;
640    }
641
642    @Override
643    public void run() {
644      // Wait a bit before transferring the queues, we may be shutting down.
645      // This sleep may not be enough in some cases.
646      try {
647        Thread.sleep(sleepBeforeFailover +
648          (long) (ThreadLocalRandom.current().nextFloat() * sleepBeforeFailover));
649      } catch (InterruptedException e) {
650        LOG.warn("Interrupted while waiting before transferring a queue.");
651        Thread.currentThread().interrupt();
652      }
653      // We try to lock that rs' queue directory
654      if (server.isStopped()) {
655        LOG.info("Not transferring queue since we are shutting down");
656        return;
657      }
658      Map<String, Set<String>> newQueues = new HashMap<>();
659      try {
660        List<String> queues = queueStorage.getAllQueues(deadRS);
661        while (!queues.isEmpty()) {
662          Pair<String, SortedSet<String>> peer = queueStorage.claimQueue(deadRS,
663            queues.get(ThreadLocalRandom.current().nextInt(queues.size())), server.getServerName());
664          long sleep = sleepBeforeFailover / 2;
665          if (!peer.getSecond().isEmpty()) {
666            newQueues.put(peer.getFirst(), peer.getSecond());
667            sleep = sleepBeforeFailover;
668          }
669          try {
670            Thread.sleep(sleep);
671          } catch (InterruptedException e) {
672            LOG.warn("Interrupted while waiting before transferring a queue.");
673            Thread.currentThread().interrupt();
674          }
675          queues = queueStorage.getAllQueues(deadRS);
676        }
677        if (queues.isEmpty()) {
678          queueStorage.removeReplicatorIfQueueIsEmpty(deadRS);
679        }
680      } catch (ReplicationException e) {
681        server.abort("Failed to claim queue from dead regionserver", e);
682        return;
683      }
684      // Copying over the failed queue is completed.
685      if (newQueues.isEmpty()) {
686        // We either didn't get the lock or the failed region server didn't have any outstanding
687        // WALs to replicate, so we are done.
688        return;
689      }
690
691      for (Map.Entry<String, Set<String>> entry : newQueues.entrySet()) {
692        String queueId = entry.getKey();
693        Set<String> walsSet = entry.getValue();
694        try {
695          // there is not an actual peer defined corresponding to peerId for the failover.
696          ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(queueId);
697          String actualPeerId = replicationQueueInfo.getPeerId();
698
699          ReplicationPeer peer = replicationPeers.getPeer(actualPeerId);
700          if (peer == null) {
701            LOG.warn("Skipping failover for peer:" + actualPeerId + " of node " + deadRS +
702              ", peer is null");
703            abortWhenFail(() -> queueStorage.removeQueue(server.getServerName(), queueId));
704            continue;
705          }
706          if (server instanceof ReplicationSyncUp.DummyServer
707              && peer.getPeerState().equals(PeerState.DISABLED)) {
708            LOG.warn("Peer {} is disbaled. ReplicationSyncUp tool will skip "
709                + "replicating data to this peer.",
710              actualPeerId);
711            continue;
712          }
713          // track sources in walsByIdRecoveredQueues
714          Map<String, SortedSet<String>> walsByGroup = new HashMap<>();
715          walsByIdRecoveredQueues.put(queueId, walsByGroup);
716          for (String wal : walsSet) {
717            String walPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(wal);
718            SortedSet<String> wals = walsByGroup.get(walPrefix);
719            if (wals == null) {
720              wals = new TreeSet<>();
721              walsByGroup.put(walPrefix, wals);
722            }
723            wals.add(wal);
724          }
725
726          ReplicationSourceInterface src = createSource(queueId, peer);
727          // synchronized on oldsources to avoid adding recovered source for the to-be-removed peer
728          synchronized (oldsources) {
729            if (!replicationPeers.getAllPeerIds().contains(src.getPeerId())) {
730              src.terminate("Recovered queue doesn't belong to any current peer");
731              removeRecoveredSource(src);
732              continue;
733            }
734            oldsources.add(src);
735            for (String wal : walsSet) {
736              src.enqueueLog(new Path(oldLogDir, wal));
737            }
738            src.startup();
739          }
740        } catch (IOException e) {
741          // TODO manage it
742          LOG.error("Failed creating a source", e);
743        }
744      }
745    }
746  }

891  /**
892   * Whether an entry can be pushed to the peer or not right now. If we enable serial replication,
893   * we can not push the entry until all entries in its region whose sequence numbers are smaller
894   * than this entry have been pushed. For each ReplicationSource, we need only check the first
895   * entry in each region, as long as it can be pushed, we can push all in this ReplicationSource.
896   * This method will be blocked until we can push.
897   * @return the first barrier of entry's region, or -1 if there is no barrier. It is used to
898   *         prevent saving positions in the region of no barrier.
899   */
900  void waitUntilCanBePushed(byte[] encodedName, long seq, String peerId)
901      throws IOException, InterruptedException {
902    /**
903     * There are barriers for this region and position for this peer. N barriers form N intervals,
904     * (b1,b2) (b2,b3) ... (bn,max). Generally, there is no logs whose seq id is not greater than
905     * the first barrier and the last interval is start from the last barrier. There are several
906     * conditions that we can push now, otherwise we should block: 1) "Serial replication" is not
907     * enabled, we can push all logs just like before. This case should not call this method. 2)
908     * There is no barriers for this region, or the seq id is smaller than the first barrier. It is
909     * mainly because we alter REPLICATION_SCOPE = 2. We can not guarantee the order of logs that is
910     * written before altering. 3) This entry is in the first interval of barriers. We can push them
911     * because it is the start of a region. But if the region is created by region split, we should
912     * check if the parent regions are fully pushed. 4) If the entry's seq id and the position are
913     * in same section, or the pos is the last number of previous section. Because when open a
914     * region we put a barrier the number is the last log's id + 1. 5) Log's seq is smaller than pos
915     * in meta, we are retrying. It may happen when a RS crashes after save replication meta and
916     * before save zk offset.
917     */
918    List<Long> barriers = MetaTableAccessor.getReplicationBarriers(connection, encodedName);
919    if (barriers.isEmpty() || seq <= barriers.get(0)) {
920      // Case 2
921      return;
922    }
923    int interval = Collections.binarySearch(barriers, seq);
924    if (interval < 0) {
925      interval = -interval - 1;// get the insert position if negative
926    }
927    if (interval == 1) {
928      // Case 3
929      // Check if there are parent regions
930      String parentValue =
931          MetaTableAccessor.getSerialReplicationParentRegion(connection, encodedName);
932      if (parentValue == null) {
933        // This region has no parent or the parent's log entries are fully pushed.
934        return;
935      }
936      while (true) {
937        boolean allParentDone = true;
938        String[] parentRegions = parentValue.split(",");
939        for (String parent : parentRegions) {
940          byte[] region = Bytes.toBytes(parent);
941          long pos = MetaTableAccessor.getReplicationPositionForOnePeer(connection, region, peerId);
942          List<Long> parentBarriers = MetaTableAccessor.getReplicationBarriers(connection, region);
943          if (parentBarriers.size() > 0 &&
944            parentBarriers.get(parentBarriers.size() - 1) - 1 > pos) {
945            allParentDone = false;
946            // For a closed region, we will write a close event marker to WAL whose sequence id is
947            // larger than final barrier but still smaller than next region's openSeqNum.
948            // So if the pos is larger than last barrier, we can say we have read the event marker
949            // which means the parent region has been fully pushed.
950            LOG.info(
951              Bytes.toString(encodedName) + " can not start pushing because parent region's" +
952                " log has not been fully pushed: parent=" + Bytes.toString(region) + " pos=" + pos +
953                " barriers=" + Arrays.toString(barriers.toArray()));
954            break;
955          }
956        }
957        if (allParentDone) {
958          return;
959        } else {
960          Thread.sleep(replicationWaitTime);
961        }
962      }
963
964    }
965
966    while (true) {
967      long pos =
968          MetaTableAccessor.getReplicationPositionForOnePeer(connection, encodedName, peerId);
969      if (seq <= pos) {
970        // Case 5
971      }
972      if (pos >= 0) {
973        // Case 4
974        int posInterval = Collections.binarySearch(barriers, pos);
975        if (posInterval < 0) {
976          posInterval = -posInterval - 1;// get the insert position if negative
977        }
978        if (posInterval == interval || pos == barriers.get(interval - 1) - 1) {
979          return;
980        }
981      }
982
983      LOG.info(Bytes.toString(encodedName) + " can not start pushing to peer " + peerId +
984        " because previous log has not been pushed: sequence=" + seq + " pos=" + pos +
985        " barriers=" + Arrays.toString(barriers.toArray()));
986      Thread.sleep(replicationWaitTime);
987    }
988  }
