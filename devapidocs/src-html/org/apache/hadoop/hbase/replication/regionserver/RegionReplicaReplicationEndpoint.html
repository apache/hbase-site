108  @Override
109  public void init(Context context) throws IOException {
110    super.init(context);
111
112    this.conf = HBaseConfiguration.create(context.getConfiguration());
113    this.tableDescriptors = context.getTableDescriptors();
114
115    // HRS multiplies client retries by 10 globally for meta operations, but we do not want this.
116    // We are resetting it here because we want default number of retries (35) rather than 10 times
117    // that which makes very long retries for disabled tables etc.
118    int defaultNumRetries = conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
119      HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
120    if (defaultNumRetries > 10) {
121      int mult = conf.getInt(HConstants.HBASE_CLIENT_SERVERSIDE_RETRIES_MULTIPLIER,
122        HConstants.DEFAULT_HBASE_CLIENT_SERVERSIDE_RETRIES_MULTIPLIER);
123      defaultNumRetries = defaultNumRetries / mult; // reset if HRS has multiplied this already
124    }
125
126    conf.setInt(HConstants.HBASE_CLIENT_SERVERSIDE_RETRIES_MULTIPLIER, 1);
127    int numRetries = conf.getInt(CLIENT_RETRIES_NUMBER, defaultNumRetries);
128    conf.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, numRetries);
129
130    this.numWriterThreads = this.conf.getInt(
131      "hbase.region.replica.replication.writer.threads", 3);
132    controller = new PipelineController();
133    entryBuffers = new EntryBuffers(controller,
134      this.conf.getInt("hbase.region.replica.replication.buffersize",
135          128*1024*1024));
136
137    // use the regular RPC timeout for replica replication RPC's
138    this.operationTimeout = conf.getInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT,
139      HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
140  }

142  @Override
143  protected void doStart() {
144    try {
145      connection = (ClusterConnection) ConnectionFactory.createConnection(this.conf);
146      this.pool = getDefaultThreadPool(conf);
147      outputSink = new RegionReplicaOutputSink(controller, tableDescriptors, entryBuffers,
148        connection, pool, numWriterThreads, operationTimeout);
149      outputSink.startWriterThreads();
150      super.doStart();
151    } catch (IOException ex) {
152      LOG.warn("Received exception while creating connection :" + ex);
153      notifyFailed(ex);
154    }
155  }
156
157  @Override
158  protected void doStop() {
159    if (outputSink != null) {
160      try {
161        outputSink.finishWritingAndClose();
162      } catch (IOException ex) {
163        LOG.warn("Got exception while trying to close OutputSink", ex);
164      }
165    }
166    if (this.pool != null) {
167      this.pool.shutdownNow();
168      try {
169        // wait for 10 sec
170        boolean shutdown = this.pool.awaitTermination(10000, TimeUnit.MILLISECONDS);
171        if (!shutdown) {
172          LOG.warn("Failed to shutdown the thread pool after 10 seconds");
173        }
174      } catch (InterruptedException e) {
175        LOG.warn("Got interrupted while waiting for the thread pool to shut down" + e);
176      }
177    }
178    if (connection != null) {
179      try {
180        connection.close();
181      } catch (IOException ex) {
182        LOG.warn("Got exception closing connection :" + ex);
183      }
184    }
185    super.doStop();
186  }

188  /**
189   * Returns a Thread pool for the RPC's to region replicas. Similar to
190   * Connection's thread pool.
191   */
192  private ExecutorService getDefaultThreadPool(Configuration conf) {
193    int maxThreads = conf.getInt("hbase.region.replica.replication.threads.max", 256);
194    if (maxThreads == 0) {
195      maxThreads = Runtime.getRuntime().availableProcessors() * 8;
196    }
197    long keepAliveTime = conf.getLong("hbase.region.replica.replication.threads.keepalivetime", 60);
198    LinkedBlockingQueue<Runnable> workQueue =
199        new LinkedBlockingQueue<>(maxThreads *
200            conf.getInt(HConstants.HBASE_CLIENT_MAX_TOTAL_TASKS,
201              HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS));
202    ThreadPoolExecutor tpe = new ThreadPoolExecutor(
203      maxThreads,
204      maxThreads,
205      keepAliveTime,
206      TimeUnit.SECONDS,
207      workQueue,
208      Threads.newDaemonThreadFactory(this.getClass().getSimpleName() + "-rpc-shared-"));
209    tpe.allowCoreThreadTimeOut(true);
210    return tpe;
211  }

213  @Override
214  public boolean replicate(ReplicateContext replicateContext) {
215    /* A note on batching in RegionReplicaReplicationEndpoint (RRRE):
216     *
217     * RRRE relies on batching from two different mechanisms. The first is the batching from
218     * ReplicationSource since RRRE is a ReplicationEndpoint driven by RS. RS reads from a single
219     * WAL file filling up a buffer of heap size "replication.source.size.capacity"(64MB) or at most
220     * "replication.source.nb.capacity" entries or until it sees the end of file (in live tailing).
221     * Then RS passes all the buffered edits in this replicate() call context. RRRE puts the edits
222     * to the WALSplitter.EntryBuffers which is a blocking buffer space of up to
223     * "hbase.region.replica.replication.buffersize" (128MB) in size. This buffer splits the edits
224     * based on regions.
225     *
226     * There are "hbase.region.replica.replication.writer.threads"(default 3) writer threads which
227     * pick largest per-region buffer and send it to the SinkWriter (see RegionReplicaOutputSink).
228     * The SinkWriter in this case will send the wal edits to all secondary region replicas in
229     * parallel via a retrying rpc call. EntryBuffers guarantees that while a buffer is
230     * being written to the sink, another buffer for the same region will not be made available to
231     * writers ensuring regions edits are not replayed out of order.
232     *
233     * The replicate() call won't return until all the buffers are sent and ack'd by the sinks so
234     * that the replication can assume all edits are persisted. We may be able to do a better
235     * pipelining between the replication thread and output sinks later if it becomes a bottleneck.
236     */
237
238    while (this.isRunning()) {
239      try {
240        for (Entry entry: replicateContext.getEntries()) {
241          entryBuffers.appendEntry(entry);
242        }
243        outputSink.flush(); // make sure everything is flushed
244        ctx.getMetrics().incrLogEditsFiltered(
245          outputSink.getSkippedEditsCounter().getAndSet(0));
246        return true;
247      } catch (InterruptedException e) {
248        Thread.currentThread().interrupt();
249        return false;
250      } catch (IOException e) {
251        LOG.warn("Received IOException while trying to replicate"
252            + StringUtils.stringifyException(e));
253      }
254    }
255
256    return false;
257  }

270  static class RegionReplicaOutputSink extends OutputSink {
271    private final RegionReplicaSinkWriter sinkWriter;
272    private final TableDescriptors tableDescriptors;
273    private final Cache<TableName, Boolean> memstoreReplicationEnabled;
274
275    public RegionReplicaOutputSink(PipelineController controller, TableDescriptors tableDescriptors,
276        EntryBuffers entryBuffers, ClusterConnection connection, ExecutorService pool,
277        int numWriters, int operationTimeout) {
278      super(controller, entryBuffers, numWriters);
279      this.sinkWriter = new RegionReplicaSinkWriter(this, connection, pool, operationTimeout);
280      this.tableDescriptors = tableDescriptors;
281
282      // A cache for the table "memstore replication enabled" flag.
283      // It has a default expiry of 5 sec. This means that if the table is altered
284      // with a different flag value, we might miss to replicate for that amount of
285      // time. But this cache avoid the slow lookup and parsing of the TableDescriptor.
286      int memstoreReplicationEnabledCacheExpiryMs = connection.getConfiguration()
287        .getInt("hbase.region.replica.replication.cache.memstoreReplicationEnabled.expiryMs", 5000);
288      this.memstoreReplicationEnabled = CacheBuilder.newBuilder()
289        .expireAfterWrite(memstoreReplicationEnabledCacheExpiryMs, TimeUnit.MILLISECONDS)
290        .initialCapacity(10)
291        .maximumSize(1000)
292        .build();
293    }
294
295    @Override
296    public void append(RegionEntryBuffer buffer) throws IOException {
297      List<Entry> entries = buffer.getEntryBuffer();
298
299      if (entries.isEmpty() || entries.get(0).getEdit().getCells().isEmpty()) {
300        return;
301      }
302
303      // meta edits (e.g. flush) are always replicated.
304      // data edits (e.g. put) are replicated if the table requires them.
305      if (!requiresReplication(buffer.getTableName(), entries)) {
306        return;
307      }
308
309      sinkWriter.append(buffer.getTableName(), buffer.getEncodedRegionName(),
310        CellUtil.cloneRow(entries.get(0).getEdit().getCells().get(0)), entries);
311    }
312
313    @Override
314    public boolean flush() throws IOException {
315      // nothing much to do for now. Wait for the Writer threads to finish up
316      // append()'ing the data.
317      entryBuffers.waitUntilDrained();
318      return super.flush();
319    }
320
321    @Override
322    public boolean keepRegionEvent(Entry entry) {
323      return true;
324    }
325
326    @Override
327    public List<Path> finishWritingAndClose() throws IOException {
328      finishWriting(true);
329      return null;
330    }
331
332    @Override
333    public Map<byte[], Long> getOutputCounts() {
334      return null; // only used in tests
335    }
336
337    @Override
338    public int getNumberOfRecoveredRegions() {
339      return 0;
340    }
341
342    AtomicLong getSkippedEditsCounter() {
343      return skippedEdits;
344    }
345
346    /**
347     * returns true if the specified entry must be replicated.
348     * We should always replicate meta operations (e.g. flush)
349     * and use the user HTD flag to decide whether or not replicate the memstore.
350     */
351    private boolean requiresReplication(final TableName tableName, final List<Entry> entries)
352        throws IOException {
353      // unit-tests may not the TableDescriptors, bypass the check and always replicate
354      if (tableDescriptors == null) return true;
355
356      Boolean requiresReplication = memstoreReplicationEnabled.getIfPresent(tableName);
357      if (requiresReplication == null) {
358        // check if the table requires memstore replication
359        // some unit-test drop the table, so we should do a bypass check and always replicate.
360        TableDescriptor htd = tableDescriptors.get(tableName);
361        requiresReplication = htd == null || htd.hasRegionMemStoreReplication();
362        memstoreReplicationEnabled.put(tableName, requiresReplication);
363      }
364
365      // if memstore replication is not required, check the entries.
366      // meta edits (e.g. flush) must be always replicated.
367      if (!requiresReplication) {
368        int skipEdits = 0;
369        java.util.Iterator<Entry> it = entries.iterator();
370        while (it.hasNext()) {
371          Entry entry = it.next();
372          if (entry.getEdit().isMetaEdit()) {
373            requiresReplication = true;
374          } else {
375            it.remove();
376            skipEdits++;
377          }
378        }
379        skippedEdits.addAndGet(skipEdits);
380      }
381      return requiresReplication;
382    }

385  static class RegionReplicaSinkWriter extends SinkWriter {
386    RegionReplicaOutputSink sink;
387    ClusterConnection connection;
388    RpcControllerFactory rpcControllerFactory;
389    RpcRetryingCallerFactory rpcRetryingCallerFactory;
390    int operationTimeout;
391    ExecutorService pool;
392    Cache<TableName, Boolean> disabledAndDroppedTables;
393
394    public RegionReplicaSinkWriter(RegionReplicaOutputSink sink, ClusterConnection connection,
395        ExecutorService pool, int operationTimeout) {
396      this.sink = sink;
397      this.connection = connection;
398      this.operationTimeout = operationTimeout;
399      this.rpcRetryingCallerFactory
400        = RpcRetryingCallerFactory.instantiate(connection.getConfiguration());
401      this.rpcControllerFactory = RpcControllerFactory.instantiate(connection.getConfiguration());
402      this.pool = pool;
403
404      int nonExistentTableCacheExpiryMs = connection.getConfiguration()
405        .getInt("hbase.region.replica.replication.cache.disabledAndDroppedTables.expiryMs", 5000);
406      // A cache for non existing tables that have a default expiry of 5 sec. This means that if the
407      // table is created again with the same name, we might miss to replicate for that amount of
408      // time. But this cache prevents overloading meta requests for every edit from a deleted file.
409      disabledAndDroppedTables = CacheBuilder.newBuilder()
410        .expireAfterWrite(nonExistentTableCacheExpiryMs, TimeUnit.MILLISECONDS)
411        .initialCapacity(10)
412        .maximumSize(1000)
413        .build();
414    }
415
416    public void append(TableName tableName, byte[] encodedRegionName, byte[] row,
417        List<Entry> entries) throws IOException {
418
419      if (disabledAndDroppedTables.getIfPresent(tableName) != null) {
420        if (LOG.isTraceEnabled()) {
421          LOG.trace("Skipping " + entries.size() + " entries because table " + tableName
422            + " is cached as a disabled or dropped table");
423          for (Entry entry : entries) {
424            LOG.trace("Skipping : " + entry);
425          }
426        }
427        sink.getSkippedEditsCounter().addAndGet(entries.size());
428        return;
429      }
430
431      // If the table is disabled or dropped, we should not replay the entries, and we can skip
432      // replaying them. However, we might not know whether the table is disabled until we
433      // invalidate the cache and check from meta
434      RegionLocations locations = null;
435      boolean useCache = true;
436      while (true) {
437        // get the replicas of the primary region
438        try {
439          locations = RegionReplicaReplayCallable
440              .getRegionLocations(connection, tableName, row, useCache, 0);
441
442          if (locations == null) {
443            throw new HBaseIOException("Cannot locate locations for "
444                + tableName + ", row:" + Bytes.toStringBinary(row));
445          }
446        } catch (TableNotFoundException e) {
447          if (LOG.isTraceEnabled()) {
448            LOG.trace("Skipping " + entries.size() + " entries because table " + tableName
449              + " is dropped. Adding table to cache.");
450            for (Entry entry : entries) {
451              LOG.trace("Skipping : " + entry);
452            }
453          }
454          disabledAndDroppedTables.put(tableName, Boolean.TRUE); // put to cache. Value ignored
455          // skip this entry
456          sink.getSkippedEditsCounter().addAndGet(entries.size());
457          return;
458        }
459
460        // check whether we should still replay this entry. If the regions are changed, or the
461        // entry is not coming from the primary region, filter it out.
462        HRegionLocation primaryLocation = locations.getDefaultRegionLocation();
463        if (!Bytes.equals(primaryLocation.getRegionInfo().getEncodedNameAsBytes(),
464          encodedRegionName)) {
465          if (useCache) {
466            useCache = false;
467            continue; // this will retry location lookup
468          }
469          if (LOG.isTraceEnabled()) {
470            LOG.trace("Skipping " + entries.size() + " entries in table " + tableName
471              + " because located region " + primaryLocation.getRegionInfo().getEncodedName()
472              + " is different than the original region " + Bytes.toStringBinary(encodedRegionName)
473              + " from WALEdit");
474            for (Entry entry : entries) {
475              LOG.trace("Skipping : " + entry);
476            }
477          }
478          sink.getSkippedEditsCounter().addAndGet(entries.size());
479          return;
480        }
481        break;
482      }
483
484      if (locations.size() == 1) {
485        return;
486      }
487
488      ArrayList<Future<ReplicateWALEntryResponse>> tasks = new ArrayList<>(locations.size() - 1);
489
490      // All passed entries should belong to one region because it is coming from the EntryBuffers
491      // split per region. But the regions might split and merge (unlike log recovery case).
492      for (int replicaId = 0; replicaId < locations.size(); replicaId++) {
493        HRegionLocation location = locations.getRegionLocation(replicaId);
494        if (!RegionReplicaUtil.isDefaultReplica(replicaId)) {
495          RegionInfo regionInfo = location == null
496              ? RegionReplicaUtil.getRegionInfoForReplica(
497                locations.getDefaultRegionLocation().getRegionInfo(), replicaId)
498              : location.getRegionInfo();
499          RegionReplicaReplayCallable callable = new RegionReplicaReplayCallable(connection,
500            rpcControllerFactory, tableName, location, regionInfo, row, entries,
501            sink.getSkippedEditsCounter());
502           Future<ReplicateWALEntryResponse> task = pool.submit(
503             new RetryingRpcCallable<>(rpcRetryingCallerFactory, callable, operationTimeout));
504           tasks.add(task);
505        }
506      }
507
508      boolean tasksCancelled = false;
509      for (Future<ReplicateWALEntryResponse> task : tasks) {
510        try {
511          task.get();
512        } catch (InterruptedException e) {
513          throw new InterruptedIOException(e.getMessage());
514        } catch (ExecutionException e) {
515          Throwable cause = e.getCause();
516          if (cause instanceof IOException) {
517            // The table can be disabled or dropped at this time. For disabled tables, we have no
518            // cheap mechanism to detect this case because meta does not contain this information.
519            // ClusterConnection.isTableDisabled() is a zk call which we cannot do for every replay
520            // RPC. So instead we start the replay RPC with retries and check whether the table is
521            // dropped or disabled which might cause SocketTimeoutException, or
522            // RetriesExhaustedException or similar if we get IOE.
523            if (cause instanceof TableNotFoundException || connection.isTableDisabled(tableName)) {
524              if (LOG.isTraceEnabled()) {
525                LOG.trace("Skipping " + entries.size() + " entries in table " + tableName
526                  + " because received exception for dropped or disabled table", cause);
527                for (Entry entry : entries) {
528                  LOG.trace("Skipping : " + entry);
529                }
530              }
531              disabledAndDroppedTables.put(tableName, Boolean.TRUE); // put to cache for later.
532              if (!tasksCancelled) {
533                sink.getSkippedEditsCounter().addAndGet(entries.size());
534                tasksCancelled = true; // so that we do not add to skipped counter again
535              }
536              continue;
537            }
538            // otherwise rethrow
539            throw (IOException)cause;
540          }
541          // unexpected exception
542          throw new IOException(cause);
543        }
544      }
545    }
546  }

548  static class RetryingRpcCallable<V> implements Callable<V> {
549    RpcRetryingCallerFactory factory;
550    RetryingCallable<V> callable;
551    int timeout;
552    public RetryingRpcCallable(RpcRetryingCallerFactory factory, RetryingCallable<V> callable,
553        int timeout) {
554      this.factory = factory;
555      this.callable = callable;
556      this.timeout = timeout;
557    }
558    @Override
559    public V call() throws Exception {
560      return factory.<V>newCaller().callWithRetries(callable, timeout);
561    }

564  /**
565   * Calls replay on the passed edits for the given set of entries belonging to the region. It skips
566   * the entry if the region boundaries have changed or the region is gone.
567   */
568  static class RegionReplicaReplayCallable extends
569      RegionAdminServiceCallable<ReplicateWALEntryResponse> {
570    private final List<Entry> entries;
571    private final byte[] initialEncodedRegionName;
572    private final AtomicLong skippedEntries;
573
574    public RegionReplicaReplayCallable(ClusterConnection connection,
575        RpcControllerFactory rpcControllerFactory, TableName tableName,
576        HRegionLocation location, RegionInfo regionInfo, byte[] row,List<Entry> entries,
577        AtomicLong skippedEntries) {
578      super(connection, rpcControllerFactory, location, tableName, row, regionInfo.getReplicaId());
579      this.entries = entries;
580      this.skippedEntries = skippedEntries;
581      this.initialEncodedRegionName = regionInfo.getEncodedNameAsBytes();
582    }
583
584    @Override
585    public ReplicateWALEntryResponse call(HBaseRpcController controller) throws Exception {
586      // Check whether we should still replay this entry. If the regions are changed, or the
587      // entry is not coming form the primary region, filter it out because we do not need it.
588      // Regions can change because of (1) region split (2) region merge (3) table recreated
589      boolean skip = false;
590      if (!Bytes.equals(location.getRegionInfo().getEncodedNameAsBytes(),
591          initialEncodedRegionName)) {
592        skip = true;
593      }
594      if (!this.entries.isEmpty() && !skip) {
595        Entry[] entriesArray = new Entry[this.entries.size()];
596        entriesArray = this.entries.toArray(entriesArray);
597
598        // set the region name for the target region replica
599        Pair<AdminProtos.ReplicateWALEntryRequest, CellScanner> p =
600            ReplicationProtbufUtil.buildReplicateWALEntryRequest(entriesArray, location
601                .getRegionInfo().getEncodedNameAsBytes(), null, null, null);
602        controller.setCellScanner(p.getSecond());
603        return stub.replay(controller, p.getFirst());
604      }
605
606      if (skip) {
607        if (LOG.isTraceEnabled()) {
608          LOG.trace("Skipping " + entries.size() + " entries in table " + tableName
609            + " because located region " + location.getRegionInfo().getEncodedName()
610            + " is different than the original region "
611            + Bytes.toStringBinary(initialEncodedRegionName) + " from WALEdit");
612          for (Entry entry : entries) {
613            LOG.trace("Skipping : " + entry);
614          }
615        }
616        skippedEntries.addAndGet(entries.size());
617      }
618      return ReplicateWALEntryResponse.newBuilder().build();
619    }
620  }
