051/**
052 * Gateway to Replication. Used by {@link org.apache.hadoop.hbase.regionserver.HRegionServer}.
053 */
054@InterfaceAudience.Private
055public class Replication implements ReplicationSourceService {
056  private static final Logger LOG =
057      LoggerFactory.getLogger(Replication.class);
058  private boolean isReplicationForBulkLoadDataEnabled;
059  private ReplicationSourceManager replicationManager;
060  private ReplicationQueueStorage queueStorage;
061  private ReplicationPeers replicationPeers;
062  private Configuration conf;
063  private SyncReplicationPeerInfoProvider syncReplicationPeerInfoProvider;
064  // Hosting server
065  private Server server;
066  private int statsPeriodInSecond;
067  // ReplicationLoad to access replication metrics
068  private ReplicationLoad replicationLoad;
069  private MetricsReplicationGlobalSourceSource globalMetricsSource;
070
071  private PeerProcedureHandler peerProcedureHandler;
072
073  /**
074   * Empty constructor
075   */
076  public Replication() {
077  }
078
079  @Override
080  public void initialize(Server server, FileSystem fs, Path logDir, Path oldLogDir,
081      WALFactory walFactory) throws IOException {
082    this.server = server;
083    this.conf = this.server.getConfiguration();
084    this.isReplicationForBulkLoadDataEnabled =
085      ReplicationUtils.isReplicationForBulkLoadDataEnabled(this.conf);
086    if (this.isReplicationForBulkLoadDataEnabled) {
087      if (conf.get(HConstants.REPLICATION_CLUSTER_ID) == null
088          || conf.get(HConstants.REPLICATION_CLUSTER_ID).isEmpty()) {
089        throw new IllegalArgumentException(HConstants.REPLICATION_CLUSTER_ID
090            + " cannot be null/empty when " + HConstants.REPLICATION_BULKLOAD_ENABLE_KEY
091            + " is set to true.");
092      }
093    }
094
095    try {
096      this.queueStorage =
097          ReplicationStorageFactory.getReplicationQueueStorage(server.getZooKeeper(), conf);
098      this.replicationPeers =
099          ReplicationFactory.getReplicationPeers(server.getZooKeeper(), this.conf);
100      this.replicationPeers.init();
101    } catch (Exception e) {
102      throw new IOException("Failed replication handler create", e);
103    }
104    UUID clusterId = null;
105    try {
106      clusterId = ZKClusterId.getUUIDForCluster(this.server.getZooKeeper());
107    } catch (KeeperException ke) {
108      throw new IOException("Could not read cluster id", ke);
109    }
110    SyncReplicationPeerMappingManager mapping = new SyncReplicationPeerMappingManager();
111    this.globalMetricsSource = CompatibilitySingletonFactory
112        .getInstance(MetricsReplicationSourceFactory.class).getGlobalSource();
113    this.replicationManager = new ReplicationSourceManager(queueStorage, replicationPeers, conf,
114      this.server, fs, logDir, oldLogDir, clusterId, walFactory, mapping, globalMetricsSource);
115    this.syncReplicationPeerInfoProvider =
116        new SyncReplicationPeerInfoProviderImpl(replicationPeers, mapping);
117    PeerActionListener peerActionListener = PeerActionListener.DUMMY;
118    // Get the user-space WAL provider
119    WALProvider walProvider = walFactory != null? walFactory.getWALProvider(): null;
120    if (walProvider != null) {
121      walProvider
122        .addWALActionsListener(new ReplicationSourceWALActionListener(conf, replicationManager));
123      if (walProvider instanceof SyncReplicationWALProvider) {
124        SyncReplicationWALProvider syncWALProvider = (SyncReplicationWALProvider) walProvider;
125        peerActionListener = syncWALProvider;
126        syncWALProvider.setPeerInfoProvider(syncReplicationPeerInfoProvider);
127        // for sync replication state change, we need to reload the state twice, you can see the
128        // code in PeerProcedureHandlerImpl, so here we need to go over the sync replication peers
129        // to see if any of them are in the middle of the two refreshes, if so, we need to manually
130        // repeat the action we have done in the first refresh, otherwise when the second refresh
131        // comes we will be in trouble, such as NPE.
132        replicationPeers.getAllPeerIds().stream().map(replicationPeers::getPeer)
133            .filter(p -> p.getPeerConfig().isSyncReplication())
134            .filter(p -> p.getNewSyncReplicationState() != SyncReplicationState.NONE)
135            .forEach(p -> syncWALProvider.peerSyncReplicationStateChange(p.getId(),
136              p.getSyncReplicationState(), p.getNewSyncReplicationState(), 0));
137      }
138    }
139    this.statsPeriodInSecond =
140        this.conf.getInt("replication.stats.thread.period.seconds", 5 * 60);
141    this.replicationLoad = new ReplicationLoad();
142
143    this.peerProcedureHandler =
144      new PeerProcedureHandlerImpl(replicationManager, peerActionListener);
145  }
146
147  @Override
148  public PeerProcedureHandler getPeerProcedureHandler() {
149    return peerProcedureHandler;
150  }
151
152  /**
153   * Stops replication service.
154   */
155  @Override
156  public void stopReplicationService() {
157    this.replicationManager.join();
158  }
159
160  /**
161   * If replication is enabled and this cluster is a master,
162   * it starts
163   */
164  @Override
165  public void startReplicationService() throws IOException {
166    this.replicationManager.init();
167    this.server.getChoreService().scheduleChore(
168      new ReplicationStatisticsChore("ReplicationSourceStatistics", server,
169          (int) TimeUnit.SECONDS.toMillis(statsPeriodInSecond)));
170    LOG.info("{} started", this.server.toString());
171  }
172
173  /**
174   * Get the replication sources manager
175   * @return the manager if replication is enabled, else returns false
176   */
177  public ReplicationSourceManager getReplicationManager() {
178    return this.replicationManager;
179  }
180
181  void addHFileRefsToQueue(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
182      throws IOException {
183    try {
184      this.replicationManager.addHFileRefs(tableName, family, pairs);
185    } catch (IOException e) {
186      LOG.error("Failed to add hfile references in the replication queue.", e);
187      throw e;
188    }
189  }
190
191  /**
192   * Statistics task. Periodically prints the cache statistics to the log.
193   */
194  private final class ReplicationStatisticsChore extends ScheduledChore {
195
196    ReplicationStatisticsChore(String name, Stoppable stopper, int period) {
197      super(name, stopper, period);
198    }
199
200    @Override
201    protected void chore() {
202      printStats(replicationManager.getStats());
203    }
204
205    private void printStats(String stats) {
206      if (!stats.isEmpty()) {
207        LOG.info(stats);
208      }
209    }
210  }
211
212  @Override
213  public ReplicationLoad refreshAndGetReplicationLoad() {
214    if (this.replicationLoad == null) {
215      return null;
216    }
217    // always build for latest data
218    List<ReplicationSourceInterface> allSources = new ArrayList<>();
219    allSources.addAll(this.replicationManager.getSources());
220    allSources.addAll(this.replicationManager.getOldSources());
221    this.replicationLoad.buildReplicationLoad(allSources, null);
222    return this.replicationLoad;
223  }
224
225  @Override
226  public SyncReplicationPeerInfoProvider getSyncReplicationPeerInfoProvider() {
227    return syncReplicationPeerInfoProvider;
228  }
229
230  @Override
231  public ReplicationPeers getReplicationPeers() {
232    return replicationPeers;
233  }
