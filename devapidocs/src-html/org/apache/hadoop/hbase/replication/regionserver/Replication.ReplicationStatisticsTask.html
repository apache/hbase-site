059/**
060 * Gateway to Replication. Used by {@link org.apache.hadoop.hbase.regionserver.HRegionServer}.
061 */
062@InterfaceAudience.Private
063public class Replication implements ReplicationSourceService, ReplicationSinkService {
064  private static final Logger LOG =
065      LoggerFactory.getLogger(Replication.class);
066  private boolean isReplicationForBulkLoadDataEnabled;
067  private ReplicationSourceManager replicationManager;
068  private ReplicationQueueStorage queueStorage;
069  private ReplicationPeers replicationPeers;
070  private ReplicationTracker replicationTracker;
071  private Configuration conf;
072  private ReplicationSink replicationSink;
073  private SyncReplicationPeerInfoProvider syncReplicationPeerInfoProvider;
074  // Hosting server
075  private Server server;
076  /** Statistics thread schedule pool */
077  private ScheduledExecutorService scheduleThreadPool;
078  private int statsThreadPeriod;
079  // ReplicationLoad to access replication metrics
080  private ReplicationLoad replicationLoad;
081  private MetricsReplicationGlobalSourceSource globalMetricsSource;
082
083  private PeerProcedureHandler peerProcedureHandler;
084
085  /**
086   * Empty constructor
087   */
088  public Replication() {
089  }

091  @Override
092  public void initialize(Server server, FileSystem fs, Path logDir, Path oldLogDir,
093      WALFactory walFactory) throws IOException {
094    this.server = server;
095    this.conf = this.server.getConfiguration();
096    this.isReplicationForBulkLoadDataEnabled =
097      ReplicationUtils.isReplicationForBulkLoadDataEnabled(this.conf);
098    this.scheduleThreadPool = Executors.newScheduledThreadPool(1,
099      new ThreadFactoryBuilder()
100        .setNameFormat(server.getServerName().toShortString() + "Replication Statistics #%d")
101        .setDaemon(true)
102        .build());
103    if (this.isReplicationForBulkLoadDataEnabled) {
104      if (conf.get(HConstants.REPLICATION_CLUSTER_ID) == null
105          || conf.get(HConstants.REPLICATION_CLUSTER_ID).isEmpty()) {
106        throw new IllegalArgumentException(HConstants.REPLICATION_CLUSTER_ID
107            + " cannot be null/empty when " + HConstants.REPLICATION_BULKLOAD_ENABLE_KEY
108            + " is set to true.");
109      }
110    }
111
112    try {
113      this.queueStorage =
114          ReplicationStorageFactory.getReplicationQueueStorage(server.getZooKeeper(), conf);
115      this.replicationPeers =
116          ReplicationFactory.getReplicationPeers(server.getZooKeeper(), this.conf);
117      this.replicationPeers.init();
118      this.replicationTracker =
119          ReplicationFactory.getReplicationTracker(server.getZooKeeper(), this.server, this.server);
120    } catch (Exception e) {
121      throw new IOException("Failed replication handler create", e);
122    }
123    UUID clusterId = null;
124    try {
125      clusterId = ZKClusterId.getUUIDForCluster(this.server.getZooKeeper());
126    } catch (KeeperException ke) {
127      throw new IOException("Could not read cluster id", ke);
128    }
129    SyncReplicationPeerMappingManager mapping = new SyncReplicationPeerMappingManager();
130    this.globalMetricsSource = CompatibilitySingletonFactory
131        .getInstance(MetricsReplicationSourceFactory.class).getGlobalSource();
132    WALProvider walProvider = walFactory.getWALProvider();
133    this.replicationManager = new ReplicationSourceManager(queueStorage, replicationPeers,
134        replicationTracker, conf, this.server, fs, logDir, oldLogDir, clusterId,
135        walProvider != null ? walProvider.getWALFileLengthProvider() : p -> OptionalLong.empty(),
136        mapping, globalMetricsSource);
137    this.syncReplicationPeerInfoProvider =
138        new SyncReplicationPeerInfoProviderImpl(replicationPeers, mapping);
139    PeerActionListener peerActionListener = PeerActionListener.DUMMY;
140    if (walProvider != null) {
141      walProvider
142        .addWALActionsListener(new ReplicationSourceWALActionListener(conf, replicationManager));
143      if (walProvider instanceof SyncReplicationWALProvider) {
144        SyncReplicationWALProvider syncWALProvider = (SyncReplicationWALProvider) walProvider;
145        peerActionListener = syncWALProvider;
146        syncWALProvider.setPeerInfoProvider(syncReplicationPeerInfoProvider);
147        // for sync replication state change, we need to reload the state twice, you can see the
148        // code in PeerProcedureHandlerImpl, so here we need to go over the sync replication peers
149        // to see if any of them are in the middle of the two refreshes, if so, we need to manually
150        // repeat the action we have done in the first refresh, otherwise when the second refresh
151        // comes we will be in trouble, such as NPE.
152        replicationPeers.getAllPeerIds().stream().map(replicationPeers::getPeer)
153            .filter(p -> p.getPeerConfig().isSyncReplication())
154            .filter(p -> p.getNewSyncReplicationState() != SyncReplicationState.NONE)
155            .forEach(p -> syncWALProvider.peerSyncReplicationStateChange(p.getId(),
156              p.getSyncReplicationState(), p.getNewSyncReplicationState(), 0));
157      }
158    }
159    this.statsThreadPeriod =
160        this.conf.getInt("replication.stats.thread.period.seconds", 5 * 60);
161    LOG.debug("Replication stats-in-log period={} seconds",  this.statsThreadPeriod);
162    this.replicationLoad = new ReplicationLoad();
163
164    this.peerProcedureHandler =
165      new PeerProcedureHandlerImpl(replicationManager, peerActionListener);
166  }
167
168  @Override
169  public PeerProcedureHandler getPeerProcedureHandler() {
170    return peerProcedureHandler;
171  }
172
173  /**
174   * Stops replication service.
175   */
176  @Override
177  public void stopReplicationService() {
178    join();
179  }
180
181  /**
182   * Join with the replication threads
183   */
184  public void join() {
185    this.replicationManager.join();
186    if (this.replicationSink != null) {
187      this.replicationSink.stopReplicationSinkServices();
188    }
189    scheduleThreadPool.shutdown();
190  }
191
192  /**
193   * Carry on the list of log entries down to the sink
194   * @param entries list of entries to replicate
195   * @param cells The data -- the cells -- that <code>entries</code> describes (the entries do not
196   *          contain the Cells we are replicating; they are passed here on the side in this
197   *          CellScanner).
198   * @param replicationClusterId Id which will uniquely identify source cluster FS client
199   *          configurations in the replication configuration directory
200   * @param sourceBaseNamespaceDirPath Path that point to the source cluster base namespace
201   *          directory required for replicating hfiles
202   * @param sourceHFileArchiveDirPath Path that point to the source cluster hfile archive directory
203   */
204  @Override
205  public void replicateLogEntries(List<WALEntry> entries, CellScanner cells,
206      String replicationClusterId, String sourceBaseNamespaceDirPath,
207      String sourceHFileArchiveDirPath) throws IOException {
208    this.replicationSink.replicateEntries(entries, cells, replicationClusterId,
209      sourceBaseNamespaceDirPath, sourceHFileArchiveDirPath);
210  }
211
212  /**
213   * If replication is enabled and this cluster is a master,
214   * it starts
215   */
216  @Override
217  public void startReplicationService() throws IOException {
218    this.replicationManager.init();
219    this.replicationSink = new ReplicationSink(this.conf);
220    this.scheduleThreadPool.scheduleAtFixedRate(
221      new ReplicationStatisticsTask(this.replicationSink, this.replicationManager),
222      statsThreadPeriod, statsThreadPeriod, TimeUnit.SECONDS);
223    LOG.info("{} started", this.server.toString());
224  }
225
226  /**
227   * Get the replication sources manager
228   * @return the manager if replication is enabled, else returns false
229   */
230  public ReplicationSourceManager getReplicationManager() {
231    return this.replicationManager;
232  }
233
234  void addHFileRefsToQueue(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
235      throws IOException {
236    try {
237      this.replicationManager.addHFileRefs(tableName, family, pairs);
238    } catch (IOException e) {
239      LOG.error("Failed to add hfile references in the replication queue.", e);
240      throw e;
241    }
242  }
243
244  /**
245   * Statistics task. Periodically prints the cache statistics to the log.
246   */
247  private final static class ReplicationStatisticsTask implements Runnable {
248
249    private final ReplicationSink replicationSink;
250    private final ReplicationSourceManager replicationManager;
251
252    public ReplicationStatisticsTask(ReplicationSink replicationSink,
253        ReplicationSourceManager replicationManager) {
254      this.replicationManager = replicationManager;
255      this.replicationSink = replicationSink;
256    }
257
258    @Override
259    public void run() {
260      printStats(this.replicationManager.getStats());
261      printStats(this.replicationSink.getStats());
262    }
263
264    private void printStats(String stats) {
265      if (!stats.isEmpty()) {
266        LOG.info(stats);
267      }
268    }
269  }
270
271  @Override
272  public ReplicationLoad refreshAndGetReplicationLoad() {
273    if (this.replicationLoad == null) {
274      return null;
275    }
276    // always build for latest data
277    buildReplicationLoad();
278    return this.replicationLoad;
279  }
280
281  private void buildReplicationLoad() {
282    List<ReplicationSourceInterface> allSources = new ArrayList<>();
283    allSources.addAll(this.replicationManager.getSources());
284    allSources.addAll(this.replicationManager.getOldSources());
285    // get sink
286    MetricsSink sinkMetrics = this.replicationSink.getSinkMetrics();
287    this.replicationLoad.buildReplicationLoad(allSources, sinkMetrics);
288  }
