073/**
074 * Class that handles the source of a replication stream.
075 * Currently does not handle more than 1 slave cluster.
076 * For each slave cluster it selects a random number of peers
077 * using a replication ratio. For example, if replication ration = 0.1
078 * and slave cluster has 100 region servers, 10 will be selected.
079 * <p>
080 * A stream is considered down when we cannot contact a region server on the
081 * peer cluster for more than 55 seconds by default.
082 * </p>
083 */
084@InterfaceAudience.Private
085public class ReplicationSource implements ReplicationSourceInterface {

211    this.maxRetriesMultiplier =
212        this.conf.getInt("replication.source.maxretriesmultiplier", 300); // 5 minutes @ 1 sec per
213    this.queueSizePerGroup = this.conf.getInt("hbase.regionserver.maxlogs", 32);
214    this.queueStorage = queueStorage;
215    this.replicationPeer = replicationPeer;
216    this.manager = manager;
217    this.fs = fs;
218    this.metrics = metrics;
219    this.clusterId = clusterId;
220
221    this.queueId = queueId;
222    this.replicationQueueInfo = new ReplicationQueueInfo(queueId);
223    this.logQueueWarnThreshold = this.conf.getInt("replication.source.log.queue.warn", 2);
224
225    defaultBandwidth = this.conf.getLong("replication.source.per.peer.node.bandwidth", 0);
226    currentBandwidth = getCurrentBandwidth();
227    this.throttler = new ReplicationThrottler((double) currentBandwidth / 10.0);
228    this.totalBufferUsed = manager.getTotalBufferUsed();
229    this.walFileLengthProvider = walFileLengthProvider;
230
231    this.abortOnError = this.conf.getBoolean("replication.source.regionserver.abort",

234    LOG.info("queueId={}, ReplicationSource: {}, currentBandwidth={}", queueId,
235      replicationPeer.getId(), this.currentBandwidth);
236  }

245  @Override
246  public void enqueueLog(Path wal) {
247    if (!this.filterInWALs.test(wal)) {
248      LOG.trace("NOT replicating {}", wal);
249      return;
250    }
251    // Use WAL prefix as the WALGroupId for this peer.
252    String walPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(wal.getName());
253    PriorityBlockingQueue<Path> queue = queues.get(walPrefix);
254    if (queue == null) {
255      queue = new PriorityBlockingQueue<>(queueSizePerGroup,
256        new AbstractFSWALProvider.WALStartTimeComparator());
257      // make sure that we do not use an empty queue when setting up a ReplicationSource, otherwise
258      // the shipper may quit immediately
259      queue.put(wal);
260      queues.put(walPrefix, queue);
261      if (this.isSourceActive() && this.walEntryFilter != null) {
262        // new wal group observed after source startup, start a new worker thread to track it
263        // notice: it's possible that wal enqueued when this.running is set but worker thread
264        // still not launched, so it's necessary to check workerThreads before start the worker
265        tryStartNewShipper(walPrefix, queue);
266      }
267    } else {
268      queue.put(wal);
269    }
270    if (LOG.isTraceEnabled()) {
271      LOG.trace("{} Added wal {} to queue of source {}.", logPeerId(), walPrefix,
272        this.replicationQueueInfo.getQueueId());
273    }
274    this.metrics.incrSizeOfLogQueue();
275    // This will wal a warning for each new wal that gets created above the warn threshold
276    int queueSize = queue.size();
277    if (queueSize > this.logQueueWarnThreshold) {
278      LOG.warn("{} WAL group {} queue size: {} exceeds value of " +
279          "replication.source.log.queue.warn {}", logPeerId(), walPrefix, queueSize,
280        logQueueWarnThreshold);
281    }
282  }

284  @Override
285  public void addHFileRefs(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
286      throws ReplicationException {
287    String peerId = replicationPeer.getId();
288    Set<String> namespaces = replicationPeer.getNamespaces();
289    Map<TableName, List<String>> tableCFMap = replicationPeer.getTableCFs();
290    if (tableCFMap != null) { // All peers with TableCFs
291      List<String> tableCfs = tableCFMap.get(tableName);
292      if (tableCFMap.containsKey(tableName)
293          && (tableCfs == null || tableCfs.contains(Bytes.toString(family)))) {
294        this.queueStorage.addHFileRefs(peerId, pairs);
295        metrics.incrSizeOfHFileRefsQueue(pairs.size());
296      } else {
297        LOG.debug("HFiles will not be replicated belonging to the table {} family {} to peer id {}",
298            tableName, Bytes.toString(family), peerId);
299      }
300    } else if (namespaces != null) { // Only for set NAMESPACES peers
301      if (namespaces.contains(tableName.getNamespaceAsString())) {
302        this.queueStorage.addHFileRefs(peerId, pairs);
303        metrics.incrSizeOfHFileRefsQueue(pairs.size());
304      } else {
305        LOG.debug("HFiles will not be replicated belonging to the table {} family {} to peer id {}",
306            tableName, Bytes.toString(family), peerId);
307      }
308    } else {
309      // user has explicitly not defined any table cfs for replication, means replicate all the
310      // data
311      this.queueStorage.addHFileRefs(peerId, pairs);
312      metrics.incrSizeOfHFileRefsQueue(pairs.size());
313    }
314  }

316  private ReplicationEndpoint createReplicationEndpoint()
317      throws InstantiationException, IllegalAccessException, ClassNotFoundException, IOException {
318    RegionServerCoprocessorHost rsServerHost = null;
319    if (server instanceof HRegionServer) {
320      rsServerHost = ((HRegionServer) server).getRegionServerCoprocessorHost();
321    }
322    String replicationEndpointImpl = replicationPeer.getPeerConfig().getReplicationEndpointImpl();
323
324    ReplicationEndpoint replicationEndpoint;
325    if (replicationEndpointImpl == null) {
326      // Default to HBase inter-cluster replication endpoint; skip reflection
327      replicationEndpoint = new HBaseInterClusterReplicationEndpoint();
328    } else {
329      try {
330        replicationEndpoint = Class.forName(replicationEndpointImpl)
331            .asSubclass(ReplicationEndpoint.class)
332            .getDeclaredConstructor()
333            .newInstance();
334      } catch (NoSuchMethodException | InvocationTargetException e) {
335        throw new IllegalArgumentException(e);
336      }
337    }
338    if (rsServerHost != null) {
339      ReplicationEndpoint newReplicationEndPoint =
340        rsServerHost.postCreateReplicationEndPoint(replicationEndpoint);
341      if (newReplicationEndPoint != null) {
342        // Override the newly created endpoint from the hook with configured end point
343        replicationEndpoint = newReplicationEndPoint;
344      }
345    }
346    return replicationEndpoint;
347  }
348
349  private void initAndStartReplicationEndpoint(ReplicationEndpoint replicationEndpoint)
350      throws IOException, TimeoutException {
351    TableDescriptors tableDescriptors = null;
352    if (server instanceof HRegionServer) {
353      tableDescriptors = ((HRegionServer) server).getTableDescriptors();
354    }
355    replicationEndpoint
356      .init(new ReplicationEndpoint.Context(server, conf, replicationPeer.getConfiguration(), fs,
357        replicationPeer.getId(), clusterId, replicationPeer, metrics, tableDescriptors, server));
358    replicationEndpoint.start();
359    replicationEndpoint.awaitRunning(waitOnEndpointSeconds, TimeUnit.SECONDS);
360  }

362  private void initializeWALEntryFilter(UUID peerClusterId) {
363    // get the WALEntryFilter from ReplicationEndpoint and add it to default filters
364    List<WALEntryFilter> filters = new ArrayList<>(this.baseFilterOutWALEntries);
365    WALEntryFilter filterFromEndpoint = this.replicationEndpoint.getWALEntryfilter();
366    if (filterFromEndpoint != null) {
367      filters.add(filterFromEndpoint);
368    }
369    filters.add(new ClusterMarkingEntryFilter(clusterId, peerClusterId, replicationEndpoint));
370    this.walEntryFilter = new ChainWALEntryFilter(filters);
371  }

373  private void tryStartNewShipper(String walGroupId, PriorityBlockingQueue<Path> queue) {
374    workerThreads.compute(walGroupId, (key, value) -> {
375      if (value != null) {
376        LOG.debug("{} preempted start of worker walGroupId={}", logPeerId(), walGroupId);
377        return value;
378      } else {
379        LOG.debug("{} starting worker for walGroupId={}", logPeerId(), walGroupId);
380        ReplicationSourceShipper worker = createNewShipper(walGroupId, queue);
381        ReplicationSourceWALReader walReader =
382            createNewWALReader(walGroupId, queue, worker.getStartPosition());
383        Threads.setDaemonThreadRunning(
384            walReader, Thread.currentThread().getName()
385            + ".replicationSource.wal-reader." + walGroupId + "," + queueId,
386          (t,e) -> this.uncaughtException(t, e, this.manager, this.getPeerId()));
387        worker.setWALReader(walReader);
388        worker.startup((t,e) -> this.uncaughtException(t, e, this.manager, this.getPeerId()));
389        return worker;
390      }
391    });
392  }

394  @Override
395  public Map<String, ReplicationStatus> getWalGroupStatus() {
396    Map<String, ReplicationStatus> sourceReplicationStatus = new TreeMap<>();
397    long ageOfLastShippedOp, replicationDelay, fileSize;
398    for (Map.Entry<String, ReplicationSourceShipper> walGroupShipper : workerThreads.entrySet()) {
399      String walGroupId = walGroupShipper.getKey();
400      ReplicationSourceShipper shipper = walGroupShipper.getValue();
401      ageOfLastShippedOp = metrics.getAgeOfLastShippedOp(walGroupId);
402      int queueSize = queues.get(walGroupId).size();
403      replicationDelay = metrics.getReplicationDelay();
404      Path currentPath = shipper.getCurrentPath();
405      fileSize = -1;
406      if (currentPath != null) {
407        try {
408          fileSize = getFileSize(currentPath);
409        } catch (IOException e) {
410          LOG.warn("Ignore the exception as the file size of HLog only affects the web ui", e);
411        }
412      } else {
413        currentPath = new Path("NO_LOGS_IN_QUEUE");
414        LOG.warn("{} No replication ongoing, waiting for new log", logPeerId());
415      }
416      ReplicationStatus.ReplicationStatusBuilder statusBuilder = ReplicationStatus.newBuilder();
417      statusBuilder.withPeerId(this.getPeerId())
418          .withQueueSize(queueSize)
419          .withWalGroup(walGroupId)
420          .withCurrentPath(currentPath)
421          .withCurrentPosition(shipper.getCurrentPosition())
422          .withFileSize(fileSize)
423          .withAgeOfLastShippedOp(ageOfLastShippedOp)
424          .withReplicationDelay(replicationDelay);
425      sourceReplicationStatus.put(this.getPeerId() + "=>" + walGroupId, statusBuilder.build());
426    }
427    return sourceReplicationStatus;
428  }

462  protected final void uncaughtException(Thread t, Throwable e,
463      ReplicationSourceManager manager, String peerId) {
464    RSRpcServices.exitIfOOME(e);
465    LOG.error("Unexpected exception in {} currentPath={}",
466      t.getName(), getCurrentPath(), e);
467    if(abortOnError){
468      server.abort("Unexpected exception in " + t.getName(), e);
469    }
470    if(manager != null){
471      while (true) {
472        try {
473          LOG.info("Refreshing replication sources now due to previous error on thread: {}",
474            t.getName());
475          manager.refreshSources(peerId);
476          break;
477        } catch (IOException e1) {
478          LOG.error("Replication sources refresh failed.", e1);
479          sleepForRetries("Sleeping before try refreshing sources again",
480            maxRetriesMultiplier);
481        }
482      }
483    }
484  }

496  @Override
497  public void tryThrottle(int batchSize) throws InterruptedException {
498    checkBandwidthChangeAndResetThrottler();
499    if (throttler.isEnabled()) {
500      long sleepTicks = throttler.getNextSleepInterval(batchSize);
501      if (sleepTicks > 0) {
502        if (LOG.isTraceEnabled()) {
503          LOG.trace("{} To sleep {}ms for throttling control", logPeerId(), sleepTicks);
504        }
505        Thread.sleep(sleepTicks);
506        // reset throttler's cycle start tick when sleep for throttling occurs
507        throttler.resetStartTick();
508      }
509    }
510  }
511
512  private void checkBandwidthChangeAndResetThrottler() {
513    long peerBandwidth = getCurrentBandwidth();
514    if (peerBandwidth != currentBandwidth) {
515      currentBandwidth = peerBandwidth;
516      throttler.setBandwidth((double) currentBandwidth / 10.0);
517      LOG.info("ReplicationSource : {} bandwidth throttling changed, currentBandWidth={}",
518        replicationPeer.getId(), currentBandwidth);
519    }
520  }

522  private long getCurrentBandwidth() {
523    long peerBandwidth = replicationPeer.getPeerBandwidth();
524    // user can set peer bandwidth to 0 to use default bandwidth
525    return peerBandwidth != 0 ? peerBandwidth : defaultBandwidth;
526  }

528  /**
529   * Do the sleeping logic
530   * @param msg Why we sleep
531   * @param sleepMultiplier by how many times the default sleeping time is augmented
532   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
533   */
534  protected boolean sleepForRetries(String msg, int sleepMultiplier) {
535    try {
536      if (LOG.isTraceEnabled()) {
537        LOG.trace("{} {}, sleeping {} times {}",
538          logPeerId(), msg, sleepForRetries, sleepMultiplier);
539      }
540      Thread.sleep(this.sleepForRetries * sleepMultiplier);
541    } catch (InterruptedException e) {
542      if(LOG.isDebugEnabled()) {
543        LOG.debug("{} Interrupted while sleeping between retries", logPeerId());
544      }
545      Thread.currentThread().interrupt();
546    }
547    return sleepMultiplier < maxRetriesMultiplier;
548  }

550  private void initialize() {
551    int sleepMultiplier = 1;
552    while (this.isSourceActive()) {
553      ReplicationEndpoint replicationEndpoint;
554      try {
555        replicationEndpoint = createReplicationEndpoint();
556      } catch (Exception e) {
557        LOG.warn("{} error creating ReplicationEndpoint, retry", logPeerId(), e);
558        if (sleepForRetries("Error creating ReplicationEndpoint", sleepMultiplier)) {
559          sleepMultiplier++;
560        }
561        continue;
562      }
563
564      try {
565        initAndStartReplicationEndpoint(replicationEndpoint);
566        this.replicationEndpoint = replicationEndpoint;
567        break;
568      } catch (Exception e) {
569        LOG.warn("{} Error starting ReplicationEndpoint, retry", logPeerId(), e);
570        replicationEndpoint.stop();
571        if (sleepForRetries("Error starting ReplicationEndpoint", sleepMultiplier)) {
572          sleepMultiplier++;
573        } else {
574          retryStartup.set(!this.abortOnError);
575          this.startupOngoing.set(false);
576          throw new RuntimeException("Exhausted retries to start replication endpoint.");
577        }
578      }
579    }
580
581    if (!this.isSourceActive()) {
582      retryStartup.set(!this.abortOnError);
583      this.startupOngoing.set(false);
584      throw new IllegalStateException("Source should be active.");
585    }
586
587    sleepMultiplier = 1;
588    UUID peerClusterId;
589    // delay this until we are in an asynchronous thread
590    for (;;) {
591      peerClusterId = replicationEndpoint.getPeerUUID();
592      if (this.isSourceActive() && peerClusterId == null) {
593        if(LOG.isDebugEnabled()) {
594          LOG.debug("{} Could not connect to Peer ZK. Sleeping for {} millis", logPeerId(),
595            (this.sleepForRetries * sleepMultiplier));
596        }
597        if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
598          sleepMultiplier++;
599        }
600      } else {
601        break;
602      }
603    }
604
605    if(!this.isSourceActive()) {
606      retryStartup.set(!this.abortOnError);
607      this.startupOngoing.set(false);
608      throw new IllegalStateException("Source should be active.");
609    }
610    LOG.info("{} queueId={} is replicating from cluster={} to cluster={}",
611      logPeerId(), this.replicationQueueInfo.getQueueId(), clusterId, peerClusterId);
612
613    initializeWALEntryFilter(peerClusterId);
614    // start workers
615    for (Map.Entry<String, PriorityBlockingQueue<Path>> entry : queues.entrySet()) {
616      String walGroupId = entry.getKey();
617      PriorityBlockingQueue<Path> queue = entry.getValue();
618      tryStartNewShipper(walGroupId, queue);
619    }
620    this.startupOngoing.set(false);
621  }
622
623  @Override
624  public void startup() {
625    // mark we are running now
626    this.sourceRunning = true;
627    startupOngoing.set(true);
628    initThread = new Thread(this::initialize);
629    Threads.setDaemonThreadRunning(initThread,
630      Thread.currentThread().getName() + ".replicationSource," + this.queueId,
631      (t,e) -> {
632        //if first initialization attempt failed, and abortOnError is false, we will
633        //keep looping in this thread until initialize eventually succeeds,
634        //while the server main startup one can go on with its work.
635        sourceRunning = false;
636        uncaughtException(t, e, null, null);
637        retryStartup.set(!this.abortOnError);
638        do {
639          if(retryStartup.get()) {
640            this.sourceRunning = true;
641            startupOngoing.set(true);
642            retryStartup.set(false);
643            try {
644              initialize();
645            } catch(Throwable error){
646              sourceRunning = false;
647              uncaughtException(t, error, null, null);
648              retryStartup.set(!this.abortOnError);
649            }
650          }
651        } while ((this.startupOngoing.get() || this.retryStartup.get()) && !this.abortOnError);
652      });
653  }

670  public void terminate(String reason, Exception cause, boolean clearMetrics,
671      boolean join) {
672    if (cause == null) {
673      LOG.info("{} Closing source {} because: {}", logPeerId(), this.queueId, reason);
674    } else {
675      LOG.error("{} Closing source {} because an error occurred: {}",
676        logPeerId(), this.queueId, reason, cause);
677    }
678    this.sourceRunning = false;
679    if (initThread != null && Thread.currentThread() != initThread) {
680      // This usually won't happen but anyway, let's wait until the initialization thread exits.
681      // And notice that we may call terminate directly from the initThread so here we need to
682      // avoid join on ourselves.
683      initThread.interrupt();
684      Threads.shutdown(initThread, this.sleepForRetries);
685    }
686    Collection<ReplicationSourceShipper> workers = workerThreads.values();
687    for (ReplicationSourceShipper worker : workers) {
688      worker.stopWorker();
689      if(worker.entryReader != null) {
690        worker.entryReader.setReaderRunning(false);
691      }
692    }
693
694    for (ReplicationSourceShipper worker : workers) {
695      if (worker.isAlive() || worker.entryReader.isAlive()) {
696        try {
697          // Wait worker to stop
698          Thread.sleep(this.sleepForRetries);
699        } catch (InterruptedException e) {
700          LOG.info("{} Interrupted while waiting {} to stop", logPeerId(), worker.getName());
701          Thread.currentThread().interrupt();
702        }
703        // If worker still is alive after waiting, interrupt it
704        if (worker.isAlive()) {
705          worker.interrupt();
706        }
707        // If entry reader is alive after waiting, interrupt it
708        if (worker.entryReader.isAlive()) {
709          worker.entryReader.interrupt();
710        }
711      }
712    }
713
714    if (this.replicationEndpoint != null) {
715      this.replicationEndpoint.stop();
716    }
717    if (join) {
718      for (ReplicationSourceShipper worker : workers) {
719        Threads.shutdown(worker, this.sleepForRetries);
720        LOG.info("{} ReplicationSourceWorker {} terminated", logPeerId(), worker.getName());
721      }
722      if (this.replicationEndpoint != null) {
723        try {
724          this.replicationEndpoint.awaitTerminated(sleepForRetries * maxRetriesMultiplier,
725            TimeUnit.MILLISECONDS);
726        } catch (TimeoutException te) {
727          LOG.warn("{} Got exception while waiting for endpoint to shutdown "
728            + "for replication source : {}", logPeerId(), this.queueId, te);
729        }
730      }
731    }
732    if (clearMetrics) {
733      // Can be null in test context.
734      if (this.metrics != null) {
735        this.metrics.clear();
736      }
737    }
738  }

774  @Override
775  public String getStats() {
776    StringBuilder sb = new StringBuilder();
777    sb.append("Total replicated edits: ").append(totalReplicatedEdits)
778        .append(", current progress: \n");
779    for (Map.Entry<String, ReplicationSourceShipper> entry : workerThreads.entrySet()) {
780      String walGroupId = entry.getKey();
781      ReplicationSourceShipper worker = entry.getValue();
782      long position = worker.getCurrentPosition();
783      Path currentPath = worker.getCurrentPath();
784      sb.append("walGroup [").append(walGroupId).append("]: ");
785      if (currentPath != null) {
786        sb.append("currently replicating from: ").append(currentPath).append(" at position: ")
787            .append(position).append("\n");
788      } else {
789        sb.append("no replication ongoing, waiting for new log");
790      }
791    }
792    return sb.toString();
793  }
