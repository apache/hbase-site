066/**
067 * Class that handles the source of a replication stream.
068 * Currently does not handle more than 1 slave
069 * For each slave cluster it selects a random number of peers
070 * using a replication ratio. For example, if replication ration = 0.1
071 * and slave cluster has 100 region servers, 10 will be selected.
072 * <p>
073 * A stream is considered down when we cannot contact a region server on the
074 * peer cluster for more than 55 seconds by default.
075 * </p>
076 */
077@InterfaceAudience.Private
078public class ReplicationSource implements ReplicationSourceInterface {

140   * @param conf configuration to use
141   * @param fs file system to use
142   * @param manager replication manager to ping to

148  @Override
149  public void init(Configuration conf, FileSystem fs, ReplicationSourceManager manager,
150      ReplicationQueueStorage queueStorage, ReplicationPeer replicationPeer, Server server,
151      String queueId, UUID clusterId, WALFileLengthProvider walFileLengthProvider,
152      MetricsSource metrics) throws IOException {
153    this.server = server;
154    this.conf = HBaseConfiguration.create(conf);
155    this.waitOnEndpointSeconds =
156      this.conf.getInt(WAIT_ON_ENDPOINT_SECONDS, DEFAULT_WAIT_ON_ENDPOINT_SECONDS);
157    decorateConf();
158    this.sleepForRetries =
159        this.conf.getLong("replication.source.sleepforretries", 1000);    // 1 second
160    this.maxRetriesMultiplier =
161        this.conf.getInt("replication.source.maxretriesmultiplier", 300); // 5 minutes @ 1 sec per
162    this.queueSizePerGroup = this.conf.getInt("hbase.regionserver.maxlogs", 32);
163    this.queueStorage = queueStorage;
164    this.replicationPeer = replicationPeer;
165    this.manager = manager;
166    this.fs = fs;
167    this.metrics = metrics;
168    this.clusterId = clusterId;
169
170    this.queueId = queueId;
171    this.replicationQueueInfo = new ReplicationQueueInfo(queueId);
172    // ReplicationQueueInfo parses the peerId out of the znode for us
173    this.peerId = this.replicationQueueInfo.getPeerId();
174    this.logQueueWarnThreshold = this.conf.getInt("replication.source.log.queue.warn", 2);
175
176    defaultBandwidth = this.conf.getLong("replication.source.per.peer.node.bandwidth", 0);
177    currentBandwidth = getCurrentBandwidth();
178    this.throttler = new ReplicationThrottler((double) currentBandwidth / 10.0);
179    this.totalBufferUsed = manager.getTotalBufferUsed();
180    this.walFileLengthProvider = walFileLengthProvider;
181    LOG.info("queueId=" + queueId + ", ReplicationSource : " + peerId
182        + ", currentBandwidth=" + this.currentBandwidth);
183  }

185  private void decorateConf() {
186    String replicationCodec = this.conf.get(HConstants.REPLICATION_CODEC_CONF_KEY);
187    if (StringUtils.isNotEmpty(replicationCodec)) {
188      this.conf.set(HConstants.RPC_CODEC_CONF_KEY, replicationCodec);
189    }

192  @Override
193  public void enqueueLog(Path log) {
194    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(log.getName());
195    PriorityBlockingQueue<Path> queue = queues.get(logPrefix);
196    if (queue == null) {
197      queue = new PriorityBlockingQueue<>(queueSizePerGroup, new LogsComparator());
198      queues.put(logPrefix, queue);
199      if (this.isSourceActive() && this.replicationEndpoint != null) {
200        // new wal group observed after source startup, start a new worker thread to track it
201        // notice: it's possible that log enqueued when this.running is set but worker thread
202        // still not launched, so it's necessary to check workerThreads before start the worker
203        tryStartNewShipper(logPrefix, queue);
204      }
205    }
206    queue.put(log);
207    this.metrics.incrSizeOfLogQueue();
208    // This will log a warning for each new log that gets created above the warn threshold
209    int queueSize = queue.size();
210    if (queueSize > this.logQueueWarnThreshold) {
211      LOG.warn("WAL group " + logPrefix + " queue size: " + queueSize
212          + " exceeds value of replication.source.log.queue.warn: " + logQueueWarnThreshold);
213    }
214  }

216  @Override
217  public void addHFileRefs(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
218      throws ReplicationException {
219    Map<TableName, List<String>> tableCFMap = replicationPeer.getTableCFs();
220    if (tableCFMap != null) {
221      List<String> tableCfs = tableCFMap.get(tableName);
222      if (tableCFMap.containsKey(tableName)
223          && (tableCfs == null || tableCfs.contains(Bytes.toString(family)))) {
224        this.queueStorage.addHFileRefs(peerId, pairs);
225        metrics.incrSizeOfHFileRefsQueue(pairs.size());
226      } else {
227        LOG.debug("HFiles will not be replicated belonging to the table " + tableName + " family "
228            + Bytes.toString(family) + " to peer id " + peerId);
229      }
230    } else {
231      // user has explicitly not defined any table cfs for replication, means replicate all the
232      // data
233      this.queueStorage.addHFileRefs(peerId, pairs);
234      metrics.incrSizeOfHFileRefsQueue(pairs.size());
235    }
236  }

238  private ReplicationEndpoint createReplicationEndpoint()
239      throws InstantiationException, IllegalAccessException, ClassNotFoundException, IOException {
240    RegionServerCoprocessorHost rsServerHost = null;
241    if (server instanceof HRegionServer) {
242      rsServerHost = ((HRegionServer) server).getRegionServerCoprocessorHost();
243    }
244    String replicationEndpointImpl = replicationPeer.getPeerConfig().getReplicationEndpointImpl();
245    if (replicationEndpointImpl == null) {
246      // Default to HBase inter-cluster replication endpoint
247      replicationEndpointImpl = HBaseInterClusterReplicationEndpoint.class.getName();
248    }
249    ReplicationEndpoint replicationEndpoint =
250      Class.forName(replicationEndpointImpl).asSubclass(ReplicationEndpoint.class).newInstance();
251    if (rsServerHost != null) {
252      ReplicationEndpoint newReplicationEndPoint =
253        rsServerHost.postCreateReplicationEndPoint(replicationEndpoint);
254      if (newReplicationEndPoint != null) {
255        // Override the newly created endpoint from the hook with configured end point
256        replicationEndpoint = newReplicationEndPoint;
257      }
258    }
259    return replicationEndpoint;
260  }
261
262  private void initAndStartReplicationEndpoint(ReplicationEndpoint replicationEndpoint)
263      throws IOException, TimeoutException {
264    TableDescriptors tableDescriptors = null;
265    if (server instanceof HRegionServer) {
266      tableDescriptors = ((HRegionServer) server).getTableDescriptors();
267    }
268    replicationEndpoint
269        .init(new ReplicationEndpoint.Context(conf, replicationPeer.getConfiguration(), fs, peerId,
270            clusterId, replicationPeer, metrics, tableDescriptors, server));
271    replicationEndpoint.start();
272    replicationEndpoint.awaitRunning(waitOnEndpointSeconds, TimeUnit.SECONDS);
273  }

275  private void initializeWALEntryFilter() {
276    // get the WALEntryFilter from ReplicationEndpoint and add it to default filters
277    ArrayList<WALEntryFilter> filters =
278      Lists.<WALEntryFilter> newArrayList(new SystemTableWALEntryFilter());
279    WALEntryFilter filterFromEndpoint = this.replicationEndpoint.getWALEntryfilter();
280    if (filterFromEndpoint != null) {
281      filters.add(filterFromEndpoint);
282    }
283    filters.add(new ClusterMarkingEntryFilter(clusterId, peerClusterId, replicationEndpoint));
284    this.walEntryFilter = new ChainWALEntryFilter(filters);
285  }

287  protected void tryStartNewShipper(String walGroupId, PriorityBlockingQueue<Path> queue) {
288    ReplicationSourceShipper worker = new ReplicationSourceShipper(conf, walGroupId, queue, this);
289    ReplicationSourceShipper extant = workerThreads.putIfAbsent(walGroupId, worker);
290    if (extant != null) {
291      LOG.debug("Someone has beat us to start a worker thread for wal group " + walGroupId);
292    } else {
293      LOG.debug("Starting up worker for wal group " + walGroupId);
294      worker.startup(this::uncaughtException);
295      worker.setWALReader(
296        startNewWALReader(worker.getName(), walGroupId, queue, worker.getStartPosition()));
297    }
298  }
299
300  protected ReplicationSourceWALReader startNewWALReader(String threadName, String walGroupId,
301      PriorityBlockingQueue<Path> queue, long startPosition) {
302    ReplicationSourceWALReader walReader =
303      new ReplicationSourceWALReader(fs, conf, queue, startPosition, walEntryFilter, this);
304    return (ReplicationSourceWALReader) Threads.setDaemonThreadRunning(walReader,
305      threadName + ".replicationSource.wal-reader." + walGroupId + "," + queueId,
306      this::uncaughtException);
307  }

309  protected final void uncaughtException(Thread t, Throwable e) {
310    RSRpcServices.exitIfOOME(e);
311    LOG.error("Unexpected exception in " + t.getName() + " currentPath=" + getCurrentPath(), e);
312    server.abort("Unexpected exception in " + t.getName(), e);
313  }

325  @Override
326  public void tryThrottle(int batchSize) throws InterruptedException {
327    checkBandwidthChangeAndResetThrottler();
328    if (throttler.isEnabled()) {
329      long sleepTicks = throttler.getNextSleepInterval(batchSize);
330      if (sleepTicks > 0) {
331        if (LOG.isTraceEnabled()) {
332          LOG.trace("To sleep " + sleepTicks + "ms for throttling control");
333        }
334        Thread.sleep(sleepTicks);
335        // reset throttler's cycle start tick when sleep for throttling occurs
336        throttler.resetStartTick();
337      }
338    }
339  }
340
341  private void checkBandwidthChangeAndResetThrottler() {
342    long peerBandwidth = getCurrentBandwidth();
343    if (peerBandwidth != currentBandwidth) {
344      currentBandwidth = peerBandwidth;
345      throttler.setBandwidth((double) currentBandwidth / 10.0);
346      LOG.info("ReplicationSource : " + peerId
347          + " bandwidth throttling changed, currentBandWidth=" + currentBandwidth);
348    }
349  }
350
351  private long getCurrentBandwidth() {
352    long peerBandwidth = replicationPeer.getPeerBandwidth();
353    // user can set peer bandwidth to 0 to use default bandwidth
354    return peerBandwidth != 0 ? peerBandwidth : defaultBandwidth;
355  }

357  /**
358   * Do the sleeping logic
359   * @param msg Why we sleep
360   * @param sleepMultiplier by how many times the default sleeping time is augmented
361   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
362   */
363  protected boolean sleepForRetries(String msg, int sleepMultiplier) {
364    try {
365      if (LOG.isTraceEnabled()) {
366        LOG.trace(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
367      }
368      Thread.sleep(this.sleepForRetries * sleepMultiplier);
369    } catch (InterruptedException e) {
370      LOG.debug("Interrupted while sleeping between retries");
371      Thread.currentThread().interrupt();
372    }
373    return sleepMultiplier < maxRetriesMultiplier;
374  }

385  private void initialize() {
386    int sleepMultiplier = 1;
387    while (this.isSourceActive()) {
388      ReplicationEndpoint replicationEndpoint;
389      try {
390        replicationEndpoint = createReplicationEndpoint();
391      } catch (Exception e) {
392        LOG.warn("error creating ReplicationEndpoint, retry", e);
393        if (sleepForRetries("Error creating ReplicationEndpoint", sleepMultiplier)) {
394          sleepMultiplier++;
395        }
396        continue;
397      }
398
399      try {
400        initAndStartReplicationEndpoint(replicationEndpoint);
401        this.replicationEndpoint = replicationEndpoint;
402        break;
403      } catch (Exception e) {
404        LOG.warn("Error starting ReplicationEndpoint, retry", e);
405        replicationEndpoint.stop();
406        if (sleepForRetries("Error starting ReplicationEndpoint", sleepMultiplier)) {
407          sleepMultiplier++;
408        }
409      }
410    }
411
412    if (!this.isSourceActive()) {
413      return;
414    }
415
416    sleepMultiplier = 1;
417    // delay this until we are in an asynchronous thread
418    while (this.isSourceActive() && this.peerClusterId == null) {
419      this.peerClusterId = replicationEndpoint.getPeerUUID();
420      if (this.isSourceActive() && this.peerClusterId == null) {
421        if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
422          sleepMultiplier++;
423        }
424      }
425    }
426
427    // In rare case, zookeeper setting may be messed up. That leads to the incorrect
428    // peerClusterId value, which is the same as the source clusterId
429    if (clusterId.equals(peerClusterId) && !replicationEndpoint.canReplicateToSameCluster()) {
430      this.terminate("ClusterId " + clusterId + " is replicating to itself: peerClusterId "
431          + peerClusterId + " which is not allowed by ReplicationEndpoint:"
432          + replicationEndpoint.getClass().getName(), null, false);
433      this.manager.removeSource(this);
434      return;
435    }
436    LOG.info("Replicating " + clusterId + " -> " + peerClusterId);
437
438    initializeWALEntryFilter();
439    // start workers
440    for (Map.Entry<String, PriorityBlockingQueue<Path>> entry : queues.entrySet()) {
441      String walGroupId = entry.getKey();
442      PriorityBlockingQueue<Path> queue = entry.getValue();
443      tryStartNewShipper(walGroupId, queue);
444    }
445  }

467  public void terminate(String reason, Exception cause, boolean join) {
468    if (cause == null) {
469      LOG.info("Closing source " + this.queueId + " because: " + reason);
470    } else {
471      LOG.error("Closing source " + this.queueId + " because an error occurred: " + reason,
472        cause);
473    }
474    this.sourceRunning = false;
475    if (initThread != null && Thread.currentThread() != initThread) {
476      // This usually won't happen but anyway, let's wait until the initialization thread exits.
477      // And notice that we may call terminate directly from the initThread so here we need to
478      // avoid join on ourselves.
479      initThread.interrupt();
480      Threads.shutdown(initThread, this.sleepForRetries);
481    }
482    Collection<ReplicationSourceShipper> workers = workerThreads.values();
483    for (ReplicationSourceShipper worker : workers) {
484      worker.stopWorker();
485      worker.entryReader.interrupt();
486      worker.interrupt();
487    }
488    if (this.replicationEndpoint != null) {
489      this.replicationEndpoint.stop();
490    }
491    if (join) {
492      for (ReplicationSourceShipper worker : workers) {
493        Threads.shutdown(worker, this.sleepForRetries);
494        LOG.info("ReplicationSourceWorker " + worker.getName() + " terminated");
495      }
496      if (this.replicationEndpoint != null) {
497        try {
498          this.replicationEndpoint.awaitTerminated(sleepForRetries * maxRetriesMultiplier,
499            TimeUnit.MILLISECONDS);
500        } catch (TimeoutException te) {
501          LOG.warn("Got exception while waiting for endpoint to shutdown for replication source :" +
502            this.queueId, te);
503        }
504      }
505    }
506  }

534  /**
535   * Comparator used to compare logs together based on their start time
536   */
537  public static class LogsComparator implements Comparator<Path> {
538
539    @Override
540    public int compare(Path o1, Path o2) {
541      return Long.compare(getTS(o1), getTS(o2));
542    }
543
544    /**
545     * Split a path to get the start time
546     * For example: 10.20.20.171%3A60020.1277499063250
547     * @param p path to split
548     * @return start time
549     */
550    private static long getTS(Path p) {
551      int tsIndex = p.getName().lastIndexOf('.') + 1;
552      return Long.parseLong(p.getName().substring(tsIndex));
553    }
554  }

556  @Override
557  public String getStats() {
558    StringBuilder sb = new StringBuilder();
559    sb.append("Total replicated edits: ").append(totalReplicatedEdits)
560        .append(", current progress: \n");
561    for (Map.Entry<String, ReplicationSourceShipper> entry : workerThreads.entrySet()) {
562      String walGroupId = entry.getKey();
563      ReplicationSourceShipper worker = entry.getValue();
564      long position = worker.getCurrentPosition();
565      Path currentPath = worker.getCurrentPath();
566      sb.append("walGroup [").append(walGroupId).append("]: ");
567      if (currentPath != null) {
568        sb.append("currently replicating from: ").append(currentPath).append(" at position: ")
569            .append(position).append("\n");
570      } else {
571        sb.append("no replication ongoing, waiting for new log");
572      }
573    }
574    return sb.toString();
575  }
