053/**
054 * Reads and filters WAL entries, groups the filtered entries into batches, and puts the batches onto a queue
055 *
056 */
057@InterfaceAudience.Private
058@InterfaceStability.Evolving
059public class ReplicationSourceWALReader extends Thread {
060  private static final Log LOG = LogFactory.getLog(ReplicationSourceWALReader.class);
061
062  private final PriorityBlockingQueue<Path> logQueue;
063  private final FileSystem fs;
064  private final Configuration conf;
065  private final WALEntryFilter filter;
066  private final ReplicationSource source;
067
068  protected final BlockingQueue<WALEntryBatch> entryBatchQueue;
069  // max (heap) size of each batch - multiply by number of batches in queue to get total
070  private final long replicationBatchSizeCapacity;
071  // max count of each batch - multiply by number of batches in queue to get total
072  protected final int replicationBatchCountCapacity;
073  // position in the WAL to start reading at
074  private long currentPosition;
075  private final long sleepForRetries;
076  private final int maxRetriesMultiplier;
077  private final boolean eofAutoRecovery;
078
079  //Indicates whether this particular worker is running
080  private boolean isReaderRunning = true;
081
082  private AtomicLong totalBufferUsed;
083  private long totalBufferQuota;
084
085  /**
086   * Creates a reader worker for a given WAL queue. Reads WAL entries off a given queue, batches the
087   * entries, and puts them on a batch queue.
088   * @param fs the files system to use
089   * @param conf configuration to use
090   * @param logQueue The WAL queue to read off of
091   * @param startPosition position in the first WAL to start reading from
092   * @param filter The filter to use while reading
093   * @param source replication source
094   */
095  public ReplicationSourceWALReader(FileSystem fs, Configuration conf,
096      PriorityBlockingQueue<Path> logQueue, long startPosition, WALEntryFilter filter,
097      ReplicationSource source) {
098    this.logQueue = logQueue;
099    this.currentPosition = startPosition;
100    this.fs = fs;
101    this.conf = conf;
102    this.filter = filter;
103    this.source = source;
104    this.replicationBatchSizeCapacity =
105        this.conf.getLong("replication.source.size.capacity", 1024 * 1024 * 64);
106    this.replicationBatchCountCapacity = this.conf.getInt("replication.source.nb.capacity", 25000);
107    // memory used will be batchSizeCapacity * (nb.batches + 1)
108    // the +1 is for the current thread reading before placing onto the queue
109    int batchCount = conf.getInt("replication.source.nb.batches", 1);
110    this.totalBufferUsed = source.getSourceManager().getTotalBufferUsed();
111    this.totalBufferQuota = conf.getLong(HConstants.REPLICATION_SOURCE_TOTAL_BUFFER_KEY,
112      HConstants.REPLICATION_SOURCE_TOTAL_BUFFER_DFAULT);
113    this.sleepForRetries =
114        this.conf.getLong("replication.source.sleepforretries", 1000);    // 1 second
115    this.maxRetriesMultiplier =
116        this.conf.getInt("replication.source.maxretriesmultiplier", 300); // 5 minutes @ 1 sec per
117    this.eofAutoRecovery = conf.getBoolean("replication.source.eof.autorecovery", false);
118    this.entryBatchQueue = new LinkedBlockingQueue<>(batchCount);
119    LOG.info("peerClusterZnode=" + source.getPeerClusterZnode()
120        + ", ReplicationSourceWALReaderThread : " + source.getPeerId()
121        + " inited, replicationBatchSizeCapacity=" + replicationBatchSizeCapacity
122        + ", replicationBatchCountCapacity=" + replicationBatchCountCapacity
123        + ", replicationBatchQueueCapacity=" + batchCount);
124  }
125
126  @Override
127  public void run() {
128    int sleepMultiplier = 1;
129    while (isReaderRunning()) { // we only loop back here if something fatal happened to our stream
130      try (WALEntryStream entryStream =
131          new WALEntryStream(logQueue, fs, conf, currentPosition, source.getSourceMetrics())) {
132        while (isReaderRunning()) { // loop here to keep reusing stream while we can
133          if (!checkQuota()) {
134            continue;
135          }
136          WALEntryBatch batch = readWALEntries(entryStream);
137          if (batch != null && (!batch.getLastSeqIds().isEmpty() || batch.getNbEntries() > 0)) {
138            if (LOG.isTraceEnabled()) {
139              LOG.trace(String.format("Read %s WAL entries eligible for replication",
140                batch.getNbEntries()));
141            }
142            entryBatchQueue.put(batch);
143            sleepMultiplier = 1;
144          } else { // got no entries and didn't advance position in WAL
145            handleEmptyWALEntryBatch(batch, entryStream.getCurrentPath());
146          }
147          currentPosition = entryStream.getPosition();
148          entryStream.reset(); // reuse stream
149        }
150      } catch (IOException | WALEntryStreamRuntimeException e) { // stream related
151        if (sleepMultiplier < maxRetriesMultiplier) {
152          LOG.debug("Failed to read stream of replication entries: " + e);
153          sleepMultiplier++;
154        } else {
155          LOG.error("Failed to read stream of replication entries", e);
156          handleEofException(e);
157        }
158        Threads.sleep(sleepForRetries * sleepMultiplier);
159      } catch (InterruptedException e) {
160        LOG.trace("Interrupted while sleeping between WAL reads");
161        Thread.currentThread().interrupt();
162      }
163    }
164  }
165
166  private WALEntryBatch readWALEntries(WALEntryStream entryStream) throws IOException {
167    WALEntryBatch batch = null;
168    while (entryStream.hasNext()) {
169      if (batch == null) {
170        batch = new WALEntryBatch(replicationBatchCountCapacity, entryStream.getCurrentPath());
171      }
172      Entry entry = entryStream.next();
173      if (updateSerialReplPos(batch, entry)) {
174        batch.lastWalPosition = entryStream.getPosition();
175        break;
176      }
177      entry = filterEntry(entry);
178      if (entry != null) {
179        WALEdit edit = entry.getEdit();
180        if (edit != null && !edit.isEmpty()) {
181          long entrySize = getEntrySize(entry);
182          batch.addEntry(entry);
183          updateBatchStats(batch, entry, entryStream.getPosition(), entrySize);
184          boolean totalBufferTooLarge = acquireBufferQuota(entrySize);
185          // Stop if too many entries or too big
186          if (totalBufferTooLarge || batch.getHeapSize() >= replicationBatchSizeCapacity
187              || batch.getNbEntries() >= replicationBatchCountCapacity) {
188            break;
189          }
190        }
191      }
192    }
193    return batch;
194  }
195
196  protected void handleEmptyWALEntryBatch(WALEntryBatch batch, Path currentPath)
197      throws InterruptedException {
198    LOG.trace("Didn't read any new entries from WAL");
199    Thread.sleep(sleepForRetries);
200  }
201
202  // if we get an EOF due to a zero-length log, and there are other logs in queue
203  // (highly likely we've closed the current log), we've hit the max retries, and autorecovery is
204  // enabled, then dump the log
205  private void handleEofException(Exception e) {
206    if (e.getCause() instanceof EOFException && logQueue.size() > 1 && this.eofAutoRecovery) {
207      try {
208        if (fs.getFileStatus(logQueue.peek()).getLen() == 0) {
209          LOG.warn("Forcing removal of 0 length log in queue: " + logQueue.peek());
210          logQueue.remove();
211          currentPosition = 0;
212        }
213      } catch (IOException ioe) {
214        LOG.warn("Couldn't get file length information about log " + logQueue.peek());
215      }
216    }
217  }
218
219  public Path getCurrentPath() {
220    // if we've read some WAL entries, get the Path we read from
221    WALEntryBatch batchQueueHead = entryBatchQueue.peek();
222    if (batchQueueHead != null) {
223      return batchQueueHead.lastWalPath;
224    }
225    // otherwise, we must be currently reading from the head of the log queue
226    return logQueue.peek();
227  }
228
229  //returns false if we've already exceeded the global quota
230  private boolean checkQuota() {
231    // try not to go over total quota
232    if (totalBufferUsed.get() > totalBufferQuota) {
233      Threads.sleep(sleepForRetries);
234      return false;
235    }
236    return true;
237  }
238
239  private Entry filterEntry(Entry entry) {
240    Entry filtered = filter.filter(entry);
241    if (entry != null && filtered == null) {
242      source.getSourceMetrics().incrLogEditsFiltered();
243    }
244    return filtered;
245  }

247  /**
248   * @return true if we should stop reading because we're at REGION_CLOSE
249   */
250  private boolean updateSerialReplPos(WALEntryBatch batch, Entry entry) throws IOException {
251    if (entry.hasSerialReplicationScope()) {
252      String key = Bytes.toString(entry.getKey().getEncodedRegionName());
253      batch.setLastPosition(key, entry.getKey().getSequenceId());
254      if (!entry.getEdit().getCells().isEmpty()) {
255        WALProtos.RegionEventDescriptor maybeEvent =
256            WALEdit.getRegionEventDescriptor(entry.getEdit().getCells().get(0));
257        if (maybeEvent != null && maybeEvent
258            .getEventType() == WALProtos.RegionEventDescriptor.EventType.REGION_CLOSE) {
259          // In serially replication, if we move a region to another RS and move it back, we may
260          // read logs crossing two sections. We should break at REGION_CLOSE and push the first
261          // section first in case of missing the middle section belonging to the other RS.
262          // In a worker thread, if we can push the first log of a region, we can push all logs
263          // in the same region without waiting until we read a close marker because next time
264          // we read logs in this region, it must be a new section and not adjacent with this
265          // region. Mark it negative.
266          batch.setLastPosition(key, -entry.getKey().getSequenceId());
267          return true;
268        }
269      }
270    }
271    return false;
272  }

274  /**
275   * Retrieves the next batch of WAL entries from the queue, waiting up to the specified time for a
276   * batch to become available
277   * @return A batch of entries, along with the position in the log after reading the batch
278   * @throws InterruptedException if interrupted while waiting
279   */
280  public WALEntryBatch take() throws InterruptedException {
281    return entryBatchQueue.take();
282  }

284  private long getEntrySize(Entry entry) {
285    WALEdit edit = entry.getEdit();
286    return edit.heapSize() + calculateTotalSizeOfStoreFiles(edit);
287  }
288
289  private void updateBatchStats(WALEntryBatch batch, Entry entry, long entryPosition, long entrySize) {
290    WALEdit edit = entry.getEdit();
291    if (edit != null && !edit.isEmpty()) {
292      batch.incrementHeapSize(entrySize);
293      Pair<Integer, Integer> nbRowsAndHFiles = countDistinctRowKeysAndHFiles(edit);
294      batch.incrementNbRowKeys(nbRowsAndHFiles.getFirst());
295      batch.incrementNbHFiles(nbRowsAndHFiles.getSecond());
296    }
297    batch.lastWalPosition = entryPosition;
298  }

300  /**
301   * Count the number of different row keys in the given edit because of mini-batching. We assume
302   * that there's at least one Cell in the WALEdit.
303   * @param edit edit to count row keys from
304   * @return number of different row keys and HFiles
305   */
306  private Pair<Integer, Integer> countDistinctRowKeysAndHFiles(WALEdit edit) {
307    List<Cell> cells = edit.getCells();
308    int distinctRowKeys = 1;
309    int totalHFileEntries = 0;
310    Cell lastCell = cells.get(0);
311
312    int totalCells = edit.size();
313    for (int i = 0; i < totalCells; i++) {
314      // Count HFiles to be replicated
315      if (CellUtil.matchingQualifier(cells.get(i), WALEdit.BULK_LOAD)) {
316        try {
317          BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cells.get(i));
318          List<StoreDescriptor> stores = bld.getStoresList();
319          int totalStores = stores.size();
320          for (int j = 0; j < totalStores; j++) {
321            totalHFileEntries += stores.get(j).getStoreFileList().size();
322          }
323        } catch (IOException e) {
324          LOG.error("Failed to deserialize bulk load entry from wal edit. "
325              + "Then its hfiles count will not be added into metric.");
326        }
327      }
328
329      if (!CellUtil.matchingRows(cells.get(i), lastCell)) {
330        distinctRowKeys++;
331      }
332      lastCell = cells.get(i);
333    }
334
335    Pair<Integer, Integer> result = new Pair<>(distinctRowKeys, totalHFileEntries);
336    return result;
337  }

339  /**
340   * Calculate the total size of all the store files
341   * @param edit edit to count row keys from
342   * @return the total size of the store files
343   */
344  private int calculateTotalSizeOfStoreFiles(WALEdit edit) {
345    List<Cell> cells = edit.getCells();
346    int totalStoreFilesSize = 0;
347
348    int totalCells = edit.size();
349    for (int i = 0; i < totalCells; i++) {
350      if (CellUtil.matchingQualifier(cells.get(i), WALEdit.BULK_LOAD)) {
351        try {
352          BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cells.get(i));
353          List<StoreDescriptor> stores = bld.getStoresList();
354          int totalStores = stores.size();
355          for (int j = 0; j < totalStores; j++) {
356            totalStoreFilesSize += stores.get(j).getStoreFileSizeBytes();
357          }
358        } catch (IOException e) {
359          LOG.error("Failed to deserialize bulk load entry from wal edit. "
360              + "Size of HFiles part of cell will not be considered in replication "
361              + "request size calculation.",
362            e);
363        }
364      }
365    }
366    return totalStoreFilesSize;
367  }

369  /**
370   * @param size delta size for grown buffer
371   * @return true if we should clear buffer and push all
372   */
373  private boolean acquireBufferQuota(long size) {
374    return totalBufferUsed.addAndGet(size) >= totalBufferQuota;
375  }

391  /**
392   * Holds a batch of WAL entries to replicate, along with some statistics
393   *
394   */
395  static class WALEntryBatch {
396    private List<Entry> walEntries;
397    // last WAL that was read
398    private Path lastWalPath;
399    // position in WAL of last entry in this batch
400    private long lastWalPosition = 0;
401    // number of distinct row keys in this batch
402    private int nbRowKeys = 0;
403    // number of HFiles
404    private int nbHFiles = 0;
405    // heap size of data we need to replicate
406    private long heapSize = 0;
407    // save the last sequenceid for each region if the table has serial-replication scope
408    private Map<String, Long> lastSeqIds = new HashMap<>();
409
410    /**
411     * @param walEntries
412     * @param lastWalPath Path of the WAL the last entry in this batch was read from
413     * @param lastWalPosition Position in the WAL the last entry in this batch was read from
414     */
415    WALEntryBatch(int maxNbEntries, Path lastWalPath) {
416      this.walEntries = new ArrayList<>(maxNbEntries);
417      this.lastWalPath = lastWalPath;
418    }
419
420    public void addEntry(Entry entry) {
421      walEntries.add(entry);
422    }
423
424    /**
425     * @return the WAL Entries.
426     */
427    public List<Entry> getWalEntries() {
428      return walEntries;
429    }
430
431    /**
432     * @return the path of the last WAL that was read.
433     */
434    public Path getLastWalPath() {
435      return lastWalPath;
436    }
437
438    /**
439     * @return the position in the last WAL that was read.
440     */
441    public long getLastWalPosition() {
442      return lastWalPosition;
443    }
444
445    public int getNbEntries() {
446      return walEntries.size();
447    }
448
449    /**
450     * @return the number of distinct row keys in this batch
451     */
452    public int getNbRowKeys() {
453      return nbRowKeys;
454    }
455
456    /**
457     * @return the number of HFiles in this batch
458     */
459    public int getNbHFiles() {
460      return nbHFiles;
461    }
462
463    /**
464     * @return total number of operations in this batch
465     */
466    public int getNbOperations() {
467      return getNbRowKeys() + getNbHFiles();
468    }
469
470    /**
471     * @return the heap size of this batch
472     */
473    public long getHeapSize() {
474      return heapSize;
475    }
476
477    /**
478     * @return the last sequenceid for each region if the table has serial-replication scope
479     */
480    public Map<String, Long> getLastSeqIds() {
481      return lastSeqIds;
482    }
483
484    private void incrementNbRowKeys(int increment) {
485      nbRowKeys += increment;
486    }
487
488    private void incrementNbHFiles(int increment) {
489      nbHFiles += increment;
490    }
491
492    private void incrementHeapSize(long increment) {
493      heapSize += increment;
494    }
495
496    private void setLastPosition(String region, Long sequenceId) {
497      getLastSeqIds().put(region, sequenceId);
498    }
499  }
