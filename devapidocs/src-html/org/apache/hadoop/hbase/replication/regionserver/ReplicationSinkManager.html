018package org.apache.hadoop.hbase.replication.regionserver;
019
020import java.io.IOException;
021import java.util.Collections;
022import java.util.List;
023import java.util.Map;
024import java.util.concurrent.ThreadLocalRandom;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.ServerName;
027import org.apache.hadoop.hbase.client.AsyncClusterConnection;
028import org.apache.hadoop.hbase.client.AsyncRegionServerAdmin;
029import org.apache.hadoop.hbase.replication.HBaseReplicationEndpoint;
030import org.apache.yetus.audience.InterfaceAudience;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034import org.apache.hbase.thirdparty.com.google.common.annotations.VisibleForTesting;
035import org.apache.hbase.thirdparty.com.google.common.collect.Lists;
036import org.apache.hbase.thirdparty.com.google.common.collect.Maps;
037
038/**
039 * Maintains a collection of peers to replicate to, and randomly selects a
040 * single peer to replicate to per set of data to replicate. Also handles
041 * keeping track of peer availability.
042 */
043@InterfaceAudience.Private
044public class ReplicationSinkManager {
045
046  private static final Logger LOG = LoggerFactory.getLogger(ReplicationSinkManager.class);
047
048  /**
049   * Default maximum number of times a replication sink can be reported as bad before
050   * it will no longer be provided as a sink for replication without the pool of
051   * replication sinks being refreshed.
052   */
053  static final int DEFAULT_BAD_SINK_THRESHOLD = 3;
054
055  /**
056   * Default ratio of the total number of peer cluster region servers to consider
057   * replicating to.
058   */
059  static final float DEFAULT_REPLICATION_SOURCE_RATIO = 0.5f;
060
061
062  private final AsyncClusterConnection conn;
063
064  private final HBaseReplicationEndpoint endpoint;
065
066  // Count of "bad replication sink" reports per peer sink
067  private final Map<ServerName, Integer> badReportCounts;
068
069  // Ratio of total number of potential peer region servers to be used
070  private final float ratio;
071
072  // Maximum number of times a sink can be reported as bad before the pool of
073  // replication sinks is refreshed
074  private final int badSinkThreshold;
075
076  // A timestamp of the last time the list of replication peers changed
077  private long lastUpdateToPeers;
078
079  // The current pool of sinks to which replication can be performed
080  private List<ServerName> sinks = Lists.newArrayList();
081
082  /**
083   * Instantiate for a single replication peer cluster.
084   * @param conn connection to the peer cluster
085   * @param endpoint replication endpoint for inter cluster replication
086   * @param conf HBase configuration, used for determining replication source ratio and bad peer
087   *          threshold
088   */
089  public ReplicationSinkManager(AsyncClusterConnection conn, HBaseReplicationEndpoint endpoint,
090      Configuration conf) {
091    this.conn = conn;
092    this.endpoint = endpoint;
093    this.badReportCounts = Maps.newHashMap();
094    this.ratio = conf.getFloat("replication.source.ratio", DEFAULT_REPLICATION_SOURCE_RATIO);
095    this.badSinkThreshold =
096      conf.getInt("replication.bad.sink.threshold", DEFAULT_BAD_SINK_THRESHOLD);
097  }
098
099  /**
100   * Get a randomly-chosen replication sink to replicate to.
101   * @return a replication sink to replicate to
102   */
103  public synchronized SinkPeer getReplicationSink() throws IOException {
104    if (endpoint.getLastRegionServerUpdate() > this.lastUpdateToPeers || sinks.isEmpty()) {
105      LOG.info("Current list of sinks is out of date or empty, updating");
106      chooseSinks();
107    }
108
109    if (sinks.isEmpty()) {
110      throw new IOException("No replication sinks are available");
111    }
112    ServerName serverName = sinks.get(ThreadLocalRandom.current().nextInt(sinks.size()));
113    return new SinkPeer(serverName, conn.getRegionServerAdmin(serverName));
114  }
115
116  /**
117   * Report a {@code SinkPeer} as being bad (i.e. an attempt to replicate to it
118   * failed). If a single SinkPeer is reported as bad more than
119   * replication.bad.sink.threshold times, it will be removed
120   * from the pool of potential replication targets.
121   *
122   * @param sinkPeer
123   *          The SinkPeer that had a failed replication attempt on it
124   */
125  public synchronized void reportBadSink(SinkPeer sinkPeer) {
126    ServerName serverName = sinkPeer.getServerName();
127    int badReportCount = (badReportCounts.containsKey(serverName)
128                    ? badReportCounts.get(serverName) : 0) + 1;
129    badReportCounts.put(serverName, badReportCount);
130    if (badReportCount > badSinkThreshold) {
131      this.sinks.remove(serverName);
132      if (sinks.isEmpty()) {
133        chooseSinks();
134      }
135    }
136  }
137
138  /**
139   * Report that a {@code SinkPeer} successfully replicated a chunk of data.
140   *
141   * @param sinkPeer
142   *          The SinkPeer that had a failed replication attempt on it
143   */
144  public synchronized void reportSinkSuccess(SinkPeer sinkPeer) {
145    badReportCounts.remove(sinkPeer.getServerName());
146  }
147
148  /**
149   * Refresh the list of sinks.
150   */
151  public synchronized void chooseSinks() {
152    List<ServerName> slaveAddresses = endpoint.getRegionServers();
153    if(slaveAddresses.isEmpty()){
154      LOG.warn("No sinks available at peer. Will not be able to replicate");
155    }
156    Collections.shuffle(slaveAddresses, ThreadLocalRandom.current());
157    int numSinks = (int) Math.ceil(slaveAddresses.size() * ratio);
158    sinks = slaveAddresses.subList(0, numSinks);
159    lastUpdateToPeers = System.currentTimeMillis();
160    badReportCounts.clear();
161  }
162
163  public synchronized int getNumSinks() {
164    return sinks.size();
165  }
166
167  @VisibleForTesting
168  protected List<ServerName> getSinksForTesting() {
169    return Collections.unmodifiableList(sinks);
170  }
171
172  /**
173   * Wraps a replication region server sink to provide the ability to identify
174   * it.
175   */
176  public static class SinkPeer {
177    private ServerName serverName;
178    private AsyncRegionServerAdmin regionServer;
179
180    public SinkPeer(ServerName serverName, AsyncRegionServerAdmin regionServer) {
181      this.serverName = serverName;
182      this.regionServer = regionServer;
183    }
184
185    ServerName getServerName() {
186      return serverName;
187    }
188
189    public AsyncRegionServerAdmin getRegionServer() {
190      return regionServer;
191    }
192  }
193}
