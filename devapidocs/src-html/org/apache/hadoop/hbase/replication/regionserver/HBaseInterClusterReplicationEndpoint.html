064/**
065 * A {@link org.apache.hadoop.hbase.replication.ReplicationEndpoint} implementation for replicating
066 * to another HBase cluster. For the slave cluster it selects a random number of peers using a
067 * replication ratio. For example, if replication ration = 0.1 and slave cluster has 100 region
068 * servers, 10 will be selected.
069 * <p>
070 * A stream is considered down when we cannot contact a region server on the peer cluster for more
071 * than 55 seconds by default.
072 * </p>
073 */
074@InterfaceAudience.Private
075public class HBaseInterClusterReplicationEndpoint extends HBaseReplicationEndpoint {
076  private static final Logger LOG =
077    LoggerFactory.getLogger(HBaseInterClusterReplicationEndpoint.class);

109  @Override
110  public void init(Context context) throws IOException {
111    super.init(context);
112    decorateConf();
113    this.maxRetriesMultiplier = this.conf.getInt("replication.source.maxretriesmultiplier", 300);
114    this.socketTimeoutMultiplier =
115      this.conf.getInt("replication.source.socketTimeoutMultiplier", maxRetriesMultiplier);
116    this.sleepForRetries = this.conf.getLong("replication.source.sleepforretries", 1000);
117    this.metrics = context.getMetrics();
118    // per sink thread pool
119    this.maxThreads = this.conf.getInt(HConstants.REPLICATION_SOURCE_MAXTHREADS_KEY,
120      HConstants.REPLICATION_SOURCE_MAXTHREADS_DEFAULT);
121    // Set the size limit for replication RPCs to 95% of the max request size.
122    // We could do with less slop if we have an accurate estimate of encoded size. Being
123    // conservative for now.
124    this.replicationRpcLimit =
125      (int) (0.95 * conf.getLong(RpcServer.MAX_REQUEST_SIZE, RpcServer.DEFAULT_MAX_REQUEST_SIZE));
126    this.dropOnDeletedTables = this.conf.getBoolean(REPLICATION_DROP_ON_DELETED_TABLE_KEY, false);
127    this.dropOnDeletedColumnFamilies =
128      this.conf.getBoolean(REPLICATION_DROP_ON_DELETED_COLUMN_FAMILY_KEY, false);
129
130    this.replicationBulkLoadDataEnabled = conf.getBoolean(
131      HConstants.REPLICATION_BULKLOAD_ENABLE_KEY, HConstants.REPLICATION_BULKLOAD_ENABLE_DEFAULT);
132    if (this.replicationBulkLoadDataEnabled) {
133      replicationClusterId = this.conf.get(HConstants.REPLICATION_CLUSTER_ID);
134    }
135    // Construct base namespace directory and hfile archive directory path
136    Path rootDir = CommonFSUtils.getRootDir(conf);
137    Path baseNSDir = new Path(HConstants.BASE_NAMESPACE_DIR);
138    baseNamespaceDir = new Path(rootDir, baseNSDir);
139    hfileArchiveDir = new Path(rootDir, new Path(HConstants.HFILE_ARCHIVE_DIRECTORY, baseNSDir));
140    isSerial = context.getPeerConfig().isSerial();
141  }

150  private void connectToPeers() {
151    int sleepMultiplier = 1;
152    // Connect to peer cluster first, unless we have to stop
153    while (this.isRunning() && getNumSinks() == 0) {
154      chooseSinks();
155      if (this.isRunning() && getNumSinks() == 0) {
156        if (sleepForRetries("Waiting for peers", sleepMultiplier)) {
157          sleepMultiplier++;
158        }
159      }
160    }
161  }
162
163  /**
164   * Do the sleeping logic
165   * @param msg             Why we sleep
166   * @param sleepMultiplier by how many times the default sleeping time is augmented
167   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
168   */
169  private boolean sleepForRetries(String msg, int sleepMultiplier) {
170    try {
171      if (LOG.isTraceEnabled()) {
172        LOG.trace("{} {}, sleeping {} times {}", logPeerId(), msg, sleepForRetries,
173          sleepMultiplier);
174      }
175      Thread.sleep(this.sleepForRetries * sleepMultiplier);
176    } catch (InterruptedException e) {
177      Thread.currentThread().interrupt();
178      if (LOG.isDebugEnabled()) {
179        LOG.debug("{} {} Interrupted while sleeping between retries", msg, logPeerId());
180      }
181    }
182    return sleepMultiplier < maxRetriesMultiplier;
183  }

185  private int getEstimatedEntrySize(Entry e) {
186    long size = e.getKey().estimatedSerializedSizeOf() + e.getEdit().estimatedSerializedSizeOf();
187    return (int) size;
188  }

190  private List<List<Entry>> createParallelBatches(final List<Entry> entries) {
191    int numSinks = Math.max(getNumSinks(), 1);
192    int n = Math.min(Math.min(this.maxThreads, entries.size() / 100 + 1), numSinks);
193    List<List<Entry>> entryLists =
194      Stream.generate(ArrayList<Entry>::new).limit(n).collect(Collectors.toList());
195    int[] sizes = new int[n];
196    for (Entry e : entries) {
197      int index = Math.abs(Bytes.hashCode(e.getKey().getEncodedRegionName()) % n);
198      int entrySize = getEstimatedEntrySize(e);
199      // If this batch has at least one entry and is over sized, move it to the tail of list and
200      // initialize the entryLists[index] to be a empty list.
201      if (sizes[index] > 0 && sizes[index] + entrySize > replicationRpcLimit) {
202        entryLists.add(entryLists.get(index));
203        entryLists.set(index, new ArrayList<>());
204        sizes[index] = 0;
205      }
206      entryLists.get(index).add(e);
207      sizes[index] += entrySize;
208    }
209    return entryLists;
210  }
211
212  private List<List<Entry>> createSerialBatches(final List<Entry> entries) {
213    Map<byte[], List<Entry>> regionEntries = new TreeMap<>(Bytes.BYTES_COMPARATOR);
214    for (Entry e : entries) {
215      regionEntries.computeIfAbsent(e.getKey().getEncodedRegionName(), key -> new ArrayList<>())
216        .add(e);
217    }
218    return new ArrayList<>(regionEntries.values());
219  }

221  /**
222   * Divide the entries into multiple batches, so that we can replicate each batch in a thread pool
223   * concurrently. Note that, for serial replication, we need to make sure that entries from the
224   * same region to be replicated serially, so entries from the same region consist of a batch, and
225   * we will divide a batch into several batches by replicationRpcLimit in method
226   * serialReplicateRegionEntries()
227   */
228  private List<List<Entry>> createBatches(final List<Entry> entries) {
229    if (isSerial) {
230      return createSerialBatches(entries);
231    } else {
232      return createParallelBatches(entries);
233    }
234  }

236  /**
237   * Check if there's an {@link TableNotFoundException} in the caused by stacktrace.
238   */
239  public static boolean isTableNotFoundException(Throwable io) {
240    if (io instanceof RemoteException) {
241      io = ((RemoteException) io).unwrapRemoteException();
242    }
243    if (io != null && io.getMessage().contains("TableNotFoundException")) {
244      return true;
245    }
246    for (; io != null; io = io.getCause()) {
247      if (io instanceof TableNotFoundException) {
248        return true;
249      }
250    }
251    return false;
252  }

272  List<List<Entry>> filterNotExistTableEdits(final List<List<Entry>> oldEntryList) {
273    List<List<Entry>> entryList = new ArrayList<>();
274    Map<TableName, Boolean> existMap = new HashMap<>();
275    try (Connection localConn = ConnectionFactory.createConnection(ctx.getLocalConfiguration());
276      Admin localAdmin = localConn.getAdmin()) {
277      for (List<Entry> oldEntries : oldEntryList) {
278        List<Entry> entries = new ArrayList<>();
279        for (Entry e : oldEntries) {
280          TableName tableName = e.getKey().getTableName();
281          boolean exist = true;
282          if (existMap.containsKey(tableName)) {
283            exist = existMap.get(tableName);
284          } else {
285            try {
286              exist = localAdmin.tableExists(tableName);
287              existMap.put(tableName, exist);
288            } catch (IOException iox) {
289              LOG.warn("Exception checking for local table " + tableName, iox);
290              // we can't drop edits without full assurance, so we assume table exists.
291              exist = true;
292            }
293          }
294          if (exist) {
295            entries.add(e);
296          } else {
297            // Would potentially be better to retry in one of the outer loops
298            // and add a table filter there; but that would break the encapsulation,
299            // so we're doing the filtering here.
300            LOG.warn("Missing table detected at sink, local table also does not exist, "
301              + "filtering edits for table '{}'", tableName);
302          }
303        }
304        if (!entries.isEmpty()) {
305          entryList.add(entries);
306        }
307      }
308    } catch (IOException iox) {
309      LOG.warn("Exception when creating connection to check local table", iox);
310      return oldEntryList;
311    }
312    return entryList;
313  }

315  List<List<Entry>> filterNotExistColumnFamilyEdits(final List<List<Entry>> oldEntryList) {
316    List<List<Entry>> entryList = new ArrayList<>();
317    Map<TableName, Set<String>> existColumnFamilyMap = new HashMap<>();
318    try (Connection localConn = ConnectionFactory.createConnection(ctx.getLocalConfiguration());
319      Admin localAdmin = localConn.getAdmin()) {
320      for (List<Entry> oldEntries : oldEntryList) {
321        List<Entry> entries = new ArrayList<>();
322        for (Entry e : oldEntries) {
323          TableName tableName = e.getKey().getTableName();
324          if (!existColumnFamilyMap.containsKey(tableName)) {
325            try {
326              Set<String> cfs = localAdmin.getDescriptor(tableName).getColumnFamilyNames().stream()
327                .map(Bytes::toString).collect(Collectors.toSet());
328              existColumnFamilyMap.put(tableName, cfs);
329            } catch (Exception ex) {
330              LOG.warn("Exception getting cf names for local table {}", tableName, ex);
331              // if catch any exception, we are not sure about table's description,
332              // so replicate raw entry
333              entries.add(e);
334              continue;
335            }
336          }
337
338          Set<String> existColumnFamilies = existColumnFamilyMap.get(tableName);
339          Set<String> missingCFs = new HashSet<>();
340          WALEdit walEdit = new WALEdit();
341          walEdit.getCells().addAll(e.getEdit().getCells());
342          WALUtil.filterCells(walEdit, cell -> {
343            String cf = Bytes.toString(CellUtil.cloneFamily(cell));
344            if (existColumnFamilies.contains(cf)) {
345              return cell;
346            } else {
347              missingCFs.add(cf);
348              return null;
349            }
350          });
351          if (!walEdit.isEmpty()) {
352            Entry newEntry = new Entry(e.getKey(), walEdit);
353            entries.add(newEntry);
354          }
355
356          if (!missingCFs.isEmpty()) {
357            // Would potentially be better to retry in one of the outer loops
358            // and add a table filter there; but that would break the encapsulation,
359            // so we're doing the filtering here.
360            LOG.warn(
361              "Missing column family detected at sink, local column family also does not exist,"
362                + " filtering edits for table '{}',column family '{}'",
363              tableName, missingCFs);
364          }
365        }
366        if (!entries.isEmpty()) {
367          entryList.add(entries);
368        }
369      }
370    } catch (IOException iox) {
371      LOG.warn("Exception when creating connection to check local table", iox);
372      return oldEntryList;
373    }
374    return entryList;
375  }

377  private long parallelReplicate(ReplicateContext replicateContext, List<List<Entry>> batches)
378    throws IOException {
379    List<CompletableFuture<Integer>> futures =
380      new ArrayList<CompletableFuture<Integer>>(batches.size());
381    for (int i = 0; i < batches.size(); i++) {
382      List<Entry> entries = batches.get(i);
383      if (entries.isEmpty()) {
384        continue;
385      }
386      if (LOG.isTraceEnabled()) {
387        LOG.trace("{} Submitting {} entries of total size {}", logPeerId(), entries.size(),
388          replicateContext.getSize());
389      }
390      // RuntimeExceptions encountered here bubble up and are handled in ReplicationSource
391      futures.add(asyncReplicate(entries, i, replicateContext.getTimeout()));
392    }
393
394    IOException iox = null;
395    long lastWriteTime = 0;
396
397    for (CompletableFuture<Integer> f : futures) {
398      try {
399        // wait for all futures, remove successful parts
400        // (only the remaining parts will be retried)
401        int index = FutureUtils.get(f);
402        List<Entry> batch = batches.get(index);
403        batches.set(index, Collections.emptyList()); // remove successful batch
404        // Find the most recent write time in the batch
405        long writeTime = batch.get(batch.size() - 1).getKey().getWriteTime();
406        if (writeTime > lastWriteTime) {
407          lastWriteTime = writeTime;
408        }
409      } catch (IOException e) {
410        iox = e;
411      } catch (RuntimeException e) {
412        iox = new IOException(e);
413      }
414    }
415    if (iox != null) {
416      // if we had any exceptions, try again
417      throw iox;
418    }
419    return lastWriteTime;
420  }

422  /**
423   * Do the shipping logic
424   */
425  @Override
426  public boolean replicate(ReplicateContext replicateContext) {
427    int sleepMultiplier = 1;
428
429    if (!peersSelected && this.isRunning()) {
430      connectToPeers();
431      peersSelected = true;
432    }
433
434    int numSinks = getNumSinks();
435    if (numSinks == 0) {
436      if (
437        (EnvironmentEdgeManager.currentTime() - lastSinkFetchTime) >= (maxRetriesMultiplier * 1000)
438      ) {
439        LOG.warn("No replication sinks found, returning without replicating. "
440          + "The source should retry with the same set of edits. Not logging this again for "
441          + "the next {} seconds.", maxRetriesMultiplier);
442        lastSinkFetchTime = EnvironmentEdgeManager.currentTime();
443      }
444      sleepForRetries("No sinks available at peer", sleepMultiplier);
445      return false;
446    }
447
448    List<List<Entry>> batches = createBatches(replicateContext.getEntries());
449    while (this.isRunning() && !this.stopping) {
450      if (!isPeerEnabled()) {
451        if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
452          sleepMultiplier++;
453        }
454        continue;
455      }
456      try {
457        // replicate the batches to sink side.
458        parallelReplicate(replicateContext, batches);
459        return true;
460      } catch (IOException ioe) {
461        if (ioe instanceof RemoteException) {
462          if (dropOnDeletedTables && isTableNotFoundException(ioe)) {
463            // Only filter the edits to replicate and don't change the entries in replicateContext
464            // as the upper layer rely on it.
465            batches = filterNotExistTableEdits(batches);
466            if (batches.isEmpty()) {
467              LOG.warn("After filter not exist table's edits, 0 edits to replicate, just return");
468              return true;
469            }
470          } else if (dropOnDeletedColumnFamilies && isNoSuchColumnFamilyException(ioe)) {
471            batches = filterNotExistColumnFamilyEdits(batches);
472            if (batches.isEmpty()) {
473              LOG.warn("After filter not exist column family's edits, 0 edits to replicate, "
474                + "just return");
475              return true;
476            }
477          } else {
478            LOG.warn("{} Peer encountered RemoteException, rechecking all sinks: ", logPeerId(),
479              ioe);
480            chooseSinks();
481          }
482        } else {
483          if (ioe instanceof SocketTimeoutException) {
484            // This exception means we waited for more than 60s and nothing
485            // happened, the cluster is alive and calling it right away
486            // even for a test just makes things worse.
487            sleepForRetries(
488              "Encountered a SocketTimeoutException. Since the "
489                + "call to the remote cluster timed out, which is usually "
490                + "caused by a machine failure or a massive slowdown",
491              this.socketTimeoutMultiplier);
492          } else if (ioe instanceof ConnectException || ioe instanceof UnknownHostException) {
493            LOG.warn("{} Peer is unavailable, rechecking all sinks: ", logPeerId(), ioe);
494            chooseSinks();
495          } else {
496            LOG.warn("{} Can't replicate because of a local or network error: ", logPeerId(), ioe);
497          }
498        }
499        if (sleepForRetries("Since we are unable to replicate", sleepMultiplier)) {
500          sleepMultiplier++;
501        }
502      }
503    }
504    return false; // in case we exited before replicating
505  }

519  protected CompletableFuture<Integer> replicateEntries(List<Entry> entries, int batchIndex,
520    int timeout) {
521    int entriesHashCode = System.identityHashCode(entries);
522    if (LOG.isTraceEnabled()) {
523      long size = entries.stream().mapToLong(this::getEstimatedEntrySize).sum();
524      LOG.trace("{} Replicating batch {} of {} entries with total size {} bytes to {}", logPeerId(),
525        entriesHashCode, entries.size(), size, replicationClusterId);
526    }
527    SinkPeer sinkPeer = null;
528    final CompletableFuture<Integer> resultCompletableFuture = new CompletableFuture<Integer>();
529    try {
530      sinkPeer = getReplicationSink();
531    } catch (IOException e) {
532      this.onReplicateWALEntryException(entriesHashCode, e, sinkPeer);
533      resultCompletableFuture.completeExceptionally(e);
534      return resultCompletableFuture;
535    }
536    assert sinkPeer != null;
537    AsyncRegionServerAdmin rsAdmin = sinkPeer.getRegionServer();
538    final SinkPeer sinkPeerToUse = sinkPeer;
539    FutureUtils.addListener(
540      ReplicationProtobufUtil.replicateWALEntry(rsAdmin, entries.toArray(new Entry[entries.size()]),
541        replicationClusterId, baseNamespaceDir, hfileArchiveDir, timeout),
542      (response, exception) -> {
543        if (exception != null) {
544          onReplicateWALEntryException(entriesHashCode, exception, sinkPeerToUse);
545          resultCompletableFuture.completeExceptionally(exception);
546          return;
547        }
548        reportSinkSuccess(sinkPeerToUse);
549        resultCompletableFuture.complete(batchIndex);
550      });
551    return resultCompletableFuture;
552  }
553
554  private void onReplicateWALEntryException(int entriesHashCode, Throwable exception,
555    final SinkPeer sinkPeer) {
556    if (LOG.isTraceEnabled()) {
557      LOG.trace("{} Failed replicating batch {}", logPeerId(), entriesHashCode, exception);
558    }
559    if (exception instanceof IOException) {
560      if (sinkPeer != null) {
561        reportBadSink(sinkPeer);
562      }
563    }
564  }

566  /**
567   * Here for {@link HBaseInterClusterReplicationEndpoint#isSerialis} is true, we iterator over the
568   * WAL {@link Entry} list, once we reached a batch limit, we send it out, and in the callback, we
569   * send the next batch, until we send all entries out.
570   */
571  private CompletableFuture<Integer> serialReplicateRegionEntries(
572    PeekingIterator<Entry> walEntryPeekingIterator, int batchIndex, int timeout) {
573    if (!walEntryPeekingIterator.hasNext()) {
574      return CompletableFuture.completedFuture(batchIndex);
575    }
576    int batchSize = 0;
577    List<Entry> batch = new ArrayList<>();
578    while (walEntryPeekingIterator.hasNext()) {
579      Entry entry = walEntryPeekingIterator.peek();
580      int entrySize = getEstimatedEntrySize(entry);
581      if (batchSize > 0 && batchSize + entrySize > replicationRpcLimit) {
582        break;
583      }
584      walEntryPeekingIterator.next();
585      batch.add(entry);
586      batchSize += entrySize;
587    }
588
589    if (batchSize <= 0) {
590      return CompletableFuture.completedFuture(batchIndex);
591    }
592    final CompletableFuture<Integer> resultCompletableFuture = new CompletableFuture<Integer>();
593    FutureUtils.addListener(replicateEntries(batch, batchIndex, timeout), (response, exception) -> {
594      if (exception != null) {
595        resultCompletableFuture.completeExceptionally(exception);
596        return;
597      }
598      if (!walEntryPeekingIterator.hasNext()) {
599        resultCompletableFuture.complete(batchIndex);
600        return;
601      }
602      FutureUtils.addListener(
603        serialReplicateRegionEntries(walEntryPeekingIterator, batchIndex, timeout),
604        (currentResponse, currentException) -> {
605          if (currentException != null) {
606            resultCompletableFuture.completeExceptionally(currentException);
607            return;
608          }
609          resultCompletableFuture.complete(batchIndex);
610        });
611    });
612    return resultCompletableFuture;
613  }

615  /**
616   * Replicate entries to peer cluster by async API.
617   */
618  protected CompletableFuture<Integer> asyncReplicate(List<Entry> entries, int batchIndex,
619    int timeout) {
620    return isSerial
621      ? serialReplicateRegionEntries(Iterators.peekingIterator(entries.iterator()), batchIndex,
622        timeout)
623      : replicateEntries(entries, batchIndex, timeout);
