058/**
059 * A {@link org.apache.hadoop.hbase.replication.ReplicationEndpoint}
060 * implementation for replicating to another HBase cluster.
061 * For the slave cluster it selects a random number of peers
062 * using a replication ratio. For example, if replication ration = 0.1
063 * and slave cluster has 100 region servers, 10 will be selected.
064 * <p>
065 * A stream is considered down when we cannot contact a region server on the
066 * peer cluster for more than 55 seconds by default.
067 * </p>
068 */
069@InterfaceAudience.Private
070public class HBaseInterClusterReplicationEndpoint extends HBaseReplicationEndpoint {
071
072  private static final Log LOG = LogFactory.getLog(HBaseInterClusterReplicationEndpoint.class);
073  private HConnection conn;
074
075  private Configuration conf;
076
077  // How long should we sleep for each retry
078  private long sleepForRetries;
079
080  // Maximum number of retries before taking bold actions
081  private int maxRetriesMultiplier;
082  // Socket timeouts require even bolder actions since we don't want to DDOS
083  private int socketTimeoutMultiplier;
084  //Metrics for this source
085  private MetricsSource metrics;
086  // Handles connecting to peer region servers
087  private ReplicationSinkManager replicationSinkMgr;
088  private boolean peersSelected = false;
089  private String replicationClusterId = "";
090  private ThreadPoolExecutor exec;
091  private int maxThreads;
092  private Path baseNamespaceDir;
093  private Path hfileArchiveDir;
094  private boolean replicationBulkLoadDataEnabled;
095
096  @Override
097  public void init(Context context) throws IOException {
098    super.init(context);
099    this.conf = HBaseConfiguration.create(ctx.getConfiguration());
100    decorateConf();
101    this.maxRetriesMultiplier = this.conf.getInt("replication.source.maxretriesmultiplier", 300);
102    this.socketTimeoutMultiplier = this.conf.getInt("replication.source.socketTimeoutMultiplier",
103        maxRetriesMultiplier);
104    // TODO: This connection is replication specific or we should make it particular to
105    // replication and make replication specific settings such as compression or codec to use
106    // passing Cells.
107    this.conn = (HConnection) ConnectionFactory.createConnection(this.conf);
108    this.sleepForRetries =
109        this.conf.getLong("replication.source.sleepforretries", 1000);
110    this.metrics = context.getMetrics();
111    // ReplicationQueueInfo parses the peerId out of the znode for us
112    this.replicationSinkMgr = new ReplicationSinkManager(conn, ctx.getPeerId(), this, this.conf);
113    // per sink thread pool
114    this.maxThreads = this.conf.getInt(HConstants.REPLICATION_SOURCE_MAXTHREADS_KEY,
115      HConstants.REPLICATION_SOURCE_MAXTHREADS_DEFAULT);
116    this.exec = new ThreadPoolExecutor(maxThreads, maxThreads, 60, TimeUnit.SECONDS,
117        new LinkedBlockingQueue<Runnable>());
118    this.exec.allowCoreThreadTimeOut(true);
119
120    this.replicationBulkLoadDataEnabled =
121        conf.getBoolean(HConstants.REPLICATION_BULKLOAD_ENABLE_KEY,
122          HConstants.REPLICATION_BULKLOAD_ENABLE_DEFAULT);
123    if (this.replicationBulkLoadDataEnabled) {
124      replicationClusterId = this.conf.get(HConstants.REPLICATION_CLUSTER_ID);
125    }
126    // Construct base namespace directory and hfile archive directory path
127    Path rootDir = FSUtils.getRootDir(conf);
128    Path baseNSDir = new Path(HConstants.BASE_NAMESPACE_DIR);
129    baseNamespaceDir = new Path(rootDir, baseNSDir);
130    hfileArchiveDir = new Path(rootDir, new Path(HConstants.HFILE_ARCHIVE_DIRECTORY, baseNSDir));
131  }
132
133  private void decorateConf() {
134    String replicationCodec = this.conf.get(HConstants.REPLICATION_CODEC_CONF_KEY);
135    if (StringUtils.isNotEmpty(replicationCodec)) {
136      this.conf.set(HConstants.RPC_CODEC_CONF_KEY, replicationCodec);
137    }
138  }
139
140  private void connectToPeers() {
141    getRegionServers();
142
143    int sleepMultiplier = 1;
144
145    // Connect to peer cluster first, unless we have to stop
146    while (this.isRunning() && replicationSinkMgr.getNumSinks() == 0) {
147      replicationSinkMgr.chooseSinks();
148      if (this.isRunning() && replicationSinkMgr.getNumSinks() == 0) {
149        if (sleepForRetries("Waiting for peers", sleepMultiplier)) {
150          sleepMultiplier++;
151        }
152      }
153    }
154  }
155
156  /**
157   * Do the sleeping logic
158   * @param msg Why we sleep
159   * @param sleepMultiplier by how many times the default sleeping time is augmented
160   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
161   */
162  protected boolean sleepForRetries(String msg, int sleepMultiplier) {
163    try {
164      if (LOG.isTraceEnabled()) {
165        LOG.trace(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
166      }
167      Thread.sleep(this.sleepForRetries * sleepMultiplier);
168    } catch (InterruptedException e) {
169      LOG.debug("Interrupted while sleeping between retries");
170    }
171    return sleepMultiplier < maxRetriesMultiplier;
172  }
173
174  /**
175   * Do the shipping logic
176   */
177  @Override
178  public boolean replicate(ReplicateContext replicateContext) {
179    CompletionService<Integer> pool = new ExecutorCompletionService<Integer>(this.exec);
180    List<Entry> entries = replicateContext.getEntries();
181    String walGroupId = replicateContext.getWalGroupId();
182    int sleepMultiplier = 1;
183    int numReplicated = 0;
184
185    if (!peersSelected && this.isRunning()) {
186      connectToPeers();
187      peersSelected = true;
188    }
189
190    int numSinks = replicationSinkMgr.getNumSinks();
191    if (numSinks == 0) {
192      LOG.warn("No replication sinks found, returning without replicating. The source should retry"
193          + " with the same set of edits.");
194      return false;
195    }
196
197    // minimum of: configured threads, number of 100-waledit batches,
198    //  and number of current sinks
199    int n = Math.min(Math.min(this.maxThreads, entries.size()/100+1), numSinks);
200
201    List<List<Entry>> entryLists = new ArrayList<List<Entry>>(n);
202    if (n == 1) {
203      entryLists.add(entries);
204    } else {
205      for (int i=0; i<n; i++) {
206        entryLists.add(new ArrayList<Entry>(entries.size()/n+1));
207      }
208      // now group by region
209      for (Entry e : entries) {
210        entryLists.get(Math.abs(Bytes.hashCode(e.getKey().getEncodedRegionName())%n)).add(e);
211      }
212    }
213    while (this.isRunning()) {
214      if (!isPeerEnabled()) {
215        if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
216          sleepMultiplier++;
217        }
218        continue;
219      }
220      try {
221        if (LOG.isTraceEnabled()) {
222          LOG.trace("Replicating " + entries.size() +
223              " entries of total size " + replicateContext.getSize());
224        }
225
226        int futures = 0;
227        for (int i=0; i<entryLists.size(); i++) {
228          if (!entryLists.get(i).isEmpty()) {
229            if (LOG.isTraceEnabled()) {
230              LOG.trace("Submitting " + entryLists.get(i).size() +
231                  " entries of total size " + replicateContext.getSize());
232            }
233            // RuntimeExceptions encountered here bubble up and are handled in ReplicationSource
234            pool.submit(createReplicator(entryLists.get(i), i));
235            futures++;
236          }
237        }
238        IOException iox = null;
239
240        for (int i=0; i<futures; i++) {
241          try {
242            // wait for all futures, remove successful parts
243            // (only the remaining parts will be retried)
244            Future<Integer> f = pool.take();
245            int index = f.get().intValue();
246            int batchSize =  entryLists.get(index).size();
247            entryLists.set(index, Collections.<Entry>emptyList());
248            // Now, we have marked the batch as done replicating, record its size
249            numReplicated += batchSize;
250          } catch (InterruptedException ie) {
251            iox =  new IOException(ie);
252          } catch (ExecutionException ee) {
253            // cause must be an IOException
254            iox = (IOException)ee.getCause();
255          }
256        }
257        if (iox != null) {
258          // if we had any exceptions, try again
259          throw iox;
260        }
261        if (numReplicated != entries.size()) {
262          // Something went wrong here and we don't know what, let's just fail and retry.
263          LOG.warn("The number of edits replicated is different from the number received,"
264              + " failing for now.");
265          return false;
266        }
267        // update metrics
268        this.metrics.setAgeOfLastShippedOp(entries.get(entries.size() - 1).getKey().getWriteTime(),
269          walGroupId);
270        return true;
271
272      } catch (IOException ioe) {
273        // Didn't ship anything, but must still age the last time we did
274        this.metrics.refreshAgeOfLastShippedOp(walGroupId);
275        if (ioe instanceof RemoteException) {
276          ioe = ((RemoteException) ioe).unwrapRemoteException();
277          LOG.warn("Can't replicate because of an error on the remote cluster: ", ioe);
278          if (ioe instanceof TableNotFoundException) {
279            if (sleepForRetries("A table is missing in the peer cluster. "
280                + "Replication cannot proceed without losing data.", sleepMultiplier)) {
281              sleepMultiplier++;
282            }
283          }
284        } else {
285          if (ioe instanceof SocketTimeoutException) {
286            // This exception means we waited for more than 60s and nothing
287            // happened, the cluster is alive and calling it right away
288            // even for a test just makes things worse.
289            sleepForRetries("Encountered a SocketTimeoutException. Since the " +
290              "call to the remote cluster timed out, which is usually " +
291              "caused by a machine failure or a massive slowdown",
292              this.socketTimeoutMultiplier);
293          } else if (ioe instanceof ConnectException) {
294            LOG.warn("Peer is unavailable, rechecking all sinks: ", ioe);
295            replicationSinkMgr.chooseSinks();
296          } else {
297            LOG.warn("Can't replicate because of a local or network error: ", ioe);
298          }
299        }
300        if (sleepForRetries("Since we are unable to replicate", sleepMultiplier)) {
301          sleepMultiplier++;
302        }
303      }
304    }
305    return false; // in case we exited before replicating
306  }
307
308  protected boolean isPeerEnabled() {
309    return ctx.getReplicationPeer().getPeerState() == PeerState.ENABLED;
310  }
311
312  @Override
313  protected void doStop() {
314    disconnect(); //don't call super.doStop()
315    if (this.conn != null) {
316      try {
317        this.conn.close();
318        this.conn = null;
319      } catch (IOException e) {
320        LOG.warn("Failed to close the connection");
321      }
322    }
323    exec.shutdownNow();
324    notifyStopped();
325  }
326
327  // is this needed? Nobody else will call doStop() otherwise
328  @Override
329  public State stopAndWait() {
330    doStop();
331    return super.stopAndWait();
332  }
333
334  @VisibleForTesting
335  protected Replicator createReplicator(List<Entry> entries, int ordinal) {
336    return new Replicator(entries, ordinal);
337  }
338
339  @VisibleForTesting
340  protected class Replicator implements Callable<Integer> {
341    private List<Entry> entries;
342    private int ordinal;
343    public Replicator(List<Entry> entries, int ordinal) {
344      this.entries = entries;
345      this.ordinal = ordinal;
346    }
347
348    @Override
349    public Integer call() throws IOException {
350      SinkPeer sinkPeer = null;
351      try {
352        sinkPeer = replicationSinkMgr.getReplicationSink();
353        BlockingInterface rrs = sinkPeer.getRegionServer();
354        ReplicationProtbufUtil.replicateWALEntry(rrs, entries.toArray(new Entry[entries.size()]),
355          replicationClusterId, baseNamespaceDir, hfileArchiveDir);
356        replicationSinkMgr.reportSinkSuccess(sinkPeer);
357        return ordinal;
358
359      } catch (IOException ioe) {
360        if (sinkPeer != null) {
361          replicationSinkMgr.reportBadSink(sinkPeer);
362        }
363        throw ioe;
364      }
365    }
366
367  }
