072/**
073 * Class that handles the source of a replication stream.
074 * Currently does not handle more than 1 slave
075 * For each slave cluster it selects a random number of peers
076 * using a replication ratio. For example, if replication ration = 0.1
077 * and slave cluster has 100 region servers, 10 will be selected.
078 * <p>
079 * A stream is considered down when we cannot contact a region server on the
080 * peer cluster for more than 55 seconds by default.
081 * </p>
082 */
083@InterfaceAudience.Private
084public class ReplicationSource implements ReplicationSourceInterface {

161    this.sleepForRetries =
162        this.conf.getLong("replication.source.sleepforretries", 1000);    // 1 second
163    this.maxRetriesMultiplier =
164        this.conf.getInt("replication.source.maxretriesmultiplier", 300); // 5 minutes @ 1 sec per
165    this.queueSizePerGroup = this.conf.getInt("hbase.regionserver.maxlogs", 32);
166    this.queueStorage = queueStorage;
167    this.replicationPeer = replicationPeer;
168    this.manager = manager;
169    this.fs = fs;
170    this.metrics = metrics;
171    this.clusterId = clusterId;
172
173    this.queueId = queueId;
174    this.replicationQueueInfo = new ReplicationQueueInfo(queueId);
175    this.logQueueWarnThreshold = this.conf.getInt("replication.source.log.queue.warn", 2);
176
177    defaultBandwidth = this.conf.getLong("replication.source.per.peer.node.bandwidth", 0);
178    currentBandwidth = getCurrentBandwidth();
179    this.throttler = new ReplicationThrottler((double) currentBandwidth / 10.0);
180    this.totalBufferUsed = manager.getTotalBufferUsed();
181    this.walFileLengthProvider = walFileLengthProvider;
182    LOG.info("queueId={}, ReplicationSource: {}, currentBandwidth={}", queueId,
183      replicationPeer.getId(), this.currentBandwidth);
184  }

193  @Override
194  public void enqueueLog(Path log) {
195    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(log.getName());
196    PriorityBlockingQueue<Path> queue = queues.get(logPrefix);
197    if (queue == null) {
198      queue = new PriorityBlockingQueue<>(queueSizePerGroup, new LogsComparator());
199      // make sure that we do not use an empty queue when setting up a ReplicationSource, otherwise
200      // the shipper may quit immediately
201      queue.put(log);
202      queues.put(logPrefix, queue);
203      if (this.isSourceActive() && this.walEntryFilter != null) {
204        // new wal group observed after source startup, start a new worker thread to track it
205        // notice: it's possible that log enqueued when this.running is set but worker thread
206        // still not launched, so it's necessary to check workerThreads before start the worker
207        tryStartNewShipper(logPrefix, queue);
208      }
209    } else {
210      queue.put(log);
211    }
212    if (LOG.isTraceEnabled()) {
213      LOG.trace("{} Added log file {} to queue of source {}.", logPeerId(), logPrefix,
214        this.replicationQueueInfo.getQueueId());
215    }
216    this.metrics.incrSizeOfLogQueue();
217    // This will log a warning for each new log that gets created above the warn threshold
218    int queueSize = queue.size();
219    if (queueSize > this.logQueueWarnThreshold) {
220      LOG.warn("{} WAL group {} queue size: {} exceeds value of "
221        + "replication.source.log.queue.warn: {}", logPeerId(),
222        logPrefix, queueSize, logQueueWarnThreshold);
223    }
224  }

226  @Override
227  public void addHFileRefs(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
228      throws ReplicationException {
229    String peerId = replicationPeer.getId();
230    Map<TableName, List<String>> tableCFMap = replicationPeer.getTableCFs();
231    if (tableCFMap != null) {
232      List<String> tableCfs = tableCFMap.get(tableName);
233      if (tableCFMap.containsKey(tableName)
234          && (tableCfs == null || tableCfs.contains(Bytes.toString(family)))) {
235        this.queueStorage.addHFileRefs(peerId, pairs);
236        metrics.incrSizeOfHFileRefsQueue(pairs.size());
237      } else {
238        LOG.debug("HFiles will not be replicated belonging to the table {} family {} to peer id {}",
239          tableName, Bytes.toString(family), peerId);
240      }
241    } else {
242      // user has explicitly not defined any table cfs for replication, means replicate all the
243      // data
244      this.queueStorage.addHFileRefs(peerId, pairs);
245      metrics.incrSizeOfHFileRefsQueue(pairs.size());
246    }
247  }

249  private ReplicationEndpoint createReplicationEndpoint()
250      throws InstantiationException, IllegalAccessException, ClassNotFoundException, IOException {
251    RegionServerCoprocessorHost rsServerHost = null;
252    if (server instanceof HRegionServer) {
253      rsServerHost = ((HRegionServer) server).getRegionServerCoprocessorHost();
254    }
255    String replicationEndpointImpl = replicationPeer.getPeerConfig().getReplicationEndpointImpl();
256
257    ReplicationEndpoint replicationEndpoint;
258    if (replicationEndpointImpl == null) {
259      // Default to HBase inter-cluster replication endpoint; skip reflection
260      replicationEndpoint = new HBaseInterClusterReplicationEndpoint();
261    } else {
262      try {
263        replicationEndpoint = Class.forName(replicationEndpointImpl)
264            .asSubclass(ReplicationEndpoint.class)
265            .getDeclaredConstructor()
266            .newInstance();
267      } catch (NoSuchMethodException | InvocationTargetException e) {
268        throw new IllegalArgumentException(e);
269      }
270    }
271    if (rsServerHost != null) {
272      ReplicationEndpoint newReplicationEndPoint =
273        rsServerHost.postCreateReplicationEndPoint(replicationEndpoint);
274      if (newReplicationEndPoint != null) {
275        // Override the newly created endpoint from the hook with configured end point
276        replicationEndpoint = newReplicationEndPoint;
277      }
278    }
279    return replicationEndpoint;
280  }
281
282  private void initAndStartReplicationEndpoint(ReplicationEndpoint replicationEndpoint)
283      throws IOException, TimeoutException {
284    TableDescriptors tableDescriptors = null;
285    if (server instanceof HRegionServer) {
286      tableDescriptors = ((HRegionServer) server).getTableDescriptors();
287    }
288    replicationEndpoint
289      .init(new ReplicationEndpoint.Context(server, conf, replicationPeer.getConfiguration(), fs,
290        replicationPeer.getId(), clusterId, replicationPeer, metrics, tableDescriptors, server));
291    replicationEndpoint.start();
292    replicationEndpoint.awaitRunning(waitOnEndpointSeconds, TimeUnit.SECONDS);
293  }

295  private void initializeWALEntryFilter(UUID peerClusterId) {
296    // get the WALEntryFilter from ReplicationEndpoint and add it to default filters
297    ArrayList<WALEntryFilter> filters =
298      Lists.<WALEntryFilter> newArrayList(new SystemTableWALEntryFilter());
299    WALEntryFilter filterFromEndpoint = this.replicationEndpoint.getWALEntryfilter();
300    if (filterFromEndpoint != null) {
301      filters.add(filterFromEndpoint);
302    }
303    filters.add(new ClusterMarkingEntryFilter(clusterId, peerClusterId, replicationEndpoint));
304    this.walEntryFilter = new ChainWALEntryFilter(filters);
305  }

307  private void tryStartNewShipper(String walGroupId, PriorityBlockingQueue<Path> queue) {
308    ReplicationSourceShipper worker = createNewShipper(walGroupId, queue);
309    ReplicationSourceShipper extant = workerThreads.putIfAbsent(walGroupId, worker);
310    if (extant != null) {
311      if(LOG.isDebugEnabled()) {
312        LOG.debug("{} Someone has beat us to start a worker thread for wal group {}", logPeerId(),
313          walGroupId);
314      }
315    } else {
316      if(LOG.isDebugEnabled()) {
317        LOG.debug("{} Starting up worker for wal group {}", logPeerId(), walGroupId);
318      }
319      ReplicationSourceWALReader walReader =
320        createNewWALReader(walGroupId, queue, worker.getStartPosition());
321      Threads.setDaemonThreadRunning(walReader, Thread.currentThread().getName() +
322        ".replicationSource.wal-reader." + walGroupId + "," + queueId, this::uncaughtException);
323      worker.setWALReader(walReader);
324      worker.startup(this::uncaughtException);
325    }
326  }

328  @Override
329  public Map<String, ReplicationStatus> getWalGroupStatus() {
330    Map<String, ReplicationStatus> sourceReplicationStatus = new TreeMap<>();
331    long ageOfLastShippedOp, replicationDelay, fileSize;
332    for (Map.Entry<String, ReplicationSourceShipper> walGroupShipper : workerThreads.entrySet()) {
333      String walGroupId = walGroupShipper.getKey();
334      ReplicationSourceShipper shipper = walGroupShipper.getValue();
335      ageOfLastShippedOp = metrics.getAgeofLastShippedOp(walGroupId);
336      int queueSize = queues.get(walGroupId).size();
337      replicationDelay = metrics.getReplicationDelay();
338      Path currentPath = shipper.getCurrentPath();
339      fileSize = -1;
340      if (currentPath != null) {
341        try {
342          fileSize = getFileSize(currentPath);
343        } catch (IOException e) {
344          LOG.warn("Ignore the exception as the file size of HLog only affects the web ui", e);
345        }
346      } else {
347        currentPath = new Path("NO_LOGS_IN_QUEUE");
348        LOG.warn("{} No replication ongoing, waiting for new log", logPeerId());
349      }
350      ReplicationStatus.ReplicationStatusBuilder statusBuilder = ReplicationStatus.newBuilder();
351      statusBuilder.withPeerId(this.getPeerId())
352          .withQueueSize(queueSize)
353          .withWalGroup(walGroupId)
354          .withCurrentPath(currentPath)
355          .withCurrentPosition(shipper.getCurrentPosition())
356          .withFileSize(fileSize)
357          .withAgeOfLastShippedOp(ageOfLastShippedOp)
358          .withReplicationDelay(replicationDelay);
359      sourceReplicationStatus.put(this.getPeerId() + "=>" + walGroupId, statusBuilder.build());
360    }
361    return sourceReplicationStatus;
362  }

364  private long getFileSize(Path currentPath) throws IOException {
365    long fileSize;
366    try {
367      fileSize = fs.getContentSummary(currentPath).getLength();
368    } catch (FileNotFoundException e) {
369      currentPath = getArchivedLogPath(currentPath, conf);
370      fileSize = fs.getContentSummary(currentPath).getLength();
371    }
372    return fileSize;
373  }

404  @Override
405  public void tryThrottle(int batchSize) throws InterruptedException {
406    checkBandwidthChangeAndResetThrottler();
407    if (throttler.isEnabled()) {
408      long sleepTicks = throttler.getNextSleepInterval(batchSize);
409      if (sleepTicks > 0) {
410        if (LOG.isTraceEnabled()) {
411          LOG.trace("{} To sleep {}ms for throttling control", logPeerId(), sleepTicks);
412        }
413        Thread.sleep(sleepTicks);
414        // reset throttler's cycle start tick when sleep for throttling occurs
415        throttler.resetStartTick();
416      }
417    }
418  }
419
420  private void checkBandwidthChangeAndResetThrottler() {
421    long peerBandwidth = getCurrentBandwidth();
422    if (peerBandwidth != currentBandwidth) {
423      currentBandwidth = peerBandwidth;
424      throttler.setBandwidth((double) currentBandwidth / 10.0);
425      LOG.info("ReplicationSource : {} bandwidth throttling changed, currentBandWidth={}",
426        replicationPeer.getId(), currentBandwidth);
427    }
428  }

430  private long getCurrentBandwidth() {
431    long peerBandwidth = replicationPeer.getPeerBandwidth();
432    // user can set peer bandwidth to 0 to use default bandwidth
433    return peerBandwidth != 0 ? peerBandwidth : defaultBandwidth;

436  /**
437   * Do the sleeping logic
438   * @param msg Why we sleep
439   * @param sleepMultiplier by how many times the default sleeping time is augmented
440   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
441   */
442  protected boolean sleepForRetries(String msg, int sleepMultiplier) {
443    try {
444      if (LOG.isTraceEnabled()) {
445        LOG.trace("{} {}, sleeping {} times {}",
446          logPeerId(), msg, sleepForRetries, sleepMultiplier);
447      }
448      Thread.sleep(this.sleepForRetries * sleepMultiplier);
449    } catch (InterruptedException e) {
450      if(LOG.isDebugEnabled()) {
451        LOG.debug("{} Interrupted while sleeping between retries", logPeerId());
452      }
453      Thread.currentThread().interrupt();
454    }
455    return sleepMultiplier < maxRetriesMultiplier;
456  }

458  private void initialize() {
459    int sleepMultiplier = 1;
460    while (this.isSourceActive()) {
461      ReplicationEndpoint replicationEndpoint;
462      try {
463        replicationEndpoint = createReplicationEndpoint();
464      } catch (Exception e) {
465        LOG.warn("{} error creating ReplicationEndpoint, retry", logPeerId(), e);
466        if (sleepForRetries("Error creating ReplicationEndpoint", sleepMultiplier)) {
467          sleepMultiplier++;
468        }
469        continue;
470      }
471
472      try {
473        initAndStartReplicationEndpoint(replicationEndpoint);
474        this.replicationEndpoint = replicationEndpoint;
475        break;
476      } catch (Exception e) {
477        LOG.warn("{} Error starting ReplicationEndpoint, retry", logPeerId(), e);
478        replicationEndpoint.stop();
479        if (sleepForRetries("Error starting ReplicationEndpoint", sleepMultiplier)) {
480          sleepMultiplier++;
481        }
482      }
483    }
484
485    if (!this.isSourceActive()) {
486      return;
487    }
488
489    sleepMultiplier = 1;
490    UUID peerClusterId;
491    // delay this until we are in an asynchronous thread
492    for (;;) {
493      peerClusterId = replicationEndpoint.getPeerUUID();
494      if (this.isSourceActive() && peerClusterId == null) {
495        if(LOG.isDebugEnabled()) {
496          LOG.debug("{} Could not connect to Peer ZK. Sleeping for {} millis", logPeerId(),
497            (this.sleepForRetries * sleepMultiplier));
498        }
499        if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
500          sleepMultiplier++;
501        }
502      } else {
503        break;
504      }
505    }
506
507    if(!this.isSourceActive()) {
508      return;
509    }
510
511    // In rare case, zookeeper setting may be messed up. That leads to the incorrect
512    // peerClusterId value, which is the same as the source clusterId
513    if (clusterId.equals(peerClusterId) && !replicationEndpoint.canReplicateToSameCluster()) {
514      this.terminate("ClusterId " + clusterId + " is replicating to itself: peerClusterId "
515          + peerClusterId + " which is not allowed by ReplicationEndpoint:"
516          + replicationEndpoint.getClass().getName(), null, false);
517      this.manager.removeSource(this);
518      return;
519    }
520    LOG.info("{} Source: {}, is now replicating from cluster: {}; to peer cluster: {};",
521      logPeerId(), this.replicationQueueInfo.getQueueId(), clusterId, peerClusterId);
522
523    initializeWALEntryFilter(peerClusterId);
524    // start workers
525    for (Map.Entry<String, PriorityBlockingQueue<Path>> entry : queues.entrySet()) {
526      String walGroupId = entry.getKey();
527      PriorityBlockingQueue<Path> queue = entry.getValue();
528      tryStartNewShipper(walGroupId, queue);
529    }
530  }

552  public void terminate(String reason, Exception cause, boolean join) {
553    if (cause == null) {
554      LOG.info("{} Closing source {} because: {}", logPeerId(), this.queueId, reason);
555    } else {
556      LOG.error("{} Closing source {} because an error occurred: {}",
557        logPeerId(), this.queueId, reason, cause);
558    }
559    this.sourceRunning = false;
560    if (initThread != null && Thread.currentThread() != initThread) {
561      // This usually won't happen but anyway, let's wait until the initialization thread exits.
562      // And notice that we may call terminate directly from the initThread so here we need to
563      // avoid join on ourselves.
564      initThread.interrupt();
565      Threads.shutdown(initThread, this.sleepForRetries);
566    }
567    Collection<ReplicationSourceShipper> workers = workerThreads.values();
568    for (ReplicationSourceShipper worker : workers) {
569      worker.stopWorker();
570      if(worker.entryReader != null) {
571        worker.entryReader.setReaderRunning(false);
572      }
573    }
574
575    for (ReplicationSourceShipper worker : workers) {
576      if (worker.isAlive() || worker.entryReader.isAlive()) {
577        try {
578          // Wait worker to stop
579          Thread.sleep(this.sleepForRetries);
580        } catch (InterruptedException e) {
581          LOG.info("{} Interrupted while waiting {} to stop", logPeerId(), worker.getName());
582          Thread.currentThread().interrupt();
583        }
584        // If worker still is alive after waiting, interrupt it
585        if (worker.isAlive()) {
586          worker.interrupt();
587        }
588        // If entry reader is alive after waiting, interrupt it
589        if (worker.entryReader.isAlive()) {
590          worker.entryReader.interrupt();
591        }
592      }
593    }
594
595    if (this.replicationEndpoint != null) {
596      this.replicationEndpoint.stop();
597    }
598    metrics.clear();
599    if (join) {
600      for (ReplicationSourceShipper worker : workers) {
601        Threads.shutdown(worker, this.sleepForRetries);
602        LOG.info("{} ReplicationSourceWorker {} terminated", logPeerId(), worker.getName());
603      }
604      if (this.replicationEndpoint != null) {
605        try {
606          this.replicationEndpoint.awaitTerminated(sleepForRetries * maxRetriesMultiplier,
607            TimeUnit.MILLISECONDS);
608        } catch (TimeoutException te) {
609          LOG.warn("{} Got exception while waiting for endpoint to shutdown "
610            + "for replication source : {}", logPeerId(), this.queueId, te);
611        }
612      }
613    }
614    this.metrics.clear();
615  }

642  /**
643   * Comparator used to compare logs together based on their start time
644   */
645  public static class LogsComparator implements Comparator<Path> {
646
647    @Override
648    public int compare(Path o1, Path o2) {
649      return Long.compare(getTS(o1), getTS(o2));
650    }
651
652    /**
653     * <p>
654     * Split a path to get the start time
655     * </p>
656     * <p>
657     * For example: 10.20.20.171%3A60020.1277499063250
658     * </p>
659     * @param p path to split
660     * @return start time
661     */
662    private static long getTS(Path p) {
663      return AbstractFSWALProvider.getWALStartTimeFromWALName(p.getName());
664    }
665  }

680  @Override
681  public String getStats() {
682    StringBuilder sb = new StringBuilder();
683    sb.append("Total replicated edits: ").append(totalReplicatedEdits)
684        .append(", current progress: \n");
685    for (Map.Entry<String, ReplicationSourceShipper> entry : workerThreads.entrySet()) {
686      String walGroupId = entry.getKey();
687      ReplicationSourceShipper worker = entry.getValue();
688      long position = worker.getCurrentPosition();
689      Path currentPath = worker.getCurrentPath();
690      sb.append("walGroup [").append(walGroupId).append("]: ");
691      if (currentPath != null) {
692        sb.append("currently replicating from: ").append(currentPath).append(" at position: ")
693            .append(position).append("\n");
694      } else {
695        sb.append("no replication ongoing, waiting for new log");
696      }
697    }
698    return sb.toString();
699  }
