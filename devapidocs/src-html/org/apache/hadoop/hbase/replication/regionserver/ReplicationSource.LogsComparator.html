072/**
073 * Class that handles the source of a replication stream.
074 * Currently does not handle more than 1 slave
075 * For each slave cluster it selects a random number of peers
076 * using a replication ratio. For example, if replication ration = 0.1
077 * and slave cluster has 100 region servers, 10 will be selected.
078 * <p>
079 * A stream is considered down when we cannot contact a region server on the
080 * peer cluster for more than 55 seconds by default.
081 * </p>
082 */

193  @Override
194  public void enqueueLog(Path log) {
195    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(log.getName());
196    PriorityBlockingQueue<Path> queue = queues.get(logPrefix);
197    if (queue == null) {
198      queue = new PriorityBlockingQueue<>(queueSizePerGroup, new LogsComparator());
199      // make sure that we do not use an empty queue when setting up a ReplicationSource, otherwise
200      // the shipper may quit immediately
201      queue.put(log);
202      queues.put(logPrefix, queue);
203      if (this.isSourceActive() && this.walEntryFilter != null) {
204        // new wal group observed after source startup, start a new worker thread to track it
205        // notice: it's possible that log enqueued when this.running is set but worker thread
206        // still not launched, so it's necessary to check workerThreads before start the worker
207        tryStartNewShipper(logPrefix, queue);
208      }
209    } else {
210      queue.put(log);
211    }
212
213    this.metrics.incrSizeOfLogQueue();
214    // This will log a warning for each new log that gets created above the warn threshold
215    int queueSize = queue.size();
216    if (queueSize > this.logQueueWarnThreshold) {
217      LOG.warn("WAL group " + logPrefix + " queue size: " + queueSize
218          + " exceeds value of replication.source.log.queue.warn: " + logQueueWarnThreshold);
219    }
220  }

222  @Override
223  public void addHFileRefs(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
224      throws ReplicationException {
225    String peerId = replicationPeer.getId();
226    Map<TableName, List<String>> tableCFMap = replicationPeer.getTableCFs();
227    if (tableCFMap != null) {
228      List<String> tableCfs = tableCFMap.get(tableName);
229      if (tableCFMap.containsKey(tableName)
230          && (tableCfs == null || tableCfs.contains(Bytes.toString(family)))) {
231        this.queueStorage.addHFileRefs(peerId, pairs);
232        metrics.incrSizeOfHFileRefsQueue(pairs.size());
233      } else {
234        LOG.debug("HFiles will not be replicated belonging to the table " + tableName + " family "
235            + Bytes.toString(family) + " to peer id " + peerId);
236      }
237    } else {
238      // user has explicitly not defined any table cfs for replication, means replicate all the
239      // data
240      this.queueStorage.addHFileRefs(peerId, pairs);
241      metrics.incrSizeOfHFileRefsQueue(pairs.size());
242    }
243  }

245  private ReplicationEndpoint createReplicationEndpoint()
246      throws InstantiationException, IllegalAccessException, ClassNotFoundException, IOException {
247    RegionServerCoprocessorHost rsServerHost = null;
248    if (server instanceof HRegionServer) {
249      rsServerHost = ((HRegionServer) server).getRegionServerCoprocessorHost();
250    }
251    String replicationEndpointImpl = replicationPeer.getPeerConfig().getReplicationEndpointImpl();
252
253    ReplicationEndpoint replicationEndpoint;
254    if (replicationEndpointImpl == null) {
255      // Default to HBase inter-cluster replication endpoint; skip reflection
256      replicationEndpoint = new HBaseInterClusterReplicationEndpoint();
257    } else {
258      try {
259        replicationEndpoint = Class.forName(replicationEndpointImpl)
260            .asSubclass(ReplicationEndpoint.class)
261            .getDeclaredConstructor()
262            .newInstance();
263      } catch (NoSuchMethodException | InvocationTargetException e) {
264        throw new IllegalArgumentException(e);
265      }
266    }
267    if (rsServerHost != null) {
268      ReplicationEndpoint newReplicationEndPoint =
269        rsServerHost.postCreateReplicationEndPoint(replicationEndpoint);
270      if (newReplicationEndPoint != null) {
271        // Override the newly created endpoint from the hook with configured end point
272        replicationEndpoint = newReplicationEndPoint;
273      }
274    }
275    return replicationEndpoint;
276  }

278  private void initAndStartReplicationEndpoint(ReplicationEndpoint replicationEndpoint)
279      throws IOException, TimeoutException {
280    TableDescriptors tableDescriptors = null;
281    if (server instanceof HRegionServer) {
282      tableDescriptors = ((HRegionServer) server).getTableDescriptors();
283    }
284    replicationEndpoint
285      .init(new ReplicationEndpoint.Context(conf, replicationPeer.getConfiguration(), fs,
286        replicationPeer.getId(), clusterId, replicationPeer, metrics, tableDescriptors, server));
287    replicationEndpoint.start();
288    replicationEndpoint.awaitRunning(waitOnEndpointSeconds, TimeUnit.SECONDS);
289  }

291  private void initializeWALEntryFilter(UUID peerClusterId) {
292    // get the WALEntryFilter from ReplicationEndpoint and add it to default filters
293    ArrayList<WALEntryFilter> filters =
294      Lists.<WALEntryFilter> newArrayList(new SystemTableWALEntryFilter());
295    WALEntryFilter filterFromEndpoint = this.replicationEndpoint.getWALEntryfilter();
296    if (filterFromEndpoint != null) {
297      filters.add(filterFromEndpoint);
298    }
299    filters.add(new ClusterMarkingEntryFilter(clusterId, peerClusterId, replicationEndpoint));
300    this.walEntryFilter = new ChainWALEntryFilter(filters);
301  }

303  private void tryStartNewShipper(String walGroupId, PriorityBlockingQueue<Path> queue) {
304    ReplicationSourceShipper worker = createNewShipper(walGroupId, queue);
305    ReplicationSourceShipper extant = workerThreads.putIfAbsent(walGroupId, worker);
306    if (extant != null) {
307      LOG.debug("Someone has beat us to start a worker thread for wal group {}", walGroupId);
308    } else {
309      LOG.debug("Starting up worker for wal group {}", walGroupId);
310      ReplicationSourceWALReader walReader =
311        createNewWALReader(walGroupId, queue, worker.getStartPosition());
312      Threads.setDaemonThreadRunning(walReader, Thread.currentThread().getName() +
313        ".replicationSource.wal-reader." + walGroupId + "," + queueId, this::uncaughtException);
314      worker.setWALReader(walReader);
315      worker.startup(this::uncaughtException);
316    }
317  }

319  @Override
320  public Map<String, ReplicationStatus> getWalGroupStatus() {
321    Map<String, ReplicationStatus> sourceReplicationStatus = new TreeMap<>();
322    long lastTimeStamp, ageOfLastShippedOp, replicationDelay, fileSize;
323    for (Map.Entry<String, ReplicationSourceShipper> walGroupShipper : workerThreads.entrySet()) {
324      String walGroupId = walGroupShipper.getKey();
325      ReplicationSourceShipper shipper = walGroupShipper.getValue();
326      lastTimeStamp = metrics.getLastTimeStampOfWalGroup(walGroupId);
327      ageOfLastShippedOp = metrics.getAgeofLastShippedOp(walGroupId);
328      int queueSize = queues.get(walGroupId).size();
329      replicationDelay =
330          ReplicationLoad.calculateReplicationDelay(ageOfLastShippedOp, lastTimeStamp, queueSize);
331      Path currentPath = shipper.getCurrentPath();
332      try {
333        fileSize = getFileSize(currentPath);
334      } catch (IOException e) {
335        LOG.warn("Ignore the exception as the file size of HLog only affects the web ui", e);
336        fileSize = -1;
337      }
338      ReplicationStatus.ReplicationStatusBuilder statusBuilder = ReplicationStatus.newBuilder();
339      statusBuilder.withPeerId(this.getPeerId())
340          .withQueueSize(queueSize)
341          .withWalGroup(walGroupId)
342          .withCurrentPath(currentPath)
343          .withCurrentPosition(shipper.getCurrentPosition())
344          .withFileSize(fileSize)
345          .withAgeOfLastShippedOp(ageOfLastShippedOp)
346          .withReplicationDelay(replicationDelay);
347      sourceReplicationStatus.put(this.getPeerId() + "=>" + walGroupId, statusBuilder.build());
348    }
349    return sourceReplicationStatus;
350  }

391  @Override
392  public void tryThrottle(int batchSize) throws InterruptedException {
393    checkBandwidthChangeAndResetThrottler();
394    if (throttler.isEnabled()) {
395      long sleepTicks = throttler.getNextSleepInterval(batchSize);
396      if (sleepTicks > 0) {
397        if (LOG.isTraceEnabled()) {
398          LOG.trace("To sleep " + sleepTicks + "ms for throttling control");
399        }
400        Thread.sleep(sleepTicks);
401        // reset throttler's cycle start tick when sleep for throttling occurs
402        throttler.resetStartTick();
403      }
404    }
405  }
406
407  private void checkBandwidthChangeAndResetThrottler() {
408    long peerBandwidth = getCurrentBandwidth();
409    if (peerBandwidth != currentBandwidth) {
410      currentBandwidth = peerBandwidth;
411      throttler.setBandwidth((double) currentBandwidth / 10.0);
412      LOG.info("ReplicationSource : {} bandwidth throttling changed, currentBandWidth={}",
413        replicationPeer.getId(), currentBandwidth);
414    }
415  }

417  private long getCurrentBandwidth() {
418    long peerBandwidth = replicationPeer.getPeerBandwidth();
419    // user can set peer bandwidth to 0 to use default bandwidth
420    return peerBandwidth != 0 ? peerBandwidth : defaultBandwidth;
421  }

423  /**
424   * Do the sleeping logic
425   * @param msg Why we sleep
426   * @param sleepMultiplier by how many times the default sleeping time is augmented
427   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
428   */
429  protected boolean sleepForRetries(String msg, int sleepMultiplier) {
430    try {
431      if (LOG.isTraceEnabled()) {
432        LOG.trace(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
433      }
434      Thread.sleep(this.sleepForRetries * sleepMultiplier);
435    } catch (InterruptedException e) {
436      LOG.debug("Interrupted while sleeping between retries");
437      Thread.currentThread().interrupt();
438    }
439    return sleepMultiplier < maxRetriesMultiplier;
440  }

442  private void initialize() {
443    int sleepMultiplier = 1;
444    while (this.isSourceActive()) {
445      ReplicationEndpoint replicationEndpoint;
446      try {
447        replicationEndpoint = createReplicationEndpoint();
448      } catch (Exception e) {
449        LOG.warn("error creating ReplicationEndpoint, retry", e);
450        if (sleepForRetries("Error creating ReplicationEndpoint", sleepMultiplier)) {
451          sleepMultiplier++;
452        }
453        continue;
454      }
455
456      try {
457        initAndStartReplicationEndpoint(replicationEndpoint);
458        this.replicationEndpoint = replicationEndpoint;
459        break;
460      } catch (Exception e) {
461        LOG.warn("Error starting ReplicationEndpoint, retry", e);
462        replicationEndpoint.stop();
463        if (sleepForRetries("Error starting ReplicationEndpoint", sleepMultiplier)) {
464          sleepMultiplier++;
465        }
466      }
467    }
468
469    if (!this.isSourceActive()) {
470      return;
471    }
472
473    sleepMultiplier = 1;
474    UUID peerClusterId;
475    // delay this until we are in an asynchronous thread
476    for (;;) {
477      peerClusterId = replicationEndpoint.getPeerUUID();
478      if (this.isSourceActive() && peerClusterId == null) {
479        if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
480          sleepMultiplier++;
481        }
482      } else {
483        break;
484      }
485    }
486
487    // In rare case, zookeeper setting may be messed up. That leads to the incorrect
488    // peerClusterId value, which is the same as the source clusterId
489    if (clusterId.equals(peerClusterId) && !replicationEndpoint.canReplicateToSameCluster()) {
490      this.terminate("ClusterId " + clusterId + " is replicating to itself: peerClusterId "
491          + peerClusterId + " which is not allowed by ReplicationEndpoint:"
492          + replicationEndpoint.getClass().getName(), null, false);
493      this.manager.removeSource(this);
494      return;
495    }
496    LOG.info("Replicating " + clusterId + " -> " + peerClusterId);
497
498    initializeWALEntryFilter(peerClusterId);
499    // start workers
500    for (Map.Entry<String, PriorityBlockingQueue<Path>> entry : queues.entrySet()) {
501      String walGroupId = entry.getKey();
502      PriorityBlockingQueue<Path> queue = entry.getValue();
503      tryStartNewShipper(walGroupId, queue);
504    }
505  }

527  public void terminate(String reason, Exception cause, boolean join) {
528    if (cause == null) {
529      LOG.info("Closing source " + this.queueId + " because: " + reason);
530    } else {
531      LOG.error("Closing source " + this.queueId + " because an error occurred: " + reason,
532        cause);
533    }
534    this.sourceRunning = false;
535    if (initThread != null && Thread.currentThread() != initThread) {
536      // This usually won't happen but anyway, let's wait until the initialization thread exits.
537      // And notice that we may call terminate directly from the initThread so here we need to
538      // avoid join on ourselves.
539      initThread.interrupt();
540      Threads.shutdown(initThread, this.sleepForRetries);
541    }
542    Collection<ReplicationSourceShipper> workers = workerThreads.values();
543    for (ReplicationSourceShipper worker : workers) {
544      worker.stopWorker();
545      worker.entryReader.setReaderRunning(false);
546    }
547
548    for (ReplicationSourceShipper worker : workers) {
549      if (worker.isAlive() || worker.entryReader.isAlive()) {
550        try {
551          // Wait worker to stop
552          Thread.sleep(this.sleepForRetries);
553        } catch (InterruptedException e) {
554          LOG.info("Interrupted while waiting " + worker.getName() + " to stop");
555          Thread.currentThread().interrupt();
556        }
557        // If worker still is alive after waiting, interrupt it
558        if (worker.isAlive()) {
559          worker.interrupt();
560        }
561        // If entry reader is alive after waiting, interrupt it
562        if (worker.entryReader.isAlive()) {
563          worker.entryReader.interrupt();
564        }
565      }
566    }
567
568    if (this.replicationEndpoint != null) {
569      this.replicationEndpoint.stop();
570    }
571    metrics.clear();
572    if (join) {
573      for (ReplicationSourceShipper worker : workers) {
574        Threads.shutdown(worker, this.sleepForRetries);
575        LOG.info("ReplicationSourceWorker " + worker.getName() + " terminated");
576      }
577      if (this.replicationEndpoint != null) {
578        try {
579          this.replicationEndpoint.awaitTerminated(sleepForRetries * maxRetriesMultiplier,
580            TimeUnit.MILLISECONDS);
581        } catch (TimeoutException te) {
582          LOG.warn("Got exception while waiting for endpoint to shutdown for replication source :" +
583            this.queueId, te);
584        }
585      }
586    }
587    this.metrics.clear();
588  }

629     * @return start time
630     */
631    private static long getTS(Path p) {
632      return AbstractFSWALProvider.getWALStartTimeFromWALName(p.getName());
633    }
634  }

636  @Override
637  public String getStats() {
638    StringBuilder sb = new StringBuilder();
639    sb.append("Total replicated edits: ").append(totalReplicatedEdits)
640        .append(", current progress: \n");
641    for (Map.Entry<String, ReplicationSourceShipper> entry : workerThreads.entrySet()) {
642      String walGroupId = entry.getKey();
643      ReplicationSourceShipper worker = entry.getValue();
644      long position = worker.getCurrentPosition();
645      Path currentPath = worker.getCurrentPath();
646      sb.append("walGroup [").append(walGroupId).append("]: ");
647      if (currentPath != null) {
648        sb.append("currently replicating from: ").append(currentPath).append(" at position: ")
649            .append(position).append("\n");
650      } else {
651        sb.append("no replication ongoing, waiting for new log");
652      }
653    }
654    return sb.toString();
655  }
