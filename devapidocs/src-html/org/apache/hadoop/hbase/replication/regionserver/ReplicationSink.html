063/**
064 * <p>
065 * This class is responsible for replicating the edits coming
066 * from another cluster.
067 * </p><p>
068 * This replication process is currently waiting for the edits to be applied
069 * before the method can return. This means that the replication of edits
070 * is synchronized (after reading from WALs in ReplicationSource) and that a
071 * single region server cannot receive edits from two sources at the same time
072 * </p><p>
073 * This class uses the native HBase client in order to replicate entries.
074 * </p>
075 *
076 * TODO make this class more like ReplicationSource wrt log handling
077 */
078@InterfaceAudience.Private
079public class ReplicationSink {
080
081  private static final Log LOG = LogFactory.getLog(ReplicationSink.class);
082  private final Configuration conf;
083  // Volatile because of note in here -- look for double-checked locking:
084  // http://www.oracle.com/technetwork/articles/javase/bloch-effective-08-qa-140880.html
085  private volatile Connection sharedHtableCon;
086  private final MetricsSink metrics;
087  private final AtomicLong totalReplicatedEdits = new AtomicLong();
088  private final Object sharedHtableConLock = new Object();
089  // Number of hfiles that we successfully replicated
090  private long hfilesReplicated = 0;
091  private SourceFSConfigurationProvider provider;
092
093  /**
094   * Create a sink for replication
095   *
096   * @param conf                conf object
097   * @param stopper             boolean to tell this thread to stop
098   * @throws IOException thrown when HDFS goes bad or bad file name
099   */
100  public ReplicationSink(Configuration conf, Stoppable stopper)
101      throws IOException {
102    this.conf = HBaseConfiguration.create(conf);
103    decorateConf();
104    this.metrics = new MetricsSink();
105
106    String className =
107        conf.get("hbase.replication.source.fs.conf.provider",
108          DefaultSourceFSConfigurationProvider.class.getCanonicalName());
109    try {
110      @SuppressWarnings("rawtypes")
111      Class c = Class.forName(className);
112      this.provider = (SourceFSConfigurationProvider) c.newInstance();
113    } catch (Exception e) {
114      throw new IllegalArgumentException("Configured source fs configuration provider class "
115          + className + " throws error.", e);
116    }
117  }
118
119  /**
120   * decorate the Configuration object to make replication more receptive to delays:
121   * lessen the timeout and numTries.
122   */
123  private void decorateConf() {
124    this.conf.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
125        this.conf.getInt("replication.sink.client.retries.number", 4));
126    this.conf.setInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT,
127        this.conf.getInt("replication.sink.client.ops.timeout", 10000));
128    String replicationCodec = this.conf.get(HConstants.REPLICATION_CODEC_CONF_KEY);
129    if (StringUtils.isNotEmpty(replicationCodec)) {
130      this.conf.set(HConstants.RPC_CODEC_CONF_KEY, replicationCodec);
131    }
132   }

134  /**
135   * Replicate this array of entries directly into the local cluster using the native client. Only
136   * operates against raw protobuf type saving on a conversion from pb to pojo.
137   * @param entries
138   * @param cells
139   * @param replicationClusterId Id which will uniquely identify source cluster FS client
140   *          configurations in the replication configuration directory
141   * @param sourceBaseNamespaceDirPath Path that point to the source cluster base namespace
142   *          directory
143   * @param sourceHFileArchiveDirPath Path that point to the source cluster hfile archive directory
144   * @throws IOException If failed to replicate the data
145   */
146  public void replicateEntries(List<WALEntry> entries, final CellScanner cells,
147      String replicationClusterId, String sourceBaseNamespaceDirPath,
148      String sourceHFileArchiveDirPath) throws IOException {
149    if (entries.isEmpty()) return;
150    if (cells == null) throw new NullPointerException("TODO: Add handling of null CellScanner");
151    // Very simple optimization where we batch sequences of rows going
152    // to the same table.
153    try {
154      long totalReplicated = 0;
155      // Map of table => list of Rows, grouped by cluster id, we only want to flushCommits once per
156      // invocation of this method per table and cluster id.
157      Map<TableName, Map<List<UUID>, List<Row>>> rowMap = new TreeMap<>();
158
159      // Map of table name Vs list of pair of family and list of hfile paths from its namespace
160      Map<String, List<Pair<byte[], List<String>>>> bulkLoadHFileMap = null;
161
162      for (WALEntry entry : entries) {
163        TableName table =
164            TableName.valueOf(entry.getKey().getTableName().toByteArray());
165        Cell previousCell = null;
166        Mutation m = null;
167        int count = entry.getAssociatedCellCount();
168        for (int i = 0; i < count; i++) {
169          // Throw index out of bounds if our cell count is off
170          if (!cells.advance()) {
171            throw new ArrayIndexOutOfBoundsException("Expected=" + count + ", index=" + i);
172          }
173          Cell cell = cells.current();
174          // Handle bulk load hfiles replication
175          if (CellUtil.matchingQualifier(cell, WALEdit.BULK_LOAD)) {
176            if (bulkLoadHFileMap == null) {
177              bulkLoadHFileMap = new HashMap<>();
178            }
179            buildBulkLoadHFileMap(bulkLoadHFileMap, table, cell);
180          } else {
181            // Handle wal replication
182            if (isNewRowOrType(previousCell, cell)) {
183              // Create new mutation
184              m =
185                  CellUtil.isDelete(cell) ? new Delete(cell.getRowArray(), cell.getRowOffset(),
186                      cell.getRowLength()) : new Put(cell.getRowArray(), cell.getRowOffset(),
187                      cell.getRowLength());
188              List<UUID> clusterIds = new ArrayList<>(entry.getKey().getClusterIdsList().size());
189              for (HBaseProtos.UUID clusterId : entry.getKey().getClusterIdsList()) {
190                clusterIds.add(toUUID(clusterId));
191              }
192              m.setClusterIds(clusterIds);
193              addToHashMultiMap(rowMap, table, clusterIds, m);
194            }
195            if (CellUtil.isDelete(cell)) {
196              ((Delete) m).add(cell);
197            } else {
198              ((Put) m).add(cell);
199            }
200            previousCell = cell;
201          }
202        }
203        totalReplicated++;
204      }
205
206      // TODO Replicating mutations and bulk loaded data can be made parallel
207      if (!rowMap.isEmpty()) {
208        LOG.debug("Started replicating mutations.");
209        for (Entry<TableName, Map<List<UUID>, List<Row>>> entry : rowMap.entrySet()) {
210          batch(entry.getKey(), entry.getValue().values());
211        }
212        LOG.debug("Finished replicating mutations.");
213      }
214
215      if (bulkLoadHFileMap != null && !bulkLoadHFileMap.isEmpty()) {
216        LOG.debug("Started replicating bulk loaded data.");
217        HFileReplicator hFileReplicator =
218            new HFileReplicator(this.provider.getConf(this.conf, replicationClusterId),
219                sourceBaseNamespaceDirPath, sourceHFileArchiveDirPath, bulkLoadHFileMap, conf,
220                getConnection());
221        hFileReplicator.replicate();
222        LOG.debug("Finished replicating bulk loaded data.");
223      }
224
225      int size = entries.size();
226      this.metrics.setAgeOfLastAppliedOp(entries.get(size - 1).getKey().getWriteTime());
227      this.metrics.applyBatch(size + hfilesReplicated, hfilesReplicated);
228      this.totalReplicatedEdits.addAndGet(totalReplicated);
229    } catch (IOException ex) {
230      LOG.error("Unable to accept edit because:", ex);
231      throw ex;
232    }
233  }

235  private void buildBulkLoadHFileMap(
236      final Map<String, List<Pair<byte[], List<String>>>> bulkLoadHFileMap, TableName table,
237      Cell cell) throws IOException {
238    BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cell);
239    List<StoreDescriptor> storesList = bld.getStoresList();
240    int storesSize = storesList.size();
241    for (int j = 0; j < storesSize; j++) {
242      StoreDescriptor storeDescriptor = storesList.get(j);
243      List<String> storeFileList = storeDescriptor.getStoreFileList();
244      int storeFilesSize = storeFileList.size();
245      hfilesReplicated += storeFilesSize;
246      for (int k = 0; k < storeFilesSize; k++) {
247        byte[] family = storeDescriptor.getFamilyName().toByteArray();
248
249        // Build hfile relative path from its namespace
250        String pathToHfileFromNS = getHFilePath(table, bld, storeFileList.get(k), family);
251
252        String tableName = table.getNameWithNamespaceInclAsString();
253        if (bulkLoadHFileMap.containsKey(tableName)) {
254          List<Pair<byte[], List<String>>> familyHFilePathsList = bulkLoadHFileMap.get(tableName);
255          boolean foundFamily = false;
256          for (int i = 0; i < familyHFilePathsList.size(); i++) {
257            Pair<byte[], List<String>> familyHFilePathsPair = familyHFilePathsList.get(i);
258            if (Bytes.equals(familyHFilePathsPair.getFirst(), family)) {
259              // Found family already present, just add the path to the existing list
260              familyHFilePathsPair.getSecond().add(pathToHfileFromNS);
261              foundFamily = true;
262              break;
263            }
264          }
265          if (!foundFamily) {
266            // Family not found, add this family and its hfile paths pair to the list
267            addFamilyAndItsHFilePathToTableInMap(family, pathToHfileFromNS, familyHFilePathsList);
268          }
269        } else {
270          // Add this table entry into the map
271          addNewTableEntryInMap(bulkLoadHFileMap, family, pathToHfileFromNS, tableName);
272        }
273      }
274    }
275  }
276
277  private void addFamilyAndItsHFilePathToTableInMap(byte[] family, String pathToHfileFromNS,
278      List<Pair<byte[], List<String>>> familyHFilePathsList) {
279    List<String> hfilePaths = new ArrayList<>(1);
280    hfilePaths.add(pathToHfileFromNS);
281    familyHFilePathsList.add(new Pair<>(family, hfilePaths));
282  }
283
284  private void addNewTableEntryInMap(
285      final Map<String, List<Pair<byte[], List<String>>>> bulkLoadHFileMap, byte[] family,
286      String pathToHfileFromNS, String tableName) {
287    List<String> hfilePaths = new ArrayList<>(1);
288    hfilePaths.add(pathToHfileFromNS);
289    Pair<byte[], List<String>> newFamilyHFilePathsPair = new Pair<>(family, hfilePaths);
290    List<Pair<byte[], List<String>>> newFamilyHFilePathsList = new ArrayList<>();
291    newFamilyHFilePathsList.add(newFamilyHFilePathsPair);
292    bulkLoadHFileMap.put(tableName, newFamilyHFilePathsList);
293  }
294
295  private String getHFilePath(TableName table, BulkLoadDescriptor bld, String storeFile,
296      byte[] family) {
297    return new StringBuilder(100).append(table.getNamespaceAsString()).append(Path.SEPARATOR)
298        .append(table.getQualifierAsString()).append(Path.SEPARATOR)
299        .append(Bytes.toString(bld.getEncodedRegionName().toByteArray())).append(Path.SEPARATOR)
300        .append(Bytes.toString(family)).append(Path.SEPARATOR).append(storeFile).toString();
301  }

303  /**
304   * @param previousCell
305   * @param cell
306   * @return True if we have crossed over onto a new row or type
307   */
308  private boolean isNewRowOrType(final Cell previousCell, final Cell cell) {
309    return previousCell == null || previousCell.getTypeByte() != cell.getTypeByte() ||
310        !CellUtil.matchingRows(previousCell, cell);
311  }

313  private java.util.UUID toUUID(final HBaseProtos.UUID uuid) {
314    return new java.util.UUID(uuid.getMostSigBits(), uuid.getLeastSigBits());
315  }

317  /**
318   * Simple helper to a map from key to (a list of) values
319   * TODO: Make a general utility method
320   * @param map
321   * @param key1
322   * @param key2
323   * @param value
324   * @return the list of values corresponding to key1 and key2
325   */
326  private <K1, K2, V> List<V> addToHashMultiMap(Map<K1, Map<K2,List<V>>> map, K1 key1, K2 key2, V value) {
327    Map<K2,List<V>> innerMap = map.get(key1);
328    if (innerMap == null) {
329      innerMap = new HashMap<>();
330      map.put(key1, innerMap);
331    }
332    List<V> values = innerMap.get(key2);
333    if (values == null) {
334      values = new ArrayList<>();
335      innerMap.put(key2, values);
336    }
337    values.add(value);
338    return values;
339  }

341  /**
342   * stop the thread pool executor. It is called when the regionserver is stopped.
343   */
344  public void stopReplicationSinkServices() {
345    try {
346      if (this.sharedHtableCon != null) {
347        synchronized (sharedHtableConLock) {
348          if (this.sharedHtableCon != null) {
349            this.sharedHtableCon.close();
350            this.sharedHtableCon = null;
351          }
352        }
353      }
354    } catch (IOException e) {
355      LOG.warn("IOException while closing the connection", e); // ignoring as we are closing.
356    }
357  }

360  /**
361   * Do the changes and handle the pool
362   * @param tableName table to insert into
363   * @param allRows list of actions
364   * @throws IOException
365   */
366  protected void batch(TableName tableName, Collection<List<Row>> allRows) throws IOException {
367    if (allRows.isEmpty()) {
368      return;
369    }
370    Table table = null;
371    try {
372      Connection connection = getConnection();
373      table = connection.getTable(tableName);
374      for (List<Row> rows : allRows) {
375        table.batch(rows, null);
376      }
377    } catch (RetriesExhaustedWithDetailsException rewde) {
378      for (Throwable ex : rewde.getCauses()) {
379        if (ex instanceof TableNotFoundException) {
380          throw new TableNotFoundException("'"+tableName+"'");
381        }
382      }
383    } catch (InterruptedException ix) {
384      throw (InterruptedIOException) new InterruptedIOException().initCause(ix);
385    } finally {
386      if (table != null) {
387        table.close();
388      }
389    }
390  }
391
392  private Connection getConnection() throws IOException {
393    // See https://en.wikipedia.org/wiki/Double-checked_locking
394    Connection connection = sharedHtableCon;
395    if (connection == null) {
396      synchronized (sharedHtableConLock) {
397        connection = sharedHtableCon;
398        if (connection == null) {
399          connection = sharedHtableCon = ConnectionFactory.createConnection(conf);
400        }
401      }
402    }
403    return connection;
404  }

406  /**
407   * Get a string representation of this sink's metrics
408   * @return string with the total replicated edits count and the date
409   * of the last edit that was applied
410   */
411  public String getStats() {
412    return this.totalReplicatedEdits.get() == 0 ? "" : "Sink: " +
413      "age in ms of last applied edit: " + this.metrics.refreshAgeOfLastAppliedOp() +
414      ", total replicated edits: " + this.totalReplicatedEdits;
415  }
