019package org.apache.hadoop.hbase.replication.regionserver;
020
021import org.apache.hadoop.metrics2.lib.MutableFastCounter;
022import org.apache.hadoop.metrics2.lib.MutableGaugeLong;
023
024public class MetricsReplicationGlobalSourceSource implements MetricsReplicationSourceSource{
025  private final MetricsReplicationSourceImpl rms;
026
027  private final MutableGaugeLong ageOfLastShippedOpGauge;
028  private final MutableGaugeLong sizeOfLogQueueGauge;
029  private final MutableFastCounter logReadInEditsCounter;
030  private final MutableFastCounter logEditsFilteredCounter;
031  private final MutableFastCounter shippedBatchesCounter;
032  private final MutableFastCounter shippedOpsCounter;
033  private final MutableFastCounter shippedBytesCounter;
034  @Deprecated
035  private final MutableFastCounter shippedKBsCounter;
036  private final MutableFastCounter logReadInBytesCounter;
037  private final MutableFastCounter shippedHFilesCounter;
038  private final MutableGaugeLong sizeOfHFileRefsQueueGauge;
039  private final MutableFastCounter unknownFileLengthForClosedWAL;
040  private final MutableFastCounter uncleanlyClosedWAL;
041  private final MutableFastCounter uncleanlyClosedSkippedBytes;
042  private final MutableFastCounter restartWALReading;
043  private final MutableFastCounter repeatedFileBytes;
044  private final MutableFastCounter completedWAL;
045  private final MutableFastCounter completedRecoveryQueue;
046
047  public MetricsReplicationGlobalSourceSource(MetricsReplicationSourceImpl rms) {
048    this.rms = rms;
049
050    ageOfLastShippedOpGauge = rms.getMetricsRegistry().getGauge(SOURCE_AGE_OF_LAST_SHIPPED_OP, 0L);
051
052    sizeOfLogQueueGauge = rms.getMetricsRegistry().getGauge(SOURCE_SIZE_OF_LOG_QUEUE, 0L);
053
054    shippedBatchesCounter = rms.getMetricsRegistry().getCounter(SOURCE_SHIPPED_BATCHES, 0L);
055
056    shippedOpsCounter = rms.getMetricsRegistry().getCounter(SOURCE_SHIPPED_OPS, 0L);
057
058    shippedKBsCounter = rms.getMetricsRegistry().getCounter(SOURCE_SHIPPED_KBS, 0L);
059
060    shippedBytesCounter = rms.getMetricsRegistry().getCounter(SOURCE_SHIPPED_BYTES, 0L);
061
062    logReadInBytesCounter = rms.getMetricsRegistry().getCounter(SOURCE_LOG_READ_IN_BYTES, 0L);
063
064    logReadInEditsCounter = rms.getMetricsRegistry().getCounter(SOURCE_LOG_READ_IN_EDITS, 0L);
065
066    logEditsFilteredCounter = rms.getMetricsRegistry().getCounter(SOURCE_LOG_EDITS_FILTERED, 0L);
067
068    shippedHFilesCounter = rms.getMetricsRegistry().getCounter(SOURCE_SHIPPED_HFILES, 0L);
069
070    sizeOfHFileRefsQueueGauge =
071        rms.getMetricsRegistry().getGauge(SOURCE_SIZE_OF_HFILE_REFS_QUEUE, 0L);
072
073    unknownFileLengthForClosedWAL = rms.getMetricsRegistry().getCounter(SOURCE_CLOSED_LOGS_WITH_UNKNOWN_LENGTH, 0L);
074    uncleanlyClosedWAL = rms.getMetricsRegistry().getCounter(SOURCE_UNCLEANLY_CLOSED_LOGS, 0L);
075    uncleanlyClosedSkippedBytes = rms.getMetricsRegistry().getCounter(SOURCE_UNCLEANLY_CLOSED_IGNORED_IN_BYTES, 0L);
076    restartWALReading = rms.getMetricsRegistry().getCounter(SOURCE_RESTARTED_LOG_READING, 0L);
077    repeatedFileBytes = rms.getMetricsRegistry().getCounter(SOURCE_REPEATED_LOG_FILE_BYTES, 0L);
078    completedWAL = rms.getMetricsRegistry().getCounter(SOURCE_COMPLETED_LOGS, 0L);
079    completedRecoveryQueue = rms.getMetricsRegistry().getCounter(SOURCE_COMPLETED_RECOVERY_QUEUES, 0L);
080  }
081
082  @Override public void setLastShippedAge(long age) {
083    ageOfLastShippedOpGauge.set(age);
084  }
085
086  @Override public void incrSizeOfLogQueue(int size) {
087    sizeOfLogQueueGauge.incr(size);
088  }
089
090  @Override public void decrSizeOfLogQueue(int size) {
091    sizeOfLogQueueGauge.decr(size);
092  }
093
094  @Override public void incrLogReadInEdits(long size) {
095    logReadInEditsCounter.incr(size);
096  }
097
098  @Override public void incrLogEditsFiltered(long size) {
099    logEditsFilteredCounter.incr(size);
100  }
101
102  @Override public void incrBatchesShipped(int batches) {
103    shippedBatchesCounter.incr(batches);
104  }
105
106  @Override public void incrOpsShipped(long ops) {
107    shippedOpsCounter.incr(ops);
108  }
109
110  @Override public void incrShippedBytes(long size) {
111    shippedBytesCounter.incr(size);
112    // obtained value maybe smaller than 1024. We should make sure that KB count
113    // eventually picks up even from multiple smaller updates.
114    incrementKBsCounter(shippedBytesCounter, shippedKBsCounter);
115  }
116
117  static void incrementKBsCounter(MutableFastCounter bytesCounter, MutableFastCounter kbsCounter) {
118    // Following code should be thread-safe.
119    long delta = 0;
120    while(true) {
121      long bytes = bytesCounter.value();
122      delta = (bytes / 1024) - kbsCounter.value();
123      if (delta > 0) {
124        kbsCounter.incr(delta);
125      } else {
126        break;
127      }
128    }
129  }
130
131  @Override public void incrLogReadInBytes(long size) {
132    logReadInBytesCounter.incr(size);
133  }
134
135  @Override public void clear() {
136  }

144    shippedHFilesCounter.incr(hfiles);
145  }
146
147  @Override
148  public void incrSizeOfHFileRefsQueue(long size) {
149    sizeOfHFileRefsQueueGauge.incr(size);
150  }
151

154    sizeOfHFileRefsQueueGauge.decr(size);
155  }

163  public void incrUnknownFileLengthForClosedWAL() {

165  }
166  @Override
167  public void incrUncleanlyClosedWALs() {
168    uncleanlyClosedWAL.incr(1L);
169  }
170  @Override
171  public void incrBytesSkippedInUncleanlyClosedWALs(final long bytes) {
172    uncleanlyClosedSkippedBytes.incr(bytes);
173  }
174  @Override
175  public void incrRestartedWALReading() {
176    restartWALReading.incr(1L);
177  }
178  @Override
179  public void incrRepeatedFileBytes(final long bytes) {
180    repeatedFileBytes.incr(bytes);
181  }
182  @Override
183  public void incrCompletedWAL() {
184    completedWAL.incr(1L);
185  }
186  @Override
187  public void incrCompletedRecoveryQueue() {
188    completedRecoveryQueue.incr(1L);
189  }

193    rms.init();
