044@InterfaceAudience.Private
045public class ReplicationPeerZKImpl extends ReplicationStateZKBase
046    implements ReplicationPeer, Abortable, Closeable {
047  private static final Log LOG = LogFactory.getLog(ReplicationPeerZKImpl.class);
048
049  private ReplicationPeerConfig peerConfig;
050  private final String id;
051  private volatile PeerState peerState;
052  private volatile Map<TableName, List<String>> tableCFs = new HashMap<TableName, List<String>>();
053  private final Configuration conf;
054  private PeerStateTracker peerStateTracker;
055  private PeerConfigTracker peerConfigTracker;
056
057
058  /**
059   * Constructor that takes all the objects required to communicate with the specified peer, except
060   * for the region server addresses.
061   * @param conf configuration object to this peer
062   * @param id string representation of this peer's identifier
063   * @param peerConfig configuration for the replication peer
064   */
065  public ReplicationPeerZKImpl(ZooKeeperWatcher zkWatcher, Configuration conf,
066                               String id, ReplicationPeerConfig peerConfig,
067                               Abortable abortable)
068      throws ReplicationException {
069    super(zkWatcher, conf, abortable);
070    this.conf = conf;
071    this.peerConfig = peerConfig;
072    this.id = id;
073  }
074
075  /**
076   * start a state tracker to check whether this peer is enabled or not
077   *
078   * @param peerStateNode path to zk node which stores peer state
079   * @throws KeeperException
080   */
081  public void startStateTracker(String peerStateNode)
082      throws KeeperException {
083    ensurePeerEnabled(peerStateNode);
084    this.peerStateTracker = new PeerStateTracker(peerStateNode, zookeeper, this);
085    this.peerStateTracker.start();
086    try {
087      this.readPeerStateZnode();
088    } catch (DeserializationException e) {
089      throw ZKUtil.convert(e);
090    }
091  }
092
093  private void readPeerStateZnode() throws DeserializationException {
094    this.peerState =
095        isStateEnabled(this.peerStateTracker.getData(false))
096          ? PeerState.ENABLED
097          : PeerState.DISABLED;
098  }
099
100  /**
101   * start a table-cfs tracker to listen the (table, cf-list) map change
102   * @param peerConfigNode path to zk node which stores table-cfs
103   * @throws KeeperException
104   */
105  public void startPeerConfigTracker(String peerConfigNode)
106    throws KeeperException {
107    this.peerConfigTracker = new PeerConfigTracker(peerConfigNode, zookeeper,
108        this);
109    this.peerConfigTracker.start();
110    this.readPeerConfig();
111  }
112
113  private ReplicationPeerConfig readPeerConfig() {
114    try {
115      byte[] data = peerConfigTracker.getData(false);
116      if (data != null) {
117        this.peerConfig = ReplicationSerDeHelper.parsePeerFrom(data);
118      }
119    } catch (DeserializationException e) {
120      LOG.error("", e);
121    }
122    return this.peerConfig;
123  }
124

183  @Override

187  }

192    // abort method is called.

201  /**
202   * Parse the raw data from ZK to get a peer's state
203   * @param bytes raw ZK data
204   * @return True if the passed in <code>bytes</code> are those of a pb serialized ENABLED state.
205   * @throws DeserializationException
206   */
207  public static boolean isStateEnabled(final byte[] bytes) throws DeserializationException {
208    ZooKeeperProtos.ReplicationState.State state = parseStateFrom(bytes);
209    return ZooKeeperProtos.ReplicationState.State.ENABLED == state;
210  }
211
212  /**
213   * @param bytes Content of a state znode.
214   * @return State parsed from the passed bytes.
215   * @throws DeserializationException
216   */
217  private static ZooKeeperProtos.ReplicationState.State parseStateFrom(final byte[] bytes)
218      throws DeserializationException {
219    ProtobufUtil.expectPBMagicPrefix(bytes);
220    int pblen = ProtobufUtil.lengthOfPBMagic();
221    ZooKeeperProtos.ReplicationState.Builder builder =
222        ZooKeeperProtos.ReplicationState.newBuilder();
223    ZooKeeperProtos.ReplicationState state;
224    try {
225      ProtobufUtil.mergeFrom(builder, bytes, pblen, bytes.length - pblen);
226      state = builder.build();
227      return state.getState();
228    } catch (IOException e) {
229      throw new DeserializationException(e);
230    }
231  }
232
233  /**
234   * Utility method to ensure an ENABLED znode is in place; if not present, we create it.
235   * @param path Path to znode to check
236   * @return True if we created the znode.
237   * @throws NodeExistsException
238   * @throws KeeperException
239   */
240  private boolean ensurePeerEnabled(final String path)
241      throws NodeExistsException, KeeperException {
242    if (ZKUtil.checkExists(zookeeper, path) == -1) {
243      // There is a race b/w PeerWatcher and ReplicationZookeeper#add method to create the
244      // peer-state znode. This happens while adding a peer.
245      // The peer state data is set as "ENABLED" by default.
246      ZKUtil.createNodeIfNotExistsAndWatch(zookeeper, path,
247        ReplicationStateZKBase.ENABLED_ZNODE_BYTES);
248      return true;
249    }
250    return false;
251  }

253  /**
254   * Tracker for state of this peer
255   */
256  public class PeerStateTracker extends ZooKeeperNodeTracker {
257
258    public PeerStateTracker(String peerStateZNode, ZooKeeperWatcher watcher,
259        Abortable abortable) {
260      super(watcher, peerStateZNode, abortable);
261    }
262
263    @Override
264    public synchronized void nodeDataChanged(String path) {
265      if (path.equals(node)) {
266        super.nodeDataChanged(path);
267        try {
268          readPeerStateZnode();
269        } catch (DeserializationException e) {
270          LOG.warn("Failed deserializing the content of " + path, e);
271        }
272      }
273    }
274  }

276  /**
277   * Tracker for PeerConfigNode of this peer
278   */
279  public class PeerConfigTracker extends ZooKeeperNodeTracker {
280
281    ReplicationPeerConfigListener listener;
282
283    public PeerConfigTracker(String peerConfigNode, ZooKeeperWatcher watcher,
284        Abortable abortable) {
285      super(watcher, peerConfigNode, abortable);
286    }
287
288    public synchronized void setListener(ReplicationPeerConfigListener listener){
289      this.listener = listener;
290    }
291
292    @Override
293    public synchronized void nodeCreated(String path) {
294      if (path.equals(node)) {
295        super.nodeCreated(path);
296        ReplicationPeerConfig config = readPeerConfig();
297        if (listener != null){
298          listener.peerConfigUpdated(config);
299        }
300      }
301    }
302
303    @Override
304    public synchronized void nodeDataChanged(String path) {
305      //superclass calls nodeCreated
306      if (path.equals(node)) {
307        super.nodeDataChanged(path);
308      }
309
310    }
311
312  }
