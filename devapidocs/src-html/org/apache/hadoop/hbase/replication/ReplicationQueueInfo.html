019package org.apache.hadoop.hbase.replication;
020
021
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.List;
025
026import org.apache.commons.logging.Log;
027import org.apache.commons.logging.LogFactory;
028
029import org.apache.yetus.audience.InterfaceAudience;
030import org.apache.hadoop.hbase.ServerName;
031
032/**
033 * This class is responsible for the parsing logic for a znode representing a queue.
034 * It will extract the peerId if it's recovered as well as the dead region servers
035 * that were part of the queue's history.
036 */
037@InterfaceAudience.Private
038public class ReplicationQueueInfo {
039  private static final Log LOG = LogFactory.getLog(ReplicationQueueInfo.class);
040
041  private final String peerId;
042  private final String peerClusterZnode;
043  private boolean queueRecovered;
044  // List of all the dead region servers that had this queue (if recovered)
045  private List<String> deadRegionServers = new ArrayList<>();
046
047  /**
048   * The passed znode will be either the id of the peer cluster or
049   * the handling story of that queue in the form of id-servername-*
050   */
051  public ReplicationQueueInfo(String znode) {
052    this.peerClusterZnode = znode;
053    String[] parts = znode.split("-", 2);
054    this.queueRecovered = parts.length != 1;
055    this.peerId = this.queueRecovered ?
056        parts[0] : peerClusterZnode;
057    if (parts.length >= 2) {
058      // extract dead servers
059      extractDeadServersFromZNodeString(parts[1], this.deadRegionServers);
060    }
061  }
062
063  /**
064   * Parse dead server names from znode string servername can contain "-" such as
065   * "ip-10-46-221-101.ec2.internal", so we need skip some "-" during parsing for the following
066   * cases: 2-ip-10-46-221-101.ec2.internal,52170,1364333181125-&lt;server name>-...
067   */
068  private static void
069      extractDeadServersFromZNodeString(String deadServerListStr, List<String> result) {
070
071    if(deadServerListStr == null || result == null || deadServerListStr.isEmpty()) return;
072
073    // valid server name delimiter "-" has to be after "," in a server name
074    int seenCommaCnt = 0;
075    int startIndex = 0;
076    int len = deadServerListStr.length();
077
078    for (int i = 0; i < len; i++) {
079      switch (deadServerListStr.charAt(i)) {
080      case ',':
081        seenCommaCnt += 1;
082        break;
083      case '-':
084        if(seenCommaCnt>=2) {
085          if (i > startIndex) {
086            String serverName = deadServerListStr.substring(startIndex, i);
087            if(ServerName.isFullServerName(serverName)){
088              result.add(serverName);
089            } else {
090              LOG.error("Found invalid server name:" + serverName);
091            }
092            startIndex = i + 1;
093          }
094          seenCommaCnt = 0;
095        }
096        break;
097      default:
098        break;
099      }
100    }
101
102    // add tail
103    if(startIndex < len - 1){
104      String serverName = deadServerListStr.substring(startIndex, len);
105      if(ServerName.isFullServerName(serverName)){
106        result.add(serverName);
107      } else {
108        LOG.error("Found invalid server name at the end:" + serverName);
109      }
110    }
111
112    LOG.debug("Found dead servers:" + result);
113  }
114
115  public List<String> getDeadRegionServers() {
116    return Collections.unmodifiableList(this.deadRegionServers);
117  }
118
119  public String getPeerId() {
120    return this.peerId;
121  }
122
123  public String getPeerClusterZnode() {
124    return this.peerClusterZnode;
125  }
126
127  public boolean isQueueRecovered() {
128    return queueRecovered;
129  }
130}
