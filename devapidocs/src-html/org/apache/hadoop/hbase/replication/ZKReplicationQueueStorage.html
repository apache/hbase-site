057/**
058 * ZK based replication queue storage.
059 * <p>
060 * The base znode for each regionserver is the regionserver name. For example:
061 *
062 * <pre>
063 * /hbase/replication/rs/hostname.example.org,6020,1234
064 * </pre>
065 *
066 * Within this znode, the region server maintains a set of WAL replication queues. These queues are
067 * represented by child znodes named using there give queue id. For example:
068 *
069 * <pre>
070 * /hbase/replication/rs/hostname.example.org,6020,1234/1
071 * /hbase/replication/rs/hostname.example.org,6020,1234/2
072 * </pre>
073 *
074 * Each queue has one child znode for every WAL that still needs to be replicated. The value of
075 * these WAL child znodes is the latest position that has been replicated. This position is updated
076 * every time a WAL entry is replicated. For example:
077 *
078 * <pre>
079 * /hbase/replication/rs/hostname.example.org,6020,1234/1/23522342.23422 [VALUE: 254]
080 * </pre>
081 */
082@InterfaceAudience.Private
083class ZKReplicationQueueStorage extends ZKReplicationStorageBase
084    implements ReplicationQueueStorage {
085

108  public ZKReplicationQueueStorage(ZKWatcher zookeeper, Configuration conf) {
109    super(zookeeper, conf);
110
111    String queuesZNodeName = conf.get("zookeeper.znode.replication.rs", "rs");
112    String hfileRefsZNodeName = conf.get(ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_KEY,
113      ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_DEFAULT);
114    this.queuesZNode = ZNodePaths.joinZNode(replicationZNode, queuesZNodeName);
115    this.hfileRefsZNode = ZNodePaths.joinZNode(replicationZNode, hfileRefsZNodeName);
116    this.regionsZNode = ZNodePaths.joinZNode(replicationZNode, conf
117        .get(ZOOKEEPER_ZNODE_REPLICATION_REGIONS_KEY, ZOOKEEPER_ZNODE_REPLICATION_REGIONS_DEFAULT));
118  }

136  /**
137   * <p>
138   * Put all regions under /hbase/replication/regions znode will lead to too many children because
139   * of the huge number of regions in real production environment. So here we will distribute the
140   * znodes to multiple directories.
141   * </p>
142   * <p>
143   * So the final znode path will be format like this:
144   *
145   * <pre>
146   * /hbase/replication/regions/dd/04/e76a6966d4ffa908ed0586764767-100
147   * </pre>
148   *
149   * Here the full encoded region name is dd04e76a6966d4ffa908ed0586764767, and we use the first two
150   * characters 'dd' as the first level directory name, and use the next two characters '04' as the
151   * second level directory name, and the rest part as the prefix of the znode, and the suffix '100'
152   * is the peer id.
153   * </p>
154   * @param encodedRegionName the encoded region name.
155   * @param peerId peer id for replication.
156   * @return ZNode path to persist the max sequence id that we've pushed for the given region and
157   *         peer.
158   */
159  @VisibleForTesting
160  String getSerialReplicationRegionPeerNode(String encodedRegionName, String peerId) {
161    if (encodedRegionName == null || encodedRegionName.length() != RegionInfo.MD5_HEX_LENGTH) {
162      throw new IllegalArgumentException(
163          "Invalid encoded region name: " + encodedRegionName + ", length should be 32.");
164    }
165    return new StringBuilder(regionsZNode).append(ZNodePaths.ZNODE_PATH_SEPARATOR)
166        .append(encodedRegionName.substring(0, 2)).append(ZNodePaths.ZNODE_PATH_SEPARATOR)
167        .append(encodedRegionName.substring(2, 4)).append(ZNodePaths.ZNODE_PATH_SEPARATOR)
168        .append(encodedRegionName.substring(4)).append("-").append(peerId).toString();
169  }

206  @Override
207  public void setWALPosition(ServerName serverName, String queueId, String fileName, long position,
208      Map<String, Long> lastSeqIds) throws ReplicationException {
209    try {
210      List<ZKUtilOp> listOfOps = new ArrayList<>();
211      if (position > 0) {
212        listOfOps.add(ZKUtilOp.setData(getFileNode(serverName, queueId, fileName),
213          ZKUtil.positionToByteArray(position)));
214      }
215      // Persist the max sequence id(s) of regions for serial replication atomically.
216      for (Entry<String, Long> lastSeqEntry : lastSeqIds.entrySet()) {
217        String peerId = new ReplicationQueueInfo(queueId).getPeerId();
218        String path = getSerialReplicationRegionPeerNode(lastSeqEntry.getKey(), peerId);
219        /*
220         * Make sure the existence of path
221         * /hbase/replication/regions/<hash>/<encoded-region-name>-<peer-id>. As the javadoc in
222         * multiOrSequential() method said, if received a NodeExistsException, all operations will
223         * fail. So create the path here, and in fact, no need to add this operation to listOfOps,
224         * because only need to make sure that update file position and sequence id atomically.
225         */
226        ZKUtil.createWithParents(zookeeper, path);
227        // Persist the max sequence id of region to zookeeper.
228        listOfOps.add(ZKUtilOp.setData(path, ZKUtil.positionToByteArray(lastSeqEntry.getValue())));
229      }
230      if (!listOfOps.isEmpty()) {
231        ZKUtil.multiOrSequential(zookeeper, listOfOps, false);
232      }
233    } catch (KeeperException e) {
234      throw new ReplicationException("Failed to set log position (serverName=" + serverName
235          + ", queueId=" + queueId + ", fileName=" + fileName + ", position=" + position + ")", e);
236    }
237  }

239  @Override
240  public long getLastSequenceId(String encodedRegionName, String peerId)
241      throws ReplicationException {
242    byte[] data;
243    try {
244      data =
245          ZKUtil.getData(zookeeper, getSerialReplicationRegionPeerNode(encodedRegionName, peerId));
246    } catch (KeeperException | InterruptedException e) {
247      throw new ReplicationException("Failed to get the last sequence id(region="
248          + encodedRegionName + ", peerId=" + peerId + ")");
249    }
250    try {
251      return ZKUtil.parseWALPositionFrom(data);
252    } catch (DeserializationException de) {
253      LOG.warn("Failed to parse log position (region=" + encodedRegionName + ", peerId=" + peerId
254          + "), data=" + Bytes.toStringBinary(data));
255    }
256    return HConstants.NO_SEQNUM;
257  }

259  @Override
260  public long getWALPosition(ServerName serverName, String queueId, String fileName)
261      throws ReplicationException {
262    byte[] bytes;
263    try {
264      bytes = ZKUtil.getData(zookeeper, getFileNode(serverName, queueId, fileName));
265    } catch (KeeperException | InterruptedException e) {
266      throw new ReplicationException("Failed to get log position (serverName=" + serverName +
267        ", queueId=" + queueId + ", fileName=" + fileName + ")", e);
268    }
269    try {
270      return ZKUtil.parseWALPositionFrom(bytes);
271    } catch (DeserializationException de) {
272      LOG.warn("Failed parse log position (serverName={}, queueId={}, fileName={})",
273          serverName, queueId, fileName);
274    }
275    // if we can not parse the position, start at the beginning of the wal file again
276    return 0;
277  }

279  @Override
280  public Pair<String, SortedSet<String>> claimQueue(ServerName sourceServerName, String queueId,
281      ServerName destServerName) throws ReplicationException {
282    LOG.info("Atomically moving {}/{}'s WALs to {}", sourceServerName, queueId, destServerName);
283    try {
284      ZKUtil.createWithParents(zookeeper, getRsNode(destServerName));
285    } catch (KeeperException e) {
286      throw new ReplicationException(
287          "Claim queue queueId=" + queueId + " from " + sourceServerName + " to " + destServerName +
288            " failed when creating the node for " + destServerName,
289          e);
290    }
291    try {
292      String oldQueueNode = getQueueNode(sourceServerName, queueId);
293      List<String> wals = ZKUtil.listChildrenNoWatch(zookeeper, oldQueueNode);
294      String newQueueId = queueId + "-" + sourceServerName;
295      if (CollectionUtils.isEmpty(wals)) {
296        ZKUtil.deleteNodeFailSilent(zookeeper, oldQueueNode);
297        LOG.info("Removed empty {}/{}", sourceServerName, queueId);
298        return new Pair<>(newQueueId, Collections.emptySortedSet());
299      }
300      String newQueueNode = getQueueNode(destServerName, newQueueId);
301      List<ZKUtilOp> listOfOps = new ArrayList<>();
302      SortedSet<String> logQueue = new TreeSet<>();
303      // create the new cluster znode
304      listOfOps.add(ZKUtilOp.createAndFailSilent(newQueueNode, HConstants.EMPTY_BYTE_ARRAY));
305      // get the offset of the logs and set it to new znodes
306      for (String wal : wals) {
307        String oldWalNode = getFileNode(oldQueueNode, wal);
308        byte[] logOffset = ZKUtil.getData(this.zookeeper, oldWalNode);
309        LOG.debug("Creating {} with data {}", wal, Bytes.toStringBinary(logOffset));
310        String newWalNode = getFileNode(newQueueNode, wal);
311        listOfOps.add(ZKUtilOp.createAndFailSilent(newWalNode, logOffset));
312        listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldWalNode));
313        logQueue.add(wal);
314      }
315      // add delete op for peer
316      listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldQueueNode));
317
318      LOG.trace("The multi list size is {}", listOfOps.size());
319      ZKUtil.multiOrSequential(zookeeper, listOfOps, false);
320
321      LOG.info("Atomically moved {}/{}'s WALs to {}", sourceServerName, queueId, destServerName);
322      return new Pair<>(newQueueId, logQueue);
323    } catch (NoNodeException | NodeExistsException | NotEmptyException | BadVersionException e) {
324      // Multi call failed; it looks like some other regionserver took away the logs.
325      // These exceptions mean that zk tells us the request can not be execute so it is safe to just
326      // return a null. For other types of exception should be thrown out to notify the upper layer.
327      LOG.info("Claim queue queueId={} from {} to {} failed with {}, someone else took the log?",
328          queueId,sourceServerName, destServerName, e.toString());
329      return null;
330    } catch (KeeperException | InterruptedException e) {
331      throw new ReplicationException("Claim queue queueId=" + queueId + " from " +
332        sourceServerName + " to " + destServerName + " failed", e);
333    }
334  }

399  @Override
400  public Set<String> getAllWALs() throws ReplicationException {
401    try {
402      for (int retry = 0;; retry++) {
403        int v0 = getQueuesZNodeCversion();
404        List<ServerName> rss = getListOfReplicators0();
405        if (rss.isEmpty()) {
406          LOG.debug("Didn't find a RegionServer that replicates, won't prevent deletions.");
407          return Collections.emptySet();
408        }
409        Set<String> wals = new HashSet<>();
410        for (ServerName rs : rss) {
411          for (String queueId : getAllQueues0(rs)) {
412            wals.addAll(getWALsInQueue0(rs, queueId));
413          }
414        }
415        int v1 = getQueuesZNodeCversion();
416        if (v0 == v1) {
417          return wals;
418        }
419        LOG.info("Replication queue node cversion changed from %d to %d, retry = %d",
420            v0, v1, retry);
421      }
422    } catch (KeeperException e) {
423      throw new ReplicationException("Failed to get all wals", e);
424    }
425  }

435  @Override
436  public void addPeerToHFileRefs(String peerId) throws ReplicationException {
437    String peerNode = getHFileRefsPeerNode(peerId);
438    try {
439      if (ZKUtil.checkExists(zookeeper, peerNode) == -1) {
440        LOG.info("Adding peer {} to hfile reference queue.", peerId);
441        ZKUtil.createWithParents(zookeeper, peerNode);
442      }
443    } catch (KeeperException e) {
444      throw new ReplicationException("Failed to add peer " + peerId + " to hfile reference queue.",
445          e);
446    }
447  }

451    String peerNode = getHFileRefsPeerNode(peerId);
452    try {
453      if (ZKUtil.checkExists(zookeeper, peerNode) == -1) {
454          LOG.debug("Peer {} not found in hfile reference queue.", peerNode);

457        ZKUtil.deleteNodeRecursively(zookeeper, peerNode);

459    } catch (KeeperException e) {

465  @Override
466  public void addHFileRefs(String peerId, List<Pair<Path, Path>> pairs)
467      throws ReplicationException {
468    String peerNode = getHFileRefsPeerNode(peerId);
469    LOG.debug("Adding hfile references {} in queue {}", pairs, peerNode);
470    List<ZKUtilOp> listOfOps = pairs.stream().map(p -> p.getSecond().getName())
471        .map(n -> getHFileNode(peerNode, n))
472        .map(f -> ZKUtilOp.createAndFailSilent(f, HConstants.EMPTY_BYTE_ARRAY)).collect(toList());
473      LOG.debug("The multi list size for adding hfile references in zk for node {} is {}",
474          peerNode, listOfOps.size());
475    try {
476      ZKUtil.multiOrSequential(this.zookeeper, listOfOps, true);
477    } catch (KeeperException e) {
478      throw new ReplicationException("Failed to add hfile reference to peer " + peerId, e);
479    }
480  }

538  @Override
539  public Set<String> getAllHFileRefs() throws ReplicationException {
540    try {
541      for (int retry = 0;; retry++) {
542        int v0 = getHFileRefsZNodeCversion();
543        List<String> peers = getAllPeersFromHFileRefsQueue();
544        if (peers.isEmpty()) {
545          LOG.debug("Didn't find any peers with hfile references, won't prevent deletions.");
546          return Collections.emptySet();
547        }
548        Set<String> hfileRefs = new HashSet<>();
549        for (String peer : peers) {
550          hfileRefs.addAll(getReplicableHFiles0(peer));
551        }
552        int v1 = getHFileRefsZNodeCversion();
553        if (v0 == v1) {
554          return hfileRefs;
555        }
556        LOG.debug("Replication hfile references node cversion changed from %d to %d, retry = %d",
557            v0, v1, retry);
558      }
559    } catch (KeeperException e) {
560      throw new ReplicationException("Failed to get all hfile refs", e);
561    }
562  }
