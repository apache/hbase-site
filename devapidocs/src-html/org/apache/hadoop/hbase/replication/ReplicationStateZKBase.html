019package org.apache.hadoop.hbase.replication;
020
021import java.io.ByteArrayOutputStream;
022import java.io.IOException;
023import java.util.List;
024
025import org.apache.hadoop.hbase.shaded.com.google.common.annotations.VisibleForTesting;
026
027import org.apache.hadoop.hbase.shaded.com.google.protobuf.CodedOutputStream;
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.hbase.Abortable;
030import org.apache.hadoop.hbase.classification.InterfaceAudience;
031import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
032import org.apache.hadoop.hbase.shaded.protobuf.generated.ReplicationProtos;
033import org.apache.hadoop.hbase.zookeeper.ZKConfig;
034import org.apache.hadoop.hbase.zookeeper.ZKUtil;
035import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
036import org.apache.zookeeper.KeeperException;
037
038
039/**
040 * This is a base class for maintaining replication state in zookeeper.
041 */
042@InterfaceAudience.Private
043public abstract class ReplicationStateZKBase {
044
045  /**
046   * The name of the znode that contains the replication status of a remote slave (i.e. peer)
047   * cluster.
048   */
049  protected final String peerStateNodeName;
050  /** The name of the base znode that contains all replication state. */
051  protected final String replicationZNode;
052  /** The name of the znode that contains a list of all remote slave (i.e. peer) clusters. */
053  protected final String peersZNode;
054  /** The name of the znode that contains all replication queues */
055  protected final String queuesZNode;
056  /** The name of the znode that contains queues of hfile references to be replicated */
057  protected final String hfileRefsZNode;
058  /** The cluster key of the local cluster */
059  protected final String ourClusterKey;
060  /** The name of the znode that contains tableCFs */
061  protected final String tableCFsNodeName;
062
063  protected final ZooKeeperWatcher zookeeper;
064  protected final Configuration conf;
065  protected final Abortable abortable;
066
067  // Public for testing
068  public static final byte[] ENABLED_ZNODE_BYTES =
069      toByteArray(ReplicationProtos.ReplicationState.State.ENABLED);
070  public static final byte[] DISABLED_ZNODE_BYTES =
071      toByteArray(ReplicationProtos.ReplicationState.State.DISABLED);
072  public static final String ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_KEY =
073      "zookeeper.znode.replication.hfile.refs";
074  public static final String ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_DEFAULT = "hfile-refs";
075
076  public ReplicationStateZKBase(ZooKeeperWatcher zookeeper, Configuration conf,
077      Abortable abortable) {
078    this.zookeeper = zookeeper;
079    this.conf = conf;
080    this.abortable = abortable;
081
082    String replicationZNodeName = conf.get("zookeeper.znode.replication", "replication");
083    String peersZNodeName = conf.get("zookeeper.znode.replication.peers", "peers");
084    String queuesZNodeName = conf.get("zookeeper.znode.replication.rs", "rs");
085    String hfileRefsZNodeName = conf.get(ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_KEY,
086      ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_DEFAULT);
087    this.peerStateNodeName = conf.get("zookeeper.znode.replication.peers.state", "peer-state");
088    this.tableCFsNodeName = conf.get("zookeeper.znode.replication.peers.tableCFs", "tableCFs");
089    this.ourClusterKey = ZKConfig.getZooKeeperClusterKey(this.conf);
090    this.replicationZNode = ZKUtil.joinZNode(this.zookeeper.znodePaths.baseZNode,
091      replicationZNodeName);
092    this.peersZNode = ZKUtil.joinZNode(replicationZNode, peersZNodeName);
093    this.queuesZNode = ZKUtil.joinZNode(replicationZNode, queuesZNodeName);
094    this.hfileRefsZNode = ZKUtil.joinZNode(replicationZNode, hfileRefsZNodeName);
095  }
096
097  public List<String> getListOfReplicators() {
098    List<String> result = null;
099    try {
100      result = ZKUtil.listChildrenNoWatch(this.zookeeper, this.queuesZNode);
101    } catch (KeeperException e) {
102      this.abortable.abort("Failed to get list of replicators", e);
103    }
104    return result;
105  }
106
107  /**
108   * @param state
109   * @return Serialized protobuf of <code>state</code> with pb magic prefix prepended suitable for
110   *         use as content of a peer-state znode under a peer cluster id as in
111   *         /hbase/replication/peers/PEER_ID/peer-state.
112   */
113  protected static byte[] toByteArray(final ReplicationProtos.ReplicationState.State state) {
114    ReplicationProtos.ReplicationState msg =
115        ReplicationProtos.ReplicationState.newBuilder().setState(state).build();
116    // There is no toByteArray on this pb Message?
117    // 32 bytes is default which seems fair enough here.
118    try (ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
119      CodedOutputStream cos = CodedOutputStream.newInstance(baos, 16);
120      msg.writeTo(cos);
121      cos.flush();
122      baos.flush();
123      return ProtobufUtil.prependPBMagic(baos.toByteArray());
124    } catch (IOException e) {
125      throw new RuntimeException(e);
126    }
127  }
128
129  protected boolean peerExists(String id) throws KeeperException {
130    return ZKUtil.checkExists(this.zookeeper, ZKUtil.joinZNode(this.peersZNode, id)) >= 0;
131  }
132
133  /**
134   * Determine if a ZK path points to a peer node.
135   * @param path path to be checked
136   * @return true if the path points to a peer node, otherwise false
137   */
138  protected boolean isPeerPath(String path) {
139    return path.split("/").length == peersZNode.split("/").length + 1;
140  }
141
142  @VisibleForTesting
143  protected String getTableCFsNode(String id) {
144    return ZKUtil.joinZNode(this.peersZNode, ZKUtil.joinZNode(id, this.tableCFsNodeName));
145  }
146
147  @VisibleForTesting
148  protected String getPeerStateNode(String id) {
149    return ZKUtil.joinZNode(this.peersZNode, ZKUtil.joinZNode(id, this.peerStateNodeName));
150  }
151  @VisibleForTesting
152  protected String getPeerNode(String id) {
153    return ZKUtil.joinZNode(this.peersZNode, id);
154  }
155}
