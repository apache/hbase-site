079  @InterfaceAudience.Private
080  public ServerLoad(ServerMetrics metrics) {
081    this.metrics = metrics;
082    this.serverLoad = ServerMetricsBuilder.toServerLoad(metrics);
083    for (RegionMetrics rl : metrics.getRegionMetrics().values()) {
084      stores += rl.getStoreCount();
085      storefiles += rl.getStoreFileCount();
086      storeUncompressedSizeMB += rl.getUncompressedStoreFileSize().get(Size.Unit.MEGABYTE);
087      storefileSizeMB += rl.getStoreFileSize().get(Size.Unit.MEGABYTE);
088      memstoreSizeMB += rl.getMemStoreSize().get(Size.Unit.MEGABYTE);
089      readRequestsCount += rl.getReadRequestCount();
090      cpRequestsCount += rl.getCpRequestCount();
091      filteredReadRequestsCount += rl.getFilteredReadRequestCount();
092      writeRequestsCount += rl.getWriteRequestCount();
093      storefileIndexSizeKB += rl.getStoreFileIndexSize().get(Size.Unit.KILOBYTE);
094      rootIndexSizeKB += rl.getStoreFileRootLevelIndexSize().get(Size.Unit.KILOBYTE);
095      totalStaticIndexSizeKB += rl.getStoreFileUncompressedDataIndexSize().get(Size.Unit.KILOBYTE);
096      totalStaticBloomSizeKB += rl.getBloomFilterSize().get(Size.Unit.KILOBYTE);
097      totalCompactingKVs += rl.getCompactingCellCount();
098      currentCompactedKVs += rl.getCompactedCellCount();
099    }
100  }

492  /**
493   * @see java.lang.Object#toString()
494   */
495  @Override
496  public String toString() {
497    StringBuilder sb = Strings.appendKeyValue(new StringBuilder(), "requestsPerSecond",
498      Double.valueOf(getRequestsPerSecond()));
499    Strings.appendKeyValue(sb, "numberOfOnlineRegions", Integer.valueOf(getNumberOfRegions()));
500    Strings.appendKeyValue(sb, "usedHeapMB", Integer.valueOf(this.getUsedHeapMB()));
501    Strings.appendKeyValue(sb, "maxHeapMB", Integer.valueOf(getMaxHeapMB()));
502    Strings.appendKeyValue(sb, "numberOfStores", Integer.valueOf(this.stores));
503    Strings.appendKeyValue(sb, "numberOfStorefiles", Integer.valueOf(this.storefiles));
504    Strings.appendKeyValue(sb, "storefileUncompressedSizeMB",
505        Integer.valueOf(this.storeUncompressedSizeMB));
506    Strings.appendKeyValue(sb, "storefileSizeMB", Integer.valueOf(this.storefileSizeMB));
507    if (this.storeUncompressedSizeMB != 0) {
508      Strings.appendKeyValue(sb, "compressionRatio", String.format("%.4f",
509          (float) this.storefileSizeMB / (float) this.storeUncompressedSizeMB));
510    }
511    Strings.appendKeyValue(sb, "memstoreSizeMB", Integer.valueOf(this.memstoreSizeMB));
512    Strings.appendKeyValue(sb, "storefileIndexSizeKB",
513        Long.valueOf(this.storefileIndexSizeKB));
514    Strings.appendKeyValue(sb, "readRequestsCount", Long.valueOf(this.readRequestsCount));
515    Strings.appendKeyValue(sb, "cpRequestsCount", Long.valueOf(this.cpRequestsCount));
516    Strings.appendKeyValue(sb, "filteredReadRequestsCount",
517        Long.valueOf(this.filteredReadRequestsCount));
518    Strings.appendKeyValue(sb, "writeRequestsCount", Long.valueOf(this.writeRequestsCount));
519    Strings.appendKeyValue(sb, "rootIndexSizeKB", Integer.valueOf(this.rootIndexSizeKB));
520    Strings.appendKeyValue(sb, "totalStaticIndexSizeKB",
521        Integer.valueOf(this.totalStaticIndexSizeKB));
522    Strings.appendKeyValue(sb, "totalStaticBloomSizeKB",
523        Integer.valueOf(this.totalStaticBloomSizeKB));
524    Strings.appendKeyValue(sb, "totalCompactingKVs", Long.valueOf(this.totalCompactingKVs));
525    Strings.appendKeyValue(sb, "currentCompactedKVs", Long.valueOf(this.currentCompactedKVs));
526    float compactionProgressPct = Float.NaN;
527    if (this.totalCompactingKVs > 0) {
528      compactionProgressPct =
529          Float.valueOf((float) this.currentCompactedKVs / this.totalCompactingKVs);
530    }
531    Strings.appendKeyValue(sb, "compactionProgressPct", compactionProgressPct);
532
533    String[] coprocessorStrings = getRsCoprocessors();
534    if (coprocessorStrings != null) {
535      Strings.appendKeyValue(sb, "coprocessors", Arrays.toString(coprocessorStrings));
536    }
537    return sb.toString();
538  }
