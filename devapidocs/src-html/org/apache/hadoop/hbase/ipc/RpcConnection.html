053/**
054 * Base class for ipc connection.
055 */
056@InterfaceAudience.Private
057abstract class RpcConnection {
058
059  private static final Logger LOG = LoggerFactory.getLogger(RpcConnection.class);
060
061  protected final ConnectionId remoteId;
062
063  protected final boolean useSasl;
064
065  protected final Token<? extends TokenIdentifier> token;
066
067  protected final SecurityInfo securityInfo;
068
069  protected final int reloginMaxBackoff; // max pause before relogin on sasl failure
070
071  protected final Codec codec;
072
073  protected final CompressionCodec compressor;
074
075  protected final MetricsConnection metrics;
076  private final Map<String, byte[]> connectionAttributes;
077
078  protected final HashedWheelTimer timeoutTimer;
079
080  protected final Configuration conf;
081
082  protected static String CRYPTO_AES_ENABLED_KEY = "hbase.rpc.crypto.encryption.aes.enabled";
083
084  protected static boolean CRYPTO_AES_ENABLED_DEFAULT = false;
085
086  // the last time we were picked up from connection pool.
087  protected long lastTouched;
088
089  protected SaslClientAuthenticationProvider provider;
090
091  protected RpcConnection(Configuration conf, HashedWheelTimer timeoutTimer, ConnectionId remoteId,
092    String clusterId, boolean isSecurityEnabled, Codec codec, CompressionCodec compressor,
093    MetricsConnection metrics, Map<String, byte[]> connectionAttributes) throws IOException {
094    this.timeoutTimer = timeoutTimer;
095    this.codec = codec;
096    this.compressor = compressor;
097    this.conf = conf;
098    this.metrics = metrics;
099    this.connectionAttributes = connectionAttributes;
100    User ticket = remoteId.getTicket();
101    this.securityInfo = SecurityInfo.getInfo(remoteId.getServiceName());
102    this.useSasl = isSecurityEnabled;
103
104    // Choose the correct Token and AuthenticationProvider for this client to use
105    SaslClientAuthenticationProviders providers =
106      SaslClientAuthenticationProviders.getInstance(conf);
107    Pair<SaslClientAuthenticationProvider, Token<? extends TokenIdentifier>> pair;
108    if (useSasl && securityInfo != null) {
109      pair = providers.selectProvider(clusterId, ticket);
110      if (pair == null) {
111        if (LOG.isTraceEnabled()) {
112          LOG.trace("Found no valid authentication method from providers={} with tokens={}",
113            providers.toString(), ticket.getTokens());
114        }
115        throw new RuntimeException("Found no valid authentication method from options");
116      }
117    } else if (!useSasl) {
118      // Hack, while SIMPLE doesn't go via SASL.
119      pair = providers.getSimpleProvider();
120    } else {
121      throw new RuntimeException("Could not compute valid client authentication provider");
122    }
123
124    this.provider = pair.getFirst();
125    this.token = pair.getSecond();
126
127    LOG.debug("Using {} authentication for service={}, sasl={}",
128      provider.getSaslAuthMethod().getName(), remoteId.serviceName, useSasl);
129    reloginMaxBackoff = conf.getInt("hbase.security.relogin.maxbackoff", 5000);
130    this.remoteId = remoteId;
131  }
132
133  protected final void scheduleTimeoutTask(final Call call) {
134    if (call.timeout > 0) {
135      call.timeoutTask = timeoutTimer.newTimeout(new TimerTask() {
136
137        @Override
138        public void run(Timeout timeout) throws Exception {
139          call.setTimeout(new CallTimeoutException(call.toShortString() + ", waitTime="
140            + (EnvironmentEdgeManager.currentTime() - call.getStartTime()) + "ms, rpcTimeout="
141            + call.timeout + "ms"));
142          callTimeout(call);
143        }
144      }, call.timeout, TimeUnit.MILLISECONDS);
145    }
146  }
147
148  protected final byte[] getConnectionHeaderPreamble() {
149    // Assemble the preamble up in a buffer first and then send it. Writing individual elements,
150    // they are getting sent across piecemeal according to wireshark and then server is messing
151    // up the reading on occasion (the passed in stream is not buffered yet).
152
153    // Preamble is six bytes -- 'HBas' + VERSION + AUTH_CODE
154    int rpcHeaderLen = HConstants.RPC_HEADER.length;
155    byte[] preamble = new byte[rpcHeaderLen + 2];
156    System.arraycopy(HConstants.RPC_HEADER, 0, preamble, 0, rpcHeaderLen);
157    preamble[rpcHeaderLen] = HConstants.RPC_CURRENT_VERSION;
158    synchronized (this) {
159      preamble[rpcHeaderLen + 1] = provider.getSaslAuthMethod().getCode();
160    }
161    return preamble;
162  }
163
164  protected final ConnectionHeader getConnectionHeader() {
165    final ConnectionHeader.Builder builder = ConnectionHeader.newBuilder();
166    builder.setServiceName(remoteId.getServiceName());
167    final UserInformation userInfoPB = provider.getUserInfo(remoteId.ticket);
168    if (userInfoPB != null) {
169      builder.setUserInfo(userInfoPB);
170    }
171    if (this.codec != null) {
172      builder.setCellBlockCodecClass(this.codec.getClass().getCanonicalName());
173    }
174    if (this.compressor != null) {
175      builder.setCellBlockCompressorClass(this.compressor.getClass().getCanonicalName());
176    }
177    if (connectionAttributes != null && !connectionAttributes.isEmpty()) {
178      HBaseProtos.NameBytesPair.Builder attributeBuilder = HBaseProtos.NameBytesPair.newBuilder();
179      for (Map.Entry<String, byte[]> attribute : connectionAttributes.entrySet()) {
180        attributeBuilder.setName(attribute.getKey());
181        attributeBuilder.setValue(UnsafeByteOperations.unsafeWrap(attribute.getValue()));
182        builder.addAttribute(attributeBuilder.build());
183      }
184    }
185    builder.setVersionInfo(ProtobufUtil.getVersionInfo());
186    boolean isCryptoAESEnable = conf.getBoolean(CRYPTO_AES_ENABLED_KEY, CRYPTO_AES_ENABLED_DEFAULT);
187    // if Crypto AES enable, setup Cipher transformation
188    if (isCryptoAESEnable) {
189      builder.setRpcCryptoCipherTransformation(
190        conf.get("hbase.rpc.crypto.encryption.aes.cipher.transform", "AES/CTR/NoPadding"));
191    }
192    return builder.build();
193  }
194
195  protected final InetSocketAddress getRemoteInetAddress(MetricsConnection metrics)
196    throws UnknownHostException {
197    if (metrics != null) {
198      metrics.incrNsLookups();
199    }
200    InetSocketAddress remoteAddr = Address.toSocketAddress(remoteId.getAddress());
201    if (remoteAddr.isUnresolved()) {
202      if (metrics != null) {
203        metrics.incrNsLookupsFailed();
204      }
205      throw new UnknownHostException(remoteId.getAddress() + " could not be resolved");
206    }
207    return remoteAddr;
208  }
209
210  protected abstract void callTimeout(Call call);
211
212  public ConnectionId remoteId() {
213    return remoteId;
214  }
215
216  public long getLastTouched() {
217    return lastTouched;
218  }
219
220  public void setLastTouched(long lastTouched) {
221    this.lastTouched = lastTouched;
222  }
223
224  /**
225   * Tell the idle connection sweeper whether we could be swept.
226   */
227  public abstract boolean isActive();
228
229  /**
230   * Just close connection. Do not need to remove from connection pool.
231   */
232  public abstract void shutdown();
233
234  public abstract void sendRequest(Call call, HBaseRpcController hrc) throws IOException;
235
236  /**
237   * Does the clean up work after the connection is removed from the connection pool
238   */
239  public abstract void cleanupConnection();
240}
