049/**
050 * Base class for ipc connection.
051 */
052@InterfaceAudience.Private
053abstract class RpcConnection {
054
055  private static final Log LOG = LogFactory.getLog(RpcConnection.class);
056
057  protected final ConnectionId remoteId;
058
059  protected final AuthMethod authMethod;
060
061  protected final boolean useSasl;
062
063  protected final Token<? extends TokenIdentifier> token;
064
065  protected final String serverPrincipal; // server's krb5 principal name
066
067  protected final int reloginMaxBackoff; // max pause before relogin on sasl failure
068
069  protected final Codec codec;
070
071  protected final CompressionCodec compressor;
072
073  protected final HashedWheelTimer timeoutTimer;
074
075  // the last time we were picked up from connection pool.
076  protected long lastTouched;
077
078  protected RpcConnection(Configuration conf, HashedWheelTimer timeoutTimer, ConnectionId remoteId,
079      String clusterId, boolean isSecurityEnabled, Codec codec, CompressionCodec compressor)
080      throws IOException {
081    if (remoteId.getAddress().isUnresolved()) {
082      throw new UnknownHostException("unknown host: " + remoteId.getAddress().getHostName());
083    }
084    this.timeoutTimer = timeoutTimer;
085    this.codec = codec;
086    this.compressor = compressor;
087
088    UserGroupInformation ticket = remoteId.getTicket().getUGI();
089    SecurityInfo securityInfo = SecurityInfo.getInfo(remoteId.getServiceName());
090    this.useSasl = isSecurityEnabled;
091    Token<? extends TokenIdentifier> token = null;
092    String serverPrincipal = null;
093    if (useSasl && securityInfo != null) {
094      AuthenticationProtos.TokenIdentifier.Kind tokenKind = securityInfo.getTokenKind();
095      if (tokenKind != null) {
096        TokenSelector<? extends TokenIdentifier> tokenSelector = AbstractRpcClient.TOKEN_HANDLERS
097            .get(tokenKind);
098        if (tokenSelector != null) {
099          token = tokenSelector.selectToken(new Text(clusterId), ticket.getTokens());
100        } else if (LOG.isDebugEnabled()) {
101          LOG.debug("No token selector found for type " + tokenKind);
102        }
103      }
104      String serverKey = securityInfo.getServerPrincipal();
105      if (serverKey == null) {
106        throw new IOException("Can't obtain server Kerberos config key from SecurityInfo");
107      }
108      serverPrincipal = SecurityUtil.getServerPrincipal(conf.get(serverKey),
109        remoteId.address.getAddress().getCanonicalHostName().toLowerCase());
110      if (LOG.isDebugEnabled()) {
111        LOG.debug("RPC Server Kerberos principal name for service=" + remoteId.getServiceName()
112            + " is " + serverPrincipal);
113      }
114    }
115    this.token = token;
116    this.serverPrincipal = serverPrincipal;
117    if (!useSasl) {
118      authMethod = AuthMethod.SIMPLE;
119    } else if (token != null) {
120      authMethod = AuthMethod.DIGEST;
121    } else {
122      authMethod = AuthMethod.KERBEROS;
123    }
124
125    if (LOG.isDebugEnabled()) {
126      LOG.debug("Use " + authMethod + " authentication for service " + remoteId.serviceName
127          + ", sasl=" + useSasl);
128    }
129    reloginMaxBackoff = conf.getInt("hbase.security.relogin.maxbackoff", 5000);
130    this.remoteId = remoteId;
131  }
132
133  private UserInformation getUserInfo(UserGroupInformation ugi) {
134    if (ugi == null || authMethod == AuthMethod.DIGEST) {
135      // Don't send user for token auth
136      return null;
137    }
138    UserInformation.Builder userInfoPB = UserInformation.newBuilder();
139    if (authMethod == AuthMethod.KERBEROS) {
140      // Send effective user for Kerberos auth
141      userInfoPB.setEffectiveUser(ugi.getUserName());
142    } else if (authMethod == AuthMethod.SIMPLE) {
143      // Send both effective user and real user for simple auth
144      userInfoPB.setEffectiveUser(ugi.getUserName());
145      if (ugi.getRealUser() != null) {
146        userInfoPB.setRealUser(ugi.getRealUser().getUserName());
147      }
148    }
149    return userInfoPB.build();
150  }
151
152  protected UserGroupInformation getUGI() {
153    UserGroupInformation ticket = remoteId.getTicket().getUGI();
154    if (authMethod == AuthMethod.KERBEROS) {
155      if (ticket != null && ticket.getRealUser() != null) {
156        ticket = ticket.getRealUser();
157      }
158    }
159    return ticket;
160  }
161
162  protected boolean shouldAuthenticateOverKrb() throws IOException {
163    UserGroupInformation loginUser = UserGroupInformation.getLoginUser();
164    UserGroupInformation currentUser = UserGroupInformation.getCurrentUser();
165    UserGroupInformation realUser = currentUser.getRealUser();
166    return authMethod == AuthMethod.KERBEROS && loginUser != null &&
167    // Make sure user logged in using Kerberos either keytab or TGT
168        loginUser.hasKerberosCredentials() &&
169        // relogin only in case it is the login user (e.g. JT)
170        // or superuser (like oozie).
171        (loginUser.equals(currentUser) || loginUser.equals(realUser));
172  }
173
174  protected void relogin() throws IOException {
175    if (UserGroupInformation.isLoginKeytabBased()) {
176      UserGroupInformation.getLoginUser().reloginFromKeytab();
177    } else {
178      UserGroupInformation.getLoginUser().reloginFromTicketCache();
179    }
180  }
181
182  protected void scheduleTimeoutTask(final Call call) {
183    if (call.timeout > 0) {
184      call.timeoutTask = timeoutTimer.newTimeout(new TimerTask() {
185
186        @Override
187        public void run(Timeout timeout) throws Exception {
188          call.setTimeout(new CallTimeoutException("Call id=" + call.id + ", waitTime="
189              + (EnvironmentEdgeManager.currentTime() - call.getStartTime()) + ", rpcTimetout="
190              + call.timeout));
191          callTimeout(call);
192        }
193      }, call.timeout, TimeUnit.MILLISECONDS);
194    }
195  }
196
197  protected byte[] getConnectionHeaderPreamble() {
198    // Assemble the preamble up in a buffer first and then send it. Writing individual elements,
199    // they are getting sent across piecemeal according to wireshark and then server is messing
200    // up the reading on occasion (the passed in stream is not buffered yet).
201
202    // Preamble is six bytes -- 'HBas' + VERSION + AUTH_CODE
203    int rpcHeaderLen = HConstants.RPC_HEADER.length;
204    byte[] preamble = new byte[rpcHeaderLen + 2];
205    System.arraycopy(HConstants.RPC_HEADER, 0, preamble, 0, rpcHeaderLen);
206    preamble[rpcHeaderLen] = HConstants.RPC_CURRENT_VERSION;
207    synchronized (this) {
208      preamble[rpcHeaderLen + 1] = authMethod.code;
209    }
210    return preamble;
211  }
212
213  protected ConnectionHeader getConnectionHeader() {
214    ConnectionHeader.Builder builder = ConnectionHeader.newBuilder();
215    builder.setServiceName(remoteId.getServiceName());
216    UserInformation userInfoPB;
217    if ((userInfoPB = getUserInfo(remoteId.ticket.getUGI())) != null) {
218      builder.setUserInfo(userInfoPB);
219    }
220    if (this.codec != null) {
221      builder.setCellBlockCodecClass(this.codec.getClass().getCanonicalName());
222    }
223    if (this.compressor != null) {
224      builder.setCellBlockCompressorClass(this.compressor.getClass().getCanonicalName());
225    }
226    builder.setVersionInfo(ProtobufUtil.getVersionInfo());
227    return builder.build();
228  }
229
230  protected abstract void callTimeout(Call call);
231
232  public ConnectionId remoteId() {
233    return remoteId;
234  }
235
236  public long getLastTouched() {
237    return lastTouched;
238  }
239
240  public void setLastTouched(long lastTouched) {
241    this.lastTouched = lastTouched;
242  }
243
244  /**
245   * Tell the idle connection sweeper whether we could be swept.
246   */
247  public abstract boolean isActive();
248
249  /**
250   * Just close connection. Do not need to remove from connection pool.
251   */
252  public abstract void shutdown();
253
254  public abstract void sendRequest(Call call, HBaseRpcController hrc) throws IOException;
255
256  /**
257   * Does the clean up work after the connection is removed from the connection pool
258   */
259  public abstract void cleanupConnection();
260}
