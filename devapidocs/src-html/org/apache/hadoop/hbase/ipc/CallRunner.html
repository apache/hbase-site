019import java.net.InetSocketAddress;
020import java.nio.channels.ClosedChannelException;
021
022import org.apache.commons.logging.Log;
023import org.apache.commons.logging.LogFactory;
024import org.apache.hadoop.hbase.CallDroppedException;
025import org.apache.hadoop.hbase.CellScanner;
026import org.apache.hadoop.hbase.HBaseInterfaceAudience;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.hbase.classification.InterfaceStability;
029import org.apache.hadoop.hbase.ipc.RpcServer.Call;
030import org.apache.hadoop.hbase.monitoring.MonitoredRPCHandler;
031import org.apache.hadoop.hbase.util.Pair;
032import org.apache.hadoop.security.UserGroupInformation;
033import org.apache.hadoop.util.StringUtils;
034import org.apache.htrace.Trace;
035import org.apache.htrace.TraceScope;
036
037import org.apache.hadoop.hbase.shaded.com.google.protobuf.Message;
038
039/**
040 * The request processing logic, which is usually executed in thread pools provided by an
041 * {@link RpcScheduler}.  Call {@link #run()} to actually execute the contained
042 * RpcServer.Call
043 */
044@InterfaceAudience.LimitedPrivate({HBaseInterfaceAudience.COPROC, HBaseInterfaceAudience.PHOENIX})
045@InterfaceStability.Evolving
046public class CallRunner {
047  private static final Log LOG = LogFactory.getLog(CallRunner.class);
048
049  private static final CallDroppedException CALL_DROPPED_EXCEPTION
050    = new CallDroppedException();
051
052  private Call call;
053  private RpcServerInterface rpcServer;
054  private MonitoredRPCHandler status;
055  private volatile boolean sucessful;
056
057  /**
058   * On construction, adds the size of this call to the running count of outstanding call sizes.
059   * Presumption is that we are put on a queue while we wait on an executor to run us.  During this
060   * time we occupy heap.
061   */
062  // The constructor is shutdown so only RpcServer in this class can make one of these.
063  CallRunner(final RpcServerInterface rpcServer, final Call call) {
064    this.call = call;
065    this.rpcServer = rpcServer;
066    // Add size of the call to queue size.
067    if (call != null && rpcServer != null) {
068      this.rpcServer.addCallSize(call.getSize());
069    }
070  }
071
072  public Call getCall() {
073    return call;
074  }
075
076  public void setStatus(MonitoredRPCHandler status) {
077    this.status = status;
078  }
079
080  /**
081   * Cleanup after ourselves... let go of references.
082   */
083  private void cleanup() {
084    this.call = null;
085    this.rpcServer = null;
086  }
087
088  public void run() {
089    try {
090      if (!call.connection.channel.isOpen()) {
091        if (RpcServer.LOG.isDebugEnabled()) {
092          RpcServer.LOG.debug(Thread.currentThread().getName() + ": skipped " + call);
093        }
094        return;
095      }
096      call.startTime = System.currentTimeMillis();
097      if (call.startTime > call.deadline) {
098        RpcServer.LOG.info("Dropping timed out call: " + call);
099        return;
100      }
101      this.status.setStatus("Setting up call");
102      this.status.setConnection(call.connection.getHostAddress(), call.connection.getRemotePort());
103      if (RpcServer.LOG.isTraceEnabled()) {
104        UserGroupInformation remoteUser = call.connection.ugi;
105        RpcServer.LOG.trace(call.toShortString() + " executing as " +
106            ((remoteUser == null) ? "NULL principal" : remoteUser.getUserName()));
107      }
108      Throwable errorThrowable = null;
109      String error = null;
110      Pair<Message, CellScanner> resultPair = null;
111      RpcServer.CurCall.set(call);
112      TraceScope traceScope = null;
113      try {
114        if (!this.rpcServer.isStarted()) {
115          InetSocketAddress address = rpcServer.getListenerAddress();
116          throw new ServerNotRunningYetException("Server " +
117              (address != null ? address : "(channel closed)") + " is not running yet");
118        }
119        if (call.tinfo != null) {
120          traceScope = Trace.startSpan(call.toTraceString(), call.tinfo);
121        }
122        // make the call
123        resultPair = this.rpcServer.call(call.service, call.md, call.param, call.cellScanner,
124          call.timestamp, this.status, call.startTime, call.timeout);
125      } catch (Throwable e) {
126        RpcServer.LOG.debug(Thread.currentThread().getName() + ": " + call.toShortString(), e);
127        errorThrowable = e;
128        error = StringUtils.stringifyException(e);
129        if (e instanceof Error) {
130          throw (Error)e;
131        }
132      } finally {
133        if (traceScope != null) {
134          traceScope.close();
135        }
136        RpcServer.CurCall.set(null);
137        if (resultPair != null) {
138          this.rpcServer.addCallSize(call.getSize() * -1);
139          sucessful = true;
140        }
141      }
142      // Set the response
143      Message param = resultPair != null ? resultPair.getFirst() : null;
144      CellScanner cells = resultPair != null ? resultPair.getSecond() : null;
145      call.setResponse(param, cells, errorThrowable, error);
146      call.sendResponseIfReady();
147      this.status.markComplete("Sent response");
148      this.status.pause("Waiting for a call");
149    } catch (OutOfMemoryError e) {
150      if (this.rpcServer.getErrorHandler() != null) {
151        if (this.rpcServer.getErrorHandler().checkOOME(e)) {
152          RpcServer.LOG.info(Thread.currentThread().getName() + ": exiting on OutOfMemoryError");
153          return;
154        }
155      } else {
156        // rethrow if no handler
157        throw e;
158      }
159    } catch (ClosedChannelException cce) {
160      InetSocketAddress address = rpcServer.getListenerAddress();
161      RpcServer.LOG.warn(Thread.currentThread().getName() + ": caught a ClosedChannelException, " +
162          "this means that the server " + (address != null ? address : "(channel closed)") +
163          " was processing a request but the client went away. The error message was: " +
164          cce.getMessage());
165    } catch (Exception e) {
166      RpcServer.LOG.warn(Thread.currentThread().getName()
167          + ": caught: " + StringUtils.stringifyException(e));
168    } finally {
169      if (!sucessful) {
170        this.rpcServer.addCallSize(call.getSize() * -1);
171      }
172      cleanup();
173    }
174  }
175
176  /**
177   * When we want to drop this call because of server is overloaded.
178   */
179  public void drop() {
180    try {
181      if (!call.connection.channel.isOpen()) {
182        if (RpcServer.LOG.isDebugEnabled()) {
183          RpcServer.LOG.debug(Thread.currentThread().getName() + ": skipped " + call);
184        }
185        return;
186      }
187
188      // Set the response
189      InetSocketAddress address = rpcServer.getListenerAddress();
190      call.setResponse(null, null, CALL_DROPPED_EXCEPTION, "Call dropped, server "
191        + (address != null ? address : "(channel closed)") + " is overloaded, please retry.");
192      call.sendResponseIfReady();
193    } catch (ClosedChannelException cce) {
194      InetSocketAddress address = rpcServer.getListenerAddress();
195      RpcServer.LOG.warn(Thread.currentThread().getName() + ": caught a ClosedChannelException, " +
196        "this means that the server " + (address != null ? address : "(channel closed)") +
197        " was processing a request but the client went away. The error message was: " +
198        cce.getMessage());
199    } catch (Exception e) {
200      RpcServer.LOG.warn(Thread.currentThread().getName()
201        + ": caught: " + StringUtils.stringifyException(e));
202    } finally {
203      if (!sucessful) {
204        this.rpcServer.addCallSize(call.getSize() * -1);
205      }
206      cleanup();
207    }
208  }
209}
