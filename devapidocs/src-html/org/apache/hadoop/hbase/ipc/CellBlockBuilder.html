045/**
046 * Helper class for building cell block.
047 */
048@InterfaceAudience.Private
049public class CellBlockBuilder {
050
051  // LOG is being used in TestCellBlockBuilder
052  static final Log LOG = LogFactory.getLog(CellBlockBuilder.class);
053
054  private final Configuration conf;
055  /**
056   * How much we think the decompressor will expand the original compressed content.
057   */
058  private final int cellBlockDecompressionMultiplier;
059
060  private final int cellBlockBuildingInitialBufferSize;
061
062  public CellBlockBuilder(final Configuration conf) {
063    this.conf = conf;
064    this.cellBlockDecompressionMultiplier = conf
065        .getInt("hbase.ipc.cellblock.decompression.buffersize.multiplier", 3);
066
067    // Guess that 16k is a good size for rpc buffer. Could go bigger. See the TODO below in
068    // #buildCellBlock.
069    this.cellBlockBuildingInitialBufferSize = ClassSize
070        .align(conf.getInt("hbase.ipc.cellblock.building.initial.buffersize", 16 * 1024));
071  }
072
073  /**
074   * Puts CellScanner Cells into a cell block using passed in <code>codec</code> and/or
075   * <code>compressor</code>.
076   * @param codec to use for encoding
077   * @param compressor to use for encoding
078   * @param cellScanner to encode
079   * @return Null or byte buffer filled with a cellblock filled with passed-in Cells encoded using
080   *         passed in <code>codec</code> and/or <code>compressor</code>; the returned buffer has
081   *         been flipped and is ready for reading. Use limit to find total size.
082   * @throws IOException if encoding the cells fail
083   */
084  public ByteBuffer buildCellBlock(final Codec codec, final CompressionCodec compressor,
085      final CellScanner cellScanner) throws IOException {
086    if (cellScanner == null) {
087      return null;
088    }
089    if (codec == null) {
090      throw new CellScannerButNoCodecException();
091    }
092    int bufferSize = this.cellBlockBuildingInitialBufferSize;
093    ByteBufferOutputStream baos = new ByteBufferOutputStream(bufferSize);
094    encodeCellsTo(baos, cellScanner, codec, compressor);
095    if (LOG.isTraceEnabled()) {
096      if (bufferSize < baos.size()) {
097        LOG.trace("Buffer grew from initial bufferSize=" + bufferSize + " to " + baos.size()
098            + "; up hbase.ipc.cellblock.building.initial.buffersize?");
099      }
100    }
101    ByteBuffer bb = baos.getByteBuffer();
102    // If no cells, don't mess around. Just return null (could be a bunch of existence checking
103    // gets or something -- stuff that does not return a cell).
104    if (!bb.hasRemaining()) return null;
105    return bb;
106  }
107
108  private void encodeCellsTo(ByteBufferOutputStream bbos, CellScanner cellScanner, Codec codec,
109      CompressionCodec compressor) throws IOException {
110    OutputStream os = bbos;
111    Compressor poolCompressor = null;
112    try {
113      if (compressor != null) {
114        if (compressor instanceof Configurable) {
115          ((Configurable) compressor).setConf(this.conf);
116        }
117        poolCompressor = CodecPool.getCompressor(compressor);
118        os = compressor.createOutputStream(os, poolCompressor);
119      }
120      Codec.Encoder encoder = codec.getEncoder(os);
121      while (cellScanner.advance()) {
122        encoder.write(cellScanner.current());
123      }
124      encoder.flush();
125    } catch (BufferOverflowException e) {
126      throw new DoNotRetryIOException(e);
127    } finally {
128      os.close();
129      if (poolCompressor != null) {
130        CodecPool.returnCompressor(poolCompressor);
131      }
132    }
133  }
134
135  /**
136   * Puts CellScanner Cells into a cell block using passed in <code>codec</code> and/or
137   * <code>compressor</code>.
138   * @param codec to use for encoding
139   * @param compressor to use for encoding
140   * @param cellScanner to encode
141   * @param pool Pool of ByteBuffers to make use of.
142   * @return Null or byte buffer filled with a cellblock filled with passed-in Cells encoded using
143   *         passed in <code>codec</code> and/or <code>compressor</code>; the returned buffer has
144   *         been flipped and is ready for reading. Use limit to find total size. If
145   *         <code>pool</code> was not null, then this returned ByteBuffer came from there and
146   *         should be returned to the pool when done.
147   * @throws IOException if encoding the cells fail
148   */
149  public ByteBufferListOutputStream buildCellBlockStream(Codec codec, CompressionCodec compressor,
150      CellScanner cellScanner, ByteBufferPool pool) throws IOException {
151    if (cellScanner == null) {
152      return null;
153    }
154    if (codec == null) {
155      throw new CellScannerButNoCodecException();
156    }
157    assert pool != null;
158    ByteBufferListOutputStream bbos = new ByteBufferListOutputStream(pool);
159    encodeCellsTo(bbos, cellScanner, codec, compressor);
160    if (bbos.size() == 0) {
161      bbos.releaseResources();
162      return null;
163    }
164    return bbos;
165  }
166
167  /**
168   * @param codec to use for cellblock
169   * @param cellBlock to encode
170   * @return CellScanner to work against the content of <code>cellBlock</code>
171   * @throws IOException if encoding fails
172   */
173  public CellScanner createCellScanner(final Codec codec, final CompressionCodec compressor,
174      final byte[] cellBlock) throws IOException {
175    // Use this method from Client side to create the CellScanner
176    ByteBuffer cellBlockBuf = ByteBuffer.wrap(cellBlock);
177    if (compressor != null) {
178      cellBlockBuf = decompress(compressor, cellBlockBuf);
179    }
180    // Not making the Decoder over the ByteBuffer purposefully. The Decoder over the BB will
181    // make Cells directly over the passed BB. This method is called at client side and we don't
182    // want the Cells to share the same byte[] where the RPC response is being read. Caching of any
183    // of the Cells at user's app level will make it not possible to GC the response byte[]
184    return codec.getDecoder(new ByteBufferInputStream(cellBlockBuf));
185  }
186
187  /**
188   * @param codec to use for cellblock
189   * @param cellBlock ByteBuffer containing the cells written by the Codec. The buffer should be
190   *          position()'ed at the start of the cell block and limit()'ed at the end.
191   * @return CellScanner to work against the content of <code>cellBlock</code>. All cells created
192   *         out of the CellScanner will share the same ByteBuffer being passed.
193   * @throws IOException if cell encoding fails
194   */
195  public CellScanner createCellScannerReusingBuffers(final Codec codec,
196      final CompressionCodec compressor, ByteBuffer cellBlock) throws IOException {
197    // Use this method from HRS to create the CellScanner
198    // If compressed, decompress it first before passing it on else we will leak compression
199    // resources if the stream is not closed properly after we let it out.
200    if (compressor != null) {
201      cellBlock = decompress(compressor, cellBlock);
202    }
203    return codec.getDecoder(cellBlock);
204  }
205
206  private ByteBuffer decompress(CompressionCodec compressor, ByteBuffer cellBlock)
207      throws IOException {
208    // GZIPCodec fails w/ NPE if no configuration.
209    if (compressor instanceof Configurable) {
210      ((Configurable) compressor).setConf(this.conf);
211    }
212    Decompressor poolDecompressor = CodecPool.getDecompressor(compressor);
213    CompressionInputStream cis = compressor.createInputStream(new ByteBufferInputStream(cellBlock),
214      poolDecompressor);
215    ByteBufferOutputStream bbos;
216    try {
217      // TODO: This is ugly. The buffer will be resized on us if we guess wrong.
218      // TODO: Reuse buffers.
219      bbos = new ByteBufferOutputStream(
220          cellBlock.remaining() * this.cellBlockDecompressionMultiplier);
221      IOUtils.copy(cis, bbos);
222      bbos.close();
223      cellBlock = bbos.getByteBuffer();
224    } finally {
225      CodecPool.returnDecompressor(poolDecompressor);
226    }
227    return cellBlock;
228  }
