045/**
046 * RPC Executor that uses different queues for reads and writes.
047 * With the options to use different queues/executors for gets and scans.
048 * Each handler has its own queue and there is no stealing.
049 */
050@InterfaceAudience.LimitedPrivate({HBaseInterfaceAudience.COPROC, HBaseInterfaceAudience.PHOENIX})
051@InterfaceStability.Evolving
052public class RWQueueRpcExecutor extends RpcExecutor {
053  private static final Log LOG = LogFactory.getLog(RWQueueRpcExecutor.class);
054
055  public static final String CALL_QUEUE_READ_SHARE_CONF_KEY =
056      "hbase.ipc.server.callqueue.read.ratio";
057  public static final String CALL_QUEUE_SCAN_SHARE_CONF_KEY =
058      "hbase.ipc.server.callqueue.scan.ratio";
059
060  private final QueueBalancer writeBalancer;
061  private final QueueBalancer readBalancer;
062  private final QueueBalancer scanBalancer;
063  private final int writeHandlersCount;
064  private final int readHandlersCount;
065  private final int scanHandlersCount;
066  private final int numWriteQueues;
067  private final int numReadQueues;
068  private final int numScanQueues;
069
070  public RWQueueRpcExecutor(final String name, final int handlerCount, final int maxQueueLength,
071      final PriorityFunction priority, final Configuration conf, final Abortable abortable) {
072    super(name, handlerCount, maxQueueLength, priority, conf, abortable);
073
074    float callqReadShare = conf.getFloat(CALL_QUEUE_READ_SHARE_CONF_KEY, 0);
075    float callqScanShare = conf.getFloat(CALL_QUEUE_SCAN_SHARE_CONF_KEY, 0);
076
077    numWriteQueues = calcNumWriters(this.numCallQueues, callqReadShare);
078    writeHandlersCount = Math.max(numWriteQueues, calcNumWriters(handlerCount, callqReadShare));
079
080    int readQueues = calcNumReaders(this.numCallQueues, callqReadShare);
081    int readHandlers = Math.max(readQueues, calcNumReaders(handlerCount, callqReadShare));
082
083    int scanQueues = Math.max(0, (int)Math.floor(readQueues * callqScanShare));
084    int scanHandlers = Math.max(0, (int)Math.floor(readHandlers * callqScanShare));
085
086    if ((readQueues - scanQueues) > 0) {
087      readQueues -= scanQueues;
088      readHandlers -= scanHandlers;
089    } else {
090      scanQueues = 0;
091      scanHandlers = 0;
092    }
093
094    numReadQueues = readQueues;
095    readHandlersCount = readHandlers;
096    numScanQueues = scanQueues;
097    scanHandlersCount = scanHandlers;
098
099    this.writeBalancer = getBalancer(numWriteQueues);
100    this.readBalancer = getBalancer(numReadQueues);
101    this.scanBalancer = numScanQueues > 0 ? getBalancer(numScanQueues) : null;
102
103    initializeQueues(numWriteQueues);
104    initializeQueues(numReadQueues);
105    initializeQueues(numScanQueues);
106
107    LOG.info(getName() + " writeQueues=" + numWriteQueues + " writeHandlers=" + writeHandlersCount
108      + " readQueues=" + numReadQueues + " readHandlers=" + readHandlersCount + " scanQueues="
109      + numScanQueues + " scanHandlers=" + scanHandlersCount);
110  }
111
112  @Override
113  protected int computeNumCallQueues(final int handlerCount, final float callQueuesHandlersFactor) {
114    // at least 1 read queue and 1 write queue
115    return Math.max(2, (int) Math.round(handlerCount * callQueuesHandlersFactor));
116  }
117
118  @Override
119  protected void startHandlers(final int port) {
120    startHandlers(".write", writeHandlersCount, queues, 0, numWriteQueues, port);
121    startHandlers(".read", readHandlersCount, queues, numWriteQueues, numReadQueues, port);
122    if (numScanQueues > 0) {
123      startHandlers(".scan", scanHandlersCount, queues, numWriteQueues + numReadQueues,
124        numScanQueues, port);
125    }
126  }
127
128  @Override
129  public boolean dispatch(final CallRunner callTask) throws InterruptedException {
130    RpcServer.Call call = callTask.getCall();
131    int queueIndex;
132    if (isWriteRequest(call.getHeader(), call.param)) {
133      queueIndex = writeBalancer.getNextQueue();
134    } else if (numScanQueues > 0 && isScanRequest(call.getHeader(), call.param)) {
135      queueIndex = numWriteQueues + numReadQueues + scanBalancer.getNextQueue();
136    } else {
137      queueIndex = numWriteQueues + readBalancer.getNextQueue();
138    }
139
140    BlockingQueue<CallRunner> queue = queues.get(queueIndex);
141    if (queue.size() >= currentQueueLimit) {
142      return false;
143    }
144    return queue.offer(callTask);
145  }
146
147  private boolean isWriteRequest(final RequestHeader header, final Message param) {
148    // TODO: Is there a better way to do this?
149    if (param instanceof MultiRequest) {
150      MultiRequest multi = (MultiRequest)param;
151      for (RegionAction regionAction : multi.getRegionActionList()) {
152        for (Action action: regionAction.getActionList()) {
153          if (action.hasMutation()) {
154            return true;
155          }
156        }
157      }
158    }
159    if (param instanceof MutateRequest) {
160      return true;
161    }
162    // Below here are methods for master. It's a pretty brittle version of this.
163    // Not sure that master actually needs a read/write queue since 90% of requests to
164    // master are writing to status or changing the meta table.
165    // All other read requests are admin generated and can be processed whenever.
166    // However changing that would require a pretty drastic change and should be done for
167    // the next major release and not as a fix for HBASE-14239
168    if (param instanceof RegionServerStatusProtos.ReportRegionStateTransitionRequest) {
169      return true;
170    }
171    if (param instanceof RegionServerStatusProtos.RegionServerStartupRequest) {
172      return true;
173    }
174    if (param instanceof RegionServerStatusProtos.RegionServerReportRequest) {
175      return true;
176    }
177    return false;
178  }
179
180  private boolean isScanRequest(final RequestHeader header, final Message param) {
181    if (param instanceof ScanRequest) {
182      // The first scan request will be executed as a "short read"
183      ScanRequest request = (ScanRequest)param;
184      return request.hasScannerId();
185    }
186    return false;
187  }
188
189  /*
190   * Calculate the number of writers based on the "total count" and the read share.
191   * You'll get at least one writer.
192   */
193  private static int calcNumWriters(final int count, final float readShare) {
194    return Math.max(1, count - Math.max(1, (int)Math.round(count * readShare)));
195  }
196
197  /*
198   * Calculate the number of readers based on the "total count" and the read share.
199   * You'll get at least one reader.
200   */
201  private static int calcNumReaders(final int count, final float readShare) {
202    return count - calcNumWriters(count, readShare);
203  }
