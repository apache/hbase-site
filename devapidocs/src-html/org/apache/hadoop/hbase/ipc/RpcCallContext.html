018package org.apache.hadoop.hbase.ipc;
019
020import java.net.InetAddress;
021
022import org.apache.hadoop.hbase.classification.InterfaceAudience;
023import org.apache.hadoop.hbase.shaded.protobuf.generated.HBaseProtos.VersionInfo;
024import org.apache.hadoop.hbase.security.User;
025
026/**
027 * Interface of all necessary to carry out a RPC service invocation on the server. This interface
028 * focus on the information needed or obtained during the actual execution of the service method.
029 */
030@InterfaceAudience.Private
031public interface RpcCallContext {
032  /**
033   * Check if the caller who made this IPC call has disconnected.
034   * If called from outside the context of IPC, this does nothing.
035   * @return &lt; 0 if the caller is still connected. The time in ms
036   *  since the disconnection otherwise
037   */
038  long disconnectSince();
039
040  /**
041   * If the client connected and specified a codec to use, then we will use this codec making
042   * cellblocks to return.  If the client did not specify a codec, we assume it does not support
043   * cellblocks and will return all content protobuf'd (though it makes our serving slower).
044   * We need to ask this question per call because a server could be hosting both clients that
045   * support cellblocks while fielding requests from clients that do not.
046   * @return True if the client supports cellblocks, else return all content in pb
047   */
048  boolean isClientCellBlockSupported();
049
050  /**
051   * Returns the user credentials associated with the current RPC request or
052   * <code>null</code> if no credentials were provided.
053   * @return A User
054   */
055  User getRequestUser();
056
057  /**
058   * @return Current request's user name or null if none ongoing.
059   */
060  String getRequestUserName();
061
062  /**
063   * @return Address of remote client in this call
064   */
065  InetAddress getRemoteAddress();
066
067  /**
068   * @return the client version info, or null if the information is not present
069   */
070  VersionInfo getClientVersionInfo();
071
072  /**
073   * Sets a callback which has to be executed at the end of this RPC call. Such a callback is an
074   * optional one for any Rpc call.
075   *
076   * @param callback
077   */
078  void setCallBack(RpcCallback callback);
079
080  boolean isRetryImmediatelySupported();
081
082  /**
083   * The size of response cells that have been accumulated so far.
084   * This along with the corresponding increment call is used to ensure that multi's or
085   * scans dont get too excessively large
086   */
087  long getResponseCellSize();
088
089  /**
090   * Add on the given amount to the retained cell size.
091   *
092   * This is not thread safe and not synchronized at all. If this is used by more than one thread
093   * then everything will break. Since this is called for every row synchronization would be too
094   * onerous.
095   */
096  void incrementResponseCellSize(long cellSize);
097
098  long getResponseBlockSize();
099
100  void incrementResponseBlockSize(long blockSize);
101
102  long getResponseExceptionSize();
103  void incrementResponseExceptionSize(long exceptionSize);
104}
