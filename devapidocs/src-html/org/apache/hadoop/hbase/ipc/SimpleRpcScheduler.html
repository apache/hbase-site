020import org.apache.hadoop.conf.Configuration;
021import org.apache.hadoop.hbase.Abortable;
022import org.apache.hadoop.hbase.HBaseInterfaceAudience;
023import org.apache.hadoop.hbase.HConstants;

025import org.apache.yetus.audience.InterfaceStability;

028/**
029 * The default scheduler. Configurable. Maintains isolated handler pools for general ('default'),
030 * high-priority ('priority'), and replication ('replication') requests. Default behavior is to
031 * balance the requests across handlers. Add configs to enable balancing by read vs writes, etc.
032 * See below article for explanation of options.
033 * @see <a href="http://blog.cloudera.com/blog/2014/12/new-in-cdh-5-2-improvements-for-running-multiple-workloads-on-a-single-hbase-cluster/">Overview on Request Queuing</a>
034 */
035@InterfaceAudience.LimitedPrivate({HBaseInterfaceAudience.COPROC, HBaseInterfaceAudience.PHOENIX})
036@InterfaceStability.Evolving
037public class SimpleRpcScheduler extends RpcScheduler implements ConfigurationObserver {
038  private int port;
039  private final PriorityFunction priority;
040  private final RpcExecutor callExecutor;
041  private final RpcExecutor priorityExecutor;
042  private final RpcExecutor replicationExecutor;
043
044  /** What level a high priority call is at. */
045  private final int highPriorityLevel;
046
047  private Abortable abortable = null;
048
049  /**
050   * @param conf
051   * @param handlerCount the number of handler threads that will be used to process calls
052   * @param priorityHandlerCount How many threads for priority handling.
053   * @param replicationHandlerCount How many threads for replication handling.
054   * @param highPriorityLevel
055   * @param priority Function to extract request priority.
056   */
057  public SimpleRpcScheduler(
058      Configuration conf,
059      int handlerCount,
060      int priorityHandlerCount,
061      int replicationHandlerCount,
062      PriorityFunction priority,
063      Abortable server,
064      int highPriorityLevel) {
065
066    int maxQueueLength = conf.getInt(RpcScheduler.IPC_SERVER_MAX_CALLQUEUE_LENGTH,
067        handlerCount * RpcServer.DEFAULT_MAX_CALLQUEUE_LENGTH_PER_HANDLER);
068    int maxPriorityQueueLength = conf.getInt(RpcScheduler.IPC_SERVER_PRIORITY_MAX_CALLQUEUE_LENGTH,
069      priorityHandlerCount * RpcServer.DEFAULT_MAX_CALLQUEUE_LENGTH_PER_HANDLER);
070    int maxReplicationQueueLength =
071        conf.getInt(RpcScheduler.IPC_SERVER_REPLICATION_MAX_CALLQUEUE_LENGTH,
072          replicationHandlerCount * RpcServer.DEFAULT_MAX_CALLQUEUE_LENGTH_PER_HANDLER);
073
074    this.priority = priority;
075    this.highPriorityLevel = highPriorityLevel;
076    this.abortable = server;
077
078    String callQueueType = conf.get(RpcExecutor.CALL_QUEUE_TYPE_CONF_KEY,
079      RpcExecutor.CALL_QUEUE_TYPE_CONF_DEFAULT);
080    float callqReadShare = conf.getFloat(RWQueueRpcExecutor.CALL_QUEUE_READ_SHARE_CONF_KEY, 0);
081
082    if (callqReadShare > 0) {
083      // at least 1 read handler and 1 write handler
084      callExecutor = new RWQueueRpcExecutor("default.RWQ", Math.max(2, handlerCount),
085        maxQueueLength, priority, conf, server);
086    } else {
087      if (RpcExecutor.isFifoQueueType(callQueueType) || RpcExecutor.isCodelQueueType(callQueueType)) {
088        callExecutor = new FastPathBalancedQueueRpcExecutor("default.FPBQ", handlerCount,
089            maxQueueLength, priority, conf, server);
090      } else {
091        callExecutor = new BalancedQueueRpcExecutor("default.BQ", handlerCount, maxQueueLength,
092            priority, conf, server);
093      }
094    }
095
096    // Create 2 queues to help priorityExecutor be more scalable.
097    this.priorityExecutor = priorityHandlerCount > 0 ? new FastPathBalancedQueueRpcExecutor(
098        "priority.FPBQ", priorityHandlerCount, RpcExecutor.CALL_QUEUE_TYPE_FIFO_CONF_VALUE,
099        maxPriorityQueueLength, priority, conf, abortable) : null;
100    this.replicationExecutor =
101        replicationHandlerCount > 0
102            ? new FastPathBalancedQueueRpcExecutor("replication.FPBQ", replicationHandlerCount,
103                RpcExecutor.CALL_QUEUE_TYPE_FIFO_CONF_VALUE, maxReplicationQueueLength, priority,
104                conf, abortable)
105            : null;
106  }
107
108
109  public SimpleRpcScheduler(
110        Configuration conf,
111        int handlerCount,
112        int priorityHandlerCount,
113        int replicationHandlerCount,
114        PriorityFunction priority,
115        int highPriorityLevel) {
116    this(conf, handlerCount, priorityHandlerCount, replicationHandlerCount, priority,
117      null, highPriorityLevel);
118  }
119
120  /**
121   * Resize call queues;
122   * @param conf new configuration
123   */
124  @Override
125  public void onConfigurationChange(Configuration conf) {
126    callExecutor.resizeQueues(conf);
127    if (priorityExecutor != null) {
128      priorityExecutor.resizeQueues(conf);
129    }
130    if (replicationExecutor != null) {
131      replicationExecutor.resizeQueues(conf);
132    }
133
134    String callQueueType = conf.get(RpcExecutor.CALL_QUEUE_TYPE_CONF_KEY,
135      RpcExecutor.CALL_QUEUE_TYPE_CONF_DEFAULT);
136    if (RpcExecutor.isCodelQueueType(callQueueType)) {
137      callExecutor.onConfigurationChange(conf);
138    }
139  }
140
141  @Override
142  public void init(Context context) {
143    this.port = context.getListenerAddress().getPort();
144  }
145
146  @Override
147  public void start() {
148    callExecutor.start(port);
149    if (priorityExecutor != null) priorityExecutor.start(port);
150    if (replicationExecutor != null) replicationExecutor.start(port);
151  }
152
153  @Override
154  public void stop() {
155    callExecutor.stop();
156    if (priorityExecutor != null) priorityExecutor.stop();
157    if (replicationExecutor != null) replicationExecutor.stop();
158  }
159
160  @Override
161  public boolean dispatch(CallRunner callTask) throws InterruptedException {
162    RpcCall call = callTask.getRpcCall();
163    int level = priority.getPriority(call.getHeader(), call.getParam(),
164        call.getRequestUser().orElse(null));
165    if (level == HConstants.PRIORITY_UNSET) {
166      level = HConstants.NORMAL_QOS;
167    }
168    if (priorityExecutor != null && level > highPriorityLevel) {
169      return priorityExecutor.dispatch(callTask);
170    } else if (replicationExecutor != null && level == HConstants.REPLICATION_QOS) {
171      return replicationExecutor.dispatch(callTask);
172    } else {
173      return callExecutor.dispatch(callTask);
174    }
175  }

253  @Override
254  public CallQueueInfo getCallQueueInfo() {
255    String queueName;
256
257    CallQueueInfo callQueueInfo = new CallQueueInfo();
258
259    if(null!=callExecutor) {
260      queueName = "Call Queue";
261      callQueueInfo.setCallMethodCount(queueName, callExecutor.getCallQueueCountsSummary());
262      callQueueInfo.setCallMethodSize(queueName, callExecutor.getCallQueueSizeSummary());
263    }
264
265    if(null!=priorityExecutor) {
266      queueName = "Priority Queue";
267      callQueueInfo.setCallMethodCount(queueName, priorityExecutor.getCallQueueCountsSummary());
268      callQueueInfo.setCallMethodSize(queueName, priorityExecutor.getCallQueueSizeSummary());
269    }
270
271    if(null!=replicationExecutor) {
272      queueName = "Replication Queue";
273      callQueueInfo.setCallMethodCount(queueName, replicationExecutor.getCallQueueCountsSummary());
274      callQueueInfo.setCallMethodSize(queueName, replicationExecutor.getCallQueueSizeSummary());
275    }
276
277    return callQueueInfo;
278  }
