051/**
052 * The netty rpc handler.
053 * @since 2.0.0
054 */
055@InterfaceAudience.Private
056class NettyRpcDuplexHandler extends ChannelDuplexHandler {
057
058  private static final Logger LOG = LoggerFactory.getLogger(NettyRpcDuplexHandler.class);
059
060  private final NettyRpcConnection conn;
061
062  private final CellBlockBuilder cellBlockBuilder;
063
064  private final Codec codec;
065
066  private final CompressionCodec compressor;
067
068  private final Map<Integer, Call> id2Call = new HashMap<>();
069
070  public NettyRpcDuplexHandler(NettyRpcConnection conn, CellBlockBuilder cellBlockBuilder,
071    Codec codec, CompressionCodec compressor) {
072    this.conn = conn;
073    this.cellBlockBuilder = cellBlockBuilder;
074    this.codec = codec;
075    this.compressor = compressor;
076
077  }
078
079  private void writeRequest(ChannelHandlerContext ctx, Call call, ChannelPromise promise)
080    throws IOException {
081    id2Call.put(call.id, call);
082    ByteBuf cellBlock = cellBlockBuilder.buildCellBlock(codec, compressor, call.cells, ctx.alloc());
083    CellBlockMeta cellBlockMeta;
084    if (cellBlock != null) {
085      CellBlockMeta.Builder cellBlockMetaBuilder = CellBlockMeta.newBuilder();
086      cellBlockMetaBuilder.setLength(cellBlock.writerIndex());
087      cellBlockMeta = cellBlockMetaBuilder.build();
088    } else {
089      cellBlockMeta = null;
090    }
091    RequestHeader requestHeader = IPCUtil.buildRequestHeader(call, cellBlockMeta);
092    int sizeWithoutCellBlock = IPCUtil.getTotalSizeWhenWrittenDelimited(requestHeader, call.param);
093    int totalSize =
094      cellBlock != null ? sizeWithoutCellBlock + cellBlock.writerIndex() : sizeWithoutCellBlock;
095    ByteBuf buf = ctx.alloc().buffer(sizeWithoutCellBlock + 4);
096    buf.writeInt(totalSize);
097    try (ByteBufOutputStream bbos = new ByteBufOutputStream(buf)) {
098      requestHeader.writeDelimitedTo(bbos);
099      if (call.param != null) {
100        call.param.writeDelimitedTo(bbos);
101      }
102      if (cellBlock != null) {
103        ChannelPromise withoutCellBlockPromise = ctx.newPromise();
104        ctx.write(buf, withoutCellBlockPromise);
105        ChannelPromise cellBlockPromise = ctx.newPromise();
106        ctx.write(cellBlock, cellBlockPromise);
107        PromiseCombiner combiner = new PromiseCombiner(ctx.executor());
108        combiner.addAll((ChannelFuture) withoutCellBlockPromise, cellBlockPromise);
109        combiner.finish(promise);
110      } else {
111        ctx.write(buf, promise);
112      }
113    }
114  }
115
116  @Override
117  public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise)
118    throws Exception {
119    if (msg instanceof Call) {
120      Call call = (Call) msg;
121      try (Scope scope = call.span.makeCurrent()) {
122        writeRequest(ctx, call, promise);
123      }
124    } else {
125      ctx.write(msg, promise);
126    }
127  }
128
129  private void finishCall(ResponseHeader responseHeader, ByteBufInputStream in, Call call)
130    throws IOException {
131    Message value;
132    if (call.responseDefaultType != null) {
133      Message.Builder builder = call.responseDefaultType.newBuilderForType();
134      if (!builder.mergeDelimitedFrom(in)) {
135        // The javadoc of mergeDelimitedFrom says returning false means the stream reaches EOF
136        // before reading any bytes out, so here we need to manually finish create the EOFException
137        // and finish the call
138        call.setException(new EOFException("EOF while reading response with type: "
139          + call.responseDefaultType.getClass().getName()));
140        return;
141      }
142      value = builder.build();
143    } else {
144      value = null;
145    }
146    CellScanner cellBlockScanner;
147    if (responseHeader.hasCellBlockMeta()) {
148      int size = responseHeader.getCellBlockMeta().getLength();
149      // Maybe we could read directly from the ByteBuf.
150      // The problem here is that we do not know when to release it.
151      byte[] cellBlock = new byte[size];
152      in.readFully(cellBlock);
153      cellBlockScanner = cellBlockBuilder.createCellScanner(this.codec, this.compressor, cellBlock);
154    } else {
155      cellBlockScanner = null;
156    }
157    call.setResponse(value, cellBlockScanner);
158  }
159
160  private void readResponse(ChannelHandlerContext ctx, ByteBuf buf) throws IOException {
161    int totalSize = buf.readInt();
162    ByteBufInputStream in = new ByteBufInputStream(buf);
163    ResponseHeader responseHeader = ResponseHeader.parseDelimitedFrom(in);
164    int id = responseHeader.getCallId();
165    if (LOG.isTraceEnabled()) {
166      LOG.trace("got response header " + TextFormat.shortDebugString(responseHeader)
167        + ", totalSize: " + totalSize + " bytes");
168    }
169    RemoteException remoteExc;
170    if (responseHeader.hasException()) {
171      ExceptionResponse exceptionResponse = responseHeader.getException();
172      remoteExc = IPCUtil.createRemoteException(exceptionResponse);
173      if (IPCUtil.isFatalConnectionException(exceptionResponse)) {
174        // Here we will cleanup all calls so do not need to fall back, just return.
175        exceptionCaught(ctx, remoteExc);
176        return;
177      }
178    } else {
179      remoteExc = null;
180    }
181    Call call = id2Call.remove(id);
182    if (call == null) {
183      // So we got a response for which we have no corresponding 'call' here on the client-side.
184      // We probably timed out waiting, cleaned up all references, and now the server decides
185      // to return a response. There is nothing we can do w/ the response at this stage. Clean
186      // out the wire of the response so its out of the way and we can get other responses on
187      // this connection.
188      if (LOG.isDebugEnabled()) {
189        int readSoFar = IPCUtil.getTotalSizeWhenWrittenDelimited(responseHeader);
190        int whatIsLeftToRead = totalSize - readSoFar;
191        LOG.debug("Unknown callId: " + id + ", skipping over this response of " + whatIsLeftToRead
192          + " bytes");
193      }
194      return;
195    }
196    if (remoteExc != null) {
197      call.setException(remoteExc);
198      return;
199    }
200    try {
201      finishCall(responseHeader, in, call);
202    } catch (IOException e) {
203      // As the call has been removed from id2Call map, if we hit an exception here, the
204      // exceptionCaught method can not help us finish the call, so here we need to catch the
205      // exception and finish it
206      // And in netty, the decoding the frame based, when reaching here we have already read a full
207      // frame, so hitting exception here does not mean the stream decoding is broken, thus we do
208      // not need to throw the exception out and close the connection.
209      call.setException(e);
210    }
211  }
212
213  @Override
214  public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
215    if (msg instanceof ByteBuf) {
216      ByteBuf buf = (ByteBuf) msg;
217      try {
218        readResponse(ctx, buf);
219      } finally {
220        buf.release();
221      }
222    } else {
223      super.channelRead(ctx, msg);
224    }
225  }
226
227  private void cleanupCalls(IOException error) {
228    for (Call call : id2Call.values()) {
229      call.setException(error);
230    }
231    id2Call.clear();
232  }
233
234  @Override
235  public void channelInactive(ChannelHandlerContext ctx) throws Exception {
236    if (!id2Call.isEmpty()) {
237      cleanupCalls(new ConnectionClosedException("Connection closed"));
238    }
239    conn.shutdown();
240    ctx.fireChannelInactive();
241  }
242
243  @Override
244  public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
245    if (!id2Call.isEmpty()) {
246      cleanupCalls(IPCUtil.toIOE(cause));
247    }
248    conn.shutdown();
249  }
250
251  @Override
252  public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
253    if (evt instanceof IdleStateEvent) {
254      IdleStateEvent idleEvt = (IdleStateEvent) evt;
255      switch (idleEvt.state()) {
256        case WRITER_IDLE:
257          if (id2Call.isEmpty()) {
258            if (LOG.isTraceEnabled()) {
259              LOG.trace("shutdown connection to " + conn.remoteId().address
260                + " because idle for a long time");
261            }
262            // It may happen that there are still some pending calls in the event loop queue and
263            // they will get a closed channel exception. But this is not a big deal as it rarely
264            // rarely happens and the upper layer could retry immediately.
265            conn.shutdown();
266          }
267          break;
268        default:
269          LOG.warn("Unrecognized idle state " + idleEvt.state());
270          break;
271      }
272    } else if (evt instanceof CallEvent) {
273      // just remove the call for now until we add other call event other than timeout and cancel.
274      id2Call.remove(((CallEvent) evt).call.id);
275    } else {
276      ctx.fireUserEventTriggered(evt);
277    }
278  }
