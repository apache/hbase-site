096  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="NP_NULL_ON_SOME_PATH",
097      justification="Can't figure why this complaint is happening... see below")
098  ServerCall(int id, BlockingService service, MethodDescriptor md, RequestHeader header,
099      Message param, CellScanner cellScanner, T connection, long size, TraceInfo tinfo,
100      InetAddress remoteAddress, long receiveTime, int timeout, ByteBufferPool reservoir,
101      CellBlockBuilder cellBlockBuilder, CallCleanup reqCleanup) {
102    this.id = id;
103    this.service = service;
104    this.md = md;
105    this.header = header;
106    this.param = param;
107    this.cellScanner = cellScanner;
108    this.connection = connection;
109    this.receiveTime = receiveTime;
110    this.response = null;
111    this.isError = false;
112    this.size = size;
113    this.tinfo = tinfo;
114    if (connection != null) {
115      this.user =  connection.user;
116      this.retryImmediatelySupported = connection.retryImmediatelySupported;
117    } else {
118      this.user = null;
119      this.retryImmediatelySupported = false;
120    }
121    this.remoteAddress = remoteAddress;
122    this.timeout = timeout;
123    this.deadline = this.timeout > 0 ? this.receiveTime + this.timeout : Long.MAX_VALUE;
124    this.reservoir = reservoir;
125    this.cellBlockBuilder = cellBlockBuilder;
126    this.reqCleanup = reqCleanup;
127  }

129  /**
130   * Call is done. Execution happened and we returned results to client. It is
131   * now safe to cleanup.
132   */
133  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "IS2_INCONSISTENT_SYNC",
134      justification = "Presume the lock on processing request held by caller is protection enough")
135  @Override
136  public void done() {
137    if (this.cellBlockStream != null) {
138      // This will return back the BBs which we got from pool.
139      this.cellBlockStream.releaseResources();
140      this.cellBlockStream = null;
141    }
142    // If the call was run successfuly, we might have already returned the BB
143    // back to pool. No worries..Then inputCellBlock will be null
144    cleanup();
145  }
146
147  @Override
148  public void cleanup() {
149    if (this.reqCleanup != null) {
150      this.reqCleanup.run();
151      this.reqCleanup = null;
152    }
153  }

176  @Override
177  public String toShortString() {
178    String serviceName = this.connection.service != null ?
179        this.connection.service.getDescriptorForType().getName() : "null";
180    return "callId: " + this.id + " service: " + serviceName +
181        " methodName: " + ((this.md != null) ? this.md.getName() : "n/a") +
182        " size: " + StringUtils.TraditionalBinaryPrefix.long2String(this.size, "", 1) +
183        " connection: " + connection.toString() +
184        " deadline: " + deadline;
185  }

187  @Override
188  public synchronized void setResponse(Message m, final CellScanner cells,
189      Throwable t, String errorMsg) {
190    if (this.isError) return;
191    if (t != null) this.isError = true;
192    BufferChain bc = null;
193    try {
194      ResponseHeader.Builder headerBuilder = ResponseHeader.newBuilder();
195      // Call id.
196      headerBuilder.setCallId(this.id);
197      if (t != null) {
198        setExceptionResponse(t, errorMsg, headerBuilder);
199      }
200      // Pass reservoir to buildCellBlock. Keep reference to returne so can add it back to the
201      // reservoir when finished. This is hacky and the hack is not contained but benefits are
202      // high when we can avoid a big buffer allocation on each rpc.
203      List<ByteBuffer> cellBlock = null;
204      int cellBlockSize = 0;
205      if (this.reservoir != null) {
206        this.cellBlockStream = this.cellBlockBuilder.buildCellBlockStream(this.connection.codec,
207          this.connection.compressionCodec, cells, this.reservoir);
208        if (this.cellBlockStream != null) {
209          cellBlock = this.cellBlockStream.getByteBuffers();
210          cellBlockSize = this.cellBlockStream.size();
211        }
212      } else {
213        ByteBuffer b = this.cellBlockBuilder.buildCellBlock(this.connection.codec,
214          this.connection.compressionCodec, cells);
215        if (b != null) {
216          cellBlockSize = b.remaining();
217          cellBlock = new ArrayList<>(1);
218          cellBlock.add(b);
219        }
220      }
221
222      if (cellBlockSize > 0) {
223        CellBlockMeta.Builder cellBlockBuilder = CellBlockMeta.newBuilder();
224        // Presumes the cellBlock bytebuffer has been flipped so limit has total size in it.
225        cellBlockBuilder.setLength(cellBlockSize);
226        headerBuilder.setCellBlockMeta(cellBlockBuilder.build());
227      }
228      Message header = headerBuilder.build();
229      ByteBuffer headerBuf =
230          createHeaderAndMessageBytes(m, header, cellBlockSize, cellBlock);
231      ByteBuffer[] responseBufs = null;
232      int cellBlockBufferSize = 0;
233      if (cellBlock != null) {
234        cellBlockBufferSize = cellBlock.size();
235        responseBufs = new ByteBuffer[1 + cellBlockBufferSize];
236      } else {
237        responseBufs = new ByteBuffer[1];
238      }
239      responseBufs[0] = headerBuf;
240      if (cellBlock != null) {
241        for (int i = 0; i < cellBlockBufferSize; i++) {
242          responseBufs[i + 1] = cellBlock.get(i);
243        }
244      }
245      bc = new BufferChain(responseBufs);
246      if (connection.useWrap) {
247        bc = wrapWithSasl(bc);
248      }
249    } catch (IOException e) {
250      RpcServer.LOG.warn("Exception while creating response " + e);
251    }
252    this.response = bc;
253    // Once a response message is created and set to this.response, this Call can be treated as
254    // done. The Responder thread will do the n/w write of this message back to client.
255    if (this.rpcCallback != null) {
256      try {
257        this.rpcCallback.run();
258      } catch (Exception e) {
259        // Don't allow any exception here to kill this handler thread.
260        RpcServer.LOG.warn("Exception while running the Rpc Callback.", e);
261      }
262    }
263  }

265  static void setExceptionResponse(Throwable t, String errorMsg,
266      ResponseHeader.Builder headerBuilder) {
267    ExceptionResponse.Builder exceptionBuilder = ExceptionResponse.newBuilder();
268    exceptionBuilder.setExceptionClassName(t.getClass().getName());
269    exceptionBuilder.setStackTrace(errorMsg);
270    exceptionBuilder.setDoNotRetry(t instanceof DoNotRetryIOException);
271    if (t instanceof RegionMovedException) {
272      // Special casing for this exception.  This is only one carrying a payload.
273      // Do this instead of build a generic system for allowing exceptions carry
274      // any kind of payload.
275      RegionMovedException rme = (RegionMovedException)t;
276      exceptionBuilder.setHostname(rme.getHostname());
277      exceptionBuilder.setPort(rme.getPort());
278    }
279    // Set the exception as the result of the method invocation.
280    headerBuilder.setException(exceptionBuilder.build());
281  }

283  static ByteBuffer createHeaderAndMessageBytes(Message result, Message header,
284      int cellBlockSize, List<ByteBuffer> cellBlock) throws IOException {
285    // Organize the response as a set of bytebuffers rather than collect it all together inside
286    // one big byte array; save on allocations.
287    // for writing the header, we check if there is available space in the buffers
288    // created for the cellblock itself. If there is space for the header, we reuse
289    // the last buffer in the cellblock. This applies to the cellblock created from the
290    // pool or even the onheap cellblock buffer in case there is no pool enabled.
291    // Possible reuse would avoid creating a temporary array for storing the header every time.
292    ByteBuffer possiblePBBuf =
293        (cellBlockSize > 0) ? cellBlock.get(cellBlock.size() - 1) : null;
294    int headerSerializedSize = 0, resultSerializedSize = 0, headerVintSize = 0,
295        resultVintSize = 0;
296    if (header != null) {
297      headerSerializedSize = header.getSerializedSize();
298      headerVintSize = CodedOutputStream.computeUInt32SizeNoTag(headerSerializedSize);
299    }
300    if (result != null) {
301      resultSerializedSize = result.getSerializedSize();
302      resultVintSize = CodedOutputStream.computeUInt32SizeNoTag(resultSerializedSize);
303    }
304    // calculate the total size
305    int totalSize = headerSerializedSize + headerVintSize
306        + (resultSerializedSize + resultVintSize)
307        + cellBlockSize;
308    int totalPBSize = headerSerializedSize + headerVintSize + resultSerializedSize
309        + resultVintSize + Bytes.SIZEOF_INT;
310    // Only if the last buffer has enough space for header use it. Else allocate
311    // a new buffer. Assume they are all flipped
312    if (possiblePBBuf != null
313        && possiblePBBuf.limit() + totalPBSize <= possiblePBBuf.capacity()) {
314      // duplicate the buffer. This is where the header is going to be written
315      ByteBuffer pbBuf = possiblePBBuf.duplicate();
316      // get the current limit
317      int limit = pbBuf.limit();
318      // Position such that we write the header to the end of the buffer
319      pbBuf.position(limit);
320      // limit to the header size
321      pbBuf.limit(totalPBSize + limit);
322      // mark the current position
323      pbBuf.mark();
324      writeToCOS(result, header, totalSize, pbBuf);
325      // reset the buffer back to old position
326      pbBuf.reset();
327      return pbBuf;
328    } else {
329      return createHeaderAndMessageBytes(result, header, totalSize, totalPBSize);
330    }
331  }
332
333  private static void writeToCOS(Message result, Message header, int totalSize, ByteBuffer pbBuf)
334      throws IOException {
335    ByteBufferUtils.putInt(pbBuf, totalSize);
336    // create COS that works on BB
337    CodedOutputStream cos = CodedOutputStream.newInstance(pbBuf);
338    if (header != null) {
339      cos.writeMessageNoTag(header);
340    }
341    if (result != null) {
342      cos.writeMessageNoTag(result);
343    }
344    cos.flush();
345    cos.checkNoSpaceLeft();
346  }
347
348  private static ByteBuffer createHeaderAndMessageBytes(Message result, Message header,
349      int totalSize, int totalPBSize) throws IOException {
350    ByteBuffer pbBuf = ByteBuffer.allocate(totalPBSize);
351    writeToCOS(result, header, totalSize, pbBuf);
352    pbBuf.flip();
353    return pbBuf;
354  }

356  protected BufferChain wrapWithSasl(BufferChain bc)
357      throws IOException {
358    if (!this.connection.useSasl) return bc;
359    // Looks like no way around this; saslserver wants a byte array.  I have to make it one.
360    // THIS IS A BIG UGLY COPY.
361    byte [] responseBytes = bc.getBytes();
362    byte [] token;
363    // synchronization may be needed since there can be multiple Handler
364    // threads using saslServer or Crypto AES to wrap responses.
365    if (connection.useCryptoAesWrap) {
366      // wrap with Crypto AES
367      synchronized (connection.cryptoAES) {
368        token = connection.cryptoAES.wrap(responseBytes, 0, responseBytes.length);
369      }
370    } else {
371      synchronized (connection.saslServer) {
372        token = connection.saslServer.wrap(responseBytes, 0, responseBytes.length);
373      }
374    }
375    if (RpcServer.LOG.isTraceEnabled()) {
376      RpcServer.LOG.trace("Adding saslServer wrapped token of size " + token.length
377          + " as call response.");
378    }
379
380    ByteBuffer[] responseBufs = new ByteBuffer[2];
381    responseBufs[0] = ByteBuffer.wrap(Bytes.toBytes(token.length));
382    responseBufs[1] = ByteBuffer.wrap(token);
383    return new BufferChain(responseBufs);
384  }

388    if (!this.connection.isConnectionOpen()) {
389      return System.currentTimeMillis() - receiveTime;
390    } else {
391      return -1L;
