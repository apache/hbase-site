108  // This is a dirty hack to address HBASE-22539. The highest bit is for rpc ref and cleanup, and
109  // the rest of the bits are for WAL reference count. We can only call release if all of them are
110  // zero. The reason why we can not use a general reference counting is that, we may call cleanup
111  // multiple times in the current implementation. We should fix this in the future.
112  // The refCount here will start as 0x80000000 and increment with every WAL reference and decrement
113  // from WAL side on release
114  private final AtomicInteger reference = new AtomicInteger(0x80000000);

120  ServerCall(int id, BlockingService service, MethodDescriptor md, RequestHeader header,
121    Message param, CellScanner cellScanner, T connection, long size, InetAddress remoteAddress,

125    this.service = service;

127    this.header = header;

130    this.connection = connection;

132    this.response = null;

134    this.size = size;
135    if (connection != null) {
136      this.user = connection.user;
137      this.retryImmediatelySupported = connection.retryImmediatelySupported;
138    } else {
139      this.user = null;
140      this.retryImmediatelySupported = false;
141    }
142    this.remoteAddress = remoteAddress;
143    this.timeout = timeout;
144    this.deadline = this.timeout > 0 ? this.receiveTime + this.timeout : Long.MAX_VALUE;
145    this.bbAllocator = byteBuffAllocator;
146    this.cellBlockBuilder = cellBlockBuilder;
147    this.reqCleanup = reqCleanup;
148    this.span = Span.current();

151  /**
152   * Call is done. Execution happened and we returned results to client. It is now safe to cleanup.
153   */
154  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "IS2_INCONSISTENT_SYNC",
155      justification = "Presume the lock on processing request held by caller is protection enough")
156  @Override
157  public void done() {
158    if (this.cellBlockStream != null) {
159      // This will return back the BBs which we got from pool.
160      this.cellBlockStream.releaseResources();
161      this.cellBlockStream = null;
162    }
163    // If the call was run successfuly, we might have already returned the BB
164    // back to pool. No worries..Then inputCellBlock will be null
165    cleanup();
166    span.end();
167  }
168
169  @Override
170  public void cleanup() {
171    for (;;) {
172      int ref = reference.get();
173      if ((ref & 0x80000000) == 0) {
174        return;
175      }
176      int nextRef = ref & 0x7fffffff;
177      if (reference.compareAndSet(ref, nextRef)) {
178        if (nextRef == 0) {
179          if (this.reqCleanup != null) {
180            this.reqCleanup.run();
181          }
182        }
183        return;
184      }
185    }
186  }
187
188  public void retainByWAL() {
189    reference.incrementAndGet();
190  }
191
192  public void releaseByWAL() {
193    // Here this method of decrementAndGet for releasing WAL reference count will work in both
194    // cases - i.e. highest bit (cleanup) 1 or 0. We will be decrementing a negative or positive
195    // value respectively in these 2 cases, but the logic will work the same way
196    if (reference.decrementAndGet() == 0) {
197      if (this.reqCleanup != null) {
198        this.reqCleanup.run();
199      }
200    }
201
202  }

221  @Override
222  public Map<String, byte[]> getRequestAttributes() {
223    if (this.requestAttributes == null) {
224      if (header.getAttributeList().isEmpty()) {
225        this.requestAttributes = Collections.emptyMap();
226      } else {
227        Map<String, byte[]> requestAttributes =
228          Maps.newHashMapWithExpectedSize(header.getAttributeList().size());
229        for (HBaseProtos.NameBytesPair nameBytesPair : header.getAttributeList()) {
230          requestAttributes.put(nameBytesPair.getName(), nameBytesPair.getValue().toByteArray());
231        }
232        this.requestAttributes = requestAttributes;
233      }
234    }
235    return this.requestAttributes;
236  }
237
238  @Override
239  public byte[] getRequestAttribute(String key) {
240    if (this.requestAttributes == null) {
241      for (HBaseProtos.NameBytesPair nameBytesPair : header.getAttributeList()) {
242        if (nameBytesPair.getName().equals(key)) {
243          return nameBytesPair.getValue().toByteArray();
244        }
245      }
246      return null;
247    }
248    return this.requestAttributes.get(key);
249  }

264      : "null";
265    return "callId: " + this.id + " service: " + serviceName + " methodName: "

271  @Override
272  public synchronized void setResponse(Message m, final CellScanner cells, Throwable t,
273    String errorMsg) {
274    if (this.isError) {
275      return;
276    }
277    if (t != null) {
278      this.isError = true;
279      TraceUtil.setError(span, t);
280    } else {
281      span.setStatus(StatusCode.OK);
282    }
283    BufferChain bc = null;
284    try {
285      ResponseHeader.Builder headerBuilder = ResponseHeader.newBuilder();
286      // Call id.
287      headerBuilder.setCallId(this.id);
288      if (t != null) {
289        setExceptionResponse(t, errorMsg, headerBuilder);
290      }
291      // Pass reservoir to buildCellBlock. Keep reference to returne so can add it back to the
292      // reservoir when finished. This is hacky and the hack is not contained but benefits are
293      // high when we can avoid a big buffer allocation on each rpc.
294      List<ByteBuffer> cellBlock = null;
295      int cellBlockSize = 0;
296      if (bbAllocator.isReservoirEnabled()) {
297        this.cellBlockStream = this.cellBlockBuilder.buildCellBlockStream(this.connection.codec,
298          this.connection.compressionCodec, cells, bbAllocator);
299        if (this.cellBlockStream != null) {
300          cellBlock = this.cellBlockStream.getByteBuffers();
301          cellBlockSize = this.cellBlockStream.size();
302        }
303      } else {
304        ByteBuffer b = this.cellBlockBuilder.buildCellBlock(this.connection.codec,
305          this.connection.compressionCodec, cells);
306        if (b != null) {
307          cellBlockSize = b.remaining();
308          cellBlock = new ArrayList<>(1);
309          cellBlock.add(b);
310        }
311      }
312
313      if (cellBlockSize > 0) {
314        CellBlockMeta.Builder cellBlockBuilder = CellBlockMeta.newBuilder();
315        // Presumes the cellBlock bytebuffer has been flipped so limit has total size in it.
316        cellBlockBuilder.setLength(cellBlockSize);
317        headerBuilder.setCellBlockMeta(cellBlockBuilder.build());
318      }
319      Message header = headerBuilder.build();
320      ByteBuffer headerBuf = createHeaderAndMessageBytes(m, header, cellBlockSize, cellBlock);
321      ByteBuffer[] responseBufs = null;
322      int cellBlockBufferSize = 0;
323      if (cellBlock != null) {
324        cellBlockBufferSize = cellBlock.size();
325        responseBufs = new ByteBuffer[1 + cellBlockBufferSize];
326      } else {
327        responseBufs = new ByteBuffer[1];
328      }
329      responseBufs[0] = headerBuf;
330      if (cellBlock != null) {
331        for (int i = 0; i < cellBlockBufferSize; i++) {
332          responseBufs[i + 1] = cellBlock.get(i);
333        }
334      }
335      bc = new BufferChain(responseBufs);
336    } catch (IOException e) {
337      RpcServer.LOG.warn("Exception while creating response " + e);
338    }
339    this.response = bc;
340    // Once a response message is created and set to this.response, this Call can be treated as
341    // done. The Responder thread will do the n/w write of this message back to client.
342    if (this.rpcCallback != null) {
343      try (Scope ignored = span.makeCurrent()) {
344        this.rpcCallback.run();
345      } catch (Exception e) {
346        // Don't allow any exception here to kill this handler thread.
347        RpcServer.LOG.warn("Exception while running the Rpc Callback.", e);
348        TraceUtil.setError(span, e);
349      }
350    }
351  }

353  static void setExceptionResponse(Throwable t, String errorMsg,
354    ResponseHeader.Builder headerBuilder) {
355    ExceptionResponse.Builder exceptionBuilder = ExceptionResponse.newBuilder();
356    exceptionBuilder.setExceptionClassName(t.getClass().getName());
357    exceptionBuilder.setStackTrace(errorMsg);
358    exceptionBuilder.setDoNotRetry(t instanceof DoNotRetryIOException);
359    if (t instanceof RegionMovedException) {
360      // Special casing for this exception. This is only one carrying a payload.
361      // Do this instead of build a generic system for allowing exceptions carry
362      // any kind of payload.
363      RegionMovedException rme = (RegionMovedException) t;
364      exceptionBuilder.setHostname(rme.getHostname());
365      exceptionBuilder.setPort(rme.getPort());
366    } else if (t instanceof HBaseServerException) {
367      HBaseServerException hse = (HBaseServerException) t;
368      exceptionBuilder.setServerOverloaded(hse.isServerOverloaded());
369    }
370    // Set the exception as the result of the method invocation.
371    headerBuilder.setException(exceptionBuilder.build());
372  }

374  static ByteBuffer createHeaderAndMessageBytes(Message result, Message header, int cellBlockSize,
375    List<ByteBuffer> cellBlock) throws IOException {
376    // Organize the response as a set of bytebuffers rather than collect it all together inside
377    // one big byte array; save on allocations.
378    // for writing the header, we check if there is available space in the buffers
379    // created for the cellblock itself. If there is space for the header, we reuse
380    // the last buffer in the cellblock. This applies to the cellblock created from the
381    // pool or even the onheap cellblock buffer in case there is no pool enabled.
382    // Possible reuse would avoid creating a temporary array for storing the header every time.
383    ByteBuffer possiblePBBuf = (cellBlockSize > 0) ? cellBlock.get(cellBlock.size() - 1) : null;
384    int headerSerializedSize = 0, resultSerializedSize = 0, headerVintSize = 0, resultVintSize = 0;
385    if (header != null) {
386      headerSerializedSize = header.getSerializedSize();
387      headerVintSize = CodedOutputStream.computeUInt32SizeNoTag(headerSerializedSize);
388    }
389    if (result != null) {
390      resultSerializedSize = result.getSerializedSize();
391      resultVintSize = CodedOutputStream.computeUInt32SizeNoTag(resultSerializedSize);
392    }
393    // calculate the total size
394    int totalSize = headerSerializedSize + headerVintSize + (resultSerializedSize + resultVintSize)
395      + cellBlockSize;
396    int totalPBSize = headerSerializedSize + headerVintSize + resultSerializedSize + resultVintSize
397      + Bytes.SIZEOF_INT;
398    // Only if the last buffer has enough space for header use it. Else allocate
399    // a new buffer. Assume they are all flipped
400    if (possiblePBBuf != null && possiblePBBuf.limit() + totalPBSize <= possiblePBBuf.capacity()) {
401      // duplicate the buffer. This is where the header is going to be written
402      ByteBuffer pbBuf = possiblePBBuf.duplicate();
403      // get the current limit
404      int limit = pbBuf.limit();
405      // Position such that we write the header to the end of the buffer
406      pbBuf.position(limit);
407      // limit to the header size
408      pbBuf.limit(totalPBSize + limit);
409      // mark the current position
410      pbBuf.mark();
411      writeToCOS(result, header, totalSize, pbBuf);
412      // reset the buffer back to old position
413      pbBuf.reset();
414      return pbBuf;
415    } else {
416      return createHeaderAndMessageBytes(result, header, totalSize, totalPBSize);
417    }
418  }
419
420  private static void writeToCOS(Message result, Message header, int totalSize, ByteBuffer pbBuf)
421    throws IOException {
422    ByteBufferUtils.putInt(pbBuf, totalSize);
423    // create COS that works on BB
424    CodedOutputStream cos = CodedOutputStream.newInstance(pbBuf);
425    if (header != null) {
426      cos.writeMessageNoTag(header);
427    }
428    if (result != null) {
429      cos.writeMessageNoTag(result);
430    }
431    cos.flush();
432    cos.checkNoSpaceLeft();
433  }
434
435  private static ByteBuffer createHeaderAndMessageBytes(Message result, Message header,
436    int totalSize, int totalPBSize) throws IOException {
437    ByteBuffer pbBuf = ByteBuffer.allocate(totalPBSize);
438    writeToCOS(result, header, totalSize, pbBuf);
439    pbBuf.flip();
440    return pbBuf;
441  }
