045/**
046 * Runs the CallRunners passed here via {@link #dispatch(CallRunner)}. Subclass and add particular
047 * scheduling behavior.
048 */
049@InterfaceAudience.Private
050public abstract class RpcExecutor {
051  private static final Log LOG = LogFactory.getLog(RpcExecutor.class);
052
053  protected static final int DEFAULT_CALL_QUEUE_SIZE_HARD_LIMIT = 250;
054  public static final String CALL_QUEUE_HANDLER_FACTOR_CONF_KEY = "hbase.ipc.server.callqueue.handler.factor";
055
056  /** max delay in msec used to bound the deprioritized requests */
057  public static final String QUEUE_MAX_CALL_DELAY_CONF_KEY = "hbase.ipc.server.queue.max.call.delay";
058
059  /**
060   * The default, 'fifo', has the least friction but is dumb. If set to 'deadline', uses a priority
061   * queue and deprioritizes long-running scans. Sorting by priority comes at a cost, reduced
062   * throughput.
063   */
064  public static final String CALL_QUEUE_TYPE_CODEL_CONF_VALUE = "codel";
065  public static final String CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE = "deadline";
066  public static final String CALL_QUEUE_TYPE_FIFO_CONF_VALUE = "fifo";
067  public static final String CALL_QUEUE_TYPE_CONF_KEY = "hbase.ipc.server.callqueue.type";
068  public static final String CALL_QUEUE_TYPE_CONF_DEFAULT = CALL_QUEUE_TYPE_FIFO_CONF_VALUE;
069
070  // These 3 are only used by Codel executor
071  public static final String CALL_QUEUE_CODEL_TARGET_DELAY = "hbase.ipc.server.callqueue.codel.target.delay";
072  public static final String CALL_QUEUE_CODEL_INTERVAL = "hbase.ipc.server.callqueue.codel.interval";
073  public static final String CALL_QUEUE_CODEL_LIFO_THRESHOLD = "hbase.ipc.server.callqueue.codel.lifo.threshold";
074
075  public static final int CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY = 100;
076  public static final int CALL_QUEUE_CODEL_DEFAULT_INTERVAL = 100;
077  public static final double CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD = 0.8;
078
079  private AtomicLong numGeneralCallsDropped = new AtomicLong();
080  private AtomicLong numLifoModeSwitches = new AtomicLong();

102  public RpcExecutor(final String name, final int handlerCount, final int maxQueueLength,
103      final PriorityFunction priority, final Configuration conf, final Abortable abortable) {
104    this(name, handlerCount, conf.get(CALL_QUEUE_TYPE_CONF_KEY,
105      CALL_QUEUE_TYPE_CONF_DEFAULT), maxQueueLength, priority, conf, abortable);
106  }
107
108  public RpcExecutor(final String name, final int handlerCount, final String callQueueType,
109      final int maxQueueLength, final PriorityFunction priority, final Configuration conf,
110      final Abortable abortable) {
111    this.name = Strings.nullToEmpty(name);
112    this.conf = conf;
113    this.abortable = abortable;
114
115    float callQueuesHandlersFactor = this.conf.getFloat(CALL_QUEUE_HANDLER_FACTOR_CONF_KEY, 0.1f);
116    this.numCallQueues = computeNumCallQueues(handlerCount, callQueuesHandlersFactor);
117    this.queues = new ArrayList<>(this.numCallQueues);
118
119    this.handlerCount = Math.max(handlerCount, this.numCallQueues);
120    this.handlers = new ArrayList<>(this.handlerCount);
121
122    this.priority = priority;
123
124    if (isDeadlineQueueType(callQueueType)) {
125      this.name += ".Deadline";
126      this.queueInitArgs = new Object[] { maxQueueLength,
127        new CallPriorityComparator(conf, this.priority) };
128      this.queueClass = BoundedPriorityBlockingQueue.class;
129    } else if (isCodelQueueType(callQueueType)) {
130      this.name += ".Codel";
131      int codelTargetDelay = conf.getInt(CALL_QUEUE_CODEL_TARGET_DELAY,
132        CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY);
133      int codelInterval = conf.getInt(CALL_QUEUE_CODEL_INTERVAL, CALL_QUEUE_CODEL_DEFAULT_INTERVAL);
134      double codelLifoThreshold = conf.getDouble(CALL_QUEUE_CODEL_LIFO_THRESHOLD,
135        CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD);
136      queueInitArgs = new Object[] { maxQueueLength, codelTargetDelay, codelInterval,
137          codelLifoThreshold, numGeneralCallsDropped, numLifoModeSwitches };
138      queueClass = AdaptiveLifoCoDelCallQueue.class;
139    } else {
140      this.name += ".Fifo";
141      queueInitArgs = new Object[] { maxQueueLength };
142      queueClass = LinkedBlockingQueue.class;
143    }
144
145    LOG.info("RpcExecutor " + name + " using " + callQueueType
146        + " as call queue; numCallQueues=" + numCallQueues + "; maxQueueLength=" + maxQueueLength
147        + "; handlerCount=" + handlerCount);
148  }
149
150  protected int computeNumCallQueues(final int handlerCount, final float callQueuesHandlersFactor) {
151    return Math.max(1, (int) Math.round(handlerCount * callQueuesHandlersFactor));
152  }
153
154  protected void initializeQueues(final int numQueues) {
155    if (queueInitArgs.length > 0) {
156      currentQueueLimit = (int) queueInitArgs[0];
157      queueInitArgs[0] = Math.max((int) queueInitArgs[0], DEFAULT_CALL_QUEUE_SIZE_HARD_LIMIT);
158    }
159    for (int i = 0; i < numQueues; ++i) {
160      queues
161          .add((BlockingQueue<CallRunner>) ReflectionUtils.newInstance(queueClass, queueInitArgs));
162    }
163  }
164
165  public void start(final int port) {
166    running = true;
167    startHandlers(port);
168  }
169
170  public void stop() {
171    running = false;
172    for (Thread handler : handlers) {
173      handler.interrupt();
174    }
175  }

177  /** Add the request to the executor queue */
178  public abstract boolean dispatch(final CallRunner callTask) throws InterruptedException;
179
180  /** Returns the list of request queues */
181  protected List<BlockingQueue<CallRunner>> getQueues() {
182    return queues;
183  }
184
185  protected void startHandlers(final int port) {
186    List<BlockingQueue<CallRunner>> callQueues = getQueues();
187    startHandlers(null, handlerCount, callQueues, 0, callQueues.size(), port, activeHandlerCount);
188  }
189
190  /**
191   * Override if providing alternate Handler implementation.
192   */
193  protected Handler getHandler(final String name, final double handlerFailureThreshhold,
194      final BlockingQueue<CallRunner> q, final AtomicInteger activeHandlerCount) {
195    return new Handler(name, handlerFailureThreshhold, q, activeHandlerCount);
196  }
197
198  /**
199   * Start up our handlers.
200   */
201  protected void startHandlers(final String nameSuffix, final int numHandlers,
202      final List<BlockingQueue<CallRunner>> callQueues, final int qindex, final int qsize,
203      final int port, final AtomicInteger activeHandlerCount) {
204    final String threadPrefix = name + Strings.nullToEmpty(nameSuffix);
205    double handlerFailureThreshhold = conf == null ? 1.0 : conf.getDouble(
206      HConstants.REGION_SERVER_HANDLER_ABORT_ON_ERROR_PERCENT,
207      HConstants.DEFAULT_REGION_SERVER_HANDLER_ABORT_ON_ERROR_PERCENT);
208    LOG.debug("Started " + handlers.size() + " " + threadPrefix +
209        " handlers, qsize=" + qsize + " on port=" + port);
210    for (int i = 0; i < numHandlers; i++) {
211      final int index = qindex + (i % qsize);
212      String name = "RpcServer." + threadPrefix + ".handler=" + handlers.size() + ",queue=" + index
213          + ",port=" + port;
214      Handler handler = getHandler(name, handlerFailureThreshhold, callQueues.get(index),
215        activeHandlerCount);
216      handler.start();
217      handlers.add(handler);
218    }
219  }
220
221  /**
222   * Handler thread run the {@link CallRunner#run()} in.
223   */
224  protected class Handler extends Thread {
225    /**
226     * Q to find CallRunners to run in.
227     */
228    final BlockingQueue<CallRunner> q;
229
230    final double handlerFailureThreshhold;
231
232    // metrics (shared with other handlers)
233    final AtomicInteger activeHandlerCount;
234
235    Handler(final String name, final double handlerFailureThreshhold,
236        final BlockingQueue<CallRunner> q, final AtomicInteger activeHandlerCount) {
237      super(name);
238      setDaemon(true);
239      this.q = q;
240      this.handlerFailureThreshhold = handlerFailureThreshhold;
241      this.activeHandlerCount = activeHandlerCount;
242    }
243
244    /**
245     * @return A {@link CallRunner}
246     * @throws InterruptedException
247     */
248    protected CallRunner getCallRunner() throws InterruptedException {
249      return this.q.take();
250    }
251
252    @Override
253    public void run() {
254      boolean interrupted = false;
255      try {
256        while (running) {
257          try {
258            run(getCallRunner());
259          } catch (InterruptedException e) {
260            interrupted = true;
261          }
262        }
263      } catch (Exception e) {
264        LOG.warn(e);
265        throw e;
266      } finally {
267        if (interrupted) {
268          Thread.currentThread().interrupt();
269        }
270      }
271    }
272
273    private void run(CallRunner cr) {
274      MonitoredRPCHandler status = RpcServer.getStatus();
275      cr.setStatus(status);
276      try {
277        this.activeHandlerCount.incrementAndGet();
278        cr.run();
279      } catch (Throwable e) {
280        if (e instanceof Error) {
281          int failedCount = failedHandlerCount.incrementAndGet();
282          if (this.handlerFailureThreshhold >= 0
283              && failedCount > handlerCount * this.handlerFailureThreshhold) {
284            String message = "Number of failed RpcServer handler runs exceeded threshhold "
285                + this.handlerFailureThreshhold + "; reason: " + StringUtils.stringifyException(e);
286            if (abortable != null) {
287              abortable.abort(message, e);
288            } else {
289              LOG.error("Error but can't abort because abortable is null: "
290                  + StringUtils.stringifyException(e));
291              throw e;
292            }
293          } else {
294            LOG.warn("Handler errors " + StringUtils.stringifyException(e));
295          }
296        } else {
297          LOG.warn("Handler  exception " + StringUtils.stringifyException(e));
298        }
299      } finally {
300        this.activeHandlerCount.decrementAndGet();
301      }
302    }
303  }

305  public static abstract class QueueBalancer {
306    /**
307     * @return the index of the next queue to which a request should be inserted
308     */
309    public abstract int getNextQueue();
310  }

312  public static QueueBalancer getBalancer(int queueSize) {
313    Preconditions.checkArgument(queueSize > 0, "Queue size is <= 0, must be at least 1");
314    if (queueSize == 1) {
315      return ONE_QUEUE;
316    } else {
317      return new RandomQueueBalancer(queueSize);
318    }
319  }

321  /**
322   * All requests go to the first queue, at index 0
323   */
324  private static QueueBalancer ONE_QUEUE = new QueueBalancer() {
325    @Override
326    public int getNextQueue() {
327      return 0;
328    }
329  };

331  /**
332   * Queue balancer that just randomly selects a queue in the range [0, num queues).
333   */
334  private static class RandomQueueBalancer extends QueueBalancer {
335    private final int queueSize;
336
337    public RandomQueueBalancer(int queueSize) {
338      this.queueSize = queueSize;
339    }
340
341    public int getNextQueue() {
342      return ThreadLocalRandom.current().nextInt(queueSize);
343    }
344  }

346  /**
347   * Comparator used by the "normal callQueue" if DEADLINE_CALL_QUEUE_CONF_KEY is set to true. It
348   * uses the calculated "deadline" e.g. to deprioritize long-running job If multiple requests have
349   * the same deadline BoundedPriorityBlockingQueue will order them in FIFO (first-in-first-out)
350   * manner.
351   */
352  private static class CallPriorityComparator implements Comparator<CallRunner> {
353    private final static int DEFAULT_MAX_CALL_DELAY = 5000;
354
355    private final PriorityFunction priority;
356    private final int maxDelay;
357
358    public CallPriorityComparator(final Configuration conf, final PriorityFunction priority) {
359      this.priority = priority;
360      this.maxDelay = conf.getInt(QUEUE_MAX_CALL_DELAY_CONF_KEY, DEFAULT_MAX_CALL_DELAY);
361    }
362
363    @Override
364    public int compare(CallRunner a, CallRunner b) {
365      RpcCall callA = a.getRpcCall();
366      RpcCall callB = b.getRpcCall();
367      long deadlineA = priority.getDeadline(callA.getHeader(), callA.getParam());
368      long deadlineB = priority.getDeadline(callB.getHeader(), callB.getParam());
369      deadlineA = callA.getReceiveTime() + Math.min(deadlineA, maxDelay);
370      deadlineB = callB.getReceiveTime() + Math.min(deadlineB, maxDelay);
371      return Long.compare(deadlineA, deadlineB);
372    }
373  }

436  /**
437   * Update current soft limit for executor's call queues
438   * @param conf updated configuration
439   */
440  public void resizeQueues(Configuration conf) {
441    String configKey = RpcScheduler.IPC_SERVER_MAX_CALLQUEUE_LENGTH;
442    if (name != null && name.toLowerCase(Locale.ROOT).contains("priority")) {
443      configKey = RpcScheduler.IPC_SERVER_PRIORITY_MAX_CALLQUEUE_LENGTH;
444    }
445    currentQueueLimit = conf.getInt(configKey, currentQueueLimit);
446  }
447
448  public void onConfigurationChange(Configuration conf) {
449    // update CoDel Scheduler tunables
450    int codelTargetDelay = conf.getInt(CALL_QUEUE_CODEL_TARGET_DELAY,
451      CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY);
452    int codelInterval = conf.getInt(CALL_QUEUE_CODEL_INTERVAL, CALL_QUEUE_CODEL_DEFAULT_INTERVAL);
453    double codelLifoThreshold = conf.getDouble(CALL_QUEUE_CODEL_LIFO_THRESHOLD,
454      CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD);
455
456    for (BlockingQueue<CallRunner> queue : queues) {
457      if (queue instanceof AdaptiveLifoCoDelCallQueue) {
458        ((AdaptiveLifoCoDelCallQueue) queue).updateTunables(codelTargetDelay, codelInterval,
459          codelLifoThreshold);
460      }
461    }
