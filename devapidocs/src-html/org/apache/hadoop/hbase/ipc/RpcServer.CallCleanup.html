245  /**
246   * Constructs a server listening on the named port and address.
247   * @param server hosting instance of {@link Server}. We will do authentications if an
248   * instance else pass null for no authentication check.
249   * @param name Used keying this rpc servers' metrics and for naming the Listener thread.
250   * @param services A list of services.
251   * @param bindAddress Where to listen
252   * @param conf
253   * @param scheduler
254   */
255  public RpcServer(final Server server, final String name,
256      final List<BlockingServiceAndInterface> services,
257      final InetSocketAddress bindAddress, Configuration conf,
258      RpcScheduler scheduler)
259      throws IOException {
260    if (conf.getBoolean("hbase.ipc.server.reservoir.enabled", true)) {
261      int poolBufSize = conf.getInt(ByteBufferPool.BUFFER_SIZE_KEY,
262          ByteBufferPool.DEFAULT_BUFFER_SIZE);
263      // The max number of buffers to be pooled in the ByteBufferPool. The default value been
264      // selected based on the #handlers configured. When it is read request, 2 MB is the max size
265      // at which we will send back one RPC request. Means max we need 2 MB for creating the
266      // response cell block. (Well it might be much lesser than this because in 2 MB size calc, we
267      // include the heap size overhead of each cells also.) Considering 2 MB, we will need
268      // (2 * 1024 * 1024) / poolBufSize buffers to make the response cell block. Pool buffer size
269      // is by default 64 KB.
270      // In case of read request, at the end of the handler process, we will make the response
271      // cellblock and add the Call to connection's response Q and a single Responder thread takes
272      // connections and responses from that one by one and do the socket write. So there is chances
273      // that by the time a handler originated response is actually done writing to socket and so
274      // released the BBs it used, the handler might have processed one more read req. On an avg 2x
275      // we consider and consider that also for the max buffers to pool
276      int bufsForTwoMB = (2 * 1024 * 1024) / poolBufSize;
277      int maxPoolSize = conf.getInt(ByteBufferPool.MAX_POOL_SIZE_KEY,
278          conf.getInt(HConstants.REGION_SERVER_HANDLER_COUNT,
279              HConstants.DEFAULT_REGION_SERVER_HANDLER_COUNT) * bufsForTwoMB * 2);
280      this.reservoir = new ByteBufferPool(poolBufSize, maxPoolSize);
281      this.minSizeForReservoirUse = getMinSizeForReservoirUse(this.reservoir);
282    } else {
283      reservoir = null;
284      this.minSizeForReservoirUse = Integer.MAX_VALUE;// reservoir itself not in place.
285    }
286    this.server = server;
287    this.services = services;
288    this.bindAddress = bindAddress;
289    this.conf = conf;
290    // See declaration above for documentation on what this size is.
291    this.maxQueueSizeInBytes =
292      this.conf.getLong("hbase.ipc.server.max.callqueue.size", DEFAULT_MAX_CALLQUEUE_SIZE);
293
294    this.warnResponseTime = conf.getInt(WARN_RESPONSE_TIME, DEFAULT_WARN_RESPONSE_TIME);
295    this.warnResponseSize = conf.getInt(WARN_RESPONSE_SIZE, DEFAULT_WARN_RESPONSE_SIZE);
296    this.minClientRequestTimeout = conf.getInt(MIN_CLIENT_REQUEST_TIMEOUT,
297        DEFAULT_MIN_CLIENT_REQUEST_TIMEOUT);
298    this.maxRequestSize = conf.getInt(MAX_REQUEST_SIZE, DEFAULT_MAX_REQUEST_SIZE);
299
300    this.metrics = new MetricsHBaseServer(name, new MetricsHBaseServerWrapperImpl(this));
301    this.tcpNoDelay = conf.getBoolean("hbase.ipc.server.tcpnodelay", true);
302    this.tcpKeepAlive = conf.getBoolean("hbase.ipc.server.tcpkeepalive", true);
303
304    this.cellBlockBuilder = new CellBlockBuilder(conf);
305
306    this.authorize = conf.getBoolean(HADOOP_SECURITY_AUTHORIZATION, false);
307    this.userProvider = UserProvider.instantiate(conf);
308    this.isSecurityEnabled = userProvider.isHBaseSecurityEnabled();
309    if (isSecurityEnabled) {
310      HBaseSaslRpcServer.init(conf);
311    }
312
313    this.scheduler = scheduler;
314  }

380  /**
381   * This is a server side method, which is invoked over RPC. On success
382   * the return response has protobuf response payload. On failure, the
383   * exception name and the stack trace are returned in the protobuf response.
384   */
385  @Override
386  public Pair<Message, CellScanner> call(RpcCall call,
387      MonitoredRPCHandler status) throws IOException {
388    try {
389      MethodDescriptor md = call.getMethod();
390      Message param = call.getParam();
391      status.setRPC(md.getName(), new Object[]{param},
392        call.getReceiveTime());
393      // TODO: Review after we add in encoded data blocks.
394      status.setRPCPacket(param);
395      status.resume("Servicing call");
396      //get an instance of the method arg type
397      HBaseRpcController controller = new HBaseRpcControllerImpl(call.getCellScanner());
398      controller.setCallTimeout(call.getTimeout());
399      Message result = call.getService().callBlockingMethod(md, controller, param);
400      long receiveTime = call.getReceiveTime();
401      long startTime = call.getStartTime();
402      long endTime = System.currentTimeMillis();
403      int processingTime = (int) (endTime - startTime);
404      int qTime = (int) (startTime - receiveTime);
405      int totalTime = (int) (endTime - receiveTime);
406      if (LOG.isTraceEnabled()) {
407        LOG.trace(CurCall.get().toString() +
408            ", response " + TextFormat.shortDebugString(result) +
409            " queueTime: " + qTime +
410            " processingTime: " + processingTime +
411            " totalTime: " + totalTime);
412      }
413      // Use the raw request call size for now.
414      long requestSize = call.getSize();
415      long responseSize = result.getSerializedSize();
416      if (call.isClientCellBlockSupported()) {
417        // Include the payload size in HBaseRpcController
418        responseSize += call.getResponseCellSize();
419      }
420
421      metrics.dequeuedCall(qTime);
422      metrics.processedCall(processingTime);
423      metrics.totalCall(totalTime);
424      metrics.receivedRequest(requestSize);
425      metrics.sentResponse(responseSize);
426      // log any RPC responses that are slower than the configured warn
427      // response time or larger than configured warning size
428      boolean tooSlow = (processingTime > warnResponseTime && warnResponseTime > -1);
429      boolean tooLarge = (responseSize > warnResponseSize && warnResponseSize > -1);
430      if (tooSlow || tooLarge) {
431        // when tagging, we let TooLarge trump TooSmall to keep output simple
432        // note that large responses will often also be slow.
433        logResponse(param,
434            md.getName(), md.getName() + "(" + param.getClass().getName() + ")",
435            (tooLarge ? "TooLarge" : "TooSlow"),
436            status.getClient(), startTime, processingTime, qTime,
437            responseSize);
438      }
439      return new Pair<>(result, controller.cellScanner());
440    } catch (Throwable e) {
441      // The above callBlockingMethod will always return a SE.  Strip the SE wrapper before
442      // putting it on the wire.  Its needed to adhere to the pb Service Interface but we don't
443      // need to pass it over the wire.
444      if (e instanceof ServiceException) {
445        if (e.getCause() == null) {
446          LOG.debug("Caught a ServiceException with null cause", e);
447        } else {
448          e = e.getCause();
449        }
450      }
451
452      // increment the number of requests that were exceptions.
453      metrics.exception(e);
454
455      if (e instanceof LinkageError) throw new DoNotRetryIOException(e);
456      if (e instanceof IOException) throw (IOException)e;
457      LOG.error("Unexpected throwable object ", e);
458      throw new IOException(e.getMessage(), e);
459    }
460  }

462  /**
463   * Logs an RPC response to the LOG file, producing valid JSON objects for
464   * client Operations.
465   * @param param The parameters received in the call.
466   * @param methodName The name of the method invoked
467   * @param call The string representation of the call
468   * @param tag  The tag that will be used to indicate this event in the log.
469   * @param clientAddress   The address of the client who made this call.
470   * @param startTime       The time that the call was initiated, in ms.
471   * @param processingTime  The duration that the call took to run, in ms.
472   * @param qTime           The duration that the call spent on the queue
473   *                        prior to being initiated, in ms.
474   * @param responseSize    The size in bytes of the response buffer.
475   */
476  void logResponse(Message param, String methodName, String call, String tag,
477      String clientAddress, long startTime, int processingTime, int qTime,
478      long responseSize) throws IOException {
479    // base information that is reported regardless of type of call
480    Map<String, Object> responseInfo = new HashMap<>();
481    responseInfo.put("starttimems", startTime);
482    responseInfo.put("processingtimems", processingTime);
483    responseInfo.put("queuetimems", qTime);
484    responseInfo.put("responsesize", responseSize);
485    responseInfo.put("client", clientAddress);
486    responseInfo.put("class", server == null? "": server.getClass().getSimpleName());
487    responseInfo.put("method", methodName);
488    responseInfo.put("call", call);
489    responseInfo.put("param", ProtobufUtil.getShortTextFormat(param));
490    if (param instanceof ClientProtos.ScanRequest && rsRpcServices != null) {
491      ClientProtos.ScanRequest request = ((ClientProtos.ScanRequest) param);
492      if (request.hasScannerId()) {
493        long scannerId = request.getScannerId();
494        String scanDetails = rsRpcServices.getScanDetailsWithId(scannerId);
495        if (scanDetails != null) {
496          responseInfo.put("scandetails", scanDetails);
497        }
498      }
499    }
500    LOG.warn("(response" + tag + "): " + MAPPER.writeValueAsString(responseInfo));
501  }

548  /**
549   * When the read or write buffer size is larger than this limit, i/o will be
550   * done in chunks of this size. Most RPC requests and responses would be
551   * be smaller.
552   */
553  protected static final int NIO_BUFFER_LIMIT = 64 * 1024; //should not be more than 64KB.
554
555  /**
556   * This is a wrapper around {@link java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)}.
557   * If the amount of data is large, it writes to channel in smaller chunks.
558   * This is to avoid jdk from creating many direct buffers as the size of
559   * ByteBuffer increases. There should not be any performance degredation.
560   *
561   * @param channel writable byte channel to write on
562   * @param buffer buffer to write
563   * @return number of bytes written
564   * @throws java.io.IOException e
565   * @see java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)
566   */
567  protected int channelRead(ReadableByteChannel channel,
568                                   ByteBuffer buffer) throws IOException {
569
570    int count = (buffer.remaining() <= NIO_BUFFER_LIMIT) ?
571           channel.read(buffer) : channelIO(channel, null, buffer);
572    if (count > 0) {
573      metrics.receivedBytes(count);
574    }
575    return count;
576  }

578  /**
579   * Helper for {@link #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)}
580   * and {@link #channelWrite(GatheringByteChannel, BufferChain)}. Only
581   * one of readCh or writeCh should be non-null.
582   *
583   * @param readCh read channel
584   * @param writeCh write channel
585   * @param buf buffer to read or write into/out of
586   * @return bytes written
587   * @throws java.io.IOException e
588   * @see #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)
589   * @see #channelWrite(GatheringByteChannel, BufferChain)
590   */
591  private static int channelIO(ReadableByteChannel readCh,
592                               WritableByteChannel writeCh,
593                               ByteBuffer buf) throws IOException {
594
595    int originalLimit = buf.limit();
596    int initialRemaining = buf.remaining();
597    int ret = 0;
598
599    while (buf.remaining() > 0) {
600      try {
601        int ioSize = Math.min(buf.remaining(), NIO_BUFFER_LIMIT);
602        buf.limit(buf.position() + ioSize);
603
604        ret = (readCh == null) ? writeCh.write(buf) : readCh.read(buf);
605
606        if (ret < ioSize) {
607          break;
608        }
609
610      } finally {
611        buf.limit(originalLimit);
612      }
613    }
614
615    int nBytes = initialRemaining - buf.remaining();
616    return (nBytes > 0) ? nBytes : ret;
617  }

619  /**
620   * This is extracted to a static method for better unit testing. We try to get buffer(s) from pool
621   * as much as possible.
622   *
623   * @param pool The ByteBufferPool to use
624   * @param minSizeForPoolUse Only for buffer size above this, we will try to use pool. Any buffer
625   *           need of size below this, create on heap ByteBuffer.
626   * @param reqLen Bytes count in request
627   */
628  @VisibleForTesting
629  static Pair<ByteBuff, CallCleanup> allocateByteBuffToReadInto(ByteBufferPool pool,
630      int minSizeForPoolUse, int reqLen) {
631    ByteBuff resultBuf;
632    List<ByteBuffer> bbs = new ArrayList<>((reqLen / pool.getBufferSize()) + 1);
633    int remain = reqLen;
634    ByteBuffer buf = null;
635    while (remain >= minSizeForPoolUse && (buf = pool.getBuffer()) != null) {
636      bbs.add(buf);
637      remain -= pool.getBufferSize();
638    }
639    ByteBuffer[] bufsFromPool = null;
640    if (bbs.size() > 0) {
641      bufsFromPool = new ByteBuffer[bbs.size()];
642      bbs.toArray(bufsFromPool);
643    }
644    if (remain > 0) {
645      bbs.add(ByteBuffer.allocate(remain));
646    }
647    if (bbs.size() > 1) {
648      ByteBuffer[] items = new ByteBuffer[bbs.size()];
649      bbs.toArray(items);
650      resultBuf = new MultiByteBuff(items);
651    } else {
652      // We are backed by single BB
653      resultBuf = new SingleByteBuff(bbs.get(0));
654    }
655    resultBuf.limit(reqLen);
656    if (bufsFromPool != null) {
657      final ByteBuffer[] bufsFromPoolFinal = bufsFromPool;
658      return new Pair<>(resultBuf, () -> {
659        // Return back all the BBs to pool
660        for (int i = 0; i < bufsFromPoolFinal.length; i++) {
661          pool.putbackBuffer(bufsFromPoolFinal[i]);
662        }
663      });
664    }
665    return new Pair<>(resultBuf, null);
666  }
