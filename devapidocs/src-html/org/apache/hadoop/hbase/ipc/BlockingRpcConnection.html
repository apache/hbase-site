118  /**
119   * If the client wants to interrupt its calls easily (i.e. call Thread#interrupt), it gets into a
120   * java issue: an interruption during a write closes the socket/channel. A way to avoid this is to
121   * use a different thread for writing. This way, on interruptions, we either cancel the writes or
122   * ignore the answer if the write is already done, but we don't stop the write in the middle. This
123   * adds a thread per region server in the client, so it's kept as an option.
124   * <p>
125   * The implementation is simple: the client threads adds their call to the queue, and then wait
126   * for an answer. The CallSender blocks on the queue, and writes the calls one after the other. On
127   * interruption, the client cancels its call. The CallSender checks that the call has not been
128   * canceled before writing it.
129   * </p>
130   * When the connection closes, all the calls not yet sent are dismissed. The client thread is
131   * notified with an appropriate exception, as if the call was already sent but the answer not yet
132   * received.
133   * </p>
134   */
135  private class CallSender extends Thread {
136
137    private final Queue<Call> callsToWrite;
138
139    private final int maxQueueSize;
140
141    public CallSender(String name, Configuration conf) {
142      int queueSize = conf.getInt("hbase.ipc.client.write.queueSize", 1000);
143      callsToWrite = new ArrayDeque<>(queueSize);
144      this.maxQueueSize = queueSize;
145      setDaemon(true);
146      setName(name + " - writer");
147    }
148
149    public void sendCall(final Call call) throws IOException {
150      if (callsToWrite.size() >= maxQueueSize) {
151        throw new IOException("Can't add the call " + call.id
152            + " to the write queue. callsToWrite.size()=" + callsToWrite.size());
153      }
154      callsToWrite.offer(call);
155      BlockingRpcConnection.this.notifyAll();
156    }
157
158    public void remove(Call call) {
159      callsToWrite.remove(call);
160      // By removing the call from the expected call list, we make the list smaller, but
161      // it means as well that we don't know how many calls we cancelled.
162      calls.remove(call.id);
163      call.setException(new CallCancelledException("Call id=" + call.id + ", waitTime="
164          + (EnvironmentEdgeManager.currentTime() - call.getStartTime()) + ", rpcTimeout="
165          + call.timeout));
166    }
167
168    /**
169     * Reads the call from the queue, write them on the socket.
170     */
171    @Override
172    public void run() {
173      synchronized (BlockingRpcConnection.this) {
174        while (!closed) {
175          if (callsToWrite.isEmpty()) {
176            // We should use another monitor object here for better performance since the read
177            // thread also uses ConnectionImpl.this. But this makes the locking schema more
178            // complicated, can do it later as an optimization.
179            try {
180              BlockingRpcConnection.this.wait();
181            } catch (InterruptedException e) {
182            }
183            // check if we need to quit, so continue the main loop instead of fallback.
184            continue;
185          }
186          Call call = callsToWrite.poll();
187          if (call.isDone()) {
188            continue;
189          }
190          try {
191            tracedWriteRequest(call);
192          } catch (IOException e) {
193            // exception here means the call has not been added to the pendingCalls yet, so we need
194            // to fail it by our own.
195            if (LOG.isDebugEnabled()) {
196              LOG.debug("call write error for call #" + call.id, e);
197            }
198            call.setException(e);
199            closeConn(e);
200          }
201        }
202      }
203    }
204
205    /**
206     * Cleans the call not yet sent when we finish.
207     */
208    public void cleanup(IOException e) {
209      IOException ie = new ConnectionClosingException(
210          "Connection to " + remoteId.address + " is closing.");
211      for (Call call : callsToWrite) {
212        call.setException(ie);
213      }
214      callsToWrite.clear();
215    }
216  }

218  BlockingRpcConnection(BlockingRpcClient rpcClient, ConnectionId remoteId) throws IOException {
219    super(rpcClient.conf, AbstractRpcClient.WHEEL_TIMER, remoteId, rpcClient.clusterId,
220        rpcClient.userProvider.isHBaseSecurityEnabled(), rpcClient.codec, rpcClient.compressor);
221    this.rpcClient = rpcClient;
222    if (remoteId.getAddress().isUnresolved()) {
223      throw new UnknownHostException("unknown host: " + remoteId.getAddress().getHostName());
224    }
225
226    this.connectionHeaderPreamble = getConnectionHeaderPreamble();
227    ConnectionHeader header = getConnectionHeader();
228    ByteArrayOutputStream baos = new ByteArrayOutputStream(4 + header.getSerializedSize());
229    DataOutputStream dos = new DataOutputStream(baos);
230    dos.writeInt(header.getSerializedSize());
231    header.writeTo(dos);
232    assert baos.size() == 4 + header.getSerializedSize();
233    this.connectionHeaderWithLength = baos.getBuffer();
234
235    UserGroupInformation ticket = remoteId.ticket.getUGI();
236    this.threadName = "IPC Client (" + this.rpcClient.socketFactory.hashCode() + ") connection to "
237        + remoteId.getAddress().toString()
238        + ((ticket == null) ? " from an unknown user" : (" from " + ticket.getUserName()));
239
240    if (this.rpcClient.conf.getBoolean(BlockingRpcClient.SPECIFIC_WRITE_THREAD, false)) {
241      callSender = new CallSender(threadName, this.rpcClient.conf);
242      callSender.start();
243    } else {
244      callSender = null;
245    }
246  }

248  // protected for write UT.
249  protected void setupConnection() throws IOException {
250    short ioFailures = 0;
251    short timeoutFailures = 0;
252    while (true) {
253      try {
254        this.socket = this.rpcClient.socketFactory.createSocket();
255        this.socket.setTcpNoDelay(this.rpcClient.isTcpNoDelay());
256        this.socket.setKeepAlive(this.rpcClient.tcpKeepAlive);
257        if (this.rpcClient.localAddr != null) {
258          this.socket.bind(this.rpcClient.localAddr);
259        }
260        NetUtils.connect(this.socket, remoteId.getAddress(), this.rpcClient.connectTO);
261        this.socket.setSoTimeout(this.rpcClient.readTO);
262        return;
263      } catch (SocketTimeoutException toe) {
264        /*
265         * The max number of retries is 45, which amounts to 20s*45 = 15 minutes retries.
266         */
267        handleConnectionFailure(timeoutFailures++, this.rpcClient.maxRetries, toe);
268      } catch (IOException ie) {
269        handleConnectionFailure(ioFailures++, this.rpcClient.maxRetries, ie);
270      }
271    }
272  }
273
274  /**
275   * Handle connection failures If the current number of retries is equal to the max number of
276   * retries, stop retrying and throw the exception; Otherwise backoff N seconds and try connecting
277   * again. This Method is only called from inside setupIOstreams(), which is synchronized. Hence
278   * the sleep is synchronized; the locks will be retained.
279   * @param curRetries current number of retries
280   * @param maxRetries max number of retries allowed
281   * @param ioe failure reason
282   * @throws IOException if max number of retries is reached
283   */
284  private void handleConnectionFailure(int curRetries, int maxRetries, IOException ioe)
285      throws IOException {
286    closeSocket();
287
288    // throw the exception if the maximum number of retries is reached
289    if (curRetries >= maxRetries || ExceptionUtil.isInterrupt(ioe)) {
290      throw ioe;
291    }
292
293    // otherwise back off and retry
294    try {
295      Thread.sleep(this.rpcClient.failureSleep);
296    } catch (InterruptedException ie) {
297      ExceptionUtil.rethrowIfInterrupt(ie);
298    }
299
300    LOG.info("Retrying connect to server: " + remoteId.getAddress() + " after sleeping "
301        + this.rpcClient.failureSleep + "ms. Already tried " + curRetries + " time(s).");
302  }
303
304  /*
305   * wait till someone signals us to start reading RPC response or it is idle too long, it is marked
306   * as to be closed, or the client is marked as not running.
307   * @return true if it is time to read a response; false otherwise.
308   */
309  private synchronized boolean waitForWork() {
310    // beware of the concurrent access to the calls list: we can add calls, but as well
311    // remove them.
312    long waitUntil = EnvironmentEdgeManager.currentTime() + this.rpcClient.minIdleTimeBeforeClose;
313    for (;;) {
314      if (thread == null) {
315        return false;
316      }
317      if (!calls.isEmpty()) {
318        return true;
319      }
320      if (EnvironmentEdgeManager.currentTime() >= waitUntil) {
321        closeConn(
322          new IOException("idle connection closed with " + calls.size() + " pending request(s)"));
323        return false;
324      }
325      try {
326        wait(Math.min(this.rpcClient.minIdleTimeBeforeClose, 1000));
327      } catch (InterruptedException e) {
328      }
329    }
330  }

332  @Override
333  public void run() {
334    if (LOG.isTraceEnabled()) {
335      LOG.trace(threadName + ": starting, connections " + this.rpcClient.connections.size());
336    }
337    while (waitForWork()) {
338      readResponse();
339    }
340    if (LOG.isTraceEnabled()) {
341      LOG.trace(threadName + ": stopped, connections " + this.rpcClient.connections.size());
342    }
343  }

361  /**
362   * If multiple clients with the same principal try to connect to the same server at the same time,
363   * the server assumes a replay attack is in progress. This is a feature of kerberos. In order to
364   * work around this, what is done is that the client backs off randomly and tries to initiate the
365   * connection again. The other problem is to do with ticket expiry. To handle that, a relogin is
366   * attempted.
367   * <p>
368   * The retry logic is governed by the {@link #shouldAuthenticateOverKrb} method. In case when the
369   * user doesn't have valid credentials, we don't need to retry (from cache or ticket). In such
370   * cases, it is prudent to throw a runtime exception when we receive a SaslException from the
371   * underlying authentication implementation, so there is no retry from other high level (for eg,
372   * HCM or HBaseAdmin).
373   * </p>
374   */
375  private void handleSaslConnectionFailure(final int currRetries, final int maxRetries,
376      final Exception ex, final UserGroupInformation user)
377      throws IOException, InterruptedException {
378    closeSocket();
379    user.doAs(new PrivilegedExceptionAction<Object>() {
380      @Override
381      public Object run() throws IOException, InterruptedException {
382        if (shouldAuthenticateOverKrb()) {
383          if (currRetries < maxRetries) {
384            if (LOG.isDebugEnabled()) {
385              LOG.debug("Exception encountered while connecting to " + "the server : " + ex);
386            }
387            // try re-login
388            relogin();
389            disposeSasl();
390            // have granularity of milliseconds
391            // we are sleeping with the Connection lock held but since this
392            // connection instance is being used for connecting to the server
393            // in question, it is okay
394            Thread.sleep(ThreadLocalRandom.current().nextInt(reloginMaxBackoff) + 1);
395            return null;
396          } else {
397            String msg = "Couldn't setup connection for "
398                + UserGroupInformation.getLoginUser().getUserName() + " to " + serverPrincipal;
399            LOG.warn(msg, ex);
400            throw (IOException) new IOException(msg).initCause(ex);
401          }
402        } else {
403          LOG.warn("Exception encountered while connecting to " + "the server : " + ex);
404        }
405        if (ex instanceof RemoteException) {
406          throw (RemoteException) ex;
407        }
408        if (ex instanceof SaslException) {
409          String msg = "SASL authentication failed."
410              + " The most likely cause is missing or invalid credentials." + " Consider 'kinit'.";
411          LOG.fatal(msg, ex);
412          throw new RuntimeException(msg, ex);
413        }
414        throw new IOException(ex);
415      }
416    });
417  }

419  private void setupIOstreams() throws IOException {
420    if (socket != null) {
421      // The connection is already available. Perfect.
422      return;
423    }
424
425    if (this.rpcClient.failedServers.isFailedServer(remoteId.getAddress())) {
426      if (LOG.isDebugEnabled()) {
427        LOG.debug("Not trying to connect to " + remoteId.address
428            + " this server is in the failed servers list");
429      }
430      throw new FailedServerException(
431          "This server is in the failed servers list: " + remoteId.address);
432    }
433
434    try {
435      if (LOG.isDebugEnabled()) {
436        LOG.debug("Connecting to " + remoteId.address);
437      }
438
439      short numRetries = 0;
440      final short MAX_RETRIES = 5;
441      while (true) {
442        setupConnection();
443        InputStream inStream = NetUtils.getInputStream(socket);
444        // This creates a socket with a write timeout. This timeout cannot be changed.
445        OutputStream outStream = NetUtils.getOutputStream(socket, this.rpcClient.writeTO);
446        // Write out the preamble -- MAGIC, version, and auth to use.
447        writeConnectionHeaderPreamble(outStream);
448        if (useSasl) {
449          final InputStream in2 = inStream;
450          final OutputStream out2 = outStream;
451          UserGroupInformation ticket = getUGI();
452          boolean continueSasl;
453          if (ticket == null) {
454            throw new FatalConnectionException("ticket/user is null");
455          }
456          try {
457            continueSasl = ticket.doAs(new PrivilegedExceptionAction<Boolean>() {
458              @Override
459              public Boolean run() throws IOException {
460                return setupSaslConnection(in2, out2);
461              }
462            });
463          } catch (Exception ex) {
464            ExceptionUtil.rethrowIfInterrupt(ex);
465            handleSaslConnectionFailure(numRetries++, MAX_RETRIES, ex, ticket);
466            continue;
467          }
468          if (continueSasl) {
469            // Sasl connect is successful. Let's set up Sasl i/o streams.
470            inStream = saslRpcClient.getInputStream();
471            outStream = saslRpcClient.getOutputStream();
472          } else {
473            // fall back to simple auth because server told us so.
474            // do not change authMethod and useSasl here, we should start from secure when
475            // reconnecting because regionserver may change its sasl config after restart.
476          }
477        }
478        this.in = new DataInputStream(new BufferedInputStream(inStream));
479        this.out = new DataOutputStream(new BufferedOutputStream(outStream));
480        // Now write out the connection header
481        writeConnectionHeader();
482        // process the response from server for connection header if necessary
483        processResponseForConnectionHeader();
484
485        break;
486      }
487    } catch (Throwable t) {
488      closeSocket();
489      IOException e = ExceptionUtil.asInterrupt(t);
490      if (e == null) {
491        this.rpcClient.failedServers.addToFailedServers(remoteId.address);
492        if (t instanceof LinkageError) {
493          // probably the hbase hadoop version does not match the running hadoop version
494          e = new DoNotRetryIOException(t);
495        } else if (t instanceof IOException) {
496          e = (IOException) t;
497        } else {
498          e = new IOException("Could not set up IO Streams to " + remoteId.address, t);
499        }
500      }
501      throw e;
502    }
503
504    // start the receiver thread after the socket connection has been set up
505    thread = new Thread(this, threadName);
506    thread.setDaemon(true);
507    thread.start();
508  }

510  /**
511   * Write the RPC header: {@code <MAGIC WORD -- 'HBas'> <ONEBYTE_VERSION> <ONEBYTE_AUTH_TYPE>}
512   */
513  private void writeConnectionHeaderPreamble(OutputStream out) throws IOException {
514    out.write(connectionHeaderPreamble);
515    out.flush();
516  }

518  /**
519   * Write the connection header.
520   */
521  private void writeConnectionHeader() throws IOException {
522    boolean isCryptoAesEnable = false;
523    // check if Crypto AES is enabled
524    if (saslRpcClient != null) {
525      boolean saslEncryptionEnabled = SaslUtil.QualityOfProtection.PRIVACY.
526          getSaslQop().equalsIgnoreCase(saslRpcClient.getSaslQOP());
527      isCryptoAesEnable = saslEncryptionEnabled && conf.getBoolean(
528          CRYPTO_AES_ENABLED_KEY, CRYPTO_AES_ENABLED_DEFAULT);
529    }
530
531    // if Crypto AES is enabled, set transformation and negotiate with server
532    if (isCryptoAesEnable) {
533      waitingConnectionHeaderResponse = true;
534    }
535    this.out.write(connectionHeaderWithLength);
536    this.out.flush();
537  }
538
539  private void processResponseForConnectionHeader() throws IOException {
540    // if no response excepted, return
541    if (!waitingConnectionHeaderResponse) return;
542    try {
543      // read the ConnectionHeaderResponse from server
544      int len = this.in.readInt();
545      byte[] buff = new byte[len];
546      int readSize = this.in.read(buff);
547      if (LOG.isDebugEnabled()) {
548        LOG.debug("Length of response for connection header:" + readSize);
549      }
550
551      RPCProtos.ConnectionHeaderResponse connectionHeaderResponse =
552          RPCProtos.ConnectionHeaderResponse.parseFrom(buff);
553
554      // Get the CryptoCipherMeta, update the HBaseSaslRpcClient for Crypto Cipher
555      if (connectionHeaderResponse.hasCryptoCipherMeta()) {
556        negotiateCryptoAes(connectionHeaderResponse.getCryptoCipherMeta());
557      }
558      waitingConnectionHeaderResponse = false;
559    } catch (SocketTimeoutException ste) {
560      LOG.fatal("Can't get the connection header response for rpc timeout, please check if" +
561          " server has the correct configuration to support the additional function.", ste);
562      // timeout when waiting the connection header response, ignore the additional function
563      throw new IOException("Timeout while waiting connection header response", ste);
564    }
565  }
566
567  private void negotiateCryptoAes(RPCProtos.CryptoCipherMeta cryptoCipherMeta)
568      throws IOException {
569    // initilize the Crypto AES with CryptoCipherMeta
570    saslRpcClient.initCryptoCipher(cryptoCipherMeta, this.rpcClient.conf);
571    // reset the inputStream/outputStream for Crypto AES encryption
572    this.in = new DataInputStream(new BufferedInputStream(saslRpcClient.getInputStream()));
573    this.out = new DataOutputStream(new BufferedOutputStream(saslRpcClient.getOutputStream()));
574  }

582  /**
583   * Initiates a call by sending the parameter to the remote server. Note: this is not called from
584   * the Connection thread, but by other threads.
585   * @see #readResponse()
586   */
587  private void writeRequest(Call call) throws IOException {
588    ByteBuffer cellBlock = this.rpcClient.cellBlockBuilder.buildCellBlock(this.codec,
589      this.compressor, call.cells);
590    CellBlockMeta cellBlockMeta;
591    if (cellBlock != null) {
592      cellBlockMeta = CellBlockMeta.newBuilder().setLength(cellBlock.limit()).build();
593    } else {
594      cellBlockMeta = null;
595    }
596    RequestHeader requestHeader = buildRequestHeader(call, cellBlockMeta);
597
598    setupIOstreams();
599
600    // Now we're going to write the call. We take the lock, then check that the connection
601    // is still valid, and, if so we do the write to the socket. If the write fails, we don't
602    // know where we stand, we have to close the connection.
603    if (Thread.interrupted()) {
604      throw new InterruptedIOException();
605    }
606
607    calls.put(call.id, call); // We put first as we don't want the connection to become idle.
608    // from here, we do not throw any exception to upper layer as the call has been tracked in the
609    // pending calls map.
610    try {
611      call.callStats.setRequestSizeBytes(write(this.out, requestHeader, call.param, cellBlock));
612    } catch (IOException e) {
613      closeConn(e);
614      return;
615    }
616    notifyAll();
617  }

619  /*
620   * Receive a response. Because only one receiver, so no synchronization on in.
621   */
622  private void readResponse() {
623    Call call = null;
624    boolean expectedCall = false;
625    try {
626      // See HBaseServer.Call.setResponse for where we write out the response.
627      // Total size of the response. Unused. But have to read it in anyways.
628      int totalSize = in.readInt();
629
630      // Read the header
631      ResponseHeader responseHeader = ResponseHeader.parseDelimitedFrom(in);
632      int id = responseHeader.getCallId();
633      call = calls.remove(id); // call.done have to be set before leaving this method
634      expectedCall = (call != null && !call.isDone());
635      if (!expectedCall) {
636        // So we got a response for which we have no corresponding 'call' here on the client-side.
637        // We probably timed out waiting, cleaned up all references, and now the server decides
638        // to return a response. There is nothing we can do w/ the response at this stage. Clean
639        // out the wire of the response so its out of the way and we can get other responses on
640        // this connection.
641        int readSoFar = getTotalSizeWhenWrittenDelimited(responseHeader);
642        int whatIsLeftToRead = totalSize - readSoFar;
643        IOUtils.skipFully(in, whatIsLeftToRead);
644        if (call != null) {
645          call.callStats.setResponseSizeBytes(totalSize);
646          call.callStats
647              .setCallTimeMs(EnvironmentEdgeManager.currentTime() - call.callStats.getStartTime());
648        }
649        return;
650      }
651      if (responseHeader.hasException()) {
652        ExceptionResponse exceptionResponse = responseHeader.getException();
653        RemoteException re = createRemoteException(exceptionResponse);
654        call.setException(re);
655        call.callStats.setResponseSizeBytes(totalSize);
656        call.callStats
657            .setCallTimeMs(EnvironmentEdgeManager.currentTime() - call.callStats.getStartTime());
658        if (isFatalConnectionException(exceptionResponse)) {
659          synchronized (this) {
660            closeConn(re);
661          }
662        }
663      } else {
664        Message value = null;
665        if (call.responseDefaultType != null) {
666          Builder builder = call.responseDefaultType.newBuilderForType();
667          ProtobufUtil.mergeDelimitedFrom(builder, in);
668          value = builder.build();
669        }
670        CellScanner cellBlockScanner = null;
671        if (responseHeader.hasCellBlockMeta()) {
672          int size = responseHeader.getCellBlockMeta().getLength();
673          byte[] cellBlock = new byte[size];
674          IOUtils.readFully(this.in, cellBlock, 0, cellBlock.length);
675          cellBlockScanner = this.rpcClient.cellBlockBuilder.createCellScanner(this.codec,
676            this.compressor, cellBlock);
677        }
678        call.setResponse(value, cellBlockScanner);
679        call.callStats.setResponseSizeBytes(totalSize);
680        call.callStats
681            .setCallTimeMs(EnvironmentEdgeManager.currentTime() - call.callStats.getStartTime());
682      }
683    } catch (IOException e) {
684      if (expectedCall) {
685        call.setException(e);
686      }
687      if (e instanceof SocketTimeoutException) {
688        // Clean up open calls but don't treat this as a fatal condition,
689        // since we expect certain responses to not make it by the specified
690        // {@link ConnectionId#rpcTimeout}.
691        if (LOG.isTraceEnabled()) {
692          LOG.trace("ignored", e);
693        }
694      } else {
695        synchronized (this) {
696          closeConn(e);
697        }
698      }
699    }
700  }
701
702  @Override
703  protected synchronized void callTimeout(Call call) {
704    // call sender
705    calls.remove(call.id);
706  }
707
708  // just close socket input and output.
709  private void closeSocket() {
710    IOUtils.closeStream(out);
711    IOUtils.closeStream(in);
712    IOUtils.closeSocket(socket);
713    out = null;
714    in = null;
715    socket = null;
716  }
717
718  // close socket, reader, and clean up all pending calls.
719  private void closeConn(IOException e) {
720    if (thread == null) {
721      return;
722    }
723    thread.interrupt();
724    thread = null;
725    closeSocket();
726    if (callSender != null) {
727      callSender.cleanup(e);
728    }
729    for (Call call : calls.values()) {
730      call.setException(e);
731    }
732    calls.clear();
733  }
734
735  // release all resources, the connection will not be used any more.
736  @Override
737  public synchronized void shutdown() {
738    closed = true;
739    if (callSender != null) {
740      callSender.interrupt();
741    }
742    closeConn(new IOException("connection to " + remoteId.address + " closed"));
743  }

750  @Override
751  public synchronized void sendRequest(final Call call, HBaseRpcController pcrc)
752      throws IOException {
753    pcrc.notifyOnCancel(new RpcCallback<Object>() {
754
755      @Override
756      public void run(Object parameter) {
757        setCancelled(call);
758        synchronized (BlockingRpcConnection.this) {
759          if (callSender != null) {
760            callSender.remove(call);
761          } else {
762            calls.remove(call.id);
763          }
764        }
765      }
766    }, new CancellationCallback() {
767
768      @Override
769      public void run(boolean cancelled) throws IOException {
770        if (cancelled) {
771          setCancelled(call);
772          return;
773        }
774        scheduleTimeoutTask(call);
775        if (callSender != null) {
776          callSender.sendCall(call);
777        } else {
778          tracedWriteRequest(call);
779        }
780      }
781    });
782  }

784  @Override
785  public synchronized boolean isActive() {
786    return thread != null;
787  }
