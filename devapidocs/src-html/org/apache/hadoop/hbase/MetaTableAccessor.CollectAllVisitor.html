075/**
076 * Read/write operations on region and assignment information store in
077 * <code>hbase:meta</code>.
078 *
079 * Some of the methods of this class take ZooKeeperWatcher as a param. The only reason
080 * for this is because when used on client-side (like from HBaseAdmin), we want to use
081 * short-living connection (opened before each operation, closed right after), while
082 * when used on HM or HRS (like in AssignmentManager) we want permanent connection.
083 */

087  /*
088   * HBASE-10070 adds a replicaId to HRI, meaning more than one HRI can be defined for the
089   * same table range (table, startKey, endKey). For every range, there will be at least one
090   * HRI defined which is called default replica.
091   *
092   * Meta layout (as of 0.98 + HBASE-10070) is like:
093   *
094   * For each table there is single row in column family 'table' formatted:
095   * <tableName> including namespace and columns are:
096   * table: state             => contains table state
097   *
098   * For each table range, there is a single row, formatted like:
099   * <tableName>,<startKey>,<regionId>,<encodedRegionName>. This row corresponds to the regionName
100   * of the default region replica.
101   * Columns are:
102   * info:regioninfo         => contains serialized HRI for the default region replica
103   * info:server             => contains hostname:port (in string form) for the server hosting
104   *                            the default regionInfo replica
105   * info:server_<replicaId> => contains hostname:port (in string form) for the server hosting the
106   *                            regionInfo replica with replicaId
107   * info:serverstartcode    => contains server start code (in binary long form) for the server
108   *                            hosting the default regionInfo replica
109   * info:serverstartcode_<replicaId> => contains server start code (in binary long form) for the
110   *                                     server hosting the regionInfo replica with replicaId
111   * info:seqnumDuringOpen    => contains seqNum (in binary long form) for the region at the time
112   *                             the server opened the region with default replicaId
113   * info:seqnumDuringOpen_<replicaId> => contains seqNum (in binary long form) for the region at
114   *                             the time the server opened the region with replicaId
115   * info:splitA              => contains a serialized HRI for the first daughter region if the
116   *                             region is split
117   * info:splitB              => contains a serialized HRI for the second daughter region if the
118   *                             region is split
119   * info:mergeA              => contains a serialized HRI for the first parent region if the
120   *                             region is the result of a merge
121   * info:mergeB              => contains a serialized HRI for the second parent region if the
122   *                             region is the result of a merge
123   * The actual layout of meta should be encapsulated inside MetaTableAccessor methods,
124   * and should not leak out of it (through Result objects, etc)
125   *
126   * For replication serially, there are three column families "rep_barrier", "rep_position" and
127   * "rep_meta" whose row key is encodedRegionName.
128   * rep_barrier:{seqid}      => in each time a RS opens a region, it saves the open sequence
129   *                                  id in this region
130   * rep_position:{peerid}    => to save the max sequence id we have pushed for each peer
131   * rep_meta:_TABLENAME_     => a special cell to save this region's table name, will used when
132   *                             we clean old data
133   * rep_meta:_DAUGHTER_      => a special cell to present this region is split or merged, in this
134   *                             cell the value is merged encoded name or two split encoded names
135   *                             separated by ","
136   * rep_meta:_PARENT_        => a special cell to present this region's parent region(s), in this
137   *                             cell the value is encoded name of one or two parent regions
138   *                             separated by ","
139   */

496   * @param tableName table we're working with
497   * @return start row for scanning META according to query type
498   */
499  public static byte[] getTableStartRowForMeta(TableName tableName, QueryType type) {
500    if (tableName == null) {
501      return null;
502    }
503    switch (type) {
504    case REGION:
505      byte[] startRow = new byte[tableName.getName().length + 2];
506      System.arraycopy(tableName.getName(), 0, startRow, 0, tableName.getName().length);
507      startRow[startRow.length - 2] = HConstants.DELIMITER;
508      startRow[startRow.length - 1] = HConstants.DELIMITER;
509      return startRow;
510    case ALL:
511    case TABLE:
512    default:
513      return tableName.getName();
514    }
515  }

517  /**
518   * @param tableName table we're working with
519   * @return stop row for scanning META according to query type
520   */
521  public static byte[] getTableStopRowForMeta(TableName tableName, QueryType type) {
522    if (tableName == null) {
523      return null;
524    }
525    final byte[] stopRow;
526    switch (type) {
527    case REGION:
528      stopRow = new byte[tableName.getName().length + 3];
529      System.arraycopy(tableName.getName(), 0, stopRow, 0, tableName.getName().length);
530      stopRow[stopRow.length - 3] = ' ';
531      stopRow[stopRow.length - 2] = HConstants.DELIMITER;
532      stopRow[stopRow.length - 1] = HConstants.DELIMITER;
533      break;
534    case ALL:
535    case TABLE:
536    default:
537      stopRow = new byte[tableName.getName().length + 1];
538      System.arraycopy(tableName.getName(), 0, stopRow, 0, tableName.getName().length);
539      stopRow[stopRow.length - 1] = ' ';
540      break;
541    }
542    return stopRow;
543  }

567  private static Scan getMetaScan(Connection connection, int rowUpperLimit) {
568    Scan scan = new Scan();
569    int scannerCaching = connection.getConfiguration()
570        .getInt(HConstants.HBASE_META_SCANNER_CACHING,
571            HConstants.DEFAULT_HBASE_META_SCANNER_CACHING);
572    if (connection.getConfiguration().getBoolean(HConstants.USE_META_REPLICAS,
573        HConstants.DEFAULT_USE_META_REPLICAS)) {
574      scan.setConsistency(Consistency.TIMELINE);
575    }
576    if (rowUpperLimit > 0) {
577      scan.setLimit(rowUpperLimit);
578      scan.setReadType(Scan.ReadType.PREAD);
579    }
580    scan.setCaching(scannerCaching);
581    return scan;
582  }

596  /**
597   * Do not use this method to get meta table regions, use methods in MetaTableLocator instead.
598   * @param connection connection we're using
599   * @param tableName table to work with, can be null for getting all regions
600   * @param excludeOfflinedSplitParents don't return split parents
601   * @return Return list of regioninfos and server addresses.
602   * @throws IOException
603   */
604  public static List<Pair<HRegionInfo, ServerName>> getTableRegionsAndLocations(
605      Connection connection, @Nullable final TableName tableName,
606      final boolean excludeOfflinedSplitParents) throws IOException {
607    if (tableName != null && tableName.equals(TableName.META_TABLE_NAME)) {
608      throw new IOException("This method can't be used to locate meta regions;"
609        + " use MetaTableLocator instead");
610    }
611    // Make a version of CollectingVisitor that collects HRegionInfo and ServerAddress
612    CollectingVisitor<Pair<HRegionInfo, ServerName>> visitor =
613      new CollectingVisitor<Pair<HRegionInfo, ServerName>>() {
614        private RegionLocations current = null;
615
616        @Override
617        public boolean visit(Result r) throws IOException {
618          current = getRegionLocations(r);
619          if (current == null || current.getRegionLocation().getRegionInfo() == null) {
620            LOG.warn("No serialized HRegionInfo in " + r);
621            return true;
622          }
623          HRegionInfo hri = current.getRegionLocation().getRegionInfo();
624          if (excludeOfflinedSplitParents && hri.isSplitParent()) return true;
625          // Else call super and add this Result to the collection.
626          return super.visit(r);
627        }
628
629        @Override
630        void add(Result r) {
631          if (current == null) {
632            return;
633          }
634          for (HRegionLocation loc : current.getRegionLocations()) {
635            if (loc != null) {
636              this.results.add(new Pair<>(loc.getRegionInfo(), loc.getServerName()));
637            }
638          }
639        }
640      };
641    scanMeta(connection,
642        getTableStartRowForMeta(tableName, QueryType.REGION),
643        getTableStopRowForMeta(tableName, QueryType.REGION),
644        QueryType.REGION, visitor);
645    return visitor.getResults();
646  }

756  /**
757   * Performs a scan of META table.
758   * @param connection connection we're using
759   * @param startRow Where to start the scan. Pass null if want to begin scan
760   *                 at first row.
761   * @param stopRow Where to stop the scan. Pass null if want to scan all rows
762   *                from the start one
763   * @param type scanned part of meta
764   * @param maxRows maximum rows to return
765   * @param visitor Visitor invoked against each row.
766   * @throws IOException
767   */
768  public static void scanMeta(Connection connection,
769      @Nullable final byte[] startRow, @Nullable final byte[] stopRow,
770      QueryType type, int maxRows, final Visitor visitor)
771  throws IOException {
772    int rowUpperLimit = maxRows > 0 ? maxRows : Integer.MAX_VALUE;
773    Scan scan = getMetaScan(connection, rowUpperLimit);
774
775    for (byte[] family : type.getFamilies()) {
776      scan.addFamily(family);
777    }
778    if (startRow != null) scan.setStartRow(startRow);
779    if (stopRow != null) scan.setStopRow(stopRow);
780
781    if (LOG.isTraceEnabled()) {
782      LOG.trace("Scanning META"
783          + " starting at row=" + Bytes.toStringBinary(startRow)
784          + " stopping at row=" + Bytes.toStringBinary(stopRow)
785          + " for max=" + rowUpperLimit
786          + " with caching=" + scan.getCaching());
787    }
788
789    int currentRow = 0;
790    try (Table metaTable = getMetaHTable(connection)) {
791      try (ResultScanner scanner = metaTable.getScanner(scan)) {
792        Result data;
793        while ((data = scanner.next()) != null) {
794          if (data.isEmpty()) continue;
795          // Break if visit returns false.
796          if (!visitor.visit(data)) break;
797          if (++currentRow >= rowUpperLimit) break;
798        }
799      }
800    }
801    if (visitor != null && visitor instanceof Closeable) {
802      try {
803        ((Closeable) visitor).close();
804      } catch (Throwable t) {
805        ExceptionUtil.rethrowIfInterrupt(t);
806        LOG.debug("Got exception in closing the meta scanner visitor", t);
807      }
808    }
809  }

970  /**
971   * Returns an HRegionLocationList extracted from the result.
972   * @return an HRegionLocationList containing all locations for the region range or null if
973   *  we can't deserialize the result.
974   */
975  @Nullable
976  public static RegionLocations getRegionLocations(final Result r) {
977    if (r == null) return null;
978    HRegionInfo regionInfo = getHRegionInfo(r, getRegionInfoColumn());
979    if (regionInfo == null) return null;
980
981    List<HRegionLocation> locations = new ArrayList<>(1);
982    NavigableMap<byte[],NavigableMap<byte[],byte[]>> familyMap = r.getNoVersionMap();
983
984    locations.add(getRegionLocation(r, regionInfo, 0));
985
986    NavigableMap<byte[], byte[]> infoMap = familyMap.get(getCatalogFamily());
987    if (infoMap == null) return new RegionLocations(locations);
988
989    // iterate until all serverName columns are seen
990    int replicaId = 0;
991    byte[] serverColumn = getServerColumn(replicaId);
992    SortedMap<byte[], byte[]> serverMap = null;
993    serverMap = infoMap.tailMap(serverColumn, false);
994
995    if (serverMap.isEmpty()) return new RegionLocations(locations);
996
997    for (Map.Entry<byte[], byte[]> entry : serverMap.entrySet()) {
998      replicaId = parseReplicaIdFromServerColumn(entry.getKey());
999      if (replicaId < 0) {
1000        break;
1001      }
1002      HRegionLocation location = getRegionLocation(r, regionInfo, replicaId);
1003      // In case the region replica is newly created, it's location might be null. We usually do not
1004      // have HRL's in RegionLocations object with null ServerName. They are handled as null HRLs.
1005      if (location == null || location.getServerName() == null) {
1006        locations.add(null);
1007      } else {
1008        locations.add(location);
1009      }
1010    }
1011
1012    return new RegionLocations(locations);
1013  }

1630  /**
1631   * Merge the two regions into one in an atomic operation. Deletes the two
1632   * merging regions in hbase:meta and adds the merged region with the information of
1633   * two merging regions.
1634   * @param connection connection we're using
1635   * @param mergedRegion the merged region
1636   * @param regionA
1637   * @param regionB
1638   * @param sn the location of the region
1639   * @param masterSystemTime
1640   * @param saveBarrier true if need save replication barrier in meta, used for serial replication
1641   * @throws IOException
1642   */
1643  public static void mergeRegions(final Connection connection, HRegionInfo mergedRegion,
1644      HRegionInfo regionA, HRegionInfo regionB, ServerName sn, int regionReplication,
1645      long masterSystemTime, boolean saveBarrier)
1646          throws IOException {
1647    Table meta = getMetaHTable(connection);
1648    try {
1649      HRegionInfo copyOfMerged = new HRegionInfo(mergedRegion);
1650
1651      // use the maximum of what master passed us vs local time.
1652      long time = Math.max(EnvironmentEdgeManager.currentTime(), masterSystemTime);
1653
1654      // Put for parent
1655      Put putOfMerged = makePutFromRegionInfo(copyOfMerged, time);
1656      putOfMerged.addImmutable(HConstants.CATALOG_FAMILY, HConstants.MERGEA_QUALIFIER,
1657        regionA.toByteArray());
1658      putOfMerged.addImmutable(HConstants.CATALOG_FAMILY, HConstants.MERGEB_QUALIFIER,
1659        regionB.toByteArray());
1660
1661      // Deletes for merging regions
1662      Delete deleteA = makeDeleteFromRegionInfo(regionA, time);
1663      Delete deleteB = makeDeleteFromRegionInfo(regionB, time);
1664
1665      // The merged is a new region, openSeqNum = 1 is fine. ServerName may be null
1666      // if crash after merge happened but before we got to here.. means in-memory
1667      // locations of offlined merged, now-closed, regions is lost. Should be ok. We
1668      // assign the merged region later.
1669      if (sn != null) addLocation(putOfMerged, sn, 1, -1, mergedRegion.getReplicaId());
1670
1671      // Add empty locations for region replicas of the merged region so that number of replicas can
1672      // be cached whenever the primary region is looked up from meta
1673      for (int i = 1; i < regionReplication; i++) {
1674        addEmptyLocation(putOfMerged, i);
1675      }
1676
1677      byte[] tableRow = Bytes.toBytes(mergedRegion.getRegionNameAsString()
1678        + HConstants.DELIMITER);
1679      Mutation[] mutations;
1680      if (saveBarrier) {
1681        Put putBarrierA = makeDaughterPut(regionA.getEncodedNameAsBytes(),
1682            mergedRegion.getEncodedNameAsBytes());
1683        Put putBarrierB = makeDaughterPut(regionB.getEncodedNameAsBytes(),
1684            mergedRegion.getEncodedNameAsBytes());
1685        Put putDaughter = makeParentPut(mergedRegion.getEncodedNameAsBytes(), Bytes.toBytes(
1686            regionA.getEncodedName() + "," + regionB.getEncodedName()));
1687        mutations = new Mutation[] { putOfMerged, deleteA, deleteB,
1688            putBarrierA, putBarrierB, putDaughter};
1689      } else {
1690        mutations = new Mutation[] { putOfMerged, deleteA, deleteB };
1691      }
1692      multiMutate(connection, meta, tableRow, mutations);
1693    } finally {
1694      meta.close();
1695    }
1696  }

1698  /**
1699   * Splits the region into two in an atomic operation. Offlines the parent
1700   * region with the information that it is split into two, and also adds
1701   * the daughter regions. Does not add the location information to the daughter
1702   * regions since they are not open yet.
1703   * @param connection connection we're using
1704   * @param parent the parent region which is split
1705   * @param splitA Split daughter region A
1706   * @param splitB Split daughter region A
1707   * @param sn the location of the region
1708   * @param saveBarrier true if need save replication barrier in meta, used for serial replication
1709   */
1710  public static void splitRegion(final Connection connection, HRegionInfo parent,
1711      HRegionInfo splitA, HRegionInfo splitB, ServerName sn, int regionReplication,
1712      boolean saveBarrier) throws IOException {
1713    Table meta = getMetaHTable(connection);
1714    try {
1715      HRegionInfo copyOfParent = new HRegionInfo(parent);
1716      copyOfParent.setOffline(true);
1717      copyOfParent.setSplit(true);
1718
1719      //Put for parent
1720      Put putParent = makePutFromRegionInfo(copyOfParent);
1721      addDaughtersToPut(putParent, splitA, splitB);
1722
1723      //Puts for daughters
1724      Put putA = makePutFromRegionInfo(splitA);
1725      Put putB = makePutFromRegionInfo(splitB);
1726
1727      addLocation(putA, sn, 1, -1, splitA.getReplicaId()); //new regions, openSeqNum = 1 is fine.
1728      addLocation(putB, sn, 1, -1, splitB.getReplicaId());
1729
1730      // Add empty locations for region replicas of daughters so that number of replicas can be
1731      // cached whenever the primary region is looked up from meta
1732      for (int i = 1; i < regionReplication; i++) {
1733        addEmptyLocation(putA, i);
1734        addEmptyLocation(putB, i);
1735      }
1736
1737      Mutation[] mutations;
1738      if (saveBarrier) {
1739        Put parentPut = makeDaughterPut(parent.getEncodedNameAsBytes(),
1740            Bytes.toBytes(splitA.getEncodedName() + "," + splitB.getEncodedName()));
1741        Put daughterPutA = makeParentPut(splitA.getEncodedNameAsBytes(),
1742            parent.getEncodedNameAsBytes());
1743        Put daughterPutB = makeParentPut(splitB.getEncodedNameAsBytes(),
1744            parent.getEncodedNameAsBytes());
1745
1746        mutations = new Mutation[]{putParent, putA, putB, parentPut, daughterPutA, daughterPutB};
1747      } else {
1748        mutations = new Mutation[]{putParent, putA, putB};
1749      }
1750      byte[] tableRow = Bytes.toBytes(parent.getRegionNameAsString() + HConstants.DELIMITER);
1751      multiMutate(connection, meta, tableRow, mutations);
1752    } finally {
1753      meta.close();
1754    }
1755  }

1987   * Adds and Removes the specified regions from hbase:meta
1988   * @param connection connection we're using
1989   * @param regionsToRemove list of regions to be deleted from META
1990   * @param regionsToAdd list of regions to be added to META
1991   * @throws IOException
1992   */
1993  public static void mutateRegions(Connection connection,
1994                                   final List<HRegionInfo> regionsToRemove,
1995                                   final List<HRegionInfo> regionsToAdd)
1996    throws IOException {
1997    List<Mutation> mutation = new ArrayList<>();
1998    if (regionsToRemove != null) {
1999      for (HRegionInfo hri: regionsToRemove) {
2000        mutation.add(makeDeleteFromRegionInfo(hri));
2001      }
2002    }
2003    if (regionsToAdd != null) {
2004      for (HRegionInfo hri: regionsToAdd) {
2005        mutation.add(makePutFromRegionInfo(hri));
2006      }
2007    }
2008    mutateMetaTable(connection, mutation);
2009    if (regionsToRemove != null && regionsToRemove.size() > 0) {
2010      LOG.debug("Deleted " + HRegionInfo.getShortNameToLog(regionsToRemove));
2011    }
2012    if (regionsToAdd != null && regionsToAdd.size() > 0) {
2013      LOG.debug("Added " + HRegionInfo.getShortNameToLog(regionsToAdd));
2014    }
2015  }

2018   * Overwrites the specified regions from hbase:meta
2019   * @param connection connection we're using
2020   * @param regionInfos list of regions to be added to META
2021   * @throws IOException
2022   */
2023  public static void overwriteRegions(Connection connection,
2024      List<HRegionInfo> regionInfos, int regionReplication) throws IOException {
2025    // use master time for delete marker and the Put
2026    long now = EnvironmentEdgeManager.currentTime();
2027    deleteRegions(connection, regionInfos, now);
2028    // Why sleep? This is the easiest way to ensure that the previous deletes does not
2029    // eclipse the following puts, that might happen in the same ts from the server.
2030    // See HBASE-9906, and HBASE-9879. Once either HBASE-9879, HBASE-8770 is fixed,
2031    // or HBASE-9905 is fixed and meta uses seqIds, we do not need the sleep.
2032    //
2033    // HBASE-13875 uses master timestamp for the mutations. The 20ms sleep is not needed
2034    addRegionsToMeta(connection, regionInfos, regionReplication, now+1);
2035    LOG.info("Overwritten " + regionInfos.size() + " regions to Meta");
2036    if (LOG.isDebugEnabled()) {
2037      LOG.debug("Overwritten regions: " + regionInfos);
2038    }
2039  }
