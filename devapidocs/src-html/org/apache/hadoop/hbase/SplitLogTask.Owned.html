018package org.apache.hadoop.hbase;
019
020import java.io.IOException;
021
022import org.apache.hadoop.hbase.classification.InterfaceAudience;
023import org.apache.hadoop.hbase.exceptions.DeserializationException;
024import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
025import org.apache.hadoop.hbase.shaded.protobuf.generated.HBaseProtos;
026import org.apache.hadoop.hbase.shaded.protobuf.generated.ZooKeeperProtos;
027import org.apache.hadoop.hbase.shaded.protobuf.generated.ZooKeeperProtos.SplitLogTask.RecoveryMode;
028import org.apache.hadoop.hbase.util.Bytes;
029
030/**
031 * State of a WAL log split during distributed splitting.  State is kept up in zookeeper.
032 * Encapsulates protobuf serialization/deserialization so we don't leak generated pb outside of
033 * this class.  Used by regionserver and master packages.
034 * <p>Immutable
035 */
036@InterfaceAudience.Private
037public class SplitLogTask {
038  private final ServerName originServer;
039  private final ZooKeeperProtos.SplitLogTask.State state;
040  private final ZooKeeperProtos.SplitLogTask.RecoveryMode mode;
041
042  public static class Unassigned extends SplitLogTask {
043    public Unassigned(final ServerName originServer, final RecoveryMode mode) {
044      super(originServer, ZooKeeperProtos.SplitLogTask.State.UNASSIGNED, mode);
045    }
046  }
047
048  public static class Owned extends SplitLogTask {
049    public Owned(final ServerName originServer) {
050      this(originServer, ZooKeeperProtos.SplitLogTask.RecoveryMode.LOG_SPLITTING);
051    }
052 
053    public Owned(final ServerName originServer, final RecoveryMode mode) {
054      super(originServer, ZooKeeperProtos.SplitLogTask.State.OWNED, mode);
055    }
056  }
057
058  public static class Resigned extends SplitLogTask {
059    public Resigned(final ServerName originServer, final RecoveryMode mode) {
060      super(originServer, ZooKeeperProtos.SplitLogTask.State.RESIGNED, mode);
061    }
062  }
063
064  public static class Done extends SplitLogTask {
065    public Done(final ServerName originServer, final RecoveryMode mode) {
066      super(originServer, ZooKeeperProtos.SplitLogTask.State.DONE, mode);
067    }
068  }
069
070  public static class Err extends SplitLogTask {
071    public Err(final ServerName originServer, final RecoveryMode mode) {
072      super(originServer, ZooKeeperProtos.SplitLogTask.State.ERR, mode);
073    }
074  }
075
076  SplitLogTask(final ZooKeeperProtos.SplitLogTask slt) {
077    this.originServer = ProtobufUtil.toServerName(slt.getServerName());
078    this.state = slt.getState();
079    this.mode = (slt.hasMode()) ? slt.getMode() :
080      ZooKeeperProtos.SplitLogTask.RecoveryMode.UNKNOWN;
081  }
082
083  SplitLogTask(final ServerName originServer, final ZooKeeperProtos.SplitLogTask.State state,
084      final ZooKeeperProtos.SplitLogTask.RecoveryMode mode) {
085    this.originServer = originServer;
086    this.state = state;
087    this.mode = mode;
088  }
089
090  public ServerName getServerName() {
091    return this.originServer;
092  }
093
094  public ZooKeeperProtos.SplitLogTask.RecoveryMode getMode() {
095    return this.mode;
096  }
097
098  public boolean isUnassigned(final ServerName sn) {
099    return this.originServer.equals(sn) && isUnassigned();
100  }
101
102  public boolean isUnassigned() {
103    return this.state == ZooKeeperProtos.SplitLogTask.State.UNASSIGNED;
104  }
105
106  public boolean isOwned(final ServerName sn) {
107    return this.originServer.equals(sn) && isOwned();
108  }
109
110  public boolean isOwned() {
111    return this.state == ZooKeeperProtos.SplitLogTask.State.OWNED;
112  }
113
114  public boolean isResigned(final ServerName sn) {
115    return this.originServer.equals(sn) && isResigned();
116  }
117
118  public boolean isResigned() {
119    return this.state == ZooKeeperProtos.SplitLogTask.State.RESIGNED;
120  }
121
122  public boolean isDone(final ServerName sn) {
123    return this.originServer.equals(sn) && isDone();
124  }
125
126  public boolean isDone() {
127    return this.state == ZooKeeperProtos.SplitLogTask.State.DONE;
128  }
129
130  public boolean isErr(final ServerName sn) {
131    return this.originServer.equals(sn) && isErr();
132  }
133
134  public boolean isErr() {
135    return this.state == ZooKeeperProtos.SplitLogTask.State.ERR;
136  }
137
138  @Override
139  public String toString() {
140    return this.state.toString() + " " + this.originServer.toString();
141  }
142
143  @Override
144  public boolean equals(Object obj) {
145    if (!(obj instanceof SplitLogTask)) return false;
146    SplitLogTask other = (SplitLogTask)obj;
147    return other.state.equals(this.state) && other.originServer.equals(this.originServer);
148  }
149
150  @Override
151  public int hashCode() {
152    int hash = 7;
153    hash = 31 * hash + this.state.hashCode();
154    return 31 * hash + this.originServer.hashCode();
155  }
156
157  /**
158   * @param data Serialized date to parse.
159   * @return An SplitLogTaskState instance made of the passed <code>data</code>
160   * @throws DeserializationException
161   * @see #toByteArray()
162   */
163  public static SplitLogTask parseFrom(final byte [] data) throws DeserializationException {
164    ProtobufUtil.expectPBMagicPrefix(data);
165    try {
166      int prefixLen = ProtobufUtil.lengthOfPBMagic();
167      ZooKeeperProtos.SplitLogTask.Builder builder = ZooKeeperProtos.SplitLogTask.newBuilder();
168      ProtobufUtil.mergeFrom(builder, data, prefixLen, data.length - prefixLen);
169      return new SplitLogTask(builder.build());
170    } catch (IOException e) {
171      throw new DeserializationException(Bytes.toStringBinary(data, 0, 64), e);
172    }
173  }
174
175  /**
176   * @return This instance serialized into a byte array
177   * @see #parseFrom(byte[])
178   */
179  public byte [] toByteArray() {
180    // First create a pb ServerName.  Then create a ByteString w/ the TaskState
181    // bytes in it.  Finally create a SplitLogTaskState passing in the two
182    // pbs just created.
183    HBaseProtos.ServerName snpb = ProtobufUtil.toServerName(this.originServer);
184    ZooKeeperProtos.SplitLogTask slts =
185      ZooKeeperProtos.SplitLogTask.newBuilder().setServerName(snpb).setState(this.state).
186      setMode(this.mode).build();
187    return ProtobufUtil.prependPBMagic(slts.toByteArray());
188  }
189}
