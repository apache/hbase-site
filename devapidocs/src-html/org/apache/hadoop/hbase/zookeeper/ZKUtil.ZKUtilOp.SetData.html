115  public static RecoverableZooKeeper connect(Configuration conf, String ensemble,
116      Watcher watcher, final String identifier)
117  throws IOException {
118    if(ensemble == null) {
119      throw new IOException("Unable to determine ZooKeeper ensemble");
120    }
121    int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
122        HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
123    if (LOG.isTraceEnabled()) {
124      LOG.trace(identifier + " opening connection to ZooKeeper ensemble=" + ensemble);
125    }
126    int retry = conf.getInt("zookeeper.recovery.retry", 3);
127    int retryIntervalMillis =
128      conf.getInt("zookeeper.recovery.retry.intervalmill", 1000);
129    int maxSleepTime = conf.getInt("zookeeper.recovery.retry.maxsleeptime", 60000);
130    zkDumpConnectionTimeOut = conf.getInt("zookeeper.dump.connection.timeout",
131        1000);
132    return new RecoverableZooKeeper(ensemble, timeout, watcher,
133        retry, retryIntervalMillis, maxSleepTime, identifier);
134  }

178  /**
179   * Log in the current process using the given configuration keys for the
180   * credential file and login principal.
181   *
182   * <p><strong>This is only applicable when running on secure hbase</strong>
183   * On regular HBase (without security features), this will safely be ignored.
184   * </p>
185   *
186   * @param conf The configuration data to use
187   * @param keytabFileKey Property key used to configure the path to the credential file
188   * @param userNameKey Property key used to configure the login principal
189   * @param hostname Current hostname to use in any credentials
190   * @param loginContextProperty property name to expose the entry name
191   * @param loginContextName jaas entry name
192   * @throws IOException underlying exception from SecurityUtil.login() call
193   */
194  private static void login(Configuration conf, String keytabFileKey,
195      String userNameKey, String hostname,
196      String loginContextProperty, String loginContextName)
197      throws IOException {
198    if (!isSecureZooKeeper(conf))
199      return;
200
201    // User has specified a jaas.conf, keep this one as the good one.
202    // HBASE_OPTS="-Djava.security.auth.login.config=jaas.conf"
203    if (System.getProperty("java.security.auth.login.config") != null)
204      return;
205
206    // No keytab specified, no auth
207    String keytabFilename = conf.get(keytabFileKey);
208    if (keytabFilename == null) {
209      LOG.warn("no keytab specified for: " + keytabFileKey);
210      return;
211    }
212
213    String principalConfig = conf.get(userNameKey, System.getProperty("user.name"));
214    String principalName = SecurityUtil.getServerPrincipal(principalConfig, hostname);
215
216    // Initialize the "jaas.conf" for keyTab/principal,
217    // If keyTab is not specified use the Ticket Cache.
218    // and set the zookeeper login context name.
219    JaasConfiguration jaasConf = new JaasConfiguration(loginContextName,
220        principalName, keytabFilename);
221    javax.security.auth.login.Configuration.setConfiguration(jaasConf);
222    System.setProperty(loginContextProperty, loginContextName);
223  }

326  /**
327   * Watch the specified znode for delete/create/change events.  The watcher is
328   * set whether or not the node exists.  If the node already exists, the method
329   * returns true.  If the node does not exist, the method returns false.
330   *
331   * @param zkw zk reference
332   * @param znode path of node to watch
333   * @return true if znode exists, false if does not exist or error
334   * @throws KeeperException if unexpected zookeeper exception
335   */
336  public static boolean watchAndCheckExists(ZooKeeperWatcher zkw, String znode)
337  throws KeeperException {
338    try {
339      Stat s = zkw.getRecoverableZooKeeper().exists(znode, zkw);
340      boolean exists = s != null ? true : false;
341      if (exists) {
342        LOG.debug(zkw.prefix("Set watcher on existing znode=" + znode));
343      } else {
344        LOG.debug(zkw.prefix("Set watcher on znode that does not yet exist, " + znode));
345      }
346      return exists;
347    } catch (KeeperException e) {
348      LOG.warn(zkw.prefix("Unable to set watcher on znode " + znode), e);
349      zkw.keeperException(e);
350      return false;
351    } catch (InterruptedException e) {
352      LOG.warn(zkw.prefix("Unable to set watcher on znode " + znode), e);
353      zkw.interruptedException(e);
354      return false;
355    }
356  }

447  /**
448   * List all the children of the specified znode, setting a watch for children
449   * changes and also setting a watch on every individual child in order to get
450   * the NodeCreated and NodeDeleted events.
451   * @param zkw zookeeper reference
452   * @param znode node to get children of and watch
453   * @return list of znode names, null if the node doesn't exist
454   * @throws KeeperException
455   */
456  public static List<String> listChildrenAndWatchThem(ZooKeeperWatcher zkw,
457      String znode) throws KeeperException {
458    List<String> children = listChildrenAndWatchForNewChildren(zkw, znode);
459    if (children == null) {
460      return null;
461    }
462    for (String child : children) {
463      watchAndCheckExists(zkw, ZNodePaths.joinZNode(znode, child));
464    }
465    return children;
466  }

847  /**
848   * Returns whether or not secure authentication is enabled
849   * (whether <code>hbase.security.authentication</code> is set to
850   * <code>kerberos</code>.
851   */
852  public static boolean isSecureZooKeeper(Configuration conf) {
853    // Detection for embedded HBase client with jaas configuration
854    // defined for third party programs.
855    try {
856      javax.security.auth.login.Configuration testConfig =
857          javax.security.auth.login.Configuration.getConfiguration();
858      if (testConfig.getAppConfigurationEntry("Client") == null
859          && testConfig.getAppConfigurationEntry(
860            JaasConfiguration.CLIENT_KEYTAB_KERBEROS_CONFIG_NAME) == null
861          && testConfig.getAppConfigurationEntry(
862              JaasConfiguration.SERVER_KEYTAB_KERBEROS_CONFIG_NAME) == null
863          && conf.get(HConstants.ZK_CLIENT_KERBEROS_PRINCIPAL) == null
864          && conf.get(HConstants.ZK_SERVER_KERBEROS_PRINCIPAL) == null) {
865
866        return false;
867      }
868    } catch(Exception e) {
869      // No Jaas configuration defined.
870      return false;
871    }
872
873    // Master & RSs uses hbase.zookeeper.client.*
874    return "kerberos".equalsIgnoreCase(conf.get("hbase.security.authentication"));
875  }

881  public static ArrayList<ACL> createACL(ZooKeeperWatcher zkw, String node,
882    boolean isSecureZooKeeper) {
883    if (!node.startsWith(zkw.znodePaths.baseZNode)) {
884      return Ids.OPEN_ACL_UNSAFE;
885    }
886    if (isSecureZooKeeper) {
887      ArrayList<ACL> acls = new ArrayList<>();
888      // add permission to hbase supper user
889      String[] superUsers = zkw.getConfiguration().getStrings(Superusers.SUPERUSER_CONF_KEY);
890      String hbaseUser = null;
891      try {
892        hbaseUser = UserGroupInformation.getCurrentUser().getShortUserName();
893      } catch (IOException e) {
894        LOG.debug("Could not acquire current User.", e);
895      }
896      if (superUsers != null) {
897        List<String> groups = new ArrayList<>();
898        for (String user : superUsers) {
899          if (AuthUtil.isGroupPrincipal(user)) {
900            // TODO: Set node ACL for groups when ZK supports this feature
901            groups.add(user);
902          } else {
903            if(!user.equals(hbaseUser)) {
904              acls.add(new ACL(Perms.ALL, new Id("sasl", user)));
905            }
906          }
907        }
908        if (!groups.isEmpty()) {
909          LOG.warn("Znode ACL setting for group " + groups
910              + " is skipped, ZooKeeper doesn't support this feature presently.");
911        }
912      }
913      // Certain znodes are accessed directly by the client,
914      // so they must be readable by non-authenticated clients
915      if (zkw.isClientReadable(node)) {
916        acls.addAll(Ids.CREATOR_ALL_ACL);
917        acls.addAll(Ids.READ_ACL_UNSAFE);
918      } else {
919        acls.addAll(Ids.CREATOR_ALL_ACL);
920      }
921      return acls;
922    } else {
923      return Ids.OPEN_ACL_UNSAFE;
924    }
925  }

1334   * these znodes including the parent using multi-update api or
1335   * sequential based on the specified configurations.
1336   * <p>

1338   * children.
1339   * <p>
1340   * If the following is true:
1341   * <ul>
1342   * <li>runSequentialOnMultiFailure is true
1343   * </ul>
1344   * on calling multi, we get a ZooKeeper exception that can be handled by a
1345   * sequential call(*), we retry the operations one-by-one (sequentially).
1346   *
1347   * @param zkw
1348   *          - zk reference
1349   * @param runSequentialOnMultiFailure
1350   *          - if true when we get a ZooKeeper exception that could retry the
1351   *          operations one-by-one (sequentially)
1352   * @param pathRoots
1353   *          - path of the parent node(s)
1354   * @throws KeeperException.NotEmptyException
1355   *           if node has children while deleting
1356   * @throws KeeperException
1357   *           if unexpected ZooKeeper exception
1358   * @throws IllegalArgumentException
1359   *           if an invalid path is specified
1360   */
1361  public static void deleteNodeRecursivelyMultiOrSequential(ZooKeeperWatcher zkw,

1363    if (pathRoots == null || pathRoots.length <= 0) {

1365      return;
1366    }
1367    List<ZKUtilOp> ops = new ArrayList<>();
1368    for (String eachRoot : pathRoots) {
1369      // ZooKeeper Watches are one time triggers; When children of parent nodes are deleted
1370      // recursively, must set another watch, get notified of delete node
1371      List<String> children = listChildrenBFSAndWatchThem(zkw, eachRoot);
1372      // Delete the leaves first and eventually get rid of the root
1373      for (int i = children.size() - 1; i >= 0; --i) {
1374        ops.add(ZKUtilOp.deleteNodeFailSilent(children.get(i)));
1375      }
1376      try {
1377        if (zkw.getRecoverableZooKeeper().exists(eachRoot, zkw) != null) {
1378          ops.add(ZKUtilOp.deleteNodeFailSilent(eachRoot));
1379        }
1380      } catch (InterruptedException e) {
1381        zkw.interruptedException(e);
1382      }
1383    }
1384    // atleast one element should exist
1385    if (ops.size() > 0) {
1386      multiOrSequential(zkw, ops, runSequentialOnMultiFailure);

1391   * BFS Traversal of all the children under path, with the entries in the list,
1392   * in the same order as that of the traversal. Lists all the children without
1393   * setting any watches.
1394   *
1395   * @param zkw
1396   *          - zk reference
1397   * @param znode
1398   *          - path of node
1399   * @return list of children znodes under the path
1400   * @throws KeeperException
1401   *           if unexpected ZooKeeper exception
1402   */
1403  private static List<String> listChildrenBFSNoWatch(ZooKeeperWatcher zkw,
1404      final String znode) throws KeeperException {
1405    Deque<String> queue = new LinkedList<>();
1406    List<String> tree = new ArrayList<>();
1407    queue.add(znode);
1408    while (true) {
1409      String node = queue.pollFirst();
1410      if (node == null) {
1411        break;
1412      }
1413      List<String> children = listChildrenNoWatch(zkw, node);
1414      if (children == null) {
1415        continue;
1416      }
1417      for (final String child : children) {
1418        final String childPath = node + "/" + child;
1419        queue.add(childPath);
1420        tree.add(childPath);
1421      }
1422    }
1423    return tree;
1424  }

1612  /**
1613   * Use ZooKeeper's multi-update functionality.
1614   *
1615   * If all of the following are true:
1616   * - runSequentialOnMultiFailure is true
1617   * - on calling multi, we get a ZooKeeper exception that can be handled by a sequential call(*)
1618   * Then:
1619   * - we retry the operations one-by-one (sequentially)
1620   *
1621   * Note *: an example is receiving a NodeExistsException from a "create" call.  Without multi,
1622   * a user could call "createAndFailSilent" to ensure that a node exists if they don't care who
1623   * actually created the node (i.e. the NodeExistsException from ZooKeeper is caught).
1624   * This will cause all operations in the multi to fail, however, because
1625   * the NodeExistsException that zk.create throws will fail the multi transaction.
1626   * In this case, if the previous conditions hold, the commands are run sequentially, which should
1627   * result in the correct final state, but means that the operations will not run atomically.
1628   *
1629   * @throws KeeperException
1630   */
1631  public static void multiOrSequential(ZooKeeperWatcher zkw, List<ZKUtilOp> ops,
1632      boolean runSequentialOnMultiFailure) throws KeeperException {
1633    if (zkw.getConfiguration().get("hbase.zookeeper.useMulti") != null) {
1634      LOG.warn("hbase.zookeeper.useMulti is deprecated. Default to true always.");
1635    }
1636    if (ops == null) return;
1637
1638    List<Op> zkOps = new LinkedList<>();
1639    for (ZKUtilOp op : ops) {
1640      zkOps.add(toZooKeeperOp(zkw, op));
1641    }
1642    try {
1643      zkw.getRecoverableZooKeeper().multi(zkOps);
1644    } catch (KeeperException ke) {
1645      switch (ke.code()) {
1646      case NODEEXISTS:
1647      case NONODE:
1648      case BADVERSION:
1649      case NOAUTH:
1650        // if we get an exception that could be solved by running sequentially
1651        // (and the client asked us to), then break out and run sequentially
1652        if (runSequentialOnMultiFailure) {
1653          LOG.info("On call to ZK.multi, received exception: " + ke.toString() + "."
1654              + "  Attempting to run operations sequentially because"
1655              + " runSequentialOnMultiFailure is: " + runSequentialOnMultiFailure + ".");
1656          processSequentially(zkw, ops);
1657          break;
1658        }
1659      default:
1660        throw ke;
1661      }
1662    } catch (InterruptedException ie) {
1663      zkw.interruptedException(ie);
1664    }
1665  }

1667  private static void processSequentially(ZooKeeperWatcher zkw, List<ZKUtilOp> ops)
1668      throws KeeperException, NoNodeException {
1669    for (ZKUtilOp op : ops) {
1670      if (op instanceof CreateAndFailSilent) {
1671        createAndFailSilent(zkw, (CreateAndFailSilent) op);
1672      } else if (op instanceof DeleteNodeFailSilent) {
1673        deleteNodeFailSilent(zkw, (DeleteNodeFailSilent) op);
1674      } else if (op instanceof SetData) {
1675        setData(zkw, (SetData) op);
1676      } else {
1677        throw new UnsupportedOperationException("Unexpected ZKUtilOp type: "
1678            + op.getClass().getName());
1679      }
1680    }
1681  }

1887   * @param timeout  The socket timeout to use.

1889   * @throws IOException When the socket communication fails.
1890   */
1891  public static String[] getServerStats(String server, int timeout)
1892  throws IOException {
1893    String[] sp = server.split(":");
1894    if (sp == null || sp.length == 0) {
1895      return null;
1896    }
1897
1898    String host = sp[0];
1899    int port = sp.length > 1 ? Integer.parseInt(sp[1])
1900        : HConstants.DEFAULT_ZOOKEPER_CLIENT_PORT;
1901
1902    InetSocketAddress sockAddr = new InetSocketAddress(host, port);
1903    try (Socket socket = new Socket()) {
1904      socket.connect(sockAddr, timeout);
1905
1906      socket.setSoTimeout(timeout);
1907      try (PrintWriter out = new PrintWriter(socket.getOutputStream(), true);
1908          BufferedReader in = new BufferedReader(new InputStreamReader(socket.getInputStream()))) {
1909        out.println("stat");
1910        out.flush();
1911        ArrayList<String> res = new ArrayList<>();
1912        while (true) {
1913          String line = in.readLine();
1914          if (line != null) {
1915            res.add(line);
1916          } else {
1917            break;
1918          }
1919        }
1920        return res.toArray(new String[res.size()]);
1921      }
1922    }

1947  /**
1948   * Waits for HBase installation's base (parent) znode to become available.
1949   * @throws IOException on ZK errors
1950   */
1951  public static void waitForBaseZNode(Configuration conf) throws IOException {
1952    LOG.info("Waiting until the base znode is available");
1953    String parentZNode = conf.get(HConstants.ZOOKEEPER_ZNODE_PARENT,
1954        HConstants.DEFAULT_ZOOKEEPER_ZNODE_PARENT);
1955    ZooKeeper zk = new ZooKeeper(ZKConfig.getZKQuorumServersString(conf),
1956        conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
1957        HConstants.DEFAULT_ZK_SESSION_TIMEOUT), EmptyWatcher.instance);
1958
1959    final int maxTimeMs = 10000;
1960    final int maxNumAttempts = maxTimeMs / HConstants.SOCKET_RETRY_WAIT_MS;
1961
1962    KeeperException keeperEx = null;
1963    try {
1964      try {
1965        for (int attempt = 0; attempt < maxNumAttempts; ++attempt) {
1966          try {
1967            if (zk.exists(parentZNode, false) != null) {
1968              LOG.info("Parent znode exists: " + parentZNode);
1969              keeperEx = null;
1970              break;
1971            }
1972          } catch (KeeperException e) {
1973            keeperEx = e;
1974          }
1975          Threads.sleepWithoutInterrupt(HConstants.SOCKET_RETRY_WAIT_MS);
1976        }
1977      } finally {
1978        zk.close();
1979      }
1980    } catch (InterruptedException ex) {
1981      Thread.currentThread().interrupt();
1982    }
1983
1984    if (keeperEx != null) {
1985      throw new IOException(keeperEx);
1986    }
1987  }

2045   * @param bytes - Content of a WAL position znode.
2046   * @return long - The current WAL position.
2047   * @throws DeserializationException
2048   */
2049  public static long parseWALPositionFrom(final byte[] bytes) throws DeserializationException {
2050    if (bytes == null) {
2051      throw new DeserializationException("Unable to parse null WAL position.");
2052    }
2053    if (ProtobufUtil.isPBMagicPrefix(bytes)) {
2054      int pblen = ProtobufUtil.lengthOfPBMagic();
2055      ReplicationProtos.ReplicationHLogPosition.Builder builder =
2056          ReplicationProtos.ReplicationHLogPosition.newBuilder();
2057      ReplicationProtos.ReplicationHLogPosition position;
2058      try {
2059        ProtobufUtil.mergeFrom(builder, bytes, pblen, bytes.length - pblen);
2060        position = builder.build();
2061      } catch (IOException e) {
2062        throw new DeserializationException(e);
2063      }
2064      return position.getPosition();
2065    } else {
2066      if (bytes.length > 0) {
2067        return Bytes.toLong(bytes);
2068      }
2069      return 0;
2070    }
