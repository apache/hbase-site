103   * from the specified configuration object using methods from {@link ZKConfig}.

124  public static RecoverableZooKeeper connect(Configuration conf, String ensemble,
125      Watcher watcher, final String identifier)
126    throws IOException {
127    if(ensemble == null) {
128      throw new IOException("Unable to determine ZooKeeper ensemble");
129    }
130    int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
131        HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
132    if (LOG.isTraceEnabled()) {
133      LOG.trace("{} opening connection to ZooKeeper ensemble={}", identifier, ensemble);
134    }
135    int retry = conf.getInt("zookeeper.recovery.retry", 3);
136    int retryIntervalMillis =
137      conf.getInt("zookeeper.recovery.retry.intervalmill", 1000);
138    int maxSleepTime = conf.getInt("zookeeper.recovery.retry.maxsleeptime", 60000);
139    zkDumpConnectionTimeOut = conf.getInt("zookeeper.dump.connection.timeout",
140        1000);
141    int multiMaxSize = conf.getInt("zookeeper.multi.max.size", 1024*1024);
142    return new RecoverableZooKeeper(ensemble, timeout, watcher,
143        retry, retryIntervalMillis, maxSleepTime, identifier, multiMaxSize);
144  }

188  /**
189   * Log in the current process using the given configuration keys for the
190   * credential file and login principal.
191   *
192   * <p><strong>This is only applicable when running on secure hbase</strong>
193   * On regular HBase (without security features), this will safely be ignored.
194   * </p>
195   *
196   * @param conf The configuration data to use
197   * @param keytabFileKey Property key used to configure the path to the credential file
198   * @param userNameKey Property key used to configure the login principal
199   * @param hostname Current hostname to use in any credentials
200   * @param loginContextProperty property name to expose the entry name
201   * @param loginContextName jaas entry name
202   * @throws IOException underlying exception from SecurityUtil.login() call
203   */
204  private static void login(Configuration conf, String keytabFileKey,
205      String userNameKey, String hostname,
206      String loginContextProperty, String loginContextName)
207      throws IOException {
208    if (!isSecureZooKeeper(conf)) {
209      return;
210    }
211
212    // User has specified a jaas.conf, keep this one as the good one.
213    // HBASE_OPTS="-Djava.security.auth.login.config=jaas.conf"
214    if (System.getProperty("java.security.auth.login.config") != null) {
215      return;
216    }
217
218    // No keytab specified, no auth
219    String keytabFilename = conf.get(keytabFileKey);
220    if (keytabFilename == null) {
221      LOG.warn("no keytab specified for: {}", keytabFileKey);
222      return;
223    }
224
225    String principalConfig = conf.get(userNameKey, System.getProperty("user.name"));
226    String principalName = SecurityUtil.getServerPrincipal(principalConfig, hostname);
227
228    // Initialize the "jaas.conf" for keyTab/principal,
229    // If keyTab is not specified use the Ticket Cache.
230    // and set the zookeeper login context name.
231    JaasConfiguration jaasConf = new JaasConfiguration(loginContextName,
232        principalName, keytabFilename);
233    javax.security.auth.login.Configuration.setConfiguration(jaasConf);
234    System.setProperty(loginContextProperty, loginContextName);
235  }

341  /**
342   * Watch the specified znode for delete/create/change events.  The watcher is
343   * set whether or not the node exists.  If the node already exists, the method
344   * returns true.  If the node does not exist, the method returns false.
345   *
346   * @param zkw zk reference
347   * @param znode path of node to watch
348   * @return true if znode exists, false if does not exist or error
349   * @throws KeeperException if unexpected zookeeper exception
350   */
351  public static boolean watchAndCheckExists(ZKWatcher zkw, String znode)
352    throws KeeperException {
353    try {
354      Stat s = zkw.getRecoverableZooKeeper().exists(znode, zkw);
355      boolean exists = s != null;
356      if (exists) {
357        LOG.debug(zkw.prefix("Set watcher on existing znode=" + znode));
358      } else {
359        LOG.debug(zkw.prefix("Set watcher on znode that does not yet exist, " + znode));
360      }
361      return exists;
362    } catch (KeeperException e) {
363      LOG.warn(zkw.prefix("Unable to set watcher on znode " + znode), e);
364      zkw.keeperException(e);
365      return false;
366    } catch (InterruptedException e) {
367      LOG.warn(zkw.prefix("Unable to set watcher on znode " + znode), e);
368      zkw.interruptedException(e);
369      return false;
370    }
371  }

461  /**
462   * List all the children of the specified znode, setting a watch for children
463   * changes and also setting a watch on every individual child in order to get
464   * the NodeCreated and NodeDeleted events.
465   * @param zkw zookeeper reference
466   * @param znode node to get children of and watch
467   * @return list of znode names, null if the node doesn't exist
468   * @throws KeeperException if a ZooKeeper operation fails
469   */
470  public static List<String> listChildrenAndWatchThem(ZKWatcher zkw,
471      String znode) throws KeeperException {
472    List<String> children = listChildrenAndWatchForNewChildren(zkw, znode);
473    if (children == null) {
474      return null;
475    }
476    for (String child : children) {
477      watchAndCheckExists(zkw, ZNodePaths.joinZNode(znode, child));
478    }
479    return children;
480  }

859  /**
860   * Returns whether or not secure authentication is enabled
861   * (whether <code>hbase.security.authentication</code> is set to
862   * <code>kerberos</code>.
863   */
864  public static boolean isSecureZooKeeper(Configuration conf) {
865    // Detection for embedded HBase client with jaas configuration
866    // defined for third party programs.
867    try {
868      javax.security.auth.login.Configuration testConfig =
869          javax.security.auth.login.Configuration.getConfiguration();
870      if (testConfig.getAppConfigurationEntry("Client") == null
871          && testConfig.getAppConfigurationEntry(
872            JaasConfiguration.CLIENT_KEYTAB_KERBEROS_CONFIG_NAME) == null
873          && testConfig.getAppConfigurationEntry(
874              JaasConfiguration.SERVER_KEYTAB_KERBEROS_CONFIG_NAME) == null
875          && conf.get(HConstants.ZK_CLIENT_KERBEROS_PRINCIPAL) == null
876          && conf.get(HConstants.ZK_SERVER_KERBEROS_PRINCIPAL) == null) {
877
878        return false;
879      }
880    } catch(Exception e) {
881      // No Jaas configuration defined.
882      return false;
883    }
884
885    // Master & RSs uses hbase.zookeeper.client.*
886    return "kerberos".equalsIgnoreCase(conf.get("hbase.security.authentication"));
887  }

890    return createACL(zkw, node, isSecureZooKeeper(zkw.getConfiguration()));

893  public static ArrayList<ACL> createACL(ZKWatcher zkw, String node,
894                                         boolean isSecureZooKeeper) {
895    if (!node.startsWith(zkw.getZNodePaths().baseZNode)) {
896      return Ids.OPEN_ACL_UNSAFE;
897    }
898    if (isSecureZooKeeper) {
899      ArrayList<ACL> acls = new ArrayList<>();
900      // add permission to hbase supper user
901      String[] superUsers = zkw.getConfiguration().getStrings(Superusers.SUPERUSER_CONF_KEY);
902      String hbaseUser = null;
903      try {
904        hbaseUser = UserGroupInformation.getCurrentUser().getShortUserName();
905      } catch (IOException e) {
906        LOG.debug("Could not acquire current User.", e);
907      }
908      if (superUsers != null) {
909        List<String> groups = new ArrayList<>();
910        for (String user : superUsers) {
911          if (AuthUtil.isGroupPrincipal(user)) {
912            // TODO: Set node ACL for groups when ZK supports this feature
913            groups.add(user);
914          } else {
915            if(!user.equals(hbaseUser)) {
916              acls.add(new ACL(Perms.ALL, new Id("sasl", user)));
917            }
918          }
919        }
920        if (!groups.isEmpty()) {
921          LOG.warn("Znode ACL setting for group {} is skipped, ZooKeeper doesn't support this " +
922            "feature presently.", groups);
923        }
924      }
925      // Certain znodes are accessed directly by the client,
926      // so they must be readable by non-authenticated clients
927      if (zkw.getZNodePaths().isClientReadable(node)) {
928        acls.addAll(Ids.CREATOR_ALL_ACL);
929        acls.addAll(Ids.READ_ACL_UNSAFE);
930      } else {
931        acls.addAll(Ids.CREATOR_ALL_ACL);
932      }
933      return acls;
934    } else {
935      return Ids.OPEN_ACL_UNSAFE;
936    }
937  }

1398   * Chunks the provided {@code ops} when their approximate size exceeds the the configured limit.
1399   * Take caution that this can ONLY be used for operations where atomicity is not important,
1400   * e.g. deletions. It must not be used when atomicity of the operations is critical.
1401   *
1402   * @param zkw reference to the {@link ZKWatcher} which contains configuration and constants
1403   * @param runSequentialOnMultiFailure if true when we get a ZooKeeper exception that could
1404   *        retry the operations one-by-one (sequentially)
1405   * @param ops list of ZKUtilOp {@link ZKUtilOp} to partition while submitting batched multi
1406   *        or sequential
1407   * @throws KeeperException unexpected ZooKeeper Exception / Zookeeper unreachable
1408   */
1409  private static void submitBatchedMultiOrSequential(ZKWatcher zkw,
1410      boolean runSequentialOnMultiFailure, List<ZKUtilOp> ops) throws KeeperException {
1411    // at least one element should exist
1412    if (ops.isEmpty()) {
1413      return;
1414    }
1415    final int maxMultiSize = zkw.getRecoverableZooKeeper().getMaxMultiSizeLimit();
1416    // Batch up the items to over smashing through jute.maxbuffer with too many Ops.
1417    final List<List<ZKUtilOp>> batchedOps = partitionOps(ops, maxMultiSize);
1418    // Would use forEach() but have to handle KeeperException
1419    for (List<ZKUtilOp> batch : batchedOps) {
1420      multiOrSequential(zkw, batch, runSequentialOnMultiFailure);
1421    }
1422  }

1424  /**
1425   * Partition the list of {@code ops} by size (using {@link #estimateSize(ZKUtilOp)}).
1426   */
1427  static List<List<ZKUtilOp>> partitionOps(List<ZKUtilOp> ops, int maxPartitionSize) {
1428    List<List<ZKUtilOp>> partitionedOps = new ArrayList<>();
1429    List<ZKUtilOp> currentPartition = new ArrayList<>();
1430    int currentPartitionSize = 0;
1431    partitionedOps.add(currentPartition);
1432    Iterator<ZKUtilOp> iter = ops.iterator();
1433    while (iter.hasNext()) {
1434      ZKUtilOp currentOp = iter.next();
1435      int currentOpSize = estimateSize(currentOp);
1436
1437      // Roll a new partition if necessary
1438      // If the current partition is empty, put the element in there anyways.
1439      // We can roll a new partition if we get another element
1440      if (!currentPartition.isEmpty() && currentOpSize + currentPartitionSize > maxPartitionSize) {
1441        currentPartition = new ArrayList<>();
1442        partitionedOps.add(currentPartition);
1443        currentPartitionSize = 0;
1444      }
1445
1446      // Add the current op to the partition
1447      currentPartition.add(currentOp);
1448      // And record its size
1449      currentPartitionSize += currentOpSize;
1450    }
1451    return partitionedOps;
1452  }

1454  static int estimateSize(ZKUtilOp op) {
1455    return Bytes.toBytes(op.getPath()).length;
1456  }

1459   * BFS Traversal of all the children under path, with the entries in the list,
1460   * in the same order as that of the traversal. Lists all the children without
1461   * setting any watches.
1462   *
1463   * @param zkw
1464   *          - zk reference
1465   * @param znode
1466   *          - path of node
1467   * @return list of children znodes under the path
1468   * @throws KeeperException
1469   *           if unexpected ZooKeeper exception
1470   */
1471  private static List<String> listChildrenBFSNoWatch(ZKWatcher zkw,
1472      final String znode) throws KeeperException {
1473    Deque<String> queue = new LinkedList<>();
1474    List<String> tree = new ArrayList<>();
1475    queue.add(znode);
1476    while (true) {
1477      String node = queue.pollFirst();
1478      if (node == null) {
1479        break;
1480      }
1481      List<String> children = listChildrenNoWatch(zkw, node);
1482      if (children == null) {
1483        continue;
1484      }
1485      for (final String child : children) {
1486        final String childPath = node + "/" + child;
1487        queue.add(childPath);
1488        tree.add(childPath);
1489      }
1490    }
1491    return tree;
1492  }

1713  /**
1714   * Use ZooKeeper's multi-update functionality.
1715   *
1716   * If all of the following are true:
1717   * - runSequentialOnMultiFailure is true
1718   * - on calling multi, we get a ZooKeeper exception that can be handled by a sequential call(*)
1719   * Then:
1720   * - we retry the operations one-by-one (sequentially)
1721   *
1722   * Note *: an example is receiving a NodeExistsException from a "create" call.  Without multi,
1723   * a user could call "createAndFailSilent" to ensure that a node exists if they don't care who
1724   * actually created the node (i.e. the NodeExistsException from ZooKeeper is caught).
1725   * This will cause all operations in the multi to fail, however, because
1726   * the NodeExistsException that zk.create throws will fail the multi transaction.
1727   * In this case, if the previous conditions hold, the commands are run sequentially, which should
1728   * result in the correct final state, but means that the operations will not run atomically.
1729   *
1730   * @throws KeeperException if a ZooKeeper operation fails
1731   */
1732  public static void multiOrSequential(ZKWatcher zkw, List<ZKUtilOp> ops,
1733                                       boolean runSequentialOnMultiFailure) throws KeeperException {
1734    if (zkw.getConfiguration().get("hbase.zookeeper.useMulti") != null) {
1735      LOG.warn("hbase.zookeeper.useMulti is deprecated. Default to true always.");
1736    }
1737    if (ops == null) {
1738      return;
1739    }
1740
1741    List<Op> zkOps = new LinkedList<>();
1742    for (ZKUtilOp op : ops) {
1743      zkOps.add(toZooKeeperOp(zkw, op));
1744    }
1745    try {
1746      zkw.getRecoverableZooKeeper().multi(zkOps);
1747    } catch (KeeperException ke) {
1748      switch (ke.code()) {
1749        case NODEEXISTS:
1750        case NONODE:
1751        case BADVERSION:
1752        case NOAUTH:
1753          // if we get an exception that could be solved by running sequentially
1754          // (and the client asked us to), then break out and run sequentially
1755          if (runSequentialOnMultiFailure) {
1756            LOG.info("On call to ZK.multi, received exception: " + ke.toString() + "."
1757                + "  Attempting to run operations sequentially because"
1758                + " runSequentialOnMultiFailure is: " + runSequentialOnMultiFailure + ".");
1759            processSequentially(zkw, ops);
1760            break;
1761          }
1762        default:
1763          throw ke;
1764      }
1765    } catch (InterruptedException ie) {
1766      zkw.interruptedException(ie);
1767    }
1768  }
1769
1770  private static void processSequentially(ZKWatcher zkw, List<ZKUtilOp> ops)
1771      throws KeeperException, NoNodeException {
1772    for (ZKUtilOp op : ops) {
1773      if (op instanceof CreateAndFailSilent) {
1774        createAndFailSilent(zkw, (CreateAndFailSilent) op);
1775      } else if (op instanceof DeleteNodeFailSilent) {
1776        deleteNodeFailSilent(zkw, (DeleteNodeFailSilent) op);
1777      } else if (op instanceof SetData) {
1778        setData(zkw, (SetData) op);
1779      } else {
1780        throw new UnsupportedOperationException("Unexpected ZKUtilOp type: "
1781            + op.getClass().getName());
1782      }
1783    }
1784  }

1919      throws KeeperException {
1920    List<String> stack = new LinkedList<>();
1921    stack.add(znode);
1922    do {
1923      String znodeToProcess = stack.remove(stack.size() - 1);
1924      sb.append("\n").append(znodeToProcess).append(": ");
1925      byte[] data;
1926      try {
1927        data = ZKUtil.getData(zkw, znodeToProcess);
1928      } catch (InterruptedException e) {
1929        zkw.interruptedException(e);
1930        return;
1931      }
1932      if (data != null && data.length > 0) { // log position
1933        long position = 0;

1938        } catch (InterruptedException e) {

1940          return;
1941        }
1942      }
1943      for (String zNodeChild : ZKUtil.listChildrenNoWatch(zkw, znodeToProcess)) {
1944        stack.add(ZNodePaths.joinZNode(znodeToProcess, zNodeChild));
1945      }
1946    } while (stack.size() > 0);
