048/**
049 * Acts as the single ZooKeeper Watcher.  One instance of this is instantiated
050 * for each Master, RegionServer, and client process.
051 *
052 * <p>This is the only class that implements {@link Watcher}.  Other internal
053 * classes which need to be notified of ZooKeeper events must register with
054 * the local instance of this watcher via {@link #registerListener}.
055 *
056 * <p>This class also holds and manages the connection to ZooKeeper.  Code to
057 * deal with connection related events and exceptions are handled here.
058 */
059@InterfaceAudience.Private
060public class ZKWatcher implements Watcher, Abortable, Closeable {
061  private static final Log LOG = LogFactory.getLog(ZKWatcher.class);

107  /**
108   * Instantiate a ZooKeeper connection and watcher.
109   * @param conf
110   * @param identifier string that is passed to RecoverableZookeeper to be used as identifier for
111   *          this instance. Use null for default.
112   * @param abortable Can be null if there is on error there is no host to abort: e.g. client
113   *          context.
114   * @param canCreateBaseZNode
115   * @throws IOException
116   * @throws ZooKeeperConnectionException
117   */
118  public ZKWatcher(Configuration conf, String identifier,
119                   Abortable abortable, boolean canCreateBaseZNode)
120  throws IOException, ZooKeeperConnectionException {
121    this.conf = conf;
122    this.quorum = ZKConfig.getZKQuorumServersString(conf);
123    this.prefix = identifier;
124    // Identifier will get the sessionid appended later below down when we
125    // handle the syncconnect event.
126    this.identifier = identifier + "0x0";
127    this.abortable = abortable;
128    this.znodePaths = new ZNodePaths(conf);
129    PendingWatcher pendingWatcher = new PendingWatcher();
130    this.recoverableZooKeeper = ZKUtil.connect(conf, quorum, pendingWatcher, identifier);
131    pendingWatcher.prepare(this);
132    if (canCreateBaseZNode) {
133      try {
134        createBaseZNodes();
135      } catch (ZooKeeperConnectionException zce) {
136        try {
137          this.recoverableZooKeeper.close();
138        } catch (InterruptedException ie) {
139          LOG.debug("Encountered InterruptedException when closing " + this.recoverableZooKeeper);
140          Thread.currentThread().interrupt();
141        }
142        throw zce;
143      }
144    }
145  }

147  private void createBaseZNodes() throws ZooKeeperConnectionException {
148    try {
149      // Create all the necessary "directories" of znodes
150      ZKUtil.createWithParents(this, znodePaths.baseZNode);
151      ZKUtil.createAndFailSilent(this, znodePaths.rsZNode);
152      ZKUtil.createAndFailSilent(this, znodePaths.drainingZNode);
153      ZKUtil.createAndFailSilent(this, znodePaths.tableZNode);
154      ZKUtil.createAndFailSilent(this, znodePaths.splitLogZNode);
155      ZKUtil.createAndFailSilent(this, znodePaths.backupMasterAddressesZNode);
156      ZKUtil.createAndFailSilent(this, znodePaths.tableLockZNode);
157      ZKUtil.createAndFailSilent(this, znodePaths.masterMaintZNode);
158    } catch (KeeperException e) {
159      throw new ZooKeeperConnectionException(
160          prefix("Unexpected KeeperException creating base node"), e);
161    }
162  }

164  /** Returns whether the znode is supposed to be readable by the client
165   * and DOES NOT contain sensitive information (world readable).*/
166  public boolean isClientReadable(String node) {
167    // Developer notice: These znodes are world readable. DO NOT add more znodes here UNLESS
168    // all clients need to access this data to work. Using zk for sharing data to clients (other
169    // than service lookup case is not a recommended design pattern.
170    return
171        node.equals(znodePaths.baseZNode) ||
172        znodePaths.isAnyMetaReplicaZNode(node) ||
173        node.equals(znodePaths.masterAddressZNode) ||
174        node.equals(znodePaths.clusterIdZNode)||
175        node.equals(znodePaths.rsZNode) ||
176        // /hbase/table and /hbase/table/foo is allowed, /hbase/table-lock is not
177        node.equals(znodePaths.tableZNode) ||
178        node.startsWith(znodePaths.tableZNode + "/");
179  }

181  /**
182   * On master start, we check the znode ACLs under the root directory and set the ACLs properly
183   * if needed. If the cluster goes from an unsecure setup to a secure setup, this step is needed
184   * so that the existing znodes created with open permissions are now changed with restrictive
185   * perms.
186   */
187  public void checkAndSetZNodeAcls() {
188    if (!ZKUtil.isSecureZooKeeper(getConfiguration())) {
189      LOG.info("not a secure deployment, proceeding");
190      return;
191    }
192
193    // Check the base znodes permission first. Only do the recursion if base znode's perms are not
194    // correct.
195    try {
196      List<ACL> actualAcls = recoverableZooKeeper.getAcl(znodePaths.baseZNode, new Stat());
197
198      if (!isBaseZnodeAclSetup(actualAcls)) {
199        LOG.info("setting znode ACLs");
200        setZnodeAclsRecursive(znodePaths.baseZNode);
201      }
202    } catch(KeeperException.NoNodeException nne) {
203      return;
204    } catch(InterruptedException ie) {
205      interruptedExceptionNoThrow(ie, false);
206    } catch (IOException|KeeperException e) {
207      LOG.warn("Received exception while checking and setting zookeeper ACLs", e);
208    }
209  }

211  /**
212   * Set the znode perms recursively. This will do post-order recursion, so that baseZnode ACLs
213   * will be set last in case the master fails in between.
214   * @param znode
215   */
216  private void setZnodeAclsRecursive(String znode) throws KeeperException, InterruptedException {
217    List<String> children = recoverableZooKeeper.getChildren(znode, false);
218
219    for (String child : children) {
220      setZnodeAclsRecursive(ZNodePaths.joinZNode(znode, child));
221    }
222    List<ACL> acls = ZKUtil.createACL(this, znode, true);
223    LOG.info("Setting ACLs for znode:" + znode + " , acl:" + acls);
224    recoverableZooKeeper.setAcl(znode, acls, -1);
225  }

227  /**
228   * Checks whether the ACLs returned from the base znode (/hbase) is set for secure setup.
229   * @param acls acls from zookeeper
230   * @return whether ACLs are set for the base znode
231   * @throws IOException
232   */
233  private boolean isBaseZnodeAclSetup(List<ACL> acls) throws IOException {
234    if (LOG.isDebugEnabled()) {
235      LOG.debug("Checking znode ACLs");
236    }
237    String[] superUsers = conf.getStrings(Superusers.SUPERUSER_CONF_KEY);
238    // Check whether ACL set for all superusers
239    if (superUsers != null && !checkACLForSuperUsers(superUsers, acls)) {
240      return false;
241    }
242
243    // this assumes that current authenticated user is the same as zookeeper client user
244    // configured via JAAS
245    String hbaseUser = UserGroupInformation.getCurrentUser().getShortUserName();
246
247    if (acls.isEmpty()) {
248      if (LOG.isDebugEnabled()) {
249        LOG.debug("ACL is empty");
250      }
251      return false;
252    }
253
254    for (ACL acl : acls) {
255      int perms = acl.getPerms();
256      Id id = acl.getId();
257      // We should only set at most 3 possible ACLs for 3 Ids. One for everyone, one for superuser
258      // and one for the hbase user
259      if (Ids.ANYONE_ID_UNSAFE.equals(id)) {
260        if (perms != Perms.READ) {
261          if (LOG.isDebugEnabled()) {
262            LOG.debug(String.format("permissions for '%s' are not correct: have 0x%x, want 0x%x",
263              id, perms, Perms.READ));
264          }
265          return false;
266        }
267      } else if (superUsers != null && isSuperUserId(superUsers, id)) {
268        if (perms != Perms.ALL) {
269          if (LOG.isDebugEnabled()) {
270            LOG.debug(String.format("permissions for '%s' are not correct: have 0x%x, want 0x%x",
271              id, perms, Perms.ALL));
272          }
273          return false;
274        }
275      } else if ("sasl".equals(id.getScheme())) {
276        String name = id.getId();
277        // If ZooKeeper recorded the Kerberos full name in the ACL, use only the shortname
278        Matcher match = NAME_PATTERN.matcher(name);
279        if (match.matches()) {
280          name = match.group(1);
281        }
282        if (name.equals(hbaseUser)) {
283          if (perms != Perms.ALL) {
284            if (LOG.isDebugEnabled()) {
285              LOG.debug(String.format("permissions for '%s' are not correct: have 0x%x, want 0x%x",
286                id, perms, Perms.ALL));
287            }
288            return false;
289          }
290        } else {
291          if (LOG.isDebugEnabled()) {
292            LOG.debug("Unexpected shortname in SASL ACL: " + id);
293          }
294          return false;
295        }
296      } else {
297        if (LOG.isDebugEnabled()) {
298          LOG.debug("unexpected ACL id '" + id + "'");
299        }
300        return false;
301      }
302    }
303    return true;
304  }

306  /*
307   * Validate whether ACL set for all superusers.
308   */
309  private boolean checkACLForSuperUsers(String[] superUsers, List<ACL> acls) {
310    for (String user : superUsers) {
311      boolean hasAccess = false;
312      // TODO: Validate super group members also when ZK supports setting node ACL for groups.
313      if (!AuthUtil.isGroupPrincipal(user)) {
314        for (ACL acl : acls) {
315          if (user.equals(acl.getId().getId())) {
316            if (acl.getPerms() == Perms.ALL) {
317              hasAccess = true;
318            } else {
319              if (LOG.isDebugEnabled()) {
320                LOG.debug(String.format(
321                  "superuser '%s' does not have correct permissions: have 0x%x, want 0x%x",
322                  acl.getId().getId(), acl.getPerms(), Perms.ALL));
323              }
324            }
325            break;
326          }
327        }
328        if (!hasAccess) {
329          return false;
330        }
331      }
332    }
333    return true;
334  }

336  /*
337   * Validate whether ACL ID is superuser.
338   */
339  public static boolean isSuperUserId(String[] superUsers, Id id) {
340    for (String user : superUsers) {
341      // TODO: Validate super group members also when ZK supports setting node ACL for groups.
342      if (!AuthUtil.isGroupPrincipal(user) && new Id("sasl", user).equals(id)) {
343        return true;
344      }
345    }
346    return false;
347  }

364  /**
365   * Get the znodes corresponding to the meta replicas from ZK
366   * @return list of znodes
367   * @throws KeeperException
368   */
369  public List<String> getMetaReplicaNodes() throws KeeperException {
370    List<String> childrenOfBaseNode = ZKUtil.listChildrenNoWatch(this, znodePaths.baseZNode);
371    List<String> metaReplicaNodes = new ArrayList<>(2);
372    if (childrenOfBaseNode != null) {
373      String pattern = conf.get("zookeeper.znode.metaserver","meta-region-server");
374      for (String child : childrenOfBaseNode) {
375        if (child.startsWith(pattern)) metaReplicaNodes.add(child);
376      }
377    }
378    return metaReplicaNodes;
379  }

452  /**
453   * Method called from ZooKeeper for events and connection status.
454   * <p>
455   * Valid events are passed along to listeners.  Connection status changes
456   * are dealt with locally.
457   */
458  @Override
459  public void process(WatchedEvent event) {
460    LOG.debug(prefix("Received ZooKeeper Event, " +
461        "type=" + event.getType() + ", " +
462        "state=" + event.getState() + ", " +
463        "path=" + event.getPath()));
464
465    switch(event.getType()) {
466
467      // If event type is NONE, this is a connection status change
468      case None: {
469        connectionEvent(event);
470        break;
471      }
472
473      // Otherwise pass along to the listeners
474
475      case NodeCreated: {
476        for(ZKListener listener : listeners) {
477          listener.nodeCreated(event.getPath());
478        }
479        break;
480      }
481
482      case NodeDeleted: {
483        for(ZKListener listener : listeners) {
484          listener.nodeDeleted(event.getPath());
485        }
486        break;
487      }
488
489      case NodeDataChanged: {
490        for(ZKListener listener : listeners) {
491          listener.nodeDataChanged(event.getPath());
492        }
493        break;
494      }
495
496      case NodeChildrenChanged: {
497        for(ZKListener listener : listeners) {
498          listener.nodeChildrenChanged(event.getPath());
499        }
500        break;
501      }
502    }
503  }

507  /**
508   * Called when there is a connection-related event via the Watcher callback.
509   * <p>
510   * If Disconnected or Expired, this should shutdown the cluster. But, since
511   * we send a KeeperException.SessionExpiredException along with the abort
512   * call, it's possible for the Abortable to catch it and try to create a new
513   * session with ZooKeeper. This is what the client does in HCM.
514   * <p>
515   * @param event
516   */
517  private void connectionEvent(WatchedEvent event) {
518    switch(event.getState()) {
519      case SyncConnected:
520        this.identifier = this.prefix + "-0x" +
521          Long.toHexString(this.recoverableZooKeeper.getSessionId());
522        // Update our identifier.  Otherwise ignore.
523        LOG.debug(this.identifier + " connected");
524        break;
525
526      // Abort the server if Disconnected or Expired
527      case Disconnected:
528        LOG.debug(prefix("Received Disconnected from ZooKeeper, ignoring"));
529        break;
530
531      case Expired:
532        String msg = prefix(this.identifier + " received expired from " +
533          "ZooKeeper, aborting");
534        // TODO: One thought is to add call to ZKListener so say,
535        // ZKNodeTracker can zero out its data values.
536        if (this.abortable != null) {
537          this.abortable.abort(msg, new KeeperException.SessionExpiredException());
538        }
539        break;
540
541      case ConnectedReadOnly:
542      case SaslAuthenticated:
543      case AuthFailed:
544        break;
545
546      default:
547        throw new IllegalStateException("Received event is not valid: " + event.getState());
548    }
549  }

551  /**
552   * Forces a synchronization of this ZooKeeper client connection.
553   * <p>
554   * Executing this method before running other methods will ensure that the
555   * subsequent operations are up-to-date and consistent as of the time that
556   * the sync is complete.
557   * <p>
558   * This is used for compareAndSwap type operations where we need to read the
559   * data of an existing node and delete or transition that node, utilizing the
560   * previously read version and data.  We want to ensure that the version read
561   * is up-to-date from when we begin the operation.
562   */
563  public void sync(String path) throws KeeperException {
564    this.recoverableZooKeeper.sync(path, null, null);
565  }

583  /**
584   * Handles InterruptedExceptions in client calls.
585   * @param ie the InterruptedException instance thrown
586   * @throws KeeperException the exception to throw, transformed from the InterruptedException
587   */
588  public void interruptedException(InterruptedException ie) throws KeeperException {
589    interruptedExceptionNoThrow(ie, true);
590    // Throw a system error exception to let upper level handle it
591    throw new KeeperException.SystemErrorException();
592  }

594  /**
595   * Log the InterruptedException and interrupt current thread
596   * @param ie The IterruptedException to log
597   * @param throwLater Whether we will throw the exception latter
598   */
599  public void interruptedExceptionNoThrow(InterruptedException ie, boolean throwLater) {
600    LOG.debug(prefix("Received InterruptedException, will interrupt current thread"
601        + (throwLater ? " and rethrow a SystemErrorException" : "")),
602      ie);
603    // At least preserve interrupt.
604    Thread.currentThread().interrupt();
605  }

626    if (this.abortable != null) this.abortable.abort(why, e);
627    else this.aborted = true;
