044/**
045 * A very simple read only zookeeper implementation without watcher support.
046 */
047@InterfaceAudience.Private
048public final class ReadOnlyZKClient implements Closeable {
049
050  private static final Log LOG = LogFactory.getLog(ReadOnlyZKClient.class);
051
052  public static final String RECOVERY_RETRY = "zookeeper.recovery.retry";
053
054  private static final int DEFAULT_RECOVERY_RETRY = 30;
055
056  public static final String RECOVERY_RETRY_INTERVAL_MILLIS =
057      "zookeeper.recovery.retry.intervalmill";
058
059  private static final int DEFAULT_RECOVERY_RETRY_INTERVAL_MILLIS = 1000;
060
061  public static final String KEEPALIVE_MILLIS = "zookeeper.keep-alive.time";
062
063  private static final int DEFAULT_KEEPALIVE_MILLIS = 60000;
064
065  private static final EnumSet<Code> FAIL_FAST_CODES = EnumSet.of(Code.NOAUTH, Code.AUTHFAILED);
066
067  private final String connectString;
068
069  private final int sessionTimeoutMs;
070
071  private final int maxRetries;
072
073  private final int retryIntervalMs;
074
075  private final int keepAliveTimeMs;
076
077  private static abstract class Task implements Delayed {
078
079    protected long time = System.nanoTime();
080
081    public boolean needZk() {
082      return false;
083    }
084
085    public void exec(ZooKeeper zk) {
086    }
087
088    public void connectFailed(IOException e) {
089    }
090
091    public void closed(IOException e) {
092    }
093
094    @Override
095    public int compareTo(Delayed o) {
096      Task that = (Task) o;
097      int c = Long.compare(time, that.time);
098      if (c != 0) {
099        return c;
100      }
101      return Integer.compare(System.identityHashCode(this), System.identityHashCode(that));
102    }
103
104    @Override
105    public long getDelay(TimeUnit unit) {
106      return unit.convert(time - System.nanoTime(), TimeUnit.NANOSECONDS);
107    }
108  }
109
110  private static final Task CLOSE = new Task() {
111  };
112
113  private final DelayQueue<Task> tasks = new DelayQueue<>();
114
115  private final AtomicBoolean closed = new AtomicBoolean(false);
116
117  private ZooKeeper zookeeper;
118
119  private String getId() {
120    return String.format("0x%08x", System.identityHashCode(this));
121  }
122
123  public ReadOnlyZKClient(Configuration conf) {
124    this.connectString = ZKConfig.getZKQuorumServersString(conf);
125    this.sessionTimeoutMs = conf.getInt(ZK_SESSION_TIMEOUT, DEFAULT_ZK_SESSION_TIMEOUT);
126    this.maxRetries = conf.getInt(RECOVERY_RETRY, DEFAULT_RECOVERY_RETRY);
127    this.retryIntervalMs =
128        conf.getInt(RECOVERY_RETRY_INTERVAL_MILLIS, DEFAULT_RECOVERY_RETRY_INTERVAL_MILLIS);
129    this.keepAliveTimeMs = conf.getInt(KEEPALIVE_MILLIS, DEFAULT_KEEPALIVE_MILLIS);
130    LOG.info("Start read only zookeeper connection " + getId() + " to " + connectString +
131        ", session timeout " + sessionTimeoutMs + " ms, retries " + maxRetries +
132        ", retry interval " + retryIntervalMs + " ms, keep alive " + keepAliveTimeMs + " ms");
133    Thread t = new Thread(this::run, "ReadOnlyZKClient");
134    t.setDaemon(true);
135    t.start();
136  }
137
138  private abstract class ZKTask<T> extends Task {
139
140    protected final String path;
141
142    private final CompletableFuture<T> future;
143
144    private final String operationType;
145
146    private int retries;
147
148    protected ZKTask(String path, CompletableFuture<T> future, String operationType) {
149      this.path = path;
150      this.future = future;
151      this.operationType = operationType;
152    }
153
154    protected final void onComplete(ZooKeeper zk, int rc, T ret, boolean errorIfNoNode) {
155      tasks.add(new Task() {
156
157        @Override
158        public void exec(ZooKeeper alwaysNull) {
159          Code code = Code.get(rc);
160          if (code == Code.OK) {
161            future.complete(ret);
162          } else if (code == Code.NONODE) {
163            if (errorIfNoNode) {
164              future.completeExceptionally(KeeperException.create(code, path));
165            } else {
166              future.complete(ret);
167            }
168          } else if (FAIL_FAST_CODES.contains(code)) {
169            future.completeExceptionally(KeeperException.create(code, path));
170          } else {
171            if (code == Code.SESSIONEXPIRED) {
172              LOG.warn(getId() + " session expired, close and reconnect");
173              try {
174                zk.close();
175              } catch (InterruptedException e) {
176              }
177            }
178            if (ZKTask.this.delay(retryIntervalMs, maxRetries)) {
179              LOG.warn(getId() + " failed for " + operationType + " of " + path + ", code = " +
180                  code + ", retries = " + ZKTask.this.retries);
181              tasks.add(ZKTask.this);
182            } else {
183              LOG.warn(getId() + " failed for " + operationType + " of " + path + ", code = " +
184                  code + ", retries = " + ZKTask.this.retries + ", give up");
185              future.completeExceptionally(KeeperException.create(code, path));
186            }
187          }
188        }
189
190        @Override
191        public void closed(IOException e) {
192          // It may happen that a request is succeeded and the onComplete has been called and pushed
193          // us into the task queue, but before we get called a close is called and here we will
194          // fail the request, although it is succeeded actually.
195          // This is not a perfect solution but anyway, it is better than hang the requests for
196          // ever, and also acceptable as if you close the zk client before actually getting the
197          // response then a failure is always possible.
198          future.completeExceptionally(e);
199        }
200      });
201    }
202
203    @Override
204    public boolean needZk() {
205      return true;
206    }
207
208    public boolean delay(long intervalMs, int maxRetries) {
209      if (retries >= maxRetries) {
210        return false;
211      }
212      retries++;
213      time = System.nanoTime() + TimeUnit.MILLISECONDS.toNanos(intervalMs);
214      return true;
215    }
216
217    @Override
218    public void connectFailed(IOException e) {
219      if (delay(retryIntervalMs, maxRetries)) {
220        LOG.warn(getId() + " failed to connect to zk for " + operationType + " of " + path +
221            ", retries = " + retries,
222          e);
223        tasks.add(this);
224      } else {
225        LOG.warn(getId() + " failed to connect to zk for " + operationType + " of " + path +
226            ", retries = " + retries + ", give up",
227          e);
228        future.completeExceptionally(e);
229      }
230    }
231
232    @Override
233    public void closed(IOException e) {
234      future.completeExceptionally(e);
235    }
236  }
237
238  private static <T> CompletableFuture<T> failed(Throwable e) {
239    CompletableFuture<T> future = new CompletableFuture<>();
240    future.completeExceptionally(e);
241    return future;
242  }
243
244  public CompletableFuture<byte[]> get(String path) {
245    if (closed.get()) {
246      return failed(new IOException("Client already closed"));
247    }
248    CompletableFuture<byte[]> future = new CompletableFuture<>();
249    tasks.add(new ZKTask<byte[]>(path, future, "get") {
250
251      @Override
252      public void exec(ZooKeeper zk) {
253        zk.getData(path, false, (rc, path, ctx, data, stat) -> onComplete(zk, rc, data, true),
254          null);
255      }
256    });
257    return future;
258  }
259
260  public CompletableFuture<Stat> exists(String path) {
261    if (closed.get()) {
262      return failed(new IOException("Client already closed"));
263    }
264    CompletableFuture<Stat> future = new CompletableFuture<>();
265    tasks.add(new ZKTask<Stat>(path, future, "exists") {
266
267      @Override
268      public void exec(ZooKeeper zk) {
269        zk.exists(path, false, (rc, path, ctx, stat) -> onComplete(zk, rc, stat, false), null);
270      }
271    });
272    return future;
273  }
274
275  private void closeZk() {
276    if (zookeeper != null) {
277      try {
278        zookeeper.close();
279      } catch (InterruptedException e) {
280      }
281      zookeeper = null;
282    }
283  }
284
285  private ZooKeeper getZk() throws IOException {
286    // may be closed when session expired
287    if (zookeeper == null || !zookeeper.getState().isAlive()) {
288      zookeeper = new ZooKeeper(connectString, sessionTimeoutMs, e -> {
289      });
290    }
291    return zookeeper;
292  }
293
294  private void run() {
295    for (;;) {
296      Task task;
297      try {
298        task = tasks.poll(keepAliveTimeMs, TimeUnit.MILLISECONDS);
299      } catch (InterruptedException e) {
300        continue;
301      }
302      if (task == CLOSE) {
303        break;
304      }
305      if (task == null) {
306        LOG.info(getId() + " no activities for " + keepAliveTimeMs +
307            " ms, close active connection. Will reconnect next time when there are new requests.");
308        closeZk();
309        continue;
310      }
311      if (!task.needZk()) {
312        task.exec(null);
313      } else {
314        ZooKeeper zk;
315        try {
316          zk = getZk();
317        } catch (IOException e) {
318          task.connectFailed(e);
319          continue;
320        }
321        task.exec(zk);
322      }
323    }
324    closeZk();
325    IOException error = new IOException("Client already closed");
326    Arrays.stream(tasks.toArray(new Task[0])).forEach(t -> t.closed(error));
327    tasks.clear();
328  }
329
330  @Override
331  public void close() {
332    if (closed.compareAndSet(false, true)) {
333      LOG.info("Close zookeeper connection " + getId() + " to " + connectString);
334      tasks.add(CLOSE);
335    }
336  }
337
338  @VisibleForTesting
339  ZooKeeper getZooKeeper() {
340    return zookeeper;
341  }
342
343  @VisibleForTesting
344  public String getConnectString() {
345    return connectString;
346  }
