018package org.apache.hadoop.hbase.zookeeper;
019
020import java.io.IOException;
021
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.Abortable;
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025import org.apache.hadoop.hbase.client.MasterSwitchType;
026import org.apache.hadoop.hbase.exceptions.DeserializationException;
027import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
028import org.apache.hadoop.hbase.protobuf.generated.ZooKeeperProtos.SwitchState;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.apache.zookeeper.KeeperException;
031
032
033/**
034 * Tracks the switch of split and merge states in ZK
035 *
036 */
037@InterfaceAudience.Private
038public class SplitOrMergeTracker {
039
040  private String splitZnode;
041  private String mergeZnode;
042
043  private SwitchStateTracker splitStateTracker;
044  private SwitchStateTracker mergeStateTracker;
045
046  public SplitOrMergeTracker(ZooKeeperWatcher watcher, Configuration conf,
047                             Abortable abortable) {
048    try {
049      if (ZKUtil.checkExists(watcher, watcher.getSwitchZNode()) < 0) {
050        ZKUtil.createAndFailSilent(watcher, watcher.getSwitchZNode());
051      }
052    } catch (KeeperException e) {
053      throw new RuntimeException(e);
054    }
055    splitZnode = ZKUtil.joinZNode(watcher.getSwitchZNode(),
056      conf.get("zookeeper.znode.switch.split", "split"));
057    mergeZnode = ZKUtil.joinZNode(watcher.getSwitchZNode(),
058      conf.get("zookeeper.znode.switch.merge", "merge"));
059    splitStateTracker = new SwitchStateTracker(watcher, splitZnode, abortable);
060    mergeStateTracker = new SwitchStateTracker(watcher, mergeZnode, abortable);
061  }
062
063  public void start() {
064    splitStateTracker.start();
065    mergeStateTracker.start();
066  }
067
068  public boolean isSplitOrMergeEnabled(MasterSwitchType switchType) {
069    switch (switchType) {
070      case SPLIT:
071        return splitStateTracker.isSwitchEnabled();
072      case MERGE:
073        return mergeStateTracker.isSwitchEnabled();
074      default:
075        break;
076    }
077    return false;
078  }
079
080  public void setSplitOrMergeEnabled(boolean enabled, MasterSwitchType switchType)
081    throws KeeperException {
082    switch (switchType) {
083      case SPLIT:
084        splitStateTracker.setSwitchEnabled(enabled);
085        break;
086      case MERGE:
087        mergeStateTracker.setSwitchEnabled(enabled);
088        break;
089      default:
090        break;
091    }
092  }
093
094  private static class SwitchStateTracker extends ZooKeeperNodeTracker {
095
096    public SwitchStateTracker(ZooKeeperWatcher watcher, String node, Abortable abortable) {
097      super(watcher, node, abortable);
098    }
099
100    /**
101     * Return true if the switch is on, false otherwise
102     */
103    public boolean isSwitchEnabled() {
104      byte [] upData = super.getData(false);
105      try {
106        // if data in ZK is null, use default of on.
107        return upData == null || parseFrom(upData).getEnabled();
108      } catch (DeserializationException dex) {
109        LOG.error("ZK state for LoadBalancer could not be parsed " + Bytes.toStringBinary(upData));
110        // return false to be safe.
111        return false;
112      }
113    }
114
115    /**
116     * Set the switch on/off
117     * @param enabled switch enabled or not?
118     * @throws KeeperException keepException will be thrown out
119     */
120    public void setSwitchEnabled(boolean enabled) throws KeeperException {
121      byte [] upData = toByteArray(enabled);
122      try {
123        ZKUtil.setData(watcher, node, upData);
124      } catch(KeeperException.NoNodeException nne) {
125        ZKUtil.createAndWatch(watcher, node, upData);
126      }
127      super.nodeDataChanged(node);
128    }
129
130    private byte [] toByteArray(boolean enabled) {
131      SwitchState.Builder builder = SwitchState.newBuilder();
132      builder.setEnabled(enabled);
133      return ProtobufUtil.prependPBMagic(builder.build().toByteArray());
134    }
135
136    private SwitchState parseFrom(byte [] bytes)
137      throws DeserializationException {
138      ProtobufUtil.expectPBMagicPrefix(bytes);
139      SwitchState.Builder builder = SwitchState.newBuilder();
140      try {
141        int magicLen = ProtobufUtil.lengthOfPBMagic();
142        ProtobufUtil.mergeFrom(builder, bytes, magicLen, bytes.length - magicLen);
143      } catch (IOException e) {
144        throw new DeserializationException(e);
145      }
146      return builder.build();
147    }
148  }
149}
