049/**
050 * A zookeeper that can handle 'recoverable' errors.
051 * To handle recoverable errors, developers need to realize that there are two
052 * classes of requests: idempotent and non-idempotent requests. Read requests
053 * and unconditional sets and deletes are examples of idempotent requests, they
054 * can be reissued with the same results.
055 * (Although, the delete may throw a NoNodeException on reissue its effect on
056 * the ZooKeeper state is the same.) Non-idempotent requests need special
057 * handling, application and library writers need to keep in mind that they may
058 * need to encode information in the data or name of znodes to detect
059 * retries. A simple example is a create that uses a sequence flag.
060 * If a process issues a create("/x-", ..., SEQUENCE) and gets a connection
061 * loss exception, that process will reissue another
062 * create("/x-", ..., SEQUENCE) and get back x-111. When the process does a
063 * getChildren("/"), it sees x-1,x-30,x-109,x-110,x-111, now it could be
064 * that x-109 was the result of the previous create, so the process actually
065 * owns both x-109 and x-111. An easy way around this is to use "x-process id-"
066 * when doing the create. If the process is using an id of 352, before reissuing
067 * the create it will do a getChildren("/") and see "x-222-1", "x-542-30",
068 * "x-352-109", x-333-110". The process will know that the original create
069 * succeeded an the znode it created is "x-352-109".
070 * @see "http://wiki.apache.org/hadoop/ZooKeeper/ErrorHandling"
071 */
072@InterfaceAudience.Private
073public class RecoverableZooKeeper {

086  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="DE_MIGHT_IGNORE",
087      justification="None. Its always been this way.")
088  public RecoverableZooKeeper(String quorumServers, int sessionTimeout,
089      Watcher watcher, int maxRetries, int retryIntervalMillis, int maxSleepTime, String identifier,
090      int maxMultiSize)
091    throws IOException {
092    // TODO: Add support for zk 'chroot'; we don't add it to the quorumServers String as we should.
093    this.retryCounterFactory =
094      new RetryCounterFactory(maxRetries+1, retryIntervalMillis, maxSleepTime);
095
096    if (identifier == null || identifier.length() == 0) {
097      // the identifier = processID@hostName
098      identifier = ManagementFactory.getRuntimeMXBean().getName();
099    }
100    LOG.info("Process identifier=" + identifier +
101      " connecting to ZooKeeper ensemble=" + quorumServers);
102    this.identifier = identifier;
103    this.id = Bytes.toBytes(identifier);
104
105    this.watcher = watcher;
106    this.sessionTimeout = sessionTimeout;
107    this.quorumServers = quorumServers;
108    this.maxMultiSize = maxMultiSize;
109
110    try {
111      checkZk();
112    } catch (Exception x) {
113      /* ignore */
114    }
115  }

128  /**
129   * Try to create a ZooKeeper connection. Turns any exception encountered into a
130   * KeeperException.OperationTimeoutException so it can retried.
131   * @return The created ZooKeeper connection object
132   * @throws KeeperException if a ZooKeeper operation fails
133   */
134  protected synchronized ZooKeeper checkZk() throws KeeperException {
135    if (this.zk == null) {
136      try {
137        this.zk = new ZooKeeper(quorumServers, sessionTimeout, watcher);
138      } catch (IOException ex) {
139        LOG.warn("Unable to create ZooKeeper Connection", ex);
140        throw new KeeperException.OperationTimeoutException();
141      }
142    }
143    return zk;
144  }
145
146  public synchronized void reconnectAfterExpiration()
147        throws IOException, KeeperException, InterruptedException {
148    if (zk != null) {
149      LOG.info("Closing dead ZooKeeper connection, session" +
150        " was: 0x"+Long.toHexString(zk.getSessionId()));
151      zk.close();
152      // reset the ZooKeeper connection
153      zk = null;
154    }
155    checkZk();
156    LOG.info("Recreated a ZooKeeper, session" +
157      " is: 0x"+Long.toHexString(zk.getSessionId()));
158  }

160  /**
161   * delete is an idempotent operation. Retry before throwing exception.
162   * This function will not throw NoNodeException if the path does not
163   * exist.
164   */
165  public void delete(String path, int version) throws InterruptedException, KeeperException {
166    try (TraceScope scope = TraceUtil.createTrace("RecoverableZookeeper.delete")) {
167      RetryCounter retryCounter = retryCounterFactory.create();
168      boolean isRetry = false; // False for first attempt, true for all retries.
169      while (true) {
170        try {
171          checkZk().delete(path, version);
172          return;
173        } catch (KeeperException e) {
174          switch (e.code()) {
175            case NONODE:
176              if (isRetry) {
177                LOG.debug("Node " + path + " already deleted. Assuming a " +
178                    "previous attempt succeeded.");
179                return;
180              }
181              LOG.debug("Node " + path + " already deleted, retry=" + isRetry);
182              throw e;
183
184            case CONNECTIONLOSS:
185              retryOrThrow(retryCounter, e, "delete");
186              break;
187            case OPERATIONTIMEOUT:
188              retryOrThrow(retryCounter, e, "delete");
189              break;
190
191            default:
192              throw e;
193          }
194        }
195        retryCounter.sleepUntilNextRetry();
196        isRetry = true;
197      }
198    }
199  }

209  private Stat exists(String path, Watcher watcher, Boolean watch)
210          throws InterruptedException, KeeperException {
211    try (TraceScope scope = TraceUtil.createTrace("RecoverableZookeeper.exists")) {
212      RetryCounter retryCounter = retryCounterFactory.create();
213      while (true) {
214        try {
215          Stat nodeStat;
216          if (watch == null) {
217            nodeStat = checkZk().exists(path, watcher);
218          } else {
219            nodeStat = checkZk().exists(path, watch);
220          }
221          return nodeStat;
222        } catch (KeeperException e) {
223          switch (e.code()) {
224            case CONNECTIONLOSS:
225              retryOrThrow(retryCounter, e, "exists");
226              break;
227            case OPERATIONTIMEOUT:
228              retryOrThrow(retryCounter, e, "exists");
229              break;
230
231            default:
232              throw e;
233          }
234        }
235        retryCounter.sleepUntilNextRetry();
236      }
237    }
238  }

248  private void retryOrThrow(RetryCounter retryCounter, KeeperException e,
249      String opName) throws KeeperException {
250    if (!retryCounter.shouldRetry()) {
251      LOG.error("ZooKeeper " + opName + " failed after "
252        + retryCounter.getMaxAttempts() + " attempts");
253      throw e;
254    }
255    LOG.debug("Retry, connectivity issue (JVM Pause?); quorum=" + quorumServers + "," +
256        "exception=" + e);
257  }

317  private byte[] getData(String path, Watcher watcher, Boolean watch, Stat stat)
318          throws InterruptedException, KeeperException {
319    try (TraceScope scope = TraceUtil.createTrace("RecoverableZookeeper.getData")) {

321      while (true) {

330        } catch (KeeperException e) {

332            case CONNECTIONLOSS:
333              retryOrThrow(retryCounter, e, "getData");
334              break;
335            case OPERATIONTIMEOUT:
336              retryOrThrow(retryCounter, e, "getData");
337              break;
338
339            default:
340              throw e;
341          }
342        }
343        retryCounter.sleepUntilNextRetry();
344      }

357  /**
358   * setData is NOT an idempotent operation. Retry may cause BadVersion Exception
359   * Adding an identifier field into the data to check whether
360   * badversion is caused by the result of previous correctly setData
361   * @return Stat instance
362   */
363  public Stat setData(String path, byte[] data, int version)
364    throws KeeperException, InterruptedException {
365    try (TraceScope scope = TraceUtil.createTrace("RecoverableZookeeper.setData")) {
366      RetryCounter retryCounter = retryCounterFactory.create();
367      byte[] newData = ZKMetadata.appendMetaData(id, data);
368      boolean isRetry = false;
369      while (true) {
370        try {
371          return checkZk().setData(path, newData, version);
372        } catch (KeeperException e) {
373          switch (e.code()) {
374            case CONNECTIONLOSS:
375              retryOrThrow(retryCounter, e, "setData");
376              break;
377            case OPERATIONTIMEOUT:
378              retryOrThrow(retryCounter, e, "setData");
379              break;
380            case BADVERSION:
381              if (isRetry) {
382                // try to verify whether the previous setData success or not
383                try{
384                  Stat stat = new Stat();
385                  byte[] revData = checkZk().getData(path, false, stat);
386                  if(Bytes.compareTo(revData, newData) == 0) {
387                    // the bad version is caused by previous successful setData
388                    return stat;
389                  }
390                } catch(KeeperException keeperException){
391                  // the ZK is not reliable at this moment. just throwing exception
392                  throw keeperException;
393                }
394              }
395            // throw other exceptions and verified bad version exceptions
396            default:
397              throw e;
398          }
399        }
400        retryCounter.sleepUntilNextRetry();
401        isRetry = true;
402      }
403    }
404  }

464  /**
465   * <p>
466   * NONSEQUENTIAL create is idempotent operation.
467   * Retry before throwing exceptions.
468   * But this function will not throw the NodeExist exception back to the
469   * application.
470   * </p>
471   * <p>
472   * But SEQUENTIAL is NOT idempotent operation. It is necessary to add
473   * identifier to the path to verify, whether the previous one is successful
474   * or not.
475   * </p>
476   *
477   * @return Path
478   */
479  public String create(String path, byte[] data, List<ACL> acl,
480      CreateMode createMode)
481    throws KeeperException, InterruptedException {
482    try (TraceScope scope = TraceUtil.createTrace("RecoverableZookeeper.create")) {
483      byte[] newData = ZKMetadata.appendMetaData(id, data);
484      switch (createMode) {
485        case EPHEMERAL:
486        case PERSISTENT:
487          return createNonSequential(path, newData, acl, createMode);
488
489        case EPHEMERAL_SEQUENTIAL:
490        case PERSISTENT_SEQUENTIAL:
491          return createSequential(path, newData, acl, createMode);
492
493        default:
494          throw new IllegalArgumentException("Unrecognized CreateMode: " +
495              createMode);
496      }
497    }
498  }
499
500  private String createNonSequential(String path, byte[] data, List<ACL> acl,
501      CreateMode createMode) throws KeeperException, InterruptedException {
502    RetryCounter retryCounter = retryCounterFactory.create();
503    boolean isRetry = false; // False for first attempt, true for all retries.
504    while (true) {
505      try {
506        return checkZk().create(path, data, acl, createMode);
507      } catch (KeeperException e) {
508        switch (e.code()) {
509          case NODEEXISTS:
510            if (isRetry) {
511              // If the connection was lost, there is still a possibility that
512              // we have successfully created the node at our previous attempt,
513              // so we read the node and compare.
514              byte[] currentData = checkZk().getData(path, false, null);
515              if (currentData != null &&
516                  Bytes.compareTo(currentData, data) == 0) {
517                // We successfully created a non-sequential node
518                return path;
519              }
520              LOG.error("Node " + path + " already exists with " +
521                  Bytes.toStringBinary(currentData) + ", could not write " +
522                  Bytes.toStringBinary(data));
523              throw e;
524            }
525            LOG.trace("Node {} already exists", path);
526            throw e;
527
528          case CONNECTIONLOSS:
529            retryOrThrow(retryCounter, e, "create");
530            break;
531          case OPERATIONTIMEOUT:
532            retryOrThrow(retryCounter, e, "create");
533            break;
534
535          default:
536            throw e;
537        }
538      }
539      retryCounter.sleepUntilNextRetry();
540      isRetry = true;
541    }
542  }

544  private String createSequential(String path, byte[] data,
545      List<ACL> acl, CreateMode createMode)
546    throws KeeperException, InterruptedException {
547    RetryCounter retryCounter = retryCounterFactory.create();
548    boolean first = true;
549    String newPath = path+this.identifier;
550    while (true) {
551      try {
552        if (!first) {
553          // Check if we succeeded on a previous attempt
554          String previousResult = findPreviousSequentialNode(newPath);
555          if (previousResult != null) {
556            return previousResult;
557          }
558        }
559        first = false;
560        return checkZk().create(newPath, data, acl, createMode);
561      } catch (KeeperException e) {
562        switch (e.code()) {
563          case CONNECTIONLOSS:
564            retryOrThrow(retryCounter, e, "create");
565            break;
566          case OPERATIONTIMEOUT:
567            retryOrThrow(retryCounter, e, "create");
568            break;
569
570          default:
571            throw e;
572        }
573      }
574      retryCounter.sleepUntilNextRetry();
575    }
576  }
577  /**
578   * Convert Iterable of {@link org.apache.zookeeper.Op} we got into the ZooKeeper.Op
579   * instances to actually pass to multi (need to do this in order to appendMetaData).
580   */
581  private Iterable<Op> prepareZKMulti(Iterable<Op> ops) throws UnsupportedOperationException {
582    if(ops == null) {
583      return null;
584    }
585
586    List<Op> preparedOps = new LinkedList<>();
587    for (Op op : ops) {
588      if (op.getType() == ZooDefs.OpCode.create) {
589        CreateRequest create = (CreateRequest)op.toRequestRecord();
590        preparedOps.add(Op.create(create.getPath(), ZKMetadata.appendMetaData(id, create.getData()),
591          create.getAcl(), create.getFlags()));
592      } else if (op.getType() == ZooDefs.OpCode.delete) {
593        // no need to appendMetaData for delete
594        preparedOps.add(op);
595      } else if (op.getType() == ZooDefs.OpCode.setData) {
596        SetDataRequest setData = (SetDataRequest)op.toRequestRecord();
597        preparedOps.add(Op.setData(setData.getPath(),
598                ZKMetadata.appendMetaData(id, setData.getData()), setData.getVersion()));
599      } else {
600        throw new UnsupportedOperationException("Unexpected ZKOp type: " + op.getClass().getName());
601      }
602    }
603    return preparedOps;
604  }

606  /**
607   * Run multiple operations in a transactional manner. Retry before throwing exception
608   */
609  public List<OpResult> multi(Iterable<Op> ops)
610    throws KeeperException, InterruptedException {
611    try (TraceScope scope = TraceUtil.createTrace("RecoverableZookeeper.multi")) {
612      RetryCounter retryCounter = retryCounterFactory.create();
613      Iterable<Op> multiOps = prepareZKMulti(ops);
614      while (true) {
615        try {
616          return checkZk().multi(multiOps);
617        } catch (KeeperException e) {
618          switch (e.code()) {
619            case CONNECTIONLOSS:
620              retryOrThrow(retryCounter, e, "multi");
621              break;
622            case OPERATIONTIMEOUT:
623              retryOrThrow(retryCounter, e, "multi");
624              break;
625
626            default:
627              throw e;
628          }
629        }
630        retryCounter.sleepUntilNextRetry();
631      }
632    }
633  }

635  private String findPreviousSequentialNode(String path)
636    throws KeeperException, InterruptedException {
637    int lastSlashIdx = path.lastIndexOf('/');
638    assert(lastSlashIdx != -1);
639    String parent = path.substring(0, lastSlashIdx);
640    String nodePrefix = path.substring(lastSlashIdx+1);
641    List<String> nodes = checkZk().getChildren(parent, false);
642    List<String> matching = filterByPrefix(nodes, nodePrefix);
643    for (String node : matching) {
644      String nodePath = parent + "/" + node;
645      Stat stat = checkZk().exists(nodePath, false);
646      if (stat != null) {
647        return nodePath;
648      }
649    }
650    return null;
651  }

675  /**
676   * Filters the given node list by the given prefixes.
677   * This method is all-inclusive--if any element in the node list starts
678   * with any of the given prefixes, then it is included in the result.
679   *
680   * @param nodes the nodes to filter
681   * @param prefixes the prefixes to include in the result
682   * @return list of every element that starts with one of the prefixes
683   */
684  private static List<String> filterByPrefix(List<String> nodes,
685      String... prefixes) {
686    List<String> lockChildren = new ArrayList<>();
687    for (String child : nodes){
688      for (String prefix : prefixes){
689        if (child.startsWith(prefix)){
690          lockChildren.add(child);
691          break;
692        }
693      }
694    }
695    return lockChildren;
696  }
