052/**
053 * Acts as the single ZooKeeper Watcher.  One instance of this is instantiated
054 * for each Master, RegionServer, and client process.
055 *
056 * <p>This is the only class that implements {@link Watcher}.  Other internal
057 * classes which need to be notified of ZooKeeper events must register with
058 * the local instance of this watcher via {@link #registerListener}.
059 *
060 * <p>This class also holds and manages the connection to ZooKeeper.  Code to
061 * deal with connection related events and exceptions are handled here.
062 */
063@InterfaceAudience.Private
064public class ZooKeeperWatcher implements Watcher, Abortable, Closeable {

111  /**
112   * Instantiate a ZooKeeper connection and watcher.
113   * @param conf
114   * @param identifier string that is passed to RecoverableZookeeper to be used as identifier for
115   *          this instance. Use null for default.
116   * @param abortable Can be null if there is on error there is no host to abort: e.g. client
117   *          context.
118   * @param canCreateBaseZNode
119   * @throws IOException
120   * @throws ZooKeeperConnectionException
121   */
122  public ZooKeeperWatcher(Configuration conf, String identifier,
123      Abortable abortable, boolean canCreateBaseZNode)
124  throws IOException, ZooKeeperConnectionException {
125    this.conf = conf;
126    this.quorum = ZKConfig.getZKQuorumServersString(conf);
127    this.prefix = identifier;
128    // Identifier will get the sessionid appended later below down when we
129    // handle the syncconnect event.
130    this.identifier = identifier + "0x0";
131    this.abortable = abortable;
132    this.znodePaths = new ZNodePaths(conf);
133    PendingWatcher pendingWatcher = new PendingWatcher();
134    this.recoverableZooKeeper = ZKUtil.connect(conf, quorum, pendingWatcher, identifier);
135    pendingWatcher.prepare(this);
136    if (canCreateBaseZNode) {
137      try {
138        createBaseZNodes();
139      } catch (ZooKeeperConnectionException zce) {
140        try {
141          this.recoverableZooKeeper.close();
142        } catch (InterruptedException ie) {
143          LOG.debug("Encountered InterruptedException when closing " + this.recoverableZooKeeper);
144          Thread.currentThread().interrupt();
145        }
146        throw zce;
147      }
148    }
149  }

151  private void createBaseZNodes() throws ZooKeeperConnectionException {
152    try {
153      // Create all the necessary "directories" of znodes
154      ZKUtil.createWithParents(this, znodePaths.baseZNode);
155      ZKUtil.createAndFailSilent(this, znodePaths.rsZNode);
156      ZKUtil.createAndFailSilent(this, znodePaths.drainingZNode);
157      ZKUtil.createAndFailSilent(this, znodePaths.tableZNode);
158      ZKUtil.createAndFailSilent(this, znodePaths.splitLogZNode);
159      ZKUtil.createAndFailSilent(this, znodePaths.backupMasterAddressesZNode);
160      ZKUtil.createAndFailSilent(this, znodePaths.tableLockZNode);
161      ZKUtil.createAndFailSilent(this, znodePaths.recoveringRegionsZNode);
162      ZKUtil.createAndFailSilent(this, znodePaths.masterMaintZNode);
163    } catch (KeeperException e) {
164      throw new ZooKeeperConnectionException(
165          prefix("Unexpected KeeperException creating base node"), e);
166    }
167  }

169  /** Returns whether the znode is supposed to be readable by the client
170   * and DOES NOT contain sensitive information (world readable).*/
171  public boolean isClientReadable(String node) {
172    // Developer notice: These znodes are world readable. DO NOT add more znodes here UNLESS
173    // all clients need to access this data to work. Using zk for sharing data to clients (other
174    // than service lookup case is not a recommended design pattern.
175    return
176        node.equals(znodePaths.baseZNode) ||
177        znodePaths.isAnyMetaReplicaZNode(node) ||
178        node.equals(znodePaths.masterAddressZNode) ||
179        node.equals(znodePaths.clusterIdZNode)||
180        node.equals(znodePaths.rsZNode) ||
181        // /hbase/table and /hbase/table/foo is allowed, /hbase/table-lock is not
182        node.equals(znodePaths.tableZNode) ||
183        node.startsWith(znodePaths.tableZNode + "/");
184  }

186  /**
187   * On master start, we check the znode ACLs under the root directory and set the ACLs properly
188   * if needed. If the cluster goes from an unsecure setup to a secure setup, this step is needed
189   * so that the existing znodes created with open permissions are now changed with restrictive
190   * perms.
191   */
192  public void checkAndSetZNodeAcls() {
193    if (!ZKUtil.isSecureZooKeeper(getConfiguration())) {
194      LOG.info("not a secure deployment, proceeding");
195      return;
196    }
197
198    // Check the base znodes permission first. Only do the recursion if base znode's perms are not
199    // correct.
200    try {
201      List<ACL> actualAcls = recoverableZooKeeper.getAcl(znodePaths.baseZNode, new Stat());
202
203      if (!isBaseZnodeAclSetup(actualAcls)) {
204        LOG.info("setting znode ACLs");
205        setZnodeAclsRecursive(znodePaths.baseZNode);
206      }
207    } catch(KeeperException.NoNodeException nne) {
208      return;
209    } catch(InterruptedException ie) {
210      interruptedException(ie);
211    } catch (IOException|KeeperException e) {
212      LOG.warn("Received exception while checking and setting zookeeper ACLs", e);
213    }
214  }
215
216  /**
217   * Set the znode perms recursively. This will do post-order recursion, so that baseZnode ACLs
218   * will be set last in case the master fails in between.
219   * @param znode
220   */
221  private void setZnodeAclsRecursive(String znode) throws KeeperException, InterruptedException {
222    List<String> children = recoverableZooKeeper.getChildren(znode, false);
223
224    for (String child : children) {
225      setZnodeAclsRecursive(ZKUtil.joinZNode(znode, child));
226    }
227    List<ACL> acls = ZKUtil.createACL(this, znode, true);
228    LOG.info("Setting ACLs for znode:" + znode + " , acl:" + acls);
229    recoverableZooKeeper.setAcl(znode, acls, -1);
230  }

232  /**
233   * Checks whether the ACLs returned from the base znode (/hbase) is set for secure setup.
234   * @param acls acls from zookeeper
235   * @return whether ACLs are set for the base znode
236   * @throws IOException
237   */
238  private boolean isBaseZnodeAclSetup(List<ACL> acls) throws IOException {
239    if (LOG.isDebugEnabled()) {
240      LOG.debug("Checking znode ACLs");
241    }
242    String[] superUsers = conf.getStrings(Superusers.SUPERUSER_CONF_KEY);
243    // Check whether ACL set for all superusers
244    if (superUsers != null && !checkACLForSuperUsers(superUsers, acls)) {
245      return false;
246    }
247
248    // this assumes that current authenticated user is the same as zookeeper client user
249    // configured via JAAS
250    String hbaseUser = UserGroupInformation.getCurrentUser().getShortUserName();
251
252    if (acls.isEmpty()) {
253      if (LOG.isDebugEnabled()) {
254        LOG.debug("ACL is empty");
255      }
256      return false;
257    }
258
259    for (ACL acl : acls) {
260      int perms = acl.getPerms();
261      Id id = acl.getId();
262      // We should only set at most 3 possible ACLs for 3 Ids. One for everyone, one for superuser
263      // and one for the hbase user
264      if (Ids.ANYONE_ID_UNSAFE.equals(id)) {
265        if (perms != Perms.READ) {
266          if (LOG.isDebugEnabled()) {
267            LOG.debug(String.format("permissions for '%s' are not correct: have 0x%x, want 0x%x",
268              id, perms, Perms.READ));
269          }
270          return false;
271        }
272      } else if (superUsers != null && isSuperUserId(superUsers, id)) {
273        if (perms != Perms.ALL) {
274          if (LOG.isDebugEnabled()) {
275            LOG.debug(String.format("permissions for '%s' are not correct: have 0x%x, want 0x%x",
276              id, perms, Perms.ALL));
277          }
278          return false;
279        }
280      } else if ("sasl".equals(id.getScheme())) {
281        String name = id.getId();
282        // If ZooKeeper recorded the Kerberos full name in the ACL, use only the shortname
283        Matcher match = NAME_PATTERN.matcher(name);
284        if (match.matches()) {
285          name = match.group(1);
286        }
287        if (name.equals(hbaseUser)) {
288          if (perms != Perms.ALL) {
289            if (LOG.isDebugEnabled()) {
290              LOG.debug(String.format("permissions for '%s' are not correct: have 0x%x, want 0x%x",
291                id, perms, Perms.ALL));
292            }
293            return false;
294          }
295        } else {
296          if (LOG.isDebugEnabled()) {
297            LOG.debug("Unexpected shortname in SASL ACL: " + id);
298          }
299          return false;
300        }
301      } else {
302        if (LOG.isDebugEnabled()) {
303          LOG.debug("unexpected ACL id '" + id + "'");
304        }
305        return false;
306      }
307    }
308    return true;
309  }

311  /*
312   * Validate whether ACL set for all superusers.
313   */
314  private boolean checkACLForSuperUsers(String[] superUsers, List<ACL> acls) {
315    for (String user : superUsers) {
316      boolean hasAccess = false;
317      // TODO: Validate super group members also when ZK supports setting node ACL for groups.
318      if (!AuthUtil.isGroupPrincipal(user)) {
319        for (ACL acl : acls) {
320          if (user.equals(acl.getId().getId())) {
321            if (acl.getPerms() == Perms.ALL) {
322              hasAccess = true;
323            } else {
324              if (LOG.isDebugEnabled()) {
325                LOG.debug(String.format(
326                  "superuser '%s' does not have correct permissions: have 0x%x, want 0x%x",
327                  acl.getId().getId(), acl.getPerms(), Perms.ALL));
328              }
329            }
330            break;
331          }
332        }
333        if (!hasAccess) {
334          return false;
335        }
336      }
337    }
338    return true;
339  }

341  /*
342   * Validate whether ACL ID is superuser.
343   */
344  public static boolean isSuperUserId(String[] superUsers, Id id) {
345    for (String user : superUsers) {
346      // TODO: Validate super group members also when ZK supports setting node ACL for groups.
347      if (!AuthUtil.isGroupPrincipal(user) && new Id("sasl", user).equals(id)) {
348        return true;
349      }
350    }
351    return false;
352  }

369  /**
370   * Get the znodes corresponding to the meta replicas from ZK
371   * @return list of znodes
372   * @throws KeeperException
373   */
374  public List<String> getMetaReplicaNodes() throws KeeperException {
375    List<String> childrenOfBaseNode = ZKUtil.listChildrenNoWatch(this, znodePaths.baseZNode);
376    List<String> metaReplicaNodes = new ArrayList<>(2);
377    if (childrenOfBaseNode != null) {
378      String pattern = conf.get("zookeeper.znode.metaserver","meta-region-server");
379      for (String child : childrenOfBaseNode) {
380        if (child.startsWith(pattern)) metaReplicaNodes.add(child);
381      }
382    }
383    return metaReplicaNodes;
384  }

457  /**
458   * Method called from ZooKeeper for events and connection status.
459   * <p>
460   * Valid events are passed along to listeners.  Connection status changes
461   * are dealt with locally.
462   */
463  @Override
464  public void process(WatchedEvent event) {
465    LOG.debug(prefix("Received ZooKeeper Event, " +
466        "type=" + event.getType() + ", " +
467        "state=" + event.getState() + ", " +
468        "path=" + event.getPath()));
469
470    switch(event.getType()) {
471
472      // If event type is NONE, this is a connection status change
473      case None: {
474        connectionEvent(event);
475        break;
476      }
477
478      // Otherwise pass along to the listeners
479
480      case NodeCreated: {
481        for(ZooKeeperListener listener : listeners) {
482          listener.nodeCreated(event.getPath());
483        }
484        break;
485      }
486
487      case NodeDeleted: {
488        for(ZooKeeperListener listener : listeners) {
489          listener.nodeDeleted(event.getPath());
490        }
491        break;
492      }
493
494      case NodeDataChanged: {
495        for(ZooKeeperListener listener : listeners) {
496          listener.nodeDataChanged(event.getPath());
497        }
498        break;
499      }
500
501      case NodeChildrenChanged: {
502        for(ZooKeeperListener listener : listeners) {
503          listener.nodeChildrenChanged(event.getPath());
504        }
505        break;
506      }
507    }
508  }

512  /**
513   * Called when there is a connection-related event via the Watcher callback.
514   * <p>
515   * If Disconnected or Expired, this should shutdown the cluster. But, since
516   * we send a KeeperException.SessionExpiredException along with the abort
517   * call, it's possible for the Abortable to catch it and try to create a new
518   * session with ZooKeeper. This is what the client does in HCM.
519   * <p>
520   * @param event
521   */
522  private void connectionEvent(WatchedEvent event) {
523    switch(event.getState()) {
524      case SyncConnected:
525        this.identifier = this.prefix + "-0x" +
526          Long.toHexString(this.recoverableZooKeeper.getSessionId());
527        // Update our identifier.  Otherwise ignore.
528        LOG.debug(this.identifier + " connected");
529        break;
530
531      // Abort the server if Disconnected or Expired
532      case Disconnected:
533        LOG.debug(prefix("Received Disconnected from ZooKeeper, ignoring"));
534        break;
535
536      case Expired:
537        String msg = prefix(this.identifier + " received expired from " +
538          "ZooKeeper, aborting");
539        // TODO: One thought is to add call to ZooKeeperListener so say,
540        // ZooKeeperNodeTracker can zero out its data values.
541        if (this.abortable != null) {
542          this.abortable.abort(msg, new KeeperException.SessionExpiredException());
543        }
544        break;
545
546      case ConnectedReadOnly:
547      case SaslAuthenticated:
548      case AuthFailed:
549        break;
550
551      default:
552        throw new IllegalStateException("Received event is not valid: " + event.getState());
553    }
554  }

556  /**
557   * Forces a synchronization of this ZooKeeper client connection.
558   * <p>
559   * Executing this method before running other methods will ensure that the
560   * subsequent operations are up-to-date and consistent as of the time that
561   * the sync is complete.
562   * <p>
563   * This is used for compareAndSwap type operations where we need to read the
564   * data of an existing node and delete or transition that node, utilizing the
565   * previously read version and data.  We want to ensure that the version read
566   * is up-to-date from when we begin the operation.
567   */
568  public void sync(String path) throws KeeperException {
569    this.recoverableZooKeeper.sync(path, null, null);
570  }

623  @Override
624  public void abort(String why, Throwable e) {
625    if (this.abortable != null) this.abortable.abort(why, e);
626    else this.aborted = true;
