062/**
063 * Utility class to perform operation (get/wait for/verify/set/delete) on znode in ZooKeeper
064 * which keeps hbase:meta region server location.
065 *
066 * Stateless class with a bunch of static methods. Doesn't manage resources passed in
067 * (e.g. Connection, ZooKeeperWatcher etc).
068 *
069 * Meta region location is set by <code>RegionServerServices</code>.
070 * This class doesn't use ZK watchers, rather accesses ZK directly.
071 *
072 * This class it stateless. The only reason it's not made a non-instantiable util class
073 * with a collection of static methods is that it'd be rather hard to mock properly in tests.
074 *
075 * TODO: rewrite using RPC calls to master to find out about hbase:meta.
076 */
077@InterfaceAudience.Private
078public class MetaTableLocator {
079  private static final Log LOG = LogFactory.getLog(MetaTableLocator.class);
080
081  // only needed to allow non-timeout infinite waits to stop when cluster shuts down
082  private volatile boolean stopped = false;

135  private List<HRegionInfo> getListOfHRegionInfos(
136      final List<Pair<HRegionInfo, ServerName>> pairs) {
137    if (pairs == null || pairs.isEmpty()) return null;
138    List<HRegionInfo> result = new ArrayList<HRegionInfo>(pairs.size());
139    for (Pair<HRegionInfo, ServerName> pair: pairs) {
140      result.add(pair.getFirst());
141    }
142    return result;
143  }

191  /**
192   * Gets the meta region location, if available, and waits for up to the
193   * specified timeout if not immediately available.
194   * Given the zookeeper notification could be delayed, we will try to
195   * get the latest data.
196   * @param zkw
197   * @param replicaId
198   * @param timeout maximum time to wait, in millis
199   * @return server name for server hosting meta region formatted as per
200   * {@link ServerName}, or null if none available
201   * @throws InterruptedException
202   * @throws NotAllMetaRegionsOnlineException
203   */
204  public ServerName waitMetaRegionLocation(ZooKeeperWatcher zkw, int replicaId, long timeout)
205  throws InterruptedException, NotAllMetaRegionsOnlineException {
206    try {
207      if (ZKUtil.checkExists(zkw, zkw.znodePaths.baseZNode) == -1) {
208        String errorMsg = "Check the value configured in 'zookeeper.znode.parent'. "
209            + "There could be a mismatch with the one configured in the master.";
210        LOG.error(errorMsg);
211        throw new IllegalArgumentException(errorMsg);
212      }
213    } catch (KeeperException e) {
214      throw new IllegalStateException("KeeperException while trying to check baseZNode:", e);
215    }
216    ServerName sn = blockUntilAvailable(zkw, replicaId, timeout);
217
218    if (sn == null) {
219      throw new NotAllMetaRegionsOnlineException("Timed out; " + timeout + "ms");
220    }
221
222    return sn;
223  }

225  /**
226   * Waits indefinitely for availability of <code>hbase:meta</code>.  Used during
227   * cluster startup.  Does not verify meta, just that something has been
228   * set up in zk.
229   * @see #waitMetaRegionLocation(org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher, long)
230   * @throws InterruptedException if interrupted while waiting
231   */
232  public void waitMetaRegionLocation(ZooKeeperWatcher zkw) throws InterruptedException {
233    long startTime = System.currentTimeMillis();
234    while (!stopped) {
235      try {
236        if (waitMetaRegionLocation(zkw, 100) != null) break;
237        long sleepTime = System.currentTimeMillis() - startTime;
238        // +1 in case sleepTime=0
239        if ((sleepTime + 1) % 10000 == 0) {
240          LOG.warn("Have been waiting for meta to be assigned for " + sleepTime + "ms");
241        }
242      } catch (NotAllMetaRegionsOnlineException e) {
243        if (LOG.isTraceEnabled()) {
244          LOG.trace("hbase:meta still not available, sleeping and retrying." +
245            " Reason: " + e.getMessage());
246        }
247      }
248    }
249  }

267  /**
268   * Verify <code>hbase:meta</code> is deployed and accessible.
269   * @param connection
270   * @param zkw
271   * @param timeout How long to wait on zk for meta address (passed through to
272   * @param replicaId
273   * @return True if the <code>hbase:meta</code> location is healthy.
274   * @throws InterruptedException
275   * @throws IOException
276   */
277  public boolean verifyMetaRegionLocation(ClusterConnection connection,
278      ZooKeeperWatcher zkw, final long timeout, int replicaId)
279  throws InterruptedException, IOException {
280    AdminProtos.AdminService.BlockingInterface service = null;
281    try {
282      service = getMetaServerConnection(connection, zkw, timeout, replicaId);
283    } catch (NotAllMetaRegionsOnlineException e) {
284      // Pass
285    } catch (ServerNotRunningYetException e) {
286      // Pass -- remote server is not up so can't be carrying root
287    } catch (UnknownHostException e) {
288      // Pass -- server name doesn't resolve so it can't be assigned anything.
289    } catch (RegionServerStoppedException e) {
290      // Pass -- server name sends us to a server that is dying or already dead.
291    }
292    return (service != null) && verifyRegionLocation(connection, service,
293            getMetaRegionLocation(zkw, replicaId), RegionReplicaUtil.getRegionInfoForReplica(
294                HRegionInfo.FIRST_META_REGIONINFO, replicaId).getRegionName());
295  }

297  /**
298   * Verify we can connect to <code>hostingServer</code> and that its carrying
299   * <code>regionName</code>.
300   * @param hostingServer Interface to the server hosting <code>regionName</code>
301   * @param address The servername that goes with the <code>metaServer</code>
302   * Interface.  Used logging.
303   * @param regionName The regionname we are interested in.
304   * @return True if we were able to verify the region located at other side of
305   * the Interface.
306   * @throws IOException
307   */
308  // TODO: We should be able to get the ServerName from the AdminProtocol
309  // rather than have to pass it in.  Its made awkward by the fact that the
310  // HRI is likely a proxy against remote server so the getServerName needs
311  // to be fixed to go to a local method or to a cache before we can do this.
312  private boolean verifyRegionLocation(final ClusterConnection connection,
313      AdminService.BlockingInterface hostingServer, final ServerName address,
314      final byte [] regionName)
315  throws IOException {
316    if (hostingServer == null) {
317      LOG.info("Passed hostingServer is null");
318      return false;
319    }
320    Throwable t;
321    HBaseRpcController controller = connection.getRpcControllerFactory().newController();
322    try {
323      // Try and get regioninfo from the hosting server.
324      return ProtobufUtil.getRegionInfo(controller, hostingServer, regionName) != null;
325    } catch (ConnectException e) {
326      t = e;
327    } catch (RetriesExhaustedException e) {
328      t = e;
329    } catch (RemoteException e) {
330      IOException ioe = e.unwrapRemoteException();
331      t = ioe;
332    } catch (IOException e) {
333      Throwable cause = e.getCause();
334      if (cause != null && cause instanceof EOFException) {
335        t = cause;
336      } else if (cause != null && cause.getMessage() != null
337          && cause.getMessage().contains("Connection reset")) {
338        t = cause;
339      } else {
340        t = e;
341      }
342    }
343    LOG.info("Failed verification of " + Bytes.toStringBinary(regionName) +
344      " at address=" + address + ", exception=" + t.getMessage());
345    return false;
346  }

348  /**
349   * Gets a connection to the server hosting meta, as reported by ZooKeeper,
350   * waiting up to the specified timeout for availability.
351   * <p>WARNING: Does not retry.  Use an {@link org.apache.hadoop.hbase.client.HTable} instead.
352   * @param connection
353   * @param zkw
354   * @param timeout How long to wait on meta location
355   * @param replicaId
356   * @return connection to server hosting meta
357   * @throws InterruptedException
358   * @throws NotAllMetaRegionsOnlineException if timed out waiting
359   * @throws IOException
360   */
361  private AdminService.BlockingInterface getMetaServerConnection(ClusterConnection connection,
362      ZooKeeperWatcher zkw, long timeout, int replicaId)
363  throws InterruptedException, NotAllMetaRegionsOnlineException, IOException {
364    return getCachedConnection(connection, waitMetaRegionLocation(zkw, replicaId, timeout));
365  }
366
367  /**
368   * @param sn ServerName to get a connection against.
369   * @return The AdminProtocol we got when we connected to <code>sn</code>
370   * May have come from cache, may not be good, may have been setup by this
371   * invocation, or may be null.
372   * @throws IOException
373   */
374  private static AdminService.BlockingInterface getCachedConnection(ClusterConnection connection,
375    ServerName sn)
376  throws IOException {
377    if (sn == null) {
378      return null;
379    }
380    AdminService.BlockingInterface service = null;
381    try {
382      service = connection.getAdmin(sn);
383    } catch (RetriesExhaustedException e) {
384      if (e.getCause() != null && e.getCause() instanceof ConnectException) {
385        // Catch this; presume it means the cached connection has gone bad.
386      } else {
387        throw e;
388      }
389    } catch (SocketTimeoutException e) {
390      LOG.debug("Timed out connecting to " + sn);
391    } catch (NoRouteToHostException e) {
392      LOG.debug("Connecting to " + sn, e);
393    } catch (SocketException e) {
394      LOG.debug("Exception connecting to " + sn);
395    } catch (UnknownHostException e) {
396      LOG.debug("Unknown host exception connecting to  " + sn);
397    } catch (FailedServerException e) {
398      if (LOG.isDebugEnabled()) {
399        LOG.debug("Server " + sn + " is in failed server list.");
400      }
401    } catch (IOException ioe) {
402      Throwable cause = ioe.getCause();
403      if (ioe instanceof ConnectException) {
404        // Catch. Connect refused.
405      } else if (cause != null && cause instanceof EOFException) {
406        // Catch. Other end disconnected us.
407      } else if (cause != null && cause.getMessage() != null &&
408        cause.getMessage().toLowerCase(Locale.ROOT).contains("connection reset")) {
409        // Catch. Connection reset.
410      } else {
411        throw ioe;
412      }
413
414    }
415    return service;
416  }

431  /**
432   * Sets the location of <code>hbase:meta</code> in ZooKeeper to the
433   * specified server address.
434   * @param zookeeper
435   * @param serverName
436   * @param replicaId
437   * @param state
438   * @throws KeeperException
439   */
440  public static void setMetaLocation(ZooKeeperWatcher zookeeper,
441      ServerName serverName, int replicaId, RegionState.State state) throws KeeperException {
442    LOG.info("Setting hbase:meta region location in ZooKeeper as " + serverName);
443    // Make the MetaRegionServer pb and then get its bytes and save this as
444    // the znode content.
445    MetaRegionServer pbrsr = MetaRegionServer.newBuilder()
446      .setServer(ProtobufUtil.toServerName(serverName))
447      .setRpcVersion(HConstants.RPC_CURRENT_VERSION)
448      .setState(state.convert()).build();
449    byte[] data = ProtobufUtil.prependPBMagic(pbrsr.toByteArray());
450    try {
451      ZKUtil.setData(zookeeper, zookeeper.znodePaths.getZNodeForReplica(replicaId), data);
452    } catch(KeeperException.NoNodeException nne) {
453      if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
454        LOG.debug("META region location doesn't exist, create it");
455      } else {
456        LOG.debug("META region location doesn't exist for replicaId " + replicaId +
457            ", create it");
458      }
459      ZKUtil.createAndWatch(zookeeper, zookeeper.znodePaths.getZNodeForReplica(replicaId), data);
460    }
461  }

470  /**
471   * Load the meta region state from the meta server ZNode.
472   * @param zkw
473   * @param replicaId
474   * @return regionstate
475   * @throws KeeperException
476   */
477  public static RegionState getMetaRegionState(ZooKeeperWatcher zkw, int replicaId)
478      throws KeeperException {
479    RegionState.State state = RegionState.State.OPEN;
480    ServerName serverName = null;
481    try {
482      byte[] data = ZKUtil.getData(zkw, zkw.znodePaths.getZNodeForReplica(replicaId));
483      if (data != null && data.length > 0 && ProtobufUtil.isPBMagicPrefix(data)) {
484        try {
485          int prefixLen = ProtobufUtil.lengthOfPBMagic();
486          ZooKeeperProtos.MetaRegionServer rl =
487            ZooKeeperProtos.MetaRegionServer.PARSER.parseFrom
488              (data, prefixLen, data.length - prefixLen);
489          if (rl.hasState()) {
490            state = RegionState.State.convert(rl.getState());
491          }
492          HBaseProtos.ServerName sn = rl.getServer();
493          serverName = ServerName.valueOf(
494            sn.getHostName(), sn.getPort(), sn.getStartCode());
495        } catch (InvalidProtocolBufferException e) {
496          throw new DeserializationException("Unable to parse meta region location");
497        }
498      } else {
499        // old style of meta region location?
500        serverName = ProtobufUtil.parseServerNameFrom(data);
501      }
502    } catch (DeserializationException e) {
503      throw ZKUtil.convert(e);
504    } catch (InterruptedException e) {
505      Thread.currentThread().interrupt();
506    }
507    if (serverName == null) {
508      state = RegionState.State.OFFLINE;
509    }
510    return new RegionState(
511        RegionReplicaUtil.getRegionInfoForReplica(HRegionInfo.FIRST_META_REGIONINFO, replicaId),
512      state, serverName);
513  }

539  /**
540   * Wait until the primary meta region is available. Get the secondary
541   * locations as well but don't block for those.
542   * @param zkw
543   * @param timeout
544   * @param conf
545   * @return ServerName or null if we timed out.
546   * @throws InterruptedException
547   */
548  public List<ServerName> blockUntilAvailable(final ZooKeeperWatcher zkw,
549      final long timeout, Configuration conf)
550          throws InterruptedException {
551    int numReplicasConfigured = 1;
552
553    List<ServerName> servers = new ArrayList<ServerName>();
554    // Make the blocking call first so that we do the wait to know
555    // the znodes are all in place or timeout.
556    ServerName server = blockUntilAvailable(zkw, timeout);
557    if (server == null) return null;
558    servers.add(server);
559
560    try {
561      List<String> metaReplicaNodes = zkw.getMetaReplicaNodes();
562      numReplicasConfigured = metaReplicaNodes.size();
563    } catch (KeeperException e) {
564      LOG.warn("Got ZK exception " + e);
565    }
566    for (int replicaId = 1; replicaId < numReplicasConfigured; replicaId++) {
567      // return all replica locations for the meta
568      servers.add(getMetaRegionLocation(zkw, replicaId));
569    }
570    return servers;
571  }

586  /**
587   * Wait until the meta region is available and is not in transition.
588   * @param zkw
589   * @param replicaId
590   * @param timeout
591   * @return ServerName or null if we timed out.
592   * @throws InterruptedException
593   */
594  public ServerName blockUntilAvailable(final ZooKeeperWatcher zkw, int replicaId,
595      final long timeout)
596  throws InterruptedException {
597    if (timeout < 0) throw new IllegalArgumentException();
598    if (zkw == null) throw new IllegalArgumentException();
599    long startTime = System.currentTimeMillis();
600    ServerName sn = null;
601    while (true) {
602      sn = getMetaRegionLocation(zkw, replicaId);
603      if (sn != null || (System.currentTimeMillis() - startTime)
604          > timeout - HConstants.SOCKET_RETRY_WAIT_MS) {
605        break;
606      }
607      Thread.sleep(HConstants.SOCKET_RETRY_WAIT_MS);
608    }
609    return sn;
610  }

615   */

618      LOG.debug("Stopping MetaTableLocator");

620    }
