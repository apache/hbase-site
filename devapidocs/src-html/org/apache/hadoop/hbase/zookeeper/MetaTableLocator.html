036/**
037 * Utility class to perform operation (get/wait for/verify/set/delete) on znode in ZooKeeper which
038 * keeps hbase:meta region server location.
039 * <p/>
040 * Stateless class with a bunch of static methods. Doesn't manage resources passed in (e.g.
041 * Connection, ZKWatcher etc).
042 * <p/>
043 * Meta region location is set by <code>RegionServerServices</code>. This class doesn't use ZK
044 * watchers, rather accesses ZK directly.
045 * <p/>
046 * TODO: rewrite using RPC calls to master to find out about hbase:meta.
047 */
048@InterfaceAudience.Private
049public final class MetaTableLocator {
050  private static final Logger LOG = LoggerFactory.getLogger(MetaTableLocator.class);
051
052  private MetaTableLocator() {
053  }
054
055  /**
056   * Gets the meta region location, if available.  Does not block.
057   * @param zkw zookeeper connection to use
058   * @return server name or null if we failed to get the data.
059   */
060  @RestrictedApi(explanation = "Should only be called in tests or ZKUtil", link = "",
061    allowedOnPath = ".*/src/test/.*|.*/ZKUtil\\.java")
062  public static ServerName getMetaRegionLocation(final ZKWatcher zkw) {
063    try {
064      RegionState state = getMetaRegionState(zkw);
065      return state.isOpened() ? state.getServerName() : null;
066    } catch (KeeperException ke) {
067      return null;
068    }
069  }
070
071  /**
072   * Gets the meta region location, if available.  Does not block.
073   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and operation
074   * @param replicaId the ID of the replica
075   * @return server name
076   */
077  @RestrictedApi(explanation = "Should only be called in self or ZKUtil", link = "",
078    allowedOnPath = ".*(MetaTableLocator|ZKUtil)\\.java")
079  public static ServerName getMetaRegionLocation(final ZKWatcher zkw, int replicaId) {
080    try {
081      RegionState state = getMetaRegionState(zkw, replicaId);
082      return state.isOpened() ? state.getServerName() : null;
083    } catch (KeeperException ke) {
084      return null;
085    }
086  }

088  /**
089   * Gets the meta region location, if available, and waits for up to the specified timeout if not
090   * immediately available. Given the zookeeper notification could be delayed, we will try to get
091   * the latest data.
092   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and operation
093   * @param timeout maximum time to wait, in millis
094   * @return server name for server hosting meta region formatted as per {@link ServerName}, or null
095   *         if none available
096   * @throws InterruptedException if interrupted while waiting
097   * @throws NotAllMetaRegionsOnlineException if a meta or root region is not online
098   */
099  @RestrictedApi(explanation = "Should only be called in tests", link = "",
100    allowedOnPath = ".*/src/test/.*")
101  public static ServerName waitMetaRegionLocation(ZKWatcher zkw, long timeout)
102      throws InterruptedException, NotAllMetaRegionsOnlineException {
103    return waitMetaRegionLocation(zkw, RegionInfo.DEFAULT_REPLICA_ID, timeout);
104  }
105
106  /**
107   * Gets the meta region location, if available, and waits for up to the specified timeout if not
108   * immediately available. Given the zookeeper notification could be delayed, we will try to get
109   * the latest data.
110   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and operation
111   * @param replicaId the ID of the replica
112   * @param timeout maximum time to wait, in millis
113   * @return server name for server hosting meta region formatted as per {@link ServerName}, or null
114   *         if none available
115   * @throws InterruptedException if waiting for the socket operation fails
116   * @throws NotAllMetaRegionsOnlineException if a meta or root region is not online
117   */
118  private static ServerName waitMetaRegionLocation(ZKWatcher zkw, int replicaId, long timeout)
119      throws InterruptedException, NotAllMetaRegionsOnlineException {
120    try {
121      if (ZKUtil.checkExists(zkw, zkw.getZNodePaths().baseZNode) == -1) {
122        String errorMsg = "Check the value configured in 'zookeeper.znode.parent'. " +
123          "There could be a mismatch with the one configured in the master.";
124        LOG.error(errorMsg);
125        throw new IllegalArgumentException(errorMsg);
126      }
127    } catch (KeeperException e) {
128      throw new IllegalStateException("KeeperException while trying to check baseZNode:", e);
129    }
130    ServerName sn = blockUntilAvailable(zkw, replicaId, timeout);
131
132    if (sn == null) {
133      throw new NotAllMetaRegionsOnlineException("Timed out; " + timeout + "ms");
134    }
135
136    return sn;
137  }

139  /**
140   * Sets the location of <code>hbase:meta</code> in ZooKeeper to the
141   * specified server address.
142   * @param zookeeper zookeeper reference
143   * @param serverName The server hosting <code>hbase:meta</code>
144   * @param state The region transition state
145   * @throws KeeperException unexpected zookeeper exception
146   */
147  @RestrictedApi(explanation = "Should only be called in tests", link = "",
148    allowedOnPath = ".*/src/test/.*")
149  public static void setMetaLocation(ZKWatcher zookeeper,
150      ServerName serverName, RegionState.State state) throws KeeperException {
151    setMetaLocation(zookeeper, serverName, RegionInfo.DEFAULT_REPLICA_ID, state);
152  }
153
154  /**
155   * Sets the location of <code>hbase:meta</code> in ZooKeeper to the specified server address.
156   * @param zookeeper reference to the {@link ZKWatcher} which also contains configuration and
157   *                  operation
158   * @param serverName the name of the server
159   * @param replicaId the ID of the replica
160   * @param state the state of the region
161   * @throws KeeperException if a ZooKeeper operation fails
162   */
163  public static void setMetaLocation(ZKWatcher zookeeper, ServerName serverName, int replicaId,
164      RegionState.State state) throws KeeperException {
165    if (serverName == null) {
166      LOG.warn("Tried to set null ServerName in hbase:meta; skipping -- ServerName required");
167      return;
168    }
169    LOG.info("Setting hbase:meta replicaId={} location in ZooKeeper as {}, state={}", replicaId,
170      serverName, state);
171    // Make the MetaRegionServer pb and then get its bytes and save this as
172    // the znode content.
173    MetaRegionServer pbrsr = MetaRegionServer.newBuilder()
174      .setServer(ProtobufUtil.toServerName(serverName))
175      .setRpcVersion(HConstants.RPC_CURRENT_VERSION)
176      .setState(state.convert()).build();
177    byte[] data = ProtobufUtil.prependPBMagic(pbrsr.toByteArray());
178    try {
179      ZKUtil.setData(zookeeper,
180          zookeeper.getZNodePaths().getZNodeForReplica(replicaId), data);
181    } catch(KeeperException.NoNodeException nne) {
182      if (replicaId == RegionInfo.DEFAULT_REPLICA_ID) {
183        LOG.debug("hbase:meta region location doesn't exist, create it");
184      } else {
185        LOG.debug("hbase:meta region location doesn't exist for replicaId=" + replicaId +
186            ", create it");
187      }
188      ZKUtil.createAndWatch(zookeeper, zookeeper.getZNodePaths().getZNodeForReplica(replicaId),
189              data);
190    }
191  }

193  /**
194   * Load the meta region state from the meta server ZNode.
195   */
196  @RestrictedApi(explanation = "Should only be called in self or tests", link = "",
197    allowedOnPath = ".*/src/test/.*|.*/MetaTableLocator\\.java")
198  public static RegionState getMetaRegionState(ZKWatcher zkw) throws KeeperException {
199    return getMetaRegionState(zkw, RegionInfo.DEFAULT_REPLICA_ID);
200  }
201
202  /**
203   * Load the meta region state from the meta region server ZNode.
204   *
205   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and operation
206   * @param replicaId the ID of the replica
207   * @return regionstate
208   * @throws KeeperException if a ZooKeeper operation fails
209   */
210  public static RegionState getMetaRegionState(ZKWatcher zkw, int replicaId)
211      throws KeeperException {
212    RegionState regionState = null;
213    try {
214      byte[] data = ZKUtil.getData(zkw, zkw.getZNodePaths().getZNodeForReplica(replicaId));
215      regionState = ProtobufUtil.parseMetaRegionStateFrom(data, replicaId);
216    } catch (DeserializationException e) {
217      throw ZKUtil.convert(e);
218    } catch (InterruptedException e) {
219      Thread.currentThread().interrupt();
220    }
221    return regionState;
222  }

224  /**
225   * Deletes the location of <code>hbase:meta</code> in ZooKeeper.
226   * @param zookeeper zookeeper reference
227   * @throws KeeperException unexpected zookeeper exception
228   */
229  @RestrictedApi(explanation = "Should only be called in tests", link = "",
230    allowedOnPath = ".*/src/test/.*")
231  public static void deleteMetaLocation(ZKWatcher zookeeper)
232    throws KeeperException {
233    deleteMetaLocation(zookeeper, RegionInfo.DEFAULT_REPLICA_ID);
234  }
235
236  public static void deleteMetaLocation(ZKWatcher zookeeper, int replicaId)
237    throws KeeperException {
238    if (replicaId == RegionInfo.DEFAULT_REPLICA_ID) {
239      LOG.info("Deleting hbase:meta region location in ZooKeeper");
240    } else {
241      LOG.info("Deleting hbase:meta for {} region location in ZooKeeper", replicaId);
242    }
243    try {
244      // Just delete the node.  Don't need any watches.
245      ZKUtil.deleteNode(zookeeper, zookeeper.getZNodePaths().getZNodeForReplica(replicaId));
246    } catch(KeeperException.NoNodeException nne) {
247      // Has already been deleted
248    }
249  }
250
251  /**
252   * Wait until the meta region is available and is not in transition.
253   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and constants
254   * @param replicaId the ID of the replica
255   * @param timeout maximum time to wait in millis
256   * @return ServerName or null if we timed out.
257   * @throws InterruptedException if waiting for the socket operation fails
258   */
259  private static ServerName blockUntilAvailable(final ZKWatcher zkw, int replicaId,
260    final long timeout) throws InterruptedException {
261    if (timeout < 0) {
262      throw new IllegalArgumentException();
263    }
264
265    if (zkw == null) {
266      throw new IllegalArgumentException();
267    }
268
269    long startTime = EnvironmentEdgeManager.currentTime();
270    ServerName sn = null;
271    while (true) {
272      sn = getMetaRegionLocation(zkw, replicaId);
273      if (sn != null || (EnvironmentEdgeManager.currentTime() - startTime) > timeout -
274        HConstants.SOCKET_RETRY_WAIT_MS) {
275        break;
276      }
277      Thread.sleep(HConstants.SOCKET_RETRY_WAIT_MS);
278    }
279    return sn;
280  }
