051/**
052 * HBase's version of ZooKeeper's QuorumPeer. When HBase is set to manage
053 * ZooKeeper, this class is used to start up QuorumPeer instances. By doing
054 * things in here rather than directly calling to ZooKeeper, we have more
055 * control over the process. This class uses {@link ZKConfig} to get settings
056 * from the hbase-site.xml file.
057 */
058@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
059@InterfaceStability.Evolving
060public final class HQuorumPeer {
061  private static final Logger LOG = LoggerFactory.getLogger(HQuorumPeer.class);
062
063  private HQuorumPeer() {
064  }
065
066  /**
067   * Parse ZooKeeper configuration from HBase XML config and run a QuorumPeer.
068   * @param args String[] of command line arguments. Not used.
069   */
070  public static void main(String[] args) {
071    Configuration conf = HBaseConfiguration.create();
072    try {
073      Properties zkProperties = ZKConfig.makeZKProps(conf);
074      writeMyID(zkProperties);
075      QuorumPeerConfig zkConfig = new QuorumPeerConfig();
076      zkConfig.parseProperties(zkProperties);
077
078      // login the zookeeper server principal (if using security)
079      ZKUtil.loginServer(conf, HConstants.ZK_SERVER_KEYTAB_FILE,
080        HConstants.ZK_SERVER_KERBEROS_PRINCIPAL,
081        zkConfig.getClientPortAddress().getHostName());
082
083      runZKServer(zkConfig);
084    } catch (Exception e) {
085      LOG.error("Failed to start ZKServer", e);
086      System.exit(-1);
087    }
088  }
089
090  private static void runZKServer(QuorumPeerConfig zkConfig)
091          throws IOException, AdminServer.AdminServerException {
092
093    /**
094     *  Start and schedule the purge task
095     *  autopurge.purgeInterval is 0 by default,so in fact the DatadirCleanupManager task will not
096     *  be started to clean the logs by default. Config is recommended only for standalone server.
097     */
098
099    DatadirCleanupManager purgeMgr=new DatadirCleanupManager(
100      zkConfig.getDataDir(),
101      zkConfig.getDataLogDir(),
102      zkConfig.getSnapRetainCount(),
103      zkConfig.getPurgeInterval());
104    purgeMgr.start();
105
106    if (zkConfig.isDistributed()) {
107      QuorumPeerMain qp = new QuorumPeerMain();
108      qp.runFromConfig(zkConfig);
109    } else {
110      ZooKeeperServerMain zk = new ZooKeeperServerMain();
111      ServerConfig serverConfig = new ServerConfig();
112      serverConfig.readFrom(zkConfig);
113      zk.runFromConfig(serverConfig);
114    }
115  }
116
117  private static boolean addressIsLocalHost(String address) {
118    return address.equals("localhost") || address.equals("127.0.0.1");
119  }
120
121  static void writeMyID(Properties properties) throws IOException {
122    long myId = -1;
123
124    Configuration conf = HBaseConfiguration.create();
125    String myAddress = Strings.domainNamePointerToHostName(DNS.getDefaultHost(
126        conf.get("hbase.zookeeper.dns.interface","default"),
127        conf.get("hbase.zookeeper.dns.nameserver","default")));
128
129    List<String> ips = new ArrayList<>();
130
131    // Add what could be the best (configured) match
132    ips.add(myAddress.contains(".") ?
133        myAddress :
134        StringUtils.simpleHostname(myAddress));
135
136    // For all nics get all hostnames and IPs
137    Enumeration<?> nics = NetworkInterface.getNetworkInterfaces();
138    while(nics.hasMoreElements()) {
139      Enumeration<?> rawAdrs =
140          ((NetworkInterface)nics.nextElement()).getInetAddresses();
141      while(rawAdrs.hasMoreElements()) {
142        InetAddress inet = (InetAddress) rawAdrs.nextElement();
143        ips.add(StringUtils.simpleHostname(inet.getHostName()));
144        ips.add(inet.getHostAddress());
145      }
146    }
147
148    for (Entry<Object, Object> entry : properties.entrySet()) {
149      String key = entry.getKey().toString().trim();
150      String value = entry.getValue().toString().trim();
151      if (key.startsWith("server.")) {
152        int dot = key.indexOf('.');
153        long id = Long.parseLong(key.substring(dot + 1));
154        String[] parts = value.split(":");
155        String address = parts[0];
156        if (addressIsLocalHost(address) || ips.contains(address)) {
157          myId = id;
158          break;
159        }
160      }
161    }
162
163    // Set the max session timeout from the provided client-side timeout
164    properties.setProperty("maxSessionTimeout", conf.get(HConstants.ZK_SESSION_TIMEOUT,
165            Integer.toString(HConstants.DEFAULT_ZK_SESSION_TIMEOUT)));
166
167    if (myId == -1) {
168      throw new IOException("Could not find my address: " + myAddress +
169                            " in list of ZooKeeper quorum servers");
170    }
171
172    String dataDirStr = properties.get("dataDir").toString().trim();
173    File dataDir = new File(dataDirStr);
174    if (!dataDir.isDirectory()) {
175      if (!dataDir.mkdirs()) {
176        throw new IOException("Unable to create data dir " + dataDir);
177      }
178    }
179
180    File myIdFile = new File(dataDir, "myid");
181    PrintWriter w = new PrintWriter(myIdFile, StandardCharsets.UTF_8.name());
182    w.println(myId);
183    w.close();
184  }
185}
