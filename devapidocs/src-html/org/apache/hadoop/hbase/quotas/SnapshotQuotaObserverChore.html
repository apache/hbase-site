047/**
048 * A Master-invoked {@code Chore} that computes the size of each snapshot which was created from
049 * a table which has a space quota.
050 */
051@InterfaceAudience.Private
052public class SnapshotQuotaObserverChore extends ScheduledChore {
053  private static final Logger LOG = LoggerFactory.getLogger(SnapshotQuotaObserverChore.class);
054  static final String SNAPSHOT_QUOTA_CHORE_PERIOD_KEY =
055      "hbase.master.quotas.snapshot.chore.period";
056  static final int SNAPSHOT_QUOTA_CHORE_PERIOD_DEFAULT = 1000 * 60 * 5; // 5 minutes in millis
057
058  static final String SNAPSHOT_QUOTA_CHORE_DELAY_KEY =
059      "hbase.master.quotas.snapshot.chore.delay";
060  static final long SNAPSHOT_QUOTA_CHORE_DELAY_DEFAULT = 1000L * 60L; // 1 minute in millis
061
062  static final String SNAPSHOT_QUOTA_CHORE_TIMEUNIT_KEY =
063      "hbase.master.quotas.snapshot.chore.timeunit";
064  static final String SNAPSHOT_QUOTA_CHORE_TIMEUNIT_DEFAULT = TimeUnit.MILLISECONDS.name();
065
066  private final Connection conn;
067  private final Configuration conf;
068  private final MetricsMaster metrics;
069  private final FileSystem fs;
070
071  public SnapshotQuotaObserverChore(HMaster master, MetricsMaster metrics) {
072    this(
073        master.getConnection(), master.getConfiguration(), master.getFileSystem(), master, metrics);
074  }
075
076  SnapshotQuotaObserverChore(
077      Connection conn, Configuration conf, FileSystem fs, Stoppable stopper,
078      MetricsMaster metrics) {
079    super(
080        QuotaObserverChore.class.getSimpleName(), stopper, getPeriod(conf),
081        getInitialDelay(conf), getTimeUnit(conf));
082    this.conn = conn;
083    this.conf = conf;
084    this.metrics = metrics;
085    this.fs = fs;
086  }
087
088  @Override
089  protected void chore() {
090    try {
091      if (LOG.isTraceEnabled()) {
092        LOG.trace("Computing sizes of snapshots for quota management.");
093      }
094      long start = System.nanoTime();
095      _chore();
096      if (null != metrics) {
097        metrics.incrementSnapshotObserverTime((System.nanoTime() - start) / 1_000_000);
098      }
099    } catch (IOException e) {
100      LOG.warn("Failed to compute the size of snapshots, will retry", e);
101    }
102  }
103
104  void _chore() throws IOException {
105    // Gets all tables with quotas that also have snapshots.
106    // This values are all of the snapshots that we need to compute the size of.
107    long start = System.nanoTime();
108    Multimap<TableName,String> snapshotsToComputeSize = getSnapshotsToComputeSize();
109    if (null != metrics) {
110      metrics.incrementSnapshotFetchTime((System.nanoTime() - start) / 1_000_000);
111    }
112
113    // For each table, compute the size of each snapshot
114    Map<String,Long> namespaceSnapshotSizes = computeSnapshotSizes(snapshotsToComputeSize);
115
116    // Write the size data by namespaces to the quota table.
117    // We need to do this "globally" since each FileArchiverNotifier is limited to its own Table.
118    persistSnapshotSizesForNamespaces(namespaceSnapshotSizes);
119  }
120
121  /**
122   * Fetches each table with a quota (table or namespace quota), and then fetch the name of each
123   * snapshot which was created from that table.
124   *
125   * @return A mapping of table to snapshots created from that table
126   */
127  Multimap<TableName,String> getSnapshotsToComputeSize() throws IOException {
128    Set<TableName> tablesToFetchSnapshotsFrom = new HashSet<>();
129    QuotaFilter filter = new QuotaFilter();
130    filter.addTypeFilter(QuotaType.SPACE);
131    try (Admin admin = conn.getAdmin()) {
132      // Pull all of the tables that have quotas (direct, or from namespace)
133      for (QuotaSettings qs : QuotaRetriever.open(conf, filter)) {
134        String ns = qs.getNamespace();
135        TableName tn = qs.getTableName();
136        if ((null == ns && null == tn) || (null != ns && null != tn)) {
137          throw new IllegalStateException(
138              "Expected only one of namespace and tablename to be null");
139        }
140        // Collect either the table name itself, or all of the tables in the namespace
141        if (null != ns) {
142          tablesToFetchSnapshotsFrom.addAll(Arrays.asList(admin.listTableNamesByNamespace(ns)));
143        } else {
144          tablesToFetchSnapshotsFrom.add(tn);
145        }
146      }
147      // Fetch all snapshots that were created from these tables
148      return getSnapshotsFromTables(admin, tablesToFetchSnapshotsFrom);
149    }
150  }
151
152  /**
153   * Computes a mapping of originating {@code TableName} to snapshots, when the {@code TableName}
154   * exists in the provided {@code Set}.
155   */
156  Multimap<TableName,String> getSnapshotsFromTables(
157      Admin admin, Set<TableName> tablesToFetchSnapshotsFrom) throws IOException {
158    Multimap<TableName,String> snapshotsToCompute = HashMultimap.create();
159    for (org.apache.hadoop.hbase.client.SnapshotDescription sd : admin.listSnapshots()) {
160      TableName tn = sd.getTableName();
161      if (tablesToFetchSnapshotsFrom.contains(tn)) {
162        snapshotsToCompute.put(tn, sd.getName());
163      }
164    }
165    return snapshotsToCompute;
166  }
167
168  /**
169   * Computes the size of each snapshot provided given the current files referenced by the table.
170   *
171   * @param snapshotsToComputeSize The snapshots to compute the size of
172   * @return A mapping of table to snapshot created from that table and the snapshot's size.
173   */
174  Map<String,Long> computeSnapshotSizes(
175      Multimap<TableName,String> snapshotsToComputeSize) throws IOException {
176    final Map<String,Long> snapshotSizesByNamespace = new HashMap<>();
177    final long start = System.nanoTime();
178    for (Entry<TableName,Collection<String>> entry : snapshotsToComputeSize.asMap().entrySet()) {
179      final TableName tn = entry.getKey();
180      final Collection<String> snapshotNames = entry.getValue();
181
182      // Get our notifier instance, this is tracking archivals that happen out-of-band of this chore
183      FileArchiverNotifier notifier = getNotifierForTable(tn);
184
185      // The total size consumed by all snapshots against this table
186      long totalSnapshotSize = notifier.computeAndStoreSnapshotSizes(snapshotNames);
187      // Bucket that size into the appropriate namespace
188      snapshotSizesByNamespace.merge(tn.getNamespaceAsString(), totalSnapshotSize, Long::sum);
189    }
190
191    // Update the amount of time it took to compute the size of the snapshots for a table
192    if (metrics != null) {
193      metrics.incrementSnapshotSizeComputationTime((System.nanoTime() - start) / 1_000_000);
194    }
195
196    return snapshotSizesByNamespace;
197  }
198
199  /**
200   * Returns the correct instance of {@link FileArchiverNotifier} for the given table name.
201   *
202   * @param tn The table name
203   * @return A {@link FileArchiverNotifier} instance
204   */
205  FileArchiverNotifier getNotifierForTable(TableName tn) {
206    return FileArchiverNotifierFactoryImpl.getInstance().get(conn, conf, fs, tn);
207  }
208
209  /**
210   * Writes the size used by snapshots for each namespace to the quota table.
211   */
212  void persistSnapshotSizesForNamespaces(
213      Map<String,Long> snapshotSizesByNamespace) throws IOException {
214    try (Table quotaTable = conn.getTable(QuotaUtil.QUOTA_TABLE_NAME)) {
215      quotaTable.put(snapshotSizesByNamespace.entrySet().stream()
216          .map(e -> QuotaTableUtil.createPutForNamespaceSnapshotSize(e.getKey(), e.getValue()))
217          .collect(Collectors.toList()));
218    }
219  }

221  /**
222   * Extracts the period for the chore from the configuration.
223   *
224   * @param conf The configuration object.
225   * @return The configured chore period or the default value.
226   */
227  static int getPeriod(Configuration conf) {
228    return conf.getInt(SNAPSHOT_QUOTA_CHORE_PERIOD_KEY,
229        SNAPSHOT_QUOTA_CHORE_PERIOD_DEFAULT);
230  }
231
232  /**
233   * Extracts the initial delay for the chore from the configuration.
234   *
235   * @param conf The configuration object.
236   * @return The configured chore initial delay or the default value.
237   */
238  static long getInitialDelay(Configuration conf) {
239    return conf.getLong(SNAPSHOT_QUOTA_CHORE_DELAY_KEY,
240        SNAPSHOT_QUOTA_CHORE_DELAY_DEFAULT);
241  }
242
243  /**
244   * Extracts the time unit for the chore period and initial delay from the configuration. The
245   * configuration value for {@link #SNAPSHOT_QUOTA_CHORE_TIMEUNIT_KEY} must correspond to
246   * a {@link TimeUnit} value.
247   *
248   * @param conf The configuration object.
249   * @return The configured time unit for the chore period and initial delay or the default value.
250   */
251  static TimeUnit getTimeUnit(Configuration conf) {
252    return TimeUnit.valueOf(conf.get(SNAPSHOT_QUOTA_CHORE_TIMEUNIT_KEY,
253        SNAPSHOT_QUOTA_CHORE_TIMEUNIT_DEFAULT));
254  }
