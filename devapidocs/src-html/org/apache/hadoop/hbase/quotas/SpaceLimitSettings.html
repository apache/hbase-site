017package org.apache.hadoop.hbase.quotas;
018
019import java.util.Objects;
020
021import org.apache.hadoop.hbase.TableName;
022import org.apache.yetus.audience.InterfaceAudience;
023import org.apache.yetus.audience.InterfaceStability;
024import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
025import org.apache.hadoop.hbase.shaded.protobuf.generated.QuotaProtos;
026import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProtos.SetQuotaRequest.Builder;
027import org.apache.hadoop.hbase.shaded.protobuf.generated.QuotaProtos.SpaceLimitRequest;
028import org.apache.hadoop.hbase.shaded.protobuf.generated.QuotaProtos.SpaceQuota;
029
030/**
031 * A {@link QuotaSettings} implementation for configuring filesystem-use quotas.
032 */
033@InterfaceAudience.Private
034@InterfaceStability.Evolving
035class SpaceLimitSettings extends QuotaSettings {
036
037  private final SpaceLimitRequest proto;
038
039  SpaceLimitSettings(TableName tableName, long sizeLimit, SpaceViolationPolicy violationPolicy) {
040    super(null, Objects.requireNonNull(tableName), null);
041    if (sizeLimit < 0L) {
042      throw new IllegalArgumentException("Size limit must be a non-negative value.");
043    }
044    proto = buildProtoAddQuota(sizeLimit, Objects.requireNonNull(violationPolicy));
045  }
046
047  /**
048   * Constructs a {@code SpaceLimitSettings} to remove a space quota on the given {@code tableName}.
049   */
050  SpaceLimitSettings(TableName tableName) {
051    super(null, Objects.requireNonNull(tableName), null);
052    proto = buildProtoRemoveQuota();
053  }
054
055  SpaceLimitSettings(String namespace, long sizeLimit, SpaceViolationPolicy violationPolicy) {
056    super(null, null, Objects.requireNonNull(namespace));
057    if (sizeLimit < 0L) {
058      throw new IllegalArgumentException("Size limit must be a non-negative value.");
059    }
060    proto = buildProtoAddQuota(sizeLimit, Objects.requireNonNull(violationPolicy));
061  }
062
063  /**
064   * Constructs a {@code SpaceLimitSettings} to remove a space quota on the given {@code namespace}.
065   */
066  SpaceLimitSettings(String namespace) {
067    super(null, null, Objects.requireNonNull(namespace));
068    proto = buildProtoRemoveQuota();
069  }
070
071  SpaceLimitSettings(TableName tableName, String namespace, SpaceLimitRequest req) {
072    super(null, tableName, namespace);
073    proto = req;
074  }
075
076  /**
077   * Build a {@link SpaceLimitRequest} protobuf object from the given {@link SpaceQuota}.
078   *
079   * @param protoQuota The preconstructed SpaceQuota protobuf
080   * @return A protobuf request to change a space limit quota
081   */
082  private SpaceLimitRequest buildProtoFromQuota(SpaceQuota protoQuota) {
083    return SpaceLimitRequest.newBuilder().setQuota(protoQuota).build();
084  }
085
086  /**
087   * Builds a {@link SpaceQuota} protobuf object given the arguments.
088   *
089   * @param sizeLimit The size limit of the quota.
090   * @param violationPolicy The action to take when the quota is exceeded.
091   * @return The protobuf SpaceQuota representation.
092   */
093  private SpaceLimitRequest buildProtoAddQuota(
094      long sizeLimit, SpaceViolationPolicy violationPolicy) {
095    return buildProtoFromQuota(SpaceQuota.newBuilder()
096            .setSoftLimit(sizeLimit)
097            .setViolationPolicy(ProtobufUtil.toProtoViolationPolicy(violationPolicy))
098            .build());
099  }
100
101  /**
102   * Builds a {@link SpaceQuota} protobuf object to remove a quota.
103   *
104   * @return The protobuf SpaceQuota representation.
105   */
106  private SpaceLimitRequest buildProtoRemoveQuota() {
107    return SpaceLimitRequest.newBuilder().setQuota(
108        SpaceQuota.newBuilder()
109            .setRemove(true)
110            .build())
111        .build();
112  }
113
114  /**
115   * Returns a copy of the internal state of <code>this</code>
116   */
117  SpaceLimitRequest getProto() {
118    return proto.toBuilder().build();
119  }
120
121  @Override
122  public QuotaType getQuotaType() {
123    return QuotaType.SPACE;
124  }
125
126  @Override
127  protected void setupSetQuotaRequest(Builder builder) {
128    // TableName/Namespace are serialized in QuotaSettings
129    builder.setSpaceLimit(proto);
130  }
131
132  /**
133   * Constructs a {@link SpaceLimitSettings} from the provided protobuf message and tablename.
134   *
135   * @param tableName The target tablename for the limit.
136   * @param proto The protobuf representation.
137   * @return A QuotaSettings.
138   */
139  static SpaceLimitSettings fromSpaceQuota(
140      final TableName tableName, final QuotaProtos.SpaceQuota proto) {
141    validateProtoArguments(proto);
142    return new SpaceLimitSettings(tableName, proto.getSoftLimit(),
143        ProtobufUtil.toViolationPolicy(proto.getViolationPolicy()));
144  }
145
146  /**
147   * Constructs a {@link SpaceLimitSettings} from the provided protobuf message and namespace.
148   *
149   * @param namespace The target namespace for the limit.
150   * @param proto The protobuf representation.
151   * @return A QuotaSettings.
152   */
153  static SpaceLimitSettings fromSpaceQuota(
154      final String namespace, final QuotaProtos.SpaceQuota proto) {
155    validateProtoArguments(proto);
156    return new SpaceLimitSettings(namespace, proto.getSoftLimit(),
157        ProtobufUtil.toViolationPolicy(proto.getViolationPolicy()));
158  }
159
160  /**
161   * Validates that the provided protobuf SpaceQuota has the necessary information to construct
162   * a {@link SpaceLimitSettings}.
163   *
164   * @param proto The protobuf message to validate.
165   */
166  static void validateProtoArguments(final QuotaProtos.SpaceQuota proto) {
167    if (!Objects.requireNonNull(proto).hasSoftLimit()) {
168      throw new IllegalArgumentException("Cannot handle SpaceQuota without a soft limit");
169    }
170    if (!proto.hasViolationPolicy()) {
171      throw new IllegalArgumentException("Cannot handle SpaceQuota without a violation policy");
172    }
173  }
174
175  @Override
176  public int hashCode() {
177    return Objects.hash(getTableName(), getNamespace(), proto);
178  }
179
180  @Override
181  public boolean equals(Object o) {
182    if (o == this) {
183      return true;
184    }
185    if (!(o instanceof SpaceLimitSettings)) {
186      return false;
187    }
188    // o is non-null and an instance of SpaceLimitSettings
189    SpaceLimitSettings other = (SpaceLimitSettings) o;
190    return Objects.equals(getTableName(), other.getTableName()) &&
191        Objects.equals(getNamespace(), other.getNamespace()) &&
192        Objects.equals(proto, other.proto);
193  }
194
195  @Override
196  public String toString() {
197    StringBuilder sb = new StringBuilder();
198    sb.append("TYPE => SPACE");
199    if (getTableName() != null) {
200      sb.append(", TABLE => ").append(getTableName());
201    }
202    if (getNamespace() != null) {
203      sb.append(", NAMESPACE => ").append(getNamespace());
204    }
205    if (proto.getQuota().getRemove()) {
206      sb.append(", REMOVE => ").append(proto.getQuota().getRemove());
207    } else {
208      sb.append(", LIMIT => ").append(sizeToString(proto.getQuota().getSoftLimit()));
209      sb.append(", VIOLATION_POLICY => ").append(proto.getQuota().getViolationPolicy());
210    }
211    return sb.toString();
212  }
213
214  @Override
215  protected QuotaSettings merge(QuotaSettings newSettings) {
216    if (newSettings instanceof SpaceLimitSettings) {
217      SpaceLimitSettings settingsToMerge = (SpaceLimitSettings) newSettings;
218
219      // The message contained the expect SpaceQuota object
220      if (settingsToMerge.proto.hasQuota()) {
221        SpaceQuota quotaToMerge = settingsToMerge.proto.getQuota();
222        if (quotaToMerge.getRemove()) {
223          return settingsToMerge;
224        } else {
225          // Validate that the two settings are for the same target.
226          // SpaceQuotas either apply to a table or a namespace (no user spacequota).
227          if (!Objects.equals(getTableName(), settingsToMerge.getTableName())
228              && !Objects.equals(getNamespace(), settingsToMerge.getNamespace())) {
229            throw new IllegalArgumentException("Cannot merge " + newSettings + " into " + this);
230          }
231          // Create a builder from the old settings
232          SpaceQuota.Builder mergedBuilder = this.proto.getQuota().toBuilder();
233          // Build a new SpaceQuotas object from merging in the new settings
234          return new SpaceLimitSettings(
235              getTableName(), getNamespace(),
236              buildProtoFromQuota(mergedBuilder.mergeFrom(quotaToMerge).build()));
237        }
238      }
239      // else, we don't know what to do, so return the original object
240    }
241    return this;
242  }
243}
