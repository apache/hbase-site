046/**
047 * Cache that keeps track of the quota settings for the users and tables that are interacting with
048 * it. To avoid blocking the operations if the requested quota is not in cache an "empty quota" will
049 * be returned and the request to fetch the quota information will be enqueued for the next refresh.
050 * TODO: At the moment the Cache has a Chore that will be triggered every 5min or on cache-miss
051 * events. Later the Quotas will be pushed using the notification system.
052 */
053@InterfaceAudience.Private
054@InterfaceStability.Evolving
055public class QuotaCache implements Stoppable {
056  private static final Logger LOG = LoggerFactory.getLogger(QuotaCache.class);
057
058  public static final String REFRESH_CONF_KEY = "hbase.quota.refresh.period";
059  private static final int REFRESH_DEFAULT_PERIOD = 5 * 60000; // 5min
060  private static final int EVICT_PERIOD_FACTOR = 5; // N * REFRESH_DEFAULT_PERIOD
061
062  // for testing purpose only, enforce the cache to be always refreshed
063  static boolean TEST_FORCE_REFRESH = false;
064
065  private final ConcurrentMap<String, QuotaState> namespaceQuotaCache = new ConcurrentHashMap<>();
066  private final ConcurrentMap<TableName, QuotaState> tableQuotaCache = new ConcurrentHashMap<>();
067  private final ConcurrentMap<String, UserQuotaState> userQuotaCache = new ConcurrentHashMap<>();
068  private final ConcurrentMap<String, QuotaState> regionServerQuotaCache =
069    new ConcurrentHashMap<>();
070  private volatile boolean exceedThrottleQuotaEnabled = false;
071  // factors used to divide cluster scope quota into machine scope quota
072  private volatile double machineQuotaFactor = 1;
073  private final ConcurrentHashMap<TableName, Double> tableMachineQuotaFactors =
074    new ConcurrentHashMap<>();
075  private final RegionServerServices rsServices;
076
077  private QuotaRefresherChore refreshChore;
078  private boolean stopped = true;
079
080  public QuotaCache(final RegionServerServices rsServices) {
081    this.rsServices = rsServices;
082  }
083
084  public void start() throws IOException {
085    stopped = false;
086
087    // TODO: This will be replaced once we have the notification bus ready.
088    Configuration conf = rsServices.getConfiguration();
089    int period = conf.getInt(REFRESH_CONF_KEY, REFRESH_DEFAULT_PERIOD);
090    refreshChore = new QuotaRefresherChore(period, this);
091    rsServices.getChoreService().scheduleChore(refreshChore);
092  }
093
094  @Override
095  public void stop(final String why) {
096    if (refreshChore != null) {
097      LOG.debug("Stopping QuotaRefresherChore chore.");
098      refreshChore.shutdown(true);
099    }
100    stopped = true;
101  }
102
103  @Override
104  public boolean isStopped() {
105    return stopped;
106  }

108  /**
109   * Returns the limiter associated to the specified user/table.
110   * @param ugi   the user to limit
111   * @param table the table to limit
112   * @return the limiter associated to the specified user/table
113   */
114  public QuotaLimiter getUserLimiter(final UserGroupInformation ugi, final TableName table) {
115    if (table.isSystemTable()) {
116      return NoopQuotaLimiter.get();
117    }
118    return getUserQuotaState(ugi).getTableLimiter(table);
119  }
120
121  /**
122   * Returns the QuotaState associated to the specified user.
123   * @param ugi the user
124   * @return the quota info associated to specified user
125   */
126  public UserQuotaState getUserQuotaState(final UserGroupInformation ugi) {
127    return computeIfAbsent(userQuotaCache, ugi.getShortUserName(), UserQuotaState::new,
128      this::triggerCacheRefresh);
129  }
130
131  /**
132   * Returns the limiter associated to the specified table.
133   * @param table the table to limit
134   * @return the limiter associated to the specified table
135   */
136  public QuotaLimiter getTableLimiter(final TableName table) {
137    return getQuotaState(this.tableQuotaCache, table).getGlobalLimiter();
138  }
139
140  /**
141   * Returns the limiter associated to the specified namespace.
142   * @param namespace the namespace to limit
143   * @return the limiter associated to the specified namespace

145  public QuotaLimiter getNamespaceLimiter(final String namespace) {
146    return getQuotaState(this.namespaceQuotaCache, namespace).getGlobalLimiter();
147  }

149  /**
150   * Returns the limiter associated to the specified region server.
151   * @param regionServer the region server to limit
152   * @return the limiter associated to the specified region server

154  public QuotaLimiter getRegionServerQuotaLimiter(final String regionServer) {
155    return getQuotaState(this.regionServerQuotaCache, regionServer).getGlobalLimiter();
156  }

158  protected boolean isExceedThrottleQuotaEnabled() {
159    return exceedThrottleQuotaEnabled;
160  }

162  /**
163   * Returns the QuotaState requested. If the quota info is not in cache an empty one will be
164   * returned and the quota request will be enqueued for the next cache refresh.
165   */
166  private <K> QuotaState getQuotaState(final ConcurrentMap<K, QuotaState> quotasMap, final K key) {
167    return computeIfAbsent(quotasMap, key, QuotaState::new, this::triggerCacheRefresh);
168  }
169
170  void triggerCacheRefresh() {
171    refreshChore.triggerNow();
172  }
173
174  long getLastUpdate() {
175    return refreshChore.lastUpdate;
176  }

194  // TODO: Remove this once we have the notification bus
195  private class QuotaRefresherChore extends ScheduledChore {
196    private long lastUpdate = 0;
197
198    public QuotaRefresherChore(final int period, final Stoppable stoppable) {
199      super("QuotaRefresherChore", stoppable, period);
200    }
201
202    @Override
203    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "GC_UNRELATED_TYPES",
204        justification = "I do not understand why the complaints, it looks good to me -- FIX")
205    protected void chore() {
206      // Prefetch online tables/namespaces
207      for (TableName table : ((HRegionServer) QuotaCache.this.rsServices).getOnlineTables()) {
208        if (table.isSystemTable()) {
209          continue;
210        }
211        QuotaCache.this.tableQuotaCache.computeIfAbsent(table, key -> new QuotaState());
212
213        final String ns = table.getNamespaceAsString();
214
215        QuotaCache.this.namespaceQuotaCache.computeIfAbsent(ns, key -> new QuotaState());
216      }
217
218      QuotaCache.this.regionServerQuotaCache
219        .computeIfAbsent(QuotaTableUtil.QUOTA_REGION_SERVER_ROW_KEY, key -> new QuotaState());
220
221      updateQuotaFactors();
222      fetchNamespaceQuotaState();
223      fetchTableQuotaState();
224      fetchUserQuotaState();
225      fetchRegionServerQuotaState();
226      fetchExceedThrottleQuota();
227      lastUpdate = EnvironmentEdgeManager.currentTime();
228    }
229
230    private void fetchNamespaceQuotaState() {
231      fetch("namespace", QuotaCache.this.namespaceQuotaCache, new Fetcher<String, QuotaState>() {
232        @Override
233        public Get makeGet(final Map.Entry<String, QuotaState> entry) {
234          return QuotaUtil.makeGetForNamespaceQuotas(entry.getKey());
235        }
236
237        @Override
238        public Map<String, QuotaState> fetchEntries(final List<Get> gets) throws IOException {
239          return QuotaUtil.fetchNamespaceQuotas(rsServices.getConnection(), gets,
240            machineQuotaFactor);
241        }
242      });
243    }
244
245    private void fetchTableQuotaState() {
246      fetch("table", QuotaCache.this.tableQuotaCache, new Fetcher<TableName, QuotaState>() {
247        @Override
248        public Get makeGet(final Map.Entry<TableName, QuotaState> entry) {
249          return QuotaUtil.makeGetForTableQuotas(entry.getKey());
250        }
251
252        @Override
253        public Map<TableName, QuotaState> fetchEntries(final List<Get> gets) throws IOException {
254          return QuotaUtil.fetchTableQuotas(rsServices.getConnection(), gets,
255            tableMachineQuotaFactors);
256        }
257      });
258    }
259
260    private void fetchUserQuotaState() {
261      final Set<String> namespaces = QuotaCache.this.namespaceQuotaCache.keySet();
262      final Set<TableName> tables = QuotaCache.this.tableQuotaCache.keySet();
263      fetch("user", QuotaCache.this.userQuotaCache, new Fetcher<String, UserQuotaState>() {
264        @Override
265        public Get makeGet(final Map.Entry<String, UserQuotaState> entry) {
266          return QuotaUtil.makeGetForUserQuotas(entry.getKey(), tables, namespaces);
267        }
268
269        @Override
270        public Map<String, UserQuotaState> fetchEntries(final List<Get> gets) throws IOException {
271          return QuotaUtil.fetchUserQuotas(rsServices.getConnection(), gets,
272            tableMachineQuotaFactors, machineQuotaFactor);
273        }
274      });
275    }
276
277    private void fetchRegionServerQuotaState() {
278      fetch("regionServer", QuotaCache.this.regionServerQuotaCache,
279        new Fetcher<String, QuotaState>() {
280          @Override
281          public Get makeGet(final Map.Entry<String, QuotaState> entry) {
282            return QuotaUtil.makeGetForRegionServerQuotas(entry.getKey());
283          }
284
285          @Override
286          public Map<String, QuotaState> fetchEntries(final List<Get> gets) throws IOException {
287            return QuotaUtil.fetchRegionServerQuotas(rsServices.getConnection(), gets);
288          }
289        });
290    }
291
292    private void fetchExceedThrottleQuota() {
293      try {
294        QuotaCache.this.exceedThrottleQuotaEnabled =
295          QuotaUtil.isExceedThrottleQuotaEnabled(rsServices.getConnection());
296      } catch (IOException e) {
297        LOG.warn("Unable to read if exceed throttle quota enabled from quota table", e);
298      }
299    }
300
301    private <K, V extends QuotaState> void fetch(final String type,
302      final ConcurrentMap<K, V> quotasMap, final Fetcher<K, V> fetcher) {
303      long now = EnvironmentEdgeManager.currentTime();
304      long refreshPeriod = getPeriod();
305      long evictPeriod = refreshPeriod * EVICT_PERIOD_FACTOR;
306
307      // Find the quota entries to update
308      List<Get> gets = new ArrayList<>();
309      List<K> toRemove = new ArrayList<>();
310      for (Map.Entry<K, V> entry : quotasMap.entrySet()) {
311        long lastUpdate = entry.getValue().getLastUpdate();
312        long lastQuery = entry.getValue().getLastQuery();
313        if (lastQuery > 0 && (now - lastQuery) >= evictPeriod) {
314          toRemove.add(entry.getKey());
315        } else if (TEST_FORCE_REFRESH || (now - lastUpdate) >= refreshPeriod) {
316          gets.add(fetcher.makeGet(entry));
317        }
318      }
319
320      for (final K key : toRemove) {
321        if (LOG.isTraceEnabled()) {
322          LOG.trace("evict " + type + " key=" + key);
323        }
324        quotasMap.remove(key);
325      }
326
327      // fetch and update the quota entries
328      if (!gets.isEmpty()) {
329        try {
330          for (Map.Entry<K, V> entry : fetcher.fetchEntries(gets).entrySet()) {
331            V quotaInfo = quotasMap.putIfAbsent(entry.getKey(), entry.getValue());
332            if (quotaInfo != null) {
333              quotaInfo.update(entry.getValue());
334            }
335
336            if (LOG.isTraceEnabled()) {
337              LOG.trace("refresh " + type + " key=" + entry.getKey() + " quotas=" + quotaInfo);
338            }
339          }
340        } catch (IOException e) {
341          LOG.warn("Unable to read " + type + " from quota table", e);
342        }
343      }
344    }
345
346    /**
347     * Update quota factors which is used to divide cluster scope quota into machine scope quota For
348     * user/namespace/user over namespace quota, use [1 / RSNum] as machine factor. For table/user
349     * over table quota, use [1 / TotalTableRegionNum * MachineTableRegionNum] as machine factor.
350     */
351    private void updateQuotaFactors() {
352      // Update machine quota factor
353      try {
354        int rsSize = rsServices.getConnection().getAdmin()
355          .getClusterMetrics(EnumSet.of(Option.SERVERS_NAME)).getServersName().size();
356        if (rsSize != 0) {
357          // TODO if use rs group, the cluster limit should be shared by the rs group
358          machineQuotaFactor = 1.0 / rsSize;
359        }
360      } catch (IOException e) {
361        LOG.warn("Get live region servers failed", e);
362      }
363
364      // Update table machine quota factors
365      for (TableName tableName : tableQuotaCache.keySet()) {
366        double factor = 1;
367        try {
368          long regionSize =
369            MetaTableAccessor.getTableRegions(rsServices.getConnection(), tableName, true).stream()
370              .filter(regionInfo -> !regionInfo.isOffline()).count();
371          if (regionSize == 0) {
372            factor = 0;
373          } else {
374            int localRegionSize = rsServices.getRegions(tableName).size();
375            factor = 1.0 * localRegionSize / regionSize;
376          }
377        } catch (IOException e) {
378          LOG.warn("Get table regions failed: {}", tableName, e);
379        }
380        tableMachineQuotaFactors.put(tableName, factor);
381      }
382    }
383  }
384
385  static interface Fetcher<Key, Value> {
386    Get makeGet(Map.Entry<Key, Value> entry);
387
388    Map<Key, Value> fetchEntries(List<Get> gets) throws IOException;
389  }
