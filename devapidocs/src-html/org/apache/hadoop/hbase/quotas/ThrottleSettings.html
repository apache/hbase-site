018package org.apache.hadoop.hbase.quotas;
019
020import java.util.concurrent.TimeUnit;
021
022import org.apache.hadoop.hbase.TableName;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.classification.InterfaceStability;
025import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
026import org.apache.hadoop.hbase.protobuf.generated.MasterProtos.SetQuotaRequest;
027import org.apache.hadoop.hbase.protobuf.generated.QuotaProtos;
028
029@InterfaceAudience.Private
030@InterfaceStability.Evolving
031class ThrottleSettings extends QuotaSettings {
032  private final QuotaProtos.ThrottleRequest proto;
033
034  ThrottleSettings(final String userName, final TableName tableName,
035      final String namespace, final QuotaProtos.ThrottleRequest proto) {
036    super(userName, tableName, namespace);
037    this.proto = proto;
038  }
039
040  public ThrottleType getThrottleType() {
041    return ProtobufUtil.toThrottleType(proto.getType());
042  }
043
044  public long getSoftLimit() {
045    return proto.hasTimedQuota() ? proto.getTimedQuota().getSoftLimit() : -1;
046  }
047
048  public TimeUnit getTimeUnit() {
049    return proto.hasTimedQuota() ?
050      ProtobufUtil.toTimeUnit(proto.getTimedQuota().getTimeUnit()) : null;
051  }
052
053  @Override
054  public QuotaType getQuotaType() {
055    return QuotaType.THROTTLE;
056  }
057
058  @Override
059  protected void setupSetQuotaRequest(SetQuotaRequest.Builder builder) {
060    builder.setThrottle(proto);
061  }
062
063  @Override
064  public String toString() {
065    StringBuilder builder = new StringBuilder();
066    builder.append("TYPE => THROTTLE");
067    if (proto.hasType()) {
068      builder.append(", THROTTLE_TYPE => ");
069      builder.append(proto.getType().toString());
070    }
071    if (proto.hasTimedQuota()) {
072      QuotaProtos.TimedQuota timedQuota = proto.getTimedQuota();
073      builder.append(", LIMIT => ");
074      if (timedQuota.hasSoftLimit()) {
075        switch (getThrottleType()) {
076          case REQUEST_NUMBER:
077          case WRITE_NUMBER:
078          case READ_NUMBER:
079            builder.append(String.format("%dreq", timedQuota.getSoftLimit()));
080            break;
081          case REQUEST_SIZE:
082          case WRITE_SIZE:
083          case READ_SIZE:
084            builder.append(sizeToString(timedQuota.getSoftLimit()));
085            break;
086        }
087      } else if (timedQuota.hasShare()) {
088        builder.append(String.format("%.2f%%", timedQuota.getShare()));
089      }
090      builder.append('/');
091      builder.append(timeToString(ProtobufUtil.toTimeUnit(timedQuota.getTimeUnit())));
092      if (timedQuota.hasScope()) {
093        builder.append(", SCOPE => ");
094        builder.append(timedQuota.getScope().toString());
095      }
096    } else {
097      builder.append(", LIMIT => NONE");
098    }
099    return builder.toString();
100  }
101
102  static ThrottleSettings fromTimedQuota(final String userName,
103      final TableName tableName, final String namespace,
104      ThrottleType type, QuotaProtos.TimedQuota timedQuota) {
105    QuotaProtos.ThrottleRequest.Builder builder = QuotaProtos.ThrottleRequest.newBuilder();
106    builder.setType(ProtobufUtil.toProtoThrottleType(type));
107    builder.setTimedQuota(timedQuota);
108    return new ThrottleSettings(userName, tableName, namespace, builder.build());
109  }
110}
