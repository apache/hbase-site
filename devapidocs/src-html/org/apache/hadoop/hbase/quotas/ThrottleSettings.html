018package org.apache.hadoop.hbase.quotas;
019
020import java.io.IOException;
021import java.util.concurrent.TimeUnit;
022
023import org.apache.hadoop.hbase.DoNotRetryIOException;
024import org.apache.hadoop.hbase.TableName;
025import org.apache.yetus.audience.InterfaceAudience;
026import org.apache.yetus.audience.InterfaceStability;
027import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProtos.SetQuotaRequest;
028import org.apache.hadoop.hbase.shaded.protobuf.generated.QuotaProtos.TimedQuota;
029import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
030import org.apache.hadoop.hbase.shaded.protobuf.generated.QuotaProtos;
031
032@InterfaceAudience.Private
033@InterfaceStability.Evolving
034class ThrottleSettings extends QuotaSettings {
035  final QuotaProtos.ThrottleRequest proto;
036
037  ThrottleSettings(final String userName, final TableName tableName,
038      final String namespace, final QuotaProtos.ThrottleRequest proto) {
039    super(userName, tableName, namespace);
040    this.proto = proto;
041  }
042
043  public ThrottleType getThrottleType() {
044    return ProtobufUtil.toThrottleType(proto.getType());
045  }
046
047  public long getSoftLimit() {
048    return proto.hasTimedQuota() ? proto.getTimedQuota().getSoftLimit() : -1;
049  }
050
051  public TimeUnit getTimeUnit() {
052    return proto.hasTimedQuota() ?
053      ProtobufUtil.toTimeUnit(proto.getTimedQuota().getTimeUnit()) : null;
054  }
055
056  @Override
057  public QuotaType getQuotaType() {
058    return QuotaType.THROTTLE;
059  }
060
061  @Override
062  protected void setupSetQuotaRequest(SetQuotaRequest.Builder builder) {
063    builder.setThrottle(proto);
064  }
065
066  @Override
067  public String toString() {
068    StringBuilder builder = new StringBuilder();
069    builder.append("TYPE => THROTTLE");
070    if (proto.hasType()) {
071      builder.append(", THROTTLE_TYPE => ");
072      builder.append(proto.getType().toString());
073    }
074    if (proto.hasTimedQuota()) {
075      QuotaProtos.TimedQuota timedQuota = proto.getTimedQuota();
076      builder.append(", LIMIT => ");
077      if (timedQuota.hasSoftLimit()) {
078        switch (getThrottleType()) {
079          case REQUEST_NUMBER:
080          case WRITE_NUMBER:
081          case READ_NUMBER:
082            builder.append(String.format("%dreq", timedQuota.getSoftLimit()));
083            break;
084          case REQUEST_SIZE:
085          case WRITE_SIZE:
086          case READ_SIZE:
087            builder.append(sizeToString(timedQuota.getSoftLimit()));
088            break;
089        }
090      } else if (timedQuota.hasShare()) {
091        builder.append(String.format("%.2f%%", timedQuota.getShare()));
092      }
093      builder.append('/');
094      builder.append(timeToString(ProtobufUtil.toTimeUnit(timedQuota.getTimeUnit())));
095      if (timedQuota.hasScope()) {
096        builder.append(", SCOPE => ");
097        builder.append(timedQuota.getScope().toString());
098      }
099    } else {
100      builder.append(", LIMIT => NONE");
101    }
102    return builder.toString();
103  }
104
105  @Override
106  protected ThrottleSettings merge(QuotaSettings other) throws IOException {
107    if (other instanceof ThrottleSettings) {
108      ThrottleSettings otherThrottle = (ThrottleSettings) other;
109
110      // Make sure this and the other target the same "subject"
111      validateQuotaTarget(other);
112
113      QuotaProtos.ThrottleRequest.Builder builder = proto.toBuilder();
114      if (!otherThrottle.proto.hasType()) {
115        return null;
116      }
117
118      QuotaProtos.ThrottleRequest otherProto = otherThrottle.proto;
119      if (otherProto.hasTimedQuota()) {
120        if (otherProto.hasTimedQuota()) {
121          validateTimedQuota(otherProto.getTimedQuota());
122        }
123
124        if (!proto.getType().equals(otherProto.getType())) {
125          throw new IllegalArgumentException(
126              "Cannot merge a ThrottleRequest for " + proto.getType() + " with " +
127                  otherProto.getType());
128        }
129        QuotaProtos.TimedQuota.Builder timedQuotaBuilder = proto.getTimedQuota().toBuilder();
130        timedQuotaBuilder.mergeFrom(otherProto.getTimedQuota());
131
132        QuotaProtos.ThrottleRequest mergedReq = builder.setTimedQuota(
133            timedQuotaBuilder.build()).build();
134        return new ThrottleSettings(getUserName(), getTableName(), getNamespace(), mergedReq);
135      }
136    }
137    return this;
138  }
139
140  private void validateTimedQuota(final TimedQuota timedQuota) throws IOException {
141    if (timedQuota.getSoftLimit() < 1) {
142      throw new DoNotRetryIOException(new UnsupportedOperationException(
143          "The throttle limit must be greater then 0, got " + timedQuota.getSoftLimit()));
144    }
145  }
146
147  static ThrottleSettings fromTimedQuota(final String userName,
148      final TableName tableName, final String namespace,
149      ThrottleType type, QuotaProtos.TimedQuota timedQuota) {
150    QuotaProtos.ThrottleRequest.Builder builder = QuotaProtos.ThrottleRequest.newBuilder();
151    builder.setType(ProtobufUtil.toProtoThrottleType(type));
152    builder.setTimedQuota(timedQuota);
153    return new ThrottleSettings(userName, tableName, namespace, builder.build());
154  }
155}
