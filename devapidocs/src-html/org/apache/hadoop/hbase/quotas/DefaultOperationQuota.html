019package org.apache.hadoop.hbase.quotas;
020
021import java.util.Arrays;
022import java.util.List;
023
024import org.apache.hadoop.conf.Configuration;
025import org.apache.yetus.audience.InterfaceAudience;
026import org.apache.yetus.audience.InterfaceStability;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029import org.apache.hadoop.hbase.client.Mutation;
030import org.apache.hadoop.hbase.client.Result;
031
032@InterfaceAudience.Private
033@InterfaceStability.Evolving
034public class DefaultOperationQuota implements OperationQuota {
035  private static final Logger LOG = LoggerFactory.getLogger(DefaultOperationQuota.class);
036
037  private final List<QuotaLimiter> limiters;
038  private final long writeCapacityUnit;
039  private final long readCapacityUnit;
040
041  private long writeAvailable = 0;
042  private long readAvailable = 0;
043  private long writeConsumed = 0;
044  private long readConsumed = 0;
045  private long writeCapacityUnitConsumed = 0;
046  private long readCapacityUnitConsumed = 0;
047  private final long[] operationSize;
048
049  public DefaultOperationQuota(final Configuration conf, final QuotaLimiter... limiters) {
050    this(conf, Arrays.asList(limiters));
051  }
052
053  /**
054   * NOTE: The order matters. It should be something like [user, table, namespace, global]
055   */
056  public DefaultOperationQuota(final Configuration conf, final List<QuotaLimiter> limiters) {
057    this.writeCapacityUnit =
058        conf.getLong(QuotaUtil.WRITE_CAPACITY_UNIT_CONF_KEY, QuotaUtil.DEFAULT_WRITE_CAPACITY_UNIT);
059    this.readCapacityUnit =
060        conf.getLong(QuotaUtil.READ_CAPACITY_UNIT_CONF_KEY, QuotaUtil.DEFAULT_READ_CAPACITY_UNIT);
061    this.limiters = limiters;
062    int size = OperationType.values().length;
063    operationSize = new long[size];
064
065    for (int i = 0; i < size; ++i) {
066      operationSize[i] = 0;
067    }
068  }
069
070  @Override
071  public void checkQuota(int numWrites, int numReads, int numScans) throws RpcThrottlingException {
072    writeConsumed = estimateConsume(OperationType.MUTATE, numWrites, 100);
073    readConsumed = estimateConsume(OperationType.GET, numReads, 100);
074    readConsumed += estimateConsume(OperationType.SCAN, numScans, 1000);
075
076    writeCapacityUnitConsumed = calculateWriteCapacityUnit(writeConsumed);
077    readCapacityUnitConsumed = calculateReadCapacityUnit(readConsumed);
078
079    writeAvailable = Long.MAX_VALUE;
080    readAvailable = Long.MAX_VALUE;
081    for (final QuotaLimiter limiter : limiters) {
082      if (limiter.isBypass()) continue;
083
084      limiter.checkQuota(numWrites, writeConsumed, numReads + numScans, readConsumed,
085        writeCapacityUnitConsumed, readCapacityUnitConsumed);
086      readAvailable = Math.min(readAvailable, limiter.getReadAvailable());
087      writeAvailable = Math.min(writeAvailable, limiter.getWriteAvailable());
088    }
089
090    for (final QuotaLimiter limiter : limiters) {
091      limiter.grabQuota(numWrites, writeConsumed, numReads + numScans, readConsumed,
092        writeCapacityUnitConsumed, readCapacityUnitConsumed);
093    }
094  }
095
096  @Override
097  public void close() {
098    // Adjust the quota consumed for the specified operation
099    long writeDiff = operationSize[OperationType.MUTATE.ordinal()] - writeConsumed;
100    long readDiff = operationSize[OperationType.GET.ordinal()]
101        + operationSize[OperationType.SCAN.ordinal()] - readConsumed;
102    long writeCapacityUnitDiff = calculateWriteCapacityUnitDiff(
103      operationSize[OperationType.MUTATE.ordinal()], writeConsumed);
104    long readCapacityUnitDiff = calculateReadCapacityUnitDiff(
105      operationSize[OperationType.GET.ordinal()] + operationSize[OperationType.SCAN.ordinal()],
106      readConsumed);
107
108    for (final QuotaLimiter limiter : limiters) {
109      if (writeDiff != 0) {
110        limiter.consumeWrite(writeDiff, writeCapacityUnitDiff);
111      }
112      if (readDiff != 0) {
113        limiter.consumeRead(readDiff, readCapacityUnitDiff);
114      }
115    }
116  }
117
118  @Override
119  public long getReadAvailable() {
120    return readAvailable;
121  }
122
123  @Override
124  public long getWriteAvailable() {
125    return writeAvailable;
126  }
127
128  @Override
129  public void addGetResult(final Result result) {
130    operationSize[OperationType.GET.ordinal()] += QuotaUtil.calculateResultSize(result);
131  }
132
133  @Override
134  public void addScanResult(final List<Result> results) {
135    operationSize[OperationType.SCAN.ordinal()] += QuotaUtil.calculateResultSize(results);
136  }
137
138  @Override
139  public void addMutation(final Mutation mutation) {
140    operationSize[OperationType.MUTATE.ordinal()] += QuotaUtil.calculateMutationSize(mutation);
141  }
142
143  private long estimateConsume(final OperationType type, int numReqs, long avgSize) {
144    if (numReqs > 0) {
145      return avgSize * numReqs;
146    }
147    return 0;
148  }
149
150  private long calculateWriteCapacityUnit(final long size) {
151    return (long) Math.ceil(size * 1.0 / this.writeCapacityUnit);
152  }
153
154  private long calculateReadCapacityUnit(final long size) {
155    return (long) Math.ceil(size * 1.0 / this.readCapacityUnit);
156  }
157
158  private long calculateWriteCapacityUnitDiff(final long actualSize, final long estimateSize) {
159    return calculateWriteCapacityUnit(actualSize) - calculateWriteCapacityUnit(estimateSize);
160  }
161
162  private long calculateReadCapacityUnitDiff(final long actualSize, final long estimateSize) {
163    return calculateReadCapacityUnit(actualSize) - calculateReadCapacityUnit(estimateSize);
164  }
165}
