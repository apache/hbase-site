051/**
052 * Master Quota Manager.
053 * It is responsible for initialize the quota table on the first-run and
054 * provide the admin operations to interact with the quota table.
055 *
056 * TODO: FUTURE: The master will be responsible to notify each RS of quota changes
057 * and it will do the "quota aggregation" when the QuotaScope is CLUSTER.
058 */
059@InterfaceAudience.Private
060@InterfaceStability.Evolving
061public class MasterQuotaManager implements RegionStateListener {
062  private static final Log LOG = LogFactory.getLog(MasterQuotaManager.class);
063  private static final Map<HRegionInfo, Long> EMPTY_MAP = Collections.unmodifiableMap(
064      new HashMap<>());

066  private final MasterServices masterServices;
067  private NamedLock<String> namespaceLocks;
068  private NamedLock<TableName> tableLocks;
069  private NamedLock<String> userLocks;
070  private boolean initialized = false;
071  private NamespaceAuditor namespaceQuotaManager;
072  private ConcurrentHashMap<HRegionInfo, SizeSnapshotWithTimestamp> regionSizes;

074  public MasterQuotaManager(final MasterServices masterServices) {
075    this.masterServices = masterServices;

078  public void start() throws IOException {
079    // If the user doesn't want the quota support skip all the initializations.
080    if (!QuotaUtil.isQuotaEnabled(masterServices.getConfiguration())) {
081      LOG.info("Quota support disabled");
082      return;
083    }
084
085    // Create the quota table if missing
086    if (!MetaTableAccessor.tableExists(masterServices.getConnection(),
087          QuotaUtil.QUOTA_TABLE_NAME)) {
088      LOG.info("Quota table not found. Creating...");
089      createQuotaTable();
090    }
091
092    LOG.info("Initializing quota support");
093    namespaceLocks = new NamedLock<>();
094    tableLocks = new NamedLock<>();
095    userLocks = new NamedLock<>();
096    regionSizes = new ConcurrentHashMap<>();
097
098    namespaceQuotaManager = new NamespaceAuditor(masterServices);
099    namespaceQuotaManager.start();
100    initialized = true;
101  }
102
103  public void stop() {
104  }
105
106  public boolean isQuotaInitialized() {
107    return initialized && namespaceQuotaManager.isInitialized();
108  }

113  public SetQuotaResponse setQuota(final SetQuotaRequest req)
114      throws IOException, InterruptedException {
115    checkQuotaSupport();
116
117    if (req.hasUserName()) {
118      userLocks.lock(req.getUserName());
119      try {
120        if (req.hasTableName()) {
121          setUserQuota(req.getUserName(), ProtobufUtil.toTableName(req.getTableName()), req);
122        } else if (req.hasNamespace()) {
123          setUserQuota(req.getUserName(), req.getNamespace(), req);
124        } else {
125          setUserQuota(req.getUserName(), req);
126        }
127      } finally {
128        userLocks.unlock(req.getUserName());
129      }
130    } else if (req.hasTableName()) {
131      TableName table = ProtobufUtil.toTableName(req.getTableName());
132      tableLocks.lock(table);
133      try {
134        setTableQuota(table, req);
135      } finally {
136        tableLocks.unlock(table);
137      }
138    } else if (req.hasNamespace()) {
139      namespaceLocks.lock(req.getNamespace());
140      try {
141        setNamespaceQuota(req.getNamespace(), req);
142      } finally {
143        namespaceLocks.unlock(req.getNamespace());
144      }
145    } else {
146      throw new DoNotRetryIOException(
147        new UnsupportedOperationException("a user, a table or a namespace must be specified"));
148    }
149    return SetQuotaResponse.newBuilder().build();
150  }

304  private void setQuota(final SetQuotaRequest req, final SetQuotaOperations quotaOps)
305      throws IOException, InterruptedException {
306    if (req.hasRemoveAll() && req.getRemoveAll() == true) {
307      quotaOps.preApply(null);
308      quotaOps.delete();
309      quotaOps.postApply(null);
310      return;
311    }
312
313    // Apply quota changes
314    GlobalQuotaSettings currentQuota = quotaOps.fetch();
315    if (LOG.isTraceEnabled()) {
316      LOG.trace(
317          "Current quota for request(" + TextFormat.shortDebugString(req)
318              + "): " + currentQuota);
319    }
320    // Call the appropriate "pre" CP hook with the current quota value (may be null)
321    quotaOps.preApply(currentQuota);
322    // Translate the protobuf request back into a POJO
323    QuotaSettings newQuota = QuotaSettings.buildFromProto(req);
324    if (LOG.isTraceEnabled()) {
325      LOG.trace("Deserialized quota from request: " + newQuota);
326    }
327
328    // Merge the current quota settings with the new quota settings the user provided.
329    //
330    // NB: while SetQuotaRequest technically allows for multi types of quotas to be set in one
331    // message, the Java API (in Admin/AsyncAdmin) does not. Assume there is only one type.
332    GlobalQuotaSettings mergedQuota = currentQuota.merge(newQuota);
333    if (LOG.isTraceEnabled()) {
334      LOG.trace("Computed merged quota from current quota and user request: " + mergedQuota);
335    }
336
337    // Submit new changes
338    if (mergedQuota == null) {
339      quotaOps.delete();
340    } else {
341      quotaOps.update(mergedQuota);
342    }
343    // Advertise the final result via the "post" CP hook
344    quotaOps.postApply(mergedQuota);
345  }

431  private void checkQuotaSupport() throws IOException {
432    if (!QuotaUtil.isQuotaEnabled(masterServices.getConfiguration())) {
433      throw new DoNotRetryIOException(
434        new UnsupportedOperationException("quota support disabled"));
435    }
436    if (!initialized) {
437      long maxWaitTime = masterServices.getConfiguration().getLong(
438        "hbase.master.wait.for.quota.manager.init", 30000); // default is 30 seconds.
439      long startTime = EnvironmentEdgeManager.currentTime();
440      do {
441        try {
442          Thread.sleep(100);
443        } catch (InterruptedException e) {
444          LOG.warn("Interrupted while waiting for Quota Manager to be initialized.");
445          break;
446        }
447      } while (!initialized && (EnvironmentEdgeManager.currentTime() - startTime) < maxWaitTime);
448      if (!initialized) {
449        throw new IOException("Quota manager is uninitialized, please retry later.");
450      }
451    }
452  }

458  private static class NamedLock<T> {
459    private HashSet<T> locks = new HashSet<>();
460
461    public void lock(final T name) throws InterruptedException {
462      synchronized (locks) {
463        while (locks.contains(name)) {
464          locks.wait();
465        }
466        locks.add(name);
467      }
468    }
469
470    public void unlock(final T name) {
471      synchronized (locks) {
472        locks.remove(name);
473        locks.notifyAll();
474      }
475    }
476  }

485  /**
486   * Holds the size of a region at the given time, millis since the epoch.
487   */
488  private static class SizeSnapshotWithTimestamp {
489    private final long size;
490    private final long time;
491
492    public SizeSnapshotWithTimestamp(long size, long time) {
493      this.size = size;
494      this.time = time;
495    }
496
497    public long getSize() {
498      return size;
499    }
500
501    public long getTime() {
502      return time;
503    }
504
505    public boolean equals(Object o) {
506      if (o instanceof SizeSnapshotWithTimestamp) {
507        SizeSnapshotWithTimestamp other = (SizeSnapshotWithTimestamp) o;
508        return size == other.size && time == other.time;
509      }
510      return false;
511    }
512
513    public int hashCode() {
514      HashCodeBuilder hcb = new HashCodeBuilder();
515      return hcb.append(size).append(time).toHashCode();
516    }
517
518    @Override
519    public String toString() {
520      StringBuilder sb = new StringBuilder(32);
521      sb.append("SizeSnapshotWithTimestamp={size=").append(size).append("B, ");
522      sb.append("time=").append(time).append("}");
523      return sb.toString();
524    }
525  }

527  @VisibleForTesting
528  void initializeRegionSizes() {
529    assert regionSizes == null;
530    this.regionSizes = new ConcurrentHashMap<>();
531  }

533  public void addRegionSize(HRegionInfo hri, long size, long time) {
534    if (regionSizes == null) {
535      return;
536    }
537    regionSizes.put(hri, new SizeSnapshotWithTimestamp(size, time));
538  }
539
540  public Map<HRegionInfo, Long> snapshotRegionSizes() {
541    if (regionSizes == null) {
542      return EMPTY_MAP;
543    }
544
545    Map<HRegionInfo, Long> copy = new HashMap<>();
546    for (Entry<HRegionInfo, SizeSnapshotWithTimestamp> entry : regionSizes.entrySet()) {
547      copy.put(entry.getKey(), entry.getValue().getSize());
548    }
549    return copy;
550  }
551
552  int pruneEntriesOlderThan(long timeToPruneBefore) {
553    if (regionSizes == null) {
554      return 0;
555    }
556    int numEntriesRemoved = 0;
557    Iterator<Entry<HRegionInfo,SizeSnapshotWithTimestamp>> iterator =
558        regionSizes.entrySet().iterator();
559    while (iterator.hasNext()) {
560      long currentEntryTime = iterator.next().getValue().getTime();
561      if (currentEntryTime < timeToPruneBefore) {
562        iterator.remove();
563        numEntriesRemoved++;
564      }
565    }
566    return numEntriesRemoved;
567  }
