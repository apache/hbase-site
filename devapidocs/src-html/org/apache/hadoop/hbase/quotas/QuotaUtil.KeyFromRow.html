162  protected static void switchExceedThrottleQuota(final Connection connection,
163      boolean exceedThrottleQuotaEnabled) throws IOException {
164    if (exceedThrottleQuotaEnabled) {
165      checkRSQuotaToEnableExceedThrottle(
166        getRegionServerQuota(connection, QuotaTableUtil.QUOTA_REGION_SERVER_ROW_KEY));
167    }
168
169    Put put = new Put(getExceedThrottleQuotaRowKey());
170    put.addColumn(QUOTA_FAMILY_INFO, QUOTA_QUALIFIER_SETTINGS,
171      Bytes.toBytes(exceedThrottleQuotaEnabled));
172    doPut(connection, put);
173  }
174
175  private static void checkRSQuotaToEnableExceedThrottle(Quotas quotas) throws IOException {
176    if (quotas != null && quotas.hasThrottle()) {
177      Throttle throttle = quotas.getThrottle();
178      // If enable exceed throttle quota, make sure that there are at least one read(req/read +
179      // num/size/cu) and one write(req/write + num/size/cu) region server throttle quotas.
180      boolean hasReadQuota = false;
181      boolean hasWriteQuota = false;
182      if (throttle.hasReqNum() || throttle.hasReqSize() || throttle.hasReqCapacityUnit()) {
183        hasReadQuota = true;
184        hasWriteQuota = true;
185      }
186      if (!hasReadQuota
187          && (throttle.hasReadNum() || throttle.hasReadSize() || throttle.hasReadCapacityUnit())) {
188        hasReadQuota = true;
189      }
190      if (!hasReadQuota) {
191        throw new DoNotRetryIOException(
192            "Please set at least one read region server quota before enable exceed throttle quota");
193      }
194      if (!hasWriteQuota && (throttle.hasWriteNum() || throttle.hasWriteSize()
195          || throttle.hasWriteCapacityUnit())) {
196        hasWriteQuota = true;
197      }
198      if (!hasWriteQuota) {
199        throw new DoNotRetryIOException("Please set at least one write region server quota "
200            + "before enable exceed throttle quota");
201      }
202      // If enable exceed throttle quota, make sure that region server throttle quotas are in
203      // seconds time unit. Because once previous requests exceed their quota and consume region
204      // server quota, quota in other time units may be refilled in a long time, this may affect
205      // later requests.
206      List<Pair<Boolean, TimedQuota>> list =
207          Arrays.asList(Pair.newPair(throttle.hasReqNum(), throttle.getReqNum()),
208            Pair.newPair(throttle.hasReadNum(), throttle.getReadNum()),
209            Pair.newPair(throttle.hasWriteNum(), throttle.getWriteNum()),
210            Pair.newPair(throttle.hasReqSize(), throttle.getReqSize()),
211            Pair.newPair(throttle.hasReadSize(), throttle.getReadSize()),
212            Pair.newPair(throttle.hasWriteSize(), throttle.getWriteSize()),
213            Pair.newPair(throttle.hasReqCapacityUnit(), throttle.getReqCapacityUnit()),
214            Pair.newPair(throttle.hasReadCapacityUnit(), throttle.getReadCapacityUnit()),
215            Pair.newPair(throttle.hasWriteCapacityUnit(), throttle.getWriteCapacityUnit()));
216      for (Pair<Boolean, TimedQuota> pair : list) {
217        if (pair.getFirst()) {
218          if (pair.getSecond().getTimeUnit() != TimeUnit.SECONDS) {
219            throw new DoNotRetryIOException("All region server quota must be "
220                + "in seconds time unit if enable exceed throttle quota");
221          }
222        }
223      }
224    } else {
225      // If enable exceed throttle quota, make sure that region server quota is already set
226      throw new DoNotRetryIOException(
227          "Please set region server quota before enable exceed throttle quota");
228    }
229  }

231  protected static boolean isExceedThrottleQuotaEnabled(final Connection connection)
232      throws IOException {
233    Get get = new Get(getExceedThrottleQuotaRowKey());
234    get.addColumn(QUOTA_FAMILY_INFO, QUOTA_QUALIFIER_SETTINGS);
235    Result result = doGet(connection, get);
236    if (result.isEmpty()) {
237      return false;
238    }
239    return Bytes.toBoolean(result.getValue(QUOTA_FAMILY_INFO, QUOTA_QUALIFIER_SETTINGS));
240  }

265    doDelete(connection, delete);

268  public static Map<String, UserQuotaState> fetchUserQuotas(final Connection connection,
269      final List<Get> gets) throws IOException {
270    long nowTs = EnvironmentEdgeManager.currentTime();
271    Result[] results = doGet(connection, gets);
272
273    Map<String, UserQuotaState> userQuotas = new HashMap<>(results.length);
274    for (int i = 0; i < results.length; ++i) {
275      byte[] key = gets.get(i).getRow();
276      assert isUserRowKey(key);
277      String user = getUserFromRowKey(key);
278
279      final UserQuotaState quotaInfo = new UserQuotaState(nowTs);
280      userQuotas.put(user, quotaInfo);
281
282      if (results[i].isEmpty()) continue;
283      assert Bytes.equals(key, results[i].getRow());
284
285      try {
286        parseUserResult(user, results[i], new UserQuotasVisitor() {
287          @Override
288          public void visitUserQuotas(String userName, String namespace, Quotas quotas) {
289            quotaInfo.setQuotas(namespace, quotas);
290          }
291
292          @Override
293          public void visitUserQuotas(String userName, TableName table, Quotas quotas) {
294            quotaInfo.setQuotas(table, quotas);
295          }
296
297          @Override
298          public void visitUserQuotas(String userName, Quotas quotas) {
299            quotaInfo.setQuotas(quotas);
300          }
301        });
302      } catch (IOException e) {
303        LOG.error("Unable to parse user '" + user + "' quotas", e);
304        userQuotas.remove(user);
305      }
306    }
307    return userQuotas;
308  }

343  public static <K> Map<K, QuotaState> fetchGlobalQuotas(final String type,
344      final Connection connection, final List<Get> gets, final KeyFromRow<K> kfr)
345  throws IOException {
346    long nowTs = EnvironmentEdgeManager.currentTime();
347    Result[] results = doGet(connection, gets);
348
349    Map<K, QuotaState> globalQuotas = new HashMap<>(results.length);
350    for (int i = 0; i < results.length; ++i) {
351      byte[] row = gets.get(i).getRow();
352      K key = kfr.getKeyFromRow(row);
353
354      QuotaState quotaInfo = new QuotaState(nowTs);
355      globalQuotas.put(key, quotaInfo);
356
357      if (results[i].isEmpty()) continue;
358      assert Bytes.equals(row, results[i].getRow());
359
360      byte[] data = results[i].getValue(QUOTA_FAMILY_INFO, QUOTA_QUALIFIER_SETTINGS);
361      if (data == null) continue;
362
363      try {
364        Quotas quotas = quotasFromData(data);
365        quotaInfo.setQuotas(quotas);
366      } catch (IOException e) {
367        LOG.error("Unable to parse " + type + " '" + key + "' quotas", e);
368        globalQuotas.remove(key);
369      }
370    }
371    return globalQuotas;
372  }

398  public static long calculateMutationSize(final Mutation mutation) {
399    long size = 0;
400    for (Map.Entry<byte[], List<Cell>> entry : mutation.getFamilyCellMap().entrySet()) {
401      for (Cell cell : entry.getValue()) {
402        size += cell.getSerializedSize();
403      }
404    }
405    return size;
406  }
407
408  public static long calculateResultSize(final Result result) {
409    long size = 0;
410    for (Cell cell : result.rawCells()) {
411      size += cell.getSerializedSize();
412    }
413    return size;
414  }
415
416  public static long calculateResultSize(final List<Result> results) {
417    long size = 0;
418    for (Result result: results) {
419      for (Cell cell : result.rawCells()) {
420        size += cell.getSerializedSize();
421      }
422    }
423    return size;
424  }
