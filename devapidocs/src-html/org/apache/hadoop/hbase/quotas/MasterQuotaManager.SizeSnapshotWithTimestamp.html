056/**
057 * Master Quota Manager.
058 * It is responsible for initialize the quota table on the first-run and
059 * provide the admin operations to interact with the quota table.
060 *
061 * TODO: FUTURE: The master will be responsible to notify each RS of quota changes
062 * and it will do the "quota aggregation" when the QuotaScope is CLUSTER.
063 */

083  public void start() throws IOException {
084    // If the user doesn't want the quota support skip all the initializations.
085    if (!QuotaUtil.isQuotaEnabled(masterServices.getConfiguration())) {
086      LOG.info("Quota support disabled");
087      return;
088    }
089
090    // Create the quota table if missing
091    if (!MetaTableAccessor.tableExists(masterServices.getConnection(),
092          QuotaUtil.QUOTA_TABLE_NAME)) {
093      LOG.info("Quota table not found. Creating...");
094      createQuotaTable();
095    }
096
097    LOG.info("Initializing quota support");
098    namespaceLocks = new NamedLock<>();
099    tableLocks = new NamedLock<>();
100    userLocks = new NamedLock<>();
101    regionSizes = new ConcurrentHashMap<>();
102
103    namespaceQuotaManager = new NamespaceAuditor(masterServices);
104    namespaceQuotaManager.start();
105    initialized = true;
106  }
107
108  public void stop() {
109  }

112    return initialized && namespaceQuotaManager.isInitialized();

118  public SetQuotaResponse setQuota(final SetQuotaRequest req)
119      throws IOException, InterruptedException {
120    checkQuotaSupport();
121
122    if (req.hasUserName()) {
123      userLocks.lock(req.getUserName());
124      try {
125        if (req.hasTableName()) {
126          setUserQuota(req.getUserName(), ProtobufUtil.toTableName(req.getTableName()), req);
127        } else if (req.hasNamespace()) {
128          setUserQuota(req.getUserName(), req.getNamespace(), req);
129        } else {
130          setUserQuota(req.getUserName(), req);
131        }
132      } finally {
133        userLocks.unlock(req.getUserName());
134      }
135    } else if (req.hasTableName()) {
136      TableName table = ProtobufUtil.toTableName(req.getTableName());
137      tableLocks.lock(table);
138      try {
139        setTableQuota(table, req);
140      } finally {
141        tableLocks.unlock(table);
142      }
143    } else if (req.hasNamespace()) {
144      namespaceLocks.lock(req.getNamespace());
145      try {
146        setNamespaceQuota(req.getNamespace(), req);
147      } finally {
148        namespaceLocks.unlock(req.getNamespace());
149      }
150    } else {
151      throw new DoNotRetryIOException(
152        new UnsupportedOperationException("a user, a table or a namespace must be specified"));
153    }
154    return SetQuotaResponse.newBuilder().build();
155  }

299  private void setQuota(final SetQuotaRequest req, final SetQuotaOperations quotaOps)
300      throws IOException, InterruptedException {
301    if (req.hasRemoveAll() && req.getRemoveAll() == true) {
302      quotaOps.preApply(null);
303      quotaOps.delete();
304      quotaOps.postApply(null);
305      return;
306    }
307
308    // Apply quota changes
309    Quotas quotas = quotaOps.fetch();
310    quotaOps.preApply(quotas);
311
312    // Copy the user request into the Quotas object
313    Quotas.Builder builder = (quotas != null) ? quotas.toBuilder() : Quotas.newBuilder();
314    if (req.hasThrottle()) applyThrottle(builder, req.getThrottle());
315    if (req.hasBypassGlobals()) applyBypassGlobals(builder, req.getBypassGlobals());
316    if (req.hasSpaceLimit()) applySpaceLimit(builder, req.getSpaceLimit());
317
318    // Submit new changes
319    quotas = builder.build();
320    if (QuotaUtil.isEmptyQuota(quotas)) {
321      quotaOps.delete();
322    } else {
323      quotaOps.update(quotas);
324    }
325    quotaOps.postApply(quotas);
326  }

389   *  Helpers to apply changes to the quotas
390   */
391  private void applyThrottle(final Quotas.Builder quotas, final ThrottleRequest req)
392      throws IOException {
393    Throttle.Builder throttle;
394
395    if (req.hasType() && (req.hasTimedQuota() || quotas.hasThrottle())) {
396      // Validate timed quota if present
397      if (req.hasTimedQuota()) validateTimedQuota(req.getTimedQuota());
398
399      // apply the new settings
400      throttle = quotas.hasThrottle() ? quotas.getThrottle().toBuilder() : Throttle.newBuilder();
401
402      switch (req.getType()) {
403        case REQUEST_NUMBER:
404          if (req.hasTimedQuota()) {
405            throttle.setReqNum(req.getTimedQuota());
406          } else {
407            throttle.clearReqNum();
408          }
409          break;
410        case REQUEST_SIZE:
411          if (req.hasTimedQuota()) {
412            throttle.setReqSize(req.getTimedQuota());
413          } else {
414            throttle.clearReqSize();
415          }
416          break;
417        case WRITE_NUMBER:
418          if (req.hasTimedQuota()) {
419            throttle.setWriteNum(req.getTimedQuota());
420          } else {
421            throttle.clearWriteNum();
422          }
423          break;
424        case WRITE_SIZE:
425          if (req.hasTimedQuota()) {
426            throttle.setWriteSize(req.getTimedQuota());
427          } else {
428            throttle.clearWriteSize();
429          }
430          break;
431        case READ_NUMBER:
432          if (req.hasTimedQuota()) {
433            throttle.setReadNum(req.getTimedQuota());
434          } else {
435            throttle.clearReqNum();
436          }
437          break;
438        case READ_SIZE:
439          if (req.hasTimedQuota()) {
440            throttle.setReadSize(req.getTimedQuota());
441          } else {
442            throttle.clearReadSize();
443          }
444          break;
445      }
446      quotas.setThrottle(throttle.build());
447    } else {
448      quotas.clearThrottle();
449    }
450  }

460  /**
461   * Adds the information from the provided {@link SpaceLimitRequest} to the {@link Quotas} builder.
462   *
463   * @param quotas The builder to update.
464   * @param req The request to extract space quota information from.
465   */
466  void applySpaceLimit(final Quotas.Builder quotas, final SpaceLimitRequest req) {
467    if (req.hasQuota()) {
468      SpaceQuota spaceQuota = req.getQuota();
469      // If we have the remove flag, unset the space quota.
470      if (spaceQuota.getRemove()) {
471        quotas.setSpace(SpaceQuota.getDefaultInstance());
472      } else {
473        // Otherwise, update the new quota
474        applySpaceQuota(quotas, req.getQuota());
475      }
476    }
477  }

479  /**
480   * Merges the provided {@link SpaceQuota} into the given {@link Quotas} builder.
481   *
482   * @param quotas The Quotas builder instance to update
483   * @param quota The SpaceQuota instance to update from
484   */
485  void applySpaceQuota(final Quotas.Builder quotas, final SpaceQuota quota) {
486    // Create a builder for Quotas
487    SpaceQuota.Builder builder = quotas.hasSpace() ? quotas.getSpace().toBuilder() :
488        SpaceQuota.newBuilder();
489    // Update the values from the provided quota into the new one and set it on Quotas.
490    quotas.setSpace(builder.mergeFrom(quota).build());
491  }

504  private void checkQuotaSupport() throws IOException {
505    if (!QuotaUtil.isQuotaEnabled(masterServices.getConfiguration())) {
506      throw new DoNotRetryIOException(
507        new UnsupportedOperationException("quota support disabled"));
508    }
509    if (!initialized) {
510      long maxWaitTime = masterServices.getConfiguration().getLong(
511        "hbase.master.wait.for.quota.manager.init", 30000); // default is 30 seconds.
512      long startTime = EnvironmentEdgeManager.currentTime();
513      do {
514        try {
515          Thread.sleep(100);
516        } catch (InterruptedException e) {
517          LOG.warn("Interrupted while waiting for Quota Manager to be initialized.");
518          break;
519        }
520      } while (!initialized && (EnvironmentEdgeManager.currentTime() - startTime) < maxWaitTime);
521      if (!initialized) {
522        throw new IOException("Quota manager is uninitialized, please retry later.");
523      }
524    }
525  }

531  private static class NamedLock<T> {
532    private HashSet<T> locks = new HashSet<>();
533
534    public void lock(final T name) throws InterruptedException {
535      synchronized (locks) {
536        while (locks.contains(name)) {
537          locks.wait();
538        }
539        locks.add(name);
540      }
541    }
542
543    public void unlock(final T name) {
544      synchronized (locks) {
545        locks.remove(name);
546        locks.notifyAll();
547      }
548    }
549  }

558  /**
559   * Holds the size of a region at the given time, millis since the epoch.
560   */
561  private static class SizeSnapshotWithTimestamp {
562    private final long size;
563    private final long time;
564
565    public SizeSnapshotWithTimestamp(long size, long time) {
566      this.size = size;
567      this.time = time;
568    }
569
570    public long getSize() {
571      return size;
572    }
573
574    public long getTime() {
575      return time;
576    }
577
578    public boolean equals(Object o) {
579      if (o instanceof SizeSnapshotWithTimestamp) {
580        SizeSnapshotWithTimestamp other = (SizeSnapshotWithTimestamp) o;
581        return size == other.size && time == other.time;
582      }
583      return false;
584    }
585
586    public int hashCode() {
587      HashCodeBuilder hcb = new HashCodeBuilder();
588      return hcb.append(size).append(time).toHashCode();
589    }
590
591    @Override
592    public String toString() {
593      StringBuilder sb = new StringBuilder(32);
594      sb.append("SizeSnapshotWithTimestamp={size=").append(size).append("B, ");
595      sb.append("time=").append(time).append("}");
596      return sb.toString();
597    }
598  }

600  @VisibleForTesting
601  void initializeRegionSizes() {
602    assert regionSizes == null;
603    this.regionSizes = new ConcurrentHashMap<>();
604  }
605
606  public void addRegionSize(HRegionInfo hri, long size, long time) {
607    if (regionSizes == null) {
608      return;
609    }
610    regionSizes.put(hri, new SizeSnapshotWithTimestamp(size, time));
611  }
612
613  public Map<HRegionInfo, Long> snapshotRegionSizes() {
614    if (regionSizes == null) {
615      return EMPTY_MAP;
616    }
617
618    Map<HRegionInfo, Long> copy = new HashMap<>();
619    for (Entry<HRegionInfo, SizeSnapshotWithTimestamp> entry : regionSizes.entrySet()) {
620      copy.put(entry.getKey(), entry.getValue().getSize());
621    }
622    return copy;
623  }
624
625  int pruneEntriesOlderThan(long timeToPruneBefore) {
626    if (regionSizes == null) {
627      return 0;
628    }
629    int numEntriesRemoved = 0;
630    Iterator<Entry<HRegionInfo,SizeSnapshotWithTimestamp>> iterator =
631        regionSizes.entrySet().iterator();
632    while (iterator.hasNext()) {
633      long currentEntryTime = iterator.next().getValue().getTime();
634      if (currentEntryTime < timeToPruneBefore) {
635        iterator.remove();
636        numEntriesRemoved++;
637      }
638    }
639    return numEntriesRemoved;
640  }
