051/**
052 * Cache that keeps track of the quota settings for the users and tables that are interacting with
053 * it. To avoid blocking the operations if the requested quota is not in cache an "empty quota" will
054 * be returned and the request to fetch the quota information will be enqueued for the next refresh.
055 * TODO: At the moment the Cache has a Chore that will be triggered every 5min or on cache-miss
056 * events. Later the Quotas will be pushed using the notification system.
057 */
058@InterfaceAudience.Private
059@InterfaceStability.Evolving
060public class QuotaCache implements Stoppable {
061  private static final Logger LOG = LoggerFactory.getLogger(QuotaCache.class);
062
063  public static final String REFRESH_CONF_KEY = "hbase.quota.refresh.period";
064
065  // defines the request attribute key which, when provided, will override the request's username
066  // from the perspective of user quotas
067  public static final String QUOTA_USER_REQUEST_ATTRIBUTE_OVERRIDE_KEY =
068    "hbase.quota.user.override.key";
069  private static final int REFRESH_DEFAULT_PERIOD = 5 * 60000; // 5min
070  private static final int EVICT_PERIOD_FACTOR = 5; // N * REFRESH_DEFAULT_PERIOD
071
072  // for testing purpose only, enforce the cache to be always refreshed
073  static boolean TEST_FORCE_REFRESH = false;
074
075  private final ConcurrentMap<String, QuotaState> namespaceQuotaCache = new ConcurrentHashMap<>();
076  private final ConcurrentMap<TableName, QuotaState> tableQuotaCache = new ConcurrentHashMap<>();
077  private final ConcurrentMap<String, UserQuotaState> userQuotaCache = new ConcurrentHashMap<>();
078  private final ConcurrentMap<String, QuotaState> regionServerQuotaCache =
079    new ConcurrentHashMap<>();
080  private volatile boolean exceedThrottleQuotaEnabled = false;
081  // factors used to divide cluster scope quota into machine scope quota
082  private volatile double machineQuotaFactor = 1;
083  private final ConcurrentHashMap<TableName, Double> tableMachineQuotaFactors =
084    new ConcurrentHashMap<>();
085  private final RegionServerServices rsServices;
086  private final String userOverrideRequestAttributeKey;
087
088  private QuotaRefresherChore refreshChore;
089  private boolean stopped = true;
090
091  public QuotaCache(final RegionServerServices rsServices) {
092    this.rsServices = rsServices;
093    this.userOverrideRequestAttributeKey =
094      rsServices.getConfiguration().get(QUOTA_USER_REQUEST_ATTRIBUTE_OVERRIDE_KEY);
095  }
096
097  public void start() throws IOException {
098    stopped = false;
099
100    // TODO: This will be replaced once we have the notification bus ready.
101    Configuration conf = rsServices.getConfiguration();
102    int period = conf.getInt(REFRESH_CONF_KEY, REFRESH_DEFAULT_PERIOD);
103    refreshChore = new QuotaRefresherChore(period, this);
104    rsServices.getChoreService().scheduleChore(refreshChore);
105  }
106
107  @Override
108  public void stop(final String why) {
109    if (refreshChore != null) {
110      LOG.debug("Stopping QuotaRefresherChore chore.");
111      refreshChore.shutdown(true);
112    }
113    stopped = true;
114  }
115
116  @Override
117  public boolean isStopped() {
118    return stopped;
119  }

121  /**
122   * Returns the limiter associated to the specified user/table.
123   * @param ugi   the user to limit
124   * @param table the table to limit
125   * @return the limiter associated to the specified user/table
126   */
127  public QuotaLimiter getUserLimiter(final UserGroupInformation ugi, final TableName table) {
128    if (table.isSystemTable()) {
129      return NoopQuotaLimiter.get();
130    }
131    return getUserQuotaState(ugi).getTableLimiter(table);
132  }
133
134  /**
135   * Returns the QuotaState associated to the specified user.
136   * @param ugi the user
137   * @return the quota info associated to specified user
138   */
139  public UserQuotaState getUserQuotaState(final UserGroupInformation ugi) {
140    return computeIfAbsent(userQuotaCache, getQuotaUserName(ugi),
141      () -> QuotaUtil.buildDefaultUserQuotaState(rsServices.getConfiguration()),
142      this::triggerCacheRefresh);
143  }

176  /**
177   * Applies a request attribute user override if available, otherwise returns the UGI's short
178   * username
179   * @param ugi The request's UserGroupInformation
180   */
181  private String getQuotaUserName(final UserGroupInformation ugi) {
182    if (userOverrideRequestAttributeKey == null) {
183      return ugi.getShortUserName();
184    }
185
186    Optional<RpcCall> rpcCall = RpcServer.getCurrentCall();
187    if (!rpcCall.isPresent()) {
188      return ugi.getShortUserName();
189    }
190
191    byte[] override = rpcCall.get().getRequestAttribute(userOverrideRequestAttributeKey);
192    if (override == null) {
193      return ugi.getShortUserName();
194    }
195    return Bytes.toString(override);
196  }

198  /**
199   * Returns the QuotaState requested. If the quota info is not in cache an empty one will be
200   * returned and the quota request will be enqueued for the next cache refresh.
201   */
202  private <K> QuotaState getQuotaState(final ConcurrentMap<K, QuotaState> quotasMap, final K key) {
203    return computeIfAbsent(quotasMap, key, QuotaState::new, this::triggerCacheRefresh);
204  }
205
206  void triggerCacheRefresh() {
207    refreshChore.triggerNow();
208  }

210  long getLastUpdate() {
211    return refreshChore.lastUpdate;
212  }

230  // TODO: Remove this once we have the notification bus
231  private class QuotaRefresherChore extends ScheduledChore {
232    private long lastUpdate = 0;
233
234    public QuotaRefresherChore(final int period, final Stoppable stoppable) {
235      super("QuotaRefresherChore", stoppable, period);
236    }
237
238    @Override
239    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "GC_UNRELATED_TYPES",
240        justification = "I do not understand why the complaints, it looks good to me -- FIX")
241    protected void chore() {
242      // Prefetch online tables/namespaces
243      for (TableName table : ((HRegionServer) QuotaCache.this.rsServices).getOnlineTables()) {
244        if (table.isSystemTable()) {
245          continue;
246        }
247        QuotaCache.this.tableQuotaCache.computeIfAbsent(table, key -> new QuotaState());
248
249        final String ns = table.getNamespaceAsString();
250
251        QuotaCache.this.namespaceQuotaCache.computeIfAbsent(ns, key -> new QuotaState());
252      }
253
254      QuotaCache.this.regionServerQuotaCache
255        .computeIfAbsent(QuotaTableUtil.QUOTA_REGION_SERVER_ROW_KEY, key -> new QuotaState());
256
257      updateQuotaFactors();
258      fetchNamespaceQuotaState();
259      fetchTableQuotaState();
260      fetchUserQuotaState();
261      fetchRegionServerQuotaState();
262      fetchExceedThrottleQuota();
263      lastUpdate = EnvironmentEdgeManager.currentTime();
264    }
265
266    private void fetchNamespaceQuotaState() {
267      fetch("namespace", QuotaCache.this.namespaceQuotaCache, new Fetcher<String, QuotaState>() {
268        @Override
269        public Get makeGet(final Map.Entry<String, QuotaState> entry) {
270          return QuotaUtil.makeGetForNamespaceQuotas(entry.getKey());
271        }
272
273        @Override
274        public Map<String, QuotaState> fetchEntries(final List<Get> gets) throws IOException {
275          return QuotaUtil.fetchNamespaceQuotas(rsServices.getConnection(), gets,
276            machineQuotaFactor);
277        }
278      });
279    }
280
281    private void fetchTableQuotaState() {
282      fetch("table", QuotaCache.this.tableQuotaCache, new Fetcher<TableName, QuotaState>() {
283        @Override
284        public Get makeGet(final Map.Entry<TableName, QuotaState> entry) {
285          return QuotaUtil.makeGetForTableQuotas(entry.getKey());
286        }
287
288        @Override
289        public Map<TableName, QuotaState> fetchEntries(final List<Get> gets) throws IOException {
290          return QuotaUtil.fetchTableQuotas(rsServices.getConnection(), gets,
291            tableMachineQuotaFactors);
292        }
293      });
294    }
295
296    private void fetchUserQuotaState() {
297      final Set<String> namespaces = QuotaCache.this.namespaceQuotaCache.keySet();
298      final Set<TableName> tables = QuotaCache.this.tableQuotaCache.keySet();
299      fetch("user", QuotaCache.this.userQuotaCache, new Fetcher<String, UserQuotaState>() {
300        @Override
301        public Get makeGet(final Map.Entry<String, UserQuotaState> entry) {
302          return QuotaUtil.makeGetForUserQuotas(entry.getKey(), tables, namespaces);
303        }
304
305        @Override
306        public Map<String, UserQuotaState> fetchEntries(final List<Get> gets) throws IOException {
307          return QuotaUtil.fetchUserQuotas(rsServices.getConnection(), gets,
308            tableMachineQuotaFactors, machineQuotaFactor);
309        }
310      });
311    }
312
313    private void fetchRegionServerQuotaState() {
314      fetch("regionServer", QuotaCache.this.regionServerQuotaCache,
315        new Fetcher<String, QuotaState>() {
316          @Override
317          public Get makeGet(final Map.Entry<String, QuotaState> entry) {
318            return QuotaUtil.makeGetForRegionServerQuotas(entry.getKey());
319          }
320
321          @Override
322          public Map<String, QuotaState> fetchEntries(final List<Get> gets) throws IOException {
323            return QuotaUtil.fetchRegionServerQuotas(rsServices.getConnection(), gets);
324          }
325        });
326    }
327
328    private void fetchExceedThrottleQuota() {
329      try {
330        QuotaCache.this.exceedThrottleQuotaEnabled =
331          QuotaUtil.isExceedThrottleQuotaEnabled(rsServices.getConnection());
332      } catch (IOException e) {
333        LOG.warn("Unable to read if exceed throttle quota enabled from quota table", e);
334      }
335    }
336
337    private <K, V extends QuotaState> void fetch(final String type,
338      final ConcurrentMap<K, V> quotasMap, final Fetcher<K, V> fetcher) {
339      long now = EnvironmentEdgeManager.currentTime();
340      long refreshPeriod = getPeriod();
341      long evictPeriod = refreshPeriod * EVICT_PERIOD_FACTOR;
342
343      // Find the quota entries to update
344      List<Get> gets = new ArrayList<>();
345      List<K> toRemove = new ArrayList<>();
346      for (Map.Entry<K, V> entry : quotasMap.entrySet()) {
347        long lastUpdate = entry.getValue().getLastUpdate();
348        long lastQuery = entry.getValue().getLastQuery();
349        if (lastQuery > 0 && (now - lastQuery) >= evictPeriod) {
350          toRemove.add(entry.getKey());
351        } else if (TEST_FORCE_REFRESH || (now - lastUpdate) >= refreshPeriod) {
352          gets.add(fetcher.makeGet(entry));
353        }
354      }
355
356      for (final K key : toRemove) {
357        if (LOG.isTraceEnabled()) {
358          LOG.trace("evict " + type + " key=" + key);
359        }
360        quotasMap.remove(key);
361      }
362
363      // fetch and update the quota entries
364      if (!gets.isEmpty()) {
365        try {
366          for (Map.Entry<K, V> entry : fetcher.fetchEntries(gets).entrySet()) {
367            V quotaInfo = quotasMap.putIfAbsent(entry.getKey(), entry.getValue());
368            if (quotaInfo != null) {
369              quotaInfo.update(entry.getValue());
370            }
371
372            if (LOG.isTraceEnabled()) {
373              LOG.trace("refresh " + type + " key=" + entry.getKey() + " quotas=" + quotaInfo);
374            }
375          }
376        } catch (IOException e) {
377          LOG.warn("Unable to read " + type + " from quota table", e);
378        }
379      }
380    }
381
382    /**
383     * Update quota factors which is used to divide cluster scope quota into machine scope quota For
384     * user/namespace/user over namespace quota, use [1 / RSNum] as machine factor. For table/user
385     * over table quota, use [1 / TotalTableRegionNum * MachineTableRegionNum] as machine factor.
386     */
387    private void updateQuotaFactors() {
388      // Update machine quota factor
389      ClusterMetrics clusterMetrics;
390      try {
391        clusterMetrics = rsServices.getConnection().getAdmin()
392          .getClusterMetrics(EnumSet.of(Option.SERVERS_NAME, Option.TABLE_TO_REGIONS_COUNT));
393      } catch (IOException e) {
394        LOG.warn("Failed to get cluster metrics needed for updating quotas", e);
395        return;
396      }
397
398      int rsSize = clusterMetrics.getServersName().size();
399      if (rsSize != 0) {
400        // TODO if use rs group, the cluster limit should be shared by the rs group
401        machineQuotaFactor = 1.0 / rsSize;
402      }
403
404      Map<TableName, RegionStatesCount> tableRegionStatesCount =
405        clusterMetrics.getTableRegionStatesCount();
406
407      // Update table machine quota factors
408      for (TableName tableName : tableQuotaCache.keySet()) {
409        if (tableRegionStatesCount.containsKey(tableName)) {
410          double factor = 1;
411          try {
412            long regionSize = tableRegionStatesCount.get(tableName).getOpenRegions();
413            if (regionSize == 0) {
414              factor = 0;
415            } else {
416              int localRegionSize = rsServices.getRegions(tableName).size();
417              factor = 1.0 * localRegionSize / regionSize;
418            }
419          } catch (IOException e) {
420            LOG.warn("Get table regions failed: {}", tableName, e);
421          }
422          tableMachineQuotaFactors.put(tableName, factor);
423        } else {
424          // TableName might have already been dropped (outdated)
425          tableMachineQuotaFactors.remove(tableName);
426        }
427      }
428    }
429  }
