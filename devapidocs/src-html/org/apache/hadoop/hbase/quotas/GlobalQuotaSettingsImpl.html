038/**
039 * Implementation of {@link GlobalQuotaSettings} to hide the Protobuf messages we use internally.
040 */
041@InterfaceAudience.Private
042public class GlobalQuotaSettingsImpl extends GlobalQuotaSettings {
043
044  private final QuotaProtos.Throttle throttleProto;
045  private final Boolean bypassGlobals;
046  private final QuotaProtos.SpaceQuota spaceProto;
047
048  protected GlobalQuotaSettingsImpl(
049      String username, TableName tableName, String namespace, QuotaProtos.Quotas quotas) {
050    this(username, tableName, namespace,
051        (quotas != null && quotas.hasThrottle() ? quotas.getThrottle() : null),
052        (quotas != null && quotas.hasBypassGlobals() ? quotas.getBypassGlobals() : null),
053        (quotas != null && quotas.hasSpace() ? quotas.getSpace() : null));
054  }
055
056  protected GlobalQuotaSettingsImpl(
057      String userName, TableName tableName, String namespace, QuotaProtos.Throttle throttleProto,
058      Boolean bypassGlobals, QuotaProtos.SpaceQuota spaceProto) {
059    super(userName, tableName, namespace);
060    this.throttleProto = throttleProto;
061    this.bypassGlobals = bypassGlobals;
062    this.spaceProto = spaceProto;
063  }
064
065  @Override
066  public List<QuotaSettings> getQuotaSettings() {
067    // Very similar to QuotaSettingsFactory
068    List<QuotaSettings> settings = new ArrayList<>();
069    if (throttleProto != null) {
070      settings.addAll(QuotaSettingsFactory.fromThrottle(
071          getUserName(), getTableName(), getNamespace(), throttleProto));
072    }
073    if (bypassGlobals != null && bypassGlobals.booleanValue()) {
074      settings.add(new QuotaGlobalsSettingsBypass(
075          getUserName(), getTableName(), getNamespace(), true));
076    }
077    if (spaceProto != null) {
078      settings.add(QuotaSettingsFactory.fromSpace(getTableName(), getNamespace(), spaceProto));
079    }
080    return settings;
081  }
082
083  protected QuotaProtos.Throttle getThrottleProto() {
084    return this.throttleProto;
085  }
086
087  protected Boolean getBypassGlobals() {
088    return this.bypassGlobals;
089  }
090
091  protected QuotaProtos.SpaceQuota getSpaceProto() {
092    return this.spaceProto;
093  }
094
095  /**
096   * Constructs a new {@link Quotas} message from {@code this}.
097   */
098  protected Quotas toQuotas() {
099    QuotaProtos.Quotas.Builder builder = QuotaProtos.Quotas.newBuilder();
100    if (getThrottleProto() != null) {
101      builder.setThrottle(getThrottleProto());
102    }
103    if (getBypassGlobals() != null) {
104      builder.setBypassGlobals(getBypassGlobals());
105    }
106    if (getSpaceProto() != null) {
107      builder.setSpace(getSpaceProto());
108    }
109    return builder.build();
110  }
111
112  @Override
113  protected GlobalQuotaSettingsImpl merge(QuotaSettings other) throws IOException {
114    // Validate the quota subject
115    validateQuotaTarget(other);
116
117    // Propagate the Throttle
118    QuotaProtos.Throttle.Builder throttleBuilder =
119        throttleProto == null ? null : throttleProto.toBuilder();
120
121    if (other instanceof ThrottleSettings) {
122      ThrottleSettings otherThrottle = (ThrottleSettings) other;
123      if (!otherThrottle.proto.hasType() || !otherThrottle.proto.hasTimedQuota()) {
124        // It means it's a remove request
125        // To prevent the "empty" row in QuotaTableUtil.QUOTA_TABLE_NAME
126        throttleBuilder = null;
127      } else {
128        QuotaProtos.ThrottleRequest otherProto = otherThrottle.proto;
129        validateTimedQuota(otherProto.getTimedQuota());
130        if (throttleBuilder == null) {
131          throttleBuilder = QuotaProtos.Throttle.newBuilder();
132        }
133
134        switch (otherProto.getType()) {
135          case REQUEST_NUMBER:
136            throttleBuilder.setReqNum(otherProto.getTimedQuota());
137            break;
138          case REQUEST_SIZE:
139            throttleBuilder.setReqSize(otherProto.getTimedQuota());
140            break;
141          case WRITE_NUMBER:
142            throttleBuilder.setWriteNum(otherProto.getTimedQuota());
143            break;
144          case WRITE_SIZE:
145            throttleBuilder.setWriteSize(otherProto.getTimedQuota());
146            break;
147          case READ_NUMBER:
148            throttleBuilder.setReadNum(otherProto.getTimedQuota());
149            break;
150          case READ_SIZE:
151            throttleBuilder.setReadSize(otherProto.getTimedQuota());
152            break;
153          case REQUEST_CAPACITY_UNIT:
154            throttleBuilder.setReqCapacityUnit(otherProto.getTimedQuota());
155            break;
156          case READ_CAPACITY_UNIT:
157            throttleBuilder.setReadCapacityUnit(otherProto.getTimedQuota());
158            break;
159          case WRITE_CAPACITY_UNIT:
160            throttleBuilder.setWriteCapacityUnit(otherProto.getTimedQuota());
161            break;
162          default:
163        }
164      }
165    }
166
167    // Propagate the space quota portion
168    QuotaProtos.SpaceQuota.Builder spaceBuilder = (spaceProto == null
169        ? null : spaceProto.toBuilder());
170    if (other instanceof SpaceLimitSettings) {
171      if (spaceBuilder == null) {
172        spaceBuilder = QuotaProtos.SpaceQuota.newBuilder();
173      }
174      SpaceLimitSettings settingsToMerge = (SpaceLimitSettings) other;
175
176      QuotaProtos.SpaceLimitRequest spaceRequest = settingsToMerge.getProto();
177
178      // The message contained the expect SpaceQuota object
179      if (spaceRequest.hasQuota()) {
180        SpaceQuota quotaToMerge = spaceRequest.getQuota();
181        // Validate that the two settings are for the same target.
182        // SpaceQuotas either apply to a table or a namespace (no user spacequota).
183        if (!Objects.equals(getTableName(), settingsToMerge.getTableName())
184            && !Objects.equals(getNamespace(), settingsToMerge.getNamespace())) {
185          throw new IllegalArgumentException(
186              "Cannot merge " + settingsToMerge + " into " + this);
187        }
188
189        if (quotaToMerge.getRemove()) {
190          // It means it's a remove request
191          // Update the builder to propagate the removal
192          spaceBuilder.setRemove(true).clearSoftLimit().clearViolationPolicy();
193        } else {
194          // Add the new settings to the existing settings
195          spaceBuilder.mergeFrom(quotaToMerge);
196        }
197      }
198    }
199
200    boolean removeSpaceBuilder =
201        (spaceBuilder == null) || (spaceBuilder.hasRemove() && spaceBuilder.getRemove());
202
203    Boolean bypassGlobals = this.bypassGlobals;
204    if (other instanceof QuotaGlobalsSettingsBypass) {
205      bypassGlobals = ((QuotaGlobalsSettingsBypass) other).getBypass();
206    }
207
208    if (throttleBuilder == null && removeSpaceBuilder && bypassGlobals == null) {
209      return null;
210    }
211
212    return new GlobalQuotaSettingsImpl(
213        getUserName(), getTableName(), getNamespace(),
214        (throttleBuilder == null ? null : throttleBuilder.build()), bypassGlobals,
215        (removeSpaceBuilder ? null : spaceBuilder.build()));
216  }
217
218  private void validateTimedQuota(final TimedQuota timedQuota) throws IOException {
219    if (timedQuota.getSoftLimit() < 1) {
220      throw new DoNotRetryIOException(new UnsupportedOperationException(
221          "The throttle limit must be greater then 0, got " + timedQuota.getSoftLimit()));
222    }
223  }
224
225  @Override
226  public String toString() {
227    StringBuilder builder = new StringBuilder();
228    builder.append("GlobalQuota: ");
229    if (throttleProto != null) {
230      Map<ThrottleType,TimedQuota> throttleQuotas = buildThrottleQuotas(throttleProto);
231      builder.append(" { TYPE => THROTTLE ");
232      for (Entry<ThrottleType,TimedQuota> entry : throttleQuotas.entrySet()) {
233        final ThrottleType type = entry.getKey();
234        final TimedQuota timedQuota = entry.getValue();
235        builder.append("{THROTTLE_TYPE => ").append(type.name()).append(", LIMIT => ");
236        if (timedQuota.hasSoftLimit()) {
237          switch (type) {
238            case REQUEST_NUMBER:
239            case WRITE_NUMBER:
240            case READ_NUMBER:
241              builder.append(String.format("%dreq", timedQuota.getSoftLimit()));
242              break;
243            case REQUEST_SIZE:
244            case WRITE_SIZE:
245            case READ_SIZE:
246              builder.append(sizeToString(timedQuota.getSoftLimit()));
247              break;
248            case REQUEST_CAPACITY_UNIT:
249            case READ_CAPACITY_UNIT:
250            case WRITE_CAPACITY_UNIT:
251              builder.append(String.format("%dCU", timedQuota.getSoftLimit()));
252            default:
253          }
254        } else if (timedQuota.hasShare()) {
255          builder.append(String.format("%.2f%%", timedQuota.getShare()));
256        }
257        builder.append('/');
258        builder.append(timeToString(ProtobufUtil.toTimeUnit(timedQuota.getTimeUnit())));
259        if (timedQuota.hasScope()) {
260          builder.append(", SCOPE => ");
261          builder.append(timedQuota.getScope().toString());
262        }
263      }
264      builder.append( "} } ");
265    } else {
266      builder.append(" {} ");
267    }
268    if (bypassGlobals != null) {
269      builder.append(" { GLOBAL_BYPASS => " + bypassGlobals + " } ");
270    }
271    if (spaceProto != null) {
272      builder.append(" { TYPE => SPACE");
273      if (getTableName() != null) {
274        builder.append(", TABLE => ").append(getTableName());
275      }
276      if (getNamespace() != null) {
277        builder.append(", NAMESPACE => ").append(getNamespace());
278      }
279      if (spaceProto.getRemove()) {
280        builder.append(", REMOVE => ").append(spaceProto.getRemove());
281      } else {
282        builder.append(", LIMIT => ").append(sizeToString(spaceProto.getSoftLimit()));
283        builder.append(", VIOLATION_POLICY => ").append(spaceProto.getViolationPolicy());
284      }
285      builder.append(" } ");
286    }
287    return builder.toString();
288  }
289
290  private Map<ThrottleType,TimedQuota> buildThrottleQuotas(Throttle proto) {
291    HashMap<ThrottleType,TimedQuota> quotas = new HashMap<>();
292    if (proto.hasReadNum()) {
293      quotas.put(ThrottleType.READ_NUMBER, proto.getReadNum());
294    }
295    if (proto.hasReadSize()) {
296      quotas.put(ThrottleType.READ_SIZE, proto.getReadSize());
297    }
298    if (proto.hasReqNum()) {
299      quotas.put(ThrottleType.REQUEST_NUMBER, proto.getReqNum());
300    }
301    if (proto.hasReqSize()) {
302      quotas.put(ThrottleType.REQUEST_SIZE, proto.getReqSize());
303    }
304    if (proto.hasWriteNum()) {
305      quotas.put(ThrottleType.WRITE_NUMBER, proto.getWriteNum());
306    }
307    if (proto.hasWriteSize()) {
308      quotas.put(ThrottleType.WRITE_SIZE, proto.getWriteSize());
309    }
310    if (proto.hasReqCapacityUnit()) {
311      quotas.put(ThrottleType.REQUEST_CAPACITY_UNIT, proto.getReqCapacityUnit());
312    }
313    if (proto.hasReadCapacityUnit()) {
314      quotas.put(ThrottleType.READ_CAPACITY_UNIT, proto.getReqCapacityUnit());
315    }
316    if (proto.hasWriteCapacityUnit()) {
317      quotas.put(ThrottleType.WRITE_CAPACITY_UNIT, proto.getWriteCapacityUnit());
318    }
319    return quotas;
320  }
