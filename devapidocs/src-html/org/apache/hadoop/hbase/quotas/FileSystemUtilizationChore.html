026import org.apache.commons.logging.Log;

040/**
041 * A chore which computes the size of each {@link HRegion} on the FileSystem hosted by the given {@link HRegionServer}.
042 */
043@InterfaceAudience.Private
044public class FileSystemUtilizationChore extends ScheduledChore {
045  private static final Log LOG = LogFactory.getLog(FileSystemUtilizationChore.class);
046  static final String FS_UTILIZATION_CHORE_PERIOD_KEY = "hbase.regionserver.quotas.fs.utilization.chore.period";
047  static final int FS_UTILIZATION_CHORE_PERIOD_DEFAULT = 1000 * 60 * 5; // 5 minutes in millis
048
049  static final String FS_UTILIZATION_CHORE_DELAY_KEY = "hbase.regionserver.quotas.fs.utilization.chore.delay";
050  static final long FS_UTILIZATION_CHORE_DELAY_DEFAULT = 1000L * 60L; // 1 minute
051
052  static final String FS_UTILIZATION_CHORE_TIMEUNIT_KEY = "hbase.regionserver.quotas.fs.utilization.chore.timeunit";
053  static final String FS_UTILIZATION_CHORE_TIMEUNIT_DEFAULT = TimeUnit.MILLISECONDS.name();
054
055  static final String FS_UTILIZATION_MAX_ITERATION_DURATION_KEY = "hbase.regionserver.quotas.fs.utilization.chore.max.iteration.millis";
056  static final long FS_UTILIZATION_MAX_ITERATION_DURATION_DEFAULT = 5000L;
057
058  private int numberOfCyclesToSkip = 0, prevNumberOfCyclesToSkip = 0;
059  private static final int CYCLE_UPPER_BOUND = 32;
060
061  private final HRegionServer rs;
062  private final long maxIterationMillis;
063  private Iterator<Region> leftoverRegions;
064
065  public FileSystemUtilizationChore(HRegionServer rs) {
066    super(FileSystemUtilizationChore.class.getSimpleName(), rs, getPeriod(rs.getConfiguration()),
067        getInitialDelay(rs.getConfiguration()), getTimeUnit(rs.getConfiguration()));
068    this.rs = rs;
069    this.maxIterationMillis = rs.getConfiguration().getLong(
070        FS_UTILIZATION_MAX_ITERATION_DURATION_KEY, FS_UTILIZATION_MAX_ITERATION_DURATION_DEFAULT);
071  }
072
073  @Override
074  protected void chore() {
075    if (numberOfCyclesToSkip > 0) {
076      numberOfCyclesToSkip--;
077      return;
078    }
079    final Map<HRegionInfo,Long> onlineRegionSizes = new HashMap<>();
080    final Set<Region> onlineRegions = new HashSet<>(rs.getOnlineRegions());
081    // Process the regions from the last run if we have any. If we are somehow having difficulty
082    // processing the Regions, we want to avoid creating a backlog in memory of Region objs.
083    Iterator<Region> oldRegionsToProcess = getLeftoverRegions();
084    final Iterator<Region> iterator;
085    final boolean processingLeftovers;
086    if (oldRegionsToProcess == null) {
087      iterator = onlineRegions.iterator();
088      processingLeftovers = false;
089    } else {
090      iterator = oldRegionsToProcess;
091      processingLeftovers = true;
092    }
093    // Reset the leftoverRegions and let the loop re-assign if necessary.
094    setLeftoverRegions(null);
095    long regionSizesCalculated = 0L;
096    long offlineRegionsSkipped = 0L;
097    long skippedSplitParents = 0L;
098    long skippedRegionReplicas = 0L;
099    final long start = EnvironmentEdgeManager.currentTime();
100    while (iterator.hasNext()) {
101      // Make sure this chore doesn't hog the thread.
102      long timeRunning = EnvironmentEdgeManager.currentTime() - start;
103      if (timeRunning > maxIterationMillis) {
104        LOG.debug("Preempting execution of FileSystemUtilizationChore because it exceeds the"
105            + " maximum iteration configuration value. Will process remaining iterators"
106            + " on a subsequent invocation.");
107        setLeftoverRegions(iterator);
108        break;
109      }
110
111      final Region region = iterator.next();
112      // If we're processing leftover regions, the region may no-longer be online.
113      // If so, we can skip it.
114      if (processingLeftovers && !onlineRegions.contains(region)) {
115        offlineRegionsSkipped++;
116        continue;
117      }
118      // Avoid computing the size of regions which are the parent of split.
119      if (region.getRegionInfo().isSplitParent()) {
120        skippedSplitParents++;
121        continue;
122      }
123      // Avoid computing the size of region replicas.
124      if (HRegionInfo.DEFAULT_REPLICA_ID != region.getRegionInfo().getReplicaId()) {
125        skippedRegionReplicas++;
126        continue;
127      }
128      final long sizeInBytes = computeSize(region);
129      onlineRegionSizes.put(region.getRegionInfo(), sizeInBytes);
130      regionSizesCalculated++;
131    }
132    if (LOG.isTraceEnabled()) {
133      LOG.trace("Computed the size of " + regionSizesCalculated + " Regions. Skipped computation"
134          + " of " + offlineRegionsSkipped + " regions due to not being online on this RS, "
135          + skippedSplitParents + " regions due to being the parent of a split, and"
136          + skippedRegionReplicas + " regions due to being region replicas.");
137    }
138    if (!reportRegionSizesToMaster(onlineRegionSizes)) {
139      // backoff reporting
140      numberOfCyclesToSkip = prevNumberOfCyclesToSkip > 0 ? 2 * prevNumberOfCyclesToSkip : 1;
141      if (numberOfCyclesToSkip > CYCLE_UPPER_BOUND) {
142        numberOfCyclesToSkip = CYCLE_UPPER_BOUND;
143      }
144      prevNumberOfCyclesToSkip = numberOfCyclesToSkip;
145    }
146  }
147
148  /**
149   * Returns an {@link Iterator} over the Regions which were skipped last invocation of the chore.
150   *
151   * @return Regions from the previous invocation to process, or null.
152   */
153  Iterator<Region> getLeftoverRegions() {
154    return leftoverRegions;
155  }
156
157  /**
158   * Sets a new collection of Regions as leftovers.
159   */
160  void setLeftoverRegions(Iterator<Region> newLeftovers) {
161    this.leftoverRegions = newLeftovers;
162  }
163
164  /**
165   * Computes total FileSystem size for the given {@link Region}.
166   *
167   * @param r The region
168   * @return The size, in bytes, of the Region.
169   */
170  long computeSize(Region r) {
171    long regionSize = 0L;
172    for (Store store : r.getStores()) {
173      regionSize += store.getHFilesSize();
174    }
175    if (LOG.isTraceEnabled()) {
176      LOG.trace("Size of " + r + " is " + regionSize);
177    }
178    return regionSize;
179  }
180
181  /**
182   * Reports the computed region sizes to the currently active Master.
183   *
184   * @param onlineRegionSizes The computed region sizes to report.
185   * @return {@code false} if FileSystemUtilizationChore should pause reporting to master,
186   *    {@code true} otherwise.
187   */
188  boolean reportRegionSizesToMaster(Map<HRegionInfo,Long> onlineRegionSizes) {
189    return this.rs.reportRegionSizesForQuotas(onlineRegionSizes);
190  }
191
192  /**
193   * Extracts the period for the chore from the configuration.
194   *
195   * @param conf The configuration object.
196   * @return The configured chore period or the default value.
197   */
198  static int getPeriod(Configuration conf) {
199    return conf.getInt(FS_UTILIZATION_CHORE_PERIOD_KEY, FS_UTILIZATION_CHORE_PERIOD_DEFAULT);
200  }
201
202  /**
203   * Extracts the initial delay for the chore from the configuration.
204   *
205   * @param conf The configuration object.
206   * @return The configured chore initial delay or the default value.
207   */
208  static long getInitialDelay(Configuration conf) {
209    return conf.getLong(FS_UTILIZATION_CHORE_DELAY_KEY, FS_UTILIZATION_CHORE_DELAY_DEFAULT);
210  }
211
212  /**
213   * Extracts the time unit for the chore period and initial delay from the configuration. The
214   * configuration value for {@link #FS_UTILIZATION_CHORE_TIMEUNIT_KEY} must correspond to a
215   * {@link TimeUnit} value.
216   *
217   * @param conf The configuration object.
218   * @return The configured time unit for the chore period and initial delay or the default value.
219   */
220  static TimeUnit getTimeUnit(Configuration conf) {
221    return TimeUnit.valueOf(conf.get(FS_UTILIZATION_CHORE_TIMEUNIT_KEY,
222        FS_UTILIZATION_CHORE_TIMEUNIT_DEFAULT));
223  }
