085  public static void resetCounters() throws Exception {
086    Class<?> cl = SplitLogCounters.class;
087    for (Field fld : cl.getDeclaredFields()) {
088      /* Guard against source instrumentation. */
089      if ((!fld.isSynthetic()) && (LongAdder.class.isAssignableFrom(fld.getType()))) {
090        ((LongAdder)fld.get(null)).reset();
091      }
092    }
093  }
