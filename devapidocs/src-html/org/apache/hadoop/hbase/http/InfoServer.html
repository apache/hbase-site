018package org.apache.hadoop.hbase.http;
019
020import java.io.IOException;
021import java.net.URI;
022import javax.servlet.ServletContext;
023import javax.servlet.http.HttpServlet;
024import javax.servlet.http.HttpServletRequest;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.CommonConfigurationKeys;
027import org.apache.hadoop.hbase.HBaseConfiguration;
028import org.apache.hadoop.security.authorize.AccessControlList;
029import org.apache.yetus.audience.InterfaceAudience;
030import org.apache.hbase.thirdparty.com.google.common.net.HostAndPort;
031import org.apache.hbase.thirdparty.org.eclipse.jetty.servlet.ServletHolder;
032
033/**
034 * Create a Jetty embedded server to answer http requests. The primary goal
035 * is to serve up status information for the server.
036 * There are three contexts:
037 *   "/stacks/" -&gt; points to stack trace
038 *   "/static/" -&gt; points to common static files (src/hbase-webapps/static)
039 *   "/" -&gt; the jsp server code from (src/hbase-webapps/&lt;name&gt;)
040 */
041@InterfaceAudience.Private
042public class InfoServer {
043  private static final String HBASE_APP_DIR = "hbase-webapps";
044  private final org.apache.hadoop.hbase.http.HttpServer httpServer;
045
046  /**
047   * Create a status server on the given port.
048   * The jsp scripts are taken from src/hbase-webapps/<code>name</code>.
049   * @param name The name of the server
050   * @param bindAddress address to bind to
051   * @param port The port to use on the server
052   * @param findPort whether the server should start at the given port and increment by 1 until it
053   *                 finds a free port.
054   * @param c the {@link Configuration} to build the server
055   * @throws IOException if getting one of the password fails or the server cannot be created
056   */
057  public InfoServer(String name, String bindAddress, int port, boolean findPort,
058      final Configuration c) throws IOException {
059    HttpConfig httpConfig = new HttpConfig(c);
060    HttpServer.Builder builder =
061      new org.apache.hadoop.hbase.http.HttpServer.Builder();
062
063    builder.setName(name).addEndpoint(URI.create(httpConfig.getSchemePrefix() +
064      HostAndPort.fromParts(bindAddress,port).toString())).
065        setAppDir(HBASE_APP_DIR).setFindPort(findPort).setConf(c);
066    String logDir = System.getProperty("hbase.log.dir");
067    if (logDir != null) {
068      builder.setLogDir(logDir);
069    }
070    if (httpConfig.isSecure()) {
071      builder.keyPassword(HBaseConfiguration
072              .getPassword(c, "ssl.server.keystore.keypassword", null))
073        .keyStore(c.get("ssl.server.keystore.location"),
074                HBaseConfiguration.getPassword(c,"ssl.server.keystore.password", null),
075                c.get("ssl.server.keystore.type", "jks"))
076        .trustStore(c.get("ssl.server.truststore.location"),
077                HBaseConfiguration.getPassword(c, "ssl.server.truststore.password", null),
078                c.get("ssl.server.truststore.type", "jks"));
079      builder.excludeCiphers(c.get("ssl.server.exclude.cipher.list"));
080    }
081    // Enable SPNEGO authentication
082    if ("kerberos".equalsIgnoreCase(c.get(HttpServer.HTTP_UI_AUTHENTICATION, null))) {
083      builder.setUsernameConfKey(HttpServer.HTTP_SPNEGO_AUTHENTICATION_PRINCIPAL_KEY)
084        .setKeytabConfKey(HttpServer.HTTP_SPNEGO_AUTHENTICATION_KEYTAB_KEY)
085        .setKerberosNameRulesKey(HttpServer.HTTP_SPNEGO_AUTHENTICATION_KRB_NAME_KEY)
086        .setSignatureSecretFileKey(
087            HttpServer.HTTP_AUTHENTICATION_SIGNATURE_SECRET_FILE_KEY)
088        .setSecurityEnabled(true);
089
090      // Set an admin ACL on sensitive webUI endpoints
091      AccessControlList acl = buildAdminAcl(c);
092      builder.setACL(acl);
093    }
094    this.httpServer = builder.build();
095  }
096
097  /**
098   * Builds an ACL that will restrict the users who can issue commands to endpoints on the UI
099   * which are meant only for administrators.
100   */
101  AccessControlList buildAdminAcl(Configuration conf) {
102    final String userGroups = conf.get(HttpServer.HTTP_SPNEGO_AUTHENTICATION_ADMIN_USERS_KEY, null);
103    final String adminGroups = conf.get(
104        HttpServer.HTTP_SPNEGO_AUTHENTICATION_ADMIN_GROUPS_KEY, null);
105    if (userGroups == null && adminGroups == null) {
106      // Backwards compatibility - if the user doesn't have anything set, allow all users in.
107      return new AccessControlList("*", null);
108    }
109    return new AccessControlList(userGroups, adminGroups);
110  }
111
112  /**
113   * Explicitly invoke {@link #addPrivilegedServlet(String, String, Class)} or
114   * {@link #addUnprivilegedServlet(String, String, Class)} instead of this method.
115   * This method will add a servlet which any authenticated user can access.
116   *
117   * @deprecated Use {@link #addUnprivilegedServlet(String, String, Class)} or
118   *    {@link #addPrivilegedServlet(String, String, Class)} instead of this
119   *    method which does not state outwardly what kind of authz rules will
120   *    be applied to this servlet.
121   */
122  @Deprecated
123  public void addServlet(String name, String pathSpec,
124          Class<? extends HttpServlet> clazz) {
125    addUnprivilegedServlet(name, pathSpec, clazz);
126  }
127
128  /**
129   * Adds a servlet in the server that any user can access.
130   * @see HttpServer#addUnprivilegedServlet(String, String, Class)
131   */
132  public void addUnprivilegedServlet(String name, String pathSpec,
133          Class<? extends HttpServlet> clazz) {
134    this.httpServer.addUnprivilegedServlet(name, pathSpec, clazz);
135  }
136
137  /**
138   * Adds a servlet in the server that any user can access.
139   * @see HttpServer#addUnprivilegedServlet(String, ServletHolder)
140   */
141  public void addUnprivilegedServlet(String name, String pathSpec, ServletHolder holder) {
142    if (name != null) {
143      holder.setName(name);
144    }
145    this.httpServer.addUnprivilegedServlet(pathSpec, holder);
146  }
147
148  /**
149   * Adds a servlet in the server that any user can access.
150   * @see HttpServer#addPrivilegedServlet(String, String, Class)
151   */
152  public void addPrivilegedServlet(String name, String pathSpec,
153          Class<? extends HttpServlet> clazz) {
154    this.httpServer.addPrivilegedServlet(name, pathSpec, clazz);
155  }
156
157  public void setAttribute(String name, Object value) {
158    this.httpServer.setAttribute(name, value);
159  }
160
161  public void start() throws IOException {
162    this.httpServer.start();
163  }
164
165  /**
166   * @return the port of the info server
167   * @deprecated Since 0.99.0
168   */
169  @Deprecated
170  public int getPort() {
171    return this.httpServer.getPort();
172  }
173
174  public void stop() throws Exception {
175    this.httpServer.stop();
176  }

179  /**
180   * Returns true if and only if UI authentication (spnego) is enabled, UI authorization is enabled,
181   * and the requesting user is defined as an administrator. If the UI is set to readonly, this
182   * method always returns false.
183   */
184  public static boolean canUserModifyUI(
185      HttpServletRequest req, ServletContext ctx, Configuration conf) {
186    if (conf.getBoolean("hbase.master.ui.readonly", false)) {
187      return false;
188    }
189    String remoteUser = req.getRemoteUser();
190    if ("kerberos".equalsIgnoreCase(conf.get(HttpServer.HTTP_UI_AUTHENTICATION)) &&
191        conf.getBoolean(CommonConfigurationKeys.HADOOP_SECURITY_AUTHORIZATION, false) &&
192        remoteUser != null) {
193      return HttpServer.userHasAdministratorAccess(ctx, remoteUser);
194    }
195    return false;
196  }
