051/**
052 * Change log level in runtime.
053 */
054@InterfaceAudience.Private
055public final class LogLevel {
056  private static final String USAGES = "\nUsage: General options are:\n"
057      + "\t[-getlevel <host:port> <classname> [-protocol (http|https)]\n"
058      + "\t[-setlevel <host:port> <classname> <level> [-protocol (http|https)]";
059
060  public static final String PROTOCOL_HTTP = "http";
061  public static final String PROTOCOL_HTTPS = "https";
062
063  public static final String READONLY_LOGGERS_CONF_KEY = "hbase.ui.logLevels.readonly.loggers";
064
065  /**
066   * A command line implementation
067   */
068  public static void main(String[] args) throws Exception {
069    CLI cli = new CLI(new Configuration());
070    System.exit(cli.run(args));
071  }
072
073  /**
074   * Valid command line options.
075   */
076  private enum Operations {
077    GETLEVEL,
078    SETLEVEL,
079    UNKNOWN
080  }
081
082  private static void printUsage() {
083    System.err.println(USAGES);
084    System.exit(-1);
085  }
086
087  public static boolean isValidProtocol(String protocol) {
088    return ((protocol.equals(PROTOCOL_HTTP) ||
089        protocol.equals(PROTOCOL_HTTPS)));
090  }
091
092  static class CLI extends Configured implements Tool {
093    private Operations operation = Operations.UNKNOWN;
094    private String protocol;
095    private String hostName;
096    private String className;
097    private String level;
098
099    CLI(Configuration conf) {
100      setConf(conf);
101    }
102
103    @Override
104    public int run(String[] args) throws Exception {
105      try {
106        parseArguments(args);
107        sendLogLevelRequest();
108      } catch (HadoopIllegalArgumentException e) {
109        printUsage();
110      }
111      return 0;
112    }
113
114    /**
115     * Send HTTP request to the daemon.
116     * @throws HadoopIllegalArgumentException if arguments are invalid.
117     * @throws Exception if unable to connect
118     */
119    private void sendLogLevelRequest()
120        throws HadoopIllegalArgumentException, Exception {
121      switch (operation) {
122        case GETLEVEL:
123          doGetLevel();
124          break;
125        case SETLEVEL:
126          doSetLevel();
127          break;
128        default:
129          throw new HadoopIllegalArgumentException(
130              "Expect either -getlevel or -setlevel");
131      }
132    }
133
134    public void parseArguments(String[] args) throws
135        HadoopIllegalArgumentException {
136      if (args.length == 0) {
137        throw new HadoopIllegalArgumentException("No arguments specified");
138      }
139      int nextArgIndex = 0;
140      while (nextArgIndex < args.length) {
141        switch (args[nextArgIndex]) {
142          case "-getlevel":
143            nextArgIndex = parseGetLevelArgs(args, nextArgIndex);
144            break;
145          case "-setlevel":
146            nextArgIndex = parseSetLevelArgs(args, nextArgIndex);
147            break;
148          case "-protocol":
149            nextArgIndex = parseProtocolArgs(args, nextArgIndex);
150            break;
151          default:
152            throw new HadoopIllegalArgumentException(
153                "Unexpected argument " + args[nextArgIndex]);
154        }
155      }
156
157      // if operation is never specified in the arguments
158      if (operation == Operations.UNKNOWN) {
159        throw new HadoopIllegalArgumentException(
160            "Must specify either -getlevel or -setlevel");
161      }
162
163      // if protocol is unspecified, set it as http.
164      if (protocol == null) {
165        protocol = PROTOCOL_HTTP;
166      }
167    }
168
169    private int parseGetLevelArgs(String[] args, int index) throws
170        HadoopIllegalArgumentException {
171      // fail if multiple operations are specified in the arguments
172      if (operation != Operations.UNKNOWN) {
173        throw new HadoopIllegalArgumentException("Redundant -getlevel command");
174      }
175      // check number of arguments is sufficient
176      if (index + 2 >= args.length) {
177        throw new HadoopIllegalArgumentException("-getlevel needs two parameters");
178      }
179      operation = Operations.GETLEVEL;
180      hostName = args[index + 1];
181      className = args[index + 2];
182      return index + 3;
183    }
184
185    private int parseSetLevelArgs(String[] args, int index) throws
186        HadoopIllegalArgumentException {
187      // fail if multiple operations are specified in the arguments
188      if (operation != Operations.UNKNOWN) {
189        throw new HadoopIllegalArgumentException("Redundant -setlevel command");
190      }
191      // check number of arguments is sufficient
192      if (index + 3 >= args.length) {
193        throw new HadoopIllegalArgumentException("-setlevel needs three parameters");
194      }
195      operation = Operations.SETLEVEL;
196      hostName = args[index + 1];
197      className = args[index + 2];
198      level = args[index + 3];
199      return index + 4;
200    }
201
202    private int parseProtocolArgs(String[] args, int index) throws
203        HadoopIllegalArgumentException {
204      // make sure only -protocol is specified
205      if (protocol != null) {
206        throw new HadoopIllegalArgumentException(
207            "Redundant -protocol command");
208      }
209      // check number of arguments is sufficient
210      if (index + 1 >= args.length) {
211        throw new HadoopIllegalArgumentException(
212            "-protocol needs one parameter");
213      }
214      // check protocol is valid
215      protocol = args[index + 1];
216      if (!isValidProtocol(protocol)) {
217        throw new HadoopIllegalArgumentException(
218            "Invalid protocol: " + protocol);
219      }
220      return index + 2;
221    }
222
223    /**
224     * Send HTTP request to get log level.
225     *
226     * @throws HadoopIllegalArgumentException if arguments are invalid.
227     * @throws Exception if unable to connect
228     */
229    private void doGetLevel() throws Exception {
230      process(protocol + "://" + hostName + "/logLevel?log=" + className);
231    }
232
233    /**
234     * Send HTTP request to set log level.
235     *
236     * @throws HadoopIllegalArgumentException if arguments are invalid.
237     * @throws Exception if unable to connect
238     */
239    private void doSetLevel() throws Exception {
240      process(protocol + "://" + hostName + "/logLevel?log=" + className
241          + "&level=" + level);
242    }
243
244    /**
245     * Connect to the URL. Supports HTTP and supports SPNEGO
246     * authentication. It falls back to simple authentication if it fails to
247     * initiate SPNEGO.
248     *
249     * @param url the URL address of the daemon servlet
250     * @return a connected connection
251     * @throws Exception if it can not establish a connection.
252     */
253    private HttpURLConnection connect(URL url) throws Exception {
254      AuthenticatedURL.Token token = new AuthenticatedURL.Token();
255      AuthenticatedURL aUrl;
256      SSLFactory clientSslFactory;
257      HttpURLConnection connection;
258      // If https is chosen, configures SSL client.
259      if (PROTOCOL_HTTPS.equals(url.getProtocol())) {
260        clientSslFactory = new SSLFactory(SSLFactory.Mode.CLIENT, this.getConf());
261        clientSslFactory.init();
262        SSLSocketFactory sslSocketF = clientSslFactory.createSSLSocketFactory();
263
264        aUrl = new AuthenticatedURL(new KerberosAuthenticator(), clientSslFactory);
265        connection = aUrl.openConnection(url, token);
266        HttpsURLConnection httpsConn = (HttpsURLConnection) connection;
267        httpsConn.setSSLSocketFactory(sslSocketF);
268      } else {
269        aUrl = new AuthenticatedURL(new KerberosAuthenticator());
270        connection = aUrl.openConnection(url, token);
271      }
272      connection.connect();
273      return connection;
274    }
275
276    /**
277     * Configures the client to send HTTP request to the URL.
278     * Supports SPENGO for authentication.
279     * @param urlString URL and query string to the daemon's web UI
280     * @throws Exception if unable to connect
281     */
282    private void process(String urlString) throws Exception {
283      URL url = new URL(urlString);
284      System.out.println("Connecting to " + url);
285
286      HttpURLConnection connection = connect(url);
287
288      HttpExceptionUtils.validateResponse(connection, 200);
289
290      // read from the servlet
291
292      try (InputStreamReader streamReader =
293            new InputStreamReader(connection.getInputStream(), StandardCharsets.UTF_8);
294           BufferedReader bufferedReader = new BufferedReader(streamReader)) {
295        bufferedReader.lines().filter(Objects::nonNull).filter(line -> line.startsWith(MARKER))
296            .forEach(line -> System.out.println(TAG.matcher(line).replaceAll("")));
297      } catch (IOException ioe) {
298        System.err.println("" + ioe);
299      }
300    }
301  }
302
303  private static final String MARKER = "<!-- OUTPUT -->";
304  private static final Pattern TAG = Pattern.compile("<[^>]*>");

306  /**
307   * A servlet implementation
308   */
309  @InterfaceAudience.Private
310  public static class Servlet extends HttpServlet {
311    private static final long serialVersionUID = 1L;
312
313    @Override
314    public void doGet(HttpServletRequest request, HttpServletResponse response)
315            throws ServletException, IOException {
316      // Do the authorization
317      if (!HttpServer.hasAdministratorAccess(getServletContext(), request,
318          response)) {
319        return;
320      }
321      // Disallow modification of the LogLevel if explicitly set to readonly
322      Configuration conf = (Configuration) getServletContext().getAttribute(
323          HttpServer.CONF_CONTEXT_ATTRIBUTE);
324      if (conf.getBoolean("hbase.master.ui.readonly", false)) {
325        sendError(
326          response,
327          HttpServletResponse.SC_FORBIDDEN,
328          "Modification of HBase via the UI is disallowed in configuration.");
329        return;
330      }
331      response.setContentType("text/html");
332      PrintWriter out;
333      try {
334        String headerPath = "header.jsp?pageTitle=Log Level";
335        request.getRequestDispatcher(headerPath).include(request, response);
336        out = response.getWriter();
337      } catch (FileNotFoundException e) {
338        // in case file is not found fall back to old design
339        out = ServletUtil.initHTML(response, "Log Level");
340      }
341      out.println(FORMS);
342
343      String logName = ServletUtil.getParameter(request, "log");
344      String level = ServletUtil.getParameter(request, "level");
345
346      String[] readOnlyLogLevels = conf.getStrings(READONLY_LOGGERS_CONF_KEY);
347
348      if (logName != null) {
349        out.println("<p>Results:</p>");
350        out.println(MARKER
351            + "Submitted Log Name: <b>" + logName + "</b><br />");
352
353        Logger log = LoggerFactory.getLogger(logName);
354        out.println(MARKER
355            + "Log Class: <b>" + log.getClass().getName() +"</b><br />");
356        if (level != null) {
357          if (!isLogLevelChangeAllowed(logName, readOnlyLogLevels)) {
358            sendError(
359              response,
360              HttpServletResponse.SC_PRECONDITION_FAILED,
361              "Modification of logger " + logName + " is disallowed in configuration.");
362            return;
363          }
364
365          out.println(MARKER + "Submitted Level: <b>" + level + "</b><br />");
366        }
367        process(log, level, out);
368      }
369
370      try {
371        String footerPath = "footer.jsp";
372        out.println("</div>");
373        request.getRequestDispatcher(footerPath).include(request, response);
374      } catch (FileNotFoundException e) {
375        out.println(ServletUtil.HTML_TAIL);
376      }
377      out.close();
378    }
379
380    private boolean isLogLevelChangeAllowed(String logger, String[] readOnlyLogLevels) {
381      if (readOnlyLogLevels == null) {
382        return true;
383      }
384      for (String readOnlyLogLevel : readOnlyLogLevels) {
385        if (logger.startsWith(readOnlyLogLevel)) {
386          return false;
387        }
388      }
389      return true;
390    }
391
392    private void sendError(HttpServletResponse response, int code, String message)
393      throws IOException {
394      response.setStatus(code, message);
395      response.sendError(code, message);
396    }

419    private static void process(Logger logger, String levelName, PrintWriter out) {
420      if (levelName != null) {
421        try {
422          Log4jUtils.setLogLevel(logger.getName(), levelName);
423          out.println(MARKER + "<div class='text-success'>" + "Setting Level to <strong>" +
424            levelName + "</strong> ...<br />" + "</div>");
425        } catch (IllegalArgumentException e) {
426          out.println(MARKER + "<div class='text-danger'>" + "Bad level : <strong>" + levelName +
427            "</strong><br />" + "</div>");
428        }
429      }
430      out.println(MARKER + "Effective level: <b>" + Log4jUtils.getEffectiveLevel(logger.getName()) +
431        "</b><br />");
432    }
