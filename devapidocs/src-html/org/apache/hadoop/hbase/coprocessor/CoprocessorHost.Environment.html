095  /**
096   * Not to be confused with the per-object _coprocessors_ (above),
097   * coprocessorNames is static and stores the set of all coprocessors ever
098   * loaded by any thread in this JVM. It is strictly additive: coprocessors are
099   * added to coprocessorNames, by loadInstance() but are never removed, since
100   * the intention is to preserve a history of all loaded coprocessors for
101   * diagnosis in case of server crash (HBASE-4014).
102   */
103  private static Set<String> coprocessorNames =
104      Collections.synchronizedSet(new HashSet<String>());
105
106  public static Set<String> getLoadedCoprocessors() {
107    synchronized (coprocessorNames) {
108      return new HashSet(coprocessorNames);
109    }
110  }

127  /**
128   * Load system coprocessors once only. Read the class names from configuration.
129   * Called by constructor.
130   */
131  protected void loadSystemCoprocessors(Configuration conf, String confKey) {
132    boolean coprocessorsEnabled = conf.getBoolean(COPROCESSORS_ENABLED_CONF_KEY,
133      DEFAULT_COPROCESSORS_ENABLED);
134    if (!coprocessorsEnabled) {
135      return;
136    }
137
138    Class<?> implClass = null;
139
140    // load default coprocessors from configure file
141    String[] defaultCPClasses = conf.getStrings(confKey);
142    if (defaultCPClasses == null || defaultCPClasses.length == 0)
143      return;
144
145    int priority = Coprocessor.PRIORITY_SYSTEM;
146    for (String className : defaultCPClasses) {
147      className = className.trim();
148      if (findCoprocessor(className) != null) {
149        // If already loaded will just continue
150        LOG.warn("Attempted duplicate loading of " + className + "; skipped");
151        continue;
152      }
153      ClassLoader cl = this.getClass().getClassLoader();
154      Thread.currentThread().setContextClassLoader(cl);
155      try {
156        implClass = cl.loadClass(className);
157        // Add coprocessors as we go to guard against case where a coprocessor is specified twice
158        // in the configuration
159        this.coprocessors.add(loadInstance(implClass, priority, conf));
160        LOG.info("System coprocessor " + className + " was loaded " +
161            "successfully with priority (" + priority + ").");
162        ++priority;
163      } catch (Throwable t) {
164        // We always abort if system coprocessors cannot be loaded
165        abortServer(className, t);
166      }
167    }
168  }

170  /**
171   * Load a coprocessor implementation into the host
172   * @param path path to implementation jar
173   * @param className the main class name
174   * @param priority chaining priority
175   * @param conf configuration for coprocessor
176   * @throws java.io.IOException Exception
177   */
178  public E load(Path path, String className, int priority,
179      Configuration conf) throws IOException {
180    String[] includedClassPrefixes = null;
181    if (conf.get(HConstants.CP_HTD_ATTR_INCLUSION_KEY) != null){
182      String prefixes = conf.get(HConstants.CP_HTD_ATTR_INCLUSION_KEY);
183      includedClassPrefixes = prefixes.split(";");
184    }
185    return load(path, className, priority, conf, includedClassPrefixes);
186  }

188  /**
189   * Load a coprocessor implementation into the host
190   * @param path path to implementation jar
191   * @param className the main class name
192   * @param priority chaining priority
193   * @param conf configuration for coprocessor
194   * @param includedClassPrefixes class name prefixes to include
195   * @throws java.io.IOException Exception
196   */
197  public E load(Path path, String className, int priority,
198      Configuration conf, String[] includedClassPrefixes) throws IOException {
199    Class<?> implClass = null;
200    LOG.debug("Loading coprocessor class " + className + " with path " +
201        path + " and priority " + priority);
202
203    ClassLoader cl = null;
204    if (path == null) {
205      try {
206        implClass = getClass().getClassLoader().loadClass(className);
207      } catch (ClassNotFoundException e) {
208        throw new IOException("No jar path specified for " + className);
209      }
210    } else {
211      cl = CoprocessorClassLoader.getClassLoader(
212        path, getClass().getClassLoader(), pathPrefix, conf);
213      try {
214        implClass = ((CoprocessorClassLoader)cl).loadClass(className, includedClassPrefixes);
215      } catch (ClassNotFoundException e) {
216        throw new IOException("Cannot load external coprocessor class " + className, e);
217      }
218    }
219
220    //load custom code for coprocessor
221    Thread currentThread = Thread.currentThread();
222    ClassLoader hostClassLoader = currentThread.getContextClassLoader();
223    try{
224      // switch temporarily to the thread classloader for custom CP
225      currentThread.setContextClassLoader(cl);
226      E cpInstance = loadInstance(implClass, priority, conf);
227      return cpInstance;
228    } finally {
229      // restore the fresh (host) classloader
230      currentThread.setContextClassLoader(hostClassLoader);
231    }
232  }

246  /**
247   * @param implClass Implementation class
248   * @param priority priority
249   * @param conf configuration
250   * @throws java.io.IOException Exception
251   */
252  public E loadInstance(Class<?> implClass, int priority, Configuration conf)
253      throws IOException {
254    if (!Coprocessor.class.isAssignableFrom(implClass)) {
255      throw new IOException("Configured class " + implClass.getName() + " must implement "
256          + Coprocessor.class.getName() + " interface ");
257    }
258
259    // create the instance
260    Coprocessor impl;
261    Object o = null;
262    try {
263      o = implClass.newInstance();
264      impl = (Coprocessor)o;
265    } catch (InstantiationException e) {
266      throw new IOException(e);
267    } catch (IllegalAccessException e) {
268      throw new IOException(e);
269    }
270    // create the environment
271    E env = createEnvironment(implClass, impl, priority, loadSequence.incrementAndGet(), conf);
272    if (env instanceof Environment) {
273      ((Environment)env).startup();
274    }
275    // HBASE-4014: maintain list of loaded coprocessors for later crash analysis
276    // if server (master or regionserver) aborts.
277    coprocessorNames.add(implClass.getName());
278    return env;
279  }

287  public void shutdown(CoprocessorEnvironment e) {
288    if (e instanceof Environment) {
289      if (LOG.isDebugEnabled()) {
290        LOG.debug("Stop coprocessor " + e.getInstance().getClass().getName());
291      }
292      ((Environment)e).shutdown();
293    } else {
294      LOG.warn("Shutdown called on unknown environment: "+
295          e.getClass().getName());
296    }

299  /**
300   * Find a coprocessor implementation by class name
301   * @param className the class name
302   * @return the coprocessor, or null if not found
303   */
304  public Coprocessor findCoprocessor(String className) {
305    for (E env: coprocessors) {
306      if (env.getInstance().getClass().getName().equals(className) ||
307          env.getInstance().getClass().getSimpleName().equals(className)) {
308        return env.getInstance();
309      }
310    }
311    return null;
312  }

314  /**
315   * Find list of coprocessors that extend/implement the given class/interface
316   * @param cls the class/interface to look for
317   * @return the list of coprocessors, or null if not found
318   */
319  public <T extends Coprocessor> List<T> findCoprocessors(Class<T> cls) {
320    ArrayList<T> ret = new ArrayList<>();
321
322    for (E env: coprocessors) {
323      Coprocessor cp = env.getInstance();
324
325      if(cp != null) {
326        if (cls.isAssignableFrom(cp.getClass())) {
327          ret.add((T)cp);
328        }
329      }
330    }
331    return ret;
332  }

369  /**
370   * Retrieves the set of classloaders used to instantiate Coprocessor classes defined in external
371   * jar files.
372   * @return A set of ClassLoader instances
373   */
374  Set<ClassLoader> getExternalClassLoaders() {
375    Set<ClassLoader> externalClassLoaders = new HashSet<>();
376    final ClassLoader systemClassLoader = this.getClass().getClassLoader();
377    for (E env : coprocessors) {
378      ClassLoader cl = env.getInstance().getClass().getClassLoader();
379      if (cl != systemClassLoader){
380        //do not include system classloader
381        externalClassLoaders.add(cl);
382      }
383    }
384    return externalClassLoaders;
385  }

387  /**
388   * Environment priority comparator.
389   * Coprocessors are chained in sorted order.
390   */
391  static class EnvironmentPriorityComparator
392      implements Comparator<CoprocessorEnvironment> {
393    @Override
394    public int compare(final CoprocessorEnvironment env1,
395        final CoprocessorEnvironment env2) {
396      if (env1.getPriority() < env2.getPriority()) {
397        return -1;
398      } else if (env1.getPriority() > env2.getPriority()) {
399        return 1;
400      }
401      if (env1.getLoadSequence() < env2.getLoadSequence()) {
402        return -1;
403      } else if (env1.getLoadSequence() > env2.getLoadSequence()) {
404        return 1;
405      }
406      return 0;
407    }
408  }

410  /**
411   * Encapsulation of the environment of each coprocessor
412   */
413  public static class Environment implements CoprocessorEnvironment {
414
415    /** The coprocessor */
416    public Coprocessor impl;
417    /** Chaining priority */
418    protected int priority = Coprocessor.PRIORITY_USER;
419    /** Current coprocessor state */
420    Coprocessor.State state = Coprocessor.State.UNINSTALLED;
421    /** Accounting for tables opened by the coprocessor */
422    protected List<Table> openTables =
423      Collections.synchronizedList(new ArrayList<Table>());
424    private int seq;
425    private Configuration conf;
426    private ClassLoader classLoader;
427
428    /**
429     * Constructor
430     * @param impl the coprocessor instance
431     * @param priority chaining priority
432     */
433    public Environment(final Coprocessor impl, final int priority,
434        final int seq, final Configuration conf) {
435      this.impl = impl;
436      this.classLoader = impl.getClass().getClassLoader();
437      this.priority = priority;
438      this.state = Coprocessor.State.INSTALLED;
439      this.seq = seq;
440      this.conf = conf;
441    }
442
443    /** Initialize the environment */
444    public void startup() throws IOException {
445      if (state == Coprocessor.State.INSTALLED ||
446          state == Coprocessor.State.STOPPED) {
447        state = Coprocessor.State.STARTING;
448        Thread currentThread = Thread.currentThread();
449        ClassLoader hostClassLoader = currentThread.getContextClassLoader();
450        try {
451          currentThread.setContextClassLoader(this.getClassLoader());
452          impl.start(this);
453          state = Coprocessor.State.ACTIVE;
454        } finally {
455          currentThread.setContextClassLoader(hostClassLoader);
456        }
457      } else {
458        LOG.warn("Not starting coprocessor "+impl.getClass().getName()+
459            " because not inactive (state="+state.toString()+")");
460      }
461    }
462
463    /** Clean up the environment */
464    protected void shutdown() {
465      if (state == Coprocessor.State.ACTIVE) {
466        state = Coprocessor.State.STOPPING;
467        Thread currentThread = Thread.currentThread();
468        ClassLoader hostClassLoader = currentThread.getContextClassLoader();
469        try {
470          currentThread.setContextClassLoader(this.getClassLoader());
471          impl.stop(this);
472          state = Coprocessor.State.STOPPED;
473        } catch (IOException ioe) {
474          LOG.error("Error stopping coprocessor "+impl.getClass().getName(), ioe);
475        } finally {
476          currentThread.setContextClassLoader(hostClassLoader);
477        }
478      } else {
479        LOG.warn("Not stopping coprocessor "+impl.getClass().getName()+
480            " because not active (state="+state.toString()+")");
481      }
482      synchronized (openTables) {
483        // clean up any table references
484        for (Table table: openTables) {
485          try {
486            ((HTableWrapper)table).internalClose();
487          } catch (IOException e) {
488            // nothing can be done here
489            LOG.warn("Failed to close " +
490                table.getName(), e);
491          }
492        }
493      }
494    }
495
496    @Override
497    public Coprocessor getInstance() {
498      return impl;
499    }
500
501    @Override
502    public ClassLoader getClassLoader() {
503      return classLoader;
504    }
505
506    @Override
507    public int getPriority() {
508      return priority;
509    }
510
511    @Override

513      return seq;
514    }

560  protected void abortServer(final String coprocessorName, final Throwable e) {
561    String message = "The coprocessor " + coprocessorName + " threw " + e.toString();
562    LOG.error(message, e);
563    if (abortable != null) {
564      abortable.abort(message, e);
565    } else {
566      LOG.warn("No available Abortable, process was not aborted");
567    }

570  /**
571   * This is used by coprocessor hooks which are declared to throw IOException
572   * (or its subtypes). For such hooks, we should handle throwable objects
573   * depending on the Throwable's type. Those which are instances of
574   * IOException should be passed on to the client. This is in conformance with
575   * the HBase idiom regarding IOException: that it represents a circumstance
576   * that should be passed along to the client for its own handling. For
577   * example, a coprocessor that implements access controls would throw a
578   * subclass of IOException, such as AccessDeniedException, in its preGet()
579   * method to prevent an unauthorized client's performing a Get on a particular
580   * table.
581   * @param env Coprocessor Environment
582   * @param e Throwable object thrown by coprocessor.
583   * @exception IOException Exception
584   */
585  // Note to devs: Class comments of all observers ({@link MasterObserver}, {@link WALObserver},
586  // etc) mention this nuance of our exception handling so that coprocessor can throw appropriate
587  // exceptions depending on situation. If any changes are made to this logic, make sure to
588  // update all classes' comments.
589  protected void handleCoprocessorThrowable(final CoprocessorEnvironment env, final Throwable e)
590      throws IOException {
591    if (e instanceof IOException) {
592      throw (IOException)e;
593    }
594    // If we got here, e is not an IOException. A loaded coprocessor has a
595    // fatal bug, and the server (master or regionserver) should remove the
596    // faulty coprocessor from its set of active coprocessors. Setting
597    // 'hbase.coprocessor.abortonerror' to true will cause abortServer(),
598    // which may be useful in development and testing environments where
599    // 'failing fast' for error analysis is desired.
600    if (env.getConfiguration().getBoolean(ABORT_ON_ERROR_KEY, DEFAULT_ABORT_ON_ERROR)) {
601      // server is configured to abort.
602      abortServer(env, e);
603    } else {
604      // If available, pull a table name out of the environment
605      if(env instanceof RegionCoprocessorEnvironment) {
606        String tableName = ((RegionCoprocessorEnvironment)env).getRegionInfo().getTable().getNameAsString();
607        LOG.error("Removing coprocessor '" + env.toString() + "' from table '"+ tableName + "'", e);
608      } else {
609        LOG.error("Removing coprocessor '" + env.toString() + "' from " +
610                "environment",e);
611      }
612
613      coprocessors.remove(env);
614      try {
615        shutdown(env);
616      } catch (Exception x) {
617        LOG.error("Uncaught exception when shutting down coprocessor '"
618            + env.toString() + "'", x);
619      }
620      throw new DoNotRetryIOException("Coprocessor: '" + env.toString() +
621          "' threw: '" + e + "' and has been removed from the active " +
622          "coprocessor set.", e);
623    }
624  }

626  /**
627   * Used to gracefully handle fallback to deprecated methods when we
628   * evolve coprocessor APIs.
629   *
630   * When a particular Coprocessor API is updated to change methods, hosts can support fallback
631   * to the deprecated API by using this method to determine if an instance implements the new API.
632   * In the event that said support is partial, then in the face of a runtime issue that prevents
633   * proper operation {@link #legacyWarning(Class, String)} should be used to let operators know.
634   *
635   * For examples of this in action, see the implementation of
636   * <ul>
637   *   <li>{@link org.apache.hadoop.hbase.regionserver.RegionCoprocessorHost}
638   *   <li>{@link org.apache.hadoop.hbase.regionserver.wal.WALCoprocessorHost}
639   * </ul>
640   *
641   * @param clazz Coprocessor you wish to evaluate
642   * @param methodName the name of the non-deprecated method version
643   * @param parameterTypes the Class of the non-deprecated method's arguments in the order they are
644   *     declared.
645   */
646  @InterfaceAudience.Private
647  protected static boolean useLegacyMethod(final Class<? extends Coprocessor> clazz,
648      final String methodName, final Class<?>... parameterTypes) {
649    boolean useLegacy;
650    // Use reflection to see if they implement the non-deprecated version
651    try {
652      clazz.getDeclaredMethod(methodName, parameterTypes);
653      LOG.debug("Found an implementation of '" + methodName + "' that uses updated method " +
654          "signature. Skipping legacy support for invocations in '" + clazz +"'.");
655      useLegacy = false;
656    } catch (NoSuchMethodException exception) {
657      useLegacy = true;
658    } catch (SecurityException exception) {
659      LOG.warn("The Security Manager denied our attempt to detect if the coprocessor '" + clazz +
660          "' requires legacy support; assuming it does. If you get later errors about legacy " +
661          "coprocessor use, consider updating your security policy to allow access to the package" +
662          " and declared members of your implementation.");
663      LOG.debug("Details of Security Manager rejection.", exception);
664      useLegacy = true;
665    }
666    return useLegacy;
667  }

669  /**
670   * Used to limit legacy handling to once per Coprocessor class per classloader.
671   */
672  private static final Set<Class<? extends Coprocessor>> legacyWarning =
673      new ConcurrentSkipListSet<>(
674          new Comparator<Class<? extends Coprocessor>>() {
675            @Override
676            public int compare(Class<? extends Coprocessor> c1, Class<? extends Coprocessor> c2) {
677              if (c1.equals(c2)) {
678                return 0;
679              }
680              return c1.getName().compareTo(c2.getName());
681            }
682          });
683
684  /**
685   * limits the amount of logging to once per coprocessor class.
686   * Used in concert with {@link #useLegacyMethod(Class, String, Class[])} when a runtime issue
687   * prevents properly supporting the legacy version of a coprocessor API.
688   * Since coprocessors can be in tight loops this serves to limit the amount of log spam we create.
689   */
690  @InterfaceAudience.Private
691  protected void legacyWarning(final Class<? extends Coprocessor> clazz, final String message) {
692    if(legacyWarning.add(clazz)) {
693      LOG.error("You have a legacy coprocessor loaded and there are events we can't map to the " +
694          " deprecated API. Your coprocessor will not see these events.  Please update '" + clazz +
695          "'. Details of the problem: " + message);
696    }
697  }
