020package org.apache.hadoop.hbase.coprocessor;
021
022import org.apache.hadoop.fs.Path;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.classification.InterfaceStability;
025import org.apache.hadoop.hbase.Coprocessor;
026import org.apache.hadoop.hbase.HBaseInterfaceAudience;
027import org.apache.hadoop.hbase.HRegionInfo;
028import org.apache.hadoop.hbase.regionserver.wal.HLogKey;
029import org.apache.hadoop.hbase.wal.WALKey;
030import org.apache.hadoop.hbase.regionserver.wal.WALEdit;
031
032import java.io.IOException;
033
034/**
035 * It's provided to have a way for coprocessors to observe, rewrite,
036 * or skip WALEdits as they are being written to the WAL.
037 *
038 * Note that implementers of WALObserver will not see WALEdits that report themselves
039 * as empty via {@link WALEdit#isEmpty()}.
040 *
041 * {@link org.apache.hadoop.hbase.coprocessor.RegionObserver} provides
042 * hooks for adding logic for WALEdits in the region context during reconstruction,
043 *
044 * Defines coprocessor hooks for interacting with operations on the
045 * {@link org.apache.hadoop.hbase.wal.WAL}.
046 */
047@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
048@InterfaceStability.Evolving
049public interface WALObserver extends Coprocessor {
050
051  /**
052   * Called before a {@link org.apache.hadoop.hbase.regionserver.wal.WALEdit}
053   * is writen to WAL.
054   *
055   * @return true if default behavior should be bypassed, false otherwise
056   */
057  // TODO: return value is not used
058  boolean preWALWrite(ObserverContext<? extends WALCoprocessorEnvironment> ctx,
059      HRegionInfo info, WALKey logKey, WALEdit logEdit) throws IOException;
060
061  /**
062   * Called before a {@link org.apache.hadoop.hbase.regionserver.wal.WALEdit}
063   * is writen to WAL.
064   *
065   * This method is left in place to maintain binary compatibility with older
066   * {@link WALObserver}s. If an implementation directly overrides
067   * {@link #preWALWrite(ObserverContext, HRegionInfo, WALKey, WALEdit)} then this version
068   * won't be called at all, barring problems with the Security Manager. To work correctly
069   * in the presence of a strict Security Manager, or in the case of an implementation that
070   * relies on a parent class to implement preWALWrite, you should implement this method
071   * as a call to the non-deprecated version.
072   *
073   * Users of this method will see all edits that can be treated as HLogKey. If there are
074   * edits that can't be treated as HLogKey they won't be offered to coprocessors that rely
075   * on this method. If a coprocessor gets skipped because of this mechanism, a log message
076   * at ERROR will be generated per coprocessor on the logger for {@link CoprocessorHost} once per
077   * classloader.
078   *
079   * @return true if default behavior should be bypassed, false otherwise
080   * @deprecated use {@link #preWALWrite(ObserverContext, HRegionInfo, WALKey, WALEdit)}
081   */
082  @Deprecated
083  boolean preWALWrite(ObserverContext<WALCoprocessorEnvironment> ctx,
084      HRegionInfo info, HLogKey logKey, WALEdit logEdit) throws IOException;
085
086  /**
087   * Called after a {@link org.apache.hadoop.hbase.regionserver.wal.WALEdit}
088   * is writen to WAL.
089   */
090  void postWALWrite(ObserverContext<? extends WALCoprocessorEnvironment> ctx,
091      HRegionInfo info, WALKey logKey, WALEdit logEdit) throws IOException;
092
093  /**
094   * Called after a {@link org.apache.hadoop.hbase.regionserver.wal.WALEdit}
095   * is writen to WAL.
096   *
097   * This method is left in place to maintain binary compatibility with older
098   * {@link WALObserver}s. If an implementation directly overrides
099   * {@link #postWALWrite(ObserverContext, HRegionInfo, WALKey, WALEdit)} then this version
100   * won't be called at all, barring problems with the Security Manager. To work correctly
101   * in the presence of a strict Security Manager, or in the case of an implementation that
102   * relies on a parent class to implement preWALWrite, you should implement this method
103   * as a call to the non-deprecated version.
104   *
105   * Users of this method will see all edits that can be treated as HLogKey. If there are
106   * edits that can't be treated as HLogKey they won't be offered to coprocessors that rely
107   * on this method. If a coprocessor gets skipped because of this mechanism, a log message
108   * at ERROR will be generated per coprocessor on the logger for {@link CoprocessorHost} once per
109   * classloader.
110   *
111   * @deprecated use {@link #postWALWrite(ObserverContext, HRegionInfo, WALKey, WALEdit)}
112   */
113  @Deprecated
114  void postWALWrite(ObserverContext<WALCoprocessorEnvironment> ctx,
115      HRegionInfo info, HLogKey logKey, WALEdit logEdit) throws IOException;
116
117  /**
118   * Called before rolling the current WAL
119   * @param oldPath the path of the current wal that we are replacing
120   * @param newPath the path of the wal we are going to create
121   */
122  void preWALRoll(ObserverContext<? extends WALCoprocessorEnvironment> ctx,
123      Path oldPath, Path newPath) throws IOException;
124
125  /**
126   * Called after rolling the current WAL
127   * @param oldPath the path of the wal that we replaced
128   * @param newPath the path of the wal we have created and now is the current
129   */
130  void postWALRoll(ObserverContext<? extends WALCoprocessorEnvironment> ctx,
131      Path oldPath, Path newPath) throws IOException;
132}
