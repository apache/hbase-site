19  package org.apache.hadoop.metrics2.lib;
20  
21  import org.apache.commons.lang.StringUtils;
22  import org.apache.hadoop.hbase.classification.InterfaceAudience;
23  import org.apache.hadoop.hbase.util.Counter;
24  import org.apache.hadoop.hbase.util.FastLongHistogram;
25  import org.apache.hadoop.metrics2.MetricHistogram;
26  import org.apache.hadoop.metrics2.MetricsInfo;
27  import org.apache.hadoop.metrics2.MetricsRecordBuilder;
28  
29  /**
30   * A histogram implementation that runs in constant space, and exports to hadoop2's metrics2 system.
31   */
32  @InterfaceAudience.Private
33  public class MutableHistogram extends MutableMetric implements MetricHistogram {
34    // Double buffer the two FastLongHistograms.
35    // As they are reset they learn how the buckets should be spaced
36    // So keep two around and use them
37    protected final FastLongHistogram histogram;
38  
39    protected final String name;
40    protected final String desc;
41    protected final Counter counter = new Counter(0);
42  
43    public MutableHistogram(MetricsInfo info) {
44      this(info.name(), info.description());
45    }
46  
47    public MutableHistogram(String name, String description) {
48      this(name, description, Integer.MAX_VALUE << 2);
49    }
50  
51    protected MutableHistogram(String name, String description, long maxExpected) {
52      this.name = StringUtils.capitalize(name);
53      this.desc = StringUtils.uncapitalize(description);
54      this.histogram = new FastLongHistogram(FastLongHistogram.DEFAULT_NBINS, 1, maxExpected);
55    }
56  
57    public void add(final long val) {
58      counter.increment();
59      histogram.add(val, 1);
60    }
61  
62    @Override
63    public synchronized void snapshot(MetricsRecordBuilder metricsRecordBuilder, boolean all) {
64      // Get a reference to the old histogram.
65      FastLongHistogram histo = histogram.reset();
66      if (histo != null) {
67        updateSnapshotMetrics(metricsRecordBuilder, histo);
68      }
69    }
70  
71    protected void updateSnapshotMetrics(MetricsRecordBuilder metricsRecordBuilder,
72                                         FastLongHistogram histo) {
73      metricsRecordBuilder.addCounter(Interns.info(name + NUM_OPS_METRIC_NAME, desc), counter.get());
74      metricsRecordBuilder.addGauge(Interns.info(name + MIN_METRIC_NAME, desc), histo.getMin());
75      metricsRecordBuilder.addGauge(Interns.info(name + MAX_METRIC_NAME, desc), histo.getMax());
76      metricsRecordBuilder.addGauge(Interns.info(name + MEAN_METRIC_NAME, desc), histo.getMean());
77  
78      long[] percentiles = histo.getQuantiles();
79  
80      metricsRecordBuilder.addGauge(Interns.info(name + TWENTY_FIFTH_PERCENTILE_METRIC_NAME, desc),
81          percentiles[0]);
82      metricsRecordBuilder.addGauge(Interns.info(name + MEDIAN_METRIC_NAME, desc),
83          percentiles[1]);
84      metricsRecordBuilder.addGauge(Interns.info(name + SEVENTY_FIFTH_PERCENTILE_METRIC_NAME, desc),
85          percentiles[2]);
86      metricsRecordBuilder.addGauge(Interns.info(name + NINETIETH_PERCENTILE_METRIC_NAME, desc),
87          percentiles[3]);
88      metricsRecordBuilder.addGauge(Interns.info(name + NINETY_FIFTH_PERCENTILE_METRIC_NAME, desc),
89          percentiles[4]);
90      metricsRecordBuilder.addGauge(Interns.info(name + NINETY_EIGHTH_PERCENTILE_METRIC_NAME, desc),
91          percentiles[5]);
92      metricsRecordBuilder.addGauge(Interns.info(name + NINETY_NINETH_PERCENTILE_METRIC_NAME, desc),
93          percentiles[6]);
94      metricsRecordBuilder.addGauge(
95          Interns.info(name + NINETY_NINE_POINT_NINETH_PERCENTILE_METRIC_NAME, desc),
96          percentiles[7]);
97    }
98  }
