19  package org.apache.hadoop.metrics2.lib;
20  
21  import org.apache.commons.lang.StringUtils;
22  import org.apache.hadoop.hbase.classification.InterfaceAudience;
23  import org.apache.hadoop.hbase.metrics.Interns;
24  import org.apache.hadoop.hbase.util.Counter;
25  import org.apache.hadoop.hbase.util.FastLongHistogram;
26  import org.apache.hadoop.metrics2.MetricHistogram;
27  import org.apache.hadoop.metrics2.MetricsInfo;
28  import org.apache.hadoop.metrics2.MetricsRecordBuilder;
29
30  /**
31   * A histogram implementation that runs in constant space, and exports to hadoop2's metrics2 system.
32   */
33  @InterfaceAudience.Private
34  public class MutableHistogram extends MutableMetric implements MetricHistogram {
35    // Double buffer the two FastLongHistograms.
36    // As they are reset they learn how the buckets should be spaced
37    // So keep two around and use them
38    protected final FastLongHistogram histogram;
39
40    protected final String name;
41    protected final String desc;
42    protected final Counter counter = new Counter(0);
43
44    private boolean metricsInfoStringInited = false;
45    private String NUM_OPS_METRIC;
46    private String MIN_METRIC;
47    private String MAX_METRIC;
48    private String MEAN_METRIC;
49    private String MEDIAN_METRIC;
50    private String TWENTY_FIFTH_PERCENTILE_METRIC;
51    private String SEVENTY_FIFTH_PERCENTILE_METRIC;
52    private String NINETIETH_PERCENTILE_METRIC;
53    private String NINETY_FIFTH_PERCENTILE_METRIC;
54    private String NINETY_EIGHTH_PERCENTILE_METRIC;
55    private String NINETY_NINETH_PERCENTILE_METRIC;
56    private String NINETY_NINE_POINT_NINETH_PERCENTILE_METRIC;
57
58    public MutableHistogram(MetricsInfo info) {
59      this(info.name(), info.description());
60    }
61  
62    public MutableHistogram(String name, String description) {
63      this(name, description, Integer.MAX_VALUE << 2);
64    }
65
66    protected MutableHistogram(String name, String description, long maxExpected) {
67      this.name = StringUtils.capitalize(name);
68      this.desc = StringUtils.uncapitalize(description);
69      this.histogram = new FastLongHistogram(FastLongHistogram.DEFAULT_NBINS, 1, maxExpected);
70    }
71
72    public void add(final long val) {
73      counter.increment();
74      histogram.add(val, 1);
75    }
76
77    @Override
78    public synchronized void snapshot(MetricsRecordBuilder metricsRecordBuilder, boolean all) {
79      // Get a reference to the old histogram.
80      FastLongHistogram histo = histogram.reset();
81      if (histo != null) {
82        updateSnapshotMetrics(metricsRecordBuilder, histo);
83      }
84    }
85
86    protected void updateSnapshotMetrics(MetricsRecordBuilder metricsRecordBuilder,
87                                         FastLongHistogram histo) {
88      if (!metricsInfoStringInited) {
89        NUM_OPS_METRIC = name + NUM_OPS_METRIC_NAME;
90        MIN_METRIC = name + MIN_METRIC_NAME;
91        MAX_METRIC = name + MAX_METRIC_NAME;
92        MEAN_METRIC = name + MEAN_METRIC_NAME;
93        MEDIAN_METRIC = name + MEDIAN_METRIC_NAME;
94        TWENTY_FIFTH_PERCENTILE_METRIC = name + TWENTY_FIFTH_PERCENTILE_METRIC_NAME;
95        SEVENTY_FIFTH_PERCENTILE_METRIC = name + SEVENTY_FIFTH_PERCENTILE_METRIC_NAME;
96        NINETIETH_PERCENTILE_METRIC = name + NINETIETH_PERCENTILE_METRIC_NAME;
97        NINETY_FIFTH_PERCENTILE_METRIC = name + NINETY_FIFTH_PERCENTILE_METRIC_NAME;
98        NINETY_EIGHTH_PERCENTILE_METRIC = name + NINETY_EIGHTH_PERCENTILE_METRIC_NAME;
99        NINETY_NINETH_PERCENTILE_METRIC = name + NINETY_NINETH_PERCENTILE_METRIC_NAME;
100       NINETY_NINE_POINT_NINETH_PERCENTILE_METRIC = name +
101           NINETY_NINE_POINT_NINETH_PERCENTILE_METRIC_NAME;
102
103       metricsInfoStringInited = true;
104     }
105
106     metricsRecordBuilder.addCounter(Interns.info(NUM_OPS_METRIC, desc), counter.get());
107     metricsRecordBuilder.addGauge(Interns.info(MIN_METRIC, desc), histo.getMin());
108     metricsRecordBuilder.addGauge(Interns.info(MAX_METRIC, desc), histo.getMax());
109     metricsRecordBuilder.addGauge(Interns.info(MEAN_METRIC, desc), histo.getMean());
110
111     long[] percentiles = histo.getQuantiles();
112
113     metricsRecordBuilder.addGauge(Interns.info(TWENTY_FIFTH_PERCENTILE_METRIC, desc),
114         percentiles[0]);
115     metricsRecordBuilder.addGauge(Interns.info(MEDIAN_METRIC, desc),
116         percentiles[1]);
117     metricsRecordBuilder.addGauge(Interns.info(SEVENTY_FIFTH_PERCENTILE_METRIC, desc),
118         percentiles[2]);
119     metricsRecordBuilder.addGauge(Interns.info(NINETIETH_PERCENTILE_METRIC, desc),
120         percentiles[3]);
121     metricsRecordBuilder.addGauge(Interns.info(NINETY_FIFTH_PERCENTILE_METRIC, desc),
122         percentiles[4]);
123     metricsRecordBuilder.addGauge(Interns.info(NINETY_EIGHTH_PERCENTILE_METRIC, desc),
124         percentiles[5]);
125     metricsRecordBuilder.addGauge(Interns.info(NINETY_NINETH_PERCENTILE_METRIC, desc),
126         percentiles[6]);
127     metricsRecordBuilder.addGauge(
128         Interns.info(NINETY_NINE_POINT_NINETH_PERCENTILE_METRIC, desc),
129         percentiles[7]);
130   }
131 }
