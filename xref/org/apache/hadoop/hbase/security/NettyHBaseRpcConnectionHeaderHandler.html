19  package org.apache.hadoop.hbase.security;
20
21  import io.netty.buffer.ByteBuf;
22  import io.netty.channel.ChannelHandlerContext;
23  import io.netty.channel.ChannelPipeline;
24  import io.netty.channel.SimpleChannelInboundHandler;
25  import io.netty.handler.codec.LengthFieldBasedFrameDecoder;
26  import io.netty.util.concurrent.Promise;
27  import org.apache.hadoop.conf.Configuration;
28  import org.apache.hadoop.hbase.classification.InterfaceAudience;
29  import org.apache.hadoop.hbase.io.crypto.aes.CryptoAES;
30  import org.apache.hadoop.hbase.shaded.protobuf.generated.RPCProtos;
31
32  /**
33   * Implement logic to deal with the rpc connection header.
34   */
35  @InterfaceAudience.Private
36  public class NettyHBaseRpcConnectionHeaderHandler extends SimpleChannelInboundHandler<ByteBuf> {
37
38    private final Promise<Boolean> saslPromise;
39
40    private final Configuration conf;
41
42    private final ByteBuf connectionHeaderWithLength;
43
44    public NettyHBaseRpcConnectionHeaderHandler(Promise<Boolean> saslPromise, Configuration conf,
45                                                ByteBuf connectionHeaderWithLength) {
46      this.saslPromise = saslPromise;
47      this.conf = conf;
48      this.connectionHeaderWithLength = connectionHeaderWithLength;
49    }
50
51    @Override
52    protected void channelRead0(ChannelHandlerContext ctx, ByteBuf msg) throws Exception {
53      // read the ConnectionHeaderResponse from server
54      int len = msg.readInt();
55      byte[] buff = new byte[len];
56      msg.readBytes(buff);
57
58      RPCProtos.ConnectionHeaderResponse connectionHeaderResponse =
59          RPCProtos.ConnectionHeaderResponse.parseFrom(buff);
60
61      // Get the CryptoCipherMeta, update the HBaseSaslRpcClient for Crypto Cipher
62      if (connectionHeaderResponse.hasCryptoCipherMeta()) {
63        CryptoAES cryptoAES = EncryptionUtil.createCryptoAES(
64            connectionHeaderResponse.getCryptoCipherMeta(), conf);
65        // replace the Sasl handler with Crypto AES handler
66        setupCryptoAESHandler(ctx.pipeline(), cryptoAES);
67      }
68
69      saslPromise.setSuccess(true);
70    }
71
72    @Override
73    public void handlerAdded(ChannelHandlerContext ctx) {
74      try {
75        // send the connection header to server first
76        ctx.writeAndFlush(connectionHeaderWithLength.retainedDuplicate());
77      } catch (Exception e) {
78        // the exception thrown by handlerAdded will not be passed to the exceptionCaught below
79        // because netty will remove a handler if handlerAdded throws an exception.
80        exceptionCaught(ctx, e);
81      }
82    }
83
84    @Override
85    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
86      saslPromise.tryFailure(cause);
87    }
88
89    /**
90     * Remove handlers for sasl encryption and add handlers for Crypto AES encryption
91     */
92    private void setupCryptoAESHandler(ChannelPipeline p, CryptoAES cryptoAES) {
93      p.remove(SaslWrapHandler.class);
94      p.remove(SaslUnwrapHandler.class);
95      String lengthDecoder = p.context(LengthFieldBasedFrameDecoder.class).name();
96      p.addAfter(lengthDecoder, null, new CryptoAESUnwrapHandler(cryptoAES));
97      p.addAfter(lengthDecoder, null, new CryptoAESWrapHandler(cryptoAES));
98    }
99  }
