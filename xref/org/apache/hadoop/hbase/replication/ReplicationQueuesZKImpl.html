43  /**
44   * This class provides an implementation of the ReplicationQueues interface using Zookeeper. The
45   * base znode that this class works at is the myQueuesZnode. The myQueuesZnode contains a list of
46   * all outstanding WAL files on this region server that need to be replicated. The myQueuesZnode is
47   * the regionserver name (a concatenation of the region serverâ€™s hostname, client port and start
48   * code). For example:
49   *
50   * /hbase/replication/rs/hostname.example.org,6020,1234
51   *
52   * Within this znode, the region server maintains a set of WAL replication queues. These queues are
53   * represented by child znodes named using there give queue id. For example:
54   *
55   * /hbase/replication/rs/hostname.example.org,6020,1234/1
56   * /hbase/replication/rs/hostname.example.org,6020,1234/2
57   *
58   * Each queue has one child znode for every WAL that still needs to be replicated. The value of
59   * these WAL child znodes is the latest position that has been replicated. This position is updated
60   * every time a WAL entry is replicated. For example:
61   *
62   * /hbase/replication/rs/hostname.example.org,6020,1234/1/23522342.23422 [VALUE: 254]
63   */
64  @InterfaceAudience.Private
65  public class ReplicationQueuesZKImpl extends ReplicationStateZKBase implements ReplicationQueues {

143   @Override
144   public long getLogPosition(String queueId, String filename) throws ReplicationException {
145     String clusterZnode = ZKUtil.joinZNode(this.myQueuesZnode, queueId);
146     String znode = ZKUtil.joinZNode(clusterZnode, filename);
147     byte[] bytes = null;
148     try {
149       bytes = ZKUtil.getData(this.zookeeper, znode);
150     } catch (KeeperException e) {
151       throw new ReplicationException("Internal Error: could not get position in log for queueId="
152           + queueId + ", filename=" + filename, e);
153     } catch (InterruptedException e) {
154       Thread.currentThread().interrupt();
155       return 0;
156     }
157     try {
158       return ZKUtil.parseWALPositionFrom(bytes);
159     } catch (DeserializationException de) {
160       LOG.warn("Failed to parse WALPosition for queueId=" + queueId + " and wal=" + filename
161           + "znode content, continuing.");
162     }
163     // if we can not parse the position, start at the beginning of the wal file
164     // again
165     return 0;
166   }

173   @Override
174   public SortedMap<String, SortedSet<String>> claimQueues(String regionserverZnode) {
175     SortedMap<String, SortedSet<String>> newQueues = new TreeMap<String, SortedSet<String>>();
176     // check whether there is multi support. If yes, use it.
177     if (conf.getBoolean(HConstants.ZOOKEEPER_USEMULTI, true)) {
178       LOG.info("Atomically moving " + regionserverZnode + "'s WALs to my queue");
179       newQueues = copyQueuesFromRSUsingMulti(regionserverZnode);
180     } else {
181       LOG.info("Moving " + regionserverZnode + "'s wals to my queue");
182       if (!lockOtherRS(regionserverZnode)) {
183         return newQueues;
184       }
185       newQueues = copyQueuesFromRS(regionserverZnode);
186       deleteAnotherRSQueues(regionserverZnode);
187     }
188     return newQueues;
189   }

229   /**
230    * Try to set a lock in another region server's znode.
231    * @param znode the server names of the other server
232    * @return true if the lock was acquired, false in every other cases
233    */
234   private boolean lockOtherRS(String znode) {
235     try {
236       String parent = ZKUtil.joinZNode(this.queuesZNode, znode);
237       if (parent.equals(this.myQueuesZnode)) {
238         LOG.warn("Won't lock because this is us, we're dead!");
239         return false;
240       }
241       String p = ZKUtil.joinZNode(parent, RS_LOCK_ZNODE);
242       ZKUtil.createAndWatch(this.zookeeper, p, lockToByteArray(this.myQueuesZnode));
243     } catch (KeeperException e) {
244       // This exception will pop up if the znode under which we're trying to
245       // create the lock is already deleted by another region server, meaning
246       // that the transfer already occurred.
247       // NoNode => transfer is done and znodes are already deleted
248       // NodeExists => lock znode already created by another RS
249       if (e instanceof KeeperException.NoNodeException
250           || e instanceof KeeperException.NodeExistsException) {
251         LOG.info("Won't transfer the queue," + " another RS took care of it because of: "
252             + e.getMessage());
253       } else {
254         LOG.info("Failed lock other rs", e);
255       }
256       return false;
257     }
258     return true;
259   }

261   /**
262    * Delete all the replication queues for a given region server.
263    * @param regionserverZnode The znode of the region server to delete.
264    */
265   private void deleteAnotherRSQueues(String regionserverZnode) {
266     String fullpath = ZKUtil.joinZNode(this.queuesZNode, regionserverZnode);
267     try {
268       List<String> clusters = ZKUtil.listChildrenNoWatch(this.zookeeper, fullpath);
269       for (String cluster : clusters) {
270         // No need to delete, it will be deleted later.
271         if (cluster.equals(RS_LOCK_ZNODE)) {
272           continue;
273         }
274         String fullClusterPath = ZKUtil.joinZNode(fullpath, cluster);
275         ZKUtil.deleteNodeRecursively(this.zookeeper, fullClusterPath);
276       }
277       // Finish cleaning up
278       ZKUtil.deleteNodeRecursively(this.zookeeper, fullpath);
279     } catch (KeeperException e) {
280       if (e instanceof KeeperException.NoNodeException
281           || e instanceof KeeperException.NotEmptyException) {
282         // Testing a special case where another region server was able to
283         // create a lock just after we deleted it, but then was also able to
284         // delete the RS znode before us or its lock znode is still there.
285         if (e.getPath().equals(fullpath)) {
286           return;
287         }
288       }
289       this.abortable.abort("Failed to delete replication queues for region server: "
290           + regionserverZnode, e);
291     }
292   }

294   /**
295    * It "atomically" copies all the wals queues from another region server and returns them all
296    * sorted per peer cluster (appended with the dead server's znode).
297    * @param znode pertaining to the region server to copy the queues from
298    * @return WAL queues sorted per peer cluster
299    */
300   private SortedMap<String, SortedSet<String>> copyQueuesFromRSUsingMulti(String znode) {
301     SortedMap<String, SortedSet<String>> queues = new TreeMap<String, SortedSet<String>>();
302     // hbase/replication/rs/deadrs
303     String deadRSZnodePath = ZKUtil.joinZNode(this.queuesZNode, znode);
304     List<String> peerIdsToProcess = null;
305     List<ZKUtilOp> listOfOps = new ArrayList<ZKUtil.ZKUtilOp>();
306     try {
307       peerIdsToProcess = ZKUtil.listChildrenNoWatch(this.zookeeper, deadRSZnodePath);
308       if (peerIdsToProcess == null) return queues; // node already processed
309       for (String peerId : peerIdsToProcess) {
310         ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(peerId);
311         if (!peerExists(replicationQueueInfo.getPeerId())) {
312           LOG.warn("Peer " + peerId + " didn't exist, skipping the replay");
313           // Protection against moving orphaned queues
314           continue;
315         }
316         String newPeerId = peerId + "-" + znode;
317         String newPeerZnode = ZKUtil.joinZNode(this.myQueuesZnode, newPeerId);
318         // check the logs queue for the old peer cluster
319         String oldClusterZnode = ZKUtil.joinZNode(deadRSZnodePath, peerId);
320         List<String> wals = ZKUtil.listChildrenNoWatch(this.zookeeper, oldClusterZnode);
321         if (wals == null || wals.size() == 0) {
322           listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldClusterZnode));
323           continue; // empty log queue.
324         }
325         // create the new cluster znode
326         SortedSet<String> logQueue = new TreeSet<String>();
327         queues.put(newPeerId, logQueue);
328         ZKUtilOp op = ZKUtilOp.createAndFailSilent(newPeerZnode, HConstants.EMPTY_BYTE_ARRAY);
329         listOfOps.add(op);
330         // get the offset of the logs and set it to new znodes
331         for (String wal : wals) {
332           String oldWalZnode = ZKUtil.joinZNode(oldClusterZnode, wal);
333           byte[] logOffset = ZKUtil.getData(this.zookeeper, oldWalZnode);
334           LOG.debug("Creating " + wal + " with data " + Bytes.toString(logOffset));
335           String newLogZnode = ZKUtil.joinZNode(newPeerZnode, wal);
336           listOfOps.add(ZKUtilOp.createAndFailSilent(newLogZnode, logOffset));
337           // add ops for deleting
338           listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldWalZnode));
339           logQueue.add(wal);
340         }
341         // add delete op for peer
342         listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldClusterZnode));
343       }
344       // add delete op for dead rs, this will update the cversion of the parent.
345       // The reader will make optimistic locking with this to get a consistent
346       // snapshot
347       listOfOps.add(ZKUtilOp.deleteNodeFailSilent(deadRSZnodePath));
348       if (LOG.isTraceEnabled()) LOG.trace(" The multi list size is: " + listOfOps.size());
349       ZKUtil.multiOrSequential(this.zookeeper, listOfOps, false);
350       if (LOG.isTraceEnabled()) LOG.trace("Atomically moved the dead regionserver logs. ");
351     } catch (KeeperException e) {
352       // Multi call failed; it looks like some other regionserver took away the logs.
353       LOG.warn("Got exception in copyQueuesFromRSUsingMulti: ", e);
354       queues.clear();
355     } catch (InterruptedException e) {
356       LOG.warn("Got exception in copyQueuesFromRSUsingMulti: ", e);
357       queues.clear();
358       Thread.currentThread().interrupt();
359     }
360     return queues;
361   }

363   /**
364    * This methods copies all the wals queues from another region server and returns them all sorted
365    * per peer cluster (appended with the dead server's znode)
366    * @param znode server names to copy
367    * @return all wals for all peers of that cluster, null if an error occurred
368    */
369   private SortedMap<String, SortedSet<String>> copyQueuesFromRS(String znode) {
370     // TODO this method isn't atomic enough, we could start copying and then
371     // TODO fail for some reason and we would end up with znodes we don't want.
372     SortedMap<String, SortedSet<String>> queues = new TreeMap<String, SortedSet<String>>();
373     try {
374       String nodePath = ZKUtil.joinZNode(this.queuesZNode, znode);
375       List<String> clusters = ZKUtil.listChildrenNoWatch(this.zookeeper, nodePath);
376       // We have a lock znode in there, it will count as one.
377       if (clusters == null || clusters.size() <= 1) {
378         return queues;
379       }
380       // The lock isn't a peer cluster, remove it
381       clusters.remove(RS_LOCK_ZNODE);
382       for (String cluster : clusters) {
383         ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(cluster);
384         if (!peerExists(replicationQueueInfo.getPeerId())) {
385           LOG.warn("Peer " + cluster + " didn't exist, skipping the replay");
386           // Protection against moving orphaned queues
387           continue;
388         }
389         // We add the name of the recovered RS to the new znode, we can even
390         // do that for queues that were recovered 10 times giving a znode like
391         // number-startcode-number-otherstartcode-number-anotherstartcode-etc
392         String newCluster = cluster + "-" + znode;
393         String newClusterZnode = ZKUtil.joinZNode(this.myQueuesZnode, newCluster);
394         String clusterPath = ZKUtil.joinZNode(nodePath, cluster);
395         List<String> wals = ZKUtil.listChildrenNoWatch(this.zookeeper, clusterPath);
396         // That region server didn't have anything to replicate for this cluster
397         if (wals == null || wals.size() == 0) {
398           continue;
399         }
400         ZKUtil.createNodeIfNotExistsAndWatch(this.zookeeper, newClusterZnode,
401           HConstants.EMPTY_BYTE_ARRAY);
402         SortedSet<String> logQueue = new TreeSet<String>();
403         queues.put(newCluster, logQueue);
404         for (String wal : wals) {
405           String z = ZKUtil.joinZNode(clusterPath, wal);
406           byte[] positionBytes = ZKUtil.getData(this.zookeeper, z);
407           long position = 0;
408           try {
409             position = ZKUtil.parseWALPositionFrom(positionBytes);
410           } catch (DeserializationException e) {
411             LOG.warn("Failed parse of wal position from the following znode: " + z
412                 + ", Exception: " + e);
413           }
414           LOG.debug("Creating " + wal + " with data " + position);
415           String child = ZKUtil.joinZNode(newClusterZnode, wal);
416           // Position doesn't actually change, we are just deserializing it for
417           // logging, so just use the already serialized version
418           ZKUtil.createAndWatch(this.zookeeper, child, positionBytes);
419           logQueue.add(wal);
420         }
421       }
422     } catch (KeeperException e) {
423       this.abortable.abort("Copy queues from rs", e);
424     } catch (InterruptedException e) {
425       LOG.warn(e);
426       Thread.currentThread().interrupt();
427     }
428     return queues;
429   }

431   /**
432    * @param lockOwner
433    * @return Serialized protobuf of <code>lockOwner</code> with pb magic prefix prepended suitable
434    *         for use as content of an replication lock during region server fail over.
435    */
436   static byte[] lockToByteArray(final String lockOwner) {
437     byte[] bytes =
438         ZooKeeperProtos.ReplicationLock.newBuilder().setLockOwner(lockOwner).build().toByteArray();
439     return ProtobufUtil.prependPBMagic(bytes);
440   }

442   @Override
443   public void addHFileRefs(String peerId, List<String> files) throws ReplicationException {
444     String peerZnode = ZKUtil.joinZNode(this.hfileRefsZNode, peerId);
445     boolean debugEnabled = LOG.isDebugEnabled();
446     if (debugEnabled) {
447       LOG.debug("Adding hfile references " + files + " in queue " + peerZnode);
448     }
449     List<ZKUtilOp> listOfOps = new ArrayList<ZKUtil.ZKUtilOp>();
450     int size = files.size();
451     for (int i = 0; i < size; i++) {
452       listOfOps.add(ZKUtilOp.createAndFailSilent(ZKUtil.joinZNode(peerZnode, files.get(i)),
453         HConstants.EMPTY_BYTE_ARRAY));
454     }
455     if (debugEnabled) {
456       LOG.debug(" The multi list size for adding hfile references in zk for node " + peerZnode
457           + " is " + listOfOps.size());
458     }
459     try {
460       ZKUtil.multiOrSequential(this.zookeeper, listOfOps, true);
461     } catch (KeeperException e) {
462       throw new ReplicationException("Failed to create hfile reference znode=" + e.getPath(), e);
463     }
464   }

477     }
478     if (debugEnabled) {
479       LOG.debug(" The multi list size for removing hfile references in zk for node " + peerZnode
480           + " is " + listOfOps.size());
481     }
482     try {
483       ZKUtil.multiOrSequential(this.zookeeper, listOfOps, true);
484     } catch (KeeperException e) {
485       LOG.error("Failed to remove hfile reference znode=" + e.getPath(), e);
