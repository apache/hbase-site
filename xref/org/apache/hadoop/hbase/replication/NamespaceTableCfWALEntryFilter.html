37  /**
38   * Filter a WAL Entry by namespaces and table-cfs config in the peer. It first filter entry
39   * by namespaces config, then filter entry by table-cfs config.
40   *
41   * 1. Set a namespace in peer config means that all tables in this namespace will be replicated.
42   * 2. If the namespaces config is null, then the table-cfs config decide which table's edit
43   *    can be replicated. If the table-cfs config is null, then the namespaces config decide
44   *    which table's edit can be replicated.
45   */
46  @InterfaceAudience.Private
47  public class NamespaceTableCfWALEntryFilter implements WALEntryFilter, WALCellFilter {
48
49    private static final Log LOG = LogFactory.getLog(NamespaceTableCfWALEntryFilter.class);
50    private final ReplicationPeer peer;
51    private BulkLoadCellFilter bulkLoadFilter = new BulkLoadCellFilter();
52
53    public NamespaceTableCfWALEntryFilter(ReplicationPeer peer) {
54      this.peer = peer;
55    }
56
57    @Override
58    public Entry filter(Entry entry) {
59      TableName tabName = entry.getKey().getTablename();
60      String namespace = tabName.getNamespaceAsString();
61      Set<String> namespaces = this.peer.getNamespaces();
62      Map<TableName, List<String>> tableCFs = getTableCfs();
63
64      // If null means user has explicitly not configured any namespaces and table CFs
65      // so all the tables data are applicable for replication
66      if (namespaces == null && tableCFs == null) {
67        return entry;
68      }
69
70      // First filter by namespaces config
71      // If table's namespace in peer config, all the tables data are applicable for replication
72      if (namespaces != null && namespaces.contains(namespace)) {
73        return entry;
74      }
75
76      // Then filter by table-cfs config
77      // return null(prevent replicating) if logKey's table isn't in this peer's
78      // replicaable namespace list and table list
79      if (tableCFs == null || !tableCFs.containsKey(tabName)) {
80        return null;
81      }
82
83      return entry;
84    }
85
86    @Override
87    public Cell filterCell(final Entry entry, Cell cell) {
88      final Map<TableName, List<String>> tableCfs = getTableCfs();
89      if (tableCfs == null) return cell;
90      TableName tabName = entry.getKey().getTablename();
91      List<String> cfs = tableCfs.get(tabName);
92      // ignore(remove) kv if its cf isn't in the replicable cf list
93      // (empty cfs means all cfs of this table are replicable)
94      if (CellUtil.matchingColumn(cell, WALEdit.METAFAMILY, WALEdit.BULK_LOAD)) {
95        cell = bulkLoadFilter.filterCell(cell, new Predicate<byte[]>() {
96          @Override
97          public boolean apply(byte[] fam) {
98            if (tableCfs != null) {
99              List<String> cfs = tableCfs.get(entry.getKey().getTablename());
100             if (cfs != null && !cfs.contains(Bytes.toString(fam))) {
101               return true;
102             }
103           }
104           return false;
105         }
106       });
107     } else {
108       if ((cfs != null) && !cfs.contains(
109         Bytes.toString(cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength()))) {
110         return null;
111       }
112     }
113     return cell;
114   }
115
116   Map<TableName, List<String>> getTableCfs() {
117     Map<TableName, List<String>> tableCFs = null;
118     try {
119       tableCFs = this.peer.getTableCFs();
120     } catch (IllegalArgumentException e) {
121       LOG.error("should not happen: can't get tableCFs for peer " + peer.getId() +
122           ", degenerate as if it's not configured by keeping tableCFs==null");
123     }
124     return tableCFs;
125   }
