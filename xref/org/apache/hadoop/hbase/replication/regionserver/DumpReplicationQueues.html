46  /**
47   * Provides information about the existing states of replication, replication peers and queues.
48   *
49   * Usage: hbase org.apache.hadoop.hbase.replication.regionserver.DumpReplicationQueues [args]

59      boolean hdfs = false;

66        this.hdfs = that.hdfs;

75        return distributed;

83        this.distributed = distributed;

87    static DumpOptions parseOpts(Queue<String> args) {
88      DumpOptions opts = new DumpOptions();
89
90      String cmd = null;
91      while ((cmd = args.poll()) != null) {
92        if (cmd.equals("-h") || cmd.equals("--h") || cmd.equals("--help")) {
93          // place item back onto queue so that caller knows parsing was incomplete
94          args.add(cmd);
95          break;
96        }
97        final String hdfs = "--hdfs";
98        if (cmd.equals(hdfs)) {
99          opts.setHdfs(true);
100         continue;
101       }
102       final String distributed = "--distributed";
103       if (cmd.equals(distributed)) {
104         opts.setDistributed(true);
105         continue;
106       } else {
107         printUsageAndExit("ERROR: Unrecognized option/command: " + cmd, -1);
108       }
109       // check that --distributed is present when --hdfs is in the arguments
110       if (!opts.isDistributed()  && opts.isHdfs()) {
111         printUsageAndExit("ERROR: --hdfs option can only be used with --distributed: " + cmd, -1);
112       }
113     }
114     return opts;
115   }

129   @Override
130   public int run(String[] args) throws Exception {
131
132     int errCode = -1;
133     LinkedList<String> argv = new LinkedList<String>();
134     argv.addAll(Arrays.asList(args));
135     DumpOptions opts = parseOpts(argv);

173   private int dumpReplicationQueues(DumpOptions opts) throws Exception {
174
175     Configuration conf = getConf();
176     HBaseAdmin.available(conf);
177     ReplicationAdmin replicationAdmin = new ReplicationAdmin(conf);
178     ClusterConnection connection = (ClusterConnection) ConnectionFactory.createConnection(conf);
179     long deleted = 0;
180
181     ZooKeeperWatcher zkw = new ZooKeeperWatcher(conf, "DumpReplicationQueues" + System.currentTimeMillis(),
182         new WarnOnlyAbortable(), true);
183
184     try {
185       // Our zk watcher
186       LOG.info("Our Quorum: " + zkw.getQuorum());
187       List<HashMap<String, String>> replicatedTables = replicationAdmin.listReplicated();
188       if (replicatedTables.isEmpty()) {
189         LOG.info("No tables with a configured replication peer were found.");
190         return(0);
191       } else {
192         LOG.info("Replicated Tables: " + replicatedTables);
193       }
194
195       Map<String, ReplicationPeerConfig> peerConfigs = replicationAdmin.listPeerConfigs();
196
197       if (peerConfigs.isEmpty()) {
198         LOG.info("Replication is enabled but no peer configuration was found.");
199       }
200
201       System.out.println("Dumping replication peers and configurations:");
202       System.out.println(dumpPeersState(replicationAdmin, peerConfigs));
203
204       if (opts.isDistributed()) {
205         LOG.info("Found [--distributed], will poll each RegionServer.");
206         System.out.println(dumpQueues(connection, zkw, opts.isHdfs(), deleted));
207         if (deleted > 0) {
208           LOG.warn("Found " + deleted +" deleted queues"
209               + ", run hbck -fixReplication in order to remove the deleted replication queues");
210         }
211       } else {
212         // use ZK instead
213         System.out.print("Dumping replication znodes via ZooKeeper:");
214         System.out.println(ZKUtil.getReplicationZnodesDump(zkw));
215       }
216       return (0);
217     } catch (IOException e) {
218       return (-1);
219     } finally {
220       zkw.close();
221     }
222   }

224   public String dumpPeersState(ReplicationAdmin replicationAdmin,
225                               Map<String, ReplicationPeerConfig> peerConfigs) throws Exception {
226     Map<String, String> currentConf;
227     StringBuilder sb = new StringBuilder();
228     for (Map.Entry<String, ReplicationPeerConfig> peer : peerConfigs.entrySet()) {
229       try {
230         sb.append("Peer: " + peer.getKey() + "\n");
231         sb.append("    " + "State: " + (replicationAdmin.getPeerState(peer.getKey()) ? "ENABLED" : "DISABLED") + "\n");
232         sb.append("    " + "Cluster Name: " + peer.getValue() + "\n");
233         currentConf = peer.getValue().getConfiguration();
234         // Only show when we have a custom configuration for the peer
235         if (currentConf.size() > 1) {
236           sb.append("    " + "Peer Configuration: " + currentConf + "\n");
237         }
238         sb.append("    " + "Peer Table CFs: " + replicationAdmin.getPeerTableCFs(peer.getKey()) + "\n");
239       } catch (ReplicationException re) {
240         sb.append("Got an exception while invoking ReplicationAdmin: " + re + "\n");
241       }
242     }
243     return sb.toString();
244   }

246   public String dumpQueues(ClusterConnection connection, ZooKeeperWatcher zkw, boolean hdfs, long deleted)
247       throws Exception {
248     ReplicationQueuesClient queuesClient;
249     ReplicationPeers replicationPeers;
250     ReplicationQueues replicationQueues;
251     ReplicationQueuesClientArguments replicationArgs =
252         new ReplicationQueuesClientArguments(getConf(), new WarnOnlyAbortable(), zkw);
253     StringBuilder sb = new StringBuilder();
254
255     queuesClient = ReplicationFactory.getReplicationQueuesClient(replicationArgs);
256     queuesClient.init();
257     replicationQueues = ReplicationFactory.getReplicationQueues(replicationArgs);
258     replicationPeers = ReplicationFactory.getReplicationPeers(zkw, getConf(), queuesClient, connection);
259     replicationPeers.init();
260
261     // Loops each peer on each RS and dumps the queues
262
263     Set<String> peerIds = new HashSet<String>(replicationPeers.getAllPeerIds());
264     try {
265       List<String> regionservers = queuesClient.getListOfReplicators();
266       for (String regionserver : regionservers) {
267         List<String> queueIds = queuesClient.getAllQueues(regionserver);
268         replicationQueues.init(regionserver);
269         for (String queueId : queueIds) {
270           ReplicationQueueInfo queueInfo = new ReplicationQueueInfo(queueId);
271           List<String> wals = queuesClient.getLogsInQueue(regionserver, queueId);
272           if (!peerIds.contains(queueInfo.getPeerId())) {
273             deleted++;
274             sb.append(formatQueue(regionserver, replicationQueues, queueInfo, queueId, wals, true, hdfs));
275           } else {
276             sb.append(formatQueue(regionserver, replicationQueues, queueInfo, queueId, wals, false, hdfs));
277           }
278         }
279       }
280     } catch (KeeperException ke) {
281       throw new IOException(ke);
282     }
283     return sb.toString();
284   }

286   private String formatQueue(String regionserver, ReplicationQueues replicationQueues, ReplicationQueueInfo queueInfo,
287                            String queueId, List<String> wals, boolean isDeleted, boolean hdfs) throws Exception {
288
289     StringBuilder sb = new StringBuilder();
290
291     List<String> deadServers ;
292
293     sb.append("Dumping replication queue info for RegionServer: [" + regionserver + "]" + "\n");
294     sb.append("    PeerID: " + queueInfo.getPeerId() + "\n");
295     sb.append("    Recovered: " + queueInfo.isQueueRecovered() + "\n");
296     deadServers = queueInfo.getDeadRegionServers();
297     if (deadServers.isEmpty()) {
298       sb.append("    No dead RegionServers found in this queue." + "\n");
299     } else {
300       sb.append("    Dead RegionServers: " + deadServers + "\n");
301     }
302     sb.append("    Was deleted: " + isDeleted + "\n");
303     sb.append("    Number of WALs in replication queue: " + wals.size() + "\n");
304     for (String wal : wals) {
305       long position = replicationQueues.getLogPosition(queueInfo.getPeerId(), wal);
306       sb.append("    Replication position for " + wal + ": " + (position > 0 ? position : "0"
307           + " (not started or nothing to replicate)") + "\n");
308     }
309
310     if (hdfs) {
311       FileSystem fs = FileSystem.get(getConf());
312       sb.append("    Total size of WALs on HDFS for this queue: "
313           + StringUtils.humanSize(getTotalWALSize(fs, wals, regionserver)) + "\n");
314     }
315     return sb.toString();
316   }
317   /**
318    *  return total size in bytes from a list of WALs
319    */
320   private long getTotalWALSize(FileSystem fs, List<String> wals, String server) throws IOException {
321     int size = 0;
322     FileStatus fileStatus;
323
324     for (String wal : wals) {
325       try {
326         fileStatus = (new WALLink(getConf(), server, wal)).getFileStatus(fs);
327       } catch (IOException e) {
328         LOG.warn("WAL " + wal + " couldn't be found, skipping");
329         break;
330       }
331       size += fileStatus.getLen();
332     }
333     return size;
334   }
