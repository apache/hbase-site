158   /**
159    * Opens a HFile. You must load the index before you can use it by calling
160    * {@link #loadFileInfo()}.
161    * @param path
162    *          Path to HFile.
163    * @param trailer
164    *          File trailer.
165    * @param fsdis
166    *          input stream.
167    * @param fileSize
168    *          Length of the stream.
169    * @param cacheConf
170    *          Cache configuration.
171    * @param hfs
172    *          The file system.
173    * @param conf
174    *          Configuration
175    */
176   @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="URF_UNREAD_PUBLIC_OR_PROTECTED_FIELD")
177   public HFileReaderImpl(final Path path, FixedFileTrailer trailer,
178       final FSDataInputStreamWrapper fsdis,
179       final long fileSize, final CacheConfig cacheConf, final HFileSystem hfs,
180       final Configuration conf)
181   throws IOException {
182     this.trailer = trailer;
183     this.compressAlgo = trailer.getCompressionCodec();
184     this.cacheConf = cacheConf;
185     this.fileSize = fileSize;
186     this.path = path;
187     this.name = path.getName();
188     this.conf = conf;
189     checkFileVersion();
190     this.hfileContext = createHFileContext(fsdis, fileSize, hfs, path, trailer);
191     this.fsBlockReader = new HFileBlock.FSReaderImpl(fsdis, fileSize, hfs, path, hfileContext);
192 
193     // Comparator class name is stored in the trailer in version 2.
194     comparator = trailer.createComparator();
195     dataBlockIndexReader = new HFileBlockIndex.CellBasedKeyBlockIndexReader(comparator,
196         trailer.getNumDataIndexLevels(), this);
197     metaBlockIndexReader = new HFileBlockIndex.ByteArrayKeyBlockIndexReader(1);
198 
199     // Parse load-on-open data.
200 
201     HFileBlock.BlockIterator blockIter = fsBlockReader.blockRange(
202         trailer.getLoadOnOpenDataOffset(),
203         fileSize - trailer.getTrailerSize());
204 
205     // Data index. We also read statistics about the block index written after
206     // the root level.
207     dataBlockIndexReader.readMultiLevelIndexRoot(
208         blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX),
209         trailer.getDataIndexCount());
210 
211     // Meta index.
212     metaBlockIndexReader.readRootIndex(
213         blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX),
214         trailer.getMetaIndexCount());
215 
216     // File info
217     fileInfo = new FileInfo();
218     fileInfo.read(blockIter.nextBlockWithBlockType(BlockType.FILE_INFO).getByteStream());
219     byte[] creationTimeBytes = fileInfo.get(FileInfo.CREATE_TIME_TS);
220     this.hfileContext.setFileCreateTime(creationTimeBytes == null?  0:
221         Bytes.toLong(creationTimeBytes));
222     if (fileInfo.get(FileInfo.LASTKEY) != null) {
223       lastKeyCell = new KeyValue.KeyOnlyKeyValue(fileInfo.get(FileInfo.LASTKEY));
224     }
225     avgKeyLen = Bytes.toInt(fileInfo.get(FileInfo.AVG_KEY_LEN));
226     avgValueLen = Bytes.toInt(fileInfo.get(FileInfo.AVG_VALUE_LEN));
227     byte [] keyValueFormatVersion = fileInfo.get(HFileWriterImpl.KEY_VALUE_VERSION);
228     includesMemstoreTS = keyValueFormatVersion != null &&
229         Bytes.toInt(keyValueFormatVersion) == HFileWriterImpl.KEY_VALUE_VER_WITH_MEMSTORE;
230     fsBlockReader.setIncludesMemstoreTS(includesMemstoreTS);
231     if (includesMemstoreTS) {
232       decodeMemstoreTS = Bytes.toLong(fileInfo.get(HFileWriterImpl.MAX_MEMSTORE_TS_KEY)) > 0;
233     }
234 
235     // Read data block encoding algorithm name from file info.
236     dataBlockEncoder = HFileDataBlockEncoderImpl.createFromFileInfo(fileInfo);
237     fsBlockReader.setDataBlockEncoder(dataBlockEncoder);
238 
239     // Store all other load-on-open blocks for further consumption.
240     HFileBlock b;
241     while ((b = blockIter.nextBlock()) != null) {
242       loadOnOpenBlocks.add(b);
243     }
244 
245     // Prefetch file blocks upon open if requested
246     if (cacheConf.shouldPrefetchOnOpen()) {
247       PrefetchExecutor.request(path, new Runnable() {
248         public void run() {
249           try {
250             long offset = 0;
251             long end = fileSize - getTrailer().getTrailerSize();
252             HFileBlock prevBlock = null;
253             while (offset < end) {
254               if (Thread.interrupted()) {
255                 break;
256               }
257               long onDiskSize = -1;
258               if (prevBlock != null) {
259                 onDiskSize = prevBlock.getNextBlockOnDiskSizeWithHeader();
260               }
261               HFileBlock block = readBlock(offset, onDiskSize, true, false, false, false,
262                 null, null);
263               // Need not update the current block. Ideally here the readBlock won't find the
264               // block in cache. We call this readBlock so that block data is read from FS and
265               // cached in BC. So there is no reference count increment that happens here.
266               // The return will ideally be a noop because the block is not of MemoryType SHARED.
267               returnBlock(block);
268               prevBlock = block;
269               offset += block.getOnDiskSizeWithHeader();
270             }
271           } catch (IOException e) {
272             // IOExceptions are probably due to region closes (relocation, etc.)
273             if (LOG.isTraceEnabled()) {
274               LOG.trace("Exception encountered while prefetching " + path + ":", e);
275             }
276           } catch (Exception e) {
277             // Other exceptions are interesting
278             LOG.warn("Exception encountered while prefetching " + path + ":", e);
279           } finally {
280             PrefetchExecutor.complete(path);
281           }
282         }
283       });
284     }
285 
286     byte[] tmp = fileInfo.get(FileInfo.MAX_TAGS_LEN);
287     // max tag length is not present in the HFile means tags were not at all written to file.
288     if (tmp != null) {
289       hfileContext.setIncludesTags(true);
290       tmp = fileInfo.get(FileInfo.TAGS_COMPRESSED);
291       if (tmp != null && Bytes.toBoolean(tmp)) {
292         hfileContext.setCompressTags(true);
293       }
294     }
295   }

297   /**
298    * File version check is a little sloppy. We read v3 files but can also read v2 files if their
299    * content has been pb'd; files written with 0.98.
300    */
301   private void checkFileVersion() {
302     int majorVersion = trailer.getMajorVersion();
303     if (majorVersion == getMajorVersion()) return;
304     int minorVersion = trailer.getMinorVersion();
305     if (majorVersion == 2 && minorVersion >= MIN_V2_MINOR_VERSION_WITH_PB) return;
306     // We can read v3 or v2 versions of hfile.
307     throw new IllegalArgumentException("Invalid HFile version: major=" +
308       trailer.getMajorVersion() + ", minor=" + trailer.getMinorVersion() + ": expected at least " +
309       "major=2 and minor=" + MAX_MINOR_VERSION);
310   }

584     protected void readKeyValueLen() {
585       // This is a hot method. We go out of our way to make this method short so it can be
586       // inlined and is not too big to compile. We also manage position in ByteBuffer ourselves
587       // because it is faster than going via range-checked ByteBuffer methods or going through a
588       // byte buffer array a byte at a time.
589       // Get a long at a time rather than read two individual ints. In micro-benchmarking, even
590       // with the extra bit-fiddling, this is order-of-magnitude faster than getting two ints.
591       // Trying to imitate what was done - need to profile if this is better or
592       // earlier way is better by doing mark and reset?
593       // But ensure that you read long instead of two ints
594       long ll = blockBuffer.getLongAfterPosition(0);
595       // Read top half as an int of key length and bottom int as value length
596       this.currKeyLen = (int)(ll >> Integer.SIZE);
597       this.currValueLen = (int)(Bytes.MASK_FOR_LOWER_INT_IN_LONG ^ ll);
598       checkKeyValueLen();
599       // Move position past the key and value lengths and then beyond the key and value
600       int p = (Bytes.SIZEOF_LONG + currKeyLen + currValueLen);
601       if (reader.getFileContext().isIncludesTags()) {
602         // Tags length is a short.
603         this.currTagsLen = blockBuffer.getShortAfterPosition(p);
604         checkTagsLen();
605         p += (Bytes.SIZEOF_SHORT + currTagsLen);
606       }
607       readMvccVersion(p);
608     }

610     private final void checkTagsLen() {

613           ". Block offset: " + curBlock.getOffset() + ", block length: " +
614             this.blockBuffer.limit() +
615           ", position: " + this.blockBuffer.position() + " (without header).");

619     /**
620      * Read mvcc. Does checks to see if we even need to read the mvcc at all.
621      * @param offsetFromPos
622      */
623     protected void readMvccVersion(final int offsetFromPos) {
624       // See if we even need to decode mvcc.
625       if (!this.reader.shouldIncludeMemstoreTS()) return;
626       if (!this.reader.isDecodeMemstoreTS()) {
627         currMemstoreTS = 0;
628         currMemstoreTSLen = 1;
629         return;
630       }
631       _readMvccVersion(offsetFromPos);
632     }

634     /**
635      * Actually do the mvcc read. Does no checks.
636      * @param offsetFromPos
637      */
638     private void _readMvccVersion(int offsetFromPos) {
639       // This is Bytes#bytesToVint inlined so can save a few instructions in this hot method; i.e.
640       // previous if one-byte vint, we'd redo the vint call to find int size.
641       // Also the method is kept small so can be inlined.
642       byte firstByte = blockBuffer.getByteAfterPosition(offsetFromPos);
643       int len = WritableUtils.decodeVIntSize(firstByte);
644       if (len == 1) {
645         this.currMemstoreTS = firstByte;
646       } else {
647         int remaining = len -1;
648         long i = 0;
649         offsetFromPos++;
650         if (remaining >= Bytes.SIZEOF_INT) {
651           i = blockBuffer.getIntAfterPosition(offsetFromPos);
652           remaining -= Bytes.SIZEOF_INT;
653           offsetFromPos += Bytes.SIZEOF_INT;
654         }
655         if (remaining >= Bytes.SIZEOF_SHORT) {
656           short s = blockBuffer.getShortAfterPosition(offsetFromPos);
657           i = i << 16;
658           i = i | (s & 0xFFFF);
659           remaining -= Bytes.SIZEOF_SHORT;
660           offsetFromPos += Bytes.SIZEOF_SHORT;
661         }
662         for (int idx = 0; idx < remaining; idx++) {
663           byte b = blockBuffer.getByteAfterPosition(offsetFromPos + idx);
664           i = i << 8;
665           i = i | (b & 0xFF);
666         }
667         currMemstoreTS = (WritableUtils.isNegativeVInt(firstByte) ? ~i : i);
668       }
669       this.currMemstoreTSLen = len;
670     }

672     /**
673      * Within a loaded block, seek looking for the last key that is smaller than
674      * (or equal to?) the key we are interested in.
675      * A note on the seekBefore: if you have seekBefore = true, AND the first
676      * key in the block = key, then you'll get thrown exceptions. The caller has
677      * to check for that case and load the previous block as appropriate.
678      * @param key
679      *          the key to find
680      * @param seekBefore
681      *          find the key before the given key in case of exact match.
682      * @return 0 in case of an exact key match, 1 in case of an inexact match,
683      *         -2 in case of an inexact match and furthermore, the input key
684      *         less than the first key of current block(e.g. using a faked index
685      *         key)
686      */
687     protected int blockSeek(Cell key, boolean seekBefore) {
688       int klen, vlen, tlen = 0;
689       int lastKeyValueSize = -1;
690       int offsetFromPos;
691       do {
692         offsetFromPos = 0;
693         // Better to ensure that we use the BB Utils here
694         long ll = blockBuffer.getLongAfterPosition(offsetFromPos);
695         klen = (int)(ll >> Integer.SIZE);
696         vlen = (int)(Bytes.MASK_FOR_LOWER_INT_IN_LONG ^ ll);
697         if (klen < 0 || vlen < 0 || klen > blockBuffer.limit()
698             || vlen > blockBuffer.limit()) {
699           throw new IllegalStateException("Invalid klen " + klen + " or vlen "
700               + vlen + ". Block offset: "
701               + curBlock.getOffset() + ", block length: " + blockBuffer.limit() + ", position: "
702               + blockBuffer.position() + " (without header).");
703         }
704         offsetFromPos += Bytes.SIZEOF_LONG;
705         blockBuffer.asSubByteBuffer(blockBuffer.position() + offsetFromPos, klen, pair);
706         bufBackedKeyOnlyKv.setKey(pair.getFirst(), pair.getSecond(), klen);
707         int comp = reader.getComparator().compareKeyIgnoresMvcc(key, bufBackedKeyOnlyKv);
708         offsetFromPos += klen + vlen;
709         if (this.reader.getFileContext().isIncludesTags()) {
710           // Read short as unsigned, high byte first
711           tlen = ((blockBuffer.getByteAfterPosition(offsetFromPos) & 0xff) << 8)
712               ^ (blockBuffer.getByteAfterPosition(offsetFromPos + 1) & 0xff);
713           if (tlen < 0 || tlen > blockBuffer.limit()) {
714             throw new IllegalStateException("Invalid tlen " + tlen + ". Block offset: "
715                 + curBlock.getOffset() + ", block length: " + blockBuffer.limit() + ", position: "
716                 + blockBuffer.position() + " (without header).");
717           }
718           // add the two bytes read for the tags.
719           offsetFromPos += tlen + (Bytes.SIZEOF_SHORT);
720         }
721         if (this.reader.shouldIncludeMemstoreTS()) {
722           // Directly read the mvcc based on current position
723           readMvccVersion(offsetFromPos);
724         }
725         if (comp == 0) {
726           if (seekBefore) {
727             if (lastKeyValueSize < 0) {
728               throw new IllegalStateException("blockSeek with seekBefore "
729                   + "at the first key of the block: key=" + CellUtil.getCellKeyAsString(key)
730                   + ", blockOffset=" + curBlock.getOffset() + ", onDiskSize="
731                   + curBlock.getOnDiskSizeWithHeader());
732             }
733             blockBuffer.moveBack(lastKeyValueSize);
734             readKeyValueLen();
735             return 1; // non exact match.
736           }
737           currKeyLen = klen;
738           currValueLen = vlen;
739           currTagsLen = tlen;
740           return 0; // indicate exact match
741         } else if (comp < 0) {
742           if (lastKeyValueSize > 0) {
743             blockBuffer.moveBack(lastKeyValueSize);
744           }
745           readKeyValueLen();
746           if (lastKeyValueSize == -1 && blockBuffer.position() == 0) {
747             return HConstants.INDEX_KEY_MAGIC;
748           }
749           return 1;
750         }
751         // The size of this key/value tuple, including key/value length fields.
752         lastKeyValueSize = klen + vlen + currMemstoreTSLen + KEY_VALUE_LEN_SIZE;
753         // include tag length also if tags included with KV
754         if (reader.getFileContext().isIncludesTags()) {
755           lastKeyValueSize += tlen + Bytes.SIZEOF_SHORT;
756         }
757         blockBuffer.skip(lastKeyValueSize);
758       } while (blockBuffer.hasRemaining());
759 
760       // Seek to the last key we successfully read. This will happen if this is
761       // the last key/value pair in the file, in which case the following call
762       // to next() has to return false.
763       blockBuffer.moveBack(lastKeyValueSize);
764       readKeyValueLen();
765       return 1; // didn't exactly find it.
766     }

778     @Override
779     public int reseekTo(Cell key) throws IOException {
780       int compared;
781       if (isSeeked()) {
782         compared = compareKey(reader.getComparator(), key);
783         if (compared < 1) {
784           // If the required key is less than or equal to current key, then
785           // don't do anything.
786           return compared;
787         } else {
788           // The comparison with no_next_index_key has to be checked
789           if (this.nextIndexedKey != null &&
790               (this.nextIndexedKey == HConstants.NO_NEXT_INDEXED_KEY || reader
791               .getComparator().compareKeyIgnoresMvcc(key, nextIndexedKey) < 0)) {
792             // The reader shall continue to scan the current data block instead
793             // of querying the
794             // block index as long as it knows the target key is strictly
795             // smaller than
796             // the next indexed key or the current data block is the last data
797             // block.
798             return loadBlockAndSeekToKey(this.curBlock, nextIndexedKey, false, key,
799                 false);
800           }
801 
802         }
803       }
804       // Don't rewind on a reseek operation, because reseek implies that we are
805       // always going forward in the file.
806       return seekTo(key, false);
807     }

810      * An internal API function. Seek to the given key, optionally rewinding to
811      * the first key of the block before doing the seek.
812      *
813      * @param key - a cell representing the key that we need to fetch
814      * @param rewind whether to rewind to the first key of the block before
815      *        doing the seek. If this is false, we are assuming we never go
816      *        back, otherwise the result is undefined.
817      * @return -1 if the key is earlier than the first key of the file,
818      *         0 if we are at the given key, 1 if we are past the given key
819      *         -2 if the key is earlier than the first key of the file while
820      *         using a faked index key
821      * @throws IOException
822      */

824       HFileBlockIndex.BlockIndexReader indexReader = reader.getDataBlockIndexReader();
825       BlockWithScanInfo blockWithScanInfo = indexReader.loadDataBlockWithScanInfo(key, curBlock,
826           cacheBlocks, pread, isCompaction, getEffectiveDataBlockEncoding());
827       if (blockWithScanInfo == null || blockWithScanInfo.getHFileBlock() == null) {
828         // This happens if the key e.g. falls before the beginning of the
829         // file.
830         return -1;
831       }
832       return loadBlockAndSeekToKey(blockWithScanInfo.getHFileBlock(),
833           blockWithScanInfo.getNextIndexedKey(), rewind, key, false);

836     @Override
837     public boolean seekBefore(Cell key) throws IOException {
838       HFileBlock seekToBlock = reader.getDataBlockIndexReader().seekToDataBlock(key, curBlock,
839           cacheBlocks, pread, isCompaction, reader.getEffectiveEncodingInCache(isCompaction));
840       if (seekToBlock == null) {
841         return false;
842       }
843       Cell firstKey = getFirstKeyCellInBlock(seekToBlock);
844       if (reader.getComparator()
845            .compareKeyIgnoresMvcc(firstKey, key) >= 0) {
846         long previousBlockOffset = seekToBlock.getPrevBlockOffset();
847         // The key we are interested in
848         if (previousBlockOffset == -1) {
849           // we have a 'problem', the key we want is the first of the file.
850           return false;
851         }
852 
853         // The first key in the current block 'seekToBlock' is greater than the given 
854         // seekBefore key. We will go ahead by reading the next block that satisfies the
855         // given key. Return the current block before reading the next one.
856         reader.returnBlock(seekToBlock);
857         // It is important that we compute and pass onDiskSize to the block
858         // reader so that it does not have to read the header separately to
859         // figure out the size.  Currently, we do not have a way to do this
860         // correctly in the general case however.
861         // TODO: See https://issues.apache.org/jira/browse/HBASE-14576
862         int prevBlockSize = -1;
863         seekToBlock = reader.readBlock(previousBlockOffset,
864             prevBlockSize, cacheBlocks,
865             pread, isCompaction, true, BlockType.DATA, getEffectiveDataBlockEncoding());
866         // TODO shortcut: seek forward in this block to the last key of the
867         // block.
868       }
869       loadBlockAndSeekToKey(seekToBlock, firstKey, true, key, true);
870       return true;
871     }

873     /**
874      * Scans blocks in the "scanned" section of the {@link HFile} until the next
875      * data block is found.
876      *
877      * @return the next block, or null if there are no more data blocks
878      * @throws IOException
879      */
880     protected HFileBlock readNextDataBlock() throws IOException {
881       long lastDataBlockOffset = reader.getTrailer().getLastDataBlockOffset();
882       if (curBlock == null)
883         return null;
884 
885       HFileBlock block = this.curBlock;
886 
887       do {
888         if (block.getOffset() >= lastDataBlockOffset)
889           return null;
890 
891         if (block.getOffset() < 0) {
892           throw new IOException("Invalid block file offset: " + block);
893         }
894 
895         // We are reading the next block without block type validation, because
896         // it might turn out to be a non-data block.
897         block = reader.readBlock(block.getOffset()
898             + block.getOnDiskSizeWithHeader(),
899             block.getNextBlockOnDiskSizeWithHeader(), cacheBlocks, pread,
900             isCompaction, true, null, getEffectiveDataBlockEncoding());
901         if (block != null && !block.getBlockType().isData()) {
902           // Whatever block we read we will be returning it unless
903           // it is a datablock. Just in case the blocks are non data blocks
904           reader.returnBlock(block);
905         }
906       } while (!block.getBlockType().isData());
907 
908       return block;
909     }

915     @Override
916     public Cell getCell() {
917       if (!isSeeked())
918         return null;
919 
920       Cell ret;
921       int cellBufSize = getCellBufSize();
922       long seqId = 0l;
923       if (this.reader.shouldIncludeMemstoreTS()) {
924         seqId = currMemstoreTS;
925       }
926       if (blockBuffer.hasArray()) {
927         // TODO : reduce the varieties of KV here. Check if based on a boolean
928         // we can handle the 'no tags' case.
929         if (currTagsLen > 0) {
930           if (this.curBlock.usesSharedMemory()) {
931             ret = new ShareableMemoryKeyValue(blockBuffer.array(), blockBuffer.arrayOffset()
932               + blockBuffer.position(), getCellBufSize(), seqId);
933           } else {
934             ret = new SizeCachedKeyValue(blockBuffer.array(), blockBuffer.arrayOffset()
935                     + blockBuffer.position(), cellBufSize, seqId);
936           }
937         } else {
938           if (this.curBlock.usesSharedMemory()) {
939             ret = new ShareableMemoryNoTagsKeyValue(blockBuffer.array(), blockBuffer.arrayOffset()
940                     + blockBuffer.position(), getCellBufSize(), seqId);
941           } else {
942             ret = new SizeCachedNoTagsKeyValue(blockBuffer.array(), blockBuffer.arrayOffset()
943                     + blockBuffer.position(), cellBufSize, seqId);
944           }
945         }
946       } else {
947         ByteBuffer buf = blockBuffer.asSubByteBuffer(cellBufSize);
948         if (buf.isDirect()) {
949           if (this.curBlock.usesSharedMemory()) {
950             ret = new ShareableMemoryOffheapKeyValue(buf, buf.position(), cellBufSize,
951                 currTagsLen > 0, seqId);
952           } else {
953             ret = new OffheapKeyValue(buf, buf.position(), cellBufSize, currTagsLen > 0, seqId);
954           }
955         } else {
956           if (this.curBlock.usesSharedMemory()) {
957             if (currTagsLen > 0) {
958               ret = new ShareableMemoryKeyValue(buf.array(), buf.arrayOffset() + buf.position(),
959                   cellBufSize, seqId);
960             } else {
961               ret = new ShareableMemoryNoTagsKeyValue(buf.array(),
962                   buf.arrayOffset() + buf.position(), cellBufSize, seqId);
963             }
964           } else {
965             if (currTagsLen > 0) {
966               ret = new SizeCachedKeyValue(buf.array(), buf.arrayOffset() + buf.position(),
967                   cellBufSize, seqId);
968             } else {
969               ret = new SizeCachedNoTagsKeyValue(buf.array(), buf.arrayOffset() + buf.position(),
970                   cellBufSize, seqId);
971             }
972           }
973         }
974       }
975       return ret;
976     }

1314    * Retrieve block from cache. Validates the retrieved block's type vs {@code expectedBlockType}
1315    * and its encoding vs. {@code expectedDataBlockEncoding}. Unpacks the block as necessary.
1316    */
1317    private HFileBlock getCachedBlock(BlockCacheKey cacheKey, boolean cacheBlock, boolean useLock,
1318        boolean isCompaction, boolean updateCacheMetrics, BlockType expectedBlockType,
1319        DataBlockEncoding expectedDataBlockEncoding) throws IOException {
1320      // Check cache for block. If found return.
1321      if (cacheConf.isBlockCacheEnabled()) {
1322        BlockCache cache = cacheConf.getBlockCache();
1323        HFileBlock cachedBlock = (HFileBlock) cache.getBlock(cacheKey, cacheBlock, useLock,
1324          updateCacheMetrics);
1325        if (cachedBlock != null) {
1326          if (cacheConf.shouldCacheCompressed(cachedBlock.getBlockType().getCategory())) {
1327            HFileBlock compressedBlock = cachedBlock;
1328            cachedBlock = compressedBlock.unpack(hfileContext, fsBlockReader);
1329            // In case of compressed block after unpacking we can return the compressed block
1330           if (compressedBlock != cachedBlock) {
1331             cache.returnBlock(cacheKey, compressedBlock);
1332           }
1333         }
1334          validateBlockType(cachedBlock, expectedBlockType);
1335 
1336          if (expectedDataBlockEncoding == null) {
1337            return cachedBlock;
1338          }
1339          DataBlockEncoding actualDataBlockEncoding =
1340                  cachedBlock.getDataBlockEncoding();
1341          // Block types other than data blocks always have
1342          // DataBlockEncoding.NONE. To avoid false negative cache misses, only
1343          // perform this check if cached block is a data block.
1344          if (cachedBlock.getBlockType().isData() &&
1345                  !actualDataBlockEncoding.equals(expectedDataBlockEncoding)) {
1346            // This mismatch may happen if a Scanner, which is used for say a
1347            // compaction, tries to read an encoded block from the block cache.
1348            // The reverse might happen when an EncodedScanner tries to read
1349            // un-encoded blocks which were cached earlier.
1350            //
1351            // Because returning a data block with an implicit BlockType mismatch
1352            // will cause the requesting scanner to throw a disk read should be
1353            // forced here. This will potentially cause a significant number of
1354            // cache misses, so update so we should keep track of this as it might
1355            // justify the work on a CompoundScanner.
1356            if (!expectedDataBlockEncoding.equals(DataBlockEncoding.NONE) &&
1357                    !actualDataBlockEncoding.equals(DataBlockEncoding.NONE)) {
1358              // If the block is encoded but the encoding does not match the
1359              // expected encoding it is likely the encoding was changed but the
1360              // block was not yet evicted. Evictions on file close happen async
1361              // so blocks with the old encoding still linger in cache for some
1362              // period of time. This event should be rare as it only happens on
1363              // schema definition change.
1364              LOG.info("Evicting cached block with key " + cacheKey +
1365                      " because of a data block encoding mismatch" +
1366                      "; expected: " + expectedDataBlockEncoding +
1367                      ", actual: " + actualDataBlockEncoding);
1368              // This is an error scenario. so here we need to decrement the
1369              // count.
1370              cache.returnBlock(cacheKey, cachedBlock);
1371              cache.evictBlock(cacheKey);
1372            }
1373            return null;
1374          }
1375          return cachedBlock;
1376        }
1377      }
1378      return null;
1379    }

1441   public HFileBlock readBlock(long dataBlockOffset, long onDiskBlockSize,
1442       final boolean cacheBlock, boolean pread, final boolean isCompaction,
1443       boolean updateCacheMetrics, BlockType expectedBlockType,
1444       DataBlockEncoding expectedDataBlockEncoding)
1445       throws IOException {
1446     if (dataBlockIndexReader == null) {
1447       throw new IOException("Block index not loaded");
1448     }
1449     if (dataBlockOffset < 0 || dataBlockOffset >= trailer.getLoadOnOpenDataOffset()) {
1450       throw new IOException("Requested block is out of range: " + dataBlockOffset +
1451         ", lastDataBlockOffset: " + trailer.getLastDataBlockOffset());
1452     }
1453     // For any given block from any given file, synchronize reads for said
1454     // block.
1455     // Without a cache, this synchronizing is needless overhead, but really
1456     // the other choice is to duplicate work (which the cache would prevent you
1457     // from doing).
1458 
1459     BlockCacheKey cacheKey = new BlockCacheKey(name, dataBlockOffset,
1460       this.isPrimaryReplicaReader());

1462     boolean useLock = false;
1463     IdLock.Entry lockEntry = null;
1464     TraceScope traceScope = Trace.startSpan("HFileReaderImpl.readBlock");
1465     try {
1466       while (true) {
1467         // Check cache for block. If found return.
1468         if (cacheConf.shouldReadBlockFromCache(expectedBlockType)) {
1469           if (useLock) {
1470             lockEntry = offsetLock.getLockEntry(dataBlockOffset);
1471           }
1472           // Try and get the block from the block cache. If the useLock variable is true then this
1473           // is the second time through the loop and it should not be counted as a block cache miss.
1474           HFileBlock cachedBlock = getCachedBlock(cacheKey, cacheBlock, useLock, isCompaction,
1475             updateCacheMetrics, expectedBlockType, expectedDataBlockEncoding);
1476           if (cachedBlock != null) {
1477             if (Trace.isTracing()) {
1478               traceScope.getSpan().addTimelineAnnotation("blockCacheHit");
1479             }
1480             assert cachedBlock.isUnpacked() : "Packed block leak.";
1481             if (cachedBlock.getBlockType().isData()) {
1482               if (updateCacheMetrics) {
1483                 HFile.dataBlockReadCnt.incrementAndGet();
1484               }
1485               // Validate encoding type for data blocks. We include encoding
1486               // type in the cache key, and we expect it to match on a cache hit.
1487               if (cachedBlock.getDataBlockEncoding() != dataBlockEncoder.getDataBlockEncoding()) {
1488                 throw new IOException("Cached block under key " + cacheKey + " "
1489                   + "has wrong encoding: " + cachedBlock.getDataBlockEncoding() + " (expected: "
1490                   + dataBlockEncoder.getDataBlockEncoding() + ")");
1491               }
1492             }
1493             // Cache-hit. Return!
1494             return cachedBlock;
1495           }
1496 
1497           if (!useLock && cacheBlock && cacheConf.shouldLockOnCacheMiss(expectedBlockType)) {
1498             // check cache again with lock
1499             useLock = true;
1500             continue;
1501           }
1502           // Carry on, please load.
1503         }
1504 
1505         if (Trace.isTracing()) {
1506           traceScope.getSpan().addTimelineAnnotation("blockCacheMiss");
1507         }
1508         // Load block from filesystem.
1509         HFileBlock hfileBlock = fsBlockReader.readBlockData(dataBlockOffset, onDiskBlockSize, -1,
1510             pread);
1511         validateBlockType(hfileBlock, expectedBlockType);
1512         HFileBlock unpacked = hfileBlock.unpack(hfileContext, fsBlockReader);
1513         BlockType.BlockCategory category = hfileBlock.getBlockType().getCategory();
1514 
1515         // Cache the block if necessary
1516         if (cacheBlock && cacheConf.shouldCacheBlockOnRead(category)) {
1517           cacheConf.getBlockCache().cacheBlock(cacheKey,
1518             cacheConf.shouldCacheCompressed(category) ? hfileBlock : unpacked,
1519             cacheConf.isInMemory(), this.cacheConf.isCacheDataInL1());
1520         }
1521 
1522         if (updateCacheMetrics && hfileBlock.getBlockType().isData()) {
1523           HFile.dataBlockReadCnt.incrementAndGet();
1524         }
1525 
1526         return unpacked;
1527       }
1528     } finally {
1529       traceScope.close();
1530       if (lockEntry != null) {
1531         offsetLock.releaseLockEntry(lockEntry);
1532       }
1533     }
