343   private static IOException makeIOExceptionOfException(Exception e) {
344     Throwable t = e;
345     if (e instanceof ServiceException) {
346       t = e.getCause();
347     }
348     if (ExceptionUtil.isInterrupt(t)) {
349       return ExceptionUtil.asInterrupt(t);
350     }
351     if (t instanceof RemoteException) {
352       t = ((RemoteException)t).unwrapRemoteException();
353     }
354     return t instanceof IOException? (IOException)t: new HBaseIOException(t);
355   }

470   /**
471    * Convert a protocol buffer Get to a client Get
472    *
473    * @param proto the protocol buffer Get to convert
474    * @return the converted client Get
475    * @throws IOException
476    */
477   public static Get toGet(final ClientProtos.Get proto) throws IOException {
478     if (proto == null) return null;
479     byte[] row = proto.getRow().toByteArray();
480     Get get = new Get(row);
481     if (proto.hasCacheBlocks()) {
482       get.setCacheBlocks(proto.getCacheBlocks());
483     }
484     if (proto.hasMaxVersions()) {
485       get.setMaxVersions(proto.getMaxVersions());
486     }
487     if (proto.hasStoreLimit()) {
488       get.setMaxResultsPerColumnFamily(proto.getStoreLimit());
489     }
490     if (proto.hasStoreOffset()) {
491       get.setRowOffsetPerColumnFamily(proto.getStoreOffset());
492     }
493     if (proto.getCfTimeRangeCount() > 0) {
494       for (HBaseProtos.ColumnFamilyTimeRange cftr : proto.getCfTimeRangeList()) {
495         TimeRange timeRange = protoToTimeRange(cftr.getTimeRange());
496         get.setColumnFamilyTimeRange(cftr.getColumnFamily().toByteArray(),
497             timeRange.getMin(), timeRange.getMax());
498       }
499     }
500     if (proto.hasTimeRange()) {
501       TimeRange timeRange = protoToTimeRange(proto.getTimeRange());
502       get.setTimeRange(timeRange.getMin(), timeRange.getMax());
503     }
504     if (proto.hasFilter()) {
505       FilterProtos.Filter filter = proto.getFilter();
506       get.setFilter(ProtobufUtil.toFilter(filter));
507     }
508     for (NameBytesPair attribute: proto.getAttributeList()) {
509       get.setAttribute(attribute.getName(), attribute.getValue().toByteArray());
510     }
511     if (proto.getColumnCount() > 0) {
512       for (Column column: proto.getColumnList()) {
513         byte[] family = column.getFamily().toByteArray();
514         if (column.getQualifierCount() > 0) {
515           for (ByteString qualifier: column.getQualifierList()) {
516             get.addColumn(family, qualifier.toByteArray());
517           }
518         } else {
519           get.addFamily(family);
520         }
521       }
522     }
523     if (proto.hasExistenceOnly() && proto.getExistenceOnly()){

568    * @return A client Put.
569    * @throws IOException
570    */
571   public static Put toPut(final MutationProto proto, final CellScanner cellScanner)
572   throws IOException {
573     // TODO: Server-side at least why do we convert back to the Client types?  Why not just pb it?
574     MutationType type = proto.getMutateType();
575     assert type == MutationType.PUT: type.name();
576     long timestamp = proto.hasTimestamp()? proto.getTimestamp(): HConstants.LATEST_TIMESTAMP;
577     Put put = proto.hasRow() ? new Put(proto.getRow().toByteArray(), timestamp) : null;
578     int cellCount = proto.hasAssociatedCellCount()? proto.getAssociatedCellCount(): 0;
579     if (cellCount > 0) {
580       // The proto has metadata only and the data is separate to be found in the cellScanner.
581       if (cellScanner == null) {
582         throw new DoNotRetryIOException("Cell count of " + cellCount + " but no cellScanner: " +
583             toShortString(proto));
584       }
585       for (int i = 0; i < cellCount; i++) {
586         if (!cellScanner.advance()) {
587           throw new DoNotRetryIOException("Cell count of " + cellCount + " but at index " + i +
588             " no cell returned: " + toShortString(proto));
589         }
590         Cell cell = cellScanner.current();
591         if (put == null) {
592           put = new Put(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(), timestamp);
593         }
594         put.add(cell);
595       }
596     } else {
597       if (put == null) {
598         throw new IllegalArgumentException("row cannot be null");
599       }
600       // The proto has the metadata and the data itself
601       for (ColumnValue column: proto.getColumnValueList()) {
602         byte[] family = column.getFamily().toByteArray();
603         for (QualifierValue qv: column.getQualifierValueList()) {
604           if (!qv.hasValue()) {
605             throw new DoNotRetryIOException(
606                 "Missing required field: qualifier value");
607           }
608           ByteBuffer qualifier =
609               qv.hasQualifier() ? qv.getQualifier().asReadOnlyByteBuffer() : null;
610           ByteBuffer value =
611               qv.hasValue() ? qv.getValue().asReadOnlyByteBuffer() : null;
612           long ts = timestamp;
613           if (qv.hasTimestamp()) {
614             ts = qv.getTimestamp();
615           }
616           byte[] allTagsBytes;
617           if (qv.hasTags()) {
618             allTagsBytes = qv.getTags().toByteArray();
619             if(qv.hasDeleteType()) {
620               byte[] qual = qv.hasQualifier() ? qv.getQualifier().toByteArray() : null;
621               put.add(new KeyValue(proto.getRow().toByteArray(), family, qual, ts,
622                   fromDeleteType(qv.getDeleteType()), null, allTagsBytes));
623             } else {
624               List<Tag> tags = TagUtil.asList(allTagsBytes, 0, (short)allTagsBytes.length);
625               Tag[] tagsArray = new Tag[tags.size()];
626               put.addImmutable(family, qualifier, ts, value, tags.toArray(tagsArray));
627             }
628           } else {
629             if(qv.hasDeleteType()) {
630               byte[] qual = qv.hasQualifier() ? qv.getQualifier().toByteArray() : null;
631               put.add(new KeyValue(proto.getRow().toByteArray(), family, qual, ts,
632                   fromDeleteType(qv.getDeleteType())));
633             } else{
634               put.addImmutable(family, qualifier, ts, value);
635             }
636           }
637         }
638       }
639     }
640     put.setDurability(toDurability(proto.getDurability()));
641     for (NameBytesPair attribute: proto.getAttributeList()) {
642       put.setAttribute(attribute.getName(), attribute.getValue().toByteArray());
643     }
644     return put;
