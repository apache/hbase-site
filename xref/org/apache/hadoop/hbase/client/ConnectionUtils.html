50    /**
51     * Calculate pause time.
52     * Built on {@link HConstants#RETRY_BACKOFF}.
53     * @param pause time to pause
54     * @param tries amount of tries
55     * @return How long to wait after <code>tries</code> retries
56     */
57    public static long getPauseTime(final long pause, final int tries) {
58      int ntries = tries;
59      if (ntries >= HConstants.RETRY_BACKOFF.length) {
60        ntries = HConstants.RETRY_BACKOFF.length - 1;
61      }
62      if (ntries < 0) {
63        ntries = 0;
64      }
65
66      long normalPause = pause * HConstants.RETRY_BACKOFF[ntries];
67      // 1% possible jitter
68      long jitter = (long) (normalPause * ThreadLocalRandom.current().nextFloat() * 0.01f);
69      return normalPause + jitter;
70    }

73    /**
74     * Adds / subs an up to 50% jitter to a pause time. Minimum is 1.
75     * @param pause the expected pause.
76     * @param jitter the jitter ratio, between 0 and 1, exclusive.
77     */
78    public static long addJitter(final long pause, final float jitter) {
79      float lag = pause * (ThreadLocalRandom.current().nextFloat() - 0.5f) * jitter;
80      long newPause = pause + (long) lag;
81      if (newPause <= 0) {
82        return 1;
83      }
84      return newPause;
85    }

97    /**
98     * Changes the configuration to set the number of retries needed when using Connection
99     * internally, e.g. for  updating catalog tables, etc.
100    * Call this method before we create any Connections.
101    * @param c The Configuration instance to set the retries into.
102    * @param log Used to log what we set in here.
103    */
104   public static void setServerSideHConnectionRetriesConfig(
105       final Configuration c, final String sn, final Log log) {
106     // TODO: Fix this. Not all connections from server side should have 10 times the retries.
107     int hcRetries = c.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
108       HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
109     // Go big.  Multiply by 10.  If we can't get to meta after this many retries
110     // then something seriously wrong.
111     int serversideMultiplier = c.getInt("hbase.client.serverside.retries.multiplier", 10);
112     int retries = hcRetries * serversideMultiplier;
113     c.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, retries);
114     log.info(sn + " server-side Connection retries=" + retries);
115   }

117   /**
118    * Creates a short-circuit connection that can bypass the RPC layer (serialization,
119    * deserialization, networking, etc..) when talking to a local server.
120    * @param conf the current configuration
121    * @param pool the thread pool to use for batch operations
122    * @param user the user the connection is for
123    * @param serverName the local server name
124    * @param admin the admin interface of the local server
125    * @param client the client interface of the local server
126    * @return an short-circuit connection.
127    * @throws IOException if IO failure occurred
128    */
129   public static ClusterConnection createShortCircuitConnection(final Configuration conf,
130     ExecutorService pool, User user, final ServerName serverName,
131     final AdminService.BlockingInterface admin, final ClientService.BlockingInterface client)
132     throws IOException {
133     if (user == null) {
134       user = UserProvider.instantiate(conf).getCurrent();
135     }
136     return new ConnectionImplementation(conf, pool, user) {
137       @Override
138       public AdminService.BlockingInterface getAdmin(ServerName sn) throws IOException {
139         return serverName.equals(sn) ? admin : super.getAdmin(sn);
140       }
141
142       @Override
143       public ClientService.BlockingInterface getClient(ServerName sn) throws IOException {
144         return serverName.equals(sn) ? client : super.getClient(sn);
145       }
146     };
147   }

183   /**
184    * Get a unique key for the rpc stub to the given server.
185    */
186   static String getStubKey(String serviceName, ServerName serverName,
187       boolean hostnameCanChange) {
188     // Sometimes, servers go down and they come back up with the same hostname but a different
189     // IP address. Force a resolution of the rsHostname by trying to instantiate an
190     // InetSocketAddress, and this way we will rightfully get a new stubKey.
191     // Also, include the hostname in the key so as to take care of those cases where the
192     // DNS name is different but IP address remains the same.
193     String hostname = serverName.getHostname();
194     int port = serverName.getPort();
195     if (hostnameCanChange) {
196       try {
197         InetAddress ip = InetAddress.getByName(hostname);
198         return serviceName + "@" + hostname + "-" + ip.getHostAddress() + ":" + port;
199       } catch (UnknownHostException e) {
200         LOG.warn("Can not resolve " + hostname + ", please check your network", e);
201       }
202     }
203     return serviceName + "@" + hostname + ":" + port;
204   }
