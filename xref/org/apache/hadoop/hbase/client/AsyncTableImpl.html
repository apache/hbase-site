56    public AsyncTableImpl(AsyncConnectionImpl conn, TableName tableName) {
57      this.conn = conn;
58      this.tableName = tableName;
59      this.readRpcTimeoutNs = conn.connConf.getReadRpcTimeoutNs();
60      this.writeRpcTimeoutNs = conn.connConf.getWriteRpcTimeoutNs();
61      this.operationTimeoutNs = tableName.isSystemTable() ? conn.connConf.getMetaOperationTimeoutNs()
62          : conn.connConf.getOperationTimeoutNs();
63    }

75    @FunctionalInterface
76    private interface Converter<D, I, S> {
77      D convert(I info, S src) throws IOException;

80    @FunctionalInterface
81    private interface RpcCall<RESP, REQ> {
82      void call(ClientService.Interface stub, HBaseRpcController controller, REQ req,
83          RpcCallback<RESP> done);
84    }
85
86    private static <REQ, PREQ, PRESP, RESP> CompletableFuture<RESP> call(
87        HBaseRpcController controller, HRegionLocation loc, ClientService.Interface stub, REQ req,
88        Converter<PREQ, byte[], REQ> reqConvert, RpcCall<PRESP, PREQ> rpcCall,
89        Converter<RESP, HBaseRpcController, PRESP> respConverter) {
90      CompletableFuture<RESP> future = new CompletableFuture<>();
91      try {
92        rpcCall.call(stub, controller, reqConvert.convert(loc.getRegionInfo().getRegionName(), req),
93          new RpcCallback<PRESP>() {
94
95            @Override
96            public void run(PRESP resp) {
97              if (controller.failed()) {
98                future.completeExceptionally(controller.getFailed());
99              } else {
100               try {
101                 future.complete(respConverter.convert(controller, resp));
102               } catch (IOException e) {
103                 future.completeExceptionally(e);
104               }
105             }
106           }
107         });
108     } catch (IOException e) {
109       future.completeExceptionally(e);
110     }
111     return future;
112   }

114   private <T> SingleRequestCallerBuilder<T> newCaller(Row row, long rpcTimeoutNs) {
115     return conn.callerFactory.<T> single().table(tableName).row(row.getRow())
116         .rpcTimeout(rpcTimeoutNs, TimeUnit.NANOSECONDS)
117         .operationTimeout(operationTimeoutNs, TimeUnit.NANOSECONDS);
118   }

120   @Override
121   public CompletableFuture<Boolean> exists(Get get) {
122     if (!get.isCheckExistenceOnly()) {
123       get = ReflectionUtils.newInstance(get.getClass(), get);
124       get.setCheckExistenceOnly(true);
125     }
126     return get(get).thenApply(r -> r.getExists());
127   }

129   @Override
130   public CompletableFuture<Result> get(Get get) {
131     return this.<Result> newCaller(get, readRpcTimeoutNs)
132         .action((controller, loc, stub) -> AsyncTableImpl
133             .<Get, GetRequest, GetResponse, Result> call(controller, loc, stub, get,
134               RequestConverter::buildGetRequest, (s, c, req, done) -> s.get(c, req, done),
135               (c, resp) -> ProtobufUtil.toResult(resp.getResult(), c.cellScanner())))
136         .call();
137   }

139   @Override
140   public CompletableFuture<Void> put(Put put) {

144             controller, loc, stub, put, RequestConverter::buildMutateRequest,
145             (s, c, req, done) -> s.mutate(c, req, done), (c, resp) -> {
146               return null;
147             }))

149   }
