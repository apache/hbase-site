18  package org.apache.hadoop.hbase.client;
19
20  import com.google.common.base.Preconditions;
21
22  import io.netty.util.HashedWheelTimer;
23
24  import java.util.concurrent.CompletableFuture;
25  import java.util.concurrent.TimeUnit;
26
27  import org.apache.hadoop.hbase.TableName;
28  import org.apache.hadoop.hbase.classification.InterfaceAudience;
29
30  /**
31   * Factory to create an AsyncRpcRetryCaller.
32   */
33  @InterfaceAudience.Private
34  class AsyncRpcRetryingCallerFactory {
35
36    private final AsyncConnectionImpl conn;
37
38    private final HashedWheelTimer retryTimer;
39
40    public AsyncRpcRetryingCallerFactory(AsyncConnectionImpl conn, HashedWheelTimer retryTimer) {
41      this.conn = conn;
42      this.retryTimer = retryTimer;
43    }
44
45    public class SingleRequestCallerBuilder<T> {
46
47      private TableName tableName;
48
49      private byte[] row;
50
51      private AsyncSingleRequestRpcRetryingCaller.Callable<T> callable;
52
53      private long operationTimeoutNs = -1L;
54
55      private long rpcTimeoutNs = -1L;
56
57      public SingleRequestCallerBuilder<T> table(TableName tableName) {
58        this.tableName = tableName;
59        return this;
60      }
61
62      public SingleRequestCallerBuilder<T> row(byte[] row) {
63        this.row = row;
64        return this;
65      }
66
67      public SingleRequestCallerBuilder<T> action(
68          AsyncSingleRequestRpcRetryingCaller.Callable<T> callable) {
69        this.callable = callable;
70        return this;
71      }
72
73      public SingleRequestCallerBuilder<T> operationTimeout(long operationTimeout, TimeUnit unit) {
74        this.operationTimeoutNs = unit.toNanos(operationTimeout);
75        return this;
76      }
77
78      public SingleRequestCallerBuilder<T> rpcTimeout(long rpcTimeout, TimeUnit unit) {
79        this.rpcTimeoutNs = unit.toNanos(rpcTimeout);
80        return this;
81      }
82
83      public AsyncSingleRequestRpcRetryingCaller<T> build() {
84        return new AsyncSingleRequestRpcRetryingCaller<>(retryTimer, conn,
85            Preconditions.checkNotNull(tableName, "tableName is null"),
86            Preconditions.checkNotNull(row, "row is null"),
87            Preconditions.checkNotNull(callable, "action is null"), conn.connConf.getPauseNs(),
88            conn.connConf.getMaxRetries(), operationTimeoutNs, rpcTimeoutNs,
89            conn.connConf.getStartLogErrorsCnt());
90      }
91
92      /**
93       * Shortcut for {@code build().call()}
94       */
95      public CompletableFuture<T> call() {
96        return build().call();
97      }
98    }
99
100   /**
101    * Create retry caller for single action, such as get, put, delete, etc.
102    */
103   public <T> SingleRequestCallerBuilder<T> single() {
104     return new SingleRequestCallerBuilder<>();
105   }
106 }
