37  /**
38   * This compactor will generate StoreFile for different time ranges.
39   */
40  @InterfaceAudience.Private
41  public class DateTieredCompactor extends AbstractMultiOutputCompactor<DateTieredMultiFileWriter> {
42  
43    private static final Log LOG = LogFactory.getLog(DateTieredCompactor.class);
44  
45    public DateTieredCompactor(Configuration conf, Store store) {
46      super(conf, store);
47    }
48  
49    private boolean needEmptyFile(CompactionRequest request) {
50      // if we are going to compact the last N files, then we need to emit an empty file to retain the
51      // maxSeqId if we haven't written out anything.
52      return StoreFile.getMaxSequenceIdInList(request.getFiles()) == store.getMaxSequenceId();
53    }
54  
55    public List<Path> compact(final CompactionRequest request, List<Long> lowerBoundaries,
56        ThroughputController throughputController, User user) throws IOException {
57      if (LOG.isDebugEnabled()) {
58        LOG.debug("Executing compaction with " + lowerBoundaries.size()
59            + "windows, lower boundaries: " + lowerBoundaries);
60      }
61  
62      DateTieredMultiFileWriter writer =
63          new DateTieredMultiFileWriter(lowerBoundaries, needEmptyFile(request));
64      return compact(writer, request, new InternalScannerFactory() {
65  
66        @Override
67        public ScanType getScanType(CompactionRequest request) {
68          return request.isRetainDeleteMarkers() ? ScanType.COMPACT_RETAIN_DELETES
69              : ScanType.COMPACT_DROP_DELETES;
70        }
71  
72        @Override
73        public InternalScanner createScanner(List<StoreFileScanner> scanners, ScanType scanType,
74            FileDetails fd, long smallestReadPoint) throws IOException {
75          return DateTieredCompactor.this.createScanner(store, scanners, scanType, smallestReadPoint,
76            fd.earliestPutTs);
77        }
78      }, throughputController, user);
79    }
80  
81    @Override
82    protected List<Path> commitMultiWriter(DateTieredMultiFileWriter writer, FileDetails fd,
83        CompactionRequest request) throws IOException {
84      return writer.commitWriters(fd.maxSeqId, request.isAllFiles());
85    }
86  }
