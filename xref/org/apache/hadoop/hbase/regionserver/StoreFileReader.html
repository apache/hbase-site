207   /**
208    * Checks whether the given scan passes the Bloom filter (if present). Only
209    * checks Bloom filters for single-row or single-row-column scans. Bloom
210    * filter checking for multi-gets is implemented as part of the store
211    * scanner system (see {@link StoreFileScanner#seekExactly}) and uses
212    * the lower-level API {@link #passesGeneralRowBloomFilter(byte[], int, int)}
213    * and {@link #passesGeneralRowColBloomFilter(Cell)}.
214    *
215    * @param scan the scan specification. Used to determine the row, and to
216    *          check whether this is a single-row ("get") scan.
217    * @param columns the set of columns. Only used for row-column Bloom
218    *          filters.
219    * @return true if the scan with the given column set passes the Bloom
220    *         filter, or if the Bloom filter is not applicable for the scan.
221    *         False if the Bloom filter is applicable and the scan fails it.
222    */
223   boolean passesBloomFilter(Scan scan, final SortedSet<byte[]> columns) {
224     // Multi-column non-get scans will use Bloom filters through the
225     // lower-level API function that this function calls.
226     if (!scan.isGetScan()) {
227       return true;
228     }
229
230     byte[] row = scan.getStartRow();
231     switch (this.bloomFilterType) {
232       case ROW:
233         return passesGeneralRowBloomFilter(row, 0, row.length);
234
235       case ROWCOL:
236         if (columns != null && columns.size() == 1) {
237           byte[] column = columns.first();
238           // create the required fake key
239           Cell kvKey = KeyValueUtil.createFirstOnRow(row, 0, row.length,
240             HConstants.EMPTY_BYTE_ARRAY, 0, 0, column, 0,
241             column.length);
242           return passesGeneralRowColBloomFilter(kvKey);
243         }
244
245         // For multi-column queries the Bloom filter is checked from the
246         // seekExact operation.
247         return true;
248
249       default:
250         return true;
251     }
252   }

254   public boolean passesDeleteFamilyBloomFilter(byte[] row, int rowOffset,
255       int rowLen) {
256     // Cache Bloom filter as a local variable in case it is set to null by
257     // another thread on an IO error.
258     BloomFilter bloomFilter = this.deleteFamilyBloomFilter;
259
260     // Empty file or there is no delete family at all
261     if (reader.getTrailer().getEntryCount() == 0 || deleteFamilyCnt == 0) {
262       return false;
263     }
264
265     if (bloomFilter == null) {
266       return true;
267     }
268 
269     try {
270       if (!bloomFilter.supportsAutoLoading()) {
271         return true;
272       }
273       return bloomFilter.contains(row, rowOffset, rowLen, null);
274     } catch (IllegalArgumentException e) {
275       LOG.error("Bad Delete Family bloom filter data -- proceeding without",
276           e);
277       setDeleteFamilyBloomFilterFaulty();
278     }
279
280     return true;
281   }

283   /**
284    * A method for checking Bloom filters. Called directly from
285    * StoreFileScanner in case of a multi-column query.
286    *
287    * @return True if passes
288    */
289   public boolean passesGeneralRowBloomFilter(byte[] row, int rowOffset, int rowLen) {
290     BloomFilter bloomFilter = this.generalBloomFilter;
291     if (bloomFilter == null) {
292       return true;
293     }
294
295     // Used in ROW bloom
296     byte[] key = null;
297     if (rowOffset != 0 || rowLen != row.length) {
298       throw new AssertionError(
299           "For row-only Bloom filters the row " + "must occupy the whole array");
300     }
301     key = row;
302     return checkGeneralBloomFilter(key, null, bloomFilter);
303   }

305   /**
306    * A method for checking Bloom filters. Called directly from
307    * StoreFileScanner in case of a multi-column query.
308    *
309    * @param cell
310    *          the cell to check if present in BloomFilter
311    * @return True if passes
312    */
313   public boolean passesGeneralRowColBloomFilter(Cell cell) {
314     BloomFilter bloomFilter = this.generalBloomFilter;
315     if (bloomFilter == null) {
316       return true;
317     }
318     // Used in ROW_COL bloom
319     Cell kvKey = null;
320     // Already if the incoming key is a fake rowcol key then use it as it is
321     if (cell.getTypeByte() == KeyValue.Type.Maximum.getCode() && cell.getFamilyLength() == 0) {
322       kvKey = cell;
323     } else {
324       kvKey = CellUtil.createFirstOnRowCol(cell);
325     }
326     return checkGeneralBloomFilter(null, kvKey, bloomFilter);
327   }

329   private boolean checkGeneralBloomFilter(byte[] key, Cell kvKey, BloomFilter bloomFilter) {
330     // Empty file
331     if (reader.getTrailer().getEntryCount() == 0) {
332       return false;
333     }
334     HFileBlock bloomBlock = null;
335     try {
336       boolean shouldCheckBloom;
337       ByteBuff bloom;
338       if (bloomFilter.supportsAutoLoading()) {
339         bloom = null;
340         shouldCheckBloom = true;
341       } else {
342         bloomBlock = reader.getMetaBlock(HFile.BLOOM_FILTER_DATA_KEY, true);
343         bloom = bloomBlock.getBufferWithoutHeader();
344         shouldCheckBloom = bloom != null;
345       }
346
347       if (shouldCheckBloom) {
348         boolean exists;
349
350         // Whether the primary Bloom key is greater than the last Bloom key
351         // from the file info. For row-column Bloom filters this is not yet
352         // a sufficient condition to return false.
353         boolean keyIsAfterLast = (lastBloomKey != null);
354         // hbase:meta does not have blooms. So we need not have special interpretation
355         // of the hbase:meta cells.  We can safely use Bytes.BYTES_RAWCOMPARATOR for ROW Bloom
356         if (keyIsAfterLast) {
357           if (bloomFilterType == BloomType.ROW) {
358             keyIsAfterLast = (Bytes.BYTES_RAWCOMPARATOR.compare(key, lastBloomKey) > 0);
359           } else {
360             keyIsAfterLast = (CellComparator.COMPARATOR.compare(kvKey, lastBloomKeyOnlyKV)) > 0;
361           }
362         }
363
364         if (bloomFilterType == BloomType.ROWCOL) {
365           // Since a Row Delete is essentially a DeleteFamily applied to all
366           // columns, a file might be skipped if using row+col Bloom filter.
367           // In order to ensure this file is included an additional check is
368           // required looking only for a row bloom.
369           Cell rowBloomKey = CellUtil.createFirstOnRow(kvKey);
370           // hbase:meta does not have blooms. So we need not have special interpretation
371           // of the hbase:meta cells.  We can safely use Bytes.BYTES_RAWCOMPARATOR for ROW Bloom
372           if (keyIsAfterLast
373               && (CellComparator.COMPARATOR.compare(rowBloomKey, lastBloomKeyOnlyKV)) > 0) {
374             exists = false;
375           } else {
376             exists =
377                 bloomFilter.contains(kvKey, bloom, BloomType.ROWCOL) ||
378                 bloomFilter.contains(rowBloomKey, bloom, BloomType.ROWCOL);
379           }
380         } else {
381           exists = !keyIsAfterLast
382               && bloomFilter.contains(key, 0, key.length, bloom);
383         }
384
385         return exists;
386       }
387     } catch (IOException e) {
388       LOG.error("Error reading bloom filter data -- proceeding without",
389           e);
390       setGeneralBloomFilterFaulty();
391     } catch (IllegalArgumentException e) {
392       LOG.error("Bad bloom filter data -- proceeding without", e);
393       setGeneralBloomFilterFaulty();
394     } finally {
395       // Return the bloom block so that its ref count can be decremented.
396       reader.returnBlock(bloomBlock);
397     }
398     return true;
399   }

401   /**
402    * Checks whether the given scan rowkey range overlaps with the current storefile's
403    * @param scan the scan specification. Used to determine the rowkey range.
404    * @return true if there is overlap, false otherwise
405    */
406   public boolean passesKeyRangeFilter(Scan scan) {
407     if (this.getFirstKey() == null || this.getLastKey() == null) {
408       // the file is empty
409       return false;
410     }
411     if (Bytes.equals(scan.getStartRow(), HConstants.EMPTY_START_ROW)
412         && Bytes.equals(scan.getStopRow(), HConstants.EMPTY_END_ROW)) {
413       return true;
414     }
415     byte[] smallestScanRow = scan.isReversed() ? scan.getStopRow() : scan.getStartRow();
416     byte[] largestScanRow = scan.isReversed() ? scan.getStartRow() : scan.getStopRow();
417     Cell firstKeyKV = this.getFirstKey();
418     Cell lastKeyKV = this.getLastKey();
419     boolean nonOverLapping = (getComparator().compareRows(firstKeyKV,
420         largestScanRow, 0, largestScanRow.length) > 0
421         && !Bytes
422         .equals(scan.isReversed() ? scan.getStartRow() : scan.getStopRow(),
423             HConstants.EMPTY_END_ROW))
424         || getComparator().compareRows(lastKeyKV, smallestScanRow, 0, smallestScanRow.length) < 0;
425     return !nonOverLapping;
426   }

428   public Map<byte[], byte[]> loadFileInfo() throws IOException {
429     Map<byte [], byte []> fi = reader.loadFileInfo();
430
431     byte[] b = fi.get(StoreFile.BLOOM_FILTER_TYPE_KEY);
432     if (b != null) {
433       bloomFilterType = BloomType.valueOf(Bytes.toString(b));
434     }
435
436     lastBloomKey = fi.get(StoreFile.LAST_BLOOM_KEY);
437     if(bloomFilterType == BloomType.ROWCOL) {
438       lastBloomKeyOnlyKV = new KeyValue.KeyOnlyKeyValue(lastBloomKey, 0, lastBloomKey.length);
439     }
440     byte[] cnt = fi.get(StoreFile.DELETE_FAMILY_COUNT);
441     if (cnt != null) {
442       deleteFamilyCnt = Bytes.toLong(cnt);
443     }
444
445     return fi;
446   }

453   public void loadBloomfilter(BlockType blockType) {
454     try {
455       if (blockType == BlockType.GENERAL_BLOOM_META) {
456         if (this.generalBloomFilter != null)
457           return; // Bloom has been loaded
458
459         DataInput bloomMeta = reader.getGeneralBloomFilterMetadata();
460         if (bloomMeta != null) {
461           // sanity check for NONE Bloom filter
462           if (bloomFilterType == BloomType.NONE) {
463             throw new IOException(
464                 "valid bloom filter type not found in FileInfo");
465           } else {
466             generalBloomFilter = BloomFilterFactory.createFromMeta(bloomMeta,
467                 reader);
468             if (LOG.isTraceEnabled()) {
469               LOG.trace("Loaded " + bloomFilterType.toString() + " "
470                 + generalBloomFilter.getClass().getSimpleName()
471                 + " metadata for " + reader.getName());
472             }
473           }
474         }
475       } else if (blockType == BlockType.DELETE_FAMILY_BLOOM_META) {
476         if (this.deleteFamilyBloomFilter != null)
477           return; // Bloom has been loaded
478
479         DataInput bloomMeta = reader.getDeleteBloomFilterMetadata();
480         if (bloomMeta != null) {
481           deleteFamilyBloomFilter = BloomFilterFactory.createFromMeta(
482               bloomMeta, reader);
483           LOG.info("Loaded Delete Family Bloom ("
484               + deleteFamilyBloomFilter.getClass().getSimpleName()
485               + ") metadata for " + reader.getName());
486         }
487       } else {
488         throw new RuntimeException("Block Type: " + blockType.toString()
489             + "is not supported for Bloom filter");
490       }
491     } catch (IOException e) {
492       LOG.error("Error reading bloom filter meta for " + blockType
493           + " -- proceeding without", e);
494       setBloomFilterFaulty(blockType);
495     } catch (IllegalArgumentException e) {
496       LOG.error("Bad bloom filter meta " + blockType
497           + " -- proceeding without", e);
498       setBloomFilterFaulty(blockType);
499     }
500   }
