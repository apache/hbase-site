52  /**
53   * The MemStore holds in-memory modifications to the Store.  Modifications
54   * are {@link Cell}s.  When asked to flush, current memstore is moved
55   * to snapshot and is cleared.  We continue to serve edits out of new memstore
56   * and backing snapshot until flusher reports in that the flush succeeded. At
57   * this point we let the snapshot go.
58   *  <p>
59   * The MemStore functions should not be called in parallel. Callers should hold
60   *  write and read locks. This is done in {@link HStore}.
61   *  </p>
62   *
63   * TODO: Adjust size of the memstore when we remove items because they have
64   * been deleted.
65   * TODO: With new KVSLS, need to make sure we update HeapSize with difference
66   * in KV size.
67   */
68  @InterfaceAudience.Private
69  public class DefaultMemStore implements MemStore {

73    static final String MSLAB_CLASS_NAME = "hbase.regionserver.mslab.class";

143   /**
144    * Creates a snapshot of the current memstore.
145    * Snapshot must be cleared by call to {@link #clearSnapshot(long)}
146    */
147   @Override
148   public MemStoreSnapshot snapshot() {
149     // If snapshot currently has entries, then flusher failed or didn't call
150     // cleanup.  Log a warning.
151     if (!this.snapshot.isEmpty()) {
152       LOG.warn("Snapshot called again without clearing previous. " +
153           "Doing nothing. Another ongoing flush or did we fail last attempt?");
154     } else {
155       this.snapshotId = EnvironmentEdgeManager.currentTime();
156       this.snapshotSize = keySize();
157       if (!this.cellSet.isEmpty()) {
158         this.snapshot = this.cellSet;
159         this.cellSet = new CellSkipListSet(this.comparator);
160         this.snapshotTimeRangeTracker = this.timeRangeTracker;
161         this.timeRangeTracker = new TimeRangeTracker();
162         // Reset heap to not include any keys
163         this.size.set(DEEP_OVERHEAD);
164         this.snapshotAllocator = this.allocator;
165         // Reset allocator so we get a fresh buffer for the new memstore
166         if (allocator != null) {
167           String className = conf.get(MSLAB_CLASS_NAME, HeapMemStoreLAB.class.getName());
168           this.allocator = ReflectionUtils.instantiateWithCustomCtor(className,
169               new Class[] { Configuration.class }, new Object[] { conf });
170         }
171         timeOfOldestEdit = Long.MAX_VALUE;
172       }
173     }
174     MemStoreSnapshot memStoreSnapshot = new MemStoreSnapshot(this.snapshotId, snapshot.size(), this.snapshotSize,
175         this.snapshotTimeRangeTracker, new CollectionBackedScanner(snapshot, this.comparator),
176         this.tagsPresent);
177     this.tagsPresent = false;
178     return memStoreSnapshot;
179   }

181   /**
182    * The passed snapshot was successfully persisted; it can be let go.
183    * @param id Id of the snapshot to clean out.
184    * @throws UnexpectedStateException
185    * @see #snapshot()
186    */
187   @Override
188   public void clearSnapshot(long id) throws UnexpectedStateException {
189     MemStoreLAB tmpAllocator = null;
190     if (this.snapshotId != id) {
191       throw new UnexpectedStateException("Current snapshot id is " + this.snapshotId + ",passed "
192           + id);
193     }
194     // OK. Passed in snapshot is same as current snapshot. If not-empty,
195     // create a new snapshot and let the old one go.
196     if (!this.snapshot.isEmpty()) {
197       this.snapshot = new CellSkipListSet(this.comparator);
198       this.snapshotTimeRangeTracker = new TimeRangeTracker();
199     }
200     this.snapshotSize = 0;
201     this.snapshotId = -1;
202     if (this.snapshotAllocator != null) {
203       tmpAllocator = this.snapshotAllocator;
204       this.snapshotAllocator = null;
205     }
206     if (tmpAllocator != null) {
207       tmpAllocator.close();
208     }
209   }

237   private boolean addToCellSet(Cell e) {
238     boolean b = this.cellSet.add(e);
239     // In no tags case this NoTagsKeyValue.getTagsLength() is a cheap call.
240     // When we use ACL CP or Visibility CP which deals with Tags during
241     // mutation, the TagRewriteCell.getTagsLength() is a cheaper call. We do not
242     // parse the byte[] to identify the tags length.
243     if(e.getTagsLength() > 0) {
244       tagsPresent = true;
245     }
246     setOldestEditTimeToNow();
247     return b;
248   }
249 
250   private boolean removeFromCellSet(Cell e) {
251     boolean b = this.cellSet.remove(e);
252     setOldestEditTimeToNow();
253     return b;
254   }

256   void setOldestEditTimeToNow() {
257     if (timeOfOldestEdit == Long.MAX_VALUE) {
258       timeOfOldestEdit = EnvironmentEdgeManager.currentTime();
259     }

262   /**
263    * Internal version of add() that doesn't clone Cells with the
264    * allocator, and doesn't take the lock.
265    *
266    * Callers should ensure they already have the read lock taken
267    */
268   private long internalAdd(final Cell toAdd) {
269     long s = heapSizeChange(toAdd, addToCellSet(toAdd));
270     timeRangeTracker.includeTimestamp(toAdd);
271     this.size.addAndGet(s);
272     return s;
273   }
274 
275   private Cell maybeCloneWithAllocator(Cell cell) {
276     if (allocator == null) {
277       return cell;
278     }
279 
280     int len = KeyValueUtil.length(cell);
281     ByteRange alloc = allocator.allocateBytes(len);
282     if (alloc == null) {
283       // The allocation was too large, allocator decided
284       // not to do anything with it.
285       return cell;
286     }
287     assert alloc.getBytes() != null;
288     KeyValueUtil.appendToByteArray(cell, alloc.getBytes(), alloc.getOffset());
289     KeyValue newKv = new KeyValue(alloc.getBytes(), alloc.getOffset(), len);
290     newKv.setSequenceId(cell.getSequenceId());
291     return newKv;
292   }

294   /**
295    * Remove n key from the memstore. Only cells that have the same key and the
296    * same memstoreTS are removed.  It is ok to not update timeRangeTracker
297    * in this call. It is possible that we can optimize this method by using
298    * tailMap/iterator, but since this method is called rarely (only for
299    * error recovery), we can leave those optimization for the future.
300    * @param cell
301    */
302   @Override
303   public void rollback(Cell cell) {
304     // If the key is in the snapshot, delete it. We should not update
305     // this.size, because that tracks the size of only the memstore and
306     // not the snapshot. The flush of this snapshot to disk has not
307     // yet started because Store.flush() waits for all rwcc transactions to
308     // commit before starting the flush to disk.
309     Cell found = this.snapshot.get(cell);
310     if (found != null && found.getSequenceId() == cell.getSequenceId()) {
311       this.snapshot.remove(cell);
312       long sz = heapSizeChange(cell, true);
313       this.snapshotSize -= sz;
314     }
315     // If the key is in the memstore, delete it. Update this.size.
316     found = this.cellSet.get(cell);
317     if (found != null && found.getSequenceId() == cell.getSequenceId()) {
318       removeFromCellSet(cell);
319       long s = heapSizeChange(cell, true);
320       this.size.addAndGet(-s);
321     }
322   }

324   /**
325    * Write a delete
326    * @param deleteCell
327    * @return approximate size of the passed key and value.
328    */
329   @Override
330   public long delete(Cell deleteCell) {
331     long s = 0;
332     Cell toAdd = maybeCloneWithAllocator(deleteCell);
333     s += heapSizeChange(toAdd, addToCellSet(toAdd));
334     timeRangeTracker.includeTimestamp(toAdd);
335     this.size.addAndGet(s);
336     return s;
337   }

348   /*
349    * @param a
350    * @param b
351    * @return Return lowest of a or b or null if both a and b are null
352    */
353   private Cell getLowest(final Cell a, final Cell b) {
354     if (a == null) {
355       return b;
356     }
357     if (b == null) {
358       return a;
359     }
360     return comparator.compareRows(a, b) <= 0? a: b;
361   }

363   /*
364    * @param key Find row that follows this one.  If null, return first.
365    * @param map Set to look in for a row beyond <code>row</code>.
366    * @return Next row or null if none found.  If one found, will be a new
367    * KeyValue -- can be destroyed by subsequent calls to this method.
368    */
369   private Cell getNextRow(final Cell key,
370       final NavigableSet<Cell> set) {
371     Cell result = null;
372     SortedSet<Cell> tail = key == null? set: set.tailSet(key);
373     // Iterate until we fall into the next row; i.e. move off current row
374     for (Cell cell: tail) {
375       if (comparator.compareRows(cell, key) <= 0)
376         continue;
377       // Note: Not suppressing deletes or expired cells.  Needs to be handled
378       // by higher up functions.
379       result = cell;
380       break;
381     }
382     return result;
383   }

385   /**
386    * Only used by tests. TODO: Remove
387    *
388    * Given the specs of a column, update it, first by inserting a new record,
389    * then removing the old one.  Since there is only 1 KeyValue involved, the memstoreTS
390    * will be set to 0, thus ensuring that they instantly appear to anyone. The underlying
391    * store will ensure that the insert/delete each are atomic. A scanner/reader will either
392    * get the new value, or the old value and all readers will eventually only see the new
393    * value after the old was removed.
394    *
395    * @param row
396    * @param family
397    * @param qualifier
398    * @param newValue
399    * @param now
400    * @return  Timestamp
401    */
402   @Override
403   public long updateColumnValue(byte[] row,
404                                 byte[] family,
405                                 byte[] qualifier,
406                                 long newValue,
407                                 long now) {
408     Cell firstCell = KeyValueUtil.createFirstOnRow(row, family, qualifier);
409     // Is there a Cell in 'snapshot' with the same TS? If so, upgrade the timestamp a bit.
410     SortedSet<Cell> snSs = snapshot.tailSet(firstCell);
411     if (!snSs.isEmpty()) {
412       Cell snc = snSs.first();
413       // is there a matching Cell in the snapshot?
414       if (CellUtil.matchingRow(snc, firstCell) && CellUtil.matchingQualifier(snc, firstCell)) {
415         if (snc.getTimestamp() == now) {
416           // poop,
417           now += 1;
418         }
419       }
420     }
421 
422     // logic here: the new ts MUST be at least 'now'. But it could be larger if necessary.
423     // But the timestamp should also be max(now, mostRecentTsInMemstore)
424 
425     // so we cant add the new Cell w/o knowing what's there already, but we also
426     // want to take this chance to delete some cells. So two loops (sad)
427 
428     SortedSet<Cell> ss = cellSet.tailSet(firstCell);
429     for (Cell cell : ss) {
430       // if this isnt the row we are interested in, then bail:
431       if (!CellUtil.matchingColumn(cell, family, qualifier)
432           || !CellUtil.matchingRow(cell, firstCell)) {
433         break; // rows dont match, bail.
434       }
435 
436       // if the qualifier matches and it's a put, just RM it out of the cellSet.
437       if (cell.getTypeByte() == KeyValue.Type.Put.getCode() &&
438           cell.getTimestamp() > now && CellUtil.matchingQualifier(firstCell, cell)) {
439         now = cell.getTimestamp();
440       }
441     }
442 
443     // create or update (upsert) a new Cell with
444     // 'now' and a 0 memstoreTS == immediately visible
445     List<Cell> cells = new ArrayList<Cell>(1);
446     cells.add(new KeyValue(row, family, qualifier, now, Bytes.toBytes(newValue)));
447     return upsert(cells, 1L);
448   }

450   /**
451    * Update or insert the specified KeyValues.
452    * <p>
453    * For each KeyValue, insert into MemStore.  This will atomically upsert the
454    * value for that row/family/qualifier.  If a KeyValue did already exist,
455    * it will then be removed.
456    * <p>
457    * Currently the memstoreTS is kept at 0 so as each insert happens, it will
458    * be immediately visible.  May want to change this so it is atomic across
459    * all KeyValues.
460    * <p>
461    * This is called under row lock, so Get operations will still see updates
462    * atomically.  Scans will only see each KeyValue update as atomic.
463    *
464    * @param cells
465    * @param readpoint readpoint below which we can safely remove duplicate KVs
466    * @return change in memstore size
467    */
468   @Override
469   public long upsert(Iterable<Cell> cells, long readpoint) {
470     long size = 0;
471     for (Cell cell : cells) {
472       size += upsert(cell, readpoint);
473     }
474     return size;
475   }

477   /**
478    * Inserts the specified KeyValue into MemStore and deletes any existing
479    * versions of the same row/family/qualifier as the specified KeyValue.
480    * <p>
481    * First, the specified KeyValue is inserted into the Memstore.
482    * <p>
483    * If there are any existing KeyValues in this MemStore with the same row,
484    * family, and qualifier, they are removed.
485    * <p>
486    * Callers must hold the read lock.
487    *
488    * @param cell
489    * @return change in size of MemStore
490    */
491   private long upsert(Cell cell, long readpoint) {
492     // Add the Cell to the MemStore
493     // Use the internalAdd method here since we (a) already have a lock
494     // and (b) cannot safely use the MSLAB here without potentially
495     // hitting OOME - see TestMemStore.testUpsertMSLAB for a
496     // test that triggers the pathological case if we don't avoid MSLAB
497     // here.
498     long addedSize = internalAdd(cell);
499 
500     // Get the Cells for the row/family/qualifier regardless of timestamp.
501     // For this case we want to clean up any other puts
502     Cell firstCell = KeyValueUtil.createFirstOnRow(
503         cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
504         cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
505         cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength());
506     SortedSet<Cell> ss = cellSet.tailSet(firstCell);
507     Iterator<Cell> it = ss.iterator();
508     // versions visible to oldest scanner
509     int versionsVisible = 0;
510     while ( it.hasNext() ) {
511       Cell cur = it.next();
512 
513       if (cell == cur) {
514         // ignore the one just put in
515         continue;
516       }
517       // check that this is the row and column we are interested in, otherwise bail
518       if (CellUtil.matchingRow(cell, cur) && CellUtil.matchingQualifier(cell, cur)) {
519         // only remove Puts that concurrent scanners cannot possibly see
520         if (cur.getTypeByte() == KeyValue.Type.Put.getCode() &&
521             cur.getSequenceId() <= readpoint) {
522           if (versionsVisible >= 1) {
523             // if we get here we have seen at least one version visible to the oldest scanner,
524             // which means we can prove that no scanner will see this version
525 
526             // false means there was a change, so give us the size.
527             long delta = heapSizeChange(cur, true);
528             addedSize -= delta;
529             this.size.addAndGet(-delta);
530             it.remove();
531             setOldestEditTimeToNow();
532           } else {
533             versionsVisible++;
534           }
535         }
536       } else {
537         // past the row or column, done
538         break;
539       }
540     }
541     return addedSize;
542   }

552   /**
553    * Check if this memstore may contain the required keys
554    * @param scan scan
555    * @param store holds reference to cf
556    * @param oldestUnexpiredTS
557    * @return False if the key definitely does not exist in this Memstore
558    */
559   public boolean shouldSeek(Scan scan, Store store, long oldestUnexpiredTS) {
560     byte[] cf = store.getFamily().getName();
561     TimeRange timeRange = scan.getColumnFamilyTimeRange().get(cf);
562     if (timeRange == null) {
563       timeRange = scan.getTimeRange();
564     }
565     return (timeRangeTracker.includesTimeRange(timeRange) ||
566         snapshotTimeRangeTracker.includesTimeRange(timeRange))
567         && (Math.max(timeRangeTracker.getMaximumTimestamp(),
568                      snapshotTimeRangeTracker.getMaximumTimestamp()) >=
569             oldestUnexpiredTS);
570   }

572   /*
573    * MemStoreScanner implements the KeyValueScanner.
574    * It lets the caller scan the contents of a memstore -- both current
575    * map and snapshot.
576    * This behaves as if it were a real scanner but does not maintain position.
577    */
578   protected class MemStoreScanner extends NonLazyKeyValueScanner {
579     // Next row information for either cellSet or snapshot
580     private Cell cellSetNextRow = null;
581     private Cell snapshotNextRow = null;
582 
583     // last iterated Cells for cellSet and snapshot (to restore iterator state after reseek)
584     private Cell cellSetItRow = null;
585     private Cell snapshotItRow = null;
586     
587     // iterator based scanning.
588     private Iterator<Cell> cellSetIt;
589     private Iterator<Cell> snapshotIt;
590 
591     // The cellSet and snapshot at the time of creating this scanner
592     private CellSkipListSet cellSetAtCreation;
593     private CellSkipListSet snapshotAtCreation;
594 
595     // the pre-calculated Cell to be returned by peek() or next()
596     private Cell theNext;
597 
598     // The allocator and snapshot allocator at the time of creating this scanner
599     volatile MemStoreLAB allocatorAtCreation;
600     volatile MemStoreLAB snapshotAllocatorAtCreation;
601     
602     // A flag represents whether could stop skipping Cells for MVCC
603     // if have encountered the next row. Only used for reversed scan
604     private boolean stopSkippingCellsIfNextRow = false;
605 
606     private long readPoint;
607 
608     /*
609     Some notes...
610 
611      So memstorescanner is fixed at creation time. this includes pointers/iterators into
612     existing kvset/snapshot.  during a snapshot creation, the kvset is null, and the
613     snapshot is moved.  since kvset is null there is no point on reseeking on both,
614       we can save us the trouble. During the snapshot->hfile transition, the memstore
615       scanner is re-created by StoreScanner#updateReaders().  StoreScanner should
616       potentially do something smarter by adjusting the existing memstore scanner.
617 
618       But there is a greater problem here, that being once a scanner has progressed
619       during a snapshot scenario, we currently iterate past the kvset then 'finish' up.
620       if a scan lasts a little while, there is a chance for new entries in kvset to
621       become available but we will never see them.  This needs to be handled at the
622       StoreScanner level with coordination with MemStoreScanner.
623 
624       Currently, this problem is only partly managed: during the small amount of time
625       when the StoreScanner has not yet created a new MemStoreScanner, we will miss
626       the adds to kvset in the MemStoreScanner.
627     */
628 
629     MemStoreScanner(long readPoint) {
630       super();
631 
632       this.readPoint = readPoint;
633       cellSetAtCreation = cellSet;
634       snapshotAtCreation = snapshot;
635       if (allocator != null) {
636         this.allocatorAtCreation = allocator;
637         this.allocatorAtCreation.incScannerCount();
638       }
639       if (snapshotAllocator != null) {
640         this.snapshotAllocatorAtCreation = snapshotAllocator;
641         this.snapshotAllocatorAtCreation.incScannerCount();
642       }
643       if (Trace.isTracing() && Trace.currentSpan() != null) {
644         Trace.currentSpan().addTimelineAnnotation("Creating MemStoreScanner");
645       }
646     }

648     /**
649      * Lock on 'this' must be held by caller.
650      * @param it
651      * @return Next Cell
652      */
653     private Cell getNext(Iterator<Cell> it) {
654       Cell startCell = theNext;
655       Cell v = null;
656       try {
657         while (it.hasNext()) {
658           v = it.next();
659           if (v.getSequenceId() <= this.readPoint) {
660             return v;
661           }
662           if (stopSkippingCellsIfNextRow && startCell != null
663               && comparator.compareRows(v, startCell) > 0) {
664             return null;
665           }
666         }
667 
668         return null;
669       } finally {
670         if (v != null) {
671           // in all cases, remember the last Cell iterated to
672           if (it == snapshotIt) {
673             snapshotItRow = v;
674           } else {
675             cellSetItRow = v;
676           }
677         }
678       }
679     }

683      *  Must be called only once: there is no thread safety between the scanner
684      *   and the memStore.

698       cellSetItRow = null;
699       snapshotItRow = null;

713       theNext = getLowest(cellSetNextRow, snapshotNextRow);

721      * Move forward on the sub-lists set previously by seek.
722      * @param key seek value (should be non-null)
723      * @return true if there is at least one KV to read, false otherwise
724      */
725     @Override
726     public synchronized boolean reseek(Cell key) {
727       /*
728       See HBASE-4195 & HBASE-3855 & HBASE-6591 for the background on this implementation.
729       This code is executed concurrently with flush and puts, without locks.
730       Two points must be known when working on this code:
731       1) It's not possible to use the 'kvTail' and 'snapshot'
732        variables, as they are modified during a flush.
733       2) The ideal implementation for performance would use the sub skip list
734        implicitly pointed by the iterators 'kvsetIt' and
735        'snapshotIt'. Unfortunately the Java API does not offer a method to
736        get it. So we remember the last keys we iterated to and restore
737        the reseeked set to at least that point.
738        */
739       cellSetIt = cellSetAtCreation.tailSet(getHighest(key, cellSetItRow)).iterator();
740       snapshotIt = snapshotAtCreation.tailSet(getHighest(key, snapshotItRow)).iterator();
741 
742       return seekInSubLists(key);
743     }

776     /*
777      * Returns the lower of the two key values, or null if they are both null.
778      * This uses comparator.compare() to compare the KeyValue using the memstore
779      * comparator.
780      */
781     private Cell getLowest(Cell first, Cell second) {
782       if (first == null && second == null) {
783         return null;
784       }
785       if (first != null && second != null) {
786         int compare = comparator.compare(first, second);
787         return (compare <= 0 ? first : second);
788       }
789       return (first != null ? first : second);
790     }

792     /*
793      * Returns the higher of the two cells, or null if they are both null.
794      * This uses comparator.compare() to compare the Cell using the memstore
795      * comparator.
796      */
797     private Cell getHighest(Cell first, Cell second) {
798       if (first == null && second == null) {
799         return null;
800       }
801       if (first != null && second != null) {
802         int compare = comparator.compare(first, second);
803         return (compare > 0 ? first : second);
804       }
805       return (first != null ? first : second);
806     }

856     /**
857      * Separately get the KeyValue before the specified key from kvset and
858      * snapshotset, and use the row of higher one as the previous row of
859      * specified key, then seek to the first KeyValue of previous row
860      */
861     @Override
862     public synchronized boolean seekToPreviousRow(Cell originalKey) {
863       boolean keepSeeking = false;
864       Cell key = originalKey;
865       do {
866         Cell firstKeyOnRow = CellUtil.createFirstOnRow(key);
867         SortedSet<Cell> cellHead = cellSetAtCreation.headSet(firstKeyOnRow);
868         Cell cellSetBeforeRow = cellHead.isEmpty() ? null : cellHead.last();
869         SortedSet<Cell> snapshotHead = snapshotAtCreation
870             .headSet(firstKeyOnRow);
871         Cell snapshotBeforeRow = snapshotHead.isEmpty() ? null : snapshotHead
872             .last();
873         Cell lastCellBeforeRow = getHighest(cellSetBeforeRow, snapshotBeforeRow);
874         if (lastCellBeforeRow == null) {
875           theNext = null;
876           return false;
877         }
878         Cell firstKeyOnPreviousRow = CellUtil.createFirstOnRow(lastCellBeforeRow);
879         this.stopSkippingCellsIfNextRow = true;
880         seek(firstKeyOnPreviousRow);
881         this.stopSkippingCellsIfNextRow = false;
882         if (peek() == null
883             || comparator.compareRows(peek(), firstKeyOnPreviousRow) > 0) {
884           keepSeeking = true;
885           key = firstKeyOnPreviousRow;
886           continue;
887         } else {
888           keepSeeking = false;
889         }
890       } while (keepSeeking);
891       return true;
892     }

894     @Override
895     public synchronized boolean seekToLastRow() {
896       Cell first = cellSetAtCreation.isEmpty() ? null : cellSetAtCreation
897           .last();
898       Cell second = snapshotAtCreation.isEmpty() ? null
899           : snapshotAtCreation.last();
900       Cell higherCell = getHighest(first, second);
901       if (higherCell == null) {
902         return false;
903       }
904       Cell firstCellOnLastRow = CellUtil.createFirstOnRow(higherCell);
905       if (seek(firstCellOnLastRow)) {
906         return true;
907       } else {
908         return seekToPreviousRow(higherCell);
909       }
