71  /**
72   * Provides the basics for a RpcClient implementation like configuration and Logging.
73   * <p>
74   * Locking schema of the current IPC implementation
75   * <ul>
76   * <li>There is a lock in {@link AbstractRpcClient} to protect the fetching or creating
77   * connection.</li>
78   * <li>There is a lock in {@link Call} to make sure that we can only finish the call once.</li>
79   * <li>The same for {@link HBaseRpcController} as {@link Call}. And see the comment of
80   * {@link HBaseRpcController#notifyOnCancel(RpcCallback, HBaseRpcController.CancellationCallback)}
81   * of how to deal with cancel.</li>
82   * <li>For connection implementation, the construction of a connection should be as fast as possible
83   * because the creation is protected under a lock. Connect to remote side when needed. There is no
84   * forced locking schema for a connection implementation.</li>
85   * <li>For the locking order, the {@link Call} and {@link HBaseRpcController}'s lock should be held
86   * at last. So the callbacks in {@link Call} and {@link HBaseRpcController} should be execute
87   * outside the lock in {@link Call} and {@link HBaseRpcController} which means the implementations
88   * of the callbacks are free to hold any lock.</li>
89   * </ul>
90   */

141    * Construct an IPC client for the cluster <code>clusterId</code>
142    * @param conf configuration
143    * @param clusterId the cluster id
144    * @param localAddr client socket bind address.
145    * @param metrics the connection metrics

147   public AbstractRpcClient(Configuration conf, String clusterId, SocketAddress localAddr,
148       MetricsConnection metrics) {
149     this.userProvider = UserProvider.instantiate(conf);
150     this.localAddr = localAddr;
151     this.tcpKeepAlive = conf.getBoolean("hbase.ipc.client.tcpkeepalive", true);
152     this.clusterId = clusterId != null ? clusterId : HConstants.CLUSTER_ID_DEFAULT;
153     this.failureSleep = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
154       HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
155     this.maxRetries = conf.getInt("hbase.ipc.client.connect.max.retries", 0);
156     this.tcpNoDelay = conf.getBoolean("hbase.ipc.client.tcpnodelay", true);
157     this.cellBlockBuilder = new CellBlockBuilder(conf);
158
159     this.minIdleTimeBeforeClose = conf.getInt(IDLE_TIME, 120000); // 2 minutes
160     this.conf = conf;
161     this.codec = getCodec();
162     this.compressor = getCompressor(conf);
163     this.fallbackAllowed = conf.getBoolean(IPC_CLIENT_FALLBACK_TO_SIMPLE_AUTH_ALLOWED_KEY,
164       IPC_CLIENT_FALLBACK_TO_SIMPLE_AUTH_ALLOWED_DEFAULT);
165     this.failedServers = new FailedServers(conf);
166     this.connectTO = conf.getInt(SOCKET_TIMEOUT_CONNECT, DEFAULT_SOCKET_TIMEOUT_CONNECT);
167     this.readTO = conf.getInt(SOCKET_TIMEOUT_READ, DEFAULT_SOCKET_TIMEOUT_READ);
168     this.writeTO = conf.getInt(SOCKET_TIMEOUT_WRITE, DEFAULT_SOCKET_TIMEOUT_WRITE);
169     this.metrics = metrics;
170
171     this.connections = new PoolMap<>(getPoolType(conf), getPoolSize(conf));
172
173     this.cleanupIdleConnectionTask = IDLE_CONN_SWEEPER.scheduleAtFixedRate(new Runnable() {
174
175       @Override
176       public void run() {
177         cleanupIdleConnections();
178       }
179     }, minIdleTimeBeforeClose, minIdleTimeBeforeClose, TimeUnit.MILLISECONDS);
180
181     if (LOG.isDebugEnabled()) {
182       LOG.debug("Codec=" + this.codec + ", compressor=" + this.compressor + ", tcpKeepAlive="
183           + this.tcpKeepAlive + ", tcpNoDelay=" + this.tcpNoDelay + ", connectTO=" + this.connectTO
184           + ", readTO=" + this.readTO + ", writeTO=" + this.writeTO + ", minIdleTimeBeforeClose="
185           + this.minIdleTimeBeforeClose + ", maxRetries=" + this.maxRetries + ", fallbackAllowed="
186           + this.fallbackAllowed + ", bind address="
187           + (this.localAddr != null ? this.localAddr : "null"));
188     }

191   private void cleanupIdleConnections() {
192     long closeBeforeTime = EnvironmentEdgeManager.currentTime() - minIdleTimeBeforeClose;
193     synchronized (connections) {
194       for (T conn : connections.values()) {
195         // remove connection if it has not been chosen by anyone for more than maxIdleTime, and the
196         // connection itself has already shutdown. The latter check is because that we may still
197         // have some pending calls on connection so we should not shutdown the connection outside.
198         // The connection itself will disconnect if there is no pending call for maxIdleTime.
199         if (conn.getLastTouched() < closeBeforeTime && !conn.isActive()) {
200           LOG.info("Cleanup idle connection to " + conn.remoteId().address);
201           connections.removeValue(conn.remoteId(), conn);
202         }
203       }
204     }
205   }

288   private int nextCallId() {
289     int id, next;
290     do {
291       id = callIdCnt.get();
292       next = id < Integer.MAX_VALUE ? id + 1 : 0;
293     } while (!callIdCnt.compareAndSet(id, next));
294     return id;
295   }

297   /**
298    * Make a blocking call. Throws exceptions if there are network problems or if the remote code
299    * threw an exception.
300    * @param ticket Be careful which ticket you pass. A new user will mean a new Connection.
301    *          {@link UserProvider#getCurrent()} makes a new instance of User each time so will be a
302    *          new Connection each time.
303    * @return A pair with the Message response and the Cell data (if any).
304    */
305   private Message callBlockingMethod(Descriptors.MethodDescriptor md, HBaseRpcController hrc,
306       Message param, Message returnType, final User ticket, final InetSocketAddress isa)
307       throws ServiceException {
308     BlockingRpcCallback<Message> done = new BlockingRpcCallback<>();
309     callMethod(md, hrc, param, returnType, ticket, isa, done);
310     Message val;
311     try {
312       val = done.get();
313     } catch (IOException e) {
314       throw new ServiceException(e);
315     }
316     if (hrc.failed()) {
317       throw new ServiceException(hrc.getFailed());
318     } else {
319       return val;
320     }
321   }

323   /**
324    * Get a connection from the pool, or create a new one and add it to the pool. Connections to a
325    * given host/port are reused.
326    */
327   private T getConnection(ConnectionId remoteId) throws IOException {
328     if (failedServers.isFailedServer(remoteId.getAddress())) {
329       if (LOG.isDebugEnabled()) {
330         LOG.debug("Not trying to connect to " + remoteId.address
331             + " this server is in the failed servers list");
332       }
333       throw new FailedServerException(
334           "This server is in the failed servers list: " + remoteId.address);
335     }
336     T conn;
337     synchronized (connections) {
338       if (!running) {
339         throw new StoppedRpcClientException();
340       }
341       conn = connections.get(remoteId);
342       if (conn == null) {
343         conn = createConnection(remoteId);
344         connections.put(remoteId, conn);
345       }
346       conn.setLastTouched(EnvironmentEdgeManager.currentTime());
347     }
348     return conn;
349   }

356   private void onCallFinished(Call call, HBaseRpcController hrc, InetSocketAddress addr,
357       RpcCallback<Message> callback) {
358     call.callStats.setCallTimeMs(EnvironmentEdgeManager.currentTime() - call.getStartTime());
359     if (metrics != null) {
360       metrics.updateRpc(call.md, call.param, call.callStats);
361     }
362     if (LOG.isTraceEnabled()) {
363       LOG.trace(
364         "Call: " + call.md.getName() + ", callTime: " + call.callStats.getCallTimeMs() + "ms");
365     }
366     if (call.error != null) {
367       if (call.error instanceof RemoteException) {
368         call.error.fillInStackTrace();
369         hrc.setFailed(call.error);
370       } else {
371         hrc.setFailed(wrapException(addr, call.error));
372       }
373       callback.run(null);
374     } else {
375       hrc.setDone(call.cells);
376       callback.run(call.response);
377     }
378   }

380   private void callMethod(final Descriptors.MethodDescriptor md, final HBaseRpcController hrc,
381       final Message param, Message returnType, final User ticket, final InetSocketAddress addr,
382       final RpcCallback<Message> callback) {
383     final MetricsConnection.CallStats cs = MetricsConnection.newCallStats();
384     cs.setStartTime(EnvironmentEdgeManager.currentTime());
385     Call call = new Call(nextCallId(), md, param, hrc.cellScanner(), returnType,
386         hrc.getCallTimeout(), hrc.getPriority(), new RpcCallback<Call>() {
387
388           @Override
389           public void run(Call call) {
390             onCallFinished(call, hrc, addr, callback);
391           }
392         }, cs);
393     ConnectionId remoteId = new ConnectionId(ticket, md.getService().getName(), addr);
394     try {
395       T connection = getConnection(remoteId);
396       connection.sendRequest(call, hrc);
397     } catch (Exception e) {
398       call.setException(toIOE(e));
399     }
400   }

402   private InetSocketAddress createAddr(ServerName sn) throws UnknownHostException {
403     InetSocketAddress addr = new InetSocketAddress(sn.getHostname(), sn.getPort());
404     if (addr.isUnresolved()) {

406     }

410   /**
411    * Interrupt the connections to the given ip:port server. This should be called if the server is
412    * known as actually dead. This will not prevent current operation to be retried, and, depending
413    * on their own behavior, they may retry on the same server. This can be a feature, for example at
414    * startup. In any case, they're likely to get connection refused (if the process died) or no
415    * route to host: i.e. their next retries should be faster and with a safe exception.
416    */
417   @Override
418   public void cancelConnections(ServerName sn) {
419     synchronized (connections) {
420       for (T connection : connections.values()) {
421         ConnectionId remoteId = connection.remoteId();
422         if (remoteId.address.getPort() == sn.getPort()
423             && remoteId.address.getHostName().equals(sn.getHostname())) {
424           LOG.info("The server on " + sn.toString() + " is dead - stopping the connection "
425               + connection.remoteId);
426           connection.shutdown();
427         }
428       }
429     }
430   }

434   @Override
435   public void close() {
436     if (LOG.isDebugEnabled()) {
437       LOG.debug("Stopping rpc client");
438     }
439     Collection<T> connToClose;
440     synchronized (connections) {
441       if (!running) {
442         return;
443       }
444       running = false;
445       connToClose = connections.values();
446       connections.clear();
447     }
448     cleanupIdleConnectionTask.cancel(true);
449     for (T conn : connToClose) {
450       conn.shutdown();
451     }
452     closeInternal();
453   }

485     /**
486      * Configure an rpc controller
487      * @param controller to configure
488      * @return configured rpc controller
489      */
490     protected HBaseRpcController configureRpcController(RpcController controller) {
491       HBaseRpcController hrc;
492       // TODO: Ideally we should not use an RpcController other than HBaseRpcController at client
493       // side. And now we may use ServerRpcController.
494       if (controller != null && controller instanceof HBaseRpcController) {
495         hrc = (HBaseRpcController) controller;
496         if (!hrc.hasCallTimeout()) {
497           hrc.setCallTimeout(rpcTimeout);
498         }
499       } else {
500         hrc = new HBaseRpcControllerImpl();
501         hrc.setCallTimeout(rpcTimeout);
502       }
503       return hrc;
504     }
