018package org.apache.hadoop.hbase.master.cleaner;
019
020import static org.apache.hadoop.hbase.HConstants.HBASE_MASTER_LOGCLEANER_PLUGINS;
021
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.LinkedList;
025import java.util.List;
026import java.util.concurrent.LinkedBlockingQueue;
027
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.fs.FileStatus;
030import org.apache.hadoop.fs.FileSystem;
031import org.apache.hadoop.fs.Path;
032import org.apache.hadoop.hbase.Stoppable;
033import org.apache.hadoop.hbase.master.procedure.MasterProcedureUtil;
034import org.apache.hadoop.hbase.wal.AbstractFSWALProvider;
035import org.apache.yetus.audience.InterfaceAudience;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038import org.apache.hbase.thirdparty.com.google.common.annotations.VisibleForTesting;
039
040/**
041 * This Chore, every time it runs, will attempt to delete the WALs and Procedure WALs in the old
042 * logs folder. The WAL is only deleted if none of the cleaner delegates says otherwise.
043 * @see BaseLogCleanerDelegate
044 */
045@InterfaceAudience.Private
046public class LogCleaner extends CleanerChore<BaseLogCleanerDelegate> {
047  private static final Logger LOG = LoggerFactory.getLogger(LogCleaner.class.getName());
048
049  public static final String OLD_WALS_CLEANER_SIZE = "hbase.oldwals.cleaner.thread.size";
050  public static final int OLD_WALS_CLEANER_DEFAULT_SIZE = 2;
051
052  private final LinkedBlockingQueue<CleanerContext> pendingDelete;
053  private List<Thread> oldWALsCleaner;
054
055  /**
056   * @param period the period of time to sleep between each run
057   * @param stopper the stopper
058   * @param conf configuration to use
059   * @param fs handle to the FS
060   * @param oldLogDir the path to the archived logs
061   */
062  public LogCleaner(final int period, final Stoppable stopper, Configuration conf, FileSystem fs,
063      Path oldLogDir) {
064    super("LogsCleaner", period, stopper, conf, fs, oldLogDir, HBASE_MASTER_LOGCLEANER_PLUGINS);
065    this.pendingDelete = new LinkedBlockingQueue<>();
066    int size = conf.getInt(OLD_WALS_CLEANER_SIZE, OLD_WALS_CLEANER_DEFAULT_SIZE);
067    this.oldWALsCleaner = createOldWalsCleaner(size);
068  }
069
070  @Override
071  protected boolean validate(Path file) {
072    return AbstractFSWALProvider.validateWALFilename(file.getName())
073        || MasterProcedureUtil.validateProcedureWALFilename(file.getName());
074  }
075
076  @Override
077  public void onConfigurationChange(Configuration conf) {
078    super.onConfigurationChange(conf);
079
080    int newSize = conf.getInt(OLD_WALS_CLEANER_SIZE, OLD_WALS_CLEANER_DEFAULT_SIZE);
081    if (newSize == oldWALsCleaner.size()) {
082      if (LOG.isDebugEnabled()) {
083        LOG.debug("Size from configuration is the same as previous which is " +
084          newSize + ", no need to update.");
085      }
086      return;
087    }
088    interruptOldWALsCleaner();
089    oldWALsCleaner = createOldWalsCleaner(newSize);
090  }
091
092  @Override
093  protected int deleteFiles(Iterable<FileStatus> filesToDelete) {
094    List<CleanerContext> results = new LinkedList<>();
095    for (FileStatus toDelete : filesToDelete) {
096      CleanerContext context = CleanerContext.createCleanerContext(toDelete);
097      if (context != null) {
098        pendingDelete.add(context);
099        results.add(context);
100      }
101    }
102
103    int deletedFiles = 0;
104    for (CleanerContext res : results) {
105      deletedFiles += res.getResult(500) ? 1 : 0;
106    }
107    return deletedFiles;
108  }
109
110  @Override
111  public synchronized void cleanup() {
112    super.cleanup();
113    interruptOldWALsCleaner();
114  }
115
116  @VisibleForTesting
117  int getSizeOfCleaners() {
118    return oldWALsCleaner.size();
119  }
120
121  private List<Thread> createOldWalsCleaner(int size) {
122    LOG.info("Creating OldWALs cleaners with size=" + size);
123
124    List<Thread> oldWALsCleaner = new ArrayList<>(size);
125    for (int i = 0; i < size; i++) {
126      Thread cleaner = new Thread(() -> deleteFile());
127      cleaner.setName("OldWALsCleaner-" + i);
128      cleaner.setDaemon(true);
129      cleaner.start();
130      oldWALsCleaner.add(cleaner);
131    }
132    return oldWALsCleaner;
133  }
134
135  private void interruptOldWALsCleaner() {
136    for (Thread cleaner : oldWALsCleaner) {
137      cleaner.interrupt();
138    }
139    oldWALsCleaner.clear();
140  }
141
142  private void deleteFile() {
143    while (true) {
144      CleanerContext context = null;
145      boolean succeed = false;
146      boolean interrupted = false;
147      try {
148        context = pendingDelete.take();
149        if (context != null) {
150          FileStatus toClean = context.getTargetToClean();
151          succeed = this.fs.delete(toClean.getPath(), false);
152        }
153      } catch (InterruptedException ite) {
154        // It's most likely from configuration changing request
155        if (context != null) {
156          LOG.warn("Interrupted while cleaning oldWALs " +
157              context.getTargetToClean() + ", try to clean it next round.");
158        }
159        interrupted = true;
160      } catch (IOException e) {
161        // fs.delete() fails.
162        LOG.warn("Failed to clean oldwals with exception: " + e);
163        succeed = false;
164      } finally {
165        if (context != null) {
166          context.setResult(succeed);
167        }
168        if (interrupted) {
169          // Restore interrupt status
170          Thread.currentThread().interrupt();
171          break;
172        }
173      }
174    }
175    if (LOG.isDebugEnabled()) {
176      LOG.debug("Exiting cleaner.");
177    }
178  }
179
180  @Override
181  public synchronized void cancel(boolean mayInterruptIfRunning) {
182    super.cancel(mayInterruptIfRunning);
183    for (Thread t : oldWALsCleaner) {
184      t.interrupt();
185    }
186  }
187
188  private static final class CleanerContext {
189    // At most waits 60 seconds
190    static final long MAX_WAIT = 60 * 1000;
191
192    final FileStatus target;
193    volatile boolean result;
194    volatile boolean setFromCleaner = false;
195
196    static CleanerContext createCleanerContext(FileStatus status) {
197      return status != null ? new CleanerContext(status) : null;
198    }
199
200    private CleanerContext(FileStatus status) {
201      this.target = status;
202      this.result = false;
203    }
204
205    synchronized void setResult(boolean res) {
206      this.result = res;
207      this.setFromCleaner = true;
208      notify();
209    }
210
211    synchronized boolean getResult(long waitIfNotFinished) {
212      long totalTime = 0;
213      try {
214        while (!setFromCleaner) {
215          wait(waitIfNotFinished);
216          totalTime += waitIfNotFinished;
217          if (totalTime >= MAX_WAIT) {
218            LOG.warn("Spend too much time to delete oldwals " + target);
219            return result;
220          }
221        }
222      } catch (InterruptedException e) {
223        LOG.warn("Interrupted while waiting deletion of " + target);
224        return result;
225      }
226      return result;
227    }
228
229    FileStatus getTargetToClean() {
230      return target;
231    }
232  }
233}
