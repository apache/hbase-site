018package org.apache.hadoop.hbase.procedure;
019
020import java.io.IOException;
021
022import org.apache.hadoop.hbase.security.User;
023import org.apache.hadoop.hbase.security.access.AccessChecker;
024import org.apache.yetus.audience.InterfaceAudience;
025import org.apache.yetus.audience.InterfaceStability;
026import org.apache.hadoop.hbase.Stoppable;
027import org.apache.hadoop.hbase.master.MasterServices;
028import org.apache.hadoop.hbase.master.MetricsMaster;
029import org.apache.hadoop.hbase.shaded.protobuf.generated.HBaseProtos.ProcedureDescription;
030import org.apache.zookeeper.KeeperException;
031
032/**
033* A life-cycle management interface for globally barriered procedures on master.
034* See the following doc on details of globally barriered procedure:
035* https://issues.apache.org/jira/secure/attachment/12555103/121127-global-barrier-proc.pdf
036*
037* To implement a custom globally barriered procedure, user needs to extend two classes:
038* {@link MasterProcedureManager} and {@link RegionServerProcedureManager}. Implementation of
039* {@link MasterProcedureManager} is loaded into {@link org.apache.hadoop.hbase.master.HMaster}
040* process via configuration parameter 'hbase.procedure.master.classes', while implementation of
041* {@link RegionServerProcedureManager} is loaded into
042* {@link org.apache.hadoop.hbase.regionserver.HRegionServer} process via
043* configuration parameter 'hbase.procedure.regionserver.classes'.
044*
045* An example of globally barriered procedure implementation is
046* {@link org.apache.hadoop.hbase.master.snapshot.SnapshotManager} and
047* {@link org.apache.hadoop.hbase.regionserver.snapshot.RegionServerSnapshotManager}.
048*
049* A globally barriered procedure is identified by its signature (usually it is the name of the
050* procedure znode). During the initialization phase, the initialize methods are called by both
051* {@link org.apache.hadoop.hbase.master.HMaster}
052* and {@link org.apache.hadoop.hbase.regionserver.HRegionServer} which create the procedure znode
053* and register the listeners. A procedure can be triggered by its signature and an instant name
054* (encapsulated in a {@link ProcedureDescription} object). When the servers are shutdown,
055* the stop methods on both classes are called to clean up the data associated with the procedure.
056*/
057@InterfaceAudience.Private
058public abstract class MasterProcedureManager extends ProcedureManager implements Stoppable {
059  /**
060   * Initialize a globally barriered procedure for master.
061   *
062   * @param master Master service interface
063   * @throws KeeperException
064   * @throws IOException
065   * @throws UnsupportedOperationException
066   */
067  public abstract void initialize(MasterServices master, MetricsMaster metricsMaster)
068      throws KeeperException, IOException, UnsupportedOperationException;
069
070  /**
071   * Execute a distributed procedure on cluster
072   *
073   * @param desc Procedure description
074   * @throws IOException
075   */
076  public void execProcedure(ProcedureDescription desc) throws IOException {}
077
078  /**
079   * Execute a distributed procedure on cluster with return data.
080   *
081   * @param desc Procedure description
082   * @return data returned from the procedure execution, null if no data
083   * @throws IOException
084   */
085  public byte[] execProcedureWithRet(ProcedureDescription desc)
086      throws IOException {
087    return null;
088  }
089
090  /**
091   * Check for required permissions before executing the procedure.
092   * @throws IOException if permissions requirements are not met.
093   */
094  public abstract void checkPermissions(ProcedureDescription desc, AccessChecker accessChecker,
095      User user) throws IOException;
096
097  /**
098   * Check if the procedure is finished successfully
099   *
100   * @param desc Procedure description
101   * @return true if the specified procedure is finished successfully
102   * @throws IOException
103   */
104  public abstract boolean isProcedureDone(ProcedureDescription desc) throws IOException;
105}
