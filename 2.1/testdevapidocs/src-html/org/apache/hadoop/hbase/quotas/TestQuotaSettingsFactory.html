054  @Test
055  public void testAllQuotasAddedToList() {
056    final SpaceQuota spaceQuota = SpaceQuota.newBuilder()
057        .setSoftLimit(1024L * 1024L * 1024L * 50L) // 50G
058        .setViolationPolicy(QuotaProtos.SpaceViolationPolicy.DISABLE) // Disable the table
059        .build();
060    final long readLimit = 1000;
061    final long writeLimit = 500;
062    final Throttle throttle = Throttle.newBuilder()
063        // 1000 read reqs/min
064        .setReadNum(TimedQuota.newBuilder().setSoftLimit(readLimit)
065            .setTimeUnit(HBaseProtos.TimeUnit.MINUTES).build())
066        // 500 write reqs/min
067        .setWriteNum(TimedQuota.newBuilder().setSoftLimit(writeLimit)
068            .setTimeUnit(HBaseProtos.TimeUnit.MINUTES).build())
069        .build();
070    final Quotas quotas = Quotas.newBuilder()
071        .setSpace(spaceQuota) // Set the FS quotas
072        .setThrottle(throttle) // Set some RPC limits
073        .build();
074    final TableName tn = TableName.valueOf("my_table");
075    List<QuotaSettings> settings = QuotaSettingsFactory.fromTableQuotas(tn, quotas);
076    assertEquals(3, settings.size());
077    boolean seenRead = false;
078    boolean seenWrite = false;
079    boolean seenSpace = false;
080    for (QuotaSettings setting : settings) {
081      if (setting instanceof ThrottleSettings) {
082        ThrottleSettings throttleSettings = (ThrottleSettings) setting;
083        switch (throttleSettings.getThrottleType()) {
084          case READ_NUMBER:
085            assertFalse("Should not have multiple read quotas", seenRead);
086            assertEquals(readLimit, throttleSettings.getSoftLimit());
087            assertEquals(TimeUnit.MINUTES, throttleSettings.getTimeUnit());
088            assertEquals(tn, throttleSettings.getTableName());
089            assertNull("Username should be null", throttleSettings.getUserName());
090            assertNull("Namespace should be null", throttleSettings.getNamespace());
091            seenRead = true;
092            break;
093          case WRITE_NUMBER:
094            assertFalse("Should not have multiple write quotas", seenWrite);
095            assertEquals(writeLimit, throttleSettings.getSoftLimit());
096            assertEquals(TimeUnit.MINUTES, throttleSettings.getTimeUnit());
097            assertEquals(tn, throttleSettings.getTableName());
098            assertNull("Username should be null", throttleSettings.getUserName());
099            assertNull("Namespace should be null", throttleSettings.getNamespace());
100            seenWrite = true;
101            break;
102          default:
103            fail("Unexpected throttle type: " + throttleSettings.getThrottleType());
104        }
105      } else if (setting instanceof SpaceLimitSettings) {
106        assertFalse("Should not have multiple space quotas", seenSpace);
107        SpaceLimitSettings spaceLimit = (SpaceLimitSettings) setting;
108        assertEquals(tn, spaceLimit.getTableName());
109        assertNull("Username should be null", spaceLimit.getUserName());
110        assertNull("Namespace should be null", spaceLimit.getNamespace());
111        assertTrue("SpaceLimitSettings should have a SpaceQuota", spaceLimit.getProto().hasQuota());
112        assertEquals(spaceQuota, spaceLimit.getProto().getQuota());
113        seenSpace = true;
114      } else {
115        fail("Unexpected QuotaSettings implementation: " + setting.getClass());
116      }
117    }
118    assertTrue("Should have seen a read quota", seenRead);
119    assertTrue("Should have seen a write quota", seenWrite);
120    assertTrue("Should have seen a space quota", seenSpace);
121  }

141  @Test
142  public void testSpaceLimitSettings() {
143    final TableName tableName = TableName.valueOf("foo");
144    final long sizeLimit = 1024L * 1024L * 1024L * 75; // 75GB
145    final SpaceViolationPolicy violationPolicy = SpaceViolationPolicy.NO_INSERTS;
146    QuotaSettings settings =
147        QuotaSettingsFactory.limitTableSpace(tableName, sizeLimit, violationPolicy);
148    assertNotNull("QuotaSettings should not be null", settings);
149    assertTrue("Should be an instance of SpaceLimitSettings",
150        settings instanceof SpaceLimitSettings);
151    SpaceLimitSettings spaceLimitSettings = (SpaceLimitSettings) settings;
152    SpaceLimitRequest protoRequest = spaceLimitSettings.getProto();
153    assertTrue("Request should have a SpaceQuota", protoRequest.hasQuota());
154    SpaceQuota quota = protoRequest.getQuota();
155    assertEquals(sizeLimit, quota.getSoftLimit());
156    assertEquals(violationPolicy, ProtobufUtil.toViolationPolicy(quota.getViolationPolicy()));
157    assertFalse("The remove attribute should be false", quota.getRemove());
158  }

160  @Test
161  public void testSpaceLimitSettingsForDeletes() {
162    final String ns = "ns1";
163    final TableName tn = TableName.valueOf("tn1");
164    QuotaSettings nsSettings = QuotaSettingsFactory.removeNamespaceSpaceLimit(ns);
165    assertNotNull("QuotaSettings should not be null", nsSettings);
166    assertTrue("Should be an instance of SpaceLimitSettings",
167        nsSettings instanceof SpaceLimitSettings);
168    SpaceLimitRequest nsProto = ((SpaceLimitSettings) nsSettings).getProto();
169    assertTrue("Request should have a SpaceQuota", nsProto.hasQuota());
170    assertTrue("The remove attribute should be true", nsProto.getQuota().getRemove());
171
172    QuotaSettings tableSettings = QuotaSettingsFactory.removeTableSpaceLimit(tn);
173    assertNotNull("QuotaSettings should not be null", tableSettings);
174    assertTrue("Should be an instance of SpaceLimitSettings",
175        tableSettings instanceof SpaceLimitSettings);
176    SpaceLimitRequest tableProto = ((SpaceLimitSettings) tableSettings).getProto();
177    assertTrue("Request should have a SpaceQuota", tableProto.hasQuota());
178    assertTrue("The remove attribute should be true", tableProto.getQuota().getRemove());
179  }
