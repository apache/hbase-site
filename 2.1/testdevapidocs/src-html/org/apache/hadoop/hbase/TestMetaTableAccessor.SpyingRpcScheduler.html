108   * Does {@link MetaTableAccessor#getRegion(Connection, byte[])} and a write
109   * against hbase:meta while its hosted server is restarted to prove our retrying

112  @Test public void testRetrying()
113  throws IOException, InterruptedException {
114    final TableName tableName = TableName.valueOf(name.getMethodName());
115    LOG.info("Started " + tableName);
116    Table t = UTIL.createMultiRegionTable(tableName, HConstants.CATALOG_FAMILY);
117    int regionCount = -1;
118    try (RegionLocator r = UTIL.getConnection().getRegionLocator(tableName)) {
119      regionCount = r.getStartKeys().length;
120    }
121    // Test it works getting a region from just made user table.
122    final List<RegionInfo> regions =
123      testGettingTableRegions(connection, tableName, regionCount);
124    MetaTask reader = new MetaTask(connection, "reader") {
125      @Override
126      void metaTask() throws Throwable {
127        testGetRegion(connection, regions.get(0));
128        LOG.info("Read " + regions.get(0).getEncodedName());
129      }
130    };
131    MetaTask writer = new MetaTask(connection, "writer") {
132      @Override
133      void metaTask() throws Throwable {
134        MetaTableAccessor.addRegionToMeta(connection, regions.get(0));
135        LOG.info("Wrote " + regions.get(0).getEncodedName());
136      }
137    };
138    reader.start();
139    writer.start();
140
141    // We're gonna check how it takes. If it takes too long, we will consider
142    //  it as a fail. We can't put that in the @Test tag as we want to close
143    //  the threads nicely
144    final long timeOut = 180000;
145    long startTime = System.currentTimeMillis();
146
147    try {
148      // Make sure reader and writer are working.
149      assertTrue(reader.isProgressing());
150      assertTrue(writer.isProgressing());
151
152      // Kill server hosting meta -- twice  . See if our reader/writer ride over the
153      // meta moves.  They'll need to retry.
154      for (int i = 0; i < 2; i++) {
155        LOG.info("Restart=" + i);
156        UTIL.ensureSomeRegionServersAvailable(2);
157        int index = -1;
158        do {
159          index = UTIL.getMiniHBaseCluster().getServerWithMeta();
160        } while (index == -1 &&
161          startTime + timeOut < System.currentTimeMillis());
162
163        if (index != -1){
164          UTIL.getMiniHBaseCluster().abortRegionServer(index);
165          UTIL.getMiniHBaseCluster().waitOnRegionServer(index);
166        }
167      }
168
169      assertTrue("reader: " + reader.toString(), reader.isProgressing());
170      assertTrue("writer: " + writer.toString(), writer.isProgressing());
171    } catch (IOException e) {
172      throw e;
173    } finally {
174      reader.stop = true;
175      writer.stop = true;
176      reader.join();
177      writer.join();
178      t.close();
179    }
180    long exeTime = System.currentTimeMillis() - startTime;
181    assertTrue("Timeout: test took " + exeTime / 1000 + " sec", exeTime < timeOut);
182  }

184  /**
185   * Thread that runs a MetaTableAccessor task until asked stop.
186   */
187  abstract static class MetaTask extends Thread {
188    boolean stop = false;
189    int count = 0;
190    Throwable t = null;
191    final Connection connection;
192
193    MetaTask(final Connection connection, final String name) {
194      super(name);
195      this.connection = connection;
196    }
197
198    @Override
199    public void run() {
200      try {
201        while(!this.stop) {
202          LOG.info("Before " + this.getName()+ ", count=" + this.count);
203          metaTask();
204          this.count += 1;
205          LOG.info("After " + this.getName() + ", count=" + this.count);
206          Thread.sleep(100);
207        }
208      } catch (Throwable t) {
209        LOG.info(this.getName() + " failed", t);
210        this.t = t;
211      }
212    }
213
214    boolean isProgressing() throws InterruptedException {
215      int currentCount = this.count;
216      while(currentCount == this.count) {
217        if (!isAlive()) return false;
218        if (this.t != null) return false;
219        Thread.sleep(10);
220      }
221      return true;
222    }
223
224    @Override
225    public String toString() {
226      return "count=" + this.count + ", t=" +
227        (this.t == null? "null": this.t.toString());
228    }
229
230    abstract void metaTask() throws Throwable;
231  }

705    @Override
706    public RpcScheduler create(Configuration conf, PriorityFunction priority, Abortable server) {
707      final RpcScheduler delegate = super.create(conf, priority, server);
708      return new SpyingRpcScheduler(delegate);
709    }

712  public static class SpyingRpcScheduler extends DelegatingRpcScheduler {
713    long numPriorityCalls = 0;
714
715    public SpyingRpcScheduler(RpcScheduler delegate) {
716      super(delegate);
717    }
718
719    @Override
720    public boolean dispatch(CallRunner task) throws IOException, InterruptedException {
721      int priority = task.getRpcCall().getPriority();
722
723      if (priority > HConstants.QOS_THRESHOLD) {
724        numPriorityCalls++;
725      }
726      return super.dispatch(task);
727    }
728  }
