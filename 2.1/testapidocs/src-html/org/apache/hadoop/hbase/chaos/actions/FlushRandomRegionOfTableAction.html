019package org.apache.hadoop.hbase.chaos.actions;
020
021import java.util.List;
022
023import org.apache.hadoop.hbase.HBaseTestingUtility;
024import org.apache.hadoop.hbase.HRegionInfo;
025import org.apache.hadoop.hbase.TableName;
026import org.apache.hadoop.hbase.chaos.monkies.PolicyBasedChaosMonkey;
027import org.apache.hadoop.hbase.client.Admin;
028
029/**
030* Action that tries to flush a random region of a table.
031*/
032public class FlushRandomRegionOfTableAction extends Action {
033  private final long sleepTime;
034  private final TableName tableName;
035
036  public FlushRandomRegionOfTableAction(TableName tableName) {
037   this (-1, tableName);
038  }
039
040  public FlushRandomRegionOfTableAction(int sleepTime, TableName tableName) {
041    this.sleepTime = sleepTime;
042    this.tableName = tableName;
043  }
044
045  @Override
046  public void perform() throws Exception {
047    HBaseTestingUtility util = context.getHBaseIntegrationTestingUtility();
048    Admin admin = util.getAdmin();
049
050    LOG.info("Performing action: Flush random region of table " + tableName);
051    List<HRegionInfo> regions = admin.getTableRegions(tableName);
052    if (regions == null || regions.isEmpty()) {
053      LOG.info("Table " + tableName + " doesn't have regions to flush");
054      return;
055    }
056
057    HRegionInfo region = PolicyBasedChaosMonkey.selectRandomItem(
058      regions.toArray(new HRegionInfo[regions.size()]));
059    LOG.debug("Flushing region " + region.getRegionNameAsString());
060    try {
061      admin.flushRegion(region.getRegionName());
062    } catch (Exception ex) {
063      LOG.warn("Flush failed, might be caused by other chaos: " + ex.getMessage());
064    }
065    if (sleepTime > 0) {
066      Thread.sleep(sleepTime);
067    }
068  }
069}
