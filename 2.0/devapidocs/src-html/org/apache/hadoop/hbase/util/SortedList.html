019package org.apache.hadoop.hbase.util;
020
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.Collections;
024import java.util.Comparator;
025import java.util.Iterator;
026import java.util.List;
027import java.util.ListIterator;
028import java.util.RandomAccess;
029
030/**
031 * Simple sorted list implementation that uses {@link java.util.ArrayList} as
032 * the underlying collection so we can support RandomAccess. All mutations
033 * create a new copy of the <code>ArrayList</code> instance, so can be
034 * expensive. This class is only intended for use on small, very rarely
035 * written collections that expect highly concurrent reads.
036 * <p>
037 * Read operations are performed on a reference to the internal list at the
038 * time of invocation, so will not see any mutations to the collection during
039 * their operation. Iterating over list elements manually using the
040 * RandomAccess pattern involves multiple operations. For this to be safe get
041 * a reference to the internal list first using get(). 
042 * <p>
043 * If constructed with a {@link java.util.Comparator}, the list will be sorted
044 * using the comparator. Adding or changing an element using an index will
045 * trigger a resort.
046 * <p>
047 * Iterators are read-only. They cannot be used to remove elements.
048 */
049@edu.umd.cs.findbugs.annotations.SuppressWarnings(value="UG_SYNC_SET_UNSYNC_GET",
050  justification="TODO: synchronization in here needs review!!!")
051public class SortedList<E> implements List<E>, RandomAccess {
052  private volatile List<E> list;
053  private final Comparator<? super E> comparator;
054
055  /**
056   * Constructs an empty list with the default initial capacity that will be
057   * sorted using the given comparator.
058   *
059   * @param comparator the comparator
060   */
061  public SortedList(Comparator<? super E> comparator) {
062    this.list = Collections.emptyList();
063    this.comparator = comparator;
064  }
065
066  /**
067   * Constructs a list containing the elements of the given collection, in the
068   * order returned by the collection's iterator, that will be sorted with the
069   * given comparator.
070   *
071   * @param c the collection
072   * @param comparator the comparator
073   */
074  public SortedList(Collection<? extends E> c, Comparator<? super E> comparator) {
075    this.list = Collections.unmodifiableList(new ArrayList<E>(c));
076    this.comparator = comparator;
077  }
078
079  /**
080   * Returns a reference to the unmodifiable list currently backing the SortedList.
081   * Changes to the SortedList will not be reflected in this list. Use this
082   * method to get a reference for iterating over using the RandomAccess
083   * pattern.
084   */
085  public List<E> get() { // FindBugs: UG_SYNC_SET_UNSYNC_GET complaint. Fix!!
086    return list;
087  }
088
089  @Override
090  public int size() {
091    return list.size();
092  }
093
094  @Override
095  public boolean isEmpty() {
096    return list.isEmpty();
097  }
098
099  @Override
100  public boolean contains(Object o) {
101    return list.contains(o);
102  }
103
104  @Override
105  public Iterator<E> iterator() {
106    return list.iterator();
107  }
108
109  @Override
110  public Object[] toArray() {
111    return list.toArray();
112  }
113
114  @Override
115  public <T> T[] toArray(T[] a) {
116    return list.toArray(a);
117  }
118
119  @Override
120  public synchronized boolean add(E e) {
121    ArrayList<E> newList = new ArrayList<>(list);
122    boolean changed = newList.add(e);
123    if (changed) {
124      Collections.sort(newList, comparator);
125    }
126    list = Collections.unmodifiableList(newList);
127    return changed;
128  }
129
130  @Override
131  public synchronized boolean remove(Object o) {
132    ArrayList<E> newList = new ArrayList<>(list);
133    // Removals in ArrayList won't break sorting
134    boolean changed = newList.remove(o);
135    list = Collections.unmodifiableList(newList);
136    return changed;
137  }
138
139  @Override
140  public boolean containsAll(Collection<?> c) {
141    return list.containsAll(c);
142  }
143
144  @Override
145  public synchronized boolean addAll(Collection<? extends E> c) {
146    ArrayList<E> newList = new ArrayList<>(list);
147    boolean changed = newList.addAll(c);
148    if (changed) {
149      Collections.sort(newList, comparator);
150    }
151    list = Collections.unmodifiableList(newList);
152    return changed;
153  }
154
155  @Override
156  public synchronized boolean addAll(int index, Collection<? extends E> c) {
157    ArrayList<E> newList = new ArrayList<>(list);
158    boolean changed = newList.addAll(index, c);
159    if (changed) {
160      Collections.sort(newList, comparator);
161    }
162    list = Collections.unmodifiableList(newList);
163    return changed;
164  }
165
166  @Override
167  public synchronized boolean removeAll(Collection<?> c) {
168    ArrayList<E> newList = new ArrayList<>(list);
169    // Removals in ArrayList won't break sorting
170    boolean changed = newList.removeAll(c);
171    list = Collections.unmodifiableList(newList);
172    return changed;
173  }
174
175  @Override
176  public synchronized boolean retainAll(Collection<?> c) {
177    ArrayList<E> newList = new ArrayList<>(list);
178    // Removals in ArrayList won't break sorting
179    boolean changed = newList.retainAll(c);
180    list = Collections.unmodifiableList(newList);
181    return changed;
182  }
183
184  @Override
185  public synchronized void clear() {
186    list = Collections.emptyList();
187  }
188
189  @Override
190  public synchronized E get(int index) {
191    return list.get(index);
192  }
193
194  @Override
195  public synchronized E set(int index, E element) {
196    ArrayList<E> newList = new ArrayList<>(list);
197    E result = newList.set(index, element);
198    Collections.sort(list, comparator);
199    list = Collections.unmodifiableList(newList);
200    return result;
201  }
202
203  @Override
204  public synchronized void add(int index, E element) {
205    ArrayList<E> newList = new ArrayList<>(list);
206    newList.add(index, element);
207    Collections.sort(list, comparator);
208    list = Collections.unmodifiableList(newList);
209  }
210
211  @Override
212  public synchronized E remove(int index) {
213    ArrayList<E> newList = new ArrayList<>(list);
214    // Removals in ArrayList won't break sorting
215    E result = newList.remove(index);
216    list = Collections.unmodifiableList(newList);
217    return result;
218  }
219
220  @Override
221  public int indexOf(Object o) {
222    return list.indexOf(o);
223  }
224
225  @Override
226  public int lastIndexOf(Object o) {
227    return list.lastIndexOf(o);
228  }
229
230  @Override
231  public ListIterator<E> listIterator() {
232    return list.listIterator();
233  }
234
235  @Override
236  public ListIterator<E> listIterator(int index) {
237    return list.listIterator(index);
238  }
239
240  @Override
241  public List<E> subList(int fromIndex, int toIndex) {
242    return list.subList(fromIndex, toIndex);
243  }
244}
