023import java.io.BufferedOutputStream;

039/**
040 * Finds the Jar for a class. If the class is in a directory in the
041 * classpath, it creates a Jar on the fly with the contents of the directory
042 * and returns the path to that Jar. If a Jar is created, it is created in
043 * the system temporary directory.
044 *
045 * This file was forked from hadoop/common/branches/branch-2@1377176.
046 */
047public class JarFinder {
048
049  private static void copyToZipStream(File file, ZipEntry entry,
050                              ZipOutputStream zos) throws IOException {
051    InputStream is = new FileInputStream(file);
052    try {
053      zos.putNextEntry(entry);
054      byte[] arr = new byte[4096];
055      int read = is.read(arr);
056      while (read > -1) {
057        zos.write(arr, 0, read);
058        read = is.read(arr);
059      }
060    } finally {
061      try {
062        is.close();
063      } finally {
064        zos.closeEntry();
065      }
066    }
067  }
068
069  public static void jarDir(File dir, String relativePath, ZipOutputStream zos)
070    throws IOException {
071    Preconditions.checkNotNull(relativePath, "relativePath");
072    Preconditions.checkNotNull(zos, "zos");
073
074    // by JAR spec, if there is a manifest, it must be the first entry in the
075    // ZIP.
076    File manifestFile = new File(dir, JarFile.MANIFEST_NAME);
077    ZipEntry manifestEntry = new ZipEntry(JarFile.MANIFEST_NAME);
078    if (!manifestFile.exists()) {
079      zos.putNextEntry(manifestEntry);
080      new Manifest().write(new BufferedOutputStream(zos));
081      zos.closeEntry();
082    } else {
083      copyToZipStream(manifestFile, manifestEntry, zos);
084    }
085    zos.closeEntry();
086    zipDir(dir, relativePath, zos, true);
087    zos.close();
088  }
089
090  private static void zipDir(File dir, String relativePath, ZipOutputStream zos,
091                             boolean start) throws IOException {
092    String[] dirList = dir.list();
093    if (dirList == null) {
094      return;
095    }
096    for (String aDirList : dirList) {
097      File f = new File(dir, aDirList);
098      if (!f.isHidden()) {
099        if (f.isDirectory()) {
100          if (!start) {
101            ZipEntry dirEntry = new ZipEntry(relativePath + f.getName() + "/");
102            zos.putNextEntry(dirEntry);
103            zos.closeEntry();
104          }
105          String filePath = f.getPath();
106          File file = new File(filePath);
107          zipDir(file, relativePath + f.getName() + "/", zos, false);
108        }
109        else {
110          String path = relativePath + f.getName();
111          if (!path.equals(JarFile.MANIFEST_NAME)) {
112            ZipEntry anEntry = new ZipEntry(path);
113            copyToZipStream(f, anEntry, zos);
114          }
115        }
116      }
117    }
118  }
119
120  private static void createJar(File dir, File jarFile) throws IOException {
121    Preconditions.checkNotNull(dir, "dir");
122    Preconditions.checkNotNull(jarFile, "jarFile");
123    File jarDir = jarFile.getParentFile();
124    if (!jarDir.exists()) {
125      if (!jarDir.mkdirs()) {
126        throw new IOException(MessageFormat.format("could not create dir [{0}]",
127                                                   jarDir));
128      }
129    }
130    try (FileOutputStream fos = new FileOutputStream(jarFile);
131         JarOutputStream jos = new JarOutputStream(fos)) {
132      jarDir(dir, "", jos);
133    }
134  }
135
136  /**
137   * Returns the full path to the Jar containing the class. It always return a
138   * JAR.
139   *
140   * @param klass class.
141   *
142   * @return path to the Jar containing the class.
143   */
144  public static String getJar(Class klass) {
145    Preconditions.checkNotNull(klass, "klass");
146    ClassLoader loader = klass.getClassLoader();
147    if (loader != null) {
148      String class_file = klass.getName().replaceAll("\\.", "/") + ".class";
149      try {
150        for (Enumeration itr = loader.getResources(class_file);
151             itr.hasMoreElements(); ) {
152          URL url = (URL) itr.nextElement();
153          String path = url.getPath();
154          if (path.startsWith("file:")) {
155            path = path.substring("file:".length());
156          }
157          path = URLDecoder.decode(path, "UTF-8");
158          if ("jar".equals(url.getProtocol())) {
159            path = URLDecoder.decode(path, "UTF-8");
160            return path.replaceAll("!.*$", "");
161          }
162          else if ("file".equals(url.getProtocol())) {
163            String klassName = klass.getName();
164            klassName = klassName.replace(".", "/") + ".class";
165            path = path.substring(0, path.length() - klassName.length());
166            File baseDir = new File(path);
167            File testDir = new File(System.getProperty("test.build.dir", "target/test-dir"));
168            testDir = testDir.getAbsoluteFile();
169            if (!testDir.exists()) {
170              testDir.mkdirs();
171            }
172            File tempJar = File.createTempFile("hadoop-", "", testDir);
173            tempJar = new File(tempJar.getAbsolutePath() + ".jar");
174            tempJar.deleteOnExit();
175            createJar(baseDir, tempJar);
176            return tempJar.getAbsolutePath();
177          }
178        }
179      }
180      catch (IOException e) {
181        throw new RuntimeException(e);
182      }
183    }
184    return null;
185  }
