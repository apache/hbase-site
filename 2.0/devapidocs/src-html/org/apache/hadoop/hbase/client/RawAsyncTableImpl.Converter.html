064/**
065 * The implementation of RawAsyncTable.
066 * <p>
067 * The word 'Raw' means that this is a low level class. The returned {@link CompletableFuture} will
068 * be finished inside the rpc framework thread, which means that the callbacks registered to the
069 * {@link CompletableFuture} will also be executed inside the rpc framework thread. So users who use
070 * this class should not try to do time consuming tasks in the callbacks.
071 * @since 2.0.0
072 * @see AsyncTableImpl
073 */

138  private static <REQ, PREQ, PRESP, RESP> CompletableFuture<RESP> call(
139      HBaseRpcController controller, HRegionLocation loc, ClientService.Interface stub, REQ req,
140      Converter<PREQ, byte[], REQ> reqConvert, RpcCall<PRESP, PREQ> rpcCall,
141      Converter<RESP, HBaseRpcController, PRESP> respConverter) {
142    CompletableFuture<RESP> future = new CompletableFuture<>();
143    try {
144      rpcCall.call(stub, controller, reqConvert.convert(loc.getRegion().getRegionName(), req),
145        new RpcCallback<PRESP>() {
146
147          @Override
148          public void run(PRESP resp) {
149            if (controller.failed()) {
150              future.completeExceptionally(controller.getFailed());
151            } else {
152              try {
153                future.complete(respConverter.convert(controller, resp));
154              } catch (IOException e) {
155                future.completeExceptionally(e);
156              }
157            }
158          }
159        });
160    } catch (IOException e) {
161      future.completeExceptionally(e);
162    }
163    return future;
164  }

354  // We need the MultiRequest when constructing the org.apache.hadoop.hbase.client.MultiResponse,
355  // so here I write a new method as I do not want to change the abstraction of call method.
356  private static <RESP> CompletableFuture<RESP> mutateRow(HBaseRpcController controller,
357      HRegionLocation loc, ClientService.Interface stub, RowMutations mutation,
358      Converter<MultiRequest, byte[], RowMutations> reqConvert,
359      Function<Result, RESP> respConverter) {
360    CompletableFuture<RESP> future = new CompletableFuture<>();
361    try {
362      byte[] regionName = loc.getRegion().getRegionName();
363      MultiRequest req = reqConvert.convert(regionName, mutation);
364      stub.multi(controller, req, new RpcCallback<MultiResponse>() {
365
366        @Override
367        public void run(MultiResponse resp) {
368          if (controller.failed()) {
369            future.completeExceptionally(controller.getFailed());
370          } else {
371            try {
372              org.apache.hadoop.hbase.client.MultiResponse multiResp =
373                ResponseConverter.getResults(req, resp, controller.cellScanner());
374              Throwable ex = multiResp.getException(regionName);
375              if (ex != null) {
376                future.completeExceptionally(ex instanceof IOException ? ex
377                  : new IOException(
378                      "Failed to mutate row: " + Bytes.toStringBinary(mutation.getRow()), ex));
379              } else {
380                future.complete(respConverter
381                    .apply((Result) multiResp.getResults().get(regionName).result.get(0)));
382              }
383            } catch (IOException e) {
384              future.completeExceptionally(e);
385            }
386          }
387        }
388      });
389    } catch (IOException e) {
390      future.completeExceptionally(e);
391    }
392    return future;
393  }

406    // always create a new scan object as we may reset the start row later.
407    Scan newScan = ReflectionUtils.newInstance(scan.getClass(), scan);

410    }
411    if (newScan.getMaxResultSize() <= 0) {
412      newScan.setMaxResultSize(defaultScannerMaxResultSize);
413    }

415  }

538  private boolean locateFinished(RegionInfo region, byte[] endKey, boolean endKeyInclusive) {
539    if (isEmptyStopRow(endKey)) {
540      if (isEmptyStopRow(region.getEndKey())) {
541        return true;
542      }
543      return false;
544    } else {
545      if (isEmptyStopRow(region.getEndKey())) {
546        return true;
547      }
548      int c = Bytes.compareTo(endKey, region.getEndKey());
549      // 1. if the region contains endKey
550      // 2. endKey is equal to the region's endKey and we do not want to include endKey.
551      return c < 0 || c == 0 && !endKeyInclusive;
552    }
553  }

555  private <S, R> void onLocateComplete(Function<RpcChannel, S> stubMaker,
556      ServiceCaller<S, R> callable, CoprocessorCallback<R> callback,
557      List<HRegionLocation> locs, byte[] endKey, boolean endKeyInclusive,
558      AtomicBoolean locateFinished, AtomicInteger unfinishedRequest, HRegionLocation loc,
559      Throwable error) {
560    if (error != null) {
561      callback.onError(error);
562      return;
563    }
564    unfinishedRequest.incrementAndGet();
565    RegionInfo region = loc.getRegion();
566    if (locateFinished(region, endKey, endKeyInclusive)) {
567      locateFinished.set(true);
568    } else {
569      conn.getLocator()
570          .getRegionLocation(tableName, region.getEndKey(), RegionLocateType.CURRENT,
571            operationTimeoutNs)
572          .whenComplete((l, e) -> onLocateComplete(stubMaker, callable, callback, locs, endKey,
573            endKeyInclusive, locateFinished, unfinishedRequest, l, e));
574    }
575    coprocessorService(stubMaker, callable, region, region.getStartKey()).whenComplete((r, e) -> {
576      if (e != null) {
577        callback.onRegionError(region, e);
578      } else {
579        callback.onRegionComplete(region, r);
580      }
581      if (unfinishedRequest.decrementAndGet() == 0 && locateFinished.get()) {
582        callback.onComplete();
583      }
584    });
585  }
