058/**
059 * An example for implementing a counter that reads is much less than writes, i.e, write heavy.
060 * <p>
061 * We will convert increment to put, and do aggregating when get. And of course the return value of
062 * increment is useless then.
063 * <p>
064 * Notice that this is only an example so we do not handle most corner cases, for example, you must
065 * provide a qualifier when doing a get.
066 */
067public class WriteHeavyIncrementObserver implements RegionCoprocessor, RegionObserver {
068
069  @Override
070  public Optional<RegionObserver> getRegionObserver() {
071    return Optional.of(this);
072  }
073
074  @Override
075  public void preFlushScannerOpen(ObserverContext<RegionCoprocessorEnvironment> c, Store store,
076      ScanOptions options, FlushLifeCycleTracker tracker) throws IOException {
077    options.readAllVersions();
078  }

080  private Cell createCell(byte[] row, byte[] family, byte[] qualifier, long ts, long value) {
081    return CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setRow(row)
082        .setType(Cell.Type.Put).setFamily(family).setQualifier(qualifier)
083        .setTimestamp(ts).setValue(Bytes.toBytes(value)).build();
084  }
085
086  private InternalScanner wrap(byte[] family, InternalScanner scanner) {
087    return new InternalScanner() {
088
089      private List<Cell> srcResult = new ArrayList<>();
090
091      private byte[] row;
092
093      private byte[] qualifier;
094
095      private long timestamp;
096
097      private long sum;
098
099      @Override
100      public boolean next(List<Cell> result, ScannerContext scannerContext) throws IOException {
101        boolean moreRows = scanner.next(srcResult, scannerContext);
102        if (srcResult.isEmpty()) {
103          if (!moreRows && row != null) {
104            result.add(createCell(row, family, qualifier, timestamp, sum));
105          }
106          return moreRows;
107        }
108        Cell firstCell = srcResult.get(0);
109        // Check if there is a row change first. All the cells will come from the same row so just
110        // check the first one once is enough.
111        if (row == null) {
112          row = CellUtil.cloneRow(firstCell);
113          qualifier = CellUtil.cloneQualifier(firstCell);
114        } else if (!CellUtil.matchingRows(firstCell, row)) {
115          result.add(createCell(row, family, qualifier, timestamp, sum));
116          row = CellUtil.cloneRow(firstCell);
117          qualifier = CellUtil.cloneQualifier(firstCell);
118          sum = 0;
119        }
120        srcResult.forEach(c -> {
121          if (CellUtil.matchingQualifier(c, qualifier)) {
122            sum += Bytes.toLong(c.getValueArray(), c.getValueOffset());
123          } else {
124            result.add(createCell(row, family, qualifier, timestamp, sum));
125            qualifier = CellUtil.cloneQualifier(c);
126            sum = Bytes.toLong(c.getValueArray(), c.getValueOffset());
127          }
128          timestamp = c.getTimestamp();
129        });
130        if (!moreRows) {
131          result.add(createCell(row, family, qualifier, timestamp, sum));
132        }
133        srcResult.clear();
134        return moreRows;
135      }
136
137      @Override
138      public void close() throws IOException {
139        scanner.close();
140      }
141    };
142  }

178  @Override
179  public void preGetOp(ObserverContext<RegionCoprocessorEnvironment> c, Get get, List<Cell> result)
180      throws IOException {
181    Scan scan =
182        new Scan().withStartRow(get.getRow()).withStopRow(get.getRow(), true).readAllVersions();
183    NavigableMap<byte[], NavigableMap<byte[], MutableLong>> sums =
184        new TreeMap<>(Bytes.BYTES_COMPARATOR);
185    get.getFamilyMap().forEach((cf, cqs) -> {
186      NavigableMap<byte[], MutableLong> ss = new TreeMap<>(Bytes.BYTES_COMPARATOR);
187      sums.put(cf, ss);
188      cqs.forEach(cq -> {
189        ss.put(cq, new MutableLong(0));
190        scan.addColumn(cf, cq);
191      });
192    });
193    List<Cell> cells = new ArrayList<>();
194    try (RegionScanner scanner = c.getEnvironment().getRegion().getScanner(scan)) {
195      boolean moreRows;
196      do {
197        moreRows = scanner.next(cells);
198        for (Cell cell : cells) {
199          byte[] family = CellUtil.cloneFamily(cell);
200          byte[] qualifier = CellUtil.cloneQualifier(cell);
201          long value = Bytes.toLong(cell.getValueArray(), cell.getValueOffset());
202          sums.get(family).get(qualifier).add(value);
203        }
204        cells.clear();
205      } while (moreRows);
206    }
207    sums.forEach((cf, m) -> m.forEach((cq, s) -> result
208        .add(createCell(get.getRow(), cf, cq, HConstants.LATEST_TIMESTAMP, s.longValue()))));
209    c.bypass();
210  }
211
212  private final int mask;
213  private final MutableLong[] lastTimestamps;
214  {
215    int stripes =
216        1 << IntMath.log2(Runtime.getRuntime().availableProcessors(), RoundingMode.CEILING);
217    lastTimestamps =
218        IntStream.range(0, stripes).mapToObj(i -> new MutableLong()).toArray(MutableLong[]::new);
219    mask = stripes - 1;
220  }
221
222  // We need make sure the different put uses different timestamp otherwise we may lost some
223  // increments. This is a known issue for HBase.
224  private long getUniqueTimestamp(byte[] row) {
225    int slot = Bytes.hashCode(row) & mask;
226    MutableLong lastTimestamp = lastTimestamps[slot];
227    long now = System.currentTimeMillis();
228    synchronized (lastTimestamp) {
229      long pt = lastTimestamp.longValue() >> 10;
230      if (now > pt) {
231        lastTimestamp.setValue(now << 10);
232      } else {
233        lastTimestamp.increment();
234      }
235      return lastTimestamp.longValue();
236    }
237  }
238
239  @Override
240  public Result preIncrement(ObserverContext<RegionCoprocessorEnvironment> c, Increment increment)
241      throws IOException {
242    byte[] row = increment.getRow();
243    Put put = new Put(row);
244    long ts = getUniqueTimestamp(row);
245    for (Map.Entry<byte[], List<Cell>> entry : increment.getFamilyCellMap().entrySet()) {
246      for (Cell cell : entry.getValue()) {
247        put.add(CellBuilderFactory.create(CellBuilderType.SHALLOW_COPY).setRow(row)
248            .setFamily(cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength())
249            .setQualifier(cell.getQualifierArray(), cell.getQualifierOffset(),
250              cell.getQualifierLength())
251            .setValue(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength())
252            .setType(Cell.Type.Put).setTimestamp(ts).build());
253      }
254    }
255    c.getEnvironment().getRegion().put(put);
256    c.bypass();
257    return Result.EMPTY_RESULT;
258  }
