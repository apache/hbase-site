058  private static final TableName[] TABLES = {
059      TableName.valueOf("restartTableOne"),
060      TableName.valueOf("restartTableTwo"),
061      TableName.valueOf("restartTableThree")
062  };
063  private static final byte [] FAMILY = Bytes.toBytes("family");

069  @Test
070  public void testClusterRestart() throws Exception {
071    UTIL.startMiniCluster(3);
072    while (!UTIL.getMiniHBaseCluster().getMaster().isInitialized()) {
073      Threads.sleep(1);
074    }
075    LOG.info("\n\nCreating tables");
076    for(TableName TABLE : TABLES) {
077      UTIL.createTable(TABLE, FAMILY);
078    }
079    for(TableName TABLE : TABLES) {
080      UTIL.waitTableEnabled(TABLE);
081    }
082
083    List<RegionInfo> allRegions = MetaTableAccessor.getAllRegions(UTIL.getConnection(), false);
084    assertEquals(4, allRegions.size());
085
086    LOG.info("\n\nShutting down cluster");
087    UTIL.shutdownMiniHBaseCluster();
088
089    LOG.info("\n\nSleeping a bit");
090    Thread.sleep(2000);
091
092    LOG.info("\n\nStarting cluster the second time");
093    UTIL.restartHBaseCluster(3);
094
095    // Need to use a new 'Configuration' so we make a new Connection.
096    // Otherwise we're reusing an Connection that has gone stale because
097    // the shutdown of the cluster also called shut of the connection.
098    allRegions = MetaTableAccessor.getAllRegions(UTIL.getConnection(), false);
099    assertEquals(4, allRegions.size());
100    LOG.info("\n\nWaiting for tables to be available");
101    for(TableName TABLE: TABLES) {
102      try {
103        UTIL.createTable(TABLE, FAMILY);
104        assertTrue("Able to create table that should already exist", false);
105      } catch(TableExistsException tee) {
106        LOG.info("Table already exists as expected");
107      }
108      UTIL.waitTableAvailable(TABLE);
109    }
110  }

112  /**
113   * This tests retaining assignments on a cluster restart
114   */
115  @Test
116  public void testRetainAssignmentOnRestart() throws Exception {
117    UTIL.startMiniCluster(2);
118    while (!UTIL.getMiniHBaseCluster().getMaster().isInitialized()) {
119      Threads.sleep(1);
120    }
121    // Turn off balancer
122    UTIL.getMiniHBaseCluster().getMaster().
123      getMasterRpcServices().synchronousBalanceSwitch(false);
124    LOG.info("\n\nCreating tables");
125    for(TableName TABLE : TABLES) {
126      UTIL.createTable(TABLE, FAMILY);
127    }
128    for(TableName TABLE : TABLES) {
129      UTIL.waitTableEnabled(TABLE);
130    }
131
132    HMaster master = UTIL.getMiniHBaseCluster().getMaster();
133    UTIL.waitUntilNoRegionsInTransition(120000);
134
135    // We don't have to use SnapshotOfRegionAssignmentFromMeta.
136    // We use it here because AM used to use it to load all user region placements
137    SnapshotOfRegionAssignmentFromMeta snapshot = new SnapshotOfRegionAssignmentFromMeta(
138      master.getConnection());
139    snapshot.initialize();
140    Map<RegionInfo, ServerName> regionToRegionServerMap
141      = snapshot.getRegionToRegionServerMap();
142
143    MiniHBaseCluster cluster = UTIL.getHBaseCluster();
144    List<JVMClusterUtil.RegionServerThread> threads = cluster.getLiveRegionServerThreads();
145    assertEquals(2, threads.size());
146    int[] rsPorts = new int[3];
147    for (int i = 0; i < 2; i++) {
148      rsPorts[i] = threads.get(i).getRegionServer().getServerName().getPort();
149    }
150    rsPorts[2] = cluster.getMaster().getServerName().getPort();
151    for (ServerName serverName: regionToRegionServerMap.values()) {
152      boolean found = false; // Test only, no need to optimize
153      for (int k = 0; k < 3 && !found; k++) {
154        found = serverName.getPort() == rsPorts[k];
155      }
156      assertTrue(found);
157    }
158
159    LOG.info("\n\nShutting down HBase cluster");
160    cluster.shutdown();
161    cluster.waitUntilShutDown();
162
163    LOG.info("\n\nSleeping a bit");
164    Thread.sleep(2000);
165
166    LOG.info("\n\nStarting cluster the second time with the same ports");
167    try {
168      cluster.getConf().setInt(
169          ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, 3);
170      master = cluster.startMaster().getMaster();
171      for (int i = 0; i < 3; i++) {
172        cluster.getConf().setInt(HConstants.REGIONSERVER_PORT, rsPorts[i]);
173        cluster.startRegionServer();
174      }
175    } finally {
176      // Reset region server port so as not to conflict with other tests
177      cluster.getConf().setInt(HConstants.REGIONSERVER_PORT, 0);
178      cluster.getConf().setInt(
179        ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, 2);
180    }
181
182    // Make sure live regionservers are on the same host/port
183    List<ServerName> localServers = master.getServerManager().getOnlineServersList();
184    assertEquals(3, localServers.size());
185    for (int i = 0; i < 3; i++) {
186      boolean found = false;
187      for (ServerName serverName: localServers) {
188        if (serverName.getPort() == rsPorts[i]) {
189          found = true;
190          break;
191        }
192      }
193      assertTrue(found);
194    }
195
196    // Wait till master is initialized and all regions are assigned
197    RegionStates regionStates = master.getAssignmentManager().getRegionStates();
198    int expectedRegions = regionToRegionServerMap.size() + 1;
199    while (!master.isInitialized()
200        || regionStates.getRegionAssignments().size() != expectedRegions) {
201      Threads.sleep(100);
202    }
203
204    snapshot = new SnapshotOfRegionAssignmentFromMeta(master.getConnection());
205    snapshot.initialize();
206    Map<RegionInfo, ServerName> newRegionToRegionServerMap =
207      snapshot.getRegionToRegionServerMap();
208    assertEquals(regionToRegionServerMap.size(), newRegionToRegionServerMap.size());
209    for (Map.Entry<RegionInfo, ServerName> entry: newRegionToRegionServerMap.entrySet()) {
210      if (TableName.NAMESPACE_TABLE_NAME.equals(entry.getKey().getTable())) continue;
211      ServerName oldServer = regionToRegionServerMap.get(entry.getKey());
212      ServerName currentServer = entry.getValue();
213      LOG.info("Key=" + entry.getKey() + " oldServer=" + oldServer + ", currentServer=" + currentServer);
214      assertEquals(entry.getKey().toString(), oldServer.getAddress(), currentServer.getAddress());
215      assertNotEquals(oldServer.getStartcode(), currentServer.getStartcode());
216    }
217  }
