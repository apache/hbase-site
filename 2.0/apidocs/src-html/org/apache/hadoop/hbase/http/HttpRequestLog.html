018package org.apache.hadoop.hbase.http;
019
020import java.util.HashMap;
021
022import org.apache.commons.logging.LogConfigurationException;
023import org.apache.commons.logging.impl.Log4JLogger;
024import org.apache.log4j.Appender;
025import org.apache.log4j.LogManager;
026import org.eclipse.jetty.server.NCSARequestLog;
027import org.eclipse.jetty.server.RequestLog;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030import org.slf4j.impl.Log4jLoggerAdapter;
031
032/**
033 * RequestLog object for use with Http
034 */
035public class HttpRequestLog {
036
037  private static final Logger LOG = LoggerFactory.getLogger(HttpRequestLog.class);
038  private static final HashMap<String, String> serverToComponent;
039
040  static {
041    serverToComponent = new HashMap<>();
042    serverToComponent.put("master", "master");
043    serverToComponent.put("region", "regionserver");
044  }
045
046  private static org.apache.log4j.Logger getLog4jLogger(String loggerName) {
047    Logger logger = LoggerFactory.getLogger(loggerName);
048
049    if (logger instanceof Log4JLogger) {
050      Log4JLogger httpLog4JLog = (Log4JLogger)logger;
051      return httpLog4JLog.getLogger();
052    } else if (logger instanceof Log4jLoggerAdapter) {
053      return LogManager.getLogger(loggerName);
054    } else {
055      return null;
056    }
057  }
058
059  public static RequestLog getRequestLog(String name) {
060
061    String lookup = serverToComponent.get(name);
062    if (lookup != null) {
063      name = lookup;
064    }
065    String loggerName = "http.requests." + name;
066    String appenderName = name + "requestlog";
067
068    org.apache.log4j.Logger httpLogger = getLog4jLogger(loggerName);
069
070    if (httpLogger == null) {
071      LOG.warn("Jetty request log can only be enabled using Log4j");
072      return null;
073    }
074
075    Appender appender = null;
076
077    try {
078      appender = httpLogger.getAppender(appenderName);
079    } catch (LogConfigurationException e) {
080      LOG.warn("Http request log for " + loggerName
081          + " could not be created");
082      throw e;
083    }
084
085    if (appender == null) {
086      LOG.info("Http request log for " + loggerName
087          + " is not defined");
088      return null;
089    }
090
091    if (appender instanceof HttpRequestLogAppender) {
092      HttpRequestLogAppender requestLogAppender
093        = (HttpRequestLogAppender)appender;
094      NCSARequestLog requestLog = new NCSARequestLog();
095      requestLog.setFilename(requestLogAppender.getFilename());
096      requestLog.setRetainDays(requestLogAppender.getRetainDays());
097      return requestLog;
098    } else {
099      LOG.warn("Jetty request log for " + loggerName
100          + " was of the wrong class");
101      return null;
102    }
103  }
104}
