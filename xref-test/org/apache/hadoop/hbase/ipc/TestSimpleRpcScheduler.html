116   public void testHandlerIsolation() throws IOException, InterruptedException {
117     CallRunner generalTask = createMockTask();
118     CallRunner priorityTask = createMockTask();
119     CallRunner replicationTask = createMockTask();
120     List<CallRunner> tasks = ImmutableList.of(
121         generalTask,
122         priorityTask,
123         replicationTask);
124     Map<CallRunner, Integer> qos = ImmutableMap.of(
125         generalTask, 0,
126         priorityTask, HConstants.HIGH_QOS + 1,
127         replicationTask, HConstants.REPLICATION_QOS);
128     PriorityFunction qosFunction = mock(PriorityFunction.class);
129     final Map<CallRunner, Thread> handlerThreads = Maps.newHashMap();
130     final CountDownLatch countDownLatch = new CountDownLatch(tasks.size());
131     Answer<Void> answerToRun = new Answer<Void>() {
132       @Override
133       public Void answer(InvocationOnMock invocationOnMock) throws Throwable {
134         synchronized (handlerThreads) {
135           handlerThreads.put(
136               (CallRunner) invocationOnMock.getMock(),
137               Thread.currentThread());
138         }
139         countDownLatch.countDown();
140         return null;
141       }
142     };
143     for (CallRunner task : tasks) {
144       task.setStatus(new MonitoredRPCHandlerImpl());
145       doAnswer(answerToRun).when(task).run();
146     }
147
148     RpcScheduler scheduler = new SimpleRpcScheduler(
149         conf, 1, 1 ,1, qosFunction, HConstants.HIGH_QOS);
150     scheduler.init(CONTEXT);
151     scheduler.start();
152     for (CallRunner task : tasks) {
153       when(qosFunction.getPriority((RPCProtos.RequestHeader) anyObject(),
154         (Message) anyObject(), (User) anyObject()))
155           .thenReturn(qos.get(task));
156       scheduler.dispatch(task);
157     }
158     for (CallRunner task : tasks) {
159       verify(task, timeout(1000)).run();
160     }
161     scheduler.stop();
162
163     // Tests that these requests are handled by three distinct threads.
164     countDownLatch.await();
165     assertEquals(3, ImmutableSet.copyOf(handlerThreads.values()).size());
166   }

181   private void testRpcScheduler(final String queueType) throws Exception {
182     Configuration schedConf = HBaseConfiguration.create();
183     schedConf.set(SimpleRpcScheduler.CALL_QUEUE_TYPE_CONF_KEY, queueType);
184
185     PriorityFunction priority = mock(PriorityFunction.class);
186     when(priority.getPriority(any(RequestHeader.class),
187       any(Message.class), any(User.class)))
188       .thenReturn(HConstants.NORMAL_QOS);
189
190     RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 1, 1, 1, priority,
191                                                     HConstants.QOS_THRESHOLD);
192     try {
193       scheduler.start();
194
195       CallRunner smallCallTask = mock(CallRunner.class);
196       RpcServer.Call smallCall = mock(RpcServer.Call.class);
197       RequestHeader smallHead = RequestHeader.newBuilder().setCallId(1).build();
198       when(smallCallTask.getCall()).thenReturn(smallCall);
199       when(smallCall.getHeader()).thenReturn(smallHead);
200
201       CallRunner largeCallTask = mock(CallRunner.class);
202       RpcServer.Call largeCall = mock(RpcServer.Call.class);
203       RequestHeader largeHead = RequestHeader.newBuilder().setCallId(50).build();
204       when(largeCallTask.getCall()).thenReturn(largeCall);
205       when(largeCall.getHeader()).thenReturn(largeHead);
206
207       CallRunner hugeCallTask = mock(CallRunner.class);
208       RpcServer.Call hugeCall = mock(RpcServer.Call.class);
209       RequestHeader hugeHead = RequestHeader.newBuilder().setCallId(100).build();
210       when(hugeCallTask.getCall()).thenReturn(hugeCall);
211       when(hugeCall.getHeader()).thenReturn(hugeHead);
212 
213       when(priority.getDeadline(eq(smallHead), any(Message.class))).thenReturn(0L);
214       when(priority.getDeadline(eq(largeHead), any(Message.class))).thenReturn(50L);
215       when(priority.getDeadline(eq(hugeHead), any(Message.class))).thenReturn(100L);
216
217       final ArrayList<Integer> work = new ArrayList<Integer>();
218       doAnswerTaskExecution(smallCallTask, work, 10, 250);
219       doAnswerTaskExecution(largeCallTask, work, 50, 250);
220       doAnswerTaskExecution(hugeCallTask, work, 100, 250);
221
222       scheduler.dispatch(smallCallTask);
223       scheduler.dispatch(smallCallTask);
224       scheduler.dispatch(smallCallTask);
225       scheduler.dispatch(hugeCallTask);
226       scheduler.dispatch(smallCallTask);
227       scheduler.dispatch(largeCallTask);
228       scheduler.dispatch(smallCallTask);
229       scheduler.dispatch(smallCallTask);
230 
231       while (work.size() < 8) {
232         Thread.sleep(100);
233       }
234
235       int seqSum = 0;
236       int totalTime = 0;
237       for (int i = 0; i < work.size(); ++i) {
238         LOG.debug("Request i=" + i + " value=" + work.get(i));
239         seqSum += work.get(i);
240         totalTime += seqSum;
241       }
242       LOG.debug("Total Time: " + totalTime);
243
244       // -> [small small small huge small large small small]
245       // -> NO REORDER   [10 10 10 100 10 50 10 10] -> 930 (FIFO Queue)
246       // -> WITH REORDER [10 10 10 10 10 10 50 100] -> 530 (Deadline Queue)
247       if (queueType.equals(SimpleRpcScheduler.CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE)) {
248         assertEquals(530, totalTime);
249       } else if (queueType.equals(SimpleRpcScheduler.CALL_QUEUE_TYPE_FIFO_CONF_VALUE)) {
250         assertEquals(930, totalTime);
251       }
252     } finally {
253       scheduler.stop();
254     }
255   }

275     Configuration schedConf = HBaseConfiguration.create();
276     schedConf.setFloat(SimpleRpcScheduler.CALL_QUEUE_HANDLER_FACTOR_CONF_KEY, 1.0f);
277     schedConf.setFloat(SimpleRpcScheduler.CALL_QUEUE_READ_SHARE_CONF_KEY, 0.7f);
278     schedConf.setFloat(SimpleRpcScheduler.CALL_QUEUE_SCAN_SHARE_CONF_KEY, 0.5f);
279 
280     PriorityFunction priority = mock(PriorityFunction.class);
281     when(priority.getPriority(any(RPCProtos.RequestHeader.class), any(Message.class),
282       any(User.class))).thenReturn(HConstants.NORMAL_QOS);
283
284     RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 3, 1, 1, priority,
285                                                     HConstants.QOS_THRESHOLD);
286     try {
287       scheduler.start();
288
289       CallRunner putCallTask = mock(CallRunner.class);
290       RpcServer.Call putCall = mock(RpcServer.Call.class);
291       putCall.param = RequestConverter.buildMutateRequest(
292           Bytes.toBytes("abc"), new Put(Bytes.toBytes("row")));
293       RequestHeader putHead = RequestHeader.newBuilder().setMethodName("mutate").build();
294       when(putCallTask.getCall()).thenReturn(putCall);
295       when(putCall.getHeader()).thenReturn(putHead);
296
297       CallRunner getCallTask = mock(CallRunner.class);
298       RpcServer.Call getCall = mock(RpcServer.Call.class);
299       RequestHeader getHead = RequestHeader.newBuilder().setMethodName("get").build();
300       when(getCallTask.getCall()).thenReturn(getCall);
301       when(getCall.getHeader()).thenReturn(getHead);
302
303       CallRunner scanCallTask = mock(CallRunner.class);
304       RpcServer.Call scanCall = mock(RpcServer.Call.class);
305       scanCall.param = ScanRequest.newBuilder().setScannerId(1).build();
306       RequestHeader scanHead = RequestHeader.newBuilder().setMethodName("scan").build();
307       when(scanCallTask.getCall()).thenReturn(scanCall);
308       when(scanCall.getHeader()).thenReturn(scanHead);
309
310       ArrayList<Integer> work = new ArrayList<Integer>();
311       doAnswerTaskExecution(putCallTask, work, 1, 1000);
312       doAnswerTaskExecution(getCallTask, work, 2, 1000);
313       doAnswerTaskExecution(scanCallTask, work, 3, 1000);
314
315       // There are 3 queues: [puts], [gets], [scans]
316       // so the calls will be interleaved
317       scheduler.dispatch(putCallTask);
318       scheduler.dispatch(putCallTask);
319       scheduler.dispatch(putCallTask);
320       scheduler.dispatch(getCallTask);
321       scheduler.dispatch(getCallTask);
322       scheduler.dispatch(getCallTask);
323       scheduler.dispatch(scanCallTask);
324       scheduler.dispatch(scanCallTask);
325       scheduler.dispatch(scanCallTask);
326 
327       while (work.size() < 6) {
328         Thread.sleep(100);
329       }
330
331       for (int i = 0; i < work.size() - 2; i += 3) {
332         assertNotEquals(work.get(i + 0), work.get(i + 1));
333         assertNotEquals(work.get(i + 0), work.get(i + 2));
334         assertNotEquals(work.get(i + 1), work.get(i + 2));
335       }
336     } finally {
337       scheduler.stop();
338     }

341   private void doAnswerTaskExecution(final CallRunner callTask,
342       final ArrayList<Integer> results, final int value, final int sleepInterval) {
343     callTask.setStatus(new MonitoredRPCHandlerImpl());
344     doAnswer(new Answer<Object>() {
345       @Override

408     @Override
409     public long currentTime() {
410       for (String threadNamePrefix : threadNamePrefixs) {
411         String threadName = Thread.currentThread().getName();
412         if (threadName.startsWith(threadNamePrefix)) {
413           return timeQ.poll().longValue() + offset;
