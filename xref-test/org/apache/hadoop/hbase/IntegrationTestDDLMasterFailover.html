49  /**
50   *
51   * Integration test that verifies Procedure V2. <br/><br/>
52   *
53   * DDL operations should go through (rollforward or rollback) when primary master is killed by
54   * ChaosMonkey (default MASTER_KILLING)<br/><br/>
55   *
56   * Multiple Worker threads are started to randomly do the following Actions in loops:<br/>
57   * Actions generating and populating tables:
58   * <ul>
59   *     <li>CreateTableAction</li>
60   *     <li>DisableTableAction</li>
61   *     <li>EnableTableAction</li>
62   *     <li>DeleteTableAction</li>
63   *     <li>AddRowAction</li>
64   * </ul>
65   * Actions performing column family DDL operations:
66   * <ul>
67   *     <li>AddColumnFamilyAction</li>
68   *     <li>AlterColumnFamilyVersionsAction</li>
69   *     <li>AlterColumnFamilyEncodingAction</li>
70   *     <li>DeleteColumnFamilyAction</li>
71   * </ul>
72   * Actions performing namespace DDL operations:
73   * <ul>
74   *     <li>AddNamespaceAction</li>
75   *     <li>AlterNamespaceAction</li>
76   *     <li>DeleteNamespaceAction</li>
77   * </ul>
78   * <br/>
79   *
80   * The threads run for a period of time (default 20 minutes) then are stopped at the end of
81   * runtime. Verification is performed towards those checkpoints:
82   * <ol>
83   *     <li>No Actions throw Exceptions.</li>
84   *     <li>No inconsistencies are detected in hbck.</li>
85   * </ol>
86   *
87   * <p>
88   * This test should be run by the hbase user since it invokes hbck at the end
89   * </p><p>
90   * Usage:
91   *  hbase org.apache.hadoop.hbase.IntegrationTestDDLMasterFailover
92   *    -Dhbase.IntegrationTestDDLMasterFailover.runtime=1200000
93   *    -Dhbase.IntegrationTestDDLMasterFailover.numThreads=20
94   *    -Dhbase.IntegrationTestDDLMasterFailover.numRegions=50 --monkey masterKilling
95   */

247   private abstract class NamespaceAction extends MasterAction {
248     final String nsTestConfigKey = "hbase.namespace.testKey";
249
250     // NamespaceAction has implemented selectNamespace() shared by multiple namespace Actions
251     protected NamespaceDescriptor selectNamespace(
252         ConcurrentHashMap<String, NamespaceDescriptor> namespaceMap) {
253       // synchronization to prevent removal from multiple threads
254       synchronized (namespaceMap) {
255         // randomly select namespace from namespaceMap
256         if (namespaceMap.isEmpty()) {
257           return null;
258         }
259         ArrayList<String> namespaceList = new ArrayList<String>(namespaceMap.keySet());
260         String randomKey = namespaceList.get(RandomUtils.nextInt(namespaceList.size()));
261         NamespaceDescriptor randomNsd = namespaceMap.get(randomKey);
262         // remove from namespaceMap
263         namespaceMap.remove(randomKey);
264         return randomNsd;
265       }
266     }
267   }

269   private class CreateNamespaceAction extends NamespaceAction {
270     @Override
271     void perform() throws IOException {
272       Admin admin = connection.getAdmin();
273       try {
274         NamespaceDescriptor nsd;
275         while (true) {
276           nsd = createNamespaceDesc();
277           try {
278             if (admin.getNamespaceDescriptor(nsd.getName()) != null) {
279               // the namespace has already existed.
280               continue;
281             } else {
282               // currently, the code never return null - always throws exception if
283               // namespace is not found - this just a defensive programming to make
284               // sure null situation is handled in case the method changes in the
285               // future.
286               break;
287             }
288           } catch (NamespaceNotFoundException nsnfe) {
289             // This is expected for a random generated NamespaceDescriptor
290             break;
291           }
292         }
293         LOG.info("Creating namespace:" + nsd);
294         admin.createNamespace(nsd);
295         NamespaceDescriptor freshNamespaceDesc = admin.getNamespaceDescriptor(nsd.getName());
296         Assert.assertTrue("Namespace: " + nsd + " was not created", freshNamespaceDesc != null);
297         LOG.info("Created namespace:" + freshNamespaceDesc);
298         namespaceMap.put(nsd.getName(), freshNamespaceDesc);
299       } catch (Exception e){
300         LOG.warn("Caught exception in action: " + this.getClass());
301         throw e;
302       } finally {
303         admin.close();
304       }
305       verifyNamespaces();
306     }
307
308     private NamespaceDescriptor createNamespaceDesc() {
309       String namespaceName = "itnamespace" + String.format("%010d",
310         RandomUtils.nextInt(Integer.MAX_VALUE));
311       NamespaceDescriptor nsd = NamespaceDescriptor.create(namespaceName).build();
312
313       nsd.setConfiguration(
314         nsTestConfigKey,
315         String.format("%010d", RandomUtils.nextInt(Integer.MAX_VALUE)));
316       return nsd;
317     }
318   }

320   private class ModifyNamespaceAction extends NamespaceAction {
321     @Override
322     void perform() throws IOException {
323       NamespaceDescriptor selected = selectNamespace(namespaceMap);
324       if (selected == null) {
325         return;
326       }
327
328       Admin admin = connection.getAdmin();
329       try {
330         String namespaceName = selected.getName();
331         LOG.info("Modifying namespace :" + selected);
332         NamespaceDescriptor modifiedNsd = NamespaceDescriptor.create(namespaceName).build();
333         String nsValueNew;
334         do {
335           nsValueNew = String.format("%010d", RandomUtils.nextInt(Integer.MAX_VALUE));
336         } while (selected.getConfigurationValue(nsTestConfigKey).equals(nsValueNew));
337         modifiedNsd.setConfiguration(nsTestConfigKey, nsValueNew);
338         admin.modifyNamespace(modifiedNsd);
339         NamespaceDescriptor freshNamespaceDesc = admin.getNamespaceDescriptor(namespaceName);
340         Assert.assertTrue(
341           "Namespace: " + selected + " was not modified",
342           freshNamespaceDesc.getConfigurationValue(nsTestConfigKey).equals(nsValueNew));
343         LOG.info("Modified namespace :" + freshNamespaceDesc);
344         namespaceMap.put(namespaceName, freshNamespaceDesc);
345       } catch (Exception e){
346         LOG.warn("Caught exception in action: " + this.getClass());
347         throw e;
348       } finally {
349         admin.close();
350       }
351       verifyNamespaces();
352     }
353   }

390     // TableAction has implemented selectTable() shared by multiple table Actions
391     protected HTableDescriptor selectTable(ConcurrentHashMap<TableName, HTableDescriptor> tableMap)
392     {
393       // synchronization to prevent removal from multiple threads
394       synchronized (tableMap){
395         // randomly select table from tableMap
396         if (tableMap.isEmpty()) {
397           return null;
398         }
399         ArrayList<TableName> tableList = new ArrayList<TableName>(tableMap.keySet());
400         TableName randomKey = tableList.get(RandomUtils.nextInt(tableList.size()));
401         HTableDescriptor randomHtd = tableMap.get(randomKey);
402         // remove from tableMap
403         tableMap.remove(randomKey);
404         return randomHtd;
405       }
406     }

409   private class CreateTableAction extends TableAction {
410
411     @Override
412     void perform() throws IOException {
413       Admin admin = connection.getAdmin();
414       try {
415         HTableDescriptor htd = createTableDesc();
416         TableName tableName = htd.getTableName();
417         if ( admin.tableExists(tableName)){
418           return;
419         }
420         String numRegionKey = String.format(NUM_REGIONS_KEY, this.getClass().getSimpleName());
421         numRegions = getConf().getInt(numRegionKey, DEFAULT_NUM_REGIONS);
422         byte[] startKey = Bytes.toBytes("row-0000000000");
423         byte[] endKey = Bytes.toBytes("row-" + Integer.MAX_VALUE);
424         LOG.info("Creating table:" + htd);
425         admin.createTable(htd, startKey, endKey, numRegions);
426         Assert.assertTrue("Table: " + htd + " was not created", admin.tableExists(tableName));
427         HTableDescriptor freshTableDesc = admin.getTableDescriptor(tableName);
428         enabledTables.put(tableName, freshTableDesc);
429         LOG.info("Created table:" + freshTableDesc);
430       } catch (Exception e) {
431         LOG.warn("Caught exception in action: " + this.getClass());
432         throw e;
433       } finally {
434         admin.close();
435       }
436       verifyTables();
437     }

439     private HTableDescriptor createTableDesc() {
440       String tableName = "ittable-" + String.format("%010d",
441         RandomUtils.nextInt(Integer.MAX_VALUE));
442       String familyName = "cf-" + Math.abs(RandomUtils.nextInt());
443       HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(tableName));
444       // add random column family
445       htd.addFamily(new HColumnDescriptor(familyName));
446       return htd;
447     }

450   private class DisableTableAction extends TableAction {
451
452     @Override
453     void perform() throws IOException {
454
455       HTableDescriptor selected = selectTable(enabledTables);
456       if (selected == null) {
457         return;
458       }
459
460       Admin admin = connection.getAdmin();
461       try {
462         TableName tableName = selected.getTableName();
463         LOG.info("Disabling table :" + selected);
464         admin.disableTable(tableName);
465         Assert.assertTrue("Table: " + selected + " was not disabled",
466             admin.isTableDisabled(tableName));
467         HTableDescriptor freshTableDesc = admin.getTableDescriptor(tableName);
468         disabledTables.put(tableName, freshTableDesc);
469         LOG.info("Disabled table :" + freshTableDesc);
470       } catch (Exception e){
471         LOG.warn("Caught exception in action: " + this.getClass());
472         // TODO workaround
473         // loose restriction for TableNotDisabledException/TableNotEnabledException thrown in sync
474         // operations
475         // 1) when enable/disable starts, the table state is changed to ENABLING/DISABLING (ZK node
476         // in 1.x), which will be further changed to ENABLED/DISABLED once the operation completes
477         // 2) if master failover happens in the middle of the enable/disable operation, the new
478         // master will try to recover the tables in ENABLING/DISABLING state, as programmed in
479         // AssignmentManager#recoverTableInEnablingState() and
480         // AssignmentManager#recoverTableInDisablingState()
481         // 3) after the new master initialization completes, the procedure tries to re-do the
482         // enable/disable operation, which was already done. Ignore those exceptions before change
483         // of behaviors of AssignmentManager in presence of PV2
484         if (e instanceof TableNotEnabledException) {
485           LOG.warn("Caught TableNotEnabledException in action: " + this.getClass());
486           e.printStackTrace();
487         } else {
488           throw e;
489         }
490       } finally {
491         admin.close();
492       }
493       verifyTables();
494     }
495   }

574   private abstract class ColumnAction extends TableAction{
575     // ColumnAction has implemented selectFamily() shared by multiple family Actions
576     protected HColumnDescriptor selectFamily(HTableDescriptor htd) {
577       if (htd == null) {
578         return null;
579       }
580       HColumnDescriptor[] families = htd.getColumnFamilies();
581       if (families.length == 0){
582         LOG.info("No column families in table: " + htd);
583         return null;
584       }
585       HColumnDescriptor randomCfd = families[RandomUtils.nextInt(families.length)];
586       return randomCfd;
587     }

590   private class AddColumnFamilyAction extends ColumnAction {
591
592     @Override
593     void perform() throws IOException {
594       HTableDescriptor selected = selectTable(disabledTables);
595       if (selected == null) {
596         return;
597       }
598
599       Admin admin = connection.getAdmin();
600       try {
601         HColumnDescriptor cfd = createFamilyDesc();
602         if (selected.hasFamily(cfd.getName())){
603           LOG.info(new String(cfd.getName()) + " already exists in table "
604               + selected.getTableName());
605           return;
606         }
607         TableName tableName = selected.getTableName();
608         LOG.info("Adding column family: " + cfd + " to table: " + tableName);
609         admin.addColumn(tableName, cfd);
610         // assertion
611         HTableDescriptor freshTableDesc = admin.getTableDescriptor(tableName);
612         Assert.assertTrue("Column family: " + cfd + " was not added",
613             freshTableDesc.hasFamily(cfd.getName()));
614         LOG.info("Added column family: " + cfd + " to table: " + tableName);
615         disabledTables.put(tableName, freshTableDesc);
616       } catch (Exception e) {
617         LOG.warn("Caught exception in action: " + this.getClass());
618         throw e;
619       } finally {
620         admin.close();
621       }
622       verifyTables();
623     }

625     private HColumnDescriptor createFamilyDesc() {
626       String familyName = "cf-" + String.format("%010d", RandomUtils.nextInt(Integer.MAX_VALUE));
627       HColumnDescriptor cfd = new HColumnDescriptor(familyName);
628       return cfd;
629     }

632   private class AlterFamilyVersionsAction extends ColumnAction {
633
634     @Override
635     void perform() throws IOException {
636       HTableDescriptor selected = selectTable(disabledTables);
637       if (selected == null) {
638         return;
639       }
640       HColumnDescriptor columnDesc = selectFamily(selected);
641       if (columnDesc == null){
642         return;
643       }
644
645       Admin admin = connection.getAdmin();
646       int versions = RandomUtils.nextInt(10) + 3;
647       try {
648         TableName tableName = selected.getTableName();
649         LOG.info("Altering versions of column family: " + columnDesc + " to: " + versions +
650             " in table: " + tableName);
651         columnDesc.setMinVersions(versions);
652         columnDesc.setMaxVersions(versions);
653         admin.modifyTable(tableName, selected);
654         // assertion
655         HTableDescriptor freshTableDesc = admin.getTableDescriptor(tableName);
656         HColumnDescriptor freshColumnDesc = freshTableDesc.getFamily(columnDesc.getName());
657         Assert.assertEquals("Column family: " + columnDesc + " was not altered",
658             freshColumnDesc.getMaxVersions(), versions);
659         Assert.assertEquals("Column family: " + freshColumnDesc + " was not altered",
660             freshColumnDesc.getMinVersions(), versions);
661         LOG.info("Altered versions of column family: " + columnDesc + " to: " + versions +
662             " in table: " + tableName);
663         disabledTables.put(tableName, freshTableDesc);
664       } catch (Exception e) {
665         LOG.warn("Caught exception in action: " + this.getClass());
666         throw e;
667       } finally {
668         admin.close();
669       }
670       verifyTables();
671     }

674   private class AlterFamilyEncodingAction extends ColumnAction {
675
676     @Override
677     void perform() throws IOException {
678       HTableDescriptor selected = selectTable(disabledTables);
679       if (selected == null) {
680         return;
681       }
682       HColumnDescriptor columnDesc = selectFamily(selected);
683       if (columnDesc == null){
684         return;
685       }
686
687       Admin admin = connection.getAdmin();
688       try {
689         TableName tableName = selected.getTableName();
690         // possible DataBlockEncoding ids
691         int[] possibleIds = {0, 2, 3, 4, 6};
692         short id = (short) possibleIds[RandomUtils.nextInt(possibleIds.length)];
693         LOG.info("Altering encoding of column family: " + columnDesc + " to: " + id +
694             " in table: " + tableName);

818     @Override
819     public void run() {
820       while (running.get()) {
821         // select random action

823         this.action = selectedAction;
824         LOG.info("Performing Action: " + selectedAction);
825
826         try {

828           case CREATE_NAMESPACE:
829             new CreateNamespaceAction().perform();
830             break;
831           case MODIFY_NAMESPACE:
832             new ModifyNamespaceAction().perform();
833             break;
834           case DELETE_NAMESPACE:
835             new DeleteNamespaceAction().perform();
836             break;
837           case CREATE_TABLE:
838             // stop creating new tables in the later stage of the test to avoid too many empty
839             // tables
840             if (create_table.get()) {
841               new CreateTableAction().perform();
842             }
843             break;
844           case ADD_ROW:
845             new AddRowAction().perform();
846             break;
847           case DISABLE_TABLE:
848             new DisableTableAction().perform();
849             break;
850           case ENABLE_TABLE:
851             new EnableTableAction().perform();
852             break;
853           case DELETE_TABLE:
854             // reduce probability of deleting table to 20%
855             if (RandomUtils.nextInt(100) < 20) {
856               new DeleteTableAction().perform();
857             }
858             break;
859           case ADD_COLUMNFAMILY:
860             new AddColumnFamilyAction().perform();
861             break;
862           case DELETE_COLUMNFAMILY:
863             // reduce probability of deleting column family to 20%
864             if (RandomUtils.nextInt(100) < 20) {
865               new DeleteColumnFamilyAction().perform();
866             }
867             break;
868           case ALTER_FAMILYVERSIONS:
869             new AlterFamilyVersionsAction().perform();
870             break;
871           case ALTER_FAMILYENCODING:
872             new AlterFamilyEncodingAction().perform();
873             break;
874           }
875         } catch (Exception ex) {
876           this.savedException = ex;
877           return;
878         }
879       }
880       LOG.info(this.getName() + " stopped");
881     }

907     LOG.info("Starting the test");
908
909     String runtimeKey = String.format(RUN_TIME_KEY, this.getClass().getSimpleName());
910     long runtime = util.getConfiguration().getLong(runtimeKey, DEFAULT_RUN_TIME);
911
912     String numThreadKey = String.format(NUM_THREADS_KEY, this.getClass().getSimpleName());
913     numThreads = util.getConfiguration().getInt(numThreadKey, DEFAULT_NUM_THREADS);
914
915     ArrayList<Worker> workers = new ArrayList<>();
916     for (int i = 0; i < numThreads; i++) {
917       checkException(workers);
918       Worker worker = new Worker();
919       LOG.info("Launching worker thread " + worker.getName());
920       workers.add(worker);
921       worker.start();
922     }
923
924     Threads.sleep(runtime / 2);
925     LOG.info("Stopping creating new tables");
926     create_table.set(false);
927     Threads.sleep(runtime / 2);
928     LOG.info("Runtime is up");
929     running.set(false);
930
931     checkException(workers);
932
933     for (Worker worker : workers) {
934       worker.join();
935     }
936     LOG.info("All Worker threads stopped");
937
938     // verify
939     LOG.info("Verify actions of all threads succeeded");
940     checkException(workers);
941     LOG.info("Verify namespaces");
942     verifyNamespaces();
943     LOG.info("Verify states of all tables");
944     verifyTables();
945
946     // RUN HBCK
947
948     HBaseFsck hbck = null;
949     try {
950       LOG.info("Running hbck");
951       hbck = HbckTestingUtil.doFsck(util.getConfiguration(), false);
952       if (HbckTestingUtil.inconsistencyFound(hbck)) {
953         // Find the inconsistency during HBCK. Leave table and namespace undropped so that
954         // we can check outside the test.
955         keepObjectsAtTheEnd = true;
956       }
957       HbckTestingUtil.assertNoErrors(hbck);
958       LOG.info("Finished hbck");
959     } finally {
960       if (hbck != null) {
961         hbck.close();
962       }
963     }
964      return 0;
