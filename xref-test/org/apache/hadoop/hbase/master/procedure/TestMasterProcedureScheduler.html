66    @Test
67    public void testConcurrentCreateDelete() throws Exception {
68      final MasterProcedureScheduler procQueue = queue;
69      final TableName table = TableName.valueOf("testtb");
70      final AtomicBoolean running = new AtomicBoolean(true);
71      final AtomicBoolean failure = new AtomicBoolean(false);
72      Thread createThread = new Thread() {
73        @Override
74        public void run() {
75          try {
76            TestTableProcedure proc = new TestTableProcedure(1, table,
77                TableProcedureInterface.TableOperationType.CREATE);
78            while (running.get() && !failure.get()) {
79              if (procQueue.tryAcquireTableExclusiveLock(proc, table)) {
80                procQueue.releaseTableExclusiveLock(proc, table);
81              }
82            }
83          } catch (Throwable e) {
84            LOG.error("create failed", e);
85            failure.set(true);
86          }
87        }
88      };
89  
90      Thread deleteThread = new Thread() {
91        @Override
92        public void run() {
93          try {
94            TestTableProcedure proc = new TestTableProcedure(2, table,
95                TableProcedureInterface.TableOperationType.DELETE);
96            while (running.get() && !failure.get()) {
97              if (procQueue.tryAcquireTableExclusiveLock(proc, table)) {
98                procQueue.releaseTableExclusiveLock(proc, table);
99              }
100             procQueue.markTableAsDeleted(table);
101           }
102         } catch (Throwable e) {
103           LOG.error("delete failed", e);
104           failure.set(true);
105         }
106       }
107     };
108 
109     createThread.start();
110     deleteThread.start();
111     for (int i = 0; i < 100 && running.get() && !failure.get(); ++i) {
112       Thread.sleep(100);
113     }
114     running.set(false);
115     createThread.join();
116     deleteThread.join();
117     assertEquals(false, failure.get());
118   }

123   @Test
124   public void testSimpleTableOpsQueues() throws Exception {
125     final int NUM_TABLES = 10;
126     final int NUM_ITEMS = 10;
127 
128     int count = 0;
129     for (int i = 1; i <= NUM_TABLES; ++i) {
130       TableName tableName = TableName.valueOf(String.format("test-%04d", i));
131       // insert items
132       for (int j = 1; j <= NUM_ITEMS; ++j) {
133         queue.addBack(new TestTableProcedure(i * 1000 + j, tableName,
134           TableProcedureInterface.TableOperationType.EDIT));
135         assertEquals(++count, queue.size());
136       }
137     }
138     assertEquals(NUM_TABLES * NUM_ITEMS, queue.size());
139 
140     for (int j = 1; j <= NUM_ITEMS; ++j) {
141       for (int i = 1; i <= NUM_TABLES; ++i) {
142         Procedure proc = queue.poll();
143         assertTrue(proc != null);
144         TableName tableName = ((TestTableProcedure)proc).getTableName();
145         queue.tryAcquireTableExclusiveLock(proc, tableName);
146         queue.releaseTableExclusiveLock(proc, tableName);
147         queue.completionCleanup(proc);
148         assertEquals(--count, queue.size());
149         assertEquals(i * 1000 + j, proc.getProcId());
150       }
151     }
152     assertEquals(0, queue.size());
153 
154     for (int i = 1; i <= NUM_TABLES; ++i) {
155       TableName tableName = TableName.valueOf(String.format("test-%04d", i));
156       // complete the table deletion
157       assertTrue(queue.markTableAsDeleted(tableName));
158     }
159   }

190    * Check that the table queue is not deletable until every procedure
191    * in-progress is completed (this is a special case for read-locks).
192    */
193   @Test
194   public void testCreateDeleteTableOperationsWithReadLock() throws Exception {
195     final TableName tableName = TableName.valueOf("testtb");
196     final int nitems = 2;
197 
198     for (int i = 1; i <= nitems; ++i) {
199       queue.addBack(new TestTableProcedure(i, tableName,
200             TableProcedureInterface.TableOperationType.READ));
201     }
202 
203     // table can't be deleted because one item is in the queue
204     assertFalse(queue.markTableAsDeleted(tableName));
205 
206     Procedure[] procs = new Procedure[nitems];
207     for (int i = 0; i < nitems; ++i) {
208       // fetch item and take a lock
209       Procedure proc = procs[i] = queue.poll();
210       assertEquals(i + 1, proc.getProcId());
211       // take the rlock
212       assertTrue(queue.tryAcquireTableSharedLock(proc, tableName));
213       // table can't be deleted because we have locks and/or items in the queue
214       assertFalse(queue.markTableAsDeleted(tableName));
215     }
216 
217     for (int i = 0; i < nitems; ++i) {
218       // table can't be deleted because we have locks
219       assertFalse(queue.markTableAsDeleted(tableName));
220       // release the rlock
221       queue.releaseTableSharedLock(procs[i], tableName);
222     }
223 
224     // there are no items and no lock in the queeu
225     assertEquals(0, queue.size());
226     // complete the table deletion
227     assertTrue(queue.markTableAsDeleted(tableName));
228   }

231    * Verify the correct logic of RWLocks on the queue
232    */
233   @Test
234   public void testVerifyRwLocks() throws Exception {
235     TableName tableName = TableName.valueOf("testtb");
236     queue.addBack(new TestTableProcedure(1, tableName,
237           TableProcedureInterface.TableOperationType.EDIT));
238     queue.addBack(new TestTableProcedure(2, tableName,
239           TableProcedureInterface.TableOperationType.READ));
240     queue.addBack(new TestTableProcedure(3, tableName,
241           TableProcedureInterface.TableOperationType.EDIT));
242     queue.addBack(new TestTableProcedure(4, tableName,
243           TableProcedureInterface.TableOperationType.READ));
244     queue.addBack(new TestTableProcedure(5, tableName,
245           TableProcedureInterface.TableOperationType.READ));
246 
247     // Fetch the 1st item and take the write lock
248     Procedure proc = queue.poll();
249     assertEquals(1, proc.getProcId());
250     assertEquals(true, queue.tryAcquireTableExclusiveLock(proc, tableName));
251 
252     // Fetch the 2nd item and verify that the lock can't be acquired
253     assertEquals(null, queue.poll(0));
254 
255     // Release the write lock and acquire the read lock
256     queue.releaseTableExclusiveLock(proc, tableName);
257 
258     // Fetch the 2nd item and take the read lock
259     Procedure rdProc = queue.poll();
260     assertEquals(2, rdProc.getProcId());
261     assertEquals(true, queue.tryAcquireTableSharedLock(rdProc, tableName));
262 
263     // Fetch the 3rd item and verify that the lock can't be acquired
264     Procedure wrProc = queue.poll();
265     assertEquals(3, wrProc.getProcId());
266     assertEquals(false, queue.tryAcquireTableExclusiveLock(wrProc, tableName));
267 
268     // release the rdlock of item 2 and take the wrlock for the 3d item
269     queue.releaseTableSharedLock(rdProc, tableName);
270     assertEquals(true, queue.tryAcquireTableExclusiveLock(wrProc, tableName));
271 
272     // Fetch 4th item and verify that the lock can't be acquired
273     assertEquals(null, queue.poll(0));
274 
275     // Release the write lock and acquire the read lock
276     queue.releaseTableExclusiveLock(wrProc, tableName);
277 
278     // Fetch the 4th item and take the read lock
279     rdProc = queue.poll();
280     assertEquals(4, rdProc.getProcId());
281     assertEquals(true, queue.tryAcquireTableSharedLock(rdProc, tableName));
282 
283     // Fetch the 4th item and take the read lock
284     Procedure rdProc2 = queue.poll();
285     assertEquals(5, rdProc2.getProcId());
286     assertEquals(true, queue.tryAcquireTableSharedLock(rdProc2, tableName));
287 
288     // Release 4th and 5th read-lock
289     queue.releaseTableSharedLock(rdProc, tableName);
290     queue.releaseTableSharedLock(rdProc2, tableName);
291 
292     // remove table queue
293     assertEquals(0, queue.size());
294     assertTrue("queue should be deleted", queue.markTableAsDeleted(tableName));
295   }

297   @Test
298   public void testVerifyNamespaceRwLocks() throws Exception {
299     String nsName1 = "ns1";
300     String nsName2 = "ns2";
301     TableName tableName1 = TableName.valueOf(nsName1, "testtb");
302     TableName tableName2 = TableName.valueOf(nsName2, "testtb");
303     queue.addBack(new TestNamespaceProcedure(1, nsName1,
304           TableProcedureInterface.TableOperationType.EDIT));
305     queue.addBack(new TestTableProcedure(2, tableName1,
306           TableProcedureInterface.TableOperationType.EDIT));
307     queue.addBack(new TestTableProcedure(3, tableName2,
308           TableProcedureInterface.TableOperationType.EDIT));
309     queue.addBack(new TestNamespaceProcedure(4, nsName2,
310           TableProcedureInterface.TableOperationType.EDIT));
311 
312     // Fetch the 1st item and take the write lock
313     Procedure procNs1 = queue.poll();
314     assertEquals(1, procNs1.getProcId());
315     assertEquals(true, queue.tryAcquireNamespaceExclusiveLock(procNs1, nsName1));
316 
317     // System tables have 2 as default priority
318     Procedure procNs2 = queue.poll();
319     assertEquals(4, procNs2.getProcId());
320     assertEquals(true, queue.tryAcquireNamespaceExclusiveLock(procNs2, nsName2));
321     queue.releaseNamespaceExclusiveLock(procNs2, nsName2);
322     queue.yield(procNs2);
323 
324     // table on ns1 is locked, so we get table on ns2
325     procNs2 = queue.poll();
326     assertEquals(3, procNs2.getProcId());
327     assertEquals(true, queue.tryAcquireTableExclusiveLock(procNs2, tableName2));
328 
329     // ns2 is not available (TODO we may avoid this one)
330     Procedure procNs2b = queue.poll();
331     assertEquals(4, procNs2b.getProcId());
332     assertEquals(false, queue.tryAcquireNamespaceExclusiveLock(procNs2b, nsName2));
333     queue.yield(procNs2b);
334 
335     // release the ns1 lock
336     queue.releaseNamespaceExclusiveLock(procNs1, nsName1);
337 
338     // we are now able to execute table of ns1
339     long procId = queue.poll().getProcId();
340     assertEquals(2, procId);
341 
342     queue.releaseTableExclusiveLock(procNs2, tableName2);
343 
344     // we are now able to execute ns2
345     procId = queue.poll().getProcId();
346     assertEquals(4, procId);
347   }

349   /**
350    * Verify that "write" operations for a single table are serialized,
351    * but different tables can be executed in parallel.
352    */
353   @Test(timeout=90000)
354   public void testConcurrentWriteOps() throws Exception {
355     final TestTableProcSet procSet = new TestTableProcSet(queue);
356 
357     final int NUM_ITEMS = 10;
358     final int NUM_TABLES = 4;
359     final AtomicInteger opsCount = new AtomicInteger(0);
360     for (int i = 0; i < NUM_TABLES; ++i) {
361       TableName tableName = TableName.valueOf(String.format("testtb-%04d", i));
362       for (int j = 1; j < NUM_ITEMS; ++j) {
363         procSet.addBack(new TestTableProcedure(i * 100 + j, tableName,
364           TableProcedureInterface.TableOperationType.EDIT));
365         opsCount.incrementAndGet();
366       }
367     }
368     assertEquals(opsCount.get(), queue.size());
369 
370     final Thread[] threads = new Thread[NUM_TABLES * 2];
371     final HashSet<TableName> concurrentTables = new HashSet<TableName>();
372     final ArrayList<String> failures = new ArrayList<String>();
373     final AtomicInteger concurrentCount = new AtomicInteger(0);
374     for (int i = 0; i < threads.length; ++i) {
375       threads[i] = new Thread() {
376         @Override
377         public void run() {
378           while (opsCount.get() > 0) {
379             try {
380               Procedure proc = procSet.acquire();
381               if (proc == null) {
382                 queue.signalAll();
383                 if (opsCount.get() > 0) {
384                   continue;
385                 }
386                 break;
387               }
388 
389               TableName tableId = procSet.getTableName(proc);
390               synchronized (concurrentTables) {
391                 assertTrue("unexpected concurrency on " + tableId, concurrentTables.add(tableId));
392               }
393               assertTrue(opsCount.decrementAndGet() >= 0);
394               try {
395                 long procId = proc.getProcId();
396                 int concurrent = concurrentCount.incrementAndGet();
397                 assertTrue("inc-concurrent="+ concurrent +" 1 <= concurrent <= "+ NUM_TABLES,
398                   concurrent >= 1 && concurrent <= NUM_TABLES);
399                 LOG.debug("[S] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
400                 Thread.sleep(2000);
401                 concurrent = concurrentCount.decrementAndGet();
402                 LOG.debug("[E] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
403                 assertTrue("dec-concurrent=" + concurrent, concurrent < NUM_TABLES);
404               } finally {
405                 synchronized (concurrentTables) {
406                   assertTrue(concurrentTables.remove(tableId));
407                 }
408                 procSet.release(proc);
409               }
410             } catch (Throwable e) {
411               LOG.error("Failed " + e.getMessage(), e);
412               synchronized (failures) {
413                 failures.add(e.getMessage());
414               }
415             } finally {
416               queue.signalAll();
417             }
418           }
419         }
420       };
421       threads[i].start();
422     }
423     for (int i = 0; i < threads.length; ++i) {
424       threads[i].join();
425     }
426     assertTrue(failures.toString(), failures.isEmpty());
427     assertEquals(0, opsCount.get());
428     assertEquals(0, queue.size());
429 
430     for (int i = 1; i <= NUM_TABLES; ++i) {
431       TableName table = TableName.valueOf(String.format("testtb-%04d", i));
432       assertTrue("queue should be deleted, table=" + table, queue.markTableAsDeleted(table));
433     }
434   }

436   public static class TestTableProcSet {
437     private final MasterProcedureScheduler queue;
438 
439     public TestTableProcSet(final MasterProcedureScheduler queue) {
440       this.queue = queue;
441     }
442 
443     public void addBack(Procedure proc) {
444       queue.addBack(proc);
445     }
446 
447     public void addFront(Procedure proc) {
448       queue.addFront(proc);
449     }
450 
451     public Procedure acquire() {
452       Procedure proc = null;
453       boolean avail = false;
454       while (!avail) {
455         proc = queue.poll();
456         if (proc == null) break;
457         switch (getTableOperationType(proc)) {
458           case CREATE:
459           case DELETE:
460           case EDIT:
461             avail = queue.tryAcquireTableExclusiveLock(proc, getTableName(proc));
462             break;
463           case READ:
464             avail = queue.tryAcquireTableSharedLock(proc, getTableName(proc));
465             break;
466           default:
467             throw new UnsupportedOperationException();
468         }
469         if (!avail) {
470           addFront(proc);
471           LOG.debug("yield procId=" + proc);
472         }
473       }
474       return proc;
475     }
476 
477     public void release(Procedure proc) {
478       switch (getTableOperationType(proc)) {
479         case CREATE:
480         case DELETE:
481         case EDIT:
482           queue.releaseTableExclusiveLock(proc, getTableName(proc));
483           break;
484         case READ:
485           queue.releaseTableSharedLock(proc, getTableName(proc));
486           break;
487       }
488     }
489 
490     public TableName getTableName(Procedure proc) {
491       return ((TableProcedureInterface)proc).getTableName();
492     }
493 
494     public TableProcedureInterface.TableOperationType getTableOperationType(Procedure proc) {
495       return ((TableProcedureInterface)proc).getTableOperationType();
496     }
497   }
