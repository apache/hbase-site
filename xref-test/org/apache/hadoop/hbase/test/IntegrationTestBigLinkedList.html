123 /**
124  * This is an integration test borrowed from goraci, written by Keith Turner,
125  * which is in turn inspired by the Accumulo test called continous ingest (ci).
126  * The original source code can be found here:
127  * https://github.com/keith-turner/goraci
128  * https://github.com/enis/goraci/
129  *
130  * Apache Accumulo [0] has a simple test suite that verifies that data is not
131  * lost at scale. This test suite is called continuous ingest. This test runs
132  * many ingest clients that continually create linked lists containing 25
133  * million nodes. At some point the clients are stopped and a map reduce job is
134  * run to ensure no linked list has a hole. A hole indicates data was lost.··
135  *
136  * The nodes in the linked list are random. This causes each linked list to
137  * spread across the table. Therefore if one part of a table loses data, then it
138  * will be detected by references in another part of the table.
139  *
140  * THE ANATOMY OF THE TEST
141  *
142  * Below is rough sketch of how data is written. For specific details look at
143  * the Generator code.
144  *
145  * 1 Write out 1 million nodes· 2 Flush the client· 3 Write out 1 million that
146  * reference previous million· 4 If this is the 25th set of 1 million nodes,
147  * then update 1st set of million to point to last· 5 goto 1
148  *
149  * The key is that nodes only reference flushed nodes. Therefore a node should
150  * never reference a missing node, even if the ingest client is killed at any
151  * point in time.
152  *
153  * When running this test suite w/ Accumulo there is a script running in
154  * parallel called the Aggitator that randomly and continuously kills server
155  * processes.·· The outcome was that many data loss bugs were found in Accumulo
156  * by doing this.· This test suite can also help find bugs that impact uptime
157  * and stability when· run for days or weeks.··
158  *
159  * This test suite consists the following· - a few Java programs· - a little
160  * helper script to run the java programs - a maven script to build it.··
161  *
162  * When generating data, its best to have each map task generate a multiple of
163  * 25 million. The reason for this is that circular linked list are generated
164  * every 25M. Not generating a multiple in 25M will result in some nodes in the
165  * linked list not having references. The loss of an unreferenced node can not
166  * be detected.
167  *
168  *
169  * Below is a description of the Java programs
170  *
171  * Generator - A map only job that generates data. As stated previously,·its best to generate data
172  * in multiples of 25M. An option is also available to allow concurrent walkers to select and walk
173  * random flushed loops during this phase.
174  *
175  * Verify - A map reduce job that looks for holes. Look at the counts after running. REFERENCED and
176  * UNREFERENCED are· ok, any UNDEFINED counts are bad. Do not run at the· same
177  * time as the Generator.
178  *
179  * Walker - A standalone program that start following a linked list· and emits timing info.··
180  *
181  * Print - A standalone program that prints nodes in the linked list
182  *
183  * Delete - A standalone program that deletes a single node
184  *
185  * This class can be run as a unit test, as an integration test, or from the command line
186  *
187  * ex:
188  * ./hbase org.apache.hadoop.hbase.test.IntegrationTestBigLinkedList
189  *    loop 2 1 100000 /temp 1 1000 50 1 0
190  *
191  */

455       @Override
456       protected void map(BytesWritable key, NullWritable value, Context output) throws IOException {
457         current[i] = new byte[key.getLength()];
458         System.arraycopy(key.getBytes(), 0, current[i], 0, key.getLength());
459         if (++i == current.length) {
460           LOG.info("Persisting current.length=" + current.length + ", count=" + count + ", id=" +
461             Bytes.toStringBinary(id) + ", current=" + Bytes.toStringBinary(current[0]) +
462             ", i=" + i);
463           persist(output, count, prev, current, id);
464           i = 0;
465 
466           if (first == null) {
467             first = current;
468           }
469           prev = current;
470           current = new byte[this.width][];
471
472           count += current.length;
473           output.setStatus("Count " + count);
474
475           if (count % wrap == 0) {
476             // this block of code turns the 1 million linked list of length 25 into one giant
477             //circular linked list of 25 million
478             circularLeftShift(first);
479             persist(output, -1, prev, first, null);
480             // At this point the entire loop has been flushed so we can add one of its nodes to the
481             // concurrent walker
482             if (numWalkers > 0) {
483               addFlushed(key.getBytes());
484               if (walkers.isEmpty()) {
485                 startWalkers(numWalkers, conf, output);
486               }
487             }
488             first = null;
489             prev = null;
490           }
491         }

507       protected void persist(Context output, long count, byte[][] prev, byte[][] current, byte[] id)
508           throws IOException {
509         for (int i = 0; i < current.length; i++) {
510           Put put = new Put(current[i]);
511           put.addColumn(FAMILY_NAME, COLUMN_PREV, prev == null ? NO_KEY : prev[i]);
512
513           if (count >= 0) {
514             put.addColumn(FAMILY_NAME, COLUMN_COUNT, Bytes.toBytes(count + i));
515           }
516           if (id != null) {
517             put.addColumn(FAMILY_NAME, COLUMN_CLIENT, id);
518           }
519           // See if we are to write multiple columns.
520           if (this.multipleUnevenColumnFamilies) {
521             // Use any column name.
522             put.addColumn(TINY_FAMILY_NAME, TINY_FAMILY_NAME, this.tinyValue);
523             // If we've not allocated bigValue, do it now. Reuse same value each time.
524             if (this.bigValue == null) {
525               this.bigValue = new byte[current[i].length * 10];
526               ThreadLocalRandom.current().nextBytes(this.bigValue);
527             }
528             // Use any column name.
529             put.addColumn(BIG_FAMILY_NAME, BIG_FAMILY_NAME, this.bigValue);
530           }
531           mutator.mutate(put);
532 
533           if (i % 1000 == 0) {
534             // Tickle progress every so often else maprunner will think us hung
535             output.progress();
536           }
537         }
538         mutator.flush();
539       }

541       private void startWalkers(int numWalkers, Configuration conf, Context context) {
542         LOG.info("Starting " + numWalkers + " concurrent walkers");
543         for (int i = 0; i < numWalkers; i++) {

550       private void joinWalkers() {
551         walkersStop = true;
552         synchronized (flushedLoops) {
553           flushedLoops.notifyAll();
554         }
555         for (Thread walker : walkers) {
556           try {
557             walker.join();
558           } catch (InterruptedException e) {
559             // no-op
560           }
561         }

564       /**
565        * Randomly selects and walks a random flushed loop concurrently with the Generator Mapper by
566        * spawning ConcurrentWalker's with specified StartNodes. These ConcurrentWalker's are
567        * configured to only log erroneous nodes.
568        */
569 
570       public class ContinuousConcurrentWalker implements Runnable {
571
572         ConcurrentWalker walker;
573         Configuration conf;
574         Context context;
575         Random rand;
576
577         public ContinuousConcurrentWalker(Configuration conf, Context context) {
578           this.conf = conf;
579           this.context = context;
580           rand = new Random();
581         }
582
583         @Override
584         public void run() {
585           while (!walkersStop) {
586             try {
587               long node = selectLoop();
588               try {
589                 walkLoop(node);
590               } catch (IOException e) {
591                 context.getCounter(Counts.IOEXCEPTION).increment(1l);
592                 return;
593               }
594             } catch (InterruptedException e) {
595               return;
596             }
597           }
598         }
599 
600         private void walkLoop(long node) throws IOException {
601           walker = new ConcurrentWalker(context);
602           walker.setConf(conf);
603           walker.run(node, wrap);
604         }
605
606         private long selectLoop () throws InterruptedException{
607           synchronized (flushedLoops) {
608             while (flushedLoops.isEmpty() && !walkersStop) {
609               flushedLoops.wait();
610             }
611             if (walkersStop) {
612               throw new InterruptedException();
613             }
614             return flushedLoops.get(rand.nextInt(flushedLoops.size()));
615           }
616         }
617       }

630           Connection connection = ConnectionFactory.createConnection(getConf());
631           Table table = connection.getTable(getTableName(getConf()));
632           long numQueries = 0;
633           // If isSpecificStart is set, only walk one list from that particular node.
634           // Note that in case of circular (or P-shaped) list it will walk forever, as is
635           // the case in normal run without startKey.
636
637           CINode node = findStartNode(table, startKey);
638           if (node == null) {
639             LOG.error("Start node not found: " + Bytes.toStringBinary(startKey));
640             throw new IOException("Start node not found: " + startKeyIn);
641           }
642           while (numQueries < maxQueries) {
643             numQueries++;
644             byte[] prev = node.prev;
645             long t1 = System.currentTimeMillis();
646             node = getNode(prev, table, node);
647             long t2 = System.currentTimeMillis();
648             if (node == null) {
649               LOG.error("ConcurrentWalker found UNDEFINED NODE: " + Bytes.toStringBinary(prev));
650               context.getCounter(Counts.UNDEFINED).increment(1l);
651             } else if (node.prev.length == NO_KEY.length) {
652               LOG.error("ConcurrentWalker found TERMINATING NODE: " +
653                   Bytes.toStringBinary(node.key));
654               context.getCounter(Counts.TERMINATING).increment(1l);
655             } else {
656               // Increment for successful walk
657               context.getCounter(Counts.SUCCESS).increment(1l);
658             }
659           }
660           table.close();
661           connection.close();
662         }
663       }
664     }

954     throws IOException, InterruptedException {

956       FileSystem fs = FileSystem.get(conf);
957       SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
958       if (!fs.exists(keysInputDir)) {
959         throw new FileNotFoundException(keysInputDir.toString());
960       }
961       if (!fs.isDirectory(keysInputDir)) {
962         throw new UnsupportedOperationException("TODO");

975     private static SortedSet<byte[]> readFileToSearch(final Configuration conf,
976         final FileSystem fs, final LocatedFileStatus keyFileStatus) throws IOException,
977         InterruptedException {
978       SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
979       // Return entries that are flagged Counts.UNDEFINED in the value. Return the row. This is
980       // what is missing.
981       TaskAttemptContext context = new TaskAttemptContextImpl(conf, new TaskAttemptID());
982       try (SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader rr =
983           new SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader()) {
984         InputSplit is =
985           new FileSplit(keyFileStatus.getPath(), 0, keyFileStatus.getLen(), new String [] {});
986         rr.initialize(is, context);
987         while (rr.nextKeyValue()) {
988           rr.getCurrentKey();
989           BytesWritable bw = rr.getCurrentValue();
990           if (Verify.VerifyReducer.whichType(bw.getBytes()) == Verify.Counts.UNDEFINED) {
991             byte[] key = new byte[rr.getCurrentKey().getLength()];
992             System.arraycopy(rr.getCurrentKey().getBytes(), 0, key, 0, rr.getCurrentKey()
993                 .getLength());
994             result.add(key);
995           }
996         }
997       }
998       return result;
999     }

1058     /**
1059      * Per reducer, we output problem rows as byte arrasy so can be used as input for
1060      * subsequent investigative mapreduce jobs. Each emitted value is prefaced by a one byte flag
1061      * saying what sort of emission it is. Flag is the Count enum ordinal as a short.
1062      */
1063     public static class VerifyReducer extends
1064         Reducer<BytesWritable, BytesWritable, BytesWritable, BytesWritable> {
1065       private ArrayList<byte[]> refs = new ArrayList<byte[]>();
1066       private final BytesWritable UNREF = new BytesWritable(addPrefixFlag(
1067         Counts.UNREFERENCED.ordinal(), new byte[] {}));
1068       private final BytesWritable LOSTFAM = new BytesWritable(addPrefixFlag(
1069         Counts.LOST_FAMILIES.ordinal(), new byte[] {}));
1070
1071       private AtomicInteger rows = new AtomicInteger(0);
1072       private Connection connection;
1073
1074       @Override
1075       protected void setup(Reducer<BytesWritable, BytesWritable, BytesWritable, BytesWritable>.Context context)
1076       throws IOException, InterruptedException {
1077         super.setup(context);
1078         this.connection = ConnectionFactory.createConnection(context.getConfiguration());
1079       }
1080
1081       @Override
1082       protected void cleanup(
1083           Reducer<BytesWritable, BytesWritable, BytesWritable, BytesWritable>.Context context)
1084           throws IOException, InterruptedException {

1086           this.connection.close();

1091       /**
1092        * @param ordinal
1093        * @param r
1094        * @return Return new byte array that has <code>ordinal</code> as prefix on front taking up
1095        * Bytes.SIZEOF_SHORT bytes followed by <code>r</code>
1096        */
1097       public static byte[] addPrefixFlag(final int ordinal, final byte [] r) {
1098         byte[] prefix = Bytes.toBytes((short)ordinal);
1099         if (prefix.length != Bytes.SIZEOF_SHORT) {
1100           throw new RuntimeException("Unexpected size: " + prefix.length);
1101         }
1102         byte[] result = new byte[prefix.length + r.length];
1103         System.arraycopy(prefix, 0, result, 0, prefix.length);
1104         System.arraycopy(r, 0, result, prefix.length, r.length);
1105         return result;
1106       }

1109        * @param bs
1110        * @return Type from the Counts enum of this row. Reads prefix added by
1111        * {@link #addPrefixFlag(int, byte[])}
1112        */
1113       public static Counts whichType(final byte [] bs) {
1114         int ordinal = Bytes.toShort(bs, 0, Bytes.SIZEOF_SHORT);
1115         return Counts.values()[ordinal];

1121        */

1123         byte[] bytes = new byte [bw.getLength() - Bytes.SIZEOF_SHORT];
1124         System.arraycopy(bw.getBytes(), Bytes.SIZEOF_SHORT, bytes, 0, bytes.length);

1128       @Override
1129       public void reduce(BytesWritable key, Iterable<BytesWritable> values, Context context)
1130           throws IOException, InterruptedException {
1131         int defCount = 0;
1132         boolean lostFamilies = false;
1133         refs.clear();
1134         for (BytesWritable type : values) {
1135           if (type.getLength() == DEF.getLength()) {
1136             defCount++;
1137             if (type.getBytes()[0] == 1) {
1138               lostFamilies = true;
1139             }
1140           } else {
1141             byte[] bytes = new byte[type.getLength()];
1142             System.arraycopy(type.getBytes(), 0, bytes, 0, type.getLength());
1143             refs.add(bytes);
1144           }
1145         }
1146
1147         // TODO check for more than one def, should not happen
1148         StringBuilder refsSb = null;
1149         String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
1150         if (defCount == 0 || refs.size() != 1) {
1151           refsSb = dumpExtraInfoOnRefs(key, context, refs);
1152           LOG.error("LinkedListError: key=" + keyString + ", reference(s)=" +
1153             (refsSb != null? refsSb.toString(): ""));
1154         }
1155         if (lostFamilies) {
1156           LOG.error("LinkedListError: key=" + keyString + ", lost big or tiny families");
1157           context.getCounter(Counts.LOST_FAMILIES).increment(1);
1158           context.write(key, LOSTFAM);
1159         }
1160
1161         if (defCount == 0 && refs.size() > 0) {
1162           // This is bad, found a node that is referenced but not defined. It must have been
1163           // lost, emit some info about this node for debugging purposes.
1164           // Write out a line per reference. If more than one, flag it.;
1165           for (int i = 0; i < refs.size(); i++) {
1166             byte[] bs = refs.get(i);
1167             int ordinal;
1168             if (i <= 0) {
1169               ordinal = Counts.UNDEFINED.ordinal();
1170               context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
1171               context.getCounter(Counts.UNDEFINED).increment(1);
1172             } else {
1173               ordinal = Counts.EXTRA_UNDEF_REFERENCES.ordinal();
1174               context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
1175             }
1176           }
1177           if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
1178             // Print out missing row; doing get on reference gives info on when the referencer
1179             // was added which can help a little debugging. This info is only available in mapper
1180             // output -- the 'Linked List error Key...' log message above. What we emit here is
1181             // useless for debugging.
1182             context.getCounter("undef", keyString).increment(1);
1183           }
1184         } else if (defCount > 0 && refs.size() == 0) {
1185           // node is defined but not referenced
1186           context.write(key, UNREF);
1187           context.getCounter(Counts.UNREFERENCED).increment(1);
1188           if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
1189             context.getCounter("unref", keyString).increment(1);
1190           }
1191         } else {
1192           if (refs.size() > 1) {
1193             // Skip first reference.
1194             for (int i = 1; i < refs.size(); i++) {
1195               context.write(key,
1196                 new BytesWritable(addPrefixFlag(Counts.EXTRAREFERENCES.ordinal(), refs.get(i))));
1197             }
1198             context.getCounter(Counts.EXTRAREFERENCES).increment(refs.size() - 1);
1199           }
1200           // node is defined and referenced
1201           context.getCounter(Counts.REFERENCED).increment(1);
1202         }
1203       }

1354      * Verify the values in the Counters against the expected number of entries written.
1355      *
1356      * @param expectedReferenced
1357      *          Expected number of referenced entrires
1358      * @param counters
1359      *          The Job's Counters object
1360      * @return True if the values match what's expected, false otherwise
1361      */
1362     protected boolean verifyExpectedValues(long expectedReferenced, Counters counters) {
1363       final Counter referenced = counters.findCounter(Counts.REFERENCED);
1364       final Counter unreferenced = counters.findCounter(Counts.UNREFERENCED);
1365       boolean success = true;
1366
1367       if (expectedReferenced != referenced.getValue()) {
1368         LOG.error("Expected referenced count does not match with actual referenced count. " +
1369             "expected referenced=" + expectedReferenced + " ,actual=" + referenced.getValue());
1370         success = false;
1371       }
1372
1373       if (unreferenced.getValue() > 0) {
1374         final Counter multiref = counters.findCounter(Counts.EXTRAREFERENCES);
1375         boolean couldBeMultiRef = (multiref.getValue() == unreferenced.getValue());
1376         LOG.error("Unreferenced nodes were not expected. Unreferenced count=" + unreferenced.getValue()
1377             + (couldBeMultiRef ? "; could be due to duplicate random numbers" : ""));
1378         success = false;
1379       }
1380
1381       return success;
