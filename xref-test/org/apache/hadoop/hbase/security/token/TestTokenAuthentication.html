529   /**
530    * A copy of the BlockingRpcCallback class for use locally. Only difference is that it makes
531    * use of non-shaded protobufs; i.e. refers to com.google.protobuf.* rather than to
532    * org.apache.hadoop.hbase.shaded.com.google.protobuf.*
533    */
534   private static class NonShadedBlockingRpcCallback<R> implements
535       com.google.protobuf.RpcCallback<R> {
536     private R result;
537     private boolean resultSet = false;
538
539     /**
540      * Called on completion of the RPC call with the response object, or {@code null} in the case of
541      * an error.
542      * @param parameter the response object or {@code null} if an error occurred
543      */
544     @Override
545     public void run(R parameter) {
546       synchronized (this) {
547         result = parameter;
548         resultSet = true;
549         this.notifyAll();
550       }
551     }
552
553     /**
554      * Returns the parameter passed to {@link #run(Object)} or {@code null} if a null value was
555      * passed.  When used asynchronously, this method will block until the {@link #run(Object)}
556      * method has been called.
557      * @return the response object or {@code null} if no response was passed
558      */
559     public synchronized R get() throws IOException {
560       while (!resultSet) {
561         try {
562           this.wait();
563         } catch (InterruptedException ie) {
564           InterruptedIOException exception = new InterruptedIOException(ie.getMessage());
565           exception.initCause(ie);
566           throw exception;
567         }
568       }
569       return result;
570     }
