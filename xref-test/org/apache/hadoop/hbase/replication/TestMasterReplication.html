224   /**
225    * Tests the cyclic replication scenario of 0 -> 1 -> 2 -> 0 by adding and deleting rows to a
226    * table in each clusters and ensuring that the each of these clusters get the appropriate
227    * mutations. It also tests the grouping scenario where a cluster needs to replicate the edits
228    * originating from itself and also the edits that it received using replication from a different
229    * cluster. The scenario is explained in HBASE-9158
230    */
231   @Test(timeout = 300000)
232   public void testCyclicReplication2() throws Exception {
233     LOG.info("testCyclicReplication1");
234     int numClusters = 3;
235     Table[] htables = null;
236     try {
237       startMiniClusters(numClusters);
238       createTableOnClusters(table);
239
240       // Test the replication scenario of 0 -> 1 -> 2 -> 0
241       addPeer("1", 0, 1);
242       addPeer("1", 1, 2);
243       addPeer("1", 2, 0);
244
245       htables = getHTablesOnClusters(tableName);
246
247       // put "row" and wait 'til it got around
248       putAndWait(row, famName, htables[0], htables[2]);
249       putAndWait(row1, famName, htables[1], htables[0]);
250       putAndWait(row2, famName, htables[2], htables[1]);
251
252       deleteAndWait(row, htables[0], htables[2]);
253       deleteAndWait(row1, htables[1], htables[0]);
254       deleteAndWait(row2, htables[2], htables[1]);
255
256       int[] expectedCounts = new int[] { 3, 3, 3 };
257       validateCounts(htables, put, expectedCounts);
258       validateCounts(htables, delete, expectedCounts);
259
260       // Test HBASE-9158
261       disablePeer("1", 2);
262       // we now have an edit that was replicated into cluster originating from
263       // cluster 0
264       putAndWait(row3, famName, htables[0], htables[1]);
265       // now add a local edit to cluster 1
266       htables[1].put(new Put(row4).addColumn(famName, row4, row4));
267       // re-enable replication from cluster 2 to cluster 0
268       enablePeer("1", 2);
269       // without HBASE-9158 the edit for row4 would have been marked with
270       // cluster 0's id
271       // and hence not replicated to cluster 0
272       wait(row4, htables[0], true);
273     } finally {
274       close(htables);
275       shutDownMiniClusters();
276     }
277   }

435   @SuppressWarnings("resource")
436   private void startMiniClusters(int numClusters) throws Exception {
437     Random random = new Random();
438     utilities = new HBaseTestingUtility[numClusters];
439     configurations = new Configuration[numClusters];
440     for (int i = 0; i < numClusters; i++) {
441       Configuration conf = new Configuration(baseConfiguration);
442       conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/" + i + random.nextInt());
443       HBaseTestingUtility utility = new HBaseTestingUtility(conf);
444       if (i == 0) {
445         utility.startMiniZKCluster();
446         miniZK = utility.getZkCluster();
447       } else {
448         utility.setZkCluster(miniZK);
449       }
450       utility.startMiniCluster();
451       utilities[i] = utility;
452       configurations[i] = conf;
453       new ZooKeeperWatcher(conf, "cluster" + i, null, true);
454     }
455   }

457   private void shutDownMiniClusters() throws Exception {
458     int numClusters = utilities.length;
459     for (int i = numClusters - 1; i >= 0; i--) {
460       if (utilities[i] != null) {
461         utilities[i].shutdownMiniCluster();
462       }
463     }
464     miniZK.shutdown();
465   }

539     for (int i = 0; i < numClusters; i++) {

577   private void loadAndValidateHFileReplication(String testName, int masterNumber,
578       int[] slaveNumbers, byte[] row, byte[] fam, Table[] tables, byte[][][] hfileRanges,
579       int numOfRows, int[] expectedCounts, boolean toValidate) throws Exception {
580     HBaseTestingUtility util = utilities[masterNumber];
581
582     Path dir = util.getDataTestDirOnTestFS(testName);
583     FileSystem fs = util.getTestFileSystem();
584     dir = dir.makeQualified(fs);
585     Path familyDir = new Path(dir, Bytes.toString(fam));
586
587     int hfileIdx = 0;
588     for (byte[][] range : hfileRanges) {
589       byte[] from = range[0];
590       byte[] to = range[1];
591       HFileTestUtil.createHFile(util.getConfiguration(), fs,

608   private void wait(int slaveNumber, Table target, int expectedCount)
609       throws IOException, InterruptedException {
610     int count = 0;
611     for (int i = 0; i < NB_RETRIES; i++) {
612       if (i == NB_RETRIES - 1) {
613         fail("Waited too much time for bulkloaded data replication. Current count=" + count
614             + ", expected count=" + expectedCount);
615       }
616       count = utilities[slaveNumber].countRows(target);
617       if (count != expectedCount) {
618         LOG.info("Waiting more time for bulkloaded data replication.");
619         Thread.sleep(SLEEP_TIME);
620       } else {
621         break;
622       }
623     }
624   }

626   private void wait(byte[] row, Table target, boolean isDeleted) throws Exception {
627     Get get = new Get(row);
628     for (int i = 0; i < NB_RETRIES; i++) {
629       if (i == NB_RETRIES - 1) {
630         fail("Waited too much time for replication. Row:" + Bytes.toString(row)
631             + ". IsDeleteReplication:" + isDeleted);
632       }
633       Result res = target.get(get);
634       boolean sleep = isDeleted ? res.size() > 0 : res.size() == 0;
635       if (sleep) {
636         LOG.info("Waiting for more time for replication. Row:"
637             + Bytes.toString(row) + ". IsDeleteReplication:" + isDeleted);
638         Thread.sleep(SLEEP_TIME);
639       } else {
640         if (!isDeleted) {
641           assertArrayEquals(res.value(), row);
642         }
643         LOG.info("Obtained row:"
644             + Bytes.toString(row) + ". IsDeleteReplication:" + isDeleted);
645         break;
646       }
647     }

650   private void rollWALAndWait(final HBaseTestingUtility utility, final TableName table,
651       final byte[] row) throws IOException {
652     final Admin admin = utility.getHBaseAdmin();
653     final MiniHBaseCluster cluster = utility.getMiniHBaseCluster();
654
655     // find the region that corresponds to the given row.
656     HRegion region = null;
657     for (HRegion candidate : cluster.getRegions(table)) {
658       if (HRegion.rowIsInRange(candidate.getRegionInfo(), row)) {
659         region = candidate;
660         break;
661       }
662     }
663     assertNotNull("Couldn't find the region for row '" + Arrays.toString(row) + "'", region);
664
665     final CountDownLatch latch = new CountDownLatch(1);
666
667     // listen for successful log rolls
668     final WALActionsListener listener = new WALActionsListener.Base() {
669           @Override
670           public void postLogRoll(final Path oldPath, final Path newPath) throws IOException {
671             latch.countDown();
672           }
673         };
674     region.getWAL().registerWALActionsListener(listener);
675
676     // request a roll
677     admin.rollWALWriter(cluster.getServerHoldingRegion(region.getTableDesc().getTableName(),
678       region.getRegionInfo().getRegionName()));
679
680     // wait
681     try {
682       latch.await();
683     } catch (InterruptedException exception) {
684       LOG.warn("Interrupted while waiting for the wal of '" + region + "' to roll. If later " +
685           "replication tests fail, it's probably because we should still be waiting.");
686       Thread.currentThread().interrupt();
687     }
688     region.getWAL().unregisterWALActionsListener(listener);
689   }

691   /**
692    * Use a coprocessor to count puts and deletes. as KVs would be replicated back with the same
693    * timestamp there is otherwise no way to count them.
694    */
695   public static class CoprocessorCounter extends BaseRegionObserver {
696     private int nCount = 0;
697     private int nDelete = 0;
698
699     @Override
700     public void prePut(final ObserverContext<RegionCoprocessorEnvironment> e, final Put put,
701         final WALEdit edit, final Durability durability) throws IOException {
702       nCount++;
703     }
704
705     @Override
706     public void postDelete(final ObserverContext<RegionCoprocessorEnvironment> c,
707         final Delete delete, final WALEdit edit, final Durability durability) throws IOException {
708       nDelete++;
709     }
710
711     @Override
712     public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> c,
713         final Get get, final List<Cell> result) throws IOException {
714       if (get.getAttribute("count") != null) {
715         result.clear();
716         // order is important!
717         result.add(new KeyValue(count, count, delete, Bytes.toBytes(nDelete)));
718         result.add(new KeyValue(count, count, put, Bytes.toBytes(nCount)));
719         c.bypass();
720       }
721     }
