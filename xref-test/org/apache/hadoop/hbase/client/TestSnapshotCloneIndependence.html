76    static void setupConf(Configuration conf) {
77      // Up the handlers; this test needs more than usual.
78      conf.setInt(HConstants.REGION_SERVER_HIGH_PRIORITY_HANDLER_COUNT, 15);
79      // enable snapshot support
80      conf.setBoolean(SnapshotManager.HBASE_SNAPSHOT_ENABLED, true);
81      // disable the ui
82      conf.setInt("hbase.regionsever.info.port", -1);
83      conf.setInt("hbase.master.info.port", -1);
84      // change the flush size to a small amount, regulating number of store files
85      conf.setInt("hbase.hregion.memstore.flush.size", 25000);
86      // so make sure we get a compaction when doing a load, but keep around
87      // some files in the store
88      conf.setInt("hbase.hstore.compaction.min", 10);
89      conf.setInt("hbase.hstore.compactionThreshold", 10);
90      // block writes if we get to 12 store files
91      conf.setInt("hbase.hstore.blockingStoreFiles", 12);
92      conf.setInt("hbase.regionserver.msginterval", 100);
93      conf.setBoolean("hbase.master.enabletable.roundrobin", true);
94      // Avoid potentially aggressive splitting which would cause snapshot to fail
95      conf.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY,
96          ConstantSizeRegionSplitPolicy.class.getName());
97      // Execute cleaner frequently to induce failures
98      conf.setInt("hbase.master.cleaner.interval", CLEANER_INTERVAL);
99      conf.setInt("hbase.master.hfilecleaner.plugins.snapshot.period", CLEANER_INTERVAL);
100     // Effectively disable TimeToLiveHFileCleaner. Don't want to fully disable it because that
101     // will even trigger races between creating the directory containing back references and
102     // the back reference itself.
103     conf.setInt("hbase.master.hfilecleaner.ttl", CLEANER_INTERVAL);
104   }

195   private static void waitOnSplit(Connection c, final Table t, int originalCount) throws Exception {
196     for (int i = 0; i < 200; i++) {
197       try {
198         Thread.sleep(500);
199       } catch (InterruptedException e) {
200         // Restore the interrupted status
201         Thread.currentThread().interrupt();
202       }
203       try (RegionLocator locator = c.getRegionLocator(t.getName())) {
204         if (locator.getAllRegionLocations().size() > originalCount) {
205           return;
206         }
207       }
208     }
209     throw new Exception("Split did not increase the number of regions");
210   }

212   /*
213    * Take a snapshot of a table, add data, and verify that this only
214    * affects one table
215    * @param online - Whether the table is online or not during the snapshot
216    */
217   private void runTestSnapshotAppendIndependent(boolean online) throws Exception {
218     FileSystem fs = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getFileSystem();
219     Path rootDir = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getRootDir();
220 
221     Admin admin = UTIL.getHBaseAdmin();
222     final long startTime = System.currentTimeMillis();
223     final TableName localTableName =
224         TableName.valueOf(STRING_TABLE_NAME + startTime);
225 
226     try (Table original = createTable(localTableName, TEST_FAM)) {
227       loadData(original, TEST_FAM);
228       final int origTableRowCount = countRows(original);
229 
230       // Take a snapshot
231       final String snapshotNameAsString = "snapshot_" + localTableName;
232       byte[] snapshotName = Bytes.toBytes(snapshotNameAsString);
233 
234       SnapshotTestingUtils.createSnapshotAndValidate(admin, localTableName, TEST_FAM_STR,
235           snapshotNameAsString, rootDir, fs, online);
236 
237       if (!online) {
238         tryDisable(admin, localTableName);
239       }
240       TableName cloneTableName = TableName.valueOf("test-clone-" + localTableName);
241       admin.cloneSnapshot(snapshotName, cloneTableName);
242 
243       try (Table clonedTable = UTIL.getConnection().getTable(cloneTableName)) {
244 
245         // Make sure that all the regions are available before starting
246         UTIL.waitUntilAllRegionsAssigned(cloneTableName);
247 
248         final int clonedTableRowCount = countRows(clonedTable);
249 
250         Assert.assertEquals(
251             "The line counts of original and cloned tables do not match after clone. ",
252             origTableRowCount, clonedTableRowCount);
253 
254         // Attempt to add data to the test
255         final String rowKey = "new-row-" + System.currentTimeMillis();
256 
257         Put p = new Put(Bytes.toBytes(rowKey));
258         p.addColumn(TEST_FAM, Bytes.toBytes("someQualifier"), Bytes.toBytes("someString"));
259         original.put(p);
260 
261         // Verify that it is not present in the original table
262         Assert.assertEquals("The row count of the original table was not modified by the put",
263             origTableRowCount + 1, countRows(original));
264         Assert.assertEquals(
265             "The row count of the cloned table changed as a result of addition to the original",
266             clonedTableRowCount, countRows(clonedTable));
267 
268         p = new Put(Bytes.toBytes(rowKey));
269         p.addColumn(TEST_FAM, Bytes.toBytes("someQualifier"), Bytes.toBytes("someString"));
270         clonedTable.put(p);
271 
272         // Verify that the new family is not in the restored table's description
273         Assert.assertEquals(
274             "The row count of the original table was modified by the put to the clone",
275             origTableRowCount + 1, countRows(original));
276         Assert.assertEquals("The row count of the cloned table was not modified by the put",
277             clonedTableRowCount + 1, countRows(clonedTable));
278       }
279     }
280   }

282   /*
283    * Take a snapshot of a table, do a split, and verify that this only affects one table
284    * @param online - Whether the table is online or not during the snapshot
285    */
286   private void runTestRegionOperationsIndependent(boolean online) throws Exception {
287     FileSystem fs = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getFileSystem();
288     Path rootDir = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getRootDir();
289 
290     // Create a table
291     Admin admin = UTIL.getHBaseAdmin();
292     final long startTime = System.currentTimeMillis();
293     final TableName localTableName =
294         TableName.valueOf(STRING_TABLE_NAME + startTime);
295     Table original = createTable(localTableName, TEST_FAM);
296     loadData(original, TEST_FAM);
297     final int loadedTableCount = countRows(original);
298     System.out.println("Original table has: " + loadedTableCount + " rows");
299 
300     final String snapshotNameAsString = "snapshot_" + localTableName;
301 
302     // Create a snapshot
303     SnapshotTestingUtils.createSnapshotAndValidate(admin, localTableName, TEST_FAM_STR,
304         snapshotNameAsString, rootDir, fs, online);
305 
306     if (!online) {
307       tryDisable(admin, localTableName);
308     }
309 
310     TableName cloneTableName = TableName.valueOf("test-clone-" + localTableName);
311 
312     // Clone the snapshot
313     byte[] snapshotName = Bytes.toBytes(snapshotNameAsString);
314     admin.cloneSnapshot(snapshotName, cloneTableName);
315 
316     // Verify that region information is the same pre-split
317     ((ClusterConnection) UTIL.getConnection()).clearRegionCache();
318     List<HRegionInfo> originalTableHRegions = admin.getTableRegions(localTableName);
319 
320     final int originalRegionCount = originalTableHRegions.size();
321     final int cloneTableRegionCount = admin.getTableRegions(cloneTableName).size();
322     Assert.assertEquals(
323         "The number of regions in the cloned table is different than in the original table.",
324         originalRegionCount, cloneTableRegionCount);
325 
326     // Split a region on the parent table
327     admin.splitRegion(originalTableHRegions.get(0).getRegionName());
328     waitOnSplit(UTIL.getConnection(), original, originalRegionCount);
329 
330     // Verify that the cloned table region is not split
331     final int cloneTableRegionCount2 = admin.getTableRegions(cloneTableName).size();
332     Assert.assertEquals(
333         "The number of regions in the cloned table changed though none of its regions were split.",
334         cloneTableRegionCount, cloneTableRegionCount2);
335   }

382     // get a list of its families
383     // assert that the family is there
384     HTableDescriptor originalTableDescriptor = original.getTableDescriptor();
385     HTableDescriptor clonedTableDescriptor = admin.getTableDescriptor(cloneTableName);
386 
387     Assert.assertTrue("The original family was not found. There is something wrong. ",
388         originalTableDescriptor.hasFamily(TEST_FAM));
389     Assert.assertTrue("The original family was not found in the clone. There is something wrong. ",
390         clonedTableDescriptor.hasFamily(TEST_FAM));
391 
392     Assert.assertTrue("The new family was not found. ",
393         originalTableDescriptor.hasFamily(TEST_FAM_2));
394     Assert.assertTrue("The new family was not found. ",
395         !clonedTableDescriptor.hasFamily(TEST_FAM_2));
396   }

398   private void tryDisable(Admin admin, TableName localTableName) throws IOException {
399     int offlineRetry = 0;
400     while ( offlineRetry < 5 && admin.isTableEnabled(localTableName)) {
401       try {
402         admin.disableTable(localTableName);
403       } catch (IOException ioe) {
404         LOG.warn("Error disabling the table", ioe);
405       }
406       offlineRetry ++;
407     }
408   }

410   /*
411    * Take a snapshot of a table, add data, and verify that deleting the snapshot does not affect
412    * either table.
413    * @param online - Whether the table is online or not during the snapshot
414    */
415   private void runTestSnapshotDeleteIndependent(boolean online) throws Exception {
416     FileSystem fs = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getFileSystem();
417     Path rootDir = UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getRootDir();
418 
419     final Admin admin = UTIL.getHBaseAdmin();
420     final long startTime = System.currentTimeMillis();
421     final TableName localTableName =
422         TableName.valueOf(STRING_TABLE_NAME + startTime);
423 
424     try (Table original = createTable(localTableName, TEST_FAM)) {
425       loadData(original, TEST_FAM);
426     }
427 
428     // Take a snapshot
429     final String snapshotNameAsString = "snapshot_" + localTableName;
430     byte[] snapshotName = Bytes.toBytes(snapshotNameAsString);
431 
432     SnapshotTestingUtils.createSnapshotAndValidate(admin, localTableName, TEST_FAM_STR,
433         snapshotNameAsString, rootDir, fs, online);
434 
435     if (!online) {
436       tryDisable(admin, localTableName);
437     }
438 
439     TableName cloneTableName = TableName.valueOf("test-clone-" + localTableName);
440     admin.cloneSnapshot(snapshotName, cloneTableName);
441 
442     UTIL.waitUntilAllRegionsAssigned(cloneTableName);
443 
444     // Ensure the original table does not reference the HFiles anymore
445     admin.majorCompact(localTableName);
446 
447     // Deleting the snapshot used to break the cloned table by deleting in-use HFiles
448     admin.deleteSnapshot(snapshotName);
449 
450     // Wait for cleaner run and DFS heartbeats so that anything that is deletable is fully deleted
451     do {
452       Thread.sleep(5000);
453     } while (!admin.listSnapshots(snapshotNameAsString).isEmpty());
454 
455     try (Table original = UTIL.getConnection().getTable(localTableName)) {
456       try (Table clonedTable = UTIL.getConnection().getTable(cloneTableName)) {
457         // Verify that all regions of both tables are readable
458         final int origTableRowCount = countRows(original);
459         final int clonedTableRowCount = countRows(clonedTable);
460         Assert.assertEquals(origTableRowCount, clonedTableRowCount);
461       }
462     }
463   }
