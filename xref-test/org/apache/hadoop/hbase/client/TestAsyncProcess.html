329   class MyAsyncProcessWithReplicas extends MyAsyncProcess {
330     private Set<byte[]> failures = new TreeSet<byte[]>(new Bytes.ByteArrayComparator());
331     private long primarySleepMs = 0, replicaSleepMs = 0;
332     private Map<ServerName, Long> customPrimarySleepMs = new HashMap<ServerName, Long>();
333     private final AtomicLong replicaCalls = new AtomicLong(0);
334
335     public void addFailures(HRegionInfo... hris) {
336       for (HRegionInfo hri : hris) {
337         failures.add(hri.getRegionName());
338       }
339     }
340
341     public long getReplicaCallCount() {
342       return replicaCalls.get();
343     }
344
345     public void setPrimaryCallDelay(ServerName server, long primaryMs) {
346       customPrimarySleepMs.put(server, primaryMs);
347     }
348
349     public MyAsyncProcessWithReplicas(ClusterConnection hc, Configuration conf) {
350       super(hc, conf);
351     }
352
353     public void setCallDelays(long primaryMs, long replicaMs) {
354       this.primarySleepMs = primaryMs;
355       this.replicaSleepMs = replicaMs;
356     }
357
358     @Override
359     protected RpcRetryingCaller<AbstractResponse> createCaller(
360         CancellableRegionServerCallable payloadCallable, int rpcTimeout) {
361       MultiServerCallable callable = (MultiServerCallable) payloadCallable;
362       final MultiResponse mr = createMultiResponse(
363           callable.getMulti(), nbMultiResponse, nbActions, new ResponseGenerator() {
364             @Override
365             public void addResponse(MultiResponse mr, byte[] regionName, Action a) {
366               if (failures.contains(regionName)) {
367                 mr.add(regionName, a.getOriginalIndex(), failure);
368               } else {
369                 boolean isStale = !RegionReplicaUtil.isDefaultReplica(a.getReplicaId());
370                 mr.add(regionName, a.getOriginalIndex(),
371                     Result.create(new Cell[0], null, isStale));
372               }
373             }
374           });
375       // Currently AsyncProcess either sends all-replica, or all-primary request.
376       final boolean isDefault = RegionReplicaUtil.isDefaultReplica(
377           callable.getMulti().actions.values().iterator().next().iterator().next().getReplicaId());
378       final ServerName server = ((MultiServerCallable)callable).getServerName();
379       String debugMsg = "Call to " + server + ", primary=" + isDefault + " with "
380           + callable.getMulti().actions.size() + " entries: ";
381       for (byte[] region : callable.getMulti().actions.keySet()) {
382         debugMsg += "[" + Bytes.toStringBinary(region) + "], ";
383       }
384       LOG.debug(debugMsg);
385       if (!isDefault) {
386         replicaCalls.incrementAndGet();
387       }
388
389       return new RpcRetryingCallerImpl<AbstractResponse>(100, 10, 9) {
390         @Override
391         public MultiResponse callWithoutRetries(RetryingCallable<AbstractResponse> callable,
392                                                 int callTimeout)
393         throws IOException, RuntimeException {
394           long sleep = -1;
395           if (isDefault) {
396             Long customSleep = customPrimarySleepMs.get(server);
397             sleep = (customSleep == null ? primarySleepMs : customSleep.longValue());
398           } else {
399             sleep = replicaSleepMs;
400           }
401           if (sleep != 0) {
402             try {
403               Thread.sleep(sleep);
404             } catch (InterruptedException e) {
405             }
406           }
407           return mr;
408         }
409       };
410     }

413   static MultiResponse createMultiResponse(final MultiAction multi,
414       AtomicInteger nbMultiResponse, AtomicInteger nbActions, ResponseGenerator gen) {
415     final MultiResponse mr = new MultiResponse();
416     nbMultiResponse.incrementAndGet();
417     for (Map.Entry<byte[], List<Action>> entry : multi.actions.entrySet()) {
418       byte[] regionName = entry.getKey();
419       for (Action a : entry.getValue()) {
420         nbActions.incrementAndGet();
421         gen.addResponse(mr, regionName, a);
422       }
423     }
424     return mr;
425   }
