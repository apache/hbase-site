37    @Test
38    public void testCompactionRatio() throws IOException {
39      TimeOffsetEnvironmentEdge edge = new TimeOffsetEnvironmentEdge();
40      EnvironmentEdgeManager.injectEdge(edge);
41      /**
42       * NOTE: these tests are specific to describe the implementation of the
43       * current compaction algorithm.  Developed to ensure that refactoring
44       * doesn't implicitly alter this.
45       */
46      long tooBig = maxSize + 1;
47
48      // default case. preserve user ratio on size
49      compactEquals(sfCreate(100,50,23,12,12), 23, 12, 12);
50      // less than compact threshold = don't compact
51      compactEquals(sfCreate(100,50,25,12,12) /* empty */);
52      // greater than compact size = skip those
53      compactEquals(sfCreate(tooBig, tooBig, 700, 700, 700), 700, 700, 700);
54      // big size + threshold
55      compactEquals(sfCreate(tooBig, tooBig, 700,700) /* empty */);
56      // small files = don't care about ratio
57      compactEquals(sfCreate(7,1,1), 7,1,1);
58  
59      // don't exceed max file compact threshold
60      // note:  file selection starts with largest to smallest.
61      compactEquals(sfCreate(7, 6, 5, 4, 3, 2, 1), 5, 4, 3, 2, 1);
62  
63      compactEquals(sfCreate(50, 10, 10 ,10, 10), 10, 10, 10, 10);
64  
65      compactEquals(sfCreate(10, 10, 10, 10, 50), 10, 10, 10, 10);
66  
67      compactEquals(sfCreate(251, 253, 251, maxSize -1), 251, 253, 251);
68
69      compactEquals(sfCreate(maxSize -1,maxSize -1,maxSize -1) /* empty */);
70
71      // Always try and compact something to get below blocking storefile count
72      this.conf.setLong("hbase.hstore.compaction.min.size", 1);
73      store.storeEngine.getCompactionPolicy().setConf(conf);
74      compactEquals(sfCreate(512,256,128,64,32,16,8,4,2,1), 4,2,1);
75      this.conf.setLong("hbase.hstore.compaction.min.size", minSize);
76      store.storeEngine.getCompactionPolicy().setConf(conf);
77
78      /* MAJOR COMPACTION */
79      // if a major compaction has been forced, then compact everything
80      compactEquals(sfCreate(50,25,12,12), true, 50, 25, 12, 12);
81      // also choose files < threshold on major compaction
82      compactEquals(sfCreate(12,12), true, 12, 12);
83      // even if one of those files is too big
84      compactEquals(sfCreate(tooBig, 12,12), true, tooBig, 12, 12);
85      // don't exceed max file compact threshold, even with major compaction
86      store.forceMajor = true;
87      compactEquals(sfCreate(7, 6, 5, 4, 3, 2, 1), 5, 4, 3, 2, 1);
88      store.forceMajor = false;
89      // if we exceed maxCompactSize, downgrade to minor
90      // if not, it creates a 'snowball effect' when files >> maxCompactSize:
91      // the last file in compaction is the aggregate of all previous compactions
92      compactEquals(sfCreate(100,50,23,12,12), true, 23, 12, 12);
93      conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, 1);
94      conf.setFloat("hbase.hregion.majorcompaction.jitter", 0);
95      store.storeEngine.getCompactionPolicy().setConf(conf);
96      try {
97        // The modTime of the mocked store file is currentTimeMillis, so we need to increase the
98        // timestamp a bit to make sure that now - lowestModTime is greater than major compaction
99        // period(1ms).
100       // trigger an aged major compaction
101       List<StoreFile> candidates = sfCreate(50, 25, 12, 12);
102       edge.increment(2);
103       compactEquals(candidates, 50, 25, 12, 12);
104       // major sure exceeding maxCompactSize also downgrades aged minors
105       candidates = sfCreate(100, 50, 23, 12, 12);
106       edge.increment(2);
107       compactEquals(candidates, 23, 12, 12);
108     } finally {
109       conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, 1000*60*60*24);
110       conf.setFloat("hbase.hregion.majorcompaction.jitter", 0.20F);
111     }
112
113     /* REFERENCES == file is from a region that was split */
114     // treat storefiles that have references like a major compaction
115     compactEquals(sfCreate(true, 100,50,25,12,12), 100, 50, 25, 12, 12);
116     // reference files shouldn't obey max threshold
117     compactEquals(sfCreate(true, tooBig, 12,12), tooBig, 12, 12);
118     // reference files should obey max file compact to avoid OOM
119     compactEquals(sfCreate(true, 7, 6, 5, 4, 3, 2, 1), 7, 6, 5, 4, 3);
120
121     // empty case
122     compactEquals(new ArrayList<StoreFile>() /* empty */);
123     // empty case (because all files are too big)
124     compactEquals(sfCreate(tooBig, tooBig) /* empty */);
125   }

142   @Test
143   public void testStuckStoreCompaction() throws IOException {
144     // Select the smallest compaction if the store is stuck.
145     compactEquals(sfCreate(99,99,99,99,99,99, 30,30,30,30), 30, 30, 30);
146     // If not stuck, standard policy applies.
147     compactEquals(sfCreate(99,99,99,99,99, 30,30,30,30), 99, 30, 30, 30, 30);
148
149     // Add sufficiently small files to compaction, though
150     compactEquals(sfCreate(99,99,99,99,99,99, 30,30,30,15), 30, 30, 30, 15);
151     // Prefer earlier compaction to latter if the benefit is not significant
152     compactEquals(sfCreate(99,99,99,99, 30,26,26,29,25,25), 30, 26, 26);
153     // Prefer later compaction if the benefit is significant.
154     compactEquals(sfCreate(99,99,99,99, 27,27,27,20,20,20), 20, 20, 20);
155   }

157   @Test
158   public void testCompactionEmptyHFile() throws IOException {
159     // Set TTL
160     ScanInfo oldScanInfo = store.getScanInfo();
161     ScanInfo newScanInfo = new ScanInfo(oldScanInfo.getConfiguration(), oldScanInfo.getFamily(),
162         oldScanInfo.getMinVersions(), oldScanInfo.getMaxVersions(), 600,
163         oldScanInfo.getKeepDeletedCells(), oldScanInfo.getTimeToPurgeDeletes(),
164         oldScanInfo.getComparator());
165     store.setScanInfo(newScanInfo);
166     // Do not compact empty store file
167     List<StoreFile> candidates = sfCreate(0);
168     for (StoreFile file : candidates) {
169       if (file instanceof MockStoreFile) {
170         MockStoreFile mockFile = (MockStoreFile) file;
171         mockFile.setTimeRangeTracker(new TimeRangeTracker(-1, -1));
172         mockFile.setEntries(0);
173       }
174     }
175     // Test Default compactions
176     CompactionRequest result = ((RatioBasedCompactionPolicy) store.storeEngine
177         .getCompactionPolicy()).selectCompaction(candidates,
178         new ArrayList<StoreFile>(), false, false, false);
179     Assert.assertTrue(result.getFiles().size() == 0);
180     store.setScanInfo(oldScanInfo);
181   }
