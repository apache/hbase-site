236   private void createRecordAndCorruptMobFile(TableName tn, byte[] row, byte[] family, byte[] qf,
237     byte[] value) throws IOException {
238     Put put1 = new Put(row);
239     put1.addColumn(family, qf, value);
240     table.put(put1);
241     admin.flush(tn);
242     Path mobFile = getFlushedMobFile(conf, fs, tn, Bytes.toString(family));
243     Assert.assertNotNull(mobFile);
244     // create new corrupt mob file.
245     Path corruptFile = new Path(mobFile.getParent(), "dummy");
246     TestHFile.truncateFile(fs, mobFile, corruptFile);
247     fs.delete(mobFile, true);
248     fs.rename(corruptFile, mobFile);
249   }

251   private Path getFlushedMobFile(Configuration conf, FileSystem fs, TableName table, String family)
252     throws IOException {
253     Path famDir = MobUtils.getMobFamilyPath(conf, table, family);
254     FileStatus[] hfFss = fs.listStatus(famDir);
255     for (FileStatus hfs : hfFss) {
256       if (!hfs.isDirectory()) {
257         return hfs.getPath();
258       }
259     }
260     return null;
261   }

329   private void testMobThreshold(boolean reversed) throws Exception {
330     TableName tn = TableName.valueOf("testMobThreshold" + reversed);
331     setUp(defaultThreshold, tn);
332     byte [] valueLess = generateMobValue((int)defaultThreshold-1);
333     byte [] valueEqual = generateMobValue((int)defaultThreshold);
334     byte [] valueGreater = generateMobValue((int)defaultThreshold+1);
335     long ts1 = System.currentTimeMillis();
336     long ts2 = ts1 + 1;
337     long ts3 = ts1 + 2;
338 
339     Put put1 = new Put(row1);
340     put1.addColumn(family, qf1, ts3, valueLess);
341     put1.addColumn(family, qf2, ts2, valueEqual);
342     put1.addColumn(family, qf3, ts1, valueGreater);
343     table.put(put1);
344 
345     admin.flush(tn);
346 
347     Scan scan = new Scan();
348     setScan(scan, reversed, true);
349 
350     Cell cellLess= null;
351     Cell cellEqual = null;
352     Cell cellGreater = null;
353     ResultScanner results = table.getScanner(scan);
354     int count = 0;
355     for (Result res : results) {
356       List<Cell> cells = res.listCells();
357       for(Cell cell : cells) {
358         // Verify the value
359         String qf = Bytes.toString(CellUtil.cloneQualifier(cell));
360         if(qf.equals(Bytes.toString(qf1))) {
361           cellLess = cell;
362         }
363         if(qf.equals(Bytes.toString(qf2))) {
364           cellEqual = cell;
365         }
366         if(qf.equals(Bytes.toString(qf3))) {
367           cellGreater = cell;
368         }
369         count++;
370       }
371     }
372     Assert.assertEquals(3, count);
373     assertNotMobReference(cellLess, row1, family, valueLess);
374     assertNotMobReference(cellEqual, row1, family, valueEqual);
375     assertIsMobReference(cellGreater, row1, family, valueGreater, tn);
376     results.close();
377   }

379   private void testGetFromArchive(boolean reversed) throws Exception {
380     TableName tn = TableName.valueOf("testGetFromArchive" + reversed);
381     setUp(defaultThreshold, tn);
382     long ts1 = System.currentTimeMillis();
383     long ts2 = ts1 + 1;
384     long ts3 = ts1 + 2;
385     byte [] value = generateMobValue((int)defaultThreshold+1);;
386     // Put some data
387     Put put1 = new Put(row1);
388     put1.addColumn(family, qf1, ts3, value);
389     put1.addColumn(family, qf2, ts2, value);
390     put1.addColumn(family, qf3, ts1, value);
391     table.put(put1);
392 
393     admin.flush(tn);
394 
395     // Get the files in the mob path
396     Path mobFamilyPath;
397     mobFamilyPath = MobUtils.getMobFamilyPath(
398       TEST_UTIL.getConfiguration(), tn, hcd.getNameAsString());
399     FileSystem fs = FileSystem.get(TEST_UTIL.getConfiguration());
400     FileStatus[] files = fs.listStatus(mobFamilyPath);
401 
402     // Get the archive path
403     Path rootDir = FSUtils.getRootDir(TEST_UTIL.getConfiguration());
404     Path tableDir = FSUtils.getTableDir(rootDir, tn);
405     HRegionInfo regionInfo = MobUtils.getMobRegionInfo(tn);
406     Path storeArchiveDir = HFileArchiveUtil.getStoreArchivePath(TEST_UTIL.getConfiguration(),
407         regionInfo, tableDir, family);
408 
409     // Move the files from mob path to archive path
410     fs.mkdirs(storeArchiveDir);
411     int fileCount = 0;
412     for(FileStatus file : files) {
413       fileCount++;
414       Path filePath = file.getPath();
415       Path src = new Path(mobFamilyPath, filePath.getName());
416       Path dst = new Path(storeArchiveDir, filePath.getName());
417       fs.rename(src, dst);
418     }
419 
420     // Verify the moving success
421     FileStatus[] files1 = fs.listStatus(mobFamilyPath);
422     Assert.assertEquals(0, files1.length);
423     FileStatus[] files2 = fs.listStatus(storeArchiveDir);
424     Assert.assertEquals(fileCount, files2.length);
425 
426     // Scan from archive
427     Scan scan = new Scan();
428     setScan(scan, reversed, false);
429     MobTestUtil.assertCellsValue(table, scan, value, 3);
430   }

435   private static void assertNotMobReference(Cell cell, byte[] row, byte[] family,

442   /**
443    * Assert the value is store in mob.
444    */
445   private static void assertIsMobReference(Cell cell, byte[] row, byte[] family,
446       byte[] value, TableName tn) throws IOException {
447     Assert.assertArrayEquals(row, CellUtil.cloneRow(cell));
448     Assert.assertArrayEquals(family, CellUtil.cloneFamily(cell));
449     Assert.assertFalse(Bytes.equals(value, CellUtil.cloneValue(cell)));
450     byte[] referenceValue = CellUtil.cloneValue(cell);
451     String fileName = MobUtils.getMobFileName(cell);
452     int valLen = Bytes.toInt(referenceValue, 0, Bytes.SIZEOF_INT);
453     Assert.assertEquals(value.length, valLen);
454     Path mobFamilyPath = MobUtils.getMobFamilyPath(
455       TEST_UTIL.getConfiguration(), tn, hcd.getNameAsString());
456     Path targetPath = new Path(mobFamilyPath, fileName);
457     FileSystem fs = FileSystem.get(TEST_UTIL.getConfiguration());
458     Assert.assertTrue(fs.exists(targetPath));
459   }
