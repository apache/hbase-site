109   @Test
110   public void testScanAcrossSnapshot2() throws IOException, CloneNotSupportedException {
111     // we are going to the scanning across snapshot with two kvs
112     // kv1 should always be returned before kv2
113     final byte[] one = Bytes.toBytes(1);
114     final byte[] two = Bytes.toBytes(2);
115     final byte[] f = Bytes.toBytes("f");
116     final byte[] q = Bytes.toBytes("q");
117     final byte[] v = Bytes.toBytes(3);
118
119     final KeyValue kv1 = new KeyValue(one, f, q, 10, v);
120     final KeyValue kv2 = new KeyValue(two, f, q, 10, v);
121
122     // use case 1: both kvs in kvset
123     this.memstore.add(kv1.clone(), null);
124     this.memstore.add(kv2.clone(), null);
125     verifyScanAcrossSnapshot2(kv1, kv2);
126
127     // use case 2: both kvs in snapshot
128     this.memstore.snapshot();
129     verifyScanAcrossSnapshot2(kv1, kv2);
130
131     // use case 3: first in snapshot second in kvset
132     this.memstore = new CompactingMemStore(HBaseConfiguration.create(),
133         CellComparator.COMPARATOR, store, regionServicesForStores);
134     this.memstore.add(kv1.clone(), null);
135     // As compaction is starting in the background the repetition
136     // of the k1 might be removed BUT the scanners created earlier
137     // should look on the OLD MutableCellSetSegment, so this should be OK...
138     this.memstore.snapshot();
139     this.memstore.add(kv2.clone(), null);
140     verifyScanAcrossSnapshot2(kv1,kv2);

167   @Override
168   @Test
169   public void testGetNextRow() throws Exception {
170     addRows(this.memstore);
171     // Add more versions to make it a little more interesting.
172     Thread.sleep(1);
173     addRows(this.memstore);
174     Cell closestToEmpty = ((CompactingMemStore)this.memstore).getNextRow(KeyValue.LOWESTKEY);
175     assertTrue(CellComparator.COMPARATOR.compareRows(closestToEmpty,
176         new KeyValue(Bytes.toBytes(0), System.currentTimeMillis())) == 0);
177     for (int i = 0; i < ROW_COUNT; i++) {
178       Cell nr = ((CompactingMemStore)this.memstore).getNextRow(new KeyValue(Bytes.toBytes(i),
179           System.currentTimeMillis()));
180       if (i + 1 == ROW_COUNT) {
181         assertEquals(nr, null);
182       } else {
183         assertTrue(CellComparator.COMPARATOR.compareRows(nr,
184             new KeyValue(Bytes.toBytes(i + 1), System.currentTimeMillis())) == 0);
185       }
186     }
187     //starting from each row, validate results should contain the starting row
188     Configuration conf = HBaseConfiguration.create();
189     for (int startRowId = 0; startRowId < ROW_COUNT; startRowId++) {
190       ScanInfo scanInfo = new ScanInfo(conf, FAMILY, 0, 1, Integer.MAX_VALUE,
191         KeepDeletedCells.FALSE, 0, this.memstore.getComparator());
192       ScanType scanType = ScanType.USER_SCAN;
193       InternalScanner scanner = new StoreScanner(new Scan(
194           Bytes.toBytes(startRowId)), scanInfo, scanType, null,
195           memstore.getScanners(0));
196       List<Cell> results = new ArrayList<Cell>();
197       for (int i = 0; scanner.next(results); i++) {
198         int rowId = startRowId + i;
199         Cell left = results.get(0);
200         byte[] row1 = Bytes.toBytes(rowId);
201         assertTrue("Row name",
202             CellComparator.COMPARATOR.compareRows(left, row1, 0, row1.length) == 0);
203         assertEquals("Count of columns", QUALIFIER_COUNT, results.size());
204         List<Cell> row = new ArrayList<Cell>();
205         for (Cell kv : results) {
206           row.add(kv);
207         }
208         isExpectedRowWithoutTimestamps(rowId, row);
209         // Clear out set.  Otherwise row results accumulate.
210         results.clear();
211       }
212     }
213   }

290    * Tests that the timeOfOldestEdit is updated correctly for the
291    * various edit operations in memstore.
292    * @throws Exception
293    */
294   @Override
295   @Test
296   public void testUpdateToTimeOfOldestEdit() throws Exception {
297     try {
298       EnvironmentEdgeForMemstoreTest edge = new EnvironmentEdgeForMemstoreTest();
299       EnvironmentEdgeManager.injectEdge(edge);
300       long t = memstore.timeOfOldestEdit();
301       assertEquals(t, Long.MAX_VALUE);
302
303       // test the case that the timeOfOldestEdit is updated after a KV add
304       memstore.add(KeyValueTestUtil.create("r", "f", "q", 100, "v"), null);
305       t = memstore.timeOfOldestEdit();
306       assertTrue(t == 1234);
307       // The method will also assert
308       // the value is reset to Long.MAX_VALUE
309       t = runSnapshot(memstore, true);
310
311       // test the case that the timeOfOldestEdit is updated after a KV delete
312       memstore.add(KeyValueTestUtil.create("r", "f", "q", 100, KeyValue.Type.Delete, "v"), null);
313       t = memstore.timeOfOldestEdit();
314       assertTrue(t == 1234);
315      t = runSnapshot(memstore, true);
316
317       // test the case that the timeOfOldestEdit is updated after a KV upsert
318       List<Cell> l = new ArrayList<Cell>();
319       KeyValue kv1 = KeyValueTestUtil.create("r", "f", "q", 100, "v");
320       kv1.setSequenceId(100);
321       l.add(kv1);
322       memstore.upsert(l, 1000, null);
323       t = memstore.timeOfOldestEdit();
324       assertTrue(t == 1234);
325     } finally {
326       EnvironmentEdgeManager.reset();
327     }
328   }

330   private long runSnapshot(final AbstractMemStore hmc, boolean useForce)
331       throws IOException {
332     // Save off old state.
333     long oldHistorySize = hmc.getSnapshot().keySize();
334     long prevTimeStamp = hmc.timeOfOldestEdit();
335
336     hmc.snapshot();
337     MemStoreSnapshot snapshot = hmc.snapshot();
338     if (useForce) {
339       // Make some assertions about what just happened.
340       assertTrue("History size has not increased", oldHistorySize < snapshot.getDataSize());
341       long t = hmc.timeOfOldestEdit();
342       assertTrue("Time of oldest edit is not Long.MAX_VALUE", t == Long.MAX_VALUE);
343       hmc.clearSnapshot(snapshot.getId());
344     } else {
345       long t = hmc.timeOfOldestEdit();
346       assertTrue("Time of oldest edit didn't remain the same", t == prevTimeStamp);
347     }
348     return prevTimeStamp;
349   }

351   private void isExpectedRowWithoutTimestamps(final int rowIndex,
352       List<Cell> kvs) {
353     int i = 0;
354     for (Cell kv : kvs) {
355       byte[] expectedColname = makeQualifier(rowIndex, i++);
356       assertTrue("Column name", CellUtil.matchingQualifier(kv, expectedColname));
357       // Value is column name as bytes.  Usually result is
358       // 100 bytes in size at least. This is the default size
359       // for BytesWriteable.  For comparison, convert bytes to
360       // String and trim to remove trailing null bytes.
361       assertTrue("Content", CellUtil.matchingValue(kv, expectedColname));
362     }
363   }

462   public void testPuttingBackChunksWithOpeningPipelineScanner()
463       throws IOException {
464
465     // set memstore to do data compaction and not to use the speculative scan
466     memstore.getConfiguration().set("hbase.hregion.compacting.memstore.type", "data-compaction");
467     ((CompactingMemStore)memstore).initiateType();
468
469     byte[] row = Bytes.toBytes("testrow");
470     byte[] fam = Bytes.toBytes("testfamily");
471     byte[] qf1 = Bytes.toBytes("testqualifier1");
472     byte[] qf2 = Bytes.toBytes("testqualifier2");
473     byte[] qf3 = Bytes.toBytes("testqualifier3");
474     byte[] val = Bytes.toBytes("testval");
475
476     // Setting up memstore
477     memstore.add(new KeyValue(row, fam, qf1, 1, val), null);
478     memstore.add(new KeyValue(row, fam, qf2, 1, val), null);
479     memstore.add(new KeyValue(row, fam, qf3, 1, val), null);
480
481     // Creating a pipeline
482     ((CompactingMemStore)memstore).disableCompaction();
483     ((CompactingMemStore)memstore).flushInMemory();
484
485     // Adding value to "new" memstore
486     assertEquals(0, memstore.getActive().getCellsCount());
487     memstore.add(new KeyValue(row, fam, qf1, 2, val), null);
488     memstore.add(new KeyValue(row, fam, qf2, 2, val), null);
489     assertEquals(2, memstore.getActive().getCellsCount());
490
491     // pipeline bucket 2
492     ((CompactingMemStore)memstore).flushInMemory();
493     // opening scanner before force flushing
494     List<KeyValueScanner> scanners = memstore.getScanners(0);
495     // Shouldn't putting back the chunks to pool,since some scanners are opening
496     // based on their data
497     ((CompactingMemStore)memstore).enableCompaction();
498     // trigger compaction
499     ((CompactingMemStore)memstore).flushInMemory();
500
501     // Adding value to "new" memstore
502     assertEquals(0, memstore.getActive().getCellsCount());
503     memstore.add(new KeyValue(row, fam, qf3, 3, val), null);
504     memstore.add(new KeyValue(row, fam, qf2, 3, val), null);

506     assertEquals(3, memstore.getActive().getCellsCount());
507
508     assertTrue(chunkPool.getPoolSize() == 0);
509
510     // Chunks will be put back to pool after close scanners;
511     for (KeyValueScanner scanner : scanners) {
512       scanner.close();
513     }
514     assertTrue(chunkPool.getPoolSize() > 0);
515
516     // clear chunks
517     chunkPool.clearChunks();
518
519     // Creating another snapshot
520
521     MemStoreSnapshot snapshot = memstore.snapshot();
522     memstore.clearSnapshot(snapshot.getId());
523
524     snapshot = memstore.snapshot();
525     // Adding more value
526     memstore.add(new KeyValue(row, fam, qf2, 4, val), null);
527     memstore.add(new KeyValue(row, fam, qf3, 4, val), null);
528     // opening scanners
529     scanners = memstore.getScanners(0);
530     // close scanners before clear the snapshot
531     for (KeyValueScanner scanner : scanners) {
532       scanner.close();
533     }
534     // Since no opening scanner, the chunks of snapshot should be put back to
535     // pool
536     memstore.clearSnapshot(snapshot.getId());
537     assertTrue(chunkPool.getPoolSize() > 0);
538   }

545
546     // set memstore to do data compaction and not to use the speculative scan
547     memstore.getConfiguration().set("hbase.hregion.compacting.memstore.type", "data-compaction");
548     ((CompactingMemStore)memstore).initiateType();

552     // test 1 bucket
553     int totalCellsLen = addRowsByKeys(memstore, keys1);
554     long totalHeapOverhead = 4 * (KeyValue.FIXED_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
555     assertEquals(totalCellsLen, regionServicesForStores.getMemstoreSize());
556     assertEquals(totalHeapOverhead, ((CompactingMemStore)memstore).heapOverhead());
557
558     MemstoreSize size = memstore.getFlushableSize();
559     ((CompactingMemStore)memstore).flushInMemory(); // push keys to pipeline and compact
560     assertEquals(0, memstore.getSnapshot().getCellsCount());
561     // One cell is duplicated and the compaction will remove it. All cells of same size so adjusting
562     // totalCellsLen
563     totalCellsLen = (totalCellsLen * 3) / 4;
564     totalHeapOverhead = 3 * (KeyValue.FIXED_OVERHEAD + ClassSize.CELL_ARRAY_MAP_ENTRY);
565     assertEquals(totalCellsLen, regionServicesForStores.getMemstoreSize());
566     assertEquals(totalHeapOverhead, ((CompactingMemStore)memstore).heapOverhead());
567
568     size = memstore.getFlushableSize();
569     MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
570     region.decrMemstoreSize(size);  // simulate flusher
571     ImmutableSegment s = memstore.getSnapshot();
572     assertEquals(3, s.getCellsCount());
573     assertEquals(0, regionServicesForStores.getMemstoreSize());
574
575     memstore.clearSnapshot(snapshot.getId());
576   }

712   private int addRowsByKeys(final AbstractMemStore hmc, String[] keys) {
713     byte[] fam = Bytes.toBytes("testfamily");
714     byte[] qf = Bytes.toBytes("testqualifier");
715     long size = hmc.getActive().keySize();
716     long heapOverhead = hmc.getActive().heapOverhead();
717     int totalLen = 0;
718     for (int i = 0; i < keys.length; i++) {
719       long timestamp = System.currentTimeMillis();
720       Threads.sleep(1); // to make sure each kv gets a different ts
721       byte[] row = Bytes.toBytes(keys[i]);
722       byte[] val = Bytes.toBytes(keys[i] + i);
723       KeyValue kv = new KeyValue(row, fam, qf, timestamp, val);
724       totalLen += kv.getLength();
725       hmc.add(kv, null);
726       LOG.debug("added kv: " + kv.getKeyString() + ", timestamp:" + kv.getTimestamp());
727     }
728     regionServicesForStores.addMemstoreSize(new MemstoreSize(hmc.getActive().keySize() - size,
729         hmc.getActive().heapOverhead() - heapOverhead));
730     return totalLen;
731   }
