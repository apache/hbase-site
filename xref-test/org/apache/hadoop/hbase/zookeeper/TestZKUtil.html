43    @Test
44    public void testGetZooKeeperClusterKey() {
45      Configuration conf = HBaseConfiguration.create();
46      conf.set(HConstants.ZOOKEEPER_QUORUM, "\tlocalhost\n");
47      conf.set(HConstants.ZOOKEEPER_CLIENT_PORT, "3333");
48      conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "hbase");
49      String clusterKey = ZKUtil.getZooKeeperClusterKey(conf, "test");
50      Assert.assertTrue(!clusterKey.contains("\t") && !clusterKey.contains("\n"));
51      Assert.assertEquals("localhost:3333:hbase,test", clusterKey);
52    }
53    
54    @Test
55    public void testCreateACL() throws ZooKeeperConnectionException, IOException {
56      Configuration conf = HBaseConfiguration.create();
57      conf.set(Superusers.SUPERUSER_CONF_KEY, "user1,@group1,user2,@group2,user3");
58      String node = "/hbase/testCreateACL";
59      ZooKeeperWatcher watcher = new ZooKeeperWatcher(conf, node, null, false);
60      List<ACL> aclList = ZKUtil.createACL(watcher, node, true);
61      Assert.assertEquals(aclList.size(), 4); // 3+1, since ACL will be set for the creator by default
62      Assert.assertTrue(!aclList.contains(new ACL(Perms.ALL, new Id("auth", "@group1")))
63          && !aclList.contains(new ACL(Perms.ALL, new Id("auth", "@group2"))));
64      Assert.assertTrue(aclList.contains(new ACL(Perms.ALL, new Id("auth", "user1")))
65          && aclList.contains(new ACL(Perms.ALL, new Id("auth", "user2")))
66          && aclList.contains(new ACL(Perms.ALL, new Id("auth", "user3"))));
67    }
