103   private HFileBlockEncodingContext getEncodingContext(Compression.Algorithm algo,
104       DataBlockEncoding encoding) {
105     DataBlockEncoder encoder = encoding.getEncoder();
106     HFileContext meta = new HFileContextBuilder()
107                         .withHBaseCheckSum(false)
108                         .withIncludesMvcc(includesMemstoreTS)
109                         .withIncludesTags(includesTags)
110                         .withCompression(algo).build();
111     if (encoder != null) {
112       return encoder.newDataBlockEncodingContext(encoding,
113           HConstants.HFILEBLOCK_DUMMY_HEADER, meta);
114     } else {
115       return new HFileBlockDefaultEncodingContext(encoding,
116           HConstants.HFILEBLOCK_DUMMY_HEADER, meta);
117     }
118   }

186   /**
187    * Test seeking while file is encoded.
188    */
189   @Test
190   public void testSeekingOnSample() throws IOException {
191     List<KeyValue> sampleKv = generator.generateTestKeyValues(NUMBER_OF_KV, includesTags);
192 
193     // create all seekers
194     List<DataBlockEncoder.EncodedSeeker> encodedSeekers = 
195         new ArrayList<DataBlockEncoder.EncodedSeeker>();
196     for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
197       LOG.info("Encoding: " + encoding);
198       // Off heap block data support not added for PREFIX_TREE DBE yet.
199       // TODO remove this once support is added. HBASE-12298
200       if (this.useOffheapData && encoding == DataBlockEncoding.PREFIX_TREE) continue;
201       DataBlockEncoder encoder = encoding.getEncoder();
202       if (encoder == null) {
203         continue;
204       }
205       LOG.info("Encoder: " + encoder);
206       ByteBuffer encodedBuffer = encodeKeyValues(encoding, sampleKv,
207           getEncodingContext(Compression.Algorithm.NONE, encoding), this.useOffheapData);
208       HFileContext meta = new HFileContextBuilder()
209                           .withHBaseCheckSum(false)
210                           .withIncludesMvcc(includesMemstoreTS)
211                           .withIncludesTags(includesTags)
212                           .withCompression(Compression.Algorithm.NONE)
213                           .build();
214       DataBlockEncoder.EncodedSeeker seeker = encoder.createSeeker(CellComparator.COMPARATOR,
215           encoder.newDataBlockDecodingContext(meta));
216       seeker.setCurrentBuffer(new SingleByteBuff(encodedBuffer));
217       encodedSeekers.add(seeker);
218     }
219     LOG.info("Testing it!");
220     // test it!
221     // try a few random seeks
222     for (boolean seekBefore : new boolean[] { false, true }) {
223       for (int i = 0; i < NUM_RANDOM_SEEKS; ++i) {
224         int keyValueId;
225         if (!seekBefore) {
226           keyValueId = randomizer.nextInt(sampleKv.size());
227         } else {
228           keyValueId = randomizer.nextInt(sampleKv.size() - 1) + 1;
229         }
230 
231         KeyValue keyValue = sampleKv.get(keyValueId);
232         checkSeekingConsistency(encodedSeekers, seekBefore, keyValue);
233       }
234     }
235 
236     // check edge cases
237     LOG.info("Checking edge cases");
238     checkSeekingConsistency(encodedSeekers, false, sampleKv.get(0));
239     for (boolean seekBefore : new boolean[] { false, true }) {
240       checkSeekingConsistency(encodedSeekers, seekBefore, sampleKv.get(sampleKv.size() - 1));
241       KeyValue midKv = sampleKv.get(sampleKv.size() / 2);
242       Cell lastMidKv =CellUtil.createLastOnRowCol(midKv);
243       checkSeekingConsistency(encodedSeekers, seekBefore, lastMidKv);
244     }
245     LOG.info("Done");
246   }

248   static ByteBuffer encodeKeyValues(DataBlockEncoding encoding, List<KeyValue> kvs,
249       HFileBlockEncodingContext encodingContext, boolean useOffheapData) throws IOException {
250     DataBlockEncoder encoder = encoding.getEncoder();
251     ByteArrayOutputStream baos = new ByteArrayOutputStream();
252     baos.write(HConstants.HFILEBLOCK_DUMMY_HEADER);
253     DataOutputStream dos = new DataOutputStream(baos);
254     encoder.startBlockEncoding(encodingContext, dos);
255     for (KeyValue kv : kvs) {
256       encoder.encode(kv, encodingContext, dos);
257     }
258     BufferGrabbingByteArrayOutputStream stream = new BufferGrabbingByteArrayOutputStream();
259     baos.writeTo(stream);
260     encoder.endBlockEncoding(encodingContext, dos, stream.getBuffer());
261     byte[] encodedData = new byte[baos.size() - ENCODED_DATA_OFFSET];
262     System.arraycopy(baos.toByteArray(), ENCODED_DATA_OFFSET, encodedData, 0, encodedData.length);
263     if (useOffheapData) {
264       ByteBuffer bb = ByteBuffer.allocateDirect(encodedData.length);
265       bb.put(encodedData);
266       bb.rewind();
267       return bb;
268     }
269     return ByteBuffer.wrap(encodedData);
270   }

272   @Test
273   public void testNextOnSample() throws IOException {
274     List<KeyValue> sampleKv = generator.generateTestKeyValues(NUMBER_OF_KV, includesTags);
275 
276     for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
277       // Off heap block data support not added for PREFIX_TREE DBE yet.
278       // TODO remove this once support is added. HBASE-12298
279       if (this.useOffheapData && encoding == DataBlockEncoding.PREFIX_TREE) continue;
280       if (encoding.getEncoder() == null) {
281         continue;
282       }
283       DataBlockEncoder encoder = encoding.getEncoder();
284       ByteBuffer encodedBuffer = encodeKeyValues(encoding, sampleKv,
285           getEncodingContext(Compression.Algorithm.NONE, encoding), this.useOffheapData);
286       HFileContext meta = new HFileContextBuilder()
287                           .withHBaseCheckSum(false)
288                           .withIncludesMvcc(includesMemstoreTS)
289                           .withIncludesTags(includesTags)
290                           .withCompression(Compression.Algorithm.NONE)
291                           .build();
292       DataBlockEncoder.EncodedSeeker seeker = encoder.createSeeker(CellComparator.COMPARATOR,
293           encoder.newDataBlockDecodingContext(meta));
294       seeker.setCurrentBuffer(new SingleByteBuff(encodedBuffer));
295       int i = 0;
296       do {
297         KeyValue expectedKeyValue = sampleKv.get(i);
298         Cell cell = seeker.getCell();
299         if (CellComparator.COMPARATOR.compareKeyIgnoresMvcc(expectedKeyValue, cell) != 0) {
300           int commonPrefix = CellUtil
301               .findCommonPrefixInFlatKey(expectedKeyValue, cell, false, true);
302           fail(String.format("next() produces wrong results "
303               + "encoder: %s i: %d commonPrefix: %d" + "\n expected %s\n actual      %s", encoder
304               .toString(), i, commonPrefix, Bytes.toStringBinary(expectedKeyValue.getBuffer(),
305               expectedKeyValue.getKeyOffset(), expectedKeyValue.getKeyLength()), CellUtil.toString(
306               cell, false)));
307         }
308         i++;
309       } while (seeker.next());
310     }
311   }

340   private void checkSeekingConsistency(List<DataBlockEncoder.EncodedSeeker> encodedSeekers,
341       boolean seekBefore, Cell keyValue) {
342     Cell expectedKeyValue = null;
343     ByteBuffer expectedKey = null;
344     ByteBuffer expectedValue = null;
345     for (DataBlockEncoder.EncodedSeeker seeker : encodedSeekers) {
346       seeker.seekToKeyInBlock(keyValue, seekBefore);
347       seeker.rewind();
348 
349       Cell actualKeyValue = seeker.getCell();
350       ByteBuffer actualKey = null;
351       if (seeker instanceof PrefixTreeSeeker) {
352         byte[] serializedKey = CellUtil.getCellKeySerializedAsKeyValueKey(seeker.getKey());
353         actualKey = ByteBuffer.wrap(KeyValueUtil.createKeyValueFromKey(serializedKey).getKey());
354       } else {
355         actualKey = ByteBuffer.wrap(((KeyValue) seeker.getKey()).getKey());
356       }
357       ByteBuffer actualValue = seeker.getValueShallowCopy();
358 
359       if (expectedKeyValue != null) {
360         assertTrue(CellUtil.equals(expectedKeyValue, actualKeyValue));
361       } else {
362         expectedKeyValue = actualKeyValue;
363       }
364 
365       if (expectedKey != null) {
366         assertEquals(expectedKey, actualKey);
367       } else {
368         expectedKey = actualKey;
369       }
370 
371       if (expectedValue != null) {
372         assertEquals(expectedValue, actualValue);
373       } else {
374         expectedValue = actualValue;
375       }
376     }
377   }

379   private void testEncodersOnDataset(List<KeyValue> kvList, boolean includesMemstoreTS,
380       boolean includesTags) throws IOException {
381     ByteBuffer unencodedDataBuf = RedundantKVGenerator.convertKvToByteBuffer(kvList,
382         includesMemstoreTS);
383     HFileContext fileContext = new HFileContextBuilder().withIncludesMvcc(includesMemstoreTS)
384         .withIncludesTags(includesTags).build();
385     for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
386       DataBlockEncoder encoder = encoding.getEncoder();
387       if (encoder == null) {
388         continue;
389       }
390       HFileBlockEncodingContext encodingContext = new HFileBlockDefaultEncodingContext(encoding,
391           HConstants.HFILEBLOCK_DUMMY_HEADER, fileContext);
392 
393       ByteArrayOutputStream baos = new ByteArrayOutputStream();
394       baos.write(HConstants.HFILEBLOCK_DUMMY_HEADER);
395       DataOutputStream dos = new DataOutputStream(baos);
396       encoder.startBlockEncoding(encodingContext, dos);
397       for (KeyValue kv : kvList) {
398         encoder.encode(kv, encodingContext, dos);
399       }
400       BufferGrabbingByteArrayOutputStream stream = new BufferGrabbingByteArrayOutputStream();
401       baos.writeTo(stream);
402       encoder.endBlockEncoding(encodingContext, dos, stream.getBuffer());
403       byte[] encodedData = baos.toByteArray();
404 
405       testAlgorithm(encodedData, unencodedDataBuf, encoder);
406     }
407   }

430   private void testAlgorithm(byte[] encodedData, ByteBuffer unencodedDataBuf,
431       DataBlockEncoder encoder) throws IOException {
432     // decode
433     ByteArrayInputStream bais = new ByteArrayInputStream(encodedData, ENCODED_DATA_OFFSET,
434         encodedData.length - ENCODED_DATA_OFFSET);
435     DataInputStream dis = new DataInputStream(bais);
436     ByteBuffer actualDataset;
437     HFileContext meta = new HFileContextBuilder().withHBaseCheckSum(false)
438         .withIncludesMvcc(includesMemstoreTS).withIncludesTags(includesTags)
439         .withCompression(Compression.Algorithm.NONE).build();
440     actualDataset = encoder.decodeKeyValues(dis, encoder.newDataBlockDecodingContext(meta));
441     actualDataset.rewind();
442 
443     // this is because in case of prefix tree the decoded stream will not have
444     // the
445     // mvcc in it.
446     assertEquals("Encoding -> decoding gives different results for " + encoder,
447         Bytes.toStringBinary(unencodedDataBuf), Bytes.toStringBinary(actualDataset));
448   }
