043  private static void checkDefaultsVersion(Configuration conf) {
044    if (conf.getBoolean("hbase.defaults.for.version.skip", Boolean.FALSE)) return;
045    String defaultsVersion = conf.get("hbase.defaults.for.version");
046    String thisVersion = VersionInfo.getVersion();
047    if (!thisVersion.equals(defaultsVersion)) {
048      throw new RuntimeException(
049        "hbase-default.xml file seems to be for an older version of HBase (" +
050        defaultsVersion + "), this version is " + thisVersion);
051    }
052  }

054  /**
055   * The hbase.ipc.server.reservoir.initial.max and hbase.ipc.server.reservoir.initial.buffer.size
056   * were introduced in HBase2.0.0, while in HBase3.0.0 the two config keys will be replaced by
057   * hbase.server.allocator.max.buffer.count and hbase.server.allocator.buffer.size.
058   * Also the hbase.ipc.server.reservoir.enabled will be replaced by
059   * hbase.server.allocator.pool.enabled. Keep the three old config keys here for HBase2.x
060   * compatibility.
061   * <br>
062   * HBASE-24667: This config hbase.regionserver.hostname.disable.master.reversedns will be
063   * replaced by hbase.unsafe.regionserver.hostname.disable.master.reversedns. Keep the old config
064   * keys here for backward compatibility.
065   */
066  private static void addDeprecatedKeys() {
067    Configuration.addDeprecations(new DeprecationDelta[]{
068      new DeprecationDelta("hbase.regionserver.hostname", "hbase.unsafe.regionserver.hostname"),
069      new DeprecationDelta("hbase.regionserver.hostname.disable.master.reversedns",
070        "hbase.unsafe.regionserver.hostname.disable.master.reversedns"),
071      new DeprecationDelta("hbase.offheapcache.minblocksize",
072        "hbase.blockcache.minblocksize"),
073      new DeprecationDelta("hbase.ipc.server.reservoir.enabled",
074        "hbase.server.allocator.pool.enabled"),
075      new DeprecationDelta("hbase.ipc.server.reservoir.initial.max",
076        "hbase.server.allocator.max.buffer.count"),
077      new DeprecationDelta("hbase.ipc.server.reservoir.initial.buffer.size",
078        "hbase.server.allocator.buffer.size"),
079      new DeprecationDelta("hlog.bulk.output", "wal.bulk.output"),
080      new DeprecationDelta("hlog.input.tables", "wal.input.tables"),
081      new DeprecationDelta("hlog.input.tablesmap", "wal.input.tablesmap"),
082      new DeprecationDelta("hbase.master.mob.ttl.cleaner.period",
083        "hbase.master.mob.cleaner.period"),
084      new DeprecationDelta("hbase.normalizer.min.region.count",
085        "hbase.normalizer.merge.min.region.count")
086    });
087  }

089  public static Configuration addHbaseResources(Configuration conf) {
090    conf.addResource("hbase-default.xml");
091    conf.addResource("hbase-site.xml");
092
093    checkDefaultsVersion(conf);
094    return conf;
095  }

097  /**
098   * Creates a Configuration with HBase resources
099   * @return a Configuration with HBase resources
100   */
101  public static Configuration create() {
102    Configuration conf = new Configuration();
103    // In case HBaseConfiguration is loaded from a different classloader than
104    // Configuration, conf needs to be set with appropriate class loader to resolve
105    // HBase resources.
106    conf.setClassLoader(HBaseConfiguration.class.getClassLoader());
107    return addHbaseResources(conf);
108  }

110  /**
111   * @param that Configuration to clone.
112   * @return a Configuration created with the hbase-*.xml files plus
113   * the given configuration.
114   */
115  public static Configuration create(final Configuration that) {
116    Configuration conf = create();
117    merge(conf, that);
118    return conf;
119  }

121  /**
122   * Merge two configurations.
123   * @param destConf the configuration that will be overwritten with items
124   *                 from the srcConf
125   * @param srcConf the source configuration
126   **/
127  public static void merge(Configuration destConf, Configuration srcConf) {
128    for (Map.Entry<String, String> e : srcConf) {
129      destConf.set(e.getKey(), e.getValue());
130    }
131  }
132
133  /**
134   * Returns a subset of the configuration properties, matching the given key prefix.
135   * The prefix is stripped from the return keys, ie. when calling with a prefix of "myprefix",
136   * the entry "myprefix.key1 = value1" would be returned as "key1 = value1".  If an entry's
137   * key matches the prefix exactly ("myprefix = value2"), it will <strong>not</strong> be
138   * included in the results, since it would show up as an entry with an empty key.
139   */
140  public static Configuration subset(Configuration srcConf, String prefix) {
141    Configuration newConf = new Configuration(false);
142    for (Map.Entry<String, String> entry : srcConf) {
143      if (entry.getKey().startsWith(prefix)) {
144        String newKey = entry.getKey().substring(prefix.length());
145        // avoid entries that would produce an empty key
146        if (!newKey.isEmpty()) {
147          newConf.set(newKey, entry.getValue());
148        }
149      }
150    }
151    return newConf;
152  }
153
154  /**
155   * Sets all the entries in the provided {@code Map<String, String>} as properties in the
156   * given {@code Configuration}.  Each property will have the specified prefix prepended,
157   * so that the configuration entries are keyed by {@code prefix + entry.getKey()}.
158   */
159  public static void setWithPrefix(Configuration conf, String prefix,
160                                   Iterable<Map.Entry<String, String>> properties) {
161    for (Map.Entry<String, String> entry : properties) {
162      conf.set(prefix + entry.getKey(), entry.getValue());
163    }
164  }

166  /**
167   * @return whether to show HBase Configuration in servlet
168   */
169  public static boolean isShowConfInServlet() {
170    boolean isShowConf = false;
171    try {
172      if (Class.forName("org.apache.hadoop.conf.ConfServlet") != null) {
173        isShowConf = true;
174      }
175    } catch (LinkageError e) {
176       // should we handle it more aggressively in addition to log the error?
177       LOG.warn("Error thrown: ", e);
178    } catch (ClassNotFoundException ce) {
179      LOG.debug("ClassNotFound: ConfServlet");
180      // ignore
181    }
182    return isShowConf;
183  }

185  /**
186   * Get the password from the Configuration instance using the
187   * getPassword method if it exists. If not, then fall back to the
188   * general get method for configuration elements.
189   *
190   * @param conf    configuration instance for accessing the passwords
191   * @param alias   the name of the password element
192   * @param defPass the default password
193   * @return String password or default password
194   * @throws IOException
195   */
196  public static String getPassword(Configuration conf, String alias,
197      String defPass) throws IOException {
198    String passwd = null;
199    try {
200      Method m = Configuration.class.getMethod("getPassword", String.class);
201      char[] p = (char[]) m.invoke(conf, alias);
202      if (p != null) {
203        LOG.debug(String.format("Config option \"%s\" was found through" +
204            " the Configuration getPassword method.", alias));
205        passwd = new String(p);
206      } else {
207        LOG.debug(String.format(
208            "Config option \"%s\" was not found. Using provided default value",
209            alias));
210        passwd = defPass;
211      }
212    } catch (NoSuchMethodException e) {
213      // this is a version of Hadoop where the credential
214      //provider API doesn't exist yet
215      LOG.debug(String.format(
216          "Credential.getPassword method is not available." +
217              " Falling back to configuration."));
218      passwd = conf.get(alias, defPass);
219    } catch (SecurityException e) {
220      throw new IOException(e.getMessage(), e);
221    } catch (IllegalAccessException e) {
222      throw new IOException(e.getMessage(), e);
223    } catch (IllegalArgumentException e) {
224      throw new IOException(e.getMessage(), e);
225    } catch (InvocationTargetException e) {
226      throw new IOException(e.getMessage(), e);
227    }
228    return passwd;
229  }

231  /**
232   * Generates a {@link Configuration} instance by applying the ZooKeeper cluster key
233   * to the base Configuration.  Note that additional configuration properties may be needed
234   * for a remote cluster, so it is preferable to use
235   * {@link #createClusterConf(Configuration, String, String)}.
236   *
237   * @param baseConf the base configuration to use, containing prefixed override properties
238   * @param clusterKey the ZooKeeper quorum cluster key to apply, or {@code null} if none
239   *
240   * @return the merged configuration with override properties and cluster key applied
241   *
242   * @see #createClusterConf(Configuration, String, String)
243   */
244  public static Configuration createClusterConf(Configuration baseConf, String clusterKey)
245      throws IOException {
246    return createClusterConf(baseConf, clusterKey, null);
247  }
248
249  /**
250   * Generates a {@link Configuration} instance by applying property overrides prefixed by
251   * a cluster profile key to the base Configuration.  Override properties are extracted by
252   * the {@link #subset(Configuration, String)} method, then the merged on top of the base
253   * Configuration and returned.
254   *
255   * @param baseConf the base configuration to use, containing prefixed override properties
256   * @param clusterKey the ZooKeeper quorum cluster key to apply, or {@code null} if none
257   * @param overridePrefix the property key prefix to match for override properties,
258   *     or {@code null} if none
259   * @return the merged configuration with override properties and cluster key applied
260   */
261  public static Configuration createClusterConf(Configuration baseConf, String clusterKey,
262      String overridePrefix) throws IOException {
263    Configuration clusterConf = HBaseConfiguration.create(baseConf);
264    if (clusterKey != null && !clusterKey.isEmpty()) {
265      applyClusterKeyToConf(clusterConf, clusterKey);
266    }
267
268    if (overridePrefix != null && !overridePrefix.isEmpty()) {
269      Configuration clusterSubset = HBaseConfiguration.subset(clusterConf, overridePrefix);
270      HBaseConfiguration.merge(clusterConf, clusterSubset);
271    }
272    return clusterConf;
273  }

275  /**
276   * Apply the settings in the given key to the given configuration, this is
277   * used to communicate with distant clusters
278   * @param conf configuration object to configure
279   * @param key string that contains the 3 required configuratins
280   */
281  private static void applyClusterKeyToConf(Configuration conf, String key)
282      throws IOException {
283    ZKConfig.ZKClusterKey zkClusterKey = ZKConfig.transformClusterKey(key);
284    conf.set(HConstants.ZOOKEEPER_QUORUM, zkClusterKey.getQuorumString());
285    conf.setInt(HConstants.ZOOKEEPER_CLIENT_PORT, zkClusterKey.getClientPort());
286    conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, zkClusterKey.getZnodeParent());
287    // Without the right registry, the above configs are useless. Also, we don't use setClass()
288    // here because the ConnectionRegistry* classes are not resolvable from this module.
289    // This will be broken if ZkConnectionRegistry class gets renamed or moved. Is there a better
290    // way?
291    LOG.info("Overriding client registry implementation to {}",
292        HConstants.ZK_CONNECTION_REGISTRY_CLASS);
293    conf.set(HConstants.CLIENT_CONNECTION_REGISTRY_IMPL_CONF_KEY,
294        HConstants.ZK_CONNECTION_REGISTRY_CLASS);
295  }
