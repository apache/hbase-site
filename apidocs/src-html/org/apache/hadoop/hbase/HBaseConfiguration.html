040  private static void checkDefaultsVersion(Configuration conf) {
041    if (conf.getBoolean("hbase.defaults.for.version.skip", Boolean.FALSE)) return;
042    String defaultsVersion = conf.get("hbase.defaults.for.version");
043    String thisVersion = VersionInfo.getVersion();
044    if (!thisVersion.equals(defaultsVersion)) {
045      throw new RuntimeException(
046        "hbase-default.xml file seems to be for an older version of HBase (" + defaultsVersion
047          + "), this version is " + thisVersion);
048    }
049  }

051  /**
052   * The hbase.ipc.server.reservoir.initial.max and hbase.ipc.server.reservoir.initial.buffer.size
053   * were introduced in HBase2.0.0, while in HBase3.0.0 the two config keys will be replaced by
054   * hbase.server.allocator.max.buffer.count and hbase.server.allocator.buffer.size. Also the
055   * hbase.ipc.server.reservoir.enabled will be replaced by hbase.server.allocator.pool.enabled.
056   * Keep the three old config keys here for HBase2.x compatibility. <br>
057   * HBASE-24667: This config hbase.regionserver.hostname.disable.master.reversedns will be replaced
058   * by hbase.unsafe.regionserver.hostname.disable.master.reversedns. Keep the old config keys here
059   * for backward compatibility. <br>
060   * Note: Before Hadoop-3.3, we must call the addDeprecations method before creating the
061   * Configuration object to work correctly. After this bug is fixed in hadoop-3.3, there will be no
062   * order problem.
063   * @see <a href="https://issues.apache.org/jira/browse/HADOOP-15708">HADOOP-15708</a>
064   */
065  private static void addDeprecatedKeys() {
066    Configuration.addDeprecations(new DeprecationDelta[] {
067      new DeprecationDelta("hbase.regionserver.hostname", "hbase.unsafe.regionserver.hostname"),
068      new DeprecationDelta("hbase.regionserver.hostname.disable.master.reversedns",
069        "hbase.unsafe.regionserver.hostname.disable.master.reversedns"),
070      new DeprecationDelta("hbase.offheapcache.minblocksize", "hbase.blockcache.minblocksize"),
071      new DeprecationDelta("hbase.ipc.server.reservoir.enabled",
072        "hbase.server.allocator.pool.enabled"),
073      new DeprecationDelta("hbase.ipc.server.reservoir.initial.max",
074        "hbase.server.allocator.max.buffer.count"),
075      new DeprecationDelta("hbase.ipc.server.reservoir.initial.buffer.size",
076        "hbase.server.allocator.buffer.size"),
077      new DeprecationDelta("hlog.bulk.output", "wal.bulk.output"),
078      new DeprecationDelta("hlog.input.tables", "wal.input.tables"),
079      new DeprecationDelta("hlog.input.tablesmap", "wal.input.tablesmap"),
080      new DeprecationDelta("hbase.master.mob.ttl.cleaner.period",
081        "hbase.master.mob.cleaner.period"),
082      new DeprecationDelta("hbase.normalizer.min.region.count",
083        "hbase.normalizer.merge.min.region.count") });
084  }

086  public static Configuration addHbaseResources(Configuration conf) {
087    conf.addResource("hbase-default.xml");
088    conf.addResource("hbase-site.xml");
089
090    checkDefaultsVersion(conf);
091    return conf;
092  }

094  /**
095   * Creates a Configuration with HBase resources
096   * @return a Configuration with HBase resources
097   */
098  public static Configuration create() {
099    Configuration conf = new Configuration();
100    // In case HBaseConfiguration is loaded from a different classloader than
101    // Configuration, conf needs to be set with appropriate class loader to resolve
102    // HBase resources.
103    conf.setClassLoader(HBaseConfiguration.class.getClassLoader());
104    return addHbaseResources(conf);
105  }
106
107  /**
108   * @param that Configuration to clone.
109   * @return a Configuration created with the hbase-*.xml files plus the given configuration.
110   */
111  public static Configuration create(final Configuration that) {
112    Configuration conf = create();
113    merge(conf, that);
114    return conf;
115  }
116
117  /**
118   * Merge two configurations.
119   * @param destConf the configuration that will be overwritten with items from the srcConf
120   * @param srcConf  the source configuration
121   **/
122  public static void merge(Configuration destConf, Configuration srcConf) {
123    for (Map.Entry<String, String> e : srcConf) {
124      destConf.set(e.getKey(), e.getValue());
125    }
126  }
127
128  /**
129   * Returns a subset of the configuration properties, matching the given key prefix. The prefix is
130   * stripped from the return keys, ie. when calling with a prefix of "myprefix", the entry
131   * "myprefix.key1 = value1" would be returned as "key1 = value1". If an entry's key matches the
132   * prefix exactly ("myprefix = value2"), it will <strong>not</strong> be included in the results,
133   * since it would show up as an entry with an empty key.
134   */
135  public static Configuration subset(Configuration srcConf, String prefix) {
136    Configuration newConf = new Configuration(false);
137    for (Map.Entry<String, String> entry : srcConf) {
138      if (entry.getKey().startsWith(prefix)) {
139        String newKey = entry.getKey().substring(prefix.length());
140        // avoid entries that would produce an empty key
141        if (!newKey.isEmpty()) {
142          newConf.set(newKey, entry.getValue());
143        }
144      }
145    }
146    return newConf;
147  }
148
149  /**
150   * Sets all the entries in the provided {@code Map<String, String>} as properties in the given
151   * {@code Configuration}. Each property will have the specified prefix prepended, so that the
152   * configuration entries are keyed by {@code prefix + entry.getKey()}.
153   */
154  public static void setWithPrefix(Configuration conf, String prefix,
155    Iterable<Map.Entry<String, String>> properties) {
156    for (Map.Entry<String, String> entry : properties) {
157      conf.set(prefix + entry.getKey(), entry.getValue());
158    }
159  }

161  /** Returns whether to show HBase Configuration in servlet */
162  public static boolean isShowConfInServlet() {
163    boolean isShowConf = false;
164    try {
165      if (Class.forName("org.apache.hadoop.conf.ConfServlet") != null) {
166        isShowConf = true;
167      }
168    } catch (LinkageError e) {
169      // should we handle it more aggressively in addition to log the error?
170      LOG.warn("Error thrown: ", e);
171    } catch (ClassNotFoundException ce) {
172      LOG.debug("ClassNotFound: ConfServlet");
173      // ignore
174    }
175    return isShowConf;
176  }

178  /**
179   * Get the password from the Configuration instance using the getPassword method if it exists. If
180   * not, then fall back to the general get method for configuration elements.
181   * @param conf    configuration instance for accessing the passwords
182   * @param alias   the name of the password element
183   * @param defPass the default password
184   * @return String password or default password n
185   */
186  public static String getPassword(Configuration conf, String alias, String defPass)
187    throws IOException {
188    String passwd;
189    char[] p = conf.getPassword(alias);
190    if (p != null) {
191      LOG.debug("Config option {} was found through the Configuration getPassword method.", alias);
192      passwd = new String(p);
193    } else {
194      LOG.debug("Config option {} was not found. Using provided default value", alias);
195      passwd = defPass;
196    }
197    return passwd;
198  }

200  /**
201   * Generates a {@link Configuration} instance by applying the ZooKeeper cluster key to the base
202   * Configuration. Note that additional configuration properties may be needed for a remote
203   * cluster, so it is preferable to use {@link #createClusterConf(Configuration, String, String)}.
204   * @param baseConf   the base configuration to use, containing prefixed override properties
205   * @param clusterKey the ZooKeeper quorum cluster key to apply, or {@code null} if none
206   * @return the merged configuration with override properties and cluster key applied
207   * @see #createClusterConf(Configuration, String, String)
208   */
209  public static Configuration createClusterConf(Configuration baseConf, String clusterKey)
210    throws IOException {
211    return createClusterConf(baseConf, clusterKey, null);
212  }
213
214  /**
215   * Generates a {@link Configuration} instance by applying property overrides prefixed by a cluster
216   * profile key to the base Configuration. Override properties are extracted by the
217   * {@link #subset(Configuration, String)} method, then the merged on top of the base Configuration
218   * and returned.
219   * @param baseConf       the base configuration to use, containing prefixed override properties
220   * @param clusterKey     the ZooKeeper quorum cluster key to apply, or {@code null} if none
221   * @param overridePrefix the property key prefix to match for override properties, or {@code null}
222   *                       if none
223   * @return the merged configuration with override properties and cluster key applied
224   */
225  public static Configuration createClusterConf(Configuration baseConf, String clusterKey,
226    String overridePrefix) throws IOException {
227    Configuration clusterConf = HBaseConfiguration.create(baseConf);
228    if (clusterKey != null && !clusterKey.isEmpty()) {
229      applyClusterKeyToConf(clusterConf, clusterKey);
230    }
231
232    if (overridePrefix != null && !overridePrefix.isEmpty()) {
233      Configuration clusterSubset = HBaseConfiguration.subset(clusterConf, overridePrefix);
234      HBaseConfiguration.merge(clusterConf, clusterSubset);
235    }
236    return clusterConf;
237  }

239  /**
240   * Apply the settings in the given key to the given configuration, this is used to communicate
241   * with distant clusters
242   * @param conf configuration object to configure
243   * @param key  string that contains the 3 required configuratins
244   */
245  private static void applyClusterKeyToConf(Configuration conf, String key) throws IOException {
246    ZKConfig.ZKClusterKey zkClusterKey = ZKConfig.transformClusterKey(key);
247    conf.set(HConstants.ZOOKEEPER_QUORUM, zkClusterKey.getQuorumString());
248    conf.setInt(HConstants.ZOOKEEPER_CLIENT_PORT, zkClusterKey.getClientPort());
249    conf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, zkClusterKey.getZnodeParent());
250    // Without the right registry, the above configs are useless. Also, we don't use setClass()
251    // here because the ConnectionRegistry* classes are not resolvable from this module.
252    // This will be broken if ZkConnectionRegistry class gets renamed or moved. Is there a better
253    // way?
254    LOG.info("Overriding client registry implementation to {}",
255      HConstants.ZK_CONNECTION_REGISTRY_CLASS);
256    conf.set(HConstants.CLIENT_CONNECTION_REGISTRY_IMPL_CONF_KEY,
257      HConstants.ZK_CONNECTION_REGISTRY_CLASS);
258  }
