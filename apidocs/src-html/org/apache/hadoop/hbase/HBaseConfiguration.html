033/**
034 * Adds HBase configuration files to a Configuration
035 */
036@InterfaceAudience.Public
037@InterfaceStability.Stable
038public class HBaseConfiguration extends Configuration {
039  private static final Log LOG = LogFactory.getLog(HBaseConfiguration.class);
040
041  /**
042   * Instantiating HBaseConfiguration() is deprecated. Please use
043   * HBaseConfiguration#create() to construct a plain Configuration
044   * @deprecated Please use create() instead.
045   */
046  @Deprecated
047  public HBaseConfiguration() {
048    //TODO:replace with private constructor, HBaseConfiguration should not extend Configuration
049    super();
050    addHbaseResources(this);
051    LOG.warn("instantiating HBaseConfiguration() is deprecated. Please use"
052        + " HBaseConfiguration#create() to construct a plain Configuration");
053  }

057   * HBaseConfiguration#create(conf) to construct a plain Configuration
058   * @deprecated Please user create(conf) instead.

060  @Deprecated

062    //TODO:replace with private constructor
063    this();
064    merge(this, c);
065  }
066
067  private static void checkDefaultsVersion(Configuration conf) {
068    if (conf.getBoolean("hbase.defaults.for.version.skip", Boolean.FALSE)) return;
069    String defaultsVersion = conf.get("hbase.defaults.for.version");
070    String thisVersion = VersionInfo.getVersion();
071    if (!thisVersion.equals(defaultsVersion)) {
072      throw new RuntimeException(
073        "hbase-default.xml file seems to be for an older version of HBase (" +
074        defaultsVersion + "), this version is " + thisVersion);
075    }
076  }
077
078  public static Configuration addHbaseResources(Configuration conf) {
079    conf.addResource("hbase-default.xml");
080    conf.addResource("hbase-site.xml");
081
082    checkDefaultsVersion(conf);
083    HeapMemorySizeUtil.checkForClusterFreeMemoryLimit(conf);
084    return conf;
085  }
086
087  /**
088   * Creates a Configuration with HBase resources
089   * @return a Configuration with HBase resources
090   */
091  public static Configuration create() {
092    Configuration conf = new Configuration();
093    // In case HBaseConfiguration is loaded from a different classloader than
094    // Configuration, conf needs to be set with appropriate class loader to resolve
095    // HBase resources.
096    conf.setClassLoader(HBaseConfiguration.class.getClassLoader());
097    return addHbaseResources(conf);
098  }
099
100  /**
101   * @param that Configuration to clone.
102   * @return a Configuration created with the hbase-*.xml files plus
103   * the given configuration.
104   */
105  public static Configuration create(final Configuration that) {
106    Configuration conf = create();
107    merge(conf, that);
108    return conf;
109  }
110
111  /**
112   * Merge two configurations.
113   * @param destConf the configuration that will be overwritten with items
114   *                 from the srcConf
115   * @param srcConf the source configuration
116   **/
117  public static void merge(Configuration destConf, Configuration srcConf) {
118    for (Entry<String, String> e : srcConf) {
119      destConf.set(e.getKey(), e.getValue());
120    }
121  }
122
123  /**
124   * Returns a subset of the configuration properties, matching the given key prefix.
125   * The prefix is stripped from the return keys, ie. when calling with a prefix of "myprefix",
126   * the entry "myprefix.key1 = value1" would be returned as "key1 = value1".  If an entry's
127   * key matches the prefix exactly ("myprefix = value2"), it will <strong>not</strong> be
128   * included in the results, since it would show up as an entry with an empty key.
129   */
130  public static Configuration subset(Configuration srcConf, String prefix) {
131    Configuration newConf = new Configuration(false);
132    for (Entry<String, String> entry : srcConf) {
133      if (entry.getKey().startsWith(prefix)) {
134        String newKey = entry.getKey().substring(prefix.length());
135        // avoid entries that would produce an empty key
136        if (!newKey.isEmpty()) {
137          newConf.set(newKey, entry.getValue());
138        }
139      }
140    }
141    return newConf;
142  }
143
144  /**
145   * @return whether to show HBase Configuration in servlet
146   */
147  public static boolean isShowConfInServlet() {
148    boolean isShowConf = false;
149    try {
150      if (Class.forName("org.apache.hadoop.conf.ConfServlet") != null) {
151        isShowConf = true;
152      }
153    } catch (LinkageError e) {
154       // should we handle it more aggressively in addition to log the error?
155       LOG.warn("Error thrown: ", e);
156    } catch (ClassNotFoundException ce) {
157      LOG.debug("ClassNotFound: ConfServlet");
158      // ignore
159    }
160    return isShowConf;
161  }
162
163  /**
164   * Get the value of the <code>name</code> property as an <code>int</code>, possibly
165   * referring to the deprecated name of the configuration property.
166   * If no such property exists, the provided default value is returned,
167   * or if the specified value is not a valid <code>int</code>,
168   * then an error is thrown.
169   *
170   * @param name property name.
171   * @param deprecatedName a deprecatedName for the property to use
172   * if non-deprecated name is not used
173   * @param defaultValue default value.
174   * @throws NumberFormatException when the value is invalid
175   * @return property value as an <code>int</code>,
176   *         or <code>defaultValue</code>.
177   */
178  // TODO: developer note: This duplicates the functionality of deprecated
179  // property support in Configuration in Hadoop 2. But since Hadoop-1 does not
180  // contain these changes, we will do our own as usual. Replace these when H2 is default.
181  public static int getInt(Configuration conf, String name,
182      String deprecatedName, int defaultValue) {
183    if (conf.get(deprecatedName) != null) {
184      LOG.warn(String.format("Config option \"%s\" is deprecated. Instead, use \"%s\""
185        , deprecatedName, name));
186      return conf.getInt(deprecatedName, defaultValue);
187    } else {
188      return conf.getInt(name, defaultValue);
189    }
190  }
191
192  /**
193   * Get the password from the Configuration instance using the
194   * getPassword method if it exists. If not, then fall back to the
195   * general get method for configuration elements.
196   *
197   * @param conf    configuration instance for accessing the passwords
198   * @param alias   the name of the password element
199   * @param defPass the default password
200   * @return String password or default password
201   * @throws IOException
202   */
203  public static String getPassword(Configuration conf, String alias,
204      String defPass) throws IOException {
205    String passwd = null;
206    try {
207      Method m = Configuration.class.getMethod("getPassword", String.class);
208      char[] p = (char[]) m.invoke(conf, alias);
209      if (p != null) {
210        LOG.debug(String.format("Config option \"%s\" was found through" +
211            " the Configuration getPassword method.", alias));
212        passwd = new String(p);
213      } else {
214        LOG.debug(String.format(
215            "Config option \"%s\" was not found. Using provided default value",
216            alias));
217        passwd = defPass;
218      }
219    } catch (NoSuchMethodException e) {
220      // this is a version of Hadoop where the credential
221      //provider API doesn't exist yet
222      LOG.debug(String.format(
223          "Credential.getPassword method is not available." +
224              " Falling back to configuration."));
225      passwd = conf.get(alias, defPass);
226    } catch (SecurityException e) {
227      throw new IOException(e.getMessage(), e);
228    } catch (IllegalAccessException e) {
229      throw new IOException(e.getMessage(), e);
230    } catch (IllegalArgumentException e) {
231      throw new IOException(e.getMessage(), e);
232    } catch (InvocationTargetException e) {
233      throw new IOException(e.getMessage(), e);
234    }
235    return passwd;
236  }

239   * For debugging.  Dump configurations to system output as xml format.
240   * Master and RS configurations can also be dumped using
241   * http services. e.g. "curl http://master:16010/dump"
242   */
243  public static void main(String[] args) throws Exception {
244    HBaseConfiguration.create().writeXml(System.out);
245  }
