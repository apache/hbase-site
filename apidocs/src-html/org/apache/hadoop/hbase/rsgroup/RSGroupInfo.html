019package org.apache.hadoop.hbase.rsgroup;
020
021import java.util.Collection;
022import java.util.Set;
023import java.util.SortedSet;
024import java.util.TreeSet;
025
026import org.apache.hadoop.hbase.TableName;
027import org.apache.hadoop.hbase.net.Address;
028import org.apache.yetus.audience.InterfaceAudience;
029
030/**
031 * Stores the group information of region server groups.
032 */
033@InterfaceAudience.Public
034public class RSGroupInfo {
035  public static final String DEFAULT_GROUP = "default";
036  public static final String NAMESPACE_DESC_PROP_GROUP = "hbase.rsgroup.name";
037
038  private final String name;
039  // Keep servers in a sorted set so has an expected ordering when displayed.
040  private final SortedSet<Address> servers;
041  // Keep tables sorted too.
042  private final SortedSet<TableName> tables;
043
044  public RSGroupInfo(String name) {
045    this(name, new TreeSet<Address>(), new TreeSet<TableName>());
046  }
047
048  RSGroupInfo(String name, SortedSet<Address> servers, SortedSet<TableName> tables) {
049    this.name = name;
050    this.servers = servers == null? new TreeSet<>(): servers;
051    this.servers.addAll(servers);
052    this.tables = new TreeSet<>(tables);
053  }
054
055  public RSGroupInfo(RSGroupInfo src) {
056    this(src.getName(), src.servers, src.tables);
057  }
058
059  /**
060   * Get group name.
061   */
062  public String getName() {
063    return name;
064  }
065
066  /**
067   * Adds the given server to the group.
068   */
069  public void addServer(Address hostPort){
070    servers.add(hostPort);
071  }
072
073  /**
074   * Adds the given servers to the group.
075   */
076  public void addAllServers(Collection<Address> hostPort){
077    servers.addAll(hostPort);
078  }
079
080  /**
081   * @param hostPort hostPort of the server
082   * @return true, if a server with hostPort is found
083   */
084  public boolean containsServer(Address hostPort) {
085    return servers.contains(hostPort);
086  }
087
088  /**
089   * Get list of servers.
090   */
091  public Set<Address> getServers() {
092    return servers;
093  }
094
095  /**
096   * Remove given server from the group.
097   */
098  public boolean removeServer(Address hostPort) {
099    return servers.remove(hostPort);
100  }
101
102  /**
103   * Get set of tables that are members of the group.
104   */
105  public SortedSet<TableName> getTables() {
106    return tables;
107  }
108
109  public void addTable(TableName table) {
110    tables.add(table);
111  }
112
113  public void addAllTables(Collection<TableName> arg) {
114    tables.addAll(arg);
115  }
116
117  public boolean containsTable(TableName table) {
118    return tables.contains(table);
119  }
120
121  public boolean removeTable(TableName table) {
122    return tables.remove(table);
123  }
124
125  @Override
126  public String toString() {
127    StringBuilder sb = new StringBuilder();
128    sb.append("Name:");
129    sb.append(this.name);
130    sb.append(", ");
131    sb.append(" Servers:");
132    sb.append(this.servers);
133    sb.append(", ");
134    sb.append(" Tables:");
135    sb.append(this.tables);
136    return sb.toString();
137
138  }
139
140  @Override
141  public boolean equals(Object o) {
142    if (this == o) {
143      return true;
144    }
145    if (o == null || getClass() != o.getClass()) {
146      return false;
147    }
148
149    RSGroupInfo RSGroupInfo = (RSGroupInfo) o;
150
151    if (!name.equals(RSGroupInfo.name)) {
152      return false;
153    }
154    if (!servers.equals(RSGroupInfo.servers)) {
155      return false;
156    }
157    if (!tables.equals(RSGroupInfo.tables)) {
158      return false;
159    }
160
161    return true;
162  }
163
164  @Override
165  public int hashCode() {
166    int result = servers.hashCode();
167    result = 31 * result + tables.hashCode();
168    result = 31 * result + name.hashCode();
169    return result;
170  }
171}
