114      FileInfo(final boolean inArchive, final long size, final boolean corrupted) {
115        this.corrupted = corrupted;
116        this.inArchive = inArchive;

126      public boolean isCorrupted() {

141        if (isCorrupted()) return "CORRUPTED";
142        if (isMissing()) return "NOT FOUND";

144        return null;
145      }

273    /** Check if for a give file in archive, if there are other snapshots/tables still
274     * reference it.
275     * @param filePath file path in archive
276     * @param snapshotFilesMap a map for store files in snapshots about how many snapshots refer
277     *                         to it.
278     * @return true or false
279     */
280    private boolean isArchivedFileStillReferenced(final Path filePath,
281        final Map<Path, Integer> snapshotFilesMap) {
282
283      Integer c = snapshotFilesMap.get(filePath);
284
285      // Check if there are other snapshots or table from clone_snapshot() (via back-reference)
286      // still reference to it.
287      if ((c != null) && (c == 1)) {
288        Path parentDir = filePath.getParent();
289        Path backRefDir = HFileLink.getBackReferencesDir(parentDir, filePath.getName());
290        try {
291          if (CommonFSUtils.listStatus(fs, backRefDir) == null) {
292            return false;
293          }
294        } catch (IOException e) {
295          // For the purpose of this function, IOException is ignored and treated as
296          // the file is still being referenced.
297        }
298      }
299      return true;
300    }

302    /**
303     * Add the specified store file to the stats
304     * @param region region encoded Name
305     * @param family family name
306     * @param storeFile store file name
307     * @param filesMap store files map for all snapshots, it may be null
308     * @return the store file information
309     */
310    FileInfo addStoreFile(final RegionInfo region, final String family,
311        final SnapshotRegionManifest.StoreFile storeFile,
312        final Map<Path, Integer> filesMap) throws IOException {
313      HFileLink link = HFileLink.build(conf, snapshotTable, region.getEncodedName(),
314              family, storeFile.getName());
315      boolean isCorrupted = false;
316      boolean inArchive = false;
317      long size = -1;
318      try {
319        if (fs.exists(link.getArchivePath())) {
320          inArchive = true;
321          size = fs.getFileStatus(link.getArchivePath()).getLen();
322          hfilesArchiveSize.addAndGet(size);
323          hfilesArchiveCount.incrementAndGet();
324
325          // If store file is not shared with other snapshots and tables,
326          // increase nonSharedHfilesArchiveSize
327          if ((filesMap != null) &&
328              !isArchivedFileStillReferenced(link.getArchivePath(), filesMap)) {
329            nonSharedHfilesArchiveSize.addAndGet(size);
330          }
331        } else if (fs.exists(link.getMobPath())) {
332          inArchive = true;
333          size = fs.getFileStatus(link.getMobPath()).getLen();
334          hfilesMobSize.addAndGet(size);
335          hfilesMobCount.incrementAndGet();
336        } else {
337          size = link.getFileStatus(fs).getLen();
338          hfilesSize.addAndGet(size);
339          hfilesCount.incrementAndGet();
340        }
341        isCorrupted = (storeFile.hasFileSize() && storeFile.getFileSize() != size);
342        if (isCorrupted) hfilesCorrupted.incrementAndGet();
343      } catch (FileNotFoundException e) {
344        hfilesMissing.incrementAndGet();
345      }
346      return new FileInfo(inArchive, size, isCorrupted);
347    }
348
349    /**
350     * Add the specified log file to the stats
351     * @param server server name
352     * @param logfile log file name
353     * @return the log information
354     */
355    FileInfo addLogFile(final String server, final String logfile) throws IOException {
356      WALLink logLink = new WALLink(conf, server, logfile);
357      long size = -1;
358      try {
359        size = logLink.getFileStatus(fs).getLen();
360        logSize.addAndGet(size);
361        logsCount.incrementAndGet();
362      } catch (FileNotFoundException e) {
363        logsMissing.incrementAndGet();
364      }
365      return new FileInfo(false, size, false);
366    }

382  @Override
383  public int doWork() throws IOException, InterruptedException {
384    if (remoteDir != null) {
385      URI defaultFs = remoteDir.getFileSystem(conf).getUri();
386      CommonFSUtils.setFsDefault(conf, new Path(defaultFs));
387      CommonFSUtils.setRootDir(conf, remoteDir);
388    }
389
390    // List Available Snapshots
391    if (listSnapshots) {
392      SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
393      System.out.printf("%-20s | %-20s | %-20s | %s%n", "SNAPSHOT", "CREATION TIME", "TTL IN SEC",
394              "TABLE NAME");
395      for (SnapshotDescription desc: getSnapshotList(conf)) {
396        System.out.printf("%-20s | %20s | %20s | %s%n", desc.getName(),
397                df.format(new Date(desc.getCreationTime())), desc.getTtl(),
398                desc.getTableNameAsString());
399      }
400      return 0;
401    }
402
403    rootDir = CommonFSUtils.getRootDir(conf);
404    fs = FileSystem.get(rootDir.toUri(), conf);
405    LOG.debug("fs=" + fs.getUri().toString() + " root=" + rootDir);
406
407    // Load snapshot information
408    if (!loadSnapshotInfo(snapshotName)) {
409      System.err.println("Snapshot '" + snapshotName + "' not found!");
410      return 1;
411    }
412
413    printInfo();
414    if (showSchema) {
415      printSchema();
416    }
417    printFiles(showFiles, showStats);
418
419    return 0;
420  }

422  /**
423   * Load snapshot info and table descriptor for the specified snapshot
424   * @param snapshotName name of the snapshot to load
425   * @return false if snapshot is not found
426   */
427  private boolean loadSnapshotInfo(final String snapshotName) throws IOException {
428    Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
429    if (!fs.exists(snapshotDir)) {
430      LOG.warn("Snapshot '" + snapshotName + "' not found in: " + snapshotDir);
431      return false;
432    }
433
434    SnapshotProtos.SnapshotDescription snapshotDesc =
435        SnapshotDescriptionUtils.readSnapshotInfo(fs, snapshotDir);
436    snapshotManifest = SnapshotManifest.open(getConf(), fs, snapshotDir, snapshotDesc);
437    return true;
438  }

468  /**
469   * Collect the hfiles and logs statistics of the snapshot and
470   * dump the file list if requested and the collected information.
471   */
472  private void printFiles(final boolean showFiles, final boolean showStats) throws IOException {
473    if (showFiles) {
474      System.out.println("Snapshot Files");
475      System.out.println("----------------------------------------");
476    }
477
478    // Collect information about hfiles and logs in the snapshot
479    final SnapshotProtos.SnapshotDescription snapshotDesc = snapshotManifest.getSnapshotDescription();
480    final String table = snapshotDesc.getTable();
481    final SnapshotDescription desc = ProtobufUtil.createSnapshotDesc(snapshotDesc);
482    final SnapshotStats stats = new SnapshotStats(this.getConf(), this.fs, desc);
483    SnapshotReferenceUtil.concurrentVisitReferencedFiles(getConf(), fs, snapshotManifest,
484        "SnapshotInfo",
485      new SnapshotReferenceUtil.SnapshotVisitor() {
486        @Override
487        public void storeFile(final RegionInfo regionInfo, final String family,
488            final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
489          if (storeFile.hasReference()) return;
490
491          SnapshotStats.FileInfo info = stats.addStoreFile(regionInfo, family, storeFile, null);
492          if (showFiles) {
493            String state = info.getStateToString();
494            System.out.printf("%8s %s/%s/%s/%s %s%n",
495              (info.isMissing() ? "-" : fileSizeToString(info.getSize())),
496              table, regionInfo.getEncodedName(), family, storeFile.getName(),
497              state == null ? "" : "(" + state + ")");
498          }
499        }
500    });
501
502    // Dump the stats
503    System.out.println();
504    if (stats.isSnapshotCorrupted()) {
505      System.out.println("**************************************************************");
506      System.out.printf("BAD SNAPSHOT: %d hfile(s) and %d log(s) missing.%n",
507        stats.getMissingStoreFilesCount(), stats.getMissingLogsCount());
508      System.out.printf("              %d hfile(s) corrupted.%n",
509        stats.getCorruptedStoreFilesCount());
510      System.out.println("**************************************************************");
511    }
512
513    if (showStats) {
514      System.out.printf("%d HFiles (%d in archive, %d in mob storage), total size %s " +
515              "(%.2f%% %s shared with the source table, %.2f%% %s in mob dir)%n",
516        stats.getStoreFilesCount(), stats.getArchivedStoreFilesCount(),
517        stats.getMobStoreFilesCount(),
518        fileSizeToString(stats.getStoreFilesSize()),
519        stats.getSharedStoreFilePercentage(),
520        fileSizeToString(stats.getSharedStoreFilesSize()),
521        stats.getMobStoreFilePercentage(),
522        fileSizeToString(stats.getMobStoreFilesSize())
523      );
524      System.out.printf("%d Logs, total size %s%n",
525        stats.getLogsCount(), fileSizeToString(stats.getLogsSize()));
526      System.out.println();
527    }
528  }

546  @Override
547  protected CommandLineParser newParser() {
548    // Commons-CLI lacks the capability to handle combinations of options, so we do it ourselves
549    // Validate in parse() to get helpful error messages instead of exploding in processOptions()
550    return new DefaultParser() {
551      @Override
552      public CommandLine parse(org.apache.hbase.thirdparty.org.apache.commons.cli.Options opts, String[] args, Properties props, boolean stop)
553        throws ParseException {
554        CommandLine cl = super.parse(opts, args, props, stop);
555        if(!cmd.hasOption(Options.LIST_SNAPSHOTS) && !cmd.hasOption(Options.SNAPSHOT)) {
556          throw new ParseException("Missing required snapshot option!");
557        }
558        return cl;
559      }
560    };
561  }

597  /**
598   * Returns the snapshot stats
599   * @param conf the {@link Configuration} to use
600   * @param snapshotDesc  HBaseProtos.SnapshotDescription to get stats from
601   * @param filesMap {@link Map} store files map for all snapshots, it may be null
602   * @return the snapshot stats
603   */
604  public static SnapshotStats getSnapshotStats(final Configuration conf,
605      final SnapshotProtos.SnapshotDescription snapshotDesc,
606      final Map<Path, Integer> filesMap) throws IOException {
607    Path rootDir = CommonFSUtils.getRootDir(conf);
608    FileSystem fs = FileSystem.get(rootDir.toUri(), conf);
609    Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotDesc, rootDir);
610    SnapshotManifest manifest = SnapshotManifest.open(conf, fs, snapshotDir, snapshotDesc);
611    final SnapshotStats stats = new SnapshotStats(conf, fs, snapshotDesc);
612    SnapshotReferenceUtil.concurrentVisitReferencedFiles(conf, fs, manifest,
613        "SnapshotsStatsAggregation", new SnapshotReferenceUtil.SnapshotVisitor() {
614          @Override
615          public void storeFile(final RegionInfo regionInfo, final String family,
616              final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
617            if (!storeFile.hasReference()) {
618              stats.addStoreFile(regionInfo, family, storeFile, filesMap);
619            }
620          }});
621    return stats;
622  }

624  /**
625   * Returns the list of available snapshots in the specified location
626   * @param conf the {@link Configuration} to use
627   * @return the list of snapshots
628   */
629  public static List<SnapshotDescription> getSnapshotList(final Configuration conf)
630      throws IOException {
631    Path rootDir = CommonFSUtils.getRootDir(conf);
632    FileSystem fs = FileSystem.get(rootDir.toUri(), conf);
633    Path snapshotDir = SnapshotDescriptionUtils.getSnapshotsDir(rootDir);
634    FileStatus[] snapshots = fs.listStatus(snapshotDir,
635        new SnapshotDescriptionUtils.CompletedSnaphotDirectoriesFilter(fs));
636    List<SnapshotDescription> snapshotLists = new ArrayList<>(snapshots.length);
637    for (FileStatus snapshotDirStat: snapshots) {
638      SnapshotProtos.SnapshotDescription snapshotDesc =
639          SnapshotDescriptionUtils.readSnapshotInfo(fs, snapshotDirStat.getPath());
640      snapshotLists.add(ProtobufUtil.createSnapshotDesc(snapshotDesc));
641    }
642    return snapshotLists;
643  }

645  /**
646   * Gets the store files map for snapshot
647   * @param conf the {@link Configuration} to use
648   * @param snapshot {@link SnapshotDescription} to get stats from
649   * @param exec the {@link ExecutorService} to use
650   * @param filesMap {@link Map} the map to put the mapping entries
651   * @param uniqueHFilesArchiveSize {@link AtomicLong} the accumulated store file size in archive
652   * @param uniqueHFilesSize {@link AtomicLong} the accumulated store file size shared
653   * @param uniqueHFilesMobSize {@link AtomicLong} the accumulated mob store file size shared
654   */
655  private static void getSnapshotFilesMap(final Configuration conf,
656      final SnapshotDescription snapshot, final ExecutorService exec,
657      final ConcurrentHashMap<Path, Integer> filesMap,
658      final AtomicLong uniqueHFilesArchiveSize, final AtomicLong uniqueHFilesSize,
659      final AtomicLong uniqueHFilesMobSize) throws IOException {
660    SnapshotProtos.SnapshotDescription snapshotDesc =
661        ProtobufUtil.createHBaseProtosSnapshotDesc(snapshot);
662    Path rootDir = CommonFSUtils.getRootDir(conf);
663    final FileSystem fs = FileSystem.get(rootDir.toUri(), conf);
664
665    Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotDesc, rootDir);
666    SnapshotManifest manifest = SnapshotManifest.open(conf, fs, snapshotDir, snapshotDesc);
667    SnapshotReferenceUtil.concurrentVisitReferencedFiles(conf, fs, manifest, exec,
668        new SnapshotReferenceUtil.SnapshotVisitor() {
669          @Override public void storeFile(final RegionInfo regionInfo, final String family,
670              final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
671            if (!storeFile.hasReference()) {
672              HFileLink link = HFileLink.build(conf, snapshot.getTableName(),
673                  regionInfo.getEncodedName(), family, storeFile.getName());
674              long size;
675              Integer count;
676              Path p;
677              AtomicLong al;
678              int c = 0;
679
680              if (fs.exists(link.getArchivePath())) {
681                p = link.getArchivePath();
682                al = uniqueHFilesArchiveSize;
683                size = fs.getFileStatus(p).getLen();
684              } else if (fs.exists(link.getMobPath())) {
685                p = link.getMobPath();
686                al = uniqueHFilesMobSize;
687                size = fs.getFileStatus(p).getLen();
688              } else {
689                p = link.getOriginPath();
690                al = uniqueHFilesSize;
691                size = link.getFileStatus(fs).getLen();
692              }
693
694              // If it has been counted, do not double count
695              count = filesMap.get(p);
696              if (count != null) {
697                c = count.intValue();
698              } else {
699                al.addAndGet(size);
700              }
701
702              filesMap.put(p, ++c);
703            }
704          }
705        });
706  }

708  /**
709   * Returns the map of store files based on path for all snapshots
710   * @param conf the {@link Configuration} to use
711   * @param uniqueHFilesArchiveSize pass out the size for store files in archive
712   * @param uniqueHFilesSize pass out the size for store files shared
713   * @param uniqueHFilesMobSize pass out the size for mob store files shared
714   * @return the map of store files
715   */
716  public static Map<Path, Integer> getSnapshotsFilesMap(final Configuration conf,
717      AtomicLong uniqueHFilesArchiveSize, AtomicLong uniqueHFilesSize,
718      AtomicLong uniqueHFilesMobSize) throws IOException {
719    List<SnapshotDescription> snapshotList = getSnapshotList(conf);
720
721
722    if (snapshotList.isEmpty()) {
723      return Collections.emptyMap();
724    }
725
726    ConcurrentHashMap<Path, Integer> fileMap = new ConcurrentHashMap<>();
727
728    ExecutorService exec = SnapshotManifest.createExecutor(conf, "SnapshotsFilesMapping");
729
730    try {
731      for (final SnapshotDescription snapshot : snapshotList) {
732        getSnapshotFilesMap(conf, snapshot, exec, fileMap, uniqueHFilesArchiveSize,
733            uniqueHFilesSize, uniqueHFilesMobSize);
734      }
735    } finally {
736      exec.shutdown();
737    }
738
739    return fileMap;
740  }
