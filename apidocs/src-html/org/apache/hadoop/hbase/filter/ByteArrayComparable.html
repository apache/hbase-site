019package org.apache.hadoop.hbase.filter;
020
021import java.nio.ByteBuffer;
022
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.classification.InterfaceStability;
025import org.apache.hadoop.hbase.exceptions.DeserializationException;
026import org.apache.hadoop.hbase.util.ByteBufferUtils;
027import org.apache.hadoop.hbase.util.Bytes;
028
029
030/** Base class for byte array comparators */
031@InterfaceAudience.Public
032@InterfaceStability.Stable
033// TODO Now we are deviating a lot from the actual Comparable<byte[]> what this implements, by
034// adding special compareTo methods. We have to clean it. Deprecate this class and replace it
035// with a more generic one which says it compares bytes (not necessary a byte array only)
036// BytesComparable implements Comparable<Byte> will work?
037public abstract class ByteArrayComparable implements Comparable<byte[]> {
038
039  byte[] value;
040
041  /**
042   * Constructor.
043   * @param value the value to compare against
044   */
045  public ByteArrayComparable(byte [] value) {
046    this.value = value;
047  }
048
049  public byte[] getValue() {
050    return value;
051  }
052
053  /**
054   * @return The comparator serialized using pb
055   */
056  public abstract byte [] toByteArray();
057
058  /**
059   * @param pbBytes A pb serialized {@link ByteArrayComparable} instance
060   * @return An instance of {@link ByteArrayComparable} made from <code>bytes</code>
061   * @throws DeserializationException
062   * @see #toByteArray
063   */
064  public static ByteArrayComparable parseFrom(final byte [] pbBytes)
065  throws DeserializationException {
066    throw new DeserializationException(
067      "parseFrom called on base ByteArrayComparable, but should be called on derived type");
068  }
069
070  /**
071   * @param other
072   * @return true if and only if the fields of the comparator that are serialized
073   * are equal to the corresponding fields in other.  Used for testing.
074   */
075  boolean areSerializedFieldsEqual(ByteArrayComparable other) {
076    if (other == this) return true;
077
078    return Bytes.equals(this.getValue(), other.getValue());
079  }
080
081  @Override
082  public int compareTo(byte [] value) {
083    return compareTo(value, 0, value.length);
084  }
085
086  /**
087   * Special compareTo method for subclasses, to avoid
088   * copying byte[] unnecessarily.
089   * @param value byte[] to compare
090   * @param offset offset into value
091   * @param length number of bytes to compare
092   * @return a negative integer, zero, or a positive integer as this object
093   *         is less than, equal to, or greater than the specified object.
094   */
095  public abstract int compareTo(byte [] value, int offset, int length);
096
097  /**
098   * Special compareTo method for subclasses, to avoid copying bytes unnecessarily.
099   * @param value bytes to compare within a ByteBuffer
100   * @param offset offset into value
101   * @param length number of bytes to compare
102   * @return a negative integer, zero, or a positive integer as this object
103   *         is less than, equal to, or greater than the specified object.
104   */
105  public int compareTo(ByteBuffer value, int offset, int length) {
106    // For BC, providing a default implementation here which is doing a bytes copy to a temp byte[]
107    // and calling compareTo(byte[]). Make sure to override this method in subclasses to avoid
108    // copying bytes unnecessarily.
109    byte[] temp = new byte[length];
110    ByteBufferUtils.copyFromBufferToArray(temp, value, offset, 0, length);
111    return compareTo(temp);
112  }
113}
