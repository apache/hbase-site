038/**
039 * Filter to support scan multiple row key ranges. It can construct the row key ranges from the
040 * passed list which can be accessed by each region server.
041 *
042 * HBase is quite efficient when scanning only one small row key range. If user needs to specify
043 * multiple row key ranges in one scan, the typical solutions are: 1. through FilterList which is a
044 * list of row key Filters, 2. using the SQL layer over HBase to join with two table, such as hive,
045 * phoenix etc. However, both solutions are inefficient. Both of them can't utilize the range info
046 * to perform fast forwarding during scan which is quite time consuming. If the number of ranges
047 * are quite big (e.g. millions), join is a proper solution though it is slow. However, there are
048 * cases that user wants to specify a small number of ranges to scan (e.g. &lt;1000 ranges). Both
049 * solutions can't provide satisfactory performance in such case. MultiRowRangeFilter is to support
050 * such usec ase (scan multiple row key ranges), which can construct the row key ranges from user
051 * specified list and perform fast-forwarding during scan. Thus, the scan will be quite efficient.
052 */
053@InterfaceAudience.Public
054@InterfaceStability.Evolving
055public class MultiRowRangeFilter extends FilterBase {
056
057  private List<RowRange> rangeList;
058
059  private static final int ROW_BEFORE_FIRST_RANGE = -1;
060  private boolean EXCLUSIVE = false;
061  private boolean done = false;
062  private boolean initialized = false;
063  private int index;
064  private RowRange range;
065  private ReturnCode currentReturnCode;
066
067  /**
068   * @param list A list of <code>RowRange</code>
069   * @throws java.io.IOException
070   *           throw an exception if the range list is not in an natural order or any
071   *           <code>RowRange</code> is invalid
072   */
073  public MultiRowRangeFilter(List<RowRange> list) throws IOException {
074    this.rangeList = sortAndMerge(list);
075  }
076
077  @Override
078  public boolean filterAllRemaining() {
079    return done;
080  }
081
082  public List<RowRange> getRowRanges() {
083    return this.rangeList;
084  }

086  @Override
087  public boolean filterRowKey(Cell firstRowCell) {
088    if (filterAllRemaining()) return true;
089    // If it is the first time of running, calculate the current range index for
090    // the row key. If index is out of bound which happens when the start row
091    // user sets is after the largest stop row of the ranges, stop the scan.
092    // If row key is after the current range, find the next range and update index.
093    byte[] rowArr = firstRowCell.getRowArray();
094    int length = firstRowCell.getRowLength();
095    int offset = firstRowCell.getRowOffset();
096    if (!initialized
097        || !range.contains(rowArr, offset, length)) {
098      byte[] rowkey = CellUtil.cloneRow(firstRowCell);
099      index = getNextRangeIndex(rowkey);
100      if (index >= rangeList.size()) {
101        done = true;
102        currentReturnCode = ReturnCode.NEXT_ROW;
103        return false;
104      }
105      if(index != ROW_BEFORE_FIRST_RANGE) {
106        range = rangeList.get(index);
107      } else {
108        range = rangeList.get(0);
109      }
110      if (EXCLUSIVE) {
111        EXCLUSIVE = false;
112        currentReturnCode = ReturnCode.NEXT_ROW;
113        return false;
114      }
115      if (!initialized) {
116        if(index != ROW_BEFORE_FIRST_RANGE) {
117          currentReturnCode = ReturnCode.INCLUDE;
118        } else {
119          currentReturnCode = ReturnCode.SEEK_NEXT_USING_HINT;
120        }
121        initialized = true;
122      } else {
123        if (range.contains(rowArr, offset, length)) {
124          currentReturnCode = ReturnCode.INCLUDE;
125        } else currentReturnCode = ReturnCode.SEEK_NEXT_USING_HINT;
126      }
127    } else {
128      currentReturnCode = ReturnCode.INCLUDE;
129    }
130    return false;
131  }
132
133  @Override
134  public ReturnCode filterKeyValue(Cell ignored) {
135    return currentReturnCode;
136  }
137
138  @Override
139  public Cell getNextCellHint(Cell currentKV) {
140    // skip to the next range's start row
141    return KeyValueUtil.createFirstOnRow(range.startRow);
142  }

144  /**
145   * @return The filter serialized using pb
146   */
147  public byte[] toByteArray() {
148    FilterProtos.MultiRowRangeFilter.Builder builder = FilterProtos.MultiRowRangeFilter
149        .newBuilder();
150    for (RowRange range : rangeList) {
151      if (range != null) {
152        FilterProtos.RowRange.Builder rangebuilder = FilterProtos.RowRange.newBuilder();
153        if (range.startRow != null)
154          rangebuilder.setStartRow(ByteStringer.wrap(range.startRow));
155        rangebuilder.setStartRowInclusive(range.startRowInclusive);
156        if (range.stopRow != null)
157          rangebuilder.setStopRow(ByteStringer.wrap(range.stopRow));
158        rangebuilder.setStopRowInclusive(range.stopRowInclusive);
159        range.isScan = Bytes.equals(range.startRow, range.stopRow) ? 1 : 0;
160        builder.addRowRangeList(rangebuilder.build());
161      }
162    }
163    return builder.build().toByteArray();
164  }

166  /**
167   * @param pbBytes A pb serialized instance
168   * @return An instance of MultiRowRangeFilter
169   * @throws org.apache.hadoop.hbase.exceptions.DeserializationException
170   */
171  public static MultiRowRangeFilter parseFrom(final byte[] pbBytes)
172      throws DeserializationException {
173    FilterProtos.MultiRowRangeFilter proto;
174    try {
175      proto = FilterProtos.MultiRowRangeFilter.parseFrom(pbBytes);
176    } catch (InvalidProtocolBufferException e) {
177      throw new DeserializationException(e);
178    }
179    int length = proto.getRowRangeListCount();
180    List<FilterProtos.RowRange> rangeProtos = proto.getRowRangeListList();
181    List<RowRange> rangeList = new ArrayList<RowRange>(length);
182    for (FilterProtos.RowRange rangeProto : rangeProtos) {
183      RowRange range = new RowRange(rangeProto.hasStartRow() ? rangeProto.getStartRow()
184          .toByteArray() : null, rangeProto.getStartRowInclusive(), rangeProto.hasStopRow() ?
185              rangeProto.getStopRow().toByteArray() : null, rangeProto.getStopRowInclusive());
186      rangeList.add(range);
187    }
188    try {
189      return new MultiRowRangeFilter(rangeList);
190    } catch (IOException e) {
191      throw new DeserializationException("Fail to instantiate the MultiRowRangeFilter", e);
192    }
193  }

195  /**
196   * @param o the filter to compare
197   * @return true if and only if the fields of the filter that are serialized are equal to the
198   *         corresponding fields in other. Used for testing.
199   */
200  boolean areSerializedFieldsEqual(Filter o) {
201    if (o == this)
202      return true;
203    if (!(o instanceof MultiRowRangeFilter))
204      return false;
205
206    MultiRowRangeFilter other = (MultiRowRangeFilter) o;
207    if (this.rangeList.size() != other.rangeList.size())
208      return false;
209    for (int i = 0; i < rangeList.size(); ++i) {
210      RowRange thisRange = this.rangeList.get(i);
211      RowRange otherRange = other.rangeList.get(i);
212      if (!(Bytes.equals(thisRange.startRow, otherRange.startRow) && Bytes.equals(
213          thisRange.stopRow, otherRange.stopRow) && (thisRange.startRowInclusive ==
214          otherRange.startRowInclusive) && (thisRange.stopRowInclusive ==
215          otherRange.stopRowInclusive))) {
216        return false;
217      }
218    }
219    return true;
220  }

222  /**
223   * calculate the position where the row key in the ranges list.
224   *
225   * @param rowKey the row key to calculate
226   * @return index the position of the row key
227   */
228  private int getNextRangeIndex(byte[] rowKey) {
229    RowRange temp = new RowRange(rowKey, true, null, true);
230    int index = Collections.binarySearch(rangeList, temp);
231    if (index < 0) {
232      int insertionPosition = -index - 1;
233      // check if the row key in the range before the insertion position
234      if (insertionPosition != 0 && rangeList.get(insertionPosition - 1).contains(rowKey)) {
235        return insertionPosition - 1;
236      }
237      // check if the row key is before the first range
238      if (insertionPosition == 0 && !rangeList.get(insertionPosition).contains(rowKey)) {
239        return ROW_BEFORE_FIRST_RANGE;
240      }
241      if (!initialized) {
242        initialized = true;
243      }
244      return insertionPosition;
245    }
246    // the row key equals one of the start keys, and the the range exclude the start key
247    if(rangeList.get(index).startRowInclusive == false) {
248      EXCLUSIVE = true;
249    }
250    return index;
251  }

253  /**
254   * sort the ranges and if the ranges with overlap, then merge them.
255   *
256   * @param ranges the list of ranges to sort and merge.
257   * @return the ranges after sort and merge.
258   */
259  public static List<RowRange> sortAndMerge(List<RowRange> ranges) {
260    if (ranges.size() == 0) {
261      throw new IllegalArgumentException("No ranges found.");
262    }
263    List<RowRange> invalidRanges = new ArrayList<RowRange>();
264    List<RowRange> newRanges = new ArrayList<RowRange>(ranges.size());
265    Collections.sort(ranges);
266    if(ranges.get(0).isValid()) {
267      if (ranges.size() == 1) {
268        newRanges.add(ranges.get(0));
269      }
270    } else {
271      invalidRanges.add(ranges.get(0));
272    }
273
274    byte[] lastStartRow = ranges.get(0).startRow;
275    boolean lastStartRowInclusive = ranges.get(0).startRowInclusive;
276    byte[] lastStopRow = ranges.get(0).stopRow;
277    boolean lastStopRowInclusive = ranges.get(0).stopRowInclusive;
278    int i = 1;
279    for (; i < ranges.size(); i++) {
280      RowRange range = ranges.get(i);
281      if (!range.isValid()) {
282        invalidRanges.add(range);
283      }
284      if(Bytes.equals(lastStopRow, HConstants.EMPTY_BYTE_ARRAY)) {
285        newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, lastStopRow,
286            lastStopRowInclusive));
287        break;
288      }
289      // with overlap in the ranges
290      if ((Bytes.compareTo(lastStopRow, range.startRow) > 0) ||
291          (Bytes.compareTo(lastStopRow, range.startRow) == 0 && !(lastStopRowInclusive == false &&
292          range.isStartRowInclusive() == false))) {
293        if(Bytes.equals(range.stopRow, HConstants.EMPTY_BYTE_ARRAY)) {
294          newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, range.stopRow,
295              range.stopRowInclusive));
296          break;
297        }
298        // if first range contains second range, ignore the second range
299        if (Bytes.compareTo(lastStopRow, range.stopRow) >= 0) {
300          if((Bytes.compareTo(lastStopRow, range.stopRow) == 0)) {
301            if(lastStopRowInclusive == true || range.stopRowInclusive == true) {
302              lastStopRowInclusive = true;
303            }
304          }
305          if ((i + 1) == ranges.size()) {
306            newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, lastStopRow,
307                lastStopRowInclusive));
308          }
309        } else {
310          lastStopRow = range.stopRow;
311          lastStopRowInclusive = range.stopRowInclusive;
312          if ((i + 1) < ranges.size()) {
313            i++;
314            range = ranges.get(i);
315            if (!range.isValid()) {
316              invalidRanges.add(range);
317            }
318          } else {
319            newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, lastStopRow,
320                lastStopRowInclusive));
321            break;
322          }
323          while ((Bytes.compareTo(lastStopRow, range.startRow) > 0) ||
324              (Bytes.compareTo(lastStopRow, range.startRow) == 0 &&
325              (lastStopRowInclusive == true || range.startRowInclusive==true))) {
326            if(Bytes.equals(range.stopRow, HConstants.EMPTY_BYTE_ARRAY)) {
327              break;
328            }
329            // if this first range contain second range, ignore the second range
330            if (Bytes.compareTo(lastStopRow, range.stopRow) >= 0) {
331              if(lastStopRowInclusive == true || range.stopRowInclusive == true) {
332                lastStopRowInclusive = true;
333              }
334              i++;
335              if (i < ranges.size()) {
336                range = ranges.get(i);
337                if (!range.isValid()) {
338                  invalidRanges.add(range);
339                }
340              } else {
341                break;
342              }
343            } else {
344              lastStopRow = range.stopRow;
345              lastStopRowInclusive = range.stopRowInclusive;
346              i++;
347              if (i < ranges.size()) {
348                range = ranges.get(i);
349                if (!range.isValid()) {
350                  invalidRanges.add(range);
351                }
352              } else {
353                break;
354              }
355            }
356          }
357          if(Bytes.equals(range.stopRow, HConstants.EMPTY_BYTE_ARRAY)) {
358            if((Bytes.compareTo(lastStopRow, range.startRow) < 0) ||
359                (Bytes.compareTo(lastStopRow, range.startRow) == 0 &&
360                lastStopRowInclusive == false && range.startRowInclusive == false)) {
361              newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, lastStopRow,
362                  lastStopRowInclusive));
363              newRanges.add(range);
364            } else {
365              newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, range.stopRow,
366                  range.stopRowInclusive));
367              break;
368            }
369          }
370          newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, lastStopRow,
371              lastStopRowInclusive));
372          if ((i + 1) == ranges.size()) {
373            newRanges.add(range);
374          }
375          lastStartRow = range.startRow;
376          lastStartRowInclusive = range.startRowInclusive;
377          lastStopRow = range.stopRow;
378          lastStopRowInclusive = range.stopRowInclusive;
379        }
380      } else {
381        newRanges.add(new RowRange(lastStartRow, lastStartRowInclusive, lastStopRow,
382            lastStopRowInclusive));
383        if ((i + 1) == ranges.size()) {
384          newRanges.add(range);
385        }
386        lastStartRow = range.startRow;
387        lastStartRowInclusive = range.startRowInclusive;
388        lastStopRow = range.stopRow;
389        lastStopRowInclusive = range.stopRowInclusive;
390      }
391    }
392    // check the remaining ranges
393    for(int j=i; j < ranges.size(); j++) {
394      if(!ranges.get(j).isValid()) {
395        invalidRanges.add(ranges.get(j));
396      }
397    }
398    // if invalid range exists, throw the exception
399    if (invalidRanges.size() != 0) {
400      throwExceptionForInvalidRanges(invalidRanges, true);
401    }
402    // If no valid ranges found, throw the exception
403    if(newRanges.size() == 0) {
404      throw new IllegalArgumentException("No valid ranges found.");
405    }
406    return newRanges;
407  }
408
409  private static void throwExceptionForInvalidRanges(List<RowRange> invalidRanges,
410      boolean details) {
411    StringBuilder sb = new StringBuilder();
412    sb.append(invalidRanges.size()).append(" invaild ranges.\n");
413    if (details) {
414      for (RowRange range : invalidRanges) {
415        sb.append(
416            "Invalid range: start row => " + Bytes.toString(range.startRow) + ", stop row => "
417                + Bytes.toString(range.stopRow)).append('\n');
418      }
419    }
420    throw new IllegalArgumentException(sb.toString());
421  }

423  @InterfaceAudience.Public
424  @InterfaceStability.Evolving
425  public static class RowRange implements Comparable<RowRange> {
426    private byte[] startRow;
427    private boolean startRowInclusive = true;
428    private byte[] stopRow;
429    private boolean stopRowInclusive = false;
430    private int isScan = 0;
431
432    public RowRange() {
433    }
434    /**
435     * If the startRow is empty or null, set it to HConstants.EMPTY_BYTE_ARRAY, means begin at the
436     * start row of the table. If the stopRow is empty or null, set it to
437     * HConstants.EMPTY_BYTE_ARRAY, means end of the last row of table.
438     */
439    public RowRange(String startRow, boolean startRowInclusive, String stopRow,
440        boolean stopRowInclusive) {
441      this((startRow == null || startRow.isEmpty()) ? HConstants.EMPTY_BYTE_ARRAY :
442        Bytes.toBytes(startRow), startRowInclusive,
443        (stopRow == null || stopRow.isEmpty()) ? HConstants.EMPTY_BYTE_ARRAY :
444        Bytes.toBytes(stopRow), stopRowInclusive);
445    }
446
447    public RowRange(byte[] startRow,  boolean startRowInclusive, byte[] stopRow,
448        boolean stopRowInclusive) {
449      this.startRow = (startRow == null) ? HConstants.EMPTY_BYTE_ARRAY : startRow;
450      this.startRowInclusive = startRowInclusive;
451      this.stopRow = (stopRow == null) ? HConstants.EMPTY_BYTE_ARRAY :stopRow;
452      this.stopRowInclusive = stopRowInclusive;
453      isScan = Bytes.equals(startRow, stopRow) ? 1 : 0;
454    }
455
456    public byte[] getStartRow() {
457      return startRow;
458    }
459
460    public byte[] getStopRow() {
461      return stopRow;
462    }
463
464    /**
465     * @return if start row is inclusive.
466     */
467    public boolean isStartRowInclusive() {
468      return startRowInclusive;
469    }
470
471    /**
472     * @return if stop row is inclusive.
473     */
474    public boolean isStopRowInclusive() {
475      return stopRowInclusive;
476    }
477
478    public boolean contains(byte[] row) {
479      return contains(row, 0, row.length);
480    }
481
482    public boolean contains(byte[] buffer, int offset, int length) {
483      if(startRowInclusive) {
484        if(stopRowInclusive) {
485          return Bytes.compareTo(buffer, offset, length, startRow, 0, startRow.length) >= 0
486              && (Bytes.equals(stopRow, HConstants.EMPTY_BYTE_ARRAY) ||
487                  Bytes.compareTo(buffer, offset, length, stopRow, 0, stopRow.length) <= isScan);
488        } else {
489          return Bytes.compareTo(buffer, offset, length, startRow, 0, startRow.length) >= 0
490              && (Bytes.equals(stopRow, HConstants.EMPTY_BYTE_ARRAY) ||
491                  Bytes.compareTo(buffer, offset, length, stopRow, 0, stopRow.length) < isScan);
492        }
493      } else {
494        if(stopRowInclusive) {
495          return Bytes.compareTo(buffer, offset, length, startRow, 0, startRow.length) > 0
496              && (Bytes.equals(stopRow, HConstants.EMPTY_BYTE_ARRAY) ||
497                  Bytes.compareTo(buffer, offset, length, stopRow, 0, stopRow.length) <= isScan);
498        } else {
499          return Bytes.compareTo(buffer, offset, length, startRow, 0, startRow.length) > 0
500              && (Bytes.equals(stopRow, HConstants.EMPTY_BYTE_ARRAY) ||
501                  Bytes.compareTo(buffer, offset, length, stopRow, 0, stopRow.length) < isScan);
502        }
503      }
504    }
505
506    @Override
507    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="EQ_COMPARETO_USE_OBJECT_EQUALS",
508      justification="This compareTo is not of this Object, but of referenced RowRange")
509    public int compareTo(RowRange other) {
510      return Bytes.compareTo(this.startRow, other.startRow);
511    }
512
513    public boolean isValid() {
514      return Bytes.equals(startRow, HConstants.EMPTY_BYTE_ARRAY)
515          || Bytes.equals(stopRow, HConstants.EMPTY_BYTE_ARRAY)
516          || Bytes.compareTo(startRow, stopRow) < 0
517          || (Bytes.compareTo(startRow, stopRow) == 0 && stopRowInclusive == true);
518    }
519  }
