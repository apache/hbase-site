018package org.apache.hadoop.hbase.filter;
019
020import java.nio.ByteBuffer;
021import org.apache.hadoop.hbase.exceptions.DeserializationException;
022import org.apache.yetus.audience.InterfaceAudience;
023
024import org.apache.hbase.thirdparty.com.google.protobuf.InvalidProtocolBufferException;
025
026import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
027import org.apache.hadoop.hbase.shaded.protobuf.generated.ComparatorProtos;
028
029/**
030 * A bit comparator which performs the specified bitwise operation on each of the bytes with the
031 * specified byte array. Then returns whether the result is non-zero.
032 */
033@InterfaceAudience.Public
034@SuppressWarnings("ComparableType") // Should this move to Comparator usage?
035public class BitComparator extends ByteArrayComparable {
036
037  /** Bit operators. */
038  @InterfaceAudience.Public
039  public enum BitwiseOp {
040    /** and */
041    AND,
042    /** or */
043    OR,
044    /** xor */
045    XOR
046  }
047
048  protected BitwiseOp bitOperator;
049
050  /**
051   * Constructor
052   * @param value       value
053   * @param bitOperator operator to use on the bit comparison
054   */
055  public BitComparator(byte[] value, BitwiseOp bitOperator) {
056    super(value);
057    this.bitOperator = bitOperator;
058  }
059
060  /** Returns the bitwise operator */
061  public BitwiseOp getOperator() {
062    return bitOperator;
063  }
064
065  /** Returns The comparator serialized using pb */
066  @Override
067  public byte[] toByteArray() {
068    ComparatorProtos.BitComparator.Builder builder = ComparatorProtos.BitComparator.newBuilder();
069    builder.setComparable(ProtobufUtil.toByteArrayComparable(this.value));
070    ComparatorProtos.BitComparator.BitwiseOp bitwiseOpPb =
071      ComparatorProtos.BitComparator.BitwiseOp.valueOf(bitOperator.name());
072    builder.setBitwiseOp(bitwiseOpPb);
073    return builder.build().toByteArray();
074  }
075
076  /**
077   * @param pbBytes A pb serialized {@link BitComparator} instance
078   * @return An instance of {@link BitComparator} made from <code>bytes</code> n * @see #toByteArray
079   */
080  public static BitComparator parseFrom(final byte[] pbBytes) throws DeserializationException {
081    ComparatorProtos.BitComparator proto;
082    try {
083      proto = ComparatorProtos.BitComparator.parseFrom(pbBytes);
084    } catch (InvalidProtocolBufferException e) {
085      throw new DeserializationException(e);
086    }
087    BitwiseOp bitwiseOp = BitwiseOp.valueOf(proto.getBitwiseOp().name());
088    return new BitComparator(proto.getComparable().getValue().toByteArray(), bitwiseOp);
089  }
090
091  /**
092   * n * @return true if and only if the fields of the comparator that are serialized are equal to
093   * the corresponding fields in other. Used for testing.
094   */
095  @Override
096  boolean areSerializedFieldsEqual(ByteArrayComparable other) {
097    if (other == this) return true;
098    if (!(other instanceof BitComparator)) return false;
099
100    BitComparator comparator = (BitComparator) other;
101    return super.areSerializedFieldsEqual(other)
102      && this.getOperator().equals(comparator.getOperator());
103  }
104
105  @Override
106  public int compareTo(byte[] value, int offset, int length) {
107    if (length != this.value.length) {
108      return 1;
109    }
110    int b = 0;
111    // Iterating backwards is faster because we can quit after one non-zero byte.
112    for (int i = length - 1; i >= 0 && b == 0; i--) {
113      switch (bitOperator) {
114        case AND:
115          b = (this.value[i] & value[i + offset]) & 0xff;
116          break;
117        case OR:
118          b = (this.value[i] | value[i + offset]) & 0xff;
119          break;
120        case XOR:
121          b = (this.value[i] ^ value[i + offset]) & 0xff;
122          break;
123      }
124    }
125    return b == 0 ? 1 : 0;
126  }
127
128  @Override
129  public int compareTo(ByteBuffer value, int offset, int length) {
130    if (length != this.value.length) {
131      return 1;
132    }
133    int b = 0;
134    // Iterating backwards is faster because we can quit after one non-zero byte.
135    for (int i = length - 1; i >= 0 && b == 0; i--) {
136      switch (bitOperator) {
137        case AND:
138          b = (this.value[i] & value.get(i + offset)) & 0xff;
139          break;
140        case OR:
141          b = (this.value[i] | value.get(i + offset)) & 0xff;
142          break;
143        case XOR:
144          b = (this.value[i] ^ value.get(i + offset)) & 0xff;
145          break;
146      }
147    }
148    return b == 0 ? 1 : 0;
149  }
150}
