036/**
037 * This is a generic filter to be used to filter by comparison.  It takes an
038 * operator (equal, greater, not equal, etc) and a byte [] comparator.
039 * <p>
040 * To filter by row key, use {@link RowFilter}.
041 * <p>
042 * To filter by column qualifier, use {@link QualifierFilter}.
043 * <p>
044 * To filter by value, use {@link SingleColumnValueFilter}.
045 * <p>
046 * These filters can be wrapped with {@link SkipFilter} and {@link WhileMatchFilter}
047 * to add more control.
048 * <p>
049 * Multiple filters can be combined using {@link FilterList}.
050 */
051@InterfaceAudience.Public
052public abstract class CompareFilter extends FilterBase {
053  /**
054   * Comparison operators. For filters only!
055   * Use {@link CompareOperator} otherwise.
056   * It (intentionally) has at least the below enums with same names.
057   * @deprecated  since 2.0.0. Will be removed in 3.0.0. Use {@link CompareOperator} instead.
058   */
059  @Deprecated
060  @InterfaceAudience.Public
061  public enum CompareOp {
062    /** less than */
063    LESS,
064    /** less than or equal to */
065    LESS_OR_EQUAL,
066    /** equals */
067    EQUAL,
068    /** not equal */
069    NOT_EQUAL,
070    /** greater than or equal to */
071    GREATER_OR_EQUAL,
072    /** greater than */
073    GREATER,
074    /** no operation */
075    NO_OP,
076  }
077
078  protected CompareOperator op;
079  protected ByteArrayComparable comparator;

224
225  static boolean compare(final CompareOp op, int compareResult) {
226    switch (op) {
227    case LESS:
228      return compareResult <= 0;
229    case LESS_OR_EQUAL:
230      return compareResult < 0;
231    case EQUAL:
232      return compareResult != 0;
233    case NOT_EQUAL:
234      return compareResult == 0;
235    case GREATER_OR_EQUAL:
236      return compareResult > 0;
237    case GREATER:
238      return compareResult >= 0;
239    default:
240      throw new RuntimeException("Unknown Compare op " + op.name());
241    }
242  }
243
244  static boolean compare(final CompareOperator op, int compareResult) {
245    switch (op) {
246      case LESS:
247        return compareResult <= 0;
248      case LESS_OR_EQUAL:
249        return compareResult < 0;
250      case EQUAL:
251        return compareResult != 0;
252      case NOT_EQUAL:
253        return compareResult == 0;
254      case GREATER_OR_EQUAL:
255        return compareResult > 0;
256      case GREATER:
257        return compareResult >= 0;
258      default:
259        throw new RuntimeException("Unknown Compare op " + op.name());
260    }
261  }
262
263  // returns an array of heterogeneous objects
264  public static ArrayList<Object> extractArguments(ArrayList<byte []> filterArguments) {
265    Preconditions.checkArgument(filterArguments.size() == 2,
266                                "Expected 2 but got: %s", filterArguments.size());
267    CompareOperator op = ParseFilter.createCompareOperator(filterArguments.get(0));
268    ByteArrayComparable comparator = ParseFilter.createComparator(
269      ParseFilter.removeQuotesFromByteArray(filterArguments.get(1)));
270
271    if (comparator instanceof RegexStringComparator ||
272        comparator instanceof SubstringComparator) {
273      if (op != CompareOperator.EQUAL &&
274          op != CompareOperator.NOT_EQUAL) {
275        throw new IllegalArgumentException ("A regexstring comparator and substring comparator" +
276                                            " can only be used with EQUAL and NOT_EQUAL");
277      }
278    }
279    ArrayList<Object> arguments = new ArrayList<>(2);
280    arguments.add(op);
281    arguments.add(comparator);
282    return arguments;
283  }

285  /**
286   * @return A pb instance to represent this instance.
287   */
288  FilterProtos.CompareFilter convert() {
289    FilterProtos.CompareFilter.Builder builder =
290      FilterProtos.CompareFilter.newBuilder();
291    HBaseProtos.CompareType compareOp = CompareType.valueOf(this.op.name());
292    builder.setCompareOp(compareOp);
293    if (this.comparator != null) builder.setComparator(ProtobufUtil.toComparator(this.comparator));
294    return builder.build();
295  }
296
297  /**
298   *
299   * @param o
300   * @return true if and only if the fields of the filter that are serialized
301   * are equal to the corresponding fields in other.  Used for testing.
302   */
303  @Override
304  boolean areSerializedFieldsEqual(Filter o) {
305    if (o == this) return true;
306    if (!(o instanceof CompareFilter)) return false;
307    CompareFilter other = (CompareFilter)o;
308    return this.getCompareOperator().equals(other.getCompareOperator()) &&
309      (this.getComparator() == other.getComparator()
310        || this.getComparator().areSerializedFieldsEqual(other.getComparator()));
311  }
312
313  @Override
314  public String toString() {
315    return String.format("%s (%s, %s)",
316        this.getClass().getSimpleName(),
317        this.op.name(),
318        Bytes.toStringBinary(this.comparator.getValue()));
319  }
