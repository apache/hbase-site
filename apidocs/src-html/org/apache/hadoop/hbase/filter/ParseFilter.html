115  /**
116   * Parses the filterString and constructs a filter using it
117   * <p>
118   * @param filterStringAsByteArray filter string given by the user
119   * @return filter object we constructed
120   */
121  public Filter parseFilterString (byte [] filterStringAsByteArray)
122    throws CharacterCodingException {
123    // stack for the operators and parenthesis
124    Stack <ByteBuffer> operatorStack = new Stack<>();
125    // stack for the filter objects
126    Stack <Filter> filterStack = new Stack<>();
127
128    Filter filter = null;
129    for (int i=0; i<filterStringAsByteArray.length; i++) {
130      if (filterStringAsByteArray[i] == ParseConstants.LPAREN) {
131        // LPAREN found
132        operatorStack.push(ParseConstants.LPAREN_BUFFER);
133      } else if (filterStringAsByteArray[i] == ParseConstants.WHITESPACE ||
134                 filterStringAsByteArray[i] == ParseConstants.TAB) {
135        // WHITESPACE or TAB found
136        continue;
137      } else if (checkForOr(filterStringAsByteArray, i)) {
138        // OR found
139        i += ParseConstants.OR_ARRAY.length - 1;
140        reduce(operatorStack, filterStack, ParseConstants.OR_BUFFER);
141        operatorStack.push(ParseConstants.OR_BUFFER);
142      } else if (checkForAnd(filterStringAsByteArray, i)) {
143        // AND found
144        i += ParseConstants.AND_ARRAY.length - 1;
145        reduce(operatorStack, filterStack, ParseConstants.AND_BUFFER);
146        operatorStack.push(ParseConstants.AND_BUFFER);
147      } else if (checkForSkip(filterStringAsByteArray, i)) {
148        // SKIP found
149        i += ParseConstants.SKIP_ARRAY.length - 1;
150        reduce(operatorStack, filterStack, ParseConstants.SKIP_BUFFER);
151        operatorStack.push(ParseConstants.SKIP_BUFFER);
152      } else if (checkForWhile(filterStringAsByteArray, i)) {
153        // WHILE found
154        i += ParseConstants.WHILE_ARRAY.length - 1;
155        reduce(operatorStack, filterStack, ParseConstants.WHILE_BUFFER);
156        operatorStack.push(ParseConstants.WHILE_BUFFER);
157      } else if (filterStringAsByteArray[i] == ParseConstants.RPAREN) {
158        // RPAREN found
159        if (operatorStack.empty()) {
160          throw new IllegalArgumentException("Mismatched parenthesis");
161        }
162        ByteBuffer argumentOnTopOfStack = operatorStack.peek();
163        if (argumentOnTopOfStack.equals(ParseConstants.LPAREN_BUFFER)) {
164          operatorStack.pop();
165          continue;
166        }
167        while (!(argumentOnTopOfStack.equals(ParseConstants.LPAREN_BUFFER))) {
168          filterStack.push(popArguments(operatorStack, filterStack));
169          if (operatorStack.empty()) {
170            throw new IllegalArgumentException("Mismatched parenthesis");
171          }
172          argumentOnTopOfStack = operatorStack.pop();
173        }
174      } else {
175        // SimpleFilterExpression found
176        byte [] filterSimpleExpression = extractFilterSimpleExpression(filterStringAsByteArray, i);
177        i+= (filterSimpleExpression.length - 1);
178        filter = parseSimpleFilterExpression(filterSimpleExpression);
179        filterStack.push(filter);
180      }
181    }
182
183    // Finished parsing filterString
184    while (!operatorStack.empty()) {
185      filterStack.push(popArguments(operatorStack, filterStack));
186    }
187    if (filterStack.empty()) {
188        throw new IllegalArgumentException("Incorrect Filter String");
189    }
190    filter = filterStack.pop();
191    if (!filterStack.empty()) {
192      throw new IllegalArgumentException("Incorrect Filter String");
193    }
194    return filter;
195  }

197/**
198 * Extracts a simple filter expression from the filter string given by the user
199 * <p>
200 * A simpleFilterExpression is of the form: FilterName('arg', 'arg', 'arg')
201 * The user given filter string can have many simpleFilterExpressions combined
202 * using operators.
203 * <p>
204 * This function extracts a simpleFilterExpression from the
205 * larger filterString given the start offset of the simpler expression
206 * <p>
207 * @param filterStringAsByteArray filter string given by the user
208 * @param filterExpressionStartOffset start index of the simple filter expression
209 * @return byte array containing the simple filter expression
210 */
211  public byte [] extractFilterSimpleExpression (byte [] filterStringAsByteArray,
212                                                int filterExpressionStartOffset)
213    throws CharacterCodingException {
214    int quoteCount = 0;
215    for (int i=filterExpressionStartOffset; i<filterStringAsByteArray.length; i++) {
216      if (filterStringAsByteArray[i] == ParseConstants.SINGLE_QUOTE) {
217        if (isQuoteUnescaped(filterStringAsByteArray, i)) {
218          quoteCount ++;
219        } else {
220          // To skip the next quote that has been escaped
221          i++;
222        }
223      }
224      if (filterStringAsByteArray[i] == ParseConstants.RPAREN && (quoteCount %2 ) == 0) {
225        byte [] filterSimpleExpression = new byte [i - filterExpressionStartOffset + 1];
226        Bytes.putBytes(filterSimpleExpression, 0, filterStringAsByteArray,
227                       filterExpressionStartOffset, i-filterExpressionStartOffset + 1);
228        return filterSimpleExpression;
229      }
230    }
231    throw new IllegalArgumentException("Incorrect Filter String");
232  }

234/**
235 * Constructs a filter object given a simple filter expression
236 * <p>
237 * @param filterStringAsByteArray filter string given by the user
238 * @return filter object we constructed
239 */
240  public Filter parseSimpleFilterExpression (byte [] filterStringAsByteArray)
241    throws CharacterCodingException {
242
243    String filterName = Bytes.toString(getFilterName(filterStringAsByteArray));
244    ArrayList<byte []> filterArguments = getFilterArguments(filterStringAsByteArray);
245    if (!filterHashMap.containsKey(filterName)) {
246      throw new IllegalArgumentException("Filter Name " + filterName + " not supported");
247    }
248    try {
249      filterName = filterHashMap.get(filterName);
250      Class<?> c = Class.forName(filterName);
251      Class<?>[] argTypes = new Class [] {ArrayList.class};
252      Method m = c.getDeclaredMethod("createFilterFromArguments", argTypes);
253      return (Filter) m.invoke(null,filterArguments);
254    } catch (ClassNotFoundException e) {
255      e.printStackTrace();
256    } catch (NoSuchMethodException e) {
257      e.printStackTrace();
258    } catch (IllegalAccessException e) {
259      e.printStackTrace();
260    } catch (InvocationTargetException e) {
261      e.printStackTrace();
262    }
263    throw new IllegalArgumentException("Incorrect filter string " +
264                                       new String(filterStringAsByteArray));
265  }

267/**
268 * Returns the filter name given a simple filter expression
269 * <p>
270 * @param filterStringAsByteArray a simple filter expression
271 * @return name of filter in the simple filter expression
272 */
273  public static byte [] getFilterName (byte [] filterStringAsByteArray) {
274    int filterNameStartIndex = 0;
275    int filterNameEndIndex = 0;
276
277    for (int i=filterNameStartIndex; i<filterStringAsByteArray.length; i++) {
278      if (filterStringAsByteArray[i] == ParseConstants.LPAREN ||
279          filterStringAsByteArray[i] == ParseConstants.WHITESPACE) {
280        filterNameEndIndex = i;
281        break;
282      }
283    }
284
285    if (filterNameEndIndex == 0) {
286      throw new IllegalArgumentException("Incorrect Filter Name");
287    }
288
289    byte [] filterName = new byte[filterNameEndIndex - filterNameStartIndex];
290    Bytes.putBytes(filterName, 0, filterStringAsByteArray, 0,
291                   filterNameEndIndex - filterNameStartIndex);
292    return filterName;
293  }

295/**
296 * Returns the arguments of the filter from the filter string
297 * <p>
298 * @param filterStringAsByteArray filter string given by the user
299 * @return an ArrayList containing the arguments of the filter in the filter string
300 */
301  public static ArrayList<byte []> getFilterArguments (byte [] filterStringAsByteArray) {
302    int argumentListStartIndex = Bytes.searchDelimiterIndex(filterStringAsByteArray, 0,
303                                                       filterStringAsByteArray.length,
304                                                       ParseConstants.LPAREN);
305    if (argumentListStartIndex == -1) {
306      throw new IllegalArgumentException("Incorrect argument list");
307    }
308
309    int argumentStartIndex = 0;
310    int argumentEndIndex = 0;
311    ArrayList<byte []> filterArguments = new ArrayList<>();
312
313    for (int i = argumentListStartIndex + 1; i<filterStringAsByteArray.length; i++) {
314
315      if (filterStringAsByteArray[i] == ParseConstants.WHITESPACE ||
316          filterStringAsByteArray[i] == ParseConstants.COMMA ||
317          filterStringAsByteArray[i] == ParseConstants.RPAREN) {
318        continue;
319      }
320
321      // The argument is in single quotes - for example 'prefix'
322      if (filterStringAsByteArray[i] == ParseConstants.SINGLE_QUOTE) {
323        argumentStartIndex = i;
324        for (int j = argumentStartIndex+1; j < filterStringAsByteArray.length; j++) {
325          if (filterStringAsByteArray[j] == ParseConstants.SINGLE_QUOTE) {
326            if (isQuoteUnescaped(filterStringAsByteArray,j)) {
327              argumentEndIndex = j;
328              i = j+1;
329              byte [] filterArgument = createUnescapdArgument(filterStringAsByteArray,
330                                                              argumentStartIndex, argumentEndIndex);
331              filterArguments.add(filterArgument);
332              break;
333            } else {
334              // To jump over the second escaped quote
335              j++;
336            }
337          } else if (j == filterStringAsByteArray.length - 1) {
338            throw new IllegalArgumentException("Incorrect argument list");
339          }
340        }
341      } else {
342        // The argument is an integer, boolean, comparison operator like <, >, != etc
343        argumentStartIndex = i;
344        for (int j = argumentStartIndex; j < filterStringAsByteArray.length; j++) {
345          if (filterStringAsByteArray[j] == ParseConstants.WHITESPACE ||
346              filterStringAsByteArray[j] == ParseConstants.COMMA ||
347              filterStringAsByteArray[j] == ParseConstants.RPAREN) {
348            argumentEndIndex = j - 1;
349            i = j;
350            byte [] filterArgument = new byte [argumentEndIndex - argumentStartIndex + 1];
351            Bytes.putBytes(filterArgument, 0, filterStringAsByteArray,
352                           argumentStartIndex, argumentEndIndex - argumentStartIndex + 1);
353            filterArguments.add(filterArgument);
354            break;
355          } else if (j == filterStringAsByteArray.length - 1) {
356            throw new IllegalArgumentException("Incorrect argument list");
357          }
358        }
359      }
360    }
361    return filterArguments;
362  }

364/**
365 * This function is called while parsing the filterString and an operator is parsed
366 * <p>
367 * @param operatorStack the stack containing the operators and parenthesis
368 * @param filterStack the stack containing the filters
369 * @param operator the operator found while parsing the filterString
370 */
371  public void reduce(Stack<ByteBuffer> operatorStack,
372                     Stack<Filter> filterStack,
373                     ByteBuffer operator) {
374    while (!operatorStack.empty() &&
375           !(ParseConstants.LPAREN_BUFFER.equals(operatorStack.peek())) &&
376           hasHigherPriority(operatorStack.peek(), operator)) {
377      filterStack.push(popArguments(operatorStack, filterStack));
378    }
379  }

381  /**
382   * Pops an argument from the operator stack and the number of arguments required by the operator
383   * from the filterStack and evaluates them
384   * <p>
385   * @param operatorStack the stack containing the operators
386   * @param filterStack the stack containing the filters
387   * @return the evaluated filter
388   */
389  public static Filter popArguments (Stack<ByteBuffer> operatorStack, Stack <Filter> filterStack) {
390    ByteBuffer argumentOnTopOfStack = operatorStack.peek();
391
392    if (argumentOnTopOfStack.equals(ParseConstants.OR_BUFFER)) {
393      // The top of the stack is an OR
394      try {
395        ArrayList<Filter> listOfFilters = new ArrayList<>();
396        while (!operatorStack.empty() && operatorStack.peek().equals(ParseConstants.OR_BUFFER)) {
397          Filter filter = filterStack.pop();
398          listOfFilters.add(0, filter);
399          operatorStack.pop();
400        }
401        Filter filter = filterStack.pop();
402        listOfFilters.add(0, filter);
403        Filter orFilter = new FilterList(FilterList.Operator.MUST_PASS_ONE, listOfFilters);
404        return orFilter;
405      } catch (EmptyStackException e) {
406        throw new IllegalArgumentException("Incorrect input string - an OR needs two filters");
407      }
408
409    } else if (argumentOnTopOfStack.equals(ParseConstants.AND_BUFFER)) {
410      // The top of the stack is an AND
411      try {
412        ArrayList<Filter> listOfFilters = new ArrayList<>();
413        while (!operatorStack.empty() && operatorStack.peek().equals(ParseConstants.AND_BUFFER)) {
414          Filter filter = filterStack.pop();
415          listOfFilters.add(0, filter);
416          operatorStack.pop();
417        }
418        Filter filter = filterStack.pop();
419        listOfFilters.add(0, filter);
420        Filter andFilter = new FilterList(FilterList.Operator.MUST_PASS_ALL, listOfFilters);
421        return andFilter;
422      } catch (EmptyStackException e) {
423        throw new IllegalArgumentException("Incorrect input string - an AND needs two filters");
424      }
425
426    } else if (argumentOnTopOfStack.equals(ParseConstants.SKIP_BUFFER)) {
427      // The top of the stack is a SKIP
428      try {
429        Filter wrappedFilter = filterStack.pop();
430        Filter skipFilter = new SkipFilter(wrappedFilter);
431        operatorStack.pop();
432        return skipFilter;
433      } catch (EmptyStackException e) {
434        throw new IllegalArgumentException("Incorrect input string - a SKIP wraps a filter");
435      }
436
437    } else if (argumentOnTopOfStack.equals(ParseConstants.WHILE_BUFFER)) {
438      // The top of the stack is a WHILE
439      try {
440        Filter wrappedFilter = filterStack.pop();
441        Filter whileMatchFilter = new WhileMatchFilter(wrappedFilter);
442        operatorStack.pop();
443        return whileMatchFilter;
444      } catch (EmptyStackException e) {
445        throw new IllegalArgumentException("Incorrect input string - a WHILE wraps a filter");
446      }
447
448    } else if (argumentOnTopOfStack.equals(ParseConstants.LPAREN_BUFFER)) {
449      // The top of the stack is a LPAREN
450      try {
451        Filter filter  = filterStack.pop();
452        operatorStack.pop();
453        return filter;
454      } catch (EmptyStackException e) {
455        throw new IllegalArgumentException("Incorrect Filter String");
456      }
457
458    } else {
459      throw new IllegalArgumentException("Incorrect arguments on operatorStack");
460    }
461  }

463/**
464 * Returns which operator has higher precedence
465 * <p>
466 * If a has higher precedence than b, it returns true
467 * If they have the same precedence, it returns false
468 */
469  public boolean hasHigherPriority(ByteBuffer a, ByteBuffer b) {
470    if ((operatorPrecedenceHashMap.get(a) - operatorPrecedenceHashMap.get(b)) < 0) {
471      return true;
472    }
473    return false;
474  }

476/**
477 * Removes the single quote escaping a single quote - thus it returns an unescaped argument
478 * <p>
479 * @param filterStringAsByteArray filter string given by user
480 * @param argumentStartIndex start index of the argument
481 * @param argumentEndIndex end index of the argument
482 * @return returns an unescaped argument
483 */
484  public static byte [] createUnescapdArgument (byte [] filterStringAsByteArray,
485                                                int argumentStartIndex, int argumentEndIndex) {
486    int unescapedArgumentLength = 2;
487    for (int i = argumentStartIndex + 1; i <= argumentEndIndex - 1; i++) {
488      unescapedArgumentLength ++;
489      if (filterStringAsByteArray[i] == ParseConstants.SINGLE_QUOTE &&
490          i != (argumentEndIndex - 1) &&
491          filterStringAsByteArray[i+1] == ParseConstants.SINGLE_QUOTE) {
492        i++;
493        continue;
494      }
495    }
496
497    byte [] unescapedArgument = new byte [unescapedArgumentLength];
498    int count = 1;
499    unescapedArgument[0] = '\'';
500    for (int i = argumentStartIndex + 1; i <= argumentEndIndex - 1; i++) {
501      if (filterStringAsByteArray [i] == ParseConstants.SINGLE_QUOTE &&
502          i != (argumentEndIndex - 1) &&
503          filterStringAsByteArray [i+1] == ParseConstants.SINGLE_QUOTE) {
504        unescapedArgument[count++] = filterStringAsByteArray [i+1];
505        i++;
506      }
507      else {
508        unescapedArgument[count++] = filterStringAsByteArray [i];
509      }
510    }
511    unescapedArgument[unescapedArgumentLength - 1] = '\'';
512    return unescapedArgument;
513  }

515/**
516 * Checks if the current index of filter string we are on is the beginning of the keyword 'OR'
517 * <p>
518 * @param filterStringAsByteArray filter string given by the user
519 * @param indexOfOr index at which an 'O' was read
520 * @return true if the keyword 'OR' is at the current index
521 */
522  public static boolean checkForOr (byte [] filterStringAsByteArray, int indexOfOr)
523    throws CharacterCodingException, ArrayIndexOutOfBoundsException {
524
525    try {
526      if (filterStringAsByteArray[indexOfOr] == ParseConstants.O &&
527          filterStringAsByteArray[indexOfOr+1] == ParseConstants.R &&
528          (filterStringAsByteArray[indexOfOr-1] == ParseConstants.WHITESPACE ||
529           filterStringAsByteArray[indexOfOr-1] == ParseConstants.RPAREN) &&
530          (filterStringAsByteArray[indexOfOr+2] == ParseConstants.WHITESPACE ||
531           filterStringAsByteArray[indexOfOr+2] == ParseConstants.LPAREN)) {
532        return true;
533      } else {
534        return false;
535      }
536    } catch (ArrayIndexOutOfBoundsException e) {
537      return false;
538    }
539  }

628/**

631 * @param array byte array in which the quote was found
632 * @param quoteIndex index of the single quote
633 * @return returns true if the quote was unescaped
634 */
635  public static boolean isQuoteUnescaped (byte [] array, int quoteIndex) {
636    if (array == null) {
637      throw new IllegalArgumentException("isQuoteUnescaped called with a null array");
638    }
639
640    if (quoteIndex == array.length - 1 || array[quoteIndex+1] != ParseConstants.SINGLE_QUOTE) {
641      return true;
642    }
643    else {
644      return false;
645    }
646  }

648/**
649 * Takes a quoted byte array and converts it into an unquoted byte array
650 * For example: given a byte array representing 'abc', it returns a
651 * byte array representing abc
652 * <p>
653 * @param quotedByteArray the quoted byte array
654 * @return Unquoted byte array
655 */
656  public static byte [] removeQuotesFromByteArray (byte [] quotedByteArray) {
657    if (quotedByteArray == null ||
658        quotedByteArray.length < 2 ||
659        quotedByteArray[0] != ParseConstants.SINGLE_QUOTE ||
660        quotedByteArray[quotedByteArray.length - 1] != ParseConstants.SINGLE_QUOTE) {
661      throw new IllegalArgumentException("removeQuotesFromByteArray needs a quoted byte array");
662    } else {
663      byte [] targetString = new byte [quotedByteArray.length - 2];
664      Bytes.putBytes(targetString, 0, quotedByteArray, 1, quotedByteArray.length - 2);
665      return targetString;
666    }
667  }

669/**
670 * Converts an int expressed in a byte array to an actual int
671 * <p>
672 * This doesn't use Bytes.toInt because that assumes
673 * that there will be {@link Bytes#SIZEOF_INT} bytes available.
674 * <p>
675 * @param numberAsByteArray the int value expressed as a byte array
676 * @return the int value
677 */
678  public static int convertByteArrayToInt (byte [] numberAsByteArray) {
679
680    long tempResult = ParseFilter.convertByteArrayToLong(numberAsByteArray);
681
682    if (tempResult > Integer.MAX_VALUE) {
683      throw new IllegalArgumentException("Integer Argument too large");
684    } else if (tempResult < Integer.MIN_VALUE) {
685      throw new IllegalArgumentException("Integer Argument too small");
686    }
687
688    int result = (int) tempResult;
689    return result;
690  }

692/**
693 * Converts a long expressed in a byte array to an actual long
694 * <p>
695 * This doesn't use Bytes.toLong because that assumes
696 * that there will be {@link Bytes#SIZEOF_INT} bytes available.
697 * <p>
698 * @param numberAsByteArray the long value expressed as a byte array
699 * @return the long value
700 */
701  public static long convertByteArrayToLong (byte [] numberAsByteArray) {
702    if (numberAsByteArray == null) {
703      throw new IllegalArgumentException("convertByteArrayToLong called with a null array");
704    }
705
706    int i = 0;
707    long result = 0;
708    boolean isNegative = false;
709
710    if (numberAsByteArray[i] == ParseConstants.MINUS_SIGN) {
711      i++;
712      isNegative = true;
713    }
714
715    while (i != numberAsByteArray.length) {
716      if (numberAsByteArray[i] < ParseConstants.ZERO ||
717          numberAsByteArray[i] > ParseConstants.NINE) {
718        throw new IllegalArgumentException("Byte Array should only contain digits");
719      }
720      result = result*10 + (numberAsByteArray[i] - ParseConstants.ZERO);
721      if (result < 0) {
722        throw new IllegalArgumentException("Long Argument too large");
723      }
724      i++;
725    }
726
727    if (isNegative) {
728      return -result;
729    } else {
730      return result;
731    }
732  }

734/**
735 * Converts a boolean expressed in a byte array to an actual boolean
736 *<p>
737 * This doesn't used Bytes.toBoolean because Bytes.toBoolean(byte [])
738 * assumes that 1 stands for true and 0 for false.
739 * Here, the byte array representing "true" and "false" is parsed
740 * <p>
741 * @param booleanAsByteArray the boolean value expressed as a byte array
742 * @return the boolean value
743 */
744  public static boolean convertByteArrayToBoolean (byte [] booleanAsByteArray) {
745    if (booleanAsByteArray == null) {
746      throw new IllegalArgumentException("convertByteArrayToBoolean called with a null array");
747    }
748
749    if (booleanAsByteArray.length == 4 &&
750        (booleanAsByteArray[0] == 't' || booleanAsByteArray[0] == 'T') &&
751        (booleanAsByteArray[1] == 'r' || booleanAsByteArray[1] == 'R') &&
752        (booleanAsByteArray[2] == 'u' || booleanAsByteArray[2] == 'U') &&
753        (booleanAsByteArray[3] == 'e' || booleanAsByteArray[3] == 'E')) {
754      return true;
755    }
756    else if (booleanAsByteArray.length == 5 &&
757             (booleanAsByteArray[0] == 'f' || booleanAsByteArray[0] == 'F') &&
758             (booleanAsByteArray[1] == 'a' || booleanAsByteArray[1] == 'A') &&
759             (booleanAsByteArray[2] == 'l' || booleanAsByteArray[2] == 'L') &&
760             (booleanAsByteArray[3] == 's' || booleanAsByteArray[3] == 'S') &&
761             (booleanAsByteArray[4] == 'e' || booleanAsByteArray[4] == 'E')) {
762      return false;
763    }
764    else {
765      throw new IllegalArgumentException("Incorrect Boolean Expression");
766    }
767  }

821/**
822 * Parses a comparator of the form comparatorType:comparatorValue form and returns a comparator
823 * <p>
824 * @param comparator the comparator in the form comparatorType:comparatorValue
825 * @return the parsed comparator
826 */
827  public static ByteArrayComparable createComparator (byte [] comparator) {
828    if (comparator == null)
829      throw new IllegalArgumentException("Incorrect Comparator");
830    byte [][] parsedComparator = ParseFilter.parseComparator(comparator);
831    byte [] comparatorType = parsedComparator[0];
832    byte [] comparatorValue = parsedComparator[1];
833
834
835    if (Bytes.equals(comparatorType, ParseConstants.binaryType))
836      return new BinaryComparator(comparatorValue);
837    else if (Bytes.equals(comparatorType, ParseConstants.binaryPrefixType))
838      return new BinaryPrefixComparator(comparatorValue);
839    else if (Bytes.equals(comparatorType, ParseConstants.regexStringType))
840      return new RegexStringComparator(new String(comparatorValue));
841    else if (Bytes.equals(comparatorType, ParseConstants.substringType))
842      return new SubstringComparator(new String(comparatorValue));
843    else
844      throw new IllegalArgumentException("Incorrect comparatorType");
845  }

847/**
848 * Splits a column in comparatorType:comparatorValue form into separate byte arrays
849 * <p>
850 * @param comparator the comparator
851 * @return the parsed arguments of the comparator as a 2D byte array
852 */
853  public static byte [][] parseComparator (byte [] comparator) {
854    final int index = Bytes.searchDelimiterIndex(comparator, 0, comparator.length,
855        ParseConstants.COLON);
856    if (index == -1) {
857      throw new IllegalArgumentException("Incorrect comparator");
858    }
859
860    byte [][] result = new byte [2][0];
861    result[0] = new byte [index];
862    System.arraycopy(comparator, 0, result[0], 0, index);
863
864    final int len = comparator.length - (index + 1);
865    result[1] = new byte[len];
866    System.arraycopy(comparator, index + 1, result[1], 0, len);
867
868    return result;
869  }
