036/**
037 * The base interface for asynchronous version of Table. Obtain an instance from a
038 * {@link AsyncConnection}.
039 * <p>
040 * The implementation is required to be thread safe.
041 * <p>
042 * Usually the implementation will not throw any exception directly. You need to get the exception
043 * from the returned {@link CompletableFuture}.
044 * @since 2.0.0
045 */
046@InterfaceAudience.Public
047public interface AsyncTableBase {
048
049  /**
050   * Gets the fully qualified table name instance of this table.
051   */
052  TableName getName();
053
054  /**
055   * Returns the {@link org.apache.hadoop.conf.Configuration} object used by this instance.
056   * <p>
057   * The reference returned is not a copy, so any change made to it will affect this instance.
058   */
059  Configuration getConfiguration();
060
061  /**
062   * Get timeout of each rpc request in this Table instance. It will be overridden by a more
063   * specific rpc timeout config such as readRpcTimeout or writeRpcTimeout.
064   * @see #getReadRpcTimeout(TimeUnit)
065   * @see #getWriteRpcTimeout(TimeUnit)
066   */
067  long getRpcTimeout(TimeUnit unit);
068
069  /**
070   * Get timeout of each rpc read request in this Table instance.
071   */
072  long getReadRpcTimeout(TimeUnit unit);
073
074  /**
075   * Get timeout of each rpc write request in this Table instance.
076   */
077  long getWriteRpcTimeout(TimeUnit unit);
078
079  /**
080   * Get timeout of each operation in Table instance.
081   */
082  long getOperationTimeout(TimeUnit unit);
083
084  /**
085   * Get the timeout of a single operation in a scan. It works like operation timeout for other
086   * operations.
087   */
088  long getScanTimeout(TimeUnit unit);
089
090  /**
091   * Test for the existence of columns in the table, as specified by the Get.
092   * <p>
093   * This will return true if the Get matches one or more keys, false if not.
094   * <p>
095   * This is a server-side call so it prevents any data from being transfered to the client.
096   * @return true if the specified Get matches one or more keys, false if not. The return value will
097   *         be wrapped by a {@link CompletableFuture}.
098   */
099  default CompletableFuture<Boolean> exists(Get get) {
100    return get(toCheckExistenceOnly(get)).thenApply(r -> r.getExists());
101  }
102
103  /**
104   * Extracts certain cells from a given row.
105   * @param get The object that specifies what data to fetch and from which row.
106   * @return The data coming from the specified row, if it exists. If the row specified doesn't
107   *         exist, the {@link Result} instance returned won't contain any
108   *         {@link org.apache.hadoop.hbase.KeyValue}, as indicated by {@link Result#isEmpty()}. The
109   *         return value will be wrapped by a {@link CompletableFuture}.
110   */
111  CompletableFuture<Result> get(Get get);
112
113  /**
114   * Puts some data to the table.
115   * @param put The data to put.
116   * @return A {@link CompletableFuture} that always returns null when complete normally.
117   */
118  CompletableFuture<Void> put(Put put);
119
120  /**
121   * Deletes the specified cells/row.
122   * @param delete The object that specifies what to delete.
123   * @return A {@link CompletableFuture} that always returns null when complete normally.
124   */
125  CompletableFuture<Void> delete(Delete delete);
126
127  /**
128   * Appends values to one or more columns within a single row.
129   * <p>
130   * This operation does not appear atomic to readers. Appends are done under a single row lock, so
131   * write operations to a row are synchronized, but readers do not take row locks so get and scan
132   * operations can see this operation partially completed.
133   * @param append object that specifies the columns and amounts to be used for the increment
134   *          operations
135   * @return values of columns after the append operation (maybe null). The return value will be
136   *         wrapped by a {@link CompletableFuture}.
137   */
138  CompletableFuture<Result> append(Append append);
139
140  /**
141   * Increments one or more columns within a single row.
142   * <p>
143   * This operation does not appear atomic to readers. Increments are done under a single row lock,
144   * so write operations to a row are synchronized, but readers do not take row locks so get and
145   * scan operations can see this operation partially completed.
146   * @param increment object that specifies the columns and amounts to be used for the increment
147   *          operations
148   * @return values of columns after the increment. The return value will be wrapped by a
149   *         {@link CompletableFuture}.
150   */
151  CompletableFuture<Result> increment(Increment increment);
152
153  /**
154   * See {@link #incrementColumnValue(byte[], byte[], byte[], long, Durability)}
155   * <p>
156   * The {@link Durability} is defaulted to {@link Durability#SYNC_WAL}.
157   * @param row The row that contains the cell to increment.
158   * @param family The column family of the cell to increment.
159   * @param qualifier The column qualifier of the cell to increment.
160   * @param amount The amount to increment the cell with (or decrement, if the amount is negative).
161   * @return The new value, post increment. The return value will be wrapped by a
162   *         {@link CompletableFuture}.
163   */
164  default CompletableFuture<Long> incrementColumnValue(byte[] row, byte[] family, byte[] qualifier,
165      long amount) {
166    return incrementColumnValue(row, family, qualifier, amount, Durability.SYNC_WAL);
167  }
168
169  /**
170   * Atomically increments a column value. If the column value already exists and is not a
171   * big-endian long, this could throw an exception. If the column value does not yet exist it is
172   * initialized to <code>amount</code> and written to the specified column.
173   * <p>
174   * Setting durability to {@link Durability#SKIP_WAL} means that in a fail scenario you will lose
175   * any increments that have not been flushed.
176   * @param row The row that contains the cell to increment.
177   * @param family The column family of the cell to increment.
178   * @param qualifier The column qualifier of the cell to increment.
179   * @param amount The amount to increment the cell with (or decrement, if the amount is negative).
180   * @param durability The persistence guarantee for this increment.
181   * @return The new value, post increment. The return value will be wrapped by a
182   *         {@link CompletableFuture}.
183   */
184  default CompletableFuture<Long> incrementColumnValue(byte[] row, byte[] family, byte[] qualifier,
185      long amount, Durability durability) {
186    Preconditions.checkNotNull(row, "row is null");
187    Preconditions.checkNotNull(family, "family is null");
188    return increment(
189      new Increment(row).addColumn(family, qualifier, amount).setDurability(durability))
190          .thenApply(r -> Bytes.toLong(r.getValue(family, qualifier)));
191  }
192
193  /**
194   * Atomically checks if a row/family/qualifier value equals to the expected value. If it does, it
195   * adds the put. If the passed value is null, the check is for the lack of column (ie:
196   * non-existence)
197   * @param row to check
198   * @param family column family to check
199   * @param qualifier column qualifier to check
200   * @param value the expected value
201   * @param put data to put if check succeeds
202   * @return true if the new put was executed, false otherwise. The return value will be wrapped by
203   *         a {@link CompletableFuture}.
204   */
205  default CompletableFuture<Boolean> checkAndPut(byte[] row, byte[] family, byte[] qualifier,
206      byte[] value, Put put) {
207    return checkAndPut(row, family, qualifier, CompareOperator.EQUAL, value, put);
208  }
209
210  /**
211   * Atomically checks if a row/family/qualifier value matches the expected value. If it does, it
212   * adds the put. If the passed value is null, the check is for the lack of column (ie:
213   * non-existence)
214   * @param row to check
215   * @param family column family to check
216   * @param qualifier column qualifier to check
217   * @param compareOp comparison operator to use
218   * @param value the expected value
219   * @param put data to put if check succeeds
220   * @return true if the new put was executed, false otherwise. The return value will be wrapped by
221   *         a {@link CompletableFuture}.
222   */
223  CompletableFuture<Boolean> checkAndPut(byte[] row, byte[] family, byte[] qualifier,
224                                         CompareOperator compareOp, byte[] value, Put put);
225
226  /**
227   * Atomically checks if a row/family/qualifier value equals to the expected value. If it does, it
228   * adds the delete. If the passed value is null, the check is for the lack of column (ie:
229   * non-existence)
230   * @param row to check
231   * @param family column family to check
232   * @param qualifier column qualifier to check
233   * @param value the expected value
234   * @param delete data to delete if check succeeds
235   * @return true if the new delete was executed, false otherwise. The return value will be wrapped
236   *         by a {@link CompletableFuture}.
237   */
238  default CompletableFuture<Boolean> checkAndDelete(byte[] row, byte[] family, byte[] qualifier,
239      byte[] value, Delete delete) {
240    return checkAndDelete(row, family, qualifier, CompareOperator.EQUAL, value, delete);
241  }
242
243  /**
244   * Atomically checks if a row/family/qualifier value matches the expected value. If it does, it
245   * adds the delete. If the passed value is null, the check is for the lack of column (ie:
246   * non-existence)
247   * @param row to check
248   * @param family column family to check
249   * @param qualifier column qualifier to check
250   * @param compareOp comparison operator to use
251   * @param value the expected value
252   * @param delete data to delete if check succeeds
253   * @return true if the new delete was executed, false otherwise. The return value will be wrapped
254   *         by a {@link CompletableFuture}.
255   */
256  CompletableFuture<Boolean> checkAndDelete(byte[] row, byte[] family, byte[] qualifier,
257                                            CompareOperator compareOp, byte[] value, Delete delete);
258
259  /**
260   * Performs multiple mutations atomically on a single row. Currently {@link Put} and
261   * {@link Delete} are supported.
262   * @param mutation object that specifies the set of mutations to perform atomically
263   * @return A {@link CompletableFuture} that always returns null when complete normally.
264   */
265  CompletableFuture<Void> mutateRow(RowMutations mutation);
266
267  /**
268   * Atomically checks if a row/family/qualifier value equals to the expected value. If it does, it
269   * performs the row mutations. If the passed value is null, the check is for the lack of column
270   * (ie: non-existence)
271   * @param row to check
272   * @param family column family to check
273   * @param qualifier column qualifier to check
274   * @param value the expected value
275   * @param mutation mutations to perform if check succeeds
276   * @return true if the new put was executed, false otherwise. The return value will be wrapped by
277   *         a {@link CompletableFuture}.
278   */
279  default CompletableFuture<Boolean> checkAndMutate(byte[] row, byte[] family, byte[] qualifier,
280      byte[] value, RowMutations mutation) {
281    return checkAndMutate(row, family, qualifier, CompareOperator.EQUAL, value, mutation);
282  }
283
284  /**
285   * Atomically checks if a row/family/qualifier value matches the expected value. If it does, it
286   * performs the row mutations. If the passed value is null, the check is for the lack of column
287   * (ie: non-existence)
288   * @param row to check
289   * @param family column family to check
290   * @param qualifier column qualifier to check
291   * @param compareOp the comparison operator
292   * @param value the expected value
293   * @param mutation mutations to perform if check succeeds
294   * @return true if the new put was executed, false otherwise. The return value will be wrapped by
295   *         a {@link CompletableFuture}.
296   */
297  CompletableFuture<Boolean> checkAndMutate(byte[] row, byte[] family, byte[] qualifier,
298                                            CompareOperator compareOp, byte[] value, RowMutations mutation);

300  /**
301   * Return all the results that match the given scan object.
302   * <p>
303   * Notice that usually you should use this method with a {@link Scan} object that has limit set.
304   * For example, if you want to get the closest row after a given row, you could do this:
305   * <p>
306   *
307   * <pre>
308   * <code>
309   * table.scanAll(new Scan().withStartRow(row, false).setLimit(1)).thenAccept(results -> {
310   *   if (results.isEmpty()) {
311   *      System.out.println("No row after " + Bytes.toStringBinary(row));
312   *   } else {
313   *     System.out.println("The closest row after " + Bytes.toStringBinary(row) + " is "
314   *         + Bytes.toStringBinary(results.stream().findFirst().get().getRow()));
315   *   }
316   * });
317   * </code>
318   * </pre>
319   * <p>
320   * If your result set is very large, you should use other scan method to get a scanner or use
321   * callback to process the results. They will do chunking to prevent OOM. The scanAll method will
322   * fetch all the results and store them in a List and then return the list to you.
323   * <p>
324   * The scan metrics will be collected background if you enable it but you have no way to get it.
325   * Usually you can get scan metrics from {@code ResultScanner}, or through
326   * {@code ScanResultConsumer.onScanMetricsCreated} but this method only returns a list of results.
327   * So if you really care about scan metrics then you'd better use other scan methods which return
328   * a {@code ResultScanner} or let you pass in a {@code ScanResultConsumer}. There is no
329   * performance difference between these scan methods so do not worry.
330   * @param scan A configured {@link Scan} object. So if you use this method to fetch a really large
331   *          result set, it is likely to cause OOM.
332   * @return The results of this small scan operation. The return value will be wrapped by a
333   *         {@link CompletableFuture}.
334   */
335  CompletableFuture<List<Result>> scanAll(Scan scan);
336
337  /**
338   * Test for the existence of columns in the table, as specified by the Gets.
339   * <p>
340   * This will return a list of booleans. Each value will be true if the related Get matches one or
341   * more keys, false if not.
342   * <p>
343   * This is a server-side call so it prevents any data from being transferred to the client.
344   * @param gets the Gets
345   * @return A list of {@link CompletableFuture}s that represent the existence for each get.
346   */
347  default List<CompletableFuture<Boolean>> exists(List<Get> gets) {
348    return get(toCheckExistenceOnly(gets)).stream()
349        .<CompletableFuture<Boolean>> map(f -> f.thenApply(r -> r.getExists())).collect(toList());
350  }
351
352  /**
353   * A simple version for batch exists. It will fail if there are any failures and you will get the
354   * whole result boolean list at once if the operation is succeeded.
355   * @param gets the Gets
356   * @return A {@link CompletableFuture} that wrapper the result boolean list.
357   */
358  default CompletableFuture<List<Boolean>> existsAll(List<Get> gets) {
359    return allOf(exists(gets));
360  }

362  /**
363   * Extracts certain cells from the given rows, in batch.
364   * <p>
365   * Notice that you may not get all the results with this function, which means some of the
366   * returned {@link CompletableFuture}s may succeed while some of the other returned
367   * {@link CompletableFuture}s may fail.
368   * @param gets The objects that specify what data to fetch and from which rows.
369   * @return A list of {@link CompletableFuture}s that represent the result for each get.
370   */
371  List<CompletableFuture<Result>> get(List<Get> gets);
372
373  /**
374   * A simple version for batch get. It will fail if there are any failures and you will get the
375   * whole result list at once if the operation is succeeded.
376   * @param gets The objects that specify what data to fetch and from which rows.
377   * @return A {@link CompletableFuture} that wrapper the result list.
378   */
379  default CompletableFuture<List<Result>> getAll(List<Get> gets) {
380    return allOf(get(gets));
381  }
382
383  /**
384   * Puts some data in the table, in batch.
385   * @param puts The list of mutations to apply.
386   * @return A list of {@link CompletableFuture}s that represent the result for each put.
387   */
388  List<CompletableFuture<Void>> put(List<Put> puts);
389
390  /**
391   * A simple version of batch put. It will fail if there are any failures.
392   * @param puts The list of mutations to apply.
393   * @return A {@link CompletableFuture} that always returns null when complete normally.
394   */
395  default CompletableFuture<Void> putAll(List<Put> puts) {
396    return allOf(put(puts)).thenApply(r -> null);
397  }
398
399  /**
400   * Deletes the specified cells/rows in bulk.
401   * @param deletes list of things to delete.
402   * @return A list of {@link CompletableFuture}s that represent the result for each delete.
403   */
404  List<CompletableFuture<Void>> delete(List<Delete> deletes);
405
406  /**
407   * A simple version of batch delete. It will fail if there are any failures.
408   * @param deletes list of things to delete.
409   * @return A {@link CompletableFuture} that always returns null when complete normally.
410   */
411  default CompletableFuture<Void> deleteAll(List<Delete> deletes) {
412    return allOf(delete(deletes)).thenApply(r -> null);
413  }
414
415  /**
416   * Method that does a batch call on Deletes, Gets, Puts, Increments and Appends. The ordering of
417   * execution of the actions is not defined. Meaning if you do a Put and a Get in the same
418   * {@link #batch} call, you will not necessarily be guaranteed that the Get returns what the Put
419   * had put.
420   * @param actions list of Get, Put, Delete, Increment, Append objects
421   * @return A list of {@link CompletableFuture}s that represent the result for each action.
422   */
423  <T> List<CompletableFuture<T>> batch(List<? extends Row> actions);
424
425  /**
426   * A simple version of batch. It will fail if there are any failures and you will get the whole
427   * result list at once if the operation is succeeded.
428   * @param actions list of Get, Put, Delete, Increment, Append objects
429   * @return A list of the result for the actions. Wrapped by a {@link CompletableFuture}.
430   */
431  default <T> CompletableFuture<List<T>> batchAll(List<? extends Row> actions) {
432    return allOf(batch(actions));
433  }
