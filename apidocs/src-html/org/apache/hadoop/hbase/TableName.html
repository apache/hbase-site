030/**
031 * Immutable POJO class for representing a table name.
032 * Which is of the form:
033 * &lt;table namespace&gt;:&lt;table qualifier&gt;
034 *
035 * Two special namespaces:
036 *
037 * 1. hbase - system namespace, used to contain hbase internal tables
038 * 2. default - tables with no explicit specified namespace will
039 * automatically fall into this namespace.
040 *
041 * ie
042 *
043 * a) foo:bar, means namespace=foo and qualifier=bar
044 * b) bar, means namespace=default and qualifier=bar
045 * c) default:bar, means namespace=default and qualifier=bar
046 *
047 *  <p>
048 * Internally, in this class, we cache the instances to limit the number of objects and
049 *  make the "equals" faster. We try to minimize the number of objects created of
050 *  the number of array copy to check if we already have an instance of this TableName. The code
051 *  is not optimize for a new instance creation but is optimized to check for existence.
052 * </p>
053 */
054@InterfaceAudience.Public
055public final class TableName implements Comparable<TableName> {
056
057  /** See {@link #createTableNameIfNecessary(ByteBuffer, ByteBuffer)} */
058  private static final Set<TableName> tableCache = new CopyOnWriteArraySet<>();
059
060  /** Namespace delimiter */
061  //this should always be only 1 byte long
062  public final static char NAMESPACE_DELIM = ':';
063
064  // A non-capture group so that this can be embedded.
065  // regex is a bit more complicated to support nuance of tables
066  // in default namespace
067  //Allows only letters, digits and '_'
068  public static final String VALID_NAMESPACE_REGEX =
069      "(?:[a-zA-Z_0-9]+)";
070  //Allows only letters, digits, '_', '-' and '.'
071  public static final String VALID_TABLE_QUALIFIER_REGEX =
072      "(?:[a-zA-Z_0-9][a-zA-Z_0-9-.]*)";
073  //Concatenation of NAMESPACE_REGEX and TABLE_QUALIFIER_REGEX,
074  //with NAMESPACE_DELIM as delimiter
075  public static final String VALID_USER_TABLE_REGEX =
076      "(?:(?:(?:"+VALID_NAMESPACE_REGEX+"\\"+NAMESPACE_DELIM+")?)" +
077         "(?:"+VALID_TABLE_QUALIFIER_REGEX+"))";

116  /**
117   * Check passed byte array, "tableName", is legal user-space table name.
118   * @return Returns passed <code>tableName</code> param
119   * @throws IllegalArgumentException if passed a tableName is null or
120   * is made of other than 'word' characters or underscores: i.e.
121   * <code>[a-zA-Z_0-9.-:]</code>. The ':' is used to delimit the namespace
122   * from the table name and can be used for nothing else.
123   *
124   * Namespace names can only contain 'word' characters
125   * <code>[a-zA-Z_0-9]</code> or '_'
126   *
127   * Qualifier names can only contain 'word' characters
128   * <code>[a-zA-Z_0-9]</code> or '_', '.' or '-'.
129   * The name may not start with '.' or '-'.
130   *
131   * Valid fully qualified table names:
132   * foo:bar, namespace=&gt;foo, table=&gt;bar
133   * org:foo.bar, namespace=org, table=&gt;foo.bar
134   */
135  public static byte [] isLegalFullyQualifiedTableName(final byte[] tableName) {
136    if (tableName == null || tableName.length <= 0) {
137      throw new IllegalArgumentException("Name is null or empty");
138    }
139
140    int namespaceDelimIndex = com.google.common.primitives.Bytes.lastIndexOf(tableName,
141        (byte) NAMESPACE_DELIM);
142    if (namespaceDelimIndex < 0){
143      isLegalTableQualifierName(tableName);
144    } else {
145      isLegalNamespaceName(tableName, 0, namespaceDelimIndex);
146      isLegalTableQualifierName(tableName, namespaceDelimIndex + 1, tableName.length);
147    }
148    return tableName;
149  }
150
151  public static byte [] isLegalTableQualifierName(final byte[] qualifierName) {
152    isLegalTableQualifierName(qualifierName, 0, qualifierName.length, false);
153    return qualifierName;
154  }

162  /**
163   * Qualifier names can only contain 'word' characters
164   * <code>[a-zA-Z_0-9]</code> or '_', '.' or '-'.
165   * The name may not start with '.' or '-'.
166   *
167   * @param qualifierName byte array containing the qualifier name
168   * @param start start index
169   * @param end end index (exclusive)
170   */
171  public static void isLegalTableQualifierName(final byte[] qualifierName,
172                                                int start,
173                                                int end) {
174      isLegalTableQualifierName(qualifierName, start, end, false);
175  }
176
177  public static void isLegalTableQualifierName(final byte[] qualifierName,
178                                                int start,
179                                                int end,
180                                                boolean isSnapshot) {
181    if(end - start < 1) {
182      throw new IllegalArgumentException(isSnapshot ? "Snapshot" : "Table" + " qualifier must not be empty");
183    }
184
185    if (qualifierName[start] == '.' || qualifierName[start] == '-') {
186      throw new IllegalArgumentException("Illegal first character <" + qualifierName[start] +
187                                         "> at 0. " + (isSnapshot ? "Snapshot" : "User-space table") +
188                                         " qualifiers can only start with 'alphanumeric " +
189                                         "characters': i.e. [a-zA-Z_0-9]: " +
190                                         Bytes.toString(qualifierName, start, end));
191    }
192    for (int i = start; i < end; i++) {
193      if (Character.isLetterOrDigit(qualifierName[i]) ||
194          qualifierName[i] == '_' ||
195          qualifierName[i] == '-' ||
196          qualifierName[i] == '.') {
197        continue;
198      }
199      throw new IllegalArgumentException("Illegal character code:" + qualifierName[i] +
200                                         ", <" + (char) qualifierName[i] + "> at " + i +
201                                         ". " + (isSnapshot ? "Snapshot" : "User-space table") +
202                                         " qualifiers can only contain " +
203                                         "'alphanumeric characters': i.e. [a-zA-Z_0-9-.]: " +
204                                         Bytes.toString(qualifierName, start, end));
205    }
206  }
207  public static void isLegalNamespaceName(byte[] namespaceName) {
208    isLegalNamespaceName(namespaceName, 0, namespaceName.length);
209  }
210
211  /**
212   * Valid namespace characters are [a-zA-Z_0-9]
213   */
214  public static void isLegalNamespaceName(final byte[] namespaceName,
215                                           final int start,
216                                           final int end) {
217    if(end - start < 1) {
218      throw new IllegalArgumentException("Namespace name must not be empty");
219    }
220    for (int i = start; i < end; i++) {
221      if (Character.isLetterOrDigit(namespaceName[i])|| namespaceName[i] == '_') {
222        continue;
223      }
224      throw new IllegalArgumentException("Illegal character <" + namespaceName[i] +
225        "> at " + i + ". Namespaces can only contain " +
226        "'alphanumeric characters': i.e. [a-zA-Z_0-9]: " + Bytes.toString(namespaceName,
227          start, end));
228    }
229  }

281  /**
282   *
283   * @throws IllegalArgumentException See {@link #valueOf(byte[])}
284   */
285  private TableName(ByteBuffer namespace, ByteBuffer qualifier) throws IllegalArgumentException {
286    this.qualifier = new byte[qualifier.remaining()];
287    qualifier.duplicate().get(this.qualifier);
288    this.qualifierAsString = Bytes.toString(this.qualifier);
289
290    if (qualifierAsString.equals(OLD_ROOT_STR)) {
291      throw new IllegalArgumentException(OLD_ROOT_STR + " has been deprecated.");
292    }
293    if (qualifierAsString.equals(OLD_META_STR)) {
294      throw new IllegalArgumentException(OLD_META_STR + " no longer exists. The table has been " +
295          "renamed to " + META_TABLE_NAME);
296    }
297
298    if (Bytes.equals(NamespaceDescriptor.DEFAULT_NAMESPACE_NAME, namespace)) {
299      // Using the same objects: this will make the comparison faster later
300      this.namespace = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME;
301      this.namespaceAsString = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR;
302      this.systemTable = false;
303
304      // The name does not include the namespace when it's the default one.
305      this.nameAsString = qualifierAsString;
306      this.name = this.qualifier;
307    } else {
308      if (Bytes.equals(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME, namespace)) {
309        this.namespace = NamespaceDescriptor.SYSTEM_NAMESPACE_NAME;
310        this.namespaceAsString = NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR;
311        this.systemTable = true;
312      } else {
313        this.namespace = new byte[namespace.remaining()];
314        namespace.duplicate().get(this.namespace);
315        this.namespaceAsString = Bytes.toString(this.namespace);
316        this.systemTable = false;
317      }
318      this.nameAsString = namespaceAsString + NAMESPACE_DELIM + qualifierAsString;
319      this.name = Bytes.toBytes(nameAsString);
320    }
321
322    this.hashCode = nameAsString.hashCode();
323
324    isLegalNamespaceName(this.namespace);
325    isLegalTableQualifierName(this.qualifier);
326  }
327
328  /**
329   * This is only for the old and meta tables.
330   */
331  private TableName(String qualifier) {
332    this.qualifier = Bytes.toBytes(qualifier);
333    this.qualifierAsString = qualifier;
334
335    this.namespace = NamespaceDescriptor.SYSTEM_NAMESPACE_NAME;
336    this.namespaceAsString = NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR;
337    this.systemTable = true;
338
339    // WARNING: nameAsString is different than name for old meta & root!
340    // This is by design.
341    this.nameAsString = namespaceAsString + NAMESPACE_DELIM + qualifierAsString;
342    this.name = this.qualifier;
343
344    this.hashCode = nameAsString.hashCode();
345  }

348  /**
349   * Check that the object does not exist already. There are two reasons for creating the objects
350   * only once:
351   * 1) With 100K regions, the table names take ~20MB.
352   * 2) Equals becomes much faster as it's resolved with a reference and an int comparison.
353   */
354  private static TableName createTableNameIfNecessary(ByteBuffer bns, ByteBuffer qns) {
355    for (TableName tn : tableCache) {
356      if (Bytes.equals(tn.getQualifier(), qns) && Bytes.equals(tn.getNamespace(), bns)) {
357        return tn;
358      }
359    }
360
361    TableName newTable = new TableName(bns, qns);
362    if (tableCache.add(newTable)) {  // Adds the specified element if it is not already present
363      return newTable;
364    }
365
366    // Someone else added it. Let's find it.
367    for (TableName tn : tableCache) {
368      if (Bytes.equals(tn.getQualifier(), qns) && Bytes.equals(tn.getNamespace(), bns)) {
369        return tn;
370      }
371    }
372    // this should never happen.
373    throw new IllegalStateException(newTable + " was supposed to be in the cache");
374  }

387  public static TableName valueOf(String namespaceAsString, String qualifierAsString) {
388    if (namespaceAsString == null || namespaceAsString.length() < 1) {
389      namespaceAsString = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR;
390    }
391
392    for (TableName tn : tableCache) {
393      if (qualifierAsString.equals(tn.getQualifierAsString()) &&
394          namespaceAsString.equals(tn.getNameAsString())) {
395        return tn;
396      }
397    }
398
399    return createTableNameIfNecessary(
400        ByteBuffer.wrap(Bytes.toBytes(namespaceAsString)),
401        ByteBuffer.wrap(Bytes.toBytes(qualifierAsString)));
402  }

405  /**
406   * @throws IllegalArgumentException if fullName equals old root or old meta. Some code
407   *  depends on this. The test is buried in the table creation to save on array comparison
408   *  when we're creating a standard table object that will be in the cache.
409   */
410  public static TableName valueOf(byte[] fullName) throws IllegalArgumentException{
411    for (TableName tn : tableCache) {
412      if (Arrays.equals(tn.getName(), fullName)) {
413        return tn;
414      }
415    }
416
417    int namespaceDelimIndex = com.google.common.primitives.Bytes.lastIndexOf(fullName,
418        (byte) NAMESPACE_DELIM);
419
420    if (namespaceDelimIndex < 0) {
421      return createTableNameIfNecessary(
422          ByteBuffer.wrap(NamespaceDescriptor.DEFAULT_NAMESPACE_NAME),
423          ByteBuffer.wrap(fullName));
424    } else {
425      return createTableNameIfNecessary(
426          ByteBuffer.wrap(fullName, 0, namespaceDelimIndex),
427          ByteBuffer.wrap(fullName, namespaceDelimIndex + 1,
428              fullName.length - (namespaceDelimIndex + 1)));
429    }
430  }

433  /**
434   * @throws IllegalArgumentException if fullName equals old root or old meta. Some code
435   *  depends on this.
436   */
437  public static TableName valueOf(String name) {
438    for (TableName tn : tableCache) {
439      if (name.equals(tn.getNameAsString())) {
440        return tn;
441      }
442    }
443
444    int namespaceDelimIndex = name.indexOf(NAMESPACE_DELIM);
445    byte[] nameB = Bytes.toBytes(name);
446
447    if (namespaceDelimIndex < 0) {
448      return createTableNameIfNecessary(
449          ByteBuffer.wrap(NamespaceDescriptor.DEFAULT_NAMESPACE_NAME),
450          ByteBuffer.wrap(nameB));
451    } else {
452      return createTableNameIfNecessary(
453          ByteBuffer.wrap(nameB, 0, namespaceDelimIndex),
454          ByteBuffer.wrap(nameB, namespaceDelimIndex + 1,
455              nameB.length - (namespaceDelimIndex + 1)));
456    }
457  }

460  public static TableName valueOf(byte[] namespace, byte[] qualifier) {
461    if (namespace == null || namespace.length < 1) {
462      namespace = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME;
463    }
464
465    for (TableName tn : tableCache) {
466      if (Arrays.equals(tn.getQualifier(), qualifier) &&
467          Arrays.equals(tn.getNamespace(), namespace)) {
468        return tn;
469      }
470    }
471
472    return createTableNameIfNecessary(
473        ByteBuffer.wrap(namespace), ByteBuffer.wrap(qualifier));
474  }
475
476  public static TableName valueOf(ByteBuffer namespace, ByteBuffer qualifier) {
477    if (namespace == null || namespace.remaining() < 1) {
478      return createTableNameIfNecessary(
479          ByteBuffer.wrap(NamespaceDescriptor.DEFAULT_NAMESPACE_NAME), qualifier);
480    }
481
482    return createTableNameIfNecessary(namespace, qualifier);
483  }

485  @Override
486  public boolean equals(Object o) {
487    if (this == o) return true;
488    if (o == null || getClass() != o.getClass()) return false;
489
490    TableName tableName = (TableName) o;
491
492    return o.hashCode() == hashCode && nameAsString.equals(tableName.nameAsString);
493  }

495  @Override

497    return hashCode;
498  }

500  /**
501   * For performance reasons, the ordering is not lexicographic.
502   */
503  @Override
504  public int compareTo(TableName tableName) {
505    if (this == tableName) return 0;
506    if (this.hashCode < tableName.hashCode()) {
507      return -1;
508    }
509    if (this.hashCode > tableName.hashCode()) {
510      return 1;
511    }
512    return this.nameAsString.compareTo(tableName.getNameAsString());
513  }
