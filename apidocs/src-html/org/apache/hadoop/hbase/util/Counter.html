018package org.apache.hadoop.hbase.util;
019
020import java.util.concurrent.atomic.AtomicBoolean;
021import java.util.concurrent.atomic.AtomicLongFieldUpdater;
022import java.util.concurrent.atomic.AtomicReference;
023
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025import org.apache.hadoop.hbase.classification.InterfaceStability;
026
027/**
028 * High scalable counter. Thread safe.
029 * @deprecated use {@link java.util.concurrent.atomic.LongAdder} instead.
030 */
031@InterfaceAudience.Public
032@InterfaceStability.Evolving
033@Deprecated
034public class Counter {
035  private static final int MAX_CELLS_LENGTH = 1 << 20;
036
037  private static class Cell {
038    // Pads are added around the value to avoid cache-line contention with
039    // another cell's value. The cache-line size is expected to be equal to or
040    // less than about 128 Bytes (= 64 Bits * 16).
041
042    @SuppressWarnings("unused")
043    volatile long p0, p1, p2, p3, p4, p5, p6;
044    volatile long value;
045    @SuppressWarnings("unused")
046    volatile long q0, q1, q2, q3, q4, q5, q6;
047
048    static final AtomicLongFieldUpdater<Cell> valueUpdater =
049        AtomicLongFieldUpdater.newUpdater(Cell.class, "value");
050
051    Cell() {}
052
053    Cell(long initValue) {
054      value = initValue;
055    }
056
057    long get() {
058      return value;
059    }
060
061    boolean add(long delta) {
062      long current = value;
063      return valueUpdater.compareAndSet(this, current, current + delta);
064    }
065  }
066
067  private static class Container {
068    /** The length should be a power of 2. */
069    final Cell[] cells;
070
071    /** True if a new extended container is going to replace this. */
072    final AtomicBoolean demoted = new AtomicBoolean();
073
074    Container(Cell cell) {
075      this(new Cell[] { cell });
076    }
077
078    /**
079     * @param cells the length should be a power of 2
080     */
081    Container(Cell[] cells) {
082      this.cells = cells;
083    }
084  }
085
086  private final AtomicReference<Container> containerRef;
087
088  public Counter() {
089    this(new Cell());
090  }
091
092  public Counter(long initValue) {
093    this(new Cell(initValue));
094  }
095
096  private Counter(Cell initCell) {
097    containerRef = new AtomicReference<Container>(new Container(initCell));
098  }
099
100  private static int hash() {
101    // The logic is borrowed from high-scale-lib's ConcurrentAutoTable.
102
103    int h = System.identityHashCode(Thread.currentThread());
104    // You would think that System.identityHashCode on the current thread
105    // would be a good hash fcn, but actually on SunOS 5.8 it is pretty lousy
106    // in the low bits.
107
108    h ^= (h >>> 20) ^ (h >>> 12); // Bit spreader, borrowed from Doug Lea
109    h ^= (h >>>  7) ^ (h >>>  4);
110    return h;
111  }
112
113  public void add(long delta) {
114    Container container = containerRef.get();
115    Cell[] cells = container.cells;
116    int mask = cells.length - 1;
117
118    int baseIndex = hash();
119    if(cells[baseIndex & mask].add(delta)) {
120      return;
121    }
122
123    int index = baseIndex + 1;
124    while(true) {
125      if(cells[index & mask].add(delta)) {
126        break;
127      }
128      index++;
129    }
130
131    if(index - baseIndex >= cells.length &&
132        cells.length < MAX_CELLS_LENGTH &&
133        container.demoted.compareAndSet(false, true)) {
134
135      if(containerRef.get() == container) {
136        Cell[] newCells = new Cell[cells.length * 2];
137        System.arraycopy(cells, 0, newCells, 0, cells.length);
138        for(int i = cells.length; i < newCells.length; i++) {
139          newCells[i] = new Cell();
140          // Fill all of the elements with instances. Creating a cell on demand
141          // and putting it into the array makes a concurrent problem about
142          // visibility or, in other words, happens-before relation, because
143          // each element of the array is not volatile so that you should
144          // establish the relation by some piggybacking.
145        }
146        containerRef.compareAndSet(container, new Container(newCells));
147      }
148    }
149  }
150
151  public void increment() {
152    add(1);
153  }
154
155  public void decrement() {
156    add(-1);
157  }
158
159  public void set(long value) {
160    containerRef.set(new Container(new Cell(value)));
161  }
162
163  public long get() {
164    long sum = 0;
165    for(Cell cell : containerRef.get().cells) {
166      sum += cell.get();
167    }
168    return sum;
169  }
170
171  @Override
172  public String toString() {
173    Cell[] cells = containerRef.get().cells;
174
175    long min = Long.MAX_VALUE;
176    long max = Long.MIN_VALUE;
177    long sum = 0;
178
179    for(Cell cell : cells) {
180      long value = cell.get();
181      sum += value;
182      if(min > value) { min = value; }
183      if(max < value) { max = value; }
184    }
185
186    return new StringBuilder(100)
187    .append("[value=").append(sum)
188    .append(", cells=[length=").append(cells.length)
189    .append(", min=").append(min)
190    .append(", max=").append(max)
191    .append("]]").toString();
192  }
193}
