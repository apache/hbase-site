018package org.apache.hadoop.hbase.io;
019
020import java.io.DataInput;
021import java.io.DataOutput;
022import java.io.IOException;
023import java.util.Arrays;
024import java.util.List;
025import org.apache.hadoop.io.BytesWritable;
026import org.apache.hadoop.io.WritableComparable;
027import org.apache.hadoop.io.WritableComparator;
028import org.apache.yetus.audience.InterfaceAudience;
029
030/**
031 * A byte sequence that is usable as a key or value. Based on
032 * {@link org.apache.hadoop.io.BytesWritable} only this class is NOT resizable and DOES NOT
033 * distinguish between the size of the sequence and the current capacity as
034 * {@link org.apache.hadoop.io.BytesWritable} does. Hence its comparatively 'immutable'. When
035 * creating a new instance of this class, the underlying byte [] is not copied, just referenced. The
036 * backing buffer is accessed when we go to serialize.
037 */
038@InterfaceAudience.Public
039@edu.umd.cs.findbugs.annotations.SuppressWarnings(
040    value = "EQ_CHECK_FOR_OPERAND_NOT_COMPATIBLE_WITH_THIS",
041    justification = "It has been like this forever")
042public class ImmutableBytesWritable implements WritableComparable<ImmutableBytesWritable> {
043  private byte[] bytes;
044  private int offset;
045  private int length;
046
047  /**
048   * Create a zero-size sequence.
049   */
050  public ImmutableBytesWritable() {
051    super();
052  }
053
054  /**
055   * Create a ImmutableBytesWritable using the byte array as the initial value.
056   * @param bytes This array becomes the backing storage for the object.
057   */
058  public ImmutableBytesWritable(byte[] bytes) {
059    this(bytes, 0, bytes.length);
060  }
061
062  /**
063   * Set the new ImmutableBytesWritable to the contents of the passed <code>ibw</code>.
064   * @param ibw the value to set this ImmutableBytesWritable to.
065   */
066  public ImmutableBytesWritable(final ImmutableBytesWritable ibw) {
067    this(ibw.get(), ibw.getOffset(), ibw.getLength());
068  }
069
070  /**
071   * Set the value to a given byte range
072   * @param bytes  the new byte range to set to
073   * @param offset the offset in newData to start at
074   * @param length the number of bytes in the range
075   */
076  public ImmutableBytesWritable(final byte[] bytes, final int offset, final int length) {
077    this.bytes = bytes;
078    this.offset = offset;
079    this.length = length;
080  }
081
082  /**
083   * Get the data from the BytesWritable.
084   * @return The data is only valid between offset and offset+length.
085   */
086  public byte[] get() {
087    if (this.bytes == null) {
088      throw new IllegalStateException(
089        "Uninitialiized. Null constructor " + "called w/o accompaying readFields invocation");
090    }
091    return this.bytes;
092  }
093
094  /**
095   * @param b Use passed bytes as backing array for this instance.
096   */
097  public void set(final byte[] b) {
098    set(b, 0, b.length);
099  }
100
101  /**
102   * @param b Use passed bytes as backing array for this instance. nn
103   */
104  public void set(final byte[] b, final int offset, final int length) {
105    this.bytes = b;
106    this.offset = offset;
107    this.length = length;
108  }
109
110  /** Returns the number of valid bytes in the buffer */
111  public int getLength() {
112    if (this.bytes == null) {
113      throw new IllegalStateException(
114        "Uninitialiized. Null constructor " + "called w/o accompaying readFields invocation");
115    }
116    return this.length;
117  }
118
119  /**
120   * n
121   */
122  public int getOffset() {
123    return this.offset;
124  }
125
126  @Override
127  public void readFields(final DataInput in) throws IOException {
128    this.length = in.readInt();
129    this.bytes = new byte[this.length];
130    in.readFully(this.bytes, 0, this.length);
131    this.offset = 0;
132  }
133
134  @Override
135  public void write(final DataOutput out) throws IOException {
136    out.writeInt(this.length);
137    out.write(this.bytes, this.offset, this.length);
138  }
139
140  // Below methods copied from BytesWritable
141  @Override
142  public int hashCode() {
143    int hash = 1;
144    for (int i = offset; i < offset + length; i++)
145      hash = (31 * hash) + (int) bytes[i];
146    return hash;
147  }
148
149  /**
150   * Define the sort order of the BytesWritable.
151   * @param that The other bytes writable
152   * @return Positive if left is bigger than right, 0 if they are equal, and negative if left is
153   *         smaller than right.
154   */
155  @Override
156  public int compareTo(ImmutableBytesWritable that) {
157    return WritableComparator.compareBytes(this.bytes, this.offset, this.length, that.bytes,
158      that.offset, that.length);
159  }
160
161  /**
162   * Compares the bytes in this object to the specified byte array n * @return Positive if left is
163   * bigger than right, 0 if they are equal, and negative if left is smaller than right.
164   */
165  public int compareTo(final byte[] that) {
166    return WritableComparator.compareBytes(this.bytes, this.offset, this.length, that, 0,
167      that.length);
168  }
169
170  /**
171   * @see java.lang.Object#equals(java.lang.Object)
172   */
173  @Override
174  public boolean equals(Object right_obj) {
175    if (right_obj instanceof byte[]) {
176      return compareTo((byte[]) right_obj) == 0;
177    }
178    if (right_obj instanceof ImmutableBytesWritable) {
179      return compareTo((ImmutableBytesWritable) right_obj) == 0;
180    }
181    return false;
182  }
183
184  /**
185   * @see java.lang.Object#toString()
186   */
187  @Override
188  public String toString() {
189    StringBuilder sb = new StringBuilder(3 * this.length);
190    final int endIdx = this.offset + this.length;
191    for (int idx = this.offset; idx < endIdx; idx++) {
192      sb.append(' ');
193      String num = Integer.toHexString(0xff & this.bytes[idx]);
194      // if it is only one digit, add a leading 0.
195      if (num.length() < 2) {
196        sb.append('0');
197      }
198      sb.append(num);
199    }
200    return sb.length() > 0 ? sb.substring(1) : "";
201  }
202
203  /**
204   * A Comparator optimized for ImmutableBytesWritable.
205   */
206  @InterfaceAudience.Public
207  public static class Comparator extends WritableComparator {
208    private BytesWritable.Comparator comparator = new BytesWritable.Comparator();
209
210    /** constructor */
211    public Comparator() {
212      super(ImmutableBytesWritable.class);
213    }
214
215    /**
216     * @see org.apache.hadoop.io.WritableComparator#compare(byte[], int, int, byte[], int, int)
217     */
218    @Override
219    public int compare(byte[] b1, int s1, int l1, byte[] b2, int s2, int l2) {
220      return comparator.compare(b1, s1, l1, b2, s2, l2);
221    }
222  }
223
224  static { // register this comparator
225    WritableComparator.define(ImmutableBytesWritable.class, new Comparator());
226  }
227
228  /**
229   * @param array List of byte [].
230   * @return Array of byte [].
231   */
232  public static byte[][] toArray(final List<byte[]> array) {
233    // List#toArray doesn't work on lists of byte [].
234    byte[][] results = new byte[array.size()][];
235    for (int i = 0; i < array.size(); i++) {
236      results[i] = array.get(i);
237    }
238    return results;
239  }
240
241  /**
242   * Returns a copy of the bytes referred to by this writable
243   */
244  public byte[] copyBytes() {
245    return Arrays.copyOfRange(bytes, offset, offset + length);
246  }
