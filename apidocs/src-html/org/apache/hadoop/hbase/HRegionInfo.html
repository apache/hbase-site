048/**
049 * Information about a region. A region is a range of keys in the whole keyspace of a table, an
050 * identifier (a timestamp) for differentiating between subset ranges (after region split)
051 * and a replicaId for differentiating the instance for the same range and some status information
052 * about the region.
053 *
054 * The region has a unique name which consists of the following fields:
055 * <ul>
056 * <li> tableName   : The name of the table </li>
057 * <li> startKey    : The startKey for the region. </li>
058 * <li> regionId    : A timestamp when the region is created. </li>
059 * <li> replicaId   : An id starting from 0 to differentiate replicas of the same region range
060 * but hosted in separated servers. The same region range can be hosted in multiple locations.</li>
061 * <li> encodedName : An MD5 encoded string for the region name.</li>
062 * </ul>
063 *
064 * <br> Other than the fields in the region name, region info contains:
065 * <ul>
066 * <li> endKey      : the endKey for the region (exclusive) </li>
067 * <li> split       : Whether the region is split </li>
068 * <li> offline     : Whether the region is offline </li>
069 * </ul>
070 *
071 * In 0.98 or before, a list of table's regions would fully cover the total keyspace, and at any
072 * point in time, a row key always belongs to a single region, which is hosted in a single server.
073 * In 0.99+, a region can have multiple instances (called replicas), and thus a range (or row) can
074 * correspond to multiple HRegionInfo's. These HRI's share the same fields however except the
075 * replicaId field. If the replicaId is not set, it defaults to 0, which is compatible with the
076 * previous behavior of a range corresponding to 1 region.
077 */
078@InterfaceAudience.Public
079public class HRegionInfo implements Comparable<HRegionInfo> {

083  /**
084   * The new format for a region name contains its encodedName at the end.
085   * The encoded name also serves as the directory name for the region
086   * in the filesystem.
087   *
088   * New region name format:
089   *    &lt;tablename>,,&lt;startkey>,&lt;regionIdTimestamp>.&lt;encodedName>.
090   * where,
091   *    &lt;encodedName> is a hex version of the MD5 hash of
092   *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
093   *
094   * The old region name format:
095   *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
096   * For region names in the old format, the encoded name is a 32-bit
097   * JenkinsHash integer value (in its decimal notation, string form).
098   *<p>
099   * **NOTE**
100   *
101   * The first hbase:meta region, and regions created by an older
102   * version of HBase (0.20 or prior) will continue to use the
103   * old region name format.
104   */
105

107   * in the new format. See description on new format above.
108   */

110  public  static final int MD5_HEX_LENGTH   = 32;

126  /**
127   * Does region name contain its encoded name?
128   * @param regionName region name
129   * @return boolean indicating if this a new format region
130   *         name which contains its encoded name.
131   */
132  private static boolean hasEncodedName(final byte[] regionName) {
133    // check if region name ends in ENC_SEPARATOR
134    if ((regionName.length >= 1)
135        && (regionName[regionName.length - 1] == ENC_SEPARATOR)) {
136      // region name is new format. it contains the encoded name.
137      return true;
138    }
139    return false;
140  }

142  /**
143   * @param regionName
144   * @return the encodedName
145   */
146  public static String encodeRegionName(final byte [] regionName) {
147    String encodedName;
148    if (hasEncodedName(regionName)) {
149      // region is in new format:
150      // <tableName>,<startKey>,<regionIdTimeStamp>/encodedName/
151      encodedName = Bytes.toString(regionName,
152          regionName.length - MD5_HEX_LENGTH - 1,
153          MD5_HEX_LENGTH);
154    } else {
155      // old format region name. First hbase:meta region also
156      // use this format.EncodedName is the JenkinsHash value.
157      HashKey<byte[]> key = new ByteArrayHashKey(regionName, 0, regionName.length);
158      int hashVal = Math.abs(JenkinsHash.getInstance().hash(key, 0));
159      encodedName = String.valueOf(hashVal);
160    }
161    return encodedName;
162  }

304  /**
305   * Construct HRegionInfo with explicit parameters
306   *
307   * @param tableName the table descriptor
308   * @param startKey first key in region
309   * @param endKey end of key range
310   * @param split true if this region has split and we have daughter regions
311   * regions that may or may not hold references to this region.
312   * @param regionid Region id to use.
313   * @param replicaId the replicaId to use
314   * @throws IllegalArgumentException
315   */
316  public HRegionInfo(final TableName tableName, final byte[] startKey,
317                     final byte[] endKey, final boolean split, final long regionid,
318                     final int replicaId)
319    throws IllegalArgumentException {
320    super();
321    if (tableName == null) {
322      throw new IllegalArgumentException("TableName cannot be null");
323    }
324    this.tableName = tableName;
325    this.offLine = false;
326    this.regionId = regionid;
327    this.replicaId = replicaId;
328    if (this.replicaId > MAX_REPLICA_ID) {
329      throw new IllegalArgumentException("ReplicaId cannot be greater than" + MAX_REPLICA_ID);
330    }
331
332    this.regionName = createRegionName(this.tableName, startKey, regionId, replicaId, true);
333
334    this.split = split;
335    this.endKey = endKey == null? HConstants.EMPTY_END_ROW: endKey.clone();
336    this.startKey = startKey == null?
337      HConstants.EMPTY_START_ROW: startKey.clone();
338    this.tableName = tableName;
339    setHashCode();
340  }

424  /**
425   * Make a region name of passed parameters.
426   * @param tableName
427   * @param startKey Can be null
428   * @param id Region id (Usually timestamp from when region was created).
429   * @param replicaId
430   * @param newFormat should we create the region name in the new format
431   * @return Region name made of passed tableName, startKey, id and replicaId
432   */
433  public static byte [] createRegionName(final TableName tableName,
434      final byte [] startKey, final byte [] id, final int replicaId, boolean newFormat) {
435    int len = tableName.getName().length + 2 + id.length +
436        (startKey == null? 0: startKey.length);
437    if (newFormat) {
438      len += MD5_HEX_LENGTH + 2;
439    }
440    byte[] replicaIdBytes = null;
441    // Special casing: replicaId is only appended if replicaId is greater than
442    // 0. This is because all regions in meta would have to be migrated to the new
443    // name otherwise
444    if (replicaId > 0) {
445      // use string representation for replica id
446      replicaIdBytes = Bytes.toBytes(String.format(REPLICA_ID_FORMAT, replicaId));
447      len += 1 + replicaIdBytes.length;
448    }
449
450    byte [] b = new byte [len];
451
452    int offset = tableName.getName().length;
453    System.arraycopy(tableName.getName(), 0, b, 0, offset);
454    b[offset++] = HConstants.DELIMITER;
455    if (startKey != null && startKey.length > 0) {
456      System.arraycopy(startKey, 0, b, offset, startKey.length);
457      offset += startKey.length;
458    }
459    b[offset++] = HConstants.DELIMITER;
460    System.arraycopy(id, 0, b, offset, id.length);
461    offset += id.length;
462
463    if (replicaIdBytes != null) {
464      b[offset++] = REPLICA_ID_DELIMITER;
465      System.arraycopy(replicaIdBytes, 0, b, offset, replicaIdBytes.length);
466      offset += replicaIdBytes.length;
467    }
468
469    if (newFormat) {
470      //
471      // Encoded name should be built into the region name.
472      //
473      // Use the region name thus far (namely, <tablename>,<startKey>,<id>_<replicaId>)
474      // to compute a MD5 hash to be used as the encoded name, and append
475      // it to the byte buffer.
476      //
477      String md5Hash = MD5Hash.getMD5AsHex(b, 0, offset);
478      byte [] md5HashBytes = Bytes.toBytes(md5Hash);
479
480      if (md5HashBytes.length != MD5_HEX_LENGTH) {
481        LOG.error("MD5-hash length mismatch: Expected=" + MD5_HEX_LENGTH +
482                  "; Got=" + md5HashBytes.length);
483      }
484
485      // now append the bytes '.<encodedName>.' to the end
486      b[offset++] = ENC_SEPARATOR;
487      System.arraycopy(md5HashBytes, 0, b, offset, MD5_HEX_LENGTH);
488      offset += MD5_HEX_LENGTH;
489      b[offset++] = ENC_SEPARATOR;
490    }
491
492    return b;
493  }

500  public static TableName getTable(final byte [] regionName) {
501    int offset = -1;
502    for (int i = 0; i < regionName.length; i++) {
503      if (regionName[i] == HConstants.DELIMITER) {
504        offset = i;
505        break;
506      }
507    }
508    byte[] buff  = new byte[offset];
509    System.arraycopy(regionName, 0, buff, 0, offset);
510    return TableName.valueOf(buff);
511  }

522  /**
523   * Separate elements of a regionName.
524   * @param regionName
525   * @return Array of byte[] containing tableName, startKey and id
526   * @throws IOException
527   */
528  public static byte [][] parseRegionName(final byte [] regionName)
529  throws IOException {
530    // Region name is of the format:
531    // tablename,startkey,regionIdTimestamp[_replicaId][.encodedName.]
532    // startkey can contain the delimiter (',') so we parse from the start and end
533
534    // parse from start
535    int offset = -1;
536    for (int i = 0; i < regionName.length; i++) {
537      if (regionName[i] == HConstants.DELIMITER) {
538        offset = i;
539        break;
540      }
541    }
542    if (offset == -1) {
543      throw new IOException(INVALID_REGION_NAME_FORMAT_MESSAGE
544        + ": " + Bytes.toStringBinary(regionName));
545    }
546    byte[] tableName = new byte[offset];
547    System.arraycopy(regionName, 0, tableName, 0, offset);
548    offset = -1;
549
550    int endOffset = regionName.length;
551    // check whether regionName contains encodedName
552    if (regionName.length > MD5_HEX_LENGTH + 2
553        && regionName[regionName.length-1] == ENC_SEPARATOR
554        && regionName[regionName.length-MD5_HEX_LENGTH-2] == ENC_SEPARATOR) {
555      endOffset = endOffset - MD5_HEX_LENGTH - 2;
556    }
557
558    // parse from end
559    byte[] replicaId = null;
560    int idEndOffset = endOffset;
561    for (int i = endOffset - 1; i > 0; i--) {
562      if (regionName[i] == REPLICA_ID_DELIMITER) { //replicaId may or may not be present
563        replicaId = new byte[endOffset - i - 1];
564        System.arraycopy(regionName, i + 1, replicaId, 0,
565          endOffset - i - 1);
566        idEndOffset = i;
567        // do not break, continue to search for id
568      }
569      if (regionName[i] == HConstants.DELIMITER) {
570        offset = i;
571        break;
572      }
573    }
574    if (offset == -1) {
575      throw new IOException(INVALID_REGION_NAME_FORMAT_MESSAGE
576        + ": " + Bytes.toStringBinary(regionName));
577    }
578    byte [] startKey = HConstants.EMPTY_BYTE_ARRAY;
579    if(offset != tableName.length + 1) {
580      startKey = new byte[offset - tableName.length - 1];
581      System.arraycopy(regionName, tableName.length + 1, startKey, 0,
582          offset - tableName.length - 1);
583    }
584    byte [] id = new byte[idEndOffset - offset - 1];
585    System.arraycopy(regionName, offset + 1, id, 0,
586      idEndOffset - offset - 1);
587    byte [][] elements = new byte[replicaId == null ? 3 : 4][];
588    elements[0] = tableName;
589    elements[1] = startKey;
590    elements[2] = id;
591    if (replicaId != null) {
592      elements[3] = replicaId;
593    }
594
595    return elements;
596  }

598  public static boolean isEncodedRegionName(byte[] regionName) throws IOException {
599    try {
600      HRegionInfo.parseRegionName(regionName);
601      return false;
602    } catch (IOException e) {
603      if (StringUtils.stringifyException(e)
604          .contains(HRegionInfo.INVALID_REGION_NAME_FORMAT_MESSAGE)) {
605        return true;
606      }
607      throw e;
608    }
609  }

624  /**
625   * @return Region name as a String for use in logging, etc.
626   */
627  public String getRegionNameAsString() {
628    if (hasEncodedName(this.regionName)) {
629      // new format region names already have their encoded name.
630      return Bytes.toStringBinary(this.regionName);
631    }
632
633    // old format. regionNameStr doesn't have the region name.
634    //
635    //

679   * Returns true if the given inclusive range of rows is fully contained
680   * by this region. For example, if the region is foo,a,g and this is
681   * passed ["b","c"] or ["a","c"] it will return true, but if this is passed
682   * ["b","z"] it will return false.
683   * @throws IllegalArgumentException if the range passed is invalid (ie. end &lt; start)
684   */
685  public boolean containsRange(byte[] rangeStartKey, byte[] rangeEndKey) {
686    if (Bytes.compareTo(rangeStartKey, rangeEndKey) > 0) {
687      throw new IllegalArgumentException(
688      "Invalid range: " + Bytes.toStringBinary(rangeStartKey) +
689      " > " + Bytes.toStringBinary(rangeEndKey));
690    }
691
692    boolean firstKeyInRange = Bytes.compareTo(rangeStartKey, startKey) >= 0;
693    boolean lastKeyInRange =
694      Bytes.compareTo(rangeEndKey, endKey) < 0 ||
695      Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY);
696    return firstKeyInRange && lastKeyInRange;
697  }

820  @Override
821  public int compareTo(HRegionInfo o) {
822    if (o == null) {
823      return 1;
824    }
825
826    // Are regions of same table?
827    int result = this.tableName.compareTo(o.tableName);
828    if (result != 0) {
829      return result;
830    }
831
832    // Compare start keys.
833    result = Bytes.compareTo(this.startKey, o.startKey);
834    if (result != 0) {
835      return result;
836    }
837
838    // Compare end keys.
839    result = Bytes.compareTo(this.endKey, o.endKey);
840
841    if (result != 0) {
842      if (this.getStartKey().length != 0
843              && this.getEndKey().length == 0) {
844          return 1; // this is last region
845      }
846      if (o.getStartKey().length != 0
847              && o.getEndKey().length == 0) {
848          return -1; // o is the last region
849      }
850      return result;
851    }
852
853    // regionId is usually milli timestamp -- this defines older stamps
854    // to be "smaller" than newer stamps in sort order.
855    if (this.regionId > o.regionId) {
856      return 1;
857    } else if (this.regionId < o.regionId) {
858      return -1;
859    }
860
861    int replicaDiff = this.getReplicaId() - o.getReplicaId();
862    if (replicaDiff != 0) return replicaDiff;
863
864    if (this.offLine == o.offLine)
865      return 0;
866    if (this.offLine == true) return -1;
867
868    return 1;
869  }

890  /**
891   * Convert a HRegionInfo to a RegionInfo
892   *
893   * @param info the HRegionInfo to convert
894   * @return the converted RegionInfo
895   */
896  public static RegionInfo convert(final HRegionInfo info) {
897    if (info == null) return null;
898    RegionInfo.Builder builder = RegionInfo.newBuilder();
899    builder.setTableName(ProtobufUtil.toProtoTableName(info.getTable()));
900    builder.setRegionId(info.getRegionId());
901    if (info.getStartKey() != null) {
902      builder.setStartKey(UnsafeByteOperations.unsafeWrap(info.getStartKey()));
903    }
904    if (info.getEndKey() != null) {
905      builder.setEndKey(UnsafeByteOperations.unsafeWrap(info.getEndKey()));
906    }
907    builder.setOffline(info.isOffline());
908    builder.setSplit(info.isSplit());
909    builder.setReplicaId(info.getReplicaId());
910    return builder.build();
911  }

913  /**
914   * Convert a RegionInfo to a HRegionInfo
915   *
916   * @param proto the RegionInfo to convert
917   * @return the converted HRegionInfho
918   */
919  public static HRegionInfo convert(final RegionInfo proto) {
920    if (proto == null) return null;
921    TableName tableName =
922        ProtobufUtil.toTableName(proto.getTableName());
923    if (tableName.equals(TableName.META_TABLE_NAME)) {
924      return RegionReplicaUtil.getRegionInfoForReplica(FIRST_META_REGIONINFO,
925          proto.getReplicaId());
926    }
927    long regionId = proto.getRegionId();
928    int replicaId = proto.hasReplicaId() ? proto.getReplicaId() : DEFAULT_REPLICA_ID;
929    byte[] startKey = null;
930    byte[] endKey = null;
931    if (proto.hasStartKey()) {
932      startKey = proto.getStartKey().toByteArray();
933    }
934    if (proto.hasEndKey()) {
935      endKey = proto.getEndKey().toByteArray();
936    }
937    boolean split = false;
938    if (proto.hasSplit()) {
939      split = proto.getSplit();
940    }
941    HRegionInfo hri = new HRegionInfo(
942        tableName,
943        startKey,
944        endKey, split, regionId, replicaId);
945    if (proto.hasOffline()) {
946      hri.setOffline(proto.getOffline());
947    }
948    return hri;
949  }

995  /**
996   * @param bytes A pb RegionInfo serialized with a pb magic prefix.
997   * @param offset starting point in the byte array
998   * @param len length to read on the byte array
999   * @return A deserialized {@link HRegionInfo}
1000   * @throws DeserializationException
1001   * @see #toByteArray()
1002   */
1003  public static HRegionInfo parseFrom(final byte [] bytes, int offset, int len)
1004      throws DeserializationException {
1005    if (ProtobufUtil.isPBMagicPrefix(bytes, offset, len)) {
1006      int pblen = ProtobufUtil.lengthOfPBMagic();
1007      try {
1008        HBaseProtos.RegionInfo.Builder builder = HBaseProtos.RegionInfo.newBuilder();
1009        ProtobufUtil.mergeFrom(builder, bytes, pblen + offset, len - pblen);
1010        HBaseProtos.RegionInfo ri = builder.build();
1011        return convert(ri);
1012      } catch (IOException e) {
1013        throw new DeserializationException(e);
1014      }
1015    } else {
1016      throw new DeserializationException("PB encoded HRegionInfo expected");
1017    }
1018  }

1081  /**
1082   * Get the region name for display. Optionally hide the start key.
1083   * @param hri
1084   * @param conf
1085   * @return region name bytes
1086   */
1087  public static byte[] getRegionNameForDisplay(HRegionInfo hri, Configuration conf) {
1088    boolean displayKey = conf.getBoolean(DISPLAY_KEYS_KEY, true);
1089    if (displayKey || hri.getTable().equals(TableName.META_TABLE_NAME)) {
1090      return hri.getRegionName();
1091    } else {
1092      // create a modified regionname with the startkey replaced but preserving
1093      // the other parts including the encodedname.
1094      try {
1095        byte[][]regionNameParts = parseRegionName(hri.getRegionName());
1096        regionNameParts[1] = HIDDEN_START_KEY; //replace the real startkey
1097        int len = 0;
1098        // get the total length
1099        for (byte[] b : regionNameParts) {
1100          len += b.length;
1101        }
1102        byte[] encodedRegionName =
1103            Bytes.toBytes(encodeRegionName(hri.getRegionName()));
1104        len += encodedRegionName.length;
1105        //allocate some extra bytes for the delimiters and the last '.'
1106        byte[] modifiedName = new byte[len + regionNameParts.length + 1];
1107        int lengthSoFar = 0;
1108        int loopCount = 0;
1109        for (byte[] b : regionNameParts) {
1110          System.arraycopy(b, 0, modifiedName, lengthSoFar, b.length);
1111          lengthSoFar += b.length;
1112          if (loopCount++ == 2) modifiedName[lengthSoFar++] = REPLICA_ID_DELIMITER;
1113          else  modifiedName[lengthSoFar++] = HConstants.DELIMITER;
1114        }
1115        // replace the last comma with '.'
1116        modifiedName[lengthSoFar - 1] = ENC_SEPARATOR;
1117        System.arraycopy(encodedRegionName, 0, modifiedName, lengthSoFar,
1118            encodedRegionName.length);
1119        lengthSoFar += encodedRegionName.length;
1120        modifiedName[lengthSoFar] = ENC_SEPARATOR;
1121        return modifiedName;
1122      } catch (IOException e) {
1123        //LOG.warn("Encountered exception " + e);
1124        throw new RuntimeException(e);
1125      }
1126    }
1127  }

1129  /**
1130   * Parses an HRegionInfo instance from the passed in stream.  Presumes the HRegionInfo was
1131   * serialized to the stream with {@link #toDelimitedByteArray()}
1132   * @param in
1133   * @return An instance of HRegionInfo.
1134   * @throws IOException
1135   */
1136  public static HRegionInfo parseFrom(final DataInputStream in) throws IOException {
1137    // I need to be able to move back in the stream if this is not a pb serialization so I can
1138    // do the Writable decoding instead.
1139    int pblen = ProtobufUtil.lengthOfPBMagic();
1140    byte [] pbuf = new byte[pblen];
1141    if (in.markSupported()) { //read it with mark()
1142      in.mark(pblen);
1143    }
1144
1145    //assumption: if Writable serialization, it should be longer than pblen.
1146    int read = in.read(pbuf);
1147    if (read != pblen) throw new IOException("read=" + read + ", wanted=" + pblen);
1148    if (ProtobufUtil.isPBMagicPrefix(pbuf)) {
1149      return convert(HBaseProtos.RegionInfo.parseDelimitedFrom(in));
1150    } else {
1151      throw new IOException("PB encoded HRegionInfo expected");
1152    }
1153  }

1155  /**
1156   * Serializes given HRegionInfo's as a byte array. Use this instead of {@link #toByteArray()} when
1157   * writing to a stream and you want to use the pb mergeDelimitedFrom (w/o the delimiter, pb reads
1158   * to EOF which may not be what you want). {@link #parseDelimitedFrom(byte[], int, int)} can
1159   * be used to read back the instances.
1160   * @param infos HRegionInfo objects to serialize
1161   * @return This instance serialized as a delimited protobuf w/ a magic pb prefix.
1162   * @throws IOException
1163   * @see #toByteArray()
1164   */
1165  public static byte[] toDelimitedByteArray(HRegionInfo... infos) throws IOException {
1166    byte[][] bytes = new byte[infos.length][];
1167    int size = 0;
1168    for (int i = 0; i < infos.length; i++) {
1169      bytes[i] = infos[i].toDelimitedByteArray();
1170      size += bytes[i].length;
1171    }
1172
1173    byte[] result = new byte[size];
1174    int offset = 0;
1175    for (byte[] b : bytes) {
1176      System.arraycopy(b, 0, result, offset, b.length);
1177      offset += b.length;
1178    }
1179    return result;
1180  }

1182  /**
1183   * Parses all the HRegionInfo instances from the passed in stream until EOF. Presumes the
1184   * HRegionInfo's were serialized to the stream with {@link #toDelimitedByteArray()}
1185   * @param bytes serialized bytes
1186   * @param offset the start offset into the byte[] buffer
1187   * @param length how far we should read into the byte[] buffer
1188   * @return All the hregioninfos that are in the byte array. Keeps reading till we hit the end.
1189   */
1190  public static List<HRegionInfo> parseDelimitedFrom(final byte[] bytes, final int offset,
1191      final int length) throws IOException {
1192    if (bytes == null) {
1193      throw new IllegalArgumentException("Can't build an object with empty bytes array");
1194    }
1195    DataInputBuffer in = new DataInputBuffer();
1196    List<HRegionInfo> hris = new ArrayList<>();
1197    try {
1198      in.reset(bytes, offset, length);
1199      while (in.available() > 0) {
1200        HRegionInfo hri = parseFrom(in);
1201        hris.add(hri);
1202      }
1203    } finally {
1204      in.close();
1205    }
1206    return hris;
1207  }

1209  /**
1210   * Check whether two regions are adjacent
1211   * @param regionA
1212   * @param regionB
1213   * @return true if two regions are adjacent
1214   */
1215  public static boolean areAdjacent(HRegionInfo regionA, HRegionInfo regionB) {
1216    if (regionA == null || regionB == null) {
1217      throw new IllegalArgumentException(
1218          "Can't check whether adjacent for null region");
1219    }
1220    HRegionInfo a = regionA;
1221    HRegionInfo b = regionB;
1222    if (Bytes.compareTo(a.getStartKey(), b.getStartKey()) > 0) {
1223      a = regionB;
1224      b = regionA;
1225    }
1226    if (Bytes.compareTo(a.getEndKey(), b.getStartKey()) == 0) {
1227      return true;
1228    }
1229    return false;
1230  }
