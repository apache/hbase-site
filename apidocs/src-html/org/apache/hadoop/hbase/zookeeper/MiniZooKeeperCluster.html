112    return (clientPortList.size() > index && clientPortList.get(index) > 0);

115  public void setDefaultClientPort(int clientPort) {
116    if (clientPort <= 0) {
117      throw new IllegalArgumentException("Invalid default ZK client port: "
118          + clientPort);
119    }
120    this.defaultClientPort = clientPort;

123  /**
124   * Selects a ZK client port.
125   *
126   * @param seedPort the seed port to start with; -1 means first time.
127   * @Returns a valid and unused client port
128   */
129  private int selectClientPort(int seedPort) {
130    int i;
131    int returnClientPort = seedPort + 1;
132    if (returnClientPort == 0) {
133      // If the new port is invalid, find one - starting with the default client port.
134      // If the default client port is not specified, starting with a random port.
135      // The random port is selected from the range between 49152 to 65535. These ports cannot be
136      // registered with IANA and are intended for dynamic allocation (see http://bit.ly/dynports).
137      if (defaultClientPort > 0) {
138        returnClientPort = defaultClientPort;
139      } else {
140        returnClientPort = 0xc000 + new Random().nextInt(0x3f00);
141      }
142    }
143    // Make sure that the port is unused.
144    while (true) {
145      for (i = 0; i < clientPortList.size(); i++) {
146        if (returnClientPort == clientPortList.get(i)) {
147          // Already used. Update the port and retry.
148          returnClientPort++;
149          break;
150        }
151      }
152      if (i == clientPortList.size()) {
153        break; // found a unused port, exit
154      }
155    }
156    return returnClientPort;
157  }

172  private static void setupTestEnv() {
173    // during the tests we run with 100K prealloc in the logs.
174    // on windows systems prealloc of 64M was seen to take ~15seconds
175    // resulting in test failure (client timeout on first session).
176    // set env and directly in order to handle static init/gc issues
177    System.setProperty("zookeeper.preAllocSize", "100");
178    FileTxnLog.setPreallocSize(100 * 1024);
179  }

181  public int startup(File baseDir) throws IOException, InterruptedException {
182    int numZooKeeperServers = clientPortList.size();
183    if (numZooKeeperServers == 0) {
184      numZooKeeperServers = 1; // need at least 1 ZK server for testing
185    }
186    return startup(baseDir, numZooKeeperServers);
187  }

189  /**
190   * @param baseDir
191   * @param numZooKeeperServers
192   * @return ClientPort server bound to, -1 if there was a
193   *         binding problem and we couldn't pick another port.
194   * @throws IOException
195   * @throws InterruptedException
196   */
197  public int startup(File baseDir, int numZooKeeperServers) throws IOException,
198      InterruptedException {
199    if (numZooKeeperServers <= 0)
200      return -1;
201
202    setupTestEnv();
203    shutdown();
204
205    int tentativePort = -1; // the seed port
206    int currentClientPort;
207
208    // running all the ZK servers
209    for (int i = 0; i < numZooKeeperServers; i++) {
210      File dir = new File(baseDir, "zookeeper_"+i).getAbsoluteFile();
211      createDir(dir);
212      int tickTimeToUse;
213      if (this.tickTime > 0) {
214        tickTimeToUse = this.tickTime;
215      } else {
216        tickTimeToUse = TICK_TIME;
217      }
218
219      // Set up client port - if we have already had a list of valid ports, use it.
220      if (hasValidClientPortInList(i)) {
221        currentClientPort = clientPortList.get(i);
222      } else {
223        tentativePort = selectClientPort(tentativePort); // update the seed
224        currentClientPort = tentativePort;
225      }
226
227      ZooKeeperServer server = new ZooKeeperServer(dir, dir, tickTimeToUse);
228      // Setting {min,max}SessionTimeout defaults to be the same as in Zookeeper
229      server.setMinSessionTimeout(configuration.getInt("hbase.zookeeper.property.minSessionTimeout", -1));
230      server.setMaxSessionTimeout(configuration.getInt("hbase.zookeeper.property.maxSessionTimeout", -1));
231      NIOServerCnxnFactory standaloneServerFactory;
232      while (true) {
233        try {
234          standaloneServerFactory = new NIOServerCnxnFactory();
235          standaloneServerFactory.configure(
236            new InetSocketAddress(currentClientPort),
237            configuration.getInt(HConstants.ZOOKEEPER_MAX_CLIENT_CNXNS, HConstants.DEFAULT_ZOOKEPER_MAX_CLIENT_CNXNS));
238        } catch (BindException e) {
239          LOG.debug("Failed binding ZK Server to client port: " +
240              currentClientPort, e);
241          // We're told to use some port but it's occupied, fail
242          if (hasValidClientPortInList(i)) {
243            return -1;
244          }
245          // This port is already in use, try to use another.
246          tentativePort = selectClientPort(tentativePort);
247          currentClientPort = tentativePort;
248          continue;
249        }
250        break;
251      }
252
253      // Start up this ZK server
254      standaloneServerFactory.startup(server);
255      // Runs a 'stat' against the servers.
256      if (!waitForServerUp(currentClientPort, connectionTimeout)) {
257        throw new IOException("Waiting for startup of standalone server");
258      }
259
260      // We have selected a port as a client port.  Update clientPortList if necessary.
261      if (clientPortList.size() <= i) { // it is not in the list, add the port
262        clientPortList.add(currentClientPort);
263      }
264      else if (clientPortList.get(i) <= 0) { // the list has invalid port, update with valid port
265        clientPortList.remove(i);
266        clientPortList.add(i, currentClientPort);
267      }
268
269      standaloneServerFactoryList.add(standaloneServerFactory);
270      zooKeeperServers.add(server);
271    }
272
273    // set the first one to be active ZK; Others are backups
274    activeZKServerIndex = 0;
275    started = true;
276    int clientPort = clientPortList.get(activeZKServerIndex);
277    LOG.info("Started MiniZooKeeperCluster and ran successful 'stat' " +
278        "on client port=" + clientPort);
279    return clientPort;
280  }

282  private void createDir(File dir) throws IOException {
283    try {
284      if (!dir.exists()) {
285        dir.mkdirs();
286      }
287    } catch (SecurityException e) {
288      throw new IOException("creating dir: " + dir, e);
289    }
290  }

292  /**
293   * @throws IOException
294   */
295  public void shutdown() throws IOException {
296    // shut down all the zk servers
297    for (int i = 0; i < standaloneServerFactoryList.size(); i++) {
298      NIOServerCnxnFactory standaloneServerFactory =
299        standaloneServerFactoryList.get(i);
300      int clientPort = clientPortList.get(i);
301
302      standaloneServerFactory.shutdown();
303      if (!waitForServerDown(clientPort, connectionTimeout)) {
304        throw new IOException("Waiting for shutdown of standalone server");
305      }
306    }
307    standaloneServerFactoryList.clear();
308
309    for (ZooKeeperServer zkServer: zooKeeperServers) {
310      //explicitly close ZKDatabase since ZookeeperServer does not close them
311      zkServer.getZKDatabase().close();
312    }
313    zooKeeperServers.clear();
314
315    // clear everything
316    if (started) {
317      started = false;
318      activeZKServerIndex = 0;
319      clientPortList.clear();
320      LOG.info("Shutdown MiniZK cluster with all ZK servers");
321    }
322  }

324  /**@return clientPort return clientPort if there is another ZK backup can run
325   *         when killing the current active; return -1, if there is no backups.
326   * @throws IOException
327   * @throws InterruptedException
328   */
329  public int killCurrentActiveZooKeeperServer() throws IOException,
330                                        InterruptedException {
331    if (!started || activeZKServerIndex < 0) {
332      return -1;
333    }
334
335    // Shutdown the current active one
336    NIOServerCnxnFactory standaloneServerFactory =
337      standaloneServerFactoryList.get(activeZKServerIndex);
338    int clientPort = clientPortList.get(activeZKServerIndex);
339
340    standaloneServerFactory.shutdown();
341    if (!waitForServerDown(clientPort, connectionTimeout)) {
342      throw new IOException("Waiting for shutdown of standalone server");
343    }
344
345    zooKeeperServers.get(activeZKServerIndex).getZKDatabase().close();
346
347    // remove the current active zk server
348    standaloneServerFactoryList.remove(activeZKServerIndex);
349    clientPortList.remove(activeZKServerIndex);
350    zooKeeperServers.remove(activeZKServerIndex);
351    LOG.info("Kill the current active ZK servers in the cluster " +
352        "on client port: " + clientPort);
353
354    if (standaloneServerFactoryList.isEmpty()) {
355      // there is no backup servers;
356      return -1;
357    }
358    clientPort = clientPortList.get(activeZKServerIndex);
359    LOG.info("Activate a backup zk server in the cluster " +
360        "on client port: " + clientPort);
361    // return the next back zk server's port
362    return clientPort;
363  }

365  /**
366   * Kill one back up ZK servers
367   * @throws IOException
368   * @throws InterruptedException
369   */
370  public void killOneBackupZooKeeperServer() throws IOException,
371                                        InterruptedException {
372    if (!started || activeZKServerIndex < 0 ||
373        standaloneServerFactoryList.size() <= 1) {
374      return ;
375    }
376
377    int backupZKServerIndex = activeZKServerIndex+1;
378    // Shutdown the current active one
379    NIOServerCnxnFactory standaloneServerFactory =
380      standaloneServerFactoryList.get(backupZKServerIndex);
381    int clientPort = clientPortList.get(backupZKServerIndex);
382
383    standaloneServerFactory.shutdown();
384    if (!waitForServerDown(clientPort, connectionTimeout)) {
385      throw new IOException("Waiting for shutdown of standalone server");
386    }
387
388    zooKeeperServers.get(backupZKServerIndex).getZKDatabase().close();
389
390    // remove this backup zk server
391    standaloneServerFactoryList.remove(backupZKServerIndex);
392    clientPortList.remove(backupZKServerIndex);
393    zooKeeperServers.remove(backupZKServerIndex);
394    LOG.info("Kill one backup ZK servers in the cluster " +
395        "on client port: " + clientPort);
396  }

398  // XXX: From o.a.zk.t.ClientBase
399  private static boolean waitForServerDown(int port, long timeout) throws IOException {
400    long start = System.currentTimeMillis();
401    while (true) {
402      try {
403        Socket sock = new Socket("localhost", port);
404        try {
405          OutputStream outstream = sock.getOutputStream();
406          outstream.write("stat".getBytes());
407          outstream.flush();
408        } finally {
409          sock.close();
410        }
411      } catch (IOException e) {
412        return true;
413      }
414
415      if (System.currentTimeMillis() > start + timeout) {
416        break;
417      }
418      try {
419        Thread.sleep(250);
420      } catch (InterruptedException e) {
421        throw (InterruptedIOException)new InterruptedIOException().initCause(e);
422      }
423    }
424    return false;
425  }

427  // XXX: From o.a.zk.t.ClientBase
428  private static boolean waitForServerUp(int port, long timeout) throws IOException {
429    long start = System.currentTimeMillis();
430    while (true) {
431      try {
432        Socket sock = new Socket("localhost", port);
433        BufferedReader reader = null;
434        try {
435          OutputStream outstream = sock.getOutputStream();
436          outstream.write("stat".getBytes());
437          outstream.flush();
438
439          Reader isr = new InputStreamReader(sock.getInputStream());
440          reader = new BufferedReader(isr);
441          String line = reader.readLine();
442          if (line != null && line.startsWith("Zookeeper version:")) {
443            return true;
444          }
445        } finally {
446          sock.close();
447          if (reader != null) {
448            reader.close();
449          }
450        }
451      } catch (IOException e) {
452        // ignore as this is expected
453        LOG.info("server localhost:" + port + " not up " + e);
454      }
455
456      if (System.currentTimeMillis() > start + timeout) {
457        break;
458      }
459      try {
460        Thread.sleep(250);
461      } catch (InterruptedException e) {
462        throw (InterruptedIOException)new InterruptedIOException().initCause(e);
463      }
464    }
465    return false;
466  }
