101  public static class CellWritableComparablePartitioner
102      extends Partitioner<CellWritableComparable, Cell> {
103    private static CellWritableComparable[] START_KEYS = null;
104    @Override
105    public int getPartition(CellWritableComparable key, Cell value,
106        int numPartitions) {
107      for (int i = 0; i < START_KEYS.length; ++i) {

191  public static class CellSortImporter
192      extends TableMapper<CellWritableComparable, Cell> {
193    private Map<byte[], byte[]> cfRenameMap;
194    private Filter filter;
195    private static final Logger LOG = LoggerFactory.getLogger(CellImporter.class);
196
197    /**
198     * @param row  The current table row key.
199     * @param value  The columns.
200     * @param context  The current context.
201     * @throws IOException When something is broken with the data.
202     */
203    @Override
204    public void map(ImmutableBytesWritable row, Result value,
205      Context context)
206    throws IOException {
207      try {
208        if (LOG.isTraceEnabled()) {
209          LOG.trace("Considering the row."
210              + Bytes.toString(row.get(), row.getOffset(), row.getLength()));
211        }
212        if (filter == null || !filter.filterRowKey(
213          PrivateCellUtil.createFirstOnRow(row.get(), row.getOffset(), (short) row.getLength()))) {
214          for (Cell kv : value.rawCells()) {
215            kv = filterKv(filter, kv);
216            // skip if we filtered it out
217            if (kv == null) continue;
218            Cell ret = convertKv(kv, cfRenameMap);
219            context.write(new CellWritableComparable(ret), ret);
220          }
221        }
222      } catch (InterruptedException e) {
223        e.printStackTrace();
224      }
225    }
226
227    @Override
228    public void setup(Context context) throws IOException {
229      cfRenameMap = createCfRenameMap(context.getConfiguration());
230      filter = instantiateFilter(context.getConfiguration());
231      int reduceNum = context.getNumReduceTasks();
232      Configuration conf = context.getConfiguration();
233      TableName tableName = TableName.valueOf(context.getConfiguration().get(TABLE_NAME));
234      try (Connection conn = ConnectionFactory.createConnection(conf);
235          RegionLocator regionLocator = conn.getRegionLocator(tableName)) {
236        byte[][] startKeys = regionLocator.getStartKeys();
237        if (startKeys.length != reduceNum) {
238          throw new IOException("Region split after job initialization");
239        }
240        CellWritableComparable[] startKeyWraps =
241            new CellWritableComparable[startKeys.length - 1];
242        for (int i = 1; i < startKeys.length; ++i) {
243          startKeyWraps[i - 1] =
244              new CellWritableComparable(KeyValueUtil.createFirstOnRow(startKeys[i]));
245        }
246        CellWritableComparablePartitioner.START_KEYS = startKeyWraps;
247      }
248    }
249  }

298  /**
299   * Write table content out to files in hdfs.
300   */
301  public static class Importer extends TableMapper<ImmutableBytesWritable, Mutation> {
302    private Map<byte[], byte[]> cfRenameMap;
303    private List<UUID> clusterIds;
304    private Filter filter;
305    private Durability durability;
306
307    /**
308     * @param row  The current table row key.
309     * @param value  The columns.
310     * @param context  The current context.
311     * @throws IOException When something is broken with the data.
312     */
313    @Override
314    public void map(ImmutableBytesWritable row, Result value,
315      Context context)
316    throws IOException {
317      try {
318        writeResult(row, value, context);
319      } catch (InterruptedException e) {
320        e.printStackTrace();
321      }
322    }
323
324    private void writeResult(ImmutableBytesWritable key, Result result, Context context)
325    throws IOException, InterruptedException {
326      Put put = null;
327      Delete delete = null;
328      if (LOG.isTraceEnabled()) {
329        LOG.trace("Considering the row."
330            + Bytes.toString(key.get(), key.getOffset(), key.getLength()));
331      }
332      if (filter == null
333          || !filter.filterRowKey(PrivateCellUtil.createFirstOnRow(key.get(), key.getOffset(),
334              (short) key.getLength()))) {
335        processKV(key, result, context, put, delete);
336      }
337    }
338
339    protected void processKV(ImmutableBytesWritable key, Result result, Context context, Put put,
340        Delete delete) throws IOException, InterruptedException {
341      for (Cell kv : result.rawCells()) {
342        kv = filterKv(filter, kv);
343        // skip if we filter it out
344        if (kv == null) continue;
345
346        kv = convertKv(kv, cfRenameMap);
347        // Deletes and Puts are gathered and written when finished
348        /*
349         * If there are sequence of mutations and tombstones in an Export, and after Import the same
350         * sequence should be restored as it is. If we combine all Delete tombstones into single
351         * request then there is chance of ignoring few DeleteFamily tombstones, because if we
352         * submit multiple DeleteFamily tombstones in single Delete request then we are maintaining
353         * only newest in hbase table and ignoring other. Check - HBASE-12065
354         */
355        if (PrivateCellUtil.isDeleteFamily(kv)) {
356          Delete deleteFamily = new Delete(key.get());
357          deleteFamily.add(kv);
358          if (durability != null) {
359            deleteFamily.setDurability(durability);
360          }
361          deleteFamily.setClusterIds(clusterIds);
362          context.write(key, deleteFamily);
363        } else if (CellUtil.isDelete(kv)) {
364          if (delete == null) {
365            delete = new Delete(key.get());
366          }
367          delete.add(kv);
368        } else {
369          if (put == null) {
370            put = new Put(key.get());
371          }
372          addPutToKv(put, kv);
373        }
374      }
375      if (put != null) {
376        if (durability != null) {
377          put.setDurability(durability);
378        }
379        put.setClusterIds(clusterIds);
380        context.write(key, put);
381      }
382      if (delete != null) {
383        if (durability != null) {
384          delete.setDurability(durability);
385        }
386        delete.setClusterIds(clusterIds);
387        context.write(key, delete);
388      }
389    }
390
391    protected void addPutToKv(Put put, Cell kv) throws IOException {
392      put.add(kv);
393    }
394
395    @Override
396    public void setup(Context context) {
397      LOG.info("Setting up " + getClass() + " mapper.");
398      Configuration conf = context.getConfiguration();
399      cfRenameMap = createCfRenameMap(conf);
400      filter = instantiateFilter(conf);
401      String durabilityStr = conf.get(WAL_DURABILITY);
402      if(durabilityStr != null){
403        durability = Durability.valueOf(durabilityStr.toUpperCase(Locale.ROOT));
404        LOG.info("setting WAL durability to " + durability);
405      } else {
406        LOG.info("setting WAL durability to default.");
407      }
408      // TODO: This is kind of ugly doing setup of ZKW just to read the clusterid.
409      ZKWatcher zkw = null;
410      Exception ex = null;
411      try {
412        zkw = new ZKWatcher(conf, context.getTaskAttemptID().toString(), null);
413        clusterIds = Collections.singletonList(ZKClusterId.getUUIDForCluster(zkw));
414      } catch (ZooKeeperConnectionException e) {
415        ex = e;
416        LOG.error("Problem connecting to ZooKeper during task setup", e);
417      } catch (KeeperException e) {
418        ex = e;
419        LOG.error("Problem reading ZooKeeper data during task setup", e);
420      } catch (IOException e) {
421        ex = e;
422        LOG.error("Problem setting up task", e);
423      } finally {
424        if (zkw != null) zkw.close();
425      }
426      if (clusterIds == null) {
427        // exit early if setup fails
428        throw new RuntimeException(ex);
429      }
430    }
431  }

433  /**
434   * Create a {@link Filter} to apply to all incoming keys ({@link KeyValue KeyValues}) to
435   * optionally not include in the job output
436   * @param conf {@link Configuration} from which to load the filter
437   * @return the filter to use for the task, or <tt>null</tt> if no filter to should be used
438   * @throws IllegalArgumentException if the filter is misconfigured
439   */
440  public static Filter instantiateFilter(Configuration conf) {
441    // get the filter, if it was configured
442    Class<? extends Filter> filterClass = conf.getClass(FILTER_CLASS_CONF_KEY, null, Filter.class);
443    if (filterClass == null) {
444      LOG.debug("No configured filter class, accepting all keyvalues.");
445      return null;
446    }
447    LOG.debug("Attempting to create filter:" + filterClass);
448    String[] filterArgs = conf.getStrings(FILTER_ARGS_CONF_KEY);
449    ArrayList<byte[]> quotedArgs = toQuotedByteArrays(filterArgs);
450    try {
451      Method m = filterClass.getMethod("createFilterFromArguments", ArrayList.class);
452      return (Filter) m.invoke(null, quotedArgs);
453    } catch (IllegalAccessException e) {
454      LOG.error("Couldn't instantiate filter!", e);
455      throw new RuntimeException(e);
456    } catch (SecurityException e) {
457      LOG.error("Couldn't instantiate filter!", e);
458      throw new RuntimeException(e);
459    } catch (NoSuchMethodException e) {
460      LOG.error("Couldn't instantiate filter!", e);
461      throw new RuntimeException(e);
462    } catch (IllegalArgumentException e) {
463      LOG.error("Couldn't instantiate filter!", e);
464      throw new RuntimeException(e);
465    } catch (InvocationTargetException e) {
466      LOG.error("Couldn't instantiate filter!", e);
467      throw new RuntimeException(e);
468    }
469  }
470
471  private static ArrayList<byte[]> toQuotedByteArrays(String... stringArgs) {
472    ArrayList<byte[]> quotedArgs = new ArrayList<>();
473    for (String stringArg : stringArgs) {
474      // all the filters' instantiation methods expected quoted args since they are coming from
475      // the shell, so add them here, though it shouldn't really be needed :-/
476      quotedArgs.add(Bytes.toBytes("'" + stringArg + "'"));
477    }
478    return quotedArgs;
479  }

481  /**
482   * Attempt to filter out the keyvalue
483   * @param c {@link Cell} on which to apply the filter
484   * @return <tt>null</tt> if the key should not be written, otherwise returns the original
485   *         {@link Cell}
486   */
487  public static Cell filterKv(Filter filter, Cell c) throws IOException {
488    // apply the filter and skip this kv if the filter doesn't apply
489    if (filter != null) {
490      Filter.ReturnCode code = filter.filterCell(c);
491      if (LOG.isTraceEnabled()) {
492        LOG.trace("Filter returned:" + code + " for the cell:" + c);
493      }
494      // if its not an accept type, then skip this kv
495      if (!(code.equals(Filter.ReturnCode.INCLUDE) || code
496          .equals(Filter.ReturnCode.INCLUDE_AND_NEXT_COL))) {
497        return null;
498      }
499    }
500    return c;
501  }

503  // helper: create a new KeyValue based on CF rename map
504  private static Cell convertKv(Cell kv, Map<byte[], byte[]> cfRenameMap) {
505    if(cfRenameMap != null) {
506      // If there's a rename mapping for this CF, create a new KeyValue
507      byte[] newCfName = cfRenameMap.get(CellUtil.cloneFamily(kv));
508      if (newCfName != null) {
509        kv = new KeyValue(kv.getRowArray(), // row buffer
510            kv.getRowOffset(),              // row offset
511            kv.getRowLength(),              // row length
512            newCfName,                      // CF buffer
513            0,                              // CF offset
514            newCfName.length,               // CF length
515            kv.getQualifierArray(),         // qualifier buffer
516            kv.getQualifierOffset(),        // qualifier offset
517            kv.getQualifierLength(),        // qualifier length
518            kv.getTimestamp(),              // timestamp
519            KeyValue.Type.codeToType(kv.getTypeByte()), // KV Type
520            kv.getValueArray(),             // value buffer
521            kv.getValueOffset(),            // value offset
522            kv.getValueLength());           // value length
523      }
524    }
525    return kv;
526  }

528  // helper: make a map from sourceCfName to destCfName by parsing a config key
529  private static Map<byte[], byte[]> createCfRenameMap(Configuration conf) {
530    Map<byte[], byte[]> cfRenameMap = null;
531    String allMappingsPropVal = conf.get(CF_RENAME_PROP);
532    if(allMappingsPropVal != null) {
533      // The conf value format should be sourceCf1:destCf1,sourceCf2:destCf2,...
534      String[] allMappings = allMappingsPropVal.split(",");
535      for (String mapping: allMappings) {
536        if(cfRenameMap == null) {
537            cfRenameMap = new TreeMap<>(Bytes.BYTES_COMPARATOR);
538        }
539        String [] srcAndDest = mapping.split(":");
540        if(srcAndDest.length != 2) {
541            continue;
542        }
543        cfRenameMap.put(srcAndDest[0].getBytes(), srcAndDest[1].getBytes());
544      }
545    }
546    return cfRenameMap;
547  }

549  /**
550   * <p>Sets a configuration property with key {@link #CF_RENAME_PROP} in conf that tells
551   * the mapper how to rename column families.
552   *
553   * <p>Alternately, instead of calling this function, you could set the configuration key
554   * {@link #CF_RENAME_PROP} yourself. The value should look like
555   * <pre>srcCf1:destCf1,srcCf2:destCf2,....</pre>. This would have the same effect on
556   * the mapper behavior.
557   *
558   * @param conf the Configuration in which the {@link #CF_RENAME_PROP} key will be
559   *  set
560   * @param renameMap a mapping from source CF names to destination CF names
561   */
562  static public void configureCfRenaming(Configuration conf,
563          Map<String, String> renameMap) {
564    StringBuilder sb = new StringBuilder();
565    for(Map.Entry<String,String> entry: renameMap.entrySet()) {
566      String sourceCf = entry.getKey();
567      String destCf = entry.getValue();
568
569      if(sourceCf.contains(":") || sourceCf.contains(",") ||
570              destCf.contains(":") || destCf.contains(",")) {
571        throw new IllegalArgumentException("Illegal character in CF names: "
572              + sourceCf + ", " + destCf);
573      }
574
575      if(sb.length() != 0) {
576        sb.append(",");
577      }
578      sb.append(sourceCf + ":" + destCf);
579    }
580    conf.set(CF_RENAME_PROP, sb.toString());
581  }

595  /**
596   * Sets up the actual job.
597   * @param conf The current configuration.
598   * @param args The command line parameters.
599   * @return The newly created job.
600   * @throws IOException When setting up the job fails.
601   */
602  public static Job createSubmittableJob(Configuration conf, String[] args)
603  throws IOException {
604    TableName tableName = TableName.valueOf(args[0]);
605    conf.set(TABLE_NAME, tableName.getNameAsString());
606    Path inputDir = new Path(args[1]);
607    Job job = Job.getInstance(conf, conf.get(JOB_NAME_CONF_KEY, NAME + "_" + tableName));
608    job.setJarByClass(Importer.class);
609    FileInputFormat.setInputPaths(job, inputDir);
610    job.setInputFormatClass(SequenceFileInputFormat.class);
611    String hfileOutPath = conf.get(BULK_OUTPUT_CONF_KEY);
612
613    // make sure we get the filter in the jars
614    try {
615      Class<? extends Filter> filter = conf.getClass(FILTER_CLASS_CONF_KEY, null, Filter.class);
616      if (filter != null) {
617        TableMapReduceUtil.addDependencyJarsForClasses(conf, filter);
618      }
619    } catch (Exception e) {
620      throw new IOException(e);
621    }
622
623    if (hfileOutPath != null && conf.getBoolean(HAS_LARGE_RESULT, false)) {
624      LOG.info("Use Large Result!!");
625      try (Connection conn = ConnectionFactory.createConnection(conf);
626          Table table = conn.getTable(tableName);
627          RegionLocator regionLocator = conn.getRegionLocator(tableName)) {
628        HFileOutputFormat2.configureIncrementalLoad(job, table.getDescriptor(), regionLocator);
629        job.setMapperClass(CellSortImporter.class);
630        job.setReducerClass(CellReducer.class);
631        Path outputDir = new Path(hfileOutPath);
632        FileOutputFormat.setOutputPath(job, outputDir);
633        job.setMapOutputKeyClass(CellWritableComparable.class);
634        job.setMapOutputValueClass(MapReduceExtendedCell.class);
635        job.getConfiguration().setClass("mapreduce.job.output.key.comparator.class",
636            CellWritableComparable.CellWritableComparator.class,
637            RawComparator.class);
638        Path partitionsPath =
639            new Path(TotalOrderPartitioner.getPartitionFile(job.getConfiguration()));
640        FileSystem fs = FileSystem.get(job.getConfiguration());
641        fs.deleteOnExit(partitionsPath);
642        job.setPartitionerClass(CellWritableComparablePartitioner.class);
643        job.setNumReduceTasks(regionLocator.getStartKeys().length);
644        TableMapReduceUtil.addDependencyJarsForClasses(job.getConfiguration(),
645            org.apache.hadoop.hbase.shaded.com.google.common.base.Preconditions.class);
646      }
647    } else if (hfileOutPath != null) {
648      LOG.info("writing to hfiles for bulk load.");
649      job.setMapperClass(CellImporter.class);
650      try (Connection conn = ConnectionFactory.createConnection(conf);
651          Table table = conn.getTable(tableName);
652          RegionLocator regionLocator = conn.getRegionLocator(tableName)){
653        job.setReducerClass(CellSortReducer.class);
654        Path outputDir = new Path(hfileOutPath);
655        FileOutputFormat.setOutputPath(job, outputDir);
656        job.setMapOutputKeyClass(ImmutableBytesWritable.class);
657        job.setMapOutputValueClass(MapReduceExtendedCell.class);
658        HFileOutputFormat2.configureIncrementalLoad(job, table.getDescriptor(), regionLocator);
659        TableMapReduceUtil.addDependencyJarsForClasses(job.getConfiguration(),
660            org.apache.hadoop.hbase.shaded.com.google.common.base.Preconditions.class);
661      }
662    } else {
663      LOG.info("writing directly to table from Mapper.");
664      // No reducers.  Just write straight to table.  Call initTableReducerJob
665      // because it sets up the TableOutputFormat.
666      job.setMapperClass(Importer.class);
667      TableMapReduceUtil.initTableReducerJob(tableName.getNameAsString(), null, job);
668      job.setNumReduceTasks(0);
669    }
670    return job;
671  }

710  /**
711   * If the durability is set to {@link Durability#SKIP_WAL} and the data is imported to hbase, we
712   * need to flush all the regions of the table as the data is held in memory and is also not
713   * present in the Write Ahead Log to replay in scenarios of a crash. This method flushes all the
714   * regions of the table in the scenarios of import data to hbase with {@link Durability#SKIP_WAL}
715   */
716  public static void flushRegionsIfNecessary(Configuration conf) throws IOException,
717      InterruptedException {
718    String tableName = conf.get(TABLE_NAME);
719    Admin hAdmin = null;
720    Connection connection = null;
721    String durability = conf.get(WAL_DURABILITY);
722    // Need to flush if the data is written to hbase and skip wal is enabled.
723    if (conf.get(BULK_OUTPUT_CONF_KEY) == null && durability != null
724        && Durability.SKIP_WAL.name().equalsIgnoreCase(durability)) {
725      LOG.info("Flushing all data that skipped the WAL.");
726      try {
727        connection = ConnectionFactory.createConnection(conf);
728        hAdmin = connection.getAdmin();
729        hAdmin.flush(TableName.valueOf(tableName));
730      } finally {
731        if (hAdmin != null) {
732          hAdmin.close();
733        }
734        if (connection != null) {
735          connection.close();
736        }
737      }
738    }
739  }

741  @Override
742  public int run(String[] args) throws Exception {
743    if (args.length < 2) {
744      usage("Wrong number of arguments: " + args.length);
745      return -1;
746    }
747    String inputVersionString = System.getProperty(ResultSerialization.IMPORT_FORMAT_VER);
748    if (inputVersionString != null) {
749      getConf().set(ResultSerialization.IMPORT_FORMAT_VER, inputVersionString);
750    }
751    Job job = createSubmittableJob(getConf(), args);
752    boolean isJobSuccessful = job.waitForCompletion(true);
753    if(isJobSuccessful){
754      // Flush all the regions of the table
755      flushRegionsIfNecessary(getConf());
756    }
757    long inputRecords = job.getCounters().findCounter(TaskCounter.MAP_INPUT_RECORDS).getValue();
758    long outputRecords = job.getCounters().findCounter(TaskCounter.MAP_OUTPUT_RECORDS).getValue();
759    if (outputRecords < inputRecords) {
760      System.err.println("Warning, not all records were imported (maybe filtered out).");
761      if (outputRecords == 0) {
762        System.err.println("If the data was exported from HBase 0.94 "+
763            "consider using -Dhbase.import.version=0.94.");
764      }
765    }
766
767    return (isJobSuccessful ? 0 : 1);
768  }
