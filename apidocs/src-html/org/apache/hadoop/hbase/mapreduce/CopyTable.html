044/**
045 * Tool used to copy a table to another one which can be on a different setup.
046 * It is also configurable with a start and time as well as a specification
047 * of the region server implementation if different from the local cluster.
048 */
049@InterfaceAudience.Public
050public class CopyTable extends Configured implements Tool {
051  private static final Log LOG = LogFactory.getLog(CopyTable.class);
052
053  final static String NAME = "copytable";
054  long startTime = 0;
055  long endTime = HConstants.LATEST_TIMESTAMP;
056  int batch = Integer.MAX_VALUE;
057  int cacheRow = -1;
058  int versions = -1;
059  String tableName = null;
060  String startRow = null;
061  String stopRow = null;
062  String dstTableName = null;
063  String peerAddress = null;
064  String families = null;
065  boolean allCells = false;
066  static boolean shuffle = false;
067
068  boolean bulkload = false;
069  Path bulkloadDir = null;
070
071  private final static String JOB_NAME_CONF_KEY = "mapreduce.job.name";
072
073  /**
074   * Sets up the actual job.
075   *
076   * @param args  The command line parameters.
077   * @return The newly created job.
078   * @throws IOException When setting up the job fails.
079   */
080  public Job createSubmittableJob(String[] args)
081  throws IOException {
082    if (!doCommandLine(args)) {
083      return null;
084    }
085    
086    Job job = Job.getInstance(getConf(), getConf().get(JOB_NAME_CONF_KEY, NAME + "_" + tableName));
087    job.setJarByClass(CopyTable.class);
088    Scan scan = new Scan();
089    
090    scan.setBatch(batch);
091    scan.setCacheBlocks(false);
092    
093    if (cacheRow > 0) {
094      scan.setCaching(cacheRow);
095    } else {
096      scan.setCaching(getConf().getInt(HConstants.HBASE_CLIENT_SCANNER_CACHING, 100));
097    }
098    
099    scan.setTimeRange(startTime, endTime);
100    
101    if (allCells) {
102      scan.setRaw(true);
103    }
104    if (shuffle) {
105      job.getConfiguration().set(TableInputFormat.SHUFFLE_MAPS, "true");
106    }
107    if (versions >= 0) {
108      scan.setMaxVersions(versions);
109    }
110
111    if (startRow != null) {
112      scan.setStartRow(Bytes.toBytesBinary(startRow));
113    }
114
115    if (stopRow != null) {
116      scan.setStopRow(Bytes.toBytesBinary(stopRow));
117    }
118
119    if(families != null) {
120      String[] fams = families.split(",");
121      Map<String,String> cfRenameMap = new HashMap<>();
122      for(String fam : fams) {
123        String sourceCf;
124        if(fam.contains(":")) {
125            // fam looks like "sourceCfName:destCfName"
126            String[] srcAndDest = fam.split(":", 2);
127            sourceCf = srcAndDest[0];
128            String destCf = srcAndDest[1];
129            cfRenameMap.put(sourceCf, destCf);
130        } else {
131            // fam is just "sourceCf"
132            sourceCf = fam;
133        }
134        scan.addFamily(Bytes.toBytes(sourceCf));
135      }
136      Import.configureCfRenaming(job.getConfiguration(), cfRenameMap);
137    }
138    job.setNumReduceTasks(0);
139
140    if (bulkload) {
141      TableMapReduceUtil.initTableMapperJob(tableName, scan, Import.KeyValueImporter.class, null,
142        null, job);
143
144      // We need to split the inputs by destination tables so that output of Map can be bulk-loaded.
145      TableInputFormat.configureSplitTable(job, TableName.valueOf(dstTableName));
146
147      FileSystem fs = FileSystem.get(getConf());
148      Random rand = new Random();
149      Path root = new Path(fs.getWorkingDirectory(), "copytable");
150      fs.mkdirs(root);
151      while (true) {
152        bulkloadDir = new Path(root, "" + rand.nextLong());
153        if (!fs.exists(bulkloadDir)) {
154          break;
155        }
156      }
157
158      System.out.println("HFiles will be stored at " + this.bulkloadDir);
159      HFileOutputFormat2.setOutputPath(job, bulkloadDir);
160      try (Connection conn = ConnectionFactory.createConnection(getConf());
161          Admin admin = conn.getAdmin()) {
162        HFileOutputFormat2.configureIncrementalLoadMap(job,
163            admin.listTableDescriptor((TableName.valueOf(dstTableName))));
164      }
165    } else {
166      TableMapReduceUtil.initTableMapperJob(tableName, scan,
167        Import.Importer.class, null, null, job);
168
169      TableMapReduceUtil.initTableReducerJob(dstTableName, null, job, null, peerAddress, null,
170        null);
171    }
172
173    return job;
174  }

177   * @param errorMsg Error message.  Can be null.
178   */
179  private static void printUsage(final String errorMsg) {
180    if (errorMsg != null && errorMsg.length() > 0) {
181      System.err.println("ERROR: " + errorMsg);
182    }
183    System.err.println("Usage: CopyTable [general options] [--starttime=X] [--endtime=Y] " +
184        "[--new.name=NEW] [--peer.adr=ADR] <tablename>");
185    System.err.println();
186    System.err.println("Options:");
187    System.err.println(" rs.class     hbase.regionserver.class of the peer cluster");
188    System.err.println("              specify if different from current cluster");
189    System.err.println(" rs.impl      hbase.regionserver.impl of the peer cluster");
190    System.err.println(" startrow     the start row");
191    System.err.println(" stoprow      the stop row");
192    System.err.println(" starttime    beginning of the time range (unixtime in millis)");
193    System.err.println("              without endtime means from starttime to forever");
194    System.err.println(" endtime      end of the time range.  Ignored if no starttime specified.");
195    System.err.println(" versions     number of cell versions to copy");
196    System.err.println(" new.name     new table's name");
197    System.err.println(" peer.adr     Address of the peer cluster given in the format");
198    System.err.println("              hbase.zookeeper.quorum:hbase.zookeeper.client"
199        + ".port:zookeeper.znode.parent");
200    System.err.println(" families     comma-separated list of families to copy");
201    System.err.println("              To copy from cf1 to cf2, give sourceCfName:destCfName. ");
202    System.err.println("              To keep the same name, just give \"cfName\"");
203    System.err.println(" all.cells    also copy delete markers and deleted cells");
204    System.err.println(" bulkload     Write input into HFiles and bulk load to the destination "
205        + "table");
206    System.err.println();
207    System.err.println("Args:");
208    System.err.println(" tablename    Name of the table to copy");
209    System.err.println();
210    System.err.println("Examples:");
211    System.err.println(" To copy 'TestTable' to a cluster that uses replication for a 1 hour window:");
212    System.err.println(" $ hbase " +
213        "org.apache.hadoop.hbase.mapreduce.CopyTable --starttime=1265875194289 --endtime=1265878794289 " +
214        "--peer.adr=server1,server2,server3:2181:/hbase --families=myOldCf:myNewCf,cf2,cf3 TestTable ");
215    System.err.println("For performance consider the following general option:\n"
216        + "  It is recommended that you set the following to >=100. A higher value uses more memory but\n"
217        + "  decreases the round trip time to the server and may increase performance.\n"
218        + "    -Dhbase.client.scanner.caching=100\n"
219        + "  The following should always be set to false, to prevent writing data twice, which may produce \n"
220        + "  inaccurate results.\n"
221        + "    -Dmapreduce.map.speculative=false");
222  }

224  private boolean doCommandLine(final String[] args) {
225    // Process command-line args. TODO: Better cmd-line processing
226    // (but hopefully something not as painful as cli options).
227    if (args.length < 1) {
228      printUsage(null);
229      return false;
230    }
231    try {
232      for (int i = 0; i < args.length; i++) {
233        String cmd = args[i];
234        if (cmd.equals("-h") || cmd.startsWith("--h")) {
235          printUsage(null);
236          return false;
237        }
238
239        final String startRowArgKey = "--startrow=";
240        if (cmd.startsWith(startRowArgKey)) {
241          startRow = cmd.substring(startRowArgKey.length());
242          continue;
243        }
244
245        final String stopRowArgKey = "--stoprow=";
246        if (cmd.startsWith(stopRowArgKey)) {
247          stopRow = cmd.substring(stopRowArgKey.length());
248          continue;
249        }
250
251        final String startTimeArgKey = "--starttime=";
252        if (cmd.startsWith(startTimeArgKey)) {
253          startTime = Long.parseLong(cmd.substring(startTimeArgKey.length()));
254          continue;
255        }
256
257        final String endTimeArgKey = "--endtime=";
258        if (cmd.startsWith(endTimeArgKey)) {
259          endTime = Long.parseLong(cmd.substring(endTimeArgKey.length()));
260          continue;
261        }
262        
263        final String batchArgKey = "--batch=";
264        if (cmd.startsWith(batchArgKey)) {
265          batch = Integer.parseInt(cmd.substring(batchArgKey.length()));
266          continue;
267        }
268        
269        final String cacheRowArgKey = "--cacheRow=";
270        if (cmd.startsWith(cacheRowArgKey)) {
271          cacheRow = Integer.parseInt(cmd.substring(cacheRowArgKey.length()));
272          continue;
273        }
274
275        final String versionsArgKey = "--versions=";
276        if (cmd.startsWith(versionsArgKey)) {
277          versions = Integer.parseInt(cmd.substring(versionsArgKey.length()));
278          continue;
279        }
280
281        final String newNameArgKey = "--new.name=";
282        if (cmd.startsWith(newNameArgKey)) {
283          dstTableName = cmd.substring(newNameArgKey.length());
284          continue;
285        }
286
287        final String peerAdrArgKey = "--peer.adr=";
288        if (cmd.startsWith(peerAdrArgKey)) {
289          peerAddress = cmd.substring(peerAdrArgKey.length());
290          continue;
291        }
292
293        final String familiesArgKey = "--families=";
294        if (cmd.startsWith(familiesArgKey)) {
295          families = cmd.substring(familiesArgKey.length());
296          continue;
297        }
298
299        if (cmd.startsWith("--all.cells")) {
300          allCells = true;
301          continue;
302        }
303
304        if (cmd.startsWith("--bulkload")) {
305          bulkload = true;
306          continue;
307        }
308
309        if (cmd.startsWith("--shuffle")) {
310          shuffle = true;
311          continue;
312        }
313
314        if (i == args.length-1) {
315          tableName = cmd;
316        } else {
317          printUsage("Invalid argument '" + cmd + "'");
318          return false;
319        }
320      }
321      if (dstTableName == null && peerAddress == null) {
322        printUsage("At least a new table name or a " +
323            "peer address must be specified");
324        return false;
325      }
326      if ((endTime != 0) && (startTime > endTime)) {
327        printUsage("Invalid time range filter: starttime=" + startTime + " >  endtime=" + endTime);
328        return false;
329      }
330
331      if (bulkload && peerAddress != null) {
332        printUsage("Remote bulkload is not supported!");
333        return false;
334      }
335
336      // set dstTableName if necessary
337      if (dstTableName == null) {
338        dstTableName = tableName;
339      }
340    } catch (Exception e) {
341      e.printStackTrace();
342      printUsage("Can't start because " + e.getMessage());
343      return false;
344    }
345    return true;
346  }
347
348  /**
349   * Main entry point.
350   *
351   * @param args  The command line parameters.
352   * @throws Exception When running the job fails.
353   */
354  public static void main(String[] args) throws Exception {
355    int ret = ToolRunner.run(HBaseConfiguration.create(), new CopyTable(), args);
356    System.exit(ret);
357  }
358
359  @Override
360  public int run(String[] args) throws Exception {
361    Job job = createSubmittableJob(args);
362    if (job == null) return 1;
363    if (!job.waitForCompletion(true)) {
364      LOG.info("Map-reduce job failed!");
365      if (bulkload) {
366        LOG.info("Files are not bulkloaded!");
367      }
368      return 1;
369    }
370    int code = 0;
371    if (bulkload) {
372      code = new LoadIncrementalHFiles(this.getConf()).run(new String[]{this.bulkloadDir.toString(),
373          this.dstTableName});
374      if (code == 0) {
375        // bulkloadDir is deleted only LoadIncrementalHFiles was successful so that one can rerun
376        // LoadIncrementalHFiles.
377        FileSystem fs = FileSystem.get(this.getConf());
378        if (!fs.delete(this.bulkloadDir, true)) {
379          LOG.error("Deleting folder " + bulkloadDir + " failed!");
380          code = 1;
381        }
382      }
383    }
384    return code;
385  }
